35     @BugPattern(name = "RectIntersectReturnValueIgnored",
36         summary = "Return value of android.graphics.Rect.intersect() must be checked",
37         explanation = "`android.graphics.Rect.intersect(Rect r)` and "
38             + "`android.graphics.Rect.intersect(int, int, int, int)` do not always modify the "
39             + "rectangle to the intersected result. If the rectangles do not intersect, no change "
40             + "is made and the original rectangle is not modified. These methods return false to "
41             + "indicate that this has happened.\n\n"
42             + "If you donâ€™t check the return value of these methods, you may end up drawing the "
43             + "wrong rectangle.",
44         category = ANDROID, severity = ERROR, maturity = MATURE)
45     public class RectIntersectReturnValueIgnored extends AbstractReturnValueIgnored {
46       @Override
47       public Matcher<? super MethodInvocationTree> specializedMatcher() {
48         return instanceMethod().onExactClass("android.graphics.Rect").named("intersect");
49       }
50     
51       @Override
52       public Description describe(MethodInvocationTree methodInvocationTree, VisitorState state) {
53         return describeMatch(methodInvocationTree);
54       }
55     }
