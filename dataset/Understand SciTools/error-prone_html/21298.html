45     /**
46      * @author bill.pugh@gmail.com (Bill Pugh)
47      * @author eaftan@google.com (Eddie Aftandilian)
48      */
49     @BugPattern(name = "BadShiftAmount",
50         summary = "Shift by an amount that is out of range",
51         explanation = "For shift operations on int types, only the five lowest-order bits of the "
52             + "shift amount are used as the shift distance.  This means that shift amounts that are "
53             + "not in the range 0 to 31, inclusive, are silently mapped to values in that range. "
54             + "For example, a shift of an int by 32 is equivalent to shifting by 0, i.e., a no-op.\n\n"
55             + "See JLS 15.19, \"Shift Operators\", for more details.",
56         category = JDK, severity = ERROR, maturity = MATURE)
57     public class BadShiftAmount extends BugChecker implements BinaryTreeMatcher {
58     
59       /**
60        * Matches if the left operand is an int, byte, short, or char, and the right operand is a
61        * literal that is not in the range 0-31 inclusive.
62        *
63        * <p>In a shift expression, byte, short, and char undergo unary numeric promotion and are
64        * promoted to int.  See JLS 5.6.1.
65        */
66       private static final Matcher<BinaryTree> BAD_SHIFT_AMOUNT_INT = new Matcher<BinaryTree>() {
67         @Override
68         public boolean matches(BinaryTree tree, VisitorState state) {
69           Type leftType = ((JCTree) tree.getLeftOperand()).type;
70           Types types = state.getTypes();
71           Symtab symtab = state.getSymtab();
72           if (!(types.isSameType(leftType, symtab.intType)) &&
73               !(types.isSameType(leftType, symtab.byteType)) &&
74               !(types.isSameType(leftType, symtab.shortType)) &&
75               !(types.isSameType(leftType, symtab.charType))) {
76             return false;
77           }
78     
79           ExpressionTree rightOperand = tree.getRightOperand();
80           if (rightOperand instanceof LiteralTree) {
81             Object rightValue = ((LiteralTree) rightOperand).getValue();
82             if (rightValue instanceof Number) {
83               int intValue = ((Number) rightValue).intValue();
84               return intValue < 0 || intValue > 31;
85             }
86           }
87     
88           return false;
89         }
90       };
91     
92       public static final Matcher<BinaryTree> BINARY_TREE_MATCHER = allOf(
93           anyOf(
94               kindIs(Kind.LEFT_SHIFT),
95               kindIs(Kind.RIGHT_SHIFT),
96               kindIs(Kind.UNSIGNED_RIGHT_SHIFT)),
97           BAD_SHIFT_AMOUNT_INT
98       );
99     
100       @Override
101       public Description matchBinary(BinaryTree tree, VisitorState state) {
102         if (!BINARY_TREE_MATCHER.matches(tree, state)) {
103           return Description.NO_MATCH;
104         }
105     
106         /*
107          * For shift amounts in [32, 63], cast the left operand to long.  Otherwise change the shift
108          * amount to whatever would actually be used.
109          */
110         int intValue = ((Number) ((LiteralTree) tree.getRightOperand()).getValue()).intValue();
111     
112         Fix fix;
113         if (intValue >= 32 && intValue <= 63) {
114           if (tree.getLeftOperand().getKind() == Kind.INT_LITERAL) {
115             fix = SuggestedFix.postfixWith(tree.getLeftOperand(), "L");
116           } else {
117             fix = SuggestedFix.prefixWith(tree, "(long) ");
118           }
119         } else {
120           JCLiteral jcLiteral = (JCLiteral) tree.getRightOperand();
121           // This is the equivalent shift distance according to JLS 15.19.
122           String actualShiftDistance = Integer.toString(intValue & 0x1f);
123           int actualStart = ASTHelpers.getActualStartPosition(jcLiteral, state.getSourceCode());
124           if (actualStart != jcLiteral.getStartPosition()) {
125             fix = SuggestedFix.replace(tree.getRightOperand(), actualShiftDistance,
126                 actualStart - jcLiteral.getStartPosition(), 0);
127           } else {
128             fix = SuggestedFix.replace(tree.getRightOperand(), actualShiftDistance);
129           }
130         }
131         return describeMatch(tree, fix);
132       }
133     }
