29     import com.sun.source.tree.ExpressionTree;
30     import com.sun.source.tree.MethodInvocationTree;
31     import com.sun.tools.javac.code.Type;
32     
33     import java.util.Arrays;
34     import java.util.HashSet;
35     import java.util.Set;

37     /**
38      * @author alexeagle@google.com (Alex Eagle)
39      */
40     @BugPattern(name = "ReturnValueIgnored",
41         altNames = {"ResultOfMethodCallIgnored", "CheckReturnValue"},
42         summary = "Return value of this method must be used",
43         explanation = "Certain library methods do nothing useful if their return value is ignored. "
44             + "For example, String.trim() has no side effects, and you must store the return value "
45             + "of String.intern() to access the interned string.  This check encodes a list of "
46             + "methods in the JDK whose return value must be used and issues an error if they "
47             + "are not.",
48         category = JDK, severity = ERROR, maturity = MATURE)
49     public class ReturnValueIgnored extends AbstractReturnValueIgnored {
50       /**
51        * A set of types which this checker should examine method calls on.
52        */
53       // There are also some high-priority return value ignored checks in FindBugs for various
54       // threading constructs which do not return the same type as the receiver.
55       // This check does not deal with them, since the fix is less straightforward.
56       // See a list of the FindBugs checks here:
57       // http://code.google.com/searchframe#Fccnll6ERQ0/trunk/findbugs/src/java/edu/umd/cs/findbugs/ba/CheckReturnAnnotationDatabase.java
58       private static final Set<String> typesToCheck = new HashSet<>(Arrays.asList(
59           "java.lang.String", "java.math.BigInteger", "java.math.BigDecimal", "java.nio.file.Path"));
60       /**
61        * Return a matcher for method invocations in which the method being called is on an instance of
62        * a type in the typesToCheck set and returns the same type (e.g. String.trim() returns a
63        * String).
64        */
65       @Override
66       public Matcher<? super MethodInvocationTree> specializedMatcher() {
67         return allOf(methodReceiverHasType(typesToCheck), methodReturnsSameTypeAsReceiver());
68       }
69     
70       /**
71        * Matches method invocations that return the same type as the receiver object.
72        */
73       private static Matcher<ExpressionTree> methodReturnsSameTypeAsReceiver() {
74         return new Matcher<ExpressionTree>() {
75           @Override
76           public boolean matches(ExpressionTree expressionTree, VisitorState state) {
77             return state.getTypes().isSameType(
78                 ASTHelpers.getReceiverType(expressionTree), ASTHelpers.getReturnType(expressionTree));
79           }
80         };
81       }
82     
83       /**
84        * Matches method calls whose receiver objects are of a type included in the set.
85        */
86       private static Matcher<ExpressionTree> methodReceiverHasType(final Set<String> typeSet) {
87         return new Matcher<ExpressionTree>() {
88           @Override
89           public boolean matches(ExpressionTree expressionTree, VisitorState state) {
90             Type receiverType = ASTHelpers.getReceiverType(expressionTree);
91             return typeSet.contains(receiverType.toString());
92           }
93         };
94       }
95     }
