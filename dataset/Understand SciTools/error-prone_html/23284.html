19     import java.util.Comparator;
20     
21     /**
22      * @author irogers@google.com (Ian Rogers)
23      */
24     public class BadComparableNegativeCases {
25       // The corrected cases of the PositiveCases test.
26       static class ComparableTest implements Comparable<ComparableTest> {
27         private final long value = 0;
28     
29         public int compareTo(ComparableTest other) {
30           return Long.compare(value, other.value);
31         }
32       }
33     
34       static class BoxedComparableTest implements Comparable<BoxedComparableTest> {
35         private final Long value = Long.valueOf(0);
36     
37         public int compareTo(BoxedComparableTest other) {
38           return value.compareTo(other.value);
39         }
40       }
41     
42       static final Comparator<Number> COMPARATOR_UNBOXED_INT_CAST =
43           new Comparator<Number>() {
44             public int compare(Number n1, Number n2) {
45               return Long.compare(n1.longValue(), n2.longValue());
46             }
47           };
48     
49       static final Comparator<Long> COMPARATOR_BOXED_INT_CAST =
50           new Comparator<Long>() {
51             public int compare(Long n1, Long n2) {
52               return n1.compareTo(n2);
53             }
54           };
55     
56       // Don't match non-Comparable or Comparator cases.
57       static class NonComparableTest {
58         private final long value = 0;
59     
60         public int compareTo(ComparableTest other) {
61           return (int) (value - other.value);
62         }
63       }
64     
65       static final Object COMPARATOR_LIKE_INT_CAST =
66           new Object() {
67             public int compare(Long n1, Long n2) {
68               return (int) (n1 - n2);
69             }
70           };
71     
72       // Narrowing conversions that don't follow the long -> int pattern.
73       static final Comparator<Number> COMPARATOR_UNBOXED_NON_PATTERN_LONG_CAST =
74           new Comparator<Number>() {
75             // To match the Comparator API.
76             @Override
77             public int compare(Number n1, Number n2) {
78               return (int) (n1.intValue() - n2.intValue());
79             }
80     
81             public short compare(int n1, int n2) {
82               return (short) (n1 - n2);
83             }
84     
85             public byte compare(long n1, long n2) {
86               return (byte) (n1 - n2);
87             }
88           };
89     
90       // Not narrowing conversions.
91       static final Comparator<Number> COMPARATOR_UNBOXED_NON_NARROW_LONG_CAST =
92           new Comparator<Number>() {
93             // To match the Comparator API.
94             @Override
95             public int compare(Number n1, Number n2) {
96               return (int) (n1.intValue() - n2.intValue());
97             }
98     
99             public long compare(long n1, long n2) {
100               return (long) (n1 - n2);
101             }
102           };
103     
104       static final Comparator<Number> COMPARATOR_UNBOXED_NON_NARROW_INT_CAST =
105           new Comparator<Number>() {
106             public int compare(Number n1, Number n2) {
107               return (int) (n1.intValue() - n2.intValue());
108             }
109           };
110     
111       static final Comparator<Number> COMPARATOR_UNBOXED_NON_NARROW_SHORT_CAST =
112           new Comparator<Number>() {
113             // To match the Comparator API.
114             @Override
115             public int compare(Number n1, Number n2) {
116               return (int) (n1.intValue() - n2.intValue());
117             }

125           new Comparator<Number>() {
126             // To match the Comparator API.
127             @Override
128             public int compare(Number n1, Number n2) {
129               return (int) (n1.intValue() - n2.intValue());

139           new Comparator<Number>() {
140             @Override
141             public int compare(Number n1, Number n2) {
142               return (char) (n1.shortValue() - n2.shortValue());
143             }
144     
145             public char compare(char n1, char n2) {
146               return (char) (n1 - n2);
147             }
148     
149             public char compare(byte n1, byte n2) {
150               return (char) (n1 - n2);
151             }
152           };
