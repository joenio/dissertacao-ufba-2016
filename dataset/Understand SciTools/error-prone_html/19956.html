111       @Test
112       public void filterWorks() {
113         ScannerSupplier ss =
114             ScannerSupplier.fromBugCheckerClasses(
115                 ArrayEquals.class, BadShiftAmount.class, StaticAccessedFromInstance.class);
116         Predicate<BugCheckerInfo> isBadShiftAmount =
117             new Predicate<BugCheckerInfo>() {
118           @Override
119               public boolean apply(BugCheckerInfo input) {
120             return input.canonicalName().equals("BadShiftAmount");
121           }
122         };
123     
124         Set<BugCheckerInfo> expected = getSuppliers(BadShiftAmount.class);
125     
126         assertThat(ss.filter(isBadShiftAmount).getEnabledChecks()).isEqualTo(expected);
127       }

143       public void applyOverridesEnablesCheck() throws Exception {
144         ScannerSupplier ss =
145             ScannerSupplier
146                 .fromBugCheckerClasses(
147                     ArrayEquals.class, BadShiftAmount.class, StaticAccessedFromInstance.class)
148             .filter(Predicates.alwaysFalse());    // disables all checks
149     
150         ErrorProneOptions epOptions = ErrorProneOptions.processArgs(
151             ImmutableList.of("-Xep:ArrayEquals", "-Xep:BadShiftAmount"));
152     
153         Set<BugCheckerInfo> expected = getSuppliers(ArrayEquals.class, BadShiftAmount.class);
154     
155         assertThat(ss.applyOverrides(epOptions).getEnabledChecks()).isEqualTo(expected);
156       }

223       @Test
224       public void applyOverridesSetsSeverity() throws Exception {
225         ScannerSupplier ss =
226             ScannerSupplier.fromBugCheckerClasses(
227                 BadShiftAmount.class, ChainingConstructorIgnoresParameter.class, StringEquality.class);
228         ErrorProneOptions epOptions = ErrorProneOptions.processArgs(
229             ImmutableList.of(
230                 "-Xep:ChainingConstructorIgnoresParameter:WARN",
231                 "-Xep:StringEquality:ERROR"));
232         ScannerSupplier overriddenScannerSupplier = ss.applyOverrides(epOptions);
233         
234         Map<String, SeverityLevel> expected = ImmutableMap.of(
235             "BadShiftAmount", SeverityLevel.ERROR,
236             "ChainingConstructorIgnoresParameter", SeverityLevel.WARNING,
237             "StringEquality", SeverityLevel.ERROR);
238     
239         assertThat(overriddenScannerSupplier.severities()).isEqualTo(expected);
240       }
