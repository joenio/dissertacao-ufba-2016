244       @BugPattern(
245           name = "InLoopChecker",
246           summary = "Checker that flags the given expression statement if the given matcher matches",
247           category = ONE_OFF, maturity = MATURE, severity = ERROR)
248       public static class InLoopChecker extends MatcherChecker {
249         public InLoopChecker() {
250           super("System.out.println();", inLoop());
251         }
252       }
253     
254       private static ScannerSupplier methodTreeCheckerSupplier(Matcher<MethodTree> matcher) {
255         return ScannerSupplier.fromScanner(new ErrorProneScanner(new MethodTreeChecker(matcher)));
256       }
257     
258       @BugPattern(
259         name = "MethodTreeChecker",
260         summary = "Checker that flags the given method declaration if the given matcher matches",
261         category = ONE_OFF,
262         maturity = MATURE,
263         severity = ERROR
264       )
265       static class MethodTreeChecker extends BugChecker
266           implements MethodTreeMatcher {
267         private final Matcher<MethodTree> matcher;
268     
269         public MethodTreeChecker(Matcher<MethodTree> matcher) {
270           this.matcher = matcher;
271         }
272     
273         @Override
274         public Description matchMethod(MethodTree tree, VisitorState state) {
275           return matcher.matches(tree, state) ? describeMatch(tree) : Description.NO_MATCH;
276         }
277       }
