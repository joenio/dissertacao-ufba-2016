17     package com.google.errorprone.matchers.method;
18     
19     import com.google.common.base.Optional;
20     import com.google.errorprone.VisitorState;
21     import com.google.errorprone.matchers.method.MethodMatchers.MethodNameMatcher;
22     import com.google.errorprone.matchers.method.MethodMatchers.ParameterMatcher;
23     
24     import com.sun.source.tree.ExpressionTree;
25     
26     import java.util.Arrays;
27     import java.util.regex.Pattern;
28     
29     /** Matchers that select on method name. */
30     public abstract class MethodNameMatcherImpl extends AbstractChainedMatcher<MatchState, MatchState>
31         implements MethodNameMatcher {
32     
33       MethodNameMatcherImpl(AbstractSimpleMatcher<MatchState> baseMatcher) {
34         super(baseMatcher);
35       }
36     
37       @Override
38       public ParameterMatcher withParameters(String... parameters) {
39         return withParameters(Arrays.asList(parameters));
40       }
41     
42       @Override
43       public ParameterMatcher withParameters(Iterable<String> parameters) {
44         return new ParameterMatcherImpl(this, parameters);
45       }
46     
47       /** Matches on exact method name. */
48       static class Exact extends MethodNameMatcherImpl {
49         private final String name;
50     
51         Exact(AbstractSimpleMatcher<MatchState> baseMatcher, String name) {
52           super(baseMatcher);
53           this.name = name;
54         }
55     
56         @Override
57         protected Optional<MatchState> matchResult(ExpressionTree item, MatchState method,
58             VisitorState state) {
59           if (!method.sym().getSimpleName().contentEquals(name)) {
60             return Optional.absent();
61           }
62           return Optional.of(method);
63         }
64       }
65     
66       /** Matches any method name. */
67       static class Any extends MethodNameMatcherImpl {
68         Any(AbstractSimpleMatcher<MatchState> baseMatcher) {
69           super(baseMatcher);
70         }
71     
72         @Override
73         protected Optional<MatchState> matchResult(ExpressionTree item, MatchState method,
74             VisitorState state) {
75           return Optional.of(method);
76         }
77       }
78     
79       /** Matches methods with a name that matches the given regular expression. */
80       static class Regex extends MethodNameMatcherImpl {
81     
82         private final Pattern regex;
83     
84         Regex(AbstractSimpleMatcher<MatchState> baseMatcher, Pattern regex) {
85           super(baseMatcher);
86           this.regex = regex;
87         }
88     
89     
90         @Override
91         protected Optional<MatchState> matchResult(ExpressionTree item, MatchState method,
92             VisitorState state) {
93           if (!regex.matcher(method.sym().getSimpleName().toString()).matches()) {
94             return Optional.absent();
95           }
96           return Optional.of(method);
97         }
98       }
99     }
