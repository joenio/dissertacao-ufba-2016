15     package com.google.errorprone.bugpatterns;
16     
17     import static com.google.errorprone.BugPattern.Category.GUICE;
18     import static com.google.errorprone.BugPattern.MaturityLevel.EXPERIMENTAL;
19     import static com.google.errorprone.BugPattern.SeverityLevel.ERROR;
20     import static com.google.errorprone.bugpatterns.BugChecker.VariableTreeMatcher;
21     
22     import com.google.errorprone.BugPattern;
23     import com.google.errorprone.VisitorState;
24     import com.google.errorprone.fixes.SuggestedFix;
25     import com.google.errorprone.matchers.Description;
26     import com.google.errorprone.matchers.Matcher;
27     import com.google.errorprone.matchers.Matchers;
28     import com.google.errorprone.util.ASTHelpers;
29     
30     import com.sun.source.tree.AnnotationTree;
31     import com.sun.source.tree.MethodTree;
32     import com.sun.source.tree.VariableTree;
33     import com.sun.tools.javac.code.Attribute.Compound;
34     import com.sun.tools.javac.code.Symbol;
35     import com.sun.tools.javac.code.Symbol.MethodSymbol;
36     
37     import javax.lang.model.element.TypeElement;
38     
39     /**
40      * @author sgoldfeder@google.com (Steven Goldfeder)
41      */
42     @BugPattern(name = "GuiceAssistedParameters", summary =
43         "A constructor cannot have two @Assisted parameters of the same type unless they are "
44         + "disambiguated with named @Assisted annotations. ",
45         explanation = "See http://google-guice.googlecode.com/git/javadoc/com/google/inject/assistedinject/FactoryModuleBuilder.html",
46         category = GUICE, severity = ERROR, maturity = EXPERIMENTAL)
47     public class GuiceAssistedParameters extends BugChecker implements VariableTreeMatcher {
48     
49       private static final String ASSISTED_ANNOTATION = "com.google.inject.assistedinject.Assisted";
50     
51       private final Matcher<VariableTree> constructorAssistedParameterMatcher =
52           new Matcher<VariableTree>() {
53             @Override
54             public boolean matches(VariableTree t, VisitorState state) {
55               Symbol modified = ASTHelpers.getSymbol(state.getPath().getParentPath().getLeaf());
56               return modified != null
57                   && modified.isConstructor()
58                   && Matchers.<VariableTree>hasAnnotation(ASSISTED_ANNOTATION).matches(t, state);
59             }
60           };
61     
62       @Override
63       public final Description matchVariable(VariableTree variableTree, VisitorState state) {
64         if (constructorAssistedParameterMatcher.matches(variableTree, state)) {
65           Compound thisParamsAssisted = null;
66           for (Compound c : ASTHelpers.getSymbol(variableTree).getAnnotationMirrors()) {
67             if (((TypeElement) c.getAnnotationType().asElement()).getQualifiedName()
68                 .contentEquals(ASSISTED_ANNOTATION)) {
69               thisParamsAssisted = c;
70             }
71           }
72           MethodTree enclosingMethod = (MethodTree) state.getPath().getParentPath().getLeaf();
73           // count the number of parameters of this type and value. One is expected since we
74           // will be iterating through all parameters including the one we're matching.
75           int numIdentical = 0;
76           for (VariableTree parameter : enclosingMethod.getParameters()) {
77             if (Matchers.<VariableTree>isSameType(variableTree).matches(parameter, state)) {
78               Compound otherParamsAssisted = null;
79               for (Compound c : ASTHelpers.getSymbol(parameter).getAnnotationMirrors()) {
80                 if (((TypeElement) c.getAnnotationType().asElement()).getQualifiedName()
81                     .contentEquals(ASSISTED_ANNOTATION)) {
82                   otherParamsAssisted = c;
83                 }
84               }
85               if (otherParamsAssisted != null) {
86                 if (thisParamsAssisted.getElementValues().isEmpty()
87                     && otherParamsAssisted.getElementValues().isEmpty()) {
88                   //both have unnamed @Assisted annotations
89                   numIdentical++;
90                 }
91                 //if value is specified, check that they are equal
92                 //also, there can only be one value which is why I didn't check for equality
93                 //in both directions
94                 for (MethodSymbol m : thisParamsAssisted.getElementValues().keySet())
95                   if (otherParamsAssisted.getElementValues().get(m).getValue()
96                       .equals(thisParamsAssisted.getElementValues().get(m).getValue())) {
97                     numIdentical++;
98                   }
99               }
100               if (numIdentical > 1) {
101                 return describe(variableTree, state);
102               }
103             }
104           }
105         }
106         return Description.NO_MATCH;
107       }
108     
109       public Description describe(VariableTree variableTree, VisitorState state) {
110         // find the @Assisted annotation to put the error on
111         for (AnnotationTree annotation : variableTree.getModifiers().getAnnotations()) {
112           if (ASTHelpers.getSymbol(annotation).equals(state.getSymbolFromString(ASSISTED_ANNOTATION))) {
113             return describeMatch(annotation, SuggestedFix.delete(annotation));
114           }
115         }
116         throw new IllegalStateException("Expected to find @Assisted on this parameter");
117       }
118     }
