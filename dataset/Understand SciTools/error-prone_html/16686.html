17     package com.google.errorprone.matchers;
18     
19     import com.google.errorprone.VisitorState;
20     import com.sun.source.tree.ExpressionTree;
21     import com.sun.source.tree.LiteralTree;
22     
23     /**
24      * @author eaftan@google.com (Eddie Aftandilian)
25      * @author pepstein@google.com (Peter Epstein)
26      */
27     public class StringLiteral implements Matcher<ExpressionTree> {
28     
29       private final String value;
30     
31       public StringLiteral(String value) {
32         this.value = value;
33       }
34     
35       @Override
36       public boolean matches(ExpressionTree expressionTree, VisitorState state) {
37         if (expressionTree instanceof LiteralTree) {
38           LiteralTree literalTree = (LiteralTree) expressionTree;
39           Object actualValue = literalTree.getValue();
40           return actualValue instanceof String && actualValue.equals(value);
41         } else {
42           return false;
43         }
44       }
45     }
