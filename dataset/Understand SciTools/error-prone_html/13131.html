17     package com.google.errorprone.refaster.testdata.template;
18     
19     import com.google.errorprone.refaster.ImportPolicy;
20     import com.google.errorprone.refaster.annotation.AfterTemplate;
21     import com.google.errorprone.refaster.annotation.BeforeTemplate;
22     import com.google.errorprone.refaster.annotation.UseImportPolicy;
23     
24     import java.util.Map;
25     
26     /**
27      * Test for an {@code ImportPolicy} importing nested classes directly.
28      *
29      * @author Louis Wasserman
30      */
31     public class ImportClassDirectlyTemplate<K, V> {
32       @BeforeTemplate void forEachKeys(Map<K, V> map) {
33         for (K k : map.keySet()) {
34           System.out.println(k + " " + map.get(k));
35         }
36       }
37     
38       @AfterTemplate
39       @UseImportPolicy(ImportPolicy.IMPORT_CLASS_DIRECTLY)
40       void forEachEntries(Map<K, V> map) {
41         for (Map.Entry<K, V> entry : map.entrySet()) {
42           System.out.println(entry.getKey() + " " + entry.getValue());
43         }
44       }
45     }
