17     package com.google.errorprone.matchers.method;
18     
19     import com.google.errorprone.matchers.Matcher;
20     import com.google.errorprone.predicates.TypePredicate;
21     import com.google.errorprone.suppliers.Supplier;
22     
23     import com.sun.source.tree.ExpressionTree;
24     import com.sun.tools.javac.code.Type;
25     
26     import java.util.regex.Pattern;
27     
28     public class MethodMatchers {
29     
30       // Language definition for fluent method matchers.
31     
32       public interface InstanceMethodMatcher extends Matcher<ExpressionTree> {
33         /** Match on types with satisfy the given predicate. */
34         MethodClassMatcher onClass(TypePredicate predicate);
35         
36         /** Match on types with the given fully-qualified name. (e.g. java.lang.String) */
37         MethodClassMatcher onExactClass(String className);
38         
39         /** Match on the given type exactly. */
40         MethodClassMatcher onExactClass(Supplier<Type> classType);
41         
42         /** Match on descendants of the given fully-qualified type name. */
43         MethodClassMatcher onDescendantOf(String className);
44         
45         /** Match on descendants of the given type. */
46         MethodClassMatcher onDescendantOf(Supplier<Type> classType);
47         
48         /** Match on types that are descendants of any of the given types. */
49         MethodClassMatcher onDescendantOfAny(String... classTypes);
50         
51         /** Match on types that are descendants of any of the given types. */
52         MethodClassMatcher onDescendantOfAny(Iterable<String> classTypes);
53         
54         /** Match on any class. */
55         MethodClassMatcher anyClass();
56       }
57     
58       public interface StaticMethodMatcher extends Matcher<ExpressionTree> {
59         /** Match on types with the given fully-qualified name. (e.g. {@code java.lang.String} */
60         MethodClassMatcher onClass(String className);
61         
62         /** Match on the given type exactly. */
63         MethodClassMatcher onClass(Supplier<Type> classType);
64         
65         /** Match on types that are equal to any of the given types. */
66         MethodClassMatcher onClassAny(Iterable<String> classNames);
67         
68         /** Match on types that are equal to any of the given types. */
69         MethodClassMatcher onClassAny(String... classNames);
70       }
71     
72       public interface AnyMethodMatcher extends Matcher<ExpressionTree> {
73         /** Match the given type exactly. */
74         MethodClassMatcher onClass(TypePredicate predicate);
75     
76         /** Match on any class. */
77         MethodClassMatcher anyClass();
78       }
79     
80       public interface MethodClassMatcher extends Matcher<ExpressionTree> {
81         /** Match methods with the given name. (e.g. {@code toString}) */
82         MethodNameMatcher named(String name);
83         
84         /** Match methods with any name. */
85         MethodNameMatcher withAnyName();
86     
87         /** Match methods with a name that matches the given regular expression. */
88         MethodNameMatcher withNameMatching(Pattern pattern);
89         
90         /**
91          * Match methods with the given signature. The implementation uses javac internals to
92          * pretty-print the signatures, and the signature format is not well-specified. This matcher
93          * should be used with caution.
94          * 
95          * Example: {@code format(java.lang.String,java.lang.Object...)}
96          */
97         MethodSignatureMatcher withSignature(String signature);
98       }
99     
100       public interface MethodSignatureMatcher extends Matcher<ExpressionTree> {
101       }
102     
103       public interface MethodNameMatcher extends Matcher<ExpressionTree> {
104         /** Match methods whose formal parameters have the given types. */
105         ParameterMatcher withParameters(String... parameters);
106         
107         /** Match methods whose formal parameters have the given types. */
108         ParameterMatcher withParameters(Iterable<String> parameters);
109       }
110     
111       public interface ConstructorMatcher extends Matcher<ExpressionTree> {
112         /** Match on the given type exactly. */
113         ConstructorClassMatcher forClass(String className);
114         
115         /** Match on the given type exactly. */
116         ConstructorClassMatcher forClass(Supplier<Type> classType);
117       }
118     
119       public interface ConstructorClassMatcher extends Matcher<ExpressionTree> {
120         /** Match constructors whose formal parameters have the given types. */
121         ParameterMatcher withParameters(String... parameters);
122         
123         /** Match constructors whose formal parameters have the given types. */
124         ParameterMatcher withParameters(Iterable<String> parameters);
125       }
126     
127       public interface ParameterMatcher extends Matcher<ExpressionTree> {
128       }
129     
130       // Method matcher factories
131     
132       public static StaticMethodMatcher staticMethod() {
133         return new StaticMethodMatcherImpl();
134       }
135     
136       public static InstanceMethodMatcher instanceMethod() {
137         return new InstanceMethodMatcherImpl();
138       }
139     
140       public static AnyMethodMatcher anyMethod() {
141         return new AnyMethodMatcherImpl();
142       }
143     
144       public static ConstructorMatcher constructor() {
145         return new ConstructorMatcherImpl();
146       }
147     }
