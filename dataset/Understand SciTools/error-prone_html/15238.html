59     /**
60      * Implementation of a template to match and replace an expression anywhere in an AST.
61      *
62      * @author lowasser@google.com (Louis Wasserman)
63      */
64     @AutoValue
65     public abstract class ExpressionTemplate extends Template<ExpressionTemplateMatch> 
66         implements Unifiable<JCExpression> {
67       private static final Logger logger = Logger.getLogger(ExpressionTemplate.class.toString());
68       
69       public static ExpressionTemplate create(
70           UExpression expression, UType returnType) {
71         return create(ImmutableMap.<String, UType>of(), expression, returnType);
72       }
73       
74       public static ExpressionTemplate create(
75           Map<String, ? extends UType> expressionArgumentTypes,
76           UExpression expression, UType returnType) {
77         return create(
78             ImmutableClassToInstanceMap.<Annotation>builder().build(),
79             ImmutableList.<UTypeVar>of(),
80             expressionArgumentTypes,
81             expression,
82             returnType);
83       }
84       
85       public static ExpressionTemplate create(
86           ImmutableClassToInstanceMap<Annotation> annotations,
87           Iterable<UTypeVar> typeVariables,
88           Map<String, ? extends UType> expressionArgumentTypes,
89           UExpression expression, UType returnType) {
90         return new AutoValue_ExpressionTemplate(
91             annotations,
92             ImmutableList.copyOf(typeVariables),
93             ImmutableMap.copyOf(expressionArgumentTypes),
94             expression,
95             returnType);
96       }
97     
98       abstract UExpression expression();
99       abstract UType returnType();
100       
101       public boolean generateNegation() {
102         return annotations().containsKey(AlsoNegation.class);
103       }
104       
105       public ExpressionTemplate negation() {
106         checkState(returnType().equals(UPrimitiveType.BOOLEAN),
107             "Return type must be boolean to generate negation, but was %s", returnType());
108         return create(annotations(), templateTypeVariables(), expressionArgumentTypes(), 
109             expression().negate(), returnType());
110       }
111     
112       /**
113        * Returns the matches of this template against the specified target AST.
114        */
115       @Override
116       public Iterable<ExpressionTemplateMatch> match(JCTree target, Context context) {
117         if (target instanceof JCExpression) {
118           JCExpression targetExpr = (JCExpression) target;
119           Optional<Unifier> unifier = unify(targetExpr, new Unifier(context)).first();
120           if (unifier.isPresent()) {
121             return ImmutableList.of(new ExpressionTemplateMatch(targetExpr, unifier.get()));
122           }
123         }
124         return ImmutableList.of();
125       }
126     
127       @Override
128       public Choice<Unifier> unify(final JCExpression target, Unifier unifier) {
129         return expression().unify(target, unifier)
130             .thenOption(new Function<Unifier, Optional<Unifier>>() {
131     
132               @Override
133               public Optional<Unifier> apply(Unifier unifier) {
134                 Inliner inliner = unifier.createInliner();
135                 try {
136                   List<Type> expectedTypes = expectedTypes(inliner);
137                   List<Type> actualTypes = actualTypes(inliner);
138                   /*
139                    * TODO(cushon): the following is not true in javac8, which can apply target-typing to
140                    * nested method invocations.
141                    *
142                    * The Java compiler's type inference doesn't directly take into account the expected
143                    * return type, so we test the return type by treating the expected return type as an
144                    * extra method argument, and the actual type of the return expression as its actual
145                    * value.
146                    */
147                   if (target.type.getTag() != TypeTag.VOID) {
148                     expectedTypes = expectedTypes.prepend(returnType().inline(inliner));
149                     actualTypes = actualTypes.prepend(target.type);
150                   }
151     
152                   return typecheck(unifier, inliner, new Warner(target), expectedTypes, actualTypes);
153                 } catch (CouldNotResolveImportException e) {
154                   logger.log(FINE, "Failure to resolve import", e);
155                   return Optional.absent();
156                 }
157               }
158             });
159       }
160     
161       /**
162        * Generates a {@link SuggestedFix} replacing the specified match (usually of another template)
163        * with this template.
164        */
165       @Override
166       public Fix replace(ExpressionTemplateMatch match) {
167         Inliner inliner = match.createInliner();
168         Context context = inliner.getContext();
169         if (annotations().containsKey(UseImportPolicy.class)) {
170           ImportPolicy.bind(context, annotations().getInstance(UseImportPolicy.class).value());
171         } else {
172           ImportPolicy.bind(context, ImportPolicy.IMPORT_TOP_LEVEL);
173         }
174         int prec = getPrecedence(match.getLocation(), context);
175         SuggestedFix.Builder fix = SuggestedFix.builder();
176         try {
177           StringWriter writer = new StringWriter();
178           pretty(inliner.getContext(), writer).printExpr(expression().inline(inliner), prec);
179           fix.replace(match.getLocation(), writer.toString());
180         } catch (CouldNotResolveImportException e) {
181           logger.log(SEVERE, "Failure to resolve in replacement", e);
182         } catch (IOException e) {
183           throw new RuntimeException(e);
184         }
185         return addImports(inliner, fix);
186       }

188       /**
189        * Returns the precedence level appropriate for unambiguously printing
190        * leaf as a subexpression of its parent.
191        */
192       private static int getPrecedence(JCTree leaf, Context context) {
193         JCCompilationUnit comp = context.get(JCCompilationUnit.class);
194         JCTree parent = TreeInfo.pathFor(leaf, comp).get(1);
195     
196         // In general, this should match the logic in com.sun.tools.javac.tree.Pretty.
197         //
198         // TODO(mdempsky): There are probably cases where we could omit parentheses
199         // by tweaking the returned precedence, but they need careful review.
200         // For example, consider a template to replace "add(a, b)" with "a + b",
201         // which applied to "x + add(y, z)" would result in "x + (y + z)".
202         // In most cases, we'd likely prefer "x + y + z" instead, but those aren't
203         // always equivalent: "0L + (Integer.MIN_VALUE + Integer.MIN_VALUE)" yields
204         // a different value than "0L + Integer.MIN_VALUE + Integer.MIN_VALUE" due
205         // to integer promotion rules.
206     
207         if (parent instanceof JCConditional) {
208           // This intentionally differs from Pretty, because Pretty appears buggy:
209           // http://mail.openjdk.java.net/pipermail/compiler-dev/2013-September/007303.html
210           JCConditional conditional = (JCConditional) parent;
211           return TreeInfo.condPrec + ((conditional.cond == leaf) ? 1 : 0);
212         } else if (parent instanceof JCAssign) {
213           JCAssign assign = (JCAssign) parent;
214           return TreeInfo.assignPrec + ((assign.lhs == leaf) ? 1 : 0);
215         } else if (parent instanceof JCAssignOp) {
216           JCAssignOp assignOp = (JCAssignOp) parent;
217           return TreeInfo.assignopPrec + ((assignOp.lhs == leaf) ? 1 : 0);
218         } else if (parent instanceof JCUnary) {
219           return TreeInfo.opPrec(parent.getTag());
220         } else if (parent instanceof JCBinary) {
221           JCBinary binary = (JCBinary) parent;
222           return TreeInfo.opPrec(parent.getTag()) + ((binary.rhs == leaf) ? 1 : 0);
223         } else if (parent instanceof JCTypeCast) {
224           JCTypeCast typeCast = (JCTypeCast) parent;
225           return (typeCast.expr == leaf) ? TreeInfo.prefixPrec : TreeInfo.noPrec;
226         } else if (parent instanceof JCInstanceOf) {
227           JCInstanceOf instanceOf = (JCInstanceOf) parent;
228           return TreeInfo.ordPrec + ((instanceOf.clazz == leaf) ? 1 : 0);
229         } else if (parent instanceof JCArrayAccess) {
230           JCArrayAccess arrayAccess = (JCArrayAccess) parent;
231           return (arrayAccess.indexed == leaf) ? TreeInfo.postfixPrec : TreeInfo.noPrec;
232         } else if (parent instanceof JCFieldAccess) {
233           JCFieldAccess fieldAccess = (JCFieldAccess) parent;
234           return (fieldAccess.selected == leaf) ? TreeInfo.postfixPrec : TreeInfo.noPrec;
235         } else {
236           return TreeInfo.noPrec;
237         }
238       }
