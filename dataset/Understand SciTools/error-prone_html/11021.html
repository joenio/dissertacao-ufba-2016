18     
19     import com.google.errorprone.scanner.ScannerSupplier;
20     
21     import com.sun.tools.javac.main.Main.Result;
22     import com.sun.tools.javac.util.Context;
23     
24     import java.io.PrintWriter;
25     import java.util.Arrays;
26     import java.util.List;
27     
28     import javax.annotation.processing.Processor;
29     import javax.tools.DiagnosticListener;
30     import javax.tools.JavaFileObject;
31     
32     /**
33      * Wraps {@link com.google.errorprone.ErrorProneCompiler}.
34      */
35     public class ErrorProneTestCompiler {
36     
37       /** Wraps {@link com.google.errorprone.ErrorProneCompiler.Builder} */
38       public static class Builder {
39     
40         final ErrorProneCompiler.Builder wrappedCompilerBuilder = new ErrorProneCompiler.Builder();
41     
42         public ErrorProneTestCompiler build() {
43           return new ErrorProneTestCompiler(wrappedCompilerBuilder.build());
44         }
45     
46         public Builder listenToDiagnostics(DiagnosticListener<? super JavaFileObject> listener) {
47           wrappedCompilerBuilder.listenToDiagnostics(listener);
48           return this;
49         }
50     
51         public Builder report(ScannerSupplier scannerSupplier) {
52           wrappedCompilerBuilder.report(scannerSupplier);
53           return this;
54         }
55     
56         public Builder named(String test) {
57           wrappedCompilerBuilder.named(test);
58           return this;
59         }
60     
61         public Builder redirectOutputTo(PrintWriter printWriter) {
62           wrappedCompilerBuilder.redirectOutputTo(printWriter);
63           return this;
64         }
65       }
66     
67       private final ErrorProneCompiler compiler;
68       private final ErrorProneInMemoryFileManager fileManager = new ErrorProneInMemoryFileManager();
69     
70       public ErrorProneInMemoryFileManager fileManager() {
71         return fileManager;
72       }
73     
74       private ErrorProneTestCompiler(ErrorProneCompiler compiler) {
75         this.compiler = compiler;
76       }
77     
78       public Result compile(List<JavaFileObject> sources) {
79         return compile(sources, null);
80       }
81     
82       public Result compile(String[] args, List<JavaFileObject> sources) {
83         return compile(args, sources, null);
84       }
85     
86       public Result compile(List<JavaFileObject> sources, List<? extends Processor> processors) {
87         return compile(new String[]{}, sources, processors);
88       }
89     
90       public Result compile(String[] args, List<JavaFileObject> sources, List<? extends Processor>
91           processors) {
92         Context context = new Context();
93         List<String> processedArgs =
94             CompilationTestHelper.disableImplicitProcessing(Arrays.asList(args));
95         String[] argsArray = processedArgs.toArray(new String[processedArgs.size()]);
96         return compiler.run(argsArray, context, fileManager, sources, processors);
97       }
98     }
