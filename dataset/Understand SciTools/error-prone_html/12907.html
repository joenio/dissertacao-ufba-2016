66       public Inliner(Context context, Bindings bindings) {
67         this.context = new SubContext(context);
68         this.bindings = new Bindings(bindings).unmodifiable();
69         this.importsToAdd = Sets.newHashSet();
70         this.staticImportsToAdd = Sets.newHashSet();
71         this.typeVarCache = Maps.newHashMap();
72       }

84       public ClassSymbol resolveClass(CharSequence qualifiedClass) 
85           throws CouldNotResolveImportException {
86         try {
87           Symbol symbol = JavaCompiler.instance(context).resolveIdent(qualifiedClass.toString());
88           if (symbol.equals(symtab().errSymbol) || !(symbol instanceof ClassSymbol)) {
89             throw new CouldNotResolveImportException(qualifiedClass);
90           } else {
91             return (ClassSymbol) symbol;
92           }
93         } catch (NullPointerException e) {
94           throw new CouldNotResolveImportException(qualifiedClass);
95         }
96       }

134       private static final Types.SimpleVisitor<JCExpression, Inliner> INLINE_AS_TREE =
135           new Types.SimpleVisitor<JCExpression, Inliner>() {
136             @Override
137             public JCExpression visitType(Type t, Inliner inliner) {
138               return inliner.maker().Type(t);
139             }
140     
141             @Override
142             public JCExpression visitClassType(ClassType type, Inliner inliner) {
143               ClassSymbol classSym = (ClassSymbol) type.tsym;
144               JCExpression classExpr = inliner.importPolicy().classReference(
145                   inliner, classSym.outermostClass().getQualifiedName().toString(),
146                   classSym.getQualifiedName().toString());
147               List<JCExpression> argExprs = List.nil();
148               for (Type argType : type.getTypeArguments()) {
149                 argExprs = argExprs.append(visit(argType, inliner));
150               }
151               return argExprs.isEmpty()
152                   ? classExpr
153                   : inliner.maker().TypeApply(classExpr, argExprs);
154             }
155     
156             @Override
157             public JCExpression visitWildcardType(WildcardType type, Inliner inliner) {
158               TreeMaker maker = inliner.maker();
159               return maker.Wildcard(
160                   maker.TypeBoundKind(type.kind),
161                   visit(type.baseType(), inliner));
162             }
163     
164             @Override
165             public JCExpression visitArrayType(ArrayType type, Inliner inliner) {
166               return inliner.maker().TypeArray(visit(type.getComponentType(), inliner));
167             }
168           };

170       /**
171        * Inlines the syntax tree representing the specified type.
172        */
173       public JCExpression inlineAsTree(Type type) {
174         return INLINE_AS_TREE.visit(type, this);
175       }

177       public <V> V getBinding(Key<V> key) {
178         V value = bindings.getBinding(key);
179         if (value == null) {
180           throw new IllegalStateException("No binding for " + key);
181         }
182         return value;
183       }
184       
185       public <V> Optional<V> getOptionalBinding(Key<V> key) {
186         return Optional.fromNullable(bindings.getBinding(key));
187       }

189       public <R> com.sun.tools.javac.util.List<R>
190           inlineList(Iterable<? extends Inlineable<? extends R>> elements)
191           throws CouldNotResolveImportException {
192         ListBuffer<R> result = new ListBuffer<>();
193         for (Inlineable<? extends R> e : elements) {
194           if (e instanceof URepeated) {
195             // URepeated is bound to a list of expressions.
196             URepeated repeated = (URepeated) e;
197             for (JCExpression expr : getBinding(repeated.key())) {
198               @SuppressWarnings("unchecked") 
199               // URepeated is an Inlineable<JCExpression>, so if e is also an Inlineable<? extends R>,
200               // then R must be ? super JCExpression.
201               R r = (R) expr;
202               result.append(r);
203             }
204           } else {
205             result.append(e.inline(this));
206           }
207         }
208         return result.toList();
209       }

219       public TypeVar inlineAsVar(UTypeVar var) throws CouldNotResolveImportException {
220         /*
221          * In order to handle recursively bounded type variables without a stack overflow,
222          * we first cache a type var with no bounds, then we inline the bounds.
223          */
224         TypeVar typeVar = typeVarCache.get(var.getName());
225         if (typeVar != null) {
226           return typeVar;
227         }
228         Name name = asName(var.getName());
229         TypeSymbol sym = new TypeVariableSymbol(0, name, null, symtab().noSymbol);
230         typeVar = new TypeVar(sym, null, null);
231         sym.type = typeVar;
232         typeVarCache.put(var.getName(), typeVar);
233         // Any recursive uses of var will point to the same TypeVar object generated above.
234         typeVar.bound = var.getUpperBound().inline(this);
235         typeVar.lower = var.getLowerBound().inline(this);
236         return typeVar;
237       }
238       
239       Type inlineTypeVar(UTypeVar var) throws CouldNotResolveImportException {
240         Optional<TypeWithExpression> typeVarBinding = getOptionalBinding(var.key());
241         if (typeVarBinding.isPresent()) {
242           return typeVarBinding.get().type();
243         } else {
244           return inlineAsVar(var);
245         }
246       }
