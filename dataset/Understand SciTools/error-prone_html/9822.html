42     /**
43      * Ban use of YYYY in a SimpleDateFormat pattern, unless it is being used for a week date.
44      * Otherwise the user almost certainly meant yyyy instead.  See the summary in the {@link
45      * BugPattern} below for more details.
46      *
47      * <p>This bug caused a Twitter outage in December 2014.
48      */
49     @BugPattern(name = "MisusedWeekYear",
50         summary = "Use of \"YYYY\" (week year) in a date pattern without \"ww\" (week in year). "
51             + "You probably meant to use \"yyyy\" (year) instead.",
52         explanation = "\"YYYY\" in a date pattern means \"week year\".  The week year is defined to "
53             + "begin at the beginning of the week that contains the year's first Thursday.  For "
54             + "example, the week year 2015 began on Monday, December 29, 2014, since January 1, 2015, "
55             + "was on a Thursday.\n\n"
56             + "\"Week year\" is intended to be used for week dates, e.g. \"2015-W01-1\", but is often "
57             + "mistakenly used for calendar dates, e.g. 2014-12-29, in which case the year may be "
58             + "incorrect during the last week of the year.  If you are formatting anything other than "
59             + "a week date, you should use the year specifier \"yyyy\" instead.",
60         category = JDK, severity = ERROR, maturity = MATURE)
61     public class MisusedWeekYear extends BugChecker
62         implements MethodInvocationTreeMatcher, NewClassTreeMatcher {
63       
64       private static final String JAVA_SIMPLE_DATE_FORMAT = "java.text.SimpleDateFormat";
65       private static final String ICU_SIMPLE_DATE_FORMAT = "com.ibm.icu.text.SimpleDateFormat";
66     
67       private static final Matcher<NewClassTree> simpleDateFormatConstructorMatcher =
68           Matchers.<NewClassTree>anyOf(
69               constructor().forClass(JAVA_SIMPLE_DATE_FORMAT)
70                   .withParameters("java.lang.String"),
71               constructor().forClass(JAVA_SIMPLE_DATE_FORMAT)
72                   .withParameters("java.lang.String", "java.text.DateFormatSymbols"),
73               constructor().forClass(JAVA_SIMPLE_DATE_FORMAT)
74                   .withParameters("java.lang.String", "java.util.Locale"),
75               constructor().forClass(ICU_SIMPLE_DATE_FORMAT)
76                   .withParameters("java.lang.String"),
77               constructor().forClass(ICU_SIMPLE_DATE_FORMAT)
78                   .withParameters("java.lang.String", "com.ibm.icu.text.DateFormatSymbols"),
79               constructor().forClass(ICU_SIMPLE_DATE_FORMAT)
80                   .withParameters(
81                       "java.lang.String", 
82                       "com.ibm.icu.text.DateFormatSymbols", 
83                       "com.ibm.icu.util.ULocale"),
84               constructor().forClass(ICU_SIMPLE_DATE_FORMAT)
85                   .withParameters("java.lang.String", "java.util.Locale"),
86               constructor().forClass(ICU_SIMPLE_DATE_FORMAT)
87                   .withParameters("java.lang.String", "java.lang.String", "com.ibm.icu.util.ULocale"),
88               constructor().forClass(ICU_SIMPLE_DATE_FORMAT)
89                   .withParameters("java.lang.String", "com.ibm.icu.util.ULocale"));
90     
91       private static final Matcher<ExpressionTree> applyPatternMatcher =
92           Matchers.<ExpressionTree>anyOf(
93               instanceMethod().onExactClass(JAVA_SIMPLE_DATE_FORMAT).named("applyPattern"),
94               instanceMethod().onExactClass(JAVA_SIMPLE_DATE_FORMAT).named("applyLocalizedPattern"),
95               instanceMethod().onExactClass(ICU_SIMPLE_DATE_FORMAT).named("applyPattern"),
96               instanceMethod().onExactClass(ICU_SIMPLE_DATE_FORMAT).named("applyLocalizedPattern"));
97     
98       /**
99        * Match uses of SimpleDateFormat.applyPattern and SimpleDateFormat.applyLocalizedPattern in
100        * which the pattern passed in contains YYYY but not ww, signifying that it was not intended to
101        * be a week date.  If the pattern is a string literal, suggest replacing the YYYY with yyyy.
102        * If the pattern is a constant, don't give a suggested fix since the fix is nonlocal.
103        */
104       @Override
105       public Description matchMethodInvocation(MethodInvocationTree tree, VisitorState state) {
106         if (!applyPatternMatcher.matches(tree, state)) {
107           return Description.NO_MATCH;
108         }
109     
110         return constructDescription(tree, tree.getArguments().get(0));
111       }
112     
113       /**
114        * Match uses of the SimpleDateFormat constructor in which the pattern passed in contains
115        * YYYY but not ww, signifying that it was not intended to be a week date.  If the pattern
116        * is a string literal, suggest replacing the YYYY with yyyy.  If the pattern is a constant, don't
117        * give a suggested fix since the fix is nonlocal.
118        */
119       @Override
120       public Description matchNewClass(NewClassTree tree, VisitorState state) {
121         if (!simpleDateFormatConstructorMatcher.matches(tree, state)) {
122           return Description.NO_MATCH;
123         }
124     
125         return constructDescription(tree, tree.getArguments().get(0));
126       }
127     
128       /**
129        * Given the {@link ExpressionTree} representing the pattern argument to the various
130        * methods in SimpleDateFormat that accept a pattern, construct the description for this matcher 
131        * to return.  May be {@link Description#NO_MATCH} if the pattern does not have a constant value,
132        * does not use the week year format specifier, or is in proper week date format.
133        */
134       private Description constructDescription(Tree tree, ExpressionTree patternArg) {
135         String pattern = (String) ASTHelpers.constValue((JCTree) patternArg);
136         if (pattern != null && pattern.contains("YYYY") && !pattern.contains("ww")) {
137           if (patternArg.getKind() == Kind.STRING_LITERAL) {
138             String replacement = patternArg.toString().replace("YYYY", "yyyy");
139             return describeMatch(tree, SuggestedFix.replace(patternArg, replacement));
140           } else {
141             return describeMatch(tree);
142           }
143         }
144     
145         return Description.NO_MATCH;
146       }
147     }
