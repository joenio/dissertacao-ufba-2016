17     package com.google.errorprone.fixes;
18     import static org.hamcrest.CoreMatchers.equalTo;
19     import static org.junit.Assert.assertNull;
20     import static org.junit.Assert.assertThat;

22     import static org.mockito.Mockito.when;
23     
24     import com.sun.tools.javac.tree.EndPosTable;
25     import com.sun.tools.javac.tree.JCTree;
26     import com.sun.tools.javac.util.Position;
27     
28     import org.junit.Test;
29     import org.junit.runner.RunWith;
30     import org.mockito.Mock;
31     import org.mockito.runners.MockitoJUnitRunner;
32     
33     import java.util.HashMap;
34     import java.util.Map;
35     
36     /**
37      * @author alexeagle@google.com (Alex Eagle)
38      */
39     @RunWith(MockitoJUnitRunner.class)
40     public class AppliedFixTest {
41       @Mock JCTree node;
42       final EndPosTable endPositions = new EndPosTable() {
43         
44         final Map<JCTree, Integer> map = new HashMap<>();
45         
46         @Override
47         public void storeEnd(JCTree tree, int endpos) {
48           map.put(tree, endpos);
49         }
50         
51         @Override
52         public int replaceTree(JCTree oldtree, JCTree newtree) {
53           Integer endpos = map.get(oldtree);
54           if (endpos == null) {
55             endpos = Position.NOPOS;
56           }
57           map.put(newtree, endpos);
58           return endpos;
59         }
60         
61         @Override
62         public int getEndPos(JCTree tree) {
63           Integer result = map.get(tree);
64           if (result == null) {
65             result = Position.NOPOS;
66           }
67           return result;
68         }
69       };
70       
71       @Test
72       public void shouldApplySingleFixOnALine() {
73         when(node.getStartPosition()).thenReturn(11);
74         when(node.getEndPosition(same(endPositions))).thenReturn(14);
75     
76         AppliedFix fix = AppliedFix.fromSource("import org.me.B;", endPositions)
77             .apply(SuggestedFix.delete(node));
78         assertThat(fix.getNewCodeSnippet().toString(), equalTo("import org.B;"));
79       }
80     
81       @Test
82       public void shouldReportOnlyTheChangedLineInNewSnippet() {
83         when(node.getStartPosition()).thenReturn(25);
84         when(node.getEndPosition(same(endPositions))).thenReturn(26);
85     
86         AppliedFix fix = AppliedFix.fromSource(
87             "public class Foo {\n" +
88             "  int 3;\n" +
89             "}", endPositions)
90             .apply(SuggestedFix.builder()
91                 .prefixWith(node, "three")
92                 .postfixWith(node, "tres")
93                 .build());
94         assertThat(fix.getNewCodeSnippet().toString(), equalTo("int three3tres;"));
95       }
96     
97       @Test(expected = IllegalStateException.class)
98       public void shouldThrowExceptionOnEmptyFix() {
99         AppliedFix.fromSource(
100             "public class Foo {}", endPositions)
101             .apply(SuggestedFix.builder().build());
102       }
103     
104       @Test
105       public void shouldReturnNullOnImportOnlyFix() {
106         AppliedFix fix = AppliedFix.fromSource(
107             "public class Foo {}", endPositions)
108             .apply(SuggestedFix.builder().addImport("foo.bar.Baz").build());
109         assertNull(fix);
110       }
111     
112       @Test(expected = IllegalArgumentException.class)
113       public void shouldThrowExceptionOnIllegalRange() {
114         AppliedFix.fromSource("public class Foo {}", endPositions)
115             .apply(SuggestedFix.replace(0, -1, ""));
116       }
117     }
