26     import com.sun.tools.javac.code.Symbol;
27     import com.sun.tools.javac.code.Type;

32     import javax.annotation.Nullable;

34     /**
35      * Logic for inspecting static imports used by
36      * {@link NonCanonicalStaticImport}, {@link NonCanonicalStaticMemberImport},
37      * and {@link UnnecessaryStaticImport}.
38      */
39     final class StaticImports {
40     
41       @AutoValue
42       public abstract static class StaticImportInfo {
43         /** @return the fully qualified name used to import the type (possibly non-canonical) */
44         abstract String importedName();
45     
46         /** @return the fully-qualified canonical name of the type */
47         abstract String canonicalName();
48     
49         /**
50          * The field or variable symbol for a static non-type member import.
51          */
52         abstract Optional<Symbol> member();
53     
54         /**
55          * Returns true if the import is canonical, i.e. the fully qualified name used to import the
56          * type matches the scopes it was declared in.
57          */
58         public boolean isCanonical() {
59           return canonicalName().equals(importedName());
60         }
61     
62         /** Builds the canonical import statement for the type. */
63         public String importStatement() {
64           if (member().isPresent()) {
65             Symbol member = member().get();
66             return String.format("import static %s.%s;", canonicalName(), member.getSimpleName());
67           } else {
68             return String.format("import %s;", canonicalName());
69           }
70         }
71     
72         private static StaticImportInfo create(
73             String importedName, String canonicalName, Optional<Symbol> member) {
74           return new AutoValue_StaticImports_StaticImportInfo(importedName, canonicalName, member);
75         }
76       }
77     
78       /**
79        * Returns a {@link StaticImports} if the given import is a static single-type import.
80        * Returns {@code null} otherwise, e.g. because the import is non-static, or an on-demand
81        * import, or statically imports a field or method.
82        */
83       @Nullable
84       public static StaticImportInfo tryCreate(ImportTree tree, VisitorState state) {
85         if (!tree.isStatic()) {
86           return null;
87         }
88         if (!(tree.getQualifiedIdentifier() instanceof JCTree.JCFieldAccess)) {
89           return null;
90         }
91         JCTree.JCFieldAccess access = (JCTree.JCFieldAccess) tree.getQualifiedIdentifier();
92         String importedName = access.toString();
93         Type result = state.getTypeFromString(importedName);
94         if (result == null) {
95           // If the full imported name isn't a type, it might be a field or
96           // method:
97           return tryAsStaticMember(access, state);
98         }
99         String canonicalName = state.getTypes().erasure(result).toString();
100         if (canonicalName == null) {
101           return null;
102         }
103         return StaticImportInfo.create(importedName, canonicalName, Optional.<Symbol>absent());
104       }
105     
106       /**
107        * Returns a {@code StaticImportInfo} for a static field or method import.
108        */
109       private static StaticImportInfo tryAsStaticMember(
110           JCTree.JCFieldAccess access, VisitorState state) {
111         if (access.getIdentifier().contentEquals("*")) {
112           // Java doesn't allow non-canonical types inside wildcard imports,
113           // so there's nothing to do here.
114           return null;
115         }
116         String importedTypeName = access.getExpression().toString();
117         Type importedType = state.getTypeFromString(importedTypeName);
118         if (importedType == null) {
119           return null;
120         }
121         Type canonicalType = state.getTypes().erasure(importedType);
122         if (canonicalType == null) {
123           return null;
124         }
125         Symbol.TypeSymbol baseType;
126         {
127           Symbol sym = ASTHelpers.getSymbol(access.getExpression());
128           if (!(sym instanceof Symbol.TypeSymbol)) {
129             return null;
130           }
131           baseType = (Symbol.TypeSymbol) sym;
132         }
133         Symbol.PackageSymbol pkgSym =
134             ((JCTree.JCCompilationUnit) state.getPath().getCompilationUnit()).packge;
135         Symbol member = lookup(baseType, baseType, access.getIdentifier(), state.getTypes(), pkgSym);
136         if (member == null) {
137           return null;
138         }
139         Type canonicalOwner = state.getTypes().erasure(member.owner.type);
140         if (canonicalOwner == null) {
141           return null;
142         }
143         return StaticImportInfo.create(
144             importedTypeName, canonicalOwner.toString(), Optional.of(member));
145       }
146     
147       /**
148        * Looks for a field or method with the given {@code identifier}, in
149        * @code typeSym} or one of it's super-types or super-interfaces,
150        * and that is visible from the {@code start} symbol.
151        */
152       // TODO(cushon): does javac really not expose this anywhere?
153       //
154       // Resolve.resolveInternal{Method,Field} almost work, but we don't want
155       // to filter on method signature.
156       private static Symbol lookup(
157           Symbol.TypeSymbol typeSym,
158           Symbol.TypeSymbol start,
159           Name identifier,
160           Types types,
161           Symbol.PackageSymbol pkg) {
162         if (typeSym == null) {
163           return null;
164         }
165     
166         Symbol result = lookup(types.supertype(typeSym.type).tsym, start, identifier, types, pkg);
167         if (result != null) {
168           return result;
169         }
170     
171         for (Type i : types.interfaces(typeSym.type)) {
172           result = lookup(i.tsym, start, identifier, types, pkg);
173           if (result != null) {
174             return result;
175           }
176         }
177     
178         OUTER:
179         for (Symbol member : typeSym.members().getSymbolsByName(identifier)) {
180           if (!member.isStatic()) {
181             continue;
182           }
183           switch ((int) (member.flags() & Flags.AccessFlags)) {
184             case Flags.PUBLIC:
185               break;
186             case Flags.PRIVATE:
187               continue OUTER;
188             case 0:
189             case Flags.PROTECTED:
190               if (member.packge() != pkg) {
191                 continue OUTER;
192               }
193             default:
194               break;
195           }
196           if (member.isMemberOf(start, types)) {
197             return member;
198           }
199         }
200     
201         return null;
202       }
203     }
