17     package com.google.errorprone.refaster;
18     
19     import com.google.auto.value.AutoValue;
20     import com.google.common.collect.ImmutableList;
21     
22     import com.sun.tools.javac.code.Type;
23     import com.sun.tools.javac.code.Type.ForAll;
24     import com.sun.tools.javac.code.Types;
25     
26     import java.util.List;
27     
28     /**
29      * {@link UType} version of {@link ForAll}.
30      *
31      * @author lowasser@google.com (Louis Wasserman)
32      */
33     @AutoValue
34     public abstract class UForAll extends UType {
35       public static UForAll create(List<UTypeVar> typeVars, UType quantifiedType) {
36         return new AutoValue_UForAll(ImmutableList.copyOf(typeVars), quantifiedType);
37       }
38     
39       public abstract List<UTypeVar> getTypeVars();
40     
41       public abstract UType getQuantifiedType();
42     
43       @Override
44       public Choice<Unifier> visitForAll(ForAll target, Unifier unifier) {
45         Types types = unifier.types();
46         try {
47           Type myType = inline(new Inliner(unifier.getContext(), Bindings.create()));
48           return Choice.condition(
49               types.overrideEquivalent(types.erasure(myType), types.erasure(target)),
50               unifier);
51         } catch (CouldNotResolveImportException e) {
52           return Choice.none();
53         }
54       }
55     
56       @Override
57       public Type inline(Inliner inliner) throws CouldNotResolveImportException {
58         return new ForAll(
59             inliner.<Type>inlineList(getTypeVars()), getQuantifiedType().inline(inliner));
60       }
61     }
