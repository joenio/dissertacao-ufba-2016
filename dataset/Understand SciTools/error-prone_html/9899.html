49     /**
50      * Scanner that outputs suggested fixes generated by a {@code RefasterMatcher}.
51      * 
52      * @author lowasser@google.com (Louis Wasserman)
53      */
54     @AutoValue
55     abstract class RefasterScanner<M extends TemplateMatch, T extends Template<M>> 
56         extends TreeScanner<Void, Context> {
57       static <M extends TemplateMatch, T extends Template<M>> RefasterScanner<M, T> create(
58           RefasterRule<M, T> rule, DescriptionListener listener) {
59         return new AutoValue_RefasterScanner<>(rule, listener);
60       }
61       
62       abstract RefasterRule<M, T> rule();
63     
64       abstract DescriptionListener listener();
65       
66       @Override
67       public Void visitClass(ClassTree node, Context context) {
68         Symbol sym = ASTHelpers.getSymbol(node);
69         if (sym == null || !sym.getQualifiedName().contentEquals(rule().qualifiedTemplateClass())) {
70           ListBuffer<JCStatement> statements = new ListBuffer<>();
71           for (Tree tree : node.getMembers()) {
72             if (tree instanceof JCStatement) {
73               statements.append((JCStatement) tree);
74             } else {
75               tree.accept(this, context);
76             }
77           }
78           scan(TreeMaker.instance(context).Block(0, statements.toList()), context);
79         }
80         return null;
81       }
82     
83       @Override
84       public Void scan(Tree tree, Context context) {
85         if (tree == null) {
86           return null;
87         }
88         JCCompilationUnit compilationUnit = context.get(JCCompilationUnit.class);
89         for (T beforeTemplate : rule().beforeTemplates()) {
90           matchLoop: for (M match : beforeTemplate.match((JCTree) tree, context)) {
91             if (rule().rejectMatchesWithComments()) {
92               String matchContents = match.getRange(compilationUnit);
93               JavaTokenizer tokenizer = new JavaTokenizer(ScannerFactory.instance(context), 
94                   CharBuffer.wrap(matchContents)) {};
95               for (Token token = tokenizer.readToken(); token.kind != TokenKind.EOF;
96                   token = tokenizer.readToken()) {
97                 if (token.comments != null && !token.comments.isEmpty()) {
98                   continue matchLoop;
99                 }
100               }
101             }
102             Description.Builder builder = Description.builder(match.getLocation(),
103                 rule().qualifiedTemplateClass(), "", SeverityLevel.WARNING, "");
104                 
105             if (rule().afterTemplates().isEmpty()) {
106               builder.addFix(SuggestedFix.prefixWith(match.getLocation(), "/* match found */ "));
107             } else {
108               for (T afterTemplate : rule().afterTemplates()) {
109                 builder.addFix(afterTemplate.replace(match));
110               }
111             }
112             listener().onDescribed(builder.build());
113           }
114         }
115         return super.scan(tree, context);
116       }
117       
118       private static final SimpleTreeVisitor<Tree, Void> SKIP_PARENS =
119           new SimpleTreeVisitor<Tree, Void>() {
120             @Override
121             public Tree visitParenthesized(ParenthesizedTree node, Void v) {
122               return node.getExpression().accept(this, null);
123             }
124     
125             @Override
126             protected Tree defaultAction(Tree node, Void v) {
127               return node;
128             }
129           };
130     
131       /* 
132        * Matching on the parentheses surrounding the condition of an if, while, or do-while
133        * is nonsensical, as those parentheses are obligatory and should never be changed.
134        */
135     
136       @Override
137       public Void visitDoWhileLoop(DoWhileLoopTree node, Context context) {
138         scan(node.getStatement(), context);
139         scan(SKIP_PARENS.visit(node.getCondition(), null), context);
140         return null;
141       }
142     
143       @Override
144       public Void visitWhileLoop(WhileLoopTree node, Context context) {
145         scan(SKIP_PARENS.visit(node.getCondition(), null), context);
146         scan(node.getStatement(), context);
147         return null;
148       }
149     
150       @Override
151       public Void visitSynchronized(SynchronizedTree node, Context context) {
152         scan(SKIP_PARENS.visit(node.getExpression(), null), context);
153         scan(node.getBlock(), context);
154         return null;
155       }
156     
157       @Override
158       public Void visitIf(IfTree node, Context context) {
159         scan(SKIP_PARENS.visit(node.getCondition(), null), context);
160         scan(node.getThenStatement(), context);
161         scan(node.getElseStatement(), context);
162         return null;
163       }
164     }
