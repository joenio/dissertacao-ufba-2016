35     /**
36      * @author cushon@google.com (Liam Miller-Cushon)
37      */
38     @BugPattern(name = "GuardedByValidator",
39         summary = "Invalid @GuardedBy expression",
40         explanation = "@GuardedBy(lock) documents that a field or method should be accessed only with "
41           + "a specific lock held. The lock argument identifies the lock that should be held when "
42           + "accessing the annotated field or method. The possible values for lock are: "
43           + "* @GuardedBy(\"this\"), meaning the intrinsic lock on the containing object (the "
44           + "object of which the method or field is a member);"
45           + "* @GuardedBy(\"fieldName\"), meaning the lock associated with the object referenced by "
46           + "the named field, either an intrinsic lock (for fields that do not refer to a Lock) or an "
47           + "explicit Lock (for fields that refer to a Lock);"
48           + "* @GuardedBy(\"ClassName.fieldName\"), like @GuardedBy(\"fieldName\"), but referencing "
49           + "a lock object held in a static field of another class;"
50           + "* @GuardedBy(\"methodName()\"), meaning the lock object that is returned by calling the "
51           + "named method;"
52           + "* @GuardedBy(\"ClassName.class\"), meaning the class literal object for the named class.",
53           category = JDK, severity = ERROR, maturity = EXPERIMENTAL)
54     public class GuardedByValidator extends BugChecker implements VariableTreeMatcher,
55         MethodTreeMatcher {
56     
57       private static final String MESSAGE_FORMAT = "Invalid @GuardedBy expression: %s";
58       
59       @Override
60       public Description matchMethod(MethodTree tree, final VisitorState state) {
61         return validate(this, tree, state);
62       }
63     
64       @Override
65       public Description matchVariable(VariableTree tree, VisitorState state) {
66         // We only want to check field declarations. The VariableTree might be for a local or a
67         // parameter, but they won't have @GuardedBy annotations.
68         return validate(this, tree, state);
69       }
70     
71       static Description validate(BugChecker checker, Tree tree, VisitorState state) {
72         GuardedByValidationResult result = GuardedByUtils.isGuardedByValid(tree, state);
73         if (result.isValid()) {
74           return Description.NO_MATCH;
75         }
76         return BugChecker.buildDescriptionFromChecker(tree, checker)
77             .setMessage(String.format(MESSAGE_FORMAT, result.message()))
78             .build();
79       }
80     }
