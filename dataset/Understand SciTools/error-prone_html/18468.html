138       /**
139        * Returns a template based on a method. One-line methods starting with a {@code return} statement
140        * are guessed to be expression templates, and all other methods are guessed to be block
141        * templates.
142        */
143       public static Template<?> createTemplate(Context context, MethodTree decl) {
144         MethodSymbol declSym = ASTHelpers.getSymbol(decl);
145         ImmutableClassToInstanceMap<Annotation> annotations = UTemplater.annotationMap(declSym);
146         ImmutableMap<String, VarSymbol> freeExpressionVars = freeExpressionVariables(decl);
147         Context subContext = new SubContext(context);
148         final UTemplater templater = new UTemplater(freeExpressionVars, subContext);
149         ImmutableMap<String, UType> expressionVarTypes = ImmutableMap.copyOf(
150             Maps.transformValues(freeExpressionVars, new Function<VarSymbol, UType>() {
151               @Override
152               public UType apply(VarSymbol sym) {
153                 return templater.template(sym.type);
154               }
155             }));
156     
157         UType genericType = templater.template(declSym.type);
158         List<UTypeVar> typeParameters;
159         UMethodType methodType;
160         if (genericType instanceof UForAll) {
161           UForAll forAllType = (UForAll) genericType;
162           typeParameters = forAllType.getTypeVars();
163           methodType = (UMethodType) forAllType.getQuantifiedType();
164         } else if (genericType instanceof UMethodType) {
165           typeParameters = ImmutableList.of();
166           methodType = (UMethodType) genericType;
167         } else {
168           throw new IllegalArgumentException(
169               "Expected genericType to be either a ForAll or a UMethodType, but was " + genericType);
170         }
171     
172         List<? extends StatementTree> bodyStatements = decl.getBody().getStatements();
173         if (bodyStatements.size() == 1
174             && Iterables.getOnlyElement(bodyStatements).getKind() == Kind.RETURN
175             && context.get(REQUIRE_BLOCK_KEY) == null) {
176           ExpressionTree expression =
177               ((ReturnTree) Iterables.getOnlyElement(bodyStatements)).getExpression();
178           return ExpressionTemplate.create(
179               annotations, typeParameters, expressionVarTypes,
180               templater.template(expression), methodType.getReturnType());
181         } else {
182           List<UStatement> templateStatements = new ArrayList<>();
183           for (StatementTree statement : bodyStatements) {
184             templateStatements.add(templater.template(statement));
185           }
186           return BlockTemplate.create(annotations, 
187               typeParameters, expressionVarTypes, templateStatements);
188         }
189       }

191       public static ImmutableMap<String, VarSymbol> freeExpressionVariables(
192           MethodTree templateMethodDecl) {
193         ImmutableMap.Builder<String, VarSymbol> builder = ImmutableMap.builder();
194         for (VariableTree param : templateMethodDecl.getParameters()) {
195           builder.put(param.getName().toString(), ASTHelpers.getSymbol(param));
196         }
197         return builder.build();
198       }

218         if (trees == null) {
219           return null;
220         }
221         ImmutableList.Builder<Tree> builder = ImmutableList.builder();
222         for (Tree tree : trees) {

334         Symbol sym = ASTHelpers.getSymbol(tree);

384       private static Tree getSingleExplicitTypeArgument(MethodInvocationTree tree) {
385         if (tree.getTypeArguments().isEmpty()) {
386           throw new IllegalArgumentException("Methods in the Refaster class must be invoked with "
387               + "an explicit type parameter; for example, 'Refaster.<T>isInstance(o)'.");
388         }
389         return Iterables.getOnlyElement(tree.getTypeArguments());
390       }

392       static <T, U extends Unifiable<? super T>> boolean anyMatch(
393           U toUnify, T target, Unifier unifier) {
394         return toUnify.unify(target, unifier).first().isPresent();
395       }

397       @Override
398       public UExpression visitMethodInvocation(MethodInvocationTree tree, Void v) {
399         if (anyMatch(ANY_OF, tree.getMethodSelect(), new Unifier(context))) {
400           return UAnyOf.create(templateExpressions(tree.getArguments()));
401         } else if (anyMatch(IS_INSTANCE, tree.getMethodSelect(), new Unifier(context))) {
402           return UInstanceOf.create(template(Iterables.getOnlyElement(tree.getArguments())),
403               templateType(getSingleExplicitTypeArgument(tree)));
404         } else if (anyMatch(CLAZZ, tree.getMethodSelect(), new Unifier(context))) {
405           Tree typeArg = getSingleExplicitTypeArgument(tree);
406           return UMemberSelect.create(templateType(typeArg), "class",
407               UClassType.create("java.lang.Class", template(((JCTree) typeArg).type)));
408         } else if (anyMatch(NEW_ARRAY, tree.getMethodSelect(), new Unifier(context))) {
409           Tree typeArg = getSingleExplicitTypeArgument(tree);
410           ExpressionTree lengthArg = Iterables.getOnlyElement(tree.getArguments());
411           return UNewArray.create(templateType(typeArg), ImmutableList.of(template(lengthArg)), null);
412         } else if (anyMatch(ENUM_VALUE_OF, tree.getMethodSelect(), new Unifier(context))) {
413           Tree typeArg = getSingleExplicitTypeArgument(tree);
414           ExpressionTree strArg = Iterables.getOnlyElement(tree.getArguments());
415           return UMethodInvocation.create(UMemberSelect.create(templateType(typeArg),
416               "valueOf", UMethodType.create(template(((JCTree) typeArg).type),
417                   UClassType.create("java.lang.String"))), template(strArg));
418         }
419         Map<MethodSymbol, PlaceholderMethod> placeholderMethods = 
420             context.get(RefasterRuleBuilderScanner.PLACEHOLDER_METHODS_KEY);
421         if (placeholderMethods != null && placeholderMethods.containsKey(ASTHelpers.getSymbol(tree))) {
422           return UPlaceholderExpression.create(
423               placeholderMethods.get(ASTHelpers.getSymbol(tree)),
424               templateExpressions(tree.getArguments()));
425         } else {
426           return UMethodInvocation.create(template(tree.getMethodSelect()),
427               templateExpressions(tree.getArguments()));
428         }
429       }

539       @Override
540       public UExpression visitIdentifier(IdentifierTree tree, Void v) {
541         Symbol sym = ASTHelpers.getSymbol(tree);
542         if (sym instanceof ClassSymbol) {
543           return UClassIdent.create((ClassSymbol) sym);
544         } else if (sym != null && sym.isStatic()) {
545           return staticMember(sym);
546         } else if (freeVariables.containsKey(tree.getName().toString())) {
547           VarSymbol symbol = freeVariables.get(tree.getName().toString());
548           checkState(symbol == sym);
549           UExpression ident = UFreeIdent.create(tree.getName());
550           Matches matches = ASTHelpers.getAnnotation(symbol, Matches.class);
551           if (matches != null) {
552             ident = UMatches.create(getValue(matches), true, ident);
553           }
554           NotMatches notMatches = ASTHelpers.getAnnotation(symbol, NotMatches.class);
555           if (notMatches != null) {
556             ident = UMatches.create(getValue(notMatches), false, ident);
557           }
558           OfKind hasKind = ASTHelpers.getAnnotation(symbol, OfKind.class);
559           if (hasKind != null) {
560             EnumSet<Kind> allowed = EnumSet.copyOf(Arrays.asList(hasKind.value()));
561             ident = UOfKind.create(ident, ImmutableSet.copyOf(allowed));
562           }
563           // @Repeated annotations need to be checked last.
564           Repeated repeated = ASTHelpers.getAnnotation(symbol, Repeated.class);
565           if (repeated != null) {
566             ident = URepeated.create(tree.getName(), ident);
567           }
568           return ident;
569         }
570         if (sym == null) {
571           return UTypeVarIdent.create(tree.getName());
572         }
573         switch (sym.getKind()) {
574           case TYPE_PARAMETER:
575             return UTypeVarIdent.create(tree.getName());
576           default:
577             return ULocalVarIdent.create(tree.getName());
578         }
579       }

583        * {@link Matches} annotation.  This roundabout solution is recommended and explained by
584        * {@link Element#getAnnotation(Class)}.
585        */
586       static Class<? extends Matcher<? super ExpressionTree>> getValue(Matches matches) {
587         String name;
588         try {
589           matches.value();
590           throw new RuntimeException("unreachable");
591         } catch (MirroredTypeException e) {
592           DeclaredType type = (DeclaredType) e.getTypeMirror();
593           name = ((TypeElement) type.asElement()).getQualifiedName().toString();
594         }
595         try {
596           return asSubclass(Class.forName(name), new TypeToken<Matcher<? super ExpressionTree>>() {});
597         } catch (ClassNotFoundException|ClassCastException e) {
598           throw new RuntimeException(e);
599         }
600       }

602       /**
603        * Returns the {@link Class} instance for the {@link Matcher} associated with the provided
604        * {@link NotMatches} annotation.  This roundabout solution is recommended and explained by
605        * {@link Element#getAnnotation(Class)}.
606        */
607       static Class<? extends Matcher<? super ExpressionTree>> getValue(NotMatches matches) {
608         String name;
609         try {
610           matches.value();
611           throw new RuntimeException("unreachable");
612         } catch (MirroredTypeException e) {
613           DeclaredType type = (DeclaredType) e.getTypeMirror();
614           name = ((TypeElement) type.asElement()).getQualifiedName().toString();
615         }
616         try {
617           return asSubclass(Class.forName(name), new TypeToken<Matcher<? super ExpressionTree>>() {});
618         } catch (ClassNotFoundException|ClassCastException e) {
619           throw new RuntimeException(e);
620         }
621       }

623       /**
624        * Similar to {@link Class#asSubclass(Class)}, but it accepts a {@link TypeToken} so it handles
625        * generics better.
626        */
627       @SuppressWarnings("unchecked")
628       private static <T> Class<? extends T> asSubclass(Class<?> klass, TypeToken<T> token)
629           throws ClassCastException {
630         if (!token.isSupertypeOf(klass)) {
631           throw new ClassCastException(klass + " is not assignable to " + token);
632         }
633         return (Class<? extends T>) klass;
634       }

652       @Override
653       public UTry visitTry(TryTree tree, Void v) {
654         @SuppressWarnings({"unchecked", "rawtypes"})
655         List<UTree<?>> resources =
656             cast(templateTrees(tree.getResources()), (Class<UTree<?>>) (Class) UTree.class);
657         UBlock block = visitBlock(tree.getBlock(), null);
658         ImmutableList.Builder<UCatch> catchesBuilder = ImmutableList.builder();
659         for (CatchTree catchTree : tree.getCatches()) {
660           catchesBuilder.add(visitCatch(catchTree, null));
661         }
662         UBlock finallyBlock = 
663             (tree.getFinallyBlock() == null) ? null : visitBlock(tree.getFinallyBlock(), null);
664         return UTry.create(resources, block, catchesBuilder.build(), finallyBlock);
665       }

674       @Nullable 
675       private PlaceholderMethod placeholder(@Nullable ExpressionTree expr) {
676         Map<MethodSymbol, PlaceholderMethod> placeholderMethods = 
677             context.get(RefasterRuleBuilderScanner.PLACEHOLDER_METHODS_KEY);
678         return (placeholderMethods != null && expr != null) 
679             ? placeholderMethods.get(ASTHelpers.getSymbol(expr)) : null;
680       }

682       @Override
683       public UStatement visitExpressionStatement(ExpressionStatementTree tree, Void v) {
684         PlaceholderMethod placeholderMethod = placeholder(tree.getExpression());
685         if (placeholderMethod != null && placeholderMethod.returnType().equals(UPrimitiveType.VOID)) {
686           MethodInvocationTree invocation = (MethodInvocationTree) tree.getExpression();
687           return UPlaceholderStatement.create(placeholderMethod,
688               templateExpressions(invocation.getArguments()), ControlFlowVisitor.Result.NEVER_EXITS);
689         }
690         return UExpressionStatement.create(template(tree.getExpression()));
691       }

693       @Override
694       public UStatement visitReturn(ReturnTree tree, Void v) {
695         PlaceholderMethod placeholderMethod = placeholder(tree.getExpression());
696         if (placeholderMethod != null) {
697           MethodInvocationTree invocation = (MethodInvocationTree) tree.getExpression();
698           return UPlaceholderStatement.create(placeholderMethod, 
699               templateExpressions(invocation.getArguments()), 
700               ControlFlowVisitor.Result.ALWAYS_RETURNS);
701         }
702         return UReturn.create(
703             (tree.getExpression() == null) ? null : template(tree.getExpression()));
704       }

811       private final Type.Visitor<UType, Void> typeTemplater = new Types.SimpleVisitor<UType, Void>() {
812         private final Map<TypeSymbol, UTypeVar> typeVariables = new HashMap<>();
813     
814         @Override
815         public UType visitType(Type type, Void v) {
816           if (UPrimitiveType.isDeFactoPrimitive(type.getKind())) {
817             return UPrimitiveType.create(type.getKind());
818           } else {
819             throw new IllegalArgumentException(
820                 "Refaster does not currently support syntax " + type.getKind());
821           }
822         }
823     
824         @Override
825         public UArrayType visitArrayType(ArrayType type, Void v) {
826           return UArrayType.create(type.getComponentType().accept(this, null));
827         }
828     
829         @Override
830         public UMethodType visitMethodType(MethodType type, Void v) {
831           return UMethodType.create(
832               type.getReturnType().accept(this, null), templateTypes(type.getParameterTypes()));
833         }
834     
835         @Override
836         public UType visitClassType(ClassType type, Void v) {
837           if (type instanceof IntersectionClassType) {
838             return UIntersectionClassType.create(
839                 templateTypes(((IntersectionClassType) type).getComponents()));
840           }
841           return UClassType.create(
842               type.tsym.getQualifiedName().toString(), templateTypes(type.getTypeArguments()));
843         }
844     
845         @Override
846         public UWildcardType visitWildcardType(WildcardType type, Void v) {
847           return UWildcardType.create(type.kind, type.type.accept(this, null));
848         }
849     
850         @Override
851         public UTypeVar visitTypeVar(TypeVar type, Void v) {
852           /*
853            * In order to handle recursively bounded type variables without a stack overflow, we first
854            * cache a type var with no bounds, then we template the bounds.
855            */
856           TypeSymbol tsym = type.asElement();
857           if (typeVariables.containsKey(tsym)) {
858             return typeVariables.get(tsym);
859           }
860           UTypeVar var = UTypeVar.create(tsym.getSimpleName().toString());
861           typeVariables.put(tsym, var); // so the type variable can be used recursively in the bounds
862           var.setLowerBound(type.getLowerBound().accept(this, null));
863           var.setUpperBound(type.getUpperBound().accept(this, null));
864           return var;
865         }
866     
867         @Override
868         public UForAll visitForAll(ForAll type, Void v) {
869           List<UTypeVar> vars = cast(templateTypes(type.getTypeVariables()), UTypeVar.class);
870           return UForAll.create(vars, type.qtype.accept(this, null));
871         }
872       };

874       @SuppressWarnings("unchecked")
875       public static ImmutableClassToInstanceMap<Annotation> annotationMap(Symbol symbol) {
876         ImmutableClassToInstanceMap.Builder<Annotation> builder = ImmutableClassToInstanceMap.builder();
877         for (Compound compound : symbol.getAnnotationMirrors()) {
878           Name qualifiedAnnotationType =
879               ((TypeElement) compound.getAnnotationType().asElement()).getQualifiedName();
880           try {
881             Class<? extends Annotation> annotationClazz = 
882                 Class.forName(qualifiedAnnotationType.toString()).asSubclass(Annotation.class);
883             builder.put((Class) annotationClazz,
884                 AnnotationProxyMaker.generateAnnotation(compound, annotationClazz));
885           } catch (ClassNotFoundException e) {
886             throw new IllegalArgumentException("Unrecognized annotation type", e);
887           }
888         }
889         return builder.build();
890       }
