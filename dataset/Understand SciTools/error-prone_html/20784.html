68       @Test
69       public void testGetVariableUses() {
70         writeFile("A.java",
71             "public class A {",
72             "  public String b;",
73             "  void foo() {}",
74             "}");
75         writeFile("B.java",
76             "public class B {",
77             "  A my;",
78             "  B bar() { return null; }",
79             "  void foo(String x, A a) {",
80             "    x.trim().intern();",
81             "    a.b.trim().intern();",
82             "    this.my.foo();",
83             "    my.foo();",
84             "    this.bar();",
85             "    String.valueOf(0);",
86             "    java.lang.String.valueOf(1);",
87             "    bar().bar();",
88             "    System.out.println();",
89             "    a.b.indexOf(x.substring(1));",
90             "  }",
91             "}");
92     
93         TestScanner scanner = new TestScanner.Builder()
94             .add("x.trim().intern()", "x")
95             .add("a.b.trim().intern()", "a")
96             .add("this.my.foo()", "this")
97             .add("my.foo()", "my")
98             .add("this.bar()", "this")
99             .add("String.valueOf(0)")
100             .add("java.lang.String.valueOf(1)")
101             .add("bar().bar()")
102             .add("System.out.println()")
103             .add("a.b.indexOf(x.substring(1))", "a", "x")
104             .build();
105         assertCompiles(scanner);
106         scanner.assertFoundAll();
107       }

109       // TODO(mdempsky): Make this more reusable.
110       private static class TestScanner extends Scanner {
111         private static class Match {
112           private final ImmutableList<String> expected;
113           private boolean found = false;
114     
115           private Match(String... expected) {
116             this.expected = ImmutableList.copyOf(expected);
117           }
118         }
119     
120         private static class Builder {
121           private final ImmutableMap.Builder<String, Match> builder = ImmutableMap.builder();
122     
123           public Builder add(String expression, String... expected) {
124             builder.put(expression, new Match(expected));
125             return this;
126           }
127     
128           public TestScanner build() {
129             return new TestScanner(builder.build());
130           }
131         }
132     
133         private final ImmutableMap<String, Match> matches;
134     
135         private TestScanner(ImmutableMap<String, Match> matches) {
136           this.matches = matches;
137         }
138     
139         @Override
140         public Void visitExpressionStatement(ExpressionStatementTree node, VisitorState state) {
141           ExpressionTree expression = node.getExpression();
142           Match match = matches.get(expression.toString());
143           if (match != null) {
144             assertMatch(expression, match.expected);
145             match.found = true;
146           }
147           return super.visitExpressionStatement(node, state);
148         }
149     
150         private void assertMatch(ExpressionTree node, List<String> expected) {
151           List<IdentifierTree> uses = PreconditionsCheckNotNullPrimitive.getVariableUses(node);
152           assertEquals("variables used in " + node, expected,
153               Lists.transform(uses, new Function<IdentifierTree, String>() {
154                 @Override
155                 public String apply(IdentifierTree ident) {
156                   return ident.toString();
157                 }
158               }));
159         }
160     
161         public void assertFoundAll() {
162           for (Map.Entry<String, Match> entry : matches.entrySet()) {
163             assertTrue("found " + entry.getKey(), entry.getValue().found);
164           }
165         }
166       }
