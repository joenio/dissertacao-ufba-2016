17     package com.google.errorprone.bugpatterns;
18     
19     import static com.google.errorprone.BugPattern.Category.JDK;
20     import static com.google.errorprone.BugPattern.MaturityLevel.MATURE;
21     import static com.google.errorprone.BugPattern.SeverityLevel.ERROR;
22     import static com.google.errorprone.matchers.method.MethodMatchers.staticMethod;
23     
24     import com.google.errorprone.BugPattern;
25     import com.google.errorprone.VisitorState;
26     import com.google.errorprone.bugpatterns.BugChecker.MethodInvocationTreeMatcher;
27     import com.google.errorprone.matchers.Description;
28     import com.google.errorprone.matchers.Matcher;
29     import com.google.errorprone.util.ASTHelpers;
30     
31     import com.sun.source.tree.ExpressionTree;
32     import com.sun.source.tree.MethodInvocationTree;
33     import com.sun.tools.javac.tree.JCTree;
34     
35     /**
36      * @author avenet@google.com (Arnaud J. Venet)
37      */
38     @BugPattern(name = "InsecureCipherMode",
39         summary = "Cipher.getInstance() is invoked using either the default settings or ECB mode",
40         explanation = "This error is triggered when a Cipher instance is created using either"
41             + " the default settings or the notoriously insecure ECB mode. In particular, Java's"
42             + " default `Cipher.getInstance(\"AES\")` returns a cipher object that operates in ECB"
43             + " mode. Dynamically constructed transformation strings are also flagged, as they may"
44             + " conceal an instance of ECB mode. The problem with ECB mode is that encrypting the same"
45             + " block of plaintext always yields the same block of ciphertext. Hence, repetitions in"
46             + " the plaintext translate into repetitions in the ciphertext, which can be readily used"
47             + " to conduct cryptanalysis.\n\n"
48         ,
49         category = JDK, severity = ERROR, documentSuppression = false,
50         maturity = MATURE)
51     public class InsecureCipherMode extends BugChecker implements MethodInvocationTreeMatcher {
52       private static final String MESSAGE_BASE = "Insecure usage of Cipher.getInstance(): ";
53     
54       private static final Matcher<ExpressionTree> GETINSTANCE_MATCHER =
55           staticMethod().onClass("javax.crypto.Cipher").named("getInstance");
56     
57     
58       private Description buildErrorMessage(MethodInvocationTree tree, String explanation) {
59         Description.Builder description = buildDescription(tree);
60         description.setMessage(MESSAGE_BASE + explanation + ".");
61         return description.build();
62       }
63     
64       @Override
65       public Description matchMethodInvocation(MethodInvocationTree tree, VisitorState state) {
66         if (!GETINSTANCE_MATCHER.matches(tree, state)) {
67           return Description.NO_MATCH;
68         }
69     
70     
71         // We analyze the first argument of all the overloads of Cipher.getInstance().
72         Object argument = ASTHelpers.constValue((JCTree) tree.getArguments().get(0));
73         if (argument == null) {
74           // We flag call sites where the transformation string is dynamically computed.
75           return buildErrorMessage(
76               tree, "the transformation is not a compile-time constant expression");
77         }
78         // Otherwise, we know that the transformation is specified by a string literal.
79         String transformation = (String) argument;
80     
81         // We exclude stream ciphers (this check only makes sense for block ciphers), i.e., the RC4
82         // cipher. The name of this algorithm is "ARCFOUR" in the SunJce and "ARC4" in Conscrypt.
83         // Some other providers like JCraft also seem to use the name "RC4".
84         if (transformation.matches("ARCFOUR.*") || transformation.matches("ARC4.*")
85             || transformation.matches("RC4.*")) {
86           return Description.NO_MATCH;
87         }
88     
89         if (!transformation.matches(".*/.*/.*")) {
90           // The mode and padding shall be explicitly specified. We don't allow default settings to be
91           // used, regardless of the algorithm and provider.
92           return buildErrorMessage(tree, "the mode and padding must be explicitly specified");
93         }
94     
95         if (transformation.matches(".*/ECB/.*") && !transformation.matches("RSA/.*")
96             && !transformation.matches("AESWrap/.*")) {
97           // Otherwise, ECB mode should be explicitly specified in order to trigger the check. RSA
98           // is an exception, as this transformation doesn't actually implement a block cipher
99           // encryption mode (the input is limited by the size of the key). AESWrap is another
100           // exception, because this algorithm is only used to encrypt encryption keys.
101           return buildErrorMessage(tree, "ECB mode must not be used");
102         }
103     
104         return Description.NO_MATCH;
105       }
106     }
