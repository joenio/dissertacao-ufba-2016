19     import static com.google.errorprone.BugPattern.Category.JDK;
20     import static com.google.errorprone.BugPattern.MaturityLevel.EXPERIMENTAL;
21     import static com.google.errorprone.BugPattern.SeverityLevel.WARNING;
22     import static com.google.errorprone.matchers.Matchers.allOf;
23     import static com.google.errorprone.matchers.Matchers.isSubtypeOf;
24     
25     import com.google.common.collect.Lists;
26     import com.google.errorprone.BugPattern;
27     import com.google.errorprone.VisitorState;
28     import com.google.errorprone.bugpatterns.BugChecker.ClassTreeMatcher;
29     import com.google.errorprone.matchers.Description;
30     import com.google.errorprone.matchers.Matcher;
31     
32     import com.sun.source.tree.ClassTree;
33     import com.sun.source.tree.Tree;
34     
35     import java.util.Iterator;
36     import java.util.List;
37     
38     /**
39      * @author jsjeon@google.com (Jinseong Jeon)
40      */
41     @BugPattern(
42       name = "IterableAndIterator",
43       summary = "Class should not implement both `Iterable` and `Iterator`",
44       explanation =
45           "An `Iterator` is a *state-ful* instance that enables you to check "
46               + "whether it has more elements (via `hasNext()`) "
47               + "and moves to the next one if any (via `next()`), "
48               + "while an `Iterable` is a representation of literally iterable elements. "
49               + "An `Iterable` can generate multiple valid `Iterator`s, though.",
50       category = JDK,
51       severity = WARNING,
52       maturity = EXPERIMENTAL
53     )
54     public class IterableAndIterator extends BugChecker implements ClassTreeMatcher {
55     
56       private static final String ITERABLE = Iterable.class.getCanonicalName();
57       private static final String ITERATOR = Iterator.class.getCanonicalName();
58     
59       /**
60        * Matches if a class/interface is subtype of Iterable
61        */
62       private static final Matcher<Tree> ITERABLE_MATCHER = isSubtypeOf(ITERABLE);
63     
64       /**
65        * Matches if a class/interface is subtype of Iterator
66        */
67       private static final Matcher<Tree> ITERATOR_MATCHER = isSubtypeOf(ITERATOR);
68     
69       /**
70        * Matches if a class/interface is subtype of Iterable _and_ Iterator
71        */
72       private static final Matcher<Tree> ITERABLE_AND_ITERATOR_MATCHER =
73           allOf(ITERABLE_MATCHER, ITERATOR_MATCHER);
74     
75       private boolean matchAnySuperType(ClassTree tree, VisitorState state) {
76         List<Tree> superTypes = Lists.<Tree>newArrayList(tree.getImplementsClause());
77         Tree superClass = tree.getExtendsClause();
78         if (superClass != null) {
79           superTypes.add(superClass);
80         }
81     
82         /* NOTE: at "Eight day", use Java 8 feature below
83         return superTypes.stream()
84             .anyMatch(superType -> ITERABLE_AND_ITERATOR_MATCHER.matches(superType, state));
85          */
86         for (Tree superType : superTypes) {
87           if (ITERABLE_AND_ITERATOR_MATCHER.matches(superType, state)) {
88             return true;
89           }
90         }
91         return false;
92       }
93     
94       @Override
95       public Description matchClass(ClassTree tree, VisitorState state) {
96         if (ITERABLE_AND_ITERATOR_MATCHER.matches(tree, state)) {
97           // Filter out inherited case to not warn again
98           if (matchAnySuperType(tree, state)) {
99             return Description.NO_MATCH;
100           }
101     
102           // TODO: Distinguish direct implementation and indirect cases
103     
104           // TODO: Suggest removing Iterable or Iterator, along with implemented methods
105           return describeMatch(tree);
106         }
107         return Description.NO_MATCH;
108       }
109     }
