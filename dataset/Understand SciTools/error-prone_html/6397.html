17     package com.google.errorprone.bugpatterns;
18     import static com.google.errorprone.BugPattern.Category.JDK;
19     import static com.google.errorprone.BugPattern.MaturityLevel.MATURE;
20     import static com.google.errorprone.BugPattern.SeverityLevel.ERROR;
21     import static com.google.errorprone.matchers.Matchers.allOf;
22     import static com.google.errorprone.matchers.Matchers.anyOf;
23     import static com.google.errorprone.matchers.Matchers.argument;
24     import static com.google.errorprone.matchers.method.MethodMatchers.instanceMethod;
25     import static com.google.errorprone.matchers.method.MethodMatchers.staticMethod;
26     
27     import com.google.errorprone.BugPattern;
28     import com.google.errorprone.VisitorState;
29     import com.google.errorprone.bugpatterns.BugChecker.MethodInvocationTreeMatcher;
30     import com.google.errorprone.fixes.SuggestedFix;
31     import com.google.errorprone.matchers.Description;
32     import com.google.errorprone.matchers.Matcher;
33     
34     import com.sun.source.tree.ExpressionTree;
35     import com.sun.source.tree.MethodInvocationTree;
36     import com.sun.tools.javac.tree.JCTree.JCExpression;
37     
38     import java.util.regex.Pattern;
39     import java.util.regex.PatternSyntaxException;
40     
41     /**
42      * @author mdempsky@google.com (Matthew Dempsky)
43      */
44     @BugPattern(name = "InvalidPatternSyntax",
45         summary = "Invalid syntax used for a regular expression",
46         explanation = "This error is triggered by calls to regex-accepting methods with invalid string "
47             + "literals.  These calls would cause a PatternSyntaxException at runtime.\n\n"
48             + "We deliberately do not check java.util.regex.Pattern#compile as many of its users "
49             + "are deliberately testing the regex compiler or using a vacuously true regex.",
50         category = JDK, severity = ERROR, maturity = MATURE)
51     public class InvalidPatternSyntax extends BugChecker implements MethodInvocationTreeMatcher {
52     
53       private static final String MESSAGE_BASE = "Invalid syntax used for a regular expression: ";
54     
55       /* Match string literals that are not valid syntax for regular expressions. */
56       private static final Matcher<ExpressionTree> BAD_REGEX_LITERAL = new Matcher<ExpressionTree>() {
57         @Override
58         public boolean matches(ExpressionTree tree, VisitorState state) {
59           Object value = ((JCExpression) tree).type.constValue();
60           return value instanceof String && !isValidSyntax((String) value);
61         }
62     
63         private boolean isValidSyntax(String regex) {
64           // Actually valid, but useless.
65           if (".".equals(regex)) {
66             return false;
67           }
68           try {
69             Pattern.compile(regex);
70             return true;
71           } catch (PatternSyntaxException e) {
72             return false;
73           }
74         }
75       };
76     
77       /*
78        * Match invocations to regex-accepting methods with bad string literals.
79        *
80        * <p>We deliberately omit Pattern.compile itself, as most of its users appear to be either
81        * passing e.g. LITERAL flags, deliberately testing the regex compiler, or deliberately
82        * using "." as the "vacuously true regex."
83        */
84       private static final Matcher<MethodInvocationTree> BAD_REGEX_USAGE =
85           allOf(
86               anyOf(
87                   instanceMethod().onDescendantOf("java.lang.String")
88                       .withSignature("matches(java.lang.String)"),
89                   instanceMethod().onDescendantOf("java.lang.String")
90                       .withSignature("replaceAll(java.lang.String,java.lang.String)"),
91                   instanceMethod().onDescendantOf("java.lang.String")
92                       .withSignature("replaceFirst(java.lang.String,java.lang.String)"),
93                   instanceMethod().onDescendantOf("java.lang.String")
94                       .withSignature("split(java.lang.String)"),
95                   instanceMethod().onDescendantOf("java.lang.String")
96                       .withSignature("split(java.lang.String,int)"),
97                   staticMethod().onClass("java.util.regex.Pattern").named("matches"),
98                   staticMethod().onClass("com.google.common.base.Splitter").named("onPattern")),
99               argument(0, BAD_REGEX_LITERAL));
100     
101       @Override
102       public Description matchMethodInvocation(
103           MethodInvocationTree methodInvocationTree, VisitorState state) {
104         if (!BAD_REGEX_USAGE.matches(methodInvocationTree, state)) {
105           return Description.NO_MATCH;
106         }
107     
108         // TODO: Suggest fixes for more situations.
109         Description.Builder descriptionBuilder = buildDescription(methodInvocationTree);
110         ExpressionTree arg = methodInvocationTree.getArguments().get(0);
111         String value = (String) ((JCExpression) arg).type.constValue();
112         String reasonInvalid = "";
113     
114         if (".".equals(value)) {
115           descriptionBuilder.addFix(SuggestedFix.replace(arg, "\"\\\\.\""));
116           reasonInvalid = "\".\" is a valid but useless regex";
117         } else {
118           try {
119             Pattern.compile(value);
120           } catch (PatternSyntaxException e) {
121             reasonInvalid = e.getMessage();
122           }
123         }
124     
125         descriptionBuilder.setMessage(MESSAGE_BASE + reasonInvalid);
126         return descriptionBuilder.build();
127       }
128     }
