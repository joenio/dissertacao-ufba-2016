17     package com.google.errorprone;
18     
19     import static java.lang.annotation.RetentionPolicy.RUNTIME;
20     
21     import java.lang.annotation.Annotation;
22     import java.lang.annotation.Retention;
23     
24     /**
25      * Describes a bug pattern detected by error-prone.  Used to generate compiler error messages,
26      * for @SuppressWarnings, and to generate the documentation that we host on our web site.
27      *
28      * @author eaftan@google.com (Eddie Aftandilian)
29      */
30     @Retention(RUNTIME)
31     public @interface BugPattern {
32       /**
33        * A unique identifier for this bug, used for @SuppressWarnings and in the compiler error
34        * message.
35        */
36       String name();
37     
38       /**
39        * Alternate identifiers for this bug, which may also be used in @SuppressWarnings.
40        */
41       String[] altNames() default {};
42     
43       /**
44        * The type of link to generate in the compiler error message.
45        */
46       LinkType linkType() default LinkType.AUTOGENERATED;
47     
48       /**
49        * The link URL to use if linkType() is LinkType.CUSTOM.
50        */
51       String link() default "";
52     
53       public enum LinkType {
54         /**
55          * Link to autogenerated documentation, hosted on the error-prone web site.
56          */
57         AUTOGENERATED,
58         /**
59          * Custom string.
60          */
61         CUSTOM,
62         /**
63          * No link should be displayed.
64          */
65         NONE
66       }
67     
68       /**
69        * The class of bug this bug checker detects.
70        */
71       Category category();
72     
73       public enum Category {
74         /**
75          * General Java or JDK errors.
76          */
77         JDK,
78         /**
79          * Errors specific to Google Guava.
80          */
81         GUAVA,
82         /**
83          * Errors specific to Google Guice.
84          */
85         GUICE,
86         /**
87          * Errors specific to Dagger.
88          */
89         DAGGER,
90         /**
91          * Errors specific to JUnit.
92          */
93         JUNIT,
94         /**
95          * One-off matchers that are not general errors.
96          */
97         ONE_OFF,
98         /**
99          *  JSR-330 errors not specific to Guice.
100          */
101         INJECT,
102         /**
103          * Errors specific to Mockito.
104          */
105         MOCKITO,
106         /**
107          * Errors specific to JMock.
108          */
109         JMOCK,
110         /**
111          * Errors specific to Android.
112          */
113         ANDROID;
114       }
115     
116       /**
117        * A short summary of the problem that this checker detects.  Used for the default compiler error
118        * message and for the short description in the generated docs.  Should not end with a period,
119        * to match javac warning/error style.
120        *
121        * <p>Markdown syntax is not allowed for this element.
122        */
123       String summary();
124     
125       /**
126        * A longer explanation of the problem that this checker detects.  Used as the main content
127        * in the generated documentation for this checker.
128        *
129        * <p>Markdown syntax is allowed for this element.
130        */
131       String explanation() default "";
132     
133       SeverityLevel severity();
134     
135       public enum SeverityLevel {
136         ERROR(true),
137         WARNING(true),
138         SUGGESTION(true),
139         /**
140          * Should not be used for general code.
141          */
142         NOT_A_PROBLEM(false);
143     
144         private final boolean enabled;
145     
146         SeverityLevel(boolean enabled) {
147           this.enabled = enabled;
148         }
149     
150         public boolean enabled() {
151           return enabled;
152         }
153       }
154     
155       MaturityLevel maturity();
156     
157       public enum MaturityLevel {
158         MATURE("On by default"),
159         EXPERIMENTAL("Experimental");
160     
161         final String description;
162         MaturityLevel(String description) {
163           this.description = description;
164         }
165       }
166     
167       /**
168        * Whether this checker should be suppressible, and if so, by what means.
169        */
170       Suppressibility suppressibility() default Suppressibility.SUPPRESS_WARNINGS;
171     
172       public enum Suppressibility {
173         /**
174          * Can be suppressed using the standard {@code SuppressWarnings("foo")} mechanism. This
175          * setting should be used unless there is a good reason otherwise, e.g. security.
176          */
177         SUPPRESS_WARNINGS(true),
178         /**
179          * Can be suppressed with a custom annotation on a parent AST node.
180          */
181         CUSTOM_ANNOTATION(false),
182         /**
183          * Cannot be suppressed.
184          */
185         UNSUPPRESSIBLE(false);
186     
187         private final boolean disableable;
188     
189         Suppressibility(boolean disableable) {
190           this.disableable = disableable;
191         }
192     
193         public boolean disableable() {
194           return disableable;
195         }
196       }
197     
198       /**
199        * A set of custom suppression annotation types to use if suppressibility is
200        * Suppressibility.CUSTOM_ANNOTATION.
201        */
202       Class<? extends Annotation>[] customSuppressionAnnotations() default {};
203     
204       /**
205        * Generate an explanation of how to suppress the check.
206        *
207        * <p>This should only be disabled if the check has a non-standard suppression
208        * mechanism that requires additional explanation. For example,
209        * {@link SuppressWarnings} cannot be applied to packages, so checks that operate
210        * at the package level need special treatment.
211        */
212       public boolean documentSuppression() default true;
213     
214       /**
215        * Generate examples from test cases.
216        *
217        * <p>By default, any positive or negative test inputs  are included in the generated
218        * documentation as examples. That behaviour can be disabled if the test inputs aren't good
219        * documentation (for example, because they're testing implementation details of the check and
220        * aren't representative of real code).
221        *
222        * <p>If this feature is disabled, make sure to include some representative examples in the
223        * explanation.
224        */
225       public boolean generateExamplesFromTestCases() default true;
226     
227     }
