17     package com.google.errorprone.matchers;
18     
19     import com.google.auto.value.AutoValue;
20     import com.google.common.base.Optional;
21     import com.google.common.collect.ImmutableList;
22     import com.google.common.collect.Iterables;
23     import com.google.errorprone.VisitorState;
24     
25     import com.sun.source.tree.Tree;
26     import com.sun.source.util.TreePath;
27     
28     import java.util.List;
29     
30     /**
31      * A {@link MultiMatcher} that applies a matcher across multiple children of a single ancestor
32      * node.  Configurable to return true if any of, all of, or the last node matches. In the any or
33      * last of cases, provides access to the node that matched.
34      *
35      * @author eaftan@google.com (Eddie Aftandilian)
36      * @param <T> the type of the node to match on
37      * @param <N> the type of the subnode that the given matcher should match
38      */
39     public abstract class ChildMultiMatcher<T extends Tree, N extends Tree>
40         implements MultiMatcher<T, N> {
41     
42       public enum MatchType {
43         ALL,
44         AT_LEAST_ONE,
45         LAST
46       }
47       
48       @AutoValue
49       abstract static class Matchable<T extends Tree> {
50         public abstract T tree();
51         public abstract VisitorState state();
52         
53         public static <T extends Tree> Matchable<T> create(T tree, VisitorState state) {
54           return new AutoValue_ChildMultiMatcher_Matchable<>(tree, state);
55         }
56       }
57       
58       @AutoValue
59       abstract static class MatchResult<T extends Tree> {
60         public abstract Optional<T> matchingNode();
61         public abstract boolean matches();
62         
63         public static <T extends Tree> MatchResult<T> none() {
64           return create(Optional.<T>absent(), false);
65         }
66         
67         public static <T extends Tree> MatchResult<T> match() {
68           return create(Optional.<T>absent(), true);
69         }
70         
71         public static <T extends Tree> MatchResult<T> match(T matchingNode) {
72           return create(Optional.of(matchingNode), true);
73         }
74         
75         private static <T extends Tree> MatchResult<T> create(
76             Optional<T> matchingNode, boolean matches) {
77           return new AutoValue_ChildMultiMatcher_MatchResult<>(matchingNode, matches);
78         }
79       }
80       
81       /**
82        * A matcher that operates over a list of nodes, each of which includes an AST node and a
83        * VisitorState with a TreePath for the given node.
84        */
85       private abstract static class ListMatcher<N extends Tree> {
86         abstract MatchResult<N> matches(List<Matchable<N>> matchables, Matcher<N> nodeMatcher);
87     
88         public static <N extends Tree> ListMatcher<N> create(MatchType matchType) {
89           switch (matchType) {
90             case ALL:
91               return new AllMatcher<>();
92             case AT_LEAST_ONE:
93               return new AtLeastOneMatcher<>();
94             case LAST:
95               return new LastMatcher<>();
96           }
97           throw new AssertionError("Unexpected match type: " + matchType);
98         }
99       }
100       
101       /**
102        * A matcher that returns true if all nodes in the list match.
103        */
104       private static class AllMatcher<N extends Tree> extends ListMatcher<N> {
105         @Override
106         public MatchResult<N> matches(List<Matchable<N>> matchables, Matcher<N> nodeMatcher) {
107           for (Matchable<N> matchable : matchables) {
108             if (!nodeMatcher.matches(matchable.tree(), matchable.state())) {
109               return MatchResult.none();
110             }
111           }
112           return MatchResult.match();
113         }
114       }
115       
116       /**
117        * A matcher that returns true if at least one node in the list matches.
118        */
119       private static class AtLeastOneMatcher<N extends Tree> extends ListMatcher<N> {
120         @Override
121         public MatchResult<N> matches(List<Matchable<N>> matchables, Matcher<N> nodeMatcher) {
122           for (Matchable<N> matchable : matchables) {
123             if (nodeMatcher.matches(matchable.tree(), matchable.state())) {
124               return MatchResult.match(matchable.tree());
125             }
126           }
127           return MatchResult.none();
128         }
129       }
130       
131       /**
132        * A matcher that returns true if the last node in the list matches.
133        */
134       private static class LastMatcher<N extends Tree> extends ListMatcher<N> {
135         @Override
136         public MatchResult<N> matches(List<Matchable<N>> matchables, Matcher<N> nodeMatcher) {
137           if (matchables.isEmpty()) {
138             return MatchResult.none();
139           }
140           Matchable<N> last = Iterables.getLast(matchables);
141           return nodeMatcher.matches(last.tree(), last.state())
142               ? MatchResult.match(last.tree())
143               : MatchResult.<N>none();
144         }
145       }
146         
147       /**
148        * The matcher to apply to the subnodes in question.
149        */
150       protected final Matcher<N> nodeMatcher;
151     
152       private final ListMatcher<N> listMatcher;
153       private MatchResult<N> matchResult;
154     
155       public ChildMultiMatcher(MatchType matchType, Matcher<N> nodeMatcher) {
156         this.nodeMatcher = nodeMatcher;
157         this.listMatcher = ListMatcher.create(matchType);
158       }
159     
160       @Override
161       public boolean matches(T tree, VisitorState state) {
162         ImmutableList.Builder<Matchable<N>> result = ImmutableList.builder();
163         for (N subnode : getChildNodes(tree, state)) {
164           TreePath newPath = new TreePath(state.getPath(), subnode);
165           result.add(Matchable.create(subnode, state.withPath(newPath)));
166         }
167         matchResult = listMatcher.matches(result.build(), nodeMatcher);
168         return matchResult.matches();
169       }
170     
171       @Override
172       public N getMatchingNode() {
173         return matchResult.matchingNode().get();
174       }
175       
176       /**
177        * Returns the set of child nodes to match. The nodes must be immediate children of the current
178        * node to ensure the TreePath calculation is correct. MultiMatchers with other requirements
179        * should not subclass ChildMultiMatcher (see {@link HasIdentifier} for an example).
180        */
181       protected abstract Iterable<? extends N> getChildNodes(T tree, VisitorState state);
182     }
