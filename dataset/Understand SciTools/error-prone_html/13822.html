17     package com.google.errorprone.bugpatterns;

19     import static com.google.errorprone.BugPattern.Category.INJECT;
20     import static com.google.errorprone.BugPattern.MaturityLevel.EXPERIMENTAL;
21     import static com.google.errorprone.BugPattern.SeverityLevel.ERROR;
22     import static com.google.errorprone.matchers.Matchers.allOf;
23     import static com.google.errorprone.matchers.Matchers.booleanLiteral;
24     import static com.google.errorprone.matchers.Matchers.hasAnnotation;
25     import static com.google.errorprone.matchers.Matchers.hasArgumentWithValue;
26     import static com.google.errorprone.matchers.Matchers.isType;
27     import static com.google.errorprone.matchers.Matchers.methodIsConstructor;

29     import com.google.errorprone.BugPattern;
30     import com.google.errorprone.VisitorState;
31     import com.google.errorprone.bugpatterns.BugChecker.MethodTreeMatcher;
32     import com.google.errorprone.fixes.SuggestedFix;
33     import com.google.errorprone.matchers.Description;
34     import com.google.errorprone.matchers.Matcher;
35     
36     import com.sun.source.tree.AnnotationTree;
37     import com.sun.source.tree.MethodTree;
38     
39     /**
40      * A checker for injected constructors with @Inject(optional=true) or binding annotations.
41      */
42     @BugPattern(name = "InjectedConstructorAnnotations",
43         summary = "Injected constructors cannot be optional nor have binding annotations",
44         explanation = "The constructor is annotated with @Inject(optional=true), or it is annotated "
45             + "with @Inject and a binding annotation. This will cause a Guice runtime error.\n\n"
46             + "See [https://code.google.com/p/google-guice/wiki/InjectionPoints] for details.",
47         category = INJECT, severity = ERROR, maturity = EXPERIMENTAL)
48     public class InjectedConstructorAnnotations extends BugChecker implements MethodTreeMatcher {
49       private static final String GUICE_INJECT_ANNOTATION = "com.google.inject.Inject";
50       private static final String GUICE_BINDING_ANNOTATION = "com.google.inject.BindingAnnotation";
51     
52       // A matcher of @Inject{optional=true}
53       private static final Matcher<AnnotationTree> OPTIONAL_INJECTION_MATCHER =
54           allOf(
55             isType(GUICE_INJECT_ANNOTATION),
56             hasArgumentWithValue("optional", booleanLiteral(true)));
57     
58       // A matcher of binding annotations
59       private static final Matcher<AnnotationTree> BINDING_ANNOTATION_MATCHER =
60         new Matcher<AnnotationTree>() {
61           @Override
62           public boolean matches(AnnotationTree annotationTree, VisitorState state) {
63             return hasAnnotation(GUICE_BINDING_ANNOTATION)
64                 .matches(annotationTree.getAnnotationType(), state);
65           }
66       };
67     
68       /**
69        * Matches injected constructors annotated with @Inject(optional=true) or binding annotations.
70        * Suggests fixes to remove the argument @code{optional=true} or binding annotations.
71        */
72       @Override
73       public Description matchMethod(MethodTree methodTree, VisitorState state) {
74         SuggestedFix.Builder fix = null;
75         if (isInjectedConstructor(methodTree, state)) {
76           for (AnnotationTree annotationTree : methodTree.getModifiers().getAnnotations()) {
77             if (OPTIONAL_INJECTION_MATCHER.matches(annotationTree, state)) {
78               // Replace the annotation with "@Inject"
79               if (fix == null) {
80                 fix = SuggestedFix.builder();
81               }
82               fix = fix.replace(annotationTree, "@Inject");
83             } else if (BINDING_ANNOTATION_MATCHER.matches(annotationTree, state)) {
84               // Remove the binding annotation
85               if (fix == null) {
86                 fix = SuggestedFix.builder();
87               }
88               fix = fix.delete(annotationTree);
89             }
90           }
91         }
92         if (fix == null) {
93           return Description.NO_MATCH;
94         } else {
95           return describeMatch(methodTree, fix.build());
96         }
97       }
98     
99       private boolean isInjectedConstructor(MethodTree methodTree, VisitorState state) {
100         return allOf(methodIsConstructor(), hasAnnotation(GUICE_INJECT_ANNOTATION))
101             .matches(methodTree, state);
102       }
103     }
