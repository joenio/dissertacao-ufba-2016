234       /**
235        * Annotation processor that checks that the context always has the same {@link ErrorProneScanner}
236        * instance at each stage of annotation processing.
237        */
238       @SupportedAnnotationTypes("*")
239       public static final class ScannerCheckingProcessor extends AbstractProcessor {
240     
241         private final ErrorProneScanner expected;
242     
243         public ScannerCheckingProcessor(ErrorProneScanner expected) {
244           this.expected = expected;
245         }
246     
247         @Override
248         public SourceVersion getSupportedSourceVersion() {
249           return SourceVersion.latest();
250         }
251     
252         @Override
253         public boolean process(Set<? extends TypeElement> annotations, RoundEnvironment roundEnv) {
254           Context context = ((JavacProcessingEnvironment) processingEnv).getContext();
255           assertSame(expected, context.get(Scanner.class));
256           return false;
257         }
258       }

285       @BugPattern(name = "SuperCallMatcher", explanation = "",
286           category = ONE_OFF, maturity = EXPERIMENTAL, severity = ERROR, summary = "")
287       static class SuperCallMatcher extends BugChecker implements MethodInvocationTreeMatcher {
288         @Override
289         public Description matchMethodInvocation(MethodInvocationTree tree, VisitorState state) {
290           Tree select = tree.getMethodSelect();
291           Name name;
292           if (select instanceof MemberSelectTree) {
293             name = ((MemberSelectTree) select).getIdentifier();
294           } else if (select instanceof IdentifierTree) {
295             name = ((IdentifierTree) select).getName();
296           } else {
297             return Description.NO_MATCH;
298           }
299           return name.contentEquals("super")
300               ? describeMatch(tree)
301               : Description.NO_MATCH;
302         }
303       }
