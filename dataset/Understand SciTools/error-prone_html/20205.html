39     
40     import java.util.List;

43     import javax.lang.model.element.Modifier;
44     import javax.lang.model.element.TypeElement;

46     /**
47      * @author sgoldfeder@google.com (Steven Goldfeder)
48      */
49     @BugPattern(name = "IncompatibleModifiers",
50         summary = "This annotation has incompatible modifiers as specified by its "
51         + "@IncompatibleModifiers annotation",
52         explanation = "The @IncompatibleModifiers annotation declares that the target annotation "
53         + "is incompatible with a set of provided modifiers. This check ensures that all "
54         + "annotations respect their @IncompatibleModifiers specifications.",
55         linkType = NONE, category = JDK, severity = WARNING, maturity = MATURE)
56     // TODO(cushon): merge the implementation with RequiredModifiersChecker
57     public class IncompatibleModifiersChecker extends BugChecker implements AnnotationTreeMatcher {
58     
59       private static final String MESSAGE_TEMPLATE = "%s has specified that it should not be used"
60           + " together with the following modifiers: %s";
61     
62     
63       // TODO(cushon): deprecate and remove
64       private static final String GUAVA_ANNOTATION =
65           "com.google.common.annotations.IncompatibleModifiers";
66     
67       private static final Function<Attribute.Enum, Modifier> TO_MODIFIER =
68           new Function<Attribute.Enum, Modifier>() {
69             @Override
70             public Modifier apply(Attribute.Enum input) {
71               return Modifier.valueOf(input.getValue().name.toString());
72             }
73           };
74     
75       private static Set<Modifier> getIncompatibleModifiers(AnnotationTree tree, VisitorState state) {
76         for (Attribute.Compound c : ASTHelpers.getSymbol(tree).getAnnotationMirrors()) {
77           if (((TypeElement) c.getAnnotationType().asElement()).getQualifiedName()
78               .contentEquals(GUAVA_ANNOTATION)) {
79             @SuppressWarnings("unchecked")
80             List<Attribute.Enum> modifiers =
81                 (List<Attribute.Enum>) c.member(state.getName("value")).getValue();
82             return ImmutableSet.copyOf(Iterables.transform(modifiers, TO_MODIFIER));
83           }
84         }
85     
86         IncompatibleModifiers annotation = ASTHelpers.getAnnotation(tree, IncompatibleModifiers.class);
87         if (annotation != null) {
88           return ImmutableSet.copyOf(annotation.value());
89         }
90     
91         return ImmutableSet.of();
92       }
93     
94       @Override
95       public Description matchAnnotation(AnnotationTree tree, VisitorState state) {
96         Set<Modifier> incompatibleModifiers = getIncompatibleModifiers(tree, state);
97         if (incompatibleModifiers.isEmpty()) {
98           return Description.NO_MATCH;
99         }
100     
101         Tree parent = state.getPath().getParentPath().getLeaf();
102         if (!(parent instanceof ModifiersTree)) {
103           // e.g. An annotated package name
104           return Description.NO_MATCH;
105         }
106     
107         Set<Modifier> incompatible = Sets.intersection(
108             incompatibleModifiers,
109             ((ModifiersTree) parent).getFlags());
110     
111         if (incompatible.isEmpty()) {
112           return Description.NO_MATCH;
113         }
114     
115         String annotationName = ASTHelpers.getAnnotationName(tree);
116         String nameString = annotationName != null
117             ? String.format("The annotation '@%s'", annotationName)
118             : "This annotation";
119         String customMessage = String.format(MESSAGE_TEMPLATE, nameString, incompatible.toString());
120         return buildDescription(tree)
121             .setMessage(customMessage)
122             .build();
123       }
124     }
