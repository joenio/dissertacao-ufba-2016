44     /**
45      * Detects invocations of methods with a parameter annotated {@code @CompileTimeConstant} such that
46      * the corresponding actual parameter is not a compile-time constant expression.
47      *
48      * <p>
49      * This type annotation checker enforces that for all method and constructor invocations, for all
50      * formal parameters of the invoked method/constructor that are annotated with the
51      * {@link com.google.errorprone.annotations.CompileTimeConstant} type annotation, the
52      * corresponding actual parameter is an expression that satisfies one of the following conditions:
53      * <ol>
54      * <li>The expression is one for which the Java compiler can determine a constant value at compile
55      * time, or</li>
56      * <li>the expression consists of the literal {@code null}, or</li>
57      * <li>the expression consists of a single identifier, where the identifier is a formal method
58      * parameter that is declared {@code final} and has the
59      * {@link com.google.errorprone.annotations.CompileTimeConstant} annotation.</li>
60      * </ol>
61      *
62      * @see CompileTimeConstantExpressionMatcher
63      */
64     @BugPattern(name = "CompileTimeConstant",
65         summary =
66             "Non-compile-time constant expression passed to parameter with "
67             + "@CompileTimeConstant type annotation.",
68         explanation =
69             "A method or constructor with one or more parameters whose declaration is "
70             + "annotated with the @CompileTimeConstant type annotation must only be invoked "
71             + "with corresponding actual parameters that are computed as compile-time constant "
72             + "expressions, such as a literal or static final constant.",
73         linkType = NONE, category = GUAVA, severity = ERROR, maturity = MATURE)
74     public class CompileTimeConstantChecker extends BugChecker
75         implements MethodInvocationTreeMatcher, NewClassTreeMatcher {
76     
77       private static final String DID_YOU_MEAN_FINAL_FMT_MESSAGE =
78           " Did you mean to make '%s' final?";
79     
80       private final Matcher<ExpressionTree> compileTimeConstExpressionMatcher =
81           new CompileTimeConstantExpressionMatcher();
82     
83       /**
84        * Matches formal parameters with
85        * {@link com.google.errorprone.annotations.CompileTimeConstant} annotations against
86        * corresponding actual parameters.
87        *
88        * @param state        the visitor state
89        * @param calleeSymbol the method whose formal parameters to consider
90        * @param actualParams the list of actual parameters
91        * @return a {@code Description} of the match <i>iff</i> for any of the actual parameters that
92        * is annotated with {@link com.google.errorprone.annotations.CompileTimeConstant}, the
93        * corresponding formal parameter is not a compile-time-constant expression in the sense of
94        * {@link CompileTimeConstantExpressionMatcher}. Otherwise returns {@code Description.NO_MATCH}.
95        */
96       private Description matchArguments(VisitorState state, final Symbol.MethodSymbol calleeSymbol,
97                                          Iterator<? extends ExpressionTree> actualParams) {
98         Symbol.VarSymbol lastFormalParam = null;
99         for (Symbol.VarSymbol formalParam : calleeSymbol.getParameters()) {
100           lastFormalParam = formalParam;
101           // It appears that for some reason, the Tree for implicit Enum constructors
102           // includes an invocation of super(), but the target symbol has the signature
103           // Enum(String, int). This resulted in NoSuchElementExceptions.
104           // It is safe to return no match in this case, since even if this could happen
105           // in another scenario, a non-existent actual parameter can't possibly
106           // be a non-constant parameter for a @CompileTimeConstant formal.
107           if (!actualParams.hasNext()) {
108             return Description.NO_MATCH;
109           }
110           ExpressionTree actualParam = actualParams.next();
111           if (hasCompileTimeConstantAnnotation(
112               state, formalParam)) {
113             if (!compileTimeConstExpressionMatcher.matches(actualParam, state)) {
114               return handleMatch(actualParam, state);
115             }
116           }
117         }
118     
119         // If the last formal parameter is a vararg and has the @CompileTimeConstant annotation,
120         // we need to check the remaining args as well.
121         if (lastFormalParam == null || (lastFormalParam.flags() & Flags.VARARGS) == 0) {
122           return Description.NO_MATCH;
123         }
124         if (!hasCompileTimeConstantAnnotation(
125             state, lastFormalParam)) {
126           return Description.NO_MATCH;
127         }
128         while (actualParams.hasNext()) {
129           ExpressionTree actualParam = actualParams.next();
130           if (!compileTimeConstExpressionMatcher.matches(actualParam, state)) {
131             return handleMatch(actualParam, state);
132           }
133         }
134         return Description.NO_MATCH;
135       }
136     
137       /**
138        * If the non-constant variable is annotated with @CompileTimeConstant, it must have been
139        * non-final. Suggest making it final in the error message.
140        */
141       private Description handleMatch(ExpressionTree actualParam, VisitorState state) {
142         Symbol sym = ASTHelpers.getSymbol(actualParam);
143         if (!(sym instanceof VarSymbol)) {
144           return describeMatch(actualParam);
145         }
146         VarSymbol var = (VarSymbol) sym;
147         if (!hasCompileTimeConstantAnnotation(state, var)) {
148           return describeMatch(actualParam);
149         }
150         return buildDescription(actualParam)
151             .setMessage(this.message()
152                 + String.format(DID_YOU_MEAN_FINAL_FMT_MESSAGE, var.getSimpleName()))
153             .build();
154       }
155     
156       @Override
157       public Description matchNewClass(NewClassTree tree, VisitorState state) {
158         JCTree.JCNewClass newClass = (JCTree.JCNewClass) tree;
159         return matchArguments(
160             state, (Symbol.MethodSymbol) newClass.constructor, tree.getArguments().iterator());
161       }
162     
163       @Override
164       public Description matchMethodInvocation(MethodInvocationTree tree, VisitorState state) {
165         ExpressionTree methodSelect = tree.getMethodSelect();
166         Symbol sym = ASTHelpers.getSymbol(methodSelect);
167         if (sym == null) {
168           return Description.NO_MATCH;
169         }
170         return matchArguments(state, (Symbol.MethodSymbol) sym, tree.getArguments().iterator());
171       }
172     }
