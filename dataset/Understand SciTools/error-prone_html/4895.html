48     /**
49      * @author rburny@google.com (Radoslaw Burny)
50      */
51     @BugPattern(name = "JUnit3TestNotRun",
52         summary = "Test method will not be run; please prefix name with \"test\"",
53         explanation = "JUnit 3 requires that test method names start with \"test\". The method that" +
54             " triggered this error looks like it is supposed to be the test, but either" +
55             " misspells the required prefix, or has @Test annotation, but no prefix." +
56             " As a consequence, JUnit 3 will ignore it.\n\n" +
57             "If you want to disable test on purpose, change the name to something more descriptive," +
58             " like \"disabledTestSomething()\". You don't need @Test annotation, but if you want to" +
59             " keep it, add @Ignore too.",
60         category = JUNIT, maturity = MATURE, severity = ERROR)
61     public class JUnit3TestNotRun extends BugChecker implements MethodTreeMatcher {
62     
63       /*
64        * Regular expression for test method name that is misspelled and should be replaced with "test".
65        * ".est" and "est"  are omitted, because they catch real words like "restore", "destroy", "best",
66        * "establish". ".test" is omitted, because people use it on purpose, to disable the test.
67        * Otherwise, I haven't found any false positives; "tes" was most common typo.
68        * There are some ambiguities in this regex that lead to bad corrections
69        * (i.e. tets -> tests, tesst -> testst), but the error is still found
70        * (those could be improved with regex lookahead, but I prefer simpler regex).
71        *  TODO(rburny): see if we can cleanup intentional ".test" misspellings
72        */
73       private static final Pattern MISSPELLED_NAME = Pattern.compile(
74           "t.est|te.st|"    +  // letter inserted
75           "tst|tet|tes|"    +  // letter removed
76           "etst|tset|tets|" +  // letters swapped
77           "t.st|te.t|"      +  // letter changed
78           "[tT][eE][sS][tT]"   // miscapitalized
79           );
80     
81       /**
82        * Matches if:
83        * 1) Method's name begins with misspelled variation of "test".
84        * 2) Method is public, returns void, and has no parameters.
85        * 3) Enclosing class is JUnit3 test (extends TestCase, has no RunWith annotation,
86        *    and is not abstract).
87        */
88       @Override
89       public Description matchMethod(MethodTree methodTree, VisitorState state) {
90         Matcher<MethodTree> methodMatcher = allOf(
91             not(methodNameStartsWith("test")),
92             Matchers.<MethodTree>hasModifier(Modifier.PUBLIC),
93             methodReturns(VOID_TYPE),
94             methodHasParameters(),
95             enclosingClass(isJUnit3TestClass));
96         if (!methodMatcher.matches(methodTree, state)) {
97           return Description.NO_MATCH;
98         }
99     
100         String name = methodTree.getName().toString();
101         String fixedName;
102         // regex.Matcher class name collides with errorprone.Matcher
103         java.util.regex.Matcher matcher = MISSPELLED_NAME.matcher(name);
104         if (matcher.lookingAt()) {
105           fixedName = matcher.replaceFirst("test");
106         } else if (wouldRunInJUnit4.matches(methodTree, state)) {
107           fixedName = "test" + name.substring(0, 1).toUpperCase() + name.substring(1);
108         } else {
109           return Description.NO_MATCH;
110         }
111         // We don't have start position for a method symbol, so we replace everything between result
112         // type and body.
113         JCMethodDecl decl = (JCMethodDecl) methodTree;
114         Fix fix = SuggestedFix.replace(
115             decl.restype.getStartPosition() + 4, decl.body.getStartPosition(), " " + fixedName + "() ");
116         return describeMatch(methodTree, fix);
117       }
118     }
