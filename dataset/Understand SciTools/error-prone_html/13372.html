17     import static com.google.common.util.concurrent.Futures.immediateFuture;
18     
19     import com.google.common.base.Function;
20     import com.google.common.util.concurrent.AsyncFunction;
21     import com.google.common.util.concurrent.ListenableFuture;
22     
23     import javax.annotation.Nullable;
24     
25     /**
26      * Negative cases for {@link AsyncFunctionReturnsNull}.
27      */
28     public class AsyncFunctionReturnsNullNegativeCases {
29       static {
30         new AsyncFunction<String, Object>() {
31           @Override
32           public ListenableFuture<Object> apply(String input) throws Exception {
33             return immediateFuture(null);
34           }
35         };
36     
37         new Function<String, Object>() {
38           @Override
39           public Object apply(String input) {
40             return null;
41           }
42         };
43     
44         new AsyncFunction<String, Object>() {
45           @Override
46           public ListenableFuture<Object> apply(String input) throws Exception {
47             return apply(input, input);
48           }
49     
50           public ListenableFuture<Object> apply(String input1, String input2) {
51             return null;
52           }
53         };
54     
55         new MyNonAsyncFunction<String, Object>() {
56           @Override
57           public ListenableFuture<Object> apply(String input) throws Exception {
58             return null;
59           }
60         };
61       }
62     
63       interface MyNonAsyncFunction<I, O> {
64         ListenableFuture<O> apply(@Nullable I input) throws Exception;
65       }
66     }
