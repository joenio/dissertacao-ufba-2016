29     import com.sun.source.tree.SynchronizedTree;
30     import com.sun.tools.javac.code.Flags;
31     import com.sun.tools.javac.code.Symbol;
32     import com.sun.tools.javac.code.Symbol.VarSymbol;
33     import com.sun.tools.javac.tree.JCTree.JCExpression;
34     import com.sun.tools.javac.tree.TreeInfo;

36     /**
37      * @author cushon@google.com (Liam Miller-Cushon)
38      */
39     @BugPattern(name = "SynchronizeOnNonFinalField",
40         summary = "Synchronizing on non-final fields is not safe: if the field is ever updated,"
41         + " different threads may end up locking on different objects.",
42         explanation = "Possible fixes:\n"
43             + "* If the field is already effectively final, add the missing 'final' modifier.\n"
44             + "* If the field needs to be mutable, create a separate lock by adding a private"
45             + "  final field and synchronizing on it to guard all accesses.",
46         category = JDK, severity = WARNING, maturity = MATURE)
47     public class SynchronizeOnNonFinalField extends BugChecker
48         implements BugChecker.SynchronizedTreeMatcher {
49     
50       @Override
51       public Description matchSynchronized(SynchronizedTree tree, VisitorState state) {
52         Symbol symbol = ASTHelpers.getSymbol(TreeInfo.skipParens((JCExpression) tree.getExpression()));
53         if (!(symbol instanceof VarSymbol)) {
54           return Description.NO_MATCH;
55         }
56     
57         // TODO(cushon): check that the receiver doesn't contain mutable state.
58         // Currently 'this.locks[i].mu' is accepted if 'mu' is final but 'locks' is non-final.
59         VarSymbol varSymbol = (VarSymbol) symbol;
60         if (varSymbol.isLocal()
61             || varSymbol.isStatic()
62             || (varSymbol.flags() & Flags.FINAL) != 0) {
63           return Description.NO_MATCH;
64         }
65     
66         return describeMatch(tree.getExpression());
67       }
68     }
