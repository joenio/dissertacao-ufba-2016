20     /**
21      * This tests the case where there is a chain of method overrides where the varargs constraint is
22      * not met, and the root has an array parameter.
23      * TODO(cushon): The original implementation tried to be clever and make this consistent, but
24      * didn't handle multiple interface inheritance.
25      *
26      * @author cushon@google.com (Liam Miller-Cushon)
27      */
28     public class OverridesPositiveCase3 {
29       abstract class Base {
30         abstract void arrayMethod(Object[] xs);
31       }
32     
33       abstract class SubOne extends Base {
34         @Override
35         // BUG: Diagnostic contains:
36         abstract void arrayMethod(Object... newNames);
37       }
38     
39       abstract class SubTwo extends SubOne {
40         @Override
41         // BUG: Diagnostic contains:
42         abstract void arrayMethod(Object[] xs);
43       }
44     
45       abstract class SubThree extends SubTwo {
46         @Override
47         // BUG: Diagnostic contains:
48         abstract void arrayMethod(Object... newNames);
49       }
50     }
