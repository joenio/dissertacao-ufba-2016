33     /**
34      * {@link UTree} representation of an {@link IfTree}.
35      *
36      * @author lowasser@google.com (Louis Wasserman)
37      */
38     @AutoValue
39     abstract class UIf implements UStatement, IfTree {
40       public static UIf create(
41           UExpression condition, UStatement thenStatement, UStatement elseStatement) {
42         return new AutoValue_UIf(condition, thenStatement, elseStatement);
43       }
44     
45       @Override
46       public abstract UExpression getCondition();
47     
48       @Override
49       public abstract UStatement getThenStatement();
50     
51       @Override
52       @Nullable
53       public abstract UStatement getElseStatement();
54     
55       @Override
56       public <R, D> R accept(TreeVisitor<R, D> visitor, D data) {
57         return visitor.visitIf(this, data);
58       }
59     
60       @Override
61       public Kind getKind() {
62         return Kind.IF;
63       }
64     
65       private static Function<Unifier, Choice<Unifier>> unifyUStatementWithSingleStatement(
66           @Nullable final UStatement toUnify, @Nullable final StatementTree target) {
67         return new Function<Unifier, Choice<Unifier>>() {
68           @Override
69           public Choice<Unifier> apply(Unifier unifier) {
70             if (toUnify == null) {
71               return (target == null) ? Choice.of(unifier) : Choice.<Unifier>none();
72             }
73             List<StatementTree> list = (target == null) ? List.<StatementTree>nil() : List.of(target);
74             return toUnify
75                 .apply(UnifierWithUnconsumedStatements.create(unifier, list))
76                 .thenOption(
77                     new Function<UnifierWithUnconsumedStatements, Optional<Unifier>>() {
78                       @Override
79                       public Optional<Unifier> apply(UnifierWithUnconsumedStatements state) {
80                         return state.unconsumedStatements().isEmpty()
81                             ? Optional.of(state.unifier())
82                             : Optional.<Unifier>absent();
83                       }
84                     });
85           }
86         };
87       }
88     
89       @Override
90       @Nullable
91       public Choice<UnifierWithUnconsumedStatements> apply(UnifierWithUnconsumedStatements state) {
92         java.util.List<? extends StatementTree> unconsumedStatements = state.unconsumedStatements();
93         if (unconsumedStatements.isEmpty()) {
94           return Choice.none();
95         }
96         final java.util.List<? extends StatementTree> unconsumedStatementsTail =
97             unconsumedStatements.subList(1, unconsumedStatements.size());
98         StatementTree firstStatement = unconsumedStatements.get(0);
99         if (firstStatement.getKind() != Kind.IF) {
100           return Choice.none();
101         }
102         final IfTree ifTree = (IfTree) firstStatement;
103         Unifier unifier = state.unifier();
104         Choice<UnifierWithUnconsumedStatements> forwardMatch =
105             getCondition()
106                 .unify(ifTree.getCondition(), unifier.fork())
107                 .thenChoose(
108                     unifyUStatementWithSingleStatement(getThenStatement(), ifTree.getThenStatement()))
109                 .thenChoose(
110                     new Function<Unifier, Choice<UnifierWithUnconsumedStatements>>() {
111                       @Override
112                       public Choice<UnifierWithUnconsumedStatements> apply(Unifier unifier) {
113                         if (getElseStatement() != null
114                             && ifTree.getElseStatement() == null
115                             && ControlFlowVisitor.INSTANCE.visitStatement(ifTree.getThenStatement())
116                                 == Result.ALWAYS_RETURNS) {
117                           Choice<UnifierWithUnconsumedStatements> result =
118                               getElseStatement()
119                                   .apply(
120                                       UnifierWithUnconsumedStatements.create(
121                                           unifier.fork(), unconsumedStatementsTail));
122                           if (getElseStatement() instanceof UBlock) {
123                             Choice<UnifierWithUnconsumedStatements> alternative =
124                                 Choice.of(
125                                     UnifierWithUnconsumedStatements.create(
126                                         unifier.fork(), unconsumedStatementsTail));
127                             for (UStatement stmt : ((UBlock) getElseStatement()).getStatements()) {
128                               alternative = alternative.thenChoose(stmt);
129                             }
130                             result = result.or(alternative);
131                           }
132                           return result;
133                         } else {
134                           return unifyUStatementWithSingleStatement(
135                                   getElseStatement(), ifTree.getElseStatement())
136                               .apply(unifier)
137                               .transform(
138                                   new Function<Unifier, UnifierWithUnconsumedStatements>() {
139                                     @Override
140                                     public UnifierWithUnconsumedStatements apply(Unifier unifier) {
141                                       return UnifierWithUnconsumedStatements.create(
142                                           unifier, unconsumedStatementsTail);
143                                     }
144                                   });
145                         }
146                       }
147                     });
148         Choice<UnifierWithUnconsumedStatements> backwardMatch =
149             getCondition()
150                 .negate()
151                 .unify(ifTree.getCondition(), unifier.fork())
152                 .thenChoose(
153                     new Function<Unifier, Choice<Unifier>>() {
154                       @Override
155                       public Choice<Unifier> apply(Unifier unifier) {
156                         if (getElseStatement() == null) {
157                           return Choice.none();
158                         }
159                         return getElseStatement()
160                             .apply(
161                                 UnifierWithUnconsumedStatements.create(
162                                     unifier, List.of(ifTree.getThenStatement())))
163                             .thenOption(
164                                 new Function<UnifierWithUnconsumedStatements, Optional<Unifier>>() {
165                                   @Override
166                                   public Optional<Unifier> apply(
167                                       UnifierWithUnconsumedStatements state) {
168                                     return state.unconsumedStatements().isEmpty()
169                                         ? Optional.of(state.unifier())
170                                         : Optional.<Unifier>absent();
171                                   }
172                                 });
173                       }
174                     })
175                 .thenChoose(
176                     new Function<Unifier, Choice<UnifierWithUnconsumedStatements>>() {
177                       @Override
178                       public Choice<UnifierWithUnconsumedStatements> apply(Unifier unifier) {
179                         if (ifTree.getElseStatement() == null
180                             && ControlFlowVisitor.INSTANCE.visitStatement(ifTree.getThenStatement())
181                                 == Result.ALWAYS_RETURNS) {
182                           Choice<UnifierWithUnconsumedStatements> result =
183                               getThenStatement()
184                                   .apply(
185                                       UnifierWithUnconsumedStatements.create(
186                                           unifier.fork(), unconsumedStatementsTail));
187                           if (getThenStatement() instanceof UBlock) {
188                             Choice<UnifierWithUnconsumedStatements> alternative =
189                                 Choice.of(
190                                     UnifierWithUnconsumedStatements.create(
191                                         unifier.fork(), unconsumedStatementsTail));
192                             for (UStatement stmt : ((UBlock) getThenStatement()).getStatements()) {
193                               alternative = alternative.thenChoose(stmt);
194                             }
195                             result = result.or(alternative);
196                           }
197                           return result;
198                         } else {
199                           return unifyUStatementWithSingleStatement(
200                                   getThenStatement(), ifTree.getElseStatement())
201                               .apply(unifier)
202                               .transform(
203                                   new Function<Unifier, UnifierWithUnconsumedStatements>() {
204                                     @Override
205                                     public UnifierWithUnconsumedStatements apply(Unifier unifier) {
206                                       return UnifierWithUnconsumedStatements.create(
207                                           unifier, unconsumedStatementsTail);
208                                     }
209                                   });
210                         }
211                       }
212                     });
213         return forwardMatch.or(backwardMatch);
214       }
215     
216       @Override
217       public List<JCStatement> inlineStatements(Inliner inliner) throws CouldNotResolveImportException {
218         return List.<JCStatement>of(
219             inliner
220                 .maker()
221                 .If(
222                     getCondition().inline(inliner),
223                     Iterables.getOnlyElement(getThenStatement().inlineStatements(inliner)),
224                     (getElseStatement() == null)
225                         ? null
226                         : Iterables.getOnlyElement(getElseStatement().inlineStatements(inliner))));
227       }
