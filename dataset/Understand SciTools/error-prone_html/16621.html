17     package com.google.errorprone.refaster.testdata.template;
18     
19     import com.google.common.collect.Maps;
20     import com.google.errorprone.refaster.ImportPolicy;
21     import com.google.errorprone.refaster.annotation.AfterTemplate;
22     import com.google.errorprone.refaster.annotation.BeforeTemplate;
23     import com.google.errorprone.refaster.annotation.UseImportPolicy;
24     
25     import java.util.AbstractMap;
26     import java.util.Map;
27     
28     /**
29      * Sample template generating references to nested classes in java.util.
30      */
31     public class SamePackageImportsTemplate<K, V> {
32       @BeforeTemplate Map.Entry<K, V> immutableEntry(K k, V v) {
33         return Maps.immutableEntry(k, v);
34       }
35     
36       @AfterTemplate
37       @UseImportPolicy(ImportPolicy.IMPORT_CLASS_DIRECTLY)
38       Map.Entry<K, V> abstractMapEntry(K k, V v) {
39         return new AbstractMap.SimpleImmutableEntry<>(k, v);
40       }
41     }
