17     package com.google.errorprone.refaster;
18     
19     import static com.google.common.base.Preconditions.checkArgument;
20     import static com.google.common.base.Preconditions.checkState;
21     
22     import com.google.auto.value.AutoValue;
23     import com.google.common.collect.ImmutableBiMap;
24     
25     import com.sun.source.tree.LiteralTree;
26     import com.sun.source.tree.TreeVisitor;
27     import com.sun.tools.javac.code.TypeTag;
28     import com.sun.tools.javac.tree.JCTree.JCLiteral;
29     
30     import java.util.Objects;
31     
32     import javax.annotation.Nullable;
33     
34     /**
35      * {@link UTree} version of {@link LiteralTree}.
36      *
37      * @author lowasser@google.com (Louis Wasserman)
38      */
39     @AutoValue
40     public abstract class ULiteral extends UExpression implements LiteralTree {
41       
42       public static ULiteral nullLit() {
43         return create(Kind.NULL_LITERAL, null);
44       }
45     
46       public static ULiteral intLit(int value) {
47         return create(Kind.INT_LITERAL, value);
48       }
49     
50       public static ULiteral longLit(long value) {
51         return create(Kind.LONG_LITERAL, value);
52       }
53       
54       public static ULiteral floatLit(float value) {
55         return create(Kind.FLOAT_LITERAL, value);
56       }
57       
58       public static ULiteral doubleLit(double value) {
59         return create(Kind.DOUBLE_LITERAL, value);
60       }
61       
62       public static ULiteral booleanLit(boolean value) {
63         return create(Kind.BOOLEAN_LITERAL, value);
64       }
65       
66       public static ULiteral charLit(char value) {
67         return create(Kind.CHAR_LITERAL, value);
68       }
69       
70       public static ULiteral stringLit(String value) {
71         return create(Kind.STRING_LITERAL, value);
72       }
73     
74       private static final ImmutableBiMap<Kind, TypeTag> LIT_KIND_TAG =
75           new ImmutableBiMap.Builder<Kind, TypeTag>()
76             .put(Kind.INT_LITERAL, TypeTag.INT)
77             .put(Kind.LONG_LITERAL, TypeTag.LONG)
78             .put(Kind.FLOAT_LITERAL, TypeTag.FLOAT)
79             .put(Kind.DOUBLE_LITERAL, TypeTag.DOUBLE)
80             .put(Kind.CHAR_LITERAL, TypeTag.CHAR)
81             .put(Kind.BOOLEAN_LITERAL, TypeTag.BOOLEAN)
82             .put(Kind.NULL_LITERAL, TypeTag.BOT)
83             .put(Kind.STRING_LITERAL, TypeTag.CLASS)
84             .build();
85       
86       public static ULiteral create(Kind kind, Object value) {
87         checkArgument(LIT_KIND_TAG.containsKey(kind), "%s is not a literal kind", kind);
88         return new AutoValue_ULiteral(kind, value);
89       }
90     
91       @Override
92       public abstract Kind getKind();
93     
94       @Override
95       @Nullable
96       public abstract Object getValue();
97     
98       private static boolean integral(@Nullable Object o) {
99         return o instanceof Integer || o instanceof Long;
100       }
101     
102       private static boolean match(@Nullable Object a, @Nullable Object b) {
103         if (a instanceof Number && b instanceof Number) {
104           return (integral(a) && integral(b))
105               ? ((Number) a).longValue() == ((Number) b).longValue()
106               : Double.compare(((Number) a).doubleValue(), ((Number) b).doubleValue()) == 0;
107         } else {
108           return Objects.equals(a, b);
109         }
110       }
111     
112       @Override
113       public Choice<Unifier> visitLiteral(LiteralTree literal, Unifier unifier) {
114         return Choice.condition(match(getValue(), literal.getValue()), unifier);
115       }
116     
117       @Override
118       public <R, D> R accept(TreeVisitor<R, D> visitor, D data) {
119         return visitor.visitLiteral(this, data);
120       }
121     
122       @Override
123       public JCLiteral inline(Inliner inliner) {
124         Object value = this.getValue();
125         switch (getKind()) {
126           case CHAR_LITERAL:
127             // Why do they do it like this?  I wish I knew.
128             value = (int) ((Character) value).charValue();
129             break;
130           case BOOLEAN_LITERAL:
131             value = ((Boolean) value) ? 1 : 0;
132             break;
133           default:
134             // do nothing
135         }
136         return inliner.maker().Literal(LIT_KIND_TAG.get(getKind()), value);
137       }
138     
139       @Override
140       public UExpression negate() {
141         checkState(getKind() == Kind.BOOLEAN_LITERAL, "Cannot negate a non-Boolean literal");
142         return booleanLit(!((Boolean) getValue()));
143       }
144     }
