17     package com.google.errorprone.matchers;
18     
19     import com.google.errorprone.VisitorState;
20     import com.google.errorprone.annotations.CompileTimeConstant;
21     import com.google.errorprone.util.ASTHelpers;
22     
23     import com.sun.source.tree.ExpressionTree;
24     import com.sun.source.tree.Tree;
25     import com.sun.tools.javac.code.Flags;
26     import com.sun.tools.javac.code.Symbol;
27     import com.sun.tools.javac.tree.JCTree;
28     
29     import javax.lang.model.element.ElementKind;
30     
31     /**
32      * A matcher for compile-time-constant expressions.
33      *
34      * <p>
35      * For the purposes of this matcher, a compile-time constant expression is one of the following:
36      * <ol>
37      * <li>Any expression for which the Java compiler can determine a constant value at compile time.
38      * <li>The expression consisting of the literal {@code null}.
39      * <li>An expression consisting of a single identifier, where the identifier is a formal method
40      * parameter that is declared {@code final} and has the {@link CompileTimeConstant} annotation.
41      * </ol>
42      */
43     public class CompileTimeConstantExpressionMatcher implements Matcher<ExpressionTree> {
44     
45       private static final String COMPILE_TIME_CONSTANT_ANNOTATION =
46           CompileTimeConstant.class.getName();
47     
48       // TODO(cushon): remove once the migration is done
49       private static final String GUAVA_COMPILE_TIME_CONSTANT_ANNOTATION =
50           "com.google.common.annotations.CompileTimeConstant";
51     
52       @SuppressWarnings("unchecked")
53       private final Matcher<ExpressionTree> matcher = Matchers.anyOf(
54           // TODO(xtof): Consider utilising mdempksy's closed-over-addition matcher (perhaps extended
55           // for other arithmetic operations).
56           new ExpressionWithConstValueMatcher(),
57           Matchers.kindIs(Tree.Kind.NULL_LITERAL),
58           new FinalCompileTimeConstantParameterMatcher()
59           );
60     
61       @Override
62       public boolean matches(ExpressionTree t, VisitorState state) {
63         return matcher.matches(t, state);
64       }
65     
66       // TODO(xtof): Perhaps some of these matchers could be generally useful, in which case they should
67       // be moved into c.g.errorprone.matchers.
68     
69       /**
70        * A matcher for {@link ExpressionTree}s for which the java compiler can compute a constant value
71        * (except a literal {@code null}).
72        */
73       private static final class ExpressionWithConstValueMatcher implements Matcher<ExpressionTree> {
74     
75         @Override
76         public boolean matches(ExpressionTree t, VisitorState state) {
77           Object constValue = ((JCTree.JCExpression) t).type.constValue();
78           return constValue != null;
79         }
80       }
81     
82       /**
83        * A matcher that matches a {@code @CompileTimeConstant final} parameter}.
84        */
85       private static final class FinalCompileTimeConstantParameterMatcher
86           implements Matcher<ExpressionTree> {
87     
88         @Override
89         public boolean matches(ExpressionTree t, VisitorState state) {
90           if (t.getKind() != Tree.Kind.IDENTIFIER) {
91             return false;
92           }
93           Symbol.VarSymbol varSymbol = (Symbol.VarSymbol) ASTHelpers.getSymbol(t);
94           Symbol owner = varSymbol.owner;
95           ElementKind ownerKind = owner.getKind();
96           // Check that the identifier is a formal method/constructor parameter.
97           if (ownerKind != ElementKind.METHOD && ownerKind != ElementKind.CONSTRUCTOR) {
98             return false;
99           }
100           // Check that the symbol is declared {@code final}.
101           if ((varSymbol.flags() & Flags.FINAL) == 0) {
102             return false;
103           }
104           // Check if the symbol has the @CompileTimeConstant annotation.
105           if (hasCompileTimeConstantAnnotation(state, varSymbol)) {
106             return true;
107           }
108           return false;
109         }
110       }
111     
112       private static boolean hasAttribute(Symbol symbol, String name, VisitorState state) {
113         Symbol annotation = state.getSymbolFromString(name);
114         // If we can't look up the annotation in the current VisitorState, then presumably it couldn't
115         // be present on a Symbol we're inspecting.
116         return annotation != null && symbol.attribute(annotation) != null;
117       }
118     
119       // default visibility, since this is also used by CompileTimeConstantTypeAnnotationChecker.
120       public static boolean hasCompileTimeConstantAnnotation(VisitorState state, Symbol symbol) {
121         return hasAttribute(symbol, COMPILE_TIME_CONSTANT_ANNOTATION, state)
122             || hasAttribute(symbol, GUAVA_COMPILE_TIME_CONSTANT_ANNOTATION, state);
123       }
124     }
