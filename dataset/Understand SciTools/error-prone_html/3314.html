24     public class CheckReturnValuePositiveCases {

26       IntValue intValue = new IntValue(0);
27       
28       @CheckReturnValue
29       private int increment(int bar) {
30         return bar + 1;
31       }
32       
33       public void foo() {
34         int i = 1;
35         // BUG: Diagnostic contains: remove this line
36         increment(i);
37         System.out.println(i);
38       }
39       
40       public void bar() {
41         // BUG: Diagnostic contains: this.intValue = this.intValue.increment()
42         this.intValue.increment();
43       }
44       
45       public void testIntValue() {
46         IntValue value = new IntValue(10);
47         // BUG: Diagnostic contains: value = value.increment()
48         value.increment();
49       }

51       public void constructor() {
52         /*
53          * We may or may not want to treat this as a bug. On the one hand, the
54          * subclass might be "using" the superclass, so it might not be being
55          * "ignored." (Plus, it would be a pain to produce a valid suggested fix
56          * that incorporates any subclass constructor body, which might even contain
57          * calls to methods in the class.) On the other hand, the more likely
58          * scenario may be a class like IteratorTester, which requires (a) that the
59          * user subclass it to implement a method and (b) that the user call test()
60          * on the constructed object. There, it would be nice if IteratorTester
61          * could be annotated with @CheckReturnValue to mean "anyone who creates an
62          * anonymous subclasses of this should still do something with that
63          * subclass." But perhaps that's an abuse of @CheckForNull.
64          *
65          * Anyway, these tests are here to ensure that subclasses don't don't crash
66          * the compiler.
67          */
68         new MyObject() {};
69     
70         class MySubObject1 extends MyObject {
71         }
72     
73         class MySubObject2 extends MyObject {
74           MySubObject2() {
75           }
76         }
77     
78         class MySubObject3 extends MyObject {
79           MySubObject3() {
80             super();
81           }
82         }
83     
84         // TODO(cpovirk): This one probably ought to be treated as a bug:
85         new MyObject();
86       }

88       private class IntValue {
89         final int i;
90         
91         public IntValue(int i) {
92           this.i = i;
93         }
94         
95         @CheckReturnValue
96         public IntValue increment() {
97           return new IntValue(i + 1);
98         }
99         
100         public void increment2() {
101           // BUG: Diagnostic contains: remove this line
102           this.increment();
103         }
104         
105         public void increment3() {
106           // BUG: Diagnostic contains: remove this line
107          increment();
108         }
109       }
