53        * A pair of Analysis and ControlFlowGraph.
54        */
55       public static interface Result<A extends AbstractValue<A>, S extends Store<S>,
56           T extends TransferFunction<A, S>> {
57         Analysis<A, S, T> getAnalysis();
58     
59         ControlFlowGraph getControlFlowGraph();
60       }

62       /*
63        * We cache both the control flow graph and the analyses that are run on it.
64        * We tuned performance to the following assumptions (which are currently true for error-prone):
65        *
66        * <ul>
67        * <li> all dataflow analyses for a method are finished before another method is analyzed
68        * <li> multiple dataflow analyses for the same method are executed in arbitrary order
69        * </ul>
70        *
71        * TODO(user): Write a test that checks these assumptions
72        */
73       private static LoadingCache<AnalysisParams, Analysis<?, ?, ?>> analysisCache =
74           CacheBuilder.newBuilder().build(
75               new CacheLoader<AnalysisParams, Analysis<?, ?, ?>>() {
76                 @Override
77                 public Analysis<?, ?, ?> load(AnalysisParams key) {
78                   final ProcessingEnvironment env = key.getEnvironment();
79                   final ControlFlowGraph cfg = key.getCFG();
80                   final TransferFunction<?, ?> transfer = key.getTransferFunction();
81     
82                   @SuppressWarnings({"unchecked", "rawtypes"})
83                   final Analysis<?, ?, ?> analysis = new Analysis(env, transfer);
84                   analysis.performAnalysis(cfg);
85                   return analysis;
86                 }
87               });
88     
89       private static LoadingCache<CFGParams, ControlFlowGraph> cfgCache =
90           CacheBuilder.newBuilder().maximumSize(1).build(
91               new CacheLoader<CFGParams, ControlFlowGraph>() {
92                 @Override
93                 public ControlFlowGraph load(CFGParams key) {
94                   final TreePath methodPath = key.getMethodPath();
95                   final MethodTree method = (MethodTree) methodPath.getLeaf();
96                   final BlockTree body = method.getBody();
97                   final TreePath bodyPath = new TreePath(methodPath, body);
98                   final ClassTree classTree = null;
99                   final UnderlyingAST ast = new UnderlyingAST.CFGMethod(method, classTree);
100                   final ProcessingEnvironment env = key.getEnvironment();
101     
102                   analysisCache.invalidateAll();
103                   CompilationUnitTree root = bodyPath.getCompilationUnit();
104                   // TODO(user), replace with faster build(bodyPath, env, ast, false, false);
105                   return CFGBuilder.build(root, env, ast, false, false);
106                 }
107               });

109       // TODO(user), remove once we merge jdk8 specific's with core
110       public static <T> TreePath findPathFromEnclosingNodeToTopLevel(TreePath path,
111           Class<T> klass) {
112         while (path != null && !(klass.isInstance(path.getLeaf()))) {
113           path = path.getParentPath();
114         }
115         return path;
116       }

118       /**
119        * Run the {@code transfer} dataflow analysis over the method which is the leaf of the
120        * {@code methodPath}.
121        *
122        * <p>For caching, we make the following assumptions:
123        * - if two paths to methods are {@code equal}, their control flow graph is the same.
124        * - if two transfer functions are {@code equal}, and are run over the same control flow graph,
125        *   the analysis result is the same.
126        * - for all contexts, the analysis result is the same.
127        */
128       public static <A extends AbstractValue<A>, S extends Store<S>,
129                      T extends TransferFunction<A, S>> Result<A, S, T>
130           methodDataflow(TreePath methodPath, Context context, T transfer) {
131         final Tree leaf = methodPath.getLeaf();
132         Preconditions.checkArgument(leaf instanceof MethodTree,
133             "Leaf of methodPath must be of type MethodTree, but was %s", leaf.getClass().getName());
134     
135         final MethodTree method = (MethodTree) leaf;
136         Preconditions.checkNotNull(method.getBody(),
137             "Method to analyze must have a body. Method passed in: %s() in file %s",
138             method.getName(),
139             methodPath.getCompilationUnit().getSourceFile().getName());
140     
141         final ProcessingEnvironment env = JavacProcessingEnvironment.instance(context);
142         final ControlFlowGraph cfg = cfgCache.getUnchecked(new CFGParams(methodPath, env));
143         final AnalysisParams aparams = new AnalysisParams(transfer, cfg, env);
144         @SuppressWarnings("unchecked")
145         final Analysis<A, S, T> analysis = (Analysis<A, S, T>) analysisCache.getUnchecked(aparams);
146     
147         return new Result<A, S, T>() {
148           @Override
149           public Analysis<A, S, T> getAnalysis() {
150             return analysis;
151           }
152     
153           @Override
154           public ControlFlowGraph getControlFlowGraph() {
155             return cfg;
156           }
157         };
158       }
159     
160       /**
161        * Run the {@code transfer} dataflow analysis to compute the abstract value of the expression
162        * which is the leaf of {@code exprPath}.
163        */
164       public static <A extends AbstractValue<A>, S extends Store<S>,
165                      T extends TransferFunction<A, S>> A
166           expressionDataflow(TreePath exprPath, Context context, T transfer) {
167         final Tree leaf = exprPath.getLeaf();
168         Preconditions.checkArgument(leaf instanceof ExpressionTree,
169             "Leaf of exprPath must be of type ExpressionTree, but was %s", leaf.getClass().getName());
170     
171         final ExpressionTree expr = (ExpressionTree) leaf;
172         final TreePath enclosingMethodPath =
173             findPathFromEnclosingNodeToTopLevel(exprPath, MethodTree.class);
174     
175         if (enclosingMethodPath == null) {
176           // TODO(user) this can happen in field initialization.
177           // Currently not supported because it only happens in ~2% of cases.
178           return null;
179         }
180     
181         final MethodTree method = (MethodTree) enclosingMethodPath.getLeaf();
182         if (method.getBody() == null) {
183           // expressions can occur in abstract methods, for example {@code Map.Entry} in:
184           //
185           //   abstract Set<Map.Entry<K, V>> entries();
186           return null;
187         }
188     
189         return methodDataflow(enclosingMethodPath, context, transfer).getAnalysis().getValue(expr);
190       }

192       private static final class CFGParams {
193         final ProcessingEnvironment env;
194         final TreePath methodPath;
195     
196         public ProcessingEnvironment getEnvironment() {
197           return env;
198         }
199     
200         public TreePath getMethodPath() {
201           return methodPath;
202         }
203     
204         public CFGParams(TreePath methodPath, ProcessingEnvironment env) {
205           this.env = env;
206           this.methodPath = methodPath;
207         }
208     
209         @Override
210         public int hashCode() {
211           return Objects.hash(methodPath);
212         }
213     
214         @Override
215         public boolean equals(Object obj) {
216           if (this == obj) {
217             return true;
218           }
219           if (obj == null) {
220             return false;
221           }
222           if (getClass() != obj.getClass()) {
223             return false;
224           }
225           CFGParams other = (CFGParams) obj;
226           return Objects.equals(methodPath, other.methodPath);
227         }
228       }
229     
230       private static final class AnalysisParams {
231         private final ProcessingEnvironment env;
232         private final ControlFlowGraph cfg;
233         private final TransferFunction<?, ?> transfer;
234     
235         public AnalysisParams(TransferFunction<?, ?> trans, ControlFlowGraph cfg,
236             ProcessingEnvironment env) {
237           this.env = env;
238           this.cfg = cfg;
239           this.transfer = trans;
240         }
241     
242         public ProcessingEnvironment getEnvironment() {
243           return env;
244         }
245     
246         public ControlFlowGraph getCFG() {
247           return cfg;
248         }
249     
250         @SuppressWarnings("rawtypes")
251         public TransferFunction getTransferFunction() {
252           return transfer;
253         }
254     
255         @Override
256         public int hashCode() {
257           return Objects.hash(cfg, transfer);
258         }
259     
260         @Override
261         public boolean equals(Object obj) {
262           if (this == obj) {
263             return true;
264           }
265           if (obj == null) {
266             return false;
267           }
268           if (getClass() != obj.getClass()) {
269             return false;
270           }
271           AnalysisParams other = (AnalysisParams) obj;
272           return Objects.equals(cfg, other.cfg)
273               && Objects.equals(transfer, other.transfer);
274         }
275       }
