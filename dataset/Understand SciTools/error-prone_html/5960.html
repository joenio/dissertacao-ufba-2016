24     import com.sun.source.tree.BlockTree;
25     import com.sun.source.tree.StatementTree;
26     import com.sun.source.tree.TreeVisitor;
27     import com.sun.tools.javac.tree.JCTree.JCBlock;

29     import com.sun.tools.javac.util.ListBuffer;
30     
31     import java.util.List;
32     
33     /**
34      * {@link UTree} representation of a {@link BlockTree}.
35      *
36      * @author lowasser@google.com (Louis Wasserman)
37      */
38     @AutoValue
39     abstract class UBlock extends USimpleStatement implements BlockTree {
40       public static UBlock create(List<UStatement> statements) {
41         return new AutoValue_UBlock(ImmutableList.copyOf(statements));
42       }
43     
44       public static UBlock create(UStatement... statements) {
45         return create(ImmutableList.copyOf(statements));
46       }
47     
48       @Override
49       public abstract List<UStatement> getStatements();
50       
51       static Choice<Unifier> unifyStatementList(Iterable<? extends UStatement> statements,
52           Iterable<? extends StatementTree> targets, Unifier unifier) {
53         Choice<UnifierWithUnconsumedStatements> choice = Choice.of(
54             UnifierWithUnconsumedStatements.create(unifier, ImmutableList.copyOf(targets)));
55         for (UStatement statement : statements) {
56           choice = choice.thenChoose(statement);
57         }
58         return choice.thenOption(new Function<UnifierWithUnconsumedStatements, Optional<Unifier>>() {
59           @Override
60           public Optional<Unifier> apply(UnifierWithUnconsumedStatements state) {
61             return state.unconsumedStatements().isEmpty()
62                 ? Optional.of(state.unifier())
63                 : Optional.<Unifier>absent();
64           }
65         });
66       }
67       
68       static com.sun.tools.javac.util.List<JCStatement> inlineStatementList(
69           Iterable<? extends UStatement> statements, Inliner inliner) 
70               throws CouldNotResolveImportException {
71         ListBuffer<JCStatement> buffer = new ListBuffer<>();
72         for (UStatement statement : statements) {
73           buffer.appendList(statement.inlineStatements(inliner));
74         }
75         return buffer.toList();
76       }
77     
78       @Override
79       public Choice<Unifier> visitBlock(BlockTree block, Unifier unifier) {
80         return unifyStatementList(getStatements(), block.getStatements(), unifier);
81       }
82     
83       @Override
84       public JCBlock inline(Inliner inliner) throws CouldNotResolveImportException {
85         return inliner.maker().Block(0, inlineStatementList(getStatements(), inliner));
86       }
87     
88       @Override
89       public Kind getKind() {
90         return Kind.BLOCK;
91       }
92     
93       @Override
94       public <R, D> R accept(TreeVisitor<R, D> visitor, D data) {
95         return visitor.visitBlock(this, data);
96       }
97     
98       @Override
99       public boolean isStatic() {
100         return false;
101       }
102     }
