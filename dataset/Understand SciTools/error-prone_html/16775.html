39     /** @author cushon@google.com (Liam Miller-Cushon) */
40     @BugPattern(
41       name = "MissingOverride",
42       summary = "method overrides method in supertype; expected @Override",
43       category = JDK,
44       severity = WARNING,
45       maturity = MATURE
46     )
47     public class MissingOverride extends BugChecker implements MethodTreeMatcher {
48     
49       @Override
50       public Description matchMethod(MethodTree tree, VisitorState state) {
51         Symbol sym = ASTHelpers.getSymbol(tree);
52         if (sym.isStatic()) {
53           return Description.NO_MATCH;
54         }
55         if (ASTHelpers.hasAnnotation(sym, Override.class, state)) {
56           return Description.NO_MATCH;
57         }
58         MethodSymbol override = getFirstOverride(sym, state.getTypes());
59         if (override == null) {
60           return Description.NO_MATCH;
61         }
62         if (ASTHelpers.hasAnnotation(override, Deprecated.class, state)) {
63           // to allow deprecated methods to be removed non-atomically, we permit overrides of
64           // @Deprecated to skip the annotation
65           return Description.NO_MATCH;
66         }
67         return buildDescription(tree)
68             .addFix(SuggestedFix.prefixWith(tree, "@Override "))
69             .setMessage(
70                 String.format(
71                     "%s %s method in %s; expected @Override",
72                     sym.getSimpleName(),
73                     override.enclClass().isInterface()
74                             || override.getModifiers().contains(Modifier.ABSTRACT)
75                         ? "implements"
76                         : "overrides",
77                     override.enclClass().getSimpleName()))
78             .build();
79       }
80     
81       /**
82        * Returns the {@link MethodSymbol} of the first method that sym overrides in its supertype
83        * closure, or {@code null} if no such method exists.
84        */
85       private MethodSymbol getFirstOverride(Symbol sym, Types types) {
86         ClassSymbol owner = sym.enclClass();
87         for (Type s : types.closure(owner.type)) {
88           if (s == owner.type) {
89             continue;
90           }
91           for (Symbol m : s.tsym.members().getSymbolsByName(sym.name)) {
92             if (!(m instanceof MethodSymbol)) {
93               continue;
94             }
95             MethodSymbol msym = (MethodSymbol) m;
96             if (msym.isStatic()) {
97               continue;
98             }
99             if (sym.overrides(msym, owner, types, /*checkReturn*/ false)) {
100               return msym;
101             }
102           }
103         }
104         return null;
105       }
106     }
