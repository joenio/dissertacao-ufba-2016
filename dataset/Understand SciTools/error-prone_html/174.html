42     /**
43      * Tests for {@link Matchers#enclosingNode}, {@link Matchers#parentNode},
44      * {@link Matchers#enclosingBlock}, {@link Enclosing.BlockOrCase}, and at least some of the code
45      * paths also used by {@link Matchers#enclosingClass} and {@link Matchers#enclosingMethod}. The
46      * tests focus on verifying that the {@link TreePath} is set correctly.
47      *
48      * @author cpovirk@google.com (Chris Povirk)
49      */
50     @RunWith(JUnit4.class)
51     public class EnclosingTest extends CompilerBasedAbstractTest {
52       private abstract static class IsInterestingLoopSubNode<T extends Tree> implements Matcher<T> {
53         @Override
54         public boolean matches(T t, VisitorState state) {
55           if (state.getPath().getParentPath() == null) {
56             return false;
57           }
58           Tree parent = state.getPath().getParentPath().getLeaf();
59           return (parent instanceof ForLoopTree && (interestingPartOfLoop((ForLoopTree) parent) == t));
60         }
61     
62         abstract Object interestingPartOfLoop(ForLoopTree loop);
63       }
64     
65       private static final Matcher<Tree> IS_LOOP_CONDITION = new IsInterestingLoopSubNode<Tree>() {
66         @Override
67         Object interestingPartOfLoop(ForLoopTree loop) {
68           return loop.getCondition();
69         }
70       };
71       private static final Matcher<BlockTree> IS_LOOP_STATEMENT =
72           new IsInterestingLoopSubNode<BlockTree>() {
73             @Override
74             Object interestingPartOfLoop(ForLoopTree loop) {
75               return loop.getStatement();
76             }
77           };
78       private static final Matcher<Tree> ENCLOSED_IN_LOOP_CONDITION = enclosingNode(IS_LOOP_CONDITION);
79       private static final Matcher<Tree> CHILD_OF_LOOP_CONDITION = parentNode(IS_LOOP_CONDITION);
80       private static final Matcher<Tree> USED_UNDER_LOOP_STATEMENT = enclosingBlock(IS_LOOP_STATEMENT);
81       private static final Matcher<Tree> USED_UNDER_LOOP_STATEMENT_ACCORDING_TO_BLOCK_OR_CASE =
82           new Enclosing.BlockOrCase<>(IS_LOOP_STATEMENT, Matchers.<CaseTree>nothing());
83     
84       final List<ScannerTest> tests = new ArrayList<ScannerTest>();
85     
86       @After
87       public void tearDown() {
88         for (ScannerTest test : tests) {
89           test.assertDone();
90         }
91       }

93       /** Tests that a node is not enclosed by itself. */
94       @Test
95       public void usedDirectlyInLoopCondition() {
96         writeFile("A.java",
97             "public class A {",
98             "  A() {",
99             "    boolean foo = true;",
100             "    for (int i = 0; foo; i++) {}",
101             "  }",
102             "}");
103         assertCompiles(fooIsUsedUnderLoopCondition(false));
104         assertCompiles(fooIsChildOfLoopCondition(false));
105         assertCompiles(fooIsUsedUnderLoopStatement(false));
106         assertCompiles(fooIsUsedUnderLoopStatementAccordingToBlockOrCase(false));
107       }
108     
109       /** Tests that a node is enclosed by its parent. */
110       @Test
111       public void usedAsChildTreeOfLoopCondition() {
112         writeFile("A.java",
113             "public class A {",
114             "  A() {",
115             "    boolean foo = true;",
116             "    for (int i = 0; !foo; i++) {}",
117             "  }",
118             "}");
119         assertCompiles(fooIsUsedUnderLoopCondition(true));
120         assertCompiles(fooIsChildOfLoopCondition(true));
121         assertCompiles(fooIsUsedUnderLoopStatement(false));
122         assertCompiles(fooIsUsedUnderLoopStatementAccordingToBlockOrCase(false));
123       }
124     
125       /** Tests that a node is enclosed by a node many levels up the tree. */
126       @Test
127       public void usedInSubTreeOfLoopCondition() {
128         writeFile("A.java",
129             "public class A {",
130             "  A() {",
131             "    boolean foo = true;",
132             "    for (int i = 0; !!!!!!!!!foo; i++) {}",
133             "  }",
134             "}");
135         assertCompiles(fooIsUsedUnderLoopCondition(true));
136         assertCompiles(fooIsChildOfLoopCondition(false));
137         assertCompiles(fooIsUsedUnderLoopStatement(false));

139       }

141       /** Tests enclosing blocks. */
142       @Test
143       public void usedInStatement() {
144         writeFile("A.java",
145             "public class A {",
146             "  A() {",
147             "    boolean foo = true;",
148             "    for (int i = 0; i < 100; i++) {",
149             "      foo = !foo;",
150             "    }",
151             "  }",
152             "}");
153         assertCompiles(fooIsUsedUnderLoopCondition(false));
154         assertCompiles(fooIsChildOfLoopCondition(false));
155         assertCompiles(fooIsUsedUnderLoopStatement(true));
156         assertCompiles(fooIsUsedUnderLoopStatementAccordingToBlockOrCase(true));
157       }

159       /** Sanity checks that the scanners are doing what we expect. */
160       @Test
161       public void usedElsewhereInLoop() {
162         writeFile("A.java",
163             "public class A {",
164             "  A() {",
165             "    boolean foo = true;",
166             "    for (int i = foo ? 0 : 1; i < 100; foo = !foo) {}",
167             "  }",
168             "}");
169         assertCompiles(fooIsUsedUnderLoopCondition(false));
170         assertCompiles(fooIsChildOfLoopCondition(false));
171         assertCompiles(fooIsUsedUnderLoopStatement(false));
172         assertCompiles(fooIsUsedUnderLoopStatementAccordingToBlockOrCase(false));
173       }

175       private abstract class ScannerTest extends Scanner {
176         abstract void assertDone();
177       }
178     
179       private Scanner fooIsUsedUnderLoopCondition(boolean shouldMatch) {
180         return fooMatches(shouldMatch, ENCLOSED_IN_LOOP_CONDITION);
181       }
182     
183       private Scanner fooIsChildOfLoopCondition(boolean shouldMatch) {
184         return fooMatches(shouldMatch, CHILD_OF_LOOP_CONDITION);
185       }
186     
187       private Scanner fooIsUsedUnderLoopStatement(boolean shouldMatch) {
188         return fooMatches(shouldMatch, USED_UNDER_LOOP_STATEMENT);
189       }
190     
191       private Scanner fooIsUsedUnderLoopStatementAccordingToBlockOrCase(boolean shouldMatch) {
192         return fooMatches(shouldMatch, USED_UNDER_LOOP_STATEMENT_ACCORDING_TO_BLOCK_OR_CASE);
193       }
194     
195       private Scanner fooMatches(final boolean shouldMatch, final Matcher<Tree> matcher) {
196         ScannerTest test = new ScannerTest() {
197           boolean matched = false;
198     
199           @Override
200           public Void visitIdentifier(IdentifierTree tree, VisitorState state) {
201             // Normally handled by ErrorProneMatcher:
202             // TODO(cpovirk): Find a way for this to be available by default to Matcher tests.
203             state = state.withPath(getCurrentPath());
204     
205             matched |= tree.getName().contentEquals("foo") && matcher.matches(tree, state);
206             return null;
207           }
208     
209           @Override
210           void assertDone() {
211             assertEquals(shouldMatch, matched);
212           }
213         };
214         tests.add(test);
215         return test;
216       }
