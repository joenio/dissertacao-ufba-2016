48     /**
49      * Test for unsuppressible checks.
50      *
51      * @author eaftan@google.com (Eddie Aftandilan)
52      */
53     @RunWith(JUnit4.class)
54     public class UnsuppressibleTest {
55     
56       @BugPattern(name = "MyChecker",
57           summary = "Test checker that is unsuppressible",
58           explanation = "Test checker that that is unsuppressible",
59           suppressibility = Suppressibility.UNSUPPRESSIBLE,
60           category = ONE_OFF, severity = ERROR, maturity = MATURE)
61       public static class MyChecker extends BugChecker implements ReturnTreeMatcher {
62         @Override
63         public Description matchReturn(ReturnTree tree, VisitorState state) {
64           return describeMatch(tree);
65         }
66       }
67     
68       private ErrorProneTestCompiler compiler;
69       private DiagnosticTestHelper diagnosticHelper;
70     
71       @Before
72       public void setUp() {
73         diagnosticHelper = new DiagnosticTestHelper();
74         compiler = new ErrorProneTestCompiler.Builder()
75             .listenToDiagnostics(diagnosticHelper.collector)
76             .report(ScannerSupplier.fromBugCheckerClasses(MyChecker.class))
77             .build();
78       }
79     
80       @Test
81       public void testPositiveCase() throws Exception {
82         List<JavaFileObject> sources =
83             compiler.fileManager().forResources(getClass(), "UnsuppressiblePositiveCases.java");
84         assertThat(compiler.compile(sources), is(Result.ERROR));
85         assertThat(diagnosticHelper.getDiagnostics().toString(), containsString("[MyChecker]"));
86       }
87     }
