41     /**
42      * Matches the behaviour of javac's divzero xlint warning.
43      *
44      * @author cushon@google.com (Liam Miller-Cushon)
45      */
46     @BugPattern(name = "DivZero", altNames = "divzero",
47         summary = "Division by integer literal zero",
48         explanation = "This code will cause a runtime arithmetic exception if it is executed.",
49         category = JDK, severity = ERROR, maturity = EXPERIMENTAL)
50     public class DivZero extends BugChecker
51         implements BinaryTreeMatcher, CompoundAssignmentTreeMatcher {
52     
53       @Override
54       public Description matchBinary(BinaryTree tree, VisitorState state) {
55         return matchDivZero(tree, tree.getRightOperand(), state);
56       }
57     
58       @Override
59       public Description matchCompoundAssignment(CompoundAssignmentTree tree, VisitorState state) {
60         return matchDivZero(tree, tree.getExpression(), state);
61       }
62     
63       private Description matchDivZero(Tree tree, ExpressionTree operand, VisitorState state) {
64         if (!anyOf(kindIs(Kind.DIVIDE), kindIs(Kind.DIVIDE_ASSIGNMENT)).matches(tree, state)) {
65           return Description.NO_MATCH;
66         }
67     
68         if (!kindIs(Kind.INT_LITERAL).matches(operand, state)) {
69           return Description.NO_MATCH;
70         }
71     
72         LiteralTree rightOperand = (LiteralTree) operand;
73         if (((Integer) rightOperand.getValue()) != 0) {
74           return Description.NO_MATCH;
75         }
76     
77         // Find and replace enclosing Statement.
78         StatementTree enclosingStmt =
79             ASTHelpers.findEnclosingNode(state.getPath(), StatementTree.class);
80         return (enclosingStmt != null)
81             ? describeMatch(tree,
82                 SuggestedFix.replace(enclosingStmt, "throw new ArithmeticException(\"/ by zero\");"))
83             : describeMatch(tree);
84       }
85     }
