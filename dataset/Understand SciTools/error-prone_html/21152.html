17     package com.google.errorprone;
18     
19     import static java.nio.charset.StandardCharsets.UTF_8;
20     
21     import com.google.auto.service.AutoService;
22     import com.google.gson.Gson;
23     
24     import java.io.IOException;
25     import java.io.OutputStreamWriter;
26     import java.io.PrintWriter;
27     import java.util.Set;
28     
29     import javax.annotation.processing.AbstractProcessor;
30     import javax.annotation.processing.ProcessingEnvironment;
31     import javax.annotation.processing.Processor;
32     import javax.annotation.processing.RoundEnvironment;
33     import javax.annotation.processing.SupportedAnnotationTypes;
34     import javax.lang.model.SourceVersion;
35     import javax.lang.model.element.Element;
36     import javax.lang.model.element.TypeElement;
37     import javax.tools.FileObject;
38     import javax.tools.StandardLocation;
39     
40     /**
41      * Annotation processor which visits all classes that have a {@code BugPattern} annotation,
42      * and writes a tab-delimited text file dumping the data found.
43      *
44      * @author eaftan@google.com (Eddie Aftandilian)
45      * @author alexeagle@google.com (Alex Eagle)
46      */
47     @AutoService(Processor.class)
48     @SupportedAnnotationTypes("com.google.errorprone.BugPattern")
49     public class DocGenProcessor extends AbstractProcessor {
50     
51       @Override
52       public SourceVersion getSupportedSourceVersion() {
53         return SourceVersion.latest();
54       }
55     
56       private final Gson gson = new Gson();
57     
58       private PrintWriter pw;
59     
60       /**
61        * {@inheritDoc}
62        */
63       @Override
64       public synchronized void init(ProcessingEnvironment processingEnv) {
65         super.init(processingEnv);
66         try {
67           FileObject manifest = processingEnv.getFiler()
68               .createResource(StandardLocation.SOURCE_OUTPUT, "", "bugPatterns.txt");
69           pw = new PrintWriter(new OutputStreamWriter(manifest.openOutputStream(), UTF_8));
70         } catch (IOException e) {
71           throw new RuntimeException(e);
72         }
73       }
74     
75       /**
76        * {@inheritDoc}
77        */
78       @Override
79       public boolean process(Set<? extends TypeElement> annotations, RoundEnvironment roundEnv) {
80         for (Element element : roundEnv.getElementsAnnotatedWith(BugPattern.class)) {
81           gson.toJson(BugPatternInstance.fromElement(element), pw);
82           pw.write("\n");
83         }
84     
85         if (roundEnv.processingOver()) {
86           // this was the last round, do cleanup
87           cleanup();
88         }
89         return true;
90       }
91     
92       /**
93        * Perform cleanup after last round of annotation processing.
94        */
95       private void cleanup() {
96         pw.close();
97       }
98     }
