17     package com.google.errorprone.refaster;
18     
19     import static com.google.common.base.Preconditions.checkNotNull;
20     
21     import com.google.auto.value.AutoValue;
22     import com.google.common.base.MoreObjects;
23     import com.google.common.base.Objects;
24     
25     import com.sun.tools.javac.code.Type;
26     import com.sun.tools.javac.code.Type.TypeVar;
27     import com.sun.tools.javac.tree.JCTree.JCExpression;
28     
29     import javax.annotation.Nullable;
30     
31     /**
32      * {@link UType} version of {@link TypeVar}.
33      *
34      * @author Louis Wasserman
35      */
36     public class UTypeVar extends UType {
37       // This can't be @AutoValue'd, since the fields are mutable.
38     
39       /**
40        * Bindings key linked to a {@code UTypeVar}.
41        */
42       public static final class Key extends Bindings.Key<TypeWithExpression> {
43         public Key(CharSequence name) {
44           super(name.toString());
45         }
46       }
47       
48       /**
49        * Tuple of an expression with an associated type.
50        */
51       @AutoValue
52       public abstract static class TypeWithExpression implements Inlineable<JCExpression> {
53         public static TypeWithExpression create(Type type, JCExpression expression) {
54           return new AutoValue_UTypeVar_TypeWithExpression(type, checkNotNull(expression));
55         }
56         
57         public static TypeWithExpression create(Type type) {
58           return new AutoValue_UTypeVar_TypeWithExpression(type, null);
59         }
60         
61         public abstract Type type();
62         @Nullable abstract JCExpression expression();
63     
64         @Override
65         public JCExpression inline(Inliner inliner) {
66           return (expression() == null)
67               ? inliner.inlineAsTree(type())
68               : expression(); 
69         }
70     
71         @Override
72         public String toString() {
73           return type().toString();
74         }
75       }
76       
77       public static UTypeVar create(String name, UType lowerBound, UType upperBound) {
78         return new UTypeVar(name, lowerBound, upperBound);
79       }
80     
81       public static UTypeVar create(String name, UType upperBound) {
82         return create(name, UPrimitiveType.NULL, upperBound);
83       }
84     
85       public static UTypeVar create(String name) {
86         return create(name, UClassType.create("java.lang.Object"));
87       }
88     
89       private final String name;
90       private UType lowerBound;
91       private UType upperBound;
92     
93       private UTypeVar(String name, UType lowerBound, UType upperBound) {
94         this.name = checkNotNull(name);
95         this.lowerBound = checkNotNull(lowerBound);
96         this.upperBound = checkNotNull(upperBound);
97       }
98     
99       @Override
100       public Choice<Unifier> visitType(Type target, Unifier unifier) {
101         // This is only called when we're trying to unify overloads, in which case
102         // type variables don't matter.
103         return Choice.condition(!target.isPrimitive(), unifier);
104       }
105       
106       public Key key() {
107         return new Key(name);
108       }
109     
110       public String getName() {
111         return name;
112       }
113     
114       public UType getLowerBound() {
115         return lowerBound;
116       }
117     
118       public UType getUpperBound() {
119         return upperBound;
120       }
121     
122       /**
123        * @param lowerBound the lowerBound to set
124        */
125       public void setLowerBound(UType lowerBound) {
126         this.lowerBound = checkNotNull(lowerBound);
127       }
128     
129       /**
130        * @param upperBound the upperBound to set
131        */
132       public void setUpperBound(UType upperBound) {
133         this.upperBound = checkNotNull(upperBound);
134       }
135     
136       @Override
137       public Type inline(Inliner inliner) throws CouldNotResolveImportException {
138         return inliner.inlineTypeVar(this);
139       }
140       
141       @Override
142       public int hashCode() {
143         return Objects.hashCode(name);
144       }
145     
146       @Override
147       public boolean equals(@Nullable Object obj) {
148         if (this == obj) {
149           return true;
150         } else if (obj instanceof UTypeVar) {
151           UTypeVar typeVar = (UTypeVar) obj;
152           return name.equals(typeVar.name)
153               && lowerBound.equals(typeVar.lowerBound)
154               && upperBound.equals(typeVar.upperBound);
155         }
156         return false;
157       }
158     
159       @Override
160       public String toString() {
161         return MoreObjects.toStringHelper(this)
162             .add("name", name)
163             .toString();
164       }
165     
166     }
