123       /**
124        * Compose several matchers together, such that the composite matches an AST node iff all the given matchers do.
125        */
126       @SafeVarargs
127       public static <T extends Tree> Matcher<T> allOf(final Matcher<? super T>... matchers) {
128         return new Matcher<T>() {
129           @Override public boolean matches(T t, VisitorState state) {
130             for (Matcher<? super T> matcher : matchers) {
131               if (!matcher.matches(t, state)) {
132                 return false;
133               }
134             }
135             return true;
136           }
137         };
138       }

310       /**
311        * Matches when the receiver of an instance method is the same reference as a particular argument to the method.
312        * For example, receiverSameAsArgument(1) would match {@code obj.method("", obj)}
313        *
314        * @param argNum The number of the argument to compare against (zero-based.
315        */
316       public static Matcher<? super MethodInvocationTree> receiverSameAsArgument(final int argNum) {
317         return new Matcher<MethodInvocationTree>() {
318           @Override
319           public boolean matches(MethodInvocationTree t, VisitorState state) {
320             List<? extends ExpressionTree> args = t.getArguments();
321             if (args.size() <= argNum) {
322               return false;
323             }
324             ExpressionTree arg = args.get(argNum);
325     
326             JCExpression methodSelect = (JCExpression) t.getMethodSelect();
327             if (methodSelect instanceof JCFieldAccess) {
328               JCFieldAccess fieldAccess = (JCFieldAccess) methodSelect;
329               return ASTHelpers.sameVariable(fieldAccess.getExpression(), arg);
330             } else if (methodSelect instanceof JCIdent) {
331               // A bare method call: "equals(foo)".  Receiver is implicitly "this".
332               return "this".equals(arg.toString());
333             }
334     
335             return false;
336           }
337         };
338       }

579       /**
580        * Matches an AST node that is enclosed by some node that matches the given matcher.
581        *
582        * TODO(eaftan): This could be used instead of enclosingBlock and enclosingClass.
583        */
584       public static <T extends Tree> Matcher<Tree> enclosingNode(final Matcher<T> matcher) {
585         return new Matcher<Tree>() {
586           @SuppressWarnings("unchecked")  // TODO(cushon): this should take a Class<T>
587           @Override
588           public boolean matches(Tree t, VisitorState state) {
589             TreePath path = state.getPath().getParentPath();
590             while (path != null) {
591               Tree node = path.getLeaf();
592               state = state.withPath(path);
593               if (matcher.matches((T) node, state)) {
594                 return true;
595               }
596               path = path.getParentPath();
597             }
598             return false;
599           }
600     
601         };
602       }

613        * Matches an AST node if it is a literal other than null.
614        */
615       public static Matcher<ExpressionTree> nonNullLiteral() {
616         return new Matcher<ExpressionTree>() {
617           @Override
618           public boolean matches(ExpressionTree tree, VisitorState state) {
619             switch (tree.getKind()) {
620               case MEMBER_SELECT:
621                 return ((MemberSelectTree) tree).getIdentifier().contentEquals("class");
622               case INT_LITERAL:
623               case LONG_LITERAL:
624               case FLOAT_LITERAL:
625               case DOUBLE_LITERAL:
626               case BOOLEAN_LITERAL:
627               case CHAR_LITERAL:
628                 // fall through
629               case STRING_LITERAL:
630                 return true;
631               default:
632                 return false;
633             }
634           }
635         };
636       }

659       /**

666           public boolean matches(ExpressionTree expressionTree, VisitorState state) {

1342        *
1343        * @param conditionMatcher The matcher to apply to the condition in the assertion, e.g. in
