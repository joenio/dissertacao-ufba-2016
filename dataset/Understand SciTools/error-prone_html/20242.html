17     package com.google.errorprone.matchers;
18     
19     import com.google.errorprone.VisitorState;
20     
21     import com.sun.source.tree.NewClassTree;
22     import com.sun.tools.javac.code.Type;
23     import com.sun.tools.javac.tree.JCTree.JCNewClass;
24     
25     import java.util.ArrayList;
26     import java.util.List;
27     
28     /**
29      * Matches a call to a constructor.
30      *
31      * @author eaftan@google.com (Eddie Aftandilian)
32      */
33     public class Constructor implements Matcher<NewClassTree>{
34     
35       private final String className;
36       private final List<String> parameterTypes;
37     
38       public Constructor(String className, List<String> parameterTypes) {
39         this.className = className;
40         this.parameterTypes = parameterTypes;
41       }
42     
43       @Override
44       public boolean matches(NewClassTree newClassTree, VisitorState state) {
45         /* TODO(eaftan): Don't catch NullPointerException.  Need to do this right now
46          * for internal use, but remember to remove later. */
47         try {
48           JCNewClass newClass = (JCNewClass) newClassTree;
49           String thisClassName = newClass.constructor.getEnclosingElement().toString();
50           com.sun.tools.javac.util.List<Type> thisParameterTypes =
51               newClass.constructor.type.getParameterTypes();
52           List<String> thisParameterTypesAsStrings = new ArrayList<>(thisParameterTypes.length());
53           for (Type t : thisParameterTypes) {
54             thisParameterTypesAsStrings.add(t.toString());
55           }
56     
57           return thisClassName.equals(className) && thisParameterTypesAsStrings.equals(parameterTypes);
58         } catch (NullPointerException e) {
59           return false;
60         }
61       }
62     }
