17     package com.google.errorprone.refaster;
18     
19     import com.google.auto.value.AutoValue;
20     import com.google.common.base.Function;
21     import com.google.common.collect.ImmutableList;
22     
23     import com.sun.source.tree.Tree;
24     import com.sun.source.tree.TreeVisitor;
25     import com.sun.tools.javac.tree.JCTree.JCExpression;
26     
27     /**
28      * {@code UExpression} allowing a match against any of a list of expressions.
29      *
30      * @author lowasser@google.com (Louis Wasserman)
31      */
32     @AutoValue
33     public abstract class UAnyOf extends UExpression {
34       public static UAnyOf create(UExpression... expressions) {
35         return create(ImmutableList.copyOf(expressions));
36       }
37     
38       public static UAnyOf create(Iterable<? extends UExpression> expressions) {
39         return new AutoValue_UAnyOf(ImmutableList.copyOf(expressions));
40       }
41     
42       abstract ImmutableList<UExpression> expressions();
43     
44       @Override
45       public UExpression negate() {
46         ImmutableList.Builder<UExpression> negations = ImmutableList.builder();
47         for (UExpression expression : expressions()) {
48           negations.add(expression.negate());
49         }
50         return create(negations.build());
51       }  
52       
53       @Override
54       protected Choice<Unifier> defaultAction(final Tree tree, final Unifier unifier) {
55         return Choice.from(expressions()).thenChoose(new Function<UExpression, Choice<Unifier>>() {
56           @Override
57           public Choice<Unifier> apply(UExpression expression) {
58             return expression.unify(UParens.skipParens(tree), unifier.fork());
59           }
60         });
61       }
62       
63       @Override
64       public JCExpression inline(Inliner inliner) throws CouldNotResolveImportException {
65         throw new UnsupportedOperationException("anyOf should not appear in an @AfterTemplate");
66       }
67     
68       @Override
69       public <R, D> R accept(TreeVisitor<R, D> visitor, D data) {
70         return expressions().get(0).accept(visitor, data);
71       }
72     
73       @Override
74       public Kind getKind() {
75         return Kind.OTHER;
76       }
77     }
