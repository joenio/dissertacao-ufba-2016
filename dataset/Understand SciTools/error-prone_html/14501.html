44     import com.sun.tools.javac.tree.JCTree.JCFieldAccess;

46     /**
47      * @author eaftan@google.com (Eddie Aftandilian)
48      */
49     @BugPattern(name = "ArrayHashCode",
50         summary = "hashcode method on array does not hash array contents",
51         category = JDK, severity = ERROR, maturity = MATURE)
52     public class ArrayHashCode extends BugChecker implements MethodInvocationTreeMatcher {
53     
54       /**
55        * Matches calls to varargs hashcode methods {@link com.google.common.base.Objects#hashCode} and
56        * {@link java.util.Objects#hash} in which at least one argument is of array type.
57        */
58       private static final Matcher<MethodInvocationTree> varargsHashCodeMethodMatcher = allOf(
59           Matchers.anyOf(
60             staticMethod().onClass("com.google.common.base.Objects").named("hashCode"),
61             staticMethod().onClass("java.util.Objects").named("hash")),
62           hasArguments(MatchType.AT_LEAST_ONE, Matchers.<ExpressionTree>isArrayType()));
63     
64       /**
65        * Matches calls to the JDK7 method {@link java.util.Objects#hashCode} with an argument of array
66        * type. This method is not varargs, so we don't need to check the number of arguments.
67        */
68       @SuppressWarnings({"unchecked"})
69       private static final Matcher<MethodInvocationTree> jdk7HashCodeMethodMatcher = allOf(
70           staticMethod().onClass("java.util.Objects").named("hashCode"),
71           argument(0, Matchers.<ExpressionTree>isArrayType()));
72     
73       /**
74        * Matches calls to the hashCode instance method on an array.
75        */
76       private static final Matcher<ExpressionTree> instanceHashCodeMethodMatcher =
77           instanceMethod().onClass(isArray()).named("hashCode");
78     
79       /**
80        * Wraps identity hashcode computations in calls to {@link java.util.Arrays#hashCode} if the
81        * array is single dimensional or {@link java.util.Arrays#deepHashCode} if the array is
82        * multidimensional.
83        *
84        * <p>If there is only one argument to the hashcode method or the instance hashcode method is
85        * used, replaces the whole method invocation.  If there are multiple arguments, wraps any that
86        * are of array type with the appropriate {@link java.util.Arrays} hashcode method.
87        */
88       @Override
89       public Description matchMethodInvocation(MethodInvocationTree tree, VisitorState state) {
90         SuggestedFix.Builder fix = null;
91         Types types = state.getTypes();
92         if (jdk7HashCodeMethodMatcher.matches(tree, state)) {
93           // java.util.Objects#hashCode takes a single argument, so rewrite the whole method call
94           // to use Arrays.hashCode/deepHashCode instead.
95           fix = SuggestedFix.builder()
96               .replace(tree, rewriteArrayArgument(tree.getArguments().get(0), types));
97         } else if (instanceHashCodeMethodMatcher.matches(tree, state)) {
98           // Rewrite call to instance hashCode method to use Arrays.hashCode/deepHashCode instead.
99           fix = SuggestedFix.builder()
100               .replace(tree,
101                   rewriteArrayArgument(
102                       ((JCFieldAccess) tree.getMethodSelect()).getExpression(),
103                       types));
104         } else if (varargsHashCodeMethodMatcher.matches(tree, state)) {
105           // Varargs hash code methods, java.util.Objects#hash and
106           // com.google.common.base.Objects#hashCode
107           if (tree.getArguments().size() == 1) {
108             // If only one argument, type must be either primitive array or multidimensional array.
109             // Types like Object[], String[], etc. are not an error because they don't get boxed
110             // in this single-argument varargs call.
111             ExpressionTree arg = tree.getArguments().get(0);
112             Type elemType = types.elemtype(ASTHelpers.getType(arg));
113             if (elemType.isPrimitive() || types.isArray(elemType)) {
114               fix = SuggestedFix.builder()
115                   .replace(tree, rewriteArrayArgument(arg, types));
116             }
117           } else {
118             // If more than one argument, wrap each argument in a call to Arrays#hashCode/deepHashCode.
119             fix = SuggestedFix.builder();
120             for (ExpressionTree arg : tree.getArguments()) {
121               if (types.isArray(ASTHelpers.getType(arg))) {
122                 fix.replace(arg, rewriteArrayArgument(arg, types));
123               }
124             }
125           }
126         }
127     
128         if (fix != null) {
129           fix.addImport("java.util.Arrays");
130           return describeMatch(tree, fix.build());
131         }
132     
133         return Description.NO_MATCH;
134       }
135     
136       /**
137        * Given an {@link ExpressionTree} that represents an argument of array type, rewrites it to wrap
138        * it in a call to either {@link java.util.Arrays#hashCode} if it is single dimensional, or
139        * {@link java.util.Arrays#deepHashCode} if it is multidimensional.
140        */
141       private static String rewriteArrayArgument(ExpressionTree arg, Types types) {
142         Type argType = ASTHelpers.getType(arg);
143         Preconditions.checkState(types.isArray(argType), "arg must be of array type");
144         if (types.isArray(types.elemtype(argType))) {
145           return "Arrays.deepHashCode(" + arg + ")";
146         } else {
147           return "Arrays.hashCode(" + arg + ")";
148         }
149       }
150     }
