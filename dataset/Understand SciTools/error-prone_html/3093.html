53       @Retention(RUNTIME)
54       public @interface EditModifiers {
55         String value() default "";
56     
57         EditKind kind() default EditKind.ADD;
58     
59         enum EditKind {
60           ADD,
61           REMOVE
62         }
63       }
64     
65       @BugPattern(
66         name = "EditModifiers",
67         category = Category.ONE_OFF,
68         summary = "Edits modifiers",
69         severity = SeverityLevel.ERROR,
70         maturity = MaturityLevel.EXPERIMENTAL
71       )
72       public static class EditModifiersChecker extends BugChecker implements VariableTreeMatcher {
73     
74         static final ImmutableMap<String, Modifier> MODIFIERS_BY_NAME = createModifiersByName();
75     
76         private static ImmutableMap<String, Modifier> createModifiersByName() {
77           ImmutableMap.Builder<String, Modifier> builder = ImmutableMap.builder();
78           for (Modifier mod : Modifier.values()) {
79             builder.put(mod.toString(), mod);
80           }
81           return builder.build();
82         }
83     
84         @Override
85         public Description matchVariable(VariableTree tree, VisitorState state) {
86           EditModifiers editModifiers =
87               ASTHelpers.getAnnotation(
88                   ASTHelpers.findEnclosingNode(state.getPath(), ClassTree.class), EditModifiers.class);
89           Modifier mod = MODIFIERS_BY_NAME.get(editModifiers.value());
90           Verify.verifyNotNull(mod, editModifiers.value());
91           Fix fix;
92           switch (editModifiers.kind()) {
93             case ADD:
94               fix = SuggestedFix.addModifier(tree, mod, state);
95               break;
96             case REMOVE:
97               fix = SuggestedFix.removeModifier(tree, mod, state);
98               break;
99             default:
100               throw new AssertionError(editModifiers.kind());
101           }
102           return describeMatch(tree, fix);
103         }
104       }

106       @Test
107       public void addModifiers() {
108         CompilationTestHelper.newInstance(EditModifiersChecker.class, getClass())
109             .addSourceLines(
110                 "Test.java",

112                 "import javax.annotation.Nullable;",

163       @BugPattern(
164         category = Category.ONE_OFF,
165         maturity = MaturityLevel.EXPERIMENTAL,
166         name = "CastReturn",
167         severity = SeverityLevel.ERROR,
168         summary = "Adds casts to returned expressions"
169       )
170       public static class CastReturn extends BugChecker implements ReturnTreeMatcher {
171     
172         @Override
173         public Description matchReturn(ReturnTree tree, VisitorState state) {
174           if (tree.getExpression() == null) {
175             return Description.NO_MATCH;
176           }
177           Type type =
178               ASTHelpers.getSymbol(
179                   ASTHelpers.findEnclosingNode(state.getPath(), MethodTree.class)).getReturnType();
180           SuggestedFix.Builder fixBuilder = SuggestedFix.builder();
181           String qualifiedTargetType = SuggestedFix.qualifyType(state, fixBuilder, type.tsym);
182           fixBuilder.prefixWith(tree.getExpression(), String.format("(%s) ", qualifiedTargetType));
183           return describeMatch(tree, fixBuilder.build());
184         }
185       }
