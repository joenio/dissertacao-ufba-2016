19     import static com.google.errorprone.BugPattern.Category.GUAVA;
20     import static com.google.errorprone.BugPattern.MaturityLevel.MATURE;
21     import static com.google.errorprone.BugPattern.SeverityLevel.ERROR;
22     import static org.junit.Assert.fail;
23     
24     import com.google.errorprone.BugPattern;
25     import com.google.errorprone.CompilationTestHelper;
26     
27     import org.junit.Before;
28     import org.junit.Test;
29     import org.junit.runner.RunWith;
30     import org.junit.runners.JUnit4;
31     
32     /**
33      * @author eaftan@google.com (Eddie Aftandilian)
34      */
35     @RunWith(JUnit4.class)
36     public class SelfEqualsTest {
37       CompilationTestHelper compilationHelper;
38     
39       @BugPattern(
40           name = "SelfEquals", summary = "An object is tested for equality to itself", category = GUAVA,
41           severity = ERROR, maturity = MATURE)
42       public static class SelfEqualsTestChecker extends SelfEquals {
43         public SelfEqualsTestChecker() {
44           super(true, true);
45         }
46       }
47     
48       @Before
49       public void setUp() {
50         compilationHelper = CompilationTestHelper.newInstance(SelfEqualsTestChecker.class, getClass());
51       }
52     
53       @Test
54       public void testPositiveCase1() throws Exception {
55         compilationHelper.addSourceFile("SelfEqualsPositiveCase1.java").doTest();
56       }
57     
58       @Test
59       public void testPositiveCase2() throws Exception {
60         compilationHelper.addSourceFile("SelfEqualsPositiveCase2.java").doTest();
61       }
62     
63       @Test
64       public void testNegativeCase() throws Exception {
65         compilationHelper.addSourceFile("SelfEqualsNegativeCases.java").doTest();
66       }
67     
68       @BugPattern(
69           name = "SelfEquals", summary = "An object is tested for equality to itself", category = GUAVA,
70           severity = ERROR, maturity = MATURE)
71       public static class SelfEquals_Guava_Equals extends SelfEquals {
72         public SelfEquals_Guava_Equals() {
73           super(true, true);
74         }
75       }
76     
77       @BugPattern(
78           name = "SelfEquals", summary = "An object is tested for equality to itself", category = GUAVA,
79           severity = ERROR, maturity = MATURE)
80       public static class SelfEquals_Guava extends SelfEquals {
81         public SelfEquals_Guava() {
82           super(true, false);
83         }
84       }
85     
86       @BugPattern(
87           name = "SelfEquals", summary = "An object is tested for equality to itself", category = GUAVA,
88           severity = ERROR, maturity = MATURE)
89       public static class SelfEquals_Equals extends SelfEquals {
90         public SelfEquals_Equals() {
91           super(false, true);
92         }
93       }
94     
95       @Test
96       public void testFlags() throws Exception {
97         Class<? extends SelfEquals> checker;
98         CompilationTestHelper compilationHelper;
99         // Both checks off.
100         try {
101           new SelfEquals(false, false);
102           fail();
103         } catch (IllegalArgumentException e) {
104           // Expected to get an exception.
105         }
106     
107         // Both checks on.
108         checker = SelfEquals_Guava_Equals.class;
109         compilationHelper = CompilationTestHelper.newInstance(checker, getClass());
110         compilationHelper.addSourceFile("SelfEqualsPositiveCase1.java").doTest();
111         compilationHelper = CompilationTestHelper.newInstance(checker, getClass());
112         compilationHelper.addSourceFile("SelfEqualsPositiveCase2.java").doTest();
113     
114         // Guava on, Equals off.
115         checker = SelfEquals_Guava.class;
116         compilationHelper = CompilationTestHelper.newInstance(checker, getClass());
117         compilationHelper.addSourceFile("SelfEqualsPositiveCase1.java").doTest();
118         compilationHelper = CompilationTestHelper.newInstance(checker, getClass());
119         compilationHelper.addSourceFile("SelfEqualsPositiveCase2.java")
120             .expectNoDiagnostics()
121             .doTest();
122     
123         // Equals on, Guava off.
124         checker = SelfEquals_Equals.class;
125         compilationHelper = CompilationTestHelper.newInstance(checker, getClass());
126         compilationHelper.addSourceFile("SelfEqualsPositiveCase1.java")
127             .expectNoDiagnostics()
128             .doTest();
129         compilationHelper = CompilationTestHelper.newInstance(checker, getClass());
130         compilationHelper.addSourceFile("SelfEqualsPositiveCase2.java").doTest();
131       }
