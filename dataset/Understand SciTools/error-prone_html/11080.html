45     /**
46      * @author bill.pugh@gmail.com (Bill Pugh)
47      * @author eaftan@google.com (Eddie Aftandilian)
48      *
49      * TODO(eaftan): Support other types of comparisons?  Are there likely to be errors in those?
50      */
51     @BugPattern(name = "ComparisonOutOfRange",
52         summary = "Comparison to value that is out of range for the compared type",
53         explanation = "This checker looks for equality comparisons to values that are out of " +
54             "range for the compared type.  For example, bytes may have a value in the range " +
55             Byte.MIN_VALUE + " to " + Byte.MAX_VALUE + ". Comparing a byte for equality with a value " +
56             "outside that range will always evaluate to false and usually indicates an error in the " +
57             "code.\n\n" +
58             "This checker currently supports checking for bad byte and character comparisons.",
59         category = JDK, severity = ERROR, maturity = MATURE)
60     public class ComparisonOutOfRange extends BugChecker implements BinaryTreeMatcher {
61     
62       private static final String MESSAGE_TEMPLATE = "%ss may have a value in the range %d to %d; "
63           + "therefore, this comparison to %s will always evaluate to %s";
64     
65       /**
66        * Matches comparisons that are out of range for the given type.  Parameterized based on the
67        * type of comparison (byte or char).
68        */
69       private static class BadComparisonMatcher implements Matcher<BinaryTree> {
70         /**
71          * The type of bad comparison matcher to create. Must be either Byte.TYPE or Character.TYPE.
72          */
73         private final Class<?> type;
74     
75         private boolean initialized = false;
76         private Type comparisonType;
77         private int maxValue;
78         private int minValue;
79     
80         public BadComparisonMatcher(Class<?> type) {
81           if (type != Byte.TYPE && type != Character.TYPE) {
82             throw new IllegalArgumentException("type must be either byte or char, but was "
83                 + type.getName());
84           }
85           this.type = type;
86         }
87     
88         /**
89          * Initialize matcher for the specific type.  We can't do this in the constructor because
90          * we need the symbol table, which isn't available at that time.
91          *
92          * @param symbolTable The compiler's symbol table
93          */
94         private void init(Symtab symbolTable) {
95           if (initialized) {
96             throw new IllegalStateException("Do not try to initialize twice!");
97           }
98     
99           // Specialize matcher based on type.
100           if (type == Byte.TYPE) {
101             comparisonType = symbolTable.byteType;
102             maxValue = Byte.MAX_VALUE;
103             minValue = Byte.MIN_VALUE;
104           } else {
105             comparisonType = symbolTable.charType;
106             maxValue = Character.MAX_VALUE;
107             minValue = Character.MIN_VALUE;
108           }
109           initialized = true;
110         }
111     
112         @Override
113         public boolean matches(BinaryTree tree, VisitorState state) {
114           if (!initialized) {
115             init(state.getSymtab());
116           }
117     
118           // Must be an == or != comparison.
119           if (tree.getKind() != Kind.EQUAL_TO && tree.getKind() != Kind.NOT_EQUAL_TO) {
120             return false;
121           }
122     
123           // Match trees that have one literal operand and another of the specified type.
124           List<ExpressionTree> binaryTreeMatches = ASTHelpers.matchBinaryTree(tree,
125               Arrays.asList(Matchers.<ExpressionTree>isInstance(JCLiteral.class),
126                   Matchers.<ExpressionTree>isSameType(comparisonType)),
127               state);
128           if (binaryTreeMatches == null) {
129             return false;
130           }
131           JCLiteral literal = (JCLiteral) binaryTreeMatches.get(0);
132     
133           // Check whether literal is out of range for the specified type.  Logic is based on
134           // JLS 5.6.2 - Binary Numeric Promotion:
135           // If either is double, other is converted to double.
136           // If either is float, other is converted to float.
137           // If either is long, other is converted to long.
138           // Otherwise, both are converted to int.
139           Object literalValue = literal.getValue();
140           switch (literal.getKind()) {
141             case DOUBLE_LITERAL:
142               double doubleValue = ((Double) literalValue).doubleValue();
143               return doubleValue < minValue || doubleValue > maxValue;
144             case FLOAT_LITERAL:
145               float floatValue = ((Float) literalValue).floatValue();
146               return floatValue < minValue || floatValue > maxValue;
147             case LONG_LITERAL:
148               long longValue = ((Long) literalValue).longValue();
149               return longValue < minValue || longValue > maxValue;
150             default:
151               // JCLiteral.getValue() can return Integer, Character, or Boolean.
152               int intValue;
153               if (literalValue instanceof Integer) {
154                 intValue = ((Integer) literalValue).intValue();
155               } else if (literalValue instanceof Character) {
156                 intValue = ((Character) literalValue).charValue();
157               } else if (literalValue instanceof Boolean) {
158                 throw new IllegalStateException("Cannot compare " + comparisonType
159                     + " to boolean literal");
160               } else {
161                 throw new IllegalStateException("Unexpected literal type: " + literal);
162               }
163               return intValue < minValue || intValue > maxValue;
164           }
165         }
166       }
167     
168       private static final Matcher<BinaryTree> BYTE_MATCHER = new BadComparisonMatcher(Byte.TYPE);
169       private static final Matcher<BinaryTree> CHAR_MATCHER = new BadComparisonMatcher(Character.TYPE);
170     
171       @Override
172       public Description matchBinary(BinaryTree tree, VisitorState state) {
173         if(anyOf(BYTE_MATCHER, CHAR_MATCHER).matches(tree, state)) {
174           return describe(tree, state);
175         }
176         return Description.NO_MATCH;
177       }
178     
179       /**
180        * Suggested fixes are as follows.  For the byte case, convert the literal to its byte
181        * representation. For example, "255" becomes "-1.  For the character case, replace the
182        * comparison with "true"/"false" since it's not clear what was intended and that is
183        * semantically equivalent.
184        *
185        * TODO(eaftan): Suggested fixes don't handle side-effecting expressions, such as
186        * (d = reader.read()) == -1.  Maybe add special case handling for assignments.
187        */
188       public Description describe(BinaryTree tree, VisitorState state) {
189         List<ExpressionTree> binaryTreeMatches = ASTHelpers.matchBinaryTree(tree,
190             Arrays.asList(Matchers.<ExpressionTree>isInstance(JCLiteral.class),
191                 Matchers.<ExpressionTree>anything()),
192             state);
193         if (binaryTreeMatches == null) {
194           throw new IllegalStateException("Expected one of the operands to be a literal");
195         }
196         JCLiteral literal = (JCLiteral) binaryTreeMatches.get(0);
197         JCTree nonLiteralOperand = (JCTree) binaryTreeMatches.get(1);
198         boolean byteMatch = state.getTypes().isSameType(nonLiteralOperand.type,
199             state.getSymtab().byteType);
200     
201         boolean willEvaluateTo = (tree.getKind() != Kind.EQUAL_TO);
202         Fix fix;
203         String customDiagnosticMessage;
204         if (byteMatch) {
205           String replacement = Byte.toString(((Number) literal.getValue()).byteValue());
206     
207           // Correct for poor javac 6 literal parsing.
208           int actualStart = ASTHelpers.getActualStartPosition(literal, state.getSourceCode());
209           if (actualStart != literal.getStartPosition()) {
210             fix =
211                 SuggestedFix.replace(literal, replacement, actualStart - literal.getStartPosition(), 0);
212           } else {
213             fix = SuggestedFix.replace(literal, replacement);
214           }
215           customDiagnosticMessage = String.format(MESSAGE_TEMPLATE, "byte", (int) Byte.MIN_VALUE,
216               (int) Byte.MAX_VALUE, literal.toString(), Boolean.toString(willEvaluateTo));
217         } else {
218           fix = SuggestedFix.replace(tree, Boolean.toString(willEvaluateTo));
219           customDiagnosticMessage = String.format(MESSAGE_TEMPLATE, "char", (int) Character.MIN_VALUE,
220               (int) Character.MAX_VALUE, literal.toString(), Boolean.toString(willEvaluateTo));
221         }
222         return buildDescription(tree)
223             .addFix(fix)
224             .setMessage(customDiagnosticMessage)
225             .build();
226       }
