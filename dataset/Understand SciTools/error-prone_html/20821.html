17     package com.google.errorprone.bugpatterns.testdata;
18     
19     import java.util.Date;
20     import java.util.concurrent.TimeUnit;
21     import java.util.concurrent.locks.Condition;
22     
23     /**
24      * @author eaftan@google.com (Eddie Aftandilian)
25      */
26     public class WaitNotInLoopPositiveCases {
27     
28       boolean flag = false;
29     
30       public void testIfInsteadOfLoop() {
31         synchronized (this) {
32           if (!flag) {
33             try {
34               // BUG: Diagnostic contains: wait() must always be called in a loop
35               // Did you mean 'while (!flag) {'?
36               wait();
37             } catch (InterruptedException e) {
38             }
39           }
40         }
41       }
42     
43       public void testWaitLong() throws InterruptedException {
44         // BUG: Diagnostic contains: wait(long) must always be called in a loop
45         wait(1000);
46       }
47     
48       public void testWaitLongInt() throws Exception {
49         // BUG: Diagnostic contains: wait(long,int) must always be called in a loop
50         wait(1000, 1000);
51       }
52     
53       public void testAwait(Condition cond) throws Exception {
54         // BUG: Diagnostic contains: await() must always be called in a loop
55         cond.await();
56       }
57     
58       public void testAwaitWithFix(Condition cond) throws Exception {
59         synchronized (this) {
60           if (!flag) {
61             try {
62               // BUG: Diagnostic contains: await() must always be called in a loop
63               // Did you mean 'while (!flag) {'?
64               cond.await();
65             } catch (InterruptedException e) {
66             }
67           }
68         }
69       }
70     
71       public void testAwaitLongTimeUnit(Condition cond) throws Exception {
72         // BUG: Diagnostic contains:
73         // await(long,java.util.concurrent.TimeUnit) must always be called in a loop
74         cond.await(1, TimeUnit.SECONDS);
75       }
76     
77       public void testAwaitNanos(Condition cond) throws Exception {
78         // BUG: Diagnostic contains: awaitNanos(long) must always be called in a loop
79         cond.awaitNanos(1000000);
80       }
81     
82       public void testAwaitUninterruptibly(Condition cond) throws Exception {
83         // BUG: Diagnostic contains: awaitUninterruptibly() must always be called in a loop
84         cond.awaitUninterruptibly();
85       }
86     
87       public void testAwaitUntil(Condition cond) throws Exception {
88         // BUG: Diagnostic contains: awaitUntil(java.util.Date) must always be called in a loop
89         cond.awaitUntil(new Date());
90       }
91     }
