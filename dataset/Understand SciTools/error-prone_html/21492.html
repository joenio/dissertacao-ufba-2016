47      * Template representing a sequence of consecutive statements.

82       abstract ImmutableList<UStatement> templateStatements();
83     
84       /**
85        * If the tree is a {@link JCBlock}, returns a list of disjoint matches corresponding to the exact
86        * list of template statements found consecutively; otherwise, returns an empty list.
87        */
88       @Override
89       public Iterable<BlockTemplateMatch> match(JCTree tree, Context context) {
90         // TODO(lowasser): consider nonconsecutive matches?
91         if (tree instanceof JCBlock) {
92           JCBlock block = (JCBlock) tree;
93           ImmutableList<JCStatement> targetStatements = ImmutableList.copyOf(block.getStatements());
94           return matchesStartingAnywhere(block, 0, targetStatements, context).first()
95               .or(List.<BlockTemplateMatch>nil());
96         }
97         return ImmutableList.of();
98       }
99     
100       private Choice<List<BlockTemplateMatch>> matchesStartingAtBeginning(
101           final JCBlock block,
102           final int offset, 
103           final ImmutableList<? extends StatementTree> statements, 
104           final Context context) {
105         if (statements.isEmpty()) {
106           return Choice.none();
107         }
108         final JCStatement firstStatement = (JCStatement) statements.get(0);
109         Choice<UnifierWithUnconsumedStatements> choice =
110             Choice.of(UnifierWithUnconsumedStatements.create(new Unifier(context), statements));
111         for (UStatement templateStatement : templateStatements()) {
112           choice = choice.thenChoose(templateStatement);
113         }
114         return choice.thenChoose(
115             new Function<UnifierWithUnconsumedStatements, Choice<List<BlockTemplateMatch>>>() {
116               @Override
117               public Choice<List<BlockTemplateMatch>> apply(UnifierWithUnconsumedStatements state) {
118                 Unifier unifier = state.unifier();
119                 Inliner inliner = unifier.createInliner();
120                 try {
121                   Optional<Unifier> checkedUnifier = typecheck(unifier, inliner,
122                       new Warner(firstStatement), expectedTypes(inliner), actualTypes(inliner));
123                   if (checkedUnifier.isPresent()) {
124                     int consumedStatements = statements.size() - state.unconsumedStatements().size();
125                     BlockTemplateMatch match = new BlockTemplateMatch(block, checkedUnifier.get(),
126                         offset, offset + consumedStatements);
127                     return matchesStartingAnywhere(block, offset + consumedStatements,
128                         statements.subList(consumedStatements, statements.size()), context).transform(
129                         prepend(match));
130                   }
131                 } catch (CouldNotResolveImportException e) {
132                   // fall through
133                 }
134                 return Choice.none();
135               }
136             });
137       }
138     
139       private static <T> Function<List<T>, List<T>> prepend(final T t) {
140         return new Function<List<T>, List<T>>() {
141           @Override
142           public List<T> apply(List<T> list) {
143             return list.prepend(t);
144           }
145         };
146       }
147     
148       private Choice<List<BlockTemplateMatch>> matchesStartingAnywhere(
149           JCBlock block, 
150           int offset,
151           final ImmutableList<? extends StatementTree> statements, 
152           final Context context) {
153         Choice<List<BlockTemplateMatch>> choice = Choice.none();
154         for (int i = 0; i < statements.size(); i++) {
155           choice = choice.or(matchesStartingAtBeginning(block, offset + i,
156               statements.subList(i, statements.size()), context));
157         }
158         return choice.or(Choice.of(List.<BlockTemplateMatch>nil()));
159       }

161       /**
162        * Returns a {@code String} representation of a statement, including semicolon.
163        */
164       private static String printStatement(Context context, JCStatement statement) {
165         StringWriter writer = new StringWriter();
166         try {
167           pretty(context, writer).printStat(statement);
168         } catch (IOException e) {
169           throw new AssertionError("StringWriter cannot throw IOExceptions");
170         }
171         return writer.toString();
172       }
173     
174       /**
175        * Returns a {@code String} representation of a sequence of statements, with semicolons and
176        * newlines.
177        */
178       private static String printStatements(
179           Context context, Iterable<JCStatement> statements) {
180         StringWriter writer = new StringWriter();
181         try {
182           pretty(context, writer).printStats(com.sun.tools.javac.util.List.from(statements));
183         } catch (IOException e) {
184           throw new AssertionError("StringWriter cannot throw IOExceptions");
185         }
186         return writer.toString();
187       }

189       @Override
190       public Fix replace(BlockTemplateMatch match) {
191         checkNotNull(match);
192         SuggestedFix.Builder fix = SuggestedFix.builder();
193         Inliner inliner = match.createInliner();
194         Context context = inliner.getContext();
195         if (annotations().containsKey(UseImportPolicy.class)) {
196           ImportPolicy.bind(context, annotations().getInstance(UseImportPolicy.class).value());
197         } else {
198           ImportPolicy.bind(context, ImportPolicy.IMPORT_TOP_LEVEL);
199         }
200         ImmutableList<JCStatement> targetStatements = match.getStatements();
201         try {
202           ImmutableList.Builder<JCStatement> inlinedStatementsBuilder = ImmutableList.builder();
203           for (UStatement statement : templateStatements()) {
204             inlinedStatementsBuilder.addAll(statement.inlineStatements(inliner));
205           }
206           ImmutableList<JCStatement> inlinedStatements = inlinedStatementsBuilder.build();
207           int nInlined = inlinedStatements.size();
208           int nTargets = targetStatements.size();
209           if (nInlined <= nTargets) {
210             for (int i = 0; i < nInlined; i++) {
211               fix.replace(targetStatements.get(i),
212                   printStatement(context, inlinedStatements.get(i)));
213             }
214             for (int i = nInlined; i < nTargets; i++) {
215               fix.delete(targetStatements.get(i));
216             }
217           } else {
218             for (int i = 0; i < nTargets - 1; i++) {
219               fix.replace(targetStatements.get(i),
220                   printStatement(context, inlinedStatements.get(i)));
221             }
222             int last = nTargets - 1;
223             ImmutableList<JCStatement> remainingInlined =
224                 inlinedStatements.subList(last, nInlined);
225             fix.replace(targetStatements.get(last), CharMatcher.is('\n').trimTrailingFrom(
226                 printStatements(context, remainingInlined)));
227           }
228         } catch (CouldNotResolveImportException e) {
229           logger.log(SEVERE, "Failure to resolve import in replacement", e);
230         }
231         return addImports(inliner, fix);
232       }
