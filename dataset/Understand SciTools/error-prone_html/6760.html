78     /**
79      * Abstract superclass for templates that can be used to search and replace in a Java syntax tree.
80      * 
81      * @author lowasser@google.com (Louis Wasserman)
82      * @param <M> Type of a match for this template.
83      */
84     public abstract class Template<M extends TemplateMatch> implements Serializable {

96       Iterable<UTypeVar> typeVariables(Context context) {
97         ImmutableList<UTypeVar> ruleTypeVars = context.get(RefasterRule.RULE_TYPE_VARS);
98         return Iterables.concat(
99             (ruleTypeVars == null) ? ImmutableList.<UTypeVar>of() : ruleTypeVars,
100             templateTypeVariables());

107       /**
108        * Returns a list of the expected types to be matched.  This consists of the argument types from
109        * the @BeforeTemplate method, concatenated with the return types of expression placeholders, 
110        * sorted by the name of the placeholder method.
111        *
112        * @throws CouldNotResolveImportException if a referenced type could not be resolved
113        */
114       protected List<Type> expectedTypes(Inliner inliner) throws CouldNotResolveImportException {
115         ArrayList<Type> result = new ArrayList<>();
116         ImmutableList<UType> types = expressionArgumentTypes().values().asList();
117         ImmutableList<String> argNames = expressionArgumentTypes().keySet().asList();
118         for (int i = 0; i < argNames.size(); i++) {
119           String argName = argNames.get(i);
120           Optional<JCExpression> singleBinding =
121               inliner.getOptionalBinding(new UFreeIdent.Key(argName));
122           if (!singleBinding.isPresent()) {
123             Optional<java.util.List<JCExpression>> exprs =
124                 inliner.getOptionalBinding(new URepeated.Key(argName));
125             if (!exprs.isPresent() || exprs.get().isEmpty()) {
126               // It is a repeated template variable and matches no expressions.
127               continue;
128             }
129           }
130           result.add(types.get(i).inline(inliner));
131         }
132         for (PlaceholderExpressionKey key :
133             Ordering.natural().immutableSortedCopy(
134                 Iterables.filter(inliner.bindings.keySet(), PlaceholderExpressionKey.class))) {
135           result.add(key.method.returnType().inline(inliner));
136         }
137         return List.from(result);
138       }

140       /**
141        * Returns a list of the actual types to be matched.  This consists of the types of the 
142        * expressions bound to the @BeforeTemplate method parameters, concatenated with the types
143        * of the expressions bound to expression placeholders, sorted by the name of the placeholder 
144        * method.
145        */
146       protected List<Type> actualTypes(Inliner inliner) {
147         ArrayList<Type> result = new ArrayList<>();
148         ImmutableList<String> argNames = expressionArgumentTypes().keySet().asList();
149         for (int i = 0; i < expressionArgumentTypes().size(); i++) {
150           String argName = argNames.get(i);
151           Optional<JCExpression> singleBinding =
152               inliner.getOptionalBinding(new UFreeIdent.Key(argName));
153           if (singleBinding.isPresent()) {
154             result.add(singleBinding.get().type);
155           } else {
156             Optional<java.util.List<JCExpression>> exprs =
157                 inliner.getOptionalBinding(new URepeated.Key(argName));
158             if (exprs.isPresent() && !exprs.get().isEmpty()) {
159               Type[] exprTys = new Type[exprs.get().size()];
160               for (int j = 0; j < exprs.get().size(); j++) {
161                 exprTys[j] = exprs.get().get(j).type;
162               }
163               // Get the least upper bound of the types of all expressions that the argument matches.
164               // In the special case where exprs is empty, returns the "bottom" type, which is a
165               // subtype of everything.
166               result.add(inliner.types().lub(List.from(exprTys)));
167             }
168           }
169         }
170         for (PlaceholderExpressionKey key :
171             Ordering.natural().immutableSortedCopy(
172                 Iterables.filter(inliner.bindings.keySet(), PlaceholderExpressionKey.class))) {
173           result.add(inliner.getBinding(key).type);
174         }
175         return List.from(result);
176       }

178       @Nullable
179       protected Optional<Unifier> typecheck(Unifier unifier, Inliner inliner, Warner warner,
180           List<Type> expectedTypes, List<Type> actualTypes) {
181         try {
182           ImmutableList<UTypeVar> freeTypeVars = freeTypeVars(unifier);
183           infer(warner,
184               inliner,
185               inliner.<Type>inlineList(freeTypeVars),
186               expectedTypes,
187               inliner.symtab().voidType,
188               actualTypes);
189     
190           for (UTypeVar var : freeTypeVars) {
191             Type instantiationForVar = infer(warner,
192                 inliner,
193                 inliner.<Type>inlineList(freeTypeVars),
194                 expectedTypes,
195                 var.inline(inliner),
196                 actualTypes);
197             unifier.putBinding(var.key(), 
198                 TypeWithExpression.create(instantiationForVar.getReturnType()));
199           }
200           
201           if (!checkBounds(unifier, inliner, warner)) {
202             return Optional.absent();
203           }
204           return Optional.of(unifier);
205         } catch (CouldNotResolveImportException e) {
206           logger.log(FINE, "Failure to resolve an import", e);
207           return Optional.absent();
208         } catch (InferException e) {
209           logger.log(FINE, "No valid instantiation found: " + e.getMessage());
210           return Optional.absent();
211         }
212       }

214       private boolean checkBounds(Unifier unifier, Inliner inliner, Warner warner)
215           throws CouldNotResolveImportException {
216         Types types = unifier.types();
217         ListBuffer<Type> varsBuffer = new ListBuffer<>();
218         ListBuffer<Type> bindingsBuffer = new ListBuffer<>();
219         for (UTypeVar typeVar : typeVariables(unifier.getContext())) {
220           varsBuffer.add(inliner.inlineAsVar(typeVar));
221           bindingsBuffer.add(unifier.getBinding(typeVar.key()).type());
222         }
223         List<Type> vars = varsBuffer.toList();
224         List<Type> bindings = bindingsBuffer.toList();
225         for (UTypeVar typeVar : typeVariables(unifier.getContext())) {
226           List<Type> bounds = types.getBounds(inliner.inlineAsVar(typeVar));
227           bounds = types.subst(bounds, vars, bindings);
228           if (!types.isSubtypeUnchecked(unifier.getBinding(typeVar.key()).type(), bounds, warner)) {
229             logger.log(FINE,
230                 String.format("%s is not a subtype of %s", inliner.getBinding(typeVar.key()), bounds));
231             return false;
232           }
233         }
234         return true;
235       }

247             @Override
248             public void visitAnnotation(JCAnnotation anno) {
249               if (anno.getArguments().isEmpty()) {
250                 try {
251                   print("@");
252                   printExpr(anno.annotationType);
253                 } catch (IOException e) {
254                   // the supertype swallows exceptions too
255                   throw new RuntimeException(e);
256                 }
257               } else {
258                 super.visitAnnotation(anno);
259               }
260             }
261     
262             @Override
263             public void printExpr(JCTree tree, int prec) throws IOException {
264               EndPosTable endPositions = unit.endPositions;
265               /*
266                * Modifiers, and specifically flags like final, appear to just need weird special
267                * handling.
268                *
269                * Note: we can't use {@code TreeInfo.getEndPos()} or {@code JCTree.getEndPosition()}
270                * here, because they will return the end position of an enclosing AST node for trees
271                * whose real end positions aren't stored.
272                */
273               int endPos = endPositions.getEndPos(tree);
274               boolean hasRealEndPosition = endPos != Position.NOPOS;
275               if (tree.getKind() != Kind.MODIFIERS && hasRealEndPosition) {
276                 writer.append(unitContents.substring(tree.getStartPosition(), endPos));
277               } else {
278                 super.printExpr(tree, prec);
279               }
280             }
281     
282             @Override
283             public void visitTry(JCTry tree) {
284               if (tree.getResources().isEmpty()) {
285                 super.visitTry(tree);
286                 return;
287               }
288               try {
289                 print("try (");
290                 boolean first = true;
291                 for (JCTree resource : tree.getResources()) {
292                   if (!first) {
293                     print(";");
294                     println();
295                   }
296                   printExpr(resource);
297                   first = false;
298                 }
299                 print(")");
300                 printStat(tree.body);
301                 for (JCCatch catchStmt : tree.getCatches()) {
302                   printStat(catchStmt);
303                 }
304                 if (tree.getFinallyBlock() != null) {
305                   print(" finally ");
306                   printStat(tree.getFinallyBlock());
307                 }
308               } catch (IOException e) {
309                 throw new RuntimeException(e);
310               }
311             }
312           };
313         } catch (IOException e) {
314           throw new RuntimeException(e);
315         }

331       /**
332        * Returns the inferred method type of the template based on the given actual argument types.
333        *
334        * @throws InferException if no instances of the specified type variables would allow the
335        *         {@code actualArgTypes} to match the {@code expectedArgTypes}
336        */
337       private Type infer(Warner warner,
338           Inliner inliner,
339           List<Type> freeTypeVariables,
340           List<Type> expectedArgTypes,
341           Type returnType,
342           List<Type> actualArgTypes) throws InferException {
343         Symtab symtab = inliner.symtab();
344     
345         Type methodType =
346             new MethodType(expectedArgTypes, returnType, List.<Type>nil(), symtab.methodClass);
347         if (!freeTypeVariables.isEmpty()) {
348           methodType = new ForAll(freeTypeVariables, methodType);
349         }
350     
351         Enter enter = inliner.enter();
352         MethodSymbol methodSymbol =
353             new MethodSymbol(0, inliner.asName("__m__"), methodType, symtab.unknownSymbol);
354     
355         Type site = symtab.methodClass.type;
356     
357         Env<AttrContext> env = enter.getTopLevelEnv(
358             TreeMaker.instance(inliner.getContext()).TopLevel(List.<JCTree>nil()));
359     
360         // Set up the resolution phase:
361         try {
362           Field field = AttrContext.class.getDeclaredField("pendingResolutionPhase");
363           field.setAccessible(true);
364           field.set(env.info, newMethodResolutionPhase(autoboxing()));
365         } catch (ReflectiveOperationException e) {
366           throw new LinkageError(e.getMessage(), e);
367         }
368     
369         Object resultInfo;
370         try {
371           Class<?> resultInfoClass = Class.forName("com.sun.tools.javac.comp.Attr$ResultInfo");
372           Constructor<?> resultInfoCtor =
373               resultInfoClass.getDeclaredConstructor(Attr.class, KindSelector.class, Type.class);
374           resultInfoCtor.setAccessible(true);
375           resultInfo =
376               resultInfoCtor.newInstance(
377                   Attr.instance(inliner.getContext()), KindSelector.PCK, Type.noType);
378         } catch (ReflectiveOperationException e) {
379           throw new LinkageError(e.getMessage(), e);
380         }
381     
382         // Type inference sometimes produces diagnostics, so we need to catch them to avoid interfering
383         // with the enclosing compilation.
384         Log.DeferredDiagnosticHandler handler =
385             new Log.DeferredDiagnosticHandler(Log.instance(inliner.getContext()));
386         try {
387           MethodType result =
388               callCheckMethod(warner, inliner, resultInfo, actualArgTypes, methodSymbol, site, env);
389           if (!handler.getDiagnostics().isEmpty()) {
390             throw new InferException(handler.getDiagnostics());
391           }
392           return result;
393         } finally {
394           Log.instance(inliner.getContext()).popDiagnosticHandler(handler);
395         }
396       }

398       /**
399        * Reflectively instantiate the package-private {@code MethodResolutionPhase} enum.
400        */
401       private static Object newMethodResolutionPhase(boolean autoboxing) {
402         for (Class<?> c : Resolve.class.getDeclaredClasses()) {
403           if (!c.getName().equals("com.sun.tools.javac.comp.Resolve$MethodResolutionPhase")) {
404             continue;
405           }
406           for (Object e : c.getEnumConstants()) {
407             if (e.toString().equals(autoboxing ? "BOX" : "BASIC")) {
408               return e;
409             }
410           }
411         }
412         return null;
413       }

415       /**
416        * Reflectively invoke Resolve.checkMethod(), which despite being package-private is apparently
417        * the only useful entry-point into javac8's type inference implementation.
418        */
419       private MethodType callCheckMethod(Warner warner,
420           Inliner inliner,
421           Object resultInfo,
422           List<Type> actualArgTypes,
423           MethodSymbol methodSymbol,
424           Type site,
425           Env<AttrContext> env) throws InferException {
426         try {
427           Method checkMethod;
428           checkMethod = Resolve.class.getDeclaredMethod(
429               "checkMethod",
430               Env.class,
431               Type.class,
432               Symbol.class,
433              Class.forName("com.sun.tools.javac.comp.Attr$ResultInfo"), // ResultInfo is package-private
434              List.class,
435              List.class,
436              Warner.class);
437           checkMethod.setAccessible(true);
438           return (MethodType) checkMethod.invoke(Resolve.instance(inliner.getContext()),
439             env,
440             site,
441             methodSymbol,
442             resultInfo,
443             actualArgTypes,
444             /*freeTypeVariables=*/List.<Type>nil(),
445             warner);
446         } catch (InvocationTargetException e) {
447           if (e.getCause() instanceof Resolve.InapplicableMethodException) {
448             throw new InferException(ImmutableList.of(
449                 ((Resolve.InapplicableMethodException) e.getTargetException()).getDiagnostic()));
450           }
451           throw new LinkageError(e.getMessage(), e.getCause());
452         } catch (ReflectiveOperationException e) {
453           throw new LinkageError(e.getMessage(), e);
454         }
455       }

457       /**
458        * Returns a list of the elements of {@code typeVariables} that are <em>not</em> bound in the
459        * specified {@link Unifier}.
460        */
461       private ImmutableList<UTypeVar> freeTypeVars(Unifier unifier) {
462         ImmutableList.Builder<UTypeVar> builder = ImmutableList.builder();
463         for (UTypeVar var : typeVariables(unifier.getContext())) {
464           if (unifier.getBinding(var.key()) == null) {
465             builder.add(var);
466           }
467         }
468         return builder.build();
469       }

471       protected static Fix addImports(Inliner inliner, SuggestedFix.Builder fix) {
472         for (String importToAdd : inliner.getImportsToAdd()) {
473           fix.addImport(importToAdd);
474         }
475         for (String staticImportToAdd : inliner.getStaticImportsToAdd()) {
476           fix.addStaticImport(staticImportToAdd);
477         }
478         return fix.build();
479       }
