44     /**
45      * Supplies {@link Scanner}s and provides access to the backing sets of all {@link
46      * BugChecker}s and enabled {@link BugChecker}s.
47      */
48     public abstract class ScannerSupplier implements Supplier<Scanner> {
49     
50       /* Static factory methods and helpers */
51     
52       /**
53        * Returns a {@link ScannerSupplier} with a specific list of {@link BugChecker} classes.
54        */
55       @SafeVarargs
56       public static ScannerSupplier fromBugCheckerClasses(
57           Class<? extends BugChecker>... checkerClasses) {
58         return fromBugCheckerClasses(Arrays.asList(checkerClasses));
59       }
60     
61       private static PMap<String, BugPattern.SeverityLevel> defaultSeverities(
62           Iterable<BugCheckerInfo> checkers) {
63         PMap<String, BugPattern.SeverityLevel> severities = HashTreePMap.empty();
64         for (BugCheckerInfo check : checkers) {
65           severities = severities.plus(check.canonicalName(), check.defaultSeverity());
66         }
67         return severities;
68       }
69     
70       /**
71        * Returns a {@link ScannerSupplier} with a specific list of {@link BugChecker} classes.
72        */
73       public static ScannerSupplier fromBugCheckerClasses(
74           Iterable<Class<? extends BugChecker>> checkers) {
75         ImmutableList.Builder<BugCheckerInfo> builder = ImmutableList.builder();
76         for (Class<? extends BugChecker> checker : checkers) {
77           builder.add(BugCheckerInfo.create(checker));
78         }
79         return fromBugCheckerInfos(builder.build());
80       }
81     
82       /**
83        * Returns a {@link ScannerSupplier} built from a list of {@link BugCheckerInfo}s.
84        */
85       public static ScannerSupplier fromBugCheckerInfos(Iterable<BugCheckerInfo> checkers) {
86         ImmutableBiMap.Builder<String, BugCheckerInfo> builder = ImmutableBiMap.builder();
87         for (BugCheckerInfo checker : checkers) {
88           builder.put(checker.canonicalName(), checker);
89         }
90         ImmutableBiMap<String, BugCheckerInfo> allChecks = builder.build();
91         return new ScannerSupplierImpl(allChecks, defaultSeverities(allChecks.values()));
92       }
93     
94       /**
95        * Returns a {@link ScannerSupplier} that just returns the {@link Scanner} that was passed in.
96        * Used mostly for testing.  Does not implement any method other than
97        * {@link ScannerSupplier#get()}.
98        */
99       public static ScannerSupplier fromScanner(Scanner scanner) {
100         return new InstanceReturningScannerSupplierImpl(scanner);
101       }
102     
103       /* Instance methods */
104     
105       /**
106        * Returns a map of check name to {@link BugCheckerInfo} for all {@link BugCheckerInfo}s in this
107        * {@link ScannerSupplier}, including disabled ones.
108        */
109       protected abstract ImmutableBiMap<String, BugCheckerInfo> getAllChecks();
110     
111       /**
112        * Returns the set of {@link BugCheckerInfo}s that are enabled in this {@link ScannerSupplier}.
113        */
114       protected abstract ImmutableSet<BugCheckerInfo> getEnabledChecks();
115     
116       protected abstract PMap<String, BugPattern.SeverityLevel> severities();
117     
118       /**
119        * Applies options to this {@link ScannerSupplier}.
120        *
121        * <p>Command-line options to override check severities may do any of the following:
122        * <ul>
123        * <li>Enable a check that is currently off</li>
124        * <li>Disable a check that is currently on</li>
125        * <li>Change the severity of a check that is on, promoting a warning to an error or demoting
126        * an error to a warning</li>
127        * </ul>
128        *
129        * @param errorProneOptions an {@link ErrorProneOptions} object that encapsulates the overrides
130        * for this compilation
131        * @throws InvalidCommandLineOptionException if the override map attempts to disable a check
132        * that may not be disabled
133        */
134       @CheckReturnValue
135       public ScannerSupplier applyOverrides(ErrorProneOptions errorProneOptions)
136           throws InvalidCommandLineOptionException {
137         Map<String, Severity> severityOverrides = errorProneOptions.getSeverityMap();
138         if (severityOverrides.isEmpty()) {
139           return this;
140         }
141     
142         // Initialize result allChecks map and enabledChecks set with current state of this Supplier.
143         ImmutableBiMap<String, BugCheckerInfo> checks = getAllChecks();
144         PMap<String, SeverityLevel> severities = severities();
145     
146         // Create a map from names (canonical and alternate) to checks. We could do this when the
147         // supplier is created, but applyOverrides() is unlikely to be called more than once per
148         // scanner instance.
149         Multimap<String, BugCheckerInfo> checksByAllNames = ArrayListMultimap.create();
150         for (BugCheckerInfo checker : getAllChecks().values()) {
151           for (String name : checker.allNames()) {
152             checksByAllNames.put(name, checker);
153           }
154         }
155         
156         // Process overrides
157         for (Entry<String, Severity> entry : severityOverrides.entrySet()) {
158           Collection<BugCheckerInfo> checksWithName = checksByAllNames.get(entry.getKey());
159           if (checksWithName.isEmpty()) {
160             if (errorProneOptions.ignoreUnknownChecks()) {
161               continue;
162             }
163             throw new InvalidCommandLineOptionException(
164                 entry.getKey() + " is not a valid checker name");
165           }
166           for (BugCheckerInfo check : checksWithName) {
167             switch (entry.getValue()) {
168               case OFF:
169                 if (!check.suppressibility().disableable()) {
170                   throw new InvalidCommandLineOptionException(
171                       check.canonicalName() + " may not be disabled");
172                 }
173                 severities = severities.plus(check.canonicalName(), SeverityLevel.NOT_A_PROBLEM);
174                 break;
175               case DEFAULT:
176                 severities = severities.plus(check.canonicalName(), check.defaultSeverity());
177                 break;
178               case WARN:
179                 // Demoting an enabled check from an error to a warning is a form of disabling
180                 if (check.severity(severities).enabled()
181                     && !check.suppressibility().disableable()
182                     && check.defaultSeverity() == SeverityLevel.ERROR) {
183                   throw new InvalidCommandLineOptionException(check.canonicalName()
184                       + " is not disableable and may not be demoted to a warning");
185                 }
186                 severities = severities.plus(check.canonicalName(), SeverityLevel.WARNING);
187                 break;
188               case ERROR:
189                 severities = severities.plus(check.canonicalName(), SeverityLevel.ERROR);
190                 break;
191               default:
192                 throw new IllegalStateException("Unexpected severity level: " + entry.getValue());
193             }
194           }
195         }
196     
197         return new ScannerSupplierImpl(checks, severities);
198       }
199     
200       /**
201        * Composes this {@link ScannerSupplier} with the {@code other} {@link ScannerSupplier}. The set
202        * of checks that are turned on is the union of the set of checks on in {@code this} and
203        * {@code other}.
204        */
205       @CheckReturnValue
206       public ScannerSupplier plus(ScannerSupplier other) {
207         ImmutableBiMap<String, BugCheckerInfo> combinedAllChecks =
208             ImmutableBiMap.<String, BugCheckerInfo>builder()
209                 .putAll(this.getAllChecks())
210                 .putAll(other.getAllChecks())
211                 .build();
212         PMap<String, SeverityLevel> combinedSeverities =
213             this.severities().plusAll(other.severities());
214         return new ScannerSupplierImpl(combinedAllChecks, combinedSeverities);
215       }
216     
217       /**
218        * Filters this {@link ScannerSupplier} based on the provided predicate. Returns a
219        * {@link ScannerSupplier} with only the checks enabled that satisfy the predicate.
220        */
221       @CheckReturnValue
222       public ScannerSupplier filter(Predicate<? super BugCheckerInfo> predicate) {
223         PMap<String, SeverityLevel> filteredSeverities = severities();
224         for (Entry<String, SeverityLevel> entry : severities().entrySet()) {
225           if (!predicate.apply(getAllChecks().get(entry.getKey()))) {
226             filteredSeverities = filteredSeverities.plus(entry.getKey(), SeverityLevel.NOT_A_PROBLEM);
227           }
228         }
229         return new ScannerSupplierImpl(getAllChecks(), filteredSeverities);
230       }
231     }
