18     package com.google.errorprone.refaster;
19     
20     import static com.google.common.base.Preconditions.checkArgument;
21     import static com.google.errorprone.refaster.Unifier.unifications;
22     
23     import com.google.auto.value.AutoValue;
24     import com.google.common.collect.ImmutableBiMap;
25     
26     import com.sun.source.tree.CompoundAssignmentTree;
27     import com.sun.source.tree.TreeVisitor;
28     import com.sun.tools.javac.tree.JCTree;
29     import com.sun.tools.javac.tree.JCTree.JCAssignOp;
30     
31     /**
32      * {@link UTree} representation of a {@link CompoundAssignmentTree}.
33      *
34      * @author lowasser@google.com (Louis Wasserman)
35      */
36     @AutoValue
37     abstract class UAssignOp extends UExpression implements CompoundAssignmentTree {
38       private static final ImmutableBiMap<Kind, JCTree.Tag> TAG =
39           ImmutableBiMap.<Kind, JCTree.Tag>builder()
40               .put(Kind.PLUS_ASSIGNMENT, JCTree.Tag.PLUS_ASG)
41               .put(Kind.MINUS_ASSIGNMENT, JCTree.Tag.MINUS_ASG)
42               .put(Kind.MULTIPLY_ASSIGNMENT, JCTree.Tag.MUL_ASG)
43               .put(Kind.DIVIDE_ASSIGNMENT, JCTree.Tag.DIV_ASG)
44               .put(Kind.REMAINDER_ASSIGNMENT, JCTree.Tag.MOD_ASG)
45               .put(Kind.LEFT_SHIFT_ASSIGNMENT, JCTree.Tag.SL_ASG)
46               .put(Kind.RIGHT_SHIFT_ASSIGNMENT, JCTree.Tag.SR_ASG)
47               .put(Kind.UNSIGNED_RIGHT_SHIFT_ASSIGNMENT, JCTree.Tag.USR_ASG)
48               .put(Kind.OR_ASSIGNMENT, JCTree.Tag.BITOR_ASG)
49               .put(Kind.AND_ASSIGNMENT, JCTree.Tag.BITAND_ASG)
50               .put(Kind.XOR_ASSIGNMENT, JCTree.Tag.BITXOR_ASG)
51               .build();
52     
53       public static UAssignOp create(UExpression variable, Kind operator, UExpression expression) {
54         checkArgument(TAG.containsKey(operator),
55             "Tree kind %s does not represent a compound assignment operator", operator);
56         return new AutoValue_UAssignOp(variable, operator, expression);
57       }
58     
59       @Override
60       public abstract UExpression getVariable();
61     
62       @Override
63       public abstract Kind getKind();
64     
65       @Override
66       public abstract UExpression getExpression();
67     
68       @Override
69       public JCAssignOp inline(Inliner inliner) throws CouldNotResolveImportException {
70         return inliner.maker().Assignop(
71             TAG.get(getKind()), getVariable().inline(inliner), getExpression().inline(inliner));
72       }
73     
74       @Override
75       public <R, D> R accept(TreeVisitor<R, D> visitor, D data) {
76         return visitor.visitCompoundAssignment(this, data);
77       }
78     
79       // TODO(lowasser): consider matching x = x ? y as well as x ?= y
80       
81       @Override
82       public Choice<Unifier> visitCompoundAssignment(
83           CompoundAssignmentTree assignOp, Unifier unifier) {
84         return Choice.condition(getKind() == assignOp.getKind(), unifier)
85             .thenChoose(unifications(getVariable(), assignOp.getVariable()))
86             .thenChoose(unifications(getExpression(), assignOp.getExpression()));
87       }
88     }
