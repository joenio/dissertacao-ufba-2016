17     package com.google.errorprone.refaster;
18     
19     import com.google.common.collect.ImmutableSet;
20     import com.google.common.testing.EqualsTester;
21     import com.google.common.testing.SerializableTester;
22     
23     import com.sun.tools.javac.code.Symbol.PackageSymbol;
24     import com.sun.tools.javac.code.Symtab;
25     
26     import org.junit.Test;
27     import org.junit.runner.RunWith;
28     import org.junit.runners.JUnit4;
29     
30     /**
31      * Tests for {@link UClassIdent}.
32      *
33      * @author lowasser@google.com (Louis Wasserman)
34      */
35     @RunWith(JUnit4.class)
36     public class UClassIdentTest extends AbstractUTreeTest {
37       @Test
38       public void equality() throws CouldNotResolveImportException {
39         new EqualsTester()
40           .addEqualityGroup(UClassIdent.create("java.util.List"))
41           .addEqualityGroup(UClassIdent.create("com.sun.tools.javac.util.List"))
42           .addEqualityGroup(UClassIdent.create("java.lang.String"),
43               UClassIdent.create(inliner.resolveClass("java.lang.String")))
44           .testEquals();
45       }
46       
47       @Test
48       public void serialization() {
49         SerializableTester.reserializeAndAssert(UClassIdent.create("java.math.BigInteger"));
50       }
51       
52       @Test
53       public void inline() {
54         ImportPolicy.bind(context, ImportPolicy.IMPORT_TOP_LEVEL);
55         context.put(PackageSymbol.class, Symtab.instance(context).rootPackage);
56         assertInlines("List", UClassIdent.create("java.util.List"));
57         assertInlines("Map.Entry", UClassIdent.create("java.util.Map.Entry"));
58       }
59     
60       @Test
61       public void importConflicts() {
62         ImportPolicy.bind(context, ImportPolicy.IMPORT_TOP_LEVEL);
63         context.put(PackageSymbol.class, Symtab.instance(context).rootPackage);
64         // Test fully qualified class names
65         inliner.addImport("package.Exception");
66         assertInlines("Exception", UClassIdent.create("package.Exception"));
67         // Will import "anotherPackage.Exception" due to conflicts
68         assertInlines("anotherPackage.Exception", UClassIdent.create("anotherPackage.Exception"));
69         new EqualsTester()
70             .addEqualityGroup(inliner.getImportsToAdd(),
71                 ImmutableSet.of("package.Exception"))
72             .testEquals();
73         // Test nested class names
74         inliner.addImport("package.subpackage.Foo.Bar");
75         // Will import "package.Foo"
76         assertInlines("Foo.Bar", UClassIdent.create("package.Foo.Bar"));
77         assertInlines("Bar", UClassIdent.create("package.subpackage.Foo.Bar"));
78         // Will not import "anotherPackage.Foo" due to conflicts
79         assertInlines("anotherPackage.Foo.Bar", UClassIdent.create("anotherPackage.Foo.Bar"));
80         new EqualsTester()
81             .addEqualityGroup(inliner.getImportsToAdd(),
82                 ImmutableSet.of("package.Exception", "package.subpackage.Foo.Bar", "package.Foo"))
83             .testEquals();
84       }
85     }
