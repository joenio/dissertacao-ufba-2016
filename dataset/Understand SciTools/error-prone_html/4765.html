17     package com.google.errorprone;
18     
19     import com.google.common.base.Function;
20     import com.google.common.collect.ImmutableSet;
21     import com.google.errorprone.internal.NonDelegatingClassLoader;
22     import org.apache.tools.ant.AntClassLoader;
23     import org.apache.tools.ant.BuildException;
24     import org.apache.tools.ant.taskdefs.compilers.DefaultCompilerAdapter;
25     
26     import java.net.MalformedURLException;
27     import java.net.URL;
28     import java.net.URLClassLoader;
29     import java.nio.file.Paths;
30     
31     /**
32      * Adapts the error-prone compiler to be used in an Ant build.
33      * @author alexeagle@google.com (Alex Eagle)
34      */
35     public class ErrorProneAntCompilerAdapter extends DefaultCompilerAdapter {
36       public static class AntRunner implements Function<String[], Boolean> {
37         @Override
38         public Boolean apply(String[] args) {
39           return ErrorProneCompiler.compile(args).isOK();
40         }
41       }
42     
43       @Override
44       public boolean execute() throws BuildException {
45         ClassLoader originalLoader = ErrorProneCompiler.class.getClassLoader();
46         URL[] urls;
47         if (originalLoader instanceof URLClassLoader) {
48           urls = ((URLClassLoader) originalLoader).getURLs();
49         } else if (originalLoader instanceof AntClassLoader) {
50           String[] pieces = ((AntClassLoader) originalLoader).getClasspath().split(":");
51           urls = new URL[pieces.length];
52           for (int i = 0; i < pieces.length; ++i) {
53             try {
54               urls[i] = Paths.get(pieces[i]).toUri().toURL();
55             } catch (MalformedURLException e) {
56               throw new BuildException(e);
57             }
58           }
59         } else {
60           throw new BuildException("Unexpected ClassLoader: " + originalLoader.getClass());
61         }
62     
63         ClassLoader loader = NonDelegatingClassLoader.create(
64             ImmutableSet.<String>of(Function.class.getName()), urls, originalLoader);
65     
66         String[] args = setupModernJavacCommand().getArguments();
67     
68         try {
69           Class<?> runnerClass = Class.forName(AntRunner.class.getName(), true, loader);
70           @SuppressWarnings("unchecked")
71           Function<String[], Boolean> runner = (Function<String[], Boolean>) runnerClass.newInstance();
72           return runner.apply(args);
73         } catch (ReflectiveOperationException e) {
74           throw new LinkageError("Unable to create runner.", e);
75         }
76       }
77     }
