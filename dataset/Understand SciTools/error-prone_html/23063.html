17     package com.google.errorprone.bugpatterns.android;
18     
19     import static com.google.errorprone.BugPattern.Category.ANDROID;
20     import static com.google.errorprone.BugPattern.MaturityLevel.EXPERIMENTAL;
21     import static com.google.errorprone.BugPattern.SeverityLevel.WARNING;
22     import static com.google.errorprone.matchers.Matchers.allOf;
23     import static com.google.errorprone.matchers.Matchers.anyOf;
24     import static com.google.errorprone.matchers.Matchers.isSubtypeOf;
25     import static com.google.errorprone.matchers.Matchers.kindIs;
26     import static com.google.errorprone.matchers.Matchers.nestingKind;
27     import static com.sun.source.tree.Tree.Kind.CLASS;
28     import static javax.lang.model.element.Modifier.ABSTRACT;
29     import static javax.lang.model.element.Modifier.PUBLIC;
30     import static javax.lang.model.element.NestingKind.MEMBER;
31     
32     import com.google.errorprone.BugPattern;
33     import com.google.errorprone.VisitorState;
34     import com.google.errorprone.bugpatterns.BugChecker;
35     import com.google.errorprone.bugpatterns.BugChecker.ClassTreeMatcher;
36     import com.google.errorprone.matchers.Description;
37     import com.google.errorprone.matchers.Matcher;
38     import com.google.errorprone.util.ASTHelpers;
39     
40     import com.sun.source.tree.ClassTree;
41     import com.sun.source.tree.MethodTree;
42     import com.sun.source.tree.Tree;
43     
44     import java.util.List;
45     
46     /**
47      * @author avenet@google.com (Arnaud J. Venet)
48      */
49     @BugPattern(
50       name = "FragmentNotInstantiable",
51       altNames = {"ValidFragment"},
52       summary =
53           "Subclasses of Fragment must be instantiable via Class#newInstance():"
54               + " the class must be public, static and have a public nullary constructor",
55       category = ANDROID,
56       severity = WARNING,
57       maturity = EXPERIMENTAL
58     )
59     public class FragmentNotInstantiable extends BugChecker implements ClassTreeMatcher {
60       private static final String MESSAGE_BASE = "Fragment is not instantiable: ";
61     
62       private static final String FRAGMENT_CLASS = "android.app.Fragment";
63     
64       // Static library support version of the framework's Fragment. Used to write apps that run on
65       // platforms prior to Android 3.0.
66       private static final String FRAGMENT_CLASS_V4 = "android.support.v4.app.Fragment";
67     
68       private static final Matcher<ClassTree> FRAGMENT_MATCHER =
69           allOf(kindIs(CLASS), anyOf(isSubtypeOf(FRAGMENT_CLASS), isSubtypeOf(FRAGMENT_CLASS_V4)));
70     
71       private Description buildErrorMessage(Tree tree, String explanation) {
72         Description.Builder description = buildDescription(tree);
73         description.setMessage(MESSAGE_BASE + explanation + ".");
74         return description.build();
75       }
76     
77       @Override
78       public Description matchClass(ClassTree classTree, VisitorState state) {
79         if (!FRAGMENT_MATCHER.matches(classTree, state)) {
80           return Description.NO_MATCH;
81         }
82     
83         String className = ASTHelpers.getSymbol(classTree).toString();
84         if (className.equals(FRAGMENT_CLASS) || className.equals(FRAGMENT_CLASS_V4)) {
85           // Do not check the base class declarations (or their stubs).
86           return Description.NO_MATCH;
87         }
88     
89         // The check doesn't apply to abstract classes.
90         if (classTree.getModifiers().getFlags().contains(ABSTRACT)) {
91           return Description.NO_MATCH;
92         }
93         // A fragment must be public.
94         if (!classTree.getModifiers().getFlags().contains(PUBLIC)) {
95           return buildErrorMessage(classTree, "a fragment must be public");
96         }
97     
98         // A fragment that is an inner class must be static.
99         if (nestingKind(MEMBER).matches(classTree, state)
100             && !ASTHelpers.getSymbol(classTree).isStatic()) {
101           return buildErrorMessage(classTree, "a fragment inner class must be static");
102         }
103     
104         List<MethodTree> constructors = ASTHelpers.getConstructors(classTree, state);
105         for (MethodTree constructor : constructors) {
106           if (constructor.getParameters().isEmpty()) {
107             // The nullary constructor exists. We must make sure that it is public.
108             if (!constructor.getModifiers().getFlags().contains(PUBLIC)) {
109               return buildErrorMessage(constructor, "the nullary constructor must be public");
110             }
111             return Description.NO_MATCH;
112           }
113         }
114     
115         // If we reach this point, we know for sure that the class has at least one constructor
116         // but no nullary constructor.
117         return buildErrorMessage(classTree, "the nullary constructor is missing");
118       }
119     }
