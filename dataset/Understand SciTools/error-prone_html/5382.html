45     /**
46      * @author eaftan@google.com (Eddie Aftandilian)
47      */
48     @BugPattern(name = "StaticAccessedFromInstance",
49         summary = "A static variable or method should not be accessed from an object instance",
50         explanation = "A static variable or method should never be accessed from an instance.  This "
51             + "hides the fact that the variable or method is static and does not depend on the value "
52             + "of the object instance on which this variable or method is being invoked.",
53         category = JDK, severity = WARNING, maturity = MATURE, altNames = "static")
54     public class StaticAccessedFromInstance extends BugChecker implements MemberSelectTreeMatcher {
55     
56       private static final String MESSAGE_TEMPLATE = "Static %s %s should not be accessed from an "
57           + "object instance; instead use %s";
58     
59       private static final Matcher<ExpressionTree> staticAccessedFromInstanceMatcher = allOf(
60           anyOf(
61               staticMethod(),
62               staticFieldAccess()),
63           kindIs(Kind.MEMBER_SELECT),
64           selectedIsInstance());
65     
66       @Override
67       public Description matchMemberSelect(MemberSelectTree tree, VisitorState state) {
68         if (!staticAccessedFromInstanceMatcher.matches(tree, state)) {
69           return Description.NO_MATCH;
70         }
71     
72         // Is the static member being accessed a method or a variable?
73         Symbol staticMemberSym = ASTHelpers.getSymbol(tree);
74         if (staticMemberSym == null) {
75           return Description.NO_MATCH;
76         }
77         boolean isMethod = staticMemberSym instanceof MethodSymbol;
78     
79         // Is the static member defined in this class?
80         Symbol ownerSym = staticMemberSym.owner;
81         Symbol whereAccessedSym = ASTHelpers.getSymbol(
82             ASTHelpers.findEnclosingNode(state.getPath().getParentPath(), ClassTree.class));
83         if (!(ownerSym instanceof ClassSymbol && whereAccessedSym instanceof ClassSymbol)) {
84           return Description.NO_MATCH;
85         }
86         boolean staticMemberDefinedHere = whereAccessedSym.equals(ownerSym);
87     
88         SuggestedFix.Builder fix = SuggestedFix.builder();
89         String replacement;
90         if (staticMemberDefinedHere && isMethod) {
91           // If the static member is defined in the enclosing class and the member is a method, then
92           // just use the bare method name. Don't do this for fields, because they may share a simple
93           // name with a local in the same scope.
94           // TODO(eaftan): If we had access to name resolution info, we could do this in all applicable
95           // cases.  Investigate Scope.Entry for this.
96           replacement = tree.getIdentifier().toString();
97         } else {
98           // Replace the operand of the field access expression with the simple name of the class.
99           replacement = ownerSym.getSimpleName() + "." + tree.getIdentifier();
100     
101           // Don't import implicitly imported packages (java.lang.* and current package).
102           // TODO(cushon): move this logic into addImport?
103           Symbol packageSym = ownerSym.packge();
104           if (!packageSym.toString().equals("java.lang")
105               && !packageSym.equals(whereAccessedSym.packge())) {
106             fix.addImport(ownerSym.toString());
107           }
108         }
109         fix.replace(tree, replacement);
110     
111         // Compute strings to interpolate into diagnostic message.
112         String memberName = staticMemberSym.getSimpleName().toString();
113         String methodOrVariable = isMethod ? "method" : "variable";
114     
115         String customDiagnosticMessage = String.format(MESSAGE_TEMPLATE,
116             methodOrVariable, memberName, replacement);
117         return buildDescription(tree)
118             .setMessage(customDiagnosticMessage)
119             .addFix(fix.build())
120             .build();
121       }
122     }
