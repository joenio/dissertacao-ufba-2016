17     package com.google.errorprone.matchers;
18     
19     import com.google.errorprone.VisitorState;
20     import com.google.errorprone.dataflow.nullnesspropagation.Nullness;
21     import com.google.errorprone.dataflow.nullnesspropagation.NullnessAnalysis;
22     
23     import com.sun.source.tree.ExpressionTree;
24     import com.sun.source.util.TreePath;
25     
26     /**
27      * Matches an expression based on the result of the nullness dataflow analysis. 
28      */
29     public class NullnessMatcher implements Matcher<ExpressionTree> {
30       private final NullnessAnalysis nullnessAnalysis;
31       private final Nullness expectedNullnessValue;
32       
33       public NullnessMatcher(NullnessAnalysis nullnessAnalysis, Nullness expectedNullnessValue) {
34         this.nullnessAnalysis = nullnessAnalysis;
35         this.expectedNullnessValue = expectedNullnessValue;
36       }
37     
38       @Override
39       public boolean matches(ExpressionTree expr, VisitorState state) {
40         TreePath exprPath = new TreePath(state.getPath(), expr);
41         return nullnessAnalysis.getNullness(exprPath, state.context) == expectedNullnessValue;
42       }
43     }
