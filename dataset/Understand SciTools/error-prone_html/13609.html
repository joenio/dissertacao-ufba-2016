56     /**
57      * @author rburny@google.com (Radoslaw Burny)
58      */
59     @BugPattern(name = "MalformedFormatString",
60         summary = "Printf-like format string does not match its arguments",
61         explanation = "Format strings for printf family of functions contain format specifiers"
62             + " (placeholders) which must match amount and type of arguments that follow them. If there"
63             + " are more arguments then specifiers, redundant ones are silently ignored. If there are"
64             + " less, or their types don't match, runtime exception is thrown.",
65         category = JDK, maturity = EXPERIMENTAL, severity = ERROR)
66     public class MalformedFormatString extends BugChecker implements MethodInvocationTreeMatcher {
67     
68       private static final String EXTRA_ARGUMENTS_MESSAGE =
69           "Too many arguments for printf-like format string: expected %d, got %d";
70     
71       private static final Matcher<ExpressionTree> isPrintfLike = anyOf(
72         instanceMethod().onDescendantOf("java.io.PrintStream")
73             .withSignature("format(java.lang.String,java.lang.Object...)"),
74         instanceMethod().onDescendantOf("java.io.PrintStream")
75             .withSignature("printf(java.lang.String,java.lang.Object...)"),
76         instanceMethod().onDescendantOf("java.io.PrintWriter")
77             .withSignature("format(java.lang.String,java.lang.Object...)"),
78         instanceMethod().onDescendantOf("java.io.PrintWriter")
79             .withSignature("printf(java.lang.String,java.lang.Object...)"),
80         instanceMethod().onDescendantOf("java.util.Formatter")
81             .withSignature("format(java.lang.String,java.lang.Object...)"),
82         staticMethod().onClass("java.lang.String")
83             .withSignature("format(java.lang.String,java.lang.Object...)")
84         );
85       private static final Matcher<ExpressionTree> isPrintfLikeWithLocale = anyOf(
86         instanceMethod().onDescendantOf("java.io.PrintStream")
87             .withSignature("format(java.util.Locale,java.lang.String,java.lang.Object...)"),
88         instanceMethod().onDescendantOf("java.io.PrintStream")
89             .withSignature("printf(java.util.Locale,java.lang.String,java.lang.Object...)"),
90         instanceMethod().onDescendantOf("java.io.PrintWriter")
91             .withSignature("printf(java.util.Locale,java.lang.String,java.lang.Object...)"),
92         instanceMethod().onDescendantOf("java.io.PrintWriter")
93             .withSignature("format(java.util.Locale,java.lang.String,java.lang.Object...)"),
94         instanceMethod().onDescendantOf("java.util.Formatter")
95             .withSignature("format(java.util.Locale,java.lang.String,java.lang.Object...)"),
96         staticMethod().onClass("java.lang.String")
97             .withSignature("format(java.util.Locale,java.lang.String,java.lang.Object...)")
98         );
99       private static final ImmutableMap<TypeKind, String> BOXED_TYPE_NAMES;
100     
101       static {
102         EnumMap<TypeKind, String> boxedTypeNames = new EnumMap<>(TypeKind.class);
103         boxedTypeNames.put(TypeKind.BYTE, Byte.class.getName());
104         boxedTypeNames.put(TypeKind.SHORT, Short.class.getName());
105         boxedTypeNames.put(TypeKind.INT, Integer.class.getName());
106         boxedTypeNames.put(TypeKind.LONG, Long.class.getName());
107         boxedTypeNames.put(TypeKind.FLOAT, Float.class.getName());
108         boxedTypeNames.put(TypeKind.DOUBLE, Double.class.getName());
109         boxedTypeNames.put(TypeKind.BOOLEAN, Boolean.class.getName());
110         boxedTypeNames.put(TypeKind.CHAR, Character.class.getName());
111         boxedTypeNames.put(TypeKind.NULL, Object.class.getName());
112         // Apparently, matcher is run even if typing phase failed. Hence, we replace missing/erroneous
113         // types with Object to prevent further failures.
114         boxedTypeNames.put(TypeKind.ERROR, Object.class.getName());
115         BOXED_TYPE_NAMES = Maps.immutableEnumMap(boxedTypeNames);
116       }
117     
118       // get type name in format accepted by Formatter.check
119       private static String getFormatterType(Type type) {
120         String boxedTypeName = BOXED_TYPE_NAMES.get(type.getKind());
121         String typeName = (boxedTypeName != null ? boxedTypeName : type.toString());
122         return ("L" + typeName.replace('.', '/') + ";");
123       }
124     
125       @Override
126       public Description matchMethodInvocation(MethodInvocationTree tree, VisitorState state) {
127         int formatIndex;
128         if (isPrintfLike.matches(tree, state)) {
129           formatIndex = 0;
130         } else if (isPrintfLikeWithLocale.matches(tree, state)) {
131           formatIndex = 1;
132         } else {
133           return Description.NO_MATCH;
134         }
135     
136         List<? extends ExpressionTree> allArgs = tree.getArguments();
137         ExpressionTree formatExpression = allArgs.get(formatIndex);
138         List<? extends ExpressionTree> formatArgs = allArgs.subList(formatIndex + 1, allArgs.size());
139         // If there's a sole argument of array type, this can be a non-varargs form call. Ignoring.
140         if (formatArgs.size() == 1
141             && ((JCExpression) formatArgs.get(0)).type.getKind() == TypeKind.ARRAY) {
142           return Description.NO_MATCH;
143         }
144         String formatString = null;
145         if (formatExpression.getKind() == Kind.STRING_LITERAL) {
146           formatString = ((JCLiteral) formatExpression).getValue().toString();
147         } else {
148           Symbol sym = ASTHelpers.getSymbol(formatExpression);
149           if (sym instanceof VarSymbol) {
150             formatString = (String) ((VarSymbol) sym).getConstValue();
151           }
152         }
153         if (formatString == null) {
154           return Description.NO_MATCH;
155         }
156     
157         List<String> argTypes = new ArrayList<>();
158         for (ExpressionTree arg : formatArgs) {
159           Type type = state.getTypes().erasure(((JCExpression) arg).type);
160           argTypes.add(getFormatterType(type));
161         }
162     
163         try {
164           Formatter.check(formatString, argTypes.toArray(new String[0]));
165         } catch (ExtraFormatArgumentsException e) {
166           int begin = state.getEndPosition((JCExpression) allArgs.get(formatIndex + e.used));
167           int end = state.getEndPosition((JCMethodInvocation) tree);
168           if (end < 0) {
169             return describeMatch(tree);
170           }
171           Fix fix = SuggestedFix.replace(begin, end - 1, "");
172           String message = String.format(EXTRA_ARGUMENTS_MESSAGE, e.used, e.provided);
173           return buildDescription(tree)
174               .setMessage(message)
175               .addFix(fix)
176               .build();
177         } catch (Exception e) {
178           // TODO(rburny): provide fixes for other problems
179         }
180         return Description.NO_MATCH;
181       }
182     }
