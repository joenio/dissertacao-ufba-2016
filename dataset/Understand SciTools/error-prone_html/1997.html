26     import com.google.errorprone.bugpatterns.BugChecker.MethodTreeMatcher;
27     import com.google.errorprone.fixes.SuggestedFix;
28     import com.google.errorprone.matchers.AnnotationType;
29     import com.google.errorprone.matchers.Description;
30     import com.google.errorprone.matchers.Matcher;
31     import com.google.errorprone.matchers.Matchers;
32     import com.google.errorprone.util.ASTHelpers;
33     
34     import com.sun.source.tree.AnnotationTree;
35     import com.sun.source.tree.ClassTree;
36     import com.sun.source.tree.MethodTree;
37     import com.sun.source.tree.Tree;

39     /**
40      * Matches classes that have two or more constructors annotated with @Inject.
41      *
42      * @author sgoldfeder@google.com (Steven Goldfeder)
43      */
44     
45     @BugPattern(name = "MoreThanOneInjectableConstructor",
46         summary = "A class may not have more than one injectable constructor.", 
47         explanation = "Having more than one injectable constructor will throw a runtime error in "
48             + "compliant JSR-330 frameworks such as Guice or Dagger",
49             category = INJECT, severity = ERROR, maturity = EXPERIMENTAL)
50     public class InjectMoreThanOneInjectableConstructor extends BugChecker
51         implements MethodTreeMatcher {
52     
53       private static final String GUICE_INJECT_ANNOTATION = "com.google.inject.Inject";
54       private static final String JAVAX_INJECT_ANNOTATION = "javax.inject.Inject";
55     
56       private static final Matcher<MethodTree> INJECTABLE_METHOD_MATCHER = Matchers.<MethodTree>anyOf(
57           hasAnnotation(GUICE_INJECT_ANNOTATION), hasAnnotation(JAVAX_INJECT_ANNOTATION));
58     
59       private static final AnnotationType JAVAX_INJECT_MATCHER =
60           new AnnotationType(JAVAX_INJECT_ANNOTATION);
61       
62       private static final AnnotationType GUICE_INJECT_MATCHER =
63           new AnnotationType(GUICE_INJECT_ANNOTATION);
64     
65       @Override
66       public Description matchMethod(MethodTree methodTree, VisitorState state) {
67         int numberOfInjectableConstructors = 0;
68         if (ASTHelpers.getSymbol(methodTree).isConstructor()
69             && INJECTABLE_METHOD_MATCHER.matches(methodTree, state)) {
70           for (Tree member : ((ClassTree) state.getPath().getParentPath().getLeaf()).getMembers()) {
71             if (ASTHelpers.getSymbol(member).isConstructor()
72                 && INJECTABLE_METHOD_MATCHER.matches((MethodTree) member, state)) {
73               numberOfInjectableConstructors++;
74             }
75           }
76         }
77         return numberOfInjectableConstructors > 1 ? describe(methodTree,state) : Description.NO_MATCH;
78       }
79     
80       public Description describe(MethodTree methodTree, VisitorState state) {
81         for (AnnotationTree annotation : methodTree.getModifiers().getAnnotations()) {
82           if (JAVAX_INJECT_MATCHER.matches(annotation, state)
83               || GUICE_INJECT_MATCHER.matches(annotation, state)) {
84             return describeMatch(
85                 annotation, SuggestedFix.delete(annotation));
86           }
87         }
88         throw new IllegalStateException(
89             "Expected to find more than once constructor annotated with @Inject");
90       }
91     }
