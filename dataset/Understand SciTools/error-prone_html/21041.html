17     package com.google.errorprone;
18     
19     import static java.nio.charset.StandardCharsets.UTF_8;
20     
21     import com.google.common.base.Optional;
22     import com.google.common.base.Preconditions;
23     import com.google.common.collect.ImmutableList;
24     import com.google.common.collect.Iterables;
25     import com.google.common.jimfs.Configuration;
26     import com.google.common.jimfs.Jimfs;
27     
28     import com.sun.tools.javac.file.JavacFileManager;
29     import com.sun.tools.javac.util.Context;
30     
31     import java.io.IOException;
32     import java.io.InputStream;
33     import java.nio.file.FileSystem;
34     import java.nio.file.Files;
35     import java.nio.file.Path;
36     import java.util.Arrays;
37     import java.util.List;
38     
39     import javax.tools.JavaFileObject;
40     
41     /**
42      * An in-memory file manager for testing that uses {@link JavacFileManager} and {@link Jimfs}.
43      */
44     public class ErrorProneInMemoryFileManager extends JavacFileManager {
45       private final FileSystem fileSystem;
46       private final Optional<Class<?>> clazz;
47     
48       /**
49        * Constructs an ErrorProneInMemoryFileManager instance.
50        *
51        * <p>Instances constructed with this constructor may not use the {@link #forResource(String)}
52        * method to create a JavaFileObject from a source file on disk.  If you wish to use that method,
53        * use the {@link #ErrorProneInMemoryFileManager(Class)} constructor instead.
54        */
55       public ErrorProneInMemoryFileManager() {
56         this(Optional.<Class<?>>absent());
57       }
58     
59       /**
60        * Constructs an ErrorProneInMemoryFileManager instance, given a class that can be used to lookup
61        * file resources, such as test inputs to compile.
62        *
63        * @param clazz the class to use to locate file resources
64        */
65       public ErrorProneInMemoryFileManager(Class<?> clazz) {
66         this(Optional.<Class<?>>of(clazz));
67       }
68     
69       private ErrorProneInMemoryFileManager(Optional<Class<?>> clazz) {
70         super(new Context(), false, UTF_8);
71         this.fileSystem = Jimfs.newFileSystem(Configuration.unix());
72         this.clazz = clazz;
73       }
74     
75       /**
76        * Loads resources of the provided class into {@link JavaFileObject}s.
77        */
78       public List<JavaFileObject> forResources(Class<?> clazz, String... fileNames) {
79         ImmutableList.Builder<JavaFileObject> result = ImmutableList.builder();
80         for (String fileName : fileNames) {
81           result.add(forResource(clazz, fileName));
82         }
83         return result.build();
84       }
85     
86       /**
87        * Loads a resource of the provided class into a {@link JavaFileObject}.
88        */
89       public JavaFileObject forResource(Class<?> clazz, String fileName) {
90         Path path = fileSystem.getPath("/", clazz.getPackage().getName().replace('.', '/'), fileName);
91         try {
92           Files.createDirectories(path.getParent());
93           InputStream inputStream = null;
94           try {
95             inputStream = clazz.getResourceAsStream(fileName);
96             if (inputStream == null) {
97               inputStream = clazz.getResourceAsStream("testdata/" + fileName);
98             }
99             Files.copy(inputStream, path);
100           } finally {
101             inputStream.close();
102           }
103         } catch (IOException e) {
104           throw new AssertionError(e);
105         }
106         return Iterables.getOnlyElement(getJavaFileObjects(path));
107       }
108     
109       /**
110        * Loads a resource of the class passed into the constructor into a {@link JavaFileObject}.
111        */
112       public JavaFileObject forResource(String fileName) {
113         Preconditions.checkState(
114             clazz.isPresent(), "clazz must be set if you want to add a source from a resource file");
115         return forResource(clazz.get(), fileName);
116       }
117     
118       /**
119        * Creates a {@link JavaFileObject} with the given name and content.
120        */
121       public JavaFileObject forSourceLines(String fileName, String... lines) {
122         if (!fileName.startsWith("/")) {
123           fileName = "/" + fileName;
124         }
125         Path path = fileSystem.getPath(fileName);
126         try {
127           Files.createDirectories(path.getParent());
128           Files.write(path, Arrays.asList(lines), UTF_8);
129         } catch (IOException e) {
130           throw new AssertionError(e);
131         }
132         return Iterables.getOnlyElement(getJavaFileObjects(path));
133       }
134     }
