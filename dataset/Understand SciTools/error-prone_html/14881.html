90     /**
91      * The {@code TransferFunction} for our nullability analysis.  This analysis determines, for all
92      * variables and parameters, whether they are definitely null ({@link Nullness#NULL}),
93      * definitely non-null ({@link Nullness#NONNULL}), possibly null
94      * ({@link Nullness#NULLABLE}), or are on an infeasible path ({@link Nullness#BOTTOM}).
95      * This analysis depends only on the code and does not take nullness annotations into account.
96      *
97      * <p>Each {@code visit*()} implementation provides us with information about nullability that
98      * applies <i>if the expression is successfully evaluated</i> (in other words, it did not throw an
99      * exception). For example, if {@code foo.toString()} is successfully evaluated, we know two things:
100      *
101      * <ol>
102      * <li>The expression itself is non-null (because {@code toString()} is in our whitelist of methods
103      *     known to return non-null values)
104      * <li>{@code foo} is non-null (because it has been dereferenced without producing a {@code
105      *     NullPointerException})
106      * </ol>
107      *
108      * <p>These particular two pieces of data also demonstrate the two connected but distinct systems
109      * that we use to track nullness:
110      *
111      * <ol>
112      * <li>We compute the nullability of each expression by applying rules that may reference only the
113      *     nullability of <i>subexpressions</i>. We make the result available only to superexpressions
114      *     (and to the {@linkplain Analysis#getValue final output of the analysis}).
115      * <li>We {@linkplain LocalVariableUpdates update} and {@linkplain LocalVariableValues read} the
116      *     nullability of <i>variables</i> in a mapping that persists from node to node. This is the
117      *     only exception to the rule that we propagate data from subexpression to superexpression only.
118      *     The mapping is read only when visiting a {@link LocalVariableNode}. That is enough to give
119      *     the {@code LocalVariableNode} a value that is then available to superexpressions.
120      * </ol>
121      *
122      * <p>A further complication is that sometimes we know the nullability of an expression only
123      * conditionally based on its result. For example, {@code foo == null} proves that {@code foo} is
124      * null in the true case (such as inside {@code if (foo == null) { ... }}) and non-null in the false
125      * case (such an inside an accompanying {@code else} block). This is handled by methods that accept
126      * multiple {@link LocalVariableUpdates} instances, one for the true case and one for the false
127      * case.
128      *
129      * @author deminguyen@google.com (Demi Nguyen)
130      */
131     class NullnessPropagationTransfer extends AbstractNullnessPropagationTransfer
132         implements Serializable {

144             ImmutableSet.of(

146               member(Files.class.getName(), "toString"),

148               member(Class.class.getName(), "getName"),
149               member(Class.class.getName(), "getSimpleName"),
150               member(Class.class.getName(), "forName"),
151               member(Charset.class.getName(), "forName"));

155             ImmutableSet.of(
156                 Optional.class.getName(),
157                 Preconditions.class.getName(),
158                 Verify.class.getName(),
159                 String.class.getName(),
160                 BigInteger.class.getName(),

166             ImmutableSet.of(
167                 // The primitive types.
168                 Boolean.class.getName(),
169                 Byte.class.getName(),
170                 Character.class.getName(),
171                 Double.class.getName(),
172                 Float.class.getName(),

175                 Short.class.getName(),

179                 Enum.class.getName(),

182         @Override
183         public boolean apply(MethodInfo methodInfo) {
184           // Any method explicitly annotated with @Nullable is assumed to be capable of returning
185           // null.
186           for (String annotation : methodInfo.annotations()) {
187             if (annotation.endsWith("Nullable")) {
188               return false;
189             }
190           }
191     
192           if (methodInfo.method().equals("valueOf")
193               && CLASSES_WITH_NON_NULLABLE_VALUE_OF_METHODS.contains(methodInfo.clazz())) {
194             return true;
195           }
196           if (methodInfo.isPrimitive()) {
197             return true;
198           }
199           if (CLASSES_WITH_NON_NULLABLE_RETURNS.contains(methodInfo.clazz())) {
200             return true;
201           }
202           MemberName searchMemberName = new MemberName(methodInfo.clazz(), methodInfo.method());
203           if (METHODS_WITH_NON_NULLABLE_RETURNS.contains(searchMemberName)) {
204             return true;
205           }
206     
207           return false;
208         }
209       }

350       @Override
351       Nullness visitAssignment(AssignmentNode node, SubNodeValues inputs,
352           LocalVariableUpdates updates) {
353         Nullness value = inputs.valueOfSubNode(node.getExpression());
354     
355         Node target = node.getTarget();
356         if (target instanceof LocalVariableNode) {
357           updates.set((LocalVariableNode) target, value);
358         }
359     
360         if (target instanceof FieldAccessNode) {
361           FieldAccessNode fieldAccess = (FieldAccessNode) target;
362           ClassAndField targetField = tryGetFieldSymbol(target.getTree());
363           setReceiverNullness(updates, fieldAccess.getReceiver(), targetField);
364         }
365     
366         /*
367          * We propagate the value of the target to the value of the assignment expressions as a whole.
368          * We do this regardless of whether the target is a local variable. For example:
369          *
370          * String s = object.field = "foo"; // Now |s| is non-null.
371          *
372          * It's not clear to me that this is technically correct, but it works in practice with the
373          * bytecode generated by both javac and ecj.
374          *
375          * http://stackoverflow.com/q/12850676/28465
376          */
377         return value;
378       }

380       /**
381        * Variables take their values from their past assignments (as far as they can be determined).
382        * Additionally, variables of primitive type are always refined to non-null.
383        *
384        * <p>(This second case is rarely of interest to us. Either the variable is being used as a
385        * primitive, in which case we probably wouldn't have bothered to run the nullness checker on it,
386        * or it's being used as an Object, in which case the compiler generates a call to {@code valueOf}
387        * (to autobox the value), which triggers {@link #visitMethodInvocation}.)
388        */
389       @Override
390       Nullness visitLocalVariable(LocalVariableNode node, LocalVariableValues values) {
391         return hasPrimitiveType(node) || hasNonNullConstantValue(node)
392             ? NONNULL
393             : values.valueOfLocalVariable(node);
394       }

396       /**
397        * Refines the receiver of a field access to type non-null after a successful field access, and
398        * refines the value of the expression as a whole to non-null if applicable (e.g., if the field
399        * has a primitive type).
400        *
401        * <p>Note: If the field access occurs when the node is an l-value, the analysis won't call this
402        * method. Instead, it will call {@link #visitAssignment}.
403        */
404       @Override
405       Nullness visitFieldAccess(FieldAccessNode node, LocalVariableUpdates updates) {
406         ClassAndField accessed = tryGetFieldSymbol(node.getTree());
407         setReceiverNullness(updates, node.getReceiver(), accessed);
408         return fieldNullness(accessed);
409       }

411       /**
412        * Refines the receiver of a method invocation to type non-null after successful invocation, and
413        * refines the value of the expression as a whole to non-null if applicable (e.g., if the method
414        * returns a primitive type).
415        */
416       @Override
417       Nullness visitMethodInvocation(MethodInvocationNode node, LocalVariableUpdates thenUpdates,
418           LocalVariableUpdates elseUpdates, LocalVariableUpdates bothUpdates) {
419         ClassAndMethod callee = tryGetMethodSymbol(node.getTree());
420         setReceiverNullness(bothUpdates, node.getTarget().getReceiver(), callee);
421         setUnconditionalArgumentNullness(bothUpdates, node.getArguments(), callee);
422         setConditionalArgumentNullness(elseUpdates, node.getArguments(), callee);
423         return returnValueNullness(callee);
424       }

437       @Override
438       Nullness visitVariableDeclaration(
439           VariableDeclarationNode node, SubNodeValues inputs, LocalVariableUpdates updates) {
440         /*
441          * We could try to handle primitives here instead of in visitLocalVariable, but it won't be
442          * enough because we don't see method parameters here.
443          */
444         if (isCatchVariable(node)) {
445           updates.set(node, NONNULL);
446         }
447         /*
448          * We can return whatever we want here because a variable declaration is not an expression and
449          * thus no one can use its value directly. We've already made any updates to the value of the
450          * variable in the store above.
451          */
452         return NULLABLE;
453       }

456         return elementFromDeclaration(node.getTree()).getKind() == EXCEPTION_PARAMETER;
457       }

459       /**
460        * Refines the {@code Nullness} of {@code LocalVariableNode}s used in an equality comparison using
461        * the greatest lower bound.
462        *
463        * @param equalTo whether the comparison is == (false for !=)
464        * @param leftNode the left-hand side of the comparison
465        * @param rightNode the right-hand side of the comparison
466        * @param inputs access to nullness values of the left and right nodes
467        * @param thenUpdates the local variables whose nullness values should be updated if the
468        *      comparison returns {@code true}
469        * @param elseUpdates the local variables whose nullness values should be updated if the
470        *      comparison returns {@code false}
471        */
472       private static void handleEqualityComparison(
473           boolean equalTo,
474           Node leftNode,
475           Node rightNode,
476           SubNodeValues inputs,
477           LocalVariableUpdates thenUpdates,
478           LocalVariableUpdates elseUpdates) {
479         Nullness leftVal = inputs.valueOfSubNode(leftNode);
480         Nullness rightVal = inputs.valueOfSubNode(rightNode);
481         Nullness equalBranchValue = leftVal.greatestLowerBound(rightVal);
482         LocalVariableUpdates equalBranchUpdates = equalTo ? thenUpdates : elseUpdates;
483         LocalVariableUpdates notEqualBranchUpdates = equalTo ? elseUpdates : thenUpdates;
484     
485         if (leftNode instanceof LocalVariableNode) {
486           LocalVariableNode localVar = (LocalVariableNode) leftNode;
487           equalBranchUpdates.set(localVar, equalBranchValue);
488           notEqualBranchUpdates.set(
489               localVar, leftVal.greatestLowerBound(rightVal.deducedValueWhenNotEqual()));
490         }
491     
492         if (rightNode instanceof LocalVariableNode) {
493           LocalVariableNode localVar = (LocalVariableNode) rightNode;
494           equalBranchUpdates.set(localVar, equalBranchValue);
495           notEqualBranchUpdates.set(
496               localVar, rightVal.greatestLowerBound(leftVal.deducedValueWhenNotEqual()));
497         }
498       }

543       private Nullness fieldNullness(@Nullable ClassAndField accessed) {
544         if (accessed == null) {
545           return NULLABLE;
546         }
547     
548         if (accessed.field.equals("class")) {
549           return NONNULL;
550         }
551         if (accessed.isEnumConstant()) {
552           return NONNULL;
553         }
554         if (accessed.isPrimitive()) {
555           return NONNULL;
556         }
557         if (accessed.hasNonNullConstantValue()) {
558           return NONNULL;
559         }
560         if (accessed.isStatic() && accessed.isFinal()) {
561           if (CLASSES_WITH_NON_NULL_CONSTANTS.contains(accessed.clazz)){
562             return NONNULL;
563           }
564           // Try to evaluate initializer.
565           // TODO(kmb): Consider handling final instance fields as well
566           Nullness initializer = fieldInitializerNullnessIfAvailable(accessed);
567           if (initializer != null) {
568             return initializer;
569           }
570         }
571     
572         return NULLABLE;
573       }
574     
575       private Nullness returnValueNullness(@Nullable ClassAndMethod callee) {
576         if (callee == null) {
577           return NULLABLE;
578         }
579     
580         return methodReturnsNonNull.apply(callee) ? NONNULL : NULLABLE;
581       }

583       @Nullable
584       private Nullness fieldInitializerNullnessIfAvailable(ClassAndField accessed) {
585         if (!traversed.add(accessed.symbol)) {
586           // Circular dependency between initializers results in null.  Note static fields can also be
587           // null if they're observed before initialized, but we're ignoring that case for simplicity.
588           // TODO(kmb): Try to recognize problems with initialization order
589           return NULL;
590         }
591     
592         try {
593           JavacProcessingEnvironment javacEnv = JavacProcessingEnvironment.instance(context);
594           TreePath fieldDeclPath = Trees.instance(javacEnv).getPath(accessed.symbol);
595           // Skip initializers in other compilation units as analysis of such nodes can fail due to
596           // missing types.
597           if (fieldDeclPath == null
598               || fieldDeclPath.getCompilationUnit() != compilationUnit
599               || !(fieldDeclPath.getLeaf() instanceof VariableTree)) {
600             return null;
601           }
602     
603           ExpressionTree initializer = ((VariableTree) fieldDeclPath.getLeaf()).getInitializer();
604           if (initializer == null) {
605             return null;
606           }
607     
608           // Run flow analysis on field initializer.  This is inefficient compared to just walking
609           // the initializer expression tree but it avoids duplicating the logic from this transfer
610           // function into a method that operates on Javac Nodes.
611           TreePath initializerPath = TreePath.getPath(fieldDeclPath, initializer);
612           UnderlyingAST ast = new UnderlyingAST.CFGStatement(initializerPath.getLeaf());
613           ControlFlowGraph cfg = CFGBuilder.build(initializerPath, javacEnv, ast,
614               /* assumeAssertionsEnabled */ false, /* assumeAssertionsDisabled */ false);
615           Analysis<Nullness, LocalStore<Nullness>, NullnessPropagationTransfer> analysis =
616               new Analysis<>(javacEnv, this);
617           analysis.performAnalysis(cfg);
618           return analysis.getValue(initializerPath.getLeaf());
619         } finally {
620           traversed.remove(accessed.symbol);
621         }
622       }

625           LocalVariableUpdates updates, Node receiver, Member member) {

627           updates.set((LocalVariableNode) receiver, NONNULL);
628         }

631       /**
632        * Records which arguments are guaranteed to be non-null if the method completes without
633        * exception. For example, if {@code checkNotNull(foo, message)} completes successfully, then
634        * {@code foo} is not null.
635        */
636       private static void setUnconditionalArgumentNullness(
637           LocalVariableUpdates bothUpdates, List<Node> arguments, ClassAndMethod callee) {
638         Set<Integer> requiredNonNullParameters = REQUIRED_NON_NULL_PARAMETERS.get(callee.name());
639         for (LocalVariableNode var : variablesAtIndexes(requiredNonNullParameters, arguments)) {
640           bothUpdates.set(var, NONNULL);
641         }
642       }
643     
644       /**
645        * Records which arguments are guaranteed to be non-null only if the method completes by returning
646        * {@code true} or only if the method completes by returning {@code false}. For example, if {@code
647        * Strings.isNullOrEmpty(s)} returns {@code false}, then {@code s} is not null.
648        */
649       private static void setConditionalArgumentNullness(
650           LocalVariableUpdates elseUpdates, List<Node> arguments, ClassAndMethod callee) {
651         Set<Integer> nullImpliesTrueParameters = NULL_IMPLIES_TRUE_PARAMETERS.get(callee.name());
652         for (LocalVariableNode var : variablesAtIndexes(nullImpliesTrueParameters, arguments)) {
653           elseUpdates.set(var, NONNULL);
654         }
655       }
656     
657       private static Iterable<LocalVariableNode> variablesAtIndexes(
658           Set<Integer> indexes, List<Node> arguments) {
659         List<LocalVariableNode> result = new ArrayList<>();
660         for (Integer i : indexes) {
661           if (i < 0) {
662             i = arguments.size() + i;
663           }
664           // TODO(cpovirk): better handling of varargs
665           if (i >= 0 && i < arguments.size()) {
666             Node argument = arguments.get(i);
667             if (argument instanceof LocalVariableNode) {
668               result.add((LocalVariableNode) argument);
669             }
670           }
671         }
672         return result;
673       }

718       static final class ClassAndMethod implements Member, MethodInfo {
719         final String clazz;
720         final String method;
721         final List<String> annotations;
722         final boolean isStatic;
723         final boolean isPrimitive;
724         final boolean isBoolean;
725     
726         private ClassAndMethod(String clazz, String method, List<String> annotations, boolean isStatic,
727             boolean isPrimitive, boolean isBoolean) {
728           this.clazz = clazz;
729           this.method = method;
730           this.annotations = ImmutableList.copyOf(annotations);
731           this.isStatic = isStatic;
732           this.isPrimitive = isPrimitive;
733           this.isBoolean = isBoolean;
734         }
735     
736         static ClassAndMethod make(MethodSymbol symbol) {
737           List<? extends AnnotationMirror> annotationMirrors = symbol.getAnnotationMirrors();
738           List<String> annotations = new ArrayList<>(annotationMirrors.size());
739           for (AnnotationMirror annotationMirror : annotationMirrors) {
740             annotations.add(annotationMirror.getAnnotationType().toString());
741           }
742           return new ClassAndMethod(symbol.owner.getQualifiedName().toString(),
743               symbol.getSimpleName().toString(),
744               annotations,
745               symbol.isStatic(),
746               symbol.getReturnType().isPrimitive(),
747               symbol.getReturnType().getTag() == BOOLEAN);
748         }
749     
750         @Override
751         public boolean isStatic() {
752           return isStatic;
753         }
754     
755         MemberName name() {
756           return new MemberName(this.clazz, this.method);
757         }
758     
759         @Override
760         public String clazz() {
761           return clazz;
762         }
763     
764         @Override
765         public String method() {
766           return method;
767         }
768     
769         @Override
770         public List<String> annotations() {
771           return annotations;
772         }
773     
774         @Override
775         public boolean isPrimitive() {
776           return isPrimitive;
777         }
778       }
