43     /**
44      * NOTE(cushon): This test does two rounds of compilation and relies on the first round producing
45      * classes that appear on the second round's class path. I'm not aware of any good ways to do this
46      * without creating actual files on disk.
47      *
48      * @author eaftan@google.com (Eddie Aftandilian)
49      */
50     @RunWith(JUnit4.class)
51     public class DescendantOfTransitiveTest extends DescendantOfAbstractTest {
52     
53       @Rule public final TemporaryFolder tempDir = new TemporaryFolder();
54       final List<ScannerTest> tests = new ArrayList<ScannerTest>();
55       final List<String> filesToCompile = new ArrayList<String>();
56     
57       private void writeFileToLocalDisk(String fileName, String... lines) throws IOException {
58         File source = new File(tempDir.getRoot(), fileName);
59         new File(source.getParent()).mkdirs();
60         PrintWriter writer = new PrintWriter(new FileWriter(source));
61         for (String line : lines) {
62           writer.println(line);
63         }
64         writer.close();
65         filesToCompile.add(source.getAbsolutePath());
66       }
67     
68       private void assertCompilesWithLocalDisk(Scanner scanner) {
69         List<String> args = new ArrayList<String>();
70         args.add("-cp");
71         args.add(tempDir.getRoot().getAbsolutePath());
72         args.add("-d");
73         args.add(tempDir.getRoot().getAbsolutePath());
74         args.addAll(filesToCompile);
75     
76         ErrorProneCompiler compiler = new ErrorProneCompiler.Builder()
77             .report(ScannerSupplier.fromScanner(scanner))
78             .build();
79         Assert.assertThat(compiler.run(args.toArray(new String[0])), is(Result.OK));
80       }
81     
82       @Before
83       public void setUp() throws IOException {
84         writeFileToLocalDisk("A.java",
85             "package com.google;",
86             "public class A { ",
87             "  public int count() {",
88             "    return 1;",
89             "  }",
90             "  public static int staticCount() {",
91             "    return 2;",
92             "  }",
93             "}"
94         );
95       }
96     
97       @After
98       public void tearDown() {
99         for (ScannerTest test : tests) {
100           test.assertDone();
101         }
102       }
103     
104       @Test
105       public void shouldMatchTransitively() throws Exception {
106         writeFileToLocalDisk("I1.java",
107             "package i;",
108             "public interface I1 {",
109             "  void count();",
110             "}"
111         );
112         writeFileToLocalDisk("I2.java",
113             "package i;",
114             "public interface I2 extends I1 {",
115             "}"
116         );
117         writeFileToLocalDisk("B.java",
118             "package b;",
119             "public class B implements i.I2 {",
120             "  public void count() {",
121             "  }",
122             "}"
123         );
124         assertCompilesWithLocalDisk(new Scanner());
125         clearSourceFiles();
126         writeFileToLocalDisk("C.java",
127             "public class C {",
128             "  public void test(b.B b) {",
129             "    b.count();",
130             "  }",
131             "}"
132         );
133         assertCompilesWithLocalDisk(memberSelectMatches(true, new DescendantOf("i.I1", "count()")));
134       }
135     }
