17     package com.google.errorprone.apply;
18     
19     import static com.google.common.base.Preconditions.checkArgument;
20     import static com.google.common.base.Preconditions.checkNotNull;
21     
22     import com.google.common.collect.Range;
23     import com.google.common.collect.RangeMap;
24     import com.google.common.collect.TreeRangeMap;
25     import com.google.errorprone.DescriptionListener;
26     import com.google.errorprone.fixes.Fix;
27     import com.google.errorprone.fixes.Replacement;
28     import com.google.errorprone.matchers.Description;
29     
30     import com.sun.tools.javac.tree.EndPosTable;
31     import com.sun.tools.javac.tree.JCTree.JCCompilationUnit;
32     
33     import java.util.ArrayList;
34     import java.util.Collections;
35     import java.util.HashSet;
36     import java.util.List;
37     import java.util.Set;
38     
39     /**
40      * Implementation of a {@link Diff} that performs the modifications that are passed to its
41      * {@link #onDescribed} method, with no formatting.
42      * 
43      * <p>If imports are changed, they are resorted as per Google Java style.
44      * 
45      * @author lowasser@google.com (Louis Wasserman)
46      */
47     public final class DescriptionBasedDiff implements DescriptionListener, Diff {
48       private final String sourcePath;
49       private final JCCompilationUnit compilationUnit;
50       private final Set<String> importsToAdd;
51       private final Set<String> importsToRemove;
52       private final EndPosTable endPositions;
53       private final RangeMap<Integer, Replacement> replacements;
54     
55       public static DescriptionBasedDiff create(JCCompilationUnit compilationUnit) {
56         return new DescriptionBasedDiff(compilationUnit);
57       }
58     
59       private DescriptionBasedDiff(JCCompilationUnit compilationUnit) {
60         this.compilationUnit = checkNotNull(compilationUnit);
61         this.sourcePath = compilationUnit.getSourceFile().toUri().getPath();
62         this.importsToAdd = new HashSet<>();
63         this.importsToRemove = new HashSet<>();
64         this.endPositions = compilationUnit.endPositions;
65         this.replacements = TreeRangeMap.create();
66       }
67     
68       @Override
69       public String getRelevantFileName() {
70         return sourcePath;
71       }
72       
73       public boolean isEmpty() {
74         return importsToAdd.isEmpty() && importsToRemove.isEmpty()
75             && replacements.asMapOfRanges().isEmpty();
76       }
77     
78       @Override
79       public void onDescribed(Description description) {
80         // Use only first (most likely) suggested fix
81         if (description.fixes.size() > 0) {
82           Fix fix = description.fixes.get(0);
83           importsToAdd.addAll(fix.getImportsToAdd());
84           importsToRemove.addAll(fix.getImportsToRemove());
85           for (Replacement replacement : fix.getReplacements(endPositions)) {
86             addReplacement(replacement);
87           }
88         }
89       }
90     
91       private void addReplacement(Replacement replacement) {
92         checkNotNull(replacement);
93         Range<Integer> range;
94         if (replacement.startPosition() == replacement.endPosition()) {
95           /*
96            * 1) Inserting Range.closedOpen(N,N) into a RangeMap is a no-op, leading to pure insertions
97            *    being discarded. Therefore, we create a fake range [n,n+1), which has a non-zero 
98            *    measure and therefore won't be discarded.
99            * 2) The RangeMap is only used for ordering insertions, not for removing or inserting text.
100            *    The Replacement still contains the right range (N,N), so no extra characters will be 
101            *    deleted.
102            * 3) This does not work when insertions overlap with replacements or other insertions 
103            *    (and will raise an Exception below in that case).
104            *    There is no general solution for this case, e.g. when inserting '(' and '[' at the same 
105            *    position, we can't tell whether to insert "([" or "[(". Therefore, insertions cannot 
106            *    overlap with replacements, and there can only be one insertion at each position.
107            *    This class is only used in tests, and overlapping replacements do not occur. 
108            */
109           range = Range.closedOpen(replacement.startPosition(), replacement.endPosition() + 1);
110         } else {
111           range = Range.closedOpen(replacement.startPosition(), replacement.endPosition());
112         }
113         RangeMap<Integer, Replacement> overlaps = replacements.subRangeMap(range);
114         checkArgument(overlaps.asMapOfRanges().isEmpty(), "Replacement %s overlaps with %s",
115             replacement, overlaps);
116     
117         replacements.put(range, replacement);
118       }
119     
120       @Override
121       public void applyDifferences(SourceFile sourceFile) throws DiffNotApplicableException {
122         /*
123          * We want to apply replacements in reverse order of start position, and we know that imports
124          * come before all the other replacements.
125          */
126         List<Replacement> replacementsInOrder = new ArrayList<>(replacements.asMapOfRanges().values());
127         Collections.reverse(replacementsInOrder);
128         if (!importsToAdd.isEmpty() || !importsToRemove.isEmpty()) {
129           ImportStatements importStatements = ImportStatements.create(compilationUnit);
130           importStatements.addAll(importsToAdd);
131           importStatements.removeAll(importsToRemove);
132           replacementsInOrder.add(Replacement.create(importStatements.getStartPos(),
133               importStatements.getEndPos(), importStatements.toString()));
134         }
135     
136         for (Replacement replacement : replacementsInOrder) {
137           sourceFile.replaceChars(replacement.startPosition(), replacement.endPosition(),
138               replacement.replaceWith());
139         }
140       }
141     }
