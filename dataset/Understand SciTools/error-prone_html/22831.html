43     /**
44      * @author cushon@google.com (Liam Miller-Cushon)
45      */
46     @BugPattern(
47       name = "IsInstanceOfClass",
48       summary = "The argument to Class#isInstance(Object) should not be a class literal",
49       category = JDK,
50       severity = ERROR,
51       maturity = MATURE
52     )
53     public class IsInstanceOfClass extends BugChecker implements MethodInvocationTreeMatcher {
54     
55       private static final Matcher<MethodInvocationTree> INSTANCE_OF_CLASS =
56           Matchers.allOf(
57               instanceMethod().onExactClass("java.lang.Class").named("isInstance"),
58               argument(
59                   0,
60                   // Class is final so we could just use isSameType, but we want to
61                   // test for the same _erased_ type.
62                   Matchers.<ExpressionTree>isSubtypeOf("java.lang.Class")));
63     
64     
65       /**
66        * Suggests removing getClass() or changing to Class.class.
67        */
68       @Override
69       public Description matchMethodInvocation(MethodInvocationTree tree, VisitorState state) {
70         if (!INSTANCE_OF_CLASS.matches(tree, state)) {
71           return Description.NO_MATCH;
72         }
73         return describeMatch(tree, SuggestedFix.replace(tree, buildReplacement(tree, state)));
74       }
75     
76       static String buildReplacement(MethodInvocationTree tree, VisitorState state) {
77     
78         Operand lhs = classify((JCTree) ASTHelpers.getReceiver(tree.getMethodSelect()), state);
79         Operand rhs = classify((JCTree) Iterables.getOnlyElement(tree.getArguments()), state);
80     
81         // expr.getClass().isInstance(Bar.class) -> expr instanceof Bar
82         if (lhs.kind() == Kind.GET_CLASS && rhs.kind() == Kind.LITERAL) {
83           return String.format("%s instanceof %s", lhs.value(), rhs.value());
84         }
85     
86         // expr1.getClass().isInstance(expr2.getClass()) -> expr2.getClass().isInstance(expr1)
87         if (lhs.kind() == Kind.GET_CLASS && rhs.kind() == Kind.GET_CLASS) {
88           return String.format("%s.getClass().isInstance(%s)", rhs.value(), lhs.value());
89         }
90     
91         // Foo.class.isInstance(Bar.class) -> Bar.class == Class.class
92         if (lhs.kind() == Kind.LITERAL && rhs.kind() == Kind.LITERAL) {
93           return String.format("%s.class == Class.class", rhs.value()); // !!
94         }
95     
96         // Foo.class.isInstance(expr.getClass()) -> expr instanceof Foo
97         if (lhs.kind() == Kind.LITERAL && rhs.kind() == Kind.GET_CLASS) {
98           return String.format("%s instanceof %s", rhs.value(), lhs.value());
99         }
100     
101         // clazz.isInstance(expr.getClass()) -> clazz.isInstance(expr)
102         if (rhs.kind() == Kind.GET_CLASS) {
103           return String.format("%s.isInstance(%s)", lhs.source(), rhs.value());
104         }
105     
106         // expr.getClass().isInstance(clazz) -> clazz.isInstance(expr)
107         if (lhs.kind() == Kind.GET_CLASS) {
108           return String.format("%s.isInstance(%s)", rhs.source(), lhs.value());
109         }
110     
111         // clazz1.isInstance(clazz2) -> clazz2.isAssignableFrom(clazz1)
112         // clazz.isInstance(Bar.class) -> Bar.class.isAssignableFrom(clazz)
113         // Foo.class.isInstance(clazz) -> clazz.isAssignableFrom(Foo.class)
114         return String.format("%s.isAssignableFrom(%s)", rhs.source(), lhs.source());
115       }
116     
117       enum Kind {
118         LITERAL,
119         GET_CLASS,
120         EXPR
121       }
122     
123       @AutoValue
124       abstract static class Operand {
125         abstract Kind kind();
126         abstract CharSequence value();
127         abstract CharSequence source();
128     
129         static Operand create(Kind kind, CharSequence value, CharSequence source) {
130           return new AutoValue_IsInstanceOfClass_Operand(kind, value, source);
131         }
132       }
133     
134       static Operand classify(JCTree tree, VisitorState state) {
135         CharSequence source = state.getSourceForNode(tree);
136         if (tree instanceof MethodInvocationTree) {
137           // expr.getClass() -> "expr"
138           MethodInvocationTree receiverInvocation = (MethodInvocationTree) tree;
139           MethodSymbol sym = ASTHelpers.getSymbol(receiverInvocation);
140           if (sym != null) {
141             if (sym.getSimpleName().contentEquals("getClass") && sym.params().isEmpty()) {
142               if (receiverInvocation.getMethodSelect() instanceof IdentifierTree) {
143                 // unqualified `getClass()`
144                 return Operand.create(Kind.EXPR, state.getSourceForNode(tree), source);
145               }
146               return Operand.create(
147                   Kind.GET_CLASS,
148                   state.getSourceForNode((JCTree) ASTHelpers.getReceiver(receiverInvocation)),
149                   source);
150             }
151           }
152         } else if (tree instanceof MemberSelectTree) {
153           // Foo.class -> "Foo"
154           MemberSelectTree select = (MemberSelectTree) tree;
155           if (select.getIdentifier().contentEquals("class")) {
156             return Operand.create(
157                 Kind.LITERAL, state.getSourceForNode((JCTree) select.getExpression()), source);
158           }
159         }
160         return Operand.create(Kind.EXPR, source, source);
161       }
162     }
