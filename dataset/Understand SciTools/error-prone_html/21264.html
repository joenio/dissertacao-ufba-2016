15     package com.google.errorprone.refaster.testdata.template;
16     
17     import com.google.errorprone.refaster.annotation.AfterTemplate;
18     import com.google.errorprone.refaster.annotation.BeforeTemplate;
19     import com.google.errorprone.refaster.annotation.OfKind;
20     
21     import com.sun.source.tree.Tree.Kind;
22     
23     import java.util.AbstractList;
24     import java.util.Collections;
25     import java.util.List;
26     
27     /**
28      * Example template using an anonymous class, demonstrating that methods may be matched in any
29      * order.
30      *
31      * @author lowasser@google.com (Louis Wasserman)
32      */
33     public class AnonymousClassTemplate {
34       @BeforeTemplate
35       List<Integer> anonymousAbstractList(@OfKind(Kind.INT_LITERAL) final int value, final int size) {
36         return new AbstractList<Integer>() {
37     
38           @Override
39           public Integer get(int index) {
40             return value;
41           }
42     
43           @Override
44           public int size() {
45             return size;
46           }
47     
48           @Override
49           public Integer set(int index, Integer element) {
50             throw new UnsupportedOperationException();
51           }
52         };
53       }
54     
55       @AfterTemplate
56       List<Integer> nCopies(final int value, final int size) {
57         return Collections.nCopies(size, value);
58       }
59     }
