15     package com.google.errorprone.analysis;
16     
17     import com.google.auto.value.AutoValue;
18     import com.google.errorprone.DescriptionListener;
19     import com.google.errorprone.VisitorState;
20     import com.google.errorprone.util.ASTHelpers;
21     
22     import com.sun.source.tree.CompilationUnitTree;
23     import com.sun.source.tree.Tree;
24     import com.sun.source.util.TreePath;
25     import com.sun.source.util.TreePathScanner;
26     import com.sun.tools.javac.util.Context;
27     
28     import java.lang.annotation.Annotation;
29     import java.util.Arrays;
30     import java.util.Collections;
31     import java.util.Set;
32     
33     /**
34      * Scans a top-level compilation unit with a local analysis, respecting local suppression
35      * annotations.
36      *
37      * @author Louis Wasserman
38      */
39     @AutoValue
40     public abstract class LocalAnalysisAsTopLevelAnalysis implements TopLevelAnalysis {
41       static LocalAnalysisAsTopLevelAnalysis wrap(LocalAnalysis analysis) {
42         return new AutoValue_LocalAnalysisAsTopLevelAnalysis(analysis);
43       }
44     
45       abstract LocalAnalysis analysis();
46     
47       LocalAnalysisAsTopLevelAnalysis() {}
48     
49       @Override
50       public Set<String> knownAnalysisNames() {
51         return analysis().allNames();
52       }
53     
54       private boolean suppressed(Tree tree, Context context) {
55         switch (analysis().suppressibility()) {
56           case UNSUPPRESSIBLE:
57             return false;
58           case CUSTOM_ANNOTATION:
59             for (Class<? extends Annotation> customSuppressionAnnotation :
60                 analysis().customSuppressionAnnotations()) {
61               if (ASTHelpers.hasAnnotation(
62                   tree, customSuppressionAnnotation, new VisitorState(context))) {
63                 return true;
64               }
65             }
66             return false;
67           case SUPPRESS_WARNINGS:
68             SuppressWarnings suppressions = ASTHelpers.getAnnotation(tree, SuppressWarnings.class);
69             return suppressions != null
70                 && !Collections.disjoint(Arrays.asList(suppressions.value()), analysis().allNames());
71           default:
72             throw new AssertionError();
73         }
74       }
75     
76       @Override
77       public void analyze(CompilationUnitTree compilationUnit, final Context context,
78           final AnalysesConfig configuration, DescriptionListener listener) {
79         new TreePathScanner<Void, DescriptionListener>() {
80           @Override
81           public Void scan(Tree tree, DescriptionListener listener) {
82             if (!suppressed(tree, context)) {
83               analysis().analyze(getCurrentPath(), context, configuration, listener);
84               super.scan(tree, listener);
85             }
86             return null;
87           }
88         }.scan(new TreePath(compilationUnit), listener);
89       }
90     }
