17     package com.google.errorprone.bugpatterns;
18     
19     import java.io.Serializable;
20     import java.util.Formatter;
21     import java.util.List;
22     import java.util.Locale;
23     import java.util.Scanner;
24     
25     /**
26      * @author rburny@google.com (Radoslaw Burny)
27      */
28     public class MalformedFormatStringNegativeCases {
29       private static final Formatter formatter = new Formatter();
30       private static final Locale locale = Locale.US;
31       private static final String FORMAT_STRING = "%s %d";
32     
33       public void literalArguments() {
34         System.out.printf("nothing here");
35         System.out.printf("%d", 42);
36         formatter.format("%f %b %s", 3.14, true, "foo");
37         String.format("%c %x %s", '$', 7, new Object());
38         String.format("%h %o", new Object(), 17);
39     
40         System.out.printf(locale, "%d", 42);
41         formatter.format(locale, "%d", 42);
42         String.format(locale, "%d", 42);
43       }
44     
45       public void variableArguments() {
46         String s = "bar";
47         Integer i = 7;
48         int j = 8;
49         float f = 2.71f;
50         Object o = new Object();
51     
52         System.out.printf("%d %s %d %f", i, s, j, f);
53         formatter.format("%d %e", i, f);
54         String.format("%d %h", j, o);
55       }
56     
57       public void specialCases() {
58         System.out.printf("%b", new Object());
59         System.err.printf("%s %d %f", null, null, null);
60         System.err.printf("%n %%");
61         System.err.printf("%d", new Object[]{17});
62         String.format("%s %s", (Object[]) new String[]{"foo", "bar"});
63       }
64     
65       public void finalVariableFormat() {
66         final String formatVar = "%s";
67         String.format("%d" + "%f" + "%s", 1, 3.14, "foo");
68         String.format(formatVar, true);
69         String.format(FORMAT_STRING, "foo", 1);
70         String.format(MalformedFormatStringNegativeCases.FORMAT_STRING, "foo", 1);
71       }
72     
73       public void dynamicallyConstructedFormat() {
74         // Using dynamically built format might be a bad practice, but it should still compile.
75         Scanner scanner = new Scanner(System.in);
76         String.format(scanner.nextLine(), 1);
77         String variable = scanner.nextLine();
78         String.format(variable, 1);
79       }
80     
81       public <T> void typeVariables(T t, List<?> list) {
82         // These calls are only safe when type variable is only ever substituted with Number. However,
83         // we do not want to throw errors on them, as cleanup could be problematic.
84         String.format("%d", t);
85         String.format("%s", list);
86         String.format("%d", list.get(0));
87       }
88     
89       public <T> void integerSuperclasses() {
90         Serializable s = 0;
91         Comparable<Integer> c = 1;
92         Number n = 2;
93         String.format("%d %d %d", s, c, n);
94       }
95       // TODO(rburny): maybe test format flags, or more advanced usage?
96     }
