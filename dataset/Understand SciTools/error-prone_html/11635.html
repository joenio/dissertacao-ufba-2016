54     /**
55      * Enforce style guide §3.3.1.
56      *
57      * <p>https://google-styleguide.googlecode.com/svn/trunk/javaguide.html#s3.3.1-wildcard-imports
58      *
59      * @author cushon@google.com (Liam Miller-Cushon)

61     @BugPattern(name = "WildcardImport",
62         summary =  "Use of wildcard imports is forbidden",
63         category = GUAVA, severity = ERROR, maturity = EXPERIMENTAL)
64     public class WildcardImport extends BugChecker implements ClassTreeMatcher {
65     
66       private static final String ASTERISK = "*";
67     
68       /**
69        * An abstraction over the logic for generating fixes. The 'test' version exists because we
70        * don't have a good way to test import-only fixes, so it pretty-prints the replacements as a
71        * suffix of each on-demand import being deleted.
72        */
73       // TODO(cushon): create a way to test import-only fixes.
74       private final FixStrategy fixStrategy;
75     
76       public WildcardImport() {
77         this(FixStrategies.PRODUCTION);
78       }
79     
80       protected WildcardImport(FixStrategy fixStrategy) {
81         this.fixStrategy = fixStrategy;
82       }

84       /** A type or member that needs to be imported. */
85       @AutoValue
86       abstract static class TypeToImport {
87     
88         /**
89          * Returns the fully-qualified canonical name.
90          */
91         abstract String typeName();
92     
93         /**
94          * Returns true if the import needs to be static (i.e. the imported is for a field or method).
95          */
96         abstract boolean isStatic();
97     
98         /**
99          * Generates an import statement.
100          */
101         String generateImportStatement() {
102           return "import " + (isStatic() ? "static " : "") + typeName() + ";";
103         }
104     
105         @Override
106         public String toString() {
107           return generateImportStatement();
108         }
109     
110         static TypeToImport create(String type, boolean stat) {
111           return new AutoValue_WildcardImport_TypeToImport(type, stat);
112         }
113       }
114     
115       @Override
116       public Description matchClass(ClassTree tree, VisitorState state) {
117         ClassSymbol sym = ASTHelpers.getSymbol(tree);
118         if (sym == null) {
119           return Description.NO_MATCH;
120         }
121         // Only match on top-level classes. The analysis walks the entire class declaration looking
122         // for types that need to be imported, so matching on nested classes will produce incomplete
123         // results.
124         if (sym.getEnclosingElement() != null
125             && sym.getEnclosingElement().getKind() != ElementKind.PACKAGE) {
126           return Description.NO_MATCH;
127         }
128     
129         // First, look for on-demand imports. If there aren't any, we're done. This ensures that we
130         // only walk the class declaration if there's an error.
131         CompilationUnitTree unit = state.getPath().getCompilationUnit();
132         ImmutableList<ImportTree> wildcardImports = getWildcardImports(unit.getImports());
133         if (wildcardImports.isEmpty()) {
134           return Description.NO_MATCH;
135         }
136     
137         if (unit.getTypeDecls().size() > 1) {
138           // There are multiple top-level types, so visiting this class isn't sufficient to find all
139           // of the imports to add. Visiting all of the classes is hard (see the docs for
140           // CompilationUnitTreeInfo), so we skip the fix.
141           return describeMatch(wildcardImports.get(0));
142         }
143     
144         // Find all of the types that need to be imported.
145         Set<TypeToImport> typesToImport = ImportCollector.collect((JCClassDecl) tree);
146     
147         // Group the imported types by the on-demand import they replace.
148         Multimap<ImportTree, TypeToImport> toFix = groupImports(wildcardImports, typesToImport);
149     
150         // Generate fixes.
151         fixStrategy.apply(tree, state, wildcardImports, toFix, this);
152     
153         // TODO(cushon): this is kind of a hack. We allow the fix strategy to generate multiple small
154         // fixes, so by the time we get here all of the fixes were emitted and there's nothing to do.
155         return Description.NO_MATCH;
156       }
157     
158       /**
159        * Creates a multimap from existing on-demand imports to the single-type imports we're going to
160        * add, e.g.: java.util.* -> [java.util.List, java.util.ArrayList]
161        */
162       private Multimap<ImportTree, TypeToImport> groupImports(
163           ImmutableList<ImportTree> wildcardImports, Set<TypeToImport> typesToImport) {
164         Multimap<ImportTree, TypeToImport> toFix = LinkedListMultimap.create();
165         for (TypeToImport type : typesToImport) {
166           ImportTree wildcard = findMatchingWildcardImport(wildcardImports, type);
167           if (wildcard != null) {
168             toFix.put(wildcard, type);
169           }
170           // The 'else' case here is interesting - it's possible that one of the types / members
171           // referenced doesn't match up to any on-demand imports. It's probably something that was
172           // already visible in the scope where it was accessed. There don't appear to be any good
173           // ways to answer the question "did we need imports to resolve this name" via accessible
174           // javac state, and the current approach works very well in practice.
175           //
176           // However, there's a chance that we're dropping a real import because the canonical name of
177           // the type doesn't match the static on-demand import that was used to import it. ¯\_(ツ)_/¯
178         }
179         return toFix;
180       }
181     
182       /**
183        * Find an on-demand import matching the given single-type import specification.
184        */
185       @Nullable
186       private ImportTree findMatchingWildcardImport(
187           ImmutableList<ImportTree> wildcardImports, TypeToImport type) {
188         for (ImportTree importTree : wildcardImports) {
189           // Get the name of the on-demand import's scope, e.g. 'java.util.*' -> 'java.util'. It's
190           // guaranteed to be a MemberSelectTree by getWildcardImports().
191           String importBase =
192               ((MemberSelectTree) importTree.getQualifiedIdentifier()).getExpression().toString();
193           if (type.typeName().startsWith(importBase)) {
194             // Only associate the single-type import with this on-demand import if the portion of
195             // the single-type import after the on-demand import's scope is a simple name.
196             //
197             // e.g. if the on-demand import is 'java.nio.*', we would associate 'java.nio.Path'
198             // but not 'java.nio.charset.Charset'.
199             int next = type.typeName().indexOf('.', importBase.length() + 1);
200             if (next == -1) {
201               return importTree;
202             }
203           }
204         }
205         return null;
206       }
207     
208       /**
209        * Collect all on-demand imports.
210        */
211       private static ImmutableList<ImportTree> getWildcardImports(List<? extends ImportTree> imports) {
212         ImmutableList.Builder<ImportTree> result = ImmutableList.builder();
213         for (ImportTree tree : imports) {
214           // javac represents on-demand imports as a member select where the selected name is '*'.
215           Tree ident = tree.getQualifiedIdentifier();
216           if (!(ident instanceof MemberSelectTree)) {
217             continue;
218           }
219           MemberSelectTree select = (MemberSelectTree) ident;
220           if (select.getIdentifier().contentEquals(ASTERISK)) {
221             result.add(tree);
222           }
223         }
224         return result.build();
225       }

227       /**
228        * Walk a top-level class declaration (possibly traversing into nested classes), and collect
229        * all types and static members that are referred to by name and may need to be imported.
230        */
231       static class ImportCollector extends TreeScanner {
232     
233         private final Set<TypeToImport> seen = new LinkedHashSet<>();
234     
235         public static Set<TypeToImport> collect(JCClassDecl tree) {
236           ImportCollector collector = new ImportCollector();
237           collector.scan(tree);
238           return collector.seen;
239         }
240     
241         @Override
242         public void visitMethodDef(JCTree.JCMethodDecl method) {
243           if (ASTHelpers.isGeneratedConstructor(method)) {
244             // Skip types in the signatures of synthetic constructors
245             scan(method.body);
246           } else {
247             super.visitMethodDef(method);
248           }
249         }
250     
251         @Override
252         public void visitIdent(JCTree.JCIdent tree) {
253           if (tree.sym != null) {
254             switch (tree.sym.kind) {
255               case TYP:
256                 addType(tree.sym.type);
257                 break;
258               case VAR:
259               case MTH:
260                 addStaticMember(tree.sym);
261                 break;
262               default: // falls through
263             }
264           }
265         }
266     
267         /**
268          * Record a static field or method that may need to be imported.
269          */
270         private void addStaticMember(Symbol sym) {
271           if (!sym.isStatic()) {
272             return;
273           }
274           if (sym.owner == null) {
275             return;
276           }
277           if (sym.isPrivate()) {
278             return;
279           }
280           String canonicalName = sym.owner.getQualifiedName() + "." + sym.getQualifiedName();
281           seen.add(TypeToImport.create(canonicalName, true));
282         }
283     
284         /**
285          * Record a type reference that may need to be imported.
286          */
287         private void addType(Type type) {
288           if (type == null) {
289             return;
290           }
291           Symbol.TypeSymbol sym = type.tsym;
292           if (sym == null) {
293             return;
294           }
295           if (sym.isPrivate()) {
296             return;
297           }
298           seen.add(TypeToImport.create(sym.getQualifiedName().toString(), false));
299         }
300       }

302       /**
303        * Interface for the fix generating logic.
304        */
305       public interface FixStrategy {
306         void apply(
307             ClassTree tree,
308             VisitorState state,
309             ImmutableList<ImportTree> wildcardImports,
310             Multimap<ImportTree, TypeToImport> toFix,
311             BugChecker checker);
312       }
313     
314       /**
315        * Fix strategies for testing and production. (See the TODO above about better testing for
316        * import-only fixes.)
317        */
318       public enum FixStrategies implements FixStrategy {
319         PRODUCTION {
320           /**
321            * Delete all of the on-demand imports, and add all of the new imports.
322            */
323           @Override
324           public void apply(
325               ClassTree tree,
326               VisitorState state,
327               ImmutableList<ImportTree> wildcardImports,
328               Multimap<ImportTree, TypeToImport> toFix,
329               BugChecker checker) {
330             SuggestedFix.Builder fix = SuggestedFix.builder();
331             for (ImportTree importToDelete : wildcardImports) {
332               String importSpecification = importToDelete.getQualifiedIdentifier().toString();
333               if (importToDelete.isStatic()) {
334                 fix.removeStaticImport(importSpecification);
335               } else {
336                 fix.removeImport(importSpecification);
337               }
338             }
339             for (TypeToImport toImport : toFix.values()) {
340               if (toImport.isStatic()) {
341                 fix.addStaticImport(toImport.typeName());
342               } else {
343                 fix.addImport(toImport.typeName());
344               }
345             }
346             if (!fix.isEmpty()) {
347               state.reportMatch(
348                   buildDescriptionFromChecker(wildcardImports.get(0), checker)
349                       .addFix(fix.build()).build());
350             }
351           }
352         },
353         TEST {
354           /**
355            * Match up the replacements with each on-demand import being deleted, and pretty print
356            * them as a suffix.
357            */
358           @Override
359           public void apply(
360               ClassTree tree,
361               VisitorState state,
362               ImmutableList<ImportTree> wildcardImports,
363               Multimap<ImportTree, TypeToImport> toFix,
364               BugChecker checker) {
365             for (ImportTree toDelete : wildcardImports) {
366               Iterable<TypeToImport> replacements = toFix.get(toDelete);
367               Fix fix = SuggestedFix.postfixWith(toDelete, replacements.toString());
368               state.reportMatch(buildDescriptionFromChecker(toDelete, checker).addFix(fix).build());
369             }
370           }
371         }
372       }
