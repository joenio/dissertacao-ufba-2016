15     package com.google.errorprone.refaster;
16     
17     import static com.google.common.base.MoreObjects.firstNonNull;
18     
19     import com.google.common.collect.ImmutableCollection;
20     import com.google.common.collect.ImmutableList;
21     import com.google.common.collect.Iterables;
22     import com.google.errorprone.util.ASTHelpers;
23     
24     import com.sun.source.tree.IdentifierTree;
25     import com.sun.source.tree.Tree;
26     import com.sun.source.util.TreeScanner;
27     import com.sun.tools.javac.tree.JCTree.JCExpression;
28     
29     import java.util.Collection;
30     import java.util.Iterator;
31     import java.util.LinkedList;
32     
33     /**
34      * Cheap visitor to verify that a tree or set of trees could conceivably match a placeholder, by
35      * potentially matching at least all of the required expressions, and not matching forbidden
36      * expressions, i.e. other variables known to Refaster.
37      *
38      * @author lowasser@google.com (Louis Wasserman)
39      */
40     final class PlaceholderVerificationVisitor extends TreeScanner<Boolean, Unifier> {
41       private final Collection<UExpression> unmatched;
42       private final ImmutableCollection<UExpression> allowed;
43     
44       /**
45        * @param required UExpressions that must potentially match at least one expression in the tested
46        *        code.
47        * @param allowed UExpressions that are explicitly allowed, and excepted from the ban on
48        *        references to known variables
49        */
50       PlaceholderVerificationVisitor(Collection<? extends UExpression> required,
51           Collection<? extends UExpression> allowed) {
52         this.unmatched = new LinkedList<>(required);
53         this.allowed = ImmutableList.copyOf(allowed);
54         assert this.allowed.containsAll(unmatched);
55       }
56     
57       public boolean allRequiredMatched() {
58         return unmatched.isEmpty();
59       }
60     
61       private boolean couldUnify(UExpression expr, Tree tree, Unifier unifier) {
62         return expr.unify(tree, unifier.fork()).first().isPresent();
63       }
64     
65       @Override
66       public Boolean scan(Tree node, Unifier unifier) {
67         Iterator<UExpression> iterator = unmatched.iterator();
68         while (iterator.hasNext()) {
69           if (couldUnify(iterator.next(), node, unifier)) {
70             iterator.remove();
71             return true;
72           }
73         }
74         for (UExpression expr : allowed) {
75           if (couldUnify(expr, node, unifier)) {
76             return true;
77           }
78         }
79         if (node instanceof JCExpression) {
80           JCExpression expr = (JCExpression) node;
81           for (UFreeIdent.Key key :
82               Iterables.filter(unifier.getBindings().keySet(), UFreeIdent.Key.class)) {
83             JCExpression keyBinding = unifier.getBinding(key);
84             if (PlaceholderUnificationVisitor.equivalentExprs(unifier, expr, keyBinding)) {
85               return false;
86             }
87           }
88         }
89         return firstNonNull(super.scan(node, unifier), true);
90       }
91     
92       @Override
93       public Boolean visitIdentifier(IdentifierTree node, Unifier unifier) {
94         for (LocalVarBinding localBinding :
95             Iterables.filter(unifier.getBindings().values(), LocalVarBinding.class)) {
96           if (localBinding.getSymbol().equals(ASTHelpers.getSymbol(node))) {
97             return false;
98           }
99         }
100         return true;
101       }
102     
103       @Override
104       public Boolean reduce(Boolean r1, Boolean r2) {
105         return firstNonNull(r1, true) && firstNonNull(r2, true);
106       }
107     
108     }
