57     /**
58      * Reads each line of the bugpatterns.txt tab-delimited data file, and generates a GitHub
59      * Jekyll page for each one.
60      *
61      * @author alexeagle@google.com (Alex Eagle)
62      */
63     class BugPatternFileGenerator implements LineProcessor<List<BugPatternInstance>> {
64     
65       private static final Joiner COMMA_JOINER = Joiner.on(", ");
66       private static final Function<String, String> ANNOTATE_AND_CODIFY =
67           new Function<String, String>() {
68             @Override
69             public String apply(String annotationName) {
70               Preconditions.checkState(annotationName.endsWith(".class"));
71               return "`@"
72                   + annotationName.substring(0, annotationName.length() - ".class".length())
73                   + "`";
74             }
75           };
76     
77       private final Path outputDir;
78       private final Path exampleDirBase;
79       private final Path explanationDir;
80       private List<BugPatternInstance> result;
81     
82       /**
83        * Enables pygments-style code highlighting blocks instead of github flavoured markdown style
84        * code fences, because the latter make jekyll unhappy.
85        */
86       private final boolean usePygments;
87     
88       /**
89        * Controls whether yaml front-matter is generated.
90        */
91       private final boolean generateFrontMatter;
92     
93       public BugPatternFileGenerator(
94           Path bugpatternDir,
95           Path exampleDirBase,
96           Path explanationDir,
97           boolean generateFrontMatter,
98           boolean usePygments) {
99         this.outputDir = bugpatternDir;
100         this.exampleDirBase = exampleDirBase;
101         this.explanationDir = explanationDir;
102         this.generateFrontMatter = generateFrontMatter;
103         this.usePygments = usePygments;
104         result = new ArrayList<>();
105       }
106     
107       private static class ExampleFilter implements DirectoryStream.Filter<Path> {
108         private Pattern matchPattern;
109     
110         public ExampleFilter(String checkerName) {
111           this.matchPattern = Pattern.compile(checkerName + "(Positive|Negative)Case.*");
112         }
113     
114         @Override
115         public boolean accept(Path entry) throws IOException {
116           return Files.isDirectory(entry)
117               || matchPattern.matcher(entry.getFileName().toString()).matches();
118         }
119       }
120     
121       /**
122        * A function to convert a test case file into an {@link ExampleInfo}.
123        */
124       private static class PathToExampleInfo implements Function<Path, ExampleInfo> {
125     
126         private final String checkerClass;
127     
128         public PathToExampleInfo(String checkerClass) {
129           this.checkerClass = checkerClass;
130         }
131     
132         @Override
133         public ExampleInfo apply(Path path) {
134           ExampleInfo.ExampleKind posOrNeg = null;
135           String fileName = path.getFileName().toString();
136           if (fileName.contains("Positive")) {
137             posOrNeg = ExampleInfo.ExampleKind.POSITIVE;
138           } else if (fileName.contains("Negative")) {
139             posOrNeg = ExampleInfo.ExampleKind.NEGATIVE;
140           } else {
141             // ExampleFilter enforces this
142             throw new AssertionError(
143                 "Example filename must contain \"Positive\" or \"Negative\", but was " + fileName);
144           }
145     
146           String code;
147           try {
148             code = new String(Files.readAllBytes(path), UTF_8).trim();
149           } catch (IOException e) {
150             throw new IOError(e);
151           }
152     
153           return ExampleInfo.create(posOrNeg, checkerClass, fileName, code);
154         }
155       }
156     
157       private static final Predicate<ExampleInfo> IS_POSITIVE =
158           new Predicate<ExampleInfo>() {
159             @Override
160             public boolean apply(ExampleInfo input) {
161               return input.type() == ExampleInfo.ExampleKind.POSITIVE;
162             }
163           };
164     
165       @Override
166       public boolean processLine(String line) throws IOException {
167         BugPatternInstance pattern = new Gson().fromJson(line, BugPatternInstance.class);
168         result.add(pattern);
169     
170         // replace spaces in filename with underscores
171         Path checkPath = Paths.get(pattern.name.replace(' ', '_') + ".md");
172     
173         try (Writer writer = Files.newBufferedWriter(
174             outputDir.resolve(checkPath), UTF_8)) {
175     
176           // load side-car explanation file, if it exists
177           Path sidecarExplanation = explanationDir.resolve(checkPath);
178           if (Files.exists(sidecarExplanation)) {
179             if (!pattern.explanation.isEmpty()) {
180               throw new AssertionError(
181                   String.format(
182                       "%s specifies an explanation via @BugPattern and side-car", pattern.name));
183             }
184             pattern.explanation = new String(Files.readAllBytes(sidecarExplanation), UTF_8).trim();
185           }
186     
187           // Construct an appropriate page for this {@code BugPattern}. Include altNames if
188           // there are any, and explain the correct way to suppress.
189     
190           ImmutableMap.Builder<String, Object> templateData =
191               ImmutableMap.<String, Object>builder()
192                   .put("category", pattern.category)
193                   .put("severity", pattern.severity)
194                   .put("maturity", pattern.maturity)
195                   .put("name", pattern.name)
196                   .put("summary", pattern.summary.trim())
197                   .put("altNames", Joiner.on(", ").join(pattern.altNames))
198                   .put("explanation", pattern.explanation.trim());
199     
200           if (generateFrontMatter) {
201             Map<String, String> frontmatterData =
202                 ImmutableMap.<String, String>builder()
203                     .put("title", pattern.name)
204                     .put("summary", pattern.summary)
205                     .put("layout", "bugpattern")
206                     .put("category", pattern.category.toString())
207                     .put("severity", pattern.severity.toString())
208                     .put("maturity", pattern.maturity.toString())
209                     .build();
210             DumperOptions options = new DumperOptions();
211             options.setDefaultFlowStyle(DumperOptions.FlowStyle.BLOCK);
212             Yaml yaml = new Yaml(options);
213             Writer yamlWriter = new StringWriter();
214             yamlWriter.write("---\n");
215             yaml.dump(frontmatterData, yamlWriter);
216             yamlWriter.write("---\n");
217             templateData.put("frontmatter", yamlWriter.toString());
218           }
219     
220           if (pattern.documentSuppression) {
221             String suppression;
222             switch (pattern.suppressibility) {
223               case SUPPRESS_WARNINGS:
224                 suppression =
225                     String.format(
226                         "Suppress false positives by adding an `@SuppressWarnings(\"%s\")` "
227                             + "annotation to the enclosing element.",
228                         pattern.name);
229                 break;
230               case CUSTOM_ANNOTATION:
231                 if (pattern.customSuppressionAnnotations.length == 1) {
232                   suppression =
233                       String.format(
234                           "Suppress false positives by adding the custom suppression annotation "
235                               + "`@%s` to the enclosing element.",
236                           pattern.customSuppressionAnnotations[0]);
237                 } else {
238                   suppression =
239                       String.format(
240                           "Suppress false positives by adding one of these custom suppression "
241                               + "annotations to the enclosing element: %s",
242                           COMMA_JOINER.join(
243                               Lists.transform(
244                                   Arrays.asList(pattern.customSuppressionAnnotations),
245                                   ANNOTATE_AND_CODIFY)));
246                 }
247                 break;
248               case UNSUPPRESSIBLE:
249                 suppression = "This check may not be suppressed.";
250                 break;
251               default:
252                 throw new AssertionError(pattern.suppressibility);
253             }
254             templateData.put("suppression", suppression);
255           }
256     
257           MustacheFactory mf = new DefaultMustacheFactory();
258           Mustache mustache = mf.compile("com/google/errorprone/resources/bugpattern.mustache");
259           mustache.execute(writer, templateData.build());
260     
261           if (pattern.generateExamplesFromTestCases) {
262             // Example filename must match example pattern.
263             List<Path> examplePaths = new ArrayList<>();
264             Filter<Path> filter =
265                 new ExampleFilter(pattern.className.substring(pattern.className.lastIndexOf('.') + 1));
266             findExamples(examplePaths, exampleDirBase, filter);
267     
268             List<ExampleInfo> exampleInfos =
269                 FluentIterable.from(examplePaths)
270                     .transform(new PathToExampleInfo(pattern.className))
271                     .toSortedList( // sort by name
272                         new Comparator<ExampleInfo>() {
273                           @Override
274                           public int compare(ExampleInfo first, ExampleInfo second) {
275                             return first.name().compareTo(second.name());
276                           }
277                         });
278             Collection<ExampleInfo> positiveExamples = Collections2.filter(exampleInfos, IS_POSITIVE);
279             Collection<ExampleInfo> negativeExamples =
280                 Collections2.filter(exampleInfos, not(IS_POSITIVE));
281     
282             if (!exampleInfos.isEmpty()) {
283               writer.write("\n----------\n\n");
284     
285               if (!positiveExamples.isEmpty()) {
286                 writer.write("### Positive examples\n");
287                 for (ExampleInfo positiveExample : positiveExamples) {
288                   writeExample(positiveExample, writer);
289                 }
290               }
291     
292               if (!negativeExamples.isEmpty()) {
293                 writer.write("### Negative examples\n");
294                 for (ExampleInfo negativeExample : negativeExamples) {
295                   writeExample(negativeExample, writer);
296                 }
297               }
298             }
299           }
300         }
301         return true;
302       }
303     
304       private void writeExample(ExampleInfo example, Writer writer) throws IOException {
305         writer.write("__" + example.name() + "__\n\n");
306         if (usePygments) {
307           writer.write("{% highlight java %}\n");
308         } else {
309           writer.write("```java\n");
310         }
311         writer.write(example.code() + "\n");
312         if (usePygments) {
313           writer.write("{% endhighlight %}\n");
314         } else {
315           writer.write("```\n");
316         }
317         writer.write("\n");
318       }
319     
320       private static void findExamples(
321           List<Path> examples, Path dir, DirectoryStream.Filter<Path> filter) throws IOException {
322         try (DirectoryStream<Path> stream = Files.newDirectoryStream(dir, filter)) {
323           for (Path entry : stream) {
324             if (Files.isDirectory(entry)) {
325               findExamples(examples, entry, filter);
326             } else {
327               examples.add(entry);
328             }
329           }
330         }
331       }
332     
333       @Override
334       public List<BugPatternInstance> getResult() {
335         return result;
336       }
