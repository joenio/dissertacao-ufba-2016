130       public void reportMatch(Description description) {
131         // TODO(cushon): creating Descriptions with the default severity and updating them here isn't
132         // ideal (we could forget to do the update), so consider removing severity from Description.
133         // Instead, there could be another method on the listener that took a description and a
134         // (separate) SeverityLevel. Adding the method to the interface would require updating the
135         // existing implementations, though. Wait for default methods?
136         SeverityLevel override = severityMap.get(description.checkName);
137         if (override != null) {
138           description = description.applySeverityOverride(override);
139         }
140         descriptionListener.onDescribed(description);
141       }

147       /**
148        * Given the string representation of a simple (non-array, non-generic) type, return the
149        * matching Type.
150        *
151        * <p>If this method returns null, the compiler doesn't have access to this type, which means
152        * that if you are comparing other types to this for equality or the subtype relation, your
153        * result would always be false even if it could create the type.  Thus it might be best to bail
154        * out early in your matcher if this method returns null on your type of interest.
155        *
156        * @param typeStr The canonical string representation of a simple type (e.g., "java.lang.Object")
157        * @return The Type that corresponds to the string, or null if it cannot be found
158        */
159       public Type getTypeFromString(String typeStr) {
160         validateTypeStr(typeStr);
161         if (isPrimitiveType(typeStr)) {
162           return getPrimitiveType(typeStr);
163         }
164         Name typeName = getName(typeStr);
165         try {
166           ClassSymbol typeSymbol = getSymtab().classes.get(typeName);
167           if (typeSymbol == null) {
168             JavaCompiler compiler = JavaCompiler.instance(context);
169             Symbol sym = compiler.resolveIdent(typeStr);
170             if (!(sym instanceof ClassSymbol)) {
171               return null;
172             }
173             typeSymbol = (ClassSymbol) sym;
174           }
175           Type type = typeSymbol.asType();
176           // Throws CompletionFailure if the source/class file for this type is not available.
177           // This is hacky but the best way I can think of to handle this case.
178           type.complete();
179           if (type.isErroneous()) {
180             return null;
181           }
182           return type;
183         } catch (CompletionFailure failure) {
184           return null;
185         }
186       }

188       /**
189        * @param symStr the string representation of a symbol
190        * @return the Symbol object, or null if it cannot be found
191        */
192       public Symbol getSymbolFromString(String symStr) {
193         try {
194           Name symName = getName(symStr);
195           Symbol result = getSymtab().classes.get(symName);
196           if (result != null) {
197             // Force a completion failure if the type is not available.
198             result.complete();
199           }
200           return result;
201         } catch (CompletionFailure failure) {
202           return null;
203         }
204       }

206       /**
207        * Build an instance of a Type.
208        */
209       public Type getType(Type baseType, boolean isArray, List<Type> typeParams) {
210         boolean isGeneric = typeParams != null && !typeParams.equals(List.nil());
211         if (!isArray && !isGeneric) {
212           // Simple type.
213           return baseType;
214         } else if (isArray && !isGeneric) {
215           // Array type, not generic.
216           ClassSymbol arraySymbol = getSymtab().arrayClass;
217           return new ArrayType(baseType, arraySymbol);
218         } else if (!isArray && isGeneric) {
219           return new ClassType(Type.noType, typeParams, baseType.tsym);
220         } else {
221           throw new IllegalArgumentException("Unsupported arguments to getType");
222         }
223       }

225       /**
226        * Build an instance of a Type.
227        */
228       public Type getType(Type baseType, boolean isArray, java.util.List<Type> typeParams) {
229         boolean isGeneric = typeParams != null && !typeParams.equals(List.nil());
230         if (!isArray && !isGeneric) {
231           // Simple type.
232           return baseType;
233         } else if (isArray && !isGeneric) {
234           // Array type, not generic.
235           ClassSymbol arraySymbol = getSymtab().arrayClass;
236           return new ArrayType(baseType, arraySymbol);
237         } else if (!isArray && isGeneric) {
238           // Generic type, not array.
239           List<Type> typeParamsCopy = List.from(typeParams.toArray(new Type[typeParams.size()]));
240           return new ClassType(Type.noType, typeParamsCopy, baseType.tsym);
241         } else {
242           throw new IllegalArgumentException("Unsupported arguments to getType");
243         }
244       }

246       /**
247        * Returns the {@link TreePath} to the nearest tree node of one of the given types. To instead
248        * retrieve the element directly, use {@link #findEnclosing(Class...)}.
249        *
250        * @return the path, or {@code null} if there is no match
251        */
252       @SafeVarargs
253       public final TreePath findPathToEnclosing(Class<? extends Tree>... classes) {
254         TreePath enclosingPath = getPath();
255         while (enclosingPath != null) {
256           for (Class<? extends Tree> clazz : classes) {
257             if (clazz.isInstance(enclosingPath.getLeaf())) {
258               return enclosingPath;
259             }
260           }
261           enclosingPath = enclosingPath.getParentPath();
262         }
263         return null;
264       }
265     
266       /**
267        * Find the first enclosing tree node of one of the given types.
268        *
269        * @return the node, or {@code null} if there is no match
270        */
271       @SuppressWarnings("unchecked") // findPathToEnclosing guarantees that the type is from |classes|
272       @SafeVarargs
273       public final <T extends Tree> T findEnclosing(Class<? extends T>... classes) {
274         TreePath pathToEnclosing = findPathToEnclosing(classes);
275         return (pathToEnclosing == null) ? null : (T) pathToEnclosing.getLeaf();
276       }

291       /**
292        * Gets the original source code that represents the given node.
293        *
294        * <p>Note that this may be different from what is returned by calling .toString() on the node.
295        * This returns exactly what is in the source code, whereas .toString() pretty-prints the node
296        * from its AST representation.
297        *
298        * @return the source code that represents the node.
299        */
300       public String getSourceForNode(Tree tree) {
301         JCTree node = (JCTree) tree;
302         int start = node.getStartPosition();
303         int end = getEndPosition(node);
304         if (end < 0) {
305           return null;
306         }
307         return getSourceCode().subSequence(start, end).toString();
308       }

311        * Returns the list of {@link Token}s for the given {@link JCTree}.

314        * only be used if a fix is already going to be emitted.

320       /**
321        * Gets the end position of the given node.
322        *
323        * @return the end position of the node, or -1 if it is not available
324        */
325       public int getEndPosition(JCTree node) {
326         JCCompilationUnit compilationUnit = (JCCompilationUnit) getPath().getCompilationUnit();
327         if (compilationUnit.endPositions == null) {
328           return -1;
329         }
330         return node.getEndPosition(compilationUnit.endPositions);
331       }

333       /**
334        * Validates a type string, ensuring it is not generic and not an array type.
335        */
336       private static void validateTypeStr(String typeStr) {
337         if (typeStr.contains("[") || typeStr.contains("]")) {
338           throw new IllegalArgumentException("Cannot convert array types, please build them using "
339               + "getType()");
340         }
341         if (typeStr.contains("<") || typeStr.contains(">")) {
342           throw new IllegalArgumentException("Cannot covnert generic types, please build them using "
343               + "getType()");
344         }
345       }

347       /**
348        * Given a string that represents a primitive type (e.g., "int"), return the corresponding Type.
349        */
350       private Type getPrimitiveType(String typeStr) {
351         if (typeStr.equals("byte")) {
352           return getSymtab().byteType;
353         } else if (typeStr.equals("short")) {
354           return getSymtab().shortType;
355         } else if (typeStr.equals("int")) {
356           return getSymtab().intType;
357         } else if (typeStr.equals("long")) {
358           return getSymtab().longType;
359         } else if (typeStr.equals("float")) {
360           return getSymtab().floatType;
361         } else if (typeStr.equals("double")) {
362           return getSymtab().doubleType;
363         } else if (typeStr.equals("boolean")) {
364           return getSymtab().booleanType;
365         } else if (typeStr.equals("char")) {
366           return getSymtab().charType;
367         } else {
368           throw new IllegalStateException("Type string " + typeStr + " expected to be primitive");
369         }
370       }
371     
372       private static boolean isPrimitiveType(String typeStr) {
373         return typeStr.equals("byte") || typeStr.equals("short") || typeStr.equals("int") ||
374             typeStr.equals("long") || typeStr.equals("float") || typeStr.equals("double") ||
375             typeStr.equals("boolean") || typeStr.equals("char");
376       }
