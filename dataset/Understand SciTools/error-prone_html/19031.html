17     package com.google.errorprone.matchers.method;
18     
19     import com.google.common.base.Function;
20     import com.google.common.base.Optional;
21     import com.google.common.collect.ImmutableList;
22     import com.google.common.collect.Iterables;
23     import com.google.errorprone.VisitorState;
24     import com.google.errorprone.matchers.method.MethodMatchers.ParameterMatcher;
25     import com.google.errorprone.suppliers.Supplier;
26     import com.google.errorprone.suppliers.Suppliers;
27     
28     import com.sun.source.tree.ExpressionTree;
29     import com.sun.tools.javac.code.Type;
30     
31     /** Matches on a method's formal parameters. */
32     public class ParameterMatcherImpl extends AbstractChainedMatcher<MatchState, MatchState>
33         implements ParameterMatcher {
34     
35       private final ImmutableList<Supplier<Type>> expected;
36     
37       // TODO(cushon): Suppliers::typeFromString
38       private static final Function<String, Supplier<Type>> GET_SUPPLIER =
39           new Function<String, Supplier<Type>>() {
40             @Override public Supplier<Type> apply(String input) {
41               return Suppliers.typeFromString(input);
42             }
43           };
44     
45       ParameterMatcherImpl(AbstractSimpleMatcher<MatchState> baseMatcher,
46           Iterable<String> parameterTypes) {
47         this(baseMatcher, ImmutableList.copyOf(Iterables.transform(parameterTypes, GET_SUPPLIER)));
48       }
49     
50       ParameterMatcherImpl(AbstractSimpleMatcher<MatchState> baseMatcher,
51           ImmutableList<Supplier<Type>> parameterTypes) {
52         super(baseMatcher);
53         this.expected = parameterTypes;
54       }
55     
56       @Override
57       protected Optional<MatchState> matchResult(ExpressionTree item, MatchState info,
58           VisitorState state) {
59         ImmutableList<Type> actual = ImmutableList.copyOf(info.paramTypes());
60         if (info.sym().isVarArgs()) {
61           if (actual.size() < expected.size()) {
62             return Optional.absent();
63           }
64         } else if (actual.size() != expected.size()) {
65           return Optional.absent();
66         }
67     
68         for (int i = 0; i < actual.size(); ++i) {
69           if (!state.getTypes().isSameType(actual.get(i), expected.get(i).get(state))) {
70             return Optional.absent();
71           }
72         }
73         return Optional.of(info);
74       }
75     }
