61       /**
62        * Entry point for compiling Java code with error-prone enabled.
63        * All default checks are run, and the compile fails if they find a bug.
64        *
65        * @param args the same args which could be passed to javac on the command line
66        */
67       public static void main(String[] args) {
68         System.exit(compile(args).exitCode);
69       }

71       /**
72        * Compiles in-process.
73        *
74        * @param listener listens to the diagnostics produced by error-prone
75        * @param args the same args which would be passed to javac on the command line
76        * @return result from the compiler invocation
77        */
78       public static Result compile(DiagnosticListener<JavaFileObject> listener, String[] args) {
79         ErrorProneCompiler compiler = new ErrorProneCompiler.Builder()
80             .listenToDiagnostics(listener)
81             .build();
82         return compiler.run(args);
83       }

166       /**
167        * Default to compiling with the same -source and -target as the host's javac.
168        *
169        * <p>This prevents, e.g., targeting Java 8 by default when using error-prone on JDK7.
170        */
171       private Iterable<String> defaultToLatestSupportedLanguageLevel(Iterable<String> args) {
172     
173         String overrideLanguageLevel;
174         switch (JAVA_SPECIFICATION_VERSION.value()) {
175           case "1.7":
176             overrideLanguageLevel = "7";
177             break;
178           case "1.8":
179             overrideLanguageLevel = "8";
180             break;
181           default:
182             return args;
183         }
184     
185         return Iterables.concat(
186             Arrays.asList(
187               // suppress xlint 'options' warnings to avoid diagnostics like:
188               // 'bootstrap class path not set in conjunction with -source 1.7'
189               "-Xlint:-options",
190               "-source", overrideLanguageLevel,
191               "-target", overrideLanguageLevel),
192             args);
193       }
194     
195       /**
196        * Sets javac's {@code -XDcompilePolicy} flag to {@code byfile}.  This ensures that all classes in
197        * a file are attributed before any of them are lowered.  Error Prone depends on this behavior
198        * when analyzing files that contain multiple classes.
199        *
200        * @throws InvalidCommandLineOptionException if the {@code -XDcompilePolicy} flag is passed
201        * in the existing arguments with a value other than {@code byfile}
202        */
203       private Iterable<String> setCompilePolicyToByFile(Iterable<String> args)
204           throws InvalidCommandLineOptionException {
205         for (String arg : args) {
206           if (arg.startsWith("-XDcompilePolicy")) {
207             String value = arg.substring(arg.indexOf('=') + 1);
208             if (!value.equals("byfile")) {
209               throw new InvalidCommandLineOptionException(
210                   "-XDcompilePolicy must be byfile for Error Prone to work properly");
211             }
212             // If there is already an "-XDcompilePolicy=byfile" flag, don't do anything.
213             return args;
214           }
215         }
216         return Iterables.concat(
217             args,
218             Arrays.asList("-XDcompilePolicy=byfile"));
219       }
220     
221       private String[] prepareCompilation(String[] argv, Context context)
222           throws InvalidCommandLineOptionException {
223     
224         Iterable<String> newArgs = defaultToLatestSupportedLanguageLevel(Arrays.asList(argv));
225         newArgs = setCompilePolicyToByFile(newArgs);
226         ErrorProneOptions epOptions = ErrorProneOptions.processArgs(newArgs);
227     
228         argv = epOptions.getRemainingArgs();
229     
230         if (diagnosticListener != null) {
231           context.put(DiagnosticListener.class, diagnosticListener);
232         }
233     
234         Scanner scanner = scannerSupplier.applyOverrides(epOptions).get();
235         CodeTransformer transformer = ErrorProneScannerTransformer.create(scanner);
236     
237         setupMessageBundle(context);
238         enableEndPositions(context);
239         ErrorProneJavacJavaCompiler.preRegister(context, transformer, epOptions);
240     
241         return argv;
242       }
243     
244       private Result run(String[] argv, Context context) {
245         try {
246           argv = prepareCompilation(argv, context);
247         } catch (InvalidCommandLineOptionException e) {
248           errOutput.println(e.getMessage());
249           errOutput.flush();
250           return Result.CMDERR;
251         }
252     
253         return new Main(compilerName, errOutput).compile(argv, context);
254       }
255     
256       public Result run(
257         String[] argv,
258         List<JavaFileObject> javaFileObjects) {
259     
260         Context context = new Context();
261         return run(argv, context, null, javaFileObjects, Collections.<Processor>emptyList());
262       }
263     
264       public Result run(
265           String[] argv,
266           Context context,
267           JavaFileManager fileManager,
268           List<JavaFileObject> javaFileObjects,
269           Iterable<? extends Processor> processors) {
270     
271         try {
272           argv = prepareCompilation(argv, context);
273         } catch (InvalidCommandLineOptionException e) {
274           errOutput.println(e.getMessage());
275           errOutput.flush();
276           return Result.CMDERR;
277         }
278     
279         JavacTool tool = JavacTool.create();
280         JavacTaskImpl task = (JavacTaskImpl) tool.getTask(
281             errOutput,
282             fileManager,
283             null,
284             Arrays.asList(argv),
285             null,
286             javaFileObjects,
287             context);
288         if (processors != null) {
289           task.setProcessors(processors);
290         }
291         return task.doCall();
292       }

301       private static final String PROPERTIES_RESOURCE =
302           "/META-INF/maven/com.google.errorprone/error_prone_core/pom.properties";
303     
304       /** Loads the Error Prone version. */
305       public static Optional<String> loadVersionFromPom() {
306         try (InputStream stream = ErrorProneCompiler.class.getResourceAsStream(PROPERTIES_RESOURCE)) {
307           if (stream == null) {
308             return Optional.absent();
309           }
310           Properties mavenProperties = new Properties();
311           mavenProperties.load(stream);
312           return Optional.of(mavenProperties.getProperty("version"));
313         } catch (IOException expected) {
314           return Optional.absent();
315         }
316       }
