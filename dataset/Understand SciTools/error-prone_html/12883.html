15     package com.google.errorprone.bugpatterns;
16     
17     import static com.google.errorprone.BugPattern.Category.INJECT;
18     import static com.google.errorprone.BugPattern.MaturityLevel.EXPERIMENTAL;
19     import static com.google.errorprone.BugPattern.SeverityLevel.ERROR;
20     import static com.google.errorprone.bugpatterns.BugChecker.ClassTreeMatcher;
21     import static com.google.errorprone.matchers.Matchers.hasAnnotation;
22     import static java.lang.annotation.ElementType.METHOD;
23     import static java.lang.annotation.ElementType.TYPE;
24     
25     import com.google.errorprone.BugPattern;
26     import com.google.errorprone.VisitorState;
27     import com.google.errorprone.fixes.SuggestedFix;
28     import com.google.errorprone.matchers.Description;
29     import com.google.errorprone.matchers.Matcher;
30     import com.google.errorprone.matchers.Matchers;
31     import com.google.errorprone.util.ASTHelpers;
32     
33     import com.sun.source.tree.AnnotationTree;
34     import com.sun.source.tree.ClassTree;
35     import com.sun.tools.javac.code.Attribute.Compound;
36     import com.sun.tools.javac.code.Flags;
37     import com.sun.tools.javac.code.Symbol;
38     
39     import java.lang.annotation.ElementType;
40     import java.lang.annotation.Target;
41     
42     /**
43      * @author sgoldfeder@google.com (Steven Goldfeder)
44      */
45     @BugPattern(name = "InjectInvalidTargetingOnScopingAnnotation",
46         summary = "The target of a scoping annotation must be set to METHOD and/or TYPE.",
47         explanation = "Scoping annotations are only appropriate for provision and therefore are only " +
48             "appropriate on @Provides methods and classes that will be provided just-in-time.",
49         category = INJECT, severity = ERROR, maturity = EXPERIMENTAL)
50     public class InjectInvalidTargetingOnScopingAnnotation extends BugChecker
51         implements ClassTreeMatcher {
52     
53       private static final String GUICE_SCOPE_ANNOTATION = "com.google.inject.ScopeAnnotation";
54       private static final String JAVAX_SCOPE_ANNOTATION = "javax.inject.Scope";
55       private static final String TARGET_ANNOTATION = "java.lang.annotation.Target";
56     
57       /**
58        * Matches classes that are annotated with @Scope or @ScopeAnnotation.
59        */
60       private static final Matcher<ClassTree> SCOPE_ANNOTATION_MATCHER = Matchers.<ClassTree>anyOf(
61           hasAnnotation(GUICE_SCOPE_ANNOTATION), hasAnnotation(JAVAX_SCOPE_ANNOTATION));
62     
63       @Override
64       public final Description matchClass(ClassTree classTree, VisitorState state) {
65         Symbol classSymbol = ASTHelpers.getSymbol(classTree);
66         if ((classSymbol.flags() & Flags.ANNOTATION) != 0
67             && SCOPE_ANNOTATION_MATCHER.matches(classTree, state)) {
68           Target target = ASTHelpers.getAnnotation(classSymbol, Target.class);
69           boolean hasExclusivelyTypeAndOrMethodTargeting = false;
70           if (target != null) {
71             for (ElementType elementType : target.value()) {
72               if (elementType != METHOD && elementType != TYPE) {
73                 return describe(classTree, state);
74               } else if (elementType == METHOD || elementType == TYPE) {
75                 hasExclusivelyTypeAndOrMethodTargeting = true;
76               }
77             }
78           }
79           if(!hasExclusivelyTypeAndOrMethodTargeting) { // true for no target set and for @Target({})
80             return describe(classTree, state);
81           }
82         }
83         return Description.NO_MATCH;
84       }
85     
86       public Description describe(ClassTree classTree, VisitorState state) {
87         Compound target = ASTHelpers.getSymbol(classTree).attribute(
88             state.getSymbolFromString(TARGET_ANNOTATION));
89         if (target == null) {
90           return describeMatch(classTree, SuggestedFix.builder()
91               .addImport("java.lang.annotation.Target")
92               .addStaticImport("java.lang.annotation.ElementType.TYPE")
93               .addStaticImport("java.lang.annotation.ElementType.METHOD")
94               .prefixWith(classTree, "@Target({TYPE, METHOD})\n")
95               .build());
96         }
97         AnnotationTree targetNode = null;
98         for (AnnotationTree annotation : classTree.getModifiers().getAnnotations()) {
99           if (ASTHelpers.getSymbol(annotation).equals(state.getSymbolFromString(TARGET_ANNOTATION))) {
100             targetNode = annotation;
101           }
102         }
103         return describeMatch(targetNode, SuggestedFix.builder()
104             .addImport("java.lang.annotation.Target")
105             .addStaticImport("java.lang.annotation.ElementType.TYPE")
106             .addStaticImport("java.lang.annotation.ElementType.METHOD")
107             .replace(targetNode, "@Target({TYPE, METHOD})")
108             .build());
109       }
110     }
