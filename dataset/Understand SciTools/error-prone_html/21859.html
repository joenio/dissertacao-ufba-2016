17     package com.google.errorprone.refaster.annotation;
18     
19     import java.lang.annotation.ElementType;
20     import java.lang.annotation.Retention;
21     import java.lang.annotation.RetentionPolicy;
22     import java.lang.annotation.Target;
23     
24     /**
25      * Indicates that a parameter to a placeholder method is not required to be used in the
26      * placeholder's implementation.  For example, the pattern
27      *
28      * <pre><code>
29      * abstract class Utf8Bytes {
30      *  {@literal @}Placeholder abstract void handleException(
31      *      {@literal @}MayOptionallyUse UnsupportedEncodingException e);
32      *  {@literal @}Placeholder abstract void useString(String str);
33      *  {@literal @}BeforeTemplate void before(byte[] array) {
34      *     try {
35      *       useString(new String(array, "UTF_8"));
36      *     } catch (UnsupportedEncodingException e) {
37      *       handleException(e);
38      *     }
39      *   }
40      *  {@literal @}AfterTemplate void after(byte[] array) {
41      *     useString(new String(array, StandardCharsets.UTF_8));
42      *   }
43      * }
44      * </code></pre>
45      *
46      * would match even if the catch statement were empty, or didn't refer to e.
47      */
48     @Target(ElementType.PARAMETER)
49     @Retention(RetentionPolicy.SOURCE)
50     public @interface MayOptionallyUse {
51     }
