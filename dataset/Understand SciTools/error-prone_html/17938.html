17     package com.google.errorprone.matchers;
18     
19     import static org.junit.Assert.assertEquals;
20     
21     import com.google.errorprone.VisitorState;
22     import com.google.errorprone.scanner.Scanner;
23     
24     import com.sun.source.tree.ExpressionTree;
25     import com.sun.source.tree.MethodInvocationTree;
26     
27     import org.junit.After;
28     import org.junit.Before;
29     
30     import java.util.ArrayList;
31     import java.util.List;
32     
33     public class DescendantOfAbstractTest extends CompilerBasedAbstractTest {
34     
35       final List<ScannerTest> tests = new ArrayList<ScannerTest>();
36     
37       @Before
38       public void setUp() throws Exception {
39         writeFile("com/google/A.java",
40             "package com.google;",
41             "public class A { ",
42             "  public int count() {",
43             "    return 1;",
44             "  }",
45             "  public static int staticCount() {",
46             "    return 2;",
47             "  }",
48             "}"
49         );
50       }
51     
52       @After
53       public void tearDown() {
54         for (ScannerTest test : tests) {
55           test.assertDone();
56         }
57       }
58     
59       protected abstract class ScannerTest extends Scanner {
60         public abstract void assertDone();
61       }
62     
63       protected Scanner memberSelectMatches(final boolean shouldMatch,
64                                             final DescendantOf toMatch) {
65         ScannerTest test = new ScannerTest() {
66           private boolean matched = false;
67     
68           @Override
69           public Void visitMethodInvocation(MethodInvocationTree node, VisitorState visitorState) {
70             visitorState = visitorState.withPath(getCurrentPath());
71             ExpressionTree methodSelect = node.getMethodSelect();
72             if (toMatch.matches(methodSelect, visitorState)) {
73               matched = true;
74             }
75             return super.visitMethodInvocation(node, visitorState);
76           }
77     
78           @Override
79           public void assertDone() {
80             assertEquals(matched, shouldMatch);
81           }
82         };
83         tests.add(test);
84         return test;
85       }
86     }
