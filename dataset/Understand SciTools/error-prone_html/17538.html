17     package com.google.errorprone.annotations;
18     
19     import static java.lang.annotation.ElementType.METHOD;
20     import static java.lang.annotation.RetentionPolicy.CLASS;
21     import static javax.lang.model.element.Modifier.FINAL;
22     import static javax.lang.model.element.Modifier.PRIVATE;
23     import static javax.lang.model.element.Modifier.PUBLIC;
24     import static javax.lang.model.element.Modifier.STATIC;
25     
26     import java.lang.annotation.Retention;
27     import java.lang.annotation.Target;
28     
29     /**
30      * Indicates that the annotated method is provided only to be overridden: it should not be
31      * <i>invoked</i> from outside its declaring source file (as if it is {@code private}), and
32      * overriding methods should not be directly invoked at all. Such a method represents a contract
33      * between a class and its <i>subclasses</i> only, and is not to be considered part of the
34      * <i>caller</i>-facing API of either class.
35      *
36      * <p>The annotated method must have protected or package-private visibility, and must not be
37      * {@code static}, {@code final} or declared in a {@code final} class. Overriding methods must have
38      * either protected or package-private visibility, although their effective visibility is actually
39      * "none".
40      */
41     @IncompatibleModifiers({PUBLIC, PRIVATE, STATIC, FINAL})
42     @Retention(CLASS) // Parent source might not be available while compiling subclass
43     @Target(METHOD)
44     public @interface ForOverride {}
