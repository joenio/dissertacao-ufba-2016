17     package com.google.errorprone.bugpatterns;
18     
19     import static com.google.errorprone.BugPattern.Category.JDK;
20     import static com.google.errorprone.BugPattern.MaturityLevel.MATURE;
21     import static com.google.errorprone.BugPattern.SeverityLevel.ERROR;
22     import static com.google.errorprone.matchers.Matchers.instanceMethod;
23     import static com.google.errorprone.predicates.TypePredicates.isArray;
24     
25     import com.google.errorprone.BugPattern;
26     import com.google.errorprone.VisitorState;
27     import com.google.errorprone.bugpatterns.BugChecker.MethodInvocationTreeMatcher;
28     import com.google.errorprone.fixes.Fix;
29     import com.google.errorprone.fixes.SuggestedFix;
30     import com.google.errorprone.matchers.Description;
31     import com.google.errorprone.matchers.Matcher;
32     import com.google.errorprone.util.ASTHelpers;
33     
34     import com.sun.source.tree.ExpressionTree;
35     import com.sun.source.tree.MethodInvocationTree;
36     
37     /**
38      * @author adgar@google.com (Mike Edgar)
39      */
40     @BugPattern(name = "ArrayToString",
41         summary = "Calling toString on an array does not provide useful information",
42         explanation =
43             "The toString method on an array will print its identity, such as [I@4488aabb. This " +
44             "is almost never needed. Use Arrays.toString to print a human-readable array summary.",
45         category = JDK, severity = ERROR, maturity = MATURE)
46     public class ArrayToString extends BugChecker implements MethodInvocationTreeMatcher {
47     
48       /**
49        * Matches calls to Throwable.getStackTrace().
50        */
51       private static final Matcher<ExpressionTree> getStackTraceMatcher =
52           instanceMethod().onDescendantOf("java.lang.Throwable").named("getStackTrace");
53     
54       /**
55        * Matches calls to a toString instance method in which the receiver is an array type.
56        */
57       @Override
58       public Description matchMethodInvocation(MethodInvocationTree methodTree, VisitorState state) {
59         if (!instanceMethod().onClass(isArray()).named("toString").matches(methodTree, state)) {
60           return Description.NO_MATCH;
61         }
62     
63         /*
64          * Fixes instances of calling toString() on an array.  If the array is the result of calling
65          * e.getStackTrace(), replaces e.getStackTrace().toString() with Guava's
66          * Throwables.getStackTraceAsString(e).
67          * Otherwise, replaces a.toString() with Arrays.toString(a).
68          */
69         Fix fix;
70     
71         ExpressionTree receiverTree = ASTHelpers.getReceiver(methodTree);
72         if (receiverTree instanceof MethodInvocationTree &&
73             getStackTraceMatcher.matches(receiverTree, state)) {
74           String throwable = ASTHelpers.getReceiver(receiverTree).toString();
75           fix = SuggestedFix.builder()
76               .replace(methodTree, "Throwables.getStackTraceAsString(" + throwable + ")")
77               .addImport("com.google.common.base.Throwables")
78               .build();
79         } else {
80           fix = SuggestedFix.builder()
81               .replace(methodTree, "Arrays.toString(" + receiverTree + ")")
82               .addImport("java.util.Arrays")
83               .build();
84         }
85         return describeMatch(methodTree, fix);
86       }
87     }
