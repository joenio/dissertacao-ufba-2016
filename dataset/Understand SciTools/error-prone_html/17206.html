18     
19     import java.math.BigInteger;
20     
21     /**
22      * Test data for {@code LabelTemplate}.
23      * 
24      * @author lowasser@google.com (Louis Wasserman)
25      */
26     public class LabelTemplateExample {
27       public void example(BigInteger[] array) {
28         StringBuilder builder = new StringBuilder("[");
29         for (int i = 0; i < array.length; i++) {
30           builder.append(array[i]);
31           if (i == array.length - 1) {
32             break;
33           }
34           builder.append(',');
35         }
36         join: for (int i = 0; i < array.length; i++) {
37           builder.append(array[i]);
38           if (i == array.length - 1) {
39             break join;
40           }
41           builder.append(',');
42         }
43         join: for (int i = 0; i < array.length; i++) {
44           builder.append(array[i]);
45           if (i == array.length - 1) {
46             continue join;
47           }
48           builder.append(',');
49         }
50         System.out.println(builder);
51       }
52     }
