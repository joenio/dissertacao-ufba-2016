17     package com.google.errorprone.refaster;
18     
19     import static com.google.errorprone.refaster.Unifier.unifications;
20     
21     import com.google.auto.value.AutoValue;
22     import com.google.common.collect.ImmutableList;
23     
24     import com.sun.tools.javac.code.Symbol.ClassSymbol;
25     import com.sun.tools.javac.code.Type;
26     import com.sun.tools.javac.code.Type.ClassType;
27     
28     import java.util.List;
29     
30     /**
31      * A representation of a type with optional generic parameters.
32      *
33      * @author Louis Wasserman
34      */
35     @AutoValue
36     public abstract class UClassType extends UType {
37       public static UClassType create(CharSequence fullyQualifiedClass, List<UType> typeArguments) {
38         return new AutoValue_UClassType(
39             StringName.of(fullyQualifiedClass), ImmutableList.copyOf(typeArguments));
40       }
41     
42       public static UClassType create(String fullyQualifiedClass, UType... typeArguments) {
43         return create(fullyQualifiedClass, ImmutableList.copyOf(typeArguments));
44       }
45       
46       abstract StringName fullyQualifiedClass();
47       abstract List<UType> typeArguments();
48     
49       @Override
50       public Choice<Unifier> visitClassType(ClassType classType, Unifier unifier) {
51         return fullyQualifiedClass().unify(classType.tsym.getQualifiedName(), unifier)
52             .thenChoose(unifications(typeArguments(), classType.getTypeArguments()));
53       }
54       
55       @Override
56       public ClassType inline(Inliner inliner) throws CouldNotResolveImportException {
57         ClassSymbol classSymbol = inliner.resolveClass(fullyQualifiedClass());
58         return new ClassType(
59             Type.noType, inliner.<Type>inlineList(typeArguments()), classSymbol);
60       }
61     }
