17     package com.google.errorprone.bugpatterns;
18     
19     import static com.google.errorprone.BugPattern.Category.JDK;
20     import static com.google.errorprone.BugPattern.MaturityLevel.MATURE;
21     import static com.google.errorprone.BugPattern.SeverityLevel.WARNING;
22     
23     import com.google.common.collect.Sets;
24     import com.google.errorprone.BugPattern;
25     import com.google.errorprone.VisitorState;
26     import com.google.errorprone.bugpatterns.BugChecker.SwitchTreeMatcher;
27     import com.google.errorprone.fixes.SuggestedFix;
28     import com.google.errorprone.matchers.Description;
29     import com.google.errorprone.matchers.Description.Builder;
30     import com.google.errorprone.util.ASTHelpers;
31     
32     import com.sun.source.tree.CaseTree;
33     import com.sun.source.tree.ExpressionTree;
34     import com.sun.source.tree.IdentifierTree;
35     import com.sun.source.tree.SwitchTree;
36     import com.sun.tools.javac.code.Symbol.TypeSymbol;
37     import com.sun.tools.javac.tree.JCTree;
38     import com.sun.tools.javac.tree.JCTree.JCSwitch;
39     import com.sun.tools.javac.tree.TreeInfo;
40     
41     import java.util.Iterator;
42     import java.util.LinkedHashSet;
43     import java.util.Set;
44     
45     import javax.lang.model.element.ElementKind;
46     
47     /**
48      * @author cushon@google.com (Liam Miller-Cushon)
49      */
50     @BugPattern(
51       name = "MissingCasesInEnumSwitch",
52       summary = "Enum switch statement is missing cases",
53       explanation =
54           "Enums on switches should either handle all possible values of the enum, or"
55               + " have an explicit default case",
56       category = JDK,
57       severity = WARNING,
58       maturity = MATURE
59     )
60     public class MissingCasesInEnumSwitch extends BugChecker implements SwitchTreeMatcher {
61     
62       @Override
63       public Description matchSwitch(SwitchTree tree, VisitorState state) {
64         TypeSymbol switchType = ((JCSwitch) tree).getExpression().type.tsym;
65     
66         if (switchType.getKind() != ElementKind.ENUM) {
67           return Description.NO_MATCH;
68         }
69     
70         if (hasDefaultCase(tree)) {
71           return Description.NO_MATCH;
72         }
73     
74         Set<String> unhandled =
75             Sets.difference(ASTHelpers.enumValues(switchType), collectEnumSwitchCases(tree));
76         if (unhandled.isEmpty()) {
77           return Description.NO_MATCH;
78         }
79     
80         Description.Builder description = buildDescription(tree).setMessage(buildMessage(unhandled));
81         buildFixes(tree, state, unhandled, description);
82         return description.build();
83       }
84     
85       /** Adds suggested fixes. */
86       private void buildFixes(
87           SwitchTree tree, VisitorState state, Set<String> unhandled, Builder description) {
88     
89         int idx = state.getEndPosition((JCTree) tree) - 1; // preserve closing '}'
90     
91         description.addFix(
92             SuggestedFix.replace(
93                 idx,
94                 idx,
95                 String.format(
96                     "default: throw new AssertionError(\"unexpected case: \" + %s);\n",
97                     state.getSourceForNode(TreeInfo.skipParens((JCTree) tree.getExpression())))));
98     
99         StringBuilder sb = new StringBuilder();
100         for (String label : unhandled) {
101           sb.append(String.format("case %s: ", label));
102         }
103         sb.append("break;\n");
104         description.addFix(SuggestedFix.replace(idx, idx, sb.toString()));
105     
106         description.addFix(SuggestedFix.replace(idx, idx, "default: break;\n"));
107       }
108     
109       /**
110        * Build the diagnostic message.
111        *
112        * <p>Examples:
113        * <ul>
114        * <li>Non-exhaustive switch, expected cases for: FOO
115        * <li>Non-exhaustive switch, expected cases for: FOO, BAR, BAZ, and 42 others.
116        * </ul>
117        */
118       private String buildMessage(Set<String> unhandled) {
119         final int maxCasesToPrint = 5;
120     
121         StringBuilder message = new StringBuilder("Non-exhaustive switch, expected cases for: ");
122     
123         boolean tooManyCasesToPrint = unhandled.size() > maxCasesToPrint;
124         int numberToShow = tooManyCasesToPrint
125             ? 3 // if there are too many to print, only show three examples.
126             : unhandled.size();
127     
128         Iterator<String> it = unhandled.iterator();
129         for (int i = 0; i < numberToShow; ++i) {
130           if (i != 0) {
131             message.append(", ");
132           }
133           message.append(it.next());
134         }
135     
136         if (tooManyCasesToPrint) {
137           message.append(String.format(", and %d others", unhandled.size() - numberToShow));
138         }
139         return message.toString();
140       }
141     
142       /** Return the enum values handled by the given switch statement's cases. */
143       private static LinkedHashSet<String> collectEnumSwitchCases(SwitchTree tree) {
144         LinkedHashSet<String> cases = new LinkedHashSet<>();
145         for (CaseTree caseTree : tree.getCases()) {
146           ExpressionTree pat = caseTree.getExpression();
147           if (pat instanceof IdentifierTree) {
148             cases.add(((IdentifierTree) pat).getName().toString());
149           }
150         }
151         return cases;
152       }
153     
154       /** Return true if the switch has a 'default' case. */
155       private static boolean hasDefaultCase(SwitchTree tree) {
156         for (CaseTree caseTree : tree.getCases()) {
157           if (isDefaultCase(caseTree)) {
158             return true;
159           }
160         }
161         return false;
162       }
163     
164       /** The 'default' case is represented in the AST as a CaseTree with a null expression. */
165       private static boolean isDefaultCase(CaseTree tree) {
166         return tree.getExpression() == null;
167       }
168     }
