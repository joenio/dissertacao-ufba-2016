17     package com.google.errorprone.refaster;
18     
19     import static org.junit.Assert.assertEquals;
20     
21     import com.google.common.testing.EqualsTester;
22     import com.google.common.testing.SerializableTester;
23     
24     import com.sun.source.tree.Tree.Kind;
25     
26     import org.junit.Test;
27     import org.junit.runner.RunWith;
28     import org.junit.runners.JUnit4;
29     
30     /**
31      * Tests for {@link ULiteral}.
32      *
33      * @author lowasser@google.com (Louis Wasserman)
34      */
35     @RunWith(JUnit4.class)
36     public class ULiteralTest extends AbstractUTreeTest {
37       @Test
38       public void nullLiteral() {
39         ULiteral lit = ULiteral.nullLit();
40         assertUnifiesAndInlines("null", lit);
41         assertEquals(Kind.NULL_LITERAL, lit.getKind());
42       }
43       
44       @Test
45       public void stringLiteral() {
46         ULiteral lit = ULiteral.stringLit("foo");
47         assertUnifiesAndInlines("\"foo\"", lit);
48         assertEquals(Kind.STRING_LITERAL, lit.getKind());
49       }
50     
51       @Test
52       public void intLiteral() {
53         ULiteral lit = ULiteral.intLit(123);
54         assertUnifiesAndInlines("123", lit);
55         assertEquals(Kind.INT_LITERAL, lit.getKind());
56       }
57     
58       @Test
59       public void longLiteral() {
60         ULiteral lit = ULiteral.longLit(123L);
61         assertUnifiesAndInlines("123L", lit);
62         assertEquals(Kind.LONG_LITERAL, lit.getKind());
63       }
64     
65       @Test
66       public void charLiteral() {
67         ULiteral lit = ULiteral.create(Kind.CHAR_LITERAL, '%');
68         assertUnifiesAndInlines("'%'", lit);
69         assertEquals(Kind.CHAR_LITERAL, lit.getKind());
70       }
71     
72       @Test
73       public void doubleLiteral() {
74         ULiteral lit = ULiteral.doubleLit(1.23);
75         assertUnifiesAndInlines("1.23", lit);
76         assertEquals(Kind.DOUBLE_LITERAL, lit.getKind());
77       }
78     
79       @Test
80       public void floatLiteral() {
81         ULiteral lit = ULiteral.floatLit(1.23F);
82         assertUnifiesAndInlines("1.23F", lit);
83         assertEquals(Kind.FLOAT_LITERAL, lit.getKind());
84       }
85     
86       @Test
87       public void booleanLiteral() {
88         ULiteral lit = ULiteral.booleanLit(false);
89         assertUnifiesAndInlines("false", lit);
90         assertEquals(Kind.BOOLEAN_LITERAL, lit.getKind());
91       }
92       
93       @Test
94       public void equality() {
95         new EqualsTester()
96             .addEqualityGroup(ULiteral.intLit(1))
97             .addEqualityGroup(ULiteral.intLit(2))
98             .addEqualityGroup(ULiteral.longLit(1L))
99             .addEqualityGroup(ULiteral.doubleLit(1.0))
100             .addEqualityGroup(ULiteral.stringLit("foo"))
101             .addEqualityGroup(ULiteral.nullLit())
102             .addEqualityGroup(ULiteral.charLit('0'))
103             .addEqualityGroup(ULiteral.intLit('0'))
104             .testEquals();
105       }
106       
107       @Test
108       public void serialization() {
109         SerializableTester.reserializeAndAssert(ULiteral.intLit(1));
110         SerializableTester.reserializeAndAssert(ULiteral.longLit(1L));
111         SerializableTester.reserializeAndAssert(ULiteral.doubleLit(1.0));
112         SerializableTester.reserializeAndAssert(ULiteral.floatLit(1.0f));
113         SerializableTester.reserializeAndAssert(ULiteral.stringLit("foo"));
114         SerializableTester.reserializeAndAssert(ULiteral.nullLit());
115         SerializableTester.reserializeAndAssert(ULiteral.booleanLit(true));
116         SerializableTester.reserializeAndAssert(ULiteral.charLit('0'));
117       }
118     }
