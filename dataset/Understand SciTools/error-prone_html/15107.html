19     import com.google.errorprone.bugpatterns.proto.ProtoTest.TestFieldProtoMessage;
20     import com.google.errorprone.bugpatterns.proto.ProtoTest.TestProtoMessage;

22     /**
23      * Positive examples for invalid null comparison of a proto message field.
24      */
25     public class ProtoFieldNullComparisonPositiveCases {
26       public static void main(String[] args) {
27         TestProtoMessage message = TestProtoMessage.newBuilder().build();
28         // BUG: Diagnostic contains: message.hasMessage()
29         if (message.getMessage() != null) {
30           System.out.println("always true");
31         // BUG: Diagnostic contains: !message.hasMessage()
32         } else if (message.getMessage() == null) {
33           System.out.println("impossible");
34         // BUG: Diagnostic contains: message.hasMessage()
35         } else if (null != message.getMessage()) {
36           System.out.println("always true");
37         // BUG: Diagnostic contains: message.getMessage().hasField()
38         } else if (message.getMessage().getField() != null) {
39           System.out.println("always true");
40         // BUG: Diagnostic contains: !message.getMultiFieldList().isEmpty()
41         } else if (message.getMultiFieldList() != null) {
42           System.out.println("always true");
43         // BUG: Diagnostic contains: message.getMultiFieldList().isEmpty()
44         } else if (null == message.getMultiFieldList()) {
45           System.out.println("impossible");
46         }
47       }
48     }
