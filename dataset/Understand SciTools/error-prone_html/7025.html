15     package com.google.errorprone.analysis;
16     
17     import com.google.auto.value.AutoValue;
18     import com.google.common.collect.ImmutableList;
19     import com.google.common.collect.ImmutableSet;
20     import com.google.errorprone.DescriptionListener;
21     
22     import com.sun.source.tree.CompilationUnitTree;
23     import com.sun.tools.javac.util.Context;
24     
25     import java.util.Set;
26     
27     /**
28      * Represents a combination of multiple top-level analyses.
29      *
30      * @author Louis Wasserman
31      */
32     @AutoValue
33     public abstract class SumTopLevelAnalysis implements TopLevelAnalysis {
34       public static SumTopLevelAnalysis create(Iterable<? extends TopLevelAnalysis> analyses) {
35         return new AutoValue_SumTopLevelAnalysis(ImmutableList.copyOf(analyses));
36       }
37     
38       SumTopLevelAnalysis() {}
39     
40       abstract ImmutableList<TopLevelAnalysis> analyses();
41     
42       @Override
43       public Set<String> knownAnalysisNames() {
44         ImmutableSet.Builder<String> builder = ImmutableSet.builder();
45         for (TopLevelAnalysis analysis : analyses()) {
46           builder.addAll(analysis.knownAnalysisNames());
47         }
48         return builder.build();
49       }
50     
51       @Override
52       public void analyze(CompilationUnitTree compilationUnit, Context context,
53           AnalysesConfig config, DescriptionListener listener) {
54         for (TopLevelAnalysis analysis : analyses()) {
55           analysis.analyze(compilationUnit, context, config, listener);
56         }
57       }
58     }
