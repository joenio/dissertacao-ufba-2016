19     import static com.google.common.base.Preconditions.checkArgument;
20     
21     import com.google.common.collect.ImmutableList;
22     
23     import com.sun.tools.javac.parser.JavaTokenizer;
24     import com.sun.tools.javac.parser.Scanner;
25     import com.sun.tools.javac.parser.ScannerFactory;
26     import com.sun.tools.javac.parser.Tokens.Comment;
27     import com.sun.tools.javac.parser.Tokens.Comment.CommentStyle;
28     import com.sun.tools.javac.parser.Tokens.TokenKind;
29     import com.sun.tools.javac.parser.UnicodeReader;
30     import com.sun.tools.javac.util.Context;

32     /** A utility for tokenizing and preserving comments. */
33     public class ErrorProneTokens {
34     
35       /** Returns the tokens for the given source text, including comments. */
36       public static ImmutableList<ErrorProneToken> getTokens(String source, Context context) {
37         if (source == null) {
38           return ImmutableList.of();
39         }
40         ScannerFactory fac = ScannerFactory.instance(context);
41         char[] buffer = source.toCharArray();
42         Scanner scanner =
43             new AccessibleScanner(fac, new CommentSavingTokenizer(fac, buffer, buffer.length));
44         ImmutableList.Builder<ErrorProneToken> tokens = ImmutableList.builder();
45         while (scanner.token().kind != TokenKind.EOF) {
46           tokens.add(new ErrorProneToken(scanner.token()));
47           scanner.nextToken();
48         }
49         return tokens.build();
50       }
51     
52       /** A {@link JavaTokenizer} that saves comments. */
53       static class CommentSavingTokenizer extends JavaTokenizer {
54         CommentSavingTokenizer(ScannerFactory fac, char[] buffer, int length) {
55           super(fac, buffer, length);
56         }
57     
58         @Override
59         protected Comment processComment(int pos, int endPos, CommentStyle style) {
60           char[] buf = reader.getRawCharacters(pos, endPos);
61           return new CommentWithTextAndPosition(
62               pos, endPos, new AccessibleReader(fac, buf, buf.length), style);
63         }
64       }
65     
66       /** A {@link Comment} that saves its text and start position. */
67       static class CommentWithTextAndPosition implements Comment {
68     
69         private final int pos;
70         private final int endPos;
71         private final AccessibleReader reader;
72         private final CommentStyle style;
73     
74         private String text = null;
75     
76         public CommentWithTextAndPosition(
77             int pos, int endPos, AccessibleReader reader, CommentStyle style) {
78           this.pos = pos;
79           this.endPos = endPos;
80           this.reader = reader;
81           this.style = style;
82         }
83     
84         /**
85          * Returns the source position of the character at index {@code index} in the comment text.
86          *
87          * <p>The handling of javadoc comments in javac has more logic to skip over leading
88          * whitespace and '*' characters when indexing into doc comments, but we don't need any
89          * of that.
90          */
91         @Override
92         public int getSourcePos(int index) {
93           checkArgument(
94               0 <= index && index < (endPos - pos),
95               "Expected %s in the range [0, %s)",
96               index,
97               endPos - pos);
98           return pos + index;
99         }
100     
101         @Override
102         public CommentStyle getStyle() {
103           return style;
104         }
105     
106         @Override
107         public String getText() {
108           String text = this.text;
109           if (text == null) {
110             this.text = text = new String(reader.getRawCharacters());
111           }
112           return text;
113         }
114     
115         /**
116          * We don't care about {@code @deprecated} javadoc tags (see the DepAnn check).
117          *
118          * @return false
119          */
120         @Override
121         public boolean isDeprecated() {
122           return false;
123         }
124     
125         @Override
126         public String toString() {
127           return String.format("Comment: '%s'", getText());
128         }
129       }
130     
131       // Scanner(ScannerFactory, JavaTokenizer) is package-private
132       static class AccessibleScanner extends Scanner {
133         protected AccessibleScanner(ScannerFactory fac, JavaTokenizer tokenizer) {
134           super(fac, tokenizer);
135         }
136       }
137     
138       // UnicodeReader(ScannerFactory, char[], int) is package-private
139       static class AccessibleReader extends UnicodeReader {
140         protected AccessibleReader(ScannerFactory fac, char[] buffer, int length) {
141           super(fac, buffer, length);
142         }
143       }
144     }
