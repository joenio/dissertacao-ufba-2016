32     import com.sun.tools.javac.tree.JCTree.JCConditional;
33     import com.sun.tools.javac.tree.JCTree.JCExpression;

37     import javax.annotation.Nullable;

39     /**
40      * {@link UTree} version of {@link UnaryTree}.
41      *
42      * @author lowasser@google.com (Louis Wasserman)
43      */
44     @AutoValue
45     abstract class UUnary extends UExpression implements UnaryTree {
46       private static final ImmutableBiMap<Kind, JCTree.Tag> UNARY_OP_CODES =
47           new ImmutableBiMap.Builder<Kind, JCTree.Tag>()
48             .put(Kind.PREFIX_INCREMENT, JCTree.Tag.PREINC)
49             .put(Kind.PREFIX_DECREMENT, JCTree.Tag.PREDEC)
50             .put(Kind.POSTFIX_INCREMENT, JCTree.Tag.POSTINC)
51             .put(Kind.POSTFIX_DECREMENT, JCTree.Tag.POSTDEC)
52             .put(Kind.UNARY_PLUS, JCTree.Tag.POS)
53             .put(Kind.UNARY_MINUS, JCTree.Tag.NEG)
54             .put(Kind.BITWISE_COMPLEMENT, JCTree.Tag.COMPL)
55             .put(Kind.LOGICAL_COMPLEMENT, JCTree.Tag.NOT)
56             .build();
57     
58       public static UUnary create(Kind unaryOp, UExpression expression) {
59         checkArgument(UNARY_OP_CODES.containsKey(unaryOp), 
60             "%s is not a recognized unary operation", unaryOp);
61         return new AutoValue_UUnary(unaryOp, expression);
62       }
63     
64       @Override
65       public abstract Kind getKind();
66     
67       @Override
68       public abstract UExpression getExpression();
69     
70       @Override
71       @Nullable
72       public Choice<Unifier> visitUnary(UnaryTree unary, @Nullable Unifier unifier) {
73         return Choice.condition(getKind().equals(unary.getKind()), unifier)
74             .thenChoose(unifications(getExpression(), UParens.skipParens(unary.getExpression())));
75       }
76     
77       @Override
78       public <R, D> R accept(TreeVisitor<R, D> visitor, D data) {
79         return visitor.visitUnary(this, data);
80       }
81     
82       @Override
83       public JCExpression inline(Inliner inliner) throws CouldNotResolveImportException {
84         JCExpression expr = getExpression().inline(inliner);
85         final TreeMaker maker = inliner.maker();
86         if (getKind() == Kind.LOGICAL_COMPLEMENT) {
87           return new TreeCopier<Void>(maker) {
88             @SuppressWarnings("unchecked")
89                 // essentially depends on T being a superclass of JCExpression
90             @Override public <T extends JCTree> T copy(T t, Void v) {
91               if (t instanceof BinaryTree || t instanceof UnaryTree
92                   || t instanceof ConditionalExpressionTree) {
93                 return super.copy(t, v);
94               } else {
95                 return (T) defaultNegation(t);
96               }
97             }
98     
99             public JCExpression defaultNegation(Tree expr) {
100               return maker.Unary(JCTree.Tag.NOT, (JCExpression) expr);
101             }
102     
103             @Override public JCExpression visitBinary(BinaryTree tree, Void v) {
104               if (UBinary.DEMORGAN.containsKey(tree.getKind())) {
105                 JCExpression negLeft = copy((JCExpression) tree.getLeftOperand());
106                 JCExpression negRight = copy((JCExpression) tree.getRightOperand());
107                 return maker.Binary(
108                     UBinary.OP_CODES.get(UBinary.DEMORGAN.get(tree.getKind())), negLeft, negRight);
109               } else if (UBinary.NEGATION.containsKey(tree.getKind())) {
110                 JCExpression left = (JCExpression) tree.getLeftOperand();
111                 JCExpression right = (JCExpression) tree.getRightOperand();
112                 return maker.Binary(
113                     UBinary.OP_CODES.get(UBinary.NEGATION.get(tree.getKind())), left, right);
114               } else {
115                 return defaultNegation(tree);
116               }
117             }
118     
119             @Override public JCExpression visitUnary(UnaryTree tree, Void v) {
120               if (tree.getKind() == Kind.LOGICAL_COMPLEMENT) {
121                 return (JCExpression) tree.getExpression();
122               } else {
123                 return defaultNegation(tree);
124               }
125             }
126     
127             @Override public JCConditional visitConditionalExpression(
128                 ConditionalExpressionTree tree, Void v) {
129               return maker.Conditional(
130                   (JCExpression) tree.getCondition(),
131                   copy((JCExpression) tree.getTrueExpression()),
132                   copy((JCExpression) tree.getFalseExpression()));
133             }
134           }.copy(expr);
135         } else {
136           return inliner.maker().Unary(UNARY_OP_CODES.get(getKind()), getExpression().inline(inliner));
137         }
138       }
139       
140       @Override
141       public UExpression negate() {
142         return (getKind() == Kind.LOGICAL_COMPLEMENT) ? getExpression() : super.negate();
143       }
144     }
