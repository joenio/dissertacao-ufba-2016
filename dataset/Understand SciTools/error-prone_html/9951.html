17     package com.google.errorprone.bugpatterns;
18     
19     import static com.google.errorprone.BugPattern.Category.DAGGER;
20     import static com.google.errorprone.BugPattern.MaturityLevel.MATURE;
21     import static com.google.errorprone.BugPattern.SeverityLevel.ERROR;

23     import com.google.errorprone.BugPattern;

37     
38     import javax.lang.model.element.AnnotationMirror;

40     /**
41      * Bug checker for null-returning methods annotated with {@code @Provides} but not
42      * {@code @Nullable}.
43      */
44     @BugPattern(
45       name = "DaggerProvidesNull",
46       summary = "Dagger @Provides methods may not return null unless annotated with @Nullable",
47       explanation =
48           "Dagger `@Provides` methods may not return null unless annotated with `@Nullable`. "
49               + "Such a method will cause a `NullPointerException` at runtime if the `return null` "
50               + "path is ever taken.\n\n"
51               + "If you believe the `return null` path can never be taken, please throw a "
52               + "`RuntimeException` instead. Otherwise, please annotate the method with `@Nullable`.",
53       category = DAGGER,
54       maturity = MATURE,
55       severity = ERROR
56     )
57     public class DaggerProvidesNull extends BugChecker implements ReturnTreeMatcher {
58     
59       /**
60        * Matches explicit "return null" statements in methods annotated with {@code @Provides} but not
61        * {@code @Nullable}.  Suggests either annotating the method with {@code @Nullable} or throwing a
62        * {@link RuntimeException} instead.
63        */
64       // TODO(eaftan): Use nullness dataflow analysis when it's ready
65       @Override
66       public Description matchReturn(ReturnTree returnTree, VisitorState state) {
67         ExpressionTree returnExpression = returnTree.getExpression();
68         if (returnExpression == null || returnExpression.getKind() != Kind.NULL_LITERAL) {
69           return Description.NO_MATCH;
70         }
71     
72         MethodTree enclosingMethod = ASTHelpers.findEnclosingNode(state.getPath(), MethodTree.class);
73         if (enclosingMethod == null) {
74           return Description.NO_MATCH;
75         }
76         MethodSymbol enclosingMethodSym = ASTHelpers.getSymbol(enclosingMethod);
77         if (enclosingMethodSym == null) {
78           return Description.NO_MATCH;
79         }
80     
81         if (!ASTHelpers.hasAnnotation(enclosingMethodSym, "dagger.Provides", state)
82             || hasAnyNullableAnnotation(enclosingMethodSym)) {
83           return Description.NO_MATCH;
84         }
85     
86         Fix addNullableFix =
87             SuggestedFix.builder()
88             .prefixWith(enclosingMethod, "@Nullable\n")
89             .addImport("javax.annotation.Nullable")
90             .build();
91     
92         CatchTree enclosingCatch = ASTHelpers.findEnclosingNode(state.getPath(), CatchTree.class);
93         if (enclosingCatch == null) {
94           // If not in a catch block, suggest adding @Nullable first, then throwing an exception.
95           Fix throwRuntimeExceptionFix =
96               SuggestedFix.replace(returnTree, "throw new RuntimeException();");
97           return buildDescription(returnTree)
98               .addFix(addNullableFix)
99               .addFix(throwRuntimeExceptionFix)
100               .build();
101         } else {
102           // If in a catch block, suggest throwing an exception first, then adding @Nullable.
103           String replacement = String.format(
104               "throw new RuntimeException(%s);",
105               enclosingCatch.getParameter().getName());
106           Fix throwRuntimeExceptionFix =
107               SuggestedFix.replace(returnTree, replacement);
108           return buildDescription(returnTree)
109               .addFix(throwRuntimeExceptionFix)
110               .addFix(addNullableFix)
111               .build();
112         }
113       }
114     
115       /**
116        * Returns true iff this method is directly annotated with <em>any</em> annotation with the simple
117        * name "Nullable".
118        */
119       private static boolean hasAnyNullableAnnotation(MethodSymbol methodSym) {
120         for (AnnotationMirror annotation : methodSym.getAnnotationMirrors()) {
121           if (annotation.getAnnotationType().asElement().getSimpleName().contentEquals("Nullable")) {
122             return true;
123           }
124         }
125         return false;
126       }
127     }
