85     @BugPattern(name = "MissingFail",
86         altNames = "missing-fail",
87         summary = "Not calling fail() when expecting an exception masks bugs",
88         category = JUNIT,
89         maturity = EXPERIMENTAL,
90         severity = WARNING)
91     public class MissingFail extends BugChecker implements TryTreeMatcher {

93       // Many test writers don't seem to know about `fail()`. They instead use synonyms of varying
94       // complexity instead.
95       //
96       // One category of synonyms replaces `fail()` with a single, equivalent statement
97       // such as `assertTrue(false)`, `assertFalse(true)` or `throw new
98       // AssertionError()`. In these cases we will simply skip any further analysis, they
99       // work perfectly fine.
100       //
101       // Other, more complex synonyms tend to use boolean variables, like such:
102       //
103       // ```java
104       // boolean thrown = false;
105       // try {
106       //   throwingExpression();
107       // } catch (SomeException e) {
108       //   thrown = true;
109       // }
110       // assertTrue(thrown);
111       // ```
112     
113       private static final Matcher<ExpressionTree> ASSERT_EQUALS = Matchers.anyOf(
114           staticMethod().onClass("org.junit.Assert").named("assertEquals"),
115           staticMethod().onClass("junit.framework.Assert").named("assertEquals"),
116           staticMethod().onClass("junit.framework.TestCase").named("assertEquals"));
117       private static final Matcher<Tree> ASSERT_UNEQUAL =
118           toType(MethodInvocationTree.class, new UnequalIntegerLiteralMatcher(ASSERT_EQUALS));
119     
120       private static final Matcher<ExpressionTree> ASSERT_TRUE = Matchers.anyOf(
121           staticMethod().onClass("org.junit.Assert").named("assertTrue"),
122           staticMethod().onClass("junit.framework.Assert").named("assertTrue"),
123           staticMethod().onClass("junit.framework.TestCase").named("assertTrue"));
124       private static final Matcher<ExpressionTree> ASSERT_FALSE = Matchers.anyOf(
125           staticMethod().onClass("org.junit.Assert").named("assertFalse"),
126           staticMethod().onClass("junit.framework.Assert").named("assertFalse"),
127           staticMethod().onClass("junit.framework.TestCase").named("assertFalse"));
128       private static final Matcher<ExpressionTree> ASSERT_TRUE_FALSE =
129           methodInvocation(ASSERT_TRUE, MatchType.AT_LEAST_ONE,
130               Matchers.anyOf(booleanLiteral(false), booleanConstant(false)));
131       private static final Matcher<ExpressionTree> ASSERT_FALSE_TRUE =
132           methodInvocation(ASSERT_FALSE, MatchType.AT_LEAST_ONE, Matchers.anyOf(
133               booleanLiteral(true), booleanConstant(true)));
134       private static final Matcher<ExpressionTree> ASSERT_TRUE_TRUE =
135           methodInvocation(ASSERT_TRUE, MatchType.AT_LEAST_ONE, Matchers.anyOf(
136               booleanLiteral(true), booleanConstant(true)));
137       private static final Matcher<ExpressionTree> ASSERT_FALSE_FALSE =
138           methodInvocation(ASSERT_FALSE, MatchType.AT_LEAST_ONE, Matchers.anyOf(
139               booleanLiteral(false), booleanConstant(false)));

141       private static final Matcher<StatementTree> JAVA_ASSERT_FALSE =
142           assertStatement(ignoreParens(Matchers.anyOf(booleanLiteral(false), booleanConstant(false))));
143     
144       private static final Matcher<ExpressionTree> LOG_CALL = methodInvocation(new LogMethodMatcher());
145       private static final Matcher<Tree> LOG_IN_BLOCK =
146           contains(toType(ExpressionTree.class, LOG_CALL));

148       private static final Pattern FAIL_PATTERN = Pattern.compile(".*(?i:fail).*");
149       private static final Matcher<ExpressionTree> FAIL =
150           anyMethod().anyClass().withNameMatching(FAIL_PATTERN);

152       private static final Matcher<ExpressionTree> ASSERT_CALL =
153           methodInvocation(new AssertMethodMatcher());
154       private static final Matcher<ExpressionTree> REAL_ASSERT_CALL = Matchers.allOf(
155           ASSERT_CALL,
156           Matchers.not(Matchers.anyOf(ASSERT_FALSE_FALSE, ASSERT_TRUE_TRUE)));
157       private static final Matcher<ExpressionTree> VERIFY_CALL =
158           methodInvocation(staticMethod().onClass("org.mockito.Mockito").named("verify"));
159       private static final MultiMatcher<TryTree, Tree> ASSERT_LAST_CALL_IN_TRY =
160           new ChildOfTryMatcher(
161               MatchType.LAST,
162               contains(toType(ExpressionTree.class, Matchers.anyOf(REAL_ASSERT_CALL, VERIFY_CALL))));
163       private static final Matcher<Tree> ASSERT_IN_BLOCK =
164           contains(toType(ExpressionTree.class, REAL_ASSERT_CALL));

166       private static final Matcher<StatementTree> THROW_STATEMENT = throwStatement(Matchers.anything());
167       private static final Matcher<Tree> THROW_OR_FAIL_IN_BLOCK =
168           contains(
169               Matchers.anyOf(
170                   toType(StatementTree.class, THROW_STATEMENT),
171                   // TODO(schmitt): Include Preconditions.checkState(false)?
172                   toType(ExpressionTree.class, ASSERT_TRUE_FALSE),
173                   toType(ExpressionTree.class, ASSERT_FALSE_TRUE),
174                   toType(ExpressionTree.class, ASSERT_UNEQUAL),
175                   toType(StatementTree.class, JAVA_ASSERT_FALSE),
176                   toType(ExpressionTree.class, FAIL)));

187           isSameType("java.lang.AssertionError"),

211       // Subtly different from JUnitMatchers: We want to match test base classes too.
212       private static final Matcher<ClassTree> JUNIT3_TEST_CLASS =
213           isSubtypeOf("junit.framework.TestCase");
214       private static final Matcher<ClassTree> TEST_CLASS = Matchers.anyOf(JUNIT3_TEST_CLASS,
215           new JUnitMatchers.JUnit4TestClassMatcher());

217       @Override
218       public Description matchTry(TryTree tree, VisitorState state) {
219         if (tryTreeMatches(tree, state)) {
220           List<? extends StatementTree> tryStatements = tree.getBlock().getStatements();
221           StatementTree lastTryStatement = tryStatements.get(tryStatements.size() - 1);
222     
223           String failCall = String.format("%nfail(\"Expected %s\");", exceptionToString(tree));
224           SuggestedFix.Builder fixBuilder = SuggestedFix.builder()
225               .postfixWith(lastTryStatement, failCall);
226     
227           // Make sure that when the fail import is added it doesn't conflict with existing ones.
228           fixBuilder.removeStaticImport("junit.framework.Assert.fail");
229           fixBuilder.removeStaticImport("junit.framework.TestCase.fail");
230           fixBuilder.addStaticImport("org.junit.Assert.fail");
231     
232           return describeMatch(lastTryStatement, fixBuilder.build());
233         } else {
234           return Description.NO_MATCH;
235         }
236       }

238       /**
239        * Returns a string describing the exception type caught by the given try tree's catch
240        * statement(s), defaulting to {@code "Exception"} if more than one exception type is caught.
241        */
242       private String exceptionToString(TryTree tree) {
243         if (tree.getCatches().size() != 1) {
244           return "Exception";
245         }
246         String exceptionType = tree.getCatches().iterator().next().getParameter().getType().toString();
247         if (exceptionType.contains("|")) {
248           return "Exception";
249         }
250         return exceptionType;
251       }

253       private boolean tryTreeMatches(TryTree tree, VisitorState state) {
254         if (!isInClass(tree, state, TEST_CLASS)) {
255           return false;
256         }
257     
258         boolean assertInCatch = hasAssertInCatch(tree, state);
259         if (!hasExpectedException(tree) && !assertInCatch) {
260           return false;
261         }
262     
263         if (hasThrowOrFail(tree, state) || isInInapplicableMethod(tree, state)
264             || returnsInTryCatchOrAfter(tree, state) || isInapplicableExceptionType(tree, state)
265             || isInLoop(state, tree) || hasWhileTrue(tree, state) || hasContinue(tree, state)
266             || hasFinally(tree) || logsInCatch(state, tree)) {
267           return false;
268         }
269     
270         if (assertInCatch
271             && (hasFieldAssignmentInCatch(tree, state)
272                 || hasBooleanAssertVariableInCatch(tree, state)
273                 || lastTryStatementIsAssert(tree, state))) {
274           return false;
275         }
276     
277         return true;
278       }

280       private boolean hasWhileTrue(TryTree tree, VisitorState state) {
281         return WHILE_TRUE_IN_BLOCK.matches(tree, state);

284       private boolean isInClass(TryTree tree, VisitorState state, Matcher<ClassTree> classTree) {
285         return Matchers.enclosingNode(toType(ClassTree.class, classTree)).matches(tree, state);
286       }

288       private boolean hasBooleanAssertVariableInCatch(TryTree tree, VisitorState state) {
289         return anyCatchBlockMatches(tree, state, BOOLEAN_ASSERT_VAR_IN_BLOCK);
290       }
291     
292       private boolean lastTryStatementIsAssert(TryTree tree, VisitorState state) {
293         return ASSERT_LAST_CALL_IN_TRY.matches(tree, state);
294       }
295     
296       private boolean hasFieldAssignmentInCatch(TryTree tree, VisitorState state) {
297         return anyCatchBlockMatches(tree, state, FIELD_ASSIGNMENT_IN_BLOCK);
298       }
299     
300       private boolean logsInCatch(VisitorState state, TryTree tree) {
301         return anyCatchBlockMatches(tree, state, LOG_IN_BLOCK);
302       }

304       private boolean hasFinally(TryTree tree) {
305         return tree.getFinallyBlock() != null;
306       }
307     
308       private boolean hasContinue(TryTree tree, VisitorState state) {
309         return CONTINUE_IN_BLOCK.matches(tree, state);
310       }

312       private boolean isInLoop(VisitorState state, TryTree tree) {
313         return IN_LOOP.matches(tree, state);
314       }

316       private boolean isInapplicableExceptionType(TryTree tree, VisitorState state) {
317         for (CatchTree catchTree : tree.getCatches()) {
318           if (INAPPLICABLE_EXCEPTION.matches(catchTree.getParameter(), state)) {
319             return true;
320           }
321         }
322         return false;
323       }
324     
325       private boolean returnsInTryCatchOrAfter(TryTree tree, VisitorState state) {
326         return RETURN_IN_BLOCK.matches(tree, state) || RETURN_AFTER.matches(tree, state);
327       }
328     
329       private boolean isInInapplicableMethod(TryTree tree, VisitorState state) {
330         return NON_TEST_METHOD.matches(tree, state);
331       }
332     
333       private boolean hasThrowOrFail(TryTree tree, VisitorState state) {
334         return THROW_OR_FAIL_IN_BLOCK.matches(tree, state);
335       }
336     
337       private boolean hasAssertInCatch(TryTree tree, VisitorState state) {
338         return anyCatchBlockMatches(tree, state, ASSERT_IN_BLOCK);
339       }
340     
341       private boolean hasExpectedException(TryTree tree) {
342         for (CatchTree catchTree : tree.getCatches()) {
343           if (catchTree.getParameter().getName().contentEquals("expected")) {
344             return true;
345           }
346         }
347         return false;
348       }
349     
350       private boolean anyCatchBlockMatches(TryTree tree, VisitorState state, Matcher<Tree> matcher) {
351         for (CatchTree catchTree : tree.getCatches()) {
352           if (matcher.matches(catchTree.getBlock(), state)) {
353             return true;
354           }
355         }
356         return false;
357       }

359       private static class AssertMethodMatcher implements Matcher<ExpressionTree> {
360     
361         @Override
362         public boolean matches(ExpressionTree expressionTree, VisitorState state) {
363           Symbol sym = ASTHelpers.getSymbol(expressionTree);
364           return sym != null && sym.getSimpleName().toString().startsWith("assert");
365         }
366       }

368       private static class LogMethodMatcher implements Matcher<ExpressionTree> {
369     
370         @Override
371         public boolean matches(ExpressionTree expressionTree, VisitorState state) {
372           Symbol sym = ASTHelpers.getSymbol(expressionTree);
373           if (sym != null && sym.getSimpleName().toString().startsWith("log")) {
374             return true;
375           }
376     
377           if (sym != null && sym.isStatic()) {
378             if (sym.owner.getQualifiedName().toString().contains("Logger")) {
379               return true;
380             }
381           } else if (expressionTree instanceof MemberSelectTree) {
382             if (((MemberSelectTree) expressionTree).getExpression().toString().startsWith("log")) {
383               return true;
384             }
385           }
386     
387           return false;
388         }
389       }

414       /**
415        * Matches any try-tree that is in a JUNit3 {@code setUp} or {@code tearDown} method, their JUnit4
416        * equivalents, a JUnit {@code suite()} method or a {@code main} method.
417        */
418       private static class IgnoredEnclosingMethodMatcher implements Matcher<TryTree> {
419     
420         @Override
421         public boolean matches(TryTree tryTree, VisitorState state) {
422           MethodTree enclosingMethodTree =
423               ASTHelpers.findEnclosingNode(state.getPath(), MethodTree.class);
424     
425           Name name = enclosingMethodTree.getName();
426           return JUnitMatchers.looksLikeJUnit3SetUp.matches(enclosingMethodTree, state)
427               || JUnitMatchers.looksLikeJUnit3TearDown.matches(enclosingMethodTree, state)
428               || name.contentEquals("main")
429               // TODO(schmitt): Move to JUnitMatchers?
430               || name.contentEquals("suite")
431               || Matchers.hasAnnotation(JUNIT_BEFORE_ANNOTATION).matches(enclosingMethodTree, state)
432               || Matchers.hasAnnotation(JUNIT_AFTER_ANNOTATION).matches(enclosingMethodTree, state);
433         }
434       }

436       /**
437        * Matches if any two of the given list of expressions are integer literals with different values.
438        */
439       private static class UnequalIntegerLiteralMatcher
440           implements MultiMatcher<MethodInvocationTree, ExpressionTree> {
441     
442         private final Matcher<ExpressionTree> methodSelectMatcher;
443     
444         private UnequalIntegerLiteralMatcher(Matcher<ExpressionTree> methodSelectMatcher) {
445           this.methodSelectMatcher = methodSelectMatcher;
446         }
447     
448         @Override
449         public boolean matches(MethodInvocationTree methodInvocationTree, VisitorState state) {
450           return methodSelectMatcher.matches(methodInvocationTree, state)
451               && matches(methodInvocationTree.getArguments());
452         }
453     
454         private boolean matches(List<? extends ExpressionTree> expressionTrees) {
455           Set<Integer> foundValues = new HashSet<>();
456           for (Tree tree : expressionTrees) {
457             if (tree instanceof LiteralTree) {
458               Object value = ((LiteralTree) tree).getValue();
459               if (value instanceof Integer) {
460                 boolean duplicate = !foundValues.add((Integer) value);
461                 if (!duplicate && foundValues.size() > 1) {
462                   return true;
463                 }
464               }
465             }
466           }
467           return false;
468         }
469     
470         @Override
471         public ExpressionTree getMatchingNode() {
472           throw new UnsupportedOperationException();
473         }
474       }
