39     /**
40      * {@code UTree} representation of a {@code ClassTree} for anonymous inner class matching.
41      * 
42      * @author lowasser@google.com (Louis Wasserman)
43      */
44     @AutoValue
45     abstract class UClassDecl extends USimpleStatement implements ClassTree {
46       public static UClassDecl create(UMethodDecl... members) {
47         return create(ImmutableList.copyOf(members));
48       }
49       
50       public static UClassDecl create(Iterable<UMethodDecl> members) {
51         return new AutoValue_UClassDecl(ImmutableList.copyOf(members));
52       }
53       
54       @AutoValue
55       abstract static class UnifierWithRemainingMembers {
56         static UnifierWithRemainingMembers create(
57             Unifier unifier, Iterable<UMethodDecl> remainingMembers) {
58           return new AutoValue_UClassDecl_UnifierWithRemainingMembers(unifier,
59               ImmutableList.copyOf(remainingMembers));
60         }
61         
62         abstract Unifier unifier();
63         abstract ImmutableList<UMethodDecl> remainingMembers();
64         
65         static final Function<Unifier, UnifierWithRemainingMembers> withRemaining(
66             final Iterable<UMethodDecl> remainingMembers) {
67           return new Function<Unifier, UnifierWithRemainingMembers>() {
68             @Override
69             public UnifierWithRemainingMembers apply(Unifier unifier) {
70               return create(unifier, remainingMembers);
71             }
72           };
73         }
74       }
75       
76       private static Function<UnifierWithRemainingMembers, Choice<UnifierWithRemainingMembers>>
77           match(final Tree tree) {
78         return new Function<UnifierWithRemainingMembers, Choice<UnifierWithRemainingMembers>>() {
79           @Override
80           public Choice<UnifierWithRemainingMembers> apply(final UnifierWithRemainingMembers state) {
81             final ImmutableList<UMethodDecl> currentMembers = state.remainingMembers();
82             Choice<Integer> methodChoice = Choice.from(ContiguousSet.create(
83                 Range.closedOpen(0, currentMembers.size()), DiscreteDomain.integers()));
84             return methodChoice.thenChoose(
85                 new Function<Integer, Choice<UnifierWithRemainingMembers>>() {
86                   @Override
87                   public Choice<UnifierWithRemainingMembers> apply(Integer i) {
88                     ImmutableList<UMethodDecl> remainingMembers = 
89                         new ImmutableList.Builder<UMethodDecl>()
90                             .addAll(currentMembers.subList(0, i))
91                             .addAll(currentMembers.subList(i + 1, currentMembers.size()))
92                             .build();
93                     UMethodDecl chosenMethod = currentMembers.get(i);
94                     Unifier unifier = state.unifier().fork();
95                     /* 
96                      * If multiple methods use the same parameter name, preserve the last parameter
97                      * name from the target code.  For example, given a @BeforeTemplate with
98                      * 
99                      *    int get(int index) {...}
100                      *    int set(int index, int value) {...}
101                      *    
102                      * and target code with the lines
103                      * 
104                      *    int get(int i) {...}
105                      *    int set(int j) {...}
106                      *    
107                      * then use "j" in place of index in the @AfterTemplates.
108                      */
109                     for (UVariableDecl param : chosenMethod.getParameters()) {
110                       unifier.clearBinding(param.key());
111                     }
112                     return chosenMethod.unify(tree, unifier)
113                         .transform(UnifierWithRemainingMembers.withRemaining(remainingMembers));
114                   }
115                 });
116           }
117         };
118       }
119     
120       @Override
121       public Choice<Unifier> visitClass(ClassTree node, Unifier unifier) {
122         Choice<UnifierWithRemainingMembers> path = Choice.of(
123             UnifierWithRemainingMembers.create(unifier, getMembers()));
124         for (Tree targetMember : node.getMembers()) {
125           if (!(targetMember instanceof MethodTree)
126               || ((MethodTree) targetMember).getReturnType() != null) {
127             // skip synthetic constructors
128             path = path.thenChoose(match(targetMember));
129           }
130         }
131         return path.thenOption(new Function<UnifierWithRemainingMembers, Optional<Unifier>>() {
132           @Override
133           public Optional<Unifier> apply(UnifierWithRemainingMembers state) {
134             if (state.remainingMembers().isEmpty()) {
135               return Optional.of(state.unifier());
136             } else {
137               return Optional.absent();
138             }
139           }});
140       }
141     
142       @Override
143       public JCClassDecl inline(Inliner inliner) throws CouldNotResolveImportException {
144         return inliner.maker().AnonymousClassDef(
145             inliner.maker().Modifiers(0L),
146             List.convert(JCTree.class, inliner.inlineList(getMembers())));
147       }
148       
149       @Override
150       public <R, D> R accept(TreeVisitor<R, D> visitor, D data) {
151         return visitor.visitClass(this, data);
152       }
153     
154       @Override
155       public Kind getKind() {
156         return Kind.CLASS;
157       }
158     
159       @Override
160       public UTree<?> getExtendsClause() {
161         return null;
162       }
163     
164       @Override
165       public ImmutableList<UTree<?>> getImplementsClause() {
166         return ImmutableList.of();
167       }
168     
169       @Override
170       public abstract ImmutableList<UMethodDecl> getMembers();
171     
172       @Override
173       public ModifiersTree getModifiers() {
174         return null;
175       }
176     
177       @Override
178       public Name getSimpleName() {
179         return null;
180       }
181

183       public ImmutableList<TypeParameterTree> getTypeParameters() {
184         return ImmutableList.of();
185       }
