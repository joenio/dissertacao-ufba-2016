46     /**
47      * Verifies that methods marked {@link com.google.errorprone.annotations.ForOverride} are only
48      * called from the defining class.
49      *
50      * <p>Specifically, all calls to the method have to occur within the context of the outermost class
51      * where the method is defined.
52      */
53     @BugPattern(name = "ForOverride",
54         summary =
55             "Method annotated @ForOverride must be protected or package-private and only invoked from "
56             + "declaring class",
57         explanation =
58             "A method that overrides a @ForOverride method should not be invoked directly. Instead, it"
59             + " should be invoked only from the class in which it was declared. For example, if"
60             + " overriding Converter.doForward, you should invoke it through Converter.convert."
61             + " For testing, factor out the code you want to run to a separate method.",
62         category = GUAVA, severity = ERROR, maturity = MATURE)
63     public class ForOverrideChecker extends BugChecker
64         implements MethodInvocationTreeMatcher, MethodTreeMatcher {
65     
66       private static final String FOR_OVERRIDE = "com.google.errorprone.annotations.ForOverride";
67       private static final String MESSAGE_BASE = "Method annotated @ForOverride ";
68     
69       //TODO(cushon): remove
70       private static final String GUAVA_FOR_OVERRIDE = "com.google.common.annotations.ForOverride";
71     
72       @Override
73       public Description matchMethodInvocation(MethodInvocationTree tree, VisitorState state) {
74         MethodSymbol method = ASTHelpers.getSymbol(tree);
75         Type currentClass = getOutermostClass(state);
76     
77         if (method.isStatic() || method.isConstructor() || currentClass == null) {
78           return Description.NO_MATCH;
79         }
80     
81         List<MethodSymbol> overriddenMethods = getOverriddenMethods(state, method);
82     
83         for (Symbol overriddenMethod : overriddenMethods) {
84           Type declaringClass = overriddenMethod.outermostClass().asType();
85           if (!declaringClass.equals(currentClass)) {
86             String customMessage = MESSAGE_BASE +  "must not be invoked directly "
87                         + "(except by the declaring class, " + declaringClass + ")";
88             return buildDescription(tree)
89                 .setMessage(customMessage)
90                 .build();
91           }
92         }
93     
94         return Description.NO_MATCH;
95       }
96     
97       @Override
98       public Description matchMethod(MethodTree tree, VisitorState state) {
99         MethodSymbol method = ASTHelpers.getSymbol(tree);
100     
101         if (method.isStatic() || method.isConstructor()) {
102           return Description.NO_MATCH;
103         }
104     
105         if (method.getModifiers().contains(Modifier.PUBLIC)
106             || method.getModifiers().contains(Modifier.PRIVATE)) {
107           List<MethodSymbol> overriddenMethods = getOverriddenMethods(state, method);
108     
109           if (!overriddenMethods.isEmpty()) {
110             String customMessage = MESSAGE_BASE + "must have protected or package-private visibility";
111             return buildDescription(tree)
112                 .setMessage(customMessage)
113                 .build();
114           }
115         }
116     
117         return Description.NO_MATCH;
118       }
119     
120       /**
121        * Get overridden @ForOverride methods.
122        *
123        * @param state the VisitorState
124        * @param method the method to find overrides for
125        * @return a list of methods annotated @ForOverride that the method overrides, including the
126        * method itself if it has the annotation
127        */
128       private List<MethodSymbol> getOverriddenMethods(VisitorState state, MethodSymbol method) {
129         // Static methods cannot override, only overload.
130         if (method.isStatic()) {
131           throw new IllegalArgumentException(
132               "getOverriddenMethods may not be called on a static method");
133         }
134     
135         List<MethodSymbol> list = new LinkedList<MethodSymbol>();
136         list.add(method);
137     
138         // Iterate over supertypes of the type that owns this method, collecting a list of all method
139         // symbols with the same name.  We intentionally exclude interface methods because interface
140         // methods cannot be annotated @ForOverride.  @ForOverride methods must have protected or
141         // package-private visibility, but interface methods have implicit public visibility.
142         Type currType = state.getTypes().supertype(method.owner.type);
143         while (currType != null
144             && !currType.equals(state.getSymtab().objectType)
145             && !currType.equals(Type.noType)) {
146           Symbol sym = currType.tsym.members().findFirst(method.name);
147           if (sym instanceof MethodSymbol) {
148             list.add((MethodSymbol) sym);
149           }
150           currType = state.getTypes().supertype(currType);
151         }
152     
153         // Remove methods that either don't have the @ForOverride annotation or don't override the
154         // method in question.
155         Iterator<MethodSymbol> iter = list.iterator();
156         while (iter.hasNext()) {
157           MethodSymbol member = iter.next();
158           if (!(hasAnnotation(FOR_OVERRIDE, member) || hasAnnotation(GUAVA_FOR_OVERRIDE, member))
159               // Note that MethodSymbol.overrides() ignores static-ness, but that's OK since we've
160               // already checked that this method is not static.
161               || !method.overrides(
162                   member, (TypeSymbol) member.owner, state.getTypes(), true)) {
163             iter.remove();
164           }
165         }
166     
167         return list;
168       }
169     
170       /**
171        * Get the outermost class/interface/enum of an element, or null if none.
172        */
173       private Type getOutermostClass(VisitorState state) {
174         TreePath path = state.getPath();
175         Type type = null;
176     
177         while (path != null) {
178           if (path.getLeaf().getKind() == Kind.CLASS
179               || path.getLeaf().getKind() == Kind.INTERFACE
180               || path.getLeaf().getKind() == Kind.ENUM) {
181             type = ASTHelpers.getSymbol(path.getLeaf()).type;
182           }
183           path = path.getParentPath();
184         }
185         return type;
186       }
187     
188       private boolean hasAnnotation(String annotation, Symbol member) {
189         for (Attribute.Compound attribute : member.getAnnotationMirrors()) {
190           if (annotation.equals(attribute.type.toString())) {
191             return true;
192           }
193         }
194         return false;
195       }
