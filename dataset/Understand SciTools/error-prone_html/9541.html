17     package com.google.errorprone.matchers;
18     
19     import com.google.errorprone.VisitorState;
20     import com.sun.source.tree.ExpressionTree;
21     import com.sun.source.tree.MethodInvocationTree;
22     import com.sun.source.tree.Tree.Kind;
23     
24     /**
25      * @author sjnickerson@google.com (Simon Nickerson)
26      *
27      */
28     public class ExpressionMethodSelect implements Matcher<ExpressionTree> {
29     
30       private final Matcher<ExpressionTree> methodSelectMatcher;
31       
32       public ExpressionMethodSelect(Matcher<ExpressionTree> methodSelectMatcher) {
33         this.methodSelectMatcher = methodSelectMatcher;
34       }
35       
36       @Override
37       public boolean matches(ExpressionTree t, VisitorState state) {
38         if (t.getKind() != Kind.METHOD_INVOCATION) {
39           return false;  
40         }
41         
42         MethodInvocationTree methodInvocation = (MethodInvocationTree) t;
43         return methodSelectMatcher.matches(methodInvocation.getMethodSelect(), state);
44       }
45     
46     }
