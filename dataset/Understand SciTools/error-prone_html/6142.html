17     package com.google.errorprone.dataflow;
18     
19     import static com.google.common.base.Preconditions.checkArgument;
20     import static com.google.common.base.Preconditions.checkNotNull;
21     import static com.google.common.collect.Sets.intersection;
22     import static javax.lang.model.element.ElementKind.EXCEPTION_PARAMETER;
23     import static javax.lang.model.element.ElementKind.LOCAL_VARIABLE;
24     import static javax.lang.model.element.ElementKind.PARAMETER;
25     import static javax.lang.model.element.ElementKind.RESOURCE_VARIABLE;
26     
27     import com.google.common.collect.ImmutableMap;
28     
29     import org.checkerframework.dataflow.analysis.AbstractValue;
30     import org.checkerframework.dataflow.analysis.FlowExpressions;
31     import org.checkerframework.dataflow.analysis.Store;
32     import org.checkerframework.dataflow.cfg.node.LocalVariableNode;
33     import org.checkerframework.dataflow.cfg.node.VariableDeclarationNode;
34     import org.checkerframework.javacutil.TreeUtils;
35     
36     import java.util.HashMap;
37     import java.util.Map;
38     
39     import javax.lang.model.element.Element;

41     /**
42      * Immutable map from each local variable to its {@link AbstractValue}. Note that, while the
43      * interface is written in terms of <b>nodes</b>, the stored data is indexed by variable
44      * <b>declaration</b>, so values persist across nodes.
45      *
46      * <p>To derive a new instance, {@linkplain #toBuilder() create a builder} from an old instance. To
47      * start from scratch, call {@link #empty()}.
48      *
49      * @author deminguyen@google.com (Demi Nguyen)
50      */
51     public final class LocalStore<V extends AbstractValue<V>> implements Store<LocalStore<V>> {
52     
53       @SuppressWarnings({"unchecked", "rawtypes"}) // fully variant
54       private static final LocalStore<?> EMPTY = new LocalStore(ImmutableMap.of());
55     
56       @SuppressWarnings("unchecked") // fully variant
57       public static <V extends AbstractValue<V>> LocalStore<V> empty() {
58         return (LocalStore<V>) EMPTY;
59       }
60     
61       private final ImmutableMap<Element, V> contents;
62     
63       private LocalStore(Map<Element, V> contents) {
64         this.contents = ImmutableMap.copyOf(contents);
65       }
66     
67       /**
68        * Returns the value for the given variable. {@code element} must come from a call to
69        * {@link LocalVariableNode#getElement()} or
70        * {@link TreeUtils#elementFromDeclaration(com.sun.source.tree.VariableTree)}(
71        * {@link VariableDeclarationNode#getTree()}).
72        */
73       public V getInformation(Element element) {
74         checkElementType(element);
75         return contents.get(checkNotNull(element));
76       }
77     
78       public Builder<V> toBuilder() {
79         return new Builder<V>(this);
80       }
81     
82       /**
83        * Builder for {@link LocalStore} instances. To obtain an instance, obtain a {@link LocalStore}
84        * (such as {@link LocalStore#empty()}), and call {@link LocalStore#toBuilder() toBuilder()} on
85        * it.
86        */
87       public static final class Builder<V extends AbstractValue<V>> {
88         private final Map<Element, V> contents;
89     
90         Builder(LocalStore<V> prototype) {
91           contents = new HashMap<>(prototype.contents);
92         }
93     
94         /**
95          * Sets the value for the given variable. {@code element} must come from a call to
96          * {@link LocalVariableNode#getElement()} or
97          * {@link TreeUtils#elementFromDeclaration(com.sun.source.tree.VariableTree)}(
98          * {@link VariableDeclarationNode#getTree()}).
99          */
100         public Builder<V> setInformation(Element element, V value) {
101           checkElementType(element);
102           contents.put(checkNotNull(element), checkNotNull(value));
103           return this;
104         }
105     
106         public LocalStore<V> build() {
107           return new LocalStore<V>(contents);
108         }
109       }
110     
111       @Override
112       public LocalStore<V> copy() {
113         // No need to copy because it's immutable.
114         return this;
115       }
116     
117       @Override
118       public LocalStore<V> leastUpperBound(LocalStore<V> other) {
119         Builder<V> result = LocalStore.<V>empty().toBuilder();
120         for (Element var : intersection(contents.keySet(), other.contents.keySet())) {
121           result.contents.put(var, contents.get(var).leastUpperBound(other.contents.get(var)));
122         }
123         return result.build();
124       }
125     
126       @Override
127       public boolean equals(Object o) {
128         if (!(o instanceof LocalStore)) {
129           return false;
130         }
131         LocalStore<?> other = (LocalStore<?>) o;
132         return contents.equals(other.contents);
133       }
134     
135       @Override
136       public int hashCode() {
137         return contents.hashCode();
138       }
139     
140       @Override
141       public String toString() {
142         return contents.toString();
143       }
144     
145       @Override
146       public boolean canAlias(FlowExpressions.Receiver a, FlowExpressions.Receiver b) {
147         return true;
148       }
149     
150       @Override
151       public boolean hasDOToutput() {
152         return false;
153       }
154     
155       @Override
156       public String toDOToutput() {
157         throw new UnsupportedOperationException("DOT output not supported");
158       }
159     
160       private static void checkElementType(Element element) {
161         checkArgument(
162             element.getKind() == LOCAL_VARIABLE
163                 || element.getKind() == PARAMETER
164                 || element.getKind() == EXCEPTION_PARAMETER
165                 || element.getKind() == RESOURCE_VARIABLE,
166             "unexpected element type: %s (%s)",
167             element.getKind(),
168             element);
169       }
170     }
