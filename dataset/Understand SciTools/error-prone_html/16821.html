34     public class GuardedByUtils {
35       static String getGuardValue(Tree tree) {
36         {
37           net.jcip.annotations.GuardedBy guardedBy =
38               ASTHelpers.getAnnotation(tree, net.jcip.annotations.GuardedBy.class);
39           if (guardedBy != null) {
40             return guardedBy.value();
41           }
42         }
43     
44         {
45           javax.annotation.concurrent.GuardedBy guardedBy =
46               ASTHelpers.getAnnotation(tree, javax.annotation.concurrent.GuardedBy.class);
47           if (guardedBy != null) {
48             return guardedBy.value();
49           }
50         }
51     
52         return null;
53       }
54     
55       public static JCTree.JCExpression parseString(String guardedByString, Context context) {
56         JavacParser parser =
57             ParserFactory.instance(context).newParser(guardedByString, false, true, false);
58         JCTree.JCExpression exp;
59         try {
60           exp = parser.parseExpression();
61         } catch (Throwable e) {
62           throw new IllegalGuardedBy(e.getMessage());
63         }
64         int len = (parser.getEndPos(exp) - exp.getStartPosition());
65         if (len != guardedByString.length()) {
66           throw new IllegalGuardedBy("Didn't parse entire string.");
67         }
68         return exp;
69       }

72       @AutoValue
73       abstract static class GuardedByValidationResult {
74         abstract String message();
75         abstract Boolean isValid();
76         
77         static GuardedByValidationResult invalid(String message) {
78           return new AutoValue_GuardedByUtils_GuardedByValidationResult(message, false);
79         }
80         
81         static GuardedByValidationResult ok() {
82           return new AutoValue_GuardedByUtils_GuardedByValidationResult("", true);
83         }
84       }
85     
86       public static GuardedByValidationResult isGuardedByValid(Tree tree, VisitorState state) {
87         String guard = GuardedByUtils.getGuardValue(tree);
88         if (guard == null) {
89           return GuardedByValidationResult.ok();
90         }
91     
92         Optional<GuardedByExpression> boundGuard =
93             GuardedByBinder.bindString(guard, GuardedBySymbolResolver.from(tree, state));
94         if (!boundGuard.isPresent()) {
95           return GuardedByValidationResult.invalid("could not resolve guard");
96         }
97     
98         Symbol treeSym = ASTHelpers.getSymbol(tree);
99         if (treeSym == null) {
100           // this shouldn't happen unless the compilation had already failed.
101           return GuardedByValidationResult.ok();
102         }
103     
104         boolean staticGuard = boundGuard.get().kind() == GuardedByExpression.Kind.CLASS_LITERAL
105             || (boundGuard.get().sym() != null && boundGuard.get().sym().isStatic());
106         if (treeSym.isStatic() && !staticGuard) {
107           return GuardedByValidationResult.invalid(
108               "static member guarded by instance");
109         }
110     
111         return GuardedByValidationResult.ok();
112       }
113     }
