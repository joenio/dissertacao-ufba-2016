20     import com.google.common.base.Function;
21     import com.google.common.base.Preconditions;

23     import com.google.common.collect.Lists;
24     import com.google.common.collect.Ordering;

26     import com.sun.tools.javac.tree.EndPosTable;
27     import com.sun.tools.javac.tree.JCTree.JCCompilationUnit;
28     import com.sun.tools.javac.tree.JCTree.JCExpression;
29     import com.sun.tools.javac.tree.JCTree.JCImport;
30     
31     import java.util.Collection;
32     import java.util.List;
33     import java.util.Set;
34     import java.util.TreeSet;
35     import java.util.regex.Matcher;
36     import java.util.regex.Pattern;

38     /**
39      * Represents a list of import statements.  Supports adding and removing 
40      * import statements and pretty printing the result as source code.  Correctly
41      * sorts the imports according to Google Java Style Guide rules. 
42      * 
43      * @author eaftan@google.com (Eddie Aftandilian)
44      */
45     public class ImportStatements {
46       
47       private int startPos = Integer.MAX_VALUE;
48       private int endPos = -1;
49       private final Set<String> importStrings;
50       private boolean hasExistingImports;
51       
52       /**
53        * An Ordering that sorts import statements based on the Google Java Style 
54        * Guide.
55        */
56       private static final Ordering<String> IMPORT_ORDERING = new Ordering<String>() {
57         @Override
58         public int compare(String s1, String s2) {
59           return ComparisonChain.start()
60               .compare(Kind.getKind(s1), Kind.getKind(s2))
61               .compare(s1, s2)
62               .result();
63         }
64       };
65       
66       /**
67        * A regex to use for finding the top-level package in an import
68        * statement.
69        */
70       private static final Pattern TOPLEVEL_PATTERN = 
71           Pattern.compile("import\\s+(static\\s+|)([^.]+).");
72       
73       public static ImportStatements create(JCCompilationUnit compilationUnit) {
74         return new ImportStatements((JCExpression) compilationUnit.getPackageName(),
75             compilationUnit.getImports(), compilationUnit.endPositions);
76       }
77       
78       public ImportStatements(JCExpression packageTree, List<JCImport> importTrees,
79           EndPosTable endPositions) {
80         
81         // find start, end positions for current list of imports (for replacement)
82         if (importTrees.isEmpty()) {
83           // start/end positions are just after the package expression
84           hasExistingImports = false;
85           startPos = packageTree.getEndPosition(endPositions) + 2;   // +2 for semicolon and newline
86           endPos = startPos;
87         } else {
88           // process list of imports and find start/end positions
89           hasExistingImports = true;
90           for (JCImport importTree : importTrees) {
91             int currStartPos = importTree.getStartPosition();
92             int currEndPos = importTree.getEndPosition(endPositions);
93     
94             startPos = Math.min(startPos, currStartPos);
95             endPos = Math.max(endPos, currEndPos);
96           }  
97         }
98         
99         // sanity check for start/end positions
100         Preconditions.checkState(startPos <= endPos);
101         
102         // convert list of JCImports to list of strings
103         importStrings = new TreeSet<>(IMPORT_ORDERING);
104         importStrings.addAll(Lists.transform(importTrees, new Function<JCImport, String>() {
105           @Override
106           public String apply(JCImport input) {
107             String importExpr = input.toString();
108             return importExpr.substring(0, importExpr.length() - 2); // snip trailing ";\n"
109           }
110         }));
111       }
112         
113       /**
114        * Return the start position of the import statements.
115        */
116       public int getStartPos() {
117         return startPos;
118       }
119     
120       /**
121        * Return the end position of the import statements.
122        */
123       public int getEndPos() {
124         return endPos;
125       }
126       
127       /**
128        * Add an import to the list of imports. If the import is already in the 
129        * list, does nothing. The import should be of the form "import foo.bar".
130        * 
131        * @param importToAdd a string representation of the import to add
132        * @return true if the import was added
133        */
134       public boolean add(String importToAdd) {
135         return importStrings.add(importToAdd);
136       }
137       
138       /**
139        * Add all imports in a collection to this list of imports. Does not add 
140        * any imports that are already in the list.
141        * 
142        * @param importsToAdd a collection of imports to add
143        * @return true if any imports were added to the list
144        */
145       public boolean addAll(Collection<String> importsToAdd) {
146         return importStrings.addAll(importsToAdd);
147       }
148       
149       /**
150        * Remove an import from the list of imports. If the import is not in the
151        * list, does nothing. The import should be of the form "import foo.bar".
152        * 
153        * @param importToRemove a string representation of the import to remove
154        * @return true if the import was removed
155        */
156       public boolean remove(String importToRemove) {
157         return importStrings.remove(importToRemove);
158       }
159       
160       /**
161        * Removes all imports in a collection to this list of imports. Does not
162        * remove any imports that are not in the list.
163        * 
164        * @param importsToRemove a collection of imports to remove
165        * @return true if any imports were removed from the list
166        */
167       public boolean removeAll(Collection<String> importsToRemove) {
168         return importStrings.removeAll(importsToRemove);
169       }
170       
171       /**
172        * Returns a string representation of the imports, as proper Java code.
173        * Includes newlines in the correct places as defined by the Google
174        * Java Style Guide.
175        */
176       @Override
177       public String toString() {
178         if (importStrings.size() == 0) {
179           return "";
180         }
181         
182         StringBuilder result = new StringBuilder();
183         
184         if (!hasExistingImports) {
185           // insert a newline after the package expression, then add imports
186           result.append('\n');
187         }
188         
189         // output sorted imports, with line breaks between sections
190         Kind prevKind = null;
191         String prevTopLevel = null;
192         for (String importString : importStrings) {
193           Kind currKind = Kind.getKind(importString);
194           String currTopLevel = getTopLevel(importString);
195           if (prevKind != null && prevKind != currKind) {
196             result.append('\n');
197           } else if (currKind == Kind.THIRD_PARTY) {
198             if (prevTopLevel != null && !prevTopLevel.equals(currTopLevel)) {
199               result.append('\n');
200             }
201           }
202           result.append(importString).append(";\n");
203           prevKind = currKind;
204           prevTopLevel = currTopLevel;
205         }
206         
207         String replacementString = result.toString();
208         if (!hasExistingImports) {
209           return replacementString;
210         } else {
211           return replacementString.substring(0, replacementString.length() - 1);    // trim last newline
212         }
213       }
214       
215       /**
216        * Given an import string, returns the top-level package for that
217        * import.
218        */
219       @VisibleForTesting
220       static String getTopLevel(String importString) {
221         Matcher m = TOPLEVEL_PATTERN.matcher(importString);
222         if (m.find()) {
223           return m.group(2);
224         } else {
225           throw new IllegalArgumentException(importString + " is not a valid import statement");
226         }
227       }
228     
229       /**
230        * An enumeration of the different kinds of import statements we might 
231        * encounter.  Each kind must be sorted into its own bucket in the import 
232        * statements.
233        */
234       private enum Kind {
235         STATIC,         // import static
236         GOOGLE,         // import com.google...
237         THIRD_PARTY,    // import org.foo.bar...
238         JAVA,           // import java...
239         JAVAX;          // import javax...
240         
241         /**
242          * Determines the Kind of an import statement.
243          * 
244          * @param importString the import statement as a string
245          * @return the kind of the import statement
246          */
247         public static Kind getKind(String importString) {
248           if (importString.startsWith("import static")) {
249             return STATIC;
250           } else if (importString.startsWith("import com.google.")) {
251             return GOOGLE;
252           } else if (importString.startsWith("import java.")) {
253             return JAVA;
254           } else if (importString.startsWith("import javax.")) {
255             return JAVAX;
256           } else {
257             return THIRD_PARTY;
258           }
259         }
260       }
