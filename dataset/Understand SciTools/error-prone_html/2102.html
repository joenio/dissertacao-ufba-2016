15     package com.google.errorprone.analysis;
16     
17     import com.google.auto.value.AutoValue;
18     import com.google.errorprone.BugPattern.SeverityLevel;
19     import com.google.errorprone.DescriptionListener;
20     import com.google.errorprone.ErrorProneOptions.Severity;
21     import com.google.errorprone.matchers.Description;
22     
23     import com.sun.source.tree.CompilationUnitTree;
24     import com.sun.tools.javac.util.Context;
25     
26     import java.util.Set;
27     
28     /**
29      * Wraps a {@code TopLevelAnalysis} to handle severity levels, taking into account
30      * {@code ErrorProneOptions} in updating the severity level.
31      *
32      * @author Louis Wasserman
33      */
34     @AutoValue
35     public abstract class TopLevelAnalysisWithSeverity implements TopLevelAnalysis {
36       static TopLevelAnalysisWithSeverity wrap(String canonicalName, boolean onByDefault,
37           SeverityLevel defaultSeverity, boolean disableable, TopLevelAnalysis analysis) {
38         return new AutoValue_TopLevelAnalysisWithSeverity(canonicalName, onByDefault, defaultSeverity,
39             disableable, analysis);
40       }
41     
42       // TODO(lowasser): consider eliminating this and respecting all configs on knownAnalysisNames
43       abstract String canonicalName();
44     
45       abstract boolean onByDefault();
46     
47       abstract SeverityLevel defaultSeverity();
48     
49       abstract boolean disableable();
50     
51       abstract TopLevelAnalysis analysis();
52     
53       @Override
54       public Set<String> knownAnalysisNames() {
55         return analysis().knownAnalysisNames();
56       }
57     
58       @Override
59       public void analyze(CompilationUnitTree compilationUnit, Context context,
60           AnalysesConfig config, final DescriptionListener listener) {
61         final SeverityLevel severity;
62         boolean on = onByDefault();
63         Severity optionsSeverity = config.errorProneOptions().getSeverityMap().get(canonicalName());
64         if (optionsSeverity != null) {
65           switch (optionsSeverity) {
66             case OFF:
67               if (!disableable()) {
68                 throw new IllegalArgumentException(canonicalName() + " may not be disabled");
69               }
70               on = false;
71               severity = SeverityLevel.NOT_A_PROBLEM;
72               break;
73             case DEFAULT:
74               on = true;
75               severity = defaultSeverity();
76               break;
77             case WARN:
78               // Demoting an enabled check from an error to a warning is a form of disabling
79               if (!disableable() && defaultSeverity() == SeverityLevel.ERROR) {
80                 throw new IllegalArgumentException(
81                     canonicalName() + " is not disableable and may not be demoted to a warning");
82               }
83               on = true;
84               severity = SeverityLevel.WARNING;
85               break;
86             case ERROR:
87               on = true;
88               severity = SeverityLevel.ERROR;
89               break;
90             default:
91               throw new IllegalStateException("Unexpected severity level: " + optionsSeverity);
92           }
93         } else {
94           severity = defaultSeverity();
95         }
96         if (on && severity.enabled()) {
97           analysis().analyze(compilationUnit, context, config, new DescriptionListener() {
98             @Override
99             public void onDescribed(Description description) {
100               listener.onDescribed(description.applySeverityOverride(severity));
101             }
102           });
103         }
104       }
105     }
