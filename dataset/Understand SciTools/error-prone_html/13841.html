43     /**
44      * Test helper for {@link CodeTransformer} implementations.
45      *
46      * @author lowasser@google.com (Louis Wasserman)
47      */
48     @AutoValue
49     public abstract class CodeTransformerTestHelper {
50       public static CodeTransformerTestHelper create(CodeTransformer transformer) {
51         return new AutoValue_CodeTransformerTestHelper(transformer);
52       }
53     
54       abstract CodeTransformer transformer();
55     
56       public JavaFileObject transform(JavaFileObject original) {
57         JavaCompiler compiler = JavacTool.create();
58         DiagnosticCollector<JavaFileObject> diagnosticsCollector =
59             new DiagnosticCollector<JavaFileObject>();
60         StandardJavaFileManager fileManager =
61             compiler.getStandardFileManager(diagnosticsCollector, Locale.ENGLISH, UTF_8);
62         JavacTaskImpl task = (JavacTaskImpl) compiler.getTask(CharStreams.nullWriter(),
63             fileManager,
64             diagnosticsCollector,
65             ImmutableList.<String>of(),
66             null,
67             ImmutableList.of(original));
68     
69         try {
70           SourceFile sourceFile = SourceFile.create(original);
71           Iterable<? extends CompilationUnitTree> trees = task.parse();
72           task.analyze();
73           JCCompilationUnit tree =
74               Iterables.getOnlyElement(Iterables.filter(trees, JCCompilationUnit.class));
75           DescriptionBasedDiff diff = DescriptionBasedDiff.create(tree);
76           transformer().apply(new TreePath(tree), task.getContext(), diff);
77           diff.applyDifferences(sourceFile);
78     
79           return JavaFileObjects.forSourceString(
80               Iterables.getOnlyElement(Iterables.filter(tree.getTypeDecls(), JCClassDecl.class))
81                   .sym
82                   .getQualifiedName()
83                   .toString(),
84               sourceFile.getSourceText());
85         } catch (IOException e) {
86           throw new RuntimeException(e);
87         }
88       }
89     }
