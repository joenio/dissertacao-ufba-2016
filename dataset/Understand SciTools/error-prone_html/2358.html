53     /**
54      * @author alexeagle@google.com (Alex Eagle)
55      */
56     @BugPattern(name = "DeadException", altNames = "ThrowableInstanceNeverThrown",
57         summary = "Exception created but not thrown",
58         explanation =
59             "The exception is created with new, but is not thrown, and the reference is lost.",
60         category = JDK, severity = ERROR, maturity = MATURE)
61     public class DeadException extends BugChecker implements NewClassTreeMatcher {
62     
63       public static final Matcher<Tree> MATCHER = allOf(
64           parentNode(kindIs(EXPRESSION_STATEMENT)),
65           isSubtypeOf(EXCEPTION_TYPE),
66           not(enclosingMethod(JUnitMatchers.wouldRunInJUnit4)),
67           anyOf(not(enclosingMethod(JUnitMatchers.isJunit3TestCase)),
68                 not(enclosingClass(JUnitMatchers.isJUnit3TestClass)))
69       );
70     
71       @Override
72       public Description matchNewClass(NewClassTree newClassTree, VisitorState state) {
73         if (!MATCHER.matches(newClassTree, state)) {
74           return Description.NO_MATCH;
75         }
76     
77         StatementTree parent = (StatementTree) state.getPath().getParentPath().getLeaf();
78     
79         boolean isLastStatement = anyOf(
80             new ChildOfBlockOrCase<>(ChildMultiMatcher.MatchType.LAST,
81                 Matchers.<StatementTree>isSame(parent)),
82             // it could also be a bare if statement with no braces
83             parentNode(parentNode(kindIs(IF))))
84             .matches(newClassTree, state);
85     
86         Fix fix;
87         if (isLastStatement) {
88           fix = SuggestedFix.prefixWith(newClassTree, "throw ");
89         } else {
90           fix = SuggestedFix.delete(parent);
91         }
92     
93         return describeMatch(newClassTree, fix);
94       }
95       
96       private static class ChildOfBlockOrCase<T extends Tree> 
97           extends ChildMultiMatcher<T, StatementTree> {
98         public ChildOfBlockOrCase(MatchType matchType, Matcher<StatementTree> nodeMatcher) {
99           super(matchType, nodeMatcher);
100         }
101     
102         @Override
103         protected Iterable<? extends StatementTree> getChildNodes(T tree, VisitorState state) {
104           Tree enclosing = state.findEnclosing(CaseTree.class, BlockTree.class);
105           if (enclosing == null) {
106             return ImmutableList.of();
107           }
108           if (enclosing instanceof BlockTree) {
109             return ((BlockTree) enclosing).getStatements();
110           } else if (enclosing instanceof CaseTree) {
111             return ((CaseTree) enclosing).getStatements();
112           } else {
113             // findEnclosing given two types must return something of one of those types
114             throw new IllegalStateException("enclosing tree not a BlockTree or CaseTree");
115           }
116         }
117       }
118     }
