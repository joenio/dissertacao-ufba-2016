35     import java.util.List;
36     import java.util.Set;

38     /**
39      * Abstract implementation of checkers for {@code @LockMethod} and{@code @UnlockMethod}.
40      *
41      * @author cushon@google.com (Liam Miller-Cushon)
42      */
43     public abstract class AbstractLockMethodChecker extends BugChecker
44         implements BugChecker.MethodTreeMatcher {
45     
46       /**
47        * Returns the lock expressions in the {@code @LockMethod}/{@code @UnlockMethod}
48        * annotation, if any.
49        */
50       protected abstract ImmutableList<String> getLockExpressions(MethodTree tree);
51     
52       /**
53        * Searches the method body for locks that are acquired/released.
54        */
55       protected abstract Set<GuardedByExpression> getActual(MethodTree tree, VisitorState state);
56     
57       /**
58        * Searches the method body for the incorrect lock operation (e.g. releasing a lock in
59        * {@code @LockMethod}, or acquiring a lock in {@code @UnlockMethod}).
60        */
61       protected abstract Set<GuardedByExpression> getUnwanted(MethodTree tree, VisitorState state);
62     
63       /**
64        * Builds the error message, given the list of locks that were not handled.
65        */
66       protected abstract String buildMessage(String unhandled);
67     
68       @Override
69       public Description matchMethod(MethodTree tree, final VisitorState state) {
70     
71         ImmutableList<String> lockExpressions = getLockExpressions(tree);
72         if (lockExpressions.isEmpty()) {
73           return Description.NO_MATCH;
74         }
75     
76         Optional<ImmutableSet<GuardedByExpression>> expected =
77             parseLockExpressions(lockExpressions, tree, state);
78         if (!expected.isPresent()) {
79           return buildDescription(tree)
80               .setMessage("Could not resolve lock expression.")
81               .build();
82         }
83     
84         Set<GuardedByExpression> unwanted = getUnwanted(tree, state);
85         SetView<GuardedByExpression> mishandled = Sets.intersection(expected.get(), unwanted);
86         if (!mishandled.isEmpty()) {
87           String message = buildMessage(formatLockString(mishandled));
88           return buildDescription(tree).setMessage(message).build();
89         }
90     
91         Set<GuardedByExpression> actual = getActual(tree, state);
92         SetView<GuardedByExpression> unhandled = Sets.difference(expected.get(), actual);
93         if (!unhandled.isEmpty()) {
94           String message = buildMessage(formatLockString(unhandled));
95           return buildDescription(tree).setMessage(message).build();
96         }
97     
98         return Description.NO_MATCH;
99       }
100     
101       private static String formatLockString(Set<GuardedByExpression> locks) {
102         ImmutableList<String> sortedUnhandled = FluentIterable.from(locks)
103             .transform(Functions.toStringFunction())
104             .toSortedList(Ordering.natural());
105         return Joiner.on(", ").join(sortedUnhandled);
106       }
107     
108       private static Optional<ImmutableSet<GuardedByExpression>> parseLockExpressions(
109           List<String> lockExpressions, Tree tree, VisitorState state) {
110         ImmutableSet.Builder<GuardedByExpression> builder = ImmutableSet.builder();
111         for (String lockExpression : lockExpressions) {
112           Optional<GuardedByExpression> guard = GuardedByBinder.bindString(
113               lockExpression, GuardedBySymbolResolver.from(tree, state));
114           if (!guard.isPresent()) {
115             return Optional.absent();
116           }
117           builder.add(guard.get());
118         }
119         return Optional.of(builder.build());
120       }
121     }
