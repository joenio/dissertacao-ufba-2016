17     package com.google.errorprone.matchers;
18     
19     import static com.google.errorprone.BugPattern.SeverityLevel.NOT_A_PROBLEM;
20     
21     import com.google.common.base.Preconditions;
22     import com.google.common.base.Predicate;
23     import com.google.common.collect.ImmutableList;
24     import com.google.common.collect.Iterables;
25     import com.google.errorprone.BugPattern;
26     import com.google.errorprone.BugPattern.SeverityLevel;
27     import com.google.errorprone.fixes.Fix;
28     
29     import com.sun.source.tree.Tree;
30     
31     import javax.annotation.CheckReturnValue;
32     import javax.annotation.Nullable;
33     
34     /**
35      * Simple data object containing the information captured about an AST match.
36      * Can be printed in a UI, or output in structured format for use by tools.
37      * @author alexeagle@google.com (Alex Eagle)
38      */
39     public class Description {
40       /** Describes the sentinel value of the case where the match failed. */
41       public static final Description NO_MATCH = new Description(null, "<no match>", "<no match>",
42           "<no match>", ImmutableList.<Fix>of(), NOT_A_PROBLEM);
43     
44       private static final String UNDEFINED_CHECK_NAME = "Undefined";
45     
46       /**
47        * The AST node that matched.
48        */
49       public final Tree node;
50     
51       /**
52        * The name of the check that produced the match.
53        */
54       public final String checkName;
55     
56       /**
57        * The raw message, not including the check name or the link.
58        */
59       private final String rawMessage;
60     
61       /**
62        * The link to include in the error message. May be null if there is no link.
63        */
64       @Nullable
65       private final String link;
66     
67       /**
68        * A list of fixes to suggest in an error message or use in automated refactoring.  Fixes are
69        * in order of decreasing preference, from most preferred to least preferred.
70        */
71       public final ImmutableList<Fix> fixes;
72     
73       /**
74        * Is this a warning, error, etc.?
75        */
76       public final BugPattern.SeverityLevel severity;
77     
78       /**
79        * Returns the message to be printed by the compiler when a match is found in interactive use.
80        * Includes the name of the check and a link for more information.
81        */
82       public String getMessage() {
83         return String.format("[%s] %s", checkName, getMessageWithoutCheckName());
84       }
85     
86       /**
87        * Returns link to be included in the error message or null if there is no link.
88        */
89       @Nullable
90       public String getLink() {
91         return link;
92       }
93     
94       /** Returns the raw message, not including a link or check name. */
95       public String getRawMessage() {
96         return rawMessage;
97       }
98     
99       /**
100        * Returns the message, not including the check name but including the link.
101        */
102       public String getMessageWithoutCheckName() {
103         return link != null
104             ? String.format("%s\n%s", rawMessage, link)
105             : String.format("%s", rawMessage);
106       }
107     
108       /** TODO(cushon): Remove this constructor and ensure that there's always a check name. */
109       public Description(Tree node, String message, Fix suggestedFix,
110                          BugPattern.SeverityLevel severity) {
111         this(node, UNDEFINED_CHECK_NAME, message, message, ImmutableList.of(suggestedFix), severity);
112         if (suggestedFix == null) {
113           throw new IllegalArgumentException("suggestedFix must not be null.");
114         }
115       }
116     
117       private Description(Tree node, String checkName, String rawMessage, String link,
118           ImmutableList<Fix> fixes, BugPattern.SeverityLevel severity) {
119         this.node = node;
120         this.checkName = checkName;
121         this.rawMessage = rawMessage;
122         this.link = link;
123         this.fixes = fixes;
124         this.severity = severity;
125       }
126     
127       @CheckReturnValue
128       public Description applySeverityOverride(SeverityLevel severity) {
129         return new Description(node, checkName, rawMessage, link, fixes, severity);
130       }
131     
132       @CheckReturnValue
133       public Description filterFixes(Predicate<? super Fix> predicate) {
134         return new Description(node,
135             checkName,
136             rawMessage,
137             link,
138             ImmutableList.copyOf(Iterables.filter(fixes, predicate)),
139             severity);
140       }
141     
142       /**
143        * Construct the link text to include in the compiler error message. Returns null if there is
144        * no link.
145        */
146       private static String linkTextForDiagnostic(String linkUrl) {
147         return linkUrl == null ? null : "  (see " + linkUrl + ")";
148       }
149     
150       /**
151        * Returns a new builder for {@link Description}s.
152        */
153       public static Builder builder(
154           Tree node, String name, @Nullable String link, SeverityLevel severity, String message) {
155         return new Builder(node, name, link, severity, message);
156       }
157     
158       /**
159        * Builder for {@code Description}s.
160        */
161       public static class Builder {
162         private final Tree node;
163         private final String name;
164         private final String linkUrl;
165         private final SeverityLevel severity;
166         private final ImmutableList.Builder<Fix> fixListBuilder = ImmutableList.builder();
167         private String rawMessage;
168     
169         private Builder(
170             Tree node,
171             String name,
172             @Nullable String linkUrl,
173             SeverityLevel severity,
174             String rawMessage) {
175           this.node = Preconditions.checkNotNull(node);
176           this.name = Preconditions.checkNotNull(name);
177           this.linkUrl = linkUrl;
178           this.severity = Preconditions.checkNotNull(severity);
179           this.rawMessage = Preconditions.checkNotNull(rawMessage);
180         }
181     
182         /**
183          * Add a suggested fix for this {@code Description}. Fixes should be added in order of
184          * decreasing preference.
185          *
186          * @param fix A suggested fix for this problem
187          */
188         public Builder addFix(Fix fix) {
189           if (fix == null) {
190             throw new IllegalArgumentException("fix must not be null");
191           }
192           fixListBuilder.add(fix);
193           return this;
194         }
195     
196         /**
197          * Set a custom error message for this {@code Description}.  The custom message will be used
198          * instead of the summary field as the text for the diagnostic message.
199          *
200          * @param message A custom error message without the check name ("[checkname]") or link
201          */
202         public Builder setMessage(String message) {
203           if (message == null) {
204             throw new IllegalArgumentException("message must not be null");
205           }
206           this.rawMessage = message;
207           return this;
208         }
209     
210         public Description build() {
211           return new Description(node, name, rawMessage, linkTextForDiagnostic(linkUrl),
212               fixListBuilder.build(), severity);
213         }
214       }
215     
216     }
