17     package com.google.errorprone.bugpatterns;
18     
19     import static com.google.errorprone.BugPattern.Category.JDK;
20     import static com.google.errorprone.BugPattern.MaturityLevel.EXPERIMENTAL;
21     import static com.google.errorprone.BugPattern.SeverityLevel.ERROR;
22     import static com.google.errorprone.matchers.Matchers.allOf;
23     import static com.google.errorprone.matchers.Matchers.anyOf;
24     import static com.google.errorprone.matchers.Matchers.instanceMethod;
25     import static com.google.errorprone.matchers.Matchers.isSameType;
26     import static com.google.errorprone.matchers.Matchers.staticMethod;
27     import static com.google.errorprone.suppliers.Suppliers.BOOLEAN_TYPE;
28     
29     import com.google.common.base.Predicate;
30     import com.google.errorprone.BugPattern;
31     import com.google.errorprone.VisitorState;
32     import com.google.errorprone.bugpatterns.BugChecker.MethodInvocationTreeMatcher;
33     import com.google.errorprone.matchers.Description;
34     import com.google.errorprone.matchers.Matcher;
35     import com.google.errorprone.util.ASTHelpers;
36     
37     import com.sun.source.tree.MethodInvocationTree;
38     import com.sun.tools.javac.code.Flags;
39     import com.sun.tools.javac.code.Symbol.ClassSymbol;
40     import com.sun.tools.javac.code.Symbol.MethodSymbol;
41     import com.sun.tools.javac.code.Type;
42     import com.sun.tools.javac.code.Types;
43     import com.sun.tools.javac.util.Name;
44     
45     import java.util.Set;
46     
47     /**
48      * @author avenet@google.com (Arnaud J. Venet)
49      */
50     @BugPattern(
51       name = "EqualsIncompatibleType",
52       summary = "An equality test between objects with incompatible types always returns false",
53       category = JDK,
54       severity = ERROR,
55       maturity = EXPERIMENTAL
56     )
57     public class EqualsIncompatibleType extends BugChecker implements MethodInvocationTreeMatcher {
58       private static final Matcher<MethodInvocationTree> STATIC_EQUALS_INVOCATION_MATCHER =
59           anyOf(
60               allOf(
61                   staticMethod()
62                       .onClass("java.util.Objects")
63                       .named("equals")
64                       .withParameters("java.lang.Object", "java.lang.Object"),
65                   isSameType(BOOLEAN_TYPE)),
66               allOf(
67                   staticMethod()
68                       .onClass("com.google.common.base.Objects")
69                       .named("equal")
70                       .withParameters("java.lang.Object", "java.lang.Object"),
71                   isSameType(BOOLEAN_TYPE)));
72     
73       private static final Matcher<MethodInvocationTree> INSTANCE_EQUALS_INVOCATION_MATCHER =
74           allOf(
75               instanceMethod().anyClass().named("equals").withParameters("java.lang.Object"),
76               isSameType(BOOLEAN_TYPE));
77     
78       @Override
79       public Description matchMethodInvocation(
80           MethodInvocationTree invocationTree, final VisitorState state) {
81         if (!STATIC_EQUALS_INVOCATION_MATCHER.matches(invocationTree, state)
82             && !INSTANCE_EQUALS_INVOCATION_MATCHER.matches(invocationTree, state)) {
83           return Description.NO_MATCH;
84         }
85     
86         // This is the type of the object on which the java.lang.Object.equals() method
87         // is called, either directly or indirectly via a static utility method. In the latter,
88         // it is the type of the second argument to the static method.
89         Type receiverType;
90         // This is the type of the argument to the java.lang.Object.equals() method.
91         // In case a static utility method is used, it is the type of the second argument
92         // to this method.
93         Type argumentType;
94     
95         if (STATIC_EQUALS_INVOCATION_MATCHER.matches(invocationTree, state)) {
96           receiverType = ASTHelpers.getType(invocationTree.getArguments().get(0));
97           argumentType = ASTHelpers.getType(invocationTree.getArguments().get(1));
98         } else {
99           receiverType = ASTHelpers.getReceiverType(invocationTree);
100           argumentType = ASTHelpers.getType(invocationTree.getArguments().get(0));
101         }
102     
103         // If one type can be cast into the other, we don't flag the equality test.
104         if (ASTHelpers.isCastable(receiverType, argumentType, state)) {
105           return Description.NO_MATCH;
106         }
107     
108         // Otherwise, we explore the superclasses of the receiver type as well as the interfaces it
109         // implements and we collect all overrides of java.lang.Object.equals(). If one of those
110         // overrides is inherited by the argument, then we don't flag the equality test.
111         final Types types = state.getTypes();
112         Predicate<MethodSymbol> equalsPredicate =
113             new Predicate<MethodSymbol>() {
114               @Override
115               public boolean apply(MethodSymbol methodSymbol) {
116                 return !methodSymbol.isStatic()
117                     && ((methodSymbol.flags() & Flags.SYNTHETIC) == 0)
118                     && types.isSameType(methodSymbol.getReturnType(), state.getSymtab().booleanType)
119                     && methodSymbol.getParameters().size() == 1
120                     && types.isSameType(
121                         methodSymbol.getParameters().get(0).type, state.getSymtab().objectType);
122               }
123             };
124         Name equalsName = state.getName("equals");
125         Set<MethodSymbol> overridesOfEquals =
126             ASTHelpers.findMatchingMethods(equalsName, equalsPredicate, receiverType, types);
127         ClassSymbol argumentClass = (ClassSymbol) argumentType.tsym;
128     
129         for (MethodSymbol method : overridesOfEquals) {
130           ClassSymbol methodClass = (ClassSymbol) method.enclClass();
131           if (argumentClass.isSubClass(methodClass, types)
132               && !methodClass.equals(state.getSymtab().objectType.tsym)) {
133             // The type of the argument shares a superclass (other then java.lang.Object) or interface
134             // with the receiver that implements an override of java.lang.Object.equals().
135             return Description.NO_MATCH;
136           }
137         }
138     
139         // When we reach this point, we know that the two following facts hold:
140         // (1) The types of the receiver and the argument to the eventual invocation of
141         //     java.lang.Object.equals() are incompatible.
142         // (2) No common superclass (other than java.lang.Object) or interface of the receiver and the
143         //     argument defines an override of java.lang.Object.equals().
144         // This equality test almost certainly evaluates to false, which is very unlikely to be the
145         // programmer's intent. Hence, this is reported as an error. There is no sensible fix to suggest
146         // in this situation.
147         Description.Builder description = buildDescription(invocationTree);
148         description.setMessage(
149             "Calling "
150                 + ASTHelpers.getSymbol(invocationTree).getSimpleName()
151                 + " on incompatible types "
152                 + receiverType
153                 + " and "
154                 + argumentType);
155         return description.build();
156       }
157     }
