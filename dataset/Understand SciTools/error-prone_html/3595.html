51     /**
52      * Wrapper around a {@code TopLevelAnalysis} that filters analysis results to remove fixes that do
53      * not compile.
54      *
55      * @author Louis Wasserman
56      */
57     @AutoValue
58     public abstract class RecompilingTopLevelAnalysis implements TopLevelAnalysis {
59       public static RecompilingTopLevelAnalysis create(TopLevelAnalysis analysis) {
60         return new AutoValue_RecompilingTopLevelAnalysis(analysis);
61       }
62       
63       RecompilingTopLevelAnalysis() {}
64       
65       abstract TopLevelAnalysis analysis();
66     
67       @Override
68       public Set<String> knownAnalysisNames() {
69         return analysis().knownAnalysisNames();
70       }
71     
72       @Override
73       public void analyze(final CompilationUnitTree compilationUnit, final Context context,
74           AnalysesConfig configuration, final DescriptionListener listener) {
75         analysis().analyze(compilationUnit, context, configuration, new DescriptionListener() {
76     
77           @Override
78           public void onDescribed(Description description) {
79             listener.onDescribed(description.filterFixes(new Predicate<Fix>() {
80               @Override
81               public boolean apply(Fix fix) {
82                 return compiles(fix, (JCCompilationUnit) compilationUnit, context);
83               }
84             }));
85           }
86         });
87     
88         
89       }
90     
91       private enum ByStartPosition implements Comparator<Replacement> {
92         INSTANCE {
93           @Override
94           public int compare(Replacement o1, Replacement o2) {
95             return Integer.compare(o1.startPosition(), o2.startPosition());
96           }
97         };
98       }
99     
100       private static boolean compiles(Fix fix, JCCompilationUnit compilationUnit, Context context) {
101         final NavigableSet<Replacement> replacements = new TreeSet<>(ByStartPosition.INSTANCE);
102         replacements.addAll(fix.getReplacements(compilationUnit.endPositions));
103         JavaFileObject modifiedFile = compilationUnit.getSourceFile();
104     
105         if (replacements.isEmpty()) {
106           return true;
107         }
108     
109         JavacTaskImpl javacTask = (JavacTaskImpl) context.get(JavacTask.class);
110         if (javacTask == null) {
111           throw new IllegalArgumentException("No JavacTask in context.");
112         }
113         List<JavaFileObject> fileObjects = new ArrayList<>(fileObjectsForTask(javacTask));
114         for (int i = 0; i < fileObjects.size(); i++) {
115           final JavaFileObject oldFile = fileObjects.get(i);
116           if (modifiedFile.toUri().equals(oldFile.toUri())) {
117             fileObjects.set(i, new SimpleJavaFileObject(modifiedFile.toUri(), Kind.SOURCE) {
118               @Override
119               public CharSequence getCharContent(boolean ignoreEncodingErrors) throws IOException {
120                 StringBuilder builder = new StringBuilder(oldFile.getCharContent(ignoreEncodingErrors));
121                 for (Replacement replacement : replacements.descendingSet()) {
122                   builder.replace(replacement.startPosition(), replacement.endPosition(),
123                       replacement.replaceWith());
124                 }
125                 return builder;
126               }
127             });
128             break;
129           }
130         }
131         JavaCompiler compiler = JavacTool.create();
132         DiagnosticCollector<JavaFileObject> diagnosticListener = new DiagnosticCollector<>();
133         JavacTask newTask = (JavacTask) compiler.getTask(CharStreams.nullWriter(),
134             context.get(JavaFileManager.class),
135             diagnosticListener,
136             asListOrNull(argsForTask(javacTask)),
137             asListOrNull(classesForTask(javacTask)),
138             fileObjects);
139         try {
140           newTask.analyze();
141         } catch (Throwable e) {
142           return false;
143         }
144         return countErrors(diagnosticListener) == 0;
145       }
146     
147       private static int countErrors(DiagnosticCollector<JavaFileObject> diagnosticCollector) {
148         int errorCount = 0;
149         for (Diagnostic<? extends JavaFileObject> diag : diagnosticCollector.getDiagnostics()) {
150           if (diag.getKind() == Diagnostic.Kind.ERROR) {
151             errorCount++;
152           }
153         }
154         return errorCount;
155       }

157       private static Object getFieldReflectively(JavacTaskImpl task, String fieldName) {
158         try {
159           Field field = JavacTaskImpl.class.getDeclaredField(fieldName);
160           field.setAccessible(true);
161           return field.get(task);
162         } catch (ReflectiveOperationException | SecurityException | IllegalArgumentException e) {
163           throw new RuntimeException(e);
164         }
165       }
166     
167       private static <E> List<E> asListOrNull(@Nullable E[] array) {
168         return (array == null) ? null : Arrays.asList(array);
169       }
170     
171       @SuppressWarnings("unchecked")
172       private static List<JavaFileObject> fileObjectsForTask(JavacTaskImpl task) {
173         return (List<JavaFileObject>) getFieldReflectively(task, "fileObjects");
174       }
175     
176       private static String[] argsForTask(JavacTaskImpl task) {
177         return (String[]) getFieldReflectively(task, "args");
178       }
179     
180       private static String[] classesForTask(JavacTaskImpl task) {
181         return (String[]) getFieldReflectively(task, "classNames");
182       }
