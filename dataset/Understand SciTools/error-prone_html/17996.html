17     import com.google.common.collect.Ordering;
18     
19     /**
20      * Test data for {@code ReturnPlaceholderTemplate}.
21      *
22      * @author lowasser@google.com (Louis Wasserman)
23      */
24     public class ReturnPlaceholderTemplateExample {
25       public static final Ordering<String> LENGTH_THEN_LOWER_CASE_ONE_LINE = new Ordering<String>() {
26         @Override
27         public int compare(String left, String right) {
28           int lengthCmp = Integer.compare(left.length(), right.length());
29           if (lengthCmp != 0) {
30             return lengthCmp;
31           }
32           return left.toLowerCase().compareTo(right.toLowerCase());
33         }
34       };
35       
36       public static final Ordering<String> LENGTH_THEN_LOWER_CASE_MULTI_LINE = new Ordering<String>() {
37         @Override
38         public int compare(String left, String right) {
39           int lengthCmp = Integer.compare(left.length(), right.length());
40           if (lengthCmp != 0) {
41             return lengthCmp;
42           }
43           String leftLower = left.toLowerCase();
44           String rightLower = right.toLowerCase();
45           return leftLower.compareTo(rightLower);
46         }
47       };
48     }
