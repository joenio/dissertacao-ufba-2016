106     /**
107      * Checks methods annotated with {@code @NoAllocation} to ensure they really do not allocate.
108      *
109      * <ol>
110      *   <li>Calls to new are disallowed.
111      *   <li>Methods statically determined to be reachable from this method must also be annotated
112      *       with {@code @NoAllocation}.
113      *   <li>Autoboxing is disallowed.
114      *   <li>String concatenation and conversions are disallowed.
115      *   <li>To make it easier to use exceptions, allocations are always allowed within a throw
116      *       statement. (But not in the methods of nested classes if they are annotated with
117      *       {@code NoAllocation}.)
118      *   <li>The check is done at the source level. The compiler or runtime may perform optimizations
119      *       or transformations that add or remove allocations in a way not visible to this check.
120      * </ol>
121      */
122     @BugPattern(name = "NoAllocation",
123         summary = "@NoAllocation was specified on this method, but something was found that would"
124           + " trigger an allocation",
125         explanation = "Like many other languages, Java provides automatic memory management. In Java,"
126           + " this feature incurs an runtime cost, and can also lead to unpredictable execution pauses."
127           + " In most cases, this is a reasonable tradeoff, but sometimes the loss of performance or"
128           + " predictability is unacceptable. Examples include pause-sensitive user interface handlers,"
129           + " high query rate server response handlers, or other soft-realtime applications.\n\n"
130           + "In these situations, you can annotate a few carefully written methods with"
131           + " @NoAllocation. Methods with this annotation will avoid allocations in most cases,"
132           + " reducing pressure on the garbage collector. Note that allocations may still occur in"
133           + " methods with @NoAllocation if the compiler or runtime system inserts them.\n\n"
134           + "To ease the use of exceptions, allocations are allowed if they occur within a throw"
135           + " statement. But if the throw statement contains a nested class with methods annotated"
136           + " with @NoAllocation, those methods will be disallowed from allocating.",
137         category = JDK, severity = ERROR, maturity = EXPERIMENTAL)
138     public class NoAllocationChecker extends BugChecker
139         implements AssignmentTreeMatcher, BinaryTreeMatcher, CompoundAssignmentTreeMatcher,
140         EnhancedForLoopTreeMatcher, MethodInvocationTreeMatcher, NewArrayTreeMatcher,
141         NewClassTreeMatcher, ReturnTreeMatcher, TypeCastTreeMatcher, UnaryTreeMatcher,
142         VariableTreeMatcher {

144       private static final String COMMON_MESSAGE_SUFFIX =
145           "is disallowed in methods annotated with @NoAllocation";
146     
147       private static final Matcher<MethodTree> noAllocationMethodMatcher =
148           hasAnnotation(NoAllocation.class.getName());
149     
150       private static final Matcher<MethodInvocationTree> noAllocationMethodInvocationMatcher =
151           hasAnnotation(NoAllocation.class.getName());
152     
153       private static final Matcher<ExpressionTree> anyExpression = anything();
154     
155       private static final Matcher<StatementTree> anyStatement = anything();
156     
157       private static final Matcher<VariableTree> anyVariable = anything();
158     
159       private static final Matcher<ExpressionTree> isString = isSameType("java.lang.String");
160     
161       private static final Matcher<ExpressionTree> arrayExpression = isArrayType();
162     
163       private static final Matcher<ExpressionTree> primitiveExpression = isPrimitiveType();
164     
165       private static final Matcher<ExpressionTree> primitiveArrayExpression = isPrimitiveArrayType();

167       private static final Set<Kind> ALL_COMPOUND_OPERATORS =
168           Collections.unmodifiableSet(
169               EnumSet.of(
170                   AND_ASSIGNMENT,
171                   DIVIDE_ASSIGNMENT,
172                   LEFT_SHIFT_ASSIGNMENT,
173                   MINUS_ASSIGNMENT,
174                   MULTIPLY_ASSIGNMENT,
175                   OR_ASSIGNMENT,
176                   PLUS_ASSIGNMENT,
177                   REMAINDER_ASSIGNMENT,
178                   RIGHT_SHIFT_ASSIGNMENT,
179                   UNSIGNED_RIGHT_SHIFT_ASSIGNMENT,
180                   XOR_ASSIGNMENT));

182       /**
183        * Matches if a Tree has a ThrowTree or an AnnotationTree before any MethodTree in its hierarchy.
184        * We don't want the throw to nullify any {@code @NoAnnotation} in a method in an anonymous
185        * class below it.
186        */
187       private static final Matcher<Tree> withinThrowOrAnnotation = new Matcher<Tree>() {
188         @Override
189         public boolean matches(Tree tree, VisitorState state) {
190           // TODO(agoode): Make this accept statements in a block that definitely will lead to a throw.
191           TreePath path = state.getPath().getParentPath();
192           while (path != null) {
193             Tree node = path.getLeaf();
194             state = state.withPath(path);
195             switch (node.getKind()) {
196               case METHOD:
197                 // We've gotten to the top of the method without finding a throw.
198                 return false;
199               case THROW:
200               case ANNOTATION:
201                 return true;
202               default:
203                 path = path.getParentPath();
204             }
205           }
206           return false;
207         }
208       };

211        * Matches a new array statement if the enclosing method is annotated with
212        * {@code @NoAllocation}.

214       private static final Matcher<NewArrayTree> newArrayMatcher =
215           allOf(
216             not(withinThrowOrAnnotation),
217             enclosingMethod(noAllocationMethodMatcher)
218           );

221        * Matches a new statement if the enclosing method is annotated with {@code @NoAllocation}.
222        */
223       private static final Matcher<NewClassTree> newClassMatcher =
224           allOf(
225             not(withinThrowOrAnnotation),
226             enclosingMethod(noAllocationMethodMatcher)
227           );

229       /**
230        * Matches if a method without {@code @NoAllocation} is invoked from a method with
231        * {@code @NoAllocation}.
232        */
233       private static final Matcher<MethodInvocationTree> methodMatcher =
234           allOf(
235             not(withinThrowOrAnnotation),
236             enclosingMethod(noAllocationMethodMatcher),
237             not(noAllocationMethodInvocationMatcher)
238           );

240       /**
241        * Matches string concatenation. Includes all string conversions.
242        */
243       private static final Matcher<BinaryTree> stringConcatenationMatcher =
244           allOf(
245             not(withinThrowOrAnnotation),
246             enclosingMethod(noAllocationMethodMatcher),
247             kindIs(PLUS),
248             binaryTree(anyExpression, isString)
249           );
250     
251       /**
252        * Matches string and boxing compound assignment.
253        */
254       private static final Matcher<CompoundAssignmentTree> compoundAssignmentMatcher =
255           allOf(
256               not(withinThrowOrAnnotation),
257               enclosingMethod(noAllocationMethodMatcher),
258               anyOf(
259                   compoundAssignment(PLUS_ASSIGNMENT, isString, anyExpression),
260                   compoundAssignment(ALL_COMPOUND_OPERATORS, not(primitiveExpression), anyExpression)));
261     
262       /**
263        * Matches if foreach is used on a non-array or if boxing occurs with an array.
264        */
265       private static final Matcher<EnhancedForLoopTree> foreachMatcher =
266           allOf(
267             not(withinThrowOrAnnotation),
268             enclosingMethod(noAllocationMethodMatcher),
269             anyOf(
270               not(enhancedForLoop(anyVariable, arrayExpression, anyStatement)),
271               enhancedForLoop(variableType(not(isPrimitiveType())), primitiveArrayExpression,
272                 anyStatement)
273             )
274           );

277        * Matches boxing assignment.
278        */
279       private static final Matcher<AssignmentTree> boxingAssignment =
280           allOf(
281             not(withinThrowOrAnnotation),
282             enclosingMethod(noAllocationMethodMatcher),
283             assignment(not(primitiveExpression), primitiveExpression)
284           );

286       /**
287        * Matches boxing during variable initialization.

289       private static final Matcher<VariableTree> boxingInitialization =
290           allOf(
291             not(withinThrowOrAnnotation),
292             enclosingMethod(noAllocationMethodMatcher),
293             variableInitializer(primitiveExpression),

298        * Matches boxing by explicit cast.
299        */
300       private static final Matcher<TypeCastTree> boxingCast =
301           allOf(
302             not(withinThrowOrAnnotation),
303             enclosingMethod(noAllocationMethodMatcher),

307       /**
308        * Matches boxing by return.
309        */
310       private static final Matcher<ReturnTree> boxingReturn = new Matcher<ReturnTree>() {
311         @Override
312         public boolean matches(ReturnTree tree, VisitorState state) {
313           return allOf(
314             not(withinThrowOrAnnotation),
315             enclosingMethod(allOf(noAllocationMethodMatcher, methodReturnsNonPrimitiveType())),
316             isPrimitiveType()
317           ).matches(tree.getExpression(), state);
318         }
319       };
320     
321       /**
322        * Matches boxing by method invocation, including varargs.
323        */
324       private static final Matcher<MethodInvocationTree> boxingInvocation =
325           new Matcher<MethodInvocationTree>() {
326         @Override
327         public boolean matches(MethodInvocationTree tree, VisitorState state) {
328           if (!enclosingMethod(noAllocationMethodMatcher).matches(tree, state)) {
329             return false;
330           }
331     
332           // Get the arguments.
333           JCMethodInvocation methodInvocation = (JCMethodInvocation) tree;
334           List<JCExpression> arguments = methodInvocation.getArguments();
335     
336           // Get the parameters.
337           MethodSymbol methodSymbol = ASTHelpers.getSymbol(tree);
338           List<VarSymbol> params = methodSymbol.getParameters();
339     
340           // If there is a length mismatch, this implies varargs boxing.
341           if (arguments.size() != params.size()) {
342             return true;
343           }
344     
345           // Check for boxing at each argument.
346           int numArgs = arguments.size();
347           int i = 0;
348           Iterator<JCExpression> argument = arguments.iterator();
349           Iterator<VarSymbol> param = params.iterator();
350           while (param.hasNext() && argument.hasNext()) {
351             JCExpression a = argument.next();
352             VarSymbol p = param.next();
353     
354             if (a.type.isPrimitive() && !p.type.isPrimitive()) {
355               // Boxing occurs here.
356               return true;
357             }
358     
359             // Check last parameter. If unassignable, this implies varargs boxing.
360             if (i == numArgs - 1) {
361               if (!state.getTypes().isAssignable(a.type, p.type)) {
362                 return true;
363               }
364             }
365             i++;
366           }
367     
368           return false;
369         }
370       };

372       /**
373        * Matches boxing by unary operator.
374        */
375       private static final Matcher<UnaryTree> boxingUnary = new Matcher<UnaryTree>() {
376         @Override
377         public boolean matches(UnaryTree tree, VisitorState state) {
378           return allOf(
379             not(withinThrowOrAnnotation),
380             enclosingMethod(noAllocationMethodMatcher),
381             anyOf(
382               kindIs(POSTFIX_DECREMENT),
383               kindIs(POSTFIX_INCREMENT),
384               kindIs(PREFIX_DECREMENT),
385               kindIs(PREFIX_INCREMENT)
386             )
387           ).matches(tree, state)
388           && not(isPrimitiveType()).matches(tree, state);
389         }
390       };

392       @Override
393       public Description matchNewArray(NewArrayTree tree, VisitorState state) {
394         if (!newArrayMatcher.matches(tree, state)) {
395           return Description.NO_MATCH;
396         }
397         return buildDescription(tree)
398             .setMessage("Allocating a new array with \"new\" or \"{ ... }\" " + COMMON_MESSAGE_SUFFIX)
399             .build();
400       }
401     
402       @Override
403       public Description matchNewClass(NewClassTree tree, VisitorState state) {
404         if (!newClassMatcher.matches(tree, state)) {
405           return Description.NO_MATCH;
406         }
407         return buildDescription(tree)
408             .setMessage("Constructing a new object " + COMMON_MESSAGE_SUFFIX)
409             .build();
410       }
411     
412       @Override
413       public Description matchMethodInvocation(MethodInvocationTree tree, VisitorState state) {
414         if (!methodMatcher.matches(tree, state) && !boxingInvocation.matches(tree, state)) {
415           return Description.NO_MATCH;
416         }
417         return buildDescription(tree)
418             .setMessage("Calling a method that is not annotated with @NoAllocation, calling a varargs"
419               + " method without exactly matching the signature, or passing a primitive value as"
420               + " non-primitive method argument " + COMMON_MESSAGE_SUFFIX)
421             .build();
422       }
423     
424       @Override
425       public Description matchBinary(BinaryTree tree, VisitorState state) {
426         if (!stringConcatenationMatcher.matches(tree, state)) {
427           return Description.NO_MATCH;
428         }
429         return buildDescription(tree)
430             .setMessage("String concatenation allocates a new String, which " + COMMON_MESSAGE_SUFFIX)
431             .build();
432       }
433     
434       @Override
435       public Description matchCompoundAssignment(CompoundAssignmentTree tree, VisitorState state) {
436         if (!compoundAssignmentMatcher.matches(tree, state)) {
437           return Description.NO_MATCH;
438         }
439         return buildDescription(tree)
440             .setMessage("Compound assignment to a String or boxed primitive allocates a new object,"
441               + " which " + COMMON_MESSAGE_SUFFIX)
442             .build();
443       }
444     
445       @Override
446       public Description matchEnhancedForLoop(EnhancedForLoopTree tree, VisitorState state) {
447         if (!foreachMatcher.matches(tree, state)) {
448           return Description.NO_MATCH;
449         }
450         return buildDescription(tree)
451             .setMessage("Iterating over a Collection or iterating over a primitive array using a"
452               + " non-primitive element type will trigger allocation, which " + COMMON_MESSAGE_SUFFIX)
453             .build();
454       }
455     
456       @Override
457       public Description matchAssignment(AssignmentTree tree, VisitorState state) {
458         if (!boxingAssignment.matches(tree, state)) {
459           return Description.NO_MATCH;
460         }
461         return buildDescription(tree)
462             .setMessage("Assigning a primitive value to a non-primitive variable or array element"
463               + " will autobox the value, which " + COMMON_MESSAGE_SUFFIX)
464             .build();
465       }
466     
467       @Override
468       public Description matchVariable(VariableTree tree, VisitorState state) {
469         if (!boxingInitialization.matches(tree, state)) {
470           return Description.NO_MATCH;
471         }
472         return buildDescription(tree)
473             .setMessage("Initializing a non-primitive variable with a primitive value will autobox the"
474               + " value, which " + COMMON_MESSAGE_SUFFIX)
475             .build();
476       }
477     
478       @Override
479       public Description matchTypeCast(TypeCastTree tree, VisitorState state) {
480         if (!boxingCast.matches(tree, state)) {
481           return Description.NO_MATCH;
482         }
483         return buildDescription(tree)
484             .setMessage("Casting a primitive value to a non-primitive type will autobox the value,"
485               + " which " + COMMON_MESSAGE_SUFFIX)
486             .build();
487       }
488     
489       @Override
490       public Description matchReturn(ReturnTree tree, VisitorState state) {
491         if (!boxingReturn.matches(tree, state)) {
492           return Description.NO_MATCH;
493         }
494         return buildDescription(tree)
495             .setMessage("Returning a primitive value from a method with a non-primitive return type"
496               + " will autobox the value, which " + COMMON_MESSAGE_SUFFIX)
497             .build();
498       }
499     
500       @Override
501       public Description matchUnary(UnaryTree tree, VisitorState state) {
502         if (!boxingUnary.matches(tree, state)) {
503           return Description.NO_MATCH;
504         }
505         return buildDescription(tree)
506             .setMessage("Pre- and post- increment/decrement operations on a non-primitive variable or"
507               + " array element will autobox the result, which " + COMMON_MESSAGE_SUFFIX)
508             .build();
509       }
