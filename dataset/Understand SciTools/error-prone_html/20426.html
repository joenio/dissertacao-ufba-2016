46       @Test
47       public void testMatches_matchesLiteralsAndStaticFinals() throws Exception {
48         String[] lines = {
49           "package test;",
50           "import com.google.errorprone.annotations.CompileTimeConstant;",
51           "public class CompileTimeConstantExpressionMatcherTestCase {",
52           "  private final String final_string = \"bap\";",
53           "  private final int final_int = 29;",
54           "  private static final int static_final_int = 29;",
55           "  public void m() { ",
56           "    String s1; s1 = \"boop\"; s1 = \"boop\" + final_string;",
57           "    int int2; int2 = 42;",
58           "    Integer int3; int3 = 42 * final_int; int3 = 12 - static_final_int;",
59           "    boolean bool4; bool4 = false;",
60           "  }",
61           "}"
62         };
63     
64         Map<String, Boolean> expectedMatches = new HashMap<String, Boolean>();
65         expectedMatches.put("s1", true);
66         expectedMatches.put("int2", true);
67         expectedMatches.put("int3", true);
68         expectedMatches.put("bool4", true);
69         assertCompilerMatchesOnAssignment(expectedMatches, lines);
70       }

72       @Test
73       public void testMatches_nullLiteral() throws Exception {
74         String[] lines = {
75           "package test;",
76           "import com.google.errorprone.annotations.CompileTimeConstant;",
77           "public class CompileTimeConstantExpressionMatcherTestCase {",
78           "  private static final String static_final_string = null;",
79           "  public void m() { ",
80           "    String s1; s1 = null;",
81           "    String s2; s2 = static_final_string;",
82           "  }",
83           "}"
84         };
85         Map<String, Boolean> expectedMatches = new HashMap<String, Boolean>();
86         expectedMatches.put("s1", true);
87         // Even though s2 has the compile-time constant value "null", it's not
88         // a literal.  I don't know how to distinguish this, but I doubt this is
89         // an important use case.
90         expectedMatches.put("s2", false);
91         assertCompilerMatchesOnAssignment(expectedMatches, lines);
92       }

180       // Helper methods.
181       private void assertCompilerMatchesOnAssignment(
182           final Map<String, Boolean> expectedMatches, String... lines) {
183         final Matcher<ExpressionTree> matcher = new CompileTimeConstantExpressionMatcher();
184         final Scanner scanner = new Scanner() {
185           @Override
186           public Void visitAssignment(AssignmentTree t, VisitorState state) {
187             ExpressionTree lhs = t.getVariable();
188             if (expectedMatches.containsKey(lhs.toString())) {
189               boolean matches = matcher.matches(t.getExpression(), state);
190               if (expectedMatches.get(lhs.toString())) {
191                 assertTrue("Matcher should match expression" + t.getExpression(), matches);
192               } else {
193                 assertFalse("Matcher should not match expression" + t.getExpression(), matches);
194               }
195             }
196             return super.visitAssignment(t, state);
197           }
198         };
199     
200         CompilationTestHelper
201             .newInstance(ScannerSupplier.fromScanner(scanner), getClass())
202             .expectResult(Result.OK)
203             .addSourceLines("test/CompileTimeConstantExpressionMatcherTestCase.java", lines)
204             .doTest();
205       }
