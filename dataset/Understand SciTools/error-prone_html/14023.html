45     /**
46      * A symbol resolver used while binding guardedby expressions from string literals.
47      *
48      * @author cushon@google.com (Liam Miller-Cushon)
49      */
50     public class GuardedBySymbolResolver implements GuardedByBinder.Resolver {
51     
52       private final ClassSymbol enclosingClass;
53       private final Tree decl;
54       private final JCTree.JCCompilationUnit compilationUnit;
55       private final Context context;
56       private final Types types;
57     
58       public static GuardedBySymbolResolver from(Tree tree, VisitorState visitorState) {
59         return GuardedBySymbolResolver.from(
60             (ClassSymbol) ASTHelpers.getSymbol(tree).owner,
61             visitorState.getPath().getCompilationUnit(),
62             visitorState.context,
63             tree);
64       }
65     
66       public static GuardedBySymbolResolver from(ClassSymbol owner,
67           CompilationUnitTree compilationUnit, Context context, Tree leaf) {
68         return new GuardedBySymbolResolver(owner, compilationUnit, context, leaf);
69       }
70     
71       private GuardedBySymbolResolver(ClassSymbol enclosingClass,
72           CompilationUnitTree compilationUnit, Context context, Tree leaf) {
73         this.compilationUnit = (JCCompilationUnit) compilationUnit;
74         this.enclosingClass = enclosingClass;
75         this.context = context;
76         this.types = Types.instance(context);
77         this.decl = leaf;
78       }
79     
80       public Context context() {
81         return context;
82       }
83     
84       public ClassSymbol enclosingClass() {
85         return enclosingClass;
86       }

88       @Override
89       public Symbol resolveIdentifier(IdentifierTree node) {
90         String name = node.getName().toString();
91     
92         if (name.equals("this")) {
93           return enclosingClass;
94         }
95     
96         // TODO(cushon): consider disallowing this? It's the only case where the lock description
97         // isn't legal java.
98         if (name.equals("itself")) {
99           if (decl instanceof VariableTree) {
100             name = ((VariableTree) decl).getName().toString();
101           } else if (decl instanceof MethodTree) {
102             name = ((MethodTree) decl).getName().toString();
103           } else {
104             throw new IllegalGuardedBy(decl.getClass().toString());
105           }
106           return getField(enclosingClass, name);
107         }
108     
109         Symbol.VarSymbol field = getField(enclosingClass, name);
110         if (field != null) {
111           return field;
112         }
113     
114         Symbol type = resolveType(name, SearchSuperTypes.YES);
115         if (type != null) {
116           return type;
117         }
118     
119         throw new IllegalGuardedBy(name);
120       }
121     
122       @Override
123       public Symbol.MethodSymbol resolveMethod(MethodInvocationTree node,
124           javax.lang.model.element.Name name) {
125         return getMethod(enclosingClass, name.toString());
126       }
127     
128       @Override
129       public Symbol.MethodSymbol resolveMethod(MethodInvocationTree node, GuardedByExpression base,
130           javax.lang.model.element.Name identifier) {
131         Symbol baseSym = base.kind() == GuardedByExpression.Kind.THIS
132             ? enclosingClass
133             : base.type().asElement();
134         return getMethod(baseSym, identifier.toString());
135       }
136     
137       private Symbol.MethodSymbol getMethod(Symbol classSymbol, String name) {
138         return getMember(Symbol.MethodSymbol.class, ElementKind.METHOD, classSymbol, name);
139       }
140     
141       @Override
142       public Symbol resolveSelect(GuardedByExpression base, MemberSelectTree node) {
143         Symbol baseSym = base.kind() == GuardedByExpression.Kind.THIS
144             ? enclosingClass
145             : base.type().asElement();
146         return getField(baseSym, node.getIdentifier().toString());
147       }
148     
149       private Symbol.VarSymbol getField(Symbol classSymbol, String name) {
150         return getMember(Symbol.VarSymbol.class, ElementKind.FIELD, classSymbol, name);
151       }
152     
153       private <T extends Symbol> T getMember(Class<T> type, ElementKind kind, Symbol classSymbol,
154           String name) {
155         if (classSymbol.type == null) {
156           return null;
157         }
158         for (Type t : types.closure(classSymbol.type)) {
159           Scope scope = t.tsym.members();
160           for (Symbol sym : scope.getSymbolsByName(getName(name))) {
161             if (sym.getKind().equals(kind)) {
162               return type.cast(sym);
163             }
164           }
165         }
166         if (classSymbol.owner != null && classSymbol != classSymbol.owner
167             && classSymbol.owner instanceof Symbol.ClassSymbol) {
168           T sym = getMember(type, kind, classSymbol.owner, name);
169           if (sym != null) {
170             return sym;
171           }
172         }
173         if (classSymbol.hasOuterInstance()) {
174           T sym = getMember(type, kind, classSymbol.type.getEnclosingType().asElement(), name);
175           if (sym != null) {
176             return sym;
177           }
178         }
179         return null;
180       }
181     
182       @Override
183       public Symbol resolveTypeLiteral(ExpressionTree expr) {
184         checkGuardedBy(expr instanceof IdentifierTree, "bad type literal: %s", expr);
185         IdentifierTree ident = (IdentifierTree) expr;
186         Symbol type = resolveType(ident.getName().toString(), SearchSuperTypes.YES);
187         if (type instanceof Symbol.ClassSymbol) {
188           return type;
189         }
190         return null;
191       }
192     
193       private static enum SearchSuperTypes { YES, NO }
194     
195       /**
196        * Resolves a simple name as a type. Considers super classes, lexically enclosing classes, and
197        * then arbitrary types available in the current environment.
198        */
199       private Symbol resolveType(String name, SearchSuperTypes searchSuperTypes) {
200         Symbol type = null;
201         if (searchSuperTypes == SearchSuperTypes.YES) {
202           type = getSuperType(enclosingClass, name);
203         }
204         if (enclosingClass.getSimpleName().contentEquals(name)) {
205           type = enclosingClass;
206         }
207         if (type == null) {
208           type = getLexicallyEnclosing(enclosingClass, name);
209         }
210         if (type == null) {
211           type = attribIdent(name);
212         }
213         checkGuardedBy(!(type instanceof Symbol.PackageSymbol),
214             "All we could find for '%s' was a package symbol.", name);
215         return type;
216       }
217     
218       private Symbol getSuperType(Symbol symbol, String name) {
219         for (Type t : types.closure(symbol.type)) {
220           if (t.asElement().getSimpleName().contentEquals(name)) {
221             return t.asElement();
222           }
223         }
224         return null;
225       }
226     
227       private Symbol getLexicallyEnclosing(ClassSymbol symbol, String name) {
228         Symbol current = symbol.owner;
229         while (true) {
230           if (current == null || current.getSimpleName().contentEquals(name)) {
231             return current;
232           }
233     
234           if (current != current.owner && current.owner instanceof Symbol.ClassSymbol) {
235             current = current.owner;
236           } else {
237             return null;
238           }
239         }
240       }
241     
242       private Symbol attribIdent(String name) {
243         Attr attr = Attr.instance(context);
244         TreeMaker tm = TreeMaker.instance(context);
245         return attr.attribIdent(tm.Ident(getName(name)), compilationUnit);
246       }
247     
248       private Name getName(String name) {
249         return Names.instance(context).fromString(name);
250       }
251     
252       @Override
253       public Symbol resolveEnclosingClass(ExpressionTree expr) {
254         checkGuardedBy(expr instanceof IdentifierTree, "bad type literal: %s", expr);
255         IdentifierTree ident = (IdentifierTree) expr;
256         Symbol type = resolveType(ident.getName().toString(), SearchSuperTypes.NO);
257         if (type instanceof Symbol.ClassSymbol) {
258           return type;
259         }
260         return null;
261       }
