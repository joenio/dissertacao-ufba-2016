15     package com.google.errorprone.refaster.testdata.template;
16     
17     import com.google.common.collect.ComparisonChain;
18     import com.google.errorprone.refaster.annotation.AfterTemplate;
19     import com.google.errorprone.refaster.annotation.BeforeTemplate;
20     import com.google.errorprone.refaster.annotation.Placeholder;
21     
22     /**
23      * Example use of placeholder methods in a {@code ComparisonChain} refactoring.
24      */
25     public abstract class ComparisonChainTemplate<A extends Comparable<A>, B extends Comparable<B>> {
26       @Placeholder(allowsIdentity = true) abstract <T> A propertyA(T t);
27       @Placeholder(allowsIdentity = true) abstract <T> B propertyB(T t);
28       
29       @BeforeTemplate <T> int before(T left, T right) {
30         int cmp = propertyA(left).compareTo(propertyA(right));
31         if (cmp == 0) {
32           return propertyB(left).compareTo(propertyB(right));
33         } else {
34           return cmp;
35         }
36       }
37       
38       @AfterTemplate <T> int after(T left, T right) {
39         return ComparisonChain.start()
40             .compare(propertyA(left), propertyA(right))
41             .compare(propertyB(left), propertyB(right))
42             .result();
43       }
44     }
