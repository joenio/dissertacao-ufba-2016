44     /**
45      * An abstract base class to match method invocations in which the return value is not used.
46      *
47      * @author eaftan@google.com (Eddie Aftandilian)
48      */
49     public abstract class AbstractReturnValueIgnored extends BugChecker
50         implements MethodInvocationTreeMatcher {
51     
52       @Override
53       public Description matchMethodInvocation(
54           MethodInvocationTree methodInvocationTree, VisitorState state) {
55         if (allOf(
56             parentNode(Matchers.<MethodInvocationTree>kindIs(Kind.EXPRESSION_STATEMENT)),
57             not(methodSelect(
58                 Matchers.<ExpressionTree>allOf(kindIs(Kind.IDENTIFIER), identifierHasName("super")))),
59             specializedMatcher())
60             .matches(methodInvocationTree, state)) {
61           return describe(methodInvocationTree, state);
62         }
63         return Description.NO_MATCH;
64       }
65     
66       /**
67        * Match whatever additional conditions concrete subclasses want to match (a list of known
68        * side-effect-free methods, has a @CheckReturnValue annotation, etc.).
69        */
70       public abstract Matcher<? super MethodInvocationTree> specializedMatcher();
71     
72       private static Matcher<ExpressionTree> identifierHasName(final String name) {
73         return new Matcher<ExpressionTree>() {
74           @Override
75           public boolean matches(ExpressionTree item, VisitorState state) {
76             return ((IdentifierTree) item).getName().contentEquals(name);
77           }
78         };
79       }
80     
81       /**
82        * Fixes the error by assigning the result of the call to the receiver reference, or deleting
83        * the method call.
84        */
85       public Description describe(MethodInvocationTree methodInvocationTree, VisitorState state) {
86         // Find the root of the field access chain, i.e. a.intern().trim() ==> a.
87         ExpressionTree identifierExpr = ASTHelpers.getRootAssignable(methodInvocationTree);
88         String identifierStr = null;
89         Type identifierType = null;
90         if (identifierExpr != null) {
91           identifierStr = identifierExpr.toString();
92           if (identifierExpr instanceof JCIdent) {
93             identifierType = ((JCIdent) identifierExpr).sym.type;
94           } else if (identifierExpr instanceof JCFieldAccess) {
95             identifierType = ((JCFieldAccess) identifierExpr).sym.type;
96           } else {
97             throw new IllegalStateException("Expected a JCIdent or a JCFieldAccess");
98           }
99         }
100     
101         Type returnType = ASTHelpers.getReturnType(
102             ((JCMethodInvocation) methodInvocationTree).getMethodSelect());
103     
104         Fix fix;
105         if (identifierStr != null && !"this".equals(identifierStr) && returnType != null &&
106             state.getTypes().isAssignable(returnType, identifierType)) {
107           // Fix by assigning the assigning the result of the call to the root receiver reference.
108           fix = SuggestedFix.prefixWith(methodInvocationTree, identifierStr + " = ");
109         } else {
110           // Unclear what the programmer intended.  Delete since we don't know what else to do.
111           Tree parent = state.getPath().getParentPath().getLeaf();
112           fix = SuggestedFix.delete(parent);
113         }
114         return describeMatch(methodInvocationTree, fix);
115       }
116     }
