15     package com.google.errorprone.refaster.annotation;
16     
17     import java.lang.annotation.ElementType;
18     import java.lang.annotation.Retention;
19     import java.lang.annotation.RetentionPolicy;
20     import java.lang.annotation.Target;
21     
22     /**
23      * Annotation to indicate a placeholder method.
24      *
25      * <p>A placeholder method is an abstract method in a Refaster template class which can represent
26      * an arbitrary expression (if the return type is nonvoid), or zero or more statements
27      * (if the return type is void), in terms of its arguments.  For example,
28      *
29      * <pre><code>
30      * abstract class ComputeIfAbsent&lt;K, V&gt; {
31      *  {@literal @}Placeholder abstract V computeValue(K key);
32      *
33      *  {@literal @}BeforeTemplate void getOrCompute(Map&lt;K, V&gt; map, K key) {
34      *     V value = map.get(key);
35      *     if (value == null) {
36      *       map.put(key, value = computeValue(key));
37      *     }
38      *   }
39      *
40      *  {@literal @}AfterTemplate void computeIfAbsent(Map&lt;K, V&gt; map, K key) {
41      *     V value = map.computeIfAbsent(key, (K k) -&gt; computeValue(k));
42      *   }
43      * }
44      * </code></pre>
45      *
46      * <p>Here, {@code computeValue} represents an arbitrary expression in terms of {@code key},
47      * and the {@code @AfterTemplate} rewrites that same expression in terms of the parameter of a
48      * lambda expression.
49      *
50      * <p>For a multi-line example, consider
51      *
52      * <pre><code>
53      * abstract class TryWithResources&lt;T extends AutoCloseable&gt; {
54      *  {@literal @}Placeholder abstract T open();
55      *  {@literal @}Placeholder void process(T resource);
56      *
57      *  {@literal @}BeforeTemplate void tryFinallyClose() {
58      *     T resource = open();
59      *     try {
60      *       process(resource);
61      *     } finally {
62      *       resource.close();
63      *     }
64      *   }
65      *
66      *  {@literal @}AfterTemplate void tryWithResource() {
67      *     try (T resource = open()) {
68      *       process(resource);
69      *     }
70      *   }
71      * }
72      * </code></pre>
73      *
74      * <p>Here, {@code process} is any block, though it must refer to {@code resource} in some way;
75      * it is not permitted to reassign the contents of {@code resource}.
76      *
77      * <p>Placeholder methods are not permitted to refer to any local variables or parameters of the
78      * {@code @BeforeTemplate} that are not passed to them as arguments, and must currently contain
79      * references to all arguments that <em>are</em> passed to them.
80      *
81      * @author lowasser@google.com (Louis Wasserman)
82      */
83     @Target(ElementType.METHOD)
84     @Retention(RetentionPolicy.SOURCE)
85     public @interface Placeholder {
86       // TODO(lowasser): consider putting forbiddenKinds here as an annotation parameter
87     
88       /**
89        * Identifies whether the placeholder is allowed to match an expression which simply returns one
90        * of the placeholder arguments unchanged.
91        */
92       boolean allowsIdentity() default false;
93     }
