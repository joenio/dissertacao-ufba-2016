52     /**
53      * @author scottjohnson@google.com (Scott Johnson)
54      */
55     @BugPattern(name = "SelfEquality",
56         summary = "Variable compared to itself",
57         explanation = "There is no good reason to test a primitive value or reference for equality " +
58               "with itself.",
59         category = JDK, severity = ERROR, maturity = EXPERIMENTAL)
60     public class SelfEquality extends BugChecker implements BinaryTreeMatcher {
61     
62       @Override
63       public Description matchBinary(BinaryTree tree, VisitorState state) {
64         if (!(anyOf(kindIs(EQUAL_TO), kindIs(NOT_EQUAL_TO)).matches(tree, state)
65             && ASTHelpers.sameVariable(tree.getLeftOperand(), tree.getRightOperand()))) {
66           return Description.NO_MATCH;
67         }
68     
69         StringBuilder fixedExpression = new StringBuilder();
70         Fix fix = null;
71     
72         ExpressionTree leftOperand = tree.getLeftOperand();
73         ExpressionTree rightOperand = tree.getRightOperand();
74         Type leftType = ((JCTree) leftOperand).type;
75         Types types = state.getTypes();
76         Symtab symtab = state.getSymtab();
77     
78         /**
79          * Try to figure out what they were trying to do.
80          * Cases:
81          * 1) (foo == foo) ==> (foo == other.foo)
82          * 2) (foo == this.foo) ==> (other.foo == this.foo)
83          * 3) (this.foo == foo) ==> (this.foo == other.foo)
84          * 4) (this.foo == this.foo) ==> (this.foo == other.foo)
85          */
86     
87         // Choose argument to replace.
88         ExpressionTree toReplace;
89         if (rightOperand.getKind() == Kind.IDENTIFIER) {
90           toReplace = rightOperand;
91         } else if (leftOperand.getKind() == Kind.IDENTIFIER) {
92           toReplace = leftOperand;
93         } else {
94           // If we don't have a good reason to replace one or the other, replace the second.
95           toReplace = rightOperand;
96         }
97     
98         // Find containing block
99         TreePath path = state.getPath();
100         while (path.getLeaf() != null && path.getLeaf().getKind() != Kind.CLASS
101             && path.getLeaf().getKind() != Kind.BLOCK) {
102           path = path.getParentPath();
103         }
104         if (path.getLeaf() != null) {
105           List<? extends JCTree> members;
106           // Must be block or class
107           if (path.getLeaf().getKind() == Kind.CLASS) {
108             members = ((JCClassDecl) path.getLeaf()).getMembers();
109           } else {
110             members = ((JCBlock) path.getLeaf()).getStatements();
111           }
112           for (JCTree jcTree : members) {
113             if (jcTree.getKind() == Kind.VARIABLE) {
114               JCVariableDecl declaration = (JCVariableDecl) jcTree;
115               TypeSymbol variableTypeSymbol = declaration.getType().type.tsym;
116     
117               if (ASTHelpers.getSymbol(toReplace).isMemberOf(variableTypeSymbol, state.getTypes())) {
118                 if (toReplace.getKind() == Kind.IDENTIFIER) {
119                   fix = SuggestedFix.prefixWith(toReplace, declaration.getName() + ".");
120                 } else {
121                   fix = SuggestedFix.replace(
122                       ((JCFieldAccess) toReplace).getExpression(), declaration.getName().toString());
123                 }
124               }
125             }
126           }
127         }
128     
129         if (fix == null) {
130           // No good replacement, let's try something else!
131     
132           // For floats or doubles, y!=y -> isNaN(y)
133           if (tree.getKind() == Tree.Kind.EQUAL_TO) {
134             fixedExpression.append("!");
135           }
136           if (types.isSameType(leftType, symtab.floatType)) {
137             fixedExpression.append("Float.isNaN(" + leftOperand + ")");
138             fix = SuggestedFix.replace(tree, fixedExpression.toString());
139           } else if (types.isSameType(leftType, symtab.doubleType)) {
140             fixedExpression.append("Double.isNaN(" + leftOperand + ")");
141             fix = SuggestedFix.replace(tree, fixedExpression.toString());
142           } else {
143     
144             // last resort, just replace with true or false
145             if (tree.getKind() == Tree.Kind.EQUAL_TO) {
146               fix = SuggestedFix.replace(tree, "true");
147             } else {
148               fix = SuggestedFix.replace(tree, "false");
149             }
150           }
151         }
152     
153         return (fix == null) ? describeMatch(tree) : describeMatch(tree, fix);
154       }
155     }
