46     /**
47      * @author ptoomey@google.com (Patrick Toomey)
48      */
49     @BugPattern(name = "StringEquality",
50         summary = "String comparison using reference equality instead of value equality",
51         explanation = "Strings are compared for reference equality/inequality using == or !="
52             + "instead of for value equality using .equals()",
53         category = JDK, severity = WARNING, maturity = MATURE)
54     public class StringEquality extends BugChecker implements BinaryTreeMatcher {
55     
56       private final NullnessAnalysis nullnessAnalysis = new NullnessAnalysis();
57     
58       /**
59        *  A {@link Matcher} that matches whether the operands in a {@link BinaryTree} are
60        *  strictly String operands.  For Example, if either operand is {@code null} the matcher
61        *  will return {@code false}
62        */
63       private static final Matcher<BinaryTree> STRING_OPERANDS = new Matcher<BinaryTree>() {
64         @Override
65         public boolean matches(BinaryTree tree, VisitorState state) {
66           Type stringType = state.getSymtab().stringType;
67           ExpressionTree leftOperand = tree.getLeftOperand();
68           Type leftType = ((JCTree.JCExpression) leftOperand).type;
69           // The left operand is not a String (ex. null) so no match
70           if (!state.getTypes().isSameType(leftType, stringType)) {
71             return false;
72           }
73           ExpressionTree rightOperand = tree.getRightOperand();
74           Type rightType = ((JCTree.JCExpression) rightOperand).type;
75           // We know that both operands are String objects
76           return state.getTypes().isSameType(rightType, stringType);
77         }
78       };
79     
80       public static final Matcher<BinaryTree> MATCHER = allOf(
81           anyOf(kindIs(EQUAL_TO), kindIs(NOT_EQUAL_TO)),
82           STRING_OPERANDS);
83     
84       /* Match string that are compared with == and != */
85       @Override
86       public Description matchBinary(BinaryTree tree, final VisitorState state) {
87         if (!MATCHER.matches(tree, state)) {
88           return Description.NO_MATCH;
89         }
90     
91         SuggestedFix.Builder fix = SuggestedFix.builder();
92     
93         // Consider one of the tree's operands. If it is "", and the other is non-null,
94         // then call isEmpty on the other.
95         StringBuilder fixExpr = considerOneOf(tree.getLeftOperand(), tree.getRightOperand(),
96             new HandleChoice<ExpressionTree, StringBuilder>() {
97               @Override
98               public StringBuilder apply(ExpressionTree it, ExpressionTree other) {
99                 return "".equals(constValue((JCTree) it)) && isNonNull(other, state)
100                        ? methodCall(other, "isEmpty") : null;
101               }
102             });
103     
104         if (fixExpr == null) {
105           // Consider one of the tree's operands. If it is non-null,
106           // then call equals on it, passing the other operand as argument.
107           fixExpr = considerOneOf(tree.getLeftOperand(), tree.getRightOperand(),
108               new HandleChoice<ExpressionTree, StringBuilder>() {
109                 @Override
110                 public StringBuilder apply(ExpressionTree it, ExpressionTree other) {
111                   return isNonNull(it, state) ? methodCall(it, "equals", other) : null;
112                 }
113               });
114     
115           if (fixExpr == null) {
116             fixExpr = methodCall(
117                 null, "Objects.equals", tree.getLeftOperand(), tree.getRightOperand());
118             fix.addImport("java.util.Objects");
119           }
120         }
121     
122         if (tree.getKind() == Tree.Kind.NOT_EQUAL_TO) {
123           fixExpr.insert(0, "!");
124         }
125     
126         fix.replace(tree, fixExpr.toString());
127         return describeMatch(tree, fix.build());
128       }
129     
130       private interface HandleChoice<T, R> {
131         R apply(T it, T other);
132       }
133     
134       private static <T, R> R considerOneOf(final T a, final T b, final HandleChoice<T, R> f) {
135         R r = f.apply(a, b);
136         return r == null ? f.apply(b, a) : r;
137       }
138     
139       private boolean isNonNull(ExpressionTree expr, VisitorState state) {
140         TreePath pathToExpr = new TreePath(state.getPath(), expr);
141         return nullnessAnalysis.getNullness(pathToExpr, state.context) == NONNULL;
142       }
143     
144       /**
145        * Create a method call {@code methodName} with parameters {@code params}.
146        * If {@code receiver} is null, the call is static or to {@code this},
147        * otherwise the call is to {@code receiver}.
148        */
149       private static StringBuilder methodCall(ExpressionTree receiver, String methodName,
150           ExpressionTree... params) {
151         final StringBuilder fixedExpression = new StringBuilder();
152         if (receiver != null) {
153           boolean isBinop = receiver instanceof BinaryTree;
154           if (isBinop) {
155             fixedExpression.append("(");
156           }
157           fixedExpression.append(receiver);
158           if (isBinop) {
159             fixedExpression.append(")");
160           }
161           fixedExpression.append(".");
162         }
163         fixedExpression.append(methodName);
164         fixedExpression.append("(");
165         fixedExpression.append(Joiner.on(", ").join(params));
166         fixedExpression.append(")");
167     
168         return fixedExpression;
169       }
