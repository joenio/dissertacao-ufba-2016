17     package com.google.errorprone.refaster;
18     
19     import static com.google.errorprone.refaster.Unifier.unifyList;
20     
21     import com.google.auto.value.AutoValue;
22     import com.google.common.annotations.VisibleForTesting;
23     import com.google.common.collect.ImmutableList;
24     
25     import com.sun.source.tree.IntersectionTypeTree;
26     import com.sun.source.tree.TreeVisitor;
27     import com.sun.tools.javac.tree.JCTree.JCTypeIntersection;
28     
29     /**
30      * {@code UTree} representation of an {@code IntersectionTypeTree}.
31      * 
32      * @author lowasser@google.com (Louis Wasserman)
33      */
34     @AutoValue
35     abstract class UIntersectionType extends UExpression implements IntersectionTypeTree {
36       @VisibleForTesting static UIntersectionType create(UExpression... bounds) {
37         return create(ImmutableList.copyOf(bounds));
38       }
39       
40       static UIntersectionType create(Iterable<? extends UExpression> bounds) {
41         return new AutoValue_UIntersectionType(ImmutableList.copyOf(bounds));
42       }
43     
44       @Override
45       public abstract ImmutableList<UExpression> getBounds();
46     
47       @Override
48       public JCTypeIntersection inline(Inliner inliner) throws CouldNotResolveImportException {
49         return inliner.maker().TypeIntersection(inliner.inlineList(getBounds()));
50       }
51     
52       @Override
53       public Kind getKind() {
54         return Kind.INTERSECTION_TYPE;
55       }
56     
57       @Override
58       public <R, D> R accept(TreeVisitor<R, D> visitor, D data) {
59         return visitor.visitIntersectionType(this, data);
60       }
61     
62       @Override
63       public Choice<Unifier> visitIntersectionType(IntersectionTypeTree node, Unifier unifier) {
64         return unifyList(unifier, getBounds(), node.getBounds());
65       }
66     }
