41     /** @author cushon@google.com (Liam Miller-Cushon) */
42     @BugPattern(
43       name = "OptionalEquality",
44       summary = "Comparison using reference equality instead of value equality",
45       explanation =
46           "Optionals should be compared for value equality using `.equals()`, and not for reference "
47               + "equality using `==` and `!=`.",
48       category = GUAVA,
49       severity = ERROR,
50       maturity = MATURE
51     )
52     public class OptionalEquality extends BugChecker implements BinaryTreeMatcher {
53     
54       private final NullnessAnalysis nullnessAnalysis = new NullnessAnalysis();
55     
56       @Override
57       public final Description matchBinary(BinaryTree tree, VisitorState state) {
58         switch (tree.getKind()) {
59           case EQUAL_TO:
60           case NOT_EQUAL_TO:
61             break;
62           default:
63             return Description.NO_MATCH;
64         }
65         if (!isOptionalType(ASTHelpers.getType(tree.getLeftOperand()), state)) {
66           return Description.NO_MATCH;
67         }
68         if (!isOptionalType(ASTHelpers.getType(tree.getRightOperand()), state)) {
69           return Description.NO_MATCH;
70         }
71     
72         Description.Builder builder = buildDescription(tree);
73         addFixes(builder, tree, state);
74         return builder.build();
75       }
76     
77       void addFixes(Description.Builder builder, BinaryTree tree, VisitorState state) {
78         String prefix = tree.getKind() == Kind.NOT_EQUAL_TO ? "!" : "";
79         String lhs = state.getSourceForNode(tree.getLeftOperand());
80         String rhs = state.getSourceForNode(tree.getRightOperand());
81         Nullness nullness = getNullness(tree.getLeftOperand(), state);
82     
83         // If the lhs is possibly-null, provide both options.
84     
85         // Never swap the order (e.g. rhs.equals(lhs) when lhs is nullable and rhs is non-null), since
86         // that has observable side-effects.
87     
88         if (nullness != NONNULL) {
89           builder.addFix(
90               SuggestedFix.builder()
91                   .replace(tree, String.format("%sObjects.equals(%s, %s)", prefix, lhs, rhs))
92                   .addImport("java.util.Objects")
93                   .build());
94         }
95         if (nullness != NULL) {
96           builder.addFix(
97               SuggestedFix.replace(tree, String.format("%s%s.equals(%s)", prefix, lhs, rhs)));
98         }
99       }
100     
101       private Nullness getNullness(ExpressionTree expr, VisitorState state) {
102         TreePath pathToExpr = new TreePath(state.getPath(), expr);
103         return nullnessAnalysis.getNullness(pathToExpr, state.context);
104       }
105     
106       private static boolean isOptionalType(Type type, VisitorState state) {
107         Type guavaOptional = state.getTypeFromString(com.google.common.base.Optional.class.getName());
108         Type utilOptional = state.getTypeFromString("java.util.Optional");
109         // TODO(cushon): write a test for the util.Optional case once we can depend on 8.
110         return ASTHelpers.isSameType(type, guavaOptional, state)
111             || ASTHelpers.isSameType(type, utilOptional, state);
112       }
113     }
