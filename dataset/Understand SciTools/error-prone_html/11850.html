17     package com.google.errorprone;
18     
19     import static com.google.common.io.Files.readLines;
20     import static java.nio.charset.StandardCharsets.UTF_8;
21     
22     import com.beust.jcommander.IStringConverter;
23     import com.beust.jcommander.JCommander;
24     import com.beust.jcommander.Parameter;
25     import com.beust.jcommander.Parameters;
26     
27     import java.io.IOException;
28     import java.io.Writer;
29     import java.nio.charset.StandardCharsets;
30     import java.nio.file.Files;
31     import java.nio.file.Path;
32     import java.nio.file.Paths;
33     
34     /**
35      * Utility main which consumes the same tab-delimited text file and generates GitHub pages for
36      * the BugPatterns.
37      */
38     public class DocGenTool {
39     
40       @Parameters(separators = "=")
41       static class Options {
42         @Parameter(names = "-bug_patterns", description = "Path to bugPatterns.txt", required = true)
43         private String bugPatterns;
44     
45         @Parameter(
46           names = "-explanations",
47           description = "Path to side-car explanations",
48           required = true
49         )
50         private String explanations;
51     
52         @Parameter(names = "-docs_repository", description = "Path to docs repository", required = true)
53         private String docsRepository;
54     
55         @Parameter(names = "-examplesDir", description = "Path to examples directory", required = true)
56         private String examplesDir;
57     
58         @Parameter(
59           names = "-target",
60           description = "Whether to target the internal or external site",
61           converter = TargetEnumConverter.class,
62           required = true
63         )
64         private Target target;
65     
66         @Parameter(
67           names = "-use_pygments_highlighting",
68           description = "Use pygments for highlighting",
69           arity = 1
70         )
71         private boolean usePygments = true;
72       }
73     
74       enum Target {
75         INTERNAL,
76         EXTERNAL
77       }
78     
79       public static class TargetEnumConverter implements IStringConverter<Target> {
80         @Override
81         public Target convert(String arg) {
82           return Target.valueOf(arg.toUpperCase());
83         }
84       }
85     
86       public static void main(String[] args) throws IOException {
87         Options options = new Options();
88         new JCommander(options, args);
89     
90         Path bugPatterns = Paths.get(options.bugPatterns);
91         if (!Files.exists(bugPatterns)) {
92           usage("Cannot find bugPatterns file: " + options.bugPatterns);
93         }
94         Path explanationDir = Paths.get(options.explanations);
95         if (!Files.exists(explanationDir)) {
96           usage("Cannot find explanations dir: " + options.explanations);
97         }
98         Path wikiDir = Paths.get(options.docsRepository);
99         Files.createDirectories(wikiDir);
100         Path exampleDirBase = Paths.get(options.examplesDir);
101         if (!Files.exists(exampleDirBase)) {
102           usage("Cannot find example directory: " + options.examplesDir);
103         }
104         Path bugpatternDir = wikiDir.resolve("bugpattern");
105         if (!Files.exists(bugpatternDir)) {
106           Files.createDirectories(bugpatternDir);
107         }
108         Files.createDirectories(wikiDir.resolve("_data"));
109         BugPatternFileGenerator generator =
110             new BugPatternFileGenerator(
111                 bugpatternDir,
112                 exampleDirBase,
113                 explanationDir,
114                 options.target == Target.EXTERNAL,
115                 options.usePygments);
116         try (Writer w =
117                 Files.newBufferedWriter(wikiDir.resolve("bugpatterns.md"), StandardCharsets.UTF_8)) {
118           new BugPatternIndexWriter()
119               .dump(readLines(bugPatterns.toFile(), UTF_8, generator), w, options.target);
120         }
121       }
122     
123       private static void usage(String err) {
124         System.err.println(err);
125         System.exit(1);
126       }
127     }
