50     @BugPattern(name = "TypeParameterUnusedInFormals",
51         summary = "Declaring a type parameter that is only used in the return type is a misuse of"
52         + " generics: operations on the type parameter are unchecked, it hides unsafe casts at"
53         + " invocations of the method, and it interacts badly with method overload resolution."
54         + " NOTE: correcting this issue is often an incompatible API change; you should check that"
55         + " all dependent code still compiles succesfully.",
56         category = JDK, severity = WARNING, maturity = MATURE)
57     public class TypeParameterUnusedInFormals extends BugChecker implements MethodTreeMatcher {
58     
59       @Override
60       public Description matchMethod(MethodTree tree, VisitorState state) {
61         MethodSymbol methodSymbol = ASTHelpers.getSymbol(tree);
62         if (methodSymbol == null) {
63           return Description.NO_MATCH;
64         }
65     
66         // Only match methods where the return type is just a type parameter.
67         // e.g. the following is OK: <T> List<T> newArrayList();
68         TypeVar retType;
69         switch (methodSymbol.getReturnType().getKind()) {
70           case TYPEVAR:
71             retType = (TypeVar) methodSymbol.getReturnType();
72             break;
73           default:
74             return Description.NO_MATCH;
75         }
76     
77         // Ignore f-bounds.
78         // e.g.: <T extends Enum<T>> T unsafeEnumDeserializer();
79         if (retType.bound != null && TypeParameterFinder.visit(retType.bound).contains(retType)) {
80           return Description.NO_MATCH;
81         }
82     
83         // Ignore cases where the type parameter is used in the declaration of a formal parameter.
84         // e.g.: <T> T noop(T t);
85         for (VarSymbol formalParam : methodSymbol.getParameters()) {
86           if (TypeParameterFinder.visit(formalParam.type).contains(retType)) {
87             return Description.NO_MATCH;
88           }
89         }
90     
91         return attemptFix(retType, tree, state);
92       }
93     
94       private Description attemptFix(final TypeVar retType, MethodTree tree, VisitorState state) {
95         CharSequence source = state.getSourceForNode((JCTree) tree);
96         if (source == null) {
97           // No fix if we don't have end positions.
98           return describeMatch(tree);
99         }
100     
101         int paramIndex = -1;
102         for (int idx = 0; idx < tree.getTypeParameters().size(); ++idx) {
103           if (((JCTypeParameter) tree.getTypeParameters().get(idx)).type.equals(retType)) {
104             paramIndex = idx;
105             break;
106           }
107         }
108     
109         if (paramIndex == -1) {
110           return Description.NO_MATCH;
111         }
112     
113         SuggestedFix.Builder fix = SuggestedFix.builder();
114         fix = removeTypeParam(paramIndex, tree.getTypeParameters(), state, fix);
115     
116         // Replace usages of the type parameter with its upper-bound, both inside the method body
117         // and in the return type.
118         // e.g. <A, B> B f(A a) { return (B) a; } -> <A> Object f(A a) { return (Object) a; }
119         String qualifiedName = retType.bound != null ? retType.bound.toString() : "Object";
120         // Always use simple names.
121         // TODO(cushon) - this isn't always correct, but it's better than defaulting to
122         // fully-qualified names. There should be a better way to do this.
123         String newType = Iterables.getLast(Splitter.on('.').split(qualifiedName));
124     
125         rewriteTypeUsages(retType, tree.getBody(), newType, fix, state);
126         rewriteTypeUsages(retType, tree.getReturnType(), newType, fix, state);
127     
128         return describeMatch(tree, fix.build());
129       }
130     
131       // Remove the type parameter declaration.
132       // e.g. <A, B> B f(A a); -> <A> B f(A a);
133       private SuggestedFix.Builder removeTypeParam(
134           int paramIndex,
135           List<? extends TypeParameterTree> tyParams,
136           VisitorState state,
137           SuggestedFix.Builder fix) {
138     
139         JCTree typeParam = (JCTree) tyParams.get(paramIndex);
140         boolean isFirst = paramIndex == 0;
141         boolean isLast = paramIndex == (tyParams.size() - 1);
142     
143         int startDeletion = isFirst
144             ? typeParam.getStartPosition()
145             : state.getEndPosition((JCTree) tyParams.get(paramIndex - 1));
146     
147         int endDeletion = isLast
148             ? state.getEndPosition(typeParam)
149             : ((JCTree) tyParams.get(paramIndex + 1)).getStartPosition();
150     
151         // if it's an interior deletion, leave a comma to separate the neighbouring params
152         String replacement = (isFirst || isLast) ? "" : ", ";
153     
154         if (isFirst && isLast) {
155           // Remove leading "<" and trailing "> "
156           startDeletion -= 1;
157           endDeletion += 2;
158         }
159     
160         fix = fix.replace(startDeletion, endDeletion, replacement);
161         return fix;
162       }
163     
164       private static void rewriteTypeUsages(
165           final TypeVar retType,
166           Tree tree,
167           final String newType,
168           final SuggestedFix.Builder fix,
169           final VisitorState state) {
170         if (tree == null) {
171           // e.g. abstract methods without bodies.
172           return;
173         }
174         ((JCTree) tree)
175             .accept(
176                 new TreeScanner() {
177                   @Override
178                   public void visitIdent(JCIdent node) {
179                     if (retType.tsym.equals(node.sym)) {
180                       fix.replace(node, newType);
181                     }
182                   }
183     
184                   @Override
185                   public void visitTypeCast(JCTypeCast node) {
186                     if (ASTHelpers.isSameType(retType.tsym.type, node.type.tsym.type, state)
187                         && newType.equals("Object")) {
188                       fix.replace(node, state.getSourceForNode(node.getExpression()).toString());
189                     } else {
190                       super.visitTypeCast(node);
191                     }
192                   }
193                 });
194       }

196       /**
197        * A visitor that records the set of {@link com.sun.tools.javac.code.Type.TypeVar}s referenced by
198        * the current type.
199        */
200       private static class TypeParameterFinder extends Types.DefaultTypeVisitor<Void, Void> {
201     
202         static Set<Type.TypeVar> visit(Type type) {
203           TypeParameterFinder visitor = new TypeParameterFinder();
204           type.accept(visitor, null);
205           return visitor.seen;
206         }
207     
208         private final Set<Type.TypeVar> seen = new HashSet<Type.TypeVar>();
209     
210         @Override
211         public Void visitClassType(Type.ClassType type, Void unused) {
212           if (type instanceof Type.IntersectionClassType) {
213             // TypeVisitor doesn't support intersection types natively, so we have
214             // to handle visiting them manually.
215             visitIntersectionClassType((Type.IntersectionClassType) type);
216           } else {
217             for (Type t : type.getTypeArguments()) {
218               t.accept(this, null);
219             }
220           }
221           return null;
222         }
223     
224         public void visitIntersectionClassType(Type.IntersectionClassType type) {
225           for (Type component : type.getComponents()) {
226             component.accept(this, null);
227           }
228         }
229     
230         @Override
231         public Void visitWildcardType(Type.WildcardType type, Void unused) {
232           if (type.getSuperBound() != null) {
233             type.getSuperBound().accept(this, null);
234           }
235           if (type.getExtendsBound() != null) {
236             type.getExtendsBound().accept(this, null);
237           }
238           return null;
239         }
240     
241         @Override
242         public Void visitArrayType(Type.ArrayType type, Void unused) {
243           type.elemtype.accept(this, null);
244           return null;
245         }
246     
247         @Override
248         public Void visitTypeVar(Type.TypeVar type, Void unused) {
249           // only visit f-bounds once:
250           if (!seen.add(type)) {
251             return null;
252           }
253           if (type.bound != null) {
254             type.bound.accept(this, null);
255           }
256           return null;
257         }
258     
259         @Override
260         public Void visitType(Type type, Void unused) {
261           return null;
262         }
263       }
