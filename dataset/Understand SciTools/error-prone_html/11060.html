42     /**
43      * @author eaftan@google.com (Eddie Aftandilian)
44      */
45     @BugPattern(name = "PrimitiveArrayPassedToVarargsMethod",
46         summary = "Passing a primitive array to a varargs method is usually wrong",
47         explanation = "When you pass a primitive array as the only argument to a varargs method, the "
48             + "primitive array is autoboxed into a single-element Object array. This is usually "
49             + "not what was intended.",
50         category = JDK, severity = WARNING, maturity = EXPERIMENTAL)
51     public class PrimitiveArrayPassedToVarargsMethod extends BugChecker
52         implements MethodInvocationTreeMatcher {
53     
54       /**
55        * Assuming the argument in the varargs position is a single one of type int[], here is the
56        * truth table:
57        *     Param type    Should return     Why
58        *     int...        false             Exact type match
59        *     int[]...      false             Exact type match for the array element type
60        *     T...          true              Will cause boxing
61        *     Object...     true              Will cause boxing
62        */
63       private static final Matcher<MethodInvocationTree> isVarargs =
64           new Matcher<MethodInvocationTree>() {
65         @Override
66         public boolean matches(MethodInvocationTree t, VisitorState state) {
67           Symbol symbol = ASTHelpers.getSymbol(t);
68           if (!(symbol instanceof MethodSymbol)) {
69             return false;
70           }
71           MethodSymbol methodSymbol = (MethodSymbol) symbol;
72     
73           // Bail out quickly if the method is not varargs
74           if (!methodSymbol.isVarArgs()) {
75             return false;
76           }
77     
78           // Last param must be varags
79           List<VarSymbol> params = methodSymbol.getParameters();
80           int varargsPosition = params.length() - 1;
81           ArrayType varargsParamType = (ArrayType) params.last().type;
82     
83           // Is the argument at the varargsPosition the only varargs argument and a primitive array?
84           JCMethodInvocation methodInvocation = (JCMethodInvocation) t;
85           List<JCExpression> arguments = methodInvocation.getArguments();
86           Types types = state.getTypes();
87           if (arguments.size() != params.length()) {
88             return false;
89           }
90           Type varargsArgumentType = arguments.get(varargsPosition).type;
91           if (!types.isArray(varargsArgumentType)
92               || !types.elemtype(varargsArgumentType).isPrimitive()) {
93             return false;
94           }
95     
96           // Do the param and argument types actually match? i.e. can boxing even happen?
97           return !(types.isSameType(varargsParamType, varargsArgumentType)
98               || types.isSameType(varargsParamType.getComponentType(), varargsArgumentType));
99     
100         }
101       };
102     
103       @Override
104       public Description matchMethodInvocation(MethodInvocationTree t, VisitorState state) {
105         return isVarargs.matches(t, state) ? describeMatch(t) : NO_MATCH;
106       }
107     }
