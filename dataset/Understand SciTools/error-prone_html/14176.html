17     package com.google.errorprone;
18     
19     import com.google.auto.value.AutoValue;
20     import com.google.common.collect.ImmutableClassToInstanceMap;
21     import com.google.common.collect.ImmutableList;
22     
23     import com.sun.source.util.TreePath;
24     import com.sun.tools.javac.util.Context;
25     
26     import java.io.Serializable;
27     import java.lang.annotation.Annotation;
28     
29     /**
30      * Combines multiple {@code CodeTransformer}s into one.
31      */
32     @AutoValue
33     public abstract class CompositeCodeTransformer implements CodeTransformer, Serializable {
34       public static CodeTransformer compose(CodeTransformer... transformers) {
35         return compose(ImmutableList.copyOf(transformers));
36       }
37     
38       public static CodeTransformer compose(Iterable<? extends CodeTransformer> transformers) {
39         return new AutoValue_CompositeCodeTransformer(ImmutableList.copyOf(transformers));
40       }
41     
42       CompositeCodeTransformer() {}
43     
44       public abstract ImmutableList<CodeTransformer> transformers();
45     
46       @Override
47       public void apply(TreePath path, Context context, DescriptionListener listener) {
48         for (CodeTransformer transformer : transformers()) {
49           transformer.apply(path, context, listener);
50         }
51       }
52     
53       @Override
54       public ImmutableClassToInstanceMap<Annotation> annotations() {
55         return ImmutableClassToInstanceMap.<Annotation>builder().build();
56       }
57     }
