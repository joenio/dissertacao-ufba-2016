17     package com.google.errorprone.bugpatterns.collectionincompatibletype;
18     
19     import static com.google.errorprone.matchers.method.MethodMatchers.instanceMethod;
20     
21     import com.google.common.collect.Iterables;
22     import com.google.errorprone.VisitorState;
23     import com.google.errorprone.matchers.Matcher;
24     import com.google.errorprone.util.ASTHelpers;
25     
26     import com.sun.source.tree.ExpressionTree;
27     import com.sun.source.tree.MethodInvocationTree;
28     import com.sun.tools.javac.code.Type;
29     
30     import java.util.Collection;
31     
32     /**
33      * Matches an instance method like {@link Collection#removeAll}, for which we need to extract
34      * the type argument to the method argument.
35      */
36     class TypeArgOfMethodArgMatcher extends AbstractCollectionIncompatibleTypeMatcher {
37     
38       private final Matcher<ExpressionTree> methodMatcher;
39       private final String receiverTypeName;
40       private final int receiverTypeArgIndex;
41       private final int methodArgIndex;
42       private final String methodArgTypeName;
43       private final int methodArgTypeArgIndex;
44     
45       /**
46        * @param receiverTypeName The fully-qualified name of the type of the method receiver whose
47        *        descendants to match on
48        * @param signature The signature of the method to match on
49        * @param receiverTypeArgIndex The index of the type argument that should match the method
50        *        argument
51        * @param methodArgIndex The index of the method argument whose type argument we should extract
52        * @param methodArgTypeName The fully-qualified name of the type of the method argument whose
53        *        type argument we should extract
54        * @param methodArgTypeArgIndex The index of the type argument to extract from the method
55        *        argument
56        */
57       public TypeArgOfMethodArgMatcher(
58           String receiverTypeName,
59           String signature,
60           int receiverTypeArgIndex,
61           int methodArgIndex,
62           String methodArgTypeName,
63           int methodArgTypeArgIndex) {
64         this.methodMatcher = instanceMethod().onDescendantOf(receiverTypeName).withSignature(signature);
65         this.receiverTypeName = receiverTypeName;
66         this.receiverTypeArgIndex = receiverTypeArgIndex;
67         this.methodArgIndex = methodArgIndex;
68         this.methodArgTypeName = methodArgTypeName;
69         this.methodArgTypeArgIndex = methodArgTypeArgIndex;
70       }
71     
72       @Override
73       Matcher<ExpressionTree> methodMatcher() {
74         return methodMatcher;
75       }
76     
77       @Override
78       ExpressionTree extractSourceTree(MethodInvocationTree tree, VisitorState state) {
79         return Iterables.get(tree.getArguments(), methodArgIndex);
80       }
81     
82       @Override
83       Type extractSourceType(MethodInvocationTree tree, VisitorState state) {
84         return extractTypeArgAsMemberOfSupertype(
85             ASTHelpers.getType(Iterables.get(tree.getArguments(), methodArgIndex)),
86             state.getSymbolFromString(methodArgTypeName),
87             methodArgTypeArgIndex,
88             state.getTypes());
89       }
90     
91       @Override
92       Type extractTargetType(MethodInvocationTree tree, VisitorState state) {
93         return extractTypeArgAsMemberOfSupertype(
94             ASTHelpers.getReceiverType(tree),
95             state.getSymbolFromString(receiverTypeName),
96             receiverTypeArgIndex,
97             state.getTypes());
98       }
99     
100       String getMethodArgTypeName() {
101         return methodArgTypeName;
102       }
103     }
