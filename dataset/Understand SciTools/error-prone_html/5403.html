41     /**
42      * @author cushon@google.com (Liam Miller-Cushon)
43      */
44     @BugPattern(name = "NarrowingCompoundAssignment",
45         summary = "Compound assignments to bytes, shorts, chars, and floats hide dangerous casts",
46         explanation = "The compound assignment E1 op= E2 could be mistaken for being equivalent to "
47             + " E1 = E1 op E2. However, this is not the case: compound "
48             + " assignment operators automatically cast the result of the computation"
49             + " to the type on the left hand side. So E1 op= E2 is actually equivalent to"
50             + " E1 = (T) (E1 op E2), where T is the type of E1. If the type of the expression is"
51             + " wider than the type of the variable (i.e. the variable is a byte, char, short, or"
52             + " float), then the compound assignment will perform a narrowing"
53             + " primitive conversion. Attempting to perform the equivalent simple assignment"
54             + " would generate a compilation error.\n\n"
55             + " For example, 'byte b = 0; b = b << 1;' does not compile, but 'byte b = 0; b <<= 1;'"
56             + " does!\n\n"
57             + " (See Puzzle #9 in 'Java Puzzlers: Traps, Pitfalls, and Corner Cases'.)",
58         category = JDK, severity = ERROR, maturity = MATURE)
59     public class NarrowingCompoundAssignment extends BugChecker
60         implements CompoundAssignmentTreeMatcher {
61     
62       // The set of types susceptible to the implicit narrowing bug, and their string names.
63       private static final EnumMap<TypeKind, String> DEFICIENT_TYPES = new EnumMap<>(TypeKind.class);
64       static {
65         DEFICIENT_TYPES.put(TypeKind.BYTE, "byte");
66         DEFICIENT_TYPES.put(TypeKind.SHORT, "short");
67         DEFICIENT_TYPES.put(TypeKind.CHAR, "char");
68         DEFICIENT_TYPES.put(TypeKind.FLOAT, "float");
69       }
70     
71       static String assignmentToString(Tree.Kind kind) {
72         switch (kind) {
73           case MULTIPLY:
74             return "*";
75           case DIVIDE:
76             return "/";
77           case REMAINDER:
78             return "%";
79           case PLUS:
80             return "+";
81           case MINUS:
82             return "-";
83           case LEFT_SHIFT:
84             return "<<";
85           case AND:
86             return "&";
87           case XOR:
88             return "^";
89           case OR:
90             return "|";
91           case RIGHT_SHIFT:
92             return ">>";
93           case UNSIGNED_RIGHT_SHIFT:
94             return ">>>";
95           default:
96             throw new IllegalArgumentException("Unexpected operator assignment kind: " + kind);
97         }
98       }
99       
100       static Kind regularAssignmentFromCompound(Kind kind) {
101         switch (kind) {
102           case MULTIPLY_ASSIGNMENT:
103             return Kind.MULTIPLY;
104           case DIVIDE_ASSIGNMENT:
105             return Kind.DIVIDE;
106           case REMAINDER_ASSIGNMENT:
107             return Kind.REMAINDER;
108           case PLUS_ASSIGNMENT:
109             return Kind.PLUS;
110           case MINUS_ASSIGNMENT:
111             return Kind.MINUS;
112           case LEFT_SHIFT_ASSIGNMENT:
113             return Kind.LEFT_SHIFT;
114           case AND_ASSIGNMENT:
115             return Kind.AND;
116           case XOR_ASSIGNMENT:
117             return Kind.XOR;
118           case OR_ASSIGNMENT:
119             return Kind.OR;
120           case RIGHT_SHIFT_ASSIGNMENT:
121             return Kind.RIGHT_SHIFT;
122           case UNSIGNED_RIGHT_SHIFT_ASSIGNMENT:
123             return Kind.UNSIGNED_RIGHT_SHIFT;
124           default:
125             throw new IllegalArgumentException("Unexpected compound assignment kind: " + kind);
126         }
127       }
128       
129       @Override
130       public Description matchCompoundAssignment(CompoundAssignmentTree tree, VisitorState state) {
131         String deficient = DEFICIENT_TYPES.get(type(tree.getVariable()).getKind());
132         if (deficient == null) {
133           return Description.NO_MATCH;
134         }
135         if (state.getTypes().isConvertible(type(tree.getExpression()), type(tree.getVariable()))) {
136           return Description.NO_MATCH;
137         }
138     
139         CharSequence var = state.getSourceForNode((JCTree) tree.getVariable());
140         CharSequence expr = state.getSourceForNode((JCTree) tree.getExpression());
141         if (var == null || expr == null) {
142           return Description.NO_MATCH;
143         }
144         switch (tree.getKind()) {
145           case RIGHT_SHIFT_ASSIGNMENT:
146           case UNSIGNED_RIGHT_SHIFT_ASSIGNMENT:
147             // Right shifts cannot cause overflow
148             return Description.NO_MATCH;
149           default:  // continue below
150         }
151         Kind regularAssignmentKind = regularAssignmentFromCompound(tree.getKind());
152         String op = assignmentToString(regularAssignmentKind);
153     
154         // Add parens to the rhs if necessary to preserve the current precedence
155         // e.g. 's -= 1 - 2' -> 's = s - (1 - 2)'
156         if (tree.getExpression() instanceof JCBinary) {
157           Kind rhsKind = ((JCBinary) tree.getExpression()).getKind();
158           if (OperatorPrecedence.from(rhsKind) == OperatorPrecedence.from(regularAssignmentKind)) {
159             expr = String.format("(%s)", expr); 
160           }
161         }
162     
163         // e.g. 's *= 42' -> 's = (short) (s * 42)'
164         String replacement = String.format("%s = (%s) (%s %s %s)", var, deficient, var, op, expr);
165         return describeMatch(tree, SuggestedFix.replace(tree, replacement));
166       }
167     
168       static Type type(Tree tree) {
169         return ((JCTree) tree).type;
170       }
171     }
