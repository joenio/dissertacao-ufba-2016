17     package com.google.errorprone.refaster;
18     
19     import static com.google.common.base.Preconditions.checkNotNull;
20     
21     import com.google.common.annotations.VisibleForTesting;
22     import com.google.common.base.Function;
23     import com.google.common.base.Functions;
24     import com.google.common.base.Optional;
25     import com.google.common.base.Predicate;
26     import com.google.common.collect.Iterables;
27     import com.google.common.collect.Iterators;
28     import com.google.errorprone.annotations.ForOverride;
29     
30     import java.util.Collection;
31     import java.util.Collections;
32     import java.util.Iterator;
33     
34     /**
35      * A representation of a choice with zero or more options, which may be evaluated lazily or
36      * strictly.
37      * 
38      * <p>This resembles the list monad in Haskell.
39      *
40      * @author lowasser@google.com (Louis Wasserman)
41      */
42     public abstract class Choice<T> {
43       /*
44        * This is currently implemented in terms of Iterators, but this may change in future, e.g. to a
45        * Stream-based implementation.
46        *
47        * Special-casing operations for Choice.none and Choice.single, which are by far the most common
48        * cases, avoids overly nested results.
49        */
50     
51       private static final Choice<Object> NONE = new Choice<Object>() {
52     
53         @Override
54         protected Iterator<Object> iterator() {
55           return Collections.emptyIterator();
56         }
57     
58         @Override
59         public <R> Choice<R> thenChoose(Function<? super Object, Choice<R>> function) {
60           checkNotNull(function);
61           return none();
62         }
63     
64         @Override
65         public <R> Choice<R> thenOption(Function<? super Object, Optional<R>> function) {
66           checkNotNull(function);
67           return none();
68         }
69     
70         @Override
71         public <R> Choice<R> transform(Function<? super Object, R> function) {
72           checkNotNull(function);
73           return none();
74         }
75     
76         @Override
77         public Choice<Object> or(Choice<Object> other) {
78           return checkNotNull(other);
79         }
80     
81         @Override
82         public Choice<Object> condition(Predicate<? super Object> predicate) {
83           checkNotNull(predicate);
84           return this;
85         }
86       };
87     
88       /**
89        * The empty {@code Choice}.
90        */
91       @SuppressWarnings("unchecked")
92       public static <T> Choice<T> none() {
93         return (Choice) NONE;
94       }
95     
96       /**
97        * Returns a {@code Choice} with only one option, {@code t}.
98        */
99       public static <T> Choice<T> of(final T t) {
100         checkNotNull(t);
101         return new Choice<T>() {
102           @Override
103           protected Iterator<T> iterator() {
104             return Iterators.singletonIterator(t);
105           }
106     
107           @Override
108           public Optional<T> first() {
109             return Optional.of(t);
110           }
111     
112           @Override
113           public Choice<T> condition(Predicate<? super T> predicate) {
114             return predicate.apply(t) ? this : Choice.<T>none();
115           }
116     
117           @Override
118           public <R> Choice<R> thenChoose(Function<? super T, Choice<R>> function) {
119             return function.apply(t);
120           }
121     
122           @Override
123           public <R> Choice<R> thenOption(Function<? super T, Optional<R>> function) {
124             return fromOptional(function.apply(t));
125           }
126     
127           @Override
128           public <R> Choice<R> transform(Function<? super T, R> function) {
129             return of(function.apply(t));
130           }
131         };
132       }
133       
134       /**
135        * Returns a {@code Choice} with {@code t} as an option if {@code condition},
136        * and no options otherwise.
137        */
138       public static <T> Choice<T> condition(boolean condition, T t) {
139         return condition ? of(t) : Choice.<T>none();
140       }
141     
142       /**
143        * Returns a choice of the optional value, if it is present, or the empty choice if it is absent.
144        */
145       public static <T> Choice<T> fromOptional(Optional<T> optional) {
146         return optional.isPresent() ? of(optional.get()) : Choice.<T>none();
147       }
148       
149       public static <T> Choice<T> from(final Collection<T> choices) {
150         switch (choices.size()) {
151           case 0:
152             return none();
153           case 1:
154             return of(Iterables.getOnlyElement(choices));
155           default:
156             return new Choice<T>() {
157               @Override
158               protected Iterator<T> iterator() {
159                 return choices.iterator();
160               }      
161             };
162         }
163       }
164     
165       /**
166        * Returns a choice between any of the options from any of the specified choices.
167        */
168       public static <T> Choice<T> any(final Collection<Choice<T>> choices) {
169         return from(choices).thenChoose(Functions.<Choice<T>>identity());
170       }
171       
172       private Choice() {}
173       
174       @VisibleForTesting
175       Iterable<T> asIterable() {
176         return new Iterable<T>() {
177           @Override
178           public Iterator<T> iterator() {
179             return Choice.this.iterator();
180           }
181         };
182       }
183     
184       // Currently, this is implemented with an Iterator, but that may change in future!
185       @ForOverride
186       protected abstract Iterator<T> iterator();
187     
188       /**
189        * Returns the first valid option from this {@code Choice}.
190        */
191       public Optional<T> first() {
192         Iterator<T> itr = iterator();
193         return itr.hasNext() ? Optional.of(itr.next()) : Optional.<T>absent();
194       }
195     
196       /**
197        * Returns all the choices obtained by choosing from this {@code Choice} and then choosing from
198        * the {@code Choice} yielded by this function on the result.
199        *
200        * <p>The function may be applied lazily or immediately, at the discretion of the implementation.
201        *
202        * <p>This is the monadic bind for {@code Choice}.
203        */
204       public <R> Choice<R> thenChoose(final Function<? super T, Choice<R>> function) {
205         checkNotNull(function);
206         if (Thread.interrupted()) {
207           throw new RuntimeException(new InterruptedException());
208         }
209         final Choice<T> thisChoice = this;
210         return new Choice<R>() {
211           @Override
212           protected Iterator<R> iterator() {
213             if (Thread.interrupted()) {
214               throw new RuntimeException(new InterruptedException());
215             }
216             return Iterators.concat(
217                 Iterators.transform(thisChoice.iterator(), new Function<T, Iterator<R>>() {
218                   @Override
219                   public Iterator<R> apply(T t) {
220                     return function.apply(t).iterator();
221                   }
222                 }));
223           }
224         };
225       }
226     
227       /**
228        * Returns all the choices obtained by choosing from this {@code Choice} and yielding a present
229        * {@code Optional}.
230        *
231        * <p>The function may be applied lazily or immediately, at the discretion of the implementation.
232        */
233       public <R> Choice<R> thenOption(final Function<? super T, Optional<R>> function) {
234         checkNotNull(function);
235         final Choice<T> thisChoice = this;
236         return new Choice<R>() {
237           @Override
238           protected Iterator<R> iterator() {
239             return Optional.presentInstances(Iterables.transform(thisChoice.asIterable(), function))
240                 .iterator();
241           }
242         };
243       }
244     
245       /**
246        * Maps the choices with the specified function.
247        */
248       public <R> Choice<R> transform(final Function<? super T, R> function) {
249         checkNotNull(function);
250         final Choice<T> thisChoice = this;
251         return new Choice<R>() {
252           @Override
253           protected Iterator<R> iterator() {
254             return Iterators.transform(thisChoice.iterator(), function);
255           }
256         };
257       }
258     
259       /**
260        * Returns a choice of the options from this {@code Choice} or from {@code other}.
261        */
262       public Choice<T> or(final Choice<T> other) {
263         checkNotNull(other);
264         if (other == none()) {
265           return this;
266         } else {
267           final Choice<T> thisChoice = this;
268           return new Choice<T>() {
269             @Override
270             protected Iterator<T> iterator() {
271               return Iterators.concat(thisChoice.iterator(), other.iterator());
272             }
273           };
274         }
275       }
276     
277       /**
278        * Returns this choice if {@code condition}, otherwise the empty choice.
279        */
280       public Choice<T> condition(boolean condition) {
281         return condition ? this : Choice.<T>none();
282       }
283     
284       /**
285        * Filters the choices to those that satisfy the provided {@code Predicate}.
286        */
287       public Choice<T> condition(final Predicate<? super T> predicate) {
288         checkNotNull(predicate);
289         final Choice<T> thisChoice = this;
290         return new Choice<T>() {
291           @Override
292           protected Iterator<T> iterator() {
293             return Iterators.filter(thisChoice.iterator(), predicate);
294           }
295         };
296       }
297     }
