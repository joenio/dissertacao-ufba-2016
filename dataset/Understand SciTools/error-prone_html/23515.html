56     /**
57      * Scanner implementation to extract a single Refaster rule from a {@code ClassTree}.
58      * 
59      * @author lowasser@google.com (Louis Wasserman)
60      */
61     public final class RefasterRuleBuilderScanner extends SimpleTreeVisitor<Void, Void> {
62       private static final Logger logger =
63           Logger.getLogger(RefasterRuleBuilderScanner.class.toString());
64       
65       static final Context.Key<Map<MethodSymbol, PlaceholderMethod>> PLACEHOLDER_METHODS_KEY
66           = new Context.Key<>();
67       
68       private final Context context;
69       private final Map<MethodSymbol, PlaceholderMethod> placeholderMethods;
70       private final List<Template<?>> beforeTemplates;
71       private final List<Template<?>> afterTemplates;
72     
73       private RefasterRuleBuilderScanner(Context context) {
74         this.context = new SubContext(context);
75         if (context.get(PLACEHOLDER_METHODS_KEY) == null) {
76           this.placeholderMethods = new HashMap<>();
77           context.put(PLACEHOLDER_METHODS_KEY, placeholderMethods);
78         } else {
79           this.placeholderMethods = context.get(PLACEHOLDER_METHODS_KEY);
80         }
81         
82         this.beforeTemplates = new ArrayList<>();
83         this.afterTemplates = new ArrayList<>();
84       }
85       
86       public static Collection<? extends CodeTransformer> extractRules(ClassTree tree,
87           Context context) {
88         ClassSymbol sym = ASTHelpers.getSymbol(tree);
89         RefasterRuleBuilderScanner scanner = new RefasterRuleBuilderScanner(context);
90         // visit abstract methods first
91         List<MethodTree> methods = new Ordering<MethodTree>() {
92           @Override public int compare(MethodTree l, MethodTree r) {
93             return Boolean.compare(
94                 l.getModifiers().getFlags().contains(Modifier.ABSTRACT),
95                 r.getModifiers().getFlags().contains(Modifier.ABSTRACT));
96           }
97         }.reverse().immutableSortedCopy(Iterables.filter(tree.getMembers(), MethodTree.class));
98         scanner.visit(methods, null);
99         
100         UTemplater templater = new UTemplater(context);
101         List<UType> types = templater.templateTypes(sym.type.getTypeArguments());
102         
103         return scanner.createMatchers(
104             Iterables.filter(types, UTypeVar.class),
105             sym.getQualifiedName().toString(),
106             UTemplater.annotationMap(sym));
107       }
108     
109       @Override
110       public Void visitMethod(MethodTree tree, Void v) {
111         try {
112           VisitorState state = new VisitorState(context);
113           logger.log(FINE, "Discovered method with name {0}", tree.getName());
114           if (ASTHelpers.hasAnnotation(tree, Placeholder.class, state)) {
115             checkArgument(tree.getModifiers().getFlags().contains(Modifier.ABSTRACT),
116                 "@Placeholder methods are expected to be abstract");
117             UTemplater templater = new UTemplater(context);
118             ImmutableMap.Builder<UVariableDecl,
119                 ImmutableClassToInstanceMap<Annotation>> params = ImmutableMap.builder();
120             for (VariableTree param : tree.getParameters()) {
121               params.put(templater.visitVariable(param, null), 
122                   UTemplater.annotationMap(ASTHelpers.getSymbol(param)));
123             }
124             MethodSymbol sym = ASTHelpers.getSymbol(tree);
125             placeholderMethods.put(sym, PlaceholderMethod.create(tree.getName(),
126                 templater.template(sym.getReturnType()), params.build(),
127                 UTemplater.annotationMap(sym)));
128           } else if (ASTHelpers.hasAnnotation(tree, BeforeTemplate.class, state)) {
129             checkState(afterTemplates.isEmpty(), "BeforeTemplate must come before AfterTemplate");
130             Template<?> template = UTemplater.createTemplate(context, tree);
131             beforeTemplates.add(template);
132             if (template instanceof BlockTemplate) {
133               context.put(UTemplater.REQUIRE_BLOCK_KEY, true);
134             }
135           } else if (ASTHelpers.hasAnnotation(tree, AfterTemplate.class, state)) {
136             afterTemplates.add(UTemplater.createTemplate(context, tree));
137           } else if (tree.getModifiers().getFlags().contains(Modifier.ABSTRACT)) {
138             throw new IllegalArgumentException(
139                 "Placeholder methods must have @Placeholder, but abstract method does not: " + tree);
140           }
141           return null;
142         } catch (Throwable t) {
143           throw new RuntimeException("Error analysing: " + tree.getName(), t);
144         }
145       }
146     
147       private Collection<? extends CodeTransformer> createMatchers(
148           Iterable<UTypeVar> typeVars,
149           String qualifiedTemplateClass, ImmutableClassToInstanceMap<Annotation> annotationMap) {
150         if (beforeTemplates.isEmpty() && afterTemplates.isEmpty()) {
151           // there's no template here
152           return ImmutableList.of();
153         } else {
154           if (annotationMap.containsKey(AllowCodeBetweenLines.class)) {
155             List<UBlank> blanks = new ArrayList<>();
156             for (int i = 0; i < beforeTemplates.size(); i++) {
157               BlockTemplate before = (BlockTemplate) beforeTemplates.get(i);
158               List<UStatement> stmtsWithBlanks = new ArrayList<>();
159               for (UStatement stmt : before.templateStatements()) {
160                 if (!stmtsWithBlanks.isEmpty()) {
161                   UBlank blank = UBlank.create();
162                   blanks.add(blank);
163                   stmtsWithBlanks.add(blank);
164                 }
165                 stmtsWithBlanks.add(stmt);
166               }
167               beforeTemplates.set(i, before.withStatements(stmtsWithBlanks));
168             }
169             for (int i = 0; i < afterTemplates.size(); i++) {
170               BlockTemplate afterBlock = (BlockTemplate) afterTemplates.get(i);
171               afterTemplates.set(i, afterBlock.withStatements(
172                   Iterables.concat(blanks, afterBlock.templateStatements())));
173             }
174           }
175           RefasterRule<?, ?> rule = RefasterRule.create(qualifiedTemplateClass, typeVars,
176               beforeTemplates, afterTemplates, annotationMap);
177     
178           List<ExpressionTemplate> negatedAfterTemplates = new ArrayList<>();
179           for (Template<?> afterTemplate : afterTemplates) {
180             if (afterTemplate.annotations().containsKey(AlsoNegation.class)) {
181               negatedAfterTemplates.add(((ExpressionTemplate) afterTemplate).negation());
182             }
183           }
184           if (!negatedAfterTemplates.isEmpty()) {
185             List<ExpressionTemplate> negatedBeforeTemplates = new ArrayList<>();
186             for (Template<?> beforeTemplate : beforeTemplates) {
187               negatedBeforeTemplates.add(((ExpressionTemplate) beforeTemplate).negation());
188             }
189             RefasterRule<?, ?> negation = RefasterRule.create(
190                 qualifiedTemplateClass, typeVars, negatedBeforeTemplates, 
191                 negatedAfterTemplates, annotationMap);
192             return ImmutableList.of(rule, negation);
193           }
194           return ImmutableList.of(rule);
195         }
196       }
