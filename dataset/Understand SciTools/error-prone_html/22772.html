45       static class Builder {
46         final ImmutableMap.Builder<String, ImmutableAnnotationInfo> mapBuilder = ImmutableMap.builder();
47     
48         public Builder addAll(Set<Class<?>> clazzs) {
49           for (Class<?> clazz : clazzs) {
50             add(clazz);
51           }
52           return this;
53         }
54     
55         public Builder add(Class<?> clazz, String... containerOf) {
56           ImmutableSet<String> containerTyParams = ImmutableSet.copyOf(containerOf);
57           HashSet<String> actualTyParams = new HashSet<>();
58           for (TypeVariable<?> x : clazz.getTypeParameters()) {
59             actualTyParams.add(x.getName());
60           }
61           SetView<String> difference = Sets.difference(containerTyParams, actualTyParams);
62           if (!difference.isEmpty()) {
63             throw new AssertionError(
64                 String.format(
65                     "Container type parameter(s) %s did not exist, expected %s",
66                     difference,
67                     actualTyParams));
68           }
69           mapBuilder.put(
70               clazz.getName(),
71               ImmutableAnnotationInfo.create(clazz.getName(), ImmutableList.copyOf(containerOf)));
72           return this;
73         }
74     
75         public Builder add(String className, String... containerOf) {
76           mapBuilder.put(
77               className, ImmutableAnnotationInfo.create(className, ImmutableList.copyOf(containerOf)));
78           return this;
79         }
80     
81         public ImmutableMap<String, ImmutableAnnotationInfo> build() {
82           return mapBuilder.build();
83         }
84       }

133       /** Types that are known to be mutable. */
134       static final ImmutableSet<String> KNOWN_UNSAFE = getKnownUnsafeClasses();
135     
136       private static ImmutableSet<String> getKnownUnsafeClasses() {
137         ImmutableSet.Builder<String> result = ImmutableSet.<String>builder();
138         for (Class<?> clazz :
139             ImmutableSet.<Class<?>>of(
140                 java.lang.Iterable.class,
141                 java.lang.Object.class,
142                 java.util.ArrayList.class,
143                 java.util.Collection.class,
144                 java.util.List.class,
145                 java.util.Map.class,
146                 java.util.Set.class,
147                 java.util.EnumSet.class,
148                 java.util.EnumMap.class)) {
149           result.add(clazz.getName());
150         }
151         return result.build();
152       }

154       // ProtocolSupport matches Message (not MessageLite) for legacy reasons
155       private static final Supplier<Type> MESSAGE_TYPE =
156           Suppliers.typeFromString("com.google.protobuf.MessageLite");
157     
158       private static final Supplier<Type> MUTABLE_MESSAGE_TYPE =
159           Suppliers.typeFromString("com.google.protobuf.MutableMessageLite");
160     
161       private static final Supplier<Type> PROTOCOL_MESSAGE_TYPE =
162           Suppliers.typeFromString("com.google.io.protocol.ProtocolMessage");
163     
164       private static boolean isAssignableTo(Type type, Supplier<Type> supplier, VisitorState state) {
165         Type to = supplier.get(state);
166         if (to == null) {
167           // the type couldn't be loaded
168           return false;
169         }
170         return state.getTypes().isAssignable(type, to);
171       }
172     
173       /**
174        * Compile-time equivalent of
175        * {@code com.google.io.protocol.ProtocolSupport#isProto2MessageClass}.
176        */
177       static boolean isProto2MessageClass(VisitorState state, Type type) {
178         checkNotNull(type);
179         return isAssignableTo(type, MESSAGE_TYPE, state)
180             && !isAssignableTo(type, PROTOCOL_MESSAGE_TYPE, state);
181       }
182     
183       /**
184        * Compile-time equivalent of
185        * {@code com.google.io.protocol.ProtocolSupport#isProto2MutableMessageClass}.
186        */
187       static boolean isProto2MutableMessageClass(VisitorState state, Type type) {
188         checkNotNull(type);
189         return isAssignableTo(type, MUTABLE_MESSAGE_TYPE, state)
190             && !isAssignableTo(type, PROTOCOL_MESSAGE_TYPE, state);
191       }
