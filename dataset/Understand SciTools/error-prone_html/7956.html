47     /**
48      * Matches the behaviour of javac's finally Xlint warning.
49      *
50      * 1) Any return statement in a finally block is an error
51      * 2) An uncaught throw statement in a finally block is an error. We can't always know whether a
52      *    specific exception will be caught, so we report errors for throw statements that are not
53      *    contained in a try with at least one catch block.
54      * 3) A continue statement in a finally block is an error if it breaks out of a (possibly labeled)
55      *    loop that is outside the enclosing finally.
56      * 4) A break statement in a finally block is an error if it breaks out of a (possibly labeled)
57      *    loop or a switch statement that is outside the enclosing finally.
58      *
59      * @author eaftan@google.com (Eddie Aftandilian)
60      * @author cushon@google.com (Liam Miller-Cushon)
61      */
62     @BugPattern(
63       name = "Finally",
64       altNames = {"finally", "ThrowFromFinallyBlock"},
65       summary =
66           "If you return or throw from a finally, then values returned or thrown from the"
67               + " try-catch block will be ignored. Consider using try-with-resources instead.",
68       category = JDK,
69       severity = WARNING,
70       maturity = MATURE,
71       generateExamplesFromTestCases = false
72     )
73     public class Finally extends BugChecker
74         implements ContinueTreeMatcher, ThrowTreeMatcher, BreakTreeMatcher, ReturnTreeMatcher {
75     
76       @Override
77       public Description matchContinue(ContinueTree tree, VisitorState state) {
78         if (new FinallyJumpMatcher((JCContinue) tree).matches(tree, state)) {
79           return describeMatch(tree);
80         }
81         return Description.NO_MATCH;
82       }
83     
84       @Override
85       public Description matchBreak(BreakTree tree, VisitorState state) {
86         if (new FinallyJumpMatcher((JCBreak) tree).matches(tree, state)) {
87           return describeMatch(tree);
88         }
89         return Description.NO_MATCH;
90       }
91     
92       @Override
93       public Description matchThrow(ThrowTree tree, VisitorState state) {
94         if (new FinallyThrowMatcher().matches(tree, state)) {
95           return describeMatch(tree);
96         }
97         return Description.NO_MATCH;
98       }
99     
100       @Override
101       public Description matchReturn(ReturnTree tree, VisitorState state) {
102         if (new FinallyCompletionMatcher<ReturnTree>().matches(tree, state)) {
103           return describeMatch(tree);
104         }
105         return Description.NO_MATCH;
106       }
107     
108       private static enum MatchResult {
109         KEEP_LOOKING, NO_MATCH, FOUND_ERROR;
110       }
111     
112       /**
113        * Base class for all finally matchers.  Walks up the tree of enclosing statements and
114        * reports an error if it finds an enclosing finally block.
115        *
116        * @param <T> The type of the tree node to match against
117        */
118       private static class FinallyCompletionMatcher<T extends StatementTree> implements Matcher<T> {
119     
120         /**
121          * Matches a StatementTree type by walking that statement's ancestor chain.
122          * @returns true if an error is found.
123          */
124         @Override
125         public boolean matches(T tree, VisitorState state) {
126     
127           TreePath path = state.getPath();
128           Tree prevTree = path.getLeaf();
129     
130           while (path != null && path.getLeaf().getKind() != Kind.METHOD
131               && path.getLeaf().getKind() != Kind.COMPILATION_UNIT) {
132             prevTree = path.getLeaf();
133             path = path.getParentPath();
134     
135             MatchResult mr = (matchAncestor(path.getLeaf(), prevTree));
136             if (mr != MatchResult.KEEP_LOOKING) {
137               return mr == MatchResult.FOUND_ERROR;
138             }
139           }
140     
141           return false;
142         }
143     
144         /**
145          * Match a tree in the ancestor chain given the ancestor's immediate descendant.
146          */
147         protected MatchResult matchAncestor(Tree leaf, Tree prevTree) {
148           if (leaf instanceof TryTree) {
149             TryTree tryTree = (TryTree) leaf;
150             if (tryTree.getFinallyBlock() != null && tryTree.getFinallyBlock().equals(prevTree)) {
151               return MatchResult.FOUND_ERROR;
152             }
153           }
154     
155           return MatchResult.KEEP_LOOKING;
156         }
157       }
158     
159       /**
160        * Ancestor matcher for statements that break or continue out of a finally block.
161        */
162       private static class FinallyJumpMatcher extends FinallyCompletionMatcher<StatementTree> {
163         private final Name label;
164         private final JumpType jumpType;
165     
166         private enum JumpType {
167           BREAK,
168           CONTINUE
169         }
170     
171         public FinallyJumpMatcher(JCContinue jcContinue) {
172           this.label = jcContinue.label;
173           this.jumpType = JumpType.CONTINUE;
174         }
175     
176         public FinallyJumpMatcher(JCBreak jcBreak) {
177           this.label = jcBreak.label;
178           this.jumpType = JumpType.BREAK;
179         }
180     
181         /**
182          * The target of a jump statement (break or continue) is
183          * (1) the enclosing loop if the jump is unlabeled
184          * (2) the enclosing LabeledStatementTree with matching label if the jump is labeled
185          * (3) the enclosing switch statement if the jump is a break
186          *
187          * If the target of a break or continue statement is encountered before reaching a finally
188          * block, return NO_MATCH.
189          */
190         @Override
191         protected MatchResult matchAncestor(Tree leaf, Tree prevTree) {
192     
193           // (1)
194           if (label == null) {
195             switch (leaf.getKind()) {
196               case WHILE_LOOP:
197               case DO_WHILE_LOOP:
198               case FOR_LOOP:
199               case ENHANCED_FOR_LOOP:
200                 return MatchResult.NO_MATCH;
201               default:
202                 break;
203             }
204           }
205     
206           // (2)
207           if (label != null
208               && leaf instanceof LabeledStatementTree
209               && label.equals(((LabeledStatementTree) leaf).getLabel())) {
210               return MatchResult.NO_MATCH;
211           }
212     
213           // (3)
214           if (jumpType == JumpType.BREAK && leaf instanceof SwitchTree) {
215             return MatchResult.NO_MATCH;
216           }
217     
218           return super.matchAncestor(leaf, prevTree);
219         }
220       }
221     
222       /**
223        * Match throw statements that are not caught.
224        */
225       private class FinallyThrowMatcher extends FinallyCompletionMatcher<ThrowTree> {
226         @Override
227         protected MatchResult matchAncestor(Tree tree, Tree prevTree) {
228           if (tree instanceof TryTree) {
229             TryTree tryTree = (TryTree) tree;
230             if (tryTree.getBlock().equals(prevTree) && !tryTree.getCatches().isEmpty()) {
231               // The current ancestor is a try block with associated catch blocks.
232               return MatchResult.NO_MATCH;
233             }
234           }
235     
236           return super.matchAncestor(tree, prevTree);
237         }
238       }
