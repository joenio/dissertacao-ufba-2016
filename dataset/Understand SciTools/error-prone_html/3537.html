44     /**
45      * Base class for JUnit4SetUp/TearDown not run. This will take care of the nitty-gritty about
46      * replacing @After with @Before, adding @Before on unannotated methods, making them public
47      * if necessary, fixing the imports of other @Before, etc.
48      *
49      * @author glorioso@google.com
50      */
51     abstract class AbstractJUnit4InitMethodNotRun extends BugChecker implements MethodTreeMatcher {
52       private static final String JUNIT_TEST = "org.junit.Test";
53       protected final JUnitMatchers.JUnit4TestClassMatcher isJUnit4TestClass;
54     
55       protected AbstractJUnit4InitMethodNotRun() {
56         this.isJUnit4TestClass = new JUnitMatchers.JUnit4TestClassMatcher();
57       }
58     
59       /**
60        * Returns a matcher that selects which methods this matcher applies to
61        * (e.g. public void setUp() without @Before/@BeforeClass annotation)
62        */
63       protected abstract Matcher<MethodTree> methodMatcher();
64     
65       /**
66        * Returns the fully qualified class name of the annotation this bugpattern should apply to
67        * matched methods.
68        *
69        * If another annotation is on the method that has the same name, the import will be replaced
70        * with the appropriate one (e.g.: com.example.Before becomes org.junit.Before)
71        */
72       protected abstract String correctAnnotation();
73     
74       /**
75        * Returns a collection of 'before-and-after' pairs of annotations that should be replaced on
76        * these methods.
77        *
78        * <p>If this method matcher finds a method annotated with
79        * {@link AnnotationReplacements#badAnnotation}, instead of applying
80        * {@link #correctAnnotation()}, instead replace it with
81        * {@link AnnotationReplacements#goodAnnotation}
82        */
83       protected abstract List<AnnotationReplacements> annotationReplacements();
84     
85     
86       /**
87        * Matches if all of the following conditions are true:
88        * 1) The method matches {@link #methodMatcher()}, (looks like setUp() or tearDown(),
89        *    and none of the overrides in the hierarchy of the method have the appropriate @Before
90        *    or @After annotations)
91        * 2) The method is not annotated with @Test
92        * 3) The enclosing class has an @RunWith annotation and does not extend TestCase. This marks
93        *    that the test is intended to run with JUnit 4.
94        */
95       @Override
96       public Description matchMethod(MethodTree methodTree, VisitorState state) {
97         boolean matches = allOf(
98             methodMatcher(),
99             not(hasAnnotationOnAnyOverriddenMethod(JUNIT_TEST)),
100             enclosingClass(isJUnit4TestClass))
101             .matches(methodTree, state);
102         if (!matches) {
103           return Description.NO_MATCH;
104         }
105     
106         // For each annotationReplacement, replace the first annotation that matches. If any of them
107         // matches, don't try and do the rest of the work.
108         Description description;
109         for (AnnotationReplacements replacement : annotationReplacements()) {
110           description = tryToReplaceAnnotation(
111               methodTree, state, replacement.badAnnotation, replacement.goodAnnotation);
112           if (description != null) {
113             return description;
114           }
115         }
116     
117         // Search for another @Before annotation on the method and replace the import
118         // if we find one
119         String correctAnnotation = correctAnnotation();
120         String unqualifiedClassName = getUnqualifiedClassName(correctAnnotation);
121         for (AnnotationTree annotationNode : methodTree.getModifiers().getAnnotations()) {
122           String annotationClassName =
123               ASTHelpers.getSymbol(annotationNode).getQualifiedName().toString();
124           if (annotationClassName.endsWith("." + unqualifiedClassName)) {
125             SuggestedFix.Builder suggestedFix = SuggestedFix.builder()
126                 .removeImport(annotationClassName)
127                 .addImport(correctAnnotation);
128             if (makeProtectedPublic(methodTree, state, unqualifiedClassName, suggestedFix, false)
129                 == null) {
130               // No source position available, don't suggest a fix
131               return describeMatch(annotationNode);
132             }
133             suggestedFix.replace(annotationNode, "@" + unqualifiedClassName);
134             return describeMatch(annotationNode, suggestedFix.build());
135           }
136         }
137     
138         // Add correctAnnotation() to the unannotated method
139         // (and convert protected to public if it is)
140         SuggestedFix.Builder suggestedFix = SuggestedFix.builder().addImport(correctAnnotation);
141     
142         // The makeProtectedPublic will take care of adding the annotation for us
143         Boolean annotationAdded =
144             makeProtectedPublic(methodTree, state, unqualifiedClassName, suggestedFix, true);
145         //
146         if (annotationAdded == null) {
147           // No source position available, don't suggest a fix
148           return describeMatch(methodTree);
149         }
150     
151         if (!annotationAdded) {
152           suggestedFix.prefixWith(methodTree, "@" + unqualifiedClassName + "\n");
153         }
154     
155         return describeMatch(methodTree, suggestedFix.build());
156       }
157     
158       // returns null if the method source isn't available for some reason. Caller should check
159       // null and not emit a fix in that case. Returns true if the method was upgraded from protected
160       // to public
161       @Nullable
162       private Boolean makeProtectedPublic(MethodTree methodTree, VisitorState state,
163           String unqualifiedClassName, SuggestedFix.Builder suggestedFix, boolean addAnnotation) {
164         if (Matchers.<MethodTree>hasModifier(Modifier.PROTECTED).matches(methodTree, state)) {
165           CharSequence methodSource = state.getSourceForNode((JCTree.JCMethodDecl) methodTree);
166           if (methodSource == null) {
167             return null;
168           }
169           String methodString = (addAnnotation ? "@" + unqualifiedClassName + "\n" : "")
170               + methodSource.toString().replaceFirst("protected ", "public ");
171           suggestedFix.replace(methodTree, methodString);
172           return true;
173         }
174         return false;
175       }
176     
177     
178       private Description tryToReplaceAnnotation(MethodTree methodTree, VisitorState state,
179                                                  String badAnnotation, String goodAnnotation) {
180         String finalName = getUnqualifiedClassName(goodAnnotation);
181         if (hasAnnotation(badAnnotation).matches(methodTree, state)) {
182           AnnotationTree annotationTree = findAnnotation(methodTree, state, badAnnotation);
183           return describeMatch(annotationTree, SuggestedFix.builder()
184               .addImport(goodAnnotation)
185               .replace(annotationTree, "@" + finalName)
186               .build());
187         } else {
188           return null;
189         }
190       }
191     
192       private String getUnqualifiedClassName(String goodAnnotation) {
193         return goodAnnotation.substring(goodAnnotation.lastIndexOf(".") + 1);
194       }
195     
196       private AnnotationTree findAnnotation(MethodTree methodTree, VisitorState state,
197                                             String annotationName) {
198         AnnotationTree annotationNode = null;
199         for (AnnotationTree annotation : methodTree.getModifiers().getAnnotations()) {
200           if (ASTHelpers.getSymbol(annotation)
201               .equals(state.getSymbolFromString(annotationName))) {
202             annotationNode = annotation;
203           }
204         }
205         return annotationNode;
206       }
207     
208       protected static class AnnotationReplacements implements Serializable {
209         private final String goodAnnotation;
210         private final String badAnnotation;
211     
212         protected AnnotationReplacements(String badAnnotation, String goodAnnotation) {
213           this.goodAnnotation = goodAnnotation;
214           this.badAnnotation = badAnnotation;
215         }
216       }
217     }
