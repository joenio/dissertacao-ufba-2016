37     import com.google.errorprone.matchers.Matchers;
38     
39     import com.sun.source.tree.ExpressionTree;
40     import com.sun.source.tree.MethodInvocationTree;
41     import com.sun.tools.javac.tree.JCTree.JCFieldAccess;
42     
43     /**
44      * @author eaftan@google.com (Eddie Aftandilian)
45      */
46     @BugPattern(name = "ArrayEquals",
47         summary = "Reference equality used to compare arrays",
48         explanation =
49             "Generally when comparing arrays for equality, the programmer intends to check that the "
50             + "the contents of the arrays are equal rather than that they are actually the same "
51             + "object.  But many commonly used equals methods compare arrays for reference equality "
52             + "rather than content equality. These include the instance .equals() method, Guava's "
53             + "com.google.common.base.Objects#equal(), and the JDK's java.util.Objects#equals().\n\n"
54             + "If reference equality is needed, == should be used instead for clarity. Otherwise, "
55             + "use java.util.Arrays#equals() to compare the contents of the arrays.",
56         category = JDK, severity = ERROR, maturity = MATURE)
57     public class ArrayEquals extends BugChecker implements MethodInvocationTreeMatcher {
58       /**
59        * Matches when the equals instance method is used to compare two arrays.
60        */
61       private static final Matcher<MethodInvocationTree> instanceEqualsMatcher = Matchers.allOf(
62           instanceMethod().onClass(isArray()).named("equals"),
63           argument(0, Matchers.<ExpressionTree>isArrayType()));
64     
65       /**
66        * Matches when the Guava com.google.common.base.Objects#equal or the JDK7
67        * java.util.Objects#equals method is used to compare two arrays.
68        */
69       private static final Matcher<MethodInvocationTree> staticEqualsMatcher = allOf(
70           anyOf(
71             staticMethod().onClass("com.google.common.base.Objects").named("equal"),
72             staticMethod().onClass("java.util.Objects").named("equals")),
73           argument(0, Matchers.<ExpressionTree>isArrayType()),
74           argument(1, Matchers.<ExpressionTree>isArrayType()));
75     
76       /**
77        * Suggests replacing with Arrays.equals(a, b). Also adds the necessary import statement for
78        * java.util.Arrays.
79        */
80       @Override
81       public Description matchMethodInvocation(MethodInvocationTree t, VisitorState state) {
82         String arg1;
83         String arg2;
84         if (instanceEqualsMatcher.matches(t, state)) {
85           arg1 = ((JCFieldAccess) t.getMethodSelect()).getExpression().toString();
86           arg2 = t.getArguments().get(0).toString();
87         } else if (staticEqualsMatcher.matches(t, state)) {
88           arg1 = t.getArguments().get(0).toString();
89           arg2 = t.getArguments().get(1).toString();
90         } else {
91           return NO_MATCH;
92         }
93     
94         Fix fix = SuggestedFix.builder()
95             .replace(t, "Arrays.equals(" + arg1 + ", " + arg2 + ")")
96             .addImport("java.util.Arrays")
97             .build();
98         return describeMatch(t, fix);
99       }
100     }
