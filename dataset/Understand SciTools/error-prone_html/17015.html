17     package com.google.errorprone.bugpatterns;
18     
19     import static com.google.errorprone.BugPattern.Category.JDK;
20     import static com.google.errorprone.BugPattern.MaturityLevel.MATURE;
21     import static com.google.errorprone.BugPattern.SeverityLevel.ERROR;
22     import static com.google.errorprone.util.ASTHelpers.isSameType;
23     
24     import com.google.errorprone.BugPattern;
25     import com.google.errorprone.VisitorState;
26     import com.google.errorprone.bugpatterns.BugChecker.MethodTreeMatcher;
27     import com.google.errorprone.fixes.SuggestedFix;
28     import com.google.errorprone.matchers.Description;
29     import com.google.errorprone.util.ASTHelpers;
30     
31     import com.sun.source.tree.MethodTree;
32     import com.sun.tools.javac.code.Symbol.MethodSymbol;
33     
34     import javax.lang.model.element.Modifier;
35     
36     /** @author cushon@google.com (Liam Miller-Cushon) */
37     @BugPattern(
38       name = "UnsynchronizedOverridesSynchronized",
39       summary = "Unsynchronized method overrides a synchronized method.",
40       category = JDK,
41       severity = ERROR,
42       maturity = MATURE
43     )
44     public class UnsynchronizedOverridesSynchronized extends BugChecker implements MethodTreeMatcher {
45       @Override
46       public Description matchMethod(MethodTree methodTree, VisitorState state) {
47         MethodSymbol methodSymbol = ASTHelpers.getSymbol(methodTree);
48     
49         if (isSynchronized(methodSymbol)) {
50           return Description.NO_MATCH;
51         }
52         for (MethodSymbol s : ASTHelpers.findSuperMethods(methodSymbol, state.getTypes())) {
53           if (isSynchronized(s)) {
54             // Input streams are typically not used across threads, so this case isn't
55             // worth enforcing.
56             if (isSameType(s.owner.type, state.getTypeFromString("java.io.InputStream"), state)) {
57               continue;
58             }
59             return buildDescription(methodTree)
60                 .addFix(SuggestedFix.addModifier(methodTree, Modifier.SYNCHRONIZED, state))
61                 .setMessage(
62                     String.format(
63                         "Unsynchronized method %s overrides synchronized method in %s",
64                         methodSymbol.getSimpleName(),
65                         s.enclClass().getSimpleName()))
66                 .build();
67           }
68         }
69         return Description.NO_MATCH;
70       }
71       
72       private static boolean isSynchronized(MethodSymbol sym) {
73         return sym.getModifiers().contains(Modifier.SYNCHRONIZED);
74       }
75     }
