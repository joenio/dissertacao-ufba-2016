162       private static Map<String, BugPattern.SeverityLevel> defaultSeverities(
163           Iterable<BugChecker> checkers) {
164         ImmutableMap.Builder<String, BugPattern.SeverityLevel> builder = ImmutableMap.builder();
165         for (BugChecker check : checkers) {
166           builder.put(check.canonicalName(), check.defaultSeverity());
167         }
168         return builder.build();
169       }

171       /**
172        * Create an error-prone scanner for a non-hardcoded set of checkers.
173        *
174        * @param checkers The checkers that this scanner should use.
175        */
176       public ErrorProneScanner(Iterable<BugChecker> checkers) {
177         this(checkers, defaultSeverities(checkers));
178       }

181        * Create an error-prone scanner for a non-hardcoded set of checkers.
182        *
183        * @param checkers The checkers that this scanner should use.

185        */
186       public ErrorProneScanner(Iterable<BugChecker> checkers, Map<String, SeverityLevel> severities) {
187         this.severities = severities;
188         for (BugChecker checker : checkers) {
189           registerNodeTypes(checker);
190         }
191       }

461       @Override
462       public Void visitAnnotation(AnnotationTree tree, VisitorState visitorState) {
463         VisitorState state = visitorState.withPath(getCurrentPath());
464         for (AnnotationTreeMatcher matcher : annotationMatchers) {
465           if (!isSuppressed(matcher, state)) {
466             reportMatch(matcher.matchAnnotation(tree, state), tree, state);
467           }
468         }
469         return super.visitAnnotation(tree, visitorState);
470       }
