64     // TODO(cushon): allow @LazyInit on fields as a suppression mechanism?
65     @BugPattern(
66         name = "DoubleCheckedLocking",
67         summary = "Double-checked locking on non-volatile fields is unsafe",
68         category = JDK,
69         severity = ERROR,
70         maturity = MATURE)
71     public class DoubleCheckedLocking extends BugChecker implements IfTreeMatcher {
72       @Override
73       public Description matchIf(IfTree outerIf, VisitorState state) {
74         DCLInfo info = findDCL(outerIf);
75         if (info == null) {
76           return Description.NO_MATCH;
77         }
78         switch (info.sym().getKind()) {
79           case FIELD:
80             return handleField(info.outerIf(), info.sym(), state);
81           case LOCAL_VARIABLE:
82             return handleLocal(info, state);
83           default:
84             return Description.NO_MATCH;
85         }
86       }
87     
88       /**
89        * Report a {@link Description} if a field used in double-checked locking is not volatile.
90        *
91        * <p>If the AST node for the field declaration can be located in the current compilation unit,
92        * suggest adding the volatile modifier.
93        */
94       private Description handleField(IfTree outerIf, VarSymbol sym, VisitorState state) {
95         if (sym.getModifiers().contains(Modifier.VOLATILE)) {
96           return Description.NO_MATCH;
97         }
98         if (isImmutable(sym.type, state)) {
99           return Description.NO_MATCH;
100         }
101         Description.Builder builder = buildDescription(outerIf);
102         JCTree fieldDecl = findFieldDeclaration(state.getPath(), sym);
103         if (fieldDecl != null) {
104           Fix fix = SuggestedFix.addModifier(fieldDecl, Modifier.VOLATILE, state);
105           if (fix != null) {
106             builder.addFix(fix);
107           }
108         }
109         return builder.build();
110       }

112       private static final ImmutableSet<String> IMMUTABLE_WHITELIST =
113           ImmutableSet.of(
114               java.lang.Boolean.class.getName(),
115               java.lang.Byte.class.getName(),
116               java.lang.Short.class.getName(),
117               java.lang.Integer.class.getName(),
118               java.lang.Character.class.getName(),
119               java.lang.Float.class.getName(),
120               java.lang.String.class.getName());
121     
122       /**
123        * Recognize a small set of known-immutable types that are safe for DCL even without
124        * a volatile field.
125        */
126       private static boolean isImmutable(Type type, VisitorState state) {
127         switch (type.getKind()) {
128           case BOOLEAN:
129           case BYTE:
130           case SHORT:
131           case INT:
132           case CHAR:
133           case FLOAT:
134             return true;
135           case LONG:
136           case DOUBLE:
137             // double-width primitives aren't written atomically
138             return true;
139           default:
140             break;
141         }
142         return IMMUTABLE_WHITELIST.contains(
143             state.getTypes().erasure(type).tsym.getQualifiedName().toString());
144       }
145     
146       /**
147        * Report a diagnostic for an instance of DCL on a local variable. A match is only reported
148        * if a non-volatile field is written to the variable after acquiring the lock and before
149        * the second null-check on the local.
150        *
151        * <p>e.g.
152        *
153        * <pre>
154        * {@code
155        * if ($X == null) {
156        *   synchronized (...) {
157        *     $X = myNonVolatileField;
158        *     if ($X == null) {
159        *       ...
160        *     }
161        *     ...
162        *   }
163        * }
164        * }
165        * </pre>
166        */
167       private Description handleLocal(DCLInfo info, VisitorState state) {
168         JCExpressionStatement expr = getChild(info.synchTree().getBlock(), JCExpressionStatement.class);
169         if (expr.getStartPosition() > ((JCTree) info.innerIf()).getStartPosition()) {
170           return Description.NO_MATCH;
171         }
172         if (!(expr.getExpression() instanceof JCAssign)) {
173           return Description.NO_MATCH;
174         }
175         JCAssign assign = (JCAssign) expr.getExpression();
176         if (!Objects.equals(ASTHelpers.getSymbol(assign.getVariable()), info.sym())) {
177           return Description.NO_MATCH;
178         }
179         Symbol sym = ASTHelpers.getSymbol(assign.getExpression());
180         if (!(sym instanceof VarSymbol)) {
181           return Description.NO_MATCH;
182         }
183         VarSymbol fvar = (VarSymbol) sym;
184         if (fvar.getKind() != ElementKind.FIELD) {
185           return Description.NO_MATCH;
186         }
187         return handleField(info.outerIf(), fvar, state);
188       }

190       /** Information about an instance of DCL. */
191       @AutoValue
192       abstract static class DCLInfo {
193         /** The outer if statement */
194         abstract IfTree outerIf();
195     
196         /** The synchronized statement */
197         abstract SynchronizedTree synchTree();
198     
199         /** The inner if statement **/
200         abstract IfTree innerIf();
201     
202         /** The variable (local or field) that is double-checked */
203         abstract VarSymbol sym();
204     
205         static DCLInfo create(
206             IfTree outerIf, SynchronizedTree synchTree, IfTree innerIf, VarSymbol sym) {
207           return new AutoValue_DoubleCheckedLocking_DCLInfo(outerIf, synchTree, innerIf, sym);
208         }
209       }
210     
211       /**
212        * Matches an instance of DCL. The canonical pattern is:
213        *
214        * <pre>
215        * {@code
216        * if ($X == null) {
217        *   synchronized (...) {
218        *     if ($X == null) {
219        *       ...
220        *     }
221        *     ...
222        *   }
223        * }
224        * }
225        * </pre>
226        *
227        * Gaps before the synchronized or inner 'if' statement are ignored, and the operands in the
228        * null-checks are accepted in either order.
229        */
230       @Nullable
231       static DCLInfo findDCL(IfTree outerIf) {
232         // TODO(cushon): Optional.ifPresent...
233         ExpressionTree outerIfTest = getNullCheckedExpression(outerIf.getCondition());
234         if (outerIfTest == null) {
235           return null;
236         }
237         SynchronizedTree synchTree = getChild(outerIf.getThenStatement(), SynchronizedTree.class);
238         if (synchTree == null) {
239           return null;
240         }
241         IfTree innerIf = getChild(synchTree.getBlock(), IfTree.class);
242         if (innerIf == null) {
243           return null;
244         }
245         ExpressionTree innerIfTest = getNullCheckedExpression(innerIf.getCondition());
246         if (innerIfTest == null) {
247           return null;
248         }
249         Symbol outerSym = ASTHelpers.getSymbol(outerIfTest);
250         if (!Objects.equals(outerSym, ASTHelpers.getSymbol(innerIfTest))) {
251           return null;
252         }
253         if (!(outerSym instanceof VarSymbol)) {
254           return null;
255         }
256         VarSymbol var = (VarSymbol) outerSym;
257         return DCLInfo.create(outerIf, synchTree, innerIf, var);
258       }

260       /**
261        * Matches comparisons to null (e.g. {@code foo == null}) and returns the expression being
262        * tested.
263        */
264       private static ExpressionTree getNullCheckedExpression(ExpressionTree condition) {
265         condition = TreeInfo.skipParens((JCExpression) condition);
266         if (!(condition instanceof BinaryTree)) {
267           return null;
268         }
269         BinaryTree bin = (BinaryTree) condition;
270         ExpressionTree other;
271         if (bin.getLeftOperand().getKind() == Kind.NULL_LITERAL) {
272           other = bin.getRightOperand();
273         } else if (bin.getRightOperand().getKind() == Kind.NULL_LITERAL) {
274           other = bin.getLeftOperand();
275         } else {
276           return null;
277         }
278         return other;
279       }

281       /**
282        * Visits (possibly nested) block statements and returns the first child statement with the
283        * given class.
284        */
285       private static <T> T getChild(StatementTree tree, final Class<T> clazz) {
286         return tree.accept(
287             new SimpleTreeVisitor<T, Void>() {
288               @Override
289               protected T defaultAction(Tree node, Void p) {
290                 if (clazz.isInstance(node)) {
291                   return clazz.cast(node);
292                 }
293                 return null;
294               }
295     
296               @Override
297               public T visitBlock(BlockTree node, Void p) {
298                 return visit(node.getStatements());
299               }
300     
301               private T visit(List<? extends Tree> tx) {
302                 for (Tree t : tx) {
303                   T r = t.accept(this, null);
304                   if (r != null) {
305                     return r;
306                   }
307                 }
308                 return null;
309               }
310             },
311             null);
312       }

314       /**
315        * Performs a best-effort search for the AST node of a field declaration.
316        *
317        * <p>It will only find fields declared in a lexically enclosing scope of the current location.
318        * Since double-checked locking should always be used on a private field, this should be
319        * reasonably effective.
320        */
321       @Nullable
322       private static JCTree findFieldDeclaration(TreePath path, VarSymbol var) {
323         for (TreePath curr = path; curr != null; curr = curr.getParentPath()) {
324           Tree leaf = curr.getLeaf();
325           if (!(leaf instanceof JCClassDecl)) {
326             continue;
327           }
328           for (JCTree tree : ((JCClassDecl) leaf).getMembers()) {
329             if (Objects.equals(var, ASTHelpers.getSymbol(tree))) {
330               return tree;
331             }
332           }
333         }
334         return null;
335       }
