17     package com.google.errorprone.refaster;
18     
19     import com.sun.source.tree.EmptyStatementTree;
20     import com.sun.source.tree.TreeVisitor;
21     import com.sun.tools.javac.tree.JCTree.JCSkip;
22     
23     /**
24      * A {@link UTree} representation of an {@link EmptyStatementTree}
25      *
26      * @author lowasser@google.com (Louis Wasserman)
27      */
28     final class USkip extends USimpleStatement implements EmptyStatementTree {
29       public static final USkip INSTANCE = new USkip();
30       
31       private USkip() {}
32       
33       Object readResolve() {
34         return INSTANCE;
35       }
36     
37       @Override
38       public JCSkip inline(Inliner inliner) {
39         return inliner.maker().Skip();
40       }
41     
42       @Override
43       public <R, D> R accept(TreeVisitor<R, D> visitor, D data) {
44         return visitor.visitEmptyStatement(this, data);
45       }
46     
47       @Override
48       public Kind getKind() {
49         return Kind.EMPTY_STATEMENT;
50       }
51     
52       @Override
53       public Choice<Unifier> visitEmptyStatement(EmptyStatementTree node, Unifier unifier) {
54         return Choice.of(unifier);
55       }
56     
57       @Override
58       public String toString() {
59         return "USkip{}";
60       }
61     }
