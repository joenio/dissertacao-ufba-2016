22     import static com.sun.source.tree.Tree.Kind.VARIABLE;
23     import static javax.lang.model.element.Modifier.FINAL;
24     
25     import com.google.errorprone.BugPattern;
26     import com.google.errorprone.VisitorState;
27     import com.google.errorprone.bugpatterns.BugChecker.VariableTreeMatcher;
28     import com.google.errorprone.fixes.SuggestedFix;
29     import com.google.errorprone.matchers.ChildMultiMatcher.MatchType;

32     import com.google.errorprone.matchers.MultiMatcher;
33     import com.google.errorprone.util.ASTHelpers;

35     import com.sun.source.tree.AnnotationTree;
36     import com.sun.source.tree.ClassTree;
37     import com.sun.source.tree.Tree;
38     import com.sun.source.tree.VariableTree;
39     import com.sun.tools.javac.code.Flags;
40     import com.sun.tools.javac.tree.JCTree.JCModifiers;
41     import com.sun.tools.javac.tree.JCTree.JCVariableDecl;
42     import com.sun.tools.javac.tree.TreeMaker;
43     
44     /**
45      * @author sgoldfeder@google.com (Steven Goldfeder)
46      */
47     @BugPattern(name = "GuiceInjectOnFinalField", summary =
48         "Although Guice allows injecting final fields, doing so is not "
49         + "recommended because the injected value may not be visible to other threads.",
50         explanation = "See https://code.google.com/p/google-guice/wiki/InjectionPoints",
51         category = GUICE, severity = WARNING, maturity = EXPERIMENTAL)
52     @SuppressWarnings("serial")
53     public class GuiceInjectOnFinalField extends BugChecker implements VariableTreeMatcher {
54     
55       private static final String GUICE_INJECT_ANNOTATION = "com.google.inject.Inject";
56     
57       private static final MultiMatcher<Tree, AnnotationTree> ANNOTATED_WITH_GUICE_INJECT_MATCHER =
58           annotations(MatchType.AT_LEAST_ONE, isType(GUICE_INJECT_ANNOTATION));
59     
60       private static final Matcher<Tree> FINAL_FIELD_MATCHER = new Matcher<Tree>() {
61         @Override
62         public boolean matches(Tree t, VisitorState state) {
63           return isField(t, state) && ((VariableTree) t).getModifiers().getFlags().contains(FINAL);
64         }
65       };
66     
67       @Override
68       public Description matchVariable(VariableTree tree, VisitorState state) {
69         if (ANNOTATED_WITH_GUICE_INJECT_MATCHER.matches(tree, state)
70                 && FINAL_FIELD_MATCHER.matches(tree, state)) {
71           JCModifiers modifiers = ((JCVariableDecl) tree).getModifiers();
72           long replacementFlags = modifiers.flags ^ Flags.FINAL;
73           JCModifiers replacementModifiers = TreeMaker.instance(state.context)
74               .Modifiers(replacementFlags, modifiers.annotations);
75           /*
76            * replace new lines with strings, trim whitespace and remove empty parens to make the
77            * suggested fixes look sane
78            */
79           String replacementModifiersString =
80               replacementModifiers.toString().replace('\n', ' ').replace("()", "").trim();
81           return describeMatch(modifiers,
82               SuggestedFix.replace(modifiers, replacementModifiersString));
83         }
84         return Description.NO_MATCH;
85       }
86     
87       private static boolean isField(Tree tree, VisitorState state) {
88         return tree.getKind().equals(VARIABLE)
89             && ASTHelpers.findEnclosingNode(state.getPath(), ClassTree.class)
90                 .getMembers().contains(tree);
91       }
92     }
