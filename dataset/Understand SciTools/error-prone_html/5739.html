38     /**
39      * Encapsulates the logic of handling suppressions, both via {@code @SuppressWarnings} and via
40      * custom suppression annotations.  To do this we have to maintains 2 sets of suppression info:
41      * 1) A set of the suppression strings in all {@code @Suppresswarnings} annotations down this path
42      *    of the AST.
43      * 2) A set of all custom suppression annotations down this path of the AST.
44      */
45     public class SuppressionHelper {
46     
47       /**
48        * The set of custom suppression annotations that this SuppressionHelper should look for.
49        */
50       private final Set<Class<? extends Annotation>> customSuppressionAnnotations;
51     
52     
53       /**
54        * @param customSuppressionAnnotations The set of custom suppression annotations that this
55        * SuppressionHelper should look for.
56        * @param state the {@link VisitorState} of the current analysis.
57        */
58       public SuppressionHelper(
59           Set<Class<? extends Annotation>> customSuppressionAnnotations, VisitorState state) {
60         if (customSuppressionAnnotations == null) {
61           throw new IllegalArgumentException("customSuppressionAnnotations must be non-null");
62         }
63         this.customSuppressionAnnotations = customSuppressionAnnotations;
64       }
65     
66       /**
67        * Container for information about suppressions.  Either reference field may be null, which
68        * indicates that the suppression sets are unchanged.
69        */
70       public static class SuppressionInfo {
71         public Set<String> suppressWarningsStrings;
72         public Set<Class<? extends Annotation>> customSuppressions;
73         public boolean inGeneratedCode;
74     
75         public SuppressionInfo(
76             Set<String> suppressWarningsStrings,
77             Set<Class<? extends Annotation>> customSuppressions,
78             boolean inGeneratedCode) {
79           this.suppressWarningsStrings = suppressWarningsStrings;
80           this.customSuppressions = customSuppressions;
81           this.inGeneratedCode = inGeneratedCode;
82         }
83       }
84     
85       /**
86        * Extend suppression sets for both {@code @SuppressWarnings} and custom suppression annotations.
87        * When we explore a new node, we have to extend the suppression sets with any new suppressed
88        * warnings or custom suppression annotations.  We also have to retain the previous suppression
89        * set so that we can reinstate it when we move up the tree.
90        *
91        * We do not modify the existing suppression sets, so they can be restored when moving up the
92        * tree.  We also avoid copying the suppression sets if the next node to explore does not have
93        * any suppressed warnings or custom suppression annotations.  This is the common case.
94        *
95        * @param sym The {@code Symbol} for the AST node currently being scanned
96        * @param suppressWarningsType The {@code Type} for {@code @SuppressWarnings}, as given by
97        *        javac's symbol table
98        * @param suppressionsOnCurrentPath The set of strings in all {@code @SuppressWarnings}
99        *        annotations on the current path through the AST
100        * @param customSuppressionsOnCurrentPath The set of all custom suppression annotations
101        */
102       public SuppressionInfo extendSuppressionSets(
103           Symbol sym,
104           Type suppressWarningsType,
105           Set<String> suppressionsOnCurrentPath,
106           Set<Class<? extends Annotation>> customSuppressionsOnCurrentPath,
107           boolean inGeneratedCode,
108           VisitorState state) {
109     
110         boolean newInGeneratedCode =
111             inGeneratedCode || ASTHelpers.hasAnnotation(sym, Generated.class, state);
112     
113         /**
114          * Handle custom suppression annotations.
115          */
116         Set<Class<? extends Annotation>> newCustomSuppressions = null;
117         for (Class<? extends Annotation> annotationType : customSuppressionAnnotations) {
118           if (ASTHelpers.hasAnnotation(sym, annotationType, state)) {
119             if (newCustomSuppressions == null) {
120               newCustomSuppressions = new HashSet<>(customSuppressionsOnCurrentPath);
121             }
122             newCustomSuppressions.add(annotationType);
123           }
124         }
125     
126         /**
127          * Handle @SuppressWarnings.
128          */
129         Set<String> newSuppressions = null;
130         // Iterate over annotations on this symbol, looking for SuppressWarnings
131         for (Attribute.Compound attr : sym.getAnnotationMirrors()) {
132           // TODO(eaftan): use JavacElements.getAnnotation instead
133           if (attr.type.tsym == suppressWarningsType.tsym) {
134             for (List<Pair<MethodSymbol,Attribute>> v = attr.values;
135                 v.nonEmpty(); v = v.tail) {
136               Pair<MethodSymbol,Attribute> value = v.head;
137               if (value.fst.name.contentEquals("value"))
138                 if (value.snd instanceof Attribute.Array) { // SuppressWarnings takes an array
139                   for (Attribute suppress : ((Attribute.Array) value.snd).values) {
140                     if (newSuppressions == null) {
141                       newSuppressions = new HashSet<>(suppressionsOnCurrentPath);
142                     }
143                     // TODO(eaftan): check return value to see if this was a new warning?
144                     newSuppressions.add((String) suppress.getValue());
145                   }
146                 } else {
147                   throw new RuntimeException("Expected SuppressWarnings annotation to take array type");
148                 }
149             }
150           }
151         }
152     
153         return new SuppressionInfo(newSuppressions, newCustomSuppressions, newInGeneratedCode);
154       }
155     
156       /**
157        * Returns true if this checker should be suppressed on the current tree path.
158        *
159        * @param suppressible Holds information about the suppressibilty of a checker
160        * @param suppressionsOnCurrentPath The set of strings in all {@code @SuppressWarnings}
161        *        annotations on the current path through the AST
162        * @param customSuppressionsOnCurrentPath The set of all custom suppression annotations
163        *        on the current path through the AST
164        * @param severityLevel of the check to be suppressed
165        * @param inGeneratedCode true if the current code is generated
166        * @param disableWarningsInGeneratedCode true if warnings in generated code should be suppressed
167        */
168       public static boolean isSuppressed(
169           Suppressible suppressible,
170           Set<String> suppressionsOnCurrentPath,
171           Set<Class<? extends Annotation>> customSuppressionsOnCurrentPath,
172           SeverityLevel severityLevel,
173           boolean inGeneratedCode,
174           boolean disableWarningsInGeneratedCode) {
175         if (suppressible.suppressibility() == Suppressibility.UNSUPPRESSIBLE) {
176           return false;
177         }
178         if (inGeneratedCode && disableWarningsInGeneratedCode && severityLevel != SeverityLevel.ERROR) {
179           return true;
180         }
181         switch (suppressible.suppressibility()) {
182           case CUSTOM_ANNOTATION:
183             return !Collections.disjoint(
184                 suppressible.customSuppressionAnnotations(), customSuppressionsOnCurrentPath);
185           case SUPPRESS_WARNINGS:
186             return !Collections.disjoint(suppressible.allNames(), suppressionsOnCurrentPath);
187           default:
188             throw new IllegalStateException("No case for: " + suppressible.suppressibility());
189         }
190       }
