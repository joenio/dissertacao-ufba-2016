38     /**
39      * A {@link UTree} representation of a local variable declaration.
40      *
41      * <p>A {@code UVariableDecl} can be unified with any variable declaration which has a matching 
42      * type and initializer. Annotations and modifiers are preserved for the corresponding replacement,
43      * as well as the variable name. {@link ULocalVarIdent} instances are used to represent references 
44      * to local variables.
45      *
46      * <p>As a result, we can modify variable declarations and initializations in target code while
47      * preserving variable names and other contextual information.
48      *
49      * @author lowasser@google.com (Louis Wasserman)
50      */
51     @AutoValue
52     public abstract class UVariableDecl extends USimpleStatement implements VariableTree {
53     
54       public static UVariableDecl create(
55           CharSequence identifier, UExpression type, @Nullable UExpression initializer) {
56         return new AutoValue_UVariableDecl(StringName.of(identifier), type, initializer);
57       }
58     
59       public static UVariableDecl create(CharSequence identifier, UExpression type) {
60         return create(identifier, type, null);
61       }
62       
63       @Override
64       public abstract StringName getName();
65       
66       @Override
67       public abstract UExpression getType();
68       
69       @Override
70       @Nullable
71       public abstract UExpression getInitializer();
72       
73       ULocalVarIdent.Key key() {
74         return new ULocalVarIdent.Key(getName());
75       }
76     
77       @Override
78       public Choice<Unifier> visitVariable(final VariableTree decl, Unifier unifier) {
79         return Choice.condition(unifier.getBinding(key()) == null, unifier)
80             .thenChoose(unifications(getType(), decl.getType()))
81             .thenChoose(unifications(getInitializer(), decl.getInitializer()))
82             .transform(new Function<Unifier, Unifier>() {
83               @Override
84               public Unifier apply(Unifier unifier) {
85                 unifier.putBinding(key(),
86                     LocalVarBinding.create(ASTHelpers.getSymbol(decl), decl.getModifiers()));
87                 return unifier;
88               }
89             });
90       }
91     
92       @Override
93       public JCVariableDecl inline(Inliner inliner) throws CouldNotResolveImportException {
94         Optional<LocalVarBinding> binding = inliner.getOptionalBinding(key());
95         JCModifiers modifiers;
96         Name name;
97         TreeMaker maker = inliner.maker();
98         if (binding.isPresent()) {
99           modifiers = (JCModifiers) binding.get().getModifiers();
100           name = binding.get().getName();
101         } else {
102           modifiers = maker.Modifiers(0L);
103           name = getName().inline(inliner);
104         }
105         return maker.VarDef(modifiers, name, getType().inline(inliner),
106             (getInitializer() == null) ? null : getInitializer().inline(inliner));
107       }
108     
109       @Override
110       public Kind getKind() {
111         return Kind.VARIABLE;
112       }
113     
114       @Override
115       public <R, D> R accept(TreeVisitor<R, D> visitor, D data) {
116         return visitor.visitVariable(this, data);
117       }
118     
119       @Override
120       public ModifiersTree getModifiers() {
121         throw new UnsupportedOperationException();
122       }
123     
124       @Override
125       public ExpressionTree getNameExpression() {
126         // TODO(cushon): java 8.
127         throw new IllegalStateException();
128       }
129     }
