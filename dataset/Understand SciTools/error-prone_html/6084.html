39     /**
40      * Checks for the existence of a JUnit3 style setUp() method in a JUnit4 test class.
41      *
42      * @author glorioso@google.com (Nick Glorioso)
43      */
44     @BugPattern(name = "JUnit4SetUpNotRun",
45         summary = "setUp() method will not be run; Please add a @Before annotation",
46         explanation = "JUnit 3 provides the method setUp(), to be overridden by subclasses"
47             + " when the test needs to perform some pre-test initialization. In JUnit 4, this"
48             + " is accomplished by annotating such a method with @Before.\n\n"
49             + " The method that triggered this error matches the definition of setUp() from"
50             + " JUnit3, but was not annotated with @Before and thus won't be run by the JUnit4"
51             + " runner.\n\n"
52             + " If you intend for this setUp() method not to run by the JUnit4 runner, but perhaps"
53             + " manually be invoked in certain test methods, please rename the method or mark"
54             + " it private. \n\n"
55             + " If the method is part of an abstract test class hierarchy"
56             + " where this class's setUp() is invoked by a superclass method that is annotated with"
57             + " @Before, then please rename the abstract method or add @Before to"
58             + " the superclass's definition of setUp()",
59         category = JUNIT, maturity = MATURE, severity = ERROR)
60     public class JUnit4SetUpNotRun extends AbstractJUnit4InitMethodNotRun {
61       @Override
62       protected Matcher<MethodTree> methodMatcher() {
63         return allOf(looksLikeJUnit3SetUp, not(hasJUnit4BeforeAnnotations));
64       }
65     
66       @Override
67       protected String correctAnnotation() {
68         return JUNIT_BEFORE_ANNOTATION;
69       }
70     
71       @Override
72       protected List<AnnotationReplacements> annotationReplacements() {
73         return Arrays.asList(
74             new AnnotationReplacements(JUNIT_AFTER_ANNOTATION, JUNIT_BEFORE_ANNOTATION),
75             new AnnotationReplacements(JUNIT_AFTER_CLASS_ANNOTATION, JUNIT_BEFORE_CLASS_ANNOTATION));
76       }
77     }
