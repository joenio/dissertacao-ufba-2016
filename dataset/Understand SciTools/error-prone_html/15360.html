39     /**
40      * Policy specifying when and how to import classes when inlining types.
41      *
42      * @author Louis Wasserman
43      */
44     public enum ImportPolicy {
45       // TODO(lowasser): add support for other policies, like "explicitly qualify everything"
46     
47       /**
48        * Import the outermost class and explicitly qualify references below that. For example, to
49        * reference {@code com.google.Foo.Bar}, we import {@code com.google.Foo} and explicitly qualify
50        * {@code Foo.Bar}.
51        */
52       IMPORT_TOP_LEVEL {
53         @Override
54         public JCExpression classReference(
55             Inliner inliner, CharSequence topLevelClazz, CharSequence fullyQualifiedClazz) {
56           List<String> allImports = getAllImports(inliner);
57           /*
58            * Check if topLevelClazz or fullyQualifiedClazz is already imported.
59            * If fullyQualifiedClazz is imported, return the class name.
60            * If topLevelClazz is imported, return the name from the top level class to the class name.
61            * If there are imports whose names conflict with topLevelClazz, return fully qualified name.
62            * Otherwise, import toplevelClazz, and return the name from the top level class
63            * to the class name.
64            */
65           checkArgument(topLevelClazz.length() > 0 && fullyQualifiedClazz.length() > 0,
66               String.format(
67                   "either topLevelClass (%s) or fullyQualifiedClazz (%s) is null or empty",
68                   topLevelClazz,
69                   fullyQualifiedClazz));
70           List<String> topLevelPath = Splitter.on('.').splitToList(topLevelClazz);
71           String topClazz = Iterables.getLast(topLevelPath);
72           List<String> qualifiedPath = Splitter.on('.').splitToList(fullyQualifiedClazz);
73           boolean importTopLevelClazz = false, conflictTopLevelClazz = false;
74           for (String importName : allImports) {
75             if (importName.contentEquals(fullyQualifiedClazz)) {
76               // fullyQualifiedClazz already imported
77               return makeSelectExpression(inliner, qualifiedPath, qualifiedPath.size() - 1);
78             }
79             importTopLevelClazz |= importName.contentEquals(topLevelClazz);
80             if (!importTopLevelClazz) {
81               conflictTopLevelClazz |= topClazz.equals(
82                   Iterables.getLast(Splitter.on('.').split(importName)));
83             }
84           }
85           if (importTopLevelClazz) {
86               return makeSelectExpression(inliner, qualifiedPath, topLevelPath.size() - 1);
87           } else if (conflictTopLevelClazz) {
88               return makeSelectExpression(inliner, qualifiedPath, 0);
89           }
90           // No conflicts
91           String packge = Joiner.on('.').join(topLevelPath.subList(0, topLevelPath.size() - 1));
92           PackageSymbol currentPackage = inliner.getContext().get(PackageSymbol.class);
93           if (currentPackage == null || !currentPackage.getQualifiedName().contentEquals(packge)) {
94             // don't import classes from the same package as the class we're refactoring
95             inliner.addImport(topLevelClazz.toString());
96           }
97           return makeSelectExpression(inliner, qualifiedPath, topLevelPath.size() - 1);
98         }
99     
100         @Override
101         public JCExpression staticReference(Inliner inliner, CharSequence topLevelClazz,
102             CharSequence fullyQualifiedClazz, CharSequence member) {
103           return inliner.maker().Select(
104               classReference(inliner, topLevelClazz, fullyQualifiedClazz), inliner.asName(member));
105         }
106     
107         /* Returns a combined list of strings from importsInSource and importsToAdd. */
108         private List<String> getAllImports(Inliner inliner) {
109           List<String> allImports = new ArrayList<>(inliner.getImportsToAdd());
110           if (inliner.getContext() != null
111               && inliner.getContext().get(JCCompilationUnit.class) != null) {
112             for (JCImport jcImport : inliner.getContext().get(JCCompilationUnit.class).getImports()) {
113               JCFieldAccess qualified = (JCFieldAccess) jcImport.getQualifiedIdentifier();
114               allImports.add(qualified.toString());
115             }
116           }
117           return allImports;
118         }
119     
120         private JCExpression makeSelectExpression(
121             Inliner inliner, List<String> qualifiedPath, int start) {
122           Iterator<String> selects = qualifiedPath.listIterator(start);
123           TreeMaker maker = inliner.maker();
124           JCExpression select = maker.Ident(inliner.asName(selects.next()));
125           while (selects.hasNext()) {
126             select = maker.Select(select, inliner.asName(selects.next()));
127           }
128           return select;
129         }
130       },
131       /**
132        * Import nested classes directly, and qualify static references from the class level.
133        */
134       IMPORT_CLASS_DIRECTLY {
135         @Override
136         public JCExpression classReference(Inliner inliner, CharSequence topLevelClazz,
137             CharSequence fullyQualifiedClazz) {
138           String packge = topLevelClazz.toString();
139           int lastDot = packge.lastIndexOf('.');
140           packge = (lastDot >= 0) ? packge.substring(0, lastDot) : "";
141           PackageSymbol currentPackage = inliner.getContext().get(PackageSymbol.class);
142           if (currentPackage == null || !currentPackage.getQualifiedName().contentEquals(packge)
143               || !topLevelClazz.equals(fullyQualifiedClazz)) {
144             // don't import classes from the same package as the class we're refactoring
145             inliner.addImport(fullyQualifiedClazz.toString());
146           }
147           String simpleName = fullyQualifiedClazz.toString();
148           simpleName = simpleName.substring(simpleName.lastIndexOf('.') + 1);
149           return inliner.maker().Ident(inliner.asName(simpleName));
150         }
151     
152         @Override
153         public JCExpression staticReference(Inliner inliner, CharSequence topLevelClazz,
154             CharSequence fullyQualifiedClazz, CharSequence member) {
155           return inliner.maker().Select(
156               classReference(inliner, topLevelClazz, fullyQualifiedClazz), 
157               inliner.asName(member));
158         }
159       },
160     
161       /**
162        * When inlining static methods, always static import the method.  Non-static references to
163        * classes are imported from the top level as in {@code IMPORT_TOP_LEVEL}.
164        */
165       STATIC_IMPORT_ALWAYS {
166         @Override
167         public JCExpression classReference(
168             Inliner inliner, CharSequence topLevelClazz, CharSequence fullyQualifiedClazz) {
169           return IMPORT_TOP_LEVEL.classReference(inliner, topLevelClazz, fullyQualifiedClazz);
170         }
171     
172         @Override
173         public JCExpression staticReference(Inliner inliner, CharSequence topLevelClazz,
174             CharSequence fullyQualifiedClazz, CharSequence member) {
175           inliner.addStaticImport(fullyQualifiedClazz + "." + member);
176           return inliner.maker().Ident(inliner.asName(member));
177         }    
178       };
179       
180       public static void bind(Context context, ImportPolicy policy) {
181         context.put(ImportPolicy.class, checkNotNull(policy));
182       }
183     
184       public static ImportPolicy instance(Context context) {
185         ImportPolicy result = context.get(ImportPolicy.class);
186         checkState(result != null, "No ImportPolicy bound in this context");
187         return result;
188       }
189     
190       public abstract JCExpression classReference(
191           Inliner inliner, CharSequence topLevelClazz, CharSequence fullyQualifiedClazz);
192     
193       public abstract JCExpression staticReference(Inliner inliner, CharSequence topLevelClazz,
194           CharSequence fullyQualifiedClazz, CharSequence member);
195     }
