15     package com.google.errorprone.analysis;
16     
17     import com.google.auto.value.AutoValue;
18     import com.google.errorprone.ErrorProneOptions;
19     import com.google.errorprone.InvalidCommandLineOptionException;
20     
21     import java.util.Set;
22     
23     /**
24      * Config for all kinds of analyses.
25      * 
26      * @author Louis Wasserman
27      */
28     @AutoValue
29     public abstract class AnalysesConfig {
30       public static AnalysesConfig create(ErrorProneOptions options) {
31         return new AutoValue_AnalysesConfig(options);
32       }
33       
34       AnalysesConfig() {}
35       
36       abstract ErrorProneOptions errorProneOptions();
37       
38       void validate(TopLevelAnalysis analysis) throws InvalidCommandLineOptionException {
39         if (!errorProneOptions().ignoreUnknownChecks()) {
40           Set<String> knownAnalyses = analysis.knownAnalysisNames();
41           for (String configured : errorProneOptions().getSeverityMap().keySet()) {
42             if (!knownAnalyses.contains(configured)) {
43               throw new InvalidCommandLineOptionException(configured + " is not a valid checker name");
44             }
45           }
46         }
47       }
48     }
