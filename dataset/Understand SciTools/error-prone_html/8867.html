17     package com.google.errorprone.refaster;
18     
19     import com.google.auto.value.AutoValue;
20     import com.google.common.collect.ImmutableList;
21     
22     import com.sun.source.tree.ModifiersTree;
23     import com.sun.source.tree.TreeVisitor;
24     import com.sun.tools.javac.code.Flags;
25     import com.sun.tools.javac.tree.JCTree.JCAnnotation;
26     import com.sun.tools.javac.tree.JCTree.JCModifiers;
27     import com.sun.tools.javac.util.List;
28     
29     import java.util.Set;
30     
31     import javax.lang.model.element.Modifier;
32     
33     
34     /**
35      * {@code UTree} representation of a {@code ModifiersTree}.
36      * 
37      * @author lowasser@google.com (Louis Wasserman)
38      */
39     @AutoValue
40     abstract class UModifiers extends UTree<JCModifiers> implements ModifiersTree {
41       public static UModifiers create(long flagBits, UAnnotation... annotations) {
42         return create(flagBits, ImmutableList.copyOf(annotations));
43       }
44       
45       public static UModifiers create(long flagBits, Iterable<? extends UAnnotation> annotations) {
46         return new AutoValue_UModifiers(flagBits, ImmutableList.copyOf(annotations));
47       }
48       
49       abstract long flagBits();
50     
51       @Override
52       public abstract ImmutableList<UAnnotation> getAnnotations();
53       
54       @Override
55       public JCModifiers inline(Inliner inliner) throws CouldNotResolveImportException {
56         return inliner.maker().Modifiers(
57             flagBits(), 
58             List.convert(JCAnnotation.class, inliner.inlineList(getAnnotations())));
59       }
60     
61       @Override
62       public Choice<Unifier> visitModifiers(ModifiersTree modifier, Unifier unifier) {
63         return Choice.condition(getFlags().equals(modifier.getFlags()), unifier);
64       }
65     
66       @Override
67       public <R, D> R accept(TreeVisitor<R, D> visitor, D data) {
68         return visitor.visitModifiers(this, data);
69       }
70     
71       @Override
72       public Kind getKind() {
73         return Kind.MODIFIERS;
74       }
75     
76       @Override
77       public Set<Modifier> getFlags() {
78         return Flags.asModifierSet(flagBits());
79       }
80     }
