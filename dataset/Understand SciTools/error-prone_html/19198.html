56     /**
57      * Checks that the 1st argument to Preconditions.checkNotNull() isn't a primitive.
58      * The primitive would be autoboxed to a non-null boxed type, and the check would trivially
59      * pass.
60      *
61      * In our experience, most of these errors are from copied-and-pasted code and should
62      * simply be removed.
63      *
64      * @author sjnickerson@google.com (Simon Nickerson)
65      * @author eaftan@google.com (Eddie Aftandilian)
66      */
67     @BugPattern(name = "PreconditionsCheckNotNullPrimitive",
68         summary = "First argument to `Preconditions.checkNotNull()` is a primitive rather "
69             + "than an object reference",
70         explanation =
71             "`Preconditions.checkNotNull()` takes as an argument a reference that should be " +
72             "non-null. Often a primitive is passed as the argument to check. The primitive " +
73             "will be [autoboxed]" +
74             "(http://docs.oracle.com/javase/7/docs/technotes/guides/language/autoboxing.html) " +
75             "into a boxed object, which is non-null, causing the check to " +
76             "always pass without the condition being evaluated.\n\n" +
77             "If the intent was to ensure that the primitive met some criterion (e.g., a boolean " +
78             "that should be non-null), please use `Preconditions.checkState()` or " +
79             "`Preconditions.checkArgument()` instead.",
80         category = GUAVA, severity = ERROR, maturity = MATURE)
81     public class PreconditionsCheckNotNullPrimitive
82         extends BugChecker implements MethodInvocationTreeMatcher {
83     
84       @Override
85       public Description matchMethodInvocation(MethodInvocationTree methodInvocationTree, VisitorState state) {
86         if (allOf(
87                 staticMethod().onClass("com.google.common.base.Preconditions").named("checkNotNull"),
88                 argument(0, Matchers.<ExpressionTree>isPrimitiveType()))
89                 .matches(methodInvocationTree, state)) {
90           return describe(methodInvocationTree, state);
91         }
92         return Description.NO_MATCH;
93       }
94     
95       /**
96        * If the call to Preconditions.checkNotNull is part of an expression (assignment, return, etc.),
97        * we substitute the argument for the method call. E.g.:
98        * {@code bar = Preconditions.checkNotNull(foo); ==> bar = foo;}
99        *
100        * <p>If the argument to Preconditions.checkNotNull is a comparison using == or != and one of the
101        * operands is null, we call checkNotNull on the non-null operand. E.g.:
102        * {@code checkNotNull(a == null); ==> checkNotNull(a);}
103        *
104        * <p>If the argument is a method call or binary tree and its return type is boolean, change it to a
105        * checkArgument/checkState. E.g.:
106        * {@code Preconditions.checkNotNull(foo.hasFoo()) ==> Preconditions.checkArgument(foo.hasFoo())}
107        *
108        * <p>Otherwise, delete the checkNotNull call. E.g.:
109        * {@code Preconditions.checkNotNull(foo); ==> [delete the line]}
110        */
111       public Description describe(MethodInvocationTree methodInvocationTree, VisitorState state) {
112         ExpressionTree arg1 = methodInvocationTree.getArguments().get(0);
113         Tree parent = state.getPath().getParentPath().getLeaf();
114     
115         // Assignment, return, etc.
116         if (parent.getKind() != Kind.EXPRESSION_STATEMENT) {
117           return describeMatch(arg1,
118               SuggestedFix.replace(methodInvocationTree, arg1.toString()));
119         }
120     
121         // Comparison to null
122         if (arg1.getKind() == Kind.EQUAL_TO || arg1.getKind() == Kind.NOT_EQUAL_TO) {
123           BinaryTree binaryExpr = (BinaryTree) arg1;
124           if (binaryExpr.getLeftOperand().getKind() == Kind.NULL_LITERAL) {
125             return describeMatch(arg1,
126                 SuggestedFix.replace(arg1, binaryExpr.getRightOperand().toString()));
127           }
128           if (binaryExpr.getRightOperand().getKind() == Kind.NULL_LITERAL) {
129             return describeMatch(arg1,
130                 SuggestedFix.replace(arg1, binaryExpr.getLeftOperand().toString()));
131           }
132         }
133     
134         if ((arg1 instanceof BinaryTree || arg1.getKind() == Kind.METHOD_INVOCATION ||
135              arg1.getKind() == Kind.LOGICAL_COMPLEMENT) &&
136             ((JCExpression) arg1).type == state.getSymtab().booleanType) {
137           return describeMatch(arg1,
138               createCheckArgumentOrStateCall(methodInvocationTree, state, arg1));
139         }
140     
141         return describeMatch(arg1, SuggestedFix.delete(parent));
142       }
143     
144       /**
145        * Creates a SuggestedFix that replaces the checkNotNull call with a checkArgument or checkState
146        * call.
147        */
148       private Fix createCheckArgumentOrStateCall(MethodInvocationTree methodInvocationTree,
149           VisitorState state, ExpressionTree arg1) {
150         SuggestedFix.Builder fix = SuggestedFix.builder();
151         String replacementMethod = "checkState";
152         if (hasMethodParameter(state.getPath(), arg1)) {
153           replacementMethod = "checkArgument";
154         }
155     
156         StringBuilder replacement = new StringBuilder();
157     
158         // Was the original call to Preconditions.checkNotNull a static import or not?
159         if (methodInvocationTree.getMethodSelect().getKind() == Kind.IDENTIFIER) {
160           replacement.append(replacementMethod + "(");
161           fix.addStaticImport("com.google.common.base.Preconditions." + replacementMethod);
162         } else {
163           replacement.append("Preconditions." + replacementMethod + "(");
164         }
165     
166         Joiner.on(", ").appendTo(replacement, methodInvocationTree.getArguments());
167     
168         replacement.append(")");
169         fix.replace(methodInvocationTree, replacement.toString());
170         return fix.build();
171       }

173       /**
174        * Determines whether the expression contains a reference to one of the
175        * enclosing method's parameters.
176        *
177        * TODO(eaftan): Extract this to ASTHelpers.
178        *
179        * @param path the path to the current tree node
180        * @param tree the node to compare against the parameters
181        * @return whether the argument is a parameter to the enclosing method
182        */
183       private static boolean hasMethodParameter(TreePath path, ExpressionTree tree) {
184         Set<Symbol> symbols = new HashSet<>();
185         for (IdentifierTree ident : getVariableUses(tree)) {
186           Symbol sym = ASTHelpers.getSymbol(ident);
187           if (sym.isLocal()) {
188             symbols.add(sym);
189           }
190         }
191     
192         // Find enclosing method declaration.
193         while (path != null && !(path.getLeaf() instanceof MethodTree)) {
194           path = path.getParentPath();
195         }
196         if (path == null) {
197           throw new IllegalStateException("Should have an enclosing method declaration");
198         }
199         MethodTree methodDecl = (MethodTree) path.getLeaf();
200         for (VariableTree param : methodDecl.getParameters()) {
201           if (symbols.contains(ASTHelpers.getSymbol(param))) {
202             return true;
203           }
204         }
205     
206         return false;
207       }

209       /**
210        * Find the root variable identifiers from an arbitrary expression.
211        *
212        * Examples:
213        *    a.trim().intern() ==> {a}
214        *    a.b.trim().intern() ==> {a}
215        *    this.intValue.foo() ==> {this}
216        *    this.foo() ==> {this}
217        *    intern() ==> {}
218        *    String.format() ==> {}
219        *    java.lang.String.format() ==> {}
220        *    x.y.z(s.t) ==> {x,s}
221        */
222       static List<IdentifierTree> getVariableUses(ExpressionTree tree) {
223         final List<IdentifierTree> freeVars = new ArrayList<>();
224     
225         new TreeScanner<Void, Void>() {
226           @Override
227           public Void visitIdentifier(IdentifierTree node, Void v) {
228             if (((JCIdent) node).sym instanceof VarSymbol) {
229               freeVars.add(node);
230             }
231             return super.visitIdentifier(node, v);
232           }
233         }.scan(tree, null);
234     
235         return freeVars;
236       }
