17     package com.google.errorprone.matchers.method;
18     
19     import com.google.common.base.Optional;
20     import com.google.common.collect.ImmutableList;
21     import com.google.errorprone.VisitorState;
22     import com.google.errorprone.matchers.method.MethodMatchers.StaticMethodMatcher;
23     import com.google.errorprone.predicates.TypePredicates;
24     import com.google.errorprone.suppliers.Supplier;
25     
26     import com.sun.source.tree.ExpressionTree;
27     import com.sun.tools.javac.code.Type;
28     
29     /** Matches static methods, allows refinement on class type. */
30     class StaticMethodMatcherImpl extends MethodMatcher implements StaticMethodMatcher {
31       @Override
32       protected Optional<MatchState> matchResult(ExpressionTree item, MatchState method,
33           VisitorState state) {
34         if (!method.sym().isStatic()) {
35           return Optional.absent();
36         }
37         return Optional.of(method);
38       }
39     
40       @Override
41       public MethodClassMatcherImpl onClass(String className) {
42         return new MethodClassMatcherImpl(this, TypePredicates.isExactType(className));
43       }
44     
45       @Override
46       public MethodClassMatcherImpl onClassAny(Iterable<String> classNames) {
47         return new MethodClassMatcherImpl(this, TypePredicates.isExactTypeAny(classNames));
48       }
49       
50       @Override
51       public MethodClassMatcherImpl onClassAny(String... classNames) {
52         return onClassAny(ImmutableList.copyOf(classNames));
53       }
54     
55       @Override
56       public MethodClassMatcherImpl onClass(Supplier<Type> classType) {
57         return new MethodClassMatcherImpl(this, TypePredicates.isExactType(classType));
58       }
59     }
