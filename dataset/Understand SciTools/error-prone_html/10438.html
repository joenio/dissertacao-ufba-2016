17     package com.google.errorprone.refaster;
18     
19     import static com.google.errorprone.refaster.Unifier.unifications;
20     import static com.google.errorprone.refaster.Unifier.unifyNullable;
21     
22     import com.google.auto.value.AutoValue;
23     import com.google.common.collect.ImmutableList;
24     
25     import com.sun.source.tree.NewClassTree;
26     import com.sun.source.tree.TreeVisitor;
27     import com.sun.tools.javac.tree.JCTree.JCExpression;
28     import com.sun.tools.javac.tree.JCTree.JCNewClass;
29     
30     import java.util.List;
31     
32     import javax.annotation.Nullable;
33     
34     /**
35      * {@link UTree} version of {@link NewClassTree}, which represents a constructor invocation.
36      *
37      * @author lowasser@google.com (Louis Wasserman)
38      */
39     @AutoValue
40     abstract class UNewClass extends UExpression implements NewClassTree {
41     
42       public static UNewClass create(UExpression enclosingExpression,
43           List<? extends UExpression> typeArguments, UExpression identifier,
44           List<UExpression> arguments, @Nullable UClassDecl classBody) {
45         return new AutoValue_UNewClass(
46             enclosingExpression, ImmutableList.copyOf(typeArguments), identifier, 
47             ImmutableList.copyOf(arguments), classBody);
48       }
49     
50       public static UNewClass create(
51           List<? extends UExpression> typeArguments, UExpression identifier, 
52           UExpression... arguments) {
53         return create(null, typeArguments, identifier, ImmutableList.copyOf(arguments), null);
54       }
55     
56       public static UNewClass create(UExpression identifier, UExpression... arguments) {
57         return create(ImmutableList.<UExpression>of(), identifier, arguments);
58       }
59     
60       @Override
61       @Nullable
62       public abstract UExpression getEnclosingExpression();
63     
64       /**
65        * Note: these are not the type arguments to the class, but to the constructor, for
66        * those extremely rare constructors that look like e.g. {@code <E> Foo(E e)}, where
67        * the type parameter is for the constructor alone and not the class. 
68        */
69       @Override
70       public abstract List<UExpression> getTypeArguments();
71     
72       @Override
73       public abstract UExpression getIdentifier();
74     
75       @Override
76       public abstract List<UExpression> getArguments();
77       
78       @Override
79       @Nullable
80       public abstract UClassDecl getClassBody();
81     
82       @Override
83       @Nullable
84       public Choice<Unifier> visitNewClass(NewClassTree newClass, @Nullable Unifier unifier) {
85         return unifyNullable(unifier, getEnclosingExpression(), newClass.getEnclosingExpression())
86             .thenChoose(unifications(getTypeArguments(), newClass.getTypeArguments()))
87             .thenChoose(unifications(getIdentifier(), newClass.getIdentifier()))
88             .thenChoose(unifications(getClassBody(), newClass.getClassBody()))
89             .thenChoose(unifications(getArguments(), newClass.getArguments()));
90       }
91     
92       @Override
93       public Kind getKind() {
94         return Kind.NEW_CLASS;
95       }
96     
97       @Override
98       public <R, D> R accept(TreeVisitor<R, D> visitor, D data) {
99         return visitor.visitNewClass(this, data);
100       }
101     
102       @Override
103       public JCNewClass inline(Inliner inliner) throws CouldNotResolveImportException {
104         return inliner.maker().NewClass(
105             (getEnclosingExpression() == null) ? null : getEnclosingExpression().inline(inliner),
106             inliner.<JCExpression>inlineList(getTypeArguments()),
107             getIdentifier().inline(inliner),
108             inliner.<JCExpression>inlineList(getArguments()),
109             (getClassBody() == null) ? null : getClassBody().inline(inliner));
110       }
111     }
