55     /**
56      * @author scottjohnson@google.com (Scott Johnson)
57      */
58     @BugPattern(name = "ModifyingCollectionWithItself",
59         summary = "Modifying a collection with itself",
60         explanation = "Modifying a collection with itself is almost never what is intended. " +
61             "collection.addAll(collection) and collection.retainAll(collection) are both no-ops, " +
62             "and collection.removeAll(collection) is equivalent to collection.clear().",
63         category = JDK, severity = ERROR, maturity = EXPERIMENTAL)
64     public class ModifyingCollectionWithItself extends BugChecker
65         implements MethodInvocationTreeMatcher {
66     
67       /**
68        * Matches calls to addAll, containsAll, removeAll, and retainAll on itself
69        */
70       @Override
71       public Description matchMethodInvocation(MethodInvocationTree t, VisitorState state) {
72         if (allOf(anyOf(
73             instanceMethod().onDescendantOf("java.util.Collection").named("addAll"),
74             instanceMethod().onDescendantOf("java.util.Collection").named("removeAll"),
75             instanceMethod().onDescendantOf("java.util.Collection").named("containsAll"),
76             instanceMethod().onDescendantOf("java.util.Collection").named("retainAll")),
77             receiverSameAsArgument(0)).matches(t, state)) {
78           return describe(t, state);
79         }
80         return Description.NO_MATCH;
81       }
82     
83       /**
84        * We expect that the lhs is a field and the rhs is an identifier, specifically
85        * a parameter to the method.  We base our suggested fixes on this expectation.
86        *
87        * Case 1: If lhs is a field and rhs is an identifier, find a method parameter
88        * of the same type and similar name and suggest it as the rhs.  (Guess that they
89        * have misspelled the identifier.)
90        *
91        * Case 2: If lhs is a field and rhs is not an identifier, find a method parameter
92        * of the same type and similar name and suggest it as the rhs.
93        *
94        * Case 3: If lhs is not a field and rhs is an identifier, find a class field
95        * of the same type and similar name and suggest it as the lhs.
96        *
97        * Case 4: Otherwise replace with literal meaning of functionality
98        */
99       public Description describe(MethodInvocationTree methodInvocationTree, VisitorState state) {
100     
101         // the statement that is the parent of the self-assignment expression
102         Tree parent = state.getPath().getParentPath().getLeaf();
103     
104         ExpressionTree lhs = ASTHelpers.getReceiver(methodInvocationTree);
105         ExpressionTree rhs = methodInvocationTree.getArguments().get(0);
106     
107         // default fix for methods
108         Fix fix = SuggestedFix.delete(parent);
109         if (instanceMethod().anyClass().named("removeAll").matches(methodInvocationTree, state)) {
110           fix = SuggestedFix.replace(methodInvocationTree, lhs + ".clear()");
111         }
112     
113         if (lhs.getKind() == MEMBER_SELECT) {
114           // find a method parameter of the same type and similar name and suggest it
115           // as the rhs
116     
117           // rhs should be either identifier or field access
118           assert(rhs.getKind() == IDENTIFIER || rhs.getKind() == MEMBER_SELECT);
119     
120           // get current name of rhs
121           String rhsName = null;
122           if (rhs.getKind() == IDENTIFIER) {
123             rhsName = ((JCIdent) rhs).name.toString();
124           } else if (rhs.getKind() == MEMBER_SELECT) {
125             rhsName = ((JCFieldAccess) rhs).name.toString();
126           }
127     
128           // find method parameters of the type "Collection"
129           TreePath path = state.getPath();
130           while (path != null && path.getLeaf().getKind() != METHOD) {
131             path = path.getParentPath();
132           }
133           JCMethodDecl method = (JCMethodDecl) path.getLeaf();
134           int minEditDistance = Integer.MAX_VALUE;
135           String replacement = null;
136           for (JCVariableDecl var : method.params) {
137             if (variableType(isSubtypeOf("java.util.Collection")).matches(var, state)) {
138               int editDistance = EditDistance.getEditDistance(rhsName, var.name.toString());
139               if (editDistance < minEditDistance) {
140                 // pick one with minimum edit distance
141                 minEditDistance = editDistance;
142                 replacement = var.name.toString();
143               }
144             }
145           }
146           if (replacement != null) {
147             // suggest replacing rhs with the parameter
148             fix = SuggestedFix.replace(rhs, replacement);
149           }
150         } else if (rhs.getKind() == IDENTIFIER) {
151           // find a field of the same type and similar name and suggest it as the lhs
152     
153           // lhs should be identifier
154           assert(lhs.getKind() == IDENTIFIER);
155     
156           // get current name of lhs
157           String lhsName = ((JCIdent) rhs).name.toString();
158     
159           // find class instance fields of the type "Collection"
160           TreePath path = state.getPath();
161           while (path != null && path.getLeaf().getKind() != CLASS) {
162             path = path.getParentPath();
163           }
164           JCClassDecl klass = (JCClassDecl) path.getLeaf();
165           int minEditDistance = Integer.MAX_VALUE;
166           String replacement = null;
167           for (JCTree member : klass.getMembers()) {
168             if (member.getKind() == VARIABLE) {
169               JCVariableDecl var = (JCVariableDecl) member;
170               if (!Flags.isStatic(var.sym)
171                   && variableType(isSubtypeOf("java.util.Collection")).matches(var, state)) {
172                 int editDistance = EditDistance.getEditDistance(lhsName, var.name.toString());
173                 if (editDistance < minEditDistance) {
174                   // pick one with minimum edit distance
175                   minEditDistance = editDistance;
176                   replacement = var.name.toString();
177                 }
178               }
179             }
180           }
181           if (replacement != null) {
182             // suggest replacing lhs with the field
183             fix = SuggestedFix.replace(lhs, "this." + replacement);
184           }
185         }
186     
187         return describeMatch(methodInvocationTree, fix);
188       }
189     }
