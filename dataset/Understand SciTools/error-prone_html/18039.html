17     package com.google.errorprone.bugpatterns;
18     
19     import static com.google.errorprone.BugPattern.Category.JDK;
20     import static com.google.errorprone.BugPattern.MaturityLevel.MATURE;
21     import static com.google.errorprone.BugPattern.SeverityLevel.ERROR;
22     
23     import com.google.errorprone.BugPattern;
24     import com.google.errorprone.VisitorState;
25     import com.google.errorprone.bugpatterns.BugChecker.BinaryTreeMatcher;
26     import com.google.errorprone.fixes.SuggestedFix;
27     import com.google.errorprone.matchers.Description;
28     import com.google.errorprone.util.ASTHelpers;
29     
30     import com.sun.source.tree.BinaryTree;
31     import com.sun.source.tree.ExpressionTree;
32     import com.sun.tools.javac.code.Symbol;
33     import com.sun.tools.javac.tree.JCTree;
34     import com.sun.tools.javac.tree.JCTree.JCExpression;
35     
36     import javax.annotation.Nullable;
37     
38     /**
39      * @author lowasser@google.com (Louis Wasserman)
40      */
41     @BugPattern(name = "EqualsNaN",
42         summary = "== NaN always returns false; use the isNaN methods instead",
43         explanation = "As per JLS 15.21.1, == NaN comparisons always return false, even NaN == NaN. "
44             + "Instead, use the isNaN methods to check for NaN.",
45         category = JDK, severity = ERROR, maturity = MATURE)
46     public class EqualsNaN extends BugChecker implements BinaryTreeMatcher {  
47       @Override
48       public Description matchBinary(BinaryTree tree, VisitorState state) {
49         String prefix;
50         switch (tree.getKind()) {
51           case EQUAL_TO:
52             prefix = "";
53             break;
54           case NOT_EQUAL_TO:
55             prefix = "!";
56             break;
57           default:
58             return Description.NO_MATCH;
59         }
60         JCExpression left = (JCExpression) tree.getLeftOperand();
61         JCExpression right = (JCExpression) tree.getRightOperand();
62         
63         String leftMatch = matchNaN(left);
64         if (leftMatch != null) {
65           return describeMatch(tree, SuggestedFix.replace(tree,
66               String.format("%s%s.isNaN(%s)", prefix, leftMatch, toString(right, state))));
67         }
68         String rightMatch = matchNaN(right);
69         if (rightMatch != null) {
70           return describeMatch(tree, SuggestedFix.replace(tree,
71               String.format("%s%s.isNaN(%s)", prefix, rightMatch, toString(left, state))));
72         }
73         return Description.NO_MATCH;
74       }
75       
76       private CharSequence toString(JCTree tree, VisitorState state) {
77         CharSequence source = state.getSourceForNode(tree);
78         return (source == null) ? tree.toString() : source;
79       }
80     
81       @Nullable
82       private String matchNaN(ExpressionTree tree) {
83         Symbol sym = ASTHelpers.getSymbol(tree);
84         if (sym != null && sym.owner != null && sym.owner.asType() != null
85             && sym.getSimpleName().contentEquals("NaN")) {
86           if (sym.owner.getQualifiedName().contentEquals("java.lang.Double")) {
87             return "Double";
88           } else if (sym.owner.getQualifiedName().contentEquals("java.lang.Float")) {
89             return "Float";
90           }
91         }
92         return null;
93       }
94     }
