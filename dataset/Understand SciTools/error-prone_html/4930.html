46     /**
47      * Representation of a {@code Refaster} placeholder method, which can represent an arbitrary
48      * operation on a specific set of expressions.
49      *
50      * @author lowasser@google.com (Louis Wasserman)
51      */
52     @AutoValue
53     abstract class PlaceholderMethod implements Serializable {
54       static PlaceholderMethod create(CharSequence name, UType returnType,
55           ImmutableMap<UVariableDecl, ImmutableClassToInstanceMap<Annotation>> parameters, 
56           ClassToInstanceMap<Annotation> annotations) {
57         final boolean allowsIdentity = annotations.getInstance(Placeholder.class).allowsIdentity();
58         final Class<? extends Matcher<? super ExpressionTree>> matchesClass =
59             annotations.containsKey(Matches.class)
60                 ? UTemplater.getValue(annotations.getInstance(Matches.class))
61                 : null;
62         final Class<? extends Matcher<? super ExpressionTree>> notMatchesClass =
63             annotations.containsKey(NotMatches.class)
64                 ? UTemplater.getValue(annotations.getInstance(NotMatches.class))
65                 : null;
66         final Predicate<Tree.Kind> allowedKinds =
67             annotations.containsKey(OfKind.class)
68                 ? Predicates.<Tree.Kind>in(Arrays.asList(annotations.getInstance(OfKind.class).value()))
69                 : Predicates.<Tree.Kind>alwaysTrue();
70         class PlaceholderMatcher implements Serializable, Matcher<ExpressionTree> {
71     
72           @Override
73           public boolean matches(ExpressionTree t, VisitorState state) {
74             try {
75               return (allowsIdentity || !(t instanceof PlaceholderParamIdent))
76                   && (matchesClass == null || matchesClass.newInstance().matches(t, state))
77                   && (notMatchesClass == null || !notMatchesClass.newInstance().matches(t, state))
78                   && allowedKinds.apply(t.getKind());
79             } catch (InstantiationException | IllegalAccessException e) {
80               throw new RuntimeException(e);
81             }
82           }
83         }
84         return new AutoValue_PlaceholderMethod(
85             StringName.of(name),
86             returnType,
87             parameters,
88             new PlaceholderMatcher(),
89             ImmutableClassToInstanceMap.<Annotation, Annotation>copyOf(annotations));
90       }
91     
92       abstract StringName name();
93     
94       abstract UType returnType();
95     
96       abstract ImmutableMap<UVariableDecl, ImmutableClassToInstanceMap<Annotation>> 
97           annotatedParameters();
98       
99       abstract Matcher<ExpressionTree> matcher();
100     
101       abstract ImmutableClassToInstanceMap<Annotation> annotations();
102     
103       ImmutableSet<UVariableDecl> parameters() {
104         return annotatedParameters().keySet();
105       }
106       
107       /**
108        * Parameters which must be referenced in any tree matched to this placeholder.
109        */
110       Set<UVariableDecl> requiredParameters() {
111         return Maps.filterValues(annotatedParameters(),
112             new Predicate<ImmutableClassToInstanceMap<Annotation>>() {
113               @Override
114               public boolean apply(ImmutableClassToInstanceMap<Annotation> annotations) {
115                 return !annotations.containsKey(MayOptionallyUse.class);
116               }
117             }).keySet();
118       }
119     
120       PlaceholderExpressionKey exprKey() {
121         return new PlaceholderExpressionKey(name().contents(), this);
122       }
123     
124       PlaceholderBlockKey blockKey() {
125         return new PlaceholderBlockKey(name().contents(), this);
126       }
127     
128       static final class PlaceholderExpressionKey extends Bindings.Key<JCExpression> 
129           implements Comparable<PlaceholderExpressionKey> {
130         final PlaceholderMethod method;
131         
132         private PlaceholderExpressionKey(String str, PlaceholderMethod method) {
133           super(str);
134           this.method = method;
135         }
136     
137         @Override
138         public int compareTo(PlaceholderExpressionKey o) {
139           return getIdentifier().compareTo(o.getIdentifier());
140         }
141       }
142     
143       static final class PlaceholderBlockKey extends Bindings.Key<List<JCStatement>> {
144         final PlaceholderMethod method;
145         
146         private PlaceholderBlockKey(String str, PlaceholderMethod method) {
147           super(str);
148           this.method = method;
149         }
150       }
151     }
