34     /**
35      * @author deminguyen@google.com (Demi Nguyen)
36      */
37     @RunWith(JUnit4.class)
38     public class NonNullLiteralTest extends CompilerBasedAbstractTest {
39     
40       @Test
41       public void shouldMatchPrimitiveLiterals() {
42         writeFile("A.java",
43             "public class A {",
44             "  public int getInt() {",
45             "    return 2;",
46             "  }",
47             "  public long getLong() {",
48             "    return 3L;",
49             "  }",
50             "  public float getFloat() {",
51             "    return 4.0f;",
52             "  }",
53             "  public double getDouble() {",
54             "    return 5.0d;",
55             "  }",
56             "  public boolean getBool() {",
57             "    return true;",
58             "  }",
59             "  public char getChar() {",
60             "    return 'c';",
61             "  }",
62             "  public String getString() {",
63             "    return \"test string\";",
64             "  }",
65             "}"
66           );
67           assertCompiles(nonNullLiteralMatches(true, Matchers.nonNullLiteral()));
68       }
69     
70       @Test
71       public void shouldMatchClassLiteral() {
72         writeFile("A.java",
73             "import java.lang.reflect.Type;",
74             "public class A {",
75             "  public void testClassLiteral() {",
76             "    Type klass = String.class;",
77             "  }",
78             "}"
79           );
80           assertCompiles(nonNullLiteralMatches(true, Matchers.nonNullLiteral()));
81       }
82     
83       @Test
84       public void shouldNotMatchClassDeclaration() {
85         writeFile("A.java",
86             "public class A {",
87             "  protected class B {",
88             "    private class C {",
89             "    }",
90             "  }",
91             "}"
92         );
93         assertCompiles(nonNullLiteralMatches(false, Matchers.nonNullLiteral()));
94       }
95     
96       @Test
97       public void shouldNotMatchMemberAccess() {
98         writeFile("A.java",
99             "package com.google;",
100             "public class A {",
101             "  public String stringVar;",
102             "  public void testMemberAccess() {",
103             "    this.stringVar = new String();",
104             "  }",
105             "}"
106         );
107     
108         writeFile("B.java",
109             "import com.google.A;",
110             "public class B {",
111             "  public void testInstanceMemberAccess() {",
112             "    A foo = new A();",
113             "    foo.stringVar = new String();",
114             "  }",
115             "}"
116         );
117         assertCompiles(nonNullLiteralMatches(false, Matchers.nonNullLiteral()));
118       }
119     
120       private Scanner nonNullLiteralMatches(
121           final boolean shouldMatch, final Matcher<ExpressionTree> toMatch) {
122         return new Scanner() {
123           @Override
124           public Void visitLiteral(LiteralTree node, VisitorState visitorState) {
125             assertTrue(node.toString(), !shouldMatch ^ toMatch.matches(node, visitorState));
126             return super.visitLiteral(node, visitorState);
127           }
128     
129           @Override
130           public Void visitMemberSelect(MemberSelectTree node, VisitorState visitorState) {
131             assertTrue(node.toString(), !shouldMatch ^ toMatch.matches(node, visitorState));
132             return super.visitMemberSelect(node, visitorState);
133           }
134     
135           @Override
136           public Void visitImport(ImportTree node, VisitorState visitorState) {
137             return null;
138           }
139     
140           @Override
141           public Void visitCompilationUnit(CompilationUnitTree node, VisitorState visitorState) {
142             return null;
143           }
144         };
145       }
146     }
