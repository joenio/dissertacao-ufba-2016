17     package com.google.errorprone;
18     
19     import com.google.common.base.Strings;
20     import com.google.common.base.Throwables;
21     
22     import com.sun.tools.javac.util.DiagnosticSource;
23     import com.sun.tools.javac.util.JCDiagnostic.DiagnosticPosition;
24     import com.sun.tools.javac.util.Log;
25     
26     import javax.tools.JavaFileObject;
27     
28     /**
29      * Wraps an unrecoverable error that occurs during analysis with the source
30      * position that triggered the crash.
31      */
32     public class ErrorProneError extends Error {
33     
34       private final Throwable cause;
35       private final DiagnosticPosition pos;
36       private final JavaFileObject source;
37     
38       public ErrorProneError(Throwable cause, DiagnosticPosition pos, JavaFileObject source) {
39         super(formatMessage(source, pos), cause);
40         this.cause = cause;
41         this.pos = pos;
42         this.source = source;
43       }
44     
45       public void logFatalError(Log log) {
46         String version = ErrorProneCompiler.loadVersionFromPom().or("unknown version");
47         JavaFileObject prev = log.currentSourceFile();
48         try {
49           log.useSource(source);
50           log.error(
51               pos, "error.prone.crash", Throwables.getStackTraceAsString(cause), version);
52         } finally {
53           log.useSource(prev);
54         }
55       }
56     
57       private static String formatMessage(JavaFileObject file, DiagnosticPosition pos) {
58         DiagnosticSource source = new DiagnosticSource(file, /*log=*/ null);
59         int column = source.getColumnNumber(pos.getStartPosition(), /*expandTabs=*/ true);
60         int line = source.getLineNumber(pos.getStartPosition());
61         String snippet = source.getLine(pos.getStartPosition());
62         StringBuilder sb = new StringBuilder();
63         sb.append(
64             String.format(
65                 "\n%s:%d: An unhandled exception was thrown by Error Prone\n",
66                 source.getFile().getName(),
67                 line));
68         sb.append(snippet).append('\n');
69         sb.append(Strings.repeat(" ", column - 1)).append("^\n");
70         return sb.toString();
71       }
72     }
