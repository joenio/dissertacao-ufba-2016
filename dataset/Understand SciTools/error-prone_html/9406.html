58     /**
59      * @author eaftan@google.com (Eddie Aftandilian)

61     @BugPattern(
62       name = "CheckReturnValue",
63       altNames = {"ResultOfMethodCallIgnored", "ReturnValueIgnored"},
64       summary = "Ignored return value of method that is annotated with @CheckReturnValue",
65       category = JDK,
66       severity = ERROR,
67       maturity = MATURE
68     )
69     public class CheckReturnValue extends AbstractReturnValueIgnored
70         implements MethodTreeMatcher, ClassTreeMatcher {
71     
72       private static Optional<Boolean> shouldCheckReturnValue(Symbol sym, VisitorState state) {
73         if (hasAnnotation(sym, CanIgnoreReturnValue.class, state)) {
74           return Optional.of(false);
75         }
76         if (hasAnnotation(sym, javax.annotation.CheckReturnValue.class, state)) {
77           return Optional.of(true);
78         }
79         return Optional.absent();
80       }
81     
82       private static Optional<Boolean> checkEnclosingClasses(MethodSymbol method, VisitorState state) {
83         Symbol enclosingClass = enclosingClass(method);
84         while (enclosingClass instanceof ClassSymbol) {
85           Optional<Boolean> result = shouldCheckReturnValue(enclosingClass, state);
86           if (result.isPresent()) {
87             return result;
88           }
89           enclosingClass = enclosingClass.owner;
90         }
91         return Optional.absent();
92       }
93     
94       private static Optional<Boolean> checkPackage(MethodSymbol method, VisitorState state) {
95         return shouldCheckReturnValue(enclosingPackage(method), state);
96       }
97     
98       private static final Matcher<MethodInvocationTree> MATCHER =
99           new Matcher<MethodInvocationTree>() {
100             @Override
101             public boolean matches(MethodInvocationTree tree, VisitorState state) {
102     
103               MethodSymbol method = ASTHelpers.getSymbol(tree);
104               if (ASTHelpers.isVoidType(method.getReturnType(), state)) {
105                 return false;
106               }
107     
108               if (mockitoInvocation(tree, state)) {
109                 return false;
110               }
111     
112               // Allow unused return values in tests that check for thrown exceptions, e.g.:
113               //
114               // try {
115               //   Foo.newFoo(-1);
116               //   fail();
117               // } catch (IllegalArgumentException expected) {
118               // }
119               //
120               StatementTree statement =
121                   ASTHelpers.findEnclosingNode(state.getPath(), StatementTree.class);
122               if (statement != null && EXPECTED_EXCEPTION_MATCHER.matches(statement, state)) {
123                 return false;
124               }
125     
126               Optional<Boolean> result = shouldCheckReturnValue(method, state);
127               if (result.isPresent()) {
128                 return result.get();
129               }
130     
131               result = checkEnclosingClasses(method, state);
132               if (result.isPresent()) {
133                 return result.get();
134               }
135     
136               result = checkPackage(method, state);
137               if (result.isPresent()) {
138                 return result.get();
139               }
140     
141               return false;
142             }
143           };
144     
145       private static final Matcher<ExpressionTree> MOCKITO_MATCHER =
146           anyOf(
147               staticMethod().onClass("org.mockito.Mockito").named("verify"),
148               instanceMethod().onDescendantOf("org.mockito.stubbing.Stubber").named("when"),
149               instanceMethod().onDescendantOf("org.mockito.InOrder").named("verify"));
150     
151       /**
152        * Return a matcher for method invocations in which the method being called has the
153        * {@code @CheckReturnValue} annotation.
154        */
155       @Override
156       public Matcher<MethodInvocationTree> specializedMatcher() {
157         return MATCHER;
158       }
159     
160       private static final String BOTH_ERROR =
161           "@CheckReturnValue and @CanIgnoreReturnValue cannot both be applied to the same %s";
162     
163       /**
164        * Validate {@link javax.annotation.CheckReturnValue} and {@link CanIgnoreReturnValue} usage on
165        * methods.
166        *
167        * <p>The annotations should not both be appled to the same method.
168        *
169        * <p>The annotations should not be applied to void-returning methods. Doing so makes no sense,
170        * because there is no return value to check.
171        */
172       @Override
173       public Description matchMethod(MethodTree tree, VisitorState state) {
174         MethodSymbol method = ASTHelpers.getSymbol(tree);
175     
176         boolean checkReturn = hasAnnotation(method, javax.annotation.CheckReturnValue.class, state);
177         boolean canIgnore = hasAnnotation(method, CanIgnoreReturnValue.class, state);
178     
179         if (checkReturn && canIgnore) {
180           return buildDescription(tree).setMessage(String.format(BOTH_ERROR, "method")).build();
181         }
182     
183         String annotationToValidate;
184         if (checkReturn) {
185           annotationToValidate = javax.annotation.CheckReturnValue.class.getSimpleName();
186         } else if (canIgnore) {
187           annotationToValidate = CanIgnoreReturnValue.class.getSimpleName();
188         } else {
189           return Description.NO_MATCH;
190         }
191         if (method.getKind() != ElementKind.METHOD) {
192           // skip contructors (which javac thinks are void-returning)
193           return Description.NO_MATCH;
194         }
195         if (!ASTHelpers.isVoidType(method.getReturnType(), state)) {
196           return Description.NO_MATCH;
197         }
198         String message =
199             String.format("@%s may not be applied to void-returning methods", annotationToValidate);
200         return buildDescription(tree).setMessage(message).build();
201       }
202     
203       /**
204        * Validate that at most one of {@link javax.annotation.CheckReturnValue} and
205        * {@link CanIgnoreReturnValue} are applied to a class (or interface or enum).
206        */
207       @Override
208       public Description matchClass(ClassTree tree, VisitorState state) {
209         if (hasAnnotation(tree, javax.annotation.CheckReturnValue.class, state)
210             && hasAnnotation(tree, CanIgnoreReturnValue.class, state)) {
211           return buildDescription(tree).setMessage(String.format(BOTH_ERROR, "class")).build();
212         }
213         return Description.NO_MATCH;
214       }
215     
216       /**
217        * Don't match the method that is invoked through {@code Mockito.verify(t)} or
218        * {@code doReturn(val).when(t)}.
219        */
220       private static boolean mockitoInvocation(MethodInvocationTree tree, VisitorState state) {
221         if (tree instanceof JCMethodInvocation
222             && ((JCMethodInvocation) tree).getMethodSelect() instanceof JCFieldAccess) {
223           ExpressionTree receiver = ASTHelpers.getReceiver(tree);
224           if (MOCKITO_MATCHER.matches(receiver, state)) {
225               return true;
226           }
227         }
228         return false;
229       }
230     
231       static final Matcher<ExpressionTree> FAIL_METHOD =
232           anyOf(
233               instanceMethod().onDescendantOf("com.google.common.truth.AbstractVerb").named("fail"),
234               staticMethod().onClass("org.junit.Assert").named("fail"),
235               staticMethod().onClass("junit.framework.Assert").named("fail"),
236               staticMethod().onClass("junit.framework.TestCase").named("fail"));
237     
238       static final Matcher<StatementTree> EXPECTED_EXCEPTION_MATCHER =
239           allOf(enclosingNode(kindIs(Kind.TRY)), nextStatement(expressionStatement(FAIL_METHOD)));
240     }
