17     package com.google.errorprone.refaster;
18     
19     import static com.google.errorprone.refaster.Unifier.unifications;
20     
21     import com.google.auto.value.AutoValue;
22     
23     import com.sun.source.tree.InstanceOfTree;
24     import com.sun.source.tree.TreeVisitor;
25     import com.sun.tools.javac.tree.JCTree.JCInstanceOf;
26     
27     import javax.annotation.Nullable;
28     
29     /**
30      * {@link UTree} representation of a {@link InstanceOfTree}.
31      * 
32      * @author lowasser@google.com (Louis Wasserman)
33      */
34     @AutoValue
35     abstract class UInstanceOf extends UExpression implements InstanceOfTree {
36       public static UInstanceOf create(UExpression expression, UTree<?> type) {
37         return new AutoValue_UInstanceOf(expression, type);
38       }
39     
40       @Override
41       public abstract UExpression getExpression();
42     
43       @Override
44       public abstract UTree<?> getType();
45     
46       @Override
47       public <R, D> R accept(TreeVisitor<R, D> visitor, D data) {
48         return visitor.visitInstanceOf(this, data);
49       }
50     
51       @Override
52       public Kind getKind() {
53         return Kind.INSTANCE_OF;
54       }
55     
56       @Override
57       public JCInstanceOf inline(Inliner inliner) throws CouldNotResolveImportException {
58         return inliner.maker().TypeTest(getExpression().inline(inliner), getType().inline(inliner));
59       }
60     
61       @Override
62       @Nullable
63       public Choice<Unifier> visitInstanceOf(InstanceOfTree instanceOf, @Nullable Unifier unifier) {
64         return getExpression().unify(instanceOf.getExpression(), unifier)
65             .thenChoose(unifications(getType(), instanceOf.getType()));
66       }
67     }
