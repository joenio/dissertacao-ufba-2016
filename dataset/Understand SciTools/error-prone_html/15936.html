51     /**
52      * @author irogers@google.com (Ian Rogers)
53      */
54     @BugPattern(
55       name = "BadComparable",
56       summary = "Possible sign flip from narrowing conversion",
57       explanation =
58           "A narrowing integral conversion can cause a sign flip, since it simply discards all"
59               + " but the n lowest order bits, where n is the number of bits used to represent"
60               + " the target type (JLS 5.1.3). In a compare or compareTo method, this can cause"
61               + " incorrect and unstable sort orders.",
62       category = JDK,
63       severity = ERROR,
64       maturity = MATURE
65     )
66     public class BadComparable extends BugChecker implements TypeCastTreeMatcher {
67       /** Matcher for the overriding method of 'int java.lang.Comparable.compareTo(T other)' */
68       private static final Matcher<MethodTree> COMPARABLE_METHOD_MATCHER =
69           allOf(
70               methodIsNamed("compareTo"),
71               methodHasVisibility(PUBLIC),
72               methodReturns(INT_TYPE),
73               methodHasArity(1));
74     
75       private static final Matcher<ClassTree> COMPARABLE_CLASS_MATCHER =
76           isSubtypeOf("java.lang.Comparable");
77     
78       /** Matcher for the overriding method of 'int java.util.Comparator.compare(T o1, T o2)' */
79       private static final Matcher<MethodTree> COMPARATOR_METHOD_MATCHER =
80           allOf(
81               methodIsNamed("compare"),
82               methodHasVisibility(PUBLIC),
83               methodReturns(INT_TYPE),
84               methodHasArity(2));
85     
86       private static final Matcher<ClassTree> COMPARATOR_CLASS_MATCHER =
87           isSubtypeOf("java.util.Comparator");
88     
89       /**
90        * Compute the type of the subtract BinaryTree. We use the type of the
91        * left/right operand except when they're not the same, in which case we
92        * prefer the type of the expression. This ensures that a byte/short
93        * subtracted from another byte/short isn't regarded as an int.
94        */
95       private static Type getTypeOfSubtract(BinaryTree expression) {
96         Type expressionType = ASTHelpers.getType(expression.getLeftOperand());
97         if (!expressionType.equals(ASTHelpers.getType(expression.getRightOperand()))) {
98           return ASTHelpers.getType(expression);
99         }
100         return expressionType;
101       }
102     
103       /**
104        * Matches if this is a narrowing integral cast between signed types where the expression is a
105        * subtract.
106        */
107       private boolean matches(TypeCastTree tree, VisitorState state) {
108         Type treeType = ASTHelpers.getType(tree.getType());
109     
110         // If the cast isn't narrowing to an int then don't implicate it in the bug pattern.
111         if (treeType.getTag() != TypeTag.INT) {
112           return false;
113         }
114     
115         // The expression should be a subtract but remove parentheses.
116         ExpressionTree expression = TreeInfo.skipParens((JCExpression) tree.getExpression());
117         if (expression.getKind() != Kind.MINUS) {
118           return false;
119         }
120     
121         // Ensure the expression type is wider and signed (ie a long) than the cast type ignoring
122         // boxing.
123         Type expressionType = getTypeOfSubtract((BinaryTree) expression);
124         TypeTag expressionTypeTag = state.getTypes().unboxedTypeOrType(expressionType).getTag();
125         return (expressionTypeTag == TypeTag.LONG);
126       }
127     
128       @Override
129       public Description matchTypeCast(TypeCastTree tree, VisitorState state) {
130         // Check for a narrowing match first as its simplest match to test.
131         if (!matches(tree, state)) {
132           return Description.NO_MATCH;
133         }
134     
135         // Test that the match is in a Comparable.compareTo or Comparator.compare method.
136         ClassTree declaringClass = ASTHelpers.findEnclosingNode(state.getPath(), ClassTree.class);
137         if (!COMPARABLE_CLASS_MATCHER.matches(declaringClass, state)
138             && !COMPARATOR_CLASS_MATCHER.matches(declaringClass, state)) {
139           return Description.NO_MATCH;
140         }
141         MethodTree method = ASTHelpers.findEnclosingNode(state.getPath(), MethodTree.class);
142         if (method == null) {
143           return Description.NO_MATCH;
144         }
145         if (!COMPARABLE_METHOD_MATCHER.matches(method, state)
146             && !COMPARATOR_METHOD_MATCHER.matches(method, state)) {
147           return Description.NO_MATCH;
148         }
149     
150         // Get the unparenthesized expression.
151         BinaryTree subtract = (BinaryTree) TreeInfo.skipParens((JCExpression) tree.getExpression());
152         ExpressionTree lhs = subtract.getLeftOperand();
153         ExpressionTree rhs = subtract.getRightOperand();
154         Fix fix;
155         if (ASTHelpers.getType(lhs).isPrimitive()) {
156           fix = SuggestedFix.replace(tree, "Long.compare(" + lhs + ", " + rhs + ")");
157         } else {
158           fix = SuggestedFix.replace(tree, lhs + ".compareTo(" + rhs + ")");
159         }
160         return describeMatch(tree, fix);
161       }
162     }
