17     package com.google.errorprone.bugpatterns;
18     
19     import static com.google.errorprone.BugPattern.Category.INJECT;
20     import static com.google.errorprone.BugPattern.MaturityLevel.EXPERIMENTAL;
21     import static com.google.errorprone.BugPattern.SeverityLevel.ERROR;
22     import static com.google.errorprone.bugpatterns.BugChecker.AnnotationTreeMatcher;
23     import static com.google.errorprone.matchers.Matchers.hasAnnotation;
24     
25     import com.google.errorprone.BugPattern;
26     import com.google.errorprone.VisitorState;
27     import com.google.errorprone.fixes.SuggestedFix;
28     import com.google.errorprone.matchers.Description;
29     import com.google.errorprone.matchers.Matcher;
30     import com.google.errorprone.matchers.Matchers;
31     
32     import com.sun.source.tree.AnnotationTree;
33     import com.sun.source.tree.ClassTree;
34     import com.sun.source.tree.ModifiersTree;
35     
36     /**
37      * This checker matches if a class has more than one annotation that is a scope
38      * annotation(that is, the annotation is either annotated with Guice's
39      * {@code @ScopeAnnotation} or Javax's {@code @Scope}).
40      *
41      * @author sgoldfeder@google.com (Steven Goldfeder)
42      */
43     
44     @BugPattern(name = "InjectMoreThanOneScopeAnnotationOnClass",
45         summary = "A class can be annotated with at most one scope annotation",
46         explanation = "Annotating a class with more than one scope annotation is "
47             + "invalid according to the JSR-330 specification. ", category = INJECT, severity = ERROR,
48         maturity = EXPERIMENTAL)
49     public class InjectMoreThanOneScopeAnnotationOnClass extends BugChecker
50         implements AnnotationTreeMatcher {
51     
52       private static final String GUICE_SCOPE_ANNOTATION = "com.google.inject.ScopeAnnotation";
53       private static final String JAVAX_SCOPE_ANNOTATION = "javax.inject.Scope";
54     
55       /**
56        * Matches annotations that are themselves annotated with with
57        * {@code @ScopeAnnotation(Guice)} or {@code @Scope(Javax)}.
58        */
59       private final Matcher<AnnotationTree> scopeAnnotationMatcher = Matchers.<AnnotationTree>anyOf(
60           hasAnnotation(GUICE_SCOPE_ANNOTATION), hasAnnotation(JAVAX_SCOPE_ANNOTATION));
61     
62       @Override
63       public final Description matchAnnotation(AnnotationTree annotationTree, VisitorState state) {
64         int numberOfScopeAnnotations = 0;
65         // check if this annotation is on a class and is a scope annotation
66         if (scopeAnnotationMatcher.matches(annotationTree, state)
67             && state.getPath().getParentPath().getParentPath().getLeaf() instanceof ClassTree) {
68           for (AnnotationTree annotation :
69               ((ModifiersTree) state.getPath().getParentPath().getLeaf()).getAnnotations()) {
70             if (scopeAnnotationMatcher.matches(annotation, state)) {
71               numberOfScopeAnnotations++;
72             }
73           }
74         }
75         if (numberOfScopeAnnotations > 1) {
76           return describeMatch(
77             annotationTree, SuggestedFix.delete(annotationTree));
78         }
79         return Description.NO_MATCH;
80       }
81     }
