17     package com.google.errorprone.annotations;
18     import static java.lang.annotation.RetentionPolicy.CLASS;
19     
20     import java.lang.annotation.Documented;
21     import java.lang.annotation.ElementType;
22     import java.lang.annotation.Retention;
23     import java.lang.annotation.Target;
24     
25     /**
26      * Annotation for method parameter declarations, which denotes that corresponding actual
27      * parameters must be compile-time constant expressions.
28      *
29      * <p>When the formal parameter of a method or constructor is annotated with the
30      *{@link CompileTimeConstant} type annotation, the corresponding actual parameter must be
31      * an expression that satisfies one of the following conditions:
32      * <ol>
33      * <li>The expression is one for which the Java compiler can determine a constant value at compile
34      * time, or
35      * <li>the expression consists of the literal {@code null}, or
36      * <li>the expression consists of a single identifier, where the identifier is a formal method
37      * parameter that is declared {@code final} and has the {@link CompileTimeConstant} annotation.
38      * </ol>
39      *
40      * <p>This constraint on call sites of methods or constructors that have one or more formal
41      * parameters with this annotation is enforced by  <a
42      * href="http://errorprone.info">error-prone</a>.
43      *
44      * <p>For example, the following code snippet is legal:
45      * <pre>
46      * {@code
47      * public class C {
48      *   private static final S = "Hello";
49      *   void m(@CompileTimeConstant final String s) { }
50      *   void n(@CompileTimeConstant final String t) {
51      *     m(S + " World!");
52      *     m(null);
53      *     m(t);
54      *   }
55      * }
56      * }</pre>
57      *
58      *
59      * <p>In contrast, the following is illegal:
60      * <pre>
61      * {@code
62      * public class C {
63      *   void m(@CompileTimeConstant final String s) { }
64      *   void n(String t) {
65      *     m(t);
66      *   }
67      * }
68      * }</pre>
69      *
70      * <p>Compile-time constant values are implicitly under the control of the trust domain of the
71      * application whose source code they are part of. Hence, this annotation is useful to constrain the
72      * use of APIs that may only be safely called with values that are under application control.
73      *
74      * <p>The current implementation of the @CompileTimeConstant checker cannot reason about more
75      * complex scenarios, for example, returning compile-time-constant values from a method, or storing
76      * compile-time-constant values in a collection.  APIs will typically accommodate such use cases
77      * via domain-specific types that capture domain-specific aspects of trustworthiness that arise from
78      */
79     @Documented
80     @Retention(CLASS)
81     @Target({ElementType.PARAMETER})
82     public @interface CompileTimeConstant {
83     }
