77       @Test
78       public void testFailImport() throws Exception {
79         TestScanner scanner = new TestScanner();
80         CompilationTestHelper compilationHelper =
81             CompilationTestHelper.newInstance(ScannerSupplier.fromScanner(scanner), getClass());
82     
83         compilationHelper
84             .addSourceLines(
85                 "test/A.java",
86                 "package test;",
87                 "import junit.framework.TestCase;",
88                 "public class A extends TestCase {",
89                 "  public void testMethod() {",
90                 "    try {",
91                 "      new String();",
92                 "    } catch (IllegalArgumentException expected) {}",
93                 "  }",
94                 "}")
95             .doTest();
96     
97         assertThat(getOnlyFix(scanner).getImportsToAdd())
98             .containsExactly("import static org.junit.Assert.fail");
99         assertThat(getOnlyFix(scanner).getImportsToRemove())
100             .containsExactly(
101                 "import static junit.framework.TestCase.fail",
102                 "import static junit.framework.Assert.fail");
103       }

105       @Test
106       public void testFailMessageMultiCatch() throws Exception {
107         TestScanner scanner = new TestScanner();
108         CompilationTestHelper compilationHelper =
109             CompilationTestHelper.newInstance(ScannerSupplier.fromScanner(scanner), getClass());
110     
111         compilationHelper
112             .addSourceLines(
113                 "test/A.java",
114                 "package test;",
115                 "import junit.framework.TestCase;",
116                 "public class A extends TestCase {",
117                 "  public void testMethod() {",
118                 "    try {",
119                 "      new String();",
120                 "    } catch (IllegalArgumentException | IllegalStateException expected) {}",
121                 "  }",
122                 "}")
123             .doTest();
124     
125         assertThat(getOnlyFix(scanner).getReplacements(new NoopEndPosTable()))
126             .containsExactly(Replacement.create(0, 0, "\nfail(\"Expected Exception\");"));
127       }

146     
147       private Fix getOnlyFix(TestScanner scanner) {
148         Description warning = Iterables.getOnlyElement(scanner.suggestedChanges);
149         return Iterables.getOnlyElement(warning.fixes);
150       }
151     
152       private static class TestScanner extends Scanner {
153     
154         final List<Description> suggestedChanges = new LinkedList<>();
155     
156         @Override
157         public Void visitTry(TryTree node, VisitorState visitorState) {
158           suggestedChanges.add(
159               new MissingFail().matchTry(node, visitorState.withPath(getCurrentPath())));
160           return super.visitTry(node, visitorState);
161         }
162       }
163     
164       private static class NoopEndPosTable implements EndPosTable {
165     
166         @Override
167         public int getEndPos(JCTree tree) {
168           return 0;
169         }
170     
171         @Override
172         public void storeEnd(JCTree tree, int endpos) {}
173     
174         @Override
175         public int replaceTree(JCTree oldtree, JCTree newtree) {
176           return 0;
177         }
178       }
