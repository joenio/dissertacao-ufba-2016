47       public void expectMatches(
48           final Template<?> template, Match... expected) {
49         final Set<Match> expectedMatches = Sets.newHashSet(expected);
50         TreeScanner matchScanner = new TreeScanner() {
51           @Override
52           public void scan(JCTree tree) {
53             if (tree == null) {
54               return;
55             }
56             for (TemplateMatch templateMatch : template.match(tree, context)) {
57               Match match = Match.create(templateMatch);
58               if (!expectedMatches.remove(match)) {
59                 fail(String.format("Unexpected match against template %s:%n%s", template, match));
60               }
61             }
62             super.scan(tree);
63           }
64         };
65         for (JCCompilationUnit unit : compilationUnits) {
66           matchScanner.scan(unit);
67         }
68         for (Match missingMatch : expectedMatches) {
69           fail(String.format(
70               "Expected match against template %s not found: %s", template, missingMatch));
71         }
72       }

74       @Test
75       public void binary() {
76         // template: (a + b) / 2
77         ExpressionTemplate template = ExpressionTemplate.create(
78             ImmutableMap.of(
79                 "a", UPrimitiveType.INT, 
80                 "b", UPrimitiveType.INT),
81             UBinary.create(Kind.DIVIDE, 
82                 UParens.create(UBinary.create(
83                     Kind.PLUS, UFreeIdent.create("a"), UFreeIdent.create("b"))),
84                 ULiteral.intLit(2)),
85             UPrimitiveType.INT);
86         
87         compile(
88             "import java.util.Random;",
89             "class BinaryExample {",
90             "  public void example(int x, int y) {",
91             // positive examples
92             "    System.out.println((3 + 5) / 2);",
93             "    System.out.println((x + y) / 2 + 20);",
94             "    System.err.println((y + new Random().nextInt()) / 2);",
95             // negative examples
96             "    System.out.println((x - y) / 2);",
97             "    System.out.println((x * y) / 2);",
98             "    System.out.println((x + y) / 3);",
99             "    System.out.println((x + 5L) / 2);",
100             "  }",
101             "}");
102         expectMatches(template, 
103             Match.create(ImmutableMap.of("a", "3", "b", "5")),
104             Match.create(ImmutableMap.of("a", "x", "b", "y")),
105             Match.create(ImmutableMap.of("a", "y", "b", "new Random().nextInt()")));
106       }

132       public void methodInvocation() {
133         // template : md.digest(str.getBytes())
134         UType byteArrayType = UArrayType.create(UPrimitiveType.BYTE);
135         ExpressionTemplate template = ExpressionTemplate.create(
136             ImmutableMap.of(
137                 "md", UClassType.create("java.security.MessageDigest"),
138                 "str", UClassType.create("java.lang.String")),
139             UMethodInvocation.create(
140                 UMemberSelect.create(UFreeIdent.create("md"), "digest",
141                     UMethodType.create(byteArrayType, byteArrayType)), 
142                 UMethodInvocation.create(
143                     UMemberSelect.create(UFreeIdent.create("str"), "getBytes", 
144                         UMethodType.create(byteArrayType)))),
145             byteArrayType);
146         
147         compile(
148             "import java.security.MessageDigest;",
149             "import java.security.NoSuchAlgorithmException;",
150             "import java.nio.charset.Charset;",
151             "class MethodInvocationExample {",
152             "  public void example(MessageDigest digest, String string)",
153             "      throws NoSuchAlgorithmException {",
154             // positive examples
155             "    MessageDigest.getInstance(\"MD5\").digest(\"foo\".getBytes());",
156             "    digest.digest(\"foo\".getBytes());",
157             "    MessageDigest.getInstance(\"SHA1\").digest(string.getBytes());",
158             "    digest.digest((string + 90).getBytes());",
159             // negative examples
160             "    System.out.println(\"foo\".getBytes());",
161             "  }",
162             "}");
163         
164         expectMatches(template, 
165             Match.create(ImmutableMap.of("md", "MessageDigest.getInstance(\"MD5\")", "str", "\"foo\"")),
166             Match.create(ImmutableMap.of("md", "digest", "str", "\"foo\"")),
167             Match.create(ImmutableMap.of("md", "MessageDigest.getInstance(\"SHA1\")", "str", "string")),
168             Match.create(ImmutableMap.of("md", "digest", "str", "(string + 90)")));
169       }

228       @Test
229       public void returnTypeMatters() {
230         /* Template:
231          * <E> List<E> template(List<E> list) {
232          *   return Collections.unmodifiableList(list);
233          * }
234          * 
235          * Note that Collections.unmodifiableList takes a List<? extends T>,
236          * but we're restricting to the case where the return type is the same.
237          */
238         UTypeVar tVar = UTypeVar.create("T");
239         UTypeVar eVar = UTypeVar.create("E");
240         ExpressionTemplate template = ExpressionTemplate.create(
241             ImmutableClassToInstanceMap.<Annotation>builder().build(),
242             ImmutableList.of(eVar),
243             ImmutableMap.of("list", UClassType.create("java.util.List", eVar)),
244             UMethodInvocation.create(
245                 UStaticIdent.create("java.util.Collections", "unmodifiableList", 
246                     UForAll.create(ImmutableList.of(tVar),
247                         UMethodType.create(
248                             UClassType.create("java.util.List", tVar), 
249                             UClassType.create("java.util.List", 
250                                 UWildcardType.create(BoundKind.EXTENDS, tVar))))), 
251                 UFreeIdent.create("list")),
252             UClassType.create("java.util.List", eVar));
253         compile(
254             "import java.util.ArrayList;",
255             "import java.util.Collections;",
256             "import java.util.List;",
257             "class ReturnTypeMattersExample {",
258             "  public void example() {",
259             // positive examples
260             "    Collections.unmodifiableList(new ArrayList<String>());",
261             "    List<Integer> ints = Collections.unmodifiableList(Collections.singletonList(1));",
262             // negative examples
263             "    List<CharSequence> seqs = Collections.<CharSequence>unmodifiableList(",
264             "        new ArrayList<String>());",
265             "  }",
266             "}");
267         expectMatches(template,
268             Match.create(ImmutableMap.of(
269                 "list", "new ArrayList<String>()",
270                 "E", "java.lang.String")),
271             Match.create(ImmutableMap.of(
272                 "list", "Collections.singletonList(1)",
273                 "E", "java.lang.Integer")));
274       }

307       @Test
308       public void blockTemplate() {
309         BlockTemplate blockTemplate = BlockTemplate.create(
310             // <E>
311             ImmutableList.of(UTypeVar.create("E")),
312             ImmutableMap.of(
313                 // Collection<E> collection
314                 "collection", UClassType.create("java.util.Collection", UTypeVar.create("E")),
315                 // Comparator<? super E> comparator
316                 "comparator", UClassType.create("java.util.Comparator", 
317                     UWildcardType.create(BoundKind.SUPER, UTypeVar.create("E")))),
318             UVariableDecl.create(
319                 "list", 
320                 UTypeApply.create("java.util.List", UTypeVarIdent.create("E")), 
321                 UNewClass.create(UTypeApply.create("java.util.ArrayList", UTypeVarIdent.create("E")),
322                     UFreeIdent.create("collection"))),
323             UExpressionStatement.create(
324                 UMethodInvocation.create(
325                     UStaticIdent.create("java.util.Collections", "sort", 
326                         UForAll.create(ImmutableList.of(UTypeVar.create("T")), 
327                             UMethodType.create(UPrimitiveType.VOID, 
328                                 UClassType.create("java.util.List", UTypeVar.create("T")), 
329                                 UClassType.create("java.util.Comparator", 
330                                     UWildcardType.create(BoundKind.SUPER, UTypeVar.create("T")))))), 
331                     ULocalVarIdent.create("list"), 
332                     UFreeIdent.create("comparator"))));
333         compile(
334             "import java.util.ArrayList;",
335             "import java.util.Comparator;",
336             "import java.util.Collections;",
337             "import java.util.List;",
338             "class BlockTemplateExample {",
339             "  public void example(Comparator<String> cmp) {",
340             "    List<String> foo = new ArrayList<String>();",
341             "    foo.add(\"bar\");",
342             "    List<String> sorted = new ArrayList<String>(foo);",
343             "    Collections.sort(sorted, cmp);",
344             "  }",
345             "}");
346         expectMatches(blockTemplate,
347             Match.create(ImmutableMap.of(
348                 "collection", "foo", 
349                 "comparator", "cmp",
350                 "E", "java.lang.String",
351                 "list", "sorted")));
352       }

354       @Test
355       public void ifBlockTemplate() {
356         /*
357          * Template:
358          * 
359          * if (cond) {
360          *   x = y;
361          * } else {
362          *   x = z;
363          * }
364          */
365         BlockTemplate blockTemplate = BlockTemplate.create(
366             ImmutableList.of(UTypeVar.create("T")),
367             ImmutableMap.of(
368                 "cond", UPrimitiveType.BOOLEAN,
369                 "x", UTypeVar.create("T"),
370                 "y", UTypeVar.create("T"),
371                 "z", UTypeVar.create("T")),
372             UIf.create(UFreeIdent.create("cond"), 
373                 UBlock.create(UExpressionStatement.create(
374                     UAssign.create(UFreeIdent.create("x"), UFreeIdent.create("y")))),
375                 UBlock.create(UExpressionStatement.create(
376                     UAssign.create(UFreeIdent.create("x"), UFreeIdent.create("z"))))));
377         
378         compile(
379             "class IfBlockExample {",
380             "  public void example(String x) {",
381             "    if (Math.random() > 0.5) {",
382             "      x = \"foo\";",
383             "    } else {",
384             "      x = \"bar\";",
385             "    }",
386             "  }",
387             "}");
388         expectMatches(blockTemplate,
389             Match.create(ImmutableMap.of(
390                 "cond", "(Math.random() > 0.5)",
391                 "x", "x",
392                 "y", "\"foo\"", 
393                 "z", "\"bar\"",
394                 "T", "java.lang.String")));
395       }
