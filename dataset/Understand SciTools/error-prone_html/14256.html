54     /**
55      * Checks, if two constructors in a class both accept {@code Foo foo} and one calls the other, that
56      * the caller passes {@code foo} as a parameter. The goal is to catch copy-paste errors:<pre>
57      *   MissileLauncher(Location target, boolean askForConfirmation) {
58      *     ...
59      *   }
60      *   MissileLauncher(Location target) {
61      *     this(target, false);
62      *   }
63      *   MissileLauncher(boolean askForConfirmation) {
64      *     this(TEST_TARGET, <b>false</b>); // should be askForConfirmation
65      *   }</pre>
66      *
67      * @author cpovirk@google.com (Chris Povirk)
68      */
69     @BugPattern(name = "ChainingConstructorIgnoresParameter",
70         maturity = MATURE, category = JDK, severity = ERROR,
71         explanation = "A constructor parameter might not be being used as expected",
72         summary = "The called constructor accepts a parameter with the same name and type as one of "
73             + "its caller's parameters, but its caller doesn't pass that parameter to it.  It's likely "
74             + "that it was intended to.")
75     public final class ChainingConstructorIgnoresParameter extends BugChecker
76         implements CompilationUnitTreeMatcher, MethodInvocationTreeMatcher, MethodTreeMatcher {
77       private final Map<MethodSymbol, List<VariableTree>> paramTypesForMethod = newHashMap();
78       private final Multimap<MethodSymbol, Caller> callersToEvaluate = ArrayListMultimap.create();
79     
80       @Override
81       public Description matchCompilationUnit(CompilationUnitTree tree, VisitorState state) {
82         /*
83          * Clear the collections to save memory. (I wonder if it also helps to handle weird cases when a
84          * class has multiple definitions. But I would expect for multiple definitions within the same
85          * compiler invocation to cause deeper problems.)
86          */
87         paramTypesForMethod.clear();
88         callersToEvaluate.clear(); // should have already been cleared
89         return NO_MATCH;
90       }
91     
92       @Override
93       public Description matchMethodInvocation(MethodInvocationTree tree, VisitorState state) {
94         MethodSymbol symbol = getSymbol(tree);
95         // TODO(cpovirk): determine whether anyone might be calling Foo.this()
96         if (!isIdentifierWithName(tree.getMethodSelect(), "this")) {
97           return NO_MATCH;
98         }
99         callersToEvaluate.put(symbol, new Caller(tree, state));
100         return evaluateCallers(symbol);
101       }
102     
103       @Override
104       public Description matchMethod(MethodTree tree, VisitorState state) {
105         MethodSymbol symbol = getSymbol(tree);
106         if (!symbol.isConstructor()) {
107           return NO_MATCH;
108         }
109         paramTypesForMethod.put(symbol, unmodifiableList(tree.getParameters()));
110         return evaluateCallers(symbol);
111       }
112     
113       private Description evaluateCallers(MethodSymbol symbol) {
114         List<VariableTree> paramTypes = paramTypesForMethod.get(symbol);
115         if (paramTypes == null) {
116           // We haven't seen the declaration yet. We'll evaluate the call when we do.
117           return NO_MATCH;
118         }
119     
120         for (Caller caller : callersToEvaluate.removeAll(symbol)) {
121           VisitorState state = caller.state;
122           MethodInvocationTree invocation = caller.tree;
123     
124           MethodTree callerConstructor = state.findEnclosing(MethodTree.class);
125           if (callerConstructor == null) {
126             continue; // impossible, at least in compilable code?
127           }
128           Map<String, Type> availableParams = indexTypeByName(callerConstructor.getParameters());
129     
130           /*
131            * TODO(cpovirk): Better handling of varargs: If the last parameter type is varargs and it is
132            * called as varargs (rather than by passing an array), then rewrite the parameter types to
133            * (p0, p1, ..., p[n-2], p[n-1] = element type of varargs parameter if an argument is
134            * supplied, p[n] = ditto, etc.). For now, we settle for not crashing in the face of a
135            * mismatch between the number of parameters declared and the number supplied.
136            *
137            * (Use MethodSymbol.isVarArgs.)
138            */
139           for (int i = 0; i < paramTypes.size() && i < invocation.getArguments().size(); i++) {
140             VariableTree formalParam = paramTypes.get(i);
141             String formalParamName = formalParam.getName().toString();
142             Type formalParamType = getType(formalParam.getType());
143     
144             Type availableParamType = availableParams.get(formalParamName);
145     
146             ExpressionTree actualParam = invocation.getArguments().get(i);
147     
148             if (
149                 /*
150                  * The caller has no param of this type. (Or if it did, we couldn't determine the type.
151                  * Does that ever happen?) If the param doesn't exist, the caller can't be failing to
152                  * pass it.
153                  */
154                 availableParamType == null
155     
156                 /*
157                  * We couldn't determine the type of the formal parameter. (Does this ever happen?)
158                  */
159                 || formalParamType == null
160     
161                 /*
162                  * The caller is passing the expected parameter (or "ImmutableList.copyOf(parameter),"
163                  * "new File(parameter)," etc.).
164                  */
165                 || referencesIdentifierWithName(formalParamName, actualParam, state)) {
166               continue;
167             }
168     
169             if (state.getTypes().isAssignable(availableParamType, formalParamType)) {
170               reportMatch(invocation, state, actualParam, formalParamName);
171             }
172             /*
173              * If formal parameter is of an incompatible type, the caller might in theory still intend
174              * to pass a dervied expression. For example, "Foo(String file)" might intend to call
175              * "Foo(File file)" by passing "new File(file)." If this comes up in practice, we could
176              * provide the dummy suggested fix "someExpression(formalParamName)." However, my research
177              * suggests that this will rarely if ever be what the user wants.
178              */
179           }
180         }
181     
182         // All matches are reported through reportMatch calls instead of return values.
183         return NO_MATCH;
184       }
185     
186       private static Map<String, Type> indexTypeByName(List<? extends VariableTree> parameters) {
187         Map<String, Type> result = newHashMap();
188         for (VariableTree parameter : parameters) {
189           result.put(parameter.getName().toString(), getType(parameter.getType()));
190         }
191         return result;
192       }
193     
194       private void reportMatch(
195           Tree diagnosticPosition, VisitorState state, Tree toReplace, String replaceWith) {
196         state.reportMatch(describeMatch(diagnosticPosition, replace(toReplace, replaceWith)));
197       }
198     
199       private static boolean referencesIdentifierWithName(
200           final String name, ExpressionTree tree, VisitorState state) {
201         Matcher<IdentifierTree> identifierMatcher = new Matcher<IdentifierTree>() {
202           @Override
203           public boolean matches(IdentifierTree tree, VisitorState state) {
204             return isIdentifierWithName(tree, name);
205           }
206         };
207         return hasIdentifier(identifierMatcher).matches(tree, state);
208       }
209     
210       private static boolean isIdentifierWithName(ExpressionTree tree, String name) {
211         return tree.getKind() == IDENTIFIER && ((IdentifierTree) tree).getName().contentEquals(name);
212       }
213     
214       private static final class Caller {
215         final MethodInvocationTree tree;
216         final VisitorState state;
217     
218         Caller(MethodInvocationTree tree, VisitorState state) {
219           this.tree = tree;
220           this.state = state;
221         }
222       }
223     }
