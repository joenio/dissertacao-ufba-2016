86       /**
87        * Determines whether two expressions refer to the same variable. Note that returning false
88        * doesn't necessarily mean the expressions do *not* refer to the same field. We don't attempt
89        * to do any complex analysis here, just catch the obvious cases.
90        */
91       public static boolean sameVariable(ExpressionTree expr1, ExpressionTree expr2) {
92         // Throw up our hands if we're not comparing identifiers and/or field accesses.
93         if ((expr1.getKind() != Kind.IDENTIFIER && expr1.getKind() != Kind.MEMBER_SELECT)
94             || (expr2.getKind() != Kind.IDENTIFIER && expr2.getKind() != Kind.MEMBER_SELECT)) {
95           return false;
96         }
97     
98         Symbol sym1 = getSymbol(expr1);
99         Symbol sym2 = getSymbol(expr2);
100         if (sym1 == null) {
101           throw new IllegalStateException("Couldn't get symbol for " + expr1);
102         } else if (sym2 == null) {
103           throw new IllegalStateException("Couldn't get symbol for " + expr2);
104         }
105     
106         if (expr1.getKind() == Kind.IDENTIFIER && expr2.getKind() == Kind.IDENTIFIER) {
107           // foo == foo?
108           return sym1.equals(sym2);
109         } else if (expr1.getKind() == Kind.MEMBER_SELECT && expr2.getKind() == Kind.MEMBER_SELECT) {
110           // foo.baz.bar == foo.baz.bar?
111           return sym1.equals(sym2)
112               && sameVariable(((JCFieldAccess) expr1).selected, ((JCFieldAccess) expr2).selected);
113         } else {
114           // this.foo == foo?
115           ExpressionTree selected = null;
116           if (expr1.getKind() == Kind.IDENTIFIER) {
117             selected = ((JCFieldAccess) expr2).selected;
118           } else {
119             selected = ((JCFieldAccess) expr1).selected;
120           }
121           // TODO(eaftan): really shouldn't be relying on .toString()
122           return selected.toString().equals("this") && sym1.equals(sym2);
123         }
124       }

128        * of the wrong type, or if {@code tree} is null.

131       public static Symbol getSymbol(Tree tree) {
132         if (tree instanceof ClassTree) {
133           return getSymbol((ClassTree) tree);
134         }
135         if (tree instanceof MethodTree) {
136           return getSymbol((MethodTree) tree);
137         }
138         if (tree instanceof VariableTree) {
139           return getSymbol((VariableTree) tree);
140         }
141         if (tree instanceof JCFieldAccess) {
142           return ((JCFieldAccess) tree).sym;
143         }
144         if (tree instanceof JCIdent) {
145           return ((JCIdent) tree).sym;
146         }
147         if (tree instanceof JCMethodInvocation) {
148           return ASTHelpers.getSymbol((MethodInvocationTree) tree);
149         }
150         if (tree instanceof JCNewClass) {
151           return ((JCNewClass) tree).constructor;
152         }
153         if (tree instanceof AnnotationTree) {
154           return getSymbol(((AnnotationTree) tree).getAnnotationType());
155         }
156         if (tree instanceof PackageTree) {
157           return getSymbol((PackageTree) tree);
158         }
159         return null;
160       }

193        * Given a TreePath, walks up the tree until it finds a node of the given type and returns
194        * the path from that node to the top-level node in the path (typically a
195        * {@code CompilationUnitTree}).
196        */
197       public static <T> TreePath findPathFromEnclosingNodeToTopLevel(TreePath path, Class<T> klass) {
198         while (path != null && !(klass.isInstance(path.getLeaf()))) {
199           path = path.getParentPath();
200         }
201         return path;

214       /**
215        * Find the root assignable expression of a chain of field accesses.  If there is no root
216        * (i.e, a bare method call or a static method call), return null.
217        *
218        * <p>Examples:
219        * <pre>
220        * {@code
221        *    a.trim().intern() ==> a
222        *    a.b.trim().intern() ==> a.b
223        *    this.intValue.foo() ==> this.intValue
224        *    this.foo() ==> this
225        *    intern() ==> null
226        *    String.format() ==> null
227        *    java.lang.String.format() ==> null
228        * }
229        * </pre>
230        */
231       public static ExpressionTree getRootAssignable(MethodInvocationTree methodInvocationTree) {
232         if (!(methodInvocationTree instanceof JCMethodInvocation)) {
233           throw new IllegalArgumentException(
234               "Expected type to be JCMethodInvocation, but was " + methodInvocationTree.getClass());
235         }
236     
237         // Check for bare method call, e.g. intern().
238         if (((JCMethodInvocation) methodInvocationTree).getMethodSelect() instanceof JCIdent) {
239           return null;
240         }
241     
242         // Unwrap the field accesses until you get to an identifier.
243         ExpressionTree expr = methodInvocationTree;
244         while (expr instanceof JCMethodInvocation) {
245           expr = ((JCMethodInvocation) expr).getMethodSelect();
246           if (expr instanceof JCFieldAccess) {
247             expr = ((JCFieldAccess) expr).getExpression();
248           }
249         }
250     
251         // We only want assignable identifiers.
252         Symbol sym = getSymbol(expr);
253         if (sym instanceof VarSymbol) {
254           return expr;
255         }
256         return null;
257       }

330       /**
331        * Given a BinaryTree to match against and a list of two matchers, applies the matchers to the
332        * operands in both orders.  If both matchers match, returns a list with the operand that
333        * matched each matcher in the corresponding position.
334        *
335        * @param tree a BinaryTree AST node
336        * @param matchers a list of matchers
337        * @param state the VisitorState
338        * @return a list of matched operands, or null if at least one did not match
339        */
340       public static List<ExpressionTree> matchBinaryTree(
341           BinaryTree tree, List<Matcher<ExpressionTree>> matchers, VisitorState state) {
342         ExpressionTree leftOperand = tree.getLeftOperand();
343         ExpressionTree rightOperand = tree.getRightOperand();
344         if (matchers.get(0).matches(leftOperand, state)
345             && matchers.get(1).matches(rightOperand, state)) {
346           return Arrays.asList(leftOperand, rightOperand);
347         } else if (matchers.get(0).matches(rightOperand, state)
348             && matchers.get(1).matches(leftOperand, state)) {
349           return Arrays.asList(rightOperand, leftOperand);
350         }
351         return null;
352       }

359       /**
360        * Hacky fix for poor javac 6 literal parsing.  javac 6 doesn't set the AST node start
361        * position correctly when a numeric literal is preceded by -. So we scan the source
362        * backwards starting at the provided start position, looking for whitespace, until we find
363        * the true start position.  javac 7 gets this right.
364        *
365        * @return The actual start position of the literal. May be the same as the start position
366        * given by the tree node itself.
367        */
368       public static int getActualStartPosition(JCLiteral tree, CharSequence source) {
369         // This only applies to negative numeric literals.
370         Object value = tree.getValue();
371         if ((value instanceof Number) && (((Number) value).doubleValue() < 0)) {
372           int start = tree.getStartPosition() - 1;
373           while (WHITESPACE_CHARS.matches(source.charAt(start))) {
374             start--;
375           }
376           if (source.charAt(start) == '-') {
377             return start;
378           }
379         }
380         return tree.getStartPosition();
381       }

383       public static Set<MethodSymbol> findSuperMethods(MethodSymbol methodSymbol, Types types) {
384         Set<MethodSymbol> supers = new HashSet<>();
385         if (methodSymbol.isStatic()) {
386           return supers;
387         }
388     
389         TypeSymbol owner = (TypeSymbol) methodSymbol.owner;
390         // Iterates over an ordered list of all super classes and interfaces.
391         for (Type sup : types.closure(owner.type)) {
392           if (sup == owner.type) {
393             continue; // Skip the owner of the method
394           }
395           Scope scope = sup.tsym.members();
396           for (Symbol sym : scope.getSymbolsByName(methodSymbol.name)) {
397             if (sym != null
398                 && !sym.isStatic()
399                 && ((sym.flags() & Flags.SYNTHETIC) == 0)
400                 && sym.name.contentEquals(methodSymbol.name)
401                 && methodSymbol.overrides(sym, owner, types, true)) {
402               supers.add((MethodSymbol) sym);
403             }
404           }
405         }
406         return supers;
407       }

409       /**
410        * Finds all methods in any superclass of {@code startClass} with a certain {@code name} that
411        * match the given {@code predicate}.
412        *
413        * @return The (possibly empty) set of methods in any superclass that match {@code predicate} and
414        * have the given {@code name}.
415        */
416       public static Set<MethodSymbol> findMatchingMethods(
417           Name name, final Predicate<MethodSymbol> predicate, Type startClass, Types types) {
418         final Filter<Symbol> filter = new Filter<Symbol>() {
419           @Override
420           public boolean accepts(Symbol symbol) {
421             if (!(symbol instanceof MethodSymbol)) {
422               return false;
423             }
424             return predicate.apply((MethodSymbol) symbol);
425           }
426         };
427         Set<MethodSymbol> matchingMethods = new HashSet<>();
428         // Iterate over all classes and interfaces that startClass inherits from.
429         for (Type superClass : types.closure(startClass)) {
430           TypeSymbol superClassSymbol = superClass.tsym;
431           // Iterate over all the methods declared in superClass.
432           for (Symbol symbol :
433               superClassSymbol.members().getSymbolsByName(name, filter, NON_RECURSIVE)) {
434             // By definition of the filter, we know that the symbol is a MethodSymbol.
435             matchingMethods.add((MethodSymbol) symbol);
436           }
437         }
438         return matchingMethods;
439       }

442        * Find a method in the enclosing class's superclass that this method overrides.
443        *
444        * @return A superclass method that is overridden by {@code method}
445        */
446       public static MethodSymbol findSuperMethod(MethodSymbol method, Types types) {
447         TypeSymbol superClass = method.enclClass().getSuperclass().tsym;
448         if (superClass == null) {
449           return null;
450         }
451         for (Symbol sym : superClass.members().getSymbols()) {
452           if (sym.name.contentEquals(method.name) && method.overrides(sym, superClass, types, true)) {
453             return (MethodSymbol) sym;
454           }
455         }
456         return null;
457       }

459       /**
460        * Determines whether a symbol has an annotation of the given type.
461        * This includes annotations inherited from superclasses due to @Inherited.
462        *
463        * @param annotationType The type of the annotation to look for (e.g, "javax.annotation.Nullable")
464        */
465       public static boolean hasAnnotation(Symbol sym, String annotationType, VisitorState state) {
466         Symbol annotationSym = state.getSymbolFromString(annotationType);
467         Symbol inheritedSym = state.getSymtab().inheritedType.tsym;
468     
469         if ((sym == null) || (annotationSym == null)) {
470           return false;
471         }
472         if ((sym instanceof ClassSymbol) && (annotationSym.attribute(inheritedSym) != null)) {
473           while (sym != null) {
474             if (sym.attribute(annotationSym) != null) {
475               return true;
476             }
477             sym = ((ClassSymbol) sym).getSuperclass().tsym;
478           }
479           return false;
480         } else {
481           return sym.attribute(annotationSym) != null;
482         }
483       }

528       /** @return all values of the given enum type, in declaration order. */
529       public static LinkedHashSet<String> enumValues(TypeSymbol enumType) {
530         if (enumType.getKind() != ElementKind.ENUM) {
531           throw new IllegalStateException();
532         }
533         Scope scope = enumType.members();
534         Deque<String> values = new ArrayDeque<>();
535         for (Symbol sym : scope.getSymbols()) {
536           if (sym instanceof VarSymbol) {
537             VarSymbol var = (VarSymbol) sym;
538             if ((var.flags() & Flags.ENUM) != 0) {
539               /**
540                * Javac gives us the members backwards, apparently. It's worth making an effort to
541                * preserve declaration order because it's useful for diagnostics (e.g. in
542                * {@link MissingCasesInEnumSwitch}).
543                */
544               values.push(sym.name.toString());
545             }
546           }
547         }
548         return new LinkedHashSet<>(values);
549       }

607         while (sym != null) {
608           if (clazz.isInstance(sym)) {

610           }
611           sym = sym.owner;
612         }
613         return null;
614       }

626       /**
627        * Returns the {@link Nullness} for an expression as determined by the nullness dataflow
628        * analysis.
629        */
630       public static Nullness getNullnessValue(
631           ExpressionTree expr, VisitorState state, NullnessAnalysis nullnessAnalysis) {
632         TreePath pathToExpr = new TreePath(state.getPath(), expr);
633         return nullnessAnalysis.getNullness(pathToExpr, state.context);
634       }

644        * Return true if the given type is 'void' or 'Void'.
645        */
646       public static boolean isVoidType(Type type, VisitorState state) {
647         if (type == null) {
648           return false;
649         }
650         return type.getKind() == TypeKind.VOID
651             || state.getTypes().isSameType(Suppliers.JAVA_LANG_VOID_TYPE.get(state), type);

688         if (tree instanceof ClassTree) {
689           return ((ClassTree) tree).getModifiers();
690         } else if (tree instanceof MethodTree) {

694         } else {
695           return null;

699       /**
700        * Returns the upper bound of a type if it has one, or the type itself if not. Correctly
701        * handles wildcards and capture variables.
702        */
703       public static Type getUpperBound(Type type, Types types) {
704         if (type.hasTag(TypeTag.WILDCARD)) {
705           return types.wildUpperBound(type);
706         }
707     
708         if (type.hasTag(TypeTag.TYPEVAR) && ((TypeVar) type).isCaptured()) {
709           return types.cvarUpperBound(type);
710         }
711     
712         if (type.getUpperBound() != null) {
713           return type.getUpperBound();
714         }
715     
716         // concrete type, e.g. java.lang.String, or a case we haven't considered
717         return type;
718       }
