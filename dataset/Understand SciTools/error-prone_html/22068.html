45     /**
46      * Abstract skeleton for tests that run the compiler on the fly.
47      * 
48      * @author lowasser@google.com (Louis Wasserman)
49      */
50     public class CompilerBasedTest {
51       protected Context context;
52       protected SourceFile sourceFile;
53       protected Iterable<JCCompilationUnit> compilationUnits;
54       private Map<String, JCMethodDecl> methods;
55       
56       protected void compile(TreeScanner scanner, JavaFileObject fileObject) {
57         JavaCompiler compiler = JavacTool.create();
58         DiagnosticCollector<JavaFileObject> diagnosticsCollector =
59             new DiagnosticCollector<JavaFileObject>();
60         StandardJavaFileManager fileManager = 
61             compiler.getStandardFileManager(diagnosticsCollector, Locale.ENGLISH, UTF_8);
62         JavacTaskImpl task = (JavacTaskImpl) compiler.getTask(CharStreams.nullWriter(), 
63             fileManager,
64             diagnosticsCollector,
65             ImmutableList.<String>of(),
66             null,
67             ImmutableList.of(fileObject));
68         try {
69           this.sourceFile = SourceFile.create(fileObject);
70           Iterable<? extends CompilationUnitTree> trees = task.parse();
71           task.analyze();
72           for (CompilationUnitTree tree : trees) {
73             scanner.scan((JCCompilationUnit) tree);
74           }
75         } catch (IOException e) {
76           throw new RuntimeException(e);
77         }
78         this.context = task.getContext();
79       }
80       
81       protected void compile(TreeScanner scanner, String... lines) {
82         compile(scanner, JavaFileObjects.forSourceLines("CompilerBasedTestInput", lines));
83       }
84     
85       protected void compile(JavaFileObject fileObject) {
86         final ImmutableMap.Builder<String, JCMethodDecl> methodsBuilder = ImmutableMap.builder();
87         final ImmutableList.Builder<JCCompilationUnit> compilationUnitsBuilder = 
88             ImmutableList.builder();
89         compile(new TreeScanner() {
90           @Override
91           public void visitMethodDef(JCMethodDecl tree) {
92             if (!TreeInfo.isConstructor(tree)) {
93               methodsBuilder.put(tree.getName().toString(), tree);
94             }
95           }
96     
97           @Override
98           public void visitTopLevel(JCCompilationUnit tree) {
99             compilationUnitsBuilder.add(tree);
100             super.visitTopLevel(tree);
101           }
102         }, fileObject);
103         this.methods = methodsBuilder.build();
104         this.compilationUnits = compilationUnitsBuilder.build();
105       }
106     
107       protected void compile(String... lines) {
108         compile(JavaFileObjects.forSourceLines("CompilerBasedTestInput", lines));
109       }
110       
111       protected JCMethodDecl getMethodDeclaration(String name) {
112         return methods.get(name);
113       }
114     }
