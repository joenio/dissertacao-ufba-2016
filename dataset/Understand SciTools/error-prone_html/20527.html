39     /**
40      * Matches the behaviour of the javac dep-ann Xlint warning.
41      *
42      * @author cushon@google.com (Liam Miller-Cushon)
43      */
44     @BugPattern(name = "DepAnn", altNames = "dep-ann",
45         summary = "Deprecated item is not annotated with @Deprecated",
46         explanation = "A declaration has the @deprecated Javadoc tag but no @Deprecated annotation. "
47             + "Please add an @Deprecated annotation to this declaration in addition to the @deprecated "
48             + "tag in the Javadoc.",
49         category = JDK, severity = ERROR, maturity = MATURE)
50     public class DepAnn extends BugChecker
51         implements MethodTreeMatcher, ClassTreeMatcher, VariableTreeMatcher {
52     
53       @Override
54       public Description matchMethod(MethodTree methodTree, VisitorState state) {
55         return checkDeprecatedAnnotation(methodTree, state);
56       }
57     
58       @Override
59       public Description matchClass(ClassTree classTree, VisitorState state) {
60         return checkDeprecatedAnnotation(classTree, state);
61       }
62     
63       @Override
64       public Description matchVariable(VariableTree variableTree, VisitorState state) {
65         return checkDeprecatedAnnotation(variableTree, state);
66       }
67     
68       /**
69        * Reports a dep-ann error for a declaration if:
70        *   (1) javadoc contains the deprecated javadoc tag
71        *   (2) the declaration is not annotated with {@link java.lang.Deprecated}
72        */
73       @SuppressWarnings("javadoc")
74       private Description checkDeprecatedAnnotation(Tree tree, VisitorState state) {
75         Symbol symbol = ASTHelpers.getSymbol(tree);
76     
77         // (1)
78         // javac sets the DEPRECATED bit in flags if the Javadoc contains @deprecated
79         if ((symbol.flags() & DEPRECATED) == 0) {
80           return Description.NO_MATCH;
81         }
82     
83         // (2)
84         if (symbol.attribute(state.getSymtab().deprecatedType.tsym) != null) {
85           return Description.NO_MATCH;
86         }
87     
88         return describeMatch(tree, SuggestedFix.prefixWith(tree, "@Deprecated\n"));
89       }
90     }
