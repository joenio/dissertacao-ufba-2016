46     /** @author cushon@google.com (Liam Miller-Cushon) */
47     @BugPattern(
48       name = "StaticGuardedByInstance",
49       category = Category.JDK,
50       summary = "Writes to static fields should not be guarded by instance locks",
51       severity = SeverityLevel.WARNING,
52       maturity = MaturityLevel.MATURE
53     )
54     public class StaticGuardedByInstance extends BugChecker implements SynchronizedTreeMatcher {
55     
56       public static final String MESSAGE =
57           "Write to static variable should not be guarded by instance lock '%s'";
58     
59       @Override
60       public Description matchSynchronized(SynchronizedTree tree, VisitorState state) {
61         Symbol lock = ASTHelpers.getSymbol(TreeInfo.skipParens((JCTree) tree.getExpression()));
62         if (!(lock instanceof VarSymbol)) {
63           return Description.NO_MATCH;
64         }
65         if (lock.isStatic()) {
66           return Description.NO_MATCH;
67         }
68         Multimap<VarSymbol, Tree> writes = WriteVisitor.scan(tree.getBlock());
69         for (Entry<VarSymbol, Tree> write : writes.entries()) {
70           if (!write.getKey().isStatic()) {
71             continue;
72           }
73           state.reportMatch(
74               buildDescription(write.getValue()).setMessage(String.format(MESSAGE, lock)).build());
75         }
76         return Description.NO_MATCH;
77       }
78     
79       static class WriteVisitor extends TreeScanner<Void, Void> {
80     
81         static Multimap<VarSymbol, Tree> scan(Tree tree) {
82           WriteVisitor visitor = new WriteVisitor();
83           tree.accept(visitor, null);
84           return visitor.writes;
85         }
86     
87         private final Multimap<VarSymbol, Tree> writes = LinkedHashMultimap.create();
88     
89         private void recordWrite(ExpressionTree variable) {
90           Symbol sym = ASTHelpers.getSymbol(variable);
91           if (sym instanceof VarSymbol) {
92             writes.put((VarSymbol) sym, variable);
93           }
94         }
95     
96         @Override
97         public Void visitAssignment(AssignmentTree node, Void unused) {
98           recordWrite(node.getVariable());
99           return super.visitAssignment(node, null);
100         }
101     
102         @Override
103         public Void visitCompoundAssignment(CompoundAssignmentTree node, Void unused) {
104           recordWrite(node.getVariable());
105           return super.visitCompoundAssignment(node, null);
106         }
107     
108         @Override
109         public Void visitUnary(UnaryTree node, Void unused) {
110           switch (node.getKind()) {
111             case PREFIX_DECREMENT:
112             case PREFIX_INCREMENT:
113             case POSTFIX_DECREMENT:
114             case POSTFIX_INCREMENT:
115               recordWrite(node.getExpression());
116               break;
117             default:
118               break;
119           }
120           return super.visitUnary(node, null);
121         }
122     
123         @Override
124         public Void visitSynchronized(SynchronizedTree node, Void aVoid) {
125           // don't descend into nested synchronized blocks
126           return null;
127         }
128     
129         @Override
130         public Void visitNewClass(NewClassTree node, Void aVoid) {
131           // don't descend into nested synchronized blocks
132           return null;
133         }
134       }
135     }
