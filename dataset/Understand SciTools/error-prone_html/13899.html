17     package com.google.errorprone.bugpatterns;
18     
19     import static com.google.errorprone.BugPattern.Category.GUAVA;
20     import static com.google.errorprone.BugPattern.MaturityLevel.MATURE;
21     import static com.google.errorprone.BugPattern.SeverityLevel.WARNING;
22     import static com.google.errorprone.matchers.Matchers.staticMethod;
23     
24     import com.google.errorprone.BugPattern;
25     import com.google.errorprone.VisitorState;
26     import com.google.errorprone.bugpatterns.BugChecker.MethodInvocationTreeMatcher;
27     import com.google.errorprone.fixes.SuggestedFix;
28     import com.google.errorprone.matchers.Description;
29     import com.google.errorprone.matchers.Matcher;
30     import com.google.errorprone.matchers.Matchers;
31     
32     import com.sun.source.tree.ExpressionTree;
33     import com.sun.source.tree.LiteralTree;
34     import com.sun.source.tree.MethodInvocationTree;
35     import com.sun.tools.javac.tree.JCTree;
36     
37     import java.util.regex.Pattern;
38     
39     /**
40      * @author Louis Wasserman
41      */
42     @BugPattern(name = "PreconditionsInvalidPlaceholder",
43         summary = "Preconditions only accepts the %s placeholder in error message strings",
44         explanation = "The Guava Preconditions checks take error message template strings that "
45             + "look similar to format strings but only accept %s as a placeholder. This check "
46             + "points out places where there is a non-%s placeholder in a Preconditions error "
47             + "message template string and the number of arguments does not match the number of "
48             + "%s placeholders.",
49         category = GUAVA, maturity = MATURE, severity = WARNING)
50     public class PreconditionsInvalidPlaceholder extends BugChecker
51         implements MethodInvocationTreeMatcher {
52     
53       private static final
54           Matcher<ExpressionTree> PRECONDITIONS_CHECK = Matchers.<ExpressionTree>anyOf(
55               staticMethod().onClass("com.google.common.base.Preconditions").named("checkArgument"),
56               staticMethod().onClass("com.google.common.base.Preconditions").named("checkNotNull"),
57               staticMethod().onClass("com.google.common.base.Preconditions").named("checkState"));
58     
59       private static int expectedArguments(String formatString) {
60         int count = 0;
61         for (int i = formatString.indexOf("%s"); i != -1; i = formatString.indexOf("%s", i + 1)) {
62           count++;
63         }
64         return count;
65       }
66     
67       @Override
68       public Description matchMethodInvocation(MethodInvocationTree t, VisitorState state) {
69         if (PRECONDITIONS_CHECK.matches(t, state) && t.getArguments().size() >= 2
70             && t.getArguments().get(1) instanceof LiteralTree) {
71           LiteralTree formatStringTree = (LiteralTree) t.getArguments().get(1);
72           if (formatStringTree.getValue() instanceof String) {
73             String formatString = (String) formatStringTree.getValue();
74             int expectedArgs = expectedArguments(formatString);
75             if (expectedArgs < t.getArguments().size() - 2
76                 && BAD_PLACEHOLDER_REGEX.matcher(formatString).find()) {
77               return describe(t, state);
78             }
79           }
80         }
81         return Description.NO_MATCH;
82       }
83     
84       /**
85        * Matches most {@code java.util.Formatter} and {@code java.text.MessageFormat}
86        * format placeholders, other than %s itself.
87        *
88        * This does not need to be completely exhaustive, since it is only used to suggest fixes.
89        */
90       private static final Pattern BAD_PLACEHOLDER_REGEX =
91           Pattern.compile("\\$s|%(?:\\d+\\$)??[dbBhHScCdoxXeEfgGaAtTn]|\\{\\d+\\}");
92     
93       public Description describe(MethodInvocationTree t, VisitorState state) {
94         LiteralTree formatTree = (LiteralTree) t.getArguments().get(1);
95     
96         String fixedFormatString =
97             BAD_PLACEHOLDER_REGEX.matcher(state.getSourceForNode((JCTree) formatTree))
98                 .replaceAll("%s");
99         if (expectedArguments(fixedFormatString) == t.getArguments().size() - 2) {
100           return describeMatch(formatTree, SuggestedFix.replace(formatTree, fixedFormatString));
101         } else {
102           int missing = t.getArguments().size() - 2 - expectedArguments(fixedFormatString);
103           StringBuilder builder = new StringBuilder(fixedFormatString);
104           builder.deleteCharAt(builder.length() - 1);
105           builder.append(" [%s");
106           for (int i = 1; i < missing; i++) {
107             builder.append(", %s");
108           }
109           builder.append("]\"");
110           return describeMatch(t, SuggestedFix.replace(formatTree, builder.toString()));
111         }
112       }
113     }
