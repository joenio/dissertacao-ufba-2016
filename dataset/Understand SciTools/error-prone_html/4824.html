17     package com.google.errorprone.refaster;
18     
19     import static com.google.errorprone.refaster.Unifier.unifications;
20     
21     import com.google.auto.value.AutoValue;
22     
23     import com.sun.tools.javac.code.BoundKind;
24     import com.sun.tools.javac.code.Type;
25     import com.sun.tools.javac.code.Type.WildcardType;
26     
27     /**
28      * {@link UType} version of {@link WildcardType}.
29      *
30      * @author lowasser@google.com (Louis Wasserman)
31      */
32     @AutoValue
33     abstract class UWildcardType extends UType {
34       public static UWildcardType create(BoundKind boundKind, UType bound) {
35         return new AutoValue_UWildcardType(boundKind, bound);
36       }
37       
38       /**
39        * This corresponds to a plain ? wildcard.
40        */
41       public static UWildcardType create() {
42         return create(BoundKind.UNBOUND, UClassType.create("java.lang.Object"));
43       }
44       
45       abstract BoundKind boundKind();
46       
47       abstract UType bound();
48     
49       @Override
50       public Choice<Unifier> visitWildcardType(WildcardType wildcard, Unifier unifier) {
51         return Choice.condition(boundKind().equals(wildcard.kind), unifier)
52             .thenChoose(unifications(bound(), wildcard.type));
53       }
54     
55       @Override
56       public Type inline(Inliner inliner) throws CouldNotResolveImportException {
57         return new WildcardType(
58             bound().inline(inliner),
59             boundKind(),
60             inliner.symtab().boundClass);
61       }
62     }
