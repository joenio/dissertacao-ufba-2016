24     public class ArrayHashCodePositiveCases {
25       private Object[] objArray = {1, 2, 3};
26       private String[] stringArray = {"1", "2", "3"};
27       private int[] intArray = {1, 2, 3};
28       private byte[] byteArray = {1, 2, 3};
29       private int[][] multidimensionalIntArray = { {1, 2, 3}, {4, 5, 6} };
30       private String[][] multidimensionalStringArray = { {"1", "2", "3"}, {"4", "5", "6"} };
31       
32       public void objectHashCode() {
33         int hashCode;
34         // BUG: Diagnostic contains: Arrays.hashCode(objArray)
35         hashCode = objArray.hashCode();
36         // BUG: Diagnostic contains: Arrays.hashCode(stringArray)
37         hashCode = stringArray.hashCode();
38         // BUG: Diagnostic contains: Arrays.hashCode(intArray)
39         hashCode = intArray.hashCode();
40         
41         // BUG: Diagnostic contains: Arrays.deepHashCode(multidimensionalIntArray)
42         hashCode = multidimensionalIntArray.hashCode();
43         // BUG: Diagnostic contains: Arrays.deepHashCode(multidimensionalStringArray)
44         hashCode = multidimensionalStringArray.hashCode();
45     
46       }
47     
48       public void guavaObjectsHashCode() {
49         int hashCode;
50         // BUG: Diagnostic contains: Arrays.hashCode(intArray)
51         hashCode = Objects.hashCode(intArray);
52         // BUG: Diagnostic contains: Arrays.hashCode(byteArray)
53         hashCode = Objects.hashCode(byteArray);
54         
55         // BUG: Diagnostic contains: Arrays.deepHashCode(multidimensionalIntArray)
56         hashCode = Objects.hashCode(multidimensionalIntArray);
57         // BUG: Diagnostic contains: Arrays.deepHashCode(multidimensionalStringArray)
58         hashCode = Objects.hashCode(multidimensionalStringArray);
59       }
60       
61       public void varargsHashCodeOnMoreThanOneArg() {
62         int hashCode;
63         // BUG: Diagnostic contains: Objects.hashCode(Arrays.hashCode(objArray), Arrays.hashCode(intArray))
64         hashCode = Objects.hashCode(objArray, intArray);
65         // BUG: Diagnostic contains: Objects.hashCode(Arrays.hashCode(stringArray), Arrays.hashCode(byteArray))
66         hashCode = Objects.hashCode(stringArray, byteArray);
67         
68         Object obj1 = new Object();
69         Object obj2 = new Object();
70         // BUG: Diagnostic contains: Objects.hashCode(obj1, obj2, Arrays.hashCode(intArray))
71         hashCode = Objects.hashCode(obj1, obj2, intArray);
72         // BUG: Diagnostic contains: Objects.hashCode(obj1, Arrays.hashCode(intArray), obj2)
73         hashCode = Objects.hashCode(obj1, intArray, obj2);
74         // BUG: Diagnostic contains: Objects.hashCode(Arrays.hashCode(intArray), obj1, obj2)
75         hashCode = Objects.hashCode(intArray, obj1, obj2);
76         
77         // BUG: Diagnostic contains: Objects.hashCode(obj1, obj2, Arrays.deepHashCode(multidimensionalIntArray))
78         hashCode = Objects.hashCode(obj1, obj2, multidimensionalIntArray);
79         // BUG: Diagnostic contains: Objects.hashCode(obj1, obj2, Arrays.deepHashCode(multidimensionalStringArray))
80         hashCode = Objects.hashCode(obj1, obj2, multidimensionalStringArray);
81       }
