153     /**
154      * Static helper class that provides {@link ScannerSupplier}s and {@link BugChecker}s for the
155      * built-in Error Prone checks, as opposed to plugin checks or checks used in tests.
156      */
157     public class BuiltInCheckerSuppliers {
158       @SafeVarargs
159       public static ImmutableSet<BugCheckerInfo> getSuppliers(Class<? extends BugChecker>... checkers) {
160         ImmutableSet.Builder<BugCheckerInfo> result = ImmutableSet.builder();
161         for (Class<? extends BugChecker> checker : checkers) {
162           result.add(BugCheckerInfo.create(checker));
163         }
164         return result.build();
165       }
166     
167       /**
168        * Returns a {@link ScannerSupplier} with all {@link BugChecker}s in Error Prone.
169        */
170       public static ScannerSupplier allChecks() {
171         return ScannerSupplier.fromBugCheckerInfos(
172             Iterables.concat(ENABLED_ERRORS, ENABLED_WARNINGS, DISABLED_CHECKS));
173       }
174     
175       /**
176        * Returns a {@link ScannerSupplier} with the {@link BugChecker}s that are in the ENABLED lists.
177        */
178       public static ScannerSupplier defaultChecks() {
179         return allChecks().filter(
180             Predicates.or(
181                 Predicates.in(ENABLED_ERRORS),
182                 Predicates.in(ENABLED_WARNINGS)));
183       }
184     
185       /**
186        * Returns a {@link ScannerSupplier} with the {@link BugChecker}s that are in the ENABLED_ERRORS
187        * list.
188        */
189       public static ScannerSupplier errorChecks() {
190         return allChecks().filter(Predicates.in(ENABLED_ERRORS));
191       }
