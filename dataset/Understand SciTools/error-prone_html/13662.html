17     package com.google.errorprone.scanner;
18     
19     import com.google.common.base.Function;
20     import com.google.common.base.Preconditions;
21     import com.google.common.base.Predicate;
22     import com.google.common.collect.FluentIterable;
23     import com.google.common.collect.ImmutableBiMap;
24     import com.google.common.collect.ImmutableSet;
25     import com.google.common.collect.Iterables;
26     import com.google.common.collect.Sets;
27     import com.google.errorprone.BugCheckerInfo;
28     import com.google.errorprone.BugPattern;
29     import com.google.errorprone.bugpatterns.BugChecker;
30     
31     import org.pcollections.PMap;
32     
33     /**
34      * An implementation of a {@link ScannerSupplier}, abstracted as a set of all known
35      * {@link BugChecker}s and a set of enabled {@link BugChecker}s. The set of enabled suppliers must
36      * be a subset of all known suppliers.
37      */
38     class ScannerSupplierImpl extends ScannerSupplier {
39       private final ImmutableBiMap<String, BugCheckerInfo> checks;
40       private final PMap<String, BugPattern.SeverityLevel> severities;
41     
42       ScannerSupplierImpl(
43           ImmutableBiMap<String, BugCheckerInfo> checks,
44           PMap<String, BugPattern.SeverityLevel> severities) {
45         Preconditions.checkArgument(
46             Sets.difference(severities.keySet(), checks.keySet()).isEmpty(),
47             "enabledChecks must be a subset of allChecks");
48         this.checks = checks;
49         this.severities = severities;
50       }
51     
52       // TODO(cushon): BugCheckerSupplier::get
53       private static final Function<BugCheckerInfo, BugChecker> INSTANTIATE_CHECKER =
54           new Function<BugCheckerInfo, BugChecker>() {
55             @Override
56             public BugChecker apply(BugCheckerInfo checkerClass) {
57               try {
58                 return checkerClass.checkerClass().newInstance();
59               } catch (InstantiationException | IllegalAccessException e) {
60                 throw new LinkageError("Could not instantiate BugChecker.", e);
61               }
62             }
63           };
64     
65       @Override
66       public ErrorProneScanner get() {
67         return new ErrorProneScanner(
68             Iterables.transform(getEnabledChecks(), INSTANTIATE_CHECKER), severities);
69       }
70     
71       @Override
72       protected ImmutableBiMap<String, BugCheckerInfo> getAllChecks() {
73         return checks;
74       }
75     
76       @Override
77       protected PMap<String, BugPattern.SeverityLevel> severities() {
78         return severities;
79       }
80     
81       @Override
82       protected ImmutableSet<BugCheckerInfo> getEnabledChecks() {
83         return FluentIterable.from(getAllChecks().values()).filter(isCheckEnabled).toSet();
84       }
85     
86       private final Predicate<BugCheckerInfo> isCheckEnabled =
87           new Predicate<BugCheckerInfo>() {
88             @Override
89             public boolean apply(BugCheckerInfo input) {
90               return input.severity(severities).enabled();
91             }
92       };
93     }
