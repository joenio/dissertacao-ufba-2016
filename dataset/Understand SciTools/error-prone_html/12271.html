17     package com.google.errorprone.bugpatterns;
18     
19     import static com.google.errorprone.BugPattern.Category.JDK;
20     import static com.google.errorprone.BugPattern.MaturityLevel.MATURE;
21     import static com.google.errorprone.BugPattern.SeverityLevel.WARNING;
22     import static com.google.errorprone.matchers.Matchers.isPrimitiveType;
23     import static com.google.errorprone.matchers.Matchers.methodReturnsNonPrimitiveType;
24     
25     import com.google.errorprone.BugPattern;
26     import com.google.errorprone.VisitorState;
27     import com.google.errorprone.fixes.SuggestedFix;
28     import com.google.errorprone.matchers.Description;
29     import com.google.errorprone.matchers.Matcher;
30     import com.google.errorprone.util.ASTHelpers;
31     
32     import com.sun.source.tree.AnnotationTree;
33     import com.sun.source.tree.MethodTree;
34     import com.sun.source.tree.Tree;
35     import com.sun.tools.javac.code.Symbol;
36     
37     /**
38      * @author sebastian.h.monte@gmail.com (Sebastian Monte)
39      */
40     @BugPattern(
41       name = "NullablePrimitive",
42       summary = "@Nullable should not be used for primitive types since they cannot be null",
43       explanation = "Primitives can never be null.",
44       category = JDK,
45       severity = WARNING,
46       maturity = MATURE
47     )
48     public class NullablePrimitive extends BugChecker implements BugChecker.AnnotationTreeMatcher {
49     
50       private static final Matcher<Tree> IS_PRIMITIVE_TYPE_MATCHER = isPrimitiveType();
51       private static final Matcher<MethodTree> METHOD_RETURNS_NON_PRIMITIVE_TYPE_MATCHER =
52           methodReturnsNonPrimitiveType();
53     
54       @Override
55       public Description matchAnnotation(AnnotationTree tree, VisitorState state) {
56         Symbol sym = ASTHelpers.getSymbol(tree);
57         if (sym == null) {
58           return Description.NO_MATCH;
59         }
60         if (!sym.name.contentEquals("Nullable")) {
61           return Description.NO_MATCH;
62         }
63         Tree annotatedNode = getAnnotatedNode(state);
64         Symbol annotatedSymbol = ASTHelpers.getSymbol(annotatedNode);
65         if (annotatedSymbol.isConstructor()) {
66           // @Nullable constructors get their own check
67           return Description.NO_MATCH;
68         }
69         if (IS_PRIMITIVE_TYPE_MATCHER.matches(annotatedNode, state)) {
70           return describeMatch(tree, SuggestedFix.delete(tree));
71         } else if (annotatedNode instanceof MethodTree
72             && !METHOD_RETURNS_NON_PRIMITIVE_TYPE_MATCHER.matches((MethodTree) annotatedNode, state)) {
73           return describeMatch(tree, SuggestedFix.delete(tree));
74         }
75         return Description.NO_MATCH;
76       }
77     
78       private static Tree getAnnotatedNode(VisitorState state) {
79         return state.getPath().getParentPath().getParentPath().getLeaf();
80       }
81     }
