17     package com.google.errorprone.apply;
18     
19     import com.google.common.base.Joiner;
20     import com.google.common.base.Preconditions;
21     import com.google.common.io.CharSource;
22     
23     import java.io.IOException;
24     import java.io.LineNumberReader;
25     import java.io.StringReader;
26     import java.nio.CharBuffer;
27     import java.util.ArrayList;
28     import java.util.List;
29     
30     import javax.tools.JavaFileObject;
31     
32     /**
33      * Representation of a mutable Java source file.
34      * 
35      * This class is not thread-safe.
36      * 
37      * @author sjnickerson@google.com (Simon Nickerson)
38      * @author alexeagle@google.com (Alex Eagle)
39      */
40     public class SourceFile {
41     
42       private final String path;
43       private final StringBuilder sourceBuilder;
44       
45       public static SourceFile create(JavaFileObject fileObject) throws IOException {
46         return new SourceFile(fileObject.toUri().getPath(), fileObject.getCharContent(false));
47       }
48     
49       public SourceFile(String path, CharSequence source) {
50         this.path = path;
51         sourceBuilder = new StringBuilder(source);
52       }
53       
54       /**
55        * Returns the path for this source file
56        */
57       public String getPath() {
58         return path;
59       }
60       
61       /**
62        * Returns a copy of code as a list of lines.
63        */
64       public List<String> getLines() {
65         try {
66           return CharSource.wrap(sourceBuilder).readLines();
67         } catch (IOException e) {
68           throw new AssertionError("IOException not possible, as the string is in-memory");
69         }
70       }
71     
72       /**
73        * Returns a copy of the code as a string.
74        */
75       public String getSourceText() {
76         return sourceBuilder.toString();
77       }
78       
79       public CharSequence getAsSequence() {
80         return CharBuffer.wrap(sourceBuilder).asReadOnlyBuffer();
81       }
82       
83       /**
84        * Clears the current source test for this SourceFile and resets it to
85        * the passed-in value.
86        */
87       public void setSourceText(CharSequence source) {
88         sourceBuilder.setLength(0);     // clear StringBuilder
89         sourceBuilder.append(source);
90       }
91     
92       /**
93        * Returns a fragment of the source code as a string.
94        * 
95        * <p>This method uses the same conventions as {@link String#substring(int, int)} for its start
96        * and end parameters.
97        */
98       public String getFragmentByChars(int startPosition, int endPosition) {
99         return sourceBuilder.substring(startPosition, endPosition);
100       }
101     
102       /**
103        * Returns a fragment of the source code between the two stated line numbers. The parameters
104        * represent <b>inclusive</b> line numbers.
105        * 
106        * <p>The returned fragment will end in a newline.
107        */
108       public String getFragmentByLines(int startLine, int endLine) {
109         Preconditions.checkArgument(startLine <= endLine);
110         return Joiner.on("\n").join(getLines(startLine, endLine)) + "\n";
111       }
112       
113       private List<String> getLines(int startLine, int endLine) {
114         LineNumberReader reader = new LineNumberReader(new StringReader(sourceBuilder.toString()));
115         List<String> lines = new ArrayList<>(endLine - startLine + 1);
116         String line;
117         try {
118           while ((line = reader.readLine()) != null) {
119             if (reader.getLineNumber() >= startLine) {
120               lines.add(line);
121             }
122             if (reader.getLineNumber() >= endLine) {
123               break;
124             }
125           }
126           return lines;
127         } catch (IOException e) {
128           throw new AssertionError("Wrapped StringReader should not produce I/O exceptions");
129         }
130       }
131       
132       /**
133        * Replace the source code with the new lines of code.
134        */
135       public void replaceLines(List<String> lines) {
136         sourceBuilder.replace(0, sourceBuilder.length(), Joiner.on("\n").join(lines) + "\n");
137       }
138       
139       /**
140        * Replace the source code between the start and end lines with some new lines of code.
141        */
142       public void replaceLines(int startLine, int endLine, List<String> replacementLines) {
143         Preconditions.checkArgument(startLine <= endLine);
144         List<String> originalLines = getLines();
145         List<String> newLines = new ArrayList<>();
146         for (int i = 0; i < originalLines.size(); i++) {
147           int lineNum = i + 1;
148           if (lineNum == startLine) {
149             newLines.addAll(replacementLines);
150           } else if (lineNum > startLine && lineNum <= endLine) {
151             // Skip
152           } else {
153             newLines.add(originalLines.get(i));
154           }
155         }
156         replaceLines(newLines);
157       }
158     
159       /**
160        * Replace the source code between the start and end character positions with a new string.
161        * 
162        * <p>This method uses the same conventions as {@link String#substring(int, int)} for its start
163        * and end parameters.
164        */
165       public void replaceChars(int startPosition, int endPosition, String replacement) {
166         try {
167           sourceBuilder.replace(startPosition, endPosition, replacement);
168         } catch (StringIndexOutOfBoundsException e) {
169           throw new IndexOutOfBoundsException(
170               String.format("Replacement cannot be made. Source file %s has length %d, requested start "
171                   + "position %d, requested end position %d, replacement %s",
172                   path, sourceBuilder.length(), startPosition, endPosition, replacement));
173         }
174       }
175     }
