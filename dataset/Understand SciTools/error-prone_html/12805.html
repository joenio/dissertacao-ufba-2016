17     package com.google.errorprone.util;
18     
19     import com.sun.source.tree.Tree;
20     
21     /**
22      * The precedence for an operator kind in the {@link com.sun.source.tree} API.
23      * 
24      * As documented at: http://docs.oracle.com/javase/tutorial/java/nutsandbolts/operators.html
25      */
26     public enum OperatorPrecedence {
27       POSTFIX(13),
28       UNARY(12),
29       MULTIPLICATIVE(11),
30       ADDITIVE(10),
31       SHIFT(9),
32       RELATIONAL(8),
33       EQUALITY(7),
34       AND(6),
35       XOR(5),
36       OR(4),
37       CONDITIONAL_AND(3),
38       CONDITIONAL_OR(2),
39       TERNARY(1),
40       ASSIGNMENT(0);
41       
42       private final int precedence;
43       
44       private OperatorPrecedence(int precedence) {
45         this.precedence = precedence;
46       }
47       
48       public boolean isHigher(OperatorPrecedence other) {
49         return precedence > other.precedence;
50       }
51       
52       public static OperatorPrecedence from(Tree.Kind kind) {
53         switch (kind) {
54           case POSTFIX_DECREMENT:
55           case POSTFIX_INCREMENT:
56             return OperatorPrecedence.POSTFIX;
57           case PREFIX_DECREMENT:
58           case PREFIX_INCREMENT:
59             return OperatorPrecedence.UNARY;
60           case MULTIPLY:
61           case DIVIDE:
62           case REMAINDER:
63             return OperatorPrecedence.MULTIPLICATIVE;
64           case PLUS:
65           case MINUS:
66             return OperatorPrecedence.ADDITIVE;
67           case RIGHT_SHIFT:
68           case UNSIGNED_RIGHT_SHIFT:
69           case LEFT_SHIFT:
70             return OperatorPrecedence.SHIFT;
71           case LESS_THAN:
72           case LESS_THAN_EQUAL:
73           case GREATER_THAN:
74           case GREATER_THAN_EQUAL:
75           case INSTANCE_OF:
76             return OperatorPrecedence.RELATIONAL;
77           case EQUAL_TO:
78           case NOT_EQUAL_TO:
79             return OperatorPrecedence.EQUALITY;
80           case AND:
81             return OperatorPrecedence.AND;
82           case XOR:
83             return OperatorPrecedence.XOR;
84           case OR:
85             return OperatorPrecedence.OR;
86           case CONDITIONAL_AND:
87             return OperatorPrecedence.CONDITIONAL_AND;
88           case CONDITIONAL_OR:
89             return OperatorPrecedence.CONDITIONAL_OR;
90           case ASSIGNMENT:
91           case MULTIPLY_ASSIGNMENT:
92           case DIVIDE_ASSIGNMENT:
93           case REMAINDER_ASSIGNMENT:
94           case PLUS_ASSIGNMENT:
95           case MINUS_ASSIGNMENT:
96           case LEFT_SHIFT_ASSIGNMENT:
97           case AND_ASSIGNMENT:
98           case XOR_ASSIGNMENT:
99           case OR_ASSIGNMENT:
100           case RIGHT_SHIFT_ASSIGNMENT:
101           case UNSIGNED_RIGHT_SHIFT_ASSIGNMENT:
102             return OperatorPrecedence.ASSIGNMENT;
103           default:
104             throw new IllegalArgumentException("Unexpected operator kind: " + kind);
105         }
106       }
107     }
