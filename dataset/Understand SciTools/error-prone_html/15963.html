135     /**
136      * Adapter from {@code BugChecker} to {@code LocalAnalysis}.
137      *
138      * @author Louis Wasserman
139      */
140     public final class BugCheckerLocalAnalysis extends SimpleTreeVisitor<Void, VisitorState>
141         implements LocalAnalysis {
142       private final BugChecker checker;
143     
144       BugCheckerLocalAnalysis(BugChecker checker) {
145         this.checker = checkNotNull(checker);
146       }
147     
148       @Override
149       public Set<String> allNames() {
150         return checker.allNames();
151       }
152     
153       @Override
154       public String canonicalName() {
155         return checker.canonicalName();
156       }

168       @Override
169       public void analyze(TreePath tree, Context context, AnalysesConfig config,
170           DescriptionListener listener) {
171         tree.getLeaf().accept(this, new VisitorState(context, listener).withPath(tree));
172       }

176         return null;

179       private void report(Description description, VisitorState state) {
180         if (description != Description.NO_MATCH) {
181           state.reportMatch(description);
182         }
183       }
