28     import com.google.errorprone.BugPattern;
29     import com.google.errorprone.VisitorState;
30     import com.google.errorprone.bugpatterns.BugChecker.MethodInvocationTreeMatcher;
31     import com.google.errorprone.fixes.Fix;
32     import com.google.errorprone.fixes.SuggestedFix;
33     import com.google.errorprone.matchers.Description;
34     import com.google.errorprone.matchers.Matcher;
35     import com.google.errorprone.matchers.Matchers;

37     import com.sun.source.tree.ExpressionTree;
38     import com.sun.source.tree.MethodInvocationTree;
39     import com.sun.source.tree.Tree;
40     import com.sun.source.tree.Tree.Kind;
41     import com.sun.tools.javac.tree.JCTree.JCFieldAccess;
42     
43     /**
44      * @author awturner@google.com (Andy Turner)
45      */
46     @BugPattern(name = "ProtoFieldPreconditionsCheckNotNull",
47         summary = "Protobuf fields cannot be null, so this check is redundant",
48         explanation = "This checker looks for comparisons of protocol buffer fields with null "
49             + "via the com.google.common.base.Preconditions.checkNotNull method. "
50             + "If a proto field is not specified, its field accessor will return a non-null default "
51             + "value. Thus, the result of calling one of these accessors can never be null, and "
52             + "comparisons like these often indicate a nearby error.\n\n"
53             + "If you meant to check whether an optional field has been set, you should use the "
54             + "hasField() method instead.",
55         category = GUAVA, severity = WARNING, maturity = MATURE)
56     public class ProtoFieldPreconditionsCheckNotNull
57         extends BugChecker implements MethodInvocationTreeMatcher {
58     
59       private static final String PROTO_SUPER_CLASS = "com.google.protobuf.GeneratedMessage";
60     
61       private static final Matcher<ExpressionTree> protoMessageReceiverMatcher =
62           instanceMethod().onDescendantOf(PROTO_SUPER_CLASS);
63     
64       private static final String LIST_INTERFACE = "java.util.List";
65     
66       private static final Matcher<Tree> returnsListMatcher =
67           Matchers.isCastableTo(LIST_INTERFACE);
68     
69       private static final Matcher<ExpressionTree> PROTO_MESSAGE_INVOCATION_MATCHER =
70           new Matcher<ExpressionTree>() {
71             @Override
72             public boolean matches(ExpressionTree tree, VisitorState state) {
73               return isProtoMessageGetInvocation(tree, state);
74             }
75           };
76     
77       @SuppressWarnings({"unchecked"})
78       private static final Matcher<MethodInvocationTree> CHECK_NOT_NULL_MATCHER = allOf(
79           staticMethod().onClass("com.google.common.base.Preconditions").named("checkNotNull"),
80           argument(0, Matchers.<ExpressionTree>allOf(
81               Matchers.<ExpressionTree>kindIs(Kind.METHOD_INVOCATION),
82               PROTO_MESSAGE_INVOCATION_MATCHER)));
83     
84       private static boolean isProtoMessageGetInvocation(ExpressionTree tree, VisitorState state) {
85         return (isGetMethodInvocation(tree, state) || isGetListMethodInvocation(tree, state))
86             && receiverIsProtoMessage(tree, state);
87       }
88     
89       private static boolean isFieldGetMethod(String methodName) {
90         return methodName.startsWith("get");
91       }
92     
93       private static boolean isGetListMethodInvocation(ExpressionTree tree, VisitorState state) {
94         if (tree.getKind() == Tree.Kind.METHOD_INVOCATION) {
95           MethodInvocationTree method = (MethodInvocationTree) tree;
96           if (!method.getArguments().isEmpty()) {
97             return false;
98           }
99           if (!returnsListMatcher.matches(method, state)) {
100             return false;
101           }
102           ExpressionTree expressionTree = method.getMethodSelect();
103           if (expressionTree instanceof JCFieldAccess) {
104             JCFieldAccess access = (JCFieldAccess) expressionTree;
105             String methodName = access.sym.getQualifiedName().toString();
106             return isFieldGetMethod(methodName);
107           }
108           return true;
109         }
110         return false;
111       }
112     
113       private static boolean isGetMethodInvocation(ExpressionTree tree, VisitorState state) {
114         if (tree.getKind() == Tree.Kind.METHOD_INVOCATION) {
115           MethodInvocationTree method = (MethodInvocationTree) tree;
116           if (!method.getArguments().isEmpty()) {
117             return false;
118           }
119           if (returnsListMatcher.matches(method, state)) {
120             return false;
121           }
122           ExpressionTree expressionTree = method.getMethodSelect();
123           if (expressionTree instanceof JCFieldAccess) {
124             JCFieldAccess access = (JCFieldAccess) expressionTree;
125             String methodName = access.sym.getQualifiedName().toString();
126             return isFieldGetMethod(methodName);
127           }
128           return true;
129         }
130         return false;
131       }
132     
133       private static boolean receiverIsProtoMessage(ExpressionTree tree, VisitorState state) {
134         return protoMessageReceiverMatcher.matches(((MethodInvocationTree) tree), state);
135       }
136     
137       @Override
138       public Description matchMethodInvocation(MethodInvocationTree tree, VisitorState state) {
139         if (!CHECK_NOT_NULL_MATCHER.matches(tree, state)
140             || tree.getArguments().isEmpty()) {
141           return Description.NO_MATCH;
142         }
143     
144         Fix fix;
145         if (parentNode(Matchers.<MethodInvocationTree>kindIs(Kind.EXPRESSION_STATEMENT))
146             .matches(tree, state)) {
147           fix = SuggestedFix.delete(state.getPath().getParentPath().getLeaf());
148         } else {
149           fix = SuggestedFix.replace(tree, tree.getArguments().get(0).toString());
150         }
151         return describeMatch(tree, fix);
152       }
153     }
