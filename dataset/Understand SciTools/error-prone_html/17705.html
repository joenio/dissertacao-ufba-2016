51     /**
52      * Checker that ensures implementations of {@link Annotation} override equals and hashCode.
53      * Otherwise, the implementation inherits equals and hashCode from {@link Object}, and those do not
54      * meet the contract specified by the {@link Annotation} interface.
55      */
56     @BugPattern(
57       name = "BadAnnotationImplementation",
58       summary = "Classes that implement Annotation must override equals and hashCode. Consider "
59           + "using AutoAnnotation instead of implementing Annotation by hand.",
60       explanation = "Implementations of `java.lang.annotation.Annotation` must override `equals` "
61           + "and `hashCode`, otherwise they inherit the implementations from `java.lang.Object`, "
62           + "and those implementations do not meet the contract specified by the `Annotation` "
63           + "interface.\n\n"
64           + "It is very difficult to write these methods correctly, so consider using [AutoAnnotation]"
65           + "(https://github.com/google/auto/blob/master/value/src/main/java/com/google/auto/value/AutoAnnotation.java) "
66           + "to generate the correct code automatically.",
67       category = JDK, severity = WARNING, maturity = MATURE)
68     public class BadAnnotationImplementation extends BugChecker implements ClassTreeMatcher {
69     
70       private static final Matcher<ClassTree> CLASS_TREE_MATCHER = allOf(
71           anyOf(kindIs(CLASS), kindIs(ENUM)),
72           isSubtypeOf(ANNOTATION_TYPE));
73     
74       @Override
75       public Description matchClass(ClassTree classTree, final VisitorState state) {
76         if (!CLASS_TREE_MATCHER.matches(classTree, state)) {
77           return Description.NO_MATCH;
78         }
79     
80         // If this is an enum that is trying to implement Annotation, give a special error message.
81         if (classTree.getKind() == Kind.ENUM) {
82           return buildDescription(classTree)
83               .setMessage(
84                   "Enums cannot correctly implement Annotation because their equals and hashCode "
85                       + "methods are final. Consider using AutoAnnotation instead of implementing "
86                       + "Annotation by hand.")
87               .build();
88         }
89     
90         // Otherwise walk up type hierarchy looking for equals and hashcode methods
91         MethodSymbol equals = null;
92         MethodSymbol hashCode = null;
93         final Types types = state.getTypes();
94         Name equalsName = state.getName("equals");
95         Predicate<MethodSymbol> equalsPredicate = new Predicate<MethodSymbol>() {
96           @Override public boolean apply(MethodSymbol methodSymbol) {
97             return !methodSymbol.isStatic()
98                 && ((methodSymbol.flags() & Flags.SYNTHETIC) == 0)
99                 && ((methodSymbol.flags() & Flags.ABSTRACT) == 0)
100                 && methodSymbol.getParameters().size() == 1
101                 && types.isSameType(
102                     methodSymbol.getParameters().get(0).type,
103                     state.getSymtab().objectType);
104           }
105         };
106         Name hashCodeName = state.getName("hashCode");
107         Predicate<MethodSymbol> hashCodePredicate = new Predicate<MethodSymbol>() {
108           @Override public boolean apply(MethodSymbol methodSymbol) {
109             return !methodSymbol.isStatic()
110                 && ((methodSymbol.flags() & Flags.SYNTHETIC) == 0)
111                 && ((methodSymbol.flags() & Flags.ABSTRACT) == 0)
112                 && methodSymbol.getParameters().isEmpty();
113           }
114         };
115     
116         for (Type sup : types.closure(ASTHelpers.getSymbol(classTree).type)) {
117           if (equals == null) {
118             equals = getMatchingMethod(sup, equalsName, equalsPredicate);
119           }
120           if (hashCode == null) {
121             hashCode = getMatchingMethod(sup, hashCodeName, hashCodePredicate);
122           }
123         }
124         Verify.verifyNotNull(equals);
125         Verify.verifyNotNull(hashCode);
126     
127         Symbol objectSymbol = state.getSymtab().objectType.tsym;
128         if (equals.owner.equals(objectSymbol) || hashCode.owner.equals(objectSymbol)) {
129           return describeMatch(classTree);
130         }
131     
132         return Description.NO_MATCH;
133       }
134     
135       private static MethodSymbol getMatchingMethod(
136           Type type, Name name, Predicate<MethodSymbol> predicate) {
137         Scope scope = type.tsym.members();
138         for (Symbol sym : scope.getSymbolsByName(name)) {
139           if (!(sym instanceof MethodSymbol)) {
140             continue;
141           }
142           MethodSymbol methodSymbol = (MethodSymbol) sym;
143           if (predicate.apply(methodSymbol)) {
144             return methodSymbol;
145           }
146         }
147         return null;
148       }
149     }
