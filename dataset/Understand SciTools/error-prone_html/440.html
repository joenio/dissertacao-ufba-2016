17     package com.google.errorprone.refaster;
18     
19     /**
20      * Static utilities to indicate special handling in Refaster templates.
21      *
22      * @author lowasser@google.com (Louis Wasserman)
23      */
24     public class Refaster {
25       private Refaster() {}
26     
27       /**
28        * Indicates that Refaster should attempt to match a target expression against each of the
29        * specified template expressions, in order, and succeed at the first match.
30        *
31        * <p>This method should only be used in Refaster templates, but should never actually be run.
32        *
33        * <p>For example, instead of writing
34        *
35        * <pre><code>
36        *{@literal @}BeforeTemplate &lt;E&gt; List&lt;E&gt; copyOfSingleton(E element) {
37        *   return ImmutableList.copyOf(Collections.singletonList(element));
38        * }
39        *{@literal @}BeforeTemplate &lt;E&gt; List&lt;E&gt; copyOfArrayList(E element) {
40        *   return ImmutableList.copyOf(Lists.newArrayList(element));
41        * }
42        * </code></pre>
43        *
44        * <p>one could alternately write
45        *
46        * <pre><code>
47        *{@literal @}BeforeTemplate &lt;E&gt; List&lt;E&gt; singleton(E element) {
48        *   return ImmutableList.copyOf(Refaster.anyOf(
49        *     Collections.singletonList(element),
50        *     Lists.newArrayList(element)));
51        * }
52        * </code></pre>
53        */
54       @SafeVarargs
55       public static <T> T anyOf(T... expressions) {
56         throw new UnsupportedOperationException();
57       }
58     
59       /**
60        * This is a placeholder for the Java instanceof operator that can be used with Refaster
61        * type variables.  The type argument must always be specified explicitly, e.g.
62        * {@code Refaster.<String>isInstance(o)}.
63        *
64        * <p>For example, instead of writing the broken
65        *
66        * <pre><code>
67        *{@literal @}AfterTemplate &lt;T&gt; boolean instanceOf(Object o) {
68        *   return o instanceof T; // you want to match this, but it won't compile
69        * }
70        * </code></pre>
71        *
72        * <p>you would instead write
73        *
74        * <pre><code>
75        *{@literal @}AfterTemplate &lt;T&gt; boolean instanceOf(Object o) {
76        *   return Refaster.&lt;T&gt;isInstance(o); // generates the replacement "o instanceof T"
77        * }
78        * </code></pre>
79        *
80        * @throws IllegalArgumentException if T is not specified explicitly.
81        */
82       public static <T> boolean isInstance(Object o) {
83         // real code wouldn't have an unused type parameter (T) or an unused argument (o)
84         throw new UnsupportedOperationException(o.toString());
85       }
86     
87       /**
88        * This is a placeholder for {@code new T[size]}. The type argument must always be specified
89        * explicitly, e.g. {@code Refaster.<String>newArray(10)}.
90        *
91        * <p>For example, instead of writing the broken
92        *
93        * <pre><code>
94        *{@literal @}AfterTemplate &lt;T&gt; T[] newTArray(int size) {
95        *   return new T[size]; // you want to generate this code, but it won't compile
96        * }
97        * </code></pre>
98        *
99        * <p>you would instead write
100        *
101        * <pre><code>
102        *{@literal @}AfterTemplate &lt;T&gt; T[] newTArray(int size) {
103        *   return Refaster.&lt;T&gt;newArray(size);
104        * }
105        * </code></pre>
106        *
107        * @throws IllegalArgumentException if T is not specified explicitly.
108        */
109       public static <T> T[] newArray(int size) {
110         throw new UnsupportedOperationException(Integer.toString(size));
111       }
112     
113       /**
114        * This is a placeholder for the expression T.class.  The type argument must always
115        * be specified explicitly, e.g. {@code Refaster.<String>clazz()}.
116        *
117        * <p>For example, instead of writing the broken
118        *
119        * <pre><code>
120        *{@literal @}AfterTemplate &lt;T&gt; T[] getEnumConstants() {
121        *   return T.class.getEnumConstants(); // you want to inline this, but it won't compile
122        * }
123        * </code></pre>
124        *
125        * you would instead write
126        *
127        * <pre><code>
128        *{@literal @}AfterTemplate &lt;T&gt; T[] getEnumConstants() {
129        *   return Refaster.&lt;T&gt;clazz().getEnumConstants();
130        * }
131        * </code></pre>
132        *
133        * @throws IllegalArgumentException if T is not specified explicitly.
134        */
135       public static <T> Class<T> clazz() {
136         throw new UnsupportedOperationException();
137       }
138     
139       /**
140        * This is a placeholder for the expression E.valueOf(string).  The type argument must always
141        * be specified explicitly, e.g. {@code Refaster.<RoundingMode>valueOf(string)}.
142        *
143        * <p>For example, instead of writing the broken
144        *
145        * <pre><code>
146        *{@literal @}BeforeTemplate &lt;E extends Enum&lt;E&gt;&gt; E valueOf(String str) {
147        *   return E.valueOf(str);
148        * }
149        * </code></pre>
150        *
151        * <p>you would instead write
152        *
153        * <pre><code>
154        *{@literal @}BeforeTemplate &lt;E extends Enum&lt;E&gt;&gt; E valueOf(String str) {
155        *   return Refaster.&lt;E&gt;enumValueOf(str);
156        * }
157        * </code></pre>
158        *
159        * @throws IllegalArgumentException if E is not specified explicitly.
160        */
161       public static <E extends Enum<E>> E enumValueOf(String string) {
162         throw new UnsupportedOperationException();
163       }
164     }
