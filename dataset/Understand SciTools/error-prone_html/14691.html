38     import javax.lang.model.SourceVersion;

40     import javax.tools.JavaCompiler;
41     import javax.tools.JavaFileManager;
42     import javax.tools.JavaFileObject;
43     import javax.tools.StandardJavaFileManager;
44     
45     /**
46      * An error-prone compiler that implements {@link javax.tools.JavaCompiler}.
47      *
48      * @author nik
49      */
50     public class ErrorProneJavaCompiler implements JavaCompiler {
51     
52       private final JavaCompiler javacTool;
53       private final ScannerSupplier scannerSupplier;
54     
55       public ErrorProneJavaCompiler() {
56         this(JavacTool.create());
57       }
58     
59       // package-private for testing
60       ErrorProneJavaCompiler(JavaCompiler javacTool) {
61         this(javacTool, BuiltInCheckerSuppliers.defaultChecks());
62       }
63     
64       public ErrorProneJavaCompiler(ScannerSupplier scannerSupplier) {
65         this(JavacTool.create(), scannerSupplier);
66       }
67     
68       private ErrorProneJavaCompiler(JavaCompiler javacTool, ScannerSupplier scannerSupplier) {
69         this.javacTool = javacTool;
70         this.scannerSupplier = scannerSupplier;
71       }
72     
73       @Override
74       public CompilationTask getTask(
75           Writer out,
76           JavaFileManager fileManager,
77           DiagnosticListener<? super JavaFileObject> diagnosticListener,
78           Iterable<String> options,
79           Iterable<String> classes,
80           Iterable<? extends JavaFileObject> compilationUnits) {
81         ErrorProneOptions errorProneOptions;
82         Scanner scanner;
83         try {
84           errorProneOptions = ErrorProneOptions.processArgs(options);
85           scanner = scannerSupplier.applyOverrides(errorProneOptions).get();
86         } catch (InvalidCommandLineOptionException e) {
87           throw new RuntimeException(e);
88         }
89         CodeTransformer transformer = ErrorProneScannerTransformer.create(scanner);
90         List<String> remainingOptions = Arrays.asList(errorProneOptions.getRemainingArgs());
91         CompilationTask task = javacTool.getTask(
92             out, fileManager, diagnosticListener, remainingOptions, classes, compilationUnits);
93         Context context = ((JavacTaskImpl) task).getContext();
94         ErrorProneJavacJavaCompiler.preRegister(context, transformer, errorProneOptions);
95         return task;
96       }
97     
98       @Override
99       public StandardJavaFileManager getStandardFileManager(
100           DiagnosticListener<? super JavaFileObject> diagnosticListener,
101           Locale locale,
102           Charset charset) {
103         return javacTool.getStandardFileManager(diagnosticListener, locale, charset);
104       }
105     
106       @Override
107       public int isSupportedOption(String option) {
108         int numberOfArgs = javacTool.isSupportedOption(option);
109         if (numberOfArgs != -1) {
110           return numberOfArgs;
111         }
112         return ErrorProneOptions.isSupportedOption(option);
113       }
114     
115       @Override
116       public int run(InputStream in, OutputStream out, OutputStream err, String... arguments) {
117         return javacTool.run(in, out, err, arguments);
118       }
119     
120       @Override
121       public Set<SourceVersion> getSourceVersions() {
122         Set<SourceVersion> filtered = EnumSet.noneOf(SourceVersion.class);
123         for (SourceVersion version : javacTool.getSourceVersions()) {
124           if (version.compareTo(SourceVersion.RELEASE_6) >= 0) {
125             filtered.add(version);
126           }
127         }
128         return filtered;
129       }
130     }
