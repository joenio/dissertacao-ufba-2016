170       /**
171        * Tests nullness propagation for references to constants defined in other compilation units. Enum
172        * constants and compile-time constants are still known to be non-null; other constants are
173        * assumed nullable.  It doesn't matter whether the referenced compilation unit is part of the
174        * same compilation or not.  Note we often do better when constants are defined in the same
175        * compilation unit.  Circular initialization dependencies between compilation units are also not
176        * recognized while we do recognize them inside a compilation unit.
177        */
178       @Test
179       public void testConstantsDefinedInOtherCompilationUnits() throws Exception {
180         compilationHelper
181             .addSourceLines("AnotherEnum.java",
182                 "package com.google.errorprone.dataflow.nullnesspropagation;",
183                 "public enum AnotherEnum {",
184                 "  INSTANCE;",
185                 "  public static final String COMPILE_TIME_CONSTANT = \"not null\";",
186                 "  public static final AnotherEnum NOT_COMPILE_TIME_CONSTANT = INSTANCE;",
187                 "  public static final String CIRCULAR = ConstantsFromOtherCompilationUnits.CIRCULAR;",
188                 "}")
189             .addSourceLines("ConstantsFromOtherCompilationUnits.java",
190                 "package com.google.errorprone.dataflow.nullnesspropagation;",
191                 "import static com.google.errorprone.dataflow.nullnesspropagation."
192                 + "NullnessPropagationTest.triggerNullnessChecker;",
193                 "public class ConstantsFromOtherCompilationUnits {",
194                 "  public static final String CIRCULAR = AnotherEnum.CIRCULAR;",
195                 "  public void referenceInsideCompilation() {",
196                 "    // BUG: Diagnostic contains: (Non-null)",
197                 "    triggerNullnessChecker(AnotherEnum.INSTANCE);",
198                 "    // BUG: Diagnostic contains: (Non-null)",
199                 "    triggerNullnessChecker(AnotherEnum.COMPILE_TIME_CONSTANT);",
200                 "    // BUG: Diagnostic contains: (Nullable)",
201                 "    triggerNullnessChecker(AnotherEnum.NOT_COMPILE_TIME_CONSTANT);",
202                 "    // BUG: Diagnostic contains: (Nullable)",
203                 "    triggerNullnessChecker(CIRCULAR);",
204                 "    // BUG: Diagnostic contains: (Nullable)",
205                 "    triggerNullnessChecker(AnotherEnum.CIRCULAR);",
206                 "  }",
207                 "",
208                 "  public void referenceOutsideCompilation() {",
209                 "    // BUG: Diagnostic contains: (Non-null)",
210                 "    triggerNullnessChecker(NullnessPropagationTest.COMPILE_TIME_CONSTANT);",
211                 "    // BUG: Diagnostic contains: (Nullable)",
212                 "    triggerNullnessChecker(NullnessPropagationTest.NOT_COMPILE_TIME_CONSTANT);",
213                 "    // BUG: Diagnostic contains: (Nullable)",
214                 "    triggerNullnessChecker(System.out);",
215                 "    // BUG: Diagnostic contains: (Non-null)",
216                 "    triggerNullnessChecker(java.math.RoundingMode.UNNECESSARY);",
217                 "  }",
218                 "}")
219             .doTest();

222       /**
223        * BugPattern to test dataflow analysis using nullness propagation
224        */
225       @BugPattern(name = "NullnessPropagationChecker",
226           summary = "Test checker for NullnessPropagationTest",
227           explanation = "Outputs an error for each call to triggerNullnessChecker, describing its "
228               + "argument as nullable or non-null",
229           category = JDK, severity = ERROR, maturity = EXPERIMENTAL)
230       public static final class NullnessPropagationChecker
231           extends BugChecker implements MethodInvocationTreeMatcher {
232         private final NullnessPropagationTransfer nullnessPropagation =
233             new NullnessPropagationTransfer();
234     
235         private static final String AMBIGUOUS_CALL_MESSAGE = "AMBIGUOUS CALL: use "
236             + "triggerNullnessCheckerOnPrimitive if you want to test the primitive for nullness";
237     
238         private static final Matcher<ExpressionTree> TRIGGER_CALL_MATCHER = anyOf(
239             staticMethod().onClass(NullnessPropagationTest.class.getName())
240                 .named("triggerNullnessCheckerOnPrimitive"),
241             staticMethod().onClass(NullnessPropagationTest.class.getName())
242                 .named("triggerNullnessCheckerOnBoxed"),
243             staticMethod().onClass(NullnessPropagationTest.class.getName())
244                 .withSignature("triggerNullnessChecker(java.lang.Object)"));
245     
246         private static final Matcher<ExpressionTree> AMBIGUOUS_CALL_FALLBACK_MATCHER =
247             staticMethod().onClass(NullnessPropagationTest.class.getName())
248                 .named("triggerNullnessChecker");
249     
250         @Override
251         public Description matchMethodInvocation(
252             MethodInvocationTree methodInvocation, VisitorState state) {
253           if (!TRIGGER_CALL_MATCHER.matches(methodInvocation, state)) {
254             if (AMBIGUOUS_CALL_FALLBACK_MATCHER.matches(methodInvocation, state)) {
255               return buildDescription(methodInvocation)
256                   .setMessage(AMBIGUOUS_CALL_MESSAGE).build();
257             }
258             return NO_MATCH;
259           }
260     
261           TreePath root = state.getPath();
262           List<Object> values = new ArrayList<>();
263           for (Tree arg : methodInvocation.getArguments()) {
264             TreePath argPath = new TreePath(root, arg);
265             nullnessPropagation.setContext(state.context).setCompilationUnit(root.getCompilationUnit());
266             values.add(expressionDataflow(argPath, state.context, nullnessPropagation));
267             nullnessPropagation.setContext(null).setCompilationUnit(null);
268           }
269     
270           String fixString = "(" + Joiner.on(", ").join(values) + ")";
271           return describeMatch(methodInvocation, replace(methodInvocation, fixString));
272         }
273       }
