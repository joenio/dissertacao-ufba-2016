40     /**
41      * Equivalent to a no-arg block placeholder invocation.
42      */
43     @AutoValue
44     abstract class UBlank implements UStatement {
45       static UBlank create() {
46         return new AutoValue_UBlank(UUID.randomUUID());
47       }
48     
49       abstract UUID unique();
50     
51       Key key() {
52         return new Key(unique());
53       }
54     
55       static class Key extends Bindings.Key<List<? extends StatementTree>> {
56         Key(UUID k) {
57           super(k.toString());
58         }
59       }
60     
61       private static final TreeScanner<Boolean, Unifier> FORBIDDEN_REFERENCE_SCANNER =
62           new TreeScanner<Boolean, Unifier>() {
63             @Override public Boolean reduce(Boolean l, Boolean r) {
64               return firstNonNull(l, false) || firstNonNull(r, false);
65             }
66     
67             @Override public Boolean scan(Tree t, Unifier unifier) {
68               if (t != null) {
69                 Boolean forbidden =
70                     t.accept(PlaceholderUnificationVisitor.FORBIDDEN_REFERENCE_VISITOR, unifier);
71                 return firstNonNull(forbidden, false) || firstNonNull(super.scan(t, unifier), false);
72               }
73               return false;
74             }
75           };
76     
77       @Override
78       public Tree.Kind getKind() {
79         return Kind.OTHER;
80       }
81     
82       @Override
83       public <R, D> R accept(TreeVisitor<R, D> visitor, D data) {
84         return visitor.visitOther(this, data);
85       }
86     
87       @Override
88       public Choice<UnifierWithUnconsumedStatements> apply(
89           final UnifierWithUnconsumedStatements state) {
90         int goodIndex = 0;
91         while (goodIndex < state.unconsumedStatements().size()) {
92           StatementTree stmt = state.unconsumedStatements().get(goodIndex);
93           if (firstNonNull(FORBIDDEN_REFERENCE_SCANNER.scan(stmt, state.unifier()), false)
94               && ControlFlowVisitor.INSTANCE.visitStatement(stmt)
95                   == ControlFlowVisitor.Result.NEVER_EXITS) {
96             break;
97           } else {
98             goodIndex++;
99           }
100         }
101         Collection<Integer> breakPoints =
102             ContiguousSet.create(Range.closed(0, goodIndex), DiscreteDomain.integers());
103         return Choice.from(breakPoints).transform(
104             new Function<Integer, UnifierWithUnconsumedStatements>() {
105               @Override public UnifierWithUnconsumedStatements apply(Integer k) {
106                 Unifier unifier = state.unifier().fork();
107                 unifier.putBinding(key(), state.unconsumedStatements().subList(0, k));
108                 List<? extends StatementTree> remaining = state.unconsumedStatements()
109                     .subList(k, state.unconsumedStatements().size());
110                 return UnifierWithUnconsumedStatements.create(unifier, remaining);
111               }
112             });
113       }
114     
115       @Override public com.sun.tools.javac.util.List<JCStatement> inlineStatements(Inliner inliner) {
116         ListBuffer<JCStatement> buffer = new ListBuffer<>();
117         for (StatementTree stmt :
118             inliner.getOptionalBinding(key()).or(ImmutableList.<StatementTree>of())) {
119           buffer.add((JCStatement) stmt);
120         }
121         return buffer.toList();
122       }
123     }
