17     package com.google.errorprone;
18     
19     import java.io.File;
20     import java.util.ArrayList;
21     import java.util.List;
22     
23     import org.apache.tools.ant.BuildException;
24     import org.apache.tools.ant.Project;
25     import org.apache.tools.ant.taskdefs.compilers.DefaultCompilerAdapter;
26     import org.apache.tools.ant.types.Commandline;
27     import org.apache.tools.ant.types.Commandline.Argument;
28     import org.apache.tools.ant.types.Path;
29     import org.apache.tools.ant.types.Reference;
30     import org.apache.tools.ant.util.JavaEnvUtils;
31     import org.apache.tools.ant.util.LoaderUtils;
32     
33     /**
34      * Ant component to launch an external javac with error-prone enabled.
35      */
36     public class ErrorProneExternalCompilerAdapter extends DefaultCompilerAdapter {
37       private Path classpath;
38       private String memoryStackSize;
39       private List<Argument> jvmArgs = new ArrayList<Argument>();
40     
41       public void setClasspath(Path classpath) {
42         this.classpath = classpath;
43       }
44     
45       public Path createClasspath() {
46         if (classpath == null) {
47           classpath = new Path(getProject());
48         }
49     
50         return classpath.createPath();
51       }
52     
53       public void setClasspathRef(Reference r) {
54         createClasspath().setRefid(r);
55       }
56     
57       public void setMemoryStackSize(String memoryStackSize) {
58         this.memoryStackSize = memoryStackSize;
59       }
60     
61       public Argument createJvmArg() {
62         Argument arg = new Argument();
63         jvmArgs.add(arg);
64         return arg;
65       }
66     
67       @Override
68       public boolean execute() throws BuildException {
69         if (getJavac().isForkedJavac()) {
70           attributes.log("Using external Error Prone compiler", Project.MSG_VERBOSE);
71           Commandline cmd = new Commandline();
72           cmd.setExecutable(JavaEnvUtils.getJdkExecutable("java"));
73           if (memoryStackSize != null) {
74             cmd.createArgument().setValue("-Xss" + memoryStackSize);
75           }
76           String memoryParameterPrefix = "-X";
77           if (memoryInitialSize != null) {
78             cmd.createArgument().setValue(memoryParameterPrefix + "ms" + this.memoryInitialSize);
79             // Prevent setupModernJavacCommandlineSwitches() from doing it also
80             memoryInitialSize = null;
81           }
82           if (memoryMaximumSize != null) {
83             cmd.createArgument().setValue(memoryParameterPrefix + "mx" + this.memoryMaximumSize);
84             // Prevent setupModernJavacCommandlineSwitches() from doing it also
85             memoryMaximumSize = null;
86           }
87           for (Argument arg : jvmArgs) {
88             for (String part : arg.getParts()) {
89               cmd.createArgument().setValue(part);
90             }
91           }
92     
93           Path bootclasspath = new Path(getProject());
94           addResourceSource(bootclasspath, "com/google/errorprone/ErrorProneExternalCompilerAdapter.class");
95           cmd.createArgument().setValue("-Xbootclasspath/p:" + bootclasspath);
96     
97           if (classpath != null) {
98             cmd.createArgument().setValue("-classpath");
99             cmd.createArgument().setPath(classpath);
100           }
101     
102           cmd.createArgument().setValue(ErrorProneCompiler.class.getName());
103           setupModernJavacCommandlineSwitches(cmd);
104           int firstFile = cmd.size();
105           logAndAddFilesToCompile(cmd);
106           return executeExternalCompile(cmd.getCommandline(), firstFile, true) == 0;
107         } else {
108           attributes.log("You must set fork=\"yes\" to use the external Error Prone compiler",
109               Project.MSG_ERR);
110           return false;
111         }
112       }
113     
114       private void addResourceSource(Path classpath, String resource) {
115         final File f = LoaderUtils.getResourceSource(
116             ErrorProneExternalCompilerAdapter.class.getClassLoader(),
117             resource);
118         if (f != null) {
119           attributes.log("Found " + f.getAbsolutePath(), Project.MSG_DEBUG);
120           classpath.createPath().setLocation(f);
121         } else {
122           attributes.log("Couldn't find " + resource, Project.MSG_DEBUG);
123         }
124       }
125     }
