49     /**
50      * Compare a file transformed as suggested by {@link BugChecker} to an expected source.
51      *
52      * Inputs are a {@link BugChecker} instance, input file and expected file.
53      *
54      * @author kurs@google.com (Jan Kurs)
55      */
56     public class BugCheckerRefactoringTestHelper {
57     
58       /**
59        * Test mode for matching refactored source against expected source.
60        */
61       public static enum TestMode {
62         TEXT_MATCH {
63           @Override
64           void verifyMatch(JavaFileObject refactoredSource,
65               JavaFileObject expectedSource) throws IOException {
66             assertThat(refactoredSource.getCharContent(false).toString())
67                 .isEqualTo(expectedSource.getCharContent(false).toString());
68           }
69         },
70         AST_MATCH {
71           @Override
72           void verifyMatch(JavaFileObject refactoredSource, JavaFileObject expectedSource) {
73             assertAbout(javaSource()).that(refactoredSource).parsesAs(expectedSource);
74           }
75         };
76     
77         abstract void verifyMatch(JavaFileObject refactoredSource,
78             JavaFileObject expectedSource) throws IOException;
79       }
80     
81       private final Map<JavaFileObject, JavaFileObject> sources = new HashMap<>();
82       private final BugChecker refactoringBugChecker;
83       private final ErrorProneInMemoryFileManager fileManager;
84     
85       private BugCheckerRefactoringTestHelper(
86           BugChecker refactoringBugChecker, Class<?> clazz) {
87         this.refactoringBugChecker = refactoringBugChecker;
88         this.fileManager = new ErrorProneInMemoryFileManager(clazz);
89       }
90     
91       public static BugCheckerRefactoringTestHelper newInstance(
92           BugChecker refactoringBugChecker, Class<?> clazz) {
93         return new BugCheckerRefactoringTestHelper(refactoringBugChecker, clazz);
94       }
95     
96       public BugCheckerRefactoringTestHelper.ExpectOutput addInput(String inputFilename) {
97         return new ExpectOutput(fileManager.forResource(inputFilename));
98       }
99     
100       public BugCheckerRefactoringTestHelper.ExpectOutput addInputLines(String path, String... input) {
101         return new ExpectOutput(fileManager.forSourceLines(path, input));
102       }
103     
104       public void doTest() throws IOException {
105         this.doTest(TestMode.AST_MATCH);
106       }
107     
108       public void doTest(TestMode testMode) throws IOException {
109         for (Map.Entry<JavaFileObject, JavaFileObject> entry : sources.entrySet()) {
110           runTestOnPair(entry.getKey(), entry.getValue(), testMode);
111         }
112       }
113     
114       private BugCheckerRefactoringTestHelper addInputAndOutput(
115           JavaFileObject input, JavaFileObject output) {
116         sources.put(input, output);
117         return this;
118       }
119     
120       private void runTestOnPair(JavaFileObject input, JavaFileObject output, TestMode testMode)
121           throws IOException {
122         JavacTaskImpl task = createJavacTask();
123         JCCompilationUnit tree = parseAndAnalyze(task, input);
124         JavaFileObject transformed = applyDiff(input, task, tree);
125     
126         testMode.verifyMatch(transformed, output);
127       }
128     
129       private JavaFileObject applyDiff(JavaFileObject sourceFileObject,
130           JavacTaskImpl task, JCCompilationUnit tree) throws IOException {
131         DescriptionBasedDiff diff = DescriptionBasedDiff.create(tree);
132         transformer(refactoringBugChecker).apply(new TreePath(tree), task.getContext(), diff);
133         SourceFile sourceFile = SourceFile.create(sourceFileObject);
134         diff.applyDifferences(sourceFile);
135     
136         JavaFileObject transformed = JavaFileObjects.forSourceString(
137             Iterables.getOnlyElement(Iterables.filter(tree.getTypeDecls(), JCClassDecl.class))
138                 .sym.getQualifiedName()
139                 .toString(),
140             sourceFile.getSourceText());
141         return transformed;
142       }
143     
144       private JCCompilationUnit parseAndAnalyze(JavacTaskImpl task, final JavaFileObject input) {
145         Iterable<? extends CompilationUnitTree> trees = task.parse();
146         task.analyze();
147         return Iterables.getOnlyElement(
148             Iterables.filter(
149                 Iterables.filter(trees, JCCompilationUnit.class),
150                 new Predicate<JCCompilationUnit>() {
151                   @Override
152                   public boolean apply(JCCompilationUnit compilation) {
153                     return compilation.getSourceFile() == input;
154                   }
155                 }));
156       }
157     
158       private JavacTaskImpl createJavacTask() {
159         JavacTool tool = JavacTool.create();
160         DiagnosticCollector<JavaFileObject> diagnosticsCollector = new DiagnosticCollector<>();
161         Context context = new Context();
162     
163         context.put(ErrorProneOptions.class, ErrorProneOptions.empty());
164     
165         JavacTaskImpl task =
166             (JavacTaskImpl)
167                 tool.getTask(
168                     CharStreams.nullWriter(),
169                     fileManager,
170                     diagnosticsCollector,
171                     ImmutableList.<String>of(),
172                     null,
173                     ImmutableList.copyOf(sources.keySet()),
174                     context);
175     
176         return task;
177       }
178     
179       private ErrorProneScannerTransformer transformer(BugChecker bugChecker) {
180         ErrorProneScanner scanner = new ErrorProneScanner(bugChecker);
181         return ErrorProneScannerTransformer.create(scanner);
182       }
183     
184       /**
185        * To assert the proper {@code .addInput().addOutput()} chain.
186        */
187       public class ExpectOutput {
188         private final JavaFileObject input;
189     
190         public ExpectOutput(JavaFileObject input) {
191           this.input = input;
192         }
193     
194         public BugCheckerRefactoringTestHelper addOutputLines(String path, String ... output) {
195           return addInputAndOutput(input, fileManager.forSourceLines(path,  output));
196         }
197     
198         public BugCheckerRefactoringTestHelper addOutput(String outputFilename) {
199           return addInputAndOutput(input, fileManager.forResource(outputFilename));
200         }
201         
202         public BugCheckerRefactoringTestHelper expectUnchanged() {
203           return addInputAndOutput(input, input);
204         }
205       }
206     }
