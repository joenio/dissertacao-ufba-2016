17     package com.google.errorprone.matchers.method;
18     
19     import com.google.common.base.Optional;
20     import com.google.errorprone.VisitorState;
21     import com.google.errorprone.matchers.method.MethodMatchers.MethodClassMatcher;
22     import com.google.errorprone.matchers.method.MethodMatchers.MethodNameMatcher;
23     import com.google.errorprone.matchers.method.MethodMatchers.MethodSignatureMatcher;
24     import com.google.errorprone.matchers.method.MethodNameMatcherImpl.Regex;
25     import com.google.errorprone.predicates.TypePredicate;
26     
27     import com.sun.source.tree.ExpressionTree;
28     
29     import java.util.regex.Pattern;
30     
31     /** Matches on the method's class type, and allows refinement on method name or signature. */
32     class MethodClassMatcherImpl extends AbstractChainedMatcher<MatchState, MatchState>
33         implements MethodClassMatcher {
34     
35       private final TypePredicate predicate;
36     
37       MethodClassMatcherImpl(AbstractSimpleMatcher<MatchState> baseMatcher, TypePredicate predicate) {
38         super(baseMatcher);
39         this.predicate = predicate;
40       }
41     
42       @Override
43       protected Optional<MatchState> matchResult(ExpressionTree item, MatchState method,
44           VisitorState state) {
45         return predicate.apply(method.ownerType(), state)
46             ? Optional.of(method)
47             : Optional.<MatchState>absent();
48       }
49     
50       @Override
51       public MethodNameMatcher named(String name) {
52         return new MethodNameMatcherImpl.Exact(this, name);
53       }
54     
55       @Override
56       public MethodNameMatcher withAnyName() {
57         return new MethodNameMatcherImpl.Any(this);
58       }
59     
60       @Override
61       public MethodSignatureMatcher withSignature(String signature) {
62         return new MethodSignatureMatcherImpl(this, signature);
63       }
64     
65       @Override
66       public MethodNameMatcher withNameMatching(Pattern pattern) {
67         return new Regex(this, pattern);
68       }
69     }
