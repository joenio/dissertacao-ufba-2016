17     package com.google.errorprone.matchers;
18     
19     import com.google.errorprone.VisitorState;
20     
21     import com.sun.source.tree.AnnotationTree;
22     import com.sun.source.tree.ClassTree;
23     import com.sun.source.tree.CompilationUnitTree;
24     import com.sun.source.tree.MethodTree;
25     import com.sun.source.tree.PackageTree;
26     import com.sun.source.tree.Tree;
27     import com.sun.source.tree.VariableTree;
28     
29     /**
30      * Matches if the given annotation matcher matches all of or any of the annotations on the tree
31      * node.
32      *
33      * @author eaftan@google.com (Eddie Aftandilian)
34      */
35     public class Annotation<T extends Tree> extends ChildMultiMatcher<T, AnnotationTree> {
36     
37       public Annotation(MatchType matchType, Matcher<AnnotationTree> nodeMatcher) {
38         super(matchType, nodeMatcher);
39       }
40     
41       @Override
42       protected Iterable<? extends AnnotationTree> getChildNodes(T tree, VisitorState state) {
43         if (tree instanceof ClassTree) {
44           return ((ClassTree) tree).getModifiers().getAnnotations();
45         } else if (tree instanceof VariableTree) {
46           return ((VariableTree) tree).getModifiers().getAnnotations();
47         } else if (tree instanceof MethodTree) {
48           return ((MethodTree) tree).getModifiers().getAnnotations();
49         } else if (tree instanceof CompilationUnitTree) {
50           return ((CompilationUnitTree) tree).getPackageAnnotations();
51         } else if (tree instanceof PackageTree) {
52           return ((PackageTree) tree).getAnnotations();
53         } else {
54           throw new IllegalArgumentException(
55               "Cannot access annotations from tree of type " + tree.getClass());
56         }
57       }
58     }
