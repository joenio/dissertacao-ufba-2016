17     package com.google.errorprone.refaster;
18     
19     import static com.google.common.base.Preconditions.checkArgument;
20     import static com.google.errorprone.refaster.Unifier.unifications;
21     
22     import com.google.auto.value.AutoValue;
23     import com.google.common.collect.BiMap;
24     import com.google.common.collect.EnumBiMap;
25     import com.google.common.collect.Maps;
26     
27     import com.sun.source.tree.TreeVisitor;
28     import com.sun.source.tree.WildcardTree;
29     import com.sun.tools.javac.code.BoundKind;
30     import com.sun.tools.javac.tree.JCTree.JCWildcard;
31     
32     import javax.annotation.Nullable;
33     
34     /**
35      * {@code UTree} representation of a {@code WildcardTree}.
36      *
37      * @author lowasser@google.com (Louis Wasserman)
38      */
39     @AutoValue
40     abstract class UWildcard extends UExpression implements WildcardTree {
41       private static final BiMap<Kind, BoundKind> BOUND_KINDS;
42     
43       static {
44         EnumBiMap<Kind, BoundKind> validKinds = EnumBiMap.create(Kind.class, BoundKind.class);
45         validKinds.put(Kind.UNBOUNDED_WILDCARD, BoundKind.UNBOUND);
46         validKinds.put(Kind.EXTENDS_WILDCARD, BoundKind.EXTENDS);
47         validKinds.put(Kind.SUPER_WILDCARD, BoundKind.SUPER);
48         BOUND_KINDS = Maps.unmodifiableBiMap(validKinds);
49       }
50       
51       static UWildcard create(Kind kind, @Nullable UTree<?> bound) {
52         checkArgument(BOUND_KINDS.containsKey(kind));
53         // verify bound is null iff kind is UNBOUNDED_WILDCARD
54         checkArgument((bound == null) == (kind == Kind.UNBOUNDED_WILDCARD));
55         return new AutoValue_UWildcard(kind, bound);
56       }
57     
58       @Override
59       public abstract Kind getKind();
60     
61       @Override
62       @Nullable
63       public abstract UTree<?> getBound();
64     
65       @Override
66       public JCWildcard inline(Inliner inliner) throws CouldNotResolveImportException {
67         return inliner.maker().Wildcard(
68             inliner.maker().TypeBoundKind(BOUND_KINDS.get(getKind())),
69             (getBound() == null) ? null : getBound().inline(inliner));
70       }
71     
72       @Override
73       public <R, D> R accept(TreeVisitor<R, D> visitor, D data) {
74         return visitor.visitWildcard(this, data);
75       }
76     
77       @Override
78       public Choice<Unifier> visitWildcard(WildcardTree node, Unifier unifier) {
79         return Choice.condition(getKind() == node.getKind(), unifier)
80             .thenChoose(unifications(getBound(), node.getBound()));
81       }
82     }
