53     /**
54      * Matchers for code patterns which appear to be JUnit-based tests.
55      * @author alexeagle@google.com (Alex Eagle)
56      * @author eaftan@google.com (Eddie Aftandillian)
57      */
58     public class JUnitMatchers {
59       public static final String JUNIT4_TEST_ANNOTATION = "org.junit.Test";
60       public static final String JUNIT_BEFORE_ANNOTATION = "org.junit.Before";
61       public static final String JUNIT_AFTER_ANNOTATION = "org.junit.After";
62       public static final String JUNIT_BEFORE_CLASS_ANNOTATION = "org.junit.BeforeClass";
63       public static final String JUNIT_AFTER_CLASS_ANNOTATION = "org.junit.AfterClass";
64       public static final String JUNIT4_RUN_WITH_ANNOTATION = "org.junit.runner.RunWith";
65       private static final String JUNIT3_TEST_CASE_CLASS = "junit.framework.TestCase";
66       private static final String JUNIT4_IGNORE_ANNOTATION = "org.junit.Ignore";
67     
68     
69       public static final Matcher<MethodTree> hasJUnitAnnotation = anyOf(
70           /* @Test, @Before, and @After are inherited by methods that override a base method with the
71            * annotation.  @BeforeClass and @AfterClass can only be applied to static methods, so they
72            * cannot be inherited. */
73           hasAnnotationOnAnyOverriddenMethod(JUNIT4_TEST_ANNOTATION),
74           hasAnnotationOnAnyOverriddenMethod(JUNIT_BEFORE_ANNOTATION),
75           hasAnnotationOnAnyOverriddenMethod(JUNIT_AFTER_ANNOTATION),
76           hasAnnotation(JUNIT_BEFORE_CLASS_ANNOTATION),
77           hasAnnotation(JUNIT_AFTER_CLASS_ANNOTATION));
78     
79       public static final Matcher<MethodTree> hasJUnit4BeforeAnnotations = anyOf(
80           hasAnnotationOnAnyOverriddenMethod(JUNIT_BEFORE_ANNOTATION),
81           hasAnnotation(JUNIT_BEFORE_CLASS_ANNOTATION));
82     
83       public static final Matcher<MethodTree> hasJUnit4AfterAnnotations = anyOf(
84           hasAnnotationOnAnyOverriddenMethod(JUNIT_AFTER_ANNOTATION),
85           hasAnnotation(JUNIT_AFTER_CLASS_ANNOTATION));
86     
87       /**
88        * Matches a class that inherits from TestCase.
89        */
90       public static final Matcher<ClassTree> isTestCaseDescendant =
91           isSubtypeOf(JUNIT3_TEST_CASE_CLASS);
92     
93       /**
94        * Match a class which appears to be missing a @RunWith annotation.
95        *
96        * Matches if:
97        * 1) The class does not have a JUnit 4 @RunWith annotation.
98        * 2) The class is concrete.
99        * 3) The class is a top-level class.
100        */
101       public static final Matcher<ClassTree> isConcreteClassWithoutRunWith = allOf(
102           not(hasAnnotation(JUNIT4_RUN_WITH_ANNOTATION)),
103           not(Matchers.<ClassTree>hasModifier(Modifier.ABSTRACT)),
104           nestingKind(TOP_LEVEL));
105     
106       /**
107        * Match a class which has one or more methods with a JUnit 4 @Test annotation.
108        */
109       public static final Matcher<ClassTree> hasJUnit4TestCases =
110           hasMethod(hasAnnotationOnAnyOverriddenMethod(JUNIT4_TEST_ANNOTATION));
111     
112       /**
113        * Match a class which appears to be a JUnit 3 test class.
114        *
115        * Matches if:
116        * 1) The class does inherit from TestCase.
117        * 2) The class does not have a JUnit 4 @RunWith annotation.
118        * 3) The class is concrete.
119        * 4) This class is a top-level class.
120        */
121       public static final Matcher<ClassTree> isJUnit3TestClass = allOf(
122           isTestCaseDescendant,
123           isConcreteClassWithoutRunWith);
124     
125       /**
126        * Match a method which appears to be a JUnit 3 test case.
127        *
128        * Matches if:
129        * 1) The method's name begins with "test".
130        * 2) The method has no parameters.
131        * 3) The method is public.
132        * 4) The method returns void
133        */
134       public static final Matcher<MethodTree> isJunit3TestCase = allOf(
135           methodNameStartsWith("test"),
136           methodHasParameters(),
137           Matchers.<MethodTree>hasModifier(Modifier.PUBLIC),
138           methodReturns(VOID_TYPE)
139       );
140     
141       /**
142        * Match a method which appears to be a JUnit 3 setUp method
143        *
144        * Matches if:
145        * 1) The method is named "setUp"
146        * 2) The method has no parameters
147        * 3) The method is a public or protected instance method that is not abstract
148        * 4) The method returns void
149        */
150       public static final Matcher<MethodTree> looksLikeJUnit3SetUp = allOf(
151           methodIsNamed("setUp"),
152           methodHasParameters(),
153           anyOf(
154               methodHasVisibility(MethodVisibility.Visibility.PUBLIC),
155               methodHasVisibility(MethodVisibility.Visibility.PROTECTED)
156           ),
157           not(Matchers.<MethodTree>hasModifier(Modifier.ABSTRACT)),
158           not(Matchers.<MethodTree>hasModifier(Modifier.STATIC)),
159           methodReturns(VOID_TYPE)
160       );
161     
162       /**
163        * Match a method which appears to be a JUnit 3 tearDown method
164        *
165        * Matches if:
166        * 1) The method is named "tearDown"
167        * 2) The method has no parameters
168        * 3) The method is a public or protected instance method that is not abstract
169        * 4) The method returns void
170        */
171       public static final Matcher<MethodTree> looksLikeJUnit3TearDown = allOf(
172           methodIsNamed("tearDown"),
173           methodHasParameters(),
174           anyOf(
175               methodHasVisibility(MethodVisibility.Visibility.PUBLIC),
176               methodHasVisibility(MethodVisibility.Visibility.PROTECTED)
177           ),
178           not(Matchers.<MethodTree>hasModifier(Modifier.ABSTRACT)),
179           not(Matchers.<MethodTree>hasModifier(Modifier.STATIC)),
180           methodReturns(VOID_TYPE)
181       );
182     
183       /**
184        * Matches a method annotated with @Test but not @Ignore.
185        */
186       public static final Matcher<MethodTree> wouldRunInJUnit4 = allOf(
187           hasAnnotationOnAnyOverriddenMethod(JUNIT4_TEST_ANNOTATION),
188           not(hasAnnotationOnAnyOverriddenMethod(JUNIT4_IGNORE_ANNOTATION)));

190       public static class JUnit4TestClassMatcher implements Matcher<ClassTree> {
191     
192         /**
193          * A list of test runners that this matcher should look for in the @RunWith annotation.
194          * Subclasses of the test runners are also matched.
195          */
196         private static final Collection<String> TEST_RUNNERS = Arrays.asList(
197             "org.mockito.runners.MockitoJUnitRunner",
198             "org.junit.runners.BlockJUnit4ClassRunner");
199     
200         /**
201          * Matches an argument of type Class<T>, where T is a subtype of one of the test runners listed
202          * in the TEST_RUNNERS field.
203          *
204          * TODO(eaftan): Support checking for an annotation that tells us whether this test runner
205          * expects tests to be annotated with @Test.
206          */
207         private static final Matcher<ExpressionTree> isJUnit4TestRunner =
208             new Matcher<ExpressionTree>() {
209           @Override
210           public boolean matches(ExpressionTree t, VisitorState state) {
211             Type type = ((JCTree) t).type;
212             // Expect a class type.
213             if (!(type instanceof ClassType)) {
214               return false;
215             }
216             // Expect one type argument, the type of the JUnit class runner to use.
217             com.sun.tools.javac.util.List<Type> typeArgs = ((ClassType) type).getTypeArguments();
218             if (typeArgs.size() != 1) {
219               return false;
220             }
221             Type runnerType = typeArgs.get(0);
222             for (String testRunner : TEST_RUNNERS) {
223               Symbol parent = state.getSymbolFromString(testRunner);
224               if (parent == null) {
225                 continue;
226               }
227               if (runnerType.tsym.isSubClass(parent, state.getTypes())) {
228                 return true;
229               }
230             }
231             return false;
232           }
233         };
234     
235         private static final Matcher<ClassTree> isJUnit4TestClass = allOf(
236             not(isTestCaseDescendant),
237             annotations(AT_LEAST_ONE, hasArgumentWithValue("value", isJUnit4TestRunner)));
238     
239         @Override
240         public boolean matches(ClassTree classTree, VisitorState state) {
241           return isJUnit4TestClass.matches(classTree, state);
242         }
243       }
