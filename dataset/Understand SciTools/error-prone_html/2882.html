17     package com.google.errorprone.bugpatterns;
18     
19     import static com.google.errorprone.BugPattern.Category.GUICE;
20     import static com.google.errorprone.BugPattern.MaturityLevel.EXPERIMENTAL;
21     import static com.google.errorprone.BugPattern.SeverityLevel.WARNING;
22     import static com.google.errorprone.matchers.Matchers.hasAnnotation;
23     
24     import com.google.errorprone.BugPattern;
25     import com.google.errorprone.VisitorState;
26     import com.google.errorprone.bugpatterns.BugChecker.MethodTreeMatcher;
27     import com.google.errorprone.fixes.SuggestedFix;
28     import com.google.errorprone.matchers.Description;
29     import com.google.errorprone.matchers.Matcher;
30     import com.google.errorprone.matchers.Matchers;
31     import com.google.errorprone.util.ASTHelpers;
32     
33     import com.sun.source.tree.MethodTree;
34     import com.sun.tools.javac.code.Attribute.Compound;
35     import com.sun.tools.javac.code.Symbol.MethodSymbol;
36     
37     import javax.lang.model.element.TypeElement;
38     
39     /**
40      * This checker matches methods that
41      *   1) are not themselves annotated with @Inject
42      *     (neither javax.inject.Inject nor com.google.inject.Inject)
43      *   2) descend from a method that is annotated with @com.google.inject.Inject
44      *
45      * @author sgoldfeder@google.com (Steven Goldfeder)
46      */
47     @BugPattern(
48       name = "OverridesGuiceInjectableMethod",
49       summary =
50           "This method is not annotated with @Inject, but it overrides a "
51               + "method that is annotated with @com.google.inject.Inject. Guice will inject this "
52               + "method, and it is recommended to annotate it explicitly.",
53       explanation =
54           "Unlike with `@javax.inject.Inject`, if a method overrides a method annotated with "
55               + "`@com.google.inject.Inject`, Guice will inject it even if it itself is not annotated. "
56               + "This differs from the behavior of methods that override `@javax.inject.Inject` "
57               + "methods since according to the JSR-330 spec, a method that overrides a method "
58               + "annotated with `@javax.inject.Inject` will not be injected unless it iself is "
59               + "annotated with `@Inject`. Because of this difference, it is recommended that you "
60               + "annotate this method explicitly.",
61       category = GUICE,
62       severity = WARNING,
63       maturity = EXPERIMENTAL
64     )
65     public class GuiceOverridesGuiceInjectableMethod extends BugChecker implements MethodTreeMatcher {
66     
67       private static final String OVERRIDE_ANNOTATION = "java.lang.Override";
68       private static final String GUICE_INJECT_ANNOTATION = "com.google.inject.Inject";
69       private static final String JAVAX_INJECT_ANNOTATION = "javax.inject.Inject";
70     
71       private static final Matcher<MethodTree> INJECTABLE_METHOD_MATCHER =
72           Matchers.<MethodTree>anyOf(
73               hasAnnotation(GUICE_INJECT_ANNOTATION), hasAnnotation(JAVAX_INJECT_ANNOTATION));
74     
75       private static final Matcher<MethodTree> OVERRIDE_METHOD_MATCHER =
76           Matchers.<MethodTree>hasAnnotation(OVERRIDE_ANNOTATION);
77     
78       @Override
79       public Description matchMethod(MethodTree methodTree, VisitorState state) {
80         // if method is itself annotated with @Inject or it has no ancestor methods, return No_MATCH;
81         if (!INJECTABLE_METHOD_MATCHER.matches(methodTree, state)
82             && OVERRIDE_METHOD_MATCHER.matches(methodTree, state)) {
83           MethodSymbol method = (MethodSymbol) ASTHelpers.getSymbol(methodTree);
84           for (MethodSymbol superMethod : ASTHelpers.findSuperMethods(method, state.getTypes())) {
85             if (isAnnotatedWith(superMethod, GUICE_INJECT_ANNOTATION)) {
86               return buildDescription(methodTree)
87                   .addFix(
88                       SuggestedFix.builder()
89                           .addImport(JAVAX_INJECT_ANNOTATION)
90                           .prefixWith(methodTree, "@Inject\n")
91                           .build())
92                   .setMessage(
93                       String.format(
94                           "This method is not annotated with @Inject, but overrides the method in %s "
95                               + "that is annotated with @com.google.inject.Inject. Guice will inject "
96                               + "this method, and it is recommended to annotate it explicitly.",
97                       ASTHelpers.enclosingClass(superMethod).getQualifiedName()))
98                   .build();
99             }
100           }
101         }
102         return Description.NO_MATCH;
103       }
104     
105       private static boolean isAnnotatedWith(MethodSymbol method, String annotation) {
106         for (Compound c : method.getAnnotationMirrors()) {
107           if (((TypeElement) c.getAnnotationType().asElement())
108               .getQualifiedName()
109               .contentEquals(annotation)) {
110             return true;
111           }
112         }
113         return false;
114       }
115     }
