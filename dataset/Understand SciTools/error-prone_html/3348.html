37     /**
38      * A representation of a block placeholder.
39      *
40      * @author lowasser@google.com (Louis Wasserman)
41      */
42     @AutoValue
43     abstract class UPlaceholderStatement implements UStatement {
44       static UPlaceholderStatement create(PlaceholderMethod placeholder,
45           Iterable<? extends UExpression> arguments,
46           ControlFlowVisitor.Result implementationFlow) {
47         ImmutableList<UVariableDecl> placeholderParams = placeholder.parameters().asList();
48         ImmutableList<UExpression> argumentsList = ImmutableList.copyOf(arguments);
49         ImmutableMap.Builder<UVariableDecl, UExpression> builder = ImmutableMap.builder();
50         for (int i = 0; i < placeholderParams.size(); i++) {
51           builder.put(placeholderParams.get(i), argumentsList.get(i));
52         }
53         return new AutoValue_UPlaceholderStatement(placeholder, builder.build(), implementationFlow);
54       }
55     
56       abstract PlaceholderMethod placeholder();
57     
58       abstract ImmutableMap<UVariableDecl, UExpression> arguments();
59       
60       abstract ControlFlowVisitor.Result implementationFlow();
61     
62       @Override
63       public Kind getKind() {
64         return Kind.OTHER;
65       }
66     
67       @Override
68       public <R, D> R accept(TreeVisitor<R, D> visitor, D data) {
69         return visitor.visitOther(this, data);
70       }

72       @AutoValue
73       abstract static class ConsumptionState {
74         static ConsumptionState empty() {
75           return new AutoValue_UPlaceholderStatement_ConsumptionState(0, List.<JCStatement>nil());
76         }
77     
78         abstract int consumedStatements();
79     
80         abstract List<JCStatement> placeholderImplInReverseOrder();
81     
82         ConsumptionState consume(JCStatement impl) {
83           return new AutoValue_UPlaceholderStatement_ConsumptionState(consumedStatements() + 1,
84               placeholderImplInReverseOrder().prepend(impl));
85         }
86       }

88       @Override
89       public Choice<UnifierWithUnconsumedStatements> apply(
90           final UnifierWithUnconsumedStatements initState) {
91         final PlaceholderUnificationVisitor visitor = PlaceholderUnificationVisitor.create(
92             TreeMaker.instance(initState.unifier().getContext()), arguments());
93         
94         PlaceholderVerificationVisitor verification = new PlaceholderVerificationVisitor(
95             Collections2.transform(placeholder().requiredParameters(), Functions.forMap(arguments())),
96             arguments().values());
97         
98         // The choices where we might conceivably have a completed placeholder match.
99         Choice<State<ConsumptionState>> realOptions = Choice.none();
100         
101         // The choice of consumption states to this point in the block. 
102         Choice<State<ConsumptionState>> choiceToHere = Choice.of(
103             State.create(List.<UVariableDecl>nil(), initState.unifier(), ConsumptionState.empty()));
104     
105         if (verification.allRequiredMatched()) {
106           realOptions = choiceToHere.or(realOptions);
107         }
108         for (final StatementTree targetStatement : initState.unconsumedStatements()) {
109           if (!verification.scan(targetStatement, initState.unifier())) {
110             break; // we saw a variable that's not allowed to be referenced
111           }
112           // Consume another statement, or if that fails, fall back to the previous choices...
113           choiceToHere = choiceToHere.thenChoose(
114               new Function<State<ConsumptionState>, Choice<State<ConsumptionState>>>() {
115                 @Override
116                 public Choice<State<ConsumptionState>> apply(
117                     final State<ConsumptionState> consumptionState) {
118                   return visitor.unifyStatement(targetStatement, consumptionState).transform(
119                       new Function<State<? extends JCStatement>, State<ConsumptionState>>() {
120                         @Override
121                         public State<ConsumptionState> apply(State<? extends JCStatement> stmtState) {
122                           return stmtState.withResult(
123                               consumptionState.result().consume(stmtState.result()));
124                         }
125                       });
126                 }
127               });
128           if (verification.allRequiredMatched()) {
129             realOptions = choiceToHere.or(realOptions);
130           }
131         }
132         return realOptions.thenOption(
133             new Function<State<ConsumptionState>, Optional<UnifierWithUnconsumedStatements>>() {
134               @Override
135               public Optional<UnifierWithUnconsumedStatements> apply(
136                   State<ConsumptionState> consumptionState) {
137                 if (ImmutableSet.copyOf(consumptionState.seenParameters()).containsAll(
138                     placeholder().requiredParameters())) {
139                   Unifier resultUnifier = consumptionState.unifier().fork();
140                   int nConsumedStatements = consumptionState.result().consumedStatements();
141                   java.util.List<? extends StatementTree> remainingStatements = initState
142                       .unconsumedStatements().subList(nConsumedStatements,
143                       initState.unconsumedStatements().size());
144                   UnifierWithUnconsumedStatements result =
145                       UnifierWithUnconsumedStatements.create(resultUnifier, remainingStatements);
146                   List<JCStatement> impl =
147                       consumptionState.result().placeholderImplInReverseOrder().reverse();
148                   ControlFlowVisitor.Result implFlow =
149                       ControlFlowVisitor.INSTANCE.visitStatements(impl);
150                   if (implFlow == implementationFlow()) {
151                     List<JCStatement> prevBinding = resultUnifier.getBinding(placeholder().blockKey());
152                     if (prevBinding != null && prevBinding.toString().equals(impl.toString())) {
153                       return Optional.of(result);
154                     } else if (prevBinding == null) {
155                       resultUnifier.putBinding(placeholder().blockKey(), impl);
156                       return Optional.of(result);
157                     }
158                   }
159                 }
160                 return Optional.absent();
161               }
162             });
163       }

165       @Override
166       public List<JCStatement> inlineStatements(final Inliner inliner) 
167           throws CouldNotResolveImportException {
168         try {
169           Optional<List<JCStatement>> binding = inliner.getOptionalBinding(placeholder().blockKey());
170           
171           // If a placeholder was used as an expression binding in the @BeforeTemplate,
172           // and as a bare statement or as a return in the @AfterTemplate, we may need to convert.
173           Optional<JCExpression> exprBinding = inliner.getOptionalBinding(placeholder().exprKey());
174           binding = binding.or(exprBinding.transform(new Function<JCExpression, List<JCStatement>>() {
175             @Override
176             public List<JCStatement> apply(JCExpression expr) {
177               switch (implementationFlow()) {
178                 case NEVER_EXITS:
179                   return List.of((JCStatement) inliner.maker().Exec(expr));
180                 case ALWAYS_RETURNS:
181                   return List.of((JCStatement) inliner.maker().Return(expr));
182                 default:
183                   throw new AssertionError();
184               }
185             }
186           }));
187           return UPlaceholderExpression.copier(arguments(), inliner).copy(binding.get(), inliner);
188         } catch (UncheckedCouldNotResolveImportException e) {
189           throw e.getCause();
190         }
191       }
