43     /**
44      * An accessor for information about a single bug checker, including the metadata in the check's
45      * {@code @BugPattern} annotation and the class that implements the check.
46      */
47     public class BugCheckerInfo implements Serializable {
48     
49       /**
50        * The BugChecker class.
51        */
52       private final Class<? extends BugChecker> checker;
53     
54       /**
55        * The canonical name of this check. Corresponds to the {@code name} attribute from its
56        * {@code BugPattern} annotation.
57        */
58       private final String canonicalName;
59     
60       /**
61        * Additional identifiers for this check, to be checked for in @SuppressWarnings annotations.
62        * Corresponds to the canonical name plus all {@code altName}s from its {@code BugPattern}
63        * annotation.
64        */
65       private final ImmutableSet<String> allNames;
66     
67       /**
68        * The error message to print in compiler diagnostics when this check triggers. Corresponds to the
69        * {@code summary} attribute from its {@code BugPattern}.
70        */
71       private final String message;
72     
73       /**
74        * The default type of diagnostic (error or warning) to emit when this check triggers.
75        */
76       private final SeverityLevel defaultSeverity;
77     
78       /**
79        * The maturity of this checker. Used to decide whether to enable this check. Corresponds to the
80        * {@code maturity} attribute from its {@code BugPattern}.
81        */
82       private final MaturityLevel maturity;
83     
84       /**
85        * The link URL to display in the diagnostic message when this check triggers. Computed from the
86        * {@code link} and {@code linkType} attributes from its {@code BugPattern}. May be null if no
87        * link should be displayed.
88        */
89       private final String linkUrl;
90     
91       /**
92        * Whether this check may be suppressed. Corresponds to the {@code suppressibility} attribute from
93        * its {@code BugPattern}.
94        */
95       private final Suppressibility suppressibility;
96     
97       /**
98        * A set of custom suppression annotations for this check. Computed from the
99        * {@code suppressibility} and {@code customSuppressionAnnotations} attributes from its {@code
100        * BugPattern}.  May be empty if there are no custom suppression annotations for this check.
101        */
102       private final Set<Class<? extends Annotation>> customSuppressionAnnotations;
103     
104       public static BugCheckerInfo create(Class<? extends BugChecker> checker) {
105         BugPattern pattern = checkNotNull(checker.getAnnotation(BugPattern.class));
106         try {
107           BugPatternValidator.validate(pattern);
108         } catch (ValidationException e) {
109           throw new IllegalStateException(e);
110         }
111         return new BugCheckerInfo(checker, pattern);
112       }
113     
114       private BugCheckerInfo(Class<? extends BugChecker> checker, BugPattern pattern) {
115         this.checker = checker;
116         canonicalName = pattern.name();
117         allNames = ImmutableSet.<String>builder().add(canonicalName).add(pattern.altNames()).build();
118         message = pattern.summary();
119         maturity = pattern.maturity();
120         defaultSeverity = pattern.severity();
121         linkUrl = createLinkUrl(pattern);
122         suppressibility = pattern.suppressibility();
123         if (suppressibility == Suppressibility.CUSTOM_ANNOTATION) {
124           customSuppressionAnnotations =
125               new HashSet<>(Arrays.asList(pattern.customSuppressionAnnotations()));
126         } else {
127           customSuppressionAnnotations = Collections.<Class<? extends Annotation>>emptySet();
128         }
129       }
130     
131       private static final String URL_FORMAT = "http://errorprone.info/bugpattern/%s";
132     
133       private static String createLinkUrl(BugPattern pattern) {
134         switch (pattern.linkType()) {
135           case AUTOGENERATED:
136             return String.format(URL_FORMAT, pattern.name());
137           case CUSTOM:
138             // annotation.link() must be provided.
139             if (pattern.link().isEmpty()) {
140               throw new IllegalStateException(
141                   "If linkType element of @BugPattern is CUSTOM, "
142                       + "a link element must also be provided.");
143             }
144             return pattern.link();
145           case NONE:
146             return null;
147           default:
148             throw new IllegalStateException(
149                 "Unexpected value for linkType element of @BugPattern: " + pattern.linkType());
150         }
151       }
152     
153       /**
154        * Returns a new builder for {@link Description}s.
155        *
156        * @param node the node where the error is
157        * @param checker the {@code BugChecker} instance that is producing this {@code Description}
158        */
159       @CheckReturnValue
160       public static Description.Builder buildDescriptionFromChecker(Tree node, BugChecker checker) {
161         return Description.builder(
162             Preconditions.checkNotNull(node),
163             checker.canonicalName(),
164             checker.linkUrl(),
165             checker.defaultSeverity(),
166             checker.message());
167       }
168     
169       public String canonicalName() {
170         return canonicalName;
171       }
172     
173       public Set<String> allNames() {
174         return allNames;
175       }
176     
177       public String message() {
178         return message;
179       }
180     
181       public MaturityLevel maturity() {
182         return maturity;
183       }
184     
185       public SeverityLevel defaultSeverity() {
186         return defaultSeverity;
187       }
188     
189       public SeverityLevel severity(Map<String, SeverityLevel> severities) {
190         return firstNonNull(severities.get(canonicalName), defaultSeverity);
191       }
192     
193       public String linkUrl() {
194         return linkUrl;
195       }
196     
197       public Suppressibility suppressibility() {
198         return suppressibility;
199       }
200     
201       public Set<Class<? extends Annotation>> customSuppressionAnnotations() {
202         return customSuppressionAnnotations;
203       }
204     
205       public Class<? extends BugChecker> checkerClass() {
206         return checker;
207       }
208     
209       @Override
210       public int hashCode() {
211         return checker.hashCode();
212       }
213     
214       @Override
215       public boolean equals(Object o) {
216         if (!(o instanceof BugCheckerInfo)) {
217           return false;
218         }
219         return checker.equals(((BugCheckerInfo) o).checker);
220       }
221     
222       @Override
223       public String toString() {
224         return canonicalName;
225       }
226     }
