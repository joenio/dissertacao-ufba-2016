17     package com.google.errorprone.refaster;
18     
19     import static com.google.errorprone.refaster.Unifier.unifyList;
20     
21     import com.google.auto.value.AutoValue;
22     import com.google.common.collect.ImmutableList;
23     
24     import com.sun.tools.javac.code.Type;
25     import com.sun.tools.javac.code.Type.MethodType;
26     
27     import java.util.List;
28     
29     import javax.annotation.Nullable;
30     
31     /**
32      * A {@code UType} representation of a {@link MethodType}. This can be used to e.g. disambiguate
33      * method overloads.
34      *
35      * @author lowasser@google.com (Louis Wasserman)
36      */
37     @AutoValue
38     public abstract class UMethodType extends UType {
39     
40       public static UMethodType create(UType returnType, UType... parameterTypes) {
41         return create(returnType, ImmutableList.copyOf(parameterTypes));
42       }
43     
44       public static UMethodType create(UType returnType, List<UType> parameterTypes) {
45         return new AutoValue_UMethodType(returnType, ImmutableList.copyOf(parameterTypes));
46       }
47     
48       public abstract UType getReturnType();
49     
50       public abstract List<UType> getParameterTypes();
51     
52       @Override
53       @Nullable
54       public Choice<Unifier> visitMethodType(MethodType methodTy, @Nullable Unifier unifier) {
55         // Don't unify the return type, which doesn't matter in overload resolution.
56         return unifyList(unifier, getParameterTypes(), methodTy.getParameterTypes());
57       }
58       
59       @Override
60       public MethodType inline(Inliner inliner) throws CouldNotResolveImportException {
61         return new MethodType(
62             inliner.<Type>inlineList(getParameterTypes()),
63             getReturnType().inline(inliner),
64             com.sun.tools.javac.util.List.<Type>nil(),
65             inliner.symtab().methodClass);
66       }
67     }
