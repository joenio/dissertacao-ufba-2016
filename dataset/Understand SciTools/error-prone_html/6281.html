17     package com.google.errorprone.matchers;
18     
19     import com.google.errorprone.VisitorState;
20     import com.sun.source.tree.Tree;
21     
22     /**
23     
24      * @author alexeagle@google.com (Alex Eagle)
25      */
26     public class ParentNode implements Matcher<Tree> {
27       private final Matcher<Tree> treeMatcher;
28     
29       public ParentNode(Matcher<Tree> treeMatcher) {
30         this.treeMatcher = treeMatcher;
31       }
32     
33       @Override
34       public boolean matches(Tree tree, VisitorState state) {
35         Tree parent = state.getPath().getParentPath().getLeaf();
36         try {
37           return treeMatcher.matches(parent, state.withPath(state.getPath().getParentPath()));
38         } catch (ClassCastException e) {
39           return false;
40         }
41       }
42     }
