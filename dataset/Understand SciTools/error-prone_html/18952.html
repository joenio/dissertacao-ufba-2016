17     package com.google.errorprone.refaster.annotation;
18     
19     import java.util.List;
20     import java.util.Map.Entry;
21     import java.util.Set;
22     
23     import javax.annotation.processing.AbstractProcessor;
24     import javax.annotation.processing.RoundEnvironment;
25     import javax.annotation.processing.SupportedAnnotationTypes;
26     import javax.lang.model.SourceVersion;
27     import javax.lang.model.element.AnnotationMirror;
28     import javax.lang.model.element.AnnotationValue;
29     import javax.lang.model.element.Element;
30     import javax.lang.model.element.ExecutableElement;
31     import javax.lang.model.element.TypeElement;
32     import javax.lang.model.type.TypeMirror;
33     import javax.lang.model.util.SimpleAnnotationValueVisitor7;
34     import javax.tools.Diagnostic.Kind;
35     
36     /**
37      * Enforces {@code @RequiredAnnotation} as an annotation processor.
38      * 
39      * @author lowasser@google.com (Louis Wasserman)
40      */
41     @SupportedAnnotationTypes("*")
42     public final class RequiredAnnotationProcessor extends AbstractProcessor {
43       @Override
44       public boolean process(Set<? extends TypeElement> annotations, RoundEnvironment roundEnv) {
45         if (annotations.isEmpty()) {
46           return false;
47         }
48         validateElements(roundEnv.getRootElements());
49         return false;
50       }
51     
52       private AnnotationMirror getAnnotationMirror(Element element, TypeMirror annotationType) {
53         for (AnnotationMirror mirror : element.getAnnotationMirrors()) {
54           if (processingEnv.getTypeUtils().isSameType(mirror.getAnnotationType(), annotationType)) {
55             return mirror;
56           }
57         }
58         return null;
59       }
60     
61       private AnnotationValue getAnnotationValue(AnnotationMirror mirror, String key) {
62         for (Entry<? extends ExecutableElement, ? extends AnnotationValue> entry :
63             mirror.getElementValues().entrySet()) {
64           if (entry.getKey().getSimpleName().contentEquals(key)) {
65             return entry.getValue();
66           }
67         }
68         return null;
69       }
70       
71       private void validateElements(Iterable<? extends Element> elements) {
72         for (Element element : elements) {
73           validateElement(element);
74         }
75       }
76     
77       private void validateElement(final Element element) {
78         TypeMirror requiredAnnotationTypeMirror =
79             processingEnv.getElementUtils().getTypeElement(RequiredAnnotation.class.getName()).asType();
80         for (final AnnotationMirror annotation :
81             processingEnv.getElementUtils().getAllAnnotationMirrors(element)) {
82           AnnotationMirror requiredAnnotationMirror = getAnnotationMirror(
83               annotation.getAnnotationType().asElement(), requiredAnnotationTypeMirror);
84           if (requiredAnnotationMirror == null) {
85             continue;
86           }
87           AnnotationValue value = getAnnotationValue(requiredAnnotationMirror, "value");
88           if (value == null) {
89             continue;
90           }
91           new SimpleAnnotationValueVisitor7<Void, Void>() {
92             @Override
93             public Void visitType(TypeMirror t, Void p) {
94               if (getAnnotationMirror(element, t) == null) {
95                 printError(element,
96                     annotation,
97                     "Annotation %s on %s also requires %s",
98                     annotation,
99                     element,
100                     t);
101               }
102               return null;
103             }
104     
105             @Override
106             public Void visitArray(List<? extends AnnotationValue> vals, Void p) {
107               for (AnnotationValue val : vals) {
108                 visit(val);
109               }
110               return null;
111             }
112           }.visit(value);
113         }
114         validateElements(element.getEnclosedElements());
115       }
116     
117       private void printError(Element element, AnnotationMirror annotation, String message,
118           Object... args) {
119         processingEnv.getMessager().printMessage(Kind.ERROR, String.format(message, args), element,
120             annotation);
121       }
122     
123       @Override
124       public SourceVersion getSupportedSourceVersion() {
125         return SourceVersion.latest();
126       }
127     
128     }
