36     /**
37      * {@code UTree} representation of a {@code MethodTree}.
38      * 
39      * @author lowasser@google.com (Louis Wasserman)
40      */
41     @AutoValue
42     abstract class UMethodDecl extends UTree<JCMethodDecl> implements MethodTree {
43       public static UMethodDecl create(UModifiers modifiers, CharSequence name, UExpression returnType,
44           Iterable<UVariableDecl> parameters, Iterable<UExpression> thrown, UBlock body) {
45         return new AutoValue_UMethodDecl(modifiers, StringName.of(name), returnType, 
46             ImmutableList.copyOf(parameters), ImmutableList.copyOf(thrown), body);
47       }
48       
49       @Override
50       public <R, D> R accept(TreeVisitor<R, D> visitor, D data) {
51         return visitor.visitMethod(this, data);
52       }
53     
54       @Override
55       public Kind getKind() {
56         return Kind.METHOD;
57       }
58     
59       @Override
60       @Nullable
61       public Choice<Unifier> visitMethod(MethodTree decl, @Nullable Unifier unifier) {
62         return getName().unify(decl.getName(), unifier)
63             .thenChoose(unifications(getReturnType(), decl.getReturnType()))
64             .thenChoose(unifications(getParameters(), decl.getParameters()))
65             .thenChoose(unifications(getThrows(), decl.getThrows()))
66             .thenChoose(unifications(getBody(), decl.getBody()));
67       }
68     
69       @Override
70       public JCMethodDecl inline(Inliner inliner) throws CouldNotResolveImportException {
71         return inliner.maker().MethodDef(
72             getModifiers().inline(inliner),
73             getName().inline(inliner),
74             getReturnType().inline(inliner),
75             List.<JCTypeParameter>nil(),
76             List.convert(JCVariableDecl.class, inliner.inlineList(getParameters())),
77             inliner.<JCExpression>inlineList(getThrows()),
78             getBody().inline(inliner),
79             null);
80       }
81     
82       @Override
83       public abstract UModifiers getModifiers();
84     
85       @Override
86       public abstract StringName getName();
87     
88       @Override
89       public abstract UExpression getReturnType();
90     
91       @Override
92       public abstract ImmutableList<UVariableDecl> getParameters();
93     
94       @Override
95       public abstract ImmutableList<UExpression> getThrows();
96     
97       @Override
98       public abstract UBlock getBody();
99     
100       @Override
101       public UTree<?> getDefaultValue() {
102         return null;
103       }
104     
105       @Override
106       public ImmutableList<? extends TypeParameterTree> getTypeParameters() {
107         return ImmutableList.of();
108       }
109     
110       @Override
111       public VariableTree getReceiverParameter() {
112         // TODO(cushon): java 8.
113         throw new IllegalStateException();
114       }
115     }
