17     package com.google.errorprone.refaster;
18     
19     import static com.google.common.base.Preconditions.checkArgument;
20     import static com.google.errorprone.refaster.Unifier.unifications;
21     
22     import com.google.auto.value.AutoValue;
23     import com.google.common.collect.ImmutableBiMap;
24     
25     import com.sun.source.tree.BinaryTree;
26     import com.sun.source.tree.TreeVisitor;
27     import com.sun.tools.javac.tree.JCTree;
28     import com.sun.tools.javac.tree.JCTree.JCBinary;
29     
30     /**
31      * {@link UTree} version of {@link BinaryTree}.
32      *
33      * @author lowasser@google.com (Louis Wasserman)
34      */
35     @AutoValue
36     abstract class UBinary extends UExpression implements BinaryTree {
37       static final ImmutableBiMap<Kind, JCTree.Tag> OP_CODES =
38           new ImmutableBiMap.Builder<Kind, JCTree.Tag>()
39               .put(Kind.PLUS, JCTree.Tag.PLUS)
40               .put(Kind.MINUS, JCTree.Tag.MINUS)
41               .put(Kind.MULTIPLY, JCTree.Tag.MUL)
42               .put(Kind.DIVIDE, JCTree.Tag.DIV)
43               .put(Kind.REMAINDER, JCTree.Tag.MOD)
44               .put(Kind.LEFT_SHIFT, JCTree.Tag.SL)
45               .put(Kind.RIGHT_SHIFT, JCTree.Tag.SR)
46               .put(Kind.UNSIGNED_RIGHT_SHIFT, JCTree.Tag.USR)
47               .put(Kind.OR, JCTree.Tag.BITOR)
48               .put(Kind.AND, JCTree.Tag.BITAND)
49               .put(Kind.XOR, JCTree.Tag.BITXOR)
50               .put(Kind.CONDITIONAL_AND, JCTree.Tag.AND)
51               .put(Kind.CONDITIONAL_OR, JCTree.Tag.OR)
52               .put(Kind.LESS_THAN, JCTree.Tag.LT)
53               .put(Kind.LESS_THAN_EQUAL, JCTree.Tag.LE)
54               .put(Kind.GREATER_THAN, JCTree.Tag.GT)
55               .put(Kind.GREATER_THAN_EQUAL, JCTree.Tag.GE)
56               .put(Kind.EQUAL_TO, JCTree.Tag.EQ)
57               .put(Kind.NOT_EQUAL_TO, JCTree.Tag.NE)
58               .build();
59       
60       static final ImmutableBiMap<Kind, Kind> NEGATION =
61           new ImmutableBiMap.Builder<Kind, Kind>()
62               .put(Kind.LESS_THAN, Kind.GREATER_THAN_EQUAL)
63               .put(Kind.LESS_THAN_EQUAL, Kind.GREATER_THAN)
64               .put(Kind.GREATER_THAN, Kind.LESS_THAN_EQUAL)
65               .put(Kind.GREATER_THAN_EQUAL, Kind.LESS_THAN)
66               .put(Kind.EQUAL_TO, Kind.NOT_EQUAL_TO)
67               .put(Kind.NOT_EQUAL_TO, Kind.EQUAL_TO)
68               .build();
69       
70       static final ImmutableBiMap<Kind, Kind> DEMORGAN =
71           new ImmutableBiMap.Builder<Kind, Kind>()
72               .put(Kind.CONDITIONAL_AND, Kind.CONDITIONAL_OR)
73               .put(Kind.CONDITIONAL_OR, Kind.CONDITIONAL_AND)
74               .put(Kind.AND, Kind.OR)
75               .put(Kind.OR, Kind.AND)
76               .build();
77     
78       public static UBinary create(Kind binaryOp, UExpression lhs, UExpression rhs) {
79         checkArgument(OP_CODES.containsKey(binaryOp), 
80             "%s is not a supported binary operation", binaryOp);
81         return new AutoValue_UBinary(binaryOp, lhs, rhs);
82       }
83     
84       @Override
85       public abstract Kind getKind();
86     
87       @Override
88       public abstract UExpression getLeftOperand();
89     
90       @Override
91       public abstract UExpression getRightOperand();
92     
93       @Override
94       public Choice<Unifier> visitBinary(BinaryTree binary, Unifier unifier) {
95         return Choice.condition(getKind().equals(binary.getKind()), unifier)
96             .thenChoose(unifications(getLeftOperand(), binary.getLeftOperand()))
97             .thenChoose(unifications(getRightOperand(), binary.getRightOperand()));
98       }
99     
100       @Override
101       public <R, D> R accept(TreeVisitor<R, D> visitor, D data) {
102         return visitor.visitBinary(this, data);
103       }
104     
105       @Override
106       public JCBinary inline(Inliner inliner) throws CouldNotResolveImportException {
107         return inliner.maker().Binary(
108             OP_CODES.get(getKind()), 
109             getLeftOperand().inline(inliner),
110             getRightOperand().inline(inliner));
111       }
112       
113       @Override
114       public UExpression negate() {
115         if (NEGATION.containsKey(getKind())) {
116           return UBinary.create(NEGATION.get(getKind()), getLeftOperand(), getRightOperand());
117         } else if (DEMORGAN.containsKey(getKind())) {
118           return UBinary.create(
119               DEMORGAN.get(getKind()),
120               getLeftOperand().negate(),
121               getRightOperand().negate());
122         } else {
123           return super.negate();
124         }
125       }
126     }
