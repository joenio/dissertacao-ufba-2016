19     /**
20      * Test data for {@code ReturnPlaceholderTemplate}.

24     public class ReturnPlaceholderTemplateExample {
25       public static final Ordering<String> LENGTH_THEN_LOWER_CASE_ONE_LINE = new Ordering<String>(){
26         
27         @Override
28         public int compare(String left, String right) {
29             return Integer.compare(left.length(), right.length());
30         }
31     }.compound(new Ordering<String>(){
32         
33         @Override
34         public int compare(String left, String right) {
35             return left.toLowerCase().compareTo(right.toLowerCase());
36         }
37     });
38       
39       public static final Ordering<String> LENGTH_THEN_LOWER_CASE_MULTI_LINE = new Ordering<String>(){
40         
41         @Override
42         public int compare(String left, String right) {
43             return Integer.compare(left.length(), right.length());
44         }
45     }.compound(new Ordering<String>(){
46         
47         @Override
48         public int compare(String left, String right) {
49             String leftLower = left.toLowerCase();
50             String rightLower = right.toLowerCase();
51             return leftLower.compareTo(rightLower);
52         }
53     });
