45     import com.sun.tools.javac.tree.JCTree.JCVariableDecl;
46     
47     import java.util.List;

49     /**
50      * @author alexeagle@google.com (Alex Eagle)
51      */
52     @BugPattern(name = "SelfEquals",
53         summary = "An object is tested for equality to itself",
54         explanation =
55             "The arguments to this equal method are the same object, so it always returns " +
56             "true.  Either change the arguments to point to different objects or substitute true.",
57         category = GUAVA, severity = ERROR, maturity = MATURE)
58     public class SelfEquals extends BugChecker implements MethodInvocationTreeMatcher {
59     
60       /**
61        * Matches calls to the Guava method Objects.equal() in which the two arguments are
62        * the same reference.
63        *
64        * Example: Objects.equal(foo, foo)
65        */
66       @SuppressWarnings({"unchecked"})
67       private static final Matcher<MethodInvocationTree> guavaMatcher = allOf(
68           staticMethod().onClass("com.google.common.base.Objects").named("equal"),
69           sameArgument(0, 1));
70     
71       /**
72        * Matches calls to any instance method called "equals" with exactly one argument in which the
73        * receiver is the same reference as the argument.
74        *
75        * Example: foo.equals(foo)
76        *
77        * TODO(eaftan): This may match too many things, if people are calling methods "equals" that
78        * don't really mean equals.
79        */
80       private static final Matcher<MethodInvocationTree> equalsMatcher = allOf(
81           instanceMethod().anyClass().named("equals"),
82           receiverSameAsArgument(0));
83     
84       /**
85        * The state of the matcher.  Caches the result of matches() for use in describe().
86        */
87       private MatchState matchState = MatchState.NONE;
88     
89       private enum MatchState {
90         NONE,
91         OBJECTS_EQUAL,
92         EQUALS
93       }
94     
95       /**
96        * Should this matcher check for Objects.equal(foo, foo)?
97        */
98       private final boolean checkGuava;
99     
100       /**
101        * Should this matcher check for foo.equals(foo)?
102        */
103       private final boolean checkEquals;
104     
105       public SelfEquals() {
106         this(true, false);
107       }
108     
109       /**
110        * Construct a new SelfEquals matcher.
111        *
112        * @param checkGuava Check for Guava Objects.equal(foo, foo) pattern?
113        * @param checkEquals Check for foo.equals(foo) pattern?
114        */
115       // TODO(alexeagle): looks like this wants to be two different checks
116       public SelfEquals(boolean checkGuava, boolean checkEquals) {
117         if (!checkGuava && !checkEquals) {
118           throw new IllegalArgumentException("SelfEquals should check something");
119         }
120         this.checkGuava = checkGuava;
121         this.checkEquals = checkEquals;
122       }
123     
124       @Override
125       public Description matchMethodInvocation(MethodInvocationTree methodInvocationTree, VisitorState state) {
126         if (checkGuava && guavaMatcher.matches(methodInvocationTree, state)) {
127           matchState = MatchState.OBJECTS_EQUAL;
128           return describe(methodInvocationTree, state);
129         } else if (checkEquals && equalsMatcher.matches(methodInvocationTree, state)) {
130           matchState = MatchState.EQUALS;
131           return describe(methodInvocationTree, state);
132         } else {
133           return Description.NO_MATCH;
134         }
135       }
136     
137       public Description describe(MethodInvocationTree methodInvocationTree, VisitorState state) {
138         if (matchState == MatchState.NONE) {
139           throw new IllegalStateException("describe() called without a match");
140         }
141     
142         // If we don't find a good field to use, then just replace with "true"
143         Fix fix = SuggestedFix.replace(methodInvocationTree, "true");
144     
145         if (matchState == MatchState.OBJECTS_EQUAL) {
146           /**
147            * Cases:
148            *    1) Objects.equal(foo, foo) ==> Objects.equal(foo, other.foo)
149            *    2) Objects.equal(foo, this.foo) ==> Objects.equal(other.foo, this.foo)
150            *    3) Objects.equal(this.foo, foo) ==> Objects.equal(this.foo, other.foo)
151            *    4) Objects.equal(this.foo, this.foo) ==> Objects.equal(this.foo, other.foo)
152            */
153           // Assumption: Both arguments are either identifiers or field accesses.
154           List<? extends ExpressionTree> args = methodInvocationTree.getArguments();
155           for (ExpressionTree arg : args) {
156             switch (arg.getKind()) {
157               case IDENTIFIER: case MEMBER_SELECT:
158                 break;
159               default:
160                 throw new IllegalStateException("Expected arg " + arg + " to be a field access or "
161                     + "identifier");
162             }
163           }
164     
165           // Choose argument to replace.
166           ExpressionTree toReplace;
167           if (args.get(1).getKind() == Kind.IDENTIFIER) {
168             toReplace = args.get(1);
169           } else if (args.get(0).getKind() == Kind.IDENTIFIER) {
170             toReplace = args.get(0);
171           } else {
172             // If we don't have a good reason to replace one or the other, replace the second.
173             toReplace = args.get(1);
174           }
175           // Find containing block
176           TreePath path = state.getPath();
177           while(path.getLeaf().getKind() != Kind.BLOCK) {
178             path = path.getParentPath();
179           }
180           JCBlock block = (JCBlock)path.getLeaf();
181           for (JCStatement jcStatement : block.getStatements()) {
182             if (jcStatement.getKind() == Kind.VARIABLE) {
183               JCVariableDecl declaration = (JCVariableDecl) jcStatement;
184               TypeSymbol variableTypeSymbol = declaration.getType().type.tsym;
185     
186               if (ASTHelpers.getSymbol(toReplace).isMemberOf(variableTypeSymbol, state.getTypes())) {
187                 if (toReplace.getKind() == Kind.IDENTIFIER) {
188                   fix = SuggestedFix.prefixWith(toReplace, declaration.getName() + ".");
189                 } else {
190                   fix = SuggestedFix.replace(((JCFieldAccess) toReplace).getExpression(),
191                       declaration.getName().toString());
192                 }
193               }
194             }
195           }
196         }
197     
198         return describeMatch(methodInvocationTree, fix);
199       }
200     }
