17     package com.google.errorprone.refaster;
18     
19     import com.google.auto.value.AutoValue;
20     
21     import com.sun.tools.javac.code.Type.ArrayType;
22     
23     import javax.annotation.Nullable;
24     
25     /**
26      * {@link UType} version of {@link ArrayType}, which represents a type {@code T[]} based on the type
27      * {@code T}.
28      *
29      * @author lowasser@google.com (Louis Wasserman)
30      */
31     @AutoValue
32     abstract class UArrayType extends UType {
33       public static UArrayType create(UType componentType) {
34         return new AutoValue_UArrayType(componentType);
35       }
36       
37       abstract UType componentType();
38     
39       @Override
40       @Nullable
41       public Choice<Unifier> visitArrayType(ArrayType arrayType, @Nullable Unifier unifier) {
42         return componentType().unify(arrayType.getComponentType(), unifier);
43       }
44       
45       @Override
46       public ArrayType inline(Inliner inliner) throws CouldNotResolveImportException {
47         return new ArrayType(componentType().inline(inliner), inliner.symtab().arrayClass); 
48       }
49     }
