24     public class BadComparablePositiveCases {
25       static class ComparableTest implements Comparable<ComparableTest> {
26         private final long value = 0;
27     
28         public int compareTo(ComparableTest other) {
29           // BUG: Diagnostic contains: return Long.compare(value, other.value);
30           return (int) (value - other.value);
31         }
32       }
33     
34       static class BoxedComparableTest implements Comparable<BoxedComparableTest> {
35         private final Long value = Long.valueOf(0);
36     
37         public int compareTo(BoxedComparableTest other) {
38           // BUG: Diagnostic contains: return value.compareTo(other.value);
39           return (int) (value - other.value);
40         }
41       }
42     
43       static final Comparator<Number> COMPARATOR_UNBOXED_INT_CAST =
44           new Comparator<Number>() {
45             public int compare(Number n1, Number n2) {
46               // BUG: Diagnostic contains: return Long.compare(n1.longValue(), n2.longValue())
47               return (int) (n1.longValue() - n2.longValue());
48             }
49           };
50     
51       static final Comparator<Long> COMPARATOR_BOXED_INT_CAST =
52           new Comparator<Long>() {
53             public int compare(Long n1, Long n2) {
54               // BUG: Diagnostic contains: return n1.compareTo(n2)
55               return (int) (n1 - n2);
56             }
57           };
58     }
