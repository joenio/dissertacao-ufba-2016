20     public class OverridesNegativeCase1 {
21       abstract class Base {
22         abstract void varargsMethod(Object... xs);
23         abstract void arrayMethod(Object[] xs);
24       }
25     
26       abstract class Child1 extends Base {
27         @Override
28         abstract void varargsMethod(final Object... newNames);
29       }
30     
31       abstract class Child2 extends Base {
32         @Override
33         abstract void arrayMethod(Object[] xs);
34       }
35     
36       static class StaticClass {
37         static void staticVarargsMethod(Object... xs) {
38         }
39         
40         static void staticArrayMethod(Object[] xs) {
41         }
42       }
43     
44       interface Interface {
45         void varargsMethod(Object... xs);
46         void arrayMethod(Object[] xs);
47       }
48     
49       abstract class ImplementsInterface implements Interface {
50         public abstract void varargsMethod(Object... xs);
51         public abstract void arrayMethod(Object[] xs);
52       }
53     }
54     
55     // Varargs methods might end up overriding synthetic (e.g. bridge) methods, which will have already
56     // been lowered into a non-varargs form. Test that we don't report errors when a varargs method
57     // overrides a synthetic non-varargs method:
58     
59     abstract class One {
60       static class Builder {
61         Builder varargsMethod(String... args) {
62           return this;
63         }
64       }
65     }
66     
67     class Two extends One {
68       static class Builder extends One.Builder {
69         @Override
70         public Builder varargsMethod(String... args) {
71           super.varargsMethod(args);
72           return this;
73         }
74       }
75     }
76     
77     class Three extends Two {
78       static class Builder extends Two.Builder {
79         @Override
80         public Builder varargsMethod(String... args) {
81           super.varargsMethod(args);
82           return this;
83         }
84       }
85     }
