53     /**
54      * @author amshali@google.com (Amin Shali)
55      * @author eaftan@google.com (Eddie Aftandilian)
56      */
57     @BugPattern(name = "ElementsCountedInLoop",
58         summary = "This code, which counts elements using a loop, can be replaced by a simpler library "
59             + "method",
60         explanation = "This code counts elements using a loop.  You can use various library methods "
61             + "(Guava's Iterables.size(), Collection.size(), array.length) to achieve the same thing "
62             + "in a cleaner way.",
63         category = JDK, severity = WARNING, maturity = MATURE)
64     public class ElementsCountedInLoop extends BugChecker
65         implements EnhancedForLoopTreeMatcher, WhileLoopTreeMatcher {
66     
67       @Override
68       public Description matchWhileLoop(WhileLoopTree tree, VisitorState state) {
69         JCWhileLoop whileLoop = (JCWhileLoop) tree;
70         JCExpression whileExpression = ((JCParens) whileLoop.getCondition()).getExpression();
71         if (whileExpression instanceof MethodInvocationTree) {
72           MethodInvocationTree methodInvocation = (MethodInvocationTree) whileExpression;
73           if (instanceMethod().onDescendantOf("java.util.Iterator").withSignature("hasNext()").matches(
74               methodInvocation, state)) {
75             IdentifierTree identifier = getIncrementedIdentifer(extractSingleStatement(whileLoop.body));
76             if (identifier != null) {
77               return describeMatch(tree);
78             }
79           }
80         }
81         return Description.NO_MATCH;
82       }
83     
84       @Override
85       public Description matchEnhancedForLoop(EnhancedForLoopTree tree, VisitorState state) {
86         JCEnhancedForLoop enhancedForLoop = (JCEnhancedForLoop) tree;
87         IdentifierTree identifier =
88             getIncrementedIdentifer(extractSingleStatement(enhancedForLoop.body));
89         if (identifier != null) {
90           ExpressionTree expression = tree.getExpression();
91           Fix fix;
92           if (isSubtypeOf("java.util.Collection").matches(expression, state)) {
93             String replacement = identifier + " += " + expression + ".size();";
94             fix = SuggestedFix.replace(tree, replacement);
95           } else if (isArrayType().matches(expression, state)) {
96             String replacement = identifier + " += " + expression + ".length;";
97             fix = SuggestedFix.replace(tree, replacement);
98           } else {
99             String replacement = identifier + " += Iterables.size(" + expression + ");";
100             fix = SuggestedFix.builder()
101                 .replace(tree, replacement)
102                 .addImport("com.google.common.collect.Iterables")
103                 .build();
104           }
105           return describeMatch(tree, fix);
106         }
107         return Description.NO_MATCH;
108       }
109     
110       /**
111        * @return the only statement in a block containing one statement or the body itself when it is
112        *         not a block.
113        */
114       private JCStatement extractSingleStatement(JCStatement body) {
115         if (body.getKind() != Kind.BLOCK) {
116           return body;
117         }
118         JCBlock block = (JCBlock) body;
119         if (block.getStatements().size() == 1) {
120           return block.getStatements().get(0);
121         }
122         return null;
123       }
124     
125       /**
126        * @return identifier which is being incremented by constant one. Returns null if no such
127        *         identifier is found.
128        */
129       private IdentifierTree getIncrementedIdentifer(JCStatement statement) {
130         if (statement == null) {
131           return null;
132         }
133         if (statement.getKind() == Kind.EXPRESSION_STATEMENT) {
134           Tree.Kind kind = ((JCExpressionStatement) statement).getExpression().getKind();
135           if (kind == Kind.PREFIX_INCREMENT || kind == Kind.POSTFIX_INCREMENT) {
136             JCUnary unary = (JCUnary) ((JCExpressionStatement) statement).getExpression();
137             if (unary.arg.getKind() == Kind.IDENTIFIER) {
138               return (IdentifierTree) unary.arg;
139             }
140             return null;
141           } else if (kind == Kind.PLUS_ASSIGNMENT) {
142             JCAssignOp assignOp = (JCAssignOp) ((JCExpressionStatement) statement).getExpression();
143             if (assignOp.lhs.getKind() == Kind.IDENTIFIER && (isConstantOne(assignOp.rhs))) {
144               return (IdentifierTree) assignOp.lhs;
145             }
146           } else if (kind == Kind.ASSIGNMENT) {
147             JCAssign assign = (JCAssign) ((JCExpressionStatement) statement).getExpression();
148             if (assign.lhs.getKind() == Kind.IDENTIFIER && assign.rhs.getKind() == Kind.PLUS) {
149               JCBinary binary = (JCBinary) assign.rhs;
150               if (binary.lhs.getKind() == Kind.IDENTIFIER) {
151                 if (((JCIdent) assign.lhs).sym == ((JCIdent) binary.lhs).sym) {
152                   if (isConstantOne(binary.rhs)) {
153                     return (IdentifierTree) binary.lhs;
154                   }
155                 }
156               }
157               if (binary.rhs.getKind() == Kind.IDENTIFIER) {
158                 if (((JCIdent) assign.lhs).sym == ((JCIdent) binary.rhs).sym) {
159                   if (isConstantOne(binary.lhs)) {
160                     return (IdentifierTree) binary.rhs;
161                   }
162                 }
163               }
164             }
165           }
166         }
167         return null;
168       }
169     
170       private boolean isConstantOne(JCExpression exp) {
171         Tree.Kind kind = exp.getKind();
172     
173         if (kind == Kind.INT_LITERAL || kind == Kind.LONG_LITERAL || kind == Kind.FLOAT_LITERAL
174             || kind == Kind.DOUBLE_LITERAL) {
175           if (exp instanceof LiteralTree) {
176             Object literalValue = ((LiteralTree) exp).getValue();
177             if (literalValue instanceof Number) {
178               int intValue = ((Number) literalValue).intValue();
179               if (intValue == 1) {
180                 return true;
181               }
182             }
183           }
184         }
185         return false;
186       }
