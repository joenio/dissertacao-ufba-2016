17     package com.google.errorprone;
18     
19     import com.google.auto.value.AutoValue;
20     
21     /**
22      * Encapsulates information about a positive or negative test case that should be included in the
23      * generated documentation.
24      */
25     @AutoValue
26     public abstract class ExampleInfo {
27     
28       public enum ExampleKind {
29         POSITIVE,
30         NEGATIVE
31       }
32     
33       /**
34        * Whether this is a positive or negative example.
35        */
36       public abstract ExampleKind type();
37     
38       /**
39        * The fully-qualified name of the checker that this example belongs to.
40        */
41       public abstract String checkerClass();
42     
43       /**
44        * The name of the example, e.g. "ArrayEqualsPositiveCases.java" or
45        * "testFlagsSimpleCovariantEqualsMethod".
46        */
47       public abstract String name();
48     
49       /**
50        * The example code.
51        */
52       public abstract String code();
53     
54       public static ExampleInfo create(
55           ExampleKind type, String checkerClass, String name, String code) {
56         return new AutoValue_ExampleInfo(type, checkerClass, name, code);
57       }
58     }
