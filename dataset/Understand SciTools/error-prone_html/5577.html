95        * A helper method to create a stubbed package JCExpression.
96        *
97        * @param endPos the end position of the package JCExpression
98        * @return a new package JCExpression stub
99        */
100       private static JCExpression stubPackage(int endPos) {
101         JCExpression result = mock(JCExpression.class);
102         when(result.getEndPosition(any(EndPosTable.class))).thenReturn(endPos);
103         return result;
104       }

106       /**
107        * A helper method to create a JCImport stub.
108        *
109        * @param typeName the fully-qualified name of the type being imported
110        * @param isStatic whether the import is static
111        * @param startPos the start position of the import statement
112        * @param endPos the end position of the import statement
113        * @return a new JCImport stub
114        */
115       private static JCImport stubImport(String typeName, boolean isStatic, int startPos, int endPos) {
116         JCImport result = mock(JCImport.class);
117         when(result.isStatic()).thenReturn(isStatic);
118         when(result.getStartPosition()).thenReturn(startPos);
119         when(result.getEndPosition(any(EndPosTable.class))).thenReturn(endPos);
120     
121         // craft import string
122         StringBuilder returnSB = new StringBuilder("import ");
123         if (isStatic) {
124           returnSB.append("static ");
125         }
126         returnSB.append(typeName);
127         returnSB.append(";\n");
128         when(result.toString()).thenReturn(returnSB.toString());
129         return result;
130       }
