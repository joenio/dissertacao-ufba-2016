58     /**
59      * TODO(eaftan): Consider cases where the parent is not a statement or there is no parent?
60      *
61      * @author eaftan@google.com (Eddie Aftandilian)
62      * @author scottjohnson@google.com (Scott Johnson)
63      */
64     @BugPattern(name = "SelfAssignment",
65         summary = "Variable assigned to itself",
66         explanation = "The left-hand side and right-hand side of this assignment are the same. " +
67             "It has no effect.\n\n" +
68             "This also handles assignments in which the right-hand side is a call to " +
69             "Preconditions.checkNotNull(), which returns the variable that was checked for " +
70             "non-nullity.  If you just intended to check that the variable is non-null, please " +
71             "don't assign the result to the checked variable; just call Preconditions.checkNotNull() " +
72             "as a bare statement.",
73         category = JDK, severity = ERROR, maturity = MATURE)
74     public class SelfAssignment extends BugChecker
75         implements AssignmentTreeMatcher, VariableTreeMatcher {
76     
77       @Override
78       public Description matchAssignment(AssignmentTree tree, VisitorState state) {
79         ExpressionTree expression = stripCheckNotNull(tree.getExpression(), state);
80         if(ASTHelpers.sameVariable(tree.getVariable(), expression)) {
81           return describeForAssignment(tree, state);
82         }
83         return Description.NO_MATCH;
84       }
85     
86       @Override
87       public Description matchVariable(VariableTree tree, VisitorState state) {
88         ExpressionTree initializer = stripCheckNotNull(tree.getInitializer(), state);
89         Tree parent = state.getPath().getParentPath().getLeaf();
90     
91         // must be a static class variable with member select initializer
92         if (initializer == null || initializer.getKind() != MEMBER_SELECT || parent.getKind() != CLASS
93             || !tree.getModifiers().getFlags().contains(STATIC)) {
94           return Description.NO_MATCH;
95         }
96     
97         MemberSelectTree rhs = (MemberSelectTree) initializer;
98         Symbol rhsClass = ASTHelpers.getSymbol(rhs.getExpression());
99         Symbol lhsClass = ASTHelpers.getSymbol(parent);
100         if (rhsClass != null && lhsClass != null
101             && rhsClass.equals(lhsClass) && rhs.getIdentifier().contentEquals(tree.getName())) {
102           return describeForVarDecl(tree, state);
103         }
104         return Description.NO_MATCH;
105       }
106     
107       /**
108        * If the given expression is a call to checkNotNull(x), returns x.
109        * Otherwise, returns the given expression.
110        *
111        * TODO(eaftan): Also match calls to Java 7's Objects.requireNonNull() method.
112        */
113       private ExpressionTree stripCheckNotNull(ExpressionTree expression, VisitorState state) {
114         if (expression != null
115             && expression.getKind() == METHOD_INVOCATION
116             && staticMethod().onClass("com.google.common.base.Preconditions").named("checkNotNull")
117                 .matches(expression, state)) {
118           return ((MethodInvocationTree) expression).getArguments().get(0);
119         }
120         return expression;
121       }
122     
123       public Description describeForVarDecl(VariableTree tree, VisitorState state) {
124         String varDeclStr = tree.toString();
125         int equalsIndex = varDeclStr.indexOf('=');
126         if (equalsIndex < 0) {
127           throw new IllegalStateException(
128               "Expected variable declaration to have an initializer: " + tree);
129         }
130         varDeclStr = varDeclStr.substring(0, equalsIndex - 1) + ";";
131     
132         // Delete the initializer but still declare the variable.
133         return describeMatch(tree, SuggestedFix.replace(tree, varDeclStr));
134       }
135     
136       /**
137        * We expect that the lhs is a field and the rhs is an identifier, specifically
138        * a parameter to the method.  We base our suggested fixes on this expectation.
139        *
140        * Case 1: If lhs is a field and rhs is an identifier, find a method parameter
141        * of the same type and similar name and suggest it as the rhs.  (Guess that they
142        * have misspelled the identifier.)
143        *
144        * Case 2: If lhs is a field and rhs is not an identifier, find a method parameter
145        * of the same type and similar name and suggest it as the rhs.
146        *
147        * Case 3: If lhs is not a field and rhs is an identifier, find a class field
148        * of the same type and similar name and suggest it as the lhs.
149        *
150        * Case 4: Otherwise suggest deleting the assignment.
151        */
152       public Description describeForAssignment(AssignmentTree assignmentTree, VisitorState state) {
153     
154         // the statement that is the parent of the self-assignment expression
155         Tree parent = state.getPath().getParentPath().getLeaf();
156     
157         // default fix is to delete assignment
158         Fix fix = SuggestedFix.delete(parent);
159     
160         ExpressionTree lhs = assignmentTree.getVariable();
161         ExpressionTree rhs = assignmentTree.getExpression();
162     
163         // if this is a method invocation, they must be calling checkNotNull()
164         if (assignmentTree.getExpression().getKind() == METHOD_INVOCATION) {
165           // change the default fix to be "checkNotNull(x)" instead of "x = checkNotNull(x)"
166           fix = SuggestedFix.replace(assignmentTree, rhs.toString());
167           // new rhs is first argument to checkNotNull()
168           rhs = stripCheckNotNull(rhs, state);
169         }
170     
171     
172         if (lhs.getKind() == MEMBER_SELECT) {
173           // find a method parameter of the same type and similar name and suggest it
174           // as the rhs
175     
176           // rhs should be either identifier or field access
177           assert(rhs.getKind() == IDENTIFIER || rhs.getKind() == MEMBER_SELECT);
178     
179           // get current name of rhs
180           String rhsName = null;
181           if (rhs.getKind() == IDENTIFIER) {
182             rhsName = ((JCIdent) rhs).name.toString();
183           } else if (rhs.getKind() == MEMBER_SELECT) {
184             rhsName = ((JCFieldAccess) rhs).name.toString();
185           }
186     
187           // find method parameters of the same type
188           Type type = ((JCFieldAccess) lhs).type;
189           TreePath path = state.getPath();
190           while (path != null && path.getLeaf().getKind() != METHOD) {
191             path = path.getParentPath();
192           }
193           JCMethodDecl method = (JCMethodDecl) path.getLeaf();
194           int minEditDistance = Integer.MAX_VALUE;
195           String replacement = null;
196           for (JCVariableDecl var : method.params) {
197             if (var.type == type) {
198               int editDistance = EditDistance.getEditDistance(rhsName, var.name.toString());
199               if (editDistance < minEditDistance) {
200                 // pick one with minimum edit distance
201                 minEditDistance = editDistance;
202                 replacement = var.name.toString();
203               }
204             }
205           }
206           if (replacement != null) {
207             // suggest replacing rhs with the parameter
208             fix = SuggestedFix.replace(rhs, replacement);
209           }
210         } else if (rhs.getKind() == IDENTIFIER) {
211           // find a field of the same type and similar name and suggest it as the lhs
212     
213           // lhs should be identifier
214           assert(lhs.getKind() == IDENTIFIER);
215     
216           // get current name of lhs
217           String lhsName = ((JCIdent) rhs).name.toString();
218     
219           // find class instance fields of the same type
220           Type type = ((JCIdent) lhs).type;
221           TreePath path = state.getPath();
222           while (path != null && !(path.getLeaf() instanceof JCClassDecl)) {
223             path = path.getParentPath();
224           }
225           if (path == null) {
226             throw new IllegalStateException("Expected to find an enclosing class declaration");
227           }
228           JCClassDecl klass = (JCClassDecl) path.getLeaf();
229           int minEditDistance = Integer.MAX_VALUE;
230           String replacement = null;
231           for (JCTree member : klass.getMembers()) {
232             if (member.getKind() == VARIABLE) {
233               JCVariableDecl var = (JCVariableDecl) member;
234               if (!Flags.isStatic(var.sym) && var.type == type) {
235                 int editDistance = EditDistance.getEditDistance(lhsName, var.name.toString());
236                 if (editDistance < minEditDistance) {
237                   // pick one with minimum edit distance
238                   minEditDistance = editDistance;
239                   replacement = var.name.toString();
240                 }
241               }
242             }
243           }
244           if (replacement != null) {
245             // suggest replacing lhs with the field
246             fix = SuggestedFix.replace(lhs, "this." + replacement);
247           }
248         }
249     
250         return describeMatch(assignmentTree, fix);
251       }
