17     package com.google.errorprone.refaster;
18     
19     import com.google.auto.value.AutoValue;
20     
21     import javax.annotation.Nullable;
22     import javax.lang.model.element.Name;
23     
24     /**
25      * A simple wrapper to view a {@code String} as a {@code Name}.
26      *
27      * @author lowasser@google.com (Louis Wasserman)
28      */
29     @AutoValue
30     public abstract class StringName 
31         implements Name, Unifiable<Name>, Inlineable<com.sun.tools.javac.util.Name> {
32       public static StringName of(CharSequence contents) {
33         return new AutoValue_StringName(contents.toString());
34       }
35       
36       abstract String contents();
37       
38       @Override
39       public String toString() {
40         return contents();
41       }
42     
43       @Override
44       public int length() {
45         return contents().length();
46       }
47     
48       @Override
49       public char charAt(int index) {
50         return contents().charAt(index);
51       }
52     
53       @Override
54       public CharSequence subSequence(int beginIndex, int endIndex) {
55         return contents().subSequence(beginIndex, endIndex);
56       }
57     
58       @Override
59       public boolean contentEquals(@Nullable CharSequence cs) {
60         return cs != null && contents().contentEquals(cs);
61       }
62     
63       @Override
64       public Choice<Unifier> unify(@Nullable Name target, Unifier unifier) {
65         return Choice.condition(contentEquals(target), unifier);
66       }
67     
68       @Override
69       public com.sun.tools.javac.util.Name inline(Inliner inliner) {
70         return inliner.asName(contents());
71       }
72     }
