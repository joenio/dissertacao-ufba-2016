17     package com.google.errorprone.dataflow.nullnesspropagation;
18     
19     import com.google.common.base.Predicate;
20     import com.google.errorprone.dataflow.DataFlow;
21     
22     import com.sun.source.util.TreePath;
23     import com.sun.tools.javac.util.Context;
24     
25     import java.io.Serializable;
26     import java.util.List;
27     
28     /**
29      * An interface to the nullness analysis.
30      */
31     public final class NullnessAnalysis implements Serializable {
32     
33       /**
34        * Represents a Java method.  Used for custom predicates to match non-null-returning methods.
35        */
36       public interface MethodInfo {
37         String clazz();
38         String method();
39         List<String> annotations();
40         boolean isStatic();
41         boolean isPrimitive();
42       }
43     
44       private final NullnessPropagationTransfer nullnessPropagation;
45     
46       /**
47        * Initializes a nullness analysis with a built-in set of non-null returning methods.
48        */
49       public NullnessAnalysis() {
50         nullnessPropagation = new NullnessPropagationTransfer();
51       }
52     
53       /**
54        * Initializes a nullness analysis with an additional set of non-null returning methods that
55        * are or'ed with the built-in set of non-null returning methods.
56        *
57        * @param additionalNonNullReturningMethods A predicate matching methods that never return null
58        */
59       public NullnessAnalysis(Predicate<MethodInfo> additionalNonNullReturningMethods) {
60         nullnessPropagation = new NullnessPropagationTransfer(additionalNonNullReturningMethods);
61       }
62     
63       /**
64        * Returns the {@link Nullness} of the leaf of {@code exprPath}.
65        *
66        * <p>If the leaf required the compiler to generate autoboxing or autounboxing calls,
67        * {@code getNullness} returns the {@code Nullness} <i>after</i> the boxing/unboxing. This implies
68        * that, in those cases, it will always return {@code NONNULL}.
69        */
70       public Nullness getNullness(TreePath exprPath, Context context) {
71         try {
72           nullnessPropagation.setContext(context).setCompilationUnit(exprPath.getCompilationUnit());
73           return DataFlow.expressionDataflow(exprPath, context, nullnessPropagation);
74         } finally {
75           nullnessPropagation.setContext(null).setCompilationUnit(null);
76         }
77       }
78     }
