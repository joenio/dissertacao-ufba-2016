40     /** @author cushon@google.com (Liam Miller-Cushon) */
41     @BugPattern(
42       name = "EmptyTopLevelDeclaration",
43       summary = "Empty top-level type declaration",
44       explanation =
45           "A semi-colon at the top level of a Java file is treated as an empty type declaration"
46               + " in the grammar, but it's confusing and unnecessary.",
47       category = JDK,
48       severity = WARNING,
49       maturity = MATURE
50     )
51     public class EmptyTopLevelDeclaration extends BugChecker implements CompilationUnitTreeMatcher {
52     
53       @Override
54       public Description matchCompilationUnit(CompilationUnitTree tree, VisitorState state) {
55         List<Tree> toDelete = new ArrayList<>();
56         for (Tree member : tree.getTypeDecls()) {
57           if (member.getKind() == Tree.Kind.EMPTY_STATEMENT) {
58             toDelete.add(member);
59           }
60         }
61         if (toDelete.isEmpty()) {
62           return Description.NO_MATCH;
63         }
64         SuggestedFix.Builder fixBuilder = SuggestedFix.builder();
65         for (Tree member : toDelete) {
66           fixBuilder.delete(member);
67         }
68         return describeMatch(toDelete.get(0), fixBuilder.build());
69       }
70     }
