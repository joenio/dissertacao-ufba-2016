17     package com.google.errorprone.bugpatterns;
18     
19     import static com.google.errorprone.BugPattern.Category.JDK;
20     import static com.google.errorprone.BugPattern.MaturityLevel.MATURE;
21     import static com.google.errorprone.BugPattern.SeverityLevel.ERROR;
22     import static com.google.errorprone.bugpatterns.BugChecker.LiteralTreeMatcher;
23     
24     import com.google.errorprone.BugPattern;
25     import com.google.errorprone.VisitorState;
26     import com.google.errorprone.fixes.Fix;
27     import com.google.errorprone.fixes.SuggestedFix;
28     import com.google.errorprone.matchers.Description;
29     import com.google.errorprone.matchers.Matcher;
30     
31     import com.sun.source.tree.LiteralTree;
32     import com.sun.source.tree.Tree.Kind;
33     import com.sun.tools.javac.tree.JCTree.JCLiteral;
34     
35     import java.util.regex.Pattern;
36     
37     /**
38      * Matcher for a <code>long</code> literal with a lower-case ell for a suffix (e.g.
39      * <code>234l</code>) rather than the more readable upper-case ell (e.g. <code>234L</code>).
40      *
41      * @author Simon Nickerson (sjnickerson@google.com)
42      */
43     @BugPattern(name = "LongLiteralLowerCaseSuffix",
44         summary = "Prefer 'L' to 'l' for the suffix to long literals",
45         explanation = "A long literal can have a suffix of 'L' or 'l', but the former is less " +
46         "likely to be confused with a '1' in most fonts.",
47         category = JDK, severity = ERROR, maturity = MATURE)
48     public class LongLiteralLowerCaseSuffix extends BugChecker implements LiteralTreeMatcher {
49     
50       private static final Matcher<LiteralTree> matcher = new Matcher<LiteralTree>() {
51         @Override
52         public boolean matches(LiteralTree literalTree, VisitorState state) {
53           if (literalTree.getKind() == Kind.LONG_LITERAL) {
54             // The javac AST doesn't seem to record whether the suffix is present, or whether it's
55             // an 'l' or 'L'. We have to look at the original source
56             String longLiteral = getLongLiteral(literalTree, state);
57             return longLiteral != null && longLiteral.endsWith("l");
58           } else {
59             return false;
60           }
61         }
62       };
63     
64       // Doesn't need to be strict, just shouldn't read past the end
65       // of the literal.
66       private static final Pattern LONG_LITERAL_PATTERN =
67           Pattern.compile("-? *(0[bBxX]?)?[0-9a-fA-F_]+[lL]?");
68     
69       /**
70        * Extracts the long literal corresponding to a given {@link LiteralTree} node from the source
71        * code as a string. Returns null if the source code is not available.
72        */
73       private static String getLongLiteral(LiteralTree literalTree, VisitorState state) {
74         JCLiteral longLiteral = (JCLiteral) literalTree;
75         CharSequence sourceFile = state.getSourceCode();
76         if (sourceFile == null) {
77           return null;
78         }
79         int start = longLiteral.getStartPosition();
80         java.util.regex.Matcher matcher = LONG_LITERAL_PATTERN.matcher(
81             sourceFile.subSequence(start, sourceFile.length()));
82         if (matcher.lookingAt()) {
83           return matcher.group();
84         }
85         return null;
86       }
87     
88       @Override
89       public Description matchLiteral(LiteralTree literalTree, VisitorState state) {
90         if (!matcher.matches(literalTree, state)) {
91           return Description.NO_MATCH;
92         }
93         StringBuilder longLiteral = new StringBuilder(getLongLiteral(literalTree, state));
94         longLiteral.setCharAt(longLiteral.length() - 1, 'L');
95         Fix fix = SuggestedFix.replace(literalTree, longLiteral.toString());
96         return describeMatch(literalTree, fix);
97       }
98     }
