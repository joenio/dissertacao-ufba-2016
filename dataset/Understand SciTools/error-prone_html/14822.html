17     package com.google.errorprone.matchers;
18     
19     import com.google.common.collect.ImmutableList;
20     import com.google.errorprone.VisitorState;
21     import com.google.errorprone.util.ASTHelpers;
22     
23     import com.sun.source.tree.ClassTree;
24     import com.sun.source.tree.MethodTree;
25     import com.sun.source.tree.Tree;
26     
27     /**
28      * Applies the given matcher to the constructor(s) of the given class.
29      *
30      * @author eaftan@google.com (Eddie Aftandilian)
31      */
32     public class ConstructorOfClass extends ChildMultiMatcher<ClassTree, MethodTree> {
33     
34       public ConstructorOfClass(MatchType matchType, Matcher<MethodTree> nodeMatcher) {
35         super(matchType, nodeMatcher);
36       }
37     
38       @Override
39       protected Iterable<? extends MethodTree> getChildNodes(ClassTree classTree, VisitorState state) {
40         ImmutableList.Builder<MethodTree> result = ImmutableList.builder();
41         // Iterate over members of class (methods and fields).
42         for (Tree member : classTree.getMembers()) {
43           // If this member is a constructor...
44           if (member instanceof MethodTree && ASTHelpers.getSymbol(member).isConstructor()) {
45             result.add((MethodTree) member);
46           }
47         }
48         return result.build();
49       }
50     }
