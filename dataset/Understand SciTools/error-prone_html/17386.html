17     package com.google.errorprone.matchers;
18     
19     import com.google.errorprone.VisitorState;
20     import com.sun.source.tree.ExpressionTree;
21     import com.sun.tools.javac.code.Type;
22     import com.sun.tools.javac.tree.JCTree.JCNewClass;
23     
24     /**
25      * Matcher for a new instance of an anonymous inner class.
26      * 
27      * @author sjnickerson@google.com (Simon Nickerson)
28      */
29     public class NewInstanceAnonymousInnerClass implements Matcher<ExpressionTree> {
30       
31       /**
32        * Fully-qualified name of the class or interface we are subclassing for our
33        * anonymous inner class, e.g. a new Comparator
34        */
35       private final String superClass;
36       
37       public NewInstanceAnonymousInnerClass(String superClass) {
38         this.superClass = superClass;
39       }
40       
41       @Override
42       public boolean matches(ExpressionTree t, VisitorState state) {
43         if (t instanceof JCNewClass) {
44           JCNewClass invocation = (JCNewClass)t;
45           Type type = invocation.getIdentifier().type;
46           if (type.tsym.getQualifiedName().contentEquals(superClass)) {
47             return true;
48           }
49         }
50         return false;
51       }
52     }
