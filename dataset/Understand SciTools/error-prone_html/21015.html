32     /**
33      * Tests for {@code DescriptionBasedDiff}.
34      *
35      * @author lowasser@google.com (Louis Wasserman)
36      */
37     @RunWith(JUnit4.class)
38     public class DescriptionBasedDiffTest extends CompilerBasedTest {
39       private JCCompilationUnit compilationUnit;
40     
41       private static final String[] lines = {
42         "package foo.bar;",
43         "import com.foo.Bar;",
44         "",
45         "class Foo {",
46         "  public static void main(String[] args) {",
47         "    System.out.println(\"foo\");",
48         "  }", 
49         "}"};
50     
51       @Before
52       public void setUp() {
53         compile(lines);
54         compilationUnit = Iterables.getOnlyElement(compilationUnits);
55       }
56     
57       @Test
58       public void noDiffs() {
59         DescriptionBasedDiff diff = DescriptionBasedDiff.create(compilationUnit);
60         diff.applyDifferences(sourceFile);
61         assertThat(sourceFile.getLines()).containsExactly((Object[]) lines).inOrder();
62       }
63     
64       @Test
65       public void oneDiff() {
66         DescriptionBasedDiff diff = DescriptionBasedDiff.create(compilationUnit);
67         diff.onDescribed(new Description(null, "message", SuggestedFix.replace(117, 120, "bar"),
68             SeverityLevel.NOT_A_PROBLEM));
69         diff.applyDifferences(sourceFile);
70         assertThat(sourceFile.getLines())
71             .containsExactly(
72                 "package foo.bar;",
73                 "import com.foo.Bar;",
74                 "",
75                 "class Foo {",
76                 "  public static void main(String[] args) {",
77                 "    System.out.println(\"bar\");",
78                 "  }",
79                 "}")
80             .inOrder();
81       }
82       
83       @Test
84       public void prefixDiff() {
85         DescriptionBasedDiff diff = DescriptionBasedDiff.create(compilationUnit);
86         diff.onDescribed(new Description(null, "message", SuggestedFix.replace(120, 120, "bar"),
87             SeverityLevel.NOT_A_PROBLEM));
88         diff.applyDifferences(sourceFile);
89         assertThat(sourceFile.getLines())
90             .containsExactly(
91                 "package foo.bar;",
92                 "import com.foo.Bar;",
93                 "",
94                 "class Foo {",
95                 "  public static void main(String[] args) {",
96                 "    System.out.println(\"foobar\");",
97                 "  }",
98                 "}")
99             .inOrder();
100       }
101     
102     
103       @Test
104       public void twoDiffs() {
105         DescriptionBasedDiff diff = DescriptionBasedDiff.create(compilationUnit);
106         diff.onDescribed(new Description(null, "message", 
107             SuggestedFix.builder()
108                 .replace(104, 107, "longer")
109                 .replace(117, 120, "bar")
110                 .build(),
111             SeverityLevel.NOT_A_PROBLEM));
112         diff.applyDifferences(sourceFile);
113         assertThat(sourceFile.getLines())
114             .containsExactly(
115                 "package foo.bar;",
116                 "import com.foo.Bar;",
117                 "",
118                 "class Foo {",
119                 "  public static void main(String[] args) {",
120                 "    System.longer.println(\"bar\");",
121                 "  }",
122                 "}")
123             .inOrder();
124       }

126       @Test
127       public void addImport() {
128         DescriptionBasedDiff diff = DescriptionBasedDiff.create(compilationUnit);
129         diff.onDescribed(new Description(null, "message", 
130             SuggestedFix.builder().addImport("com.google.foo.Bar").build(),
131             SeverityLevel.NOT_A_PROBLEM));
132         diff.applyDifferences(sourceFile);
133         assertThat(sourceFile.getLines())
134             .containsExactly(
135                 "package foo.bar;",
136                 "import com.google.foo.Bar;",
137                 "",
138                 "import com.foo.Bar;",
139                 "",
140                 "class Foo {",
141                 "  public static void main(String[] args) {",
142                 "    System.out.println(\"foo\");",
143                 "  }",
144                 "}")
145             .inOrder();
146       }

168       @Test
169       public void twoDiffsWithImport() {
170         DescriptionBasedDiff diff = DescriptionBasedDiff.create(compilationUnit);
171         diff.onDescribed(new Description(null, "message", 
172             SuggestedFix.builder()
