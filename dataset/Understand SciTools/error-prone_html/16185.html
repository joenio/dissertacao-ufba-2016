57     /**
58      * Bug checker for equals methods that don't actually override equals.
59      */
60     @BugPattern(name = "NonOverridingEquals",
61         summary = "equals method doesn't override Object.equals",
62         category = JDK, maturity = MATURE, severity = WARNING)
63     public class NonOverridingEquals extends BugChecker implements MethodTreeMatcher {
64     
65       private static final String MESSAGE_BASE = "equals method doesn't override Object.equals";
66     
67       /**
68        * Matches any method definition that:
69        * 1) is named `equals`
70        * 2) takes a single argument of a type other than Object
71        * 3) returns a boolean or Boolean
72        */
73       private static final Matcher<MethodTree> MATCHER = allOf(
74           methodIsNamed("equals"),
75           methodHasParameters(variableType(not(isSameType("java.lang.Object")))),
76           anyOf(methodReturns(BOOLEAN_TYPE), methodReturns(JAVA_LANG_BOOLEAN_TYPE)));
77     
78       /**
79        * Matches if the enclosing class overrides Object#equals.
80        */
81       private static final Matcher<MethodTree> enclosingClassOverridesEquals =
82           enclosingClass(hasMethod(allOf(
83               methodIsNamed("equals"),
84               methodReturns(BOOLEAN_TYPE),
85               methodHasParameters(variableType(isSameType(OBJECT_TYPE))),
86               not(isStatic()))));
87     
88       /**
89        * Matches method declarations for which we cannot provide a fix.  Our default fix rewrites
90        * the equals method to override Object.equals.  In these (uncommon) cases, our rewrite
91        * algorithm doesn't work:
92        * <ul>
93        * <li>the method is static
94        * <li>the method is not public
95        * <li>the method returns a boxed Boolean
96        * </ul>
97        */
98       private static final Matcher<MethodTree> noFixMatcher = anyOf(
99           isStatic(),
100           not(methodHasVisibility(Visibility.PUBLIC)),
101           methodReturns(JAVA_LANG_BOOLEAN_TYPE));
102     
103       @Override
104       public Description matchMethod(MethodTree methodTree, VisitorState state) {
105         if (!MATCHER.matches(methodTree, state)) {
106           return Description.NO_MATCH;
107         }
108     
109         // If an overriding equals method has already been defined in the enclosing class, assume
110         // this is a type-specific helper method and give advice to either inline it or rename it.
111         if (enclosingClassOverridesEquals.matches(methodTree, state)) {
112           return buildDescription(methodTree)
113               .setMessage(MESSAGE_BASE + "; if this is a type-specific helper for a method that does "
114                   + "override Object.equals, either inline it into the callers or rename it to avoid "
115                   + "ambiguity in overload resolution")
116               .build();
117         }
118     
119         // Don't provide a fix if the method is static, non-public, or returns a boxed Boolean
120         if (noFixMatcher.matches(methodTree, state)) {
121           return describeMatch(methodTree);
122         }
123     
124         JCClassDecl cls = (JCClassDecl) state.findEnclosing(ClassTree.class);
125     
126         if ((cls.getModifiers().flags & ENUM) != 0) {
127           /* If the enclosing class is an enum, then just delete the equals method since enums
128            * should always be compared for reference equality. Enum defines a final equals method for
129            * just this reason. */
130           return buildDescription(methodTree)
131               .setMessage(MESSAGE_BASE + "; enum instances can safely be compared by reference "
132                   + "equality, so please delete this")
133               .addFix(SuggestedFix.delete(methodTree))
134               .build();
135         } else {
136           /* Otherwise, change the covariant equals method to override Object.equals. */
137     
138           SuggestedFix.Builder fix = SuggestedFix.builder();
139     
140           // Add @Override annotation if not present.
141           if (ASTHelpers.getAnnotation(methodTree, Override.class) == null) {
142             fix.prefixWith(methodTree, "@Override\n");
143           }
144     
145           // Change method signature, substituting Object for parameter type.
146           JCTree parameterType = (JCTree) methodTree.getParameters().get(0).getType();
147           Name parameterName = ((JCVariableDecl) methodTree.getParameters().get(0)).getName();
148           fix.replace(parameterType, "Object");
149     
150           // If there is a method body...
151           if (methodTree.getBody() != null) {
152     
153             // Add type check at start
154             String typeCheckStmt = "if (!(" + parameterName + " instanceof " + parameterType + ")) {\n"
155                 + "  return false;\n"
156                 + "}\n";
157             fix.prefixWith(methodTree.getBody().getStatements().get(0), typeCheckStmt);
158     
159             // Cast all uses of the parameter name using a recursive TreeScanner.
160             new CastScanner().scan(methodTree.getBody(), new CastState(parameterName,
161                 parameterType.toString(), fix));
162           }
163     
164           return describeMatch(methodTree, fix.build());
165         }
166     
167       }
168     
169       private static class CastState {
170         final Name name;
171         final String castToType;
172         final SuggestedFix.Builder fix;
173     
174         public CastState(Name name, String castToType, SuggestedFix.Builder fix) {
175           this.name = name;
176           this.castToType = castToType;
177           this.fix = fix;
178         }
179       }
180     
181       /**
182        * A Scanner used to replace all references to a variable with
183        * a casted version.
184        */
185       private static class CastScanner extends TreeScanner<Void, CastState> {
186         @Override
187         public Void visitIdentifier(IdentifierTree node, CastState state) {
188           if (state.name.equals(node.getName())) {
189             state.fix.replace(node, "((" + state.castToType + ") " + state.name + ")");
190           }
191     
192           return super.visitIdentifier(node, state);
193         }
194       }
195     }
