19     import java.io.File;

21     /**
22      * @author cpovirk@google.com (Chris Povirk)
23      */
24     public class ChainingConstructorIgnoresParameterNegativeCases {
25       static class ImplicitThisCall {
26         ImplicitThisCall() {}
27     
28         ImplicitThisCall(String foo) {}
29       }
30     
31       static class ExplicitNoArgThisCall {
32         ExplicitNoArgThisCall() {}
33     
34         ExplicitNoArgThisCall(String foo) {
35           this();
36         }
37       }
38     
39       static class ParameterNotAvailable {
40         ParameterNotAvailable(String foo, boolean bar) {}
41     
42         ParameterNotAvailable(String foo) {
43           this(foo, false);
44         }
45       }
46     
47       static class ParameterDifferentType {
48         ParameterDifferentType(File foo) {}
49     
50         ParameterDifferentType(String foo) {
51           this(new File("/tmp"));
52         }
53       }
54     
55       static class ParameterUsedInExpression {
56         ParameterUsedInExpression(String foo, boolean bar) {}
57     
58         ParameterUsedInExpression(String foo) {
59           this(foo.substring(0), false);
60         }
61       }
62     
63       /** Make sure that we don't confuse a nested class's constructor with the containing class's. */
64       static class HasNestedClass {
65         HasNestedClass(String foo) {
66           this("somethingElse", false);
67         }
68     
69         static class NestedClass {
70           NestedClass(String foo, boolean bar) {}
71         }
72     
73         HasNestedClass(String notFoo, boolean bar) {}
74       }
75     
76       static class HasNestedClassesWithSameName {
77         static class Outer1 {
78           static class Inner {
79             Inner(String foo, boolean bar) {}
80           }
81         }
82     
83         static class Outer2 {
84           static class Inner {
85             Inner(String foo) {
86               this("somethingElse", false);
87             }
88     
89             Inner(String notFoo, boolean bar) {}
90           }
91         }
92       }
93     
94       class NonStaticClass {
95         NonStaticClass(String foo, boolean bar) {}
96     
97         NonStaticClass(String foo) {
98           this(foo, false);
99         }
100       }
101     
102       static class Varargs1 {
103         Varargs1(String foo, boolean... bar) {}
104     
105         Varargs1() {
106           this("something", false, false);
107         }
108       }
109     
110       static class Varargs2 {
111         Varargs2(String foo, boolean... bar) {}
112     
113         Varargs2() {
114           this("something");
115         }
116       }
117     }
