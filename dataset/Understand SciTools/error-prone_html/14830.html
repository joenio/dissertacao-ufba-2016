17     package com.google.errorprone.bugpatterns.threadsafety;
18     
19     import org.pcollections.Empty;
20     import org.pcollections.PSet;
21     
22     import java.util.Collection;
23     
24     import javax.annotation.CheckReturnValue;
25     
26     /**
27      * A set of held locks.
28      *
29      * <p>Wrapper around a {@link PSet} of {@link GuardedByExpression}s. Using a persistent collection
30      * makes it easy to handle adding locks to the set only while visiting the scope where those
31      * locks are held, without mutating the underlying collection.
32      *
33      * @author cushon@google.com (Liam Miller-Cushon)
34      */
35     class HeldLockSet {
36     
37       final PSet<GuardedByExpression> locks;
38     
39       private HeldLockSet() {
40         this(Empty.<GuardedByExpression>set());
41       }
42     
43       private HeldLockSet(PSet<GuardedByExpression> locks) {
44         this.locks = locks;
45       }
46     
47       static HeldLockSet empty() {
48         return new HeldLockSet();
49       }
50     
51       @CheckReturnValue
52       public HeldLockSet plus(GuardedByExpression lock) {
53         return new HeldLockSet(locks.plus(lock));
54       }
55     
56       @CheckReturnValue
57       public HeldLockSet plusAll(Collection<GuardedByExpression> locks) {
58         return new HeldLockSet(this.locks.plusAll(locks));
59       }
60     
61       public Collection<GuardedByExpression> allLocks() {
62         return locks;
63       }
64     
65       @Override
66       public String toString() {
67         return locks.toString();
68       }
69     }
