54     unsigned int string_hash_function(byte* p) { 
55       unsigned int h = 0, g;
56       while(*p) { 
57         h = (h << 4) + *p++;
58         if ((g = h & 0xF0000000) != 0) { 
59           h ^= g >> 24;
60         }
61         h &= ~g;
62       }
63       return h;
64     }

78     int get_number_of_parameters(utf_string const& str)
79     {
80       const char* p = str.as_asciz();
81       assert(*p++ =='(');
82       int n_params = 0;
83       while (*p != ')') { 
84         switch (*p++) { 
85         case '[':
86           while (*p == '[') p += 1;
87           if (*p++ == 'L') { 
88             while (*p++ != ';');
89           }
90           n_params += 1;
91           break;
92         case 'D':
93         case 'J':
94           n_params += 2;
95           break;
96         case 'L':
97           while (*p++ != ';');
98           nobreak;
99         default:
100           n_params += 1;
101         }
102       }
103       return n_params;
104     }

106     int get_type(utf_string const& str) 
107     { 
108       const char* p = str.as_asciz();
109       if (*p == '(') { 
110         while (*++p != ')'); 
111         p += 1;
112       }
113       int indirect = 0;
114       while (*p == '[') { 
115         p += 1;
116         indirect += 1;
117       }
118       type_tag tag = tp_object;
119       switch (*p) { 
120       case 'I': tag = tp_int; break;
121       case 'S': tag = tp_short; break;
122       case 'D': tag = tp_double; break;
123       case 'J': tag = tp_long; break;
124       case 'F': tag = tp_float; break;
125       case 'V': tag = tp_void; break;
126       case 'B': tag = tp_byte; break;
127       case 'C': tag = tp_char; break;
128       case 'Z': tag = tp_bool; break;
129       default:
130         if (strcmp(p, "Ljava/lang/String;") == 0) tag = tp_string;
131       }
132       return int(tag) + (indirect << 8);
133     }

139     void format_message(int code, utf_string const& file, int line, __VALIST ap)
140     {
141       static int loop_id;
142       static message_node *first, *last;
143       static char* compound_message;
144       const void* parameter[MAX_MSG_PARAMETERS];
145       int   n_parameters = 2;
146       
147       if (code == msg_loop || code == msg_sync_loop) { // extract loop identifier
148         parameter[n_parameters++] = va_arg(ap, void*);
149       }
150       if (history != NULL) { 
151         if (compound_message != NULL
152             && ((loop_id != 0 
153                  && ((code != msg_loop && code != msg_sync_loop)
154                      || (int)(long)parameter[2] != loop_id))
155                 || (loop_id == 0 && code != msg_wait_path)))
156           {
157             if (!message_node::find(compound_message)) { 
158               message_node *msg = first, *next;
159               do { 
160                 next = msg->next;
161                 fprintf(stdout, "%s\n", msg->text);
162                 delete msg;
163                 n_messages += 1;
164               } while ((msg = next) != NULL);
165               fprintf(history, "%s\n", compound_message);  
166             }
167             delete[] compound_message;
168             compound_message = NULL;
169           }
170       }
171       if ((reported_message_mask & msg_table[code].category) 
172           && msg_table[code].enabled)
173         {
174           static int prev_line = 0;
175       
176           char  msg_buf[MAX_MSG_LENGTH];
177           char  his_buf[MAX_MSG_LENGTH];
178           char* hp = his_buf;
179       
180           if (line == 0) { 
181             line = ++prev_line; // avoid several messages with 0 line number
182           } else { 
183             prev_line = 0;
184           }
185           parameter[0] = file.as_asciz();
186           parameter[1] = (void*)line;
187     
188           if (history) { 
189             hp += sprintf(hp, "%s", msg_table[code].name);
190           }
191           
192           char const* src = msg_table[code].format;
193           char* dst = msg_buf;
194     
195           if (code == msg_done) { 
196             // do not output location prefix for termination message
197             src += strlen(MSG_LOCATION_PREFIX);
198             parameter[0] = (void*)n_messages;
199           }
200           while (*src != 0) { 
201             if (*src == '%') { 
202               int index;
203               int pos;
204               int n = sscanf(++src, "%d%n", &index, &pos); 
205               assert(n == 1);
206               assert(index < MAX_MSG_PARAMETERS);
207               while (index >= n_parameters) { 
208                 parameter[n_parameters++] = va_arg(ap, void*);
209               }
210               src += pos;
211               char* save_dst = dst;
212               switch (*src++) {
213               case 's': // zero terminated string
214                 dst += sprintf(dst, "%s", (char*)parameter[index]);  
215                 break;
216               case 'm': // method descriptor
217                 dst += ((method_desc*)parameter[index])->
218                   demangle_method_name(dst);
219                 break;
220               case 'u': // utf8 string
221                 dst += sprintf(dst, "%s", 
222                                ((utf_string*)parameter[index])->as_asciz()); 
223                 break;
224               case 'c': // class descriptor
225                 dst += sprintf(dst, "%s", ((class_desc*)parameter[index])->
226                                name.as_asciz()); 
227                 break;
228               case 'd': // integer
229                 dst += sprintf(dst, "%d", (int)(long)parameter[index]);  
230                 break;
231               default:
232                 assert(false/*bad message parameter format*/);
233               }
234               if (history) { 
235                 // append parameeter to history buffer
236                 if ((index >= 2 || msg_table[code].position_dependent)
237                     && (code != msg_sync_loop || index > 2) 
238                     && (code != msg_loop || index > 3))
239                   {
240                     // Do not inlude loop number in history message
241                     hp += sprintf(hp, ":%.*s", (int)(dst - save_dst), save_dst);
242                   }
243               }
244             } else { 
245               *dst++ = *src++;
246             }
247           }
248           *dst++ = '.';
249           *dst = 0;
250           if (history != NULL) { 
251             if (compound_message != NULL) { 
252               char* new_msg = new char[strlen(compound_message)
253                                       +strlen(his_buf)+2];
254               sprintf(new_msg, "%s;%s", compound_message, his_buf);
255               last = last->next = new message_node(msg_buf);
256               delete[] compound_message;
257               compound_message = new_msg;
258             } else { 
259               if (code == msg_loop || code == msg_sync_loop 
260                   || code == msg_wait) 
261                 { 
262                   compound_message = strdup(his_buf);
263                   first = last = new message_node(msg_buf);
264                   if (code != msg_wait) { 
265                     loop_id = (int)(long)parameter[2];
266                   }
267                 } else if (!message_node::find(his_buf)) { 
268                   fprintf(stdout, "%s\n", msg_buf);
269                   if (code != msg_done) { 
270                     fprintf(history, "%s\n", his_buf);
271                     n_messages += 1;
272                   }
273                 }
274             }
275           } else { 
276             fprintf(stdout, "%s\n", msg_buf);
277             if (code != msg_done) { 
278               n_messages += 1;
279             }
280           }
281         } 
282     }

295     void set_class_source_path(class_desc* cls)
296     {
297       if (source_file_path_len != 0) { 
298         const char* class_file_name = cls->source_file.as_asciz();
299         if (!source_path_redefined) { 
300           const char* dirend = strrchr(class_file_name, '/');
301           if (dirend != NULL) { 
302             int dirlen = dirend - class_file_name;
303             if (dirlen <= source_file_path_len
304                 && memcmp(class_file_name, 
305                           source_file_path + source_file_path_len - dirlen,
306                           dirlen) == 0)
307               {
308                 class_file_name = dirend+1;
309               }
310           }
311         }
312         char file_name_buf[MAX_MSG_LENGTH];
313         int len = sprintf(file_name_buf, "%.*s%c%s", 
314                           source_file_path_len, source_file_path, 
315                           FILE_SEP, class_file_name);
316         cls->source_file = utf_string(len, (byte*)file_name_buf);
317       } 
318     }

320     bool parse_class_file(byte* fp)
321     {
322       int i;
323       unsigned magic = unpack4(fp); 
324       fp += 4;
325       if (magic != 0xCAFEBABE) return false;
326       int minor_version = unpack2(fp);
327       fp += 2;
328       int major_version = unpack2(fp);
329       fp += 2;
330       int constant_pool_count = unpack2(fp);
331       fp += 2;
332       constant** constant_pool = new constant*[constant_pool_count];
333       memset(constant_pool, 0, sizeof(constant*)*constant_pool_count);
334     
335       int name_index = 0;
336       for (i = 1; i < constant_pool_count; i++) { 
337         constant* cp = NULL;
338         int n_extra_cells = 0;
339         switch (*fp) { 
340         case c_utf8:
341           cp = new const_utf8(fp);
342           if (!strcmp(((const_utf8*)cp)->as_asciz(), "this")) {
343             name_index = i;
344           }
345           break;
346         case c_integer:
347           cp = new const_int(fp);
348           break;
349         case c_float:
350           cp = new const_float(fp);
351           break;
352         case c_long:
353           cp = new const_long(fp);
354           n_extra_cells += 1;
355           break;
356         case c_double:
357           cp = new const_double(fp);
358           n_extra_cells += 1;
359           break;
360         case c_class:
361           cp = new const_class(fp);
362           break;
363         case c_string:
364           cp = new const_string(fp);
365           break;
366         case c_field_ref:
367         case c_method_ref:
368         case c_interface_method_ref:
369           cp = new const_ref(fp);
370           break;
371         case c_name_and_type:
372           cp = new const_name_and_type(fp);
373           break;
374         }
375         assert(cp != NULL);
376         fp += cp->length();
377         constant_pool[i] = cp;
378         i += n_extra_cells;
379       }      
380       int access_flags = unpack2(fp);
381       fp += 2;
382       int this_class_name = unpack2(fp);
383       fp += 2;
384       int super_class_name = unpack2(fp);
385       fp += 2;
386       int interfaces_count = unpack2(fp);
387       fp += 2;
388         
389       class_desc* this_class = 
390         class_desc::get(*(const_utf8*)constant_pool
391                         [((const_class*)constant_pool[this_class_name])->name]);
392     
393       set_class_source_path(this_class);
394     
395       // init. is_this
396       field_desc* is_this = new field_desc(utf_string("<this>"), this_class, NULL);
397       // assign name_and_type for is_this - find name entry "this" and obj. type
398       is_this->name_and_type =
399         new const_name_and_type(name_index,
400                                 ((const_class*)constant_pool[this_class_name])->name);
401       is_this->equals = is_this;
402       is_this->cls = this_class;
403     
404       // init. is_const
405       field_desc* is_const = new field_desc(utf_string("<const>"), NULL, NULL);
406     
407       this_class->attr = access_flags;
408       if (super_class_name == 0) { // Object class
409         assert(interfaces_count == 0); 
410         this_class->n_bases = 0;
411       } else { 
412         this_class->n_bases = interfaces_count + 1;
413         this_class->bases = new class_desc*[interfaces_count + 1];
414         this_class->bases[0] =
415           class_desc::get(*(const_utf8*)constant_pool
416                           [((const_class*)constant_pool[super_class_name])->name]);
417         
418         for (i = 1; i <= interfaces_count; i++) { 
419           int interface_name = unpack2(fp);
420           fp += 2;
421           this_class->bases[i] = 
422             class_desc::get(*(const_utf8*)constant_pool
423                             [((const_class*)constant_pool[interface_name])->name]);
424         }
425       }
426       int fields_count = unpack2(fp);
427       fp += 2;
428     
429       while (--fields_count >= 0) {
430         int access_flags = unpack2(fp); fp += 2;
431         int name_index = unpack2(fp); fp += 2;
432         int desc_index = unpack2(fp); fp += 2;
433         int attr_count = unpack2(fp); fp += 2;
434         field_desc* field = 
435           this_class->get_field(*(const_utf8*)constant_pool[name_index]);
436         field->attr |= access_flags;
437         while (--attr_count >= 0) { 
438           int attr_len = unpack4(fp+2); 
439           fp += 6 + attr_len;
440         }
441       }
442        
443       int methods_count = unpack2(fp);
444       fp += 2;
445         
446       //
447       // We need "SourceFile" attribute first, so remember position and 
448       // skip methods definitions
449       //
450       byte* method_part = fp;
451       for (i = 0; i < methods_count; i++) { 
452         int attr_count = unpack2(fp+6); 
453         fp += 8;
454         while (--attr_count >= 0) { 
455           int attr_len = unpack4(fp+2); 
456           fp += 6 + attr_len;
457         }
458       }
459       int attr_count = unpack2(fp); fp += 2;
460       while (--attr_count >= 0) { 
461         int attr_name = unpack2(fp); fp += 2; 
462         int attr_len = unpack4(fp); fp += 4;
463         utf_string attr(*(const_utf8*)constant_pool[attr_name]);
464         if (attr == "SourceFile") { 
465           int source_name = unpack2(fp); fp += 2;
466           int file_name_idx = this_class->source_file.rindex(FILE_SEP);
467           utf_string* file_name = (const_utf8*)constant_pool[source_name];
468           if (file_name_idx >= 0) { 
469             this_class->source_file.append(file_name_idx+1, *file_name);
470           } else { 
471             this_class->source_file = *file_name;
472           }
473         } else { 
474           fp += attr_len;
475         }
476       }
477       fp = method_part;
478     
479       while (--methods_count >= 0) { 
480         int access_flags = unpack2(fp); fp += 2;
481         int name_index = unpack2(fp); fp += 2;
482         int desc_index = unpack2(fp); fp += 2;
483         int attr_count = unpack2(fp); fp += 2;
484       
485         const_utf8* mth_name = (const_utf8*)constant_pool[name_index];
486         const_utf8* mth_desc = (const_utf8*)constant_pool[desc_index];
487     
488         method_desc* method = this_class->get_method(*mth_name, *mth_desc);
489         method->attr |= access_flags;
490         method->vars = NULL;
491       
492         while (--attr_count >= 0) { 
493           int attr_name = unpack2(fp); fp += 2;
494           int attr_len = unpack4(fp); fp += 4;
495           utf_string attr(*(const_utf8*)constant_pool[attr_name]);
496           if (attr == "Code") { 
497             int max_stack = unpack2(fp); fp += 2;
498             int max_locals = unpack2(fp); fp += 2;
499             int code_length = unpack4(fp); fp += 4;
500             method->code = fp;
501             method->code_length = code_length;
502             fp += code_length;
503     
504             method->n_vars = max_locals;
505             method->vars = max_locals > 0 ? new var_desc[max_locals] : NULL;
506     
507             method->line_table = new word[code_length];
508             memset(method->line_table, 0, sizeof(word)*code_length);
509     
510             method->context = new local_context*[code_length+1];
511             memset(method->context, 0, 
512                    sizeof(local_context*)*(code_length+1));
513     
514             int exception_table_length = unpack2(fp); fp += 2;
515     
516     	/* add new entry for each distinct "byte code adress of handle".
517     	**
518     	** if an exception handler at byte code "pos" handles exception of
519     	** more than one byte code range, call
520     	** "new ctx_entry_point(&method->context[pos]);" only once! Because
521     	** otherwise the stack gets out of control. 
522     	**
523     	** in the following example there are two different handle adresses
524     	** 16 and 25. and for each of them 
525     	** "new ctx_entry_point(&method->context[handler_pc]);" is called 
526     	** exactly once. Therefore the program calls :
527     	** new ctx_entry_point(&method->context[16]);
528     	** new ctx_entry_point(&method->context[25]);
529     	**********************************************************************
530     	** Example Exception Table:                                         **
531     	** -------------------------------------                            **
532     	**                                                                  **
533     	**                      byte code adress                            **
534     	** from         to        of handle                                 **
535     	**  2           10           16                                     **
536     	** 12           14           16                                     **
537     	** 20           23           25                                     **
538     	**********************************************************************
539     	**
540     	** it is expected that the byte code adresses of the handles are 
541     	** ordered. If this would not be the case, a simple comparison of
542     	** handler_pc and old_handler_pc would not be sufficient!
543     	*/
544     
545     	int old_handler_pc = -1;
546     
547             while (--exception_table_length >= 0) { 
548               int handler_pc = unpack2(fp+4);
549     	  if ( handler_pc != old_handler_pc) {
550     	    new ctx_entry_point(&method->context[handler_pc]); 
551     	  }
552     	  fp += 8;
553     	  old_handler_pc = handler_pc;
554             }
555     
556             int method_attr_count = unpack2(fp); fp += 2;
557             bool line_table_present = false;
558             bool local_var_table_present = false;
559             while (--method_attr_count >= 0) { 
560               int mth_attr_name = unpack2(fp); fp += 2;
561               int mth_attr_len = unpack4(fp); fp += 4;
562               utf_string* attr = 
563                 (const_utf8*)constant_pool[mth_attr_name];
564               if (*attr == "LineNumberTable") {
565                 int table_length = unpack2(fp); fp += 2;
566                 while (--table_length >= 0) { 
567                   int start_pc = unpack2(fp); fp += 2;
568                   int line_no = unpack2(fp); fp += 2;
569                   method->line_table[start_pc] = line_no;
570                 }
571                 method->first_line = method->line_table[0];
572                 if (method->first_line != 0) method->first_line -= 1;
573                 line_table_present = true;
574               } else if (*attr == "LocalVariableTable") { 
575                 method->local_variable_table_present = true;
576                 int table_length = unpack2(fp); fp += 2;
577                 while (--table_length >= 0) { 
578                   int start_pc = unpack2(fp); fp += 2;
579                   int length = unpack2(fp); fp += 2;
580                   int name  = unpack2(fp); fp += 2;
581                   int desc  = unpack2(fp); fp += 2;
582                   int index = unpack2(fp); fp += 2;
583                   int type = (index == 0 && 
584                               !(access_flags & method_desc::m_static))
585                     ? tp_self 
586                     : get_type(*(const_utf8*)constant_pool[desc]);
587                   new ctx_push_var(&method->context[start_pc], 
588                                    (const_utf8*)constant_pool[name],
589                                    type, index, start_pc);
590                   new ctx_pop_var(&method->context[start_pc+length],
591                                   index);
592                 }
593                 local_var_table_present = true;
594               } else {
595                 fp += mth_attr_len;
596               }
597             }
598             if (verbose && !(access_flags & method_desc::m_native)) { 
599               char buf[MAX_MSG_LENGTH];
600               method->demangle_method_name(buf);
601               if (!line_table_present) { 
602                 fprintf(stderr, "No line table present "
603                         "for method %s\n", buf);
604               }
605               if (!local_var_table_present) { 
606                 fprintf(stderr, "No local variable table present "
607                         "for method %s\n", buf);
608               }
609             }
610             method->parse_code(constant_pool, is_this);
611           } else { 
612             fp += attr_len;
613           }
614         }
615       }
616     
617       for (i = 1; i < constant_pool_count; i++) {
618         if (constant_pool[i] != NULL)
619           delete constant_pool[i];
620       }
621       
622       /*
623       for (method_desc* m = this_class->methods; m != NULL; m = m->next) {
624         m->locksAtEntry.clear();
625       }
626       */
627       delete[] constant_pool;
628       delete is_this->name_and_type;
629       //delete is_this;
630       delete is_const;
631     
632       monitor_stack::const_iterator it;
633       for (it = this_class->usedLocks.begin();
634            it != this_class->usedLocks.end(); it++) {
635         if (!((*it)->writes.empty())) {
636           vector<int>::const_iterator i;
637           for (i = (*it)->writes.begin(); i != (*it)->writes.end(); i++) {
638             message_at(msg_lock_assign, this_class->source_file, *i, *it);
639             // *i: line number; *it = field_desc*
640           }
641         }
642       }
643       
644       return true;
645     }

651     inline int stricmp(const char* p, const char* q) 
652     {
653       do { 
654         while (*p == '_') p += 1; // ignore '_'
655         while (*q == '_') q += 1; // ignore '_'
656         int diff = toupper(*(byte*)p) - toupper(*(byte*)q);
657         if (diff != 0) {
658           return diff;
659         }
660         q += 1;
661       } while (*p++ != 0);
662       return 0;
663     }

665     void proceed_file(char* file_name, bool recursive = false)
666     {
667     #ifdef _WIN32
668       HANDLE dir;
669       char dir_path[MAX_PATH];
670       WIN32_FIND_DATA file_data;
671       if (recursive != 0) { 
672         sprintf(dir_path, "%s\\*", file_name);
673         if ((dir=FindFirstFile(dir_path, &file_data)) != INVALID_HANDLE_VALUE) 
674           { 
675             file_name = dir_path; 
676           }
677       } else { 
678         if (strcmp(file_name, "..") == 0 || strcmp(file_name, ".") == 0) { 
679           proceed_file(file_name, 1);
680           return;
681         }
682         if ((dir=FindFirstFile(file_name, &file_data)) == INVALID_HANDLE_VALUE)
683           { 
684             fprintf(stderr, "Failed to open file '%s'\n", file_name);
685             return;
686           }
687       }
688       if (dir != INVALID_HANDLE_VALUE) {
689         do {
690           if (!recursive || *file_data.cFileName != '.') { 
691             char file_path[MAX_PATH];
692             char* file_dir = strrchr(file_name, '\\');
693             char* file_name_with_path;
694             if (file_dir != NULL) { 
695               int dir_len = file_dir - file_name + 1;
696               memcpy(file_path, file_name, dir_len);
697               strcpy(file_path+dir_len, file_data.cFileName);
698               file_name_with_path = file_path;
699             } else { 
700               file_name_with_path = file_data.cFileName;
701             }
702             proceed_file(file_name_with_path, recursive+1);
703           } 
704         } while (FindNextFile(dir, &file_data));
705         FindClose(dir);
706         return;
707       }
708     #else
709       DIR* dir = opendir(file_name);
710       if (dir != NULL) { 
711         struct dirent* entry;
712         while ((entry = readdir(dir)) != NULL) { 
713           if (*entry->d_name != '.') { 
714             char full_name[MAX_PATH];
715             sprintf(full_name, "%s/%s", file_name, entry->d_name);
716             proceed_file(full_name, 2);
717           }
718         } 
719         closedir(dir);
720         return;
721       }
722     #endif
723       if (recursive >= 2) { 
724         int len = strlen(file_name); 
725         if (len < 6 || stricmp(file_name + len - 6, ".class") != 0) {
726           return;
727         }
728       }
729       	
730       FILE* f = fopen(file_name, "rb");
731       if (f == NULL) { 
732         fprintf(stderr, "Failed to open file '%s'\n", file_name);
733         return;
734       }
735       char* suf = file_name + strlen(file_name) - 4;
736       if (suf > file_name && (stricmp(suf,".zip")==0 || stricmp(suf,".jar")==0)){
737         // extract files from ZIP archive
738         byte hdr[ECREC_SIZE+4];
739         if (fseek(f, 0, SEEK_END) != 0
740             || ftell(f) < ECREC_SIZE+4 
741             || fseek(f, -(ECREC_SIZE+4), SEEK_CUR) != 0
742             || fread(hdr, ECREC_SIZE+4, 1, f) != 1) 
743           { 
744             fprintf(stderr, "Bad format of ZIP file '%s'\n", file_name);
745             return;
746           }
747         int count = unpack2_le(&hdr[TOTAL_ENTRIES_CENTRAL_DIR]);
748         int dir_size = unpack4_le(&hdr[SIZE_CENTRAL_DIRECTORY]);
749         byte* directory = new byte[dir_size];
750         if (fseek(f, -(dir_size+ECREC_SIZE+4), SEEK_CUR) != 0
751             || fread(directory, dir_size, 1, f) != 1)
752           {
753             fprintf(stderr, "Bad format of ZIP file '%s'\n", file_name);
754             delete[] directory;
755             return;
756           }
757         byte* dp = directory;
758     
759         while (--count >= 0) {
760           int compression_method = unpack2_le(&dp[4+C_COMPRESSION_METHOD]);
761           int compressed_size = unpack4_le(&dp[4+C_COMPRESSED_SIZE]);
762           int uncompressed_size = unpack4_le(&dp[4+C_UNCOMPRESSED_SIZE]);
763           int filename_length = unpack2_le(&dp[4+C_FILENAME_LENGTH]);
764           int cextra_length = unpack2_le(&dp[4+C_EXTRA_FIELD_LENGTH]);
765     
766           if ((dp-directory)+filename_length+CREC_SIZE+4 > dir_size) { 
767             fprintf(stderr, "Bad format of ZIP file '%s'\n", file_name);
768             break;
769           }
770       	
771           byte local_rec_buffer[LREC_SIZE+4];
772           int local_header_offset = 
773             unpack4_le(&dp[4+C_RELATIVE_OFFSET_LOCAL_HEADER]);
774     
775           if (fseek(f, local_header_offset, SEEK_SET) != 0
776               || fread(local_rec_buffer, sizeof local_rec_buffer, 1, f) != 1
777               || memcmp(&local_rec_buffer[1], LOCAL_HDR_SIG, 3) != 0)
778             {
779               fprintf(stderr, "Bad format of ZIP file '%s'\n", file_name);
780               break;
781             }
782       	
783           int file_start = local_header_offset + (LREC_SIZE+4) +
784             unpack2_le(&local_rec_buffer[4+L_FILENAME_LENGTH]) +
785             unpack2_le(&local_rec_buffer[4+L_EXTRA_FIELD_LENGTH]);
786     
787           int filename_offset = CREC_SIZE+4;
788     
789           if (compression_method == C_UNCOMPRESSED)
790     	{
791               if (uncompressed_size != 0) { 
792                 byte* buffer = new byte[uncompressed_size];
793                 if (fseek(f, file_start, SEEK_SET) != 0 
794                     || fread(buffer, uncompressed_size, 1, f) != 1) 
795                   { 
796                     fprintf(stderr, "Failed to read ZIP file '%s'\n", 
797                             file_name);
798                     return;
799                   } else { 
800                     if (verbose) { 
801                       fprintf(stderr, "Verify file '%.*s'\n", 
802                               filename_length, dp + filename_offset); 
803                     }
804                     if (!parse_class_file(buffer)) { 
805                       fprintf(stderr, "File '%.*s' isn't correct Java class "
806                               "file\n", filename_length, dp+filename_offset);
807                     }
808                   }
809                 delete[] buffer;
810               }
811           }
812           else if (compression_method == C_DEFLATE) {
813     
814               if ((uncompressed_size != 0) && (compressed_size != 0)) { 
815                 byte* uncbuffer = new byte[uncompressed_size];
816                 byte* cbuffer = new byte[compressed_size];
817                 if (fseek(f, file_start, SEEK_SET) != 0 
818                     || fread(cbuffer, compressed_size, 1, f) != 1) 
819                   { 
820                     fprintf(stderr, "Failed to read ZIP file '%s'\n", 
821                             file_name);
822                     return;
823                   } else { 
824                     z_stream c_stream;
825     
826                     c_stream.zalloc = (alloc_func) 0;
827                     c_stream.zfree = (free_func)0;
828                     c_stream.opaque = (voidpf)0;
829     
830                     int err = inflateInit2(&c_stream, -MAX_WBITS);
831                     if (err != Z_OK)
832                     {
833                         fprintf(stderr, "Error %s in inflateInit while uncompressing %.*s.\n",
834                             c_stream.msg, filename_length, dp+filename_offset);
835                     }
836                     else
837                     {
838                         c_stream.next_in = cbuffer;
839                         c_stream.next_out = uncbuffer;
840                         c_stream.avail_in = compressed_size;
841                         c_stream.avail_out = uncompressed_size;
842                         c_stream.total_in = 0;
843                         c_stream.total_out = 0;
844     
845                         while ((err=inflate(&c_stream, Z_SYNC_FLUSH)) == Z_OK);
846                         if ((err != Z_STREAM_END) && (err != Z_OK))
847                         {
848                             fprintf(stderr, "Error %s in inflate while uncompressing %.*s.\n",
849                                 c_stream.msg, filename_length, dp+filename_offset);
850                         }
851                         else
852                         {
853                             inflateEnd(&c_stream);
854     
855                             if (verbose) { 
856                                 fprintf(stderr, "Verify file '%.*s'\n", 
857                                 filename_length, dp + filename_offset); 
858                             }
859                             if (!parse_class_file(uncbuffer)) { 
860                                 fprintf(stderr, "File '%.*s' isn't correct Java class "
861                                     "file\n", filename_length, dp+filename_offset);
862                             }
863                         }
864                     }
865                   }
866                 delete[] cbuffer;
867                 delete[] uncbuffer;
868               }
869           }
870           dp += filename_offset + filename_length + cextra_length; 
871         }
872         delete[] directory;
873       } else { 
874         fseek(f, 0, SEEK_END);
875         size_t file_size = ftell(f);
876         fseek(f, 0, SEEK_SET);
877         byte* buffer = new byte[file_size];
878         if (fread(buffer, file_size, 1, f) != 1) { 
879           fprintf(stderr, "Failed to read file '%s'\n", file_name);
880         } else { 
881           if (verbose) { 
882             fprintf(stderr, "Verify file '%s'\n", file_name); 
883           }
884           if (!parse_class_file(buffer)) { 
885             fprintf(stderr, "File '%s' isn't correct Java class file\n", 
886                     file_name);
887           }
888         }
889         delete[] buffer;
890       }
891       fclose(f);
892     }

949     void usage()
950     {
951       fprintf(stderr, "\
952     Usage: jlint {option|java_class_file}\n\
953     Options:\n\
954       -help : print this text\n\
955       -source <source_path> : path to directory with .java files\n\
956       -history <file_name> : use history file to avoid repeated messages\n\
957       -max_shown_paths <number> : max. shown paths between two sync. methods\n\
958       (+|-)verbose : output more/less information about program activity\n\
959       (+|-)message_category : select categories of messages to be reported\n\
960       (+|-)message_code : enable/disable concrete message\n\
961     Message categories:\n");
962       message_category group = msg_category_option[0].msg_cat;
963       for (unsigned i = 0; i < items(msg_category_option); i++) { 
964         message_category msg_cat = msg_category_option[i].msg_cat;
965         if ((msg_cat & ~group) != 0) { 
966           group = msg_cat;
967           fprintf(stderr, "\n");
968         }
969         fprintf(stderr, "  %s : %s\n", 
970                 msg_category_option[i].cat_name, 
971                 msg_category_option[i].cat_desc);
972       }
973       if (verbose) { 
974         fprintf(stderr, "\nMessages: (category:code: \"text\")\n");
975         for (int i = 0; i < msg_last; i++) { 
976           for (unsigned j = 0; j < items(msg_category_option); j++) { 
977             if (msg_table[i].category == msg_category_option[j].msg_cat) {
978               fprintf(stderr, "  %s:%s: \"%s\"\n", 
979                       msg_category_option[j].cat_name,
980                       msg_table[i].name, msg_table[i].format);
981               break;
982             }
983           }
984         }
985       }
986     }

988     int main(int argc, char* argv[])
989     {
990       int i, j;   
991     
992       if (argc == 1) { 
993         usage();
994         return EXIT_FAILURE;
995       }
996     
997       for (i = 1; i < argc; i++) { 
998         if (*argv[i] == '-' || *argv[i] == '+') { 
999           bool enabled = *argv[i] == '+';
1000           char* option = &argv[i][1];
1001           int n_cat = items(msg_category_option);
1002           msg_select_category_option* msg = msg_category_option;  
1003     
1004           if (stricmp(option, "source") == 0 && i+1 < argc) {
1005             source_file_path = argv[++i];
1006             source_file_path_len = strlen(source_file_path);
1007             if (source_file_path[source_file_path_len] == FILE_SEP) {
1008               source_file_path_len -= 1;
1009             }
1010             source_path_redefined = true;
1011             continue;
1012           } 
1013           if (stricmp(option, "history") == 0 && i+1 < argc) {
1014             history = fopen(argv[++i], "a+");
1015             if (history != NULL) { 
1016               char his_buf[MAX_MSG_LENGTH];
1017               fseek(history, 0, SEEK_SET);
1018               while (fgets(his_buf, sizeof his_buf, history)) { 
1019                 int len = strlen(his_buf); 
1020                 if (len > 0) { 
1021                   his_buf[len-1] = '\0';
1022                   message_node::add_to_hash(his_buf);
1023                 }
1024               }
1025               fseek(history, 0, SEEK_END);
1026             } else { 
1027               fprintf(stderr, "Failed to open history file '%s'\n",
1028                       argv[i]);
1029             }
1030             continue;
1031           }
1032           if (stricmp(option, "max_shown_paths") == 0 && i+1 < argc) {
1033             if (sscanf(argv[++i], "%d", &max_shown_paths) != 1) { 
1034               fprintf(stderr, "Bad parameter value for -max_shown_paths "
1035                       "option: '%s'\n", argv[i]);
1036             }
1037             continue;
1038           } 
1039           if (stricmp(option, "verbose") == 0) { 
1040             verbose = enabled;
1041             if (verbose) { 
1042               fprintf(stderr, 
1043                       "Jlint - program correctness verifier for Java, "
1044                       "version %s ("__DATE__").\n", VERSION);
1045             }
1046             continue;
1047           }
1048           if (stricmp(option, "help") == 0) { 
1049             usage();
1050             return EXIT_SUCCESS;
1051           }
1052           for (j = 0; j < n_cat; j++) { 
1053             if (stricmp(msg[j].cat_name, option) == 0) { 
1054               if (enabled) { 
1055                 reported_message_mask |= msg[j].msg_cat;
1056               } else { 
1057                 reported_message_mask &= ~msg[j].msg_cat;
1058               }
1059               break;
1060             }
1061           }
1062           if (j == n_cat) { 
1063             for (j = 0; j < msg_last; j++) { 
1064               if (stricmp(msg_table[j].name, option) == 0) { 
1065                 msg_table[j].enabled = enabled;
1066                 break;
1067               }
1068             }
1069             if (j == msg_last) { 
1070               fprintf(stderr, "Unrecognized option %s\n", argv[i]);
1071               usage();
1072             }
1073             return EXIT_FAILURE;
1074           }		 
1075         } else { 
1076           char* file_name = argv[i];
1077           if (!source_path_redefined) { 
1078             source_file_path = file_name;
1079             char* dirend = strrchr(source_file_path, FILE_SEP);
1080             source_file_path_len = (dirend != NULL)
1081               ? dirend - source_file_path : 0; 
1082           }
1083           proceed_file(file_name);
1084         }
1085       }
1086       class_desc::global_analysis();
1087       message_at(msg_done, utf_string(""), 0, (void*)n_messages);
1088       return EXIT_SUCCESS;
1089     }
