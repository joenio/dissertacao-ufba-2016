23     #ifdef HASH_TABLE
24     struct eqstr
25     {
26       bool operator()(const char* s1, const char* s2) const
27       {
28         return strcmp(s1, s2) == 0;
29       }
30     };
31     typedef hash_set<const char*, hash<const char*>, eqstr> string_table;
32     #else
33     struct ltstr
34     {
35       bool operator()(const char* s1, const char* s2) const
36       {
37         return strcmp(s1, s2) < 0;
38       }
39     };
40     typedef set<const char*, ltstr> string_table;
41     #endif
42     
43     class string_pool {
44     public:
45       string_table pool;
46     
47       const char* add(const char* el) {
48         char* entry;
49         string_table::iterator it;
50         if ((it = pool.find(el)) == pool.end()) {
51           // allocate element!
52           entry = strdup(el);
53           pool.insert(entry);
54           return const_cast<const char*>(entry);
55         } else {
56           return *it;
57         }
58       }
59     
60       void remove(const char* el) {
61         string_table::iterator entry;
62         if ((entry = pool.find(el)) != pool.end()) {
63           // free space for element
64           free(const_cast<char*>(*entry));
65           pool.erase(entry);
66         }
67       }
68     
69       void clear() {
70         for (string_table::iterator it = pool.begin(); it != pool.end(); it++) {
71           free(const_cast<char*>(*it));
72           pool.erase(it);
73         }
74       }
75     };
