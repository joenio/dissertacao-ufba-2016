4     bool Locks::acquire(Lock lock) { // acquire lock Lock (add to lockset)
5       // return true if lock is new
6       monitor_table::iterator entry = monTable.find(lock);
7       if (entry == monTable.end()) { // new lock, insert
8         monTable.insert(monitor_table::value_type(lock, 1));
9         monStack.push_front(lock);
10         return true;
11       } else {
12         // lock exists, do nothing (for now); later version: incr. count
13     #ifdef DUMP_MONITOR
14         printf("Lock %s already acquired.\n", lock->name.as_asciz());
15     #endif
16       }
17       return false;
18     }
19        
20     bool Locks::release(Lock lock) { // release lock Lock
21       // return true on success
22       if (getInnermost() == lock) {
23         monStack.pop_front();
24         monitor_table::iterator entry = monTable.find(lock);
25         if (entry != monTable.end()) { // ignore counter value for now
26           monTable.erase(entry);
27         }
28         return true;
29       } else {
30     #ifdef DUMP_MONITOR
31         Lock curr = getInnermost();
32         printf("Lock %s != %s could not be released.\n", 
33                lock->name.as_asciz(), curr ? curr->name.as_asciz() : "<none>");
34     #endif
35         return false; // could not delete lock; was not innermost lock
36       }
37     }
