3     void graph_edge::message(int loop_id)
4     {
5         caller->print_call_path_to(invocation, loop_id, 0);
6         class_desc* abstract_class = invocation->method->cls; 
7         invocation->method->cls = vertex->cls;
8         if (!strcmp(invocation->method->name.as_asciz(), "<synch>")) {
9           // special error message about synchronized blocks
10           if (invocation->method->attr & method_desc::m_sync_block) {
11             const char* lock1 = invocation->self_class->name.as_asciz();
12             // pretty print lock set
13             string lockset = "";
14             monitor_stack::const_iterator entry = caller->locksAtEntry.begin();
15             while (entry != caller->locksAtEntry.end()) {
16               if ((*entry)->cls == NULL)
17                 break;
18               const char* full_name =
19                 compound_name((*entry)->cls->name.as_asciz(),
20                               (*entry)->name.as_asciz());
21               if (strcmp(full_name, lock1)) {
22                 lockset += string(", ");
23                 if (strcmp(full_name, "")) {
24                   lockset += full_name;
25                 } else {
26                   lockset += "<unknown>";
27                 }
28               }
29               entry++;
30             }
31             if (caller->locksAtEntry.nLocks() > 2) {
32               lockset += "}";
33               lockset.replace(0, 2, "set {");
34             } else {
35               lockset.replace(0, 2, "");
36             }
37             if (lockset == "") {
38               lockset = "<unknown>";
39             }
40             invocation->message(msg_lock, lock1, lockset.c_str());
41           } else {
42             invocation->message(msg_lock, invocation->self_class->name.as_asciz(), 
43                                 caller->cls->name.as_asciz());
44           }
45         } else {
46           if (!(invocation->method->attr & method_desc::m_sync_block)) {
47             invocation->message(msg_sync_loop, (void*)loop_id, invocation->method);
48           }
49         }
50         invocation->method->cls = abstract_class;
51         mask |= (1 << (loop_id & 31));
52     }

54     void graph_vertex::verify()
55     {
56       graph_edge** stack = new graph_edge*[n_vertexes];
57       int marker = 0;
58       int n_loops = 0;
59         
60       for (graph_vertex* root = graph; root != NULL; root = root->next) { 
61         if (root->marker <= marker || root->visited >= max_shown_paths) { 
62           continue;
63         }
64         int sp = 0;
65         int i;
66       
67         graph_edge* edge = root->edges; 
68         root->visited |= flag_vertex_on_path;
69         root->marker = ++marker;
70     
71         while (edge != NULL) {
72           while (edge->vertex->marker >= marker || 
73                  edge->vertex->visited < max_shown_paths) 
74             {
75               graph_vertex* vertex = edge->vertex;
76               stack[sp++] = edge;
77     
78               if (vertex->visited & flag_vertex_on_path) { 
79                 // we found loop in the graph
80                 unsigned mask = edge->mask;
81                 for (i = sp-1; --i >= 0 && stack[i]->vertex != vertex;) { 
82                   mask &= stack[i]->mask;
83                 }
84                 if (mask == 0) { 
85                   n_loops += 1;
86                   while (++i < sp) { 
87                     stack[i]->message(n_loops);
88                   }
89                 }
90               } else { 
91                 // Pass through the graph in depth first order
92                 if (vertex->edges != NULL) { 
93                   vertex->visited = 
94                     (vertex->visited + 1) | flag_vertex_on_path;
95                   vertex->marker = marker;
96                   vertex->n_loops = n_loops;
97                   edge = vertex->edges;
98                   continue;
99                 }
100               }
101               sp -= 1;
102               break;
103             } // end of depth-first loop
104     
105           while (edge->next == NULL) { 
106             if (--sp < 0) break;
107             edge = stack[sp];
108             edge->vertex->visited &= ~flag_vertex_on_path;
109             if (edge->vertex->n_loops == n_loops) { 
110               // This path doesn't lead to deadlock: avoid future
111               // visits of this vertex
112               edge->vertex->marker = 0;
113             }
114           }
115           edge = edge->next;
116         }
117         root->visited &= ~flag_vertex_on_path;
118       }
119     }
