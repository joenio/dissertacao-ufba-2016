17     package edu.ksu.cis.indus.common.datastructures;
18     
19     import edu.ksu.cis.indus.annotations.Functional;
20     import edu.ksu.cis.indus.annotations.Immutable;
21     import edu.ksu.cis.indus.annotations.NonNull;
22     import edu.ksu.cis.indus.interfaces.IPoolable;
23     
24     import java.util.Collection;
25     import java.util.Collections;
26     import java.util.Iterator;
27     
28     /**
29      * This is a object pool aware work bag. All duplicate objects that are poolable are returned to the pool.
30      * 
31      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
32      * @author $Author: rvprasad $
33      * @version $Revision: 1.11 $ $Date: 2007/02/10 19:08:38 $
34      * @param <T> The type of work handled by this work bag.
35      */
36     public final class PoolAwareWorkBag<T>
37     		implements IWorkBag<T> {
38     
39     	/**
40     	 * The container that actual contains the work peices.
41     	 */
42     	private final IWorkBag<T> container;
43     
44     	/**
45     	 * Creates a new PoolAwareWorkBag object.
46     	 * 
47     	 * @param theContainer is the container that actually contains the work peices.
48     	 * @pre theContainer != null
49     	 */
50     	public PoolAwareWorkBag(@NonNull @Immutable final IWorkBag<T> theContainer) {
51     		container = theContainer;
52     	}
53     
54     	/**
55     	 * {@inheritDoc}
56     	 */
57     	public T getWork() {
58     		return container.getWork();
59     	}
60     
61     	/**
62     	 * {@inheritDoc}
63     	 */
64     	public void addAllWork(@NonNull @Immutable final Collection<? extends T> c) {
65     		container.addAllWork(c);
66     	}
67     
68     	/**
69     	 * Adds the given collection of work to the bag. Duplicate work peices are returned to the pool.
70     	 * 
71     	 * @param c is the collection of poolable objects.
72     	 * @return empty collection
73     	 * @pre c != null
74     	 * @post result != null and result.size() == 0
75     	 */
76     	@NonNull public Collection<T> addAllWorkNoDuplicates(@NonNull @Immutable final Collection<? extends T> c) {
77     		final Collection<T> _coll = container.addAllWorkNoDuplicates(c);
78     
79     		for (final Iterator<T> _i = _coll.iterator(); _i.hasNext();) {
80     			final T _o = _i.next();
81     
82     			if (_o instanceof IPoolable) {
83     				final IPoolable _poolable = (IPoolable) _o;
84     				_poolable.returnToPool();
85     			}
86     		}
87     		return Collections.emptyList();
88     	}
89     
90     	/**
91     	 * {@inheritDoc}
92     	 */
93     	public void addWork(@NonNull @Immutable final T o) {
94     		container.addWork(o);
95     	}
96     
97     	/**
98     	 * Adds the given work to the bag. If it is a duplicate work peice, it is returned to the pool.
99     	 * 
100     	 * @param o is the work peice.
101     	 * @return <code>true</code> if the work peice was added to the work bag; <code>false</code>, otherwise. In the
102     	 *         latter case the work peice is returned to he pool.
103     	 * @pre o != null
104     	 */
105     	public boolean addWorkNoDuplicates(@NonNull @Immutable final T o) {
106     		final boolean _result = container.addWorkNoDuplicates(o);
107     
108     		if (!_result && o instanceof IPoolable) {
109     			((IPoolable) o).returnToPool();
110     		}
111     		return _result;
112     	}
113     
114     	/**
115     	 * {@inheritDoc}
116     	 */
117     	public void clear() {
118     		container.clear();
119     	}
120     
121     	/**
122     	 * {@inheritDoc}
123     	 */
124     	@Functional public boolean hasWork() {
125     		return container.hasWork();
126     	}
127     }
