51     	/**
52     	 * Tests <code>isReachable</code>.
53     	 */
54     	public void localtestIsReachable() {
55     		final Collection _reachables = cgi.getReachableMethods();
56     		final Collection _heads = cgi.getEntryMethods();
57     
58     		for (final Iterator _i = ofa.getEnvironment().getClasses().iterator(); _i.hasNext();) {
59     			final SootClass _sc = (SootClass) _i.next();
60     
61     			for (final Iterator _j = _sc.getMethods().iterator(); _j.hasNext();) {
62     				final SootMethod _sm = (SootMethod) _j.next();
63     				assertEquals(cgi.isReachable(_sm), _reachables.contains(_sm));
64     
65     				if (cgi.isReachable(_sm)) {
66     					boolean _t = false;
67     
68     					for (final Iterator _k = _heads.iterator(); _k.hasNext();) {
69     						_t |= cg.isReachable(cg.getNode(_k.next()), cg.getNode(_sm), true);
70     					}
71     					assertTrue(_t || _heads.contains(_sm));
72     				}
73     			}
74     		}
75     	}

95     	/**
96     	 * Tests the tags on the reachable methods based on tags used during object flow analysis.
97     	 */
98     	public void testTagsOnReachableMethods() {
99     		final Context _ctxt = new Context();
100     		final Collection _reachables = cgi.getReachableMethods();
101     		assertNotNull(_reachables);
102     
103     		for (final Iterator _i = _reachables.iterator(); _i.hasNext();) {
104     			final SootMethod _o = (SootMethod) _i.next();
105     			assertTrue(_o.hasTag(FATestSetup.TAG_NAME));
106     			assertTrue(_o.getDeclaringClass().hasTag(FATestSetup.TAG_NAME));
107     
108     			if (!_o.isStatic()) {
109     				_ctxt.setRootMethod(_o);
110     				assertNotNull(ofa.getValuesForThis(_ctxt));
111     			}
112     		}
113     
114     		Collection _methods = new HashSet();
115     
116     		for (final Iterator _i = ofa.getEnvironment().getClasses().iterator(); _i.hasNext();) {
117     			_methods.addAll(((SootClass) _i.next()).getMethods());
118     		}
119     		_methods = SetUtils.difference(_methods, _reachables);
120     
121     		for (final Iterator _i = _methods.iterator(); _i.hasNext();) {
122     			final SootMethod _sm = (SootMethod) _i.next();
123     
124     			if (!_sm.isAbstract()) {
125     				assertFalse(_sm.hasTag(FATestSetup.TAG_NAME));
126     			}
127     		}
128     	}
