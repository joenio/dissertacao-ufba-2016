39     /**
40      * This class generates unique id's for each part of the system represented as Jimple. The most atomic parts are program
41      * points or <code>ValueBox</code> soot types. Given a system containing a set of classes, this generator will generate the
42      * different id's for the same class, method, field, local, statement, or program point over different runs. The user should
43      * use an external controller with deterministic traversal order to ensure that the id's for same entities are identical over
44      * different runs.
45      * 
46      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
47      * @author $Author: rvprasad $
48      * @version $Revision: 1.21 $ $Date: 2007/02/10 19:08:38 $
49      */
50     public final class UniqueJimpleIDGenerator
51     		implements IJimpleIDGenerator {
52     
53     	/**
54     	 * This maps classes to a sequence of fields that occur in it. This is used to generate unique id.
55     	 */
56     	private final Map<SootClass, List<SootField>> class2fields = new HashMap<SootClass, List<SootField>>();
57     
58     	/**
59     	 * This maps classes to a sequence of methods that occur in it. This is used to generate unique id.
60     	 * 
61     	 * @invariant class2methods.oclIsKindOf(Map(SootClass, Sequence(SootMethod)))
62     	 */
63     	private final Map<SootClass, List<SootMethod>> class2methods = new HashMap<SootClass, List<SootMethod>>();
64     
65     	/**
66     	 * This is a sequence of classes. This is used to generate unique id.
67     	 */
68     	private List<SootClass> classes = new ArrayList<SootClass>();
69     
70     	/**
71     	 * This maps methods to a sequence of locals that occur in it. This is used to generate unique id.
72     	 * 
73     	 * @invariant method2locals.oclIsKindOf(Map(SootMethod, Sequence(Local)))
74     	 */
75     	private final Map<SootMethod, List<Local>> method2locals = new HashMap<SootMethod, List<Local>>();
76     
77     	/**
78     	 * Creates an instance of this class.
79     	 */
80     	public UniqueJimpleIDGenerator() {
81     		super();
82     	}
83     
84     	/**
85     	 * Sorts the given elements based on the return value of <code>toString()</code> call on them.
86     	 * 
87     	 * @param <T> the type of elements in the collection.
88     	 * @param collection to be sorted.
89     	 * @return the sorted collection.
90     	 * @pre collection != null
91     	 * @post result != null and result.containsAll(collection) and collection.containsAll(result)
92     	 */
93     	private static <T> List<T> sort(final Collection<T> collection) {
94     		final List<T> _result = new ArrayList<T>(collection);
95     		Collections.sort(_result, ToStringBasedComparator.getComparator());
96     		return _result;
97     	}
98     
99     	/**
100     	 * @see edu.ksu.cis.indus.xmlizer.IJimpleIDGenerator#getIdForClass(SootClass)
101     	 */
102     	public String getIdForClass(final SootClass clazz) {
103     		if (!classes.contains(clazz)) {
104     			classes.add(clazz);
105     		}
106     		return "c" + String.valueOf(classes.indexOf(clazz));
107     	}
108     
109     	/**
110     	 * @see edu.ksu.cis.indus.xmlizer.IJimpleIDGenerator#getIdForField(soot.SootField)
111     	 */
112     	public String getIdForField(final SootField field) {
113     		final SootClass _declClass = field.getDeclaringClass();
114     
115     		if (!class2fields.containsKey(_declClass)) {
116     			@SuppressWarnings("unchecked") final List<SootField> _f = new ArrayList<SootField>(_declClass.getFields());
117     			class2fields.put(_declClass, sort(_f));
118     		}
119     
120     		final List<SootField> _fields = class2fields.get(_declClass);
121     		return getIdForClass(_declClass) + "_f" + _fields.indexOf(field);
122     	}
123     
124     	/**
125     	 * @see edu.ksu.cis.indus.xmlizer.IJimpleIDGenerator#getIdForLocal(Local, SootMethod)
126     	 */
127     	public String getIdForLocal(final Local v, final SootMethod method) {
128     		if (!method2locals.containsKey(method)) {
129     			@SuppressWarnings("unchecked") final List<Local> _l = new ArrayList<Local>(method.getActiveBody().getLocals());
130     			method2locals.put(method, sort(_l));
131     		}
132     		final List<Local> _locals = method2locals.get(method);
133     		return getIdForMethod(method) + "_l" + _locals.indexOf(v);
134     	}
135     
136     	/**
137     	 * @see edu.ksu.cis.indus.xmlizer.IJimpleIDGenerator#getIdForMethod(SootMethod)
138     	 */
139     	public String getIdForMethod(final SootMethod method) {
140     		final SootClass _declClass = method.getDeclaringClass();
141     
142     		if (!class2methods.containsKey(_declClass)) {
143     			@SuppressWarnings("unchecked") final List<SootMethod> _c = new ArrayList<SootMethod>(_declClass.getMethods());
144     			class2methods.put(_declClass, sort(_c));
145     		}
146     
147     		final List<SootMethod> _methods = class2methods.get(_declClass);
148     		return getIdForClass(method.getDeclaringClass()) + "_m" + _methods.indexOf(method);
149     	}
150     
151     	/**
152     	 * @see edu.ksu.cis.indus.xmlizer.IJimpleIDGenerator#getIdForStmt(soot.jimple.Stmt, soot.SootMethod)
153     	 */
154     	public String getIdForStmt(final Stmt stmt, final SootMethod method) {
155     		String _result = "?";
156     
157     		if (method.isConcrete()) {
158     			@SuppressWarnings("unchecked") final Collection<Stmt> _c = method.getActiveBody().getUnits();
159     			int _count = 0;
160     
161     			for (final Iterator<Stmt> _i = _c.iterator(); _i.hasNext();) {
162     				if (stmt == _i.next()) {
163     					break;
164     				}
165     				_count++;
166     			}
167     			_result = String.valueOf(_count);
168     		}
169     		return getIdForMethod(method) + "_s" + _result;
170     	}
171     
172     	/**
173     	 * @see edu.ksu.cis.indus.xmlizer.IJimpleIDGenerator#getIdForType(soot.Type)
174     	 */
175     	public String getIdForType(final Type type) {
176     		String _result;
177     
178     		if (type instanceof RefType) {
179     			_result = getIdForClass(((RefType) type).getSootClass());
180     		} else if (type instanceof ArrayType) {
181     			final ArrayType _arrayType = (ArrayType) type;
182     			final StringBuffer _t = new StringBuffer(getIdForType(_arrayType.baseType));
183     			_t.append(".." + _arrayType.numDimensions);
184     			_result = _t.toString();
185     		} else {
186     			_result = type.toString().replaceAll("[\\[\\]]", "_.").replaceAll("\\p{Blank}", "");
187     		}
188     		return _result;
189     	}
190     
191     	/**
192     	 * @see edu.ksu.cis.indus.xmlizer.IJimpleIDGenerator#getIdForValueBox(ValueBox, Stmt, SootMethod)
193     	 */
194     	public String getIdForValueBox(final ValueBox box, final Stmt stmt, final SootMethod method) {
195     		final List _vBoxes = stmt.getUseAndDefBoxes();
196     		return getIdForStmt(stmt, method) + "_v" + _vBoxes.indexOf(box);
197     	}
198     
199     	/**
200     	 * Resets the internal data structures.
201     	 */
202     	public void reset() {
203     		method2locals.clear();
204     		class2fields.clear();
205     		class2methods.clear();
206     		classes.clear();
207     	}
208     }
