66     /**
67      * This class represents Equivalence Class-based analysis to calculate escape information of objects. Escape information is
68      * provided in terms of share-ability of the object bound to a given value in a given method. This analysis is overloaded as a
69      * symbolic analysis to calculate information that can be used to prune ready-dependence edges.
70      * <p>
71      * This analysis requires <code>local splitting</code> option of Soot framework to be enabled while generating the Jimple
72      * for the system being analyzed.
73      * </p>
74      * <p>
75      * The implementation is based on the techreport <a
76      * href="http://www.cis.ksu.edu/santos/papers/technicalReports/SAnToS-TR2003-6.pdf">Honing the Detection of Interference and
77      * Ready Dependence for Slicing Concurrent Java Programs.</a>
78      * </p>
79      * 
80      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
81      * @author $Author: rvprasad $
82      * @version $Revision: 1.120 $
83      */

291     	/**
292     	 * Executes phase 2 and 3 as mentioned in the technical report. It processed each methods in the call-graph bottom-up
293     	 * propogating the alias set information in a collective fashion. It then propogates the information top-down in the
294     	 * call-graph.
295     	 */
296     	@Override public void analyze() {
297     		unstable();
298     		escapeInfo.unstableAdapter();
299     		objectReadWriteInfo.unstableAdapter();
300     
301     		if (LOGGER.isInfoEnabled()) {
302     			LOGGER.info("BEGIN: Equivalence Class-based and Symbol-based Escape Analysis");
303     		}
304     
305     		multiThreadedSystem = false;
306     
307     		final List<List<SootMethod>> _sccs = new ArrayList<List<SootMethod>>(cgi.getSCCs(false));
308     
309     		performPhase2(_sccs);
310     
311     		Collections.reverse(_sccs);
312     
313     		performPhase3(_sccs);
314     
315     		if (LOGGER.isDebugEnabled()) {
316     			LOGGER.debug("analyze() - " + toString());
317     		}
318     
319     		if (LOGGER.isInfoEnabled()) {
320     			LOGGER.info("END: Equivalence Class-based and Symbol-based Escape Analysis");
321     		}
322     
323     		processGlobalAliasSets();
324     		processMethodsForAccessToGlobalData();
325     		stable();
326     		escapeInfo.stableAdapter();
327     		objectReadWriteInfo.stableAdapter();
328     	}

365     	 * Checks if the given method is unconditionally atomic.
366     	 * 
367     	 * @param sm is the method of interest.
368     	 * @return <code>true</code> if the method is unconditionally atomic; <code>false</code>, otherwise.
369     	 */
370     	public boolean isMethodAtomic(final SootMethod sm) {
371     		final MethodContext _m = getMethodInfo(sm).getFirst();
372     		boolean _result = _m.getThrownAS().isSharedDataReachable();
373     
374     		if (!_result && _m.getThisAS() != null) {
375     			_result |= _m.getThisAS().isSharedDataReachable();
376     		}
377     
378     		if (!_result && _m.getReturnAS() != null) {
379     			_result |= _m.getReturnAS().isSharedDataReachable();
380     		}
381     
382     		final int _iEnd = sm.getParameterCount();
383     		for (int _i = 0; _i < _iEnd && !_result; _i++) {
384     			final AliasSet _arg = _m.getParamAS(_i);
385     			if (_arg != null) {
386     				_result |= _arg.isSharedDataReachable();
387     			}
388     		}
389     
390     		if (!_result) {
391     			for (final AliasSet _as : method2Triple.get(sm).getSecond().values()) {
392     				if (_as != null) {
393     					_result |= _as.isSharedDataReachable();
394     				}
395     			}
396     		}
397     
398     		return !_result;
399     	}

609     	 * Retrieves the alias set corresponding to the given value. This method cannot handled <code>CaughtExceptionRef</code>.
610     	 * 
611     	 * @param v is the value for which the alias set is requested.
612     	 * @param sm is the method in which <code>v</code> occurs.
613     	 * @return the alias set corresponding to <code>v</code>. <code>null</code> if <code>sm</code> was not analyzed.
614     	 * @pre v.isOclKindOf(Local) or v.isOclKindOf(ArrayRef) or v.isOclKindOf(FieldRef) or v.isOclKindOf(ArrayRef) or
615     	 *      v.isOclKindOf(InstanceFieldRef) or v.isOclIsKindOf(ParameterRef)
616     	 */
617     	AliasSet queryAliasSetFor(final Value v, final SootMethod sm) {
618     		if (LOGGER.isDebugEnabled()) {
619     			LOGGER.debug("BEGIN - queryAliasSetFor(v = " + v.getClass() + ", sm = " + sm + ")");
620     		}
621     
622     		final Triple<MethodContext, Map<Local, AliasSet>, Map<CallTriple, MethodContext>> _trp = method2Triple.get(sm);
623     
624     		final AliasSet _result;
625     		if (_trp == null) {
626     			if (LOGGER.isWarnEnabled()) {
627     				LOGGER.warn("Non-existent method triple for " + sm);
628     			}
629     			_result = null;
630     		} else {
631     			final Map<Local, AliasSet> _local2AS = _trp.getSecond();
632     
633     			if (v instanceof InstanceFieldRef) {
634     				final InstanceFieldRef _i = (InstanceFieldRef) v;
635     				final AliasSet _temp = _local2AS.get(_i.getBase());
636     				_result = _temp.getASForField(_i.getField().getSignature());
637     			} else if (v instanceof StaticFieldRef) {
638     				final SootField _field = ((StaticFieldRef) v).getField();
639     				final AliasSet _base = getASForClass(_field.getDeclaringClass());
640     				_result = _base.getASForField(_field.getSignature());
641     			} else if (v instanceof ArrayRef) {
642     				final ArrayRef _a = (ArrayRef) v;
643     				final AliasSet _temp = _local2AS.get(_a.getBase());
644     				_result = _temp.getASForField(IReadWriteInfo.ARRAY_FIELD);
645     			} else if (v instanceof Local) {
646     				_result = _local2AS.get(v);
647     			} else if (v instanceof ThisRef) {
648     				_result = _trp.getFirst().getThisAS();
649     			} else if (v instanceof ParameterRef) {
650     				_result = _trp.getFirst().getParamAS(((ParameterRef) v).getIndex());
651     			} else if (v instanceof CaughtExceptionRef) {
652     				final String _msg = "CaughtExceptionRef cannot be handled.";
653     				LOGGER.error(_msg);
654     				throw new IllegalArgumentException(_msg);
655     			} else {
656     				_result = null;
657     			}
658     		}
659     
660     		if (LOGGER.isDebugEnabled()) {
661     			LOGGER.debug("END - queryAliasSetFor(Value, SootMethod): " + _result);
662     		}
663     
664     		return _result;
665     	}

705     	/**
706     	 * Rewires the method context, local variable alias sets, and site contexts such that they contain only representative
707     	 * alias sets and no the nominal(indirectional) alias sets.
708     	 * 
709     	 * @param method for which this processing should occur.
710     	 * @pre method != null
711     	 */
712     	private void discardReferentialAliasSets(final SootMethod method) {
713     		if (localASsCache.isEmpty()) {
714     			localASsCache = Collections.emptyMap();
715     		} else {
716     			for (final Iterator<Map.Entry<Local, AliasSet>> _i = localASsCache.entrySet().iterator(); _i.hasNext();) {
717     				final Map.Entry<Local, AliasSet> _entry = _i.next();
718     				final AliasSet _as = _entry.getValue();
719     				final AliasSet _equiv = _as.find();
720     
721     				if (_equiv != _as) {
722     					_entry.setValue(_equiv);
723     				}
724     			}
725     		}
726     
727     		if (scCache.isEmpty()) {
728     			scCache = Collections.emptyMap();
729     		} else {
730     			for (final Iterator<Map.Entry<CallTriple, MethodContext>> _i = scCache.entrySet().iterator(); _i.hasNext();) {
731     				final Map.Entry<CallTriple, MethodContext> _entry = _i.next();
732     				final MethodContext _mc = _entry.getValue();
733     				final MethodContext _mcRep = _mc.find();
734     
735     				if (_mcRep != _mc) {
736     					_entry.setValue(_mcRep);
737     				}
738     				_mcRep.discardReferentialAliasSets();
739     			}
740     		}
741     		methodCtxtCache.discardReferentialAliasSets();
742     		methodCtxtCache.find(); // cleanup mere indirection contexts.
743     
744     		/*
745     		 * We don't store methodCtxtCache.find() in the triple as we need access to the original context for handling null
746     		 * argument scenarios. Read about "null-arguments scenario" in ValueProcessor.
747     		 */
748     		method2Triple.put(method, new Triple<MethodContext, Map<Local, AliasSet>, Map<CallTriple, MethodContext>>(
749     				methodCtxtCache, localASsCache, scCache));
750     	}

752     	/**
753     	 * Performs phase 2 processing as described in the paper mentioned in the documentation of this class.
754     	 * 
755     	 * @param sccsInBottomUpOrder is the SCCs in bottom up order.
756     	 */
757     	private void performPhase2(final List<List<SootMethod>> sccsInBottomUpOrder) {
758     		final Collection<BasicBlock> _processed = new HashSet<BasicBlock>();
759     		final IWorkBag<BasicBlock> _wb = new HistoryAwareFIFOWorkBag<BasicBlock>(_processed);
760     
761     		// Phase 2: The SCCs are ordered bottom up.
762     		for (final Iterator<List<SootMethod>> _i = sccsInBottomUpOrder.iterator(); _i.hasNext();) {
763     			final List<SootMethod> _nodes = _i.next();
764     
765     			for (final Iterator<SootMethod> _j = _nodes.iterator(); _j.hasNext();) {
766     				final SootMethod _sm = _j.next();
767     
768     				if (LOGGER.isDebugEnabled()) {
769     					LOGGER.debug("Bottom-up processing method " + _sm);
770     				}
771     
772     				final Triple<MethodContext, Map<Local, AliasSet>, Map<CallTriple, MethodContext>> _triple = getMethodInfo(_sm);
773     
774     				if (!_sm.isConcrete()) {
775     					if (LOGGER.isWarnEnabled()) {
776     						LOGGER.warn("NO BODY: " + _sm.getSignature());
777     					}
778     
779     					continue;
780     				}
781     
782     				methodCtxtCache = _triple.getFirst();
783     				localASsCache = _triple.getSecond();
784     				scCache = _triple.getThird();
785     				context.setRootMethod(_sm);
786     
787     				final BasicBlockGraph _bbg = bbm.getBasicBlockGraph(_sm);
788     				_wb.clear();
789     				_processed.clear();
790     				_wb.addWork(_bbg.getHead());
791     
792     				while (_wb.hasWork()) {
793     					final BasicBlock _bb = _wb.getWork();
794     
795     					for (final Iterator<Stmt> _k = _bb.getStmtsOf().iterator(); _k.hasNext();) {
796     						final Stmt _stmt = _k.next();
797     						context.setStmt(_stmt);
798     						stmtProcessor.process(_stmt);
799     					}
800     					_wb.addAllWorkNoDuplicates(_bb.getSuccsOf());
801     				}
802     
803     				// discard alias sets that serve as a mere indirection level.
804     				discardReferentialAliasSets(_sm);
805     			}
806     		}
807     
808     		if (multiThreadedSystem) {
809     			for (final Iterator<AliasSet> _i = class2aliasSet.values().iterator(); _i.hasNext();) {
810     				final AliasSet _as = _i.next();
811     				_as.markAsCrossingThreadBoundary();
812     				AliasSet.selfUnify(_as);
813     			}
814     		}
815     	}

818     	 * Performs phase 3 processing as described in the paper described in the documentation of this class.
819     	 * 
820     	 * @param sccsInTopDownOrder is the SCCs in top-down order.
821     	 */
822     	private void performPhase3(final List<List<SootMethod>> sccsInTopDownOrder) {
823     		// Phase 2: The SCCs are ordered bottom up.
824     		for (final Iterator<List<SootMethod>> _i = sccsInTopDownOrder.iterator(); _i.hasNext();) {
825     			final List<SootMethod> _nodes = _i.next();
826     
827     			for (final Iterator<SootMethod> _j = _nodes.iterator(); _j.hasNext();) {
828     				final SootMethod _caller = _j.next();
829     				if (LOGGER.isDebugEnabled()) {
830     					LOGGER.debug("Top-down processing method : CALLER : " + _caller);
831     				}
832     
833     				final Triple<MethodContext, Map<Local, AliasSet>, Map<CallTriple, MethodContext>> _callerTriple = method2Triple
834     						.get(_caller);
835     				final Map<CallTriple, MethodContext> _ctrp2sc = _callerTriple.getThird();
836     				final Collection<CallTriple> _callees = cgi.getCallees(_caller);
837     
838     				for (final Iterator<CallTriple> _k = _callees.iterator(); _k.hasNext();) {
839     					final CallTriple _ctrp = _k.next();
840     					final SootMethod _callee = _ctrp.getMethod();
841     
842     					if (LOGGER.isDebugEnabled()) {
843     						LOGGER.debug("Top-down processing : CALLEE : " + _callee);
844     					}
845     
846     					final Triple<MethodContext, Map<Local, AliasSet>, Map<CallTriple, MethodContext>> _calleeTriple;
847     					_calleeTriple = method2Triple.get(_callee);
848     
849     					final MethodContext _calleeMethodContext = _calleeTriple.getFirst();
850     					final CallTriple _callerTrp = new CallTriple(_caller, _ctrp.getStmt(), _ctrp.getExpr());
851     					final MethodContext _calleeSiteContext = _ctrp2sc.get(_callerTrp);
852     					_calleeSiteContext.propogateInfoFromTo(_calleeMethodContext);
853     				}
854     			}
855     		}

862     		final Collection<AliasSet> _proc = new HashSet<AliasSet>();
863     		final IWorkBag<AliasSet> _wb = new HistoryAwareFIFOWorkBag<AliasSet>(_proc);
864     		for (final AliasSet _as : class2aliasSet.values()) {
865     			_wb.clear();
866     			_proc.clear();
867     			_wb.addWork(_as.find());
868     			while (_wb.hasWork()) {
869     				final AliasSet _a = _wb.getWork();
870     				_a.setGlobal();

872     					_wb.addWork(_fs.find());

879     	 * Processes methods to calculate access to global data.
880     	 */
881     	private void processMethodsForAccessToGlobalData() {
882     		final Collection<AliasSet> _proc = new HashSet<AliasSet>();
883     		final IWorkBag<AliasSet> _wb = new HistoryAwareFIFOWorkBag<AliasSet>(_proc);
884     
885     		for (final SootMethod _sm : method2Triple.keySet()) {
886     			final Triple<MethodContext, ?, ?> _t = method2Triple.get(_sm);
887     			final MethodContext _m = _t.getFirst();
888     			_wb.clear();
889     			_proc.clear();
890     
891     			if (_m.getThisAS() != null) {
892     				_wb.addWork(_m.getThisAS());
893     			}
894     
895     			if (_m.getReturnAS() != null) {
896     				_wb.addWork(_m.getReturnAS());
897     			}
898     
899     			final int _iEnd = _sm.getParameterCount();
900     			for (int _iIndex = 0; _iIndex < _iEnd; _iIndex++) {
901     				final AliasSet _arg = _m.getParamAS(_iIndex);
902     				if (_arg != null) {
903     					_wb.addWork(_arg);
904     				}
905     			}
906     
907     			_wb.addWork(_m.getThrownAS());
908     
909     			for (final AliasSet _as : method2Triple.get(_sm).getSecond().values()) {
910     				if (_as != null) {
911     					_wb.addWork(_as);
912     				}
913     			}
914     
915     			boolean _read = _m.isGlobalDataRead();
916     			boolean _written = _m.isGlobalDataWritten();
917     			while (_wb.hasWork() && !(_read && _written)) {
918     				final AliasSet _a = _wb.getWork();
919     				if (_a.isGlobal()) {
920     					_read |= _a.wasAnyFieldRead();
921     					_written |= _a.wasAnyFieldWritten();
922     				}
923     				for (final AliasSet _fs : _a.getFieldMap().values()) {
924     					_wb.addWork(_fs.find());
925     				}
926     			}
927     
928     			if (_read) {
929     				_m.globalDataWasRead();
930     			}
931     
932     			if (_written) {
933     				_m.globalDataWasWritten();
934     			}
935     		}
936     	}
