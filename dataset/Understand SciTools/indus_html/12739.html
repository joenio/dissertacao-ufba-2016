97     	/**
98     	 * Retrieves the name that serves as the base for the file names into which info will be dumped along with the root
99     	 * methods to be considered in one execution of the analyses.
100     	 * 
101     	 * @param root is the object based on which base name should be generated.
102     	 * @param methods is the collection that will contain the root methods upon return.
103     	 * @return a name along with the root methods via <code>methods</code>.
104     	 * @pre root != null and methods != null
105     	 * @post result != null and (methods.contains(root) or methods.containsAll(root))
106     	 */
107     	public static String getBaseNameOfFileAndRootMethods(final Object root, final Collection<SootMethod> methods) {
108     		final String _result;
109     
110     		if (root instanceof SootMethod) {
111     			final SootMethod _sm = (SootMethod) root;
112     			_result = (_sm.getDeclaringClass().getJavaStyleName() + "_" + _sm.getSubSignature()).replaceAll(" ", "_");
113     			methods.add(_sm);
114     		} else {
115     			_result = "cumulative_" + System.currentTimeMillis();
116     			methods.addAll((Collection) root);
117     		}
118     		return _result;
119     	}

121     	/**
122     	 * The entry point to the program via command line.
123     	 * 
124     	 * @param args is the command line arguments.
125     	 * @throws RuntimeException when object flow analysis fails.
126     	 */
127     	public static void main(final String[] args) {
128     		final Options _options = new Options();
129     		Option _option = new Option("c", "cumulative", false, "Consider all root methods in the same execution.");
130     		_options.addOption(_option);
131     		_option = new Option("o", "output", true, "Directory into which xml files will be written into.");
132     		_option.setArgs(1);
133     		_options.addOption(_option);
134     		_option = new Option("j", "jimple", false, "Dump xmlized jimple.");
135     		_options.addOption(_option);
136     		_option = new Option("h", "help", false, "Display message.");
137     		_options.addOption(_option);
138     		_option = new Option("p", "soot-classpath", false, "Prepend this to soot class path.");
139     		_option.setArgs(1);
140     		_option.setArgName("classpath");
141     		_option.setOptionalArg(false);
142     		_options.addOption(_option);
143     		_option = new Option("t", "ofa-type", false, "Type of analysis : fioi, fsoi, fios, fsos, fioirt, fsoirt.");
144     		_option.setArgs(1);
145     		_option.setArgName("type");
146     		_option.setOptionalArg(false);
147     		_option.setRequired(true);
148     		_options.addOption(_option);
149     		_option = new Option("S", "scope", true, "The scope that should be analyzed.");
150     		_option.setArgs(1);
151     		_option.setArgName("scope");
152     		_option.setRequired(false);
153     		_options.addOption(_option);
154     
155     		final PosixParser _parser = new PosixParser();
156     
157     		try {
158     			final CommandLine _cl = _parser.parse(_options, args);
159     
160     			if (_cl.hasOption("h")) {
161     				printUsage(_options);
162     				System.exit(1);
163     			}
164     
165     			String _outputDir = _cl.getOptionValue('o');
166     
167     			if (_outputDir == null) {
168     				if (LOGGER.isWarnEnabled()) {
169     					LOGGER.warn("Defaulting to current directory for output.");
170     				}
171     				_outputDir = ".";
172     			}
173     
174     			if (_cl.getArgList().isEmpty()) {
175     				throw new MissingArgumentException("Please specify atleast one class.");
176     			}
177     
178     			final OFAXMLizerCLI _cli = new OFAXMLizerCLI();
179     			_cli.xmlizer.setXmlOutputDir(_outputDir);
180     			_cli.xmlizer.setGenerator(new UniqueJimpleIDGenerator());
181     			_cli.setCumulative(_cl.hasOption('c'));
182     			_cli.setClassNames(_cl.getArgList());
183     			_cli.type = _cl.getOptionValue('t');
184     
185     			if (_cl.hasOption('p')) {
186     				_cli.addToSootClassPath(_cl.getOptionValue('p'));
187     			}
188     
189     			if (_cl.hasOption('S')) {
190     				_cli.setScopeSpecFile(_cl.getOptionValue('S'));
191     			}
192     
193     			_cli.initialize();
194     
195     			_cli.<ITokens> execute(_cl.hasOption('j'));
196     		} catch (final ParseException _e) {
197     			LOGGER.error("Error while parsing command line.", _e);
198     			System.out.println("Error while parsing command line. \n" + _e);
199     			printUsage(_options);
200     		} catch (final Throwable _e) {
201     			LOGGER.error("Beyond our control. May day! May day!", _e);
202     			throw new RuntimeException(_e);
203     		}
204     	}

217     	/**
218     	 * Xmlize the given system.
219     	 * 
220     	 * @param dumpJimple <code>true</code> indicates that the jimple should be xmlized as well; <code>false</code>,
221     	 *            otherwise.
222     	 * @param <T> dummy type parameter.
223     	 */
224     	private <T extends ITokens<T, Value>> void execute(final boolean dumpJimple) {
225     		setInfoLogger(LOGGER);
226     
227     		final String _tagName = "OFAXMLizer:FA";
228     		final IValueAnalyzer<Value> _aa;
229     		if (type.equals("fioi")) {
230     			_aa = OFAnalyzer.getFIOIAnalyzer(_tagName,
231     					TokenUtil.<T, Value, Type> getTokenManager(new SootValueTypeManager()), getStmtGraphFactory());
232     		} else if (type.equals("fios")) {
233     			_aa = OFAnalyzer.getFIOSAnalyzer(_tagName,
234     					TokenUtil.<T, Value, Type> getTokenManager(new SootValueTypeManager()), getStmtGraphFactory());
235     		} else if (type.equals("fsoi")) {
236     			_aa = OFAnalyzer.getFSOIAnalyzer(_tagName,
237     					TokenUtil.<T, Value, Type> getTokenManager(new SootValueTypeManager()), getStmtGraphFactory());
238     		} else if (type.equals("fsos")) {
239     			_aa = OFAnalyzer.getFSOSAnalyzer(_tagName,
240     					TokenUtil.<T, Value, Type> getTokenManager(new SootValueTypeManager()), getStmtGraphFactory());
241     		} else if (type.equals("fioirt")) {
242     			_aa = OFAnalyzer.getFSOIRTAnalyzer(_tagName, TokenUtil
243     					.<T, Value, Type> getTokenManager(new SootValueTypeManager()), getStmtGraphFactory());
244     		} else if (type.equals("fsoirt")) {
245     			_aa = OFAnalyzer.getFSOIRTAnalyzer(_tagName, TokenUtil
246     					.<T, Value, Type> getTokenManager(new SootValueTypeManager()), getStmtGraphFactory());
247     		} else {
248     			throw new IllegalArgumentException("ofa-type has to be one of the following: fioi, fsoi, fios, fsos, fioirt,"
249     					+ " fsoirt.");
250     		}
251     
252     		final ValueAnalyzerBasedProcessingController _pc = new ValueAnalyzerBasedProcessingController();
253     		final Collection<IProcessor> _processors = new ArrayList<IProcessor>();
254     		final CallGraphInfo _cgi = new CallGraphInfo(new PairManager(false, true));
255     		final OFABasedCallInfoCollector _callGraphInfoCollector = new OFABasedCallInfoCollector();
256     		final Collection<SootMethod> _rm = new ArrayList<SootMethod>();
257     		final ProcessingController _xmlcgipc = new ProcessingController();
258     		final MetricsProcessor _countingProcessor = new MetricsProcessor();
259     		final OneAllStmtSequenceRetriever _ssr = new OneAllStmtSequenceRetriever();
260     		_ssr.setStmtGraphFactory(getStmtGraphFactory());
261     		_pc.setStmtSequencesRetriever(_ssr);
262     		_pc.setAnalyzer(_aa);
263     
264     		final IProcessingFilter _tagFilter = new TagBasedProcessingFilter(_tagName);
265     		_pc.setProcessingFilter(_tagFilter);
266     		_xmlcgipc.setEnvironment(_aa.getEnvironment());
267     
268     		final IProcessingFilter _xmlFilter = new CGBasedXMLizingProcessingFilter(_cgi);
269     		_xmlFilter.chain(_tagFilter);
270     		_xmlcgipc.setProcessingFilter(_xmlFilter);
271     		_xmlcgipc.setStmtSequencesRetriever(_ssr);
272     
273     		final Map _info = new HashMap();
274     		_info.put(IValueAnalyzer.ID, _aa);
275     		_info.put(IValueAnalyzer.TAG_ID, _tagName);
276     
277     		final List<Object> _roots = new ArrayList<Object>();
278     
279     		if (cumulative) {
280     			_roots.add(getRootMethods());
281     		} else {
282     			_roots.addAll(getRootMethods());
283     		}
284     		Collections.sort(_roots, ToStringBasedComparator.getComparator());
285     		writeInfo("Root methods are: " + _roots.size() + "\n" + _roots);
286     
287     		for (final Iterator<Object> _k = _roots.iterator(); _k.hasNext();) {
288     			_rm.clear();
289     
290     			final Object _root = _k.next();
291     			final String _fileBaseName = getBaseNameOfFileAndRootMethods(_root, _rm);
292     
293     			writeInfo("RootMethod: " + _root);
294     			writeInfo("BEGIN: FA");
295     
296     			final long _start = System.currentTimeMillis();
297     			_aa.reset();
298     			getBbm().reset();
299     
300     			_aa.analyze(getEnvironment(), _rm);
301     
302     			final long _stop = System.currentTimeMillis();
303     			addTimeLog("FA", _stop - _start);
304     			writeInfo("END: FA");
305     
306     			_callGraphInfoCollector.reset();
307     			_processors.clear();
308     			_processors.add(_callGraphInfoCollector);
309     			_processors.add(_countingProcessor);
310     			_pc.reset();
311     			_pc.driveProcessors(_processors);
312     			_processors.clear();
313     			_cgi.reset();
314     			_cgi.createCallGraphInfo(_callGraphInfoCollector.getCallInfo());
315     
316     			final ByteArrayOutputStream _stream = new ByteArrayOutputStream();
317     			new PrintWriter(_stream).write("STATISTICS: "
318     					+ MapUtils
319     							.verbosePrint(new TreeMap<MetricKeys, Map<Object, Integer>>(_countingProcessor.getStatistics())));
320     			writeInfo(_stream.toString());
321     
322     			_info.put(AbstractXMLizer.FILE_NAME_ID, _fileBaseName);
323     			_info.put(IStmtGraphFactory.ID, getStmtGraphFactory());
324     			xmlizer.writeXML(_info);
325     
326     			if (dumpJimple) {
327     				((AbstractXMLizer) xmlizer).dumpJimple(_fileBaseName, xmlizer.getXmlOutputDir(), _xmlcgipc);
328     			}
329     
330     			final OFAStringizer _str = new OFAStringizer();
331     			_str.getOFAInfoAsString(_info, new PrintWriter(System.out));
332     		}
333     	}
334     
335     	/**
336     	 * Sets cumulative mode.
337     	 * 
338     	 * @param option <code>true</code> indicates all root methods should be analyzed in one go; <code>false</code>
339     	 *            indicates analysis should be executed once for each root method.
340     	 */
341     	private void setCumulative(final boolean option) {
342     		cumulative = option;
343     	}
