86     	public final void testFindMethodInSuperClassesAndInterfaces() {

88     		assertFalse(_temp.isEmpty());
89     
90     		final List _params = new ArrayList();
91     		_params.add(scene.loadClassAndSupport("java.lang.Object").getType());

93     		assertFalse(_temp.isEmpty());

109     	/**
110     	 * Tests <code>getAncestors</code>.
111     	 */
112     	public final void testGetAncestors() {
113     		final Collection _classes = Util.getAncestors(sc);
114     		final Collection _ancestors = new ArrayList();
115     		_ancestors.add("java.lang.Object");
116     		_ancestors.add("java.util.Collection");
117     		_ancestors.add("java.util.Set");
118     		_ancestors.add("java.lang.Cloneable");
119     		_ancestors.add("java.io.Serializable");
120     		_ancestors.add("java.util.AbstractSet");
121     		_ancestors.add("java.util.AbstractCollection");
122     		_ancestors.add("java.lang.Iterable");
123     		assertEquals(_ancestors.size(), _classes.size());
124     
125     		for (final Iterator _i = _classes.iterator(); _i.hasNext();) {
126     			final SootClass _class = (SootClass) _i.next();
127     			_ancestors.remove(_class.getName());
128     		}
129     		assertTrue(_ancestors.isEmpty());
130     	}

201     	 * Tests <code>removeMethodsWithSameSignature</code>.
202     	 */
203     	public final void testRemoveMethodsWithSameSignature() {
204     		final SootClass _object = scene.getSootClass("java.lang.Object");
205     		Collection _t1 = new ArrayList(sc.getMethods());
206     		int _preSize = _t1.size();
207     		Util.removeMethodsWithSameSignature(_t1, _object.getMethods());
208     		assertTrue(_t1.size() == _preSize - 3);  // 3 for <init>, <clinit>, and clone
209     		_t1 = new ArrayList(_object.getMethods());
210     		_preSize = _t1.size();
211     		Util.removeMethodsWithSameSignature(_t1, _object.getMethods());
212     		assertTrue(_t1.size() != _preSize);
213     		assertTrue(_t1.isEmpty());
214     	}

219     	public final void testRetainMethodsWithSameSignature() {
220     		final SootClass _object = scene.getSootClass("java.lang.Object");
221     		Collection _t1 = new ArrayList(sc.getMethods());
222     		int _preSize = _t1.size();
223     		Util.retainMethodsWithSameSignature(_t1, _object.getMethods());
224     
225     		assertTrue(_t1.size() != _preSize);
226     		assertTrue(_t1.size() == 3);  // 3 for <init>, <clinit>, and clone
227     		_t1 = new ArrayList(_object.getMethods());
228     		_preSize = _t1.size();
229     		Util.retainMethodsWithSameSignature(_t1, _object.getMethods());
230     		assertTrue(_t1.size() == _preSize);
