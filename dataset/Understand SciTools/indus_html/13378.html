32     /**
33      * This implementation modifies the class hierarchy to minimize the number of required classes.
34      *
35      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
36      * @author $Author: rvprasad $
37      * @version $Revision: 1.9 $ $Date: 2007/02/10 19:06:02 $
38      */
39     public class ExecutableSlicePostProcessorAndModifier
40     		extends ExecutableSlicePostProcessor {
41     
42     	/**
43     	 * The names of the classes to be retained.
44     	 */
45     	private final Collection<String> retain;
46     
47     	/**
48     	 * The environment on which this processor is operating.
49     	 */
50     	private final IEnvironment environment;
51     
52     	/**
53     	 * Creates an instance of this class.
54     	 *
55     	 * @param env on which this processor should operate on.
56     	 * @param classesToRetain is the collection of FQN of classes that need to be retained in the slice.
57     	 * @pre env != null and classesToRetain != null and classesToRetain.oclIsKindOf(Collection(String))
58     	 */
59     	public ExecutableSlicePostProcessorAndModifier(final IEnvironment env, final Collection<String> classesToRetain) {
60     		environment = env;
61     		retain = classesToRetain;
62     	}
63     
64     	/**
65     	 * @see ExecutableSlicePostProcessor#getClassHierarchyContainingClasses(Collection)
66     	 */
67     	@Override protected IClassHierarchy getClassHierarchyContainingClasses(final Collection<SootClass> classes) {
68     		final Collection<SootClass> _cl = new ArrayList<SootClass>(classes);
69     		final ClassHierarchy _ch = ClassHierarchy.createClassHierarchyFrom(_cl);
70     		final Iterator<String> _i = retain.iterator();
71     		final int _iEnd = retain.size();
72     
73     		for (int _iIndex = 0; _iIndex < _iEnd; _iIndex++) {
74     			final String _scName = _i.next();
75     
76     			if (environment.hasClass(_scName)) {
77     				final SootClass _sc = environment.getClass(_scName);
78     				collector.includeInSlice(_sc);
79     				_cl.add(_sc);
80     			}
81     		}
82     		_ch.confine(_cl, true);
83     		_ch.updateEnvironment();
84     		return _ch;
85     	}
86     }
