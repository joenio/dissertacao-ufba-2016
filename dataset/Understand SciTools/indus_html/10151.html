17     package edu.ksu.cis.indus.slicer;
18     
19     import edu.ksu.cis.indus.annotations.Functional;
20     import edu.ksu.cis.indus.annotations.NonNull;
21     
22     /**
23      * The enumeration of types of slices.
24      * 
25      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
26      * @author $Author: rvprasad $
27      * @version $Revision: 1.3 $
28      */
29     public enum SliceType {
30     	/**
31     	 * Backward slice request.
32     	 */
33     	BACKWARD_SLICE,
34     
35     	/**
36     	 * Complete slice request.
37     	 */
38     	COMPLETE_SLICE,
39     
40     	/**
41     	 * Forward slice request.
42     	 */
43     	FORWARD_SLICE;
44     
45     	/**
46     	 * Provides the name of the slice type.
47     	 * 
48     	 * @param v is the slice type of interest.
49     	 * @return the name of the slice type.
50     	 */
51     	@NonNull @Functional public static String name(@NonNull final SliceType v) {
52     		return v.name();
53     	}
54     }
