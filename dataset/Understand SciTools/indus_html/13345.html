17     package edu.ksu.cis.indus.common.graph;
18     
19     import edu.ksu.cis.indus.annotations.Immutable;
20     import edu.ksu.cis.indus.annotations.NonNull;
21     import edu.ksu.cis.indus.annotations.NonNullContainer;
22     import edu.ksu.cis.indus.common.datastructures.Pair;
23     
24     import java.util.Collection;
25     import java.util.List;
26     
27     /**
28      * This is an interface to a directed graph in which nodes are represented by <code>INode</code> objects. The nodes in the
29      * graph are to be ordered. The concrete implementation can determine the ordering, but it needs to be fixed over the lifetime
30      * of the graph.
31      * <p>
32      * A node <i>a</i> is reachable from node <i>b</i> if there is an explicit edge from <i>a</i> to <i>b</i>. This also holds
33      * when <i>a=b</i>.
34      * </p>
35      * 
36      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
37      * @author $Author: rvprasad $
38      * @version $Revision: 1.23 $ $Date: 2007/10/15 19:09:41 $
39      * @param <N> is the type of the node in this graph.
40      */
41     public interface IDirectedGraph<N extends INode<N>> {
42     
43     	/**
44     	 * Retrieves the back edges in this graph.
45     	 * 
46     	 * @return a collection of pairs containing the backedge. The source and the destination nodes of the edge are the first
47     	 *         and the secondelement of the pair, respectively.
48     	 * @post getNodes().contains(result.getFirst()) and getNodes().contains(result.getFirst())
49     	 */
50     	@NonNull @NonNullContainer Collection<Pair<N, N>> getBackEdges();
51     
52     	/**
53     	 * Returns the intersection of the nodes reachable from the given nodes in the given direction. This is equivalent to
54     	 * <code>CollectionUtils.intersection(getReachablesFrom(node1, forward1), getReachablesFrom(node2, forward2))</code>.
55     	 * 
56     	 * @param node1 of interest.
57     	 * @param forward1 direction of reachability from <code>node1</code>.
58     	 * @param node2 of interest.
59     	 * @param forward2 direction of reachability from <code>node2</code>.
60     	 * @return a collection of nodes.
61     	 * @post getReachablesFrom(node2, forward2)->forall(o | getReachablesFrom(node1, forward2).contains(o) implies
62     	 *       result.contains(o))
63     	 * @post result->forall(o | getReachablesFrom(node2, forward2).contains(o) and getReachableFrom(node1,
64     	 *       forward1).contains(o))
65     	 */
66     	@NonNull @NonNullContainer Collection<N> getCommonReachablesFrom(@NonNull @Immutable N node1, boolean forward1,
67     			@NonNull @Immutable N node2, boolean forward2);
68     
69     	/**
70     	 * Returns the minimum set of nodes that are needed to ensure that the given nodes remain reachable from or can reach
71     	 * common nodes. The removal of this set of nodes will ensure that there are no common ancestors (descendants) of the
72     	 * given nodes when forward is true (false).
73     	 * 
74     	 * @param node1 is one node of interest.
75     	 * @param node2 is another node of interest.
76     	 * @param forward <code>true</code> indicates that common nodes that can reach the given nodes are required;
77     	 *            <code>false</code> indicates that the common nodes that are reachable from the given nodes are required.
78     	 * @return a collection of nodes.
79     	 * @post getNodes().containsAll(result)
80     	 * @post result->forall(o | isReachable(o, node1, forward) and isReachable(o, node2, forward))
81     	 */
82     	@NonNull @NonNullContainer Collection<N> getConnectivityNodesFor(@NonNull @Immutable N node1,
83     			@NonNull @Immutable N node2, boolean forward);
84     
85     	/**
86     	 * Returns the cycles that occur in the graph.
87     	 * 
88     	 * @return a collection of list of nodes which form cycles in this graph. The head of the list is the initiator/head of
89     	 *         the cycle.
90     	 * @post result->forall(o | getNodes().containsAll(o))
91     	 */
92     	@NonNull @NonNullContainer Collection<List<N>> getCycles();
93     
94     	/**
95     	 * Returns the directed-acyclic graph of this graph. The objects in the nodes in the returned graph are nodes in this
96     	 * graph. For each edge in the returned graph, there will be an edges between the nodes corresponding to the source and
97     	 * destination nodes in this graph.
98     	 * 
99     	 * @return a DAG.
100     	 */
101     	@NonNull IObjectDirectedGraph<? extends IObjectNode<?, N>, N> getDAG();
102     
103     	/**
104     	 * Retrieves the nodes in the graph. The order of the nodes should be the same across calls to this method, if no nodes
105     	 * are added or removed. In case of addition and removal, the relative ordering between the old nodes should not change
106     	 * across calls.
107     	 * 
108     	 * @return the nodes(<code>INode</code>) in the graph.
109     	 */
110     	@NonNull @NonNullContainer List<N> getNodes();
111     
112     	/**
113     	 * Retrieves the nodes that occur on the paths between the given nodes.
114     	 * 
115     	 * @param nodes of interest.
116     	 * @return a collection of nodes that occur on path between the given nodes.
117     	 */
118     	@NonNull @NonNullContainer Collection<N> getNodesOnPathBetween(@NonNull @Immutable @NonNullContainer Collection<N> nodes);
119     
120     	/**
121     	 * Retrieves the nodes reachable from <code>root</code> in the requested direction.
122     	 * 
123     	 * @param root is the node from which the reachability needs to be calculated.
124     	 * @param forward <code>true</code> indicates that nodes reachable by following the edges from <code>root</code> are
125     	 *            requirested. <code>false</code> indicates that nodes reachables by following the edges in the reverse
126     	 *            direction from <code>root</code> are required.
127     	 * @return the collection of reachable nodes.
128     	 */
129     	@NonNull @NonNullContainer Collection<N> getReachablesFrom(@NonNull @Immutable N root, boolean forward);
130     
131     	/**
132     	 * Returns a sequence of strongly-connected components in this graph.
133     	 * 
134     	 * @param topDown <code>true</code> indicates returned sccs should be in the top-down order; <code>false</code>,
135     	 *            indicates bottom-up.
136     	 * @return a sequence of <code>List</code> of <code>INode</code>s that form SCCs in this graph.
137     	 * @post result->forall(o | getNodes().containsAll(o))
138     	 */
139     	@NonNull @NonNullContainer List<List<N>> getSCCs(boolean topDown);
140     
141     	/**
142     	 * Retrieves the sink nodes of this graph.
143     	 * 
144     	 * @return the sink nodes(<code>INode</code>) of this graph.
145     	 * @post result->forall(o | o.getSuccsOf()->size() == 0) and getNodes().containsAll(result)
146     	 */
147     	@NonNull @NonNullContainer Collection<N> getSinks();
148     
149     
150     	/**
151     	 * Retrieves a spanning tree of this graph.
152     	 * 
153     	 * @return a spanning tree in which each node is associated with a node in this graph.
154     	 */
155     	@NonNull IObjectDirectedGraph<?, N> getSpanningForest();
156     	
157     	/**
158     	 * Retrieves the source nodes of this graph.
159     	 * 
160     	 * @return the source nodes(<code>INode</code>) of this graph.
161     	 * @post result->forall(o | o.getPredsOf()->size() == 0) and getNodes().containsAll(result)
162     	 */
163     	@NonNull @NonNullContainer Collection<N> getSources();
164     
165     	/**
166     	 * Retrieves the tails of the given graph. A Tail is a sink node or an arbitrary node in a strongly-connected component
167     	 * that acts as a sink (only the nodes of the SCC are reachable from the nodes in the SCC).
168     	 * 
169     	 * @return a collection of nodes.
170     	 * @post result->forall(o | getNodes().contains(o))
171     	 * @post result.containsAll(getSinks()) and getNodes().containsAll(result)
172     	 */
173     	@NonNull @NonNullContainer Collection<N> getTails();
174     
175     	/**
176     	 * Checks if there is any node that can be reached from the give nodes in the given direction.
177     	 * 
178     	 * @param node1 is a node of interest.
179     	 * @param forward1 <code>true</code> indicates forward direction; <code>false</code> indicates backward direction.
180     	 * @param node2 is another node of interest.
181     	 * @param forward2 <code>true</code> indicates forward direction; <code>false</code> indicates backward direction.
182     	 * @return <code>true</code> if there is common node that is reachable from <code>node1</code> in
183     	 *         <code>forward1</code> direction and is reachable from <code>node2</code> in <code>forward2</code>
184     	 *         direction.
185     	 */
186     	boolean hasCommonReachablesFrom(@NonNull @Immutable N node1, boolean forward1, @NonNull @Immutable N node2,
187     			boolean forward2);
188     
189     	/**
190     	 * Checks if the given nodes have ancestral relationship. A node is considered as an ancestor of itself.
191     	 * 
192     	 * @param ancestor in the relationship.
193     	 * @param descendent in the relationship.
194     	 * @return <code>true</code> if <code>ancestor</code> is the ancestor of <code>descendent</code>;
195     	 *         <code>false</code>, otherwise.
196     	 * @pre getNodes().contains(ancestor) and getNodes().contains(descendent)
197     	 */
198     	boolean isAncestorOf(@NonNull @Immutable N ancestor, @NonNull @Immutable N descendent);
199     
200     	/**
201     	 * Checks if the given destination node is reachable from the given source node in the given direction via outgoing edges.
202     	 * 
203     	 * @param src is the source node in the graph.
204     	 * @param dest is the destination node in the graph.
205     	 * @param forward <code>true</code> indicates by forward traversal; <code>false</code> indicates backward traversal.
206     	 * @return <code>true</code> if <code>dest</code> is reachable from <code>src</code> in the given direction;
207     	 *         <code>false</code>, otherwise.
208     	 * @pre getNodes().contains(src) and getNodes().contains(dest)
209     	 */
210     	boolean isReachable(@NonNull @Immutable N src, @NonNull @Immutable N dest, boolean forward);
211     
212     	/**
213     	 * Performs (pseudo-)topological sort of the given nodes in the given direction.
214     	 * 
215     	 * @param topdown <code>true</code> indicates follow the forward edges while sorting; <code>false</code> indicates
216     	 *            follow the backward edges.
217     	 * @return a list containing the nodes but in the sorted order.
218     	 * @post getNodes().containsAll(result)
219     	 */
220     	@NonNull @NonNullContainer List<N> performTopologicalSort(boolean topdown);
221     }
