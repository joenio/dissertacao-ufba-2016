17     package edu.ksu.cis.indus.common.soot;
18     
19     import edu.ksu.cis.indus.annotations.Functional;
20     import edu.ksu.cis.indus.annotations.NonNull;
21     import edu.ksu.cis.indus.annotations.NonNullContainer;
22     
23     import java.util.ArrayList;
24     import java.util.Collection;
25     import java.util.Collections;
26     
27     import soot.jimple.JimpleBody;
28     
29     /**
30      * This class provides <code>ExceptionFlowSensitiveStmtGraph</code>s.
31      * 
32      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
33      * @author $Author: rvprasad $
34      * @version $Revision: 1.16 $ $Date: 2007/02/10 19:08:36 $
35      */
36     public class ExceptionFlowSensitiveStmtGraphFactory
37     		extends AbstractStmtGraphFactory<ExceptionFlowSensitiveStmtGraph> {
38     
39     	/**
40     	 * The collection of exception names that are relevant while dealing with synchronization constructs.
41     	 */
42     	public static final Collection<String> SYNC_RELATED_EXCEPTIONS = Collections.singleton("java.lang.Throwable");
43     
44     	/**
45     	 * The names of the exceptions via which the control flow should be ignored.
46     	 */
47     	@NonNullContainer @NonNull private Collection<String> exceptionsToIgnore = new ArrayList<String>();
48     
49     	/**
50     	 * This flag indicates if the unit graph should be like complete unit graph or like trap unit graph in terms considering
51     	 * control from the statement before the try block.
52     	 */
53     	private final boolean flag;
54     
55     	/**
56     	 * Creates a new instance of this class. This is identical to calling <code>new
57     	 * ExceptionFlowSensitiveStmtGraphFactory(SYNC_RELATED_EXCEPTIONS, true)</code>.
58     	 */
59     	public ExceptionFlowSensitiveStmtGraphFactory() {
60     		this(SYNC_RELATED_EXCEPTIONS, true);
61     	}
62     
63     	/**
64     	 * Creates a new ExceptionFlowSensitiveStmtGraphFactory object.
65     	 * 
66     	 * @param namesOfExceptionToIgnore are the fully qualified names of the exceptions that determine the control edges to be
67     	 *            ignored.
68     	 * @param dontAddEdgeFromStmtBeforeAreaOfProtectionToCatchBlock <code>true</code> indicates if the edge from the unit
69     	 *            before the unit that begins the trap protected region to the handler unit should be omitted;
70     	 *            <code>false</code>, otherwise.
71     	 * @pre namesOfExceptionToIgnore->forall(o | ClassLoader.getSystemClassLoader().loadClass(o) != null)
72     	 */
73     	public ExceptionFlowSensitiveStmtGraphFactory(
74     			@NonNull @NonNullContainer final Collection<String> namesOfExceptionToIgnore,
75     			final boolean dontAddEdgeFromStmtBeforeAreaOfProtectionToCatchBlock) {
76     		flag = dontAddEdgeFromStmtBeforeAreaOfProtectionToCatchBlock;
77     		exceptionsToIgnore.addAll(namesOfExceptionToIgnore);
78     	}
79     
80     	/**
81     	 * {@inheritDoc}
82     	 */
83     	@Functional @Override protected ExceptionFlowSensitiveStmtGraph getStmtGraphForBody(@NonNull final JimpleBody body) {
84     		return new ExceptionFlowSensitiveStmtGraph(body, exceptionsToIgnore, flag);
85     	}
86     }
