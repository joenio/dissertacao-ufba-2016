17     package edu.ksu.cis.indus.staticanalyses.flow.modes.sensitive.allocation;
18     
19     import edu.ksu.cis.indus.processing.Context;

24     import java.util.regex.Pattern;
25     
26     import org.slf4j.Logger;
27     import org.slf4j.LoggerFactory;
28     
29     import soot.ArrayType;
30     import soot.RefType;
31     import soot.Type;
32     import soot.Value;

34     /**
35      * This class manages indices associated with fields and array components in allocation-site sensitive mode. In reality, it
36      * provides the implementation to create new indices. Created: Tue Mar 5 14:08:18 2002.
37      * 
38      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
39      * @version $Revision: 1.15 $
40      * @param <E> is the type of the entity that has been indexed.
41      */
42     public class AllocationSiteSensitiveIndexManager<E>
43     		extends AbstractIndexManager<OneContextInfoIndex<E, Object>, E> {
44     
45     	/**
46     	 * The logger used by instances of this class to log messages.
47     	 */
48     	private static final Logger LOGGER = LoggerFactory.getLogger(AllocationSiteSensitiveIndexManager.class);
49     
50     	/**
51     	 * The pattern that defines an object-based scope in which value flow through fields is object sensitive.
52     	 */
53     	private final Pattern pattern;
54     
55     	/**
56     	 * This indicates if value flow through arrays should be object sensitive.
57     	 */
58     	private final boolean objectSensitiveArrayTracking;
59     
60     	/**
61     	 * Creates an instance of this class.
62     	 */
63     	public AllocationSiteSensitiveIndexManager() {
64     		final String _p = Constants.getObjectSensitivityScopePattern();
65     
66     		if (_p != null) {
67     			pattern = Pattern.compile(_p);
68     		} else {
69     			pattern = null;
70     		}
71     
72     		objectSensitiveArrayTracking = Constants.getObjectSensitiveArrayTracking();
73     	}
74     
75     	/**
76     	 * Returns an index corresponding to the given entity and context.
77     	 * 
78     	 * @param o the entity for which the index in required. Although it is not enforced, this should be of type
79     	 *            <code>FielRef</code> or <code>ArrayRef</code>.
80     	 * @param c the context in which information pertaining to <code>o</code> needs to be captured.
81     	 * @return the index that uniquely identifies <code>o</code> in context, <code>c</code>.
82     	 * @pre o != null and c != null and
83     	 *      c.oclIsTypeOf(edu.ksu.cis.indus.staticanalyses.flow.modes.sensitive.allocation.AllocationContext)
84     	 */
85     	@Override protected OneContextInfoIndex<E, Object> createIndex(final E o, final Context c) {
86     		if (LOGGER.isDebugEnabled()) {
87     			LOGGER.debug("Getting index for " + o + " in " + c);
88     		}
89     
90     		final AllocationContext _ctxt = (AllocationContext) c;
91     		final Type _type = ((Value) _ctxt.allocationSite).getType();
92     		if ((_type instanceof RefType && (pattern == null || pattern.matcher(((RefType) _type).getClassName()).matches()))
93     				|| (objectSensitiveArrayTracking && _type instanceof ArrayType)) {
94     			return new OneContextInfoIndex<E, Object>(o, _ctxt.getAllocationSite());
95     		}
96     		return new OneContextInfoIndex<E, Object>(o, null);
97     	}
98     }
