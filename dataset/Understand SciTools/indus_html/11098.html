17     package edu.ksu.cis.indus.staticanalyses.dependency;
18     
19     import edu.ksu.cis.indus.annotations.Empty;
20     import edu.ksu.cis.indus.common.datastructures.Pair;
21     
22     import java.util.Collection;
23     import java.util.HashSet;
24     
25     /**
26      * This implementation of <code>IDependenceRetriever</code> is used in instance when the dependence information is provided
27      * in terms of statements and method pair. Use this in cases such as ready dependence.
28      * 
29      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
30      * @author $Author: rvprasad $
31      * @version $Revision: 1.7 $ $Date: 2007/02/10 19:07:01 $
32      * @param <E1> is the type of dependee object in the context of dependee-to-dependent info maintenance.
33      * @param <C1> is the type of context object in the context of dependee-to-dependent info maintenance.
34      * @param <T1> is the type of dependent object in the context of dependee-to-dependent info maintenance.
35      * @param <T2> is the type of dependent object in the context of dependent-to-dependee info maintenance.
36      * @param <C2> is the type of context object in the context of dependent-to-dependee info maintenance.
37      * @param <E2> is the type of dependee object in the context of dependent-to-dependee info maintenance.
38      */
39     final class PairRetriever<T1, C1, E1, E2, C2, T2>
40     		extends AbstractDependenceRetriever<T1, C1, Pair<E1, C1>, E2, C2, Pair<T2, C2>> {
41     
42     	/**
43     	 * Creates an instance of this class.
44     	 */
45     	@Empty public PairRetriever() {
46     		// does nothing
47     	}
48     
49     	/**
50     	 * @see IDependenceRetriever#convertToConformantDependees(java.util.Collection, java.lang.Object, java.lang.Object)
51     	 */
52     	public Collection<Pair<E2, C2>> convertToConformantDependees(Collection<Pair<T2, C2>> dependents, final E2 base,
53     			final C2 context) {
54     		final Collection<Pair<E2, C2>> _result = new HashSet<Pair<E2, C2>>();
55     		for (final Pair<T2, C2> _pair : dependents) {
56     			_result.add((Pair<E2, C2>) _pair);
57     		}
58     		return _result;
59     	}
60     
61     	/**
62     	 * @see IDependenceRetriever#convertToConformantDependents(java.util.Collection, java.lang.Object, java.lang.Object)
63     	 */
64     	public Collection<Pair<T1, C1>> convertToConformantDependents(final Collection<Pair<E1, C1>> dependees, final T1 base,
65     			final C1 context) {
66     		final Collection<Pair<T1, C1>> _result = new HashSet<Pair<T1, C1>>();
67     		for (final Pair<E1, C1> _pair : dependees) {
68     			_result.add((Pair<T1, C1>) _pair);
69     		}
70     		return _result;
71     	}
72     }
