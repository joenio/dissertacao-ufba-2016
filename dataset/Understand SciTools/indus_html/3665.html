49     /**
50      * This implementation calculates call information based on class-hierarchy information.
51      *
52      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
53      * @author $Author: rvprasad $
54      * @version $Revision: 1.11 $ $Date: 2007/02/10 19:07:01 $
55      */
56     public final class CHABasedCallInfoCollector
57     		extends AbstractProcessor
58     		implements ICallInfoCollector {
59     
60     	/**
61     	 * The logger used by instances of this class to log messages.
62     	 */
63     	private static final Logger LOGGER = LoggerFactory.getLogger(CHABasedCallInfoCollector.class);
64     
65     	/**
66     	 * This holds call information.
67     	 */
68     	private final CallInfo callInfoHolder = new CallInfo();
69     
70     	/**
71     	 * The class hierarchy analysis to be used.
72     	 */
73     	private IClassHierarchy cha;
74     
75     	/**
76     	 * This maps invoked methods (not resolved) to caller-site triple.
77     	 */
78     	private final Map<SootMethod, Collection<CallTriple>> invokedMethod2callerTriple = new HashMap<SootMethod, Collection<CallTriple>>();
79     
80     	/**
81     	 * @see edu.ksu.cis.indus.processing.AbstractProcessor#callback(soot.SootMethod)
82     	 */
83     	@Override public void callback(final SootMethod method) {
84     		callInfoHolder.addReachable(method);
85     	}
86     
87     	/**
88     	 * @see edu.ksu.cis.indus.processing.AbstractProcessor#callback(soot.ValueBox, edu.ksu.cis.indus.processing.Context)
89     	 */
90     	@Override public void callback(final ValueBox vBox, final Context context) {
91     		if (LOGGER.isDebugEnabled()) {
92     			LOGGER.debug("callback(ValueBox vBox = " + vBox + ", Context context = " + context + ") - BEGIN");
93     		}
94     
95     		final Value _val = vBox.getValue();
96     		final Stmt _stmt = context.getStmt();
97     		final SootMethod _caller = context.getCurrentMethod();
98     		final InvokeExpr _expr = (InvokeExpr) _val;
99     		final SootMethod _callee = _expr.getMethod();
100     		final Collection<CallTriple> _callees = MapUtils.getCollectionFromMap(callInfoHolder.caller2callees, _caller);
101     		final CallTriple _callerTriple = new CallTriple(_caller, _stmt, _expr);
102     
103     		if (_expr instanceof StaticInvokeExpr
104     				|| (_expr instanceof SpecialInvokeExpr && (_callee.getName().equals("<init>") || _callee.isPrivate()))) {
105     			final Collection<CallTriple> _callers = MapUtils.getCollectionFromMap(callInfoHolder.callee2callers, _callee);
106     			_callers.add(_callerTriple);
107     			_callees.add(new CallTriple(_callee, _stmt, _expr));
108     		} else {
109     			MapUtils.putIntoCollectionInMap(invokedMethod2callerTriple, _callee, _callerTriple);
110     		}
111     
112     		if (LOGGER.isDebugEnabled()) {
113     			LOGGER.debug("callback() - END");
114     		}
115     	}
116     
117     	/**
118     	 * @see edu.ksu.cis.indus.processing.AbstractProcessor#consolidate()
119     	 */
120     	@Override public void consolidate() {
121     		final Collection<SootMethod> _temp = new HashSet<SootMethod>();
122     		final Collection<SootClass> _implClasses = new HashSet<SootClass>();
123     		final Set<Map.Entry<SootMethod, Collection<CallTriple>>> _entrySet = invokedMethod2callerTriple.entrySet();
124     		final Iterator<Map.Entry<SootMethod, Collection<CallTriple>>> _j = _entrySet.iterator();
125     		final int _jEnd = _entrySet.size();
126     
127     		for (int _jIndex = 0; _jIndex < _jEnd; _jIndex++) {
128     			final Map.Entry<SootMethod, Collection<CallTriple>> _entry = _j.next();
129     			final SootMethod _invokedMethod = _entry.getKey();
130     			final Collection<CallTriple> _callerTriples = _entry.getValue();
131     			final String _invokedMethodSubSignature = _invokedMethod.getSubSignature();
132     			final SootClass _declInterface = _invokedMethod.getDeclaringClass();
133     			_implClasses.clear();
134     			_implClasses.addAll(cha.getProperSubclassesOf(_declInterface));
135     			_implClasses.add(_declInterface);
136     
137     			final Iterator<SootClass> _i = _implClasses.iterator();
138     			final int _iEnd = _implClasses.size();
139     
140     			_temp.clear();
141     
142     			for (int _iIndex = 0; _iIndex < _iEnd; _iIndex++) {
143     				final SootClass _impl = _i.next();
144     
145     				if (_impl.declaresMethod(_invokedMethodSubSignature)
146     						&& !_impl.getMethod(_invokedMethodSubSignature).isAbstract()) {
147     					final SootMethod _implMethod = _impl.getMethod(_invokedMethodSubSignature);
148     					MapUtils.putAllIntoCollectionInMap(callInfoHolder.callee2callers, _implMethod, _callerTriples);
149     					_temp.add(_implMethod);
150     				}
151     			}
152     
153     			final Iterator<CallTriple> _k = _callerTriples.iterator();
154     			final int _kEnd = _callerTriples.size();
155     
156     			for (int _kIndex = 0; _kIndex < _kEnd; _kIndex++) {
157     				final CallTriple _callerTriple = _k.next();
158     				final SootMethod _caller = _callerTriple.getMethod();
159     				final Stmt _stmt = _callerTriple.getStmt();
160     				final InvokeExpr _expr = _callerTriple.getExpr();
161     				final Collection<CallTriple> _callees = MapUtils.getCollectionFromMap(callInfoHolder.caller2callees, _caller);
162     				final Iterator<SootMethod> _l = _temp.iterator();
163     				final int _lEnd = _temp.size();
164     
165     				for (int _lIndex = 0; _lIndex < _lEnd; _lIndex++) {
166     					final SootMethod _callee = _l.next();
167     					_callees.add(new CallTriple(_callee, _stmt, _expr));
168     				}
169     			}
170     		}
171     
172     		callInfoHolder.fixupMethodsHavingZeroCallersAndCallees();
173     
174     		invokedMethod2callerTriple.clear();
175     
176     		stable();
177     
178     		if (LOGGER.isInfoEnabled()) {
179     			LOGGER.info("END - Call info collection");
180     		}
181     	}
182     
183     	/**
184     	 * @see edu.ksu.cis.indus.staticanalyses.callgraphs.ICallInfoCollector#getCallInfo()
185     	 */
186     	public CallGraphInfo.ICallInfo getCallInfo() {
187     		return callInfoHolder;
188     	}

190     	/**
191     	 * @see edu.ksu.cis.indus.processing.IProcessor#hookup(edu.ksu.cis.indus.processing.ProcessingController)
192     	 */
193     	public void hookup(final ProcessingController ppc) {
194     		ppc.register(this);
195     		ppc.register(InterfaceInvokeExpr.class, this);
196     		ppc.register(SpecialInvokeExpr.class, this);
197     		ppc.register(StaticInvokeExpr.class, this);
198     		ppc.register(VirtualInvokeExpr.class, this);
199     	}

202     	 * Creates an instance of this class.

204     	 * @param analysis to be used.
205     	 */

207     		this.cha = analysis;
208     	}

214     		unstable();
215

217     			LOGGER.info("BEGIN - Call info collection");
218     		}
219     	}

234     		ppc.unregister(this);
235     		ppc.unregister(InterfaceInvokeExpr.class, this);
236     		ppc.unregister(SpecialInvokeExpr.class, this);
237     		ppc.unregister(StaticInvokeExpr.class, this);
238     		ppc.unregister(VirtualInvokeExpr.class, this);
239     	}
