58     /**
59      * This class provides interference dependency information. This implementation refers to the technical report <a
60      * href="http://www.cis.ksu.edu/santos/papers/technicalReports.html">A Formal Study of Slicing for Multi-threaded Program with
61      * JVM Concurrency Primitives"</a>.
62      * <p>
63      * The calculated information is very pessimistic. This analysis uses points-to analysis to check if the primaries of two
64      * field references may be aliased.
65      * </p>
66      * <p>
67      * This analysis assumes that there can be no interference between field references in class initializers. This is true for
68      * good programs.
69      * </p>
70      * <p>
71      * This analysis should be <code>setup</code> before preprocessing.
72      * </p>
73      * 
74      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
75      * @author $Author: rvprasad $
76      * @version $Revision: 1.71 $
77      */

205     		}

207     		// we return immediately if there are no start sites in the system.

221     					.get(_o);

223     					.get(_o);
