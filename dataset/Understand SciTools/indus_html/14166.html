16     package edu.ksu.cis.indus.common.collections;
17     
18     import edu.ksu.cis.indus.annotations.Functional;
19     import edu.ksu.cis.indus.annotations.NonNull;
20     
21     /**
22      * This predicate performs <code>instanceof</code> check.
23      * 
24      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
25      * @author $Author: rvprasad $
26      * @version $Revision: 1.7 $ $Date: 2007/10/09 15:59:13 $
27      * @param <T> is the type in the instanceof check.
28      * @param <V> is the type of value that participates in the instanceof check.
29      */
30     public class InstanceOfPredicate<T, V>
31     		implements IPredicate<V> {
32     
33     	/**
34     	 * The class of the type that participates in the instanceof check.
35     	 */
36     	private Class<T> clazz;
37     
38     	/**
39     	 * Creates an instance of this class.
40     	 * 
41     	 * @param instanceClass is the class of the type involved in the instanceof check.
42     	 */
43     	public InstanceOfPredicate(@NonNull final Class<T> instanceClass) {
44     		super();
45     		clazz = instanceClass;
46     	}
47     
48     	/**
49     	 * {@inheritDoc}
50     	 */
51     	@Functional public  boolean evaluate(final V t) {
52     		return clazz.isInstance(t);
53     	}
54     
55     }
