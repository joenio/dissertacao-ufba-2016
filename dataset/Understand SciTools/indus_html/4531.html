64     	 * Tests <code>hashCode</code> and <code>equals</code>.
65     	 */
66     	public final void testHashCodeAndEquals() {
67     		final Pair _p3 = new Pair("first", "second");
68     		assertTrue(pair1.hashCode() != pair2.hashCode());
69     		assertTrue(pair1.hashCode() == _p3.hashCode());
70     		assertFalse(pair1.equals(pair2));
71     		assertTrue(pair1.equals(_p3));
72     
73     		final Pair _p4 = new Pair(null, null);
74     		final Pair _p5 = new Pair(null, null);
75     		assertTrue(_p4.equals(_p5));
76     		assertTrue(_p4.hashCode() == _p5.hashCode());
77     		assertFalse(_p4.equals("hi"));
78     	}

80     	/**
81     	 * Tests <code>mapify</code>.
82     	 */
83     	public final void testMapify() {
84     		final Collection _temp = new ArrayList();
85     		_temp.add(pair1);
86     		_temp.add(pair2);
87     
88     		Map _map = Pair.mapify(_temp, true);
89     		final Collection _c1 = (Collection) _map.get(pair1.getFirst());
90     		assertTrue(_c1.size() == 1);
91     		assertTrue(_c1.contains(pair1.getSecond()));
92     
93     		final Collection _c2 = (Collection) _map.get(pair2.getFirst());
94     		assertTrue(_c2.size() == 1);
95     		assertTrue(_c2.contains(pair2.getSecond()));
96     
97     		final Pair _p1 = new Pair("first", "fifth");
98     		_temp.add(_p1);
99     		_map = Pair.mapify(_temp, true);
100     		assertTrue(_map.get(pair1.getFirst()) instanceof Collection);
101     		assertTrue(((Collection) _map.get(pair1.getFirst())).contains(_p1.getSecond()));
102     		assertTrue(((Collection) _map.get(pair1.getFirst())).contains(pair1.getSecond()));
103     
104     		_map = Pair.mapify(_temp, false);
105     		assertTrue(((Collection) _map.get(pair1.getSecond())).contains(pair1.getFirst()));
106     		assertTrue(((Collection) _map.get(_p1.getSecond())).contains(_p1.getFirst()));
107     		assertTrue(((Collection) _map.get(pair2.getSecond())).contains(pair2.getFirst()));
108     
109     		assertTrue(Pair.mapify(Collections.EMPTY_LIST, true).isEmpty());
110     	}

112     	/**
113     	 * Tests <code>optimize</code> and <code>unoptimize</code>.
114     	 */
115     	public final void testOptimizeAndUnoptimize1() {
116     		final Collection _second = new ArrayList();
117     		_second.add("first");
118     
119     		final Pair _p1 = new Pair("first", _second, true, false);
120     		final int _hash1 = _p1.hashCode();
121     		_second.add("second");
122     
123     		assertTrue(_p1.hashCode() == _hash1);
124     		_p1.unoptimizeHashCode();
125     		assertTrue(_p1.hashCode() != _hash1);
126     		_p1.optimizeHashCode();
127     		assertTrue(_p1.hashCode() != _hash1);
128     	}

135     		_second.append("first");

150     	 */
151     	public final void testPairManager() {
152     		final PairManager _pmgr = new PairManager(true, true);
153     		final Pair _p1 = _pmgr.getPair("first", "second");
154     		final Pair _p2 = _pmgr.getPair("first", "second");
155     		final Pair _p3 = _pmgr.getPair("first", "second");
156     		final Pair _p4 = _pmgr.getPair("first", "second");
157     		assertTrue(_p1 == _p2);
158     		assertTrue(_p3 == _p4);
159     		assertTrue(_p1 == _p3);
160     		_pmgr.reset();
161     
162     		final Pair _p5 = _pmgr.getPair("first", "second");
163     		assertTrue(_p1 != _p5);
164     	}

168     	 */
169     	public final void testToString() {
170     		final StringBuffer _second = new StringBuffer();
171     		_second.append(true);

174     		_p1.optimizeToString();

186     		assertTrue(_str2.equals(_str3));
