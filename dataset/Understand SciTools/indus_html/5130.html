148     	/**
149     	 * Tests <code>pause()</code> and <code>resume()</code> in asynchronous mode.
150     	 */
151     	public final void testPauseAndResumeInAsyncMode() {
152     		try {
153     			testTool.pause();
154     		} catch (final Exception _e) {
155     			fail("Should be able to pause tool that is not running.");
156     		}
157     
158     		final Phase _ph = Phase.createPhase();
159     		testTool.pause();
160     		testTool.run(_ph, null, false);
161     		_ph.nextMinorPhase();
162     		assertTrue(testTool.localPH.isEarlierThan(_ph));
163     		testTool.resume();
164     
165     		while (!testTool.finished) {
166     			;
167     		}
168     		assertTrue(_ph.isEarlierThan(testTool.localPH));
169     	}
