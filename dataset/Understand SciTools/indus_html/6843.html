38     /**
39      * This is an abstract implementation of slice criteria contextualizer.
40      *
41      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
42      * @author $Author: rvprasad $
43      * @version $Revision: 1.8 $ $Date: 2007/02/10 19:06:03 $
44      */
45     public abstract class AbstractSliceCriteriaCallStackContextualizer
46       implements ISliceCriteriaContextualizer {
47     	/** 
48     	 * This is the slicer tool provides the context in which filtering occurs.
49     	 */
50     	private SlicerTool<?> slicerTool;
51     
52     	/**
53     	 * Sets the value of <code>slicerTool</code>.
54     	 *
55     	 * @param slicer the new value of <code>slicerTool</code>.
56     	 */
57     	public void setSlicerTool(final SlicerTool<?> slicer) {
58     		slicerTool = slicer;
59     	}
60     
61     	/**
62     	 * @see ISliceCriteriaContextualizer#processCriteriaBasedOnProgramPoint(Context, Collection)
63     	 */
64     	public final void processCriteriaBasedOnProgramPoint(final Context programPoint, final Collection<ISliceCriterion> baseCriteria) {
65     		final Collection<Stack<CallTriple>> _contexts = getCallingContextsForProgramPoint(programPoint);
66     		contextualize(baseCriteria, _contexts);
67     	}
68     
69     	/**
70     	 * @see ISliceCriteriaContextualizer#processCriteriaBasedOnThis(SootMethod, Collection)
71     	 */
72     	public final void processCriteriaBasedOnThis(final SootMethod method, final Collection<ISliceCriterion> baseCriteria) {
73     		final Collection<Stack<CallTriple>> _contexts = getCallingContextsForThis(method);
74     		contextualize(baseCriteria, _contexts);
75     	}
76     
77     	/**
78     	 * Retrieves the value in <code>slicerTool</code>.
79     	 *
80     	 * @return the value in <code>slicerTool</code>.
81     	 */
82     	protected final SlicerTool<?> getSlicerTool() {
83     		return slicerTool;
84     	}
85     
86     	/**
87     	 * Retrieves the calling contexts based on given program point.
88     	 *
89     	 * @param programPoint obviously.
90     	 *
91     	 * @return a collection of calling contexts.  The call triples in the calling contexts correspond to the caller side,
92     	 * 		   i.e, they contain caller and call-site information.
93     	 *
94     	 * @pre programPoint != null
95     	 * @post result != null
96     	 */
97     	protected abstract Collection<Stack<CallTriple>> getCallingContextsForProgramPoint(final Context programPoint);
98     
99     	/**
100     	 * Retrieves the calling contexts based on "this" variable of the given method.
101     	 *
102     	 * @param method in which "this" occurs.
103     	 *
104     	 * @return a collection of calling contexts. The call triples in the calling contexts correspond to the caller side, i.e,
105     	 * 		   they contain caller and call-site information.
106     	 *
107     	 * @pre method != null
108     	 * @post result != null
109     	 */
110     	protected abstract Collection<Stack<CallTriple>> getCallingContextsForThis(final SootMethod method);
111     
112     	/**
113     	 * Injects the given criterion with the given call stack and adds it to result.
114     	 *
115     	 * @param criterion to be modified.
116     	 * @param stack to be injected.
117     	 * @param result into which <code>criterion</code> needs to be added.
118     	 *
119     	 * @pre criterion != null and result != null
120     	 * @post result.containsAll(result$pre)
121     	 * @post stack.equals(stack$pre)
122     	 */
123     	private void addCriteriaWithGivenCallStackToResult(final ISliceCriterion criterion, final Stack<CallTriple> stack,
124     		final Collection<ISliceCriterion> result) {
125     		criterion.setCallStack(stack);
126     		result.add(criterion);
127     	}
128     
129     	/**
130     	 * Contextualizes the given criteria with given contexts.
131     	 *
132     	 * @param baseCriteria is the collection of criteria to be contextualized.
133     	 * @param contexts to be injected into the criteria.  <code>null</code> context indicates an open context.
134     	 *
135     	 * @pre contexts->forall(o | not o.isEmpty())
136     	 * @post baseCriteria->forall(o | not baseCriteria$pre.contains(o))
137     	 * @post baseCriteria$pre->forall(o | contexts->exists(p | p.peek().equals(o.getOccurringMethod()) and
138     	 * 		 baseCriteria.contains(o.setCallStack(p))) or baseCriteria->forall(t | not t.setCallStack(null).equals(o)))
139     	 */
140     	private void contextualize(final Collection<ISliceCriterion> baseCriteria, final Collection<Stack<CallTriple>> contexts) {
141     		final Collection<ISliceCriterion> _result = new HashSet<ISliceCriterion>();
142     		final SliceCriteriaFactory _criteriaFactory = SliceCriteriaFactory.getFactory();
143     		final Iterator<ISliceCriterion> _j = baseCriteria.iterator();
144     		final int _jEnd = baseCriteria.size();
145     		final ICallGraphInfo _cgi = getSlicerTool().getCallGraph();
146     
147     		for (int _jIndex = 0; _jIndex < _jEnd; _jIndex++) {
148     			final ISliceCriterion _criterion = _j.next();
149     			final Iterator<Stack<CallTriple>> _i = contexts.iterator();
150     			final int _iEnd = contexts.size();
151     
152     			for (int _iIndex = 0; _iIndex < _iEnd; _iIndex++) {
153     				final Stack<CallTriple> _callStack = _i.next();
154     				final ISliceCriterion _temp = _criteriaFactory.clone(_criterion);
155     
156     				if (_callStack == null) {
157     					addCriteriaWithGivenCallStackToResult(_temp, null, _result);
158     				} else if (!_callStack.isEmpty() && _cgi.getCallers(_temp.getOccurringMethod()).contains(_callStack.peek())) {
159                         addCriteriaWithGivenCallStackToResult(_temp, _callStack.clone(), _result);
160                     }
161     			}
162     			_j.remove();
163     		}
164     		baseCriteria.addAll(_result);
165     	}
166     }
