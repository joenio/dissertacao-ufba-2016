17     package edu.ksu.cis.indus.slicer;
18     
19     import edu.ksu.cis.indus.common.datastructures.Pair;
20     import edu.ksu.cis.indus.staticanalyses.dependency.IDependencyAnalysis;
21     
22     import java.util.Collection;
23     import java.util.HashSet;
24     
25     import soot.Local;
26     import soot.SootMethod;
27     import soot.ValueBox;
28     import soot.jimple.IdentityStmt;
29     import soot.jimple.Stmt;
30     
31     /**
32      * This class provides the logic to detect parts of a complete (backward and forward) slice.
33      * 
34      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
35      * @author $Author: rvprasad $
36      * @version $Revision: 1.13 $ $Date: 2007/02/10 19:05:58 $
37      */
38     public class CompleteSlicingPart
39     		implements IDirectionSensitivePartOfSlicingEngine {
40     
41     	/**
42     	 * This part provides the backward slice creation logic.
43     	 */
44     	private final BackwardSlicingPart backwardPart;
45     
46     	/**
47     	 * The engine with which this part is a part of.
48     	 */
49     	private final SlicingEngine engine;
50     
51     	/**
52     	 * This part provides the forward slice creation logic.
53     	 */
54     	private final ForwardSlicingPart forwardPart;
55     
56     	/**
57     	 * Creates an instance of this class.
58     	 * 
59     	 * @param theEngine of which this part is a part of.
60     	 * @pre theEngine != null
61     	 */
62     	public CompleteSlicingPart(final SlicingEngine theEngine) {
63     		engine = theEngine;
64     		backwardPart = new BackwardSlicingPart(engine);
65     		forwardPart = new ForwardSlicingPart(engine);
66     	}
67     
68     	/**
69     	 * {@inheritDoc}
70     	 * 
71     	 * @see edu.ksu.cis.indus.slicer.IDirectionSensitivePartOfSlicingEngine#continueProcessing()
72     	 */
73     	public boolean continueProcessing() {
74     		return backwardPart.continueProcessing() || forwardPart.continueProcessing();
75     	}
76     
77     	/**
78     	 * {@inheritDoc}
79     	 * 
80     	 * @see IDirectionSensitivePartOfSlicingEngine#generateCriteriaForTheCallToMethod(soot.SootMethod, soot.SootMethod,
81     	 *      soot.jimple.Stmt)
82     	 */
83     	public void generateCriteriaForTheCallToMethod(final SootMethod callee, final SootMethod caller, final Stmt callStmt) {
84     		backwardPart.generateCriteriaForTheCallToMethod(callee, caller, callStmt);
85     		forwardPart.generateCriteriaForTheCallToMethod(callee, caller, callStmt);
86     	}
87     
88     	/**
89     	 * {@inheritDoc}
90     	 * 
91     	 * @see edu.ksu.cis.indus.slicer.IDirectionSensitivePartOfSlicingEngine#generateCriteriaToIncludeCallees(soot.jimple.Stmt,
92     	 *      soot.SootMethod, java.util.Collection)
93     	 */
94     	public void generateCriteriaToIncludeCallees(final Stmt stmt, final SootMethod caller, final Collection callees) {
95     		backwardPart.generateCriteriaToIncludeCallees(stmt, caller, callees);
96     		forwardPart.generateCriteriaToIncludeCallees(stmt, caller, callees);
97     	}
98     
99     	/**
100     	 * {@inheritDoc}
101     	 * 
102     	 * @see DependenceExtractor.IDependenceRetriver#getDependences(IDependencyAnalysis, Object, SootMethod)
103     	 */
104     	public Collection<Object> getDependences(final IDependencyAnalysis analysis, final Object entity, final SootMethod method) {
105     		final Collection<Object> _result = new HashSet<Object>();
106     		if (entity instanceof Pair) {
107     			_result.addAll(backwardPart.getDependences(analysis, ((Pair) entity).getSecond(), method));
108     			_result.addAll(forwardPart.getDependences(analysis, ((Pair) entity).getSecond(), method));
109     		} else {
110     			_result.addAll(backwardPart.getDependences(analysis, entity, method));
111     			_result.addAll(forwardPart.getDependences(analysis, entity, method));			
112     		}
113     		return _result;
114     	}
115     
116     	/**
117     	 * {@inheritDoc}
118     	 * 
119     	 * @see DependenceExtractor.IDependenceRetriver#getEntityForIdentifierBasedDataDA(soot.Local, soot.jimple.Stmt)
120     	 */
121     	public Object getEntityForIdentifierBasedDataDA(final Local local, final Stmt stmt) {
122     		return new Pair<Object, Object>(backwardPart.getEntityForIdentifierBasedDataDA(local, stmt), forwardPart
123     				.getEntityForIdentifierBasedDataDA(local, stmt));
124     	}
125     
126     	/**
127     	 * {@inheritDoc}
128     	 * 
129     	 * @see IDirectionSensitivePartOfSlicingEngine#processLocalAt(Local, Stmt, SootMethod)
130     	 */
131     	public void processLocalAt(final Local local, final Stmt stmt, final SootMethod method) {
132     		backwardPart.processLocalAt(local, stmt, method);
133     		forwardPart.processLocalAt(local, stmt, method);
134     	}
135     
136     	/**
137     	 * {@inheritDoc}
138     	 * 
139     	 * @see IDirectionSensitivePartOfSlicingEngine#processNewExpr(Stmt, SootMethod)
140     	 */
141     	public void processNewExpr(final Stmt stmt, final SootMethod method) {
142     		backwardPart.processNewExpr(stmt, method);
143     		forwardPart.processNewExpr(stmt, method);
144     	}
145     
146     	/**
147     	 * {@inheritDoc}
148     	 * 
149     	 * @see IDirectionSensitivePartOfSlicingEngine#processParameterRef(IdentityStmt, SootMethod)
150     	 */
151     	public void processParameterRef(final IdentityStmt stmt, final SootMethod method) {
152     		backwardPart.processParameterRef(stmt, method);
153     		forwardPart.processParameterRef(stmt, method);
154     	}
155     
156     	/**
157     	 * {@inheritDoc}
158     	 * 
159     	 * @see edu.ksu.cis.indus.slicer.IDirectionSensitivePartOfSlicingEngine#reset()
160     	 */
161     	public void reset() {
162     		backwardPart.reset();
163     		forwardPart.reset();
164     	}
165     
166     	/**
167     	 * {@inheritDoc}
168     	 * 
169     	 * @see IDirectionSensitivePartOfSlicingEngine#retrieveValueBoxesToTransformExpr(ValueBox, Stmt)
170     	 */
171     	public Collection<ValueBox> retrieveValueBoxesToTransformExpr(final ValueBox valueBox, final Stmt stmt) {
172     		final Collection<ValueBox> _result = new HashSet<ValueBox>();
173     		_result.addAll(backwardPart.retrieveValueBoxesToTransformExpr(valueBox, stmt));
174     		_result.addAll(forwardPart.retrieveValueBoxesToTransformExpr(valueBox, stmt));
175     		return _result;
176     	}
177     
178     	/**
179     	 * {@inheritDoc}
180     	 * 
181     	 * @see IDirectionSensitivePartOfSlicingEngine#retrieveValueBoxesToTransformStmt(Stmt)
182     	 */
183     	public Collection<ValueBox> retrieveValueBoxesToTransformStmt(final Stmt stmt) {
184     		final Collection<ValueBox> _result = new HashSet<ValueBox>();
185     		_result.addAll(backwardPart.retrieveValueBoxesToTransformStmt(stmt));
186     		_result.addAll(forwardPart.retrieveValueBoxesToTransformStmt(stmt));
187     		return _result;
188     	}
189     }
