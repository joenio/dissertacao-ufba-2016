18     
19     import edu.ksu.cis.indus.staticanalyses.tokens.ITokens;
20     
21     import soot.SootMethod;

23     /**
24      * This is factory interface to create method variants in the flow analysis framework.
25      * 
26      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
27      * @author $Author: rvprasad $
28      * @version $Revision: 1.7 $
29      * @param <SYM> is the type of symbol whose flow is being analyzed.
30      * @param <T> is the type of the token set object.
31      * @param <N> is the type of the summary node in the flow analysis.
32      * @param <R> is the type of the symbol types.
33      */
34     public interface IMethodVariantFactory<SYM, T extends ITokens<T, SYM>, N extends IFGNode<SYM, T, N>, R> {
35     
36     	/**
37     	 * Creates a method variant.
38     	 * 
39     	 * @param sootMethod for which the method variant needs to be created.
40     	 * @param astVM to be used by the created method variant.
41     	 * @param fa is the flow analysis framework instance in which the created variant operates.
42     	 * @return a method variant
43     	 * @pre sootMethod != null and fa != null
44     	 */
45     	IMethodVariant<N> create(SootMethod sootMethod, ASTVariantManager<SYM, T, N, R> astVM, FA<SYM, T, N, R> fa);
46     }
