17     package edu.ksu.cis.indus.staticanalyses.flow;
18     
19     import edu.ksu.cis.indus.processing.Context;
20     import edu.ksu.cis.indus.staticanalyses.tokens.ITokens;
21     
22     import org.slf4j.Logger;
23     import org.slf4j.LoggerFactory;
24     
25     import soot.Value;
26     import soot.ValueBox;
27     import soot.jimple.AbstractJimpleValueSwitch;
28     
29     /**
30      * The expression visitor class. This class provides the default method implementations for all the expressions that need to
31      * be dealt at Jimple level. The class is tagged as <code>abstract</code> to force the users to extend the class as
32      * required.
33      * 
34      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
35      * @author $Author: rvprasad $
36      * @version $Revision: 1.24 $
37      * @param <E> is the type of the expression visitor.
38      * @param <SYM> is the type of symbol whose flow is being analyzed.
39      * @param <T> is the type of the token set object.
40      * @param <N> is the type of the summary node in the flow analysis.
41      */
42     public abstract class AbstractExprSwitch<E extends AbstractExprSwitch<E, SYM, T, N, R>, SYM, T extends ITokens<T, SYM>, N extends IFGNode<SYM, T, N>, R>
43     		extends AbstractJimpleValueSwitch
44     		implements IExprSwitch<N> {
45     
46     	/**
47     	 * The logger used by instances of this class to log messages.
48     	 */
49     	private static final Logger LOGGER = LoggerFactory.getLogger(AbstractExprSwitch.class);
50     
51     	/**
52     	 * The object used to connect flow graph nodes corresponding to AST and non-AST entities. This provides the flexibility to
53     	 * use the same implementation of the visitor with different connectors to process LHS and RHS entities.
54     	 */
55     	protected final IFGNodeConnector<N> connector;
56     
57     	/**
58     	 * This visitor works in the context given by <code>context</code>.
59     	 */
60     	protected final Context context;
61     
62     	/**
63     	 * The instance of the underlying flow analysis framework.
64     	 */
65     	protected final FA<SYM, T, N, R> fa;
66     
67     	/**
68     	 * This visitor is used to visit the expressions in the <code>method</code> variant.
69     	 */
70     	protected final IMethodVariant<N> method;
71     
72     	/**
73     	 * This visitor is used by <code>stmt</code> to walk the embedded expressions.
74     	 */
75     	protected final IStmtSwitch stmtSwitch;
76     
77     	/**
78     	 * This captures the flow node resulting from visiting an expression.
79     	 */
80     	private N resultFlowNode;
81     
82     	/**
83     	 * Creates a new <code>AbstractExprSwitch</code> instance. In non-prototype mode, all of the fields (declared in this
84     	 * class) will be non-null after returning from the constructor.
85     	 * 
86     	 * @param stmtVisitor the statement visitor which shall use this expression visitor.
87     	 * @param connectorToUse the connector to be used by this expression visitor to connect flow graph nodes corresponding to
88     	 *            AST and non-AST entities.
89     	 * @pre connectorToUse != null
90     	 */
91     	protected AbstractExprSwitch(final IStmtSwitch stmtVisitor, final IFGNodeConnector<N> connectorToUse) {
92     		this.stmtSwitch = stmtVisitor;
93     		this.connector = connectorToUse;
94     
95     		if (stmtSwitch != null) {
96     			context = ((AbstractStmtSwitch) stmtSwitch).context;
97     			method = ((AbstractStmtSwitch) stmtSwitch).method;
98     			fa = ((AbstractStmtSwitch) stmtSwitch).method.getFA();
99     		} else {
100     			context = null;
101     			method = null;
102     			fa = null;
103     		}
104     	}
105     
106     	/**
107     	 * Provides the default implementation when any expression is not handled by the visitor. It sets the flow node associated
108     	 * with the AST as the result.
109     	 * 
110     	 * @param o the expression which is not handled by the visitor.
111     	 */
112     	@Override public void defaultCase(final Object o) {
113     		setResult(method.getASTNode((Value) o, context));
114     
115     		if (LOGGER.isDebugEnabled()) {
116     			LOGGER.debug(o + "(" + o.getClass() + ") is not handled.");
117     		}
118     	}
119     
120     	/**
121     	 * This method will throw <code>UnsupportedOperationException</code>.
122     	 * 
123     	 * @return (This method raises an exception.)
124     	 * @throws UnsupportedOperationException as this method is not supported.
125     	 */
126     	public E getClone(@SuppressWarnings("unused") final Object... o) {
127     		throw new UnsupportedOperationException("Parameterless prototype method is not supported.");
128     	}
129     
130     	/**
131     	 * @see edu.ksu.cis.indus.staticanalyses.flow.IExprSwitch#getFlowNode()
132     	 */
133     	public N getFlowNode() {
134     		return resultFlowNode;
135     	}
136     
137     	/**
138     	 * Processes the expression at the given program point, <code>v</code>.
139     	 * 
140     	 * @param v the program point at which the to-be-processed expression occurs.
141     	 * @pre v != null
142     	 */
143     	public void process(final ValueBox v) {
144     		if (LOGGER.isDebugEnabled()) {
145     			LOGGER.debug("Started to process expression: " + v.getValue());
146     		}
147     
148     		final ValueBox _temp = context.setProgramPoint(v);
149     		v.getValue().apply(this);
150     		context.setProgramPoint(_temp);
151     
152     		if (LOGGER.isDebugEnabled()) {
153     			LOGGER.debug("Finished processing expression: " + v.getValue() + "\n" + getResult());
154     		}
155     	}
156     
157     	/**
158     	 * {@inheritDoc}
159     	 */
160     	public void setFlowNode(final N node) {
161     		resultFlowNode = node;
162     	}
163     }
