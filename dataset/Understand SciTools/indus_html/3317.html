16     package edu.ksu.cis.indus.common.collections;
17     
18     import edu.ksu.cis.indus.annotations.Functional;
19     import edu.ksu.cis.indus.annotations.NonNull;
20     
21     import java.lang.reflect.InvocationTargetException;
22     import java.lang.reflect.Method;
23     
24     import org.slf4j.Logger;
25     import org.slf4j.LoggerFactory;
26     
27     /**
28      * This transformer dynamically invokes a method on a input object and provides the return value as the result.
29      * 
30      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
31      * @author $Author: rvprasad $
32      * @version $Revision: 1.4 $ $Date: 2007/02/10 19:08:38 $
33      * @param <R> is the type of receiver object of the invoked method.
34      * @param <O> is the type of return value from the invoked method.
35      */
36     public class InvokeTransformer<R, O>
37     		implements ITransformer<R, O> {
38     
39     	/**
40     	 * The logger used by instances of this class to log messages.
41     	 */
42     	private static final Logger LOGGER = LoggerFactory.getLogger(InvokeTransformer.class);
43     
44     	/**
45     	 * 
46     	 */
47     	private final String name;
48     
49     	/**
50     	 * Creates an instance of this class. The method should return a value.
51     	 * 
52     	 * @param methodName is the name of the method to be invoked.
53     	 */
54     	public InvokeTransformer(@NonNull final String methodName) {
55     		super();
56     		name = methodName;
57     	}
58     
59     	/**
60     	 * {@inheritDoc}
61     	 */
62     	@Functional public O transform(@NonNull final R input) {
63     		final Class<?> _t = input.getClass();
64     		final Method _m;
65     		try {
66     			_m = _t.getMethod(name, (Class[]) null);
67     			@SuppressWarnings("unchecked") final O _invoke = (O) _m.invoke(input, (Object[]) null);
68     			return _invoke;
69     		} catch (final SecurityException _e) {
70     			LOGGER.error("Not enough permission to invoke method " + name + " on object " + input.toString(), _e);
71     			throw new RuntimeException(_e);
72     		} catch (final NoSuchMethodException _e) {
73     			LOGGER.error("Method with name " + name + " does not exist in type " + input.getClass(), _e);
74     			throw new RuntimeException(_e);
75     		} catch (final IllegalArgumentException _e) {
76     			LOGGER.error("Illegal argument to invoke method named " + name + " on object " + input.toString(), _e);
77     			throw new RuntimeException(_e);
78     		} catch (final IllegalAccessException _e) {
79     			LOGGER.error("Illegal access to method " + name + " on class " + input.getClass(), _e);
80     			throw new RuntimeException(_e);
81     		} catch (final InvocationTargetException _e) {
82     			LOGGER.error("Error invoking method " + name + " on object " + input.getClass(), _e);
83     			throw new RuntimeException(_e);
84     		}
85     	}
86     }
