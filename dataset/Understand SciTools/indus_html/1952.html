54     /**
55      * This class provides intra-thread aliased use-def information which is based on types and points-to information. If the use
56      * is reachable from the def via the control flow graph or via the CFG, then def and use site are related by use-def relation.
57      * The only exception for this case is when the def occurs in the class initializer. In this case, the defs can reach almost
58      * all methods even if they are executed in a different thread from the use site.
59      * 
60      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
61      * @author $Author$
62      * @version $Revision$
63      */

65     		extends AbstractValueAnalyzerBasedProcessor<Value>
66     		implements IUseDefInfo<Pair<DefinitionStmt, SootMethod>, Pair<DefinitionStmt, SootMethod>>, IIdentification {

116     	public AliasedUseDefInfo(final IValueAnalyzer<Value> iva, final BasicBlockGraphMgr bbgManager,
117     			final PairManager pairManager, final CFGAnalysis analysis) {
118     		cfgAnalysis = analysis;
119     		analyzer = iva;
120     		bbgMgr = bbgManager;
121     		pairMgr = pairManager;
122     		def2usesMap = new HashMap<Object, Map<Pair<DefinitionStmt, SootMethod>, Collection<Pair<DefinitionStmt, SootMethod>>>>(
123     				Constants.getNumOfFieldsInApplication());
124     		use2defsMap = new HashMap<Object, Map<Pair<DefinitionStmt, SootMethod>, Collection<Pair<DefinitionStmt, SootMethod>>>>(
125     				Constants.getNumOfFieldsInApplication());
126     	}

128     	/**
129     	 * @see edu.ksu.cis.indus.staticanalyses.interfaces.IValueAnalyzerBasedProcessor#callback(Stmt, Context)
130     	 */
131     	@Override public void callback(final Stmt stmt, final Context context) {
132     		final DefinitionStmt _as = (DefinitionStmt) stmt;
133     
134     		if (_as.containsArrayRef() || _as.containsFieldRef()) {
135     			final Object _key;
136     
137     			if (_as.containsArrayRef()) {
138     				_key = _as.getArrayRef().getBase().getType();
139     			} else {
140     				_key = _as.getFieldRef().getField();
141     			}
142     
143     			final Value _ref = _as.getRightOp();
144     			final Map<Object, Map<Pair<DefinitionStmt, SootMethod>, Collection<Pair<DefinitionStmt, SootMethod>>>> _map;
145     			if (_ref instanceof ArrayRef || _ref instanceof FieldRef) {
146     				_map = use2defsMap;
147     			} else {
148     				_map = def2usesMap;
149     			}
150     			final Map<Pair<DefinitionStmt, SootMethod>, Collection<Pair<DefinitionStmt, SootMethod>>> _key2info = MapUtils
151     					.getMapFromMap(_map, _key);
152     			_key2info.put(pairMgr.getPair(_as, context.getCurrentMethod()), null);
153     		}
154     	}

156     	/**
157     	 * Records naive interprocedural data dependence. All it does it records dependence between type conformant writes and
158     	 * reads.
159     	 * 
160     	 * @see edu.ksu.cis.indus.staticanalyses.interfaces.IValueAnalyzerBasedProcessor#consolidate()
161     	 */
162     	@Override public void consolidate() {
163     		if (LOGGER.isDebugEnabled()) {
164     			LOGGER.debug("BEGIN: consolidating");
165     		}
166     
167     		final Collection<Pair<DefinitionStmt, SootMethod>> _uses = new HashSet<Pair<DefinitionStmt, SootMethod>>();
168     
169     		for (final Iterator<Map.Entry<Object, Map<Pair<DefinitionStmt, SootMethod>, Collection<Pair<DefinitionStmt, SootMethod>>>>> _i = def2usesMap
170     				.entrySet().iterator(); _i.hasNext();) {
171     			final Map.Entry<Object, Map<Pair<DefinitionStmt, SootMethod>, Collection<Pair<DefinitionStmt, SootMethod>>>> _entry = _i
172     					.next();
173     			final Object _key = _entry.getKey();
174     			final Map<Pair<DefinitionStmt, SootMethod>, Collection<Pair<DefinitionStmt, SootMethod>>> _defsite2usesites = _entry
175     					.getValue();
176     			final Map<Pair<DefinitionStmt, SootMethod>, Collection<Pair<DefinitionStmt, SootMethod>>> _usesite2defsites = use2defsMap
177     					.get(_key);
178     
179     			if (_usesite2defsites != null) {
180     				final Iterator<Pair<DefinitionStmt, SootMethod>> _k = _defsite2usesites.keySet().iterator();
181     				final int _kEnd = _defsite2usesites.keySet().size();
182     
183     				for (int _kIndex = 0; _kIndex < _kEnd; _kIndex++) {
184     					final Pair<DefinitionStmt, SootMethod> _defSite = _k.next();
185     					final Iterator<Pair<DefinitionStmt, SootMethod>> _l = _usesite2defsites.keySet().iterator();
186     					final int _lEnd = _usesite2defsites.keySet().size();
187     
188     					for (int _lIndex = 0; _lIndex < _lEnd; _lIndex++) {
189     						final Pair<DefinitionStmt, SootMethod> _useSite = _l.next();
190     
191     						if (areDefUseRelated(_defSite, _useSite)) {
192     							/*
193     							 * Check if the use method and the def method are the same. If so, use CFG reachability. If not,
194     							 * use call graph reachability within the locality of a thread.
195     							 */
196     							if (doesDefReachUse(_defSite, _useSite)) {
197     								MapUtils.putIntoCollectionInMap(_usesite2defsites, _useSite, _defSite);
198     								_uses.add(_useSite);
199     							}
200     						}
201     					}
202     
203     					if (!_uses.isEmpty()) {
204     						MapUtils.putAllIntoCollectionInMap(_defsite2usesites, _defSite, _uses);
205     						_uses.clear();
206     					}
207     				}
208     			}
209     		}
210     
211     		if (LOGGER.isDebugEnabled()) {
212     			LOGGER.debug("END: consolidating - " + toString());
213     		}
214     	}

366     	 */
367     	protected boolean isReachableViaInterProceduralControlFlow(@SuppressWarnings("unused") final SootMethod defMethod,
368     			@SuppressWarnings("unused") final Stmt defStmt, @SuppressWarnings("unused") final SootMethod useMethod,
369     			@SuppressWarnings("unused") final Stmt useStmt) {
370     		return true;

373     	/**
374     	 * Checks if the given definition and use are related.
375     	 * 
376     	 * @param defSite is the definition site.
377     	 * @param useSite is the use site.
378     	 * @return <code>true</code> if the def and use site are related; <code>false</code>, otherwise.
379     	 * @pre defSite != null and useSite != null
380     	 * @pre defSite.oclIsKindOf(Pair(Stmt, SootMethod)) and useSite.oclIsKindOf(Pair(Stmt, SootMethod))
381     	 */
382     	private boolean areDefUseRelated(final Pair<DefinitionStmt, SootMethod> defSite,
383     			final Pair<DefinitionStmt, SootMethod> useSite) {
384     		boolean _result = false;
385     		final Context _context = new Context();
386     		final DefinitionStmt _defStmt = defSite.getFirst();
387     		final DefinitionStmt _useStmt = useSite.getFirst();
388     		final SootMethod _defMethod = defSite.getSecond();
389     		final SootMethod _useMethod = useSite.getSecond();
390     
391     		if (_defStmt.containsArrayRef()) {
392     			final ValueBox _vBox1 = _useStmt.getArrayRef().getBaseBox();
393     			_context.setRootMethod(_useMethod);
394     			_context.setStmt(_useStmt);
395     			_context.setProgramPoint(_vBox1);
396     
397     			final Collection<Value> _c1 = analyzer.getValues(_vBox1.getValue(), _context);
398     			final ValueBox _vBox2 = _defStmt.getArrayRef().getBaseBox();
399     			_context.setRootMethod(_defMethod);
400     			_context.setStmt(_defStmt);
401     			_context.setProgramPoint(_vBox2);
402     
403     			// if the primaries of the access expression alias atleast one object
404     			_result = CollectionUtils.containsAny(_c1, analyzer.getValues(_vBox2.getValue(), _context));
405     		} else if (_defStmt.containsFieldRef()) {
406     			final FieldRef _fr = _useStmt.getFieldRef();
407     
408     			// set the initial value to true assuming dependency in case of static field ref
409     			_result = true;
410     
411     			if (_fr instanceof InstanceFieldRef) {
412     				final ValueBox _vBox1 = ((InstanceFieldRef) _useStmt.getFieldRef()).getBaseBox();
413     				_context.setRootMethod(_useMethod);
414     				_context.setStmt(_useStmt);
415     				_context.setProgramPoint(_vBox1);
416     
417     				final Collection<Value> _c1 = analyzer.getValues(_vBox1.getValue(), _context);
418     				final ValueBox _vBox2 = ((InstanceFieldRef) _defStmt.getFieldRef()).getBaseBox();
419     				_context.setRootMethod(_defMethod);
420     				_context.setStmt(_defStmt);
421     				_context.setProgramPoint(_vBox2);
422     
423     				// if the primaries of the access expression alias atleast one object.
424     				_result = CollectionUtils.containsAny(_c1, analyzer.getValues(_vBox2.getValue(), _context));
425     			}
426     		}
427     
428     		if (LOGGER.isDebugEnabled()) {
429     			LOGGER.debug("Def: " + defSite + " / Use: " + useSite + " --> related:" + _result);
430     		}
431     		return _result;
432     	}

434     	/**
435     	 * Checks if the def reaches the use site. If either of the methods are class initializers, <code>true</code> is
436     	 * returned.
437     	 * 
438     	 * @param defSite is the definition site.
439     	 * @param useSite is the use site.
440     	 * @return <code>true</code> if the def and use site are related; <code>false</code>, otherwise.
441     	 * @pre defSite != null and useSite != null
442     	 */
443     	private boolean doesDefReachUse(final Pair<DefinitionStmt, SootMethod> defSite,
444     			final Pair<DefinitionStmt, SootMethod> useSite) {
445     		boolean _result;
446     		final DefinitionStmt _defStmt = defSite.getFirst();
447     		final DefinitionStmt _useStmt = useSite.getFirst();
448     		final SootMethod _defMethod = defSite.getSecond();
449     		final SootMethod _useMethod = useSite.getSecond();
450     
451     		if (_useMethod.equals(_defMethod)) {
452     			// if def and use occur in the same method
453     			_result = cfgAnalysis.doesControlFlowPathExistBetween(_defStmt, _useStmt, _useMethod);
454     		} else if (_defStmt.containsFieldRef()
455     				&& ((_defMethod.getName().equals("<clinit>") && _defStmt.getFieldRef().getField().isStatic()) || (_defMethod
456     						.getName().equals("<init>") && !_defStmt.getFieldRef().getField().isStatic()))
457     				&& _defStmt.getFieldRef().getField().getDeclaringClass().equals(_defMethod.getDeclaringClass())) {
458     			// if the def/use involves the same field and either the field is static and is being defined in clinit or the
459     			// field is instance-based and is being defined in init
460     			_result = true;
461     		} else {
462     			_result = isReachableViaInterProceduralControlFlow(_defMethod, _defStmt, _useMethod, _useStmt);
463     		}
464     
465     		if (LOGGER.isDebugEnabled()) {
466     			LOGGER.debug("Def: " + defSite + " / Use: " + useSite + " --> related:" + _result);
467     		}
468     		return _result;
469     	}
