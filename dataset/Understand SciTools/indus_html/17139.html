86     	/**
87     	 * {@inheritDoc}
88     	 * 
89     	 * @see IDirectionSensitivePartOfSlicingEngine#generateCriteriaForTheCallToMethod(soot.SootMethod, soot.SootMethod,
90     	 *      soot.jimple.Stmt)
91     	 */
92     	public void generateCriteriaForTheCallToMethod(final SootMethod callee, final SootMethod caller, final Stmt callStmt) {
93     		/*
94     		 * _stmt may be an assignment statement. Hence, we want the control to reach the statement but not leave it. However,
95     		 * the execution of the invoke expression should be considered as it is requied to reach the callee. Likewise, we want
96     		 * to include the expression but not all arguments. We rely on the reachable parameters to suck in the arguments. So,
97     		 * we generate criteria only for the invocation expression and not the arguments. Refer to
98     		 * transformAndGenerateToNewCriteriaForXXXX for information about how invoke expressions are handled differently.
99     		 */
100     		engine.generateStmtLevelSliceCriterion(callStmt, caller, true);
101     		engine.includeInSlice(callStmt.getInvokeExprBox());
102     
103     		if (!callee.isStatic()) {
104     			final ValueBox _vBox = ((InstanceInvokeExpr) callStmt.getInvokeExpr()).getBaseBox();
105     			engine.generateExprLevelSliceCriterion(_vBox, callStmt, caller, false);
106     		}
107     
108     		if (callStmt instanceof AssignStmt) {
109     			final AssignStmt _defStmt = (AssignStmt) callStmt;
110     			engine.generateExprLevelSliceCriterion(_defStmt.getLeftOpBox(), callStmt, caller, true);
111     		}
112     	}

114     	/**
115     	 * {@inheritDoc}
116     	 * 
117     	 * @see edu.ksu.cis.indus.slicer.IDirectionSensitivePartOfSlicingEngine#generateCriteriaToIncludeCallees(soot.jimple.Stmt,
118     	 *      soot.SootMethod, java.util.Collection)
119     	 */
120     	public void generateCriteriaToIncludeCallees(final Stmt stmt, final SootMethod caller,
121     			final Collection<SootMethod> callees) {
122     		final InvokeExpr _expr = stmt.getInvokeExpr();
123     
124     		if (_expr instanceof InstanceInvokeExpr) {
125     			final Iterator<SootMethod> _i = callees.iterator();
126     			final int _iEnd = callees.size();
127     
128     			engine.enterMethod(new CallTriple(caller, stmt, _expr));
129     
130     			for (int _iIndex = 0; _iIndex < _iEnd; _iIndex++) {
131     				final SootMethod _callee = _i.next();
132     				final Collection<Stmt> _units = engine.getBasicBlockGraphManager().getStmtList(_callee);
133     
134     				for (final Iterator<Stmt> _j = _units.iterator(); _j.hasNext();) {
135     					final Stmt _stmt = _j.next();
136     
137     					if (_stmt instanceof IdentityStmt) {
138     						final IdentityStmt _idStmt = (IdentityStmt) _stmt;
139     						final Value _rightOp = _idStmt.getRightOp();
140     
141     						if (_rightOp instanceof ThisRef) {
142     							engine.generateStmtLevelSliceCriterion(_idStmt, _callee, false);
143     							break;
144     						}
145     					}
146     				}
147     			}
148     			engine.returnFromMethod();
149     		}
150     	}

152     	/**
153     	 * {@inheritDoc}
154     	 * 
155     	 * @see DependenceExtractor.IDependenceRetriver#getDependences(IDependencyAnalysis, Object, SootMethod )
156     	 */
157     	public Collection<Object> getDependences(final IDependencyAnalysis analysis, final Object entity, final SootMethod method) {
158     		final Collection<Object> _result = new HashSet<Object>();
159     		final Object _dir = analysis.getDirection();
160     
161     		if (_dir.equals(Direction.FORWARD_DIRECTION)) {
162     			_result.addAll(analysis.getDependees(entity, method));
163     		} else if (_dir.equals(Direction.BI_DIRECTIONAL)) {
164     			_result.addAll(analysis.getDependents(entity, method));
165     		} else if (LOGGER.isWarnEnabled()) {
166     			LOGGER.warn("Trying to retrieve FORWARD dependence from a dependence analysis that is BACKWARD in direction. -- "
167     					+ analysis.getClass() + " - " + _dir);
168     		}
169     
170     		return _result;
171     	}

188     	/**
189     	 * {@inheritDoc}
190     	 * 
191     	 * @see IDirectionSensitivePartOfSlicingEngine#processLocalAt(Local, Stmt, SootMethod)
192     	 */
193     	public void processLocalAt(final Local local, final Stmt stmt, final SootMethod method) {
194     		engine.generateStmtLevelSliceCriterion(stmt, method, true);
195     
196     		if (stmt instanceof DefinitionStmt) {
197           final Collection<ValueBox> _boxes = new ArrayList<ValueBox>(); 
198           final ValueBox _leftOpBox = ((DefinitionStmt) stmt).getLeftOpBox();
199           _boxes.add(_leftOpBox);
200           _boxes.addAll(_leftOpBox.getValue().getUseBoxes());
201           for (final ValueBox _box : _boxes) {
202             engine.generateExprLevelSliceCriterion(_box, stmt, method, false);
203           }
204     		}
205     
206     		if (stmt.containsInvokeExpr()) {
207     			final Collection<ValueBox> _useBoxes = stmt.getInvokeExpr().getUseBoxes();
208     			final Iterator<ValueBox> _j = _useBoxes.iterator();
209     			final int _jEnd = _useBoxes.size();
210     
211     			for (int _jIndex = 0; _jIndex < _jEnd; _jIndex++) {
212     				final ValueBox _vb = _j.next();
213     
214     				if (_vb.getValue().equals(local)) {
215     					engine.generateExprLevelSliceCriterion(_vb, stmt, method, true);
216     				}
217     			}
218     
219     			final InvokeExpr _invokeExpr = stmt.getInvokeExpr();
220     			final List<Value> _args = _invokeExpr.getArgs();
221     			final int _argIndex = _args.indexOf(local);
222     
223     			if (_argIndex > -1) {
224     				final Context _ctxt = new Context();
225     				_ctxt.setRootMethod(method);
226     				_ctxt.setStmt(stmt);
227     
228     				final Collection<SootMethod> _callees = engine.getCgi().getCallees(_invokeExpr, _ctxt);
229     
230     				engine.enterMethod(new CallTriple(method, stmt, stmt.getInvokeExpr()));
231     				generateCriteriaToIncludeArgumentReadStmts(_argIndex, _callees);
232     				engine.returnFromMethod();
233     			}
234     		}
235     	}

267     	/**
268     	 * {@inheritDoc}
269     	 * 
270     	 * @see IDirectionSensitivePartOfSlicingEngine#retrieveValueBoxesToTransformExpr(ValueBox, Stmt)
271     	 */
272     	public Collection<ValueBox> retrieveValueBoxesToTransformExpr(final ValueBox valueBox, final Stmt stmt) {
273     		final Collection<ValueBox> _valueBoxes = new HashSet<ValueBox>();
274     		_valueBoxes.add(valueBox);
275     
276     		final Value _value = valueBox.getValue();
277     
278     		// if it is an invocation expression, we do not want to include the arguments/sub-expressions.
279     		// in case of instance invocation, we do want to include the receiver position expression.
280     		if (_value instanceof InvokeExpr) {
281     			_valueBoxes.addAll(_value.getUseBoxes());
282     
283     			if (_value instanceof InstanceInvokeExpr) {
284     				_valueBoxes.add(((InstanceInvokeExpr) _value).getBaseBox());
285     			}
286     		}
287     
288     		/*
289     		 * Note that l-position is the lhs of an assignment statement whereas an l-value is value that occurs the l-position
290     		 * and is defined. In a[i] = v; a is the l-value whereas i is a r-value in the l-position and v is r-value in the
291     		 * r-position.
292     		 */
293     
294     		// we include the unincluded l-values in case the given value box appears in the r-position.
295     		if (stmt instanceof DefinitionStmt && ((DefinitionStmt) stmt).getLeftOp().getUseBoxes().contains(valueBox)) {
296     			_valueBoxes.addAll(engine.getCollector().getUncollected(stmt.getDefBoxes()));
297     		}
298     
299     		return _valueBoxes;
300     	}

311     	/**
312     	 * Generates criteria to include statements that pop the arguments of the call stack in the callees.
313     	 * 
314     	 * @param argIndex is the index of the argument whose read should be included.
315     	 * @param callees are the methods called.
316     	 * @pre argIndex != null and callees != null
317     	 * @pre callees.oclIsKindOf(Collection(SootMethod))
318     	 * @pre callees->forall(o | o.getParameterCount() > argIndex)
319     	 */
320     	private void generateCriteriaToIncludeArgumentReadStmts(final int argIndex, final Collection<SootMethod> callees) {
321     		final Iterator<SootMethod> _i = callees.iterator();
322     		final int _iEnd = callees.size();
323     
324     		for (int _iIndex = 0; _iIndex < _iEnd; _iIndex++) {
325     			final SootMethod _callee = _i.next();
326     			final Collection<Stmt> _units = engine.getBasicBlockGraphManager().getStmtList(_callee);
327     
328     			for (final Iterator<Stmt> _j = _units.iterator(); _j.hasNext();) {
329     				final Stmt _stmt = _j.next();
330     
331     				if (_stmt instanceof IdentityStmt) {
332     					final IdentityStmt _idStmt = (IdentityStmt) _stmt;
333     					final Value _rightOp = _idStmt.getRightOp();
334     
335     					if (_rightOp instanceof ParameterRef && ((ParameterRef) _rightOp).getIndex() == argIndex) {
336     						engine.generateStmtLevelSliceCriterion(_idStmt, _callee, false);
337     						break;
338     					}
339     				}
340     			}
341     		}
342     	}
