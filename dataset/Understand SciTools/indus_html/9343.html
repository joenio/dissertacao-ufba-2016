17     package edu.ksu.cis.indus.common.collections;
18     
19     import edu.ksu.cis.indus.annotations.Functional;
20     import edu.ksu.cis.indus.annotations.Immutable;
21     import edu.ksu.cis.indus.annotations.NonNull;
22     import edu.ksu.cis.indus.annotations.NumericalConstraint;
23     import edu.ksu.cis.indus.annotations.NumericalConstraint.NumericalValue;
24     
25     import java.util.AbstractList;
26     import java.util.AbstractSet;
27     import java.util.ArrayList;
28     import java.util.Collection;
29     import java.util.Collections;
30     import java.util.HashSet;
31     import java.util.Iterator;
32     import java.util.List;
33     import java.util.Set;
34     
35     /**
36      * This is a ordered set that maintains insertion ordering.
37      * 
38      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
39      * @author $Author: rvprasad $
40      * @version $Revision: 1.5 $ $Date: 2007/02/10 19:08:38 $
41      * @param <E> is the type of object in the set.
42      */
43     public class ListOrderedSet<E>
44     		extends AbstractSet<E> {
45     
46     	/**
47     	 * This stores the elements for order related operation.
48     	 */
49     	@NonNull final List<E> list;
50     
51     	/**
52     	 * This stores the elements in a set for fast access.
53     	 */
54     	@NonNull private final Set<E> set;
55     
56     	/**
57     	 * Creates an instance of this class.
58     	 */
59     	public ListOrderedSet() {
60     		super();
61     		set = new HashSet<E>();
62     		list = new ArrayList<E>();
63     	}
64     
65     	/**
66     	 * Creates an instance of this class.
67     	 * 
68     	 * @param c is the collection whose contents will be added to this set.
69     	 */
70     	public ListOrderedSet(@NonNull @Immutable final Collection<E> c) {
71     		set = new HashSet<E>(c);
72     		list = new ArrayList<E>(c);
73     	}
74     
75     	/**
76     	 * Creates an instance of this class.
77     	 * 
78     	 * @param initialCapacity <i>see super class constructor.</i>
79     	 */
80     	public ListOrderedSet(@NumericalConstraint(value = NumericalValue.NON_NEGATIVE) final int initialCapacity) {
81     		set = new HashSet<E>(initialCapacity);
82     		list = new ArrayList<E>(initialCapacity);
83     	}
84     
85     	/**
86     	 * Creates an instance of this class.
87     	 * 
88     	 * @param initialCapacity <i>see super class constructor.</i>
89     	 * @param loadFactor <i>see super class constructor.</i>
90     	 */
91     	public ListOrderedSet(@NumericalConstraint(value = NumericalValue.NON_NEGATIVE) final int initialCapacity,
92     			@NumericalConstraint(value = NumericalValue.NON_NEGATIVE) final float loadFactor) {
93     		set = new HashSet<E>(initialCapacity, loadFactor);
94     		list = new ArrayList<E>(initialCapacity);
95     	}
96     
97     	/**
98     	 * {@inheritDoc}
99     	 */
100     	@Override public boolean add(@Immutable final E o) {
101     		final boolean _t = set.add(o);
102     		if (_t) {
103     			list.add(o);
104     		}
105     		return _t;
106     	}
107     
108     	/**
109     	 * {@inheritDoc}
110     	 */
111     	@Override public void clear() {
112     		set.clear();
113     		list.clear();
114     	}
115     
116     	/**
117     	 * {@inheritDoc}
118     	 */
119     	@Functional @Override public boolean contains(final Object o) {
120     		return set.contains(o);
121     	}
122     
123     	/**
124     	 * {@inheritDoc}
125     	 */
126     	@Functional @Override public boolean containsAll(@NonNull final Collection<?> c) {
127     		return set.containsAll(c);
128     	}
129     
130     	/**
131     	 * {@inheritDoc}
132     	 */
133     	@Functional @Override public boolean equals(@NonNull @Immutable final Object o) {
134     		if (o instanceof ListOrderedSet) {
135     			final ListOrderedSet<?> _t = (ListOrderedSet) o;
136     			return set.equals(_t.list) && list.equals(_t.list);
137     		}
138     		return false;
139     	}
140     
141     	/**
142     	 * Retrieves the i-th element according to the insertion order.
143     	 * 
144     	 * @param i is the index.
145     	 * @return the element, if it exists.
146     	 * @throws IndexOutOfBoundsException when <code>i</code> is less than 0 or when <code>i</code> is greater than or
147     	 *             equal to size().
148     	 */
149     	@Functional public E get(final int i) throws IndexOutOfBoundsException {
150     		return list.get(i);
151     	}
152     
153     	/**
154     	 * Retrieves a view of the set as an unmodifiable list.
155     	 * 
156     	 * @return the unmodifiable sequential view of the set.
157     	 */
158     	@Functional @NonNull public List<E> getUnmodifiableList() {
159     		return Collections.unmodifiableList(new AbstractList<E>() {
160     
161     			@Override public E get(final int index) {
162     				return list.get(index);
163     			}
164     
165     			@Override public int size() {
166     				return list.size();
167     			}
168     
169     		});
170     	}
171     
172     	/**
173     	 * {@inheritDoc}
174     	 */
175     	@Functional @Override public int hashCode() {
176     		return set.hashCode() + list.hashCode();
177     	}
178     
179     	/**
180     	 * Retrieves the index of the given object corresponding to its first insertion into the set.
181     	 * 
182     	 * @param object of interest.
183     	 * @return a non-negative index if <code>object</code> exists in the set; -1, otherwise.
184     	 */
185     	@Functional public int indexOf(final E object) {
186     		return list.indexOf(object);
187     	}
188     
189     	/**
190     	 * {@inheritDoc}
191     	 */
192     	@Functional @Override public boolean isEmpty() {
193     		return set.isEmpty();
194     	}
195     
196     	/**
197     	 * {@inheritDoc}
198     	 */
199     	@Functional @NonNull @Override public Iterator<E> iterator() {
200     		return list.iterator();
201     	}
202     
203     	/**
204     	 * Removes the i-th inserted element.
205     	 * 
206     	 * @param i is the index of the element that should be removed.
207     	 * @return the removed element, if it existed.
208     	 * @throws IndexOutOfBoundsException when <code>i</code> is less than 0 or when <code>i</code> is greater than or
209     	 *             equal to size().
210     	 */
211     	@Functional @NonNull public E remove(final int i) throws IndexOutOfBoundsException {
212     		final E _t = get(i);
213     		set.remove(_t);
214     		return _t;
215     	}
216     
217     	/**
218     	 * {@inheritDoc}
219     	 */
220     	@Override public boolean remove(@Immutable final Object o) {
221     		list.remove(o);
222     		return set.remove(o);
223     	}
224     
225     	/**
226     	 * {@inheritDoc}
227     	 */
228     	@Override public boolean removeAll(@NonNull @Immutable final Collection<?> c) {
229     		return super.removeAll(c);
230     	}
231     
232     	/**
233     	 * {@inheritDoc}
234     	 */
235     	@Override public boolean retainAll(@NonNull @Immutable final Collection<?> c) {
236     		list.retainAll(c);
237     		return set.retainAll(c);
238     	}
239     
240     	/**
241     	 * {@inheritDoc}
242     	 */
243     	@Functional @Override public int size() {
244     		return set.size();
245     	}
246     
247     	/**
248     	 * {@inheritDoc}
249     	 */
250     	@Functional @NonNull @Override public Object[] toArray() {
251     		return list.toArray();
252     	}
253     
254     	/**
255     	 * {@inheritDoc}
256     	 */
257     	@Functional @Override public <T> T[] toArray(@NonNull final T[] a) {
258     		return list.toArray(a);
259     	}
260     
261     }
