43     /**
44      * This implementation facilitates the extraction of calling-contexts based on multithread data sharing (more precise than
45      * escape information).
46      * 
47      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
48      * @author $Author: rvprasad $
49      * @version $Revision: 1.23 $ $Date: 2007/02/10 19:07:04 $
50      */
51     public class ThreadEscapeInfoBasedCallingContextRetrieverV2
52     		extends ThreadEscapeInfoBasedCallingContextRetriever {
53     
54     	/**
55     	 * The logger used by instances of this class to log messages.
56     	 */
57     	private static final Logger LOGGER = LoggerFactory.getLogger(ThreadEscapeInfoBasedCallingContextRetrieverV2.class);
58     
59     	/**
60     	 * This indicates if this instance retrieves context required to preserve interference dependence.
61     	 */
62     	protected final boolean interferenceBased;
63     
64     	/**
65     	 * This indicates if this instance retrieves context required to preserve ready dependence.
66     	 */
67     	protected final boolean readyBased;
68     
69     	/**
70     	 * Creates an instance of this instance.
71     	 * 
72     	 * @param callContextLenLimit <i>refer to the constructor of the super class</i>.
73     	 * @param preserveReady <code>true</code> indicates ready dependence should be preserved; <code>false</code>,
74     	 *            otherwise.
75     	 * @param preserveInterference <code>true</code> indicates interference dependence should be preserved;
76     	 *            <code>false</code>, otherwise.
77     	 */
78     	public ThreadEscapeInfoBasedCallingContextRetrieverV2(final int callContextLenLimit, final boolean preserveReady,
79     			final boolean preserveInterference) {
80     		super(callContextLenLimit);
81     		readyBased = preserveReady;
82     		interferenceBased = preserveInterference;
83     	}
84     
85     	/**
86     	 * {@inheritDoc}
87     	 * 
88     	 * @see ThreadEscapeInfoBasedCallingContextRetriever#considerProgramPoint(Context)
89     	 */
90     	@Override protected boolean considerProgramPoint(final Context context) {
91     		boolean _result = super.considerProgramPoint(context);
92     
93     		if (_result) {
94     			final Value _value = context.getProgramPoint().getValue();
95     			final Stmt _stmt = context.getStmt();
96     			final SootMethod _currentMethod = context.getCurrentMethod();
97     
98     			if (interferenceBased) {
99     				if (_stmt.containsFieldRef()) {
100     					final FieldRef _fr = _stmt.getFieldRef();
101     
102     					if (_fr instanceof InstanceFieldRef && ((InstanceFieldRef) _fr).getBase() == _value) {
103     						_result = escapesInfo.fieldAccessShared(_value, _currentMethod, _fr.getField().getSignature(),
104     								IEscapeInfo.READ_WRITE_SHARED_ACCESS);
105     					} else if (_fr instanceof StaticFieldRef && _fr == _value) {
106     						final SootField _field = _fr.getField();
107     						final SootClass _declaringClass = _field.getDeclaringClass();
108     						final String _signature = _field.getSignature();
109     						_result = escapesInfo.staticfieldAccessShared(_declaringClass, _currentMethod, _signature,
110     								IEscapeInfo.READ_WRITE_SHARED_ACCESS);
111     					}
112     				} else if (_stmt.containsArrayRef() && _stmt.getArrayRef().getBase() == _value) {
113     					_result = escapesInfo.fieldAccessShared(_value, _currentMethod, IEscapeInfo.READ_WRITE_SHARED_ACCESS);
114     				}
115     			} else if (readyBased) {
116     				if (_stmt instanceof MonitorStmt && ((MonitorStmt) _stmt).getOp() == _value) {
117     					_result = escapesInfo.lockUnlockShared(_value, _currentMethod);
118     				} else if (_stmt.containsInvokeExpr()) {
119     					final InvokeExpr _ex = _stmt.getInvokeExpr();
120     					final SootMethod _invokedMethod = _ex.getMethod();
121     
122     					if (_ex instanceof VirtualInvokeExpr
123     							&& (Util.isWaitMethod(_invokedMethod) || Util.isNotifyMethod(_invokedMethod))
124     							&& ((VirtualInvokeExpr) _ex).getBase() == _value) {
125     						_result = escapesInfo.waitNotifyShared(_value, _currentMethod);
126     					}
127     				}
128     			}
129     		}
130     
131     		if (LOGGER.isDebugEnabled()) {
132     			LOGGER.debug("considerProgramPoint() - result =" + _result);
133     		}
134     
135     		return _result;
136     	}
137     
138     	/**
139     	 * {@inheritDoc}
140     	 * 
141     	 * @see ThreadEscapeInfoBasedCallingContextRetriever#considerThis(Context)
142     	 */
143     	@Override protected boolean considerThis(final Context methodContext) {
144     		final SootMethod _method = methodContext.getCurrentMethod();
145     		final boolean _result1 = interferenceBased
146     				&& escapesInfo.thisFieldAccessShared(_method, IEscapeInfo.READ_WRITE_SHARED_ACCESS);
147     		final boolean _result2 = readyBased
148     				&& (escapesInfo.thisWaitNotifyShared(_method) || escapesInfo.thisLockUnlockShared(_method));
149     
150     		if (LOGGER.isDebugEnabled()) {
151     			LOGGER.debug("considerThis() -  : _result = " + (_result1 || _result2));
152     		}
153     
154     		return _result1 || _result2;
155     	}
156     
157     	/**
158     	 * {@inheritDoc}
159     	 * 
160     	 * @see ThreadEscapeInfoBasedCallingContextRetriever#getCallerSideToken(java.lang.Object, soot.SootMethod,
161     	 *      edu.ksu.cis.indus.interfaces.ICallGraphInfo.CallTriple, Stack)
162     	 */
163     	@Override protected Object getCallerSideToken(final Object token, final SootMethod callee, final CallTriple callsite,
164     			final Stack<CallTriple> calleeCallStack) {
165     
166     		Object _result = super.getCallerSideToken(token, callee, callsite, calleeCallStack);
167     
168     		if (!(_result instanceof Tokens)) {
169     			final AliasSet _callerSideToken = (AliasSet) _result;
170     			final AliasSet _calleeSideToken = (AliasSet) token;
171     
172     			if (_callerSideToken != null && _calleeSideToken != null
173     					&& shouldCallerSideTokenBeDiscarded(_callerSideToken, _calleeSideToken)) {
174     				if (Util.isStartMethod(callee)) {
175     					_result = Tokens.ACCEPT_NON_TERMINAL_CONTEXT_TOKEN;
176     				} else {
177     					_result = Tokens.DISCARD_CONTEXT_TOKEN;
178     				}
179     			}
180     		}
181     
182     		return _result;
183     	}
184     
185     	/**
186     	 * Checks if the given caller side token can be discarded.
187     	 * 
188     	 * @param callerSideToken of interest.
189     	 * @param calleeSideToken of interest.
190     	 * @return <code>true</code> if the given caller side token can be discarded; <code>false</code>, otherwise.
191     	 */
192     	protected boolean shouldCallerSideTokenBeDiscarded(final AliasSet callerSideToken, final AliasSet calleeSideToken) {
193     		final boolean _considerForInterference;
194     		if (interferenceBased) {
195     			final Collection<?> _callerRWEntities = callerSideToken.getReadWriteShareEntities();
196     			final Collection<?> _calleeRWEntities = calleeSideToken.getReadWriteShareEntities();
197     			_considerForInterference = _callerRWEntities != null && _calleeRWEntities != null
198     					&& CollectionUtils.containsAny(_callerRWEntities, _calleeRWEntities);
199     		} else {
200     			_considerForInterference = false;
201     		}
202     
203     		final boolean _considerForReady;
204     		final boolean _considerForLock;
205     		if (readyBased) {
206     			final Collection<?> _callerReadyEntities = callerSideToken.getReadyEntities();
207     			final Collection<?> _calleeReadyEntities = calleeSideToken.getReadyEntities();
208     			_considerForReady = _callerReadyEntities != null && _calleeReadyEntities != null
209     					&& CollectionUtils.containsAny(_callerReadyEntities, _calleeReadyEntities);
210     			final Collection<?> _callerLockEntities = callerSideToken.getLockEntities();
211     			final Collection<?> _calleeLockEntities = calleeSideToken.getLockEntities();
212     			_considerForLock = _callerLockEntities != null && _calleeLockEntities != null
213     					&& CollectionUtils.containsAny(_callerLockEntities, _calleeLockEntities);
214     		} else {
215     			_considerForReady = false;
216     			_considerForLock = false;
217     		}
218     		return !_considerForInterference && !_considerForReady && !_considerForLock;
219     	}

221     }
