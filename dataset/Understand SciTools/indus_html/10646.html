110     	 * Flushes the writes associated with each xmlizers.

112     	 * @param xmlizers to be flushed.
113     	 * @param ctrl to unhook the xmlizers from.
114     	 * @pre xmlizers != null and ctrl != null
115     	 */
116     	public void flushXMLizers(final Map<StmtAndMethodBasedDependencyXMLizer<?, ?>, Writer> xmlizers,
117     			final ProcessingController ctrl) {
118     		for (final Iterator<StmtAndMethodBasedDependencyXMLizer<?, ?>> _i = xmlizers.keySet().iterator(); _i.hasNext();) {
119     			final IProcessor _p = _i.next();
120     			_p.unhook(ctrl);
121     
122     			try {
123     				final Writer _f = xmlizers.get(_p);
124     				_f.flush();
125     				_f.close();
126     			} catch (IOException _e) {
127     				_e.printStackTrace();
128     				LOGGER.error("Failed to close the xml file based on " + _p.getClass(), _e);
129     			}
130     		}
131     	}

165     	/**
166     	 * Retrives the xmlizer for the given dependence analysis based on the properties.
167     	 * 
168     	 * @param <T1> is the type of the dependent entity.
169     	 * @param <E2> is the type of the dependee entity.
170     	 * @param writer to be used by the xmlizer.
171     	 * @param da is the dependence analysis for which the xmlizer is requested.
172     	 * @return the xmlizer.
173     	 * @pre writer != null and da != null
174     	 * @post result != null
175     	 */
176     	private <T1 extends Stmt, E2 extends Stmt> StmtAndMethodBasedDependencyXMLizer<T1, E2> getXMLizerFor(final Writer writer,
177     			final IDependencyAnalysis<T1, SootMethod, ?, E2, SootMethod, ?> da) {
178     		StmtAndMethodBasedDependencyXMLizer<T1, E2> _result = null;
179     		final List<IDependencyAnalysis.DependenceSort> _t = new ArrayList<IDependencyAnalysis.DependenceSort>(da.getIds());
180     		Collections.sort(_t);
181     
182     		final String _xmlizerId = _t.toString();
183     		final String _temp = PROPERTIES.getProperty(_xmlizerId);
184     
185     		if (_temp.equals(DependencyXMLizer.STMT_LEVEL_DEPENDENCY)) {
186     			try {
187     				_result = new StmtAndMethodBasedDependencyXMLizer<T1, E2>(new CustomXMLOutputter(writer), getIdGenerator(),
188     						da);
189     			} catch (final UnsupportedEncodingException _e) {
190     				LOGGER.error("UTF-8 encoding is unsupported.  Now, this contradicts the documentation!!", _e);
191     			}
192     		} else {
193     			LOGGER.error("Unknown dependency xmlizer type requested.  Bailing on this.");
194     		}
195     		return _result;
196     	}

198     	/**
199     	 * Initializes the xmlizers.
200     	 * 
201     	 * @param info is the name of the root method.
202     	 * @param ctrl is the controller to be used to initialize the xmlizers and to which to hook up the xmlizers to xmlize the
203     	 *            dependence information.
204     	 * @return a map of xmlizers and the associated writers.
205     	 * @throws IllegalStateException when output directory is unspecified.
206     	 * @pre rootname != null and ctrl != null
207     	 * @post result != null
208     	 */
209     	private Map<StmtAndMethodBasedDependencyXMLizer<?, ?>, Writer> initXMLizers(final Map info,
210     			final ProcessingController ctrl) {
211     		final Map<StmtAndMethodBasedDependencyXMLizer<?, ?>, Writer> _result = new HashMap<StmtAndMethodBasedDependencyXMLizer<?, ?>, Writer>();
212     
213     		if (getXmlOutputDir() == null) {
214     			LOGGER.error("Please specify an output directory while using the xmlizer.");
215     			throw new IllegalStateException("Please specify an output directory while using the xmlizer.");
216     		}
217     
218     		for (final Iterator<Comparable<?>> _i = AbstractDependencyAnalysis.IDENTIFIERS.iterator(); _i.hasNext();) {
219     			final Object _id = _i.next();
220     			final Collection<IDependencyAnalysis<? extends Stmt, SootMethod, ?, ? extends Stmt, SootMethod, ?>> _col = (Collection) info
221     					.get(_id);
222     
223     			if (_col != null) {
224     				for (final Iterator<IDependencyAnalysis<? extends Stmt, SootMethod, ?, ? extends Stmt, SootMethod, ?>> _j = _col
225     						.iterator(); _j.hasNext();) {
226     					final IDependencyAnalysis<? extends Stmt, SootMethod, ?, ? extends Stmt, SootMethod, ?> _da = _j.next();
227     					String _providedFileName = (String) info.get(FILE_NAME_ID);
228     
229     					if (_providedFileName == null) {
230     						_providedFileName = getDAPartOfFileName(_da);
231     					}
232     
233     					final String _filename = getFileName(_providedFileName);
234     					filenames.add(_filename);
235     
236     					final File _f = new File(getXmlOutputDir() + File.separator + _filename);
237     
238     					try {
239     						final FileWriter _writer = new FileWriter(_f);
240     						final StmtAndMethodBasedDependencyXMLizer<?, ?> _xmlizer = getXMLizerFor(_writer, _da);
241     
242     						if (_xmlizer == null) {
243     							LOGGER.error("No xmlizer specified for dependency calculated by " + _da.getClass()
244     									+ ".  No xml file written.");
245     							_writer.close();
246     						} else {
247     							_xmlizer.hookup(ctrl);
248     							_result.put(_xmlizer, _writer);
249     						}
250     					} catch (IOException _e) {
251     						LOGGER.error("Failed to write the xml file based on " + _da.getClass(), _e);
252     					}
253     				}
254     			}
255     		}
256     		return _result;
257     	}
