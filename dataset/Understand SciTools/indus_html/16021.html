72     /**
73      * This class provides ready dependency information. This implementation refers to the technical report <a
74      * href="http://www.cis.ksu.edu/santos/papers/technicalReports.html">A Formal Study of Slicing for Multi-threaded Program with
75      * JVM Concurrency Primitives"</a>. This implementation by default does not consider call-sites for dependency calculation.
76      * <p>
77      * <i>Ready Dependence</i>: In a thread, all statements reachable from an enter-monitor statement or a <code>wait()</code>
78      * call-site via intra-procedural control-flow path with no intervening enter-monitor statement or <code>wait()</code>
79      * call-site are ready dependent on the enter-monitor statement. Across different threads, enter-monitor statements and
80      * <code>wait()</code> call-sites in a thread are ready dependent on corresponding exit-monitor statements and
81      * <code>notify()/notifyAll()</code> call-sites, respectively, occurring in a different thread.
82      * </p>
83      * <p>
84      * By default, all rules are considered for the analysis. This can be changed via <code>setRules()</code>. This class will
85      * also use OFA information if it is configured to do so.
86      * </p>
87      * <p>
88      * Ready dependence information pertaining to entry and exit points of synchronized methods cause a divergence in the way
89      * information is provided via <code>getDependees</code> and <code>getDependents</code>.
90      * </p>
91      * <p>
92      * In case the body of the synchronized method is unavailable, then any dependence involving entry and exit points of the
93      * method will use null as the dependee/dependent statement.
94      * </p>
95      * 
96      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
97      * @author $Author: rvprasad $
98      * @version $Revision: 1.91 $
99      */

115     		/**
116     		 * Collects monitor statements and statements with <code>Object.wait()</code> and <code>Object.notifyXX()</code>
117     		 * call-sites.
118     		 * 
119     		 * @param stmt to be preprocessed.
120     		 * @param context in which <code>stmt</code> occurs.
121     		 * @pre stmt.oclIsTypeOf(ExitMonitorStmt) or stmt.oclIsTypeOf(EnterMonitorStmt) or stmt.oclIsTypeOf(InvokeStmt)
122     		 * @pre stmt != null and context.getCurrentMethod() != null
123     		 */
124     		@Override public void callback(final Stmt stmt, final Context context) {
125     			final SootMethod _method = context.getCurrentMethod();
126     			final InvokeExpr _expr = stmt.getInvokeExpr();
127     
128     			if (_expr instanceof VirtualInvokeExpr) {
129     				final VirtualInvokeExpr _invokeExpr = (VirtualInvokeExpr) _expr;
130     				final SootMethod _callee = _invokeExpr.getMethod();
131     
132     				Map<SootMethod, Collection<InvokeStmt>> _method2stmts = null;
133     
134     				if (Util.isWaitMethod(_callee)) {
135     					_method2stmts = waits;
136     				} else if (Util.isNotifyMethod(_callee)) {
137     					_method2stmts = notifies;
138     				}
139     
140     				if (_method2stmts != null) {
141     					MapUtils.putIntoCollectionInMap(_method2stmts, _method, (InvokeStmt) stmt);
142     				}
143     			}
144     		}

323     	 * Calculates ready dependency for the methods provided at initialization. It considers only the rules specified by via
324     	 * <code>setRules</code> method. By default, all rules are considered for the analysis.
325     	 * 
326     	 * @see edu.ksu.cis.indus.staticanalyses.dependency.AbstractDependencyAnalysis#analyze()
327     	 */
328     	@Override public void analyze() {
329     		unstable();
330     
331     		if (monitorInfo.isStable() && callgraph.isStable() && threadgraph.isStable()
332     				&& (!useSafeLockAnalysis || safelockAnalysis.isStable())) {
333     			if (LOGGER.isInfoEnabled()) {
334     				LOGGER.info("BEGIN: Ready Dependence [" + this.getClass() + "] processing");
335     			}
336     
337     			if (!threadgraph.getCreationSites().isEmpty()) {
338     				processRules();
339     			}
340     
341     			if (LOGGER.isDebugEnabled()) {
342     				LOGGER.debug("analyze() - " + toString());
343     			}
344     
345     			if (LOGGER.isInfoEnabled()) {
346     				LOGGER.info("END: Ready Dependence processing");
347     			}
348     
349     			stable();
350     		} else {
351     			if (LOGGER.isInfoEnabled()) {
352     				LOGGER.info("Delaying execution as prerequisites are unsatisfied.");
353     			}
354     		}

518     	/**
519     	 * Checks if the given wait invocation is dependent on the notify invocation according to rule 4 based of OFA information.
520     	 * 
521     	 * @param waitPair is the wait invocation statement and containg method pair.
522     	 * @param notifyPair is the notify invocation statement and containg method pair.
523     	 * @return <code>true</code> if there is a dependence; <code>false</code>, otherwise.
524     	 * @pre waitPair.getFirst() != null and waitPair.getSecond() != null
525     	 * @pre notifyPair.getFirst() != null and notifyPair.getSecond() != null
526     	 */
527     	protected final boolean ifDependentOnBasedOnOFAByRule4(final Pair<InvokeStmt, SootMethod> waitPair,
528     			final Pair<InvokeStmt, SootMethod> notifyPair) {
529     		boolean _result;
530     
531     		final SootMethod _waitMethod = waitPair.getSecond();
532     		final SootMethod _notifyMethod = notifyPair.getSecond();
533     		final InvokeStmt _wStmt = waitPair.getFirst();
534     		final InvokeStmt _nStmt = notifyPair.getFirst();
535     		final Context _context = new AllocationContext();
536     
537     		final InstanceInvokeExpr _wExpr = (InstanceInvokeExpr) _wStmt.getInvokeExpr();
538     		_context.setProgramPoint(_wExpr.getBaseBox());
539     		_context.setStmt(_wStmt);
540     		_context.setRootMethod(_waitMethod);
541     
542     		final Collection<?> _col1 = ofa.getValues(_wExpr.getBase(), _context);
543     
544     		final InstanceInvokeExpr _nExpr = (InstanceInvokeExpr) _nStmt.getInvokeExpr();
545     		_context.setProgramPoint(_nExpr.getBaseBox());
546     		_context.setStmt(_nStmt);
547     		_context.setRootMethod(_notifyMethod);
548     
549     		final Collection<?> _col2 = ofa.getValues(_nExpr.getBase(), _context);
550     
551     		final NullConstant _n = NullConstant.v();

1202     			final SootClass _nClass = enterMethod.getDeclaringClass();

1204     				if (exit == null) { // synchronized method
1205     					final SootClass _xClass = exitMethod.getDeclaringClass();
