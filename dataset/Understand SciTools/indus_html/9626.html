28     /**
29      * This class contains static utility methods that are useful in the context of <code>java.util.List</code> instances.
30      * 
31      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
32      * @author $Author: rvprasad $
33      * @version $Revision: 1.6 $ $Date: 2007/02/10 19:08:38 $
34      */
35     public final class ListUtils {
36     
37     	// / CLOVER:OFF
38     	/**
39     	 * Creates an instance of this class.
40     	 */
41     	private ListUtils() {
42     		super();
43     	}

47     	/**
48     	 * Ensures that <code>list.size()</code> returns <code>finalSize</code>. This is achieved by appending
49     	 * <code>defaultValue</code> appropriate number of times to <code>list</code>.
50     	 * 
51     	 * @param <T> The type of the elements in the list.
52     	 * @param list to be modified.
53     	 * @param finalSize of <code>list</code>
54     	 * @param defaultValue to be injected into the list, if needed.
55     	 * @post list.size() = finalSize
56     	 * @post list$pre.size() &lt; finalSize implies list.contains(defaultValue)
57     	 * @post list$pre.size() &gt;= finalSize implies list$pre.equals(list)
58     	 */
59     	@Functional public static <T> void ensureSize(@NonNull final List<T> list,
60     			@NumericalConstraint(value = NumericalValue.NON_NEGATIVE) final int finalSize, final T defaultValue) {
61     		final int _size = list.size();
62     
63     		if (finalSize > _size) {
64     			for (int _i = finalSize - _size; _i > 0; _i--) {
65     				list.add(defaultValue);
66     			}
67     		}
68     	}
69     
70     	/**
71     	 * Retrieves the element at the given index from the given list; if none exists, then the list is extended to accomodate
72     	 * the index, the default value is inserted at the given index, and the default value is returned.
73     	 * 
74     	 * @param <T> is the type of object in the list.
75     	 * @param list of interest.
76     	 * @param index of interest.
77     	 * @param defaultValue to be used, if needed.
78     	 * @return the value in <code>list</code> at index <code>i</code>.
79     	 * @post list$pre.size() &lt; index implies (list.size() = index and list.get(index) = defaultValue)
80     	 * @post list.containsAll(list$pre)
81     	 * @post result = list.get(index)
82     	 */
83     	@Immutable public static <T> T getAtIndexFromList(@NonNull final List<T> list,
84     			@NumericalConstraint(value = NumericalValue.NON_NEGATIVE) final int index, @Immutable final T defaultValue) {
85     		if (index < 0) {
86     			throw new IndexOutOfBoundsException("invalid index: " + index + " < 0");
87     		}
88     
89     		if (index >= list.size()) {
90     			ensureSize(list, index + 1, null);
91     			list.set(index, defaultValue);
92     		} else if (list.get(index) == null) {
93     			list.set(index, defaultValue);
94     		}
95     		return list.get(index);
96     	}
97     
98     	/**
99     	 * Retrieves an element at the given index in the list. If the value at the given index is <code>null</code> then the
100     	 * <code>null</code> value is replaced by the defaultValue and the same is returned. If the index does not occur in the
101     	 * list (list is short) then the list is extended with null values, defaultValue is added at the appropriate index, and
102     	 * the same is returned.
103     	 * 
104     	 * @param <T> is the type of the elements of the list.
105     	 * @param <T2> is the type of object created by the factory.
106     	 * @param list from which to retrieve the value.
107     	 * @param index in <code>list</code> from which to retrive the value.
108     	 * @param factory provides the default value to be injected and returned if none exists or if <code>null</code> exists.
109     	 * @return the value at <code>index</code> in <code>list</code>.
110     	 * @throws IndexOutOfBoundsException when <code>index</code> is less than 0.
111     	 * @post list$pre.get(index) = null or list$pre.size() &lt; index implies result = factory.create()
112     	 * @post list.get(index) = result
113     	 */
114     	@NonNull @Functional public static <T, T2 extends T> T getAtIndexFromListUsingFactory(@NonNull final List<T> list,
115     			@NumericalConstraint(value = NumericalValue.NON_NEGATIVE) final int index, @NonNull final IFactory<T2> factory)
116     			throws IndexOutOfBoundsException {
117     		if (index < 0) {
118     			throw new IndexOutOfBoundsException("invalid index: " + index + " < 0");
119     		}
120     
121     		if (index >= list.size()) {
122     			ensureSize(list, index + 1, null);
123     			list.set(index, factory.create());
124     		} else if (list.get(index) == null) {
125     			list.set(index, factory.create());
126     		}
127     		return list.get(index);
128     	}
129     
130     	/**
131     	 * Retrieves a list factory.
132     	 * 
133     	 * @param <T> is the type of object stored in the list created by the returned factory.
134     	 * @return a factory.
135     	 */
136     	@NonNull @Functional public static <T> IFactory<List<T>> getFactory() {
137     		return new IFactory<List<T>>() {
138     
139     			public List<T> create() {
140     				return new ArrayList<T>();
141     			}
142     		};
143     	}
144     
145     }
