45     /**
46      * This is the setup in which various tests of flow analyses are run.  The classes to be processed during the test can be
47      * configured via the command line or via specifying <code>FARegressionTestSuite.FA_TEST_PROPERTIES_FILE</code> system
48      * property. The syntax for both these options is a space separated list of class names.
49      *
50      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
51      * @author $Author: rvprasad $
52      * @version $Revision: 1.33 $ $Date: 2007/02/10 19:07:27 $
53      */
54     public class FATestSetup
55       extends AbstractXMLBasedTestSetup {
56     	/**
57     	 * The tag used by the flow analysis instance.
58     	 */
59     	public static final String TAG_NAME = "indus.staticanalyses.flow.FATestSetup:FA";
60     
61     	/**
62     	 * The value analyzer used during testing.
63     	 */
64     	protected IValueAnalyzer valueAnalyzer;
65     
66     	/**
67     	 * The names of the class to analyze.
68     	 */
69     	protected final String classNames;
70     
71     	/**
72     	 * The class path to use with soot in this setup.
73     	 */
74     	private final String sootClassPath;
75     
76     	/**
77     	 * Creates a new FATestSetup object.
78     	 *
79     	 * @param test is the test to run in this setup.
80     	 * @param theNameOfClasses is the list of classes.
81     	 * @param classpath to be used to find the classes.
82     	 *
83     	 * @pre test != null and theNameOfClasses != null
84     	 */
85     	protected FATestSetup(final TestSuite test, final String theNameOfClasses, final String classpath) {
86     		super(test);
87     		sootClassPath = classpath;
88     		classNames = theNameOfClasses;
89     	}
90     
91     	/**
92     	 * @see junit.extensions.TestSetup#setUp()
93     	 */
94     	protected void setUp()
95     	  throws Exception {
96     		super.setUp();
97     
98     		final SootBasedDriver _driver = new SootBasedDriver();
99     		_driver.addToSootClassPath(sootClassPath);
100     		_driver.setClassNames(Arrays.asList(classNames.split(" ")));
101     		_driver.initialize();
102     		valueAnalyzer = OFAnalyzer.getFSOIAnalyzer(FATestSetup.TAG_NAME, new BitSetTokenManager<Value, Type>(new SootValueTypeManager()), _driver.getStmtGraphFactory());
103     		valueAnalyzer.analyze(_driver.getEnvironment(), _driver.getRootMethods());
104     
105     		final Collection _temp1 = TestHelper.getTestCasesReachableFromSuite((TestSuite) getTest(), IFATest.class);
106     
107     		for (final Iterator _i = _temp1.iterator(); _i.hasNext();) {
108     			final IFATest _test = (IFATest) _i.next();
109     			_test.setAnalyzer(valueAnalyzer);
110     			_test.setFATagName(TAG_NAME);
111     		}
112     
113             final Collection _temp2 = TestHelper.getTestCasesReachableFromSuite((TestSuite) getTest(), FATest.class);
114     
115             for (final Iterator _i = _temp2.iterator(); _i.hasNext();) {
116                 final FATest _test = (FATest) _i.next();
117                 _test.setFA(((OFAnalyzer) valueAnalyzer).fa);
118             }
119     
120     		if (dumpLocation != null) {
121     			JimpleXMLizerCLI.writeJimpleAsXML(_driver.getEnvironment(), dumpLocation, null, idGenerator,
122     				new TagBasedProcessingFilter(TAG_NAME));
123     		}
124     	}
125     
126     	/**
127     	 * @see junit.extensions.TestSetup#tearDown()
128     	 */
129     	protected void tearDown()
130     	  throws Exception {
131     		G.reset();
132     		valueAnalyzer.reset();
133     		valueAnalyzer = null;
134     		idGenerator.reset();
135     		super.tearDown();
136     	}
137     }
