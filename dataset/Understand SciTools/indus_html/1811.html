28      * <p>

46     	/**

51     			+ ".LoadMethodBodiesDuringInit";
52     
53     	/**
54     	 * This is the name of the property that control the number of classes in the system. It's name is

56     	 */

59     	/**

62     	 */

64     
65     	/**
66     	 * This is the name of the property that control the number of methods in the system. It's name is
67     	 * "edu.ksu.cis.indus.NumOfMethodsInApplication".

69     	public static final String NUM_OF_METHODS_IN_APPLICATION_PROPERTY = "edu.ksu.cis.indus.NumOfMethodsInApplication";

74     	 * <code>edu.ksu.cis.indus.soot.RootMethodTrapper</code>.
75     	 */
76     	public static final String ROOT_METHOD_TRAPPER_CLASS_PROPERTY = "edu.ksu.cis.indus.common.soot.RootMethodTrapper.class";
77     
78     	/**
79     	 * The name of the property via which the name of the statement graph factory class can be specified. It's name is
80     	 * "edu.ksu.cis.indus.common.soot.SootBasedDriver.StmtGraphFactory.class".
81     	 */
82     	public static final String STMT_GRAPH_FACTORY_CLASS_PROPERTY = "edu.ksu.cis.indus.common.soot.SootBasedDriver"
83     			+ ".StmtGraphFactory.class";

85     	/**
86     	 * This contains the constants.
87     	 */
88     	private static final Properties CONFIGURATIONS = new Properties();
89     
90     	/**
91     	 * This is the default for the number of classes in the system.
92     	 */
93     	private static final int DEFAULT_NUM_OF_CLASSES_IN_APPLICATION = 1000;
94     
95     	/**
96     	 * This is the default for the number of fields in the system.
97     	 */
98     	private static final int DEFAULT_NUM_OF_FIELDS_IN_APPLICATION = 3000;
99     
100     	/**
101     	 * This is the default for the number of classes in the system.
102     	 */
103     	private static final int DEFAULT_NUM_OF_METHODS_IN_APPLICATION = 5000;
104     
105     	static {
106     		final String _propFileName = System.getProperty(CONSTANTS_CONFIGURATION_FILE_PROPERTY);
107     
108     		if (_propFileName != null) {
109     			try {
110     				final InputStream _stream = ClassLoader.getSystemResourceAsStream(_propFileName);
111     				CONFIGURATIONS.load(_stream);
112     			} catch (final IOException _e) {
113     				System.err.println("Well, error loading property file.  Bailing.");
114     				throw new RuntimeException(_e);
115     			}
116     		}
117     	}

130     	/**
131     	 * Returns the approximate number of classes that the application may contain. This defaults to 1000.
132     	 * 
133     	 * @return the approximate number of classes that the application may contain.
134     	 */
135     	public static int getNumOfClassesInApplication() {
136     		final int _defaultValue = DEFAULT_NUM_OF_CLASSES_IN_APPLICATION;
137     		final String _key = NUM_OF_CLASSES_IN_APPLICATION_PROPERTY;
138     		final int _result;
139     		_result = retrieveIntValue(_defaultValue, _key, CONFIGURATIONS);
140     		return _result;
141     	}

147     	 */
148     	public static int getNumOfFieldsInApplication() {
149     		final int _defaultValue = DEFAULT_NUM_OF_FIELDS_IN_APPLICATION;
150     		final String _key = NUM_OF_FIELDS_IN_APPLICATION_PROPERTY;
151     		final int _result;
152     		_result = retrieveIntValue(_defaultValue, _key, CONFIGURATIONS);

154     	}

159     	 * @return the approximate number of methods that the application may contain.
160     	 */
161     	public static int getNumOfMethodsInApplication() {
162     		final int _defaultValue = DEFAULT_NUM_OF_METHODS_IN_APPLICATION;
163     		final String _key = NUM_OF_METHODS_IN_APPLICATION_PROPERTY;
164     		final int _result;
165     		_result = retrieveIntValue(_defaultValue, _key, CONFIGURATIONS);
166     		return _result;
167     	}

169     	/**
170     	 * Retrieves the name of the root method trapper class. If not specified, it defaults to
171     	 * <code>edu.ksu.cis.indus.common.soot.RootMethodTrapper$MainMethodTrapper</code>.
172     	 * 
173     	 * @return the class name if specified.
174     	 */
175     	@NonNull public static String getRootMethodTrapperClassName() {
176     		String _result = CONFIGURATIONS.getProperty(ROOT_METHOD_TRAPPER_CLASS_PROPERTY);
177     
178     		if (_result == null) {
179     			_result = "edu.ksu.cis.indus.common.soot.RootMethodTrapper$MainMethodTrapper";
180     		}
181     		return _result;
182     	}
183     
184     	/**
185     	 * Retrieves the name of the statement graph factory class. If not specified, it defaults to
186     	 * <code>edu.ksu.cis.indus.common.soot.ExceptionFlowSensitiveStmtGraphFactory</code>.
187     	 * 
188     	 * @return the class name if specified.
189     	 */
190     	@NonNull public static String getStmtGraphFactoryClassName() {
191     		String _result = CONFIGURATIONS.getProperty(STMT_GRAPH_FACTORY_CLASS_PROPERTY);
192     
193     		if (_result == null) {
194     			_result = "edu.ksu.cis.indus.common.soot.CompleteStmtGraphFactory";
195     		}
196     		return _result;
197     	}
198     
199     	/**
200     	 * Retrieves an integer constant. <i>This method is not for public use.</i>
201     	 * 
202     	 * @param defaultValue obviously.
203     	 * @param key of interest.
204     	 * @param props from which to retrieve the values.
205     	 * @return the constant.
206     	 * @throws NumberFormatException when configuration file is syntactically incorrect.
207     	 */
208     	public static int retrieveIntValue(final int defaultValue, final String key, final Properties props)
209     			throws NumberFormatException {
210     		final int _result;
211     		final String _temp = props.getProperty(key);
212     
213     		if (_temp != null) {
214     			_result = Integer.parseInt(_temp);
215     		} else {
216     			_result = defaultValue;
217     		}
218     		return _result;
219     	}
220     
221     	/**
222     	 * Checks if method bodies should be loaded during initialization. If not specified, it defaults to <code>false</code>.
223     	 * 
224     	 * @return <code>true</code> if the bodies should be loaded; <code>false</code>, otherwise.
225     	 */
226     	public static boolean shouldLoadMethodBodiesDuringInitialization() {
227     		final boolean _defaultValue = false;
228     		final boolean _result;
229     		final String _temp = CONFIGURATIONS.getProperty(LOAD_METHOD_BODIES_DURING_INITIALIZATION);
230     
231     		if (_temp != null) {
232     			_result = Boolean.parseBoolean(_temp);
233     		} else {
234     			_result = _defaultValue;
235     		}
236     		return _result;
237     	}
238     }
