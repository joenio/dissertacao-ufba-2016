17     package edu.ksu.cis.indus.staticanalyses.flow.modes.sensitive.flow;
18     
19     import edu.ksu.cis.indus.interfaces.IPrototype;
20     import edu.ksu.cis.indus.processing.Context;
21     
22     import edu.ksu.cis.indus.staticanalyses.flow.AbstractIndexManager;
23     import edu.ksu.cis.indus.staticanalyses.flow.modes.sensitive.OneContextInfoIndex;
24     
25     import org.slf4j.Logger;
26     import org.slf4j.LoggerFactory;
27     
28     import soot.ValueBox;
29     
30     /**
31      * This class manages indices associated with entities in flow sensitive mode. In reality, it provides the implementation to
32      * create new indices.
33      *
34      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
35      * @version $Revision: 1.13 $
36      * @param <E> is the type of the entity that has been indexed.
37     
38      */
39     public class FlowSensitiveIndexManager<E>
40     		extends AbstractIndexManager<OneContextInfoIndex<E, ValueBox>, E>
41     		implements IPrototype<FlowSensitiveIndexManager<E>> {
42     
43     	/**
44     	 * The logger used by instances of this class to log messages.
45     	 */
46     	private static final Logger LOGGER = LoggerFactory.getLogger(FlowSensitiveIndexManager.class);
47     
48     	/**
49     	 * Returns an index corresponding to the given entity and context. The index is dependent on the program point stored in
50     	 * the context.
51     	 *
52     	 * @param o the entity for which the index in required.
53     	 * @param c the context which captures program point needed to generate the index.
54     	 * @return the index that uniquely identifies <code>o</code> at the program point captured in <code>c</code>.
55     	 * @pre o != null and c != null
56     	 */
57     	@Override protected OneContextInfoIndex<E, ValueBox> createIndex(final E o, final Context c) {
58     		if (LOGGER.isDebugEnabled()) {
59     			LOGGER.debug("Getting index for " + o + " in " + c);
60     		}
61     
62     		return new OneContextInfoIndex<E, ValueBox>(o, c.getProgramPoint());
63     	}
64     
65     	/**
66     	 * @see edu.ksu.cis.indus.interfaces.IPrototype#getClone(java.lang.Object[])
67     	 */
68     	public FlowSensitiveIndexManager<E> getClone(Object... o) {
69     		return new FlowSensitiveIndexManager<E>();
70     	}
71     }
