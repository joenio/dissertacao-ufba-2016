17     package edu.ksu.cis.indus.slicer;
18     
19     import java.util.Collection;
20     
21     import soot.Local;
22     import soot.SootMethod;
23     import soot.ValueBox;
24     import soot.jimple.IdentityStmt;
25     import soot.jimple.Stmt;
26     
27     /**
28      * This interface provides methods to control the direction of slicing.
29      * 
30      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
31      * @author $Author: rvprasad $
32      * @version $Revision: 1.13 $
33      */
34     interface IDirectionSensitivePartOfSlicingEngine
35     		extends DependenceExtractor.IDependenceRetriver {
36     
37     	/**
38     	 * Controls if processing of slice criteria can end. This is involed by the engine when there are no criteria/work left to
39     	 * process. This method would be a good place to inject new criteria/work based on some caching mechanism. So, if the
40     	 * implementation adds new criteria/work then it should return <code>true</code> as the return value.
41     	 * 
42     	 * @return <code>true</code> if the processing should continue; <code>false</code>, otherwise.
43     	 */
44     	boolean continueProcessing();
45     
46     	/**
47     	 * Generates new criteria to capture the call to the given method <code>callee</code>.
48     	 * <p>
49     	 * This should be called from within the caller's context (callStack containing the call to the caller as TOS).
50     	 * </p>
51     	 * 
52     	 * @param callee obviously.
53     	 * @param caller obviously.
54     	 * @param callStmt in the <code>caller</code> at which <code>callee</code> was called.
55     	 * @pre callee != null and caller != null and callStmt != null
56     	 * @pre callStmt.containsInvokeExpr()
57     	 */
58     	void generateCriteriaForTheCallToMethod(SootMethod callee, SootMethod caller, Stmt callStmt);
59     
60     	/**
61     	 * Generate new criteria to include the given called methods at given statement in the caller.
62     	 * <p>
63     	 * This should be called from within the caller's context (callStack does not contain the call to the callee as TOS).
64     	 * </p>
65     	 * 
66     	 * @param callStmt at which the invocation occurs.
67     	 * @param caller in which the invocation occurs.
68     	 * @param callees that are invoked.
69     	 * @pre callStmt != null and caller != null and callees != null
70     	 * @pre callStmt.containsInvokeExpr()
71     	 */
72     	void generateCriteriaToIncludeCallees(Stmt callStmt, SootMethod caller, Collection<SootMethod> callees);
73     
74     	/**
75     	 * Process the local in the given statement for inclusion in the slice.
76     	 * 
77     	 * @param local obviosly.
78     	 * @param depStmt in which <code>local</code> occurs.
79     	 * @param method in which <code>stmt</code> occurs.
80     	 * @pre local != null and depStmt != null and method != null
81     	 */
82     	void processLocalAt(Local local, Stmt depStmt, SootMethod method);
83     
84     	/**
85     	 * Process the new expression that occurs in the given statement and method for inclusion in the slice.
86     	 * <p>
87     	 * This should be called from within the callee's context (callStack containing the call to the callee as TOS).
88     	 * </p>
89     	 * 
90     	 * @param stmt containing the new expression.
91     	 * @param method containing <code>stamt</code>.
92     	 * @pre stmt != null and method != null
93     	 * @pre stmt.getUseBoxes()->exists(o | o.getValue().oclIsKindOf(NewExpr))
94     	 */
95     	void processNewExpr(final Stmt stmt, final SootMethod method);
96     
97     	/**
98     	 * Process the parameter reference in <code>stmt</code> for inclusion in the slice.
99     	 * <p>
100     	 * This should be called from within the method's context (callStack containing the call to method as TOS).
101     	 * </p>
102     	 * 
103     	 * @param stmt in which the parameter is referenced.
104     	 * @param method containting <code>stmt</code>.
105     	 * @pre method != null and stmt != null
106     	 */
107     	void processParameterRef(IdentityStmt stmt, SootMethod method);
108     
109     	/**
110     	 * Reset the part.
111     	 */
112     	void reset();
113     
114     	/**
115     	 * Retrieves the value boxes at the given given program point that should be considered while transforming the given
116     	 * program point. Other value boxes occurring in the statement may also be included in the result.
117     	 * 
118     	 * @param valueBox is the program point being transformed.
119     	 * @param stmt in which the program point occurs.
120     	 * @return a collection of value boxes.
121     	 * @pre valueBox != null and stmt != null
122     	 * @post result != null
123     	 */
124     	Collection<ValueBox> retrieveValueBoxesToTransformExpr(ValueBox valueBox, Stmt stmt);
125     
126     	/**
127     	 * Retrieves the value boxes reachable from the given statement (via containment relationship) that should be considered
128     	 * while transforming the given statement.
129     	 * 
130     	 * @param stmt is the statement being transformed.
131     	 * @return a collection of value boxes.
132     	 * @pre stmt != null
133     	 * @post result != null
134     	 */
135     	Collection<ValueBox> retrieveValueBoxesToTransformStmt(Stmt stmt);
136     }
