43     /**
44      * This class provides an implementation of <code>INewExpr2InitMapper</code> based on object flow information. The approach
45      * pairs the instance creation expression and the invocation expression only when the object created at the creation
46      * expression flows into the primary at the invocation site.
47      * 
48      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
49      * @author $Author$
50      * @version $Revision$ $Date$
51      */
52     public class NewExpr2InitMapper
53     		extends AbstractValueAnalyzerBasedProcessor<Value>
54     		implements INewExpr2InitMapper {
55     
56     	/**
57     	 * This is a cache of the context.
58     	 */
59     	private Context contextCache = new Context();
60     
61     	/**
62     	 * This maps methods to a map from new expression occurring statement to init invocation expression occurring statement.
63     	 * 
64     	 * @invariant method2map != null
65     	 * @invariant method2map.keySet()->forall(o | method2map.get(o)->forall(p | (
66     	 *            p.getValue().getInvokeExpr().oclIsKindOf(SpecialInvokeExpr))))
67     	 */
68     	private final Map<SootMethod, Map<Stmt, InvokeStmt>> method2initStmt = new HashMap<SootMethod, Map<Stmt, InvokeStmt>>(
69     			Constants.getNumOfMethodsInApplication());
70     
71     	/**
72     	 * This maps methods to a map from new expressions to statements in which they occur.
73     	 * 
74     	 * @invariant method2newExprStmt.keySet()->forall(o | method2newExprStmt.get(o)->keySet()->forall(p | (
75     	 *            p.oclIsKindOf(NewExpr) or p.oclIsKindOf(NewArrayExpr) or p.oclIsKindOf(NewMultiArrayExpr) or
76     	 *            p.oclIsKindOf(StringConstant))))
77     	 */
78     	private final Map<SootMethod, Map<Value, Stmt>> method2newExprStmt = new HashMap<SootMethod, Map<Value, Stmt>>(Constants
79     			.getNumOfMethodsInApplication());
80     
81     	/**
82     	 * This is the object flow information to be used to improve precision.
83     	 */
84     	private IValueAnalyzer<Value> ofa;
85     
86     	/**
87     	 * {@inheritDoc}
88     	 * 
89     	 * @see edu.ksu.cis.indus.processing.IProcessor#callback(soot.ValueBox, edu.ksu.cis.indus.processing.Context)
90     	 */
91     	@Override public void callback(final ValueBox vBox, final Context context) {
92     		final Value _value = vBox.getValue();
93     
94     		if (_value instanceof NewExpr) {
95     			final Stmt _stmt = context.getStmt();
96     			final SootMethod _method = context.getCurrentMethod();
97     
98     			final Map<Value, Stmt> _ne2init = MapUtils.getMapFromMap(method2newExprStmt, _method);
99     			_ne2init.put(_value, _stmt);
100     		} else if (_value instanceof SpecialInvokeExpr && context.getStmt() instanceof InvokeStmt) {
101     			final InvokeStmt _stmt = (InvokeStmt) context.getStmt();
102     			final SootMethod _method = context.getCurrentMethod();
103     			final SpecialInvokeExpr _expr = (SpecialInvokeExpr) _value;
104     			final SootMethod _sm = _expr.getMethod();
105     
106     			if (_sm.getName().equals("<init>")) {
107     				final Map<Stmt, InvokeStmt> _ne2initStmt = MapUtils.getMapFromMap(method2initStmt, _method);
108     				final Map<Value, Stmt> _ne2newStmt = MapUtils.getMapFromMap(method2newExprStmt, _method);
109     				contextCache.setRootMethod(_method);
110     				contextCache.setStmt(_stmt);
111     				contextCache.setProgramPoint(_expr.getBaseBox());
112     
113     				final Collection<Value> _values = ofa.getValues(_expr.getBase(), contextCache);
114     
115     				for (final Iterator<Value> _i = _values.iterator(); _i.hasNext();) {
116     					final Value _o = _i.next();
117     					final Stmt _newStmt = _ne2newStmt.remove(_o);
118     					_ne2initStmt.put(_newStmt, _stmt);
119     				}
120     			}
121     		}
122     	}
123     
124     	/**
125     	 * {@inheritDoc}
126     	 * 
127     	 * @see edu.ksu.cis.indus.processing.AbstractProcessor#consolidate()
128     	 */
129     	@Override public void consolidate() {
130     		super.consolidate();
131     		contextCache.setStmt(null);
132     		contextCache.setProgramPoint(null);
133     		contextCache.setRootMethod(null);
134     	}
135     
136     	/**
137     	 * {@inheritDoc}
138     	 * 
139     	 * @see INewExpr2InitMapper#getInitCallStmtForNewExprStmt(Stmt,SootMethod)
140     	 */
141     	public InvokeStmt getInitCallStmtForNewExprStmt(final Stmt newExprStmt, final SootMethod method) {
142     		final Map<Stmt, InvokeStmt> _ne2init = method2initStmt.get(method);
143     		InvokeStmt _result = null;
144     
145     		if (_ne2init != null) {
146     			_result = _ne2init.get(newExprStmt);
147     		}
148     		return _result;
149     	}
150     
151     	/**
152     	 * {@inheritDoc}
153     	 * 
154     	 * @see edu.ksu.cis.indus.processing.IProcessor#hookup(edu.ksu.cis.indus.processing.ProcessingController)
155     	 */
156     	public void hookup(final ProcessingController ppc) {
157     		ppc.register(NewExpr.class, this);
158     		ppc.register(NewArrayExpr.class, this);
159     		ppc.register(NewMultiArrayExpr.class, this);
160     		ppc.register(SpecialInvokeExpr.class, this);
161     		ppc.register(StringConstant.class, this);
162     	}
163     
164     	/**
165     	 * {@inheritDoc}
166     	 * 
167     	 * @see edu.ksu.cis.indus.processing.IProcessor#processingBegins()
168     	 */
169     	@Override public void processingBegins() {
170     		method2initStmt.clear();
171     	}
172     
173     	/**
174     	 * Resets the data structures.
175     	 */
176     	@Override public final void reset() {
177     		method2initStmt.clear();
178     	}
179     
180     	/**
181     	 * {@inheritDoc}
182     	 * 
183     	 * @see AbstractValueAnalyzerBasedProcessor#setAnalyzer(IValueAnalyzer)
184     	 */
185     	@Override public void setAnalyzer(final IValueAnalyzer<Value> analyzer) {
186     		ofa = analyzer;
187     	}
188     
189     	/**
190     	 * {@inheritDoc}
191     	 * 
192     	 * @see edu.ksu.cis.indus.processing.IProcessor#unhook(edu.ksu.cis.indus.processing.ProcessingController)
193     	 */
194     	public void unhook(final ProcessingController ppc) {
195     		ppc.unregister(NewExpr.class, this);
196     		ppc.unregister(NewArrayExpr.class, this);
197     		ppc.unregister(NewMultiArrayExpr.class, this);
198     		ppc.unregister(SpecialInvokeExpr.class, this);
199     		ppc.unregister(StringConstant.class, this);
200     	}
201     }
