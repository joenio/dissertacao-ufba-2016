17     package edu.ksu.cis.indus.interfaces;
18     
19     import edu.ksu.cis.indus.common.datastructures.Triple;
20     
21     import edu.ksu.cis.indus.processing.Context;
22     
23     import java.util.Collection;
24     import java.util.List;
25     
26     import soot.SootMethod;
27     
28     import soot.jimple.InvokeExpr;
29     import soot.jimple.Stmt;
30     
31     /**
32      * This interface provides call graph information pertaining to the analyzed system. It is adviced that any post processor
33      * which provides Call graph information should provide it via this interface.
34      * <p>
35      * Subtypes of this class have to return the constant <code>ID</code> defined in this class as a result of
36      * <code>getId</code>.
37      * </p>
38      * <p>
39      * A method <i>a</i> is reachable from method <i>b</i> if there is an explicit call from <i>a</i> to <i>b</i>. This is
40      * also holds when <i>a=b</i>.
41      * </p>
42      *
43      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
44      * @author $Author: rvprasad $
45      * @version $Revision: 1.22 $
46      */
47     public interface ICallGraphInfo
48     		extends IStatus, IIdentification {
49     
50     	/**
51     	 * This class captures in the information pertaining to a call relation. It provides the expression, statement, and the
52     	 * method in which the call occurs. However, depending on the call relation context, the method may indicate the method
53     	 * called at an expression and a statement.
54     	 *
55     	 * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
56     	 * @author $Author: rvprasad $
57     	 * @version $Revision: 1.22 $
58     	 */
59     	final class CallTriple
60     			extends Triple<InvokeExpr, Stmt, SootMethod> {
61     
62     		/**
63     		 * Creates a new CallTriple object. It is assumed that none of the components of this triple will not change their
64     		 * state in ways that will affect equality test or hash code value of the component.
65     		 *
66     		 * @param method is the caller or the callee in the relation.
67     		 * @param stmt in which the call occurs.
68     		 * @param expr is the call.
69     		 * @pre method != null and stmt != null and expr != null
70     		 */
71     		public CallTriple(final SootMethod method, final Stmt stmt, final InvokeExpr expr) {
72     			super(expr, stmt, method);
73     			optimize();
74     		}
75     
76     		/**
77     		 * Returns the call expression.
78     		 *
79     		 * @return the call expression.
80     		 * @post result != null
81     		 */
82     		public InvokeExpr getExpr() {
83     			return getFirst();
84     		}
85     
86     		/**
87     		 * Returns the caller or the callee.
88     		 *
89     		 * @return the caller/callee.
90     		 * @post result != null
91     		 */
92     		public SootMethod getMethod() {
93     			return getThird();
94     		}
95     
96     		/**
97     		 * Returns the statement in which the call occurs.
98     		 *
99     		 * @return the statement containing the call.
100     		 * @post result != null
101     		 */
102     		public Stmt getStmt() {
103     			return getSecond();
104     		}
105     
106     		/**
107     		 * Provides a stringized representation of this object.
108     		 *
109     		 * @return stringized representation of this object.
110     		 * @post result != null
111     		 */
112     		@Override protected String stringize() {
113     			return getStmt() + "@" + getMethod();
114     		}
115     	}
116     
117     	/**
118     	 * The id of this interface.
119     	 */
120     	Comparable<?> ID = "Callgraph Information";
121     
122     	/**
123     	 * Checks if any of the given methods are reachable from the given caller.
124     	 *
125     	 * @param methods that should be reached.
126     	 * @param caller containing the call-site.
127     	 * @return <code>true</code> if any of the <code>methods</code> are reachable from <code>caller</code>;
128     	 *         <code>false</code>, otherwise.
129     	 * @pre mehtod != null and caller != null
130     	 * @pre stmt.containsInvokeExpr()
131     	 */
132     	boolean areAnyMethodsReachableFrom(Collection<SootMethod> methods, SootMethod caller);
133     
134     	/**
135     	 * Checks if any of the given methods are reachable from the given call-site.
136     	 *
137     	 * @param methods that should be reached.
138     	 * @param stmt containing the call-site.
139     	 * @param caller containing the call-site.
140     	 * @return <code>true</code> if any of the <code>methods</code> are reachable from call-site at <code>stmt</code> in
141     	 *         <code>caller</code>; <code>false</code>, otherwise.
142     	 * @pre mehtod != null and stmt != null and caller != null
143     	 * @pre stmt.containsInvokeExpr()
144     	 */
145     	boolean areAnyMethodsReachableFrom(Collection<SootMethod> methods, Stmt stmt, SootMethod caller);
146     
147     	/**
148     	 * Returns the set of methods called in the given <code>expr</code> in the given <code>context</code>.
149     	 *
150     	 * @param expr that is the call.
151     	 * @param context in which the expr occurs. The calling method should occurs in the call string of the context as the
152     	 *            current method.
153     	 * @return a collection of methods.
154     	 * @pre expr != null and context != null
155     	 * @post result != null
156     	 */
157     	Collection<SootMethod> getCallees(InvokeExpr expr, Context context);
158     
159     	/**
160     	 * Returns the set of methods called in <code>caller</code>.
161     	 *
162     	 * @param caller of interest.
163     	 * @return a collection of call-sites.
164     	 * @pre caller != null
165     	 * @post result != null
166     	 */
167     	Collection<CallTriple> getCallees(SootMethod caller);
168     
169     	/**
170     	 * Returns the set of methods that call <code>callee</code>.
171     	 *
172     	 * @param callee is the method invoked.
173     	 * @return a colleciton of call-sites.
174     	 * @pre callee != null
175     	 * @post result != null
176     	 */
177     	Collection<CallTriple> getCallers(SootMethod callee);
178     
179     	/**
180     	 * Returns the intersection of the methods reachable from the given methods in the given direction. This is equivalent to
181     	 * <code>CollectionUtils.intersection(getMethodsReachableFrom(method1, forward1), getMethodsReachableFrom(method2,
182     	 * forward2))</code>.
183     	 *
184     	 * @param method1 of interest.
185     	 * @param forward1 direction of reachability from <code>method1</code>.
186     	 * @param method2 of interest.
187     	 * @param forward2 direction of reachability from <code>method2</code>.
188     	 * @return a collection of methods.
189     	 * @pre method1 != null and method2 != null
190     	 * @post result != null
191     	 * @post getReachableFrom(method2, forward2)->forall(o | getReachableFrom(method1, forward2).contains(o) implies
192     	 *       result.contains(o))
193     	 * @post result->forall(o | getReachableFrom(method2, forward2).contains(o) and getReachableFrom(method1,
194     	 *       forward1).contains(o))
195     	 */
196     	Collection<SootMethod> getCommonMethodsReachableFrom(SootMethod method1, boolean forward1, SootMethod method2,
197     			boolean forward2);
198     
199     	/**
200     	 * Retrieves the minimum set of callees that ensure the given methods reach common callees.
201     	 *
202     	 * @param method1 is one method of interest.
203     	 * @param method2 is another method of interest.
204     	 * @return a collection of methods.
205     	 * @pre method1 != null and method2 != null
206     	 * @post result != null
207     	 * @post getReachableMethods().containsAll(result)
208     	 * @post
209     	 * @post result->forall(o | isReachable(o, method, false) and isReachable(o, method, false))
210     	 */
211     	Collection<SootMethod> getConnectivityCalleesFor(SootMethod method1, SootMethod method2);
212     
213     	/**
214     	 * Retrieves the minimum set of callers that ensure the common callers can reach the given methods.
215     	 *
216     	 * @param method1 is one method of interest.
217     	 * @param method2 is another method of interest.
218     	 * @return a collection of methods.
219     	 * @pre method1 != null and method2 != null
220     	 * @post result != null
221     	 * @post getReachableMethods().containsAll(result)
222     	 * @post
223     	 * @post result->forall(o | isReachable(o, method, true) and isReachable(o, method, true))
224     	 */
225     	Collection<SootMethod> getConnectivityCallersFor(SootMethod method1, SootMethod method2);
226     
227     	/**
228     	 * Returns the methods from which the system starts.
229     	 *
230     	 * @return a colleciton of <code>SootMethod</code>s.
231     	 * @post result != null
232     	 */
233     	Collection<SootMethod> getEntryMethods();
234     
235     	/**
236     	 * Retrieves the methods in the call graph in topological order.
237     	 *
238     	 * @param topdown <code>true</code> indicates in top-down order; <code>false</code> indicates bottom-up order.
239     	 * @return the methods in the call graph.
240     	 * @post result != null
241     	 */
242     	List<SootMethod> getMethodsInTopologicalOrder(boolean topdown);
243     
244     	/**
245     	 * Returns the methods that are reachable from withing the given method via a call chain.
246     	 *
247     	 * @param root in which the method invocation occurs.
248     	 * @param forward <code>true</code> indicates that methods reachable by following a call chain from <code>root</code>
249     	 *            are required. <code>false</code> indicates that methods that can reach <code>root</code> by following a
250     	 *            call chain are required.
251     	 * @return a collection of reachable methods.
252     	 * @pre root != null
253     	 * @post result != null
254     	 */
255     	Collection<SootMethod> getMethodsReachableFrom(SootMethod root, boolean forward);
256     
257     	/**
258     	 * Returns the methods that are reachable from the given invocation point via a call chain.
259     	 *
260     	 * @param stmt in which the method invocation occurs.
261     	 * @param root in which the method invocation occurs.
262     	 * @return a collection of reachable methods.
263     	 * @pre stmt != null and root != null and stmt.containsInvokeExpr() == true
264     	 * @post result != null
265     	 */
266     	Collection<SootMethod> getMethodsReachableFrom(Stmt stmt, SootMethod root);
267     
268     	/**
269     	 * Returns a collection of methods that can be reached in the analyzed system.
270     	 *
271     	 * @return a collection of methods.
272     	 * @post result != null
273     	 */
274     	Collection<SootMethod> getReachableMethods();
275     
276     	/**
277     	 * Returns a sequence of strongly connected components in the given call graph.
278     	 *
279     	 * @param topDown <code>true</code> indicates returned sccs should be in the top-down order; <code>false</code>,
280     	 *            indicates bottom-up.
281     	 * @return a sequence of sequence of methods.
282     	 * @post result != null
283     	 */
284     	List<List<SootMethod>> getSCCs(boolean topDown);
285     
286     	/**
287     	 * Checks if the <code>callee</code> is reachable <code>caller</code>.
288     	 *
289     	 * @param callee is the method that needs to be reached.
290     	 * @param caller is the method to start the search from.
291     	 * @return <code>true</code> if <code>callee</code> can be reached from <code>caller</code>; <code>false</code>,
292     	 *         otherwise.
293     	 * @pre callee != null and caller != null
294     	 * @post not (isReachable(callee) and isReachable(caller)) implies result == false
295     	 */
296     	boolean isCalleeReachableFromCaller(SootMethod callee, SootMethod caller);
297     
298     	/**
299     	 * Checks if the <code>callee</code> is reachable <code>caller</code>.
300     	 *
301     	 * @param callee is the method that needs to be reached.
302     	 * @param stmt in the <code>caller</code> at which invocation occurs.
303     	 * @param caller is the method to start the search from.
304     	 * @return <code>true</code> if <code>callee</code> can be reached from <code>caller</code> via the call at
305     	 *         <code>stmt</code>; <code>false</code>, otherwise.
306     	 * @pre callee != null and caller != null and stmt != null
307     	 * @post not (isReachable(callee) and isReachable(caller)) implies result == false
308     	 * @post result implies isCalleeReachableFromCaller(callee, caller)
309     	 */
310     	boolean isCalleeReachableFromCallSite(SootMethod callee, Stmt stmt, SootMethod caller);
311     
312     	/**
313     	 * Checks if the <code>method</code> is reachable in the analyzed system.
314     	 *
315     	 * @param method to be checked for reachability.
316     	 * @return <code>true</code> if <code>method</code> can be reached in the analyzed system; <code>false</code>,
317     	 *         otherwise.
318     	 * @pre method != null
319     	 */
320     	boolean isReachable(SootMethod method);
321     }
