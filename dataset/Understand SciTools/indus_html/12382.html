48     	/**
49     	 * Composes the given iterable in the given order.
50     	 * 
51     	 * @param <T3> is the type of objects provided by the resulting iterable.
52     	 * @param <T1> is the type of objects provided by the first component iterable.
53     	 * @param <T2> is the type of objects provided by the second component iterable.
54     	 * @param i1 is one of the component iterable.
55     	 * @param i2 is the other component iterable.
56     	 * @return a iterable that will provide the elements from <code>i1</code> and then from <code>i2</code>.
57     	 */
58     	@Functional @NonNull public static <T3, T1 extends T3, T2 extends T3> Iterable<T3> chainedIterable(
59     			@NonNull @Immutable final Iterable<T1> i1, @NonNull @Immutable final Iterable<T2> i2) {
60     		return new Iterable<T3>() {
61     
62     			public Iterator<T3> iterator() {
63     				return chainedIterator(i1.iterator(), i2.iterator());
64     			}
65     		};
66     	}
67     
68     	/**
69     	 * Composes the given iterators in the given order.
70     	 * 
71     	 * @param <T3> is the type of objects provided by the resulting iterator.
72     	 * @param <T1> is the type of objects provided by the first component iterator.
73     	 * @param <T2> is the type of objects provided by the second component iterator.
74     	 * @param i1 is one of the component iterator.
75     	 * @param i2 is the other component iterator.
76     	 * @return a iterator that will provide the elements from <code>i1</code> and then from <code>i2</code>.
77     	 */
78     	@Functional @NonNull public static <T3, T1 extends T3, T2 extends T3> Iterator<T3> chainedIterator(
79     			@NonNull @Immutable final Iterator<T1> i1, @NonNull @Immutable final Iterator<T2> i2) {
80     		return new Iterator<T3>() {
81     
82     			private Iterator<? extends T3> curr = i1;
83     
84     			public boolean hasNext() {
85     				if (curr.hasNext()) {
86     					return true;
87     				}
88     				curr = i2;
89     				return curr.hasNext();
90     			}
91     
92     			public T3 next() {
93     				if (hasNext()) {
94     					return curr.next();
95     				}
96     				throw new NoSuchElementException();
97     			}
98     
99     			public void remove() {
100     				curr.remove();
101     			}
102     		};
103     	}

105     	/**
106     	 * Provides an iterable for the given collection that returns only the elements that satisfy the given predicate.
107     	 * 
108     	 * @param <T1> is the type of object in the collection.
109     	 * @param <T2> is the type of input object to the predicate.
110     	 * @param c is the given collection.
111     	 * @param p is the given predicate.
112     	 * @return an iterable.
113     	 */
114     	@Functional @NonNull public static <T1, T2 extends T1> Iterable<T2> filteredIterable(
115     			@NonNull @Immutable final Collection<T2> c, @NonNull @Immutable final IPredicate<T1> p) {
116     		return new Iterable<T2>() {
117     
118     			public Iterator<T2> iterator() {
119     				return filteredIterator(c.iterator(), p);
120     			}
121     		};
122     	}
123     
124     	/**
125     	 * Provides an iterator that returns only the elements from the given iterator that satisfy the given predicate.
126     	 * 
127     	 * @param <T1> is the type of object returned by the given iterator.
128     	 * @param <T2> is the type of input object to the predicate.
129     	 * @param i is the given iterator.
130     	 * @param p is the given predicate.
131     	 * @return an iterator.
132     	 */
133     	@Functional @NonNull public static <T1, T2 extends T1> Iterator<T2> filteredIterator(
134     			@NonNull @Immutable final Iterator<T2> i, @NonNull @Immutable final IPredicate<T1> p) {
135     		return new Iterator<T2>() {
136     
137     			T2 nextItem;
138     
139     			boolean nextItemUsed = true;
140     
141     			public boolean hasNext() {
142     				final boolean _result;
143     				if (nextItemUsed) {
144     					_result = moveToNextObject();
145     				} else {
146     					_result = true;
147     				}
148     				return _result;
149     			}
150     
151     			public T2 next() {
152     				if (nextItemUsed) {
153     					if (!moveToNextObject()) {
154     						throw new NoSuchElementException();
155     					}
156     				}
157     				nextItemUsed = true;
158     				return nextItem;
159     			}
160     
161     			public void remove() {
162     				if (nextItemUsed) {
163     					i.remove();
164     				} else {
165     					throw new IllegalStateException("Either next() was not called or remove() was called successively.");
166     				}
167     			}
168     
169     			private boolean moveToNextObject() {
170     				while (i.hasNext()) {
171     					final T2 _t = i.next();
172     					if (p.evaluate(_t)) {
173     						nextItem = _t;
174     						nextItemUsed = false;
175     						return true;
176     					}
177     				}
178     				return false;
179     			}
180     		};
181     	}
182     
183     	/**
184     	 * Provides the content of the iterable as a list.
185     	 * 
186     	 * @param <T> is the type of object in the iterable.
187     	 * @param i is the iterable.
188     	 * @return the list containing the contents accessible via <code>i</code>.
189     	 */
190     	@Functional @NonNull public static <T> List<T> toList(@NonNull @Immutable final Iterable<T> i) {
191     		final List<T> _r = new ArrayList<T>();
192     		for (final T _t : i) {
193     			_r.add(_t);
194     		}
195     		return _r;
196     	}
197     
198     	/**
199     	 * Provides the content of the iterator as a list.
200     	 * 
201     	 * @param <T> is the type of object in the iterator.
202     	 * @param i is the iterator.
203     	 * @return the list containing the contents accessible via <code>i</code>.
204     	 */
205     	@Functional @NonNull public static <T> List<T> toList(@NonNull @Immutable final Iterator<T> i) {
206     		final List<T> _r = new ArrayList<T>();
207     		for (; i.hasNext();) {
208     			_r.add(i.next());
209     		}
210     		return _r;
211     	}
