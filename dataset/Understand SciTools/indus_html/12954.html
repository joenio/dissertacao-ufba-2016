17     package edu.ksu.cis.indus.common.collections;
18     
19     import edu.ksu.cis.indus.annotations.Functional;
20     import edu.ksu.cis.indus.annotations.Immutable;
21     import edu.ksu.cis.indus.annotations.NonNull;
22     
23     import java.util.AbstractMap;
24     import java.util.Map;
25     import java.util.Set;
26     
27     /**
28      * A factory based lazy map that uses the factory to creates values when there is no mapping for a given key.
29      * 
30      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
31      * @author $Author: rvprasad $
32      * @version $Revision: 1.5 $ $Date: 2007/02/10 19:08:38 $
33      * @param <K> is the type of the key.
34      * @param <V> is the type of the value.
35      */
36     public class FactoryBasedLazyMap<K, V>
37     		extends AbstractMap<K, V> {
38     
39     	/**
40     	 * The map that actually stores the mapping.
41     	 */
42     	@NonNull private final Map<K, V> backedMap;
43     
44     	/**
45     	 * The factory used to create values.
46     	 */
47     	@NonNull private final IFactory<V> factory;
48     
49     	/**
50     	 * Creates an instance of this class.
51     	 * 
52     	 * @param map to be used to store the actual mappings.
53     	 * @param valueFactory to be used to create values.
54     	 */
55     	public FactoryBasedLazyMap(@NonNull @Immutable final Map<K, V> map, @NonNull @Immutable final IFactory<V> valueFactory) {
56     		backedMap = map;
57     		factory = valueFactory;
58     	}
59     
60     	/**
61     	 * {@inheritDoc}
62     	 */
63     	@Functional @Override public boolean containsKey(final Object key) {
64     		return get(key) != null;
65     	}
66     
67     	/**
68     	 * {@inheritDoc}
69     	 */
70     	@Functional @Override @NonNull public Set<Map.Entry<K, V>> entrySet() {
71     		return backedMap.entrySet();
72     	}
73     
74     	/**
75     	 * {@inheritDoc}
76     	 */
77     	@Functional @Override public boolean equals(final Object o) {
78     		return backedMap.equals(o);
79     	}
80     
81     	/**
82     	 * {@inheritDoc}
83     	 */
84     	@Functional @Override public V get(final Object key) {
85     		if (!backedMap.containsKey(key)) {
86     			final V _c = factory.create();
87     			@SuppressWarnings("unchecked") final K _k = (K) key;
88     			backedMap.put(_k, _c);
89     			return _c;
90     		}
91     		return backedMap.get(key);
92     	}
93     
94     	/**
95     	 * {@inheritDoc}
96     	 */
97     	@Functional @Override public int hashCode() {
98     		return backedMap.hashCode();
99     	}
100     
101     	/**
102     	 * {@inheritDoc}
103     	 */
104     	@Override public V put(@Immutable final K key, @Immutable final V value) {
105     		return backedMap.put(key, value);
106     	}
107     }
