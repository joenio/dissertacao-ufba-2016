17     package edu.ksu.cis.indus.common.scoping;
18     
19     import edu.ksu.cis.indus.annotations.Empty;
20     import edu.ksu.cis.indus.annotations.Functional;
21     import edu.ksu.cis.indus.annotations.Immutable;
22     import edu.ksu.cis.indus.annotations.NonNull;
23     import edu.ksu.cis.indus.interfaces.IEnvironment;
24     
25     import java.util.regex.Pattern;
26     
27     import org.apache.commons.lang.builder.ToStringBuilder;
28     import org.slf4j.Logger;
29     import org.slf4j.LoggerFactory;
30     
31     import soot.SootField;
32     
33     /**
34      * This class represents field-level scope specification.
35      * 
36      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
37      * @author $Author: rvprasad $
38      * @version $Revision: 1.12 $ $Date: 2007/02/10 19:08:39 $
39      */
40     public final class FieldSpecification
41     		extends AbstractSpecification {
42     
43     	/**
44     	 * The logger used by instances of this class to log messages.
45     	 */
46     	private static final Logger LOGGER = LoggerFactory.getLogger(FieldSpecification.class);
47     
48     	/**
49     	 * This is the specification of the type of the class that declares the field.
50     	 */
51     	@NonNull private TypeSpecification declaringClassSpec;
52     
53     	/**
54     	 * This is the specification of the type of the field.
55     	 */
56     	@NonNull private TypeSpecification fieldTypeSpec;
57     
58     	/**
59     	 * The pattern of the field's name.
60     	 */
61     	@NonNull private Pattern namePattern;
62     
63     	/**
64     	 * Creates an instance of this class.
65     	 */
66     	@Empty public FieldSpecification() {
67     		super();
68     	}
69     
70     	/**
71     	 * Retrieves the specification of the class that declares the field.
72     	 * 
73     	 * @return the specification.
74     	 */
75     	@NonNull @Functional public TypeSpecification getDeclaringClassSpec() {
76     		return declaringClassSpec;
77     	}
78     
79     	/**
80     	 * Retrieves the specification of the field's name.
81     	 * 
82     	 * @return the specification.
83     	 */
84     	@NonNull @Functional public String getFieldNameSpec() {
85     		return namePattern.pattern();
86     	}
87     
88     	/**
89     	 * Retrieves the specification of the type of the field.
90     	 * 
91     	 * @return the specification.
92     	 */
93     	@NonNull @Functional public TypeSpecification getFieldTypeSpec() {
94     		return fieldTypeSpec;
95     	}
96     
97     	/**
98     	 * Checks if the given field is in the scope of this specification in the given environment.
99     	 * 
100     	 * @param field to be checked for scope constraints.
101     	 * @param system in which the check the constraints.
102     	 * @return <code>true</code> if the given field lies within the scope defined by this specification; <code>false</code>,
103     	 *         otherwise.
104     	 */
105     	@Functional public boolean isInScope(@NonNull final SootField field, @NonNull final IEnvironment system) {
106     		boolean _result = accessConformant(new AccessSpecifierWrapper(field));
107     		_result = _result && fieldTypeSpec.conformant(field.getType(), system);
108     		_result = _result && declaringClassSpec.conformant(field.getDeclaringClass().getType(), system);
109     		_result = _result && namePattern.matcher(field.getName()).matches();
110     
111     		if (!isInclusion()) {
112     			_result = !_result;
113     		}
114     
115     		if (LOGGER.isDebugEnabled()) {
116     			LOGGER.debug(this + " " + field + " " + _result);
117     		}
118     
119     		return _result;
120     	}
121     
122     	/**
123     	 * Sets the specification of the class that declares the field.
124     	 * 
125     	 * @param spec the specification.
126     	 */
127     	public void setDeclaringClassSpec(@NonNull @Immutable final TypeSpecification spec) {
128     		declaringClassSpec = spec;
129     	}
130     
131     	/**
132     	 * Sets the specification of the field's name.
133     	 * 
134     	 * @param spec is a regular expression.
135     	 */
136     	public void setFieldNameSpec(@NonNull @Immutable final String spec) {
137     		namePattern = Pattern.compile(spec);
138     	}
139     
140     	/**
141     	 * Sets the specification of the type of the field.
142     	 * 
143     	 * @param spec the specification.
144     	 */
145     	public void setFieldTypeSpec(@NonNull @Immutable final TypeSpecification spec) {
146     		fieldTypeSpec = spec;
147     	}
148     
149     	/**
150     	 * {@inheritDoc}
151     	 */
152     	@Functional @Override public String toString() {
153     		return new ToStringBuilder(this).appendSuper(super.toString()).append("namePattern", this.namePattern.pattern())
154     				.append("fieldTypeSpec", this.fieldTypeSpec).append("declaringClassSpec", this.declaringClassSpec).toString();
155     	}
156     }
