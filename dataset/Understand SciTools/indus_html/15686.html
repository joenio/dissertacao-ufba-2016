17     package edu.ksu.cis.indus.common.scoping;
18     
19     import edu.ksu.cis.indus.annotations.Functional;
20     import edu.ksu.cis.indus.annotations.NonNull;
21     
22     import java.util.EnumSet;
23     
24     import org.apache.commons.lang.builder.ToStringBuilder;
25     
26     /**
27      * This class represents access control specification for the purpose of serialization and deserialization in the realm of
28      * scope definition.
29      * 
30      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
31      * @author $Author: rvprasad $
32      * @version $Revision: 1.12 $ $Date: 2007/02/10 19:08:39 $
33      */
34     public final class AccessSpecification {
35     
36     	/**
37     	 * This contains the access specification encoded in terms of XXX_ACCESS constants defined in
38     	 * <code>AccessSpecificationEnum</code>.
39     	 */
40     	private final EnumSet<AccessSpecificationEnum> access = EnumSet.noneOf(AccessSpecificationEnum.class);
41     
42     	/**
43     	 * Creates a new AccessSpecification object with default access set to public and protected.
44     	 */
45     	public AccessSpecification() {
46     		access.add(AccessSpecificationEnum.PROTECTED_ACCESS);
47     		access.add(AccessSpecificationEnum.PUBLIC_ACCESS);
48     	}
49     
50     	/**
51     	 * Checks if the given access control specifier confirms to this specification.
52     	 * 
53     	 * @param specifier to be checked for conformance.
54     	 * @return <code>true</code> if <code>specifier</code> confirms to this spec; <code>false</code>, otherwise.
55     	 */
56     	public boolean conformant(@NonNull final IAccessSpecifiers specifier) {
57     		return isDefaultAccess() == specifier.isDefaultAccess() || isProtectedAccess() == specifier.isProtectedAccess()
58     				|| isPublicAccess() == specifier.isPublicAccess() || isPrivateAccess() == specifier.isPrivateAccess();
59     	}
60     
61     	/**
62     	 * Retrieves default access control level.
63     	 * 
64     	 * @return <code>true</code> if enabled; <code>false</code>, otherwise.
65     	 */
66     	public boolean isDefaultAccess() {
67     		return access.contains(AccessSpecificationEnum.DEFAULT_ACCESS);
68     	}
69     
70     	/**
71     	 * Retrieves private access control level.
72     	 * 
73     	 * @return <code>true</code> if enabled; <code>false</code>, otherwise.
74     	 */
75     	public boolean isPrivateAccess() {
76     		return access.contains(AccessSpecificationEnum.PRIVATE_ACCESS);
77     	}
78     
79     	/**
80     	 * Retrieves protected access control level.
81     	 * 
82     	 * @return <code>true</code> if enabled; <code>false</code>, otherwise.
83     	 */
84     	public boolean isProtectedAccess() {
85     		return access.contains(AccessSpecificationEnum.PROTECTED_ACCESS);
86     	}
87     
88     	/**
89     	 * Retrieves public access control level.
90     	 * 
91     	 * @return <code>true</code> if enabled; <code>false</code>, otherwise.
92     	 */
93     	public boolean isPublicAccess() {
94     		return access.contains(AccessSpecificationEnum.PUBLIC_ACCESS);
95     	}
96     
97     	/**
98     	 * Sets the default access control (package-private) level.
99     	 * 
100     	 * @param value to be set.
101     	 */
102     	public void setDefaultAccess(final boolean value) {
103     		if (value) {
104     			access.add(AccessSpecificationEnum.DEFAULT_ACCESS);
105     		} else {
106     			access.remove(AccessSpecificationEnum.DEFAULT_ACCESS);
107     		}
108     	}
109     
110     	/**
111     	 * Sets the private access control level.
112     	 * 
113     	 * @param value to be set.
114     	 */
115     	public void setPrivateAccess(final boolean value) {
116     		if (value) {
117     			access.add(AccessSpecificationEnum.PRIVATE_ACCESS);
118     		} else {
119     			access.remove(AccessSpecificationEnum.PRIVATE_ACCESS);
120     		}
121     	}
122     
123     	/**
124     	 * Sets the protected access control level.
125     	 * 
126     	 * @param value to be set.
127     	 */
128     	public void setProtectedAccess(final boolean value) {
129     		if (value) {
130     			access.add(AccessSpecificationEnum.PROTECTED_ACCESS);
131     		} else {
132     			access.remove(AccessSpecificationEnum.PROTECTED_ACCESS);
133     		}
134     	}
135     
136     	/**
137     	 * Sets the public access control level.
138     	 * 
139     	 * @param value to be set.
140     	 */
141     	public void setPublicAccess(final boolean value) {
142     		if (value) {
143     			access.add(AccessSpecificationEnum.PUBLIC_ACCESS);
144     		} else {
145     			access.remove(AccessSpecificationEnum.PUBLIC_ACCESS);
146     		}
147     	}
148     
149     	/**
150     	 * {@inheritDoc}
151     	 */
152     	@NonNull @Functional @Override public String toString() {
153     		final StringBuffer _access = new StringBuffer(access + " ");
154     
155     		if (isPrivateAccess()) {
156     			_access.append("Private | ");
157     		}
158     
159     		if (isDefaultAccess()) {
160     			_access.append("Package Private | ");
161     		}
162     
163     		if (isProtectedAccess()) {
164     			_access.append("Protected | ");
165     		}
166     
167     		if (isPublicAccess()) {
168     			_access.append("Public | ");
169     		}
170     		return new ToStringBuilder(this).appendSuper(super.toString()).append("access", _access).toString();
171     	}
172     }
