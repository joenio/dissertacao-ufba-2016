17     package edu.ksu.cis.indus.staticanalyses.flow;
18     
19     import edu.ksu.cis.indus.interfaces.IPrototype;
20     import edu.ksu.cis.indus.staticanalyses.tokens.ITokens;
21     
22     import org.slf4j.Logger;
23     import org.slf4j.LoggerFactory;
24     
25     import soot.SootMethod;
26     import soot.Value;
27     
28     /**
29      * This class manages of method variants. This only provides the implementation to create new method variants. The super class
30      * is responsible of managing the variants.
31      *
32      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
33      * @version $Revision: 1.21 $
34      * @param <SYM> is the type of symbol whose flow is being analyzed.
35      * @param <T>  is the type of the token set object.
36      * @param <N> is the type of the summary node in the flow analysis.
37      */
38     class MethodVariantManager<SYM, T extends ITokens<T, SYM>, N extends IFGNode<SYM, T, N>, R>
39     		extends AbstractVariantManager<IMethodVariant<N>, SootMethod, SYM, T, N, R> {
40     
41     	/**
42     	 * The logger used by instances of this class to log messages.
43     	 */
44     	private static final Logger LOGGER = LoggerFactory.getLogger(MethodVariantManager.class);
45     
46     	/**
47     	 * A prototype object used to create index managers related to AST nodes. Objects created via this prototype object are
48     	 * used by <code>MethodVariant</code>s to manage the variants corresponding to the AST nodes that exists in them.
49     	 */
50     	protected final IPrototype<? extends IIndexManager<? extends IIndex<?>, Value>> astIMPrototype;
51     
52     	/**
53     	 * The factory used to create method variants.
54     	 */
55     	private final IMethodVariantFactory<SYM, T, N, R> mvFactory;
56     
57     	/**
58     	 * Creates a new <code>MethodVariantManager</code> instance.
59     	 *
60     	 * @param theAnalysis the instance of the framework in which this object is used. This parameter cannot be
61     	 *            <code>null</code>.
62     	 * @param indexManager the manager to indices which are used to map methods to their variants. This parameter cannot be
63     	 *            <code>null</code>.
64     	 * @param astIndexManagerPrototype the prototype object used to create index managers related to AST nodes. This
65     	 *            implementation should support <code>getClone()</code>.
66     	 * @param factory used to create method variants.
67     	 * @pre theAnalysis != null and indexManager != null and astIndexManagerPrototype != null
68     	 */
69     	MethodVariantManager(final FA<SYM, T, N, R> theAnalysis,
70     			final IIndexManager<? extends IIndex<?>, SootMethod> indexManager,
71     			final IPrototype<? extends IIndexManager<? extends IIndex<?>, Value>> astIndexManagerPrototype,
72     			final IMethodVariantFactory<SYM, T, N, R> factory) {
73     		super(theAnalysis, indexManager);
74     		this.astIMPrototype = astIndexManagerPrototype;
75     		mvFactory = factory;
76     	}
77     
78     	/**
79     	 * Returns a new variant of the method represented by <code>o</code>.
80     	 *
81     	 * @param o the method whose variant is to be returned.
82     	 * @return the new <code>MethodVariant</code> corresponding to method <code>o</code>.
83     	 * @pre o != null and o.oclIsKindOf(SootMethod)
84     	 * @post result != null
85     	 */
86     	@Override protected IMethodVariant<N> getNewVariant(final SootMethod o) {
87     		if (LOGGER.isDebugEnabled()) {
88     			LOGGER.debug("STATS: Processing method: " + o + "\t number: " + (getVariantCount() + 1));
89     		}
90     
91     		final ASTVariantManager<SYM, T, N, R> _astVM = new ASTVariantManager<SYM, T, N, R>(fa, astIMPrototype.getClone());
92     		final SootMethod _sootMethod = o;
93     		return mvFactory.create(_sootMethod, _astVM, fa);
94     	}
95     }
