175     	 * {@inheritDoc}
176     	 * 
177     	 * @see edu.ksu.cis.indus.slicer.IDirectionSensitivePartOfSlicingEngine#continueProcessing()
178     	 */
179     	public boolean continueProcessing() {
180     		if (LOGGER.isDebugEnabled()) {
181     			LOGGER.debug("continueProcessing() - BEGIN");
182     		}
183     
184     		boolean _result = false;
185     
186     		for (final Iterator<Map.Entry<SootMethod, Collection<Triple<Stmt, SootMethod, Stack<CallTriple>>>>> _i = callee2callsites
187     				.entrySet().iterator(); _i.hasNext();) {
188     			final Map.Entry<SootMethod, Collection<Triple<Stmt, SootMethod, Stack<CallTriple>>>> _entry = _i.next();
189     			final SootMethod _callee = _entry.getKey();
190     			final BitSet _params = method2params.get(_callee);
191     			if (_params != null) {
192     				final Collection<Triple<Stmt, SootMethod, Stack<CallTriple>>> _temp = _entry.getValue();
193     				for (final Iterator<Triple<Stmt, SootMethod, Stack<CallTriple>>> _k = _temp.iterator(); _k.hasNext();) {
194     					final Triple<Stmt, SootMethod, Stack<CallTriple>> _triple = _k.next();
195     					final Stmt _stmt = _triple.getFirst();
196     					final SootMethod _caller = _triple.getSecond();
197     					final Stack<CallTriple> _stack = _triple.getThird();
198     					final InvokeExpr _expr = _stmt.getInvokeExpr();
199     					for (int _j = _params.nextSetBit(0); _j >= 0; _j = _params.nextSetBit(_j + 1)) {
200     						_result |= engine.generateExprLevelSliceCriterion(_expr.getArgBox(_j), _stmt, _caller,
201     								CONSIDER_EXECUTION, _stack);
202     					}
203     				}
204     				if (_params.cardinality() == _callee.getParameterCount()) {
205     					_i.remove();
206     				}
207     			}
208     		}
209     
210     		if (LOGGER.isDebugEnabled()) {
211     			LOGGER.debug("continueProcessing() - END");
212     		}
213     
214     		return _result;
215     	}

217     	/**
218     	 * {@inheritDoc}
219     	 * 
220     	 * @see IDirectionSensitivePartOfSlicingEngine#generateCriteriaForTheCallToMethod(soot.SootMethod, soot.SootMethod,
221     	 *      soot.jimple.Stmt)
222     	 */
223     	public void generateCriteriaForTheCallToMethod(final SootMethod callee, final SootMethod caller, final Stmt callStmt) {
224     		if (LOGGER.isDebugEnabled()) {
225     			LOGGER.debug("generateCriteriaForTheCallToMethod(Stmt callStmt = " + callStmt + "SootMethod callee = " + callee
226     					+ ", SootMethod caller = " + caller + ", stack = " + engine.getCopyOfCallStackCache() + ") - BEGIN");
227     		}
228     
229     		/*
230     		 * _stmt may be an assignment statement. Hence, we want the control to reach the statement but not leave it. However,
231     		 * the execution of the invoke expression should be considered as it is requied to reach the callee. Likewise, we want
232     		 * to include the expression but not all arguments. We rely on the reachable parameters to suck in the arguments. So,
233     		 * we generate criteria only for the invocation expression and not the arguments. Refer to
234     		 * transformAndGenerateToNewCriteriaForXXXX for information about how invoke expressions are handled differently.
235     		 */
236     		engine.generateStmtLevelSliceCriterion(callStmt, caller, false);
237     		engine.includeInSlice(callStmt.getInvokeExprBox());
238     		generateCriteriaForReceiverOfAt(callStmt, caller);
239     		recordCallInfoForProcessingArgsTo(callStmt, caller, callee);
240     
241     		if (LOGGER.isDebugEnabled()) {
242     			LOGGER.debug("generateCriteriaForTheCallToMethod() - END");
243     		}
244     	}

302     	 *

307     		if (LOGGER.isDebugEnabled()) {
308     			LOGGER.debug("processLocalAt(Local local = " + local + ", Stmt stmt = " + stmt + ", SootMethod method = "
309     					+ method + ", stack = " + engine.getCopyOfCallStackCache() + ") - BEGIN");
310     		}
311     
312     		boolean _flag = false;
313     		
314     		if (stmt instanceof DefinitionStmt) {
315     			final Iterator<ValueBox> _i = stmt.getDefBoxes().iterator();
316     			final int _iEnd = stmt.getDefBoxes().size();
317     
318     			for (int _iIndex = 0; _iIndex < _iEnd && !_flag; _iIndex++) {
319     				final ValueBox _vb = _i.next();
320     				final Value _v = _vb.getValue();
321     				_flag = _v.equals(local);
322     			}
323     		}
324     		
325     		engine.generateStmtLevelSliceCriterion(stmt, method, _flag);
326     		
327     		if (stmt.containsInvokeExpr() && _flag) {
328     			final Context _ctxt = new Context();
329     			_ctxt.setRootMethod(method);
330     			_ctxt.setStmt(stmt);
331     			processTailsOf(engine.getCgi().getCallees(stmt.getInvokeExpr(), _ctxt), stmt, method,
332     					returnValueInclClosure);
333     		}
334     
335     		if (LOGGER.isDebugEnabled()) {
336     			LOGGER.debug("processLocalAt() , stack = " + engine.getCopyOfCallStackCache() + "- END");
337     		}
338     	}

340     	/**
341     	 * Processes new expressions to include corresponding init statements into the slice.
342     	 * 
343     	 * @param stmt is the statement containing the new expression.
344     	 * @param method contains <code>stmt</code>.
345     	 * @pre stmt != null and method != null
346     	 */
347     	public void processNewExpr(final Stmt stmt, final SootMethod method) {
348     		if (LOGGER.isDebugEnabled()) {
349     			LOGGER.debug("processNewExpr(Stmt stmt = " + stmt + ", SootMethod method = " + method + ", stack = "
350     					+ engine.getCopyOfCallStackCache() + ") - BEGIN");
351     		}
352     
353     		/*
354     		 * Here we make some assumptions. new expressions will always be assigned to a variable in order to call the
355     		 * constructor. Hence, they will always occur in assignment statement. It is common practice in compilers to emit code
356     		 * to construct the instance just after emitting for creating the instance. Hence, we should be able to find the
357     		 * <init> call site by following use-def chain. However, this may fail in case where the <init> call is made on a
358     		 * variable other than the one to which the new expression was assigned to. r1 = new <X>; r2 = r1; r2.<init>(); To
359     		 * handle such cases, we use OFA.
360     		 */
361     		if (stmt instanceof AssignStmt) {
362     			final AssignStmt _as = (AssignStmt) stmt;
363     
364     			if (_as.getRightOp() instanceof NewExpr) {
365     				final Stmt _def = engine.getInitMapper().getInitCallStmtForNewExprStmt(stmt, method);
366     				if (_def != null) {
367     					engine.generateStmtLevelSliceCriterion(_def, method, true);
368     				} else if (LOGGER.isWarnEnabled()) {
369     					LOGGER.warn("Could not find the <init> for " + stmt + "@" + method);
370     				}
371     			}
372     		}
373     
374     		if (LOGGER.isDebugEnabled()) {
375     			LOGGER.debug("processNewExpr() - END");
376     		}
377     	}

379     	/**
380     	 * Generates new slicing criteria which captures inter-procedural dependences due to call-sites.
381     	 * <p>
382     	 * This should be called from within the callee's context (callStack containing the call to the callee).
383     	 * </p>
384     	 * 
385     	 * @param stmt is the statement in which <code>pBox</code> occurs.
386     	 * @param callee in which<code>stmt</code> occurs.
387     	 * @pre method != null and stmt != null
388     	 */
389     	public void processParameterRef(final IdentityStmt stmt, final SootMethod callee) {
390     		if (LOGGER.isDebugEnabled()) {
391     			LOGGER.debug("processParameterRef(ValueBox pBox = " + stmt.getRightOpBox() + ", SootMethod callee = " + callee
392     					+ ", stack = " + engine.getCopyOfCallStackCache() + ") - BEGIN");
393     		}
394     
395     		final ParameterRef _param = (ParameterRef) stmt.getRightOp();
396     		final int _index = _param.getIndex();
397     
398     		BitSet _params = method2params.get(callee);
399     
400     		if (_params == null) {
401     			// we size the bitset on a maximum sane value for arguments to improve efficiency.
402     			final int _maxArguments = 8;
403     			_params = new BitSet(_maxArguments);
404     			method2params.put(callee, _params);
405     		}
406     		_params.set(_param.getIndex());
407     
408     		if (LOGGER.isDebugEnabled()) {
409     			LOGGER.debug("Parameters required for " + callee + " are " + _params);
410     		}
411     
412     		if (engine.ifInsideContext()) {
413     			final CallTriple _temp = engine.returnFromMethod();
414     			if (_temp != null) {
415     				final SootMethod _caller = _temp.getMethod();
416     				final Stmt _stmt = _temp.getStmt();
417     				final ValueBox _argBox = _temp.getExpr().getArgBox(_index);
418     				engine.generateExprLevelSliceCriterion(_argBox, _stmt, _caller, true);
419     				generateCriteriaForReceiverOfAt(_stmt, _caller);
420     			}
421     			engine.enterMethod(_temp);
422     		} else {
423     			for (final Iterator<CallTriple> _i = engine.getCgi().getCallers(callee).iterator(); _i.hasNext();) {
424     				final CallTriple _ctrp = _i.next();
425     				final SootMethod _caller = _ctrp.getMethod();
426     				final Stmt _stmt = _ctrp.getStmt();
427     				final ValueBox _argBox = _ctrp.getExpr().getArgBox(_index);
428     				engine.generateExprLevelSliceCriterion(_argBox, _stmt, _caller, true);
429     				generateCriteriaForReceiverOfAt(_stmt, _caller);
430     			}
431     		}
432     
433     		if (LOGGER.isDebugEnabled()) {
434     			LOGGER.debug("processParameterRef() - END");
435     		}
436     	}

448     	 * {@inheritDoc}
449     	 * 
450     	 * @see IDirectionSensitivePartOfSlicingEngine#retrieveValueBoxesToTransformExpr(ValueBox, Stmt)
451     	 */
452     	public Collection<ValueBox> retrieveValueBoxesToTransformExpr(final ValueBox valueBox, final Stmt stmt) {
453     		final Collection<ValueBox> _valueBoxes = new HashSet<ValueBox>();
454     		_valueBoxes.add(valueBox);
455     
456     		final Value _value = valueBox.getValue();
457     
458     		// if it is an invocation expression, we do not want to include the arguments/sub-expressions.
459     		// in case of instance invocation, we do want to include the receiver position expression.
460     		if (!(_value instanceof InvokeExpr)) {
461     			_valueBoxes.addAll(_value.getUseBoxes());
462     		} else if (_value instanceof InstanceInvokeExpr) {
463     			_valueBoxes.add(((InstanceInvokeExpr) _value).getBaseBox());
464     		}
465     
466     		/*
467     		 * Note that l-position is the lhs of an assignment statement whereas an l-value is value that occurs the l-position
468     		 * and is defined. In a[i] = v; a is the l-value whereas i is a r-value in the l-position and v is r-value in the
469     		 * r-position.
470     		 */
471     
472     		// we include the unincluded r-values in case the given value box represents a l-value.
473     		if (stmt instanceof DefinitionStmt && stmt.getDefBoxes().contains(valueBox)) {
474     			_valueBoxes.addAll(engine.getCollector().getUncollected(stmt.getUseBoxes()));
475     		}
476     
477     		return _valueBoxes;
478     	}

481     	 * {@inheritDoc}

483     	 * @see IDirectionSensitivePartOfSlicingEngine#retrieveValueBoxesToTransformStmt(Stmt)
484     	 */
485     	public Collection<ValueBox> retrieveValueBoxesToTransformStmt(final Stmt stmt) {
486     		final Collection<ValueBox> _valueBoxes = new HashSet<ValueBox>(stmt.getUseAndDefBoxes());
487     
488     		// if it contains an invocation expression, we do not want to include the arguments/sub-expressions.
489     		if (stmt.containsInvokeExpr()) {
490     			final InvokeExpr _invokeExpr = stmt.getInvokeExpr();
491     
492     			_valueBoxes.removeAll(_invokeExpr.getUseBoxes());
493     
494     			// in case of instance invocation, we do want to include the receiver position expression.
495     			if (_invokeExpr instanceof InstanceInvokeExpr) {
496     				_valueBoxes.add(((InstanceInvokeExpr) _invokeExpr).getBaseBox());
497     			}
498     		}
499     
500     		return _valueBoxes;
501     	}

503     	/**
504     	 * Checks if the given called method's return points should be considered to generate new slice criterion. The callee
505     	 * should be marked as invoked or required before calling this method.
506     	 * 
507     	 * @param callee is the method in question.
508     	 * @param expr indicates if the expression in the tail statement should be checked to determine if criteria should be
509     	 *            generated.
510     	 * @return <code>true</code> if method's return points of callee should be considered to generate new slice criterion;
511     	 *         <code>false</code>, otherwise.
512     	 */
513     	private boolean considerMethodExitForCriteriaGeneration(final SootMethod callee, final boolean expr) {
514     		boolean _result = false;
515     
516     		if (!exitTransformedMethods2exitpoints.containsKey(callee)) {
517     			_result = true;
518     		} else if (expr) {
519     			final Collection<Stmt> _temp = MapUtils.queryCollection(exitTransformedMethods2exitpoints, callee);
520     			final Iterator<Stmt> _i = _temp.iterator();
521     			final int _iEnd = _temp.size();
522     
523     			for (int _iIndex = 0; _iIndex < _iEnd && !_result; _iIndex++) {
524     				final Stmt _tail = _i.next();
525     
526     				if (_tail instanceof ReturnStmt) {
527     					_result |= !engine.getCollector().hasBeenCollected(((ReturnStmt) _tail).getOpBox());
528     				} else if (_tail instanceof ThrowStmt) {
529     					_result |= !engine.getCollector().hasBeenCollected(((ThrowStmt) _tail).getOpBox());
530     				}
531     			}
532     		}
533     
534     		if (LOGGER.isDebugEnabled()) {
535     			LOGGER.debug("Consider Method exit for method " + callee + " is " + _result);
536     		}
537     
538     		return _result;
539     	}

557     		if (!callStmt.getInvokeExpr().getMethod().isStatic()) {

567     	/**
568     	 * Processes the init call to the super class inside init method.
569     	 * 
570     	 * @param initMethod is the init method.
571     	 * @param bbg is the basic block graph of <code>callee</code>.
572     	 * @pre initMethod != null and bbg != null and engine.getCallStackCache() != null
573     	 */
574     	private void processSuperInitInInit(final SootMethod initMethod, final BasicBlockGraph bbg) {
575     		if (LOGGER.isDebugEnabled()) {
576     			LOGGER.debug("processSuperInitInInit(SootMethod initMethod = " + initMethod + ", BasicBlockGraph bbg = " + bbg
577     					+ ", stack = " + engine.getCopyOfCallStackCache() + ") - BEGIN");
578     		}
579     
580     		/*
581     		 * if we are sucking in an init we better suck in the super <init> invoke expression as well. By JLS, this has to be
582     		 * the first statement in the constructor. However, if it accepts arguments, the arguments will be set up before the
583     		 * call. Hence, it is safe to suck in the first <init> invoke expression in the <init> method being sucked in.
584     		 * However, care must be taken to suck in the first <init> invocation that is invokes <init> from the same class as
585     		 * the enclosing <init> method. As we process invocation expressions, we are bound to suck in any other required
586     		 * <init>'s from other higher super classes.
587     		 */
588     		if (initMethod.getName().equals("<init>") && initMethod.getDeclaringClass().hasSuperclass()) {
589     			final LocalUseDefAnalysisv2 _udl = new LocalUseDefAnalysisv2(engine.getBasicBlockGraphManager()
590     					.getBasicBlockGraph(initMethod));
591     			final Collection<Pair<Local, Stmt>> _uses = _udl.getUses((DefinitionStmt) bbg.getHead().getLeaderStmt(),
592     					initMethod);
593     
594     			for (final Iterator<Pair<Local, Stmt>> _i = _uses.iterator(); _i.hasNext();) {
595     				final Pair<Local, Stmt> _pair = _i.next();
596     				final Stmt _stmt = _pair.getSecond();
597     
598     				if (_stmt instanceof InvokeStmt) {
599     					final SootMethod _called = _stmt.getInvokeExpr().getMethod();
600     
601     					if (_called.getName().equals("<init>")
602     							&& _called.getDeclaringClass().equals(initMethod.getDeclaringClass().getSuperclass())) {
603     						engine.generateStmtLevelSliceCriterion(_stmt, initMethod, true);
604     						break;
605     					}
606     				}
607     			}
608     		}
609     
610     		if (LOGGER.isDebugEnabled()) {
611     			LOGGER.debug("processSuperInitInInit() - END");
612     		}
613     	}

615     	/**
616     	 * Processes the callees called at the given statements in the caller with the given closure to include tails of the
617     	 * callees along with any arguments at the call-site.
618     	 * <p>
619     	 * This should be called from the caller's context (callStack containing the call to the caller).
620     	 * </p>
621     	 * 
622     	 * @param callees are the methods called at the statement.
623     	 * @param stmt is the statment containing the invocation.
624     	 * @param caller is the method in which <code>invocationStmt</code> occurs.
625     	 * @param closure to be executed.
626     	 * @pre stmt != null and caller != null and callees != null
627     	 */
628     	private void processTailsOf(final Collection<SootMethod> callees, final Stmt stmt, final SootMethod caller,
629     			final IClosure<Pair<Stmt, SootMethod>> closure) {
630     		if (LOGGER.isDebugEnabled()) {
631     			LOGGER.debug("processTailsOf(Collection callees = " + callees + ", Stmt stmt = " + stmt
632     					+ ", SootMethod caller = " + caller + ", IClosure closure = " + closure + ", stack = "
633     					+ engine.getCopyOfCallStackCache() + ") - BEGIN");
634     		}
635     
636     		final BasicBlockGraphMgr _bbgMgr = engine.getBasicBlockGraphManager();
637     
638     		for (final Iterator<SootMethod> _i = callees.iterator(); _i.hasNext();) {
639     			final SootMethod _callee = _i.next();
640     
641     			if (considerMethodExitForCriteriaGeneration(_callee, closure == returnValueInclClosure)) {
642     				engine.includeInSlice(_callee);
643     
644     				if (_callee.isConcrete()) {
645     					final BasicBlockGraph _calleeBasicBlockGraph = _bbgMgr.getBasicBlockGraph(_callee);
646     					final CallTriple _callTriple = new CallTriple(caller, stmt, stmt.getInvokeExpr());
647     					engine.enterMethod(_callTriple);
648     					processSuperInitInInit(_callee, _calleeBasicBlockGraph);
649     
650     					final Collection<Stmt> _temp = new HashSet<Stmt>();
651     
652     					for (final Iterator<BasicBlock> _j = _calleeBasicBlockGraph.getSinks().iterator(); _j.hasNext();) {
653     						final BasicBlock _bb = _j.next();
654     						final Stmt _trailer = _bb.getTrailerStmt();
655     						closure.execute(new Pair<Stmt, SootMethod>(_trailer, _callee));
656     						_temp.add(_trailer);
657     					}
658     					exitTransformedMethods2exitpoints.put(_callee, _temp);
659     					engine.returnFromMethod();
660     				} else {
661     					final InvokeExpr _invokeExpr = stmt.getInvokeExpr();
662     
663     					for (int _j = _callee.getParameterCount() - 1; _j >= 0; _j--) {
664     						engine.generateExprLevelSliceCriterion(_invokeExpr.getArgBox(_j), stmt, caller, true);
665     					}
666     				}
667     			}
668     			recordCallInfoForProcessingArgsTo(stmt, caller, _callee);
669     		}
670     		generateCriteriaForReceiverOfAt(stmt, caller);
671     
672     		if (LOGGER.isDebugEnabled()) {
673     			LOGGER.debug("processTailsOf() , stack = " + engine.getCopyOfCallStackCache() + "- END");
674     		}
675     	}
