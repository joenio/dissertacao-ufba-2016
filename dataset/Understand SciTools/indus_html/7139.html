20     
21     import edu.ksu.cis.indus.interfaces.IEscapeInfo;

23     import edu.ksu.cis.indus.staticanalyses.InitializationException;

25     import soot.SootMethod;
26     import soot.Value;
27     
28     import soot.jimple.ArrayRef;
29     import soot.jimple.AssignStmt;
30     import soot.jimple.InstanceFieldRef;
31     import soot.jimple.StaticFieldRef;
32     
33     /**
34      * This class uses escape-analysis information as calculated by {@link
35      * edu.ksu.cis.indus.staticanalyses.concurrency.escape.EquivalenceClassBasedEscapeAnalysis
36      * EquivalenceClassBasedEscapeAnalysis} to prune the interference dependence edges as calculated by it's parent class. This
37      * can be further spruced by symbolic-analysis.
38      * 
39      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
40      * @author $Author: rvprasad $
41      * @version $Revision: 1.32 $
42      * @see InterferenceDAv1
43      */
44     public class InterferenceDAv2
45     		extends InterferenceDAv1 {
46     
47     	/**
48     	 * This provide information shared access in the analyzed system. This is required by the analysis.
49     	 */
50     	protected IEscapeInfo ecba;
51     
52     	/**
53     	 * Creates an instance of this class.
54     	 */
55     	public InterferenceDAv2() {
56     		super();
57     	}
58     
59     	/**
60     	 * @see edu.ksu.cis.indus.staticanalyses.interfaces.AbstractAnalysis#analyze()
61     	 */
62     	@Override public void analyze() {
63     		unstable();
64     
65     		if (ecba.isStable()) {
66     			super.analyze();
67     		}
68     	}
69     
70     	/**
71     	 * @see InterferenceDAv1#isArrayDependentOn(Pair, Pair, ArrayRef, ArrayRef)
72     	 */
73     	@Override protected boolean isArrayDependentOn(final Pair<AssignStmt, SootMethod> dependent,
74     			final Pair<AssignStmt, SootMethod> dependee, final ArrayRef dependentArrayRef, final ArrayRef dependeeArrayRef) {
75     		boolean _result = super.isArrayDependentOn(dependent, dependee, dependentArrayRef, dependeeArrayRef);
76     
77     		if (_result) {
78     			final SootMethod _deMethod = dependee.getSecond();
79     			final SootMethod _dtMethod = dependent.getSecond();
80     			final Value _de = dependeeArrayRef.getBase();
81     			final Value _dt = dependentArrayRef.getBase();
82     			_result = ecba.escapes(_de, _deMethod) && ecba.escapes(_dt, _dtMethod);
83     		}
84     		return _result;
85     	}
86     
87     	/**
88     	 * @see InterferenceDAv1#isInstanceFieldDependentOn(Pair, Pair, InstanceFieldRef, InstanceFieldRef)
89     	 */
90     	@Override protected boolean isInstanceFieldDependentOn(final Pair<AssignStmt, SootMethod> dependent,
91     			final Pair<AssignStmt, SootMethod> dependee, final InstanceFieldRef dependentFieldRef,
92     			final InstanceFieldRef dependeeFieldRef) {
93     		boolean _result = super.isInstanceFieldDependentOn(dependent, dependee, dependentFieldRef, dependeeFieldRef);
94     
95     		if (_result) {
96     			final SootMethod _deMethod = dependee.getSecond();
97     			final SootMethod _dtMethod = dependent.getSecond();
98     			final Value _de = dependeeFieldRef.getBase();
99     			final Value _dt = dependentFieldRef.getBase();
100     			_result = ecba.escapes(_de, _deMethod) && ecba.escapes(_dt, _dtMethod);
101     		}
102     		return _result;
103     	}
104     
105     	/**
106     	 * @see InterferenceDAv1#isStaticFieldDependentOn(Pair, Pair, StaticFieldRef, StaticFieldRef)
107     	 */
108     	@Override protected boolean isStaticFieldDependentOn(final Pair<AssignStmt, SootMethod> dependent,
109     			final Pair<AssignStmt, SootMethod> dependee, final StaticFieldRef dependentFieldRef,
110     			final StaticFieldRef dependeeFieldRef) {
111     		final boolean _result = super.isStaticFieldDependentOn(dependent, dependee, dependentFieldRef, dependeeFieldRef);
112     
113     		if (_result) {
114     			ecba.escapes(dependeeFieldRef.getField().getDeclaringClass(), dependee.getSecond());
115     		}
116     		return _result;
117     	}
118     
119     	/**
120     	 * Extracts information provided by the environment via <code>info</code> parameter to {@link #initialize(java.util.Map)
121     	 * initialize}.
122     	 * 
123     	 * @throws InitializationException when and instance of pair managing service or interference analysis is not provided.
124     	 * @pre info.get(IEscapeInfo.ID) != null
125     	 * @see InterferenceDAv1#setup()
126     	 */
127     	@Override protected void setup() throws InitializationException {
128     		super.setup();
129     
130     		ecba = (IEscapeInfo) info.get(IEscapeInfo.ID);
131     
132     		if (pairMgr == null) {
133     			throw new InitializationException(IEscapeInfo.ID + " was not provided in info.");
134     		}
135     	}
136     }
