46     /**
47      * This class xmlizes call graphs.
48      *
49      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
50      * @author $Author: rvprasad $
51      * @version $Revision: 1.3 $ $Date: 2007/02/10 19:07:05 $
52      */
53     final class CallGraphXMLizer
54       extends AbstractXMLizer {
55     	/** 
56     	 * The logger used by instances of this class to log messages.
57     	 */
58     	private static final Logger LOGGER = LoggerFactory.getLogger(CallGraphXMLizer.class);
59     
60     	/**
61     	 * @see edu.ksu.cis.indus.xmlizer.IXMLizer#getFileName(String)
62     	 */
63     	public String getFileName(final String name) {
64     		return "callgraph_" + xmlizeString(name) + ".xml";
65     	}
66     
67     	/**
68     	 * Writes the call graph in XML.
69     	 *
70     	 * @param info is a map of id's to implementation that satisfies the interface associated with the id.
71     	 *
72     	 * @pre rootname != null and info != null
73     	 * @pre info.oclIsKindOf(Map(Object, Object))
74     	 * @pre info.get(ICallGraphInfo.ID) != null and info.get(ICallGraphInfo.ID).oclIsKindOf(ICallGraphInfo)
75     	 * @pre info.get(FILE_NAME_ID) != null and info.get(FILE_NAME_ID).oclIsKindOf(String)
76     	 */
77     	public void writeXML(final Map info) {
78     		final File _f = new File(getXmlOutputDir() + File.separator + getFileName((String) info.get(FILE_NAME_ID)));
79     		final FileWriter _writer;
80     
81     		try {
82     			_writer = new FileWriter(_f);
83     
84     			final XMLOutputter _xmlWriter = new CustomXMLOutputter(_writer);
85     			final ICallGraphInfo _cgi = (ICallGraphInfo) info.get(ICallGraphInfo.ID);
86     			_xmlWriter.declaration();
87     			_xmlWriter.startTag("callgraph");
88     
89     			// Control the order in which methods are processed. 
90     			final IProcessingFilter _filter = new XMLizingProcessingFilter();
91     			final Collection _temp = new HashSet();
92     
93     			for (final Iterator _i = _filter.filterMethods(_cgi.getReachableMethods()).iterator(); _i.hasNext();) {
94     				final SootMethod _method = (SootMethod) _i.next();
95     				_xmlWriter.startTag("method");
96     				_xmlWriter.attribute("id", getIdGenerator().getIdForMethod(_method));
97     				_temp.clear();
98     
99     				for (final Iterator _j = _cgi.getCallees(_method).iterator(); _j.hasNext();) {
100     					final CallTriple _ctrp = (CallTriple) _j.next();
101     					_temp.add(_ctrp.getMethod());
102     				}
103     
104     				for (final Iterator _j = _filter.filterMethods(_temp).iterator(); _j.hasNext();) {
105     					final SootMethod _callee = (SootMethod) _j.next();
106     					_xmlWriter.startTag("callee");
107     					_xmlWriter.attribute("calleeId", getIdGenerator().getIdForMethod(_callee));
108     					_xmlWriter.endTag();
109     				}
110     
111     				_temp.clear();
112     
113     				for (final Iterator _j = _cgi.getCallers(_method).iterator(); _j.hasNext();) {
114     					final CallTriple _ctrp = (CallTriple) _j.next();
115     					_temp.add(_ctrp.getMethod());
116     				}
117     
118     				for (final Iterator _j = _filter.filterMethods(_temp).iterator(); _j.hasNext();) {
119     					final SootMethod _caller = (SootMethod) _j.next();
120     					_xmlWriter.startTag("caller");
121     					_xmlWriter.attribute("callerId", getIdGenerator().getIdForMethod(_caller));
122     					_xmlWriter.endTag();
123     				}
124     
125     				_xmlWriter.endTag();
126     			}
127     			_xmlWriter.startTag("reachables");
128     
129     			for (final Iterator _i = _cgi.getReachableMethods().iterator(); _i.hasNext();) {
130     				final SootMethod _sm = (SootMethod) _i.next();
131     				_xmlWriter.startTag("method");
132     				_xmlWriter.attribute("methodId", getIdGenerator().getIdForMethod(_sm));
133     				_xmlWriter.endTag();
134     			}
135     			_xmlWriter.endDocument();
136     			_writer.flush();
137     			_writer.close();
138     		} catch (final IOException _e) {
139     			LOGGER.error("Error while xmlizing call graph", _e);
140     		}
141     	}
142     }
