17     package edu.ksu.cis.indus.staticanalyses.interfaces;
18     
19     import edu.ksu.cis.indus.interfaces.IActivePart;
20     import edu.ksu.cis.indus.processing.Context;
21     
22     import java.util.Collection;
23     
24     import soot.SootMethod;
25     import soot.Value;
26     import soot.jimple.InvokeExpr;
27     
28     /**
29      * This is the interface to be provided by an analysis that operates on values (which may be symbolic). The analysis that
30      * implement this interface are behavioral analysis rather than structural analysis.
31      *
32      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
33      * @author $Author: rvprasad $
34      * @version $Revision: 1.21 $
35      * @param <V> is the type of the value object.
36      */
37     public interface IValueAnalyzer<V>
38     		extends IAnalyzer {
39     
40     	/**
41     	 * The id of this interface.
42     	 */
43     	Comparable<? extends Object> ID = "value flow analyzer";
44     
45     	/**
46     	 * The id of the tag used by the underlying flow analysis.
47     	 */
48     	String TAG_ID = "id of tag used by flow analysis";
49     
50     	/**
51     	 * Returns the active part of this object.
52     	 *
53     	 * @return the active part.
54     	 */
55     	public IActivePart getActivePart();
56     
57     	/**
58     	 * Returns the values associated with thrown exceptions for the given invocation expression and <code>this.context</code>.
59     	 *
60     	 * @param e is the method invoke expression.
61     	 * @param context in which the returned values will be associatd with the invocation site.
62     	 * @return the collection of values of type of the given exception class.
63     	 * @pre e != null and context != null
64     	 * @post result != null
65     	 */
66     	Collection<V> getThrownValues(InvokeExpr e, Context context);
67     
68     	/**
69     	 * Returns the values associated with exceptions thrown by the given method in the given context..
70     	 *
71     	 * @param method of interest
72     	 * @param context in which the returned values will be associatd with the method.
73     	 * @return the collection of values of type of the given exception class.
74     	 * @pre method != null and context != null
75     	 * @post result != null
76     	 */
77     	Collection<V> getThrownValues(SootMethod method, Context context);
78     
79     	/**
80     	 * Retrieves the values associated with the given value expression in the given context.
81     	 *
82     	 * @param value expression for which values are requested.
83     	 * @param context in which the returned values will be associated with the entity.
84     	 * @return the collection of values.
85     	 * @pre context != null
86     	 * @pre value != null
87     	 * @post result != null
88     	 */
89     	Collection<V> getValues(Value value, Context context);
90     
91     	/**
92     	 * Retrieves the values associated with the given parameter position in the given context.
93     	 *
94     	 * @param paramIndex is the position of the parameter.
95     	 * @param context in which value is requested.
96     	 * @return the collection of values
97     	 * @pre context != null and 0 &lt;= paramIndex &lt context.getCurrentMethod().getParameterCount()
98     	 * @pre context.getCurrentMethod() != null
99     	 * @post result != null
100     	 */
101     	Collection<V> getValuesForParameter(int paramIndex, Context context);
102     
103     	/**
104     	 * Retrieves the values associated with <code>this</code> variable in the given context.
105     	 *
106     	 * @param context in which the returned values will be associatd with <code>this</code> variable.
107     	 * @return the collection of values
108     	 * @pre context != null
109     	 * @pre context.getCurrentMethod() != null
110     	 * @post result != null
111     	 */
112     	Collection<V> getValuesForThis(Context context);
113     }
