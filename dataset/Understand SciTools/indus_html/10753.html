49     	/**
50     	 * This is the label of the data dependence arch across method boundaries.
51     	 */
52     	private static final Object INTER_PROCEDURAL_DATA_DEPENDENCE = "Inter-procedural data dependence";

54     	/**
55     	 * The collection of classes that define the scope in which dependence nodes have both incoming and outgoing edges.
56     	 */
57     	private final Collection<SootClass> coreClasses;
58     
59     	/**
60     	 * This is collection of dependence analyses from which dependences need to be captured.
61     	 */
62     	private final Collection<IDependencyAnalysis> deps;
63     
64     	/**
65     	 * This is the call graph. If this is <code>null</code> then dependences such as control dependence that are based on
66     	 * call-site will not be captured.

68     	private final ICallGraphInfo callgraph;

78     	private final SimpleEdgeGraphBuilder<Pair<Stmt, SootMethod>> segb;

80     	/**
81     	 * Creates an instance of this class.
82     	 *
83     	 * @param dependences is collection of dependence analyses from which dependences need to be captured.
84     	 * @param cgi is the call graph. If this is <code>null</code> then dependences such as control dependence that are based
85     	 *            on call-site will not be captured.
86     	 * @param classes to be considered while building the graph.
87     	 * @pre dependences != null and cgi != null and classes != null
88     	 */
89     	private SystemDependenceGraphBuilder(final Collection<IDependencyAnalysis> dependences,
90     			final ICallGraphInfo cgi, final Collection<SootClass> classes) {
91     		pairMgr = new PairManager(true, true);
92     		segb = new SimpleEdgeGraphBuilder<Pair<Stmt, SootMethod>>();
93     		deps = dependences;
94     		callgraph = cgi;
95     		coreClasses = classes;
96     	}
97     
98     	/**
99     	 * Creates a system dependence graph.
100     	 *
101     	 * @param dependences is collection of dependence analyses from which dependences need to be captured.
102     	 * @param cgi is the call graph. If this is <code>null</code> then dependences such as control dependence that are based
103     	 *            on call-site will not be captured.
104     	 * @param classes is the collection of classes that define the scope in which dependence nodes have both incoming and
105     	 *            outgoing edges.
106     	 * @return a dependence graph
107     	 * @pre dependences != null
108     	 * @pre classes != null
109     	 * @post result != null
110     	 * @post result.getNodes()->forall(o | o.getIncomingEdgeLabels()->forall(p | dependences->exists(q |
111     	 *       q.getIds().contains(p)) or p.equals(INTER_PROCEDURAL_DATA_DEPENDENCE)))
112     	 */
113     	public static SimpleEdgeGraph<Pair<Stmt, SootMethod>> getSystemDependenceGraph(
114     			final Collection<IDependencyAnalysis> dependences, final ICallGraphInfo cgi,
115     			final Collection<SootClass> classes) {
116     		final SystemDependenceGraphBuilder _builder = new SystemDependenceGraphBuilder(dependences, cgi, classes);
117     		return _builder.createGraph();
118     	}
119     
120     	/**
121     	 * Adds edges to the given graph builder that capture the dependences edges available from the provided analyses onto the
122     	 * given statment in the given method.
123     	 *
124     	 * @param stmt that serves as the destination.
125     	 * @param method that contains <code>stmt</code>.
126     	 * @pre stmt != null and method != null
127     	 */
128     	private void addEdgesFor(final Stmt stmt, final SootMethod method) {
129     		final Pair<Stmt, SootMethod> _dest = pairMgr.getPair(stmt, method);
130     		final Collection<Pair<Stmt, SootMethod>> _sources = new ArrayList<Pair<Stmt, SootMethod>>();
131     		final Iterator<IDependencyAnalysis> _i = deps.iterator();
132     		final int _iEnd = deps.size();
133     
134     		for (int _iIndex = 0; _iIndex < _iEnd; _iIndex++) {
135     			final IDependencyAnalysis _da = _i.next();
136     			final Collection<? extends Comparable<?>> _ids = _da.getIds();
137     			final Collection<?> _dees;
138     			
139     			// This block is a HACK!!!
140     			if (_da instanceof IdentifierBasedDataDA) {
141     				_dees = ((IdentifierBasedDataDA) _da).getDependees(stmt, method);
142     			} else if (_da instanceof IdentifierBasedDataDAv2) {
143     				_dees = ((IdentifierBasedDataDAv2) _da).getDependees(stmt, method);
144     			} else { 
145     				_dees = _da.getDependees(stmt, method);
146     			}
147     			
148     			final Iterator<?> _j = _dees.iterator();
149     			final int _jEnd = _dees.size();
150     			_sources.clear();
151     
152     			for (int _jIndex = 0; _jIndex < _jEnd; _jIndex++) {
153     				final Object _o = _j.next();
154     
155     				if (_o instanceof Pair) {
156     					_sources.add((Pair) _o);
157     				} else {
158     					_sources.add(pairMgr.getPair((Stmt) _o, method));
159     				}
160     			}
161     
162     			if (_ids.contains(IDependencyAnalysis.DependenceSort.SYNCHRONIZATION_DA)) {
163     				segb.addEdgeFromTo(_sources, IDependencyAnalysis.DependenceSort.SYNCHRONIZATION_DA, _dest);
164     			}
165     
166     			if (_ids.contains(IDependencyAnalysis.DependenceSort.READY_DA)) {
167     				segb.addEdgeFromTo(_sources, IDependencyAnalysis.DependenceSort.READY_DA, _dest);
168     			}
169     
170     			if (_ids.contains(IDependencyAnalysis.DependenceSort.INTERFERENCE_DA)) {
171     				segb.addEdgeFromTo(_sources, IDependencyAnalysis.DependenceSort.INTERFERENCE_DA, _dest);
172     			}
173     
174     			if (_ids.contains(IDependencyAnalysis.DependenceSort.REFERENCE_BASED_DATA_DA)) {
175     				segb.addEdgeFromTo(_sources, IDependencyAnalysis.DependenceSort.REFERENCE_BASED_DATA_DA, _dest);
176     			}
177     
178     			if (_ids.contains(IDependencyAnalysis.DependenceSort.CONTROL_DA)) {
179     				segb.addEdgeFromTo(_sources, IDependencyAnalysis.DependenceSort.CONTROL_DA, _dest);
180     			}
181     
182     			if (_ids.contains(IDependencyAnalysis.DependenceSort.IDENTIFIER_BASED_DATA_DA)) {
183     				segb.addEdgeFromTo(_sources, IDependencyAnalysis.DependenceSort.IDENTIFIER_BASED_DATA_DA, _dest);
184     			}
185     		}
186     		
187     		if (callgraph != null) {
188     			final Collection<CallTriple> _callers = callgraph.getCallers(method);
189     			final Iterator<CallTriple> _k = _callers.iterator();
190     			final int _kEnd = _callers.size();
191     
192     			for (int _kIndex = 0; _kIndex < _kEnd; _kIndex++) {
193     				final CallTriple _ctrp = _k.next();
194     				_sources.add(pairMgr.getPair(_ctrp.getStmt(), _ctrp.getMethod()));
195     			}
196     		}
197     
198     		processForInterProceduralEdges(_dest);
199     	}

201     	/**
202     	 * Creates the system dependence graph.
203     	 *
204     	 * @return the SDG
205     	 * @post result != null
206     	 */
207     	private SimpleEdgeGraph<Pair<Stmt, SootMethod>> createGraph() {
208     		segb.createGraph();
209     
210     		final Iterator<SootClass> _j = coreClasses.iterator();
211     		final int _jEnd = coreClasses.size();
212     
213     		for (int _jIndex = 0; _jIndex < _jEnd; _jIndex++) {
214     			final SootClass _sc = _j.next();
215     			final Collection<SootMethod> _methods = _sc.getMethods();
216     			final Iterator<SootMethod> _k = _methods.iterator();
217     			final int _kEnd = _methods.size();
218     
219     			for (int _kIndex = 0; _kIndex < _kEnd; _kIndex++) {
220     				final SootMethod _sm = _k.next();
221     
222     				if (_sm.hasActiveBody()) {
223     					final Iterator<Stmt> _l = _sm.getActiveBody().getUnits().iterator();
224     					final int _lEnd = _sm.getActiveBody().getUnits().size();
225     
226     					for (int _lIndex = 0; _lIndex < _lEnd; _lIndex++) {
227     						final Stmt _stmt = _l.next();
228     						addEdgesFor(_stmt, _sm);
229     					}
230     				}
231     			}
232     		}
233     
234     		segb.finishBuilding();
235     		return segb.getBuiltGraph();
236     	}

238     	/**
239     	 * Adds data dependence edges across procedure boundaries.
240     	 *
241     	 * @param node that needs to be processed.
242     	 * @pre node != null
243     	 */
244     	private void processForInterProceduralEdges(final Pair<Stmt, SootMethod> node) {
245     		final Stmt _stmt = node.getFirst();
246     		final SootMethod _sm = node.getSecond();
247     
248     		if (_stmt instanceof ReturnStmt) {
249     			final Iterator<CallTriple> _i = callgraph.getCallers(_sm).iterator();
250     			final int _iEnd = callgraph.getCallers(_sm).size();
251     
252     			for (int _iIndex = 0; _iIndex < _iEnd; _iIndex++) {
253     				final CallTriple _ctrp = _i.next();
254     				final Stmt _dest = _ctrp.getStmt();
255     
256     				if (_dest instanceof AssignStmt) {
257     					segb.addEdgeFromTo(node, INTER_PROCEDURAL_DATA_DEPENDENCE, pairMgr.getPair(_dest, _ctrp.getMethod()));
258     				}
259     			}
260     		} else if (_stmt instanceof IdentityStmt) {
261     			final IdentityStmt _s = (IdentityStmt) _stmt;
262     
263     			if (_s.getRightOp() instanceof ParameterRef) {
264     				final Iterator<CallTriple> _i = callgraph.getCallers(_sm).iterator();
265     				final int _iEnd = callgraph.getCallers(_sm).size();
266     
267     				for (int _iIndex = 0; _iIndex < _iEnd; _iIndex++) {
268     					final CallTriple _ctrp = _i.next();
269     					segb.addEdgeFromTo(pairMgr.getPair(_ctrp.getStmt(), _ctrp.getMethod()), INTER_PROCEDURAL_DATA_DEPENDENCE,
270     							node);
271     				}
272     			}
273     		}
274     	}
