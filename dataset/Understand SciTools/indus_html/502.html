44     /**
45      * This utility class can be used to xmlize jimple.

51     public final class JimpleXMLizerCLI {

53     	/**
54     	 * The logger used by instances of this class to log messages.

56     	private static final Logger LOGGER = LoggerFactory.getLogger(JimpleXMLizerCLI.class);

69     	/**
70     	 * The entry point to execute this xmlizer from command prompt.
71     	 * 
72     	 * @param s is the command-line arguments.
73     	 * @throws RuntimeException when jimple xmlization fails.
74     	 * @pre s != null
75     	 */
76     	public static void main(final String[] s) {
77     		final Scene _scene = Scene.v();
78     
79     		final Options _options = new Options();
80     		Option _o = new Option("d", "dump directory", true, "The directory in which to write the xml files.  "
81     				+ "If unspecified, the xml output will be directed standard out.");
82     		_o.setArgs(1);
83     		_o.setArgName("path");
84     		_o.setOptionalArg(false);
85     		_options.addOption(_o);
86     		_o = new Option("h", "help", false, "Display message.");
87     		_options.addOption(_o);
88     		_o = new Option("p", "soot-classpath", true, "Prepend this to soot class path.");
89     		_o.setArgs(1);
90     		_o.setArgName("classpath");
91     		_o.setOptionalArg(false);
92     		_options.addOption(_o);
93     
94     		final HelpFormatter _help = new HelpFormatter();
95     
96     		try {
97     			final CommandLine _cl = (new BasicParser()).parse(_options, s);
98     			final String[] _args = _cl.getArgs();
99     
100     			if (_cl.hasOption('h')) {
101     				final String _cmdLineSyn = "java " + JimpleXMLizerCLI.class.getName() + "<options> <class names>";
102     				_help.printHelp(_cmdLineSyn.length(), _cmdLineSyn, "", _options, "", true);
103     			} else {
104     				if (_args.length > 0) {
105     					if (_cl.hasOption('p')) {
106     						_scene.setSootClassPath(_cl.getOptionValue('p') + File.pathSeparator + _scene.getSootClassPath());
107     					}
108     
109     					final NamedTag _tag = new NamedTag("JimpleXMLizer");
110     
111     					for (int _i = 0; _i < _args.length; _i++) {
112     						final SootClass _sc = _scene.loadClassAndSupport(_args[_i]);
113     						_sc.addTag(_tag);
114     					}
115     					final IProcessingFilter _filter = new TagBasedProcessingFilter(_tag.getName());
116     					writeJimpleAsXML(new Environment(_scene), _cl.getOptionValue('d'), null, new UniqueJimpleIDGenerator(),
117     							_filter);
118     				} else {
119     					System.out.println("No classes were specified.");
120     				}
121     			}
122     		} catch (final ParseException _e) {
123     			LOGGER.error("Error while parsing command line.", _e);
124     			printUsage(_options);
125     		} catch (final Throwable _e) {
126     			LOGGER.error("Beyond our control. May day! May day!", _e);
127     			throw new RuntimeException(_e);
128     		}
129     	}
130     
131     	/**
132     	 * Writes the jimple in the scene via the writer.
133     	 * 
134     	 * @param env in which the jimple to be dumped resides.
135     	 * @param directory with which jimple is dumped. If <code>null</code>, the output will be redirected to standarad
136     	 *            output.
137     	 * @param suffix to be appended each file name. If <code>null</code>, no suffix is appended.
138     	 * @param jimpleIDGenerator is the id generator to be used during xmlization.
139     	 * @param processingFilter to be used to control the parts of the system that should be jimplified.
140     	 * @pre scene != null and jimpleIDGenerator != null
141     	 */
142     	public static void writeJimpleAsXML(final IEnvironment env, final String directory, final String suffix,
143     			final IJimpleIDGenerator jimpleIDGenerator, final IProcessingFilter processingFilter) {
144     		final JimpleXMLizer _xmlizer = new JimpleXMLizer(jimpleIDGenerator);
145     		final ProcessingController _pc = new ProcessingController();
146     		final IStmtGraphFactory<?> _sgf = new CompleteStmtGraphFactory();
147     		final OneAllStmtSequenceRetriever _ssr = new OneAllStmtSequenceRetriever();
148     		_ssr.setStmtGraphFactory(_sgf);
149     		_pc.setStmtSequencesRetriever(_ssr);
150     		_pc.setEnvironment(env);
151     
152     		final XMLizingProcessingFilter _xmlFilter = new XMLizingProcessingFilter();
153     
154     		if (processingFilter != null) {
155     			_xmlFilter.chain(processingFilter);
156     		}
157     		_pc.setProcessingFilter(_xmlFilter);
158     		_xmlizer.setDumpOptions(directory, suffix);
159     		_xmlizer.hookup(_pc);
160     		_pc.process();
161     		_xmlizer.unhook(_pc);
162     	}
163     
164     	/**
165     	 * Prints the help/usage info for this class.
166     	 * 
167     	 * @param options is the command line option.
168     	 * @pre options != null
169     	 */
170     	private static void printUsage(final Options options) {
171     		final String _cmdLineSyn = "java " + JimpleXMLizerCLI.class.getName() + " <options> <classnames>";
172     		(new HelpFormatter()).printHelp(_cmdLineSyn, "Options are: ", options, "");
173     	}
174     }
