113     			_r = 0;
114     			for (final T1 _t : col) {
115     				if (_t == obj) {

117     				}
118     			}

165     	 * Checks if the given collections have a common element.
166     	 * 
167     	 * @param <T1> is the type of the objects in the first collection.
168     	 * @param <T2> is the type of the objects in the second collection.
169     	 * @param col1 is one of the collection of interest.
170     	 * @param col2 is another collection of interest.
171     	 * @return <code>true</code> if <code>col1</code> and <code>col2</code> have a common element; <code>false</code>,
172     	 *         otherwise.
173     	 * @post col1->exists(o | col2->contains(o))
174     	 */
175     	@Functional public static <T1, T2> boolean containsAny(@NonNull final Collection<T1> col1,
176     			@NonNull final Collection<T2> col2) {
177     		final Collection<?> _minCol;

179     		if (col1.size() > col2.size()) {
180     			_minCol = col2;
181     			_maxCol = col1;
182     		} else {
183     			_minCol = col1;
184     			_maxCol = col2;
185     		}
186     		for (final Object _t2 : _minCol) {
187     			if (_maxCol.contains(_t2)) {
188     				return true;
189     			}
190     		}
191     		return false;
192     	}
