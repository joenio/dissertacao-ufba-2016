44     /**

51     public class SootValueTypeManager
52     		extends Observable
53     		implements ITypeManager<Type, Value> {
54     
55     	/**
56     	 * This is a dummy implementation of <code>IType</code>.
57     	 * 
58     	 * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>

61     	 */
62     	static class DummyType
63     			implements IType {
64     
65     		/**
66     		 * The soot type represented by this instance.
67     		 */
68     		final Type sootType;
69     
70     		/**
71     		 * Creates an instance of this class.
72     		 * 
73     		 * @param sType is the represented Soot type.
74     		 * @pre sType != null
75     		 */
76     		public DummyType(final Type sType) {
77     			sootType = sType;
78     		}
79     	}
80     
81     	/**
82     	 * This is the dynamic token-type relation evaluator.
83     	 */
84     	private final IDynamicTokenTypeRelationDetector<Value> evaluator;
85     
86     	/**
87     	 * This maps soot types to user's type.
88     	 */
89     	private final Map<Type, IType> sootType2Type;
90     
91     	/**
92     	 * The type corresponding to the soot type representing <code>java.lang.Object</code>.
93     	 */
94     	private IType tokenTypeForObjectType;
95     
96     	/**
97     	 * This predicate is used to detect types that can hold null constant.
98     	 */
99     	private final IPredicate<Type> typesForNullConstPredicate;
100     
101     	/**
102     	 * Creates an instance of this class.
103     	 */
104     	public SootValueTypeManager() {
105     		super();
106     		typesForNullConstPredicate = new InstanceOfPredicate<RefLikeType, Type>(RefLikeType.class);
107     		evaluator = new SootDynamicTokenTypeRelationDetector();
108     		sootType2Type = new HashMap<Type, IType>(Constants.getNumOfClassesInApplication());
109     	}
110     
111     	/**
112     	 * {@inheritDoc}
113     	 */
114     	public Collection<IType> getAllTypes(final Value value) {
115     		final Value _theValue = value;
116     		final Type _type = _theValue.getType();
117     		final Collection<IType> _result = new ArrayList<IType>();
118     
119     		_result.add(getTokenTypeForRepType(_type));
120     
121     		if (_type instanceof RefType) {
122     			for (final Iterator _i = Util.getAncestors(((RefType) _type).getSootClass()).iterator(); _i.hasNext();) {
123     				_result.add(getTokenTypeForRepType(((SootClass) _i.next()).getType()));
124     			}
125     		}
126     
127     		if (_type instanceof NullType) {
128     			final Collection<Type> _s = CollectionUtils.collect(sootType2Type.keySet(), typesForNullConstPredicate);
129     
130     			for (final Iterator<Type> _i = _s.iterator(); _i.hasNext();) {
131     				final Type _t = _i.next();
132     				_result.add(sootType2Type.get(_t));
133     			}
134     		} else if (_type instanceof ArrayType) {
135     			final IType _t = getTokenTypeForObjectType();
136     
137     			if (_t != null) {
138     				_result.add(_t);
139     			}
140     		}
141     
142     		return _result;
143     	}
144     
145     	/**
146     	 * @see edu.ksu.cis.indus.staticanalyses.tokens.ITypeManager#getDynamicTokenTypeRelationEvaluator()
147     	 */
148     	public IDynamicTokenTypeRelationDetector<Value> getDynamicTokenTypeRelationEvaluator() {
149     		return evaluator;
150     	}
151     
152     	/**
153     	 * {@inheritDoc}
154     	 */
155     	public IType getExactType(final Value value) {
156     		return getTokenTypeForRepType(value.getType());
157     	}
158     
159     	/**
160     	 * @see edu.ksu.cis.indus.staticanalyses.tokens.ITypeManager#getTokenTypeForRepType(Object)
161     	 */
162     	public IType getTokenTypeForRepType(final Type sootType) {
163     		IType _result = sootType2Type.get(sootType);
164     
165     		if (_result == null) {
166     			_result = new DummyType(sootType);
167     			sootType2Type.put(sootType, _result);
168     			setChanged();
169     			notifyObservers(new NewTypeCreated(_result));
170     		}
171     		return _result;
172     	}
173     
174     	/**
175     	 * Resets the manager.
176     	 */
177     	public void reset() {
178     		sootType2Type.clear();
179     		evaluator.reset();
180     		tokenTypeForObjectType = null;
181     	}

183     	/**
184     	 * Retrieves the type for the soot type representing <code>java.lang.Object</code>.
185     	 * 
186     	 * @return the type
187     	 */
188     	private IType getTokenTypeForObjectType() {
189     		if (tokenTypeForObjectType == null) {
190     			final Set<Map.Entry<Type, IType>> _entrySet = sootType2Type.entrySet();
191     			final Iterator<Map.Entry<Type, IType>> _i = _entrySet.iterator();
192     			final int _iEnd = _entrySet.size();
193     
194     			for (int _iIndex = 0; _iIndex < _iEnd; _iIndex++) {
195     				final Map.Entry<Type, IType> _entry = _i.next();
196     				final Type _sootType = _entry.getKey();
197     
198     				if (_sootType instanceof RefType) {
199     					final SootClass _sc = ((RefType) _sootType).getSootClass();
200     
201     					if (_sc.getName().equals("java.lang.Object")) {
202     						tokenTypeForObjectType = _entry.getValue();
203     						break;
204     					}
205     				}
206     			}
207     		}
208     		return tokenTypeForObjectType;
209     	}
210     }
