17     package edu.ksu.cis.indus.common.scoping;
18     
19     import edu.ksu.cis.indus.annotations.Functional;
20     import edu.ksu.cis.indus.annotations.Immutable;
21     import edu.ksu.cis.indus.annotations.NonNull;
22     
23     /**
24      * This is the interface to a scope specification.
25      * 
26      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
27      * @author $Author$
28      * @version $Revision$ $Date$
29      */
30     interface ISpecification {
31     
32     	/**
33     	 * Retrieves the value in <code>name</code>.
34     	 * 
35     	 * @return the value in <code>name</code>.
36     	 */
37     	@Functional String getName();
38     
39     	/**
40     	 * Sets the access specification. If not set, default access specification is attached to this specification.
41     	 * 
42     	 * @param accessSpecification for this specification.
43     	 */
44     	void setAccessSpec(@NonNull @Immutable final AccessSpecification accessSpecification);
45     
46     	/**
47     	 * Sets the value of <code>inclusion</code>.
48     	 * 
49     	 * @param value the new value of <code>inclusion</code>.
50     	 */
51     	void setInclusion(final boolean value);
52     
53     	/**
54     	 * Sets the value of <code>name</code>.
55     	 * 
56     	 * @param nameOfTheSpec the new value of <code>name</code>.
57     	 */
58     	void setName(@NonNull @Immutable final String nameOfTheSpec);
59     
60     }
