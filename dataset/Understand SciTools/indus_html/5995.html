40     /**
41      * This class is provides the control class for the analyses suite. The analyses progress in phases. It may be so that some
42      * application require a particular sequence in which each analysis should progress. Hence, the applications provide an
43      * implementation of controller interface to drive the analyses in a particular sequence of phases.
44      * <p>
45      * This implementation will drive the given analyses such that each analysei is executed only when the anlaysis indicates that
46      * all it's prerequesites have been fulfilled.
47      * </p>
48      *
49      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
50      * @author $Author: rvprasad $
51      * @version $Revision: 1.9 $
52      */
53     public class AnalysesController {
54     
55     	/**
56     	 * The logger used by instances of this class to log messages.
57     	 */
58     	private static final Logger LOGGER = LoggerFactory.getLogger(AnalysesController.class);
59     
60     	/**
61     	 * The map of analysis being controlled by this object. It maps names of analysis to the analysis object.
62     	 *
63     	 * @invariant participatingAnalyses != null
64     	 */
65     	protected final Map<Comparable, Collection<? extends IAnalysis>> participatingAnalyses;
66     
67     	/**
68     	 * This is the preprocessing controlling agent.
69     	 *
70     	 * @invariant preprocessController != null;
71     	 */
72     	protected final ProcessingController preprocessController;
73     
74     	/**
75     	 * The object used to realize the "active" part of this object.
76     	 */
77     	private final IActivePart.ActivePart activePart = new IActivePart.ActivePart();
78     
79     	/**
80     	 * This provides basic block graphs for the analyses.
81     	 */
82     	private BasicBlockGraphMgr basicBlockGraphMgr;
83     
84     	/**
85     	 * This is a map of name to objects which provide information that maybe used by analyses, but is of no use to the
86     	 * controller.
87     	 */
88     	private Map<Comparable<?>, Object> info;
89     
90     	/**
91     	 * Creates a new AbstractAnalysesController object.
92     	 *
93     	 * @param infoPrm is a map of name to objects which provide information that maybe used by analyses, but is of no use to
94     	 *            the controller.
95     	 * @param pc is the preprocessing controller.
96     	 * @param bbgMgr provides basic blocks graphs for methods. If this is non-null then the analyses are initialized with this
97     	 *            graph manager. If not, the graph managers of the analyses will not be initialized. Hence, it should be done
98     	 *            by the application.
99     	 * @pre pc != null
100     	 */
101     	public AnalysesController(final Map<Comparable<?>, Object> infoPrm, final ProcessingController pc,
102     			final BasicBlockGraphMgr bbgMgr) {
103     		participatingAnalyses = new HashMap<Comparable, Collection<? extends IAnalysis>>();
104     		info = infoPrm;
105     		preprocessController = pc;
106     		basicBlockGraphMgr = bbgMgr;
107     	}
108     
109     	/**
110     	 * Adds the implementations to be used for analysis.
111     	 *
112     	 * @param id of the analysis.
113     	 * @param analyses are the implementations of the named analysis.
114     	 * @pre id != null and analyses != null and analysis->forall(o | o != null)
115     	 */
116     	public final void addAnalyses(final Comparable<?> id, final Collection<? extends IAnalysis> analyses) {
117     		MapUtils.putAllIntoCollectionInMap(participatingAnalyses, id, analyses);
118     	}
119     
120     	/**
121     	 * Executes the analyses in the registered order.
122     	 */
123     	public void execute() {
124     		boolean _analyzing;
125     		final Collection<IAnalysis> _done = new ArrayList<IAnalysis>();
126     
127     		do {
128     			_analyzing = false;
129     
130     			for (final Iterator<Comparable> _i = participatingAnalyses.keySet().iterator(); _i.hasNext()
131     					&& activePart.canProceed();) {
132     				final Comparable<?> _daName = _i.next();
133     				final Collection<? extends IAnalysis> _c = participatingAnalyses.get(_daName);
134     
135     				for (final Iterator<? extends IAnalysis> _j = _c.iterator(); _j.hasNext();) {
136     					final IAnalysis _analysis = _j.next();
137     
138     					if (_analysis != null && !_done.contains(_analysis)) {
139     						_analysis.analyze();
140     
141     						final boolean _t = _analysis.isStable();
142     
143     						if (_t) {
144     							_done.add(_analysis);
145     						}
146     						_analyzing |= _t;
147     					}
148     				}
149     			}
150     		} while (_analyzing);
151     	}
152     
153     	/**
154     	 * Returns the active part of this object.
155     	 *
156     	 * @return the active part.
157     	 */
158     	public IActivePart getActivePart() {
159     		return activePart;
160     	}
161     
162     	/**
163     	 * Provides the implementation registered for the given analysis purpose.
164     	 *
165     	 * @param id of the requested analyses. This has to be one of the names(XXX_DA) defined in this class.
166     	 * @return the implementation registered for the given purpose. Changes to this collection is visible to the controller.
167     	 * @post result != null
168     	 */
169     	public final Collection<? extends IAnalysis> getAnalyses(final Comparable<?> id) {
170     		final Collection<? extends IAnalysis> _result;
171     
172     		if (participatingAnalyses != null) {
173     			_result = participatingAnalyses.get(id);
174     		} else {
175     			_result = null;
176     		}
177     		return _result;
178     	}
179     
180     	/**
181     	 * Initializes the controller. Analyses are initialized and then driven to preprocess the system (in that order only).
182     	 */
183     	public void initialize() {
184     		final Collection<Comparable<?>> _failed = new ArrayList<Comparable<?>>();
185     		final Collection<IProcessor> _preprocessors = new HashSet<IProcessor>();
186     
187     		for (final Iterator<Comparable> _k = participatingAnalyses.keySet().iterator(); _k.hasNext()
188     				&& activePart.canProceed();) {
189     			final Comparable<?> _key = _k.next();
190     			final Collection<? extends IAnalysis> _c = participatingAnalyses.get(_key);
191     
192     			for (final Iterator<? extends IAnalysis> _j = _c.iterator(); _j.hasNext() && activePart.canProceed();) {
193     				final IAnalysis _analysis = _j.next();
194     
195     				try {
196     					_analysis.initialize(info);
197     
198     					if (_analysis.doesPreProcessing()) {
199     						final IProcessor _p = _analysis.getPreProcessor();
200     						_p.hookup(preprocessController);
201     						_preprocessors.add(_p);
202     					}
203     
204     					if (basicBlockGraphMgr != null) {
205     						_analysis.setBasicBlockGraphManager(basicBlockGraphMgr);
206     					}
207     				} catch (final InitializationException _e) {
208     					if (LOGGER.isWarnEnabled()) {
209     						LOGGER.warn(_analysis.getClass() + " failed to initialize, hence, it will not executed.", _e);
210     					}
211     					_failed.add(_key);
212     
213     					if (_analysis.doesPreProcessing()) {
214     						_preprocessors.remove(_analysis.getPreProcessor());
215     					}
216     				}
217     			}
218     
219     			for (final Iterator<Comparable<?>> _i = _failed.iterator(); _i.hasNext();) {
220     				_c.remove(_i.next());
221     			}
222     		}
223     
224     		if (activePart.canProceed()) {
225     			preprocessController.process();
226     		}
227     
228     		for (final Iterator<IProcessor> _i = _preprocessors.iterator(); _i.hasNext();) {
229     			_i.next().unhook(preprocessController);
230     		}
231     	}
232     
233     	/**
234     	 * Resets the internal data structures of the controller. This resets the participating analyses. This does not reset the
235     	 * Object Flow Analysis instance.
236     	 */
237     	public void reset() {
238     		for (final Iterator<Collection<? extends IAnalysis>> _i = participatingAnalyses.values().iterator(); _i.hasNext();) {
239     			final Collection<? extends IAnalysis> _c = _i.next();
240     
241     			for (final Iterator<? extends IAnalysis> _j = _c.iterator(); _j.hasNext();) {
242     				final IAnalysis _analysis = _j.next();
243     				_analysis.reset();
244     			}
245     		}
246     		participatingAnalyses.clear();
247     		activePart.activate();
248     	}
249     }
