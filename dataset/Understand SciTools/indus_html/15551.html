17     package edu.ksu.cis.indus.tools.slicer.criteria.generators;
18     
19     import edu.ksu.cis.indus.annotations.NonNull;
20     import edu.ksu.cis.indus.annotations.NonNullContainer;
21     import edu.ksu.cis.indus.common.collections.IPredicate;
22     import edu.ksu.cis.indus.processing.Context;
23     import edu.ksu.cis.indus.slicer.ISliceCriterion;
24     import edu.ksu.cis.indus.tools.slicer.SlicerTool;
25     import edu.ksu.cis.indus.tools.slicer.contextualizers.ISliceCriteriaContextualizer;
26     import edu.ksu.cis.indus.tools.slicer.criteria.predicates.ISliceCriteriaPredicate;
27     
28     import java.util.Collection;
29     
30     import soot.SootMethod;
31     
32     /**
33      * This is an abstract implementation of <code>ISliceCriteriaGenerator</code>.
34      * 
35      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
36      * @author $Author: rvprasad $
37      * @version $Revision: 1.10 $ $Date: 2007/02/10 19:06:00 $
38      * @param <T1> is the type of objects that are selected in expectation of containing slice criteria.
39      * @param <T2> is the type of objects that will be considered for being used as slice criteria.
40      */
41     public abstract class AbstractSliceCriteriaGenerator<T1, T2>
42     		implements ISliceCriteriaGenerator<T1, T2> {
43     
44     	/**
45     	 * The contextualizer to use.
46     	 */
47     	private ISliceCriteriaContextualizer contextualizer;
48     
49     	/**
50     	 * The filter to use.
51     	 */
52     	private ISliceCriteriaPredicate<T2> criteriaPredicate;
53     
54     	/**
55     	 * The predicate to select sites in which the criteria may occur.
56     	 */
57     	private IPredicate<T1> siteSelectionPredicate;
58     
59     	/**
60     	 * The slicer that defines the context in which generator functions.
61     	 */
62     	private SlicerTool<?> slicerTool;
63     
64     	/**
65     	 * {@inheritDoc}
66     	 */
67     	public final Collection<ISliceCriterion> getCriteria(final SlicerTool<?> slicer) {
68     		slicerTool = slicer;
69     
70     		if (contextualizer != null) {
71     			contextualizer.setSlicerTool(slicerTool);
72     		}
73     
74     		if (criteriaPredicate != null) {
75     			criteriaPredicate.setSlicerTool(slicerTool);
76     		}
77     		return getCriteriaTemplateMethod();
78     	}
79     
80     	/**
81     	 * {@inheritDoc}
82     	 */
83     	public final void setCriteriaContextualizer(final ISliceCriteriaContextualizer theContextualizer) {
84     		contextualizer = theContextualizer;
85     	}
86     
87     	/**
88     	 * {@inheritDoc}
89     	 */
90     	public final void setCriteriaFilterPredicate(final ISliceCriteriaPredicate<T2> thePredicate) {
91     		criteriaPredicate = thePredicate;
92     	}
93     
94     	/**
95     	 * {@inheritDoc}
96     	 */
97     	public final void setSiteSelectionPredicate(final IPredicate<T1> thePredicate) {
98     		siteSelectionPredicate = thePredicate;
99     	}
100     
101     	/**
102     	 * Contextualizes the given criteria.
103     	 * 
104     	 * @param context in which <code>baseCriteria</code> were generated.
105     	 * @param baseCriteria is the collection of criteria to be contextualized. This collection will be modified upon return
106     	 *            with contextualized criteria.
107     	 */
108     	protected final void contextualizeCriteriaBasedOnProgramPoint(@NonNull final Context context,
109     			@NonNull @NonNullContainer final Collection<ISliceCriterion> baseCriteria) {
110     		if (contextualizer != null) {
111     			contextualizer.processCriteriaBasedOnProgramPoint(context, baseCriteria);
112     		}
113     	}
114     
115     	/**
116     	 * Contextualizes the given criteria.
117     	 * 
118     	 * @param method in which <code>baseCriteria</code> were generated.
119     	 * @param baseCriteria is the collection of criteria to be contextualized. This collection will be modified upon return
120     	 *            with contextualized criteria.
121     	 */
122     	protected final void contextualizeCriteriaBasedOnThis(@NonNull final SootMethod method,
123     			@NonNull @NonNullContainer final Collection<ISliceCriterion> baseCriteria) {
124     		if (contextualizer != null) {
125     			contextualizer.processCriteriaBasedOnThis(method, baseCriteria);
126     		}
127     	}
128     
129     	/**
130     	 * Checks if the given site should be considered for criteria generation.
131     	 * 
132     	 * @param site to be checked.
133     	 * @return <code>true</code> if site should be considered; <code>false</code>, otherwise.
134     	 */
135     	protected final boolean shouldConsiderSite(final T1 site) {
136     		final boolean _result;
137     
138     		if (siteSelectionPredicate != null) {
139     			_result = siteSelectionPredicate.evaluate(site);
140     		} else {
141     			_result = true;
142     		}
143     		return _result;
144     	}
145     
146     	/**
147     	 * This is a template method that the subclasses should implement to generate the criteria.
148     	 * 
149     	 * @return a collection of criteria.
150     	 */
151     	@NonNull @NonNullContainer protected abstract Collection<ISliceCriterion> getCriteriaTemplateMethod();
152     
153     	/**
154     	 * Retrieves the value in <code>slicerTool</code>.
155     	 * 
156     	 * @return the value in <code>slicerTool</code>.
157     	 */
158     	protected SlicerTool<?> getSlicerTool() {
159     		return slicerTool;
160     	}
161     
162     	/**
163     	 * Checks if criteria should be generated based on the given entity.
164     	 * 
165     	 * @param entity forms the base for the criteria.
166     	 * @return <code>true</code>
167     	 */
168     	protected final boolean shouldGenerateCriteriaFrom(@NonNull final T2 entity) {
169     		final boolean _result;
170     
171     		if (criteriaPredicate != null) {
172     			_result = criteriaPredicate.evaluate(entity);
173     		} else {
174     			_result = true;
175     		}
176     		return _result;
177     	}
178     }
