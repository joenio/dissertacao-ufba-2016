18     package edu.ksu.cis.indus;
19     
20     import edu.ksu.cis.indus.common.soot.IStmtGraphFactory;
21     
22     import edu.ksu.cis.indus.xmlizer.IJimpleIDGenerator;
23     
24     import java.util.Collection;
25     import java.util.Iterator;
26     
27     import junit.extensions.TestSetup;
28     
29     import junit.framework.TestSuite;
30     
31     
32     /**
33      * This class is a test setup class that is used in conjunction with xml data-based testing.
34      *
35      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
36      * @author $Author: rvprasad $
37      * @version $Revision: 1.18 $ $Date: 2007/02/10 19:08:51 $
38      */
39     public class AbstractXMLBasedTestSetup
40       extends TestSetup
41       implements IXMLBasedTest {
42     	/** 
43     	 * ID generator used while xmlizing documents which will often be the case (xmlize and test the xmlized data).
44     	 */
45     	protected IJimpleIDGenerator idGenerator;
46     
47     	/** 
48     	 * This is the location where the jimple xml should be dumpled.
49     	 */
50     	protected String dumpLocation;
51     
52     	/** 
53     	 * The statement graph (CFG) factory used during testing.
54     	 */
55     	private IStmtGraphFactory stmtGraphFactory;
56     
57     	/** 
58     	 * The directory in which one of the xml-based testing input is read from.
59     	 */
60     	private String xmlFirstInputDir;
61     
62     	/** 
63     	 * The directory in which the other xml-based testing input is read from.
64     	 */
65     	private String xmlSecondInputDir;
66     
67     	/**
68     	 * Creates an instance of this class.
69     	 *
70     	 * @param test to be enclosed by this setup.
71     	 *
72     	 * @pre test != null
73     	 *
74     	 * @see TestSetup#TestSetup(Test)
75     	 */
76     	public AbstractXMLBasedTestSetup(final TestSuite test) {
77     		super(test);
78     	}
79     
80     	/**
81     	 * Sets the location where jimple xml should be dumped.
82     	 *
83     	 * @param location to dump jimple xml.
84     	 *
85     	 * @pre location != null
86     	 */
87     	public final void setJimpleXMLDumpLocation(final String location) {
88     		dumpLocation = location;
89     	}
90     
91     	/**
92     	 * @see edu.ksu.cis.indus.IXMLBasedTest#setIdGenerator(edu.ksu.cis.indus.xmlizer.IJimpleIDGenerator)
93     	 */
94     	public void setIdGenerator(final IJimpleIDGenerator generator) {
95     		idGenerator = generator;
96     	}
97     
98     	/**
99     	 * Sets the CFG factory to be used during testing.
100     	 *
101     	 * @param cfgFactory is the factory to be used.
102     	 */
103     	public void setStmtGraphFactory(final IStmtGraphFactory cfgFactory) {
104     		stmtGraphFactory = cfgFactory;
105     	}
106     
107     	/**
108     	 * Retrieves the CFG factory used during testing.
109     	 *
110     	 * @return the cfg factory
111     	 */
112     	public IStmtGraphFactory getStmtGraphFactory() {
113     		return stmtGraphFactory;
114     	}
115     
116     	/**
117     	 * @see IXMLBasedTest#setXMLControlDir(String)
118     	 */
119     	public void setXMLControlDir(final String xmlInDir) {
120     		xmlFirstInputDir = xmlInDir;
121     	}
122     
123     	/**
124     	 * Retrieves the directory from which one of the xml-based testing input is read from.
125     	 *
126     	 * @return directory in which one of the xml-based testing input is read from.
127     	 */
128     	public String getXMLControlDir() {
129     		return xmlFirstInputDir;
130     	}
131     
132     	/**
133     	 * @see IXMLBasedTest#setXMLTestDir(String)
134     	 */
135     	public void setXMLTestDir(final String xmlInDir) {
136     		xmlSecondInputDir = xmlInDir;
137     	}
138     
139     	/**
140     	 * Retrieves the directory from which the other xml-based testing input is read from.
141     	 *
142     	 * @return directory in which the other xml-based testing input is read from.
143     	 */
144     	public String getXMLTestDir() {
145     		return xmlSecondInputDir;
146     	}
147     
148     	/**
149     	 * @see junit.extensions.TestSetup#setUp()
150     	 */
151     	protected void setUp()
152     	  throws Exception {
153     		final Collection _temp = TestHelper.getTestCasesReachableFromSuite((TestSuite) getTest(), IXMLBasedTest.class);
154     
155     		for (final Iterator _i = _temp.iterator(); _i.hasNext();) {
156     			final IXMLBasedTest _tester = (IXMLBasedTest) _i.next();
157     			_tester.setXMLTestDir(xmlSecondInputDir);
158     			_tester.setXMLControlDir(xmlFirstInputDir);
159     			_tester.setStmtGraphFactory(stmtGraphFactory);
160     			_tester.setIdGenerator(idGenerator);
161     		}
162     	}
163     
164     	/**
165     	 * @see junit.extensions.TestSetup#tearDown()
166     	 */
167     	protected void tearDown()
168     	  throws Exception {
169     		if (stmtGraphFactory != null) {
170     			stmtGraphFactory.reset();
171     			stmtGraphFactory = null;
172     		}
173     
174     		if (idGenerator != null) {
175     			idGenerator.reset();
176     			idGenerator = null;
177     		}
178     		System.gc();
179     	}
180     }
