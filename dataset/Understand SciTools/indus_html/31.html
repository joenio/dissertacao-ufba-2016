285     	/**
286     	 * Checks if the given method either reads or writes global data.
287     	 * 
288     	 * @param method of interest.
289     	 * @param retriever to be used.
290     	 * @param read <code>true</code> indicates read information is requested; <code>false</code> indidates write info is
291     	 *            requested.
292     	 * @return <code>true</code> if any global data was read when <code>read</code> is <code>true</code> and
293     	 *         <code>false</code> if it was not read when <code>read</code> was <code>true</code>. <code>true</code>
294     	 *         if any global data was written when <code>read</code> is <code>false</code> and <code>false</code> if it
295     	 *         was not written when <code>read</code> was <code> false</code>.
296     	 * @pre method != null and retriever != null
297     	 */
298     	private boolean globalDataReadWriteInfoHelper(final SootMethod method,
299     			final ITransformer<SootMethod, MethodContext> retriever, final boolean read) {
300     		final boolean _result;
301     		final MethodContext _ctxt = retriever.transform(method);
302     
303     		if (_ctxt != null) {
304     			if (read) {
305     				_result = _ctxt.isGlobalDataRead();
306     			} else {
307     				_result = _ctxt.isGlobalDataWritten();
308     			}
309     		} else {
310     			_result = read ? readDefaultValue : writeDefaultValue;
311     
312     			if (LOGGER.isWarnEnabled()) {
313     				LOGGER.warn("No recorded information for " + method + " is available.  Returning default value - " + _result);
314     			}
315     		}
316     		return _result;
317     	}

319     	/**
320     	 * Calculates the read-write information based on an access path rooted at the given method using the given transformer.
321     	 * 
322     	 * @param method in which the accesspath is rooted.
323     	 * @param accesspath of interest.
324     	 * @param recurse <code>true</code> indicates that read/write beyond the end point should be considered.
325     	 *            <code>false</code>, otherwise.
326     	 * @param retriever to be used to get the method context and the alias set.
327     	 * @param read <code>true</code> indicates read information is requested; <code>false</code> indidates write info is
328     	 *            requested.
329     	 * @return <code>true</code> if the given access path was read when <code>read</code> is <code>true</code> and
330     	 *         <code>false</code> if it was not read when <code>read</code> was <code>true</code>. <code>true</code>
331     	 *         if the given access path was written when <code>read</code> is <code>false</code> and <code>false</code>
332     	 *         if it was not written when <code>read</code> was <code> false</code>.
333     	 * @pre method != null and accesspath != null and retriever != null
334     	 */
335     	private boolean instanceDataReadWriteHelper(final SootMethod method, final String[] accesspath, final boolean recurse,
336     			final ITransformer<SootMethod, AliasSet> retriever, final boolean read) {
337     		final AliasSet _aliasSet = retriever.transform(method);
338     		final int _pathLength = accesspath.length;
339     		final boolean _zeroLenghtPath = _pathLength == 0;
340     		final AliasSet _endPoint;
341     
342     		if (_aliasSet == null) {
343     			_endPoint = null;
344     		} else {
345     			final String[] _s;
346     
347     			if (_zeroLenghtPath) {
348     				_s = new String[0];
349     			} else {
350     				_s = new String[_pathLength - 1];
351     				System.arraycopy(accesspath, 0, _s, 0, _s.length);
352     			}
353     
354     			final Pair<AliasSet, String[]> _pair = new Pair<AliasSet, String[]>(_aliasSet.find(), _s);
355     
356     			if (this.analysis.query2handle.containsKey(_pair)) {
357     				_endPoint = this.analysis.query2handle.get(_pair);
358     			} else {
359     				_endPoint = _aliasSet.getAccessPathEndPoint(_s);
360     				this.analysis.query2handle.put(_pair, _endPoint);
361     			}
362     		}
363     
364     		final boolean _result;
365     
366     		if (_endPoint == null) {
367     			_result = read ? readDefaultValue : writeDefaultValue;
368     
369     			if (LOGGER.isWarnEnabled()) {
370     				LOGGER.warn("isAccessPathOperatedHelper(method = " + method + ", accesspath = " + Arrays.asList(accesspath)
371     						+ ", recurse = " + recurse + ", retriver = " + retriever + ") - No recorded information for "
372     						+ method + " is available.  Returning default value - " + _result);
373     			}
374     		} else {
375     			if (read) {
376     				if (_zeroLenghtPath) {
377     					_result = _endPoint.isAccessed();
378     				} else {
379     					_result = _endPoint.wasFieldRead(accesspath[_pathLength - 1], recurse);
380     				}
381     			} else {
382     				if (_zeroLenghtPath) {
383     					_result = false;
384     				} else {
385     					_result = _endPoint.wasFieldWritten(accesspath[_pathLength - 1], recurse);
386     				}
387     			}
388     		}
389     		return _result;
390     	}
