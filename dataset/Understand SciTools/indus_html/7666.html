63     	/**
64     	 * Creates a new <code>OFAnalyzer</code> instance.
65     	 * 
66     	 * @param tagName is the name of the tag used by the instance of the flow analysis framework associated with this analysis
67     	 *            instance to tag parts of the AST. Refer to <code>FA.FA(AbstractAnalyzer, String)</code> for more detail.
68     	 * @param astim the prototype of the index manager to be used in conjunction with AST nodes.
69     	 * @param arrayIM the prototype of the index manager to be used in conjunction with arrays.
70     	 * @param instancefieldIM the prototype of the index manager to be used in conjunction with fields.
71     	 * @param lexpr the LHS expression visitor prototype.
72     	 * @param rexpr the RHS expression visitor prototype.
73     	 * @param stmt the statement visitor prototype.
74     	 * @param tokenMgr manages the tokens for the objects in OFA.
75     	 * @param stmtGrphFctry the statement graph factory to use.
76     	 * @pre astim != null and allocationim != null and lexpr != null and rexpr != null and stmt != null and tokenMgr != null
77     	 */
78     	private OFAnalyzer(final String tagName, final IPrototype<? extends IIndexManager<? extends IIndex<?>, Value>> astim,
79     			final IIndexManager<? extends IIndex<?>, ArrayType> arrayIM,
80     			final IIndexManager<? extends IIndex<?>, SootField> instancefieldIM, final IExprSwitch<OFAFGNode<T>> lexpr,
81     			final IExprSwitch<OFAFGNode<T>> rexpr, final IStmtSwitch stmt, final ITokenManager<T, Value, Type> tokenMgr,
82     			final IStmtGraphFactory<?> stmtGrphFctry) {
83     		super(new AllocationContext(), tagName, tokenMgr);
84     
85     		fa.setupInstanceFieldVariantManager(instancefieldIM);
86     		fa.setupArrayVariantManager(arrayIM);
87     		fa.setupMethodVariantManager(new IndexManager<SootMethod>(), astim, new MethodVariantFactory<T>(Constants
88     				.getFAScopePattern(), stmtGrphFctry));
89     		fa.setupStaticFieldVariantManager(new IndexManager<SootField>());
90     
91     		fa.setNodePrototype(new OFAFGNode<T>(null, tokenMgr));
92     		fa.setLhsVisitorPrototype(lexpr);
93     		fa.setRhsVisitorPrototype(rexpr);
94     		fa.setStmtVisitorPrototype(stmt);
95     	}

223     	/**
224     	 * Returns values associated with the given array type associated with the given allocation sites.
225     	 * 
226     	 * @param t the array type reqarding which information is requested.
227     	 * @param sites the collection of allocation sites that are of interest when extracting field information.
228     	 * @return a collection of values the array type <code>t</code> may evaluate when associated with object created at
229     	 *         allocation sites given by <code>sites</code>.
230     	 * @pre t != null and sites != null
231     	 */
232     	public Collection<Value> getValues(final ArrayType t, final Collection<Value> sites) {
233     		Object _temp = null;
234     		Collection<Value> _retValues;
235     		final AllocationContext _ctxt = (AllocationContext) context;
236     
237     		_retValues = new HashSet<Value>();
238     		_temp = _ctxt.getAllocationSite();
239     
240     		for (final Iterator<Value> _i = sites.iterator(); _i.hasNext();) {
241     			_ctxt.setAllocationSite(_i.next());
242     			_retValues.addAll(getValues(t));
243     		}
244     		_ctxt.setAllocationSite(_temp);
245     
246     		return _retValues.isEmpty() ? Collections.<Value> emptySet() : _retValues;
247     	}
248     
249     	/**
250     	 * Returns values associated with the given field associated with the given allocation sites.
251     	 * 
252     	 * @param f the field reqarding which information is requested.
253     	 * @param sites the collection of allocation sites that are of interest when extracting field information.
254     	 * @return a collection of values the field <code>f</code> may evaluate when associated with object created at
255     	 *         allocation sites given by <code>sites</code>.
256     	 * @pre f != null and sites != null
257     	 */
258     	public Collection<Value> getValues(final SootField f, final Collection<Value> sites) {
259     		Object _temp = null;
260     		Collection<Value> _retValues;
261     		final AllocationContext _ctxt = (AllocationContext) context;
262     
263     		if (Modifier.isStatic(f.getModifiers())) {
264     			_retValues = getValues(f);
265     		} else {
266     			_retValues = new HashSet<Value>();
267     			_temp = _ctxt.getAllocationSite();
268     
269     			for (final Iterator<Value> _i = sites.iterator(); _i.hasNext();) {
270     				_ctxt.setAllocationSite(_i.next());
271     				_retValues.addAll(getValues(f));
272     			}
273     			_ctxt.setAllocationSite(_temp);
274     		}
275     		return _retValues.isEmpty() ? Collections.<Value> emptySet() : _retValues;
276     	}
