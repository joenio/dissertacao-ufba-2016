43     /**
44      * This class provides the an abstract implementation of <code>IStmtGraphFactory</code> via which unit graphs can be
45      * retrieved. The subclasses should provide suitable unit graph implementation. The control flow edges in the provided unit
46      * graphs are pruned by matching the thrown exceptions to the enclosing catch blocks. Refer to
47      * <code>Util.pruneExceptionBasedControlFlow()</code> for more information.
48      * 
49      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
50      * @author $Author: rvprasad $
51      * @version $Revision: 1.21 $ $Date: 2007/02/10 19:08:36 $
52      * @param <T> the type of the manufactored graph.
53      */
54     public abstract class AbstractStmtGraphFactory<T extends UnitGraph>
55     		implements IStmtGraphFactory<T> {
56     
57     	/**
58     	 * The logger used by instances of this class to log messages.
59     	 */
60     	private static final Logger LOGGER = LoggerFactory.getLogger(AbstractStmtGraphFactory.class);
61     
62     	/**
63     	 * The scope specification.
64     	 */
65     	protected SpecificationBasedScopeDefinition scope;
66     
67     	/**
68     	 * The environment with which this factory is associated with.
69     	 */
70     	@NonNull private IEnvironment environment;
71     
72     	/**
73     	 * This maps methods to unit graphs.
74     	 */
75     	private final Map<SootMethod, T> method2UnitGraph = new HashMap<SootMethod, T>(Constants.getNumOfMethodsInApplication());
76     
77     	/**
78     	 * Retrieves the unit graph of the given method.
79     	 * 
80     	 * @param method for which the unit graph is requested.
81     	 * @return the requested unit graph.
82     	 * @post method.isConcrete() implies result.getBody() = method.getBody()
83     	 * @post (not method.isConcrete()) implies result.getBody() != method.getBody()
84     	 */
85     	@NonNull public final T getStmtGraph(@NonNull @Immutable final SootMethod method) {
86     		if (LOGGER.isDebugEnabled()) {
87     			LOGGER.debug("getStmtGraph(method = " + method + ")");
88     		}
89     
90     		T _result = method2UnitGraph.get(method);
91     
92     		if (_result == null) {
93     			_result = getBodyForMethod(method);
94     
95     			if (_result == null) {
96     				// stub in an empty graph.
97     				final Jimple _jimple = Jimple.v();
98     				final JimpleBody _body = _jimple.newBody();
99     				@SuppressWarnings("unchecked") final Collection<Stmt> _units = _body.getUnits();
100     				_body.setMethod(method);
101     
102     				if (!method.isStatic()) {
103     					final RefType _thisType = method.getDeclaringClass().getType();
104     					final Local _newLocal = _jimple.newLocal("r0", _thisType);
105     					_body.getLocals().add(_newLocal);
106     					_units.add(_jimple.newIdentityStmt(_newLocal, _jimple.newThisRef(_thisType)));
107     				}
108     
109     				if (method.getParameterCount() > 0) {
110     					int _j = 0;
111     					for (@SuppressWarnings("unchecked") final Iterator<Type> _i = method.getParameterTypes().iterator(); _i
112     							.hasNext();) {
113     						final Type _type = _i.next();
114     						final Local _newLocal = _jimple.newLocal("p" + _j, _type);
115     						_body.getLocals().add(_newLocal);
116     						_units.add(_jimple.newIdentityStmt(_newLocal, _jimple.newParameterRef(_type,
117     								_j++)));
118     					}
119     				}
120     
121     				if (method.getReturnType() instanceof VoidType) {
122     					_units.add(_jimple.newReturnVoidStmt());
123     				} else {
124     					_units.add(_jimple.newReturnStmt(Util.getDefaultValueFor(method.getReturnType())));
125     				}
126     				_result = getStmtGraphForBody(_body);
127     			}
128     			method2UnitGraph.put(method, _result);
129     		}
130     		return _result;
131     	}
132     
133     	/**
134     	 * Resets all internal datastructures.
135     	 */
136     	public final void reset() {
137     		method2UnitGraph.clear();
138     	}
139     
140     	/**
141     	 * {@inheritDoc}
142     	 */
143     	public void setScope(final SpecificationBasedScopeDefinition scopeDef, final IEnvironment env) {
144     		scope = scopeDef;
145     		environment = env;
146     	}
147     
148     	/**
149     	 * Retrieves the body for the given method.
150     	 * 
151     	 * @param method of interest.
152     	 * @return the method body.
153     	 */
154     	protected T getBodyForMethod(@NonNull final SootMethod method) {
155     		T _result = null;
156     		if (isInScope(method)) {
157     			final JimpleBody _body = (JimpleBody) method.retrieveActiveBody();
158     			_result = getStmtGraphForBody(_body);
159     		} else if (LOGGER.isDebugEnabled()) {
160     			LOGGER.debug("Method " + method + " is out of scope or is not concrete.");
161     		}
162     		return _result;
163     	}
164     
165     	/**
166     	 * Retreives the unit graph (of a particular implementation) for the given body.
167     	 * 
168     	 * @param body to be represented as a graph.
169     	 * @return a unit graph.
170     	 */
171     	@NonNull protected abstract T getStmtGraphForBody(@NonNull final JimpleBody body);
172     
173     	/**
174     	 * Checks if the given method is in the scope.
175     	 * 
176     	 * @param method of interest.
177     	 * @return <code>true</code> if the method is in scope; <code>false</code>, otherwise.
178     	 */
179     	@Functional protected final boolean isInScope(@NonNull final SootMethod method) {
180     		return (scope == null || scope.isInScope(method, environment)) && method.isConcrete();
181     	}
182     }
