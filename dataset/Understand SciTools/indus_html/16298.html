45     /**
46      * This analysis performs a conservative static analysis to detect statements that may participate in deadlocks.
47      * 
48      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
49      * @author $Author$
50      * @version $Revision$
51      */
52     public final class DeadlockAnalysis {
53     
54     	/**
55     	 * This provides monitor information.
56     	 */
57     	private final IMonitorInfo<?> monitorInfo;
58     
59     	/**
60     	 * This provides object flow information.
61     	 */
62     	private final IValueAnalyzer<Value> ofa;
63     
64     	/**
65     	 * This provides lock based equivalence information.
66     	 */
67     	private final LockAcquisitionBasedEquivalence lbe;
68     
69     	/**
70     	 * This provides escape information.
71     	 */
72     	private final IEscapeInfo escapeInfo;
73     
74     	/**
75     	 * Creates an instance of this class.
76     	 * 
77     	 * @param m to be used.
78     	 * @param va to be used.
79     	 * @param l to be used.
80     	 * @param e to be used.
81     	 * @pre m != null and va != null
82     	 */
83     	public DeadlockAnalysis(final IMonitorInfo<?> m, final IValueAnalyzer<Value> va, final LockAcquisitionBasedEquivalence l,
84     			final IEscapeInfo e) {
85     		monitorInfo = m;
86     		ofa = va;
87     		lbe = l;
88     		escapeInfo = e;
89     	}
90     
91     	/**
92     	 * Retrieves the collection of monitors that may participate in deadlock.
93     	 * 
94     	 * @return a collection of monitors.
95     	 */
96     	public Collection<Triple<EnterMonitorStmt, ExitMonitorStmt, SootMethod>> getDeadlockingMonitors() {
97     		final SimpleNodeGraphBuilder<Triple<EnterMonitorStmt, ExitMonitorStmt, SootMethod>> _sng = new SimpleNodeGraphBuilder<Triple<EnterMonitorStmt, ExitMonitorStmt, SootMethod>>();
98     		final Collection<Triple<EnterMonitorStmt, ExitMonitorStmt, SootMethod>> _result = new HashSet<Triple<EnterMonitorStmt, ExitMonitorStmt, SootMethod>>();
99     		final Collection<Triple<EnterMonitorStmt, ExitMonitorStmt, SootMethod>> _monitors = monitorInfo.getMonitorTriples();
100     		final Collection<Triple<EnterMonitorStmt, ExitMonitorStmt, SootMethod>> _processed = new ArrayList<Triple<EnterMonitorStmt, ExitMonitorStmt, SootMethod>>();
101     		final IWorkBag<Triple<EnterMonitorStmt, ExitMonitorStmt, SootMethod>> _wb = new HistoryAwareFIFOWorkBag<Triple<EnterMonitorStmt, ExitMonitorStmt, SootMethod>>(
102     				_processed);
103     
104     		_sng.createGraph();
105     
106     		for (final Triple<EnterMonitorStmt, ExitMonitorStmt, SootMethod> _m : _monitors) {
107     			_wb.addWork(_m);
108     			while (_wb.hasWork()) {
109     				final Triple<EnterMonitorStmt, ExitMonitorStmt, SootMethod> _e = _wb.getWork();
110     				final Collection<Triple<EnterMonitorStmt, ExitMonitorStmt, SootMethod>> _enclosingMonitorTriples = monitorInfo
111     						.getEnclosingMonitorTriples(_m.getFirst(), _m.getThird(), false);
112     				_sng.addEdgeFromTo(_e, _enclosingMonitorTriples);
113     				_wb.addAllWork(_enclosingMonitorTriples);
114     			}
115     		}
116     
117     		for (final Triple<EnterMonitorStmt, ExitMonitorStmt, SootMethod> _m1 : _processed) {
118     			for (final Triple<EnterMonitorStmt, ExitMonitorStmt, SootMethod> _m2 : _processed) {
119     				if (_m1 != _m2 && areMonitorsRelated(_m1, _m2)) {
120     					_sng.addEdgeFromTo(_m1, _m2);
121     				}
122     			}
123     		}
124     
125     		_sng.finishBuilding();
126     		final SimpleNodeGraph<Triple<EnterMonitorStmt, ExitMonitorStmt, SootMethod>> _sn = _sng.getBuiltGraph();
127     		for (final List<SimpleNode<Triple<EnterMonitorStmt, ExitMonitorStmt, SootMethod>>> _nodes : _sn.getSCCs(true)) {
128     			for (final SimpleNode<Triple<EnterMonitorStmt, ExitMonitorStmt, SootMethod>> _node : _nodes) {
129     				_result.add(_sn.getObjectExtractor().transform(_node));
130     			}
131     		}
132     		return _result;
133     	}
134     
135     	/**
136     	 * Checks if the given monitors are related.
137     	 * 
138     	 * @param m1 is a monitor of interest.
139     	 * @param m2 is another monitor of interest.
140     	 * @return <code>true</code> if the monitors are related; <code>false</code>, otherwise.
141     	 */
142     	private boolean areMonitorsRelated(final Triple<EnterMonitorStmt, ExitMonitorStmt, SootMethod> m1,
143     			final Triple<EnterMonitorStmt, ExitMonitorStmt, SootMethod> m2) {
144     		boolean _result = false;
145     		final SootMethod _sm1 = m1.getThird();
146     		final SootMethod _sm2 = m2.getThird();
147     		final Pair<Stmt, SootMethod> _p1 = new Pair<Stmt, SootMethod>(m1.getSecond(), m1.getThird());
148     		final Pair<Stmt, SootMethod> _p2 = new Pair<Stmt, SootMethod>(m2.getSecond(), m2.getThird());
149     
150     		if (lbe == null || lbe.getLockAcquisitionsInEquivalenceClassOf(_p1).contains(_p2)) {
151     			final Context _ctxt = new Context();
152     			if (m1.getFirst() == null && m2.getFirst() == null) {
153     				if (_sm1.isStatic() && _sm2.isStatic()) {
154     					_result = _sm1.getDeclaringClass().equals(_sm2.getDeclaringClass());
155     				} else if (!_sm1.isStatic() && !_sm2.isStatic()) {
156     					_ctxt.setRootMethod(_sm1);
157     					final Collection<Value> _c1 = ofa.getValuesForThis(_ctxt);
158     					_ctxt.setRootMethod(_sm2);
159     					final Collection<Value> _c2 = ofa.getValuesForThis(_ctxt);
160     					_result = CollectionUtils.containsAny(_c1, _c2)
161     							&& (escapeInfo == null || (escapeInfo.escapes(_sm2.getDeclaringClass(), _sm2) && escapeInfo
162     									.escapes(_sm1.getDeclaringClass(), _sm1)));
163     				}
164     			} else {
165     				final Collection<Value> _c1;
166     				boolean _flag1 = true;
167     				boolean _flag2 = true;
168     				if (m1.getFirst() == null) {
169     					if (_sm1.isStatic()) {
170     						_c1 = Collections.emptySet();
171     					} else {
172     						_ctxt.setRootMethod(_sm1);
173     						_c1 = ofa.getValuesForThis(_ctxt);
174     						_flag1 = escapeInfo == null || escapeInfo.escapes(_sm1.getDeclaringClass(), _sm1);
175     					}
176     				} else {
177     					_ctxt.setRootMethod(_sm1);
178     					_ctxt.setStmt(m1.getFirst());
179     					_ctxt.setProgramPoint(m1.getFirst().getOpBox());
180     					_c1 = ofa.getValues(m1.getFirst().getOp(), _ctxt);
181     					_flag1 = escapeInfo == null || escapeInfo.escapes(m1.getFirst().getOp(), _sm1);
182     				}
183     
184     				final Collection<Value> _c2;
185     				if (m2.getFirst() == null) {
186     					if (_sm2.isStatic()) {
187     						_c2 = Collections.emptySet();
188     					} else {
189     						_ctxt.setRootMethod(_sm2);
190     						_c2 = ofa.getValuesForThis(_ctxt);
191     						_flag2 = escapeInfo == null || escapeInfo.escapes(_sm2.getDeclaringClass(), _sm2);
192     					}
193     				} else {
194     					_ctxt.setRootMethod(_sm2);
195     					_ctxt.setStmt(m2.getFirst());
196     					_ctxt.setProgramPoint(m2.getFirst().getOpBox());
197     					_c2 = ofa.getValues(m2.getFirst().getOp(), _ctxt);
198     					_flag2 = escapeInfo == null || escapeInfo.escapes(m2.getFirst().getOp(), _sm2);
199     				}
200     				_result = _flag1 && _flag2 && CollectionUtils.containsAny(_c1, _c2);
201     			}
202     		}
203     		return _result;
204     	}
205     }
