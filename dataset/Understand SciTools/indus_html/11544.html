17     package edu.ksu.cis.indus.staticanalyses.tokens;
18     
19     import edu.ksu.cis.indus.annotations.Empty;
20     import edu.ksu.cis.indus.staticanalyses.tokens.ITypeManager.NewTypeCreated;
21     
22     import java.util.Collection;
23     import java.util.Collections;
24     import java.util.HashMap;
25     import java.util.Map;
26     import java.util.Observable;
27     import java.util.Observer;
28     
29     /**
30      * This class provides the abstract implementation of <code>ITokenmanager</code>. It is advised that all token managers
31      * extend this class.
32      * 
33      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
34      * @author $Author: rvprasad $
35      * @version $Revision: 1.14 $ $Date: 2007/02/10 19:07:04 $
36      * @param <T> is the type of token set objects.
37      * @param <V> is the type of the value objects (in the representation).
38      * @param <R> is the type of the representation types.
39      */
40     public abstract class AbstractTokenManager<T extends ITokens<T, V>, V, R>
41     		implements ITokenManager<T, V, R>, Observer {
42     
43     	/**
44     	 * This provides the logic to update token to type relation. A simple situation is that in Java <code>null</code> is a
45     	 * valid value/token of all reference types in the system. Hence, token to type relation will change as more types are
46     	 * loaded into the system after <code>null</code> has been considered.
47     	 */
48     	protected final IDynamicTokenTypeRelationDetector<V> onlineTokenTypeRelationEvalutator;
49     
50     	/**
51     	 * The type manager that manages the types of the tokens managed by this object.
52     	 * 
53     	 * @invariant typeMgr != null
54     	 */
55     	protected final ITypeManager<R, V> typeMgr;
56     
57     	/**
58     	 * The mapping between types to the type based filter.
59     	 */
60     	private final Map<IType, ITokenFilter<T, V>> type2filter = new HashMap<IType, ITokenFilter<T, V>>();
61     
62     	/**
63     	 * Creates an instance of this class.
64     	 * 
65     	 * @param typeManager manages the types of the tokens managed by this object. The client should relinquish ownership of
66     	 *            the given argument. This argument is provided for configurability.
67     	 * @pre typeManager != null
68     	 */
69     	public AbstractTokenManager(final ITypeManager<R, V> typeManager) {
70     		typeMgr = typeManager;
71     		onlineTokenTypeRelationEvalutator = typeManager.getDynamicTokenTypeRelationEvaluator();
72     	}
73     
74     	/**
75     	 * {@inheritDoc}
76     	 */
77     	public final ITokenFilter<T, V> getTypeBasedFilter(final IType type) {
78     		ITokenFilter<T, V> _result = type2filter.get(type);
79     
80     		if (_result == null) {
81     			_result = getNewFilterForType(type);
82     			type2filter.put(type, _result);
83     		}
84     
85     		return _result;
86     	}
87     
88     	/**
89     	 * {@inheritDoc}
90     	 */
91     	public ITypeManager<R, V> getTypeManager() {
92     		return typeMgr;
93     	}
94     
95     	/**
96     	 * @see edu.ksu.cis.indus.staticanalyses.tokens.ITokenManager#reset()
97     	 */
98     	public void reset() {
99     		typeMgr.reset();
100     		type2filter.clear();
101     	}
102     
103     	/**
104     	 * {@inheritDoc}
105     	 */
106     	public final void update(@SuppressWarnings("unused") final Observable observer, final Object arg) {
107     		if (arg instanceof NewTypeCreated && onlineTokenTypeRelationEvalutator != null) {
108     			final IType _type = ((NewTypeCreated) arg).getCreatedType();
109     			final Collection<V> _values = onlineTokenTypeRelationEvalutator.getValuesConformingTo(getValues(), _type);
110     			if (!_values.isEmpty()) {
111     				recordNewTokenTypeRelations(_values, _type);
112     			}
113     		}
114     	}
115     
116     	/**
117     	 * Retrieves a new token filter for the given type.
118     	 * 
119     	 * @param type for which the filter is requested.
120     	 * @return a new token filter.
121     	 * @pre type != null
122     	 */
123     	protected abstract ITokenFilter<T, V> getNewFilterForType(final IType type);
124     
125     	/**
126     	 * Retrieves the values being managed by this manager. This implementation returns an empty collection. This is used to
127     	 * update value-token-type relation on-the-fly.
128     	 * 
129     	 * @return the values being managed by this manager.
130     	 */
131     	protected Collection<V> getValues() {
132     		return Collections.emptySet();
133     	}
134     
135     	/**
136     	 * Records the new token-type relations. This implementation does nothing. This method will be called only be called if
137     	 * new token-type relations are discovered on-the-fly.
138     	 * 
139     	 * @param values whose type has been incrementally changed. This is guaranteed to be the objects in the collection
140     	 *            <code>values</code> provided to <code>fixupTokenTypeRelation</code> method.
141     	 * @param type is the new additional type of <code>values</code>. This is guaranteed to be one of the objects in the
142     	 *            collection <code>types</code> provided to <code>fixupTokenTypeRelation</code> method.
143     	 * @pre values != null and type != null
144     	 * @pre not values.isEmpty()
145     	 */
146     	@Empty protected void recordNewTokenTypeRelations(@SuppressWarnings("unused") final Collection<V> values,
147     			@SuppressWarnings("unused") final IType type) {
148     		// does nothing
149     	}
150     }
