28     /**
29      * This class provides intraprocedural data dependency information based on identifiers. Local variables in a method enable
30      * such dependence. Given a def site, the use site is tracked based on the id being defined and used. Hence, information about
31      * field/array access via primaries which are local variables is inaccurate in such a setting, hence, it is not provided by
32      * this class. Please refer to {@link ReferenceBasedDataDA ReferenceBasedDataDA} for such information.
33      * <p>
34      * This implementation is based on <code>edu.ksu.cis.indus.staticanalyses.cfg.LocalUseDefAnalysisv2</code> class.
35      * </p>
36      * 
37      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
38      * @author $Author: rvprasad $
39      * @version $Revision: 1.7 $
40      */
41     public final class IdentifierBasedDataDAv3
42     		extends IdentifierBasedDataDAv2 {
43     
44     	/**
45     	 * @see edu.ksu.cis.indus.staticanalyses.dependency.IdentifierBasedDataDAv2#getLocalUseDefAnalysis(soot.SootMethod)
46     	 */
47     	@Override protected IUseDefInfo<DefinitionStmt, Pair<Local, Stmt>> getLocalUseDefAnalysis(final SootMethod method) {
48     		return new LocalUseDefAnalysisv2(getBasicBlockGraph(method));
49     	}
50     }
