18     package edu.ksu.cis.indus.toolkits.bandera;
19     
20     import edu.ksu.cis.bandera.tool.ToolConfigurationView;
21     import edu.ksu.cis.bandera.tool.ToolIconView;
22     
23     import edu.ksu.cis.indus.tools.IToolConfigurator;
24     
25     import org.eclipse.swt.widgets.Shell;
26     
27     
28     /**
29      * This class exposes {@link edu.ksu.cis.indus.tools.AbstractToolConfigurator AbstractToolConfigurator} and  {@link
30      * edu.ksu.cis.indus.tools.slicer.SlicerConfigurator SlicerConfigurator} in a way amenable to the  tool configuration API in
31      * bandera toolkit.  One can say it adapts <code>AbstractToolConfigurator</code> to Bandera tool  configuration API.
32      *
33      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
34      * @author $Author: rvprasad $
35      * @version $Revision: 1.14 $ $Date: 2007/02/10 19:06:03 $
36      */
37     public final class SlicerConfigurationView
38       implements ToolConfigurationView,
39     	  ToolIconView {
40     	/** 
41     	 * The configurator used to configure <code>configuration</code>.
42     	 *
43     	 * @invariant configurator != null
44     	 */
45     	private final IToolConfigurator configurator;
46     
47     	/**
48     	 * Creates a new SlicerConfigurationView object.
49     	 *
50     	 * @param theConfigurator is used to configure the slicer.
51     	 *
52     	 * @pre theConfigurator != null
53     	 */
54     	SlicerConfigurationView(final IToolConfigurator theConfigurator) {
55     		configurator = theConfigurator;
56     	}
57     
58     	/**
59     	 * @see edu.ksu.cis.bandera.tool.ToolIconView#getActiveIcon()
60     	 */
61     	public Object getActiveIcon() {
62     		return null;
63     	}
64     
65     	/**
66     	 * @see edu.ksu.cis.bandera.tool.ToolIconView#getDisabledIcon()
67     	 */
68     	public Object getDisabledIcon() {
69     		return null;
70     	}
71     
72     	/**
73     	 * @see edu.ksu.cis.bandera.tool.ToolIconView#getIcon()
74     	 */
75     	public Object getIcon() {
76     		return null;
77     	}
78     
79     	/**
80     	 * @see edu.ksu.cis.bandera.tool.ToolIconView#getText()
81     	 */
82     	public String getText() {
83     		return "Configure Slicer";
84     	}
85     
86     	/**
87     	 * @see edu.ksu.cis.bandera.tool.ToolIconView#getToolTipText()
88     	 */
89     	public String getToolTipText() {
90     		return "Click here to configure the slicer";
91     	}
92     
93     	/**
94     	 * @see edu.ksu.cis.bandera.tool.ToolConfigurationView#configure()
95     	 */
96     	public void configure() {
97     		final Shell _shell = new Shell();
98     		_shell.setText("Configure Slicer");
99     		configurator.initialize(_shell);
100     		_shell.pack();
101     		_shell.open();
102     	}
103     }
