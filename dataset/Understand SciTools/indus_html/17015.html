17     package edu.ksu.cis.indus.annotations;
18     
19     import java.lang.annotation.Documented;
20     import java.lang.annotation.ElementType;
21     import java.lang.annotation.Target;
22     
23     /**
24      * This annotation indicates the degree to which a method is functional (free of observational side-effect). Parameters and
25      * return values of the method are immutable independent of the degree of functional-ness.
26      * 
27      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
28      * @author $Author: rvprasad $
29      * @version $Revision: 1.3 $ $Date: 2007/02/10 19:08:37 $
30      */
31     @Target({ ElementType.METHOD }) @Documented public @interface Functional {
32     
33     	/**
34     	 * The enumeration of various access specifiers in Java.
35     	 * 
36     	 * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
37     	 * @author $Author: rvprasad $
38     	 * @version $Revision: 1.3 $ $Date: 2007/02/10 19:08:37 $
39     	 */
40     	public enum AccessSpecifier {
41     		/**
42     		 * Private access specification.
43     		 */
44     		PRIVATE,
45     		/**
46     		 * Package private access specification.
47     		 */
48     		PACKAGE,
49     		/**
50     		 * Protected access specification.
51     		 */
52     		PROTECTED,
53     		/**
54     		 * Public access specification.
55     		 */
56     		PUBLIC
57     	};
58     
59     	/**
60     	 * Provides the access specification level at and beyond (less exposure) which the method is functional.
61     	 * 
62     	 * @return the access specification level.
63     	 */
64     	AccessSpecifier level() default AccessSpecifier.PRIVATE;
65     }
