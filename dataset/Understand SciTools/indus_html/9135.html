168     		return defaultValue;

239     	 * Inverts the given map. In the returned map, each key is mapped to a collection. This is to address situation where
240     	 * multiple keys may map to the same value in the given map.
241     	 * 
242     	 * @param <K> the type of the keys in the given map.
243     	 * @param <V> the type of the elements in the given map.
244     	 * @param map to be inverted.
245     	 * @return the inverted map.
246     	 * @post map.values().containsAll(result.keySet())
247     	 * @post result.keySet().containsAll(map.values())
248     	 * @post result.values()->forall(o | map.keySet().containsAll(o))
249     	 * @post map.keySet()->forall(o | result.values()->exists(p | p.contains(o)))
250     	 */
251     	@NonNull @Immutable public static <K, V> Map<V, Set<K>> invertMap(@NonNull final Map<K, V> map) {
252     		final Map<V, Set<K>> _result = new HashMap<V, Set<K>>();
253     		final Collection<K> _values = map.keySet();
254     		final Iterator<K> _i = _values.iterator();
255     		final int _iEnd = _values.size();
256     
257     		for (int _iIndex = 0; _iIndex < _iEnd; _iIndex++) {
258     			final K _valueInResult = _i.next();
259     			final V _keyInResult = map.get(_valueInResult);
260     			if (!_result.containsKey(_keyInResult)) {
261     				final Set<K> _set = new HashSet<K>();
262     				_result.put(_keyInResult, _set);
263     			}
264     			final Collection<K> _t = _result.get(_keyInResult);
265     			_t.add(_valueInResult);
266     		}
267     		return _result;
268     	}
