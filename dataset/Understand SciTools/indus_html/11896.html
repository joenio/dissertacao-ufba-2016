108     	 * This predicate allows only one statement to be considered as a criterion.
109     	 * 
110     	 * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>

114     	private class CustomCriteriaPredicate
115     			extends AbstractSliceCriteriaPredicate<Stmt> {
116     
117     		/**
118     		 * This indicates if the predicate is counting. Only in counting status will the predicate accept criteria.
119     		 */
120     		private boolean countingStatus = true;
121     
122     		/**
123     		 * The collection of statements that have been considered as criterion.
124     		 */
125     		private final Collection<Stmt> stmts;
126     
127     		/**
128     		 * Creates an instance of this class.
129     		 */
130     		CustomCriteriaPredicate() {
131     			super();
132     			stmts = new HashSet<Stmt>();
133     		}
134     
135     		/**
136     		 * {@inheritDoc}
137     		 */
138     		public boolean evaluate(final Stmt t) {
139     			final boolean _flag;
140     			if (countingStatus) {
141     				_flag = stmts.add(t);
142     				if (_flag) {
143     					countingStatus = false;
144     				}
145     			} else {
146     				_flag = false;
147     			}
148     			return _flag;
149     		}
150     
151     		/**
152     		 * Retrieves the number of slices generated so far.
153     		 * 
154     		 * @return the number of slices.
155     		 */
156     		int getSliceCount() {
157     			return stmts.size();
158     		}
159     
160     		/**
161     		 * Sets the predicate into counting status.
162     		 */
163     		void setCountingStatus() {
164     			countingStatus = true;
165     		}
166     	}

286     	 * The entry point to the driver.
287     	 * 
288     	 * @param args contains the command line arguments.
289     	 */
290     	public static void main(final String[] args) {
291     		final SliceXMLizerCLI _xmlizer = new SliceXMLizerCLI();
292     
293     		// parse command line arguments
294     		parseCommandLine(args, _xmlizer);
295     
296     		_xmlizer.initialize();
297     
298     		if (!_xmlizer.generateSeparateSlicesForEachThrowStmt) {
299     			final long _startTime = System.currentTimeMillis();
300     
301     			_xmlizer.executeForSingleSlice();
302     
303     			final long _stopTime = System.currentTimeMillis();
304     
305     			if (LOGGER.isInfoEnabled()) {
306     				LOGGER.info("It took " + (_stopTime - _startTime) + "ms to identify the slice.");
307     			}
308     
309     			_xmlizer.setIDGenerator(new UniqueJimpleIDGenerator());
310     			_xmlizer.writeSliceXML();
311     			_xmlizer.outputStats(_xmlizer.nameOfSliceTag);
312     			_xmlizer.residualize();
313     			_xmlizer.slicer.reset();
314     			_xmlizer.reset();
315     		} else {
316     			_xmlizer.executeForMultipleSlices();
317     		}
318     	}

371     	/**
372     	 * Executes the slicer to generate multiple slices -- one for each throw statement. The slice will not be residualized in
373     	 * this mode.
374     	 */
375     	private void executeForMultipleSlices() {
376     		final StmtTypeBasedSliceCriteriaGenerator _generator = new StmtTypeBasedSliceCriteriaGenerator();
377     		_generator.setStmtTypes(Collections.singleton(ThrowStmt.class));
378     
379     		if (preserveThrowStmtsInAppClassesOnly) {
380     			_generator.setSiteSelectionPredicate(new ApplicationClassesOnlyPredicate());
381     		}
382     
383     		final CustomCriteriaPredicate _filter = new CustomCriteriaPredicate();
384     
385     		_generator.setCriteriaFilterPredicate(_filter);
386     		slicer.addCriteriaGenerator(_generator);
387     
388     		// execute the slicer
389     		final Collection<ISliceCriterion> _criteria = processCriteriaSpecFile();
390     		slicer.addCriteria(_criteria);
391     		slicer.setSystem(new Environment(scene));
392     		slicer.setRootMethods(rootMethods);
393     		slicer.setSliceScopeDefinition(sliceScope);
394     		slicer.addToolProgressListener(this);
395     
396     		slicer.setTagName(nameOfSliceTag);
397     		slicer.run(Phase.STARTING_PHASE, SlicerTool.SLICE_MAJOR_PHASE, true);
398     
399     		int _prevSize;
400     		int _count = 0;
401     		do {
402     			_prevSize = _filter.getSliceCount();
403     			final String _tag = nameOfSliceTag + ":" + _prevSize;
404     			slicer.setTagName(_tag);
405     			slicer.run(SlicerTool.SLICE_MAJOR_PHASE, Phase.FINISHED_PHASE, true);
406     			outputStats(_tag);
407     			dumpJimple("_" + String.valueOf(_count++), true, false, _tag);
408     			_filter.setCountingStatus();
409     			if (LOGGER.isInfoEnabled()) {
410     				try {
411     					LOGGER.info("Criteria specification after slicing: \n"
412     							+ SliceCriteriaParser.serialize(slicer.getCriteria()));
413     				} catch (final JiBXException _e) {
414     					LOGGER.info("JiBX failed during serialization.", _e);
415     				}
416     			}
417     		} while (_filter.getSliceCount() == _count);
418     	}

420     	/**
421     	 * Executes the slicer to generate a slice.
422     	 */
423     	void executeForSingleSlice() {
424     		// execute the slicer
425     		slicer.setTagName(nameOfSliceTag);
426     		slicer.setSystem(new Environment(scene));
427     		slicer.setRootMethods(rootMethods);
428     
429     		if (preserveThrowStatements) {
430     			final StmtTypeBasedSliceCriteriaGenerator _generator = new StmtTypeBasedSliceCriteriaGenerator();
431     			_generator.setStmtTypes(Collections.singleton(ThrowStmt.class));
432     			if (preserveThrowStmtsInAppClassesOnly) {
433     				_generator.setSiteSelectionPredicate(new IPredicate<SootMethod>() {
434     
435     					public boolean evaluate(final SootMethod t) {
436     						return t.getDeclaringClass().isApplicationClass();
437     					}
438     				});
439     			}
440     			slicer.addCriteriaGenerator(_generator);
441     		}
442     
443     		if (lineBasedCriteriaGenerator != null) {
444     			slicer.addCriteriaGenerator(lineBasedCriteriaGenerator);
445     		}
446     
447     		slicer.setSliceScopeDefinition(sliceScope);
448     		
449     		final Collection<ISliceCriterion> _criteria = processCriteriaSpecFile();
450     		slicer.addCriteria(_criteria);
451     		slicer.addToolProgressListener(this);
452     		slicer.run(Phase.STARTING_PHASE, null, true);
453     
454     		if (LOGGER.isInfoEnabled()) {
455     			try {
456     				LOGGER.info("Criteria specification after slicing: \n" + SliceCriteriaParser.serialize(slicer.getCriteria()));
457     			} catch (final JiBXException _e) {
458     				LOGGER.info("JiBX failed during serialization.", _e);
459     			}
460     		}
461     	}

506     			if (LOGGER.isInfoEnabled()) {
507     				LOGGER.info("BEGIN: Dumping XMLized Jimple");
508     			}
509     
510     			final ProcessingController _ctrl = new ProcessingController();
511     			final IProcessingFilter _filter = new XMLizingProcessingFilter();
512     			final OneAllStmtSequenceRetriever _ssr = new OneAllStmtSequenceRetriever();
513     			_ssr.setStmtGraphFactory(getStmtGraphFactory());
514     			_ctrl.setStmtSequencesRetriever(_ssr);
515     			_ctrl.setEnvironment(getEnvironment());
516     			_filter.chain(new TagBasedProcessingFilter(nameOfSliceTag));
517     			_ctrl.setProcessingFilter(_filter);
518     			((AbstractXMLizer) _xmlizer).dumpJimple(base, jimpleXMLDumpDir, _ctrl);
519     
520     			if (LOGGER.isInfoEnabled()) {
521     				LOGGER.info("END: Dumping XMLized Jimple");
522     			}
523     		}

846     	 * Dumps jimple for the classes in the scene. The jimple file names will end with the given suffix.
847     	 * 
848     	 * @param suffix to be appended to the file name.
849     	 * @param jimpleFile <code>true</code> indicates that class files should be dumped as well; <code>false</code>,
850     	 *            otherwise.
851     	 * @param classFile <code>true</code> indicates that class files should be dumped as well; <code>false</code>,
852     	 *            otherwise.
853     	 * @param tag to jimplify.
854     	 */
855     	private void dumpJimple(final String suffix, final boolean jimpleFile, final boolean classFile, final String tag) {
856     		final Printer _printer = Printer.v();
857     
858     		for (@SuppressWarnings("unchecked") final Iterator<SootClass> _i = scene.getClasses().iterator(); _i.hasNext();) {
859     			final SootClass _sc = _i.next();
860     
861     			if (!_sc.hasTag(tag)) {
862     				continue;
863     			}
864     
865     			if (LOGGER.isDebugEnabled()) {
866     				LOGGER.debug("Dumping jimple for " + _sc + " with suffix " + suffix);
867     			}
868     
869     			for (@SuppressWarnings("unchecked") final Iterator<SootMethod> _j = _sc.getMethods().iterator(); _j.hasNext();) {
870     				final SootMethod _sm = _j.next();
871     
872     				if (_sm.isConcrete()) {
873     					try {
874     						_sm.retrieveActiveBody();
875     					} catch (final RuntimeException _e) {
876     						LOGGER.error("Failed to retrieve body for method " + _sm, _e);
877     					}
878     				}
879     			}
880     
881     			PrintWriter _writer = null;
882     
883     			try {
884     				// write .jimple file
885     				if (jimpleFile) {
886     					final File _file = new File(outputDirectory + File.separator + _sc.getName() + ".jimple" + suffix);
887     					_writer = new PrintWriter(new FileWriter(_file));
888     					_printer.printTo(_sc, _writer);
889     				}
890     
891     				// write .class file
892     				if (classFile) {
893     					_printer.write(_sc, outputDirectory);
894     				}
895     			} catch (final IOException _e) {
896     				LOGGER.error("Error while writing " + _sc, _e);
897     			} catch (final RuntimeException _e) {
898     				LOGGER.error("Error while writing class file of " + _sc, _e);
899     			} finally {
900     				if (_writer != null) {
901     					_writer.flush();
902     					_writer.close();
903     				}
904     			}
905     		}
906     	}

987     	 * Residualize the slice as jimple files in the output directory.
988     	 */
989     	private void residualize() {
990     		if (preResXMLJimpleDump) {
991     			dumpJimpleAsXML("unsliced");
992     		}
993     
994     		if (preResJimpleDump) {
995     			dumpJimple("_preRes", true, false, nameOfSliceTag);
996     		}
997     
998     		if (residualize) {
999     			if (retentionList != null) {
1000     				SlicerToolHelper.optimizeForSpaceBeforeResidualization(slicer, retentionList);
1001     			}
1002     			destructivelyUpdateJimple();
1003     
1004     			if (retentionList != null) {
1005     				final Collection<SootClass> _c = SlicerToolHelper.optimizeForSpaceAfterResidualization(slicer, retentionList);
1006     
1007     				if (LOGGER.isDebugEnabled()) {
1008     					LOGGER.debug("Retained classes are " + _c);
1009     				}
1010     			}
1011     			dumpJimple("", false, true, nameOfSliceTag);
1012     		}
1013     
1014     		if (postResJimpleDump) {
1015     			dumpJimple("_postRes", true, false, nameOfSliceTag);
1016     		}
1017     
1018     		if (postResXMLJimpleDump) {
1019     			dumpJimpleAsXML("sliced");
1020     		}
1021     	}
