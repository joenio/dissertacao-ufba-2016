17     package edu.ksu.cis.indus.processing;
18     
19     import soot.SootClass;
20     import soot.SootField;
21     import soot.SootMethod;
22     import soot.ValueBox;
23     
24     import soot.jimple.Stmt;
25     
26     /**
27      * This is the interface to be provided by higher level analyses to a controlling agent. The controlling agent walks over the
28      * analyzed system and calls the interested processors at each value and statement in the analyzed system. Upon callback, the
29      * processors suitably collect/process the information available from the low-level analyzer.
30      *
31      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
32      * @author $Author: rvprasad $
33      * @version $Revision: 1.13 $
34      */
35     public interface IProcessor {
36     	/**
37     	 * This method will be called by the controlling agent for each class in the analyzed system.
38     	 *
39     	 * @param clazz to be processed.
40     	 * @pre clazz != null
41     	 */
42     	void callback(SootClass clazz);
43     
44     	/**
45     	 * This method will be called by the controlling agent for each field in each class in the analyzed system.
46     	 *
47     	 * @param field to be processed.
48     	 * @pre field != null
49     	 */
50     	void callback(SootField field);
51     
52     	/**
53     	 * This method will be called by the controlling agent for each method for each class in the analyzed system.
54     	 *
55     	 * @param method to be processed.
56     	 * @pre method != null
57     	 */
58     	void callback(SootMethod method);
59     
60     	/**
61     	 * This method will be called by the controlling agent upon walking a statement in the analyzed system.
62     	 *
63     	 * @param stmt to be processed.
64     	 * @param context in which to <code>stmt</code> should be processed.
65     	 * @pre stmt != null
66     	 */
67     	void callback(Stmt stmt, Context context);
68     
69     	/**
70     	 * This method will be called by the controlling agent upon walking a value in the analyzed system.
71     	 *
72     	 * @param vBox to be processed.
73     	 * @param context in which to <code>value</code> should be processed.
74     	 * @pre vBox != null
75     	 */
76     	void callback(ValueBox vBox, Context context);
77     
78     	/**
79     	 * This gives the processors to consolidate before the information is available to the user. This <i>should</i> be called
80     	 * before the processors are queried for the results of the processing.
81     	 */
82     	void consolidate();
83     
84     	/**
85     	 * This method will be called by the application. The processor should register it's interest with the controller via this
86     	 * method.
87     	 *
88     	 * @param ppc is the processing controller.
89     	 * @pre ppc != null
90     	 */
91     	void hookup(ProcessingController ppc);
92     
93     	/**
94     	 * This method indicates to the processors that the processing will begin. Implementation can suitably initialize in this
95     	 * method.
96     	 */
97     	void processingBegins();
98     
99     	/**
100     	 * Resets the processor.
101     	 */
102     	void reset();
103     
104     	/**
105     	 * This method will be called by the application. The processor should unregister it's interest with the controller via
106     	 * this method after it has participated in processing.
107     	 *
108     	 * @param ppc is the processing controller.
109     	 * @pre ppc != null
110     	 */
111     	void unhook(ProcessingController ppc);
112     }
