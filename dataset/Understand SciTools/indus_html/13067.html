17     package edu.ksu.cis.indus.interfaces;
18     
19     import edu.ksu.cis.indus.common.datastructures.Pair;
20     import edu.ksu.cis.indus.common.datastructures.Triple;
21     
22     import java.util.Collection;
23     
24     import soot.SootClass;
25     import soot.SootMethod;
26     
27     import soot.jimple.AssignStmt;
28     import soot.jimple.InvokeStmt;
29     
30     /**
31      * This interface provides thread graph information pertaining to the analyzed system. It is adviced that any post processor
32      * which provides Thread graph information should provide it via this interface.
33      * <p>
34      * Subtypes of this class have to return the constant <code>ID</code> defined in this class as a result of
35      * <code>getId</code>.
36      * </p>
37      *
38      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
39      * @author $Author: rvprasad $
40      * @version $Revision: 1.18 $
41      */
42     public interface IThreadGraphInfo
43     		extends IStatus, IIdentification {
44     
45     	/**
46     	 * The id of this interface.
47     	 */
48     	Comparable<?> ID = "Threadgraph Information";
49     
50     	/**
51     	 * Checks if there is a class initializing thread in the given collection of threads.
52     	 *
53     	 * @param executionThreads to be tested.
54     	 * @return <code>true</code> if one of the threads in <code>executionThreads</code> did initialize a class;
55     	 *         <code>false</code>, otherwise.
56     	 * @pre executionThreads != null
57     	 */
58     	boolean containsClassInitThread(Collection<Triple<InvokeStmt, SootMethod, SootClass>> executionThreads);
59     
60     	/**
61     	 * Returns a collection of thread allocation sites in the system.
62     	 *
63     	 * @return a collection of pairs of creation statements and their enclosing methods. The returned pairs consists of a
64     	 *         statement and the method in which it occurs. 
65     	 * @post result != null 
66     	 */
67     	Collection<Pair<AssignStmt, SootMethod>> getAllocationSites();
68     
69     	/**
70     	 * Returns the sites which create new threads, i.e., <code>java.lang.Thread.start()</code> call-sites.
71     	 *
72     	 * @return a collection of call-sites which captures the start sites for threads. The returned pairs consists of a
73     	 *         invocation statement and the method in which it occurs. However, system threads are represented as pairs of
74     	 *         simple objects. So, the caller is adviced to check for the types of the Pair before using them.
75     	 * @post result != null
76     	 */
77     	Collection<Pair<InvokeStmt, SootMethod>> getCreationSites();
78     
79     	/**
80     	 * Returns the methods executed in the <code>Thread</code> created at the given <code>startStmt</code> in the given
81     	 * context <code>ctxt</code>.
82     	 *
83     	 * @param startStmt is the statement in which the thread is created.
84     	 * @param method is the method in which the statement occurs.
85     	 * @return a collection of methods executed in this thread.
86     	 * @pre startStmt != null and method != null
87     	 * @post result != null
88     	 */
89     	Collection<SootMethod> getExecutedMethods(InvokeStmt startStmt, SootMethod method);
90     
91     	/**
92     	 * Returns the threads in which the given method is executed.
93     	 *
94     	 * @param sm is the method which is executed.
95     	 * @return a collection of thread creation sites (invocation statement and method) along with the class that provided that
96     	 *         body of the thread in which the given method executes is returned.
97     	 * @pre sm != null
98     	 * @post result != null
99     	 */
100     	Collection<Triple<InvokeStmt, SootMethod, SootClass>> getExecutionThreads(SootMethod sm);
101     
102     	/**
103     	 * Returns the methods that act as the entry points in to the threads. These are implementations of
104     	 * <code>java.lang.Runnable.run()</code> or class initializer methods.
105     	 *
106     	 * @return a collection of methods.
107     	 */
108     	Collection<SootMethod> getThreadEntryPoints();
109     
110     	/**
111     	 * Checks if the methods will always occur in the different threads.
112     	 *
113     	 * @param methodOne obviously contains the definition.
114     	 * @param methodTwo obviously contains the use.
115     	 * @return <code>true</code> if the given methods will only occur in different threads; <code>false</code>,
116     	 *         otherwise.
117     	 * @pre methodOne != null and methodTwo != null
118     	 */
119     	boolean mustOccurInDifferentThread(SootMethod methodOne, SootMethod methodTwo);
120     
121     	/**
122     	 * Checks if the methods will only occur in the same thread.
123     	 *
124     	 * @param methodOne obviously contains the definition.
125     	 * @param methodTwo obviously contains the use.
126     	 * @return <code>true</code> if the given methods will only occur in the same thread; <code>false</code>, otherwise.
127     	 * @pre methodOne != null and methodTwo != null
128     	 */
129     	boolean mustOccurInSameThread(SootMethod methodOne, SootMethod methodTwo);
130     }
