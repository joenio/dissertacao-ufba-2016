48     /**
49      * This class provides the logic required to process the given slice in order to include goto statements such that it realizes
50      * the control as in the original program but as required in the slice.
51      * 
52      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
53      * @author $Author: rvprasad $
54      * @version $Revision: 1.35 $ $Date: 2007/02/10 19:05:58 $
55      */
56     public final class SliceGotoProcessor {
57     
58     	/**
59     	 * This filter out statements that are not of type <code>GotoStmt</code>.
60     	 */
61     	public static final IPredicate<Stmt> GOTO_STMT_PREDICATE = new InstanceOfPredicate<GotoStmt, Stmt>(GotoStmt.class);
62     
63     	/**
64     	 * The logger used by instances of this class to log messages.
65     	 */
66     	private static final Logger LOGGER = LoggerFactory.getLogger(SliceGotoProcessor.class);
67     
68     	/**
69     	 * The slice collector.
70     	 */
71     	protected final SliceCollector sliceCollector;
72     
73     	/**
74     	 * The method being processed.
75     	 */
76     	protected SootMethod method;
77     
78     	/**
79     	 * Creates a new AbstractSliceGotoProcessor object.
80     	 * 
81     	 * @param collector collects the slice.
82     	 * @pre collector != null
83     	 */
84     	public SliceGotoProcessor(final SliceCollector collector) {
85     		sliceCollector = collector;
86     	}
87     
88     	/**
89     	 * Process the given methods.
90     	 * 
91     	 * @param methods to be processed.
92     	 * @param bbgMgr provides the basic block required to process the methods.
93     	 * @pre methods != null and bbgMgr != null
94     	 * @pre methods.oclIsKindOf(Collection(SootMethod))
95     	 */
96     	public void process(final Collection<SootMethod> methods, final BasicBlockGraphMgr bbgMgr) {
97     		// include all gotos required to recreate the control flow of the system.
98     		for (final Iterator<SootMethod> _i = methods.iterator(); _i.hasNext();) {
99     			final SootMethod _sm = _i.next();
100     			final BasicBlockGraph _bbg = bbgMgr.getBasicBlockGraph(_sm);
101     
102     			if (_bbg != null) {
103     				process(_sm, _bbg);
104     			}
105     		}
106     	}
107     
108     	/**
109     	 * Process the current method's body for goto-based control flow retention.
110     	 * 
111     	 * @param theMethod to be processed.
112     	 * @param bbg is the basic block graph of <code>theMethod</code>.
113     	 * @pre theMethod != null
114     	 * @pre bbg != null
115     	 */
116     	private void process(final SootMethod theMethod, final BasicBlockGraph bbg) {
117     		if (LOGGER.isDebugEnabled()) {
118     			LOGGER.debug("process(SootMethod theMethod = " + theMethod + ") - BEGIN");
119     		}
120     
121     		method = theMethod;
122     
123     		// process basic blocks to include all gotos in basic blocks with slice statements.
124     		final Collection<BasicBlock> _bbInSlice = processForIntraBasicBlockGotos(bbg);
125     
126     		if (methodBodyContainsNonTrivialSlice(_bbInSlice)) {
127     			final SimpleNodeGraph<BasicBlock> _dag = bbg.getDAG();
128     			final Collection<SimpleNode<BasicBlock>> _bbInSliceInDAG = new ArrayList<SimpleNode<BasicBlock>>();
129     			final Iterator<BasicBlock> _i = _bbInSlice.iterator();
130     			final int _iEnd = _bbInSlice.size();
131     
132     			for (int _iIndex = 0; _iIndex < _iEnd; _iIndex++) {
133     				final BasicBlock _bb = _i.next();
134     				_bbInSliceInDAG.add(_dag.queryNode(_bb));
135     			}
136     
137     			// find basic blocks between slice basic blocks to include the gotos in them into the slice.
138     			final Collection<SimpleNode<BasicBlock>> _bbToBeIncludedInSlice = _dag.getNodesOnPathBetween(_bbInSliceInDAG);
139     
140     			// find basic blocks that are part of cycles (partially or completely) in the slice.
141     			final Collection<Pair<BasicBlock, BasicBlock>> _backedges = bbg.getBackEdges();
142     			final Iterator<Pair<BasicBlock, BasicBlock>> _k = _backedges.iterator();
143     			final int _kEnd = _backedges.size();
144     
145     			for (int _kIndex = 0; _kIndex < _kEnd; _kIndex++) {
146     				final Pair<BasicBlock, BasicBlock> _edge = _k.next();
147     				_bbInSliceInDAG.clear();
148     				_bbInSliceInDAG.add(_dag.queryNode(_edge.getFirst()));
149     				_bbInSliceInDAG.add(_dag.queryNode(_edge.getSecond()));
150     
151     				final Collection<SimpleNode<BasicBlock>> _nodes = _dag.getNodesOnPathBetween(_bbInSliceInDAG);
152     
153     				if (CollectionUtils.containsAny(_nodes, _bbInSlice)
154     						|| CollectionUtils.containsAny(_nodes, _bbToBeIncludedInSlice)) {
155     					_bbToBeIncludedInSlice.addAll(_nodes);
156     				}
157     			}
158     
159     			final Collection<BasicBlock> _t = CollectionUtils.collect(_bbToBeIncludedInSlice, _dag.getObjectExtractor());
160     
161     			// include the gotos in the found basic blocks in the slice.
162     			final Iterator _j = _t.iterator();
163     			final int _jEnd = _t.size();
164     
165     			for (int _jIndex = 0; _jIndex < _jEnd; _jIndex++) {
166     				final BasicBlock _bb = (BasicBlock) _j.next();
167     				_bbInSlice.add(_bb);
168     				final List<Stmt> _stmtsOf = new ArrayList<Stmt>(_bb.getStmtsOf());
169     				CollectionUtils.filter(_stmtsOf, GOTO_STMT_PREDICATE);
170     				sliceCollector.includeInSlice(_stmtsOf);
171     			}
172     
173     			final List<Stmt> _stmtList = new ArrayList<Stmt>(bbg.getStmtGraph().getBody().getUnits());
174     			for (final BasicBlock _bb : _bbInSlice) {
175     				final Stmt _leader = _bb.getLeaderStmt();
176     				final int _i1 = _stmtList.indexOf(_leader) - 1;
177     				if (_i1 >= 0) {
178     					final Stmt _prev = _stmtList.get(_i1);
179     					if (_prev instanceof ThrowStmt || _prev instanceof ReturnStmt || _prev instanceof ReturnVoidStmt) {
180     						sliceCollector.includeInSlice(_prev);
181     					}
182     				}
183     			}
184     		}
185     
186     		if (LOGGER.isDebugEnabled()) {
187     			LOGGER.debug("process() - END");
188     		}
189     	}
190     
191     	/**
192     	 * Checks if the given basic blocks contain parts of a non-trivial slice, i.e. contains an expression with side effect
193     	 * beyond the scope of the method.
194     	 * 
195     	 * @param bbInSlice are the basic blocks to check in.
196     	 * @return <code>true</code> if the blocks contain parts of a non-trivial slice; <code>false</code>, otherwise.
197     	 */
198     	private boolean methodBodyContainsNonTrivialSlice(@NonNull @NonNullContainer final Collection<BasicBlock> bbInSlice) {
199     		boolean _result = false;
200     		for (final BasicBlock _block : bbInSlice) {
201     			for (final Stmt _s : _block.getStmtsOf()) {
202     				if (sliceCollector.hasBeenCollected(_s) && !(_s instanceof IdentityStmt) && !(_s instanceof ReturnStmt)
203     						&& !(_s instanceof ReturnVoidStmt) && !(_s instanceof ThrowStmt)) {
204     					_result = true;
205     					break;
206     				}
207     			}
208     		}
209     		return _result;
210     	}
211     
212     	/**
213     	 * Process the basic block to consider intra basic block gotos to reconstruct the control flow.
214     	 * 
215     	 * @param bb is the basic block to be processed.
216     	 * @param bbInSlice is the collection of basic blocks containing atleast one statement in the slice. This is an out param.
217     	 * @post bbInSlice.containsAll(bbInSlice$pre)
218     	 */
219     	private void processForIntraBasicBlockGotos(@NonNull final BasicBlock bb,
220     			@NonNull @NonNullContainer final Collection<BasicBlock> bbInSlice) {
221     		for (final Iterator<Stmt> _i = bb.getStmtsOf().iterator(); _i.hasNext();) {
222     			final Stmt _stmt = _i.next();
223     
224     			if (sliceCollector.hasBeenCollected(_stmt)) {
225     				bbInSlice.add(bb);
226     
227     				final List<Stmt> _stmtsOf = new ArrayList<Stmt>(bb.getStmtsOf());
228     				CollectionUtils.filter(_stmtsOf, GOTO_STMT_PREDICATE);
229     				sliceCollector.includeInSlice(_stmtsOf);
230     				break;
231     			}
232     		}
233     	}
234     
235     	/**
236     	 * Process the basic block graph to consider intra basic block gotos to reconstruct the control flow.
237     	 * 
238     	 * @param bbg is the basic block graph containing the basic blocks to be processed.
239     	 * @return the basic blocks containing atleast one statement in the slice.
240     	 * @pre bbg != null
241     	 * @post result != null and result.oclIsKindOf(Collection(BasicBloc))
242     	 * @post bbg.getNodes().containsAll(result)
243     	 */
244     	private Collection<BasicBlock> processForIntraBasicBlockGotos(final BasicBlockGraph bbg) {
245     		final Collection<BasicBlock> _result = new HashSet<BasicBlock>();
246     
247     		for (final Iterator<BasicBlock> _j = bbg.getNodes().iterator(); _j.hasNext();) {
248     			final BasicBlock _bb = _j.next();
249     			processForIntraBasicBlockGotos(_bb, _result);
250     		}
251     		return _result;
252     	}
253     }
