17     package edu.ksu.cis.indus.common.graph;
18     
19     import edu.ksu.cis.indus.annotations.Immutable;
20     import edu.ksu.cis.indus.annotations.NonNull;
21     import edu.ksu.cis.indus.annotations.NonNullContainer;
22     
23     import java.util.Collection;
24     
25     /**
26      * This interface is used to build edge-labelled directed object graphs.
27      * 
28      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
29      * @author $Author: rvprasad $
30      * @version $Revision: 1.5 $ $Date: 2007/02/10 19:08:36 $
31      * @param <O> the type of object stored in the nodes of this graph.
32      */
33     public interface IEdgeLabelledObjectDirectedGraphBuilder<O> {
34     
35     	/**
36     	 * Adds a labelled edge between each of the source nodes and the destination node.
37     	 * 
38     	 * @param sources are the source nodes.
39     	 * @param label is the edge label.
40     	 * @param dest is the destination node.
41     	 */
42     	void addEdgeFromTo(@NonNull @Immutable @NonNullContainer final Collection<O> sources,
43     			@NonNull @Immutable final Object label, @NonNull final O dest);
44     
45     	/**
46     	 * Adds a labelled edge between the source node and each of the destination nodes.
47     	 * 
48     	 * @param src is the source node.
49     	 * @param label is the edge label.
50     	 * @param destinations are the destination nodes.
51     	 */
52     	void addEdgeFromTo(@NonNull final O src, @NonNull @Immutable final Object label,
53     			@NonNull @Immutable @NonNullContainer final Collection<O> destinations);
54     
55     	/**
56     	 * Adds a labelled edge between 2 nodes.
57     	 * 
58     	 * @param src is the source node.
59     	 * @param label is the edge label.
60     	 * @param dest is the destination node.
61     	 */
62     	void addEdgeFromTo(@NonNull final O src, @NonNull @Immutable final Object label, @NonNull final O dest);
63     }
