17     package edu.ksu.cis.indus.staticanalyses.flow;
18     
19     import edu.ksu.cis.indus.processing.Context;
20     
21     /**
22      * This interface is used to manage indices.
23      *
24      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
25      * @author $Author: rvprasad $
26      * @version $Revision: 1.5 $
27      * @param <I> is the type of the index.
28      * @param <E> is the type of the indexed entity.
29      */
30     public interface IIndexManager<I extends IIndex<I>, E> {
31     
32     	/**
33     	 * Returns the index corresponding to the given entity in the given context, if one exists. If none exist, a new index is
34     	 * created and returned.
35     	 *
36     	 * @param o the entity whose index is to be returned.
37     	 * @param c the context in which the entity's index is requested.
38     	 * @return the index corresponding to the entity in the given context.
39     	 * @pre o != null and c != null
40     	 * @post result != null
41     	 */
42     	I getIndex(final E o, final Context c);
43     
44     	/**
45     	 * Reset the manager. Flush all the internal data structures.
46     	 */
47     	void reset();
48     }
