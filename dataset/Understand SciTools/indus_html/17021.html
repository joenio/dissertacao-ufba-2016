68     /**
69      * This is a command line interface to exercise deadlock analysis.
70      * 
71      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
72      * @author $Author$
73      * @version $Revision$ $Date$
74      */
75     public class DeadlockAnalysisCLI
76     		extends SootBasedDriver {
77     
78     	/**
79     	 * The logger used by instances of this class to log messages.
80     	 */
81     	private static final Logger LOGGER = LoggerFactory.getLogger(DeadlockAnalysisCLI.class);
82     
83     	/**
84     	 * The entry point to this class.
85     	 * 
86     	 * @param args command line arguments.
87     	 * @throws RuntimeException when escape information and side-effect information calculation fails.
88     	 */
89     	public static void main(final String[] args) {
90     		final Options _options = new Options();
91     		Option _option = new Option("h", "help", false, "Display message.");
92     		_option.setOptionalArg(false);
93     		_options.addOption(_option);
94     		_option = new Option("p", "soot-classpath", false, "Prepend this to soot class path.");
95     		_option.setArgs(1);
96     		_option.setArgName("classpath");
97     		_option.setOptionalArg(false);
98     		_options.addOption(_option);
99     		_option = new Option("S", "scope", true, "The scope that should be analyzed.");
100     		_option.setArgs(1);
101     		_option.setArgName("scope");
102     		_option.setRequired(false);
103     		_options.addOption(_option);
104     
105     		final CommandLineParser _parser = new GnuParser();
106     
107     		try {
108     			final CommandLine _cl = _parser.parse(_options, args);
109     
110     			if (_cl.hasOption("h")) {
111     				final String _cmdLineSyn = "java " + DeadlockAnalysisCLI.class.getName() + " <options> <classnames>";
112     				(new HelpFormatter()).printHelp(_cmdLineSyn, _options);
113     				System.exit(1);
114     			}
115     
116     			if (_cl.getArgList().isEmpty()) {
117     				throw new MissingArgumentException("Please specify atleast one class.");
118     			}
119     
120     			final DeadlockAnalysisCLI _cli = new DeadlockAnalysisCLI();
121     
122     			if (_cl.hasOption('p')) {
123     				_cli.addToSootClassPath(_cl.getOptionValue('p'));
124     			}
125     
126     			if (_cl.hasOption('S')) {
127     				_cli.setScopeSpecFile(_cl.getOptionValue('S'));
128     			}
129     			_cli.setClassNames(_cl.getArgList());
130     			_cli.<ITokens> execute();
131     		} catch (final ParseException _e) {
132     			LOGGER.error("Error while parsing command line.", _e);
133     			System.out.println("Error while parsing command line." + _e);
134     			final String _cmdLineSyn = "java " + DeadlockAnalysisCLI.class.getName() + " <options> <classnames>";
135     			(new HelpFormatter()).printHelp(_cmdLineSyn, "Options are:", _options, "");
136     		} catch (final Throwable _e) {
137     			LOGGER.error("Beyond our control. May day! May day!", _e);
138     			throw new RuntimeException(_e);
139     		}
140     	}
141     
142     	/**
143     	 * This contains the driver logic.
144     	 * 
145     	 * @param <T> dummy type parameter.
146     	 */
147     	private <T extends ITokens<T, Value>> void execute() {
148     		setInfoLogger(LOGGER);
149     
150     		final String _tagName = "SideEffect:FA";
151     		final IValueAnalyzer<Value> _aa = OFAnalyzer.getFSOSAnalyzer(_tagName, TokenUtil
152     				.<T, Value, Type> getTokenManager(new SootValueTypeManager()), getStmtGraphFactory());
153     		final ValueAnalyzerBasedProcessingController _pc = new ValueAnalyzerBasedProcessingController();
154     		final Collection<IProcessor> _processors = new ArrayList<IProcessor>();
155     		final PairManager _pairManager = new PairManager(false, true);
156     		final CallGraphInfo _cgi = new CallGraphInfo(new PairManager(false, true));
157     		final MonitorAnalysis _monitorInfo = new MonitorAnalysis();
158     		final OFABasedCallInfoCollector _callGraphInfoCollector = new OFABasedCallInfoCollector();
159     		final IThreadGraphInfo _tgi = new ThreadGraph(_cgi, new CFGAnalysis(_cgi, getBbm()), _pairManager);
160     		final ValueAnalyzerBasedProcessingController _cgipc = new ValueAnalyzerBasedProcessingController();
161     		final OneAllStmtSequenceRetriever _ssr = new OneAllStmtSequenceRetriever();
162     
163     		_ssr.setStmtGraphFactory(getStmtGraphFactory());
164     
165     		_pc.setStmtSequencesRetriever(_ssr);
166     		_pc.setAnalyzer(_aa);
167     		_pc.setProcessingFilter(new TagBasedProcessingFilter(_tagName));
168     
169     		_cgipc.setAnalyzer(_aa);
170     		_cgipc.setProcessingFilter(new CGBasedProcessingFilter(_cgi));
171     		_cgipc.setStmtSequencesRetriever(_ssr);
172     
173     		final Map _info = new HashMap();
174     		_info.put(ICallGraphInfo.ID, _cgi);
175     		_info.put(IThreadGraphInfo.ID, _tgi);
176     		_info.put(PairManager.ID, _pairManager);
177     		_info.put(IEnvironment.ID, _aa.getEnvironment());
178     		_info.put(IValueAnalyzer.ID, _aa);
179     
180     		initialize();
181     		_aa.analyze(getEnvironment(), getRootMethods());
182     
183     		_processors.clear();
184     		_processors.add(_callGraphInfoCollector);
185     		_pc.reset();
186     		_pc.driveProcessors(_processors);
187     		_cgi.reset();
188     		_cgi.createCallGraphInfo(_callGraphInfoCollector.getCallInfo());
189     		writeInfo("CALL GRAPH:\n" + _cgi.toString());
190     
191     		_processors.clear();
192     		((ThreadGraph) _tgi).reset();
193     		_processors.add((IProcessor) _tgi);
194     		_cgipc.reset();
195     		_cgipc.driveProcessors(_processors);
196     		writeInfo("THREAD GRAPH:\n" + ((ThreadGraph) _tgi).toString());
197     		final EquivalenceClassBasedEscapeAnalysis _ecba = new EquivalenceClassBasedEscapeAnalysis(_cgi, _tgi, getBbm());
198     		final IEscapeInfo _escapeInfo = _ecba.getEscapeInfo();
199     		final AnalysesController _ac = new AnalysesController(_info, _cgipc, getBbm());
200     		_ac.addAnalyses(EquivalenceClassBasedEscapeAnalysis.ID, Collections.singleton(_ecba));
201     		_ac.addAnalyses(IMonitorInfo.ID, Collections.singleton(_monitorInfo));
202     		_ac.initialize();
203     		_ac.execute();
204     		writeInfo("END: Escape analysis");
205     
206     		final LockAcquisitionBasedEquivalence _lbe = new LockAcquisitionBasedEquivalence(_escapeInfo, _cgi);
207     		_lbe.hookup(_cgipc);
208     		_cgipc.process();
209     		_lbe.unhook(_cgipc);
210     
211     		System.out.println("Deadlock Analysis:");
212     		System.out.println("Total number of monitors: " + _monitorInfo.getMonitorTriples().size());
213     		calculateDeadlockInfo(_aa, _monitorInfo, null, null);
214     		calculateDeadlockInfo(_aa, _monitorInfo, _escapeInfo, null);
215     		calculateDeadlockInfo(_aa, _monitorInfo, null, _lbe);
216     		calculateDeadlockInfo(_aa, _monitorInfo, _escapeInfo, _lbe);
217     	}
218     
219     	/**
220     	 * Calculates deadlock information based on the given inputs.
221     	 * 
222     	 * @param aa is the OFA to be used.
223     	 * @param monitorInfo is the monitor analysis to be used.
224     	 * @param escapeInfo is the escape analysis to be used.
225     	 * @param lbe is the lock based equivalence analysis to be used.
226     	 */
227     	private void calculateDeadlockInfo(final IValueAnalyzer<Value> aa, final MonitorAnalysis monitorInfo,
228     			final IEscapeInfo escapeInfo, final LockAcquisitionBasedEquivalence lbe) {
229     		final DeadlockAnalysis _dla = new DeadlockAnalysis(monitorInfo, aa, lbe, escapeInfo);
230     		final Collection<Triple<EnterMonitorStmt, ExitMonitorStmt, SootMethod>> _deadlockingMonitors = _dla
231     				.getDeadlockingMonitors();
232     		System.out.println("Deadlocking Monitors: " + _deadlockingMonitors.size() + " -- using escapeInfo :"
233     				+ (escapeInfo != null) + " -- using locking based equiv : " + (lbe != null));
234     		for (final Triple<EnterMonitorStmt, ExitMonitorStmt, SootMethod> _m : _deadlockingMonitors) {
235     			System.out.println("\t" + _m);
236     		}
237     	}
238     }
