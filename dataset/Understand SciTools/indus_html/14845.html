105     	 *

111     			@Immutable @NonNull @NonNullContainer final Collection<SootClass> classes, @NonNull final IEnvironment env) {
112     		final Collection<SootClass> _result = new ArrayList<SootClass>(classes);
113     		final ProcessingController _pc = new ProcessingController();

117     		_pc.setEnvironment(env);
118

120     		_ece.hookup(_pc);
121     		_pc.process();
122     		_ece.unhook(_pc);
123     		_pc.reset();
124     
125     		@SuppressWarnings("unchecked") final Iterator<SootClass> _j = classes.iterator();
126     		final int _jEnd = classes.size();
127     
128     		for (int _jIndex = 0; _jIndex < _jEnd; _jIndex++) {
129     			final SootClass _o = _j.next();
130

139     	 * Returns the class, starting from the given class and above it in the class hierarchy, that declares the given method.
140     	 * 
141     	 * @param sc the class from which to start the search in the class hierarchy. This parameter cannot be <code>null</code>.
142     	 * @param sm the method to search for in the class hierarchy. This parameter cannot be <code>null</code>.
143     	 * @return the <code>SootMethod</code> corresponding to the implementation of <code>sm</code>.
144     	 * @throws IllegalStateException if <code>sm</code> is not available in the given branch of the class hierarchy.
145     	 */
146     	@Functional @NonNull public static SootMethod findDeclaringMethod(@NonNull final SootClass sc,
147     			@NonNull final SootMethod sm) {
148     		final SootMethod _result;
149     
150     		if (sc.declaresMethod(sm.getName(), sm.getParameterTypes(), sm.getReturnType())) {
151     			_result = sc.getMethod(sm.getName(), sm.getParameterTypes(), sm.getReturnType());
152     		} else if (hasSuperclass(sc)) {
153     			_result = findDeclaringMethod(sc.getSuperclass(), sm);
154     		} else {
155     			throw new IllegalStateException("Method " + sm + " not available in class " + sc + ".");
156     		}
157     		return _result;
158     	}

201     		Collection<SootMethod> _result = new HashSet<SootMethod>();

203     
204     		@SuppressWarnings("unchecked") final List<Type> _parameterTypes = method.getParameterTypes();
205     		final Type _retType = method.getReturnType();
206     		final String _methodName = method.getName();
207     
208     		while (_toProcess.hasWork()) {
209     			final SootClass _sc = _toProcess.getWork();
210     
211     			if (_sc.declaresMethod(_methodName, _parameterTypes, _retType)) {
212     				_result.add(_sc.getMethod(_methodName, _parameterTypes, _retType));

216     				final SootClass _superClass = _sc.getSuperclass();
217     
218     				_toProcess.addWorkNoDuplicates(_superClass);
219     			}
220     
221     			for (@SuppressWarnings("unchecked") final Iterator<SootClass> _i = _sc.getInterfaces().iterator(); _i.hasNext();) {
222     				final SootClass _interface = _i.next();
223     
224     				_toProcess.addWorkNoDuplicates(_interface);
225     			}
226     		}

228     		if (_result.isEmpty()) {
229     			_result = Collections.emptyList();
230     		}
231     		return _result;
232     	}

235     	 * Fixes the body of <code>java.lang.Thread.start()</code> (only if it is native) to call <code>run()</code> on the
236     	 * target or self. This is required to complete the call graph. This leaves the body untouched if it is not native.
237     	 * 
238     	 * @param scm is the scene in which the alteration occurs.
239     	 */
240     	@SuppressWarnings("unchecked") @Immutable public static void fixupThreadStartBody(@NonNull final Scene scm) {
241     		final SootClass _declClass = scm.getSootClass("java.lang.Thread");
242     
243     		if (_declClass != null) {
244     			final SootMethod _sm = _declClass.getMethodByName("start");
245     
246     			if (_sm != null) {
247     				JimpleBody _threadStartBody = null;
248     
249     				if (_sm.isConcrete()) {
250     					_threadStartBody = (JimpleBody) _sm.retrieveActiveBody();
251     				}
252     
253     				if (_threadStartBody == null) {
254     					_declClass.setApplicationClass();
255     					_declClass.setPhantom(false);
256     					_sm.setModifiers(_sm.getModifiers() & ~Modifier.NATIVE);
257     					_sm.setPhantom(false);
258     
259     					final Jimple _jimple = Jimple.v();
260     					_threadStartBody = _jimple.newBody(_sm);
261     					final PatchingChain _sl = _threadStartBody.getUnits();
262     					final Local _thisRef = _jimple.newLocal("$this", RefType.v(_declClass.getName()));
263     					_threadStartBody.getLocals().add(_thisRef);
264     					// adds $this := @this;
265     					_sl.addFirst(_jimple.newIdentityStmt(_thisRef, _jimple.newThisRef(RefType.v(_declClass))));
266     
267     					// adds $this.virtualinvoke[java.lang.Thread.run()]:void;
268     					final VirtualInvokeExpr _ve = _jimple.newVirtualInvokeExpr(_thisRef, _declClass.getMethodByName("run"),
269     							Collections.EMPTY_LIST);
270     					_sl.addLast(_jimple.newInvokeStmt(_ve));
271     					_sl.addLast(_jimple.newReturnVoidStmt());
272     					_sm.setActiveBody(_threadStartBody);
273     
274     					if (LOGGER.isWarnEnabled()) {
275     						LOGGER.warn("Fixed up java.lang.Thread.start() body");
276     					}
277     				}
278     			}
279     		}
