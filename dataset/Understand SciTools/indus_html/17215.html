45     /**
46      * This class provides synchronization dependency information. This implementation refers to the technical report <a
47      * href="http://www.cis.ksu.edu/santos/papers/technicalReports.html">A Formal Study of Slicing for Multi-threaded Program with
48      * JVM Concurrency Primitives"</a>.
49      * <p>
50      * <i>Synchronization dependence</i>: All non-monitor statement in a method are synchronization dependent on the immediately
51      * enclosing monitor statements in the same method.
52      * </p>
53      * <p>
54      * In case of synchronized methods, the statements in the method not enclosed by monitor statements are dependent on the entry
55      * and exit into the method which is tied to the call-sites. Hence, <code>getDependents()</code> and
56      * <code>getDependees()</code> do not include this dependence as it is application specific and can be derived from the
57      * control-flow. If the return points and entry point are assumed to comprise the monitor then there may be more than one
58      * monitor pair as there are many return points, hence, not all statements in the method may be dependent on the same monitor
59      * pair.
60      * </p>
61      * 
62      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
63      * @author $Author: rvprasad $
64      * @version $Revision: 1.67 $
65      */
66     public final class SynchronizationDA
67     		extends
68     		AbstractDependencyAnalysis<Stmt, SootMethod, MonitorStmt, SootMethod, Map<Stmt, Collection<MonitorStmt>>, MonitorStmt, SootMethod, Stmt, SootMethod, Map<MonitorStmt, Collection<Stmt>>> {

96     	 * @see edu.ksu.cis.indus.staticanalyses.dependency.AbstractDependencyAnalysis#analyze()
97     	 */
98     	@Override public void analyze() {
99     		unstable();
100     
101     		if (monitorInfo.isStable()) {
102     			if (LOGGER.isInfoEnabled()) {
103     				LOGGER.info("BEGIN: Synchronization Dependence processing");
104     			}
105     
106     			stable();
107     
108     			if (LOGGER.isDebugEnabled()) {
109     				LOGGER.debug("analyze() - " + toString());
110     			}
111     
112     			if (LOGGER.isInfoEnabled()) {
113     				LOGGER.info("END: Synchronization Dependence processing");
114     			}
115     		} else {
116     			if (LOGGER.isInfoEnabled()) {
117     				LOGGER.info("Monitor Info is unstable. So, passing up .");
118     			}
119     		}
120     	}

122     	/**
123     	 * Returns the enter and exit monitor statements on which the given statement is dependent on in the given method.
124     	 * 
125     	 * @param dependentStmt is a statement in the method.
126     	 * @param method in which <code>dependentStmt</code> occurs.
127     	 * @return a collection of enter and exit monitor statements.
128     	 * @see edu.ksu.cis.indus.staticanalyses.dependency.AbstractDependencyAnalysis#getDependees(java.lang.Object,
129     	 *      java.lang.Object)
130     	 */
131     	public Collection<MonitorStmt> getDependees(final Stmt dependentStmt, final SootMethod method) {
132     		final Map<Stmt, Collection<MonitorStmt>> _temp = MapUtils.getMapFromMap(dependent2dependee, method);
133     		Collection<MonitorStmt> _result = _temp.get(dependentStmt);
134     
135     		if (_result == null) {
136     			_result = new HashSet<MonitorStmt>(monitorInfo.getEnclosingMonitorStmts(dependentStmt, method, false));
137     
138     			if (dependentStmt instanceof MonitorStmt) {
139     				final Collection<Triple<EnterMonitorStmt, ExitMonitorStmt, SootMethod>> _monitorTriples = monitorInfo
140     						.getMonitorTriplesFor((MonitorStmt) dependentStmt, method);
141     
142     				for (final Iterator<Triple<EnterMonitorStmt, ExitMonitorStmt, SootMethod>> _i = _monitorTriples.iterator(); _i
143     						.hasNext();) {
144     					final Triple<EnterMonitorStmt, ExitMonitorStmt, SootMethod> _triple = _i.next();
145     					final EnterMonitorStmt _enter = _triple.getFirst();
146     					final ExitMonitorStmt _exit = _triple.getSecond();
147     
148     					if (_enter.equals(dependentStmt) || _exit.equals(dependentStmt)) {
149     						_result.add(_enter);
150     						_result.add(_exit);
151     					}
152     				}
153     			}
154     			_temp.put(dependentStmt, _result);
155     		}
156     		return _result;
157     	}
158     
159     	/**
160     	 * Returns the statements which depend on the given enter or exit monitor statement in the given method.
161     	 * 
162     	 * @param dependeeStmt is the enter or exit monitor statement.
163     	 * @param method in which<code>dependeeStmt</code> occurs.
164     	 * @return a collection of statements.
165     	 * @see edu.ksu.cis.indus.staticanalyses.dependency.AbstractDependencyAnalysis#getDependees(java.lang.Object,
166     	 *      java.lang.Object)
167     	 */
168     	public Collection<Stmt> getDependents(final MonitorStmt dependeeStmt, final SootMethod method) {
169     		final Map<MonitorStmt, Collection<Stmt>> _temp = MapUtils.getMapFromMap(dependee2dependent, method);
170     		Collection<Stmt> _result = _temp.get(dependeeStmt);
171     
172     		if (_result == null) {
173     			final Collection<Triple<EnterMonitorStmt, ExitMonitorStmt, SootMethod>> _monitors = monitorInfo
174     					.getMonitorTriplesFor(dependeeStmt, method);
175     			_result = new HashSet<Stmt>();
176     
177     			final Iterator<Triple<EnterMonitorStmt, ExitMonitorStmt, SootMethod>> _i = _monitors.iterator();
178     			final int _iEnd = _monitors.size();
179     
180     			for (int _iIndex = 0; _iIndex < _iEnd; _iIndex++) {
181     				final Triple<EnterMonitorStmt, ExitMonitorStmt, SootMethod> _monitor = _i.next();
182     				final EnterMonitorStmt _enter = _monitor.getFirst();
183     				final ExitMonitorStmt _exit = _monitor.getSecond();
184     
185     				if (dependeeStmt.equals(_enter) || dependeeStmt.equals(_exit)) {
186     					final Collection<Stmt> _enclosedStmts = monitorInfo.getEnclosedStmts(_monitor, false);
187     					_result.addAll(_enclosedStmts);
188     					_result.add(_enter);
189     					_result.add(_exit);
190     				}
191     			}
192     			_temp.put(dependeeStmt, _result);
193     		}
194     		return _result;
195     	}

230     
231     			public Collection<Pair<MonitorStmt, SootMethod>> convertToConformantDependees(final Collection<Stmt> dependents,
232     					@SuppressWarnings("unused") final MonitorStmt base, final SootMethod context) {
233     				final Collection<Pair<MonitorStmt, SootMethod>> _result = new HashSet<Pair<MonitorStmt, SootMethod>>();
234     				for (final Stmt _t2 : dependents) {
235     					if (_t2 instanceof MonitorStmt) {
236     						_result.add(new Pair<MonitorStmt, SootMethod>((MonitorStmt) _t2, context));
237     					}
238     				}
239     				return _result;
240     			}
241     
242     			public Collection<Pair<Stmt, SootMethod>> convertToConformantDependents(final Collection<MonitorStmt> dependees,
243     					@SuppressWarnings("unused") final Stmt base, final SootMethod context) {
244     				final Collection<Pair<Stmt, SootMethod>> _result = new HashSet<Pair<Stmt, SootMethod>>();
245     				for (final Stmt _e1 : dependees) {
246     					_result.add(new Pair<Stmt, SootMethod>(_e1, context));
247     				}
248     				return _result;
249     			}
250     
251     		};
