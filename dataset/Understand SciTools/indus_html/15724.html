17     package edu.ksu.cis.indus.staticanalyses.dependency.direction;
18     
19     import edu.ksu.cis.indus.common.soot.BasicBlockGraph.BasicBlock;
20     
21     import java.util.Collection;
22     import java.util.List;
23     
24     import soot.jimple.Stmt;
25     
26     /**
27      * This class provides information to drive the analysis to generate information that is forward in direction.
28      * 
29      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
30      * @author $Author: rvprasad $
31      * @version $Revision: 1.6 $ $Date: 2007/02/10 19:07:49 $
32      */
33     public final class ForwardDirectionSensitiveInfo
34     		implements IDirectionSensitiveInfo {
35     
36     	/**
37     	 * @see IDirectionSensitiveInfo#getFirstStmtInBB(edu.ksu.cis.indus.common.soot.BasicBlockGraph.BasicBlock)
38     	 */
39     	public Stmt getFirstStmtInBB(final BasicBlock bb) {
40     		return bb.getTrailerStmt();
41     	}
42     
43     	/**
44     	 * @see IDirectionSensitiveInfo#getFollowersOfBB(edu.ksu.cis.indus.common.soot.BasicBlockGraph.BasicBlock)
45     	 */
46     	public Collection<BasicBlock> getFollowersOfBB(final BasicBlock bb) {
47     		return bb.getPredsOf();
48     	}
49     
50     	/**
51     	 * {@inheritDoc} This provides information for forward direciton.
52     	 */
53     	public List<Stmt> getIntraBBDependents(final BasicBlock bb, final Stmt stmt) {
54     		final List<Stmt> _stmtsFrom = bb.getStmtsFromTo(bb.getLeaderStmt(), stmt);
55     		_stmtsFrom.remove(stmt);
56     		return _stmtsFrom;
57     	}
58     }
