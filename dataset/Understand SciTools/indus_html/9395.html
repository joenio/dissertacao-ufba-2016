115     	 * 
116     	 * @param methods to be analyzed.
117     	 * @pre methods != null and not method->includes(null)
118     	 */
119     	public void analyze(final Collection<SootMethod> methods) {
120     		unstable();
121     
122     		for (final Iterator<SootMethod> _i = methods.iterator(); _i.hasNext();) {
123     			final SootMethod _currMethod = _i.next();

129     			}

132     				LOGGER.debug("Processing method: " + _currMethod.getSignature());
133     			}
134

138

140     		nodesWithChildrenCache = null;
141     		pairMgr.reset();

144     			LOGGER.debug("analyze() - " + toString());

195     	 * 
196     	 * @param tokenSets is the collection of token sets of the nodes in the graph.
197     	 * @return an array of bitsets. The length of the array and each of the bitset in it is equal to the number of nodes in

210     			final BasicBlock _controlPoint = _i.next();
211     			final int _cpIndex = nodesCache.indexOf(_controlPoint);
212     			final int _succsSize = getFanoutNumOf(_controlPoint);
213     
214     			for (int _k = nodesCache.size() - 1; _k >= 0; _k--) {
215     				final BitSet _tokens = tokenSets[_k][_cpIndex];

217     				if (_tokens != null) {

237     	 * Calculates the control dependency from a basic block graph. This calculates the dependence information in terms of
238     	 * nodes in the graph. This is later translated to statement level information by {@link
239     	 * NonTerminationSensitiveEntryControlDA#fixupMaps(BitSet[], SootMethod) fixupMaps}.
240     	 * 
241     	 * @param graph for which control dependency should be calculated.
242     	 * @return an array of bitsets. The length of the array and each of the bitset in it is equal to the number of nodes in
243     	 *         the graph. The nth bitset captures the dependence information via set bits. The BitSets capture
244     	 *         dependent->dependee information.
245     	 * @post result->size() == graph.getNodes().size()
246     	 * @post result->forall(o | o.size() == graph.getNodes().size())
247     	 */

249     		nodesCache = graph.getNodes();
250     		nodesWithChildrenCache = getNodesWithChildren();
251     
252     		final int _size = nodesCache.size();
253     		final BitSet[][] _tokenSets = new BitSet[_size][_size];
254     
255     		for (int _i = _size - 1; _i >= 0; _i--) {
256     			for (int _j = _size - 1; _j >= 0; _j--) {
257     				_tokenSets[_i][_j] = new BitSet();
258     			}
259     		}
260     
261     		final IWorkBag<Pair<BasicBlock, Boolean>> _wb = injectTokensAndGenerateWorkForTokenPropagation(_tokenSets);
262     
263     		while (_wb.hasWork()) {
264     			final Pair<BasicBlock, Boolean> _pair = _wb.getWork();
265     			_wb.addAllWorkNoDuplicates(processNode(_pair, _tokenSets));
266     
267     			if (LOGGER.isDebugEnabled()) {
268     				final StringBuffer _msg = new StringBuffer();
269     
270     				for (int _i = _size - 1; _i >= 0; _i--) {
271     					_msg.append("{");
272     
273     					for (int _j = _size - 1; _j >= 0; _j--) {
274     						_msg.append(_tokenSets[_i][_j].toString() + ",");
275     					}
276     					_msg.append("}");
277     				}
278     
279     				LOGGER.debug("computeControlDependency() - Node being processed : _node = " + _pair.getFirst()
280     						+ "nodeIndex = " + nodesCache.indexOf(_pair.getFirst()) + " _tokenSets = " + _msg);
281     			}
282     		}
283     
284     		return calculateCDFromTokenInfo(_tokenSets);
285     	}

289     	 * to the same ancestors at the node at <code>dest</code>.
290     	 * 
291     	 * @param src is the index of the node whose ancestor's tokens need to be propagated.
292     	 * @param dest is the index of the node into which the tokens will be propagated to.
293     	 * @param tokenSets is the collection of token sets of the nodes in the graph.
294     	 * @return <code>true</code>if bits/tokens were added to the destination node; <code>false</code>, otherwise.
295     	 * @pre tokenSets != null

297     	 * @pre 0 &lt;= dest &lt; tokensSets.length
298     	 */

300     		boolean _result = false;
301     		final BitSet _temp = new BitSet();
302     		final BitSet[] _srcBitSets = tokenSets[src];
303     		final BitSet[] _destBitSets = tokenSets[dest];
304     		final Iterator<BasicBlock> _i = nodesWithChildrenCache.iterator();
305     		final int _iEnd = nodesWithChildrenCache.size();
306     
307     		for (int _iIndex = 0; _iIndex < _iEnd; _iIndex++) {
308     			final BasicBlock _ancestor = _i.next();

314     				_temp.clear();
315     				_temp.or(_srcsAncestorBitSet);
316     				_temp.andNot(_destAndAncestorBitSet);

325     	 * Translates the dependence information as captured in <code>bbCDBitSets</code> to statement level info and populates
326     	 * the dependeXXMap fields.
327     	 * 
328     	 * @param bbCDBitSets is the array that contains the basic block level dependence information as calculated by {@link
329     	 *            #computeControlDependency(BasicBlockGraph) computeControlDependency}.
330     	 * @param method for which the maps are being populated.
331     	 * @pre graph != null and bbCDBitSets != null and method != null
332     	 * @post dependee2dependent.get(method) != null
333     	 * @post dependee2dependent.values()->forall(o | o->forall(p | p != null()))
334     	 * @post dependent2dependee.get(method) != null
335     	 * @post dependent2dependee.values()->forall(o | o->forall(p | p != null()))
336     	 */
337     	private void fixupMaps(final BitSet[] bbCDBitSets, final SootMethod method) {
338     		final List<Stmt> _sl = getStmtList(method);
339     		final List<Collection<Stmt>> _mDependee = new ArrayList<Collection<Stmt>>();
340     		final List<Collection<Stmt>> _mDependent = new ArrayList<Collection<Stmt>>();
341     
342     		for (int _i = _sl.size(); _i > 0; _i--) {
343     			_mDependee.add(null);
344     			_mDependent.add(null);
345     		}
346     
347     		boolean _flag = false;
348     
349     		for (int _i = bbCDBitSets.length - 1; _i >= 0; _i--) {
350     			final BitSet _cd = bbCDBitSets[_i];
351     			_flag |= _cd != null;
352     
353     			if (_cd != null) {

356     
357     				for (final Iterator<Stmt> _j = _bb.getStmtsOf().iterator(); _j.hasNext();) {
358     					_mDependee.set(_sl.indexOf(_j.next()), _cdp);

361     				for (int _j = _cd.nextSetBit(0); _j != -1; _j = _cd.nextSetBit(_j + 1)) {
362     					final BasicBlock _cdbb = nodesCache.get(_j);

366     					final int _deIndex = _sl.indexOf(_cdStmt);
367     					Collection<Stmt> _dees = _mDependent.get(_deIndex);
368     
369     					if (_dees == null) {
370     						_dees = new ArrayList<Stmt>();
371     						_mDependent.set(_deIndex, _dees);
372     					}
373     					_dees.addAll(_bb.getStmtsOf());
374     				}
375     			}
376     		}
377     
378     		if (_flag) {
379     			dependee2dependent.put(method, new ArrayList<Collection<Stmt>>(_mDependent));
380     			dependent2dependee.put(method, new ArrayList<Collection<Stmt>>(_mDependee));
381     		} else {

383     			dependent2dependee.put(method, null);
384     		}

442     	/**
443     	 * Processes the given node. Basically, it propagates the tokens to it's successor and returns the successor whose token
444     	 * sets were modified.
445     	 * 
446     	 * @param pair to be processed.
447     	 * @param tokenSets is the collection of token sets of the nodes in the graph. The first subscript is the index of the
448     	 *            dependent basic block in the sequence of basic blocks. The second subscript is the index of the control
449     	 *            point basic block in the sequence of basic blocks. The bit set at these subscript indicate the number of
450     	 *            tokens (corresponding to the successors of the control point) that have been accumulated at the dependent
451     	 *            basic block.
452     	 * @return the collection of nodes whose token sets were modified.
453     	 * @pre pair != null and tokenSets != null
454     	 * @pre 0 &lt;= graphCache.getNumOfSuccsOf(pair.getFirst()) &lt;= 1
455     	 * @post result != null
456     	 * @post pair.getFirst().getSuccsOf().containsAll(result)
457     	 */

462     
463     		Collection<Pair<BasicBlock, Boolean>> _result = Collections.emptySet();
464     
465     		if (_addedDueToTokePropagation || _accumlatedTokens) {
466     			final int _size = getFanoutNumOf(_node);
467     
468     			if (_size == 1) {
469     				_result = processNodeWithOneChild(_node, tokenSets);
470     			} else if (_size > 1) {
471     				final int _nodeIndex = nodesCache.indexOf(_node);
472     				_result = new HashSet<Pair<BasicBlock, Boolean>>();
473     
474     				final int _iEnd = nodesCache.size();
475     
476     				for (int _iIndex = 0; _iIndex < _iEnd; _iIndex++) {
477     					final BitSet _bitSet = tokenSets[_iIndex][_nodeIndex];
478     
479     					if (_bitSet.cardinality() == _size) {
480     						_result.add(pairMgr.getPair(nodesCache.get(_iIndex), Boolean.valueOf(_accumlatedTokens)));
481     					}
482     				}
483     			}
484     		}
485     		return _result;
486     	}

489     	 * Processes the given node. Basically, it propagates the tokens to it's successor and returns the successor whose token
490     	 * sets were modified.
491     	 * 
492     	 * @param node to be processed.
493     	 * @param tokenSets is the collection of token sets of the nodes in the graph. The first subscript is the index of the
494     	 *            dependent basic block in the sequence of basic blocks. The second subscript is the index of the control
495     	 *            point basic block in the sequence of basic blocks. The bit set at these subscript indicate the number of
496     	 *            tokens (corresponding to the successors of the control point) that have been accumulated at the dependent
497     	 *            basic block.
498     	 * @return the collection of nodes whose token sets were modified.
499     	 * @pre node != null and tokenSets != null and 0 &lt;= node.getSuccsOf().size() &lt;= 1
500     	 * @post result != null
501     	 * @post node.getSuccsOf().containsAll(result)
502     	 */
503     	private Collection<Pair<BasicBlock, Boolean>> processNodeWithOneChild(final BasicBlock node, final BitSet[][] tokenSets) {
504     		boolean _addflag = false;
505     		final int _nodeIndex = nodesCache.indexOf(node);
506     		final BitSet[] _nodeBitSets = tokenSets[_nodeIndex];
507     		final BasicBlock _succ = node.getSuccsOf().iterator().next();
508     		final int _succIndex = nodesCache.indexOf(_succ);
509     		final BitSet[] _succBitSets = tokenSets[_succIndex];
510     		final BitSet _temp = new BitSet();
511     		final Iterator<BasicBlock> _i = nodesWithChildrenCache.iterator();
512     		final int _iEnd = nodesWithChildrenCache.size();
513     
514     		for (int _iIndex = 0; _iIndex < _iEnd; _iIndex++) {
515     			final BasicBlock _ancestor = _i.next();
516     			final int _ancIndex = nodesCache.indexOf(_ancestor);

518     			final BitSet _succAncestorTokenSet = _succBitSets[_ancIndex];
519     
520     			_temp.clear();
521     			_temp.or(_nodeAncestorTokenSet);
