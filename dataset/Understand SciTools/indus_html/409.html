18     package edu.ksu.cis.indus.staticanalyses.flow.instances.ofa;

20     import edu.ksu.cis.indus.common.soot.IStmtGraphFactory;
21     import edu.ksu.cis.indus.staticanalyses.flow.ASTVariantManager;
22     import edu.ksu.cis.indus.staticanalyses.flow.FA;

27     import java.util.regex.Pattern;

29     import soot.SootMethod;
30     import soot.Type;
31     import soot.Value;

33     
34     /**
35      * This implementation creates instances of <code>MethodVariant</code>.
36      *
37      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
38      * @author $Author: rvprasad $
39      * @version $Revision: 1.8 $ $Date: 2007/02/10 19:07:03 $
40      * @param <T>  is the type of the token set object.
41      */
42     class MethodVariantFactory<T extends ITokens<T, Value>>
43       implements IMethodVariantFactory<Value, T, OFAFGNode<T>, Type> {
44     	/**
45     	 * The pattern used to decide if a stub variant or a complete variant needs to be returned during <code>create()</code>
46     	 * call.
47     	 */
48     	private final Pattern pattern;
49     
50     	/**
51     	 * The statement graph to use to retrieve method bodies.
52     	 */
53     	private final IStmtGraphFactory<?> stmtGraphFactory;
54     
55     	/**
56     	 * Creates an instance of this class.
57     	 *
58     	 * @param actualBodyScopePattern the scope in which the variants are based on actual body.
59     	 * @param factory provides the statement graphs to be used construct method variants.
60     	 * @pre factory != null
61     	 */
62     	public MethodVariantFactory(final String actualBodyScopePattern, final IStmtGraphFactory<?> factory) {
63     		if (actualBodyScopePattern != null) {
64     			pattern = Pattern.compile(actualBodyScopePattern);
65     		} else {
66     			pattern = null;
67     		}
68     		stmtGraphFactory = factory;
69     	}
70     
71     	/**
72     	 * @see edu.ksu.cis.indus.staticanalyses.flow.IMethodVariantFactory#create(soot.SootMethod,
73     	 * 		edu.ksu.cis.indus.staticanalyses.flow.ASTVariantManager, edu.ksu.cis.indus.staticanalyses.flow.FA)
74     	 */
75     	public IMethodVariant<OFAFGNode<T>> create(final SootMethod sootMethod,
76     			final ASTVariantManager<Value, T, OFAFGNode<T>, Type> astVM, final FA<Value, T, OFAFGNode<T>, Type> fa) {
77     		final IMethodVariant<OFAFGNode<T>> _result;
78     
79     		if (pattern == null || pattern.matcher(sootMethod.getDeclaringClass().getName()).matches()) {
80     			_result = new MethodVariant<T>(sootMethod, astVM, fa, stmtGraphFactory);
81     		} else {
82     			_result = new StubMethodVariant<T>(sootMethod, astVM, fa);
83     		}
84     		return _result;
85     	}
86     }
