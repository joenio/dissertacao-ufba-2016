19     import org.eclipse.swt.events.DisposeEvent;

21     
22     import org.eclipse.swt.widgets.Composite;

24     /**
25      * This is the interface exposed by the tool for configuring it via GUI.
26      * 
27      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
28      * @author $Author: rvprasad $
29      * @version $Revision: 1.5 $
30      */
31     public interface IToolConfigurator
32     		extends DisposeListener {
33     
34     	/**
35     	 * Initializes the configurator with the given composite on which it should provide the UI.
36     	 * 
37     	 * @param composite on which the UI is provided.
38     	 * @pre composite != null
39     	 */
40     	void initialize(final Composite composite);
41     
42     	/**
43     	 * Sets the configuration to be configured.
44     	 * 
45     	 * @param toolConfiguration is the configuration to be edited.
46     	 * @pre toolConfiguration != null
47     	 */
48     	void setConfiguration(final IToolConfiguration toolConfiguration);
49     
50     	/**
51     	 * Called when the parent widget is disposed. Subclasses should override this method appropriately.
52     	 * 
53     	 * @see org.eclipse.swt.events.DisposeListener#widgetDisposed(org.eclipse.swt.events.DisposeEvent)
54     	 */
55     	void widgetDisposed(final DisposeEvent evt);
56     }
