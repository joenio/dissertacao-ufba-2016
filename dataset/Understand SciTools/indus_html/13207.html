17     package edu.ksu.cis.indus.staticanalyses.flow.instances.ofa;
18     
19     import edu.ksu.cis.indus.processing.Context;
20     import edu.ksu.cis.indus.staticanalyses.flow.AbstractTokenProcessingWork;
21     import edu.ksu.cis.indus.staticanalyses.flow.IMethodVariant;
22     import edu.ksu.cis.indus.staticanalyses.tokens.ITokens;
23     
24     import soot.Value;
25     import soot.ValueBox;
26     
27     /**
28      * This class encapsulates the logic and data related to work to be done in correspondence to access expressions.
29      * 
30      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
31      * @version $Revision: 1.17 $
32      * @param <T> is the type of the token set object.
33      */
34     abstract class AbstractAccessExprWork<T extends ITokens<T, Value>>
35     		extends AbstractTokenProcessingWork<T> {
36     
37     	/**
38     	 * The context in which the access occurs.
39     	 *
40     	 * @invariant context != null
41     	 */
42     	protected final Context context;
43     
44     	/**
45     	 * The method in which the access occurs.
46     	 *
47     	 * @invariant caller != null
48     	 */
49     	protected final IMethodVariant<OFAFGNode<T>> caller;
50     
51     	/**
52     	 * The program point at which the entity occurs.
53     	 *
54     	 * @invariant accessExprBox != null
55     	 */
56     	protected final ValueBox accessExprBox;
57     
58     	/**
59     	 * Creates a new <code>AbstractAccessExprWork</code> instance.
60     	 *
61     	 * @param callerMethod the method in which the access expression occurs.
62     	 * @param accessContext the context in which the access occurs.
63     	 * @param tokenSet to be used by this work object to store the tokens whose flow should be instrumented.
64     	 * @pre callerMethod != null and accessContext != null and tokenSet != null
65     	 */
66     	protected AbstractAccessExprWork(final IMethodVariant<OFAFGNode<T>> callerMethod, final Context accessContext,
67     			final T tokenSet) {
68     		super(tokenSet);
69     		accessExprBox = accessContext.getProgramPoint();
70     		caller = callerMethod;
71     		context = accessContext.clone();
72     	}
73     }
