17     package edu.ksu.cis.indus.interfaces;
18     
19     import edu.ksu.cis.indus.annotations.NonNull;
20     import edu.ksu.cis.indus.annotations.NonNullContainer;
21     
22     import java.util.Collection;
23     
24     import soot.Local;
25     import soot.SootMethod;
26     import soot.jimple.DefinitionStmt;
27     import soot.jimple.Stmt;
28     
29     /**
30      * This interface will be used to retrieve use-def information of a system.
31      * <p>
32      * Subtypes of this class have to return the constant <code>ID</code> defined in this class as a result of
33      * <code>getId</code>.
34      * </p>
35      * 
36      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
37      * @author $Author: rvprasad $
38      * @version $Revision: 1.13 $
39      * @param <D> is the type of the definition AST nodes.
40      * @param <U> is the type of the use AST nodes.
41      */
42     public interface IUseDefInfo<D, U>
43     		extends IStatus, IIdentification {
44     
45     	/**
46     	 * This is an ID of this interface. This is used in conjuction with instance-based reference use-def info.
47     	 */
48     	Comparable<String> ALIASED_USE_DEF_ID = "Aliased Use-Def Information";
49     
50     	/**
51     	 * This is an ID of this interface. This is used in conjuction with class-based reference use-def info.
52     	 */
53     	Comparable<String> GLOBAL_USE_DEF_ID = "Global Use-Def Information";
54     
55     	/**
56     	 * This is an ID of this interface. This is used in conjuction with method local variable use-def info.
57     	 */
58     	Comparable<String> LOCAL_USE_DEF_ID = "Local Use-Def Information";
59     
60     	/**
61     	 * Retrieves the def sites that reach the given use site in the given context.
62     	 * 
63     	 * @param useStmt is the statement containing the use site.
64     	 * @param method in which the use-site occurs.
65     	 * @return a collection of def sites.
66     	 */
67     	@NonNull @NonNullContainer Collection<D> getDefs(@NonNull Stmt useStmt, @NonNull SootMethod method);
68     
69     	/**
70     	 * Retrieves the def sites that reach the given local at the given use site in the given context.
71     	 * 
72     	 * @param local for which the definition is requested.
73     	 * @param useStmt is the statement containing the use site.
74     	 * @param method in which the use-site occurs.
75     	 * @return a collection of def sites.
76     	 * @throws UnsupportedOperationException when the implementation does not support this operation.
77     	 */
78     	@NonNull @NonNullContainer Collection<D> getDefs(@NonNull Local local, @NonNull Stmt useStmt, @NonNull SootMethod method)
79     			throws UnsupportedOperationException;
80     
81     	/**
82     	 * Retrieves the use sites that reach the given def site in the given context.
83     	 * 
84     	 * @param defStmt is the statement containing the def site.
85     	 * @param method in which the def-site occurs.
86     	 * @return a collection of use sites.
87     	 */
88     	@NonNull @NonNullContainer Collection<U> getUses(@NonNull DefinitionStmt defStmt, @NonNull SootMethod method);
89     }
