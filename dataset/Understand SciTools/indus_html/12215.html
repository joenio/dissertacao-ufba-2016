54     /**
55      * This is generic driver that provides basic support to process a system represented in Jimple.
56      * <p>
57      * The classes in which to search for root methods can be configured via setting 1 system property named
58      * <code>indus.common.soot.SootBasedDriver.rootClasses</code>. The syntax of this property a regular expression which will
59      * be matched against the name of the class in which the method is declared. Note if this is unspecified, then root methods
60      * are searched only in the classes specified and not the classes which are required. So, "edu.ksu.cis.indus.Processing" will
61      * find root methods occurring in classes defined in class <code>edu.ksu.cis.indus.Processing</code>. Likewise, the user
62      * can control which methods are picked as root methods based on their signature by specifying the pattern via the system
63      * property, <code>indus.common.soot.SootBasedDriver.rootMethods</code>.
64      * </p>
65      * <p>
66      * The user can provide a root method trapper object to be used to identify root methods. However, if the user does not
67      * provide one, then an instance of the class named via <code>indus.common.soot.RootMethodTrapper.class</code> property will
68      * be used. This named class should be a subclass of <code>edu.ksu.cis.indus.common.soot.RootMethodTrapper</code>. As
69      * reflection is used on to instantiate an object, the specified class by the property should have a no-argument constructor.
70      * </p>
71      * <p>
72      * The user can provide a statement graph factory object to be used to identify root methods. However, if the user does not
73      * provide one, then an instance of the class named via <code>indus.common.soot.SootBasedDriver.StmtGraphFactory.class</code>
74      * property will be used. This named class should be a subclass of
75      * <code>edu.ksu.cis.indus.common.soot.IStmtGraphFactory</code>. As reflection is used on to instantiate an object, the
76      * specified class by the property should have a no-argument constructor.
77      * </p>
78      * <p>
79      * Please refer to <code>edu.ksu.cis.indus.Constants</code> for a file-based approach to specifying these properties.
80      * </p>
81      * 
82      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
83      * @author $Author: rvprasad $
84      * @version $Revision: 1.54 $ $Date: 2007/02/10 19:08:36 $
85      */

462     	 * Loads up the classes specified via <code>setClassNames()</code> and also collects the possible entry points into the
463     	 * system being analyzed. All <code>public static void main()</code> methods defined in <code>public</code> classes

465     	 * <code>addToSootClassPath</code>.

467     	 * @param options to be used while setting up Soot infrastructure.
468     	 * @return a soot scene that provides the classes to be analyzed.
469     	 */

477     		} else {

479     		}

482     		for (final Iterator<String> _i = classNames.iterator(); _i.hasNext();) {
483     			final SootClass _sc = _result.loadClassAndSupport(_i.next());

485     		}

487     		final Collection<SootClass> _mc = new HashSet<SootClass>();
488     		_mc.addAll(_result.getClasses());

499     			final SootClass _sc = _i.next();

501     			if (_rmt.considerClassForEntryPoint(_sc)) {

503

511     			}
512     		}
513     		Util.fixupThreadStartBody(_result);
514     
515     		if (Constants.shouldLoadMethodBodiesDuringInitialization()) {
516     			loadupMethodBodies();
517     		}

519     		return _result;
520     	}
