17     package edu.ksu.cis.indus.interfaces;
18     
19     import java.util.Collection;
20     
21     import soot.SootClass;
22     import soot.SootMethod;
23     
24     /**
25      * This interface exposes the information pertaining to the environment in which the analyses function. It provides the
26      * non-functional information about the system.
27      *
28      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
29      * @author $Author: rvprasad $
30      * @version $Revision: 1.11 $
31      */
32     public interface IEnvironment {
33     
34     	/**
35     	 * The id of this interface.
36     	 */
37     	Comparable<String> ID = "The Environment";
38     
39     	/**
40     	 * Returns the Jimple representation of the given class.
41     	 *
42     	 * @param className is the name of the class whose Jimple representation is to be returned.
43     	 * @return the requested class.
44     	 * @post result.oclType = SootClass
45     	 */
46     	SootClass getClass(final String className);
47     
48     	/**
49     	 * Returns the classes that form the system.
50     	 *
51     	 * @return the classes the form the system.
52     	 */
53     	Collection<SootClass> getClasses();
54     
55     	/**
56     	 * Retrieves the methods that serve as the entry points or as "roots" of the system being analyzed.
57     	 *
58     	 * @return a collection of methods that are the "roots".
59     	 * @post result != null
60     	 */
61     	Collection<SootMethod> getRoots();
62     
63     	/**
64     	 * Checks if a class by the givne name exists in the environment.
65     	 *
66     	 * @param scName if the FQN of the class of interest.
67     	 * @return <code>true</code> if it exists; <code>false</code>, otherwise.
68     	 */
69     	boolean hasClass(String scName);
70     
71     	/**
72     	 * Removes the given class from the environment.
73     	 *
74     	 * @param clazz to be removed from the system.
75     	 * @pre clazz != null
76     	 */
77     	void removeClass(SootClass clazz);
78     }
