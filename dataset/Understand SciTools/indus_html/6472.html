140     	 * Tests if the basic block graph is minimal.
141     	 */
142     	public final void testBasicBlockGraphAgainstStmtGraph() {
143     		final UnitGraph _unitGraph = bbGraph.getStmtGraph();
144     		final Collection _unitsInGraph = new ArrayList();
145     		final Collection _unitsInBB = new ArrayList();
146     
147     		CollectionUtils.addAll(_unitsInGraph, _unitGraph.iterator());
148     
149     		for (final Iterator _i = bbGraph.getNodes().iterator(); _i.hasNext();) {
150     			final BasicBlock _bb = (BasicBlock) _i.next();
151     			_unitsInBB.addAll(_bb.getStmtsOf());
152     		}
153     
154     		assertTrue(_unitGraph.getClass().getName() + ":" + _unitGraph.getBody().getMethod() + " - "
155     				+ SetUtils.difference(_unitsInBB, _unitsInGraph).toString(), _unitsInGraph.containsAll(_unitsInBB));
156     		assertTrue(_unitGraph.getClass().getName() + ":" + _unitGraph.getBody().getMethod() + " - "
157     				+ SetUtils.difference(_unitsInGraph, _unitsInBB).toString(), _unitsInBB.containsAll(_unitsInGraph));
158     	}

160     	/**
161     	 * Tests the structure of basic block graph.
162     	 */
163     	public final void testBasicBlockGraphStructure() {
164     		final UnitGraph _unitGraph = bbGraph.getStmtGraph();
165     		final Collection _units = new ArrayList();
166     
167     		for (final Iterator _i = bbGraph.getNodes().iterator(); _i.hasNext();) {
168     			final BasicBlock _bb = (BasicBlock) _i.next();
169     			final Stmt _leader = _bb.getLeaderStmt();
170     			final Collection _preds = _unitGraph.getPredsOf(_leader);
171     
172     			if (!_preds.isEmpty()) {
173     				boolean _flag = _preds.size() > 1;
174     
175     				if (!_flag) {
176     					for (final Iterator _j = _preds.iterator(); _j.hasNext();) {
177     						final Stmt _pred = (Stmt) _j.next();
178     						_flag |= _unitGraph.getSuccsOf(_pred).size() > 1;
179     					}
180     				}
181     				assertTrue(_leader.toString() + _unitGraph.getBody().getMethod(), _flag);
182     			}
183     
184     			final Stmt _trailer = _bb.getTrailerStmt();
185     			final Collection _succs = _unitGraph.getSuccsOf(_trailer);
186     
187     			if (!_succs.isEmpty()) {
188     				boolean _flag = _succs.size() > 1;
189     
190     				if (!_flag) {
191     					for (final Iterator _j = _succs.iterator(); _j.hasNext();) {
192     						final Stmt _succ = (Stmt) _j.next();
193     						_flag |= _unitGraph.getPredsOf(_succ).size() > 1;
194     					}
195     				}
196     				assertTrue(_trailer.toString() + _unitGraph.getBody().getMethod(), _flag);
197     			}
198     
199     			_units.clear();
200     			_units.addAll(_bb.getStmtsOf());
201     			_units.remove(_leader);
202     			_units.remove(_trailer);
203     
204     			for (final Iterator _iter = _units.iterator(); _iter.hasNext();) {
205     				final Stmt _unit = (Stmt) _iter.next();
206     				assertTrue(_unitGraph.getPredsOf(_unit).size() == 1);
207     				assertTrue(_unitGraph.getSuccsOf(_unit).size() == 1);
208     			}
209     		}
210     	}
