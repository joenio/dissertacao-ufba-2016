325     	/**
326     	 * This method constructs the BIR representation of the name of a method.
327     	 * 
328     	 * @param sm The soot method whose name will be compiled here.
329     	 * @return the BIR representation of the name of sm.
330     	 */
331     	static String constructMethodName(final SootMethod sm) {
332     		final SootClass _sc = sm.getDeclaringClass();
333     		final String _className = _sc.getName();
334     		final String _methodName = sm.getName();
335     		final int _size = sm.getParameterCount();
336     		final String[] _paramTypeNames = new String[_size];
337     
338     		// get the array of the soot types of the parameters of the method
339     		for (int _i = 0; _i < _size; _i++) {
340     			_paramTypeNames[_i] = sm.getParameterType(_i).toString();
341     		}
342     
343     		// construct the method name.
344     		final StringBuffer _sb = new StringBuffer(METHOD_PREFIX);
345     		_sb.append(_className);
346     		_sb.append('.');
347     		_sb.append(_methodName);
348     		_sb.append('(');
349     
350     		final int _size1 = _paramTypeNames.length;
351     
352     		if (_size1 > 0) {
353     			_sb.append(_paramTypeNames[0]);
354     
355     			for (int _i = 1; _i < _size1; _i++) {
356     				_sb.append(',');
357     				_sb.append(_paramTypeNames[_i]);
358     			}
359     		}
360     
361     		_sb.append(')');
362     		_sb.append(METHOD_SUFFIX);
363     
364     		return _sb.toString();
365     	}

479     	/**
480     	 * Generates the bir locations for the given statement-method pair.
481     	 * 
482     	 * @param p of interest.
483     	 * @param getUnlocking <code>true</code> indicates that locations names of unlocking transitions are also required if
484     	 *            the method is synchronized and the statement is <code>null</code>; <code>false</code>, otherwise. This
485     	 *            is applicable only to entry-exit synchronization of sychronized methods.
486     	 * @return the bir location.
487     	 * @throws IllegalStateException when the given statement does not occur in the system.
488     	 * @pre p != null and p.getSecond() != null
489     	 * @post result != null
490     	 */
491     	Collection<String> generateBIRRep(final Pair<? extends Stmt, SootMethod> p, final boolean getUnlocking) {
492     		final Stmt _stmt = p.getFirst();
493     		final SootMethod _method = p.getSecond();
494     		final String _sig;
495     
496     		if (method2birsig.containsKey(_method)) {
497     			_sig = method2birsig.get(_method);
498     		} else {
499     			_sig = RelativeDependenceInfoTool.constructMethodName(_method);
500     			method2birsig.put(_method, _sig);
501     		}
502     
503     		final List<Stmt> _sl = new ArrayList<Stmt>(_method.retrieveActiveBody().getUnits());
504     		final int _index = _sl.indexOf(_stmt);
505     		final Collection<String> _result = new ArrayList<String>();
506     
507     		if (_index != -1) {
508     			_result.add(_sig + " loc" + _index);
509     		} else if (_stmt == null) {
510     			_result.add(_sig + " sync");
511     
512     			if (getUnlocking) {
513     				_result.add(_sig + " unsync");
514     				_result.add(_sig + " unsyncEx");
515     				_result.add(_sig + " throwEx");
516     			}
517     		} else {
518     			throw new IllegalStateException("Hmm");
519     		}
520     		return _result;
521     	}

534     				.<T, Value, Type> getTokenManager(new SootValueTypeManager()), _stmtGraphFactory);
535     
536     		if (abort) {
537     			return;
538     		}
539     		_aa.analyze(environment, entryPointMethods);
540     
541     		final BasicBlockGraphMgr _bbm = new BasicBlockGraphMgr();
542     		final ValueAnalyzerBasedProcessingController _pc = new ValueAnalyzerBasedProcessingController();
543     		final Collection<IProcessor> _processors = new ArrayList<IProcessor>();
544     		final PairManager _pairManager = new PairManager(false, true);
545     		final CallGraphInfo _cgi = new CallGraphInfo(_pairManager);
546     		final CFGAnalysis _cfgAnalysis = new CFGAnalysis(_cgi, _bbm);
547     		final IThreadGraphInfo _tgi = new ThreadGraph(_cgi, _cfgAnalysis, _pairManager);
548     		final ValueAnalyzerBasedProcessingController _cgipc = new ValueAnalyzerBasedProcessingController();
549     		final OFABasedCallInfoCollector _callGraphInfoCollector = new OFABasedCallInfoCollector();
550     		final OneAllStmtSequenceRetriever _ssr = new OneAllStmtSequenceRetriever();
551     		_bbm.setStmtGraphFactory(_stmtGraphFactory);
