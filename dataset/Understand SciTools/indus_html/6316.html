17     package edu.ksu.cis.indus.common.collections;
18     
19     import edu.ksu.cis.indus.annotations.Functional;
20     import edu.ksu.cis.indus.annotations.Immutable;
21     import edu.ksu.cis.indus.annotations.NonNull;
22     
23     import java.util.ArrayList;
24     import java.util.EmptyStackException;
25     
26     /**
27      * A stack implementation.
28      * 
29      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
30      * @author $Author: rvprasad $
31      * @version $Revision: 1.5 $ $Date: 2007/02/10 19:08:38 $
32      * @param <T> is the type of object stored in this stack.
33      */
34     public final class Stack<T>
35     		extends ArrayList<T>
36     		implements Cloneable {
37     
38     	/**
39     	 * The serial version ID.
40     	 */
41     	private static final long serialVersionUID = -3292985677102986589L;
42     
43     	/**
44     	 * The index at which the top of the stack occurs.
45     	 */
46     	private int top;
47     
48     	/**
49     	 * Creates an instance of this class.
50     	 */
51     	public Stack() {
52     		super();
53     		top = -1;
54     	}
55     
56     	/**
57     	 * {@inheritDoc}
58     	 */
59     	@Override public void clear() {
60     		super.clear();
61     		top = -1;
62     	}
63     
64     	/**
65     	 * {@inheritDoc}
66     	 */
67     	@Functional @Override public Stack<T> clone() {
68     		@SuppressWarnings("unchecked") final Stack<T> _stack = (Stack) super.clone();
69     		return _stack;
70     	}
71     
72     	/**
73     	 * Checks if the stack is empty.
74     	 * 
75     	 * @return <code>true</code> if the stack is empty; <code>false</code>, otherwise.
76     	 */
77     	@Functional public boolean empty() {
78     		return top == -1;
79     	}
80     
81     	/**
82     	 * Retrieves the top of the stack element without popping it.
83     	 * 
84     	 * @return the top of the stack element.
85     	 * @throws EmptyStackException when the stack is empty.
86     	 */
87     	@Functional public T peek() {
88     		if (empty()) {
89     			throw new EmptyStackException();
90     		}
91     		return get(top);
92     	}
93     
94     	/**
95     	 * Pops the element at the top of this stack.
96     	 * 
97     	 * @return the top of the stack element.
98     	 * @throws EmptyStackException when the stack is empty.
99     	 */
100     	public T pop() {
101     		if (empty()) {
102     			throw new EmptyStackException();
103     		}
104     		final T _item = remove(top);
105     		top--;
106     		return _item;
107     	}
108     
109     	/**
110     	 * Pushes the given item onto this stack.
111     	 * 
112     	 * @param item to be pushed onto the stack.
113     	 * @return the item that was pushed.
114     	 */
115     	public T push(@NonNull @Immutable final T item) {
116     		add(++top, item);
117     		return item;
118     	}
119     
120     	/**
121     	 * Retrieves the index at which the given input occurs on this stack.
122     	 * 
123     	 * @param <T1> is the type of the input.
124     	 * @param o is the input to search for.
125     	 * @return the index of the input from the bottom of the stack; -1, if the input does not exist in the stack.
126     	 */
127     	@Functional public <T1 extends T> int search(final T1 o) {
128     		for (int _i = top; _i >= 0; _i--) {
129     			if (get(_i).equals(o)) {
130     				return _i + 1;
131     			}
132     		}
133     		return -1;
134     	}
135     }
