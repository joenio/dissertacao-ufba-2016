272     	/**
273     	 * Calculates SCC according to the algorithm in Udi Manber's book.
274     	 * 
275     	 * @param nodes of interest.
276     	 * @param node2srd maps nodes to an instance of <code>SCCRelatedData</code>.
277     	 * @param node to be explored.
278     	 * @param stack to be used.
279     	 * @param sccs is the outgoing argument to contains the SCCs.
280     	 * @post sccs.containsAll(sccs$pre)
281     	 * @post sccs->forall(o | o->forall(p | nodes.contains(p)))
282     	 * @param <T> the type of nodes being processed.
283     	 */
284     	@Immutable private static <T extends INode<T>> void calculateSCCs(
285     			@NonNull @NonNullContainer @Immutable final Collection<T> nodes,
286     			@NonNull @NonNullContainer final Map<T, SCCRelatedData> node2srd,
287     			@NonNull @NonNullContainer @Immutable final T node, @NonNull @NonNullContainer final Stack<T> stack,
288     			@NonNull @NonNullContainer final Collection<List<T>> sccs) {
289     		final SCCRelatedData _nodeSRD = node2srd.get(node);
290     		_nodeSRD.setDfsNum(dfsNum);
291     		_nodeSRD.setHigh(dfsNum);
292     		stack.push(node);
293     		dfsNum--;
294     
295     		final Iterator<T> _j = IteratorUtils.filteredIterator(node.getSuccsOf().iterator(), new MembershipPredicate<T>(true,
296     				nodes));
297     
298     		for (; _j.hasNext();) {
299     			final T _succ = _j.next();
300     			final SCCRelatedData _succSRD = node2srd.get(_succ);
301     
302     			if (_succSRD.getDfsNum() == 0) {
303     				calculateSCCs(nodes, node2srd, _succ, stack, sccs);
304     				_nodeSRD.setHigh(Math.max(_nodeSRD.getHigh(), _succSRD.getHigh()));
305     			} else if (_succSRD.getDfsNum() > _nodeSRD.getDfsNum() && _succSRD.getComponentNum() == 0) {
306     				_nodeSRD.setHigh(Math.max(_nodeSRD.getHigh(), _succSRD.getDfsNum()));
307     			}
308     		}
309     
310     		if (_nodeSRD.getHigh() == _nodeSRD.getDfsNum()) {
311     			currCompNum++;
312     			_nodeSRD.setComponentNum(currCompNum);
313     
314     			T _o;
315     			final List<T> _scc = new ArrayList<T>();
316     
317     			do {
318     				_o = stack.pop();
319     				node2srd.put(_o, _nodeSRD);
320     				_scc.add(_o);
321     			} while (_o != node);
322     			sccs.add(_scc);
323     		}
324     	}
