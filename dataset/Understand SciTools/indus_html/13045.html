17     package edu.ksu.cis.indus.staticanalyses.flow;
18     
19     import edu.ksu.cis.indus.staticanalyses.tokens.ITokens;
20     
21     /**
22      * This class manages variants corresponding to entities that have values.
23      * 
24      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
25      * @version $Revision: 1.7 $
26      * @param <E> is the type of the entity whose variance is being tracked.!
27      * @param <SYM> is the type of symbol whose flow is being analyzed.
28      * @param <T> is the type of the token set object.
29      * @param <N> is the type of the summary node in the flow analysis.
30      * @param <R> is the type of the symbol types.
31      */
32     class ValuedVariantManager<E, SYM, T extends ITokens<T, SYM>, N extends IFGNode<SYM, T, N>, R>
33     		extends AbstractVariantManager<ValuedVariant<N>, E, SYM, T, N, R> {
34     
35     	/**
36     	 * Creates a new <code>ValuedVariantManager</code> instance.
37     	 * 
38     	 * @param theAnalysis the instance of the framework in which this object is used.
39     	 * @param indexManager the manager of indices which map array variants to arrays.
40     	 * @pre theAnalysis != null and indexManager != null
41     	 */
42     	ValuedVariantManager(final FA<SYM, T, N, R> theAnalysis, final IIndexManager<? extends IIndex<?>, E> indexManager) {
43     		super(theAnalysis, indexManager);
44     	}
45     
46     	/**
47     	 * Returns a new valued variant corresponding to the given ast object.
48     	 * 
49     	 * @param o the ast object whose variant is to be returned.
50     	 * @return a new variant corresponding to <code>o</code>.
51     	 */
52     	@Override protected ValuedVariant<N> getNewVariant(@SuppressWarnings("unused") final E o) {
53     		return new ValuedVariant<N>(fa.getNewFGNode());
54     	}
55     }
