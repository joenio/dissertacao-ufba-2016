33     /**
34      * This class tests results from dependency analysis which can be queried for dependence on a statement in a method.
35      *
36      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
37      * @author $Author: rvprasad $
38      * @version $Revision: 1.6 $ $Date: 2007/02/10 19:07:02 $
39      */
40     class StmtAndMethodBasedDependencyAnalysisTest
41       extends AbstractDependencyAnalysisTest {
42     	/** 
43     	 * The logger used by instances of this class to log messages.
44     	 */
45     	private static final Logger LOGGER = LoggerFactory.getLogger(StmtAndMethodBasedDependencyAnalysisTest.class);
46     
47     	/**
48     	 * @see AbstractDependencyAnalysisTest#verifyDAFor(soot.jimple.Stmt, soot.SootMethod)
49     	 */
50     	protected void verifyDAFor(final Stmt unit, final SootMethod sm) {
51     		final IDependencyAnalysis _da = getDA();
52     		verifyDependentPosition(unit, sm, _da);
53     		verifyDependeePosition(unit, sm, _da);
54     	}
55     
56     	/**
57     	 * Verifies dependency analysis from dependee position, i.e., checks if the given unit occurs as a dependee in the
58     	 * dependees set of all it's dependents.
59     	 *
60     	 * @param dependee is the dependee
61     	 * @param method is the method containing <code>stmt</code>.
62     	 * @param dependencyAnalysis is the analysis to be tested.
63     	 *
64     	 * @throws IllegalStateException when the given dependency analysis does not report info as pairs or as statements.
65     	 *
66     	 * @pre dependee != null and method != null and dependencyAnalysis != null
67     	 */
68     	private void verifyDependeePosition(final Stmt dependee, final SootMethod method,
69     		final IDependencyAnalysis dependencyAnalysis) {
70     		final Collection _dependents = dependencyAnalysis.getDependents(dependee, method);
71     
72     		if (!_dependents.isEmpty()) {
73     			final Object _temp = _dependents.iterator().next();
74     			final Object _dependee;
75     
76     			if (_temp instanceof Pair) {
77     				_dependee = new Pair(dependee, method);
78     			} else {
79     				_dependee = dependee;
80     			}
81     
82     			for (final Iterator _i = _dependents.iterator(); _i.hasNext();) {
83     				final Object _o = _i.next();
84     				Stmt _stmt;
85     				SootMethod _method = method;
86     
87     				if (_o instanceof Pair) {
88     					final Pair _pair = (Pair) _o;
89     					_stmt = (Stmt) _pair.getFirst();
90     					_method = (SootMethod) _pair.getSecond();
91     				} else if (_o instanceof Stmt) {
92     					_stmt = (Stmt) _o;
93     				} else {
94     					final String _msg = "This class can only handle Pair and Stmt as targets of dependence information.";
95     					LOGGER.error(_msg);
96     					throw new IllegalStateException(_msg);
97     				}
98     
99     				assertTrue("Dependee:" + getDA().getClass().getName() + ": (" + dependee + ", " + method + ") -> (" + _stmt
100     					+ ", " + _method + ")", dependencyAnalysis.getDependees(_stmt, _method).contains(_dependee));
101     			}
102     		}
103     	}
104     
105     	/**
106     	 * Verifies dependency analysis from dependent position, i.e., checks if the given unit occurs as a dependent in the
107     	 * dependents set of all it's dependees.
108     	 *
109     	 * @param dependent is the dependent
110     	 * @param method is the method containing <code>stmt</code>.
111     	 * @param dependencyAnalysis is the analysis to be tested.
112     	 *
113     	 * @throws IllegalStateException when the given dependency analysis does not report info as pairs or as statements.
114     	 *
115     	 * @pre dependent != null and method != null and dependencyAnalysis != null
116     	 */
117     	private void verifyDependentPosition(final Stmt dependent, final SootMethod method,
118     		final IDependencyAnalysis dependencyAnalysis) {
119     		final Collection _dependees = dependencyAnalysis.getDependees(dependent, method);
120     
121     		if (!_dependees.isEmpty()) {
122     			final Object _temp = _dependees.iterator().next();
123     			final Object _dependent;
124     
125     			if (_temp instanceof Pair) {
126     				_dependent = new Pair(dependent, method);
127     			} else {
128     				_dependent = dependent;
129     			}
130     
131     			for (final Iterator _i = _dependees.iterator(); _i.hasNext();) {
132     				final Object _o = _i.next();
133     				Stmt _stmt;
134     				SootMethod _method = method;
135     
136     				if (_o instanceof Pair) {
137     					final Pair _pair = (Pair) _o;
138     					_stmt = (Stmt) _pair.getFirst();
139     					_method = (SootMethod) _pair.getSecond();
140     				} else if (_o instanceof Stmt) {
141     					_stmt = (Stmt) _o;
142     				} else {
143     					final String _msg = "This class can only handle Pair and Stmt as targets of dependence information.";
144     					LOGGER.error(_msg);
145     					throw new IllegalStateException(_msg);
146     				}
147     
148     				assertTrue("Dependent:" + getDA().getClass().getName() + ": (" + dependent + ", " + method + ") -> (" + _stmt
149     					+ ", " + _method + ")", dependencyAnalysis.getDependents(_stmt, _method).contains(_dependent));
150     			}
151     		}
152     	}
153     }
