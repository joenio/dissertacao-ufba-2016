236     	/**
237     	 * Executes the tool. The tool is multithreaded. However, the user can run it in synchronous mode and if tool fails in
238     	 * this mode, any subsequent calls to <code>isStable()</code> until a following call to <code>run()</code> will return
239     	 * <code>false</code>.
240     	 * 
241     	 * @param phase is the suggestive phase to start execution in.
242     	 * @param lastPhase is the phase that should be executed last before exiting.
243     	 * @param synchronous <code>true</code> indicates that this method should behave synchronously and return only after the
244     	 *            tool's run has completed; <code>false</code> indicates that this method can return once the tool has
245     	 *            started it's run.
246     	 * @throws RuntimeException when the tool fails.
247     	 * @throws IllegalStateException when this method is called on a paused tool.
248     	 */
249     	public final synchronized void run(final Phase phase, final Phase lastPhase, final boolean synchronous) {
250     		if (!pause || isNotAlive()) {
251     			checkConfiguration();
252     			childException = null;
253     			unstable();
254     			activateActiveParts();
255     			thread = new Thread() {
256     
257     				@Override public final void run() {
258     					Throwable _temp = null;
259     					try {
260     						// we do this to respect any pre-run pause calls.
261     						movingToNextPhase();
262     
263     						execute(phase, lastPhase);
264     					} catch (final InterruptedException _e) {
265     						LOGGER.error("Interrupted while executing the tool.", _e);
266     						_temp = _e;
267     					} catch (final Throwable _e) {
268     						LOGGER.error("Tool failed.", _e);
269     						_temp = _e;
270     					} finally {
271     						if (_temp != null) {
272     							childException = _temp;
273     						}
274     						pause = false;
275     						executor.shutdown();
276     						executor = Executors.newSingleThreadExecutor();
277     					}
278     				}
279     			};
280     			thread.start();
281     
282     			if (synchronous) {
283     				try {
284     					thread.join();
285     
286     					if (childException != null) {
287     						throw new RuntimeException(childException);
288     					}
289     					stable();
290     				} catch (final InterruptedException _e) {
291     					LOGGER.error("Interrupted while waiting on the run to complete.", _e);
292     					throw new RuntimeException(_e);
293     				}
294     			} else {
295     				final Thread _temp = new Thread() {
296     
297     					@Override public void run() {
298     						try {
299     							thread.join();
300     
301     							if (childException != null) {
302     								throw new RuntimeException(childException);
303     							}
304     							stable();
305     						} catch (final InterruptedException _e) {
306     							LOGGER.error("Interrupted while waiting on the helper thread.", _e);
307     						}
308     					}
309     				};
310     				_temp.start();
311     			}
312     		} else {
313     			throw new IllegalStateException("run() should be called when the tool is paused or running.");
314     		}
315     	}

346     	 * Reports the given tool progress information to any registered listeners. The listeners will receive the events in the
347     	 * order they were fired.
348     	 * 
349     	 * @param message about the progress of the tool.
350     	 * @param info anything the tool may want to convey to the listener.
351     	 * @throws RuntimeException is thrown by the message delivery thread when it is interrupted.
352     	 * @pre message != null and info != null
353     	 */
354     	protected void fireToolProgressEvent(final String message, final Object info) {
355     		executor.execute(new Runnable() {
356     
357     			public void run() {
358     				synchronized (listeners) {
359     					final ToolProgressEvent _evt = new ToolProgressEvent(this, message, info);
360     
361     					for (final Iterator<IToolProgressListener> _i = listeners.iterator(); _i.hasNext();) {
362     						final IToolProgressListener _listener = _i.next();
363     						_listener.toolProgess(_evt);
364     					}
365     				}
366     			}
367     		});
368     	}

370     	/**
371     	 * Used to suspend the tool execution. This indicates that the tool implementation is moving onto a new phase, hence, it
372     	 * is at a point where it is safe to pause/suspend execution. If the application had requested the tool to pause via
373     	 * <code>pause()</code>, this method will suspend the execution of the tool.
374     	 * 
375     	 * @throws InterruptedException when the thread in which the tool has paused is interrupted.
376     	 */
377     	protected final void movingToNextPhase() throws InterruptedException {
378     		synchronized (control) {
379     			if (pause) {
380     				control.wait();
381     			} else if (!activePart.canProceed()) {

383     				fireToolProgressEvent(_string, null);

385     		}
