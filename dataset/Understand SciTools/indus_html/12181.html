31     /**
32      * This class uses symbolic- and escape-analysis as calculated by <code>EquivalenceClassBasedEscapeAnalysis</code> to prune
33      * the ready dependency information calculated by it's parent class. This class will also use OFA information if it is
34      * configured to do so.
35      * 
36      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
37      * @author $Author: rvprasad $
38      * @version $Revision: 1.17 $
39      * @see edu.ksu.cis.indus.staticanalyses.dependency.ReadyDAv1
40      */
41     public final class ReadyDAv3
42     		extends ReadyDAv2 {
43     
44     	/**
45     	 * Creates an instance of this class.
46     	 * 
47     	 * @param directionSensitiveInfo that controls the direction.
48     	 * @param direction of the analysis
49     	 * @pre info != null and direction != null
50     	 */
51     	private ReadyDAv3(final IDirectionSensitiveInfo directionSensitiveInfo, final Direction direction) {
52     		super(directionSensitiveInfo, direction);
53     	}
54     
55     	/**
56     	 * Retrieves an instance of ready dependence analysis that calculates information in backward direction.
57     	 * 
58     	 * @return an instance of ready dependence.
59     	 * @post result != null
60     	 */
61     	public static ReadyDAv1 getBackwardReadyDA() {
62     		return new ReadyDAv3(new BackwardDirectionSensitiveInfo(), Direction.BACKWARD_DIRECTION);
63     	}
64     
65     	/**
66     	 * Retrieves an instance of ready dependence analysis that calculates information in forward direction.
67     	 * 
68     	 * @return an instance of ready dependence.
69     	 * @post result != null
70     	 */
71     	public static ReadyDAv1 getForwardReadyDA() {
72     		return new ReadyDAv3(new ForwardDirectionSensitiveInfo(), Direction.FORWARD_DIRECTION);
73     	}
74     
75     	/**
76     	 * Checks if the given enter monitor statement/synchronized method is dependent on the exit monitor statement/synchronized
77     	 * method according to rule 2. The results of escape analysis info calculated {@link
78     	 * edu.ksu.cis.indus.staticanalyses.concurrency.escape.EquivalenceClassBasedEscapeAnalysis
79     	 * EquivalenceClassBasedEscapeAnalysis} analysis along with lock entities information is used to determine the dependence.
80     	 * 
81     	 * @param enterPair is the enter monitor statement and containg statement pair.
82     	 * @param exitPair is the exit monitor statement and containg statement pair.
83     	 * @return <code>true</code> if there is a dependence; <code>false</code>, otherwise.
84     	 * @pre enterPair.getSecond() != null and exitPair.getSecond() != null
85     	 * @see ReadyDAv2#ifDependentOnByRule2(Pair, Pair)
86     	 */
87     	@Override protected boolean ifDependentOnByRule2(final Pair<EnterMonitorStmt, SootMethod> enterPair,
88     			final Pair<ExitMonitorStmt, SootMethod> exitPair) {
89     		boolean _result = super.ifDependentOnByRule2(enterPair, exitPair);
90     
91     		if (_result) {
92     			final SootMethod _enterMethod = enterPair.getSecond();
93     			final SootMethod _exitMethod = exitPair.getSecond();
94     
95     			if (!(_enterMethod.isStatic() && _exitMethod.isStatic())) {
96     				final EnterMonitorStmt _enter = enterPair.getFirst();
97     				final ExitMonitorStmt _exit = exitPair.getFirst();
98     				_result = ecba.areMonitorsCoupled(_enter, _enterMethod, _exit, _exitMethod);
99     			}
100     		}
101     		return _result;
102     	}
103     
104     	/**
105     	 * Checks if the given <code>wait()</code> call-site is dependent on the <code>notifyXX()</code> call-site according
106     	 * to rule 2. The symbolic and escape analysis infomration from {@link
107     	 * edu.ksu.cis.indus.staticanalyses.concurrency.escape.EquivalenceClassBasedEscapeAnalysis
108     	 * EquivalenceClassBasedEscapeAnalysis} analysis is used to determine the dependence.
109     	 * 
110     	 * @param wPair is the statement in which <code>java.lang.Object.wait()</code> is invoked.
111     	 * @param nPair is the statement in which <code>java.lang.Object.notifyXX()</code> is invoked.
112     	 * @return <code>true</code> if there is a dependence; <code>false</code>, otherwise.
113     	 * @pre wPair.getSecond() != null and nPair.getSecond() != null
114     	 * @see ReadyDAv1#ifDependentOnByRule4(Pair, Pair)
115     	 */
116     	@Override protected boolean ifDependentOnByRule4(final Pair<InvokeStmt, SootMethod> wPair,
117     			final Pair<InvokeStmt, SootMethod> nPair) {
118     		boolean _result = super.ifDependentOnByRule4(wPair, nPair);
119     
120     		if (_result) {
121     			final InvokeStmt _notify = nPair.getFirst();
122     			final InvokeStmt _wait = wPair.getFirst();
123     			final SootMethod _wMethod = wPair.getSecond();
124     			final SootMethod _nMethod = nPair.getSecond();
125     			_result = ecba.areWaitAndNotifyCoupled(_wait, _wMethod, _notify, _nMethod);
126     		}
127     		return _result;
128     	}
129     }
