46     /**
47      * This class provides abstract implementation that can be used to do xml data based testing.
48      *
49      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
50      * @author $Author: rvprasad $
51      * @version $Revision: 1.23 $ $Date: 2007/02/10 19:08:51 $
52      */
53     public abstract class AbstractXMLBasedTest
54       extends XMLTestCase
55       implements IXMLBasedTest {
56     	/** 
57     	 * The logger used by instances of this class to log messages.
58     	 */
59     	private static final Logger LOGGER = LoggerFactory.getLogger(AbstractXMLBasedTest.class);
60     
61     	/** 
62     	 * The xmlizer used to xmlize analysis info before testing.
63     	 */
64     	protected IXMLizer xmlizer;
65     
66     	/** 
67     	 * The map of interface id to interface implementation instances.
68     	 */
69     	protected final Map info = new HashMap();
70     
71     	/** 
72     	 * The directory in which the test control input is read from.
73     	 */
74     	protected String xmlControlDir;
75     
76     	/** 
77     	 * The directory in which the test input is read from.
78     	 */
79     	protected String xmlTestDir;
80     
81     	/** 
82     	 * ID generator used while xmlizing documents which will often be the case (xmlize and test the xmlized data).
83     	 */
84     	private IJimpleIDGenerator idGenerator;
85     
86     	/** 
87     	 * The statement graph (CFG) factory used during testing.
88     	 */
89     	private IStmtGraphFactory stmtGraphFactory;
90     
91     	/** 
92     	 * The name of the method being run.
93     	 */
94     	private String testMethodName = "";
95     
96     	/** 
97     	 * The name of the test case instance.
98     	 */
99     	private String testName = "";
100     
101     	/**
102     	 * @see IXMLBasedTest#setIdGenerator(IJimpleIDGenerator)
103     	 */
104     	public final void setIdGenerator(final IJimpleIDGenerator generator) {
105     		idGenerator = generator;
106     	}
107     
108     	/**
109     	 * @see junit.framework.TestCase#setName(java.lang.String)
110     	 */
111     	public void setName(final String name) {
112     		testMethodName = name;
113     	}
114     
115     	/**
116     	 * @see junit.framework.TestCase#getName()
117     	 */
118     	public String getName() {
119     		return testName;
120     	}
121     
122     	/**
123     	 * Sets the CFG factory to be used during testing.
124     	 *
125     	 * @param cfgFactory is the factory to be used.
126     	 */
127     	public void setStmtGraphFactory(final IStmtGraphFactory cfgFactory) {
128     		stmtGraphFactory = cfgFactory;
129     	}
130     
131     	/**
132     	 * Sets the name of the test instance.
133     	 *
134     	 * @param name of the test instance.
135     	 *
136     	 * @pre name != null
137     	 */
138     	public void setTestName(final String name) {
139     		testName = name;
140     		super.setName(name);
141     	}
142     
143     	/**
144     	 * @see IXMLBasedTest#setXMLControlDir(String)
145     	 */
146     	public void setXMLControlDir(final String xmlInDir) {
147     		xmlControlDir = xmlInDir;
148     	}
149     
150     	/**
151     	 * @see IXMLBasedTest#setXMLTestDir(String)
152     	 */
153     	public void setXMLTestDir(final String xmlInDir) {
154     		xmlTestDir = xmlInDir;
155     	}
156     
157     	/**
158     	 * Tests the inforamtion generated from the associated fixture. This uses <i>XMLUnit</i>.
159     	 */
160     	public void testXMLSimilarity() {
161     		final String _outfileName = xmlTestDir + File.separator + getFileName();
162     
163     		try {
164     			final Reader _current = new FileReader(new File(_outfileName));
165     			final Reader _previous = new FileReader(new File(xmlControlDir + File.separator + getFileName()));
166     
167     			final Diff _diff = new Diff(_previous, _current);
168     			_diff.overrideElementQualifier(new ElementNameAndAttributeQualifier());
169     
170     			final boolean _verdict = _diff.similar();
171     			final String _difference = _diff.toString();
172     			_current.close();
173     			_previous.close();
174     			assertTrue(_difference, _verdict);
175     		} catch (IOException _e) {
176     			LOGGER.error("Failed to read the xml file " + _outfileName, _e);
177     			fail(_e.getMessage());
178     		} catch (SAXException _e) {
179     			LOGGER.error("Exception while parsing XML", _e);
180     			fail(_e.getMessage());
181     		} catch (ParserConfigurationException _e) {
182     			LOGGER.error("XML parser configuration related exception", _e);
183     			fail(_e.getMessage());
184     		}
185     	}
186     
187     	/**
188     	 * Retrieves the name of the file that contains the test data.  The default implementation constructs a name from the
189     	 * xmlizer.
190     	 *
191     	 * @return the name of the file.
192     	 */
193     	protected String getFileName() {
194     		return xmlizer.getFileName(getName());
195     	}
196     
197     	/**
198     	 * @see junit.framework.TestCase#setUp()
199     	 */
200     	protected final void setUp()
201     	  throws Exception {
202     		xmlizer = getXMLizer();
203     		xmlizer.setXmlOutputDir(xmlTestDir);
204     		xmlizer.setGenerator(idGenerator);
205     		localSetup();
206     		xmlizer.writeXML(info);
207     	}
208     
209     	/**
210     	 * Retrieve the xmlizer to be used to generate the xml data for testing purpose.
211     	 *
212     	 * @return the xmlizer.
213     	 *
214     	 * @post result != null
215     	 */
216     	protected abstract IXMLizer getXMLizer();
217     
218     	/**
219     	 * Local test setup to be provided by subclasses.  Default implementation will add the name obtained via
220     	 * <code>getName()</code> into the <code>info</code> map against the key <code>AbstractXMLizer.FILE_NAME_ID</code> along
221     	 * with a <code>IStmtGraphFactory.ID</code> to <code>stmtGraphFactory</code> mapping.
222     	 *
223     	 * @throws Exception <i>not thrown by this implementation.</i>
224     	 */
225     	protected void localSetup()
226     	  throws Exception {
227     		info.put(AbstractXMLizer.FILE_NAME_ID, getName());
228     		info.put(IStmtGraphFactory.ID, stmtGraphFactory);
229     	}
230     
231     	/**
232     	 * @see junit.framework.TestCase#runTest()
233     	 */
234     	protected void runTest()
235     	  throws Throwable {
236     		super.setName(testMethodName);
237     		super.runTest();
238     		super.setName(testName);
239     	}
240     
241     	/**
242     	 * @see junit.framework.TestCase#tearDown()
243     	 */
244     	protected void tearDown()
245     	  throws Exception {
246     		info.clear();
247     
248     		if (stmtGraphFactory != null) {
249     			stmtGraphFactory.reset();
250     			stmtGraphFactory = null;
251     		}
252     		xmlizer = null;
253     		super.tearDown();
254     	}
255     }
