18     package edu.ksu.cis.indus.interfaces;
19     
20     /**
21      * This interface is used to manage information based on ID.
22      *
23      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
24      * @author $Author: rvprasad $
25      * @version $Revision: 1.5 $
26      */
27     public interface IIDBasedInfoManagement {
28     	/**
29     	 * Sets the information for the given id.
30     	 *
31     	 * @param infoID is the id.
32     	 * @param info is the implemenation that provides the info.
33     	 *
34     	 * @return the previous implementation that provided the info.
35     	 *
36     	 * @pre infoID != null and info != null
37     	 */
38     	Object setInfoFor(final Comparable<?> infoID, final Object info);
39     
40     	/**
41     	 * Clears all ID to info mapping.
42     	 */
43     	void clearInfo();
44     
45     	/**
46     	 * Removes the ID to info mapping.
47     	 *
48     	 * @param infoID for which the mapping is to be removed.
49     	 *
50     	 * @return the implementation that provide the info.
51     	 *
52     	 * @pre infoID != null
53     	 */
54     	Object removeInfo(final Comparable<?> infoID);
55     }
