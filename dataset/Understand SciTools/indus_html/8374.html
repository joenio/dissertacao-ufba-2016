17     package edu.ksu.cis.indus.staticanalyses.processing;
18     
19     import edu.ksu.cis.indus.processing.IProcessor;
20     import edu.ksu.cis.indus.processing.ProcessingController;
21     
22     import edu.ksu.cis.indus.staticanalyses.interfaces.IValueAnalyzer;
23     import edu.ksu.cis.indus.staticanalyses.interfaces.IValueAnalyzerBasedProcessor;
24     
25     import java.util.Collection;
26     import java.util.HashSet;
27     import java.util.Iterator;
28     import java.util.Set;
29     
30     /**
31      * This class controls the post processing for an analysis. The analysis as realised by FA is very low-level. The information
32      * is raw. This needs to be massaged via post processing. Each post processor can registered interest in particular types of
33      * AST chunks. The controller will walk over the analyzed system and call the registered post processors. The post processors
34      * then collect information from the analysis in form which is more accessible to the other applications. This visitor will
35      * notify the interested post processors with the given AST node and then visit it's children.
36      * <p>
37      * Please note that the processor should be registered/unregistered separately for interface-level (class/method) processing
38      * and functional (method-body) processing.
39      * </p>
40      * 
41      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
42      * @author $Author: rvprasad $
43      * @version $Revision: 1.7 $
44      */
45     public class ValueAnalyzerBasedProcessingController
46     		extends ProcessingController {
47     
48     	/**
49     	 * The analyzer instance that provides the low-level analysis information to be be further processed.
50     	 * 
51     	 * @invariant analyzer != null
52     	 */
53     	protected IValueAnalyzer<?> analyzer;
54     
55     	/**
56     	 * Sets the analyzer which provides the information to be processed. This implementation retrieves the environment from
57     	 * the provided analyzer.
58     	 * 
59     	 * @param analyzerParam an instance of the FA.
60     	 */
61     	public void setAnalyzer(final IValueAnalyzer<?> analyzerParam) {
62     		analyzer = analyzerParam;
63     		setEnvironment(analyzer.getEnvironment());
64     	}
65     
66     	/**
67     	 * Sets the analyzer on all the processors which require the analyzer.
68     	 */
69     	@Override protected void initializeProcessors() {
70     		final Collection<IProcessor> _processors = new HashSet<IProcessor>();
71     		_processors.addAll(interfaceProcessors);
72     
73     		for (final Iterator<Set<IProcessor>> _i = class2processors.values().iterator(); _i.hasNext();) {
74     			_processors.addAll(_i.next());
75     		}
76     
77     		for (final Iterator _i = _processors.iterator(); _i.hasNext();) {
78     			final Object _o = _i.next();
79     
80     			if (_o instanceof IValueAnalyzerBasedProcessor) {
81     				((IValueAnalyzerBasedProcessor) _o).setAnalyzer(analyzer);
82     			}
83     		}
84     	}
85     }
