129     	/**
130     	 * Processes the given methods.
131     	 *
132     	 * @param taggedMethods are the methods to process.
133     	 * @param basicBlockMgr is the basic block manager to be used to retrieve basic blocks while processing methods.
134     	 * @param theCollector is the slice collector to extend the slice.
135     	 *
136     	 * @pre taggedMethods != null and basicBlockMgr != null and theCollector != null
137     	 * @pre taggedMethods.oclIsKindOf(Collection(SootMethod))
138     	 */
139     	public final void process(final Collection<SootMethod> taggedMethods, final BasicBlockGraphMgr basicBlockMgr,
140     		final SliceCollector theCollector) {
141     		if (LOGGER.isInfoEnabled()) {
142     			LOGGER.info("BEGIN: Post Processing");
143     		}
144     		
145     		if (LOGGER.isDebugEnabled()) {
146     			LOGGER.debug("Slice before processing: " + theCollector.toString());
147     		}
148     
149     		collector = theCollector;
150     		bbgMgr = basicBlockMgr;
151     		cd.setBasicBlockGraphManager(basicBlockMgr);
152     		methodWorkBag.addAllWorkNoDuplicates(taggedMethods);
153     
154     		// process the methods and gather the collected classes
155     		while (methodWorkBag.hasWork()) {
156     			final SootMethod _method = methodWorkBag.getWork();
157     
158     			processMethod(_method);
159     
160     			if (_method.isConcrete()) {
161     				if (LOGGER.isDebugEnabled()) {
162     					LOGGER.debug("Post Processing method " + _method);
163     				}
164     				stmtCollected = false;
165     				processStmts(_method);
166     
167     				if (stmtCollected) {
168     					pickReturnPoints(_method);
169     				}
170     			} else {
171     				if (LOGGER.isDebugEnabled()) {
172     					LOGGER.debug("Could not process method " + _method.getSignature());
173     				}
174     			}
175     		}
176     
177     		fixupAbstractMethodsInClassHierarchy();
178     
179     		if (LOGGER.isInfoEnabled()) {
180     			LOGGER.info("END: Post Processing");
181     		}
182     		if (LOGGER.isDebugEnabled()) {
183     			LOGGER.debug("Slice after processing: " + theCollector.toString());
184     		}
185     
186     	}

212     	/**
213     	 * Fix up class hierarchy such that all abstract methods have an implemented counterpart in the slice.
214     	 */
215     	private void fixupAbstractMethodsInClassHierarchy() {
216             if (LOGGER.isDebugEnabled()) {
217                 LOGGER.debug("BEGIN: Fixing Class Hierarchy - " + getClass() + " " + collector.getClassesInSlice());
218             }
219             
220     		// setup the variables for fixing the class hierarchy
221     		final IClassHierarchy _ch = getClassHierarchyContainingClasses(collector.getClassesInSlice());
222     		final Collection<SootClass> _topologicallyOrderedClasses = _ch.getClassesInTopologicalOrder(true);
223     		final Map<SootClass, Collection<SootMethod>> _class2abstractMethods = new HashMap<SootClass, Collection<SootMethod>>();
224             collector.includeInSlice(_ch.getClasses());
225             collector.includeInSlice(_ch.getInterfaces());
226     
227     		if (LOGGER.isDebugEnabled()) {
228     			LOGGER.debug("Topological Sort: " + _topologicallyOrderedClasses);
229     		}
230     
231     		// fixup methods with respect the class hierarchy  
232     		for (final Iterator<SootClass> _i = _topologicallyOrderedClasses.iterator(); _i.hasNext();) {
233     			final SootClass _currClass = _i.next();
234     			final Collection<SootMethod> _abstractMethodsAtCurrClass =
235     				gatherCollectedAbstractMethodsInSuperClasses(_class2abstractMethods, _currClass);
236     			final List<SootMethod> _methods = _currClass.getMethods();
237     			final Collection<SootMethod> _collectedMethods =  collector.getCollected(_methods);
238     			final Collection<SootMethod> _unCollectedMethods = SetUtils.difference(_methods, _collectedMethods);
239     
240     			if (LOGGER.isDebugEnabled()) {
241     				LOGGER.debug("Fixing up " + _currClass);
242     			}
243     
244     			// remove all abstract methods which are overridden by collected methods of this class.
245     			Util.removeMethodsWithSameSignature(_abstractMethodsAtCurrClass, _collectedMethods);
246     			// gather uncollected methods of this class which have the same signature as any gathered "super" abstract 
247     			// methods.
248     			Util.retainMethodsWithSameSignature(_unCollectedMethods, _abstractMethodsAtCurrClass);
249     			// remove abstract counterparts of all methods that were included in the slice in the previous step.
250     			Util.removeMethodsWithSameSignature(_abstractMethodsAtCurrClass, _unCollectedMethods);
251     			// include the uncollected concrete methods into the slice and process them
252     			collector.includeInSlice(_unCollectedMethods);
253     
254     			// gather collected abstract methods in this class/interface
255     			if (_currClass.isInterface()) {
256     				_abstractMethodsAtCurrClass.addAll(_collectedMethods);
257     			} else if (_currClass.isAbstract()) {
258     				for (final Iterator<SootMethod> _j = _collectedMethods.iterator(); _j.hasNext();) {
259     					final SootMethod _sm = _j.next();
260     
261     					if (_sm.isAbstract()) {
262     						_abstractMethodsAtCurrClass.add(_sm);
263     					}
264     				}
265     			}
266     
267     			// record the abstract methods
268     			if (!_abstractMethodsAtCurrClass.isEmpty()) {
269     				_class2abstractMethods.put(_currClass, new ArrayList<SootMethod>(_abstractMethodsAtCurrClass));
270     			}
271     		}
272     
273     		if (LOGGER.isDebugEnabled()) {
274     			LOGGER.debug("Class -> abstract Method mapping:\n" + _class2abstractMethods);
275     			LOGGER.debug("END: Fixing Class Hierarchy");
276     		}
277     	}

279     	/**
280     	 * Gathers the collected abstract methods that belong to the super classes/interface of the given class.
281     	 *
282     	 * @param class2abstractMethods maps classes to collected abstract methods.
283     	 * @param clazz for which the collection should be performed.
284     	 *
285     	 * @return a collection of collected abstract methods belonging to the super classes.
286     	 *
287     	 * @pre class2abstractMethods != null
288     	 * @pre clazz != null
289     	 * @post result != null
290     	 * @post result->forall(o | class2abstractMethods->exists(p | p.values().includes(o)))
291     	 */
292     	private Collection<SootMethod> gatherCollectedAbstractMethodsInSuperClasses(final Map<SootClass, Collection<SootMethod>> class2abstractMethods, final SootClass clazz) {
293     		final Collection<SootMethod> _methods = new HashSet<SootMethod>();
294     
295     		// gather collected abstract methods from super interfaces and classes.
296     		for (@SuppressWarnings("unchecked") final Iterator<SootClass> _j = clazz.getInterfaces().iterator(); _j.hasNext();) {
297     			final SootClass _interface =  _j.next();
298     
299     			if (collector.hasBeenCollected(_interface)) {
300     				final Collection<SootMethod> _abstractMethods = class2abstractMethods.get(_interface);
301     
302     				if (_abstractMethods != null) {
303     					_methods.addAll(_abstractMethods);
304     				}
305     			}
306     		}
307     
308     		if (clazz.hasSuperclass()) {
309     			final SootClass _superClass = clazz.getSuperclass();
310     
311     			if (collector.hasBeenCollected(_superClass)) {
312     				final Collection<SootMethod> _abstractMethods = class2abstractMethods.get(_superClass);
313     
314     				if (_abstractMethods != null) {
315     					_methods.addAll(_abstractMethods);
316     				}
317     			}
318     		}
319     		return _methods;
320     	}

322     	/**
323     	 * Picks a return point from the given set of return points.  All other options should be tried and it should be certain
324     	 * that a  random choice is safe.
325     	 *
326     	 * @param returnPoints from which to pick a random one.
327     	 *
328     	 * @pre returnPoints != null and returnPoints.oclIsKindOf(Collection(Stmt))
329     	 */
330     	private void pickARandomReturnPoint(final Collection<BasicBlock> returnPoints) {
331     		Stmt _exitStmt = null;
332     
333     		if (LOGGER.isDebugEnabled()) {
334     			LOGGER.debug("pickARandomReturnPoint(returnPoints = " + returnPoints + ")");
335     		}
336     
337     		for (final Iterator<BasicBlock> _i = returnPoints.iterator(); _i.hasNext();) {
338     			final BasicBlock _bb = _i.next();
339     			final Stmt _stmt = _bb.getTrailerStmt();
340     
341     			if (_stmt instanceof ReturnStmt || _stmt instanceof ReturnVoidStmt) {
342     				// if there exists a return statement grab it.
343     				_exitStmt = _stmt;
344     				break;
345     			} else if (_stmt instanceof ThrowStmt) {
346     				// if there have been no return statements, then grab a throw statement and continue to look.
347     				_exitStmt = _stmt;
348     			} else if (_exitStmt == null) {
349     				// if there have been no exit points, then grab the psedu exit and continue to look.
350     				_exitStmt = _stmt;
351     			}
352     		}
353     		processAndIncludeExitStmt(_exitStmt);
354     	}

356     	/**
357     	 * Picks the return points of the method required to make it's slice executable.
358     	 *
359     	 * @param method to be processed.
360     	 *
361     	 * @pre method != null
362     	 */
363     	private void pickReturnPoints(final SootMethod method) {
364     		if (LOGGER.isDebugEnabled()) {
365     			LOGGER.debug("BEGIN: Picking return points in " + method);
366     		}
367     
368     		// pick all return/throw points in the methods.
369     		final String _tagName = collector.getTagName();
370     		final BasicBlockGraph _bbg = bbgMgr.getBasicBlockGraph(method);
371     		final Collection<BasicBlock> _tails = new HashSet<BasicBlock>();
372     		_tails.addAll(_bbg.getTails());
373     		cd.analyze(Collections.singleton(method));
374     
375     		if (_tails.size() == 1) {
376     			// If there is only one tail then include the statement
377     			final BasicBlock _bb = _tails.iterator().next();
378     			processAndIncludeExitStmt(_bb.getTrailerStmt());
379     		} else {
380     			boolean _tailWasNotPicked = true;
381     
382     			// if there are more than one tail then pick only the ones that are reachable via collected statements
383     			for (final Iterator<BasicBlock> _j = _tails.iterator(); _j.hasNext();) {
384     				final BasicBlock _bb = _j.next();
385     				final Stmt _stmt = _bb.getTrailerStmt();
386     				final Collection<Stmt> _dependees = cd.getDependees(_stmt, method);
387     
388     				if (!collector.hasBeenCollected(_stmt)
389     					  && (_dependees.isEmpty() || !Util.getHostsWithTag(_dependees, _tagName).isEmpty())) {
390     					processAndIncludeExitStmt(_stmt);
391     					_tailWasNotPicked = false;
392     
393     					if (LOGGER.isDebugEnabled()) {
394     						LOGGER.debug("Picked " + _stmt + " in " + method);
395     					}
396     				}
397     			}
398     
399     			/*
400     			 * It might be the case that all tails are control dependent on some dependee and none of these dependees are
401     			 * in the slice.  In this case, _flag will always be false in the above scenario.  In such cases, we pick the
402     			 * first return statement.  If none found, then first throw statetement. If none found, some arbitrary tail node.
403     			 */
404     			if (_tailWasNotPicked) {
405     				pickARandomReturnPoint(_tails);
406     			}
407     		}
408     
409     		if (LOGGER.isDebugEnabled()) {
410     			LOGGER.debug("END: Picking return points in " + method);
411     		}
412     	}

414     	/**
415     	 * Processes the given exit statement and also includes it into the slice.   It handles <code>throw</code> statements
416     	 * approriately.  Refer to <code>processThrowStmt</code> for details.
417     	 *
418     	 * @param exitStmt to be included.
419     	 *
420     	 * @pre exitStmt != null
421     	 */
422     	private void processAndIncludeExitStmt(final Stmt exitStmt) {
423     		if (exitStmt instanceof ThrowStmt) {
424     			processThrowStmt((ThrowStmt) exitStmt);
425     		}
426     		collector.includeInSlice(exitStmt);
427     	}

429     	/**
430     	 * Process assignment statements to include the classes of the types appearing in them.
431     	 *
432     	 * @param assignStmt to be processed.
433     	 *
434     	 * @pre assignStmt != null
435     	 */
436     	private void processAssignmentsForTypes(final Stmt assignStmt) {
437     		final Value _rightOp = ((AssignStmt) assignStmt).getRightOp();
438     		Type _type = null;
439     
440     		if (_rightOp instanceof CastExpr) {
441     			final CastExpr _v = (CastExpr) _rightOp;
442     			_type = _v.getCastType();
443     		} else if (_rightOp instanceof InstanceOfExpr) {
444     			final InstanceOfExpr _v = (InstanceOfExpr) _rightOp;
445     			_type = _v.getCheckType();
446     		}
447     
448     		if (_type != null) {
449     			if (_type instanceof ArrayType) {
450     				_type = ((ArrayType) _type).baseType;
451     			}
452     
453     			if (_type instanceof RefType) {
454     				final SootClass _sootClass = ((RefType) _type).getSootClass();
455     				collector.includeInSlice(_sootClass);
456     			}
457     		}
458     	}

460     	/**
461     	 * Marks the traps to be included in the slice.  The statement is assumed to be in the slice.
462     	 *
463     	 * @param method in which the statement occurs.
464     	 * @param stmt will trigger the traps to include.
465     	 * @param bbg is the basic block graph of the method.
466     	 *
467     	 * @pre method != null and stmt != null and bbg != null and collector.hasBeenCollected(stmt)
468     	 */
469     	private void processHandlers(final SootMethod method, final Stmt stmt, final BasicBlockGraph bbg) {
470     		if (LOGGER.isDebugEnabled()) {
471     			LOGGER.debug("BEGIN: Pruning handlers " + stmt + "@" + method);
472     		}
473     
474     		/*
475     		 * Include the first statement of the handler for all traps found to cover atleast one statement included in the
476     		 * slice.
477     		 */
478     		for (final Iterator<Trap> _i = TrapManager.getTrapsAt(stmt, method.retrieveActiveBody()).iterator(); _i.hasNext();) {
479     			final Trap _trap = _i.next();
480     			final IdentityStmt _handlerUnit = (IdentityStmt) _trap.getHandlerUnit();
481     
482     			if (bbg.getEnclosingBlock(_handlerUnit) != null) {
483     				collector.includeInSlice(_handlerUnit);
484     				collector.includeInSlice(_handlerUnit.getLeftOpBox());
485     				collector.includeInSlice(_handlerUnit.getRightOpBox());
486     
487     				final SootClass _exceptionClass = _trap.getException();
488     				collector.includeInSlice(Util.getAncestors(_exceptionClass));
489     				collector.includeInSlice(_exceptionClass);
490     			}
491     		}
492     
493     		if (LOGGER.isDebugEnabled()) {
494     			LOGGER.debug("END: Pruning handlers " + stmt + "@" + method);
495     		}
496     	}

498     	/**
499     	 * For the given method, this method includes the declarations/definitions of methods with identical signature in the
500     	 * super classes to make the slice executable.
501     	 *
502     	 * @param method in which <code>stmt</code> occurs.
503     	 *
504     	 * @pre method != null
505     	 */
506     	private void processMethod(final SootMethod method) {
507     		final Collection<Type> _temp = new HashSet<Type>();
508     
509     		for (final Iterator<SootMethod> _i = Util.findMethodInSuperClassesAndInterfaces(method).iterator(); _i.hasNext();) {
510     			final SootMethod _sm =  _i.next();
511     			collector.includeInSlice(_sm.getDeclaringClass());
512     			collector.includeInSlice(_sm);
513     			_temp.clear();
514     			_temp.add(_sm.getReturnType());
515     			_temp.addAll(_sm.getParameterTypes());
516     
517     			for (final Iterator<Type> _j = _temp.iterator(); _j.hasNext();) {
518     				final Type _type = _j.next();
519     
520     				if (_type instanceof RefType) {
521     					collector.includeInSlice(((RefType) _type).getSootClass());
522     				} else if (_type instanceof ArrayType) {
523     					final Type _baseType = ((ArrayType) _type).baseType;
524     
525     					if (_baseType instanceof RefType) {
526     						collector.includeInSlice(((RefType) _baseType).getSootClass());
527     					}
528     				}
529     			}
530     			methodWorkBag.addWorkNoDuplicates(_sm);
531     		}
532     	}

534     	/**
535     	 * Process the statements in the slice body of the given method.
536     	 *
537     	 * @param method whose statements need to be processed.
538     	 *
539     	 * @pre method != null and method.isConcrete()
540     	 */
541     	private void processStmts(final SootMethod method) {
542     		if (LOGGER.isDebugEnabled()) {
543     			LOGGER.debug("Picking up identity statements and methods required in" + method);
544     		}
545     
546     		final BasicBlockGraph _bbg = bbgMgr.getBasicBlockGraph(method);
547     		final UnitGraph _unitGraph = _bbg.getStmtGraph();
548     
549     		for (final Iterator<Stmt> _i = _unitGraph.iterator(); _i.hasNext();) {
550     			final Stmt _stmt = _i.next();
551     
552     			if (_stmt instanceof IdentityStmt) {
553     				final IdentityStmt _identityStmt = (IdentityStmt) _stmt;
554     				final Value _rhs = _identityStmt.getRightOp();
555     
556     				if (_rhs instanceof ThisRef || _rhs instanceof ParameterRef) {
557     					collector.includeInSlice(_identityStmt.getLeftOpBox());
558     					collector.includeInSlice(_identityStmt.getRightOpBox());
559     					collector.includeInSlice(_identityStmt);
560     
561     					if (LOGGER.isDebugEnabled()) {
562     						LOGGER.debug("Picked " + _identityStmt + " in " + method);
563     					}
564     					processHandlers(method, _stmt, _bbg);
565     					stmtCollected = true;
566     				}
567     			} else if (collector.hasBeenCollected(_stmt)) {
568     				if (_stmt.containsInvokeExpr() && !(_stmt.getInvokeExpr() instanceof StaticInvokeExpr)) {
569     					/*
570     					 * If an invoke expression occurs in the slice, the slice will include only the invoked method and not any
571     					 * incarnations of it in it's ancestral classes.  This will lead to unverifiable system of classes.
572     					 * This can be fixed by sucking all the method definitions that need to make the system verifiable
573     					 * and empty bodies will be substituted for such methods.
574     					 */
575     					processMethod(_stmt.getInvokeExpr().getMethod());
576     
577     					if (LOGGER.isDebugEnabled()) {
578     						LOGGER.debug("Included method invoked at " + _stmt + " in " + method);
579     					}
580     				} else if (_stmt instanceof ThrowStmt) {
581     					processThrowStmt((ThrowStmt) _stmt);
582     				} else if (_stmt instanceof AssignStmt) {
583     					processAssignmentsForTypes(_stmt);
584     				}
585     				processHandlers(method, _stmt, _bbg);
586     				stmtCollected = true;
587     			}
588     		}
589     	}
590     
591     	/**
592     	 * Processes the throw statement to include any required classes, if necessary.
593     	 *
594     	 * @param throwStmt to be processed.
595     	 *
596     	 * @pre throwStmt != null
597     	 */
598     	private void processThrowStmt(final ThrowStmt throwStmt) {
599     		if (!collector.hasBeenCollected(throwStmt.getOpBox())) {
600     			final SootClass _exceptionClass = ((RefType) throwStmt.getOp().getType()).getSootClass();
601     			collector.includeInSlice(_exceptionClass);
602     
603     			// the ancestors will be added when we fix up the class hierarchy.
604     			if (LOGGER.isDebugEnabled()) {
605     				LOGGER.debug("Included classes of exception thrown at " + throwStmt);
606     			}
607     		}
608     	}
609     }
