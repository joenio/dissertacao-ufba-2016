46     /**
47      * This class provides use-def information for local variables in a method. The analysis is performed at statement level.
48      * 
49      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
50      * @author $Author: rvprasad $
51      * @version $Revision: 1.19 $ $Date: 2007/02/10 19:07:31 $
52      */
53     public final class LocalUseDefAnalysis
54     		implements IUseDefInfo<DefinitionStmt, Pair<Local, Stmt>> {
55     
56     	/**
57     	 * A map from local and statement pair to a collection of def statement.
58     	 */
59     	private final Map<Pair<Local, Stmt>, Collection<DefinitionStmt>> defInfo = new HashMap<Pair<Local, Stmt>, Collection<DefinitionStmt>>();
60     
61     	/**
62     	 * A list of statements in the given method.
63     	 */
64     	private List<Stmt> stmtList;
65     
66     	/**
67     	 * The control flow graph used to calculate the use-def info.
68     	 */
69     	private UnitGraph unitGraph;
70     
71     	/**
72     	 * A map from definition statement to a collection of statements.
73     	 */
74     	private final Map<DefinitionStmt, Collection<Pair<Local, Stmt>>> useInfo = new HashMap<DefinitionStmt, Collection<Pair<Local, Stmt>>>();
75     
76     	/**
77     	 * Creates a new LocalDefsAnalysis object.
78     	 * 
79     	 * @param graph is the control flow graph used to calculate the use-def info.
80     	 * @pre graph != null
81     	 */
82     	public LocalUseDefAnalysis(final UnitGraph graph) {
83     		final Body _body = graph.getBody();
84     		final PatchingChain _stmts = _body.getUnits();
85     		final BitSet[][] _l2defs = new BitSet[_stmts.size()][_body.getLocalCount()];
86     		stmtList = new ArrayList<Stmt>();
87     		stmtList.addAll(_stmts);
88     
89     		final List<Local> _listOfLocals = new ArrayList<Local>();
90     		_listOfLocals.addAll(_body.getLocals());
91     		unitGraph = graph;
92     		analyze(_l2defs, _listOfLocals);
93     		extract(_l2defs, _listOfLocals);
94     		unitGraph = null;
95     		stmtList = null;
96     	}
97     
98     	/**
99     	 * Retrieves the definitions of <code>local</code> that reach <code>stmt</code>.
100     	 * 
101     	 * @param local variable.
102     	 * @param stmt in which <code>local</code> occurs.
103     	 * @param method <i>ignored</i>.
104     	 * @return a collection of def statements.
105     	 * @pre local != null and stmt != null
106     	 * @post result != null
107     	 */
108     	public Collection<DefinitionStmt> getDefs(final Local local, final Stmt stmt,
109     			@SuppressWarnings("unused") final SootMethod method) {
110     		return Collections.unmodifiableCollection(MapUtils.queryCollection(defInfo, new Pair<Local, Stmt>(local, stmt)));
111     	}
112     
113     	/**
114     	 * {@inheritDoc}
115     	 * 
116     	 * @param method <i>ignored</i>.
117     	 */
118     	public Collection<DefinitionStmt> getDefs(final Stmt useStmt, @SuppressWarnings("unused") final SootMethod method) {
119     		final Collection<DefinitionStmt> _result = new HashSet<DefinitionStmt>();
120     
121     		for (final Iterator<ValueBox> _i = useStmt.getUseBoxes().iterator(); _i.hasNext();) {
122     			final ValueBox _vb = _i.next();
123     			final Value _value = _vb.getValue();
124     
125     			if (_value instanceof Local) {
126     				_result.addAll(getDefs((Local) _value, useStmt, null));
127     			}
128     		}
129     		return _result;
130     	}

162     	/**
163     	 * Performs the analysis to calculate the info.
164     	 * 
165     	 * @param local2defs maps a statment to a collection of def sets of each locals in the method.
166     	 * @param listOfLocals is a list of the locals.
167     	 * @pre local2defs != null and listOfLocals != null
168     	 */
169     	private void analyze(final BitSet[][] local2defs, final List<Local> listOfLocals) {
170     		final IWorkBag<Stmt> _wb = new FIFOWorkBag<Stmt>();
171     		final BitSet _temp = new BitSet();
172     		final Map<Stmt, Collection<Integer>> _stmt2localIndices = new HashMap<Stmt, Collection<Integer>>();
173     
174     		_wb.addAllWork(seedDefInfo(local2defs, listOfLocals, _stmt2localIndices));
175     
176     		while (_wb.hasWork()) {
177     			final Stmt _stmt = _wb.getWork();
178     			final BitSet[] _defsAtStmt = local2defs[stmtList.indexOf(_stmt)];
179     			final Collection<Integer> _localIndices = _stmt2localIndices.get(_stmt);
180     
181     			// if the current statement is a def stmt then remove the index of the local being killed via definition
182     			if (_stmt instanceof DefinitionStmt) {
183     				final Value _leftOp = ((DefinitionStmt) _stmt).getLeftOp();
184     
185     				if (_leftOp instanceof Local) {
186     					_localIndices.remove(new Integer(listOfLocals.indexOf(_leftOp)));
187     				}
188     			}
189     
190     			// propagate the local defs to the successors
191     			for (final Iterator<Integer> _i = _localIndices.iterator(); _i.hasNext();) {
192     				final Integer _localIndexValue = _i.next();
193     				final int _localIndex = _localIndexValue.intValue();
194     				final BitSet _defsOfLocalAtStmt = _defsAtStmt[_localIndex];
195     
196     				for (final Iterator<Stmt> _j = unitGraph.getSuccsOf(_stmt).iterator(); _j.hasNext();) {
197     					final Stmt _succ = _j.next();
198     					final int _succIndex = stmtList.indexOf(_succ);
199     					final BitSet[] _defsAtSucc = local2defs[_succIndex];
200     
201     					if (_defsAtSucc[_localIndex] == null) {
202     						_defsAtSucc[_localIndex] = new BitSet();
203     					}
204     
205     					_temp.clear();
206     					_temp.or(_defsOfLocalAtStmt);
207     					_temp.andNot(_defsAtSucc[_localIndex]);
208     
209     					if (_temp.cardinality() > 0) {
210     						_defsAtSucc[_localIndex].or(_defsOfLocalAtStmt);
211     						_wb.addWorkNoDuplicates(_succ);
212     						_stmt2localIndices.get(_succ).add(_localIndexValue);
213     					}
214     				}
215     			}
216     			_localIndices.clear();
217     		}
218     	}

220     	/**
221     	 * Extracts the info encoded in bits and captures as occurring between entities.
222     	 * 
223     	 * @param local2defs maps a statment to a collection of def sets of each locals in the method.
224     	 * @param localList is a list of the locals.
225     	 * @pre local2defs != null and listOfLocals != null
226     	 */
227     	private void extract(final BitSet[][] local2defs, final List<Local> localList) {
228     		final PairManager _pairMgr = new PairManager(false, true);
229     
230     		for (final Iterator<Stmt> _i = unitGraph.iterator(); _i.hasNext();) {
231     			final Stmt _stmt = _i.next();
232     			final int _stmtIndex = stmtList.indexOf(_stmt);
233     			final BitSet[] _stmtDefSet = local2defs[_stmtIndex];
234     
235     			for (final Iterator<ValueBox> _j = _stmt.getUseBoxes().iterator(); _j.hasNext();) {
236     				final ValueBox _ub = _j.next();
237     
238     				if (_ub.getValue() instanceof Local) {
239     					final Local _local = (Local) _ub.getValue();
240     					final Pair<Local, Stmt> _pair = _pairMgr.getPair(_local, _stmt);
241     					final int _lindex = localList.indexOf(_local);
242     					final BitSet _defs = _stmtDefSet[_lindex];
243     
244     					if (_defs != null && !_defs.isEmpty()) {
245     
246     						for (int _k = _defs.nextSetBit(0); _k >= 0; _k = _defs.nextSetBit(_k + 1)) {
247     							final DefinitionStmt _defStmt = (DefinitionStmt) stmtList.get(_k);
248     							final Local _l = (Local) _defStmt.getLeftOp();
249     							if (_l == _local) {
250     								MapUtils.putIntoCollectionInMap(useInfo, _defStmt, _pair);
251     								MapUtils.putIntoCollectionInMap(defInfo, _pair, _defStmt);
252     							}
253     						}
254     					}
255     				}
256     			}
257     		}
258     	}

260     	/**
261     	 * Captures definition info into <code>local2defs</code>.
262     	 * 
263     	 * @param local2defs maps a statment to a collection of def sets of each locals in the method.
264     	 * @param listOfLocals is a list of the locals.
265     	 * @param stmt2localIndices maps a stmt to the set of local indices
266     	 * @return a collection of definition statements.
267     	 * @pre local2defs != null and listOfLocals != null and stmt2localIndices != null
268     	 * @post result != null
269     	 */
270     	private Collection<Stmt> seedDefInfo(final BitSet[][] local2defs, final List<Local> listOfLocals,
271     			final Map<Stmt, Collection<Integer>> stmt2localIndices) {
272     		final Collection<Stmt> _result = new ArrayList<Stmt>();
273     		final Collection<DefinitionStmt> _defStmts = new ArrayList<DefinitionStmt>();
274     
275     		for (final Iterator<Stmt> _i = unitGraph.iterator(); _i.hasNext();) {
276     			final Stmt _stmt = _i.next();
277     			stmt2localIndices.put(_stmt, new HashSet<Integer>());
278     
279     			if (_stmt instanceof DefinitionStmt) {
280     				_defStmts.add((DefinitionStmt) _stmt);
281     			}
282     		}
283     
284     		for (final Iterator<DefinitionStmt> _i = _defStmts.iterator(); _i.hasNext();) {
285     			final DefinitionStmt _stmt = _i.next();
286     			final Value _value = _stmt.getLeftOp();
287     
288     			if (_value instanceof Local && listOfLocals.contains(_value)) {
289     				final int _localIndex = listOfLocals.indexOf(_value);
290     				final int _stmtIndex = stmtList.indexOf(_stmt);
291     
292     				for (final Iterator<Stmt> _j = unitGraph.getSuccsOf(_stmt).iterator(); _j.hasNext();) {
293     					final Stmt _succ = _j.next();
294     					final int _succIndex = stmtList.indexOf(_succ);
295     					BitSet _temp = local2defs[_succIndex][_localIndex];
296     
297     					if (_temp == null) {
298     						_temp = new BitSet();
299     						local2defs[_succIndex][_localIndex] = _temp;
300     					}
301     					_temp.set(_stmtIndex, true);
302     					_result.add(_succ);
303     					stmt2localIndices.get(_succ).add(new Integer(_localIndex));
304     				}
305     			}
306     		}
307     		return _result;
308     	}
