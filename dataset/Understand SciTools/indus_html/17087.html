102     	/**
103     	 * Returns a stringized representation of this analysis. The representation includes the results of the analysis.
104     	 *
105     	 * @return a stringized representation of this object.
106     	 * @post result != null
107     	 */
108     	@Override public final String toString() {
109     		final StringBuffer _result = new StringBuffer("Statistics for control dependence as calculated by "
110     				+ getClass().getName() + "\n");
111     		int _localEdgeCount;
112     		int _localEntryPointDep;
113     		int _edgeCount = 0;
114     		int _entryPointDep = 0;
115     
116     		final StringBuffer _temp = new StringBuffer();
117     
118     		for (final Iterator<Map.Entry<SootMethod, List<Collection<Stmt>>>> _i = dependent2dependee.entrySet().iterator(); _i
119     				.hasNext();) {
120     			final Map.Entry<SootMethod, List<Collection<Stmt>>> _entry = _i.next();
121     			final SootMethod _method = _entry.getKey();
122     			final List<Stmt> _stmts = getStmtList(_method);
123     			final List<Collection<Stmt>> _cd = _entry.getValue();
124     			_localEdgeCount = 0;
125     			_localEntryPointDep = _stmts.size();
126     
127     			for (int _j = 0; _j < _stmts.size(); _j++) {
128     				if (_cd == null || _cd.isEmpty()) {
129     					continue;
130     				}
131     
132     				final Collection<Stmt> _dees = _cd.get(_j);
133     
134     				if (_dees != null) {
135     					_temp.append("\t\t" + _stmts.get(_j) + " --> " + _dees + "\n");
136     					_localEdgeCount += _dees.size();
137     					_localEntryPointDep--;
138     				}
139     			}
140     
141     			_result.append("\tFor " + _entry.getKey() + " there are " + _localEdgeCount + " control dependence edges with "
142     					+ _localEntryPointDep + " entry point dependences.\n");
143     			_result.append(_temp);
144     			_temp.delete(0, _temp.length());
145     			_edgeCount += _localEdgeCount;
146     			_entryPointDep += _localEntryPointDep;
147     		}
148     		_result.append("A total of " + _edgeCount + " control dependence edges exists with " + _entryPointDep
149     				+ " entry point dependences.");
150     		return _result.toString();
151     	}

160     	/**
161     	 * Retrieves the fan out number of the given basic block.
162     	 *
163     	 * @param basicblock of interest.
164     	 * @return the fan out number.
165     	 * @pre basicblock != null
166     	 */
167     	protected final int getFanoutNumOf(final BasicBlock basicblock) {
168     		if (!node2fanout.containsKey(basicblock)) {
169     			int _fanout = basicblock.getSuccsOf().size();
170     
171     			if (_fanout > 0 && basicblock.isAnExitBlock()) {
172     				_fanout++;
173     			}
174     			node2fanout.put(basicblock, _fanout);
175     		}
176     		return node2fanout.get(basicblock);
177     	}

200     	/**
201     	 * A helper method to extract dependence.
202     	 *
203     	 * @param stmt for which dependence is requested.
204     	 * @param method in which <code>stmt</code> occurs.
205     	 * @param list from which to extract dependence.
206     	 * @return the collection of statement on which <code>stmt</code> is control dependent on.
207     	 * @pre stmt != null and method != null and list.oclIsKindOf(Collection(Stmt))
208     	 * @post result != null
209     	 */
210     	private Collection<Stmt> getDependenceHelper(final Object stmt, final Object method, final List<Collection<Stmt>> list) {
211     		Collection<Stmt> _result = Collections.emptyList();
212     
213     		if (list != null) {
214     			final int _index = getStmtList((SootMethod) method).indexOf(stmt);
215     
216     			if (_index > -1) {
217     				final List<Stmt> _l = Collections.emptyList();
218     				_result = ListUtils.getAtIndexFromList(list, _index, _l);
219     			}
220     		}
221     		return _result;
222     	}
