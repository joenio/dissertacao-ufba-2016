17     package edu.ksu.cis.indus.common.datastructures;
18     
19     import edu.ksu.cis.indus.annotations.Empty;
20     import edu.ksu.cis.indus.annotations.Functional;
21     import edu.ksu.cis.indus.annotations.Immutable;
22     import edu.ksu.cis.indus.annotations.NonNull;
23     
24     import java.util.ArrayList;
25     import java.util.List;
26     
27     /**
28      * This class provides the basic implementation for elements to be used in fast-union-find algorithm as defined by Aho,
29      * Ullman, and Sethi in the "Dragon" book.
30      * 
31      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
32      * @author $Author: rvprasad $
33      * @version $Revision: 1.11 $ $Date: 2007/02/10 19:08:38 $
34      * @param <T> the type parameter for this class.
35      */
36     public class FastUnionFindElement<T extends FastUnionFindElement<T>> {
37     
38     	/**
39     	 * A sequence of children elements of this element.
40     	 */
41     	protected List<T> children;
42     
43     	/**
44     	 * This is the set to which this element belongs to.
45     	 */
46     	protected T set;
47     
48     	/**
49     	 * This is the type associated with this element.
50     	 */
51     	protected Object type;
52     
53     	/**
54     	 * Creates an instance of this class.
55     	 */
56     	@Empty public FastUnionFindElement() {
57     		// does nothing
58     	}
59     
60     	/**
61     	 * Adds a new child to this element.
62     	 * 
63     	 * @param child to be added.
64     	 */
65     	public final void addChild(@NonNull @Immutable final T child) {
66     		if (children == null) {
67     			children = new ArrayList<T>();
68     		}
69     		children.add(child);
70     	}
71     
72     	/**
73     	 * Retrieves the element that represents the equivalence class to which this element belongs to.
74     	 * 
75     	 * @return the representative element.
76     	 */
77     	@NonNull public final T find() {
78     		@SuppressWarnings("unchecked") T _result = (T) this;
79     
80     		while (_result.set != null) {
81     			_result = _result.set;
82     		}
83     
84     		if (_result != this) {
85     			set = _result;
86     		}
87     		return _result;
88     	}
89     
90     	/**
91     	 * Retrieves the type of this element.
92     	 * 
93     	 * @return the type of this element.
94     	 */
95     	@Functional public final Object getType() {
96     		return find().type;
97     	}
98     
99     	/**
100     	 * Checks if this element has any children.
101     	 * 
102     	 * @return <code>true</code> if it has children; <code>false</code>, otherwise.
103     	 */
104     	@Functional public final boolean isAtomic() {
105     		return children == null || children.size() == 0;
106     	}
107     
108     	/**
109     	 * Checks if this element is bound to a type.
110     	 * 
111     	 * @return <code>true</code> if it is bound to a type; <code>false</code>, otherwise.
112     	 */
113     	@Functional public final boolean isBound() {
114     		return find().type != null;
115     	}
116     
117     	/**
118     	 * Checks if the given element and this element represent the same type. As this implementation does not deal with types,
119     	 * it assumes no two element belong to the same type, hence, always returns <code>false</code>.
120     	 * 
121     	 * @param e is the element to checked for equivalence.
122     	 * @return <code>false</code>
123     	 */
124     	@Functional public final boolean sameType(@NonNull @Immutable final T e) {
125     		boolean _result = false;
126     
127     		if (e.type != null && type != null) {
128     			_result = e.type.equals(type);
129     		}
130     		return _result;
131     	}
132     
133     	/**
134     	 * Set the type of the element.
135     	 * 
136     	 * @param theType of the element.
137     	 * @throws IllegalStateException when the type of the element is already set.
138     	 */
139     	public final void setType(@NonNull @Immutable final Object theType) {
140     		if (set == null) {
141     			if (type == null) {
142     				type = theType;
143     			} else {
144     				throw new IllegalStateException("Cannot set a type on an element with a fixed type.");
145     			}
146     		} else {
147     			find().setType(theType);
148     		}
149     	}
150     
151     	/**
152     	 * Unifies the given element with this element.
153     	 * 
154     	 * @param e is the element to be unified with this element.
155     	 * @return <code>true</code> if this element was unified with the given element; <code>false</code>, otherwise.
156     	 */
157     	public final boolean unify(@NonNull final T e) {
158     		boolean _result = false;
159     		final T _a = find();
160     		final T _b = e.find();
161     
162     		if (_a == _b || _a.sameType(_b)) {
163     			_result = true;
164     		} else if (!(_a.isAtomic() || _b.isAtomic())) {
165     			_a.union(_b);
166     			_result = _a.unifyChildren(_b);
167     		} else if (!(_a.isBound() && _b.isBound())) {
168     			_a.union(_b);
169     			_result = true;
170     		}
171     
172     		return _result;
173     	}
174     
175     	/**
176     	 * Unifies the children of this element with that of the given element.
177     	 * 
178     	 * @param e is the element whose children needs to be unified with that of this element.
179     	 * @return <code>true</code>, if the tree rooted at this element and at <code>e</code> was unified successfully;
180     	 *         <code>false</code>, otherwise.
181     	 */
182     	public final boolean unifyChildren(@NonNull final T e) {
183     		boolean _result = false;
184     
185     		if (children != null && e.children != null && children.size() == e.children.size()) {
186     			_result = true;
187     
188     			for (int _i = children.size() - 1; _i >= 0 && _result; _i--) {
189     				final T _c1 = children.get(_i);
190     				final T _c2 = e.children.get(_i);
191     				_result &= _c1.unify(_c2);
192     			}
193     		}
194     		return _result;
195     	}
196     
197     	/**
198     	 * Merges the equivalence classes containing <code>e</code> and this element.
199     	 * 
200     	 * @param e is the element in an equivalence class that needs to be merged.
201     	 */
202     	public final void union(@NonNull final FastUnionFindElement<T> e) {
203     		final T _a = find();
204     		final T _b = e.find();
205     
206     		if (_a != _b) {
207     			if (_b.isBound()) {
208     				_a.set = _b;
209     			} else {
210     				// if a.isBound() or neither is bound
211     				_b.set = _a;
212     			}
213     		}
214     	}
215     }
