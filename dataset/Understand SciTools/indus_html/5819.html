22     /**
23      * This class tests directed graph and simple node graph implementations with a different graph instance.  All the local
24      * tests are overloaded suitably.
25      * 
26      * <p>
27      * This represents the CFG as generated by <b>javac</b> for the following snippet.
28      * </p>
29      * <pre>
30      *  public static void main(String[] s) {
31      *       int i = 0;
32      *       do {
33      *           while(i > 0) {
34      *               i--;
35      *           };
36      *           i++;
37      *       } while (i < 10);
38      *       System.out.println("Hi");
39      *   }
40      * </pre>
41      *
42      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
43      * @author $Author: rvprasad $
44      * @version $Revision: 1.11 $ $Date: 2007/02/10 19:08:51 $
45      */
46     public class JavacBasedDirectedAndSimpleNodeGraphTest
47       extends JikesBasedDirectedAndSimpleNodeGraphTest {
48     	/**
49     	 * @see edu.ksu.cis.indus.common.graph.SimpleNodeGraphTest#testlocalGetSinks()
50     	 */
51     	public void testlocalGetSinks() {
52     		assertFalse(dg.getSinks().isEmpty());
53     		assertTrue(dg.getSinks().contains(name2node.get("e")));
54     	}
55     
56     	/**
57     	 * @see JikesBasedDirectedAndSimpleNodeGraphTest#setUp()
58     	 */
59     	protected void setUp()
60     	  throws Exception {
61     		final SimpleNodeGraph<String> _sng = new SimpleNodeGraph<String>();
62     		name2node.put("a", _sng.getNode("a"));
63     		name2node.put("b", _sng.getNode("b"));
64     		name2node.put("c", _sng.getNode("c"));
65     		name2node.put("d", _sng.getNode("d"));
66     		name2node.put("e", _sng.getNode("e"));
67     
68     		_sng.addEdgeFromTo( name2node.get("a"),  name2node.get("b"));
69     		_sng.addEdgeFromTo( name2node.get("b"),  name2node.get("c"));
70     		_sng.addEdgeFromTo( name2node.get("b"),  name2node.get("d"));
71     		_sng.addEdgeFromTo( name2node.get("d"),  name2node.get("e"));
72     		// add loop edges
73     		_sng.addEdgeFromTo( name2node.get("c"),  name2node.get("b"));
74     		_sng.addEdgeFromTo( name2node.get("d"),  name2node.get("b"));
75     		setSNG(_sng);
76     
77     		numberOfCycles = 2;
78     	}
79     
80     	/**
81     	 * @see edu.ksu.cis.indus.common.graph.AbstractDirectedGraphTest#localtestIsAncestorOf()
82     	 */
83     	protected void localtestIsAncestorOf() {
84     		assertTrue(sng.isAncestorOf( name2node.get("a"),  name2node.get("a")));
85     		assertTrue(sng.isAncestorOf( name2node.get("a"),  name2node.get("e")));
86     		assertFalse(sng.isAncestorOf( name2node.get("e"),  name2node.get("b")));
87     	}
88     
89     	/**
90     	 * @see edu.ksu.cis.indus.common.graph.AbstractDirectedGraphTest#localtestIsReachable()
91     	 */
92     	protected void localtestIsReachable() {
93     		assertTrue(sng.isReachable( name2node.get("a"),  name2node.get("e"), true));
94     		assertTrue(sng.isReachable( name2node.get("e"),  name2node.get("a"), false));
95     		assertTrue(sng.isReachable( name2node.get("c"),  name2node.get("e"), true));
96     		assertTrue(sng.isReachable( name2node.get("d"),  name2node.get("c"), true));
97     	}
98     }
