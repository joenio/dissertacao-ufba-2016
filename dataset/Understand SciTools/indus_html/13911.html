17     package edu.ksu.cis.indus.common.scoping;
18     
19     import edu.ksu.cis.indus.annotations.Functional;
20     import edu.ksu.cis.indus.annotations.NonNull;
21     
22     import org.apache.commons.lang.builder.ToStringBuilder;
23     
24     /**
25      * This class captures attributes common all scope specifications.
26      * 
27      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
28      * @author $Author: rvprasad $
29      * @version $Revision: 1.9 $ $Date: 2007/02/10 19:08:39 $
30      */
31     abstract class AbstractSpecification
32     		implements ISpecification {
33     
34     	/**
35     	 * This is the access control specification.
36     	 */
37     	private AccessSpecification accessSpec = new AccessSpecification();
38     
39     	/**
40     	 * This indicates if the specification should be interpreted as inclusive or exclusive.
41     	 */
42     	private boolean inclusion = true;
43     
44     	/**
45     	 * The name of this specification.
46     	 */
47     	@NonNull private String name;
48     
49     	/**
50     	 * {@inheritDoc}
51     	 */
52     	@Functional public final String getName() {
53     		return name;
54     	}
55     
56     	/**
57     	 * {@inheritDoc}
58     	 */
59     	public final void setAccessSpec(final AccessSpecification accessSpecification) {
60     		this.accessSpec = accessSpecification;
61     	}
62     
63     	/**
64     	 * {@inheritDoc}
65     	 */
66     	public final void setInclusion(final boolean value) {
67     		this.inclusion = value;
68     	}
69     
70     	/**
71     	 * {@inheritDoc}
72     	 */
73     	public final void setName(final String nameOfTheSpec) {
74     		this.name = nameOfTheSpec;
75     	}
76     
77     	/**
78     	 * {@inheritDoc}
79     	 */
80     	@Functional @NonNull @Override public String toString() {
81     		return new ToStringBuilder(this).append("inclusion", this.inclusion).append("accessSpec", this.accessSpec).append(
82     				"name", this.name).toString();
83     	}
84     
85     	/**
86     	 * Checks if the given access specifier conformant.
87     	 * 
88     	 * @param accessSpecifier to be tested.
89     	 * @return <code>true</code> if it is conformant; <code>false</code>, otherwise.
90     	 */
91     	protected final boolean accessConformant(@NonNull final AccessSpecifierWrapper accessSpecifier) {
92     		return accessSpec.conformant(accessSpecifier);
93     	}
94     
95     	/**
96     	 * Retrieves the value in <code>inclusion</code>.
97     	 * 
98     	 * @return the value in <code>inclusion</code>.
99     	 */
100     	boolean isInclusion() {
101     		return inclusion;
102     	}
103     }
