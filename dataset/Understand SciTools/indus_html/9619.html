17     package edu.ksu.cis.indus.staticanalyses.flow.instances.ofa;
18     
19     import edu.ksu.cis.indus.staticanalyses.flow.IFGNode;
20     import edu.ksu.cis.indus.staticanalyses.flow.IFGNodeConnector;
21     
22     /**
23      * This class encapsulates the logic to connect ast flow graph nodes with non-ast flow graph nodes when the ast nodes
24      * correspond to l-values.
25      * <p>
26      * Created: Wed Jan 30 15:19:44 2002
27      * </p>
28      *
29      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
30      * @version $Revision: 1.10 $
31      * @param <N>  is the type of the token set object.
32      */
33     class LHSConnector<N extends IFGNode<?, ?, N>>
34     		implements IFGNodeConnector<N> {
35     
36     	/**
37     	 * Connects the given ast flow graph node to the non-ast flow graph node. This is used to connect flow nodes corresponding
38     	 * to LHS expressions.
39     	 *
40     	 * @param ast the ast flow graph node to be connected.
41     	 * @param nonast the non-ast flow graph node to be connnected.
42     	 * @pre ast != null and nonast != null
43     	 */
44     	public void connect(final N ast, final N nonast) {
45     		ast.addSucc(nonast);
46     	}
47     }
