17     package edu.ksu.cis.indus.common.soot;
18     
19     import edu.ksu.cis.indus.annotations.Immutable;
20     import edu.ksu.cis.indus.annotations.NonNull;
21     import edu.ksu.cis.indus.common.scoping.SpecificationBasedScopeDefinition;
22     import edu.ksu.cis.indus.interfaces.IEnvironment;
23     
24     import soot.SootMethod;
25     import soot.toolkits.graph.UnitGraph;
26     
27     /**
28      * This is the interface via which the user can plugin various sorts of unit graphs into the analyses and also reuse the same
29      * implementation at many places.
30      * 
31      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
32      * @author $Author: rvprasad $
33      * @version $Revision: 1.10 $ $Date: 2007/02/10 19:08:36 $
34      * @param <T> the type of the manufactured graph.
35      */
36     public interface IStmtGraphFactory<T extends UnitGraph> {
37     
38     	/**
39     	 * The id of this interface.
40     	 */
41     	Comparable<?> ID = "Statement Graph IFactory";
42     
43     	/**
44     	 * Retrieves the unit graph of the given method.
45     	 * 
46     	 * @param method for which the unit graph is requested.
47     	 * @return the requested unit graph.
48     	 */
49     	@NonNull T getStmtGraph(final SootMethod method);
50     
51     	/**
52     	 * Sets the scope specification.
53     	 * 
54     	 * @param scopeDef is the scope definition.
55     	 * @param env is the environment in which the scope is defined.
56     	 */
57     	void setScope(@Immutable final SpecificationBasedScopeDefinition scopeDef, @NonNull @Immutable final IEnvironment env);
58     
59     	/**
60     	 * Resets all internal datastructures.
61     	 */
62     	void reset();
63     }
