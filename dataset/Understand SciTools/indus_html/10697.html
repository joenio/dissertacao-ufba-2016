44     /**
45      * This is a specialized version of <code>UnitGraph</code> in which the control flow edges based on exceptions can be
46      * controlled. The user can specify the names of exceptions and the control flow via the throw of these exceptions will not be
47      * included in the graph.
48      * 
49      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
50      * @author $Author: rvprasad $
51      * @version $Revision: 1.22 $ $Date: 2007/02/10 19:08:36 $
52      */
53     final class ExceptionFlowSensitiveStmtGraph
54     		extends UnitGraph {
55     
56     	/**
57     	 * The CFG edges based on the exceptions named here should not be considered during CFG pruning.
58     	 */
59     	private static final List<String> NAMES_OF_EXCEPTIONS_TO_IGNORE_WHILE_CFG_PRUNING;
60     
61     	static {
62     		NAMES_OF_EXCEPTIONS_TO_IGNORE_WHILE_CFG_PRUNING = new ArrayList<String>();
63     		NAMES_OF_EXCEPTIONS_TO_IGNORE_WHILE_CFG_PRUNING.add("java.lang.RuntimeException");
64     		NAMES_OF_EXCEPTIONS_TO_IGNORE_WHILE_CFG_PRUNING.add("java.lang.Exception");
65     		NAMES_OF_EXCEPTIONS_TO_IGNORE_WHILE_CFG_PRUNING.add("java.lang.Throwable");
66     		NAMES_OF_EXCEPTIONS_TO_IGNORE_WHILE_CFG_PRUNING.add("java.lang.Error");
67     	}
68     
69     	/**
70     	 * The sequence of units represented in this graph.
71     	 */
72     	@NonNull @NonNullContainer private List<Stmt> nodes;
73     
74     	/**
75     	 * A cache of the nodes for which predecessors need to be fixed after processing.
76     	 */
77     	@NonNull @NonNullContainer private Collection<Stmt> predsToBeProcessedCache;
78     
79     	/**
80     	 * A cache of the nodes for which successors need to be fixed after processing.
81     	 */
82     	@NonNull @NonNullContainer private Collection<Stmt> succsToBeProcessedCache;
83     
84     	/**
85     	 * Creates an instance of this unit graph corresponding to the given body and options.
86     	 * 
87     	 * @param unitBody is the body for which the graph should be constructed.
88     	 * @param namesOfExceptionsToIgnore is the fully qualified names of the exceptions. Control flow based on these exceptions
89     	 *            are not captured in the graph.
90     	 * @param exceptionEdges <code>true</code> indicates that the edges from the predecessors of excepting statements should
91     	 *            be included; <code>false</code>, otherwise.
92     	 */
93     	@SuppressWarnings("unchecked") ExceptionFlowSensitiveStmtGraph(@NonNull final JimpleBody unitBody,
94     			@NonNull final Collection<String> namesOfExceptionsToIgnore, final boolean exceptionEdges) {
95     		super(unitBody, true, exceptionEdges);
96     		predsToBeProcessedCache = new HashSet<Stmt>();
97     		succsToBeProcessedCache = new HashSet<Stmt>();
98     		deleteEdgesResultingFromTheseExceptions(namesOfExceptionsToIgnore);
99     		pruneExceptionBasedControlFlow();
100     
101     		for (final Iterator<Stmt> _i = succsToBeProcessedCache.iterator(); _i.hasNext();) {
102     			final Stmt _unit = _i.next();
103     			unitToSuccs.put(_unit, Collections.unmodifiableCollection((Collection) unitToSuccs.get(_unit)));
104     		}
105     
106     		for (final Iterator<Stmt> _i = predsToBeProcessedCache.iterator(); _i.hasNext();) {
107     			final Stmt _unit = _i.next();
108     			unitToPreds.put(_unit, Collections.unmodifiableCollection((Collection) unitToPreds.get(_unit)));
109     		}
110     		predsToBeProcessedCache = null;
111     		succsToBeProcessedCache = null;
112     		fixupMapsAndIterator();
113     	}
114     
115     	/**
116     	 * Returns an iterator over the statements represented in this graph. The order of the statements will reflect the order
117     	 * of the statements in the sequence of statements obtained from the method. {@inheritDoc}
118     	 */
119     	@Functional @NonNull @Override public Iterator<Stmt> iterator() {
120     		return nodes.iterator();
121     	}
122     
123     	/**
124     	 * Deletes all edges resulting from exceptions named in <code>namesOfExceptionsToIgnore</code>.
125     	 * 
126     	 * @param namesOfExceptionsToIgnore is the fully qualified names of exceptions. Control flow based on these exceptions is
127     	 *            deleted from the graph.
128     	 */
129     	@SuppressWarnings("unchecked") private void deleteEdgesResultingFromTheseExceptions(
130     			@NonNullContainer @NonNull final Collection<String> namesOfExceptionsToIgnore) {
131     		final Chain _traps = body.getTraps();
132     		final Chain _units = body.getUnits();
133     
134     		for (@SuppressWarnings("unchecked") final Iterator<Trap> _j = _traps.iterator(); _j.hasNext();) {
135     			final Trap _trap = _j.next();
136     
137     			if (namesOfExceptionsToIgnore.contains(_trap.getException().getName())) {
138     				final Stmt _handler = (Stmt) _trap.getHandlerUnit();
139     				final Stmt _endUnit = (Stmt) _units.getPredOf(_trap.getEndUnit());
140     
141     				final List<Stmt> _preds = new ArrayList<Stmt>((Collection) unitToPreds.get(_handler));
142     
143     				for (final Iterator<Stmt> _i = _units.iterator(_trap.getBeginUnit(), _endUnit); _i.hasNext();) {
144     					final Stmt _unit = _i.next();
145     					final List<Stmt> _succs = new ArrayList<Stmt>((Collection) unitToSuccs.get(_unit));
146     					_succs.remove(_handler);
147     					unitToSuccs.put(_unit, _succs);
148     					succsToBeProcessedCache.add(_unit);
149     					_preds.remove(_unit);
150     				}
151     				unitToPreds.put(_handler, _preds);
152     				predsToBeProcessedCache.add(_handler);
153     				_j.remove();
154     			}
155     		}
156     	}
157     
158     	/**
159     	 * Fixes up the maps for unreachable statements and collects the reachable nodes to be made available for iterators.
160     	 */
161     	@SuppressWarnings("unchecked") private void fixupMapsAndIterator() {
162     		final List<Object> _temp = new ArrayList<Object>();
163     		final IWorkBag<Object> _wb = new HistoryAwareFIFOWorkBag<Object>(_temp);
164     		final PatchingChain _units = getBody().getUnits();
165     
166     		// find reachable units.
167     		_wb.addWork(_units.getFirst());
168     
169     		while (_wb.hasWork()) {
170     			final Stmt _unit = (Stmt) _wb.getWork();
171     			_wb.addAllWork(getSuccsOf(_unit));
172     		}
173     
174     		nodes = new ArrayList<Stmt>(_units);
175     		nodes.retainAll(_temp);
176     		nodes = Collections.unmodifiableList(nodes);
177     
178     		// fix the maps such that there is no info pertaining to unreachable statements
179     		final Collection<Stmt> _stmts = new ArrayList<Stmt>(_units);
180     		_stmts.removeAll(_temp);
181     
182     		for (final Iterator<?> _i = _stmts.iterator(); _i.hasNext();) {
183     			final Object _stmt = _i.next();
184     			unitToPreds.remove(_stmt);
185     			unitToSuccs.remove(_stmt);
186     		}
187     	}
188     
189     	/**
190     	 * Removes control flow edges based on the matching of the exceptions resulting from source expression and the exception
191     	 * being handled.
192     	 */
193     	@SuppressWarnings("unchecked") private void pruneExceptionBasedControlFlow() {
194     		// process each trapped unit
195     		for (final Iterator<Stmt> _i = TrapManager.getTrappedUnitsOf(body).iterator(); _i.hasNext();) {
196     			final Stmt _unit = _i.next();
197     			final List<Trap> _traps = TrapManager.getTrapsAt(_unit, body);
198     
199     			// gather all the exception types that is assumed to be thrown by the current unit.
200     			for (final Iterator<Trap> _j = _traps.iterator(); _j.hasNext();) {
201     				final Trap _trap = _j.next();
202     				final SootClass _exception = _trap.getException();
203     
204     				if (NAMES_OF_EXCEPTIONS_TO_IGNORE_WHILE_CFG_PRUNING.contains(_exception.getName())) {
205     					continue;
206     				}
207     
208     				final Stmt _handler = (Stmt) _trap.getHandlerUnit();
209     				final boolean _hasArrayRef = _unit.containsArrayRef();
210     				final boolean _hasFieldRef = _unit.containsFieldRef();
211     				final boolean _hasInstanceFieldRef = _hasFieldRef && _unit.getFieldRef() instanceof InstanceFieldRef;
212     				final boolean _hasInvokeExpr = _unit.containsInvokeExpr();
213     				final InvokeExpr _invokeExpr = _hasInvokeExpr ? _unit.getInvokeExpr() : null;
214     				final boolean _hasInstanceInvokeExpr = _hasInvokeExpr && _invokeExpr instanceof InstanceInvokeExpr;
215     
216     				// for the declared caught exception type, validate the declaration and tailor the graph as needed.
217     				boolean _retainflag = (_hasArrayRef || _hasInstanceFieldRef || _hasInstanceInvokeExpr)
218     						&& Util.isDescendentOf(_exception, "java.lang.NullPointerException");
219     				_retainflag |= _hasArrayRef && Util.isDescendentOf(_exception, "java.lang.ArrayIndexOutOfBoundsException");
220     
221     				if (_hasInvokeExpr) {
222     					for (final Iterator<SootClass> _l = _invokeExpr.getMethod().getExceptions().iterator(); _l.hasNext();) {
223     						final SootClass _thrown = _l.next();
224     						_retainflag |= Util.isDescendentOf(_thrown, _exception);
225     					}
226     				}
227     
228     				if (!_retainflag) {
229     					final List<?> _preds = new ArrayList<Stmt>((Collection) unitToPreds.get(_handler));
230     					_preds.remove(_unit);
231     					unitToPreds.put(_handler, _preds);
232     					predsToBeProcessedCache.add(_handler);
233     
234     					final List<?> _succs = new ArrayList<Stmt>((Collection) unitToSuccs.get(_unit));
235     					_succs.remove(_handler);
236     					unitToSuccs.put(_unit, _succs);
237     					succsToBeProcessedCache.add(_unit);
238     				}
239     			}
240     		}
241     	}
242     }
