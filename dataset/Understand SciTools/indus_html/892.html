57     /**
58      * This is the setup in which various tests of dependence analyses are run. The classes to be processed during the test can be
59      * configured via the command line or via specifying
60      * <code>DepedencyAnalysisRegressionTestSuite.DEPENDENCYANALYSIS_TEST_PROPERTIES_FILE</code> system property. The syntax for
61      * both these options is a space separated list of class names.
62      *
63      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
64      * @author $Author: rvprasad $
65      * @version $Revision: 1.40 $ $Date: 2007/02/10 19:07:02 $
66      */
67     public class DependencyAnalysisTestSetup
68     		extends ValueAnalysisTestSetup {
69     
70     	/**
71     	 * The instance of aliased use-def info to use.
72     	 */
73     	private AliasedUseDefInfo aliasUD;
74     
75     	/**
76     	 * The instance of basic block graph manager to use.
77     	 */
78     	private BasicBlockGraphMgr bbgMgr;
79     
80     	/**
81     	 * The collection of dependency analyses being tested.
82     	 */
83     	private Collection<IDependencyAnalysis<?, ?, ?, ?, ?, ?>> das;
84     
85     	/**
86     	 * The instance of equivalence class based escape analysis to use.
87     	 */
88     	private EquivalenceClassBasedEscapeAnalysis ecba;
89     
90     	/**
91     	 * A map used to communicate arguments to various analyses.
92     	 */
93     	private Map info;
94     
95     	/**
96     	 * This provides monitor information.
97     	 */
98     	private MonitorAnalysis monitorInfo;
99     
100     	/**
101     	 * The thread graph to be used.
102     	 */
103     	private ThreadGraph tgiImpl;
104     
105     	/**
106     	 * @see ValueAnalysisTestSetup#ValueAnalysisTestSetup(TestSuite,String,String)
107     	 */
108     	protected DependencyAnalysisTestSetup(final TestSuite test, final String theNameOfClasses, final String classpath) {
109     		super(test, theNameOfClasses, classpath);
110     	}
111     
112     	/**
113     	 * @see junit.extensions.TestSetup#setUp()
114     	 */
115     	@Override protected void setUp() throws Exception {
116     		super.setUp();
117     		bbgMgr = new BasicBlockGraphMgr();
118     		bbgMgr.setStmtGraphFactory(getStmtGraphFactory());
119     
120     		// setup level 1 analysis here.
121     		final ValueAnalyzerBasedProcessingController _pc = new ValueAnalyzerBasedProcessingController();
122     		final OneAllStmtSequenceRetriever _ssr = new OneAllStmtSequenceRetriever();
123     		_ssr.setStmtGraphFactory(getStmtGraphFactory());
124     		_pc.setStmtSequencesRetriever(_ssr);
125     		_pc.setAnalyzer(valueAnalyzer);
126     		_pc.setEnvironment(valueAnalyzer.getEnvironment());
127     		_pc.setProcessingFilter(new TagBasedProcessingFilter(FATestSetup.TAG_NAME));
128     
129     		final PairManager _pairManager = new PairManager(false, true);
130     		tgiImpl = new ThreadGraph(cgiImpl, new CFGAnalysis(cgiImpl, bbgMgr), _pairManager);
131     		tgiImpl.hookup(_pc);
132     		_pc.process();
133     		tgiImpl.unhook(_pc);
134     		aliasUD = new AliasedUseDefInfov2(valueAnalyzer, cgiImpl, tgiImpl, bbgMgr, _pairManager);
135     		ecba = new EquivalenceClassBasedEscapeAnalysis(cgiImpl, null, bbgMgr);
136     		monitorInfo = new MonitorAnalysis();
137     
138     		// setup info
139     		info = new HashMap();
140     		info.put(ICallGraphInfo.ID, cgiImpl);
141     		info.put(IThreadGraphInfo.ID, tgiImpl);
142     		info.put(PairManager.ID, _pairManager);
143     		info.put(IEnvironment.ID, valueAnalyzer.getEnvironment());
144     		info.put(IValueAnalyzer.ID, valueAnalyzer);
145     		info.put(IUseDefInfo.ALIASED_USE_DEF_ID, aliasUD);
146     		info.put(IEscapeInfo.ID, ecba.getEscapeInfo());
147     		info.put(IMonitorInfo.ID, monitorInfo);
148     
149     		// retrieve dependence analysis
150     		das = new HashSet<IDependencyAnalysis<?, ?, ?, ?, ?, ?>>();
151     
152     		for (final Iterator<IDependencyAnalysisTest> _i = TestHelper.getTestCasesReachableFromSuite((TestSuite) getTest(),
153     				IDependencyAnalysisTest.class).iterator(); _i.hasNext();) {
154     			final IDependencyAnalysisTest _test = _i.next();
155     			_test.setEnvironment(valueAnalyzer.getEnvironment());
156     
157     			final IDependencyAnalysis<?, ?, ?, ?, ?, ?> _da = _test.getDA();
158     			das.add(_da);
159     
160     			if (_da.getIds().contains(IDependencyAnalysis.DependenceSort.CONTROL_DA)
161     					&& (_da.getDirection().equals(IDependencyAnalysis.Direction.BI_DIRECTIONAL) || _da.getDirection().equals(
162     							IDependencyAnalysis.Direction.BACKWARD_DIRECTION))) {
163     				MapUtils.putIntoSetInMap(info, IDependencyAnalysis.DependenceSort.CONTROL_DA, _da);
164     			}
165     		}
166     
167     		for (final Iterator<XMLBasedDependencyAnalysisTest> _i = TestHelper.getTestCasesReachableFromSuite(
168     				(TestSuite) getTest(), XMLBasedDependencyAnalysisTest.class).iterator(); _i.hasNext();) {
169     			final XMLBasedDependencyAnalysisTest _test = _i.next();
170     			_test.setCallGraph(cgiImpl);
171     		}
172     
173     		// drive the analysis.
174     		_pc.setProcessingFilter(new CGBasedProcessingFilter(cgiImpl));
175     		aliasUD.hookup(_pc);
176     		_pc.process();
177     		aliasUD.unhook(_pc);
178     
179     		// drive dependency analyses
180     		final AnalysesController _ac = new AnalysesController(info, _pc, bbgMgr);
181     		_ac.addAnalyses(IMonitorInfo.ID, Collections.singleton(monitorInfo));
182     		_ac.addAnalyses(IEscapeInfo.ID, Collections.singleton(ecba));
183     
184     		for (final Iterator<IDependencyAnalysis<?, ?, ?, ?, ?, ?>> _i1 = das.iterator(); _i1.hasNext();) {
185     			final IDependencyAnalysis<?, ?, ?, ?, ?, ?> _da1 = _i1.next();
186     			_da1.reset();
187     
188     			for (final Iterator<? extends Comparable<?>> _i2 = _da1.getIds().iterator(); _i2.hasNext();) {
189     				final Comparable<?> _id = _i2.next();
190     				_ac.addAnalyses(_id, Collections.singleton(_da1));
191     			}
192     		}
193     		_ac.initialize();
194     		_ac.execute();
195     	}
196     
197     	/**
198     	 * @see junit.extensions.TestSetup#tearDown()
199     	 */
200     	@Override protected void tearDown() throws Exception {
201     		bbgMgr.reset();
202     		bbgMgr = null;
203     		// teardown TGI and ECBA here
204     		tgiImpl.reset();
205     		tgiImpl = null;
206     		ecba.reset();
207     		ecba = null;
208     		monitorInfo.reset();
209     		monitorInfo = null;
210     		aliasUD.reset();
211     		aliasUD = null;
212     		info.clear();
213     		info = null;
214     
215     		// teardown the dependency analysis
216     		for (final Iterator<IDependencyAnalysis<?, ?, ?, ?, ?, ?>> _i = das.iterator(); _i.hasNext();) {
217     			_i.next().reset();
218     		}
219     		das.clear();
220     		das = null;
221     		super.tearDown();
222     	}
223     }
