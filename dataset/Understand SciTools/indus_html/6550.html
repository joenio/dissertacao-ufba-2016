17     package edu.ksu.cis.indus.staticanalyses.flow.instances.ofa;
18     
19     import edu.ksu.cis.indus.common.soot.Util;
20     import edu.ksu.cis.indus.staticanalyses.flow.AbstractStmtSwitch;
21     import edu.ksu.cis.indus.staticanalyses.flow.IMethodVariant;
22     import edu.ksu.cis.indus.staticanalyses.tokens.ITokens;
23     
24     import soot.Value;
25     import soot.jimple.DefinitionStmt;
26     
27     /**
28      * This is used to process statements in object flow analysis. This class in turn uses a expression visitor to process
29      * expressions that occur in a statement.
30      * 
31      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
32      * @version $Revision: 1.10 $
33      * @param <T>  is the type of the token set object.
34      */
35     class StmtSwitch<T extends ITokens<T, Value>>
36     		extends AbstractStmtSwitch<StmtSwitch<T>, OFAFGNode<T>> {
37     
38     	/**
39     	 * Creates a new <code>StmtSwitch</code> instance.
40     	 * 
41     	 * @param m the <code>IMethodVariant</code> which uses this object.
42     	 * @pre m != null
43     	 */
44     	public StmtSwitch(final IMethodVariant<OFAFGNode<T>> m) {
45     		super(m);
46     	}
47     
48     	/**
49     	 * Returns a new instance of this class.
50     	 * 
51     	 * @param o the method variant which uses this object. This is of type <code>MethodVariant</code>.
52     	 * @return the new instance of this class.
53     	 * @pre o != null and o[0].oclIsKindOf(IMethodVariant)
54     	 * @post result != null
55     	 */
56     	@Override public StmtSwitch<T> getClone(final Object... o) {
57     		return new StmtSwitch<T>((IMethodVariant) o[0]);
58     	}
59     
60     	/**
61     	 * AbstractStmtSwitch#processDefinitionStmt(DefinitionStmt)
62     	 */
63     	@Override protected void processDefinitionStmt(final DefinitionStmt stmt) {
64     		lexpr.process(stmt.getLeftOpBox());
65     
66     		final OFAFGNode<T> _left = lexpr.getFlowNode();
67     
68     		rexpr.process(stmt.getRightOpBox());
69     
70     		final OFAFGNode<T> _right = rexpr.getFlowNode();
71     
72     		if (Util.isReferenceType(stmt.getRightOp().getType())) {
73     			_right.addSucc(_left);
74     		}
75     	}
76     }
