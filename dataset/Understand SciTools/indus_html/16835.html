24     import java.io.StringWriter;
25     import java.util.Collection;
26     
27     import org.jibx.runtime.BindingDirectory;
28     import org.jibx.runtime.IBindingFactory;
29     import org.jibx.runtime.IMarshallingContext;
30     import org.jibx.runtime.IUnmarshallingContext;
31     import org.jibx.runtime.JiBXException;
32     import org.slf4j.Logger;
33     import org.slf4j.LoggerFactory;

38     /**
39      * This class provides facility to serialize and deserialize slice criteria.
40      *
41      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
42      * @author $Author: rvprasad $
43      * @version $Revision: 1.8 $ $Date: 2007/02/10 19:06:02 $
44      */
45     public final class SliceCriteriaParser {
46     	/** 
47     	 * The logger used by instances of this class to log messages.
48     	 */
49     	private static final Logger LOGGER = LoggerFactory.getLogger(SliceCriteriaParser.class);
50     
51     	/** 
52     	 * Indentation to be used during serialization.
53     	 */
54     	private static final int INDENT = 4;
55     
56     	///CLOVER:OFF
57     
58     	/**
59     	 * Creates a new SliceCriteriaParser object.
60     	 */
61     	@Empty private SliceCriteriaParser() {
62     		// Does nothing
63     	}
64     
65     	///CLOVER:ON
66     
67     	/**
68     	 * Deserializes the given criteria specification into criteria objects.
69     	 *
70     	 * @param criteriaSpec is the criteria specification in serialized form.
71     	 * @param scene relative on which the deserialization occurs.
72     	 *
73     	 * @return a collection of criteria.
74     	 *
75     	 * @throws JiBXException when either JiBX cannot be setup or error occurs during deserialization.
76     	 *
77     	 * @pre criteriaSpec != null and scene != null
78     	 * @post result != null
79     	 */
80     	public static Collection<ISliceCriterion> deserialize(final String criteriaSpec, final Scene scene)
81     	  throws JiBXException {
82     		final Collection<ISliceCriterion>_result;
83     		IBindingFactory _bindingFactory;
84     		IUnmarshallingContext _unmarshallingContext;
85     
86     		try {
87     			_bindingFactory = BindingDirectory.getFactory(SliceCriteriaSpec.class);
88     			_unmarshallingContext = _bindingFactory.createUnmarshallingContext();
89     		} catch (final JiBXException _e) {
90     			LOGGER.error("Error while setting up JiBX.  Aborting.", _e);
91     			throw _e;
92     		}
93     
94     		try {
95     			final StringReader _reader = new StringReader(criteriaSpec);
96     			final SliceCriteriaSpec _sliceCriteriaSpec =
97     				(SliceCriteriaSpec) _unmarshallingContext.unmarshalDocument(_reader, null);
98     			_result = _sliceCriteriaSpec.getCriteria(scene);
99     		} catch (final JiBXException _e) {
100     			LOGGER.error("Error while deserializing slice criteria.", _e);
101     			throw _e;
102     		}
103     
104     		return _result;
105     	}
106     
107     	/**
108     	 * Serializes the given criteria into criteria specification.
109     	 *
110     	 * @param criteria is the criteria to be serialized.
111     	 *
112     	 * @return the criteria specification.
113     	 *
114     	 * @throws JiBXException when either JiBX cannot be setup or error occurs during serialization.
115     	 *
116     	 * @pre criteria != null
117     	 * @pre criteria.oclIsKindOf(ISliceCriterion)
118     	 * @post result != null
119     	 */
120     	public static String serialize(final Collection<ISliceCriterion> criteria)
121     	  throws JiBXException {
122     		final String _result;
123     		IBindingFactory _bindingFactory;
124     		IMarshallingContext _marshallingContext;
125     
126     		try {
127     			_bindingFactory = BindingDirectory.getFactory(SliceCriteriaSpec.class);
128     			_marshallingContext = _bindingFactory.createMarshallingContext();
129     			_marshallingContext.setIndent(INDENT);
130     		} catch (final JiBXException _e) {
131     			LOGGER.error("Error while setting up JiBX.", _e);
132     			throw _e;
133     		}
134     
135     		try {
136     			final StringWriter _writer = new StringWriter();
137     			_marshallingContext.marshalDocument(SliceCriteriaSpec.getSliceCriteriaSpec(criteria), "UTF-8", Boolean.TRUE,
138     				_writer);
139     			_writer.flush();
140     			_result = _writer.toString();
141     		} catch (final JiBXException _e) {
142     			LOGGER.error("Error while marshalling Slicer configuration.", _e);
143     			throw _e;
144     		}
145     		return _result;
146     	}
147     }
