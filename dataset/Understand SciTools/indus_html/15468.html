23     import java.util.Collection;
24     import java.util.List;

27     /**
28      * This class tests <code>AbstractWorkBag</code> class.
29      *
30      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
31      * @author $Author: rvprasad $
32      * @version $Revision: 1.8 $ $Date: 2007/02/10 19:08:35 $
33      */
34     public abstract class AbstractWorkBagTest
35       extends IndusTestCase {
36     	/** 
37     	 * The workbag.
38     	 */
39     	protected IWorkBag wb;
40     
41     	/**
42     	 * Tests <code>addAllWork</code>.
43     	 */
44     	public void testAddAllWork() {
45     		final List _workPieces = getWorkPieces();
46     		wb.addAllWork(_workPieces);
47     
48     		while (wb.hasWork()) {
49     			assertTrue(_workPieces.contains(wb.getWork()));
50     		}
51     	}
52     
53     	/**
54     	 * Tests <code>addAllWorkNoDuplicates</code>.
55     	 */
56     	public void testAddAllWorkNoDuplicates() {
57     		final List _workPieces = getWorkPieces();
58     		wb.addAllWork(_workPieces);
59     
60     		final int _two = 2;
61     		final int _five = 5;
62     
63     		final List _subList = new ArrayList(_workPieces.subList(_two, _five));
64     		_subList.add(new Object());
65     
66     		final Collection _temp = wb.addAllWorkNoDuplicates(_subList);
67     		assertTrue(_subList.containsAll(_temp));
68     		assertTrue(!_temp.containsAll(_subList));
69     	}
70     
71     	/**
72     	 * Tests <code>addWorkNoDuplicate</code>.
73     	 */
74     	public void testAddWorkNoDuplicates() {
75     		final List _workPieces = getWorkPieces();
76     		wb.addAllWork(_workPieces);
77     		assertFalse(wb.addWorkNoDuplicates(_workPieces.get(0)));
78     		assertTrue(wb.addWorkNoDuplicates(new Object()));
79     	}
80     
81     	/**
82     	 * Tests <code>clear</code>.
83     	 */
84     	public final void testClear() {
85     		wb.clear();
86     		assertFalse(wb.hasWork());
87     
88     		try {
89     			wb.getWork();
90     			///CLOVER:OFF            
91     			fail("This is incorrect behavior.");
92     			///CLOVER:ON
93     		} catch (IllegalStateException _e) {
94     			_e.printStackTrace();
95     		}
96     	}
97     
98     	/**
99     	 * Tests <code>getWork</code> and <code>hasWork</code>.
100     	 */
101     	public void testGetWorkandHasWork() {
102     		final Object _object = new Object();
103     		wb.addWork(_object);
104     		assertTrue(wb.hasWork());
105     
106     		final Object _temp = wb.getWork();
107     		assertTrue(_temp != null);
108     		assertTrue(_temp == _object);
109     	}
110     
111     	/**
112     	 * Provides new collection of work pieces.
113     	 *
114     	 * @return a collection of work pieces.
115     	 */
116     	protected List getWorkPieces() {
117     		final List _result = new ArrayList();
118     		final int _ten = 10;
119     
120     		for (int _i = 1; _i <= _ten; _i++) {
121     			_result.add(String.valueOf(_i));
122     		}
123     		return _result;
124     	}
125     }
