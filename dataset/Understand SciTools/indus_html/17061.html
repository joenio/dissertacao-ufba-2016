23     /**
24      * This interface exposes independence information in the context of concurrency.
25      *
26      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
27      * @author $Author: rvprasad $
28      * @version $Revision: 1.5 $
29      */
30     public interface IConcurrentIndependenceInfo
31       extends IIdentification {
32     	/**
33     	 * This is the unique identifier that can be used to identify an instance of this class.
34     	 */
35     	Comparable<String> ID = "Concurrent Independence Analysis";
36     
37     	/**
38     	 * Checks if the statement is independent.
39     	 *
40     	 * @param stmt to be checked.
41     	 *
42     	 * @return <code>true</code> if <code>stmt</code> is independent; <code>false</code>, otherwise.
43     	 *
44     	 * @pre stmt != null
45     	 */
46     	boolean isIndependent(final Stmt stmt);
47     }
