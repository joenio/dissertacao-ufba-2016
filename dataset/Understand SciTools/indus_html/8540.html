17     package edu.ksu.cis.indus.staticanalyses.dependency;
18     
19     import edu.ksu.cis.indus.interfaces.IIdentification;
20     import edu.ksu.cis.indus.interfaces.IStatus;
21     import edu.ksu.cis.indus.staticanalyses.interfaces.IAnalysis;
22     
23     import java.util.Collection;
24     
25     /**
26      * The interface to dependency analysis information.
27      * <p>
28      * There is room for confusion due to <code>BI_DIRECTIONAL</code> behavior of an analysis. To avoid this, it is required
29      * that in a bi-directional analysis implement <code>getDependees</code> and <code>getDependents</code> to provide
30      * dependee and dependent info, respectively, in backward direction while <code>getDependents</code> and
31      * <code>getDependees</code> provide dependee and dependent info, respectively, in forward direction.
32      * </p>
33      * 
34      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
35      * @author $Author: rvprasad $
36      * @version $Revision: 1.17 $
37      * @param <E1> is the type of dependee object in the context of dependee-to-dependent info maintenance.
38      * @param <C1> is the type of context object in the context of dependee-to-dependent info maintenance.
39      * @param <T1> is the type of dependent object in the context of dependee-to-dependent info maintenance.
40      * @param <T2> is the type of dependent object in the context of dependent-to-dependee info maintenance.
41      * @param <C2> is the type of context object in the context of dependent-to-dependee info maintenance.
42      * @param <E2> is the type of dependee object in the context of dependent-to-dependee info maintenance.
43      */
44     public interface IDependencyAnalysis<T1, C1, E1, E2, C2, T2>
45     		extends IIdentification, IStatus, IAnalysis {
46     
47     	/**
48     	 * The enumeration of sorts of dependences.
49     	 * 
50     	 * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
51     	 * @author $Author: rvprasad $
52     	 * @version $Revision: 1.17 $
53     	 */
54     	enum DependenceSort {
55     		/**
56     		 * This identifies control dependency analysis.
57     		 */
58     		CONTROL_DA,
59     
60     		/**
61     		 * This identifies divergence dependency analysis.
62     		 */
63     		DIVERGENCE_DA,
64     
65     		/**
66     		 * This identifies identifier based data dependency analysis.
67     		 */
68     		IDENTIFIER_BASED_DATA_DA,
69     
70     		/**
71     		 * This identifies inteference dependency analysis.
72     		 */
73     		INTERFERENCE_DA,
74     
75     		/**
76     		 * This identifies ready dependency analysis.
77     		 */
78     		READY_DA,
79     
80     		/**
81     		 * This identifies class-level data dependency analysis.
82     		 */
83     		REFERENCE_BASED_DATA_DA,
84     
85     		/**
86     		 * This identifies synchronization dependency analysis.
87     		 */
88     		SYNCHRONIZATION_DA
89     	}
90     
91     	/**
92     	 * The enumeration of the dependence direction.
93     	 * 
94     	 * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
95     	 * @author $Author: rvprasad $
96     	 * @version $Revision: 1.17 $
97     	 */
98     	enum Direction {
99     		/**
100     		 * This identifies backward directional analaysis.
101     		 */
102     		BACKWARD_DIRECTION,
103     		/**
104     		 * This identifies bi-directional dependence analysis. This means that if such an analysis is designed for forward
105     		 * direction then the backward direction information can be obtained by switching <code>getDependees()</code> and
106     		 * <code>getDependents()</code>.
107     		 */
108     		BI_DIRECTIONAL,
109     
110     		/**
111     		 * This identifies forward directional analaysis.
112     		 */
113     		FORWARD_DIRECTION
114     	};
115     
116     	/**
117     	 * Return the entities on which the <code>dependent</code> depends on in the given <code>context</code>.
118     	 * 
119     	 * @param dependent of interest.
120     	 * @param context in which the dependency information is requested.
121     	 * @return a collection of objects.
122     	 * @pre dependent != null
123     	 * @post result != null
124     	 */
125     	Collection<E1> getDependees(final T1 dependent, final C1 context);
126     
127     	/**
128     	 * Returns the entities which depend on the <code>dependee</code> in the given <code>context</code>.
129     	 * 
130     	 * @param dependee of interest.
131     	 * @param context in which the dependency information is requested.
132     	 * @return a collection of objects. The subclasses will further specify the types of these entities.
133     	 * @pre dependee != null
134     	 * @post result != null
135     	 */
136     	Collection<T2> getDependents(final E2 dependee, final C2 context);
137     
138     	/**
139     	 * Returns the direction of the analysis. In analysis that are <code>BACKWARD_DIRECTION</code> oriented, the dependent
140     	 * is related to the dependee via the dependence against the flow of control. In analysis that are
141     	 * <code>FORWARD_DIRECTION</code> oriented, the dependent is related to the dependee via the dependence along the flow
142     	 * of control. Analysis that are direction independent should return <code>BI_DIRECTIONAL</code>.
143     	 * 
144     	 * @return the direction of the implementation.
145     	 * @post result != null
146     	 */
147     	Direction getDirection();
148     
149     	/**
150     	 * Retrieves dependence analysis that provides indirect dependence information corresponding to this dependence analysis.
151     	 * 
152     	 * @return a dependence analysis.
153     	 * @post result != null
154     	 */
155     	IDependencyAnalysis<T1, C1, E1, E2, C2, T2> getIndirectVersionOfDependence();
156     
157     	/**
158     	 * @see edu.ksu.cis.indus.interfaces.IStatus#isStable()
159     	 */
160     	boolean isStable();
161     
162     	/**
163     	 * Resets all internal data structures. General protocol is that data acquired via setup is not reset or forgotten.
164     	 */
165     	void reset();
166     
167     	/**
168     	 * @see edu.ksu.cis.indus.interfaces.IIdentification#getIds()
169     	 */
170     	Collection<IDependencyAnalysis.DependenceSort> getIds();
171     }
