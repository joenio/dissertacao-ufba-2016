239     	 * Unifies the given object with itself. This is required when the alias set is occurs in the context of a site which is
240     	 * executed multiple times, in particular, reachable from a call-site which may be executed multiple times.
241     	 * 
242     	 * @param as the alias set to be unified with itself.
243     	 * @pre as != null
244     	 */
245     	static void selfUnify(final AliasSet as) {
246     		final Collection<AliasSet> _processed = new HashSet<AliasSet>();
247     		final IWorkBag<AliasSet> _wb = new HistoryAwareLIFOWorkBag<AliasSet>(_processed);
248     		_wb.addWork(as.find());
249     
250     		while (_wb.hasWork()) {
251     			final AliasSet _repr = _wb.getWork();
252     			_repr.unifyThreadEscapeInfo(_repr);
253     			_wb.addAllWorkNoDuplicates(CollectionUtils.collect(_repr.fieldMap.values(), REPRESENTATIVE_ALIAS_SET_RETRIEVER));
254     		}
255     	}
