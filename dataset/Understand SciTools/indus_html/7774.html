17     package edu.ksu.cis.indus.common.collections;
18     
19     import edu.ksu.cis.indus.annotations.Functional;
20     import edu.ksu.cis.indus.annotations.Immutable;
21     import edu.ksu.cis.indus.annotations.NonNull;
22     
23     /**
24      * This class can be used to check if a given class object is the super type of a sub type.
25      * 
26      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
27      * @author $Author: rvprasad $
28      * @version $Revision: 1.5 $ $Date: 2007/10/09 15:50:20 $
29      * @param <T> is the subtype.
30      */
31     public class ReflectionBasedSupertypePredicate<T>
32     		implements IPredicate<Class<? extends T>> {
33     
34     	/**
35     	 * The sub type against which the check will occur.
36     	 */
37     	private Class<? extends T> subtype;
38     
39     	/**
40     	 * Creates an instance of this class.
41     	 */
42     	public ReflectionBasedSupertypePredicate() {
43     		super();
44     	}
45     
46     	/**
47     	 * Checks if the given object is a super type of the subtype set on this object.
48     	 * 
49     	 * @param object to be checked.
50     	 * @return <code>true</code> if <code>object</code> is a class object and it is a super type of the subtype set on
51     	 *         this object; <code>false</code>, otherwise.
52     	 */
53     	@Functional public boolean evaluate(@NonNull final Class object) {
54     		return object.isAssignableFrom(subtype);
55     	}
56     
57     	/**
58     	 * Sets the subtype for the check to be performed by this object.
59     	 * 
60     	 * @param clazz is the subtype.
61     	 */
62     	public void setsubType(@NonNull @Immutable final Class<? extends T> clazz) {
63     		subtype = clazz;
64     	}
65     }
