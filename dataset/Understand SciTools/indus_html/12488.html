17     package edu.ksu.cis.indus.common.graph;
18     
19     import edu.ksu.cis.indus.annotations.Immutable;
20     import edu.ksu.cis.indus.annotations.NonNull;
21     import edu.ksu.cis.indus.common.collections.IPredicate;
22     
23     /**
24      * This implementation checks if there is a path from the node representing the given object to the node representing an
25      * object in the graph in the direction specified at initialization time. <code>evaluate</code> method will return
26      * <code>true</code> only if both objects are represented in the node.
27      * 
28      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
29      * @author $Author: rvprasad $
30      * @version $Revision: 1.8 $ $Date: 2007/10/09 15:59:13 $
31      * @param <N> the type of nodes in the graph processed by this predicate.
32      * @param <O> the type of object being represented by the object graph.
33      */
34     public final class GraphReachabilityPredicate<N extends IObjectNode<N, O>, O>
35     		implements IPredicate<O> {
36     
37     	/**
38     	 * The node to which the path leads to.
39     	 */
40     	private final N destNode;
41     
42     	/**
43     	 * The direction of the path.
44     	 */
45     	private final boolean forward;
46     
47     	/**
48     	 * The graph in which reachability is calculated.
49     	 */
50     	private final IObjectDirectedGraph<N, O> graph;
51     
52     	/**
53     	 * Creates a new GraphReachabilityPredicate object.
54     	 * 
55     	 * @param theDestNode is a node in <code>theGraph</code>.
56     	 * @param forwardDir <code>true</code> indicates forward direction (following the edges); <code>false</code> indicates
57     	 *            backward direction (following the edges in the reverse direction).
58     	 * @param theGraph of interest.
59     	 */
60     	public GraphReachabilityPredicate(@NonNull @Immutable final N theDestNode, final boolean forwardDir,
61     			@NonNull @Immutable final IObjectDirectedGraph<N, O> theGraph) {
62     		destNode = theDestNode;
63     		forward = forwardDir;
64     		graph = theGraph;
65     	}
66     
67     	/**
68     	 * {@inheritDoc}
69     	 */
70     	public  boolean evaluate(@Immutable final O srcObject) {
71     		final N _srcNode = graph.queryNode(srcObject);
72     		return destNode != null && _srcNode != null && graph.isReachable(_srcNode, destNode, forward);
73     	}
74     }
