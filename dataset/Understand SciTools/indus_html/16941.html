47     /**
48      * This class provides data dependency information based on identifiers. Local variables in a method enable such dependence.
49      * Given a def site, the use site is tracked based on the id being defined and used. Hence, information about field/array
50      * access via primaries which are local variables is inaccurate in such a setting, hence, it is not provided by this class.
51      * Please refer to {@link ReferenceBasedDataDA ReferenceBasedDataDA} for such information.
52      * <p>
53      * This implementation is based on <code>soot.toolkits.scalar.SimpleLocalDefs</code> and
54      * <code>soot.toolkits.scalar.SimpleLocalUses</code> classes .
55      * </p>
56      * 
57      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
58      * @author $Author: rvprasad $
59      * @version $Revision: 1.54 $
60      */

71     	/*
72     	 * The dependent information is stored as follows: For each method, a list of length equal to the number of statements in
73     	 * the methods is maintained. In case of dependent information, at each location corresponding to the statement a set of
74     	 * dependent statements is maintained in the list. In case of dependee information, at each location corresponding to the
75     	 * statement a map is maintained in the list. The map maps a value box in the statement to a collection of dependee
76     	 * statements. The rational for the way the information is maintained is only one local can be defined in a statement.
77     	 * Also, if the definition of a local reaches a statement, then all occurrences of that local at that statement must be
78     	 * dependent on the same reaching def.
79     	 */

98     	/**
99     	 * Returns the statements on which the locals in <code>stmt</code> depends in the given <code>method</code>.
100     	 * 
101     	 * @param stmt in which the locals occur.
102     	 * @param method in which <code>programPoint</code> occurs.
103     	 * @param da is the analysis from which to retrieve dependence information.
104     	 * @return a collection of statements on which <code>programPoint</code> depends.
105     	 */
106     	final static Collection<DefinitionStmt> getDependeesHelper(final Stmt stmt, final SootMethod method,
107     			final IDependencyAnalysis<Pair<Local, Stmt>, SootMethod, DefinitionStmt, ?, ?, ?> da) {
108     		Collection<DefinitionStmt> _result = Collections.emptySet();
109     
110     		for (final Iterator<ValueBox> _i = stmt.getUseBoxes().iterator(); _i.hasNext();) {
111     			final ValueBox _vb = _i.next();
112     			final Value _v = _vb.getValue();
113     
114     			if (_v instanceof Local) {
115     				final Collection<DefinitionStmt> _c = da.getDependees(new Pair<Local, Stmt>((Local) _v, stmt), method);
116     
117     				if (_c != null) {
118     					_result = Collections.unmodifiableCollection(_c);
119     				} else {
120     					if (LOGGER.isWarnEnabled()) {
121     						LOGGER.warn("No dependence information available for " + stmt + " in " + method);
122     					}
123     				}
124     			}
125     		}
126     
127     		return _result;
128     	}

130     	/**
131     	 * Calculates the dependency information for locals in the methods provided during initialization.
132     	 * 
133     	 * @see edu.ksu.cis.indus.staticanalyses.dependency.AbstractDependencyAnalysis#analyze()
134     	 */
135     	@Override public void analyze() {
136     		unstable();
137     
138     		if (LOGGER.isInfoEnabled()) {
139     			LOGGER.info("BEGIN: Identifier Based Data Dependence processing");
140     		}
141     
142     		for (final Iterator<SootMethod> _i = callgraph.getReachableMethods().iterator(); _i.hasNext();) {
143     			final SootMethod _currMethod = _i.next();
144     			final UnitGraph _unitGraph = getUnitGraph(_currMethod);
145     
146     			if (_unitGraph == null) {
147     				if (LOGGER.isWarnEnabled()) {
148     					LOGGER.warn("Method " + _currMethod.getSignature() + " does not have a unit graph.");
149     				}
150     				continue;
151     			}
152     
153     			if (LOGGER.isDebugEnabled()) {
154     				LOGGER.debug("Processing " + _currMethod.getSignature());
155     			}
156     
157     			calculateDAForMethod(_currMethod, _unitGraph);
158     		}
159     		stable();
160     
161     		if (LOGGER.isDebugEnabled()) {
162     			LOGGER.debug("analyze() - " + toString());
163     		}
164     
165     		if (LOGGER.isInfoEnabled()) {
166     			LOGGER.info("END:  Identifier Based Data Dependence processing");
167     		}
168     	}

170     	/**
171     	 * Returns the statements on which <code>o</code>, depends in the given <code>method</code>.
172     	 * 
173     	 * @param programPoint is the program point at which a local occurs in the statement.
174     	 * @param method in which <code>programPoint</code> occurs.
175     	 * @return a collection of statements on which <code>programPoint</code> depends.
176     	 * @pre programPoint.oclTypeOf(Pair).getFirst() != null and programPoint.oclTypeOf(Pair).getSecond() != null
177     	 */
178     	public Collection<DefinitionStmt> getDependees(final Pair<Local, Stmt> programPoint, final SootMethod method) {
179     		Collection<DefinitionStmt> _result = Collections.emptySet();
180     
181     		final Stmt _stmt = programPoint.getSecond();
182     		final Local _local = programPoint.getFirst();
183     		final List<Map<Local, Collection<DefinitionStmt>>> _dependees = dependent2dependee.get(method);
184     		if (_dependees != null) {
185     			final Map<Local, Collection<DefinitionStmt>> _local2defs = _dependees.get(getStmtList(method).indexOf(_stmt));
186     			final Collection<DefinitionStmt> _c = _local2defs.get(_local);
187     
188     			if (_c != null) {
189     				_result = Collections.unmodifiableCollection(_c);
190     			} else {
191     				if (LOGGER.isWarnEnabled()) {
192     					LOGGER.warn("No dependence information available for " + programPoint + " in " + method);
193     				}
194     			}
195     		}
196     		return _result;
197     	}

210     	/**
211     	 * Returns the statement and the program point in it which depends on statement provided via <code>programPoint</code>
212     	 * occurring in the given <code>method</code>.
213     	 * 
214     	 * @param programPoint is the definition statement or a pair containing the definition statement as the first element.
215     	 *            Although, only one variable can be defined in a Jimple statement, we allow for a pair in this query to make
216     	 *            <code>getDependees</code> and <code>getDependents</code> symmetrical for ease of usage.
217     	 * @param method is the method in which the statement provided by <code>programPoint</code> occurs.
218     	 * @return a collection of statement and program points in them which depend on the definition at
219     	 *         <code>programPoint</code>.
220     	 */
221     	public Collection<Pair<Local, Stmt>> getDependents(final DefinitionStmt programPoint, final SootMethod method) {
222     		Collection<Pair<Local, Stmt>> _result = Collections.emptySet();
223     
224     		final List<Collection<Pair<Local, Stmt>>> _dependents = dependee2dependent.get(method);
225     
226     		if (_dependents != null) {
227     			final Collection<Pair<Local, Stmt>> _temp = _dependents.get(getStmtList(method).indexOf(programPoint));
228     			if (_temp != null) {
229     				_result = Collections.unmodifiableCollection(_temp);
230     			}
231     		}
232     		return _result;
233     	}

244     	/**
245     	 * Returns a stringized representation of this analysis. The representation includes the results of the analysis.
246     	 * 
247     	 * @return a stringized representation of this object.
248     	 */
249     	@Override public String toString() {
250     		final StringBuffer _result = new StringBuffer("Statistics for Identifier-based Data dependence as calculated by "
251     				+ this.getClass().getName() + "\n");
252     		int _localEdgeCount = 0;
253     		int _edgeCount = 0;
254     
255     		final StringBuffer _temp = new StringBuffer();
256     
257     		for (final Iterator<Map.Entry<SootMethod, List<Collection<Pair<Local, Stmt>>>>> _i = dependee2dependent.entrySet()
258     				.iterator(); _i.hasNext();) {
259     			final Map.Entry<SootMethod, List<Collection<Pair<Local, Stmt>>>> _entry = _i.next();
260     			_localEdgeCount = 0;
261     
262     			final List<Stmt> _stmts = getStmtList(_entry.getKey());
263     			int _count = 0;
264     
265     			for (final Iterator<Collection<Pair<Local, Stmt>>> _j = _entry.getValue().iterator(); _j.hasNext();) {
266     				final Collection<Pair<Local, Stmt>> _c = _j.next();
267     				final Stmt _stmt = _stmts.get(_count++);
268     
269     				for (final Iterator<?> _k = _c.iterator(); _k.hasNext();) {
270     					_temp.append("\t\t" + _stmt + " <-- " + _k.next() + "\n");
271     				}
272     				_localEdgeCount += _c.size();
273     			}
274     			_result.append("\tFor " + _entry.getKey() + " there are " + _localEdgeCount
275     					+ " Identifier-based Data dependence edges.\n");
276     			_result.append(_temp);
277     			_temp.delete(0, _temp.length());
278     			_edgeCount += _localEdgeCount;
279     		}
280     		_result.append("A total of " + _edgeCount + " Identifier-based Data dependence edges exist.");
281     		return _result.toString();
282     	}

309     	/**
310     	 * Calculates dependence info for the given method.
311     	 * 
312     	 * @param method for which to calculate dependence.
313     	 * @param unitGraph of <code>method</code>.
314     	 * @pre method != null and unitGraph != null
315     	 */
316     	private void calculateDAForMethod(final SootMethod method, final UnitGraph unitGraph) {
317     		final SimpleLocalDefs _defs = new SimpleLocalDefs(unitGraph);
318     		final SimpleLocalUses _uses = new SimpleLocalUses(unitGraph, _defs);
319     		final Collection<Stmt> _t = getStmtList(method);
320     		final List<Map<Local, Collection<DefinitionStmt>>> _dependees = new ArrayList<Map<Local, Collection<DefinitionStmt>>>(
321     				_t.size());
322     		final List<Collection<Pair<Local, Stmt>>> _dependents = new ArrayList<Collection<Pair<Local, Stmt>>>(_t.size());
323     
324     		for (final Iterator<Stmt> _j = _t.iterator(); _j.hasNext();) {
325     			final Stmt _currStmt = _j.next();
326     			Collection<Pair<Local, Stmt>> _currUses = Collections.emptySet();
327     
328     			if (_currStmt instanceof DefinitionStmt) {
329     				final Collection<UnitValueBoxPair> _temp = _uses.getUsesOf(_currStmt);
330     
331     				if (_temp.size() != 0) {
332     					_currUses = new ArrayList<Pair<Local, Stmt>>();
333     
334     					for (final Iterator<UnitValueBoxPair> _k = _temp.iterator(); _k.hasNext();) {
335     						final UnitValueBoxPair _p = _k.next();
336     						_currUses.add(new Pair<Local, Stmt>((Local) _p.getValueBox().getValue(), (Stmt) _p.getUnit()));
337     					}
338     				}
339     			}
340     			_dependents.add(_currUses);
341     
342     			Map<Local, Collection<DefinitionStmt>> _currDefs = Collections.emptyMap();
343     
344     			if (_currStmt.getUseBoxes().size() > 0) {
345     				_currDefs = new HashMap<Local, Collection<DefinitionStmt>>();
346     
347     				for (final Iterator<ValueBox> _k = _currStmt.getUseBoxes().iterator(); _k.hasNext();) {
348     					final ValueBox _currValueBox = _k.next();
349     					final Value _value = _currValueBox.getValue();
350     
351     					if (_value instanceof Local && !_currDefs.containsKey(_value)) {
352     						final Local _local = (Local) _value;
353     						_currDefs.put(_local, _defs.getDefsOfAt(_local, _currStmt));
354     					}
355     				}
356     			}
357     			_dependees.add(_currDefs);
358     		}
359     		dependee2dependent.put(method, _dependents);
360     		dependent2dependee.put(method, _dependees);
361     	}
