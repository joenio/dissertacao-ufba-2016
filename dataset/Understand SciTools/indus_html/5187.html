56     /**
57      * This implementation of <code>CallGraphInfo.ICallInfo</code> generates call info for a system based on the information
58      * available from object flow information.
59      * 
60      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
61      * @author $Author: rvprasad $
62      * @version $Revision: 1.12 $ $Date: 2007/02/10 19:07:01 $
63      */
64     public class OFABasedCallInfoCollector
65     		extends AbstractValueAnalyzerBasedProcessor<Value>
66     		implements ICallInfoCollector {
67     
68     	/**
69     	 * The logger used by instances of this class to log messages.
70     	 */
71     	static final Logger LOGGER = LoggerFactory.getLogger(OFABasedCallInfoCollector.class);
72     
73     	/**
74     	 * The FA instance which implements object flow analysis. This instance is used to calculate call graphCache information.
75     	 * 
76     	 * @invariant analyzer.oclIsKindOf(OFAnalyzer)
77     	 */
78     	private IValueAnalyzer<Value> analyzer;
79     
80     	/**
81     	 * This holds call information.
82     	 */
83     	private final CallInfo callInfoHolder = new CallInfo();
84     
85     	/**
86     	 * @see edu.ksu.cis.indus.processing.IProcessor#callback(soot.SootMethod)
87     	 */
88     	@Override public void callback(final SootMethod method) {
89     		// all method marked by the object flow analyses are reachable.
90     		callInfoHolder.addReachable(method);
91     	}
92     
93     	/**
94     	 * Called by the post process controller when it walks a jimple value AST node.
95     	 * 
96     	 * @param vBox is the AST node to be processed.
97     	 * @param context in which value should be processed.
98     	 * @pre context != null and vBox != null
99     	 * @see edu.ksu.cis.indus.staticanalyses.interfaces.IValueAnalyzerBasedProcessor#callback(ValueBox,Context)
100     	 */
101     	@Override public void callback(final ValueBox vBox, final Context context) {
102     		if (LOGGER.isDebugEnabled()) {
103     			LOGGER.debug("callback(ValueBox vBox = " + vBox + ", Context context = " + context + ") - BEGIN");
104     		}
105     
106     		final Stmt _stmt = context.getStmt();
107     		final SootMethod _caller = context.getCurrentMethod();
108     		final Value _value = vBox.getValue();
109     		final InvokeExpr _invokeExpr = (InvokeExpr) _value;
110     		final SootMethod _callee = _invokeExpr.getMethod();
111     
112     		if (_value instanceof StaticInvokeExpr || _value instanceof SpecialInvokeExpr) {
113     			final Collection<CallTriple> _callees = MapUtils.getCollectionFromMap(callInfoHolder.caller2callees, _caller);
114     			final CallTriple _triple1 = new CallTriple(_callee, _stmt, _invokeExpr);
115     			_callees.add(_triple1);
116     
117     			final Collection<CallTriple> _callers = MapUtils.getCollectionFromMap(callInfoHolder.callee2callers, _callee);
118     			final CallTriple _triple2 = new CallTriple(_caller, _stmt, _invokeExpr);
119     			_callers.add(_triple2);
120     		} else {
121     			callBackOnInstanceInvokeExpr(context, (InstanceInvokeExpr) _value);
122     		}
123     
124     		if (LOGGER.isDebugEnabled()) {
125     			LOGGER.debug("callback() - END");
126     		}
127     	}
128     
129     	/**
130     	 * This calculates information such as heads, tails, and such.
131     	 * 
132     	 * @see edu.ksu.cis.indus.staticanalyses.interfaces.IValueAnalyzerBasedProcessor#consolidate()
133     	 */
134     	@Override public void consolidate() {
135     		if (LOGGER.isInfoEnabled()) {
136     			LOGGER.info("BEGIN: call graph consolidation");
137     		}
138     
139     		callInfoHolder.fixupMethodsHavingZeroCallersAndCallees();
140     
141     		if (LOGGER.isDebugEnabled()) {
142     			LOGGER.debug("Call Graph Info - " + callInfoHolder.toString());
143     		}
144     		
145     		if (LOGGER.isInfoEnabled()) {
146     			LOGGER.info("END: call graph consolidation");
147     		}
148     		
149     		stable();
150     	}
151     
152     	/**
153     	 * @see ICallInfoCollector#getCallInfo()
154     	 */
155     	public CallGraphInfo.ICallInfo getCallInfo() {
156     		return callInfoHolder;
157     	}
158     
159     	/**
160     	 * @see edu.ksu.cis.indus.processing.IProcessor#hookup(ProcessingController)
161     	 */
162     	public void hookup(final ProcessingController ppc) {
163     		unstable();
164     		ppc.register(VirtualInvokeExpr.class, this);
165     		ppc.register(InterfaceInvokeExpr.class, this);
166     		ppc.register(StaticInvokeExpr.class, this);
167     		ppc.register(SpecialInvokeExpr.class, this);
168     		ppc.register(this);
169     	}
170     
171     	/**
172     	 * @see edu.ksu.cis.indus.processing.AbstractProcessor#processingBegins()
173     	 */
174     	@Override public void processingBegins() {
175     		unstable();
176     	}
177     
178     	/**
179     	 * @see edu.ksu.cis.indus.processing.AbstractProcessor#reset()
180     	 */
181     	@Override public void reset() {
182     		callInfoHolder.reset();
183     	}
184     
185     	/**
186     	 * Sets the analyzer to be used to calculate call graph information upon call back.
187     	 * 
188     	 * @param objFlowAnalyzer that provides the information to create the call graph.
189     	 * @pre objFlowAnalyzer != null and objFlowAnalyzer.oclIsKindOf(OFAnalyzer)
190     	 * @see edu.ksu.cis.indus.staticanalyses.interfaces.IValueAnalyzerBasedProcessor#setAnalyzer(IValueAnalyzer)
191     	 */
192     	@Override public void setAnalyzer(final IValueAnalyzer<Value> objFlowAnalyzer) {
193     		analyzer = objFlowAnalyzer;
194     	}
195     
196     	/**
197     	 * @see edu.ksu.cis.indus.processing.IProcessor#unhook(ProcessingController)
198     	 */
199     	public void unhook(final ProcessingController ppc) {
200     		ppc.unregister(VirtualInvokeExpr.class, this);
201     		ppc.unregister(InterfaceInvokeExpr.class, this);
202     		ppc.unregister(StaticInvokeExpr.class, this);
203     		ppc.unregister(SpecialInvokeExpr.class, this);
204     		ppc.unregister(this);
205     		stable();
206     	}
207     
208     	/**
209     	 * Called as a result of callback durign processing the AST when instance invoke expression is encountered.
210     	 * 
211     	 * @param context in which expression should be processed.
212     	 * @param expr is the expression.
213     	 * @pre context != null and stmt != null and caller != null and expr != null
214     	 */
215     	private void callBackOnInstanceInvokeExpr(final Context context, final InstanceInvokeExpr expr) {
216     		if (LOGGER.isDebugEnabled()) {
217     			LOGGER.debug("callBackOnInstanceInvokeExpr(Context context = " + context + ", InstanceInvokeExpr expr = " + expr
218     					+ ") - BEGIN");
219     		}
220     
221     		final Stmt _stmt = context.getStmt();
222     		final SootMethod _caller = context.getCurrentMethod();
223     		final SootMethod _calleeMethod = expr.getMethod();
224     		context.setProgramPoint(expr.getBaseBox());
225     
226     		final Collection<?> _values = analyzer.getValues(expr.getBase(), context);
227     
228     		if (!_values.isEmpty()) {
229     			final Map<SootMethod, Collection<CallTriple>> _callee2callers = callInfoHolder.callee2callers;
230     			final Map<SootMethod, Collection<CallTriple>> _caller2callees = callInfoHolder.caller2callees;
231     			final Collection<CallTriple> _callees = MapUtils.getCollectionFromMap(_caller2callees, _caller);
232     			final CallTriple _ctrp = new CallTriple(_caller, _stmt, expr);
233     
234     			for (final Iterator<?> _i = _values.iterator(); _i.hasNext();) {
235     				final Object _t = _i.next();
236     				SootClass _accessClass = null;
237     
238     				if (_t instanceof NewExpr) {
239     					final NewExpr _newExpr = (NewExpr) _t;
240     					_accessClass = analyzer.getEnvironment().getClass(_newExpr.getBaseType().getClassName());
241     				} else if (_t instanceof StringConstant) {
242     					_accessClass = analyzer.getEnvironment().getClass("java.lang.String");
243     				} else if (_t instanceof NewArrayExpr || _t instanceof NewMultiArrayExpr) {
244     					_accessClass = analyzer.getEnvironment().getClass("java.lang.Object");
245     				} else {
246     					continue;
247     				}
248     
249     				final String _methodName = _calleeMethod.getName();
250     				final List<Type> _parameterTypes = _calleeMethod.getParameterTypes();
251     				final Type _returnType = _calleeMethod.getReturnType();
252     				final SootMethod _callee = Util.findMethodImplementation(_accessClass, _methodName, _parameterTypes,
253     						_returnType);
254     				final CallTriple _triple = new CallTriple(_callee, _stmt, expr);
255     				_callees.add(_triple);
256     
257     				final Collection<CallTriple> _callers = MapUtils.getCollectionFromMap(_callee2callers, _callee);
258     				_callers.add(_ctrp);
259     			}
260     		}
261     
262     		if (LOGGER.isDebugEnabled()) {
263     			LOGGER.debug("callBackOnInstanceInvokeExpr() - END");
264     		}
265     	}
266     }
