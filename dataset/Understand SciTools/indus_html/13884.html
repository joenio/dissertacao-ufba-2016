17     package edu.ksu.cis.indus.slicer;
18     
19     import edu.ksu.cis.indus.annotations.Empty;
20     import soot.ValueBox;
21     import soot.jimple.Stmt;
22     
23     /**
24      * This is a helper class to generate criteria specicfication. This is intended for internal use.
25      * 
26      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
27      * @author $Author: rvprasad $
28      * @version $Revision: 1.10 $ $Date: 2007/02/10 19:05:58 $
29      */
30     public final class CriteriaSpecHelper {
31     
32     	// /CLOVER:OFF
33     
34     	/**
35     	 * Creates a new CriteriaSpecHelper object.
36     	 */
37     	@Empty private CriteriaSpecHelper() {
38     		// does nothing
39     	}
40     
41     	// /CLOVER:ON
42     
43     	/**
44     	 * Retrieves the criterion expression.
45     	 * 
46     	 * @param criterion of interest.
47     	 * @return the criterion expression, if available; <code>null</code>, otherwise.
48     	 * @pre criterion != null
49     	 */
50     	public static ValueBox getOccurringExpr(final ISliceCriterion criterion) {
51     		final ValueBox _result;
52     
53     		if (criterion instanceof ExprLevelSliceCriterion) {
54     			_result = (ValueBox) ((ExprLevelSliceCriterion) criterion).getCriterion();
55     		} else {
56     			_result = null;
57     		}
58     		return _result;
59     	}
60     
61     	/**
62     	 * Retrieves the statement in which the criterion occurs.
63     	 * 
64     	 * @param criterion of interest.
65     	 * @return the statement in which the criterion occurs.
66     	 * @throws IllegalArgumentException if the criterion is not a valid criterion type.
67     	 * @pre criterion != null
68     	 * @post result != null
69     	 */
70     	public static Stmt getOccurringStmt(final ISliceCriterion criterion) {
71     		final Stmt _result;
72     
73     		if (criterion instanceof ExprLevelSliceCriterion) {
74     			_result = ((ExprLevelSliceCriterion) criterion).getOccurringStmt();
75     		} else if (criterion instanceof StmtLevelSliceCriterion) {
76     			_result = (Stmt) ((StmtLevelSliceCriterion) criterion).getCriterion();
77     		} else {
78     			throw new IllegalArgumentException(
79     					"The type of \"criterion\" has to be one of StmtLevelSliceCriterion or ExprLevelSliceCriterion.");
80     		}
81     		return _result;
82     	}
83     
84     	/**
85     	 * Checks if the criterion's execution is considered during slicing.
86     	 * 
87     	 * @param criterion of interest.
88     	 * @return <code>true</code> if the criterion's execution is considered during slicing; <code>false</code>,
89     	 *         otherwise.
90     	 * @pre criterion != null
91     	 */
92     	public static boolean isConsiderExecution(final ISliceCriterion criterion) {
93     		return ((AbstractSliceCriterion) criterion).isConsiderExecution();
94     	}
95     }
