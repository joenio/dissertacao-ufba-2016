193                     final Stmt _stmt = _pair.getFirst();

200                             final CallTriple _ctrp = (CallTriple) _i.next();

210             /**
211              * This helps <code>getInterProcedurallyEnclosedStmts</code> methods
212              * to calculate enclosures.
213              * 
214              * @param method of interest.
215              * @param transitive <code>true</code> if transitive closure is
216              *            required; <code>false</code>, otherwise.
217              * @param method2stmts an out parameter in which each method will be
218              *            mapped to a set of statements.
219              * @param stmtsWithInvokeExpr is an iterator over statements with invoke
220              *            expresssions.
221              * @pre method != null
222              * @pre methdo2stmts != null
223              * @pre stmtsWithInvokeExpr != null
224              * @post method2stmts.oclIsKindOf(Map(SootMethod, Collection(Stmt)))
225              */
226             private void calculateInterprocedurallyEnclosedStmts(final SootMethod method, final boolean transitive,
227                     final Map<SootMethod, Collection<Stmt>> method2stmts, final Iterator<Stmt> stmtsWithInvokeExpr) {
228                 final Context _context = new Context();
229                 final IWorkBag<Pair<Stmt, SootMethod>> _wb = new HistoryAwareLIFOWorkBag<Pair<Stmt, SootMethod>>(
230                         new HashSet<Pair<Stmt, SootMethod>>());
231     
232                 for (final Iterator<Stmt> _i = stmtsWithInvokeExpr; _i.hasNext();) {
233                     final Stmt _stmt = _i.next();
234                     _wb.addWork(pairMgr.getPair(_stmt, method));
235                 }
236     
237                 while (_wb.hasWork()) {
238                     final Pair<Stmt, SootMethod> _pair = _wb.getWork();
239                     final Stmt _stmt = _pair.getFirst();
240                     final SootMethod _caller = _pair.getSecond();
241                     _context.setStmt(_stmt);
242                     _context.setRootMethod(_caller);
243     
244                     final Collection<SootMethod> _callees = cgi.getCallees(_stmt.getInvokeExpr(), _context);
245                     final Iterator<SootMethod> _i = _callees.iterator();
246                     final int _iEnd = _callees.size();
247     
248                     for (int _iIndex = 0; _iIndex < _iEnd; _iIndex++) {
249                         final SootMethod _callee = _i.next();
250     
251                         if (!_callee.isSynchronized() || (_callee.isSynchronized() && transitive)) {
252                             final Collection<Stmt> _stmts = getUnenclosedStmtsOf(_callee);
253                             MapUtils.putAllIntoCollectionInMap(method2stmts, _callee, _stmts);
254     
255                             for (final Stmt _s : IteratorUtils.filteredIterable(_stmts,
256                                     SootPredicatesAndTransformers.INVOKING_STMT_PREDICATE)) {
257                                 _wb.addWork(pairMgr.getPair(_s, _callee));
258                             }
259                         }
260                     }
261                 }

472         /**
473          * Calculates the synchronization dependency information for the methods
474          * provided during initialization.
475          * 
476          * @see edu.ksu.cis.indus.staticanalyses.dependency.AbstractDependencyAnalysis#analyze()
477          */
478         @Override
479         public void analyze() {
480             unstable();
481     
482             if (LOGGER.isInfoEnabled()) {
483                 LOGGER.info("BEGIN: Monitor Analysis processing");
484             }
485     
486             final Map<EnterMonitorStmt, Collection<ExitMonitorStmt>> _enter2exits = new HashMap<EnterMonitorStmt, Collection<ExitMonitorStmt>>();
487             final Collection<EnterMonitorStmt> _processedMonitors = new HashSet<EnterMonitorStmt>();
488     
489             /*
490              * Calculating monitor info is not as simple as it looks in the presence
491              * of exceptions. The exit monitors are the tricky ones. The exit
492              * monitors are guarded by catch block generated by the compiler. The
493              * user cannot generate such complex catch blocks, but nevertheless we
494              * cannot assume about the compiler. So, we proceed by calculating the
495              * monitor info over a complete unit graph and use object flow
496              * information to arbitrate suspicious enter-exit matches.
497              */
498             for (final Iterator<Map.Entry<SootMethod, Collection<EnterMonitorStmt>>> _i = method2enterMonitors.entrySet()
499                     .iterator(); _i.hasNext();) {
500                 final Map.Entry<SootMethod, Collection<EnterMonitorStmt>> _entry = _i.next();
501                 final SootMethod _method = _entry.getKey();
502                 _enter2exits.clear();
503     
504                 for (final Iterator<EnterMonitorStmt> _j = _entry.getValue().iterator(); _j.hasNext();) {
505                     final EnterMonitorStmt _enterMonitor = _j.next();
506                     final Map<Triple<EnterMonitorStmt, ExitMonitorStmt, SootMethod>, Collection<Stmt>> _monitor2enclosedStmts = MapUtils
507                             .getMapFromMap(method2monitor2enclosedStmts, _method);
508     
509                     if (!_processedMonitors.contains(_enterMonitor) && _monitor2enclosedStmts.get(_enterMonitor) == null) {
510                         _processedMonitors.addAll(processMonitor(_processedMonitors, _enterMonitor, _method, _enter2exits));
511                     }
512                 }
513             }
514     
515             method2enterMonitors.clear();
516             stable();
517     
518             if (LOGGER.isDebugEnabled()) {
519                 LOGGER.debug("analyze() - " + toString());
520             }
521     
522             if (LOGGER.isInfoEnabled()) {
523                 LOGGER.info("END: Monitor Analysis processing");
524             }
