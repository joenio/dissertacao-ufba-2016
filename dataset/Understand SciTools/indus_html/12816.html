17     package edu.ksu.cis.indus.staticanalyses.flow;
18     
19     import edu.ksu.cis.indus.processing.Context;
20     
21     /**
22      * This interface is used to manage variants corresponding to entities.
23      *
24      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
25      * @author $Author: rvprasad $
26      * @version $Revision: 1.5 $
27      * @param <V> is the type of the variant.
28      * @param <E> is the type of the entity whose variance is being managed.
29      */
30     public interface IVariantManager<V extends IVariant, E> {
31     
32     	/**
33     	 * Returns the variant corresponding to the given entity in the given context, if one exists.
34     	 *
35     	 * @param o the entity whose variant is to be returned.
36     	 * @param context the context corresponding to which the variant is requested.
37     	 * @return the variant correponding to the entity in the given context, if one exists. <code>null</code> if none exist.
38     	 * @pre o != null and context != null
39     	 */
40     	V query(final E o, final Context context);
41     
42     	/**
43     	 * Resets the manager. All internal data structures are reset to enable a new session of usage.
44     	 */
45     	void reset();
46     
47     	/**
48     	 * Returns the variant corresponding to the given entity in the given context. If a variant does not exist, a new one is
49     	 * created. If one exists, it shall be returned.
50     	 *
51     	 * @param o the entity whose variant is to be returned.
52     	 * @param context the context corresponding to which the variant is requested.
53     	 * @return the variant correponding to the entity in the given context.
54     	 * @pre o != null and context != null
55     	 * @post result != null
56     	 */
57     	V select(final E o, final Context context);
58     }
