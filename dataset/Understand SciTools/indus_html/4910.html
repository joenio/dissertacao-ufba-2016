26     import java.util.List;
27     import java.util.MissingResourceException;

29     import org.slf4j.Logger;
30     import org.slf4j.LoggerFactory;

32     import soot.Scene;

35     /**
36      * This class is the criteria specification.  <i>This is intended for internal use only.  Clients should not depend on or use
37      * this class.</i>
38      *
39      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
40      * @author $Author: rvprasad $
41      * @version $Revision: 1.4 $ $Date: 2007/02/10 19:06:02 $
42      */
43     public final class SliceCriteriaSpec {
44     	/**
45     	 * The logger used by instances of this class to log messages.
46     	 */
47     	private static final Logger LOGGER = LoggerFactory.getLogger(SliceCriteriaSpec.class);
48     
49     	/**
50     	 * The collection of criterion specification.
51     	 */
52     	private List<SliceCriterionSpec> criteria;
53     
54     	/**
55     	 * Creates an instance of this class.
56     	 */
57     	private SliceCriteriaSpec() {
58     		criteria = createCriteriaContainer();
59     	}
60     
61     	/**
62     	 * Generates criteria relative on the given scene.
63     	 *
64     	 * @param scene relative to which the criteria will be generated.
65     	 *
66     	 * @return the collection of slice criteria.
67     	 *
68     	 * @pre scene != null
69     	 * @post result != null
70     	 */
71     	public Collection<ISliceCriterion> getCriteria(final Scene scene) {
72     		final Collection<ISliceCriterion> _result = new HashSet<ISliceCriterion>();
73     
74     		for (final Iterator<SliceCriterionSpec> _i = criteria.iterator(); _i.hasNext();) {
75     			final SliceCriterionSpec _critSpec = _i.next();
76     
77     			try {
78     				_result.addAll(_critSpec.getCriteria(scene));
79     			} catch (final MissingResourceException _e) {
80     				LOGGER.error("Discarding criteria - " + _critSpec, _e);
81     			}
82     		}
83     		return _result;
84     	}
85     
86     	/**
87     	 * Retrieves the criteria specification for the given collection of slicing criteria.
88     	 *
89     	 * @param criteria is a collection of slice criterion.
90     	 *
91     	 * @return the criteria specification.
92     	 *
93     	 * @pre criteria != null
94     	 * @post result != null
95     	 */
96     	static SliceCriteriaSpec getSliceCriteriaSpec(final Collection<ISliceCriterion> criteria) {
97     		final List<SliceCriterionSpec> _temp = new ArrayList<SliceCriterionSpec>();
98     
99     		for (final Iterator<ISliceCriterion> _i = criteria.iterator(); _i.hasNext();) {
100     			final ISliceCriterion _criterion = _i.next();
101     			_temp.addAll(SliceCriterionSpec.getCriterionSpec(_criterion));
102     		}
103     
104     		final SliceCriteriaSpec _result = new SliceCriteriaSpec();
105     		_result.criteria = _temp;
106     		return _result;
107     	}
108     
109     	/**
110     	 * Creates an instance of the container for criteria.  This is primarily used during deserialization.
111     	 *
112     	 * @return the instance of the container.
113     	 *
114     	 * @post result != null
115     	 */
116     	private static List<SliceCriterionSpec> createCriteriaContainer() {
117     		return new ArrayList<SliceCriterionSpec>();
118     	}
119     }
