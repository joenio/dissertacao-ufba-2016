17     package edu.ksu.cis.indus.tools.slicer.criteria.generators;
18     
19     import edu.ksu.cis.indus.annotations.NonNull;
20     import edu.ksu.cis.indus.annotations.NonNullContainer;
21     import edu.ksu.cis.indus.common.collections.CollectionUtils;
22     import edu.ksu.cis.indus.common.collections.ReflectionBasedSupertypePredicate;
23     
24     import java.util.Collection;
25     import java.util.HashSet;
26     
27     import org.slf4j.Logger;
28     import org.slf4j.LoggerFactory;
29     
30     import soot.jimple.Stmt;
31     
32     /**
33      * This class can be used to generate slice criteria based on statements of certain types/classes specified by the user. The
34      * criteria is also guided by a specification matcher.
35      * 
36      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
37      * @author $Author: rvprasad $
38      * @version $Revision: 1.8 $ $Date: 2007/02/10 19:06:00 $
39      */
40     public final class StmtTypeBasedSliceCriteriaGenerator
41     		extends AbstractStmtBasedSliceCriteriaGenerator<Stmt> {
42     
43     	/**
44     	 * The logger used by instances of this class to log messages.
45     	 */
46     	private static final Logger LOGGER = LoggerFactory.getLogger(StmtTypeBasedSliceCriteriaGenerator.class);
47     
48     	/**
49     	 * The types of the statements to be considered as slice criteria.
50     	 */
51     	private final Collection<Class<? extends Stmt>> stmtTypes = new HashSet<Class<? extends Stmt>>();
52     
53     	/**
54     	 * This is used to check type conformance.
55     	 */
56     	private final ReflectionBasedSupertypePredicate<Stmt> subClassPredicate = new ReflectionBasedSupertypePredicate<Stmt>();
57     
58     	/**
59     	 * Sets the types of the statements to be considered as slice criteria.
60     	 * 
61     	 * @param <T> is the type of the statement AST nodes.
62     	 * @param types of the statements.
63     	 */
64     	public <T extends Stmt> void setStmtTypes(@NonNull @NonNullContainer final Collection<Class<T>> types) {
65     		stmtTypes.clear();
66     		stmtTypes.addAll(types);
67     	}
68     
69     	/**
70     	 * {@inheritDoc}
71     	 */
72     	@Override protected boolean shouldConsiderStmt(final Stmt stmt) {
73     		subClassPredicate.setsubType(stmt.getClass());
74     		final boolean _result = CollectionUtils.exists(stmtTypes, subClassPredicate);
75     
76     		if (LOGGER.isDebugEnabled()) {
77     			LOGGER.debug("shouldConsiderStmt(Stmt stmt = " + stmt + ":" + stmt.getClass() + ") = " + _result);
78     		}
79     
80     		return _result;
81     	}
82     }
