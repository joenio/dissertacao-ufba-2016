40     /**
41      * This class provides a graphical interface to configure composite tool configurations. It typically provides the support to
42      * pick each configuration and delegates the configuration of each set of configuration information to specific configurators.
43      * 
44      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
45      * @author $Author: rvprasad $
46      * @version $Revision: 1.23 $ $Date: 2007/02/10 19:08:40 $
47      */
48     public final class CompositeToolConfigurator
49     		extends AbstractToolConfigurator {
50     
51     	/**
52     	 * This is composite on which the child configurator will be displayed.
53     	 */
54     	private Composite childComposite;
55     
56     	/**
57     	 * This is the child configurator to be used to configure each instance of configuration.
58     	 * 
59     	 * @invariant childConfigurator != null
60     	 */
61     	private IToolConfigurator childConfigurator;
62     
63     	/**
64     	 * This is the composite configuration being configured.
65     	 * 
66     	 * @invariant compositeConfiguration != null
67     	 */
68     	private CompositeToolConfiguration compositeConfiguration;
69     
70     	/**
71     	 * This combo presents the available configurations.
72     	 */
73     	private Combo configCombo;
74     
75     	/**
76     	 * This is the index that is selected in the combo.
77     	 */
78     	private int selectedIndex;
79     
80     	/**
81     	 * The factory used to create a new configuration instance.
82     	 * 
83     	 * @invariant toolConfigFactory != null
84     	 */
85     	private IToolConfigurationFactory toolConfigFactory;
86     
87     	/**
88     	 * Creates a new CompositeToolConfigurator object.
89     	 * 
90     	 * @param compositeConfigs is the composite configuration.
91     	 * @param child is the configurator to be used for each configuration instance.
92     	 * @param factory is used to create tool configuration of a specific type when there exists none.
93     	 * @pre compositeConfigs != null and child != null and factory != null
94     	 */
95     	public CompositeToolConfigurator(final CompositeToolConfiguration compositeConfigs, final IToolConfigurator child,
96     			final IToolConfigurationFactory factory) {
97     		compositeConfiguration = compositeConfigs;
98     		childConfigurator = child;
99     		toolConfigFactory = factory;
100     	}
101     
102     	/**
103     	 * @see edu.ksu.cis.indus.tools.AbstractToolConfigurator#checkConfiguration(edu.ksu.cis.indus.tools.IToolConfiguration)
104     	 */
105     	@Empty @Override protected void checkConfiguration(@SuppressWarnings("unused") final IToolConfiguration t) {
106     		// does nothing
107     	}
108     
109     	/**
110     	 * {@inheritDoc}
111     	 */
112     	@Override protected void setup() {
113     		final GridLayout _gridLayout = new GridLayout();
114     		_gridLayout.numColumns = 3;
115     		parent.setLayout(_gridLayout);
116     
117     		final Label _label = new Label(parent, SWT.NONE);
118     		GridData _gridData = new GridData();
119     		_gridData.horizontalSpan = 1;
120     		_label.setLayoutData(_gridData);
121     		_label.setText("Configurations:");
122     
123     		configCombo = new Combo(parent, SWT.DROP_DOWN);
124     		configCombo.setItems(new String[0]);
125     		configCombo.setLayoutData(new GridData(GridData.FILL_HORIZONTAL));
126     		configCombo.addSelectionListener(new SelectionListener() {
127     
128     			public void widgetDefaultSelected(final SelectionEvent evt) {
129     				widgetSelected(evt);
130     			}
131     
132     			public void widgetSelected(@SuppressWarnings("unused") final SelectionEvent evt) {
133     				recordSelection();
134     				displayChild();
135     				parent.layout();
136     				parent.pack();
137     			}
138     		});
139     
140     		configCombo.addFocusListener(new FocusAdapter() {
141     
142     			@Override public void focusGained(@SuppressWarnings("unused") final FocusEvent evt) {
143     				recordSelection();
144     			}
145     
146     			@Override public void focusLost(@SuppressWarnings("unused") final FocusEvent evt) {
147     				updateConfigName();
148     			}
149     		});
150     
151     		configCombo.setVisible(true);
152     
153     		final Button _newConfig = new Button(parent, SWT.PUSH);
154     		_newConfig.setText("Create");
155     		_gridData = new GridData(GridData.HORIZONTAL_ALIGN_END);
156     		_gridData.horizontalIndent = _newConfig.getText().length();
157     		_newConfig.setLayoutData(_gridData);
158     		_newConfig.addSelectionListener(new SelectionListener() {
159     
160     			public void widgetDefaultSelected(final SelectionEvent evt) {
161     				widgetSelected(evt);
162     			}
163     
164     			public void widgetSelected(@SuppressWarnings("unused") final SelectionEvent evt) {
165     				createNewConfiguration();
166     				displayChild();
167     				parent.layout();
168     				parent.pack();
169     			}
170     		});
171     
172     		if (compositeConfiguration.configurations.isEmpty()) {
173     			compositeConfiguration.configurations.add(toolConfigFactory.createToolConfiguration());
174     		}
175     
176     		for (final Iterator _i = compositeConfiguration.configurations.iterator(); _i.hasNext();) {
177     			final IToolConfiguration _config = (IToolConfiguration) _i.next();
178     			configCombo.add(_config.getConfigName());
179     		}
180     
181     		final IToolConfiguration _c = compositeConfiguration.getActiveToolConfiguration();
182     		configCombo.select(compositeConfiguration.configurations.indexOf(_c));
183     		displayChild();
184     	}
185     
186     	/**
187     	 * Creates a new configuration.
188     	 */
189     	void createNewConfiguration() {
190     		final IToolConfiguration _temp = toolConfigFactory.createToolConfiguration();
191     		_temp.setConfigName("tool_configuration_" + compositeConfiguration.configurations.size());
192     		compositeConfiguration.addToolConfiguration(_temp);
193     		configCombo.add(_temp.getConfigName());
194     		configCombo.select(compositeConfiguration.configurations.indexOf(_temp));
195     	}
196     
197     	/**
198     	 * Displays the child configurator.
199     	 */
200     	void displayChild() {
201     		if (childComposite != null) {
202     			childComposite.dispose();
203     		}
204     		childComposite = new Composite(parent, SWT.NONE);
205     
206     		final GridData _gridData = new GridData(GridData.FILL_HORIZONTAL);
207     		_gridData.horizontalSpan = 3;
208     		childComposite.setLayoutData(_gridData);
209     		childComposite.setVisible(true);
210     
211     		final int _index = configCombo.getSelectionIndex();
212     
213     		if (_index != -1) {
214     			final IToolConfiguration _tc = compositeConfiguration.configurations.get(_index);
215     			compositeConfiguration.setActiveToolConfiguration(_tc);
216     			childConfigurator.setConfiguration(_tc);
217     			childConfigurator.initialize(childComposite);
218     		}
219     	}
220     
221     	/**
222     	 * Records the index of the current selection in the Combo.
223     	 */
224     	void recordSelection() {
225     		selectedIndex = configCombo.getSelectionIndex();
226     	}
227     
228     	/**
229     	 * Updates the selected configuration's name if it was edited in the text box.
230     	 */
231     	void updateConfigName() {
232     		final int _selIndex = configCombo.getSelectionIndex();
233     
234     		// if text was changed, selection index is -1 when the name is changed.
235     		if (_selIndex < 0) {
236     			// retrive the text at previously selected index and the corresponding configuration
237     			final List _configurations = compositeConfiguration.configurations;
238     			final IToolConfiguration _selConfig = (IToolConfiguration) _configurations.get(selectedIndex);
239     
240     			// retrieve the new Text in the text box.
241     			final String _newText = configCombo.getText();
242     			boolean _noDuplicate = true;
243     
244     			// check if the new name will lead to duplicate entries.
245     			for (int _i = configCombo.getItemCount() - 1; _i >= 0 && _noDuplicate; _i--) {
246     				if (((IToolConfiguration) _configurations.get(_i)).getConfigName().equals(_newText)) {
247     					final MessageBox _msgBox = new MessageBox(parent.getShell(), SWT.OK | SWT.ICON_INFORMATION);
248     					_msgBox.setMessage("A configuration with the name of \"" + _newText
249     							+ "\" exists.  \nNo changes will be made.");
250     					_msgBox.open();
251     					_noDuplicate = false;
252     				}
253     			}
254     
255     			// if there will be no duplicate entries, then...
256     			if (_noDuplicate) {
257     				_selConfig.setConfigName(_newText);
258     				configCombo.remove(selectedIndex);
259     				configCombo.add(_newText, selectedIndex);
260     			}
261     		}
262     	}
263     }
