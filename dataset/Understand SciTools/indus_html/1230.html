34     /**
35      * This class hides the work involved in the creation of slice criteria from the environment.
36      * <p>
37      * The interesting issue while creating slicing criteria is the inclusion information. Please refer to {@link SlicingEngine
38      * SlicingEngine} for the sort of slices we discuss here. In case of expression-level slice criterion, inclusion would mean
39      * that the entire statement containing the expression needs to be included independent of the slice type, i.e., backward or
40      * complete. The same applies to statement-level slice criterion.
41      * </p>
42      * <p>
43      * On the otherhand, non-inclusion in the case complete slicing does not make sense. However, in case of backward slicing
44      * considering expresison-level slice criterion would mean that the expression should not be included in the sliced system
45      * which only makes sense, but this introduces a dependency between the expression and the containing statement. In
46      * particular, the expression is control dependent on the statement, and so the statement should be included for capturing
47      * control dependency. This is exactly what our implementation does.
48      * </p>
49      * <p>
50      * On mentioning a statement as a non-inclusive slice criterion, we interpret that as all artifacts that affect the reaching
51      * of that statement. On mentioning an expression as a non-inclusive slice criterion, we interpret that as all artifacts
52      * leading to the value of the expression in the statement it occurs in, hence we will include the statement containing the
53      * expression in a non-inclusive manner to capture control dependency leading to that statement.
54      * </p>
55      * <p>
56      * As we are interested in meaningful slices, inclusive expression-level slice criterion is the same as inclusive
57      * statement-level slice criterion which refers to the statement that contains the expression.
58      * </p>
59      * <p>
60      * When an expression/statement is specified as a criteria, does this mean that
61      * <ul>
62      * <li> the value of the expression should be preserved (the expression is executed), or </li>
63      * <li> the control reaching the expression should be preserved (the expression is not executed)? </li>
64      * </ul>
65      * To enable the user to specify this information, each factory method includes a <code>considerExecution</code> parameter
66      * that incides if the execution of the expression/statement should be captured or not. If you think that this is overly fine,
67      * I would be interested in a discussion.
68      * </p>
69      *
70      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
71      * @author $Author: rvprasad $
72      * @version $Revision: 1.24 $
73      */

117     	/**
118     	 * Clones the given criterion.
119     	 *
120     	 * @param criterion to be cloned.
121     	 * @return the clone.
122     	 * @throws IllegalArgumentException if criterion is not of type <code>ISliceCriterion</code>.
123     	 * @pre criterion != null
124     	 */
125     	public ISliceCriterion clone(final ISliceCriterion criterion) {
126     		final ISliceCriterion _result;
127     
128     		if (criterion instanceof StmtLevelSliceCriterion) {
129     			final StmtLevelSliceCriterion _t = (StmtLevelSliceCriterion) criterion;
130     			_result = getStmtCriteria(_t.getOccurringMethod(), (Stmt) _t.getCriterion(), _t.isConsiderExecution());
131     
132     			final Stack<CallTriple> _callStack = _t.getCallStack();
133     
134     			if (_callStack != null) {
135     				_result.setCallStack(_callStack.clone());
136     			}
137     		} else if (criterion instanceof ExprLevelSliceCriterion) {
138     			final ExprLevelSliceCriterion _t = (ExprLevelSliceCriterion) criterion;
139     			_result = getExprCriteria(_t.getOccurringMethod(), _t.getOccurringStmt(), (ValueBox) _t.getCriterion(), _t
140     					.isConsiderExecution());
141     
142     			final Stack<CallTriple> _callStack = _t.getCallStack();
143     
144     			if (_callStack != null) {
145     				_result.setCallStack(_callStack.clone());
146     			}
147     		} else if (criterion instanceof MethodLevelSliceCriterion) {
148     			final MethodLevelSliceCriterion _m = (MethodLevelSliceCriterion) criterion;
149     			_result = getMethodCriteria(_m.getOccurringMethod(), _m.isConsiderExecution());
150     			final Stack<CallTriple> _callStack = _m.getCallStack();
151     
152     			if (_callStack != null) {
153     				_result.setCallStack(_callStack.clone());
154     			}
155     		} else {
156     			throw new IllegalArgumentException("criterion's type " + criterion.getClass() + " is unknown.");
157     		}
158     		return _result;
159     	}

161     	/**
162     	 * Returns a collection of criteria which include all occurrences of the given local in the given collection of
163     	 * statements.
164     	 *
165     	 * @param local is the local variable whose all occurrences in <code>method</code> should be captured as slice criterion
166     	 * @param stmts in which the occurrences of the <code>local</code> should be captured.
167     	 * @param method in which the given statements occur.
168     	 * @param considerExecution <code>true</code> indicates the result of executing the criterion or the control reaching
169     	 *            the end of criterion is required; <code>false</code> indicates the result of the control reaching the
170     	 *            criterion is required.
171     	 * @return a collection of slice criteria.
172     	 * @pre stmts != null and local != null and method != null
173     	 * @pre method.retrieveActiveBody().getUnits().containsAll(stmts)
174     	 */
175     	public Collection<ISliceCriterion> getCriteria(final Local local, final Collection<Stmt> stmts, final SootMethod method,
176     			final boolean considerExecution) {
177     		final Collection<ISliceCriterion> _result = new HashSet<ISliceCriterion>();
178     
179     		for (final Iterator<Stmt> _i = stmts.iterator(); _i.hasNext();) {
180     			final Stmt _stmt = _i.next();
181     
182     			for (final Iterator<ValueBox> _j = _stmt.getUseAndDefBoxes().iterator(); _j.hasNext();) {
183     				final ValueBox _vBox = _j.next();
184     
185     				if (_vBox.getValue().equals(local)) {
186     					final ExprLevelSliceCriterion _exprCriterion = getExprCriteria(method, _stmt, _vBox, considerExecution);
187     					_result.add(_exprCriterion);
188     				}
189     			}
190     		}
191     
192     		return _result;
193     	}

225     	/**
226     	 * Creates slice criteria from the given statement. If <code>descend</code> is <code>true</code>, then a criterion is
227     	 * created for each use/def site in the given excpression.
228     	 *
229     	 * @param method in which the criterion occurs.
230     	 * @param stmt is the criterion.
231     	 * @param descend <code>true</code> indicates if a criterion should be generated for each def/use site in the statement;
232     	 *            <code>false</code>, otherwise.
233     	 * @param considerExecution <code>true</code> indicates the result of executing the criterion or the control reaching
234     	 *            the end of criterion is required; <code>false</code> indicates the result of the control reaching the
235     	 *            criterion is required.
236     	 * @return a collection of slice criterion corresponding to the given criterion.
237     	 * @pre method != null and stmt != null
238     	 */
239     	public Collection<ISliceCriterion> getCriteria(final SootMethod method, final Stmt stmt, final boolean descend,
240     			final boolean considerExecution) {
241     		final Collection<ISliceCriterion> _result = new HashSet<ISliceCriterion>();
242     		final StmtLevelSliceCriterion _stmtCriterion = getStmtCriteria(method, stmt, considerExecution);
243     		_result.add(_stmtCriterion);
244     
245     		if (descend) {
246     			for (final Iterator<ValueBox> _i = stmt.getUseAndDefBoxes().iterator(); _i.hasNext();) {
247     				final ValueBox _vBox = _i.next();
248     				final ExprLevelSliceCriterion _temp = getExprCriteria(method, stmt, _vBox, considerExecution);
249     				_result.add(_temp);
250     			}
251     		}
252     
253     		return _result;
254     	}

283     	 * @param considerExecution <code>true</code> indicates the result of executing the criterion or the control reaching
284     	 *            the end of criterion is required; <code>false</code> indicates the result of the control reaching the

286     	 * @return a collection of slice criterion objects corresponding to the given criterion.
287     	 * @pre method != null and stmt != null and expr != null

289     	public Collection<ISliceCriterion> getCriteria(final SootMethod method, final Stmt stmt, final ValueBox expr,
290     			final boolean descend, final boolean considerExecution) {
291     		final Collection<ISliceCriterion> _result = new HashSet<ISliceCriterion>();
292     		final ExprLevelSliceCriterion _exprCriterion = getExprCriteria(method, stmt, expr, considerExecution);
293     		_result.add(_exprCriterion);
294     
295     		if (descend) {
296     			for (final Iterator<ValueBox> _i = expr.getValue().getUseBoxes().iterator(); _i.hasNext();) {
297     				final ValueBox _useBox = _i.next();
298     				final ExprLevelSliceCriterion _temp = getExprCriteria(method, stmt, _useBox, considerExecution);
299     				_result.add(_temp);
300     			}
301     		}
302     		return _result;
303     	}
