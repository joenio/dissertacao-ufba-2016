177     	 * Fixes up the field maps of the alias sets in the given map. When alias sets are cloned, the field maps are cloned.
178     	 * Hence, they are shallow copied. This method clones the relation between the alias sets among their clones.
179     	 * 
180     	 * @param src2clone maps an representative alias set to it's clone. This is also an out parameter that will contain new
181     	 *            mappings.
182     	 * @throws CloneNotSupportedException when <code>clone()</code> fails.
183     	 */
184     	private static void fixUpFieldMapsOfClone(final Map<AliasSet, AliasSet> src2clone) throws CloneNotSupportedException {
185     		final IWorkBag<AliasSet> _wb = new HistoryAwareLIFOWorkBag<AliasSet>(new HashSet<AliasSet>());
186     
187     		_wb.addAllWork(src2clone.keySet());
188     
189     		while (_wb.hasWork()) {
190     			final AliasSet _src = _wb.getWork();
191     			final AliasSet _clone = src2clone.get(_src);
192     			final Map<String, AliasSet> _srcASFieldMap = _src.getFieldMap();
193     			final Set<String> _srcASFields = _srcASFieldMap.keySet();
194     			final Iterator<String> _i = _srcASFields.iterator();

197     			for (int _iIndex = 0; _iIndex < _iEnd; _iIndex++) {
198     				final String _field = _i.next();
199     
200     				/*
201     				 * We use the representative alias set as it is possible that a field may have 2 alias sets in different
202     				 * contexts but the same representative alias set in both contexts. We don't do the same for clones as they
203     				 * are the representatives until they are unified, which happens in the following block.
204     				 */
205     				final AliasSet _srcFieldAS = _src.getASForField(_field).find();
206     				final AliasSet _cloneFieldAS;
207     
208     				_cloneFieldAS = getCloneOf(src2clone, _srcFieldAS);
209     				// if the clone is different from the src only then we need to further process src.
210     				if (_cloneFieldAS != _srcFieldAS) {
211     					_wb.addWork(_srcFieldAS);
212     				}
213     				_clone.putASForField(_field, _cloneFieldAS);
214     			}
215     		}
216     
217     		// Unify the clones to reflect the relation between their originators.
218     		for (final Iterator<AliasSet> _i = src2clone.keySet().iterator(); _i.hasNext();) {
219     			final AliasSet _k1 = _i.next();
220     
221     			for (final Iterator<AliasSet> _j = src2clone.keySet().iterator(); _j.hasNext();) {
222     				final AliasSet _k2 = _j.next();
223     
224     				if (_k1 != _k2 && _k1.find() == _k2.find()) {
225     					final AliasSet _v1 = src2clone.get(_k1);
226     					final AliasSet _v2 = src2clone.get(_k2);
227     					_v1.unifyAliasSetHelper(_v2, false);
228     				}
229     			}
230     		}

232     	}

236     	 * 
237     	 * @param src2clone maps alias set to alias set.
238     	 * @param as for which the clone is required.

260     	 * Clones this object.

263     	 * @throws CloneNotSupportedException if <code>java.lang.Object.clone()</code> fails.

608     
609     		if (_mThis != null) {
610     			AliasSet.selfUnify(_mThis);
611     		}
612     	}
613     
614     	/**
615     	 * Unifies this context with the given context.
616     	 * <p>
617     	 * There is asymmetry in alias sets in case of contexts. It is possible that this context may not have alias set for an
618     	 * argument location where as <code>p</code> may have one. This happens in cases where this context corresponds to a
619     	 * site-context where <code>null</code> is being passed, hence, that arg position does not require an alias set whereas
620     	 * <code>p</code> being the context of the method being called requires alias set as there may be sites where the
621     	 * non-null argument may be provided. In such cases, we safely warn and continue. However, this cannot happen for thrown
622     	 * exception, this, or return references. associated with call sites and methods.
623     	 * </p>
624     	 * 
625     	 * @param p is the context with which the unification should occur.
626     	 */
627     	void unifyMethodContext(final MethodContext p) {
628     		if (p == null) {
629     			LOGGER.error("Unification with null requested.");
630     			throw new IllegalArgumentException("Unification with null requested.");
631     		}
632     
633     		final MethodContext _m = find();
634     		final MethodContext _n = p.find();
635     
636     		if (_m != _n) {
637     			_m.union(_n);
638     
639     			final MethodContext _representative = _m.find();
640     			final MethodContext _represented;
641     
642     			if (_representative == _m) {
643     				_represented = _n;
644     			} else {
645     				_represented = _m;
646     			}
647     			_representative.globalDataWritten |= _represented.globalDataWritten;
648     			_representative.globalDataRead |= _represented.globalDataRead;
649     
650     			final int _paramCount = method.getParameterCount();
651     
652     			for (int _i = 0; _i < _paramCount; _i++) {
653     				if (EquivalenceClassBasedEscapeAnalysis.canHaveAliasSet(method.getParameterType(_i))) {
654     					final AliasSet _mAS = _representative.argAliasSets.get(_i);
655     					final AliasSet _nAS = _represented.argAliasSets.get(_i);
656     					if (_mAS == null) {
657     						_representative.argAliasSets.set(_i, _nAS);
658     					} else {
659     						_mAS.unifyAliasSet(_nAS);
660     					}
661     				}
662     			}
663     			unifyAliasSets(_representative.ret, _represented.ret);
664     			unifyAliasSets(_representative.thrown, _represented.thrown);
665     			unifyAliasSets(_representative.thisAS, _represented.thisAS);
666     		}
667     	}
668     
669     	/**
670     	 * Unifies the given alias sets.
671     	 * 
672     	 * @param representative is one of the alias set to be unified.
673     	 * @param represented is the other alias set to be unified.
674     	 * @pre representative != null and represented != null
675     	 */
676     	private void unifyAliasSets(final AliasSet representative, final AliasSet represented) {
677     		if ((representative == null && represented != null) || (representative != null && represented == null)) {
678     			LOGGER.error("Incompatible method contexts being unified - representative - " + representative
679     					+ "\n represented - " + represented);
680     			throw new IllegalStateException("Unifying null aliasSet");
681     		} else if (representative != null) {
682     			representative.unifyAliasSet(represented);
683     		}
684     	}
685     }
