17     package edu.ksu.cis.indus.tools.slicer.criteria.predicates;
18     
19     import edu.ksu.cis.indus.common.scoping.SpecificationBasedScopeDefinition;
20     import soot.SootField;
21     
22     /**
23      * This class provides facility to filter field based slice criteria based on scope specification.
24      *
25      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
26      * @author $Author: rvprasad $
27      * @version $Revision: 1.5 $ $Date: 2007/10/09 15:58:59 $
28      */
29     public class ScopeBasedFieldPredicate
30     		extends AbstractSliceCriteriaPredicate<SootField> {
31     
32     	/**
33     	 * The specification-based matcher to be used.
34     	 */
35     	private SpecificationBasedScopeDefinition matcher;
36     
37     	/**
38     	 * Checks if criteria should be generated from the given entity.
39     	 *
40     	 * @param entity that serves as the basis for the slice criteria.
41     	 * @return <code>true</code> if criteria should be generated; <code>false</code>, otherwise.
42     	 * @pre entity.oclIsKindOf(SootField)
43     	 */
44     	public boolean evaluate(final SootField entity) {
45     		return matcher.isInScope(entity, getSlicerTool().getSystem());
46     	}
47     
48     	/**
49     	 * Sets the value of <code>matcher</code>. This method should be called before using this generator.
50     	 *
51     	 * @param theMatcher the new value of <code>matcher</code>.
52     	 * @pre theMatcher != null
53     	 */
54     	public void setMatcher(final SpecificationBasedScopeDefinition theMatcher) {
55     		matcher = theMatcher;
56     	}
57     }
