93     /**
94      * This is a facade that exposes the slicer as a tool. This is recommended interface to interact with the slicer if the slicer
95      * is being used as a tool in a tool chain.
96      * <p>
97      * The term "immediate slice" in the context of this file implies the slice containing only entities on which the given term
98      * depends on, not the transitive closure.
99      * </p>
100      * <p>
101      * There are 3 types of slices: forward, backward, and complete(forward and backward). Also, there are 2 flavours of slices:
102      * executable and non-executable.
103      * </p>
104      * <p>
105      * Backward slicing is inclusion of anything that leads to the slice criterion from the given entry points to the system. This
106      * can provide a executable system which will simulate the given system along all paths from the entry points leading to the
107      * slice criterion independent of the input. In case the input causes a divergence in this path then the simulation ends
108      * there.
109      * </p>
110      * <p>
111      * However, in case of forward slicing, one would include everything that is affected by the slice criterion. This will never
112      * lead to an semantically meaningful executable slice as the part of the system that leads to the slice criterion is not
113      * captured. Rather a more meaningful notion is that of a complete slice. This includes everything that affects the given
114      * slice criterion and everything affected by the slice criterion.
115      * </p>
116      * <p>
117      * Due to the above view we only support non-executable slices of all types and only executable slices of backward and
118      * complete type.
119      * </p>
120      * <p>
121      * If slice for deadlock option is true in the slicer configuration, then criteria to preserve deadlocking are generated.
122      * However, these are generated on behalf of the application/driver (as a convenience) and not for internal use. Hence, the
123      * application/driver is responsible for the disposal of these criteria.
124      * </p>
125      * 
126      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
127      * @author $Author: rvprasad $
128      * @version $Revision: 1.181 $ $Date: 2007/02/10 19:06:01 $
129      * @param <T> dummy type parameter.
130      */

302     	/**
303     	 * Creates a new SlicerTool object. The client should relinquish control/ownership of the arguments as they are provided
304     	 * to configure the tool.
305     	 * 
306     	 * @param tokenMgr is the token manager to be used with this instance of slicer tool.
307     	 * @param stmtGraphFactoryToUse is the statement graph factory to use.
308     	 * @pre tokenMgr != null and stmtGraphFactoryToUse != null
309     	 */
310     	public SlicerTool(final ITokenManager<T, Value, Type> tokenMgr, final IStmtGraphFactory<?> stmtGraphFactoryToUse) {
311     		theTokenMgr = tokenMgr;
312     		phase = Phase.createPhase();
313     
314     		rootMethods = new HashSet<SootMethod>();
315     		criteria = new HashSet<ISliceCriterion>();
316     		info = new HashMap<Comparable<?>, Object>();
317     		criteriaGenerators = new HashSet<ISliceCriteriaGenerator<?, ?>>();
318     
319     		stmtGraphFactory = stmtGraphFactoryToUse;
320     
321     		// create the flow analysis.
322     		ofa = OFAnalyzer.getFIOSAnalyzer(FLOW_ANALYSIS_TAG_NAME, tokenMgr, getStmtGraphFactory());
323     		addActivePart(ofa.getActivePart());
324     
325     		// create the pre processor for call graph construction.
326     		cgPreProcessCtrl = new ValueAnalyzerBasedProcessingController();
327     		cgPreProcessCtrl.setAnalyzer(ofa);
328     		cgPreProcessCtrl.setProcessingFilter(new TagBasedProcessingFilter(FLOW_ANALYSIS_TAG_NAME));
329     
330     		final OneAllStmtSequenceRetriever _ssr = new OneAllStmtSequenceRetriever();
331     		cgPreProcessCtrl.setStmtSequencesRetriever(_ssr);
332     
333     		addActivePart(cgPreProcessCtrl.getActivePart());
334     
335     		// create pair manager
336     		pairMgr = new Pair.PairManager(false, true);
337     		// create the call graph.
338     		callGraph = new CallGraphInfo(pairMgr);
339     
340     		// create the pre processor for thread graph construction.
341     		cgBasedPreProcessCtrl = new ValueAnalyzerBasedProcessingController();
342     		cgBasedPreProcessCtrl.setProcessingFilter(new CGBasedProcessingFilter(callGraph));
343     		cgBasedPreProcessCtrl.setAnalyzer(ofa);
344     		cgBasedPreProcessCtrl.setStmtSequencesRetriever(_ssr);
345     
346     		addActivePart(cgBasedPreProcessCtrl.getActivePart());
347     
348     		// create basic block graph manager
349     		bbgMgr = new BasicBlockGraphMgr();
350     		bbgMgr.setStmtGraphFactory(getStmtGraphFactory());
351     		_ssr.setBbgFactory(bbgMgr);
352     		// create the thread graph.
353     		threadGraph = new ThreadGraph(callGraph, new CFGAnalysis(callGraph, bbgMgr), pairMgr);
354     		// create equivalence class-based escape analysis.
355     		ecba = new EquivalenceClassBasedEscapeAnalysis(callGraph, null, bbgMgr);
356     		// create monitor analysis
357     		monitorInfo = new MonitorAnalysis();
358     		// create safe lock analysis
359     		safelockAnalysis = new SafeLockAnalysis();
360     		// create alias use def analysis
361     		aliasUD = new AliasedUseDefInfov2(ofa, callGraph, threadGraph, bbgMgr, pairMgr);
362     		// create static field use def analysis
363     		staticFieldUD = new StaticFieldUseDefInfo();
364     
365     		// set up data required for dependency analyses.
366     		info.put(ICallGraphInfo.ID, callGraph);
367     		info.put(IThreadGraphInfo.ID, threadGraph);
368     		info.put(IEnvironment.ID, ofa.getEnvironment());
369     		info.put(IUseDefInfo.ALIASED_USE_DEF_ID, aliasUD);
370     		info.put(IUseDefInfo.GLOBAL_USE_DEF_ID, staticFieldUD);
371     		info.put(PairManager.ID, pairMgr);
372     		info.put(IValueAnalyzer.ID, ofa);
373     		info.put(IEscapeInfo.ID, ecba.getEscapeInfo());
374     		info.put(IMonitorInfo.ID, monitorInfo);
375     		info.put(SafeLockAnalysis.ID, safelockAnalysis);
376     
377     		// create the slicing engine.
378     		engine = new SlicingEngine();
379     		addActivePart(engine.getActivePart());
380     
381     		// create the <init> call to new expr mapper
382     		initMapper = new NewExpr2InitMapper();
383     
384     		addActivePart(activePart);
385     	}

422     	public boolean destringizeConfiguration(final String stringizedForm) {
423     		IBindingFactory _bindingFactory;
424     		IUnmarshallingContext _unmarshallingContext;
425     		boolean _result = false;
426     
427     		try {
428     			_bindingFactory = BindingDirectory.getFactory(CompositeToolConfiguration.class);
429     			_unmarshallingContext = _bindingFactory.createUnmarshallingContext();
430     		} catch (final JiBXException _e) {
431     			LOGGER.error("Error while setting up JiBX.  Aborting.", _e);
432     			throw new RuntimeException(_e);
433     		}
434     
435     		configurationInfo = null;
436     
437     		if (stringizedForm != null && stringizedForm.length() != 0) {
438     			try {
439     				final StringReader _reader = new StringReader(stringizedForm);
440     				configurationInfo = (IToolConfiguration) _unmarshallingContext.unmarshalDocument(_reader, null);
441     				_result = true;
442     			} catch (final JiBXException _e) {
443     				LOGGER.error("Error while unmarshalling Slicer configurationCollection. Recovering with new clean"
444     						+ " configuration.", _e);
445     				configurationInfo = null;
446     			}
447     		}
448     
449     		if (configurationInfo == null) {
450     			initialize();
451     		}
452     
453     		configurator = new CompositeToolConfigurator((CompositeToolConfiguration) configurationInfo,
454     				new SlicerConfigurator(), SlicerConfiguration.getFactory());
455     		return _result;
456     	}

458     	/**
459     	 * {@inheritDoc}
460     	 * <p>
461     	 * This implementation of the tool remembers the last phase in which it was stopped. This saved phase is used when the
462     	 * tool is executed again with a <code>null</code> valued <code>phaseParam</code>. If a non-null
463     	 * <code>phaseParam</code> is provided, the tool starts executing from the earliest of the saved phase and the given
464     	 * phase. <code>lastPhase</code> controls the last phase to have finished execution when the tools stops.
465     	 * </p>
466     	 */
467     	@Override public void execute(final Phase phaseParam, final Phase lastPhase) throws InterruptedException {
468     		if (LOGGER.isInfoEnabled()) {
469     			LOGGER.info("BEGIN: Execution of the slicer tool");
470     		}
471     
472     		Phase _ph = phase;
473     
474     		if (phaseParam != null && phaseParam.isEarlierThan(phase)) {
475     			phase = phaseParam.clone();
476     			_ph = phase;
477     		}
478     
479     		if (_ph.equalsMajor(Phase.STARTING_PHASE)) {
480     			fireToolProgressEvent("Performing low level analyses", _ph);
481     			lowLevelAnalysisPhase();
482     		}
483     
484     		if (phase.equals(lastPhase)) {
485     			return;
486     		}
487     
488     		movingToNextPhase();
489     
490     		final SlicerConfiguration _slicerConfig = (SlicerConfiguration) getActiveConfiguration();
491     		setCurrentConfiguration(_slicerConfig);
492     		_slicerConfig.setupForUse();
493     
494     		if (_ph.equalsMajor(DEPENDENCE_MAJOR_PHASE)) {
495     			fireToolProgressEvent("Performing dependence analyses", _ph);
496     			dependencyAnalysisPhase(_slicerConfig);
497     		}
498     
499     		if (phase.equals(lastPhase)) {
500     			return;
501     		}
502     
503     		movingToNextPhase();
504     
505     		if (_ph.equalsMajor(SLICE_MAJOR_PHASE)) {
506     			fireToolProgressEvent("Performing slicing", _ph);
507     			slicingPhase(_slicerConfig);
508     		}
509     		phase.finished();
510     
511     		if (LOGGER.isInfoEnabled()) {
512     			LOGGER.info("END: Execution of the slicer tool");
513     		}
514     	}

790     	/**
791     	 * Executes dependency analyses and monitor analysis.
792     	 * 
793     	 * @param slicerConfig provides the configuration.
794     	 * @pre slicerConfig != null
795     	 */
796     	private void dependencyAnalysisPhase(final SlicerConfiguration slicerConfig) {
797     		if (LOGGER.isInfoEnabled()) {
798     			LOGGER.info("BEGIN: dependence analyses phase");
799     		}
800     
801     		// perform dependency analyses
802     		// create dependency analyses controller
803     		final BasicBlockGraphMgr _b;
804     
805     		if (slicerConfig.isExplicitExceptionalExitSensitiveControlDependenceUsed()) {
806     			final ExceptionRaisingAnalysis _e = new ExceptionRaisingAnalysis(getStmtGraphFactory(), getCallGraph(),
807     					getSystem());
808     
809     			if (slicerConfig.areCommonUncheckedExceptionsConsidered()) {
810     				_e.setupForCommonUncheckedExceptions();
811     			}
812     			_e.hookup(cgBasedPreProcessCtrl);
813     			cgBasedPreProcessCtrl.process();
814     			_e.unhook(cgBasedPreProcessCtrl);
815     
816     			_b = new BasicBlockGraphMgr(_e);
817     			_b.setStmtGraphFactory(getStmtGraphFactory());
818     		} else {
819     			_b = getBasicBlockGraphManager();
820     		}
821     		daController = new AnalysesController(info, cgBasedPreProcessCtrl, _b);
822     		addActivePart(daController.getActivePart());
823     
824     		final Collection<IDependencyAnalysis<?, ?, ?, ?, ?, ?>> _controlDAs = slicerConfig
825     				.getDependenceAnalyses(IDependencyAnalysis.DependenceSort.CONTROL_DA);
826     		final Collection<IDependencyAnalysis<?, ?, ?, ?, ?, ?>> _deps = (Collection<IDependencyAnalysis<?, ?, ?, ?, ?, ?>>) MapUtils
827     				.getFromMap(info, IDependencyAnalysis.DependenceSort.CONTROL_DA,
828     						new ArrayList<IDependencyAnalysis<?, ?, ?, ?, ?, ?>>());
829     		_deps.addAll(_controlDAs);
830     
831     		// drive the analyses
832     		for (final Iterator<IDependencyAnalysis.DependenceSort> _i = slicerConfig.getIDsOfDAsToUse().iterator(); _i.hasNext();) {
833     			final IDependencyAnalysis.DependenceSort _id = _i.next();
834     			final Collection<IDependencyAnalysis<?, ?, ?, ?, ?, ?>> _c = slicerConfig.getDependenceAnalyses(_id);
835     			daController.addAnalyses(_id, _c);
836     		}
837     		daController.addAnalyses(IMonitorInfo.ID, Collections.singleton(monitorInfo));
838     		daController.addAnalyses(EquivalenceClassBasedEscapeAnalysis.ID, Collections.singleton(ecba));
839     
840     		if (slicerConfig.isSafeLockAnalysisUsedForReady()) {
841     			daController.addAnalyses(SafeLockAnalysis.ID, Collections.singleton(safelockAnalysis));
842     		}
843     		daController.initialize();
844     		daController.execute();
845     		removeActivePart(daController.getActivePart());
846     
847     		final String _deadlockCriteriaSelectionStrategy = slicerConfig.getDeadlockCriteriaSelectionStrategy();
848     
849     		if (!slicerConfig.getPropertyAware()
850     				&& !_deadlockCriteriaSelectionStrategy.equals(SlicerConfiguration.CONTEXT_SENSITIVE_ESCAPING_SYNC_CONSTRUCTS)) {
851     			ecba.flushSiteContexts();
852     		}
853     
854     		phase.nextMajorPhase();
855     
856     		if (LOGGER.isInfoEnabled()) {
857     			LOGGER.info("END: dependence analyses phase");
858     		}
859     	}

861     	/**
862     	 * Executes low level analyses.
863     	 * 
864     	 * @throws InterruptedException when the tool is interrupted when moving between phases.
865     	 */
866     	private void lowLevelAnalysisPhase() throws InterruptedException {
867     		if (LOGGER.isInfoEnabled()) {
868     			LOGGER.info("BEGIN: low level static analyses phase");
869     		}
870     
871     		fireToolProgressEvent("LOW LEVEL ANALYSES: Performing object flow analysis", phase);
872     
873     		phase.reset();
874     		// do the flow analyses
875     		ofa.reset();
876     		bbgMgr.reset();
877     		stmtGraphFactory.reset();
878     		ofa.analyze(system, rootMethods);
879     		phase.nextMinorPhase();
880     
881     		movingToNextPhase();
882     
883     		fireToolProgressEvent("LOW LEVEL ANALYSES: Constructing call graph", phase);
884     
885     		// process flow information into a more meaningful call graph
886     		final OFABasedCallInfoCollector _callGraphInfoCollector = new OFABasedCallInfoCollector();
887     		cgPreProcessCtrl.reset();
888     		_callGraphInfoCollector.hookup(cgPreProcessCtrl);
889     		cgPreProcessCtrl.process();
890     		_callGraphInfoCollector.unhook(cgPreProcessCtrl);
891     
892     		callGraph.reset();
893     		callGraph.createCallGraphInfo(_callGraphInfoCollector.getCallInfo());
894     		phase.nextMinorPhase();
895     
896     		if (LOGGER.isDebugEnabled()) {
897     			LOGGER.debug("Call Graph:\n" + callGraph);
898     		}
899     
900     		movingToNextPhase();
901     
902     		fireToolProgressEvent("LOW LEVEL ANALYSES: Constructing thread graph and misc.", phase);
903     
904     		// process flow information into a more meaningful thread graph. Also, initialize <init> call to new expression
905     		// mapper.
906     		threadGraph.reset();
907     		cgBasedPreProcessCtrl.reset();
908     		initMapper.hookup(cgBasedPreProcessCtrl);
909     		threadGraph.hookup(cgBasedPreProcessCtrl);
910     		cgBasedPreProcessCtrl.process();
911     		threadGraph.unhook(cgBasedPreProcessCtrl);
912     		initMapper.unhook(cgBasedPreProcessCtrl);
913     		phase.nextMinorPhase();
914     
915     		if (LOGGER.isDebugEnabled()) {
916     			LOGGER.debug("Thread Graph:\n" + threadGraph);
917     		}
918     
919     		movingToNextPhase();
920     
921     		fireToolProgressEvent("LOW LEVEL ANALYSES: Calculating inter-procedural use-def information", phase);
922     
923     		// process alias use-def analyses.
924     		cgBasedPreProcessCtrl.reset();
925     		aliasUD.reset();
926     		aliasUD.hookup(cgBasedPreProcessCtrl);
927     		staticFieldUD.reset();
928     		staticFieldUD.hookup(cgBasedPreProcessCtrl);
929     		cgBasedPreProcessCtrl.process();
930     		aliasUD.unhook(cgBasedPreProcessCtrl);
931     		staticFieldUD.unhook(cgBasedPreProcessCtrl);
932     
933     		phase.nextMajorPhase();
934     
935     		if (LOGGER.isInfoEnabled()) {
936     			LOGGER.info("END: low level static analyses phase");
937     		}
938     	}

952     	/**
953     	 * Executes the slicer.
954     	 * 
955     	 * @param slicerConfig provides the configuration.
956     	 * @pre slicerConfig != null
957     	 */
958     	private void slicingPhase(final SlicerConfiguration slicerConfig) {
959     		if (LOGGER.isInfoEnabled()) {
960     			LOGGER.info("BEGIN: slicing phase");
961     		}
962     
963     		// perform slicing
964     		engine.reset();
965     
966     		fireToolProgressEvent("SLICING: adding criteria", phase);
967     
968     		for (final Iterator<ISliceCriteriaGenerator<?, ?>> _i = criteriaGenerators.iterator(); _i.hasNext();) {
969     			final ISliceCriteriaGenerator<?, ?> _e = _i.next();
970     			criteria.addAll(_e.getCriteria(this));
971     		}
972     
973     		for (final Iterator<ISliceCriteriaGenerator<?, ?>> _j = slicerConfig.getSliceCriteriaGenerators().iterator(); _j
974     				.hasNext();) {
975     			final ISliceCriteriaGenerator<?, ?> _generator = _j.next();
976     			criteria.addAll(_generator.getCriteria(this));
977     		}
978     
979     		if (!criteria.isEmpty()) {
980     			fireToolProgressEvent("SLICING: Calculating the slice", phase);
981     
982     			// setup the slicing engine and slice
983     			engine.setCgi(callGraph);
984     			engine.setSliceType((SliceType) slicerConfig.getProperty(SlicerConfiguration.SLICE_TYPE));
985     			engine.setInitMapper(initMapper);
986     			engine.setBasicBlockGraphManager(bbgMgr);
987     			engine.setAnalysesControllerAndDependenciesToUse(daController, slicerConfig.getIDsOfDAsToUse());
988     			engine.setSliceCriteria(criteria);
989     			engine.setSliceScopeDefinition(sliceScopeDefinition);
990     			engine.setSystem(system);
991     
992     			if (slicerConfig.getPropertyAware()) {
993     				final Map<IDependencyAnalysis.DependenceSort, ICallingContextRetriever> _map = new HashMap<IDependencyAnalysis.DependenceSort, ICallingContextRetriever>();
994     				final int _callingContextLimit = slicerConfig.getCallingContextLimit();
995     				final ThreadEscapeInfoBasedCallingContextRetriever _t1 = new ThreadEscapeInfoBasedCallingContextRetrieverV2(
996     						_callingContextLimit, true, false);
997     				// new ThreadEscapeInfoBasedCallingContextRetriever(_callingContextLimit);
998     				_t1.setEscapeInfo(getEscapeInfo());
999     				_t1.setECBA(ecba);
1000     				_t1.setCallGraph(getCallGraph());
1001     				_map.put(IDependencyAnalysis.DependenceSort.READY_DA, _t1);
1002     
1003     				final ThreadEscapeInfoBasedCallingContextRetriever _t2 = new ThreadEscapeInfoBasedCallingContextRetrieverV2(
1004     						_callingContextLimit, false, true);
1005     				// new ThreadEscapeInfoBasedCallingContextRetriever(_callingContextLimit);
1006     				_t2.setEscapeInfo(getEscapeInfo());
1007     				_t2.setECBA(ecba);
1008     				_t2.setCallGraph(getCallGraph());
1009     				_map.put(IDependencyAnalysis.DependenceSort.INTERFERENCE_DA, _t2);
1010     
1011     				final DataAliasBasedCallingContextRetrieverV3 _t3 = new DataAliasBasedCallingContextRetrieverV3(
1012     						_callingContextLimit);
1013     				_t3.setCallGraph(getCallGraph());
1014     				_t3.setThreadGraph(threadGraph);
1015     				_t3.setCfgAnalysis(new CFGAnalysis(getCallGraph(), getBasicBlockGraphManager()));
1016     				_t3.setECBA(ecba); // in case _t3 is of type DataAliasBasedCallingContextRetriever
1017     				_map.put(IDependencyAnalysis.DependenceSort.REFERENCE_BASED_DATA_DA, _t3);
1018     				engine.setDepID2ContextRetrieverMapping(_map);
1019     			}
1020     			engine.initialize();
1021     			engine.slice();
1022     			phase.nextMinorPhase();
1023     
1024     			fireToolProgressEvent("SLICING: Post processing the slice", phase);
1025     			// post process the slice as required
1026     			postProcessSlice();
1027     
1028     			phase.nextMajorPhase();
1029     		} else {
1030     			if (LOGGER.isWarnEnabled()) {
1031     				LOGGER.warn("No slicing criteria were specified. Hence, no slicing was done.\nIf \"slice for deadlock\" was "
1032     						+ "selected then the system did not have any synchronized methods or synchronized blocks.");
1033     			}
1034     		}
1035     
1036     		if (LOGGER.isInfoEnabled()) {
1037     			LOGGER.info("END: slicing phase");
1038     		}
1039     	}
