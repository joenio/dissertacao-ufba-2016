67     /**
68      * This is a command line interface to exercise side effect and escape information analysis.

70      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>

74     public class EscapeAndReadWriteCLI
75     		extends SootBasedDriver {
76     
77     	/**
78     	 * The logger used by instances of this class to log messages.
79     	 */
80     	private static final Logger LOGGER = LoggerFactory.getLogger(EscapeAndReadWriteCLI.class);
81     
82     	/**
83     	 * The entry point to this class.
84     	 * 
85     	 * @param args command line arguments.
86     	 * @throws RuntimeException when escape information and side-effect information calculation fails.
87     	 */
88     	public static void main(final String[] args) {
89     		final Options _options = new Options();
90     		Option _option = new Option("h", "help", false, "Display message.");
91     		_option.setOptionalArg(false);
92     		_options.addOption(_option);
93     		_option = new Option("p", "soot-classpath", false, "Prepend this to soot class path.");
94     		_option.setArgs(1);
95     		_option.setArgName("classpath");
96     		_option.setOptionalArg(false);
97     		_options.addOption(_option);
98     		_option = new Option("S", "scope", true, "The scope that should be analyzed.");
99     		_option.setArgs(1);
100     		_option.setArgName("scope");
101     		_option.setRequired(false);
102     		_options.addOption(_option);
103     
104     		final CommandLineParser _parser = new GnuParser();
105     
106     		try {
107     			final CommandLine _cl = _parser.parse(_options, args);
108     
109     			if (_cl.hasOption("h")) {
110     				final String _cmdLineSyn = "java " + EscapeAndReadWriteCLI.class.getName() + " <options> <classnames>";
111     				(new HelpFormatter()).printHelp(_cmdLineSyn, _options);
112     				System.exit(1);
113     			}
114     
115     			if (_cl.getArgList().isEmpty()) {
116     				throw new MissingArgumentException("Please specify atleast one class.");
117     			}
118     
119     			final EscapeAndReadWriteCLI _cli = new EscapeAndReadWriteCLI();
120     
121     			if (_cl.hasOption('p')) {
122     				_cli.addToSootClassPath(_cl.getOptionValue('p'));
123     			}
124     
125     			if (_cl.hasOption('S')) {
126     				_cli.setScopeSpecFile(_cl.getOptionValue('S'));
127     			}
128     			_cli.setClassNames(_cl.getArgList());
129     			_cli.<ITokens> execute();
130     		} catch (final ParseException _e) {
131     			LOGGER.error("Error while parsing command line.", _e);
132     			System.out.println("Error while parsing command line." + _e);
133     			final String _cmdLineSyn = "java " + EscapeAndReadWriteCLI.class.getName() + " <options> <classnames>";
134     			(new HelpFormatter()).printHelp(_cmdLineSyn, "Options are:", _options, "");
135     		} catch (final Throwable _e) {
136     			LOGGER.error("Beyond our control. May day! May day!", _e);
137     			throw new RuntimeException(_e);
138     		}
139     	}
140     
141     	/**
142     	 * This contains the driver logic.
143     	 * 
144     	 * @param <T> dummy type parameter.
145     	 */
146     	private <T extends ITokens<T, Value>> void execute() {
147     		setInfoLogger(LOGGER);
148     
149     		final String _tagName = "SideEffect:FA";
150     		final IValueAnalyzer<Value> _aa = OFAnalyzer.getFSOSAnalyzer(_tagName, TokenUtil
151     				.<T, Value, Type> getTokenManager(new SootValueTypeManager()), getStmtGraphFactory());
152     		final ValueAnalyzerBasedProcessingController _pc = new ValueAnalyzerBasedProcessingController();
153     		final Collection<IProcessor> _processors = new ArrayList<IProcessor>();
154     		final PairManager _pairManager = new PairManager(false, true);
155     		final CallGraphInfo _cgi = new CallGraphInfo(_pairManager);
156     		final OFABasedCallInfoCollector _callGraphInfoCollector = new OFABasedCallInfoCollector();
157     		final IThreadGraphInfo _tgi = new ThreadGraph(_cgi, new CFGAnalysis(_cgi, getBbm()), _pairManager);
158     		final ValueAnalyzerBasedProcessingController _cgipc = new ValueAnalyzerBasedProcessingController();
159     		final OneAllStmtSequenceRetriever _ssr = new OneAllStmtSequenceRetriever();
160     
161     		_ssr.setStmtGraphFactory(getStmtGraphFactory());
162     
163     		_pc.setStmtSequencesRetriever(_ssr);
164     		_pc.setAnalyzer(_aa);
165     		_pc.setProcessingFilter(new TagBasedProcessingFilter(_tagName));
166     
167     		_cgipc.setAnalyzer(_aa);
168     		_cgipc.setProcessingFilter(new CGBasedProcessingFilter(_cgi));
169     		_cgipc.setStmtSequencesRetriever(_ssr);
170     
171     		final Map _info = new HashMap();
172     		_info.put(ICallGraphInfo.ID, _cgi);
173     		_info.put(IThreadGraphInfo.ID, _tgi);
174     		_info.put(PairManager.ID, _pairManager);
175     		_info.put(IEnvironment.ID, _aa.getEnvironment());
176     		_info.put(IValueAnalyzer.ID, _aa);
177     
178     		initialize();
179     		_aa.analyze(getEnvironment(), getRootMethods());
180     
181     		_processors.clear();
182     		_processors.add(_callGraphInfoCollector);
183     		_pc.reset();
184     		_pc.driveProcessors(_processors);
185     		_cgi.createCallGraphInfo(_callGraphInfoCollector.getCallInfo());
186     		writeInfo("CALL GRAPH:\n" + _cgi.toString());
187     
188     		_processors.clear();
189     		((ThreadGraph) _tgi).reset();
190     		_processors.add((IProcessor) _tgi);
191     		_cgipc.reset();
192     		_cgipc.driveProcessors(_processors);
193     		writeInfo("THREAD GRAPH:\n" + ((ThreadGraph) _tgi).toString());
194     		final EquivalenceClassBasedEscapeAnalysis _ecba = new EquivalenceClassBasedEscapeAnalysis(_cgi, _tgi, getBbm());
195     		final IReadWriteInfo _rwInfo = _ecba.getReadWriteInfo();
196     		final IEscapeInfo _escapeInfo = _ecba.getEscapeInfo();
197     		final AnalysesController _ac = new AnalysesController(_info, _cgipc, getBbm());
198     		_ac.addAnalyses(EquivalenceClassBasedEscapeAnalysis.ID, Collections.singleton(_ecba));
199     		_ac.initialize();
200     		_ac.execute();
201     		writeInfo("END: Escape analysis");
202     
203     		System.out.println("ReadWrite-Effect and Escape Information:");
204     		final String[] _emptyStringArray = new String[0];
205     
206     		for (final Iterator<SootMethod> _i = _cgi.getReachableMethods().iterator(); _i.hasNext();) {
207     			final SootMethod _sm = _i.next();
208     			System.out.println("Method: " + _sm.getSignature());
209     			System.out.println("\tsealed: " + _ecba.isMethodSealed(_sm) + ", atomic: " + _ecba.isMethodAtomic(_sm));
210     			if (!_sm.isStatic()) {
211     				System.out.println("\tthis:");
212     				System.out.println("\t\tread =  " + _rwInfo.isThisBasedAccessPathRead(_sm, _emptyStringArray, true));
213     				System.out.println("\t\twritten =  " + _rwInfo.isThisBasedAccessPathWritten(_sm, _emptyStringArray, true));
214     				System.out.println("\t\tescapes = " + _escapeInfo.thisEscapes(_sm));
215     				System.out.println("\t\tmulti-thread RW = " + _escapeInfo.thisFieldAccessShared(_sm, IEscapeInfo.READ_WRITE_SHARED_ACCESS));
216     				System.out.println("\t\tmulti-thread LU = " + _escapeInfo.thisLockUnlockShared(_sm));
217     				System.out.println("\t\tmulti-thread WN = " + _escapeInfo.thisWaitNotifyShared(_sm));
218     				System.out.println("\t\tfield reading threads = " + _escapeInfo.getReadingThreadsOfThis(_sm));
219     				System.out.println("\t\tfield writing threads = " + _escapeInfo.getWritingThreadsOfThis(_sm));
220     
221     				for (final Iterator<SootField> _j = _sm.getDeclaringClass().getFields().iterator(); _j.hasNext();) {
222     					final String[] _accessPath = {_j.next().getSignature()};
223     					System.out.println("\t\t\t" + _accessPath[0] + ": ["
224     							+ _rwInfo.isThisBasedAccessPathRead(_sm, _accessPath, true) + ", "
225     							+ _rwInfo.isThisBasedAccessPathWritten(_sm, _accessPath, true) + "]");
226     				}
227     			}
228     
229     			for (int _j = 0; _j < _sm.getParameterCount(); _j++) {
230     				System.out.println("\tParam" + (_j + 1) + "[" + _sm.getParameterType(_j) + "]:");
231     				System.out.println("\t\tread = " + _rwInfo.isParameterBasedAccessPathRead(_sm, _j, _emptyStringArray, true));
232     				System.out.println("\t\twritten = "
233     						+ _rwInfo.isParameterBasedAccessPathWritten(_sm, _j, _emptyStringArray, true));
234     				System.out.println("\t\tfield reading threads: " + _escapeInfo.getReadingThreadsOf(_j, _sm));
235     				System.out.println("\t\tfield writing threads: " + _escapeInfo.getWritingThreadsOf(_j, _sm));
236     			}
237     
238     			if (_sm.hasActiveBody()) {
239     				final Body _body = _sm.getActiveBody();
240     
241     				for (final Iterator<Local> _j = _body.getLocals().iterator(); _j.hasNext();) {
242     					final Local _local = _j.next();
243     					System.out.println("\tLocal " + _local.getName() + "[" + _local.getType() + "] : ");
244     					System.out.println("\t\tescapes = " + _escapeInfo.escapes(_local, _sm));
245     					System.out.println("\t\tshared RW = " + _escapeInfo.fieldAccessShared(_local, _sm, IEscapeInfo.READ_WRITE_SHARED_ACCESS));
246     					System.out.println("\t\tshared WW = " + _escapeInfo.fieldAccessShared(_local, _sm, IEscapeInfo.WRITE_WRITE_SHARED_ACCESS));
247     					System.out.println("\t\tlocking = " + _escapeInfo.lockUnlockShared(_local, _sm));
248     					System.out.println("\t\twaitNotify  = " + _escapeInfo.waitNotifyShared(_local, _sm));
249     					System.out.println("\t\tfield reading threads: " + _escapeInfo.getReadingThreadsOf(_local, _sm));
250     					System.out.println("\t\tfield writing threads: " + _escapeInfo.getWritingThreadsOf(_local, _sm));
251     				}
252     			}
253     		}
254     	}
255     }
