17     package edu.ksu.cis.indus.slicer;
18     
19     import edu.ksu.cis.indus.common.CustomToStringStyle;
20     
21     import org.apache.commons.lang.builder.EqualsBuilder;
22     import org.apache.commons.lang.builder.HashCodeBuilder;
23     import org.apache.commons.lang.builder.ToStringBuilder;
24     
25     import soot.SootMethod;
26     import soot.ValueBox;
27     
28     import soot.jimple.Stmt;
29     
30     /**
31      * This class represents expression-level slice criterion. This class supports object pooling.
32      *
33      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
34      * @author $Author: rvprasad $
35      * @version $Revision: 1.9 $
36      */
37     class ExprLevelSliceCriterion
38     		extends AbstractSliceCriterion {
39     
40     	/**
41     	 * The statement associated with this criterion.
42     	 */
43     	protected Stmt stmt;
44     
45     	/**
46     	 * The expression associated with this criterion.
47     	 */
48     	protected ValueBox expr;
49     
50     	/**
51     	 * @see java.lang.Object#equals(Object)
52     	 */
53     	@Override public boolean equals(final Object object) {
54     		if (object == this) {
55     			return true;
56     		}
57     
58     		if (!(object instanceof ExprLevelSliceCriterion)) {
59     			return false;
60     		}
61     
62     		final ExprLevelSliceCriterion _rhs = (ExprLevelSliceCriterion) object;
63     		return new EqualsBuilder().appendSuper(super.equals(object)).append(this.expr, _rhs.expr)
64     				.append(this.stmt, _rhs.stmt).isEquals();
65     	}
66     
67     	/**
68     	 * @see java.lang.Object#hashCode()
69     	 */
70     	@Override public int hashCode() {
71     		return new HashCodeBuilder(17, 37).appendSuper(super.hashCode()).append(stmt).append(expr).toHashCode();
72     	}
73     
74     	/**
75     	 * @see java.lang.Object#toString()
76     	 */
77     	@Override public String toString() {
78     		return new ToStringBuilder(this, CustomToStringStyle.HASHCODE_AT_END_STYLE).appendSuper(super.toString()).append(
79     				"stmt", this.stmt).append("expr", this.expr).toString();
80     	}
81     
82     	/**
83     	 * {@inheritDoc}
84     	 *
85     	 * @return the expression(<code>ValueBox</code>) associated with criterion.
86     	 * @post result != null and result.oclIsKindOf(ValueBox)
87     	 * @see AbstractSliceCriterion#getCriterion()
88     	 */
89     	@Override protected Object getCriterion() {
90     		return expr;
91     	}
92     
93     	/**
94     	 * Provides the statement in which the slice expression occurs.
95     	 *
96     	 * @return the statement in which the slice expression occurs.
97     	 * @post result != null
98     	 */
99     	protected final Stmt getOccurringStmt() {
100     		return stmt;
101     	}
102     
103     	/**
104     	 * Initializes this object.
105     	 *
106     	 * @param occurringMethod in which the criterion containing statement occurs.
107     	 * @param occurringStmt in which the criterion containing expression occurs.
108     	 * @param criterion is the slicing criterion.
109     	 * @pre expr != null and stmt != null and method != null
110     	 */
111     	void initialize(final SootMethod occurringMethod, final Stmt occurringStmt, final ValueBox criterion) {
112     		super.initialize(occurringMethod);
113     		this.stmt = occurringStmt;
114     		this.expr = criterion;
115     	}
116     }
