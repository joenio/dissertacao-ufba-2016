17     package edu.ksu.cis.indus.common;
18     
19     import edu.ksu.cis.indus.annotations.Empty;
20     import edu.ksu.cis.indus.annotations.Functional;
21     
22     import java.util.Comparator;
23     
24     /**
25      * This class compares given objects based on the return value of <code>toString()</code> method.
26      * 
27      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
28      * @author $Author: rvprasad $
29      * @version $Revision: 1.11 $ $Date: 2007/02/10 19:09:03 $
30      * @param <T> the type of the objects that this comparator can compare.
31      */
32     public final class ToStringBasedComparator<T>
33     		implements Comparator<T> {
34     
35     	// / CLOVER:ON
36     
37     	/**
38     	 * Creates an instance of this class.
39     	 */
40     	@Empty private ToStringBasedComparator() {
41     		// does nothing.
42     	}
43     
44     	// / CLOVER:ON
45     
46     	/**
47     	 * A single of this class.
48     	 * 
49     	 * @param <T1> is the type of object that the return comparator will handle.
50     	 * @return the comparator
51     	 * @post result != null
52     	 */
53     	public static <T1> Comparator<T1> getComparator() {
54     		return new ToStringBasedComparator<T1>();
55     	}
56     
57     	/**
58     	 * {@inheritDoc}
59     	 */
60     	@Functional public int compare(final T o1, final T o2) {
61     		return String.valueOf(o1).compareTo(String.valueOf(o2));
62     	}
63     }
