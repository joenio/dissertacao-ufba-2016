17     package edu.ksu.cis.indus.processing;
18     
19     import edu.ksu.cis.indus.common.soot.BasicBlockGraphMgr;
20     import edu.ksu.cis.indus.common.soot.IStmtGraphFactory;
21     
22     /**
23      * This implementation provides infrastructure to store information pertaining to higher-level structures (stmt graph or basic
24      * block graph) based on which concrete implementation can generate statement sequences. <i>The user should call atleast one
25      * of <code>setBbgFactory()</code> or <code>setStmtGraphFactory()</code> with a non-null argument before using an instance
26      * of this class.</i>
27      * 
28      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
29      * @author $Author: rvprasad $
30      * @version $Revision: 1.4 $ $Date: 2007/02/10 19:08:37 $
31      */
32     public abstract class AbstractStmtSequenceRetriever
33     		implements IStmtSequencesRetriever {
34     
35     	/**
36     	 * The basic block graph factory to use.
37     	 */
38     	private BasicBlockGraphMgr bbgFactory;
39     
40     	/**
41     	 * The statement graph factory to use.
42     	 */
43     	private IStmtGraphFactory<?> stmtGraphFactory;
44     
45     	/**
46     	 * Sets the value of <code>bbgMgr</code>.
47     	 * 
48     	 * @param factory the new value of <code>bbgMgr</code>.
49     	 */
50     	public final void setBbgFactory(final BasicBlockGraphMgr factory) {
51     		this.bbgFactory = factory;
52     	}
53     
54     	/**
55     	 * Sets the value of <code>stmtGraphFactory</code>.
56     	 * 
57     	 * @param factory the new value of <code>stmtGraphFactory</code>.
58     	 */
59     	public final void setStmtGraphFactory(final IStmtGraphFactory<?> factory) {
60     		this.stmtGraphFactory = factory;
61     	}
62     
63     	/**
64     	 * Retrieves the factory that provides basic block graphs.
65     	 * 
66     	 * @return basic block graph factory.
67     	 */
68     	protected final BasicBlockGraphMgr getBbgFactory() {
69     		return bbgFactory;
70     	}
71     
72     	/**
73     	 * Retrieves the factory that provides statement graph factory.
74     	 * 
75     	 * @return statement graph factory.
76     	 */
77     	protected final IStmtGraphFactory<?> getStmtGraphFactory() {
78     		return stmtGraphFactory;
79     	}
80     }
