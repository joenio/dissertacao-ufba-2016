17     package edu.ksu.cis.indus.common.fa;
18     
19     import edu.ksu.cis.indus.annotations.Functional;
20     import edu.ksu.cis.indus.annotations.Immutable;
21     import edu.ksu.cis.indus.annotations.NonNull;
22     import edu.ksu.cis.indus.common.collections.CollectionUtils;
23     import edu.ksu.cis.indus.common.collections.ITransformer;
24     import edu.ksu.cis.indus.common.fa.ITransitionLabel.IEpsilonLabelFactory;
25     import edu.ksu.cis.indus.common.graph.SimpleEdgeGraph;
26     import edu.ksu.cis.indus.common.graph.SimpleEdgeLabelledNode;
27     
28     import java.util.Collection;
29     import java.util.Collections;
30     import java.util.HashSet;
31     
32     /**
33      * This is an implementation of non-deterministic finite automaton.
34      * 
35      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
36      * @author $Author: rvprasad $
37      * @version $Revision: 1.7 $ $Date: 2007/02/10 19:08:39 $ *
38      * @param <S> the type of the implementation of this interface.
39      * @param <L> the type of the implementation of this interface.
40      */
41     public class NFA<S extends IState<S>, L extends ITransitionLabel<L>>
42     		implements IAutomaton<S, L> {
43     
44     	/**
45     	 * The epsilon label factory.
46     	 */
47     	@NonNull protected final IEpsilonLabelFactory<L> epsilonFactory;
48     
49     	/**
50     	 * The current state of the NFA. This is <code>null</code> if the automaton is not running.
51     	 */
52     	private S currentState;
53     
54     	/**
55     	 * The final states of this automaton.
56     	 */
57     	private final Collection<S> finalStates = new HashSet<S>();
58     
59     	/**
60     	 * The object extractor.
61     	 */
62     	private final ITransformer<SimpleEdgeLabelledNode<S>, S> objectExtractor;
63     
64     	/**
65     	 * This graph is used to represent the shape of the automaton.
66     	 */
67     	private final SimpleEdgeGraph<S> seg = new SimpleEdgeGraph<S>();
68     
69     	/**
70     	 * The start state of the the automaton.
71     	 */
72     	private S startState;
73     
74     	/**
75     	 * Creates an instance of this class.
76     	 * 
77     	 * @param eFactory the epsilon factory.
78     	 */
79     	public NFA(@NonNull @Immutable final ITransitionLabel.IEpsilonLabelFactory<L> eFactory) {
80     		super();
81     		epsilonFactory = eFactory;
82     		objectExtractor = new ITransformer<SimpleEdgeLabelledNode<S>, S>() {
83     
84     			@Functional public S transform(final SimpleEdgeLabelledNode<S> input) {
85     				return input.getObject();
86     			}
87     		};
88     	}
89     
90     	/**
91     	 * Adds a final state to the automaton.
92     	 * 
93     	 * @param state to be added as a final state.
94     	 */
95     	public void addFinalState(@NonNull @Immutable final S state) {
96     		assert state != null;
97     
98     		seg.getNode(state);
99     		finalStates.add(state);
100     	}
101     
102     	/**
103     	 * Adds a transition labelled with the given label from the given source state to the destination state.
104     	 * 
105     	 * @param src is the source state.
106     	 * @param label is the label of the transition.
107     	 * @param dest is the destination state.
108     	 * @throws IllegalStateException if the automaton is altered while it's running, i.e. <code>start()</code> has been
109     	 *             called but <code>stop()</code> has not been called.
110     	 */
111     	public void addLabelledTransitionFromTo(@NonNull @Immutable final S src, @NonNull @Immutable final L label,
112     			@NonNull @Immutable final S dest) {
113     		if (currentState != null) {
114     			throw new IllegalStateException(
115     					"The automata should be altered when it is not not running (prior to starting it or"
116     							+ "after it is stopped).");
117     		}
118     		seg.addEdgeFromTo(seg.getNode(src), label, seg.getNode(dest));
119     	}
120     
121     	/**
122     	 * {@inheritDoc}
123     	 */
124     	public boolean canPerformTransition(@NonNull @Immutable final L label) {
125     		final SimpleEdgeLabelledNode<S> _node = seg.queryNode(currentState);
126     		final boolean _result;
127     
128     		if (_node != null) {
129     			_result = _node.hasOutgoingEdgeLabelled(label);
130     		} else {
131     			_result = false;
132     		}
133     		return _result;
134     	}
135     
136     	/**
137     	 * {@inheritDoc}
138     	 */
139     	@Override @SuppressWarnings("unchecked") public NFA<S, L> clone() {
140     		try {
141     			return (NFA) super.clone();
142     		} catch (final CloneNotSupportedException _e) {
143     			throw new RuntimeException(_e);
144     		}
145     	}
146     
147     	/**
148     	 * {@inheritDoc}
149     	 */
150     	@Functional public S getCurrentState() {
151     		return currentState;
152     	}
153     
154     	/**
155     	 * {@inheritDoc}
156     	 */
157     	@Functional @NonNull public Collection<S> getFinalStates() {
158     		return finalStates;
159     	}
160     
161     	/**
162     	 * Retrieves the states that can be reached by taking a transition labelled with the given label from the given state of
163     	 * the automaton.
164     	 * 
165     	 * @param state of interest.
166     	 * @param label of the outgoing transitions from <code>state</code>.
167     	 * @return the collection of states.
168     	 */
169     	@NonNull public Collection<S> getResultingStates(@NonNull @Immutable final S state, @NonNull @Immutable final L label) {
170     		assert state != null && label != null;
171     
172     		final SimpleEdgeLabelledNode<S> _node = seg.queryNode(state);
173     		final Collection<S> _result;
174     
175     		if (_node != null) {
176     			final Collection<SimpleEdgeLabelledNode<S>> _dests = _node.getSuccsViaEdgesLabelled(label);
177     			_result = CollectionUtils.collect(_dests, objectExtractor);
178     		} else {
179     			_result = Collections.emptySet();
180     		}
181     		return _result;
182     	}
183     
184     	/**
185     	 * Retrieves the start state of the automaton.
186     	 * 
187     	 * @return the start state.
188     	 */
189     	@Functional @NonNull public S getStartState() {
190     		return startState;
191     	}
192     
193     	/**
194     	 * {@inheritDoc}
195     	 */
196     	@Functional public boolean isDeterministic() {
197     		return false;
198     	}
199     
200     	/**
201     	 * {@inheritDoc}
202     	 */
203     	@Functional public boolean isInFinalState() {
204     		return finalStates.contains(currentState);
205     	}
206     
207     	/**
208     	 * {@inheritDoc}
209     	 */
210     	public void performTransitionOn(@NonNull @Immutable final L label) {
211     		final SimpleEdgeLabelledNode<S> _node = seg.queryNode(currentState);
212     
213     		if (_node != null) {
214     			final Collection<SimpleEdgeLabelledNode<S>> _dests = _node.getSuccsViaEdgesLabelled(label);
215     			currentState = _dests.iterator().next().getObject();
216     		} else {
217     			throw new IllegalStateException("There are no transition with the label " + label + " from the current state.");
218     		}
219     	}
220     
221     	/**
222     	 * Removes the transition labelled with the given label from the given source state to the destination state.
223     	 * 
224     	 * @param src is the source state.
225     	 * @param label of the transition to be removed.
226     	 * @param dest is the destination state.
227     	 * @return <code>true</code> if such a transition existed and it was removed; <code>false</code>, otherwise.
228     	 * @throws IllegalStateException if the automaton is altered while it's running, i.e. <code>start()</code> has been
229     	 *             called but <code>stop()</code> has not been called.
230     	 */
231     	public boolean removeLabelledTransitionFromTo(@NonNull @Immutable final S src, @NonNull @Immutable final L label,
232     			@NonNull @Immutable final S dest) {
233     		if (currentState != null) {
234     			throw new IllegalStateException("The automata should be altered when it is not active (prior to starting it or"
235     					+ "after it is stopped).");
236     		}
237     		return seg.removeEdgeFromTo(seg.queryNode(src), label, seg.queryNode(dest));
238     	}
239     
240     	/**
241     	 * Sets the start state of the automaton.
242     	 * 
243     	 * @param state to be the start state.
244     	 */
245     	public void setStartState(@NonNull @Immutable final S state) {
246     		assert state != null;
247     
248     		seg.getNode(state);
249     		startState = state;
250     	}
251     
252     	/**
253     	 * {@inheritDoc}
254     	 */
255     	public void start() {
256     		currentState = startState;
257     	}
258     
259     	/**
260     	 * {@inheritDoc}
261     	 */
262     	public void stop() {
263     		currentState = null;
264     	}
265     }
