104     		super(statementSwitch, nodeConnector);

116     	 * Processes array access expressions. Current implementation processes the primary and connects a node associated with
117     	 * the primary to a <code>FGAccessNode</code> which monitors this access expressions for new values in the primary.
118     	 * 
119     	 * @param e the array access expressions.
120     	 * @pre e != null
121     	 */
122     	@Override public void caseArrayRef(final ArrayRef e) {
123     		process(e.getBaseBox());
124     
125     		final OFAFGNode<T> _baseNode = getFlowNode();
126     		final OFAFGNode<T> _ast = method.getASTNode(e, context);
127     		MethodVariant.setFilterOfBasedOn(_ast, e.getType(), tokenMgr);
128     
129     		final ITokenProcessingWork<T> _work = new ArrayAccessExprWork<T>(method, context, _ast, connector, tokenMgr
130     				.getNewTokenSet());
131     		final OFAFGNode<T> _temp = new FGAccessNode<T>(_work, fa, tokenMgr);
132     		MethodVariant.setFilterOfBasedOn(_temp, e.getBase().getType(), tokenMgr);
133     		_baseNode.addSucc(_temp);
134     		process(e.getIndexBox());
135     		setFlowNode(_ast);
136     	}

152     			MethodVariant.setFilterOfBasedOn(_cast, e.getType(), tokenMgr);
153     			_base.addSucc(_cast);

155     		}

269     	 * @pre e != null

271     	@Override public void caseNewMultiArrayExpr(final NewMultiArrayExpr e) {
272     		final ArrayType _arrayType = e.getBaseType();
273     		final Type _baseType = _arrayType.baseType;
274     
275     		Object _temp = null;
276     
277     		final boolean _flag = context instanceof AllocationContext;
278     
279     		if (_flag) {
280     			_temp = ((AllocationContext) context).setAllocationSite(e);
281     		}
282     
283     		for (int _i = _arrayType.numDimensions, _sizes = e.getSizeCount(); _i > 0 && _sizes > 0; _i--, _sizes--) {
284     			final ArrayType _aType = ArrayType.v(_baseType, _i);
285     			final ValuedVariant<OFAFGNode<T>> _array = fa.getArrayVariant(_aType, context);
286     			process(e.getSizeBox(_sizes - 1));
287     			_array.getFGNode().injectValue(valueRetriever.getValue(e));
288     		}
289     
290     		if (_flag) {
291     			((AllocationContext) context).setAllocationSite(_temp);
292     		}
293     
294     		final OFAFGNode<T> _ast = method.getASTNode(e, context);
295     		MethodVariant.setFilterOfBasedOn(_ast, e.getType(), tokenMgr);
296     		_ast.injectValue(valueRetriever.getValue(e));
297     		setFlowNode(_ast);
298     	}

399     		final Value _v = (Value) o;

402     			final BinopExpr _temp = (BinopExpr) _v;
403     			process(_temp.getOp1Box());

425     	/**
426     	 * Processes the invoke expressions that require resolution by creating nodes to various data components present at the
427     	 * call-site and making them available to be connected when new method implementations are plugged in.
428     	 * 
429     	 * @param e the invoke expression to be processed.
430     	 * @pre e != null
431     	 */
432     	protected void processInstanceInvokeExpr(final InstanceInvokeExpr e) {
433     		if (LOGGER.isDebugEnabled()) {
434     			LOGGER.debug("BEGIN: processing " + e);
435     		}
436     		fa.processClass(e.getMethod().getDeclaringClass());
437     		process(e.getBaseBox());
438     
439     		final OFAFGNode<T> _receiverNode = getFlowNode();
440     		MethodVariant.setFilterOfBasedOn(_receiverNode, e.getBase().getType(), tokenMgr);
441     
442     		for (int _i = 0; _i < e.getArgCount(); _i++) {
443     			process(e.getArgBox(_i));
444     		}
445     
446     		final InvocationVariant<OFAFGNode<T>> _iv = (InvocationVariant) method.getASTVariant(e, context);
447     
448     		if (Util.isReferenceType(e.getType())) {
449     			final OFAFGNode<T> _ast = _iv.getFGNode();
450     			MethodVariant.setFilterOfBasedOn(_ast, e.getType(), tokenMgr);
451     			setFlowNode(_ast);
452     		} else {
453     			setFlowNode(null);
454     		}
455     
456     		final ITokenProcessingWork<T> _work = new InvokeExprWork<T>(method, context, tokenMgr.getNewTokenSet());
457     		final FGAccessNode<T> _baseNode = new FGAccessNode<T>(_work, fa, tokenMgr);
458     		MethodVariant.setFilterOfBasedOn(_baseNode, e.getBase().getType(), tokenMgr);
459     		_receiverNode.addSucc(_baseNode);
460     
461     		if (LOGGER.isDebugEnabled()) {
462     			LOGGER.debug("END: processed " + e);
463     		}
464     	}

466     	/**
467     	 * Retrieves the canonical string constant that represents the given string constant.
468     	 * 
469     	 * @param e is the string constant that needs to be canonicalized.
470     	 * @return the canonical string constant.
471     	 */
472     	private StringConstant getCanonicalStringConstant(final StringConstant e) {
473     		if (STR_TO_CONST.containsKey(e.value)) {
474     			return STR_TO_CONST.get(e.value);
475     		}
476     		STR_TO_CONST.put(e.value, e);
477     		return e;
478     	}

480     	/**
481     	 * Processes the invoke expressions that do not require resolution by creating nodes to various data components present at
482     	 * the call-site and making them available to be connected when new method implementations are plugged in.
483     	 * 
484     	 * @param e the invoke expression to be processed.
485     	 * @pre e != null
486     	 */
487     	private void processInvokedMethod(final InvokeExpr e) {
488     		if (LOGGER.isDebugEnabled()) {
489     			LOGGER.debug("BEGIN: processing " + e);
490     		}
491     
492     		final IMethodVariant<OFAFGNode<T>> _callee = fa.getMethodVariant(e.getMethod(), context);
493     
494     		if (e instanceof SpecialInvokeExpr) {
495     			final SpecialInvokeExpr _expr = (SpecialInvokeExpr) e;
496     			final OFAFGNode<T> _thisNode = _callee.queryThisNode();
497     			process(_expr.getBaseBox());
498     
499     			final OFAFGNode<T> _thisArgNode = getFlowNode();
500     			MethodVariant.setFilterOfBasedOn(_thisArgNode, _expr.getBase().getType(), tokenMgr);
501     
502     			_thisArgNode.addSucc(_thisNode);
503     		}
504     
505     		for (int _i = 0; _i < e.getArgCount(); _i++) {
506     			if (Util.isReferenceType(e.getArg(_i).getType())) {
507     				process(e.getArgBox(_i));
508     
509     				final OFAFGNode<T> _argNode = getFlowNode();
510     				MethodVariant.setFilterOfBasedOn(_argNode, e.getArg(_i).getType(), tokenMgr);
511     
512     				_argNode.addSucc(_callee.queryParameterNode(_i));
513     			}
514     		}
515     
516     		final InvocationVariant<OFAFGNode<T>> _iv = (InvocationVariant) method.getASTVariant(e, context);
517     		final OFAFGNode<T> _throwNode = _iv.getThrowNode();
518     		MethodVariant.setFilterOfBasedOn(_throwNode, fa.getClass("java.lang.Throwable").getType(), tokenMgr);
519     		_callee.queryThrownNode().addSucc(_throwNode);
520     
521     		if (Util.isReferenceType(e.getType())) {
522     			final OFAFGNode<T> _ast = _iv.getFGNode();
523     			MethodVariant.setFilterOfBasedOn(_ast, e.getType(), tokenMgr);
524     			_callee.queryReturnNode().addSucc(_ast);
525     			setFlowNode(_ast);
526     		} else {
527     			setFlowNode(null);
528     		}
529     
530     		if (LOGGER.isDebugEnabled()) {
531     			LOGGER.debug("END: processed " + e);
532     		}
533     	}
