17     package edu.ksu.cis.indus.common;
18     
19     import edu.ksu.cis.indus.annotations.Empty;
20     import edu.ksu.cis.indus.annotations.NonNull;
21     
22     import org.apache.commons.lang.builder.ToStringStyle;
23     
24     /**
25      * This class customizes the output of <code>toString()</code> by outputting the hashcode at the end instead of in the
26      * beginning.
27      * 
28      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
29      * @author $Author: rvprasad $
30      * @version $Revision: 1.9 $ $Date: 2007/02/10 19:09:03 $
31      */
32     public final class CustomToStringStyle
33     		extends ToStringStyle {
34     
35     	/**
36     	 * This is the singleton object of this class.
37     	 */
38     	public static final ToStringStyle HASHCODE_AT_END_STYLE = new CustomToStringStyle();
39     
40     	/**
41     	 * The serial version UID.
42     	 */
43     	static final long serialVersionUID = -2838615041966055043L;
44     
45     	// / CLOVER:OFF
46     
47     	/**
48     	 * Creates an instance of this class.
49     	 */
50     	@Empty private CustomToStringStyle() {
51     		// does nothing
52     	}
53     
54     	// / CLOVER:ON
55     
56     	/**
57     	 * {@inheritDoc}
58     	 */
59     	@Override public void appendEnd(@NonNull final StringBuffer buffer, final Object object) {
60     		super.appendEnd(buffer, object);
61     		appendIdentityHashCode(buffer, object);
62     	}
63     
64     	/**
65     	 * {@inheritDoc}
66     	 */
67     	@Override public void appendStart(@NonNull final StringBuffer buffer, final Object object) {
68     		appendClassName(buffer, object);
69     		appendContentStart(buffer);
70     
71     		if (isFieldSeparatorAtStart()) {
72     			appendFieldSeparator(buffer);
73     		}
74     	}
75     }
