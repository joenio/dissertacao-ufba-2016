182     	/**
183     	 * {@inheritDoc}
184     	 * 
185     	 * @see edu.ksu.cis.indus.staticanalyses.interfaces.AbstractAnalysis#analyze()
186     	 */
187     	@Override public void analyze() {
188     		unstable();
189     
190     		if (monitorInfo.isStable() && callgraphInfo.isStable()) {
191     			final IMonitorGraph<? extends IObjectNode<?, Triple<EnterMonitorStmt, ExitMonitorStmt, SootMethod>>> _monitorGraph = monitorInfo
192     					.getMonitorGraph(callgraphInfo);
193     			processMonitorAndWaitsForLockBasedRelation();
194     			processMonitorsAndWaitsForEnclosureBaseRelation(_monitorGraph);
195     
196     			/*
197     			 * Assume there is an API that can be used to check if waits and notifies are valid. For each method do the
198     			 * following. [processMethod] - Find all the statements immediately enclosed in monitors. - If any of these
199     			 * statements correspond to a invalid wait, then the enclosing monitor is unsafe. Hence, the lock of objects that
200     			 * are used as monitor variable are unsafe. {monitorIsSafeBasedOnWaitSafety} - For each such statement set in safe
201     			 * monitors do the following [waitFreeCyclesExistInMonitor] - find the basic blocks containing these statements -
202     			 * Check if there is a cycle in the basic blocks (excluding those containing the enclosing monitor) with no
203     			 * wait(). - If so, the monitor is unsafe. Hence, the lock of objects that are used as monitor variable are
204     			 * unsafe. Retrieve monitor containment graph for the system. Do a fixed point - propagate the unsafe=ness
205     			 * information in the graph. - for each monitor newly marked as unsafe, mark related waits with their enclosing
206     			 * monitors and related monitors as unsafe.
207     			 */
208     			final Collection<Triple<EnterMonitorStmt, ExitMonitorStmt, SootMethod>> _monitors = monitorInfo
209     					.getMonitorTriples();
210     			final Collection<Triple<EnterMonitorStmt, ExitMonitorStmt, SootMethod>> _seedUnsafeMonitors = new HashSet<Triple<EnterMonitorStmt, ExitMonitorStmt, SootMethod>>();
211     			final Iterator<Triple<EnterMonitorStmt, ExitMonitorStmt, SootMethod>> _i = _monitors.iterator();
212     			final int _iEnd = _monitors.size();
213     
214     			for (int _iIndex = 0; _iIndex < _iEnd; _iIndex++) {
215     				final Triple<EnterMonitorStmt, ExitMonitorStmt, SootMethod> _monitor = _i.next();
216     
217     				if (!_seedUnsafeMonitors.contains(_monitor)) {
218     					if (!(safeByCondition2(_monitor) && safeByCondition1(_monitor, _monitorGraph))) {
219     						_seedUnsafeMonitors.addAll(monitorInfo.getMonitorTriplesOf(_monitor));
220     					}
221     				}
222     			}
223     
224     			propagateSafetyInformation(_seedUnsafeMonitors, _monitorGraph);
225     			stable();
226     
227     			if (LOGGER.isDebugEnabled()) {
228     				final Collection<Triple<EnterMonitorStmt, ExitMonitorStmt, SootMethod>> _safeMonitors = SetUtils.difference(
229     						monitorInfo.getMonitorTriples(), unsafeMonitors);
230     				LOGGER.debug("Unsafe Monitors: \n" + CollectionUtils.prettyPrint(unsafeMonitors));
231     				LOGGER.debug("Safe Monitors: \n" + CollectionUtils.prettyPrint(_safeMonitors));
232     
233     				final Collection<Triple<EnterMonitorStmt, ExitMonitorStmt, SootMethod>> _temp = SetUtils.difference(
234     						monitorInfo.getMonitorTriples(), _safeMonitors);
235     				_temp.removeAll(unsafeMonitors);
236     				LOGGER.debug("Unaccounted Monitors: \n" + CollectionUtils.prettyPrint(_temp));
237     				LOGGER.debug("Overlapping Monitors: \n"
238     						+ CollectionUtils.prettyPrint(SetUtils.intersection(_safeMonitors, unsafeMonitors)));
239     			}
240     		}
241     	}

243     	/**
244     	 * Checks if the lock associated with the monitor of the given synchronized method is safe.
245     	 * 
246     	 * @param method of interest.
247     	 * @return <code>true</code> if the lock associated with the monitor of the given synchronized method is safe;
248     	 *         <code>false</code>, otherwise.
249     	 * @pre method != null and method.isSynchronized()
250     	 */
251     	public boolean isLockSafe(final SootMethod method) {
252     		return !unsafeMonitors.contains(new Triple<EnterMonitorStmt, ExitMonitorStmt, SootMethod>(null, null, method));
253     	}

255     	/**
256     	 * Checks if the lock associated with the monitor occurring in the given statement in the given method is safe.
257     	 * 
258     	 * @param stmt is a monitor (enter/exit) statement.
259     	 * @param method containing <code>stmt</code>.
260     	 * @return <code>true</code> if the lock associated with the monitor occurring in the given statement in the given
261     	 *         method is safe; <code>false</code>, otherwise.
262     	 * @pre method != null and stmt != null
263     	 */
264     	public boolean isLockSafe(final MonitorStmt stmt, final SootMethod method) {
265     		return !CollectionUtils.containsAny(unsafeMonitors, monitorInfo.getMonitorTriplesFor(stmt, method));

324     	 * Checks if the given basic blocks are safe by condition 1 of safe lock.
325     	 * 
326     	 * @param bbg containing the given basic blocks.
327     	 * @param basicBlocks enclosed by the monitor being checked.
328     	 * @param method in which the basic blocks occur.
329     	 * @param waitMethods are the collection of wait methods in the system.
330     	 * @return <code>true</code> if the given blocks are safe by condition 1; <code>false</code>, otherwise.
331     	 * @pre bbg != null and basicBlocks != null and waitMethods != null and method != null
332     	 * @pre bbg.getNodes().containsAll(basicBlocks)
333     	 */
334     	private boolean isMonitorSafeByCond1InTheseBasicBlocks(final BasicBlockGraph bbg,
335     			final Collection<BasicBlock> basicBlocks, final SootMethod method, final Collection<SootMethod> waitMethods) {
336     		boolean _result = true;
337     		final Collection<List<BasicBlock>> _t = AbstractDirectedGraph.findCycles(basicBlocks, bbg.getBackEdges());
338     		final Iterator<List<BasicBlock>> _j = _t.iterator();
339     		final int _jEnd = _t.size();
340     
341     		for (int _jIndex = 0; _jIndex < _jEnd && _result; _jIndex++) {
342     			final Collection<BasicBlock> _cycle = _j.next();
343     			boolean _cycleIsUnsafe = true;
344     			final Collection<Stmt> _invocationStmts = bbg.getEnclosedStmts(_cycle);
345     			final Iterator<Stmt> _filteredIterator = IteratorUtils.filteredIterator(_invocationStmts.iterator(),
346     					SootPredicatesAndTransformers.INVOKING_STMT_PREDICATE);
347     
348     			while (_filteredIterator.hasNext() && _cycleIsUnsafe) {
349     				_cycleIsUnsafe = !callgraphInfo.areAnyMethodsReachableFrom(waitMethods, _filteredIterator.next(), method);
350     			}

352     		}

354     	}

361     				.getMonitorTriples();

366     		final int _iEnd = _monitorTriples.size();
