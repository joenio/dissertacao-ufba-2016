18     package edu.ksu.cis.indus.staticanalyses.flow;
19     
20     import edu.ksu.cis.indus.annotations.Empty;
21     
22     /**
23      * This class represents the variants of entities associated with AST nodes and fields.  This class should be extended as
24      * required for such entities.
25      *
26      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
27      * @version $Revision: 1.13 $
28      * @param <N> is the type of the summary node in the flow analysis.
29      */
30     public class ValuedVariant <N extends IFGNode<?, ?, N>>
31       implements IVariant {
32     	/** 
33     	 * <p>
34     	 * The flow graph node associated with this variant.
35     	 * </p>
36     	 */
37     	protected final N node;
38     
39     	/**
40     	 * Creates a new <code>ValuedVariant</code> instance.
41     	 *
42     	 * @param flowNode the flow graph node associated with this variant.
43     	 *
44     	 * @pre flowNode != null
45     	 */
46     	ValuedVariant(final N flowNode) {
47     		this.node = flowNode;
48     	}
49     
50     	/**
51     	 * Returns the flow graph node associated with this node.
52     	 *
53     	 * @return the flow graph node associated with this node.
54     	 *
55     	 * @post result != null
56     	 */
57     	public N getFGNode() {
58     		return node;
59     	}
60     
61     	/**
62     	 * Performs nothing.  This will be called after a variant is created and should be implemented by subclasses.
63     	 */
64     	@Empty public void process() {
65     		// does nothing
66     	}
67     }
