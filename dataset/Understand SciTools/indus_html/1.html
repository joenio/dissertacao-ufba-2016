18     package edu.ksu.cis.indus.interfaces;
19     
20     import org.apache.commons.pool.ObjectPool;
21     
22     import org.slf4j.Logger;
23     import org.slf4j.LoggerFactory;
24     
25     
26     /**
27      * This provides an abstract implementation of <code>IPoolable</code>.  Concrete implementation should extend this class.
28      *
29      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
30      * @author $Author: rvprasad $
31      * @version $Revision: 1.8 $ $Date: 2007/02/10 19:08:35 $
32      */
33     public abstract class AbstractPoolable
34       implements IPoolable {
35     	/** 
36     	 * The logger used by instances of this class to log messages.
37     	 */
38     	private static final Logger LOGGER = LoggerFactory.getLogger(AbstractPoolable.class);
39     
40     	/** 
41     	 * The pool to which this object should be returned to.
42     	 */
43     	private ObjectPool pool;
44     
45     	/**
46     	 * {@inheritDoc}
47     	 *
48     	 * @throws RuntimeException if the returning of the object to it's pool failed.
49     	 */
50     	public void returnToPool() {
51     		if (pool != null) {
52     			try {
53     				pool.returnObject(this);
54     			} catch (final Exception _e) {
55     				if (LOGGER.isWarnEnabled()) {
56     					LOGGER.warn("How can this happen?", _e);
57     				}
58     				throw new RuntimeException(_e);
59     			}
60     		}
61     	}
62     
63     	/**
64     	 * {@inheritDoc}
65     	 */
66     	public final void setPool(final ObjectPool thePool) {
67     		if (thePool == null) {
68     			throw new IllegalArgumentException("Invalid argument: null");
69     		}
70     		pool = thePool;
71     	}
72     }
