18     
19     import edu.ksu.cis.indus.annotations.Empty;

23     import edu.ksu.cis.indus.interfaces.AbstractStatus;
24     
25     import edu.ksu.cis.indus.processing.IProcessor;

27     import edu.ksu.cis.indus.staticanalyses.InitializationException;
28     
29     import java.util.ArrayList;
30     import java.util.Collections;
31     import java.util.HashMap;
32     import java.util.List;
33     import java.util.Map;
34     
35     import soot.SootMethod;
36     
37     import soot.jimple.Stmt;
38     import soot.toolkits.graph.UnitGraph;

40     /**
41      * This class is the skeletal implementation of the interface of analyses used to execute them.
42      *
43      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
44      * @author $Author: rvprasad $
45      * @version $Revision: 1.29 $
46      */
47     public abstract class AbstractAnalysis
48     		extends AbstractStatus
49     		implements IAnalysis {
50     
51     	/**
52     	 * This contains auxiliary information required by the subclasses. It is recommended that this represent
53     	 * <code>java.util.Properties</code> but map a <code>String</code> to an <code>Object</code>.
54     	 */
55     	protected final Map<Comparable<?>, Object> info = new HashMap<Comparable<?>, Object>();
56     
57     	/**
58     	 * The pre-processor for this analysis, if one exists.
59     	 */
60     	protected IProcessor preprocessor;
61     
62     	/**
63     	 * This manages the basic block graphs of methods.
64     	 */
65     	private BasicBlockGraphMgr graphManager;
66     
67     	/**
68     	 * @see IAnalysis#analyze()
69     	 */
70     	public abstract void analyze();
71     
72     	/**
73     	 * {@inheritDoc}
74     	 * <p>
75     	 * Subclasses need not override this method. Rather they can set <code>preprocessor</code> field to a preprocessor and
76     	 * this method will use that to provide the correct information to the caller.
77     	 * </p>
78     	 */
79     	public boolean doesPreProcessing() {
80     		return preprocessor != null;
81     	}
82     
83     	/**
84     	 * @see IAnalysis#getPreProcessor()
85     	 */
86     	public IProcessor getPreProcessor() {
87     		return preprocessor;
88     	}
89     
90     	/**
91     	 * Returns the statistics about this analysis in the form of a <code>String</code>.
92     	 *
93     	 * @return the statistics about this analysis.
94     	 */
95     	public String getStatistics() {
96     		return getClass() + " does not implement this method.";
97     	}
98     
99     	/**
100     	 * {@inheritDoc}
101     	 * <p>
102     	 * Refer to {@link #info info} and subclass documenation for more details.
103     	 * </p>
104     	 */
105     	public final void initialize(final Map<Comparable<?>, Object> infoParam) throws InitializationException {
106     		info.putAll(infoParam);
107     		setup();
108     	}
109     
110     	/**
111     	 * {@inheritDoc}
112     	 *
113     	 * @post info.size() == 0
114     	 */
115     	public void reset() {
116     		unstable();
117     		info.clear();
118     	}
119     
120     	/**
121     	 * @see IAnalysis#setBasicBlockGraphManager(BasicBlockGraphMgr)
122     	 */
123     	public void setBasicBlockGraphManager(final BasicBlockGraphMgr bbm) {
124     		graphManager = bbm;
125     	}
126     
127     	/**
128     	 * Returns the basic block graph for the given method, if available. If not it will try to acquire the unit graph from the
129     	 * application. From that unit graph it will construct a basic block graph and return it.
130     	 *
131     	 * @param method for which the basic block graph is requested.
132     	 * @return the basic block graph corresponding to <code>method</code>.
133     	 * @pre method != null
134     	 */
135     	protected BasicBlockGraph getBasicBlockGraph(final SootMethod method) {
136     		return graphManager.getBasicBlockGraph(method);
137     	}
138     
139     	/**
140     	 * Returns a list of statements in the given method, if it exists. This implementation retrieves the statement list from
141     	 * the basic block graph manager, if it is available. If not, it retrieves the statement list from the method body
142     	 * directly. It will return an unmodifiable list of statements.
143     	 *
144     	 * @param method of interest.
145     	 * @return an unmodifiable list of statements.
146     	 * @pre method != null
147     	 * @post result != null
148     	 */
149     	protected List<Stmt> getStmtList(final SootMethod method) {
150     		List<Stmt> _result;
151     
152     		if (graphManager != null) {
153     			_result = graphManager.getStmtList(method);
154     		} else {
155     			final UnitGraph _stmtGraph = graphManager.getStmtGraph(method);
156     
157     			if (_stmtGraph != null) {
158     				_result = Collections.unmodifiableList(new ArrayList<Stmt>(_stmtGraph.getBody().getUnits()));
159     			} else {
160     				_result = Collections.emptyList();
161     			}
162     		}
163     		return _result;
164     	}
165     
166     	/**
167     	 * Retrives the unit graph of the given method.
168     	 *
169     	 * @param method for which the unit graph is requested.
170     	 * @return the unit graph of the method.
171     	 * @post result != null
172     	 */
173     	protected UnitGraph getUnitGraph(final SootMethod method) {
174     		return graphManager.getStmtGraph(method);
175     	}
176     
177     	/**
178     	 * Setup data structures after initialization. This is a convenience method for subclasses to do processing after the
179     	 * calls to <code>initialize</code> and before the call to <code>preprocess</code>.
180     	 *
181     	 * @throws InitializationException is never thrown by this implementation.
182     	 */
183     	@Empty protected void setup() throws InitializationException {
184     		// does nothing
185     	}
186     }
