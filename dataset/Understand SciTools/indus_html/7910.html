43     /**

46      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>

50     public final class SlicerToolHelper {
51     
52     	/**
53     	 * Logger for this class
54     	 */
55     	private static final Logger LOGGER = LoggerFactory.getLogger(SlicerToolHelper.class);
56     
57     	/**
58     	 * Creates an instance of this class.
59     	 */
60     	private SlicerToolHelper() {
61     		super();
62     	}

64     	/**
65     	 * Applies the given post processor o the slice calculated by the given tool.
66     	 * 
67     	 * @param tool of interest.
68     	 * @param processor to be applied.
69     	 * @pre tool != null and processor != null
70     	 */
71     	public static void applyPostProcessor(final SlicerTool<?> tool, final ISlicePostProcessor processor) {
72     		final SliceCollector _collector = tool.getSliceCollector();
73     		final Collection<SootMethod> _methods = _collector.getMethodsInSlice();
74     		final SliceGotoProcessor _gotoProcessor = new SliceGotoProcessor(_collector);
75     		final BasicBlockGraphMgr _bbgMgr = tool.getBasicBlockGraphManager();
76     		processor.process(_methods, _bbgMgr, _collector);
77     		_gotoProcessor.process(_methods, _bbgMgr);
78     	}
79     
80     	/**
81     	 * Loads the configuration in the named file.
82     	 * 
83     	 * @param configFileName is the name of the configuration file to load; if this is <code>null</code> then the default
84     	 *            configuration file will be loaded.
85     	 * @return the configuration as stored in the named file.
86     	 * @throws IllegalStateException if the named file or the default file cannot be found (in that order) or when there is an
87     	 *             while operating on the file.
88     	 */
89     	public static String loadConfigurationInFile(final String configFileName) {
90     		InputStream _inStream = null;
91     		String _result = null;
92     		final URL _filename;
93     		
94     		if (configFileName == null) {
95     			LOGGER.warn("Trying to use default configuration.");
96     			_filename = ClassLoader.getSystemResource("edu/ksu/cis/indus/tools/slicer/default_slicer_configuration.xml");
97     		} else {
98     			try {
99     				_filename = new File(configFileName).toURI().toURL();
100     			} catch (final MalformedURLException _e) {
101     				LOGGER.warn("The specified path " + configFileName + " does not exist.");
102     				throw new IllegalArgumentException("The specified path is invalid. - " + configFileName, _e);
103     			}
104     		}
105     
106     		try {
107     			_inStream = _filename.openStream();
108     			_result = IOUtils.toString(_inStream);
109     		} catch (final IOException _e2) {
110     			LOGGER.error("Could not retrieve a handle to configuration file or an ."
111     					+ "IO error while reading configuration file. ", _e2);
112     			throw new IllegalStateException(_e2);
113     		} finally {
114     			IOUtils.closeQuietly(_inStream);
115     		}
116     		return _result;
117     	}

119     	/**
120     	 * Optimizes the slice calculated by the given tool for space. This method should be called after residualization.
121     	 * 
122     	 * @param tool in which the slice should be optimized.
123     	 * @param classesToRetain is the collection of FQN of classes that need to be retained in the slice.
124     	 * @return the unspecified classes that were retained.
125     	 * @pre tool != null and classesToRetain != null
126     	 * @post result != null
127     	 */
128     	public static Collection<SootClass> optimizeForSpaceAfterResidualization(final SlicerTool<?> tool,
129     			final Collection<String> classesToRetain) {
130     		final Collection<SootClass> _classesToErase = new HashSet<SootClass>();
131     		final Collection<SootClass> _classes = tool.getSystem().getClasses();
132     		final Iterator<SootClass> _i = _classes.iterator();
133     		final int _iEnd = _classes.size();
134     
135     		for (int _iIndex = 0; _iIndex < _iEnd; _iIndex++) {
136     			final SootClass _sc = _i.next();
137     
138     			if (_sc.getMethods().size() == 0 && _sc.getFields().size() == 0 && !classesToRetain.contains(_sc.getName())) {
139     				_classesToErase.add(_sc);
140     			}
141     		}
142     
143     		final Collection<SootClass> _c = Util.eraseClassesFrom(_classesToErase, tool.getSystem());
144     		_classesToErase.removeAll(_c);
145     		return _c;
146     	}
147     
148     	/**
149     	 * Optimizes the slice calculated by the given tool for space. This method should be called before residualization.
150     	 * 
151     	 * @param tool in which the slice should be optimized.
152     	 * @param classesToRetain is the collection of FQN of classes that need to be retained in the slice.
153     	 * @pre tool != null and classesToRetain != null
154     	 */
155     	public static void optimizeForSpaceBeforeResidualization(final SlicerTool<?> tool,
156     			final Collection<String> classesToRetain) {
157     		final IEnvironment _system = tool.getSystem();
158     		final ExecutableSlicePostProcessorAndModifier _processor = new ExecutableSlicePostProcessorAndModifier(_system,
159     				classesToRetain);
160     		applyPostProcessor(tool, _processor);
161     	}
162     }
