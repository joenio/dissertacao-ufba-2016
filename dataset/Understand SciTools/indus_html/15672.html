17     package edu.ksu.cis.indus.staticanalyses.flow.processors;
18     
19     import edu.ksu.cis.indus.common.datastructures.Pair.PairManager;
20     import edu.ksu.cis.indus.common.soot.BasicBlockGraphMgr;
21     import edu.ksu.cis.indus.interfaces.ICallGraphInfo;
22     import edu.ksu.cis.indus.interfaces.IThreadGraphInfo;
23     import edu.ksu.cis.indus.staticanalyses.cfg.CFGAnalysis;
24     import edu.ksu.cis.indus.staticanalyses.interfaces.IValueAnalyzer;
25     import soot.SootMethod;
26     import soot.Value;
27     import soot.jimple.Stmt;
28     
29     /**
30      * This class provides intra-thread aliased use-def information which is based on types, points-to information, and call
31      * graph. If the use is reachable from the def via the control flow graph or via the CFG and the call graph, then def and use
32      * site are related by use-def relation. This does not consider any intervening definition when calculating the def-use
33      * relation.
34      * 
35      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
36      * @author $Author$
37      * @version $Revision$ $Date$
38      * @see AliasedUseDefInfo
39      */
40     public final class AliasedUseDefInfov2
41     		extends AliasedUseDefInfo {
42     
43     	/**
44     	 * This provides the call graph of the system.
45     	 */
46     	protected final ICallGraphInfo cgi;
47     
48     	/**
49     	 * This provides the thread graph of the system.
50     	 */
51     	protected final IThreadGraphInfo tgi;
52     
53     	/**
54     	 * Creates an instance of this class.
55     	 * 
56     	 * @param iva is the value analyzer to use.
57     	 * @param cg is the call graph to use.
58     	 * @param tg is the thread graph to use. If this parameter is <code>null</code> then it is assumed all methods execute
59     	 *            in the same thread.
60     	 * @param bbgManager is the basic block graph manager to use.
61     	 * @param pairManager is the pair object manager to use.
62     	 * @pre iva != null and cg != null and bbgManager != null and pairManager != null
63     	 */
64     	public AliasedUseDefInfov2(final IValueAnalyzer<Value> iva, final ICallGraphInfo cg, final IThreadGraphInfo tg,
65     			final BasicBlockGraphMgr bbgManager, final PairManager pairManager) {
66     		super(iva, bbgManager, pairManager, new CFGAnalysis(cg, bbgManager));
67     		cgi = cg;
68     		tgi = tg;
69     	}
70     
71     	/**
72     	 * @see edu.ksu.cis.indus.processing.IProcessor#consolidate()
73     	 */
74     	@Override public void consolidate() {
75     		super.consolidate();
76     		cfgAnalysis.reset();
77     	}
78     
79     	/**
80     	 * {@inheritDoc}
81     	 * <p>
82     	 * This implementation actually checks if the use site is reachable from the def site via control flow. However, it does
83     	 * not check for the existence of any overriding definitions alike strong updates.
84     	 * </p>
85     	 * 
86     	 * @pre defMethod != null and defStmt != null and useMethod != null and useStmt != null
87     	 */
88     	@Override public boolean isReachableViaInterProceduralControlFlow(final SootMethod defMethod, final Stmt defStmt,
89     			final SootMethod useMethod, final Stmt useStmt) {
90     		return cfgAnalysis.isReachableViaInterProceduralControlFlow(defMethod, defStmt, useMethod, useStmt, tgi, false);
91     	}
92     }
