210     		_gridData1.horizontalSpan = 2;

214     		_rowLayout1.type = SWT.VERTICAL;
215     		_rowLayout1.fill = true;
216     		_deadlockGroup.setLayout(_rowLayout1);
217     		_deadlockGroup.setText("Preservation of Deadlocking property");
218     
219     		final Button _button = new Button(_deadlockGroup, SWT.CHECK);
220     		_button.setText("Preserve Deadlock");
221     
222     		final Button _applclasses = new Button(_deadlockGroup, SWT.CHECK);
223     		_applclasses.setText("Preserve sync constructs in application classes only");
224     		_applclasses.setSelection(cfg.areSynchronizationsOnlyInAppClassesConsidered());
225     		_applclasses.addSelectionListener(new BooleanPropertySelectionListener(

229

232     
233     		final RowLayout _rowLayout2 = new RowLayout();
234     		_rowLayout2.type = SWT.VERTICAL;
235     		_group1.setLayout(_rowLayout2);
236     
237     		final Button _allSyncStrategy = new Button(_group1, SWT.RADIO);
238     		_allSyncStrategy.setText("All Synchronization constructs");
239     		_allSyncStrategy.setToolTipText("Synchronization constructs are preserved independent of any property");
240     
241     		final Button _escapingSyncStrategy = new Button(_group1, SWT.RADIO);
242     		_escapingSyncStrategy.setText("Escaping Sychronization constructs");
243     		_escapingSyncStrategy.setToolTipText("Only synchronization constructs involving escaping lock object are preserved.");
244     
245     		final Button _ctxtsensEscapingSyncStrategy = new Button(_group1, SWT.RADIO);
246     		_ctxtsensEscapingSyncStrategy.setText("Escaping Sychronization constructs with their contexts");
247     		_ctxtsensEscapingSyncStrategy.setToolTipText("Calling contexs of the preserved synchronization constructs "
248     			+ "are considered");
249     
250     		final SelectionListener _sl4 =
251     			new SelectionListener() {
252     				public void widgetSelected(final SelectionEvent evt) {
253     					Object _value = null;
254     
255     					if (evt.widget == _ctxtsensEscapingSyncStrategy) {
256     						_value = SlicerConfiguration.CONTEXT_SENSITIVE_ESCAPING_SYNC_CONSTRUCTS;
257     					} else if (evt.widget == _escapingSyncStrategy) {
258     						_value = SlicerConfiguration.ESCAPING_SYNC_CONSTRUCTS;
259     					} else if (evt.widget == _allSyncStrategy) {
260     						_value = SlicerConfiguration.ALL_SYNC_CONSTRUCTS;
261     					}
262     
263     					if (_value != null) {
264     						cfg.setProperty(SlicerConfiguration.DEADLOCK_CRITERIA_SELECTION_STRATEGY, _value);
265     					}
266     				}
267     
268     				public void widgetDefaultSelected(final SelectionEvent evt) {
269     					widgetSelected(evt);
270     				}
271     			};
272     		_allSyncStrategy.addSelectionListener(_sl4);
273     		_escapingSyncStrategy.addSelectionListener(_sl4);
274     		_ctxtsensEscapingSyncStrategy.addSelectionListener(_sl4);
275     
276     		final Object _temp = cfg.getDeadlockCriteriaSelectionStrategy();
277     
278     		if (_temp.equals(SlicerConfiguration.ALL_SYNC_CONSTRUCTS)) {
279     			_allSyncStrategy.setSelection(true);
280     		} else if (_temp.equals(SlicerConfiguration.ESCAPING_SYNC_CONSTRUCTS)) {
281     			_escapingSyncStrategy.setSelection(true);
282     		} else if (_temp.equals(SlicerConfiguration.CONTEXT_SENSITIVE_ESCAPING_SYNC_CONSTRUCTS)) {
283     			_ctxtsensEscapingSyncStrategy.setSelection(true);
284     		}
285     
286     		final SelectionListener _sl5 =
287     			new BooleanPropertySelectionListener(SlicerConfiguration.SLICE_FOR_DEADLOCK, _button, cfg) {
288     				@Override public void widgetSelected(@SuppressWarnings("unused") final SelectionEvent evt) {
289     					final boolean _value = button.getSelection();
290     					containingConfiguration.setProperty(id, Boolean.valueOf(_value));
291     					_group1.setEnabled(_value);
292     					_applclasses.setEnabled(_value);
293     					_allSyncStrategy.setEnabled(_value);
294     					_escapingSyncStrategy.setEnabled(_value);
295     					_ctxtsensEscapingSyncStrategy.setEnabled(_value);
296     				}
297     			};
298     		_button.addSelectionListener(_sl5);
299     
300     		_button.setSelection(cfg.getSliceForDeadlock());
301     		_button.notifyListeners(SWT.Selection, null);
302     	}
