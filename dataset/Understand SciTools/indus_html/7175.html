38     /**
39      * This class uses escape-analysis as calculated by <code>EquivalenceClassBasedEscapeAnalysis</code> to prune the ready
40      * dependency information calculated by it's parent class. This class will also use OFA information if it is configured to do
41      * so.
42      * 
43      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
44      * @author $Author: rvprasad $
45      * @version $Revision: 1.30 $
46      * @see edu.ksu.cis.indus.staticanalyses.dependency.ReadyDAv1
47      */
48     public class ReadyDAv2
49     		extends ReadyDAv1 {
50     
51     	/**
52     	 * The logger used by instances of this class to log messages.
53     	 */
54     	private static final Logger LOGGER = LoggerFactory.getLogger(ReadyDAv2.class);
55     
56     	/**
57     	 * This provides information to prune ready dependence edges.
58     	 */
59     	protected IEscapeInfo ecba;
60     
61     	/**
62     	 * Creates an instance of this class.
63     	 * 
64     	 * @param directionSensitiveInfo that controls the direction.
65     	 * @param direction of the analysis
66     	 * @pre info != null and direction != null
67     	 */
68     	protected ReadyDAv2(final IDirectionSensitiveInfo directionSensitiveInfo, final Direction direction) {
69     		super(directionSensitiveInfo, direction);
70     	}
71     
72     	/**
73     	 * Retrieves an instance of ready dependence analysis that calculates information in backward direction.
74     	 * 
75     	 * @return an instance of ready dependence.
76     	 * @post result != null
77     	 */
78     	public static ReadyDAv1 getBackwardReadyDA() {
79     		return new ReadyDAv2(new BackwardDirectionSensitiveInfo(), Direction.BACKWARD_DIRECTION);
80     	}
81     
82     	/**
83     	 * Retrieves an instance of ready dependence analysis that calculates information in forward direction.
84     	 * 
85     	 * @return an instance of ready dependence.
86     	 * @post result != null
87     	 */
88     	public static ReadyDAv1 getForwardReadyDA() {
89     		return new ReadyDAv2(new ForwardDirectionSensitiveInfo(), Direction.FORWARD_DIRECTION);
90     	}
91     
92     	/**
93     	 * @see edu.ksu.cis.indus.staticanalyses.interfaces.AbstractAnalysis#analyze()
94     	 */
95     	@Override public void analyze() {
96     		unstable();
97     
98     		if (ecba.isStable()) {
99     			super.analyze();
100     		}
101     	}
102     
103     	/**
104     	 * Checks if the given enter monitor statement/synchronized method is dependent on the exit monitor statement/synchronized
105     	 * method according to rule 2. The results of escape analysis info calculated {@link
106     	 * edu.ksu.cis.indus.staticanalyses.concurrency.escape.EquivalenceClassBasedEscapeAnalysis
107     	 * EquivalenceClassBasedEscapeAnalysis} analysis is used to determine the dependence.
108     	 * 
109     	 * @param enterPair is the enter monitor statement and containg statement pair.
110     	 * @param exitPair is the exit monitor statement and containg statement pair.
111     	 * @return <code>true</code> if there is a dependence; <code>false</code>, otherwise.
112     	 * @pre enterPair.getSecond() != null and exitPair.getSecond() != null
113     	 * @see ReadyDAv1#ifDependentOnByRule2(Pair, Pair)
114     	 */
115     	@Override protected boolean ifDependentOnByRule2(final Pair<EnterMonitorStmt, SootMethod> enterPair,
116     			final Pair<ExitMonitorStmt, SootMethod> exitPair) {
117     		boolean _result = super.ifDependentOnByRule2(enterPair, exitPair);
118     
119     		if (_result) {
120     			final SootMethod _enterMethod = enterPair.getSecond();
121     			final SootMethod _exitMethod = exitPair.getSecond();
122     			final EnterMonitorStmt _o1 = enterPair.getFirst();
123     			final ExitMonitorStmt _o2 = exitPair.getFirst();
124     			boolean _flag1;
125     			boolean _flag2;
126     
127     			if (_o1 == null) {
128     				_flag1 = ecba.thisEscapes(_enterMethod);
129     			} else {
130     				final Value _enter = _o1.getOp();
131     				_flag1 = ecba.escapes(_enter, _enterMethod);
132     			}
133     
134     			if (_o2 == null) {
135     				_flag2 = ecba.thisEscapes(_exitMethod);
136     			} else {
137     				final Value _exit = _o2.getOp();
138     				_flag2 = ecba.escapes(_exit, _exitMethod);
139     			}
140     			_result = _flag1 && _flag2;
141     		}
142     		return _result;
143     	}
144     
145     	/**
146     	 * Checks if the given <code>wait()</code> call-site is dependent on the <code>notifyXX()</code> call-site according
147     	 * to rule 2. The results of escape analysis info calculated by <code>EquivalenceClassbasedAnalysis</code>analysis is
148     	 * used to determine the dependence.
149     	 * 
150     	 * @param wPair is the statement in which <code>java.lang.Object.wait()</code> is invoked.
151     	 * @param nPair is the statement in which <code>java.lang.Object.notifyXX()</code> is invoked.
152     	 * @return <code>true</code> if there is a dependence; <code>false</code>, otherwise.
153     	 * @pre wPair.getSecond() != null and nPair.getSecond() != null
154     	 * @see ReadyDAv1#ifDependentOnByRule4(Pair, Pair)
155     	 */
156     	@Override protected boolean ifDependentOnByRule4(final Pair<InvokeStmt, SootMethod> wPair,
157     			final Pair<InvokeStmt, SootMethod> nPair) {
158     		boolean _result = super.ifDependentOnByRule4(wPair, nPair);
159     
160     		if (_result) {
161     			final Value _notify = ((VirtualInvokeExpr) (nPair.getFirst()).getInvokeExpr()).getBase();
162     			final Value _wait = ((VirtualInvokeExpr) (wPair.getFirst()).getInvokeExpr()).getBase();
163     			final SootMethod _wMethod = wPair.getSecond();
164     			final SootMethod _nMethod = nPair.getSecond();
165     			_result = ecba.escapes(_notify, _nMethod) && ecba.escapes(_wait, _wMethod);
166     		}
167     		return _result;
168     	}
169     
170     	/**
171     	 * Extracts information as provided by environment at initialization time. It collects <code>wait</code> and
172     	 * <code>notifyXX</code> methods as represented in the AST system. It also extract call graph info, pair manaing
173     	 * service, and environment from the <code>info</code> member.
174     	 * 
175     	 * @throws InitializationException when call graph info, pair managing service, or environment is not available in
176     	 *             <code>info</code> member.
177     	 * @see edu.ksu.cis.indus.staticanalyses.interfaces.AbstractAnalysis#setup()
178     	 */
179     	@Override protected void setup() throws InitializationException {
180     		super.setup();
181     
182     		ecba = (IEscapeInfo) info.get(IEscapeInfo.ID);
183     
184     		if (ecba == null) {
185     			LOGGER.error(IEscapeInfo.ID + " was not provided in info.");
186     			throw new InitializationException(IEscapeInfo.ID + " was not provided in info.");
187     		}
188     	}
189     }
