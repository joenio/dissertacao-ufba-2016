26     import soot.ArrayType;

28     import soot.jimple.ArrayRef;

31     /**
32      * This class is the counter part of <code>FieldAccessExprWork</code>.  It encapsulates the logic to instrument the flow
33      * values through array components.
34      *
35      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
36      * @version $Revision: 1.19 $
37      * @param <T>  is the type of the token set object.
38      */
39     class ArrayAccessExprWork<T extends ITokens<T, Value>>
40       extends AbstractMemberDataAccessExprWork<T> {
41     	/**
42     	 * Creates a new <code>ArrayAccessExprWork</code> instance.
43     	 *
44     	 * @param callerMethod the method in which the access occurs.
45     	 * @param accessContext the context in which the access occurs.
46     	 * @param accessNode the flow graph node associated with the access expression.
47     	 * @param connectorToUse the connector to use to connect the ast node to the non-ast node.
48     	 * @param tokenSet used to store the tokens that trigger the execution of this work peice.
49     	 *
50     	 * @pre callerMethod != null and accessProgramPoint != null and accessContext != null and accessNode != null and
51     	 * 		connectorToUse != null and tokenSet != null
52     	 */
53     	public ArrayAccessExprWork(final IMethodVariant<OFAFGNode<T>> callerMethod, final Context accessContext, final OFAFGNode<T> accessNode,
54     		final IFGNodeConnector<OFAFGNode<T>> connectorToUse, final T tokenSet) {
55     		super(callerMethod, accessContext, accessNode, connectorToUse, tokenSet);
56     	}
57     
58     	/**
59     	 * @see edu.ksu.cis.indus.staticanalyses.flow.instances.ofa.AbstractMemberDataAccessExprWork#getFGNodeForMemberData()
60     	 */
61     	@Override protected  OFAFGNode<T> getFGNodeForMemberData() {
62     		final ArrayType _atype = (ArrayType) ((ArrayRef) accessExprBox.getValue()).getBase().getType();
63     		final FA<?, ?, OFAFGNode<T>, ?> _fa = caller.getFA();
64     		return _fa.getArrayVariant(_atype, context).getFGNode();
65     	}
66     }
