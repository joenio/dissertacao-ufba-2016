46     /**
47      * This class generates seed slicing criteria to preserve the deadlocking property of the system being sliced.
48      * 
49      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
50      * @author $Author: rvprasad $
51      * @version $Revision: 1.12 $ $Date: 2007/02/10 19:06:00 $
52      */
53     public final class DeadlockPreservingCriteriaGenerator
54     		extends AbstractSliceCriteriaGenerator<SootMethod, Triple<EnterMonitorStmt, ExitMonitorStmt, SootMethod>> {
55     
56     	/**
57     	 * The logger used by instances of this class to log messages.
58     	 */
59     	private static final Logger LOGGER = LoggerFactory.getLogger(DeadlockPreservingCriteriaGenerator.class);
60     
61     	/**
62     	 * {@inheritDoc}
63     	 * 
64     	 * @see edu.ksu.cis.indus.tools.slicer.criteria.generators.AbstractSliceCriteriaGenerator#getCriteriaTemplateMethod()
65     	 */
66     	@Override protected Collection<ISliceCriterion> getCriteriaTemplateMethod() {
67     		if (LOGGER.isDebugEnabled()) {
68     			LOGGER.debug("BEGIN: creating deadlock criteria.");
69     		}
70     
71     		final Collection<ISliceCriterion> _result = new HashSet<ISliceCriterion>();
72     		final Collection<ISliceCriterion> _subResult = new HashSet<ISliceCriterion>();
73     		final Context _context = new Context();
74     		final SlicerTool<?> _slicer = getSlicerTool();
75     		final BasicBlockGraphMgr _bbgMgr = _slicer.getBasicBlockGraphManager();
76     		final SliceCriteriaFactory _criteriaFactory = SliceCriteriaFactory.getFactory();
77     
78     		final Collection<Triple<EnterMonitorStmt, ExitMonitorStmt, SootMethod>> _monitorTriples = _slicer.getMonitorInfo()
79     				.getMonitorTriples();
80     		for (final Iterator<Triple<EnterMonitorStmt, ExitMonitorStmt, SootMethod>> _i = _monitorTriples.iterator(); _i
81     				.hasNext();) {
82     			final Triple<EnterMonitorStmt, ExitMonitorStmt, SootMethod> _mTriple = _i.next();
83     			final SootMethod _method = _mTriple.getThird();
84     
85     			if (shouldConsiderSite(_method)) {
86     				_subResult.clear();
87     				_context.setRootMethod(_method);
88     
89     				if (shouldGenerateCriteriaFrom(_mTriple)) {
90     					if (_mTriple.getFirst() == null) {
91     						// add all entry points and return points (including throws) of the method as the criteria
92     						_subResult.addAll(generateCriteriaForMethodBoundary(_method, _bbgMgr, _criteriaFactory));
93     						contextualizeCriteriaBasedOnThis(_method, _subResult);
94     					} else {
95     						final EnterMonitorStmt _enterStmt = _mTriple.getFirst();
96     
97     						_subResult.addAll(_criteriaFactory.getCriteria(_method, _enterStmt, true, true));
98     						_subResult.addAll(_criteriaFactory.getCriteria(_method, _mTriple.getSecond(), true, true));
99     						_context.setStmt(_enterStmt);
100     						_context.setProgramPoint(_enterStmt.getOpBox());
101     						contextualizeCriteriaBasedOnProgramPoint(_context, _subResult);
102     					}
103     				}
104     				_result.addAll(_subResult);
105     			}
106     		}
107     
108     		final InstanceOfPredicate<InvokeStmt, Stmt> _instanceOfPredicate = new InstanceOfPredicate<InvokeStmt, Stmt>(
109     				InvokeStmt.class);
110     		for (final Iterator<SootMethod> _i = _slicer.getCallGraph().getReachableMethods().iterator(); _i.hasNext();) {
111     			final SootMethod _caller = _i.next();
112     			final Iterator<Stmt> _sl = _bbgMgr.getStmtList(_caller).iterator();
113     			_context.setRootMethod(_caller);
114     
115     			for (final Iterator<Stmt> _j = IteratorUtils.filteredIterator(_sl, _instanceOfPredicate); _j.hasNext();) {
116     				final InvokeStmt _stmt = (InvokeStmt) _j.next();
117     				final InvokeExpr _ve = _stmt.getInvokeExpr();
118     				final SootMethod _callee = _ve.getMethod();
119     
120     				if (_callee.getName().equals("join") && _callee.getParameterCount() == 0
121     						&& _callee.getReturnType() == VoidType.v()
122     						&& _callee.getDeclaringClass().getName().equals("java.lang.Thread")) {
123     					_subResult.clear();
124     					_subResult.addAll(_criteriaFactory.getCriteria(_caller, _stmt, true, true));
125     					_context.setStmt(_stmt);
126     					_context.setProgramPoint(((VirtualInvokeExpr) _ve).getBaseBox());
127     					contextualizeCriteriaBasedOnProgramPoint(_context, _subResult);
128     					_result.addAll(_subResult);
129     				}
130     			}
131     		}
132     
133     		if (LOGGER.isDebugEnabled()) {
134     			LOGGER.debug("END: creating deadlock criteria. - " + _result);
135     		}
136     
137     		return _result;
138     	}
139     
140     	/**
141     	 * Generates the slice criteria to capture the boundary statements of the given method.
142     	 * 
143     	 * @param method of interest
144     	 * @param bbgMgr is the basic block graph manager to use.
145     	 * @param criteriaFactory is the slice criteria factory to use.
146     	 * @return the collection of slice criteria
147     	 * @throws IllegalStateException when the basic block graph manager is improperly initialized.
148     	 * @pre method != null and bbgMgr != null and criteriaFactory != null
149     	 */
150     	private Collection<ISliceCriterion> generateCriteriaForMethodBoundary(final SootMethod method,
151     			final BasicBlockGraphMgr bbgMgr, final SliceCriteriaFactory criteriaFactory) throws IllegalStateException {
152     		final Collection<ISliceCriterion> _result = new ArrayList<ISliceCriterion>();
153     		final BasicBlockGraph _bbg = bbgMgr.getBasicBlockGraph(method);
154     
155     		if (_bbg != null) {
156     			final Collection<ISliceCriterion> _criteria = criteriaFactory.getCriteria(method, bbgMgr.getStmtList(method).get(
157     					0), false);
158     			_result.addAll(_criteria);
159     
160     			for (final Iterator<BasicBlockGraph.BasicBlock> _j = _bbg.getSinks().iterator(); _j.hasNext();) {
161     				final BasicBlockGraph.BasicBlock _bb = _j.next();
162     				final Stmt _stmt = _bb.getTrailerStmt();
163     				_result.addAll(criteriaFactory.getCriteria(method, _stmt, false));
164     			}
165     		} else {
166     			if (LOGGER.isWarnEnabled()) {
167     				LOGGER.warn("Could not retrieve the basic block graph for " + method.getSignature() + ".  Moving on.");
168     			}
169     		}
170     		return _result;
171     	}
172     }
