17     package edu.ksu.cis.indus.staticanalyses.flow;
18     
19     import edu.ksu.cis.indus.processing.Context;
20     import edu.ksu.cis.indus.staticanalyses.tokens.ITokens;
21     
22     import java.util.Collection;
23     import java.util.HashMap;
24     import java.util.Map;
25     
26     import org.slf4j.Logger;
27     import org.slf4j.LoggerFactory;
28     
29     /**
30      * This class manages variants. An variant manager classes should extend this class. This class embodies the logic to manage
31      * the variants.
32      * 
33      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
34      * @version $Revision: 1.20 $
35      * @param <V> is the type of the variant.
36      * @param <E> is the type of the entity whose variance is being tracked.
37      * @param <SYM> is the type of symbol whose flow is being analyzed.
38      * @param <N> is the type of the summary node in the flow analysis.
39      */
40     public abstract class AbstractVariantManager<V extends IVariant, E, SYM, T extends ITokens<T, SYM>, N extends IFGNode<SYM, T, N>, R>
41     		implements IVariantManager<V, E> {
42     
43     	/**
44     	 * The logger used by instances of this class to log messages.
45     	 */
46     	private static final Logger LOGGER = LoggerFactory.getLogger(AbstractVariantManager.class);
47     
48     	/**
49     	 * The instance of the framework in which this object is used.
50     	 * 
51     	 * @invariant fa != null
52     	 */
53     	protected final FA<SYM, T, N, R> fa;
54     
55     	/**
56     	 * A manager of indices that map entities to variants.
57     	 * 
58     	 * @invariant indexManager != null
59     	 */
60     	private final IIndexManager<? extends IIndex<?>, E> idxManager;
61     
62     	/**
63     	 * A map from indices to variants.
64     	 * 
65     	 * @invariant index2variant != null
66     	 */
67     	private final Map<IIndex<?>, V> index2variant = new HashMap<IIndex<?>, V>();
68     
69     	/**
70     	 * Creates a new <code>AbstractVariantManager</code> instance.
71     	 * 
72     	 * @param theAnalysis the instance of the framework in which this object is used.
73     	 * @param indexManager the manager of indices that map the entities to variants.
74     	 * @pre theAnalysis != null and indexManager != null
75     	 */
76     	AbstractVariantManager(final FA<SYM, T, N, R> theAnalysis, final IIndexManager<? extends IIndex<?>, E> indexManager) {
77     		this.fa = theAnalysis;
78     		this.idxManager = indexManager;
79     	}
80     
81     	/**
82     	 * Returns the variant corresponding to the given entity in the given context, if one exists.
83     	 * 
84     	 * @param o the entity whose variant is to be returned.
85     	 * @param context the context corresponding to which the variant is requested.
86     	 * @return the variant correponding to the entity in the given context, if one exists. <code>null</code> if none exist.
87     	 * @pre o != null and context != null
88     	 */
89     	public final V query(final E o, final Context context) {
90     		return index2variant.get(idxManager.getIndex(o, context));
91     	}
92     
93     	/**
94     	 * Resets the manager. All internal data structures are reset to enable a new session of usage.
95     	 */
96     	public void reset() {
97     		if (LOGGER.isDebugEnabled()) {
98     			LOGGER.debug("IVariant manager being reset.");
99     		}
100     		index2variant.clear();
101     		idxManager.reset();
102     	}
103     
104     	/**
105     	 * Returns the variant corresponding to the given entity in the given context. If a variant does not exist, a new one is
106     	 * created. If one exists, it shall be returned.
107     	 * 
108     	 * @param o the entity whose variant is to be returned.
109     	 * @param context the context corresponding to which the variant is requested.
110     	 * @return the variant correponding to the entity in the given context.
111     	 * @pre o != null and context != null
112     	 * @post result != null
113     	 */
114     	public final V select(final E o, final Context context) {
115     		final IIndex<?> _index = idxManager.getIndex(o, context);
116     		V _temp = null;
117     
118     		if (LOGGER.isDebugEnabled()) {
119     			LOGGER.debug("Entering - IIndex: " + _index + "\n" + o + "\n" + context);
120     		}
121     
122     		if (index2variant.containsKey(_index)) {
123     			_temp = index2variant.get(_index);
124     		} else if (!fa.getAnalyzer().isStable()) {
125     			_temp = getNewVariant(o);
126     			index2variant.put(_index, _temp);
127     			_temp.process();
128     		}
129     
130     		if (LOGGER.isDebugEnabled()) {
131     			LOGGER.debug("Exiting - IIndex");
132     		}
133     
134     		return _temp;
135     	}
136     
137     	/**
138     	 * Returns the new variant correponding to the given object. This is a template method to be provided by concrete
139     	 * implementations.
140     	 * 
141     	 * @param o the object whose corresponding variant is to be returned.
142     	 * @return the new variant corresponding to the given object.
143     	 * @pre o != null
144     	 * @post result != null
145     	 */
146     	protected abstract V getNewVariant(final E o);
147     
148     	/**
149     	 * Returns the total variants managed by this manager.
150     	 * 
151     	 * @return number of variants managed.
152     	 */
153     	protected int getVariantCount() {
154     		return index2variant.values().size();
155     	}
156     
157     	/**
158     	 * Retrieves the variants managed by this object.
159     	 * 
160     	 * @return the variants.
161     	 * @post result != null
162     	 */
163     	Collection<V> getVariants() {
164     		return index2variant.values();
165     	}
166     }
