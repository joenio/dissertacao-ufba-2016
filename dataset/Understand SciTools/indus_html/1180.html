17     package edu.ksu.cis.indus.common.graph;
18     
19     import edu.ksu.cis.indus.annotations.Immutable;
20     import edu.ksu.cis.indus.annotations.NonNull;
21     import edu.ksu.cis.indus.annotations.NonNullContainer;
22     
23     import java.util.Collection;
24     
25     /**
26      * This class extends <code>INode</code> such that the resulting node can mutated.
27      * 
28      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
29      * @author $Author: rvprasad $
30      * @version $Revision: 1.3 $ $Date: 2007/02/10 19:08:36 $
31      * @param <T> the subtype of this type.
32      */
33     public class MutableNode<T extends MutableNode<T>>
34     		extends Node<T>
35     		implements IMutableNode<T> {
36     
37     	/**
38     	 * Creates a new MutableNode object.
39     	 * 
40     	 * @param preds is the reference to the collection of predecessors.
41     	 * @param succs is the reference to the collection of successors.
42     	 */
43     	protected MutableNode(@NonNull @NonNullContainer final Collection<T> preds,
44     			@NonNull @NonNullContainer final Collection<T> succs) {
45     		super(preds, succs);
46     	}
47     
48     	/**
49     	 * {@inheritDoc}
50     	 */
51     	public final boolean addPredecessor(@Immutable @NonNull final T node) {
52     		return predecessors.add(node);
53     	}
54     
55     	/**
56     	 * {@inheritDoc}
57     	 */
58     	public final boolean addSuccessor(@Immutable @NonNull final T node) {
59     		return successors.add(node);
60     	}
61     
62     	/**
63     	 * {@inheritDoc}
64     	 */
65     	public boolean removePredecessor(@Immutable @NonNull final T node) {
66     		return predecessors.remove(node);
67     	}
68     
69     	/**
70     	 * {@inheritDoc}
71     	 */
72     	public boolean removeSuccessor(@Immutable @NonNull final T node) {
73     		return successors.remove(node);
74     	}
75     }
