119     	public static Test suite() {
120     		final TestSuite _suite = new TestSuite("Test for edu.ksu.cis.indus.staticanalyses.dependency");

131     	/**
132     	 * Retrieves the test instance to test the given analysis instance.
133     	 * 
134     	 * @param analysis is the instance whose output should be tested.
135     	 * @return the test instance.
136     	 * @pre analysis != null
137     	 */
138     	private static Test getDATestFor(final IDependencyAnalysis analysis) {
139     		Test _result = null;
140     
141     		final String _daClassName = analysis.getClass().getName();
142     		final String _daTestClassName = TEST_CLASSES_PROPERTIES.getProperty(_daClassName);
143     
144     		if (_daTestClassName == null) {
145     			LOGGER.error("How can the test class for a dependency analysis be undefined? " + _daClassName);
146     		} else {
147     			try {
148     				final Class _daTestClass = Class.forName(_daTestClassName);
149     				final IDependencyAnalysisTest _daTest = (AbstractDependencyAnalysisTest) _daTestClass.newInstance();
150     				((AbstractDependencyAnalysisTest) _daTest).setDA(analysis);
151     				_result = _daTest;
152     			} catch (final InstantiationException _e1) {
153     				LOGGER.error("Error while creating an object of type. " + _daTestClassName, _e1);
154     			} catch (final IllegalAccessException _e1) {
155     				LOGGER.error("Error while accessing the constructor of type. " + _daTestClassName, _e1);
156     			} catch (final ClassNotFoundException _e1) {
157     				LOGGER.error("Error while finding type. " + _daTestClassName, _e1);
158     			} catch (final SecurityException _e1) {
159     				LOGGER.error("Security violation while accessing the constructor of type. " + _daTestClassName, _e1);
160     			}
161     		}
162     		return _result;
163     	}

165     	/**
166     	 * Sets up the test fixture.
167     	 * 
168     	 * @param propFileName is the name of the file with the data to setup the test fixture.
169     	 * @param suite will contain new tests based on the fixture data (upon return).
170     	 * @throws IllegalArgumentException when the fixture data is invalid.
171     	 * @pre propFileName != null and suite != null
172     	 */
173     	private static void setupTests(final String propFileName, final TestSuite suite) {
174     		final Properties _props = new Properties();
175     
176     		try {
177     			_props.load(new FileInputStream(new File(propFileName)));
178     
179     			final String[] _configs = _props.getProperty("configs").split(" ");
180     			final Collection _das = new ArrayList();
181     			final IStmtGraphFactory _stmtGraphFactory = new CompleteStmtGraphFactory();
182     
183     			for (int _i = 0; _i < _configs.length; _i++) {
184     				final String _config = _configs[_i];
185     				final String _classNames = _props.getProperty(_config + ".classNames");
186     				final String _xmlTestDir = _props.getProperty(_config + IXMLBasedTest.XML_TEST_DIR_PROP_SUFFIX);
187     				final String _xmlControlDir = _props.getProperty(_config + IXMLBasedTest.XML_CONTROL_DIR_PROP_SUFFIX);
188     				final String _classpath = _props.getProperty(_config + ".classpath");
189     				final String _jimpleXMLDumpDir = _props.getProperty(_config + ".jimpleXMLDumpDir");
190     				final String _str = TestHelper.checkXMLBasedTestExecutability(_config, _xmlTestDir, _xmlControlDir);
191     
192     				try {
193     					final TestSuite _temp = new TestSuite(_config);
194     
195     					if (_str.length() > 0) {
196     						_temp.addTest(new ErringTestCase(_str));
197     						TestHelper.appendSuiteNameToTestsIn(_temp, true);
198     						suite.addTest(_temp);
199     					} else {
200     						_das.clear();
201     						_das.add(DivergenceDA.getDivergenceDA(IDependencyAnalysis.Direction.FORWARD_DIRECTION));
202     						_das.add(DivergenceDA.getDivergenceDA(IDependencyAnalysis.Direction.BACKWARD_DIRECTION));
203     						_das
204     								.add(InterProceduralDivergenceDA
205     										.getDivergenceDA(IDependencyAnalysis.Direction.FORWARD_DIRECTION));
206     						_das.add(InterProceduralDivergenceDA
207     								.getDivergenceDA(IDependencyAnalysis.Direction.BACKWARD_DIRECTION));
208     						_das.add(new NonTerminationSensitiveEntryControlDA());
209     						_das.add(new NonTerminationInsensitiveEntryControlDA());
210     						_das.add(new ExitControlDA());
211     						_das.add(new IdentifierBasedDataDA());
212     						_das.add(new InterferenceDAv1());
213     						_das.add(new InterferenceDAv2());
214     						_das.add(new InterferenceDAv3());
215     						_das.add(ReadyDAv1.getForwardReadyDA());
216     						_das.add(ReadyDAv1.getBackwardReadyDA());
217     						_das.add(ReadyDAv2.getForwardReadyDA());
218     						_das.add(ReadyDAv2.getBackwardReadyDA());
219     						_das.add(ReadyDAv3.getForwardReadyDA());
220     						_das.add(ReadyDAv3.getBackwardReadyDA());
221     						_das.add(new ReferenceBasedDataDA());
222     						_das.add(new SynchronizationDA());
223     
224     						String _ignoreDARegex = _props.getProperty("ignore.das");
225     
226     						if (_ignoreDARegex == null) {
227     							_ignoreDARegex = "^$";
228     						}
229     
230     						for (final Iterator _j = _das.iterator(); _j.hasNext();) {
231     							final Object _da = _j.next();
232     
233     							if (!Pattern.matches(_ignoreDARegex, _da.getClass().getName())) {
234     								final Test _test = getDATestFor((IDependencyAnalysis) _da);
235     
236     								if (_test != null) {
237     									_temp.addTest(_test);
238     								}
239     							}
240     						}
241     
242     						final DependencyXMLizer _xmlizer = new DependencyXMLizer();
243     
244     						for (final Iterator _j = _das.iterator(); _j.hasNext();) {
245     							final Object _da = _j.next();
246     
247     							if (!Pattern.matches(_ignoreDARegex, _da.getClass().getName())) {
248     								final XMLBasedDependencyAnalysisTest _xmlTest = new XMLBasedDependencyAnalysisTest(
249     										(IDependencyAnalysis) _da, _xmlizer);
250     								_temp.addTest(_xmlTest);
251     							}
252     						}
253     						_temp.addTestSuite(XMLBasedCallGraphTest.class);
254     						_temp.addTestSuite(OFABasedCallGraphTest.class);
255     						_temp.addTestSuite(XMLBasedOFATest.class);
256     						_temp.addTestSuite(FATest.class);
257     						TestHelper.appendSuiteNameToTestsIn(_temp, true);
258     
259     						final DependencyAnalysisTestSetup _test = new DependencyAnalysisTestSetup(_temp, _classNames,
260     								_classpath);
261     						_test.setIdGenerator(new UniqueJimpleIDGenerator());
262     						_test.setJimpleXMLDumpLocation(_jimpleXMLDumpDir);
263     						_test.setStmtGraphFactory(_stmtGraphFactory);
264     						_test.setXMLTestDir(_xmlTestDir);
265     						_test.setXMLControlDir(_xmlControlDir);
266     						suite.addTest(_test);
267     					}
268     				} catch (final IllegalArgumentException _e) {
269     					_e.printStackTrace();
270     				}
271     			}
272     		} catch (final IOException _e) {
273     			throw new IllegalArgumentException("Specified property file does not exist.");
274     		}
275     	}
