18     
19     import edu.ksu.cis.indus.common.collections.CollectionUtils;
20     
21     import java.util.Collection;
22     
23     import soot.SootMethod;
24     import soot.Value;
25     import soot.jimple.DefinitionStmt;
26     
27     /**
28      * This implementation provides program-point-relative intra-thread calling contexts based on equivalence-class based
29      * information and context coupling.
30      * 
31      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
32      * @author $Author$
33      * @version $Revision$ $Date$
34      */
35     public class DataAliasBasedCallingContextRetrieverV3
36     		extends DataAliasBasedCallingContextRetrieverV2 {
37     
38     	/**
39     	 * Creates an instance of this instance.
40     	 * 
41     	 * @param callingContextLengthLimit <i>refer to the constructor of the super class</i>.
42     	 */
43     	public DataAliasBasedCallingContextRetrieverV3(final int callingContextLengthLimit) {
44     		super(callingContextLengthLimit);
45     	}
46     
47     	/**
48     	 * {@inheritDoc}
49     	 */
50     	@Override protected boolean shouldConsiderCallerSideToken(final Collection<Object> callerSideEntities) {
51     		final DefinitionStmt _stmt = (DefinitionStmt) getInfoFor(Identifiers.SRC_ENTITY);
52     		final Value _value;
53     
54     		if (_stmt.containsArrayRef()) {
55     			_value = _stmt.getArrayRef();
56     		} else {
57     			_value = _stmt.getFieldRef();
58     		}
59     
60     		boolean _result = true;
61     		final AliasSet _as = ecba.queryAliasSetFor(_value, (SootMethod) getInfoFor(Identifiers.SRC_METHOD));
62     		if (_as != null) {
63     			final Collection<?> _o = _as.getIntraProcRefEntities();
64     			if (_o != null) {
65     				_result = CollectionUtils.containsAny(callerSideEntities, _o);
66     			}
67     		}
68     		return _result;
69     	}
70     }
