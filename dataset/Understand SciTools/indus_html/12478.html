19     import edu.ksu.cis.indus.annotations.Empty;

23     import soot.Local;
24     import soot.Value;
25     import soot.ValueBox;
26     import soot.jimple.ArrayRef;
27     import soot.jimple.EnterMonitorStmt;
28     import soot.jimple.InstanceFieldRef;
29     import soot.jimple.NewExpr;
30     import soot.jimple.NullConstant;
31     import soot.jimple.ParameterRef;
32     import soot.jimple.StaticFieldRef;
33     import soot.jimple.Stmt;
34     import soot.jimple.ThisRef;
35     
36     /**
37      * This class contains <i>jakarta commons collections</i> related predicates and transformers that are specific to Soot AST
38      * types.
39      * 
40      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
41      * @author $Author: rvprasad $
42      * @version $Revision: 1.9 $ $Date: 2007/10/09 15:59:12 $
43      */
44     public final class SootPredicatesAndTransformers {
45     
46     	/**
47     	 * A predicate used to filter <code>EnterMonitorStmt</code>.
48     	 */
49     	public static final IPredicate<Stmt> ENTER_MONITOR_STMT_PREDICATE = new InstanceOfPredicate<EnterMonitorStmt, Stmt>(
50     			EnterMonitorStmt.class);
51     
52     	/**
53     	 * This filter is used to identify AST chunks that may represent references that can escape.
54     	 */
55     	public static final IPredicate<ValueBox> ESCAPABLE_EXPR_FILTER = new IPredicate<ValueBox>() {
56     
57     		public boolean evaluate(final ValueBox object) {
58     			final Value _v = object.getValue();
59     			return _v instanceof StaticFieldRef || _v instanceof InstanceFieldRef || _v instanceof ArrayRef
60     					|| _v instanceof Local || _v instanceof ThisRef || _v instanceof ParameterRef;
61     		}
62     	};
63     
64     	/**
65     	 * A predicate used to filter statements with invoke expressions. Filter expression is
66     	 * <code>((Stmt)o).containsInvokeExpr()</code>.
67     	 */
68     	public static final IPredicate<Stmt> INVOKING_STMT_PREDICATE = new IPredicate<Stmt>() {
69     
70     		public  boolean evaluate(final Stmt object) {
71     			return object.containsInvokeExpr();
72     		}
73     	};
74     
75     	/**
76     	 * A predicate used to filter <code>EnterMonitorStmt</code>.
77     	 */
78     	public static final IPredicate<Value> NEW_EXPR_PREDICATE = new InstanceOfPredicate<NewExpr, Value>(NewExpr.class);
79     
80     	/**
81     	 * This predicate filters out <code>NullConstant</code> values.
82     	 */
83     	public static final IPredicate<Value> NULL_PREDICATE = new InstanceOfPredicate<NullConstant, Value>(NullConstant.class);
84     
85     	/**
86     	 * Creates an instance of this class.
87     	 */
88     	@Empty public SootPredicatesAndTransformers() {
89     		super();
90     	}
91     }
