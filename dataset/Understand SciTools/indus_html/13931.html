55     /**
56      * This class encapsulates the logic to process the expressions during escape analysis. Alias sets are created as required.
57      * The class relies on <code>AliasSet</code> to decide if alias set needs to be created for a type of value.
58      * <p>
59      * The arguments to any of the overridden methods cannot be <code>null</code>.
60      * </p>
61      * 
62      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
63      * @author $Author: rvprasad $
64      * @version $Revision: 1.25 $
65      */

85     	 * This indicates if the value occurs as a rhs-value or a lhs-value in an assignment statement. <code>true</code>
86     	 * indicates that it value occurs as a rhs-value in an assignment statement. <code>false</code> indicates that the value
87     	 * occurs as a lhs-value in an assignment statement. This is used to mark alias sets of primaries in access expressions in
88     	 * a manner appropriate to the analysis. For example, in side-effect analysis, the primaries of array expressions are read
89     	 * as rhs-value and are written to as lhs-value.

109     	 * Provides the alias set associated with the array element being referred. All elements in a dimension of an array are
110     	 * abstracted by a single alias set. {@inheritDoc}
111     	 * 
112     	 * @see soot.jimple.RefSwitch#caseArrayRef(soot.jimple.ArrayRef)
113     	 */
114     	@Override public void caseArrayRef(final ArrayRef v) {
115     		final boolean _temp = rhs;
116     		rhs = true;
117     		process(v.getBase());
118     		rhs = _temp;
119     
120     		final AliasSet _base = (AliasSet) getResult();
121     		final AliasSet _elt = processField(v.getType(), _base, IReadWriteInfo.ARRAY_FIELD);
122     
123     		setResult(_elt);
124     	}

163     	 * {@inheritDoc}
164     	 * 
165     	 * @see soot.jimple.JimpleValueSwitch#caseLocal(Local)

167     	@Override public void caseLocal(final Local v) {
168     		AliasSet _s = ecba.localASsCache.get(v);
169     
170     		if (_s == null) {
171     			_s = AliasSet.getASForType(v.getType());
172     
173     			if (_s != null) {
174     				ecba.localASsCache.put(v, _s);
175     			}
176     		}
177     
178     		if (_s != null && markLocals) {
179     			recordAccessInfo(_s);
180     		}
181     
182     		setResult(_s);
183     	}

244     	/**
245     	 * Creates an alias set if <code>o</code> is of type <code>Value</code>. It uses <code>AliasSet</code> to decide if
246     	 * the given type requires an alias set. If not, <code>null</code> is provided as the alias set. This is also the case
247     	 * when <code>o</code> is not of type <code>Value</code>.
248     	 * 
249     	 * @param o is a piece of IR to be processed.
250     	 */
251     	@Override public void defaultCase(final Object o) {
252     		if (o instanceof StringConstant) {
253     			final StringConstant _stringConstant = (StringConstant) o;
254     			AliasSet _r = stringConstant2aliasSet.get(_stringConstant);
255     			if (_r == null) {
256     				_r = AliasSet.createAliasSet();
257     				stringConstant2aliasSet.put(_stringConstant, _r);
258     			}
259     			setResult(_r);
260     		} else {
261     			setResult(null);
262     		}
263     	}

309     	/**
310     	 * Process the arguments of the invoke expression.
311     	 * 
312     	 * @param v is the invoke expressions containing the arguments to be processed.
313     	 * @return the list of alias sets corresponding to the arguments.
314     	 * @pre v != null
315     	 * @post result != null and result.oclIsKindOf(Sequence(AliasSet))
316     	 */
317     	private List<AliasSet> processArguments(final InvokeExpr v) {
318     		// fix up arg alias sets.
319     		final List<AliasSet> _argASs;
320     		final int _paramCount = v.getMethod().getParameterCount();
321     
322     		if (_paramCount == 0) {
323     			_argASs = Collections.emptyList();
324     		} else {
325     			_argASs = new ArrayList<AliasSet>();
326     
327     			for (int _i = 0; _i < _paramCount; _i++) {
328     				final Value _val = v.getArg(_i);
329     				AliasSet _temp = null;
330     
331     				if (EquivalenceClassBasedEscapeAnalysis.canHaveAliasSet(_val.getType())) {
332     					process(v.getArg(_i));
333     					_temp = (AliasSet) ecba.valueProcessor.getResult();
334     				}
335     
336     				_argASs.add(_temp);
337     			}
338     		}
339     		return _argASs;
340     	}

342     	/**
343     	 * Process the callees in a caller.
344     	 * 
345     	 * @param callees is the collection of methods called.
346     	 * @param caller is the calling method.
347     	 * @param primaryAliasSet is the alias set of the primary in the invocation expression.
348     	 * @param siteContext corresponding to the invocation expression.
349     	 * @throws RuntimeException when cloning fails.
350     	 * @pre callees != null and caller != null and primaryAliasSet != null and MethodContext != null
351     	 */
352     	private void processCallees(final Collection<SootMethod> callees, final SootMethod caller,
353     			final AliasSet primaryAliasSet, final MethodContext siteContext) {
354     		for (final Iterator<SootMethod> _i = callees.iterator(); _i.hasNext();) {
355     			final SootMethod _callee = _i.next();
356     			final Triple<MethodContext, Map<Local, AliasSet>, Map<CallTriple, MethodContext>> _triple = ecba
357     					.getMethodInfo(_callee);
358     
359     			assert _triple != null : "method info was null for " + _callee + " in the calling context of " + caller;
360     
361     			// retrieve the method context of the callee
362     			MethodContext _mc = _triple.getFirst();
363     			final boolean _notInSameSCC = ecba.cfgAnalysis.notInSameSCC(caller, _callee);
364     			/*
365     			 * If the caller and callee occur in different SCCs then clone the callee method context and then unify it with
366     			 * the site context. If not, unify the method context with site-context as precision will be lost any which way.
367     			 */
368     			/*
369     			 * NULL-ARGUMENT SCENARIO: This above approach has the ill effect -- if argument position p is null then this
370     			 * approach results in the unnecessary combination of alias sets for this position from various callees. This can
371     			 * be addressed during parameter alias set retrival in MethodContext.
372     			 */
373     			if (_notInSameSCC) {
374     				try {
375     					_mc = _mc.clone();
376     				} catch (final CloneNotSupportedException _e) {
377     					LOGGER.error("Hell NO!  This should not happen.", _e);
378     					throw new RuntimeException(_e);
379     				}
380     			}
381     
382     			processNotifyWaitSync(primaryAliasSet, _callee);
383     
384     			if (Util.isStartMethod(_callee)) {
385     				ecba.markMultiThreadedSystem();
386     				_mc.markAsCrossingThreadBoundary();
387     				if (_notInSameSCC) {
388     					_mc.eraseIntraThreadInterProcRefEntities();
389     				}
390     			}
391     
392     			siteContext.unifyMethodContext(_mc);
393     
394     			/*
395     			 * Ruf's analysis mandates that the allocation sites that are executed multiple times pollute escape information.
396     			 * But this is untrue, as all the data that can be shared across threads have been exposed and marked rightly so
397     			 * at allocation sites. By equivalence class-based unification, it is guaranteed that the corresponding alias set
398     			 * at the caller side is unified atleast twice in case these threads are started at different sites. In case the
399     			 * threads are started at the same site, then the processing of call-site during phase 2 (bottom-up) will ensure
400     			 * that the alias sets are unified with themselves. Hence, the program structure and the language semantics along
401     			 * with the rules above ensure that the escape information is polluted (pessimistic) only when necessary. It would
402     			 * suffice to unify the method context with it self in the case of loop enclosure as this is more semantically
403     			 * close to what happens during execution.
404     			 */
405     			if (Util.isStartMethod(_callee) && ecba.cfgAnalysis.executedMultipleTimes(ecba.context.getStmt(), caller)) {
406     				siteContext.selfUnify();
407     			}
408     		}
409     	}

411     	/**
412     	 * Processes the fields.
413     	 * 
414     	 * @param t type of the access expression.
415     	 * @param base is the alias set of the primary.
416     	 * @param fieldSig is the signature of the accessed field.
417     	 * @return the alias set for the field.
418     	 */
419     	private AliasSet processField(final Type t, final AliasSet base, final String fieldSig) {
420     		AliasSet _field = base.getASForField(fieldSig);
421     
422     		if (_field == null) {
423     			_field = AliasSet.getASForType(t);
424     
425     			if (_field != null) {
426     				base.putASForField(fieldSig, _field);
427     			}
428     		}
429     
430     		if (_field != null) {
431     			recordAccessInfo(_field);
432     		}
433     
434     		if (rhs) {
435     			base.addReadField(fieldSig);
436     		} else {
437     			base.addWrittenField(fieldSig);
438     		}
439     		return _field;
440     	}

442     	/**
443     	 * Processes invoke expressions/call-sites.
444     	 * 
445     	 * @param expr invocation expresison to be processed.
446     	 */
447     	private void processInvokeExpr(final InvokeExpr expr) {
448     		final Collection<SootMethod> _callees = new ArrayList<SootMethod>();
449     		final SootMethod _caller = ecba.context.getCurrentMethod();
450     		final SootMethod _sm = expr.getMethod();
451     
452     		// fix up "return" alias set.
453     		AliasSet _retAS = null;
454     
455     		_retAS = AliasSet.getASForType(_sm.getReturnType());
456     
457     		// fix up "primary" alias set.
458     		AliasSet _primaryAS = null;
459     
460     		if (!_sm.isStatic()) {
461     			process(((InstanceInvokeExpr) expr).getBase());
462     			_primaryAS = (AliasSet) getResult();
463     		}
464     
465     		final List<AliasSet> _argASs = processArguments(expr);
466     
467     		// create a site-context of the given expression and store it into the associated site-context cache.
468     		final MethodContext _sc = new MethodContext(_sm, _primaryAS, _argASs, _retAS, AliasSet.createAliasSet(), ecba);
469     		ecba.scCache.put(new CallTriple(_caller, ecba.context.getStmt(), expr), _sc);
470     
471     		if (expr instanceof StaticInvokeExpr) {
472     			_callees.add(_sm);
473     		} else if (expr instanceof InterfaceInvokeExpr || expr instanceof VirtualInvokeExpr
474     				|| expr instanceof SpecialInvokeExpr) {
475     			_callees.addAll(ecba.cgi.getCallees(expr, ecba.context));
476     		}
477     
478     		processCallees(_callees, _caller, _primaryAS, _sc);
479     
480     		setResult(_retAS);
481     	}

483     	/**
484     	 * Process the called method for <code>notify(), nofityAll(),</code>, and variants of <code>wait</code> methods.
485     	 * 
486     	 * @param primaryAliasSet is the alias set corresponding to the primary of the invocation expression.
487     	 * @param callee being called.
488     	 * @pre primaryAliasSet != null and callee != null
489     	 */
490     	private void processNotifyWaitSync(final AliasSet primaryAliasSet, final SootMethod callee) {
491     		if (Util.isWaitMethod(callee)) {
492     			primaryAliasSet.setWaits();
493     			primaryAliasSet.setLocked();
494     		} else if (Util.isNotifyMethod(callee)) {
495     			primaryAliasSet.setNotifies();
496     		}
497     	}

499     	/**
500     	 * Helper method to record threads in which alias is accessed.
501     	 * 
502     	 * @param as is the alias set to be marked.
503     	 * @pre as != null
504     	 */
505     	private void recordAccessInfo(final AliasSet as) {
506     		as.setAccessed();
507     
508     		if (ecba.tgi != null) {
509     			final SootMethod _currentMethod = ecba.context.getCurrentMethod();
510     			final Collection _executionThreads = ecba.tgi.getExecutionThreads(_currentMethod);
511     			if (rhs) {
512     				as.addReadThreads(_executionThreads);
513     			} else {
514     				as.addWriteThreads(_executionThreads);
515     			}
516     		}
517     	}
