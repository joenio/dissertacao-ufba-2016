18     package edu.ksu.cis.indus.interfaces;
19     
20     /**
21      * This class is used to indicate the status of an object before dispatching any methods on it.  The intent of this interface
22      * is that the users of information-providing interfaces should be able to check if the object providing that interface is
23      * in a stable state to do so.  Hence, it is adviced that information-providing interfaces inherit this interface.  On a
24      * general note,  this interface can be used in situations where a support to query status is required.
25      * 
26      * <p>
27      * A note on stability.  In most cases, active entities like analysis may provide different answers for identical queries at
28      * different times when they are active, i.e, when analysis is happening. However, while not active, it will usually provide
29      * the same answer for identical queries at different times.  Nevertheless the answer may be incorrect, but the analysis is
30      * consistent in it's answer. It is this state in which the analysis provide consistent answers that we refer to as stable
31      * state,  a state in which the external behavior of an object will be consistent.  Hence, implementation can use
32      * <code>isStable</code> to indicate/detect activeness.
33      * </p>
34      *
35      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
36      * @author $Author: rvprasad $
37      * @version $Revision: 1.7 $
38      */
39     public interface IStatus {
40     	/**
41     	 * Checks if the object is in a stable state so that it can be queried for information.
42     	 *
43     	 * @return <code>true</code> if the implementation is in a stable state; <code>false</code>, otherwise.
44     	 */
45     	boolean isStable();
46     }
