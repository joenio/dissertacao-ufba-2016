18     
19     import edu.ksu.cis.indus.tools.slicer.SlicerTool;
20     
21     /**
22      * This is an abstract implementation of slice criteria filter.
23      * 
24      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
25      * @author $Author: rvprasad $
26      * @version $Revision: 1.6 $ $Date: 2007/02/10 19:06:02 $
27      * @param <T> is the type of the input objects to the predicate.
28      */
29     public abstract class AbstractSliceCriteriaPredicate<T>
30     		implements ISliceCriteriaPredicate<T> {
31     
32     	/**
33     	 * This is the slicer tool provides the context in which filtering occurs.
34     	 */
35     	private SlicerTool<?> slicerTool;
36     
37     	/**
38     	 * {@inheritDoc}
39     	 */
40     	public void setSlicerTool(final SlicerTool<?> slicer) {
41     		slicerTool = slicer;
42     	}
43     
44     	/**
45     	 * Retrieves the value in <code>slicerTool</code>.
46     	 * 
47     	 * @return the value in <code>slicerTool</code>.
48     	 */
49     	protected final SlicerTool<?> getSlicerTool() {
50     		return slicerTool;
51     	}
52     }
