17     package edu.ksu.cis.indus.processing;
18     
19     import java.util.Iterator;
20     
21     import org.slf4j.Logger;
22     import org.slf4j.LoggerFactory;
23     
24     import soot.tagkit.Host;
25     import soot.tagkit.Tag;
26     
27     /**
28      * This class filters out classes and methods that do have a tag of the given name.
29      * 
30      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
31      * @author $Author: rvprasad $
32      * @version $Revision: 1.9 $ $Date: 2007/02/10 19:08:37 $
33      */
34     public class AntiTagBasedProcessingFilter
35     		extends TagBasedProcessingFilter {
36     
37     	/**
38     	 * The logger used by instances of this class to log messages.
39     	 */
40     	private static final Logger LOGGER = LoggerFactory.getLogger(AntiTagBasedProcessingFilter.class);
41     
42     	/**
43     	 * Creates a new TagBasedProcessingFilter object.
44     	 * 
45     	 * @param theTagName is the name of the tag used during filtering.
46     	 * @pre theTagName != null
47     	 */
48     	public AntiTagBasedProcessingFilter(final String theTagName) {
49     		super(theTagName);
50     	}
51     
52     	/**
53     	 * @see edu.ksu.cis.indus.processing.TagBasedProcessingFilter#isFilterate(soot.tagkit.Host)
54     	 */
55     	@Override protected boolean isFilterate(final Host host) {
56     		if (LOGGER.isDebugEnabled()) {
57     			LOGGER.debug("Host " + host + " is tagged:" + host.hasTag(tagName) + " [" + tagName + "]");
58     
59     			if (!host.getTags().isEmpty()) {
60     				LOGGER.debug("Host " + host + " has the following tags: ");
61     
62     				for (@SuppressWarnings("unchecked") final Iterator<Tag> _i = host.getTags().iterator(); _i.hasNext();) {
63     					final Tag _tag = _i.next();
64     
65     					if (LOGGER.isDebugEnabled()) {
66     						LOGGER.debug(_tag.getName());
67     					}
68     				}
69     			}
70     		}
71     		return !host.hasTag(tagName);
72     	}
73     }
