37     /**
38      * This class provides dependence information closure. In other words, given a dependence analysis, it provides the indirect
39      * version of it.
40      *
41      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
42      * @author $Author: rvprasad $
43      * @version $Revision: 1.18 $ $Date: 2007/02/10 19:07:02 $
44      * @param <E1> is the type of dependee object in the context of dependee-to-dependent info maintenance.
45      * @param <C1> is the type of context object in the context of dependee-to-dependent info maintenance.
46      * @param <T1> is the type of dependent object in the context of dependee-to-dependent info maintenance.
47      * @param <KE> is the type of key object in the context of dependee-to-dependent info maintenance.
48      * @param <VT> is the type of value object in the context of dependee-to-dependent info maintenance.
49      * @param <T2> is the type of dependent object in the context of dependent-to-dependee info maintenance.
50      * @param <C2> is the type of context object in the context of dependent-to-dependee info maintenance.
51      * @param <E2> is the type of dependee object in the context of dependent-to-dependee info maintenance.
52      * @param <KT> is the type of key object in the context of dependent-to-dependee info maintenance.
53      * @param <VE> is the type of value object in the context of dependent-to-dependee info maintenance.
54      */
55     final class IndirectDependenceAnalysis<T1, C1, E1, KE, VT, E2, C2, T2, KT, VE>
56     		implements IDependencyAnalysis<T1, C1, E1, E2, C2, T2> {
57     
58     	/**
59     	 * This analysis provides seed dependence information.
60     	 */
61     	private final IDependencyAnalysis<T1, C1, E1, E2, C2, T2> da;
62     
63     	/**
64     	 * This is similar to <code>dependent2dependee</code> except the direction is dependee->dependent. Hence, it is
65     	 * recommended that the subclass use this store dependence information.
66     	 *
67     	 * @invariant dependee2dependent != null
68     	 */
69     	private final Map<Pair<E2, C2>, Collection<T2>> dependee2dependent = new HashMap<Pair<E2, C2>, Collection<T2>>(Constants
70     			.getNumOfMethodsInApplication());
71     
72     	/**
73     	 * This can used to store dependent->dependee direction of dependence information. Hence, it is recommended that the
74     	 * subclass use this store dependence information.
75     	 *
76     	 * @invariant dependent2dependee != null
77     	 */
78     	private final Map<Pair<T1, C1>, Collection<E1>> dependent2dependee = new HashMap<Pair<T1, C1>, Collection<E1>>(Constants
79     			.getNumOfMethodsInApplication());
80     
81     	/**
82     	 * This retrieves dependence from the given analysis.
83     	 */
84     	private final IDependenceRetriever<T1, C1, E1, E2, C2, T2> retriever;
85     
86     	/**
87     	 * Creates an instance of this class.
88     	 *
89     	 * @param dependenceAnalysis for which indirect dependence info (or dependence closure) is to be provided.
90     	 * @param daRetriever should be used to retrieve dependence information from the given analysis.
91     	 * @pre dependenceAnalysis != null and daRetriever != null
92     	 */
93     	IndirectDependenceAnalysis(final IDependencyAnalysis<T1, C1, E1, E2, C2, T2> dependenceAnalysis,
94     			final IDependenceRetriever<T1, C1, E1, E2, C2, T2> daRetriever) {
95     		da = dependenceAnalysis;
96     		retriever = daRetriever;
97     	}

103     		// does nothing
104     	}

106     	/**
107     	 * @see edu.ksu.cis.indus.staticanalyses.interfaces.IAnalysis#doesPreProcessing()
108     	 */
109     	@Empty public boolean doesPreProcessing() {
110     		return false;
111     	}
112     
113     	/**
114     	 * @see edu.ksu.cis.indus.staticanalyses.dependency.IDependencyAnalysis#getDependees(Object, Object)
115     	 */
116     	public Collection<E1> getDependees(final T1 dependent, final C1 context) {
117     		final Pair<T1, C1> _key = new Pair<T1, C1>(dependent, context);
118     		Collection<E1> _result = dependent2dependee.get(_key);
119     
120     		if (_result == null) {
121     			_result = new ArrayList<E1>();
122     			final Collection<Pair<T1, C1>> _processed = new ArrayList<Pair<T1, C1>>();
123     			final IWorkBag<Pair<T1, C1>> _wb = new HistoryAwareFIFOWorkBag<Pair<T1, C1>>(_processed);
124     			final Collection<E1> _dependees = da.getDependees(dependent, context);
125     			_result.addAll(_dependees);
126     			_wb.addAllWork(retriever.convertToConformantDependents(_dependees, dependent, context));
127     
128     			while (_wb.hasWork()) {
129     				final Pair<T1, C1> _work = _wb.getWork();
130     				final Collection<E1> _dependees2 = retriever.getDependees(da, _work.getFirst(), _work.getSecond());
131     				_result.addAll(_dependees);
132     				_wb.addAllWork(retriever.convertToConformantDependents(_dependees2, _work.getFirst(), _work.getSecond()));
133     			}
134     			dependent2dependee.put(_key, _result);
135     		}
136     		return Collections.unmodifiableCollection(_result);
137     	}
138     
139     	/**
140     	 * @see edu.ksu.cis.indus.staticanalyses.dependency.IDependencyAnalysis#getDependents(java.lang.Object, java.lang.Object)
141     	 */
142     	public Collection<T2> getDependents(final E2 dependee, final C2 context) {
143     		final Pair<E2, C2> _key = new Pair<E2, C2>(dependee, context);
144     		Collection<T2> _result = dependee2dependent.get(_key);
145     
146     		if (_result == null) {
147     			_result = new ArrayList<T2>();
148     			final Collection<Pair<E2, C2>> _processed = new ArrayList<Pair<E2, C2>>();
149     			final IWorkBag<Pair<E2, C2>> _wb = new HistoryAwareFIFOWorkBag<Pair<E2, C2>>(_processed);
150     			final Collection<T2> _dependents = da.getDependents(dependee, context);
151     			_result.addAll(_dependents);
152     			_wb.addAllWork(retriever.convertToConformantDependees(_dependents, dependee, context));
153     
154     			while (_wb.hasWork()) {
155     				final Pair<E2, C2> _work = _wb.getWork();
156     				final Collection<T2> _dependents2 = retriever.getDependents(da, _work.getFirst(), _work.getSecond());
157     				_result.addAll(_dependents2);
158     				_wb.addAllWork(retriever.convertToConformantDependees(_dependents2, _work.getFirst(), _work.getSecond()));
159     			}
160     
161     			dependee2dependent.put(_key, _result);
162     		}
163     		return Collections.unmodifiableCollection(_result);
164     	}
