17     package edu.ksu.cis.indus.staticanalyses.dependency;
18     
19     import edu.ksu.cis.indus.common.datastructures.Pair.PairManager;
20     import edu.ksu.cis.indus.common.soot.Constants;
21     import edu.ksu.cis.indus.staticanalyses.InitializationException;
22     import edu.ksu.cis.indus.staticanalyses.interfaces.AbstractAnalysis;
23     
24     import java.util.Collection;
25     import java.util.HashMap;
26     import java.util.HashSet;
27     import java.util.Map;

29     /**
30      * This class provides generic framework and support required by analyses (DA) to calculate dependence information. It is
31      * adviced that specific analyses extend this class.
32      * <p>
33      * It is an abstract class as it does not implement the method that actually does the analysis. Also, it contains member data
34      * that are necessary to store any sort dependency information. However, it is the responsibility of the subclasses to store
35      * the data and provide the same via concrete implementation of abstract methods. It is required to call
36      * <code>initialize()</code> before any processing is triggered on the analysis.
37      * </p>
38      * 
39      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
40      * @author $Author: rvprasad $
41      * @version $Revision: 1.17 $
42      * @param <E1> is the type of dependee object in the context of dependee-to-dependent info maintenance.
43      * @param <C1> is the type of context object in the context of dependee-to-dependent info maintenance.
44      * @param <T1> is the type of dependent object in the context of dependee-to-dependent info maintenance.
45      * @param <KE> is the type of key object in the context of dependee-to-dependent info maintenance.
46      * @param <VT> is the type of value object in the context of dependee-to-dependent info maintenance.
47      * @param <T2> is the type of dependent object in the context of dependent-to-dependee info maintenance.
48      * @param <C2> is the type of context object in the context of dependent-to-dependee info maintenance.
49      * @param <E2> is the type of dependee object in the context of dependent-to-dependee info maintenance.
50      * @param <KT> is the type of key object in the context of dependent-to-dependee info maintenance.
51      * @param <VE> is the type of value object in the context of dependent-to-dependee info maintenance.
52      * @invariant doesPreProcessing() implies getPreProcessor() != null
53      * @invariant getPreProcessing() != null implies doesPreProcessing()
54      */
55     public abstract class AbstractDependencyAnalysis<T1, C1, E1, KT, VE, E2, C2, T2, KE, VT>
56     		extends AbstractAnalysis
57     		implements IDependencyAnalysis<T1, C1, E1, E2, C2, T2> {
58     
59     	/**
60     	 * The collection of dependence analysis identifiers.
61     	 */
62     	public static final Collection<Comparable<?>> IDENTIFIERS;
63     
64     	static {
65     		IDENTIFIERS = new HashSet<Comparable<?>>();
66     		IDENTIFIERS.add(DependenceSort.CONTROL_DA);
67     		IDENTIFIERS.add(DependenceSort.IDENTIFIER_BASED_DATA_DA);
68     		IDENTIFIERS.add(DependenceSort.REFERENCE_BASED_DATA_DA);
69     		IDENTIFIERS.add(DependenceSort.SYNCHRONIZATION_DA);
70     		IDENTIFIERS.add(DependenceSort.DIVERGENCE_DA);
71     		IDENTIFIERS.add(DependenceSort.INTERFERENCE_DA);
72     		IDENTIFIERS.add(DependenceSort.READY_DA);
73     	}
74     
75     	/**
76     	 * This is similar to <code>dependent2dependee</code> except the direction is dependee->dependent. Hence, it is
77     	 * recommended that the subclass use this store dependence information.
78     	 * 
79     	 * @invariant dependee2dependent != null
80     	 */
81     	protected final Map<KE, VT> dependee2dependent = new HashMap<KE, VT>(Constants.getNumOfMethodsInApplication());
82     
83     	/**
84     	 * This can used to store dependent->dependee direction of dependence information. Hence, it is recommended that the
85     	 * subclass use this store dependence information.
86     	 * 
87     	 * @invariant dependent2dependee != null
88     	 */
89     	protected final Map<KT, VE> dependent2dependee = new HashMap<KT, VE>(Constants.getNumOfMethodsInApplication());
90     
91     	/**
92     	 * This manages pair objects.
93     	 */
94     	private PairManager pairMgr;
95     
96     	/**
97     	 * The direction of the analysis.
98     	 */
99     	private final Direction theDirection;
100     
101     	/**
102     	 * Creates an instance of this class.
103     	 * 
104     	 * @param direction of the analysis.
105     	 */
106     	public AbstractDependencyAnalysis(final Direction direction) {
107     		super();
108     		theDirection = direction;
109     	}
110     
111     	/**
112     	 * @see edu.ksu.cis.indus.staticanalyses.dependency.IDependencyAnalysis#getDirection()
113     	 */
114     	public final Direction getDirection() {
115     		return theDirection;
116     	}
117     
118     	/**
119     	 * @see edu.ksu.cis.indus.staticanalyses.dependency.IDependencyAnalysis#getIndirectVersionOfDependence()
120     	 */
121     	public IDependencyAnalysis<T1, C1, E1, E2, C2, T2> getIndirectVersionOfDependence() {
122     		return new IndirectDependenceAnalysis<T1, C1, E1, KT, VE, E2, C2, T2, KE, VT>(this, getDependenceRetriever());
123     	}
124     
125     	/**
126     	 * Resets all internal data structures. General protocol is that data acquired via setup is not reset or forgotten.
127     	 * 
128     	 * @post dependent2dependee.size() == 0 and dependee2dependent.size() == 0
129     	 */
130     	@Override public void reset() {
131     		dependent2dependee.clear();
132     		dependee2dependent.clear();
133     		super.reset();
134     	}
135     
136     	/**
137     	 * Provides the dependence retriever.
138     	 * 
139     	 * @return the dependence retriever.
140     	 */
141     	protected abstract IDependenceRetriever<T1, C1, E1, E2, C2, T2> getDependenceRetriever();
142     
143     	/**
144     	 * {@inheritDoc}
145     	 * 
146     	 * @throws InitializationException when pair manager is not provided.
147     	 * @pre info.get(PairManager.ID) != null and info.get(PairManager.ID).oclIsTypeOf(PairManager)
148     	 * @see edu.ksu.cis.indus.staticanalyses.interfaces.AbstractAnalysis#setup()
149     	 */
150     	@Override protected void setup() throws InitializationException {
151     		super.setup();
152     		pairMgr = (PairManager) info.get(PairManager.ID);
153     
154     		if (pairMgr == null) {
155     			throw new InitializationException(PairManager.ID + " was not provided in the info.");
156     		}
157     	}
158     
159     	/**
160     	 * Retrieves an object that can be used key that represents the combination of <code>entity</code> and
161     	 * <code>method</code>.
162     	 * 
163     	 * @param entity to be represented.
164     	 * @param method to be represented.
165     	 * @return a key object.
166     	 * @post result != null
167     	 */
168     	Object getKeyFor(final Object entity, final Object method) {
169     		return pairMgr.getPair(entity, method);
170     	}
171     }
