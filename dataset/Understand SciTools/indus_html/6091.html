30     /**
31      * This class can be used to generated slice criteria based on statements identified by line numbers.
32      * 
33      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
34      * @author $Author$
35      * @version $Revision$ $Date$
36      */
37     public class LineNumberBasedCriteriaGenerator
38     		extends AbstractStmtBasedSliceCriteriaGenerator<Stmt> {
39     
40     	/**
41     	 * A mapping from class names to the line numbers in the class.
42     	 */
43     	private Map<String, Collection<String>> className2lineNos;
44     
45     	/**
46     	 * Creates an instance of this class.
47     	 * 
48     	 * @param props is mapping from fully qualified class names to a comma separated string of line numbers from which the
49     	 *            slice criteria are generated.
50     	 */
51     	public LineNumberBasedCriteriaGenerator(final Properties props) {
52     		className2lineNos = new HashMap<String, Collection<String>>();
53     		for (final Object _key : props.keySet()) {
54     			final String _className = (String) _key;
55     			final List<String> _lineNos = Arrays.asList(props.getProperty(_className).split(","));
56     			className2lineNos.put(_className, _lineNos);
57     		}
58     	}
59     
60     	/**
61     	 * {@inheritDoc}
62     	 */
63     	@Override protected boolean shouldConsiderStmt(final Stmt stmt) {
64     		boolean _result = false;
65     		final LineNumberTag _lnt = (LineNumberTag) stmt.getTag("LineNumberTag");
66     		if (_lnt != null) {
67     			final String _className = getProcessingMethod().getDeclaringClass().getName();
68     			final Collection<String> _lineNos = MapUtils.queryCollection(className2lineNos, _className);
69     			_result = _lineNos.contains(_lnt.toString());
70     		}
71     		return _result;
72     	}
73     }
