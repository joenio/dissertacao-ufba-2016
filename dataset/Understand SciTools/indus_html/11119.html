81     	public StaticFieldUseDefInfo() {
82     		def2usesMap = new HashMap<SootField, Map<Pair<DefinitionStmt, SootMethod>, Collection<Pair<Stmt, SootMethod>>>>();
83     		def2usesMapShadow = new FactoryBasedLazyMap<SootField, Map<Pair<DefinitionStmt, SootMethod>, Collection<Pair<Stmt, SootMethod>>>>(
84     				def2usesMap, MapUtils.<Pair<DefinitionStmt, SootMethod>, Collection<Pair<Stmt, SootMethod>>> getFactory());
85     		use2defsMap = new HashMap<SootField, Map<Pair<Stmt, SootMethod>, Collection<Pair<DefinitionStmt, SootMethod>>>>();
86     		use2defsMapShadow = new FactoryBasedLazyMap<SootField, Map<Pair<Stmt, SootMethod>, Collection<Pair<DefinitionStmt, SootMethod>>>>(
87     				use2defsMap, MapUtils.<Pair<Stmt, SootMethod>, Collection<Pair<DefinitionStmt, SootMethod>>> getFactory());
88     		unstable();
89     	}

94     	@Override public void callback(final ValueBox vBox, final Context context) {
95     		final StaticFieldRef _value = (StaticFieldRef) vBox.getValue();
96     		final Stmt _stmt = context.getStmt();
97     		final SootMethod _currentMethod = context.getCurrentMethod();
98     		final SootField _field = _value.getField();
99     
100     		if (_stmt.getUseBoxes().contains(vBox)) {
101     			final Map<Pair<Stmt, SootMethod>, Collection<Pair<DefinitionStmt, SootMethod>>> _temp = use2defsMapShadow
102     					.get(_field);
103     			_temp.put(new Pair<Stmt, SootMethod>(_stmt, _currentMethod), null);
104     		} else {
105     			final Map<Pair<DefinitionStmt, SootMethod>, Collection<Pair<Stmt, SootMethod>>> _temp = def2usesMapShadow
106     					.get(_field);
107     			_temp.put(new Pair<DefinitionStmt, SootMethod>((DefinitionStmt) _stmt, _currentMethod), null);
108     		}
109     	}

113     	 */
114     	@Override public void consolidate() {
115     		final Collection<SootField> _keySet = SetUtils.intersection(def2usesMap.keySet(), use2defsMap.keySet());
116     		final Iterator<SootField> _i = _keySet.iterator();
117     		final int _iEnd = _keySet.size();
118     
119     		for (int _iIndex = 0; _iIndex < _iEnd; _iIndex++) {
120     			final SootField _field = _i.next();
121     			final Map<Pair<DefinitionStmt, SootMethod>, Collection<Pair<Stmt, SootMethod>>> _defsite2uses = MapUtils
122     					.getMapFromMap(def2usesMap, _field);

124     					.getMapFromMap(use2defsMap, _field);
125     			final Set<Pair<Stmt, SootMethod>> _usesiteKeySet = _usesite2defs.keySet();
126     			final Set<Pair<DefinitionStmt, SootMethod>> _defsiteKeySet = _defsite2uses.keySet();
127     
128     			if (!_defsiteKeySet.isEmpty() && !_usesiteKeySet.isEmpty()) {
129     				final Iterator<Pair<Stmt, SootMethod>> _j = _usesiteKeySet.iterator();
130     				final int _jEnd = _usesiteKeySet.size();
131     
132     				for (int _jIndex = 0; _jIndex < _jEnd; _jIndex++) {
133     					_usesite2defs.put(_j.next(), _defsiteKeySet);
134     				}
135     
136     				final Iterator<Pair<DefinitionStmt, SootMethod>> _k = _defsiteKeySet.iterator();
137     				final int _kEnd = _defsiteKeySet.size();
138     
139     				for (int _kIndex = 0; _kIndex < _kEnd; _kIndex++) {
140     					_defsite2uses.put(_k.next(), _usesiteKeySet);
141     				}
142     			}
143     		}
144     	}
