17     package edu.ksu.cis.indus.interfaces;
18     
19     import edu.ksu.cis.indus.annotations.NonNull;
20     import edu.ksu.cis.indus.annotations.NonNullContainer;
21     import edu.ksu.cis.indus.common.datastructures.Triple;
22     import edu.ksu.cis.indus.common.graph.IObjectDirectedGraph;
23     import edu.ksu.cis.indus.common.graph.IObjectNode;
24     
25     import java.util.Collection;
26     import java.util.Map;
27     
28     import soot.SootMethod;
29     import soot.jimple.EnterMonitorStmt;
30     import soot.jimple.ExitMonitorStmt;
31     import soot.jimple.MonitorStmt;
32     import soot.jimple.Stmt;
33     
34     /**
35      * This interface provides the information pertaining to Java monitors in the analyzed system.
36      * <p>
37      * Subtypes of this class have to return the constant <code>ID</code> defined in this class as a result of
38      * <code>getId</code>.
39      * </p>
40      * 
41      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
42      * @author $Author: rvprasad $
43      * @version $Revision: 1.29 $
44      * @param <N> is the type of the node of the graph used to store the monitor information.
45      */
46     public interface IMonitorInfo<N extends IObjectNode<N, Triple<EnterMonitorStmt, ExitMonitorStmt, SootMethod>>>
47     		extends IStatus, IIdentification {
48     
49     	/**
50     	 * This is the interface to monitor graphs.
51     	 * 
52     	 * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
53     	 * @author $Author: rvprasad $
54     	 * @version $Revision: 1.29 $
55     	 * @param <N> <i>refer to the type parameter of the same name in the super class.</i>
56     	 */
57     	interface IMonitorGraph<N extends IObjectNode<N, Triple<EnterMonitorStmt, ExitMonitorStmt, SootMethod>>>
58     			extends IObjectDirectedGraph<N, Triple<EnterMonitorStmt, ExitMonitorStmt, SootMethod>> {
59     
60     		/**
61     		 * Retrieves the statements enclosed by the given monitor triple, both intra and interprocedurally.
62     		 * 
63     		 * @param monitorTriple describes the monitor of interest.
64     		 * @param transitive <code>true</code> indicates transitive closure is required; <code>false</code>, otherwise.
65     		 * @return a map from a method to the statements of that method that are enclosed by the given monitor.
66     		 * @pre monitorTriple.getThird() != null
67     		 * @post (not transitive) implies result.size() = 1
68     		 */
69     		@NonNull Map<SootMethod, Collection<Stmt>> getInterProcedurallyEnclosedStmts(
70     				@NonNull final Triple<EnterMonitorStmt, ExitMonitorStmt, SootMethod> monitorTriple, final boolean transitive);
71     
72     		/**
73     		 * Retrieves the monitor triples for monitors enclosing the given statement in the given method, both intra and
74     		 * interprocedurally.
75     		 * 
76     		 * @param stmt obviously.
77     		 * @param method in which the monitor occurs.
78     		 * @param transitive <code>true</code> indicates transitive closure is required; <code>false</code>, otherwise.
79     		 * @return a collection of triples
80     		 * @post (not transitive) implies result.size() = 1
81     		 */
82     		@NonNull Map<SootMethod, Collection<Triple<EnterMonitorStmt, ExitMonitorStmt, SootMethod>>> getInterProcedurallyEnclosingMonitorTriples(
83     				@NonNull final Stmt stmt, @NonNull final SootMethod method, final boolean transitive);
84     	}
85     
86     	/**
87     	 * The id of this interface.
88     	 */
89     	Comparable<String> ID = "Monitor Information";
90     
91     	/**
92     	 * Retrieves the statements enclosed by the given monitor triple. Only the statements occurring in the method in which the
93     	 * monitor occurs are returned.
94     	 * 
95     	 * @param monitorTriple describes the monitor of interest.
96     	 * @param transitive <code>true</code> indicates transitive closure is required; <code>false</code>, otherwise.
97     	 * @return a collection of statements.
98     	 * @pre monitorTriple.getThird() != null
99     	 */
100     	@NonNull @NonNullContainer Collection<Stmt> getEnclosedStmts(@NonNull final Triple<EnterMonitorStmt, ExitMonitorStmt, SootMethod> monitorTriple,
101     			final boolean transitive);
102     
103     	/**
104     	 * Retrieves the monitor statements enclosing the given statement in the given method. Only the monitors occurring in the
105     	 * method in which the statement occurs are returned.
106     	 * 
107     	 * @param stmt obviously.
108     	 * @param method in which the monitor occurs.
109     	 * @param transitive <code>true</code> indicates transitive closure is required; <code>false</code>, otherwise.
110     	 * @return a collection of statements
111     	 */
112     	@NonNull @NonNullContainer Collection<MonitorStmt> getEnclosingMonitorStmts(@NonNull final Stmt stmt, @NonNull final SootMethod method, final boolean transitive);
113     
114     	/**
115     	 * Retrieves the monitor triples for monitors enclosing the given statement in the given method. Only the monitors
116     	 * occurring in the method in which the statement occurs are returned.
117     	 * 
118     	 * @param stmt obviously.
119     	 * @param method in which the monitor occurs.
120     	 * @param transitive <code>true</code> indicates transitive closure is required; <code>false</code>, otherwise.
121     	 * @return a collection of triples
122     	 */
123     	@NonNull @NonNullContainer Collection<Triple<EnterMonitorStmt, ExitMonitorStmt, SootMethod>> getEnclosingMonitorTriples(@NonNull final Stmt stmt,
124     			@NonNull final SootMethod method, final boolean transitive);
125     
126     	/**
127     	 * Retrieves the monitor graph based on the shape of the call graph and the monitors in the method. Each monitor triple is
128     	 * represented as a node. An outgoing edges indicates that the monitor represented by the destination node is reachable
129     	 * from within the monitor (it is directly nested or nested in a method reachable via a call in the monitor) in the source
130     	 * node.
131     	 * 
132     	 * @param callgraphInfo to be used to generate an interprocedural graph. If this parameter is <code>null</code>,
133     	 *            intraprocedural monitor graph is generated.
134     	 * @return a graph
135     	 */
136     	@NonNull IMonitorGraph<N> getMonitorGraph(final ICallGraphInfo callgraphInfo);
137     
138     	/**
139     	 * Returns a collection of <code>Triple</code>s of <code>EnterMonitorStmt</code>, <code>ExitMonitorStmt</code>,
140     	 * and <code>SootMethod</code> in the system. The third element is the method in which the monitor occurs. In case the
141     	 * first and the second element of the triple are <code>null</code> then this means the method is a synchronized.
142     	 * 
143     	 * @return collection of monitors in the analyzed system.
144     	 * @post result->forall(o | o.getThird() ! = null)
145     	 */
146     	@NonNull @NonNullContainer Collection<Triple<EnterMonitorStmt, ExitMonitorStmt, SootMethod>> getMonitorTriples();
147     
148     	/**
149     	 * Returns a collection of <code>Triple</code>s of <code>EnterMonitorStmt</code>, <code>ExitMonitorStmt</code>,
150     	 * and <code>SootMethod</code> corresponding to monitor represented by <code>monitorStmt</code> in <code>method</code>.
151     	 * 
152     	 * @param monitorStmt obviously.
153     	 * @param method in which monitorStmt occurs.
154     	 * @return collection of monitors in the analyzed system.
155     	 * @post result->forall(o | o.getThird().equals(method))
156     	 * @post result->forall(o | o.getThird() ! = null)
157     	 */
158     	@NonNull @NonNullContainer Collection<Triple<EnterMonitorStmt, ExitMonitorStmt, SootMethod>> getMonitorTriplesFor(final MonitorStmt monitorStmt,
159     			@NonNull final SootMethod method);
160     
161     	/**
162     	 * Returns a collection of <code>Triple</code>s of <code>EnterMonitorStmt</code>, <code>ExitMonitorStmt</code>,
163     	 * and <code>SootMethod</code> corresponding to the monitors in <code>method</code>. The third element is the method
164     	 * in which the monitor occurs. In case the first and the second element of the triple are <code>null</code> then this
165     	 * means the method is a synchronized.
166     	 * 
167     	 * @param method in which the monitors occur.
168     	 * @return collection of monitors in the analyzed system.
169     	 * @post result->forall(o | o.getThird().equals(method))
170     	 * @post result->forall(o | o.getThird() ! = null)
171     	 */
172     	@NonNull @NonNullContainer Collection<Triple<EnterMonitorStmt, ExitMonitorStmt, SootMethod>> getMonitorTriplesIn(@NonNull final SootMethod method);
173     
174     	/**
175     	 * Retrieves all the monitor triples corresponding to the given monitor.
176     	 * 
177     	 * @param monitor of interest
178     	 * @return the collection of monitor triples
179     	 */
180     	@NonNull @NonNullContainer Collection<Triple<EnterMonitorStmt, ExitMonitorStmt, SootMethod>> getMonitorTriplesOf(
181     			@NonNull final Triple<EnterMonitorStmt, ExitMonitorStmt, SootMethod> monitor);
182     
183     	/**
184     	 * Retrieves the statements that form the given monitor.
185     	 * 
186     	 * @param monitor of interest.
187     	 * @return a collection of statements.
188     	 * @post monitor.getFirst() = null implies result.isEmpty()
189     	 */
190     	@NonNull @NonNullContainer Collection<? extends MonitorStmt> getStmtsOfMonitor(@NonNull final Triple<EnterMonitorStmt, ExitMonitorStmt, SootMethod> monitor);
191     
192     	/**
193     	 * Retreives the statements of the given method not enclosed by a monitor in that method.
194     	 * 
195     	 * @param method of interest.
196     	 * @return the collection of statements.
197     	 */
198     	@NonNull @NonNullContainer Collection<Stmt> getUnenclosedStmtsOf(@NonNull final SootMethod method);
199     }
