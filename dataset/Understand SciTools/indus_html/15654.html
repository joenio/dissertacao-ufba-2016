36     /**
37      * This class provides data dependence information which considers references. Hence, it considers the effects of aliasing. It
38      * is an adapter for an interprocedural use-def analysis which considers the effects of aliasing. It can be configured to
39      * provide dependence based on static field references.
40      * 
41      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
42      * @author $Author: rvprasad $
43      * @version $Revision: 1.42 $
44      */
45     public class ReferenceBasedDataDA
46     		extends
47     		AbstractDependencyAnalysis<AssignStmt, SootMethod, Pair<AssignStmt, SootMethod>, Object, Map<Pair<AssignStmt, SootMethod>, Collection<Pair<AssignStmt, SootMethod>>>, AssignStmt, SootMethod, Pair<AssignStmt, SootMethod>, Object, Map<Pair<AssignStmt, SootMethod>, Collection<Pair<AssignStmt, SootMethod>>>> {
48     
49     	/**
50     	 * This predicate can be used to check if an object of this class type.
51     	 */
52     	public static final IPredicate<IDependencyAnalysis<?, ?, ?, ?, ?, ?>> INSTANCEOF_PREDICATE = new InstanceOfPredicate<ReferenceBasedDataDA, IDependencyAnalysis<?, ?, ?, ?, ?, ?>>(
53     			ReferenceBasedDataDA.class);
54     
55     	/**
56     	 * The logger used by instances of this class to log messages.
57     	 */
58     	private static final Logger LOGGER = LoggerFactory.getLogger(ReferenceBasedDataDA.class);
59     
60     	/**
61     	 * This provides inter-procedural use-def information which considers the effects of aliasing.
62     	 */
63     	private IUseDefInfo<Pair<AssignStmt, SootMethod>, Pair<AssignStmt, SootMethod>> aliasedUD;
64     
65     	/**
66     	 * This provides static field use-def information.
67     	 */
68     	private IUseDefInfo<Pair<AssignStmt, SootMethod>, Pair<AssignStmt, SootMethod>> staticFieldRefUD;
69     
70     	/**
71     	 * Creates an instance of this class.
72     	 */
73     	public ReferenceBasedDataDA() {
74     		super(Direction.BI_DIRECTIONAL);
75     	}
76     
77     	/**
78     	 * @see edu.ksu.cis.indus.staticanalyses.interfaces.AbstractAnalysis#analyze()
79     	 */
80     	@Override public void analyze() {
81     		if (LOGGER.isInfoEnabled()) {
82     			LOGGER.info("BEGIN: Reference Based Data Dependence processing");
83     		}
84     
85     		if (LOGGER.isDebugEnabled()) {
86     			LOGGER.debug("ReferenceBasedDataDA.analyze() - " + toString());
87     		}
88     
89     		if (aliasedUD.isStable()) {
90     			stable();
91     		} else {
92     			unstable();
93     		}
94     
95     		if (LOGGER.isInfoEnabled()) {
96     			LOGGER.info("END: Reference Based Data Dependence processing");
97     		}
98     	}
99     
100     	/**
101     	 * Return the statements on which field/array access in <code>stmt</code> in <code>method</code> depends on.
102     	 * 
103     	 * @param stmt in which aliased data is read.
104     	 * @param method in which <code>stmt</code> occurs.
105     	 * @return a collection of statements which affect the data being read in <code>stmt</code>.
106     	 * @post result->forall(o | o.getFirst().getLeftOf().oclIsKindOf(FieldRef) or
107     	 *       o.getFirst().getLeftOf().oclIsKindOf(ArrayRef))
108     	 * @see edu.ksu.cis.indus.staticanalyses.dependency.AbstractDependencyAnalysis#getDependees(java.lang.Object,
109     	 *      java.lang.Object)
110     	 */
111     	public Collection<Pair<AssignStmt, SootMethod>> getDependees(final AssignStmt stmt, final SootMethod method) {
112     		final Collection<Pair<AssignStmt, SootMethod>> _result;
113     
114     		if (stmt.containsArrayRef() || stmt.containsFieldRef()) {
115     			if (staticFieldRefUD != null) {
116     				_result = SetUtils.union(aliasedUD.getDefs(stmt, method), staticFieldRefUD.getDefs(stmt, method));
117     			} else {
118     				_result = aliasedUD.getDefs(stmt, method);
119     			}
120     		} else {
121     			_result = Collections.emptyList();
122     		}
123     
124     		return _result;
125     	}
126     
127     	/**
128     	 * Return the statements which depend on the field/array access in <code>stmt</code> in <code>method</code>.
129     	 * 
130     	 * @param stmt in which aliased data is written.
131     	 * @param method in which <code>stmt</code> occurs.
132     	 * @return a collection of statements which are affectted by the data write in <code>stmt</code>.
133     	 * @post result->forall(o | o.getFirst().getRightOp().oclIsKindOf(FieldRef) or
134     	 *       o.getFirst().getRightOp().oclIsKindOf(ArrayRef))
135     	 * @see edu.ksu.cis.indus.staticanalyses.dependency.AbstractDependencyAnalysis#getDependents(java.lang.Object,
136     	 *      java.lang.Object)
137     	 */
138     	public Collection<Pair<AssignStmt, SootMethod>> getDependents(final AssignStmt stmt, final SootMethod method) {
139     		final Collection<Pair<AssignStmt, SootMethod>> _result;
140     
141     		if (stmt.containsArrayRef() || stmt.containsFieldRef()) {
142     			if (staticFieldRefUD != null) {
143     				_result = SetUtils.union(aliasedUD.getUses(stmt, method), staticFieldRefUD.getUses(stmt, method));
144     			} else {
145     				_result = aliasedUD.getUses(stmt, method);
146     			}
147     		} else {
148     			_result = Collections.emptyList();
149     		}
150     
151     		return _result;
152     	}
153     
154     	/**
155     	 * @see edu.ksu.cis.indus.staticanalyses.dependency.IDependencyAnalysis#getIds()
156     	 */
157     	public Collection<IDependencyAnalysis.DependenceSort> getIds() {
158     		return Collections.singleton(IDependencyAnalysis.DependenceSort.REFERENCE_BASED_DATA_DA);
159     	}
160

163     	/**
164     	 * Returns a stringized representation of this analysis. The representation includes the results of the analysis.
165     	 * 
166     	 * @return a stringized representation of this object.
167     	 */
168     	@Override public String toString() {
169     		return aliasedUD + "\n" + staticFieldRefUD;
170     	}
171     
172     	// /CLOVER:ON
173     
174     	/**
175     	 * @see edu.ksu.cis.indus.staticanalyses.dependency.AbstractDependencyAnalysis#getDependenceRetriever()
176     	 */
177     	@Override protected IDependenceRetriever<AssignStmt, SootMethod, Pair<AssignStmt, SootMethod>, AssignStmt, SootMethod, Pair<AssignStmt, SootMethod>> getDependenceRetriever() {
178     		return new PairRetriever<AssignStmt, SootMethod, AssignStmt, AssignStmt, SootMethod, AssignStmt>();
179     	}
180     
181     	/**
182     	 * Extracts information provided by environment at initialization time. The user can configure this analysis to include
183     	 * static field reference based dependence information by passing in an implementation of <code>IUseDefInfo</code>
184     	 * implementation mapped to <code>IUseDefInfo.GLOBAL_USE_DEF_ID</code> constant in the information map.
185     	 * 
186     	 * @throws InitializationException if an implementation that provides aliased interprocedural use-def information is not
187     	 *             provided.
188     	 * @pre info.get(IUseDefInfo.ALIASED_USE_DEF_ID) != null
189     	 * @see edu.ksu.cis.indus.staticanalyses.interfaces.AbstractAnalysis#setup()
190     	 */
191     	@Override protected void setup() throws InitializationException {
192     		super.setup();
193     
194     		aliasedUD = (IUseDefInfo) info.get(IUseDefInfo.ALIASED_USE_DEF_ID);
195     
196     		if (aliasedUD == null) {
197     			throw new InitializationException(IUseDefInfo.ALIASED_USE_DEF_ID + " was not provided.");
198     		}
199     
200     		staticFieldRefUD = (IUseDefInfo) info.get(IUseDefInfo.GLOBAL_USE_DEF_ID);
201     
202     		if (staticFieldRefUD == null) {
203     			LOGGER.info(IUseDefInfo.GLOBAL_USE_DEF_ID + " was not provided.  Hence, static field reference based dependence"
204     					+ " info will not be provided.");
205     		}
206     	}
207     }
