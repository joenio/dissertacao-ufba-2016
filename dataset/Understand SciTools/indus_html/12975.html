18     package edu.ksu.cis.indus.xmlizer;
19     
20     import java.util.Map;
21     
22     
23     /**
24      * This is the interface for xmlizing jimple-based systems.
25      *
26      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
27      * @author $Author: rvprasad $
28      * @version $Revision: 1.10 $
29      */
30     public interface IXMLizer {
31     	/**
32     	 * Retrieves the a name based on the given name for the file into which the xml data will be written into.
33     	 *
34     	 * @param basename to be considered when coming up with the file name.
35     	 *
36     	 * @return the name of the file.
37     	 *
38     	 * @pre basename != null
39     	 * @post result != null
40     	 */
41     	String getFileName(final String basename);
42     
43     	/**
44     	 * Set the xml id generator to be used in xml data generation.
45     	 *
46     	 * @param generator generates the id used in xml data.
47     	 *
48     	 * @pre generator != null
49     	 */
50     	void setGenerator(final IJimpleIDGenerator generator);
51     
52     	/**
53     	 * Retrieves the xml id generator used by this object.
54     	 *
55     	 * @return the xml id generator.
56     	 */
57     	IJimpleIDGenerator getIdGenerator();
58     
59     	/**
60     	 * Set the directory into which xml data should be dumped.
61     	 *
62     	 * @param xmlOutputDir is the directory into which xml data should be dumped.
63     	 *
64     	 * @pre xmlOutputDir != null
65     	 */
66     	void setXmlOutputDir(final String xmlOutputDir);
67     
68     	/**
69     	 * Retrieves the directory into which xml data will be dumped into.
70     	 *
71     	 * @return the directory into which xml data will be dumped.
72     	 */
73     	String getXmlOutputDir();
74     
75     	/**
76     	 * Writes information in XML form.
77     	 *
78     	 * @param info is a map in which information required for xmlization will be provided.
79     	 *
80     	 * @pre info != null
81     	 */
82     	void writeXML(final Map info);
83     }
