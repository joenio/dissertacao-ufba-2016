491     	 * Slices the system provided at initialization for the initialized criteria to generate the given type of slice..
492     	 */
493     	public void slice() {
494     		workbag.addAllWorkNoDuplicates(criteria);
495     
496     		boolean _flag = true;
497     
498     		while (_flag && activePart.canProceed()) {
499     
500     			while (workbag.hasWork() && activePart.canProceed()) {
501     				final Object _work = workbag.getWork();
502     
503     				if (LOGGER.isDebugEnabled()) {
504     					LOGGER.debug("BEGIN - Processing criterion - " + _work);
505     				}
506     
507     				if (_work instanceof ExprLevelSliceCriterion) {
508     					final ExprLevelSliceCriterion _sliceExpr = (ExprLevelSliceCriterion) _work;
509     					final SootMethod _sm = _sliceExpr.getOccurringMethod();
510     
511     					if (sliceScope == null || sliceScope.isInScope(_sm, system)) {
512     						callStackCache = _sliceExpr.getCallStack();
513     						transformAndGenerateNewCriteriaForExpr(_sliceExpr);
514     					}
515     				} else if (_work instanceof StmtLevelSliceCriterion) {
516     					final StmtLevelSliceCriterion _sliceStmt = (StmtLevelSliceCriterion) _work;
517     					final SootMethod _sm = _sliceStmt.getOccurringMethod();
518     
519     					if (sliceScope == null || sliceScope.isInScope(_sm, system)) {
520     						final Stmt _stmt = (Stmt) _sliceStmt.getCriterion();
521     						callStackCache = _sliceStmt.getCallStack();
522     						transformAndGenerateNewCriteriaForStmt(_stmt, _sm, _sliceStmt.isConsiderExecution());
523     					}
524     				} else if (_work instanceof MethodLevelSliceCriterion) {
525     					final MethodLevelSliceCriterion _sliceMethod = (MethodLevelSliceCriterion) _work;
526     					final SootMethod _sm = _sliceMethod.getOccurringMethod();
527     
528     					if (sliceScope == null || sliceScope.isInScope(_sm, system)) {
529     						callStackCache = _sliceMethod.getCallStack();
530     						transformAndGenerateNewCriteriaForMethod(_sm);
531     					}
532     				}
533     
534     				if (LOGGER.isDebugEnabled()) {
535     					LOGGER.debug("END - Processing criterion - " + _work);
536     				}
537     			}
538     			_flag = directionSensitiveInfo.continueProcessing();
539     		}
540     
541     		if (activePart.canProceed()) {
542     			collector.completeSlicing();
543     		}
544     	}

550     	 * @pre method != null

558     			includeMethodAndDeclaringClassInSlice(method);
559     		}

562     			if (ifInsideContext()) {
563     				if (callStackCache.peek() != null) {
564     					final CallTriple _top = callStackCache.pop();
565     					final SootMethod _caller = _top.getMethod();
566     					final Stmt _stmt = _top.getStmt();
567     					directionSensitiveInfo.generateCriteriaForTheCallToMethod(method, _caller, _stmt);
568     					callStackCache.push(_top);
569     				}
570     			} else {
571     				markAsCollectedAllInvocationSites(method);
572     
573     				for (final Iterator<CallTriple> _i = cgi.getCallers(method).iterator(); _i.hasNext();) {
574     					final CallTriple _ctrp = _i.next();
575     					final SootMethod _caller = _ctrp.getMethod();
576     					final Stmt _stmt = _ctrp.getStmt();
577     					directionSensitiveInfo.generateCriteriaForTheCallToMethod(method, _caller, _stmt);
578     				}
579     			}
580     		}
581     
582     		if (LOGGER.isDebugEnabled()) {
583     			LOGGER.debug("END: Generating criteria for call-sites (callee-caller)");
584     		}
