53     /**
54      * The variant that represents a method implementation. It maintains variant specific information about local variables and
55      * the AST nodes in associated method. It also maintains information about the parameters, this variable, and return values,
56      * if any are present.
57      * 
58      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
59      * @version $Revision: 1.17 $ $Name:  $
60      * @param <T>  is the type of the token set object.
61      */
62     class MethodVariant<T extends ITokens<T, Value>>
63     		extends AbstractMethodVariant<Value, T, OFAFGNode<T>, Type> {
64     
65     	/**
66     	 * The logger used by instances of this class to log messages.
67     	 */
68     	private static final Logger LOGGER = LoggerFactory.getLogger(MethodVariant.class);
69     
70     	/**
71     	 * The statement graph to use to retrieve method bodies.
72     	 */
73     	private IStmtGraphFactory<?> stmtGraphFactory;
74     
75     	/**
76     	 * Creates a new <code>MethodVariant</code> instance. This will not process the statements of this method. That is
77     	 * accomplished via call to <code>process()</code>. This will also mark the field with the flow analysis tag.
78     	 * 
79     	 * @param sm the method represented by this variant. This parameter cannot be <code>null</code>.
80     	 * @param astVariantManager the manager of flow graph nodes corresponding to the AST nodes of<code>sm</code>. This
81     	 *            parameter cannot be <code>null</code>.
82     	 * @param theFA the instance of <code>FA</code> which was responsible for the creation of this variant. This parameter
83     	 *            cannot be <code>null</code>.
84     	 * @param factory provides the CFGs.
85     	 * @pre sm != null and astVariantManager != null and theFA != null and factory != null
86     	 */
87     	protected MethodVariant(final SootMethod sm, final IVariantManager<ValuedVariant<OFAFGNode<T>>, Value> astVariantManager,
88     			final FA<Value, T, OFAFGNode<T>, Type> theFA, final IStmtGraphFactory<?> factory) {
89     		super(sm, astVariantManager, theFA);
90     
91     		if (LOGGER.isDebugEnabled()) {
92     			LOGGER.debug("BEGIN: preprocessing of " + sm);
93     		}
94     
95     		/*
96     		 * NOTE: This is required to filter out values which are descendents of a higher common type but which are
97     		 * incompatible. An example is all objects entering run() site will have a run() method defined. However, it is false
98     		 * to assume that all such objects can be considered as receivers for all run() implementations plugged into the run()
99     		 * site.
100     		 */
101     		final ITokenManager<T, Value, Type> _tokenMgr = fa.getTokenManager();
102     
103     		if (thisVar != null) {
104     			final RefType _sootType = sm.getDeclaringClass().getType();
105     			setFilterOfBasedOn(thisVar, _sootType, _tokenMgr);
106     		}
107     
108     		// We also want to use retrieve acceptable values from other interfacial data entities.
109     		if (returnVar != null) {
110     			setFilterOfBasedOn(returnVar, sm.getReturnType(), _tokenMgr);
111     		}
112     
113     		for (int _i = parameters.size() - 1; _i >= 0; _i--) {
114     			final OFAFGNode<T> _pNode = parameters.get(_i);
115     
116     			if (_pNode != null) {
117     				setFilterOfBasedOn(_pNode, sm.getParameterType(_i), _tokenMgr);
118     			}
119     		}
120     
121     		setFilterOfBasedOn(thrownNode, this.fa.getClass("java.lang.Throwable").getType(), _tokenMgr);
122     
123     		stmtGraphFactory = factory;
124     
125     		if (LOGGER.isDebugEnabled()) {
126     			LOGGER.debug("END: preprocessed " + sm);
127     		}
128     	}
129     
130     	/**
131     	 * Sets the out filter based on the given type for the given node.
132     	 * 
133     	 * @param <T>  is the type of the token set object.
134     	 * @param node of interest.
135     	 * @param type for the filter.
136     	 * @param tokenMgr used in the creation of the type-based filter.
137     	 * @pre node != null and type != null and tokenMgr != null
138     	 */
139     	static <T extends ITokens<T, Value>> void setFilterOfBasedOn(final OFAFGNode<T> node, final Type type,
140     			final ITokenManager<T, Value, Type> tokenMgr) {
141     		if (node != null) {
142     			final IType _baseType = tokenMgr.getTypeManager().getTokenTypeForRepType(type);
143     			final ITokenFilter<T, Value> _baseFilter = tokenMgr.getTypeBasedFilter(_baseType);
144     			node.setFilter(_baseFilter);
145     		}
146     	}
147     
148     	/**
149     	 * Processes the body of the method implementation associated with this variant.
150     	 */
151     	public void process() {
152     		if (LOGGER.isDebugEnabled()) {
153     			LOGGER.debug("BEGIN: processing of " + method);
154     		}
155     
156     		// We assume the user has closed the system.
157     		if (method.isConcrete()) {
158     			final JimpleBody _jb = (JimpleBody) stmtGraphFactory.getStmtGraph(method).getBody();
159     			final List<Stmt> _stmtList = new ArrayList<Stmt>(_jb.getUnits());
160     
161     			for (final Iterator<Stmt> _i = _stmtList.iterator(); _i.hasNext();) {
162     				stmt.process(_i.next());
163     			}
164     
165     			processBody(_jb, _stmtList);
166     		} else {
167     			if (LOGGER.isDebugEnabled()) {
168     				LOGGER.debug(method + " is not a concrete method. Hence, it's body could not be retrieved.");
169     			}
170     		}
171     
172     		stmtGraphFactory = null;
173     
174     		if (LOGGER.isDebugEnabled()) {
175     			LOGGER.debug("END: processing of " + method);
176     		}
177     	}

180     	 * @see edu.ksu.cis.indus.staticanalyses.flow.AbstractMethodVariant#shouldConsider(soot.Type)
181     	 */
182     	@Override protected boolean shouldConsider(final Type type) {
183     		return Util.isReferenceType(type);
184     	}

186     	/**
187     	 * Connects the nodes corresponding the exceptions thrown (throw and method invocations) in the body to the nodd
188     	 * corresponding to the expression thrown by the method.
189     	 * 
190     	 * @param body of the method.
191     	 * @param stmtList is the list of statements that make up the body.
192     	 * @pre body != null and stmtList != null
193     	 */
194     	private void connectThrowNodesToThrownNode(final JimpleBody body, final List<Stmt> stmtList) {
195     		final Context _ctxt = new Context();
196     		_ctxt.setRootMethod(method);
197     
198     		final Iterator<Stmt> _j = stmtList.iterator();
199     		final int _jEnd = stmtList.size();
200     
201     		for (int _jIndex = 0; _jIndex < _jEnd; _jIndex++) {
202     			final Stmt _stmt = _j.next();
203     			_ctxt.setStmt(_stmt);
204     
205     			if (_stmt instanceof ThrowStmt) {
206     				final Collection<Trap> _t = TrapManager.getTrapsAt(_stmt, body);
207     				final ThrowStmt _throwStmt = (ThrowStmt) _stmt;
208     				final SootClass _exp = ((RefType) _throwStmt.getOp().getType()).getSootClass();
209     				boolean _exceptionIsUncaught = true;
210     				for (final Iterator<Trap> _i = _t.iterator(); _i.hasNext() && _exceptionIsUncaught;) {
211     					final Trap _trap = _i.next();
212     					final SootClass _sc = _trap.getException();
213     					_exceptionIsUncaught = !_exp.equals(_sc) && !Util.isDescendentOf(_exp, _sc);
214     				}
215     
216     				if (_t.isEmpty() || _exceptionIsUncaught) {
217     					_ctxt.setProgramPoint(_throwStmt.getOpBox());
218     					queryASTNode(_throwStmt.getOp(), _ctxt).addSucc(thrownNode);
219     				}
220     			} else if (_stmt.containsInvokeExpr()) {
221     				_ctxt.setProgramPoint(_stmt.getInvokeExprBox());
222     				queryThrowNode(_stmt.getInvokeExpr(), _ctxt).addSucc(thrownNode);
223     			}
224     		}
225     	}
226     
227     	/**
228     	 * Process the body.
229     	 * 
230     	 * @param body to be processed.
231     	 * @param stmtList is the list of statements that make up the body.
232     	 * @pre body != null and stmtList != null
233     	 */
234     	private void processBody(final JimpleBody body, final List<Stmt> stmtList) {
235     		final Collection<Stmt> _caught = new HashSet<Stmt>();
236     		final Context _exprCtxt = new Context();
237     		final Context _catchCtxt = new Context();
238     
239     		_exprCtxt.setRootMethod(method);
240     		_catchCtxt.setRootMethod(method);
241     
242     		for (final Iterator<Trap> _i = body.getTraps().iterator(); _i.hasNext();) {
243     			final Trap _trap = _i.next();
244     			final Stmt _begin = (Stmt) _trap.getBeginUnit();
245     			final Stmt _end = (Stmt) _trap.getEndUnit();
246     
247     			// we assume that the first statement in the handling block will be the identity statement that retrieves the
248     			// caught expression.
249     			final IdentityStmt _handlerStmt = (IdentityStmt) _trap.getHandlerUnit();
250     			final CaughtExceptionRef _catchRef = (CaughtExceptionRef) _handlerStmt.getRightOp();
251     			final SootClass _exception = _trap.getException();
252     
253     			_catchCtxt.setStmt(_handlerStmt);
254     			_catchCtxt.setProgramPoint(_handlerStmt.getRightOpBox());
255     
256     			final int _k = stmtList.indexOf(_end);
257     
258     			for (int _j = stmtList.indexOf(_begin); _j < _k; _j++) {
259     				final Stmt _tmp = stmtList.get(_j);
260     
261     				if (!_caught.contains(_tmp)) {
262     					_exprCtxt.setStmt(_tmp);
263     
264     					if (_tmp instanceof ThrowStmt) {
265     						final ThrowStmt _ts = (ThrowStmt) _tmp;
266     						final Value _op = _ts.getOp();
267     						final SootClass _scTemp = fa.getClass(((RefType) _op.getType()).getClassName());
268     
269     						if (Util.isDescendentOf(_scTemp, _exception)) {
270     							_exprCtxt.setProgramPoint(_ts.getOpBox());
271     							final OFAFGNode<T> _throwNode = getASTNode(_op, _exprCtxt);
272     							_throwNode.addSucc(queryASTNode(_catchRef, _catchCtxt));
273     							_caught.add(_ts);
274     						}
275     					} else if (_tmp.containsInvokeExpr()) {
276     						_exprCtxt.setProgramPoint(_tmp.getInvokeExprBox());
277     						final OFAFGNode<T> _tempNode = queryThrowNode(_tmp.getInvokeExpr(), _exprCtxt);
278     
279     						if (_tempNode != null) {
280     							_tempNode.addSucc(queryASTNode(_catchRef, _catchCtxt));
281     						}
282     					}
283     				}
284     			}
285     		}
286     
287     		connectThrowNodesToThrownNode(body, stmtList);
288     	}
289     }
