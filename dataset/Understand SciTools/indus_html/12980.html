58     	 * The logger used by instances of this class to log messages.
59     	 */
60     	private static final Logger LOGGER = LoggerFactory.getLogger(SliceCriterionSpec.class);

62     	/** 
63     	 * The singleton slice criteria factory.
64     	 */
65     	private static final SliceCriteriaFactory CRITERIA_FACTORY = SliceCriteriaFactory.getFactory();

67     	/** 
68     	 * The sequence of names of parameter types of the method immediately enclosing the criterion.
69     	 *
70     	 * @invariant parameterTypeNames.oclIsKindOf(Sequence(String))
71     	 */
72     	private List<String> parameterTypeNames;
73     
74     	/** 
75     	 * The name of the class immediately enclosing the criterion.
76     	 */
77     	private String className;
78     
79     	/** 
80     	 * The name of the method immediately enclosing the criterion.
81     	 */
82     	private String methodName;
83     
84     	/** 
85     	 * The name of the return type of the method  immediately enclosing the criterion.
86     	 */
87     	private String returnTypeName;
88     
89     	/** 
90     	 * This indicates if all expressions in the statement should be considered as slice criterion.
91     	 */
92     	private boolean considerEntireStmt;
93     
94     	/** 
95     	 * This indicates if the execution of the statement should be considered as slice criterion.
96     	 */
97     	private boolean considerExecution;
98     
99     	/** 
100     	 * The index of the use or def value box, in the statement, that is the slice criterion.
101     	 */
102     	private int exprIndex = -1;  // Initialization is required due to a limitation of JiBX (known problem 3 on JiBX site)
103     
104     	/** 
105     	 * The index of the statement, in the method body, in which the slice criterion occurs.
106     	 */
107     	private int stmtIndex = -1;  // Initialization is required due to a limitation of JiBX (known problem 3 on JiBX site)
108     
109     	/**
110     	 * Creates an instance of this class.
111     	 */
112     	private SliceCriterionSpec() {
113     		parameterTypeNames = createParameterTypeNamesContainer();
114     	}
115     
116     	/**
117     	 * Retrieves the criteria represented by this spec relative to the given scene.
118     	 *
119     	 * @param scene relative to which the criteria will be generated.
120     	 *
121     	 * @return the collection of slice criterion.
122     	 *
123     	 * @throws MissingResourceException when the any of the types used in the criterion specification cannot be found.
124     	 * @throws IllegalStateException when the method in the spec does not have a body or the specified statment/expr does not
125     	 * 		   exists.
126     	 */
127     	public Collection<ISliceCriterion> getCriteria(final Scene scene) {
128     		trim();
129     
130     		final SootClass _sc = scene.getSootClass(className);
131     
132     		if (_sc == null) {
133     			final String _msg = className + " is not available in the System.";
134     			LOGGER.error(_msg);
135     			throw new MissingResourceException("Given class not available in the System.", className, null);
136     		}
137     
138     		final List<Type> _parameterTypes = new ArrayList<Type>();
139     
140     		for (final Iterator<String> _i = parameterTypeNames.iterator(); _i.hasNext();) {
141     			final String _name =  _i.next();
142     			_parameterTypes.add(Util.getTypeFor(_name, scene));
143     		}
144     
145     		final SootMethod _sm = _sc.getMethod(methodName, _parameterTypes, Util.getTypeFor(returnTypeName, scene));
146     		_sc.setApplicationClass();
147     
148     		final Body _body = _sm.retrieveActiveBody();
149     
150     		if (_body == null) {
151     			final String _msg = returnTypeName + " " + methodName + "(" + parameterTypeNames + ") does not have a body.";
152     			LOGGER.error(_msg);
153     			throw new IllegalStateException(_msg);
154     		}
155     
156     		final List<Stmt> _stmts = Collections.list(Collections.<Stmt>enumeration(_body.getUnits()));
157     
158     		if (_stmts.size() < stmtIndex + 1) {
159     			final String _msg =
160     				returnTypeName + " " + methodName + "(" + parameterTypeNames + ") has only " + _stmts.size()
161     				+ " statements. [" + stmtIndex + "]";
162     			LOGGER.error(_msg);
163     			throw new IllegalStateException(_msg);
164     		}
165     
166     		final Collection<ISliceCriterion> _result;
167     
168     		if (stmtIndex == -1) {
169     			_result = CRITERIA_FACTORY.getCriteria(_sm);
170     		} else {
171     			final Stmt _stmt = _stmts.get(stmtIndex);
172     
173     			if (exprIndex == -1) {
174     				_result = CRITERIA_FACTORY.getCriteria(_sm, _stmt, considerEntireStmt, considerExecution);
175     			} else {
176     				_result =
177     					CRITERIA_FACTORY.getCriteria(_sm, _stmt, (ValueBox) _stmt.getUseAndDefBoxes().get(exprIndex),
178     						considerExecution);
179     			}
180     		}
181     
182     		return _result;
183     	}
184     
185     	/**
186     	 * Extracts the names of the given type.
187     	 *
188     	 * @param parameterTypes is the collection of types whose names should be extracted.
189     	 *
190     	 * @return the names of the type.
191     	 *
192     	 * @pre paramterTypes != null
193     	 * @post result != null
194     	 */
195     	public static List<String> getNamesOfTypes(final List<Type> parameterTypes) {
196     		final List<String> _result = new ArrayList<String>();
197     
198     		for (final Iterator <Type>_i = parameterTypes.iterator(); _i.hasNext();) {
199     			_result.add( _i.next().toString().replace(']', ' ').trim());
200     		}
201     		return _result;
202     	}

204     	/**
205     	 * @see java.lang.Object#toString()
206     	 */
207     	@Override public String toString() {
208     		trim();
209     		return new ToStringBuilder(this).append("className", this.className).append("returnTypeName", this.returnTypeName)
210     										  .append("methodName", this.methodName)
211     										  .append("parameterTypeNames", this.parameterTypeNames)
212     										  .append("stmtIndex", this.stmtIndex).append("exprIndex", this.exprIndex)
213     										  .append("considerExecution", this.considerExecution)
214     										  .append("considerEntireStmt", this.considerEntireStmt).toString();
215     	}

217     	/**
218     	 * Retrieves the specification for the given criterion relative to <code>scene</code>.
219     	 *
220     	 * @param criterion for which the specification should be generated.
221     	 *
222     	 * @return the specification.
223     	 *
224     	 * @throws IllegalStateException when an error occurs while creating the criterion.
225     	 *
226     	 * @pre criterion != null and scene != null
227     	 * @post result != null
228     	 */
229     	static Collection<SliceCriterionSpec> getCriterionSpec(final ISliceCriterion criterion) {
230     		final SootMethod _method = criterion.getOccurringMethod();
231     		final SootClass _declaringClass = _method.getDeclaringClass();
232     		final String _prefix = _declaringClass.getPackageName();
233     		final String _shortJavaStyleName = _declaringClass.getShortJavaStyleName();
234     		final String _className;
235     
236     		if (_prefix.length() != 0) {
237     			_className = _prefix + "." + _shortJavaStyleName;
238     		} else {
239     			_className = _shortJavaStyleName;
240     		}
241     
242     		final String _methodName = _method.getName();
243     		final Collection<SliceCriterionSpec> _result;
244     		if (!_method.isAbstract() && !_method.isNative()) {
245     			final Body _body  = _method.retrieveActiveBody();
246     
247     			if (_body == null) {
248     				final String _msg = _method.getSignature() + " does not have a body.";
249     				LOGGER.error(_msg);
250     				throw new IllegalStateException(_msg);
251     			}
252     	
253     			final List<Stmt> _stmts = Collections.list(Collections.<Stmt>enumeration(_body.getUnits()));
254     			final Stmt _occurringStmt = CriteriaSpecHelper.getOccurringStmt(criterion);
255     			final int _stmtIndex = _stmts.indexOf(_occurringStmt);
256     			final boolean _considerExecution = CriteriaSpecHelper.isConsiderExecution(criterion);
257     	
258     			final SliceCriterionSpec _spec = new SliceCriterionSpec();
259     			_spec.className = _className;
260     			_spec.methodName = _methodName;
261     			_spec.parameterTypeNames = getNamesOfTypes(_method.getParameterTypes());
262     			_spec.returnTypeName = _method.getReturnType().toString();
263     			_spec.stmtIndex = _stmtIndex;
264     			_spec.exprIndex = -1;
265     			_spec.considerExecution = _considerExecution;
266     			_spec.considerEntireStmt = false;
267     	
268     			_result = new ArrayList<SliceCriterionSpec>();
269     			final ValueBox _occurringExprBox = CriteriaSpecHelper.getOccurringExpr(criterion);
270     	
271     			if (_occurringExprBox != null) {
272     				final Value _expr = _occurringExprBox.getValue();
273     				final int _size = _expr.getUseBoxes().size();
274     	
275     				if (_size == 0) {
276     					_spec.exprIndex = _occurringStmt.getUseAndDefBoxes().indexOf(_occurringExprBox);
277     					_result.add(_spec);
278     				} else {
279     					for (int _i = _size - 1; _i >= 0; _i--) {
280     						try {
281     							final SliceCriterionSpec _temp = (SliceCriterionSpec) _spec.clone();
282     							_temp.exprIndex = _i;
283     							_result.add(_temp);
284     						} catch (final CloneNotSupportedException _e) {
285     							final String _msg = "Low level Error while creating criterion specification.";
286     							LOGGER.error(_msg);
287     							throw new IllegalStateException(_msg);
288     						}
289     					}
290     				}
291     			} else {
292     				_result.add(_spec);
293     			}
294     		} else {
295     			_result = Collections.emptySet();
296     		}
297     
298     		return _result;
299     	}

301     	/**
302     	 * Creates a container for names of paramter types. This is used mainly during deserialization.
303     	 *
304     	 * @return the container.
305     	 *
306     	 * @post result != null
307     	 */
308     	private static List<String> createParameterTypeNamesContainer() {
309     		return new ArrayList<String>();
310     	}
311     	
312     	/**
313     	 * Trims the string data in the specification.
314     	 */
315     	private void trim() {
316     		if (className != null) {
317     			className = className.trim();
318     		}
319     
320     		if (methodName != null) {
321     			methodName = methodName.trim();
322     		}
323     
324     		if (returnTypeName != null) {
325     			returnTypeName = returnTypeName.trim();
326     		}
327     
328     		for (final ListIterator<String> _i = parameterTypeNames.listIterator(); _i.hasNext();) {
329     			final String _str = _i.next();
330     
331     			if (_str != null) {
332     				_i.remove();
333     				_i.add(_str.trim());
334     			}
335     		}
336     	}
