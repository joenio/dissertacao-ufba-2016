17     package edu.ksu.cis.indus.staticanalyses.flow;
18     
19     import edu.ksu.cis.indus.annotations.InternalUse;
20     import edu.ksu.cis.indus.staticanalyses.tokens.ITokens;
21     
22     import org.slf4j.Logger;
23     import org.slf4j.LoggerFactory;
24     
25     /**
26      * This class represents a peice of work to inject a set of tokens into a flow graph node.
27      * 
28      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
29      * @author $Author: rvprasad $
30      * @version $Revision: 1.14 $ $Date: 2007/02/10 19:07:01 $
31      * @param <SYM> is the type of symbol whose flow is being analyzed.
32      * @param <T> is the type of the token set object.
33      * @param <N> is the type of the summary node in the flow analysis.
34      */
35     @InternalUse public class SendTokensWork<SYM, T extends ITokens<T, SYM>, N extends IFGNode<SYM, T, N>>
36     		extends AbstractTokenProcessingWork<T> {
37     
38     	/**
39     	 * The logger used by instances of this class to log messages.
40     	 */
41     	private static final Logger LOGGER = LoggerFactory.getLogger(SendTokensWork.class);
42     
43     	/**
44     	 * The flow graph node associated with this work.
45     	 */
46     	private N node;
47     
48     	/**
49     	 * Creates an instance of this class.
50     	 * 
51     	 * @param associatedNode is the desitination node on which this work operates.
52     	 * @param tokenSet to be used by this work object to store the tokens whose flow should be instrumented.
53     	 */
54     	public SendTokensWork(final N associatedNode, final T tokenSet) {
55     		super(tokenSet);
56     		node = associatedNode;
57     	}
58     
59     	/**
60     	 * Injects the tokens into the associated node.
61     	 */
62     	public final void execute() {
63     
64     		if (LOGGER.isDebugEnabled()) {
65     			LOGGER.debug("execute() - Propagating tokens - " + tokens.getValues() + " into node " + node);
66     		}
67     
68     		node.injectTokens(tokens);
69     
70     		if (node instanceof AbstractFGNode) {
71     			((AbstractFGNode) node).forgetSendTokensWork();
72     		}
73     
74     		tokens.clear();
75     	}
76     }
