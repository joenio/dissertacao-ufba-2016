29     /**
30      * This filter can be used to identify code that are involved in <code>assert</code> statements <b>as compiled by Sun's java
31      * compiler</b>.
32      *
33      * @author <a href="http://www.cis.ksu.edu/~rvprasad">Venkatesh Prasad Ranganath</a>
34      * @author $Author: rvprasad $
35      * @version $Revision: 1.8 $ $Date: 2007/10/09 15:58:59 $
36      */
37     public final class AssertionSliceCriteriaPredicate
38     		extends AbstractSliceCriteriaPredicate<Stmt> {
39     
40     	/**
41     	 * The logger used by instances of this class to log messages.
42     	 */
43     	private static final Logger LOGGER = LoggerFactory.getLogger(AssertionSliceCriteriaPredicate.class);
44     
45     	/**
46     	 * Creates an instance of this class.
47     	 */
48     	public AssertionSliceCriteriaPredicate() {
49     		super();
50     	}
51     
52     	/**
53     	 * {@inheritDoc}
54     	 */
55     	public boolean evaluate(final Stmt stmt) {
56     		if (LOGGER.isDebugEnabled()) {
57     			LOGGER.debug("evaluate(entity = " + stmt + ":" + stmt.getClass() + ")");
58     		}
59     
60     		boolean _result = false;
61     
62     		for (final Iterator<ValueBox> _i = stmt.getUseAndDefBoxes().iterator(); _i.hasNext() && !_result;) {
63     			final ValueBox _vb = _i.next();
64     			final Value _v = _vb.getValue();
65     
66     			if (_v.getType() instanceof RefType) {
67     				final RefType _rt = (RefType) _v.getType();
68     				final String _name = _rt.getClassName();
69     				_result |= _name.equals("java.lang.AssertionError");
70     			}
71     		}
72     		return _result;
73     	}
74     }
