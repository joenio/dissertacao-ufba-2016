18     package com.h3xstream.findsecbugs.crypto;
19     
20     import edu.umd.cs.findbugs.BugInstance;
21     import edu.umd.cs.findbugs.BugReporter;
22     import edu.umd.cs.findbugs.OpcodeStack;
23     import edu.umd.cs.findbugs.Priorities;
24     import edu.umd.cs.findbugs.bcel.OpcodeStackDetector;
25     import org.apache.bcel.Constants;
26     
27     /**
28      * Identifies the use of MD2, MD5 and SHA1 hash function and recommends the
29      * use of modern functions.
30      */
31     public class WeakMessageDigestDetector extends OpcodeStackDetector {
32     
33         private static final String WEAK_MESSAGE_DIGEST_MD5_TYPE = "WEAK_MESSAGE_DIGEST_MD5";
34         private static final String WEAK_MESSAGE_DIGEST_SHA1_TYPE = "WEAK_MESSAGE_DIGEST_SHA1";
35     
36         private final BugReporter bugReporter;
37     
38         public WeakMessageDigestDetector(BugReporter bugReporter) {
39             this.bugReporter = bugReporter;
40         }
41     
42         @Override
43         public void sawOpcode(int seen) {
44             if (seen != Constants.INVOKESTATIC) {
45                 return;
46             }
47             String className = getClassConstantOperand();
48             String methodName = getNameConstantOperand();
49             if (className.equals("java/security/MessageDigest") && methodName.equals("getInstance")) {
50                 final OpcodeStack.Item item;
51                 String methodSig = getSigConstantOperand();
52                 if (methodSig.equals("(Ljava/lang/String;)Ljava/security/MessageDigest;")) {
53                     //Extract the value being pushed..
54                     item = stack.getStackItem(0);
55                 } else if (methodSig.equals("(Ljava/lang/String;Ljava/lang/String;)Ljava/security/MessageDigest;")) {
56                     item = stack.getStackItem(1);
57                 } else if (methodSig.equals("(Ljava/lang/String;Ljava/security/Provider;)Ljava/security/MessageDigest;")) {
58                     item = stack.getStackItem(1);
59                 } else {
60                     return;
61                 }
62                 String algorithm = (String) item.getConstant(); //Null if the value passed isn't constant
63                 checkHashFunction(algorithm);
64             } else if (className.equals("org/apache/commons/codec/digest/DigestUtils")) {
65                 if (methodName.equals("getMd2Digest")
66                         || methodName.equals("md2")
67                         || methodName.equals("md2Hex")) {
68                     checkHashFunction("md2");
69                 } else if(methodName.equals("getMd5Digest")
70                         || methodName.equals("md5")
71                         || methodName.equals("md5Hex")) {
72                     checkHashFunction("md5");
73                 } else if(methodName.equals("getSha1Digest")
74                         || methodName.equals("getShaDigest")
75                         || methodName.equals("sha1")
76                         || methodName.equals("sha")
77                         || methodName.equals("sha1Hex")
78                         || methodName.equals("shaHex")) {
79                     checkHashFunction("sha1");
80                 } else if(methodName.equals("getDigest")) {
81                     //Extract the value being pushed..
82                     OpcodeStack.Item top = stack.getStackItem(0);
83                     String algorithm = (String) top.getConstant(); //Null if the value passed isn't constant
84                     checkHashFunction(algorithm);
85                 }
86             } else if (className.equals("java/security/Signature") && methodName.equals("getInstance")) {
87                 final OpcodeStack.Item item;
88                 String methodSig = getSigConstantOperand();
89                 if (methodSig.equals("(Ljava/lang/String;)Ljava/security/Signature;")) {
90                     item = stack.getStackItem(0);
91                 } else if (methodSig.equals("(Ljava/lang/String;Ljava/security/Provider;)Ljava/security/Signature;")) {
92                     item = stack.getStackItem(1);
93                 } else if (methodSig.equals("(Ljava/lang/String;Ljava/lang/String;)Ljava/security/Signature;")) {
94                     item = stack.getStackItem(1);
95                 } else {
96                     return;
97                 }
98                 String algorithm = (String) item.getConstant();
99                 if (algorithm != null) {
100                     int index = algorithm.indexOf("with");
101                     algorithm = algorithm.substring(0, index > 0 ? index : 0);
102                     checkHashFunction(algorithm);
103                 }
104             }
105         }
106     
107         private void checkHashFunction(String algorithm) {
108             if (algorithm == null) {
109                 return;
110             }
111             algorithm = algorithm.toUpperCase();
112             if ("MD2".equals(algorithm) || "MD4".equals(algorithm) || "MD5".equals(algorithm)) {
113                 bugReporter.reportBug(new BugInstance(this, WEAK_MESSAGE_DIGEST_MD5_TYPE, Priorities.HIGH_PRIORITY) //
114                         .addClass(this).addMethod(this).addSourceLine(this) //
115                         .addString(algorithm));
116             } else if ("SHA1".equals(algorithm) || "SHA-1".equals(algorithm)) { //Lower priority for SHA-1
117                 bugReporter.reportBug(new BugInstance(this, WEAK_MESSAGE_DIGEST_SHA1_TYPE, Priorities.NORMAL_PRIORITY) //
118                         .addClass(this).addMethod(this).addSourceLine(this) //
119                         .addString(algorithm));
120             }
121         }
122     }
