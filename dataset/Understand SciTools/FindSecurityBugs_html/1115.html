18     package com.h3xstream.findsecbugs.cookie;
19     
20     import edu.umd.cs.findbugs.BugInstance;
21     import edu.umd.cs.findbugs.BugReporter;
22     import edu.umd.cs.findbugs.Priorities;
23     import edu.umd.cs.findbugs.bcel.OpcodeStackDetector;
24     import org.apache.bcel.Constants;
25     
26     public class CookieReadDetector extends OpcodeStackDetector {
27     
28         private static final String COOKIE_USAGE_TYPE = "COOKIE_USAGE";
29     
30         private BugReporter bugReporter;
31     
32         public CookieReadDetector(BugReporter bugReporter) {
33             this.bugReporter = bugReporter;
34         }
35     
36         @Override
37         public void sawOpcode(int seen) {
38     
39             if (seen == Constants.INVOKEVIRTUAL && getClassConstantOperand().equals("javax/servlet/http/Cookie")
40                     && (getNameConstantOperand().equals("getName") || getNameConstantOperand().equals("getValue") ||
41                     getNameConstantOperand().equals("getPath"))) {
42     
43                 bugReporter.reportBug(new BugInstance(this, COOKIE_USAGE_TYPE, Priorities.LOW_PRIORITY) //
44                         .addClass(this).addMethod(this).addSourceLine(this));
45             }
46         }
47     }
