18     package com.h3xstream.findsecbugs.endpoint;
19     
20     import edu.umd.cs.findbugs.BugInstance;
21     import edu.umd.cs.findbugs.BugReporter;
22     import edu.umd.cs.findbugs.Detector;
23     import edu.umd.cs.findbugs.Priorities;
24     import edu.umd.cs.findbugs.ba.ClassContext;
25     import org.apache.bcel.classfile.Constant;
26     import org.apache.bcel.classfile.ConstantPool;
27     import org.apache.bcel.classfile.ConstantUtf8;
28     import org.apache.bcel.classfile.JavaClass;
29     
30     /**
31      * Identify endpoints using the web framework Tapestry.
32      * <p>
33      * <a href="http://tapestry.apache.org/">Official Website</a>
34      */
35     public class TapestryEndpointDetector implements Detector {
36     
37         private static final String TAPESTRY_ENDPOINT_TYPE = "TAPESTRY_ENDPOINT";
38     
39         private BugReporter bugReporter;
40     
41         public TapestryEndpointDetector(BugReporter bugReporter) {
42             this.bugReporter = bugReporter;
43         }
44     
45         @Override
46         public void visitClassContext(ClassContext classContext) {
47     
48             JavaClass javaClass = classContext.getJavaClass();
49     
50             if (!javaClass.getPackageName().contains(".pages")) {
51                 return;
52             }
53     
54             //The following some false positive
55             if (javaClass.getPackageName().endsWith(".pages")) {
56                 bugReporter.reportBug(new BugInstance(this, TAPESTRY_ENDPOINT_TYPE, Priorities.LOW_PRIORITY) //
57                         .addClass(javaClass));
58             }
59     
60             //The package contains ".pages" and has some references to tapestry
61             // then it must be an endpoint.
62             //The constants pool contains all references that are reused in the bytecode
63             // including full class name and interface name.
64             if (javaClass.getPackageName().contains(".pages")) {
65                 ConstantPool constants = javaClass.getConstantPool();
66                 for (Constant c : constants.getConstantPool()) {
67                     if (c instanceof ConstantUtf8) {
68                         ConstantUtf8 utf8 = (ConstantUtf8) c;
69                         String constantValue = String.valueOf(utf8.getBytes());
70                         if (constantValue.startsWith("Lorg/apache/tapestry5/annotations")) {
71                             bugReporter.reportBug(new BugInstance(this, TAPESTRY_ENDPOINT_TYPE, Priorities.LOW_PRIORITY) //
72                                     .addClass(javaClass));
73                             return;
74                         }
75                     }
76                 }
77             }
78     
79         }
80     
81         @Override
82         public void report() {
83     
84         }
85     
86     }
