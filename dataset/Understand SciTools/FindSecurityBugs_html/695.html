45     public class CustomInjectionSource implements InjectionSource {

47         private static final Logger LOG = Logger.getLogger(CustomInjectionSource.class.getName());
48         private static final String INJECTION_SOURCE_RESOURCE_NAME = "CustomInjectionSource.properties";
49         private static final Map<String, InjectionSource> instanceMap = new HashMap<String, InjectionSource>();
50     
51         private static final String PROPERTY_SEPARATOR = "\\Q|\\E";
52         private static final String PARAM_IDX_SEPARATOR = ",";
53     
54         private static final String SYSTEM_PROPERTY = "findsecbugs.injection.sources";
55         private static final String CUSTOM_INJECTION_TYPE = "CUSTOM_INJECTION";
56     
57         static String toResourceBaseName(Class<? extends LegacyInjectionDetector> that) {
58             return that.getPackage().getName().replaceAll("\\.", "/");
59         }
60     
61         public static InjectionSource getInstance(Class<? extends LegacyInjectionDetector> that) {
62             return getInstance(toResourceBaseName(that));
63         }
64     
65         public static InjectionSource getInstance(String resourceBaseName) {
66             if (instanceMap.containsKey(resourceBaseName)) {
67                 return instanceMap.get(resourceBaseName);
68             }
69     
70             Properties properties = getInjectionSourceProperties(resourceBaseName);
71             CustomInjectionSource injectionSource = new CustomInjectionSource(properties);
72             instanceMap.put(resourceBaseName, injectionSource);
73             return injectionSource;
74         }

76         static Properties getInjectionSourceProperties(String resourceBaseName) {
77             String propertyValue = System.getProperty(SYSTEM_PROPERTY);
78             String[] resourcePaths = propertyValue == null ? new String[0] : propertyValue.split("\\Q" + File.pathSeparatorChar + "\\E");
79             List<URL> urls = new ArrayList<URL>(resourcePaths.length);
80             for (String resourcePath : resourcePaths) {
81                 File file = new File(resourcePath);
82                 if (!file.exists()) {
83                     LOG.warning(file + " not found.");
84                     continue;
85                 }
86                 try {
87                     urls.add(file.toURI().toURL());
88                 } catch (MalformedURLException e) {
89                     LOG.log(Level.SEVERE, file + " did not load.", e);
90                 }
91             }
92             String resourceName = resourceBaseName + "/" + INJECTION_SOURCE_RESOURCE_NAME;
93             URL url = CustomInjectionSource.class.getClassLoader().getResource(resourceName);
94             if (url == null && FindSecBugsGlobalConfig.getInstance().isPrintCustomInjectionWarning()) {
95                 LOG.info("The optional configuration for additional injection sources (" + resourceName + ") was not found. " +
96                         "This message can be ignored if no custom API are intended to be configured. " +
97                         "For more info: http://h3xstream.github.io/find-sec-bugs/bugs.htm#CUSTOM_INJECTION");
98             }
99             return loadProperties(urls.toArray(new URL[urls.size()]), loadProperties(url, null));
100         }
101     
102         private static Properties loadProperties(URL url, Properties defaults) {
103             Properties properties = new Properties(defaults);
104             if (url == null) {
105                 return null;
106             }
107     
108             try {
109                 InputStream input = url.openStream();
110                 properties.load(input);
111             } catch (IOException e) {
112                 LOG.log(Level.SEVERE, url + " did not open.", e);
113             }
114     
115             return properties;
116         }
117     
118         private static Properties loadProperties(URL[] urls, Properties defaults) {
119             return loadProperties(urls, defaults, urls.length);
120         }
121     
122         private static Properties loadProperties(URL[] urls, Properties previous, int index) {
123             return index == 0 ? previous : loadProperties(urls, loadProperties(urls[--index], previous), index);
124         }

126         private static Map<InvokeIdentifier, InjectionPoint>  toInjectableParametersMap(Properties properties) {
127             Map<InvokeIdentifier, InjectionPoint> map = new HashMap<InvokeIdentifier, InjectionPoint>(properties.size());
128             for (String propertyName : properties.stringPropertyNames()) {
129                 //
130                 String[] value = properties.getProperty(propertyName).split(PROPERTY_SEPARATOR);
131                 String[] values = value[0].split(PARAM_IDX_SEPARATOR);
132                 int[] parameters = new int[values.length];
133                 for (int i = 0; i < values.length; i++) {
134                     parameters[i] = Integer.parseInt(values[i].trim());
135                 }
136                 String bugType = value.length < 2 ? CUSTOM_INJECTION_TYPE : value[1].trim();
137     
138                 InjectionPoint ip = new InjectionPoint(parameters, bugType);
139     
140                 InvokeIdentifier invoke = InvokeIdentifier.valueOf(propertyName);
141                 ip.setInjectableMethod(invoke.className + "." + invoke.methodName);
142     
143                 map.put(invoke,ip);
144             }
145             return map;
146         }
147     
148         private final Map<InvokeIdentifier, InjectionPoint> injectableParametersMap;
149     
150         public CustomInjectionSource(Properties properties) {
151             this(toInjectableParametersMap(properties == null ? new Properties() : properties));
152         }
153     
154         public CustomInjectionSource(Map<InvokeIdentifier, InjectionPoint> injectableParametersMap) {
155             this.injectableParametersMap = injectableParametersMap;
156         }
157     
158         @Override
159         public InjectionPoint getInjectableParameters(InvokeInstruction ins, ConstantPoolGen cpg, InstructionHandle insHandle) {
160             //ByteCode.printOpCode(ins, cpg);
161             InvokeIdentifier identifier = new InvokeIdentifier(ins, cpg, insHandle);
162             InjectionPoint injectionPoint = injectableParametersMap.get(identifier);
163             return injectionPoint == null ? InjectionPoint.NONE : injectionPoint;
164         }

166         public static class InvokeIdentifier {
167     
168             private final String methodName;
169             private final String methodSignature;
170             private final String className;
171             private final Kind kind;
172     
173             InvokeIdentifier(String className, String methodName, String methodSignature, Kind kind) {
174                 this.methodName = methodName;
175                 this.methodSignature = methodSignature;
176                 this.className = className;
177                 this.kind = kind;
178             }
179     
180             InvokeIdentifier(InvokeInstruction ins, ConstantPoolGen cpg, InstructionHandle insHandle) {
181                 this.methodName = ins.getMethodName(cpg);
182                 this.methodSignature = ins.getSignature(cpg);
183                 this.className = ins.getClassName(cpg).replaceAll("\\.", "/");
184                 this.kind = Kind.valueOf(ins);
185             }
186     
187             enum Kind {
188     
189                 INVOKEVIRTUAL(Constants.INVOKEVIRTUAL),
190                 INVOKESPECIAL(Constants.INVOKESPECIAL),
191                 INVOKESTATIC(Constants.INVOKESTATIC),
192                 INVOKEINTERFACE(Constants.INVOKEINTERFACE);
193     
194                 private final short kind;
195     
196                 private Kind(short kind) {
197                     this.kind = kind;
198                 }
199     
200                 static Kind valueOf(InvokeInstruction ins) {
201                     return Kind.valueOf(ins.getClass().getSimpleName());
202                 }
203             }
204     
205             public static InvokeIdentifier valueOf(String value) {
206                 String[] parts = value.split(PROPERTY_SEPARATOR, 4);
207                 return new InvokeIdentifier(parts[1], parts[2], parts[3], Kind.valueOf(parts[0]));
208             }
209     
210             @Override
211             public String toString() {
212                 return kind + " " + className + "." + methodName + ":" + methodSignature;
213             }
214     
215             @Override
216             public int hashCode() {
217                 return toString().hashCode();
218             }
219     
220             @Override
221             public boolean equals(Object obj) {
222                 if (obj == null || getClass() != obj.getClass()) {
223                     return false;
224                 }
225                 return toString().equals(obj.toString());
226             }
227         }
