33     /**
34      * This detector does minimal effort to find potential REDOS.
35      * <p>
36      * It will identify pattern similar to : <code>(( )+)+</code>
37      * </p>
38      * <p>
39      * It will not identify pattern of equivalence (such as<code>(aa|a)</code>).
40      * It is far more complex to identify.
41      * </p>
42      * <p>
43      * For more advanced Regex analysis: <a href="http://code.google.com/p/saferegex/">Safe Regex</a>
44      * </p>
45      */
46     public class ReDosDetector extends OpcodeStackDetector {
47     
48         private static final boolean DEBUG = true;
49     
50         private static final String REDOS_TYPE = "REDOS";
51     
52         private static final char[] OPENING_CHAR = {'(', '['};
53     
54         private static final char[] CLOSING_CHAR = {')', ']'};
55     
56         private static final char[] PLUS_CHAR = {'+', '*', '?'};
57     
58         private static final InvokeMatcherBuilder PATTERN_COMPILE = invokeInstruction().atClass("java/util/regex/Pattern")
59                 .atMethod("compile").withArgs("(Ljava/lang/String;)Ljava/util/regex/Pattern;");
60         private static final InvokeMatcherBuilder STRING_MATCHES = invokeInstruction().atClass("java/lang/String")
61                 .atMethod("matches").withArgs("(Ljava/lang/String;)Z");
62     
63         private BugReporter bugReporter;
64     
65         public ReDosDetector(BugReporter bugReporter) {
66             this.bugReporter = bugReporter;
67         }
68     
69         @Override
70         public void sawOpcode(int seen) {
71             //printOpCode(seen);
72             if (seen == Constants.INVOKESTATIC && PATTERN_COMPILE.matches(this)) {
73                 OpcodeStack.Item item = stack.getStackItem(0);
74                 if (!StackUtils.isVariableString(item)) {
75                     String value = (String) item.getConstant();
76                     analyseRegexString(value);
77                 }
78             } else if (seen == Constants.INVOKEVIRTUAL && STRING_MATCHES.matches(this)) {
79                 OpcodeStack.Item item = stack.getStackItem(0);
80                 if (!StackUtils.isVariableString(item)) {
81                     String value = (String) item.getConstant();
82                     analyseRegexString(value);
83                 }
84             }
85         }
86     
87         public void analyseRegexString(String regex) {
88             if (regex.length() > 0) {
89                 recurAnalyseRegex(regex, regex.length() - 1, 0);
90             }
91         }
92     
93         private int recurAnalyseRegex(String regex, int startPosition, int level) {
94     //        print(level, "level = " + level);
95             if (level == 2) {
96     
97                 MethodDescriptor md = this.getMethodDescriptor();
98                 FieldDescriptor fd = this.getFieldDescriptor();
99     
100                 BugInstance bug = new BugInstance(this, REDOS_TYPE, Priorities.NORMAL_PRIORITY) //
101                         .addString(regex).addClass(this);
102                 if (md != null)
103                     bug.addMethod(md);
104                 if (fd != null)
105                     bug.addField(fd);
106     
107                 try {
108                     bug.addSourceLine(this);
109                 } catch (IllegalStateException e) {
110                 }
111     
112                 bugReporter.reportBug(bug);
113                 return 0;
114             }
115     
116     
117     //        print(level, "Analysing " + regex.substring(0, startPosition + 1));
118     
119             boolean openingMode = false;
120     
121             for (int i = startPosition; i >= 0; i--) {
122     //            print(level, "[" + i + "] = '" + regex.charAt(i) + "'");
123     
124                 if (isChar(regex, i, OPENING_CHAR)) {
125     //                print(level, "<<<<");
126                     return i;
127                 }
128     
129                 if (isChar(regex, i, CLOSING_CHAR)) {
130                     int newLevel = level;
131                     if (i + 1 < regex.length() && isChar(regex, i + 1, PLUS_CHAR)) {
132                         newLevel += 1;
133                     }
134     //                print(level, ">>>>");
135                     openingMode = true;
136                     i = recurAnalyseRegex(regex, i - 1, newLevel);
137                     if (i == -1) {
138                         return 0;
139                     }
140     //                print(level, "Restarting at " + i);
141                 }
142             }
143     
144     //        print(level, "END!");
145     
146             return 0;
147         }
148     
149         /**
150          * @param value
151          * @param position
152          * @param charToTest
153          * @return
154          */
155         private boolean isChar(String value, int position, char[] charToTest) {
156             char actualChar = value.charAt(position);
157             boolean oneCharFound = false;
158             for (char ch : charToTest) {
159                 if (actualChar == ch) {
160                     oneCharFound = true;
161                     break;
162                 }
163             }
164             return oneCharFound && (position == 0 || value.charAt(position - 1) != '\\');
165         }
