30     
31     import java.util.Iterator;
32     
33     /**
34      * Similar to the blowfish key size detector
35      */
36     public class InsufficientKeySizeRsaDetector implements Detector {
37     
38         private static final boolean DEBUG = false;
39         private static final String RSA_KEY_SIZE_TYPE = "RSA_KEY_SIZE";
40     
41         private BugReporter bugReporter;
42     
43         public InsufficientKeySizeRsaDetector(BugReporter bugReporter) {
44             this.bugReporter = bugReporter;
45         }
46     
47         @Override
48         public void visitClassContext(ClassContext classContext) {
49             JavaClass javaClass = classContext.getJavaClass();
50     
51             Method[] methodList = javaClass.getMethods();
52     
53             for (Method m : methodList) {
54     
55                 try {
56                     analyzeMethod(m, classContext);
57                 } catch (CFGBuilderException e) {
58                 } catch (DataflowAnalysisException e) {
59                 }
60             }
61         }
62     
63         private void analyzeMethod(Method m, ClassContext classContext) throws CFGBuilderException, DataflowAnalysisException {
64     
65             //Conditions that needs to fill to identify the vulnerability
66             boolean createRsaKeyGen = false;
67             boolean initializeWeakKeyLength = false;
68             Location locationWeakness = null;
69     
70             ConstantPoolGen cpg = classContext.getConstantPoolGen();
71             CFG cfg = classContext.getCFG(m);
72     
73             for (Iterator<Location> i = cfg.locationIterator(); i.hasNext(); ) {
74                 Location location = i.next();
75     
76                 Instruction inst = location.getHandle().getInstruction();
77     //            ByteCode.printOpCode(inst, cpg);
78     
79                 if (inst instanceof INVOKESTATIC) { //KeyPairGenerator.getInstance is called
80                     INVOKESTATIC invoke = (INVOKESTATIC) inst;
81                     if ("java.security.KeyPairGenerator".equals(invoke.getClassName(cpg)) && "getInstance".equals(invoke.getMethodName(cpg))) {
82                         String value = ByteCode.getConstantLDC(location.getHandle().getPrev(), cpg, String.class);
83                         if (value != null && value.toUpperCase().startsWith("RSA")) {
84                             createRsaKeyGen = true;
85                         }
86                     }
87                 } else if (inst instanceof INVOKEVIRTUAL) { //KeyPairGenerator.initialize
88                     INVOKEVIRTUAL invoke = (INVOKEVIRTUAL) inst;
89                     if ("java.security.KeyPairGenerator".equals(invoke.getClassName(cpg)) && "initialize".equals(invoke.getMethodName(cpg))) {
90                         Number n = null;
91                         //init() with one parameter
92                         if ("(I)V".equals(invoke.getSignature(cpg))) {
93                             n = ByteCode.getPushNumber(location.getHandle().getPrev());
94                         }
95                         //init() with a second parameter an instance of SecureRandom
96                         else if ("(ILjava/security/SecureRandom;)V".equals(invoke.getSignature(cpg))) {
97     
98                             SIPUSH push = ByteCode.getPrevInstruction(location.getHandle(), SIPUSH.class);
99                             if (push != null) {
100                                 n = push.getValue();
101                             }
102                         }
103     
104                         if (n != null && n.intValue() < 1024) {
105                             initializeWeakKeyLength = true;
106                             locationWeakness = location;
107                         }
108                     }
109                 } else if (inst instanceof INVOKESPECIAL) { // new RSAKeyGenParameterSpec() is called
110                     INVOKESPECIAL invoke = (INVOKESPECIAL) inst;
111                     if ("java.security.spec.RSAKeyGenParameterSpec".equals(invoke.getClassName(cpg)) && "<init>".equals(invoke.getMethodName(cpg))) {
112                         Number n = null;
113                         //init() with one parameter
114                         if ("(ILjava/math/BigInteger;)V".equals(invoke.getSignature(cpg))) {
115                             SIPUSH push = ByteCode.getPrevInstruction(location.getHandle(), SIPUSH.class);
116                             if (push != null) {
117                                 n = push.getValue();
118                             }
119                         }
120     
121                         if (n != null && n.intValue() < 1024) {
122                             initializeWeakKeyLength = true;
123                             locationWeakness = location;
124                         }
125                     }
126                 }
127             }
128     
129             //Both condition have been found in the same method
130             if (createRsaKeyGen && initializeWeakKeyLength) {
131                 JavaClass clz = classContext.getJavaClass();
132                 bugReporter.reportBug(new BugInstance(this, RSA_KEY_SIZE_TYPE, Priorities.NORMAL_PRIORITY) //
133                         .addClass(clz)
134                         .addMethod(clz, m)
135                         .addSourceLine(classContext, m, locationWeakness));
136             }
137         }
138     
139         @Override
140         public void report() {
141         }
142     }
