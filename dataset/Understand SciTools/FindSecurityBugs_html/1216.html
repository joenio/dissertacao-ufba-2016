18     package com.h3xstream.findsecbugs.crypto;
19     
20     import com.h3xstream.findsecbugs.common.ByteCode;
21     import edu.umd.cs.findbugs.BugInstance;
22     import edu.umd.cs.findbugs.Priorities;
23     import edu.umd.cs.findbugs.ba.*;
24     import org.apache.bcel.generic.*;
25     import edu.umd.cs.findbugs.BugReporter;
26     import edu.umd.cs.findbugs.Detector;
27     import org.apache.bcel.classfile.JavaClass;
28     import org.apache.bcel.classfile.Method;
29     
30     import java.util.Iterator;
31     
32     public class InsufficientKeySizeBlowfishDetector implements Detector {
33     
34         private static final String BLOWFISH_KEY_SIZE_TYPE = "BLOWFISH_KEY_SIZE";
35         private BugReporter bugReporter;
36     
37         public InsufficientKeySizeBlowfishDetector(BugReporter bugReporter) {
38             this.bugReporter = bugReporter;
39         }
40     
41         @Override
42         public void visitClassContext(ClassContext classContext) {
43             JavaClass javaClass = classContext.getJavaClass();
44     
45             Method[] methodList = javaClass.getMethods();
46     
47             for (Method m : methodList) {
48     
49                 try {
50                     analyzeMethod(m, classContext);
51                 } catch (CFGBuilderException e) {
52                 } catch (DataflowAnalysisException e) {
53                 }
54             }
55         }
56     
57         private void analyzeMethod(Method m, ClassContext classContext) throws CFGBuilderException, DataflowAnalysisException {
58     
59             //Conditions that needs to fill to identify the vulnerability
60             boolean createBlowfishKeyGen = false;
61             boolean initializeWeakKeyLength = false;
62             Location locationWeakness = null;
63     
64             ConstantPoolGen cpg = classContext.getConstantPoolGen();
65             CFG cfg = classContext.getCFG(m);
66     
67             for (Iterator<Location> i = cfg.locationIterator(); i.hasNext(); ) {
68                 Location location = i.next();
69     
70                 Instruction inst = location.getHandle().getInstruction();
71     //            ByteCode.printOpCode(inst, cpg);
72     
73                 if (inst instanceof INVOKESTATIC) { //MessageDigest.digest is called
74                     INVOKESTATIC invoke = (INVOKESTATIC) inst;
75                     if ("javax.crypto.KeyGenerator".equals(invoke.getClassName(cpg)) && "getInstance".equals(invoke.getMethodName(cpg))) {
76                         String value = ByteCode.getConstantLDC(location.getHandle().getPrev(), cpg, String.class);
77                         if ("Blowfish".equals(value)) {
78                             createBlowfishKeyGen = true;
79                         }
80                     }
81                 } else if (inst instanceof INVOKEVIRTUAL) {
82                     INVOKEVIRTUAL invoke = (INVOKEVIRTUAL) inst;
83                     if ("javax.crypto.KeyGenerator".equals(invoke.getClassName(cpg)) && "init".equals(invoke.getMethodName(cpg))) {
84                         Number n = null;
85                         //init() with one parameter
86                         if ("(I)V".equals(invoke.getSignature(cpg))) {
87                             n = ByteCode.getPushNumber(location.getHandle().getPrev());
88                         }
89                         //init() with a second parameter an instance of SecureRandom
90                         else if ("(ILjava/security/SecureRandom;)V".equals(invoke.getSignature(cpg))) {
91     
92                             BIPUSH push = ByteCode.getPrevInstruction(location.getHandle(), BIPUSH.class);
93                             if (push != null) {
94                                 n = push.getValue();
95                             }
96                         }
97     
98                         if (n != null && n.intValue() < 128) {
99                             initializeWeakKeyLength = true;
100                             locationWeakness = location;
101                         }
102                     }
103                 }
104             }
105     
106             //Both condition have been found in the same method
107             if (createBlowfishKeyGen && initializeWeakKeyLength) {
108                 JavaClass clz = classContext.getJavaClass();
109                 bugReporter.reportBug(new BugInstance(this, BLOWFISH_KEY_SIZE_TYPE, Priorities.NORMAL_PRIORITY) //
110                         .addClass(clz)
111                         .addMethod(clz, m)
112                         .addSourceLine(classContext, m, locationWeakness));
113             }
114         }
115     
116         @Override
117         public void report() {
118         }
119     }
