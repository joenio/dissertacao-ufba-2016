37     /**
38      * <p>
39      *  The main goal of the this detector is to find encryption being done with static initialization vector (IV).
40      *  By design, the IV should be change for every message encrypt by a system.
41      * </p>
42      * <h3>Note on the implementation</h3>
43      * <p>
44      *  The strategy to find those occurrences is not to backtrack to find the potential source of the bytes being passed.
45      *  It will not be trigger if SecureRandom instance is use. Therefor, it is very likely to trigger false positive if the
46      *  encryption is separate from the IV generation.
47      * </p>
48      */
49     public class StaticIvDetector implements Detector {
50     
51         private static final boolean DEBUG = false;
52         private static final String STATIC_IV = "STATIC_IV";
53         private BugReporter bugReporter;
54     
55         public StaticIvDetector(BugReporter bugReporter) {
56             this.bugReporter = bugReporter;
57         }
58     
59     
60         @Override
61         public void visitClassContext(ClassContext classContext) {
62             JavaClass javaClass = classContext.getJavaClass();
63     
64             Method[] methodList = javaClass.getMethods();
65     
66             for (Method m : methodList) {
67     
68                 try {
69                     analyzeMethod(m,classContext);
70                 } catch (CFGBuilderException e) {
71                 } catch (DataflowAnalysisException e) {
72                 }
73             }
74         }
75     
76     
77         private void analyzeMethod(Method m, ClassContext classContext) throws CFGBuilderException, DataflowAnalysisException {
78     
79             ConstantPoolGen cpg = classContext.getConstantPoolGen();
80             CFG cfg = classContext.getCFG(m);
81     
82             boolean foundSafeIvGeneration = false;
83             //Detect if the method is doing decryption only. If it is the case, IV should not be generated from this point
84             //therefore it is a false positive
85             boolean atLeastOneDecryptCipher = false;
86             boolean atLeastOneEncryptCipher = false;
87     
88             boolean ivFetchFromCipher = false;
89     
90             //First pass : it look for encryption and decryption mode to detect if the method does decryption only
91             for (Iterator<Location> i = cfg.locationIterator(); i.hasNext(); ) {
92                 Location location = nextLocation(i, cpg);
93                 Instruction inst = location.getHandle().getInstruction();
94     
95     
96                 //ByteCode.printOpCode(inst,cpg);
97     
98     
99                 if (inst instanceof INVOKEVIRTUAL) {
100                     INVOKEVIRTUAL invoke = (INVOKEVIRTUAL) inst;
101     
102                     //INVOKEVIRTUAL javax/crypto/Cipher.init ((ILjava/security/Key;)V)
103                     if (("javax.crypto.Cipher").equals(invoke.getClassName(cpg)) &&
104                             "init".equals(invoke.getMethodName(cpg))) {
105     
106                         ICONST iconst = ByteCode.getPrevInstruction(location.getHandle(), ICONST.class);
107                         if (iconst != null) {
108                             int mode = iconst.getValue().intValue();
109     
110                             switch (mode) {
111                                 case Cipher.ENCRYPT_MODE:
112                                     atLeastOneEncryptCipher = true;
113                                     break;
114                                 case Cipher.DECRYPT_MODE:
115                                     atLeastOneDecryptCipher = true;
116                                     break;
117                             }
118                         }
119                     }
120     
121                     //INVOKEVIRTUAL javax/crypto/Cipher.getIV (()[B)
122                     else if (("javax.crypto.Cipher").equals(invoke.getClassName(cpg)) &&
123                             "getIV".equals(invoke.getMethodName(cpg))) {
124                         ivFetchFromCipher = true;
125                     }
126                 }
127             }
128     
129             //Second pass : It look for encryption method and a potential preceding SecureRandom usage
130             for (Iterator<Location> i = cfg.locationIterator(); i.hasNext(); ) {
131                 Location location = nextLocation(i, cpg);
132                 Instruction inst = location.getHandle().getInstruction();
133     
134                 if (inst instanceof INVOKEVIRTUAL) {
135                     INVOKEVIRTUAL invoke = (INVOKEVIRTUAL) inst;
136                     if ("java.security.SecureRandom".equals(invoke.getClassName(cpg)) &&
137                             "nextBytes".equals(invoke.getMethodName(cpg))) {
138                         foundSafeIvGeneration = true;
139                     }
140                 }
141     
142                 else if (inst instanceof INVOKESPECIAL &&
143                         !ivFetchFromCipher //IV was generate with the KeyGenerator
144                         && (!atLeastOneDecryptCipher || atLeastOneEncryptCipher) //The cipher is in decrypt mode (no iv generation)
145                         && !foundSafeIvGeneration) {
146                     INVOKESPECIAL invoke = (INVOKESPECIAL) inst;
147                     if (("javax.crypto.spec.IvParameterSpec").equals(invoke.getClassName(cpg)) &&
148                             "<init>".equals(invoke.getMethodName(cpg))) {
149     
150                         JavaClass clz = classContext.getJavaClass();
151                         bugReporter.reportBug(new BugInstance(this, STATIC_IV, Priorities.NORMAL_PRIORITY) //
152                                 .addClass(clz)
153                                 .addMethod(clz,m)
154                                 .addSourceLine(classContext,m,location));
155                     }
156                 }
157             }
158         }
159     
160         private Location nextLocation(Iterator<Location> i,ConstantPoolGen cpg) {
161             Location loc = i.next();
162             if(DEBUG) ByteCode.printOpCode(loc.getHandle().getInstruction(), cpg);
163             return loc;
164         }

169     
170         }
