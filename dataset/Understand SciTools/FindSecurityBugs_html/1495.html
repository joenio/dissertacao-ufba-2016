21     import edu.umd.cs.findbugs.ba.CFG;
22     import edu.umd.cs.findbugs.ba.DepthFirstSearch;
23     import edu.umd.cs.findbugs.classfile.CheckedAnalysisException;

25     import edu.umd.cs.findbugs.classfile.IMethodAnalysisEngine;
26     import edu.umd.cs.findbugs.classfile.MethodDescriptor;
27     import java.io.BufferedWriter;
28     import java.io.FileNotFoundException;
29     import java.io.FileOutputStream;
30     import java.io.IOException;
31     import java.io.InputStream;
32     import java.io.OutputStreamWriter;
33     import java.io.UnsupportedEncodingException;
34     import java.io.Writer;
35     import org.apache.bcel.generic.MethodGen;
36     
37     /**
38      * Requests or creates needed objects and execute taint analysis,
39      * extends taint method summaries with analyzed methods
40      * 
41      * @author David Formanek
42      */
43     public class TaintDataflowEngine implements IMethodAnalysisEngine<TaintDataflow> {
44     
45         private static final String METHODS_SUMMARIES_PATH = "taint-config/";
46         private static final String[] METHODS_SUMMARIES_FILENAMES = {
47             "java-lang.txt",
48             "java-ee.txt",
49             "collections.txt",
50             "java-net.txt",
51             "scala.txt",
52             "other.txt",
53         };
54         private static final String SAFE_ENCODERS_PATH = "safe-encoders/";
55         private static final String[] SAFE_ENCODERS_FILENAMES = {
56             "xss-encoders.txt"
57         };
58     
59         private final TaintMethodSummaryMap methodSummaries = new TaintMethodSummaryMap();
60     
61         private static Writer writer = null;
62         
63         static {
64             if (FindSecBugsGlobalConfig.getInstance().isDebugOutputSummaries()) {
65                 try {
66                     writer = new BufferedWriter(new OutputStreamWriter(
67                             new FileOutputStream("derived-summaries.txt"), "utf-8"));
68                 } catch (UnsupportedEncodingException ex) {
69                     assert false : ex.getMessage();
70                 } catch (FileNotFoundException ex) {
71                     System.err.println(ex.getMessage());
72                 }
73             }
74         }
75     
76         public TaintDataflowEngine() {
77             for (String path : METHODS_SUMMARIES_FILENAMES) {
78                 loadMethodSummaries(METHODS_SUMMARIES_PATH.concat(path));
79             }
80             for (String path : SAFE_ENCODERS_FILENAMES) {
81                 loadMethodSummaries(SAFE_ENCODERS_PATH.concat(path));
82             }
83         }
84         
85         private void loadMethodSummaries(String path) {
86             assert path != null && !path.isEmpty();
87             InputStream stream = null;
88             try {
89                 stream = getClass().getClassLoader().getResourceAsStream(path);
90                 methodSummaries.load(stream);
91             } catch (IOException ex) {
92                 throw new RuntimeException("cannot load summaries from " + path, ex);
93             } finally {
94                 if (stream != null) {
95                     try {
96                         stream.close();
97                     } catch (IOException ex) {
98                         throw new RuntimeException("cannot close stream", ex);
99                     }
100                 }
101             }
102         }
103         
104         @Override
105         public TaintDataflow analyze(IAnalysisCache cache, MethodDescriptor descriptor)
106                 throws CheckedAnalysisException {
107             CFG cfg = cache.getMethodAnalysis(CFG.class, descriptor);
108             DepthFirstSearch dfs = cache.getMethodAnalysis(DepthFirstSearch.class, descriptor);
109             MethodGen methodGen = cache.getMethodAnalysis(MethodGen.class, descriptor);
110             TaintAnalysis analysis = new TaintAnalysis(methodGen, dfs, descriptor, methodSummaries);
111             TaintDataflow flow = new TaintDataflow(cfg, analysis);
112             flow.execute();
113             analysis.finishAnalysis();
114             if (FindSecBugsGlobalConfig.getInstance().isDebugOutputSummaries() && writer != null) {
115                 TaintMethodSummary derivedSummary = methodSummaries.get(getSlashedMethodName(methodGen));
116                 if (derivedSummary != null) {
117                     try {
118                         writer.append(getSlashedMethodName(methodGen) + ":" + derivedSummary + "\n");
119                         writer.flush();
120                     } catch (IOException ex) {
121                         System.out.println("cannot write: " + ex.getMessage());
122                     }
123                 }
124             }
125             return flow;
126         }
127     
128         private static String getSlashedMethodName(MethodGen methodGen) {
129             String methodNameWithSignature = methodGen.getName() + methodGen.getSignature();
130             String slashedClassName = methodGen.getClassName().replace('.', '/');
131             return slashedClassName + "." + methodNameWithSignature;
132         }
133         
134         @Override
135         public void registerWith(IAnalysisCache iac) {
136             iac.registerMethodAnalysisEngine(TaintDataflow.class, this);
137         }
138     }
