18     package com.h3xstream.findsecbugs.injection;
19     
20     import edu.umd.cs.findbugs.BugReporter;
21     import java.io.BufferedReader;
22     import java.io.IOException;
23     import java.io.InputStream;
24     import java.io.InputStreamReader;
25     
26     /**
27      * Detector designed for extension to detect basic injection with methods
28      * configured in resource files using specified format
29      * 
30      * @author David Formanek (Y Soft Corporation, a.s.)
31      */
32     public abstract class ConfiguredBasicInjectionDetector extends BasicInjectionDetector {
33     
34         private static final String SINK_CONFIG_PATH = "injection-sinks/";
35         
36         public ConfiguredBasicInjectionDetector(BugReporter bugReporter) {
37             super(bugReporter);
38         }
39         
40         protected void loadConfiguredSinks(String filename, String bugType) {
41             assert filename != null && bugType != null && !bugType.isEmpty();
42             InputStream stream = null;
43             try {
44                 stream = getClass().getClassLoader()
45                         .getResourceAsStream(SINK_CONFIG_PATH.concat(filename));
46                 loadSinks(stream, bugType);
47             } catch (IOException ex) {
48                 throw new RuntimeException("cannot load resources", ex);
49             } finally {
50                 if (stream != null) {
51                     try {
52                         stream.close();
53                     } catch (IOException ex) {
54                         throw new RuntimeException("cannot close stream", ex);
55                     }
56                 }
57             }
58         }
59         
60         protected void loadSinks(InputStream input, String bugType) throws IOException {
61             assert input != null && bugType != null && !bugType.isEmpty();
62             BufferedReader reader = new BufferedReader(new InputStreamReader(input, "UTF-8"));
63             for (;;) {
64                 String line = reader.readLine();
65                 if (line == null) {
66                     break;
67                 }
68                 line = line.trim();
69                 if (line.isEmpty() || line.startsWith("-")) {
70                     continue;
71                 }
72                 loadSink(line, bugType);
73             }
74         }
75         
76         protected void loadSink(String line, String bugType) {
77             assert line != null && bugType != null && !bugType.isEmpty();
78             String[] split = line.split("\\:");
79             if (split.length != 2) {
80                 throw new IllegalArgumentException("There must be exactly one ':' in " + line);
81             }
82             String[] stringArguments = split[1].split(",");
83             int length = stringArguments.length;
84             if (length == 0) {
85                 throw new IllegalArgumentException("no injectable parameters specified in " + line);
86             }
87             
88             int[] intArguments = new int[length];
89             for (int i = 0; i < length; i++) {
90                 try {
91                     intArguments[i] = Integer.parseInt(stringArguments[i]);
92                 } catch (NumberFormatException ex) {
93                     throw new IllegalArgumentException("cannot parse " + stringArguments[i], ex);
94                 }
95             }
96             addSink(split[0], intArguments, bugType);
97         }
98     }
