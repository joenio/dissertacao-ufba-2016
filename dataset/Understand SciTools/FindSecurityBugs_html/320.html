26     import org.apache.bcel.classfile.JavaClass;

28     import org.apache.bcel.generic.*;
29     
30     import java.util.Iterator;
31     
32     /**
33      * <p>
34      * The first reflex for developer that encounter web services that have unsigned certificate
35      * is often to trust all certificates.
36      * </p>
37      * <p>
38      * To trust everything, the standard API for SSL communication requires the implementation of a child
39      * interface of "javax.net.ssl.TrustManager" (marker interface). Commonly, X509TrustManager is being used.
40      * </p>
41      * <a href="http://stackoverflow.com/a/1201102/89769">Sample of code being used</a>
42      *
43      * @see javax.net.ssl.TrustManager
44      * @see javax.net.ssl.X509TrustManager
45      */
46     public class WeakTrustManagerDetector implements Detector {
47     
48         private static final boolean DEBUG = false;
49         private static final String WEAK_TRUST_MANAGER_TYPE = "WEAK_TRUST_MANAGER";
50         private static final String WEAK_HOSTNAME_VERIFIER_TYPE = "WEAK_HOSTNAME_VERIFIER";
51         private BugReporter bugReporter;
52     
53         public WeakTrustManagerDetector(BugReporter bugReporter) {
54             this.bugReporter = bugReporter;
55         }
56     
57         @Override
58         public void visitClassContext(ClassContext classContext) {
59             JavaClass javaClass = classContext.getJavaClass();
60     
61             //The class extends X509TrustManager
62             boolean isTrustManager = InterfaceUtils.isSubtype(javaClass, "javax.net.ssl.X509TrustManager");
63             boolean isHostnameVerifier = InterfaceUtils.isSubtype(javaClass, "javax.net.ssl.HostnameVerifier");
64     
65             //Not the target of this detector
66             if (!isTrustManager && !isHostnameVerifier) return;
67     
68             Method[] methodList = javaClass.getMethods();
69     
70             for (Method m : methodList) {
71                 MethodGen methodGen = classContext.getMethodGen(m);
72     
73                 if (DEBUG) System.out.println(">>> Method: " + m.getName());
74     
75                 if (isTrustManager &&
76                         (m.getName().equals("checkClientTrusted") ||
77                         m.getName().equals("checkServerTrusted") ||
78                         m.getName().equals("getAcceptedIssuers"))) {
79     
80     
81                     if(isEmptyImplementation(methodGen)) {
82                         bugReporter.reportBug(new BugInstance(this, WEAK_TRUST_MANAGER_TYPE, Priorities.NORMAL_PRIORITY) //
83                                 .addClassAndMethod(javaClass, m));
84                     }
85                 }
86                 else if (isHostnameVerifier && m.getName().equals("verify")) {
87     
88                     if(isEmptyImplementation(methodGen)) {
89                         bugReporter.reportBug(new BugInstance(this, WEAK_HOSTNAME_VERIFIER_TYPE, Priorities.NORMAL_PRIORITY) //
90                                 .addClassAndMethod(javaClass, m));
91                     }
92                 }
93     
94             }
95         }
96     
97         /**
98          * Currently the detection is pretty weak.
99          * It will catch Dummy implementation that have empty method implementation
100          *
101          * @return If the implementation is "empty" (direct return or dummy code)
102          */
103         private boolean isEmptyImplementation(MethodGen methodGen){
104             boolean invokeInst = false;
105             boolean loadField = false;
106     
107             for (Iterator itIns = methodGen.getInstructionList().iterator();itIns.hasNext();) {
108                 Instruction inst = ((InstructionHandle) itIns.next()).getInstruction();
109                 if (DEBUG)
110                     System.out.println(inst.toString(true));
111     
112                 if (inst instanceof InvokeInstruction) {
113                     invokeInst = true;
114                 }
115                 if (inst instanceof GETFIELD) {
116                     loadField = true;
117                 }
118             }
119             return !invokeInst && !loadField;
120         }
121     
122         @Override
123         public void report() {
124     
125         }
126     }
