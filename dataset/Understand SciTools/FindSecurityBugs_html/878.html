48     /**
49      * Detector designed for extension to detect injection vulnerabilities
50      *
51      * @author David Formanek (Y Soft Corporation, a.s.)
52      */
53     public abstract class AbstractInjectionDetector extends AbstractTaintDetector {
54         
55         protected final Map<String, Set<TaintSink>> methodsWithSinks = new HashMap<String, Set<TaintSink>>();
56         
57         protected AbstractInjectionDetector(BugReporter bugReporter) {
58             super(bugReporter);
59         }
60     
61         /**
62          * Once the analysis is completed, all the collected sink are report as bug.
63          */
64         @Override
65         public void report() {
66             //The HashSet data structure was chosen to remove duplicate bug instances
67             Set<BugInstance> bugs = new HashSet<BugInstance>();
68             for (Set<TaintSink> sinkSet : methodsWithSinks.values()) {
69                 for (TaintSink sink : sinkSet) {
70                     bugs.add(sink.getBugInstance());
71                 }
72             }
73             for (BugInstance bug : bugs) {
74                 bugReporter.reportBug(bug);
75             }
76         }
77         
78         @Override
79         protected void analyzeLocation(ClassContext classContext, Method method, InstructionHandle handle,
80                 ConstantPoolGen cpg, InvokeInstruction invoke, TaintFrame fact, String currentMethod)
81                 throws DataflowAnalysisException {
82             SourceLineAnnotation sourceLine = SourceLineAnnotation.fromVisitedInstruction(classContext, method, handle);
83             checkTaintSink(cpg, invoke, fact, sourceLine, currentMethod);
84             InjectionPoint injectionPoint = getInjectionPoint(invoke, cpg, handle);
85     
86             for (int offset : injectionPoint.getInjectableArguments()) {
87                 Taint parameterTaint = fact.getStackValue(offset);
88                 int priority = getPriority(parameterTaint);
89                 if (priority == Priorities.IGNORE_PRIORITY) {
90                     continue;
91                 }
92                 BugInstance bugInstance = new BugInstance(this, injectionPoint.getBugType(), priority);
93                 bugInstance.addClassAndMethod(classContext.getJavaClass(), method);
94                 bugInstance.addSourceLine(sourceLine);
95                 if (injectionPoint.getInjectableMethod() != null) {
96                     bugInstance.addString(injectionPoint.getInjectableMethod());
97                 }
98                 reportBug(bugInstance, parameterTaint, currentMethod);
99                 return;
100             }
101         }
102     
103         /**
104          * The default implementation of <code>getPriority()</code> can be overridden if the severity and the confidence for risk
105          * is particular.
106          *
107          * By default, injection will be rated "High" if the complete link between source and sink is made.
108          * If it is not the case but concatenation with external source is made, "Medium" is used.
109          *
110          * @param taint Detail about the state of the value passed (Cumulative information leading to the variable passed).
111          * @return Priorities interface values from 1 to 5 (Enum-like interface)
112          */
113         protected int getPriority(Taint taint) {
114             if (taint.isTainted()) {
115                 return Priorities.HIGH_PRIORITY;
116             } else if (!taint.isSafe()) {
117                 return Priorities.NORMAL_PRIORITY;
118             } else {
119                 return Priorities.IGNORE_PRIORITY;
120             }
121         }
122         
123         private void checkTaintSink(ConstantPoolGen cpg, InvokeInstruction invoke, TaintFrame fact,
124                 SourceLineAnnotation sourceLine, String currentMethod) throws DataflowAnalysisException {
125             for (TaintSink sink : getSinks(cpg, invoke, fact)) {
126                 Taint sinkTaint = sink.getTaint();
127                 Set<Integer> taintParameters = sinkTaint.getParameters();
128                 Taint finalTaint = Taint.valueOf(sinkTaint.getNonParametricState());
129                 for (Integer offset : taintParameters) {
130                     Taint parameterTaint = fact.getStackValue(offset);
131                     finalTaint = Taint.merge(finalTaint, parameterTaint);
132                 }
133                 if (finalTaint == null) {
134                     continue;
135                 }
136                 if (finalTaint.isTainted() || finalTaint.hasParameters()) {
137                     BugInstance bugInstance = sink.getBugInstance();
138                     bugInstance.addSourceLine(sourceLine);
139                     addSourceLines(finalTaint.getLocations(), bugInstance);
140                     if (finalTaint.isTainted()) {
141                         bugInstance.setPriority(getPriority(finalTaint));
142                     } else {
143                         assert finalTaint.isUnknown();
144                         delayBugToReport(currentMethod, finalTaint, bugInstance);
145                     }
146                 }
147             }
148         }
149     
150         private Set<TaintSink> getSinks(ConstantPoolGen cpg, InvokeInstruction invoke, TaintFrame frame) {
151             String className = getInstanceClassName(cpg, invoke, frame);
152             String methodName = "." + invoke.getMethodName(cpg) + invoke.getSignature(cpg);
153             Set<TaintSink> sinks = methodsWithSinks.get(className.concat(methodName));
154             if (sinks != null) {
155                 return sinks;
156             }
157             try {
158                 JavaClass javaClass = Repository.lookupClass(className);
159                 assert javaClass != null;
160                 return getSuperSinks(javaClass, methodName);
161             } catch (ClassNotFoundException ex) {
162                 AnalysisContext.reportMissingClass(ex);
163             }
164             return Collections.emptySet();
165         }
166         
167         private Set<TaintSink> getSuperSinks(JavaClass javaClass, String method) throws ClassNotFoundException {
168             for (JavaClass superClass : javaClass.getSuperClasses()) {
169                 String fullMethodName = superClass.getClassName().replace('.', '/').concat(method);
170                 Set<TaintSink> sinks = methodsWithSinks.get(fullMethodName);
171                 if (sinks != null) {
172                     return sinks;
173                 }
174             }
175             for (JavaClass interfaceClass : javaClass.getAllInterfaces()) {
176                 String fullMethodName = interfaceClass.getClassName().replace('.', '/').concat(method);
177                 Set<TaintSink> sinks = methodsWithSinks.get(fullMethodName);
178                 if (sinks != null) {
179                     return sinks;
180                 }
181             }
182             return Collections.emptySet();
183         }
184         
185         private void reportBug(BugInstance bugInstance, Taint taint, String currentMethod) {
186             addSourceLines(taint.getLocations(), bugInstance);
187             if (shouldBugBeDelayed(taint, bugInstance)) {
188                 delayBugToReport(currentMethod, taint, bugInstance);
189             } else {
190                 bugReporter.reportBug(bugInstance);
191             }
192         }
193     
194         private boolean shouldBugBeDelayed(Taint taint, BugInstance bugInstance) {
195             if (!taint.hasParameters()) {
196                 return false;
197             }
198             return bugInstance.getPriority() == getPriority(new Taint(Taint.State.UNKNOWN));
199         }
200         
201         private void delayBugToReport(String method, Taint taint, BugInstance bug) {
202             TaintSink taintSink = new TaintSink(taint, bug);
203             Set<TaintSink> sinkSet = methodsWithSinks.get(method);
204             if (sinkSet == null) {
205                 sinkSet = new HashSet<TaintSink>();
206             }
207             sinkSet.add(taintSink);
208             methodsWithSinks.put(method, sinkSet);
209         }

211         private static void addSourceLines(Collection<TaintLocation> locations, BugInstance bugInstance) {
212             List<SourceLineAnnotation> annotations = new LinkedList<SourceLineAnnotation>();
213             for (TaintLocation location : locations) {
214                 SourceLineAnnotation taintSource = SourceLineAnnotation.fromVisitedInstruction(
215                         location.getMethodDescriptor(), location.getPosition());
216                 annotations.add(taintSource);
217             }
218             Collections.sort(annotations);
219             SourceLineAnnotation annotation = null;
220             for (Iterator<SourceLineAnnotation> it = annotations.iterator(); it.hasNext();) {
221                 SourceLineAnnotation prev = annotation;
222                 annotation = it.next();
223                 if (prev != null && prev.getClassName().equals(annotation.getClassName())
224                         && prev.getStartLine() == annotation.getStartLine()) {
225                     // keep only one annotation per line
226                     it.remove();
227                 }
228             }
229             for (SourceLineAnnotation sourceLine : annotations) {
230                 bugInstance.addSourceLine(sourceLine);
231             }
232         }

234         private static String getInstanceClassName(ConstantPoolGen cpg, InvokeInstruction invoke, TaintFrame frame) {
235             try {
236                 int instanceIndex = frame.getNumArgumentsIncludingObjectInstance(invoke, cpg) - 1;
237                 if (instanceIndex != -1) {
238                     assert instanceIndex < frame.getStackDepth();
239                     Taint instanceTaint = frame.getStackValue(instanceIndex);
240                     String className = instanceTaint.getRealInstanceClassName();
241                     if (className != null) {
242                         return className;
243                     }
244                 }
245             } catch (DataflowAnalysisException ex) {
246                 assert false : ex.getMessage();
247             }
248             String dottedClassName = invoke.getReferenceType(cpg).toString();
249             return ClassName.toSlashedClassName(dottedClassName);
250         }
