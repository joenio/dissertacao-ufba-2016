61         public TaintAnalysis(MethodGen methodGen, DepthFirstSearch dfs,
62                 MethodDescriptor descriptor, TaintMethodSummaryMap methodSummaries) {
63             super(dfs);
64             this.methodGen = methodGen;
65             this.methodDescriptor = (MethodInfo) descriptor;
66             this.visitor = new TaintFrameModelingVisitor(methodGen.getConstantPool(), descriptor, methodSummaries);
67             computeParametersInfo(descriptor.getSignature(), descriptor.isStatic());
68         }

70         @Override
71         protected void mergeValues(TaintFrame frame, TaintFrame result, int i)
72                 throws DataflowAnalysisException {
73             result.setValue(i, Taint.merge(result.getValue(i), frame.getValue(i)));
74         }

76         @Override
77         public void transferInstruction(InstructionHandle handle, BasicBlock block, TaintFrame fact)
78                 throws DataflowAnalysisException {
79             visitor.setFrameAndLocation(fact, new Location(handle, block));
80             visitor.analyzeInstruction(handle.getInstruction());
81         }

83         @Override
84         public TaintFrame createFact() {
85             return new TaintFrame(methodGen.getMaxLocals());
86         }

88         /**
89          * Initialize the initial state of a TaintFrame.
90          * @param fact Initial frame
91          */
92         @Override
93         public void initEntryFact(TaintFrame fact) {
94             fact.setValid();
95             fact.clearStack();
96             boolean inMainMethod = isInMainMethod();
97             int numSlots = fact.getNumSlots();
98             int numLocals = fact.getNumLocals();
99             for (int i = 0; i < numSlots; ++i) {
100                 Taint value = new Taint(Taint.State.UNKNOWN);
101                 if (i < numLocals) {
102                     if (i < parameterStackSize) {
103                         if (inMainMethod || isTaintedByAnnotation(i - 1)) {
104                             value = new Taint(Taint.State.TAINTED);
105                             // this would add line number for the first instruction in the main method
106                             //value.addLocation(new TaintLocation(methodDescriptor, 0), true);
107                         } else {
108                             int stackOffset = parameterStackSize - i - 1;
109                             value.addParameter(stackOffset);
110                         }
111                     }
112                     value.setVariableIndex(i);
113                 }
114                 fact.setValue(i, value);
115             }
116         }
117     
118         /**
119          * @return true if the method is the startup point of a console or gui application
120          * ("public static void main(String[] args)"), false otherwise
121          */
122         private boolean isInMainMethod() {
123             return methodDescriptor.isStatic()
124                     && "main".equals(methodDescriptor.getName())
125                     && "([Ljava/lang/String;)V".equals(methodDescriptor.getSignature())
126                     && methodGen.getMethod().isPublic();
127         }
128     
129         /**
130          * Determine if the slot value is tainted based on added framework annotations.
131          * <hr/>
132          * Example of mapping done:<br/>
133          *
134          * Method analyzed: "test(String,String,double)V"
135          * For the slot no 2, it look at annotations on the parameter 2.
136          * <pre>
137          *          0               1               2               3
138          *                                    +++++++++++++
139          *  [ double param3 | double param3 | String param2 | String param1 ]
140          * </pre>
141          * <br/>
142          * (Reminder : double and long take two slots for one parameter)
143          *
144          * @param slotNo
145          * @return
146          */
147         private boolean isTaintedByAnnotation(int slotNo) {
148             if (slotNo >= 0 && methodDescriptor.hasParameterAnnotations()) {
149                 int parameter = slotToParameter.get(slotNo);
150                 Collection<AnnotationValue> annotations = methodDescriptor.getParameterAnnotations(parameter);
151                 for (AnnotationValue annotation : annotations) {
152                     if (TAINTED_ANNOTATIONS.contains(annotation.getAnnotationClass().getClassName())) {
153                         return true;
154                     }
155                 }
156             }
157             return false;
158         }

160         @Override
161         public void meetInto(TaintFrame fact, Edge edge, TaintFrame result)
162                 throws DataflowAnalysisException {
163             if (fact.isValid() && edge.isExceptionEdge()) {
164                 TaintFrame copy = null;
165                 // creates modifiable copy
166                 copy = modifyFrame(fact, copy);
167                 copy.clearStack();
168                 // do not trust values that are safe just when an exception occurs
169                 copy.pushValue(new Taint(Taint.State.UNKNOWN));
170                 fact = copy;
171             }
172             mergeInto(fact, result);
173         }

182         /**
183          * Compute two values:
184          *  <li>The number of value require on the stack for a specific call.</li>
185          *  <li>The parameter association with each slot.</li>
186          *
187          * <hr/>
188          *
189          * For exemple : "(I[Ljava/lang/String;Ljava/lang/Object;)V" =>  void a(Integer, String, Object)<br/><br/>
190          * Expected Stack prior the method invocation :<br/>
191          * <pre>
192          * +--------------+
193          * | 0: Object    |
194          * +--------------+
195          * | 1: String    |
196          * +--------------+
197          * | 2: int       |
198          * +--------------+
199          * </pre>
200          * <hr/>
201          *
202          * Special note: double and long (primitive types) value take two slots.<br/>
203          * For exemple : "(IDJ)V" =>  void a(Integer, Double, Long)
204          * <pre>
205          * +--------------+
206          * | 0: long  pt1 |
207          * +--------------+
208          * | 1: long  pt2 |
209          * +--------------+
210          * | 2: double pt1|
211          * +--------------+
212          * | 3: double pt2|
213          * +--------------+
214          * | 4: int       |
215          * +--------------+
216          * </pre>
217          *
218          * @param signature
219          * @param isStatic
220          * @return
221          */
222         private void computeParametersInfo(String signature, boolean isStatic) {
223             assert signature != null && !signature.isEmpty();
224             // static methods does not have reference to this
225             int stackSize = isStatic ? 0 : 1;
226             GenericSignatureParser parser = new GenericSignatureParser(signature);
227             Iterator<String> iterator = parser.parameterSignatureIterator();
228             int paramIdx = 0;
229             slotToParameter = new ArrayList<Integer>();
230             while (iterator.hasNext()) {
231                 String parameter = iterator.next();
232                 if (parameter.equals("D") || parameter.equals("J")) {
233                     // double and long types takes two slots
234                     stackSize += 2;
235                     slotToParameter.add(paramIdx);
236                     slotToParameter.add(paramIdx);
237                 } else {
238                     stackSize++;
239                     slotToParameter.add(paramIdx);
240                 }
241                 paramIdx++;
242             }
243             parameterStackSize = stackSize;
244         }

246         private static List<String> loadFileContent(String path) {
247             try {
248                 InputStream in = TaintAnalysis.class.getClassLoader().getResourceAsStream(path);
249                 BufferedReader stream = new BufferedReader(new InputStreamReader(in));
250                 String line;
251                 List<String> content = new ArrayList<String>();
252                 while ((line = stream.readLine()) != null) {
253                     content.add(line.trim());
254                 }
255                 stream.close();
256                 return content;
257             } catch (IOException e) {
258                 LOG.log(Level.SEVERE, "Unable to load data from {0}", path);
259             }
260             return new ArrayList<String>();
261         }
