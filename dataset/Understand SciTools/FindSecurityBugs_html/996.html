34     import java.util.Iterator;

36     /**
37      * The SaxParser use the Xerces XML Parser engine.
38      * <ul>
39      *   <li>SAXParser/XMLReader -&gt; Xerces 1</li>
40      *   <li>DocumentBuilder -&gt; Xerces 2</li>
41      * </ul>
42      *
43      * 3 equivalent APIs are covered by this detector.
44      * <ul>
45      *   <li>SAXParser.parse()</li>
46      *   <li>XMLReader.parse()</li>
47      *   <li>DocumentBuilder.parse()</li>
48      * </ul>
49      *
50      * References:
51      * <ul>
52      *   <li>https://www.securecoding.cert.org/confluence/pages/viewpage.action?pageId=61702260</li>
53      *   <li>https://www.owasp.org/index.php/XML_External_Entity_%28XXE%29_Processing</li>
54      * </ul>
55      */
56     public class SaxParserXxeDetector extends OpcodeStackDetector {
57         private static final String XXE_SAX_PARSER_TYPE = "XXE_SAXPARSER";
58         private static final String XXE_XML_READER_TYPE = "XXE_XMLREADER";
59         private static final String XXE_DOCUMENT_TYPE = "XXE_DOCUMENT";
60     
61         private static final String FEATURE_DISALLOW_DTD = "http://apache.org/xml/features/disallow-doctype-decl";
62         private static final String FEATURE_SECURE_PROCESSING = "http://javax.xml.XMLConstants/feature/secure-processing";
63     
64         private static final String FEATURE_GENERAL_ENTITIES = "http://xml.org/sax/features/external-general-entities";
65         private static final String FEATURE_EXTERNAL_ENTITIES = "http://xml.org/sax/features/external-parameter-entities";
66     
67         private BugReporter bugReporter;
68     
69         public SaxParserXxeDetector(BugReporter bugReporter) {
70             this.bugReporter = bugReporter;
71         }
72     
73         @Override
74         public void sawOpcode(int seen) {
75             if(seen != Constants.INVOKEVIRTUAL && seen != INVOKEINTERFACE) return;
76     
77             String fullClassName = getClassConstantOperand();
78             String method = getNameConstantOperand();
79     
80             //The method call is doing XML parsing (see class javadoc)
81             if ((seen == Constants.INVOKEVIRTUAL &&
82                     fullClassName.equals("javax/xml/parsers/SAXParser") &&
83                     method.equals("parse")) ||
84                     (seen == Constants.INVOKEINTERFACE &&
85                             fullClassName.equals("org/xml/sax/XMLReader") &&
86                             method.equals("parse")) ||
87                     (seen == Constants.INVOKEVIRTUAL &&
88                             getClassConstantOperand().equals("javax/xml/parsers/DocumentBuilder") &&
89                             method.equals("parse"))) {
90     
91                 JavaClass javaClass = getThisClass();
92     
93                 //(1rst solution for secure parsing proposed by the CERT) Sandbox in an action with limited privileges
94                 if (InterfaceUtils.isSubtype(javaClass, "java.security.PrivilegedExceptionAction")) {
95                     return; //Assuming the proper right are apply to the sandbox
96                 }
97     
98                 ClassContext classCtx = getClassContext();
99                 ConstantPoolGen cpg = classCtx.getConstantPoolGen();
100     
101                 CFG cfg = null;
102                 try {
103                     cfg = classCtx.getCFG(getMethod());
104                 } catch (CFGBuilderException e) {
105                 }
106     
107                 //The combination of the 4 following is consider safe (expand option is only available with DocumentBuilderFactory)
108                 boolean hasSetXIncludeAware = false;
109                 boolean hasExpandEntityReferences = getClassConstantOperand().equals("javax/xml/parsers/DocumentBuilder") ? false : true;
110                 boolean hasFeatureGeneralEntities = false;
111                 boolean hasFeatureExternalEntities = false;
112     
113                 for (Iterator<Location> i = cfg.locationIterator(); i.hasNext();) {
114                     Location location = i.next();
115                     Instruction inst = location.getHandle().getInstruction();
116                     //ByteCode.printOpCode(inst, cpg);
117     
118     
119                     //(2nd solution for secure parsing proposed by the CERT) Look for entity custom resolver
120                     if (inst instanceof INVOKEINTERFACE) { //XMLReader.setEntityResolver is called
121                         INVOKEINTERFACE invoke = (INVOKEINTERFACE) inst;
122                         if ("setEntityResolver".equals(invoke.getMethodName(cpg))) {
123                             return;
124                         }
125                     }
126     
127                     //DTD disallow
128                     //SAXParser.setFeature and DocumentBuilder.setFeature => INVOKEVIRTUAL
129                     //XMLReader.setFeature => INVOKEINTERFACE
130                     if(inst instanceof INVOKEVIRTUAL || inst instanceof INVOKEINTERFACE) {
131                         InvokeInstruction invoke = (InvokeInstruction) inst;
132                         if ("setFeature".equals(invoke.getMethodName(cpg))) {
133                             LDC loadConst = ByteCode.getPrevInstruction(location.getHandle(), LDC.class);
134     
135                             if (loadConst != null) {
136                                 if (FEATURE_DISALLOW_DTD.equals(loadConst.getValue(cpg))){
137                                     return;
138                                 }
139                                 else if (FEATURE_SECURE_PROCESSING.equals(loadConst.getValue(cpg))){
140                                     return;
141                                 }
142                                 else if (FEATURE_GENERAL_ENTITIES.equals(loadConst.getValue(cpg))){
143                                     hasFeatureGeneralEntities = true;
144                                 }
145                                 else if (FEATURE_EXTERNAL_ENTITIES.equals(loadConst.getValue(cpg))){
146                                     hasFeatureExternalEntities = true;
147                                 }
148                             }
149                         }
150     
151                         else if ("setXIncludeAware".equals(invoke.getMethodName(cpg))) {
152                             ICONST boolConst = ByteCode.getPrevInstruction(location.getHandle(), ICONST.class);
153                             if (boolConst != null && boolConst.getValue().equals(0)) {
154                                 hasSetXIncludeAware = true;
155                             }
156                         }
157     
158                         else if ("setExpandEntityReferences".equals(invoke.getMethodName(cpg))) {
159                             ICONST boolConst = ByteCode.getPrevInstruction(location.getHandle(), ICONST.class);
160                             if (boolConst != null && boolConst.getValue().equals(0)) {
161                                 hasExpandEntityReferences = true;
162                             }
163                         }
164                     }
165     
166     
167                 }
168     
169                 //Manual configuration include all the suggested settings
170                 if(hasFeatureExternalEntities && hasFeatureGeneralEntities && hasSetXIncludeAware && hasExpandEntityReferences) {
171                     return;
172                 }
173     
174     
175                 String simpleClassName = fullClassName.substring(fullClassName.lastIndexOf('/')+1);
176     
177                 //Raise a bug
178                 if(fullClassName.equals("javax/xml/parsers/SAXParser")) {
179                     bugReporter.reportBug(new BugInstance(this, XXE_SAX_PARSER_TYPE, Priorities.NORMAL_PRIORITY) //
180                             .addClass(this).addMethod(this).addSourceLine(this)
181                             .addString(simpleClassName + "." + method + "(...)"));
182                 }
183                 else if(fullClassName.equals("org/xml/sax/XMLReader")) {
184                     bugReporter.reportBug(new BugInstance(this, XXE_XML_READER_TYPE, Priorities.NORMAL_PRIORITY) //
185                             .addClass(this).addMethod(this).addSourceLine(this)
186                             .addString(simpleClassName + "." + method + "(...)"));
187                 }
188                 else if(fullClassName.equals("javax/xml/parsers/DocumentBuilder")) {
189                     bugReporter.reportBug(new BugInstance(this, XXE_DOCUMENT_TYPE, Priorities.NORMAL_PRIORITY) //
190                             .addClass(this).addMethod(this).addSourceLine(this)
191                             .addString(simpleClassName + "." + method + "(...)"));
192                 }
193             }
194         }
195     }
