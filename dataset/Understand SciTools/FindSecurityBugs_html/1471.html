18     package com.h3xstream.findsecbugs.common.matcher;
19     
20     import edu.umd.cs.findbugs.bcel.OpcodeStackDetector;
21     import org.apache.bcel.generic.ConstantPoolGen;
22     import org.apache.bcel.generic.InvokeInstruction;
23     
24     import java.util.ArrayList;
25     import java.util.List;
26     
27     public class InvokeMatcherBuilder {
28     
29         private List<String> classesNames = new ArrayList<String>();
30         private List<String> methodNames = new ArrayList<String>();
31         private List<String> argSignatures = new ArrayList<String>();
32     
33     
34         public InvokeMatcherBuilder atClass(String... classesNames) {
35             for(String clz : classesNames) {
36                 this.classesNames.add(clz);
37             }
38             return this;
39         }
40     
41         public InvokeMatcherBuilder atMethod(String... methodNames) {
42             for(String method : methodNames) {
43                 this.methodNames.add(method);
44             }
45             return this;
46         }
47     
48         public InvokeMatcherBuilder withArgs(String... argSignatures) {
49             for(String method : argSignatures) {
50                 this.argSignatures.add(method);
51             }
52             return this;
53         }
54     
55         public boolean matches(OpcodeStackDetector opcodeStackDetector) {
56             if(classesNames.size() != 0 && !classesNames.contains(opcodeStackDetector.getClassConstantOperand())) {
57                 return false;
58             }
59             if(methodNames.size() != 0 && !methodNames.contains(opcodeStackDetector.getNameConstantOperand())) {
60                 return false;
61             }
62             if(argSignatures.size() != 0 && !argSignatures.contains(opcodeStackDetector.getSigConstantOperand())) {
63                 return false;
64             }
65             return true;
66         }
67     
68     
69         public boolean matches(InvokeInstruction invokeInstruction,ConstantPoolGen cpg) {
70             if (invokeInstruction != null) {
71                 if (classesNames.size() != 0 && !classesNames.contains(invokeInstruction.getClassName(cpg))) {
72                     return false;
73                 }
74                 if (methodNames.size() != 0 && !methodNames.contains(invokeInstruction.getMethodName(cpg))) {
75                     return false;
76                 }
77                 if (argSignatures.size() != 0 && !argSignatures.contains(invokeInstruction.getSignature(cpg))) {
78                     return false;
79                 }
80             }
81             return true;
82         }
83     }
