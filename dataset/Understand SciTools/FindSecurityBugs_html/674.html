18     package com.h3xstream.findsecbugs.taintanalysis;
19     
20     import edu.umd.cs.findbugs.classfile.MethodDescriptor;
21     
22     /**
23      * Global specification of a taint source location
24      * 
25      * @author David Formanek (Y Soft Corporation, a.s.)
26      */
27     public class TaintLocation {
28         
29         private final MethodDescriptor methodDescriptor;
30         private final int position;
31     
32         public TaintLocation(MethodDescriptor methodDescriptor, int position) {
33             if (methodDescriptor == null) {
34                 throw new NullPointerException("method not specified");
35             }
36             if (position < 0) {
37                 throw new IllegalArgumentException("postition is negative");
38             }
39             this.methodDescriptor = methodDescriptor;
40             this.position = position;
41         }
42     
43         public MethodDescriptor getMethodDescriptor() {
44             return methodDescriptor;
45         }
46     
47         public int getPosition() {
48             return position;
49         }
50     
51         @Override
52         public boolean equals(Object obj) {
53             if (obj == null) {
54                 return false;
55             }
56             if (!(obj instanceof TaintLocation)) {
57                 return false;
58             }
59             final TaintLocation other = (TaintLocation) obj;
60             return methodDescriptor.equals(other.methodDescriptor) && position == other.position;
61         }
62     
63         @Override
64         public int hashCode() {
65             return methodDescriptor.hashCode() + 11 * position;
66         }
67     
68         @Override
69         public String toString() {
70             return methodDescriptor.toString() + " " + position;
71         }
72     }
