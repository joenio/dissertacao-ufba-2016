112         private static Collection<Integer> getMutableStackIndices(String signature) {
113             assert signature != null && !signature.isEmpty();
114             ArrayList<Integer> indices = new ArrayList<Integer>();
115             int stackIndex = 0;
116             GenericSignatureParser parser = new GenericSignatureParser(signature);
117             Iterator<String> iterator = parser.parameterSignatureIterator();
118             while (iterator.hasNext()) {
119                 String parameter = iterator.next();
120                 if ((parameter.startsWith("L") || parameter.startsWith("["))
121                         && !IMMUTABLE_OBJECT_TYPES.contains(parameter)) {
122                     indices.add(stackIndex);
123                 }
124                 if (parameter.equals("D") || parameter.equals("J")) {
125                     // double and long types takes two slots
126                     stackIndex += 2;
127                 } else {
128                     stackIndex++;
129                 }
130             }
131             for (int i = 0; i < indices.size(); i++) {
132                 int reverseIndex = stackIndex - indices.get(i) - 1;
133                 assert reverseIndex >= 0;
134                 indices.set(i, reverseIndex);
135             }
136             return indices;
137         }

230         @Override
231         public void handleStoreInstruction(StoreInstruction obj) {
232             try {
233                 int numConsumed = obj.consumeStack(cpg);
234                 if (numConsumed == Constants.UNPREDICTABLE) {
235                     throw new InvalidBytecodeException("Unpredictable stack consumption");
236                 }
237                 int index = obj.getIndex();
238                 while (numConsumed-- > 0) {
239                     Taint value = new Taint(getFrame().popValue());
240                     value.setVariableIndex(index);
241                     getFrame().setValue(index++, value);
242                 }
243             } catch (DataflowAnalysisException ex) {
244                 throw new InvalidBytecodeException(ex.toString(), ex);
245             }
246         }

248         @Override
249         public void handleLoadInstruction(LoadInstruction obj) {
250             int numProduced = obj.produceStack(cpg);
251             if (numProduced == Constants.UNPREDICTABLE) {
252                 throw new InvalidBytecodeException("Unpredictable stack production");
253             }
254             int index = obj.getIndex() + numProduced;
255             while (numProduced-- > 0) {
256                 Taint value = getFrame().getValue(--index);
257                 assert value.hasValidVariableIndex() : "index not set in " + methodDescriptor;
258                 assert index == value.getVariableIndex() : "bad index in " + methodDescriptor;
259                 getFrame().pushValue(new Taint(value));
260             }
261         }

298         @Override
299         public void visitAASTORE(AASTORE obj) {
300             try {
301                 Taint valueTaint = getFrame().popValue();
302                 getFrame().popValue(); // array index
303                 Taint arrayTaint = getFrame().popValue();
304                 Taint merge = Taint.merge(valueTaint, arrayTaint);
305                 setLocalVariableTaint(merge, arrayTaint);
306                 Taint stackTop = null;
307                 if (getFrame().getStackDepth() > 0) {
308                     stackTop = getFrame().getTopValue();
309                 }
310                 // varargs use duplicated values
311                 if (stackTop == arrayTaint) {
312                     getFrame().popValue();
313                     getFrame().pushValue(new Taint(merge));
314                 }
315             } catch (DataflowAnalysisException ex) {
316                 throw new InvalidBytecodeException("Not enough values on the stack", ex);
317             }
318         }

347         /**
348          * Regroup the method invocations (INVOKEINTERFACE, INVOKESPECIAL,
349          * INVOKESTATIC, INVOKEVIRTUAL)
350          *
351          * @param obj one of the invoke instructions
352          */
353         private void visitInvoke(InvokeInstruction obj) {
354             assert obj != null;
355             TaintMethodSummary methodSummary = getMethodSummary(obj);
356             ObjectType realInstanceClass = (methodSummary == null) ?
357                     null : methodSummary.getOutputTaint().getRealInstanceClass();
358             Taint taint = getMethodTaint(methodSummary);
359             assert taint != null;
360             if(FindSecBugsGlobalConfig.getInstance().isDebugTaintState()) {
361                 taint.setDebugInfo(obj.getMethodName(cpg) + "()");
362             }
363             if (taint.isUnknown()) {
364                 taint.addLocation(getTaintLocation(), false);
365             }
366             taintMutableArguments(methodSummary, obj);
367             transferTaintToMutables(methodSummary, taint); // adds variable index to taint too
368             Taint taintCopy = new Taint(taint);
369             // return type is not the instance type always
370             taintCopy.setRealInstanceClass(realInstanceClass);
371             modelInstruction(obj, getNumWordsConsumed(obj), getNumWordsProduced(obj), taintCopy);
372         }

374         private TaintMethodSummary getMethodSummary(InvokeInstruction obj) {
375             String signature = obj.getSignature(cpg);
376             String returnType = getReturnType(signature);
377             if (SAFE_OBJECT_TYPES.contains(returnType)) {
378                 return TaintMethodSummary.SAFE_SUMMARY;
379             }
380             String className = getInstanceClassName(obj);
381             String methodName = obj.getMethodName(cpg);
382             String methodId = "." + methodName + signature;
383             TaintMethodSummary summary = methodSummaries.get(className.concat(methodId));
384             if (summary != null) {
385                 return summary;
386             }
387             summary = getSuperMethodSummary(className, methodId);
388             if (summary != null) {
389                 return summary;
390             }
391             if (Constants.CONSTRUCTOR_NAME.equals(methodName)
392                     && !SAFE_OBJECT_TYPES.contains("L" + className + ";")) {
393                 try {
394                     int stackSize = getFrame().getNumArgumentsIncludingObjectInstance(obj, cpg);
395                     return TaintMethodSummary.getDefaultConstructorSummary(stackSize);
396                 } catch (DataflowAnalysisException ex) {
397                     throw new InvalidBytecodeException(ex.getMessage(), ex);
398                 }
399             }
400             return null;
401         }

403         private String getInstanceClassName(InvokeInstruction invoke) {
404             try {
405                 int instanceIndex = getFrame().getNumArgumentsIncludingObjectInstance(invoke, cpg) - 1;
406                 if (instanceIndex != -1) {
407                     assert instanceIndex < getFrame().getStackDepth();
408                     Taint instanceTaint = getFrame().getStackValue(instanceIndex);
409                     String className = instanceTaint.getRealInstanceClassName();
410                     if (className != null) {
411                         return className;
412                     }
413                 }
414             } catch (DataflowAnalysisException ex) {
415                 assert false : ex.getMessage();
416             }
417             String dottedClassName = invoke.getReferenceType(cpg).toString();
418             return ClassName.toSlashedClassName(dottedClassName);
419         }

421         private TaintMethodSummary getSuperMethodSummary(String className, String methodId) {
422             try {
423                 JavaClass javaClass = Repository.lookupClass(className);
424                 assert javaClass != null;
425                 TaintMethodSummary summary = getSuperMethodSummary(javaClass.getSuperClasses(), methodId);
426                 if (summary != null) {
427                     return summary;
428                 }
429                 return getSuperMethodSummary(javaClass.getAllInterfaces(), methodId);
430             } catch (ClassNotFoundException ex) {
431                 AnalysisContext.reportMissingClass(ex);
432                 return null;
433             }
434         }

436         private TaintMethodSummary getSuperMethodSummary(JavaClass[] javaClasses, String method) {
437             assert javaClasses != null;
438             for (JavaClass classOrInterface : javaClasses) {
439                 String fullMethodName = classOrInterface.getClassName().replace('.', '/').concat(method);
440                 TaintMethodSummary summary = methodSummaries.get(fullMethodName);
441                 if (summary != null) {
442                     return summary;
443                 }
444             }
445             return null;
446         }

449             assert signature != null && signature.contains(")");
450             return signature.substring(signature.indexOf(')') + 1);
451         }

453         private Taint getMethodTaint(TaintMethodSummary methodSummary) {
454             if (methodSummary == null) {
455                 return getDefaultValue();
456             }
457             Taint taint = methodSummary.getOutputTaint();
458             assert taint != null;
459             assert taint != methodSummary.getOutputTaint() : "defensive copy not made";
460             if (taint.isUnknown() && taint.hasParameters()) {
461                 Taint merge = mergeTransferParameters(taint.getParameters());
462                 assert merge != null;
463                 return Taint.merge(Taint.valueOf(taint.getNonParametricState()), merge);
464             }
465             if (taint.isTainted()) {
466                 taint.addLocation(getTaintLocation(), true);
467             }
468             return taint;
469         }

471         private void taintMutableArguments(TaintMethodSummary methodSummary, InvokeInstruction obj) {
472             if (methodSummary != null && methodSummary.isConfigured()) {
473                 return;
474             }
475             Collection<Integer> mutableStackIndices = getMutableStackIndices(obj.getSignature(cpg));
476             for (Integer index : mutableStackIndices) {
477                 assert index >= 0 && index < getFrame().getStackDepth();
478                 try {
479                     Taint stackValue = getFrame().getStackValue(index);
480                     Taint taint = Taint.merge(stackValue, getDefaultValue());
481                     if (stackValue.hasValidVariableIndex()) {
482                         // set back the index removed during merging
483                         taint.setVariableIndex(stackValue.getVariableIndex());
484                     }
485                     taint.setRealInstanceClass(stackValue.getRealInstanceClass());
486                     taint.addLocation(getTaintLocation(), false);
487                     getFrame().setValue(getFrame().getStackLocation(index), taint);
488                     setLocalVariableTaint(taint, taint);
489                 } catch (DataflowAnalysisException ex) {
490                     throw new InvalidBytecodeException("Not enough values on the stack", ex);
491                 }
492             }
493         }

495         private Taint mergeTransferParameters(Collection<Integer> transferParameters) {
496             assert transferParameters != null && !transferParameters.isEmpty();
497             Taint taint = null;
498             for (Integer transferParameter : transferParameters) {
499                 try {
500                     Taint value = getFrame().getStackValue(transferParameter);
501                     taint = Taint.merge(taint, value);
502                 } catch (DataflowAnalysisException ex) {
503                     throw new RuntimeException("Bad transfer parameter specification", ex);
504                 }
505             }
506             assert taint != null;
507             return taint;
508         }

510         private void transferTaintToMutables(TaintMethodSummary methodSummary, Taint taint) {
511             assert taint != null;
512             if (methodSummary == null || !methodSummary.hasMutableStackIndeces()) {
513                 return;
514             }
515             try {
516                 int stackDepth = getFrame().getStackDepth();
517                 for (Integer mutableStackIndex : methodSummary.getMutableStackIndeces()) {
518                     assert mutableStackIndex >= 0;
519                     if (mutableStackIndex >= stackDepth) {
520                         if (!Constants.CONSTRUCTOR_NAME.equals(methodDescriptor.getName()) && !Constants.STATIC_INITIALIZER_NAME.equals(methodDescriptor.getName())) {
521                             assert false : "Out of bounds mutables in " + methodDescriptor;
522                         }
523                         continue; // ignore if assertions disabled or if in constructor
524                     }
525                     Taint stackValue = getFrame().getStackValue(mutableStackIndex);
526                     setLocalVariableTaint(taint, stackValue);
527                     Taint taintCopy = new Taint(taint);
528                     // do not set instance to return values, can be different type
529                     taintCopy.setRealInstanceClass(stackValue.getRealInstanceClass());
530                     getFrame().setValue(getFrame().getStackLocation(mutableStackIndex), taintCopy);
531                 }
532             } catch (DataflowAnalysisException ex) {
533                 assert false : ex.getMessage(); // stack depth is checked
534             }
535         }
536     
537         private void setLocalVariableTaint(Taint valueTaint, Taint indexTaint) {
538             assert valueTaint != null && indexTaint != null;
539             if (!indexTaint.hasValidVariableIndex()) {
540                 return;
541             }
542             int index = indexTaint.getVariableIndex();
543             if (index >= getFrame().getNumLocals()) {
544                 assert false : "Out of bounds local variable index in " + methodDescriptor;
545                 return; // ignore if assertions disabled
546             }
547             valueTaint.setVariableIndex(index);
548             getFrame().setValue(index, valueTaint);
549         }

572         /**
573          * This method must be called from outside at the end of the method analysis
574          */
575         public void finishAnalysis() {
576             assert analyzedMethodSummary != null;
577             Taint outputTaint = analyzedMethodSummary.getOutputTaint();
578             if (outputTaint == null) {
579                 // void methods
580                 return;
581             }
582             String returnType = getReturnType(methodDescriptor.getSignature());
583             if (SAFE_OBJECT_TYPES.contains(returnType) && outputTaint.getState() != Taint.State.NULL) {
584                 // we do not have to store summaries with safe output
585                 return;
586             }
587             String realInstanceClassName = outputTaint.getRealInstanceClassName();
588             if (returnType.equals("L" + realInstanceClassName + ";")) {
589                 // storing it in method summary is useless
590                 outputTaint.setRealInstanceClass(null);
591                 analyzedMethodSummary.setOuputTaint(outputTaint);
592             }
593             String className = methodDescriptor.getSlashedClassName();
594             String methodId = "." + methodDescriptor.getName() + methodDescriptor.getSignature();
595             if (analyzedMethodSummary.isInformative()
596                     || getSuperMethodSummary(className, methodId) != null) {
597                 String fullMethodName = className.concat(methodId);
598                 if (!methodSummaries.containsKey(fullMethodName)) {
599                     // prefer configured summaries to derived
600                     methodSummaries.put(fullMethodName, analyzedMethodSummary);
601                 }
602             }
603         }

605         /**
606          * For debugging purpose.
607          * Print the state of the stack with information about the values in place.
608          */
609         private void printStackState() {
610     
611             try {
612                 System.out.println("============================");
613                 System.out.println("[[ Stack ]]");
614                 int stackDepth = getFrame().getStackDepth();
615                 for (int i = 0; i < stackDepth; i++) {
616                     Taint taintValue = getFrame().getStackValue(i);
617                     System.out.println(String.format("%s. %s {%s}", i, taintValue.getState().toString(), taintValue.getDebugInfo()));
618                 }
619                 if(stackDepth == 0) {
620                     System.out.println("Empty");
621                 }
622                 System.out.println("============================");
623     
624             }
625             catch (DataflowAnalysisException e) {
