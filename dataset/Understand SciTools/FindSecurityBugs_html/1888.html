18     package com.h3xstream.findsecbugs.crypto;
19     
20     import com.h3xstream.findsecbugs.common.ByteCode;
21     import edu.umd.cs.findbugs.Priorities;
22     import org.apache.bcel.classfile.JavaClass;
23     import org.apache.bcel.classfile.Method;
24     import org.apache.bcel.generic.*;
25     
26     import edu.umd.cs.findbugs.BugInstance;
27     import edu.umd.cs.findbugs.BugReporter;
28     import edu.umd.cs.findbugs.Detector;
29     import edu.umd.cs.findbugs.ba.ClassContext;
30     
31     import java.util.Iterator;
32     
33     public class BadHexadecimalConversionDetector implements Detector {
34     
35         /**
36          * For testing purpose
37          */
38         protected static boolean DEBUG = false;
39         private static final String BAD_HEXA_CONVERSION_TYPE = "BAD_HEXA_CONVERSION";
40         private BugReporter bugReporter;
41     
42         public BadHexadecimalConversionDetector(BugReporter bugReporter) {
43             this.bugReporter = bugReporter;
44         }
45     
46         @Override
47         public void visitClassContext(ClassContext classContext) {
48             JavaClass javaClass = classContext.getJavaClass();
49     
50             Method[] methodList = javaClass.getMethods();
51     
52             for (Method m : methodList) {
53                 MethodGen methodGen = classContext.getMethodGen(m);
54     
55                 if (DEBUG) {
56                     System.out.println(">>> Method: " + m.getName());
57                 }
58     
59                 //To suspect that an invalid String representation is being build,
60                 //we identify the construction of a MessageDigest and
61                 //the use of a function that trim leading 0.
62                 boolean invokeMessageDigest = false;
63                 boolean invokeToHexString = false;
64     
65                 ConstantPoolGen cpg = classContext.getConstantPoolGen();
66                 if (methodGen == null || methodGen.getInstructionList() == null) {
67                     continue; //No instruction .. nothing to do
68                 }
69                 for (Iterator itIns = methodGen.getInstructionList().iterator();itIns.hasNext();) {
70                     Instruction inst = ((InstructionHandle) itIns.next()).getInstruction();
71                     if (DEBUG) {
72                         ByteCode.printOpCode(inst, cpg);
73                     }
74     
75                     if (inst instanceof INVOKEVIRTUAL) { //MessageDigest.digest is called
76                         INVOKEVIRTUAL invoke = (INVOKEVIRTUAL) inst;
77                         if ("java.security.MessageDigest".equals(invoke.getClassName(cpg)) && "digest".equals(invoke.getMethodName(cpg))) {
78                             invokeMessageDigest = true;
79                         }
80                     } else if (inst instanceof INVOKESTATIC && invokeMessageDigest) { //The conversion must occurs after the digest was created
81                         INVOKESTATIC invoke = (INVOKESTATIC) inst;
82                         if ("java.lang.Integer".equals(invoke.getClassName(cpg)) && "toHexString".equals(invoke.getMethodName(cpg))) {
83                             invokeToHexString = true;
84                         }
85                     }
86                 }
87     
88                 if (invokeMessageDigest && invokeToHexString) {
89                     bugReporter.reportBug(new BugInstance(this, BAD_HEXA_CONVERSION_TYPE, Priorities.NORMAL_PRIORITY) //
90                             .addClassAndMethod(javaClass, m));
91                 }
92             }
93         }
94     
95         @Override
96         public void report() {
97     
98         }
99     }
