18     package com.h3xstream.findsecbugs.injection;
19     
20     import edu.umd.cs.findbugs.BugReporter;
21     import edu.umd.cs.findbugs.util.ClassName;
22     import java.util.HashMap;
23     import java.util.Map;
24     import org.apache.bcel.Constants;
25     import org.apache.bcel.generic.ConstantPoolGen;
26     import org.apache.bcel.generic.InstructionHandle;
27     import org.apache.bcel.generic.InvokeInstruction;
28     
29     /**
30      * Detector designed for extension to detect basic injections with a list of
31      * full method names with specified injectable arguments as taint sinks
32      * 
33      * @author David Formanek (Y Soft Corporation, a.s.)
34      */
35     public abstract class BasicInjectionDetector extends AbstractInjectionDetector {
36     
37         private final Map<String, InjectionPoint> injectionMap = new HashMap<String, InjectionPoint>();
38         
39         public BasicInjectionDetector(BugReporter bugReporter) {
40             super(bugReporter);
41         }
42     
43         @Override
44         protected InjectionPoint getInjectionPoint(InvokeInstruction invoke, ConstantPoolGen cpg,
45                 InstructionHandle handle) {
46             assert invoke != null && cpg != null;
47             InjectionPoint injectionPoint = injectionMap.get(getFullMethodName(invoke, cpg));
48             if (injectionPoint == null) {
49                 return InjectionPoint.NONE;
50             }
51             return injectionPoint;
52         }
53         
54         protected void addSink(String fullMethodName, int[] injectableParameters, String bugType) {
55             assert fullMethodName != null && !fullMethodName.isEmpty();
56             assert injectableParameters != null && injectableParameters.length != 0;
57             assert !injectionMap.containsKey(fullMethodName);
58             InjectionPoint injectionPoint = new InjectionPoint(injectableParameters, bugType);
59             String classAndMethodName = fullMethodName.substring(0, fullMethodName.indexOf('('));
60             int slashIndex = classAndMethodName.lastIndexOf('/');
61             String shortName = classAndMethodName.substring(slashIndex + 1);
62             if (shortName.endsWith(Constants.CONSTRUCTOR_NAME)) {
63                 shortName = shortName.substring(0, shortName.indexOf('.'));
64             }
65             injectionPoint.setInjectableMethod(shortName.concat("(...)"));
66             injectionMap.put(fullMethodName, injectionPoint);
67         }
68         
69         private String getFullMethodName(InvokeInstruction invoke, ConstantPoolGen cpg) {
70             return ClassName.toSlashedClassName(invoke.getReferenceType(cpg).toString())
71                     + "." + invoke.getMethodName(cpg) + invoke.getSignature(cpg);
72         }
73     }
