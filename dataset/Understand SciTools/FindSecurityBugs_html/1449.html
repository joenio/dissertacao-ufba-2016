18     package com.h3xstream.findsecbugs.android;
19     
20     import com.h3xstream.findsecbugs.common.StackUtils;
21     import edu.umd.cs.findbugs.BugInstance;
22     import edu.umd.cs.findbugs.BugReporter;
23     import edu.umd.cs.findbugs.OpcodeStack;
24     import edu.umd.cs.findbugs.Priorities;
25     import edu.umd.cs.findbugs.bcel.OpcodeStackDetector;
26     import org.apache.bcel.Constants;
27     
28     public class WorldWritableDetector extends OpcodeStackDetector {
29     
30         private static final String ANDROID_WORLD_WRITABLE_TYPE = "ANDROID_WORLD_WRITABLE";
31         private BugReporter bugReporter;
32     
33         public WorldWritableDetector(BugReporter bugReporter) {
34             this.bugReporter = bugReporter;
35         }
36     
37         @Override
38         public void sawOpcode(int seen) {
39             //printOpCode(seen);
40     
41             if (seen == Constants.INVOKEVIRTUAL && getNameConstantOperand().equals("openFileOutput")) {
42                 OpcodeStack.Item item = stack.getStackItem(0); //First item on the stack is the last
43                 if(StackUtils.isConstantInteger(item)) {
44                     Integer value = (Integer) item.getConstant();
45                     if(value == null || value != 0) {
46                         bugReporter.reportBug(new BugInstance(this, ANDROID_WORLD_WRITABLE_TYPE, Priorities.NORMAL_PRIORITY) //
47                                 .addClass(this).addMethod(this).addSourceLine(this));
48                     }
49                 }
50             }
51         }
52     }
