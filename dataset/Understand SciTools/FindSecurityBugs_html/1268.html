18     package com.h3xstream.findsecbugs.injection;
19     
20     import edu.umd.cs.findbugs.BugReporter;
21     import org.apache.bcel.generic.ConstantPoolGen;
22     import org.apache.bcel.generic.InstructionHandle;
23     import org.apache.bcel.generic.InvokeInstruction;
24     
25     /**
26      * Detector designed for extension to detect injection vulnerabilities using
27      * the original mechanism with {@link InjectionSource} class
28      *
29      * @author David Formanek (Y Soft Corporation, a.s.)
30      */
31     public abstract class LegacyInjectionDetector extends AbstractInjectionDetector {
32     
33         protected LegacyInjectionDetector(BugReporter bugReporter) {
34             super(bugReporter);
35         }
36     
37         @Override
38         protected InjectionPoint getInjectionPoint(InvokeInstruction invoke, ConstantPoolGen cpg,
39                 InstructionHandle handle) {
40             InjectionPoint injectionPoint = null;
41             for (InjectionSource source : getInjectionSource()) {
42                 injectionPoint = source.getInjectableParameters(invoke, cpg, handle);
43                 if (injectionPoint != InjectionPoint.NONE) {
44                     break;
45                 }
46             }
47             if (injectionPoint == null) {
48                 injectionPoint = InjectionPoint.NONE;
49             }
50             return injectionPoint;
51         }
52     
53         public abstract InjectionSource[] getInjectionSource();
54     }
