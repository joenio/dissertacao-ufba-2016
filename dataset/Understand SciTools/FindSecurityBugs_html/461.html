18     package com.h3xstream.findsecbugs;
19     
20     import com.h3xstream.findsecbugs.common.InterfaceUtils;
21     import edu.umd.cs.findbugs.BugInstance;
22     import edu.umd.cs.findbugs.BugReporter;
23     import edu.umd.cs.findbugs.bcel.OpcodeStackDetector;
24     import org.apache.bcel.Repository;
25     import org.apache.bcel.classfile.JavaClass;
26     
27     /**
28      * Detect Java object deserialization
29      *
30      * @author Alexander Minozhenko
31      */
32     public class ObjectDeserializationDetector extends OpcodeStackDetector {
33         private final BugReporter bugReporter;
34     
35         public ObjectDeserializationDetector(BugReporter bugReporter) {
36             this.bugReporter = bugReporter;
37         }
38     
39         @Override
40         public void sawOpcode(int code) {
41             //printOpCode(code);
42             if ((code == INVOKEVIRTUAL)) {
43                 if("java/io/ObjectInputStream".equals(getClassConstantOperand()) || getClassConstantOperand().contains("InputStream")) {
44     
45                     String methodName = getNameConstantOperand();
46                     if (methodName.equals("readObject")) {
47                         bugReporter.reportBug(new BugInstance(this, "OBJECT_DESERIALIZATION", HIGH_PRIORITY).addClassAndMethod(this).addSourceLine(this));
48                     }
49                 }
50             }
51         }
52     }
