47         public Collection<Integer> getMutableStackIndeces() {
48             if (!hasMutableStackIndeces()) {
49                 throw new IllegalStateException("stack indeces not set");
50             }
51             return Collections.unmodifiableCollection(mutableStackIndices);
52         }
53     
54         public boolean hasMutableStackIndeces() {
55             assert mutableStackIndices != null;
56             return !mutableStackIndices.isEmpty();
57         }
58     
59         public void addMutableStackIndex(int mutableStackIndex) {
60             if (mutableStackIndex < 0) {
61                 throw new IllegalArgumentException("negative index");
62             }
63             mutableStackIndices.add(mutableStackIndex);
64         }
65     
66         public Taint getOutputTaint() {
67             if (outputTaint == null) {
68                 return null;
69             }
70             return new Taint(outputTaint);
71         }
72     
73         public void setOuputTaint(Taint taint) {
74             if (taint == null) {
75                 this.outputTaint = null;
76                 return;
77             }
78             Taint taintCopy = new Taint(taint);
79             taintCopy.invalidateVariableIndex();
80             this.outputTaint = taintCopy;
81         }

83         public static TaintMethodSummary getDefaultConstructorSummary(int stackSize) {
84             if (stackSize < 1) {
85                 throw new IllegalArgumentException("stack size less than 1");
86             }
87             TaintMethodSummary summary = new TaintMethodSummary(false);
88             summary.outputTaint = new Taint(Taint.State.UNKNOWN);
89             summary.mutableStackIndices.add(stackSize - 1);
90             summary.mutableStackIndices.add(stackSize);
91             return summary;
92         }

101         public boolean isInformative() {
102             if (this == SAFE_SUMMARY) {
103                 // these are loaded automatically, do not need to store them
104                 return false;
105             }
106             if (outputTaint == null) {
107                 return false;
108             }
109             if (!outputTaint.isUnknown()) {
110                 return true;
111             }
112             if (outputTaint.hasParameters()) {
113                 return true;
114             }
115             if (outputTaint.getRealInstanceClass() != null) {
116                 return true;
117             }
118             return false;
119         }

121         public boolean isConfigured() {
122             return isConfigured;
123         }

125         @Override
126         public String toString() {
127             if (outputTaint == null) {
128                 return "";
129             }
130             StringBuilder sb = new StringBuilder();
131             if (outputTaint.isUnknown() && outputTaint.hasParameters()) {
132                 appendJoined(sb, outputTaint.getParameters());
133                 Taint.State nonParametricState = outputTaint.getNonParametricState();
134                 assert nonParametricState != null;
135                 if (nonParametricState != Taint.State.INVALID) {
136                     sb.append(",").append(nonParametricState.name());
137                 }
138             } else {
139                 sb.append(outputTaint.getState().name());
140             }
141             if (hasMutableStackIndeces()) {
142                 sb.append("#");
143                 appendJoined(sb, mutableStackIndices);
144             }
145             String realInstanceClassName = outputTaint.getRealInstanceClassName();
146             if (realInstanceClassName != null) {
147                 sb.append(" (").append(realInstanceClassName).append(")");
148             }
149             return sb.toString();
150         }
151     
152         private static void appendJoined(StringBuilder sb, Collection<Integer> objects) {
153             assert sb != null && objects != null;
154             int count = objects.size();
155             Integer[] array = objects.toArray(new Integer[count]);
156             sb.append(array[0]);
157             for (int i = 1; i < count; i++) {
158                 sb.append(",").append(array[i]);
159             }
160         }

162         /**
163          * Loads method summary from String
164          * 
165          * @param str (state or parameter indices to merge separated by comma)#mutable position
166          * @return initialized object with taint method summary
167          * @throws java.io.IOException for bad format of paramter
168          */
169         public static TaintMethodSummary load(String str) throws IOException {
170             if (str == null) {
171                 throw new NullPointerException("string is null");
172             }
173             str = str.trim();
174             String[] tuple = str.split("#");
175             TaintMethodSummary summary = new TaintMethodSummary(true);
176             if (tuple.length == 2) {
177                 str = tuple[0];
178                 try {
179                     String[] indices = tuple[1].split(",");
180                     for (String index : indices) {
181                         summary.addMutableStackIndex(Integer.parseInt(index.trim()));
182                     }
183                 } catch (NumberFormatException ex) {
184                     throw new IOException("Cannot parse mutable stack offsets", ex);
185                 }
186             } else if (tuple.length != 1) {
187                 throw new IOException("Bad format: only one '#' expected");
188             }
189             if (str.isEmpty()) {
190                 throw new IOException("No taint information set");
191             } else if (isTaintStateValue(str)) {
192                 summary.setOuputTaint(Taint.valueOf(str));
193             } else {
194                 tuple = str.split(",");
195                 int count = tuple.length;
196                 Taint taint = new Taint(Taint.State.UNKNOWN);
197                 for (int i = 0; i < count; i++) {
198                     String indexOrState = tuple[i].trim();
199                     if (isTaintStateValue(indexOrState)) {
200                         taint.setNonParametricState(Taint.State.valueOf(indexOrState));
201                     } else {
202                         try {
203                             taint.addParameter(Integer.parseInt(indexOrState));
204                         } catch (NumberFormatException ex) {
205                             throw new IOException("Cannot parse parameter offset " + i, ex);
206                         }
207                     }
208                 }
209                 summary.setOuputTaint(taint);
210             }
211             return summary;
212         }
213     
214         private static boolean isTaintStateValue(String value) {
215             assert value != null && !value.isEmpty();
216             Taint.State[] states = Taint.State.values();
217             for (Taint.State state : states) {
218                 if (state.name().equals(value)) {
219                     return true;
220                 }
221             }
222             return false;
223         }
