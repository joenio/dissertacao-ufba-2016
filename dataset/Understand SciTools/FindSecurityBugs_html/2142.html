18     package com.h3xstream.findsecbugs.injection.redirect;
19     
20     import com.h3xstream.findsecbugs.common.ByteCode;
21     import com.h3xstream.findsecbugs.injection.InjectionPoint;
22     import com.h3xstream.findsecbugs.injection.InjectionSource;
23     import org.apache.bcel.generic.ConstantPoolGen;
24     import org.apache.bcel.generic.INVOKEINTERFACE;
25     import org.apache.bcel.generic.InstructionHandle;
26     import org.apache.bcel.generic.InvokeInstruction;
27     import org.apache.bcel.generic.LDC;
28     
29     public class RedirectionSource implements InjectionSource {
30     
31         private static final String UNVALIDATED_REDIRECT_TYPE = "UNVALIDATED_REDIRECT";
32     
33         @Override
34         public InjectionPoint getInjectableParameters(InvokeInstruction ins, ConstantPoolGen cpg, InstructionHandle insHandle) {
35             //ByteCode.printOpCode(ins, cpg);
36     
37             if (ins instanceof INVOKEINTERFACE) {
38                 String methodName = ins.getMethodName(cpg);
39                 String className = ins.getClassName(cpg);
40     
41                 if (className.equals("javax.servlet.http.HttpServletResponse")) {
42                     if (methodName.equals("sendRedirect")) {
43                         InjectionPoint ip = new InjectionPoint(new int[]{0}, UNVALIDATED_REDIRECT_TYPE);
44                         ip.setInjectableMethod("HttpServletResponse.sendRedirect(...)");
45                         return ip;
46                         
47                     } else if (methodName.equals("addHeader")) {
48     
49                         LDC ldc = ByteCode.getPrevInstruction(insHandle, LDC.class);
50                         if (ldc != null) {
51                             Object value = ldc.getValue(cpg);
52                             if ("Location".equals(value)) {
53                                 InjectionPoint ip = new InjectionPoint(new int[]{0},UNVALIDATED_REDIRECT_TYPE);
54                                 ip.setInjectableMethod("HttpServletResponse.addHeader(\"Location\", ...)");
55                                 return ip;
56                             }
57                         }
58     
59                     }
60                 }
61             }
62             return InjectionPoint.NONE;
63         }
64     }
