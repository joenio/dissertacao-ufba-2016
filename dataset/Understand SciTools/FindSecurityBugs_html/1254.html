30     /**
31      * Various flavor of XSSRequestWrapper exist to do some debatable prevention. It can be considered as a Web Application
32      * Firewall.
33      * <br>
34      * Some implementations to detect:
35      * http://java.dzone.com/articles/stronger-anti-cross-site
36      * http://www.javacodegeeks.com/2012/07/anti-cross-site-scripting-xss-filter.html
37      * http://ricardozuasti.com/2012/stronger-anti-cross-site-scripting-xss-filter-for-java-web-apps/
38      */
39     public class XSSRequestWrapperDetector implements Detector {
40     
41         private static final String XSS_REQUEST_WRAPPER_TYPE = "XSS_REQUEST_WRAPPER";
42     
43         private BugReporter bugReporter;
44     
45         public XSSRequestWrapperDetector(BugReporter bugReporter) {
46             this.bugReporter = bugReporter;
47         }
48     
49         @Override
50         public void visitClassContext(ClassContext classContext) {
51             JavaClass javaClass = classContext.getJavaClass();
52     
53             //The class extends HttpServletRequestWrapper
54             boolean isRequestWrapper = InterfaceUtils.isSubtype(javaClass, "javax.servlet.http.HttpServletRequestWrapper");
55     
56             //Not the target of this detector
57             if (!isRequestWrapper) return;
58     
59             Method[] methodList = javaClass.getMethods();
60     
61             for (Method m : methodList) {
62                 if (m.getName().equals("stripXSS")) {
63                     bugReporter.reportBug(new BugInstance(this, XSS_REQUEST_WRAPPER_TYPE, Priorities.NORMAL_PRIORITY) //
64                             .addClassAndMethod(javaClass, m));
65                     return;
66                 }
67             }
68     
69         }
70     
71         @Override
72         public void report() {
73     
74         }
75     }
