18     package com.h3xstream.findsecbugs.taintanalysis;
19     
20     import java.io.BufferedReader;
21     import java.io.IOException;
22     import java.io.InputStream;
23     import java.io.InputStreamReader;
24     import java.io.PrintStream;
25     import java.util.HashMap;
26     import java.util.TreeSet;
27     
28     /**
29      * Map of taint summaries for all known methods
30      *
31      * This class extends HashMap:
32      * <ul>
33      *  <li>The key is the method signature (ie :
34      org/hibernate/Session.createQuery(Ljava/lang/String;)Lorg/hibernate/Query;)</li>
35      *  <li>The value is the behavior of the method
36      *  ("0" for param index 0 is tainted,
37      *  "UNKNOWN" if the method does not become tainted base on the value,
38      *  "TAINTED" if the result must be consider unsafe)</li>
39      * </ul>
40      *
41      * @author David Formanek (Y Soft Corporation, a.s.)
42      */
43     public class TaintMethodSummaryMap extends HashMap<String, TaintMethodSummary> {
44         private static final long serialVersionUID = 1L;
45         
46         public void load(InputStream input) throws IOException {
47             BufferedReader reader = new BufferedReader(new InputStreamReader(input, "UTF-8"));
48             for (;;) {
49                     String line = reader.readLine();
50                     if (line == null) {
51                         break;
52                     }
53                     line = line.trim();
54                     if (line.isEmpty()) {
55                         continue;
56                     }
57                     putFromLine(line);
58                 }
59         }
60         
61         public void dump(PrintStream output) {
62             TreeSet<String> keys = new TreeSet<String>(keySet());
63             for (String key : keys) {
64                 output.println(key + ":" + get(key));
65             }
66         }
67         
68         private void putFromLine(String line) throws IOException {
69             if (line.startsWith("-")) {
70                 // for comments or removing summary temporarily
71                 return;
72             }
73             String[] tuple = line.split("\\:");
74             if (tuple.length != 2) {
75                 throw new IOException("Line format is not 'method name:summary info'");
76             }
77             put(tuple[0].trim(), TaintMethodSummary.load(tuple[1]));
78         }
79     }
