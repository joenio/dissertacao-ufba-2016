44     public class JstlOutDetector implements Detector {
45         private static final boolean DEBUG = false;
46         private static final String JSP_JSTL_OUT = "JSP_JSTL_OUT";
47     
48         private static final InvokeMatcherBuilder OUT_TAG_ESCAPE_XML = invokeInstruction().atClass("org.apache.taglibs.standard.tag.rt.core.OutTag",
49                 "org.apache.taglibs.standard.tag.el.core.OutTag",
50                 "com.caucho.jstl.el.CoreOutTag",
51                 "com.caucho.jstl.rt.OutTag",
52                 "org.apache.taglibs.standard.tag.compat.core.OutTag",
53                 "org.appfuse.webapp.taglib.OutTag")
54                 .atMethod("setEscapeXml").withArgs("(Z)V","(Ljava/lang/String;)V");
55     
56         private final BugReporter bugReporter;
57     
58     
59     
60         public JstlOutDetector(BugReporter bugReporter) {
61             this.bugReporter = bugReporter;
62         }
63     
64         @Override
65         public void visitClassContext(ClassContext classContext) {
66             JavaClass javaClass = classContext.getJavaClass();
67     
68             try {
69                 if(!Hierarchy.isSubtype(javaClass.getClassName(), "javax.servlet.http.HttpServlet")) {
70                     return;
71                 }
72             } catch (ClassNotFoundException e) {
73             }
74             for (Method m : javaClass.getMethods()) {
75                 try {
76                     analyzeMethod(m, classContext);
77                 } catch (CFGBuilderException e) {
78                 } catch (DataflowAnalysisException e) {
79                 }
80             }
81         }
82     
83         private void analyzeMethod(Method m, ClassContext classContext) throws CFGBuilderException, DataflowAnalysisException {
84     
85             //Conditions that needs to fill to identify the vulnerability
86             boolean escapeXmlSetToFalse = false;
87             boolean escapeXmlValueUnknown = false;
88             Location locationWeakness = null;
89     
90             ConstantPoolGen cpg = classContext.getConstantPoolGen();
91             CFG cfg = classContext.getCFG(m);
92     
93             LinkedList<Instruction> instructionVisited = new LinkedList<Instruction>();
94     
95             for (Iterator<Location> i = cfg.locationIterator(); i.hasNext(); ) {
96                 Location location = i.next();
97     
98                 Instruction inst = location.getHandle().getInstruction();
99                 //ByteCode.printOpCode(inst, cpg);
100     
101                 instructionVisited.add(inst);
102     
103     //     JspSpringEvalDetector: [0047]  aload   4
104     //     JspSpringEvalDetector: [0049]  iconst_1
105     //     JspSpringEvalDetector: [0050]  invokevirtual   org/apache/taglibs/standard/tag/rt/core/OutTag.setEscapeXml (Z)V
106     
107                 if (inst instanceof InvokeInstruction) {
108                     InvokeInstruction invoke = (InvokeInstruction) inst;
109     
110                     if(OUT_TAG_ESCAPE_XML.matches(invoke,cpg)) {
111                         Integer booleanValue = ByteCode.getConstantInt(location.getHandle().getPrev());
112                         if (booleanValue != null && booleanValue == 0) {
113                             escapeXmlSetToFalse = true;
114                             locationWeakness = location;
115                         } else {
116                             //Some JSP compiler convert boolean value at runtime (WebLogic)
117                             String stringValue = JspUtils.getContanstBooleanAsString(instructionVisited, cpg);
118                             if (stringValue != null && stringValue.equals("false")) {
119                                 escapeXmlSetToFalse = true;
120                                 locationWeakness = location;
121                             }
122     
123                             if(booleanValue == null && stringValue == null) {
124                                 escapeXmlValueUnknown = true;
125                                 locationWeakness = location;
126                             }
127                         }
128                     }
129                 }
130             }
131     
132             //Both condition have been found in the same method
133             if (escapeXmlSetToFalse) {
134                 JavaClass clz = classContext.getJavaClass();
135                 bugReporter.reportBug(new BugInstance(this, JSP_JSTL_OUT, Priorities.NORMAL_PRIORITY) //
136                         .addClass(clz)
137                         .addMethod(clz, m)
138                         .addSourceLine(classContext, m, locationWeakness));
139             } else if (escapeXmlValueUnknown) {
140                 JavaClass clz = classContext.getJavaClass();
141                 bugReporter.reportBug(new BugInstance(this, JSP_JSTL_OUT, Priorities.LOW_PRIORITY) //
142                         .addClass(clz)
143                         .addMethod(clz, m)
144                         .addSourceLine(classContext, m, locationWeakness));
145             }
146         }

151         }
