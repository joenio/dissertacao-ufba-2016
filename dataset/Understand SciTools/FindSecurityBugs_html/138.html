40     /**
41      * Detector designed for extension to allow usage of taint analysis
42      *
43      * @author David Formanek (Y Soft Corporation, a.s.)
44      */
45     public abstract class AbstractTaintDetector implements Detector {
46         
47         protected final BugReporter bugReporter;
48         
49         protected AbstractTaintDetector(BugReporter bugReporter) {
50             this.bugReporter = bugReporter;
51         }
52     
53         /**
54          * Allow any concrete implementation of taint detector to skip the analysis of certain files.
55          * The purpose can be for optimisation or to trigger bug in specific context.
56          *
57          * The default implementation returns true to all classes visited.
58          *
59          * @param classContext Information about the class that is about to be analyzed
60          * @return If the given class should be analyze.
61          */
62         public boolean shouldAnalyzeClass(ClassContext classContext) {
63             return true;
64         }
65     
66         @Override
67         public void visitClassContext(ClassContext classContext) {
68             if(!shouldAnalyzeClass(classContext)) return;
69     
70             for (Method method : classContext.getMethodsInCallOrder()) {
71                 if (classContext.getMethodGen(method) == null) {
72                     continue;
73                 }
74                 try {
75                     analyzeMethod(classContext, method);
76                 } catch (CheckedAnalysisException e) {
77                     logException(classContext, method, e);
78                 } catch (RuntimeException e) {
79                     logException(classContext, method, e);
80                 }
81             }
82         }
83     
84         @Override
85         public void report() {
86         }
87         
88         protected void analyzeMethod(ClassContext classContext, Method method)
89                 throws DataflowAnalysisException, CheckedAnalysisException {
90             TaintDataflow dataflow = getTaintDataFlow(classContext, method);
91             ConstantPoolGen cpg = classContext.getConstantPoolGen();
92             String currentMethod = getFullMethodName(classContext.getMethodGen(method));
93             for (Iterator<Location> i = getLocationIterator(classContext, method); i.hasNext();) {
94                 Location location = i.next();
95                 InstructionHandle handle = location.getHandle();
96                 Instruction instruction = handle.getInstruction();
97                 if (!(instruction instanceof InvokeInstruction)) {
98                     continue;
99                 }
100                 InvokeInstruction invoke = (InvokeInstruction) instruction;
101                 TaintFrame fact = dataflow.getFactAtLocation(location);
102                 assert fact != null;
103                 if (!fact.isValid()) {
104                     continue;
105                 }
106                 analyzeLocation(classContext, method, handle, cpg, invoke, fact, currentMethod);
107             }
108         }
109         
110         private static Iterator<Location> getLocationIterator(ClassContext classContext, Method method)
111                 throws CheckedAnalysisException {
112             try {
113                 return classContext.getCFG(method).locationIterator();
114             } catch (CFGBuilderException ex) {
115                 throw new CheckedAnalysisException("cannot get control flow graph", ex);
116             }
117         }
118         
119         private static TaintDataflow getTaintDataFlow(ClassContext classContext, Method method)
120                 throws CheckedAnalysisException {
121             MethodDescriptor descriptor = BCELUtil.getMethodDescriptor(classContext.getJavaClass(), method);
122             return Global.getAnalysisCache().getMethodAnalysis(TaintDataflow.class, descriptor);
123         }
124     
125         private static String getFullMethodName(MethodGen methodGen) {
126             String methodNameWithSignature = methodGen.getName() + methodGen.getSignature();
127             String slashedClassName = methodGen.getClassName().replace('.', '/');
128             return slashedClassName + "." + methodNameWithSignature;
129         }
130         
131         private void logException(ClassContext classContext, Method method, Exception ex) {
132             bugReporter.logError("Exception while analyzing "
133                     + classContext.getFullyQualifiedMethodName(method), ex);
134         }
135         
136         abstract protected void analyzeLocation(ClassContext classContext, Method method, InstructionHandle handle,
137                 ConstantPoolGen cpg, InvokeInstruction invoke, TaintFrame fact, String currentMethod)
138                 throws DataflowAnalysisException;
139     }
