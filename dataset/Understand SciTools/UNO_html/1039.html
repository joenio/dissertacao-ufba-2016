22     Heap *
23     CreateHeap(unsigned int chunk_size, unsigned int chnk_hunk_ratio )
24     {	Heap    *heap;
25     	unsigned int     unit;
26     
27     	heap = (Heap *) emalloc(sizeof(Heap));
28     
29     	unit  = max(chunk_size, MIN_SZE);
30     	if (unit % ALGN_SZE != 0)
31     		unit += ALGN_SZE - (unit % ALGN_SZE);
32     	heap->chnk_sze = unit;
33     
34     	if (chnk_hunk_ratio)
35     		heap->ch_ratio = chnk_hunk_ratio; 
36     	else
37     		heap->ch_ratio = DEFLT_RATIO;
38     
39     	heap->num_alloc = 0; 
40     	heap->num_free  = 0; 
41     	heap->num_hunks = 0; 
42     
43     	heap->next_chunk = 0;
44     	heap->hunk_array_sze = MIN_HUNK_ARRY;
45     	heap->free_list = (void *) NULL;
46     
47     	heap->hunks = (void **) emalloc(MIN_HUNK_ARRY * PNTR_SZE);
48     
49     	return(heap);
50     }

64     #ifdef DEBUG
65     int
66     check_chunks(void *y)
67     {	FC *x;
68     
69     	for (x = free_chunks; x; x = x->fc.nxt)
70     	{	if ((int) x < 1000)
71     		{	printf("memory corrupted \n");
72     			abort();
73     		}
74     		if (x == y)
75     		{	printf("double release %p\n", x);
76     			x = (FC *) 0; x = x->fc.nxt;	/* cause crash */
77     			return 0;
78     		}
79     	}
80     	return 1;
81     }
82     #endif

84     void
85     rel_chunk(void *x)
86     {	FC *y;
87     
88     #ifdef DEBUG
89     	if (!check_chunks(x)) return;
90     #endif
91     	memset(x, 0, MX_NODE_SZE);
92     	y = (FC *) x;
93     	y->fc.nxt = free_chunks;
94     	free_chunks = y;
95     
96     #ifdef DEBUG
97     	printf("++ %p	-- %d\n", x, ++gh_sz);
98     #endif
99     }

121     void *
122     HeapAlloc_Gen(Heap *heap)
123     {	int     asze, chnk;
124     	void   *avail, *pnt;
125     
126     	if (!heap)
127     		return ((void *) NULL);
128     
129     	if (free_chunks)
130     	{
131     #ifdef DEBUG
132     		printf("-- %p	%d\n", free_chunks, --gh_sz);
133     #endif
134     		avail = (void *) free_chunks;
135     		free_chunks = free_chunks->fc.nxt;
136     		memset(avail, 0, sizeof(void *));
137     		return avail;
138     	}
139     
140     	if (heap->num_hunks)
141     	{	chnk = (heap->next_chunk)++;
142     
143     		if (chnk == (int) heap->ch_ratio)
144     		{
145     			if (heap->num_hunks == heap->hunk_array_sze)
146     			{	asze = heap->hunk_array_sze * 2;	/* double array size */
147     				pnt = (void *) emalloc(asze * PNTR_SZE);
148     	
149     				memcpy(pnt, (void *) heap->hunks, (heap->hunk_array_sze * PNTR_SZE));
150     	
151     				efree(heap->hunks);
152     				heap->hunks = pnt;
153     				heap->hunk_array_sze = asze;
154     			}
155     	
156     			heap->hunks[heap->num_hunks] =
157     				(void *) emalloc(heap->chnk_sze * heap->ch_ratio);
158     	
159     #ifdef DEBUG
160     	printf("2 chunk allocate at %p -- %d * %d bytes\n",
161     		heap->hunks[heap->num_hunks],
162     		heap->chnk_sze, heap->ch_ratio);
163     #endif
164     			(heap->num_hunks)++;
165     			heap->next_chunk = 1;
166     			(heap->num_alloc)++;
167     			return (heap->hunks[(heap->num_hunks)-1]);
168     		} else
169     		{	(heap->num_alloc)++;
170     			return((void *) ((char*) (heap->hunks[(heap->num_hunks)-1])
171     				+ (heap->chnk_sze * chnk))); 
172     		}
173     	} else
174     	{	heap->hunks[0] = (void *) emalloc(heap->chnk_sze * heap->ch_ratio);
175     #ifdef DEBUG
176     	printf("1 chunk allocate at %p -- %d * %d bytes\n",
177     		heap->hunks[0], heap->chnk_sze, heap->ch_ratio);
178     #endif
179     		heap->num_hunks = 1;
180     		heap->next_chunk = 1;
181     		(heap->num_alloc)++;
182     		return (heap->hunks[0]);
183     	}
184     }
