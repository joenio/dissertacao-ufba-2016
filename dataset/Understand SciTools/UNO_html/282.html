22     struct Trans {
23     	treenode *cond;	/* if conditional, expr to be matched for this branch */
24     	SymRef	*knz;	/* vars known to be nonzero after this branch */
25     	State *branch;	/* successor chain */
26     	Trans *nxt;	/* next successor chain */
27     };

31     	int	status;
32     	Stack	*nxt;
33     };

40     struct SymRef {
41     	symentry_t	*sm;
42     	treenode	*n;
43     	int		status;	/* typically def/use info */
44     	int		s_val;	/* in uno_generic, symbol can be given state value */
45     	SymRef		*nxt;
46     };
47     
48     struct SymExt {
49     	symentry_t	*sm;
50     	treenode	*n;
51     	int		status;
52     	SymExt		*nxt;
53     };
54     
55     struct DfStack {
56     	SymRef		*symrefs;	/* tracks use of locals */
57     	SymRef		*globrefs;	/* tracks use of globals */
58     	PathCond	*state;
59     	DfStack		*nxt;
60     };
61     
62     struct State {
63     	treenode *n;		/* defining node */
64     
65     	SymRef	*snapshot;	/* union of all versions of dfstack->symrefs when visited */
66     
67     	SymRef	*gi;		/* vars of dfstack->globrefs seen at this state */
68     	SymRef	*il;		/* intersection: vars common to all earlier visits */
69     
70     	PathCond *pc;		/* pathconds seen at earlier visits */
71     	PathCond *ip;		/* intersection: pathconds common to all earlier visits */
72     
73     	short	seenempty;	/* visited state with zero entries in globrefs? */
74     	short	seennone;	/* visited state with zero entries in knzv? */
75     
76     	ArBound *pvb;		/* previously seen bounds at this stmnt */
77     	unsigned long	uno_state;	/* bitvector of uno_states seen - dfs_generic */
78     
79     	short	iscond;		/* set if conditional branch point */
80     	short	visited;	/* dfs */
81     
82     	Trans	*succ;		/* successor chain(s) */
83     	Trans	*direct;	/* direct route for nut connections */
84     	State	*nxt;		/* temporary slot for immediate successor */
85     
86     	State	*all;		/* linked list of all states created */
87     };
88     
89     struct Graphs {
90     	char	*fctnm;		/* name of this fct, or _globals_ */
91     	State	*cfg;		/* start node for a procedure cfg */
92     	State	*all;		/* linked list of all states in this graph */
93     
94     	SymRef	*def_use;	/* records def and useb4def info on globals */
95     	SymRef	*locs;		/* list of locals, declared but not yet used in fct */
96     
97     	SymExt	*fcalls;	/* list of all fcts called in this fct */
98     	SymExt	*suspect;	/* globals with local useb4def + fcts called so far */
99     
100     #if 0
101     	Graphs	*prev;		/* to record calling chains in ana_graph */
102     #endif
103     	char	*scope;		/* set if function declared static */
104     	short	visited;
105     	short	status;		/* records if values are (always/never) returned */
106     	short	hasnuts;	/* has info in cfg for global defuse analysis */
107     	Graphs	*nxt;		/* all cfg's created */
108     };
