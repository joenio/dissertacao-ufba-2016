60     struct DuG {		/* variable dependency graph */
61     	struct	symentry *sm;	/* ptr to symbol table */
62     	int	marks;		/* decl, def, use, hide info */
63     	int	rdcls;		/* set inside or outside procedure */
64     	DuGP	*d_e;	/* llist of outgoing edges */
65     	DuG	*nxt;	/* llist of all nodes */
66     };

69     	DuG	*ptr;
