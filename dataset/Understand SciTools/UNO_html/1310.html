89     	fprintf(stderr, "\t-cfg fct\tgenerate control flow graph (from fct) in dot format\n");

158     void
159     add_pieces(FILE *fp, char *f)
160     {	FILE *x;
161     	char buf[1024];
162     
163     	if (!nopre) fprintf(fp, "#define UnoType typedef int\n");
164     
165     	fprintf(fp, "typedef int __builtin_va_list;\n"); /* gcc-ism */
166     	fprintf(fp, "typedef int __w64;\n"); /* cl-ism */
167     	fprintf(fp, "#define __builtin_va_start(v,l)\n");
168     	fprintf(fp, "#define __builtin_va_arg(v,l)	(l)0\n");
169     	fprintf(fp, "#define __builtin_va_end(v)\n");
170     
171     	if (no_va)
172     	{	fprintf(fp, "#undef va_start\n");
173     		fprintf(fp, "#define va_start(v,l)\n");
174     		fprintf(fp, "#undef va_arg\n");
175     		fprintf(fp, "#define va_arg(v,l)\t(l)0\n");
176     		fprintf(fp, "#undef va_end\n");
177     		fprintf(fp, "#define va_end(v)\n");
178     	}
179     	fprintf(fp, "#define UNO\n");
180     	fprintf(fp, "#define __builtin_offsetof(TYPE, MEMBER) ((size_t) &((TYPE *)0)->MEMBER)\n");
181     
182     	if ((x = fopen(master_dfn, "r")) != NULL)
183     	{	fprintf(fp, "/* from %s */\n", master_dfn);
184     		while (fgets(buf, sizeof(buf), x))
185     		{
186     			if (nopre
187     			&&  strncmp(buf, "UnoType", strlen("UnoType")) == 0)
188     				fprintf(fp, "typedef int %s", &buf[strlen("UnoType")+1]);
189     			else
190     				fprintf(fp, "%s", buf);
191     		}
192     		fclose(x);
193     		fprintf(fp, "/* end */\n");
194     	}
195     
196     	if (!f) return;
197     
198     	fprintf(fp, "#include \"%s\"\n", cur_file);
199     
200     	if (never) add_never(fp);
201     }

203     /*
204        Split pathname into directory (possibly empty) and filename
205        sets global var cur_path to a copy of f
206        sets global var cur_dir  to directory component of f (including
207        the trailing directory separator), and
208        sets global var cur_file to point into cur_path at filename 
209     */
210     
211     static void
212     split_filename(const char *f)
213     {	char *last_slash;
214     
215     	cur_path = (char *) emalloc(strlen (f) + 1);
216     	strcpy(cur_path, f);
217     
218     	last_slash = strrchr(f, '/');
219     #ifdef PC
220     	if (last_slash == NULL)
221     	{	last_slash = strrchr(f, '\\');
222     		if (last_slash == NULL)
223     			last_slash = strrchr(f, ':');
224     	}
225     #endif
226     	if (last_slash == NULL)
227     	{	cur_dir = (char *) emalloc(1);  /* cur_dir already contains a zero. */
228     		cur_file = cur_path;
229     	} else
230     	{	size_t dirlen = last_slash - f + 1;
231     		cur_dir = (char *) emalloc(dirlen + 1);
232     		strncpy(cur_dir, f, dirlen);    /* cur_dir has a zero afterward */
233     		cur_file = &cur_path[dirlen];
234     	}
235     }

246     void
247     process_input(char *f)
248     {	FILE *fp;
249     
250     	if (!CPRE)		/* not set on command line */
251     	{
252     #ifdef CPP
253     	CPRE = CPP;		/* set at compilation time */
254     #else
255     	#ifdef PC
256     	CPRE = "gcc -E";		/* cygwin */
257     	/* CPRE = "cl -E -nologo";	 microsoft visual studio */
258     	#else
259     		#ifdef PLAN9
260     		CPRE = "cpp";		/* plan9, brazil */
261     		#else
262     		CPRE = "gcc -E";	/* unix, linux */
263     		#endif
264     	#endif
265     #endif
266     	}
267     
268     	if (!f || strstr(f, ".c") == NULL)
269     		Usage("uno_local");
270     
271     	split_filename(f);
272     
273     	if (working_dir != NULL)
274     	{	cpp2 = (char *) emalloc(strlen(working_dir)+1+2+strlen(cur_file)+1);
275     		sprintf(cpp2, "%s/__%s", working_dir, cur_file);
276     
277     		cpp1 = (char *) emalloc(strlen(working_dir)+1+1+strlen(cur_file)+1);
278     		sprintf(cpp1, "%s/_%s", working_dir, cur_file);
279     	} else
280     	{	cpp2 = (char *) emalloc(strlen(cur_dir)+2+strlen(cur_file)+1);
281     		sprintf(cpp2, "%s__%s", cur_dir, cur_file);
282     
283     		cpp1 = (char *) emalloc(strlen(cur_dir)+1+strlen(cur_file)+1);
284     		sprintf(cpp1, "%s_%s", cur_dir, cur_file);
285     	}
286     
287     	cpp_cmd = (char *) emalloc(sizeof(char)*
288     			(strlen(CPRE)+
289     			(preproc_info?strlen(preproc_info):0)+
290     			strlen(f)+
291     			strlen(cpp2)+
292     			strlen("-I. ")+		/* just in case */
293     			32 ) );
294     
295     	if (nopre)
296     	{	FILE *f1, *f2;
297     		char *x, buf[4096];
298     
299     		if ((f1 = fopen(f, "r")) == NULL
300     		||  (f2 = fopen(cpp2, "w")) == NULL)
301     		{	fprintf(stderr, "uno: no file '%s', or cannot create '%s'\n", f, cpp2);
302     			exit(1);
303     		}
304     		add_pieces(f2, (char *) 0);
305     		fprintf(f2, "#line 1 \"%s\"\n", f);
306     		while (fgets(buf, sizeof(buf), f1))
307     		{	for (x = buf; *x == ' ' || *x == '\t'; x++)
308     				;
309     			if (strncmp(x, "#include", strlen("#include")) != 0) /* not an include */
310     				fprintf(f2, "%s", buf);
311     			else
312     				fprintf(f2, "\n");	/* maintain line count */
313     		}
314     		if (never) add_never(f2);
315     
316     		fclose(f1);
317     		fclose(f2);
318     	} else
319     	{	if ((fp = fopen(cpp2, "w")) == NULL)
320     		{	fprintf(stderr, "uno: cannot create '%s'\n", cpp2);
321     			exit(1);
322     		}
323     		add_pieces(fp, f);
324     		fclose(fp);
325     	}
326     
327     	if (working_dir != NULL)
328     		sprintf(cpp_cmd, "%s %s -I. %s >%s",
329     		CPRE, preproc_info?preproc_info:"", cpp2, cpp1);
330     	else
331     		sprintf(cpp_cmd, "%s %s %s >%s",
332     		CPRE, preproc_info?preproc_info:"", cpp2, cpp1);
333     
334     	if (Verbose>1) printf("%s\n", cpp_cmd);
335     
336     	if (system(cpp_cmd) != 0)
337     	{	fprintf(stderr, "preprocessing command '%s' failed\n", cpp_cmd);
338     		goto done;
339     	}
340     
341     	if ((fp = fopen(cpp1, "r")) == NULL)
342     	{	fprintf(stderr, "uno: cannot open '%s'\n", cpp2);
343     		goto done;
344     	}
345     
346     	init_nmetab();
347     	ParseStack = new_treestk();
348     	DoneStack  = new_treestk();
349     	contxt     = new_context();
350     	ParseStack->contxt = contxt;
351     	handle_new_file(ParseStack, fp, file_name);
352     	enter_scope(contxt);
353     	tree_parse(ParseStack, 0);	/* gram.y */
354     	name_scope(ParseStack->contxt, current_filename(), NONE_T);
355     	parse_tree = (top_of_stack(DoneStack))->parse_tree;
356     	fclose(fp);
357     
358     	if (!parse_tree)
359     		fprintf(stderr, "uno: no parsetree for %s\n", f);
360     	else if (do_dflow)
361     	{	dflow_reset();
362     		walk_tree(parse_tree, 0);	/* dflow.c */
363     	}
364     done:
365     	clean_tmps();
366     
367     	exit_scope(contxt);
368     }

370     int
371     main(int argc, char **argv)
372     {	char *arg = NULL;
373     
374     	argc--; argv++;
375     	file_name = "-";
376     	while (argc-- > 0)
377     	{	arg = *argv++;
378     
379     		if (strcmp(arg, "-allerr") == 0)
380     		{	allerr = allerrs = 1;
381     			continue;
382     		} else if (strcmp(arg, "-cfg") == 0)
383     		{	uno = 2;
384     			if (argc > 1 && *(*argv) != '-')
385     			{	want = *argv++;
386     			}
387     			continue;
388     		} else if (strcmp(arg, "-cyclo") == 0)
389     		{	cyclo = 1;
390     			continue;
391     		} else if (strcmp(arg, "-localonly") == 0)
392     		{	localonly = 1;
393     			continue;
394     		} else if (strcmp(arg, "-picky") == 0)
395     		{	lintlike = picky = 1;
396     			usecheck = 1;
397     			continue;
398     		} else if (strcmp(arg, "-fullpaths") == 0)
399     		{	nopaths = 1 - nopaths;
400     			continue;
401     		} else if (strcmp(arg, "-use") == 0)
402     		{	usecheck = 1;
403     			continue;
404     		} else if (strcmp(arg, "-nova") == 0)
405     		{	no_va = 0;
406     			continue;
407     		} else if (strcmp(arg, "-nopre") == 0)
408     		{	nopre = 1;
409     			continue;
410     		} else if (strcmp(arg, "-local") == 0)
411     		{	continue;	/* backward compatibility with modex */
412     		} else if (strcmp(arg, "-prop") == 0)
413     		{	never = (char *) emalloc(strlen(*argv)+1);
414     			strcpy(never, *argv);
415     			argc--; argv++;
416     			continue;
417     		} else if (strcmp(arg, "-static") == 0)
418     		{	see_static_fcts = 1;
419     			continue;
420     		} else if (strcmp(arg, "-shared") == 0)
421     		{	show_sharing = localonly = 1;
422     			continue;
423     		} else if (strcmp(arg, "-typedefs") == 0)
424     		{	list_typedefs = 1;
425     			continue;
426     		} else if (strcmp(arg, "-extern") == 0)
427     		{	see_extern_fcts = 1;
428     			continue;
429     		} else if (strcmp(arg, "-exit") == 0)
430     		{	custom_exit((const char *) *argv);
431     			argc--; argv++;
432     			continue;
433     		} else if (strcmp(arg, "-master") == 0)
434     		{	master_dfn = (char *) emalloc(strlen(*argv)+1);
435     			strcpy(master_dfn, *argv);
436     			argc--; argv++;
437     			continue;
438     		} else if (strncmp(arg, "-W", 2) == 0 && strlen(arg) > 2)
439     		{	working_dir = (char *) emalloc(strlen(arg)+1-2);
440     			strcpy(working_dir, &arg[2]);
441     			continue;
442     		} else if (strcmp(arg, "-V") == 0)
443     		{	printf("%s\n", VERSION);
444     			exit(0);
445     		} else if (strncmp(arg, "-v", 2) == 0)
446     		{	Verbose++;	/* -v or e.g. -verbose */
447     			continue;
448     		} else if (strncmp(arg, "-w", 2) == 0)
449     		{	keeptmps = 1;	/* intermediate files */
450     			continue;
451     		} else if (strcmp(arg, "-s") == 0)
452     		{	show_syms = 1;
453     			continue;
454     		} else if (strcmp(arg, "-t") == 0)
455     		{	type_check = 1;
456     			continue;
457     		}
458     
459     		if (strncmp(arg, "-CPP=", strlen("-CPP=")) == 0)
460     		{	CPRE = arg+strlen("-CPP=");
461     			continue;
462     		}
463     
464     		if (strncmp(arg, "-U", 2) == 0
465     		||  strncmp(arg, "-D", 2) == 0
466     		||  strncmp(arg, "-I", 2) == 0)
467     		{	if (!preproc_info)
468     			{	preproc_info = (char *) emalloc(strlen(arg)+1);
469     				strcpy(preproc_info, arg);
470     			} else
471     			{	char *pp = (char *) emalloc(strlen(arg)+strlen(preproc_info)+1+1+2);
472     				strcpy(pp, preproc_info);
473     				strcat(pp, " \""); /* quote all but the first one */
474     				strcat(pp, arg);
475     				strcat(pp, "\"");
476     				preproc_info = pp;
477     			}
478     			continue;
479     		}
480     		break;
481     	}
482     
483     	custom_exit("exit");		/* fcts known not to return */
484     	custom_exit("fatal");
485     	custom_exit("panic");
486     
487     	if (!arg) Usage("uno");
488     
489     	read_suppress();	/* read an optional uno_suppress file */
490     	
491     	for (file_name = arg; file_name && argc >= 0; file_name = *argv++, argc--)
492     	{	if (Verbose)
493     			printf("uno: %s\n", file_name);
494     
495     		find_suppress_lines(file_name);
496     
497     		process_input(file_name);	/* parses file */
498     
499     		if (parse_tree
500     		&&  !see_extern_fcts
501     		&&  !see_static_fcts)
502     			lts_start(parse_tree);		/* uno_lts.c   */
503     
504     		if (show_syms)
505     			show_symtab(contxt->syms, stdout);
506     
507     		if (Verbose)
508     			memstats();
509     
510     		free_context(contxt);	/* done with this file */
511     
512     		while (!is_empty(ParseStack))
513     			delete_stk_item(pop(ParseStack));
514     
515     		if (Verbose>1) printf("parsestack\n");
516     
517     		while (!is_empty(DoneStack))
518     			delete_stk_item(Parse_TOS = pop(DoneStack));
519     		if (Verbose>1) printf("donestack\n");
520     
521     		Parse_TOS  = (Stk_Item  *) 0;
522     		DoneStack  = (TreeStack *) 0;
523     		parse_tree = (treenode *) 0;
524     		ParseStack = (TreeStack *) 0;
525     	}
526     	if (Verbose>1) printf("done\n");
527     
528     	return 0;
529     }
