26     static Stk_Item *freestk = (Stk_Item *) 0;
27     
28     Stk_Item  *
29     new_stk_item( FILE *fp, char *fname )
30     {
31         Stk_Item *stk_item;
32     
33     	if (freestk)
34     	{	stk_item = freestk;
35     		freestk = freestk->next;
36     		memset(stk_item, 0, STK_ITEM_SZE);
37     	} else
38     		stk_item = emalloc(STK_ITEM_SZE);
39     
40         stk_item->filename = emalloc(strlen(fname)+1);
41         strcpy(stk_item->filename, fname);
42     
43         if ((stk_item->yybuff = yy_create_buffer(fp, YYBUFF_SIZE )) == NULL){
44             efree(stk_item->filename);
45             efree(stk_item);
46             return((Stk_Item *) NULL);
47         }
48     
49         if ((stk_item->node_heap  = CreateHeap(MX_NODE_SZE, 0)) == NULL){
50             efree(stk_item->filename);
51             yy_delete_buffer( stk_item->yybuff );
52             efree(stk_item);
53             return((Stk_Item *) NULL);
54         }
55     
56         stk_item->yyin     = fp;
57     
58         stk_item->yylineno = 1;
59         stk_item->yycolno  = 0;
60         stk_item->yynxtcol = 0;
61     
62         stk_item->parse_tree = (treenode *) NULL;
63     
64         stk_item->next = (Stk_Item *) NULL;
65     
66         return(stk_item);
67     }

69     void
70     delete_stk_item(Stk_Item *stk_item)
71     {
72     	if (!stk_item)
73     		return;
74     #ifdef PC
75     	if (stk_item->yybuff)
76     		yy_delete_buffer(stk_item->yybuff);
77     #endif
78     	if (stk_item->node_heap)
79     	{	DestroyHeap(stk_item->node_heap);
80     		memset(stk_item, 0, sizeof(Stk_Item));
81     	}
82     
83     	stk_item->next = freestk;
84     	freestk = stk_item;
85     }

87     TreeStack *
88     new_treestk(void)
89     {	TreeStack *treestk;
90     
91     	treestk = (TreeStack *) emalloc(sizeof(TreeStack));
92     	treestk->top    = NULL;
93     	treestk->bottom = NULL;
94     
95     	treestk->contxt = NULL;
96     	return treestk;
97     }

99     void
100     push(TreeStack *treestk, Stk_Item *stk_item )
101     {
102         if (! treestk || ! stk_item)
103             return;
104     
105         stk_item->next = treestk->top;
106     
107         if (! treestk->top)
108             treestk->bottom = stk_item;
109     
110         treestk->top = stk_item;
111     }
112     
113     Stk_Item *
114     pop(TreeStack *treestk)
115     {
116         Stk_Item *stk_item, *nxt_item;
117     
118         if (is_empty(treestk))
119             return((Stk_Item *) NULL);
120     
121         stk_item = treestk->top;
122         nxt_item = treestk->top->next;
123     
124         if (! nxt_item)
125             treestk->bottom = (Stk_Item *) NULL;
126     
127         treestk->top = nxt_item;
128     
129         return(stk_item);
130     }

132     int
133     is_empty(TreeStack *treestk)
134     {
135     	if (!treestk)
136     		return(1);
137     
138     	return(treestk->top == NULL);
139     }

141     Stk_Item *
142     top_of_stack(TreeStack *treestk)
143     {
144         if (is_empty(treestk))
145             return((Stk_Item *) NULL);
146         else
147             return(treestk->top);
148     }

150     FILE *
151     top_file(TreeStack *treestk)
152     {
153         if (is_empty(treestk))
154             return((FILE *) NULL);
155         else
156             return(treestk->top->yyin);
157     }

159     void
160     reset_position(TreeStack *treestk)
161     {
162         if ( !(Parse_TOS = top_of_stack(treestk)))
163             return;
164     
165         yy_switch_to_buffer(Parse_TOS->yybuff);
166         yyin = top_file(treestk);
167     }

169     int
170     tree_parse(TreeStack *treestk, int parse_all)
171     {
172         int cnt = 0;
173     
174         if (is_empty(treestk))
175             return(0);
176     
177         do {
178     	yyparse();	/* returns 1 on error 0 on accept */
179             cnt++;
180             get_next_file(treestk);
181     
182             if (is_empty(treestk))
183                 break;
184     
185         } while (parse_all);
186     
187         return(cnt);
188     }
189     
190     void
191     handle_new_file(TreeStack *treestk, FILE *fp, char *fname)
192     {
193         Stk_Item *stk_item;
194     
195         if ((stk_item = new_stk_item(fp, fname)) == NULL)
196             return;
197     
198         push(treestk, stk_item);
199         reset_position(treestk);
200     }
201     
202     int
203     get_next_file(TreeStack *treestk)
204     {
205         if (is_empty(treestk)){
206             Parse_TOS = (Stk_Item *) NULL;
207             return(0);
208         }
209     
210         if (DoneStack)
211             push(DoneStack, pop(treestk));
212         else {
213             fputs("DoneStack was NULL.\n", stdout);
214             delete_stk_item(pop(treestk));
215         }
216     
217         reset_position(treestk);
218     
219         if (is_empty(treestk))
220             return(0);
221     
222         return(1);
223     }
