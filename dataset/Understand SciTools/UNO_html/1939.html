134     get_suppressors(FILE *fd)
135     {	char *p, buf[1024];
136     	ST *nst;
137     
138     	while (fgets(buf, sizeof(buf), fd))
139     	{	if ((p = strchr(buf, '\t')) == NULL)

141     		} else
142     		{	*p++ = '\0';
143     			nst = (ST *) emalloc(sizeof(ST));
144     			nst->name = (char *) emalloc(strlen(buf)+1);
145     			strcpy(nst->name, buf);
146     			if (!isdigit(*p))
147     			{	nst->line = 0;	/* meaning all lines */
148     			} else
149     			{	nst->line = atoi(p);
150     			}
151     			nst->nxt = st;
152     			st = nst;
153     	}	}

239     void
240     uno_warn(char *msg, treenode *n, symentry_t *s)
241     {	char *str;
242     
243     	if (n)
244     	{	if (suppress(n->hdr.fnm, n->hdr.line))
245     			return;
246     	} else if (s)
247     	{	if (suppress(s->fn, s->ln))
248     			return;
249     	}
250     
251     	uno_assert(curgraph && s, "bad input to uno_warn");
252     
253     	if (strcmp(msg, "variable never used (evaluated)") == 0)
254     	{
255     		if (strncmp(s->nme->str, "unused", strlen("unused")) == 0)
256     			return;
257     
258     		if (n)
259     			str = x_stmnt(n);
260     		else
261     			str = "(declaration)";
262     
263     		fprintf(stdout, "uno: ");
264     
265     		if (n)	fprintf(stdout, "%s:%d:", n->hdr.fnm, n->hdr.line);
266     		else if (s) fprintf(stdout, "%s:%d", s->fn, s->ln);
267     
268     		fprintf(stdout, "\tin fct %s, %s '%s' ",
269     			curgraph->fctnm, msg, s->nme->str);
270     
271     	} else if (n && n->hdr.type != TN_EMPTY)
272     	{	if (strstr(x_stmnt(s->node), "va_list") != NULL)
273     			return;
274     
275     		fprintf(stdout, "uno: in fct %s, %s '%s'",
276     			curgraph->fctnm, msg, s->nme->str);
277     
278     		fprintf(stdout, "\n     statement  : %s:%d: ",
279     			n->hdr.fnm, n->hdr.line);
280     
281     		if (n->hdr.type != TN_IF)
282     		{	str = x_stmnt(n);
283     			if (!str || strlen(str) == 0)
284     				str = "<stmnt>";
285     			fprintf(stdout, "%s", str);	/* src of offending stmnt */
286     		} else
287     		{	str = x_stmnt(((if_node *)n)->cond);
288     			if (!str || strlen(str) == 0)
289     				str = "<cond>";
290     			fprintf(stdout, "%s", str);	/* src of offending stmnt */
291     		}
292     
293     		if (n->hdr.line != s->ln)		/* if different from above */
294     			fprintf(stdout, "\n     declaration: %s:%d: %s",
295     				s->fn, s->ln, x_stmnt(s->node)); /* decl src */
296     	} else
297     	{	fprintf(stdout, "uno: in fct %s, %s '%s'",
298     			curgraph->fctnm, msg, s->nme->str);
299     	}
300     
301     	fprintf(stdout, "\n");
302     }

400     {	Stack *f;
401     	int cnt = 0;
402     	Stack *lst = (Stack *) 0;
403     	char *s, *t;

405     	if (!fallthru) return;

408     	{	if (lst && lst->n->hdr.line == f->n->hdr.line + 1)
409     		{	f->status = 1;

414     
415     	for (f = fallthru; f; lst = f, f = f->nxt)
416     		if (f->status == 0)
417     			cnt++;
418     
419     	if (cnt == 0) return;
420     
421     	fprintf(stdout, "uno: fallthroughs in switch stmnt:");
422     	for (f = fallthru; f; f = f->nxt)
423     	{
424     		if (f->status != 0)
425     			continue;
426     
427     		s = x_stmnt(f->n);
428     		if (strlen(s) > 0)
429     		{	t = strchr(s, ':');
430     			if (t) *t = '\0';
431     		}
432     		fprintf(stdout, "\n");
433     		fprintf(stdout, "\t%s:%d:",
434     			f->n->hdr.fnm, f->n->hdr.line);
435     		if (strlen(s) > 0)
436     			fprintf(stdout, " <%s>", s);

562     static State *
563     lts_redirect(State *s)
564     {	State *st = s;
565     	leafnode *ln;
566     	SwStack *frame;
567     
568     	if (!s || !s->n) return s;
569     
570     	for (frame = stck; frame; frame = frame->nxt)
571     		if (frame->s == s)
572     		{	fprintf(stderr, "uno: %s:%d: goto cycle\n",
573     				s->n->hdr.fnm,
574     				s->n->hdr.line);
575     			return s;
576     		}
577     
578     	frame = (SwStack *) emalloc(sizeof(SwStack));
579     	frame->s = s;
580     	frame->nxt = stck;
581     	stck = frame;
582     	
583     	if (s->n->hdr.which == NODE_T)
584     	{	switch (s->n->hdr.type) {
585     		case TN_JUMP:
586     			if (s->n->hdr.tok == RETURN)
587     				break;
588     			if ((ln = (leafnode *) s->n->lnode) != NULL)	/* assign */
589     			{	st = lts_label_find(ln->data.sval->str);
590     				if (!st) st = s;
591     			}
592     			break;
593     
594     		case TN_BLOCK:
595     		case TN_STEMNT:
596     		case TN_LABEL:	/* skip fillers, if unique successor */
597     			if (!s->succ)	/* not yet set, but already known to be unique */
598     				st = s->nxt;
599     			else if (!s->succ->nxt)
600     				st = s->succ->branch;
601     
602     			if (!st) st = s;
603     			break;
604     		default:
605     			break;
606     	}	}
607     
608     	if (st != s)
609     		st = lts_redirect(st);
610     
611     	stck = stck->nxt;	/* pop frame stack */
612     
613     	return st;
614     }

694     static int
695     killpath(treenode *z)
696     {	int hasval = 0, thisval = 0;
697     
698     	if (z->hdr.type == TN_EXPR
699     	&&  z->hdr.tok  == COMMA)	/* foo(),goo() */
700     		return killpath(z->rnode) || killpath(z->lnode);
701     
702     	if (z->hdr.type == TN_FUNC_CALL
703     	&&  z->lnode->hdr.type == TN_IDENT) /* function call */
704     	{
705     		/* check if the first argument is an int */
706     		if (z->rnode)
707     		{	treenode *t = z->rnode;
708     
709     			while (t && t->hdr.type == TN_EXPR_LIST)
710     				t = t->lnode; /* leftmost */
711     		 	if (t
712     		 	&& t->hdr.which == LEAF_T
713     		 	&& t->hdr.type == TN_INT)
714     			{	hasval = 1;
715     				thisval = ((leafnode *)t)->data.ival;
716     		}	}
717     
718     		return (no_return(((leafnode *)z->lnode)->data.sval->str, hasval, thisval));
719     	}
720     
721     	return 0;
722     }

724     static void
725     prep_graph(Graphs *g)
726     {	State *s;
727     	Trans *t;
728     	int cnt;
729     
730     	curgraph = g;
731     
732     	for (s = g->all; s; s = s->all)
733     	{	s->visited = 0;
734     		if (!s->succ)
735     		{	if (s->n)
736     			{	if (s->n->hdr.tok == RETURN)
737     					continue;
738     
739     				if (killpath(s->n))	/* non-returning fct called */
740     					continue;
741     			}
742     
743     			if (s->n
744     			&&  s->n->hdr.tok != GOTO
745     			&&  s->n->hdr.tok != COLON
746     			&& !s->nxt
747     			&&  (!s->n->hdr.fnm || strcmp(s->n->hdr.fnm, "-1") != 0))
748     			{	g->status |= 2;	/* no value returned */
749     
750     				if (0) printf("no value returned at %s:%d: -- %s = %d\n",
751     					s->n->hdr.fnm, s->n->hdr.line,
752     					toksym(s->n->hdr.tok,0),
753     					s->n->hdr.tok);
754     				if (debug)
755     				printf("prepgraph %s adds 2 <%d:%d> %s:%d\n",
756     					g->fctnm,
757     					(int) s->n->hdr.type,
758     					s->n->hdr.tok,
759     					s->n->hdr.fnm,
760     					s->n->hdr.line);
761     			}
762     
763     			t = get_trans();
764     			t->branch = lts_redirect(s->nxt);
765     			t->nxt = (Trans *) 0;
766     			s->succ = t;
767     		} else
768     		{	cnt = 0;
769     			for (t = s->succ; t; t = t->nxt)
770     			{	t->branch = lts_redirect(t->branch);
771     				cnt++;
772     			}
773     			s->iscond = (cnt > 1);
774     
775     			/* find cases like "if (c);" */
776     			if (picky == 1
777     			&&  s->n
778     			&&  suppress(s->n->hdr.fnm, s->n->hdr.line) == 0
779     			&&  cnt == 2
780     			&&  s->n->hdr.type == TN_IF
781     			&&  s->succ->branch == s->succ->nxt->branch)
782     				printf("uno: %s:%d: in fct %s, 'if' followed by null-statement\n",
783     					s->n->hdr.fnm, s->n->hdr.line, curgraph->fctnm);
784     	}	}
785     }

788     nut_prepare(State *from, State *to)	/* insert forward arcs to nodes with a nut attached */
789     {	Trans *t;
790     
791     	if (!to || !to->n || to->visited)
792     		return;
793     	to->visited = 1;

795     	if (to->n->hdr.nuts
796     	|| !(to->succ && to->succ->branch))
797     	{	for (t = from->direct; t; t = t->nxt)
798     			if (t->branch == to)
799     				break;

805     		}
806     		return;
807     	}

809     	for (t = to->succ; t; t = t->nxt)
810     		nut_prepare(from, t->branch);

1222     static void
1223     uno_guse(void)
1224     {	SymRef *r;
1225     	Graphs *g;
1226     
1227     	/* flags: 1 - used, 2 - static, 4 - extern, 8 - def-ed, 16 - useb4def, 32 - alias */
1228     	/* 64 - fct param, 128 - array decl */
1229     
1230     	/* local check of unused globals in this file */
1231     
1232     	if (usecheck)
1233     	for (r = globuse; r; r = r->nxt)
1234     	{	int st;
1235     
1236     		if (0)	printf("%s - %d \t%s:%d: <-> %s\n",
1237     			r->sm->nme->str,
1238     			r->status,
1239     			r->n->hdr.fnm,
1240     			r->n->hdr.line,
1241     			file_name);
1242     
1243     		if (r->n
1244     		&&  r->n->hdr.fnm
1245     		&&  strcmp(r->n->hdr.fnm, file_name) != 0)
1246     			continue;
1247     
1248     		st = r->status&~128;
1249     		if (r->status & 128) st |= 8; /* array decl */
1250     
1251     		if (!(st & 32)	/* no alias taken */
1252     		&&  !(st & 64))	/* no fct param */
1253     		switch (st) {
1254     		case   0: /* normal -def -use */
1255     			uno_complain(r, "", "not assigned to or used in this file");
1256     			break;
1257     
1258     		case     8: /* normal +def -use */
1259     			if (picky || Verbose)
1260     			uno_complain(r, "", "assigned to but not used in this file");
1261     			break;
1262     
1263     		case   2: /* static -def -use */
1264     			uno_complain(r, "static ", "never assigned to or used");
1265     			break;
1266     
1267     		case 2|8: /* static +def -use */
1268     			uno_complain(r, "static ", "assigned to but not used");
1269     			break;
1270     
1271     		case 1|2: /* static -def +use */
1272     		case 1|2|16: /* static -def  +use +useb4def */
1273     			if (on_glob(r->sm))
1274     			uno_complain(r, "static ", "used but never assigned to");
1275     			break;
1276     
1277     		case 1|2|8|16: /* static +def  +use +useb4def */
1278     			if (lintlike && on_glob(r->sm))	/* picky mode */
1279     			uno_complain(r, "static ", "may be used before set");

1942     static int
1943     covered(State *s, SymRef *gr)
1944     {	SymRef *r, *q, *lq;
1945     	int cnt = 0;
1946     #if 0
1947     	this check secures coverage of global variables use only
1948     	we are interested in U or R before D errors
1949     	so the presence or absence of a DEF (1) on any gvar is important
1950     	and determines its uniqueness

1952     	individual variables have been seen with certain flags

1955     	/* 1: each var in gr must have appeared in at least one
1956     	 *    previous visit with the same DEF status, as recorded
1957     	 *    in a list s->gi stored at state s for this purpose
1958     	 */

1960     	{	if (!s->seenempty)
1961     		{	s->seenempty = 1;
1962     			cnt++;
1963     		}
1964     		goto done;
1965     	}
1966     
1967     	if (!(s->gi))	/* first visit */
1968     	{	s->gi = copy_list(gr);	/* seed initial visit list */
1969     		s->il = copy_list(gr);	/* seed intersection list */
1970     		cnt++;
1971     		goto done;
1972     	} else
1973     	for (r = gr; r; r = r->nxt)		/* each var in gr (dfstack->globrefs) */
1974     	{	for (q = s->gi; q; q = q->nxt)	/* see if already covered in s->gi */
1975     			if (q->sm == r->sm
1976     			&&  (r->status & DEF) == (q->status & DEF))
1977     				break;
1978     		if (!q)	/* not covered yet */
1979     		{	q = uno_copy_ref(r);
1980     			q->nxt = s->gi;
1981     			s->gi = q;
1982     			cnt++;
1983     	}	}
1984     
1985     	/* 2: each var absent from gr
1986     		must have been absent in at least one previous visit
1987     		i.e., must be absent from the intersection list
1988     	 */
1989     	lq = (SymRef *) 0;
1990     	for (r = s->il; r; r = r->nxt)
1991     	{	if (!in_lst(gr, r))	/* should drop r from il */
1992     		{	if (!lq)
1993     				s->il = r->nxt;
1994     			else
1995     				lq->nxt = r->nxt;
1996     			cnt++;
1997     		} else
1998     			lq = r;		/* good, it's there */
1999     	}
2000     done:
2001     	if (debug)
2002     	{	printf("Covered %s:\n", (cnt==0)?"yes":"no");
2003     		for (r = gr; r; r = r->nxt)
2004     			printf("	%s	%d\n", r->sm->nme->str, r->status);
2005     		printf("GI:\n");

2007     			printf("	%s	%d\n", r->sm->nme->str, r->status);
2008     		printf("IL:\n");
2009     		for (r = s->il; r; r = r->nxt)
2010     			printf("	%s	%d\n", r->sm->nme->str, r->status);
2011     	}
2012     
2013     	return (cnt == 0);	/* true if nothing needed adding */
2014     }

2102     global_var(SymList *s, treenode *n)	/* track global, default-initialized, pointers */
2103     {	leafnode *l;

2105     
2106     	if (s->mark & DECL)
2107     	{	l = leftmost(n);
2108     
2109     		if (s->mark & DEF)
2110     			extra |= 8;
2111     
2112     		if (s->mark & ARRAY_DECL)
2113     			extra |= 128;		/* was 8 */
2114     
2115     		if ((s->mark & PARAM)
2116     		&&  (s->mark & USE))		/* comes from prototype decl or enum */

2142     		||  (s->mark & DEREF))

2144     			add_gstack(s->sm, n, USE);	/* global used in this procedure */
2145     		}

2150     		}
2151     
2152     		if ((s->mark & (DEF|UNO_CONST)) && !(s->mark & DEREF))
2153     		{	mark_guse(s->sm, n, 8);		/* at least one DEF */
2154     			add_gstack(s->sm, n, DEF);	/* global set in this procedure */
2155     		}

2157     		if (on_glob(s->sm)	/* of interest: an uninitialized global ptr */
2158     		&& !on_safe(s->sm))	/* not already declared moot on this path */

2160     			&& !(s->mark & REF0)

2165     			}

2168     				if (checkpathcond(s->sm, n, 1))	/* unless value known */
2169     					add_uses(s->sm, n);	/* use_b4_def in fct */
2170     			}
2171     			if ((s->mark & REF0)		/* ref0->x */
2172     			||  (s->mark == DEREF)		/* *ptr = x; */
2173     			|| ((s->mark & DEREF) && (s->mark & (USE|DEF|FCALL))))	/* ptr deref */
2174     			{

2463     {

2465     
2466     	if (ex->hdr.type == TN_IDENT)
2467     		return 1;	/* s */
2468     
2469     	if (ex->hdr.type != TN_EXPR
2470     	||  ex->hdr.tok == AND
2471     	||  ex->hdr.tok == OR)
2472     		return 0;

2475     	&&  ex->lnode == ZT)

2487     		&&  ex->rnode->hdr.tok == EQUAL
2488     		&&  ex->rnode->lnode

2490     		&&  is_zero_val(ex->rnode->rnode))
2491     			return 1;	/* !(s == 0) */
2492     	} else {

2494     		&&  ex->lnode
2495     		&&  ex->lnode->hdr.type == TN_IDENT
2496     		&&  is_zero_val(ex->rnode))
2497     			return 1;	/* (s == 0) or (s != 0) */
2498     	}
2499     	return 0;

3612     			t->hdr.deftyp = forn->test->hdr.deftyp;
3613     #endif
3614     		}

3620     		now->succ = tr;			/* with its two branches */
