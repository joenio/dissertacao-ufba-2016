13     /*  
14         SymTab:  A tree of scope tables,
15                  one for each scope that declares something, or has a
16                  child scope that declares something:
17     
18              Level
19                1              external
20                              /        \
21                2          file        file 
22                           scope       scope 
23                                     /       \
24                3              prototype    function
25                                 scope        scope
26                                             /     \
27                4                         block    block
28                                          scope    scope    
29                                                      \
30                5                                     block
31                                                      scope    
32                                                         \
33                                                        (etc.)
34     
35         At any particular point you can see all the symbols 
36         (variables, types, functions, etc) declared above you in the tree.
37     
38         The scope tables are recreated in a lazy fashion, so entering
39         and exiting scopes that don't declare new symbols is cheap.
40      */

50     /* in the same namespace: */
51     #define TYPEDEF_ENTRY	(1)	/* type def */
52     #define FUNCDEF_ENTRY	(2)	/* function def */
53     #define VARDECL_ENTRY	(3)	/* variable decl */
54     #define ENUM_ENTRY	(4)	/* enum constants. */
55     
56     /* in seperate namespaces: */
57     #define LABEL_ENTRY	(5)	/* label def */
58     #define TAG_ENTRY	(6)	/* struct/union/enum tags */
59     
60     /* separate namespace for each struct/union: */
61     #define COMP_ENTRY	(7)	/* components of struct/union */
62     
63     #define PARAM_ENTRY	(8)	/* added gjh - set and checked in dflow.c */
64     
65     #define CURRENT_SCOPE	(0)	/* default */
66     #define EXTERN_SCOPE	(1)
67     #define FILE_SCOPE	(2)
68     #define FUNCTION_SCOPE	(3)
69     #define BLOCK_SCOPE	(4)	/* really 4 and up */

71     typedef struct symentry {
72     	int		kind;	/* TYPEDEF, ENUM, FUNCDEF, VARDECL, PARAM, LABEL, TAG, COMP */
73     	str_t		*nme;	/* name */
74     	treenode	*node;	/* ptr to def/decl */
75     
76     	struct scopetab	*nes;	/* ptr to enclosing scope */
77     	short		decl_level;	/* added gjh */
78     	short		used;	/* for struct elements, to see if used */
79     	int		ln;	/*	linenumber */
80     	char		*fn;	/*	file name */
81     
82     	treenode	*container; /* for struct/union component, ptr to def of container */
83     	struct symentry	*next;
84     } symentry_t;

114     typedef struct scopetab {
115     	int               nsyms;	/* nr of syms declared at this scope. */
116     	int               level;	/* This scopetab's scoping level. */
117     	hashtab_t        *htab;		/* hash table - to store the symbols. */
118     	struct scopetab  *parent;	/* The scope enclosing us, if any. */
119     
120     	char *owner;			/* name of owner of this scope, functio/file/object */
121     	int   owner_t;			/* type: TN_OBJ_DEF, TN_FUNC_DEF or NONE_T for Filescope */
122     
123     	int		  visited;	/* gjh -- facilitate searches */
124     	int               nchild;
125     	int               size;
126     	struct scopetab **children;	/* doubling array of scopes we enclose */
127     } scopetab_t;
