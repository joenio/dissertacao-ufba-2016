18     #define BITS_IN_int	(sizeof(int) * CHAR_BIT)
19     #define THREE_QUARTERS	((int) ((BITS_IN_int * 3) / 4))
20     #define ONE_EIGHTH	((int) (BITS_IN_int / 8))
21     #define HIGH_BITS	(~((unsigned int)(~0) >> ONE_EIGHTH))
22     
23     #define HASH_ITEM_SZE	(sizeof(HashItem))
24     #define MAX_HASH_BCKTS	1023
25     
26     typedef struct hi {
27         str_t      sym;
28         struct hi *next;
29     } HashItem;
30     
31     static HashItem *NmeTab[MAX_HASH_BCKTS];
32     
33     extern void *emalloc(size_t);
34     
35     char *
36     nmestr(str_t *sym)
37     {
38     	if (sym) return sym->str;
39     
40     	return NULL;
41     }
42     
43     static unsigned int
44     calc_hash(char *str)	/* pjw's hash */
45     {	unsigned int hsh = 0, i;
46     
47     	while (*str)
48     	{	hsh = (hsh << ONE_EIGHTH) + *str++;
49     		if ((i = hsh & HIGH_BITS) != 0)
50     			hsh = (hsh ^ ( i >> THREE_QUARTERS)) & ~HIGH_BITS;
51     	}
52     	return hsh;
53     }
54     
55     void
56     init_nmetab(void)
57     {	int j;
58     
59     	for (j = 0; j < MAX_HASH_BCKTS; j++)
60     		NmeTab[j] = (HashItem *) NULL;
61     }
62     
63     str_t *
64     nmelook(char *sym, int len)
65     {	HashItem *hptr;
66     	unsigned int hsh;
67     	int bckt;
68     
69     	hsh  = calc_hash(sym);
70     	bckt = hsh % MAX_HASH_BCKTS;
71     
72     	for (hptr=NmeTab[bckt]; hptr; hptr=hptr->next)
73     	{	if ((hptr->sym.hash == hsh)
74     		&&  (strcmp(sym, hptr->sym.str) == 0))
75     			return(&(hptr->sym));
76     	} 
77     
78     	hptr = (HashItem *) emalloc(HASH_ITEM_SZE);
79     
80     	if (!len) len = strlen(sym);
81     
82     	hptr->sym.str = emalloc(len+1);
83     	hptr->sym.hash = hsh;
84     	strncpy(hptr->sym.str, sym, len);
85     
86     	hptr->next = NmeTab[bckt];
87     	NmeTab[bckt] = hptr;
88     
89     	return(&(hptr->sym));
90     }
