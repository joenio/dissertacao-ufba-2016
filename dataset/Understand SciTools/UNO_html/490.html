11     /* dataflow tags used in dflow.c, uno_lts.c, and uno_global.c:
12      * bitfields stored in a 32bit unsigned integer field called 'mark'
13      */
14     
15     #define DEF		1		/*       1  x = y */
16     #define FCALL		(1<<1)		/*       2  f()   */
17     #define USE		(1<<2)		/*       4  y = x */
18     #define REF0		(1<<3)		/*       8  x->y  */
19     #define REF1		(1<<4)		/*      16  x.y   */
20     #define REF2		(1<<5)		/*      32  y.x or y->x */
21     #define DEREF		(1<<6)		/*      64  *x    */
22     #define ALIAS		(1<<7)		/*     128  &x    */
23     #define ARRAY_DECL	(1<<8)		/*     256  x[]   */
24     #define HIDE		(1<<9)		/*     512        */
25     #define DECL		(1<<10)		/*    1024        */
26     #define USEafterdef	(1<<11)		/*    2048  y = x++ */
27     #define USEbeforedef	(1<<12)		/*    4096  x++   */
28     #define UNO_CONST	(1<<13)		/*    8192  f(x)  */
29     #define PARAM		(1<<14)		/*   16384        */
30     #define INCOND		(1<<15)		/*   32768  if (x) */
31     #define ISTATIC		(1<<16)		/*   65536  static int x */
32     #define IN_SIZEOF	(1<<17)		/*  131072  sizeof(x) */
33     #define IS_PTR		(1<<18)		/*  262144  decl of ptr */
34     
35     #define NONE		0
36     #define ANY		((1<<20)-1)
37     
38     #define SELECTED	(1<<24)		/* 16777216 outside ANY, below 31 */
