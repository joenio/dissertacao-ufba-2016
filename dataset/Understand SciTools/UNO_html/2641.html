98     static char *
99     new_str(char *s)
100     {	Pool *p;
101     	char *t;
102     	int n;
103     
104     	n = strlen(s);
105     	if (n > 0 && n < 512 && pool[n])
106     	{	p = pool[n];
107     		pool[n] = p->nxt;
108     		p->nxt = free_pool;
109     		free_pool = p;
110     		t = p->s;
111     	} else
112     		t = (char *) emalloc(n+1);
113     	strcpy(t, s);
114     	return t;
115     }

213     void
214     add_fct(char *s)
215     {	BFct *f;
216     
217     	if (0 && debug) printf("adding fct %s\n", s);
218     
219     	for (f = fcts; f; f = f->nxt)
220     		if (strcmp(f->fnm, s) == 0)
221     			return;	/* reported in uno_global.c */
222     
223     	f = (BFct *) emalloc(sizeof(BFct));
224     	f->fnm = new_str(s);
225     	f->nxt = fcts;
226     	fcts = f;
227     
228     	f->root = (Node *) emalloc(sizeof(Node));
229     	f->root->nid = 0;
230     	return;

233     void
234     add_call(char *fnm, char *c)
235     {	BFct *f;
236     	BSym *s;
237     
238     	f = find_function(fnm);
239     	if (!f)
240     	{	fprintf(stderr, "error: bad input sequence (%s)\n",
241     			fnm);
242     		exit(1);
243     	}
244     
245     	for (s = f->calls; s; s = s->nxt)
246     		if (strcmp(s->s, c) == 0)
247     			return;
248     
249     	s = (BSym *) emalloc(sizeof(BSym));
250     	s->s = new_str(c);
251     	s->nxt = f->calls;
252     	f->calls = s;
253     }

257     void
258     find_roots(void)
259     {	BFct *f, *g;
260     	BSym *h;
261     	int ncalls;
262     
263     	for (f = fcts; f; f = f->nxt)
264     	{	if (strcmp(f->fnm, "_global_") == 0)
265     			continue;
266     		ncalls = 0;
267     		for (g = fcts; g; g = g->nxt)
268     		{	if (g != f)
269     			{	for (h = g->calls; h; h = h->nxt)
270     					if (strcmp(h->s, f->fnm) == 0)
271     					{	ncalls++;
272     						break;
273     					}
274     				if (ncalls > 0)
275     					break;
276     			}
277     			if (ncalls > 0)
278     				break;
279     		}
280     		if (ncalls == 0 && indirect_calls(f->fnm) == 0)
281     			printf("root: %s\n", f->fnm);
282     	}
283     	fflush(stdout);
284     }

317     static void
318     add_label(BFct *f, Arc *a, char *s)
319     {	Label *n;
320     	char  *p;
321     
322     	while ((p = strchr(s, '\t')) != NULL)
323     		*p = ' ';
324     
325     	n = (Label *) emalloc(sizeof(Label));
326     	n->label = (unsigned char *) new_str(s);
327     
328     	if (strcmp(f->fnm, glob_prop) == 0)
329     	{	n->nxt = a->lab;	/* prop: labels on arcs */
330     		a->lab = n;
331     		return;
332     	} else
333     	{	n->nxt = a->to->lab;	/* sys: labels on nodes */
334     		a->to->lab = n;
335     	}
336     
337     	if (strlen(s) <= 2)
338     		return;
339     
340     	switch (s[1]) {
341     	case ' ':	p = strchr(&s[2], ' '); break;
342     	case '\t':	p = strchr(&s[2], '\t'); break;
343     	default:
344     bad:		fprintf(stderr, "uno:global: unexpected type of label: '%s'\n", s);
345     		return;
346     	}
347     
348     	if (!p) goto bad;
349     	*p = '\0';
350     	switch (s[0]) {
351     	case 'C': 
352     	case 'c': 
353     	case 'b': 
354     	case 'h': 
355     	case 'i': 
356     	case 'j':
357     		/* add_call(f->fnm, &s[2]); */
358     		break;
359     	case 'N':
360     	case 'G':
361     	case 'D':
362     	case 'U':
363     	case 'R': break;
364     	default : goto bad;
365     	}
366     	*p = ' ';	/* restore */
367     }

369     static void
370     ini_prop(void)
371     {	Node *n;
372     
373     	prop_fct = find_function(glob_prop);
374     	if (!prop_fct)
375     	{	if (verbose)
376     			printf("uno:global: using builtin property check\n");
377     		builtin_check();
378     		prop_fct = find_function(glob_prop);
379     		if (!prop_fct)
380     		{	fprintf(stderr, "uno:global: error, could not load builtin property\n");
381     			exit(1);
382     		}
383     	}
384     	n = find_node(prop_fct, 0);
385     	if (n && n->succ)
386     		prop_start = n->succ->to;
387     }

395     		;

405     static int
406     n_reported(Stack *s)
407     {	Report *r;
408     	int n;
409     
410     	if (s && s->move && s->move->to)
411     		n = s->move->to->nid;
412     	else
413     		return 0;
414     
415     	for (r = report; r; r = r->nxt)
416     		if (r->n > n-3
417     		&&  r->n < n+3)	/* within a range of subsequent states */
418     			return 1;
419     	r = (Report *) emalloc(sizeof(Report));
420     	r->n = n;
421     	r->nxt = report;
422     	report = r;
423     
424     	if (debug) printf("NID=%d\n", n);
425     
426     	return 0;
427     }

449     print_stack(Stack *s, int d)
450     {	Arc *a;
451     	int i;
452     
453     	if (!s) return;
454     	print_stack(s->nxt, d-1);
455

457     
458     	if (a
459     	&&  a->to->lab
460     	&&  strlen((char *) a->to->lab->label) > 0)
461     	{	for (i = 0; i < tabs; i++)
462     			printf("   ");
463     		printf("%3d:\t[%d]\t%s\n", d,

468     	{	tabs++;
469     		for (i = 0; i < tabs; i++)
470     			printf("   ");

472     	}
473     	if (s->fr)
474     	{	for (i = 0; i < tabs; i++)
475     			printf("   ");
476     		tabs--;
477     		printf("%3d: Return to %s\n", d, s->fr->fnm);
478     	}

481     static void
482     do_fct_call(BFct *f, Node *sys)
483     {	Rstack *r, *or;
484     	BFct *g;
485     	BSym *s;
486     	Node *n;
487     
488     	f->visited = 1;	/* prevent fct call cycles */
489     
490     	if (free_rstack)
491     	{	r = free_rstack;
492     		free_rstack = free_rstack->nxt;
493     		memset(r, 0, sizeof(Rstack));
494     	} else
495     		r = (Rstack *) emalloc(sizeof(Rstack));
496     
497     	r->f = f;
498     	r->n = sys;	/* continuation point for return from fct in end_of_path */
499     	r->nxt = rstack;
500     	rstack = r;
501     
502     	stack->fc = f;	/* remember for error trace */
503     
504     	n = find_node(f, 0);
505     	if (n && n->succ)
506     	{	if (debug) printf("%3d: dive in\n", depth);
507     		check_prop(n->succ);
508     		if (debug) printf("%3d: undive\n", depth);
509     	} else
510     	{	if (debug) printf("%3d: no visible operations (calls: %p)\n", depth, f->calls);
511     
512     		/* if we're here, then probably f->calls is also nil, or else
513     		 * there would be an abstract call graph for this fct with the
514     		 * call points recorded...
515     		 */
516     
517     		if (!f->calls)	/* expected */
518     		{	end_of_path();	/* continuation with sys */
519     		} else
520     		{	fprintf(stderr, "%3d: unexpected...\n", depth);
521     			for (s = f->calls; s; s = s->nxt)
522     			{	g = find_function(s->s);
523     				if (g && !g->visited)
524     				{	if (debug) printf("	ByPass Call on %s\n", s->s);
525     					do_fct_call(g, (Node *) 0); /* no continuation within this fct */
526     					if (debug) printf("	Return from ByPass Call on %s\n", s->s);
527     	}	}	}	}
528     
529     	stack->fc = (BFct *) 0;
530     
531     	or = rstack;
532     	rstack = rstack->nxt;
533     	or->nxt = free_rstack;
534     	free_rstack = or;
535     
536     	f->visited = 0;
537     }

539     static int
540     same_markings(Vis *v)
541     {	Var   *p;
542     	VList *q;
543     	int cnt = 0;
544     
545     	if (!v->v && !stack->sels)
546     	{	if (v->zeromarks)
547     			return 1;
548     		v->zeromarks = 1;
549     		return 0;
550     	}
551     
552     	for (p = stack->sels; p; p = p->nxt)
553     	{	for (q = v->v; q; q = q->nxt)
554     		{	if (strcmp(q->v->s, p->s) == 0
555     			&&  q->v->mark == p->mark)
556     				break;
557     		}
558     		if (!q)	/* add p */
559     		{	if (free_v)
560     			{	q = free_v;
561     				free_v = q->nxt;
562     				memset(q, 0, sizeof(VList));
563     			} else
564     				q = (VList *) emalloc(sizeof(VList));
565     			
566     			q->v = new_var(p->s);
567     			q->v->mark = p->mark;
568     			q->nxt = v->v;
569     			v->v = q;
570     			cnt++;
571     	}	}
572     
573     	return (cnt == 0);
574     }

576     static int
577     same_callpts(Vis *v)
578     {	Rstack *p;
579     	NList *q;
580     	int cnt = 0;
581     
582     	if (!v->r && !rstack)
583     	{	if (v->zerostack)
584     			return 1;
585     		v->zerostack = 1;
586     		return 0;
587     	}
588     
589     	for (p = rstack; p; p = p->nxt)
590     	{	for (q = v->r; q; q = q->nxt)
591     		{	if (p->n == q->n)
592     				break;
593     		}
594     		if (!q)	/* add p->n */
595     		{	if (free_v)
596     			{	q = free_n;
597     				free_n = q->nxt;
598     				memset(q, 0, sizeof(NList));
599     			} else
600     				q = (NList *) emalloc(sizeof(NList));
601     			q->n = p->n;
602     			q->nxt = v->r;
603     			v->r = q;
604     			cnt++;
605     	}	}
606     
607     	return (cnt == 0);
608     }

610     static int
611     same_unostate(Vis *v)
612     {
613     	if (v->uno_state & (1<<stack->uno_state))
614     		return 1;
615     	v->uno_state |= (1<<stack->uno_state);
616     	return 0;
617     }

619     static int
620     same_state(Arc *in)
621     {	Stack	*s;
622     	Var	*e, *f;
623     	BSym	*x, *y;
624     	Node	*sys = in->to;
625     
626     	if (!sys->vis)
627     		sys->vis = (Vis *) emalloc(sizeof(Vis));
628     
629     	if (same_unostate(sys->vis)
630     	&&  same_callpts(sys->vis)
631     	&&  same_markings(sys->vis))
632     		return 1;			/* been here before with that state */
633     
634     	if (free_stack)
635     	{	s = free_stack;
636     		free_stack = s->nxt;
637     		memset(s, 0, sizeof(Stack));
638     	} else
639     		s = (Stack *) emalloc(sizeof(Stack));
640     	s->move = in;
641     	s->uno_state = stack->uno_state;
642     
643     	for (e = stack->sels; e; e = e->nxt)		/* copy marked vars forward */
644     	{	f = new_var(e->s);
645     		f->loc = e->loc;
646     		f->sel = 0;
647     		f->stat = e->stat;
648     		f->mark = e->mark;
649     		f->nxt = s->sels;
650     		s->sels = f;
651     	}
652     	for (x = stack->knz; x; x = x->nxt)
653     	{	y = new_sym(x->s);
654     		y->nxt = s->knz;
655     		s->knz = y;
656     	}
657     
658     	s->nxt = stack;
659     	stack = s;					/* stack trace for error reports */
660     
661     	return 0;
662     }

664     static void
665     end_of_path(void)
666     {	Arc	*a;
667     	Rstack	*r;
668     
669     	if (!rstack) return;
670     
671     	r = rstack;
672     	rstack = rstack->nxt;
673     
674     	stack->fr = rstack?rstack->f:(BFct *) 0;	/* remember for error traces */
675     
676     	if (debug)
677     	printf("%3d:\tFCT EXITS -- %s -- return to %s\n", depth,
678     		r->f->fnm, rstack?rstack->f->fnm:"--");
679     
680     	if (!r->n || !r->n->succ)		/* no continuation point */
681     	{	if (debug) printf("%3d: no continuation point\n", depth);
682     		end_of_path();			/* ? */
683     	} else
684     		for (a = r->n->succ; a; a = a->nxt)	/* continuation in caller */
685     		{	if (debug) printf("%3d:\tCONTinuation in %s [%d->%d]\n", depth,
686     				rstack?rstack->f->fnm:"--", r->n->nid, a->to->nid);
687     			check_prop(a);
688     			if (debug) printf("%3d:\tUN_CONT %s [%d->%d]\n", depth,
689     				rstack?rstack->f->fnm:"--", a->to->nid, r->n->nid);
690     		}
691     	r->nxt = rstack;
692     	rstack = r;
693     
694     	stack->fr = (BFct *) 0;
695     
696     	if (debug)
697     	printf("%3d:\tFCT UN_EXITS -- %s\n", depth, r->f->fnm);
698     }

709     static int
710     has_fct_match(Label *n, Label *m)
711     {	char *p, *q = (char *) 0, *r, *s = (char *) 0;
712     	char took = '(';
713     	char c = '\"';
714     	int t;
715     
716     	if (strncmp((char *) n->label, "C ", strlen("C ")) == 0
717     	||  strncmp((char *) n->label, "c ", strlen("c ")) == 0)
718     	{	/* either: fct_call("qlock()") or fct_call("qlock") */
719     		p = strchr((char *) m->label, c);	/* start of fct name */
720     
721     		if (p)
722     		{	q = strchr(p+1, '(');	/* has () */
723     		} else
724     		{	c = '\'';		/* single quotes? */
725     			p = strchr((char *) m->label, c);
726     			if (!p) return 0;	/* give up */
727     			q = strchr(p+1, '(');
728     		}
729     
730     		if (!q)
731     		{	q = strchr(p+1, c);	/* no (), find closing quote */
732     			took = c;
733     		}
734     	
735     		r = (char *) &(n->label[2]);
736     		if (r) s = strchr(r+1, ' ');
737     
738     		f2d_assert((int) (p && q && r && s && q > p+1), "bad fct_call() arg");
739     
740     		*q = *s = '\0';
741     		t = (strcmp(p+1, r) == 0);
742     		*q = took; *s = ' ';	/* restore */
743     		if (t) return 1;
744     	}
745     	return 0;
746     }

803     static int
804     has_sel_match(Label *n, int match, int dont)
805     {	char *r, *s;
806     	int typ = 0;
807     
808     	if (strlen((char *) n->label) <= 2
809     	||  n->label[1] != ' ')
810     		return 0;
811     
812     	switch (n->label[0]) {
813     	case 'G':
814     		if (!(match & (DECL|DEF)) || (dont & (DECL|DEF))) return 0;
815     		typ = DECL|DEF;
816     		break;
817     	case 'D':
818     		if (!(match & DEF) || (dont & DEF)) return 0;
819     		typ = DEF;
820     		break;
821     	case 'U':
822     		if (!(match & USE) || (dont & USE)) return 0;
823     		typ = USE;
824     		break;
825     	case 'R':
826     		if (!(match & DEREF) || (dont & DEREF)) return 0;
827     		typ = DEREF;
828     		break;
829     	default:
830     		return 0;
831     	}
832     
833     	r = (char *) &(n->label[2]);
834     	s = strchr(r+1, ' ');
835     
836     	if (!(r && s && s > r)) fprintf(stderr, "bad label: '%s' -- %p %p\n", n->label, r, s);
837     	f2d_assert((int) (r && s && s > r), "bad sel match");
838     	*s = '\0';
839     	set_select(r, s+1, typ); /* r = name, s+1 = location */
840     	*s = ' ';	/* restore */
841     
842     	return 1;
843     }

845     static int
846     has_ref_match(int mark, int match, int dont)
847     {	Var *v;
848     	int hasmatch = 0;
849     
850     	if (debug)
851     	for (v = stack->sels; v; v = v->nxt)
852     		printf("\tSel: '%s', stat %d, mark %d -- looking for mark %d, +%d -%d\n",
853     			v->s, v->stat, v->mark, mark, match, dont);
854     
855     	for (v = stack->sels; v; v = v->nxt)
856     		if (v->sel == 1
857     		&&  v->mark == mark
858     		&& (v->stat & match)
859     		&& !(v->stat & dont))
860     		{	hasmatch = 1;
861     			break;
862     		}
863     
864     	if (hasmatch)
865     	for (v = stack->sels; v; v = v->nxt)
866     		if (v->sel == 1
867     		&& (v->mark != mark
868     		|| !(v->stat & match)
869     		||  (v->stat & dont)))
870     			v->sel = 0;
871     
872     	return hasmatch;
873     }

913     take_action(Label *m)
914     {	char *p, *q;
915     	Label *lb;
916     	int x, y;

918     
919     	if (strncmp((char *) m->label, "error(", strlen("error(")) == 0)
920     	{	p = strchr((char *) m->label, c);
921     		if (p)
922     		{	q = strchr(p+1, c);
923     		} else
924     		{	c = '\'';
925     			p = strchr((char *) m->label, c);
926     			if (!p) return 0;
927     			q = strchr(p+1, c);
928     		}
929     		f2d_assert((int) (p && q && q > p+1), "bad error() arg");

973     	{	list_sel();
974     		if (debug) printf("	list\n");
975     		return 1;
976     	}
977     	if (strncmp((char *) m->label, "no_error()", strlen("no_error()")) == 0)
978     	{	p = strchr((char *) m->label, c);
979     		if (p)
980     		{	q = strchr(p+1, c);
981     			f2d_assert((int) (p && q && q > p+1), "bad no_error() arg");

993     			*q = c;

995     		if (debug) printf("	no_error\n");
996     		return 1;
997     	}
998     
999     	/* other types of stmnts, e.g., uno_state++, uno_state--, uno_state = N */
1000     	if (strncmp((char *) m->label, "uno_state--", strlen("uno_state--")) == 0)
1001     		stack->uno_state--;

1003     		stack->uno_state++;
1004     	else if (strncmp((char *) m->label, "uno_state=", strlen("uno_state=")) == 0
1005     	     && isdigit(m->label[strlen("uno_state=")]))
1006     		stack->uno_state = atoi((char *) &(m->label[strlen("uno_state=")]));

1008     		notyet(m, "expecting uno_state = N");
1009     
1010     	if (debug) printf("\taction '%s' (uno_state = %d)\n", m->label, stack->uno_state);
1011     	return 1;
1012     }

1015     act_cond(Label *m, int t)	/* match or non_zero -- conditionals indep of sys labels */
1016     {	int ret = 0;
1017     	int x, y, z;
1018     	int truth = t;
1019     
1020     	if (strncmp((char *) m->label, "!match(", strlen("!match(")) == 0)
1021     	{	truth = (truth == 1) ? -1 : 1;
1022     		x = strlen("!match(,");
1023     		goto n_ref;
1024     	}
1025     	if (strncmp((char *) m->label, "!known_nonzero()", strlen("!known_nonzero()")) == 0)
1026     	{	truth = (truth == 1) ? -1 : 1;
1027     		goto n_nz;
1028     	}
1029     
1030     	if (strncmp((char *) m->label, "match(", strlen("match(")) == 0)
1031     	{	x = strlen("match(");		/* match(mark, domatch, nomatch) */
1032     n_ref:		if (!isdigit(m->label[x])
1033     		||  sscanf((char *) &(m->label[x]), "%d,%d,%d", &z, &x, &y) != 3)
1034     			notyet(m, "args to match");
1035     
1036     		if ((x|y) & ~(DEF|USE|DEREF))	/* only D, U, or R marks matter */
1037     			notyet(m, "2nd or 3rd arg to match");
1038     
1039     		ret = has_ref_match(z, x, y);
1040     		if (truth == -1) ret = 1 - ret;
1041     
1042     		if (debug) printf("ac: %s -- returns %d\n", m->label, ret);
1043     	} else if (strncmp((char *) m->label, "known_nonzero()", strlen("known_nonzero()")) == 0)
1044     	{
1045     n_nz:		ret = known_nz();
1046     		if (truth == -1) ret = 1 - ret;
1047     
1048     		if (debug) printf("ac: %s -- returns %d\n", m->label, ret);
1049     	}
1050     
1051     	return ret;
1052     }

1054     static int
1055     act_select(Label *n, Label *m, int t)
1056     {	unsigned char *p;
1057     	int ret = 0;
1058     	int x, y, xx;
1059     	int truth = t;
1060     
1061     	/*
1062     	 *  n = sys  -- e.g. "C nm file line -- R x file line"
1063     	 *  m = prop -- e.g. "ftc_call('nm') -- select('x', DEREF, NONE)"
1064     	 */
1065     
1066     	if (strncmp((char *) m->label, "!fct_call(", strlen("!fct_call(")) == 0)
1067     	{	truth = (truth == 1) ? -1 : 1;
1068     		goto n_fct;
1069     	}
1070     	if (strncmp((char *) m->label, "!select(", strlen("!select(")) == 0)
1071     	{	truth = (truth == 1) ? -1 : 1;
1072     		xx = x = strlen("select('',");
1073     		goto n_sel;
1074     	}
1075     	if (strncmp((char *) m->label, "!(uno_state", strlen("!(uno_state")) == 0)
1076     	{	truth = (truth == 1) ? -1 : 1;
1077     		p = &(m->label[strlen("!(uno_state")]);
1078     		goto n_uno;
1079     	}
1080     
1081     	if (strncmp((char *) m->label, "fct_call(", strlen("fct_call(")) == 0)
1082     	{
1083     n_fct:		if (has_fct_match(n, m))
1084     		{	if (debug)
1085     			printf("\tfct match '%s' <-> '%s' (want %d)\n",
1086     				n->label, m->label, truth);
1087     
1088     			if (truth ==  1) return 1;
1089     			if (truth == -1) return 0;
1090     			return 1;	/* fct_call as stmnt, matching */
1091     		} else
1092     		{	if (truth ==  1) return 0;
1093     			if (truth == -1) return 1;
1094     			return 0;	/* fct_call as stmnt, not matching */
1095     	}	}
1096     
1097     	if (strncmp((char *) m->label, "select(", strlen("select(")) == 0)
1098     	{
1099     		/* 4 args:  name, domatch, nomatch, mark */
1100     		/* name must be '' for now */
1101     		xx = strlen("select('',");
1102     n_sel:
1103     		if (!strstr((char *) m->label, "('',")
1104     		||  !isdigit(m->label[xx]))
1105     			notyet(m, "args to select1");
1106     if (0) printf("<%s>\n", &(m->label[xx]));
1107     if (0) fflush(stdout);
1108     		if (sscanf((char *) &(m->label[xx]), "%d,%d", &x, &y) != 2)
1109     			notyet(m, "args to select2");
1110     
1111     		if ((x|y) & ~(DEF|USE|DEREF))	/* only D, U, or R marks matter */
1112     			notyet(m, "2nd or 3rd arg to select");
1113     
1114     		ret = has_sel_match(n, x, y);
1115     		if (truth == -1) ret = 1 - ret;
1116     
1117     		if (debug) printf("lm: %s <-> %s -- returns %d\n", n->label, m->label, ret);
1118     		return ret;
1119     	}
1120     
1121     	if (strncmp((char *) m->label, "(uno_state", strlen("(uno_state")) == 0)
1122     	{	p = &(m->label[strlen("(uno_state")]);
1123     n_uno:		switch (*p) {
1124     		case '!':
1125     			if (*(p+1) != '=' || !isdigit(*(p+2))) notyet(m, "operator");
1126     			ret = (stack->uno_state != atoi((char *) p+2));
1127     			break;
1128     		case '=':
1129     			if (*(p+1) != '=' || !isdigit(*(p+2))) notyet(m, "operator");
1130     			ret = (stack->uno_state == atoi((char *) p+2));
1131     			break;
1132     		case '>':
1133     			if (*(p+1) == '=')
1134     			{	if (!isdigit(*(p+2))) notyet(m, "operator");
1135     				ret = (stack->uno_state >= atoi((char *) p+2));
1136     			} else
1137     			{	if (!isdigit(*(p+1))) notyet(m, "operator");
1138     				ret = (stack->uno_state > atoi((char *) p+1));
1139     			}
1140     			break;
1141     		case '<':
1142     			if (*(p+1) == '=')
1143     			{	if (!isdigit(*(p+2))) notyet(m, "operator");
1144     				ret = (stack->uno_state <= atoi((char *) p+2));
1145     			} else
1146     			{	if (!isdigit(*(p+1))) notyet(m, "operator");
1147     				ret = (stack->uno_state < atoi((char *) p+1));
1148     			}
1149     			break;
1150     		default:
1151     			notyet(m, "operator");
1152     		}
1153     		if (truth == -1) ret = 1 - ret;
1154     
1155     		if (debug)
1156     		printf("\tcmd '%s' == %d\n", m->label, ret);
1157     		return ret;
1158     	}
1159     	notyet(m, "unrecnogized command");
1160     	return ret;
1161     }

1181     static int
1182     eval_prop(Node *sys, Node *prop)
1183     {	Arc	*b;
1184     	Label	*n, *m;
1185     	int	nzs = 0, truth = 0, x;
1186     
1187     	if (!prop) return 1;			/* reached end of prop */
1188     
1189     	for (n = sys->lab; n; n = n->nxt)	/* is this a knz inserted state? */
1190     		if (strncmp((char *) n->label, "N ", strlen("N ")) == 0)
1191     		{	addtostack((char *) &n->label[2]);
1192     			nzs = 1;
1193     		}
1194     	if (nzs) return 1;
1195     
1196     #if 0
1197     	the only possible labels, so far, are:
1198     	condnts:	select, fct_call	<- true/false arcs, dependent on sys
1199     	condnts:	match, known_nonzero	<- true/false arcs, independent of sys
1200     	actions:	error, no_error, mark	<- single label and independent of sys
1201     #endif
1202     
1203     	for (b = prop->succ; b; b = b->nxt)	/* all edges to succ states of prop */
1204     	for (m = b->lab; m; m = m->nxt)		/* all tags on those states */
1205     	{
1206     		if (debug) printf("	>proplabel '%s'\n", m->label);
1207     
1208     		if (strstr((char *) m->label, "||")
1209     		||  strstr((char *) m->label, "&&"))
1210     			notyet(m, "composite expr");
1211     
1212     		if (strstr((char *) m->label, "== _true") != NULL)
1213     			truth = 1;	/* true branch */
1214     		else if (strstr((char *) m->label, "== _false") != NULL)
1215     			truth = -1;	/* false branch */
1216     		else if (strstr((char *) m->label, " == ") != NULL)
1217     			notyet(m, "case switch");
1218     		else
1219     		{	/* action: error, no_error, mark */
1220     			if (!take_action(m))	/* independent of sys labels */
1221     				return 0;	/* error - do not proceed */
1222     			x = 1;			/* no_error, mark */
1223     			goto down;
1224     		}
1225     		if (strstr((char *) m->label, "match(")
1226     		||  strstr((char *) m->label, "known_nonzero("))
1227     			x = act_cond(m, truth);	/* eval independent of sys labs */
1228     		else
1229     		{	x = 0;
1230     			if (strstr((char *) m->label, "select("))
1231     				unselect();			/* new selection erases old */
1232     			for (n = sys->lab; n; n = n->nxt)	/* tags on current system state */
1233     				x += act_select(n, m, truth);	/* collect info from all matches */
1234     			/* unless 'notyet' is declared as an exit fct, uno
1235     			   will not be able to tell that 'truth' was initialized
1236     			 */
1237     		}
1238     down:		if (x)
1239     		{	if (debug) printf("	>down in prop\n");
1240     			eval_prop(sys, b->to);
1241     			if (debug) printf("	>up in prop\n");
1242     		}
1243     	}
1244     
1245     	if (debug) printf("uno:global: end of property\n");
1246     	return 1;	/* reached the end */
1247     }

1249     static void
1250     check_prop(Arc	*in)
1251     {	Stack	*os;
1252     	BFct	*f;
1253     	Label	*n;
1254     	Arc	*a;
1255     	Node	*sys;
1256     	int	hit;
1257     
1258     	if (!in || !in->to)
1259     	{	end_of_path();	/* find continuations in callstack */
1260     		return;
1261     	}
1262     
1263     	sys = in->to;
1264     
1265     	if (same_state(in))
1266     	{	if (debug)
1267     		printf("\tREVISIT nid %d state_bits %lu - callpt %d\n",
1268     			sys->nid,
1269     			sys->vis->uno_state,	/* bit mask */
1270     			(rstack && rstack->n)?rstack->n->nid:-1);
1271     		return;
1272     	}
1273     
1274     	depth++;
1275     
1276     	if (debug)
1277     	printf("%3d: [%d, %d] callpt %d -- %s\n", depth, sys?sys->nid:0,
1278     		stack->uno_state,
1279     		(rstack && rstack->n)?rstack->n->nid:-1,
1280     		(sys && sys->lab)?(char *) sys->lab->label:"--");
1281     
1282     	if (debug) printf("%3d: evalprop starts (%p)\n", depth, prop_start);
1283     
1284     	if (!eval_prop(sys, prop_start))		/* complains on errors */
1285     		goto done;
1286     
1287     	if (debug) printf("%3d: evalprop ended\n", depth);
1288     
1289     	hit = 0;
1290     	for (n = sys->lab; n; n = n->nxt)		/* check tags on current state */
1291     	{	f = fct_in_lab((char *) n->label);		/* to find all fct calls */
1292     		if (f && !f->visited)			/* effect not already in call chain */
1293     		{	if (debug) printf("%3d:\tCALL to %s [%d]\n", depth, f->fnm, sys->nid);
1294     			do_fct_call(f, sys);
1295     			if (debug) printf("%3d:\tUN_CALL %s [%d]\n", depth, f->fnm, sys->nid);
1296     			hit++;	/* continuation is explored on return from fct */
1297     	}	}
1298     	if (hit) goto done;
1299     
1300     	if (!sys->succ) end_of_path();
1301     
1302     	for (a = sys->succ; a; a = a->nxt)		/* explore all next states */
1303     	{	if (debug)
1304     		{ printf("%3d:\tDOWN (uno_state=%d)\n", depth, stack->uno_state); list_sel(); }
1305     		check_prop(a);	/* are explored */
1306     		if (debug)
1307     		{ printf("%3d:\tUP (uno_state=%d)\n", depth, stack->uno_state); list_sel(); }
1308     	}
1309     done:
1310     	depth--;
1311     	os = stack;
1312     	stack->sels = rev_release(stack->sels);
1313     	stack->knz  = rev_symrel(stack->knz);
1314     	stack = stack->nxt;
1315     	os->nxt = free_stack;
1316     	free_stack = os;
1317     }

1319     void
1320     run_check(void)
1321     {	BFct *f;
1322     
1323     	if (!fcts) return;
1324     
1325     	stack = &init;
1326     	memset(stack, 0, sizeof(Stack));
1327     	ini_prop();
1328     	add_glob_defs();
1329     
1330     	f = find_function("_global_");	/* start execution with global defs */
1331     	if (f)
1332     	{	if (debug) printf("	CHECK %s\n", f->fnm);
1333     		do_fct_call(f, (Node *) 0);
1334     	} else
1335     		printf("uno:global: internal error, no _global_\n");
1336     	if (verbose)
1337     	printf("uno:global: %d scenarios reported in property check\n", ErrCnt-1);
1338     }

1340     void
1341     handle_fct(char *p_in)
1342     {	static BFct *f;
1343     	static Arc *a;
1344     	int id;
1345     	char *t;
1346     	char p[512];
1347     
1348     	if (strlen(p_in) >= 512)
1349     	{	fprintf(stderr, "uno:global: very long fct name <%s>\n", p_in);
1350     		exit(1);
1351     	}
1352     	strcpy(p, p_in);	/* p_in could be a constant string */
1353     
1354     	t = strchr(p, '\r');
1355     	if (t) *t = '\0';
1356     	t = strchr(p, '\n');
1357     	if (t) *t = '\0';
1358     
1359     	if (0) { printf("'%s' (%d)\n", p, (int) strlen(p)); fflush(stdout); }
1360     
1361     	if (p)
1362     	switch (*p) {
1363     	case ':':	/* reminder of fct name - already created */
1364     		t = strchr(p+2, '\t');
1365     		if (!t) goto bad1;
1366     		*t = '\0';
1367     		f = find_function(p+2);
1368     		if (!f)
1369     		{
1370     bad1:			fprintf(stderr, "uno:global: fct reminder invalid '%s'\n", p);
1371     			exit(1);
1372     		}
1373     		*t = '\t';
1374     		oid = 0;
1375     		break;
1376     
1377     	case '{':
1378     		if (sscanf(p, "{%d}", &id) != 1)
1379     bad:			fprintf(stderr, "uno:global: bad fct delimiter: '%s'\n", p);
1380     		else
1381     			a = add_arc(f, oid, id);	/* uno can't tell that f was initialized */
1382     		break;
1383     
1384     	case '<':
1385     		if (sscanf(p, "<%d>", &id) != 1)
1386     			goto bad;
1387     		else
1388     			oid = id;
1389     		break;
1390     
1391     	case '>':
1392     		if (sscanf(p, ">%d>", &id) != 1)
1393     			goto bad;
1394     		else
1395     		{	a = add_arc(f, oid, id);
1396     			oid = id;
1397     		}
1398     		break;
1399     
1400     	case '[':
1401     		t = strrchr(p, ']');
1402     		if (!t) goto bad;
1403     		*t = '\0';
1404     		if (t > p+1)
1405     		{	if (!a)
1406     			{	fprintf(stderr, "uno:global: zero arc error\n");
1407     				exit(1);
1408     			}
1409     			add_label(f, a, p+1);
1410     		}
1411     		*t = ']';
1412     		break;
1413     	}
1414     }
