86          */
87         @Test
88         public void testEncodingOption() throws Exception {
89             String origEncoding = System.getProperty("file.encoding");
90     
91             // set the default encoding under Windows
92             System.setProperty("file.encoding", "Cp1252");
93     
94             runCPD("--minimum-tokens", "34", "--language", "java",
95                     "--files", "src/test/resources/net/sourceforge/pmd/cpd/clitest/",
96                     "--ignore-identifiers",
97                     "--format", "xml",
98             // request UTF-8 for CPD
99                     "--encoding", "UTF-8");
100             // reset default encoding
101             System.setProperty("file.encoding", origEncoding);
102     
103             String out = bufferStdout.toString("UTF-8");
104             Assert.assertTrue(out.startsWith("<?xml version=\"1.0\" encoding=\"UTF-8\"?>"));
105             Assert.assertTrue(Pattern.compile("System\\.out\\.println\\([ij] \\+ \"ä\"\\);").matcher(out).find());
106             Assert.assertEquals(4, Integer.parseInt(System.getProperty(CPDCommandLineInterface.STATUS_CODE_PROPERTY)));
107         }
