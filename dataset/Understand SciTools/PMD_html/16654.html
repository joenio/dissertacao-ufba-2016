15     
16         /**
17          * Count the size of all non-constructor methods.

22         }
23     
24         @Override
25         public Object visit(ExecutableCode node, Object data) {
26     	return super.visit(node, data);
27         }
28     
29         @Override
30         public Object[] getViolationParameters(DataPoint point) {

33         }
