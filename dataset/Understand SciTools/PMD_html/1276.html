850         /**
851          * will sort the AST element in list in alphabetical order and if tag name 
852          * is the same it will sort against o1.getBeginColumn() +""+ o1.getBeginLine().
853          * so first criteria is the name, then the second is the column +""+line string.
854          * @param elements
855          * @return
856          */
857         private List<ASTElement> sortNodesByName(Set<ASTElement> elements){
858         	List<ASTElement> list = new ArrayList<>();
859         	list.addAll(elements);
860         	Collections.sort(list, new Comparator<ASTElement>() {
861         		public int compare(ASTElement o1, ASTElement o2) {
862         			if (o1.getName() == null)
863         				return Integer.MIN_VALUE;
864         			if (o2.getName() == null)
865         				return Integer.MAX_VALUE;
866         			if (o1.getName().equals(o2.getName())){
867         				String o1Value = o1.getBeginColumn() +""+ o1.getBeginLine();
868         				String o2Value = o2.getBeginColumn() +""+ o2.getBeginLine();
869         				return o1Value.compareTo(o2Value);
870         			}
871         			return o1.getName().compareTo(o2.getName());
872         		}
873     		});
874         	return list;
875         }

877         /**
878          * will sort the AST node by the image name.
879          * @param elements
880          * @return
881          */
882         private <T extends Node> List<T> sortByImage(Set<T> elements){
883         	List<T> list = new ArrayList<>();
884         	list.addAll(elements);
885         	Collections.sort(list, new Comparator<Node>() {
886         		public int compare(Node o1, Node o2) {
887         			if (o1.getImage() == null)
888         				return Integer.MIN_VALUE;
889         			if (o2.getImage() == null)
890         				return Integer.MAX_VALUE;
891         			if (o1.getImage().equals(o2.getImage())){
892         				String o1Value = o1.getBeginColumn() +""+ o1.getBeginLine();
893         				String o2Value = o2.getBeginColumn() +""+ o2.getBeginLine();
894         				return o1Value.compareTo(o2Value);
895         			}
896         			return o1.getImage().compareTo(o2.getImage());
897         		}
898     		});
899         	return list;
