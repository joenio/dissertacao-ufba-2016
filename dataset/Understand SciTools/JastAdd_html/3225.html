74       public Object visit(ASTAspectMethodDeclaration node, Object data) {
75         // AspectMethodDeclaration = Modifiers() [ TypeArguments() ] ResultType() MethodDeclarator() [ Block() ]
76         for (int i = 0; i < node.jjtGetNumChildren(); ++i) {
77           if (node.jjtGetChild(i) instanceof ASTMethodDeclarator) {
78             return node.jjtGetChild(i).jjtAccept(this, data);
79           }
80         }
81         return "";
82       }
