26     package org.jastadd.util;
27     
28     /**
29      * @author Jesper Öqvist <jesper.oqvist@cs.lth.se>
30      */
31     public class StringLiterals {
32     	/**
33     	 * @param theString
34     	 * @return Escaped Java string literal
35     	 */
36     	public static String buildStringLiteral(String theString) {
37     		StringBuilder buf = new StringBuilder(theString.length());
38     		for (int i = 0; i < theString.length(); ++i) {
39     			char c = theString.charAt(i);
40     			if (c >= 0x20 && c <= 0x7E) {
41     				switch (c) {
42     				case '"':
43     					buf.append("\\\"");
44     					break;
45     				case '\\':
46     					buf.append("\\\\");
47     					break;
48     				default:
49     					buf.append(c);
50     				}
51     			} else if (c < 0x20) {
52     				buf.append(String.format("\\%03o", (int) c));
53     			} else {
54     				buf.append(String.format("\\u%04X", (int) c));
55     			}
56     		}
57     		return buf.toString();
58     	}
59     }
