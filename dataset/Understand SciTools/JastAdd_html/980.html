26     package org.jastadd.tinytemplate;
27     
28     import java.io.PrintStream;
29     import java.io.PrintWriter;
30     import java.util.HashMap;
31     import java.util.Map;
32     
33     /**
34      * A simple template context
35      * 
36      * @author Jesper Öqvist <jesper.oqvist@cs.lth.se>
37      */
38     public class SimpleContext extends TemplateContext {
39     	
40     	private final TemplateContext parentContext;
41     	
42     	private final Map<String, Object> variables = new HashMap<String, Object>();
43     
44     	private final Object contextObject;
45     
46     	/**
47     	 * Create a new simple context
48     	 * @param parent The parent context
49     	 * @param context The context object
50     	 */
51     	public SimpleContext(TemplateContext parent, Object context) {
52     		parentContext = parent;
53     		contextObject = context;
54     	}
55     
56     	@Override
57     	public Object evalVariable(String varName) {
58     		Object var = variables.get(varName);
59     		if (var != null) {
60     			return var;
61     		} else {
62     			return parentContext.evalVariable(varName);
63     		}
64     	}
65     
66     	@Override
67     	public Object evalAttribute(String attribute) {
68     		return TinyTemplate.evalAttribute(attribute, contextObject);
69     	}
70     
71     	@Override
72     	public String evalIndentation(int levels) {
73     		return parentContext.evalIndentation(levels);
74     	}
75     
76     	@Override
77     	public void expand(TemplateContext tc, String templateName, PrintStream out) {
78     		parentContext.expand(tc, templateName, out);
79     	}
80     
81     	@Override
82     	public void expand(TemplateContext tc, String templateName, PrintWriter out) {
83     		parentContext.expand(tc, templateName, out);
84     	}
85     
86     	@Override
87     	public void expand(TemplateContext tc, String templateName, StringBuffer out) {
88     		parentContext.expand(tc, templateName, out);
89     	}
90     
91     	@Override
92     	public void expand(TemplateContext tc, String templateName, StringBuilder out) {
93     		parentContext.expand(tc, templateName, out);
94     	}
95     
96     	@Override
97     	public void flushVariables() {
98     		variables.clear();
99     	}
100     	
101     	@Override
102     	public void bind(String varName, Object value) {
103     		variables.put(varName, value);
104     	}
105     
106     	@Override
107     	public void bind(String varName, boolean value) {
108     		variables.put(varName, "" + value);
109     	}
110     
111     }
