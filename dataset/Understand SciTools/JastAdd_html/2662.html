26     package org.jastadd.util;
27     
28     import java.io.PrintStream;
29     import java.util.ArrayList;
30     import java.util.Stack;
31     
32     /**
33      * @author Jesper Öqvist <jesper.oqvist@cs.lth.se>
34      */
35     public class PrettyPrinter {
36     	private final String indentation;
37     	private final java.util.List<String> ind = new ArrayList<String>(32);
38     	{
39     		ind.add("");
40     	}
41     	private final Stack<Integer> indentStack = new Stack<Integer>();
42     	{
43     		indentStack.push(0);
44     	}
45     	private int currentIndent = 0;
46     
47     	private PrintStream out = System.out;
48     	private boolean newline = false;
49     
50     	/**
51     	 * Construct a new PrettyPrinter with the given indentation
52     	 * @param ind
53     	 */
54     	public PrettyPrinter(String ind) {
55     		this.indentation = ind;
56     	}
57     
58     	/**
59     	 * Set target print stream
60     	 * @param target
61     	 */
62     	public void setTarget(PrintStream target) {
63     		out = target;
64     	}
65     
66     	/**
67      	 * @param level The level of indentation
68      	 * @return The indentation string for the given indentation level
69      	 */
70     	public String getIndentation(int level) {
71     		while (ind.size() < (level+1)) {
72     			ind.add(ind.get(ind.size()-1) + indentation);
73     		}
74     		return ind.get(level);
75     	}
76     
77     	/**
78     	 * @param str
79     	 */
80     	public void print(String str) {
81     		indentNewline();
82     		out.print(str);
83     	}
84     
85     	/**
86     	 * Print newline
87     	 */
88     	public void println() {
89     		out.println();
90     		newline = true;
91     	}
92     
93     	/**
94     	 * Pretty-print a node
95     	 * @param node
96     	 */
97     	public void print(PrettyPrintable node) {
98     		pushIndentation();
99     		node.prettyPrint(this);
100     		popIndentation();
101     	}
102     
103     	/**
104     	 * Print indentation
105     	 * @param level
106     	 */
107     	public void indent(int level) {
108     		currentIndent = level;
109     	}
110     
111     	private void pushIndentation() {
112     		indentStack.push(currentIndent + indentStack.peek());
113     		currentIndent = 0;
114     	}
115     
116     	private void popIndentation() {
117     		currentIndent = indentStack.pop();
118     		currentIndent -= indentStack.peek();
119     	}
120     
121     	private void indentNewline() {
122     		if (newline) {
123     			out.print(getIndentation(indentStack.peek()));
124     			newline = false;
125     		}
126     	}
127     }
