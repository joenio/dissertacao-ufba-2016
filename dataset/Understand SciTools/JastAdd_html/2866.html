26     package org.jastadd.tinytemplate.fragment;
27     
28     import java.io.PrintStream;
29     
30     import org.jastadd.tinytemplate.Indentation;
31     import org.jastadd.tinytemplate.TemplateContext;
32     
33     /**
34      * Represents the system-dependent newline character sequence.
35      *
36      * This is a singleton class.
37      *
38      * @author Jesper Öqvist <jesper@llbit.se>
39      */
40     public class NewlineFragment extends AbstractFragment {
41     
42     	private static final String SYS_NL = System.getProperty("line.separator");
43     
44     	/**
45     	 * Singleton instance
46     	 */
47     	public static Fragment INSTANCE = new NewlineFragment();
48     
49     	private NewlineFragment() {}
50     
51     	@Override
52     	public void expand(TemplateContext context, StringBuilder out) {
53     		out.append(SYS_NL);
54     	}
55     
56     	@Override
57     	public String toString() {
58     		return SYS_NL;
59     	}
60     
61     	@Override
62     	public boolean isNewline() {
63     		return true;
64     	}
65     
66     	@Override
67     	public boolean isWhitespace() {
68     		return true;
69     	}
70     
71     	@Override
72     	public void printAspectCode(Indentation ind, int lvl, PrintStream out) {
73     		out.println(ind.get(lvl) + "out.println();");
74     	}
75     }
