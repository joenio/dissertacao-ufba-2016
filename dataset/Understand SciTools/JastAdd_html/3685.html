3     class Util {
4         public static String addUnicodeEscapes(String str) {
5             String retval = "";
6             char ch;
7             for (int i = 0; i < str.length(); i++) {
8                 ch = str.charAt(i);
9                 if ((ch < 0x20 || ch > 0x7e) &&
10                     ch != '\t' && ch != '\n' && ch != '\r' && ch != '\f') {
11                     String s = "0000" + Integer.toString(ch, 16);
12                     retval += "\\u" + s.substring(s.length() - 4, s.length());
13                 } else {
14                     retval += ch;
15                 }
16             }
17             return retval;
18          }
