26     package org.jastadd.tinytemplate.fragment;
27     
28     import java.io.PrintStream;
29     
30     import org.jastadd.tinytemplate.Indentation;
31     import org.jastadd.tinytemplate.TemplateContext;
32     
33     /**
34      * @author Jesper Öqvist <jesper@llbit.se>
35      */
36     public class Include extends NestedIndentationFragment {
37     
38     	private final String template;
39     
40     	/**
41     	 * @param templateName
42     	 */
43     	public Include(String templateName) {
44     		this.template = templateName;
45     	}
46     
47     	@Override
48     	public void expand(TemplateContext context, StringBuilder out) {
49     		expandWithIndentation(context.expand(template), context, out);
50     	}
51     
52     	@Override
53     	public boolean isExpansion() {
54     		return true;
55     	}
56     
57     	@Override
58     	public void printAspectCode(Indentation ind, int lvl, PrintStream out) {
59     		out.println(ind.get(lvl) + template + "(out);");
60     	}
61     }
