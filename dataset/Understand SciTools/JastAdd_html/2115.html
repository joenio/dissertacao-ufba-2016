26     package org.jastadd.tinytemplate.test;
27     
28     import static org.junit.Assert.assertEquals;
29     
30     import org.jastadd.tinytemplate.SimpleContext;
31     import org.jastadd.tinytemplate.TemplateContext;
32     import org.jastadd.tinytemplate.TemplateParser.SyntaxError;
33     import org.jastadd.tinytemplate.TinyTemplate;
34     import org.junit.Test;
35     
36     /**
37      * Tests for template include directives
38      * @author Jesper Öqvist <jesper.oqvist@cs.lth.se>
39      */
40     public class TestInclude {
41     
42     	private static final String NL = System.getProperty("line.separator");
43     
44     	/**
45     	 * Constructor
46     	 */
47     	public TestInclude() {
48     		TinyTemplate.printWarnings(false);
49     		TinyTemplate.throwExceptions(false);
50     	}
51     
52     	/**
53     	 * Tests expanding an included template
54     	 * @throws SyntaxError
55     	 */
56     	@Test
57     	public void testInclude_1() throws SyntaxError {
58     		TinyTemplate tt = new TinyTemplate(
59     				"a = [[boop]]\n" +
60     				"b = [[$include(a)]]");
61     
62     		assertEquals("boop", tt.expand("b"));
63     	}
64     
65     	/**
66     	 * Tests expanding a variable in an included template
67     	 * @throws SyntaxError
68     	 */
69     	@Test
70     	public void testInclude_2() throws SyntaxError {
71     		TinyTemplate tt = new TinyTemplate(
72     				"a = [[$boop]]\n" +
73     				"b = [[$include(a)]]");
74     		TemplateContext tc = new SimpleContext(tt, this);
75     		tc.bind("boop", "beep");
76     		assertEquals("beep", tc.expand("b"));
77     	}
78     
79     	/**
80     	 * Test the include keyword with a hash instead of dollar sign
81     	 * @throws SyntaxError
82     	 */
83     	@Test
84     	public void testInclude_3() throws SyntaxError {
85     		TinyTemplate tt = new TinyTemplate(
86     				"a = [[$boop]]\n" +
87     				"b = [[#include(a)]]");
88     		TemplateContext tc = new SimpleContext(tt, this);
89     		tc.bind("boop", "beep");
90     		assertEquals("beep", tc.expand("b"));
91     	}
92     
93     	/**
94     	 * Tests indentation in an including template
95     	 * @throws SyntaxError
96     	 */
97     	@Test
98     	public void testIndentation_1() throws SyntaxError {
99     		TinyTemplate tt = new TinyTemplate(
100     				"a = [[a\nb]]\n" +
101     				"b = [[  #include(a)]]");
102     		TemplateContext tc = new SimpleContext(tt, this);
103     		assertEquals(
104     				"  a" + NL +
105     				"  b",
106     				tc.expand("b"));
107     	}
108     }
