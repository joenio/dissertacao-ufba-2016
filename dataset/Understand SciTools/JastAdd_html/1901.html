26     package org.jastadd.tinytemplate;
27     
28     import java.io.BufferedOutputStream;
29     import java.io.OutputStream;
30     import java.io.PrintStream;
31     import java.io.PrintWriter;
32     
33     /**
34      * Template context is needed to expand a template. The template context
35      * is responsible for evaluating variables and attributes.
36      * @author Jesper Öqvist <jesper.oqvist@cs.lth.se>
37      */
38     public abstract class TemplateContext {
39     	/**
40      	 * Lookup variable on the variable stack and return the variable expansion
41      	 * if it was found.
42     	 * @param varName
43     	 * @return The variable value, or the string "&lt;unbound variable varName&gt;"
44     	 * if the variable was not bound
45     	 */
46     	abstract public Object evalVariable(String varName);
47     
48     	/**
49     	 * Evaluate an attribute
50     	 * @param attrName
51     	 * @return The string value returned from the attribute
52     	 */
53     	abstract public Object evalAttribute(String attrName);
54     
55     	/**
56      	 * @param level Number of indentation levels
57      	 * @return The cumulative indentation corresponding to the given
58      	 * indentation level
59      	 */
60     	abstract public String evalIndentation(int level);
61     
62     	/**
63     	 * Expand a template
64     	 * @param templateName
65     	 * @return The template expansion
66     	 */
67     	final public String expand(String templateName) {
68     		StringBuilder buf = new StringBuilder();
69     		expand(this, templateName, buf);
70     		return buf.toString();
71     	}
72     
73     	/**
74     	 * Expand a template
75     	 * @param templateName
76     	 * @param out
77     	 */
78     	final public void expand(String templateName, OutputStream out) {
79     		PrintStream ps = new PrintStream(new BufferedOutputStream(out));
80     		expand(templateName, ps);
81     		ps.close();
82     	}
83     
84     	/**
85     	 * Expand a template
86     	 * @param templateName
87     	 * @param out
88     	 */
89     	final public void expand(String templateName, PrintStream out) {
90     		expand(this, templateName, out);
91     	}
92     
93     	/**
94     	 * Expand a template
95     	 * @param templateName
96     	 * @param out
97     	 */
98     	final public void expand(String templateName, PrintWriter out) {
99     		expand(this, templateName, out);
100     	}
101     
102     	/**
103     	 * Expand a template
104     	 * @param templateName
105     	 * @param buf
106     	 */
107     	final public void expand(String templateName, StringBuffer buf) {
108     		expand(this, templateName, buf);
109     	}
110     
111     	/**
112     	 * Expand a template
113     	 * @param tc Context to expand the template in
114     	 * @param templateName
115     	 * @param out
116     	 */
117     	abstract public void expand(TemplateContext tc, String templateName, PrintStream out);
118     
119     	/**
120     	 * Expand a template
121     	 * @param tc Context to expand the template in
122     	 * @param templateName
123     	 * @param out
124     	 */
125     	abstract public void expand(TemplateContext tc, String templateName, PrintWriter out);
126     
127     	/**
128     	 * Expand a template
129     	 * @param tc Context to expand the template in
130     	 * @param templateName
131     	 * @param out
132     	 */
133     	abstract public void expand(TemplateContext tc, String templateName, StringBuffer out);
134     
135     	/**
136     	 * Expand a template
137     	 * @param tc Context to expand the template in
138     	 * @param templateName
139     	 * @param out
140     	 */
141     	abstract public void expand(TemplateContext tc, String templateName, StringBuilder out);
142     
143     	/**
144     	 * Bind a string value to a variable
145     	 * @param varName Variable to bind
146     	 * @param value Value to bind
147     	 */
148     	abstract public void bind(String varName, Object value);
149     
150     	/**
151     	 * Bind a boolean value to a variable
152     	 * @param varName Variable to bind
153     	 * @param value Value to bind
154     	 */
155     	abstract public void bind(String varName, boolean value);
156     
157     	/**
158     	 * Bind a template expansion to a variable.
159     	 * Synonymous to <code>bind(varName, expand(templateName))</code>.
160     	 * @param varName Variable to bind
161     	 * @param templateName The template to expand
162     	 */
163     	final public void bindExpansion(String varName, String templateName) {
164     		bind(varName, expand(templateName));
165     	}
166     
167     	/**
168      	 * Unbinds all currently bound variables.
169      	 */
170     	abstract public void flushVariables();
171     }
