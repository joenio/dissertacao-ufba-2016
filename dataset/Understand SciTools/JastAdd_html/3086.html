3     /**
4        Represents an association between a class body of a jrag file
5        and the name of the jrag file where it was found.
6        The file name is used to generate meaningful error messages.
7     */
8     
9     import org.jastadd.jrag.AST.SimpleNode;
10     import java.io.File;
11     
12     public class ClassBodyObject {
13       public String fileName;
14       public int line;
15       public SimpleNode node;
16       public String refinesAspect;
17       public String replaceAspect;
18       public String comments;
19       public String modifiers = "";
20       public String aspectName;
21     
22       public ClassBodyObject(SimpleNode node, String fileName, int line, String comments,
23           String aspectName) {
24         this.fileName = fileName;
25         this.line = line;
26         this.node = node;
27         this.comments = comments;
28         this.aspectName = aspectName;
29       }
30     
31       public ClassBodyObject(SimpleNode node, String fileName, int line, String aspectName) {
32         this(node, fileName, line, Unparser.unparseComment(node), aspectName);
33       }
34     
35       public String signature() {
36         String signature = (String) node.jjtAccept(new SignatureUnparser(), null);
37         String s = signature.replace('.', '_');
38         s = s.replace('<', '_');
39         s = s.replace('>', '_');
40         return s;
41       }
42     
43       public String getFileName() {
44         return fileName;
45       }
46     
47       public int getStartLine() {
48         return line;
49       }
50     
51       public String getAspectName() {
52         return aspectName;
53       }
54     
55       public String legacyAspectName() {
56         String name = fileName;
57         if (name.endsWith(".jrag")) {
58           name = name.substring(0, name.length() - 5);
59         } else if (name.endsWith(".jadd")) {
60           name = name.substring(0, name.length() - 5);
61         } else if (name.endsWith(".ast")) {
62           name = name.substring(0, name.length() - 4);
63         }
64         String pattern = File.separator.equals("\\") ? pattern = "\\\\" : File.separator;
65         String[] names = name.split(pattern);
66         return names[names.length-1];
67       }
68     
69       public String aspectName() {
70         return getAspectName();
71       }
72     
73     	public String modifiers() {
74     		if (modifiers.isEmpty()) {
75     			return "";
76     		} else {
77     			return modifiers + " ";
78         }
79     	}
