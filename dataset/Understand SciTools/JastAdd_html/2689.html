26     package org.jastadd.io;
27     import java.io.*;
28     
29     /**
30      * @author Jesper Öqvist <jesper@llbit.se>
31      */
32     public class LookaheadReader extends FilterReader {
33     
34     	/**
35     	 * Maximum read buffer size
36     	 */
37     	private static final int BUFF_SIZE = 1024;
38     
39     	/**
40     	 * The lookahead
41     	 */
42     	private final int lookahead;
43     
44     	private final char[] buffer = new char[BUFF_SIZE];
45     
46     	/**
47     	 * Current position in the read buffer
48     	 */
49     	private int pos = 0;
50     
51     	/**
52     	 * Number of valid characters in the read buffer
53     	 */
54     	private int length = 0;
55     
56     	/**
57     	 * Create a new lookahead reader
58     	 * @param in
59     	 * @param nlook
60     	 */
61     	public LookaheadReader(Reader in, int nlook) {
62     		super(in);
63     		lookahead = nlook;
64     		if (lookahead >= BUFF_SIZE) {
65     			throw new IllegalArgumentException("Too large lookahead");
66     		}
67     	}
68     
69     	/**
70     	 * Create a new lookahead reader
71     	 * @param in
72     	 * @param nlook
73     	 */
74     	public LookaheadReader(InputStream in, int nlook) {
75     		this(new InputStreamReader(in), nlook);
76     	}
77     
78     	/**
79     	 * Skip some input.
80     	 * @param num Number of characters to skip forward
81     	 */
82     	public void consume(int num) {
83     		pos += num;
84     	}
85     
86     	@Override
87     	public long skip(long num) {
88     		pos += num;
89     		return num;
90     	}
91     
92     	/**
93     	 * @return The next character, or -1 if the next character
94     	 * is past the end of the input stream
95     	 * @throws IOException
96     	 */
97     	public int peek() throws IOException {
98     		refill();
99     		if (pos < length)
100     			return buffer[pos];
101     		else
102     			return -1;
103     	}
104     
105     	/**
106     	 * Look ahead in the input stream.
107     	 * @param index Number of characters to look ahead
108     	 * @return The character at the given position, or -1 if the given
109     	 * position is outside the read buffer
110     	 * @throws IOException
111     	 */
112     	public int peek(int index) throws IOException {
113     		refill();
114     		if ((pos+index) < length)
115     			return buffer[pos+index];
116     		else
117     			return -1;
118     	}
119     
120     	/**
121     	 * Pop next character in the stream.
122     	 * @return The next character, or -1 if the end has been reached
123     	 * @throws IOException
124     	 */
125     	public int pop() throws IOException {
126     		refill();
127     		if (pos < length)
128     			return buffer[pos++];
129     		else
130     			return -1;
131     	}
132     
133     	/**
134     	 * Refills the input buffer if it does not currently satisfy the required
135     	 * lookahead.
136     	 */
137     	private void refill() throws IOException {
138     		if (length - pos <= lookahead) {
139     			if (length-pos > 0) {
140     				System.arraycopy(buffer, pos, buffer, 0, length-pos);
141     				length = length-pos;
142     				pos = 0;
143     			} else {
144     				length = 0;
145     				pos = 0;
146     			}
147     			int i = super.read(buffer, length, BUFF_SIZE-length);
148     			length += (i != -1) ? i : 0;
149     		}
150     	}
151     
152     	@Override
153     	public int read() throws java.io.IOException {
154     		return pop();
155     	}
156     
157     	@Override
158     	public int read(char cbuf[], int off, int len) throws IOException {
159     		if (!ready())
160     			return -1;
161     		len += off;
162     
163     		for (int i=off; i<len; i++) {
164     			int c = read();
165     			if (c < 0) return i-off;
166     			else cbuf[i] = (char) c;
167     		}
168     		return len-off;
169     	}
170     
171     	@Override
172     	public boolean ready() throws IOException {
173     		return pos < length || super.ready();
174     	}
175     
176     }
