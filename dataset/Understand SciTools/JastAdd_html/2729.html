70       /**
71        * Builds an indentation string equal to a certain level of
72        * indentation.
73        *
74        * @param level the required indentation level
75        * @return the indentation string
76        */
77       public final String ind(int level) {
78         while (indList.size() <= level) {
79           if (indList.size() == 0) {
80             indList.add("");
81           } else {
82             indList.add(indList.get(indList.size()-1) + indent);
83           }
84         }
85         return indList.get(level);
86       }

508       /**
509        * Initialize a grammar object with the current configuration.
510        * @return Root Grammar node for the generated AST
511        */
512       public Grammar buildRoot() {
513         Grammar root = new Grammar();
514         root.setConfiguration(this);
515     
516         indent = indent();
517         license = license();
518     
519         // Configuration object must be set before creating root template context!
520         TemplateContext tt = root.templateContext();
521     
522         // Configure global locking.
523         // TODO(joqvist): remove this.
524         tt.bind("SynchBegin", synchronizedBlockBegin(tt));
525         tt.bind("SynchEnd", synchronizedBlockEnd(tt));
526     
527         for (Option<?> option: allOptions()) {
528           option.bind(tt);
529         }
530     
531         // Bind global template variables:
532         String packageName = packageName();
533         if (packageName.isEmpty()) {
534           tt.bind("PackageDecl", "");
535         } else {
536           tt.bind("PackageDecl", "package " + packageName + ";");
537         }
538     
539         // Default attribute cache sets/maps.
540         tt.bind("DefaultMapType", typeDefaultMap());
541         tt.bind("DefaultSetType", typeDefaultSet());
542     
543         // Rewrite options.
544         tt.bind("RewriteEnabled", rewriteEnabled());
545         tt.bind("RewriteCircularNTA", rewriteCircularNTA());
546         tt.bind("LegacyRewrite", legacyRewrite());
547     
548         // Flush options.
549         tt.bind("FlushEnabled", flushEnabled());
550         tt.bind("FlushAttr", flushAttr());
551         tt.bind("FlushColl", flushColl());
552     
553         // Incremental options.
554         tt.bind("IncrementalEnabled", incremental());
555         tt.bind("IncrementalLevelParam", incrementalLevelParam());
556         tt.bind("IncrementalLevelAttr", incrementalLevelAttr());
557         tt.bind("IncrementalLevelNode", incrementalLevelNode());
558         tt.bind("IncrementalLevelRegion", incrementalLevelRegion());
559         tt.bind("IncrementalChangeFlush", incrementalChangeFlush());
560         tt.bind("IncrementalChangeMark", incrementalChangeMark());
561         tt.bind("IncrementalPropFull", incrementalPropFull());
562         tt.bind("IncrementalPropLimit", incrementalPropLimit());
563         tt.bind("IncrementalDebug", incrementalDebug());
564         tt.bind("IncrementalTrack", incrementalTrack());
565         tt.bind("DDGNodeName", astNodeType() + "$DepGraphNode");
566     
567         // Tracing options.
568         tt.bind("TracingEnabled", tracingEnabled());
569         tt.bind("TraceCompute", traceCompute());
570         tt.bind("TraceCache", traceCache());
571         tt.bind("TraceRewrite", traceRewrite());
572         tt.bind("TraceCircularNTA", traceCircularNTA());
573         tt.bind("TraceCircular", traceCircular());
574         tt.bind("TraceCopy", traceCopy());
575         tt.bind("TraceFlush", traceFlush());
576     
577         // Cache options.
578         tt.bind("CacheAnalyzeEnabled", cacheAnalyzeEnabled());
579     
580         // Set template variables to accommodate deprecated options
581         // (the deprecated options may alter the value of the template variable).
582         tt.bind("VisitCheckEnabled", visitCheckEnabled());
583         tt.bind("CacheCycle", cacheCycle());
584         tt.bind("StaticState", staticState());
585         tt.bind("LazyMaps", lazyMaps());
586     
587         return root;
588       }

590       /**
591        * @param out Where to print warning/error messages.
592        * @return {@code true} if configuration has fatal errors
593        */
594       public boolean checkProblems(PrintStream out) {
595     
596         if (unknownOptions) {
597           return true;
598         }
599     
600         if (jjtreeOption.value() && grammarOption.value().isEmpty()) {
601           out.println("Error: No grammar name given. A grammar name is required in JJTree-mode!");
602           return true;
603         }
604     
605         Collection<String> grammarFiles = new LinkedList<String>();
606         for (String filename: filenames) {
607           if (filename.endsWith(".ast")) {
608             grammarFiles.add(filename);
609           }
610         }
611     
612         if (grammarFiles.isEmpty()) {
613           out.println("Error: No grammar files specified.");
614           return true;
615         }
616     
617         for (String filename: filenames) {
618           if (!(filename.endsWith(".ast")
619               || filename.endsWith(".jrag")
620               || filename.endsWith(".jadd"))) {
621             out.format("Error: Unrecognized file extension: %s%n", filename);
622             return true;
623           }
624         }
625     
626         File outputDir = outputDir();
627         if (!outputDir.exists()) {
628           out.format("Error: Output directory %s does not exist%n", outputDir.getAbsolutePath());
629           return true;
630         }
631     
632         if (!outputDir.isDirectory()) {
633           out.format("Error: Output directory %s is not a directory%n", outputDir.getAbsolutePath());
634           return true;
635         }
636     
637         if (!outputDir.canWrite()) {
638           out.format("Error: Output directory %s is write protected%n", outputDir.getAbsolutePath());
639           return true;
640         }
641     
642         if (!checkIncrementalConfig(out)) {
643           return true;
644         }
645     
646         if (!checkCacheConfig(out)) {
647           return true;
648         }
649     
650         if (jjtreeOption.value() && !ASTNodeSuperOption.value().isEmpty()) {
651           out.println("Error: Cannot use --jjtree and --ASTNodeSuper at the same time!");
652           return true;
653         }
654     
655         if (beaverOption.value() && !ASTNodeSuperOption.value().isEmpty()) {
656           out.println("Error: Cannot use --beaver and --ASTNodeSuper at the same time!");
657           return true;
658         }
659     
660         if (beaverOption.value() && jjtreeOption.value()) {
661           out.println("Error: Cannot use --beaver and --jjtree at the same time!");
662           return true;
663         }
664     
665         return false;
666       }

681       /**
682        * Checks the incremental configuration for errors.
683        * @param out Error output stream
684        * @return true if no errors
685        */
686       private boolean checkIncrementalConfig(PrintStream out) {
687         // Check level: only one level at a time.
688         if (incrementalLevelAttr() && incrementalLevelNode()
689             || incrementalLevelAttr() && incrementalLevelParam()
690             || incrementalLevelNode() && incrementalLevelParam()
691             || incrementalLevelParam() && incrementalLevelRegion()
692             || incrementalLevelAttr() && incrementalLevelRegion()
693             || incrementalLevelNode() && incrementalLevelRegion()) {
694           out.println("error: Conflict in incremental evaluation configuration. "
695               + "Cannot combine \"param\", \"attr\", \"node\" and \"region\".");
696           return false;
697         }
698         // Check invalidate: only one strategy at a time.
699         if (incrementalChangeFlush() && incrementalChangeMark()) {
700           out.println("error: Conflict in incremental evaluation configuration. "
701               + "Cannot combine \"flush\" and \"mark\".");
702           return false;
703         }
704         // Check invalidate: currently not supporting mark strategy -- "mark".
705         if (incrementalChangeMark()) {
706           out.println("error: Unsupported incremental evaluation configuration: \"mark\".");
707           return false;
708         }
709         // Check propagation: only one strategy at a time.
710         if (incrementalPropFull() && incrementalPropLimit()) {
711           out.println("error: Conflict in incremental evaluation configuration. "
712               + "Cannot combine \"full\" and \"limit\".");
713           return false;
714         }
715         return true;
716       }

718       private static final String readFile(String name) throws IOException {
719         StringBuilder buf = new StringBuilder();
720         Reader reader = new BufferedReader(new FileReader(name));
721         char[] cbuf = new char[1024];
722         int i = 0;
723         while((i = reader.read(cbuf)) != -1) {
724           buf.append(String.valueOf(cbuf, 0, i));
725         }
726         reader.close();
727         return buf.toString();
728       }
