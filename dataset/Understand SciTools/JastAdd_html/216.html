18       public static void getImports(SimpleNode self, Set imports) {
19         if (self instanceof ASTImportDeclaration) {
20           Token t = new Token();
21           t.next = self.firstToken;
22           StringBuilder buf = new StringBuilder(64);
23           while (t != null && t != self.lastToken) {
24             t = t.next;
25             if (t.specialToken != null) {
26               buf.append(' ');
27             }
28             buf.append(Util.addUnicodeEscapes(t.image));
29           }
30           imports.add(buf.toString().trim());
31         }
32       }

34       public static void unparseComment(SimpleNode node, StringBuffer buf) {
35         Token tt = node.firstToken.specialToken;
36         if (tt != null) {
37           while (tt.specialToken != null) {
38             tt = tt.specialToken;
39           }
40           while (tt != null) {
41             buf.append(Util.addUnicodeEscapes(tt.image));
42             tt = tt.next;
43           }
44         }
45       }

68       /**
69        * Unparse a node to a string buffer
70        */
71       public static void unparseSimple(JragParserVisitor visitor, SimpleNode node, StringBuffer buf) {
72         Token t1 = node.firstToken;
73         Token t = new Token();
74         t.next = t1;
75     
76         SimpleNode n;
77         for (int i = 0; i < node.jjtGetNumChildren(); i++) {
78           n = (SimpleNode) node.jjtGetChild(i);
79           if (n != null) {
80             while (true) {
81               // Unparse linked tokens until the first token of the current child is found.
82               t = t.next;
83               if (t == n.firstToken) {
84                 break;
85               }
86               unparseToken(t, buf);
87             }
88             // Unparse the current child.
89             n.jjtAccept(visitor, buf);
90             t = n.lastToken;
91           }
92         }
93     
94         while (t != node.lastToken && t != null) {
95           t = t.next;
96           unparseToken(t, buf);
97         }
98       }

100       /**
101        * Unparse a token to a string buffer
102        */
103       public static void unparseToken(Token t, StringBuffer buf) {
104         if (t == null) {
105           return;
106         }
107         Token tt = t.specialToken;
108         if (tt != null) {
109           while (tt.specialToken != null) {
110             tt = tt.specialToken;
111           }
112           while (tt != null) {
113             buf.append(Util.addUnicodeEscapes(tt.image));
114             tt = tt.next;
115           }
116         }
117         if (t instanceof Token.GTToken) {
118           buf.append(">");
119         } else {
120           buf.append(t.image);
121         }
122       }

211         // ClassBody <- UnmodifiedClassDecl <- ClassDecl <- TypeDecl <- CompilationUnit
212         org.jastadd.jrag.AST.Node node = self;
213         for (int i = 0; node != null && !(node instanceof ASTCompilationUnit) && i < 8; i++) {
214           node = node.jjtGetParent();

217           // Remove optional "Class." before IdDecl in method declaration.

225               t2.specialToken = t;
226               t1.next = t2;
