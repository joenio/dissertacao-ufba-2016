27     package org.jastadd.tinytemplate.fragment;
28     
29     import java.io.PrintStream;
30     
31     import org.jastadd.tinytemplate.Indentation;
32     import org.jastadd.tinytemplate.TemplateContext;
33     import org.jastadd.tinytemplate.TemplateExpansionWarning;
34     import org.jastadd.tinytemplate.TemplateParser.SyntaxError;
35     
36     /**
37      * A concatenation statement
38      * @author Niklas Fors <niklas.fors@cs.lth.se>
39      */
40     public class Concat extends NestedIndentationFragment {
41     	private String iterable;
42     	private final String sep;
43     	private boolean isAttribute;
44     
45     	/**
46     	 * @param iterable
47     	 * @throws SyntaxError
48     	 */
49     	public Concat(String iterable) throws SyntaxError {
50     		this(iterable, null);
51     	}
52     
53     	/**
54     	 * @param iterable
55     	 * @param separator
56     	 * @throws SyntaxError
57     	 */
58     	public Concat(String iterable, String separator) throws SyntaxError {
59     		if (iterable.startsWith("#")) {
60     			this.iterable = iterable.substring(1);
61     			isAttribute = true;
62     		} else if (iterable.startsWith("$")) {
63     			this.iterable = iterable.substring(1);
64     			isAttribute = false;
65     		}
66     		if (separator == null) {
67     			// separator must be non-null!!!
68     			throw new NullPointerException();
69     		}
70     		this.sep = separator;
71     	}
72     
73     	@Override
74     	public void expand(TemplateContext context, StringBuilder out) {
75     		if (isAttribute) {
76     			expandAttribute(context, out);
77     		} else {
78     			throw new TemplateExpansionWarning("Variable " + iterable + " is not iterable");
79     		}
80     	}
81     
82     	private void expandAttribute(TemplateContext context, StringBuilder out) {
83     		Object value = context.evalAttribute(iterable);
84     		if (value instanceof Iterable) {
85     			Iterable<?> itr = (Iterable<?>) value;
86     			StringBuilder sb = new StringBuilder();
87     			boolean first = true;
88     			for (Object o: itr) {
89     				if (sep != null && !first) {
90     					sb.append(sep);
91     				}
92     				first = false;
93     				sb.append(String.valueOf(o));
94     			}
95     			expandWithIndentation(sb.toString(), context, out);
96     		} else {
97     			throw new TemplateExpansionWarning("Attribute " + iterable + " is not iterable");
98     		}
99     	}
100     
101     	@Override
102     	public boolean isConditional() {
103     		return false;
104     	}
105     
106     	@Override
107     	public boolean isExpansion() {
108     		return true;
109     	}
110     
111     	@Override
112     	public void printAspectCode(Indentation ind, int lvl, PrintStream out) {
113     		out.println(ind.get(lvl++) + "{");
114     		if (!sep.isEmpty()) {
115     			out.println(ind.get(lvl) + "boolean first = true;");
116     		}
117     		out.print(ind.get(lvl++) + "for (PrettyPrintable p: ");
118     		if (isAttribute) {
119     			out.print(iterable + "()");
120     		} else {
121     			out.print("get" + iterable + "()");
122     		}
123     		out.println(") {");
124     		if (!sep.isEmpty()) {
125     			out.println(ind.get(lvl++) + "if (!first) {");
126     			StringBuilder buf = new StringBuilder();
127     			for (int i = 0; i < sep.length(); ++i) {
128     				if ((i+1) < sep.length() && sep.charAt(i) == '\\' &&
129     						sep.charAt(i+1) == 'n') {
130     					i += 1;
131     					if (buf.length() > 0) {
132     						out.println(ind.get(lvl) + "out.print(\"" + buf.toString() + "\");");
133     					}
134     					out.println(ind.get(lvl) + "out.println();");
135     					buf.setLength(0);
136     				} else {
137     					buf.append(sep.charAt(i));
138     				}
139     			}
140     			if (buf.length() > 0) {
141     				out.println(ind.get(lvl) + "out.print(\"" + buf.toString() + "\");");
142     			}
143     			out.println(ind.get(--lvl) + "}");
144     			out.println(ind.get(lvl) + "first = false;");
145     		}
146     		out.println(ind.get(lvl) + "out.print(p);");
147     		out.println(ind.get(--lvl) + "}");
148     		out.println(ind.get(--lvl) + "}");
149     	}
150     }
