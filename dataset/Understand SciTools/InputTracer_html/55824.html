31     /* 
32        Overview
33     
34        Valgrind tries to emulate the kernel's threading as closely as
35        possible.  The client does all threading via the normal syscalls
36        (on Linux: clone, etc).  Valgrind emulates this by creating exactly
37        the same process structure as would be created without Valgrind.
38        There are no extra threads.
39     
40        The main difference is that Valgrind only allows one client thread
41        to run at once.  This is controlled with the CPU Big Lock,
42        "the_BigLock".  Any time a thread wants to run client code or
43        manipulate any shared state (which is anything other than its own
44        ThreadState entry), it must hold the_BigLock.
45     
46        When a thread is about to block in a blocking syscall, it releases
47        the_BigLock, and re-takes it when it becomes runnable again (either
48        because the syscall finished, or we took a signal).
49     
50        VG_(scheduler) therefore runs in each thread.  It returns only when
51        the thread is exiting, either because it exited itself, or it was
52        told to exit by another thread.
53     
54        This file is almost entirely OS-independent.  The details of how
55        the OS handles threading and signalling are abstracted away and
56        implemented elsewhere.  [Some of the functions have worked their
57        way back for the moment, until we do an OS port in earnest...]
58      */

245     /* 
246        Set a thread into a sleeping state, and give up exclusive access to
247        the CPU.  On return, the thread must be prepared to block until it
248        is ready to run again (generally this means blocking in a syscall,
249        but it may mean that we remain in a Runnable state and we're just
250        yielding the CPU to another thread).
251      */
252     void VG_(release_BigLock)(ThreadId tid, ThreadStatus sleepstate, HChar* who)
253     {
254        ThreadState *tst = VG_(get_ThreadState)(tid);
255     
256        vg_assert(tst->status == VgTs_Runnable);
257     
258        vg_assert(sleepstate == VgTs_WaitSys ||
259     	     sleepstate == VgTs_Yielding);
260     
261        tst->status = sleepstate;
262     
263        vg_assert(VG_(running_tid) == tid);
264        VG_(running_tid) = VG_INVALID_THREADID;
265     
266        if (VG_(clo_trace_sched)) {
267           Char buf[200];
268           vg_assert(VG_(strlen)(who) <= 200-100);
269           VG_(sprintf)(buf, "releasing lock (%s) -> %s",
270                             who, VG_(name_of_ThreadStatus)(sleepstate));
271           print_sched_event(tid, buf);
272        }
273     
274        /* Release the_BigLock; this will reschedule any runnable
275           thread. */
276        ML_(sema_up)(&the_BigLock, False/*not LL*/);
277     }

292     /* Clear out the ThreadState and release the semaphore. Leaves the
293        ThreadState in VgTs_Zombie state, so that it doesn't get
294        reallocated until the caller is really ready. */
295     void VG_(exit_thread)(ThreadId tid)
296     {
297        vg_assert(VG_(is_valid_tid)(tid));
298        vg_assert(VG_(is_running_thread)(tid));
299        vg_assert(VG_(is_exiting)(tid));
300     
301        mostly_clear_thread_record(tid);
302        VG_(running_tid) = VG_INVALID_THREADID;
303     
304        /* There should still be a valid exitreason for this thread */
305        vg_assert(VG_(threads)[tid].exitreason != VgSrc_None);
306     
307        if (VG_(clo_trace_sched))
308           print_sched_event(tid, "release lock in VG_(exit_thread)");
309     
310        ML_(sema_up)(&the_BigLock, False/*not LL*/);
311     }

433     static 
434     void mostly_clear_thread_record ( ThreadId tid )
435     {
436        vki_sigset_t savedmask;
437     
438        vg_assert(tid >= 0 && tid < VG_N_THREADS);
439        VG_(cleanup_thread)(&VG_(threads)[tid].arch);
440        VG_(threads)[tid].tid = tid;
441     
442        /* Leave the thread in Zombie, so that it doesn't get reallocated
443           until the caller is finally done with the thread stack. */
444        VG_(threads)[tid].status               = VgTs_Zombie;
445     
446        VG_(sigemptyset)(&VG_(threads)[tid].sig_mask);
447        VG_(sigemptyset)(&VG_(threads)[tid].tmp_sig_mask);
448     
449        os_state_clear(&VG_(threads)[tid]);
450     
451        /* start with no altstack */
452        VG_(threads)[tid].altstack.ss_sp = (void *)0xdeadbeef;
453        VG_(threads)[tid].altstack.ss_size = 0;
454        VG_(threads)[tid].altstack.ss_flags = VKI_SS_DISABLE;
455     
456        VG_(clear_out_queued_signals)(tid, &savedmask);
457     
458        VG_(threads)[tid].sched_jmpbuf_valid = False;
459     }

461     /*                                                                             
462        Called in the child after fork.  If the parent has multiple
463        threads, then we've inherited a VG_(threads) array describing them,
464        but only the thread which called fork() is actually alive in the
465        child.  This functions needs to clean up all those other thread
466        structures.
467                                                                                    
468        Whichever tid in the parent which called fork() becomes the                 
469        master_tid in the child.  That's because the only living slot in            
470        VG_(threads) in the child after fork is VG_(threads)[tid], and it           
471        would be too hard to try to re-number the thread and relocate the           
472        thread state down to VG_(threads)[1].                                       
473                                                                                    
474        This function also needs to reinitialize the_BigLock, since
475        otherwise we may end up sharing its state with the parent, which
476        would be deeply confusing.
477     */                                          
478     static void sched_fork_cleanup(ThreadId me)
479     {
480        ThreadId tid;
481        vg_assert(VG_(running_tid) == me);
482     
483     #  if defined(VGO_darwin)
484        // GrP fixme hack reset Mach ports
485        VG_(mach_init)();
486     #  endif
487     
488        VG_(threads)[me].os_state.lwpid = VG_(gettid)();
489        VG_(threads)[me].os_state.threadgroup = VG_(getpid)();
490     
491        /* clear out all the unused thread slots */
492        for (tid = 1; tid < VG_N_THREADS; tid++) {
493           if (tid != me) {
494              mostly_clear_thread_record(tid);
495     	 VG_(threads)[tid].status = VgTs_Empty;
496              VG_(clear_syscallInfo)(tid);
497           }
498        }
499     
500        /* re-init and take the sema */
501        ML_(sema_deinit)(&the_BigLock);
502        ML_(sema_init)(&the_BigLock);
503        ML_(sema_down)(&the_BigLock, False/*not LL*/);
504     }

507     /* First phase of initialisation of the scheduler.  Initialise the
508        bigLock, zeroise the VG_(threads) structure and decide on the
509        ThreadId of the root thread.
510     */
511     ThreadId VG_(scheduler_init_phase1) ( void )
512     {
513        Int i;
514        ThreadId tid_main;
515     
516        VG_(debugLog)(1,"sched","sched_init_phase1\n");
517     
518        ML_(sema_init)(&the_BigLock);
519     
520        for (i = 0 /* NB; not 1 */; i < VG_N_THREADS; i++) {
521           /* Paranoia .. completely zero it out. */
522           VG_(memset)( & VG_(threads)[i], 0, sizeof( VG_(threads)[i] ) );
523     
524           VG_(threads)[i].sig_queue = NULL;
525     
526           os_state_init(&VG_(threads)[i]);
527           mostly_clear_thread_record(i);
528     
529           VG_(threads)[i].status                    = VgTs_Empty;
530           VG_(threads)[i].client_stack_szB          = 0;
531           VG_(threads)[i].client_stack_highest_word = (Addr)NULL;
532        }
533     
534        tid_main = VG_(alloc_ThreadState)();
535     
536        /* Bleh.  Unfortunately there are various places in the system that
537           assume that the main thread has a ThreadId of 1.
538           - Helgrind (possibly)
539           - stack overflow message in default_action() in m_signals.c
540           - definitely a lot more places
541        */
542        vg_assert(tid_main == 1);
543     
544        return tid_main;
545     }

548     /* Second phase of initialisation of the scheduler.  Given the root
549        ThreadId computed by first phase of initialisation, fill in stack
550        details and acquire bigLock.  Initialise the scheduler.  This is
551        called at startup.  The caller subsequently initialises the guest
552        state components of this main thread.
553     */
554     void VG_(scheduler_init_phase2) ( ThreadId tid_main,
555                                       Addr     clstack_end, 
556                                       SizeT    clstack_size )
557     {
558        VG_(debugLog)(1,"sched","sched_init_phase2: tid_main=%d, "
559                        "cls_end=0x%lx, cls_sz=%ld\n",
560                        tid_main, clstack_end, clstack_size);
561     
562        vg_assert(VG_IS_PAGE_ALIGNED(clstack_end+1));
563        vg_assert(VG_IS_PAGE_ALIGNED(clstack_size));
564     
565        VG_(threads)[tid_main].client_stack_highest_word 
566           = clstack_end + 1 - sizeof(UWord);
567        VG_(threads)[tid_main].client_stack_szB 
568           = clstack_size;
569     
570        VG_(atfork)(NULL, NULL, sched_fork_cleanup);
571     }

597     /* Do various guest state alignment checks prior to running a thread.
598        Specifically, check that what we have matches Vex's guest state
599        layout requirements.  See libvex.h for details, but in short the
600        requirements are: There must be no holes in between the primary
601        guest state, its two copies, and the spill area.  In short, all 4
602        areas must have a 16-aligned size and be 16-aligned, and placed
603        back-to-back. */
604     static void do_pre_run_checks ( ThreadState* tst )
605     {
606        Addr a_vex     = (Addr) & tst->arch.vex;
607        Addr a_vexsh1  = (Addr) & tst->arch.vex_shadow1;
608        Addr a_vexsh2  = (Addr) & tst->arch.vex_shadow2;
609        Addr a_spill   = (Addr) & tst->arch.vex_spill;
610        UInt sz_vex    = (UInt) sizeof tst->arch.vex;
611        UInt sz_vexsh1 = (UInt) sizeof tst->arch.vex_shadow1;
612        UInt sz_vexsh2 = (UInt) sizeof tst->arch.vex_shadow2;
613        UInt sz_spill  = (UInt) sizeof tst->arch.vex_spill;
614     
615        if (0)
616        VG_(printf)("gst %p %d, sh1 %p %d, "
617                    "sh2 %p %d, spill %p %d\n",
618                    (void*)a_vex, sz_vex,
619                    (void*)a_vexsh1, sz_vexsh1,
620                    (void*)a_vexsh2, sz_vexsh2,
621                    (void*)a_spill, sz_spill );
622     
623        vg_assert(VG_IS_16_ALIGNED(sz_vex));
624        vg_assert(VG_IS_16_ALIGNED(sz_vexsh1));
625        vg_assert(VG_IS_16_ALIGNED(sz_vexsh2));
626        vg_assert(VG_IS_16_ALIGNED(sz_spill));
627     
628        vg_assert(VG_IS_16_ALIGNED(a_vex));
629        vg_assert(VG_IS_16_ALIGNED(a_vexsh1));
630        vg_assert(VG_IS_16_ALIGNED(a_vexsh2));
631        vg_assert(VG_IS_16_ALIGNED(a_spill));
632     
633        /* Check that the guest state and its two shadows have the same
634           size, and that there are no holes in between.  The latter is
635           important because Memcheck assumes that it can reliably access
636           the shadows by indexing off a pointer to the start of the
637           primary guest state area. */
638        vg_assert(sz_vex == sz_vexsh1);
639        vg_assert(sz_vex == sz_vexsh2);
640        vg_assert(a_vex + 1 * sz_vex == a_vexsh1);
641        vg_assert(a_vex + 2 * sz_vex == a_vexsh2);
642        /* Also check there's no hole between the second shadow area and
643           the spill area. */
644        vg_assert(sz_spill == LibVEX_N_SPILL_BYTES);
645        vg_assert(a_vex + 3 * sz_vex == a_spill);
646     
647     #  if defined(VGA_amd64)
648        /* x86/amd64 XMM regs must form an array, ie, have no
649           holes in between. */
650        vg_assert(
651           (offsetof(VexGuestAMD64State,guest_XMM16)
652            - offsetof(VexGuestAMD64State,guest_XMM0))
653           == (17/*#regs*/-1) * 16/*bytes per reg*/
654        );
655     #  endif
656     
657     #  if defined(VGA_ppc32) || defined(VGA_ppc64)
658        /* ppc guest_state vector regs must be 16 byte aligned for
659           loads/stores.  This is important! */
660        vg_assert(VG_IS_16_ALIGNED(& tst->arch.vex.guest_VR0));
661        vg_assert(VG_IS_16_ALIGNED(& tst->arch.vex_shadow1.guest_VR0));
662        vg_assert(VG_IS_16_ALIGNED(& tst->arch.vex_shadow2.guest_VR0));
663        /* be extra paranoid .. */
664        vg_assert(VG_IS_16_ALIGNED(& tst->arch.vex.guest_VR1));
665        vg_assert(VG_IS_16_ALIGNED(& tst->arch.vex_shadow1.guest_VR1));
666        vg_assert(VG_IS_16_ALIGNED(& tst->arch.vex_shadow2.guest_VR1));
667     #  endif
668     
669     #  if defined(VGA_arm)
670        /* arm guest_state VFP regs must be 8 byte aligned for
671           loads/stores. */
672        vg_assert(VG_IS_8_ALIGNED(& tst->arch.vex.guest_D0));
673        vg_assert(VG_IS_8_ALIGNED(& tst->arch.vex_shadow1.guest_D0));
674        vg_assert(VG_IS_8_ALIGNED(& tst->arch.vex_shadow2.guest_D0));
675        /* be extra paranoid .. */
676        vg_assert(VG_IS_8_ALIGNED(& tst->arch.vex.guest_D1));
677        vg_assert(VG_IS_8_ALIGNED(& tst->arch.vex_shadow1.guest_D1));
678        vg_assert(VG_IS_8_ALIGNED(& tst->arch.vex_shadow2.guest_D1));
679     #  endif
680     
681     #  if defined(VGA_s390x)
682        /* no special requirements */
683     #  endif
684     }

687     /* Run the thread tid for a while, and return a VG_TRC_* value
688        indicating why VG_(run_innerloop) stopped. */
689     static UInt run_thread_for_a_while ( ThreadId tid )
690     {
691        volatile Int          jumped;
692        volatile ThreadState* tst = NULL; /* stop gcc complaining */
693        volatile UInt         trc;
694        volatile Int          dispatch_ctr_SAVED;
695        volatile Int          done_this_time;
696     
697        /* Paranoia */
698        vg_assert(VG_(is_valid_tid)(tid));
699        vg_assert(VG_(is_running_thread)(tid));
700        vg_assert(!VG_(is_exiting)(tid));
701     
702        tst = VG_(get_ThreadState)(tid);
703        do_pre_run_checks( (ThreadState*)tst );
704        /* end Paranoia */
705     
706        trc = 0;
707        dispatch_ctr_SAVED = VG_(dispatch_ctr);
708     
709     #  if defined(VGP_ppc32_aix5) || defined(VGP_ppc64_aix5)
710        /* On AIX, we need to get a plausible value for SPRG3 for this
711           thread, since it's used I think as a thread-state pointer.  It
712           is presumably set by the kernel for each dispatched thread and
713           cannot be changed by user space.  It therefore seems safe enough
714           to copy the host's value of it into the guest state at the point
715           the thread is dispatched.
716           (Later): Hmm, looks like SPRG3 is only used in 32-bit mode.
717           Oh well. */
718        { UWord host_sprg3;
719          __asm__ __volatile__( "mfspr %0,259\n" : "=b"(host_sprg3) );
720         VG_(threads)[tid].arch.vex.guest_SPRG3_RO = host_sprg3;
721         vg_assert(sizeof(VG_(threads)[tid].arch.vex.guest_SPRG3_RO) == sizeof(void*));
722        }
723     #  endif
724     
725        /* there should be no undealt-with signals */
726        //vg_assert(VG_(threads)[tid].siginfo.si_signo == 0);
727     
728        if (0) {
729           vki_sigset_t m;
730           Int i, err = VG_(sigprocmask)(VKI_SIG_SETMASK, NULL, &m);
731           vg_assert(err == 0);
732           VG_(printf)("tid %d: entering code with unblocked signals: ", tid);
733           for (i = 1; i <= _VKI_NSIG; i++)
734              if (!VG_(sigismember)(&m, i))
735                 VG_(printf)("%d ", i);
736           VG_(printf)("\n");
737        }
738     
739        // Tell the tool this thread is about to run client code
740        VG_TRACK( start_client_code, tid, bbs_done );
741     
742        vg_assert(VG_(in_generated_code) == False);
743        VG_(in_generated_code) = True;
744     
745        SCHEDSETJMP(
746           tid, 
747           jumped, 
748           trc = (UInt)VG_(run_innerloop)( (void*)&tst->arch.vex,
749                                           VG_(clo_profile_flags) > 0 ? 1 : 0 )
750        );
751     
752        vg_assert(VG_(in_generated_code) == True);
753        VG_(in_generated_code) = False;
754     
755        if (jumped) {
756           /* We get here if the client took a fault that caused our signal
757              handler to longjmp. */
758           vg_assert(trc == 0);
759           trc = VG_TRC_FAULT_SIGNAL;
760           block_signals();
761        } 
762     
763        done_this_time = (Int)dispatch_ctr_SAVED - (Int)VG_(dispatch_ctr) - 0;
764     
765        vg_assert(done_this_time >= 0);
766        bbs_done += (ULong)done_this_time;
767     
768        // Tell the tool this thread has stopped running client code
769        VG_TRACK( stop_client_code, tid, bbs_done );
770     
771        return trc;
772     }

775     /* Run a no-redir translation just once, and return the resulting
776        VG_TRC_* value. */
777     static UInt run_noredir_translation ( Addr hcode, ThreadId tid )
778     {
779        volatile Int          jumped;
780        volatile ThreadState* tst; 
781        volatile UWord        argblock[4];
782        volatile UInt         retval;
783     
784        /* Paranoia */
785        vg_assert(VG_(is_valid_tid)(tid));
786        vg_assert(VG_(is_running_thread)(tid));
787        vg_assert(!VG_(is_exiting)(tid));
788     
789        tst = VG_(get_ThreadState)(tid);
790        do_pre_run_checks( (ThreadState*)tst );
791        /* end Paranoia */
792     
793     #  if defined(VGA_ppc32) || defined(VGA_ppc64)
794        /* I don't think we need to clear this thread's guest_RESVN here,
795           because we can only get here if run_thread_for_a_while() has
796           been used immediately before, on this same thread. */
797     #  endif
798     
799        /* There can be 3 outcomes from VG_(run_a_noredir_translation):
800     
801           - a signal occurred and the sighandler longjmp'd.  Then both [2]
802             and [3] are unchanged - hence zero.
803     
804           - translation ran normally, set [2] (next guest IP) and set [3]
805             to whatever [1] was beforehand, indicating a normal (boring)
806             jump to the next block.
807     
808           - translation ran normally, set [2] (next guest IP) and set [3]
809             to something different from [1] beforehand, which indicates a
810             TRC_ value.
811        */
812        argblock[0] = (UWord)hcode;
813        argblock[1] = (UWord)&VG_(threads)[tid].arch.vex;
814        argblock[2] = 0; /* next guest IP is written here */
815        argblock[3] = 0; /* guest state ptr afterwards is written here */
816     
817        // Tell the tool this thread is about to run client code
818        VG_TRACK( start_client_code, tid, bbs_done );
819     
820        vg_assert(VG_(in_generated_code) == False);
821        VG_(in_generated_code) = True;
822     
823        SCHEDSETJMP(
824           tid, 
825           jumped, 
826           VG_(run_a_noredir_translation)( &argblock[0] )
827        );
828     
829        VG_(in_generated_code) = False;
830     
831        if (jumped) {
832           /* We get here if the client took a fault that caused our signal
833              handler to longjmp. */
834           vg_assert(argblock[2] == 0); /* next guest IP was not written */
835           vg_assert(argblock[3] == 0); /* trc was not written */
836           block_signals();
837           retval = VG_TRC_FAULT_SIGNAL;
838        } else {
839           /* store away the guest program counter */
840           VG_(set_IP)( tid, argblock[2] );
841           if (argblock[3] == argblock[1])
842              /* the guest state pointer afterwards was unchanged */
843              retval = VG_TRC_BORING;
844           else
845              retval = (UInt)argblock[3];
846        }
847     
848        bbs_done++;
849     
850        // Tell the tool this thread has stopped running client code
851        VG_TRACK( stop_client_code, tid, bbs_done );
852     
853        return retval;
854     }

887     {
888        ThreadState * volatile tst = VG_(get_ThreadState)(tid);
889        Bool jumped; 
890     
891        /* Syscall may or may not block; either way, it will be

908        
909        if (jumped) {
910           block_signals();
911           VG_(poll_signals)(tid);
912        }

951     /* 
952        Run a thread until it wants to exit.
953        
954        We assume that the caller has already called VG_(acquire_BigLock) for
955        us, so we own the VCPU.  Also, all signals are blocked.
956      */
957     VgSchedReturnCode VG_(scheduler) ( ThreadId tid )
958     {
959        UInt     trc;
960        ThreadState *tst = VG_(get_ThreadState)(tid);
961     
962        if (VG_(clo_trace_sched))
963           print_sched_event(tid, "entering VG_(scheduler)");      
964     
965        /* set the proper running signal mask */
966        block_signals();
967        
968        vg_assert(VG_(is_running_thread)(tid));
969     
970        VG_(dispatch_ctr) = SCHEDULING_QUANTUM + 1;
971     
972        while (!VG_(is_exiting)(tid)) {
973     
974           if (VG_(dispatch_ctr) == 1) {
975     
976     #        if defined(VGP_ppc32_aix5) || defined(VGP_ppc64_aix5)
977              /* Note: count runnable threads before dropping The Lock. */
978              Int rt = VG_(count_runnable_threads)();
979     #        endif
980     
981     	 /* Our slice is done, so yield the CPU to another thread.  On
982                 Linux, this doesn't sleep between sleeping and running,
983                 since that would take too much time.  On AIX, we have to
984                 prod the scheduler to get it consider other threads; not
985                 doing so appears to cause very long delays before other
986                 runnable threads get rescheduled. */
987     
988     	 /* 4 July 06: it seems that a zero-length nsleep is needed to
989                 cause async thread cancellation (canceller.c) to terminate
990                 in finite time; else it is in some kind of race/starvation
991                 situation and completion is arbitrarily delayed (although
992                 this is not a deadlock).
993     
994                 Unfortunately these sleeps cause MPI jobs not to terminate
995                 sometimes (some kind of livelock).  So sleeping once
996                 every N opportunities appears to work. */
997     
998     	 /* 3 Aug 06: doing sys__nsleep works but crashes some apps.
999                 sys_yield also helps the problem, whilst not crashing apps. */
1000     
1001     	 VG_(release_BigLock)(tid, VgTs_Yielding, 
1002                                        "VG_(scheduler):timeslice");
1003     	 /* ------------ now we don't have The Lock ------------ */
1004     
1005     #        if defined(VGP_ppc32_aix5) || defined(VGP_ppc64_aix5)
1006              { static Int ctr=0;
1007                vg_assert(__NR_AIX5__nsleep != __NR_AIX5_UNKNOWN);
1008                vg_assert(__NR_AIX5_yield   != __NR_AIX5_UNKNOWN);
1009                if (1 && rt > 0 && ((++ctr % 3) == 0)) { 
1010                   //struct vki_timespec ts;
1011                   //ts.tv_sec = 0;
1012                   //ts.tv_nsec = 0*1000*1000;
1013                   //VG_(do_syscall2)(__NR_AIX5__nsleep, (UWord)&ts, (UWord)NULL);
1014     	      VG_(do_syscall0)(__NR_AIX5_yield);
1015                }
1016              }
1017     #        endif
1018     
1019     	 VG_(acquire_BigLock)(tid, "VG_(scheduler):timeslice");
1020     	 /* ------------ now we do have The Lock ------------ */
1021     
1022     	 /* OK, do some relatively expensive housekeeping stuff */
1023     	 scheduler_sanity(tid);
1024     	 VG_(sanity_check_general)(False);
1025     
1026     	 /* Look for any pending signals for this thread, and set them up
1027     	    for delivery */
1028     	 VG_(poll_signals)(tid);
1029     
1030     	 if (VG_(is_exiting)(tid))
1031     	    break;		/* poll_signals picked up a fatal signal */
1032     
1033     	 /* For stats purposes only. */
1034     	 n_scheduling_events_MAJOR++;
1035     
1036     	 /* Figure out how many bbs to ask vg_run_innerloop to do.  Note
1037     	    that it decrements the counter before testing it for zero, so
1038     	    that if tst->dispatch_ctr is set to N you get at most N-1
1039     	    iterations.  Also this means that tst->dispatch_ctr must
1040     	    exceed zero before entering the innerloop.  Also also, the
1041     	    decrement is done before the bb is actually run, so you
1042     	    always get at least one decrement even if nothing happens. */
1043              VG_(dispatch_ctr) = SCHEDULING_QUANTUM + 1;
1044     
1045     	 /* paranoia ... */
1046     	 vg_assert(tst->tid == tid);
1047     	 vg_assert(tst->os_state.lwpid == VG_(gettid)());
1048           }
1049     
1050           /* For stats purposes only. */
1051           n_scheduling_events_MINOR++;
1052     
1053           if (0)
1054              VG_(message)(Vg_DebugMsg, "thread %d: running for %d bbs\n", 
1055                                        tid, VG_(dispatch_ctr) - 1 );
1056     
1057           trc = run_thread_for_a_while ( tid );
1058     
1059           if (VG_(clo_trace_sched) && VG_(clo_verbosity) > 2) {
1060     	 Char buf[50];
1061     	 VG_(sprintf)(buf, "TRC: %s", name_of_sched_event(trc));
1062     	 print_sched_event(tid, buf);
1063           }
1064     
1065           if (trc == VEX_TRC_JMP_NOREDIR) {
1066              /* If we got a request to run a no-redir version of
1067                 something, do so now -- handle_noredir_jump just (creates
1068                 and) runs that one translation.  The flip side is that the
1069                 noredir translation can't itself return another noredir
1070                 request -- that would be nonsensical.  It can, however,
1071                 return VG_TRC_BORING, which just means keep going as
1072                 normal. */
1073              trc = handle_noredir_jump(tid);
1074              vg_assert(trc != VEX_TRC_JMP_NOREDIR);
1075           }
1076     
1077           switch (trc) {
1078           case VG_TRC_BORING:
1079              /* no special event, just keep going. */
1080              break;
1081     
1082           case VG_TRC_INNER_FASTMISS:
1083     	 vg_assert(VG_(dispatch_ctr) > 1);
1084     	 handle_tt_miss(tid);
1085     	 break;
1086     	    
1087           case VEX_TRC_JMP_CLIENTREQ:
1088     	 do_client_request(tid);
1089     	 break;
1090     
1091           case VEX_TRC_JMP_SYS_INT128:  /* x86-linux */
1092           case VEX_TRC_JMP_SYS_INT129:  /* x86-darwin */
1093           case VEX_TRC_JMP_SYS_INT130:  /* x86-darwin */
1094           case VEX_TRC_JMP_SYS_SYSCALL: /* amd64-linux, ppc32-linux, amd64-darwin */
1095     	 handle_syscall(tid, trc);
1096     	 if (VG_(clo_sanity_level) > 2)
1097     	    VG_(sanity_check_general)(True); /* sanity-check every syscall */
1098     	 break;
1099     
1100           case VEX_TRC_JMP_YIELD:
1101     	 /* Explicit yield, because this thread is in a spin-lock
1102     	    or something.  Only let the thread run for a short while
1103                 longer.  Because swapping to another thread is expensive,
1104                 we're prepared to let this thread eat a little more CPU
1105                 before swapping to another.  That means that short term
1106                 spins waiting for hardware to poke memory won't cause a
1107                 thread swap. */
1108     	 if (VG_(dispatch_ctr) > 2000) 
1109                 VG_(dispatch_ctr) = 2000;
1110     	 break;
1111     
1112           case VG_TRC_INNER_COUNTERZERO:
1113     	 /* Timeslice is out.  Let a new thread be scheduled. */
1114     	 vg_assert(VG_(dispatch_ctr) == 1);
1115     	 break;
1116     
1117           case VG_TRC_FAULT_SIGNAL:
1118     	 /* Everything should be set up (either we're exiting, or
1119     	    about to start in a signal handler). */
1120     	 break;
1121     
1122           case VEX_TRC_JMP_MAPFAIL:
1123              /* Failure of arch-specific address translation (x86/amd64
1124                 segment override use) */
1125              /* jrs 2005 03 11: is this correct? */
1126              VG_(synth_fault)(tid);
1127              break;
1128     
1129           case VEX_TRC_JMP_EMWARN: {

1663     static
1664     void scheduler_sanity ( ThreadId tid )
1665     {
1666        Bool bad = False;
1667        static UInt lasttime = 0;
1668        UInt now;
1669        Int lwpid = VG_(gettid)();
1670     
1671        if (!VG_(is_running_thread)(tid)) {
1672           VG_(message)(Vg_DebugMsg,
1673     		   "Thread %d is supposed to be running, "
1674                        "but doesn't own the_BigLock (owned by %d)\n", 
1675     		   tid, VG_(running_tid));
1676           bad = True;
1677        }
1678     
1679        if (lwpid != VG_(threads)[tid].os_state.lwpid) {
1680           VG_(message)(Vg_DebugMsg,
1681                        "Thread %d supposed to be in LWP %d, but we're actually %d\n",
1682                        tid, VG_(threads)[tid].os_state.lwpid, VG_(gettid)());
1683           bad = True;
1684        }
1685     
1686     #if !defined(VGO_darwin)
1687        // GrP fixme
1688        if (lwpid != the_BigLock.owner_lwpid) {
1689           VG_(message)(Vg_DebugMsg,
1690                        "Thread (LWPID) %d doesn't own the_BigLock\n",
1691                        tid);
1692           bad = True;
1693        }
1694     #endif
1695     
1696        /* Periodically show the state of all threads, for debugging
1697           purposes. */
1698        now = VG_(read_millisecond_timer)();
1699        if (0 && (!bad) && (lasttime + 4000/*ms*/ <= now)) {
1700           lasttime = now;
1701           VG_(printf)("\n------------ Sched State at %d ms ------------\n",
1702                       (Int)now);
1703           VG_(show_sched_status)();
1704        }
1705     
1706        /* core_panic also shows the sched status, which is why we don't
1707           show it above if bad==True. */
1708        if (bad)
1709           VG_(core_panic)("scheduler_sanity: failed");
1710     }

1712     void VG_(sanity_check_general) ( Bool force_expensive )
1713     {
1714        ThreadId tid;
1715     
1716        static UInt next_slow_check_at = 1;
1717        static UInt slow_check_interval = 25;
1718     
1719        if (VG_(clo_sanity_level) < 1) return;
1720     
1721        /* --- First do all the tests that we can do quickly. ---*/
1722     
1723        sanity_fast_count++;
1724     
1725        /* Check stuff pertaining to the memory check system. */
1726     
1727        /* Check that nobody has spuriously claimed that the first or
1728           last 16 pages of memory have become accessible [...] */
1729        if (VG_(needs).sanity_checks) {
1730           vg_assert(VG_TDICT_CALL(tool_cheap_sanity_check));
1731        }
1732     
1733        /* --- Now some more expensive checks. ---*/
1734     
1735        /* Once every now and again, check some more expensive stuff.
1736           Gradually increase the interval between such checks so as not to
1737           burden long-running programs too much. */
1738        if ( force_expensive
1739             || VG_(clo_sanity_level) > 1
1740             || (VG_(clo_sanity_level) == 1 
1741                 && sanity_fast_count == next_slow_check_at)) {
1742     
1743           if (0) VG_(printf)("SLOW at %d\n", sanity_fast_count-1);
1744     
1745           next_slow_check_at = sanity_fast_count - 1 + slow_check_interval;
1746           slow_check_interval++;
1747           sanity_slow_count++;
1748     
1749           if (VG_(needs).sanity_checks) {
1750               vg_assert(VG_TDICT_CALL(tool_expensive_sanity_check));
1751           }
1752     
1753           /* Look for stack overruns.  Visit all threads. */
1754           for (tid = 1; tid < VG_N_THREADS; tid++) {
1755     	 SizeT    remains;
1756              VgStack* stack;
1757     
1758     	 if (VG_(threads)[tid].status == VgTs_Empty ||
1759     	     VG_(threads)[tid].status == VgTs_Zombie)
1760     	    continue;
1761     
1762              stack 
1763                 = (VgStack*)
1764                   VG_(get_ThreadState)(tid)->os_state.valgrind_stack_base;
1765              SizeT limit
1766                 = 4096; // Let's say.  Checking more causes lots of L2 misses.
1767     	 remains 
1768                 = VG_(am_get_VgStack_unused_szB)(stack, limit);
1769     	 if (remains < limit)
1770     	    VG_(message)(Vg_DebugMsg, 
1771                              "WARNING: Thread %d is within %ld bytes "
1772                              "of running out of stack!\n",
1773     		         tid, remains);
1774           }
1775        }
1776     
1777        if (VG_(clo_sanity_level) > 1) {
1778           /* Check sanity of the low-level memory manager.  Note that bugs
1779              in the client's code can cause this to fail, so we don't do
1780              this check unless specially asked for.  And because it's
1781              potentially very expensive. */
1782           VG_(sanity_check_malloc_all)();
1783        }
1784     }
