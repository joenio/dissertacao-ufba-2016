47     void ppIRType ( IRType ty )
48     {
49        switch (ty) {
50           case Ity_INVALID: vex_printf("Ity_INVALID"); break;
51           case Ity_I1:      vex_printf( "I1");   break;
52           case Ity_I8:      vex_printf( "I8");   break;
53           case Ity_I16:     vex_printf( "I16");  break;
54           case Ity_I32:     vex_printf( "I32");  break;
55           case Ity_I64:     vex_printf( "I64");  break;
56           case Ity_I128:    vex_printf( "I128"); break;
57           case Ity_F32:     vex_printf( "F32");  break;
58           case Ity_F64:     vex_printf( "F64");  break;
59           case Ity_F128:    vex_printf( "F128"); break;
60           case Ity_V128:    vex_printf( "V128"); break;
61           default: vex_printf("ty = 0x%x\n", (Int)ty);
62                    vpanic("ppIRType");
63        }
64     }

66     void ppIRConst ( IRConst* con )
67     {
68        union { ULong i64; Double f64; UInt i32; Float f32; } u;
69        vassert(sizeof(ULong) == sizeof(Double));
70        switch (con->tag) {
71           case Ico_U1:   vex_printf( "%d:I1",        con->Ico.U1 ? 1 : 0); break;
72           case Ico_U8:   vex_printf( "0x%x:I8",      (UInt)(con->Ico.U8)); break;
73           case Ico_U16:  vex_printf( "0x%x:I16",     (UInt)(con->Ico.U16)); break;
74           case Ico_U32:  vex_printf( "0x%x:I32",     (UInt)(con->Ico.U32)); break;
75           case Ico_U64:  vex_printf( "0x%llx:I64",   (ULong)(con->Ico.U64)); break;
76           case Ico_F32:  u.f32 = con->Ico.F32;
77                          vex_printf( "F32{0x%x}",   u.i32);
78                          break;
79           case Ico_F32i: vex_printf( "F32i{0x%x}",   con->Ico.F32i); break;
80           case Ico_F64:  u.f64 = con->Ico.F64;
81                          vex_printf( "F64{0x%llx}",  u.i64);
82                          break;
83           case Ico_F64i: vex_printf( "F64i{0x%llx}", con->Ico.F64i); break;
84           case Ico_V128: vex_printf( "V128{0x%04x}", (UInt)(con->Ico.V128)); break;
85           default: vpanic("ppIRConst");
86        }
87     }

89     void ppIRCallee ( IRCallee* ce )
90     {
91        vex_printf("%s", ce->name);
92        if (ce->regparms > 0)
93           vex_printf("[rp=%d]", ce->regparms);
94        if (ce->mcx_mask > 0)
95           vex_printf("[mcx=0x%x]", ce->mcx_mask);
96        vex_printf("{%p}", (void*)ce->addr);
97     }

1138     void ppIRStmt ( IRStmt* s )
1139     {
1140        if (!s) {
1141           vex_printf("!!! IRStmt* which is NULL !!!");
1142           return;
1143        }
1144        switch (s->tag) {
1145           case Ist_NoOp:
1146              vex_printf("IR-NoOp");
1147              break;
1148           case Ist_IMark:
1149              vex_printf( "------ IMark(0x%llx, %d) ------", 
1150                          s->Ist.IMark.addr, s->Ist.IMark.len);
1151              break;
1152           case Ist_AbiHint:
1153              vex_printf("====== AbiHint(");
1154              ppIRExpr(s->Ist.AbiHint.base);
1155              vex_printf(", %d, ", s->Ist.AbiHint.len);
1156              ppIRExpr(s->Ist.AbiHint.nia);
1157              vex_printf(") ======");
1158              break;
1159           case Ist_Put:
1160              vex_printf( "PUT(%d) = ", s->Ist.Put.offset);
1161              ppIRExpr(s->Ist.Put.data);
1162              break;
1163           case Ist_PutI:
1164              vex_printf( "PUTI" );
1165              ppIRRegArray(s->Ist.PutI.descr);
1166              vex_printf("[");
1167              ppIRExpr(s->Ist.PutI.ix);
1168              vex_printf(",%d] = ", s->Ist.PutI.bias);
1169              ppIRExpr(s->Ist.PutI.data);
1170              break;
1171           case Ist_WrTmp:
1172              ppIRTemp(s->Ist.WrTmp.tmp);
1173              vex_printf( " = " );
1174              ppIRExpr(s->Ist.WrTmp.data);
1175              break;
1176           case Ist_Store:
1177              vex_printf( "ST%s(", s->Ist.Store.end==Iend_LE ? "le" : "be" );
1178              ppIRExpr(s->Ist.Store.addr);
1179              vex_printf( ") = ");
1180              ppIRExpr(s->Ist.Store.data);
1181              break;
1182           case Ist_CAS:
1183              ppIRCAS(s->Ist.CAS.details);
1184              break;
1185           case Ist_LLSC:
1186              if (s->Ist.LLSC.storedata == NULL) {
1187                 ppIRTemp(s->Ist.LLSC.result);
1188                 vex_printf(" = LD%s-Linked(",
1189                            s->Ist.LLSC.end==Iend_LE ? "le" : "be");
1190                 ppIRExpr(s->Ist.LLSC.addr);
1191                 vex_printf(")");
1192              } else {
1193                 ppIRTemp(s->Ist.LLSC.result);
1194                 vex_printf(" = ( ST%s-Cond(",
1195                            s->Ist.LLSC.end==Iend_LE ? "le" : "be");
1196                 ppIRExpr(s->Ist.LLSC.addr);
1197                 vex_printf(") = ");
1198                 ppIRExpr(s->Ist.LLSC.storedata);
1199                 vex_printf(" )");
1200              }
1201              break;
1202           case Ist_Dirty:
1203              ppIRDirty(s->Ist.Dirty.details);
1204              break;
1205           case Ist_MBE:
1206              vex_printf("IR-");
1207              ppIRMBusEvent(s->Ist.MBE.event);
1208              break;
1209           case Ist_Exit:
1210              vex_printf( "if (" );
1211              ppIRExpr(s->Ist.Exit.guard);
1212              vex_printf( ") goto {");
1213              ppIRJumpKind(s->Ist.Exit.jk);
1214              vex_printf("} ");
1215              ppIRConst(s->Ist.Exit.dst);
1216              break;
1217           default: 
1218              vpanic("ppIRStmt");
1219        }
1220     }

2705     /*--- Sanity checking -- FLATNESS                             ---*/
2706     /*---------------------------------------------------------------*/
2707     
2708     /* Check that the canonical flatness constraints hold on an
2709        IRStmt. The only place where any expression is allowed to be
2710        non-atomic is the RHS of IRStmt_Tmp. */
2711     
2712     /* Relies on:
2713        inline static Bool isAtom ( IRExpr* e ) {
2714           return e->tag == Iex_RdTmp || e->tag == Iex_Const;
2715        }
2716     */
2717     
2718     Bool isFlatIRStmt ( IRStmt* st )
2719     {
2720        Int      i;
2721        IRExpr*  e;
2722        IRDirty* di;
2723        IRCAS*   cas;
2724     
2725        switch (st->tag) {
2726           case Ist_AbiHint:
2727              return isIRAtom(st->Ist.AbiHint.base)
2728                     && isIRAtom(st->Ist.AbiHint.nia);
2729           case Ist_Put:
2730              return isIRAtom(st->Ist.Put.data);
2731           case Ist_PutI:
2732              return toBool( isIRAtom(st->Ist.PutI.ix) 
2733                             && isIRAtom(st->Ist.PutI.data) );
2734           case Ist_WrTmp:
2735              /* This is the only interesting case.  The RHS can be any
2736                 expression, *but* all its subexpressions *must* be
2737                 atoms. */
2738              e = st->Ist.WrTmp.data;
2739              switch (e->tag) {
2740                 case Iex_Binder: return True;
2741                 case Iex_Get:    return True;
2742                 case Iex_GetI:   return isIRAtom(e->Iex.GetI.ix);
2743                 case Iex_RdTmp:  return True;
2744                 case Iex_Qop:    return toBool(
2745                                         isIRAtom(e->Iex.Qop.arg1) 
2746                                         && isIRAtom(e->Iex.Qop.arg2)
2747                                         && isIRAtom(e->Iex.Qop.arg3)
2748                                         && isIRAtom(e->Iex.Qop.arg4));
2749                 case Iex_Triop:  return toBool(
2750                                         isIRAtom(e->Iex.Triop.arg1) 
2751                                         && isIRAtom(e->Iex.Triop.arg2)
2752                                         && isIRAtom(e->Iex.Triop.arg3));
2753                 case Iex_Binop:  return toBool(
2754                                         isIRAtom(e->Iex.Binop.arg1) 
2755                                         && isIRAtom(e->Iex.Binop.arg2));
2756                 case Iex_Unop:   return isIRAtom(e->Iex.Unop.arg);
2757                 case Iex_Load:   return isIRAtom(e->Iex.Load.addr);
2758                 case Iex_Const:  return True;
2759                 case Iex_CCall:  for (i = 0; e->Iex.CCall.args[i]; i++)
2760                                     if (!isIRAtom(e->Iex.CCall.args[i])) 
2761                                        return False;
2762                                  return True;
2763                 case Iex_Mux0X:  return toBool (
2764                                         isIRAtom(e->Iex.Mux0X.cond) 
2765                                         && isIRAtom(e->Iex.Mux0X.expr0) 
2766                                         && isIRAtom(e->Iex.Mux0X.exprX));
2767                 default:         vpanic("isFlatIRStmt(e)");
2768              }
2769              /*notreached*/
2770              vassert(0);
2771           case Ist_Store:
2772              return toBool( isIRAtom(st->Ist.Store.addr) 
2773                             && isIRAtom(st->Ist.Store.data) );
2774           case Ist_CAS:
2775              cas = st->Ist.CAS.details;
2776              return toBool( isIRAtom(cas->addr)
2777                             && (cas->expdHi ? isIRAtom(cas->expdHi) : True)
2778                             && isIRAtom(cas->expdLo)
2779                             && (cas->dataHi ? isIRAtom(cas->dataHi) : True)
2780                             && isIRAtom(cas->dataLo) );
2781           case Ist_LLSC:
2782              return toBool( isIRAtom(st->Ist.LLSC.addr)
2783                             && (st->Ist.LLSC.storedata
2784                                    ? isIRAtom(st->Ist.LLSC.storedata) : True) );
2785           case Ist_Dirty:
2786              di = st->Ist.Dirty.details;
2787              if (!isIRAtom(di->guard)) 
2788                 return False;
2789              for (i = 0; di->args[i]; i++)
2790                 if (!isIRAtom(di->args[i])) 
2791                    return False;
2792              if (di->mAddr && !isIRAtom(di->mAddr)) 
2793                 return False;
2794              return True;
2795           case Ist_NoOp:
2796           case Ist_IMark:
2797           case Ist_MBE:
2798              return True;
2799           case Ist_Exit:
2800              return isIRAtom(st->Ist.Exit.guard);
2801           default: 
2802              vpanic("isFlatIRStmt(st)");
2803        }
2804     }

3241                 sanityCheckFail(bb,stmt,"IRStmt.PutI.descr: invalid descr");
3242              break;

3244              tcExpr( bb, stmt, stmt->Ist.WrTmp.data, gWordTy );
3245              if (typeOfIRTemp(tyenv, stmt->Ist.WrTmp.tmp)
3246                  != typeOfIRExpr(tyenv, stmt->Ist.WrTmp.data))
3247                 sanityCheckFail(bb,stmt,"IRStmt.Put.Tmp: tmp and expr do not match");

3253                 sanityCheckFail(bb,stmt,"IRStmt.Store.addr: not :: guest word type");

3261              /* make sure it's definitely either a CAS or a DCAS */
3262              if (cas->oldHi == IRTemp_INVALID

3269                 /* fine; it's a double cas */

3356                 sanityCheckFail(bb,stmt,"IRStmt.Dirty.guard not :: Ity_I1");

3371              break;
3372           case Ist_MBE:

3374                 case Imbe_Fence:
3375                    break;
3376                 default: sanityCheckFail(bb,stmt,"IRStmt.MBE.event: unknown");
3377                    break;
3378              }
3379              break;

3381              tcExpr( bb, stmt, stmt->Ist.Exit.guard, gWordTy );
3382              if (typeOfIRExpr(tyenv,stmt->Ist.Exit.guard) != Ity_I1)
3383                 sanityCheckFail(bb,stmt,"IRStmt.Exit.guard: not :: Ity_I1");

3385                 sanityCheckFail(bb,stmt,"IRStmt.Exit.dst: bad dst");
3386              if (typeOfIRConst(stmt->Ist.Exit.dst) != gWordTy)
3387                 sanityCheckFail(bb,stmt,"IRStmt.Exit.dst: not :: guest word type");

3428                 sanityCheckFail(bb, stmt, "IRStmt: is NULL");
3429              if (!isFlatIRStmt(stmt))
3430                 sanityCheckFail(bb, stmt, "IRStmt: is not flat");
3431           }
3432           if (!isIRAtom(bb->next))
3433              sanityCheckFail(bb, NULL, "bb->next is not an atom");
3434        }
3435     
3436        /* Count the defs of each temp.  Only one def is allowed.
3437           Also, check that each used temp has already been defd. */
3438     
3439        for (i = 0; i < n_temps; i++)
3440           def_counts[i] = 0;
3441     
3442        for (i = 0; i < bb->stmts_used; i++) {
3443           IRDirty* d;
3444           IRCAS*   cas;
3445           stmt = bb->stmts[i];
3446           /* Check any temps used by this statement. */
3447           useBeforeDef_Stmt(bb,stmt,def_counts);
3448     
3449           /* Now make note of any temps defd by this statement. */
3450           switch (stmt->tag) {
3451           case Ist_WrTmp:
3452              if (stmt->Ist.WrTmp.tmp < 0 || stmt->Ist.WrTmp.tmp >= n_temps)
3453                 sanityCheckFail(bb, stmt, 
3454                    "IRStmt.Tmp: destination tmp is out of range");
3455              def_counts[stmt->Ist.WrTmp.tmp]++;
3456              if (def_counts[stmt->Ist.WrTmp.tmp] > 1)
3457                 sanityCheckFail(bb, stmt, 
3458                    "IRStmt.Tmp: destination tmp is assigned more than once");
3459              break;
3460           case Ist_Store:
3461              break;
3462           case Ist_Dirty:
3463              if (stmt->Ist.Dirty.details->tmp != IRTemp_INVALID) {
3464                 d = stmt->Ist.Dirty.details;
3465                 if (d->tmp < 0 || d->tmp >= n_temps)
3466                    sanityCheckFail(bb, stmt, 
3467                       "IRStmt.Dirty: destination tmp is out of range");
3468                 def_counts[d->tmp]++;
3469                 if (def_counts[d->tmp] > 1)
3470                    sanityCheckFail(bb, stmt, 
3471                       "IRStmt.Dirty: destination tmp is assigned more than once");
3472              }
3473              break;
3474           case Ist_CAS:
3475              cas = stmt->Ist.CAS.details;
3476              if (cas->oldHi != IRTemp_INVALID) {
3477                 if (cas->oldHi < 0 || cas->oldHi >= n_temps)
3478                     sanityCheckFail(bb, stmt, 
3479                        "IRStmt.CAS: destination tmpHi is out of range");
3480                  def_counts[cas->oldHi]++;
3481                  if (def_counts[cas->oldHi] > 1)
3482                     sanityCheckFail(bb, stmt, 
3483                        "IRStmt.CAS: destination tmpHi is assigned more than once");
3484              }
3485              if (cas->oldLo < 0 || cas->oldLo >= n_temps)
3486                 sanityCheckFail(bb, stmt, 
3487                    "IRStmt.CAS: destination tmpLo is out of range");
3488              def_counts[cas->oldLo]++;
3489              if (def_counts[cas->oldLo] > 1)
3490                 sanityCheckFail(bb, stmt, 
3491                    "IRStmt.CAS: destination tmpLo is assigned more than once");
3492              break;
3493           case Ist_LLSC:
3494              if (stmt->Ist.LLSC.result < 0 || stmt->Ist.LLSC.result >= n_temps)
3495                 sanityCheckFail(bb, stmt,
3496                    "IRStmt.LLSC: destination tmp is out of range");
3497              def_counts[stmt->Ist.LLSC.result]++;
3498              if (def_counts[stmt->Ist.LLSC.result] > 1)
3499                 sanityCheckFail(bb, stmt,
3500                    "IRStmt.LLSC: destination tmp is assigned more than once");
3501              break;
3502           default:
3503              /* explicitly handle the rest, so as to keep gcc quiet */
3504              break;
3505           }
3506        }
3507     
3508        /* Typecheck everything. */
3509        for (i = 0; i < bb->stmts_used; i++)
3510           if (bb->stmts[i])
3511              tcStmt( bb, bb->stmts[i], guest_word_size );
3512        if (typeOfIRExpr(bb->tyenv,bb->next) != guest_word_size)
3513           sanityCheckFail(bb, NULL, "bb->next field has wrong type");
3514     }

3522        if (c1->tag != c2->tag)
3523           return False;
3524     
3525        switch (c1->tag) {
3526           case Ico_U1:  return toBool( (1 & c1->Ico.U1) == (1 & c2->Ico.U1) );
3527           case Ico_U8:  return toBool( c1->Ico.U8  == c2->Ico.U8 );
3528           case Ico_U16: return toBool( c1->Ico.U16 == c2->Ico.U16 );
3529           case Ico_U32: return toBool( c1->Ico.U32 == c2->Ico.U32 );
3530           case Ico_U64: return toBool( c1->Ico.U64 == c2->Ico.U64 );
3531           case Ico_F32: return toBool( c1->Ico.F32 == c2->Ico.F32 );
3532           case Ico_F32i: return toBool( c1->Ico.F32i == c2->Ico.F32i );
