69     /* If true, then this Segment may be mentioned in the core */
70     static Bool may_dump(const NSegment *seg)
71     {
72        if (seg->kind == SkAnonC ||
73            seg->kind == SkShmC ||
74            (seg->kind == SkFileC &&
75             !VKI_S_ISCHR(seg->mode) && !VKI_S_ISBLK(seg->mode)))
76           return True;
77     
78        return False;
79     }
80     
81     /* If true, then this Segment's contents will be in the core */
82     static Bool should_dump(const NSegment *seg)
83     {
84        return may_dump(seg); // && seg->hasW;
85     }

87     static void fill_ehdr(ESZ(Ehdr) *ehdr, Int num_phdrs)
88     {
89        VG_(memset)(ehdr, 0, sizeof(*ehdr));
90     
91        VG_(memcpy)(ehdr->e_ident, ELFMAG, SELFMAG);
92        ehdr->e_ident[EI_CLASS]   = VG_ELF_CLASS;
93        ehdr->e_ident[EI_DATA]    = VG_ELF_DATA2XXX;
94        ehdr->e_ident[EI_VERSION] = EV_CURRENT;
95     
96        ehdr->e_type = ET_CORE;
97        ehdr->e_machine = VG_ELF_MACHINE;
98        ehdr->e_version = EV_CURRENT;
99        ehdr->e_entry = 0;
100        ehdr->e_phoff = sizeof(ESZ(Ehdr));
101        ehdr->e_shoff = 0;
102        ehdr->e_flags = 0;
103        ehdr->e_ehsize = sizeof(ESZ(Ehdr));
104        ehdr->e_phentsize = sizeof(ESZ(Phdr));
105        ehdr->e_phnum = num_phdrs;
106        ehdr->e_shentsize = 0;
107        ehdr->e_shnum = 0;
108        ehdr->e_shstrndx = 0;
109     
110     }

112     static void fill_phdr(ESZ(Phdr) *phdr, const NSegment *seg, UInt off, Bool write)
113     {
114        SizeT len = seg->end - seg->start;
115     
116        write = write && should_dump(seg);
117     
118        VG_(memset)(phdr, 0, sizeof(*phdr));
119     
120        phdr->p_type = PT_LOAD;
121        phdr->p_offset = off;
122        phdr->p_vaddr = seg->start;
123        phdr->p_paddr = 0;
124        phdr->p_filesz = write ? len : 0;
125        phdr->p_memsz = len;
126        phdr->p_flags = 0;
127     
128        if (seg->hasR)
129           phdr->p_flags |= PF_R;
130        if (seg->hasW)
131           phdr->p_flags |= PF_W;
132        if (seg->hasX)
133           phdr->p_flags |= PF_X;
134     
135        phdr->p_align = VKI_PAGE_SIZE;
136     }

145     {
146        return sizeof(ESZ(Nhdr)) + VG_ROUNDUP(VG_(strlen)(n->name)+1, 4) + VG_ROUNDUP(n->note.n_descsz, 4);

149     static void add_note(struct note **list, const Char *name, UInt type, const void *data, UInt datasz)
150     {
151        Int namelen = VG_(strlen)(name)+1;
152        Int notelen = sizeof(struct note) + 
153           VG_ROUNDUP(namelen, 4) + 
154           VG_ROUNDUP(datasz, 4);
155        struct note *n = VG_(arena_malloc)(VG_AR_CORE, "coredump-elf.an.1", notelen);
156     
157        VG_(memset)(n, 0, notelen);
158     
159        n->next = *list;
160        *list = n;
161     
162        n->note.n_type = type;
163        n->note.n_namesz = namelen;
164        n->note.n_descsz = datasz;
165     
166        VG_(memcpy)(n->name, name, namelen);
167        VG_(memcpy)(n->name+VG_ROUNDUP(namelen,4), data, datasz);
168     }

175     static void fill_prpsinfo(const ThreadState *tst, struct vki_elf_prpsinfo *prpsinfo)
176     {
177        static Char name[VKI_PATH_MAX];
178     
179        VG_(memset)(prpsinfo, 0, sizeof(*prpsinfo));
180     
181        switch(tst->status) {
182        case VgTs_Runnable:
183        case VgTs_Yielding:
184           prpsinfo->pr_sname = 'R';
185           break;
186     
187        case VgTs_WaitSys:
188           prpsinfo->pr_sname = 'S';
189           break;
190     
191        case VgTs_Zombie:
192           prpsinfo->pr_sname = 'Z';
193           break;
194     
195        case VgTs_Empty:
196        case VgTs_Init:
197           prpsinfo->pr_sname = '?';
198           break;
199        }
200     
201        prpsinfo->pr_uid = 0;
202        prpsinfo->pr_gid = 0;
203        
204        if (VG_(resolve_filename)(VG_(cl_exec_fd), name, VKI_PATH_MAX)) {
205           Char *n = name+VG_(strlen)(name)-1;
206     
207           while (n > name && *n != '/')
208     	 n--;
209           if (n != name)
210     	 n++;
211     
212           VG_(strncpy)(prpsinfo->pr_fname, n, sizeof(prpsinfo->pr_fname));
213        }
214     }

468     static
469     void make_elf_coredump(ThreadId tid, const vki_siginfo_t *si, UInt max_size)
470     {
471        Char* buf = NULL;
472        Char *basename = "vgcore";
473        Char *coreext = "";
474        Int seq = 0;
475        Int core_fd;
476        NSegment const * seg;
477        ESZ(Ehdr) ehdr;
478        ESZ(Phdr) *phdrs;
479        Int num_phdrs;
480        Int i, idx;
481        UInt off;
482        struct note *notelist, *note;
483        UInt notesz;
484        struct vki_elf_prpsinfo prpsinfo;
485        struct vki_elf_prstatus prstatus;
486        Addr *seg_starts;
487        Int n_seg_starts;
488     
489        if (VG_(clo_log_fname_expanded) != NULL) {
490           coreext = ".core";
491           basename = VG_(expand_file_name)(
492                         "--log-file (while creating core filename)",
493                         VG_(clo_log_fname_expanded));
494        }
495     
496        vg_assert(coreext);
497        vg_assert(basename);
498        buf = VG_(malloc)( "coredump-elf.mec.1", 
499                           VG_(strlen)(coreext) + VG_(strlen)(basename)
500                              + 100/*for the two %ds. */ );
501        vg_assert(buf);
502     
503        for(;;) {
504           SysRes sres;
505     
506           if (seq == 0)
507     	 VG_(sprintf)(buf, "%s%s.%d",
508     		      basename, coreext, VG_(getpid)());
509           else
510     	 VG_(sprintf)(buf, "%s%s.%d.%d",
511     		      basename, coreext, VG_(getpid)(), seq);
512           seq++;
513     
514           sres = VG_(open)(buf, 			   
515                            VKI_O_CREAT|VKI_O_WRONLY|VKI_O_EXCL|VKI_O_TRUNC, 
516                            VKI_S_IRUSR|VKI_S_IWUSR);
517           if (!sr_isError(sres)) {
518              core_fd = sr_Res(sres);
519     	 break;
520           }
521     
522           if (sr_isError(sres) && sr_Err(sres) != VKI_EEXIST)
523     	 return;		/* can't create file */
524        }
525     
526        /* Get the segments */
527        seg_starts = VG_(get_segment_starts)(&n_seg_starts);
528     
529        /* First, count how many memory segments to dump */
530        num_phdrs = 1;		/* start with notes */
531        for(i = 0; i < n_seg_starts; i++) {
532           if (!may_dump(VG_(am_find_nsegment(seg_starts[i]))))
533     	 continue;
534     
535           num_phdrs++;
536        }
537     
538        fill_ehdr(&ehdr, num_phdrs);
539     
540        notelist = NULL;
541     
542        /* Second, work out their layout */
543        phdrs = VG_(arena_malloc)(VG_AR_CORE, "coredump-elf.mec.1", 
544                                  sizeof(*phdrs) * num_phdrs);
545     
546        for(i = 1; i < VG_N_THREADS; i++) {
547           vki_elf_fpregset_t  fpu;
548     
549           if (VG_(threads)[i].status == VgTs_Empty)
550     	 continue;
551     
552     #if defined(VGP_x86_linux)
553           {
554              vki_elf_fpxregset_t xfpu;
555              fill_xfpu(&VG_(threads)[i], &xfpu);
556              add_note(&notelist, "LINUX", NT_PRXFPREG, &xfpu, sizeof(xfpu));
557           }
558     #endif
559     
560           fill_fpu(&VG_(threads)[i], &fpu);
561           add_note(&notelist, "CORE", NT_FPREGSET, &fpu, sizeof(fpu));
562     
563           fill_prstatus(&VG_(threads)[i], &prstatus, si);
564           add_note(&notelist, "CORE", NT_PRSTATUS, &prstatus, sizeof(prstatus));
565        }
566     
567        fill_prpsinfo(&VG_(threads)[tid], &prpsinfo);
568        add_note(&notelist, "CORE", NT_PRPSINFO, &prpsinfo, sizeof(prpsinfo));
569     
570        for(note = notelist, notesz = 0; note != NULL; note = note->next)
571           notesz += note_size(note);
572     
573        off = sizeof(ehdr) + sizeof(*phdrs) * num_phdrs;
574     
575        phdrs[0].p_type = PT_NOTE;
576        phdrs[0].p_offset = off;
577        phdrs[0].p_vaddr = 0;
578        phdrs[0].p_paddr = 0;
579        phdrs[0].p_filesz = notesz;
580        phdrs[0].p_memsz = 0;
581        phdrs[0].p_flags = 0;
582        phdrs[0].p_align = 0;
583     
584        off += notesz;
585     
586        off = VG_PGROUNDUP(off);
587     
588        for(i = 0, idx = 1; i < n_seg_starts; i++) {
589           seg = VG_(am_find_nsegment(seg_starts[i]));
590     
591           if (!may_dump(seg))
592     	 continue;
593     
594           fill_phdr(&phdrs[idx], seg, off, (seg->end - seg->start + off) < max_size);
595           
596           off += phdrs[idx].p_filesz;
597     
598           idx++;
599        }
600     
601        /* write everything out */
602        VG_(write)(core_fd, &ehdr, sizeof(ehdr));
603        VG_(write)(core_fd, phdrs, sizeof(*phdrs) * num_phdrs);
604     
605        for(note = notelist; note != NULL; note = note->next)
606           write_note(core_fd, note);
607        
608        VG_(lseek)(core_fd, phdrs[1].p_offset, VKI_SEEK_SET);
609     
610        for(i = 0, idx = 1; i < n_seg_starts; i++) {
611           seg = VG_(am_find_nsegment(seg_starts[i]));
612     
613           if (!should_dump(seg))
614     	 continue;
615     
616           if (phdrs[idx].p_filesz > 0) {
617     	 vg_assert(VG_(lseek)(core_fd, phdrs[idx].p_offset, VKI_SEEK_SET) == phdrs[idx].p_offset);
618     	 vg_assert(seg->end - seg->start >= phdrs[idx].p_filesz);
619     
620     	 (void)VG_(write)(core_fd, (void *)seg->start, phdrs[idx].p_filesz);
621           }
622           idx++;
623        }
624     
625        VG_(free)(seg_starts);
626     
627        VG_(close)(core_fd);
628     }
629     
630     void VG_(make_coredump)(ThreadId tid, const vki_siginfo_t *si, UInt max_size)
631     {
632        make_elf_coredump(tid, si, max_size);
633     }
