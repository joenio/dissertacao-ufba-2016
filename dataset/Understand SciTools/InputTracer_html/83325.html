7     void cpuid ( UInt* eax, UInt* ebx, UInt* ecx, UInt* edx, 
8                  UInt index, UInt ecx_in )
9     {
10        UInt a,b,c,d;
11        asm volatile ("cpuid"
12                      : "=a" (a), "=b" (b), "=c" (c), "=d" (d) 
13                      : "0" (index), "2"(ecx_in) );
14        *eax = a; *ebx = b; *ecx = c; *edx = d;
15        printf("%08x %08x -> %08x %08x %08x %08x\n",
16               index,ecx_in, a,b,c,d );
17     }
