50     /* Main instrumentation callback */
51     IRSB* IT_(instrument)(VgCallbackClosure* closure,
52                           IRSB* sbIn, 
53                           VexGuestLayout* layout, 
54                           VexGuestExtents* vge,
55                           IRType gWordTy, IRType hWordTy)
56     {
57        Int        i;
58        IRSB*      sbOut;
59        IRType     type;
60        IRTypeEnv* tyenv;
61        
62        Addr32     curInstrAddr = 0x0;
63        Int        curInstrLen  = 0;
64        
65        // Propagation exclusion flag for current executable
66        Bool       isExcludedImage = False;
67     
68        // Buffer for storing the current executable's name
69        static Char imageNameBuffer[1024];
70     
71        // Handle for functions in it_clo.c for the current executable
72        Word       curExecutableHandle = -1;
73     
74        // Taint sink exclusion flag for current executable
75        Bool       isSinkExcludedImage = False;
76     
77        // Taint sink INclusion flag for current instruction
78        Bool       isSinkIncludedInstr = False;
79     
80        if (gWordTy != hWordTy) {
81           /* We don't currently support this case. */
82           VG_(tool_panic)("host/guest word size mismatch");
83        }
84        
85        /* Set up SB */
86        sbOut = deepCopyIRSBExceptStmts(sbIn);
87        tyenv = sbOut->tyenv;
88        
89        // Copy verbatim any IR preamble preceding the first IMark
90        i = 0;
91        while (i < sbIn->stmts_used && sbIn->stmts[i]->tag != Ist_IMark) {
92           addStmtToIRSB( sbOut, sbIn->stmts[i] );
93           i++;
94        }
95     
96        /* Handle first IMark to set curInstrXXX variables and check for
97           propagation and sink inclusions/exclusions. The check for excluded images
98           only needs to be performed at the beginning of a superblock, or (possibly)
99           after a chased function call (see below). The check for explicitly
100           included code sections need to be performed at each IMark */
101     
102        tl_assert( sbIn->stmts[i]->tag == Ist_IMark );
103        curInstrAddr = (Addr32)sbIn->stmts[i]->Ist.IMark.addr;
104        curInstrLen  = sbIn->stmts[i]->Ist.IMark.len;
105     
106        // Propagation exclusion handling
107        isExcludedImage = IT_(is_in_excluded_range_prop)(curInstrAddr);
108     
109        // Sink exclusion handling.
110        if(VG_(get_objname)(curInstrAddr, imageNameBuffer,sizeof(imageNameBuffer) ) )
111        {
112     	   curExecutableHandle =
113     			   IT_(get_sink_filter_handle_from_path)(imageNameBuffer);
114     	   isSinkExcludedImage =
115     			   !isExcludedImage &&
116     			   IT_(is_in_sink_exclude_pattern)(imageNameBuffer, curInstrAddr);
117        } else {
118     	   curExecutableHandle = -1;
119     	   isSinkExcludedImage = False;
120        }
121     
122        // Must be done at each IMark
123        if(curExecutableHandle == -1)
124     	   isSinkIncludedInstr = !isSinkExcludedImage;
125        else {
126     	   isSinkIncludedInstr =
127        			!isSinkExcludedImage ||
128        			IT_(is_included_in_sink_filter)(
129     						   curExecutableHandle,
130     						   curInstrAddr);
131        }
132     
133        addStmtToIRSB( sbOut, sbIn->stmts[i] );
134        i++;
135        
136        IT_(init_instrumentation)();
137     
138        for (/*use current i*/; i < sbIn->stmts_used; i++) {
139           IRStmt* st = sbIn->stmts[i];
140           if (!st || st->tag == Ist_NoOp) continue;
141          
142           // If instruction is in an excluded image, don't add any instrumentation.
143           // Note that Exits and IMarks needs special treatment.
144           if(isExcludedImage && st->tag != Ist_Exit && st->tag != Ist_IMark) {
145         	  addStmtToIRSB( sbOut, st );
146         	  continue;
147           }
148     
149           switch (st->tag) {
150              case Ist_NoOp:
151              case Ist_AbiHint:
152              case Ist_MBE:
153                 addStmtToIRSB( sbOut, st );
154                 break;
155     
156              case Ist_Put: {
157                 IRExpr* data = st->Ist.Put.data;
158                 Int offset = st->Ist.Put.offset;
159                 
160                 IT_(instrument_put)(sbOut, tyenv, data, offset);
161                 
162                 // Remember last instruction that modified EFLAGS, for later
163                 // taint sink events.
164                 IT_(instrument_eflags_update)(sbOut, offset, curInstrAddr);
165     
166                 addStmtToIRSB( sbOut, st );
167                 break;
168              }
169                 
170              case Ist_PutI:
171                 // Do nothing (FP/SIMD operation)
172                 addStmtToIRSB( sbOut, st );
173                 break;
174                 
175              case Ist_IMark: 
176                 // Check if we have chased to another block of code, if so
177                 // the stack trace update flag should be set to True.
178                 if(st->Ist.IMark.addr != curInstrAddr + curInstrLen) {
179                    IT_(instrument_set_stack_trace_update_flag)(sbOut);
180     
181                    // We (might) need to re-check for an excluded image here, if we
182                    // have chased to a function in another image. (Can that ever
183                    // happen?)
184     
185                    // Propagation exclusion handling
186                    isExcludedImage =
187                 		   IT_(is_in_excluded_range_prop)(st->Ist.IMark.addr);
188     
189                    // Sink exclusion handling.
190                    if(VG_(get_objname)(
191                 		   st->Ist.IMark.addr,
192                 		   imageNameBuffer,
193                 		   sizeof(imageNameBuffer) ) )
194                    {
195                 	   curExecutableHandle =
196                 	      IT_(get_sink_filter_handle_from_path)(imageNameBuffer);
197                 	   isSinkExcludedImage =
198                 			   !isExcludedImage &&
199                 			   IT_(is_in_sink_exclude_pattern)(
200     								  imageNameBuffer,
201                 					   curInstrAddr);
202                    } else {
203                 	   curExecutableHandle = -1;
204                 	   isSinkExcludedImage = False;
205                    }
206     
207                    // Check if last instruction was a CALL.
208                    // In this case the dynamic stack trace needs updating
209                    UChar opcode = *((UChar*)(curInstrAddr));
210                    if(opcode == 0xE8 || opcode == 0xFF) {
211                       IT_(instrument_push_stacktrace)(sbOut, tyenv, curInstrAddr);
212                    }
213                       
214                 }
215                
216                 curInstrAddr = st->Ist.IMark.addr;
217                 curInstrLen  = st->Ist.IMark.len;
218     
219                 // Must be done at each IMark
220                 if(curExecutableHandle == -1)
221                 	isSinkIncludedInstr = !isSinkExcludedImage;
222                 else {
223                 	isSinkIncludedInstr =
224                 			!isSinkExcludedImage ||
225                 			IT_(is_included_in_sink_filter)(
226              						   curExecutableHandle,
227              						   curInstrAddr);
228                 }
229     
230                 addStmtToIRSB( sbOut, st );
231                 break;
232                 
233              case Ist_WrTmp: {
234                 /* When writing to a temp, we only need to perform some 
235                    instrumentation if the input operands are non-constans (i.e. 
236                    can hold taint), since temps are SSA and can never be untainted.
237                 */
238                 IRTemp target = st->Ist.WrTmp.tmp;
239                 IRExpr* expr = st->Ist.WrTmp.data;
240                 type = typeOfIRExpr(sbOut->tyenv, expr);
241                 tl_assert(type != Ity_INVALID);
242                 
243                 switch (expr->tag) {
244                    case Iex_Get:
245                       IT_(instrument_get)(sbOut, tyenv, target, expr->Iex.Get.offset);
246                       break;
247                       
248                    case Iex_GetI:
249                       // Do nothing (FP/SIMD operation)
250                       break;
251                       
252                    case Iex_RdTmp:
253                       IT_(instrument_copy)(sbOut, tyenv, target, 
254                                            expr->Iex.RdTmp.tmp);
255                       break;
256                       
257                    case Iex_Unop:
258                       // Only care about non-constants
259                       if(expr->Iex.Unop.arg->tag == Iex_RdTmp) {
260                          IRTemp arg = expr->Iex.Unop.arg->Iex.RdTmp.tmp;
261                          IROp op = expr->Iex.Unop.op;
262                          
263                          switch(op) {
264                             
265                             /* Negation */
266                             case Iop_Not1:
267                             case Iop_Not8:
268                             case Iop_Not16:
269                             case Iop_Not32:
270                             case Iop_Not64:
271                                IT_(instrument_copy)(sbOut, tyenv, target, arg);
272                                break;
273                                
274                             
275                             /* Count leading/trailing zeroes */
276                             
277                             case Iop_Clz64:
278                             case Iop_Ctz64:
279                             case Iop_Clz32:
280                             case Iop_Ctz32:
281                                IT_(instrument_full_union)(sbOut, tyenv, target,
282                                                           expr, NULL);
283                                break;
284                                
285                                
286                             /* Type (size) conversions.
287                                Rules for taint propagation are as follows:
288                                
289                                -- Narrowing conversions: Just copy taint from 
290                                   relevant source bytes to destination.
291                                   
292                                -- Unsigned widening conversions: Copy taint
293                                   from source to destination bytes, setting 
294                                   additional destination bytes to 'not-tainted'.
295                                   
296                                -- Signed widening conversions: Copy taint
297                                   from source to destination bytes, and copy 
298                                   taint lables from the most significant source
299                                   byte to all additional destination bytes.
300                                   This is because sign extension can cause the

1131                          case Iop_MSubF32:
1132                             break;
1133                             
1134                          default:
1135                             ppIROp(op);
1136                             VG_(tool_panic)("InputTracer: Unknown Qop");
1137                       }
1138                       break;
1139                    }
1140                    
1141                    case Iex_Load:
1142                       // Should only be litte-endian on the x86
1143                       tl_assert(expr->Iex.Load.end == Iend_LE);
1144     
1145                       IRExpr* addr = expr->Iex.Load.addr;
1146                       IT_(instrument_load)(sbOut, tyenv, target, addr);
1147                       break;
1148                       
1149                 /* case Iex_Const: Needs not be considered. These temps are 
1150                                    treated like e.g. those in the preamble, for which
1151                                    no shadow temps exist. */
1152                    
1153                    case Iex_CCall:
1154                       IT_(instrument_clean_call)(sbOut, tyenv, target,
1155                                                  expr->Iex.CCall.cee,
1156                                                  expr->Iex.CCall.args);
1157                       break; 
1158                       
1159                    
1160                    case Iex_Mux0X:
1161                       IT_(instrument_Mux0X)(sbOut, 
1162                                             tyenv, 
1163                                             target,
1164                                             expr->Iex.Mux0X.cond, 
1165                                             expr->Iex.Mux0X.expr0,
1166                                             expr->Iex.Mux0X.exprX);
1167                       break;
1168     
1169                    default:
1170                       break;
1171                 }
1172     
1173                 addStmtToIRSB( sbOut, st );
1174                 break; /* case Ist_WrTmp */
1175              }
1176              
1177              case Ist_Store: {
1178                 // Shold only be litte-endian on x86
1179                 tl_assert(st->Ist.Store.end == Iend_LE);
1180                 
1181                 IRExpr* data = st->Ist.Store.data;
1182                 IRExpr* addr = st->Ist.Store.addr;
1183     
1184                 IT_(instrument_store)(sbOut, tyenv, data, addr);
1185                 
1186                 // Memory write taint sink handling
1187                 if(data->tag == Iex_RdTmp && isSinkIncludedInstr) {
1188                    IRTemp dataTemp = data->Iex.RdTmp.tmp;
1189                    IT_(instrument_mem_write_sink)(sbOut, tyenv, dataTemp);
1190                 }
1191                 
1192                 addStmtToIRSB( sbOut, st );
1193                 break;
1194              }
1195              
1196              case Ist_CAS: 
1197                 // Shold only be litte-endian on x86
1198                 tl_assert(st->Ist.CAS.details->end == Iend_LE);
1199                 
1200                 addStmtToIRSB( sbOut, st );
1201                 /* Instrumentation must happen _after_ the actual statement in
1202                    order to decide if the swap really happend. */
1203                 
1204                 IT_(instrument_CAS)(sbOut, tyenv, st->Ist.CAS.details);   
1205                 break;
1206     
1207              case Ist_LLSC:
1208                 // x86 doesn't have a LL/SC instruction...
1209                 VG_(tool_panic)("InputTracer: Unexpected LL/SC");
1210     
1211              
1212              case Ist_Dirty:
1213                 /* Dirty calls are ignored for now, since they are rare and as it
1214                    seems that there are no "built in" dirty calls that can
1215                    propagate taint */
1216                 addStmtToIRSB( sbOut, st );
1217                 break;
1218     
1219              case Ist_Exit:
1220                 if( st->Ist.Exit.guard->tag == Iex_RdTmp && isSinkIncludedInstr) {
1221                    IRTemp guard = st->Ist.Exit.guard->Iex.RdTmp.tmp;
1222     
1223                    // Create VEX code to record conditional branch taint sink event
1224                    IT_(instrument_cond_branch_sink)(sbOut, tyenv, guard);
1225                 }
1226              
1227                 // Call or Ret means that call stack is updated
1228                 // Note that the call stack should be updated _after_ any taint
1229                 // sink events are registered, so that the events call stack
1230                 // reflects the state before the Call/Ret happend.
1231                 if(st->Ist.Exit.jk == Ijk_Call) {
1232                    IT_(instrument_set_stack_trace_update_flag)(sbOut);
1233                    IT_(instrument_push_stacktrace)(sbOut, tyenv, curInstrAddr);
1234                 } else if(st->Ist.Exit.jk == Ijk_Ret) {
1235                    // Should never happen?
1236                    tl_assert(0);
1237                 }
1238                 
1239                 IT_(instrument_cleanup)(sbOut, tyenv, st->Ist.Exit.guard);
1240                 
1241                 addStmtToIRSB( sbOut, st );
1242                 break;
1243     
1244              default:
1245                 tl_assert(0);
1246           }
1247        }
1248     
1249        // Check if last exit is to a non-constant address and generate taint
1250        // sink event if so.
1251        if( sbIn->next->tag == Iex_RdTmp && isSinkIncludedInstr) {
1252           IRTemp target = sbIn->next->Iex.RdTmp.tmp;
1253           IT_(instrument_uncond_branch_sink)(sbOut, tyenv, target);
1254        }
1255        
1256        // Check final jump for Call and Ret
1257        // Note that the call stack should be updated _after_ any taint
1258        // sink events are registered, so that the events call stack
1259        // reflects the state before the Call/Ret happend.   
1260        if(sbIn->jumpkind == Ijk_Call) {
1261           IT_(instrument_set_stack_trace_update_flag)(sbOut);
1262           IT_(instrument_push_stacktrace)(sbOut, tyenv, curInstrAddr);
1263        } else if(sbIn->jumpkind == Ijk_Ret) {
1264           IT_(instrument_set_stack_trace_update_flag)(sbOut);
1265           IT_(instrument_pop_stacktrace)(sbOut, tyenv, sbIn->next->Iex.RdTmp.tmp);
1266        }
1267        
1268        IT_(finish_instrumentation)(sbOut, tyenv); 
1269     
1270        return sbOut;
1271     }
