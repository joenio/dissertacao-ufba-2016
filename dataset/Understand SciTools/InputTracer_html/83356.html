7     UInt read16le ( void* a )
8     {
9       UInt res;
10       __asm volatile(
11         "   lhbrx   %0,0,%1         \n"     // Get half word and reverse the bytes
12         : "=b" (res)  // %0 - Output operand
13         : "b" (a)        // %1 - Input operand
14         : "memory"       // Consider memory clobberred for aliasing
15       );
16       return res;
17     }
