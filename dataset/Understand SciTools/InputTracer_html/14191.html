28      * addresses have been accessed for reading or writing within a given
29      * segment.
30      */

42     /* Defines. */
43     
44     #define LHS_R (1<<0)
45     #define LHS_W (1<<1)
46     #define RHS_R (1<<2)
47     #define RHS_W (1<<3)
48     #define HAS_RACE(a) ((((a) & RHS_W) && ((a) & (LHS_R | LHS_W)))         \
49                          || (((a) & LHS_W) && ((a) & (RHS_R | RHS_W))))

69     /* Complete bitmap. */

73        OSet*                oset;
74     };
