43     // Helper function for balance_treap
44     static Int treap_node_comparator(void* e1, void* e2)
45     {
46        UInt p1 = TREAP_GET_PRIO(*(TreapNodeIndex*)e1);
47        UInt p2 = TREAP_GET_PRIO(*(TreapNodeIndex*)e2);
48        
49        /*
50        if(p1 < p2) return -1;
51        else if(p1 > p2) return 1;
52        else return 0;
53        */
54     
55        return p1 - p2;
56     }

58     static void balance_treap(Treap* t)
59     {
60        TreapNodeIndex i, current, parent;
61        TreapNodeIndex* indices;
62        TreapNode* array = t->array;
63        UInt treap_size = t->current_idx;
64           
65        if(treap_size == 0)
66           return;
67     
68        indices = VG_(malloc)("it_treap.c: balance_treap",
69                              treap_size * sizeof(TreapNodeIndex));
70           
71        for(i = 0; i < treap_size; i++)
72           indices[i] = i;
73           
74        VG_(ssort)(indices,
75     		      treap_size,
76     		      sizeof(TreapNodeIndex),
77     		      treap_node_comparator);
78        
79        // Re-insert first element as root
80        t->root = indices[0];
81        array[indices[0]].left = array[indices[0]].right = TREAP_INVALID_REF;
82        
83        // Re-insert all elements, updating their logical order in the tree
84        // without re-allocating them in the array.
85        for(i = 1; i < treap_size; i++) {
86           TreapNodeIndex newIdx = indices[i];
87           current = t->root; // Start from root
88           while(current != TREAP_INVALID_REF) {
89              parent = current;
90              if(array[newIdx].value > array[current].value)
91                 current = array[current].right;
92              else if(array[newIdx].value < array[current].value)
93                 current = array[current].left;
94              else
95             	 tl_assert(0); // Shouldn't happen
96           }
97           
98           array[newIdx].left = array[newIdx].right = TREAP_INVALID_REF;      
99           
100           if(array[newIdx].value > array[parent].value)
101              array[parent].right = newIdx;
102           else
103              array[parent].left = newIdx;
104        }
105              
106        VG_(free)(indices);
107     }

109     static inline TreapNodeIndex treap_allocate_node(Treap* t)
110     {
111        if(t->current_size == t->current_idx) {
112           t->array = VG_(realloc)(
113              "it_treap.c: treap_allocate_node",
114              t->array,
115              2 * t->current_size * sizeof(TreapNode));
116           t->current_size *= 2;
117        }
118     
119        return t->current_idx++;
120     }

124     inline void IT_(init_treap)(Treap* t)
125     {
126        t->current_size = TREAP_INITIAL_CAPACITY;
127        t->current_idx = 0;
128        t->root = TREAP_INVALID_REF;
129        t->array = VG_(malloc)("it_treap.c: IT_(init_treap)",
130     		                   TREAP_INITIAL_CAPACITY * sizeof(TreapNode));
131     }

134     {
135        VG_(free)(t->array);

138     void IT_(treap_add_node)(Treap* treap, TreapValueType value)
139     {
140        TreapNodeIndex current, parent;
141        UInt treap_size = treap->current_idx;
142        UInt priority;
143        UInt stack_top;
144        
145        if(treap_size == TREAP_MAX_SIZE)
146           return;
147        
148     #if TREAP_BALANCE_THRES
149        if(treap_size == TREAP_BALANCE_THRES)
150           balance_treap(treap);
151     #endif
152     
153        // Keep copy of array pointer to avoid unnecessary memory dereferences
154        TreapNode* array = treap->array;
155        
156        // Stack to store the chain of nodes during traversal, and if the child was
157        // a left or right child. Left/right info is stored by masking the MSB
158        // of each entry.
159        static UInt stack[TREAP_MAX_SIZE];
160        int stackp = -1;
161     // Bitmasks
162     #define TREAP_LEFT_REF  0x80000000
163     #define TREAP_RIGHT_REF 0x00000000
164        
165        current = treap->root;
166        parent = TREAP_INVALID_REF;   // Avoid gcc "unitialized" warnings
167        
168        // Find insertion point. Two cases with code duplication here in order
169        // to avoid an 'if(treap_size < TREAP_BALANCE_THRES)' in every loop 
170        // iteration.
171        if(treap_size < TREAP_BALANCE_THRES) {
172           while(current != TREAP_INVALID_REF) {
173              parent = current;
174              if(value > array[current].value)
175                 current = array[current].right;
176              else if(value < array[current].value)
177                 current = array[current].left;
178              else
179                 return; // Nothing to do, no duplicates allowed
180           }
181        } else {
182           while(current != TREAP_INVALID_REF) {
183              if(value > array[current].value) {
184                 stack[++stackp] = (UInt)current /*| TREAP_RIGHT_REF unnecessary*/;
185                 current = array[current].right;
186              } else if(value < array[current].value) {
187                 stack[++stackp] = (UInt)current | TREAP_LEFT_REF;
188                 current = array[current].left;
189              } 
190              else
191                 return; // Nothing to do, no duplicates allowed
192           }
193        }
194        
195        // Allocate new node
196        current = treap_allocate_node(treap);
197        treap->array[current].value = value;
198        treap->array[current].left  = TREAP_INVALID_REF;
199        treap->array[current].right = TREAP_INVALID_REF;
200     
201        // If this is the first added element, set is as root and return
202        if(current == 0) {
203           treap->root = current;
204           return;
205        }
206     
207        // If beyond balancing threshold, get parent node from top of stack, 
208        // otherwise 'parent' was already set during the traversal phase
209        if(treap_size >= TREAP_BALANCE_THRES)
210           parent = stack[stackp] & 0x7FFFFFFF; // Mask away left/right marker
211        
212        // VERY IMPORTANT: treap_allocate_node could have changed the array address,
213        // re-fetch array address before proceeding
214        array = treap->array;
215         
216        // Insert node
217        if(value > array[parent].value)
218           array[parent].right = current;
219        else
220           array[parent].left = current;
221     
222     #if TREAP_BALANCE_THRES
223        // If we're below balancing threshold, we are done now.
224        if(treap_size < TREAP_BALANCE_THRES)
225           return;
226     #endif
227     
228        // Perform rotations to maintain heap property
229        priority = treap_randomness[current];
230        stack_top = stack[stackp--];
231        while( TREAP_GET_PRIO(parent) > TREAP_GET_PRIO(current) ) {
232           if(stack_top & TREAP_LEFT_REF) {
233              // Right rotation
234              array[parent].left = array[current].right;
235              array[current].right = parent;
236           } else {
237              // Left rotation
238              array[parent].right = array[current].left;
239              array[current].left = parent;
240           }
241        
242           // If stackp is below zero, we have reached the root   
243           if(stackp < 0) {
244              treap->root = current;
245              return;
246           }
247           
248           stack_top = stack[stackp--];
249           parent = stack_top & 0x7FFFFFFF;
250        }
251        // Set last grandparent's reference to point to the last current node to
252        // complete the last rotation
253        if(stack_top & TREAP_LEFT_REF)
254           array[parent].left = current;
255        else
256           array[parent].right = current;   
257     }
