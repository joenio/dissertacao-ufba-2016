67     /* Load the client whose name is VG_(argv_the_exename). */
68     
69     static void load_client ( /*OUT*/ExeInfo* info, 
70                               /*OUT*/Addr*    client_ip,
71     			  /*OUT*/Addr*    client_toc)
72     {
73        HChar* exe_name;
74        Int    ret;
75        SysRes res;
76     
77        vg_assert( VG_(args_the_exename) != NULL);
78        exe_name = ML_(find_executable)( VG_(args_the_exename) );
79     
80        if (!exe_name) {
81           VG_(printf)("valgrind: %s: command not found\n", VG_(args_the_exename));
82           VG_(exit)(127);      // 127 is Posix NOTFOUND
83        }
84     
85        VG_(memset)(info, 0, sizeof(*info));
86        ret = VG_(do_exec)(exe_name, info);
87        if (ret < 0) {
88           VG_(printf)("valgrind: could not execute '%s'\n", exe_name);
89           VG_(exit)(1);
90        }
91     
92        // The client was successfully loaded!  Continue.
93     
94        /* Get hold of a file descriptor which refers to the client
95           executable.  This is needed for attaching to GDB. */
96        res = VG_(open)(exe_name, VKI_O_RDONLY, VKI_S_IRUSR);
97        if (!sr_isError(res))
98           VG_(cl_exec_fd) = sr_Res(res);
99     
100        /* Copy necessary bits of 'info' that were filled in */
101        *client_ip  = info->init_ip;
102        *client_toc = info->init_toc;
103        VG_(brk_base) = VG_(brk_limit) = VG_PGROUNDUP(info->brkbase);
104     }

111     /* Prepare the client's environment.  This is basically a copy of our
112        environment, except:
113     
114          LD_PRELOAD=$VALGRIND_LIB/vgpreload_core-PLATFORM.so:
115                     ($VALGRIND_LIB/vgpreload_TOOL-PLATFORM.so:)?
116                     $LD_PRELOAD
117     
118        If this is missing, then it is added.
119     
120        Also, remove any binding for VALGRIND_LAUNCHER=.  The client should
121        not be able to see this.
122     
123        If this needs to handle any more variables it should be hacked
124        into something table driven.  The copy is VG_(malloc)'d space.
125     */
126     static HChar** setup_client_env ( HChar** origenv, const HChar* toolname)
127     {
128        HChar* preload_core    = "vgpreload_core";
129        HChar* ld_preload      = "LD_PRELOAD=";
130        HChar* v_launcher      = VALGRIND_LAUNCHER "=";
131        Int    ld_preload_len  = VG_(strlen)( ld_preload );
132        Int    v_launcher_len  = VG_(strlen)( v_launcher );
133        Bool   ld_preload_done = False;
134        Int    vglib_len       = VG_(strlen)(VG_(libdir));
135        Bool   debug           = False;
136     
137        HChar** cpp;
138        HChar** ret;
139        HChar*  preload_tool_path;
140        Int     envc, i;
141     
142        /* Alloc space for the vgpreload_core.so path and vgpreload_<tool>.so
143           paths.  We might not need the space for vgpreload_<tool>.so, but it
144           doesn't hurt to over-allocate briefly.  The 16s are just cautious
145           slop. */
146        Int preload_core_path_len = vglib_len + sizeof(preload_core) 
147                                              + sizeof(VG_PLATFORM) + 16;
148        Int preload_tool_path_len = vglib_len + VG_(strlen)(toolname) 
149                                              + sizeof(VG_PLATFORM) + 16;
150        Int preload_string_len    = preload_core_path_len + preload_tool_path_len;
151        HChar* preload_string     = VG_(malloc)("initimg-linux.sce.1",
152                                                preload_string_len);
153        vg_assert(origenv);
154        vg_assert(toolname);
155        vg_assert(preload_string);
156     
157        /* Determine if there's a vgpreload_<tool>_<platform>.so file, and setup
158           preload_string. */
159        preload_tool_path = VG_(malloc)("initimg-linux.sce.2", preload_tool_path_len);
160        vg_assert(preload_tool_path);
161        VG_(snprintf)(preload_tool_path, preload_tool_path_len,
162                      "%s/vgpreload_%s-%s.so", VG_(libdir), toolname, VG_PLATFORM);
163        if (VG_(access)(preload_tool_path, True/*r*/, False/*w*/, False/*x*/) == 0) {
164           VG_(snprintf)(preload_string, preload_string_len, "%s/%s-%s.so:%s", 
165                         VG_(libdir), preload_core, VG_PLATFORM, preload_tool_path);
166        } else {
167           VG_(snprintf)(preload_string, preload_string_len, "%s/%s-%s.so", 
168                         VG_(libdir), preload_core, VG_PLATFORM);
169        }
170        VG_(free)(preload_tool_path);
171     
172        VG_(debugLog)(2, "initimg", "preload_string:\n");
173        VG_(debugLog)(2, "initimg", "  \"%s\"\n", preload_string);
174     
175        /* Count the original size of the env */
176        if (debug) VG_(printf)("\n\n");
177        envc = 0;
178        for (cpp = origenv; cpp && *cpp; cpp++) {
179           envc++;
180           if (debug) VG_(printf)("XXXXXXXXX: BEFORE %s\n", *cpp);
181        }
182     
183        /* Allocate a new space */
184        ret = VG_(malloc) ("initimg-linux.sce.3",
185                           sizeof(HChar *) * (envc+1+1)); /* 1 new entry + NULL */
186        vg_assert(ret);
187     
188        /* copy it over */
189        for (cpp = ret; *origenv; ) {
190           if (debug) VG_(printf)("XXXXXXXXX: COPY   %s\n", *origenv);
191           *cpp++ = *origenv++;
192        }
193        *cpp = NULL;
194        
195        vg_assert(envc == (cpp - ret));
196     
197        /* Walk over the new environment, mashing as we go */
198        for (cpp = ret; cpp && *cpp; cpp++) {
199           if (VG_(memcmp)(*cpp, ld_preload, ld_preload_len) == 0) {
200              Int len = VG_(strlen)(*cpp) + preload_string_len;
201              HChar *cp = VG_(malloc)("initimg-linux.sce.4", len);
202              vg_assert(cp);
203     
204              VG_(snprintf)(cp, len, "%s%s:%s",
205                            ld_preload, preload_string, (*cpp)+ld_preload_len);
206     
207              *cpp = cp;
208     
209              ld_preload_done = True;
210           }
211           if (debug) VG_(printf)("XXXXXXXXX: MASH   %s\n", *cpp);
212        }
213     
214        /* Add the missing bits */
215        if (!ld_preload_done) {
216           Int len = ld_preload_len + preload_string_len;
217           HChar *cp = VG_(malloc) ("initimg-linux.sce.5", len);
218           vg_assert(cp);
219     
220           VG_(snprintf)(cp, len, "%s%s", ld_preload, preload_string);
221     
222           ret[envc++] = cp;
223           if (debug) VG_(printf)("XXXXXXXXX: ADD    %s\n", cp);
224        }
225     
226        /* ret[0 .. envc-1] is live now. */
227        /* Find and remove a binding for VALGRIND_LAUNCHER. */
228        for (i = 0; i < envc; i++)
229           if (0 == VG_(memcmp(ret[i], v_launcher, v_launcher_len)))
230              break;
231     
232        if (i < envc) {
233           for (; i < envc-1; i++)
234              ret[i] = ret[i+1];
235           envc--;
236        }
237     
238        VG_(free)(preload_string);
239        ret[envc] = NULL;
240     
241        for (i = 0; i < envc; i++) {
242           if (debug) VG_(printf)("XXXXXXXXX: FINAL  %s\n", ret[i]);
243        }
244     
245        return ret;
246     }

289     /* Add a string onto the string table, and return its address */
290     static char *copy_str(char **tab, const char *str)
291     {
292        char *cp = *tab;
293        char *orig = cp;
294     
295        while(*str)
296           *cp++ = *str++;
297        *cp++ = '\0';
298     
299        if (0)
300           VG_(printf)("copied %p \"%s\" len %lld\n", orig, orig, (Long)(cp-orig));
301     
302        *tab = cp;
303     
304        return orig;
305     }

308     /* ----------------------------------------------------------------
309      
310        This sets up the client's initial stack, containing the args,
311        environment and aux vector.
312     
313        The format of the stack is:
314     
315        higher address +-----------------+ <- clstack_end
316                       |                 |
317     		  : string table    :
318     		  |                 |
319     		  +-----------------+
320     		  | AT_NULL         |
321     		  -                 -
322     		  | auxv            |
323     		  +-----------------+
324     		  | NULL            |
325     		  -                 -
326     		  | envp            |
327     		  +-----------------+
328     		  | NULL            |
329     		  -                 -
330     		  | argv            |
331     		  +-----------------+
332     		  | argc            |
333        lower address  +-----------------+ <- sp
334                       | undefined       |
335     		  :                 :
336     
337        Allocate and create the initial client stack.  It is allocated down
338        from clstack_end, which was previously determined by the address
339        space manager.  The returned value is the SP value for the client.
340     
341        The client's auxv is created by copying and modifying our own one.
342        As a side effect of scanning our own auxv, some important bits of
343        info are collected:
344     
345           VG_(cache_line_size_ppc32) // ppc32 only -- cache line size
346           VG_(have_altivec_ppc32)    // ppc32 only -- is Altivec supported?
347     
348        ---------------------------------------------------------------- */
349     
350     struct auxv
351     {
352        Word a_type;
353        union {
354           void *a_ptr;
355           Word a_val;
356        } u;
357     };
358     
359     static
360     struct auxv *find_auxv(UWord* sp)
361     {
362        sp++;                // skip argc (Nb: is word-sized, not int-sized!)
363     
364        while (*sp != 0)     // skip argv
365           sp++;
366        sp++;
367     
368        while (*sp != 0)     // skip env
369           sp++;
370        sp++;
371        
372     #if defined(VGA_ppc32) || defined(VGA_ppc64)
373     # if defined AT_IGNOREPPC
374        while (*sp == AT_IGNOREPPC)        // skip AT_IGNOREPPC entries
375           sp += 2;
376     # endif
377     #endif
378     
379        return (struct auxv *)sp;
380     }
381     
382     static 
383     Addr setup_client_stack( void*  init_sp,
384                              char** orig_envp, 
385                              const ExeInfo* info,
386                              UInt** client_auxv,
387                              Addr   clstack_end,
388                              SizeT  clstack_max_size )
389     {
390        SysRes res;
391        char **cpp;
392        char *strtab;		/* string table */
393        char *stringbase;
394        Addr *ptr;
395        struct auxv *auxv;
396        const struct auxv *orig_auxv;
397        const struct auxv *cauxv;
398        unsigned stringsize;		/* total size of strings in bytes */
399        unsigned auxsize;		/* total size of auxv in bytes */
400        Int argc;			/* total argc */
401        Int envc;			/* total number of env vars */
402        unsigned stacksize;		/* total client stack size */
403        Addr client_SP;	        /* client stack base (initial SP) */
404        Addr clstack_start;
405        Int i;
406        Bool have_exename;
407     
408        vg_assert(VG_IS_PAGE_ALIGNED(clstack_end+1));
409        vg_assert( VG_(args_for_client) );
410     
411        /* use our own auxv as a prototype */
412        orig_auxv = find_auxv(init_sp);
413     
414        /* ==================== compute sizes ==================== */
415     
416        /* first of all, work out how big the client stack will be */
417        stringsize   = 0;
418        have_exename = VG_(args_the_exename) != NULL;
419     
420        /* paste on the extra args if the loader needs them (ie, the #! 
421           interpreter and its argument) */
422        argc = 0;
423        if (info->interp_name != NULL) {
424           argc++;
425           stringsize += VG_(strlen)(info->interp_name) + 1;
426        }
427        if (info->interp_args != NULL) {
428           argc++;
429           stringsize += VG_(strlen)(info->interp_args) + 1;
430        }
431     
432        /* now scan the args we're given... */
433        if (have_exename)
434           stringsize += VG_(strlen)( VG_(args_the_exename) ) + 1;
435     
436        for (i = 0; i < VG_(sizeXA)( VG_(args_for_client) ); i++) {
437           argc++;
438           stringsize += VG_(strlen)( * (HChar**) 
439                                        VG_(indexXA)( VG_(args_for_client), i ))
440                         + 1;
441        }
442     
443        /* ...and the environment */
444        envc = 0;
445        for (cpp = orig_envp; cpp && *cpp; cpp++) {
446           envc++;
447           stringsize += VG_(strlen)(*cpp) + 1;
448        }
449     
450        /* now, how big is the auxv? */
451        auxsize = sizeof(*auxv);	/* there's always at least one entry: AT_NULL */
452        for (cauxv = orig_auxv; cauxv->a_type != AT_NULL; cauxv++) {
453           if (cauxv->a_type == AT_PLATFORM ||
454               cauxv->a_type == AT_BASE_PLATFORM)
455     	 stringsize += VG_(strlen)(cauxv->u.a_ptr) + 1;
456           else if (cauxv->a_type == AT_RANDOM)
457     	 stringsize += 16;
458           else if (cauxv->a_type == AT_EXECFN)
459     	 stringsize += VG_(strlen)(VG_(args_the_exename)) + 1;
460           auxsize += sizeof(*cauxv);
461        }
462     
463     #  if defined(VGP_ppc32_linux) || defined(VGP_ppc64_linux)
464        auxsize += 2 * sizeof(*cauxv);
465     #  endif
466     
467        /* OK, now we know how big the client stack is */
468        stacksize =
469           sizeof(Word) +                          /* argc */
470           (have_exename ? sizeof(char **) : 0) +  /* argc[0] == exename */
471           sizeof(char **)*argc +                  /* argv */
472           sizeof(char **) +	                      /* terminal NULL */
473           sizeof(char **)*envc +                  /* envp */
474           sizeof(char **) +	                      /* terminal NULL */
475           auxsize +                               /* auxv */
476           VG_ROUNDUP(stringsize, sizeof(Word));   /* strings (aligned) */
477     
478        if (0) VG_(printf)("stacksize = %d\n", stacksize);
479     
480        /* client_SP is the client's stack pointer */
481        client_SP = clstack_end - stacksize;
482        client_SP = VG_ROUNDDN(client_SP, 16); /* make stack 16 byte aligned */
483     
484        /* base of the string table (aligned) */
485        stringbase = strtab = (char *)clstack_end 
486                              - VG_ROUNDUP(stringsize, sizeof(int));
487     
488        clstack_start = VG_PGROUNDDN(client_SP);
489     
490        /* The max stack size */
491        clstack_max_size = VG_PGROUNDUP(clstack_max_size);
492     
493        /* Record stack extent -- needed for stack-change code. */
494        VG_(clstk_base) = clstack_start;
495        VG_(clstk_end)  = clstack_end;
496     
497        if (0)
498           VG_(printf)("stringsize=%d auxsize=%d stacksize=%d maxsize=0x%x\n"
499                       "clstack_start %p\n"
500                       "clstack_end   %p\n",
501     	          stringsize, auxsize, stacksize, (Int)clstack_max_size,
502                       (void*)clstack_start, (void*)clstack_end);
503     
504        /* ==================== allocate space ==================== */
505     
506        { SizeT anon_size   = clstack_end - clstack_start + 1;
507          SizeT resvn_size  = clstack_max_size - anon_size;
508          Addr  anon_start  = clstack_start;
509          Addr  resvn_start = anon_start - resvn_size;
510          SizeT inner_HACK  = 0;
511          Bool  ok;
512     
513          /* So far we've only accounted for space requirements down to the
514             stack pointer.  If this target's ABI requires a redzone below
515             the stack pointer, we need to allocate an extra page, to
516             handle the worst case in which the stack pointer is almost at
517             the bottom of a page, and so there is insufficient room left
518             over to put the redzone in.  In this case the simple thing to
519             do is allocate an extra page, by shrinking the reservation by
520             one page and growing the anonymous area by a corresponding
521             page. */
522          vg_assert(VG_STACK_REDZONE_SZB >= 0);
523          vg_assert(VG_STACK_REDZONE_SZB < VKI_PAGE_SIZE);
524          if (VG_STACK_REDZONE_SZB > 0) {
525             vg_assert(resvn_size > VKI_PAGE_SIZE);
526             resvn_size -= VKI_PAGE_SIZE;
527             anon_start -= VKI_PAGE_SIZE;
528             anon_size += VKI_PAGE_SIZE;
529          }
530     
531          vg_assert(VG_IS_PAGE_ALIGNED(anon_size));
532          vg_assert(VG_IS_PAGE_ALIGNED(resvn_size));
533          vg_assert(VG_IS_PAGE_ALIGNED(anon_start));
534          vg_assert(VG_IS_PAGE_ALIGNED(resvn_start));
535          vg_assert(resvn_start == clstack_end + 1 - clstack_max_size);
536     
537     #    ifdef ENABLE_INNER
538          inner_HACK = 1024*1024; // create 1M non-fault-extending stack
539     #    endif
540     
541          if (0)
542             VG_(printf)("%#lx 0x%lx  %#lx 0x%lx\n",
543                         resvn_start, resvn_size, anon_start, anon_size);
544     
545          /* Create a shrinkable reservation followed by an anonymous
546             segment.  Together these constitute a growdown stack. */
547          res = VG_(mk_SysRes_Error)(0);
548          ok = VG_(am_create_reservation)(
549                  resvn_start,
550                  resvn_size -inner_HACK,
551                  SmUpper, 
552                  anon_size +inner_HACK
553               );
554          if (ok) {
555             /* allocate a stack - mmap enough space for the stack */
556             res = VG_(am_mmap_anon_fixed_client)(
557                      anon_start -inner_HACK,
558                      anon_size +inner_HACK,
559     	         VKI_PROT_READ|VKI_PROT_WRITE|VKI_PROT_EXEC
560     	      );
561          }
562          if ((!ok) || sr_isError(res)) {
563             /* Allocation of the stack failed.  We have to stop. */
564             VG_(printf)("valgrind: "
565                         "I failed to allocate space for the application's stack.\n");
566             VG_(printf)("valgrind: "
567                         "This may be the result of a very large --main-stacksize=\n");
568             VG_(printf)("valgrind: setting.  Cannot continue.  Sorry.\n\n");
569             VG_(exit)(0);
570          }
571     
572          vg_assert(ok);
573          vg_assert(!sr_isError(res)); 
574        }
575     
576        /* ==================== create client stack ==================== */
577     
578        ptr = (Addr*)client_SP;
579     
580        /* --- client argc --- */
581        *ptr++ = argc + (have_exename ? 1 : 0);
582     
583        /* --- client argv --- */
584        if (info->interp_name) {
585           *ptr++ = (Addr)copy_str(&strtab, info->interp_name);
586           VG_(free)(info->interp_name);
587        }
588        if (info->interp_args) {
589           *ptr++ = (Addr)copy_str(&strtab, info->interp_args);
590           VG_(free)(info->interp_args);
591        }
592     
593        if (have_exename)
594           *ptr++ = (Addr)copy_str(&strtab, VG_(args_the_exename));
595     
596        for (i = 0; i < VG_(sizeXA)( VG_(args_for_client) ); i++) {
597           *ptr++ = (Addr)copy_str(
598                            &strtab, 
599                            * (HChar**) VG_(indexXA)( VG_(args_for_client), i )
600                          );
601        }
602        *ptr++ = 0;
603     
604        /* --- envp --- */
605        VG_(client_envp) = (Char **)ptr;
606        for (cpp = orig_envp; cpp && *cpp; ptr++, cpp++)
607           *ptr = (Addr)copy_str(&strtab, *cpp);
608        *ptr++ = 0;
609     
610        /* --- auxv --- */
611        auxv = (struct auxv *)ptr;
612        *client_auxv = (UInt *)auxv;
613     
614     #  if defined(VGP_ppc32_linux) || defined(VGP_ppc64_linux)
615        auxv[0].a_type  = AT_IGNOREPPC;
616        auxv[0].u.a_val = AT_IGNOREPPC;
617        auxv[1].a_type  = AT_IGNOREPPC;
618        auxv[1].u.a_val = AT_IGNOREPPC;
619        auxv += 2;
620     #  endif
621     
622        for (; orig_auxv->a_type != AT_NULL; auxv++, orig_auxv++) {
623     
624           /* copy the entry... */
625           *auxv = *orig_auxv;
626     
627           /* ...and fix up / examine the copy */
628           switch(auxv->a_type) {
629     
630              case AT_IGNORE:
631              case AT_PHENT:
632              case AT_PAGESZ:
633              case AT_FLAGS:
634              case AT_NOTELF:
635              case AT_UID:
636              case AT_EUID:
637              case AT_GID:
638              case AT_EGID:
639              case AT_CLKTCK:
640              case AT_FPUCW:
641                 /* All these are pointerless, so we don't need to do
642                    anything about them. */
643                 break;
644     
645              case AT_PHDR:
646                 if (info->phdr == 0)
647                    auxv->a_type = AT_IGNORE;
648                 else
649                    auxv->u.a_val = info->phdr;
650                 break;
651     
652              case AT_PHNUM:
653                 if (info->phdr == 0)
654                    auxv->a_type = AT_IGNORE;
655                 else
656                    auxv->u.a_val = info->phnum;
657                 break;
658     
659              case AT_BASE:
660                 auxv->u.a_val = info->interp_base;
661                 break;
662     
663              case AT_PLATFORM:
664              case AT_BASE_PLATFORM:
665                 /* points to a platform description string */
666                 auxv->u.a_ptr = copy_str(&strtab, orig_auxv->u.a_ptr);
667                 break;
668     
669              case AT_ENTRY:
670                 auxv->u.a_val = info->entry;
671                 break;
672     
673              case AT_HWCAP:
674     #           if defined(VGP_arm_linux)
675                 { Bool has_neon = (auxv->u.a_val & VKI_HWCAP_NEON) > 0;
676                   VG_(debugLog)(2, "initimg",
677                                    "ARM has-neon from-auxv: %s\n",
678                                    has_neon ? "YES" : "NO");
679                   VG_(machine_arm_set_has_NEON)( has_neon );
680                 }
681     #           endif
682                 break;
683     
684              case AT_DCACHEBSIZE:
685              case AT_ICACHEBSIZE:
686              case AT_UCACHEBSIZE:
687     #           if defined(VGP_ppc32_linux)
688                 /* acquire cache info */
689                 if (auxv->u.a_val > 0) {
690                    VG_(machine_ppc32_set_clszB)( auxv->u.a_val );
691                    VG_(debugLog)(2, "initimg", 
692                                     "PPC32 cache line size %u (type %u)\n", 
693                                     (UInt)auxv->u.a_val, (UInt)auxv->a_type );
694                 }
695     #           elif defined(VGP_ppc64_linux)
696                 /* acquire cache info */
697                 if (auxv->u.a_val > 0) {
698                    VG_(machine_ppc64_set_clszB)( auxv->u.a_val );
699                    VG_(debugLog)(2, "initimg", 
700                                     "PPC64 cache line size %u (type %u)\n", 
701                                     (UInt)auxv->u.a_val, (UInt)auxv->a_type );
702                 }
703     #           endif
704                 break;
705     
706     #        if defined(VGP_ppc32_linux) || defined(VGP_ppc64_linux)
707              case AT_IGNOREPPC:
708                 break;
709     #        endif
710     
711              case AT_SECURE:
712                 /* If this is 1, then it means that this program is
713                    running suid, and therefore the dynamic linker should
714                    be careful about LD_PRELOAD, etc.  However, since
715                    stage1 (the thing the kernel actually execve's) should
716                    never be SUID, and we need LD_PRELOAD to work for the
717                    client, we set AT_SECURE to 0. */
718                 auxv->u.a_val = 0;
719                 break;
720     
721              case AT_SYSINFO:
722                 /* Trash this, because we don't reproduce it */
723                 auxv->a_type = AT_IGNORE;
724                 break;
725     
726     #        if !defined(VGP_ppc32_linux) && !defined(VGP_ppc64_linux)
727              case AT_SYSINFO_EHDR: {
728                 /* Trash this, because we don't reproduce it */
729                 const NSegment* ehdrseg = VG_(am_find_nsegment)((Addr)auxv->u.a_ptr);
730                 vg_assert(ehdrseg);
731                 VG_(am_munmap_valgrind)(ehdrseg->start, ehdrseg->end - ehdrseg->start);
732                 auxv->a_type = AT_IGNORE;
733                 break;
734              }
735     #        endif
736     
737              case AT_RANDOM:
738                 /* points to 16 random bytes - we need to ensure this is
739                    propagated to the client as glibc will assume it is
740                    present if it is built for kernel 2.6.29 or later */
741                 auxv->u.a_ptr = strtab;
742                 VG_(memcpy)(strtab, orig_auxv->u.a_ptr, 16);
743                 strtab += 16;
744                 break;
745     
746              case AT_EXECFN:
747                 /* points to the executable filename */
748                 auxv->u.a_ptr = copy_str(&strtab, VG_(args_the_exename));
749                 break;
750     
751              default:
752                 /* stomp out anything we don't know about */
753                 VG_(debugLog)(2, "initimg",
754                                  "stomping auxv entry %lld\n", 
755                                  (ULong)auxv->a_type);
756                 auxv->a_type = AT_IGNORE;
757                 break;
758           }
759        }
760        *auxv = *orig_auxv;
761        vg_assert(auxv->a_type == AT_NULL);
762     
763        vg_assert((strtab-stringbase) == stringsize);
764     
765        /* client_SP is pointing at client's argc/argv */
766     
767        if (0) VG_(printf)("startup SP = %#lx\n", client_SP);
768        return client_SP;
769     }

772     /* Allocate the client data segment.  It is an expandable anonymous
773        mapping abutting a shrinkable reservation of size max_dseg_size.
774        The data segment starts at VG_(brk_base), which is page-aligned,
775        and runs up to VG_(brk_limit), which isn't. */
776     
777     static void setup_client_dataseg ( SizeT max_size )
778     {
779        Bool   ok;
780        SysRes sres;
781        Addr   anon_start  = VG_(brk_base);
782        SizeT  anon_size   = VKI_PAGE_SIZE;
783        Addr   resvn_start = anon_start + anon_size;
784        SizeT  resvn_size  = max_size - anon_size;
785     
786        vg_assert(VG_IS_PAGE_ALIGNED(anon_size));
787        vg_assert(VG_IS_PAGE_ALIGNED(resvn_size));
788        vg_assert(VG_IS_PAGE_ALIGNED(anon_start));
789        vg_assert(VG_IS_PAGE_ALIGNED(resvn_start));
790     
791        /* Because there's been no brk activity yet: */
792        vg_assert(VG_(brk_base) == VG_(brk_limit));
793     
794        /* Try to create the data seg and associated reservation where
795           VG_(brk_base) says. */
796        ok = VG_(am_create_reservation)( 
797                resvn_start, 
798                resvn_size, 
799                SmLower, 
800                anon_size
801             );
802     
803        if (!ok) {
804           /* Hmm, that didn't work.  Well, let aspacem suggest an address
805              it likes better, and try again with that. */
806           anon_start = VG_(am_get_advisory_client_simple)
807                           ( 0/*floating*/, anon_size+resvn_size, &ok );
808           if (ok) {
809              resvn_start = anon_start + anon_size;
810              ok = VG_(am_create_reservation)( 
811                      resvn_start, 
812                      resvn_size, 
813                      SmLower, 
814                      anon_size
815                   );
816              if (ok)
817                 VG_(brk_base) = VG_(brk_limit) = anon_start;
818           }
819           /* that too might have failed, but if it has, we're hosed: there
820              is no Plan C. */
821        }
822        vg_assert(ok);
823     
824        /* We make the data segment (heap) executable because LinuxThreads on
825           ppc32 creates trampolines in this area.  Also, on x86/Linux the data
826           segment is RWX natively, at least according to /proc/self/maps.
827           Also, having a non-executable data seg would kill any program which
828           tried to create code in the data seg and then run it. */
829        sres = VG_(am_mmap_anon_fixed_client)( 
830                  anon_start, 
831                  anon_size, 
832                  VKI_PROT_READ|VKI_PROT_WRITE|VKI_PROT_EXEC
833               );
834        vg_assert(!sr_isError(sres));
835        vg_assert(sr_Res(sres) == anon_start);
836     }

843     /* Create the client's initial memory image. */
844     IIFinaliseImageInfo VG_(ii_create_image)( IICreateImageInfo iicii )
845     {
846        ExeInfo info;
847        HChar** env = NULL;
848     
849        IIFinaliseImageInfo iifii;
850        VG_(memset)( &iifii, 0, sizeof(iifii) );
851     
852        //--------------------------------------------------------------
853        // Load client executable, finding in $PATH if necessary
854        //   p: get_helprequest_and_toolname()  [for 'exec', 'need_help']
855        //   p: layout_remaining_space          [so there's space]
856        //--------------------------------------------------------------
857        VG_(debugLog)(1, "initimg", "Loading client\n");
858     
859        if (VG_(args_the_exename) == NULL)
860           VG_(err_missing_prog)();
861     
862        load_client(&info, &iifii.initial_client_IP, &iifii.initial_client_TOC);
863     
864        //--------------------------------------------------------------
865        // Set up client's environment
866        //   p: set-libdir                   [for VG_(libdir)]
867        //   p: get_helprequest_and_toolname [for toolname]
868        //--------------------------------------------------------------
869        VG_(debugLog)(1, "initimg", "Setup client env\n");
870        env = setup_client_env(iicii.envp, iicii.toolname);
871     
872        //--------------------------------------------------------------
873        // Setup client stack, eip, and VG_(client_arg[cv])
874        //   p: load_client()     [for 'info']
875        //   p: fix_environment() [for 'env']
876        //--------------------------------------------------------------
877        {
878           /* When allocating space for the client stack on Linux, take
879              notice of the --main-stacksize value.  This makes it possible
880              to run programs with very large (primary) stack requirements
881              simply by specifying --main-stacksize. */
882           /* Logic is as follows:
883              - by default, use the client's current stack rlimit
884              - if that exceeds 16M, clamp to 16M
885              - if a larger --main-stacksize value is specified, use that instead
886              - in all situations, the minimum allowed stack size is 1M
887           */
888           void* init_sp = iicii.argv - 1;
889           SizeT m1  = 1024 * 1024;
890           SizeT m16 = 16 * m1;
891           SizeT szB = (SizeT)VG_(client_rlimit_stack).rlim_cur;
892           if (szB < m1) szB = m1;
893           if (szB > m16) szB = m16;
894           if (VG_(clo_main_stacksize) > 0) szB = VG_(clo_main_stacksize);
895           if (szB < m1) szB = m1;
896           szB = VG_PGROUNDUP(szB);
897           VG_(debugLog)(1, "initimg",
898                            "Setup client stack: size will be %ld\n", szB);
899     
900           iifii.clstack_max_size = szB;
901     
902           iifii.initial_client_SP
903              = setup_client_stack( init_sp, env, 
904                                    &info, &iifii.client_auxv, 
905                                    iicii.clstack_top, iifii.clstack_max_size );
906     
907           VG_(free)(env);
908     
909           VG_(debugLog)(2, "initimg",
910                            "Client info: "
911                            "initial_IP=%p initial_TOC=%p brk_base=%p\n",
912                            (void*)(iifii.initial_client_IP), 
913                            (void*)(iifii.initial_client_TOC),
914                            (void*)VG_(brk_base) );
915           VG_(debugLog)(2, "initimg",
916                            "Client info: "
917                            "initial_SP=%p max_stack_size=%ld\n",
918                            (void*)(iifii.initial_client_SP),
919                            (SizeT)iifii.clstack_max_size );
920        }
921     
922        //--------------------------------------------------------------
923        // Setup client data (brk) segment.  Initially a 1-page segment
924        // which abuts a shrinkable reservation. 
925        //     p: load_client()     [for 'info' and hence VG_(brk_base)]
926        //--------------------------------------------------------------
927        { 
928           SizeT m1 = 1024 * 1024;
929           SizeT m8 = 8 * m1;
930           SizeT dseg_max_size = (SizeT)VG_(client_rlimit_data).rlim_cur;
931           VG_(debugLog)(1, "initimg", "Setup client data (brk) segment\n");
932           if (dseg_max_size < m1) dseg_max_size = m1;
933           if (dseg_max_size > m8) dseg_max_size = m8;
934           dseg_max_size = VG_PGROUNDUP(dseg_max_size);
935     
936           setup_client_dataseg( dseg_max_size );
937        }
938     
939        return iifii;
940     }

947     /* Just before starting the client, we may need to make final
948        adjustments to its initial image.  Also we need to set up the VEX
949        guest state for thread 1 (the root thread) and copy in essential
950        starting values.  This is handed the IIFinaliseImageInfo created by
951        VG_(ii_create_image).
952     */
953     void VG_(ii_finalise_image)( IIFinaliseImageInfo iifii )
954     {
955        ThreadArchState* arch = &VG_(threads)[1].arch;
956     
957        /* On Linux we get client_{ip/sp/toc}, and start the client with
958           all other registers zeroed. */
959     
960     #  if defined(VGP_x86_linux)
961        vg_assert(0 == sizeof(VexGuestX86State) % 16);
962     
963        /* Zero out the initial state, and set up the simulated FPU in a
964           sane way. */
965        LibVEX_GuestX86_initialise(&arch->vex);
966     
967        /* Zero out the shadow areas. */
968        VG_(memset)(&arch->vex_shadow1, 0, sizeof(VexGuestX86State));
969        VG_(memset)(&arch->vex_shadow2, 0, sizeof(VexGuestX86State));
970     
971        /* Put essential stuff into the new state. */
972        arch->vex.guest_ESP = iifii.initial_client_SP;
973        arch->vex.guest_EIP = iifii.initial_client_IP;
974     
975        /* initialise %cs, %ds and %ss to point at the operating systems
976           default code, data and stack segments */
977        asm volatile("movw %%cs, %0" : : "m" (arch->vex.guest_CS));
978        asm volatile("movw %%ds, %0" : : "m" (arch->vex.guest_DS));
979        asm volatile("movw %%ss, %0" : : "m" (arch->vex.guest_SS));
980     
981     #  elif defined(VGP_amd64_linux)
982        vg_assert(0 == sizeof(VexGuestAMD64State) % 16);
983     
984        /* Zero out the initial state, and set up the simulated FPU in a
985           sane way. */
986        LibVEX_GuestAMD64_initialise(&arch->vex);
987     
988        /* Zero out the shadow areas. */
989        VG_(memset)(&arch->vex_shadow1, 0, sizeof(VexGuestAMD64State));
990        VG_(memset)(&arch->vex_shadow2, 0, sizeof(VexGuestAMD64State));
991     
992        /* Put essential stuff into the new state. */
993        arch->vex.guest_RSP = iifii.initial_client_SP;
994        arch->vex.guest_RIP = iifii.initial_client_IP;
995     
996     #  elif defined(VGP_ppc32_linux)
997        vg_assert(0 == sizeof(VexGuestPPC32State) % 16);
998     
999        /* Zero out the initial state, and set up the simulated FPU in a
1000           sane way. */
1001        LibVEX_GuestPPC32_initialise(&arch->vex);
1002     
1003        /* Zero out the shadow areas. */
1004        VG_(memset)(&arch->vex_shadow1, 0, sizeof(VexGuestPPC32State));
1005        VG_(memset)(&arch->vex_shadow2, 0, sizeof(VexGuestPPC32State));
1006     
1007        /* Put essential stuff into the new state. */
1008        arch->vex.guest_GPR1 = iifii.initial_client_SP;
1009        arch->vex.guest_CIA  = iifii.initial_client_IP;
1010     
1011     #  elif defined(VGP_ppc64_linux)
1012        vg_assert(0 == sizeof(VexGuestPPC64State) % 16);
1013     
1014        /* Zero out the initial state, and set up the simulated FPU in a
1015           sane way. */
1016        LibVEX_GuestPPC64_initialise(&arch->vex);
1017     
1018        /* Zero out the shadow areas. */
1019        VG_(memset)(&arch->vex_shadow1, 0, sizeof(VexGuestPPC64State));
1020        VG_(memset)(&arch->vex_shadow2, 0, sizeof(VexGuestPPC64State));
1021     
1022        /* Put essential stuff into the new state. */
1023        arch->vex.guest_GPR1 = iifii.initial_client_SP;
1024        arch->vex.guest_GPR2 = iifii.initial_client_TOC;
1025        arch->vex.guest_CIA  = iifii.initial_client_IP;
1026     
1027     #   elif defined(VGP_arm_linux)
1028        /* Zero out the initial state, and set up the simulated FPU in a
1029           sane way. */
1030        LibVEX_GuestARM_initialise(&arch->vex);
1031     
1032        /* Zero out the shadow areas. */
1033        VG_(memset)(&arch->vex_shadow1, 0, sizeof(VexGuestARMState));
1034        VG_(memset)(&arch->vex_shadow2, 0, sizeof(VexGuestARMState));
1035     
1036        arch->vex.guest_R13  = iifii.initial_client_SP;
1037        arch->vex.guest_R15T = iifii.initial_client_IP;
1038     
1039        /* This is just EABI stuff. */
1040        // FIXME jrs: what's this for?
1041        arch->vex.guest_R1 =  iifii.initial_client_SP;
1042     
1043     #  elif defined(VGP_s390x_linux)
1044        vg_assert(0 == sizeof(VexGuestS390XState) % 16);
1045     
1046        /* Zero out the initial state. This also sets the guest_fpc to 0, which
1047           is also done by the kernel for the fpc during execve. */
1048        LibVEX_GuestS390X_initialise(&arch->vex);
1049     
1050        /* Zero out the shadow area. */
1051        VG_(memset)(&arch->vex_shadow1, 0, sizeof(VexGuestS390XState));
1052        VG_(memset)(&arch->vex_shadow2, 0, sizeof(VexGuestS390XState));
1053     
1054        /* Put essential stuff into the new state. */
1055        arch->vex.guest_SP = iifii.initial_client_SP;
1056        arch->vex.guest_IA = iifii.initial_client_IP;
1057     
1058     #  else
1059     #    error Unknown platform
1060     #  endif
1061     
1062        /* Tell the tool that we just wrote to the registers. */
1063        VG_TRACK( post_reg_write, Vg_CoreStartup, /*tid*/1, /*offset*/0,
1064                  sizeof(VexGuestArchState));
1065     }
