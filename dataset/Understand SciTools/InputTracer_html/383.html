29     #ifndef __IT_TREAP_H
30     #define __IT_TREAP_H
31     
32     #include "it_include.h"
33     
34     typedef TaintLabel TreapValueType;
35     typedef UShort TreapNodeIndex;
36     
37     #define TREAP_INVALID_REF 0xFFFF
38     
39     #define TREAP_FETCH(p_treap, index) (p_treap->data.array[index])
40     #define TREAP_GET_PRIO(index) (treap_randomness[index])
41     
42     typedef struct
43     {
44        TreapValueType value;
45        TreapNodeIndex left;
46        TreapNodeIndex right;
47     } TreapNode;
48     
49     typedef struct
50     {
51        UInt current_size;           // Size of internal storage array
52        TreapNodeIndex current_idx;  // Next _free_ index in storage array
53        TreapNodeIndex root;         // Index in the array of the root element
54        TreapNode* array;            // Pointer to the storage array
55     } Treap;
56     
57     inline void IT_(init_treap)(Treap* t);
58     inline void IT_(destroy_treap)(Treap* t);
59     void IT_(treap_add_node)(Treap* treap, TreapValueType value);
60     
61     #endif /* __IT_TREAP_H */
