46     #if defined(PLAT_amd64_linux) || defined(PLAT_x86_linux) \
47         || defined(PLAT_amd64_darwin) || defined(PLAT_x86_darwin)
48     #  define XCHG_M_R(_addr,_lval) \
49          __asm__ __volatile__( \
50             "xchgl %0, %1" \
51             : /*out*/ "+r"(_lval) \
52             : /*in*/  "m"(_addr) \
53             : "memory", "cc" \
54          )
55     #  define XCHG_M_R_with_redundant_LOCK(_addr,_lval) \
56          __asm__ __volatile__( \
57             "lock xchgl %0, %1" \
58             : /*out*/ "+r"(_lval) \
59             : /*in*/  "m"(_addr) \
60             : "memory", "cc" \
61          )
62     
63     #elif defined(PLAT_ppc32_linux) || defined(PLAT_ppc64_linux) \
64           || defined(PLAT_ppc32_aix5) || defined(PLAT_ppc64_aix5) \
65           || defined(PLAT_arm_linux) || defined(PLAT_s390x_linux)
66     #  if defined(HAVE_BUILTIN_ATOMIC)
67     #    define XCHG_M_R(_addr,_lval)                                           \
68             do {                                                                \
69               int tmp;                                                          \
70               while ((tmp = *(int*)(& _addr)),                                  \
71                      ! __sync_bool_compare_and_swap((int*)&_addr, tmp, _lval))  \
72                 ;                                                               \
73               _lval = tmp;                                                      \
74             } while (0)
75     #  else
76     #    warning "XCHG_M_R() implementation is missing. Either" \
77                  "provide one or use a newer gcc version."
78     #    define XCHG_M_R(_addr,_lval) \
79             do { int tmp = *(int*)(& _addr); \
80                  *(int*)(& _addr) = (_lval); \
81                  _lval = tmp; \
82             } while (0)
83     #  endif
84     #  define XCHG_M_R_with_redundant_LOCK(_addr,_lval) \
85           XCHG_M_R(_addr,_lval)
86

92     int x = 0;

94     void* child_fn ( void* arg )
95     {
96        int v = 12345;
97        XCHG_M_R_with_redundant_LOCK(x, v);
98        assert(v == 0 || v == 6789);
99        return NULL;
100     }

102     int main ( void )
103     {
104        int v = 6789;
105        pthread_t child;
106     
107        if (pthread_create(&child, NULL, child_fn, NULL)) {
108           perror("pthread_create");
109           exit(1);
110        }
111     
112        XCHG_M_R(x, v);
113        assert(v == 0 || v == 12345);
114     
115        if (pthread_join(child, NULL)) {
116           perror("pthread join");
117           exit(1);
118        }
119
