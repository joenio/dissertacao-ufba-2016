69     /* Stabs entry types, from:
70      *   The "stabs" debug format
71      *   Menapace, Kingdon and MacKenzie
72      *   Cygnus Support
73      */
74     typedef enum { N_UNDEF = 0,	/* undefined symbol, new stringtab  */
75     	       N_GSYM  = 32,    /* Global symbol                    */
76                    N_FUN   = 36,    /* Function start or end            */
77                    N_STSYM = 38,    /* Data segment file-scope variable */
78                    N_LCSYM = 40,    /* BSS segment file-scope variable  */
79                    N_RSYM  = 64,    /* Register variable                */
80                    N_SLINE = 68,    /* Source line number               */
81                    N_SO    = 100,   /* Source file path and name        */
82                    N_LSYM  = 128,   /* Stack variable or type           */
83     	       N_BINCL = 130,	/* Beginning of an include file	    */
84                    N_SOL   = 132,   /* Include file name                */
85     	       N_PSYM  = 160,   /* Function parameter               */
86     	       N_EINCL = 162,	/* End of an include file           */
87                    N_LBRAC = 192,   /* Start of lexical block           */
88     	       N_EXCL  = 194,	/* Placeholder for an include file  */
89                    N_RBRAC = 224    /* End   of lexical block           */
90                  } stab_types;
91           
92     
93     /* Read stabs-format debug info.  This is all rather horrible because
94        stabs is a underspecified, kludgy hack.
95     */
96     void ML_(read_debuginfo_stabs) ( DebugInfo* di,
97                                      UChar* stabC,   Int stab_sz, 
98                                      UChar* stabstr, Int stabstr_sz )
99     {
100        const Bool debug     = False;
101        const Bool contdebug = False;
102        Int    i;
103        Int    n_stab_entries;
104        struct nlist* stab = (struct nlist*)stabC;
105        UChar *next_stabstr = NULL;
106        /* state for various things */
107        struct {
108           Addr     start;         /* start address */
109           Addr     end;           /* end address */
110           Int      line;          /* first line */
111        } func = { 0, 0, -1 };
112        struct {
113           Char     *name;
114           Bool     same;
115        } file = { NULL, True };
116        struct {
117           Int      prev;          /* prev line */
118           Int      no;            /* current line */
119           Int      ovf;           /* line wrap */
120           Addr     addr;          /* start of this line */
121           Bool     first;         /* first line in function */
122        } line = { 0, 0, 0, 0, False };
123     
124        /* Ok.  It all looks plausible.  Go on and read debug data. 
125              stab kinds: 100   N_SO     a source file name
126                           68   N_SLINE  a source line number
127                           36   N_FUN    start of a function
128     
129           In this loop, we maintain a current file name, updated as 
130           N_SO/N_SOLs appear, and a current function base address, 
131           updated as N_FUNs appear.  Based on that, address ranges for 
132           N_SLINEs are calculated, and stuffed into the line info table.
133     
134           Finding the instruction address range covered by an N_SLINE is
135           complicated;  see the N_SLINE case below.
136        */
137        file.name     = ML_(addStr)(di,"???", -1);
138     
139        n_stab_entries = stab_sz/(int)sizeof(struct nlist);
140     
141        for (i = 0; i < n_stab_entries; i++) {
142           const struct nlist *st = &stab[i];
143           Char *string;
144     
145           if (di->trace_symtab) {
146              VG_(printf) ( "%2d  type=%d   othr=%d   desc=%d   "
147                            "value=0x%x   strx=%d  %s\n", i,
148                            st->n_type, st->n_other, st->n_desc, 
149                            (Int)st->n_value,
150                            (Int)st->n_un.n_strx, 
151                            stabstr + st->n_un.n_strx );
152           }
153     
154           /* handle continued string stabs */
155           {
156              Int   qbuflen = 0;
157              Int   qidx = 0;
158              Char* qbuf = NULL;
159              Int   qlen;
160              Bool  qcontinuing = False;
161              UInt  qstringidx;
162     
163              qstringidx = st->n_un.n_strx;
164              string = stabstr + qstringidx;
165              qlen = VG_(strlen)(string);
166     
167              while (string 
168                     && qlen > 0 
169                     && (qcontinuing || string[qlen-1] == '\\')) {
170                 /* Gak, we have a continuation. Skip forward through
171                    subsequent stabs to gather all the parts of the
172                    continuation.  Increment i, but keep st pointing at
173                    current stab. */
174     
175                 qcontinuing = string[qlen-1] == '\\';
176     
177                 /* remove trailing \ */
178                 while (string[qlen-1] == '\\' && qlen > 0)
179                    qlen--;
180     
181                 if (contdebug)
182                    VG_(printf)("found extension string: \"%s\" "
183                                "len=%d(%c) idx=%d buflen=%d\n", 
184                                string, qlen, string[qlen-1], qidx, qbuflen);
185     
186                 /* XXX this is silly.  The si->strtab should have a way of
187                    appending to the last added string... */
188                 if ((qidx + qlen) >= qbuflen) {
189                    Char *n;
190                    
191                    if (qbuflen == 0)
192                       qbuflen = 16;
193                    while ((qidx + qlen) >= qbuflen)
194                       qbuflen *= 2;
195                    n = ML_(dinfo_zalloc)("di.readstabs.rds.1", qbuflen);
196                    VG_(memcpy)(n, qbuf, qidx);
197                    
198                    if (qbuf != NULL)
199                       ML_(dinfo_free)(qbuf);
200                    qbuf = n;
201                 }
202     
203                 VG_(memcpy)(&qbuf[qidx], string, qlen);
204                 qidx += qlen;
205                 if (contdebug) {
206                    qbuf[qidx] = '\0';
207                    VG_(printf)("working buf=\"%s\"\n", qbuf);
208                 }
209     
210                 i++;
211                 if (i >= n_stab_entries)
212                    break;
213     
214                 if (stab[i].n_un.n_strx) {
215                    string = stabstr + stab[i].n_un.n_strx;
216                    qlen = VG_(strlen)(string);
217                 } else {
218                    string = NULL;
219                    qlen = 0;
220                 }
221              }
222     
223              if (qbuf != NULL) {
224                 i--;                        /* overstepped */
225                 string = ML_(addStr)(di, qbuf, qidx);
226                 ML_(dinfo_free)(qbuf);
227                 if (contdebug)
228                    VG_(printf)("made composite: \"%s\"\n", string);
229              }
230           }
231     
232           switch(st->n_type) {
233              case N_UNDEF:
234                 /* new string table base */
235                 if (next_stabstr != NULL) {
236                    stabstr_sz -= next_stabstr - stabstr;
237                    stabstr = next_stabstr;
238                    if (stabstr_sz <= 0) {
239                       VG_(printf)(" @@ bad stabstr size %d\n", stabstr_sz);
240                       return;
241                    }
242                 }
243                 next_stabstr = stabstr + st->n_value;
244                 break;
245     
246              case N_BINCL: {
247                 break;
248              }
249     
250              case N_EINCL:
251                 break;
252     
253              case N_EXCL:
254                 break;
255     
256              case N_SOL:                /* sub-source (include) file */
257                 if (line.ovf != 0) 
258                    VG_(message)(Vg_UserMsg, 
259                                 "Warning: file %s is very big (> 65535 lines) "
260                                 "Line numbers and annotation for this file might "
261                                 "be wrong.  Sorry.\n",
262                                 file.name);
263                 /* FALLTHROUGH */
264     
265              case N_SO: {                /* new source file */
266                 UChar *nm = string;
267                 UInt len = VG_(strlen)(nm);
268                 Addr addr = func.start + st->n_value;
269     
270                 if (line.addr != 0) {
271                    /* finish off previous line */
272                    ML_(addLineInfo)(di, file.name, NULL, line.addr,
273                                     addr, line.no + line.ovf * LINENO_OVERFLOW, i);
274                 }
275     
276                 /* reset line state */
277                 line.ovf = 0;            
278                 line.addr = 0;
279                 line.prev = 0;
280                 line.no = 0;
281     
282                 if (len > 0 && nm[len-1] != '/') {
283                    file.name = ML_(addStr)(di, nm, -1);
284                    if (debug)
285                       VG_(printf)("new source: %s\n", file.name);
286                 } else if (len == 0)
287                    file.name = ML_(addStr)(di, "?1\0", -1);
288     
289                 break;
290              }
291     
292              case N_SLINE: {        /* line info */
293                 Addr addr = func.start + st->n_value;
294     
295                 if (line.addr != 0) {
296                    /* there was a previous */
297                    ML_(addLineInfo)(di, file.name, NULL, line.addr,
298                                     addr, line.no + line.ovf * LINENO_OVERFLOW, i);
299                 }
300     
301                 line.addr = addr;
302                 line.prev = line.no;
303                 line.no = (Int)((UShort)st->n_desc);
304     
305                 if (line.prev > line.no + OVERFLOW_DIFFERENCE && file.same) {
306                    VG_(message)(Vg_DebugMsg, 
307                       "Line number overflow detected (%d --> %d) in %s\n", 
308                       line.prev, line.no, file.name);
309                    line.ovf++;
310                 }
311                 file.same = True;
312     
313                 /* This is pretty horrible.  If this is the first line of
314                    the function, then bind any unbound symbols to the arg
315                    scope, since they're probably arguments. */
316                 if (line.first) {
317                    line.first = False;
318                    
319                    /* remember first line of function */
320                    if (func.start != 0) {
321                       func.line = line.no;
322                    }
323                 }
324                 break;
325              }
326     
327              case N_FUN: {                /* function start/end */
328                 Addr addr = 0;        /* end address for prev line/scope */
329     
330                 /* if this the end of the function or we haven't
331                    previously finished the previous function... */
332                 if (*string == '\0' || func.start != 0) {
333                    /* end of function */
334                    line.first = False;
335     
336                    /* end line at end of function */
337                    addr = func.start + st->n_value;
338     
339                    /* now between functions */
340                    func.start = 0;
341     
342                    // XXXX DEAD POINT XXXX
343                 }
344     
345                 if (*string != '\0') {
346                    /* new function */
347                    line.first = True;
348     
349                    /* line ends at start of next function */
350                    addr = di->text_debug_bias + st->n_value;
351     
352                    func.start = addr;
353                 }
354     
355                 if (line.addr) {
356                    ML_(addLineInfo)(di, file.name, NULL, line.addr,
357                                     addr, line.no + line.ovf * LINENO_OVERFLOW, i);
358                    line.addr = 0;
359                 }
360     
361                 //DEAD POINT
362                 //DEAD POINT
363                 break;
364              }
365     
366              case N_LBRAC: {
367                 /* open new scope */
368                 // DEAD POINT
369                 break;
370              }
371     
372              case N_RBRAC: {
373                 /* close scope */
374                 // DEAD POINT
375                 break;
376              }
377     
378              case N_GSYM:                /* global variable */
379              case N_STSYM:                /* static in data segment */
380              case N_LCSYM:                /* static in bss segment */
381              case N_PSYM:                /* function parameter */
382              case N_LSYM:                /* stack variable */
383              case N_RSYM:                  /* register variable */
384                 break;
385           }
386        }
387     }
