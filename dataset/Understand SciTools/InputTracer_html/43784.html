62     // Run a thread from beginning to end and return the thread's
63     // scheduler-return-code.
64     static VgSchedReturnCode thread_wrapper(Word /*ThreadId*/ tidW)
65     {
66        VgSchedReturnCode ret;
67        ThreadId     tid = (ThreadId)tidW;
68        ThreadState* tst = VG_(get_ThreadState)(tid);
69     
70        VG_(debugLog)(1, "syswrap-linux", 
71                         "thread_wrapper(tid=%lld): entry\n", 
72                         (ULong)tidW);
73     
74        vg_assert(tst->status == VgTs_Init);
75     
76        /* make sure we get the CPU lock before doing anything significant */
77        VG_(acquire_BigLock)(tid, "thread_wrapper(starting new thread)");
78     
79        if (0)
80           VG_(printf)("thread tid %d started: stack = %p\n",
81     		  tid, &tid);
82     
83        VG_TRACK(pre_thread_first_insn, tid);
84     
85        tst->os_state.lwpid = VG_(gettid)();
86        /* Set the threadgroup for real.  This overwrites the provisional
87           value set in do_clone() syswrap-*-linux.c.  See comments in
88           do_clone for background, also #226116. */
89        tst->os_state.threadgroup = VG_(getpid)();
90     
91        /* Thread created with all signals blocked; scheduler will set the
92           appropriate mask */
93     
94        ret = VG_(scheduler)(tid);
95     
96        vg_assert(VG_(is_exiting)(tid));
97        
98        vg_assert(tst->status == VgTs_Runnable);
99        vg_assert(VG_(is_running_thread)(tid));
100     
101        VG_(debugLog)(1, "syswrap-linux", 
102                         "thread_wrapper(tid=%lld): exit\n", 
103                         (ULong)tidW);
104     
105        /* Return to caller, still holding the lock. */
106        return ret;
107     }

114     /* Run a thread all the way to the end, then do appropriate exit actions
115        (this is the last-one-out-turn-off-the-lights bit).  */
116     static void run_a_thread_NORETURN ( Word tidW )
117     {
118        ThreadId          tid = (ThreadId)tidW;
119        VgSchedReturnCode src;
120        Int               c;
121     
122        VG_(debugLog)(1, "syswrap-linux", 
123                         "run_a_thread_NORETURN(tid=%lld): pre-thread_wrapper\n",
124                         (ULong)tidW);
125     
126        /* Run the thread all the way through. */
127        src = thread_wrapper(tid);  
128     
129        VG_(debugLog)(1, "syswrap-linux", 
130                         "run_a_thread_NORETURN(tid=%lld): post-thread_wrapper\n",
131                         (ULong)tidW);
132     
133        c = VG_(count_living_threads)();
134        vg_assert(c >= 1); /* stay sane */
135     
136        // Tell the tool this thread is exiting
137        VG_TRACK( pre_thread_ll_exit, tid );
138     
139        if (c == 1) {
140     
141           VG_(debugLog)(1, "syswrap-linux", 
142                            "run_a_thread_NORETURN(tid=%lld): "
143                               "last one standing\n",
144                               (ULong)tidW);
145     
146           /* We are the last one standing.  Keep hold of the lock and
147              carry on to show final tool results, then exit the entire system. 
148              Use the continuation pointer set at startup in m_main. */
149           ( * VG_(address_of_m_main_shutdown_actions_NORETURN) ) (tid, src);
150     
151        } else {
152     
153           ThreadState *tst;
154     
155           VG_(debugLog)(1, "syswrap-linux", 
156                            "run_a_thread_NORETURN(tid=%lld): "
157                               "not last one standing\n",
158                               (ULong)tidW);
159     
160           /* OK, thread is dead, but others still exist.  Just exit. */
161           tst = VG_(get_ThreadState)(tid);
162     
163           /* This releases the run lock */
164           VG_(exit_thread)(tid);
165           vg_assert(tst->status == VgTs_Zombie);
166     
167           /* We have to use this sequence to terminate the thread to
168              prevent a subtle race.  If VG_(exit_thread)() had left the
169              ThreadState as Empty, then it could have been reallocated,
170              reusing the stack while we're doing these last cleanups.
171              Instead, VG_(exit_thread) leaves it as Zombie to prevent
172              reallocation.  We need to make sure we don't touch the stack
173              between marking it Empty and exiting.  Hence the
174              assembler. */
175     #if defined(VGP_x86_linux)
176           asm volatile (
177              "movl	%1, %0\n"	/* set tst->status = VgTs_Empty */
178              "movl	%2, %%eax\n"    /* set %eax = __NR_exit */
179              "movl	%3, %%ebx\n"    /* set %ebx = tst->os_state.exitcode */
180              "int	$0x80\n"	/* exit(tst->os_state.exitcode) */
181              : "=m" (tst->status)
182              : "n" (VgTs_Empty), "n" (__NR_exit), "m" (tst->os_state.exitcode));
183     #elif defined(VGP_amd64_linux)
184           asm volatile (
185              "movl	%1, %0\n"	/* set tst->status = VgTs_Empty */
186              "movq	%2, %%rax\n"    /* set %rax = __NR_exit */
187              "movq	%3, %%rdi\n"    /* set %rdi = tst->os_state.exitcode */
188              "syscall\n"		/* exit(tst->os_state.exitcode) */
189              : "=m" (tst->status)
190              : "n" (VgTs_Empty), "n" (__NR_exit), "m" (tst->os_state.exitcode));
191     #elif defined(VGP_ppc32_linux) || defined(VGP_ppc64_linux)
192           { UInt vgts_empty = (UInt)VgTs_Empty;
193             asm volatile (
194               "stw %1,%0\n\t"          /* set tst->status = VgTs_Empty */
195               "li  0,%2\n\t"           /* set r0 = __NR_exit */
196               "lwz 3,%3\n\t"           /* set r3 = tst->os_state.exitcode */
197               "sc\n\t"                 /* exit(tst->os_state.exitcode) */
198               : "=m" (tst->status)
199               : "r" (vgts_empty), "n" (__NR_exit), "m" (tst->os_state.exitcode));
200           }
201     #elif defined(VGP_arm_linux)
202           asm volatile (
203              "str  %1, %0\n"      /* set tst->status = VgTs_Empty */
204              "mov  r7, %2\n"      /* set %r7 = __NR_exit */
205              "ldr  r0, %3\n"      /* set %r0 = tst->os_state.exitcode */
206              "svc  0x00000000\n"  /* exit(tst->os_state.exitcode) */
207              : "=m" (tst->status)
208              : "r" (VgTs_Empty), "n" (__NR_exit), "m" (tst->os_state.exitcode));
209     #elif defined(VGP_s390x_linux)
210           asm volatile (
211              "st   %1, %0\n"        /* set tst->status = VgTs_Empty */
212              "lg   2, %3\n"         /* set r2 = tst->os_state.exitcode */
213              "svc %2\n"             /* exit(tst->os_state.exitcode) */
214              : "=m" (tst->status)
215              : "d" (VgTs_Empty), "n" (__NR_exit), "m" (tst->os_state.exitcode)
216              : "2");
217     #else
218     # error Unknown platform
219     #endif
220     
221           VG_(core_panic)("Thread exit failed?\n");
222        }
223     
224        /*NOTREACHED*/
225        vg_assert(0);
226     }

238     /* Allocate a stack for this thread, if it doesn't already have one.
239        They're allocated lazily, and never freed.  Returns the initial stack
240        pointer value to use, or 0 if allocation failed. */
241     Addr ML_(allocstack)(ThreadId tid)
242     {
243        ThreadState* tst = VG_(get_ThreadState)(tid);
244        VgStack*     stack;
245        Addr         initial_SP;
246     
247        /* Either the stack_base and stack_init_SP are both zero (in which
248           case a stack hasn't been allocated) or they are both non-zero,
249           in which case it has. */
250     
251        if (tst->os_state.valgrind_stack_base == 0)
252           vg_assert(tst->os_state.valgrind_stack_init_SP == 0);
253     
254        if (tst->os_state.valgrind_stack_base != 0)
255           vg_assert(tst->os_state.valgrind_stack_init_SP != 0);
256     
257        /* If no stack is present, allocate one. */
258     
259        if (tst->os_state.valgrind_stack_base == 0) {
260           stack = VG_(am_alloc_VgStack)( &initial_SP );
261           if (stack) {
262              tst->os_state.valgrind_stack_base    = (Addr)stack;
263              tst->os_state.valgrind_stack_init_SP = initial_SP;
264           }
265        }
266     
267        if (0)
268           VG_(printf)( "stack for tid %d at %p; init_SP=%p\n",
269                        tid, 
270                        (void*)tst->os_state.valgrind_stack_base, 
271                        (void*)tst->os_state.valgrind_stack_init_SP );
272                       
273        return tst->os_state.valgrind_stack_init_SP;
274     }

276     /* Allocate a stack for the main thread, and run it all the way to the
277        end.  Although we already have a working VgStack
278        (VG_(interim_stack)) it's better to allocate a new one, so that
279        overflow detection works uniformly for all threads.
280     */
281     void VG_(main_thread_wrapper_NORETURN)(ThreadId tid)
282     {
283        Addr sp;
284        VG_(debugLog)(1, "syswrap-linux", 
285                         "entering VG_(main_thread_wrapper_NORETURN)\n");
286     
287        sp = ML_(allocstack)(tid);
288     
289     #if defined(VGP_ppc32_linux)
290        /* make a stack frame */
291        sp -= 16;
292        sp &= ~0xF;
293        *(UWord *)sp = 0;
294     #elif defined(VGP_ppc64_linux)
295        /* make a stack frame */
296        sp -= 112;
297        sp &= ~((Addr)0xF);
298        *(UWord *)sp = 0;
299     #elif defined(VGP_s390x_linux)
300        /* make a stack frame */
301        sp -= 160;
302        sp &= ~((Addr)0xF);
303        *(UWord *)sp = 0;
304     #endif
305     
306        /* If we can't even allocate the first thread's stack, we're hosed.
307           Give up. */
308        vg_assert2(sp != 0, "Cannot allocate main thread's stack.");
309     
310        /* shouldn't be any other threads around yet */
311        vg_assert( VG_(count_living_threads)() == 1 );
312     
313        ML_(call_on_new_stack_0_1)( 
314           (Addr)sp,               /* stack */
315           0,                      /* bogus return address */
316           run_a_thread_NORETURN,  /* fn to call */
317           (Word)tid               /* arg to give it */
318        );
319     
320        /*NOTREACHED*/
321        vg_assert(0);
322     }

325     /* Do a clone which is really a fork() */
326     SysRes ML_(do_fork_clone) ( ThreadId tid, UInt flags,
327                                 Int* parent_tidptr, Int* child_tidptr )
328     {
329        vki_sigset_t fork_saved_mask;
330        vki_sigset_t mask;
331        SysRes       res;
332     
333        if (flags & (VKI_CLONE_SETTLS | VKI_CLONE_FS | VKI_CLONE_VM 
334                     | VKI_CLONE_FILES | VKI_CLONE_VFORK))
335           return VG_(mk_SysRes_Error)( VKI_EINVAL );
336     
337        /* Block all signals during fork, so that we can fix things up in
338           the child without being interrupted. */
339        VG_(sigfillset)(&mask);
340        VG_(sigprocmask)(VKI_SIG_SETMASK, &mask, &fork_saved_mask);
341     
342        VG_(do_atfork_pre)(tid);
343     
344        /* Since this is the fork() form of clone, we don't need all that
345           VG_(clone) stuff */
346     #if defined(VGP_x86_linux) \
347         || defined(VGP_ppc32_linux) || defined(VGP_ppc64_linux) \
348         || defined(VGP_arm_linux)
349        res = VG_(do_syscall5)( __NR_clone, flags, 
350                                (UWord)NULL, (UWord)parent_tidptr, 
351                                (UWord)NULL, (UWord)child_tidptr );
352     #elif defined(VGP_amd64_linux)
353        /* note that the last two arguments are the opposite way round to x86 and
354           ppc32 as the amd64 kernel expects the arguments in a different order */
355        res = VG_(do_syscall5)( __NR_clone, flags, 
356                                (UWord)NULL, (UWord)parent_tidptr, 
357                                (UWord)child_tidptr, (UWord)NULL );
358     #elif defined(VGP_s390x_linux)
359        /* Note that s390 has the stack first and then the flags */
360        res = VG_(do_syscall4)( __NR_clone, (UWord) NULL, flags,
361                               (UWord)parent_tidptr, (UWord)child_tidptr);
362     #else
363     # error Unknown platform
364     #endif
365     
366        if (!sr_isError(res) && sr_Res(res) == 0) {
367           /* child */
368           VG_(do_atfork_child)(tid);
369     
370           /* restore signal mask */
371           VG_(sigprocmask)(VKI_SIG_SETMASK, &fork_saved_mask, NULL);
372     
373           /* If --child-silent-after-fork=yes was specified, set the
374              output file descriptors to 'impossible' values.  This is
375              noticed by send_bytes_to_logging_sink in m_libcprint.c, which
376              duly stops writing any further output. */
377           if (VG_(clo_child_silent_after_fork)) {
378              if (!VG_(log_output_sink).is_socket)
379                 VG_(log_output_sink).fd = -1;
380              if (!VG_(xml_output_sink).is_socket)
381                 VG_(xml_output_sink).fd = -1;
382           }
383        } 
384        else 
385        if (!sr_isError(res) && sr_Res(res) > 0) {
386           /* parent */
387           VG_(do_atfork_parent)(tid);
388     
389           if (VG_(clo_trace_syscalls))
390     	  VG_(printf)("   clone(fork): process %d created child %ld\n",
391                           VG_(getpid)(), sr_Res(res));
392     
393           /* restore signal mask */
394           VG_(sigprocmask)(VKI_SIG_SETMASK, &fork_saved_mask, NULL);
395        }
396     
397        return res;
398     }

599     PRE(sys_exit_group)
600     {
601        ThreadId     t;
602        ThreadState* tst;
603     
604        PRINT("exit_group( %ld )", ARG1);
605        PRE_REG_READ1(void, "exit_group", int, status);
606     
607        tst = VG_(get_ThreadState)(tid);
608     
609        /* A little complex; find all the threads with the same threadgroup
610           as this one (including this one), and mark them to exit */
611        for (t = 1; t < VG_N_THREADS; t++) {
612           if ( /* not alive */
613                VG_(threads)[t].status == VgTs_Empty 
614                ||
615     	   /* not our group */
616                VG_(threads)[t].os_state.threadgroup != tst->os_state.threadgroup
617              )
618              continue;
619     
620           VG_(threads)[t].exitreason = VgSrc_ExitThread;
621           VG_(threads)[t].os_state.exitcode = ARG1;
622     
623           if (t != tid)
624     	 VG_(get_thread_out_of_syscall)(t); /* unblock it, if blocked */
625        }
626     
627        /* We have to claim the syscall already succeeded. */
628        SET_STATUS_Success(0);
629     }

1471

1915     /* Returns True if running on 2.6.22, else False (or False if
1916        cannot be determined). */
1917     static Bool linux_kernel_2_6_22(void)
1918     {
1919        static Int result = -1;
1920        Int fd, read;
1921        HChar release[64];
1922        SysRes res;
1923     
1924        if (result == -1) {
1925           res = VG_(open)("/proc/sys/kernel/osrelease", 0, 0);
1926           if (sr_isError(res))
1927              return False;
1928           fd = sr_Res(res);
1929           read = VG_(read)(fd, release, sizeof(release) - 1);
1930           vg_assert(read >= 0);
1931           release[read] = 0;
1932           VG_(close)(fd);
1933           //VG_(printf)("kernel release = %s\n", release);
1934           result = (VG_(strncmp)(release, "2.6.22", 6) == 0
1935                     && (release[6] < '0' || release[6] > '9'));
1936        }
1937        vg_assert(result == 0 || result == 1);
1938        return result == 1;
1939     }

2427     // This has different signatures for different platforms.
2428     //
2429     //  x86:   int  sys_pipe(unsigned long __user *fildes);
2430     //  AMD64: long sys_pipe(int *fildes);
2431     //  ppc32: int  sys_pipe(int __user *fildes);
2432     //  ppc64: int  sys_pipe(int __user *fildes);

2434     // The type of the argument is most important, and it is an array of 32 bit
2435     // values in all cases.  (The return type differs across platforms, but it
2436     // is not used.)  So we use 'int' as its type.  This fixed bug #113230 which
2437     // was caused by using an array of 'unsigned long's, which didn't work on
2438     // AMD64.

2451           VG_(close)(p[1]);
2452           SET_STATUS_Failure( VKI_EMFILE );

2454           POST_MEM_WRITE( ARG1, 2*sizeof(int) );

5791           }
