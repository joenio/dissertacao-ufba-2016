92     static void add_stacktrace_to_list(ThreadId tid)
93     {
94        static UWord stacktrace_counter = 0;
95     
96        UInt n_ips;
97        StackTrace st = NULL;
98        StackTraceNode* STNode;
99        
100        n_ips = IT_(get_stack_trace)(&st, tid);
101     
102        STNode = VG_(malloc)( "it_sinks.c: add_stacktrace_to_list",
103                              sizeof(StackTraceNode) );
104     
105        STNode->st     = st;
106        STNode->n_ips  = n_ips;
107        STNode->seq_nr = stacktrace_counter;
108        STNode->next   = NULL;
109     
110        if( last_stacktrace ) {
111           last_stacktrace->next = STNode;
112           last_stacktrace = STNode;
113        } else
114           last_stacktrace = IT_(stacktrace_list) = STNode;
115     
116        stacktrace_counter++;
117     }

120     /* Returns the current thread specific stack trace in 'store'.
121        The function first checks the stack trace updadate flag in 'store' 
122        and updates the global list and thread-specific stack trace in 'store' 
123        if the flag is True. */ 
124     static StackTraceNode* get_current_stacktrace(ThreadId tid,
125                                                   ShadowStateStore* store)
126     {
127        // If update flag is True, add a new stack trace to the global list
128        // and update the thread-specific current stack trace.
129        if(store->flag) {
130           add_stacktrace_to_list(tid);
131           store->data->last_stacktrace = last_stacktrace;
132     
133           // Reset flag to False      
134           store->flag = False;
135        }
136        
137        return store->data->last_stacktrace;
138     }

160     static UWord syscall_param_offset_to_index(UInt offset)
161     {
162        // Divide everything by 4, so that all offsets that point to the middle
163        // of a register is handled properly
164        switch(offset / 4) {
165           case offsetof(VexGuestX86State, guest_EBX) / 4: return 0;
166           case offsetof(VexGuestX86State, guest_ECX) / 4: return 1;
167           case offsetof(VexGuestX86State, guest_EDX) / 4: return 2;
168           case offsetof(VexGuestX86State, guest_ESI) / 4: return 3;
169           case offsetof(VexGuestX86State, guest_EDI) / 4: return 4;
170           case offsetof(VexGuestX86State, guest_EBP) / 4: return 5;
171           default: tl_assert(0);
172        }
173        
174        return 0; // Never happens
175     }

177     // Adds an event to the global list and increments the event sequence number
178     static void add_event_list_node(IT_TaintSinkEvent* event)
179     {
180        static IT_EventListNode** next_event = &IT_(events_list);
181        
182        IT_EventListNode* listNode;
183     
184        listNode = VG_(malloc)("it_sinks.c: add_event_list_node",
185                                sizeof(IT_EventListNode) );
186        
187        // Set sequence number
188        event->seq_nr = event_counter;
189        event_counter++;
190        
191        // Create list node
192        listNode->data = event;
193        listNode->next = NULL;
194        
195        // Insert in linked list
196        *next_event = listNode;
197        next_event  = &listNode->next;
198     }

200     // Convenience function for checking if an instruction at 'a' should be
201     // excluded from taint sink generation.
202     static Bool is_excluded_from_sinks(Addr a)
203     {
204     	static Char imageNameBuffer[1024];
205     
206     	if( IT_(is_in_excluded_range_prop)(a) )
207     		return True;
208     
209     	if(VG_(get_objname)(a, imageNameBuffer, sizeof(imageNameBuffer) ) )
210     	{
211     		Word handle = IT_(get_sink_filter_handle_from_path)(imageNameBuffer);
212     		Bool isSinkExcludedImage =
213     				IT_(is_in_sink_exclude_pattern)(imageNameBuffer, a);
214     
215     		if(handle == -1)
216     			return isSinkExcludedImage;
217     		else
218     			return (isSinkExcludedImage ||
219     						!IT_(is_included_in_sink_filter)(handle, a));
220     	}
221     
222     	return False;
223     }

243     // Initializes data in the shadow state for a new thread
244     void IT_(sinks_pre_thread)(ThreadId tid)
245     {
246        static UInt thread_counter = 0;
247        
248        ShadowStateStore toShadowState;
249        
250        SinkData* data = VG_(malloc)("it_sinks.c: IT_(sinks_pre_thread)",
251                                           sizeof(SinkData));
252        data->thread_nr       = thread_counter;
253        data->current_event   = NULL;
254        data->last_stacktrace = NULL;
255        
256        toShadowState.data = data;
257        toShadowState.flag = True;
258        
259        // Store pointer to SinkData struct + update flag in shadow state
260        VG_(set_shadow_regs_area) ( 
261           tid,
262           1 /* First shadow area */,
263           SINKS_STORE_START,
264           sizeof(toShadowState),
265           (const UChar*)&toShadowState );
266        
267        thread_counter++;
268     }

270     void IT_(sinks_pre_thread_exit)(ThreadId tid)
271     {
272        SinkData* data;
273     
274        VG_(get_shadow_regs_area)( 
275           tid, 
276           (UChar*)&data,
277           1 /* First shadow area */, 
278           SINK_DATA_POINTER_OFFSET,
279           sizeof(SinkData*) );
280        
281        // Free up resources
282        // TODO: There shouldn't be any outstanding taint sink events at this
283        //       point, so why is there sometimes? Seems post_syscall is not always 
284        //       called before thread_exit? Do we miss a possible taint sink event 
285        //       here?
286        if(data->current_event) {
287           if(data->current_event->aux_inv)
288              VG_(free)(data->current_event->aux_inv);
289           VG_(free)(data->current_event);
290        }
291     
292        VG_(free)(data);
293     }

295     /* Creates a new taint sink event and stores a pointer 
296        to it in the shadow state */
297     void IT_(sinks_pre_syscall)(ThreadId tid, UInt syscallno, 
298                                 UWord* args, UInt nArgs)
299     {
300        SinkData* data;
301        IT_TaintSinkEvent* event;
302        Bool isExcludedImage = is_excluded_from_sinks( VG_(get_IP)(tid) );
303     
304        if( ! IT_(clo_syscall_sink) || isExcludedImage )
305           return;
306        
307        // Get data from shadow state
308        VG_(get_shadow_regs_area)( 
309           tid, 
310           (UChar*)&data,
311           1 /* First shadow area */, 
312           SINK_DATA_POINTER_OFFSET,
313           sizeof(data) );
314        
315        // This can happen sometimes, probably when system calls are interrupted 
316        // by a signal.
317        if( data->current_event != NULL ) {
318           event = data->current_event;
319     
320           if(event->input_vector)
321              IT_(delete_labels)(event->input_vector);
322     
323           if(event->aux_inv) {
324              Int i = 0;
325              for( ; i < event->aux_inv->num_vectors; i++) {
326                 if(event->aux_inv->vectors[i])
327                    IT_(delete_labels)(event->aux_inv->vectors[i]);
328              }
329              VG_(free)(event->aux_inv);
330           }
331     
332           VG_(free)(event);
333        }
334           
335        event = get_empty_taint_sink_event();
336        
337        event->type           = SyscallSink;
338        event->thread_nr      = data->thread_nr;
339        event->location_1     = VG_(get_IP)(tid);
340        event->call_specifier = syscallno;
341        
342        data->current_event = event;
343     }

346     void IT_(sinks_post_syscall)(ThreadId tid, UInt syscallno, 
347                                  UWord* args, UInt nArgs, SysRes res)
348     {
349        ShadowStateStore   store;
350        IT_TaintSinkEvent* event;
351        Bool isExcludedImage = is_excluded_from_sinks( VG_(get_IP)(tid) );
352        
353        if( ! IT_(clo_syscall_sink) || isExcludedImage )
354           return;
355        
356        // Get data from shadow state
357        VG_(get_shadow_regs_area)( 
358           tid, 
359           (UChar*)&store,
360           1 /* First shadow area */, 
361           SINKS_STORE_START,
362           sizeof(store) );
363           
364        event = store.data->current_event;
365        
366        tl_assert(event);
367          
368        // Check if there actually is any taint to record
369        if(event->input_vector || event->aux_inv) {
370           event->stack_trace = get_current_stacktrace(tid, &store);
371     
372           add_event_list_node(event);
373           
374           // Write back to shadow state
375           VG_(set_shadow_regs_area) ( 
376              tid,
377              1 /* First shadow area */,
378              SINKS_STORE_START,
379              sizeof(store),
380              (const UChar*)&store );
381        } else {
382           // Throw away temporary taint sink event if there is no taint in it
383           VG_(free)(event);
384        }
385        
386        store.data->current_event = NULL;
387     }

390     void IT_(sinks_pre_reg_read)(CorePart part, ThreadId tid, Char* s, 
391                                  PtrdiffT guest_state_offset, SizeT size)
392     {
393        SinkData*          data;
394        LabelSet*          labels;
395        IT_TaintSinkEvent* event;
396        UWord              index;
397        Bool isExcludedImage = is_excluded_from_sinks( VG_(get_IP)(tid) );
398     
399        if( ! IT_(clo_syscall_sink) || 
400            part != Vg_CoreSysCall || 
401            isExcludedImage ||
402            guest_state_offset == 
403              offsetof(VexGuestX86State, guest_EAX) // Don't care about syscall no
404          )
405           return;
406     
407        // There should never be any reads from multiple registers at a time?
408        tl_assert(size <= 4);
409     
410        // Get SinkData pointer from shadow state
411        VG_(get_shadow_regs_area)( 
412           tid, 
413           (UChar*)&data,
414           1 /* First shadow area */, 
415           SINK_DATA_POINTER_OFFSET,
416           sizeof(SinkData*) );
417        
418        event = data->current_event;
419        tl_assert(event);
420        
421        labels = IT_(get_reg_taint_union)(guest_state_offset, size);
422     
423        if(labels) {
424     
425           if( ! event->aux_inv ) {
426              // Allocate zero-filled struct
427              event->aux_inv = VG_(calloc)("it_sinks.c: IT_(sinks_pre_reg_read)",
428                                          1, sizeof(IT_AuxInputVectors) );
429           }
430     
431           index = syscall_param_offset_to_index(guest_state_offset);
432     
433           // Get taint labels for register and set the corresponding slot in
434           // the auxilliary input vector array
435           IT_(set_labels)( 
436              &event->aux_inv->vectors[ index ],
437              labels );                             
438     
439           // Number of arguments to the call is the highest accessed index plus 1
440           event->aux_inv->num_vectors = MAX(event->aux_inv->num_vectors, index+1);
441        }
442     }

444     void IT_(sinks_pre_mem_read)(CorePart part, ThreadId tid, 
445                                  Char* s, Addr a, SizeT size)
446     {
447        SinkData* data;
448        Bool isExcludedImage = is_excluded_from_sinks( VG_(get_IP)(tid) );
449     
450        if( ! IT_(clo_syscall_sink) || part != Vg_CoreSysCall || isExcludedImage )
451           return;
452     
453        // Get SinkData pointer from shadow state
454        VG_(get_shadow_regs_area)( 
455           tid, 
456           (UChar*)&data,
457           1 /* First shadow area */, 
458           SINK_DATA_POINTER_OFFSET,
459           sizeof(SinkData*) );
460     
461           tl_assert(data->current_event);
462     
463        // Get taint labels for memory and set the primary input vector
464        IT_(set_labels)( 
465           &data->current_event->input_vector,
466           IT_(get_mem_taint_union)(a, size) );
467     }

480     // TODO: Update for handling memory write sinks
481     void IT_(print_taint_sink_events)(void)
482     {
483        const Char* syscall_param_strings[] = 
484           { "First parameter  (EBX): ",
485             "Second parameter (ECX): ",
486             "Third parameter  (EDX): ",
487             "Fourth parameter (ESI): ",
488             "Fifth parameter  (EDI): ",
489             "Sixth parameter  (EBP): " 
490            };
491       
492        VG_(printf)("*** Taint sink events ***\n");
493       
494        IT_EventListNode* curNode = IT_(events_list);
495        while(curNode) {
496           
497           IT_TaintSinkEvent* event = curNode->data;
498           Char* typeStr;
499           
500           switch(event->type) {
501              case UncondBranchSink: typeStr = "Unconditional Branch"; break;
502              case CondBranchSink:   typeStr = "Conditional Branch"; break;
503              case SyscallSink:      typeStr = "System Call"; break;
504              default:               typeStr = "*** Invalid Type ***";
505           }
506           
507           VG_(printf)(
508     "Taint sink event number %u:\n"
509     "   Type: %s\n",
510              event->seq_nr, typeStr);
511           
512           switch(event->type) {
513              case UncondBranchSink:
514                 VG_(printf)(
515     "   Address: 0x%08lx\n"
516     "   Taint Lables: ",
517                    event->location_1);
518                    
519                 if(event->input_vector) {
520                    IT_(print_label_list)(event->input_vector); 
521                    VG_(printf)("\n");
522                 } else 
523                    VG_(printf)("none\n");
524                    
525                 break;
526                 
527              case CondBranchSink:
528                 VG_(printf)(
529     "   Address of test:   0x%08lx\n"
530     "   Address of branch: 0x%08lx\n"
531     "   Taint Lables: ",
532                    event->location_1, event->location_2);
533                 
534                 if(event->input_vector) {
535                    IT_(print_label_list)(event->input_vector); 
536                    VG_(printf)("\n");
537                 } else 
538                    VG_(printf)("none\n");
539                 
540                 break;
541     
542              case SyscallSink:
543                 VG_(printf)(
544     "   System call number: %u\n"
545     "   Address: 0x%08lx\n"
546     "   Taint Lables:\n"
547     "      Read memory: ",
548                    event->call_specifier, event->location_1);
549                 
550                 if(event->input_vector) {
551                    IT_(print_label_list)(event->input_vector); 
552                    VG_(printf)("\n");
553                 } else 
554                    VG_(printf)("none\n");
555                    
556                 VG_(printf)(
557     "      Input parameters:\n");
558     
559                 UWord i;
560                 for(i=0; i<IT_MAX_AUX_INPV; i++) {
561                    VG_(printf)(
562     "         %s",
563                       syscall_param_strings[i]);
564                    
565                    if(event->aux_inv && event->aux_inv->vectors[i]) {
566                       IT_(print_label_list)(event->aux_inv->vectors[i]); 
567                       VG_(printf)("\n");
568                    } else 
569                       VG_(printf)("none\n");
570                 }
571                 break;
572                 
573              default:
574                 break;
575           }
576           
577           VG_(printf)(
578     "   Thread: %u\n"
579     "   Stack Trace:\n",
580              event->thread_nr);
581          
582           if(event->stack_trace->n_ips)
583              VG_(pp_StackTrace)( event->stack_trace->st, event->stack_trace->n_ips );    
584     
585           VG_(printf)("\n");
586        
587           curNode = curNode->next;
588        }
589        
590        VG_(printf)("*** End of taint sink events ***\n");
591     }

598     {
599        ShadowStateStore store;
600        IT_TaintSinkEvent* event;

604           return;

619        event->location_1  = VG_(get_IP)(tid);
620        
621        IT_(set_labels)( 
622           &event->input_vector,
623           labels );
624           
625        add_event_list_node(event);
626              
627        // Write back to shadow state

629           tid,
630           1 /* First shadow area */,

731     // in the shadow state
732     void IT_(instrument_eflags_update)(IRSB* sb, Int offset, Addr addr)
733     {
734        // If this is a PUT to the CC_OP pseudo register, assume that the current
735        // native instruction altered EFLAGS
736        if( offset == offsetof(VexGuestX86State, guest_CC_OP) ) {
737           addStmtToIRSB( 
738              sb,
739              IRStmt_Put( 
740                 sizeof(VexGuestX86State) + // Skip to first shadow state
741                    LAST_TEST_ADDRESS_OFFSET,
742                 IRExpr_Const( IRConst_U32(addr) ) ) );
743        }
744     }
