54     /* Do the low-level send of a message to the logging sink. */
55     static
56     void send_bytes_to_logging_sink ( OutputSink* sink, Char* msg, Int nbytes )
57     {
58        if (sink->is_socket) {
59           Int rc = VG_(write_socket)( sink->fd, msg, nbytes );
60           if (rc == -1) {
61              // For example, the listener process died.  Switch back to stderr.
62              sink->is_socket = False;
63              sink->fd = 2;
64              VG_(write)( sink->fd, msg, nbytes );
65           }
66        } else {
67           /* sink->fd could have been set to -1 in the various
68              sys-wrappers for sys_fork, if --child-silent-after-fork=yes
69              is in effect.  That is a signal that we should not produce
70              any more output. */
71           if (sink->fd >= 0)
72              VG_(write)( sink->fd, msg, nbytes );
73        }
74     }

83     typedef 
84        struct {
85           HChar       buf[512];
86           Int         buf_used;
87           OutputSink* sink;
88        } 
89        printf_buf_t;
90     
91     // Adds a single char to the buffer.  When the buffer gets sufficiently
92     // full, we write its contents to the logging sink.
93     static void add_to__printf_buf ( HChar c, void *p )
94     {
95        printf_buf_t *b = (printf_buf_t *)p;
96        
97        if (b->buf_used > sizeof(b->buf) - 2 ) {
98           send_bytes_to_logging_sink( b->sink, b->buf, b->buf_used );
99           b->buf_used = 0;
100        }
101        b->buf[b->buf_used++] = c;
102        b->buf[b->buf_used]   = 0;
103        tl_assert(b->buf_used < sizeof(b->buf));
104     }
105     
106     static UInt vprintf_to_buf ( printf_buf_t* b,
107                                  const HChar *format, va_list vargs )
108     {
109        UInt ret = 0;
110        if (b->sink->fd >= 0) {
111           ret = VG_(debugLog_vprintf) 
112                    ( add_to__printf_buf, b, format, vargs );
113        }
114        return ret;
115     }
116     
117     static UInt vprintf_WRK ( OutputSink* sink,
118                               const HChar *format, va_list vargs )
119     {
120        printf_buf_t myprintf_buf
121           = { "", 0, sink };
122        UInt ret
123           = vprintf_to_buf(&myprintf_buf, format, vargs);
124        // Write out any chars left in the buffer.
125        if (myprintf_buf.buf_used > 0) {
126           send_bytes_to_logging_sink( myprintf_buf.sink,
127                                       myprintf_buf.buf,
128                                       myprintf_buf.buf_used );
129        }
130        return ret;
131     }

177     /* If we had an explicit buf structure here, it would contain only one
178        field, indicating where the next char is to go.  So use p directly
179        for that, rather than having it be a pointer to a structure. */
180     
181     static void add_to__sprintf_buf ( HChar c, void *p )
182     {
183        HChar** b = p;
184        *(*b)++ = c;
185     }
186     
187     UInt VG_(vsprintf) ( Char* buf, const HChar *format, va_list vargs )
188     {
189        Int ret;
190        HChar* sprintf_ptr = buf;
191     
192        ret = VG_(debugLog_vprintf) 
193                 ( add_to__sprintf_buf, &sprintf_ptr, format, vargs );
194        add_to__sprintf_buf('\0', &sprintf_ptr);
195     
196        vg_assert(VG_(strlen)(buf) == ret);
197     
198        return ret;
199     }

214     typedef 
215        struct {
216           HChar* buf;
217           Int    buf_size;
218           Int    buf_used;
219        } 
220        snprintf_buf_t;
221     
222     static void add_to__snprintf_buf ( HChar c, void* p )
223     {
224        snprintf_buf_t* b = p;
225        if (b->buf_size > 0 && b->buf_used < b->buf_size) {
226           b->buf[b->buf_used++] = c;
227           if (b->buf_used < b->buf_size)
228              b->buf[b->buf_used] = 0;
229           else
230              b->buf[b->buf_size-1] = 0; /* pre: b->buf_size > 0 */
231        } 
232     }
233     
234     UInt VG_(vsnprintf) ( Char* buf, Int size, const HChar *format, va_list vargs )
235     {
236        snprintf_buf_t b;
237        b.buf      = buf;
238        b.buf_size = size < 0 ? 0 : size;
239        b.buf_used = 0;
240     
241        (void) VG_(debugLog_vprintf) 
242                  ( add_to__snprintf_buf, &b, format, vargs );
243     
244        return b.buf_used;
245     }

273     // Percentify n/m with d decimal places.  Includes the '%' symbol at the end.
274     // Right justifies in 'buf'.
275     void VG_(percentify)(ULong n, ULong m, UInt d, Int n_buf, char buf[]) 
276     {
277        Int i, len, space;
278        ULong p1;
279        Char fmt[32];
280     
281        if (m == 0) {
282           // Have to generate the format string in order to be flexible about
283           // the width of the field.
284           VG_(sprintf)(fmt, "%%-%ds", n_buf);
285           // fmt is now "%<n_buf>s" where <d> is 1,2,3...
286           VG_(sprintf)(buf, fmt, "--%");
287           return;
288        }
289        
290        p1 = (100*n) / m;
291         
292        if (d == 0) {
293           VG_(sprintf)(buf, "%lld%%", p1);
294        } else {
295           ULong p2;
296           UInt  ex;
297           switch (d) {
298           case 1: ex = 10;    break;
299           case 2: ex = 100;   break;
300           case 3: ex = 1000;  break;
301           default: VG_(tool_panic)("Currently can only handle 3 decimal places");
302           }
303           p2 = ((100*n*ex) / m) % ex;
304           // Have to generate the format string in order to be flexible about
305           // the width of the post-decimal-point part.
306           VG_(sprintf)(fmt, "%%lld.%%0%dlld%%%%", d);
307           // fmt is now "%lld.%0<d>lld%%" where <d> is 1,2,3...
308           VG_(sprintf)(buf, fmt, p1, p2);
309        }
310     
311        len = VG_(strlen)(buf);
312        space = n_buf - len;
313        if (space < 0) space = 0;     /* Allow for v. small field_width */
314        i = len;
315     
316        /* Right justify in field */
317        for (     ; i >= 0;    i--)  buf[i + space] = buf[i];
318        for (i = 0; i < space; i++)  buf[i] = ' ';
319     }

322     /* ---------------------------------------------------------------------
323        elapsed_wallclock_time()
324        ------------------------------------------------------------------ */
325     
326     /* Get the elapsed wallclock time since startup into buf, which must
327        16 chars long.  This is unchecked.  It also relies on the
328        millisecond timer having been set to zero by an initial read in
329        m_main during startup. */
330     
331     void VG_(elapsed_wallclock_time) ( /*OUT*/HChar* buf )
332     {
333        UInt t, ms, s, mins, hours, days;
334     
335        t  = VG_(read_millisecond_timer)(); /* milliseconds */
336     
337        ms = t % 1000;
338        t /= 1000; /* now in seconds */
339     
340        s = t % 60;
341        t /= 60; /* now in minutes */
342     
343        mins = t % 60;
344        t /= 60; /* now in hours */
345     
346        hours = t % 24;
347        t /= 24; /* now in days */
348     
349        days = t;
350     
351        VG_(sprintf)(buf, "%02u:%02u:%02u:%02u.%03u ", days, hours, mins, s, ms);
352     }

356        message()

359     /* A buffer for accumulating VG_(message) style output.  This is
360        pretty much the same as VG_(printf)'s scheme, with two differences:
361     
362        * The message buffer persists between calls, so that multiple
363          calls to VG_(message) can build up output.
364     
365        * Whenever the first character on a line is emitted, the
366          ==PID== style preamble is stuffed in before it.
367     */
368     typedef 
369        struct {
370           HChar buf[512+128];
371           Int   buf_used;
372           Bool  atLeft; /* notionally, is the next char position at the
373                            leftmost column? */
374           /* Current message kind - changes from call to call */
375           VgMsgKind kind;
376           /* destination */
377           OutputSink* sink;
378        } 
379        vmessage_buf_t;
380     
381     static vmessage_buf_t vmessage_buf
382        = { "", 0, True, Vg_UserMsg, &VG_(log_output_sink) };
383     
384     
385     // Adds a single char to the buffer.  We aim to have at least 128
386     // bytes free in the buffer, so that it's always possible to emit
387     // the preamble into the buffer if c happens to be the character
388     // following a \n.  When the buffer gets too full, we write its
389     // contents to the logging sink.
390     static void add_to__vmessage_buf ( HChar c, void *p )
391     {
392        HChar tmp[64];
393        vmessage_buf_t* b = (vmessage_buf_t*)p;
394     
395        vg_assert(b->buf_used >= 0 && b->buf_used < sizeof(b->buf)-128);
396     
397        if (UNLIKELY(b->atLeft)) {
398           // insert preamble
399           HChar ch;
400           Int   i, depth;
401     
402           // Print one '>' in front of the messages for each level of
403           // self-hosting being performed.
404           depth = RUNNING_ON_VALGRIND;
405           if (depth > 10)
406              depth = 10; // ?!?!
407           for (i = 0; i < depth; i++) {
408              b->buf[b->buf_used++] = '>';
409           }
410     
411           if (Vg_FailMsg == b->kind) {
412              // "valgrind: " prefix.
413              b->buf[b->buf_used++] = 'v';
414              b->buf[b->buf_used++] = 'a';
415              b->buf[b->buf_used++] = 'l';
416              b->buf[b->buf_used++] = 'g';
417              b->buf[b->buf_used++] = 'r';
418              b->buf[b->buf_used++] = 'i';
419              b->buf[b->buf_used++] = 'n';
420              b->buf[b->buf_used++] = 'd';
421              b->buf[b->buf_used++] = ':';
422              b->buf[b->buf_used++] = ' ';
423           } else {
424              switch (b->kind) {
425                 case Vg_UserMsg:       ch = '='; break;
426                 case Vg_DebugMsg:      ch = '-'; break;
427                 case Vg_ClientMsg:     ch = '*'; break;
428                 default:               ch = '?'; break;
429              }
430     
431              b->buf[b->buf_used++] = ch;
432              b->buf[b->buf_used++] = ch;
433     
434              if (VG_(clo_time_stamp)) {
435                 VG_(memset)(tmp, 0, sizeof(tmp));
436                 VG_(elapsed_wallclock_time)(tmp);
437                 tmp[sizeof(tmp)-1] = 0;
438                 for (i = 0; tmp[i]; i++)
439                    b->buf[b->buf_used++] = tmp[i];
440              }
441     
442              VG_(sprintf)(tmp, "%d", VG_(getpid)());
443              tmp[sizeof(tmp)-1] = 0;
444              for (i = 0; tmp[i]; i++)
445                 b->buf[b->buf_used++] = tmp[i];
446     
447              b->buf[b->buf_used++] = ch;
448              b->buf[b->buf_used++] = ch;
449              b->buf[b->buf_used++] = ' ';
450           }
451     
452           /* We can't possibly have stuffed 96 chars in merely as a result
453              of making the preamble (can we?) */
454           vg_assert(b->buf_used < sizeof(b->buf)-32);
455        }
456     
457        b->buf[b->buf_used++] = c;
458        b->buf[b->buf_used]   = 0;
459        
460        if (b->buf_used >= sizeof(b->buf) - 128) {
461           send_bytes_to_logging_sink( b->sink, b->buf, b->buf_used );
462           b->buf_used = 0;
463        }
464     
465        b->atLeft = c == '\n';
466     }

469     UInt VG_(vmessage) ( VgMsgKind kind, const HChar* format, va_list vargs )
470     {
471        UInt ret;
472     
473        /* Note (carefully) that the buf persists from call to call, unlike
474           with the other printf variants in earlier parts of this file. */
475        vmessage_buf_t* b = &vmessage_buf; /* shorthand for convenience */
476     
477        /* We have to set this each call, so that the correct flavour
478           of preamble is emitted at each \n. */
479        b->kind = kind;
480     
481        ret = VG_(debugLog_vprintf) ( add_to__vmessage_buf,
482                                      b, format, vargs );
483     
484        /* If the message finished exactly with a \n, then flush it at this
485           point.  If not, assume more bits of the same line will turn up
486           in later messages, so don't bother to flush it right now. */
487     
488        if (b->atLeft && b->buf_used > 0) {
489           send_bytes_to_logging_sink( b->sink, b->buf, b->buf_used );
490           b->buf_used = 0;
491        }
492     
493        return ret;
494     }
