32     #ifndef __PUB_TOOL_MALLOCFREE_H
33     #define __PUB_TOOL_MALLOCFREE_H
34     
35     // These can be for allocating memory used by tools.
36     // Nb: the allocators *always succeed* -- they never return NULL (Valgrind
37     // will abort if they can't allocate the memory).
38     // The 'cc' is a string that identifies the allocation point.  It's used when
39     // --profile-heap=yes is specified.
40     extern void* VG_(malloc)         ( HChar* cc, SizeT nbytes );
41     extern void  VG_(free)           ( void* p );
42     extern void* VG_(calloc)         ( HChar* cc, SizeT n, SizeT bytes_per_elem );
43     extern void* VG_(realloc)        ( HChar* cc, void* p, SizeT size );
44     extern Char* VG_(strdup)         ( HChar* cc, const Char* s );
45     
46     // Returns the usable size of a heap-block.  It's the asked-for size plus
47     // possibly some more due to rounding up.
48     extern SizeT VG_(malloc_usable_size)( void* p );
49     
50     // TODO: move somewhere else
51     // Call here to bomb the system when out of memory (mmap anon fails)
52     __attribute__((noreturn))
53     extern void VG_(out_of_memory_NORETURN) ( HChar* who, SizeT szB );
54     
55     #endif   // __PUB_TOOL_MALLOCFREE_H
