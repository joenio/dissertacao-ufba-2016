34     /* Note, this header requires pub_{core,tool}_xarray.h to be
35        included ahead of it. */
36     
37     // Command line pieces, after they have been extracted from argv in
38     // m_main.main().  These are all NULL-terminated vectors.
39     
40     /* Args for the client. */
41     extern XArray* /* of HChar* */ VG_(args_for_client);
42     
43     /* Args for V.  This is the concatenation of the following:
44        - contents of ~/.valgrindrc
45        - contents of $VALGRIND_OPTS
46        - contents of ./.valgrindrc
47        - args from the command line
48        in the stated order.
49     
50        Only the last of these is passed onwards to child Valgrinds at
51        client sys_execve, since the children will re-acquire the first 3
52        categories for themselves.  Therefore we also record the number of
53        these no-pass-at-execve arguments -- that is what
54        VG_(args_for_valgrind_noexecpass) is. */
55     extern XArray* /* of HChar* */ VG_(args_for_valgrind);
56     
57     /* Number of leading args in VG_(args_for_valgrind) not to pass on at
58        exec time. */
59     extern Int VG_(args_for_valgrind_noexecpass);
60     
61     /* The name of the client executable, as specified on the command
62        line. */
63     extern const HChar* VG_(args_the_exename);
64     
65     
66     #endif   // __PUB_TOOL_CLIENTSTATE_H
