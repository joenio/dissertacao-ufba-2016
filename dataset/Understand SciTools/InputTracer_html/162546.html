54     __attribute__ ((noreturn))
55     void ML_(am_exit)( Int status )
56     {
57     #  if defined(VGO_linux)
58        (void)VG_(do_syscall1)(__NR_exit_group, status);
59     #  endif
60        (void)VG_(do_syscall1)(__NR_exit, status);
61        /* Why are we still alive here? */
62        /*NOTREACHED*/
63        *(volatile Int *)0 = 'x';
64        aspacem_assert(2+2 == 5);
65     }

68     {
69        VG_(debugLog)(0, "aspacem", "Valgrind: FATAL: %s\n", what);
70        VG_(debugLog)(0, "aspacem", "Exiting now.\n");
71        ML_(am_exit)(1);
72     }

105     // A simple sprintf implementation, so as to avoid dependence on
106     // m_libcprint.
107     
108     static void local_add_to_aspacem_sprintf_buf ( HChar c, void *p )
109     {
110        HChar** aspacem_sprintf_ptr = p;
111        *(*aspacem_sprintf_ptr)++ = c;
112     }
113     
114     static
115     UInt local_vsprintf ( HChar* buf, const HChar *format, va_list vargs )
116     {
117        Int ret;
118        Char *aspacem_sprintf_ptr = buf;
119     
120        ret = VG_(debugLog_vprintf)
121                 ( local_add_to_aspacem_sprintf_buf, 
122                   &aspacem_sprintf_ptr, format, vargs );
123        local_add_to_aspacem_sprintf_buf('\0', &aspacem_sprintf_ptr);
124     
125        return ret;
126     }
127     
128     UInt ML_(am_sprintf) ( HChar* buf, const HChar *format, ... )
129     {
130        UInt ret;
131        va_list vargs;
132     
133        va_start(vargs,format);
134        ret = local_vsprintf(buf, format, vargs);
135        va_end(vargs);
136     
137        return ret;
138     }

149     /* Note: this is VG_, not ML_. */
150     SysRes VG_(am_do_mmap_NO_NOTIFY)( Addr start, SizeT length, UInt prot, 
151                                       UInt flags, Int fd, Off64T offset)
152     {
153        SysRes res;
154        aspacem_assert(VG_IS_PAGE_ALIGNED(offset));
155     #  if defined(VGP_x86_linux) || defined(VGP_ppc32_linux) \
156           || defined(VGP_arm_linux)
157        /* mmap2 uses 4096 chunks even if actual page size is bigger. */
158        aspacem_assert((offset % 4096) == 0);
159        res = VG_(do_syscall6)(__NR_mmap2, (UWord)start, length,
160                               prot, flags, fd, offset / 4096);
161     #  elif defined(VGP_amd64_linux) || defined(VGP_ppc64_linux) \
162             || defined(VGP_ppc32_aix5) || defined(VGP_ppc64_aix5) \
163             || defined(VGP_s390x_linux)
164        res = VG_(do_syscall6)(__NR_mmap, (UWord)start, length, 
165                              prot, flags, fd, offset);
166     #  elif defined(VGP_x86_darwin)
167        if (fd == 0  &&  (flags & VKI_MAP_ANONYMOUS)) {
168            fd = -1;  // MAP_ANON with fd==0 is EINVAL
169        }
170        res = VG_(do_syscall7)(__NR_mmap, (UWord)start, length,
171                               prot, flags, fd, offset & 0xffffffff, offset >> 32);
172     #  elif defined(VGP_amd64_darwin)
173        if (fd == 0  &&  (flags & VKI_MAP_ANONYMOUS)) {
174            fd = -1;  // MAP_ANON with fd==0 is EINVAL
175        }
176        res = VG_(do_syscall6)(__NR_mmap, (UWord)start, length,
177                               prot, flags, (UInt)fd, offset);
178     #  else
179     #    error Unknown platform
180     #  endif
181        return res;
182     }

227     {
228        /* Move the mapping old_addr .. old_addr+old_len-1 to the new

288     /* Get the dev, inode and mode info for a file descriptor, if
289        possible.  Returns True on success. */
290     Bool ML_(am_get_fd_d_i_m)( Int fd, 
291                                /*OUT*/ULong* dev, 
292                                /*OUT*/ULong* ino, /*OUT*/UInt* mode )
293     {
294        SysRes          res;
295        struct vki_stat buf;
296     #  if defined(VGO_linux) && defined(__NR_fstat64)
297        /* Try fstat64 first as it can cope with minor and major device
298           numbers outside the 0-255 range and it works properly for x86
299           binaries on amd64 systems where fstat seems to be broken. */
300        struct vki_stat64 buf64;
301        res = VG_(do_syscall2)(__NR_fstat64, fd, (UWord)&buf64);
302        if (!sr_isError(res)) {
303           *dev  = (ULong)buf64.st_dev;
304           *ino  = (ULong)buf64.st_ino;
305           *mode = (UInt) buf64.st_mode;
306           return True;
307        }
308     #  endif
309        res = VG_(do_syscall2)(__NR_fstat, fd, (UWord)&buf);
310        if (!sr_isError(res)) {
311           *dev  = (ULong)buf.st_dev;
312           *ino  = (ULong)buf.st_ino;
313           *mode = (UInt) buf.st_mode;
314           return True;
315        }
316        return False;
317     }

319     Bool ML_(am_resolve_filename) ( Int fd, /*OUT*/HChar* buf, Int nbuf )
320     {
321     #if defined(VGO_linux)
322        Int i;
323        HChar tmp[64];
324        for (i = 0; i < nbuf; i++) buf[i] = 0;
325        ML_(am_sprintf)(tmp, "/proc/self/fd/%d", fd);
326        if (ML_(am_readlink)(tmp, buf, nbuf) > 0 && buf[0] == '/')
327           return True;
328        else
329           return False;
330     
331     #elif defined(VGO_aix5)
332        I_die_here; /* maybe just return False? */
333        return False;
334     
335     #elif defined(VGO_darwin)
336        HChar tmp[VKI_MAXPATHLEN+1];
337        if (0 == ML_(am_fcntl)(fd, VKI_F_GETPATH, (UWord)tmp)) {
338           if (nbuf > 0) {
339              VG_(strncpy)( buf, tmp, nbuf < sizeof(tmp) ? nbuf : sizeof(tmp) );
340              buf[nbuf-1] = 0;
341           }
342           if (tmp[0] == '/') return True;
343        }
344        return False;
345     
346     #  else
347     #     error Unknown OS
348     #  endif
349     }

360     /* Allocate and initialise a VgStack (anonymous valgrind space).
361        Protect the stack active area and the guard areas appropriately.
362        Returns NULL on failure, else the address of the bottom of the
363        stack.  On success, also sets *initial_sp to what the stack pointer
364        should be set to. */
365     
366     VgStack* VG_(am_alloc_VgStack)( /*OUT*/Addr* initial_sp )
367     {
368        Int      szB;
369        SysRes   sres;
370        VgStack* stack;
371        UInt*    p;
372        Int      i;
373     
374        /* Allocate the stack. */
375        szB = VG_STACK_GUARD_SZB 
376              + VG_STACK_ACTIVE_SZB + VG_STACK_GUARD_SZB;
377     
378        sres = VG_(am_mmap_anon_float_valgrind)( szB );
379        if (sr_isError(sres))
380           return NULL;
381     
382        stack = (VgStack*)(AddrH)sr_Res(sres);
383     
384        aspacem_assert(VG_IS_PAGE_ALIGNED(szB));
385        aspacem_assert(VG_IS_PAGE_ALIGNED(stack));
386     
387        /* Protect the guard areas. */
388        sres = local_do_mprotect_NO_NOTIFY( 
389                  (Addr) &stack[0], 
390                  VG_STACK_GUARD_SZB, VKI_PROT_NONE 
391               );
392        if (sr_isError(sres)) goto protect_failed;
393        VG_(am_notify_mprotect)( 
394           (Addr) &stack->bytes[0], 
395           VG_STACK_GUARD_SZB, VKI_PROT_NONE 
396        );
397     
398        sres = local_do_mprotect_NO_NOTIFY( 
399                  (Addr) &stack->bytes[VG_STACK_GUARD_SZB + VG_STACK_ACTIVE_SZB], 
400                  VG_STACK_GUARD_SZB, VKI_PROT_NONE 
401               );
402        if (sr_isError(sres)) goto protect_failed;
403        VG_(am_notify_mprotect)( 
404           (Addr) &stack->bytes[VG_STACK_GUARD_SZB + VG_STACK_ACTIVE_SZB],
405           VG_STACK_GUARD_SZB, VKI_PROT_NONE 
406        );
407     
408        /* Looks good.  Fill the active area with junk so we can later
409           tell how much got used. */
410     
411        p = (UInt*)&stack->bytes[VG_STACK_GUARD_SZB];
412        for (i = 0; i < VG_STACK_ACTIVE_SZB/sizeof(UInt); i++)
413           p[i] = 0xDEADBEEF;
414     
415        *initial_sp = (Addr)&stack->bytes[VG_STACK_GUARD_SZB + VG_STACK_ACTIVE_SZB];
416        *initial_sp -= 8;
417        *initial_sp &= ~((Addr)0x1F); /* 32-align it */
418     
419        VG_(debugLog)( 1,"aspacem","allocated thread stack at 0x%llx size %d\n",
420                       (ULong)(Addr)stack, szB);
421        ML_(am_do_sanity_check)();
422        return stack;
423     
424       protect_failed:
425        /* The stack was allocated, but we can't protect it.  Unmap it and
426           return NULL (failure). */
427        (void)ML_(am_do_munmap_NO_NOTIFY)( (Addr)stack, szB );
428        ML_(am_do_sanity_check)();
429        return NULL;
430     }

433     /* Figure out how many bytes of the stack's active area have not
434        been used.  Used for estimating if we are close to overflowing it. */
435     
436     SizeT VG_(am_get_VgStack_unused_szB)( VgStack* stack, SizeT limit )
437     {
438        SizeT i;
439        UInt* p;
440     
441        p = (UInt*)&stack->bytes[VG_STACK_GUARD_SZB];
442        for (i = 0; i < VG_STACK_ACTIVE_SZB/sizeof(UInt); i++) {
443           if (p[i] != 0xDEADBEEF)
444              break;
445           if (i * sizeof(UInt) >= limit)
446              break;
447        }
448     
449        return i * sizeof(UInt);
450     }
