42     /* Calls initialization routines in various modules. 
43        Note that the order of calls is important! */
44     static void it_post_clo_init(void)
45     {
46        IT_(post_clo)();        // Must be called first of all
47        IT_(sources_init)();
48        IT_(tainting_init)();
49        IT_(output_init)();
50     }
51     
52     /* Calls finalization routines in various modules */
53     static void it_fini(Int exitcode)
54     {
55     //   IT_(print_shadow_mem)();
56     //   IT_(print_taint_sink_events)();
57        IT_(write_ouput)();
58     }
59     
60     // Main pre-syscall callback.
61     static void it_pre_syscall(ThreadId tid, UInt syscallno, UWord* args, UInt nArgs)
62     {
63        IT_(sinks_pre_syscall)(tid, syscallno, args, nArgs);
64     }
65     
66     // Main post-syscall callback.
67     static void it_post_syscall(ThreadId tid, UInt syscallno, 
68                                 UWord* args, UInt nArgs, SysRes res)
69     {
70        IT_(sources_post_syscall)(tid, syscallno, args, nArgs, res);
71        IT_(tainting_post_syscall)(tid, syscallno, args, nArgs, res);
72        IT_(sinks_post_syscall)(tid, syscallno, args, nArgs, res);
73        IT_(output_post_syscall)(tid, syscallno, args, nArgs, res);
74     }
75     
76     // Main pre mem read callback
77     static void it_pre_mem_read(CorePart part, ThreadId tid, 
78                                 Char* s, Addr a, SizeT size)
79     {
80        IT_(tainting_pre_mem_read)(part, tid, s, a, size);
81        IT_(sinks_pre_mem_read)(part, tid, s, a, size);
82     }
83     
84     // Main pre mem read asciiz callback
85     static void it_pre_mem_read_asciiz(CorePart part, ThreadId tid,
86                                        Char* s, Addr a)
87     {
88        IT_(tainting_pre_mem_read_asciiz)(part, tid, s, a);
89        IT_(sinks_pre_mem_read_asciiz)(part, tid, s, a);
90     }
91     
92     // Main pre reg read callback
93     static void it_pre_reg_read(CorePart part, ThreadId tid, Char* s, 
94                                 PtrdiffT guest_state_offset, SizeT size)
95     {
96        IT_(tainting_pre_reg_read)(part, tid, s, guest_state_offset, size);
97        IT_(sinks_pre_reg_read)(part, tid, s, guest_state_offset, size);
98     }
99     
100     // Main pre-thread-first-instruction callback.
101     static void it_pre_thread(ThreadId tid)
102     {
103        IT_(tainting_pre_thread)(tid);
104        IT_(sinks_pre_thread)(tid);
105        IT_(init_stacktracing)(tid);
106     }
107     
108     // Main pre-thread-exit callback.
109     static void it_pre_thread_exit(ThreadId tid)
110     {
111        IT_(sinks_pre_thread_exit)(tid);
112        IT_(finish_stacktracing)(tid);
113     }
114     
115     static void it_pre_clo_init(void)
116     {
117        VG_(details_name)            ("InputTracer");
118        VG_(details_version)         (NULL);
119        VG_(details_description)     ("a prototype taint analysis tool");
120        VG_(details_copyright_author)(
121           "Copyright (C) 2011-2012 by Ulf Kargén.");
122        VG_(details_bug_reports_to)  ("No one");
123     
124        VG_(details_avg_translation_sizeB) ( 275 );
125     
126        VG_(basic_tool_funcs)        (it_post_clo_init,
127                                      IT_(instrument),
128                                      it_fini);
129        VG_(needs_command_line_options)(IT_(process_cmd_line_option),
130                                        IT_(print_usage),
131                                        IT_(print_debug_usage));
132     
133        /* Wrappers for syscalls etc. */      
134        VG_(needs_syscall_wrapper)    (it_pre_syscall, it_post_syscall);
135        VG_(track_pre_reg_read)       (it_pre_reg_read);
136        VG_(track_post_reg_write)     (IT_(tainting_post_reg_write));
137        VG_(track_pre_mem_read)       (it_pre_mem_read);
138        VG_(track_pre_mem_read_asciiz)(it_pre_mem_read_asciiz);
139        VG_(track_pre_mem_write)      (IT_(tainting_pre_mem_write));
140        VG_(track_die_mem_brk)        (IT_(tainting_dealloc_mem_brk));
141        VG_(track_die_mem_munmap)     (IT_(tainting_dealloc_mem_munmap));
142        VG_(track_copy_mem_remap)     (IT_(tainting_copy_mem));
143        VG_(track_pre_thread_first_insn) (it_pre_thread);
144        VG_(track_pre_thread_ll_exit) (it_pre_thread_exit);   
145     }
