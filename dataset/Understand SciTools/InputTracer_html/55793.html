37     // Extract from aspacem a vector of the current segment start
38     // addresses.  The vector is dynamically allocated and should be freed
39     // by the caller when done.  REQUIRES m_mallocfree to be running.
40     // Writes the number of addresses required into *n_acquired.
41     Addr* VG_(get_segment_starts) ( /*OUT*/Int* n_acquired )
42     {
43        Addr* starts;
44        Int   n_starts, r = 0;
45     
46        n_starts = 1;
47        while (True) {
48           starts = VG_(malloc)( "main.gss.1", n_starts * sizeof(Addr) );
49           if (starts == NULL)
50              break;
51           r = VG_(am_get_segment_starts)( starts, n_starts );
52           if (r >= 0)
53              break;
54           VG_(free)(starts);
55           n_starts *= 2;
56        }
57     
58        if (starts == NULL) {
59          *n_acquired = 0;
60          return NULL;
61        }
62     
63        *n_acquired = r;
64        return starts;
65     }
