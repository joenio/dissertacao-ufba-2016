50     /* Given the index of a function argument, return the number of the
51        general purpose register in which it is being passed. Arguments are
52        counted 0, 1, 2, ... and they are being passed in r2, r3, r4, ... */
53     static __inline__ unsigned
54     s390_gprno_from_arg_index(unsigned ix)
55     {
56        return ix + 2;
57     }

61     /* These are the address modes:
62        (1) b12:  base register + 12-bit unsigned offset   (e.g. RS)
63        (2) b20:  base register + 20-bit signed offset     (e.g. RSY)
64        (3) bx12: base register + index register + 12-bit unsigned offset (e.g. RX)
65        (4) bx20: base register + index register + 20-bit signed offset   (e.g. RXY)
66        fixs390: There is also pc-relative stuff.. e.g. LARL
67     */
68

70        S390_AMODE_B12,
71        S390_AMODE_B20,

221     /* Condition code. The encoding of the enumerators matches the value of
222        the mask field in the various branch opcodes. */
223     typedef enum {
224        S390_CC_NEVER=  0,
225        S390_CC_OVFL =  1,   /* overflow */
226        S390_CC_H    =  2,   /* A > B ; high */
227        S390_CC_NLE  =  3,   /* not low or equal */
228        S390_CC_L    =  4,   /* A < B ; low */
229        S390_CC_NHE  =  5,   /* not high or equal */
230        S390_CC_LH   =  6,   /* low or high */
231        S390_CC_NE   =  7,   /* A != B ; not zero */
232        S390_CC_E    =  8,   /* A == B ; zero */
233        S390_CC_NLH  =  9,   /* not low or high */
234        S390_CC_HE   = 10,   /* A >= B ; high or equal*/
235        S390_CC_NL   = 11,   /* not low */
236        S390_CC_LE   = 12,   /* A <= B ; low or equal */
237        S390_CC_NH   = 13,   /* not high */
238        S390_CC_NO   = 14,   /* not overflow */
239        S390_CC_ALWAYS = 15
240     } s390_cc_t;

250        S390_ROUND_NEGINF       = 7

254     /* Invert the condition code */
255     static __inline__ s390_cc_t
256     s390_cc_invert(s390_cc_t cond)
257     {
258        return S390_CC_ALWAYS - cond;

262     typedef struct s390_insn {
263        s390_insn_tag tag;
264        UChar size;            /* size of the result in bytes */
265        union {
266           struct {
267              HReg        dst;
268              s390_amode *src;
269           } load;
270           struct {
271              s390_amode *dst;
272              HReg        src;
273           } store;
274           struct {
275              HReg        dst;
276              HReg        src;
277           } move;
278           struct {
279              s390_cc_t     cond;
280              HReg          dst;
281              s390_opnd_RMI src;
282           } cond_move;
283           struct {
284              HReg        dst;
285              ULong       value;  /* not sign extended */
286           } load_immediate;
287           /* add, and, or, xor */
288           struct {
289              s390_alu_t    tag;
290              HReg          dst; /* op1 */
291              s390_opnd_RMI op2;
292           } alu;
293           struct {
294              Bool          signed_multiply;
295              HReg          dst_hi;  /*           r10 */
296              HReg          dst_lo;  /* also op1  r11 */
297              s390_opnd_RMI op2;
298           } mul;
299           struct {
300              Bool          signed_divide;
301              HReg          op1_hi;  /* also remainder   r10 */
302              HReg          op1_lo;  /* also quotient    r11 */
303              s390_opnd_RMI op2;
304           } div;
305           struct {
306              HReg          rem; /* remainder      r10 */
307              HReg          op1; /* also quotient  r11 */
308              s390_opnd_RMI op2;
309           } divs;
310           struct {
311              HReg          num_bits; /* number of leftmost '0' bits  r10 */
312              HReg          clobber;  /* unspecified                  r11 */
313              s390_opnd_RMI src;
314           } clz;
315           struct {
316              s390_unop_t   tag;
317              HReg          dst;
318              s390_opnd_RMI src;
319           } unop;
320           struct {
321              Bool          signed_comparison;
322              HReg          src1;
323              s390_opnd_RMI src2;
324           } compare;
325           struct {
326              HReg          dst;  /* condition code in s390 encoding */
327              HReg          op1;
328              HReg          op2;
329           } bfp_compare;
330           struct {
331              s390_opnd_RMI src;
332           } test;
333           /* Convert the condition code to a boolean value. */
334           struct {
335              s390_cc_t cond;
336              HReg      dst;
337           } cc2bool;
338           struct {
339              HReg        op1;
340              s390_amode *op2;
341              HReg        op3;
342              HReg        old_mem;
343           } cas;
344           struct {
345              IRJumpKind    kind;
346              s390_cc_t     cond;
347              s390_opnd_RMI dst;
348           } branch;
349           /* Pseudo-insn for representing a helper call.
350              TARGET is the absolute address of the helper function
351              NUM_ARGS says how many arguments are being passed.
352              All arguments have integer type and are being passed according to ABI,
353              i.e. in registers r2, r3, r4, r5, and r6, with argument #0 being
354              passed in r2 and so forth. */
355           struct {
356              s390_cc_t cond;
357              Addr64    target;
358              UInt      num_args;
359              HChar    *name;      /* callee's name (for debugging) */
360           } helper_call;
361           struct {
362              s390_bfp_triop_t tag;
363              s390_round_t     rounding_mode;
364              HReg             dst; /* first operand */
365              HReg             op2; /* second operand */
366              HReg             op3; /* third operand */
367           } bfp_triop;
368           struct {
369              s390_bfp_binop_t tag;
370              s390_round_t     rounding_mode;
371              HReg             dst; /* left operand */
372              HReg             op2; /* right operand */
373           } bfp_binop;
374           struct {
375              s390_bfp_unop_t tag;
376              s390_round_t    rounding_mode;
377              HReg            dst;  /* result */
378              HReg            op;   /* operand */
379           } bfp_unop;
380           struct {
381              s390_bfp_binop_t tag;
382              s390_round_t     rounding_mode;
383              HReg             dst_hi; /* left operand; high part */
384              HReg             dst_lo; /* left operand; low part */
385              HReg             op2_hi; /* right operand; high part */
386              HReg             op2_lo; /* right operand; low part */
387           } bfp128_binop;
388           /* This variant is also used by the BFP128_CONVERT_TO and
389              BFP128_CONVERT_FROM insns. */
390           struct {
391              s390_bfp_unop_t  tag;
392              s390_round_t     rounding_mode;
393              HReg             dst_hi; /* result; high part */
394              HReg             dst_lo; /* result; low part */
395              HReg             op_hi;  /* operand; high part */
396              HReg             op_lo;  /* operand; low part */
397           } bfp128_unop;
398           struct {
399              HReg             dst;    /* condition code in s390 encoding */
400              HReg             op1_hi; /* left operand; high part */
401              HReg             op1_lo; /* left operand; low part */
402              HReg             op2_hi; /* right operand; high part */
403              HReg             op2_lo; /* right operand; low part */
404           } bfp128_compare;
405        } variant;
406     } s390_insn;
