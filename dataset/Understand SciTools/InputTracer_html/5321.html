141     /* Details about the capabilities of the underlying (host) CPU.  These
142        details are acquired by (1) enquiring with the CPU at startup, or
143        (2) from the AT_SYSINFO entries the kernel gave us (ppc cache
144        line size).  It's a bit nasty in the sense that there's no obvious
145        way to stop uses of some of this info before it's ready to go.
146     
147        Current dependencies are:
148     
149        x86:   initially:  call VG_(machine_get_hwcaps)
150     
151               then safe to use VG_(machine_get_VexArchInfo) 
152                            and VG_(machine_x86_have_mxcsr)
153        -------------
154        amd64: initially:  call VG_(machine_get_hwcaps)
155     
156               then safe to use VG_(machine_get_VexArchInfo) 
157        -------------
158        ppc32: initially:  call VG_(machine_get_hwcaps)
159                           call VG_(machine_ppc32_set_clszB)
160     
161               then safe to use VG_(machine_get_VexArchInfo) 
162                            and VG_(machine_ppc32_has_FP)
163                            and VG_(machine_ppc32_has_VMX)
164        -------------
165        ppc64: initially:  call VG_(machine_get_hwcaps)
166                           call VG_(machine_ppc64_set_clszB)
167     
168               then safe to use VG_(machine_get_VexArchInfo) 
169                            and VG_(machine_ppc64_has_VMX)
170        -------------
171        arm:   initially:  call VG_(machine_get_hwcaps)
172                           call VG_(machine_arm_set_has_NEON)
173     
174               then safe to use VG_(machine_get_VexArchInfo) 
175     
176        VG_(machine_get_hwcaps) may use signals (although it attempts to
177        leave signal state unchanged) and therefore should only be
178        called before m_main sets up the client's signal state.
179     */
180     
181     /* Determine what insn set and insn set variant the host has, and
182        record it.  To be called once at system startup.  Returns False if
183        this a CPU incapable of running Valgrind. */
184     extern Bool VG_(machine_get_hwcaps)( void );
185     
186     /* Fetch host cpu info, as per above comment. */
187     extern void VG_(machine_get_VexArchInfo)( /*OUT*/VexArch*,

203     /* X86: set to 1 if the host is able to do {ld,st}mxcsr (load/store
204        the SSE control/status register), else zero.  Is referenced from
205        assembly code, so do not change from a 32-bit int. */
206     #if defined(VGA_x86)
207     extern UInt VG_(machine_x86_have_mxcsr);
