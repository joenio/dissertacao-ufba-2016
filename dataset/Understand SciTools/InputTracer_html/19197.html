60     /* Useful info which needs to be recorded somewhere:
61        Use of registers in syscalls is:
62     
63               NUM   ARG1 ARG2 ARG3 ARG4 ARG5 ARG6 ARG7 ARG8 RESULT
64        LINUX:
65        x86    eax   ebx  ecx  edx  esi  edi  ebp  n/a  n/a  eax       (== NUM)
66        amd64  rax   rdi  rsi  rdx  r10  r8   r9   n/a  n/a  rax       (== NUM)
67        ppc32  r0    r3   r4   r5   r6   r7   r8   n/a  n/a  r3+CR0.SO (== ARG1)
68        ppc64  r0    r3   r4   r5   r6   r7   r8   n/a  n/a  r3+CR0.SO (== ARG1)
69        arm    r7    r0   r1   r2   r3   r4   r5   n/a  n/a  r0        (== ARG1)
70        On s390x the svc instruction is used for system calls. The system call
71        number is encoded in the instruction (8 bit immediate field). Since Linux
72        2.6 it is also allowed to use svc 0 with the system call number in r1.
73        This was introduced for system calls >255, but works for all. It is
74        also possible to see the svc 0 together with an EXecute instruction, that
75        fills in the immediate field.
76        s390x r1/SVC r2   r3   r4   r5   r6   r7   n/a  n/a  r2        (== ARG1)
77        AIX:
78        ppc32  r2  r3   r4   r5   r6   r7   r8   r9   r10  r3(res),r4(err)
79        ppc64  r2  r3   r4   r5   r6   r7   r8   r9   r10  r3(res),r4(err)
80     
81        DARWIN:
82        x86    eax +4   +8   +12  +16  +20  +24  +28  +32  edx:eax, eflags.c
83        amd64  rax rdi  rsi  rdx  rcx  r8   r9   +8   +16  rdx:rax, rflags.c
84     
85        For x86-darwin, "+N" denotes "in memory at N(%esp)"; ditto
86        amd64-darwin.  Apparently 0(%esp) is some kind of return address
87        (perhaps for syscalls done with "sysenter"?)  I don't think it is
88        relevant for syscalls done with "int $0x80/1/2".

91     /* This is the top level of the system-call handler module.  All
92        system calls are channelled through here, doing two things:
93     
94        * notify the tool of the events (mem/reg reads, writes) happening
95     
96        * perform the syscall, usually by passing it along to the kernel
97          unmodified.
98     
99        A magical piece of assembly code, do_syscall_for_client_WRK, in
100        syscall-$PLATFORM.S does the tricky bit of passing a syscall to the
101        kernel, whilst having the simulator retain control.
102     */
103     
104     /* The main function is VG_(client_syscall).  The simulation calls it
105        whenever a client thread wants to do a syscall.  The following is a
106        sketch of what it does.
107     
108        * Ensures the root thread's stack is suitably mapped.  Tedious and
109          arcane.  See big big comment in VG_(client_syscall).
110     
111        * First, it rounds up the syscall number and args (which is a
112          platform dependent activity) and puts them in a struct ("args")
113          and also a copy in "orig_args".
114     
115          The pre/post wrappers refer to these structs and so no longer
116          need magic macros to access any specific registers.  This struct
117          is stored in thread-specific storage.
118     
119     
120        * The pre-wrapper is called, passing it a pointer to struct
121          "args".
122     
123     
124        * The pre-wrapper examines the args and pokes the tool
125          appropriately.  It may modify the args; this is why "orig_args"
126          is also stored.
127     
128          The pre-wrapper may choose to 'do' the syscall itself, and
129          concludes one of three outcomes:
130     
131            Success(N)    -- syscall is already complete, with success;
132                             result is N
133     
134            Fail(N)       -- syscall is already complete, with failure;
135                             error code is N
136     
137            HandToKernel  -- (the usual case): this needs to be given to
138                             the kernel to be done, using the values in
139                             the possibly-modified "args" struct.
140     
141          In addition, the pre-wrapper may set some flags:
142     
143            MayBlock   -- only applicable when outcome==HandToKernel
144     
145            PostOnFail -- only applicable when outcome==HandToKernel or Fail
146     
147     
148        * If the pre-outcome is HandToKernel, the syscall is duly handed
149          off to the kernel (perhaps involving some thread switchery, but
150          that's not important).  This reduces the possible set of outcomes
151          to either Success(N) or Fail(N).
152     
153     
154        * The outcome (Success(N) or Fail(N)) is written back to the guest
155          register(s).  This is platform specific:
156     
157          x86:    Success(N) ==>  eax = N
158                  Fail(N)    ==>  eax = -N
159     
160          ditto amd64
161     
162          ppc32:  Success(N) ==>  r3 = N, CR0.SO = 0
163                  Fail(N) ==>     r3 = N, CR0.SO = 1
164     
165          Darwin:
166          x86:    Success(N) ==>  edx:eax = N, cc = 0
167                  Fail(N)    ==>  edx:eax = N, cc = 1
168     
169          s390x:  Success(N) ==>  r2 = N
170                  Fail(N)    ==>  r2 = -N
171     
172        * The post wrapper is called if:
173     
174          - it exists, and
175          - outcome==Success or (outcome==Fail and PostOnFail is set)
176     
177          The post wrapper is passed the adulterated syscall args (struct
178          "args"), and the syscall outcome (viz, Success(N) or Fail(N)).
179     
180        There are several other complications, primarily to do with
181        syscalls getting interrupted, explained in comments in the code.
182     */
183     
184     /* CAVEATS for writing wrappers.  It is important to follow these!
185     
186        The macros defined in priv_types_n_macros.h are designed to help
187        decouple the wrapper logic from the actual representation of
188        syscall args/results, since these wrappers are designed to work on
189        multiple platforms.
190     
191        Sometimes a PRE wrapper will complete the syscall itself, without
192        handing it to the kernel.  It will use one of SET_STATUS_Success,
193        SET_STATUS_Failure or SET_STATUS_from_SysRes to set the return
194        value.  It is critical to appreciate that use of the macro does not
195        immediately cause the underlying guest state to be updated -- that
196        is done by the driver logic in this file, when the wrapper returns.
197     
198        As a result, PRE wrappers of the following form will malfunction:
199     
200        PRE(fooble) 
201        {
202           ... do stuff ...
203           SET_STATUS_Somehow(...)
204     
205           // do something that assumes guest state is up to date
206        }
207     
208        In particular, direct or indirect calls to VG_(poll_signals) after
209        setting STATUS can cause the guest state to be read (in order to
210        build signal frames).  Do not do this.  If you want a signal poll
211        after the syscall goes through, do "*flags |= SfPollAfter" and the
212        driver logic will do it for you.
213     
214        -----------
215     
216        Another critical requirement following introduction of new address
217        space manager (JRS, 20050923):
218     
219        In a situation where the mappedness of memory has changed, aspacem
220        should be notified BEFORE the tool.  Hence the following is
221        correct:
222     
223           Bool d = VG_(am_notify_munmap)(s->start, s->end+1 - s->start);
224           VG_TRACK( die_mem_munmap, s->start, s->end+1 - s->start );
225           if (d)
226              VG_(discard_translations)(s->start, s->end+1 - s->start);
227     
228        whilst this is wrong:
229     
230           VG_TRACK( die_mem_munmap, s->start, s->end+1 - s->start );
231           Bool d = VG_(am_notify_munmap)(s->start, s->end+1 - s->start);
232           if (d)
233              VG_(discard_translations)(s->start, s->end+1 - s->start);
234     
235        The reason is that the tool may itself ask aspacem for more shadow
236        memory as a result of the VG_TRACK call.  In such a situation it is
237        critical that aspacem's segment array is up to date -- hence the
238        need to notify aspacem first.
239     
240        -----------
241     
242        Also .. take care to call VG_(discard_translations) whenever
243        memory with execute permissions is unmapped.
244     */

252     /* Perform a syscall on behalf of a client thread, using a specific
253        signal mask.  On completion, the signal mask is set to restore_mask
254        (which presumably blocks almost everything).  If a signal happens
255        during the syscall, the handler should call
256        VG_(fixup_guest_state_after_syscall_interrupted) to adjust the
257        thread's context to do the right thing.
258     
259        The _WRK function is handwritten assembly, implemented per-platform
260        in coregrind/m_syswrap/syscall-$PLAT.S.  It has some very magic
261        properties.  See comments at the top of
262        VG_(fixup_guest_state_after_syscall_interrupted) below for details.
263     
264        This function (these functions) are required to return zero in case
265        of success (even if the syscall itself failed), and nonzero if the
266        sigprocmask-swizzling calls failed.  We don't actually care about
267        the failure values from sigprocmask, although most of the assembly
268        implementations do attempt to return that, using the convention
269        0 for success, or 0x8000 | error-code for failure.
270     */

310     static
311     void do_syscall_for_client ( Int syscallno,
312                                  ThreadState* tst,
313                                  const vki_sigset_t* syscall_mask )
314     {
315        vki_sigset_t saved;
316        UWord err;
317     #  if defined(VGO_linux)
318        err = ML_(do_syscall_for_client_WRK)(
319                 syscallno, &tst->arch.vex, 
320                 syscall_mask, &saved, sizeof(vki_sigset_t)
321              );
322     #  elif defined(VGO_aix5)
323        err = ML_(do_syscall_for_client_WRK)(
324                 syscallno, &tst->arch.vex, 
325                 syscall_mask, &saved, 0/*unused:sigsetSzB*/,
326                 __NR_rt_sigprocmask
327              );
328     #  elif defined(VGO_darwin)
329        switch (VG_DARWIN_SYSNO_CLASS(syscallno)) {
330           case VG_DARWIN_SYSCALL_CLASS_UNIX:
331              err = ML_(do_syscall_for_client_unix_WRK)(
332                       VG_DARWIN_SYSNO_FOR_KERNEL(syscallno), &tst->arch.vex, 
333                       syscall_mask, &saved, 0/*unused:sigsetSzB*/
334                    );
335              break;
336           case VG_DARWIN_SYSCALL_CLASS_MACH:
337              err = ML_(do_syscall_for_client_mach_WRK)(
338                       VG_DARWIN_SYSNO_FOR_KERNEL(syscallno), &tst->arch.vex, 
339                       syscall_mask, &saved, 0/*unused:sigsetSzB*/
340                    );
341              break;
342           case VG_DARWIN_SYSCALL_CLASS_MDEP:
343              err = ML_(do_syscall_for_client_mdep_WRK)(
344                       VG_DARWIN_SYSNO_FOR_KERNEL(syscallno), &tst->arch.vex, 
345                       syscall_mask, &saved, 0/*unused:sigsetSzB*/
346                    );
347              break;
348           default:
349              vg_assert(0);
350              /*NOTREACHED*/
351              break;
352        }
353     #  else
354     #    error "Unknown OS"
355     #  endif
356        vg_assert2(
357           err == 0,
358           "ML_(do_syscall_for_client_WRK): sigprocmask error %d",
359           (Int)(err & 0xFFF)
360        );
361     }

369     Bool eq_SyscallArgs ( SyscallArgs* a1, SyscallArgs* a2 )
370     {
371        return a1->sysno == a2->sysno
372               && a1->arg1 == a2->arg1
373               && a1->arg2 == a2->arg2
374               && a1->arg3 == a2->arg3
375               && a1->arg4 == a2->arg4
376               && a1->arg5 == a2->arg5
377               && a1->arg6 == a2->arg6
378               && a1->arg7 == a2->arg7
379               && a1->arg8 == a2->arg8;
380     }

382     static
383     Bool eq_SyscallStatus ( SyscallStatus* s1, SyscallStatus* s2 )
384     {
385        /* was: return s1->what == s2->what && sr_EQ( s1->sres, s2->sres ); */
386        if (s1->what == s2->what && sr_EQ( s1->sres, s2->sres ))
387           return True;
388     #  if defined(VGO_darwin)
389        /* Darwin-specific debugging guff */
390        vg_assert(s1->what == s2->what);
391        VG_(printf)("eq_SyscallStatus:\n");
392        VG_(printf)("  {%lu %lu %u}\n", s1->sres._wLO, s1->sres._wHI, s1->sres._mode);
393        VG_(printf)("  {%lu %lu %u}\n", s2->sres._wLO, s2->sres._wHI, s2->sres._mode);
394        vg_assert(0);
395     #  endif
396        return False;
397     }

552           canonical->sysno = VG_DARWIN_SYSCALL_CONSTRUCT_UNIX(canonical->sysno);
553           break;
554        case VEX_TRC_JMP_SYS_SYSENTER:
555           // syscall = Unix, 32-bit result
556           // OR        Mach, 32-bit result
557           if (canonical->sysno >= 0) {
558              // GrP fixme hack:  0xffff == I386_SYSCALL_NUMBER_MASK
559              canonical->sysno = VG_DARWIN_SYSCALL_CONSTRUCT_UNIX(canonical->sysno
560                                                                  & 0xffff);
561           } else {
562              canonical->sysno = VG_DARWIN_SYSCALL_CONSTRUCT_MACH(-canonical->sysno);
563           }
564           break;
565        case VEX_TRC_JMP_SYS_INT129:
566           // int $0x81 = Mach, 32-bit result
567           vg_assert(canonical->sysno < 0);
568           canonical->sysno = VG_DARWIN_SYSCALL_CONSTRUCT_MACH(-canonical->sysno);
569           break;
570        case VEX_TRC_JMP_SYS_INT130:
571           // int $0x82 = mdep, 32-bit result
572           vg_assert(canonical->sysno >= 0);
573           canonical->sysno = VG_DARWIN_SYSCALL_CONSTRUCT_MDEP(canonical->sysno);
574           break;
575        default: 
576           vg_assert(0);
577           break;
578        }
579        
580     #elif defined(VGP_amd64_darwin)
581        VexGuestAMD64State* gst = (VexGuestAMD64State*)gst_vanilla;
582        UWord *stack = (UWord *)gst->guest_RSP;
583     
584        vg_assert(trc == VEX_TRC_JMP_SYS_SYSCALL);
585     
586        // GrP fixme hope syscalls aren't called with really shallow stacks...
587        canonical->sysno = gst->guest_RAX;
588        if (canonical->sysno != __NR_syscall) {
589           // stack[0] is return address
590           canonical->arg1  = gst->guest_RDI;
591           canonical->arg2  = gst->guest_RSI;
592           canonical->arg3  = gst->guest_RDX;
593           canonical->arg4  = gst->guest_R10;  // not rcx with syscall insn
594           canonical->arg5  = gst->guest_R8;
595           canonical->arg6  = gst->guest_R9;
596           canonical->arg7  = stack[1];
597           canonical->arg8  = stack[2];
598        } else {
599           // GrP fixme hack handle syscall()
600           // GrP fixme what about __syscall() ?
601           // stack[0] is return address
602           // DDD: the tool can't see that the params have been shifted!  Can
603           //      lead to incorrect checking, I think, because the PRRAn/PSARn
604           //      macros will mention the pre-shifted args.
605           canonical->sysno = VG_DARWIN_SYSCALL_CONSTRUCT_UNIX(gst->guest_RDI);
606           vg_assert(canonical->sysno != __NR_syscall);
607           canonical->arg1  = gst->guest_RSI;
608           canonical->arg2  = gst->guest_RDX;
609           canonical->arg3  = gst->guest_R10;  // not rcx with syscall insn
610           canonical->arg4  = gst->guest_R8;
611           canonical->arg5  = gst->guest_R9;
612           canonical->arg6  = stack[1];
613           canonical->arg7  = stack[2];
614           canonical->arg8  = stack[3];
615           
616           PRINT("SYSCALL[%d,?](%s) syscall(%s, ...); please stand by...\n",
617                 VG_(getpid)(), /*tid,*/
618                 VG_SYSNUM_STRING(0), VG_SYSNUM_STRING(canonical->sysno));
619        }
620     
621        // no canonical->sysno adjustment needed
622     
623     #elif defined(VGP_s390x_linux)
624        VexGuestS390XState* gst = (VexGuestS390XState*)gst_vanilla;
625        canonical->sysno = gst->guest_SYSNO;
626        canonical->arg1  = gst->guest_r2;
627        canonical->arg2  = gst->guest_r3;
628        canonical->arg3  = gst->guest_r4;
629        canonical->arg4  = gst->guest_r5;
630        canonical->arg5  = gst->guest_r6;
631        canonical->arg6  = gst->guest_r7;
632        canonical->arg7  = 0;
633        canonical->arg8  = 0;
634     #else
635     #  error "getSyscallArgsFromGuestState: unknown arch"
636     #endif
637     }

1310     /* --- This is the main function of this file. --- */
1311     
1312     void VG_(client_syscall) ( ThreadId tid, UInt trc )
1313     {
1314        Word                     sysno;
1315        ThreadState*             tst;
1316        const SyscallTableEntry* ent;
1317        SyscallArgLayout         layout;
1318        SyscallInfo*             sci;
1319     
1320        ensure_initialised();
1321     
1322        vg_assert(VG_(is_valid_tid)(tid));
1323        vg_assert(tid >= 1 && tid < VG_N_THREADS);
1324        vg_assert(VG_(is_running_thread)(tid));
1325     
1326        tst = VG_(get_ThreadState)(tid);
1327     
1328        /* BEGIN ensure root thread's stack is suitably mapped */
1329        /* In some rare circumstances, we may do the syscall without the
1330           bottom page of the stack being mapped, because the stack pointer
1331           was moved down just a few instructions before the syscall
1332           instruction, and there have been no memory references since
1333           then, that would cause a call to VG_(extend_stack) to have
1334           happened.
1335     
1336           In native execution that's OK: the kernel automagically extends
1337           the stack's mapped area down to cover the stack pointer (or sp -
1338           redzone, really).  In simulated normal execution that's OK too,
1339           since any signals we get from accessing below the mapped area of
1340           the (guest's) stack lead us to VG_(extend_stack), where we
1341           simulate the kernel's stack extension logic.  But that leaves
1342           the problem of entering a syscall with the SP unmapped.  Because
1343           the kernel doesn't know that the segment immediately above SP is
1344           supposed to be a grow-down segment, it causes the syscall to
1345           fail, and thereby causes a divergence between native behaviour
1346           (syscall succeeds) and simulated behaviour (syscall fails).
1347     
1348           This is quite a rare failure mode.  It has only been seen
1349           affecting calls to sys_readlink on amd64-linux, and even then it
1350           requires a certain code sequence around the syscall to trigger
1351           it.  Here is one:
1352     
1353           extern int my_readlink ( const char* path );
1354           asm(
1355           ".text\n"
1356           ".globl my_readlink\n"
1357           "my_readlink:\n"
1358           "\tsubq    $0x1008,%rsp\n"
1359           "\tmovq    %rdi,%rdi\n"              // path is in rdi
1360           "\tmovq    %rsp,%rsi\n"              // &buf[0] -> rsi
1361           "\tmovl    $0x1000,%edx\n"           // sizeof(buf) in rdx
1362           "\tmovl    $"__NR_READLINK",%eax\n"  // syscall number
1363           "\tsyscall\n"
1364           "\taddq    $0x1008,%rsp\n"
1365           "\tret\n"
1366           ".previous\n"
1367           );
1368     
1369           For more details, see bug #156404
1370           (https://bugs.kde.org/show_bug.cgi?id=156404).
1371     
1372           The fix is actually very simple.  We simply need to call
1373           VG_(extend_stack) for this thread, handing it the lowest
1374           possible valid address for stack (sp - redzone), to ensure the
1375           pages all the way down to that address, are mapped.  Because
1376           this is a potentially expensive and frequent operation, we
1377           filter in two ways:
1378     
1379           First, only the main thread (tid=1) has a growdown stack.  So
1380           ignore all others.  It is conceivable, although highly unlikely,
1381           that the main thread exits, and later another thread is
1382           allocated tid=1, but that's harmless, I believe;
1383           VG_(extend_stack) will do nothing when applied to a non-root
1384           thread.
1385     
1386           Secondly, first call VG_(am_find_nsegment) directly, to see if
1387           the page holding (sp - redzone) is mapped correctly.  If so, do
1388           nothing.  This is almost always the case.  VG_(extend_stack)
1389           calls VG_(am_find_nsegment) twice, so this optimisation -- and
1390           that's all it is -- more or less halves the number of calls to
1391           VG_(am_find_nsegment) required.
1392     
1393           TODO: the test "seg->kind == SkAnonC" is really inadequate,
1394           because although it tests whether the segment is mapped
1395           _somehow_, it doesn't check that it has the right permissions
1396           (r,w, maybe x) ?  We could test that here, but it will also be
1397           necessary to fix the corresponding test in VG_(extend_stack).
1398     
1399           All this guff is of course Linux-specific.  Hence the ifdef.
1400        */
1401     #  if defined(VGO_linux)
1402        if (tid == 1/*ROOT THREAD*/) {
1403           Addr     stackMin   = VG_(get_SP)(tid) - VG_STACK_REDZONE_SZB;
1404           NSegment const* seg = VG_(am_find_nsegment)(stackMin);
1405           if (seg && seg->kind == SkAnonC) {
1406              /* stackMin is already mapped.  Nothing to do. */
1407           } else {
1408              (void)VG_(extend_stack)( stackMin,
1409                                       tst->client_stack_szB );
1410           }
1411        }
1412     #  endif
1413        /* END ensure root thread's stack is suitably mapped */
1414     
1415        /* First off, get the syscall args and number.  This is a
1416           platform-dependent action. */
1417     
1418        sci = & syscallInfo[tid];
1419        vg_assert(sci->status.what == SsIdle);
1420     
1421        getSyscallArgsFromGuestState( &sci->orig_args, &tst->arch.vex, trc );
1422     
1423        /* Copy .orig_args to .args.  The pre-handler may modify .args, but
1424           we want to keep the originals too, just in case. */
1425        sci->args = sci->orig_args;
1426     
1427        /* Save the syscall number in the thread state in case the syscall 
1428           is interrupted by a signal. */
1429        sysno = sci->orig_args.sysno;
1430     
1431     #  if defined(VGO_darwin)
1432        /* Record syscall class.  But why?  Because the syscall might be
1433           interrupted by a signal, and in the signal handler (which will
1434           be m_signals.async_signalhandler) we will need to build a SysRes
1435           reflecting the syscall return result.  In order to do that we
1436           need to know the syscall class.  Hence stash it in the guest
1437           state of this thread.  This madness is not needed on Linux or
1438           AIX5, because those OSs only have a single syscall return
1439           convention and so there is no ambiguity involved in converting
1440           the post-signal machine state into a SysRes. */
1441        tst->arch.vex.guest_SC_CLASS = VG_DARWIN_SYSNO_CLASS(sysno);
1442     #  endif
1443     
1444        /* The default what-to-do-next thing is hand the syscall to the
1445           kernel, so we pre-set that here.  Set .sres to something
1446           harmless looking (is irrelevant because .what is not
1447           SsComplete.) */
1448        sci->status.what = SsHandToKernel;
1449        sci->status.sres = VG_(mk_SysRes_Error)(0);
1450        sci->flags       = 0;
1451     
1452        /* Fetch the syscall's handlers.  If no handlers exist for this
1453           syscall, we are given dummy handlers which force an immediate
1454           return with ENOSYS. */
1455        ent = get_syscall_entry(sysno);
1456     
1457        /* Fetch the layout information, which tells us where in the guest
1458           state the syscall args reside.  This is a platform-dependent
1459           action.  This info is needed so that the scalar syscall argument
1460           checks (PRE_REG_READ calls) know which bits of the guest state
1461           they need to inspect. */
1462        getSyscallArgLayout( &layout );
1463     
1464        /* Make sure the tmp signal mask matches the real signal mask;
1465           sigsuspend may change this. */
1466        vg_assert(VG_(iseqsigset)(&tst->sig_mask, &tst->tmp_sig_mask));
1467     
1468        /* Right, we're finally ready to Party.  Call the pre-handler and
1469           see what we get back.  At this point: 
1470     
1471             sci->status.what  is Unset (we don't know yet).
1472             sci->orig_args    contains the original args.
1473             sci->args         is the same as sci->orig_args.
1474             sci->flags        is zero.
1475        */
1476     
1477        PRINT("SYSCALL[%d,%d](%s) ",
1478           VG_(getpid)(), tid, VG_SYSNUM_STRING(sysno));
1479     
1480        /* Do any pre-syscall actions */
1481        if (VG_(needs).syscall_wrapper) {
1482           UWord tmpv[8];
1483           tmpv[0] = sci->orig_args.arg1;
1484           tmpv[1] = sci->orig_args.arg2;
1485           tmpv[2] = sci->orig_args.arg3;
1486           tmpv[3] = sci->orig_args.arg4;
1487           tmpv[4] = sci->orig_args.arg5;
1488           tmpv[5] = sci->orig_args.arg6;
1489           tmpv[6] = sci->orig_args.arg7;
1490           tmpv[7] = sci->orig_args.arg8;
1491           VG_TDICT_CALL(tool_pre_syscall, tid, sysno,
1492                         &tmpv[0], sizeof(tmpv)/sizeof(tmpv[0]));
1493        }
1494     
1495        vg_assert(ent);
1496        vg_assert(ent->before);
1497        (ent->before)( tid,
1498                       &layout, 
1499                       &sci->args, &sci->status, &sci->flags );
1500        
1501        /* The pre-handler may have modified:
1502              sci->args
1503              sci->status
1504              sci->flags
1505           All else remains unchanged. 
1506           Although the args may be modified, pre handlers are not allowed
1507           to change the syscall number.
1508        */
1509        /* Now we proceed according to what the pre-handler decided. */
1510        vg_assert(sci->status.what == SsHandToKernel
1511                  || sci->status.what == SsComplete);
1512        vg_assert(sci->args.sysno == sci->orig_args.sysno);
1513     
1514        if (sci->status.what == SsComplete && !sr_isError(sci->status.sres)) {
1515           /* The pre-handler completed the syscall itself, declaring
1516              success. */
1517           if (sci->flags & SfNoWriteResult) {
1518              PRINT(" --> [pre-success] NoWriteResult");
1519           } else {
1520              PRINT(" --> [pre-success] Success(0x%llx:0x%llx)",
1521                    (ULong)sr_ResHI(sci->status.sres),
1522                    (ULong)sr_Res(sci->status.sres));
1523           }                                      
1524           /* In this case the allowable flags are to ask for a signal-poll
1525              and/or a yield after the call.  Changing the args isn't
1526              allowed. */
1527           vg_assert(0 == (sci->flags 
1528                           & ~(SfPollAfter | SfYieldAfter | SfNoWriteResult)));
1529           vg_assert(eq_SyscallArgs(&sci->args, &sci->orig_args));
1530        }
1531     
1532        else
1533        if (sci->status.what == SsComplete && sr_isError(sci->status.sres)) {
1534           /* The pre-handler decided to fail syscall itself. */
1535           PRINT(" --> [pre-fail] Failure(0x%llx)", (ULong)sr_Err(sci->status.sres));
1536           /* In this case, the pre-handler is also allowed to ask for the
1537              post-handler to be run anyway.  Changing the args is not
1538              allowed. */
1539           vg_assert(0 == (sci->flags & ~(SfMayBlock | SfPostOnFail | SfPollAfter)));
1540           vg_assert(eq_SyscallArgs(&sci->args, &sci->orig_args));
1541        }
1542     
1543        else
1544        if (sci->status.what != SsHandToKernel) {
1545           /* huh?! */
1546           vg_assert(0);
1547        }
1548     
1549        else /* (sci->status.what == HandToKernel) */ {
1550           /* Ok, this is the usual case -- and the complicated one.  There
1551              are two subcases: sync and async.  async is the general case
1552              and is to be used when there is any possibility that the
1553              syscall might block [a fact that the pre-handler must tell us
1554              via the sci->flags field.]  Because the tidying-away /
1555              context-switch overhead of the async case could be large, if
1556              we are sure that the syscall will not block, we fast-track it
1557              by doing it directly in this thread, which is a lot
1558              simpler. */
1559     
1560           /* Check that the given flags are allowable: MayBlock, PollAfter
1561              and PostOnFail are ok. */
1562           vg_assert(0 == (sci->flags & ~(SfMayBlock | SfPostOnFail | SfPollAfter)));
1563     
1564           if (sci->flags & SfMayBlock) {
1565     
1566              /* Syscall may block, so run it asynchronously */
1567              vki_sigset_t mask;
1568     
1569              PRINT(" --> [async] ... \n");
1570     
1571              mask = tst->sig_mask;
1572              sanitize_client_sigmask(&mask);
1573     
1574              /* Gack.  More impedance matching.  Copy the possibly
1575                 modified syscall args back into the guest state. */
1576              /* JRS 2009-Mar-16: if the syscall args are possibly modified,
1577                 then this assertion is senseless:
1578                   vg_assert(eq_SyscallArgs(&sci->args, &sci->orig_args));
1579                 The case that exposed it was sys_posix_spawn on Darwin,
1580                 which heavily modifies its arguments but then lets the call
1581                 go through anyway, with SfToBlock set, hence we end up here. */
1582              putSyscallArgsIntoGuestState( &sci->args, &tst->arch.vex );
1583     
1584              /* Drop the bigLock */
1585              VG_(release_BigLock)(tid, VgTs_WaitSys, "VG_(client_syscall)[async]");
1586              /* Urr.  We're now in a race against other threads trying to
1587                 acquire the bigLock.  I guess that doesn't matter provided
1588                 that do_syscall_for_client only touches thread-local
1589                 state. */
1590     
1591              /* Do the call, which operates directly on the guest state,
1592                 not on our abstracted copies of the args/result. */
1593              do_syscall_for_client(sysno, tst, &mask);
1594     
1595              /* do_syscall_for_client may not return if the syscall was
1596                 interrupted by a signal.  In that case, flow of control is
1597                 first to m_signals.async_sighandler, which calls
1598                 VG_(fixup_guest_state_after_syscall_interrupted), which
1599                 fixes up the guest state, and possibly calls
1600                 VG_(post_syscall).  Once that's done, control drops back
1601                 to the scheduler.  */
1602     
1603              /* Darwin: do_syscall_for_client may not return if the 
1604                 syscall was workq_ops(WQOPS_THREAD_RETURN) and the kernel 
1605                 responded by starting the thread at wqthread_hijack(reuse=1)
1606                 (to run another workqueue item). In that case, wqthread_hijack 
1607                 calls ML_(wqthread_continue), which is similar to 
1608                 VG_(fixup_guest_state_after_syscall_interrupted). */
1609     
1610              /* Reacquire the lock */
1611              VG_(acquire_BigLock)(tid, "VG_(client_syscall)[async]");
1612     
1613              /* Even more impedance matching.  Extract the syscall status
1614                 from the guest state. */
1615              getSyscallStatusFromGuestState( &sci->status, &tst->arch.vex );
1616              vg_assert(sci->status.what == SsComplete);
1617     
1618              /* Be decorative, if required. */
1619              if (VG_(clo_trace_syscalls)) {
1620                 Bool failed = sr_isError(sci->status.sres);
1621                 if (failed) {
1622                    PRINT("SYSCALL[%d,%d](%s) ... [async] --> Failure(0x%llx)",
1623                          VG_(getpid)(), tid, VG_SYSNUM_STRING(sysno),
1624                          (ULong)sr_Err(sci->status.sres));
1625                 } else {
1626                    PRINT("SYSCALL[%d,%d](%s) ... [async] --> "
1627                          "Success(0x%llx:0x%llx)",
1628                          VG_(getpid)(), tid, VG_SYSNUM_STRING(sysno),
1629                          (ULong)sr_ResHI(sci->status.sres),
1630                          (ULong)sr_Res(sci->status.sres) );
1631                 }
1632              }
1633     
1634           } else {
1635     
1636              /* run the syscall directly */
1637              /* The pre-handler may have modified the syscall args, but
1638                 since we're passing values in ->args directly to the
1639                 kernel, there's no point in flushing them back to the
1640                 guest state.  Indeed doing so could be construed as
1641                 incorrect. */
1642              SysRes sres 
1643                 = VG_(do_syscall)(sysno, sci->args.arg1, sci->args.arg2, 
1644                                          sci->args.arg3, sci->args.arg4, 
1645                                          sci->args.arg5, sci->args.arg6,
1646                                          sci->args.arg7, sci->args.arg8 );
1647              sci->status = convert_SysRes_to_SyscallStatus(sres);
1648     
1649              /* Be decorative, if required. */
1650              if (VG_(clo_trace_syscalls)) {
1651                 Bool failed = sr_isError(sci->status.sres);
1652                 if (failed) {
1653                    PRINT("[sync] --> Failure(0x%llx)",
1654                          (ULong)sr_Err(sci->status.sres) );
1655                 } else {
1656                    PRINT("[sync] --> Success(0x%llx:0x%llx)",
1657                          (ULong)sr_ResHI(sci->status.sres),
1658                          (ULong)sr_Res(sci->status.sres) );
1659                 }
1660              }
1661           }
1662        }
1663     
1664        vg_assert(sci->status.what == SsComplete);
1665     
1666        vg_assert(VG_(is_running_thread)(tid));
1667     
1668        /* Dump the syscall result back in the guest state.  This is
1669           a platform-specific action. */
1670        if (!(sci->flags & SfNoWriteResult))
1671           putSyscallStatusIntoGuestState( tid, &sci->status, &tst->arch.vex );
1672     
1673        /* Situation now:
1674           - the guest state is now correctly modified following the syscall
1675           - modified args, original args and syscall status are still
1676             available in the syscallInfo[] entry for this syscall.
1677     
1678           Now go on to do the post-syscall actions (read on down ..)
1679        */
1680        PRINT(" ");
1681        VG_(post_syscall)(tid);
1682        PRINT("\n");
1683     }
1684     
1685     
1686     /* Perform post syscall actions.  The expected state on entry is
1687        precisely as at the end of VG_(client_syscall), that is:
1688     
1689        - guest state up to date following the syscall
1690        - modified args, original args and syscall status are still
1691          available in the syscallInfo[] entry for this syscall.
1692        - syscall status matches what's in the guest state.
1693     
1694        There are two ways to get here: the normal way -- being called by
1695        VG_(client_syscall), and the unusual way, from
1696        VG_(fixup_guest_state_after_syscall_interrupted).
1697        Darwin: there's a third way, ML_(wqthread_continue). 
1698     */
1699     void VG_(post_syscall) (ThreadId tid)
1700     {
1701        SyscallInfo*             sci;
1702        const SyscallTableEntry* ent;
1703        SyscallStatus            test_status;
1704        ThreadState*             tst;
1705        Word sysno;
1706     
1707        /* Preliminaries */
1708        vg_assert(VG_(is_valid_tid)(tid));
1709        vg_assert(tid >= 1 && tid < VG_N_THREADS);
1710        vg_assert(VG_(is_running_thread)(tid));
1711     
1712        tst = VG_(get_ThreadState)(tid);
1713        sci = & syscallInfo[tid];
1714     
1715        /* m_signals.sigvgkill_handler might call here even when not in
1716           a syscall. */
1717        if (sci->status.what == SsIdle || sci->status.what == SsHandToKernel) {
1718           sci->status.what = SsIdle;
1719           return;
1720        }
1721     
1722        /* Validate current syscallInfo entry.  In particular we require
1723           that the current .status matches what's actually in the guest
1724           state.  At least in the normal case where we have actually
1725           previously written the result into the guest state. */
1726        vg_assert(sci->status.what == SsComplete);
1727     
1728        getSyscallStatusFromGuestState( &test_status, &tst->arch.vex );
1729        if (!(sci->flags & SfNoWriteResult))
1730           vg_assert(eq_SyscallStatus( &sci->status, &test_status ));
1731        /* Failure of the above assertion on Darwin can indicate a problem
1732           in the syscall wrappers that pre-fail or pre-succeed the
1733           syscall, by calling SET_STATUS_Success or SET_STATUS_Failure,
1734           when they really should call SET_STATUS_from_SysRes.  The former
1735           create a UNIX-class syscall result on Darwin, which may not be
1736           correct for the syscall; if that's the case then this assertion
1737           fires.  See PRE(thread_fast_set_cthread_self) for an example.  On
1738           non-Darwin platforms this assertion is should never fail, and this
1739           comment is completely irrelevant. */
1740        /* Ok, looks sane */
1741     
1742        /* Get the system call number.  Because the pre-handler isn't
1743           allowed to mess with it, it should be the same for both the
1744           original and potentially-modified args. */
1745        vg_assert(sci->args.sysno == sci->orig_args.sysno);
1746        sysno = sci->args.sysno;
1747        ent = get_syscall_entry(sysno);
1748     
1749        /* pre: status == Complete (asserted above) */
1750        /* Consider either success or failure.  Now run the post handler if:
1751           - it exists, and
1752           - Success or (Failure and PostOnFail is set)
1753        */
1754        if (ent->after
1755            && ((!sr_isError(sci->status.sres))
1756                || (sr_isError(sci->status.sres)
1757                    && (sci->flags & SfPostOnFail) ))) {
1758     
1759           (ent->after)( tid, &sci->args, &sci->status );
1760        }
1761     
1762        /* Because the post handler might have changed the status (eg, the
1763           post-handler for sys_open can change the result from success to
1764           failure if the kernel supplied a fd that it doesn't like), once
1765           again dump the syscall result back in the guest state.*/
1766        if (!(sci->flags & SfNoWriteResult))
1767           putSyscallStatusIntoGuestState( tid, &sci->status, &tst->arch.vex );
1768     
1769        /* Do any post-syscall actions required by the tool. */
1770        if (VG_(needs).syscall_wrapper) {
1771           UWord tmpv[8];
1772           tmpv[0] = sci->orig_args.arg1;
1773           tmpv[1] = sci->orig_args.arg2;
1774           tmpv[2] = sci->orig_args.arg3;
1775           tmpv[3] = sci->orig_args.arg4;
1776           tmpv[4] = sci->orig_args.arg5;
1777           tmpv[5] = sci->orig_args.arg6;
1778           tmpv[6] = sci->orig_args.arg7;
1779           tmpv[7] = sci->orig_args.arg8;
1780           VG_TDICT_CALL(tool_post_syscall, tid, 
1781                         sysno,
1782                         &tmpv[0], sizeof(tmpv)/sizeof(tmpv[0]),
1783                         sci->status.sres);
1784        }
1785     
1786        /* The syscall is done. */
1787        vg_assert(sci->status.what == SsComplete);
1788        sci->status.what = SsIdle;
1789     
1790        /* The pre/post wrappers may have concluded that pending signals
1791           might have been created, and will have set SfPollAfter to
1792           request a poll for them once the syscall is done. */
1793        if (sci->flags & SfPollAfter)
1794           VG_(poll_signals)(tid);
1795     
1796        /* Similarly, the wrappers might have asked for a yield
1797           afterwards. */
1798        if (sci->flags & SfYieldAfter)
1799           VG_(vg_yield)();
1800     }

1860     /* Back up guest state to restart a system call. */
1861     
1862     void ML_(fixup_guest_state_to_restart_syscall) ( ThreadArchState* arch )
1863     {
1864     #if defined(VGP_x86_linux)
1865        arch->vex.guest_EIP -= 2;             // sizeof(int $0x80)
1866     
1867        /* Make sure our caller is actually sane, and we're really backing
1868           back over a syscall.
1869     
1870           int $0x80 == CD 80 
1871        */
1872        {
1873           UChar *p = (UChar *)arch->vex.guest_EIP;
1874           
1875           if (p[0] != 0xcd || p[1] != 0x80)
1876              VG_(message)(Vg_DebugMsg,
1877                           "?! restarting over syscall at %#x %02x %02x\n",
1878                           arch->vex.guest_EIP, p[0], p[1]); 
1879     
1880           vg_assert(p[0] == 0xcd && p[1] == 0x80);
1881        }
1882     
1883     #elif defined(VGP_amd64_linux)
1884        arch->vex.guest_RIP -= 2;             // sizeof(syscall)
1885     
1886        /* Make sure our caller is actually sane, and we're really backing
1887           back over a syscall.
1888     
1889           syscall == 0F 05 
1890        */
1891        {
1892           UChar *p = (UChar *)arch->vex.guest_RIP;
1893           
1894           if (p[0] != 0x0F || p[1] != 0x05)
1895              VG_(message)(Vg_DebugMsg,
1896                           "?! restarting over syscall at %#llx %02x %02x\n",
1897                           arch->vex.guest_RIP, p[0], p[1]); 
1898     
1899           vg_assert(p[0] == 0x0F && p[1] == 0x05);
1900        }
1901     
1902     #elif defined(VGP_ppc32_linux) || defined(VGP_ppc64_linux)
1903        arch->vex.guest_CIA -= 4;             // sizeof(ppc32 instr)
1904     
1905        /* Make sure our caller is actually sane, and we're really backing
1906           back over a syscall.
1907     
1908           sc == 44 00 00 02
1909        */
1910        {
1911           UChar *p = (UChar *)arch->vex.guest_CIA;
1912     
1913           if (p[0] != 0x44 || p[1] != 0x0 || p[2] != 0x0 || p[3] != 0x02)
1914              VG_(message)(Vg_DebugMsg,
1915                           "?! restarting over syscall at %#llx %02x %02x %02x %02x\n",
1916                           arch->vex.guest_CIA + 0ULL, p[0], p[1], p[2], p[3]);
1917     
1918           vg_assert(p[0] == 0x44 && p[1] == 0x0 && p[2] == 0x0 && p[3] == 0x2);
1919        }
1920     
1921     #elif defined(VGP_arm_linux)
1922        if (arch->vex.guest_R15T & 1) {
1923           // Thumb mode.  SVC is a encoded as
1924           //   1101 1111 imm8
1925           // where imm8 is the SVC number, and we only accept 0.
1926           arch->vex.guest_R15T -= 2;   // sizeof(thumb 16 bit insn)
1927           UChar* p     = (UChar*)(arch->vex.guest_R15T - 1);
1928           Bool   valid = p[0] == 0 && p[1] == 0xDF;
1929           if (!valid) {
1930              VG_(message)(Vg_DebugMsg,
1931                           "?! restarting over (Thumb) syscall that is not syscall "
1932                           "at %#llx %02x %02x\n",
1933                           arch->vex.guest_R15T - 1ULL, p[0], p[1]);
1934           }
1935           vg_assert(valid);
1936           // FIXME: NOTE, this really isn't right.  We need to back up
1937           // ITSTATE to what it was before the SVC instruction, but we
1938           // don't know what it was.  At least assert that it is now
1939           // zero, because if it is nonzero then it must also have
1940           // been nonzero for the SVC itself, which means it was
1941           // conditional.  Urk.
1942           vg_assert(arch->vex.guest_ITSTATE == 0);
1943        } else {
1944           // ARM mode.  SVC is encoded as 
1945           //   cond 1111 imm24
1946           // where imm24 is the SVC number, and we only accept 0.
1947           arch->vex.guest_R15T -= 4;   // sizeof(arm instr)
1948           UChar* p     = (UChar*)arch->vex.guest_R15T;
1949           Bool   valid = p[0] == 0 && p[1] == 0 && p[2] == 0
1950                          && (p[3] & 0xF) == 0xF;
1951           if (!valid) {
1952              VG_(message)(Vg_DebugMsg,
1953                           "?! restarting over (ARM) syscall that is not syscall "
1954                           "at %#llx %02x %02x %02x %02x\n",
1955                           arch->vex.guest_R15T + 0ULL, p[0], p[1], p[2], p[3]);
1956           }
1957           vg_assert(valid);
1958        }
1959     
1960     #elif defined(VGP_ppc32_aix5) || defined(VGP_ppc64_aix5)
1961        /* Hmm.  This is problematic, because on AIX the kernel resumes
1962           after a syscall at LR, not at the insn following SC.  Hence
1963           there is no obvious way to figure out where the SC is.  Current
1964           solution is to have a pseudo-register in the guest state,
1965           CIA_AT_SC, which holds the address of the most recent SC
1966           executed.  Backing up to that syscall then simply involves
1967           copying that value back into CIA (the program counter). */
1968        arch->vex.guest_CIA = arch->vex.guest_CIA_AT_SC;
1969     
1970        /* Make sure our caller is actually sane, and we're really backing
1971           back over a syscall.
1972     
1973           sc == 44 00 00 02
1974        */
1975        {
1976           UChar *p = (UChar *)arch->vex.guest_CIA;
1977     
1978           if (p[0] != 0x44 || p[1] != 0x0 || p[2] != 0x0 || p[3] != 0x02)
1979              VG_(message)(Vg_DebugMsg,
1980                           "?! restarting over syscall at %#lx %02x %02x %02x %02x\n",
1981                           (UWord)arch->vex.guest_CIA, p[0], p[1], p[2], p[3]);
1982     
1983           vg_assert(p[0] == 0x44 && p[1] == 0x0 && p[2] == 0x0 && p[3] == 0x2);
1984        }
1985     
1986     #elif defined(VGP_x86_darwin)
1987        arch->vex.guest_EIP = arch->vex.guest_IP_AT_SYSCALL; 
1988     
1989        /* Make sure our caller is actually sane, and we're really backing
1990           back over a syscall.
1991     
1992           int $0x80 == CD 80
1993           int $0x81 == CD 81
1994           int $0x82 == CD 82
1995           sysenter  == 0F 34
1996        */
1997        {
1998            UChar *p = (UChar *)arch->vex.guest_EIP;
1999            Bool  ok = (p[0] == 0xCD && p[1] == 0x80) 
2000                       || (p[0] == 0xCD && p[1] == 0x81)
2001                       || (p[0] == 0xCD && p[1] == 0x82)  
2002                       || (p[0] == 0x0F && p[1] == 0x34);
2003            if (!ok)
2004                VG_(message)(Vg_DebugMsg,
2005                             "?! restarting over syscall at %#x %02x %02x\n",
2006                             arch->vex.guest_EIP, p[0], p[1]);
2007            vg_assert(ok);
2008        }
2009        
2010     #elif defined(VGP_amd64_darwin)
2011        // DDD: #warning GrP fixme amd64 restart unimplemented
2012        vg_assert(0);
2013        
2014     #elif defined(VGP_s390x_linux)
2015        arch->vex.guest_IA -= 2;             // sizeof(syscall)
2016     
2017        /* Make sure our caller is actually sane, and we're really backing
2018           back over a syscall.
2019     
2020           syscall == 0A <num>
2021        */
2022        {
2023           UChar *p = (UChar *)arch->vex.guest_IA;
2024           if (p[0] != 0x0A)
2025              VG_(message)(Vg_DebugMsg,
2026                           "?! restarting over syscall at %#llx %02x %02x\n",
2027                           arch->vex.guest_IA, p[0], p[1]);
2028     
2029           vg_assert(p[0] == 0x0A);
2030        }
2031     #else
2032     #  error "ML_(fixup_guest_state_to_restart_syscall): unknown plat"
2033     #endif
2034     }

2036     /* 
2037        Fix up the guest state when a syscall is interrupted by a signal
2038        and so has been forced to return 'sysret'.
2039     
2040        To do this, we determine the precise state of the syscall by
2041        looking at the (real) IP at the time the signal happened.  The
2042        syscall sequence looks like:
2043     
2044          1. unblock signals
2045          2. perform syscall
2046          3. save result to guest state (EAX, RAX, R3+CR0.SO)
2047          4. re-block signals
2048     
2049        If a signal
2050        happens at      Then     Why?
2051        [1-2)           restart  nothing has happened (restart syscall)
2052        [2]             restart  syscall hasn't started, or kernel wants to restart
2053        [2-3)           save     syscall complete, but results not saved
2054        [3-4)           syscall complete, results saved
2055     
2056        Sometimes we never want to restart an interrupted syscall (because
2057        sigaction says not to), so we only restart if "restart" is True.
2058     
2059        This will also call VG_(post_syscall) if the syscall has actually
2060        completed (either because it was interrupted, or because it
2061        actually finished).  It will not call VG_(post_syscall) if the
2062        syscall is set up for restart, which means that the pre-wrapper may
2063        get called multiple times.
2064     */
2065     
2066     void 
2067     VG_(fixup_guest_state_after_syscall_interrupted)( ThreadId tid, 
2068                                                       Addr     ip, 
2069                                                       SysRes   sres,
2070                                                       Bool     restart)
2071     {
2072        /* Note that we don't know the syscall number here, since (1) in
2073           general there's no reliable way to get hold of it short of
2074           stashing it in the guest state before the syscall, and (2) in
2075           any case we don't need to know it for the actions done by this
2076           routine.
2077     
2078           Furthermore, 'sres' is only used in the case where the syscall
2079           is complete, but the result has not been committed to the guest
2080           state yet.  In any other situation it will be meaningless and
2081           therefore ignored. */
2082     
2083        ThreadState*     tst;
2084        SyscallStatus    canonical;
2085        ThreadArchState* th_regs;
2086        SyscallInfo*     sci;
2087     
2088        /* Compute some Booleans indicating which range we're in. */
2089        Bool outside_range, 
2090             in_setup_to_restart,      // [1,2) in the .S files
2091             at_restart,               // [2]   in the .S files
2092             in_complete_to_committed, // [3,4) in the .S files
2093             in_committed_to_finished; // [4,5) in the .S files
2094     
2095     #  if defined(VGO_linux) || defined(VGO_aix5)
2096        outside_range
2097           = ip < ML_(blksys_setup) || ip >= ML_(blksys_finished);
2098        in_setup_to_restart
2099           = ip >= ML_(blksys_setup) && ip < ML_(blksys_restart); 
2100        at_restart
2101           = ip == ML_(blksys_restart); 
2102        in_complete_to_committed
2103           = ip >= ML_(blksys_complete) && ip < ML_(blksys_committed); 
2104        in_committed_to_finished
2105           = ip >= ML_(blksys_committed) && ip < ML_(blksys_finished);
2106     #  elif defined(VGO_darwin)
2107        outside_range
2108           =  (ip < ML_(blksys_setup_MACH) || ip >= ML_(blksys_finished_MACH))
2109           && (ip < ML_(blksys_setup_MDEP) || ip >= ML_(blksys_finished_MDEP))
2110           && (ip < ML_(blksys_setup_UNIX) || ip >= ML_(blksys_finished_UNIX));
2111        in_setup_to_restart
2112           =  (ip >= ML_(blksys_setup_MACH) && ip < ML_(blksys_restart_MACH))
2113           || (ip >= ML_(blksys_setup_MDEP) && ip < ML_(blksys_restart_MDEP))
2114           || (ip >= ML_(blksys_setup_UNIX) && ip < ML_(blksys_restart_UNIX));
2115        at_restart
2116           =  (ip == ML_(blksys_restart_MACH))
2117           || (ip == ML_(blksys_restart_MDEP))
2118           || (ip == ML_(blksys_restart_UNIX));
2119        in_complete_to_committed
2120           =  (ip >= ML_(blksys_complete_MACH) && ip < ML_(blksys_committed_MACH))
2121           || (ip >= ML_(blksys_complete_MDEP) && ip < ML_(blksys_committed_MDEP))
2122           || (ip >= ML_(blksys_complete_UNIX) && ip < ML_(blksys_committed_UNIX));
2123        in_committed_to_finished
2124           =  (ip >= ML_(blksys_committed_MACH) && ip < ML_(blksys_finished_MACH))
2125           || (ip >= ML_(blksys_committed_MDEP) && ip < ML_(blksys_finished_MDEP))
2126           || (ip >= ML_(blksys_committed_UNIX) && ip < ML_(blksys_finished_UNIX));
2127        /* Wasn't that just So Much Fun?  Does your head hurt yet?  Mine does. */
2128     #  else
2129     #    error "Unknown OS"
2130     #  endif
2131     
2132        if (VG_(clo_trace_signals))
2133           VG_(message)( Vg_DebugMsg,
2134                         "interrupted_syscall: tid=%d, ip=0x%llx, "
2135                         "restart=%s, sres.isErr=%s, sres.val=%lld\n", 
2136                         (Int)tid,
2137                         (ULong)ip, 
2138                         restart ? "True" : "False", 
2139                         sr_isError(sres) ? "True" : "False",
2140                         (Long)(sr_isError(sres) ? sr_Err(sres) : sr_Res(sres)) );
2141     
2142        vg_assert(VG_(is_valid_tid)(tid));
2143        vg_assert(tid >= 1 && tid < VG_N_THREADS);
2144        vg_assert(VG_(is_running_thread)(tid));
2145     
2146        tst     = VG_(get_ThreadState)(tid);
2147        th_regs = &tst->arch;
2148        sci     = & syscallInfo[tid];
2149     
2150        /* Figure out what the state of the syscall was by examining the
2151           (real) IP at the time of the signal, and act accordingly. */
2152        if (outside_range) {
2153           if (VG_(clo_trace_signals))
2154              VG_(message)( Vg_DebugMsg,
2155                            "  not in syscall at all: hmm, very suspicious\n" );
2156           /* Looks like we weren't in a syscall at all.  Hmm. */
2157           vg_assert(sci->status.what != SsIdle);
2158           return;
2159        }
2160     
2161        /* We should not be here unless this thread had first started up
2162           the machinery for a syscall by calling VG_(client_syscall).
2163           Hence: */
2164        vg_assert(sci->status.what != SsIdle);
2165     
2166        /* now, do one of four fixup actions, depending on where the IP has
2167           got to. */
2168     
2169        if (in_setup_to_restart) {
2170           /* syscall hasn't even started; go around again */
2171           if (VG_(clo_trace_signals))
2172              VG_(message)( Vg_DebugMsg, "  not started: restarting\n");
2173           vg_assert(sci->status.what == SsHandToKernel);
2174           ML_(fixup_guest_state_to_restart_syscall)(th_regs);
2175        } 
2176     
2177        else 
2178        if (at_restart) {
2179           /* We're either about to run the syscall, or it was interrupted
2180              and the kernel restarted it.  Restart if asked, otherwise
2181              EINTR it. */
2182           if (restart) {
2183              if (VG_(clo_trace_signals))
2184                 VG_(message)( Vg_DebugMsg, "  at syscall instr: restarting\n");
2185              ML_(fixup_guest_state_to_restart_syscall)(th_regs);
2186           } else {
2187              if (VG_(clo_trace_signals))
2188                 VG_(message)( Vg_DebugMsg, "  at syscall instr: returning EINTR\n");
2189              canonical = convert_SysRes_to_SyscallStatus( 
2190                             VG_(mk_SysRes_Error)( VKI_EINTR ) 
2191                          );
2192              if (!(sci->flags & SfNoWriteResult))
2193                 putSyscallStatusIntoGuestState( tid, &canonical, &th_regs->vex );
2194              sci->status = canonical;
2195              VG_(post_syscall)(tid);
2196           }
2197        }
2198     
2199        else 
2200        if (in_complete_to_committed) {
2201           /* Syscall complete, but result hasn't been written back yet.
2202              Write the SysRes we were supplied with back to the guest
2203              state. */
2204           if (VG_(clo_trace_signals))
2205              VG_(message)( Vg_DebugMsg,
2206                            "  completed, but uncommitted: committing\n");
2207           canonical = convert_SysRes_to_SyscallStatus( sres );
2208           if (!(sci->flags & SfNoWriteResult))
2209              putSyscallStatusIntoGuestState( tid, &canonical, &th_regs->vex );
2210           sci->status = canonical;
2211           VG_(post_syscall)(tid);
2212        } 
2213     
2214        else  
2215        if (in_committed_to_finished) {
2216           /* Result committed, but the signal mask has not been restored;
2217              we expect our caller (the signal handler) will have fixed
2218              this up. */
2219           if (VG_(clo_trace_signals))
2220              VG_(message)( Vg_DebugMsg,
2221                            "  completed and committed: nothing to do\n");
2222           getSyscallStatusFromGuestState( &sci->status, &th_regs->vex );
2223           vg_assert(sci->status.what == SsComplete);
2224           VG_(post_syscall)(tid);
2225        } 
2226     
2227        else
2228           VG_(core_panic)("?? strange syscall interrupt state?");
2229     
2230        /* In all cases, the syscall is now finished (even if we called
2231           ML_(fixup_guest_state_to_restart_syscall), since that just
2232           re-positions the guest's IP for another go at it).  So we need
2233           to record that fact. */
2234        sci->status.what = SsIdle;
2235     }
2236     
2237     
2238     #if defined(VGO_darwin)
2239     // Clean up after workq_ops(WQOPS_THREAD_RETURN) jumped to wqthread_hijack. 
2240     // This is similar to VG_(fixup_guest_state_after_syscall_interrupted).
2241     // This longjmps back to the scheduler.
2242     void ML_(wqthread_continue_NORETURN)(ThreadId tid)
2243     {
2244        ThreadState*     tst;
2245        SyscallInfo*     sci;
2246     
2247        VG_(acquire_BigLock)(tid, "wqthread_continue_NORETURN");
2248     
2249        PRINT("SYSCALL[%d,%d](%s) workq_ops() starting new workqueue item\n", 
2250              VG_(getpid)(), tid, VG_SYSNUM_STRING(__NR_workq_ops));
2251     
2252        vg_assert(VG_(is_valid_tid)(tid));
2253        vg_assert(tid >= 1 && tid < VG_N_THREADS);
2254        vg_assert(VG_(is_running_thread)(tid));
2255     
2256        tst     = VG_(get_ThreadState)(tid);
2257        sci     = & syscallInfo[tid];
2258        vg_assert(sci->status.what != SsIdle);
2259        vg_assert(tst->os_state.wq_jmpbuf_valid);  // check this BEFORE post_syscall
2260     
2261        // Pretend the syscall completed normally, but don't touch the thread state.
2262        sci->status = convert_SysRes_to_SyscallStatus( VG_(mk_SysRes_Success)(0) );
2263        sci->flags |= SfNoWriteResult;
2264        VG_(post_syscall)(tid);
2265     
2266        sci->status.what = SsIdle;
2267     
2268        vg_assert(tst->sched_jmpbuf_valid);
2269        __builtin_longjmp(tst->sched_jmpbuf, True);
2270     
2271        /* NOTREACHED */
2272        vg_assert(0);
2273     }
2274     #endif
