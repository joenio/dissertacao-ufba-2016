31     #ifndef __PUB_TOOL_REPLACEMALLOC_H
32     #define __PUB_TOOL_REPLACEMALLOC_H
33     
34     /* If a tool replaces malloc() et al, the easiest way to do so is to
35        link libreplacemalloc_toolpreload.o into its vgpreload_*.so file, and
36        use the functions declared below.  You can do it from scratch,
37        though, if you enjoy that sort of thing. */
38     
39     /* Can be called from VG_(tdict).malloc_malloc et al to do the actual
40      * alloc/freeing. */
41     extern void* VG_(cli_malloc) ( SizeT align, SizeT nbytes );
42     extern void  VG_(cli_free)   ( void* p );
43     
44     /* If a tool uses deferred freeing (e.g. memcheck to catch accesses to
45        freed memory) it can maintain number and total size of queued blocks
46        in these variable to provide more accurate statistics about client
47        memory usage. Currently used by mallinfo(). */
48     extern Long VG_(free_queue_volume);
49     extern Long VG_(free_queue_length);
50     
51     /* Check if an address is within a range, allowing for redzones at edges */
52     extern Bool VG_(addr_is_in_block)( Addr a, Addr start,
53                                        SizeT size, SizeT rz_szB );
54     
55     /* ------------------------------------------------------------------ */
56     /* Some options that can be used by a tool if malloc() et al are replaced.
57        The tool should call the functions in the appropriate places to give
58        control over these aspects of Valgrind's version of malloc(). */
59     
60     /* DEBUG: print malloc details?  default: NO */
61     extern Bool VG_(clo_trace_malloc);
62     /* Minimum alignment in functions that don't specify alignment explicitly.
63        default: VG_MIN_MALLOC_SZB */
64     extern UInt VG_(clo_alignment);
65     
66     extern Bool VG_(replacement_malloc_process_cmd_line_option) ( Char* arg );
67     
68     #endif   // __PUB_TOOL_REPLACEMALLOC_H
