13     UInt do_s_crc32b ( UInt crcIn, UChar b )
14     {
15        UInt i, crc = (b & 0xFF) ^ crcIn;
16        for (i = 0; i < 8; i++)
17           crc = (crc >> 1) ^ ((crc & 1) ? 0x82f63b78 : 0);
18        return crc;
19     }

145       UInt al, i;
