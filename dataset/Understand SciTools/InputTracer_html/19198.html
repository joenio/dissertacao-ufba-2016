31     #ifndef __PUB_CORE_STACKTRACE_H
32     #define __PUB_CORE_STACKTRACE_H
33     
34     //--------------------------------------------------------------------
35     // PURPOSE: This module deals with stack traces:  getting them, 
36     // traversing them, and printing them.
37     //--------------------------------------------------------------------
38     
39     #include "pub_tool_stacktrace.h"
40     
41     // Variant that gives a little more control over the stack-walking
42     // (this is the "worker" function that actually does the walking).
43     // If you know what the thread ID for this stack is, send that
44     // as the first parameter, else send zero.  This helps generate
45     // better stack traces on ppc64-linux and has no effect on other
46     // platforms.
47     //
48     // The acquired IP values are placed in
49     // ips[0 .. min(n_ips,return_value)].  If sps and fps are non-NULL,
50     // the corresponding frame-pointer and stack-pointer values for each
51     // frame are stored there.
52     
53     UInt VG_(get_StackTrace_wrk) ( ThreadId tid_if_known,
54                                    /*OUT*/Addr* ips, UInt n_ips,
55                                    /*OUT*/Addr* sps, /*OUT*/Addr* fps,
56                                    UnwindStartRegs* startRegs,
57                                    Addr fp_max_orig );
58     
59     #endif   // __PUB_CORE_STACKTRACE_H
