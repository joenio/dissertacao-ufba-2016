64     /*------------------------------------------------------------*/
65     /*--- Filter management                                    ---*/
66     /*------------------------------------------------------------*/ 
67     
68     /* Simple linked list for storing ranges. */
69     typedef struct IT_RListElem IT_RListElem;
70     struct IT_RListElem 
71     {
72        ULong          min;
73        ULong          max;
74        IT_RListElem*  next;
75     };
76     
77     /* Special "file name" of standard input */
78     #define STDIN_IDENTIFIER "STDIN"
79     
80     /* Return True if file is in filter array */
81     Bool IT_(is_in_input_filter_list)(UWord filedes);
82     
83     /* Returns an input filter handle for the given file descriptor,
84        or -1 if no matching file name is found OR if no filter
85        exists for the given file.
86        TODO: This is ugly; should return handle even if no ranges are defined,
87        change it! */
88     Word IT_(get_input_filter_handle)(UWord filedes);
89     
90     /* Returns sink filter handle for the file path,
91        or -1 if no matching file name is found. DOES return a handle if no ranges
92        are defined. */
93     Word IT_(get_sink_filter_handle_from_path)(const Char* path);
94     
95     /* Applies byte filter and returns a sorted list of all subranges between
96        'min' and 'max' that match the filter. If no matches are found, NULL is
97        returned. The list is allocated using VG_(malloc) and may be freed later. */
98     IT_RListElem* IT_(get_filtered_ranges_input)(Word handle, ULong min, ULong max);
99     
100     /* Returns True if 'path' matches any of the registered file exclusion
101        patterns */
102     Bool IT_(is_excluded_file)(const Char* path);
103     
104     /* Returns True if 'path' matches any of the registered propagation image
105        exclusion patterns */
106     Bool IT_(is_excluded_image_prop)(const Char* path);
107     
108     /* Returns True if 'path' matches any of the registered propagation image
109        inclusion patterns */
110     Bool IT_(is_included_image_prop)(const Char* path);
111     
112     /* Convenience function for checking if an instruction is inside an executable
113        excluded from taint propagation. */
114     Bool IT_(is_in_excluded_range_prop)(Addr a);
115     
116     /* Returns True if 'a' is in an executable matched by a sink exclusion
117        pattern */
118     Bool IT_(is_in_sink_exclude_pattern)(const Char* path, Addr a);
119     
120     /* Returns True if 'a' is inside an included range in the filter for the file
121        pointed to by 'handle' */
122     Bool IT_(is_included_in_sink_filter)(Word handle, Addr a);
