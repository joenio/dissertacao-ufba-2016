144     /* Figure out if any part of the guest state contained in minoff
145        .. maxoff requires precise memory exceptions.  If in doubt return
146        True (but this is generates significantly slower code).  */
147     Bool
148     guest_s390x_state_requires_precise_mem_exns(Int minoff, Int maxoff)
149     {
150        Int lr_min = offsetof(VexGuestS390XState, guest_LR);
151        Int lr_max = lr_min + 8 - 1;
152        Int sp_min = offsetof(VexGuestS390XState, guest_SP);
153        Int sp_max = sp_min + 8 - 1;
154        Int fp_min = offsetof(VexGuestS390XState, guest_FP);
155        Int fp_max = fp_min + 8 - 1;
156        Int ia_min = offsetof(VexGuestS390XState, guest_IA);
157        Int ia_max = ia_min + 8 - 1;
158     
159        if (maxoff < lr_min || minoff > lr_max) {
160           /* No overlap with LR */
161        } else {
162           return True;
163        }
164     
165        if (maxoff < sp_min || minoff > sp_max) {
166           /* No overlap with SP */
167        } else {
168           return True;
169        }
170     
171        if (maxoff < fp_min || minoff > fp_max) {
172           /* No overlap with FP */
173        } else {
174           return True;
175        }
176     
177        if (maxoff < ia_min || minoff > ia_max) {
178           /* No overlap with IA */
179        } else {
180           return True;
181        }
182     
183        return False;
184     }

241     #define S390_CC_FOR_BINARY(opcode,cc_dep1,cc_dep2) \
242     ({ \
243        __asm__ volatile ( \
244             opcode " %[op1],%[op2]\n\t" \
245             "ipm %[psw]\n\t"           : [psw] "=d"(psw), [op1] "+d"(cc_dep1) \
246                                        : [op2] "d"(cc_dep2) \
247                                        : "cc");\
248        psw >> 28;   /* cc */ \
249     })
250     
251     #define S390_CC_FOR_TERNARY_SUBB(opcode,cc_dep1,cc_dep2,cc_ndep) \
252     ({ \
253        /* Recover the original DEP2 value. See comment near s390_cc_thunk_put3 \
254           for rationale. */ \
255        cc_dep2 = cc_dep2 ^ cc_ndep; \
256        __asm__ volatile ( \
257     	"lghi 0,1\n\t" \
258     	"sr 0,%[op3]\n\t" /* borrow to cc */ \
259             opcode " %[op1],%[op2]\n\t" /* then redo the op */\
260             "ipm %[psw]\n\t"           : [psw] "=d"(psw), [op1] "+&d"(cc_dep1) \
261                                        : [op2] "d"(cc_dep2), [op3] "d"(cc_ndep) \
262                                        : "0", "cc");\
263        psw >> 28;   /* cc */ \
264     })
265     
266     #define S390_CC_FOR_TERNARY_ADDC(opcode,cc_dep1,cc_dep2,cc_ndep) \
267     ({ \
268        /* Recover the original DEP2 value. See comment near s390_cc_thunk_put3 \
269           for rationale. */ \
270        cc_dep2 = cc_dep2 ^ cc_ndep; \
271        __asm__ volatile ( \
272     	"lgfr 0,%[op3]\n\t" /* first load cc_ndep */ \
273     	"aghi 0,0\n\t" /* and convert it into a cc */ \
274             opcode " %[op1],%[op2]\n\t" /* then redo the op */\
275             "ipm %[psw]\n\t"           : [psw] "=d"(psw), [op1] "+&d"(cc_dep1) \
276                                        : [op2] "d"(cc_dep2), [op3] "d"(cc_ndep) \
277                                        : "0", "cc");\
278        psw >> 28;   /* cc */ \
279     })
280     
281     
282     #define S390_CC_FOR_BFP_RESULT(opcode,cc_dep1) \
283     ({ \
284        __asm__ volatile ( \
285             opcode " 0,%[op]\n\t" \
286             "ipm %[psw]\n\t"           : [psw] "=d"(psw) \
287                                        : [op]  "f"(cc_dep1) \
288                                        : "cc", "f0");\
289        psw >> 28;   /* cc */ \
290     })
291     
292     #define S390_CC_FOR_BFP128_RESULT(hi,lo) \
293     ({ \
294        __asm__ volatile ( \
295             "ldr   4,%[high]\n\t" \
296             "ldr   6,%[low]\n\t" \
297             "ltxbr 0,4\n\t" \
298             "ipm %[psw]\n\t"           : [psw] "=d"(psw) \
299                                        : [high] "f"(hi), [low] "f"(lo) \
300                                        : "cc", "f0", "f2", "f4", "f6");\
301        psw >> 28;   /* cc */ \
302     })
303     
304     #define S390_CC_FOR_BFP_CONVERT(opcode,cc_dep1) \
305     ({ \
306        __asm__ volatile ( \
307             opcode " 0,0,%[op]\n\t" \
308             "ipm %[psw]\n\t"           : [psw] "=d"(psw) \
309                                        : [op]  "f"(cc_dep1) \
310                                        : "cc", "r0");\
311        psw >> 28;   /* cc */ \
312     })
313     
314     #define S390_CC_FOR_BFP128_CONVERT(opcode,hi,lo) \
315     ({ \
316        __asm__ volatile ( \
317             "ldr   4,%[high]\n\t" \
318             "ldr   6,%[low]\n\t" \
319             opcode " 0,0,4\n\t" \
320             "ipm %[psw]\n\t"           : [psw] "=d"(psw) \
321                                        : [high] "f"(hi), [low] "f"(lo) \
322                                        : "cc", "r0", "f4", "f6");\
323        psw >> 28;   /* cc */ \
324     })
325     
326     #define S390_CC_FOR_BFP_TDC(opcode,cc_dep1,cc_dep2) \
327     ({ \
328        __asm__ volatile ( \
329             opcode " %[value],0(%[class])\n\t" \
330             "ipm %[psw]\n\t"           : [psw] "=d"(psw) \
331                                        : [value] "f"(cc_dep1), \
332                                          [class] "a"(cc_dep2)  \
333                                        : "cc");\
334        psw >> 28;   /* cc */ \
335     })
336     
337     #define S390_CC_FOR_BFP128_TDC(cc_dep1,cc_dep2,cc_ndep) \
338     ({ \
339        /* Recover the original DEP2 value. See comment near s390_cc_thunk_put1f128Z \
340           for rationale. */ \
341        cc_dep2 = cc_dep2 ^ cc_ndep; \
342        __asm__ volatile ( \
343             "ldr  4,%[high]\n\t" \
344             "ldr  6,%[low]\n\t" \
345             "tcxb 4,0(%[class])\n\t" \
346             "ipm  %[psw]\n\t"          : [psw] "=d"(psw) \
347                                        : [high] "f"(cc_dep1), [low] "f"(cc_dep2), \
348                                          [class] "a"(cc_ndep)  \
349                                        : "cc", "f4", "f6");\
350        psw >> 28;   /* cc */ \
351     })

354     /* Return the value of the condition code from the supplied thunk parameters.
355        This is not the value of the PSW. It is the value of the 2 CC bits within
356        the PSW. The returned value is thusly in the interval [0:3]. */
357     UInt
358     s390_calculate_cc(ULong cc_op, ULong cc_dep1, ULong cc_dep2, ULong cc_ndep)
359     {
360     #if defined(VGA_s390x)
361        UInt psw;
362     
363        switch (cc_op) {
364     
365        case S390_CC_OP_BITWISE:
366           return S390_CC_FOR_BINARY("ogr", cc_dep1, (ULong)0);
367     
368        case S390_CC_OP_SIGNED_COMPARE:
369           return S390_CC_FOR_BINARY("cgr", cc_dep1, cc_dep2);
370     
371        case S390_CC_OP_UNSIGNED_COMPARE:
372           return S390_CC_FOR_BINARY("clgr", cc_dep1, cc_dep2);
373     
374        case S390_CC_OP_SIGNED_ADD_64:
375           return S390_CC_FOR_BINARY("agr", cc_dep1, cc_dep2);
376     
377        case S390_CC_OP_SIGNED_ADD_32:
378           return S390_CC_FOR_BINARY("ar", cc_dep1, cc_dep2);
379     
380        case S390_CC_OP_SIGNED_SUB_64:
381           return S390_CC_FOR_BINARY("sgr", cc_dep1, cc_dep2);
382     
383        case S390_CC_OP_SIGNED_SUB_32:
384           return S390_CC_FOR_BINARY("sr", cc_dep1, cc_dep2);
385     
386        case S390_CC_OP_UNSIGNED_ADD_64:
387           return S390_CC_FOR_BINARY("algr", cc_dep1, cc_dep2);
388     
389        case S390_CC_OP_UNSIGNED_ADD_32:
390           return S390_CC_FOR_BINARY("alr", cc_dep1, cc_dep2);
391     
392        case S390_CC_OP_UNSIGNED_ADDC_64:
393           return S390_CC_FOR_TERNARY_ADDC("alcgr", cc_dep1, cc_dep2, cc_ndep);
394     
395        case S390_CC_OP_UNSIGNED_ADDC_32:
396           return S390_CC_FOR_TERNARY_ADDC("alcr", cc_dep1, cc_dep2, cc_ndep);
397     
398        case S390_CC_OP_UNSIGNED_SUB_64:
399           return S390_CC_FOR_BINARY("slgr", cc_dep1, cc_dep2);
400     
401        case S390_CC_OP_UNSIGNED_SUB_32:
402           return S390_CC_FOR_BINARY("slr", cc_dep1, cc_dep2);
403     
404        case S390_CC_OP_UNSIGNED_SUBB_64:
405           return S390_CC_FOR_TERNARY_SUBB("slbgr", cc_dep1, cc_dep2, cc_ndep);
406     
407        case S390_CC_OP_UNSIGNED_SUBB_32:
408           return S390_CC_FOR_TERNARY_SUBB("slbr", cc_dep1, cc_dep2, cc_ndep);
409     
410        case S390_CC_OP_LOAD_AND_TEST:
411           /* Like signed comparison with 0 */
412           return S390_CC_FOR_BINARY("cgr", cc_dep1, (Long)0);
413     
414        case S390_CC_OP_TEST_AND_SET:
415           /* Shift the sign bit into the LSB. Note, that the tested value is an
416              8-bit value which has been zero-extended to 32/64 bit. */
417           return cc_dep1 >> 7;
418     
419        case S390_CC_OP_LOAD_POSITIVE_32:
420           __asm__ volatile (
421                "lpr  %[result],%[op]\n\t"
422                "ipm  %[psw]\n\t"            : [psw] "=d"(psw), [result] "=d"(cc_dep1)
423                                             : [op] "d"(cc_dep1)
424                                             : "cc");
425           return psw >> 28;   /* cc */
426     
427        case S390_CC_OP_LOAD_POSITIVE_64:
428           __asm__ volatile (
429                "lpgr %[result],%[op]\n\t"
430                "ipm  %[psw]\n\t"            : [psw] "=d"(psw), [result] "=d"(cc_dep1)
431                                             : [op] "d"(cc_dep1)
432                                             : "cc");
433           return psw >> 28;   /* cc */
434     
435        case S390_CC_OP_TEST_UNDER_MASK_8: {
436           UChar value  = cc_dep1;
437           UChar mask   = cc_dep2;
438     
439           __asm__ volatile (
440                "bras %%r2,1f\n\t"             /* %r2 = address of next insn */
441                "tm %[value],0\n\t"            /* this is skipped, then EXecuted */
442                "1: ex %[mask],0(%%r2)\n\t"    /* EXecute TM after modifying mask */
443                "ipm %[psw]\n\t"             : [psw] "=d"(psw)
444                                             : [value] "m"(value), [mask] "a"(mask)
445                                             : "r2", "cc");
446           return psw >> 28;   /* cc */
447        }
448     
449        case S390_CC_OP_TEST_UNDER_MASK_16: {
450           /* Create a TMLL insn with the mask as given by cc_dep2 */
451           UInt insn  = (0xA701 << 16) | cc_dep2;
452           UInt value = cc_dep1;
453     
454           __asm__ volatile (
455                "lr   1,%[value]\n\t"
456                "lhi  2,0x10\n\t"
457                "ex   2,%[insn]\n\t"
458                "ipm  %[psw]\n\t"       : [psw] "=d"(psw)
459                                        : [value] "d"(value), [insn] "m"(insn)
460                                        : "r1", "r2", "cc");
461           return psw >> 28;   /* cc */
462        }
463     
464        case S390_CC_OP_SHIFT_LEFT_32:
465           __asm__ volatile (
466                "sla  %[op],0(%[amount])\n\t"
467                "ipm  %[psw]\n\t"            : [psw] "=d"(psw), [op] "+d"(cc_dep1)
468                                             : [amount] "a"(cc_dep2)
469                                             : "cc");
470           return psw >> 28;   /* cc */
471     
472        case S390_CC_OP_SHIFT_LEFT_64: {
473           Int high = (Int)(cc_dep1 >> 32);
474           Int low  = (Int)(cc_dep1 & 0xFFFFFFFF);
475     
476           __asm__ volatile (
477                "lr   2,%[high]\n\t"
478                "lr   3,%[low]\n\t"
479                "slda 2,0(%[amount])\n\t"
480                "ipm %[psw]\n\t"             : [psw] "=d"(psw), [high] "+d"(high), [low] "+d"(low)
481                                             : [amount] "a"(cc_dep2)
482                                             : "cc", "r2", "r3");
483           return psw >> 28;   /* cc */
484        }
485     
486        case S390_CC_OP_INSERT_CHAR_MASK_32: {
487           Int inserted = 0;
488           Int msb = 0;
489     
490           if (cc_dep2 & 1) {
491              inserted |= cc_dep1 & 0xff;
492              msb = 0x80;
493           }
494           if (cc_dep2 & 2) {
495              inserted |= cc_dep1 & 0xff00;
496              msb = 0x8000;
497           }
498           if (cc_dep2 & 4) {
499              inserted |= cc_dep1 & 0xff0000;
500              msb = 0x800000;
501           }
502           if (cc_dep2 & 8) {
503              inserted |= cc_dep1 & 0xff000000;
504              msb = 0x80000000;
505           }
506     
507           if (inserted & msb)  // MSB is 1
508              return 1;
509           if (inserted > 0)
510              return 2;
511           return 0;
512        }
513     
514        case S390_CC_OP_BFP_RESULT_32:
515           return S390_CC_FOR_BFP_RESULT("ltebr", cc_dep1);
516     
517        case S390_CC_OP_BFP_RESULT_64:
518           return S390_CC_FOR_BFP_RESULT("ltdbr", cc_dep1);
519     
520        case S390_CC_OP_BFP_RESULT_128:
521           return S390_CC_FOR_BFP128_RESULT(cc_dep1, cc_dep2);
522     
523        case S390_CC_OP_BFP_32_TO_INT_32:
524           return S390_CC_FOR_BFP_CONVERT("cfebr", cc_dep1);
525     
526        case S390_CC_OP_BFP_64_TO_INT_32:
527           return S390_CC_FOR_BFP_CONVERT("cfdbr", cc_dep1);
528     
529        case S390_CC_OP_BFP_128_TO_INT_32:
530           return S390_CC_FOR_BFP128_CONVERT("cfxbr", cc_dep1, cc_dep2);
531     
532        case S390_CC_OP_BFP_32_TO_INT_64:
533           return S390_CC_FOR_BFP_CONVERT("cgebr", cc_dep1);
534     
535        case S390_CC_OP_BFP_64_TO_INT_64:
536           return S390_CC_FOR_BFP_CONVERT("cgdbr", cc_dep1);
537     
538        case S390_CC_OP_BFP_128_TO_INT_64:
539           return S390_CC_FOR_BFP128_CONVERT("cgxbr", cc_dep1, cc_dep2);
540     
541        case S390_CC_OP_BFP_TDC_32:
542           return S390_CC_FOR_BFP_TDC("tceb", cc_dep1, cc_dep2);
543     
544        case S390_CC_OP_BFP_TDC_64:
545           return S390_CC_FOR_BFP_TDC("tcdb", cc_dep1, cc_dep2);
546     
547        case S390_CC_OP_BFP_TDC_128:
548           return S390_CC_FOR_BFP128_TDC(cc_dep1, cc_dep2, cc_ndep);
549     
550        case S390_CC_OP_SET:
551           return cc_dep1;
552     
553        default:
554           break;
555        }
556     #endif
557        vpanic("s390_calculate_cc");
558     }

568     /* Note that this does *not* return a Boolean value. The result needs to be
569        explicitly tested against zero. */
570     UInt
571     s390_calculate_cond(ULong mask, ULong op, ULong dep1, ULong dep2, ULong ndep)
572     {
573        UInt cc = s390_calculate_cc(op, dep1, dep2, ndep);
574     
575        return ((mask << cc) & 0x8);
576     }

579     /*--- spechelper for performance                           ---*/
580     /*------------------------------------------------------------*/
581     
582     
583     /* Convenience macros */
584     #define unop(op,a1) IRExpr_Unop((op),(a1))
585     #define binop(op,a1,a2) IRExpr_Binop((op),(a1),(a2))
586     #define mkU64(v) IRExpr_Const(IRConst_U64(v))
587     #define mkU32(v) IRExpr_Const(IRConst_U32(v))
588     #define mkU8(v)  IRExpr_Const(IRConst_U8(v))
589     
590     
591     static inline Bool
592     isC64(IRExpr *expr)
593     {
594        return expr->tag == Iex_Const && expr->Iex.Const.con->tag == Ico_U64;
595     }
596     
597     
598     /* The returned expression is NULL if no specialization was found. In that
599        case the helper function will be called. Otherwise, the expression has
600        type Ity_I32 and a Boolean value. */
601     IRExpr *
602     guest_s390x_spechelper(HChar *function_name, IRExpr **args,
603                            IRStmt **precedingStmts, Int n_precedingStmts)
604     {
605        UInt i, arity = 0;
606     
607        for (i = 0; args[i]; i++)
608           arity++;
609     
610     #  if 0
611        vex_printf("spec request:\n");
612        vex_printf("   %s  ", function_name);
613        for (i = 0; i < arity; i++) {
614           vex_printf("  ");
615           ppIRExpr(args[i]);
616        }
617        vex_printf("\n");
618     #  endif
619     
620        /* --------- Specialising "s390_calculate_cond" --------- */
621     
622        if (vex_streq(function_name, "s390_calculate_cond")) {
623           IRExpr *cond_expr, *cc_op_expr, *cc_dep1, *cc_dep2;
624           ULong cond, cc_op;
625     
626           vassert(arity == 5);
627     
628           cond_expr  = args[0];
629           cc_op_expr = args[1];
630     
631           /* The necessary requirement for all optimizations here is that the
632              condition and the cc_op are constant. So check that upfront. */
633           if (! isC64(cond_expr))  return NULL;
634           if (! isC64(cc_op_expr)) return NULL;
635     
636           cond    = cond_expr->Iex.Const.con->Ico.U64;
637           cc_op   = cc_op_expr->Iex.Const.con->Ico.U64;
638     
639           vassert(cond <= 15);
640     
641           /*
642             +------+---+---+---+---+
643             | cc   | 0 | 1 | 2 | 3 |
644             | cond | 8 | 4 | 2 | 1 |
645             +------+---+---+---+---+
646           */
647           cc_dep1 = args[2];
648           cc_dep2 = args[3];
649     
650           /* S390_CC_OP_SIGNED_COMPARE */
651           if (cc_op == S390_CC_OP_SIGNED_COMPARE) {
652              /*
653                 cc == 0  --> cc_dep1 == cc_dep2   (cond == 8)
654                 cc == 1  --> cc_dep1 <  cc_dep2   (cond == 4)
655                 cc == 2  --> cc_dep1 >  cc_dep2   (cond == 2)
656     
657                 Because cc == 3 cannot occur the rightmost bit of cond is
658                 a don't care.
659              */
660              if (cond == 8 || cond == 8 + 1) {
661                 return unop(Iop_1Uto32, binop(Iop_CmpEQ64, cc_dep1, cc_dep2));
662              }
663              if (cond == 4 + 2 || cond == 4 + 2 + 1) {
664                 return unop(Iop_1Uto32, binop(Iop_CmpNE64, cc_dep1, cc_dep2));
665              }
666              if (cond == 4 || cond == 4 + 1) {
667                 return unop(Iop_1Uto32, binop(Iop_CmpLT64S, cc_dep1, cc_dep2));
668              }
669              if (cond == 8 + 4 || cond == 8 + 4 + 1) {
670                 return unop(Iop_1Uto32, binop(Iop_CmpLE64S, cc_dep1, cc_dep2));
671              }
672              /* cc_dep1 > cc_dep2  ---->  cc_dep2 < cc_dep1 */
673              if (cond == 2 || cond == 2 + 1) {
674                 return unop(Iop_1Uto32, binop(Iop_CmpLT64S, cc_dep2, cc_dep1));
675              }
676              if (cond == 8 + 2 || cond == 8 + 2 + 1) {
677                 return unop(Iop_1Uto32, binop(Iop_CmpLE64S, cc_dep2, cc_dep1));
678              }
679              if (cond == 8 + 4 + 2 || cond == 8 + 4 + 2 + 1) {
680                 return mkU32(1);
681              }
682              /* Remaining case */
683              return mkU32(0);
684           }
685     
686           /* S390_CC_OP_UNSIGNED_COMPARE */
687           if (cc_op == S390_CC_OP_UNSIGNED_COMPARE) {
688              /*
689                 cc == 0  --> cc_dep1 == cc_dep2   (cond == 8)
690                 cc == 1  --> cc_dep1 <  cc_dep2   (cond == 4)
691                 cc == 2  --> cc_dep1 >  cc_dep2   (cond == 2)
692     
693                 Because cc == 3 cannot occur the rightmost bit of cond is
694                 a don't care.
695              */
696              if (cond == 8 || cond == 8 + 1) {
697                 return unop(Iop_1Uto32, binop(Iop_CmpEQ64, cc_dep1, cc_dep2));
698              }
699              if (cond == 4 + 2 || cond == 4 + 2 + 1) {
700                 return unop(Iop_1Uto32, binop(Iop_CmpNE64, cc_dep1, cc_dep2));
701              }
702              if (cond == 4 || cond == 4 + 1) {
703                 return unop(Iop_1Uto32, binop(Iop_CmpLT64U, cc_dep1, cc_dep2));
704              }
705              if (cond == 8 + 4 || cond == 8 + 4 + 1) {
706                 return unop(Iop_1Uto32, binop(Iop_CmpLE64U, cc_dep1, cc_dep2));
707              }
708              /* cc_dep1 > cc_dep2  ---->  cc_dep2 < cc_dep1 */
709              if (cond == 2 || cond == 2 + 1) {
710                 return unop(Iop_1Uto32, binop(Iop_CmpLT64U, cc_dep2, cc_dep1));
711              }
712              if (cond == 8 + 2 || cond == 8 + 2 + 1) {
713                 return unop(Iop_1Uto32, binop(Iop_CmpLE64U, cc_dep2, cc_dep1));
714              }
715              if (cond == 8 + 4 + 2 || cond == 8 + 4 + 2 + 1) {
716                 return mkU32(1);
717              }
718              /* Remaining case */
719              return mkU32(0);
720           }
721     
722           /* S390_CC_OP_LOAD_AND_TEST */
723           if (cc_op == S390_CC_OP_LOAD_AND_TEST) {
724              /*
725                 cc == 0  --> cc_dep1 == 0   (cond == 8)
726                 cc == 1  --> cc_dep1 <  0   (cond == 4)
727                 cc == 2  --> cc_dep1 >  0   (cond == 2)
728     
729                 Because cc == 3 cannot occur the rightmost bit of cond is
730                 a don't care.
731              */
732              if (cond == 8 || cond == 8 + 1) {
733                 return unop(Iop_1Uto32, binop(Iop_CmpEQ64, cc_dep1, mkU64(0)));
734              }
735              if (cond == 4 + 2 || cond == 4 + 2 + 1) {
736                 return unop(Iop_1Uto32, binop(Iop_CmpNE64, cc_dep1, mkU64(0)));
737              }
738              if (cond == 4 || cond == 4 + 1) {
739                 return unop(Iop_1Uto32, binop(Iop_CmpLT64S, cc_dep1, mkU64(0)));
740              }
741              if (cond == 8 + 4 || cond == 8 + 4 + 1) {
742                 return unop(Iop_1Uto32, binop(Iop_CmpLE64S, cc_dep1, mkU64(0)));
743              }
744              /* cc_dep1 > 0  ---->  0 < cc_dep1 */
745              if (cond == 2 || cond == 2 + 1) {
746                 return unop(Iop_1Uto32, binop(Iop_CmpLT64S, mkU64(0), cc_dep1));
747              }
748              if (cond == 8 + 2 || cond == 8 + 2 + 1) {
749                 return unop(Iop_1Uto32, binop(Iop_CmpLE64S, mkU64(0), cc_dep1));
750              }
751              if (cond == 8 + 4 + 2 || cond == 8 + 4 + 2 + 1) {
752                 return mkU32(1);
753              }
754              /* Remaining case */
755              return mkU32(0);
756           }
757     
758           /* S390_CC_OP_BITWISE */
759           if (cc_op == S390_CC_OP_BITWISE) {
760              /*
761                 cc_dep1 is the result of the boolean operation.
762     
763                 cc == 0  --> cc_dep1 == 0   (cond == 8)
764                 cc == 1  --> cc_dep1 != 0   (cond == 4)
765     
766                 Because cc == 2 and cc == 3 cannot occur the two rightmost bits of
767                 cond are don't cares. Therefore:
768     
769                 cond == 00xx  -> always false
770                 cond == 01xx  -> not equal
771                 cond == 10xx  -> equal
772                 cond == 11xx  -> always true
773              */
774              if ((cond & (8 + 4)) == 8 + 4) {
775                 return mkU32(1);
776              }
777              if (cond & 8) {
778                 return unop(Iop_1Uto32, binop(Iop_CmpEQ64, cc_dep1, mkU64(0)));
779              }
780              if (cond & 4) {
781                 return unop(Iop_1Uto32, binop(Iop_CmpNE64, cc_dep1, mkU64(0)));
782              }
783              /* Remaining case */
784              return mkU32(0);
785           }
786     
787           /* S390_CC_OP_INSERT_CHAR_MASK_32
788              Since the mask comes from an immediate field in the opcode, we
789              expect the mask to be a constant here. That simplifies matters. */
790           if (cc_op == S390_CC_OP_INSERT_CHAR_MASK_32) {
791              ULong mask;
792              UInt imask = 0, shift = 0;
793              IRExpr *word;
794     
795              if (! isC64(cc_dep2)) goto missed;
796     
797              mask = cc_dep2->Iex.Const.con->Ico.U64;
798     
799              /* Extract the 32-bit value from the thunk */
800     
801              word = unop(Iop_64to32, cc_dep1);
802     
803              switch (mask) {
804              case 0:  shift =  0; imask = 0x00000000; break;
805              case 1:  shift = 24; imask = 0x000000FF; break;
806              case 2:  shift = 16; imask = 0x0000FF00; break;
807              case 3:  shift = 16; imask = 0x0000FFFF; break;
808              case 4:  shift =  8; imask = 0x00FF0000; break;
809              case 5:  shift =  8; imask = 0x00FF00FF; break;
810              case 6:  shift =  8; imask = 0x00FFFF00; break;
811              case 7:  shift =  8; imask = 0x00FFFFFF; break;
812              case 8:  shift =  0; imask = 0xFF000000; break;
813              case 9:  shift =  0; imask = 0xFF0000FF; break;
814              case 10: shift =  0; imask = 0xFF00FF00; break;
815              case 11: shift =  0; imask = 0xFF00FFFF; break;
816              case 12: shift =  0; imask = 0xFFFF0000; break;
817              case 13: shift =  0; imask = 0xFFFF00FF; break;
818              case 14: shift =  0; imask = 0xFFFFFF00; break;
819              case 15: shift =  0; imask = 0xFFFFFFFF; break;
820              }
821     
822              /* Select the bits that were inserted */
823              word = binop(Iop_And32, word, mkU32(imask));
824     
825              /* cc == 0  --> all inserted bits zero or mask == 0   (cond == 8)
826                 cc == 1  --> leftmost inserted bit is one          (cond == 4)
827                 cc == 2  --> leftmost inserted bit is zero and not (cond == 2)
828                              all inserted bits are zero
829     
830                 Because cc == 0,1,2 the rightmost bit of the mask is a don't care */
831              if (cond == 8 || cond == 8 + 1) {
832                 return unop(Iop_1Uto32, binop(Iop_CmpEQ32, word, mkU32(0)));
833              }
834              if (cond == 4 + 2 || cond == 4 + 2 + 1) {
835                 return unop(Iop_1Uto32, binop(Iop_CmpNE32, word, mkU32(0)));
836              }
837     
838              /* Sign extend */
839              if (shift != 0) {
840                 word = binop(Iop_Sar32, binop(Iop_Shl32, word, mkU8(shift)),
841                              mkU8(shift));
842              }
843     
844              if (cond == 4 || cond == 4 + 1) {  /* word < 0 */
845                 return unop(Iop_1Uto32, binop(Iop_CmpLT32S, word, mkU32(0)));
846              }
847              if (cond == 2 || cond == 2 + 1) {  /* word > 0 */
848                 return unop(Iop_1Uto32, binop(Iop_CmpLT32S, mkU32(0), word));
849              }
850              if (cond == 8 + 4 || cond == 8 + 4 + 1) {
851                 return unop(Iop_1Uto32, binop(Iop_CmpLE32S, word, mkU32(0)));
852              }
853              if (cond == 8 + 2 || cond == 8 + 2 + 1) {
854                 return unop(Iop_1Uto32, binop(Iop_CmpLE32S, mkU32(0), word));
855              }
856              if (cond == 8 + 4 + 2 || cond == 8 + 4 + 2 + 1) {
857                 return mkU32(1);
858              }
859              /* Remaining case */
860              return mkU32(0);
861           }
862     
863           /* S390_CC_OP_TEST_UNDER_MASK_8
864              Since the mask comes from an immediate field in the opcode, we
865              expect the mask to be a constant here. That simplifies matters. */
866           if (cc_op == S390_CC_OP_TEST_UNDER_MASK_8) {
867              ULong mask16;
868     
869              if (! isC64(cc_dep2)) goto missed;
870     
871              mask16 = cc_dep2->Iex.Const.con->Ico.U64;
872     
873              /* Get rid of the mask16 == 0 case first. Some of the simplifications
874                 below (e.g. for OVFL) only hold if mask16 == 0.  */
875              if (mask16 == 0) {   /* cc == 0 */
876                 if (cond & 0x8) return mkU32(1);
877                 return mkU32(0);
878              }
879     
880              /* cc == 2 is a don't care */
881              if (cond == 8 || cond == 8 + 2) {
882                 return unop(Iop_1Uto32, binop(Iop_CmpEQ64,
883                                               binop(Iop_And64, cc_dep1, cc_dep2),
884                                               mkU64(0)));
885              }
886              if (cond == 7 || cond == 7 - 2) {
887                 return unop(Iop_1Uto32, binop(Iop_CmpNE64,
888                                               binop(Iop_And64, cc_dep1, cc_dep2),
889                                               mkU64(0)));
890              }
891              if (cond == 1 || cond == 1 + 2) {
892                 return unop(Iop_1Uto32, binop(Iop_CmpEQ64,
893                                               binop(Iop_And64, cc_dep1, cc_dep2),
894                                               cc_dep2));
895              }
896              if (cond == 14 || cond == 14 - 2) {  /* ! OVFL */
897                 return unop(Iop_1Uto32, binop(Iop_CmpNE64,
898                                               binop(Iop_And64, cc_dep1, cc_dep2),
899                                               cc_dep2));
900              }
901              goto missed;
902           }
903     
904           /* S390_CC_OP_TEST_UNDER_MASK_16
905              Since the mask comes from an immediate field in the opcode, we
906              expect the mask to be a constant here. That simplifies matters. */
907           if (cc_op == S390_CC_OP_TEST_UNDER_MASK_16) {
908              ULong mask16;
909              UInt msb;
910     
911              if (! isC64(cc_dep2)) goto missed;
912     
913              mask16 = cc_dep2->Iex.Const.con->Ico.U64;
914     
915              /* Get rid of the mask16 == 0 case first. Some of the simplifications
916                 below (e.g. for OVFL) only hold if mask16 == 0.  */
917              if (mask16 == 0) {   /* cc == 0 */
918                 if (cond & 0x8) return mkU32(1);
919                 return mkU32(0);
920              }
921     
922              if (cond == 8) {
923                 return unop(Iop_1Uto32, binop(Iop_CmpEQ64,
924                                               binop(Iop_And64, cc_dep1, cc_dep2),
925                                               mkU64(0)));
926              }
927              if (cond == 7) {
928                 return unop(Iop_1Uto32, binop(Iop_CmpNE64,
929                                               binop(Iop_And64, cc_dep1, cc_dep2),
930                                               mkU64(0)));
931              }
932              if (cond == 1) {
933                 return unop(Iop_1Uto32, binop(Iop_CmpEQ64,
934                                               binop(Iop_And64, cc_dep1, cc_dep2),
935                                               mkU64(mask16)));
936              }
937              if (cond == 14) {  /* ! OVFL */
938                 return unop(Iop_1Uto32, binop(Iop_CmpNE64,
939                                               binop(Iop_And64, cc_dep1, cc_dep2),
940                                               mkU64(mask16)));
941              }
942     
943              /* Find MSB in mask */
944              msb = 0x8000;
945              while (msb > mask16)
946                 msb >>= 1;
947     
948              if (cond == 2) {  /* cc == 2 */
949                 IRExpr *c1, *c2;
950     
951                 /* (cc_dep & msb) != 0 && (cc_dep & mask16) != mask16 */
952                 c1 = binop(Iop_CmpNE64,
953                            binop(Iop_And64, cc_dep1, mkU64(msb)), mkU64(0));
954                 c2 = binop(Iop_CmpNE64,
955                            binop(Iop_And64, cc_dep1, cc_dep2),
956                            mkU64(mask16));
957                 return binop(Iop_And32, unop(Iop_1Uto32, c1),
958                              unop(Iop_1Uto32, c2));
959              }
960     
961              if (cond == 4) {  /* cc == 1 */
962                 IRExpr *c1, *c2;
963     
964                 /* (cc_dep & msb) == 0 && (cc_dep & mask16) != 0 */
965                 c1 = binop(Iop_CmpEQ64,
966                            binop(Iop_And64, cc_dep1, mkU64(msb)), mkU64(0));
967                 c2 = binop(Iop_CmpNE64,
968                            binop(Iop_And64, cc_dep1, cc_dep2),
969                            mkU64(0));
970                 return binop(Iop_And32, unop(Iop_1Uto32, c1),
971                              unop(Iop_1Uto32, c2));
972              }
973     
974              if (cond == 11) {  /* cc == 0,2,3 */
975                 IRExpr *c1, *c2;
976     
977                 c1 = binop(Iop_CmpNE64,
978                            binop(Iop_And64, cc_dep1, mkU64(msb)), mkU64(0));
979                 c2 = binop(Iop_CmpEQ64,
980                            binop(Iop_And64, cc_dep1, cc_dep2),
981                            mkU64(0));
982                 return binop(Iop_Or32, unop(Iop_1Uto32, c1),
983                              unop(Iop_1Uto32, c2));
984              }
985     
986              if (cond == 3) {  /* cc == 2 || cc == 3 */
987                 return unop(Iop_1Uto32,
988                             binop(Iop_CmpNE64,
989                                   binop(Iop_And64, cc_dep1, mkU64(msb)),
990                                   mkU64(0)));
991              }
992              if (cond == 12) { /* cc == 0 || cc == 1 */
993                 return unop(Iop_1Uto32,
994                             binop(Iop_CmpEQ64,
995                                   binop(Iop_And64, cc_dep1, mkU64(msb)),
996                                   mkU64(0)));
997              }
998              // vex_printf("TUM mask = 0x%llx\n", mask16);
999              goto missed;
1000           }
1001     
1002           /* S390_CC_OP_UNSIGNED_SUB_64/32 */
1003           if (cc_op == S390_CC_OP_UNSIGNED_SUB_64 ||
1004               cc_op == S390_CC_OP_UNSIGNED_SUB_32) {
1005              /*
1006                 cc_dep1, cc_dep2 are the zero extended left and right operands
1007     
1008                 cc == 1  --> result != 0, borrow    (cond == 4)
1009                 cc == 2  --> result == 0, no borrow (cond == 2)
1010                 cc == 3  --> result != 0, no borrow (cond == 1)
1011     
1012                 cc = (cc_dep1 == cc_dep2) ? 2
1013                                           : (cc_dep1 > cc_dep2) ? 3 : 1;
1014     
1015                 Because cc == 0 cannot occur the leftmost bit of cond is
1016                 a don't care.
1017              */
1018              if (cond == 1 || cond == 1 + 8) {  /* cc == 3   op2 < op1 */
1019                 return unop(Iop_1Uto32, binop(Iop_CmpLT64U, cc_dep2, cc_dep1));
1020              }
1021              if (cond == 2 || cond == 2 + 8) {  /* cc == 2 */
1022                 return unop(Iop_1Uto32, binop(Iop_CmpEQ64, cc_dep1, cc_dep2));
1023              }
1024              if (cond == 4 || cond == 4 + 8) {  /* cc == 1 */
1025                 return unop(Iop_1Uto32, binop(Iop_CmpLT64U, cc_dep1, cc_dep2));
1026              }
1027              if (cond == 3 || cond == 3 + 8) {  /* cc == 2 || cc == 3 */
1028                 return unop(Iop_1Uto32, binop(Iop_CmpLE64U, cc_dep2, cc_dep1));
1029              }
1030              if (cond == 6 || cond == 6 + 8) {  /* cc == 2 || cc == 1 */
1031                 return unop(Iop_1Uto32, binop(Iop_CmpLE64U, cc_dep1, cc_dep2));
1032              }
1033     
1034              if (cond == 5 || cond == 5 + 8) {  /* cc == 3 || cc == 1 */
1035                 return unop(Iop_1Uto32, binop(Iop_CmpNE64, cc_dep1, cc_dep2));
1036              }
1037              if (cond == 7 || cond == 7 + 8) {
1038                 return mkU32(1);
1039              }
1040              /* Remaining case */
1041              return mkU32(0);
1042           }
1043     
1044           /* S390_CC_OP_UNSIGNED_ADD_64 */
1045           if (cc_op == S390_CC_OP_UNSIGNED_ADD_64) {
1046              /*
1047                 cc_dep1, cc_dep2 are the zero extended left and right operands
1048     
1049                 cc == 0  --> result == 0, no carry  (cond == 8)
1050                 cc == 1  --> result != 0, no carry  (cond == 4)
1051                 cc == 2  --> result == 0, carry     (cond == 2)
1052                 cc == 3  --> result != 0, carry     (cond == 1)
1053              */
1054              if (cond == 8) { /* cc == 0 */
1055                 /* Both inputs are 0 */
1056                 return unop(Iop_1Uto32, binop(Iop_CmpEQ64,
1057                                               binop(Iop_Or64, cc_dep1, cc_dep2),
1058                                               mkU64(0)));
1059              }
1060              if (cond == 7) { /* cc == 1,2,3 */
1061                 /* Not both inputs are 0 */
1062                 return unop(Iop_1Uto32, binop(Iop_CmpNE64,
1063                                               binop(Iop_Or64, cc_dep1, cc_dep2),
1064                                               mkU64(0)));
1065              }
1066              if (cond == 8 + 2) {  /* cc == 0,2  -> result is zero */
1067                 return unop(Iop_1Uto32, binop(Iop_CmpEQ64,
1068                                               binop(Iop_Add64, cc_dep1, cc_dep2),
1069                                               mkU64(0)));
1070              }
1071              if (cond == 4 + 1) {  /* cc == 1,3  -> result is not zero */
1072                 return unop(Iop_1Uto32, binop(Iop_CmpNE64,
1073                                               binop(Iop_Add64, cc_dep1, cc_dep2),
1074                                               mkU64(0)));
1075              }
1076              goto missed;
1077           }
1078     
1079           /* S390_CC_OP_UNSIGNED_ADD_32 */
1080           if (cc_op == S390_CC_OP_UNSIGNED_ADD_32) {
1081              /*
1082                 cc_dep1, cc_dep2 are the zero extended left and right operands
1083     
1084                 cc == 0  --> result == 0, no carry  (cond == 8)
1085                 cc == 1  --> result != 0, no carry  (cond == 4)
1086                 cc == 2  --> result == 0, carry     (cond == 2)
1087                 cc == 3  --> result != 0, carry     (cond == 1)
1088              */
1089              if (cond == 8) { /* cc == 0 */
1090                 /* Both inputs are 0 */
1091                 return unop(Iop_1Uto32, binop(Iop_CmpEQ64,
1092                                               binop(Iop_Or64, cc_dep1, cc_dep2),
1093                                               mkU64(0)));
1094              }
1095              if (cond == 7) { /* cc == 1,2,3 */
1096                 /* Not both inputs are 0 */
1097                 return unop(Iop_1Uto32, binop(Iop_CmpNE64,
1098                                               binop(Iop_Or64, cc_dep1, cc_dep2),
1099                                               mkU64(0)));
1100              }
1101              if (cond == 8 + 2) {  /* cc == 0,2  -> result is zero */
1102                 return unop(Iop_1Uto32, binop(Iop_CmpEQ32,
1103                                               binop(Iop_Add32,
1104                                                     unop(Iop_64to32, cc_dep1),
1105                                                     unop(Iop_64to32, cc_dep2)),
1106                                               mkU32(0)));
1107              }
1108              if (cond == 4 + 1) {  /* cc == 1,3  -> result is not zero */
1109                 return unop(Iop_1Uto32, binop(Iop_CmpNE32,
1110                                               binop(Iop_Add32,
1111                                                     unop(Iop_64to32, cc_dep1),
1112                                                     unop(Iop_64to32, cc_dep2)),
1113                                               mkU32(0)));
1114              }
1115              goto missed;
1116           }
1117     
1118           /* S390_CC_OP_SET */
1119           if (cc_op == S390_CC_OP_SET) {
1120              /* cc_dep1 is the condition code
1121     
1122                 Return 1, if ((cond << cc_dep1) & 0x8) != 0 */
1123     
1124             return unop(Iop_1Uto32,
1125                         binop(Iop_CmpNE64,
1126                               binop(Iop_And64,
1127                                     binop(Iop_Shl64, cond_expr,
1128                                           unop(Iop_64to8, cc_dep1)),
1129                                     mkU64(8)),
1130                               mkU64(0)));
1131           }
1132     
1133           /* S390_CC_OP_TEST_AND_SET */
1134           if (cc_op == S390_CC_OP_TEST_AND_SET) {
1135              /* cc_dep1 is the zero-extended loaded value
1136     
1137                 cc == 0  --> leftmost bit is zero  (cond == 8)
1138                 cc == 1  --> leftmost bit is one   (cond == 4)
1139     
1140                 As cc is either 0 or 1, only the two leftmost bits of the mask
1141                 are relevant. */
1142              IRExpr *bit = binop(Iop_Shr64, cc_dep1, mkU8(7));
1143     
1144              switch (cond & (8 + 4)) {
1145              case 0:     return mkU32(0);
1146              case 4:     return unop(Iop_1Uto32, binop(Iop_CmpNE64, bit, mkU64(0)));
1147              case 8:     return unop(Iop_1Uto32, binop(Iop_CmpEQ64, bit, mkU64(0)));
1148              case 8 + 4: return mkU32(1);
1149              }
1150              /* not reached */
1151           }
1152     
1153     missed:
1154           ;
1155        }
1156     
1157        return NULL;
1158     }
