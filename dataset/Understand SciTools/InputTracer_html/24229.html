119     // - The SP aliases are held in an array which is used as a circular buffer.
120     //   This misses very few constant updates of SP (ie. < 0.1%) while using a
121     //   small, constant structure that will also never fill up and cause
122     //   execution to abort.
123     // - Unused slots have a .temp value of 'IRTemp_INVALID'.
124     // - 'next_SP_alias_slot' is the index where the next alias will be stored.
125     // - If the buffer fills, we circle around and start over-writing
126     //   non-IRTemp_INVALID values.  This is rare, and the overwriting of a
127     //   value that would have subsequently be used is even rarer.
128     // - Every slot below next_SP_alias_slot holds a non-IRTemp_INVALID value.
129     //   The rest either all won't (if we haven't yet circled around) or all
130     //   will (if we have circled around).
131     
132     typedef 
133        struct {
134           IRTemp temp;
135           Long   delta;
136        }
137        SP_Alias;
138     
139     // With 32 slots the buffer fills very rarely -- eg. once in a run of GCC.
140     // And I've tested with smaller values and the wrap-around case works ok.
141     #define N_ALIASES    32
142     static SP_Alias SP_aliases[N_ALIASES];
143     static Int      next_SP_alias_slot = 0;
144     
145     static void clear_SP_aliases(void)
146     {
147        Int i;
148        for (i = 0; i < N_ALIASES; i++) {
149           SP_aliases[i].temp  = IRTemp_INVALID;
150           SP_aliases[i].delta = 0;
151        }
152        next_SP_alias_slot = 0;
153     }
154     
155     static void add_SP_alias(IRTemp temp, Long delta)
156     {
157        vg_assert(temp != IRTemp_INVALID);
158        SP_aliases[ next_SP_alias_slot ].temp  = temp;
159        SP_aliases[ next_SP_alias_slot ].delta = delta;
160        next_SP_alias_slot++;
161        if (N_ALIASES == next_SP_alias_slot) next_SP_alias_slot = 0;
162     }
163     
164     static Bool get_SP_delta(IRTemp temp, ULong* delta)
165     {
166        Int i;      // i must be signed!
167        vg_assert(IRTemp_INVALID != temp);
168        // Search backwards between current buffer position and the start.
169        for (i = next_SP_alias_slot-1; i >= 0; i--) {
170           if (temp == SP_aliases[i].temp) {
171              *delta = SP_aliases[i].delta;
172              return True;
173           }
174        }
175        // Search backwards between the end and the current buffer position.
176        for (i = N_ALIASES-1; i >= next_SP_alias_slot; i--) {
177           if (temp == SP_aliases[i].temp) {
178              *delta = SP_aliases[i].delta;
179              return True;
180           }
181        }
182        return False;
183     }
184     
185     static void update_SP_aliases(Long delta)
186     {
187        Int i;
188        for (i = 0; i < N_ALIASES; i++) {
189           if (SP_aliases[i].temp == IRTemp_INVALID) {
190              return;
191           }
192           SP_aliases[i].delta += delta;
193        }
194     }

196     /* Given a guest IP, get an origin tag for a 1-element stack trace,
197        and wrap it up in an IR atom that can be passed as the origin-tag
198        value for a stack-adjustment helper function. */
199     static IRExpr* mk_ecu_Expr ( Addr64 guest_IP )
200     {
201        UInt ecu;
202        ExeContext* ec
203           = VG_(make_depth_1_ExeContext_from_Addr)( (Addr)guest_IP );
204        vg_assert(ec);
205        ecu = VG_(get_ECU_from_ExeContext)( ec );
206        vg_assert(VG_(is_plausible_ECU)(ecu));
207        /* This is always safe to do, since ecu is only 32 bits, and
208           HWord is 32 or 64. */
209        return mkIRExpr_HWord( (HWord)ecu );
210     }

213     /* For tools that want to know about SP changes, this pass adds
214        in the appropriate hooks.  We have to do it after the tool's
215        instrumentation, so the tool doesn't have to worry about the C calls
216        it adds in, and we must do it before register allocation because
217        spilled temps make it much harder to work out the SP deltas.
218        This it is done with Vex's "second instrumentation" pass.
219     
220        Basically, we look for GET(SP)/PUT(SP) pairs and track constant
221        increments/decrements of SP between them.  (This requires tracking one or
222        more "aliases", which are not exact aliases but instead are tempregs
223        whose value is equal to the SP's plus or minus a known constant.)
224        If all the changes to SP leading up to a PUT(SP) are by known, small
225        constants, we can do a specific call to eg. new_mem_stack_4, otherwise
226        we fall back to the case that handles an unknown SP change.
227     
228        There is some extra complexity to deal correctly with updates to
229        only parts of SP.  Bizarre, but it has been known to happen.
230     */
231     static
232     IRSB* vg_SP_update_pass ( void*             closureV,
233                               IRSB*             sb_in, 
234                               VexGuestLayout*   layout, 
235                               VexGuestExtents*  vge,
236                               IRType            gWordTy, 
237                               IRType            hWordTy )
238     {
239        Int         i, j, minoff_ST, maxoff_ST, sizeof_SP, offset_SP;
240        Int         first_SP, last_SP, first_Put, last_Put;
241        IRDirty     *dcall, *d;
242        IRStmt*     st;
243        IRExpr*     e;
244        IRRegArray* descr;
245        IRType      typeof_SP;
246        Long        delta, con;
247     
248        /* Set up stuff for tracking the guest IP */
249        Bool   curr_IP_known = False;
250        Addr64 curr_IP       = 0;
251     
252        /* Set up BB */
253        IRSB* bb     = emptyIRSB();
254        bb->tyenv    = deepCopyIRTypeEnv(sb_in->tyenv);
255        bb->next     = deepCopyIRExpr(sb_in->next);
256        bb->jumpkind = sb_in->jumpkind;
257     
258        delta = 0;
259     
260        sizeof_SP = layout->sizeof_SP;
261        offset_SP = layout->offset_SP;
262        typeof_SP = sizeof_SP==4 ? Ity_I32 : Ity_I64;
263        vg_assert(sizeof_SP == 4 || sizeof_SP == 8);
264     
265        /* --- Start of #defines --- */
266     
267     #  define IS_ADD(op) (sizeof_SP==4 ? ((op)==Iop_Add32) : ((op)==Iop_Add64))
268     #  define IS_SUB(op) (sizeof_SP==4 ? ((op)==Iop_Sub32) : ((op)==Iop_Sub64))
269     
270     #  define IS_ADD_OR_SUB(op) (IS_ADD(op) || IS_SUB(op))
271     
272     #  define GET_CONST(con)                                                \
273            (sizeof_SP==4 ? (Long)(Int)(con->Ico.U32)                        \
274                          : (Long)(con->Ico.U64))
275     
276     #  define DO_NEW(syze, tmpp)                                            \
277           do {                                                              \
278              Bool vanilla, w_ecu;                                           \
279              vg_assert(curr_IP_known);                                      \
280              vanilla = NULL != VG_(tdict).track_new_mem_stack_##syze;       \
281              w_ecu   = NULL != VG_(tdict).track_new_mem_stack_##syze##_w_ECU; \
282              vg_assert(!(vanilla && w_ecu)); /* can't have both */          \
283              if (!(vanilla || w_ecu))                                       \
284                 goto generic;                                               \
285                                                                             \
286              /* I don't know if it's really necessary to say that the */    \
287              /* call reads the stack pointer.  But anyway, we do. */        \
288              if (w_ecu) {                                                   \
289                 dcall = unsafeIRDirty_0_N(                                  \
290                            2/*regparms*/,                                   \
291                            "track_new_mem_stack_" #syze "_w_ECU",           \
292                            VG_(fnptr_to_fnentry)(                           \
293                               VG_(tdict).track_new_mem_stack_##syze##_w_ECU ), \
294                            mkIRExprVec_2(IRExpr_RdTmp(tmpp),                \
295                                          mk_ecu_Expr(curr_IP))              \
296                         );                                                  \
297              } else {                                                       \
298                 dcall = unsafeIRDirty_0_N(                                  \
299                            1/*regparms*/,                                   \
300                            "track_new_mem_stack_" #syze ,                   \
301                            VG_(fnptr_to_fnentry)(                           \
302                               VG_(tdict).track_new_mem_stack_##syze ),      \
303                            mkIRExprVec_1(IRExpr_RdTmp(tmpp))                \
304                         );                                                  \
305              }                                                              \
306              dcall->nFxState = 1;                                           \
307              dcall->fxState[0].fx     = Ifx_Read;                           \
308              dcall->fxState[0].offset = layout->offset_SP;                  \
309              dcall->fxState[0].size   = layout->sizeof_SP;                  \
310                                                                             \
311              addStmtToIRSB( bb, IRStmt_Dirty(dcall) );                      \
312                                                                             \
313              tl_assert(syze > 0);                                           \
314              update_SP_aliases(syze);                                       \
315                                                                             \
316              n_SP_updates_fast++;                                           \
317                                                                             \
318           } while (0)
319     
320     #  define DO_DIE(syze, tmpp)                                            \
321           do {                                                              \
322              if (!VG_(tdict).track_die_mem_stack_##syze)                    \
323                 goto generic;                                               \
324                                                                             \
325              /* I don't know if it's really necessary to say that the */    \
326              /* call reads the stack pointer.  But anyway, we do. */        \
327              dcall = unsafeIRDirty_0_N(                                     \
328                         1/*regparms*/,                                      \
329                         "track_die_mem_stack_" #syze,                       \
330                         VG_(fnptr_to_fnentry)(                              \
331                            VG_(tdict).track_die_mem_stack_##syze ),         \
332                         mkIRExprVec_1(IRExpr_RdTmp(tmpp))                   \
333                      );                                                     \
334              dcall->nFxState = 1;                                           \
335              dcall->fxState[0].fx     = Ifx_Read;                           \
336              dcall->fxState[0].offset = layout->offset_SP;                  \
337              dcall->fxState[0].size   = layout->sizeof_SP;                  \
338                                                                             \
339              addStmtToIRSB( bb, IRStmt_Dirty(dcall) );                      \
340                                                                             \
341              tl_assert(syze > 0);                                           \
342              update_SP_aliases(-(syze));                                    \
343                                                                             \
344              n_SP_updates_fast++;                                           \
345                                                                             \
346           } while (0)
347     
348        /* --- End of #defines --- */
349     
350        clear_SP_aliases();
351     
352        for (i = 0; i <  sb_in->stmts_used; i++) {
353     
354           st = sb_in->stmts[i];
355     
356           if (st->tag == Ist_IMark) {
357              curr_IP_known = True;
358              curr_IP       = st->Ist.IMark.addr;
359           }
360     
361           /* t = Get(sp):   curr = t, delta = 0 */
362           if (st->tag != Ist_WrTmp) goto case2;
363           e = st->Ist.WrTmp.data;
364           if (e->tag != Iex_Get)              goto case2;
365           if (e->Iex.Get.offset != offset_SP) goto case2;
366           if (e->Iex.Get.ty != typeof_SP)     goto case2;
367           vg_assert( typeOfIRTemp(bb->tyenv, st->Ist.WrTmp.tmp) == typeof_SP );
368           add_SP_alias(st->Ist.WrTmp.tmp, 0);
369           addStmtToIRSB( bb, st );
370           continue;
371     
372          case2:
373           /* t' = curr +/- const:   curr = t',  delta +=/-= const */
374           if (st->tag != Ist_WrTmp) goto case3;
375           e = st->Ist.WrTmp.data;
376           if (e->tag != Iex_Binop) goto case3;
377           if (e->Iex.Binop.arg1->tag != Iex_RdTmp) goto case3;
378           if (!get_SP_delta(e->Iex.Binop.arg1->Iex.RdTmp.tmp, &delta)) goto case3;
379           if (e->Iex.Binop.arg2->tag != Iex_Const) goto case3;
380           if (!IS_ADD_OR_SUB(e->Iex.Binop.op)) goto case3;
381           con = GET_CONST(e->Iex.Binop.arg2->Iex.Const.con);
382           vg_assert( typeOfIRTemp(bb->tyenv, st->Ist.WrTmp.tmp) == typeof_SP );
383           if (IS_ADD(e->Iex.Binop.op)) {
384              add_SP_alias(st->Ist.WrTmp.tmp, delta + con);
385           } else {
386              add_SP_alias(st->Ist.WrTmp.tmp, delta - con);
387           }
388           addStmtToIRSB( bb, st );
389           continue;
390     
391          case3:
392           /* t' = curr:   curr = t' */
393           if (st->tag != Ist_WrTmp) goto case4;
394           e = st->Ist.WrTmp.data;
395           if (e->tag != Iex_RdTmp) goto case4;
396           if (!get_SP_delta(e->Iex.RdTmp.tmp, &delta)) goto case4;
397           vg_assert( typeOfIRTemp(bb->tyenv, st->Ist.WrTmp.tmp) == typeof_SP );
398           add_SP_alias(st->Ist.WrTmp.tmp, delta);
399           addStmtToIRSB( bb, st );
400           continue;
401     
402          case4:
403           /* Put(sp) = curr */
404           /* More generally, we must correctly handle a Put which writes
405              any part of SP, not just the case where all of SP is
406              written. */
407           if (st->tag != Ist_Put) goto case5;
408           first_SP  = offset_SP;
409           last_SP   = first_SP + sizeof_SP - 1;
410           first_Put = st->Ist.Put.offset;
411           last_Put  = first_Put
412                       + sizeofIRType( typeOfIRExpr( bb->tyenv, st->Ist.Put.data ))
413                       - 1;
414           vg_assert(first_SP <= last_SP);
415           vg_assert(first_Put <= last_Put);
416     
417           if (last_Put < first_SP || last_SP < first_Put)
418              goto case5; /* no overlap */
419     
420           if (st->Ist.Put.data->tag == Iex_RdTmp
421               && get_SP_delta(st->Ist.Put.data->Iex.RdTmp.tmp, &delta)) {
422              IRTemp tttmp = st->Ist.Put.data->Iex.RdTmp.tmp;
423              /* Why should the following assertion hold?  Because any
424                 alias added by put_SP_alias must be of a temporary which
425                 has the same type as typeof_SP, and whose value is a Get
426                 at exactly offset_SP of size typeof_SP.  Each call to
427                 put_SP_alias is immediately preceded by an assertion that
428                 we are putting in a binding for a correctly-typed
429                 temporary. */
430              vg_assert( typeOfIRTemp(bb->tyenv, tttmp) == typeof_SP );
431              /* From the same type-and-offset-correctness argument, if 
432                 we found a useable alias, it must for an "exact" write of SP. */
433              vg_assert(first_SP == first_Put);
434              vg_assert(last_SP == last_Put);
435              switch (delta) {
436                 case    0:                      addStmtToIRSB(bb,st); continue;
437                 case    4: DO_DIE(  4,  tttmp); addStmtToIRSB(bb,st); continue;
438                 case   -4: DO_NEW(  4,  tttmp); addStmtToIRSB(bb,st); continue;
439                 case    8: DO_DIE(  8,  tttmp); addStmtToIRSB(bb,st); continue;
440                 case   -8: DO_NEW(  8,  tttmp); addStmtToIRSB(bb,st); continue;
441                 case   12: DO_DIE(  12, tttmp); addStmtToIRSB(bb,st); continue;
442                 case  -12: DO_NEW(  12, tttmp); addStmtToIRSB(bb,st); continue;
443                 case   16: DO_DIE(  16, tttmp); addStmtToIRSB(bb,st); continue;
444                 case  -16: DO_NEW(  16, tttmp); addStmtToIRSB(bb,st); continue;
445                 case   32: DO_DIE(  32, tttmp); addStmtToIRSB(bb,st); continue;
446                 case  -32: DO_NEW(  32, tttmp); addStmtToIRSB(bb,st); continue;
447                 case  112: DO_DIE( 112, tttmp); addStmtToIRSB(bb,st); continue;
448                 case -112: DO_NEW( 112, tttmp); addStmtToIRSB(bb,st); continue;
449                 case  128: DO_DIE( 128, tttmp); addStmtToIRSB(bb,st); continue;
450                 case -128: DO_NEW( 128, tttmp); addStmtToIRSB(bb,st); continue;
451                 case  144: DO_DIE( 144, tttmp); addStmtToIRSB(bb,st); continue;
452                 case -144: DO_NEW( 144, tttmp); addStmtToIRSB(bb,st); continue;
453                 case  160: DO_DIE( 160, tttmp); addStmtToIRSB(bb,st); continue;
454                 case -160: DO_NEW( 160, tttmp); addStmtToIRSB(bb,st); continue;
455                 default:  
456                    /* common values for ppc64: 144 128 160 112 176 */
457                    n_SP_updates_generic_known++;
458                    goto generic;
459              }
460           } else {
461              /* Deal with an unknown update to SP.  We're here because
462                 either:
463                 (1) the Put does not exactly cover SP; it is a partial update.
464                     Highly unlikely, but has been known to happen for 16-bit
465                     Windows apps running on Wine, doing 16-bit adjustments to
466                     %sp.
467                 (2) the Put does exactly cover SP, but we are unable to
468                     determine how the value relates to the old SP.  In any 
469                     case, we cannot assume that the Put.data value is a tmp;
470                     we must assume it can be anything allowed in flat IR (tmp
471                     or const).
472              */
473              IRTemp  old_SP;
474              n_SP_updates_generic_unknown++;
475     
476              // Nb: if all is well, this generic case will typically be
477              // called something like every 1000th SP update.  If it's more than
478              // that, the above code may be missing some cases.
479             generic:
480              /* Pass both the old and new SP values to this helper.  Also,
481                 pass an origin tag, even if it isn't needed. */
482              old_SP = newIRTemp(bb->tyenv, typeof_SP);
483              addStmtToIRSB( 
484                 bb,
485                 IRStmt_WrTmp( old_SP, IRExpr_Get(offset_SP, typeof_SP) ) 
486              );
487     
488              /* Now we know what the old value of SP is.  But knowing the new
489                 value is a bit tricky if there is a partial write. */
490              if (first_Put == first_SP && last_Put == last_SP) {
491                /* The common case, an exact write to SP.  So st->Ist.Put.data
492                   does hold the new value; simple. */
493                 vg_assert(curr_IP_known);
494                 dcall = unsafeIRDirty_0_N( 
495                            3/*regparms*/, 
496                            "VG_(unknown_SP_update)", 
497                            VG_(fnptr_to_fnentry)( &VG_(unknown_SP_update) ),
498                            mkIRExprVec_3( IRExpr_RdTmp(old_SP), st->Ist.Put.data,
499                                           mk_ecu_Expr(curr_IP) ) 
500                         );
501                 addStmtToIRSB( bb, IRStmt_Dirty(dcall) );
502                 /* don't forget the original assignment */
503                 addStmtToIRSB( bb, st );
504              } else {
505                 /* We have a partial update to SP.  We need to know what
506                    the new SP will be, and hand that to the helper call,
507                    but when the helper call happens, SP must hold the
508                    value it had before the update.  Tricky.
509                    Therefore use the following kludge:
510                    1. do the partial SP update (Put)
511                    2. Get the new SP value into a tmp, new_SP
512                    3. Put old_SP
513                    4. Call the helper
514                    5. Put new_SP
515                 */
516                 IRTemp new_SP;
517                 /* 1 */
518                 addStmtToIRSB( bb, st );
519                 /* 2 */
520                 new_SP = newIRTemp(bb->tyenv, typeof_SP);
521                 addStmtToIRSB( 
522                    bb,
523                    IRStmt_WrTmp( new_SP, IRExpr_Get(offset_SP, typeof_SP) ) 
524                 );
525                 /* 3 */
526                 addStmtToIRSB( bb, IRStmt_Put(offset_SP, IRExpr_RdTmp(old_SP) ));
527                 /* 4 */
528                 vg_assert(curr_IP_known);
529                 dcall = unsafeIRDirty_0_N( 
530                            3/*regparms*/, 
531                            "VG_(unknown_SP_update)", 
532                            VG_(fnptr_to_fnentry)( &VG_(unknown_SP_update) ),
533                            mkIRExprVec_3( IRExpr_RdTmp(old_SP),
534                                           IRExpr_RdTmp(new_SP), 
535                                           mk_ecu_Expr(curr_IP) )
536                         );
537                 addStmtToIRSB( bb, IRStmt_Dirty(dcall) );
538                 /* 5 */
539                 addStmtToIRSB( bb, IRStmt_Put(offset_SP, IRExpr_RdTmp(new_SP) ));
540              }
541     
542              /* Forget what we already know. */
543              clear_SP_aliases();
544     
545              /* If this is a Put of a tmp that exactly updates SP,
546                 start tracking aliases against this tmp. */
547     
548              if (first_Put == first_SP && last_Put == last_SP
549                  && st->Ist.Put.data->tag == Iex_RdTmp) {
550                 vg_assert( typeOfIRTemp(bb->tyenv, st->Ist.Put.data->Iex.RdTmp.tmp)
551                            == typeof_SP );
552                 add_SP_alias(st->Ist.Put.data->Iex.RdTmp.tmp, 0);
553              }
554              continue;
555           }
556     
557          case5:
558           /* PutI or Dirty call which overlaps SP: complain.  We can't
559              deal with SP changing in weird ways (well, we can, but not at
560              this time of night).  */
561           if (st->tag == Ist_PutI) {
562              descr = st->Ist.PutI.descr;
563              minoff_ST = descr->base;
564              maxoff_ST = descr->base 
565                          + descr->nElems * sizeofIRType(descr->elemTy) - 1;
566              if (!(offset_SP > maxoff_ST
567                    || (offset_SP + sizeof_SP - 1) < minoff_ST))
568                 goto complain;
569           }
570           if (st->tag == Ist_Dirty) {
571              d = st->Ist.Dirty.details;
572              for (j = 0; j < d->nFxState; j++) {
573                 minoff_ST = d->fxState[j].offset;
574                 maxoff_ST = d->fxState[j].offset + d->fxState[j].size - 1;
575                 if (d->fxState[j].fx == Ifx_Read || d->fxState[j].fx == Ifx_None)
576                    continue;
577                 if (!(offset_SP > maxoff_ST
578                       || (offset_SP + sizeof_SP - 1) < minoff_ST))
579                    goto complain;
580              }
581           }
582     
583           /* well, not interesting.  Just copy and keep going. */
584           addStmtToIRSB( bb, st );
585     
586        } /* for (i = 0; i < sb_in->stmts_used; i++) */
587     
588        return bb;
589     
590       complain:
591        VG_(core_panic)("vg_SP_update_pass: PutI or Dirty which overlaps SP");
592     
593     #undef IS_ADD
594     #undef IS_SUB
595     #undef IS_ADD_OR_SUB
596     #undef GET_CONST
597     #undef DO_NEW
598     #undef DO_DIE
599     }

605     /* Extra comments re self-checking translations and self-modifying
606        code.  (JRS 14 Oct 05).
607     
608        There are 3 modes:
609        (1) no checking: all code assumed to be not self-modifying
610        (2) partial: known-problematic situations get a self-check
611        (3) full checking: all translations get a self-check
612     
613        As currently implemented, the default is (2).  (3) is always safe,
614        but very slow.  (1) works mostly, but fails for gcc nested-function
615        code which uses trampolines on the stack; this situation is
616        detected and handled by (2).
617     
618        ----------
619     
620        A more robust and transparent solution, which is not currently
621        implemented, is a variant of (2): if a translation is made from an
622        area which aspacem says does not have 'w' permission, then it can
623        be non-self-checking.  Otherwise, it needs a self-check.
624     
625        This is complicated by Vex's basic-block chasing.  If a self-check
626        is requested, then Vex will not chase over basic block boundaries
627        (it's too complex).  However there is still a problem if it chases
628        from a non-'w' area into a 'w' area.
629     
630        I think the right thing to do is:
631     
632        - if a translation request starts in a 'w' area, ask for a
633          self-checking translation, and do not allow any chasing (make
634          chase_into_ok return False).  Note that the latter is redundant
635          in the sense that Vex won't chase anyway in this situation.
636     
637        - if a translation request starts in a non-'w' area, do not ask for
638          a self-checking translation.  However, do not allow chasing (as
639          determined by chase_into_ok) to go into a 'w' area.
640     
641        The result of this is that all code inside 'w' areas is self
642        checking.
643     
644        To complete the trick, there is a caveat: we must watch the
645        client's mprotect calls.  If pages are changed from non-'w' to 'w'
646        then we should throw away all translations which intersect the
647        affected area, so as to force them to be redone with self-checks.
648     
649        ----------
650     
651        The above outlines the conditions under which bb chasing is allowed
652        from a self-modifying-code point of view.  There are other
653        situations pertaining to function redirection in which it is
654        necessary to disallow chasing, but those fall outside the scope of
655        this comment.
656     */

734     /* This is a callback passed to LibVEX_Translate.  It stops Vex from
735        chasing into function entry points that we wish to redirect.
736        Chasing across them obviously defeats the redirect mechanism, with
737        bad effects for Memcheck, Addrcheck, and possibly others.
738     
739        Also, we must stop Vex chasing into blocks for which we might want
740        to self checking.
741     */
742     static Bool chase_into_ok ( void* closureV, Addr64 addr64 )
743     {
744        Addr               addr    = (Addr)addr64;
745        NSegment const*    seg     = VG_(am_find_nsegment)(addr);
746     
747        /* Work through a list of possibilities why we might not want to
748           allow a chase. */
749     
750        /* Destination not in a plausible segment? */
751        if (!translations_allowable_from_seg(seg))
752           goto dontchase;
753     
754        /* Destination is redirected? */
755        if (addr != VG_(redir_do_lookup)(addr, NULL))
756           goto dontchase;
757     
758     #  if defined(VG_PLAT_USES_PPCTOC)
759        /* This needs to be at the start of its own block.  Don't chase. Re
760           ULong_to_Ptr, be careful to ensure we only compare 32 bits on a
761           32-bit target.*/
762        if (ULong_to_Ptr(addr64)
763            == (void*)&VG_(ppctoc_magic_redirect_return_stub))
764           goto dontchase;
765     #  endif
766     
767        /* overly conservative, but .. don't chase into the distinguished
768           address that m_transtab uses as an empty-slot marker for
769           VG_(tt_fast). */
770        if (addr == TRANSTAB_BOGUS_GUEST_ADDR)
771           goto dontchase;
772     
773        /* well, ok then.  go on and chase. */
774        return True;
775     
776        vg_assert(0);
777        /*NOTREACHED*/
778     
779       dontchase:
780        if (0) VG_(printf)("not chasing into 0x%lx\n", addr);
781        return False;
782     }

809     /* Generate code to push word-typed expression 'e' onto this thread's
810        redir stack, checking for stack overflow and generating code to
811        bomb out if so. */
812     
813     static void gen_PUSH ( IRSB* bb, IRExpr* e )
814     {
815        IRRegArray* descr;
816        IRTemp      t1;
817        IRExpr*     one;
818     
819     #  if defined(VGP_ppc64_linux) || defined(VGP_ppc64_aix5)
820        Int    stack_size       = VEX_GUEST_PPC64_REDIR_STACK_SIZE;
821        Int    offB_REDIR_SP    = offsetof(VexGuestPPC64State,guest_REDIR_SP);
822        Int    offB_REDIR_STACK = offsetof(VexGuestPPC64State,guest_REDIR_STACK);
823        Int    offB_EMWARN      = offsetof(VexGuestPPC64State,guest_EMWARN);
824        Bool   is64             = True;
825        IRType ty_Word          = Ity_I64;
826        IROp   op_CmpNE         = Iop_CmpNE64;
827        IROp   op_Sar           = Iop_Sar64;
828        IROp   op_Sub           = Iop_Sub64;
829        IROp   op_Add           = Iop_Add64;
830        IRExpr*(*mkU)(ULong)    = mkU64;
831        vg_assert(VG_WORDSIZE == 8);
832     #  else
833        Int    stack_size       = VEX_GUEST_PPC32_REDIR_STACK_SIZE;
834        Int    offB_REDIR_SP    = offsetof(VexGuestPPC32State,guest_REDIR_SP);
835        Int    offB_REDIR_STACK = offsetof(VexGuestPPC32State,guest_REDIR_STACK);
836        Int    offB_EMWARN      = offsetof(VexGuestPPC32State,guest_EMWARN);
837        Bool   is64             = False;
838        IRType ty_Word          = Ity_I32;
839        IROp   op_CmpNE         = Iop_CmpNE32;
840        IROp   op_Sar           = Iop_Sar32;
841        IROp   op_Sub           = Iop_Sub32;
842        IROp   op_Add           = Iop_Add32;
843        IRExpr*(*mkU)(UInt)     = mkU32;
844        vg_assert(VG_WORDSIZE == 4);
845     #  endif
846     
847        vg_assert(sizeof(void*) == VG_WORDSIZE);
848        vg_assert(sizeof(Word)  == VG_WORDSIZE);
849        vg_assert(sizeof(Addr)  == VG_WORDSIZE);
850     
851        descr = mkIRRegArray( offB_REDIR_STACK, ty_Word, stack_size );
852        t1    = newIRTemp( bb->tyenv, ty_Word );
853        one   = mkU(1);
854     
855        vg_assert(typeOfIRExpr(bb->tyenv, e) == ty_Word);
856     
857        /* t1 = guest_REDIR_SP + 1 */
858        addStmtToIRSB(
859           bb, 
860           IRStmt_WrTmp(
861              t1, 
862              IRExpr_Binop(op_Add, IRExpr_Get( offB_REDIR_SP, ty_Word ), one)
863           )
864        );
865     
866        /* Bomb out if t1 >=s stack_size, that is, (stack_size-1)-t1 <s 0.
867           The destination (0) is a bit bogus but it doesn't matter since
868           this is an unrecoverable error and will lead to Valgrind
869           shutting down.  _EMWARN is set regardless - that's harmless
870           since is only has a meaning if the exit is taken. */
871        addStmtToIRSB(
872           bb,
873           IRStmt_Put(offB_EMWARN, mkU32(EmWarn_PPC64_redir_overflow))
874        );
875        addStmtToIRSB(
876           bb,
877           IRStmt_Exit(
878              IRExpr_Binop(
879                 op_CmpNE,
880                 IRExpr_Binop(
881                    op_Sar,
882                    IRExpr_Binop(op_Sub,mkU(stack_size-1),IRExpr_RdTmp(t1)),
883                    mkU8(8 * VG_WORDSIZE - 1)
884                 ),
885                 mkU(0)
886              ),
887              Ijk_EmFail,
888              is64 ? IRConst_U64(0) : IRConst_U32(0)
889           )
890        );
891     
892        /* guest_REDIR_SP = t1 */
893        addStmtToIRSB(bb, IRStmt_Put(offB_REDIR_SP, IRExpr_RdTmp(t1)));
894     
895        /* guest_REDIR_STACK[t1+0] = e */
896        /* PutI/GetI have I32-typed indexes regardless of guest word size */
897        addStmtToIRSB(
898           bb, 
899           IRStmt_PutI(descr, narrowTo32(bb->tyenv,IRExpr_RdTmp(t1)), 0, e)
900        );
901     }

904     /* Generate code to pop a word-sized value from this thread's redir
905        stack, binding it to a new temporary, which is returned.  As with
906        gen_PUSH, an overflow check is also performed. */
907     
908     static IRTemp gen_POP ( IRSB* bb )
909     {
910     #  if defined(VGP_ppc64_linux) || defined(VGP_ppc64_aix5)
911        Int    stack_size       = VEX_GUEST_PPC64_REDIR_STACK_SIZE;
912        Int    offB_REDIR_SP    = offsetof(VexGuestPPC64State,guest_REDIR_SP);
913        Int    offB_REDIR_STACK = offsetof(VexGuestPPC64State,guest_REDIR_STACK);
914        Int    offB_EMWARN      = offsetof(VexGuestPPC64State,guest_EMWARN);
915        Bool   is64             = True;
916        IRType ty_Word          = Ity_I64;
917        IROp   op_CmpNE         = Iop_CmpNE64;
918        IROp   op_Sar           = Iop_Sar64;
919        IROp   op_Sub           = Iop_Sub64;
920        IRExpr*(*mkU)(ULong)    = mkU64;
921     #  else
922        Int    stack_size       = VEX_GUEST_PPC32_REDIR_STACK_SIZE;
923        Int    offB_REDIR_SP    = offsetof(VexGuestPPC32State,guest_REDIR_SP);
924        Int    offB_REDIR_STACK = offsetof(VexGuestPPC32State,guest_REDIR_STACK);
925        Int    offB_EMWARN      = offsetof(VexGuestPPC32State,guest_EMWARN);
926        Bool   is64             = False;
927        IRType ty_Word          = Ity_I32;
928        IROp   op_CmpNE         = Iop_CmpNE32;
929        IROp   op_Sar           = Iop_Sar32;
930        IROp   op_Sub           = Iop_Sub32;
931        IRExpr*(*mkU)(UInt)     = mkU32;
932     #  endif
933     
934        IRRegArray* descr = mkIRRegArray( offB_REDIR_STACK, ty_Word, stack_size );
935        IRTemp      t1    = newIRTemp( bb->tyenv, ty_Word );
936        IRTemp      res   = newIRTemp( bb->tyenv, ty_Word );
937        IRExpr*     one   = mkU(1);
938     
939        vg_assert(sizeof(void*) == VG_WORDSIZE);
940        vg_assert(sizeof(Word)  == VG_WORDSIZE);
941        vg_assert(sizeof(Addr)  == VG_WORDSIZE);
942     
943        /* t1 = guest_REDIR_SP */
944        addStmtToIRSB(
945           bb, 
946           IRStmt_WrTmp( t1, IRExpr_Get( offB_REDIR_SP, ty_Word ) )
947        );
948     
949        /* Bomb out if t1 < 0.  Same comments as gen_PUSH apply. */
950        addStmtToIRSB(
951           bb,
952           IRStmt_Put(offB_EMWARN, mkU32(EmWarn_PPC64_redir_underflow))
953        );
954        addStmtToIRSB(
955           bb,
956           IRStmt_Exit(
957              IRExpr_Binop(
958                 op_CmpNE,
959                 IRExpr_Binop(
960                    op_Sar,
961                    IRExpr_RdTmp(t1),
962                    mkU8(8 * VG_WORDSIZE - 1)
963                 ),
964                 mkU(0)
965              ),
966              Ijk_EmFail,
967              is64 ? IRConst_U64(0) : IRConst_U32(0)
968           )
969        );
970     
971        /* res = guest_REDIR_STACK[t1+0] */
972        /* PutI/GetI have I32-typed indexes regardless of guest word size */
973        addStmtToIRSB(
974           bb,
975           IRStmt_WrTmp(
976              res, 
977              IRExpr_GetI(descr, narrowTo32(bb->tyenv,IRExpr_RdTmp(t1)), 0)
978           )
979        );
980     
981        /* guest_REDIR_SP = t1-1 */
982        addStmtToIRSB(
983           bb, 
984           IRStmt_Put(offB_REDIR_SP, IRExpr_Binop(op_Sub, IRExpr_RdTmp(t1), one))
985        );
986     
987        return res;
988     }

1022     static void gen_pop_R2_LR_then_bLR ( IRSB* bb )
1023     {
1024     #  if defined(VGP_ppc64_linux) || defined(VGP_ppc64_aix5)
1025        Int    offB_GPR2 = offsetof(VexGuestPPC64State,guest_GPR2);
1026        Int    offB_LR   = offsetof(VexGuestPPC64State,guest_LR);
1027        IRTemp old_R2    = newIRTemp( bb->tyenv, Ity_I64 );
1028        IRTemp old_LR    = newIRTemp( bb->tyenv, Ity_I64 );
1029        /* Restore R2 */
1030        old_R2 = gen_POP( bb );
1031        addStmtToIRSB( bb, IRStmt_Put( offB_GPR2, IRExpr_RdTmp(old_R2)) );
1032        /* Restore LR */
1033        old_LR = gen_POP( bb );
1034        addStmtToIRSB( bb, IRStmt_Put( offB_LR, IRExpr_RdTmp(old_LR)) );
1035        /* Branch to LR */
1036        /* re boring, we arrived here precisely because a wrapped fn did a
1037           blr (hence Ijk_Ret); so we should just mark this jump as Boring,
1038           else one _Call will have resulted in two _Rets. */
1039        bb->jumpkind = Ijk_Boring;
1040        bb->next = IRExpr_Binop(Iop_And64, IRExpr_RdTmp(old_LR), mkU64(~(3ULL)));
1041     
1042     #  elif defined(VGP_ppc32_aix5)
1043        Int    offB_GPR2 = offsetof(VexGuestPPC32State,guest_GPR2);
1044        Int    offB_LR   = offsetof(VexGuestPPC32State,guest_LR);
1045        IRTemp old_R2    = newIRTemp( bb->tyenv, Ity_I32 );
1046        IRTemp old_LR    = newIRTemp( bb->tyenv, Ity_I32 );
1047        /* Restore R2 */
1048        old_R2 = gen_POP( bb );
1049        addStmtToIRSB( bb, IRStmt_Put( offB_GPR2, IRExpr_RdTmp(old_R2)) );
1050        /* Restore LR */
1051        old_LR = gen_POP( bb );
1052        addStmtToIRSB( bb, IRStmt_Put( offB_LR, IRExpr_RdTmp(old_LR)) );
1053     
1054        /* Branch to LR */
1055        /* re boring, we arrived here precisely because a wrapped fn did a
1056           blr (hence Ijk_Ret); so we should just mark this jump as Boring,
1057           else one _Call will have resulted in two _Rets. */
1058        bb->jumpkind = Ijk_Boring;
1059        bb->next = IRExpr_Binop(Iop_And32, IRExpr_RdTmp(old_LR), mkU32(~3));
1060     
1061     #  else
1062     #    error Platform is not TOC-afflicted, fortunately
1063     #  endif
1064     }

1066     static
1067     Bool mk_preamble__ppctoc_magic_return_stub ( void* closureV, IRSB* bb )
1068     {
1069        VgCallbackClosure* closure = (VgCallbackClosure*)closureV;
1070        /* Since we're creating the entire IRSB right here, give it a
1071           proper IMark, as it won't get one any other way, and cachegrind
1072           will barf if it doesn't have one (fair enough really). */
1073        addStmtToIRSB( bb, IRStmt_IMark( closure->readdr, 4 ) );
1074        /* Generate the magic sequence:
1075              pop R2 from hidden stack
1076              pop LR from hidden stack
1077              goto LR
1078        */
1079        gen_pop_R2_LR_then_bLR(bb);
1080        return True; /* True == this is the entire BB; don't disassemble any
1081                        real insns into it - just hand it directly to
1082                        optimiser/instrumenter/backend. */
1083     }

1089     /* This is the IR preamble generator used for replacement
1090        functions.  It adds code to set the guest_NRADDR{_GPR2} to zero
1091        (technically not necessary, but facilitates detecting mixups in
1092        which a replacement function has been erroneously declared using
1093        VG_REPLACE_FUNCTION_Z{U,Z} when instead it should have been written
1094        using VG_WRAP_FUNCTION_Z{U,Z}).
1095     
1096        On with-TOC platforms the follow hacks are also done: LR and R2 are
1097        pushed onto a hidden stack, R2 is set to the correct value for the
1098        replacement function, and LR is set to point at the magic
1099        return-stub address.  Setting LR causes the return of the
1100        wrapped/redirected function to lead to our magic return stub, which
1101        restores LR and R2 from said stack and returns for real.
1102     
1103        VG_(get_StackTrace_wrk) understands that the LR value may point to
1104        the return stub address, and that in that case it can get the real
1105        LR value from the hidden stack instead. */
1106     static 
1107     Bool mk_preamble__set_NRADDR_to_zero ( void* closureV, IRSB* bb )
1108     {
1109        Int nraddr_szB
1110           = sizeof(((VexGuestArchState*)0)->guest_NRADDR);
1111        vg_assert(nraddr_szB == 4 || nraddr_szB == 8);
1112        vg_assert(nraddr_szB == VG_WORDSIZE);
1113        addStmtToIRSB( 
1114           bb,
1115           IRStmt_Put( 
1116              offsetof(VexGuestArchState,guest_NRADDR),
1117              nraddr_szB == 8 ? mkU64(0) : mkU32(0)
1118           )
1119        );
1120     #  if defined(VG_PLAT_USES_PPCTOC)
1121        { VgCallbackClosure* closure = (VgCallbackClosure*)closureV;
1122          addStmtToIRSB(
1123             bb,
1124             IRStmt_Put(
1125                offsetof(VexGuestArchState,guest_NRADDR_GPR2),
1126                VG_WORDSIZE==8 ? mkU64(0) : mkU32(0)
1127             )
1128          );
1129          gen_push_and_set_LR_R2 ( bb, VG_(get_tocptr)( closure->readdr ) );
1130        }
1131     #  endif
1132        return False;
1133     }

1194          if (seg && seg->hasR) {
1195             switch ( *(UInt*)(Addr)bl_target ) {
1196                case 0xd9c1ff70: /* stfd f14,-144(r1) */

1199                case 0xda21ff88: /* stfd f17,-120(r1) */

1203                case 0xdaa1ffa8: /* stfd f21,-88(r1) */
1204                case 0xdac1ffb0: /* stfd f22,-80(r1) */
1205                case 0xdae1ffb8: /* stfd f23,-72(r1) */
1206                case 0xdb01ffc0: /* stfd f24,-64(r1) */

1221     /* --------------- main translation function --------------- */
1222     
1223     /* Note: see comments at top of m_redir.c for the Big Picture on how
1224        redirections are managed. */
1225     
1226     typedef 
1227        enum {
1228           /* normal translation, redir neither requested nor inhibited */
1229           T_Normal, 
1230           /* redir translation, function-wrap (set _NRADDR) style */
1231           T_Redir_Wrap,
1232           /* redir translation, replacement (don't set _NRADDR) style */
1233           T_Redir_Replace,
1234           /* a translation in which redir is specifically disallowed */
1235           T_NoRedir
1236        }
1237        T_Kind;
1238     
1239     /* Translate the basic block beginning at NRADDR, and add it to the
1240        translation cache & translation table.  Unless
1241        DEBUGGING_TRANSLATION is true, in which case the call is being done
1242        for debugging purposes, so (a) throw away the translation once it
1243        is made, and (b) produce a load of debugging output.  If
1244        ALLOW_REDIRECTION is False, do not attempt redirection of NRADDR,
1245        and also, put the resulting translation into the no-redirect tt/tc
1246        instead of the normal one.
1247     
1248        TID is the identity of the thread requesting this translation.
1249     */
1250     
1251     Bool VG_(translate) ( ThreadId tid, 
1252                           Addr64   nraddr,
1253                           Bool     debugging_translation,
1254                           Int      debugging_verbosity,
1255                           ULong    bbs_done,
1256                           Bool     allow_redirection )
1257     {
1258        Addr64             addr;
1259        T_Kind             kind;
1260        Int                tmpbuf_used, verbosity, i;
1261        Bool               do_self_check;
1262        Bool (*preamble_fn)(void*,IRSB*);
1263        VexArch            vex_arch;
1264        VexArchInfo        vex_archinfo;
1265        VexAbiInfo         vex_abiinfo;
1266        VexGuestExtents    vge;
1267        VexTranslateArgs   vta;
1268        VexTranslateResult tres;
1269        VgCallbackClosure  closure;
1270     
1271        /* Make sure Vex is initialised right. */
1272     
1273        static Bool vex_init_done = False;
1274     
1275        if (!vex_init_done) {
1276           LibVEX_Init ( &failure_exit, &log_bytes, 
1277                         1,     /* debug_paranoia */ 
1278                         False, /* valgrind support */
1279                         &VG_(clo_vex_control) );
1280           vex_init_done = True;
1281        }
1282     
1283        /* Establish the translation kind and actual guest address to
1284           start from.  Sets (addr,kind). */
1285        if (allow_redirection) {
1286           Bool isWrap;
1287           Addr64 tmp = VG_(redir_do_lookup)( nraddr, &isWrap );
1288           if (tmp == nraddr) {
1289              /* no redirection found */
1290              addr = nraddr;
1291              kind = T_Normal;
1292           } else {
1293              /* found a redirect */
1294              addr = tmp;
1295              kind = isWrap ? T_Redir_Wrap : T_Redir_Replace;
1296           }
1297        } else {
1298           addr = nraddr;
1299           kind = T_NoRedir;
1300        }
1301     
1302        /* Established: (nraddr, addr, kind) */
1303     
1304        /* Printing redirection info. */
1305     
1306        if ((kind == T_Redir_Wrap || kind == T_Redir_Replace)
1307            && (VG_(clo_verbosity) >= 2 || VG_(clo_trace_redir))) {
1308           Bool ok;
1309           Char name1[64] = "";
1310           Char name2[64] = "";
1311           name1[0] = name2[0] = 0;
1312           ok = VG_(get_fnname_w_offset)(nraddr, name1, 64);
1313           if (!ok) VG_(strcpy)(name1, "???");
1314           ok = VG_(get_fnname_w_offset)(addr, name2, 64);
1315           if (!ok) VG_(strcpy)(name2, "???");
1316           VG_(message)(Vg_DebugMsg, 
1317                        "REDIR: 0x%llx (%s) redirected to 0x%llx (%s)\n",
1318                        nraddr, name1,
1319                        addr, name2 );
1320        }
1321     
1322        if (!debugging_translation)
1323           VG_TRACK( pre_mem_read, Vg_CoreTranslate, 
1324                                   tid, "(translator)", addr, 1 );
1325     
1326        /* If doing any code printing, print a basic block start marker */
1327        if (VG_(clo_trace_flags) || debugging_translation) {
1328           Char fnname[64] = "";
1329           VG_(get_fnname_w_offset)(addr, fnname, 64);
1330           VG_(printf)(
1331                   "==== SB %d [tid %d] %s(0x%llx) SBs exec'd %lld ====\n",
1332                   VG_(get_bbs_translated)(), (Int)tid, fnname, addr, 
1333                   bbs_done);
1334        }
1335     
1336        /* Are we allowed to translate here? */
1337     
1338        { /* BEGIN new scope specially for 'seg' */
1339        NSegment const* seg = VG_(am_find_nsegment)(addr);
1340     
1341        if ( (!translations_allowable_from_seg(seg))
1342             || addr == TRANSTAB_BOGUS_GUEST_ADDR ) {
1343           if (VG_(clo_trace_signals))
1344              VG_(message)(Vg_DebugMsg, "translations not allowed here (0x%llx)"
1345                                        " - throwing SEGV\n", addr);
1346           /* U R busted, sonny.  Place your hands on your head and step
1347              away from the orig_addr. */
1348           /* Code address is bad - deliver a signal instead */
1349           if (seg != NULL) {
1350              /* There's some kind of segment at the requested place, but we
1351                 aren't allowed to execute code here. */
1352              VG_(synth_fault_perms)(tid, addr);
1353           } else {
1354             /* There is no segment at all; we are attempting to execute in
1355                the middle of nowhere. */
1356              VG_(synth_fault_mapping)(tid, addr);
1357           }
1358           return False;
1359        }
1360     
1361        /* Do we want a self-checking translation? */
1362        do_self_check = self_check_required( seg, tid );
1363     
1364        /* True if a debug trans., or if bit N set in VG_(clo_trace_codegen). */
1365        verbosity = 0;
1366        if (debugging_translation) {
1367           verbosity = debugging_verbosity;
1368        }
1369        else
1370        if ( (VG_(clo_trace_flags) > 0
1371             && VG_(get_bbs_translated)() >= VG_(clo_trace_notbelow) )) {
1372           verbosity = VG_(clo_trace_flags);
1373        }
1374     
1375        /* Figure out which preamble-mangling callback to send. */
1376        preamble_fn = NULL;
1377        if (kind == T_Redir_Replace)
1378           preamble_fn = mk_preamble__set_NRADDR_to_zero;
1379        else 
1380        if (kind == T_Redir_Wrap)
1381           preamble_fn = mk_preamble__set_NRADDR_to_nraddr;
1382     
1383     #  if defined(VG_PLAT_USES_PPCTOC)
1384        if (ULong_to_Ptr(nraddr)
1385            == (void*)&VG_(ppctoc_magic_redirect_return_stub)) {
1386           /* If entering the special return stub, this means a wrapped or
1387              redirected function is returning.  Make this translation one
1388              which restores R2 and LR from the thread's hidden redir
1389              stack, and branch to the (restored) link register, thereby
1390              really causing the function to return. */
1391           vg_assert(kind == T_Normal);
1392           vg_assert(nraddr == addr);
1393           preamble_fn = mk_preamble__ppctoc_magic_return_stub;
1394        }
1395     #  endif
1396     
1397        /* ------ Actually do the translation. ------ */
1398        tl_assert2(VG_(tdict).tool_instrument,
1399                   "you forgot to set VgToolInterface function 'tool_instrument'");
1400     
1401        /* Get the CPU info established at startup. */
1402        VG_(machine_get_VexArchInfo)( &vex_arch, &vex_archinfo );
1403     
1404        /* Set up 'abiinfo' structure with stuff Vex needs to know about
1405           the guest and host ABIs. */
1406     
1407        LibVEX_default_VexAbiInfo( &vex_abiinfo );
1408        vex_abiinfo.guest_stack_redzone_size = VG_STACK_REDZONE_SZB;
1409     
1410     #  if defined(VGP_amd64_linux)
1411        vex_abiinfo.guest_amd64_assume_fs_is_zero  = True;
1412     #  endif
1413     #  if defined(VGP_amd64_darwin)
1414        vex_abiinfo.guest_amd64_assume_gs_is_0x60  = True;
1415     #  endif
1416     #  if defined(VGP_ppc32_linux)
1417        vex_abiinfo.guest_ppc_zap_RZ_at_blr        = False;
1418        vex_abiinfo.guest_ppc_zap_RZ_at_bl         = NULL;
1419        vex_abiinfo.host_ppc32_regalign_int64_args = True;
1420     #  endif
1421     #  if defined(VGP_ppc64_linux)
1422        vex_abiinfo.guest_ppc_zap_RZ_at_blr        = True;
1423        vex_abiinfo.guest_ppc_zap_RZ_at_bl         = const_True;
1424        vex_abiinfo.host_ppc_calls_use_fndescrs    = True;
1425     #  endif
1426     #  if defined(VGP_ppc32_aix5) || defined(VGP_ppc64_aix5)
1427        vex_abiinfo.guest_ppc_zap_RZ_at_blr        = False;
1428        vex_abiinfo.guest_ppc_zap_RZ_at_bl         = bl_RZ_zap_ok_for_AIX;
1429        vex_abiinfo.guest_ppc_sc_continues_at_LR   = True;
1430        vex_abiinfo.host_ppc_calls_use_fndescrs    = True;
1431     #  endif
1432     
1433        /* Set up closure args. */
1434        closure.tid    = tid;
1435        closure.nraddr = nraddr;
1436        closure.readdr = addr;
1437     
1438        /* Set up args for LibVEX_Translate. */
1439        vta.arch_guest       = vex_arch;
1440        vta.archinfo_guest   = vex_archinfo;
1441        vta.arch_host        = vex_arch;
1442        vta.archinfo_host    = vex_archinfo;
1443        vta.abiinfo_both     = vex_abiinfo;
1444        vta.guest_bytes      = (UChar*)ULong_to_Ptr(addr);
1445        vta.guest_bytes_addr = (Addr64)addr;
1446        vta.callback_opaque  = (void*)&closure;
1447        vta.chase_into_ok    = chase_into_ok;
1448        vta.preamble_function = preamble_fn;
1449        vta.guest_extents    = &vge;
1450        vta.host_bytes       = tmpbuf;
1451        vta.host_bytes_size  = N_TMPBUF;
1452        vta.host_bytes_used  = &tmpbuf_used;
1453        { /* At this point we have to reconcile Vex's view of the
1454             instrumentation callback - which takes a void* first argument
1455             - with Valgrind's view, in which the first arg is a
1456             VgCallbackClosure*.  Hence the following longwinded casts.
1457             They are entirely legal but longwinded so as to maximise the
1458             chance of the C typechecker picking up any type snafus. */
1459          IRSB*(*f)(VgCallbackClosure*,
1460                    IRSB*,VexGuestLayout*,VexGuestExtents*,
1461                    IRType,IRType)
1462            = VG_(tdict).tool_instrument;
1463          IRSB*(*g)(void*,
1464                    IRSB*,VexGuestLayout*,VexGuestExtents*,
1465                    IRType,IRType)
1466            = (IRSB*(*)(void*,IRSB*,VexGuestLayout*,VexGuestExtents*,IRType,IRType))f;
1467          vta.instrument1    = g;
1468        }
1469        /* No need for type kludgery here. */
1470        vta.instrument2      = need_to_handle_SP_assignment()
1471                                  ? vg_SP_update_pass
1472                                  : NULL;
1473        vta.finaltidy        = VG_(needs).final_IR_tidy_pass
1474                                  ? VG_(tdict).tool_final_IR_tidy_pass
1475                                  : NULL;
1476        vta.do_self_check    = do_self_check;
1477        vta.traceflags       = verbosity;
1478     
1479        /* Set up the dispatch-return info.  For archs without a link
1480           register, vex generates a jump back to the specified dispatch
1481           address.  Else, it just generates a branch-to-LR. */
1482     #  if defined(VGA_x86) || defined(VGA_amd64)
1483        vta.dispatch 
1484           = (!allow_redirection)
1485             ? /* It's a no-redir translation.  Will be run with the nonstandard
1486                dispatcher VG_(run_a_noredir_translation)
1487                and so needs a nonstandard return point. */
1488               (void*) &VG_(run_a_noredir_translation__return_point)
1489     
1490             : /* normal translation.  Uses VG_(run_innerloop).  Return
1491                  point depends on whether we're profiling bbs or not. */
1492               VG_(clo_profile_flags) > 0
1493               ? (void*) &VG_(run_innerloop__dispatch_profiled)
1494               : (void*) &VG_(run_innerloop__dispatch_unprofiled);
1495     #  elif defined(VGA_ppc32) || defined(VGA_ppc64) \
1496             || defined(VGA_arm) || defined(VGA_s390x)
1497        /* See comment libvex.h; machine has link register --> dipatch = NULL */
1498        vta.dispatch = NULL;
1499     #  else
1500     #    error "Unknown arch"
1501     #  endif
1502     
1503        /* Sheesh.  Finally, actually _do_ the translation! */
1504        tres = LibVEX_Translate ( &vta );
1505     
1506        vg_assert(tres == VexTransOK);
1507        vg_assert(tmpbuf_used <= N_TMPBUF);
1508        vg_assert(tmpbuf_used > 0);
1509     
1510        /* Tell aspacem of all segments that have had translations taken
1511           from them.  Optimisation: don't re-look up vge.base[0] since seg
1512           should already point to it. */
1513     
1514        vg_assert( vge.base[0] == (Addr64)addr );
1515        /* set 'translations taken from this segment' flag */
1516        VG_(am_set_segment_hasT_if_SkFileC_or_SkAnonC)( (NSegment*)seg );
1517        } /* END new scope specially for 'seg' */
1518     
1519        for (i = 1; i < vge.n_used; i++) {
1520           NSegment const* seg 
1521              = VG_(am_find_nsegment)( vge.base[i] );
1522           /* set 'translations taken from this segment' flag */
1523           VG_(am_set_segment_hasT_if_SkFileC_or_SkAnonC)( (NSegment*)seg );
1524        }
1525     
1526        /* Copy data at trans_addr into the translation cache. */
1527        vg_assert(tmpbuf_used > 0 && tmpbuf_used < 65536);
1528     
1529        // If debugging, don't do anything with the translated block;  we
1530        // only did this for the debugging output produced along the way.
1531        if (!debugging_translation) {
1532     
1533           if (kind != T_NoRedir) {
1534               // Put it into the normal TT/TC structures.  This is the
1535               // normal case.
1536     
1537               // Note that we use nraddr (the non-redirected address), not
1538               // addr, which might have been changed by the redirection
1539               VG_(add_to_transtab)( &vge,
1540                                     nraddr,
1541                                     (Addr)(&tmpbuf[0]), 
1542                                     tmpbuf_used,
1543                                     do_self_check );
1544           } else {
1545               VG_(add_to_unredir_transtab)( &vge,
1546                                             nraddr,
1547                                             (Addr)(&tmpbuf[0]), 
1548                                             tmpbuf_used );
1549           }
1550        }
1551     
1552        return True;
1553     }
