101        else if VG_INT_CLO (arg, "--segment-merging-interval", segment_merge_interval)
102        {}
103        else if VG_BOOL_CLO(arg, "--show-confl-seg",      show_confl_seg) {}
104        else if VG_BOOL_CLO(arg, "--show-stack-usage",    s_show_stack_usage) {}
105        else if VG_BOOL_CLO(arg, "--trace-alloc",         s_trace_alloc) {}
106        else if VG_BOOL_CLO(arg, "--trace-barrier",       trace_barrier) {}
107        else if VG_BOOL_CLO(arg, "--trace-clientobj",     trace_clientobj) {}
108        else if VG_BOOL_CLO(arg, "--trace-cond",          trace_cond) {}

126        if (check_stack_accesses != -1)

157           DRD_(barrier_set_trace)(trace_barrier);
158        if (trace_clientobj != -1)
159           DRD_(clientobj_set_trace)(trace_clientobj);
160        if (trace_cond != -1)
161           DRD_(cond_set_trace)(trace_cond);
162        if (trace_csw != -1)
163           DRD_(thread_trace_context_switches)(trace_csw);
164        if (trace_fork_join != -1)
165           DRD_(thread_set_trace_fork_join)(trace_fork_join);
166        if (trace_hb != -1)

168        if (trace_conflict_set != -1)
169           DRD_(thread_trace_conflict_set)(trace_conflict_set);
170        if (trace_conflict_set_bm != -1)
171           DRD_(thread_trace_conflict_set_bm)(trace_conflict_set_bm);

261     static void drd_pre_mem_read_asciiz(const CorePart part,
262                                         const ThreadId tid,
263                                         Char* const s,
264                                         const Addr a)
265     {
266        const char* p = (void*)a;
267        SizeT size = 0;
268     
269        // Don't segfault if the string starts in an obviously stupid
270        // place.  Actually we should check the whole string, not just
271        // the start address, but that's too much trouble.  At least
272        // checking the first byte is better than nothing.  See #255009.
273        if (!VG_(am_is_valid_for_client) (a, 1, VKI_PROT_READ))
274           return;
275     
276        /* Note: the expression '*p' reads client memory and may crash if the */
277        /* client provided an invalid pointer !                               */
278        while (*p)
279        {
280           p++;
281           size++;
282        }
283        if (size > 0)
284        {
285           DRD_(trace_load)(a, size);
286        }
287     }

301     static __inline__
302     void drd_start_using_mem(const Addr a1, const SizeT len,
303                              const Bool is_stack_mem)
304     {
305        const Addr a2 = a1 + len;
306     
307        tl_assert(a1 <= a2);
308     
309        if (!is_stack_mem && s_trace_alloc)
310           VG_(message)(Vg_UserMsg, "Started using memory range 0x%lx + %ld%s\n",
311                        a1, len, DRD_(running_thread_inside_pthread_create)()
312                        ? " (inside pthread_create())" : "");
313     
314        if (!is_stack_mem && DRD_(g_free_is_write))
315           DRD_(thread_stop_using_mem)(a1, a2);
316     
317        if (UNLIKELY(DRD_(any_address_is_traced)()))
318        {
319           DRD_(trace_mem_access)(a1, len, eStart);
320        }
321     
322        if (UNLIKELY(DRD_(running_thread_inside_pthread_create)()))
323        {
324           DRD_(start_suppression)(a1, a2, "pthread_create()");
325        }
326     }

342     static __inline__
343     void drd_stop_using_mem(const Addr a1, const SizeT len,
344                             const Bool is_stack_mem)
345     {
346        const Addr a2 = a1 + len;
347     
348        tl_assert(a1 <= a2);
349     
350        if (UNLIKELY(DRD_(any_address_is_traced)()))
351           DRD_(trace_mem_access)(a1, len, eEnd);
352     
353        if (!is_stack_mem && s_trace_alloc)
354           VG_(message)(Vg_UserMsg, "Stopped using memory range 0x%lx + %ld\n",
355                        a1, len);
356     
357        if (!is_stack_mem || DRD_(get_check_stack_accesses)())
358        {
359           if (is_stack_mem || !DRD_(g_free_is_write))
360     	 DRD_(thread_stop_using_mem)(a1, a2);
361           else if (DRD_(g_free_is_write))
362     	 DRD_(trace_store)(a1, len);
363           DRD_(clientobj_stop_using_mem)(a1, a2);
364           DRD_(suppression_stop_using_mem)(a1, a2);
365        }
366     }

385     /**
386      * Suppress data race reports on all addresses contained in .plt and
387      * .got.plt sections inside the address range [ a, a + len [. The data in
388      * these sections is modified by _dl_relocate_object() every time a function
389      * in a shared library is called for the first time. Since the first call
390      * to a function in a shared library can happen from a multithreaded context,
391      * such calls can cause conflicting accesses. See also Ulrich Drepper's
392      * paper "How to Write Shared Libraries" for more information about relocation
393      * (http://people.redhat.com/drepper/dsohowto.pdf).
394      */
395     static void DRD_(suppress_relocation_conflicts)(const Addr a, const SizeT len)
396     {
397        const DebugInfo* di;
398     
399     #if 0
400        VG_(printf)("Evaluating range @ 0x%lx size %ld\n", a, len);
401     #endif
402     
403        for (di = VG_(next_DebugInfo)(0); di; di = VG_(next_DebugInfo)(di))
404        {
405           Addr  avma;
406           SizeT size;
407     
408           avma = VG_(DebugInfo_get_plt_avma)(di);
409           size = VG_(DebugInfo_get_plt_size)(di);
410           tl_assert((avma && size) || (avma == 0 && size == 0));
411           if (size > 0)
412           {
413     #if 0
414              VG_(printf)("Suppressing .plt @ 0x%lx size %ld\n", avma, size);
415     #endif
416              tl_assert(VG_(DebugInfo_sect_kind)(NULL, 0, avma) == Vg_SectPLT);
417              DRD_(start_suppression)(avma, avma + size, ".plt");
418           }
419     
420           avma = VG_(DebugInfo_get_gotplt_avma)(di);
421           size = VG_(DebugInfo_get_gotplt_size)(di);
422           tl_assert((avma && size) || (avma == 0 && size == 0));
423           if (size > 0)
424           {
425     #if 0
426              VG_(printf)("Suppressing .got.plt @ 0x%lx size %ld\n", avma, size);
427     #endif
428              tl_assert(VG_(DebugInfo_sect_kind)(NULL, 0, avma) == Vg_SectGOTPLT);
429              DRD_(start_suppression)(avma, avma + size, ".gotplt");
430           }
431        }
432     }

447     /* the addresses in range [ a, a + len [ may now be used by the client.  */
448     /* Assumption: stacks grow downward.                                     */
449     static __inline__
450     void drd_start_using_mem_stack(const Addr a, const SizeT len)

452        DRD_(thread_set_stack_min)(DRD_(thread_get_running_tid)(),

460     /* the addresses [ a, a + len [ are no longer accessible for the client.   */
461     /* Assumption: stacks grow downward.                                       */
462     static __inline__
463     void drd_stop_using_mem_stack(const Addr a, const SizeT len)
464     {
465        DRD_(thread_set_stack_min)(DRD_(thread_get_running_tid)(),
466                                   a + len - VG_STACK_REDZONE_SZB);
467        drd_stop_using_mem(a - VG_STACK_REDZONE_SZB, len + VG_STACK_REDZONE_SZB,
468                           True);
469     }

471     static
472     Bool on_alt_stack(const Addr a)
473     {
474        ThreadId vg_tid;
475        Addr alt_min;
476        SizeT alt_size;
477     
478        vg_tid = VG_(get_running_tid)();
479        alt_min = VG_(thread_get_altstack_min)(vg_tid);
480        alt_size = VG_(thread_get_altstack_size)(vg_tid);
481        return (SizeT)(a - alt_min) < alt_size;
482     }

498     /**
499      * Callback function invoked by the Valgrind core before a signal is delivered.
500      */
501     static
502     void drd_pre_deliver_signal(const ThreadId vg_tid, const Int sigNo,
503                                 const Bool alt_stack)
504     {
505        DrdThreadId drd_tid;
506     
507        drd_tid = DRD_(VgThreadIdToDrdThreadId)(vg_tid);
508        DRD_(thread_set_on_alt_stack)(drd_tid, alt_stack);
509        if (alt_stack)
510        {
511           /*
512            * As soon a signal handler has been invoked on the alternate stack,
513            * switch to stack memory handling functions that can handle the
514            * alternate stack.
515            */
516           VG_(track_new_mem_stack)(drd_start_using_mem_alt_stack);
517           VG_(track_die_mem_stack)(drd_stop_using_mem_alt_stack);
518        }
519     }

521     /**
522      * Callback function invoked by the Valgrind core after a signal is delivered,
523      * at least if the signal handler did not longjmp().
524      */
525     static
526     void drd_post_deliver_signal(const ThreadId vg_tid, const Int sigNo)
527     {
528        DrdThreadId drd_tid;
529     
530        drd_tid = DRD_(VgThreadIdToDrdThreadId)(vg_tid);
531        DRD_(thread_set_on_alt_stack)(drd_tid, False);
532        if (DRD_(thread_get_threads_on_alt_stack)() == 0)
533        {
534           VG_(track_new_mem_stack)(drd_start_using_mem_stack);
535           VG_(track_die_mem_stack)(drd_stop_using_mem_stack);
536        }
537     }

542      *
543      * @param[in] a   Start of address range.
544      * @param[in] len Address range length.
545      * @param[in] tid Valgrind thread ID for whom the signal frame is being
546      *                constructed.
547      */
548     static void drd_start_using_mem_stack_signal(const Addr a, const SizeT len,
549                                                  ThreadId tid)
550     {
551        DRD_(thread_set_vg_running_tid)(VG_(get_running_tid)());
552        drd_start_using_mem(a, len, True);
553     }
554     
555     static void drd_stop_using_mem_stack_signal(Addr a, SizeT len)
556     {
557        drd_stop_using_mem(a, len, True);
558     }

560     static
561     void drd_pre_thread_create(const ThreadId creator, const ThreadId created)
562     {
563        const DrdThreadId drd_creator = DRD_(VgThreadIdToDrdThreadId)(creator);
564        tl_assert(created != VG_INVALID_THREADID);
565        DRD_(thread_pre_create)(drd_creator, created);
566        if (DRD_(IsValidDrdThreadId)(drd_creator))
567        {
568           DRD_(thread_new_segment)(drd_creator);
569        }
570        if (DRD_(thread_get_trace_fork_join)())
571        {
572           VG_(message)(Vg_DebugMsg,
573                        "drd_pre_thread_create creator = %d, created = %d\n",
574                        drd_creator, created);
575        }
576     }

578     /* Called by Valgrind's core before any loads or stores are performed on */
579     /* the context of thread "created". At startup, this function is called  */
580     /* with arguments (0,1).                                                 */
581     static
582     void drd_post_thread_create(const ThreadId vg_created)
583     {
584        DrdThreadId drd_created;
585     
586        tl_assert(vg_created != VG_INVALID_THREADID);
587     
588        drd_created = DRD_(thread_post_create)(vg_created);
589        if (DRD_(thread_get_trace_fork_join)())
590        {
591           VG_(message)(Vg_DebugMsg,
592                        "drd_post_thread_create created = %d\n",
593                        drd_created);
594        }
595        if (! DRD_(get_check_stack_accesses)())
596        {
597           DRD_(start_suppression)(DRD_(thread_get_stack_max)(drd_created)
598                                   - DRD_(thread_get_stack_size)(drd_created),
599                                   DRD_(thread_get_stack_max)(drd_created),
600                                   "stack");
601        }
602     }
603     
604     /* Called after a thread has performed its last memory access. */
605     static void drd_thread_finished(ThreadId vg_tid)
606     {
607        DrdThreadId drd_tid;
608     
609        tl_assert(VG_(get_running_tid)() == vg_tid);
610     
611        drd_tid = DRD_(VgThreadIdToDrdThreadId)(vg_tid);
612        if (DRD_(thread_get_trace_fork_join)())
613        {
614           VG_(message)(Vg_DebugMsg,
615                        "drd_thread_finished tid = %d%s\n",
616                        drd_tid,
617                        DRD_(thread_get_joinable)(drd_tid)
618                        ? ""
619                        : " (which is a detached thread)");
620        }
621        if (s_show_stack_usage)
622        {
623           const SizeT stack_size = DRD_(thread_get_stack_size)(drd_tid);
624           const SizeT used_stack
625              = (DRD_(thread_get_stack_max)(drd_tid)
626                 - DRD_(thread_get_stack_min_min)(drd_tid));
627           VG_(message)(Vg_UserMsg,
628                        "thread %d%s finished and used %ld bytes out of %ld"
629                        " on its stack. Margin: %ld bytes.\n",
630                        drd_tid,
631                        DRD_(thread_get_joinable)(drd_tid)
632                        ? ""
633                        : " (which is a detached thread)",
634                        used_stack,
635                        stack_size,
636                        stack_size - used_stack);
637     
638        }
639        drd_stop_using_mem(DRD_(thread_get_stack_min)(drd_tid),
640                           DRD_(thread_get_stack_max)(drd_tid)
641                           - DRD_(thread_get_stack_min)(drd_tid),
642                           True);
643        DRD_(thread_set_record_loads)(drd_tid, False);
644        DRD_(thread_set_record_stores)(drd_tid, False);
645        DRD_(thread_finished)(drd_tid);
646     }

650      * only surviving thread in the child process. Cleans up thread state.
651      * See also http://pubs.opengroup.org/onlinepubs/9699919799/functions/pthread_atfork.html for a detailed discussion of using fork() in combination with mutexes.

684     static void DRD_(fini)(Int exitcode)
685     {
686        // thread_print_all();
687        if (VG_(clo_verbosity) == 1 && !VG_(clo_xml)) {
688           VG_(message)(Vg_UserMsg,
689                        "For counts of detected and suppressed errors, "
690                        "rerun with: -v\n");
691        }
692     
693        if (VG_(clo_stats) || s_print_stats)
694        {
695           ULong pu = DRD_(thread_get_update_conflict_set_count)();
696           ULong pu_seg_cr = DRD_(thread_get_update_conflict_set_new_sg_count)();
697           ULong pu_mtx_cv = DRD_(thread_get_update_conflict_set_sync_count)();
698           ULong pu_join   = DRD_(thread_get_update_conflict_set_join_count)();
699     
700           VG_(message)(Vg_UserMsg,
701                        "   thread: %lld context switches.\n",
702                        DRD_(thread_get_context_switch_count)());
703           VG_(message)(Vg_UserMsg,
704                        "confl set: %lld full updates and %lld partial updates;\n",
705     		   DRD_(thread_get_compute_conflict_set_count)(),
706     		   pu);
707           VG_(message)(Vg_UserMsg,
708                        "           %lld partial updates during segment creation,\n",
709     		   pu_seg_cr);
710           VG_(message)(Vg_UserMsg,
711                        "           %lld because of mutex/sema/cond.var. operations,\n",
712     		   pu_mtx_cv);
713           VG_(message)(Vg_UserMsg,
714                        "           %lld because of barrier/rwlock operations and\n",
715     		   pu - pu_seg_cr - pu_mtx_cv - pu_join);
716           VG_(message)(Vg_UserMsg,
717                        "           %lld partial updates because of thread join"
718     		   " operations.\n",
719     		   pu_join);
720           VG_(message)(Vg_UserMsg,
721                        " segments: created %lld segments, max %lld alive,\n",
722                        DRD_(sg_get_segments_created_count)(),
723                        DRD_(sg_get_max_segments_alive_count)());
724           VG_(message)(Vg_UserMsg,
725                        "           %lld discard points and %lld merges.\n",
726                        DRD_(thread_get_discard_ordered_segments_count)(),
727                        DRD_(sg_get_segment_merge_count)());
728           VG_(message)(Vg_UserMsg,
729                        "segmnt cr: %lld mutex, %lld rwlock, %lld semaphore and"
730                        " %lld barrier.\n",
731                        DRD_(get_mutex_segment_creation_count)(),
732                        DRD_(get_rwlock_segment_creation_count)(),
733                        DRD_(get_semaphore_segment_creation_count)(),
734                        DRD_(get_barrier_segment_creation_count)());
735           VG_(message)(Vg_UserMsg,
736                        "  bitmaps: %lld level one"
737                        " and %lld level two bitmaps were allocated.\n",
738                        DRD_(bm_get_bitmap_creation_count)(),
739                        DRD_(bm_get_bitmap2_creation_count)());
740           VG_(message)(Vg_UserMsg,
741                        "    mutex: %lld non-recursive lock/unlock events.\n",
742                        DRD_(get_mutex_lock_count)());
743           DRD_(print_malloc_stats)();
744        }
