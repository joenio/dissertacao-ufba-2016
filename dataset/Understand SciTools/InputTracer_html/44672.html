17     /* Cause memcheck to complain about the address "a" and so to print
18        its best guess as to what "a" actually is.  a must be
19        addressible. */
20     
21     void croak ( void* aV )
22     {
23       char* a = (char*)aV;
24       char* undefp = malloc(1);
25       char saved = *a;
26       assert(undefp);
27       *a = *undefp;
28       VALGRIND_CHECK_MEM_IS_DEFINED(a, 1);
29       *a = saved;
30       free(undefp);
31     }
