6     int base256(int val)
7     {
8     /* interpret the  bitstream representing val as a base 256 number for testing
9      * the parity instrs
10      */
11        int sum = 0;
12        int scale = 1;
13        int i;
14     
15        for (i = 0; i < 8; i++) {
16           int bit = val & 1;
17           sum = sum + bit * scale;
18           val <<= 1;
19           scale *= 256;
20        }
21        return sum;
22     }
