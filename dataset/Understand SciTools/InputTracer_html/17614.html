53     /* This module creates and removes signal frames for signal deliveries
54        on ppc32-linux.
55     
56        Note, this file contains kernel-specific knowledge in the form of
57        'struct sigframe' and 'struct rt_sigframe'.  How does that relate
58        to the vki kernel interface stuff?
59     
60        Either a 'struct sigframe' or a 'struct rtsigframe' is pushed 
61        onto the client's stack.  This contains a subsidiary
62        vki_ucontext.  That holds the vcpu's state across the signal, 
63        so that the sighandler can mess with the vcpu state if it
64        really wants.
65     
66        FIXME: sigcontexting is basically broken for the moment.  When
67        delivering a signal, the integer registers and %eflags are
68        correctly written into the sigcontext, however the FP and SSE state
69        is not.  When returning from a signal, only the integer registers
70        are restored from the sigcontext; the rest of the CPU state is
71        restored to what it was before the signal.
72     
73        This will be fixed.
74     */
75

81     // A structure in which to save the application's registers
82     // during the execution of signal handlers.
83     
84     // Linux has 2 signal frame structures: one for normal signal
85     // deliveries, and one for SA_SIGINFO deliveries (also known as RT
86     // signals).
87     //
88     // In theory, so long as we get the arguments to the handler function
89     // right, it doesn't matter what the exact layout of the rest of the
90     // frame is.  Unfortunately, things like gcc's exception unwinding
91     // make assumptions about the locations of various parts of the frame,
92     // so we need to duplicate it exactly.
93     
94     /* Structure containing bits of information that we want to save
95        on signal delivery. */
96     struct vg_sig_private {
97        UInt magicPI;
98        UInt sigNo_private;
99        VexGuestPPC32State vex_shadow1;
100        VexGuestPPC32State vex_shadow2;
101     };
102     
103     /* Structure put on stack for signal handlers with SA_SIGINFO clear. */
104     struct nonrt_sigframe {
105        UInt gap1[16];
106        struct vki_sigcontext sigcontext;
107        struct vki_mcontext mcontext;
108        struct vg_sig_private priv;
109        unsigned char abigap[224];
110     };
111     
112     /* Structure put on stack for signal handlers with SA_SIGINFO set. */
113     struct rt_sigframe {
114        UInt gap1[20];
115        vki_siginfo_t siginfo;
116        struct vki_ucontext ucontext;
117        struct vg_sig_private priv;
118        unsigned char abigap[224];
119     };

136     static 
137     void stack_mcontext ( struct vki_mcontext *mc, 
138                           ThreadState* tst, 
139                           Bool use_rt_sigreturn,
140                           UInt fault_addr )
141     {
142        VG_TRACK( pre_mem_write, Vg_CoreSignal, tst->tid, "signal frame mcontext",
143                  (Addr)mc, sizeof(struct vki_pt_regs) );
144     
145     #  define DO(gpr)  mc->mc_gregs[VKI_PT_R0+gpr] = tst->arch.vex.guest_GPR##gpr
146        DO(0);  DO(1);  DO(2);  DO(3);  DO(4);  DO(5);  DO(6);  DO(7);
147        DO(8);  DO(9);  DO(10); DO(11); DO(12); DO(13); DO(14); DO(15);
148        DO(16); DO(17); DO(18); DO(19); DO(20); DO(21); DO(22); DO(23);
149        DO(24); DO(25); DO(26); DO(27); DO(28); DO(29); DO(30); DO(31);
150     #  undef DO
151     
152        mc->mc_gregs[VKI_PT_NIP]     = tst->arch.vex.guest_CIA;
153        mc->mc_gregs[VKI_PT_MSR]     = 0xf032;   /* pretty arbitrary */
154        mc->mc_gregs[VKI_PT_ORIG_R3] = tst->arch.vex.guest_GPR3;
155        mc->mc_gregs[VKI_PT_CTR]     = tst->arch.vex.guest_CTR;
156        mc->mc_gregs[VKI_PT_LNK]     = tst->arch.vex.guest_LR;
157        mc->mc_gregs[VKI_PT_XER]     = LibVEX_GuestPPC32_get_XER(&tst->arch.vex);
158        mc->mc_gregs[VKI_PT_CCR]     = LibVEX_GuestPPC32_get_CR(&tst->arch.vex);
159        mc->mc_gregs[VKI_PT_MQ]      = 0;
160        mc->mc_gregs[VKI_PT_TRAP]    = 0;
161        mc->mc_gregs[VKI_PT_DAR]     = fault_addr;
162        mc->mc_gregs[VKI_PT_DSISR]   = 0;
163        mc->mc_gregs[VKI_PT_RESULT]  = 0;
164        VG_TRACK( post_mem_write, Vg_CoreSignal, tst->tid, 
165                  (Addr)mc, sizeof(struct vki_pt_regs) );
166     
167        /* XXX should do FP and vector regs */
168     
169        /* set up signal return trampoline */
170        /* NB.  5 Sept 07.  mc->mc_pad[0..1] used to contain a the code to
171           which the signal handler returns, and it just did sys_sigreturn
172           or sys_rt_sigreturn.  But this doesn't work if the stack is
173           non-executable, and it isn't consistent with the x86-linux and
174           amd64-linux scheme for removing the stack frame.  So instead be
175           consistent and use a stub in m_trampoline.  Then it doesn't
176           matter whether or not the (guest) stack is executable.  This
177           fixes #149519 and #145837. */
178        VG_TRACK(pre_mem_write, Vg_CoreSignal, tst->tid, "signal frame mcontext",
179                 (Addr)&mc->mc_pad, sizeof(mc->mc_pad));
180        mc->mc_pad[0] = 0; /* invalid */
181        mc->mc_pad[1] = 0; /* invalid */
182        VG_TRACK( post_mem_write,  Vg_CoreSignal, tst->tid, 
183                  (Addr)&mc->mc_pad, sizeof(mc->mc_pad) );
184        /* invalidate any translation of this area */
185        VG_(discard_translations)( (Addr64)(Addr)&mc->mc_pad, 
186                                   sizeof(mc->mc_pad), "stack_mcontext" );   
187     
188        /* set the signal handler to return to the trampoline */
189        SET_SIGNAL_LR(tst, (Addr)(use_rt_sigreturn 
190                                    ? (Addr)&VG_(ppc32_linux_SUBST_FOR_rt_sigreturn)
191                                    : (Addr)&VG_(ppc32_linux_SUBST_FOR_sigreturn)
192                           ));
193     }

275     //:: static inline unsigned short twd_i387_to_fxsr( unsigned short twd )
276     //:: {
277     //::    unsigned int tmp; /* to avoid 16 bit prefixes in the code */
278     //::  
279     //::    /* Transform each pair of bits into 01 (valid) or 00 (empty) */
280     //::    tmp = ~twd;

282     //::    /* and move the valid bits to the lower byte. */
283     //::    tmp = (tmp | (tmp >> 1)) & 0x3333; /* 00VV00VV00VV00VV */
284     //::    tmp = (tmp | (tmp >> 2)) & 0x0f0f; /* 0000VVVV0000VVVV */

504     /* Extend the stack segment downwards if needed so as to ensure the
505        new signal frames are mapped to something.  Return a Bool
506        indicating whether or not the operation was successful.
507     */
508     static Bool extend ( ThreadState *tst, Addr addr, SizeT size )
509     {
510        ThreadId        tid = tst->tid;
511        NSegment const* stackseg = NULL;
512     
513        if (VG_(extend_stack)(addr, tst->client_stack_szB)) {
514           stackseg = VG_(am_find_nsegment)(addr);
515           if (0 && stackseg)
516     	 VG_(printf)("frame=%#lx seg=%#lx-%#lx\n",
517     		     addr, stackseg->start, stackseg->end);
518        }
519     
520        if (stackseg == NULL || !stackseg->hasR || !stackseg->hasW) {
521           VG_(message)(
522              Vg_UserMsg,
523              "Can't extend stack to %#lx during signal delivery for thread %d:\n",
524              addr, tid);
525           if (stackseg == NULL)
526              VG_(message)(Vg_UserMsg, "  no stack segment\n");
527           else
528              VG_(message)(Vg_UserMsg, "  too small or bad protection modes\n");
529     
530           /* set SIGSEGV to default handler */
531           VG_(set_default_handler)(VKI_SIGSEGV);
532           VG_(synth_fault_mapping)(tid, addr);
533     
534           /* The whole process should be about to die, since the default
535     	 action of SIGSEGV to kill the whole process. */
536           return False;
537        }
538     
539        /* For tracking memory events, indicate the entire frame has been
540           allocated. */

542                  size + VG_STACK_REDZONE_SZB, tid );

544        return True;

667     /* EXPORTED */
668     void VG_(sigframe_create)( ThreadId tid, 
669                                Addr sp_top_of_frame,
670                                const vki_siginfo_t *siginfo,
671                                const struct vki_ucontext *siguc,
672                                void *handler, 
673                                UInt flags,
674                                const vki_sigset_t *mask,
675     		           void *restorer )
676     {
677        struct vg_sig_private *priv;
678        Addr sp;
679        ThreadState *tst;
680        Int sigNo = siginfo->si_signo;
681        Addr faultaddr;
682     
683        /* Stack must be 16-byte aligned */
684        sp_top_of_frame &= ~0xf;
685     
686        if (flags & VKI_SA_SIGINFO) {
687           sp = sp_top_of_frame - sizeof(struct rt_sigframe);
688        } else {
689           sp = sp_top_of_frame - sizeof(struct nonrt_sigframe);
690        }
691     
692        tst = VG_(get_ThreadState)(tid);
693     
694        if (!extend(tst, sp, sp_top_of_frame - sp))
695           return;
696     
697        vg_assert(VG_IS_16_ALIGNED(sp));
698     
699        /* Set up the stack chain pointer */
700        VG_TRACK( pre_mem_write, Vg_CoreSignal, tid, "signal handler frame",
701                  sp, sizeof(UWord) );
702        *(Addr *)sp = tst->arch.vex.guest_GPR1;
703        VG_TRACK( post_mem_write, Vg_CoreSignal, tid, 
704                  sp, sizeof(UWord) );
705     
706        faultaddr = (Addr)siginfo->_sifields._sigfault._addr;
707        if (sigNo == VKI_SIGILL && siginfo->si_code > 0)
708           faultaddr = tst->arch.vex.guest_CIA;
709     
710        if (flags & VKI_SA_SIGINFO) {
711           struct rt_sigframe *frame = (struct rt_sigframe *) sp;
712           struct vki_ucontext *ucp = &frame->ucontext;
713     
714           VG_TRACK( pre_mem_write, Vg_CoreSignal, tid, "signal frame siginfo",
715                     (Addr)&frame->siginfo, sizeof(frame->siginfo) );
716           VG_(memcpy)(&frame->siginfo, siginfo, sizeof(*siginfo));
717           VG_TRACK( post_mem_write, Vg_CoreSignal, tid, 
718                     (Addr)&frame->siginfo, sizeof(frame->siginfo) );
719     
720           VG_TRACK( pre_mem_write, Vg_CoreSignal, tid, "signal frame ucontext",
721                     (Addr)ucp, offsetof(struct vki_ucontext, uc_pad) );
722           ucp->uc_flags = 0;
723           ucp->uc_link = 0;
724           ucp->uc_stack = tst->altstack;
725           VG_TRACK( post_mem_write, Vg_CoreSignal, tid, (Addr)ucp,
726                     offsetof(struct vki_ucontext, uc_pad) );
727     
728           VG_TRACK( pre_mem_write, Vg_CoreSignal, tid, "signal frame ucontext",
729                     (Addr)&ucp->uc_regs,
730                     sizeof(ucp->uc_regs) + sizeof(ucp->uc_sigmask) );
731           ucp->uc_regs = &ucp->uc_mcontext;
732           ucp->uc_sigmask = tst->sig_mask;
733           VG_TRACK( post_mem_write, Vg_CoreSignal, tid, 
734                     (Addr)&ucp->uc_regs,
735                     sizeof(ucp->uc_regs) + sizeof(ucp->uc_sigmask) );
736     
737           stack_mcontext(&ucp->uc_mcontext, tst, True/*use_rt_sigreturn*/, faultaddr);
738           priv = &frame->priv;
739     
740           SET_SIGNAL_GPR(tid, 4, (Addr) &frame->siginfo);
741           SET_SIGNAL_GPR(tid, 5, (Addr) ucp);
742           /* the kernel sets this, though it doesn't seem to be in the ABI */
743           SET_SIGNAL_GPR(tid, 6, (Addr) &frame->siginfo);
744     
745        } else {
746           /* non-RT signal delivery */
747           struct nonrt_sigframe *frame = (struct nonrt_sigframe *) sp;
748           struct vki_sigcontext *scp = &frame->sigcontext;
749     
750           VG_TRACK( pre_mem_write, Vg_CoreSignal, tid, "signal frame sigcontext",
751                     (Addr)&scp->_unused[3], sizeof(*scp) - 3 * sizeof(UInt) );
752           scp->signal = sigNo;
753           scp->handler = (Addr) handler;
754           scp->oldmask = tst->sig_mask.sig[0];
755           scp->_unused[3] = tst->sig_mask.sig[1];
756           VG_TRACK( post_mem_write, Vg_CoreSignal, tid,
757                     (Addr)&scp->_unused[3], sizeof(*scp) - 3 * sizeof(UInt) );
758     
759           stack_mcontext(&frame->mcontext, tst, False/*!use_rt_sigreturn*/, faultaddr);
760           priv = &frame->priv;
761     
762           SET_SIGNAL_GPR(tid, 4, (Addr) scp);
763        }
764     
765        priv->magicPI       = 0x31415927;
766        priv->sigNo_private = sigNo;
767        priv->vex_shadow1   = tst->arch.vex_shadow1;
768        priv->vex_shadow2   = tst->arch.vex_shadow2;
769     
770        SET_SIGNAL_GPR(tid, 1, sp);
771        SET_SIGNAL_GPR(tid, 3, sigNo);
772        tst->arch.vex.guest_CIA = (Addr) handler;
773     
774     //..    Addr		esp;
775     //..    ThreadState* tst = VG_(get_ThreadState)(tid);
776     //.. 
777     //..    if (flags & VKI_SA_SIGINFO)
778     //..       esp = build_rt_sigframe(tst, esp_top_of_frame, siginfo, 
779     //..                                    handler, flags, mask, restorer);
780     //..    else
781     //..       esp = build_sigframe(tst, esp_top_of_frame, 
782     //..                                 siginfo, handler, flags, mask, restorer);
783     //.. 
784     //..    /* Set the thread so it will next run the handler. */
785     //..    /* tst->m_esp  = esp; */
786     //..    SET_SIGNAL_ESP(tid, esp);
787     //.. 
788     //..    //VG_(printf)("handler = %p\n", handler);
789     //..    tst->arch.vex.guest_CIA = (Addr) handler;
790     //..    /* This thread needs to be marked runnable, but we leave that the
791     //..       caller to do. */
792     
793        if (0)
794           VG_(printf)("pushed signal frame; %%R1 now = %#lx, "
795                       "next %%CIA = %#x, status=%d\n",
796     		  sp, tst->arch.vex.guest_CIA, tst->status);
797     }

876     /* EXPORTED */
877     void VG_(sigframe_destroy)( ThreadId tid, Bool isRT )
878     {
879        ThreadState *tst;
880        struct vg_sig_private *priv;
881        Addr sp;
882        UInt frame_size;
883        struct vki_mcontext *mc;
884        Int sigNo;
885        Bool has_siginfo = isRT;
886     
887        vg_assert(VG_(is_valid_tid)(tid));
888        tst = VG_(get_ThreadState)(tid);
889     
890        /* Check that the stack frame looks valid */
891        sp = tst->arch.vex.guest_GPR1;
892        vg_assert(VG_IS_16_ALIGNED(sp));
893        /* JRS 17 Nov 05: This code used to check that *sp -- which should
894           have been set by the stwu at the start of the handler -- points
895           to just above the frame (ie, the previous frame).  However, that
896           isn't valid when delivering signals on alt stacks.  So I removed
897           it.  The frame is still sanity-checked using the priv->magicPI
898           field. */
899     
900        if (has_siginfo) {
901           struct rt_sigframe *frame = (struct rt_sigframe *)sp;
902           frame_size = sizeof(*frame);
903           mc = &frame->ucontext.uc_mcontext;
904           priv = &frame->priv;
905           vg_assert(priv->magicPI == 0x31415927);
906           tst->sig_mask = frame->ucontext.uc_sigmask;
907        } else {
908           struct nonrt_sigframe *frame = (struct nonrt_sigframe *)sp;
909           frame_size = sizeof(*frame);
910           mc = &frame->mcontext;
911           priv = &frame->priv;
912           vg_assert(priv->magicPI == 0x31415927);
913           tst->sig_mask.sig[0] = frame->sigcontext.oldmask;
914           tst->sig_mask.sig[1] = frame->sigcontext._unused[3];
915        }
916        tst->tmp_sig_mask = tst->sig_mask;
917     
918        sigNo = priv->sigNo_private;
919     
920     #  define DO(gpr)  tst->arch.vex.guest_GPR##gpr = mc->mc_gregs[VKI_PT_R0+gpr]
921        DO(0);  DO(1);  DO(2);  DO(3);  DO(4);  DO(5);  DO(6);  DO(7);
922        DO(8);  DO(9);  DO(10); DO(11); DO(12); DO(13); DO(14); DO(15);
923        DO(16); DO(17); DO(18); DO(19); DO(20); DO(21); DO(22); DO(23);
924        DO(24); DO(25); DO(26); DO(27); DO(28); DO(29); DO(30); DO(31);
925     #  undef DO
926     
927        tst->arch.vex.guest_CIA = mc->mc_gregs[VKI_PT_NIP];
928     
929        // Umm ... ? (jrs 2005 July 8)
930        // tst->arch.m_orig_gpr3 = mc->mc_gregs[VKI_PT_ORIG_R3];
931     
932        LibVEX_GuestPPC32_put_CR( mc->mc_gregs[VKI_PT_CCR], &tst->arch.vex );
933     
934        tst->arch.vex.guest_LR  = mc->mc_gregs[VKI_PT_LNK];
935        tst->arch.vex.guest_CTR = mc->mc_gregs[VKI_PT_CTR];
936        LibVEX_GuestPPC32_put_XER( mc->mc_gregs[VKI_PT_XER], &tst->arch.vex );
937     
938        tst->arch.vex_shadow1 = priv->vex_shadow1;
939        tst->arch.vex_shadow2 = priv->vex_shadow2;
940     
941        VG_TRACK(die_mem_stack_signal, sp, frame_size);
942     
943        if (VG_(clo_trace_signals))
944           VG_(message)(Vg_DebugMsg,
945                        "vg_pop_signal_frame (thread %d): "
946                        "isRT=%d valid magic; EIP=%#x\n",
947                        tid, has_siginfo, tst->arch.vex.guest_CIA);
948     
949        /* tell the tools */
950        VG_TRACK( post_deliver_signal, tid, sigNo );
951     
952     //..    Addr          esp;
953     //..    ThreadState*  tst;
954     //..    SizeT	 size;
955     //..    Int		 sigNo;
956     //.. 
957     //..    tst = VG_(get_ThreadState)(tid);
958     //.. 
959     //..    /* Correctly reestablish the frame base address. */
960     //..    esp   = tst->arch.vex.guest_ESP;
961     //.. 
962     //..    if (!isRT)
963     //..       size = restore_sigframe(tst, (struct sigframe *)esp, &sigNo);
964     //..    else
965     //..       size = restore_rt_sigframe(tst, (struct rt_sigframe *)esp, &sigNo);
966     //.. 
967     //..    VG_TRACK( die_mem_stack_signal, esp, size );
968     //.. 
969     //..    if (VG_(clo_trace_signals))
970     //..       VG_(message)(
971     //..          Vg_DebugMsg, 
972     //..          "VG_(signal_return) (thread %d): isRT=%d valid magic; EIP=%p", 
973     //..          tid, isRT, tst->arch.vex.guest_EIP);
974     //.. 
975     //..    /* tell the tools */
976     //..    VG_TRACK( post_deliver_signal, tid, sigNo );
977     }
