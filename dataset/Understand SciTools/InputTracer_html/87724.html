45     void* HG_(zalloc) ( HChar* cc, SizeT n )
46     {
47        void* p;
48        tl_assert(n > 0);
49        p = VG_(malloc)( cc, n );
50        tl_assert(p);
51        VG_(memset)(p, 0, n);
52        return p;
53     }
54     
55     void HG_(free) ( void* p )
56     {
57        tl_assert(p);
58        VG_(free)(p);
59     }
60     
61     Char* HG_(strdup) ( HChar* cc, const Char* s )
62     {
63        return VG_(strdup)( cc, s );
64     }
