39     /* Taint sink types */
40     typedef enum { 
41        InvalidSink=0, UncondBranchSink, CondBranchSink, SyscallSink, MemWriteSink 
42     } IT_SinkType;  
43     
44     /* Type for a node in the linked list of stack traces */
45     typedef struct _StackTraceNode StackTraceNode;
46     struct _StackTraceNode {
47        StackTrace      st;
48        UWord           n_ips;
49        UWord           seq_nr;
50        StackTraceNode* next;
51     };
52     
53     /* 'IT_AuxInputVectors' is used to hold up to IT_MAX_AUX_INPV extra
54        input vectors */ 
55     #define IT_MAX_AUX_INPV 6
56     typedef struct {
57        UInt       num_vectors;
58        LabelSet*  vectors[IT_MAX_AUX_INPV];
59     } IT_AuxInputVectors;
60     
61     /* Holds information about a taint sink event */
62     typedef struct {
63        IT_SinkType     type;
64        UInt            seq_nr;
65     
66        /* Unique thread-id for the entire program lifetime. 
67           The first created thread gets tid=0, the next tid=1, and so on.
68           Note: This is NOT the same as the tid used by Valgrind */
69        UInt            thread_nr;
70        
71        StackTraceNode* stack_trace;
72        Addr            location_1;
73        Addr            location_2;       /* Only used for conditional branches */
74        UInt            call_specifier;   /* Only used for syscalls (holds syscall nr) */
75        LabelSet*       input_vector;
76        
77        IT_AuxInputVectors* aux_inv;  /* May be used for syscalls */
78     } IT_TaintSinkEvent;
79     
80     /* Node in a linked list of taint sink events */
81     typedef struct _IT_EventListNode IT_EventListNode;
82     struct _IT_EventListNode {
83        IT_TaintSinkEvent* data;
84        IT_EventListNode*  next;
85     };
