320     /* Start debugger and get it to attach to this process.  Called if the
321        user requests this service after an error has been shown, so she can
322        poke around and look at parameters, memory, etc.  You can't
323        meaningfully get the debugger to continue the program, though; to
324        continue, quit the debugger.  */
325     void VG_(start_debugger) ( ThreadId tid )
326     {
327     #  define N_BUF 4096
328        Int pid, rc;
329     
330        pid = VG_(fork)();
331     
332        if (pid == 0) {
333           /* child */
334           rc = VG_(ptrace)(VKI_PTRACE_TRACEME, 0, NULL, NULL);
335           vg_assert(rc == 0);
336           rc = VG_(kill)(VG_(getpid)(), VKI_SIGSTOP);
337           vg_assert(rc == 0);
338     
339        } else if (pid > 0) {
340           /* parent */
341           Int status;
342           Int res;
343     
344           if ((res = VG_(waitpid)(pid, &status, 0)) == pid &&
345               WIFSTOPPED(status) && WSTOPSIG(status) == VKI_SIGSTOP &&
346               ptrace_setregs(pid, &(VG_(threads)[tid].arch.vex)) == 0 &&
347               VG_(kill)(pid, VKI_SIGSTOP) == 0 &&
348               VG_(ptrace)(VKI_PTRACE_DETACH, pid, NULL, 0) == 0)
349           {
350              Char pidbuf[15];
351              Char file[50];
352              Char buf[N_BUF];
353              Char *bufptr;
354              Char *cmdptr;
355              
356              VG_(sprintf)(pidbuf, "%d", pid);
357              VG_(sprintf)(file, "/proc/%d/fd/%d", pid, VG_(cl_exec_fd));
358      
359              bufptr = buf;
360              cmdptr = VG_(clo_db_command);
361              
362              while (*cmdptr) {
363                 /* each iteration can advance bufptr by at most the length
364                    of file[], so the following assertion is generously
365                    over-paranoid. */
366                 vg_assert(bufptr - buf < N_BUF-15-50-10/*paranoia*/);
367                 switch (*cmdptr) {
368                    case '%':
369                       switch (*++cmdptr) {
370                          case 'f':
371                             VG_(memcpy)(bufptr, file, VG_(strlen)(file));
372                             bufptr += VG_(strlen)(file);
373                             cmdptr++;
374                             break;
375                          case 'p':
376                             VG_(memcpy)(bufptr, pidbuf, VG_(strlen)(pidbuf));
377                             bufptr += VG_(strlen)(pidbuf);
378                             cmdptr++;
379                             break;
380                          default:
381                             *bufptr++ = *cmdptr++;
382                             break;
383                       }
384                       break;
385                    default:
386                       *bufptr++ = *cmdptr++;
387                       break;
388                 }
389                 vg_assert(bufptr - buf < N_BUF-15-50-10/*paranoia*/);
390              }
391              
392              *bufptr++ = '\0';
393       
394              VG_(message)(Vg_UserMsg, "starting debugger with cmd: %s\n", buf);
395              res = VG_(system)(buf);
396              if (res == 0) {      
397                 VG_(message)(Vg_UserMsg, "\n");
398                 VG_(message)(Vg_UserMsg, 
399                              "Debugger has detached.  Valgrind regains control."
400                              "  We continue.\n");
401              } else {
402                 VG_(message)(Vg_UserMsg, 
403                              "Warning: Debugger attach failed! (sys_system)\n");
404                 VG_(message)(Vg_UserMsg, "\n");
405              }
406           } else {
407              VG_(message)(Vg_UserMsg, 
408                           "Warning: Debugger attach failed! (ptrace problem?)\n");
409              VG_(message)(Vg_UserMsg, "\n");
410           }
411     
412           VG_(kill)(pid, VKI_SIGKILL);
413           VG_(waitpid)(pid, &status, 0);
414        }
415     #  undef N_BUF
416     }
