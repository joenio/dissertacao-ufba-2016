79     static void mullS64 ( Long u, Long v, Long* rHi, Long* rLo )
80     {
81        ULong u0, v0, w0;
82         Long u1, v1, w1, w2, t;
83        u0   = u & 0xFFFFFFFFULL; 
84        u1   = u >> 32;
85        v0   = v & 0xFFFFFFFFULL;
86        v1   = v >> 32;
87        w0   = u0 * v0;
88        t    = u1 * v0 + (w0 >> 32);
89        w1   = t & 0xFFFFFFFFULL;
90        w2   = t >> 32;
91        w1   = u0 * v1 + w1;
92        *rHi = u1 * v1 + w2 + (w1 >> 32);
93        *rLo = u * v;
94     }

96     static void mullU64 ( ULong u, ULong v, ULong* rHi, ULong* rLo )
97     {
98        ULong u0, v0, w0;
99        ULong u1, v1, w1,w2,t;
100        u0   = u & 0xFFFFFFFFULL;
101        u1   = u >> 32;
102        v0   = v & 0xFFFFFFFFULL;
103        v1   = v >> 32;
104        w0   = u0 * v0;
105        t    = u1 * v0 + (w0 >> 32);
106        w1   = t & 0xFFFFFFFFULL;
107        w2   = t >> 32;
108        w1   = u0 * v1 + w1;
109        *rHi = u1 * v1 + w2 + (w1 >> 32);
110        *rLo = u * v;
111     }

165     #define PREAMBLE(__data_bits)					\
166        /* const */ ULong DATA_MASK 					\
167           = __data_bits==8                                          \
168                ? 0xFFULL 					        \
169                : (__data_bits==16                                   \
170                     ? 0xFFFFULL 		                        \
171                     : (__data_bits==32                              \
172                          ? 0xFFFFFFFFULL                            \
173                          : 0xFFFFFFFFFFFFFFFFULL));                 \
174        /* const */ ULong SIGN_MASK = 1ULL << (__data_bits - 1);     \
175        /* const */ ULong CC_DEP1 = cc_dep1_formal;			\
176        /* const */ ULong CC_DEP2 = cc_dep2_formal;			\
177        /* const */ ULong CC_NDEP = cc_ndep_formal;			\
178        /* Four bogus assignments, which hopefully gcc can     */	\
179        /* optimise away, and which stop it complaining about  */	\
180        /* unused variables.                                   */	\
181        SIGN_MASK = SIGN_MASK;					\
182        DATA_MASK = DATA_MASK;					\
183        CC_DEP2 = CC_DEP2;						\
184        CC_NDEP = CC_NDEP;

189     #define ACTIONS_ADD(DATA_BITS,DATA_UTYPE)			\
190     {								\
191        PREAMBLE(DATA_BITS);						\
192        { Long cf, pf, af, zf, sf, of;				\
193          Long argL, argR, res;					\
194          argL = CC_DEP1;						\
195          argR = CC_DEP2;						\
196          res  = argL + argR;					\
197          cf = (DATA_UTYPE)res < (DATA_UTYPE)argL;			\
198          pf = parity_table[(UChar)res];				\
199          af = (res ^ argL ^ argR) & 0x10;				\
200          zf = ((DATA_UTYPE)res == 0) << 6;				\
201          sf = lshift(res, 8 - DATA_BITS) & 0x80;			\
202          of = lshift((argL ^ argR ^ -1) & (argL ^ res), 		\
203                      12 - DATA_BITS) & AMD64G_CC_MASK_O;		\
204          return cf | pf | af | zf | sf | of;			\
205        }								\
206     }

465          mullU64( (ULong)CC_DEP1, (ULong)CC_DEP2, &hi, &lo );       \
466          cf = (hi != 0);                                            \
467          pf = parity_table[(UChar)lo];                              \
468          af = 0; /* undefined */                                    \
469          zf = (lo == 0) << 6;                                       \
470          sf = lshift(lo, 8 - 64) & 0x80;                            \
471          of = cf << 11;                                             \
472          return cf | pf | af | zf | sf | of;                        \
473        }								\
474     }

482          Long lo, hi;                                               \
483          mullS64( (Long)CC_DEP1, (Long)CC_DEP2, &hi, &lo );         \
484          cf = (hi != (lo >>/*s*/ (64-1)));                          \

486          af = 0; /* undefined */                                    \
487          zf = (lo == 0) << 6;                                       \
488          sf = lshift(lo, 8 - 64) & 0x80;                            \
489          of = cf << 11;                                             \
490          return cf | pf | af | zf | sf | of;                        \
491        }								\
492     }

733     /* CALLED FROM GENERATED CODE: CLEAN HELPER */
734     /* returns 1 or 0 */
735     ULong amd64g_calculate_condition ( ULong/*AMD64Condcode*/ cond, 
736                                        ULong cc_op, 
737                                        ULong cc_dep1, 
738                                        ULong cc_dep2,
739                                        ULong cc_ndep )
740     {
741        ULong rflags = amd64g_calculate_rflags_all_WRK(cc_op, cc_dep1, 
742                                                       cc_dep2, cc_ndep);
743        ULong of,sf,zf,cf,pf;
744        ULong inv = cond & 1;
745     
746     #  if PROFILE_RFLAGS
747        if (!initted) initCounts();
748        tab_cond[cc_op][cond]++;
749        n_calc_cond++;
750        if (SHOW_COUNTS_NOW) showCounts();
751     #  endif
752     
753        switch (cond) {
754           case AMD64CondNO:
755           case AMD64CondO: /* OF == 1 */
756              of = rflags >> AMD64G_CC_SHIFT_O;
757              return 1 & (inv ^ of);
758     
759           case AMD64CondNZ:
760           case AMD64CondZ: /* ZF == 1 */
761              zf = rflags >> AMD64G_CC_SHIFT_Z;
762              return 1 & (inv ^ zf);
763     
764           case AMD64CondNB:
765           case AMD64CondB: /* CF == 1 */
766              cf = rflags >> AMD64G_CC_SHIFT_C;
767              return 1 & (inv ^ cf);
768              break;
769     
770           case AMD64CondNBE:
771           case AMD64CondBE: /* (CF or ZF) == 1 */
772              cf = rflags >> AMD64G_CC_SHIFT_C;
773              zf = rflags >> AMD64G_CC_SHIFT_Z;
774              return 1 & (inv ^ (cf | zf));
775              break;
776     
777           case AMD64CondNS:
778           case AMD64CondS: /* SF == 1 */
779              sf = rflags >> AMD64G_CC_SHIFT_S;
780              return 1 & (inv ^ sf);
781     
782           case AMD64CondNP:
783           case AMD64CondP: /* PF == 1 */
784              pf = rflags >> AMD64G_CC_SHIFT_P;
785              return 1 & (inv ^ pf);
786     
787           case AMD64CondNL:
788           case AMD64CondL: /* (SF xor OF) == 1 */
789              sf = rflags >> AMD64G_CC_SHIFT_S;
790              of = rflags >> AMD64G_CC_SHIFT_O;
791              return 1 & (inv ^ (sf ^ of));
792              break;
793     
794           case AMD64CondNLE:
795           case AMD64CondLE: /* ((SF xor OF) or ZF)  == 1 */
796              sf = rflags >> AMD64G_CC_SHIFT_S;
797              of = rflags >> AMD64G_CC_SHIFT_O;
798              zf = rflags >> AMD64G_CC_SHIFT_Z;
799              return 1 & (inv ^ ((sf ^ of) | zf));
800              break;
801     
802           default:
803              /* shouldn't really make these calls from generated code */
804              vex_printf("amd64g_calculate_condition"
805                         "( %llu, %llu, 0x%llx, 0x%llx, 0x%llx )\n",
806                         cond, cc_op, cc_dep1, cc_dep2, cc_ndep );
807              vpanic("amd64g_calculate_condition");
808        }
809     }

872     IRExpr* guest_amd64_spechelper ( HChar* function_name,
873                                      IRExpr** args,
874                                      IRStmt** precedingStmts,
875                                      Int      n_precedingStmts )
876     {
877     #  define unop(_op,_a1) IRExpr_Unop((_op),(_a1))
878     #  define binop(_op,_a1,_a2) IRExpr_Binop((_op),(_a1),(_a2))
879     #  define mkU64(_n) IRExpr_Const(IRConst_U64(_n))
880     #  define mkU8(_n)  IRExpr_Const(IRConst_U8(_n))
881     
882        Int i, arity = 0;
883        for (i = 0; args[i]; i++)
884           arity++;
885     #  if 0
886        vex_printf("spec request:\n");
887        vex_printf("   %s  ", function_name);
888        for (i = 0; i < arity; i++) {
889           vex_printf("  ");
890           ppIRExpr(args[i]);
891        }
892        vex_printf("\n");
893     #  endif
894     
895        /* --------- specialising "amd64g_calculate_condition" --------- */
896     
897        if (vex_streq(function_name, "amd64g_calculate_condition")) {
898           /* specialise calls to above "calculate condition" function */
899           IRExpr *cond, *cc_op, *cc_dep1, *cc_dep2;
900           vassert(arity == 5);
901           cond    = args[0];
902           cc_op   = args[1];
903           cc_dep1 = args[2];
904           cc_dep2 = args[3];
905     
906           /*---------------- ADDQ ----------------*/
907     
908           if (isU64(cc_op, AMD64G_CC_OP_ADDQ) && isU64(cond, AMD64CondZ)) {
909              /* long long add, then Z --> test (dst+src == 0) */
910              return unop(Iop_1Uto64,
911                          binop(Iop_CmpEQ64, 
912                                binop(Iop_Add64, cc_dep1, cc_dep2),
913                                mkU64(0)));
914           }
915     
916           /*---------------- SUBQ ----------------*/
917     
918           if (isU64(cc_op, AMD64G_CC_OP_SUBQ) && isU64(cond, AMD64CondZ)) {
919              /* long long sub/cmp, then Z --> test dst==src */
920              return unop(Iop_1Uto64,
921                          binop(Iop_CmpEQ64,cc_dep1,cc_dep2));
922           }
923           if (isU64(cc_op, AMD64G_CC_OP_SUBQ) && isU64(cond, AMD64CondNZ)) {
924              /* long long sub/cmp, then NZ --> test dst!=src */
925              return unop(Iop_1Uto64,
926                          binop(Iop_CmpNE64,cc_dep1,cc_dep2));
927           }
928     
929           if (isU64(cc_op, AMD64G_CC_OP_SUBQ) && isU64(cond, AMD64CondL)) {
930              /* long long sub/cmp, then L (signed less than) 
931                 --> test dst <s src */
932              return unop(Iop_1Uto64,
933                          binop(Iop_CmpLT64S, cc_dep1, cc_dep2));
934           }
935     
936           if (isU64(cc_op, AMD64G_CC_OP_SUBQ) && isU64(cond, AMD64CondB)) {
937              /* long long sub/cmp, then B (unsigned less than)
938                 --> test dst <u src */
939              return unop(Iop_1Uto64,
940                          binop(Iop_CmpLT64U, cc_dep1, cc_dep2));
941           }
942           if (isU64(cc_op, AMD64G_CC_OP_SUBQ) && isU64(cond, AMD64CondNB)) {
943              /* long long sub/cmp, then NB (unsigned greater than or equal)
944                 --> test src <=u dst */
945              /* Note, args are opposite way round from the usual */
946              return unop(Iop_1Uto64,
947                          binop(Iop_CmpLE64U, cc_dep2, cc_dep1));
948           }
949     
950           if (isU64(cc_op, AMD64G_CC_OP_SUBQ) && isU64(cond, AMD64CondBE)) {
951              /* long long sub/cmp, then BE (unsigned less than or equal)
952                 --> test dst <=u src */
953              return unop(Iop_1Uto64,
954                          binop(Iop_CmpLE64U, cc_dep1, cc_dep2));
955           }
956     
957           /*---------------- SUBL ----------------*/
958     
959           if (isU64(cc_op, AMD64G_CC_OP_SUBL) && isU64(cond, AMD64CondZ)) {
960              /* long sub/cmp, then Z --> test dst==src */
961              return unop(Iop_1Uto64,
962                          binop(Iop_CmpEQ64, 
963                                binop(Iop_Shl64,cc_dep1,mkU8(32)),
964                                binop(Iop_Shl64,cc_dep2,mkU8(32))));
965           }
966           if (isU64(cc_op, AMD64G_CC_OP_SUBL) && isU64(cond, AMD64CondNZ)) {
967              /* long sub/cmp, then NZ --> test dst!=src */
968              return unop(Iop_1Uto64,
969                          binop(Iop_CmpNE64, 
970                                binop(Iop_Shl64,cc_dep1,mkU8(32)),
971                                binop(Iop_Shl64,cc_dep2,mkU8(32))));
972           }
973     
974           if (isU64(cc_op, AMD64G_CC_OP_SUBL) && isU64(cond, AMD64CondL)) {
975              /* long sub/cmp, then L (signed less than) 
976                 --> test dst <s src */
977              return unop(Iop_1Uto64,
978                          binop(Iop_CmpLT64S, 
979                                binop(Iop_Shl64,cc_dep1,mkU8(32)),
980                                binop(Iop_Shl64,cc_dep2,mkU8(32))));
981           }
982     
983           if (isU64(cc_op, AMD64G_CC_OP_SUBL) && isU64(cond, AMD64CondLE)) {
984              /* long sub/cmp, then LE (signed less than or equal) 
985                 --> test dst <=s src */
986              return unop(Iop_1Uto64,
987                          binop(Iop_CmpLE64S, 
988                                binop(Iop_Shl64,cc_dep1,mkU8(32)),
989                                binop(Iop_Shl64,cc_dep2,mkU8(32))));
990     
991           }
992           if (isU64(cc_op, AMD64G_CC_OP_SUBL) && isU64(cond, AMD64CondNLE)) {
993              /* long sub/cmp, then NLE (signed greater than) 
994                 --> test !(dst <=s src)
995                 --> test (dst >s src)
996                 --> test (src <s dst) */
997              return unop(Iop_1Uto64,
998                          binop(Iop_CmpLT64S,
999                                binop(Iop_Shl64,cc_dep2,mkU8(32)),
1000                                binop(Iop_Shl64,cc_dep1,mkU8(32))));
1001     
1002           }
1003     
1004           if (isU64(cc_op, AMD64G_CC_OP_SUBL) && isU64(cond, AMD64CondBE)) {
1005              /* long sub/cmp, then BE (unsigned less than or equal)
1006                 --> test dst <=u src */
1007              return unop(Iop_1Uto64,
1008                          binop(Iop_CmpLE64U, 
1009                                binop(Iop_Shl64,cc_dep1,mkU8(32)),
1010                                binop(Iop_Shl64,cc_dep2,mkU8(32))));
1011           }
1012           if (isU64(cc_op, AMD64G_CC_OP_SUBL) && isU64(cond, AMD64CondNBE)) {
1013              /* long sub/cmp, then NBE (unsigned greater than)
1014                 --> test src <u dst */
1015              /* Note, args are opposite way round from the usual */
1016              return unop(Iop_1Uto64,
1017                          binop(Iop_CmpLT64U, 
1018                                binop(Iop_Shl64,cc_dep2,mkU8(32)),
1019                                binop(Iop_Shl64,cc_dep1,mkU8(32))));
1020           }
1021     
1022           if (isU64(cc_op, AMD64G_CC_OP_SUBL) && isU64(cond, AMD64CondS)) {
1023              /* long sub/cmp, then S (negative) --> test (dst-src <s 0) */
1024              return unop(Iop_1Uto64,
1025                          binop(Iop_CmpLT64S,
1026                                binop(Iop_Sub64,
1027                                      binop(Iop_Shl64, cc_dep1, mkU8(32)), 
1028                                      binop(Iop_Shl64, cc_dep2, mkU8(32))),
1029                                mkU64(0)));
1030           }
1031     
1032           /*---------------- SUBW ----------------*/
1033     
1034           if (isU64(cc_op, AMD64G_CC_OP_SUBW) && isU64(cond, AMD64CondZ)) {
1035              /* word sub/cmp, then Z --> test dst==src */
1036              return unop(Iop_1Uto64,
1037                          binop(Iop_CmpEQ16, 
1038                                unop(Iop_64to16,cc_dep1),
1039                                unop(Iop_64to16,cc_dep2)));
1040           }
1041           if (isU64(cc_op, AMD64G_CC_OP_SUBW) && isU64(cond, AMD64CondNZ)) {
1042              /* word sub/cmp, then NZ --> test dst!=src */
1043              return unop(Iop_1Uto64,
1044                          binop(Iop_CmpNE16, 
1045                                unop(Iop_64to16,cc_dep1),
1046                                unop(Iop_64to16,cc_dep2)));
1047           }
1048     
1049           if (isU64(cc_op, AMD64G_CC_OP_SUBW) && isU64(cond, AMD64CondLE)) {
1050              /* word sub/cmp, then LE (signed less than or equal) 
1051                 --> test dst <=s src */
1052              return unop(Iop_1Uto64,
1053                          binop(Iop_CmpLE64S, 
1054                                binop(Iop_Shl64,cc_dep1,mkU8(48)),
1055                                binop(Iop_Shl64,cc_dep2,mkU8(48))));
1056     
1057           }
1058     
1059           /*---------------- SUBB ----------------*/
1060     
1061           if (isU64(cc_op, AMD64G_CC_OP_SUBB) && isU64(cond, AMD64CondZ)) {
1062              /* byte sub/cmp, then Z --> test dst==src */
1063              return unop(Iop_1Uto64,
1064                          binop(Iop_CmpEQ8, 
1065                                unop(Iop_64to8,cc_dep1),
1066                                unop(Iop_64to8,cc_dep2)));
1067           }
1068           if (isU64(cc_op, AMD64G_CC_OP_SUBB) && isU64(cond, AMD64CondNZ)) {
1069              /* byte sub/cmp, then NZ --> test dst!=src */
1070              return unop(Iop_1Uto64,
1071                          binop(Iop_CmpNE8, 
1072                                unop(Iop_64to8,cc_dep1),
1073                                unop(Iop_64to8,cc_dep2)));
1074           }
1075     
1076           if (isU64(cc_op, AMD64G_CC_OP_SUBB) && isU64(cond, AMD64CondS)
1077                                               && isU64(cc_dep2, 0)) {
1078              /* byte sub/cmp of zero, then S --> test (dst-0 <s 0)
1079                                              --> test dst <s 0
1080                                              --> (ULong)dst[7]
1081                 This is yet another scheme by which gcc figures out if the
1082                 top bit of a byte is 1 or 0.  See also LOGICB/CondS below. */
1083              /* Note: isU64(cc_dep2, 0) is correct, even though this is
1084                 for an 8-bit comparison, since the args to the helper
1085                 function are always U64s. */
1086              return binop(Iop_And64,
1087                           binop(Iop_Shr64,cc_dep1,mkU8(7)),
1088                           mkU64(1));
1089           }
1090           if (isU64(cc_op, AMD64G_CC_OP_SUBB) && isU64(cond, AMD64CondNS)
1091                                               && isU64(cc_dep2, 0)) {
1092              /* byte sub/cmp of zero, then NS --> test !(dst-0 <s 0)
1093                                               --> test !(dst <s 0)
1094                                               --> (ULong) !dst[7]
1095              */
1096              return binop(Iop_Xor64,
1097                           binop(Iop_And64,
1098                                 binop(Iop_Shr64,cc_dep1,mkU8(7)),
1099                                 mkU64(1)),
1100                           mkU64(1));
1101           }
1102     
1103           /*---------------- LOGICQ ----------------*/
1104     
1105           if (isU64(cc_op, AMD64G_CC_OP_LOGICQ) && isU64(cond, AMD64CondZ)) {
1106              /* long long and/or/xor, then Z --> test dst==0 */
1107              return unop(Iop_1Uto64,
1108                          binop(Iop_CmpEQ64, cc_dep1, mkU64(0)));
1109           }
1110     
1111           if (isU64(cc_op, AMD64G_CC_OP_LOGICQ) && isU64(cond, AMD64CondL)) {
1112              /* long long and/or/xor, then L
1113                 LOGIC sets SF and ZF according to the
1114                 result and makes OF be zero.  L computes SF ^ OF, but
1115                 OF is zero, so this reduces to SF -- which will be 1 iff
1116                 the result is < signed 0.  Hence ...
1117              */
1118              return unop(Iop_1Uto64,
1119                          binop(Iop_CmpLT64S, 
1120                                cc_dep1, 
1121                                mkU64(0)));
1122           }
1123     
1124           /*---------------- LOGICL ----------------*/
1125     
1126           if (isU64(cc_op, AMD64G_CC_OP_LOGICL) && isU64(cond, AMD64CondZ)) {
1127              /* long and/or/xor, then Z --> test dst==0 */
1128              return unop(Iop_1Uto64,
1129                          binop(Iop_CmpEQ64, 
1130                                binop(Iop_Shl64,cc_dep1,mkU8(32)), 
1131                                mkU64(0)));
1132           }
1133     
1134           if (isU64(cc_op, AMD64G_CC_OP_LOGICL) && isU64(cond, AMD64CondNZ)) {
1135              /* long and/or/xor, then NZ --> test dst!=0 */
1136              return unop(Iop_1Uto64,
1137                          binop(Iop_CmpNE64, 
1138                                binop(Iop_Shl64,cc_dep1,mkU8(32)), 
1139                                mkU64(0)));
1140           }
1141     
1142           if (isU64(cc_op, AMD64G_CC_OP_LOGICL) && isU64(cond, AMD64CondLE)) {
1143              /* long and/or/xor, then LE
1144                 This is pretty subtle.  LOGIC sets SF and ZF according to the
1145                 result and makes OF be zero.  LE computes (SF ^ OF) | ZF, but
1146                 OF is zero, so this reduces to SF | ZF -- which will be 1 iff
1147                 the result is <=signed 0.  Hence ...
1148              */
1149              return unop(Iop_1Uto64,
1150                          binop(Iop_CmpLE64S, 
1151                                binop(Iop_Shl64,cc_dep1,mkU8(32)), 
1152                                mkU64(0)));
1153           }
1154     
1155           /*---------------- LOGICB ----------------*/
1156     
1157           if (isU64(cc_op, AMD64G_CC_OP_LOGICB) && isU64(cond, AMD64CondZ)) {
1158              /* byte and/or/xor, then Z --> test dst==0 */
1159              return unop(Iop_1Uto64,
1160                          binop(Iop_CmpEQ64, binop(Iop_And64,cc_dep1,mkU64(255)), 
1161                                             mkU64(0)));
1162           }
1163           if (isU64(cc_op, AMD64G_CC_OP_LOGICB) && isU64(cond, AMD64CondNZ)) {
1164              /* byte and/or/xor, then NZ --> test dst!=0 */
1165              return unop(Iop_1Uto64,
1166                          binop(Iop_CmpNE64, binop(Iop_And64,cc_dep1,mkU64(255)), 
1167                                             mkU64(0)));
1168           }
1169     
1170           if (isU64(cc_op, AMD64G_CC_OP_LOGICB) && isU64(cond, AMD64CondS)) {
1171              /* this is an idiom gcc sometimes uses to find out if the top
1172                 bit of a byte register is set: eg testb %al,%al; js ..
1173                 Since it just depends on the top bit of the byte, extract
1174                 that bit and explicitly get rid of all the rest.  This
1175                 helps memcheck avoid false positives in the case where any
1176                 of the other bits in the byte are undefined. */
1177              /* byte and/or/xor, then S --> (UInt)result[7] */
1178              return binop(Iop_And64,
1179                           binop(Iop_Shr64,cc_dep1,mkU8(7)),
1180                           mkU64(1));
1181           }
1182           if (isU64(cc_op, AMD64G_CC_OP_LOGICB) && isU64(cond, AMD64CondNS)) {
1183              /* byte and/or/xor, then NS --> (UInt)!result[7] */
1184              return binop(Iop_Xor64,
1185                           binop(Iop_And64,
1186                                 binop(Iop_Shr64,cc_dep1,mkU8(7)),
1187                                 mkU64(1)),
1188                           mkU64(1));
1189           }
1190     
1191           /*---------------- INCB ----------------*/
1192     
1193           if (isU64(cc_op, AMD64G_CC_OP_INCB) && isU64(cond, AMD64CondLE)) {
1194              /* 8-bit inc, then LE --> sign bit of the arg */
1195              return binop(Iop_And64,
1196                           binop(Iop_Shr64,
1197                                 binop(Iop_Sub64, cc_dep1, mkU64(1)),
1198                                 mkU8(7)),
1199                           mkU64(1));
1200           }
1201     
1202           /*---------------- INCW ----------------*/
1203     
1204           if (isU64(cc_op, AMD64G_CC_OP_INCW) && isU64(cond, AMD64CondZ)) {
1205              /* 16-bit inc, then Z --> test dst == 0 */
1206              return unop(Iop_1Uto64,
1207                          binop(Iop_CmpEQ64, 
1208                                binop(Iop_Shl64,cc_dep1,mkU8(48)), 
1209                                mkU64(0)));
1210           }
1211     
1212           /*---------------- DECL ----------------*/
1213     
1214           if (isU64(cc_op, AMD64G_CC_OP_DECL) && isU64(cond, AMD64CondZ)) {
1215              /* dec L, then Z --> test dst == 0 */
1216              return unop(Iop_1Uto64,
1217                          binop(Iop_CmpEQ64,
1218                                binop(Iop_Shl64,cc_dep1,mkU8(32)),
1219                                mkU64(0)));
1220           }
1221     
1222           /*---------------- DECW ----------------*/
1223     
1224           if (isU64(cc_op, AMD64G_CC_OP_DECW) && isU64(cond, AMD64CondNZ)) {
1225              /* 16-bit dec, then NZ --> test dst != 0 */
1226              return unop(Iop_1Uto64,
1227                          binop(Iop_CmpNE64, 
1228                                binop(Iop_Shl64,cc_dep1,mkU8(48)), 
1229                                mkU64(0)));
1230           }
1231     
1232           /*---------------- COPY ----------------*/
1233           /* This can happen, as a result of amd64 FP compares: "comisd ... ;
1234              jbe" for example. */
1235     
1236           if (isU64(cc_op, AMD64G_CC_OP_COPY) && 
1237               (isU64(cond, AMD64CondBE) || isU64(cond, AMD64CondNBE))) {
1238              /* COPY, then BE --> extract C and Z from dep1, and test (C
1239                 or Z == 1). */
1240              /* COPY, then NBE --> extract C and Z from dep1, and test (C
1241                 or Z == 0). */
1242              ULong nnn = isU64(cond, AMD64CondBE) ? 1 : 0;
1243              return
1244                 unop(
1245                    Iop_1Uto64,
1246                    binop(
1247                       Iop_CmpEQ64,
1248                       binop(
1249                          Iop_And64,
1250                          binop(
1251                             Iop_Or64,
1252                             binop(Iop_Shr64, cc_dep1, mkU8(AMD64G_CC_SHIFT_C)),
1253                             binop(Iop_Shr64, cc_dep1, mkU8(AMD64G_CC_SHIFT_Z))
1254                          ),
1255                          mkU64(1)
1256                       ),
1257                       mkU64(nnn)
1258                    )
1259                 );
1260           }
1261           
1262           if (isU64(cc_op, AMD64G_CC_OP_COPY) && isU64(cond, AMD64CondB)) {
1263              /* COPY, then B --> extract C dep1, and test (C == 1). */
1264              return
1265                 unop(
1266                    Iop_1Uto64,
1267                    binop(
1268                       Iop_CmpNE64,
1269                       binop(
1270                          Iop_And64,
1271                          binop(Iop_Shr64, cc_dep1, mkU8(AMD64G_CC_SHIFT_C)),
1272                          mkU64(1)
1273                       ),
1274                       mkU64(0)
1275                    )
1276                 );
1277           }
1278     
1279           if (isU64(cc_op, AMD64G_CC_OP_COPY) 
1280               && (isU64(cond, AMD64CondZ) || isU64(cond, AMD64CondNZ))) {
1281              /* COPY, then Z --> extract Z from dep1, and test (Z == 1). */
1282              /* COPY, then NZ --> extract Z from dep1, and test (Z == 0). */
1283              UInt nnn = isU64(cond, AMD64CondZ) ? 1 : 0;
1284              return
1285                 unop(
1286                    Iop_1Uto64,
1287                    binop(
1288                       Iop_CmpEQ64,
1289                       binop(
1290                          Iop_And64,
1291                          binop(Iop_Shr64, cc_dep1, mkU8(AMD64G_CC_SHIFT_Z)),
1292                          mkU64(1)
1293                       ),
1294                       mkU64(nnn)
1295                    )
1296                 );
1297           }
1298     
1299           if (isU64(cc_op, AMD64G_CC_OP_COPY) && isU64(cond, AMD64CondP)) {
1300              /* COPY, then P --> extract P from dep1, and test (P == 1). */
1301              return
1302                 unop(
1303                    Iop_1Uto64,
1304                    binop(
1305                       Iop_CmpNE64,
1306                       binop(
1307                          Iop_And64,
1308                          binop(Iop_Shr64, cc_dep1, mkU8(AMD64G_CC_SHIFT_P)),
1309                          mkU64(1)
1310                       ),
1311                       mkU64(0)
1312                    )
1313                 );
1314           }
1315     
1316           return NULL;
1317        }
1318     
1319        /* --------- specialising "amd64g_calculate_rflags_c" --------- */
1320     
1321        if (vex_streq(function_name, "amd64g_calculate_rflags_c")) {
1322           /* specialise calls to above "calculate_rflags_c" function */
1323           IRExpr *cc_op, *cc_dep1, *cc_dep2, *cc_ndep;
1324           vassert(arity == 4);
1325           cc_op   = args[0];
1326           cc_dep1 = args[1];
1327           cc_dep2 = args[2];
1328           cc_ndep = args[3];
1329     
1330           if (isU64(cc_op, AMD64G_CC_OP_SUBQ)) {
1331              /* C after sub denotes unsigned less than */
1332              return unop(Iop_1Uto64,
1333                          binop(Iop_CmpLT64U, 
1334                                cc_dep1,
1335                                cc_dep2));
1336           }
1337           if (isU64(cc_op, AMD64G_CC_OP_SUBL)) {
1338              /* C after sub denotes unsigned less than */
1339              return unop(Iop_1Uto64,
1340                          binop(Iop_CmpLT64U, 
1341                                binop(Iop_Shl64,cc_dep1,mkU8(32)), 
1342                                binop(Iop_Shl64,cc_dep2,mkU8(32))));
1343           }
1344           if (isU64(cc_op, AMD64G_CC_OP_SUBB)) {
1345              /* C after sub denotes unsigned less than */
1346              return unop(Iop_1Uto64,
1347                          binop(Iop_CmpLT64U, 
1348                                binop(Iop_And64,cc_dep1,mkU64(0xFF)),
1349                                binop(Iop_And64,cc_dep2,mkU64(0xFF))));
1350           }
1351           if (isU64(cc_op, AMD64G_CC_OP_LOGICQ)
1352               || isU64(cc_op, AMD64G_CC_OP_LOGICL)
1353               || isU64(cc_op, AMD64G_CC_OP_LOGICW)
1354               || isU64(cc_op, AMD64G_CC_OP_LOGICB)) {
1355              /* cflag after logic is zero */
1356              return mkU64(0);
1357           }
1358           if (isU64(cc_op, AMD64G_CC_OP_DECL) || isU64(cc_op, AMD64G_CC_OP_INCL)
1359               || isU64(cc_op, AMD64G_CC_OP_DECQ) || isU64(cc_op, AMD64G_CC_OP_INCQ)) {
1360              /* If the thunk is dec or inc, the cflag is supplied as CC_NDEP. */
1361              return cc_ndep;
1362           }
1363     
1364     #     if 0
1365           if (cc_op->tag == Iex_Const) {
1366              vex_printf("CFLAG "); ppIRExpr(cc_op); vex_printf("\n");
1367           }
1368     #     endif
1369     
1370           return NULL;
1371        }
1372     
1373     #  undef unop
1374     #  undef binop
1375     #  undef mkU64
1376     #  undef mkU8
1377     
1378        return NULL;
1379     }

1393     /* Inspect a value and its tag, as per the x87 'FXAM' instruction. */
1394     /* CALLED FROM GENERATED CODE: CLEAN HELPER */
1395     ULong amd64g_calculate_FXAM ( ULong tag, ULong dbl ) 
1396     {
1397        Bool   mantissaIsZero;
1398        Int    bexp;
1399        UChar  sign;
1400        UChar* f64;
1401     
1402        vassert(host_is_little_endian());
1403     
1404        /* vex_printf("calculate_FXAM ( %d, %llx ) .. ", tag, dbl ); */
1405     
1406        f64  = (UChar*)(&dbl);
1407        sign = toUChar( (f64[7] >> 7) & 1 );
1408     
1409        /* First off, if the tag indicates the register was empty,
1410           return 1,0,sign,1 */
1411        if (tag == 0) {
1412           /* vex_printf("Empty\n"); */
1413           return AMD64G_FC_MASK_C3 | 0 | (sign << AMD64G_FC_SHIFT_C1) 
1414                                        | AMD64G_FC_MASK_C0;
1415        }
1416     
1417        bexp = (f64[7] << 4) | ((f64[6] >> 4) & 0x0F);
1418        bexp &= 0x7FF;
1419     
1420        mantissaIsZero
1421           = toBool(
1422                (f64[6] & 0x0F) == 0 
1423                && (f64[5] | f64[4] | f64[3] | f64[2] | f64[1] | f64[0]) == 0
1424             );
1425     
1426        /* If both exponent and mantissa are zero, the value is zero.
1427           Return 1,0,sign,0. */
1428        if (bexp == 0 && mantissaIsZero) {
1429           /* vex_printf("Zero\n"); */
1430           return AMD64G_FC_MASK_C3 | 0 
1431                                    | (sign << AMD64G_FC_SHIFT_C1) | 0;
1432        }
1433        
1434        /* If exponent is zero but mantissa isn't, it's a denormal.
1435           Return 1,1,sign,0. */
1436        if (bexp == 0 && !mantissaIsZero) {
1437           /* vex_printf("Denormal\n"); */
1438           return AMD64G_FC_MASK_C3 | AMD64G_FC_MASK_C2 
1439                                    | (sign << AMD64G_FC_SHIFT_C1) | 0;
1440        }
1441     
1442        /* If the exponent is 7FF and the mantissa is zero, this is an infinity.
1443           Return 0,1,sign,1. */
1444        if (bexp == 0x7FF && mantissaIsZero) {
1445           /* vex_printf("Inf\n"); */
1446           return 0 | AMD64G_FC_MASK_C2 | (sign << AMD64G_FC_SHIFT_C1) 
1447                                        | AMD64G_FC_MASK_C0;
1448        }
1449     
1450        /* If the exponent is 7FF and the mantissa isn't zero, this is a NaN.
1451           Return 0,0,sign,1. */
1452        if (bexp == 0x7FF && !mantissaIsZero) {
1453           /* vex_printf("NaN\n"); */
1454           return 0 | 0 | (sign << AMD64G_FC_SHIFT_C1) | AMD64G_FC_MASK_C0;
1455        }
1456     
1457        /* Uh, ok, we give up.  It must be a normal finite number.
1458           Return 0,1,sign,0.
1459        */
1460        /* vex_printf("normal\n"); */
1461        return 0 | AMD64G_FC_MASK_C2 | (sign << AMD64G_FC_SHIFT_C1) | 0;
1462     }

1465     /* This is used to implement both 'frstor' and 'fldenv'.  The latter
1466        appears to differ from the former only in that the 8 FP registers
1467        themselves are not transferred into the guest state. */
1468     static
1469     VexEmWarn do_put_x87 ( Bool moveRegs,
1470                            /*IN*/UChar* x87_state,
1471                            /*OUT*/VexGuestAMD64State* vex_state )
1472     {
1473        Int        stno, preg;
1474        UInt       tag;
1475        ULong*     vexRegs = (ULong*)(&vex_state->guest_FPREG[0]);
1476        UChar*     vexTags = (UChar*)(&vex_state->guest_FPTAG[0]);
1477        Fpu_State* x87     = (Fpu_State*)x87_state;

1480        UInt       fpucw   = x87->env[FP_ENV_CTRL];
1481        UInt       c3210   = x87->env[FP_ENV_STAT] & 0x4700;
1482        VexEmWarn  ew;
1483        UInt       fpround;
1484        ULong      pair;
1485     
1486        /* Copy registers and tags */
1487        for (stno = 0; stno < 8; stno++) {
1488           preg = (stno + ftop) & 7;
1489           tag = (tagw >> (2*preg)) & 3;
1490           if (tag == 3) {
1491              /* register is empty */
1492              /* hmm, if it's empty, does it still get written?  Probably
1493                 safer to say it does.  If we don't, memcheck could get out
1494                 of sync, in that it thinks all FP registers are defined by
1495                 this helper, but in reality some have not been updated. */
1496              if (moveRegs)
1497                 vexRegs[preg] = 0; /* IEEE754 64-bit zero */
1498              vexTags[preg] = 0;
1499           } else {
1500              /* register is non-empty */
1501              if (moveRegs)
1502                 convert_f80le_to_f64le( &x87->reg[10*stno], 
1503                                         (UChar*)&vexRegs[preg] );
1504              vexTags[preg] = 1;
1505           }
1506        }
1507     
1508        /* stack pointer */
1509        vex_state->guest_FTOP = ftop;
1510     
1511        /* status word */
1512        vex_state->guest_FC3210 = c3210;
1513     
1514        /* handle the control word, setting FPROUND and detecting any
1515           emulation warnings. */
1516        pair    = amd64g_check_fldcw ( (ULong)fpucw );
1517        fpround = (UInt)pair;
1518        ew      = (VexEmWarn)(pair >> 32);
1519        
1520        vex_state->guest_FPROUND = fpround & 3;
1521     
1522        /* emulation warnings --> caller */
1523        return ew;
1524     }

1527     /* Create an x87 FPU state from the guest state, as close as
1528        we can approximate it. */
1529     static
1530     void do_get_x87 ( /*IN*/VexGuestAMD64State* vex_state,
1531                       /*OUT*/UChar* x87_state )
1532     {
1533        Int        i, stno, preg;
1534        UInt       tagw;
1535        ULong*     vexRegs = (ULong*)(&vex_state->guest_FPREG[0]);
1536        UChar*     vexTags = (UChar*)(&vex_state->guest_FPTAG[0]);
1537        Fpu_State* x87     = (Fpu_State*)x87_state;
1538        UInt       ftop    = vex_state->guest_FTOP;
1539        UInt       c3210   = vex_state->guest_FC3210;
1540     
1541        for (i = 0; i < 14; i++)
1542           x87->env[i] = 0;
1543     
1544        x87->env[1] = x87->env[3] = x87->env[5] = x87->env[13] = 0xFFFF;
1545        x87->env[FP_ENV_STAT] 
1546           = toUShort(((ftop & 7) << 11) | (c3210 & 0x4700));
1547        x87->env[FP_ENV_CTRL] 
1548           = toUShort(amd64g_create_fpucw( vex_state->guest_FPROUND ));
1549     
1550        /* Dump the register stack in ST order. */
1551        tagw = 0;
1552        for (stno = 0; stno < 8; stno++) {
1553           preg = (stno + ftop) & 7;
1554           if (vexTags[preg] == 0) {
1555              /* register is empty */
1556              tagw |= (3 << (2*preg));
1557              convert_f64le_to_f80le( (UChar*)&vexRegs[preg], 
1558                                      &x87->reg[10*stno] );
1559           } else {
1560              /* register is full. */
1561              tagw |= (0 << (2*preg));
1562              convert_f64le_to_f80le( (UChar*)&vexRegs[preg], 
1563                                      &x87->reg[10*stno] );
1564           }
1565        }
1566        x87->env[FP_ENV_TAG] = toUShort(tagw);
1567     }

1576           vendor_id       : AuthenticAMD
1577           cpu family      : 15
1578           model           : 12
1579           model name      : AMD Athlon(tm) 64 Processor 3200+
1580           stepping        : 0
1581           cpu MHz         : 2200.000
1582           cache size      : 512 KB
1583        */

1586        UShort*   addrS = (UShort*)addr;
1587        UChar*    addrC = (UChar*)addr;
1588        U128*     xmm   = (U128*)(addr + 160);

1602        addrS[1]  = tmp.env[FP_ENV_STAT]; /* FCW: fpu status word */
1603

1605        summary_tags = 0;
1606        fp_tags = tmp.env[FP_ENV_TAG];

1629        addrS[10] = 0; /* BOGUS */

1689        VexEmWarn warnXMM = EmWarn_NONE;

1720        COPY_U128( gst->guest_XMM15, xmm[15] );
1721     
1722     #  undef COPY_U128
1723     
1724        /* Copy the x87 registers out of the image, into a temporary
1725           Fpu_State struct. */
1726        for (i = 0; i < 14; i++) tmp.env[i] = 0;
1727        for (i = 0; i < 80; i++) tmp.reg[i] = 0;
1728        /* fill in tmp.reg[0..7] */
1729        for (stno = 0; stno < 8; stno++) {

1731           UShort* srcS = (UShort*)(&addrS[16 + 8*stno]);

1740        tmp.env[FP_ENV_STAT] = addrS[1]; /* FCW: fpu status word */
1741

1743        for (r = 0; r < 8; r++) {
1744           if (addrC[4] & (1<<r))
1745              fp_tags |= (0 << (2*r)); /* EMPTY */
1746           else 
1747              fp_tags |= (3 << (2*r)); /* VALID -- not really precise enough. */
1748        }

1754        { UInt w32 = (((UInt)addrS[12]) & 0xFFFF)
1755                     | ((((UInt)addrS[13]) & 0xFFFF) << 16);

1759     
1760          gst->guest_SSEROUND = w64 & 0xFFFFFFFFULL;
1761        }
1762     
1763        /* Prefer an X87 emwarn over an XMM one, if both exist. */
1764        if (warnX87 != EmWarn_NONE)
1765           return warnX87;
1766        else
1767           return warnXMM;
1768     }

1805     /* mxcsr[15:0] contains a SSE native format MXCSR value.
1806        Extract from it the required SSEROUND value and any resulting
1807        emulation warning, and return (warn << 32) | sseround value.
1808     */
1809     ULong amd64g_check_ldmxcsr ( ULong mxcsr )
1810     {
1811        /* Decide on a rounding mode.  mxcsr[14:13] holds it. */
1812        /* NOTE, encoded exactly as per enum IRRoundingMode. */
1813        ULong rmode = (mxcsr >> 13) & 3;
1814     
1815        /* Detect any required emulation warnings. */
1816        VexEmWarn ew = EmWarn_NONE;
1817     
1818        if ((mxcsr & 0x1F80) != 0x1F80) {
1819           /* unmasked exceptions! */
1820           ew = EmWarn_X86_sseExns;
1821        }
1822        else 
1823        if (mxcsr & (1<<15)) {
1824           /* FZ is set */
1825           ew = EmWarn_X86_fz;
1826        } 
1827        else
1828        if (mxcsr & (1<<6)) {
1829           /* DAZ is set */
1830           ew = EmWarn_X86_daz;
1831        }
1832     
1833        return (((ULong)ew) << 32) | ((ULong)rmode);
1834     }

2775     /* Helps with PCMP{I,E}STR{I,M}.
2776     
2777        CALLED FROM GENERATED CODE: DIRTY HELPER(s).  (But not really,
2778        actually it could be a clean helper, but for the fact that we can't
2779        pass by value 2 x V128 to a clean helper, nor have one returned.)
2780        Reads guest state, writes to guest state for the xSTRM cases, no
2781        accesses of memory, is a pure function.
2782     
2783        opc_and_imm contains (4th byte of opcode << 8) | the-imm8-byte so
2784        the callee knows which I/E and I/M variant it is dealing with and
2785        what the specific operation is.  4th byte of opcode is in the range
2786        0x60 to 0x63:
2787            istri  66 0F 3A 63
2788            istrm  66 0F 3A 62
2789            estri  66 0F 3A 61
2790            estrm  66 0F 3A 60
2791     
2792        gstOffL and gstOffR are the guest state offsets for the two XMM
2793        register inputs.  We never have to deal with the memory case since
2794        that is handled by pre-loading the relevant value into the fake
2795        XMM16 register.
2796     
2797        For ESTRx variants, edxIN and eaxIN hold the values of those two
2798        registers.
2799     
2800        In all cases, the bottom 16 bits of the result contain the new
2801        OSZACP %rflags values.  For xSTRI variants, bits[31:16] of the
2802        result hold the new %ecx value.  For xSTRM variants, the helper
2803        writes the result directly to the guest XMM0.
2804     
2805        Declarable side effects: in all cases, reads guest state at
2806        [gstOffL, +16) and [gstOffR, +16).  For xSTRM variants, also writes
2807        guest_XMM0.
2808     
2809        Is expected to be called with opc_and_imm combinations which have
2810        actually been validated, and will assert if otherwise.  The front
2811        end should ensure we're only called with verified values.
2812     */
2813     ULong amd64g_dirtyhelper_PCMPxSTRx ( 
2814               VexGuestAMD64State* gst,
2815               HWord opc4_and_imm,
2816               HWord gstOffL, HWord gstOffR,
2817               HWord edxIN, HWord eaxIN
2818            )
2819     {
2820        HWord opc4 = (opc4_and_imm >> 8) & 0xFF;
2821        HWord imm8 = opc4_and_imm & 0xFF;
2822        HWord isISTRx = opc4 & 2;
2823        HWord isxSTRM = (opc4 & 1) ^ 1;
2824        vassert((opc4 & 0xFC) == 0x60); /* 0x60 .. 0x63 */
2825        vassert((imm8 & 1) == 0); /* we support byte-size cases only */
2826     
2827        // where the args are
2828        V128* argL = (V128*)( ((UChar*)gst) + gstOffL );
2829        V128* argR = (V128*)( ((UChar*)gst) + gstOffR );
2830     
2831        /* Create the arg validity masks, either from the vectors
2832           themselves or from the supplied edx/eax values. */
2833        // FIXME: this is only right for the 8-bit data cases.
2834        // At least that is asserted above.
2835        UInt zmaskL, zmaskR;
2836        if (isISTRx) {
2837           zmaskL = zmask_from_V128(argL);
2838           zmaskR = zmask_from_V128(argR);
2839        } else {
2840           Int tmp;
2841           tmp = edxIN & 0xFFFFFFFF;
2842           if (tmp < -16) tmp = -16;
2843           if (tmp > 16)  tmp = 16;
2844           if (tmp < 0)   tmp = -tmp;
2845           vassert(tmp >= 0 && tmp <= 16);
2846           zmaskL = (1 << tmp) & 0xFFFF;
2847           tmp = eaxIN & 0xFFFFFFFF;
2848           if (tmp < -16) tmp = -16;
2849           if (tmp > 16)  tmp = 16;
2850           if (tmp < 0)   tmp = -tmp;
2851           vassert(tmp >= 0 && tmp <= 16);
2852           zmaskR = (1 << tmp) & 0xFFFF;
2853        }
2854     
2855        // temp spot for the resulting flags and vector.
2856        V128 resV;
2857        UInt resOSZACP;
2858     
2859        // do the meyaath
2860        Bool ok = compute_PCMPxSTRx ( 
2861                     &resV, &resOSZACP, argL, argR, 
2862                     zmaskL, zmaskR, imm8, (Bool)isxSTRM
2863                  );
2864     
2865        // front end shouldn't pass us any imm8 variants we can't
2866        // handle.  Hence:
2867        vassert(ok);
2868     
2869        // So, finally we need to get the results back to the caller.
2870        // In all cases, the new OSZACP value is the lowest 16 of
2871        // the return value.
2872        if (isxSTRM) {
2873           /* gst->guest_XMM0 = resV; */ // gcc don't like that
2874           gst->guest_XMM0[0] = resV.w32[0];
2875           gst->guest_XMM0[1] = resV.w32[1];
2876           gst->guest_XMM0[2] = resV.w32[2];
2877           gst->guest_XMM0[3] = resV.w32[3];
2878           return resOSZACP & 0x8D5;
2879        } else {
2880           UInt newECX = resV.w32[0] & 0xFFFF;
2881           return (newECX << 16) | (resOSZACP & 0x8D5);
2882        }
2883     }
