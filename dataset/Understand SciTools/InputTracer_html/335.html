61     /* Sets taint labels for a range of memory in the following way:
62        (Char*)start[0] <-- startLabel
63        (Char*)start[1] <-- startLabel + 1,
64        ...
65        (Char*)start[end-start] <-- startLabel + end-start
66     
67        (I.e. 'end' is included in the range)
68        
69        The 'overwrite' parameter specifies if existing taint labels should be
70        overwritten or if the new taint shall be merged with existing taint.
71        This is used by the 'read' syscall wrapper; the syscall taint propagation 
72        mechanisms will already have propagated taint labels from the syscalls input
73        to the shadow memory of the output buffer when the read post-wrapper is
74        called. It should then merge the new taint with the existing taint in the
75        output buffer, rather than overwriting it. If syscall taint propagation is
76        not used, 'overwrite' will be False.
77     */
78     void IT_(set_mem_taint_range)(Addr start, 
79                                   Addr end, 
80                                   TaintLabel startLabel,
81                                   Bool overwrite);
