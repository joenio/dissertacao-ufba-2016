35     /* File entry for an open file */
36     typedef struct IT_FileEntry
37     {
38        Int     fd;
39        Char*   filename;
40        ULong   filepos;
41        Bool    seek_flag;
42        Word    filter_handle;
43     } IT_FileEntry;

45     /* Linked list with mappings from taint labels to file positions */
46     typedef struct IT_LabelMappingElem IT_LabelMappingElem;
47     struct IT_LabelMappingElem
48     {
49        ULong             filepos;
50        TaintLabel        label;
51        IT_FileEntry*     entry;
52        IT_LabelMappingElem* next; 
53     };
