124     static void pc_post_clo_init ( void )
125     {
126        h_post_clo_init();
127        sg_post_clo_init();
128     #  if defined(VGA_x86) || defined(VGA_amd64)
129        /* nothing */
130     #  elif defined(VGA_ppc32) || defined(VGA_ppc64) || defined(VGA_arm)
131        if (VG_(clo_verbosity) >= 1 && sg_clo_enable_sg_checks) {
132           VG_(message)(Vg_UserMsg, 
133              "WARNING: exp-ptrcheck on ppc32/ppc64/arm platforms: "
134              "stack and global array\n");
135           VG_(message)(Vg_UserMsg, 
136              "WARNING: checking is not currently supported.  "
137              "Only heap checking is\n");
138           VG_(message)(Vg_UserMsg, 
139              "WARNING: supported.  Disabling s/g checks "
140              "(like --enable-sg-checks=no).\n");
141        }
142        sg_clo_enable_sg_checks = False;
143     #  elif defined(VGA_s390x)
144        /* fixs390: to be done. */
145           VG_(message)(Vg_UserMsg,
146              "ERROR: exp-ptrcheck on s390x platform is not supported yet.\n");
147     #  else
148     #    error "Unsupported architecture"
149     #  endif
150     }

152     static void pc_pre_clo_init(void)
153     {
154     #if defined(VGO_darwin)
155        // This makes the (all-failing) regtests run much faster.
156        VG_(printf)("Ptrcheck doesn't work on Darwin yet, sorry.\n");
157        VG_(exit)(1);
158     #endif
159     
160        VG_(details_name)            ("exp-ptrcheck");
161        VG_(details_version)         (NULL);
162        VG_(details_description)     ("a heap, stack and global array "
163                                      "overrun detector");
164        VG_(details_copyright_author)(
165           "Copyright (C) 2003-2010, and GNU GPL'd, by OpenWorks Ltd et al.");
166        VG_(details_bug_reports_to)  (VG_BUGS_TO);
167        VG_(details_avg_translation_sizeB) ( 496 );
168     
169        VG_(basic_tool_funcs)        (pc_post_clo_init,
170                                      h_instrument,
171                                      pc_fini);
172     
173        VG_(needs_malloc_replacement)( h_replace_malloc,
174                                       h_replace___builtin_new,
175                                       h_replace___builtin_vec_new,
176                                       h_replace_memalign,
177                                       h_replace_calloc,
178                                       h_replace_free,
179                                       h_replace___builtin_delete,
180                                       h_replace___builtin_vec_delete,
181                                       h_replace_realloc,
182                                       h_replace_malloc_usable_size,
183                                       0 /* no need for client heap redzones */ );
184     
185        VG_(needs_var_info)          ();
186     
187        VG_(needs_core_errors)       ();
188        VG_(needs_tool_errors)       (pc_eq_Error,
189                                      pc_before_pp_Error,
190                                      pc_pp_Error,
191                                      True,/*show TIDs for errors*/
192                                      pc_update_Error_extra,
193                                      pc_is_recognised_suppression,
194                                      pc_read_extra_suppression_info,
195                                      pc_error_matches_suppression,
196                                      pc_get_error_name,
197                                      pc_get_extra_suppression_info);
198     
199        VG_(needs_xml_output)        ();
200     
201        VG_(needs_syscall_wrapper)( h_pre_syscall,
202                                    h_post_syscall );
203     
204        VG_(needs_command_line_options)( pc_process_cmd_line_options,
205                                         pc_print_usage,
206                                         pc_print_debug_usage );
207     
208        VG_(track_die_mem_stack)        ( pc_die_mem_stack );
209        VG_(track_pre_thread_ll_create) ( pc_pre_thread_ll_create );
210        VG_(track_pre_thread_first_insn)( pc_pre_thread_first_insn );
211     
212        VG_(track_new_mem_mmap)         ( pc_new_mem_mmap );
213        VG_(track_new_mem_startup)      ( pc_new_mem_startup);
214        VG_(track_die_mem_munmap)       ( pc_die_mem_munmap );
215     
216        VG_(track_pre_mem_read)         ( pc_pre_mem_read );
217        VG_(track_pre_mem_read_asciiz)  ( pc_pre_mem_read_asciiz );
218        VG_(track_pre_mem_write)        ( pc_pre_mem_write );
219     
220        VG_(track_post_reg_write_clientcall_return) ( h_post_reg_write_clientcall );
221        VG_(track_post_reg_write)( h_post_reg_write_demux );
222     
223        h_pre_clo_init();
224        sg_pre_clo_init();
225     
226        VG_(clo_vex_control).iropt_unroll_thresh = 0;
227        VG_(clo_vex_control).guest_chase_thresh = 0;
228     }
