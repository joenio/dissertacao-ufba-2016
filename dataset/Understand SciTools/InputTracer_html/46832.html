74     /*--- The _svma / _avma / _image / _bias naming scheme     ---*/
75     /*------------------------------------------------------------*/
76     
77     /* JRS 11 Jan 07: I find the different kinds of addresses involved in
78        debuginfo reading confusing.  Recently I arrived at some
79        terminology which makes it clearer (to me, at least).  There are 3
80        kinds of address used in the debuginfo reading process:
81      
82        stated VMAs - the address where (eg) a .so says a symbol is, that
83                      is, what it tells you if you consider the .so in
84                      isolation
85      
86        actual VMAs - the address where (eg) said symbol really wound up
87                      after the .so was mapped into memory
88      
89        image addresses - pointers into the copy of the .so (etc)
90                          transiently mmaped aboard whilst we read its info
91     
92        Additionally I use the term 'bias' to denote the difference
93        between stated and actual VMAs for a given entity.
94     
95        This terminology is not used consistently, but a start has been
96        made.  readelf.c and the call-frame info reader in readdwarf.c now
97        use it.  Specifically, various variables and structure fields have
98        been annotated with _avma / _svma / _image / _bias.  In places _img
99        is used instead of _image for the sake of brevity.
100     */

119     /* Find 'di' in the debugInfo_list and move it one step closer the the
120        front of the list, so as to make subsequent searches for it
121        cheaper.  When used in a controlled way, makes a major improvement
122        in some DebugInfo-search-intensive situations, most notably stack
123        unwinding on amd64-linux. */
124     static void move_DebugInfo_one_step_forward ( DebugInfo* di )
125     {
126        DebugInfo *di0, *di1, *di2;
127        if (di == debugInfo_list)
128           return; /* already at head of list */
129        vg_assert(di != NULL);
130        di0 = debugInfo_list;
131        di1 = NULL;
132        di2 = NULL;
133        while (True) {
134           if (di0 == NULL || di0 == di) break;
135           di2 = di1;
136           di1 = di0;
137           di0 = di0->next;
138        }
139        vg_assert(di0 == di);
140        if (di0 != NULL && di1 != NULL && di2 != NULL) {
141           DebugInfo* tmp;
142           /* di0 points to di, di1 to its predecessor, and di2 to di1's
143              predecessor.  Swap di0 and di1, that is, move di0 one step
144              closer to the start of the list. */
145           vg_assert(di2->next == di1);
146           vg_assert(di1->next == di0);
147           tmp = di0->next;
148           di2->next = di0;
149           di0->next = di1;
150           di1->next = tmp;
151        }
152        else
153        if (di0 != NULL && di1 != NULL && di2 == NULL) {
154           /* it's second in the list. */
155           vg_assert(debugInfo_list == di1);
156           vg_assert(di1->next == di0);
157           di1->next = di0->next;
158           di0->next = di1;
159           debugInfo_list = di0;
160        }
161     }

173     /* Allocate and zero out a new DebugInfo record. */
174     static 
175     DebugInfo* alloc_DebugInfo( const UChar* filename,
176                                 const UChar* memname )
177     {
178        Bool       traceme;
179        DebugInfo* di;
180     
181        vg_assert(filename);
182     
183        di = ML_(dinfo_zalloc)("di.debuginfo.aDI.1", sizeof(DebugInfo));
184        di->handle    = handle_counter++;
185        di->filename  = ML_(dinfo_strdup)("di.debuginfo.aDI.2", filename);
186        di->memname   = memname ? ML_(dinfo_strdup)("di.debuginfo.aDI.3", memname)
187                                : NULL;
188     
189        /* Everything else -- pointers, sizes, arrays -- is zeroed by calloc.
190           Now set up the debugging-output flags. */
191        traceme 
192           = VG_(string_match)( VG_(clo_trace_symtab_patt), filename )
193             || (memname && VG_(string_match)( VG_(clo_trace_symtab_patt), 
194                                               memname ));
195        if (traceme) {
196           di->trace_symtab = VG_(clo_trace_symtab);
197           di->trace_cfi    = VG_(clo_trace_cfi);
198           di->ddump_syms   = VG_(clo_debug_dump_syms);
199           di->ddump_line   = VG_(clo_debug_dump_line);
200           di->ddump_frames = VG_(clo_debug_dump_frames);
201        }
202     
203        return di;
204     }

207     /* Free a DebugInfo, and also all the stuff hanging off it. */
208     static void free_DebugInfo ( DebugInfo* di )
209     {
210        Word i, j, n;
211        struct strchunk *chunk, *next;
212        TyEnt* ent;
213        GExpr* gexpr;
214     
215        vg_assert(di != NULL);
216        if (di->filename)   ML_(dinfo_free)(di->filename);
217        if (di->symtab)     ML_(dinfo_free)(di->symtab);
218        if (di->loctab)     ML_(dinfo_free)(di->loctab);
219        if (di->cfsi)       ML_(dinfo_free)(di->cfsi);
220        if (di->cfsi_exprs) VG_(deleteXA)(di->cfsi_exprs);
221        if (di->fpo)        ML_(dinfo_free)(di->fpo);
222     
223        for (chunk = di->strchunks; chunk != NULL; chunk = next) {
224           next = chunk->next;
225           ML_(dinfo_free)(chunk);
226        }
227     
228        /* Delete the two admin arrays.  These lists exist primarily so
229           that we can visit each object exactly once when we need to
230           delete them. */
231        if (di->admin_tyents) {
232           n = VG_(sizeXA)(di->admin_tyents);
233           for (i = 0; i < n; i++) {
234              ent = (TyEnt*)VG_(indexXA)(di->admin_tyents, i);
235              /* Dump anything hanging off this ent */
236              ML_(TyEnt__make_EMPTY)(ent);
237           }
238           VG_(deleteXA)(di->admin_tyents);
239           di->admin_tyents = NULL;
240        }
241     
242        if (di->admin_gexprs) {
243           n = VG_(sizeXA)(di->admin_gexprs);
244           for (i = 0; i < n; i++) {
245              gexpr = *(GExpr**)VG_(indexXA)(di->admin_gexprs, i);
246              ML_(dinfo_free)(gexpr);
247           }
248           VG_(deleteXA)(di->admin_gexprs);
249           di->admin_gexprs = NULL;
250        }
251     
252        /* Dump the variable info.  This is kinda complex: we must take
253           care not to free items which reside in either the admin lists
254           (as we have just freed them) or which reside in the DebugInfo's
255           string table. */
256        if (di->varinfo) {
257           for (i = 0; i < VG_(sizeXA)(di->varinfo); i++) {
258              OSet* scope = *(OSet**)VG_(indexXA)(di->varinfo, i);
259              if (!scope) continue;
260              /* iterate over all entries in 'scope' */
261              VG_(OSetGen_ResetIter)(scope);
262              while (True) {
263                 DiAddrRange* arange = VG_(OSetGen_Next)(scope);
264                 if (!arange) break;
265                 /* for each var in 'arange' */
266                 vg_assert(arange->vars);
267                 for (j = 0; j < VG_(sizeXA)( arange->vars ); j++) {
268                    DiVariable* var = (DiVariable*)VG_(indexXA)(arange->vars,j);
269                    vg_assert(var);
270                    /* Nothing to free in var: all the pointer fields refer
271                       to stuff either on an admin list, or in
272                       .strchunks */
273                 }
274                 VG_(deleteXA)(arange->vars);
275                 /* Don't free arange itself, as OSetGen_Destroy does
276                    that */
277              }
278              VG_(OSetGen_Destroy)(scope);
279           }
280           VG_(deleteXA)(di->varinfo);
281        }
282     
283        ML_(dinfo_free)(di);
284     }

287     /* 'si' is a member of debugInfo_list.  Find it, remove it from the
288        list, notify m_redir that this has happened, and free all storage
289        reachable from it.
290     */
291     static void discard_DebugInfo ( DebugInfo* di )
292     {
293     #  if defined(VGP_ppc32_aix5)
294        HChar* reason = "__unload";
295     #  elif defined(VGP_ppc64_aix5)
296        HChar* reason = "kunload64";
297     #  else
298        HChar* reason = "munmap";
299     #  endif
300     
301        DebugInfo** prev_next_ptr = &debugInfo_list;
302        DebugInfo*  curr          =  debugInfo_list;
303     
304        while (curr) {
305           if (curr == di) {
306              /* Found it;  remove from list and free it. */
307              if (curr->have_dinfo
308                  && (VG_(clo_verbosity) > 1 || VG_(clo_trace_redir)))
309                 VG_(message)(Vg_DebugMsg, 
310                              "Discarding syms at %#lx-%#lx in %s due to %s()\n",
311                              di->text_avma, 
312                              di->text_avma + di->text_size,
313                              curr->filename ? curr->filename : (UChar*)"???",
314                              reason);
315              vg_assert(*prev_next_ptr == curr);
316              *prev_next_ptr = curr->next;
317              if (curr->have_dinfo)
318                 VG_(redir_notify_delete_DebugInfo)( curr );
319              free_DebugInfo(curr);
320              return;
321           }
322           prev_next_ptr = &curr->next;
323           curr          =  curr->next;
324        }
325     
326        /* Not found. */
327     }

330     /* Repeatedly scan debugInfo_list, looking for DebugInfos with text
331        AVMAs intersecting [start,start+length), and call discard_DebugInfo
332        to get rid of them.  This modifies the list, hence the multiple
333        iterations.  Returns True iff any such DebugInfos were found.
334     */
335     static Bool discard_syms_in_range ( Addr start, SizeT length )
336     {
337        Bool       anyFound = False;
338        Bool       found;
339        DebugInfo* curr;
340     
341        while (True) {
342           found = False;
343     
344           curr = debugInfo_list;
345           while (True) {
346              if (curr == NULL)
347                 break;
348              if (curr->text_present
349                  && curr->text_size > 0
350                  && (start+length - 1 < curr->text_avma 
351                      || curr->text_avma + curr->text_size - 1 < start)) {
352                 /* no overlap */
353     	 } else {
354     	    found = True;
355     	    break;
356     	 }
357     	 curr = curr->next;
358           }
359     
360           if (!found) break;
361           anyFound = True;
362           discard_DebugInfo( curr );
363        }
364     
365        return anyFound;
366     }

369     /* Does [s1,+len1) overlap [s2,+len2) ?  Note: does not handle
370        wraparound at the end of the address space -- just asserts in that
371        case. */
372     static Bool ranges_overlap (Addr s1, SizeT len1, Addr s2, SizeT len2 )
373     {
374        Addr e1, e2;
375        if (len1 == 0 || len2 == 0) 
376           return False;
377        e1 = s1 + len1 - 1;
378        e2 = s2 + len2 - 1;
379        /* Assert that we don't have wraparound.  If we do it would imply
380           that file sections are getting mapped around the end of the
381           address space, which sounds unlikely. */
382        vg_assert(s1 <= e1);
383        vg_assert(s2 <= e2);
384        if (e1 < s2 || e2 < s1) return False;
385        return True;
386     }

389     /* Do the basic rx_ and rw_ mappings of the two DebugInfos overlap in
390        any way? */
391     static Bool do_DebugInfos_overlap ( DebugInfo* di1, DebugInfo* di2 )
392     {
393        vg_assert(di1);
394        vg_assert(di2);
395     
396        if (di1->have_rx_map && di2->have_rx_map
397            && ranges_overlap(di1->rx_map_avma, di1->rx_map_size,
398                              di2->rx_map_avma, di2->rx_map_size))
399           return True;
400     
401        if (di1->have_rx_map && di2->have_rw_map
402            && ranges_overlap(di1->rx_map_avma, di1->rx_map_size,
403                              di2->rw_map_avma, di2->rw_map_size))
404           return True;
405     
406        if (di1->have_rw_map && di2->have_rx_map
407            && ranges_overlap(di1->rw_map_avma, di1->rw_map_size,
408                              di2->rx_map_avma, di2->rx_map_size))
409           return True;
410     
411        if (di1->have_rw_map && di2->have_rw_map
412            && ranges_overlap(di1->rw_map_avma, di1->rw_map_size,
413                              di2->rw_map_avma, di2->rw_map_size))
414           return True;
415     
416        return False;
417     }

444     /* Discard any elements of debugInfo_list which overlap with diRef.
445        Clearly diRef must have its rx_ and rw_ mapping information set to
446        something sane. */
447     #if defined(VGO_aix5)
448     __attribute__((unused))
449     #endif
450     static void discard_DebugInfos_which_overlap_with ( DebugInfo* diRef )
451     {
452        DebugInfo* di;
453        /* Mark all the DebugInfos in debugInfo_list that need to be
454           deleted.  First, clear all the mark bits; then set them if they
455           overlap with siRef.  Since siRef itself is in this list we at
456           least expect its own mark bit to be set. */
457        for (di = debugInfo_list; di; di = di->next) {
458           di->mark = do_DebugInfos_overlap( di, diRef );
459           if (di == diRef) {
460              vg_assert(di->mark);
461              di->mark = False;
462           }
463        }
464        discard_marked_DebugInfos();
465     }

468     /* Find the existing DebugInfo for (memname,filename) or if not found,
469        create one.  In the latter case memname and filename are strdup'd
470        into VG_AR_DINFO, and the new DebugInfo is added to
471        debugInfo_list. */
472     static
473     DebugInfo* find_or_create_DebugInfo_for ( UChar* filename, UChar* memname )
474     {
475        DebugInfo* di;
476        vg_assert(filename);
477        for (di = debugInfo_list; di; di = di->next) {

495     /* Debuginfo reading for 'di' has just been successfully completed.
496        Check that the invariants stated in
497        "Comment_on_IMPORTANT_CFSI_REPRESENTATIONAL_INVARIANTS" in
498        priv_storage.h are observed. */
499     static void check_CFSI_related_invariants ( DebugInfo* di )
500     {
501        DebugInfo* di2 = NULL;
502        vg_assert(di);
503        /* This fn isn't called until after debuginfo for this object has
504           been successfully read.  And that shouldn't happen until we have
505           both a r-x and rw- mapping for the object.  Hence: */
506        vg_assert(di->have_rx_map);
507        vg_assert(di->have_rw_map);
508        /* degenerate case: r-x section is empty */
509        if (di->rx_map_size == 0) {
510           vg_assert(di->cfsi == NULL);
511           return;
512        }
513        /* normal case: r-x section is nonempty */
514        /* invariant (0) */
515        vg_assert(di->rx_map_size > 0);
516        /* invariant (1) */
517        for (di2 = debugInfo_list; di2; di2 = di2->next) {
518           if (di2 == di)
519              continue;
520           if (di2->rx_map_size == 0)
521              continue;
522           vg_assert(di->rx_map_avma + di->rx_map_size <= di2->rx_map_avma
523                     || di2->rx_map_avma + di2->rx_map_size <= di->rx_map_avma);
524        }
525        di2 = NULL;
526        /* invariant (2) */
527        if (di->cfsi) {
528           vg_assert(di->cfsi_minavma <= di->cfsi_maxavma); /* duh! */
529           vg_assert(di->cfsi_minavma >= di->rx_map_avma);
530           vg_assert(di->cfsi_maxavma < di->rx_map_avma + di->rx_map_size);
531        }
532        /* invariants (3) and (4) */
533        if (di->cfsi) {
534           Word i;
535           vg_assert(di->cfsi_used > 0);
536           vg_assert(di->cfsi_size > 0);
537           for (i = 0; i < di->cfsi_used; i++) {
538              DiCfSI* cfsi = &di->cfsi[i];
539              vg_assert(cfsi->len > 0);
540              vg_assert(cfsi->base >= di->cfsi_minavma);
541              vg_assert(cfsi->base + cfsi->len - 1 <= di->cfsi_maxavma);
542              if (i > 0) {
543                 DiCfSI* cfsip = &di->cfsi[i-1];
544                 vg_assert(cfsip->base + cfsip->len <= cfsi->base);
545              }
546           }
547        } else {
548           vg_assert(di->cfsi_used == 0);
549           vg_assert(di->cfsi_size == 0);
550        }
551     }

581     /* The debug info system is driven by notifications that a text
582        segment has been mapped in, or unmapped.  When that happens it
583        tries to acquire/discard whatever info is available for the
584        corresponding object.  This section contains the notification

586     
587     /* Notify the debuginfo system about a new mapping.  This is the way
588        new debug information gets loaded.  If allow_SkFileV is True, it
589        will try load debug info if the mapping at 'a' belongs to Valgrind;
590        whereas normally (False) it will not do that.  This allows us to
591        carefully control when the thing will read symbols from the
592        Valgrind executable itself.
593     
594        If a call to VG_(di_notify_mmap) causes debug info to be read, then
595        the returned ULong is an abstract handle which can later be used to
596        refer to the debuginfo read as a result of this specific mapping,
597        in later queries to m_debuginfo.  In this case the handle value
598        will be one or above.  If the returned value is zero, no debug info
599        was read. */
600     
601     ULong VG_(di_notify_mmap)( Addr a, Bool allow_SkFileV )
602     {
603        NSegment const * seg;
604        HChar*     filename;
605        Bool       ok, is_rx_map, is_rw_map;
606        DebugInfo* di;
607        ULong      di_handle;
608        SysRes     fd;
609        Int        nread, oflags;
610        HChar      buf1k[1024];
611        Bool       debug = False;
612        SysRes     statres;
613        struct vg_stat statbuf;
614     
615        /* In short, figure out if this mapping is of interest to us, and
616           if so, try to guess what ld.so is doing and when/if we should
617           read debug info. */
618        seg = VG_(am_find_nsegment)(a);
619        vg_assert(seg);
620     
621        if (debug)
622           VG_(printf)("di_notify_mmap-1: %#lx-%#lx %c%c%c\n",
623                       seg->start, seg->end, 
624                       seg->hasR ? 'r' : '-',
625                       seg->hasW ? 'w' : '-',seg->hasX ? 'x' : '-' );
626     
627        /* guaranteed by aspacemgr-linux.c, sane_NSegment() */
628        vg_assert(seg->end > seg->start);
629     
630        /* Ignore non-file mappings */
631        if ( ! (seg->kind == SkFileC
632                || (seg->kind == SkFileV && allow_SkFileV)) )
633           return 0;
634     
635        /* If the file doesn't have a name, we're hosed.  Give up. */
636        filename = VG_(am_get_filename)( (NSegment*)seg );
637        if (!filename)
638           return 0;
639     
640        if (debug)
641           VG_(printf)("di_notify_mmap-2: %s\n", filename);
642     
643        /* Only try to read debug information from regular files.  */
644        statres = VG_(stat)(filename, &statbuf);
645     
646        /* stat dereferences symlinks, so we don't expect it to succeed and
647           yet produce something that is a symlink. */
648        vg_assert(sr_isError(statres) || ! VKI_S_ISLNK(statbuf.mode));
649     
650        /* Don't let the stat call fail silently.  Filter out some known
651           sources of noise before complaining, though. */
652        if (sr_isError(statres)) {
653           DebugInfo fake_di;
654           Bool quiet = VG_(strstr)(filename, "/var/run/nscd/") != NULL;
655           if (!quiet && VG_(clo_verbosity) > 1) {
656              VG_(memset)(&fake_di, 0, sizeof(fake_di));
657              fake_di.filename = filename;
658              ML_(symerr)(&fake_di, True, "failed to stat64/stat this file");
659           }
660           return 0;
661        }
662     
663        /* Finally, the point of all this stattery: if it's not a regular file,
664           don't try to read debug info from it. */
665        if (! VKI_S_ISREG(statbuf.mode))
666           return 0;
667     
668        /* no uses of statbuf below here. */
669     
670        /* Now we have to guess if this is a text-like mapping, a data-like
671           mapping, neither or both.  The rules are:
672     
673             text if:   x86-linux    r and x
674                        other-linux  r and x and not w
675     
676             data if:   x86-linux    r and w
677                        other-linux  r and w and not x
678     
679           Background: On x86-linux, objects are typically mapped twice:
680     
681           1b8fb000-1b8ff000 r-xp 00000000 08:02 4471477 vgpreload_memcheck.so
682           1b8ff000-1b900000 rw-p 00004000 08:02 4471477 vgpreload_memcheck.so
683     
684           whereas ppc32-linux mysteriously does this:
685     
686           118a6000-118ad000 r-xp 00000000 08:05 14209428 vgpreload_memcheck.so
687           118ad000-118b6000 ---p 00007000 08:05 14209428 vgpreload_memcheck.so
688           118b6000-118bd000 rwxp 00000000 08:05 14209428 vgpreload_memcheck.so
689     
690           The third mapping should not be considered to have executable
691           code in.  Therefore a test which works for both is: r and x and
692           NOT w.  Reading symbols from the rwx segment -- which overlaps
693           the r-x segment in the file -- causes the redirection mechanism
694           to redirect to addresses in that third segment, which is wrong
695           and causes crashes.
696     
697           JRS 28 Dec 05: unfortunately icc 8.1 on x86 has been seen to
698           produce executables with a single rwx segment rather than a
699           (r-x,rw-) pair. That means the rules have to be modified thusly:
700     
701           x86-linux:   consider if r and x
702           all others:  consider if r and x and not w
703     
704           2009 Aug 16: apply similar kludge to ppc32-linux.
705           See http://bugs.kde.org/show_bug.cgi?id=190820
706     
707           There are two modes on s390x: with and without the noexec kernel
708           parameter. Together with some older kernels, this leads to several
709           variants:
710           executable: r and x
711           data:       r and w and x
712           or
713           executable: r and x
714           data:       r and w
715        */
716        is_rx_map = False;
717        is_rw_map = False;
718     #  if defined(VGA_x86) || defined(VGA_ppc32)
719        is_rx_map = seg->hasR && seg->hasX;
720        is_rw_map = seg->hasR && seg->hasW;
721     #  elif defined(VGA_amd64) || defined(VGA_ppc64) || defined(VGA_arm)
722        is_rx_map = seg->hasR && seg->hasX && !seg->hasW;
723        is_rw_map = seg->hasR && seg->hasW && !seg->hasX;
724     #  elif defined(VGP_s390x_linux)
725        is_rx_map = seg->hasR && seg->hasX && !seg->hasW;
726        is_rw_map = seg->hasR && seg->hasW;
727     #  else
728     #    error "Unknown platform"
729     #  endif
730     
731        if (debug)
732           VG_(printf)("di_notify_mmap-3: is_rx_map %d, is_rw_map %d\n",
733                       (Int)is_rx_map, (Int)is_rw_map);
734     
735        /* If it is neither text-ish nor data-ish, we're not interested. */
736        if (!(is_rx_map || is_rw_map))
737           return 0;
738     
739        /* Peer at the first few bytes of the file, to see if it is an ELF */
740        /* object file. Ignore the file if we do not have read permission. */
741        VG_(memset)(buf1k, 0, sizeof(buf1k));
742        oflags = VKI_O_RDONLY;
743     #  if defined(VKI_O_LARGEFILE)
744        oflags |= VKI_O_LARGEFILE;
745     #  endif
746        fd = VG_(open)( filename, oflags, 0 );
747        if (sr_isError(fd)) {
748           if (sr_Err(fd) != VKI_EACCES) {
749              DebugInfo fake_di;
750              VG_(memset)(&fake_di, 0, sizeof(fake_di));
751              fake_di.filename = filename;
752              ML_(symerr)(&fake_di, True, "can't open file to inspect ELF header");
753           }
754           return 0;
755        }
756        nread = VG_(read)( sr_Res(fd), buf1k, sizeof(buf1k) );
757        VG_(close)( sr_Res(fd) );
758     
759        if (nread == 0)
760           return 0;
761        if (nread < 0) {
762           DebugInfo fake_di;
763           VG_(memset)(&fake_di, 0, sizeof(fake_di));
764           fake_di.filename = filename;
765           ML_(symerr)(&fake_di, True, "can't read file to inspect ELF header");
766           return 0;
767        }
768        vg_assert(nread > 0 && nread <= sizeof(buf1k) );
769     
770        /* We're only interested in mappings of object files. */
771        // Nb: AIX5 doesn't use this file and so isn't represented here.
772     #if defined(VGO_linux)
773        if (!ML_(is_elf_object_file)( buf1k, (SizeT)nread ))
774           return 0;
775     #elif defined(VGO_darwin)
776        if (!ML_(is_macho_object_file)( buf1k, (SizeT)nread ))
777           return 0;
778     #else
779     #  error "unknown OS"
780     #endif
781     
782        /* See if we have a DebugInfo for this filename.  If not,
783           create one. */
784        di = find_or_create_DebugInfo_for( filename, NULL/*membername*/ );
785        vg_assert(di);
786     
787        if (is_rx_map) {
788           /* We have a text-like mapping.  Note the details. */
789           if (!di->have_rx_map) {
790              di->have_rx_map = True;
791              di->rx_map_avma = a;
792              di->rx_map_size = seg->end + 1 - seg->start;
793              di->rx_map_foff = seg->offset;
794           } else {
795              /* FIXME: complain about a second text-like mapping */
796           }
797        }
798     
799        if (is_rw_map) {
800           /* We have a data-like mapping.  Note the details. */
801           if (!di->have_rw_map) {
802              di->have_rw_map = True;
803              di->rw_map_avma = a;
804              di->rw_map_size = seg->end + 1 - seg->start;
805              di->rw_map_foff = seg->offset;
806           } else {
807              /* FIXME: complain about a second data-like mapping */
808           }
809        }
810     
811        /* If we don't have an rx and rw mapping, or if we already have
812           debuginfo for this mapping for whatever reason, go no
813           further. */
814        if ( ! (di->have_rx_map && di->have_rw_map && !di->have_dinfo) )
815           return 0;
816     
817        /* Ok, so, finally, let's try to read the debuginfo. */
818        vg_assert(di->filename);
819        TRACE_SYMTAB("\n");
820        TRACE_SYMTAB("------ start ELF OBJECT "
821                     "------------------------------\n");
822        TRACE_SYMTAB("------ name = %s\n", di->filename);
823        TRACE_SYMTAB("\n");
824     
825        /* We're going to read symbols and debug info for the avma
826           ranges [rx_map_avma, +rx_map_size) and [rw_map_avma,
827           +rw_map_size).  First get rid of any other DebugInfos which
828           overlap either of those ranges (to avoid total confusion). */
829        discard_DebugInfos_which_overlap_with( di );
830     
831        /* .. and acquire new info. */
832        // Nb: AIX5 doesn't use this file and so isn't represented here.
833     #if defined(VGO_linux)
834        ok = ML_(read_elf_debug_info)( di );
835     #elif defined(VGO_darwin)
836        ok = ML_(read_macho_debug_info)( di );
837     #else
838     #  error "unknown OS"
839     #endif
840     
841        if (ok) {
842     
843           TRACE_SYMTAB("\n------ Canonicalising the "
844                        "acquired info ------\n");
845           /* invalidate the CFI unwind cache. */
846           cfsi_cache__invalidate();
847           /* prepare read data for use */
848           ML_(canonicaliseTables)( di );
849           /* notify m_redir about it */
850           TRACE_SYMTAB("\n------ Notifying m_redir ------\n");
851           VG_(redir_notify_new_DebugInfo)( di );
852           /* Note that we succeeded */
853           di->have_dinfo = True;
854           tl_assert(di->handle > 0);
855           di_handle = di->handle;
856           /* Check invariants listed in
857              Comment_on_IMPORTANT_REPRESENTATIONAL_INVARIANTS in
858              priv_storage.h. */
859           check_CFSI_related_invariants(di);
860     
861        } else {
862           TRACE_SYMTAB("\n------ ELF reading failed ------\n");
863           /* Something went wrong (eg. bad ELF file).  Should we delete
864              this DebugInfo?  No - it contains info on the rw/rx
865              mappings, at least. */
866           di_handle = 0;
867           vg_assert(di->have_dinfo == False);
868        }
869     
870        TRACE_SYMTAB("\n");
871        TRACE_SYMTAB("------ name = %s\n", di->filename);
872        TRACE_SYMTAB("------ end ELF OBJECT "
873                     "------------------------------\n");
874        TRACE_SYMTAB("\n");
875     
876        return di_handle;
877     }

909     /*--------- PDB (windows debug info) reading --------- */
910     
911     /* this should really return ULong, as per VG_(di_notify_mmap). */
912     void VG_(di_notify_pdb_debuginfo)( Int fd_obj, Addr avma_obj,
913                                        SizeT total_size,
914                                        PtrdiffT unknown_purpose__reloc )
915     {
916        Int    i, r, sz_exename;
917        ULong  obj_mtime, pdb_mtime;
918        Char   exename[VKI_PATH_MAX];
919        Char*  pdbname = NULL;
920        Char*  dot;
921        SysRes sres;
922        Int    fd_pdbimage;
923        SizeT  n_pdbimage;
924        struct vg_stat stat_buf;
925     
926        if (VG_(clo_verbosity) > 0) {
927           VG_(message)(Vg_UserMsg, "\n");
928           VG_(message)(Vg_UserMsg,
929              "LOAD_PDB_DEBUGINFO: clreq:   fd=%d, avma=%#lx, total_size=%lu, "
930              "uu_reloc=%#lx\n", 
931              fd_obj, avma_obj, total_size, unknown_purpose__reloc
932           );
933        }
934     
935        /* 'fd' refers to the .exe/.dll we're dealing with.  Get its modification
936           time into obj_mtime. */
937        r = VG_(fstat)(fd_obj, &stat_buf);
938        if (r == -1)
939           goto out; /* stat failed ?! */
940        vg_assert(r == 0);
941        obj_mtime = stat_buf.mtime;
942     
943        /* and get its name into exename[]. */
944        vg_assert(VKI_PATH_MAX > 100); /* to ensure /proc/self/fd/%d is safe */
945        VG_(memset)(exename, 0, sizeof(exename));
946        VG_(sprintf)(exename, "/proc/self/fd/%d", fd_obj);
947        /* convert exename from a symlink to real name .. overwrites the
948           old contents of the buffer.  Ick. */
949        sz_exename = VG_(readlink)(exename, exename, sizeof(exename)-2 );
950        if (sz_exename == -1)
951           goto out; /* readlink failed ?! */
952        vg_assert(sz_exename >= 0 && sz_exename < sizeof(exename));
953        vg_assert(exename[sizeof(exename)-1] == 0);
954     
955        if (VG_(clo_verbosity) > 0) {
956           VG_(message)(Vg_UserMsg, "LOAD_PDB_DEBUGINFO: objname: %s\n", exename);
957        }
958     
959        /* Try to get the PDB file name from the executable. */
960        pdbname = ML_(find_name_of_pdb_file)(exename);
961        if (pdbname) {
962           vg_assert(VG_(strlen)(pdbname) >= 5); /* 5 = strlen("X.pdb") */
963           /* So we successfully extracted a name from the PE file.  But it's
964              likely to be of the form
965                 e:\foo\bar\xyzzy\wibble.pdb
966              and we need to change it into something we can actually open
967              in Wine-world, which basically means turning it into
968                 $HOME/.wine/drive_e/foo/bar/xyzzy/wibble.pdb
969              We also take into account $WINEPREFIX, if it is set.
970              For the moment, if the name isn't fully qualified, just forget it
971              (we'd have to root around to find where the pdb actually is)
972           */
973           /* Change all the backslashes to forward slashes */
974           for (i = 0; pdbname[i]; i++) {
975              if (pdbname[i] == '\\')
976                 pdbname[i] = '/';
977           }
978           Bool is_quald
979              = ('a' <= VG_(tolower)(pdbname[0]) && VG_(tolower)(pdbname[0]) <= 'z')
980                && pdbname[1] == ':'
981                && pdbname[2] == '/';
982           HChar* home = VG_(getenv)("HOME");
983           HChar* wpfx = VG_(getenv)("WINEPREFIX");
984           if (is_quald && wpfx) {
985              /* Change e:/foo/bar/xyzzy/wibble.pdb
986                     to $WINEPREFIX/drive_e/foo/bar/xyzzy/wibble.pdb
987              */
988              Int mashedSzB = VG_(strlen)(pdbname) + VG_(strlen)(wpfx) + 50/*misc*/;
989              HChar* mashed = ML_(dinfo_zalloc)("di.debuginfo.dnpdi.1", mashedSzB);
990              VG_(sprintf)(mashed, "%s/drive_%c%s",
991                           wpfx, pdbname[0], &pdbname[2]);
992              vg_assert(mashed[mashedSzB-1] == 0);
993              ML_(dinfo_free)(pdbname);
994              pdbname = mashed;
995           }
996           else if (is_quald && home && !wpfx) {
997              /* Change e:/foo/bar/xyzzy/wibble.pdb
998                     to $HOME/.wine/drive_e/foo/bar/xyzzy/wibble.pdb
999              */
1000              Int mashedSzB = VG_(strlen)(pdbname) + VG_(strlen)(home) + 50/*misc*/;
1001              HChar* mashed = ML_(dinfo_zalloc)("di.debuginfo.dnpdi.2", mashedSzB);
1002              VG_(sprintf)(mashed, "%s/.wine/drive_%c%s",
1003                           home, pdbname[0], &pdbname[2]);
1004              vg_assert(mashed[mashedSzB-1] == 0);
1005              ML_(dinfo_free)(pdbname);
1006              pdbname = mashed;
1007           } else {
1008              /* It's not a fully qualified path, or neither $HOME nor $WINE
1009                 are set (strange).  Give up. */
1010              ML_(dinfo_free)(pdbname);
1011              pdbname = NULL;
1012           }
1013        }
1014     
1015        /* Try s/exe/pdb/ if we don't have a valid pdbname. */
1016        if (!pdbname) {
1017           /* Try to find a matching PDB file from which to read debuginfo.
1018              Windows PE files have symbol tables and line number information,
1019              but MSVC doesn't seem to use them. */
1020           /* Why +5 ?  Because in the worst case, we could find a dot as the
1021              last character of pdbname, and we'd then put "pdb" right after
1022              it, hence extending it a bit. */
1023           pdbname = ML_(dinfo_zalloc)("di.debuginfo.lpd1", sz_exename+5);
1024           VG_(strcpy)(pdbname, exename);
1025           vg_assert(pdbname[sz_exename+5-1] == 0);
1026           dot = VG_(strrchr)(pdbname, '.');
1027           if (!dot)
1028              goto out; /* there's no dot in the exe's name ?! */
1029           if (dot[1] == 0)
1030              goto out; /* hmm, path ends in "." */
1031     
1032           if ('A' <= dot[1] && dot[1] <= 'Z')
1033              VG_(strcpy)(dot, ".PDB");
1034           else
1035              VG_(strcpy)(dot, ".pdb");
1036     
1037           vg_assert(pdbname[sz_exename+5-1] == 0);
1038        }
1039     
1040        /* See if we can find it, and check it's in-dateness. */
1041        sres = VG_(stat)(pdbname, &stat_buf);
1042        if (sr_isError(sres)) {
1043           VG_(message)(Vg_UserMsg, "Warning: Missing or un-stat-able %s\n",
1044                                    pdbname);
1045        if (VG_(clo_verbosity) > 0)
1046           VG_(message)(Vg_UserMsg, "LOAD_PDB_DEBUGINFO: missing: %s\n", pdbname);
1047           goto out;
1048        }
1049        pdb_mtime = stat_buf.mtime;
1050     
1051        if (obj_mtime > pdb_mtime + 60ULL) {
1052           /* PDB file is older than PE file.  Really, the PDB should be
1053              newer than the PE, but that doesn't always seem to be the
1054              case.  Allow the PDB to be up to one minute older.
1055              Otherwise, it's probably out of date, in which case ignore it
1056              or we will either (a) print wrong stack traces or more likely
1057              (b) crash.
1058           */
1059           VG_(message)(Vg_UserMsg,
1060                        "Warning:       %s (mtime = %llu)\n"
1061                        " is older than %s (mtime = %llu)\n",
1062                        pdbname, pdb_mtime, exename, obj_mtime);
1063        }
1064     
1065        sres = VG_(open)(pdbname, VKI_O_RDONLY, 0);
1066        if (sr_isError(sres)) {
1067           VG_(message)(Vg_UserMsg, "Warning: Can't open %s\n", pdbname);
1068           goto out;
1069        }
1070     
1071        /* Looks promising; go on to try and read stuff from it.  But don't
1072           mmap the file.  Instead mmap free space and read the file into
1073           it.  This is because files on CIFS filesystems that are mounted
1074           '-o directio' can't be mmap'd, and that mount option is needed
1075           to make CIFS work reliably.  (See
1076           http://www.nabble.com/Corrupted-data-on-write-to-
1077                                 Windows-2003-Server-t2782623.html)
1078           This is slower, but at least it works reliably. */
1079        fd_pdbimage = sr_Res(sres);
1080        n_pdbimage  = stat_buf.size;
1081        if (n_pdbimage == 0 || n_pdbimage > 0x7FFFFFFF) {
1082           // 0x7FFFFFFF: why?  Because the VG_(read) just below only
1083           // can deal with a signed int as the size of data to read,
1084           // so we can't reliably check for read failure for files
1085           // greater than that size.  Hence just skip them; we're
1086           // unlikely to encounter a PDB that large anyway.
1087           VG_(close)(fd_pdbimage);
1088           goto out;
1089        }
1090        sres = VG_(am_mmap_anon_float_valgrind)( n_pdbimage );
1091        if (sr_isError(sres)) {
1092           VG_(close)(fd_pdbimage);
1093           goto out;
1094        }
1095     
1096        void* pdbimage = (void*)sr_Res(sres);
1097        r = VG_(read)( fd_pdbimage, pdbimage, (Int)n_pdbimage );
1098        if (r < 0 || r != (Int)n_pdbimage) {
1099           VG_(am_munmap_valgrind)( (Addr)pdbimage, n_pdbimage );
1100           VG_(close)(fd_pdbimage);
1101           goto out;
1102        }
1103     
1104        if (VG_(clo_verbosity) > 0)
1105           VG_(message)(Vg_UserMsg, "LOAD_PDB_DEBUGINFO: pdbname: %s\n", pdbname);
1106     
1107        /* play safe; always invalidate the CFI cache.  I don't know if
1108           this is necessary, but anyway .. */
1109        cfsi_cache__invalidate();
1110        /* dump old info for this range, if any */
1111        discard_syms_in_range( avma_obj, total_size );
1112     
1113        { DebugInfo* di = find_or_create_DebugInfo_for(exename, NULL/*membername*/ );
1114     
1115          /* this di must be new, since we just nuked any old stuff in the range */
1116          vg_assert(di && !di->have_rx_map && !di->have_rw_map);
1117          vg_assert(!di->have_dinfo);
1118     
1119          /* don't set up any of the di-> fields; let
1120             ML_(read_pdb_debug_info) do it. */
1121          ML_(read_pdb_debug_info)( di, avma_obj, unknown_purpose__reloc,
1122                                    pdbimage, n_pdbimage, pdbname, pdb_mtime );
1123          // JRS fixme: take notice of return value from read_pdb_debug_info,
1124          // and handle failure
1125          vg_assert(di->have_dinfo); // fails if PDB read failed
1126          VG_(am_munmap_valgrind)( (Addr)pdbimage, n_pdbimage );
1127          VG_(close)(fd_pdbimage);
1128     
1129          if (VG_(clo_verbosity) > 0) {
1130             VG_(message)(Vg_UserMsg, "LOAD_PDB_DEBUGINFO: done:    "
1131                                      "%lu syms, %lu src locs, %lu fpo recs\n",
1132                          di->symtab_used, di->loctab_used, di->fpo_size);
1133          }
1134        }
1135     
1136       out:
1137        if (pdbname) ML_(dinfo_free)(pdbname);
1138     }

1140     #endif /* defined(VGO_linux) || defined(VGO_darwin) */

1368     Bool get_sym_name ( Bool do_cxx_demangling, Bool do_z_demangling,
1369                         Bool do_below_main_renaming,
1370                         Addr a, Char* buf, Int nbuf,
1371                         Bool match_anywhere_in_sym, Bool show_offset,
1372                         Bool findText, /*OUT*/PtrdiffT* offsetP )
1373     {
1374        DebugInfo* di;
1375        Word       sno;
1376        PtrdiffT   offset;
1377     
1378        search_all_symtabs ( a, &di, &sno, match_anywhere_in_sym, findText );
1379        if (di == NULL) 
1380           return False;
1381     
1382        VG_(demangle) ( do_cxx_demangling, do_z_demangling,
1383                        di->symtab[sno].name, buf, nbuf );
1384     
1385        /* Do the below-main hack */
1386        // To reduce the endless nuisance of multiple different names 
1387        // for "the frame below main()" screwing up the testsuite, change all
1388        // known incarnations of said into a single name, "(below main)", if
1389        // --show-below-main=yes.
1390        if ( do_below_main_renaming && ! VG_(clo_show_below_main) &&
1391             Vg_FnNameBelowMain == VG_(get_fnname_kind)(buf) )
1392        {
1393           VG_(strncpy_safely)(buf, "(below main)", nbuf);
1394        }
1395        offset = a - di->symtab[sno].addr;
1396        if (offsetP) *offsetP = offset;
1397     
1398        if (show_offset && offset != 0) {
1399           Char     buf2[12];
1400           Char*    symend = buf + VG_(strlen)(buf);
1401           Char*    end = buf + nbuf;

1403     
1404           len = VG_(sprintf)(buf2, "%c%ld",
1405     			 offset < 0 ? '-' : '+',
1406     			 offset < 0 ? -offset : offset);

1410     	 Char *cp = buf2;
1411     	 VG_(memcpy)(symend, cp, len+1);
1412           }
1413        }
1414     
1415        buf[nbuf-1] = 0; /* paranoia */
1416     
1417        return True;

1748     /* Copy str into buf starting at n, but not going past buf[n_buf-1]
1749        and always ensuring that buf is zero-terminated. */

1752     {
1753        vg_assert(n_buf > 0);
1754        vg_assert(n >= 0 && n < n_buf);
1755        for (; n < n_buf-1 && *str != 0; n++,str++)

1762     /* Same as putStr, but escaping chars for XML output, and
1763        also not adding more than count chars to n_buf. */
1764     
1765     static Int putStrEsc ( Int n, Int n_buf, Int count, Char* buf, Char* str ) 
1766     {
1767        Char alt[2];
1768        vg_assert(n_buf > 0);
1769        vg_assert(count >= 0 && count < n_buf);
1770        vg_assert(n >= 0 && n < n_buf);
1771        for (; *str != 0; str++) {
1772           vg_assert(count >= 0);
1773           if (count <= 0)
1774              goto done;
1775           switch (*str) {
1776              case '&': 
1777                 if (count < 5) goto done;
1778                 n = putStr( n, n_buf, buf, "&amp;"); 
1779                 count -= 5;
1780                 break;
1781              case '<': 
1782                 if (count < 4) goto done;
1783                 n = putStr( n, n_buf, buf, "&lt;"); 
1784                 count -= 4;
1785                 break;
1786              case '>': 
1787                 if (count < 4) goto done;
1788                 n = putStr( n, n_buf, buf, "&gt;"); 
1789                 count -= 4;
1790                 break;
1791              default:
1792                 if (count < 1) goto done;
1793                 alt[0] = *str;
1794                 alt[1] = 0;
1795                 n = putStr( n, n_buf, buf, alt );
1796                 count -= 1;
1797                 break;
1798           }
1799        }
1800       done:
1801        vg_assert(count >= 0); /* should not go -ve in loop */
1802        vg_assert(n >= 0 && n < n_buf);
1803        return n;

1977     /* Evaluate the CfiExpr rooted at ix in exprs given the context eec.
1978        *ok is set to False on failure, but not to True on success.  The
1979        caller must set it to True before calling. */
1980     __attribute__((noinline))
1981     static
1982     UWord evalCfiExpr ( XArray* exprs, Int ix, 
1983                         CfiExprEvalContext* eec, Bool* ok )
1984     {
1985        UWord wL, wR;
1986        Addr  a;
1987        CfiExpr* e;
1988        vg_assert(sizeof(Addr) == sizeof(UWord));
1989        e = VG_(indexXA)( exprs, ix );
1990        switch (e->tag) {
1991           case Cex_Binop:
1992              wL = evalCfiExpr( exprs, e->Cex.Binop.ixL, eec, ok );
1993              if (!(*ok)) return 0;
1994              wR = evalCfiExpr( exprs, e->Cex.Binop.ixR, eec, ok );
1995              if (!(*ok)) return 0;
1996              switch (e->Cex.Binop.op) {
1997                 case Cop_Add: return wL + wR;
1998                 case Cop_Sub: return wL - wR;
1999                 case Cop_And: return wL & wR;
2000                 case Cop_Mul: return wL * wR;
2001                 default: goto unhandled;
2002              }
2003              /*NOTREACHED*/
2004           case Cex_CfiReg:
2005              switch (e->Cex.CfiReg.reg) {
2006     #           if defined(VGA_x86) || defined(VGA_amd64)
2007                 case Creg_IA_IP: return eec->uregs->xip;
2008                 case Creg_IA_SP: return eec->uregs->xsp;
2009                 case Creg_IA_BP: return eec->uregs->xbp;
2010     #           elif defined(VGA_arm)
2011                 case Creg_ARM_R15: return eec->uregs->r15;
2012                 case Creg_ARM_R14: return eec->uregs->r14;
2013                 case Creg_ARM_R13: return eec->uregs->r13;
2014                 case Creg_ARM_R12: return eec->uregs->r12;
2015     #           elif defined(VGA_s390x)
2016                 case Creg_IA_IP: return eec->uregs->ia;
2017                 case Creg_IA_SP: return eec->uregs->sp;
2018                 case Creg_IA_BP: return eec->uregs->fp;
2019                 case Creg_S390_R14: return eec->uregs->lr;
2020     #           elif defined(VGA_ppc32) || defined(VGA_ppc64)
2021     #           else
2022     #             error "Unsupported arch"
2023     #           endif
2024                 default: goto unhandled;
2025              }
2026              /*NOTREACHED*/
2027           case Cex_Const:
2028              return e->Cex.Const.con;
2029           case Cex_Deref:
2030              a = evalCfiExpr( exprs, e->Cex.Deref.ixAddr, eec, ok );
2031              if (!(*ok)) return 0;
2032              if (a < eec->min_accessible
2033                  || (a + sizeof(UWord) - 1) > eec->max_accessible) {
2034                 *ok = False;
2035                 return 0;
2036              }
2037              /* let's hope it doesn't trap! */
2038              return * ((UWord*)a);
2039           default: 
2040              goto unhandled;
2041        }
2042        /*NOTREACHED*/
2043       unhandled:
2044        VG_(printf)("\n\nevalCfiExpr: unhandled\n");
2045        ML_(ppCfiExpr)( exprs, ix );
2046        VG_(printf)("\n");
2047        vg_assert(0);
2048        /*NOTREACHED*/
2049        return 0;
2050     }

2053     /* Search all the DebugInfos in the entire system, to find the DiCfSI
2054        that pertains to 'ip'. 
2055     
2056        If found, set *diP to the DebugInfo in which it resides, and
2057        *ixP to the index in that DebugInfo's cfsi array.
2058     
2059        If not found, set *diP to (DebugInfo*)1 and *ixP to zero.
2060     */
2061     __attribute__((noinline))
2062     static void find_DiCfSI ( /*OUT*/DebugInfo** diP, 
2063                               /*OUT*/Word* ixP,
2064                               Addr ip )
2065     {
2066        DebugInfo* di;
2067        Word       i = -1;
2068     
2069        static UWord n_search = 0;
2070        static UWord n_steps = 0;
2071        n_search++;
2072     
2073        if (0) VG_(printf)("search for %#lx\n", ip);
2074     
2075        for (di = debugInfo_list; di != NULL; di = di->next) {
2076           Word j;
2077           n_steps++;
2078     
2079           /* Use the per-DebugInfo summary address ranges to skip
2080              inapplicable DebugInfos quickly. */
2081           if (di->cfsi_used == 0)
2082              continue;
2083           if (ip < di->cfsi_minavma || ip > di->cfsi_maxavma)
2084              continue;
2085     
2086           /* It might be in this DebugInfo.  Search it. */
2087           j = ML_(search_one_cfitab)( di, ip );
2088           vg_assert(j >= -1 && j < (Word)di->cfsi_used);
2089     
2090           if (j != -1) {
2091              i = j;
2092              break; /* found it */
2093           }
2094        }
2095     
2096        if (i == -1) {
2097     
2098           /* we didn't find it. */
2099           *diP = (DebugInfo*)1;
2100           *ixP = 0;
2101     
2102        } else {
2103     
2104           /* found it. */
2105           /* ensure that di is 4-aligned (at least), so it can't possibly
2106              be equal to (DebugInfo*)1. */
2107           vg_assert(di && VG_IS_4_ALIGNED(di));
2108           vg_assert(i >= 0 && i < di->cfsi_used);
2109           *diP = di;
2110           *ixP = i;
2111     
2112           /* Start of performance-enhancing hack: once every 64 (chosen
2113              hackily after profiling) successful searches, move the found
2114              DebugInfo one step closer to the start of the list.  This
2115              makes future searches cheaper.  For starting konqueror on
2116              amd64, this in fact reduces the total amount of searching
2117              done by the above find-the-right-DebugInfo loop by more than
2118              a factor of 20. */
2119           if ((n_search & 0xF) == 0) {
2120              /* Move di one step closer to the start of the list. */
2121              move_DebugInfo_one_step_forward( di );
2122           }
2123           /* End of performance-enhancing hack. */
2124     
2125           if (0 && ((n_search & 0x7FFFF) == 0))
2126              VG_(printf)("find_DiCfSI: %lu searches, "
2127                          "%lu DebugInfos looked at\n", 
2128                          n_search, n_steps);
2129     
2130        }
2131     
2132     }

2135     /* Now follows a mechanism for caching queries to find_DiCfSI, since
2136        they are extremely frequent on amd64-linux, during stack unwinding.
2137     
2138        Each cache entry binds an ip value to a (di, ix) pair.  Possible
2139        values:
2140     
2141        di is non-null, ix >= 0  ==>  cache slot in use, "di->cfsi[ix]"
2142        di is (DebugInfo*)1      ==>  cache slot in use, no associated di
2143        di is NULL               ==>  cache slot not in use
2144     
2145        Hence simply zeroing out the entire cache invalidates all
2146        entries.
2147     
2148        Why not map ip values directly to DiCfSI*'s?  Because this would
2149        cause problems if/when the cfsi array is moved due to resizing.
2150        Instead we cache .cfsi array index value, which should be invariant
2151        across resizing.  (That said, I don't think the current
2152        implementation will resize whilst during queries, since the DiCfSI
2153        records are added all at once, when the debuginfo for an object is
2154        read, and is not changed ever thereafter. */
2155     
2156     #define N_CFSI_CACHE 511
2157     
2158     typedef
2159        struct { Addr ip; DebugInfo* di; Word ix; }
2160        CFSICacheEnt;
2161     
2162     static CFSICacheEnt cfsi_cache[N_CFSI_CACHE];
2163     
2164     static void cfsi_cache__invalidate ( void ) {
2165        VG_(memset)(&cfsi_cache, 0, sizeof(cfsi_cache));
2166     }

2169     static inline CFSICacheEnt* cfsi_cache__find ( Addr ip )
2170     {
2171        UWord         hash = ip % N_CFSI_CACHE;
2172        CFSICacheEnt* ce = &cfsi_cache[hash];
2173        static UWord  n_q = 0, n_m = 0;
2174     
2175        n_q++;

2177           VG_(printf)("QQQ %lu %lu\n", n_q, n_m);
2178     
2179        if (LIKELY(ce->ip == ip) && LIKELY(ce->di != NULL)) {
2180           /* found an entry in the cache .. */

2183           n_m++;
2184           ce->ip = ip;

2186        }
2187     
2188        if (UNLIKELY(ce->di == (DebugInfo*)1)) {

2264              break;
2265           default:

2267        }
2268        return cfa;
2269     }

2441                              Addr max_accessible )

2443        Word       i;

2446        Addr       spHere;
2447     
2448        static UWord n_search = 0;
2449        static UWord n_steps = 0;
2450        n_search++;
2451     
2452        if (0) VG_(printf)("search FPO for %#lx\n", *ipP);
2453     
2454        for (di = debugInfo_list; di != NULL; di = di->next) {
2455           n_steps++;
2456     
2457           /* Use the per-DebugInfo summary address ranges to skip
2458              inapplicable DebugInfos quickly. */
2459           if (di->fpo == NULL)
2460              continue;
2461           if (*ipP < di->fpo_minavma || *ipP > di->fpo_maxavma)
2462              continue;
2463     
2464           i = ML_(search_one_fpotab)( di, *ipP );
2465           if (i != -1) {
2466              Word j;
2467              if (0) {
2468                 /* debug printing only */
2469                 VG_(printf)("look for %#lx  size %ld i %ld\n",
2470                             *ipP, di->fpo_size, i);
2471                 for (j = 0; j < di->fpo_size; j++)
2472                    VG_(printf)("[%02ld] %#x %d\n", 
2473                                 j, di->fpo[j].ulOffStart, di->fpo[j].cbProcSize);
2474              }
2475              vg_assert(i >= 0 && i < di->fpo_size);
2476              fpo = &di->fpo[i];
2477              break;
2478           }
2479        }
2480     
2481        if (fpo == NULL)
2482           return False;
2483     
2484        if (0 && ((n_search & 0x7FFFF) == 0))
2485           VG_(printf)("VG_(use_FPO_info): %lu searches, "
2486                       "%lu DebugInfos looked at\n",
2487                       n_search, n_steps);
2488     
2489     
2490        /* Start of performance-enhancing hack: once every 64 (chosen
2491           hackily after profiling) successful searches, move the found
2492           DebugInfo one step closer to the start of the list.  This makes
2493           future searches cheaper.  For starting konqueror on amd64, this
2494           in fact reduces the total amount of searching done by the above
2495           find-the-right-DebugInfo loop by more than a factor of 20. */
2496        if ((n_search & 0x3F) == 0) {
2497           /* Move si one step closer to the start of the list. */
2498           //move_DebugInfo_one_step_forward( di );
2499        }

2501     
2502        if (0) {
2503           VG_(printf)("found fpo: ");
2504           //ML_(ppFPO)(fpo);
2505        }
2506     
2507        /*
2508        Stack layout is:
2509        %esp->
2510           4*.cbRegs  {%edi, %esi, %ebp, %ebx}
2511           4*.cdwLocals
2512           return_pc
2513           4*.cdwParams
2514        prior_%esp->
2515     
2516        Typical code looks like:
2517           sub $4*.cdwLocals,%esp
2518              Alternative to above for >=4KB (and sometimes for smaller):
2519                 mov $size,%eax
2520                 call __chkstk  # WinNT performs page-by-page probe!
2521                    __chkstk is much like alloc(), except that on return
2522                    %eax= 5+ &CALL.  Thus it could be used as part of
2523                    Position Independent Code to locate the Global Offset Table.
2524           push %ebx
2525           push %ebp
2526           push %esi
2527              Other once-only instructions often scheduled >here<.
2528           push %edi
2529     
2530        If the pc is within the first .cbProlog bytes of the function,
2531        then you must disassemble to see how many registers have been pushed,
2532        because instructions in the prolog may be scheduled for performance.
2533        The order of PUSH is always %ebx, %ebp, %esi, %edi, with trailing
2534        registers not pushed when .cbRegs < 4.  This seems somewhat strange
2535        because %ebp is the register whose usage you want to minimize,
2536        yet it is in the first half of the PUSH list.
2537     
2538        I don't know what happens when the compiler constructs an outgoing CALL.
2539        %esp could move if outgoing parameters are PUSHed, and this affects
2540        traceback for errors during the PUSHes. */
2541      
2542        spHere = *spP;
2543     
2544        *ipP = *(Addr *)(spHere + 4*(fpo->cbRegs + fpo->cdwLocals));
2545        *spP =           spHere + 4*(fpo->cbRegs + fpo->cdwLocals + 1 
2546                                                 + fpo->cdwParams);
2547        *fpP = *(Addr *)(spHere + 4*2);
2548        return True;
2549     }

2875                    data_addr, residual_offset, ro_plural, var->name,
2876                    (char*)(VG_(indexXA)(described,0)) );

2878                    "a global variable");
2879           }

2881        else
2882        if ( frameNo >= -1 && have_srcloc && have_descr ) {
2883           /* Location 0x7fefff6cf is 2 bytes inside a[3].xyzzy[21].c2,
2884              a global variable declared at dsyms7.c:17 */
2885           if (xml) {

2888                    "Location 0x%lx is %lu byte%s inside %t%t,",
2889                    data_addr, residual_offset, ro_plural, var->name,
2890                    (HChar*)(VG_(indexXA)(described,0)) );

2892              XAGL( dn2 );
2893              TXTL( dn2 );
2894              p2XA( dn2,
2895                    "a global variable declared at %t:%d",
2896                    var->fileName, var->lineNo);
2897              TXTR( dn2 );
2898              // FIXME: also do <dir>
2899              p2XA( dn2,
2900                    " <file>%t</file> <line>%d</line> ",
2901                    var->fileName, var->lineNo );
2902              XAGR( dn2 );
2903           } else {
2904              p2XA( dn1,
2905                    "Location 0x%lx is %lu byte%s inside %s%s,",
2906                    data_addr, residual_offset, ro_plural, var->name,
2907                    (HChar*)(VG_(indexXA)(described,0)) );
2908              p2XA( dn2,
2909                    "a global variable declared at %s:%d",
2910                    var->fileName, var->lineNo);
2911           }
2912        }
2913        else 
2914           vg_assert(0);
2915     
2916        /* Zero terminate both strings */
2917        zterm_XA( dn1 );

2929     /* Determine if data_addr is a local variable in the frame
2930        characterised by (ip,sp,fp), and if so write its description at the
2931        ends of DNAME{1,2}, which are XArray*s of HChar, that have been
2932        initialised by the caller, zero terminate both, and return True.
2933        If it's not a local variable in said frame, return False. */
2934     static 
2935     Bool consider_vars_in_frame ( /*MOD*/XArray* /* of HChar */ dname1,
2936                                   /*MOD*/XArray* /* of HChar */ dname2,
2937                                   Addr data_addr,
2938                                   Addr ip, Addr sp, Addr fp,
2939                                   /* shown to user: */
2940                                   ThreadId tid, Int frameNo )
2941     {
2942        Word       i;
2943        DebugInfo* di;
2944        RegSummary regs;
2945        Bool debug = False;
2946     
2947        static UInt n_search = 0;
2948        static UInt n_steps = 0;
2949        n_search++;
2950        if (debug)
2951           VG_(printf)("QQQQ: cvif: ip,sp,fp %#lx,%#lx,%#lx\n", ip,sp,fp);
2952        /* first, find the DebugInfo that pertains to 'ip'. */
2953        for (di = debugInfo_list; di; di = di->next) {
2954           n_steps++;
2955           /* text segment missing? unlikely, but handle it .. */
2956           if (!di->text_present || di->text_size == 0)
2957              continue;
2958           /* Ok.  So does this text mapping bracket the ip? */
2959           if (di->text_avma <= ip && ip < di->text_avma + di->text_size)
2960              break;
2961        }
2962      
2963        /* Didn't find it.  Strange -- means ip is a code address outside
2964           of any mapped text segment.  Unlikely but not impossible -- app
2965           could be generating code to run. */
2966        if (!di)
2967           return False;
2968     
2969        if (0 && ((n_search & 0x1) == 0))
2970           VG_(printf)("consider_vars_in_frame: %u searches, "
2971                       "%u DebugInfos looked at\n", 
2972                       n_search, n_steps);
2973        /* Start of performance-enhancing hack: once every ??? (chosen
2974           hackily after profiling) successful searches, move the found
2975           DebugInfo one step closer to the start of the list.  This makes
2976           future searches cheaper. */
2977        if ((n_search & 0xFFFF) == 0) {
2978           /* Move si one step closer to the start of the list. */
2979           move_DebugInfo_one_step_forward( di );
2980        }
2981        /* End of performance-enhancing hack. */
2982     
2983        /* any var info at all? */
2984        if (!di->varinfo)
2985           return False;
2986     
2987        /* Work through the scopes from most deeply nested outwards,
2988           looking for code address ranges that bracket 'ip'.  The
2989           variables on each such address range found are in scope right
2990           now.  Don't descend to level zero as that is the global
2991           scope. */
2992        regs.ip = ip;
2993        regs.sp = sp;
2994        regs.fp = fp;
2995     
2996        /* "for each scope, working outwards ..." */
2997        for (i = VG_(sizeXA)(di->varinfo) - 1; i >= 1; i--) {
2998           XArray*      vars;
2999           Word         j;
3000           DiAddrRange* arange;
3001           OSet*        this_scope 
3002              = *(OSet**)VG_(indexXA)( di->varinfo, i );
3003           if (debug)
3004              VG_(printf)("QQQQ:   considering scope %ld\n", (Word)i);
3005           if (!this_scope)
3006              continue;
3007           /* Find the set of variables in this scope that
3008              bracket the program counter. */
3009           arange = VG_(OSetGen_LookupWithCmp)(
3010                       this_scope, &ip, 
3011                       ML_(cmp_for_DiAddrRange_range)
3012                    );
3013           if (!arange)
3014              continue;
3015           /* stay sane */
3016           vg_assert(arange->aMin <= arange->aMax);
3017           /* It must bracket the ip we asked for, else
3018              ML_(cmp_for_DiAddrRange_range) is somehow broken. */
3019           vg_assert(arange->aMin <= ip && ip <= arange->aMax);
3020           /* It must have an attached XArray of DiVariables. */
3021           vars = arange->vars;
3022           vg_assert(vars);
3023           /* But it mustn't cover the entire address range.  We only
3024              expect that to happen for the global scope (level 0), which
3025              we're not looking at here.  Except, it may cover the entire
3026              address range, but in that case the vars array must be
3027              empty. */
3028           vg_assert(! (arange->aMin == (Addr)0
3029                        && arange->aMax == ~(Addr)0
3030                        && VG_(sizeXA)(vars) > 0) );
3031           for (j = 0; j < VG_(sizeXA)( vars ); j++) {
3032              DiVariable* var = (DiVariable*)VG_(indexXA)( vars, j );
3033              PtrdiffT    offset;
3034              if (debug)
3035                 VG_(printf)("QQQQ:    var:name=%s %#lx-%#lx %#lx\n",
3036                             var->name,arange->aMin,arange->aMax,ip);
3037              if (data_address_is_in_var( &offset, di->admin_tyents,
3038                                          var, &regs,
3039                                          data_addr, di )) {
3040                 PtrdiffT residual_offset = 0;
3041                 XArray* described = ML_(describe_type)( &residual_offset,
3042                                                         di->admin_tyents, 
3043                                                         var->typeR, offset );
3044                 format_message( dname1, dname2,
3045                                 data_addr, var, offset, residual_offset,
3046                                 described, frameNo, tid );
3047                 VG_(deleteXA)( described );
3048                 return True;
3049              }
3050           }
3051        }
3052     
3053        return False;
3054     }

3056     /* Try to form some description of DATA_ADDR by looking at the DWARF3
3057        debug info we have.  This considers all global variables, and all
3058        frames in the stacks of all threads.  Result is written at the ends
3059        of DNAME{1,2}V, which are XArray*s of HChar, that have been
3060        initialised by the caller, and True is returned.  If no description
3061        is created, False is returned.  Regardless of the return value,
3062        DNAME{1,2}V are guaranteed to be zero terminated after the call.
3063     
3064        Note that after the call, DNAME{1,2} may have more than one
3065        trailing zero, so callers should establish the useful text length
3066        using VG_(strlen) on the contents, rather than VG_(sizeXA) on the
3067        XArray itself.
3068     */
3069     Bool VG_(get_data_description)( 
3070             /*MOD*/ void* /* really, XArray* of HChar */ dname1v,
3071             /*MOD*/ void* /* really, XArray* of HChar */ dname2v,
3072             Addr data_addr
3073          )
3074     {
3075     #  define N_FRAMES 8
3076        Addr ips[N_FRAMES], sps[N_FRAMES], fps[N_FRAMES];
3077        UInt n_frames;
3078     
3079        Addr       stack_min, stack_max;
3080        ThreadId   tid;
3081        Bool       found;
3082        DebugInfo* di;
3083        Word       j;
3084     
3085        XArray*    dname1 = (XArray*)dname1v;
3086        XArray*    dname2 = (XArray*)dname2v;
3087     
3088        if (0) VG_(printf)("get_data_description: dataaddr %#lx\n", data_addr);
3089        /* First, see if data_addr is (or is part of) a global variable.
3090           Loop over the DebugInfos we have.  Check data_addr against the
3091           outermost scope of all of them, as that should be a global
3092           scope. */
3093        for (di = debugInfo_list; di != NULL; di = di->next) {
3094           OSet*        global_scope;
3095           Word         gs_size;
3096           Addr         zero;
3097           DiAddrRange* global_arange;
3098           Word         i;
3099           XArray*      vars;
3100     
3101           /* text segment missing? unlikely, but handle it .. */
3102           if (!di->text_present || di->text_size == 0)
3103              continue;
3104           /* any var info at all? */
3105           if (!di->varinfo)
3106              continue;
3107           /* perhaps this object didn't contribute any vars at all? */
3108           if (VG_(sizeXA)( di->varinfo ) == 0)
3109              continue;
3110           global_scope = *(OSet**)VG_(indexXA)( di->varinfo, 0 );
3111           vg_assert(global_scope);
3112           gs_size = VG_(OSetGen_Size)( global_scope );
3113           /* The global scope might be completely empty if this
3114              compilation unit declared locals but nothing global. */
3115           if (gs_size == 0)
3116               continue;
3117           /* But if it isn't empty, then it must contain exactly one
3118              element, which covers the entire address range. */
3119           vg_assert(gs_size == 1);
3120           /* Fish out the global scope and check it is as expected. */
3121           zero = 0;
3122           global_arange 
3123              = VG_(OSetGen_Lookup)( global_scope, &zero );
3124           /* The global range from (Addr)0 to ~(Addr)0 must exist */
3125           vg_assert(global_arange);
3126           vg_assert(global_arange->aMin == (Addr)0
3127                     && global_arange->aMax == ~(Addr)0);
3128           /* Any vars in this range? */
3129           if (!global_arange->vars)
3130              continue;
3131           /* Ok, there are some vars in the global scope of this
3132              DebugInfo.  Wade through them and see if the data addresses
3133              of any of them bracket data_addr. */
3134           vars = global_arange->vars;
3135           for (i = 0; i < VG_(sizeXA)( vars ); i++) {
3136              PtrdiffT offset;
3137              DiVariable* var = (DiVariable*)VG_(indexXA)( vars, i );
3138              vg_assert(var->name);
3139              /* Note we use a NULL RegSummary* here.  It can't make any
3140                 sense for a global variable to have a location expression
3141                 which depends on a SP/FP/IP value.  So don't supply any.
3142                 This means, if the evaluation of the location
3143                 expression/list requires a register, we have to let it
3144                 fail. */
3145              if (data_address_is_in_var( &offset, di->admin_tyents, var, 
3146                                          NULL/* RegSummary* */, 
3147                                          data_addr, di )) {
3148                 PtrdiffT residual_offset = 0;
3149                 XArray* described = ML_(describe_type)( &residual_offset,
3150                                                         di->admin_tyents,
3151                                                         var->typeR, offset );
3152                 format_message( dname1, dname2,
3153                                 data_addr, var, offset, residual_offset,
3154                                 described, -1/*frameNo*/, tid );
3155                 VG_(deleteXA)( described );
3156                 zterm_XA( dname1 );
3157                 zterm_XA( dname2 );
3158                 return True;
3159              }
3160           }
3161        }
3162     
3163        /* Ok, well it's not a global variable.  So now let's snoop around
3164           in the stacks of all the threads.  First try to figure out which
3165           thread's stack data_addr is in. */
3166     
3167        /* --- KLUDGE --- Try examining the top frame of all thread stacks.
3168           This finds variables which are not stack allocated but are not
3169           globally visible either; specifically it appears to pick up
3170           variables which are visible only within a compilation unit.
3171           These will have the address range of the compilation unit and
3172           tend to live at Scope level 1. */
3173        VG_(thread_stack_reset_iter)(&tid);
3174        while ( VG_(thread_stack_next)(&tid, &stack_min, &stack_max) ) {
3175           if (stack_min >= stack_max)
3176              continue; /* ignore obviously stupid cases */
3177           if (consider_vars_in_frame( dname1, dname2,
3178                                       data_addr,
3179                                       VG_(get_IP)(tid),
3180                                       VG_(get_SP)(tid), 
3181                                       VG_(get_FP)(tid), tid, 0 )) {
3182              zterm_XA( dname1 );
3183              zterm_XA( dname2 );
3184              return True;
3185           }
3186        }
3187        /* --- end KLUDGE --- */
3188     
3189        /* Perhaps it's on a thread's stack? */
3190        found = False;
3191        VG_(thread_stack_reset_iter)(&tid);
3192        while ( VG_(thread_stack_next)(&tid, &stack_min, &stack_max) ) {
3193           if (stack_min >= stack_max)
3194              continue; /* ignore obviously stupid cases */
3195           if (stack_min - VG_STACK_REDZONE_SZB <= data_addr
3196               && data_addr <= stack_max) {
3197              found = True;
3198              break;
3199           }
3200        }
3201        if (!found) {
3202           zterm_XA( dname1 );
3203           zterm_XA( dname2 );
3204           return False;
3205        }
3206     
3207        /* We conclude data_addr is in thread tid's stack.  Unwind the
3208           stack to get a bunch of (ip,sp,fp) triples describing the
3209           frames, and for each frame, consider the local variables. */
3210        n_frames = VG_(get_StackTrace)( tid, ips, N_FRAMES,
3211                                        sps, fps, 0/*first_ip_delta*/ );
3212     
3213        /* As a result of KLUDGE above, starting the loop at j = 0
3214           duplicates examination of the top frame and so isn't necessary.
3215           Oh well. */
3216        vg_assert(n_frames >= 0 && n_frames <= N_FRAMES);
3217        for (j = 0; j < n_frames; j++) {
3218           if (consider_vars_in_frame( dname1, dname2,
3219                                       data_addr,
3220                                       ips[j], 
3221                                       sps[j], fps[j], tid, j )) {
3222              zterm_XA( dname1 );
3223              zterm_XA( dname2 );
3224              return True;
3225           }
3226           /* Now, it appears that gcc sometimes appears to produce
3227              location lists whose ranges don't actually cover the call
3228              instruction, even though the address of the variable in
3229              question is passed as a parameter in the call.  AFAICS this
3230              is simply a bug in gcc - how can the variable be claimed not
3231              exist in memory (on the stack) for the duration of a call in
3232              which its address is passed?  But anyway, in the particular
3233              case I investigated (memcheck/tests/varinfo6.c, call to croak
3234              on line 2999, local var budget declared at line 3115
3235              appearing not to exist across the call to mainSort on line
3236              3143, "gcc.orig (GCC) 3.4.4 20050721 (Red Hat 3.4.4-2)" on
3237              amd64), the variable's location list does claim it exists
3238              starting at the first byte of the first instruction after the
3239              call instruction.  So, call consider_vars_in_frame a second
3240              time, but this time add 1 to the IP.  GDB handles this
3241              example with no difficulty, which leads me to believe that
3242              either (1) I misunderstood something, or (2) GDB has an
3243              equivalent kludge. */
3244           if (j > 0 /* this is a non-innermost frame */
3245               && consider_vars_in_frame( dname1, dname2,
3246                                          data_addr,
3247                                          ips[j] + 1, 
3248                                          sps[j], fps[j], tid, j )) {
3249              zterm_XA( dname1 );
3250              zterm_XA( dname2 );
3251              return True;
3252           }
3253        }
3254     
3255        /* We didn't find anything useful. */
3256        zterm_XA( dname1 );
3257        zterm_XA( dname2 );
3258        return False;
3259     #  undef N_FRAMES
3260     }

3269     /* Figure out if the variable 'var' has a location that is linearly
3270        dependent on a stack pointer value, or a frame pointer value, and
3271        if it is, add a description of it to 'blocks'.  Otherwise ignore
3272        it.  If 'arrays_only' is True, also ignore it unless it has an
3273        array type. */
3274     
3275     static 
3276     void analyse_deps ( /*MOD*/XArray* /* of FrameBlock */ blocks,
3277                         XArray* /* TyEnt */ tyents,
3278                         Addr ip, const DebugInfo* di, DiVariable* var,
3279                         Bool arrays_only )
3280     {
3281        GXResult   res_sp_6k, res_sp_7k, res_fp_6k, res_fp_7k;
3282        RegSummary regs;
3283        MaybeULong mul;
3284        Bool       isVec;
3285        TyEnt*     ty;
3286     
3287        Bool debug = False;
3288        if (0&&debug)
3289           VG_(printf)("adeps: var %s\n", var->name );
3290     
3291        /* Figure out how big the variable is. */
3292        mul = ML_(sizeOfType)(tyents, var->typeR);
3293        /* If this var has a type whose size is unknown, zero, or
3294           impossibly large, it should never have been added.  ML_(addVar)
3295           should have rejected it. */
3296        vg_assert(mul.b == True);
3297        vg_assert(mul.ul > 0);
3298        if (sizeof(void*) == 4) vg_assert(mul.ul < (1ULL << 32));
3299        /* After this point, we assume we can truncate mul.ul to a host word
3300           safely (without loss of info). */
3301     
3302        /* skip if non-array and we're only interested in arrays */
3303        ty = ML_(TyEnts__index_by_cuOff)( tyents, NULL, var->typeR );
3304        vg_assert(ty);
3305        vg_assert(ty->tag == Te_UNKNOWN || ML_(TyEnt__is_type)(ty));
3306        if (ty->tag == Te_UNKNOWN)
3307           return; /* perhaps we should complain in this case? */
3308        isVec = ty->tag == Te_TyArray;
3309        if (arrays_only && !isVec)
3310           return;
3311     
3312        if (0) {ML_(pp_TyEnt_C_ishly)(tyents, var->typeR);
3313                VG_(printf)("  %s\n", var->name);}
3314     
3315        /* Do some test evaluations of the variable's location expression,
3316           in order to guess whether it is sp-relative, fp-relative, or
3317           none.  A crude hack, which can be interpreted roughly as finding
3318           the first derivative of the location expression w.r.t. the
3319           supplied frame and stack pointer values. */
3320        regs.fp   = 0;
3321        regs.ip   = ip;
3322        regs.sp   = 6 * 1024;
3323        res_sp_6k = ML_(evaluate_GX)( var->gexpr, var->fbGX, &regs, di );
3324     
3325        regs.fp   = 0;
3326        regs.ip   = ip;
3327        regs.sp   = 7 * 1024;
3328        res_sp_7k = ML_(evaluate_GX)( var->gexpr, var->fbGX, &regs, di );
3329     
3330        regs.fp   = 6 * 1024;
3331        regs.ip   = ip;
3332        regs.sp   = 0;
3333        res_fp_6k = ML_(evaluate_GX)( var->gexpr, var->fbGX, &regs, di );
3334     
3335        regs.fp   = 7 * 1024;
3336        regs.ip   = ip;
3337        regs.sp   = 0;
3338        res_fp_7k = ML_(evaluate_GX)( var->gexpr, var->fbGX, &regs, di );
3339     
3340        vg_assert(res_sp_6k.kind == res_sp_7k.kind);
3341        vg_assert(res_sp_6k.kind == res_fp_6k.kind);
3342        vg_assert(res_sp_6k.kind == res_fp_7k.kind);
3343     
3344        if (res_sp_6k.kind == GXR_Addr) {
3345           StackBlock block;
3346           GXResult res;
3347           UWord sp_delta = res_sp_7k.word - res_sp_6k.word;
3348           UWord fp_delta = res_fp_7k.word - res_fp_6k.word;
3349           tl_assert(sp_delta == 0 || sp_delta == 1024);
3350           tl_assert(fp_delta == 0 || fp_delta == 1024);
3351     
3352           if (sp_delta == 0 && fp_delta == 0) {
3353              /* depends neither on sp nor fp, so it can't be a stack
3354                 local.  Ignore it. */
3355           }
3356           else
3357           if (sp_delta == 1024 && fp_delta == 0) {
3358              regs.sp = regs.fp = 0;
3359              regs.ip = ip;
3360              res = ML_(evaluate_GX)( var->gexpr, var->fbGX, &regs, di );
3361              tl_assert(res.kind == GXR_Addr);
3362              if (debug)
3363              VG_(printf)("   %5ld .. %5ld (sp) %s\n",
3364                          res.word, res.word + ((UWord)mul.ul) - 1, var->name);
3365              block.base  = res.word;
3366              block.szB   = (SizeT)mul.ul;
3367              block.spRel = True;
3368              block.isVec = isVec;
3369              VG_(memset)( &block.name[0], 0, sizeof(block.name) );
3370              if (var->name)
3371                 VG_(strncpy)( &block.name[0], var->name, sizeof(block.name)-1 );
3372              block.name[ sizeof(block.name)-1 ] = 0;
3373              VG_(addToXA)( blocks, &block );
3374           }
3375           else
3376           if (sp_delta == 0 && fp_delta == 1024) {
3377              regs.sp = regs.fp = 0;
3378              regs.ip = ip;
3379              res = ML_(evaluate_GX)( var->gexpr, var->fbGX, &regs, di );
3380              tl_assert(res.kind == GXR_Addr);
3381              if (debug)
3382              VG_(printf)("   %5ld .. %5ld (FP) %s\n",
3383                          res.word, res.word + ((UWord)mul.ul) - 1, var->name);
3384              block.base  = res.word;
3385              block.szB   = (SizeT)mul.ul;
3386              block.spRel = False;
3387              block.isVec = isVec;
3388              VG_(memset)( &block.name[0], 0, sizeof(block.name) );
3389              if (var->name)
3390                 VG_(strncpy)( &block.name[0], var->name, sizeof(block.name)-1 );
3391              block.name[ sizeof(block.name)-1 ] = 0;
3392              VG_(addToXA)( blocks, &block );
3393           }
3394           else {
3395              vg_assert(0);
3396           }
3397        }
3398     }

3401     /* Get an XArray of StackBlock which describe the stack (auto) blocks
3402        for this ip.  The caller is expected to free the XArray at some
3403        point.  If 'arrays_only' is True, only array-typed blocks are
3404        returned; otherwise blocks of all types are returned. */
3405     
3406     void* /* really, XArray* of StackBlock */
3407           VG_(di_get_stack_blocks_at_ip)( Addr ip, Bool arrays_only )
3408     {
3409        /* This is a derivation of consider_vars_in_frame() above. */
3410        Word       i;
3411        DebugInfo* di;
3412        Bool debug = False;
3413     
3414        XArray* res = VG_(newXA)( ML_(dinfo_zalloc), "di.debuginfo.dgsbai.1",
3415                                  ML_(dinfo_free),
3416                                  sizeof(StackBlock) );
3417     
3418        static UInt n_search = 0;
3419        static UInt n_steps = 0;
3420        n_search++;
3421        if (debug)
3422           VG_(printf)("QQQQ: dgsbai: ip %#lx\n", ip);
3423        /* first, find the DebugInfo that pertains to 'ip'. */
3424        for (di = debugInfo_list; di; di = di->next) {
3425           n_steps++;
3426           /* text segment missing? unlikely, but handle it .. */
3427           if (!di->text_present || di->text_size == 0)
3428              continue;
3429           /* Ok.  So does this text mapping bracket the ip? */
3430           if (di->text_avma <= ip && ip < di->text_avma + di->text_size)
3431              break;
3432        }
3433      
3434        /* Didn't find it.  Strange -- means ip is a code address outside
3435           of any mapped text segment.  Unlikely but not impossible -- app
3436           could be generating code to run. */
3437        if (!di)
3438           return res; /* currently empty */
3439     
3440        if (0 && ((n_search & 0x1) == 0))
3441           VG_(printf)("VG_(di_get_stack_blocks_at_ip): %u searches, "
3442                       "%u DebugInfos looked at\n", 
3443                       n_search, n_steps);
3444        /* Start of performance-enhancing hack: once every ??? (chosen
3445           hackily after profiling) successful searches, move the found
3446           DebugInfo one step closer to the start of the list.  This makes
3447           future searches cheaper. */
3448        if ((n_search & 0xFFFF) == 0) {
3449           /* Move si one step closer to the start of the list. */
3450           move_DebugInfo_one_step_forward( di );
3451        }
3452        /* End of performance-enhancing hack. */
3453     
3454        /* any var info at all? */
3455        if (!di->varinfo)
3456           return res; /* currently empty */
3457     
3458        /* Work through the scopes from most deeply nested outwards,
3459           looking for code address ranges that bracket 'ip'.  The
3460           variables on each such address range found are in scope right
3461           now.  Don't descend to level zero as that is the global
3462           scope. */
3463     
3464        /* "for each scope, working outwards ..." */
3465        for (i = VG_(sizeXA)(di->varinfo) - 1; i >= 1; i--) {
3466           XArray*      vars;
3467           Word         j;
3468           DiAddrRange* arange;
3469           OSet*        this_scope 
3470              = *(OSet**)VG_(indexXA)( di->varinfo, i );
3471           if (debug)
3472              VG_(printf)("QQQQ:   considering scope %ld\n", (Word)i);
3473           if (!this_scope)
3474              continue;
3475           /* Find the set of variables in this scope that
3476              bracket the program counter. */
3477           arange = VG_(OSetGen_LookupWithCmp)(
3478                       this_scope, &ip, 
3479                       ML_(cmp_for_DiAddrRange_range)
3480                    );
3481           if (!arange)
3482              continue;
3483           /* stay sane */
3484           vg_assert(arange->aMin <= arange->aMax);
3485           /* It must bracket the ip we asked for, else
3486              ML_(cmp_for_DiAddrRange_range) is somehow broken. */
3487           vg_assert(arange->aMin <= ip && ip <= arange->aMax);
3488           /* It must have an attached XArray of DiVariables. */
3489           vars = arange->vars;
3490           vg_assert(vars);
3491           /* But it mustn't cover the entire address range.  We only
3492              expect that to happen for the global scope (level 0), which
3493              we're not looking at here.  Except, it may cover the entire
3494              address range, but in that case the vars array must be
3495              empty. */
3496           vg_assert(! (arange->aMin == (Addr)0
3497                        && arange->aMax == ~(Addr)0
3498                        && VG_(sizeXA)(vars) > 0) );
3499           for (j = 0; j < VG_(sizeXA)( vars ); j++) {
3500              DiVariable* var = (DiVariable*)VG_(indexXA)( vars, j );
3501              if (debug)
3502                 VG_(printf)("QQQQ:    var:name=%s %#lx-%#lx %#lx\n", 
3503                             var->name,arange->aMin,arange->aMax,ip);
3504              analyse_deps( res, di->admin_tyents, ip,
3505                            di, var, arrays_only );
3506           }
3507        }
3508     
3509        return res;
3510     }

3513     /* Get an array of GlobalBlock which describe the global blocks owned
3514        by the shared object characterised by the given di_handle.  Asserts
3515        if the handle is invalid.  The caller is responsible for freeing
3516        the array at some point.  If 'arrays_only' is True, only
3517        array-typed blocks are returned; otherwise blocks of all types are
3518        returned. */
3519     
3520     void* /* really, XArray* of GlobalBlock */
3521           VG_(di_get_global_blocks_from_dihandle) ( ULong di_handle,
3522                                                     Bool  arrays_only )
3523     {
3524        /* This is a derivation of consider_vars_in_frame() above. */
3525     
3526        DebugInfo* di;
3527        XArray* gvars; /* XArray* of GlobalBlock */
3528        Word nScopes, scopeIx;
3529     
3530        /* The first thing to do is find the DebugInfo that
3531           pertains to 'di_handle'. */
3532        tl_assert(di_handle > 0);
3533        for (di = debugInfo_list; di; di = di->next) {
3534           if (di->handle == di_handle)
3535              break;
3536        }
3537     
3538        /* If this fails, we were unable to find any DebugInfo with the
3539           given handle.  This is considered an error on the part of the
3540           caller. */
3541        tl_assert(di != NULL);
3542     
3543        /* we'll put the collected variables in here. */
3544        gvars = VG_(newXA)( ML_(dinfo_zalloc), "di.debuginfo.dggbfd.1",
3545                            ML_(dinfo_free), sizeof(GlobalBlock) );
3546        tl_assert(gvars);
3547     
3548        /* any var info at all? */
3549        if (!di->varinfo)
3550           return gvars;
3551     
3552        /* we'll iterate over all the variables we can find, even if
3553           it seems senseless to visit stack-allocated variables */
3554        /* Iterate over all scopes */
3555        nScopes = VG_(sizeXA)( di->varinfo );
3556        for (scopeIx = 0; scopeIx < nScopes; scopeIx++) {
3557     
3558           /* Iterate over each (code) address range at the current scope */
3559           DiAddrRange* range;
3560           OSet* /* of DiAddrInfo */ scope
3561              = *(OSet**)VG_(indexXA)( di->varinfo, scopeIx );
3562           tl_assert(scope);
3563           VG_(OSetGen_ResetIter)(scope);
3564           while ( (range = VG_(OSetGen_Next)(scope)) ) {
3565     
3566              /* Iterate over each variable in the current address range */
3567              Word nVars, varIx;
3568              tl_assert(range->vars);
3569              nVars = VG_(sizeXA)( range->vars );
3570              for (varIx = 0; varIx < nVars; varIx++) {
3571     
3572                 Bool        isVec;
3573                 GXResult    res;
3574                 MaybeULong  mul;
3575                 GlobalBlock gb;
3576                 TyEnt*      ty;
3577                 DiVariable* var = VG_(indexXA)( range->vars, varIx );
3578                 tl_assert(var->name);
3579                 if (0) VG_(printf)("at depth %ld var %s ", scopeIx, var->name );
3580     
3581                 /* Now figure out if this variable has a constant address
3582                    (that is, independent of FP, SP, phase of moon, etc),
3583                    and if so, what the address is.  Any variable with a
3584                    constant address is deemed to be a global so we collect
3585                    it. */
3586                 if (0) { VG_(printf)("EVAL: "); ML_(pp_GX)(var->gexpr);
3587                          VG_(printf)("\n"); }
3588                 res = ML_(evaluate_trivial_GX)( var->gexpr, di );
3589     
3590                 /* Not a constant address => not interesting */
3591                 if (res.kind != GXR_Addr) {
3592                    if (0) VG_(printf)("FAIL\n");
3593                    continue;
3594                 }
3595     
3596                 /* Ok, it's a constant address.  See if we want to collect
3597                    it. */
3598                 if (0) VG_(printf)("%#lx\n", res.word);
3599     
3600                 /* Figure out how big the variable is. */
3601                 mul = ML_(sizeOfType)(di->admin_tyents, var->typeR);
3602     
3603                 /* If this var has a type whose size is unknown, zero, or
3604                    impossibly large, it should never have been added.
3605                    ML_(addVar) should have rejected it. */
3606                 vg_assert(mul.b == True);
3607                 vg_assert(mul.ul > 0);
3608                 if (sizeof(void*) == 4) vg_assert(mul.ul < (1ULL << 32));
3609                 /* After this point, we assume we can truncate mul.ul to a
3610                    host word safely (without loss of info). */
3611     
3612                 /* skip if non-array and we're only interested in
3613                    arrays */
3614                 ty = ML_(TyEnts__index_by_cuOff)( di->admin_tyents, NULL,
3615                                                   var->typeR );
3616                 vg_assert(ty);
3617                 vg_assert(ty->tag == Te_UNKNOWN || ML_(TyEnt__is_type)(ty));
3618                 if (ty->tag == Te_UNKNOWN)
3619                    continue; /* perhaps we should complain in this case? */
3620     
3621                 isVec = ty->tag == Te_TyArray;
3622                 if (arrays_only && !isVec) continue;
3623     
3624                 /* Ok, so collect it! */
3625                 tl_assert(var->name);
3626                 tl_assert(di->soname);
3627                 if (0) VG_(printf)("XXXX %s %s %d\n", var->name,
3628                                     var->fileName?(HChar*)var->fileName
3629                                                  :"??",var->lineNo);
3630                 VG_(memset)(&gb, 0, sizeof(gb));
3631                 gb.addr  = res.word;
3632                 gb.szB   = (SizeT)mul.ul;
3633                 gb.isVec = isVec;
3634                 VG_(strncpy)(&gb.name[0], var->name, sizeof(gb.name)-1);
3635                 VG_(strncpy)(&gb.soname[0], di->soname, sizeof(gb.soname)-1);
3636                 tl_assert(gb.name[ sizeof(gb.name)-1 ] == 0);
3637                 tl_assert(gb.soname[ sizeof(gb.soname)-1 ] == 0);
3638     
3639                 VG_(addToXA)( gvars, &gb );
3640     
3641              } /* for (varIx = 0; varIx < nVars; varIx++) */
3642     
3643           } /* while ( (range = VG_(OSetGen_Next)(scope)) ) */
3644     
3645        } /* for (scopeIx = 0; scopeIx < nScopes; scopeIx++) */
3646     
3647        return gvars;
3648     }

3752     /* Given an address 'a', make a guess of which section of which object
3753        it comes from.  If name is non-NULL, then the last n_name-1
3754        characters of the object's name is put in name[0 .. n_name-2], and
3755        name[n_name-1] is set to zero (guaranteed zero terminated). */

3762     
3763        for (di = debugInfo_list; di != NULL; di = di->next) {
3764     
3765           if (0)
3766              VG_(printf)(
3767                 "addr=%#lx di=%p %s got=%#lx,%ld plt=%#lx,%ld "
3768                 "data=%#lx,%ld bss=%#lx,%ld\n",
3769                 a, di, di->filename,
3770                 di->got_avma,  di->got_size,
3771                 di->plt_avma,  di->plt_size,
3772                 di->data_avma, di->data_size,
3773                 di->bss_avma,  di->bss_size);
3774     
3775           if (di->text_present
3776               && di->text_size > 0
3777               && a >= di->text_avma && a < di->text_avma + di->text_size) {
3778              res = Vg_SectText;
3779              break;
3780           }
3781           if (di->data_present
3782               && di->data_size > 0
3783               && a >= di->data_avma && a < di->data_avma + di->data_size) {
3784              res = Vg_SectData;
3785              break;
3786           }
3787           if (di->sdata_present
3788               && di->sdata_size > 0
3789               && a >= di->sdata_avma && a < di->sdata_avma + di->sdata_size) {
3790              res = Vg_SectData;
3791              break;

3794               && di->bss_size > 0
3795               && a >= di->bss_avma && a < di->bss_avma + di->bss_size) {
3796              res = Vg_SectBSS;
3797              break;
3798           }
3799           if (di->sbss_present
3800               && di->sbss_size > 0
3801               && a >= di->sbss_avma && a < di->sbss_avma + di->sbss_size) {
3802              res = Vg_SectBSS;
3803              break;

3805           if (di->plt_present
3806               && di->plt_size > 0
3807               && a >= di->plt_avma && a < di->plt_avma + di->plt_size) {
3808              res = Vg_SectPLT;

3820              res = Vg_SectGOTPLT;

3826              res = Vg_SectOPD;

3829           /* we could also check for .eh_frame, if anyone really cares */
3830        }
3831     
3832        vg_assert( (di == NULL && res == Vg_SectUnknown)
3833                   || (di != NULL && res != Vg_SectUnknown) );

3859        }
