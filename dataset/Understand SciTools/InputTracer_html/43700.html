30     /* This module produces stack traces by instrumenting all Call and Ret
31        instructions to record the current call stack. The mechanism is
32        implemented (almost) entierly in VEX IR, which makes it fairly fast.
33        
34        If the command line option for using dynamic stack traces are enabled,
35        the function IT_(get_stack_trace) returns the dynamically generated
36        stack trace. Otherwise Valgrinds own stack trace is used, which is
37        generated by walking the stack traditionally.
38        
39        Stack traces from this module never contains the current EIP
40        as the topmost element (even when dynamic stack tracing is disabled), 
41        in contrast to Valgrinds stack traces. The first element is the address
42        of the last executed Call instruction.
43     
44        When dynamic stack traces are enabled the addresses in the traces are
45        the actual starting address of the Call instructions, while stack 
46        traces generated Valgrind has addresses that points to the last byte
47        of the Call instructions.
48        
49        Call stacks are saved for each thread in the shadow state. The shadow
50        state containts a pointer to the first element of the call stack 
51        (the "bottom") and a pointer to the next free position on the stack
52        (the "top). Call stacks grows _from higher to lower addresses_, that is
53        the bottom is the highest element in the array of addresses and the most
54        recent entry in the stack is at the lowest address. This follows the same
55        convention as Valgrinds stack traces.
56     */

72     #define ST_BOTTOM_OFFSET     offsetof(VexGuestX86State, guest_SC_CLASS)
73     #define ST_BOTTOM_ABS_OFFSET (sizeof(VexGuestX86State) + ST_BOTTOM_OFFSET)
74     
75     #define ST_TOP_OFFSET     offsetof(VexGuestX86State, guest_IP_AT_SYSCALL)
76     #define ST_TOP_ABS_OFFSET (sizeof(VexGuestX86State) + ST_TOP_OFFSET)
77     
78     #define ST_MAX_STACK_SIZE 4096
79     
80     static const char* 
81     overflow_error_msg = "Reached max size of dynamic stack trace!";
82     
83     UInt IT_(get_stack_trace)(/*OUT*/StackTrace* st, ThreadId tid)
84     {
85        UInt n_ips; // Number of entries in the stack trace
86        
87        if( IT_(clo_dynamic_stack_trace) ) {
88     
89           Addr* stack_top;                                          
90           Addr* stack_bottom;
91        
92           VG_(get_shadow_regs_area)( 
93              tid, 
94              (UChar*)&stack_top,
95              1 /* First shadow area */, 
96              ST_TOP_OFFSET,
97              sizeof(Addr*) );  
98         
99           VG_(get_shadow_regs_area)( 
100              tid, 
101              (UChar*)&stack_bottom,
102              1 /* First shadow area */, 
103              ST_BOTTOM_OFFSET,
104              sizeof(Addr*) );
105         
106           n_ips = ( (UInt)stack_bottom - (UInt)stack_top )/sizeof(Addr);
107           
108           if(n_ips != 0) {
109              *st = VG_(malloc)("it_stacktrace.c: IT_(get_stack_trace)",
110                                n_ips * sizeof(Addr) );
111              
112              // Copy current stack traceVG_(get_running_tid)(),
113              VG_(memcpy)(*st, stack_top+1, n_ips*sizeof(Addr));
114           } else {
115              *st = NULL;
116           }
117        } else {
118     
119           *st = VG_(malloc)("it_sinks.c: add_stacktrace_to_list",
120                            sizeof(Addr)*ST_MAX_STACK_SIZE);
121     
122           n_ips = VG_(get_StackTrace)(tid, *st, ST_MAX_STACK_SIZE, NULL, NULL, 0);
123     
124           if(n_ips != 0) {
125              // Adjust allocated size to the actual number of entries in the stack trace  
126              *st = VG_(realloc)("it_sinks.c: add_stacktrace_to_list",
127                                 *st, n_ips*sizeof(Addr) );
128              
129              // Adjust stack trace to point to the first entry after the current
130              // EIP, just like the dynamic stack traces.         
131              *st = *st+1;
132              n_ips--;
133              
134           } else {
135              VG_(free)(*st);
136              *st = NULL;
137           }
138        }
139        
140        return n_ips; 
141     }

143     void IT_(free_stacktrace)(StackTrace st)
144     {
145        if( IT_(clo_dynamic_stack_trace) )
146           VG_(free)(st);
147        else
148           VG_(free)(st-1);  // See comment in IT_(get_stack_trace)
149     }

151     void IT_(init_stacktracing)(ThreadId tid)
152     {
153        if( ! IT_(clo_dynamic_stack_trace) )
154           return;
155     
156        // Allocate new stack
157        Addr* saved_call_stack = 
158           VG_(malloc)("it_stacktrace.c: IT_(init_stacktrace)",
159                       sizeof(Addr)*ST_MAX_STACK_SIZE);
160                                               
161        Addr* stack_bottom = &saved_call_stack[ST_MAX_STACK_SIZE-1];
162        
163        // Write pointer to bottom of stack to shadow state
164        VG_(set_shadow_regs_area) ( 
165           tid,
166           1 /* First shadow area */,
167           ST_BOTTOM_OFFSET,
168           sizeof(Addr*),
169           (const UChar*)&stack_bottom );
170           
171           // Write pointer to current stack top to shadow state
172        VG_(set_shadow_regs_area) ( 
173           tid,
174           1 /* First shadow area */,
175           ST_TOP_OFFSET,
176           sizeof(Addr*),
177           (const UChar*)&stack_bottom );
178     }
179     
180     void IT_(finish_stacktracing)(ThreadId tid)
181     {
182        Addr* stack_bottom;
183        Addr* stack_base; // Beginning of allocated area                                          
184        
185        if( ! IT_(clo_dynamic_stack_trace) )
186           return;
187        
188        VG_(get_shadow_regs_area)( 
189           tid, 
190           (UChar*)&stack_bottom,
191           1 /* First shadow area */, 
192           ST_BOTTOM_OFFSET,
193           sizeof(Addr*) );
194        
195        stack_base = stack_bottom - (ST_MAX_STACK_SIZE-1);
196        
197        VG_(free)(stack_base);
198     }

208     void IT_(instrument_push_stacktrace)(IRSB* sb, 
209                                          IRTypeEnv* tyenv,
210                                          Addr addr)
211     {
212        IRDirty* di;
213        
214        if( ! IT_(clo_dynamic_stack_trace) )
215           return;
216        
217        // Allocate temps
218        IRTemp stackTopTemp    = newIRTemp(tyenv, Ity_I32);
219        IRTemp newStackTopTemp = newIRTemp(tyenv, Ity_I32);
220        IRTemp stackBottomTemp = newIRTemp(tyenv, Ity_I32);
221        IRTemp offsetTemp      = newIRTemp(tyenv, Ity_I32);
222        IRTemp guardTemp       = newIRTemp(tyenv, Ity_I1);
223        
224        // Get address of current top of stack
225        addStmtToIRSB( 
226           sb,
227           IRStmt_WrTmp( 
228              stackTopTemp,
229              IRExpr_Get(
230                 ST_TOP_ABS_OFFSET, 
231                 Ity_I32 ) ) );
232        
233        // Get address of stack bottom
234        addStmtToIRSB( 
235           sb,
236           IRStmt_WrTmp( 
237              stackBottomTemp,
238              IRExpr_Get(
239                 ST_BOTTOM_ABS_OFFSET, 
240                 Ity_I32 ) ) );
241     
242        // Calculate offset from bottom of stack to current top of stack
243        addStmtToIRSB( 
244           sb,
245           IRStmt_WrTmp( 
246              offsetTemp,
247              IRExpr_Binop(
248                 Iop_Sub32, 
249                 IRExpr_RdTmp(stackBottomTemp),
250                 IRExpr_RdTmp(stackTopTemp) ) ) );
251        
252        // Check if we have reached the max size of the stack...
253        addStmtToIRSB( 
254           sb,
255           IRStmt_WrTmp( 
256              guardTemp,
257              IRExpr_Binop(
258                 Iop_CmpLE32U, 
259                 IRExpr_Const(
260                    IRConst_U32( ST_MAX_STACK_SIZE*sizeof(Addr) ) ),
261                 IRExpr_RdTmp(offsetTemp) ) ) );
262     
263        // ... and Panic if so.
264        di = unsafeIRDirty_0_N (
265           0, 
266           "VG_(tool_panic)", 
267           VG_(fnptr_to_fnentry)( &VG_(tool_panic) ), 
268           mkIRExprVec_1( 
269              IRExpr_Const( 
270                 IRConst_U32( 
271                    (UInt)overflow_error_msg ) ) ) );
272     
273        di->guard = IRExpr_RdTmp(guardTemp),
274     
275        addStmtToIRSB( sb, IRStmt_Dirty(di) ); 
276        
277        // Store 'addr' at top of stack        
278        addStmtToIRSB( 
279           sb,
280           IRStmt_Store(
281              Iend_LE, 
282              IRExpr_RdTmp(stackTopTemp),
283              IRExpr_Const( IRConst_U32(addr) ) ) );   
284     
285        // Calculate new stack top
286        addStmtToIRSB( 
287           sb,
288           IRStmt_WrTmp( 
289              newStackTopTemp,
290              IRExpr_Binop(
291                 Iop_Sub32, 
292                 IRExpr_RdTmp(stackTopTemp),
293                 IRExpr_Const(
294                    IRConst_U32( sizeof(Addr) ) ) ) ) );
295     
296        // Write new stack top to shadow state
297        addStmtToIRSB( 
298           sb,
299           IRStmt_Put( 
300              ST_TOP_ABS_OFFSET,
301              IRExpr_RdTmp(newStackTopTemp) ) );
302     
303     /*
304        di = unsafeIRDirty_0_N (
305           0, 
306           "print_push", 
307           VG_(fnptr_to_fnentry)( &print_push ), 
308           mkIRExprVec_2( IRExpr_RdTmp(newStackTopTemp),
309                          IRExpr_RdTmp(stackTopTemp) ) ); 
310        
311        addStmtToIRSB( sb, IRStmt_Dirty(di) );    
312     */
313     }

315     // Clean helper used by IT_(instrument_pop_stacktrace)
316     static VG_REGPARM(2) UInt 
317     calculate_pop_amount(Addr bottom, Addr top)
318     {
319        return top < bottom ? sizeof(Addr) : 0;
320     }

322     static VG_REGPARM(2) void
323     fix_stacktrace(Addr retAddr, Addr topElemAddr)
324     {
325     	Addr callAddr = *((Addr*)topElemAddr);
326     	Addr stackBottom;
327     	ThreadId tid;
328     
329     	// If we appear to be returning right after the call instruction,
330     	// everything is fine.
331     	if(retAddr > callAddr && retAddr < callAddr + 8)
332     		return;
333     
334     	tid = VG_(get_running_tid)();
335     
336     	// Get stack bottom address
337         VG_(get_shadow_regs_area)(
338     	   tid,
339            (UChar*)&stackBottom,
340            1 /* First shadow area */,
341            ST_BOTTOM_OFFSET,
342            sizeof(Addr*) );
343     
344         // See if we can recover by checking if an address further down in the call
345         // stack matches.
346     	Addr currentElem = topElemAddr;
347     	callAddr = *((Addr*)currentElem);
348     	while(currentElem <= stackBottom &&
349     			(retAddr <= callAddr || retAddr > callAddr + 8)) {
350     		currentElem += sizeof(Addr);
351     		callAddr = *((Addr*)currentElem);
352     	}
353     
354     	// If a matching call was found, pop all addresses down to and including
355     	// that one.
356     	if(currentElem <= stackBottom) {
357     		VG_(set_shadow_regs_area) (
358     			tid,
359     			1 /* First shadow area */,
360     			ST_TOP_OFFSET,
361     			sizeof(Addr*),
362     			(const UChar*)&currentElem );
363     	} else {
364     		// otherwise, the RET did not represent a real return from a function
365     		// call, and the stack trace should be restored to its previous state
366     		currentElem = topElemAddr - sizeof(Addr); // Previous stack top
367     		VG_(set_shadow_regs_area) (
368     			tid,
369     			1 /* First shadow area */,
370     			ST_TOP_OFFSET,
371     			sizeof(Addr*),
372     			(const UChar*)&currentElem );
373     	}
374     }

384     void IT_(instrument_pop_stacktrace)(IRSB* sb, 
385                                         IRTypeEnv* tyenv,
386                                         IRTemp retAddr)
387     {
388        if( ! IT_(clo_dynamic_stack_trace) )
389           return;
390        
391        // Allocate temps
392        IRTemp stackTopTemp    = newIRTemp(tyenv, Ity_I32);
393        IRTemp newStackTopTemp = newIRTemp(tyenv, Ity_I32);
394        IRTemp stackBottomTemp = newIRTemp(tyenv, Ity_I32);
395        IRTemp popAmountTemp   = newIRTemp(tyenv, Ity_I32);
396        IRTemp guardTemp       = newIRTemp(tyenv, Ity_I1);
397        IRDirty* di;
398        
399        // Get address of current top of stack
400        addStmtToIRSB( 
401           sb,
402           IRStmt_WrTmp( 
403              stackTopTemp,
404              IRExpr_Get(
405                 ST_TOP_ABS_OFFSET, 
406                 Ity_I32 ) ) );
407        
408        // Get address of stack bottom
409        addStmtToIRSB( 
410           sb,
411           IRStmt_WrTmp( 
412              stackBottomTemp,
413              IRExpr_Get(
414                 ST_BOTTOM_ABS_OFFSET, 
415                 Ity_I32 ) ) );
416     
417        addStmtToIRSB( 
418           sb,
419           IRStmt_WrTmp( 
420              popAmountTemp,
421              mkIRExprCCall( 
422                 Ity_I32, 
423                 2, 
424                 "calculate_pop_amount", 
425                 VG_(fnptr_to_fnentry)( &calculate_pop_amount ), 
426                 mkIRExprVec_2(
427                    IRExpr_RdTmp(stackBottomTemp),
428                    IRExpr_RdTmp(stackTopTemp) ) ) ) );
429           
430        // Calculate new stack top (perform pop)
431        addStmtToIRSB( 
432           sb,
433           IRStmt_WrTmp( 
434              newStackTopTemp,
435              IRExpr_Binop(
436                 Iop_Add32, 
437                 IRExpr_RdTmp(stackTopTemp),
438                 IRExpr_RdTmp(popAmountTemp) ) ) );
439     
440        // Write new stack top to shadow state
441        addStmtToIRSB( 
442           sb,
443           IRStmt_Put( 
444              ST_TOP_ABS_OFFSET,
445              IRExpr_RdTmp(newStackTopTemp) ) );
446     
447        // Check if we need to call the dirty helper; if no modification has been
448        // done to the call stack, we can skip the dirty call.
449        addStmtToIRSB(
450           sb,
451     	  IRStmt_WrTmp(
452     	    guardTemp,
453     		IRExpr_Binop(
454     		   Iop_CmpNE32,
455     		   IRExpr_Const(IRConst_U32(0)),
456     		   IRExpr_RdTmp(popAmountTemp) ) ) );
457     
458        // Call dirty helper to repair stack trace in case of weird use of CALL and
459        // RET instructions
460        di = unsafeIRDirty_0_N (
461              2,
462              "fix_stacktrace",
463              VG_(fnptr_to_fnentry)(&fix_stacktrace),
464              mkIRExprVec_2(
465     		    IRExpr_RdTmp(retAddr),
466     		    IRExpr_RdTmp(newStackTopTemp /*old top element*/) ) );
467        di->guard = IRExpr_RdTmp(guardTemp);
468     
469        addStmtToIRSB( sb, IRStmt_Dirty(di) );
470     
471     /*
472        IRDirty* di;
473        di = unsafeIRDirty_0_N (
474           0, 
475           "print_pop", 
476           VG_(fnptr_to_fnentry)( &print_pop ), 
477           mkIRExprVec_2( IRExpr_RdTmp(newStackTopTemp),
478                          IRExpr_RdTmp(stackTopTemp) ) );   
479        
480        addStmtToIRSB( sb, IRStmt_Dirty(di) );  
481     */
482     }

484     /* For debugging */
485     void IT_(print_stacktrace)(ThreadId tid)
486     {
487        Addr* stack_top;                                          
488        Addr* stack_bottom;
489        UInt  n_ips;
490        
491        if( ! IT_(clo_dynamic_stack_trace) )
492           return;   
493        
494        VG_(get_shadow_regs_area)( 
495           tid, 
496           (UChar*)&stack_top,
497           1 /* First shadow area */, 
498           ST_TOP_OFFSET,
499           sizeof(Addr*) );  
500      
501        VG_(get_shadow_regs_area)( 
502           tid, 
503           (UChar*)&stack_bottom,
504           1 /* First shadow area */, 
505           ST_BOTTOM_OFFSET,
506           sizeof(Addr*) );
507     
508        n_ips = ( (UInt)stack_bottom - (UInt)stack_top )/sizeof(Addr);
509     
510        if(n_ips)
511           VG_(pp_StackTrace)(stack_top+1, n_ips);
512        else
513           VG_(printf)("Empty stacktrace\n");
514     }
