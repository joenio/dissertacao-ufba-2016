70     /* Call f(arg1), but first switch stacks, using 'stack' as the new
71        stack, and use 'retaddr' as f's return-to address.  Also, clear all
72        the integer registers before entering f.*/
73     __attribute__((noreturn))
74     void ML_(call_on_new_stack_0_1) ( Addr stack,
75     			          Addr retaddr,
76     			          void (*f)(Word),
77                                       Word arg1 );
78     //  4(%esp) == stack
79     //  8(%esp) == retaddr
80     // 12(%esp) == f
81     // 16(%esp) == arg1
82     asm(
83     ".text\n"
84     ".globl vgModuleLocal_call_on_new_stack_0_1\n"
85     "vgModuleLocal_call_on_new_stack_0_1:\n"
86     "   movl %esp, %esi\n"     // remember old stack pointer
87     "   movl 4(%esi), %esp\n"  // set stack
88     "   pushl 16(%esi)\n"      // arg1 to stack
89     "   pushl  8(%esi)\n"      // retaddr to stack
90     "   pushl 12(%esi)\n"      // f to stack
91     "   movl $0, %eax\n"       // zero all GP regs
92     "   movl $0, %ebx\n"
93     "   movl $0, %ecx\n"
94     "   movl $0, %edx\n"
95     "   movl $0, %esi\n"
96     "   movl $0, %edi\n"
97     "   movl $0, %ebp\n"
98     "   ret\n"                 // jump to f
99     "   ud2\n"                 // should never get here
100     ".previous\n"
101     );

104     /*
105             Perform a clone system call.  clone is strange because it has
106             fork()-like return-twice semantics, so it needs special
107             handling here.
108     
109             Upon entry, we have:
110     
111                 int (fn)(void*)     in  0+FSZ(%esp)
112                 void* child_stack   in  4+FSZ(%esp)
113                 int flags           in  8+FSZ(%esp)
114                 void* arg           in 12+FSZ(%esp)
115                 pid_t* child_tid    in 16+FSZ(%esp)
116                 pid_t* parent_tid   in 20+FSZ(%esp)
117                 void* tls_ptr       in 24+FSZ(%esp)
118     
119             System call requires:
120     
121                 int    $__NR_clone  in %eax
122                 int    flags        in %ebx
123                 void*  child_stack  in %ecx
124                 pid_t* parent_tid   in %edx
125                 pid_t* child_tid    in %edi
126                 void*  tls_ptr      in %esi
127     
128     	Returns an Int encoded in the linux-x86 way, not a SysRes.
129      */
130     #define FSZ               "4+4+4+4" /* frame size = retaddr+ebx+edi+esi */
131     #define __NR_CLONE        VG_STRINGIFY(__NR_clone)
132     #define __NR_EXIT         VG_STRINGIFY(__NR_exit)
133     
134     extern
135     Int do_syscall_clone_x86_linux ( Word (*fn)(void *), 
136                                      void* stack, 
137                                      Int   flags, 
138                                      void* arg,
139                                      Int*  child_tid, 
140                                      Int*  parent_tid, 
141                                      vki_modify_ldt_t * );
142     asm(
143     ".text\n"
144     "do_syscall_clone_x86_linux:\n"
145     "        push    %ebx\n"
146     "        push    %edi\n"
147     "        push    %esi\n"
148     
149              /* set up child stack with function and arg */
150     "        movl     4+"FSZ"(%esp), %ecx\n"    /* syscall arg2: child stack */
151     "        movl    12+"FSZ"(%esp), %ebx\n"    /* fn arg */
152     "        movl     0+"FSZ"(%esp), %eax\n"    /* fn */
153     "        lea     -8(%ecx), %ecx\n"          /* make space on stack */
154     "        movl    %ebx, 4(%ecx)\n"           /*   fn arg */
155     "        movl    %eax, 0(%ecx)\n"           /*   fn */
156     
157              /* get other args to clone */
158     "        movl     8+"FSZ"(%esp), %ebx\n"    /* syscall arg1: flags */
159     "        movl    20+"FSZ"(%esp), %edx\n"    /* syscall arg3: parent tid * */
160     "        movl    16+"FSZ"(%esp), %edi\n"    /* syscall arg5: child tid * */
161     "        movl    24+"FSZ"(%esp), %esi\n"    /* syscall arg4: tls_ptr * */
162     "        movl    $"__NR_CLONE", %eax\n"
163     "        int     $0x80\n"                   /* clone() */
164     "        testl   %eax, %eax\n"              /* child if retval == 0 */
165     "        jnz     1f\n"
166     
167              /* CHILD - call thread function */
168     "        popl    %eax\n"
169     "        call    *%eax\n"                   /* call fn */
170     
171              /* exit with result */
172     "        movl    %eax, %ebx\n"              /* arg1: return value from fn */
173     "        movl    $"__NR_EXIT", %eax\n"
174     "        int     $0x80\n"
175     
176              /* Hm, exit returned */
177     "        ud2\n"
178     
179     "1:\n"   /* PARENT or ERROR */
180     "        pop     %esi\n"
181     "        pop     %edi\n"
182     "        pop     %ebx\n"
183     "        ret\n"
184     ".previous\n"
185     );
186     
187     #undef FSZ
188     #undef __NR_CLONE
189     #undef __NR_EXIT

196     /* 
197        When a client clones, we need to keep track of the new thread.  This means:
198        1. allocate a ThreadId+ThreadState+stack for the the thread
199     
200        2. initialize the thread's new VCPU state
201     
202        3. create the thread using the same args as the client requested,
203        but using the scheduler entrypoint for EIP, and a separate stack
204        for ESP.
205      */
206     static SysRes do_clone ( ThreadId ptid, 
207                              UInt flags, Addr esp, 
208                              Int* parent_tidptr, 
209                              Int* child_tidptr, 
210                              vki_modify_ldt_t *tlsinfo)
211     {
212        static const Bool debug = False;
213     
214        ThreadId     ctid = VG_(alloc_ThreadState)();
215        ThreadState* ptst = VG_(get_ThreadState)(ptid);
216        ThreadState* ctst = VG_(get_ThreadState)(ctid);
217        UWord*       stack;
218        NSegment const* seg;
219        SysRes       res;
220        Int          eax;
221        vki_sigset_t blockall, savedmask;
222     
223        VG_(sigfillset)(&blockall);
224     
225        vg_assert(VG_(is_running_thread)(ptid));
226        vg_assert(VG_(is_valid_tid)(ctid));
227     
228        stack = (UWord*)ML_(allocstack)(ctid);
229        if (stack == NULL) {
230           res = VG_(mk_SysRes_Error)( VKI_ENOMEM );
231           goto out;
232        }
233     
234        /* Copy register state
235     
236           Both parent and child return to the same place, and the code
237           following the clone syscall works out which is which, so we
238           don't need to worry about it.
239     
240           The parent gets the child's new tid returned from clone, but the
241           child gets 0.
242     
243           If the clone call specifies a NULL esp for the new thread, then
244           it actually gets a copy of the parent's esp.
245        */
246        /* Note: the clone call done by the Quadrics Elan3 driver specifies
247           clone flags of 0xF00, and it seems to rely on the assumption
248           that the child inherits a copy of the parent's GDT.  
249           setup_child takes care of setting that up. */
250        setup_child( &ctst->arch, &ptst->arch, True );
251     
252        /* Make sys_clone appear to have returned Success(0) in the
253           child. */
254        ctst->arch.vex.guest_EAX = 0;
255     
256        if (esp != 0)
257           ctst->arch.vex.guest_ESP = esp;
258     
259        ctst->os_state.parent = ptid;
260     
261        /* inherit signal mask */
262        ctst->sig_mask     = ptst->sig_mask;
263        ctst->tmp_sig_mask = ptst->sig_mask;
264     
265        /* Start the child with its threadgroup being the same as the
266           parent's.  This is so that any exit_group calls that happen
267           after the child is created but before it sets its
268           os_state.threadgroup field for real (in thread_wrapper in
269           syswrap-linux.c), really kill the new thread.  a.k.a this avoids
270           a race condition in which the thread is unkillable (via
271           exit_group) because its threadgroup is not set.  The race window
272           is probably only a few hundred or a few thousand cycles long.
273           See #226116. */
274        ctst->os_state.threadgroup = ptst->os_state.threadgroup;
275     
276        /* We don't really know where the client stack is, because its
277           allocated by the client.  The best we can do is look at the
278           memory mappings and try to derive some useful information.  We
279           assume that esp starts near its highest possible value, and can
280           only go down to the start of the mmaped segment. */
281        seg = VG_(am_find_nsegment)((Addr)esp);
282        if (seg && seg->kind != SkResvn) {
283           ctst->client_stack_highest_word = (Addr)VG_PGROUNDUP(esp);
284           ctst->client_stack_szB = ctst->client_stack_highest_word - seg->start;
285     
286           VG_(register_stack)(seg->start, ctst->client_stack_highest_word);
287     
288           if (debug)
289     	 VG_(printf)("tid %d: guessed client stack range %#lx-%#lx\n",
290     		     ctid, seg->start, VG_PGROUNDUP(esp));
291        } else {
292           VG_(message)(Vg_UserMsg,
293                        "!? New thread %d starts with ESP(%#lx) unmapped\n",
294     		   ctid, esp);
295           ctst->client_stack_szB  = 0;
296        }
297     
298        /* Assume the clone will succeed, and tell any tool that wants to
299           know that this thread has come into existence.  We cannot defer
300           it beyond this point because sys_set_thread_area, just below,
301           causes tCheck to assert by making references to the new ThreadId
302           if we don't state the new thread exists prior to that point.
303           If the clone fails, we'll send out a ll_exit notification for it
304           at the out: label below, to clean up. */
305        VG_TRACK ( pre_thread_ll_create, ptid, ctid );
306     
307        if (flags & VKI_CLONE_SETTLS) {
308           if (debug)
309     	 VG_(printf)("clone child has SETTLS: tls info at %p: idx=%d "
310                          "base=%#lx limit=%x; esp=%#x fs=%x gs=%x\n",
311     		     tlsinfo, tlsinfo->entry_number, 
312                          tlsinfo->base_addr, tlsinfo->limit,
313     		     ptst->arch.vex.guest_ESP,
314     		     ctst->arch.vex.guest_FS, ctst->arch.vex.guest_GS);
315           res = sys_set_thread_area(ctid, tlsinfo);
316           if (sr_isError(res))
317     	 goto out;
318        }
319     
320        flags &= ~VKI_CLONE_SETTLS;
321     
322        /* start the thread with everything blocked */
323        VG_(sigprocmask)(VKI_SIG_SETMASK, &blockall, &savedmask);
324     
325        /* Create the new thread */
326        eax = do_syscall_clone_x86_linux(
327                 ML_(start_thread_NORETURN), stack, flags, &VG_(threads)[ctid],
328                 child_tidptr, parent_tidptr, NULL
329              );
330        res = VG_(mk_SysRes_x86_linux)( eax );
331     
332        VG_(sigprocmask)(VKI_SIG_SETMASK, &savedmask, NULL);
333     
334       out:
335        if (sr_isError(res)) {
336           /* clone failed */
337           VG_(cleanup_thread)(&ctst->arch);
338           ctst->status = VgTs_Empty;
339           /* oops.  Better tell the tool the thread exited in a hurry :-) */
340           VG_TRACK( pre_thread_ll_exit, ctid );
341        }
342     
343        return res;
344     }

351     /* Details of the LDT simulation
352        ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
353       
354        When a program runs natively, the linux kernel allows each *thread*
355        in it to have its own LDT.  Almost all programs never do this --
356        it's wildly unportable, after all -- and so the kernel never
357        allocates the structure, which is just as well as an LDT occupies
358        64k of memory (8192 entries of size 8 bytes).
359     
360        A thread may choose to modify its LDT entries, by doing the
361        __NR_modify_ldt syscall.  In such a situation the kernel will then
362        allocate an LDT structure for it.  Each LDT entry is basically a
363        (base, limit) pair.  A virtual address in a specific segment is
364        translated to a linear address by adding the segment's base value.
365        In addition, the virtual address must not exceed the limit value.
366     
367        To use an LDT entry, a thread loads one of the segment registers
368        (%cs, %ss, %ds, %es, %fs, %gs) with the index of the LDT entry (0
369        .. 8191) it wants to use.  In fact, the required value is (index <<
370        3) + 7, but that's not important right now.  Any normal instruction
371        which includes an addressing mode can then be made relative to that
372        LDT entry by prefixing the insn with a so-called segment-override
373        prefix, a byte which indicates which of the 6 segment registers
374        holds the LDT index.
375     
376        Now, a key constraint is that valgrind's address checks operate in
377        terms of linear addresses.  So we have to explicitly translate
378        virtual addrs into linear addrs, and that means doing a complete
379        LDT simulation.
380     
381        Calls to modify_ldt are intercepted.  For each thread, we maintain
382        an LDT (with the same normally-never-allocated optimisation that
383        the kernel does).  This is updated as expected via calls to
384        modify_ldt.
385     
386        When a thread does an amode calculation involving a segment
387        override prefix, the relevant LDT entry for the thread is
388        consulted.  It all works.
389     
390        There is a conceptual problem, which appears when switching back to
391        native execution, either temporarily to pass syscalls to the
392        kernel, or permanently, when debugging V.  Problem at such points
393        is that it's pretty pointless to copy the simulated machine's
394        segment registers to the real machine, because we'd also need to
395        copy the simulated LDT into the real one, and that's prohibitively
396        expensive.
397     
398        Fortunately it looks like no syscalls rely on the segment regs or
399        LDT being correct, so we can get away with it.  Apart from that the
400        simulation is pretty straightforward.  All 6 segment registers are
401        tracked, although only %ds, %es, %fs and %gs are allowed as
402        prefixes.  Perhaps it could be restricted even more than that -- I
403        am not sure what is and isn't allowed in user-mode.
404     */

406     /* Translate a struct modify_ldt_ldt_s to a VexGuestX86SegDescr, using
407        the Linux kernel's logic (cut-n-paste of code in
408        linux/kernel/ldt.c).  */
409     
410     static
411     void translate_to_hw_format ( /* IN  */ vki_modify_ldt_t* inn,
412                                   /* OUT */ VexGuestX86SegDescr* out,
413                                             Int oldmode )
414     {
415        UInt entry_1, entry_2;
416        vg_assert(8 == sizeof(VexGuestX86SegDescr));
417     
418        if (0)
419           VG_(printf)("translate_to_hw_format: base %#lx, limit %d\n",
420                       inn->base_addr, inn->limit );
421     
422        /* Allow LDTs to be cleared by the user. */
423        if (inn->base_addr == 0 && inn->limit == 0) {
424           if (oldmode ||
425               (inn->contents == 0      &&
426                inn->read_exec_only == 1   &&
427                inn->seg_32bit == 0      &&
428                inn->limit_in_pages == 0   &&
429                inn->seg_not_present == 1   &&
430                inn->useable == 0 )) {
431              entry_1 = 0;
432              entry_2 = 0;
433              goto install;
434           }
435        }
436     
437        entry_1 = ((inn->base_addr & 0x0000ffff) << 16) |
438                  (inn->limit & 0x0ffff);
439        entry_2 = (inn->base_addr & 0xff000000) |
440                  ((inn->base_addr & 0x00ff0000) >> 16) |
441                  (inn->limit & 0xf0000) |
442                  ((inn->read_exec_only ^ 1) << 9) |
443                  (inn->contents << 10) |
444                  ((inn->seg_not_present ^ 1) << 15) |
445                  (inn->seg_32bit << 22) |
446                  (inn->limit_in_pages << 23) |
447                  0x7000;
448        if (!oldmode)
449           entry_2 |= (inn->useable << 20);
450     
451        /* Install the new entry ...  */
452       install:
453        out->LdtEnt.Words.word1 = entry_1;
454        out->LdtEnt.Words.word2 = entry_2;
455     }

529     /*
530      * read_ldt() is not really atomic - this is not a problem since
531      * synchronization of reads and writes done to the LDT has to be
532      * assured by user-space anyway. Writes are atomic, to protect
533      * the security checks done on new descriptors.
534      */
535     static
536     SysRes read_ldt ( ThreadId tid, UChar* ptr, UInt bytecount )
537     {
538        SysRes res;
539        UInt   i, size;
540        UChar* ldt;
541     
542        if (0)
543           VG_(printf)("read_ldt: tid = %d, ptr = %p, bytecount = %d\n",
544                       tid, ptr, bytecount );
545     
546        vg_assert(sizeof(HWord) == sizeof(VexGuestX86SegDescr*));
547        vg_assert(8 == sizeof(VexGuestX86SegDescr));
548     
549        ldt = (Char*)(VG_(threads)[tid].arch.vex.guest_LDT);
550        res = VG_(mk_SysRes_Success)( 0 );
551        if (ldt == NULL)
552           /* LDT not allocated, meaning all entries are null */
553           goto out;
554     
555        size = VEX_GUEST_X86_LDT_NENT * sizeof(VexGuestX86SegDescr);
556        if (size > bytecount)
557           size = bytecount;
558     
559        res = VG_(mk_SysRes_Success)( size );
560        for (i = 0; i < size; i++)
561           ptr[i] = ldt[i];
562     
563       out:
564        return res;
565     }

569     SysRes write_ldt ( ThreadId tid, void* ptr, UInt bytecount, Int oldmode )

581        vg_assert(sizeof(HWord) == sizeof(VexGuestX86SegDescr*));
582     
583        ldt      = (VexGuestX86SegDescr*)VG_(threads)[tid].arch.vex.guest_LDT;
584        ldt_info = (vki_modify_ldt_t*)ptr;
585     
586        res = VG_(mk_SysRes_Error)( VKI_EINVAL );
587        if (bytecount != sizeof(vki_modify_ldt_t))
588           goto out;
589     
590        res = VG_(mk_SysRes_Error)( VKI_EINVAL );
591        if (ldt_info->entry_number >= VEX_GUEST_X86_LDT_NENT)
592           goto out;
593        if (ldt_info->contents == 3) {
594           if (oldmode)
595              goto out;
596           if (ldt_info->seg_not_present == 0)
597              goto out;
598        }
599     
600        /* If this thread doesn't have an LDT, we'd better allocate it
601           now. */
602        if (ldt == NULL) {
603           ldt = alloc_zeroed_x86_LDT();
604           VG_(threads)[tid].arch.vex.guest_LDT = (HWord)ldt;
605        }
606     
607        /* Install the new entry ...  */
608        translate_to_hw_format ( ldt_info, &ldt[ldt_info->entry_number], oldmode );
609        res = VG_(mk_SysRes_Success)( 0 );
610     
611       out:
612        return res;
613     }

642     static SysRes sys_set_thread_area ( ThreadId tid, vki_modify_ldt_t* info )
643     {
644        Int                  idx;
645        VexGuestX86SegDescr* gdt;
646     
647        vg_assert(8 == sizeof(VexGuestX86SegDescr));
648        vg_assert(sizeof(HWord) == sizeof(VexGuestX86SegDescr*));
649     
650        if (info == NULL)
651           return VG_(mk_SysRes_Error)( VKI_EFAULT );
652     
653        gdt = (VexGuestX86SegDescr*)VG_(threads)[tid].arch.vex.guest_GDT;
654     
655        /* If the thread doesn't have a GDT, allocate it now. */
656        if (!gdt) {
657           gdt = alloc_zeroed_x86_GDT();
658           VG_(threads)[tid].arch.vex.guest_GDT = (HWord)gdt;
659        }
660     
661        idx = info->entry_number;
662     
663        if (idx == -1) {
664           /* Find and use the first free entry.  Don't allocate entry
665              zero, because the hardware will never do that, and apparently
666              doing so confuses some code (perhaps stuff running on
667              Wine). */
668           for (idx = 1; idx < VEX_GUEST_X86_GDT_NENT; idx++) {
669              if (gdt[idx].LdtEnt.Words.word1 == 0 
670                  && gdt[idx].LdtEnt.Words.word2 == 0)
671                 break;
672           }
673     
674           if (idx == VEX_GUEST_X86_GDT_NENT)
675              return VG_(mk_SysRes_Error)( VKI_ESRCH );
676        } else if (idx < 0 || idx == 0 || idx >= VEX_GUEST_X86_GDT_NENT) {
677           /* Similarly, reject attempts to use GDT[0]. */
678           return VG_(mk_SysRes_Error)( VKI_EINVAL );
679        }
680     
681        translate_to_hw_format(info, &gdt[idx], 0);
682     
683        VG_TRACK( pre_mem_write, Vg_CoreSysCall, tid,
684                  "set_thread_area(info->entry)",
685                  (Addr) & info->entry_number, sizeof(unsigned int) );
686        info->entry_number = idx;
687        VG_TRACK( post_mem_write, Vg_CoreSysCall, tid,
688                  (Addr) & info->entry_number, sizeof(unsigned int) );
689     
690        return VG_(mk_SysRes_Success)( 0 );
691     }

694     static SysRes sys_get_thread_area ( ThreadId tid, vki_modify_ldt_t* info )
695     {
696        Int idx;
697        VexGuestX86SegDescr* gdt;
698     
699        vg_assert(sizeof(HWord) == sizeof(VexGuestX86SegDescr*));
700        vg_assert(8 == sizeof(VexGuestX86SegDescr));
701     
702        if (info == NULL)
703           return VG_(mk_SysRes_Error)( VKI_EFAULT );
704     
705        idx = info->entry_number;
706     
707        if (idx < 0 || idx >= VEX_GUEST_X86_GDT_NENT)
708           return VG_(mk_SysRes_Error)( VKI_EINVAL );
709     
710        gdt = (VexGuestX86SegDescr*)VG_(threads)[tid].arch.vex.guest_GDT;
711     
712        /* If the thread doesn't have a GDT, allocate it now. */
713        if (!gdt) {
714           gdt = alloc_zeroed_x86_GDT();
715           VG_(threads)[tid].arch.vex.guest_GDT = (HWord)gdt;
716        }
717     
718        info->base_addr = ( gdt[idx].LdtEnt.Bits.BaseHi << 24 ) |
719                          ( gdt[idx].LdtEnt.Bits.BaseMid << 16 ) |
720                          gdt[idx].LdtEnt.Bits.BaseLow;
721        info->limit = ( gdt[idx].LdtEnt.Bits.LimitHi << 16 ) |
722                        gdt[idx].LdtEnt.Bits.LimitLow;
723        info->seg_32bit = gdt[idx].LdtEnt.Bits.Default_Big;
724        info->contents = ( gdt[idx].LdtEnt.Bits.Type >> 2 ) & 0x3;
725        info->read_exec_only = ( gdt[idx].LdtEnt.Bits.Type & 0x1 ) ^ 0x1;
726        info->limit_in_pages = gdt[idx].LdtEnt.Bits.Granularity;
727        info->seg_not_present = gdt[idx].LdtEnt.Bits.Pres ^ 0x1;
728        info->useable = gdt[idx].LdtEnt.Bits.Sys;
729        info->reserved = 0;
730     
731        return VG_(mk_SysRes_Success)( 0 );
732     }

746     static void setup_child ( /*OUT*/ ThreadArchState *child, 
747                               /*IN*/  ThreadArchState *parent,
748                               Bool inherit_parents_GDT )
749     {
750        /* We inherit our parent's guest state. */
751        child->vex = parent->vex;
752        child->vex_shadow1 = parent->vex_shadow1;
753        child->vex_shadow2 = parent->vex_shadow2;
754     
755        /* We inherit our parent's LDT. */
756        if (parent->vex.guest_LDT == (HWord)NULL) {
757           /* We hope this is the common case. */
758           child->vex.guest_LDT = (HWord)NULL;
759        } else {
760           /* No luck .. we have to take a copy of the parent's. */
761           child->vex.guest_LDT = (HWord)alloc_zeroed_x86_LDT();
762           copy_LDT_from_to( (VexGuestX86SegDescr*)parent->vex.guest_LDT, 
763                             (VexGuestX86SegDescr*)child->vex.guest_LDT );
764        }
765     
766        /* Either we start with an empty GDT (the usual case) or inherit a
767           copy of our parents' one (Quadrics Elan3 driver -style clone
768           only). */
769        child->vex.guest_GDT = (HWord)NULL;
770     
771        if (inherit_parents_GDT && parent->vex.guest_GDT != (HWord)NULL) {
772           child->vex.guest_GDT = (HWord)alloc_zeroed_x86_GDT();
773           copy_GDT_from_to( (VexGuestX86SegDescr*)parent->vex.guest_GDT,
774                             (VexGuestX86SegDescr*)child->vex.guest_GDT );
775        }
776     }

846     PRE(sys_clone)
847     {
848        UInt cloneflags;
849        Bool badarg = False;
850     
851        PRINT("sys_clone ( %lx, %#lx, %#lx, %#lx, %#lx )",ARG1,ARG2,ARG3,ARG4,ARG5);
852        PRE_REG_READ2(int, "clone",
853                      unsigned long, flags,
854                      void *, child_stack);
855     
856        if (ARG1 & VKI_CLONE_PARENT_SETTID) {
857           if (VG_(tdict).track_pre_reg_read) {
858              PRA3("clone", int *, parent_tidptr);
859           }
860           PRE_MEM_WRITE("clone(parent_tidptr)", ARG3, sizeof(Int));
861           if (!VG_(am_is_valid_for_client)(ARG3, sizeof(Int), 
862                                                  VKI_PROT_WRITE)) {
863              badarg = True;
864           }
865        }
866        if (ARG1 & VKI_CLONE_SETTLS) {
867           if (VG_(tdict).track_pre_reg_read) {
868              PRA4("clone", vki_modify_ldt_t *, tlsinfo);
869           }
870           PRE_MEM_READ("clone(tlsinfo)", ARG4, sizeof(vki_modify_ldt_t));
871           if (!VG_(am_is_valid_for_client)(ARG4, sizeof(vki_modify_ldt_t), 
872                                                  VKI_PROT_READ)) {
873              badarg = True;
874           }
875        }
876        if (ARG1 & (VKI_CLONE_CHILD_SETTID | VKI_CLONE_CHILD_CLEARTID)) {
877           if (VG_(tdict).track_pre_reg_read) {
878              PRA5("clone", int *, child_tidptr);
879           }
880           PRE_MEM_WRITE("clone(child_tidptr)", ARG5, sizeof(Int));
881           if (!VG_(am_is_valid_for_client)(ARG5, sizeof(Int), 
882                                                  VKI_PROT_WRITE)) {
883              badarg = True;
884           }
885        }
886     
887        if (badarg) {
888           SET_STATUS_Failure( VKI_EFAULT );
889           return;
890        }
891     
892        cloneflags = ARG1;
893     
894        if (!ML_(client_signal_OK)(ARG1 & VKI_CSIGNAL)) {
895           SET_STATUS_Failure( VKI_EINVAL );
896           return;
897        }
898     
899        /* Be ultra-paranoid and filter out any clone-variants we don't understand:
900           - ??? specifies clone flags of 0x100011
901           - ??? specifies clone flags of 0x1200011.
902           - NPTL specifies clone flags of 0x7D0F00.
903           - The Quadrics Elan3 driver specifies clone flags of 0xF00.
904           - Newer Quadrics Elan3 drivers with NTPL support specify 0x410F00.
905           Everything else is rejected. 
906        */
907        if (
908             1 ||
909             /* 11 Nov 05: for the time being, disable this ultra-paranoia.
910                The switch below probably does a good enough job. */
911               (cloneflags == 0x100011 || cloneflags == 0x1200011
912                                       || cloneflags == 0x7D0F00
913                                       || cloneflags == 0x790F00
914                                       || cloneflags == 0x3D0F00
915                                       || cloneflags == 0x410F00
916                                       || cloneflags == 0xF00
917                                       || cloneflags == 0xF21)) {
918          /* OK */
919        }
920        else {
921           /* Nah.  We don't like it.  Go away. */
922           goto reject;
923        }
924     
925        /* Only look at the flags we really care about */
926        switch (cloneflags & (VKI_CLONE_VM | VKI_CLONE_FS 
927                              | VKI_CLONE_FILES | VKI_CLONE_VFORK)) {
928        case VKI_CLONE_VM | VKI_CLONE_FS | VKI_CLONE_FILES:
929           /* thread creation */
930           SET_STATUS_from_SysRes(
931              do_clone(tid,
932                       ARG1,         /* flags */
933                       (Addr)ARG2,   /* child ESP */
934                       (Int *)ARG3,  /* parent_tidptr */
935                       (Int *)ARG5,  /* child_tidptr */
936                       (vki_modify_ldt_t *)ARG4)); /* set_tls */
937           break;
938     
939        case VKI_CLONE_VFORK | VKI_CLONE_VM: /* vfork */
940           /* FALLTHROUGH - assume vfork == fork */
941           cloneflags &= ~(VKI_CLONE_VFORK | VKI_CLONE_VM);
942     
943        case 0: /* plain fork */
944           SET_STATUS_from_SysRes(
945              ML_(do_fork_clone)(tid,
946                            cloneflags,      /* flags */
947                            (Int *)ARG3,     /* parent_tidptr */
948                            (Int *)ARG5));   /* child_tidptr */
949           break;
950     
951        default:
952        reject:
953           /* should we just ENOSYS? */
954           VG_(message)(Vg_UserMsg, "\n");
955           VG_(message)(Vg_UserMsg, "Unsupported clone() flags: 0x%lx\n", ARG1);
956           VG_(message)(Vg_UserMsg, "\n");
957           VG_(message)(Vg_UserMsg, "The only supported clone() uses are:\n");
958           VG_(message)(Vg_UserMsg, " - via a threads library (LinuxThreads or NPTL)\n");
959           VG_(message)(Vg_UserMsg, " - via the implementation of fork or vfork\n");
960           VG_(message)(Vg_UserMsg, " - for the Quadrics Elan3 user-space driver\n");
961           VG_(unimplemented)
962              ("Valgrind does not support general clone().");
963        }
964     
965        if (SUCCESS) {
966           if (ARG1 & VKI_CLONE_PARENT_SETTID)
967              POST_MEM_WRITE(ARG3, sizeof(Int));
968           if (ARG1 & (VKI_CLONE_CHILD_SETTID | VKI_CLONE_CHILD_CLEARTID))
969              POST_MEM_WRITE(ARG5, sizeof(Int));
970     
971           /* Thread creation was successful; let the child have the chance
972              to run */
973           *flags |= SfYieldAfter;
974        }
975     }
