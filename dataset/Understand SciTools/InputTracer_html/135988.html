7       double block[2];
8       block[0] = x;

10          "lfd %%f1, 0(%0)\n\t"
11          "fsqrt %%f1,%%f1\n\t"
12          "stfd %%f1, 8(%0)"

14          : /*in*/ "b" (&block[0])
