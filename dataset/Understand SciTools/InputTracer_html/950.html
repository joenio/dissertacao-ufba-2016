65     /* For a given architecture, these specify extra capabilities beyond
66        the minimum supported (baseline) capabilities.  They may be OR'd
67        together, although some combinations don't make sense.  (eg, SSE2
68        but not SSE1).  LibVEX_Translate will check for nonsensical
69        combinations. */
70     
71     /* x86: baseline capability is Pentium-1 (FPU, MMX, but no SSE), with
72        cmpxchg8b. */
73     #define VEX_HWCAPS_X86_SSE1    (1<<1)  /* SSE1 support (Pentium III) */
74     #define VEX_HWCAPS_X86_SSE2    (1<<2)  /* SSE2 support (Pentium 4) */
75     #define VEX_HWCAPS_X86_SSE3    (1<<3)  /* SSE3 support (>= Prescott) */
76     #define VEX_HWCAPS_X86_LZCNT   (1<<4)  /* SSE4a LZCNT insn */
77     
78     /* amd64: baseline capability is SSE2, with cmpxchg8b but not
79        cmpxchg16b. */
80     #define VEX_HWCAPS_AMD64_SSE3  (1<<5)  /* SSE3 support */
81     #define VEX_HWCAPS_AMD64_CX16  (1<<6)  /* cmpxchg16b support */
82     #define VEX_HWCAPS_AMD64_LZCNT (1<<7)  /* SSE4a LZCNT insn */
83     
84     /* ppc32: baseline capability is integer only */
85     #define VEX_HWCAPS_PPC32_F     (1<<8)  /* basic (non-optional) FP */
86     #define VEX_HWCAPS_PPC32_V     (1<<9)  /* Altivec (VMX) */
87     #define VEX_HWCAPS_PPC32_FX    (1<<10) /* FP extns (fsqrt, fsqrts) */
88     #define VEX_HWCAPS_PPC32_GX    (1<<11) /* Graphics extns
89                                               (fres,frsqrte,fsel,stfiwx) */
90     
91     /* ppc64: baseline capability is integer and basic FP insns */
92     #define VEX_HWCAPS_PPC64_V     (1<<12) /* Altivec (VMX) */
93     #define VEX_HWCAPS_PPC64_FX    (1<<13) /* FP extns (fsqrt, fsqrts) */
94     #define VEX_HWCAPS_PPC64_GX    (1<<14) /* Graphics extns
95                                               (fres,frsqrte,fsel,stfiwx) */
96     
97     /* s390x: baseline capability is z/Architecture with long displacement */
98     #define VEX_HWCAPS_S390X_LDISP (1<<13)  /* Long-displacement facility */
99     #define VEX_HWCAPS_S390X_EIMM  (1<<14)  /* Extended-immediate facility */
100     #define VEX_HWCAPS_S390X_GIE   (1<<15)  /* General-instruction-extension
101                                                facility */
102     #define VEX_HWCAPS_S390X_DFP   (1<<16)  /* Decimal floating point facility */
103     /* Special value representing all available s390x hwcaps */
104     #define VEX_HWCAPS_S390X_ALL   (VEX_HWCAPS_S390X_LDISP | \
105                                     VEX_HWCAPS_S390X_EIMM  | \
106                                     VEX_HWCAPS_S390X_GIE   | \
107                                     VEX_HWCAPS_S390X_DFP)
108     
109     /* arm: baseline capability is ARMv4 */
110     /* Bits 5:0 - architecture level (e.g. 5 for v5, 6 for v6 etc) */
111     #define VEX_HWCAPS_ARM_VFP    (1<<6)  /* VFP extension */
112     #define VEX_HWCAPS_ARM_VFP2   (1<<7)  /* VFPv2 */
113     #define VEX_HWCAPS_ARM_VFP3   (1<<8)  /* VFPv3 */
114     /* Bits 15:10 reserved for (possible) future VFP revisions */
115     #define VEX_HWCAPS_ARM_NEON   (1<<16) /* Advanced SIMD also known as NEON */
116

300     /* Allocate in Vex's temporary allocation area.  Be careful with this.
301        You can only call it inside an instrumentation or optimisation
302        callback that you have previously specified in a call to
303        LibVEX_Translate.  The storage allocated will only stay alive until

310     
311     static inline void* LibVEX_Alloc ( Int nbytes )
312     {
313     #if 0
314       /* Nasty debugging hack, do not use. */
315       return malloc(nbytes);
316     #else
317        HChar* curr;
318        HChar* next;
319        Int    ALIGN;
320        ALIGN  = sizeof(void*)-1;
321        nbytes = (nbytes + ALIGN) & ~ALIGN;
322        curr   = private_LibVEX_alloc_curr;
323        next   = curr + nbytes;
324        if (next >= private_LibVEX_alloc_last)
325           private_LibVEX_alloc_OOM();
326        private_LibVEX_alloc_curr = next;
327        return curr;
328     #endif
329     }

346     typedef
347        struct {
348           /* Total size of the guest state, in bytes.  Must be
349              8-aligned. */
350           Int total_sizeB;
351           /* Whereabouts is the stack pointer? */
352           Int offset_SP;
353           Int sizeof_SP; /* 4 or 8 */
354           /* Whereabouts is the frame pointer? */
355           Int offset_FP;
356           Int sizeof_FP; /* 4 or 8 */
357           /* Whereabouts is the instruction pointer? */
358           Int offset_IP;
359           Int sizeof_IP; /* 4 or 8 */
360           /* Describe parts of the guest state regarded as 'always
361              defined'. */
362           Int n_alwaysDefd;

364              Int offset;
365              Int size;
366           } alwaysDefd[VEXGLO_N_ALWAYSDEFD];

370     /* A note about guest state layout.
371     
372        LibVEX defines the layout for the guest state, in the file
373        pub/libvex_guest_<arch>.h.  The struct will have an 16-aligned
374        size.  Each translated bb is assumed to be entered with a specified
375        register pointing at such a struct.  Beyond that is two copies of
376        the shadow state area with the same size as the struct.  Beyond
377        that is a spill area that LibVEX may spill into.  It must have size
378        LibVEX_N_SPILL_BYTES, and this must be a 16-aligned number.
379     
380        On entry, the baseblock pointer register must be 16-aligned.
381     
382        There must be no holes in between the primary guest state, its two
383        copies, and the spill area.  In short, all 4 areas must have a
384        16-aligned size and be 16-aligned, and placed back-to-back.
385     */
386     
387     #define LibVEX_N_SPILL_BYTES 4096

533

543                unchanged from whatever it was at entry to the
544                translation).  'dispatch' must be NULL.
545     
546              The aim is to get back and forth between translations and the
547              dispatcher without creating memory traffic to store return
548              addresses.
549           */

551        }

558     /* A subtlety re interaction between self-checking translations and
559        bb-chasing.  The supplied chase_into_ok function should say NO
560        (False) when presented with any address for which you might want to
561        make a self-checking translation.
562     
563        If it doesn't do that, you may end up with Vex chasing from BB #1
564        to BB #2 (fine); but if you wanted checking for #2 and not #1, that
565        would not be the result.  Therefore chase_into_ok should disallow
566        following into #2.  That will force the caller to eventually
567        request a new translation starting at #2, at which point Vex will
568        correctly observe the make-a-self-check flag.  */

582     /* Code generation conventions that need to be recorded somewhere.
583        ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
584     
585        x86
586        ~~~
587        Generated code should be entered using a JMP instruction.  On
588        entry, %ebp should point to the guest state, and %esp should be a
589        valid stack pointer.  The generated code may change %eax, %ebx,
590        %ecx, %edx, %esi, %edi, all the FP registers and control state, and
591        all the XMM registers.
592     
593        On entry, the FPU control word should be set to 0x027F, and the SSE
594        control word (%mxcsr) should be set to 0x1F80.  On exit, they
595        should still have those values (after masking off the lowest 6 bits
596        of %mxcsr).  If they don't, there is a bug in VEX-generated code.
597     
598        Generated code returns to the scheduler using a JMP instruction, to
599        the address specified in the .dispatch field of VexTranslateArgs.
600        %eax (or %eax:%edx, if simulating a 64-bit target) will contain the
601        guest address of the next block to execute.  %ebp may be changed
602        to a VEX_TRC_ value, otherwise it should be as it was at entry.
603     
604        CRITICAL ISSUES in x86 code generation.  The only known critical
605        issue is that the host FPU and SSE state is not properly saved
606        across calls to helper functions.  If any helper references any
607        such state, it is likely (1) to misbehave itself, since the FP
608        stack tags will not be as expected, and (2) after returning to
609        generated code, the generated code is likely to go wrong.  This

629        ~~~~~~~~~~~~~~~~~~~~~~~
630        The guest state must contain two pseudo-registers, guest_TISTART
631        and guest_TILEN.  These are used to pass the address of areas of
