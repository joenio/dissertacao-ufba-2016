281     /* Peer at previously set up VG_(args_for_valgrind) and do some
282        minimal command line processing that must happen early on:
283     
284        - show the version string, if requested (-v)
285        - extract any request for help (--help, -h, --help-debug)
286        - get the toolname (--tool=)
287        - set VG_(clo_max_stackframe) (--max-stackframe=)
288        - set VG_(clo_main_stacksize) (--main-stacksize=)
289     
290        That's all it does.  The main command line processing is done below
291        by main_process_cmd_line_options.  Note that
292        main_process_cmd_line_options has to handle but ignore the ones we
293        have handled here.
294     */
295     static void early_process_cmd_line_options ( /*OUT*/Int* need_help,
296                                                  /*OUT*/HChar** tool )
297     {
298        UInt   i;
299        HChar* str;
300     
301        vg_assert( VG_(args_for_valgrind) );
302     
303        /* parse the options we have (only the options we care about now) */
304        for (i = 0; i < VG_(sizeXA)( VG_(args_for_valgrind) ); i++) {
305     
306           str = * (HChar**) VG_(indexXA)( VG_(args_for_valgrind), i );
307           vg_assert(str);
308     
309           // Nb: the version string goes to stdout.
310           if VG_XACT_CLO(str, "--version", VG_(log_output_sink).fd, 1) {
311              VG_(log_output_sink).is_socket = False;
312              VG_(printf)("valgrind-" VERSION "\n");
313              VG_(exit)(0);
314           }
315           else if VG_XACT_CLO(str, "--help", *need_help, *need_help+1) {}
316           else if VG_XACT_CLO(str, "-h",     *need_help, *need_help+1) {}
317     
318           else if VG_XACT_CLO(str, "--help-debug", *need_help, *need_help+2) {}
319     
320           // The tool has already been determined, but we need to know the name
321           // here.
322           else if VG_STR_CLO(str, "--tool", *tool) {} 
323     
324           // Set up VG_(clo_max_stackframe) and VG_(clo_main_stacksize).
325           // These are needed by VG_(ii_create_image), which happens
326           // before main_process_cmd_line_options().
327           else if VG_INT_CLO(str, "--max-stackframe", VG_(clo_max_stackframe)) {}
328           else if VG_INT_CLO(str, "--main-stacksize", VG_(clo_main_stacksize)) {}
329        }
330     }

332     /* The main processing for command line options.  See comments above
333        on early_process_cmd_line_options.
334     
335        Comments on how the logging options are handled:
336     
337        User can specify:
338           --log-fd=      for a fd to write to (default setting, fd = 2)
339           --log-file=    for a file name to write to
340           --log-socket=  for a socket to write to
341     
342        As a result of examining these and doing relevant socket/file
343        opening, a final fd is established.  This is stored in
344        VG_(log_output_sink) in m_libcprint.  Also, if --log-file=STR was
345        specified, then STR, after expansion of %p and %q templates within
346        it, is stored in VG_(clo_log_fname_expanded), in m_options, just in
347        case anybody wants to know what it is.
348     
349        When printing, VG_(log_output_sink) is consulted to find the
350        fd to send output to.
351     
352        Exactly analogous actions are undertaken for the XML output
353        channel, with the one difference that the default fd is -1, meaning
354        the channel is disabled by default.
355     */
356     static
357     void main_process_cmd_line_options ( /*OUT*/Bool* logging_to_fd,
358                                          /*OUT*/Char** xml_fname_unexpanded,
359                                          const HChar* toolname )
360     {
361        // VG_(clo_log_fd) is used by all the messaging.  It starts as 2 (stderr)
362        // and we cannot change it until we know what we are changing it to is
363        // ok.  So we have tmp_log_fd to hold the tmp fd prior to that point.
364        SysRes sres;
365        Int    i, tmp_log_fd, tmp_xml_fd;
366        Int    toolname_len = VG_(strlen)(toolname);
367        Char*  tmp_str;         // Used in a couple of places.
368        enum {
369           VgLogTo_Fd,
370           VgLogTo_File,
371           VgLogTo_Socket
372        } log_to = VgLogTo_Fd,   // Where is logging output to be sent?
373          xml_to = VgLogTo_Fd;   // Where is XML output to be sent?
374     
375        /* Temporarily holds the string STR specified with
376           --{log,xml}-{name,socket}=STR.  'fs' stands for
377           file-or-socket. */
378        Char* log_fsname_unexpanded = NULL;
379        Char* xml_fsname_unexpanded = NULL;
380     
381        /* Log to stderr by default, but usage message goes to stdout.  XML
382           output is initially disabled. */
383        tmp_log_fd = 2; 
384        tmp_xml_fd = -1;
385      
386        /* Check for sane path in ./configure --prefix=... */
387        if (VG_LIBDIR[0] != '/') 
388           VG_(err_config_error)("Please use absolute paths in "
389                                 "./configure --prefix=... or --libdir=...");
390     
391        vg_assert( VG_(args_for_valgrind) );
392     
393        /* BEGIN command-line processing loop */
394     
395        for (i = 0; i < VG_(sizeXA)( VG_(args_for_valgrind) ); i++) {
396     
397           HChar* arg   = * (HChar**) VG_(indexXA)( VG_(args_for_valgrind), i );
398           HChar* colon = arg;
399     
400           // Look for a colon in the option name.
401           while (*colon && *colon != ':' && *colon != '=')
402              colon++;
403     
404           // Does it have the form "--toolname:foo"?  We have to do it at the start
405           // in case someone has combined a prefix with a core-specific option,
406           // eg.  "--memcheck:verbose".
407           if (*colon == ':') {
408              if (VG_STREQN(2,            arg,                "--") && 
409                  VG_STREQN(toolname_len, arg+2,              toolname) &&
410                  VG_STREQN(1,            arg+2+toolname_len, ":"))
411              {
412                 // Prefix matches, convert "--toolname:foo" to "--foo".
413                 // Two things to note:
414                 // - We cannot modify the option in-place.  If we did, and then
415                 //   a child was spawned with --trace-children=yes, the
416                 //   now-non-prefixed option would be passed and could screw up
417                 //   the child.
418                 // - We create copies, and never free them.  Why?  Non-prefixed
419                 //   options hang around forever, so tools need not make copies
420                 //   of strings within them.  We need to have the same behaviour
421                 //   for prefixed options.  The pointer to the copy will be lost
422                 //   once we leave this function (although a tool may keep a
423                 //   pointer into it), but the space wasted is insignificant.
424                 //   (In bug #142197, the copies were being freed, which caused
425                 //   problems for tools that reasonably assumed that arguments
426                 //   wouldn't disappear on them.)
427                 if (0)
428                    VG_(printf)("tool-specific arg: %s\n", arg);
429                 arg = VG_(strdup)("main.mpclo.1", arg + toolname_len + 1);
430                 arg[0] = '-';
431                 arg[1] = '-';
432     
433              } else {
434                 // prefix doesn't match, skip to next arg
435                 continue;
436              }
437           }
438           
439           /* Ignore these options - they've already been handled */
440           if      VG_STREQN( 7, arg, "--tool=")              {}
441           else if VG_STREQN(20, arg, "--command-line-only=") {}
442           else if VG_STREQ(     arg, "--")                   {}
443           else if VG_STREQ(     arg, "-d")                   {}
444           else if VG_STREQN(16, arg, "--max-stackframe")     {}
445           else if VG_STREQN(16, arg, "--main-stacksize")     {}
446           else if VG_STREQN(14, arg, "--profile-heap")       {}
447     
448           // These options are new.
449           else if (VG_STREQ(arg, "-v") ||
450                    VG_STREQ(arg, "--verbose"))
451              VG_(clo_verbosity)++;
452     
453           else if (VG_STREQ(arg, "-q") ||
454                    VG_STREQ(arg, "--quiet"))
455              VG_(clo_verbosity)--;
456     
457           else if VG_BOOL_CLO(arg, "--stats",          VG_(clo_stats)) {}
458           else if VG_BOOL_CLO(arg, "--xml",            VG_(clo_xml)) {}
459           else if VG_BOOL_CLO(arg, "--db-attach",      VG_(clo_db_attach)) {}
460           else if VG_BOOL_CLO(arg, "--demangle",       VG_(clo_demangle)) {}
461           else if VG_BOOL_CLO(arg, "--error-limit",    VG_(clo_error_limit)) {}
462           else if VG_INT_CLO (arg, "--error-exitcode", VG_(clo_error_exitcode)) {}
463           else if VG_BOOL_CLO(arg, "--show-emwarns",   VG_(clo_show_emwarns)) {}
464     
465           else if VG_BOOL_CLO(arg, "--run-libc-freeres", VG_(clo_run_libc_freeres)) {}
466           else if VG_BOOL_CLO(arg, "--show-below-main",  VG_(clo_show_below_main)) {}
467           else if VG_BOOL_CLO(arg, "--time-stamp",       VG_(clo_time_stamp)) {}
468           else if VG_BOOL_CLO(arg, "--track-fds",        VG_(clo_track_fds)) {}
469           else if VG_BOOL_CLO(arg, "--trace-children",   VG_(clo_trace_children)) {}
470           else if VG_BOOL_CLO(arg, "--child-silent-after-fork",
471                                 VG_(clo_child_silent_after_fork)) {}
472           else if VG_BOOL_CLO(arg, "--trace-sched",      VG_(clo_trace_sched)) {}
473           else if VG_BOOL_CLO(arg, "--trace-signals",    VG_(clo_trace_signals)) {}

955     // Write the name and value of log file qualifiers to the xml file.
956     static void print_file_vars(Char* format)
957     {
958        Int i = 0;
959        
960        while (format[i]) {
961           if (format[i] == '%') {
962              // We saw a '%'.  What's next...
963              i++;
964     	 if ('q' == format[i]) {
965                 i++;
966                 if ('{' == format[i]) {
967     	       // Get the env var name, print its contents.
968     	       Char* qualname;
969                    Char* qual;
970                    i++;
971                    qualname = &format[i];
972                    while (True) {
973     		  if ('}' == format[i]) {
974                          // Temporarily replace the '}' with NUL to extract var
975                          // name.
976     		     format[i] = 0;
977                          qual = VG_(getenv)(qualname);

1206     
1207     /* Number of file descriptors that Valgrind tries to reserve for
1208        it's own use - just a small constant. */
1209     #define N_RESERVED_FDS (10)
1210     
1211     static void setup_file_descriptors(void)
1212     {
1213        struct vki_rlimit rl;
1214        Bool show = False;
1215     
1216        /* Get the current file descriptor limits. */
1217        if (VG_(getrlimit)(VKI_RLIMIT_NOFILE, &rl) < 0) {
1218           rl.rlim_cur = 1024;
1219           rl.rlim_max = 1024;
1220        }
1221     
1222     #  if defined(VGO_darwin)
1223        /* Darwin lies. It reports file max as RLIM_INFINITY but
1224           silently disallows anything bigger than 10240. */
1225        if (rl.rlim_cur >= 10240  &&  rl.rlim_max == 0x7fffffffffffffffULL) {
1226           rl.rlim_max = 10240;
1227        }
1228     #  endif
1229     
1230        if (show)
1231           VG_(printf)("fd limits: host, before: cur %lu max %lu\n", 
1232                       (UWord)rl.rlim_cur, (UWord)rl.rlim_max);
1233     
1234     #  if defined(VGP_ppc32_aix5) || defined(VGP_ppc64_aix5)
1235        /* I don't know why this kludge is needed; however if rl.rlim_cur
1236           is RLIM_INFINITY, then VG_(safe_fd)'s attempts using VG_(fcntl)
1237           to lift V's file descriptors above the threshold RLIM_INFINITY -
1238           N_RESERVED_FDS fail.  So just use a relatively conservative
1239           value in this case. */
1240        if (rl.rlim_cur > 1024)
1241           rl.rlim_cur = 1024;
1242     #  endif
1243     
1244        /* Work out where to move the soft limit to. */
1245        if (rl.rlim_cur + N_RESERVED_FDS <= rl.rlim_max) {
1246           rl.rlim_cur = rl.rlim_cur + N_RESERVED_FDS;
1247        } else {
1248           rl.rlim_cur = rl.rlim_max;
1249        }
1250     
1251        /* Reserve some file descriptors for our use. */
1252        VG_(fd_soft_limit) = rl.rlim_cur - N_RESERVED_FDS;
1253        VG_(fd_hard_limit) = rl.rlim_cur - N_RESERVED_FDS;
1254     
1255        /* Update the soft limit. */
1256        VG_(setrlimit)(VKI_RLIMIT_NOFILE, &rl);
1257     
1258        if (show) {
1259           VG_(printf)("fd limits: host,  after: cur %lu max %lu\n",
1260                       (UWord)rl.rlim_cur, (UWord)rl.rlim_max);
1261           VG_(printf)("fd limits: guest       : cur %u max %u\n",
1262                       VG_(fd_soft_limit), VG_(fd_hard_limit));
1263        }
1264     
1265        if (VG_(cl_exec_fd) != -1)
1266           VG_(cl_exec_fd) = VG_(safe_fd)( VG_(cl_exec_fd) );
1267     }

3162        This procedure performs unsigned "long division" i.e., division of a
3163     64-bit unsigned dividend by a 32-bit unsigned divisor, producing a
3164     32-bit quotient.  In the overflow cases (divide by 0, or quotient
3165     exceeds 32 bits), it returns a remainder of 0xFFFFFFFF (an impossible
3166     value).
3167        The dividend is u1 and u0, with u1 being the most significant word.
3168     The divisor is parameter v. The value returned is the quotient.

3174        if (x == 0) return(32);
3175        n = 0;

3177        if (x <= 0x00FFFFFF) {n = n + 8; x = x << 8;}
3178        if (x <= 0x0FFFFFFF) {n = n + 4; x = x << 4;}
3179        if (x <= 0x3FFFFFFF) {n = n + 2; x = x << 2;}
3180        if (x <= 0x7FFFFFFF) {n = n + 1;}

3185        division as a primitive. */

3187     {

3189        UInt un1, un0,            // Norm. dividend LSD's.
3190             vn1, vn0,            // Norm. divisor digits.
3191             q1, q0,              // Quotient digits.
3192             un32, un21, un10,    // Dividend digit pairs.
3193             rhat;                // A remainder.
3194        Int s;                    // Shift amount for norm.
3195     
3196        if (u1 >= v) {            // If overflow, set rem.
3197           if (r != NULL)         // to an impossible value,
3198              *r = 0xFFFFFFFF;    // and return the largest
3199           return 0xFFFFFFFF;}    // possible quotient.
3200     
3201        s = nlz32(v);             // 0 <= s <= 31.
3202        v = v << s;               // Normalize divisor.
3203        vn1 = v >> 16;            // Break divisor up into
3204        vn0 = v & 0xFFFF;         // two 16-bit digits.
3205     
3206        un32 = (u1 << s) | ((u0 >> (32 - s)) & (-s >> 31));
3207        un10 = u0 << s;           // Shift dividend left.
3208     
3209        un1 = un10 >> 16;         // Break right half of
3210        un0 = un10 & 0xFFFF;      // dividend into two digits.

3214      again1:

3216          q1 = q1 - 1;
3217          rhat = rhat + vn1;
3218          if (rhat < b) goto again1;}
3219

3221

3223        rhat = un21 - q0*vn1;     // quotient digit, q0.
3224      again2:

3231           *r = (un21*b + un0 - q0*v) >> s;     // return it.

3344     /* 64 x 64 ==> 64 unsigned division, using only 32 x 32 ==> 32
3345        division as a primitive. */
3346     static ULong udivdi3(ULong u, ULong v)
3347     {
3348        ULong u0, u1, v1, q0, q1, k, n;
3349     
3350        if (v >> 32 == 0) {          // If v < 2**32:
3351           if (u >> 32 < v)          // If u/v cannot overflow,
3352              return DIVU(u, v)      // just do one division.
3353                 & 0xFFFFFFFF;
3354           else {                    // If u/v would overflow:
3355              u1 = u >> 32;          // Break u up into two
3356              u0 = u & 0xFFFFFFFF;   // halves.
3357              q1 = DIVU(u1, v)       // First quotient digit.
3358                 & 0xFFFFFFFF;
3359              k = u1 - q1*v;         // First remainder, < v.
3360              q0 = DIVU((k << 32) + u0, v) // 2nd quot. digit.
3361                 & 0xFFFFFFFF;
3362              return (q1 << 32) + q0;
3363           }
3364        }
3365                                     // Here v >= 2**32.
3366        n = nlz64(v);                // 0 <= n <= 31.
3367        v1 = (v << n) >> 32;         // Normalize the divisor
3368                                     // so its MSB is 1.
3369        u1 = u >> 1;                 // To ensure no overflow.
3370        q1 = DIVU(u1, v1)            // Get quotient from
3371            & 0xFFFFFFFF;            // divide unsigned insn.
3372        q0 = (q1 << n) >> 31;        // Undo normalization and
3373                                     // division of u by 2.
3374        if (q0 != 0)                 // Make q0 correct or
3375           q0 = q0 - 1;              // too small by 1.
3376        if ((u - q0*v) >= v)
3377           q0 = q0 + 1;              // Now q0 is correct.
3378        return q0;
3379     }
