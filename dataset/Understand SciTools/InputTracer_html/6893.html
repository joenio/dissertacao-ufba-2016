36     #ifndef __LIBVEX_TRC_VALUES_H
37     #define __LIBVEX_TRC_VALUES_H
38     
39     
40     /* Magic values that the guest state pointer might be set to when
41        returning to the dispatcher.  The only other legitimate value is to
42        point to the start of the thread's VEX guest state.
43     
44        This file may get included in assembly code, so do not put
45        C-specific constructs in it.
46     
47        These values should be 61 or above so as not to conflict
48        with Valgrind's VG_TRC_ values, which are 60 or below.
49     */
50     
51     #define VEX_TRC_JMP_TINVAL     61  /* invalidate translations before
52                                           continuing */
53     #define VEX_TRC_JMP_NOREDIR    81  /* jump to undirected guest addr */
54     #define VEX_TRC_JMP_SIGTRAP    85  /* deliver trap (SIGTRAP) before
55                                           continuing */
56     #define VEX_TRC_JMP_SIGSEGV    87  /* deliver segv (SIGSEGV) before
57                                           continuing */
58     #define VEX_TRC_JMP_SIGBUS     93  /* deliver SIGBUS before continuing */
59     
60     #define VEX_TRC_JMP_EMWARN     63  /* deliver emulation warning before
61                                           continuing */
62     #define VEX_TRC_JMP_EMFAIL     83  /* emulation fatal error; abort system */
63     
64     #define VEX_TRC_JMP_CLIENTREQ  65  /* do a client req before continuing */
65     #define VEX_TRC_JMP_YIELD      67  /* yield to thread sched 
66                                           before continuing */
67     #define VEX_TRC_JMP_NODECODE   69  /* next instruction is not decodable */
68     #define VEX_TRC_JMP_MAPFAIL    71  /* address translation failed */
69     
70     #define VEX_TRC_JMP_SYS_SYSCALL  73 /* do syscall before continuing */
71     #define VEX_TRC_JMP_SYS_INT32    75 /* do syscall before continuing */
72     #define VEX_TRC_JMP_SYS_INT128   77 /* do syscall before continuing */
73     #define VEX_TRC_JMP_SYS_INT129   89 /* do syscall before continuing */
74     #define VEX_TRC_JMP_SYS_INT130   91 /* do syscall before continuing */
75     
76     #define VEX_TRC_JMP_SYS_SYSENTER 79 /* do syscall before continuing */
77     
78     #endif /* ndef __LIBVEX_TRC_VALUES_H */
