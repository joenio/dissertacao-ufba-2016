190     /* Adds a label to a list of labels. Performs a deep copy of the set
191        if ref_count > 1 */
192     static inline void add_to_labels(LabelSet** pLabels, TaintLabel newLabel)
193     {
194        LabelSet* labels = *pLabels;
195     
196        tl_assert(pLabels);
197        tl_assert(*pLabels);
198        
199        if(labels->ref_count > 1) {
200           // Original set has one reference less after the deep copy
201     	  labels->ref_count--;
202     
203           labels = deep_copy_labels(labels);
204           *pLabels = labels;
205           labels->ref_count = 1;
206        }
207          
208        IT_(treap_add_node)(&labels->data, newLabel);
209     }

233     static inline void set_labels_mem(Addr addr, LabelSet* labels)
234     {
235        UWord page = addr >> SECMAP_BITS;
236        LabelSet* oldLabels;
237     //   static UWord page_count = 0;   
238        if(primary_map[page] == null_page) {
239     //      page_count++;
240     //      VG_(printf)("Number of pages: %lu\n", page_count);
241           
242           primary_map[page] = VG_(calloc)("it_tainting.c: set_labels_mem",
243                                           SECMAP_SIZE, 
244                                           sizeof(LabelSet*)
245                                          );
246        } else {
247           /* Delete previous taint labels if they exist.
248              TODO: This probably degrades performance.
249                    Would garbage collection be better?
250           */
251           oldLabels = primary_map[page][addr & SECMAP_MASK];
252           if(oldLabels)
253              delete_labels(oldLabels);
254        }
255     
256        set_labels( &(primary_map[page][addr & SECMAP_MASK]), labels );
257     }

259     static inline void untaint_mem(Addr addr)
260     {
261        LabelSet** pShMem = get_labels_mem(addr);
262     
263        if(*pShMem) {
264           delete_labels(*pShMem);
265           *pShMem = NULL;
266        }
267     }

269     static void VG_REGPARM(2) untaint_mem_range(Addr start, SizeT size)
270     {
271        Addr a = start;
272        UWord first_empty_page    = (start + SECMAP_SIZE-1) >> SECMAP_BITS;
273        UWord first_nonempty_page = (start + size) >> SECMAP_BITS;
274        UWord page = first_empty_page;
275        
276        // Remove taint labels
277        for( ; a < start + size; a++)
278           untaint_mem(a);
279           
280        // Remove secondary maps
281        for( ; page < first_nonempty_page; page++) {
282           if(primary_map[page] != null_page)
283     		  VG_(free)(primary_map[page]);
284           primary_map[page] = null_page;
285        }
286     }

437     static inline Bool isFlagRegister(Int offset)
438     {
439        // Divide all offsets by 4 to check if offset is somewhere within
440        // each 4-byte shadow register.
441        offset >>= 2;
442        
443        return ( offset >= (offsetof(VexGuestX86State, guest_CC_OP) >> 2) &&
444                 offset <  (offsetof(VexGuestX86State, guest_DFLAG) >> 2) );
445     }
446     
447     static inline Bool isInterestingRegister(Int offset)
448     {
449        // Divide all offsets by 4 to check if offset is somewhere within
450        // each 4-byte shadow register.
451        offset >>= 2;
452        
453        // Only EAX,EBX,ECX,EDX,ESI,EDI,EBP and pseudo registers
454        // for EFLAGS (except CC_OP) are taint traced.
455        return ( offset >= (offsetof(VexGuestX86State, guest_EAX)   >> 2) &&
456                 offset <  (offsetof(VexGuestX86State, guest_DFLAG) >> 2) &&
457                 offset != (offsetof(VexGuestX86State, guest_ESP)   >> 2) &&
458                 offset != (offsetof(VexGuestX86State, guest_CC_OP) >> 2) );
459     }

461     /* Adds all labels from all bytes at the specified offset to 'dst'.
462        If *dst is NULL, a new list is created at dst. */
463     static void add_labels_reg_to_union(UInt offset, UWord size, 
464                                         LabelSet** dst)
465     {
466        LabelSet* regLabels[4];
467        UWord i, j;
468        
469        tl_assert( isInterestingRegister(offset) );
470     
471        // Assume only one register is read at a time
472        tl_assert(size <= 4);
473        
474        // Copy register labels to temporary storage
475        copy_labels_from_reg(offset, size, regLabels);
476        
477        for(i=0 ; i < size; i++) {
478     	   if(regLabels[i] != NULL) {
479     		   if(*dst == NULL)
480     			   *dst = create_label_set();
481     
482     		   for(j=0; j < regLabels[i]->data.current_idx; j++)
483     			   add_to_labels(dst, regLabels[i]->data.array[j].value);
484     
485     		   // Delete labels in temporary storage
486     		   delete_labels(regLabels[i]);
487     	   }
488        }
489     }

496     /* Returns the number of shadow pointers needed to 
497        store shadow info for 'temp' */
498     static UWord sizeof_shadow_temp(IRTemp temp, IRTypeEnv* tyenv)
499     {
500        IRType type = typeOfIRTemp(tyenv, temp);
501        
502        switch(type) {
503           case Ity_I1:
504           case Ity_I8:
505              return 1;
506              
507           case Ity_I16:
508           case Ity_I32:
509           case Ity_I64:
510           case Ity_I128:  
511           case Ity_V128 /* Does this ever happen? */:
512              return sizeofIRType(type);
513              
514           case Ity_INVALID:
515           case Ity_F32:   
516           case Ity_F64:
517           case Ity_F128:
518              // Should never happen...
519              tl_assert(0);
520        }
521        
522        return 0; // Never happens
523     }

525     /* Allocates space in the shadow temp pool and creates a hash table entry
526        associating 'temp' with the allocated pool index. 
527        Returns the allocated pool index. */
528     static UWord allocate_shadow_temp( IRTemp temp, IRTypeEnv* tyenv)
529     {
530        ShadowTempHashNode* node = NULL;
531        UWord size = sizeof_shadow_temp(temp, tyenv);   
532     
533        node = VG_(malloc)( "it_tainting.c: allocate_shadow_temp",
534                            sizeof(ShadowTempHashNode)
535                          );
536     
537        node->key      = temp;
538        node->pool_idx = shadow_pool_top;
539        
540        shadow_pool_top += size;
541        tl_assert( shadow_pool_top <= SHADOW_POOL_SIZE );
542     
543        VG_(HT_add_node)(shadowTempsHT, node);
544        
545        return node->pool_idx;
546     }

678     dirty_instrument_bytewise_union(UInt src1PoolIdx, 
679                                     UInt src2PoolIdx,
680                                     UInt dstPoolIdx, 
681                                     UInt size)
682     {

760     			}

797     
798     static VG_REGPARM(3) void dirty_apply_addr_taint_load(UInt addrTmpIdx,
799                                                           UInt dstIdx, 
800                                                           UInt dstSize)
801     {
802        UWord i, j;
803        LabelSet* addrLabels = NULL;
804        LabelSet* curLabels;
805     
806        // Create union of all labels in all positions of the tainted address
807     	for (i = 0; i < sizeof(HWord); i++) {
808     		curLabels = shadow_pool[addrTmpIdx + i];
809     
810     		if (curLabels) {
811     			if (!addrLabels)
812     				set_labels(&addrLabels, create_label_set());
813     
814     			for (j = 0; j < curLabels->data.current_idx; j++)
815     				add_to_labels(&addrLabels, curLabels->data.array[j].value);
816     		}
817     	}
818     
819     	// If address didn't contain taint, just quit
820     	if (!addrLabels)
821     		return;
822     
823     	// Add the union of the address' labels to all positions of destination
824     	for (i = 0; i < dstSize; i++) {
825     		if (!shadow_pool[dstIdx + i]) {
826     			set_labels_temp(dstIdx + i, addrLabels);
827     			continue;
828     		}
829     
830     		for (j = 0; j < addrLabels->data.current_idx; j++) {
831     			add_to_labels(
832     					&shadow_pool[dstIdx + i],
833     					addrLabels->data.array[j].value);
834     		}
835     	}
836     
837     	delete_labels(addrLabels);
838     }

841                                                            Addr addr,
842                                                            UInt dstSize)
843     {
844        UWord i, j;
845        LabelSet* addrLabels = NULL;
846        LabelSet* curLabels;
847     
848        // Create union of all labels in all positions of the tainted address
849     	for (i = 0; i < sizeof(HWord); i++) {

857     				add_to_labels(&addrLabels, curLabels->data.array[j].value);

859     	}

861     	// If address didn't contain taint, just quit

877     
878     	delete_labels(addrLabels);
879     }

943     /* Inserts instrumentation to untaint all tainted temps at the end of a
944        superblock, and clears the hash table. */
945     void IT_(finish_instrumentation)(IRSB* sb, IRTypeEnv* tyenv)
946     {
947        ShadowTempHashNode* node;
948     
949        // Cleanup shadow temps (constant True guard means always perform cleanup).
950        IT_(instrument_cleanup)( sb, tyenv, IRExpr_Const( IRConst_U1(True) ) );
951     
952        // Clear hash table
953        VG_(HT_ResetIter)(shadowTempsHT);
954        node = VG_(HT_Next)(shadowTempsHT);
955        while(node) {
956           VG_(HT_remove)(shadowTempsHT, node->key );
957           VG_(free)(node);
958           
959           // Reset necessary, since table has been modified above
960           VG_(HT_ResetIter)(shadowTempsHT);
961           node = VG_(HT_Next)(shadowTempsHT);
962        }
963        
964        // Clear shadow pool   
965        shadow_pool_top = 0;
966     }

1031                                             IRTypeEnv* tyenv, 
1032                                             IRExpr* data, 
1033                                             IRExpr* addr)
1034     {
1035        UInt poolIdx, size;
1036        ShadowTempHashNode* node;
1037        IRDirty* di;
1038        
1039        // If this is an untraced type, just treat this as an untainting
1040        if( ! isInterestingType( typeOfIRExpr(tyenv, data) ) )
1041           goto untaint;
1042     
1043        tl_assert( typeOfIRExpr(tyenv, addr) == Ity_I32 );
1044        
1045        if( data->tag == Iex_Const )
1046           goto untaint;
1047        else if( data->tag == Iex_RdTmp ) {
1048           IRTemp dataTemp = data->Iex.RdTmp.tmp;
1049           node = VG_(HT_lookup)(shadowTempsHT, dataTemp);
1050           
1051           if(!node)
1052              goto untaint;
1053           
1054           poolIdx = node->pool_idx;
1055           size    = sizeof_shadow_temp(dataTemp, tyenv);
1056           di = unsafeIRDirty_0_N (
1057              3, 
1058              "copy_labels_to_mem", 
1059              VG_(fnptr_to_fnentry)(&copy_labels_to_mem), 
1060              mkIRExprVec_3( 
1061                 addr, 
1062                 IRExpr_Const( IRConst_U32(size) ),
1063                 IRExpr_Const( IRConst_U32( (UInt)(&shadow_pool[poolIdx]) ) ) ) );
1064        
1065           return di;
1066     
1067        } else
1068           tl_assert(0);
1069        
1070        /* Faster case when all memory is just to be untainted */
1071        untaint:
1072        size = sizeofIRType( typeOfIRExpr(tyenv, data) );
1073      
1074        di = unsafeIRDirty_0_N ( 
1075           2, 
1076           "untaint_mem_range", 
1077           VG_(fnptr_to_fnentry)(&untaint_mem_range), 
1078           mkIRExprVec_2(
1079              addr, 
1080              IRExpr_Const( IRConst_U32(size) ) ) );
1081        
1082        return di;
1083     }

1124

1126        IRExpr* addr     = details->addr;

1129

1132

1911                                   TaintLabel startLabel,
1912                                   Bool overwrite)
1913     {
1914     //VG_(printf)("Tainting %p -- %p starting with taint label %lu\n", startAddr, endAddr, startLabel);
1915        
1916        TaintLabel curLabel = startLabel;
1917        Addr curAddr = startAddr;
1918        LabelSet* labels;
1919        LabelSet** pShMem;
1920        
1921        for(; curAddr <= endAddr; curAddr++, curLabel++) {
1922           if( !overwrite ) {
1923              pShMem = get_labels_mem(curAddr);
1924              if(*pShMem) {
1925                 add_to_labels(pShMem, curLabel);
1926                 continue;
1927              }
1928           }
1929           labels = create_label_set();
1930           add_to_labels(&labels, curLabel);
1931           set_labels_mem(curAddr, labels);
1932        }
1933     
1934        //IT_(print_shadow_mem)();
1935     }

1950     void IT_(tainting_pre_reg_read)(CorePart part, ThreadId tid, Char* s, 
1951                                     PtrdiffT guest_state_offset, SizeT size)
1952     {
1953     	LabelSet* tmp_syscall_labels;
1954     	Bool isExcludedImage = IT_(is_in_excluded_range_prop)( VG_(get_IP)(tid) );
1955     
1956        if( !IT_(clo_syscall_prop) || part != Vg_CoreSysCall || isExcludedImage )
1957           return;
1958        
1959     //VG_(dmsg)("Pre reg read, offset %u\n", guest_state_offset);
1960     
1961        // Copy syscall labels to temporary storage
1962        tmp_syscall_labels = get_labels_syscall(tid);
1963        
1964        // Calculate union of all register labels and add to syscall labels
1965        add_labels_reg_to_union(guest_state_offset, size, &tmp_syscall_labels);
1966        
1967        // Copy back unioned labels to syscall label storage in shadow state
1968        set_labels_syscall(tmp_syscall_labels, tid);
1969     }

1971     void IT_(tainting_post_reg_write)(CorePart part, ThreadId tid,
1972                                       PtrdiffT guest_state_offset, SizeT size)
1973     {
1974     	LabelSet* tmp_syscall_labels;
1975        UWord i;
1976     	Bool isExcludedImage = IT_(is_in_excluded_range_prop)( VG_(get_IP)(tid) );
1977     
1978       if( !IT_(clo_syscall_prop) || part != Vg_CoreSysCall || isExcludedImage )
1979          return;
1980     
1981     //VG_(printf)("Pre reg write, offset %u\n", guest_state_offset);
1982     //IT_(print_label_list)(tmp_syscall_labels);
1983     //VG_(printf)("\n\n");
1984     
1985        // Only care about "regular" x86 registers
1986        if( ! isInterestingRegister(guest_state_offset) )
1987           return;
1988       
1989        // Assume only one register is written at a time
1990        tl_assert(size <= 4);
1991        
1992        tmp_syscall_labels = get_labels_syscall(tid);
1993        for(i=0 ; i < size; i++)
1994           copy_labels_to_reg(guest_state_offset+i, 1, &tmp_syscall_labels);
1995     }
