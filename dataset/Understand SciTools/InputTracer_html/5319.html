71     /* ---------------------------------------------------------------------
72        A struct to hold starting values for stack unwinding.
73        ------------------------------------------------------------------ */
74     
75     /* This really shouldn't be here.  But putting it elsewhere leads to a
76        veritable swamp of new module cycles. */
77     
78     /* To support CFA-based stack unwinding, and stack unwinding in
79        general, we need to be able to get hold of the values of specific
80        registers, in order to start the unwinding process.  This is
81        unavoidably arch and platform dependent.  Here is a struct which
82        holds the relevant values.  All platforms must have a program
83        counter and a stack pointer register, but the other fields (frame
84        pointer? link register? misc other regs?) are ad-hoc.  Note, the
85        common fields are 64-bit, so as to make this host-independent. */
86     
87     typedef
88        struct {
89           ULong r_pc; /* x86:EIP, amd64:RIP, ppc:CIA, arm:R15 */
90           ULong r_sp; /* x86:ESP, amd64:RSP, ppc:R1,  arm:R13 */
91           union {
92              struct {
93                 UInt r_ebp;
94              } X86;
95              struct {
96                 ULong r_rbp;
97              } AMD64;
98              struct {
99                 UInt r_lr;
100              } PPC32;
101              struct {
102                 ULong r_lr;
103              } PPC64;
104              struct {
105                 UInt r14;
106                 UInt r12;
107                 UInt r11;
108                 UInt r7;
109              } ARM;
110              struct {
111                 ULong r_fp;
112                 ULong r_lr;
113              } S390X;
114           } misc;
115        }
116        UnwindStartRegs;
