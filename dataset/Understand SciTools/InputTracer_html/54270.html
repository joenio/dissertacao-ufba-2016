104     /* Disassemble a complete basic block, starting at guest_IP_start, 
105        returning a new IRSB.  The disassembler may chase across basic
106        block boundaries if it wishes and if chase_into_ok allows it.
107        The precise guest address ranges from which code has been taken
108        are written into vge.  guest_IP_bbstart is taken to be the IP in
109        the guest's address space corresponding to the instruction at
110        &guest_code[0].  
111     
112        dis_instr_fn is the arch-specific fn to disassemble on function; it
113        is this that does the real work.
114     
115        do_self_check indicates that the caller needs a self-checking
116        translation.
117     
118        preamble_function is a callback which allows the caller to add
119        its own IR preamble (following the self-check, if any).  May be
120        NULL.  If non-NULL, the IRSB under construction is handed to 
121        this function, which presumably adds IR statements to it.  The
122        callback may optionally complete the block and direct bb_to_IR
123        not to disassemble any instructions into it; this is indicated
124        by the callback returning True.
125     
126        offB_TIADDR and offB_TILEN are the offsets of guest_TIADDR and
127        guest_TILEN.  Since this routine has to work for any guest state,
128        without knowing what it is, those offsets have to passed in.
129     
130        callback_opaque is a caller-supplied pointer to data which the
131        callbacks may want to see.  Vex has no idea what it is.
132        (In fact it's a VgInstrumentClosure.)
133     */
134     
135     IRSB* bb_to_IR ( /*OUT*/VexGuestExtents* vge,
136                      /*IN*/ void*            callback_opaque,
137                      /*IN*/ DisOneInstrFn    dis_instr_fn,
138                      /*IN*/ UChar*           guest_code,
139                      /*IN*/ Addr64           guest_IP_bbstart,
140                      /*IN*/ Bool             (*chase_into_ok)(void*,Addr64),
141                      /*IN*/ Bool             host_bigendian,
142                      /*IN*/ VexArch          arch_guest,
143                      /*IN*/ VexArchInfo*     archinfo_guest,
144                      /*IN*/ VexAbiInfo*      abiinfo_both,
145                      /*IN*/ IRType           guest_word_type,
146                      /*IN*/ Bool             do_self_check,
147                      /*IN*/ Bool             (*preamble_function)(void*,IRSB*),
148                      /*IN*/ Int              offB_TISTART,
149                      /*IN*/ Int              offB_TILEN )
150     {
151        Long       delta;
152        Int        i, n_instrs, first_stmt_idx;
153        Bool       resteerOK, need_to_put_IP, debug_print;
154        DisResult  dres;
155        IRStmt*    imark;
156        static Int n_resteers = 0;
157        Int        d_resteers = 0;
158        Int        selfcheck_idx = 0;
159        IRSB*      irsb;
160        Addr64     guest_IP_curr_instr;
161        IRConst*   guest_IP_bbstart_IRConst = NULL;
162        Int        n_cond_resteers_allowed = 2;
163     
164        Bool (*resteerOKfn)(void*,Addr64) = NULL;
165     
166        debug_print = toBool(vex_traceflags & VEX_TRACE_FE);
167     
168        /* Note: for adler32 to work without % operation for the self
169           check, need to limit length of stuff it scans to 5552 bytes.
170           Therefore limiting the max bb len to 100 insns seems generously
171           conservative. */
172     
173        /* check sanity .. */
174        vassert(sizeof(HWord) == sizeof(void*));
175        vassert(vex_control.guest_max_insns >= 1);
176        vassert(vex_control.guest_max_insns < 100);
177        vassert(vex_control.guest_chase_thresh >= 0);
178        vassert(vex_control.guest_chase_thresh < vex_control.guest_max_insns);
179        vassert(guest_word_type == Ity_I32 || guest_word_type == Ity_I64);
180     
181        /* Start a new, empty extent. */
182        vge->n_used  = 1;
183        vge->base[0] = guest_IP_bbstart;
184        vge->len[0]  = 0;
185     
186        /* And a new IR superblock to dump the result into. */
187        irsb = emptyIRSB();
188     
189        /* Delta keeps track of how far along the guest_code array we have
190           so far gone. */
191        delta    = 0;
192        n_instrs = 0;
193     
194        /* Guest addresses as IRConsts.  Used in the two self-checks
195           generated. */
196        if (do_self_check) {
197           guest_IP_bbstart_IRConst
198              = guest_word_type==Ity_I32 
199                   ? IRConst_U32(toUInt(guest_IP_bbstart))
200                   : IRConst_U64(guest_IP_bbstart);
201        }
202     
203        /* If asked to make a self-checking translation, leave 15 spaces in
204           which to put the check statements (up to 3 extents, and 5 stmts
205           required for each).  We'll fill them in later when we know the
206           extents and checksums of the areas to check. */
207        if (do_self_check) {
208           selfcheck_idx = irsb->stmts_used;
209           for (i = 0; i < 3 * 5; i++)
210              addStmtToIRSB( irsb, IRStmt_NoOp() );
211        }
212     
213        /* If the caller supplied a function to add its own preamble, use
214           it now. */
215        if (preamble_function) {
216           Bool stopNow = preamble_function( callback_opaque, irsb );
217           if (stopNow) {
218              /* The callback has completed the IR block without any guest
219                 insns being disassembled into it, so just return it at
220                 this point, even if a self-check was requested - as there
221                 is nothing to self-check.  The five self-check no-ops will
222                 still be in place, but they are harmless. */
223              return irsb;
224           }
225        }
226     
227        /* Process instructions. */
228        while (True) {
229           vassert(n_instrs < vex_control.guest_max_insns);
230     
231           /* Regardless of what chase_into_ok says, is chasing permissible
232              at all right now?  Set resteerOKfn accordingly. */
233           resteerOK 
234              = toBool(
235                   n_instrs < vex_control.guest_chase_thresh
236                   /* we can't afford to have a resteer once we're on the
237                      last extent slot. */
238                   && vge->n_used < 3
239                );
240     
241           resteerOKfn
242              = resteerOK ? chase_into_ok : const_False;
243     
244           /* n_cond_resteers_allowed keeps track of whether we're still
245              allowing dis_instr_fn to chase conditional branches.  It
246              starts (at 2) and gets decremented each time dis_instr_fn
247              tells us it has chased a conditional branch.  We then
248              decrement it, and use it to tell later calls to dis_instr_fn
249              whether or not it is allowed to chase conditional
250              branches. */
251           vassert(n_cond_resteers_allowed >= 0 && n_cond_resteers_allowed <= 2);
252     
253           /* This is the IP of the instruction we're just about to deal
254              with. */
255           guest_IP_curr_instr = guest_IP_bbstart + delta;
256     
257           /* This is the irsb statement array index of the first stmt in
258              this insn.  That will always be the instruction-mark
259              descriptor. */
260           first_stmt_idx = irsb->stmts_used;
261     
262           /* Add an instruction-mark statement.  We won't know until after
263              disassembling the instruction how long it instruction is, so
264              just put in a zero length and we'll fix it up later. */
265           addStmtToIRSB( irsb, IRStmt_IMark( guest_IP_curr_instr, 0 ));
266     
267           /* for the first insn, the dispatch loop will have set
268              %IP, but for all the others we have to do it ourselves. */
269           need_to_put_IP = toBool(n_instrs > 0);
270     
271           /* Finally, actually disassemble an instruction. */
272           dres = dis_instr_fn ( irsb,
273                                 need_to_put_IP,
274                                 resteerOKfn,
275                                 toBool(n_cond_resteers_allowed > 0),
276                                 callback_opaque,
277                                 guest_code,
278                                 delta,
279                                 guest_IP_curr_instr,
280                                 arch_guest,
281                                 archinfo_guest,
282                                 abiinfo_both,
283                                 host_bigendian );
284     
285           /* stay sane ... */
286           vassert(dres.whatNext == Dis_StopHere
287                   || dres.whatNext == Dis_Continue
288                   || dres.whatNext == Dis_ResteerU
289                   || dres.whatNext == Dis_ResteerC);
290           /* ... disassembled insn length is sane ... */
291           vassert(dres.len >= 0 && dres.len <= 20);
292           /* ... continueAt is zero if no resteer requested ... */
293           if (dres.whatNext != Dis_ResteerU && dres.whatNext != Dis_ResteerC)
294              vassert(dres.continueAt == 0);
295           /* ... if we disallowed conditional resteers, check that one
296                  didn't actually happen anyway ... */
297           if (n_cond_resteers_allowed == 0)
298              vassert(dres.whatNext != Dis_ResteerC);
299     
300           /* Fill in the insn-mark length field. */
301           vassert(first_stmt_idx >= 0 && first_stmt_idx < irsb->stmts_used);
302           imark = irsb->stmts[first_stmt_idx];
303           vassert(imark);
304           vassert(imark->tag == Ist_IMark);
305           vassert(imark->Ist.IMark.len == 0);
306           imark->Ist.IMark.len = toUInt(dres.len);
307     
308           /* Print the resulting IR, if needed. */
309           if (vex_traceflags & VEX_TRACE_FE) {
310              for (i = first_stmt_idx; i < irsb->stmts_used; i++) {
311                 vex_printf("              ");
312                 ppIRStmt(irsb->stmts[i]);
313                 vex_printf("\n");
314              }
315           }
316     
317           /* If dis_instr_fn terminated the BB at this point, check it
318              also filled in the irsb->next field. */
319           if (dres.whatNext == Dis_StopHere) {
320              vassert(irsb->next != NULL);
321              if (debug_print) {
322                 vex_printf("              ");
323                 vex_printf( "goto {");
324                 ppIRJumpKind(irsb->jumpkind);
325                 vex_printf( "} ");
326                 ppIRExpr( irsb->next );
327                 vex_printf( "\n");
328              }
329           }
330     
331           /* Update the VexGuestExtents we are constructing. */
332           /* If vex_control.guest_max_insns is required to be < 100 and
333              each insn is at max 20 bytes long, this limit of 5000 then
334              seems reasonable since the max possible extent length will be
335              100 * 20 == 2000. */
336           vassert(vge->len[vge->n_used-1] < 5000);
337           vge->len[vge->n_used-1] 
338              = toUShort(toUInt( vge->len[vge->n_used-1] + dres.len ));
339           n_instrs++;
340           if (debug_print) 
341              vex_printf("\n");
342     
343           /* Advance delta (inconspicuous but very important :-) */
344           delta += (Long)dres.len;
345     
346           switch (dres.whatNext) {
347              case Dis_Continue:
348                 vassert(irsb->next == NULL);
349                 if (n_instrs < vex_control.guest_max_insns) {
350                    /* keep going */
351                 } else {
352                    /* We have to stop. */
353                    irsb->next 
354                       = IRExpr_Const(
355                            guest_word_type == Ity_I32
356                               ? IRConst_U32(toUInt(guest_IP_bbstart+delta))
357                               : IRConst_U64(guest_IP_bbstart+delta)
358                         );
359                    goto done;
360                 }
361                 break;
362              case Dis_StopHere:
363                 vassert(irsb->next != NULL);
364                 goto done;
365              case Dis_ResteerU:
366              case Dis_ResteerC:
367                 /* Check that we actually allowed a resteer .. */
368                 vassert(resteerOK);
369                 vassert(irsb->next == NULL);
370                 if (dres.whatNext == Dis_ResteerC) {
371                    vassert(n_cond_resteers_allowed > 0);
372                    n_cond_resteers_allowed--;
373                 }
374                 /* figure out a new delta to continue at. */
375                 vassert(resteerOKfn(callback_opaque,dres.continueAt));
376                 delta = dres.continueAt - guest_IP_bbstart;
377                 /* we now have to start a new extent slot. */
378                 vge->n_used++;
379                 vassert(vge->n_used <= 3);
380                 vge->base[vge->n_used-1] = dres.continueAt;
381                 vge->len[vge->n_used-1] = 0;
382                 n_resteers++;
383                 d_resteers++;
384                 if (0 && (n_resteers & 0xFF) == 0)
385                 vex_printf("resteer[%d,%d] to 0x%llx (delta = %lld)\n",
386                            n_resteers, d_resteers,
387                            dres.continueAt, delta);
388                 break;
389              default:
390                 vpanic("bb_to_IR");
391           }
392        }
393        /*NOTREACHED*/
394        vassert(0);
395     
396       done:
397        /* We're done.  The only thing that might need attending to is that
398           a self-checking preamble may need to be created.
399     
400           The scheme is to compute a rather crude checksum of the code
401           we're making a translation of, and add to the IR a call to a
402           helper routine which recomputes the checksum every time the
403           translation is run, and requests a retranslation if it doesn't
404           match.  This is obviously very expensive and considerable
405           efforts are made to speed it up:
406     
407           * the checksum is computed from all the 32-bit words that
408             overlap the translated code.  That means it could depend on up
409             to 3 bytes before and 3 bytes after which aren't part of the
410             translated area, and so if those change then we'll
411             unnecessarily have to discard and retranslate.  This seems
412             like a pretty remote possibility and it seems as if the
413             benefit of not having to deal with the ends of the range at
414             byte precision far outweigh any possible extra translations
415             needed.
416     
417           * there's a generic routine and 12 specialised cases, which
418             handle the cases of 1 through 12-word lengths respectively.
419             They seem to cover about 90% of the cases that occur in
420             practice.
421        */
422        if (do_self_check) {
423     
424           Addr64   base2check;
425           UInt     len2check;
426           HWord    expectedhW;
427           IRTemp   tistart_tmp, tilen_tmp;
428           HWord    VEX_REGPARM(2) (*fn_generic)(HWord, HWord);
429           HWord    VEX_REGPARM(1) (*fn_spec)(HWord);
430           HChar*   nm_generic;
431           HChar*   nm_spec;
432           HWord    fn_generic_entry = 0;
433           HWord    fn_spec_entry = 0;
434           UInt     host_word_szB = sizeof(HWord);
435           IRType   host_word_type = Ity_INVALID;
436     
437           if (host_word_szB == 4) host_word_type = Ity_I32;
438           if (host_word_szB == 8) host_word_type = Ity_I64;
439           vassert(host_word_type != Ity_INVALID);
440     
441           vassert(vge->n_used >= 1 && vge->n_used <= 3);
442           for (i = 0; i < vge->n_used; i++) {
443     
444              /* the extent we're generating a check for */
445              base2check = vge->base[i];
446              len2check  = vge->len[i];
447     
448              /* stay sane */
449              vassert(len2check >= 0 && len2check < 1000/*arbitrary*/);
450     
451              /* Skip the check if the translation involved zero bytes */
452              if (len2check == 0)
453                 continue;
454     
455              HWord first_hW = ((HWord)base2check)
456                               & ~(HWord)(host_word_szB-1);
457              HWord last_hW  = (((HWord)base2check) + len2check - 1)
458                               & ~(HWord)(host_word_szB-1);
459              vassert(first_hW <= last_hW);
460              HWord hW_diff = last_hW - first_hW;
461              vassert(0 == (hW_diff & (host_word_szB-1)));
462              HWord hWs_to_check = (hW_diff + host_word_szB) / host_word_szB;
463              vassert(hWs_to_check > 0
464                      && hWs_to_check < 1004/*arbitrary*/ / host_word_szB);
465     
466              /* vex_printf("%lx %lx  %ld\n", first_w32, last_w32, w32s_to_check); */
467     
468              if (host_word_szB == 8) {
469                 fn_generic =  (VEX_REGPARM(2) HWord(*)(HWord, HWord))
470                               genericg_compute_checksum_8al;
471                 nm_generic = "genericg_compute_checksum_8al";
472              } else {
473                 fn_generic =  (VEX_REGPARM(2) HWord(*)(HWord, HWord))
474                               genericg_compute_checksum_4al;
475                 nm_generic = "genericg_compute_checksum_4al";
476              }
477     
478              fn_spec = NULL;
479              nm_spec = NULL;
480     
481              if (host_word_szB == 8) {
482                 HChar* nm = NULL;
483                 ULong  VEX_REGPARM(1) (*fn)(HWord)  = NULL;
484                 switch (hWs_to_check) {
485                    case 1:  fn =  genericg_compute_checksum_8al_1;
486                             nm = "genericg_compute_checksum_8al_1"; break;
487                    case 2:  fn =  genericg_compute_checksum_8al_2;
488                             nm = "genericg_compute_checksum_8al_2"; break;
489                    case 3:  fn =  genericg_compute_checksum_8al_3;
490                             nm = "genericg_compute_checksum_8al_3"; break;
491                    case 4:  fn =  genericg_compute_checksum_8al_4;
492                             nm = "genericg_compute_checksum_8al_4"; break;
493                    case 5:  fn =  genericg_compute_checksum_8al_5;
494                             nm = "genericg_compute_checksum_8al_5"; break;
495                    case 6:  fn =  genericg_compute_checksum_8al_6;
496                             nm = "genericg_compute_checksum_8al_6"; break;
497                    case 7:  fn =  genericg_compute_checksum_8al_7;
498                             nm = "genericg_compute_checksum_8al_7"; break;
499                    case 8:  fn =  genericg_compute_checksum_8al_8;
500                             nm = "genericg_compute_checksum_8al_8"; break;
501                    case 9:  fn =  genericg_compute_checksum_8al_9;
502                             nm = "genericg_compute_checksum_8al_9"; break;
503                    case 10: fn =  genericg_compute_checksum_8al_10;
504                             nm = "genericg_compute_checksum_8al_10"; break;
505                    case 11: fn =  genericg_compute_checksum_8al_11;
506                             nm = "genericg_compute_checksum_8al_11"; break;
507                    case 12: fn =  genericg_compute_checksum_8al_12;
508                             nm = "genericg_compute_checksum_8al_12"; break;
509                    default: break;
510                 }
511                 fn_spec = (VEX_REGPARM(1) HWord(*)(HWord)) fn;
512                 nm_spec = nm;
513              } else {
514                 HChar* nm = NULL;
515                 UInt   VEX_REGPARM(1) (*fn)(HWord) = NULL;
516                 switch (hWs_to_check) {
517                    case 1:  fn =  genericg_compute_checksum_4al_1;
518                             nm = "genericg_compute_checksum_4al_1"; break;
519                    case 2:  fn =  genericg_compute_checksum_4al_2;
520                             nm = "genericg_compute_checksum_4al_2"; break;
521                    case 3:  fn =  genericg_compute_checksum_4al_3;
522                             nm = "genericg_compute_checksum_4al_3"; break;
523                    case 4:  fn =  genericg_compute_checksum_4al_4;
524                             nm = "genericg_compute_checksum_4al_4"; break;
525                    case 5:  fn =  genericg_compute_checksum_4al_5;
526                             nm = "genericg_compute_checksum_4al_5"; break;
527                    case 6:  fn =  genericg_compute_checksum_4al_6;
528                             nm = "genericg_compute_checksum_4al_6"; break;
529                    case 7:  fn =  genericg_compute_checksum_4al_7;
530                             nm = "genericg_compute_checksum_4al_7"; break;
531                    case 8:  fn =  genericg_compute_checksum_4al_8;
532                             nm = "genericg_compute_checksum_4al_8"; break;
533                    case 9:  fn =  genericg_compute_checksum_4al_9;
534                             nm = "genericg_compute_checksum_4al_9"; break;
535                    case 10: fn =  genericg_compute_checksum_4al_10;
536                             nm = "genericg_compute_checksum_4al_10"; break;
537                    case 11: fn =  genericg_compute_checksum_4al_11;
538                             nm = "genericg_compute_checksum_4al_11"; break;
539                    case 12: fn =  genericg_compute_checksum_4al_12;
540                             nm = "genericg_compute_checksum_4al_12"; break;
541                    default: break;
542                 }
543                 fn_spec = (VEX_REGPARM(1) HWord(*)(HWord))fn;
544                 nm_spec = nm;
545              }
546     
547              expectedhW = fn_generic( first_hW, hWs_to_check );
548              /* If we got a specialised version, check it produces the same
549                 result as the generic version! */
550              if (fn_spec) {
551                 vassert(nm_spec);
552                 vassert(expectedhW == fn_spec( first_hW ));
553              } else {
554                 vassert(!nm_spec);
555              }
556     
557              /* Set TISTART and TILEN.  These will describe to the despatcher
558                 the area of guest code to invalidate should we exit with a
559                 self-check failure. */
560     
561              tistart_tmp = newIRTemp(irsb->tyenv, guest_word_type);
562              tilen_tmp   = newIRTemp(irsb->tyenv, guest_word_type);
563     
564              IRConst* base2check_IRConst
565                 = guest_word_type==Ity_I32 ? IRConst_U32(toUInt(base2check))
566                                            : IRConst_U64(base2check);
567              IRConst* len2check_IRConst
568                 = guest_word_type==Ity_I32 ? IRConst_U32(len2check)
569                                            : IRConst_U64(len2check);
570     
571              irsb->stmts[selfcheck_idx + i * 5 + 0]
572                 = IRStmt_WrTmp(tistart_tmp, IRExpr_Const(base2check_IRConst) );
573     
574              irsb->stmts[selfcheck_idx + i * 5 + 1]
575                 = IRStmt_WrTmp(tilen_tmp, IRExpr_Const(len2check_IRConst) );
576     
577              irsb->stmts[selfcheck_idx + i * 5 + 2]
578                 = IRStmt_Put( offB_TISTART, IRExpr_RdTmp(tistart_tmp) );
579     
580              irsb->stmts[selfcheck_idx + i * 5 + 3]
581                 = IRStmt_Put( offB_TILEN, IRExpr_RdTmp(tilen_tmp) );
582     
583              /* Generate the entry point descriptors */
584              if (abiinfo_both->host_ppc_calls_use_fndescrs) {
585                 HWord* descr = (HWord*)fn_generic;
586                 fn_generic_entry = descr[0];
587                 if (fn_spec) {
588                    descr = (HWord*)fn_spec;
589                    fn_spec_entry = descr[0];
590                 } else {
591                    fn_spec_entry = (HWord)NULL;
592                 }
593              } else {
594                 fn_generic_entry = (HWord)fn_generic;
595                 if (fn_spec) {
596                    fn_spec_entry = (HWord)fn_spec;
597                 } else {
598                    fn_spec_entry = (HWord)NULL;
599                 }
600              }
601     
602              IRExpr* callexpr = NULL;
603              if (fn_spec) {
604                 callexpr = mkIRExprCCall( 
605                               host_word_type, 1/*regparms*/, 
606                               nm_spec, (void*)fn_spec_entry,
607                               mkIRExprVec_1(
608                                  mkIRExpr_HWord( (HWord)first_hW )
609                               )
610                            );
611              } else {
612                 callexpr = mkIRExprCCall( 
613                               host_word_type, 2/*regparms*/, 
614                               nm_generic, (void*)fn_generic_entry,
615                               mkIRExprVec_2(
616                                  mkIRExpr_HWord( (HWord)first_hW ),
617                                  mkIRExpr_HWord( (HWord)hWs_to_check )
618                               )
619                            );
620              }
621     
622              irsb->stmts[selfcheck_idx + i * 5 + 4]
623                 = IRStmt_Exit( 
624                      IRExpr_Binop( 
625                         host_word_type==Ity_I64 ? Iop_CmpNE64 : Iop_CmpNE32,
626                         callexpr,
627                            host_word_type==Ity_I64
628                               ? IRExpr_Const(IRConst_U64(expectedhW))
629                               : IRExpr_Const(IRConst_U32(expectedhW))
630                      ),
631                      Ijk_TInval,
632                      /* Where we must restart if there's a failure: at the
633                         first extent, regardless of which extent the
634                         failure actually happened in. */
635                      guest_IP_bbstart_IRConst
636                   );
637           }
638        }
639     
640        return irsb;
641     }

651     /* Compute a checksum of host memory at [addr .. addr+len-1], as fast
652        as possible.  All _4al versions assume that the supplied address is
653        4 aligned.  All length values are in 4-byte chunks.  These fns
654        arecalled once for every use of a self-checking translation, so
655        they needs to be as fast as possible. */
656     
657     /* --- 32-bit versions, used only on 32-bit hosts --- */
658     
659     static inline UInt ROL32 ( UInt w, Int n ) {
660        w = (w << n) | (w >> (32-n));
661        return w;
662     }
663     
664     VEX_REGPARM(2)
665     static UInt genericg_compute_checksum_4al ( HWord first_w32, HWord n_w32s )
666     {
667        UInt  sum1 = 0, sum2 = 0;
668        UInt* p = (UInt*)first_w32;
669        /* unrolled */
670        while (n_w32s >= 4) {
671           UInt  w;
672           w = p[0];  sum1 = ROL32(sum1 ^ w, 31);  sum2 += w;
673           w = p[1];  sum1 = ROL32(sum1 ^ w, 31);  sum2 += w;
674           w = p[2];  sum1 = ROL32(sum1 ^ w, 31);  sum2 += w;
675           w = p[3];  sum1 = ROL32(sum1 ^ w, 31);  sum2 += w;
676           p += 4;
677           n_w32s -= 4;
678           sum1 ^= sum2;
679        }
680        while (n_w32s >= 1) {
681           UInt  w;
682           w = p[0];  sum1 = ROL32(sum1 ^ w, 31);  sum2 += w;
683           p += 1;
684           n_w32s -= 1;
685           sum1 ^= sum2;
686        }
687        return sum1 + sum2;
688     }
