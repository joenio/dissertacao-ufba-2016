68     /* Returns True iff address range is something the client can
69        plausibly mess with: all of it is either already belongs to the
70        client or is free or a reservation. */
71     
72     Bool ML_(valid_client_addr)(Addr start, SizeT size, ThreadId tid,
73                                        const Char *syscallname)
74     {
75        Bool ret;
76     
77        if (size == 0)
78           return True;
79     
80        ret = VG_(am_is_valid_for_client_or_free_or_resvn)
81                 (start,size,VKI_PROT_NONE);
82     
83        if (0)
84           VG_(printf)("%s: test=%#lx-%#lx ret=%d\n",
85     		  syscallname, start, start+size-1, (Int)ret);
86     
87        if (!ret && syscallname != NULL) {
88           VG_(message)(Vg_UserMsg, "Warning: client syscall %s tried "
89                                    "to modify addresses %#lx-%#lx\n",
90                                    syscallname, start, start+size-1);
91           if (VG_(clo_verbosity) > 1) {
92              VG_(get_and_pp_StackTrace)(tid, VG_(clo_backtrace_size));
93           }
94        }
95     
96        return ret;
97     }

125     /* AFAICT from kernel sources (mm/mprotect.c) and general experimentation,
126        munmap, mprotect (and mremap??) work at the page level.  So addresses
127        and lengths must be adjusted for this. */

129     /* Mash around start and length so that the area exactly covers
130        an integral number of pages.  If we don't do that, memcheck's
131        idea of addressible memory diverges from that of the
132        kernel's, which causes the leak detector to crash. */
133     static 
134     void page_align_addr_and_len( Addr* a, SizeT* len)
135     {
136        Addr ra;
137        
138        ra = VG_PGROUNDDN(*a);
139        *len = VG_PGROUNDUP(*a + *len) - ra;
140        *a = ra;
141     }

145     {
146        Bool d;
147     
148        /* 'a' is the return value from a real kernel mmap, hence: */
149        vg_assert(VG_IS_PAGE_ALIGNED(a));

155        if (d)

238     /* Expand (or shrink) an existing mapping, potentially moving it at
239        the same time (controlled by the MREMAP_MAYMOVE flag).  Nightmare.
240     */
241     static
242     SysRes do_mremap( Addr old_addr, SizeT old_len, 
243                       Addr new_addr, SizeT new_len,
244                       UWord flags, ThreadId tid )
245     {
246     #  define MIN_SIZET(_aa,_bb) (_aa) < (_bb) ? (_aa) : (_bb)
247     
248        Bool      ok, d;
249        NSegment const* old_seg;
250        Addr      advised;
251        Bool      f_fixed   = toBool(flags & VKI_MREMAP_FIXED);
252        Bool      f_maymove = toBool(flags & VKI_MREMAP_MAYMOVE);
253     
254        if (0)
255           VG_(printf)("do_remap (old %#lx %ld) (new %#lx %ld) %s %s\n",
256                       old_addr,old_len,new_addr,new_len, 
257                       flags & VKI_MREMAP_MAYMOVE ? "MAYMOVE" : "",
258                       flags & VKI_MREMAP_FIXED ? "FIXED" : "");
259        if (0)
260           VG_(am_show_nsegments)(0, "do_remap: before");
261     
262        if (flags & ~(VKI_MREMAP_FIXED | VKI_MREMAP_MAYMOVE))
263           goto eINVAL;
264     
265        if (!VG_IS_PAGE_ALIGNED(old_addr))
266           goto eINVAL;
267     
268        old_len = VG_PGROUNDUP(old_len);
269        new_len = VG_PGROUNDUP(new_len);
270     
271        if (new_len == 0)
272           goto eINVAL;
273     
274        /* kernel doesn't reject this, but we do. */
275        if (old_len == 0)
276           goto eINVAL;
277     
278        /* reject wraparounds */
279        if (old_addr + old_len < old_addr
280            || new_addr + new_len < new_len)
281           goto eINVAL;
282     
283        /* kernel rejects all fixed, no-move requests (which are
284           meaningless). */
285        if (f_fixed == True && f_maymove == False)
286           goto eINVAL;
287     
288        /* Stay away from non-client areas. */
289        if (!ML_(valid_client_addr)(old_addr, old_len, tid, "mremap(old_addr)"))
290           goto eINVAL;
291     
292        /* In all remaining cases, if the old range does not fall within a
293           single segment, fail. */
294        old_seg = VG_(am_find_nsegment)( old_addr );
295        if (old_addr < old_seg->start || old_addr+old_len-1 > old_seg->end)
296           goto eINVAL;
297        if (old_seg->kind != SkAnonC && old_seg->kind != SkFileC)
298           goto eINVAL;
299     
300        vg_assert(old_len > 0);
301        vg_assert(new_len > 0);
302        vg_assert(VG_IS_PAGE_ALIGNED(old_len));
303        vg_assert(VG_IS_PAGE_ALIGNED(new_len));
304        vg_assert(VG_IS_PAGE_ALIGNED(old_addr));
305     
306        /* There are 3 remaining cases:
307     
308           * maymove == False
309     
310             new space has to be at old address, so:
311                 - shrink    -> unmap end
312                 - same size -> do nothing
313                 - grow      -> if can grow in-place, do so, else fail
314     
315           * maymove == True, fixed == False
316     
317             new space can be anywhere, so:
318                 - shrink    -> unmap end
319                 - same size -> do nothing
320                 - grow      -> if can grow in-place, do so, else 
321                                move to anywhere large enough, else fail
322     
323           * maymove == True, fixed == True
324     
325             new space must be at new address, so:
326     
327                 - if new address is not page aligned, fail
328                 - if new address range overlaps old one, fail
329                 - if new address range cannot be allocated, fail
330                 - else move to new address range with new size
331                 - else fail
332        */
333     
334        if (f_maymove == False) {
335           /* new space has to be at old address */
336           if (new_len < old_len)
337              goto shrink_in_place;
338           if (new_len > old_len)
339              goto grow_in_place_or_fail;
340           goto same_in_place;
341        }
342     
343        if (f_maymove == True && f_fixed == False) {
344           /* new space can be anywhere */
345           if (new_len < old_len)
346              goto shrink_in_place;
347           if (new_len > old_len)
348              goto grow_in_place_or_move_anywhere_or_fail;
349           goto same_in_place;
350        }
351     
352        if (f_maymove == True && f_fixed == True) {
353           /* new space can only be at the new address */
354           if (!VG_IS_PAGE_ALIGNED(new_addr)) 
355              goto eINVAL;
356           if (new_addr+new_len-1 < old_addr || new_addr > old_addr+old_len-1) {
357              /* no overlap */
358           } else {
359              goto eINVAL;
360           }
361           if (new_addr == 0) 
362              goto eINVAL; 
363              /* VG_(am_get_advisory_client_simple) interprets zero to mean
364                 non-fixed, which is not what we want */
365           advised = VG_(am_get_advisory_client_simple)(new_addr, new_len, &ok);
366           if (!ok || advised != new_addr)
367              goto eNOMEM;
368           ok = VG_(am_relocate_nooverlap_client)
369                   ( &d, old_addr, old_len, new_addr, new_len );
370           if (ok) {
371              VG_TRACK( copy_mem_remap, old_addr, new_addr, 
372                                        MIN_SIZET(old_len,new_len) );
373              if (new_len > old_len)
374                 VG_TRACK( new_mem_mmap, new_addr+old_len, new_len-old_len,
375                           old_seg->hasR, old_seg->hasW, old_seg->hasX,
376                           0/*di_handle*/ );
377              VG_TRACK(die_mem_munmap, old_addr, old_len);
378              if (d) {
379                 VG_(discard_translations)( old_addr, old_len, "do_remap(1)" );
380                 VG_(discard_translations)( new_addr, new_len, "do_remap(2)" );
381              }
382              return VG_(mk_SysRes_Success)( new_addr );
383           }
384           goto eNOMEM;
385        }
386     
387        /* end of the 3 cases */
388        /*NOTREACHED*/ vg_assert(0);
389     
390       grow_in_place_or_move_anywhere_or_fail: 
391        { 
392        /* try growing it in-place */
393        Addr   needA = old_addr + old_len;
394        SSizeT needL = new_len - old_len;
395     
396        vg_assert(needL > 0);
397        if (needA == 0)
398           goto eINVAL; 
399           /* VG_(am_get_advisory_client_simple) interprets zero to mean
400              non-fixed, which is not what we want */
401        advised = VG_(am_get_advisory_client_simple)( needA, needL, &ok );
402        if (ok) {
403           /* VG_(am_get_advisory_client_simple) (first arg == 0, meaning
404              this-or-nothing) is too lenient, and may allow us to trash
405              the next segment along.  So make very sure that the proposed
406              new area really is free.  This is perhaps overly
407              conservative, but it fixes #129866. */
408           NSegment const* segLo = VG_(am_find_nsegment)( needA );
409           NSegment const* segHi = VG_(am_find_nsegment)( needA + needL - 1 );
410           if (segLo == NULL || segHi == NULL 
411               || segLo != segHi || segLo->kind != SkFree)
412              ok = False;
413        }
414        if (ok && advised == needA) {
415           ok = VG_(am_extend_map_client)( &d, (NSegment*)old_seg, needL );
416           if (ok) {
417              VG_TRACK( new_mem_mmap, needA, needL, 
418                                      old_seg->hasR, 
419                                      old_seg->hasW, old_seg->hasX,
420                                      0/*di_handle*/ );
421              if (d) 
422                 VG_(discard_translations)( needA, needL, "do_remap(3)" );
423              return VG_(mk_SysRes_Success)( old_addr );
424           }
425        }
426     
427        /* that failed.  Look elsewhere. */
428        advised = VG_(am_get_advisory_client_simple)( 0, new_len, &ok );
429        if (ok) {
430           Bool oldR = old_seg->hasR;
431           Bool oldW = old_seg->hasW;
432           Bool oldX = old_seg->hasX;
433           /* assert new area does not overlap old */
434           vg_assert(advised+new_len-1 < old_addr 
435                     || advised > old_addr+old_len-1);
436           ok = VG_(am_relocate_nooverlap_client)
437                   ( &d, old_addr, old_len, advised, new_len );
438           if (ok) {
439              VG_TRACK( copy_mem_remap, old_addr, advised, 
440                                        MIN_SIZET(old_len,new_len) );
441              if (new_len > old_len)
442                 VG_TRACK( new_mem_mmap, advised+old_len, new_len-old_len,
443                           oldR, oldW, oldX, 0/*di_handle*/ );
444              VG_TRACK(die_mem_munmap, old_addr, old_len);
445              if (d) {
446                 VG_(discard_translations)( old_addr, old_len, "do_remap(4)" );
447                 VG_(discard_translations)( advised, new_len, "do_remap(5)" );
448              }
449              return VG_(mk_SysRes_Success)( advised );
450           }
451        }
452        goto eNOMEM;
453        }
454        /*NOTREACHED*/ vg_assert(0);
455     
456       grow_in_place_or_fail:
457        {
458        Addr  needA = old_addr + old_len;
459        SizeT needL = new_len - old_len;
460        if (needA == 0) 
461           goto eINVAL;
462           /* VG_(am_get_advisory_client_simple) interprets zero to mean
463              non-fixed, which is not what we want */
464        advised = VG_(am_get_advisory_client_simple)( needA, needL, &ok );
465        if (ok) {
466           /* VG_(am_get_advisory_client_simple) (first arg == 0, meaning
467              this-or-nothing) is too lenient, and may allow us to trash
468              the next segment along.  So make very sure that the proposed
469              new area really is free. */
470           NSegment const* segLo = VG_(am_find_nsegment)( needA );
471           NSegment const* segHi = VG_(am_find_nsegment)( needA + needL - 1 );
472           if (segLo == NULL || segHi == NULL 
473               || segLo != segHi || segLo->kind != SkFree)
474              ok = False;
475        }
476        if (!ok || advised != needA)
477           goto eNOMEM;
478        ok = VG_(am_extend_map_client)( &d, (NSegment*)old_seg, needL );
479        if (!ok)
480           goto eNOMEM;
481        VG_TRACK( new_mem_mmap, needA, needL, 
482                                old_seg->hasR, old_seg->hasW, old_seg->hasX,
483                                0/*di_handle*/ );
484        if (d)
485           VG_(discard_translations)( needA, needL, "do_remap(6)" );
486        return VG_(mk_SysRes_Success)( old_addr );
487        }
488        /*NOTREACHED*/ vg_assert(0);
489     
490       shrink_in_place:
491        {
492        SysRes sres = VG_(am_munmap_client)( &d, old_addr+new_len, old_len-new_len );
493        if (sr_isError(sres))
494           return sres;
495        VG_TRACK( die_mem_munmap, old_addr+new_len, old_len-new_len );
496        if (d)
497           VG_(discard_translations)( old_addr+new_len, old_len-new_len, 
498                                      "do_remap(7)" );
499        return VG_(mk_SysRes_Success)( old_addr );
500        }
501        /*NOTREACHED*/ vg_assert(0);
502     
503       same_in_place:
504        return VG_(mk_SysRes_Success)( old_addr );
505        /*NOTREACHED*/ vg_assert(0);
506     
507       eINVAL:
508        return VG_(mk_SysRes_Error)( VKI_EINVAL );
509       eNOMEM:
510        return VG_(mk_SysRes_Error)( VKI_ENOMEM );
511     
512     #  undef MIN_SIZET
513     }

521     /* One of these is allocated for each open file descriptor.  */
522     typedef struct OpenFd
523     {
524        Int fd;                        /* The file descriptor */
525        Char *pathname;                /* NULL if not a regular file or unknown */
526        ExeContext *where;             /* NULL if inherited from parent */
527        struct OpenFd *next, *prev;
528     } OpenFd;
529     
530     /* List of allocated file descriptors. */
531     static OpenFd *allocated_fds = NULL;
532     
533     /* Count of open file descriptors. */
534     static Int fd_count = 0;
535     
536     
537     /* Note the fact that a file descriptor was just closed. */
538     static
539     void record_fd_close(Int fd)
540     {
541        OpenFd *i = allocated_fds;
542     
543        if (fd >= VG_(fd_hard_limit))
544           return;			/* Valgrind internal */
545     
546        while(i) {
547           if(i->fd == fd) {
548              if(i->prev)
549                 i->prev->next = i->next;
550              else
551                 allocated_fds = i->next;
552              if(i->next)
553                 i->next->prev = i->prev;
554              if(i->pathname) 
555                 VG_(arena_free) (VG_AR_CORE, i->pathname);
556              VG_(arena_free) (VG_AR_CORE, i);
557              fd_count--;
558              break;
559           }
560           i = i->next;
561        }
562     }
563     
564     /* Note the fact that a file descriptor was just opened.  If the
565        tid is -1, this indicates an inherited fd.  If the pathname is NULL,
566        this either indicates a non-standard file (i.e. a pipe or socket or
567        some such thing) or that we don't know the filename.  If the fd is
568        already open, then we're probably doing a dup2() to an existing fd,
569        so just overwrite the existing one. */
570     void ML_(record_fd_open_with_given_name)(ThreadId tid, Int fd, char *pathname)
571     {
572        OpenFd *i;
573     
574        if (fd >= VG_(fd_hard_limit))
575           return;			/* Valgrind internal */
576     
577        /* Check to see if this fd is already open. */
578        i = allocated_fds;
579        while (i) {
580           if (i->fd == fd) {
581              if (i->pathname) VG_(arena_free)(VG_AR_CORE, i->pathname);
582              break;
583           }
584           i = i->next;
585        }
586     
587        /* Not already one: allocate an OpenFd */
588        if (i == NULL) {
589           i = VG_(arena_malloc)(VG_AR_CORE, "syswrap.rfdowgn.1", sizeof(OpenFd));
590     
591           i->prev = NULL;
592           i->next = allocated_fds;
593           if(allocated_fds) allocated_fds->prev = i;

743        have /proc support compiled in, or a non-Linux kernel), then we need to
744        find out what file descriptors we inherited from our parent process the

1021     /*   +--------+------------+
1022          | anon   |    resvn   |
1023          +--------+------------+
1024     
1025          ^     ^  ^
1026          |     |  boundary is page aligned
1027          |     VG_(brk_limit) -- no alignment constraint
1028          VG_(brk_base) -- page aligned -- does not move
1029     
1030          Both the anon part and the reservation part are always at least
1031          one page.  
1032     */
1033     
1034     /* Set the new data segment end to NEWBRK.  If this succeeds, return
1035        NEWBRK, else return the current data segment end. */
1036     
1037     static Addr do_brk ( Addr newbrk )
1038     {
1039        NSegment const* aseg;
1040        NSegment const* rseg;
1041        Addr newbrkP;
1042        SizeT delta;
1043        Bool ok;
1044        Bool debug = False;
1045     
1046        if (debug)
1047           VG_(printf)("\ndo_brk: brk_base=%#lx brk_limit=%#lx newbrk=%#lx\n",
1048     		  VG_(brk_base), VG_(brk_limit), newbrk);
1049     
1050     #  if 0
1051        if (0) show_segments("in_brk");
1052     #  endif
1053     
1054        if (newbrk < VG_(brk_base))
1055           /* Clearly impossible. */
1056           goto bad;
1057     
1058        if (newbrk >= VG_(brk_base) && newbrk < VG_(brk_limit)) {
1059           /* shrinking the data segment.  Be lazy and don't munmap the
1060              excess area. */
1061           NSegment const * seg = VG_(am_find_nsegment)(newbrk);
1062           if (seg && seg->hasT)
1063              VG_(discard_translations)( newbrk, VG_(brk_limit) - newbrk, 
1064                                         "do_brk(shrink)" );
1065           /* Since we're being lazy and not unmapping pages, we have to
1066              zero out the area, so that if the area later comes back into
1067              circulation, it will be filled with zeroes, as if it really
1068              had been unmapped and later remapped.  Be a bit paranoid and
1069              try hard to ensure we're not going to segfault by doing the
1070              write - check both ends of the range are in the same segment
1071              and that segment is writable. */
1072           if (seg) {
1073              /* pre: newbrk < VG_(brk_limit) 
1074                   => newbrk <= VG_(brk_limit)-1 */
1075              NSegment const * seg2;
1076              vg_assert(newbrk < VG_(brk_limit));
1077              seg2 = VG_(am_find_nsegment)( VG_(brk_limit)-1 );
1078              if (seg2 && seg == seg2 && seg->hasW)
1079                 VG_(memset)( (void*)newbrk, 0, VG_(brk_limit) - newbrk );
1080           }
1081     
1082           VG_(brk_limit) = newbrk;
1083           return newbrk;
1084        }
1085     
1086        /* otherwise we're expanding the brk segment. */
1087        if (VG_(brk_limit) > VG_(brk_base))
1088           aseg = VG_(am_find_nsegment)( VG_(brk_limit)-1 );
1089        else
1090           aseg = VG_(am_find_nsegment)( VG_(brk_limit) );
1091        rseg = VG_(am_next_nsegment)( (NSegment*)aseg, True/*forwards*/ );
1092     
1093        /* These should be assured by setup_client_dataseg in m_main. */
1094        vg_assert(aseg);
1095        vg_assert(rseg);
1096        vg_assert(aseg->kind == SkAnonC);
1097        vg_assert(rseg->kind == SkResvn);
1098        vg_assert(aseg->end+1 == rseg->start);
1099     
1100        vg_assert(newbrk >= VG_(brk_base));
1101        if (newbrk <= rseg->start) {
1102           /* still fits within the anon segment. */
1103           VG_(brk_limit) = newbrk;
1104           return newbrk;
1105        }
1106     
1107        if (newbrk > rseg->end+1 - VKI_PAGE_SIZE) {
1108           /* request is too large -- the resvn would fall below 1 page,
1109              which isn't allowed. */
1110           goto bad;
1111        }
1112     
1113        newbrkP = VG_PGROUNDUP(newbrk);
1114        vg_assert(newbrkP > rseg->start && newbrkP <= rseg->end+1 - VKI_PAGE_SIZE);
1115        delta = newbrkP - rseg->start;
1116        vg_assert(delta > 0);
1117        vg_assert(VG_IS_PAGE_ALIGNED(delta));
1118        
1119        ok = VG_(am_extend_into_adjacent_reservation_client)( (NSegment*)aseg, delta );
1120        if (!ok) goto bad;
1121     
1122        VG_(brk_limit) = newbrk;
1123        return newbrk;
1124     
1125       bad:
1126        return VG_(brk_limit);
1127     }

1133     /* 
1134     > - what does the "Bool soft" parameter mean?
1135     
1136     (Tom Hughes, 3 Oct 05):
1137     
1138     Whether or not to consider a file descriptor invalid if it is above
1139     the current soft limit.
1140     
1141     Basically if we are testing whether a newly created file descriptor is
1142     valid (in a post handler) then we set soft to true, and if we are
1143     testing whether a file descriptor that is about to be used (in a pre
1144     handler) is valid [viz, an already-existing fd] then we set it to false.
1145     
1146     The point is that if the (virtual) soft limit is lowered then any
1147     existing descriptors can still be read/written/closed etc (so long as
1148     they are below the valgrind reserved descriptors) but no new
1149     descriptors can be created above the new soft limit.
1150     
1151     (jrs 4 Oct 05: in which case, I've renamed it "isNewFd")
1152     */

1154     /* Return true if we're allowed to use or create this fd */
1155     Bool ML_(fd_allowed)(Int fd, const Char *syscallname, ThreadId tid, Bool isNewFd)
1156     {
1157        Bool allowed = True;
1158     
1159        /* hard limits always apply */
1160        if (fd < 0 || fd >= VG_(fd_hard_limit))
1161           allowed = False;
1162     
1163        /* hijacking the output fds is never allowed */
1164        if (fd == VG_(log_output_sink).fd || fd == VG_(xml_output_sink).fd)
1165           allowed = False;
1166     
1167        /* if creating a new fd (rather than using an existing one), the
1168           soft limit must also be observed */
1169        if (isNewFd && fd >= VG_(fd_soft_limit))
1170           allowed = False;
1171     
1172        /* this looks like it ought to be included, but causes problems: */
1173        /*
1174        if (fd == 2 && VG_(debugLog_getLevel)() > 0)
1175           allowed = False;
1176        */
1177        /* The difficulty is as follows: consider a program P which expects
1178           to be able to mess with (redirect) its own stderr (fd 2).
1179           Usually to deal with P we would issue command line flags to send
1180           logging somewhere other than stderr, so as not to disrupt P.
1181           The problem is that -d unilaterally hijacks stderr with no
1182           consultation with P.  And so, if this check is enabled, P will
1183           work OK normally but fail if -d is issued.
1184     
1185           Basically -d is a hack and you take your chances when using it.
1186           It's very useful for low level debugging -- particularly at
1187           startup -- and having its presence change the behaviour of the
1188           client is exactly what we don't want.  */
1189     
1190        /* croak? */
1191        if ((!allowed) && VG_(showing_core_errors)() ) {
1192           VG_(message)(Vg_UserMsg, 
1193              "Warning: invalid file descriptor %d in syscall %s()\n",
1194              fd, syscallname);
1195           if (fd == VG_(log_output_sink).fd && VG_(log_output_sink).fd >= 0)
1196     	 VG_(message)(Vg_UserMsg, 
1197                 "   Use --log-fd=<number> to select an alternative log fd.\n");
1198           if (fd == VG_(xml_output_sink).fd && VG_(xml_output_sink).fd >= 0)
1199     	 VG_(message)(Vg_UserMsg, 
1200                 "   Use --xml-fd=<number> to select an alternative XML "
1201                 "output fd.\n");
1202           // DDD: consider always printing this stack trace, it's useful.
1203           // Also consider also making this a proper core error, ie.
1204           // suppressible and all that.
1205           if (VG_(clo_verbosity) > 1) {
1206              VG_(get_and_pp_StackTrace)(tid, VG_(clo_backtrace_size));
1207           }
1208        }
1209     
1210        return allowed;
1211     }

1735     UWord
1736     ML_(generic_PRE_sys_shmat) ( ThreadId tid,
1737                                  UWord arg0, UWord arg1, UWord arg2 )
1738     {
1739        /* void *shmat(int shmid, const void *shmaddr, int shmflg); */
1740        UInt  segmentSize = get_shm_size ( arg0 );
1741        UWord tmp;
1742        Bool  ok;
1743        if (arg1 == 0) {
1744           /* arm-linux only: work around the fact that
1745              VG_(am_get_advisory_client_simple) produces something that is
1746              VKI_PAGE_SIZE aligned, whereas what we want is something
1747              VKI_SHMLBA aligned, and VKI_SHMLBA >= VKI_PAGE_SIZE.  Hence
1748              increase the request size by VKI_SHMLBA - VKI_PAGE_SIZE and
1749              then round the result up to the next VKI_SHMLBA boundary.
1750              See bug 222545 comment 15.  So far, arm-linux is the only
1751              platform where this is known to be necessary. */
1752           vg_assert(VKI_SHMLBA >= VKI_PAGE_SIZE);
1753           if (VKI_SHMLBA > VKI_PAGE_SIZE) {
1754              segmentSize += VKI_SHMLBA - VKI_PAGE_SIZE;
1755           }
1756           tmp = VG_(am_get_advisory_client_simple)(0, segmentSize, &ok);

1939        Generic handler for mmap
1940        ------------------------------------------------------------------ */
1941     
1942     /*
1943      * Although mmap is specified by POSIX and the argument are generally
1944      * consistent across platforms the precise details of the low level
1945      * argument passing conventions differ. For example:
1946      *
1947      * - On x86-linux there is mmap (aka old_mmap) which takes the
1948      *   arguments in a memory block and the offset in bytes; and
1949      *   mmap2 (aka sys_mmap2) which takes the arguments in the normal
1950      *   way and the offset in pages.
1951      *
1952      * - On ppc32-linux there is mmap (aka sys_mmap) which takes the
1953      *   arguments in the normal way and the offset in bytes; and
1954      *   mmap2 (aka sys_mmap2) which takes the arguments in the normal
1955      *   way and the offset in pages.
1956      *
1957      * - On amd64-linux everything is simple and there is just the one
1958      *   call, mmap (aka sys_mmap)  which takes the arguments in the
1959      *   normal way and the offset in bytes.
1960      *
1961      * - On s390x-linux there is mmap (aka old_mmap) which takes the
1962      *   arguments in a memory block and the offset in bytes. mmap2
1963      *   is also available (but not exported via unistd.h) with
1964      *   arguments in a memory block and the offset in pages.
1965      *
1966      * To cope with all this we provide a generic handler function here
1967      * and then each platform implements one or more system call handlers
1968      * which call this generic routine after extracting and normalising
1969      * the arguments.
1970      */
1971     
1972     SysRes
1973     ML_(generic_PRE_sys_mmap) ( ThreadId tid,
1974                                 UWord arg1, UWord arg2, UWord arg3,
1975                                 UWord arg4, UWord arg5, Off64T arg6 )
1976     {
1977        Addr       advised;
1978        SysRes     sres;
1979        MapRequest mreq;
1980        Bool       mreq_ok;
1981     
1982     #if defined(VGO_darwin)
1983        // Nb: we can't use this on Darwin, it has races:
1984        // * needs to RETRY if advisory succeeds but map fails  
1985        //   (could have been some other thread in a nonblocking call)
1986        // * needs to not use fixed-position mmap() on Darwin
1987        //   (mmap will cheerfully smash whatever's already there, which might 
1988        //   be a new mapping from some other thread in a nonblocking call)
1989        VG_(core_panic)("can't use ML_(generic_PRE_sys_mmap) on Darwin");
1990     #endif
1991     
1992        if (arg2 == 0) {
1993           /* SuSV3 says: If len is zero, mmap() shall fail and no mapping
1994              shall be established. */
1995           return VG_(mk_SysRes_Error)( VKI_EINVAL );
1996        }
1997     
1998        if (!VG_IS_PAGE_ALIGNED(arg1)) {
1999           /* zap any misaligned addresses. */
2000           /* SuSV3 says misaligned addresses only cause the MAP_FIXED case
2001              to fail.   Here, we catch them all. */
2002           return VG_(mk_SysRes_Error)( VKI_EINVAL );
2003        }
2004     
2005        if (!VG_IS_PAGE_ALIGNED(arg6)) {
2006           /* zap any misaligned offsets. */
2007           /* SuSV3 says: The off argument is constrained to be aligned and
2008              sized according to the value returned by sysconf() when
2009              passed _SC_PAGESIZE or _SC_PAGE_SIZE. */
2010           return VG_(mk_SysRes_Error)( VKI_EINVAL );
2011        }
2012     
2013        /* Figure out what kind of allocation constraints there are
2014           (fixed/hint/any), and ask aspacem what we should do. */
2015        mreq.start = arg1;
2016        mreq.len   = arg2;
2017        if (arg4 & VKI_MAP_FIXED) {
2018           mreq.rkind = MFixed;
2019        } else
2020        if (arg1 != 0) {
2021           mreq.rkind = MHint;
2022        } else {
2023           mreq.rkind = MAny;
2024        }
2025     
2026        /* Enquire ... */
2027        advised = VG_(am_get_advisory)( &mreq, True/*client*/, &mreq_ok );
2028        if (!mreq_ok) {
2029           /* Our request was bounced, so we'd better fail. */
2030           return VG_(mk_SysRes_Error)( VKI_EINVAL );
2031        }
2032     
2033        /* Otherwise we're OK (so far).  Install aspacem's choice of
2034           address, and let the mmap go through.  */
2035        sres = VG_(am_do_mmap_NO_NOTIFY)(advised, arg2, arg3,
2036                                         arg4 | VKI_MAP_FIXED,
2037                                         arg5, arg6);
2038     
2039        /* A refinement: it may be that the kernel refused aspacem's choice
2040           of address.  If we were originally asked for a hinted mapping,
2041           there is still a last chance: try again at any address.
2042           Hence: */
2043        if (mreq.rkind == MHint && sr_isError(sres)) {
2044           mreq.start = 0;
2045           mreq.len   = arg2;
2046           mreq.rkind = MAny;
2047           advised = VG_(am_get_advisory)( &mreq, True/*client*/, &mreq_ok );
2048           if (!mreq_ok) {
2049              /* Our request was bounced, so we'd better fail. */
2050              return VG_(mk_SysRes_Error)( VKI_EINVAL );
2051           }
2052           /* and try again with the kernel */
2053           sres = VG_(am_do_mmap_NO_NOTIFY)(advised, arg2, arg3,
2054                                            arg4 | VKI_MAP_FIXED,
2055                                            arg5, arg6);
2056        }
2057     
2058        if (!sr_isError(sres)) {
2059           ULong di_handle;
2060           /* Notify aspacem. */
2061           notify_core_of_mmap(
2062              (Addr)sr_Res(sres), /* addr kernel actually assigned */
2063              arg2, /* length */
2064              arg3, /* prot */
2065              arg4, /* the original flags value */
2066              arg5, /* fd */
2067              arg6  /* offset */
2068           );
2069           /* Load symbols? */
2070           di_handle = VG_(di_notify_mmap)( (Addr)sr_Res(sres), 
2071                                            False/*allow_SkFileV*/ );
2072           /* Notify the tool. */
2073           notify_tool_of_mmap(
2074              (Addr)sr_Res(sres), /* addr kernel actually assigned */
2075              arg2, /* length */
2076              arg3, /* prot */
2077              di_handle /* so the tool can refer to the read debuginfo later,
2078                           if it wants. */
2079           );
2080        }
2081     
2082        /* Stay sane */
2083        if (!sr_isError(sres) && (arg4 & VKI_MAP_FIXED))
2084           vg_assert(sr_Res(sres) == arg1);
2085     
2086        return sres;
2087     }

2518     // XXX: prototype here seemingly doesn't match the prototype for i386-linux,
2519     // but it seems to work nonetheless...
2520     PRE(sys_execve)
2521     {
2522        Char*        path = NULL;       /* path to executable */
2523        Char**       envp = NULL;
2524        Char**       argv = NULL;
2525        Char**       arg2copy;
2526        Char*        launcher_basename = NULL;
2527        ThreadState* tst;
2528        Int          i, j, tot_args;
2529        SysRes       res;
2530        Bool         setuid_allowed, trace_this_child;
2531     
2532        PRINT("sys_execve ( %#lx(%s), %#lx, %#lx )", ARG1, (char*)ARG1, ARG2, ARG3);
2533        PRE_REG_READ3(vki_off_t, "execve",
2534                      char *, filename, char **, argv, char **, envp);
2535        PRE_MEM_RASCIIZ( "execve(filename)", ARG1 );
2536        if (ARG2 != 0)
2537           pre_argv_envp( ARG2, tid, "execve(argv)", "execve(argv[i])" );
2538        if (ARG3 != 0)
2539           pre_argv_envp( ARG3, tid, "execve(envp)", "execve(envp[i])" );
2540     
2541        vg_assert(VG_(is_valid_tid)(tid));
2542        tst = VG_(get_ThreadState)(tid);
2543     
2544        /* Erk.  If the exec fails, then the following will have made a
2545           mess of things which makes it hard for us to continue.  The
2546           right thing to do is piece everything together again in
2547           POST(execve), but that's close to impossible.  Instead, we make
2548           an effort to check that the execve will work before actually
2549           doing it. */
2550     
2551        /* Check that the name at least begins in client-accessible storage. */
2552        if (ARG1 == 0 /* obviously bogus */
2553            || !VG_(am_is_valid_for_client)( ARG1, 1, VKI_PROT_READ )) {
2554           SET_STATUS_Failure( VKI_EFAULT );
2555           return;
2556        }
2557     
2558        // debug-only printing
2559        if (0) {
2560           VG_(printf)("ARG1 = %p(%s)\n", (void*)ARG1, (HChar*)ARG1);
2561           if (ARG2) {
2562              VG_(printf)("ARG2 = ");
2563              Int q;
2564              HChar** vec = (HChar**)ARG2;
2565              for (q = 0; vec[q]; q++)
2566                 VG_(printf)("%p(%s) ", vec[q], vec[q]);
2567              VG_(printf)("\n");
2568           } else {
2569              VG_(printf)("ARG2 = null\n");
2570           }
2571        }
2572     
2573        // Decide whether or not we want to follow along
2574        { // Make 'child_argv' be a pointer to the child's arg vector
2575          // (skipping the exe name)
2576          HChar** child_argv = (HChar**)ARG2;
2577          if (child_argv && child_argv[0] == NULL)
2578             child_argv = NULL;
2579          trace_this_child = VG_(should_we_trace_this_child)( (HChar*)ARG1, child_argv );
2580        }
2581     
2582        // Do the important checks:  it is a file, is executable, permissions are
2583        // ok, etc.  We allow setuid executables to run only in the case when
2584        // we are not simulating them, that is, they to be run natively.
2585        setuid_allowed = trace_this_child  ? False  : True;
2586        res = VG_(pre_exec_check)((const Char*)ARG1, NULL, setuid_allowed);
2587        if (sr_isError(res)) {
2588           SET_STATUS_Failure( sr_Err(res) );
2589           return;
2590        }
2591     
2592        /* If we're tracing the child, and the launcher name looks bogus
2593           (possibly because launcher.c couldn't figure it out, see
2594           comments therein) then we have no option but to fail. */
2595        if (trace_this_child 
2596            && (VG_(name_of_launcher) == NULL
2597                || VG_(name_of_launcher)[0] != '/')) {
2598           SET_STATUS_Failure( VKI_ECHILD ); /* "No child processes" */
2599           return;
2600        }
2601     
2602        /* After this point, we can't recover if the execve fails. */
2603        VG_(debugLog)(1, "syswrap", "Exec of %s\n", (Char*)ARG1);
2604     
2605        /* Resistance is futile.  Nuke all other threads.  POSIX mandates
2606           this. (Really, nuke them all, since the new process will make
2607           its own new thread.) */
2608        VG_(nuke_all_threads_except)( tid, VgSrc_ExitThread );
2609        VG_(reap_threads)(tid);
2610     
2611        // Set up the child's exe path.
2612        //
2613        if (trace_this_child) {
2614     
2615           // We want to exec the launcher.  Get its pre-remembered path.
2616           path = VG_(name_of_launcher);
2617           // VG_(name_of_launcher) should have been acquired by m_main at
2618           // startup.
2619           vg_assert(path);
2620     
2621           launcher_basename = VG_(strrchr)(path, '/');
2622           if (launcher_basename == NULL || launcher_basename[1] == 0) {
2623              launcher_basename = path;  // hmm, tres dubious
2624           } else {
2625              launcher_basename++;
2626           }
2627     
2628        } else {
2629           path = (Char*)ARG1;
2630        }
2631     
2632        // Set up the child's environment.
2633        //
2634        // Remove the valgrind-specific stuff from the environment so the
2635        // child doesn't get vgpreload_core.so, vgpreload_<tool>.so, etc.  
2636        // This is done unconditionally, since if we are tracing the child,
2637        // the child valgrind will set up the appropriate client environment.
2638        // Nb: we make a copy of the environment before trying to mangle it
2639        // as it might be in read-only memory (this was bug #101881).
2640        //
2641        // Then, if tracing the child, set VALGRIND_LIB for it.
2642        //
2643        if (ARG3 == 0) {
2644           envp = NULL;
2645        } else {
2646           envp = VG_(env_clone)( (Char**)ARG3 );
2647           if (envp == NULL) goto hosed;
2648           VG_(env_remove_valgrind_env_stuff)( envp );
2649        }
2650     
2651        if (trace_this_child) {
2652           // Set VALGRIND_LIB in ARG3 (the environment)
2653           VG_(env_setenv)( &envp, VALGRIND_LIB, VG_(libdir));
2654        }
2655     
2656        // Set up the child's args.  If not tracing it, they are
2657        // simply ARG2.  Otherwise, they are
2658        //
2659        // [launcher_basename] ++ VG_(args_for_valgrind) ++ [ARG1] ++ ARG2[1..]
2660        //
2661        // except that the first VG_(args_for_valgrind_noexecpass) args
2662        // are omitted.
2663        //
2664        if (!trace_this_child) {
2665           argv = (Char**)ARG2;
2666        } else {
2667           vg_assert( VG_(args_for_valgrind) );
2668           vg_assert( VG_(args_for_valgrind_noexecpass) >= 0 );
2669           vg_assert( VG_(args_for_valgrind_noexecpass) 
2670                        <= VG_(sizeXA)( VG_(args_for_valgrind) ) );
2671           /* how many args in total will there be? */
2672           // launcher basename
2673           tot_args = 1;
2674           // V's args
2675           tot_args += VG_(sizeXA)( VG_(args_for_valgrind) );
2676           tot_args -= VG_(args_for_valgrind_noexecpass);
2677           // name of client exe
2678           tot_args++;
2679           // args for client exe, skipping [0]
2680           arg2copy = (Char**)ARG2;
2681           if (arg2copy && arg2copy[0]) {
2682              for (i = 1; arg2copy[i]; i++)
2683                 tot_args++;
2684           }
2685           // allocate
2686           argv = VG_(malloc)( "di.syswrap.pre_sys_execve.1",
2687                               (tot_args+1) * sizeof(HChar*) );
2688           if (argv == 0) goto hosed;
2689           // copy
2690           j = 0;
2691           argv[j++] = launcher_basename;
2692           for (i = 0; i < VG_(sizeXA)( VG_(args_for_valgrind) ); i++) {
2693              if (i < VG_(args_for_valgrind_noexecpass))
2694                 continue;
2695              argv[j++] = * (HChar**) VG_(indexXA)( VG_(args_for_valgrind), i );
2696           }
2697           argv[j++] = (Char*)ARG1;
2698           if (arg2copy && arg2copy[0])
2699              for (i = 1; arg2copy[i]; i++)
2700                 argv[j++] = arg2copy[i];
2701           argv[j++] = NULL;
2702           // check
2703           vg_assert(j == tot_args+1);
2704        }
2705     
2706        /* restore the DATA rlimit for the child */
2707        VG_(setrlimit)(VKI_RLIMIT_DATA, &VG_(client_rlimit_data));
2708     
2709        /*
2710           Set the signal state up for exec.
2711     
2712           We need to set the real signal state to make sure the exec'd
2713           process gets SIG_IGN properly.
2714     
2715           Also set our real sigmask to match the client's sigmask so that
2716           the exec'd child will get the right mask.  First we need to
2717           clear out any pending signals so they they don't get delivered,
2718           which would confuse things.
2719     
2720           XXX This is a bug - the signals should remain pending, and be
2721           delivered to the new process after exec.  There's also a
2722           race-condition, since if someone delivers us a signal between
2723           the sigprocmask and the execve, we'll still get the signal. Oh
2724           well.
2725        */
2726        {
2727           vki_sigset_t allsigs;
2728           vki_siginfo_t info;
2729     
2730           /* What this loop does: it queries SCSS (the signal state that
2731              the client _thinks_ the kernel is in) by calling
2732              VG_(do_sys_sigaction), and modifies the real kernel signal
2733              state accordingly. */
2734           for (i = 1; i < VG_(max_signal); i++) {
2735              vki_sigaction_fromK_t sa_f;
2736              vki_sigaction_toK_t   sa_t;
2737              VG_(do_sys_sigaction)(i, NULL, &sa_f);
2738              VG_(convert_sigaction_fromK_to_toK)(&sa_f, &sa_t);
2739              if (sa_t.ksa_handler == VKI_SIG_IGN)
2740                 VG_(sigaction)(i, &sa_t, NULL);
2741              else {
2742                 sa_t.ksa_handler = VKI_SIG_DFL;
2743                 VG_(sigaction)(i, &sa_t, NULL);
2744              }
2745           }
2746     
2747           VG_(sigfillset)(&allsigs);
2748           while(VG_(sigtimedwait_zero)(&allsigs, &info) > 0)
2749              ;
2750     
2751           VG_(sigprocmask)(VKI_SIG_SETMASK, &tst->sig_mask, NULL);
2752        }
2753     
2754        if (0) {
2755           Char **cpp;
2756           VG_(printf)("exec: %s\n", path);
2757           for (cpp = argv; cpp && *cpp; cpp++)
2758              VG_(printf)("argv: %s\n", *cpp);
2759           if (0)
2760              for (cpp = envp; cpp && *cpp; cpp++)
2761                 VG_(printf)("env: %s\n", *cpp);
2762        }
2763     
2764        SET_STATUS_from_SysRes( 
2765           VG_(do_syscall3)(__NR_execve, (UWord)path, (UWord)argv, (UWord)envp) 
2766        );
2767     
2768        /* If we got here, then the execve failed.  We've already made way
2769           too much of a mess to continue, so we have to abort. */
2770       hosed:
2771        vg_assert(FAILURE);
2772        VG_(message)(Vg_UserMsg, "execve(%#lx(%s), %#lx, %#lx) failed, errno %ld\n",
2773                     ARG1, (char*)ARG1, ARG2, ARG3, ERR);
2774        VG_(message)(Vg_UserMsg, "EXEC FAILED: I can't recover from "
2775                                 "execve() failing, so I'm dying.\n");
2776        VG_(message)(Vg_UserMsg, "Add more stringent tests in PRE(sys_execve), "
2777                                 "or work out how to recover.\n");
2778        VG_(exit)(101);
2779     }

2794     PRE(sys_brk)
2795     {
2796        Addr brk_limit = VG_(brk_limit);
2797        Addr brk_new; 
2798     
2799        /* libc   says: int   brk(void *end_data_segment);
2800           kernel says: void* brk(void* end_data_segment);  (more or less)
2801     
2802           libc returns 0 on success, and -1 (and sets errno) on failure.
2803           Nb: if you ask to shrink the dataseg end below what it
2804           currently is, that always succeeds, even if the dataseg end
2805           doesn't actually change (eg. brk(0)).  Unless it seg faults.
2806     
2807           Kernel returns the new dataseg end.  If the brk() failed, this
2808           will be unchanged from the old one.  That's why calling (kernel)
2809           brk(0) gives the current dataseg end (libc brk() just returns
2810           zero in that case).
2811     
2812           Both will seg fault if you shrink it back into a text segment.
2813        */
2814        PRINT("sys_brk ( %#lx )", ARG1);
2815        PRE_REG_READ1(unsigned long, "brk", unsigned long, end_data_segment);
2816     
2817        brk_new = do_brk(ARG1);
2818        SET_STATUS_Success( brk_new );
2819     
2820        if (brk_new == ARG1) {
2821           /* brk() succeeded */
2822           if (brk_new < brk_limit) {
2823              /* successfully shrunk the data segment. */
2824              VG_TRACK( die_mem_brk, (Addr)ARG1,
2825     		   brk_limit-ARG1 );
2826           } else
2827           if (brk_new > brk_limit) {
2828              /* successfully grew the data segment */
2829              VG_TRACK( new_mem_brk, brk_limit,
2830                        ARG1-brk_limit, tid );
2831           }
2832        } else {
2833           /* brk() failed */
2834           vg_assert(brk_limit == brk_new);
2835        }
2836     }

3344     /* 
3345        If we're sending a SIGKILL to one of our own threads, then simulate
3346        it rather than really sending the signal, so that the target thread
3347        gets a chance to clean up.  Returns True if we did the killing (or
3348        no killing is necessary), and False if the caller should use the
3349        normal kill syscall.
3350        
3351        "pid" is any pid argument which can be passed to kill; group kills
3352        (< -1, 0), and owner kills (-1) are ignored, on the grounds that
3353        they'll most likely hit all the threads and we won't need to worry
3354        about cleanup.  In truth, we can't fully emulate these multicast
3355        kills.
3356     
3357        "tgid" is a thread group id.  If it is not -1, then the target
3358        thread must be in that thread group.
3359      */
3360     Bool ML_(do_sigkill)(Int pid, Int tgid)
3361     {
3362        ThreadState *tst;
3363        ThreadId tid;
3364     
3365        if (pid <= 0)
3366           return False;
3367     
3368        tid = VG_(lwpid_to_vgtid)(pid);
3369        if (tid == VG_INVALID_THREADID)
3370           return False;		/* none of our threads */
3371     
3372        tst = VG_(get_ThreadState)(tid);
3373        if (tst == NULL || tst->status == VgTs_Empty)
3374           return False;		/* hm, shouldn't happen */
3375     
3376        if (tgid != -1 && tst->os_state.threadgroup != tgid)
3377           return False;		/* not the right thread group */
3378     
3379        /* Check to see that the target isn't already exiting. */
3380        if (!VG_(is_exiting)(tid)) {
3381           if (VG_(clo_trace_signals))
3382     	 VG_(message)(Vg_DebugMsg,
3383                           "Thread %d being killed with SIGKILL\n", 
3384                           tst->tid);
3385           
3386           tst->exitreason = VgSrc_FatalSig;
3387           tst->os_state.fatalsig = VKI_SIGKILL;
3388           
3389           if (!VG_(is_running_thread)(tid))
3390     	 VG_(get_thread_out_of_syscall)(tid);
3391        }
3392        
3393        return True;
3394     }

3396     PRE(sys_kill)
3397     {
3398        PRINT("sys_kill ( %ld, %ld )", ARG1,ARG2);
3399        PRE_REG_READ2(long, "kill", int, pid, int, sig);
3400        if (!ML_(client_signal_OK)(ARG2)) {
3401           SET_STATUS_Failure( VKI_EINVAL );
3402           return;
3403        }
3404     
3405        /* If we're sending SIGKILL, check to see if the target is one of
3406           our threads and handle it specially. */
3407        if (ARG2 == VKI_SIGKILL && ML_(do_sigkill)(ARG1, -1))
3408           SET_STATUS_Success(0);
3409        else
3410           /* re syscall3: Darwin has a 3rd arg, which is a flag (boolean)
3411              affecting how posix-compliant the call is.  I guess it is
3412              harmless to pass the 3rd arg on other platforms; hence pass
3413              it on all. */
3414           SET_STATUS_from_SysRes( VG_(do_syscall3)(SYSNO, ARG1, ARG2, ARG3) );
3415     
3416        if (VG_(clo_trace_signals))
3417           VG_(message)(Vg_DebugMsg, "kill: sent signal %ld to pid %ld\n",
3418     		   ARG2, ARG1);
3419
