118     /* Simple binary search tree for sorting ranges */
119     typedef struct RTreeElem RTreeElem;
120     struct RTreeElem 
121     {
122        ULong       min;
123        ULong       max;
124        RTreeElem*  left;
125        RTreeElem*  right;
126     };
127     
128     static RTreeElem* create_range_node(ULong min, ULong max) 
129     {   
130        RTreeElem* node = VG_(malloc)("it_clo.c: create_range_node", 
131                                      sizeof(RTreeElem));
132        
133        node->min   = min;
134        node->max   = max;
135        node->left  = NULL;
136        node->right = NULL;
137        
138        return node;
139     }
140     
141     /* Deletes a node and returns the address of the element that takes  the
142        deleted node's place */
143     static RTreeElem*
144     delete_range_node(RTreeElem* node, 
145                       RTreeElem* parent /* NULL = delete root */) 
146     {
147        RTreeElem *new, *cur;
148          
149        if(node->right != NULL) {
150           new = node->right;
151           cur = new;
152           
153           // Find lowest 'corner' of right subtree...   
154           while(cur->left)
155              cur = cur->left;
156     
157           // ...and link in left subtree
158           cur->left = node->left;
159        } else if(node->left != NULL)
160           new = node->left;
161        else
162           new = NULL;
163     
164        // Unlink node, set parent to point to right subtree
165        if(parent != NULL) {
166           if(parent->left == node)
167              parent->left = new;
168           else
169              parent->right = new;
170        }
171        
172        // Delete node
173        VG_(free)(node);
174        
175        return new;
176     }

178     /* Inserts a range into the binary sort tree. In case of overlap with an
179        existing range, the existing range is removed and the union of that range
180        and the new one is re-added. This may be repeated several times to
181        achive correct ordering */
182     static void insert_range_node(RTreeElem** tree, ULong min, ULong max) 
183     {
184        RTreeElem* current = *tree; // <-- root node
185        RTreeElem* old = NULL;
186        
187        while(True) {
188     
189           // Add as root node if tree is empty
190           if(*tree == NULL) {
191              *tree = create_range_node(min, max);      
192              return;
193           }
194           
195           
196           // Merge with adjacent ranges
197           if(min == current->max+1) {
198              min = current->min;
199              
200              // Start over
201              current = *tree;
202              continue;
203           }
204           if(max == current->min-1) {
205              max = current->max;
206              
207              // Start over
208              current = *tree;
209              continue;
210           }  
211           
212     
213           if(min > current->max) {
214           
215              if(current->right == NULL) {
216                 current->right=create_range_node(min,max);
217                 return;
218              }
219              
220              old = current;
221              current = current->right;
222              
223           } else if(max < current->min) {
224           
225              if(current->left == NULL) {
226                 current->left=create_range_node(min,max);
227                 return;
228              }
229               
230              old = current;
231              current = current->left;
232           
233           // Handle overlapping range   
234           } else {
235              // Create new min, max
236              min = MIN(min,current->min);
237              max = MAX(max,current->max);         
238         
239              // Delete overlapping node...
240              if(old == NULL)   
241                 *tree = delete_range_node(current, old);  // Replace root node
242              else
243                 delete_range_node(current, old);
244     
245              // ...and start over again to add updated interval
246              current = *tree;
247           }
248        }  
249     }
250     
251     static void delete_range_tree(RTreeElem* root) 
252     {
253        if(root->left)
254           delete_range_tree(root->left);
255     
256        if(root->right)
257           delete_range_tree(root->right);
258      
259        VG_(free)(root);
260     }

276     /* Adds ranges to a linked list. Must first be initialized with a 
277        pointer to a IT_RListElem pointer. Subsequent calls with the 'list' 
278        parameter equal to NULL instead adds elements to the list. */
279     static void add_to_range_list(IT_RListElem** list, ULong min, ULong max)
280     {
281        static IT_RListElem** current = NULL;
282     
283        if(list != NULL) {
284           current = list;
285        } else {
286           *current = VG_(malloc)("it_clo.c: add_to_range_list", 
287                                  sizeof(IT_RListElem));
288           (*current)->min  = min;
289           (*current)->max  = max;
290           (*current)->next = NULL;
291           current = &((*current)->next);
292        }
293     }

295     /* Creates a sorted linked list from a range tree. The add_to_range_list
296        function must be initialized before a call to this function. */
297     static void create_range_list(const RTreeElem* root)
298     {
299        if(root->left)
300           create_range_list(root->left);
301     
302        add_to_range_list(NULL, root->min, root->max);
303     
304        if(root->right)
305           create_range_list(root->right);
306     }

309     static void print_range_list(const IT_RListElem* head)
310     {
311        head->min == head->max ? 
312           VG_(printf)("%llu", head->min) : 
313           VG_(printf)("%llu-%llu", head->min, head->max);
314         
315        if(head->next) {
316           VG_(printf)(",");
317           print_range_list(head->next);
318        }

321     // Finds the overlap between the ranges starting with 'head' and the range
322     // specified with 'range_min' and 'range_max'.
323     static IT_RListElem* find_range_overlap(IT_RListElem* head, 
324                                             ULong range_min, 
325                                             ULong range_max)
326     {
327        IT_RListElem* current = head;
328        IT_RListElem* overlapping = NULL;
329        ULong cur_min, cur_max;
330        
331        add_to_range_list(&overlapping, 0, 0); // Initialize
332         
333        while(True) {
334           if(current == NULL || current->min > range_max)
335              break;
336         
337           if(current->max < range_min) {
338              current = current->next;
339              continue;
340           }
341           
342           if(current->max >= range_min) {
343              if(current->min < range_min)
344                 cur_min = range_min;
345              else
346                 cur_min = current->min;
347           
348              if(current->max < range_max)
349                 cur_max = current->max;
350              else
351                 cur_max = range_max;
352           }
353           
354           add_to_range_list(NULL, cur_min, cur_max);
355           
356           current = current->next;
357        }
358        
359        return overlapping;
360     }

362     // Checks if 'n' is inside any range in the range list
363     static Bool is_in_range(IT_RListElem* head, ULong n)
364     {
365        IT_RListElem* current = head;
366     
367        while(current != NULL) {
368     	   if(current->min <= n && current->max >= n)
369     		   return True;
370     	  current = current->next;
371        }
372     
373        return False;
374     }

376     /* Parses a byte filter string and returns a linked list with sorted ranges, 
377        or NULL on failure */
378     
379     typedef enum { ParseBase10, ParseBase16 } ParseBase;
380     
381     static IT_RListElem* parse_range_filter(const Char* str, ParseBase base)
382     {
383        RTreeElem* range_tree=NULL;
384        Char *cur_range=NULL, *range_end=NULL, *endptr=NULL;
385        Char *tmp = NULL;
386        Long min=0, max=0;
387        IT_RListElem* rlist = NULL;
388        
389        tl_assert(str);
390        
391        // We need to modify the string below, so make a copy
392        tmp = VG_(strdup)("it_clo.c: parse_range_filter", str);
393        
394        cur_range = tmp;      
395        range_end = VG_(strchr)(cur_range, ',');
396        while(True) {
397           
398           if(range_end)
399              range_end[0] = '\0';
400     
401           if(base == ParseBase10)
402         	  min = VG_(strtoll10)(cur_range, &endptr);
403           else
404         	  min = VG_(strtoll16)(cur_range, &endptr);
405     
406           if(min < 0)
407              return NULL;
408     
409           switch(*endptr) {
410              case '-' :
411                 if(*(endptr+1) == '\0')  // Missing 'max' range
412                    return NULL;
413     
414                 if(base == ParseBase10)
415                 	max = VG_(strtoll10)(endptr+1, &endptr);
416                 else
417                 	max = VG_(strtoll16)(endptr+1, &endptr);
418     
419                 if(*endptr) // Trailing non-digits after 'max'
420                    return NULL;
421                       
422                 break;
423                 
424              case 0 :
425                 max = min;
426                 break;
427                 
428              default:
429                 return NULL;
430           }
431         
432           if(min > max)
433              return NULL;
434           
435           insert_range_node(&range_tree, (ULong)min, (ULong)max);
436           
437           if(!range_end)
438              break;
439              
440           cur_range = range_end+1;
441           range_end = VG_(strchr)(cur_range, ',');
442        }
443     
444        VG_(free)(tmp);
445     
446        add_to_range_list(&rlist, 0, 0); // Initialize
447        
448        if(range_tree) {
449           create_range_list(range_tree);
450           delete_range_tree(range_tree);
451        }
452          
453        return rlist;
454     }

461     static Bool check_duplicates(const FileID* id, FileRangeMap* map)
462     {
463        Word i;
464        for(i=0; i < map->current_idx; i++)
465           if FILEID_EQ(*id, map->files[i]) return True;
466        
467        return False;
468     }

470     static Bool get_file_id_from_path(const Char* path, FileID* id)
471     {
472        SysRes res;
473        struct vg_stat info;
474        
475        res = VG_(stat)(path, &info);
476        
477        if(res._isError)
478           return False;
479           
480        id->device = info.dev;
481        id->inode  = info.ino;
482        return True;
483     }

485     static Bool get_file_id_from_fd(UWord fd, FileID* id)
486     {
487        struct vg_stat info;
488        
489        if(fd == 0) { // standard input
490           SET_TO_STDIN(id);
491           return True;
492        }
493        
494        if( VG_(fstat)(fd, &info) != 0 )
495           return False;
496           
497        id->device = info.dev;
498        id->inode  = info.ino;
499        return True;
500     }

503     /* Helper function for adding input filters. Performs appropriate checks and
504        exits with an error message if invalid input is detected. */
505     static void add_input_filter(const Char* filter)
506     {
507        Char *filterCopy, *delimiter;
508        FileID id;
509        
510        if(!filter[0])
511           VG_(fmsg_bad_option)(
512              "--add-filter",
513              "Empty filter string\n");
514      
515        if(input_filter_map.current_idx >= MAX_FILE_FILTERS)
516           VG_(fmsg_bad_option)(
517              "--add-filter",
518              "Too many filters specified.\n");
519        
520        filterCopy = VG_(strdup)("it_clo.c: add_input_filter", filter);
521        
522        delimiter = VG_(strchr)(filterCopy, ':');
523        if(!delimiter)
524           VG_(fmsg_bad_option)(
525              "--add-filter",
526              "Malformed filter. Format is: file_path:byte_ranges\n");
527        
528        delimiter[0] = '\0';
529        
530        /* File path parsing */
531        if(VG_STREQ(filterCopy, STDIN_IDENTIFIER)) // Check for special case of STDIN
532           SET_TO_STDIN(&id);
533        else {
534           if( !get_file_id_from_path(filterCopy, &id) )
535              VG_(fmsg_bad_option)(
536                 "--add-filter",
537                 "Could not stat file: %s\n", filterCopy);
538        }
539        
540        if(check_duplicates(&id, &input_filter_map))
541           VG_(fmsg_bad_option)(
542              "--add-filter",
543              "Error: multiple filters for %s\n", filterCopy);
544     
545        input_filter_map.files[input_filter_map.current_idx] = id;
546           
547        /* Byte range list parsing */
548        if(VG_STREQ(delimiter+1, ALL_FILTER_RANGE)) // Check for special case of ALL
549     	   input_filter_map.ranges[input_filter_map.current_idx] = NULL /* No filter */;
550        else {
551     	   input_filter_map.ranges[input_filter_map.current_idx] =
552     			   parse_range_filter(delimiter+1, ParseBase10);
553     
554     	  // Check that parsing succeeded and that the input file range start from
555     	  // 1 or higher.
556           if(!input_filter_map.ranges[input_filter_map.current_idx] ||
557              input_filter_map.ranges[input_filter_map.current_idx]->min < 1)
558              VG_(fmsg_bad_option)(
559                 "--add-filter",
560                 "Invalid byte range list: %s\n", delimiter+1);   
561        }
562        
563        VG_(free)(filterCopy);
564        input_filter_map.current_idx++;
565     }

567     /* Helper function for adding taint sink filters for executables.
568        Performs appropriate checks and exits with an error message if invalid
569        input is detected. */
570     static void add_sink_filter(const Char* filter)
571     {
572        Char *filterCopy, *delimiter;
573        FileID id;
574     
575        if(!filter[0])
576           VG_(fmsg_bad_option)(
577              "--add-sink-filter",
578              "Empty filter string\n");
579     
580        if(sink_filter_map.current_idx >= MAX_FILE_FILTERS)
581           VG_(fmsg_bad_option)(
582     		 "--add-sink-filter",
583              "Too many filters specified.\n");
584     
585        filterCopy = VG_(strdup)("it_clo.c: add_sink_filter", filter);
586     
587        delimiter = VG_(strchr)(filterCopy, ':');
588        if(!delimiter)
589           VG_(fmsg_bad_option)(
590              "--add-sink-filter",
591              "Malformed filter. Format is: file_path:offset_ranges\n");
592     
593        delimiter[0] = '\0';
594     
595        /* File path parsing */
596     	if( !get_file_id_from_path(filterCopy, &id) )
597     		VG_(fmsg_bad_option)(
598                 "--add-sink-filter",
599     			"Could not stat file: %s\n", filterCopy);
600     
601        if(check_duplicates(&id, &sink_filter_map))
602           VG_(fmsg_bad_option)(
603              "--add-sink-filter",
604              "Error: multiple filters for %s\n", filterCopy);
605     
606        sink_filter_map.files[sink_filter_map.current_idx] = id;
607     
608        /* Byte range list parsing */
609        if(VG_STREQ(delimiter+1, ALL_FILTER_RANGE)) // Check for special case of ALL
610     	   sink_filter_map.ranges[sink_filter_map.current_idx] = NULL /* No filter */;
611        else {
612     	   sink_filter_map.ranges[sink_filter_map.current_idx] =
613     			   parse_range_filter(delimiter+1, ParseBase16);
614           if(!sink_filter_map.ranges[sink_filter_map.current_idx])
615              VG_(fmsg_bad_option)(
616                  "--add-sink-filter",
617                 "Invalid offset range list: %s\n", delimiter+1);
618        }
619     
620        VG_(free)(filterCopy);
621        sink_filter_map.current_idx++;
622     }

693     Bool IT_(process_cmd_line_option)(Char* arg)
694     {
695        Char* tmp;   
696        Word i;
697        
698        if VG_STR_CLO(arg, "--output", IT_(clo_output_filename) ) {}
699        else if VG_BOOL_CLO(arg, "--trace-files",         IT_(clo_trace_files)) {}
700        else if VG_BOOL_CLO(arg, "--trace-stdin",         IT_(clo_trace_stdin)) {}
701        else if VG_BOOL_CLO(arg, "--address-propagation", IT_(clo_address_prop)) {} 
702        else if VG_BOOL_CLO(arg, "--syscall-propagation", IT_(clo_syscall_prop)) {}
703        else if VG_STR_CLO(arg, "--taint-sinks", tmp) {
704           IT_(clo_uncond_br_sink) = False;
705           IT_(clo_cond_br_sink)   = False;
706           IT_(clo_syscall_sink)   = False;
707           IT_(clo_mem_write_sink) = False;
708           
709           for(i=0; i < VG_(strlen)(tmp); i++) {
710              switch(tmp[i]) {
711                 case 'U' : IT_(clo_uncond_br_sink)  = True; break;
712                 case 'C' : IT_(clo_cond_br_sink)    = True; break;
713                 case 'S' : IT_(clo_syscall_sink)    = True; break;
714                 case 'M' : IT_(clo_mem_write_sink)  = True; break;
715                 default  : 
716                    VG_(fmsg_bad_option)(
717                       arg, "Unrecoginzed taint sink selector: %c\n", tmp[i]);
718              }
719           }
720        }
721        else if VG_BOOL_CLO(
722                   arg, "--dynamic-stacktrace", IT_(clo_dynamic_stack_trace)) {} 
723        else if VG_STR_CLO(arg, "--add-filter", tmp) { 
724           tl_assert(tmp);
725           add_input_filter(tmp);      
726        }
727        else if VG_STR_CLO(arg, "--exclude-file", tmp) { 
728           tl_assert(tmp);
729           add_file_exclusion(tmp, &input_file_exclusions);
730        }
731        else if VG_STR_CLO(arg, "--add-sink-filter", tmp) {
732           tl_assert(tmp);
733           add_sink_filter(tmp);
734        }
735        else if VG_STR_CLO(arg, "--exclude-sinks", tmp) {
736           tl_assert(tmp);
737           add_file_exclusion(tmp, &sink_image_exclusions);
738        }
739        else if VG_STR_CLO(arg, "--exclude-propagation", tmp) {
740           tl_assert(tmp);
741           add_prop_image_exclusion(tmp);
742        }
743        else if VG_STR_CLO(arg, "--include-propagation", tmp) {
744              tl_assert(tmp);
745              add_prop_image_inclusion(tmp);
746           }
747        else
748           return False;
749     
750        return True;
751     }

846     static Bool is_in_filter_list(FileRangeMap* map, UWord filedes)
847     {
848        FileID id;
849        Word i;
850     
851        if( !get_file_id_from_fd(filedes, &id) )
852           return False;
853     
854        for(i=0; i < map->current_idx; i++)
855           if FILEID_EQ(map->files[i], id)
856              return True;
857     
858        return False;
859     }
860     
861     inline Bool IT_(is_in_input_filter_list)(UWord filedes)
862     {
863        return is_in_filter_list(&input_filter_map, filedes);
864     }

866     static Word get_filter_handle(UWord filedes, FileRangeMap* map)
867     {
868        FileID id;
869        Word i;
870        
871        if( !get_file_id_from_fd(filedes, &id) )
872           return -1;
873        
874        for(i=0; i < map->current_idx; i++)
875           if ( FILEID_EQ(map->files[i], id) && map->ranges[i] != NULL)
876              return i;
877              
878        return -1;
879     }

881     static Word get_filter_handle_from_path(const Char* path, FileRangeMap* map)
882     {
883        FileID id;
884        Word i;
885     
886        if( !get_file_id_from_path(path, &id) )
887           return -1;
888     
889        for(i=0; i < map->current_idx; i++) {
890     	   if ( FILEID_EQ(map->files[i], id))
891     		   return i;
892        }
893     
894        return -1;
895     }

898     {
899        return get_filter_handle(filedes, &input_filter_map);
900     }

907     static IT_RListElem*
908     get_filtered_ranges(FileRangeMap* map, Word handle, ULong min, ULong max)
909     {
910        IT_RListElem* list;
911     
912        tl_assert(handle >= 0);
913        tl_assert(handle < map->current_idx);
914     
915        if(map->ranges[handle] == NULL) { // No filter, include all bytes
916           list = VG_(malloc)("it_clo.c: IT_(get_filtered_ranges)",
917                              sizeof(IT_RListElem));
918           list->min = min;
919           list->max = max;
920           list->next = NULL;
921        } else
922           list = find_range_overlap(map->ranges[handle], min, max);
923     
924        return list;
925     }

944     static Bool has_match_file_pattern(const Char* path, XArray* patterns)
945     {
946        Int i;
947        Char** array;
948        Word size;
949        
950        if( ! patterns )
951           return False;
952     
953        VG_(getContentsXA_UNSAFE)(patterns, (void**)&array, &size);
954     
955        for(i=0 ; i<size; i++) {
956           if( VG_(string_match)(array[i], path) )
957              return True;
958        }
959        
960        return False;
961     }
