9     // Reg test for bug #93328: we were using too-big sigset types, and thus
10     // trashing memory when we wrote out the 'oldset' param from sigprocmask().
11     
12     int main(void)
13     {
14     #if defined(__NR_sigprocmask)        \
15         && !defined(__powerpc64__)       \
16         && !defined(__s390x__)           \
17         && !defined(_AIX)                \
18         && !defined(__arm__)
19     
20        // arm-linux uses rt_sigprocmask, so no sigset mangling takes place
21     
22        int x[6], *s, *os, i;
23     
24        x[0] = 0x11111111;
25        x[1] = 0x89abcdef;
26        x[2] = 0x22222222;
27        x[3] = 0x33333333;
28        x[4] = 0x0;
29        x[5] = 0x44444444;
30     
31        s  = &x[1];
32        os = &x[4];
33     
34        // Make sure the system is in a known state with no signals
35        // blocked as perl has been known to leave some signals blocked
36        // when starting child processes which can cause failures in
37        // this test unless we reset things here.
38        syscall(__NR_sigprocmask, SIG_SETMASK, os, NULL);
39     
40        fprintf(stderr, "before\n");
41        for (i = 0; i < 6; i++) {
42           fprintf(stderr, "%x ", x[i]);
43        }
44        fprintf(stderr, "\n");
45     
46        syscall(__NR_sigprocmask, SIG_BLOCK, s, os);
47     
48        fprintf(stderr, "after1\n");
49        for (i = 0; i < 6; i++) {
50           fprintf(stderr, "%x ", x[i]);
51        }
52        fprintf(stderr, "\n");
53        
54        syscall(__NR_sigprocmask, SIG_BLOCK, s, os);
55     
56        fprintf(stderr, "after2\n");
57        for (i = 0; i < 6; i++) {
58           fprintf(stderr, "%x ", x[i]);
59        }
60        fprintf(stderr, "\n");
61     
62     #else
63     
64        fprintf(stderr, "__NR_sigprocmask not supported on this platform\n");
65     
66     #endif
67     
68        return(0);
69     }
