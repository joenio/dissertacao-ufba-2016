31     /* 
32        Signal handling.
33     
34        There are 4 distinct classes of signal:
35     
36        1. Synchronous, instruction-generated (SIGILL, FPE, BUS, SEGV and
37        TRAP): these are signals as a result of an instruction fault.  If
38        we get one while running client code, then we just do the
39        appropriate thing.  If it happens while running Valgrind code, then
40        it indicates a Valgrind bug.  Note that we "manually" implement
41        automatic stack growth, such that if a fault happens near the
42        client process stack, it is extended in the same way the kernel
43        would, and the fault is never reported to the client program.
44     
45        2. Asynchronous variants of the above signals: If the kernel tries
46        to deliver a sync signal while it is blocked, it just kills the
47        process.  Therefore, we can't block those signals if we want to be
48        able to report on bugs in Valgrind.  This means that we're also
49        open to receiving those signals from other processes, sent with
50        kill.  We could get away with just dropping them, since they aren't
51        really signals that processes send to each other.
52     
53        3. Synchronous, general signals.  If a thread/process sends itself
54        a signal with kill, its expected to be synchronous: ie, the signal
55        will have been delivered by the time the syscall finishes.
56        
57        4. Asynchronous, general signals.  All other signals, sent by
58        another process with kill.  These are generally blocked, except for
59        two special cases: we poll for them each time we're about to run a
60        thread for a time quanta, and while running blocking syscalls.
61     
62     
63        In addition, we reserve one signal for internal use: SIGVGKILL.
64        SIGVGKILL is used to terminate threads.  When one thread wants
65        another to exit, it will set its exitreason and send it SIGVGKILL
66        if it appears to be blocked in a syscall.
67     
68     
69        We use a kernel thread for each application thread.  When the
70        thread allows itself to be open to signals, it sets the thread
71        signal mask to what the client application set it to.  This means
72        that we get the kernel to do all signal routing: under Valgrind,
73        signals get delivered in the same way as in the non-Valgrind case
74        (the exception being for the sync signal set, since they're almost
75        always unblocked).
76      */
77     
78     /*
79        Some more details...
80     
81        First off, we take note of the client's requests (via sys_sigaction
82        and sys_sigprocmask) to set the signal state (handlers for each
83        signal, which are process-wide, + a mask for each signal, which is
84        per-thread).  This info is duly recorded in the SCSS (static Client
85        signal state) in m_signals.c, and if the client later queries what
86        the state is, we merely fish the relevant info out of SCSS and give
87        it back.
88     
89        However, we set the real signal state in the kernel to something
90        entirely different.  This is recorded in SKSS, the static Kernel
91        signal state.  What's nice (to the extent that anything is nice w.r.t
92        signals) is that there's a pure function to calculate SKSS from SCSS,
93        calculate_SKSS_from_SCSS.  So when the client changes SCSS then we
94        recompute the associated SKSS and apply any changes from the previous
95        SKSS through to the kernel.
96     
97        Now, that said, the general scheme we have now is, that regardless of
98        what the client puts into the SCSS (viz, asks for), what we would
99        like to do is as follows:
100     
101        (1) run code on the virtual CPU with all signals blocked
102     
103        (2) at convenient moments for us (that is, when the VCPU stops, and
104           control is back with the scheduler), ask the kernel "do you have
105           any signals for me?"  and if it does, collect up the info, and
106           deliver them to the client (by building sigframes).
107     
108        And that's almost what we do.  The signal polling is done by
109        VG_(poll_signals), which calls through to VG_(sigtimedwait_zero) to
110        do the dirty work.  (of which more later).
111     
112        By polling signals, rather than catching them, we get to deal with
113        them only at convenient moments, rather than having to recover from
114        taking a signal while generated code is running.
115     
116        Now unfortunately .. the above scheme only works for so-called async
117        signals.  An async signal is one which isn't associated with any
118        particular instruction, eg Control-C (SIGINT).  For those, it doesn't
119        matter if we don't deliver the signal to the client immediately; it
120        only matters that we deliver it eventually.  Hence polling is OK.
121     
122        But the other group -- sync signals -- are all related by the fact
123        that they are various ways for the host CPU to fail to execute an
124        instruction: SIGILL, SIGSEGV, SIGFPU.  And they can't be deferred,
125        because obviously if a host instruction can't execute, well then we
126        have to immediately do Plan B, whatever that is.
127     
128        So the next approximation of what happens is:
129     
130        (1) run code on vcpu with all async signals blocked
131     
132        (2) at convenient moments (when NOT running the vcpu), poll for async
133           signals.
134     
135        (1) and (2) together imply that if the host does deliver a signal to
136           async_signalhandler while the VCPU is running, something's
137           seriously wrong.
138     
139        (3) when running code on vcpu, don't block sync signals.  Instead
140           register sync_signalhandler and catch any such via that.  Of
141           course, that means an ugly recovery path if we do -- the
142           sync_signalhandler has to longjump, exiting out of the generated
143           code, and the assembly-dispatcher thingy that runs it, and gets
144           caught in m_scheduler, which then tells m_signals to deliver the
145           signal.
146     
147        Now naturally (ha ha) even that might be tolerable, but there's
148        something worse: dealing with signals delivered to threads in
149        syscalls.
150     
151        Obviously from the above, SKSS's signal mask (viz, what we really run
152        with) is way different from SCSS's signal mask (viz, what the client
153        thread thought it asked for).  (eg) It may well be that the client
154        did not block control-C, so that it just expects to drop dead if it
155        receives ^C whilst blocked in a syscall, but by default we are
156        running with all async signals blocked, and so that signal could be
157        arbitrarily delayed, or perhaps even lost (not sure).
158     
159        So what we have to do, when doing any syscall which SfMayBlock, is to
160        quickly switch in the SCSS-specified signal mask just before the
161        syscall, and switch it back just afterwards, and hope that we don't
162        get caught up in some wierd race condition.  This is the primary
163        purpose of the ultra-magical pieces of assembly code in
164        coregrind/m_syswrap/syscall-<plat>.S
165     
166        -----------
167     
168        The ways in which V can come to hear of signals that need to be
169        forwarded to the client as are follows:
170     
171         sync signals: can arrive at any time whatsoever.  These are caught
172                       by sync_signalhandler
173     
174         async signals:
175     
176            if    running generated code
177            then  these are blocked, so we don't expect to catch them in
178                  async_signalhandler
179     
180            else
181            if    thread is blocked in a syscall marked SfMayBlock
182            then  signals may be delivered to async_sighandler, since we
183                  temporarily unblocked them for the duration of the syscall,
184                  by using the real (SCSS) mask for this thread
185     
186            else  we're doing misc housekeeping activities (eg, making a translation,
187                  washing our hair, etc).  As in the normal case, these signals are
188                  blocked, but we can  and do poll for them using VG_(poll_signals).
189     
190        Now, re VG_(poll_signals), it polls the kernel by doing
191        VG_(sigtimedwait_zero).  This is trivial on Linux, since it's just a
192        syscall.  But on Darwin and AIX, we have to cobble together the
193        functionality in a tedious, longwinded and probably error-prone way.
194      */

464        static inline SysRes VG_UCONTEXT_SYSCALL_SYSRES( void* ucV,
465                                                         UWord scclass ) {
466           /* this is complicated by the problem that there are 3 different
467              kinds of syscalls, each with its own return convention.
468              NB: scclass is a host word, hence UWord is good for both
469              amd64-darwin and x86-darwin */
470           ucontext_t* uc = (ucontext_t*)ucV;
471           struct __darwin_mcontext32* mc = uc->uc_mcontext;
472           struct __darwin_i386_thread_state* ss = &mc->__ss;
473           /* duplicates logic in m_syswrap.getSyscallStatusFromGuestState */
474           UInt carry = 1 & ss->__eflags;
475           UInt err = 0;
476           UInt wLO = 0;
477           UInt wHI = 0;
478           switch (scclass) {
479              case VG_DARWIN_SYSCALL_CLASS_UNIX:
480                 err = carry;
481                 wLO = ss->__eax;
482                 wHI = ss->__edx;
483                 break;
484              case VG_DARWIN_SYSCALL_CLASS_MACH:
485                 wLO = ss->__eax;
486                 break;
487              case VG_DARWIN_SYSCALL_CLASS_MDEP:
488                 wLO = ss->__eax;
489                 break;
490              default: 
491                 vg_assert(0);
492                 break;
493           }
494           return VG_(mk_SysRes_x86_darwin)( scclass, err ? True : False, 
495                                             wHI, wLO );
496        }

625     /* Let's do:

627          so that at least everything is always consistent

645              for all signals to be blocked in our sighandlers. */
646           /* Also there is no skss_restorer. */

669     static 
670     void pp_SKSS ( void )
671     {
672        Int sig;
673        VG_(printf)("\n\nSKSS:\n");
674        for (sig = 1; sig <= _VKI_NSIG; sig++) {
675           VG_(printf)("sig %d:  handler %p,  flags 0x%x\n", sig,
676                       skss.skss_per_sig[sig].skss_handler,
677                       skss.skss_per_sig[sig].skss_flags );
678     
679        }
680     }

682     /* This is the core, clever bit.  Computation is as follows:
683     
684        For each signal
685           handler = if client has a handler, then our handler
686                     else if client is DFL, then our handler as well
687                     else (client must be IGN)
688     			then hander is IGN
689     */
690     static
691     void calculate_SKSS_from_SCSS ( SKSS* dst )
692     {
693        Int   sig;
694        UInt  scss_flags;
695        UInt  skss_flags;
696     
697        for (sig = 1; sig <= _VKI_NSIG; sig++) {
698           void *skss_handler;
699           void *scss_handler;
700           
701           scss_handler = scss.scss_per_sig[sig].scss_handler;
702           scss_flags   = scss.scss_per_sig[sig].scss_flags;
703     
704           switch(sig) {
705           case VKI_SIGSEGV:
706           case VKI_SIGBUS:
707           case VKI_SIGFPE:
708           case VKI_SIGILL:
709           case VKI_SIGTRAP:
710     	 /* For these, we always want to catch them and report, even
711     	    if the client code doesn't. */
712     	 skss_handler = sync_signalhandler;
713     	 break;
714     
715           case VKI_SIGCONT:
716     	 /* Let the kernel handle SIGCONT unless the client is actually
717     	    catching it. */
718           case VKI_SIGCHLD:
719           case VKI_SIGWINCH:
720           case VKI_SIGURG:
721              /* For signals which are have a default action of Ignore,
722                 only set a handler if the client has set a signal handler.
723                 Otherwise the kernel will interrupt a syscall which
724                 wouldn't have otherwise been interrupted. */
725     	 if (scss.scss_per_sig[sig].scss_handler == VKI_SIG_DFL)
726     	    skss_handler = VKI_SIG_DFL;
727     	 else if (scss.scss_per_sig[sig].scss_handler == VKI_SIG_IGN)
728     	    skss_handler = VKI_SIG_IGN;
729     	 else
730     	    skss_handler = async_signalhandler;
731     	 break;
732     
733           default:
734              // VKI_SIGVG* are runtime variables, so we can't make them            
735              // cases in the switch, so we handle them in the 'default' case.
736     	 if (sig == VG_SIGVGKILL)
737     	    skss_handler = sigvgkill_handler;
738     	 else {
739     	    if (scss_handler == VKI_SIG_IGN)
740     	       skss_handler = VKI_SIG_IGN;
741     	    else 
742     	       skss_handler = async_signalhandler;
743     	 }
744     	 break;
745           }
746     
747           /* Flags */
748     
749           skss_flags = 0;
750     
751           /* SA_NOCLDSTOP, SA_NOCLDWAIT: pass to kernel */
752           skss_flags |= scss_flags & (VKI_SA_NOCLDSTOP | VKI_SA_NOCLDWAIT);
753     
754           /* SA_ONESHOT: ignore client setting */
755           
756           /* SA_RESTART: ignore client setting and always set it for us.
757     	 Though we never rely on the kernel to restart a
758     	 syscall, we observe whether it wanted to restart the syscall
759     	 or not, which is needed by 
760              VG_(fixup_guest_state_after_syscall_interrupted) */
761           skss_flags |= VKI_SA_RESTART;
762     
763           /* SA_NOMASK: ignore it */
764     
765           /* SA_ONSTACK: client setting is irrelevant here */
766           /* We don't set a signal stack, so ignore */
767     
768           /* always ask for SA_SIGINFO */
769           skss_flags |= VKI_SA_SIGINFO;
770     
771           /* use our own restorer */
772           skss_flags |= VKI_SA_RESTORER;
773     
774           /* Create SKSS entry for this signal. */
775           if (sig != VKI_SIGKILL && sig != VKI_SIGSTOP)
776              dst->skss_per_sig[sig].skss_handler = skss_handler;
777           else
778              dst->skss_per_sig[sig].skss_handler = VKI_SIG_DFL;
779     
780           dst->skss_per_sig[sig].skss_flags   = skss_flags;
781        }
782     
783        /* Sanity checks. */
784        vg_assert(dst->skss_per_sig[VKI_SIGKILL].skss_handler == VKI_SIG_DFL);
785        vg_assert(dst->skss_per_sig[VKI_SIGSTOP].skss_handler == VKI_SIG_DFL);
786     
787        if (0)
788           pp_SKSS();
789     }

889     static void handle_SCSS_change ( Bool force_update )
890     {
891        Int  res, sig;
892        SKSS skss_old;
893        vki_sigaction_toK_t   ksa;
894        vki_sigaction_fromK_t ksa_old;
895     
896        /* Remember old SKSS and calculate new one. */
897        skss_old = skss;
898        calculate_SKSS_from_SCSS ( &skss );
899     
900        /* Compare the new SKSS entries vs the old ones, and update kernel
901           where they differ. */
902        for (sig = 1; sig <= VG_(max_signal); sig++) {
903     
904           /* Trying to do anything with SIGKILL is pointless; just ignore
905              it. */
906           if (sig == VKI_SIGKILL || sig == VKI_SIGSTOP)
907              continue;
908     
909           if (!force_update) {
910              if ((skss_old.skss_per_sig[sig].skss_handler
911                   == skss.skss_per_sig[sig].skss_handler)
912                  && (skss_old.skss_per_sig[sig].skss_flags
913                      == skss.skss_per_sig[sig].skss_flags))
914                 /* no difference */
915                 continue;
916           }
917     
918           ksa.ksa_handler = skss.skss_per_sig[sig].skss_handler;
919           ksa.sa_flags    = skss.skss_per_sig[sig].skss_flags;
920     #     if !defined(VGP_ppc32_linux) && \
921              !defined(VGP_ppc32_aix5) && !defined(VGP_ppc64_aix5) && \
922              !defined(VGP_x86_darwin) && !defined(VGP_amd64_darwin)
923           ksa.sa_restorer = my_sigreturn;
924     #     endif
925           /* Re above ifdef (also the assertion below), PaulM says:
926              The sa_restorer field is not used at all on ppc.  Glibc
927              converts the sigaction you give it into a kernel sigaction,
928              but it doesn't put anything in the sa_restorer field.
929           */
930     
931           /* block all signals in handler */
932           VG_(sigfillset)( &ksa.sa_mask );
933           VG_(sigdelset)( &ksa.sa_mask, VKI_SIGKILL );
934           VG_(sigdelset)( &ksa.sa_mask, VKI_SIGSTOP );
935     
936           if (VG_(clo_trace_signals) && VG_(clo_verbosity) > 2)
937              VG_(dmsg)("setting ksig %d to: hdlr %p, flags 0x%lx, "
938                        "mask(msb..lsb) 0x%llx 0x%llx\n",
939                        sig, ksa.ksa_handler,
940                        (UWord)ksa.sa_flags,
941                        _VKI_NSIG_WORDS > 1 ? (ULong)ksa.sa_mask.sig[1] : 0,
942                        (ULong)ksa.sa_mask.sig[0]);
943     
944           res = VG_(sigaction)( sig, &ksa, &ksa_old );
945           vg_assert(res == 0);
946     
947           /* Since we got the old sigaction more or less for free, might
948              as well extract the maximum sanity-check value from it. */
949           if (!force_update) {
950              vg_assert(ksa_old.ksa_handler 
951                        == skss_old.skss_per_sig[sig].skss_handler);
952              vg_assert(ksa_old.sa_flags 
953                        == skss_old.skss_per_sig[sig].skss_flags);
954     #        if !defined(VGP_ppc32_linux) && \
955                 !defined(VGP_ppc32_aix5) && !defined(VGP_ppc64_aix5) && \
956                 !defined(VGP_x86_darwin) && !defined(VGP_amd64_darwin)
957              vg_assert(ksa_old.sa_restorer 
958                        == my_sigreturn);
959     #        endif
960              VG_(sigaddset)( &ksa_old.sa_mask, VKI_SIGKILL );
961              VG_(sigaddset)( &ksa_old.sa_mask, VKI_SIGSTOP );
962              vg_assert(VG_(isfullsigset)( &ksa_old.sa_mask ));
963           }
964        }
965     }

993     SysRes VG_(do_sys_sigaltstack) ( ThreadId tid, vki_stack_t* ss, vki_stack_t* oss )
994     {
995        Addr m_SP;
996     
997        vg_assert(VG_(is_valid_tid)(tid));
998        m_SP  = VG_(get_SP)(tid);
999     
1000        if (VG_(clo_trace_signals))
1001           VG_(dmsg)("sys_sigaltstack: tid %d, "
1002                     "ss %p{%p,sz=%llu,flags=0x%llx}, oss %p (current SP %p)\n",
1003                     tid, (void*)ss, 
1004                     ss ? ss->ss_sp : 0,
1005                     (ULong)(ss ? ss->ss_size : 0),
1006                     (ULong)(ss ? ss->ss_flags : 0),
1007                     (void*)oss, (void*)m_SP);
1008     
1009        if (oss != NULL) {
1010           oss->ss_sp    = VG_(threads)[tid].altstack.ss_sp;
1011           oss->ss_size  = VG_(threads)[tid].altstack.ss_size;
1012           oss->ss_flags = VG_(threads)[tid].altstack.ss_flags
1013                           | sas_ss_flags(tid, m_SP);
1014        }
1015     
1016        if (ss != NULL) {
1017           if (on_sig_stack(tid, VG_(get_SP)(tid))) {
1018              return VG_(mk_SysRes_Error)( VKI_EPERM );
1019           }
1020           if (ss->ss_flags != VKI_SS_DISABLE 
1021               && ss->ss_flags != VKI_SS_ONSTACK 
1022               && ss->ss_flags != 0) {
1023              return VG_(mk_SysRes_Error)( VKI_EINVAL );
1024           }
1025           if (ss->ss_flags == VKI_SS_DISABLE) {
1026              VG_(threads)[tid].altstack.ss_flags = VKI_SS_DISABLE;
1027           } else {
1028              if (ss->ss_size < VKI_MINSIGSTKSZ) {
1029                 return VG_(mk_SysRes_Error)( VKI_ENOMEM );
1030              }
1031     
1032     	 VG_(threads)[tid].altstack.ss_sp    = ss->ss_sp;
1033     	 VG_(threads)[tid].altstack.ss_size  = ss->ss_size;
1034     	 VG_(threads)[tid].altstack.ss_flags = 0;
1035           }
1036        }
1037        return VG_(mk_SysRes_Success)( 0 );
1038     }

1041     SysRes VG_(do_sys_sigaction) ( Int signo, 
1042                                    const vki_sigaction_toK_t* new_act, 
1043                                    vki_sigaction_fromK_t* old_act )
1044     {
1045        if (VG_(clo_trace_signals))
1046           VG_(dmsg)("sys_sigaction: sigNo %d, "
1047                     "new %#lx, old %#lx, new flags 0x%llx\n",
1048                     signo, (UWord)new_act, (UWord)old_act,
1049                     (ULong)(new_act ? new_act->sa_flags : 0));
1050     
1051        /* Rule out various error conditions.  The aim is to ensure that if
1052           when the call is passed to the kernel it will definitely
1053           succeed. */
1054     
1055        /* Reject out-of-range signal numbers. */
1056        if (signo < 1 || signo > VG_(max_signal)) goto bad_signo;
1057     
1058        /* don't let them use our signals */
1059        if ( (signo > VG_SIGVGRTUSERMAX)
1060     	&& new_act
1061     	&& !(new_act->ksa_handler == VKI_SIG_DFL 
1062                  || new_act->ksa_handler == VKI_SIG_IGN) )
1063           goto bad_signo_reserved;
1064     
1065        /* Reject attempts to set a handler (or set ignore) for SIGKILL. */
1066        if ( (signo == VKI_SIGKILL || signo == VKI_SIGSTOP)
1067            && new_act
1068            && new_act->ksa_handler != VKI_SIG_DFL)
1069           goto bad_sigkill_or_sigstop;
1070     
1071        /* If the client supplied non-NULL old_act, copy the relevant SCSS
1072           entry into it. */
1073        if (old_act) {
1074           old_act->ksa_handler = scss.scss_per_sig[signo].scss_handler;
1075           old_act->sa_flags    = scss.scss_per_sig[signo].scss_flags;
1076           old_act->sa_mask     = scss.scss_per_sig[signo].scss_mask;
1077     #     if !defined(VGP_ppc32_aix5) && !defined(VGP_ppc64_aix5) && \
1078              !defined(VGP_x86_darwin) && !defined(VGP_amd64_darwin)
1079           old_act->sa_restorer = scss.scss_per_sig[signo].scss_restorer;
1080     #     endif
1081        }
1082     
1083        /* And now copy new SCSS entry from new_act. */
1084        if (new_act) {
1085           scss.scss_per_sig[signo].scss_handler  = new_act->ksa_handler;
1086           scss.scss_per_sig[signo].scss_flags    = new_act->sa_flags;
1087           scss.scss_per_sig[signo].scss_mask     = new_act->sa_mask;
1088     
1089           scss.scss_per_sig[signo].scss_restorer = NULL;
1090     #     if !defined(VGP_ppc32_aix5) && !defined(VGP_ppc64_aix5) && \
1091              !defined(VGP_x86_darwin) && !defined(VGP_amd64_darwin)
1092           scss.scss_per_sig[signo].scss_restorer = new_act->sa_restorer;
1093     #     endif
1094     
1095           scss.scss_per_sig[signo].scss_sa_tramp = NULL;
1096     #     if defined(VGP_x86_darwin) || defined(VGP_amd64_darwin)
1097           scss.scss_per_sig[signo].scss_sa_tramp = new_act->sa_tramp;
1098     #     endif
1099     
1100           VG_(sigdelset)(&scss.scss_per_sig[signo].scss_mask, VKI_SIGKILL);
1101           VG_(sigdelset)(&scss.scss_per_sig[signo].scss_mask, VKI_SIGSTOP);
1102        }
1103     
1104        /* All happy bunnies ... */
1105        if (new_act) {
1106           handle_SCSS_change( False /* lazy update */ );
1107        }
1108        return VG_(mk_SysRes_Success)( 0 );
1109     
1110       bad_signo:
1111        if (VG_(showing_core_errors)() && !VG_(clo_xml)) {
1112           VG_(umsg)("Warning: bad signal number %d in sigaction()\n", signo);
1113        }
1114        return VG_(mk_SysRes_Error)( VKI_EINVAL );
1115     
1116       bad_signo_reserved:
1117        if (VG_(showing_core_errors)() && !VG_(clo_xml)) {
1118           VG_(umsg)("Warning: ignored attempt to set %s handler in sigaction();\n",
1119                     signame(signo));
1120           VG_(umsg)("         the %s signal is used internally by Valgrind\n", 
1121                     signame(signo));
1122        }
1123        return VG_(mk_SysRes_Error)( VKI_EINVAL );
1124     
1125       bad_sigkill_or_sigstop:
1126        if (VG_(showing_core_errors)() && !VG_(clo_xml)) {
1127           VG_(umsg)("Warning: ignored attempt to set %s handler in sigaction();\n",
1128                     signame(signo));
1129           VG_(umsg)("         the %s signal is uncatchable\n", 
1130                     signame(signo));
1131        }
1132        return VG_(mk_SysRes_Error)( VKI_EINVAL );
1133     }

1182        This updates the thread's signal mask.  There's no such thing as a
1183        process-wide signal mask.
1184     
1185        Note that the thread signal masks are an implicit part of SCSS,
1186        which is why this routine is allowed to mess with them.  
1187     */
1188     static
1189     void do_setmask ( ThreadId tid,
1190                       Int how,
1191                       vki_sigset_t* newset,
1192     		  vki_sigset_t* oldset )
1193     {
1194        if (VG_(clo_trace_signals))
1195           VG_(dmsg)("do_setmask: tid = %d how = %d (%s), newset = %p (%s)\n", 
1196                     tid, how,
1197                     how==VKI_SIG_BLOCK ? "SIG_BLOCK" : (
1198                        how==VKI_SIG_UNBLOCK ? "SIG_UNBLOCK" : (
1199                           how==VKI_SIG_SETMASK ? "SIG_SETMASK" : "???")),
1200                     newset, newset ? format_sigset(newset) : "NULL" );
1201     
1202        /* Just do this thread. */
1203        vg_assert(VG_(is_valid_tid)(tid));
1204        if (oldset) {
1205           *oldset = VG_(threads)[tid].sig_mask;
1206           if (VG_(clo_trace_signals))
1207              VG_(dmsg)("\toldset=%p %s\n", oldset, format_sigset(oldset));
1208        }
1209        if (newset) {
1210           do_sigprocmask_bitops (how, &VG_(threads)[tid].sig_mask, newset );
1211           VG_(sigdelset)(&VG_(threads)[tid].sig_mask, VKI_SIGKILL);
1212           VG_(sigdelset)(&VG_(threads)[tid].sig_mask, VKI_SIGSTOP);
1213           VG_(threads)[tid].tmp_sig_mask = VG_(threads)[tid].sig_mask;
1214        }
1215     }

1280     /* Set up a stack frame (VgSigContext) for the client's signal
1281        handler. */
1282     static
1283     void push_signal_frame ( ThreadId tid, const vki_siginfo_t *siginfo,
1284                                            const struct vki_ucontext *uc )
1285     {
1286        Addr         esp_top_of_frame;
1287        ThreadState* tst;
1288        Int		sigNo = siginfo->si_signo;
1289     
1290        vg_assert(sigNo >= 1 && sigNo <= VG_(max_signal));
1291        vg_assert(VG_(is_valid_tid)(tid));
1292        tst = & VG_(threads)[tid];
1293     
1294        if (VG_(clo_trace_signals)) {
1295           VG_(dmsg)("push_signal_frame (thread %d): signal %d\n", tid, sigNo);
1296           VG_(get_and_pp_StackTrace)(tid, 10);
1297        }
1298     
1299        if (/* this signal asked to run on an alt stack */
1300            (scss.scss_per_sig[sigNo].scss_flags & VKI_SA_ONSTACK )
1301            && /* there is a defined and enabled alt stack, which we're not
1302                  already using.  Logic from get_sigframe in
1303                  arch/i386/kernel/signal.c. */
1304               sas_ss_flags(tid, VG_(get_SP)(tid)) == 0
1305           ) {
1306           esp_top_of_frame 
1307              = (Addr)(tst->altstack.ss_sp) + tst->altstack.ss_size;
1308           if (VG_(clo_trace_signals))
1309              VG_(dmsg)("delivering signal %d (%s) to thread %d: "
1310                        "on ALT STACK (%p-%p; %ld bytes)\n",
1311                        sigNo, signame(sigNo), tid, tst->altstack.ss_sp,
1312                        (UChar *)tst->altstack.ss_sp + tst->altstack.ss_size,
1313                        (Word)tst->altstack.ss_size );
1314     
1315           /* Signal delivery to tools */
1316           VG_TRACK( pre_deliver_signal, tid, sigNo, /*alt_stack*/True );
1317           
1318        } else {
1319           esp_top_of_frame = VG_(get_SP)(tid) - VG_STACK_REDZONE_SZB;
1320     
1321           /* Signal delivery to tools */
1322           VG_TRACK( pre_deliver_signal, tid, sigNo, /*alt_stack*/False );
1323        }
1324     
1325        vg_assert(scss.scss_per_sig[sigNo].scss_handler != VKI_SIG_IGN);
1326        vg_assert(scss.scss_per_sig[sigNo].scss_handler != VKI_SIG_DFL);
1327     
1328        /* This may fail if the client stack is busted; if that happens,
1329           the whole process will exit rather than simply calling the
1330           signal handler. */
1331        VG_(sigframe_create) (tid, esp_top_of_frame, siginfo, uc,
1332                              scss.scss_per_sig[sigNo].scss_handler,
1333                              scss.scss_per_sig[sigNo].scss_flags,
1334                              &tst->sig_mask,
1335                              scss.scss_per_sig[sigNo].scss_restorer);
1336     }

1427     // The si_code describes where the signal came from.  Some come from the
1428     // kernel, eg.: seg faults, illegal opcodes.  Some come from the user, eg.:
1429     // from kill() (SI_USER), or timer_settime() (SI_TIMER), or an async I/O
1430     // request (SI_ASYNCIO).  There's lots of implementation-defined leeway in
1431     // POSIX, but the user vs. kernal distinction is what we want here.  We also
1432     // pass in some other details that can help when si_code is unreliable.
1433     static Bool is_signal_from_kernel(ThreadId tid, int signum, int si_code)
1434     {
1435     #if defined(VGO_linux) || defined(VGO_aix5)
1436        // On Linux, SI_USER is zero, negative values are from the user, positive
1437        // values are from the kernel.  There are SI_FROMUSER and SI_FROMKERNEL
1438        // macros but we don't use them here because other platforms don't have
1439        // them.
1440        return ( si_code > VKI_SI_USER ? True : False );
1441     #elif defined(VGO_darwin)
1442        // On Darwin 9.6.0, the si_code is completely unreliable.  It should be the
1443        // case that 0 means "user", and >0 means "kernel".  But:
1444        // - For SIGSEGV, it seems quite reliable.
1445        // - For SIGBUS, it's always 2.
1446        // - For SIGFPE, it's often 0, even for kernel ones (eg.
1447        //   div-by-integer-zero always gives zero).
1448        // - For SIGILL, it's unclear.
1449        // - For SIGTRAP, it's always 1.
1450        // You can see the "NOTIMP" (not implemented) status of a number of the
1451        // sub-cases in sys/signal.h.  Hopefully future versions of Darwin will
1452        // get this right.
1453     
1454        // If we're blocked waiting on a syscall, it must be a user signal, because
1455        // the kernel won't generate sync signals within syscalls.
1456        if (VG_(threads)[tid].status == VgTs_WaitSys) {
1457           return False;
1458     
1459        // If it's a SIGSEGV, use the proper condition, since it's fairly reliable.
1460        } else if (SIGSEGV == signum) {
1461           return ( si_code > 0 ? True : False );
1462     
1463        // If it's anything else, assume it's kernel-generated.  Reason being that
1464        // kernel-generated sync signals are more common, and it's probable that
1465        // misdiagnosing a user signal as a kernel signal is better than the
1466        // opposite.
1467        } else {
1468           return True;
1469        }
1470     #else
1471     #  error Unknown OS
1472     #endif
1473     }
1474     
1475     // This is an arbitrary si_code that we only use internally.  It corresponds
1476     // to the value SI_KERNEL on Linux, but that's not really of any significance
1477     // as far as I can determine.
1478     #define VKI_SEGV_MADE_UP_GPF    0x80
1479     
1480     /* 
1481        Perform the default action of a signal.  If the signal is fatal, it
1482        marks all threads as needing to exit, but it doesn't actually kill
1483        the process or thread.
1484     
1485        If we're not being quiet, then print out some more detail about
1486        fatal signals (esp. core dumping signals).
1487      */
1488     static void default_action(const vki_siginfo_t *info, ThreadId tid)
1489     {
1490        Int  sigNo     = info->si_signo;
1491        Bool terminate = False;	/* kills process         */
1492        Bool core      = False;	/* kills process w/ core */
1493        struct vki_rlimit corelim;
1494        Bool could_core;
1495     
1496        vg_assert(VG_(is_running_thread)(tid));
1497        
1498        switch(sigNo) {
1499           case VKI_SIGQUIT:	/* core */
1500           case VKI_SIGILL:	/* core */
1501           case VKI_SIGABRT:	/* core */
1502           case VKI_SIGFPE:	/* core */
1503           case VKI_SIGSEGV:	/* core */
1504           case VKI_SIGBUS:	/* core */
1505           case VKI_SIGTRAP:	/* core */
1506           case VKI_SIGXCPU:	/* core */
1507           case VKI_SIGXFSZ:	/* core */
1508              terminate = True;
1509              core = True;
1510              break;
1511     
1512           case VKI_SIGHUP:	/* term */
1513           case VKI_SIGINT:	/* term */
1514           case VKI_SIGKILL:	/* term - we won't see this */
1515           case VKI_SIGPIPE:	/* term */
1516           case VKI_SIGALRM:	/* term */
1517           case VKI_SIGTERM:	/* term */
1518           case VKI_SIGUSR1:	/* term */
1519           case VKI_SIGUSR2:	/* term */
1520           case VKI_SIGIO:	/* term */
1521     #     if defined(VKI_SIGPWR)
1522           case VKI_SIGPWR:	/* term */
1523     #     endif
1524           case VKI_SIGSYS:	/* term */
1525           case VKI_SIGPROF:	/* term */
1526           case VKI_SIGVTALRM:	/* term */
1527     #     if defined(VKI_SIGRTMIN) && defined(VKI_SIGRTMAX)
1528           case VKI_SIGRTMIN ... VKI_SIGRTMAX: /* term */
1529     #     endif
1530              terminate = True;
1531              break;
1532        }
1533     
1534        vg_assert(!core || (core && terminate));
1535     
1536        if (VG_(clo_trace_signals))
1537           VG_(dmsg)("delivering %d (code %d) to default handler; action: %s%s\n",
1538                     sigNo, info->si_code, terminate ? "terminate" : "ignore",
1539                     core ? "+core" : "");
1540     
1541        if (!terminate)
1542           return;			/* nothing to do */
1543     
1544        could_core = core;
1545     
1546        if (core) {
1547           /* If they set the core-size limit to zero, don't generate a
1548     	 core file */
1549     	 
1550           VG_(getrlimit)(VKI_RLIMIT_CORE, &corelim);
1551     
1552           if (corelim.rlim_cur == 0)
1553     	 core = False;
1554        }
1555     
1556        if ( (VG_(clo_verbosity) > 1 ||
1557              (could_core && is_signal_from_kernel(tid, sigNo, info->si_code))
1558             ) &&
1559             !VG_(clo_xml) ) {
1560           VG_(umsg)(
1561              "\n"
1562              "Process terminating with default action of signal %d (%s)%s\n",
1563              sigNo, signame(sigNo), core ? ": dumping core" : "");
1564     
1565           /* Be helpful - decode some more details about this fault */
1566           if (is_signal_from_kernel(tid, sigNo, info->si_code)) {
1567     	 const Char *event = NULL;
1568     	 Bool haveaddr = True;
1569     
1570     	 switch(sigNo) {
1571     	 case VKI_SIGSEGV:
1572     	    switch(info->si_code) {
1573     	    case VKI_SEGV_MAPERR: event = "Access not within mapped region";
1574                                       break;
1575     	    case VKI_SEGV_ACCERR: event = "Bad permissions for mapped region";
1576                                       break;
1577     	    case VKI_SEGV_MADE_UP_GPF:
1578     	       /* General Protection Fault: The CPU/kernel
1579     		  isn't telling us anything useful, but this
1580     		  is commonly the result of exceeding a
1581     		  segment limit. */
1582     	       event = "General Protection Fault"; 
1583     	       haveaddr = False;
1584     	       break;
1585     	    }
1586     #if 0
1587                 {
1588                   HChar buf[110];
1589                   VG_(am_show_nsegments)(0,"post segfault");
1590                   VG_(sprintf)(buf, "/bin/cat /proc/%d/maps", VG_(getpid)());
1591                   VG_(system)(buf);
1592                 }
1593     #endif
1594     	    break;
1595     
1596     	 case VKI_SIGILL:
1597     	    switch(info->si_code) {
1598     	    case VKI_ILL_ILLOPC: event = "Illegal opcode"; break;
1599     	    case VKI_ILL_ILLOPN: event = "Illegal operand"; break;

1690     /* 
1691        This does the business of delivering a signal to a thread.  It may
1692        be called from either a real signal handler, or from normal code to
1693        cause the thread to enter the signal handler.
1694     
1695        This updates the thread state, but it does not set it to be
1696        Runnable.
1697     */
1698     static void deliver_signal ( ThreadId tid, const vki_siginfo_t *info,
1699                                                const struct vki_ucontext *uc )
1700     {
1701        Int			sigNo = info->si_signo;
1702        SCSS_Per_Signal	*handler = &scss.scss_per_sig[sigNo];
1703        void			*handler_fn;
1704        ThreadState		*tst = VG_(get_ThreadState)(tid);
1705     
1706        if (VG_(clo_trace_signals))
1707           VG_(dmsg)("delivering signal %d (%s):%d to thread %d\n", 
1708                     sigNo, signame(sigNo), info->si_code, tid );
1709     
1710        if (sigNo == VG_SIGVGKILL) {
1711           /* If this is a SIGVGKILL, we're expecting it to interrupt any
1712     	 blocked syscall.  It doesn't matter whether the VCPU state is
1713     	 set to restart or not, because we don't expect it will
1714     	 execute any more client instructions. */
1715           vg_assert(VG_(is_exiting)(tid));
1716           return;
1717        }
1718     
1719        /* If the client specifies SIG_IGN, treat it as SIG_DFL.
1720     
1721           If deliver_signal() is being called on a thread, we want
1722           the signal to get through no matter what; if they're ignoring
1723           it, then we do this override (this is so we can send it SIGSEGV,
1724           etc). */
1725        handler_fn = handler->scss_handler;
1726        if (handler_fn == VKI_SIG_IGN) 
1727           handler_fn = VKI_SIG_DFL;
1728     
1729        vg_assert(handler_fn != VKI_SIG_IGN);
1730     
1731        if (handler_fn == VKI_SIG_DFL) {
1732           default_action(info, tid);
1733        } else {
1734           /* Create a signal delivery frame, and set the client's %ESP and
1735     	 %EIP so that when execution continues, we will enter the
1736     	 signal handler with the frame on top of the client's stack,
1737     	 as it expects.
1738     
1739     	 Signal delivery can fail if the client stack is too small or
1740     	 missing, and we can't push the frame.  If that happens,
1741     	 push_signal_frame will cause the whole process to exit when
1742     	 we next hit the scheduler.
1743           */
1744           vg_assert(VG_(is_valid_tid)(tid));
1745     
1746           push_signal_frame ( tid, info, uc );
1747     
1748           if (handler->scss_flags & VKI_SA_ONESHOT) {
1749     	 /* Do the ONESHOT thing. */
1750     	 handler->scss_handler = VKI_SIG_DFL;
1751     
1752     	 handle_SCSS_change( False /* lazy update */ );
1753           }
1754     
1755           /* At this point:
1756     	 tst->sig_mask is the current signal mask
1757     	 tst->tmp_sig_mask is the same as sig_mask, unless we're in sigsuspend
1758     	 handler->scss_mask is the mask set by the handler
1759     
1760     	 Handler gets a mask of tmp_sig_mask|handler_mask|signo
1761            */
1762           tst->sig_mask = tst->tmp_sig_mask;
1763           if (!(handler->scss_flags & VKI_SA_NOMASK)) {
1764     	 VG_(sigaddset_from_set)(&tst->sig_mask, &handler->scss_mask);
1765     	 VG_(sigaddset)(&tst->sig_mask, sigNo);
1766     	 tst->tmp_sig_mask = tst->sig_mask;
1767           }
1768        }
1769     
1770        /* Thread state is ready to go - just add Runnable */
1771     }

1773     static void resume_scheduler(ThreadId tid)
1774     {
1775        ThreadState *tst = VG_(get_ThreadState)(tid);
1776     
1777        vg_assert(tst->os_state.lwpid == VG_(gettid)());
1778     
1779        if (tst->sched_jmpbuf_valid) {
1780           /* Can't continue; must longjmp back to the scheduler and thus
1781              enter the sighandler immediately. */
1782           __builtin_longjmp(tst->sched_jmpbuf, True);
1783        }
1784     }

1896     /* Make a signal pending for a thread, for later delivery.
1897        VG_(poll_signals) will arrange for it to be delivered at the right
1898        time. 
1899     
1900        tid==0 means add it to the process-wide queue, and not sent it to a
1901        specific thread.
1902     */
1903     static 
1904     void queue_signal(ThreadId tid, const vki_siginfo_t *si)
1905     {
1906        ThreadState *tst;
1907        SigQueue *sq;
1908        vki_sigset_t savedmask;
1909     
1910        tst = VG_(get_ThreadState)(tid);
1911     
1912        /* Protect the signal queue against async deliveries */
1913        block_all_host_signals(&savedmask);
1914     
1915        if (tst->sig_queue == NULL) {
1916           tst->sig_queue = VG_(arena_malloc)(VG_AR_CORE, "signals.qs.1",
1917                                              sizeof(*tst->sig_queue));
1918           VG_(memset)(tst->sig_queue, 0, sizeof(*tst->sig_queue));
1919        }
1920        sq = tst->sig_queue;
1921     
1922        if (VG_(clo_trace_signals))
1923           VG_(dmsg)("Queueing signal %d (idx %d) to thread %d\n",
1924                     si->si_signo, sq->next, tid);
1925     
1926        /* Add signal to the queue.  If the queue gets overrun, then old
1927           queued signals may get lost. 
1928     
1929           XXX We should also keep a sigset of pending signals, so that at
1930           least a non-siginfo signal gets deliviered.
1931        */
1932        if (sq->sigs[sq->next].si_signo != 0)
1933           VG_(umsg)("Signal %d being dropped from thread %d's queue\n",
1934                     sq->sigs[sq->next].si_signo, tid);
1935     
1936        sq->sigs[sq->next] = *si;
1937        sq->next = (sq->next+1) % N_QUEUED_SIGNALS;
1938     
1939        restore_all_host_signals(&savedmask);
1940     }

1942     /*
1943        Returns the next queued signal for thread tid which is in "set".
1944        tid==0 means process-wide signal.  Set si_signo to 0 when the
1945        signal has been delivered.
1946     
1947        Must be called with all signals blocked, to protect against async
1948        deliveries.
1949     */
1950     static vki_siginfo_t *next_queued(ThreadId tid, const vki_sigset_t *set)
1951     {
1952        ThreadState *tst = VG_(get_ThreadState)(tid);
1953        SigQueue *sq;
1954        Int idx;
1955        vki_siginfo_t *ret = NULL;
1956     
1957        sq = tst->sig_queue;
1958        if (sq == NULL)
1959           goto out;
1960        
1961        idx = sq->next;
1962        do {
1963           if (0)
1964     	 VG_(printf)("idx=%d si_signo=%d inset=%d\n", idx,
1965     		     sq->sigs[idx].si_signo,
1966                          VG_(sigismember)(set, sq->sigs[idx].si_signo));
1967     
1968           if (sq->sigs[idx].si_signo != 0 
1969               && VG_(sigismember)(set, sq->sigs[idx].si_signo)) {
1970     	 if (VG_(clo_trace_signals))
1971                 VG_(dmsg)("Returning queued signal %d (idx %d) for thread %d\n",
1972                           sq->sigs[idx].si_signo, idx, tid);
1973     	 ret = &sq->sigs[idx];
1974     	 goto out;
1975           }
1976     
1977           idx = (idx + 1) % N_QUEUED_SIGNALS;
1978        } while(idx != sq->next);
1979       out:   
1980        return ret;
1981     }

2002     /* 
2003        Receive an async signal from the kernel.
2004     
2005        This should only happen when the thread is blocked in a syscall,
2006        since that's the only time this set of signals is unblocked.
2007     */
2008     static 
2009     void async_signalhandler ( Int sigNo,
2010                                vki_siginfo_t *info, struct vki_ucontext *uc )
2011     {
2012        ThreadId     tid = VG_(lwpid_to_vgtid)(VG_(gettid)());
2013        ThreadState* tst = VG_(get_ThreadState)(tid);
2014        SysRes       sres;
2015     
2016        /* The thread isn't currently running, make it so before going on */
2017        vg_assert(tst->status == VgTs_WaitSys);
2018        VG_(acquire_BigLock)(tid, "async_signalhandler");
2019     
2020        info->si_code = sanitize_si_code(info->si_code);
2021     
2022        if (VG_(clo_trace_signals))
2023           VG_(dmsg)("async signal handler: signal=%d, tid=%d, si_code=%d\n",
2024                     sigNo, tid, info->si_code);
2025     
2026        /* Update thread state properly.  The signal can only have been
2027           delivered whilst we were in
2028           coregrind/m_syswrap/syscall-<PLAT>.S, and only then in the
2029           window between the two sigprocmask calls, since at all other
2030           times, we run with async signals on the host blocked.  Hence
2031           make enquiries on the basis that we were in or very close to a
2032           syscall, and attempt to fix up the guest state accordingly.
2033     
2034           (normal async signals occurring during computation are blocked,
2035           but periodically polled for using VG_(sigtimedwait_zero), and
2036           delivered at a point convenient for us.  Hence this routine only
2037           deals with signals that are delivered to a thread during a
2038           syscall.) */
2039     
2040        /* First, extract a SysRes from the ucontext_t* given to this
2041           handler.  If it is subsequently established by
2042           VG_(fixup_guest_state_after_syscall_interrupted) that the
2043           syscall was complete but the results had not been committed yet
2044           to the guest state, then it'll have to commit the results itself
2045           "by hand", and so we need to extract the SysRes.  Of course if
2046           the thread was not in that particular window then the
2047           SysRes will be meaningless, but that's OK too because
2048           VG_(fixup_guest_state_after_syscall_interrupted) will detect
2049           that the thread was not in said window and ignore the SysRes. */
2050     
2051        /* To make matters more complex still, on Darwin we need to know
2052           the "class" of the syscall under consideration in order to be
2053           able to extract the a correct SysRes.  The class will have been
2054           saved just before the syscall, by VG_(client_syscall), into this
2055           thread's tst->arch.vex.guest_SC_CLASS.  Hence: */
2056     #  if defined(VGO_darwin)
2057        sres = VG_UCONTEXT_SYSCALL_SYSRES(uc, tst->arch.vex.guest_SC_CLASS);
2058     #  else
2059        sres = VG_UCONTEXT_SYSCALL_SYSRES(uc);
2060     #  endif
2061     
2062        /* (1) */
2063        VG_(fixup_guest_state_after_syscall_interrupted)(
2064           tid, 
2065           VG_UCONTEXT_INSTR_PTR(uc), 
2066           sres,  
2067           !!(scss.scss_per_sig[sigNo].scss_flags & VKI_SA_RESTART)
2068        );
2069     
2070        /* (2) */
2071        /* Set up the thread's state to deliver a signal */
2072        if (!is_sig_ign(info->si_signo))
2073           deliver_signal(tid, info, uc);
2074     
2075        /* It's crucial that (1) and (2) happen in the order (1) then (2)
2076           and not the other way around.  (1) fixes up the guest thread
2077           state to reflect the fact that the syscall was interrupted --
2078           either to restart the syscall or to return EINTR.  (2) then sets
2079           up the thread state to deliver the signal.  Then we resume
2080           execution.  First, the signal handler is run, since that's the
2081           second adjustment we made to the thread state.  If that returns,
2082           then we resume at the guest state created by (1), viz, either
2083           the syscall returns EINTR or is restarted.
2084     
2085           If (2) was done before (1) the outcome would be completely
2086           different, and wrong. */
2087     
2088        /* longjmp back to the thread's main loop to start executing the
2089           handler. */
2090        resume_scheduler(tid);
2091     
2092        VG_(core_panic)("async_signalhandler: got unexpected signal "
2093                        "while outside of scheduler");
2094     }

2096     /* Extend the stack to cover addr.  maxsize is the limit the stack can grow to.
2097     
2098        Returns True on success, False on failure.
2099     
2100        Succeeds without doing anything if addr is already within a segment.
2101     
2102        Failure could be caused by:
2103        - addr not below a growable segment
2104        - new stack size would exceed maxsize
2105        - mmap failed for some other reason
2106      */
2107     Bool VG_(extend_stack)(Addr addr, UInt maxsize)
2108     {
2109        SizeT udelta;
2110     
2111        /* Find the next Segment above addr */
2112        NSegment const* seg
2113           = VG_(am_find_nsegment)(addr);
2114        NSegment const* seg_next 
2115           = seg ? VG_(am_next_nsegment)( (NSegment*)seg, True/*fwds*/ )
2116                 : NULL;
2117     
2118        if (seg && seg->kind == SkAnonC)
2119           /* addr is already mapped.  Nothing to do. */
2120           return True;
2121     
2122        /* Check that the requested new base is in a shrink-down
2123           reservation section which abuts an anonymous mapping that
2124           belongs to the client. */
2125        if ( ! (seg
2126                && seg->kind == SkResvn
2127                && seg->smode == SmUpper
2128                && seg_next
2129                && seg_next->kind == SkAnonC
2130                && seg->end+1 == seg_next->start))
2131           return False;
2132     
2133        udelta = VG_PGROUNDUP(seg_next->start - addr);
2134        VG_(debugLog)(1, "signals", 
2135                         "extending a stack base 0x%llx down by %lld\n",
2136                         (ULong)seg_next->start, (ULong)udelta);
2137        if (! VG_(am_extend_into_adjacent_reservation_client)
2138                 ( (NSegment*)seg_next, -(SSizeT)udelta )) {
2139           VG_(debugLog)(1, "signals", "extending a stack base: FAILED\n");
2140           return False;
2141        }
2142     
2143        /* When we change the main stack, we have to let the stack handling
2144           code know about it. */

2166     void sync_signalhandler_from_user ( ThreadId tid,
2167              Int sigNo, vki_siginfo_t *info, struct vki_ucontext *uc )
2168     {
2169        ThreadId qtid;
2170     
2171        /* If some user-process sent us a sync signal (ie. it's not the result
2172           of a faulting instruction), then how we treat it depends on when it
2173           arrives... */
2174     
2175        if (VG_(threads)[tid].status == VgTs_WaitSys) {
2176           /* Signal arrived while we're blocked in a syscall.  This means that
2177              the client's signal mask was applied.  In other words, so we can't
2178              get here unless the client wants this signal right now.  This means
2179              we can simply use the async_signalhandler. */
2180           if (VG_(clo_trace_signals))
2181              VG_(dmsg)("Delivering user-sent sync signal %d as async signal\n",
2182                        sigNo);
2183     
2184           async_signalhandler(sigNo, info, uc);
2185           VG_(core_panic)("async_signalhandler returned!?\n");
2186     
2187        } else {
2188           /* Signal arrived while in generated client code, or while running
2189              Valgrind core code.  That means that every thread has these signals
2190              unblocked, so we can't rely on the kernel to route them properly, so
2191              we need to queue them manually. */
2192           if (VG_(clo_trace_signals))
2193              VG_(dmsg)("Routing user-sent sync signal %d via queue\n", sigNo);
2194     
2195     #     if defined(VGO_linux)
2196           /* On Linux, first we have to do a sanity check of the siginfo. */
2197           if (info->VKI_SIGINFO_si_pid == 0) {
2198              /* There's a per-user limit of pending siginfo signals.  If
2199                 you exceed this, by having more than that number of
2200                 pending signals with siginfo, then new signals are
2201                 delivered without siginfo.  This condition can be caused
2202                 by any unrelated program you're running at the same time
2203                 as Valgrind, if it has a large number of pending siginfo
2204                 signals which it isn't taking delivery of.
2205     
2206                 Since we depend on siginfo to work out why we were sent a
2207                 signal and what we should do about it, we really can't
2208                 continue unless we get it. */
2209              VG_(umsg)("Signal %d (%s) appears to have lost its siginfo; "
2210                        "I can't go on.\n", sigNo, signame(sigNo));
2211              VG_(printf)(

2213     "  siginfo structures.  For more information, see:\n"
2214     "    http://kerneltrap.org/mailarchive/1/message/25599/thread\n"
2215     "  Basically, some program on your system is building up a large queue of\n"
2216     "  pending signals, and this causes the siginfo data for other signals to\n"
2217     "  be dropped because it's exceeding a system limit.  However, Valgrind\n"
2218     "  absolutely needs siginfo for SIGSEGV.  A workaround is to track down the\n"
2219     "  offending program and avoid running it while using Valgrind, but there\n"
2220     "  is no easy way to do this.  Apparently the problem was fixed in kernel\n"
2221     "  2.6.12.\n");
2222     
2223              /* It's a fatal signal, so we force the default handler. */
2224              VG_(set_default_handler)(sigNo);
2225              deliver_signal(tid, info, uc);
2226              resume_scheduler(tid);
2227              VG_(exit)(99);       /* If we can't resume, then just exit */
2228           }
2229     #     endif
2230     
2231           qtid = 0;         /* shared pending by default */
2232     #     if defined(VGO_linux)
2233           if (info->si_code == VKI_SI_TKILL)
2234              qtid = tid;    /* directed to us specifically */
2235     #     endif
2236           queue_signal(qtid, info);
2237        }

2253     /* Returns True if the sync signal was due to the stack requiring extension
2254        and the extension was successful.
2255     */
2256     static Bool extend_stack_if_appropriate(ThreadId tid, vki_siginfo_t* info)
2257     {
2258        Addr fault;
2259        Addr esp;
2260        NSegment const* seg;
2261        NSegment const* seg_next;
2262     
2263        if (info->si_signo != VKI_SIGSEGV)
2264           return False;
2265     
2266        fault    = (Addr)info->VKI_SIGINFO_si_addr;
2267        esp      = VG_(get_SP)(tid);
2268        seg      = VG_(am_find_nsegment)(fault);
2269        seg_next = seg ? VG_(am_next_nsegment)( (NSegment*)seg, True/*fwds*/ )
2270                       : NULL;
2271     
2272        if (VG_(clo_trace_signals)) {
2273           if (seg == NULL)
2274              VG_(dmsg)("SIGSEGV: si_code=%d faultaddr=%#lx tid=%d ESP=%#lx "
2275                        "seg=NULL\n",
2276                        info->si_code, fault, tid, esp);
2277           else
2278              VG_(dmsg)("SIGSEGV: si_code=%d faultaddr=%#lx tid=%d ESP=%#lx "
2279                        "seg=%#lx-%#lx\n",
2280                        info->si_code, fault, tid, esp, seg->start, seg->end);
2281        }
2282     
2283        if (info->si_code == VKI_SEGV_MAPERR
2284            && seg
2285            && seg->kind == SkResvn
2286            && seg->smode == SmUpper
2287            && seg_next
2288            && seg_next->kind == SkAnonC
2289            && seg->end+1 == seg_next->start
2290            && fault >= fault_mask(esp - VG_STACK_REDZONE_SZB)) {
2291           /* If the fault address is above esp but below the current known
2292              stack segment base, and it was a fault because there was
2293              nothing mapped there (as opposed to a permissions fault),
2294              then extend the stack segment. 
2295            */
2296           Addr base = VG_PGROUNDDN(esp - VG_STACK_REDZONE_SZB);
2297           if (VG_(extend_stack)(base, VG_(threads)[tid].client_stack_szB)) {
2298              if (VG_(clo_trace_signals))
2299                 VG_(dmsg)("       -> extended stack base to %#lx\n",
2300                           VG_PGROUNDDN(fault));
2301              return True;
2302           } else {
2303              VG_(umsg)("Stack overflow in thread %d: can't grow stack to %#lx\n",
2304                        tid, fault);
2305              return False;
2306           }
2307        } else {
2308           return False;
2309        }

2312     static
2313     void sync_signalhandler_from_kernel ( ThreadId tid,
2314              Int sigNo, vki_siginfo_t *info, struct vki_ucontext *uc )
2315     {
2316        /* Check to see if some part of Valgrind itself is interested in faults.
2317           The fault catcher should never be set whilst we're in generated code, so
2318           check for that.  AFAIK the only use of the catcher right now is
2319           memcheck's leak detector. */
2320        if (fault_catcher) {
2321           vg_assert(VG_(in_generated_code) == False);
2322     
2323           (*fault_catcher)(sigNo, (Addr)info->VKI_SIGINFO_si_addr);
2324           /* If the catcher returns, then it didn't handle the fault,
2325              so carry on panicking. */
2326        }
2327     
2328        if (extend_stack_if_appropriate(tid, info)) {
2329           /* Stack extension occurred, so we don't need to do anything else; upon
2330              returning from this function, we'll restart the host (hence guest)
2331              instruction. */
2332        } else {
2333           /* OK, this is a signal we really have to deal with.  If it came
2334              from the client's code, then we can jump back into the scheduler
2335              and have it delivered.  Otherwise it's a Valgrind bug. */
2336           ThreadState *tst = VG_(get_ThreadState)(tid);
2337     
2338           if (VG_(sigismember)(&tst->sig_mask, sigNo)) {
2339              /* signal is blocked, but they're not allowed to block faults */
2340              VG_(set_default_handler)(sigNo);
2341           }
2342     
2343           if (VG_(in_generated_code)) {
2344              /* Can't continue; must longjmp back to the scheduler and thus
2345                 enter the sighandler immediately. */
2346              deliver_signal(tid, info, uc);
2347              resume_scheduler(tid);
2348           }
2349     
2350           /* If resume_scheduler returns or its our fault, it means we
2351              don't have longjmp set up, implying that we weren't running
2352              client code, and therefore it was actually generated by
2353              Valgrind internally.
2354            */
2355           VG_(dmsg)("VALGRIND INTERNAL ERROR: Valgrind received "
2356                     "a signal %d (%s) - exiting\n",
2357                     sigNo, signame(sigNo));
2358     
2359           VG_(dmsg)("si_code=%x;  Faulting address: %p;  sp: %#lx\n",
2360                     info->si_code, info->VKI_SIGINFO_si_addr,
2361                     VG_UCONTEXT_STACK_PTR(uc));
2362     
2363           if (0)
2364              VG_(kill_self)(sigNo);  /* generate a core dump */
2365     
2366           //if (tid == 0)            /* could happen after everyone has exited */
2367           //  tid = VG_(master_tid);
2368           vg_assert(tid != 0);
2369     
2370           UnwindStartRegs startRegs;
2371           VG_(memset)(&startRegs, 0, sizeof(startRegs));
2372     
2373           VG_UCONTEXT_TO_UnwindStartRegs(&startRegs, uc);
2374           VG_(core_panic_at)("Killed by fatal signal", &startRegs);
2375        }
2376     }

2514        Poll for pending signals, and set the next one up for delivery.
2515      */
2516     void VG_(poll_signals)(ThreadId tid)
2517     {
2518        vki_siginfo_t si, *sip;
2519        vki_sigset_t pollset;
2520        ThreadState *tst = VG_(get_ThreadState)(tid);
2521        vki_sigset_t saved_mask;
2522     
2523        /* look for all the signals this thread isn't blocking */
2524        /* pollset = ~tst->sig_mask */
2525        VG_(sigcomplementset)( &pollset, &tst->sig_mask );
2526     
2527        block_all_host_signals(&saved_mask); // protect signal queue
2528     
2529        /* First look for any queued pending signals */
2530        sip = next_queued(tid, &pollset); /* this thread */
2531     
2532        if (sip == NULL)
2533           sip = next_queued(0, &pollset); /* process-wide */
2534     
2535        /* If there was nothing queued, ask the kernel for a pending signal */
2536        if (sip == NULL && VG_(sigtimedwait_zero)(&pollset, &si) > 0) {

2539                        si.si_signo, tid);
2540           sip = &si;
2541        }

2553     				   where it came from */

2670           will set the appropriate mask at the appropriate time. */
