18         static AbstractCfgNode get(String marker, TacFunction function, TacConverter tac) {
19             // EFF: distinguish between markers more efficiently
20             AbstractCfgNode retMe = null;
21     
22             if (marker.startsWith("_test_")) {
23                 // a tester node:
24                 //
25                 // _test_<whatToTest>_<numList>
26                 // e.g.:
27                 // _test_taint_0
28                 // _test_taint_3_7
29                 // _test_arrayLabel_12_24
30     
31                 // find the underline after the "whatToTest" substring
32                 int delimiter = marker.indexOf('_', 6);
33     
34                 if (delimiter == -1) {
35                     throw new RuntimeException("Error: Invalid '~_test_' marker in builtin functions file");
36                 }
37     
38                 // the "whatToTest" substring
39                 String whatToTest = marker.substring(6, delimiter);
40     
41                 // find out what to test...
42                 int whatToTestInt;
43                 switch (whatToTest) {
44                     case "taint":
45                         whatToTestInt = Tester.TEST_TAINT;
46                         break;
47                     case "arrayLabel":
48                         whatToTestInt = Tester.TEST_ARRAYLABEL;
49                         break;
50                     default:
51                         throw new RuntimeException("Error: Invalid '~_test_' marker in builtin functions file");
52                 }
53     
54                 Set<Integer> numSet;
55                 try {
56                     numSet = makeNumSet(marker.substring(delimiter + 1, marker.length()));
57                 } catch (NumberFormatException ex) {
58                     throw new RuntimeException("Error: Invalid '~_test_' marker in builtin functions file");
59                 }
60     
61                 retMe = new Tester(whatToTestInt, numSet);
62             } else if (marker.startsWith("_hotspot")) {
63                 // hotspots are only used for JUnit tests
64                 try {
65                     Integer hotspotId = Integer.valueOf(marker.substring(8));
66                     retMe = new Hotspot(hotspotId);
67                     tac.addHotspot((Hotspot) retMe);
68                 } catch (IndexOutOfBoundsException ex) {
69                     throw new RuntimeException("Illegal hotspot marker: no ID suffix");
70                 } catch (NumberFormatException ex) {
71                     throw new RuntimeException("Illegal hotspot marker: non-numeric ID suffix");
72                 }
73             } else {
74                 System.out.println("Unkown marker: " + marker);
75                 throw new RuntimeException("SNH");
76             }
77             return retMe;
78         }

80         // input: "num1" or "num1_num2" or "num1_num2_num3" ...
81         // with numX from {0, 1, 2, ...};
82         // output: Set of Integers
83         private static Set<Integer> makeNumSet(String numString)
84             throws NumberFormatException {
85     
86             Set<Integer> numSet = new HashSet<>();
87     
88             boolean findNext = true;
89             int from = 0;
90             int to;
91     
92             while (findNext) {
93     
94                 to = numString.indexOf('_', from);
95                 String num;
96                 if (to == -1) {
97                     // this is the last one
98                     num = numString.substring(from, numString.length());
99                     findNext = false;
100                 } else {
101                     // there are others
102                     num = numString.substring(from, to);
103                     findNext = true;
104                     from = to + 1;
105                 }
106                 Integer numInt = Integer.valueOf(num);
107                 numSet.add(numInt);
108             }
109     
110             return numSet;
111         }
