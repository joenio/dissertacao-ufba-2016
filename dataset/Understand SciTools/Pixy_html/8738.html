15     public class TypeLatticeElement extends AbstractLatticeElement {
16         private Map<Variable, Set<Type>> var2Type;

18         // an empty lattice element (the analysis starts with this one)
19         public TypeLatticeElement() {
20             this.var2Type = new HashMap<>();
21         }
22     
23         // clones the given element
24         public TypeLatticeElement(TypeLatticeElement element) {
25             this.var2Type = new HashMap<>(element.var2Type);
26         }

28         // lubs the given lattice element over <<this>> lattice element
29         public void lub(AbstractLatticeElement foreignX) {
30             // for all foreign mappings...
31             for (Map.Entry<Variable, Set<Type>> entry : ((TypeLatticeElement) foreignX).var2Type.entrySet()) {
32                 Variable foreignVar = entry.getKey();
33                 Set<Type> foreignTypes = entry.getValue();
34                 Set<Type> myTypes = this.var2Type.get(foreignVar);
35                 if (this.var2Type.containsKey(foreignVar)) {
36                     // if we already have a mapping for this variable:
37                     // union over the types
38                     myTypes.addAll(foreignTypes);
39                 } else {
40                     // if we don't have a mapping for this variable yet:
41                     // add this mapping
42                     this.var2Type.put(foreignVar, foreignTypes);
43                 }
44             }
45         }

47         public void setTypeString(Variable var, String className) {
48             Set<Type> types = new HashSet<>();
49             types.add(Type.getTypeForClass(className));
50             this.setType(var, types);
51         }
52     
53         private void setType(Variable var, Set<Type> types) {
54     
55             if (var.isMember()) {
56                 // we don't want to modify the special member variable
57                 return;
58             }
59     
60             if (types == null) {
61                 // simply remove the mapping
62                 this.var2Type.remove(var);
63             } else {
64                 this.var2Type.put(var, types);
65             }
66         }
67     
68         // can also return null
69         public Set<Type> getType(Variable var) {
70             return this.var2Type.get(var);
71         }
72     
73         public void assign(Variable left, AbstractTacPlace right) {
74             if (!(right instanceof Variable)) {
75                 // we are only tracking object types at the moment
76                 this.var2Type.remove(left);
77                 return;
78             }
79             Variable rightVar = (Variable) right;
80             Set<Type> rightTypes = this.var2Type.get(rightVar);
81             this.setType(left, rightTypes);
82         }

84         public void assignUnary(Variable left) {
85             // simply delete all info about left
86             this.setType(left, null);
87         }

89         public void assignBinary(Variable left) {
90             // simply delete all info about left
91             this.setType(left, null);
92         }

94         public void unset(Variable var) {

98         public void assignArray(Variable var) {

106         // resets all variables that belong to the given symbol table
107         public void resetVariables(SymbolTable symTab) {
108             for (Iterator<Map.Entry<Variable, Set<Type>>> iter = this.var2Type.entrySet().iterator(); iter.hasNext(); ) {
109                 Map.Entry<Variable, Set<Type>> entry = iter.next();
110                 Variable var = entry.getKey();
111     
112                 if (var.belongsTo(symTab)) {
113                     iter.remove();
114                 }
115             }
116         }
117     
118         // resets all temporaries that belong to the given symbol table
119         public void resetTemporaries(SymbolTable symTab) {
120             for (Iterator<Map.Entry<Variable, Set<Type>>> iter = this.var2Type.entrySet().iterator(); iter.hasNext(); ) {
121                 Map.Entry<Variable, Set<Type>> entry = iter.next();
122                 Variable var = entry.getKey();
123     
124                 if (!var.isTemp()) {
125                     // nothing to do for non-temporaries
126                     continue;
127                 }
128     
129                 if (var.belongsTo(symTab)) {
130                     iter.remove();
131                 }
132             }
133         }

135         // copies the mappings for "global-like" places
136         // from interIn (i.e., global variables, superglobal variables, and
137         // constants)
138         public void copyGlobalLike(TypeLatticeElement interIn) {
139             for (Map.Entry<Variable, Set<Type>> entry : interIn.var2Type.entrySet()) {
140                 Variable origVar = entry.getKey();
141                 Set<Type> origTypes = entry.getValue();
142     
143                 if (origVar.isGlobal() || origVar.isSuperGlobal()) {
144                     this.setType(origVar, origTypes);
145                 }
146             }
147         }
148     
149         // copies the mappings for local temporaries of the main function
150         public void copyMainTemporaries(TypeLatticeElement origElement) {
151             for (Map.Entry<Variable, Set<Type>> entry : origElement.var2Type.entrySet()) {
152                 Variable origVar = entry.getKey();
153                 Set<Type> origTypes = entry.getValue();
154     
155                 // nothing to do for non-main's and non-temporaries
156                 SymbolTable symTab = origVar.getSymbolTable();
157                 if (!symTab.isMain()) {
158                     continue;
159                 }
160                 if (!origVar.isTemp()) {
161                     continue;
162                 }
163     
164                 this.setType(origVar, origTypes);
165             }
166         }
167     
168         // sets the temporary responsible for catching the return value
169         // and resets the return variable
170         public void handleReturnValue(Call callNode,
171                                       TypeLatticeElement calleeIn, TacFunction callee) {
172     
173             Variable tempVar = callNode.getTempVar();
174             Set<Type> types = calleeIn.getType(callNode.getRetVar());
175             Variable retVar = callNode.getRetVar();
176     
177             // if the callee is a constructor, return the appropriate class as type
178             if (callee.isConstructor()) {
179                 types = new HashSet<>();
180                 types.add(Type.getTypeForClass(callee.getClassName()));
181             }
182     
183             this.setType(tempVar, types);
184             this.var2Type.remove(retVar);
185         }
186     
187         // copies the mappings for local variables from origElement
188         public void copyLocals(TypeLatticeElement origElement) {
189     
190             for (Map.Entry<Variable, Set<Type>> entry : origElement.var2Type.entrySet()) {
191     
192                 Variable origVar = entry.getKey();
193                 Set<Type> origTypes = entry.getValue();
194     
195                 // nothing to do for non-locals
196                 if (!origVar.isLocal()) {
197                     continue;
198                 }
199     
200                 this.setType(origVar, origTypes);
201             }
202         }
