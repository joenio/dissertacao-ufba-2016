15     /**
16      * Computes a reverse post-order for the whole, interprocedural cfg.
17      *
18      * Currently only works for call-string analysis.
19      *
20      * @author Nenad Jovanovic <enji@seclab.tuwien.ac.at>
21      */
22     public class InterproceduralWorklistOrder {
23         // this is what we want to compute: a mapping of interprocedural
24         // worklist elements to some number (order)
25         private Map<InterproceduralWorklistElement, Integer> order;
26     
27     //  ********************************************************************************
28     
29         public InterproceduralWorklistOrder(TacConverter tac, ConnectorComputation cc) {
30             this.order = new HashMap<>();
31     
32             TacFunction mainFunction = tac.getMainFunction();
33             AbstractCfgNode startNode = mainFunction.getControlFlowGraph().getHead();
34     
35             Map<TacFunction, EncodedCallStrings> function2ECS = cc.getFunction2ECS();
36             EncodedCallStrings mainEncodedCallStrings = function2ECS.get(mainFunction);
37             if (mainEncodedCallStrings.size() != 1) {
38                 throw new RuntimeException("SNH");
39             }
40     
41             InterproceduralWorklistElement start = new InterproceduralWorklistElement(startNode, new CallStringContext(0));
42             LinkedList<InterproceduralWorklistElement> postorder = this.getPostorder(start, cc);
43     
44             // get *reverse* postorder
45             ListIterator<InterproceduralWorklistElement> iter = postorder.listIterator(postorder.size());
46             int i = 0;
47             while (iter.hasPrevious()) {
48                 InterproceduralWorklistElement iwle = iter.previous();
49                 this.order.put(iwle, i);
50                 i++;
51             }
52         }
53     
54     //  ********************************************************************************

56         // non-recursive postorder
57         private LinkedList<InterproceduralWorklistElement> getPostorder(
58             InterproceduralWorklistElement start, ConnectorComputation cc) {
59     
60             // this is what we want to compute
61             LinkedList<InterproceduralWorklistElement> postorder = new LinkedList<>();
62     
63             // auxiliary stack and visited set
64             LinkedList<InterproceduralWorklistElement> stack = new LinkedList<>();
65             Set<InterproceduralWorklistElement> visited = new HashSet<>();
66     
67             // begin with start element
68             stack.add(start);
69     
70             // how it works:
71             // while there is something on the stack:
72             // - mark the top stack element as visited
73             // - try to get an unvisited successor of this element
74             // - if there is such a successor: push it on the stack and continue
75             // - else: pop the stack and add the popped element to the postorder list
76             while (!stack.isEmpty()) {
77     
78                 // mark the top stack element as visited
79                 InterproceduralWorklistElement element = stack.getLast();
80                 visited.add(element);
81     
82                 // interior of this element
83                 AbstractCfgNode cfgNode = element.getCfgNode();
84                 CallStringContext context = (CallStringContext) element.getContext();
85     
86                 // we will try to get an unvisited successor element
87                 InterproceduralWorklistElement nextElement = null;
88     
89                 if (cfgNode instanceof Call) {
90     
91                     // in case of a call node, we have to distinguish between
92                     // unknown calls (no callee available) and known calls
93     
94                     Call callNode = (Call) cfgNode;
95                     TacFunction callee = callNode.getCallee();
96                     if (callee == null) {
97                         // for unknown calls:
98                         // simply move on to the callret node; context stays the same
99     
100                         AbstractCfgNode retNode = callNode.getSuccessor(0);
101                         nextElement = new InterproceduralWorklistElement(retNode, context);
102     
103                         if (visited.contains(nextElement)) {
104                             nextElement = null;
105                         }
106                     } else {
107                         // for normal calls:
108                         // enter function under corresponding context
109     
110                         CfgEntry entryNode = (CfgEntry) callee.getControlFlowGraph().getHead();
111                         AbstractContext propagationContext = cc.getTargetContext(callNode, context.getPosition());
112                         if (propagationContext == null) {
113                             throw new RuntimeException("SNH: " + callNode.getLoc());
114                         }
115                         nextElement = new InterproceduralWorklistElement(entryNode, propagationContext);
116     
117                         if (visited.contains(nextElement)) {
118                             nextElement = null;
119                         }
120                     }
121                 } else if (cfgNode instanceof CfgExit) {
122                     CfgExit exitNode = (CfgExit) cfgNode;
123                     TacFunction exitedFunction = exitNode.getEnclosingFunction();
124     
125                     // only proceed if this is not the exit node of the main function
126                     if (!exitedFunction.isMain()) {
127     
128                         // an exit node can have several "reverse targets";
129                         // a reverse target consists of one call node and one or more contexts
130     
131                         Iterator<ReverseTarget> revTargetsIter = cc.getReverseTargets(exitedFunction, context.getPosition()).iterator();
132                         while ((nextElement == null) && revTargetsIter.hasNext()) {
133     
134                             ReverseTarget revTarget = revTargetsIter.next();
135                             Call revCall = revTarget.getCallNode();
136                             AbstractCfgNode revRet = revCall.getSuccessor(0);
137                             Iterator<? extends AbstractContext> reverseContextsIter = revTarget.getContexts().iterator();
138     
139                             while ((nextElement == null) && reverseContextsIter.hasNext()) {
140     
141                                 AbstractContext reverseContext = reverseContextsIter.next();
142                                 nextElement = new InterproceduralWorklistElement(revRet, reverseContext);
143     
144                                 if (visited.contains(nextElement)) {
145                                     // try the next one
146                                     nextElement = null;
147                                 } else {
148                                     // found it!
149                                 }
150                             }
151                         }
152                     }
153                 } else {
154     
155                     // handle successors
156                     for (int i = 0; (i < 2) && (nextElement == null); i++) {
157                         CfgEdge outEdge = cfgNode.getOutEdge(i);
158                         if (outEdge != null) {
159                             AbstractCfgNode succNode = outEdge.getDestination();
160                             nextElement = new InterproceduralWorklistElement(succNode, context);
161                             if (visited.contains(nextElement)) {
162                                 // try next one
163                                 nextElement = null;
164                             } else {
165                                 // found it!
166                             }
167                         }
168                     }
169                 }
170     
171                 if (nextElement == null) {
172                     // pop from stack and add it to the postorder list
173                     postorder.add(stack.removeLast());
174                 } else {
175                     // push to stack
176                     stack.add(nextElement);
177                 }
178             }
179     
180             return postorder;
181         }

184     
185         public Integer getReversePostOrder(InterproceduralWorklistElement element) {
186             return this.order.get(element);
187         }
