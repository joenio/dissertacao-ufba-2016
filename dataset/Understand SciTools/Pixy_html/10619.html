45     //  createTf ***********************************************************************
46     
47         // creates a transfer function for the given node;
48         // the enclosingNode is either an enclosing basic block (if you already know
49         // that it is enclosed by a basic block) or the node itself
50         protected AbstractTransferFunction createTf(AbstractCfgNode cfgNodeX, TacFunction traversedFunction, AbstractCfgNode enclosingNode) {
51     
52             // EFF: more efficient implementation (hashmap?)
53     
54             // CAUTION: check for basic block first!
55             if (cfgNodeX instanceof BasicBlock) {
56     
57                 BasicBlock cfgNode = (BasicBlock) cfgNodeX;
58                 return this.makeBasicBlockTf(cfgNode, traversedFunction);
59             } else if (cfgNodeX instanceof AssignSimple) {
60     
61                 return this.assignSimple(cfgNodeX, enclosingNode);
62             } else if (cfgNodeX instanceof AssignUnary) {
63     
64                 return this.assignUnary(cfgNodeX, enclosingNode);
65             } else if (cfgNodeX instanceof AssignBinary) {
66     
67                 return this.assignBinary(cfgNodeX, enclosingNode);
68             } else if (cfgNodeX instanceof AssignReference) {
69     
70                 return this.assignRef(cfgNodeX);
71             } else if (cfgNodeX instanceof Unset) {
72     
73                 return this.unset(cfgNodeX);
74             } else if (cfgNodeX instanceof AssignArray) {
75     
76                 return this.assignArray(cfgNodeX);
77             } else if (cfgNodeX instanceof Isset) {
78     
79                 return this.isset(cfgNodeX);
80             } else if (cfgNodeX instanceof CallPreparation) {
81     
82                 return this.callPrep(cfgNodeX, traversedFunction);
83             } else if (cfgNodeX instanceof CfgEntry) {
84     
85                 return this.entry(traversedFunction);
86             } else if (cfgNodeX instanceof CallReturn) {
87     
88                 return this.callRet(cfgNodeX, traversedFunction);
89             } else if (cfgNodeX instanceof CallBuiltinFunction) {
90     
91                 return this.callBuiltin(cfgNodeX, traversedFunction);
92             } else if (cfgNodeX instanceof CallUnknownFunction) {
93     
94                 return this.callUnknown(cfgNodeX, traversedFunction);
95             } else if (cfgNodeX instanceof Global) {
96     
97                 return this.global(cfgNodeX);
98             } else if (cfgNodeX instanceof Define) {
99     
100                 return this.define(cfgNodeX);
101             } else if (cfgNodeX instanceof Tester) {
102     
103                 return this.tester(cfgNodeX);
104             } else if (cfgNodeX instanceof Echo) {
105     
106                 return this.echo(cfgNodeX, traversedFunction);
107             } else if (cfgNodeX instanceof Static) {
108     
109                 return this.staticNode();
110             } else if (cfgNodeX instanceof Include) {
111     
112                 return this.include(cfgNodeX);
113             } else if (cfgNodeX instanceof IncludeStart) {
114     
115                 return this.includeStart(cfgNodeX);
116             } else if (cfgNodeX instanceof IncludeEnd) {
117     
118                 return this.includeEnd(cfgNodeX);
119             } else {
120                 // ID transfer function for all remaining cfg node types
121                 return TransferFunctionId.INSTANCE;
122             }
123         }
124     
125     //  traverseCfg ********************************************************************
126     
127         protected void traverseCfg(ControlFlowGraph controlFlowGraph, TacFunction traversedFunction) {
128     
129             for (AbstractCfgNode cfgNodeX : controlFlowGraph.dfPreOrder()) {
130     
131                 AbstractTransferFunction tf = this.createTf(cfgNodeX, traversedFunction, cfgNodeX);
132                 if (tf == null) {
133                     System.out.println(cfgNodeX.getLoc());
134                     throw new RuntimeException("SNH");
135                 }
136                 this.genericAnalysisInformation.add(cfgNodeX, this.makeAnalysisNode(
137                     cfgNodeX, tf));
138             }
139         }

169         // creates a transfer function for a whole basic block
170         protected AbstractTransferFunction makeBasicBlockTf(BasicBlock basicBlock, TacFunction traversedFunction) {
171     
172             CompositeTransferFunction ctf = new CompositeTransferFunction();
173     
174             for (AbstractCfgNode cfgNodeX : basicBlock.getContainedNodes()) {
175                 ctf.add(this.createTf(cfgNodeX, traversedFunction, basicBlock));
176             }
177             return ctf;
178         }
