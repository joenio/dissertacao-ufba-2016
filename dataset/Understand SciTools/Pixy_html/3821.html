18     public class Define extends AbstractTransferFunction {
19         private AbstractTacPlace setMe;
20         private AbstractTacPlace caseInsensitive;

23         private LiteralAnalysis literalAnalysis;
24         private AbstractCfgNode cfgNode;

30         public Define(ConstantsTable table, LiteralAnalysis literalAnalysis,
31                       at.ac.tuwien.infosys.www.pixy.conversion.cfgnodes.Define cfgNode) {
32     
33             this.setMe = cfgNode.getSetMe();
34             this.caseInsensitive = cfgNode.getCaseInsensitive();
35             this.constantsTable = table;
36             this.literalAnalysis = literalAnalysis;
37             this.cfgNode = cfgNode;
38         }

41     // OTHER ***************************************************************************
42     // *********************************************************************************
43     
44         public AbstractLatticeElement transfer(AbstractLatticeElement inX) {
45     
46             DependencyLatticeElement in = (DependencyLatticeElement) inX;
47             DependencyLatticeElement out = new DependencyLatticeElement(in);
48     
49             // retrieve the literal of the constant to be set
50             Literal constantLit;
51             if (this.setMe instanceof Literal) {
52                 // e.g.: define('foo', ...);
53                 constantLit = (Literal) setMe;
54             } else {
55                 // ask literals analysis
56                 constantLit = literalAnalysis.getLiteral(setMe, cfgNode);
57             }
58     
59             // if we can't resolve the constant that is to be set, we can't do
60             // anything;
61             // to be precise, we would have to set all constants to
62             // TOP (or better: those that are still undefined), but since this
63             // case is rather seldom, we just issue a warning;
64             if (constantLit == Literal.TOP) {
65                 // warning was already issued by literals analysis
66                 return out;
67             }
68     
69             // determine the (boolean) literal of the case flag
70             if (this.caseInsensitive == Constant.TRUE) {
71                 // define insensitive constant
72                 // all constants in setMe's insensitivity group have to be set
73                 List<Constant> insensitiveGroup = this.constantsTable.getInsensitiveGroup(constantLit);
74                 if (insensitiveGroup != null) {
75                     for (Constant constant : insensitiveGroup) {
76                         out.defineConstant(constant, this.cfgNode);
77                     }
78                 } else {
79                     // this case happens when the user defines a constant which is never
80                     // used (not even in a case-insensitive way);
81                     System.out.println("Warning: a constant is defined, but never used");
82                     System.out.println("- name:    " + constantLit.toString());
83                     System.out.println("- defined: " + this.cfgNode.getLoc());
84                 }
85             } else if (this.caseInsensitive == Constant.FALSE) {
86                 // define sensitive constant
87     
88                 Constant constant = this.constantsTable.getConstant(constantLit.toString());
89                 if (constant == null) {
90                     // happens if the constant being defined is never used
91                     System.out.println("Warning: a constant is defined, but never used");
92                     System.out.println("- name:    " + constantLit.toString());
93                     System.out.println("- defined: " + this.cfgNode.getLoc());
94                 } else {
95                     out.defineConstant(constant, this.cfgNode);
96                 }
97             } else {
98     
99                 // we don't know the exact value of this flag;
100                 // hence, we perform a strong update for the immediate constant in
101                 // question, and a weak update for all constants in its insensitivity group
102     
103                 Constant constant = this.constantsTable.getConstant(constantLit.toString());
104                 if (constant == null) {
105                     // happens if the constant being defined is never used
106                     System.out.println("Warning: a constant is defined, but never used");
107                     System.out.println("- name:    " + constantLit.toString());
108                     System.out.println("- defined: " + this.cfgNode.getLoc());
109                 } else {
110                     out.defineConstant(constant, this.cfgNode);
111                 }
112     
113                 // all constants in setMe's insensitivity group have to undergo a weak update
114                 // (except setMe itself)
115                 List<Constant> insensitiveGroup = this.constantsTable.getInsensitiveGroup(constantLit);
116                 if (insensitiveGroup != null) {
117                     for (Constant weakConstant : insensitiveGroup) {
118                         if (!weakConstant.equals(constant)) {
119                             out.defineConstantWeak(weakConstant, this.cfgNode);
120                         }
121                     }
122                 } else {
123                     // this case happens when the user defines a constant which is never
124                     // used (not even in a case-insensitive way);
125                     System.out.println("Warning: a constant is defined, but never used");
126                     System.out.println("- name:    " + constantLit.toString());
127                     System.out.println("- defined: " + this.cfgNode.getLoc());
128                 }
129             }
130     
131             return out;
132         }
