41         private void mySetUp(String testFile, boolean functional) {
42     
43             Checker checker = new Checker(this.path + testFile);
44             MyOptions.option_A = true;    // perform alias analysis!
45             MyOptions.setAnalyses("xss");
46     
47             // initialize & analyze
48             TacConverter tac = checker.initialize().getTac();
49             checker.analyzeTaint(tac, functional);
50             this.dependencyAnalysis = checker.gta.dependencyAnalysis;
51             this.xssAnalysis = (XssAnalysis) checker.gta.getAbstractVulnerabilityAnalyses().get(0);
52     
53             // collect sinks
54             this.sinks = this.xssAnalysis.collectSinks();
55             Collections.sort(sinks);
56         }

58         // returns the contents of the given file as string
59         private String readFile(String fileName) {
60             StringBuilder ret = new StringBuilder();
61             try {
62                 FileReader fr = new FileReader(fileName);
63                 int c;
64                 ret = new StringBuilder();
65                 while ((c = fr.read()) != -1) {
66                     ret.append((char) c);
67                 }
68             } catch (FileNotFoundException e) {
69                 Assert.fail("File not found: " + fileName);
70             } catch (IOException e) {
71                 Assert.fail(e.getMessage());
72             }
73             return ret.toString();
74         }

83         private void performTest(String testNum, int sinkNum, int graphNum, boolean generate, boolean functional) {
84     
85             //generate = true;
86     
87             mySetUp("test" + testNum + ".php", functional);
88     
89             Assert.assertTrue("Sinks real: " + sinks.size() + ", expected: "
90                 + sinkNum, sinks.size() == sinkNum);
91     
92             // collect dependencyGraphs
93             List<DependencyGraph> dependencyGraphs = new LinkedList<>();
94             for (Sink sink : sinks) {
95                 dependencyGraphs.addAll(dependencyAnalysis.getDependencyGraphsForSink(sink));
96             }
97     
98             Assert.assertTrue("Graphs real: " + dependencyGraphs.size() + ", expected: "
99                 + graphNum, dependencyGraphs.size() == graphNum);
100     
101             int graphCount = 0;
102             for (DependencyGraph dependencyGraph : dependencyGraphs) {
103                 graphCount++;
104                 String fileName = "test" + testNum + "_" + graphCount;
105                 if (generate) {
106                     dependencyGraph.dumpDotUnique(fileName, this.path);
107                 } else {
108                     String encountered = dependencyGraph.makeDotUnique(fileName);
109                     String expected = this.readFile(this.path + fileName + ".dot");
110                     Assert.assertEquals(expected, encountered);
111                 }
112             }
113     
114             if (generate) {
115                 // just to make sure that you don't accidentally forget
116                 // to switch generation off, and turn checking on
117                 Assert.fail("no check performed");
118             }
119         }
