43         // returns the global variables that are may-aliases of the given variable
44         // (a set of Variables)
45         public Set<Variable> getGlobalAliases(Variable var) {
46             Set<Variable> retMe = new HashSet<>();
47             for (MayAliasPair pair : this.pairs) {
48                 Variable globalMayAlias = pair.getGlobalMayAlias(var);
49                 if (globalMayAlias != null) {
50                     retMe.add(globalMayAlias);
51                 }
52             }
53             return retMe;
54         }

56         // returns the local variables that are may-aliases of the given variable
57         // (a set of Variables)
58         public Set<Variable> getLocalAliases(Variable var) {
59             Set<Variable> retMe = new HashSet<>();
60             for (MayAliasPair pair : this.pairs) {
61                 Variable localMayAlias = pair.getLocalMayAlias(var);
62                 if (localMayAlias != null) {
63                     retMe.add(localMayAlias);
64                 }
65             }
66             return retMe;
67         }
68     
69         // returns a set of variables that are may-aliases of the given variable
70         public Set<Variable> getAliases(Variable var) {
71             Set<Variable> retMe = new HashSet<>();
72             for (MayAliasPair pair : this.pairs) {
73                 Variable mayAlias = pair.getMayAlias(var);
74                 if (mayAlias != null) {
75                     retMe.add(mayAlias);

95         // copies all pairs in which "right" appears and replaces "right"
96         // through "left" in these copies
97         public void addAliasFor(Variable left, Variable right) {
98             HashSet<MayAliasPair> newPairs = new HashSet<>();
99             for (MayAliasPair pair : this.pairs) {
100                 if (pair.contains(right)) {
101                     MayAliasPair newPair = new MayAliasPair(pair);
102                     newPair.replaceBy(right, left);
103                     newPairs.add(newPair);
104                 }
105             }
106             this.pairs.addAll(newPairs);
107         }
108     
109         // removes all pairs that contain the given variable
110         public void removePairsWith(Variable var) {
111             // EFF: it would be faster to locate the pairs to be
112             // removed by maintaining a map from variables to sets
113             // of pairs
114             for (Iterator<MayAliasPair> iter = this.pairs.iterator(); iter.hasNext(); ) {
115                 MayAliasPair pair = iter.next();
116                 if (pair.contains(var)) {
117                     iter.remove();
118                 }
119             }
120         }
121     
122         public void removeLocals() {
123             for (Iterator<MayAliasPair> iter = this.pairs.iterator(); iter.hasNext(); ) {
124                 MayAliasPair pair = iter.next();
125                 // if the pair contains locals, it can be removed as a whole
126                 if (pair.containsLocals()) {
127                     iter.remove();
128                 }
129             }
130         }
131     
132         public void removeGlobals() {
133             for (Iterator<MayAliasPair> iter = this.pairs.iterator(); iter.hasNext(); ) {
134                 MayAliasPair pair = iter.next();
135                 // if the pair contains locals, it can be removed as a whole
136                 if (pair.containsGlobals()) {
137                     iter.remove();
138                 }
139             }
140         }
141     
142         public void removeVariables(SymbolTable symTab) {
143             for (Iterator<MayAliasPair> iter = this.pairs.iterator(); iter.hasNext(); ) {
144                 MayAliasPair pair = iter.next();
145                 // if the pair contains one or two variables from the given
146                 // symbol table, it can be removed as a whole
147                 if (pair.containsVariables(symTab)) {
148                     iter.remove();
149                 }
150             }
151         }

153         // for all may-alias-pairs that contain "findMe":
154         // a copy of this pair is added, and "findMe" is replaced by "replacer" in
155         // the copy
156         public void createAdjustedPairCopies(Variable findMe, Variable replacer) {
157     
158             // EFF: there must be faster ways to do this
159     
160             // start by cloning the pair set; we will add adjusted copies to the
161             // new pair set and in the end, put it in place of the old pair set
162             Set<MayAliasPair> newPairs = new HashSet<>(this.pairs);
163     
164             // for each pair...
165             for (MayAliasPair pair : this.pairs) {
166                 // if it contains the searched variable...
167                 if (pair.contains(findMe)) {
168     
169                     // create a copy of this set to work on
170                     Set<Variable> pairSet = new HashSet<>(pair.getPair());
171     
172                     // we are only interested in the other member of the pair
173                     pairSet.remove(findMe);
174                     if (pairSet.size() != 1) {
175                         throw new RuntimeException("SNH");
176                     }
177     
178                     // create the adjusted copy and add it to the new set of pairs
179                     MayAliasPair newPair =
180                         new MayAliasPair(replacer, pairSet.iterator().next());
181                     newPairs.add(newPair);
182                 }
183             }
184     
185             this.pairs = newPairs;
186         }

188         public void replace(Map<Variable, Variable> replacements) {
189             for (MayAliasPair pair : this.pairs) {
190                 pair.replace(replacements);
191             }
192         }
