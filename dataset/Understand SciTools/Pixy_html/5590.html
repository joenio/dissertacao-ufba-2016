12     /**
13      * Based on Florian Martin's PhD thesis.
14      *
15      * Output:
16      * - for every function, the set EncodedCallStrings
17      * - for every call node, a connector function
18      *
19      * @author Nenad Jovanovic <enji@seclab.tuwien.ac.at>
20      */
21     public class ConnectorComputation {
22         // worklist
23         ConnectorWorkList workList;
24     
25         // TacFunction -> EncodedCallStrings;
26         // ECS stands for "encoded call strings"; ECS(p), where p is a procedure,
27         // is the set of all call strings that reach a procedure (i.e., these
28         // are all call-strings that are used in the header of this procedure's
29         // phi table);
30         // if a function has an empty EncodedCallStrings, it means that it is not called
31         // from anywhere (or that we are working with kSize == 0);
32         // the main procedure has only the empty call string
33         Map<TacFunction, EncodedCallStrings> function2ECS;
34     
35         // Call -> ConnectorFunction;
36         // a connector function maps the context of the caller to the target
37         // context in the callee;
38         // a call node must never have an empty connector function
39         Map<Call, ConnectorFunction> call2ConnectorFunction;
40     
41         // useful by-product of the computation;
42         // associates every function with a list of contained calls
43         Map<TacFunction, List<Call>> containedCalls;
44     
45         private CallGraph callGraph;
46     
47         private TacFunction mainFunction;
48     
49         // call-string length
50         int kSize;
51     
52         public ConnectorComputation(List<TacFunction> functions, TacFunction mainFunction, int kSize) {
53             this.kSize = kSize;
54             this.mainFunction = mainFunction;
55     
56             // start with empty EncodedCallStrings for each function;
57             // by the way, build list with call nodes
58             // and the call graph
59     
60             List<Call> callNodes = new LinkedList<>();
61             this.function2ECS = new HashMap<>();
62             this.containedCalls = new HashMap<>();
63     
64             for (TacFunction function : functions) {
65                 this.function2ECS.put(function, new EncodedCallStrings());
66     
67                 List<Call> calls = function.getContainedCalls();
68                 callNodes.addAll(calls);
69     
70                 this.containedCalls.put(function, calls);
71             }
72     
73             // initialize EncodedCallStrings for main function with empty call string
74             CallString emptyCallString = new CallString();
75             this.function2ECS.put(mainFunction, new EncodedCallStrings(emptyCallString));
76     
77             // initialize worklist
78             this.workList = new ConnectorWorkList();
79             this.workList.add(new ConnectorWorkListElement(mainFunction, emptyCallString));
80     
81             // initialize connector functions
82             this.call2ConnectorFunction = new HashMap<>();
83             for (Call callNode : callNodes) {
84                 this.call2ConnectorFunction.put(callNode, new ConnectorFunction());
85             }
86         }

89             return this.callGraph;

92         // computes the call graph
93         private void makeCallGraph() {
94     
95             this.callGraph = new CallGraph(this.mainFunction);
96     
97             // how we construct the call graph:
98             // - enter the calls contained in the main function into a queue
99             // - mark the main function as "visited"; if a function is marked
100             //   as visited, it means that its contained call nodes have already
101             //   been added to the queue
102             // - while there is something in the queue:
103             //   - remove the next call node from the queue
104             //   - draw the appropriate "caller -> callee" part into the call graph
105             //   - if the callee has not been visited yet:
106             //     - add the callee's contained call nodes to the queue
107             //     - mark the callee as visited
108             List<Call> processUs = this.containedCalls.get(this.mainFunction);
109             Set<TacFunction> visited = new HashSet<>();
110             visited.add(this.mainFunction);
111             while (!processUs.isEmpty()) {
112                 Call callNode = processUs.remove(0);
113                 TacFunction caller = callNode.getEnclosingFunction();
114                 TacFunction callee = callNode.getCallee();
115                 if (callee != null) {
116                     callGraph.add(caller, callee, callNode);
117                     if (!visited.contains(callee)) {
118                         processUs.addAll(this.containedCalls.get(callee));
119                         visited.add(callee);
120                     }
121                 }
122             }
123         }

125         public void compute() {
126     
127             while (this.workList.hasNext()) {
128     
129                 // extract element
130                 ConnectorWorkListElement element = this.workList.removeNext();
131                 TacFunction p = element.getFunction();
132                 CallString gamma = element.getCallString();
133     
134                 // get position of the current call string  in the current procedure
135                 EncodedCallStrings encodedCallStrings_p = this.function2ECS.get(p);
136                 int pos = encodedCallStrings_p.getPosition(gamma);
137                 if (pos == -1) {
138                     throw new RuntimeException("SNH");
139                 }
140     
141                 // for all calls in function p...
142                 for (Call callNode : this.containedCalls.get(p)) {
143                     TacFunction q = callNode.getCallee();
144                     if (q == null) {
145                         // callee is still unknown
146                         continue;
147                     }
148                     CallString gamma_2 = gamma.append(callNode, this.kSize);
149                     EncodedCallStrings encodedCallStrings_q = this.function2ECS.get(q);
150                     int pos_2 = encodedCallStrings_q.getPosition(gamma_2);
151                     if (pos_2 == -1) {
152     
153                         // create new "column"
154                         pos_2 = encodedCallStrings_q.append(gamma_2);
155     
156                         // expand worklist
157                         this.workList.add(new ConnectorWorkListElement(q, gamma_2));
158                     }
159     
160                     // expand connector function for this call
161                     ConnectorFunction conFunc = this.getConFunc(callNode);
162                     conFunc.add(pos, pos_2);
163                 }
164             }
165     
166             // generate the call graph
167             this.makeCallGraph();
168         }

170         public CallStringContext getTargetContext(Call callNode, int sourcePosition) {
171     
172             // retrieve connector function for the given call node
173             ConnectorFunction conFunc = this.getConFunc(callNode);
174     
175             if (conFunc == null) {
176                 throw new RuntimeException("SNH: " + callNode.getFunctionNamePlace());
177             }
178     
179             // query connector function and return result
180             return conFunc.apply(sourcePosition);
181         }

183         // returns a list of ReverseTarget objects;
184         // don't call this with exitedFunction == main
185         public List<ReverseTarget> getReverseTargets(TacFunction exitedFunction, int sourcePosition) {
186     
187             if (exitedFunction.isMain()) {
188                 throw new RuntimeException("SNH");
189             }
190     
191             List<ReverseTarget> reverseTargets = new LinkedList<>();
192     
193             // determine call nodes to which we have to return
194             if (this.kSize == 0) {
195     
196                 // in this case, we are performing a context-insensitive analysis
197                 // and have to return to all calls to this function;
198                 // note that context call strings are always the empty call string
199                 // here, so there is only one position for each function;
200                 // i.e., |EncodedCallStrings| = 1 for each function
201     
202                 // prepare one-element context set
203                 Set<AbstractContext> contextSet = new HashSet<>();
204                 contextSet.add(new CallStringContext(0));
205     
206                 // for each call to this function...
207                 Set<Call> callNodes = this.callGraph.getCallsTo(exitedFunction);
208                 for (Call callNode : callNodes) {
209                     reverseTargets.add(new ReverseTarget(callNode, contextSet));
210                 }
211             } else {
212     
213                 // get the call node at the end of the call string (given by
214                 // exit node and source position)
215                 EncodedCallStrings exitedEncodedCallStrings = this.function2ECS.get(exitedFunction);
216                 CallString exitedCallString = exitedEncodedCallStrings.getCallString(sourcePosition);
217                 Call returnToMe = exitedCallString.getLast();
218                 ConnectorFunction returnToMeCF = this.getConFunc(returnToMe);
219                 Set<CallStringContext> returnToMePositions = returnToMeCF.reverseApply(sourcePosition);
220     
221                 reverseTargets.add(new ReverseTarget(returnToMe, returnToMePositions));
222             }
223     
224             return reverseTargets;
225         }

228             return this.call2ConnectorFunction.get(callNode);

240         public void stats(boolean verbose) {
241             int sumPhiEntries = 0;
242             int sumCfgNodes = 0;
243             for (Map.Entry<TacFunction, EncodedCallStrings> entry : this.function2ECS.entrySet()) {
244                 TacFunction function = entry.getKey();
245                 EncodedCallStrings encodedCallStrings = entry.getValue();
246                 int cfgNodes = function.size();
247                 int phiEntries = (cfgNodes * encodedCallStrings.size());
248                 sumPhiEntries += phiEntries;
249                 sumCfgNodes += cfgNodes;
250                 if (verbose) {
251                     System.out.println("function " + function.getName() + ": "
252                         + cfgNodes + " cfg nodes, " + encodedCallStrings.size() + " contexts, => " +
253                         phiEntries + " phi entries");
254                 }
255             }
256             if (MyOptions.optionV) {
257                 System.out.println("Total phi entries: " + sumPhiEntries);
258                 System.out.println("Total cfg nodes: " + sumCfgNodes);
259             }
260         }

262         public String dump() {
263             StringBuilder b = new StringBuilder();
264             for (Map.Entry<TacFunction, EncodedCallStrings> entry : function2ECS.entrySet()) {
265                 TacFunction function = entry.getKey();
266                 EncodedCallStrings encodedCallStrings = entry.getValue();
267                 if (function.isMain()) {
268                     // main function is not of interest
269                     continue;
270                 }
271                 if (encodedCallStrings.isEmpty()) {
272                     // functions that are not called from anywhere are not of interest
273                     continue;
274                 }
275                 b.append(function.getName());
276                 b.append(" called by:\n");
277                 b.append(encodedCallStrings.dump());
278                 b.append("\n");
279             }
280             return b.toString();
281         }
