63         // returns the head of this cfg by walking backwards;
64         // only use this for linear cfgs (e.g., default param cfgs)
65         public static AbstractCfgNode getHead(AbstractCfgNode cfgNode) {
66             boolean goOn = true;
67     
68             while (goOn) {
69                 List<AbstractCfgNode> pre = cfgNode.getPredecessors();
70                 if (pre.size() == 0) {
71                     // found the head!
72                     goOn = false;
73                 } else if (pre.size() == 1) {
74                     // continue with predecessor
75                     cfgNode = pre.get(0);
76                 } else {
77                     // more than one predecessor...
78                     System.out.println(cfgNode.getLoc());
79                     throw new RuntimeException("SNH");
80                 }
81             }
82             return cfgNode;
83         }

89         public void assignReversePostOrder() {
90             LinkedList<AbstractCfgNode> postorder = this.dfPostOrder();
91             ListIterator<AbstractCfgNode> iter = postorder.listIterator(postorder.size());
92             int i = 0;
93             while (iter.hasPrevious()) {
94                 AbstractCfgNode cfgNode = iter.previous();
95                 cfgNode.setReversePostOrder(i);
96                 i++;
97             }
98         }

108         // helper function for size();
109         // recursively calculates the number of successor nodes
110         int size(AbstractCfgNode node, Set<AbstractCfgNode> visited) {
111             if (!visited.contains(node)) {
112                 visited.add(node);
113                 int size = 1;
114                 CfgEdge[] outEdges = node.getOutEdges();
115                 for (CfgEdge outEdge : outEdges) {
116                     if (outEdge != null) {
117                         size += size(outEdge.getDestination(), visited);
118                     }
119                 }
120                 return size;
121             } else {
122                 return 0;
123             }
124         }

126         // breadth first iterator;
127         // NOTE: when iterating over large CFGs, use should better use
128         // dfPreOrderIterator; breadthFirstIterator tends to produce stack overflows
129         public Iterator<AbstractCfgNode> bfIterator() {
130             // list for the iterator
131             LinkedList<AbstractCfgNode> list = new LinkedList<>();
132     
133             // queue for nodes that still have to be visited
134             LinkedList<AbstractCfgNode> queue = new LinkedList<>();
135     
136             Set<AbstractCfgNode> visited = new HashSet<>();
137     
138             queue.add(this.head);
139             visited.add(this.head);
140     
141             this.bfIteratorHelper(list, queue, visited);
142     
143             return list.iterator();
144         }
145     
146         private void bfIteratorHelper(
147             List<AbstractCfgNode> list, LinkedList<AbstractCfgNode> queue, Set<AbstractCfgNode> visited
148         ) {
149             AbstractCfgNode cfgNode = queue.removeFirst();
150             list.add(cfgNode);
151     
152             // handle successors
153             for (int i = 0; i < 2; i++) {
154                 CfgEdge outEdge = cfgNode.getOutEdge(i);
155                 if (outEdge != null) {
156                     AbstractCfgNode succ = outEdge.getDestination();
157                     // for all successors that have not been visited yet...
158                     if (!visited.contains(succ)) {
159                         // add it to the queue
160                         queue.add(succ);
161                         // mark it as visited
162                         visited.add(succ);
163                     }
164                 }
165             }
166     
167             // if the queue is non-empty: recurse
168             if (queue.size() > 0) {
169                 bfIteratorHelper(list, queue, visited);
170             }
171         }

189         // uses the given lists as containers for preorder and postorder
190         private void dfIterator(LinkedList<AbstractCfgNode> preorder, LinkedList<AbstractCfgNode> postorder) {
191             // auxiliary stack and visited set
192             LinkedList<AbstractCfgNode> stack = new LinkedList<>();
193             Set<AbstractCfgNode> visited = new HashSet<>();
194     
195             // visit head:
196             // mark it as visited, add it to stack and preorder
197             AbstractCfgNode current = this.head;
198             visited.add(current);
199             stack.add(current);
200             preorder.add(current);
201     
202             // how it works:
203             // while there is something on the stack:
204             // - mark the top stack element as visited
205             // - add it to the preorder list
206             // - try to get an unvisited successor of this element
207             // - if there is such a successor: push it on the stack and continue
208             // - else: pop the stack and add the popped element to the postorder list
209             while (!stack.isEmpty()) {
210                 // inspect the top stack element
211                 current = stack.getLast();
212     
213                 // we will try to get an unvisited successor element
214                 AbstractCfgNode next = null;
215                 for (int i = 0; (i < 2) && (next == null); i++) {
216                     CfgEdge outEdge = current.getOutEdge(i);
217                     if (outEdge != null) {
218                         next = outEdge.getDestination();
219                         if (visited.contains(next)) {
220                             // try another one
221                             next = null;
222                         } else {
223                             // found it!
224                         }
225                     }
226                 }
227     
228                 if (next == null) {
229                     // pop from stack and add it to the postorder list
230                     postorder.add(stack.removeLast());
231                 } else {
232                     // visit next
233                     visited.add(next);
234                     stack.add(next);
235                     preorder.add(next);
236                 }
237             }
238         }
