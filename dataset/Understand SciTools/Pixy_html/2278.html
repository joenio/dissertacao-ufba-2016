17     public class AssignUnary extends AbstractTransferFunction {

26     // CONSTRUCTORS ********************************************************************

29         // mustAliases, mayAliases: of setMe
30         public AssignUnary(
31             AbstractTacPlace left, AbstractTacPlace right, int op, Set<Variable> mustAliases, Set<Variable> mayAliases, AbstractCfgNode cfgNode
32         ) {
33             this.left = (Variable) left;  // must be a variable
34             this.right = right;
35             this.op = op;
36             this.mustAliases = mustAliases;
37             this.mayAliases = mayAliases;
38             this.cfgNode = cfgNode;
39         }
40     
41     // *********************************************************************************
42     // OTHER ***************************************************************************
43     // *********************************************************************************
44     
45         public AbstractLatticeElement transfer(AbstractLatticeElement inX) {
46     
47             DependencyLatticeElement in = (DependencyLatticeElement) inX;
48             DependencyLatticeElement out = new DependencyLatticeElement(in);
49     
50             // let the lattice element handle the details
51             out.assign(left, mustAliases, mayAliases, cfgNode);
52     
53             return out;
54         }
