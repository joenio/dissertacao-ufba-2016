32         public AbstractLatticeElement transfer(AbstractLatticeElement inX) {
33     
34             LiteralLatticeElement in = (LiteralLatticeElement) inX;
35             LiteralLatticeElement out = new LiteralLatticeElement(in);
36     
37             // SIMULATION OF BUILTIN FUNCTIONS
38             // ...by letting the php binary do the work
39     
40             String functionName = this.cfgNode.getFunctionName();
41     
42             if (MyOptions.phpBin == null) {
43     
44                 // don't simulate
45                 out.handleReturnValueBuiltin(this.cfgNode.getTempVar());
46             } else {
47                 switch (functionName) {
48                     case "realpath": {
49                         Literal resultLit = this.simulate(in, functionName);
50                         out.handleReturnValue(this.cfgNode.getTempVar(), resultLit);
51                         break;
52                     }
53                     case "dirname": {
54                         Literal resultLit = this.simulate(in, functionName);
55                         out.handleReturnValue(this.cfgNode.getTempVar(), resultLit);
56                         break;
57                     }
58                     default:
59                         // this function is not explicitly modelled
60                         out.handleReturnValueBuiltin(this.cfgNode.getTempVar());
61                         break;
62                 }
63             }
64     
65             return out;
66         }

68         private Literal simulate(LiteralLatticeElement in, String functionName) {
69     
70             // retrieve the necessary parameters
71             List<TacActualParameter> paramList = this.cfgNode.getParamList();
72             if (paramList.size() < 1) {
73                 // wrong number of params
74                 return Literal.TOP;
75             }
76             AbstractTacPlace param0 = paramList.get(0).getPlace();
77     
78             // check if we can resolve the necessary params
79             Literal lit0 = in.getLiteral(param0);
80             if (lit0 == Literal.TOP) {
81                 return Literal.TOP;
82             }
83     
84             Runtime runtime = Runtime.getRuntime();
85     
86             // assemble command.....
87     
88             StringBuilder command = new StringBuilder("<? ");
89     
90             // change working directory
91             command.append("chdir('");
92             command.append(MyOptions.entryFile.getParent());
93             command.append("');");
94     
95             // use "var_dump" to retrieve information about the output
96             command.append("var_dump(");
97     
98             // funcname plus opening bracket
99             command.append(functionName);
100             command.append("(");
101     
102             // params
103             command.append("'");
104             command.append(lit0);
105             command.append("'");
106     
107             // finish
108             command.append(")); ?>");
109     
110             Literal resultLit = null;
111             try {
112                 Process p = runtime.exec(MyOptions.phpBin);
113                 OutputStream outstream = p.getOutputStream();   // input for php
114                 InputStream instream = p.getInputStream();      // output of php
115     
116                 Writer outwriter = new OutputStreamWriter(outstream);
117                 outwriter.write(command.toString());
118                 outwriter.flush();
119                 outwriter.close();
120                 p.waitFor();
121     
122                 resultLit = this.parseOutput(instream);
123             } catch (IOException | InterruptedException e) {
124                 throw new RuntimeException(e.getMessage());
125             }
126     
127             return resultLit;
128         }

130         // parses the output returned by the php invocation into a literal
131         private Literal parseOutput(InputStream instream) {
132     
133             String resultString = null;
134             try {
135                 BufferedReader inreader = new BufferedReader(new InputStreamReader(instream));
136                 resultString = inreader.readLine();
137                 inreader.close();
138             } catch (IOException e) {
139                 throw new RuntimeException(e.getMessage());
140             }
141     
142             if (resultString.startsWith("bool(")) {
143                 if (resultString.equals("bool(true)")) {
144                     return Literal.TRUE;
145                 } else {
146                     return Literal.FALSE;
147                 }
148             } else if (resultString.startsWith("string(")) {
149                 return new Literal(resultString.substring(
150                     resultString.indexOf(')') + 3,
151                     resultString.length() - 1));
152             } else {
153                 return Literal.TOP;
154             }
155         }
