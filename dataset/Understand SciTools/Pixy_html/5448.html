74         private MyTransducer str_replace_transducer(String search, String replace)
75             throws Exception {
76     
77             if (search.length() == 0 || replace.length() == 0) {
78                 throw new RuntimeException("not yet");
79             }
80     
81             TransducerNivat t = new TransducerNivat();
82     
83             // the start state
84             State start = t.addState(true, true);
85             // the first state of the acceptor
86             State firstAccept = null;
87             // the "loopless trap" state
88             State trap = t.addState(false, true);
89     
90             // from start to first state of the acceptor
91             firstAccept = t.addState(false, false);
92             t.addTransition(new Transition(start, new TransducerRelation(search.charAt(0), null), firstAccept));
93     
94             State current = firstAccept;
95             List<Object> pushback = new LinkedList<>();
96             // for all but the first and the last state of the acceptor
97             for (int i = 1; i < search.length(); i++) {
98                 State next = t.addState(false, false);
99     
100                 pushback.add(search.charAt(i - 1));
101     
102                 // outgoing linear acceptor transition
103                 t.addTransition(new Transition(current, new TransducerRelation(search.charAt(i), null), next));
104     
105                 // trap transition
106                 makeCompositeTransition(t, pushback, null, current, trap);
107     
108                 // "back to first" transition
109                 makeCompositeTransition(t, pushback, search.charAt(0), current, firstAccept);
110     
111                 // "back to start" transition
112                 addPrefixedRemainingTransitions(t, alphabet, pushback, current, start);
113     
114                 current = next;
115             }
116     
117             // the last acceptor state
118             State lastAccept = current;
119     
120             // back to start with replacement
121             List<Object> replaceWithList = new LinkedList<>();
122             for (char c : replace.toCharArray()) {
123                 replaceWithList.add(c);
124             }
125             makeCompositeTransition(t, replaceWithList, null, lastAccept, start);
126     
127             // create loop transitions for start state
128             addRemainingTransitions(t, alphabet, start, start);
129     
130             return new MyTransducer(t);
131         }

133         public static void makeCompositeTransition(
134             TransducerNivat t, List<Object> composite, Object firstInLabel, State from, State to)
135             throws Exception {
136     
137             if (composite.size() == 0) {
138                 throw new RuntimeException("SNH");
139             }
140             if (composite.size() == 1) {
141                 Object compositeElement = composite.iterator().next();
142                 t.addTransition(new Transition(from, new TransducerRelation(firstInLabel, compositeElement), to));
143                 return;
144             }
145     
146             int i = 0;
147             State previous = from;
148             for (Object compositeElement : composite) {
149                 Object inLabel;
150                 if (i == 0) {
151                     inLabel = firstInLabel;
152                 } else {
153                     inLabel = null;
154                 }
155     
156                 State targetState;
157                 if (i == composite.size() - 1) {
158                     targetState = to;
159                 } else {
160                     targetState = t.addState(false, false);
161                 }
162     
163                 t.addTransition(new Transition(previous, new TransducerRelation(inLabel, compositeElement), targetState));
164                 previous = targetState;
165     
166                 i++;
167             }
168         }

170         public static void addPrefixedRemainingTransitions(
171             TransducerNivat t, MyAlphabet alphabet, List<Object> prefix, State from, State to
172         ) throws Exception {
173             @SuppressWarnings("unchecked") Set<Transition> fromTrans = t.delta(from);
174             Set<Object> existingInLabels = new HashSet<>();
175     
176             // collect existing in-labels from all transitions leaving "from"
177             for (Transition transition : fromTrans) {
178                 TransducerRelation relation = (TransducerRelation) transition.label();
179                 Object inLabel = relation.getIn();
180                 existingInLabels.add(inLabel);
181             }
182     
183             // for each label in the alphabet:
184             // add transition if this label was not found in the previous loop;
185             for (Character c : alphabet.getAlphabet()) {
186                 if (!existingInLabels.contains(c)) {
187                     List<Object> composite = new LinkedList<>(prefix);
188                     composite.add(c);
189                     makeCompositeTransition(t, composite, c, from, to);
190                 }
191             }
192         }

194         // adds id-transducer-transitions for all labels for which there is no
195         // transition yet
196         public static void addRemainingTransitions(
197             TransducerNivat t, MyAlphabet alphabet, State from, State to
198         ) throws Exception {
199     
200             @SuppressWarnings("unchecked") Set<Transition> fromTrans = t.delta(from);
201             Set<Object> existingInLabels = new HashSet<>();
202     
203             // collect existing in-labels from all transitions leaving "from"
204             for (Transition trans : fromTrans) {
205                 TransducerRelation rel = (TransducerRelation) trans.label();
206                 Object inLabel = rel.getIn();
207                 existingInLabels.add(inLabel);
208             }
209     
210             // for each label in the alphabet:
211             // add transition if this label was not found in the previous loop;
212             // use the same label as out-label
213             for (Character c : alphabet.getAlphabet()) {
214                 if (!existingInLabels.contains(c)) {
215                     t.addTransition(new Transition(from, new TransducerRelation(c, c), to));
216                 }
217             }
218         }
