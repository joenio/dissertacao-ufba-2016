10      * Transfer function for adding include dominators.
11      *

14     public class Add extends AbstractTransferFunction {
15         private AbstractCfgNode cfgNode;
16         private InclusionDominatorAnalysis inclusionDominatorAnalysis;
17     
18     // *********************************************************************************
19     // CONSTRUCTORS ********************************************************************
20     // *********************************************************************************
21     
22         public Add(AbstractCfgNode cfgNode, InclusionDominatorAnalysis inclusionDominatorAnalysis) {
23             this.cfgNode = cfgNode;
24             this.inclusionDominatorAnalysis = inclusionDominatorAnalysis;
25         }
26     
27     // *********************************************************************************
28     // OTHER ***************************************************************************
29     // *********************************************************************************
30     
31         public AbstractLatticeElement transfer(AbstractLatticeElement inX) {
32     
33             InclusionDominatorLatticeElement in = (InclusionDominatorLatticeElement) inX;
34             InclusionDominatorLatticeElement out = new InclusionDominatorLatticeElement(in);
35             out.add(this.cfgNode);
36     
37             // recycle
38             out = (InclusionDominatorLatticeElement) this.inclusionDominatorAnalysis.recycle(out);
39     
40             return out;
41         }
42     }
