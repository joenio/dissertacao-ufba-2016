16     public class Call extends AbstractCfgNode {

20         private TacFunction callee;
21     
22         // the return variable of the called function
23         private Variable retVar;
24         // temporary variable to hold the return value
25         private Variable tempVar;
26     
27         // a list of actual params (TacActualParameter objects)
28         private List<TacActualParameter> paramList;
29     
30         // list of cbr params; description see getCbrParams()
31         private List<List<Variable>> cbrParamList;
32     
33         // if this is a method call, this field CAN contain the name of the
34         // class that contains the callee method; is null if the name could
35         // not be resolved during tac conversion
36         private String calleeClassName;
37     
38         // object upon which this method was invoked;
39         // is null if
40         // - this is not a method invocation
41         // - or if it is a static one
42         // - of it it is a constructor invocation ("new")
43         private Variable object;

45         // if you pass "null" for "function", don't forget to call "setFunction" later
46         public Call(
47             AbstractTacPlace functionNamePlace, TacFunction calledFunction, ParseNode node,
48             TacFunction enclosingFunction, Variable retVar, AbstractTacPlace tempPlace,
49             List<TacActualParameter> paramList, Variable object
50         ) {
51     
52             super(node);
53             this.functionNamePlace = functionNamePlace;
54             //this.callee = calledFunction;
55             if (calledFunction != null) {
56                 calledFunction.addCalledFrom(this);
57             }
58             this.setEnclosingFunction(enclosingFunction);
59     
60             this.retVar = retVar;
61             this.tempVar = (Variable) tempPlace;    // must be a variable
62     
63             this.paramList = paramList;
64             this.cbrParamList = null;
65     
66             this.calleeClassName = null;
67             this.object = object;

78         public List<Variable> getVariables() {
79             // only the params are relevant for globals replacement
80             List<Variable> retMe = new LinkedList<>();
81             for (TacActualParameter param : this.paramList) {
82                 AbstractTacPlace paramPlace = param.getPlace();
83                 if (paramPlace instanceof Variable) {
84                     retMe.add((Variable) paramPlace);
85                 } else {
86                     retMe.add(null);

104         // returns a list consisting of two-element-lists consisting of
105         // (actual cbr-param, formal cbr-param) (Variable objects)
106         public List<List<Variable>> getCbrParams() {
107             if (this.cbrParamList != null) {
108                 return this.cbrParamList;
109             }
110     
111             List<TacActualParameter> actualParams = this.paramList;
112             List<TacFormalParameter> formalParams = this.getCallee().getParams();
113     
114             this.cbrParamList = new LinkedList<>();
115     
116             Iterator<TacActualParameter> actualIter = actualParams.iterator();
117             Iterator<TacFormalParameter> formalIter = formalParams.iterator();
118     
119             while (actualIter.hasNext()) {
120                 TacActualParameter actualParam = actualIter.next();
121                 TacFormalParameter formalParam = formalIter.next();
122     
123                 // if this is a cbr-param...
124                 if (actualParam.isReference() || formalParam.isReference()) {
125                     // the actual part of a cbr-param must always be a variable
126                     if (!(actualParam.getPlace() instanceof Variable)) {
127                         throw new RuntimeException("Error in the PHP file!");
128                     }
129     
130                     Variable actualVar = (Variable) actualParam.getPlace();
131                     Variable formalVar = formalParam.getVariable();
132     
133                     // check for unsupported features;
134                     // none of the variables must be an array or etc.;
135                     // in such a case, ignore it and continue with the next cbr-param
136                     boolean supported = AliasAnalysis.isSupported(
137                         formalVar, actualVar, true, this.getOriginalLineNumber());
138     
139                     if (!supported) {
140                         continue;
141                     }
142     
143                     List<Variable> pairList = new LinkedList<>();
144                     pairList.add(actualVar);
145                     pairList.add(formalVar);
146                     cbrParamList.add(pairList);
147                 }
148             }
149     
150             return cbrParamList;
151         }
