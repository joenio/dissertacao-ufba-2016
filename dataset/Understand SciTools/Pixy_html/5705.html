3     import at.ac.tuwien.infosys.www.phpparser.ParseNode;
4     import at.ac.tuwien.infosys.www.pixy.conversion.AbstractTacPlace;
5     import at.ac.tuwien.infosys.www.pixy.conversion.TacFunction;
6     import at.ac.tuwien.infosys.www.pixy.conversion.Variable;
7     
8     import java.io.File;
9     import java.util.LinkedList;
10     import java.util.List;
11     
12     /**
13      * "temp = include <place>" (can also be require or *_once).
14      *
15      * @author Nenad Jovanovic <enji@seclab.tuwien.ac.at>
16      */
17     public class Include extends AbstractCfgNode implements Comparable<Include> {
18         private Variable temp;
19         private AbstractTacPlace includeMe;
20         private File file;  // file in which this node occurs
21         private TacFunction includeFunction; // function in which this node occurs
22     
23     //  CONSTRUCTORS *******************************************************************
24     
25         public Include(AbstractTacPlace temp, AbstractTacPlace includeMe,
26                        File file, TacFunction includeFunction, ParseNode parseNode) {
27             super(parseNode);
28             this.temp = (Variable) temp;
29             this.includeMe = includeMe;
30             this.file = file;
31             this.includeFunction = includeFunction;
32         }
33     
34     //  GET ****************************************************************************
35     
36         public AbstractTacPlace getTemp() {
37             return this.temp;
38         }
39     
40         public AbstractTacPlace getIncludeMe() {
41             return this.includeMe;
42         }
43     
44         public File getFile() {
45             return this.file;
46         }
47     
48         public TacFunction getIncludeFunction() {
49             return this.includeFunction;
50         }
51     
52         // is the file to be included given by a simple literal?
53         public boolean isLiteral() {
54             return this.includeMe.isLiteral();
55         }
56     
57         public List<Variable> getVariables() {
58             List<Variable> retMe = new LinkedList<>();
59             retMe.add(this.temp);
60             if (this.includeMe instanceof Variable) {
61                 retMe.add((Variable) this.includeMe);
62             } else {
63                 retMe.add(null);
64             }
65             return retMe;
66         }
67     
68     //  SET ****************************************************************************
69     
70         // use this function if this include node is inside the main function and
71         // is inlined into some other file and function
72         public void setIncludeFunction(TacFunction function) {
73             this.includeFunction = function;
74         }
75     
76         public void replaceVariable(int index, Variable replacement) {
77             switch (index) {
78                 case 0:
79                     this.temp = replacement;
80                     break;
81                 case 1:
82                     this.includeMe = replacement;
83                     break;
84                 default:
85                     throw new RuntimeException("SNH");
86             }
87         }
88     
89         public int compareTo(Include o) {
90             if (o == this) {
91                 return 0;
92             }
93             Include comp = o;
94             int fileComp = this.file.compareTo(comp.file);
95             if (fileComp != 0) {
96                 return fileComp;
97             } else {
98                 return new Integer(this.getOriginalLineNumber()).compareTo(comp.getOriginalLineNumber());
99             }
100         }
101     }
