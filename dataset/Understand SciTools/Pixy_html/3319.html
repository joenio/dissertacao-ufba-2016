5     import java.util.LinkedList;
6     import java.util.SortedMap;
7     import java.util.TreeMap;

9     /**
10      * This work list contains two lists: one for cfg nodes that have been
11      * assigned a reverse pre-order, and one for nodes without such an order.
12      *
13      * If it contains nodes without order, these nodes are returned first
14      * (queued, FIFO). If there are no nodes without order, the ordered
15      * nodes are returned accordingly.
16      *
17      * @author Nenad Jovanovic <enji@seclab.tuwien.ac.at>
18      */
19     public final class InterproceduralWorklistPoor implements InterproceduralWorklist {
20         private LinkedList<InterproceduralWorklistElement> unsortedWorkList;
21         private SortedMap<Integer, InterproceduralWorklistElement> sortedWorkList;
22     
23     // *********************************************************************************
24     // CONSTRUCTORS ********************************************************************
25     // *********************************************************************************
26     
27         public InterproceduralWorklistPoor() {
28             this.unsortedWorkList = new LinkedList<>();
29             this.sortedWorkList = new TreeMap<>();
30         }
31     
32     // *********************************************************************************
33     // OTHER ***************************************************************************
34     // *********************************************************************************
35     
36         public void add(AbstractCfgNode cfgNode, AbstractContext context) {
37             // null contexts are not allowed
38             if (context == null) {
39                 throw new RuntimeException("SNH");
40             }
41             InterproceduralWorklistElement element = new InterproceduralWorklistElement(cfgNode, context);
42     
43             int rpo = cfgNode.getReversePostOrder();
44             if (rpo == -1) {
45                 this.unsortedWorkList.add(element);
46             } else {
47                 this.sortedWorkList.put(rpo, element);
48             }
49         }
50     
51         public InterproceduralWorklistElement removeNext() {
52             if (!this.unsortedWorkList.isEmpty()) {
53                 return this.unsortedWorkList.removeFirst();
54             } else {
55                 Integer key = this.sortedWorkList.firstKey();
56                 return this.sortedWorkList.remove(key);
57             }
58         }
59     
60         public boolean hasNext() {
61             return !(this.unsortedWorkList.isEmpty() && this.sortedWorkList.isEmpty());
62         }
63     }
