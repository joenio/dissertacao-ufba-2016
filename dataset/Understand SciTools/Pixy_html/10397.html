124         public void convert() {
125             convertEntryFile();
126     
127             List<Include> processUs = baseTac.getIncludeNodes();
128             boolean goOn = true;  // start a new iteration?
129     
130             // STATS
131             //
132             // include nodes that could not be resolved
133             Set<Include> topIncludes = new HashSet<>();
134             //
135             // literal includes for which the target file could not be found
136             SortedMap<Include, String> notFoundLiteralIncludes = new TreeMap<>();
137             //
138             // dynamic includes for which the target file could not be found
139             SortedMap<Include, String> notFoundDynamicIncludes = new TreeMap<>();
140             //
141             // number of included literal includes
142             int resolvedLit = 0;
143             //
144             // number of included non-literal includes;
145             int resolvedNonLit = 0;
146             //
147             // number of cyclic includes;
148             int cyclic = 0;
149             //
150             // number of "not found" includes
151             int notFound = 0;
152             //
153             // number of iterations
154             int iteration = 0;
155     
156             while (goOn && !MyOptions.optionW) {
157                 iteration++;
158     
159                 goOn = false;
160                 boolean nonLiteralIncludes = false; // are there non-literal includes?
161                 List<Include> weComeAfterwards;
162     
163                 // resolution of literal includes *****************************
164     
165                 System.out.println();
166                 System.out.println("*** resolving literal includes ***");
167                 System.out.println();
168     
169                 while (!processUs.isEmpty()) {
170                     // auxiliary list: will be filled with the include nodes
171                     // contained in included files
172                     weComeAfterwards = new LinkedList<>();
173     
174                     // process all literal include nodes in "processUs"
175                     for (Include includeNode : processUs) {
176                         if (this.skipUs.contains(includeNode)) {
177                             continue;
178                         }
179                         if (!includeNode.isLiteral()) {
180                             nonLiteralIncludes = true;
181                             continue;
182                         }
183                         InclusionStatus status = this.include(includeNode.getIncludeMe().toString(), includeNode,
184                             includeNode.getIncludeFunction(), weComeAfterwards);
185     
186                         switch (status) {
187                             case INCLUDED:
188                                 // fine!
189                                 resolvedLit++;
190                                 break;
191                             case NOT_FOUND:
192                                 // a literal include that was not found;
193                                 // there is no need to retry
194                                 this.skipUs.add(includeNode);
195                                 notFound++;
196                                 notFoundLiteralIncludes.put(includeNode, includeNode.getIncludeMe().toString());
197                                 break;
198                             case CYCLIC:
199                                 this.skipUs.add(includeNode);
200                                 cyclic++;
201                                 break;
202                             default:
203                                 throw new RuntimeException("SNH");
204                         }
205                     }
206     
207                     processUs = weComeAfterwards;
208                 }
209                 System.out.println();
210     
211                 // assign functions to cfg nodes
212                 this.baseTac.assignFunctionsToControlFlowGraphNodes();
213     
214                 // resolution of non-literal includes **********************************
215     
216                 // if there are no non-literal includes, we don't have anything
217                 // to resolve
218                 if (!nonLiteralIncludes) {
219                     break;
220                 }
221     
222                 System.out.println();
223                 System.out.println("*** resolving non-literal includes ***");
224                 System.out.println();
225     
226                 // requires literal analysis and hence, preliminary backpatching
227                 this.baseTac.backpatch();
228     
229                 int kSize = 1;
230                 ConnectorComputation connectorComp = new ConnectorComputation(
231                     baseTac.getAllFunctions(), baseTac.getMainFunction(), kSize);
232                 connectorComp.compute();
233                 InterproceduralWorklist workList = new InterproceduralWorklistBetter(new InterproceduralWorklistOrder(baseTac, connectorComp));
234                 connectorComp.stats(false);
235     
236                 AliasAnalysis aliasAnalysis = new DummyAliasAnalysis();
237     
238                 literalAnalysis = new LiteralAnalysis(
239                     baseTac, aliasAnalysis, new CallStringAnalysis(connectorComp), workList);
240                 literalAnalysis.analyze();
241     
242                 processUs = literalAnalysis.getIncludeNodes();
243                 weComeAfterwards = new LinkedList<>();
244                 notFoundDynamicIncludes = new TreeMap<>();
245                 topIncludes = new HashSet<>();
246     
247                 for (Include includeNode : processUs) {
248                     if (this.skipUs.contains(includeNode)) {
249                         continue;
250                     }
251     
252                     Literal includedLit = literalAnalysis.getLiteral(includeNode.getIncludeMe(), includeNode);
253                     String includedString = null;
254     
255                     if (includedLit == Literal.TOP) {
256     
257                         // try heuristics
258                         List<String> includeTargets = ParseNodeHeuristics.getPossibleIncludeTargets(
259                             includeNode, literalAnalysis, notFoundDynamicIncludes,
260                             this.workingDirectoryFile.getPath());
261     
262                         if (includeTargets == null) {
263                             // more than one possibility
264                             topIncludes.add(includeNode);
265                             continue;
266                         } else if (includeTargets.isEmpty()) {
267                             continue;
268                         } else if (includeTargets.size() == 1) {
269                             // heuristics were successful!
270                             includedString = includeTargets.get(0);
271                         } else {
272                             throw new RuntimeException("SNH");
273                         }
274                     } else {
275                         includedString = includedLit.toString();
276                     }
277     
278                     // include!
279                     InclusionStatus status = this.include(includedString, includeNode, includeNode.getIncludeFunction(),
280                         weComeAfterwards);
281     
282                     switch (status) {
283                         case INCLUDED:
284                             // we have included something, so continue with the next iteration
285                             goOn = true;
286                             resolvedNonLit++;
287                             break;
288                         case CYCLIC:
289                             this.skipUs.add(includeNode);
290                             cyclic++;
291                             break;
292                         case NOT_FOUND:
293                             // a non-literal include that was not found:
294                             // perhaps we will succeed in a later iteration...
295                             notFoundDynamicIncludes.put(includeNode, includedString);
296                             break;
297                         default:
298                             throw new RuntimeException("SNH");
299                     }
300                 }
301     
302                 System.out.println();
303     
304                 // assign functions to cfg nodes
305                 this.baseTac.assignFunctionsToControlFlowGraphNodes();
306     
307                 processUs = weComeAfterwards;
308                 processUs.addAll(topIncludes);  // maybe they will become resolvable in the next iteration
309                 processUs.addAll(notFoundDynamicIncludes.keySet()); // _,,_
310             }
311     
312             // don't generate warnings for unreachables
313             this.removeUnreachables(topIncludes, notFoundDynamicIncludes);
314     
315             // not needed any more ( => save memory)
316             this.literalAnalysis = null;
317     
318             if (this.useAliasAnalysis) {
319                 // final, verbose backpatching
320                 this.baseTac.backpatch(true, true, null, null);
321     
322                 // shadow generation
323                 this.baseTac.generateShadows();
324             } else {
325                 // approximate effects of "global" statements
326                 this.baseTac.replaceGlobals();
327     
328                 // backpatch with methods so that type analysis can do its work
329                 this.baseTac.backpatch(true, false, null, null);
330     
331                 // if there were ambiguous method calls:
332                 // try to resolve them by means of a type analysis;
333                 // EFF: perform type analysis conditionally; currently, type analysis
334                 // is outside of the condition (for testing purposes)
335     
336                 System.out.println();
337                 System.out.println("*** performing type analysis ***");
338                 System.out.println();
339     
340                 ConnectorComputation connectorComp = new ConnectorComputation(
341                     baseTac.getAllFunctions(), baseTac.getMainFunction(), 0);
342                 connectorComp.compute();
343                 InterproceduralWorklist workList = new InterproceduralWorklistBetter(new InterproceduralWorklistOrder(baseTac, connectorComp));
344                 this.typeAnalysis = new TypeAnalysis(
345                     this.baseTac, new CallStringAnalysis(connectorComp), workList);
346                 typeAnalysis.analyze();
347     
348                 // final, verbose backpatching
349                 this.baseTac.backpatch(true, true, typeAnalysis, connectorComp.getCallGraph());
350             }
351     
352             // summarize into basic blocks;
353             // leads to a reduced amount of used memory
354             if (MyOptions.optionV) {
355                 System.out.println("creating basic blocks");
356             }
357             this.baseTac.createBasicBlocks();
358     
359             // assign functions to cfg nodes
360             this.baseTac.assignFunctionsToControlFlowGraphNodes();
361     
362             if (this.countLines) {
363                 System.out.println("Lines: " + this.numberOfLines);
364             }
365     
366             // statistics output
367             if (!MyOptions.optionW) {
368                 System.out.println();
369                 System.out.println("inclusion iterations:            " + iteration);
370                 System.out.println("resolved literal includes:       " + resolvedLit);
371                 System.out.println("resolved non-literal includes:   " + resolvedNonLit);
372                 System.out.println("cyclic includes:                 " + cyclic);
373                 System.out.println("not found includes:              " +
374                     (notFoundLiteralIncludes.size() + notFoundDynamicIncludes.size()));
375                 for (Map.Entry<Include, String> entry : notFoundLiteralIncludes.entrySet()) {
376                     System.out.println("- " + entry.getKey().getLoc());
377                     System.out.println("   [" + entry.getValue() + "]");
378                 }
379                 for (Map.Entry<Include, String> entry : notFoundDynamicIncludes.entrySet()) {
380                     System.out.println("- " + entry.getKey().getLoc());
381                     System.out.println("   [" + entry.getValue() + "]");
382                 }
383                 System.out.println("unresolved non-literal includes: " + topIncludes.size());
384                 List<Include> topIncludesList = new LinkedList<>(topIncludes);
385                 Collections.sort(topIncludesList);
386                 for (Include includeNode : topIncludesList) {
387                     System.out.println("- " + includeNode.getLoc());
388                 }
389                 System.out.println();
390     
391                 // dump include relationships
392                 Utils.writeToFile(this.includeGraph.dump(),
393                     MyOptions.graphPath + "/includes_" + MyOptions.entryFile.getName() + ".txt");
394             }
395     
396             // we don't need these any more
397             this.literalAnalysis = null;
398             this.includeGraph = null;
399             this.skipUs = null;
400     
401             this.baseTac.addSuperGlobalElements();
402     
403             if (!MyOptions.optionB && !MyOptions.optionW) {
404                 this.baseTac.stats();
405             }
406             System.out.println();
407     
408             // final node order
409             // EFF: only needed for functional analyses
410             baseTac.assignReversePostOrder();
411         }

426         // - input: a set of unresolved or not found includes
427         // - output: the same set, but without those that are inside functions
428         //   that are never called
429         private void removeUnreachables(Set<Include> includeSet, Map<Include, String> includeMap) {
430             for (Iterator<Include> iter = includeSet.iterator(); iter.hasNext(); ) {
431                 Include includeNode = iter.next();
432                 AbstractInterproceduralAnalysisNode incAnNode = literalAnalysis.getAnalysisNode(includeNode);
433                 if (incAnNode.getPhi().isEmpty()) {
434                     iter.remove();
435                 }
436             }
437     
438             for (Iterator<Map.Entry<Include, String>> iter = includeMap.entrySet().iterator(); iter.hasNext(); ) {
439                 Map.Entry<Include, String> entry = iter.next();
440                 Include includeNode = entry.getKey();
441                 AbstractInterproceduralAnalysisNode incAnNode = literalAnalysis.getAnalysisNode(includeNode);
442                 if (incAnNode.getPhi().isEmpty()) {
443                     iter.remove();
444                 }
445             }
446         }

448         public ParseTree parse(String fileName) {
449             String canonicalFileName = createUniqueFileName(fileName);
450     
451             ParseTree parseTree = null;
452             try {
453                 PhpLexer lexer = new PhpLexer(new FileReader(canonicalFileName));
454                 lexer.setFileName(canonicalFileName);
455                 PhpParser parser = new PhpParser(lexer);
456                 ParseNode rootNode = (ParseNode) parser.parse().value;
457                 parseTree = new ParseTree(rootNode);
458             } catch (FileNotFoundException e) {
459                 Utils.bail("File not found: " + canonicalFileName);
460             } catch (Exception e) {
461                 if (!MyOptions.optionW) {
462                     Utils.bail("Error parsing " + canonicalFileName);
463                 } else {
464                     Utils.bail();
465                 }
466             }
467     
468             if (this.countLines) {
469                 this.numberOfLines += this.countLines(canonicalFileName);
470             }
471     
472             return parseTree;
473         }

484         // turns the name of a file to be included into a File object;
485         // returns null if there is no such file
486         private File makeFile(String fileName, File includingFile) {
487             // first check whether the given file name is absolute
488             File findMe = new File(fileName);
489             if (findMe.isAbsolute()) {
490                 if (findMe.isFile()) {
491                     // found it!
492                     return findMe;
493                 }
494             }
495     
496             // try to include the file relative to working directory first:
497             // for each specified include path, append it to the working directory
498             // and search there for the file that is to be included
499             for (File includePath : MyOptions.includePaths) {
500                 File searchIn;  // directory to search in
501                 if (includePath.isAbsolute()) {
502                     searchIn = includePath;
503                 } else {
504                     searchIn = new File(this.workingDirectoryFile, includePath.getPath());
505                 }
506     
507                 findMe = new File(searchIn, fileName);
508                 if (findMe.isFile()) {
509                     // found it!
510                     return findMe;
511                 }
512             }
513     
514             // next: inclusion relative to the directory of the script file
515             // that contains the include node (includingFile);
516             // but only if the given file name doesn't start with "." or ".."
517     
518             if (!(fileName.startsWith("./") || fileName.startsWith("../"))) {
519                 for (File includePath : MyOptions.includePaths) {
520                     // we've already dealt with absolute include paths in the previous loop
521                     if (includePath.isAbsolute()) {
522                         continue;
523                     }
524     
525                     try {
526                         File searchIn = new File(
527                             includingFile.getCanonicalFile().getParentFile(),
528                             includePath.getPath());
529                         findMe = new File(searchIn, fileName);
530                         if (findMe.isFile()) {
531                             // found it!
532                             return findMe;
533                         }
534                     } catch (IOException e) {
535                         throw new RuntimeException(e.getMessage());
536                     }
537                 }
538             }
539     
540             // file not found
541             return null;
542         }

546         // function: the one that contains the given include node;
547         public InclusionStatus include(String includedFileName, Include includeNode, TacFunction function,
548                                  List<Include> includeNodes) {
549             File includingFile = includeNode.getFile();
550             File includedFile = this.makeFile(includedFileName, includingFile);
551             if (includedFile == null) {
552                 return InclusionStatus.NOT_FOUND;
553             }
554             try {
555                 this.allFiles.add(includedFile.getCanonicalFile());
556             } catch (IOException e) {
557                 throw new RuntimeException(e.getMessage());
558             }
559     
560             // approximation for handling recursions:
561             // if this is a recursive include (include graph would get cyclic), ignore it
562             boolean acyclic = this.includeGraph.addAcyclicEdge(includeNode.getFile(), includedFile);
563     
564             // get canonical path of included file
565             String includedFilePath = null;
566             try {
567                 includedFilePath = includedFile.getCanonicalPath();
568             } catch (IOException e) {
569                 throw new RuntimeException(e.getMessage());
570             }
571     
572             if (acyclic) {
573                 if (!MyOptions.optionB) {
574                     System.out.print(".");
575                 }
576                 ParseTree parseTree = this.parse(includedFilePath);
577                 TacConverter tac = new TacConverter(
578                     parseTree, this.specialNodes, this.numberOfConvertedFiles++, includedFile, this);
579                 tac.convert();
580                 this.baseTac.include(tac, includeNode, function);
581                 includeNodes.addAll(tac.getIncludeNodes());
582                 return InclusionStatus.INCLUDED;
583             } else {
584                 return InclusionStatus.CYCLIC;
585             }
586         }

590         private int countLines(String fileName) {
591             int lines = 0;
592             try {
593                 File countMe = new File(fileName);
594                 BufferedReader reader = new BufferedReader(new FileReader(countMe));
595                 while (reader.readLine() != null) {
596                     lines++;
597                 }
598             } catch (IOException e) {
599                 throw new RuntimeException(e.getMessage());
600             }
601             return lines;
602         }
