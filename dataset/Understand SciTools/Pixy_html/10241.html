202         /**
203          * Returns the automaton representation of the given dependency graph.
204          *
205          * This is done by decorating the nodes of the graph with automata bottom-up, and returning the automaton that
206          * eventually decorates the root.
207          *
208          * Beware: this also eliminates cycles!
209          *
210          * @param dependencyGraph
211          * @param origDependencyGraph
212          * @param deco
213          *
214          * @return
215          */
216         protected FSAAutomaton toAutomatonSanit(
217             DependencyGraph dependencyGraph, DependencyGraph origDependencyGraph, Map<AbstractNode, FSAAutomaton> deco
218         ) {
219             dependencyGraph.eliminateCycles();
220             AbstractNode root = dependencyGraph.getRootNode();
221             Set<AbstractNode> visited = new HashSet<>();
222             this.decorateSanit(root, deco, visited, dependencyGraph, origDependencyGraph, true);
223             FSAAutomaton rootDeco = deco.get(root).clone();
224     
225             return rootDeco;
226         }

228         /**
229          * Decorates the given node (and all its successors) with an automaton.
230          *
231          * @param node
232          * @param deco
233          * @param visited
234          * @param dependencyGraph
235          * @param origDependencyGraph
236          * @param trimAllowed
237          */
238         private void decorateSanit(
239             AbstractNode node, Map<AbstractNode, FSAAutomaton> deco, Set<AbstractNode> visited,
240             DependencyGraph dependencyGraph, DependencyGraph origDependencyGraph, boolean trimAllowed
241         ) {
242             visited.add(node);
243     
244             TrimInfo trimInfo;
245             if (trimAllowed) {
246                 trimInfo = this.checkTrim(node);
247             } else {
248                 trimInfo = new TrimInfo();
249                 trimInfo.setDefaultTrim(false);
250             }
251     
252             // if this node has successors, decorate them first (if not done yet)
253             List<AbstractNode> successors = dependencyGraph.getSuccessors(node);
254             if (successors != null && !successors.isEmpty()) {
255                 int i = 0;
256                 for (AbstractNode succ : successors) {
257                     if (!visited.contains(succ) && deco.get(succ) == null) {
258                         decorateSanit(succ, deco, visited, dependencyGraph, origDependencyGraph, trimInfo.mayTrim(i));
259                     }
260                     i++;
261                 }
262             }
263     
264             // now that all successors are decorated, we can decorate this node
265             FSAAutomaton auto = null;
266             if (node instanceof NormalNode) {
267                 NormalNode normalNode = (NormalNode) node;
268                 if (successors == null || successors.isEmpty()) {
269                     // this should be a string leaf node
270                     AbstractTacPlace place = normalNode.getPlace();
271                     if (place.isLiteral()) {
272                         if (trimUntainted && trimAllowed) {
273                             auto = FSAAutomaton.makeString("");
274                         } else {
275                             auto = FSAAutomaton.makeString(place.toString());
276                         }
277                     } else {
278                         // this case should not happen any longer (now that
279                         // we have "uninit" nodes, see below)
280                         throw new RuntimeException("SNH: " + place + ", " + normalNode.getCfgNode().getFileName() + "," +
281                             normalNode.getCfgNode().getOriginalLineNumber());
282                     }
283                 } else {
284                     // this is an interior node, not a leaf node;
285                     // the automaton for this node is the union of all the
286                     // successor automatas
287                     for (AbstractNode succ : successors) {
288                         if (succ == node) {
289                             // a simple loop, can be ignored
290                             continue;
291                         }
292                         FSAAutomaton succAuto = deco.get(succ);
293                         if (succAuto == null) {
294                             throw new RuntimeException("SNH");
295                         }
296                         if (auto == null) {
297                             auto = succAuto;  // cloning not necessary here
298                         } else {
299                             auto = auto.union(succAuto);
300                         }
301                     }
302                 }
303             } else if (node instanceof BuiltinFunctionNode) {
304                 auto = this.makeAutoForOp((BuiltinFunctionNode) node, deco, dependencyGraph, trimAllowed);
305             } else if (node instanceof CompleteGraphNode) {
306                 // for SCC nodes, we generate a coarse string approximation (.* automaton);
307                 // the taint value depends on the taint value of the successors:
308                 // if any of the successors is tainted in any way, we make the resulting
309                 // automaton tainted as well
310     
311                 /*
312                  * this approach works under the assumption that the SCC contains
313                  * no "evil" functions (functions that always return a tainted value),
314                  * and that the direct successors of the SCC are not <uninit> nodes
315                  * (see there, below);
316                  * it is primarily based on the fact that SCCs can never contain
317                  * <uninit> nodes, because <uninit> nodes are always leaf nodes in the dependency
318                  * graph (since they have no successors);
319                  * under the above assumptions and observations, it is valid to
320                  * say that the taint value of an SCC node solely depends on
321                  * the taint values of its successors, and that is exactly what we
322                  * do here
323                  *
324                  */
325                 if (trimUntainted && trimAllowed) {
326                     auto = FSAAutomaton.makeString("");
327     
328                     for (AbstractNode succ : successors) {
329                         if (succ == node) {
330                             // a simple loop, should be part of the SCC
331                             throw new RuntimeException("SNH");
332                         }
333                         FSAAutomaton succAuto = deco.get(succ);
334                         if (succAuto == null) {
335                             throw new RuntimeException("SNH");
336                         }
337                         if (succAuto.isEmpty()) {
338                             auto = FSAAutomaton.makeAnyString();
339                             break;
340                         }
341                     }
342                 } else {
343                     auto = FSAAutomaton.makeAnyString();
344                 }
345             } else if (node instanceof UninitializedNode) {
346                 // retrieve predecessor
347                 Set<AbstractNode> predecessors = dependencyGraph.getPredecessors(node);
348                 if (predecessors.size() != 1) {
349                     throw new RuntimeException("SNH");
350                 }
351                 AbstractNode pre = predecessors.iterator().next();
352     
353                 if (pre instanceof NormalNode) {
354                     NormalNode preNormal = (NormalNode) pre;
355                     switch (this.getInitialTaintForPlace(preNormal.getPlace())) {
356                         case ALWAYS:
357                         case IF_REGISTER_GLOBALS:
358                             auto = FSAAutomaton.makeAnyString();
359                             break;
360                         case NEVER:
361                             if (trimUntainted && trimAllowed) {
362                                 auto = FSAAutomaton.makeString("");
363                             } else {
364                                 auto = FSAAutomaton.makeAnyString();
365                             }
366                             break;
367                         default:
368                             throw new RuntimeException("SNH");
369                     }
370                 } else if (pre instanceof CompleteGraphNode) {
371                     // this case can really happen (e.g.: dcpportal: advertiser.php, forums.php);
372     
373                     // take a look at the "real" predecessors (i.e., take a look "into"
374                     // the SCC node): if there is exactly one predecessor, namely a
375                     // NormalNode, and if the contained place is initially untainted,
376                     // there is no danger from here; else: we will have to set it to tainted
377                     Set<AbstractNode> origPreds = origDependencyGraph.getPredecessors(node);
378                     if (origPreds.size() == 1) {
379                         AbstractNode origPre = origPreds.iterator().next();
380                         if (origPre instanceof NormalNode) {
381                             NormalNode origPreNormal = (NormalNode) origPre;
382     
383                             switch (this.getInitialTaintForPlace(origPreNormal.getPlace())) {
384                                 case ALWAYS:
385                                 case IF_REGISTER_GLOBALS:
386                                     auto = FSAAutomaton.makeAnyString();
387                                     break;
388                                 case NEVER:
389                                     if (trimUntainted && trimAllowed) {
390                                         auto = FSAAutomaton.makeString("");
391                                     } else {
392                                         auto = FSAAutomaton.makeAnyString();
393                                     }
394                                     break;
395                                 default:
396                                     throw new RuntimeException("SNH");
397                             }
398                         } else {
399                             auto = FSAAutomaton.makeAnyString();
400                         }
401                     } else {
402                         // conservative decision for this SCC
403                         auto = FSAAutomaton.makeAnyString();
404                     }
405                 } else {
406                     throw new RuntimeException("SNH: " + pre.getClass());
407                 }
408             } else {
409                 throw new RuntimeException("SNH");
410             }
411     
412             if (auto == null) {
413                 throw new RuntimeException("SNH");
414             }
415     
416             deco.put(node, auto);
417         }

419         /**
420          * Returns an automaton for the given operation node.
421          *
422          * @param node
423          * @param deco
424          * @param dependencyGraph
425          * @param trimAllowed
426          *
427          * @return
428          */
429         private FSAAutomaton makeAutoForOp(
430             BuiltinFunctionNode node, Map<AbstractNode, FSAAutomaton> deco, DependencyGraph dependencyGraph,
431             boolean trimAllowed
432         ) {
433             List<AbstractNode> successors = dependencyGraph.getSuccessors(node);
434             if (successors == null) {
435                 successors = new LinkedList<>();
436             }
437     
438             FSAAutomaton automaton = null;
439     
440             String opName = node.getName();
441     
442             List<Integer> multiList = new LinkedList<>();
443     
444             if (!node.isBuiltin()) {
445                 // call to function or method for which no definition
446                 // could be found
447                 AbstractCfgNode cfgNodeX = node.getCfgNode();
448                 if (cfgNodeX instanceof CallUnknownFunction) {
449                     CallUnknownFunction cfgNode = (CallUnknownFunction) cfgNodeX;
450                     if (cfgNode.isMethod()) {
451                         if (trimUntainted && trimAllowed) {
452                             automaton = FSAAutomaton.makeString("");
453                         } else {
454                             automaton = FSAAutomaton.makeAnyString();
455                         }
456                     } else {
457                         automaton = FSAAutomaton.makeAnyString();
458                     }
459                 } else {
460                     throw new RuntimeException("SNH");
461                 }
462             } else if (opName.equals(".")) {
463                 // CONCAT
464                 for (AbstractNode succ : successors) {
465                     FSAAutomaton succAuto = deco.get(succ);
466                     if (automaton == null) {
467                         automaton = succAuto;
468                     } else {
469                         automaton = automaton.concatenate(succAuto);
470                     }
471                 }
472     
473                 // TRANSDUCIBLES ****************************************
474             } else if (opName.equals("preg_replace")) {
475                 if (successors.size() < 3) {
476                     throw new RuntimeException("SNH");
477                 }
478                 FSAAutomaton searchAuto = deco.get(successors.get(0));
479                 FSAAutomaton replaceAuto = deco.get(successors.get(1));
480                 FSAAutomaton subjectAuto = deco.get(successors.get(2));
481     
482                 // if the replacement is evil, be conservative
483                 if (trimUntainted && !replaceAuto.isEmpty()) {
484                     return FSAAutomaton.makeAnyString();
485                 }
486     
487                 FSAAutomaton transduced = FSAUtils.reg_replace(searchAuto, replaceAuto,
488                     subjectAuto, true, node.getCfgNode());
489                 return transduced;
490             } else if (opName.equals("ereg_replace")) {
491                 if (successors.size() < 3) {
492                     throw new RuntimeException("SNH");
493                 }
494                 FSAAutomaton searchAuto = deco.get(successors.get(0));
495                 FSAAutomaton replaceAuto = deco.get(successors.get(1));
496                 FSAAutomaton subjectAuto = deco.get(successors.get(2));
497     
498                 // if the replacement is evil, be conservative
499                 if (trimUntainted && !replaceAuto.isEmpty()) {
500                     return FSAAutomaton.makeAnyString();
501                 }
502     
503                 FSAAutomaton transduced = FSAUtils.reg_replace(searchAuto, replaceAuto,
504                     subjectAuto, false, node.getCfgNode());
505                 return transduced;
506             } else if (opName.equals("str_replace")) {
507                 if (successors.size() < 3) {
508                     throw new RuntimeException("SNH");
509                 }
510                 FSAAutomaton searchAuto = deco.get(successors.get(0));
511                 FSAAutomaton replaceAuto = deco.get(successors.get(1));
512                 FSAAutomaton subjectAuto = deco.get(successors.get(2));
513     
514                 // if the replacement is evil, be conservative
515                 if (trimUntainted && !replaceAuto.isEmpty()) {
516                     return FSAAutomaton.makeAnyString();
517                 }
518     
519                 FSAAutomaton transduced = FSAUtils.str_replace(
520                     searchAuto, replaceAuto, subjectAuto, node.getCfgNode());
521                 return transduced;
522             } else if (opName.equals("addslashes")) {
523                 if (successors.size() != 1) {
524                     throw new RuntimeException("SNH");
525                 }
526                 FSAAutomaton paramAuto = deco.get(successors.get(0));
527     
528                 FSAAutomaton transduced = FSAUtils.addslashes(
529                     paramAuto, node.getCfgNode());
530                 return transduced;
531     
532                 // WEAK SANITIZATION FUNCTIONS *******************************
533                 // ops that perform sanitization, but which are insufficient
534                 // in cases where the output is not enclosed by quotes in an SQL query
535             } else if (isWeakSanitation(opName, multiList)) {
536                 if (trimUntainted && trimAllowed) {
537                     automaton = FSAAutomaton.makeString("");
538                 } else {
539                     automaton = FSAAutomaton.makeAnyString();
540                 }
541     
542                 // STRONG SANITIZATION FUNCTIONS *******************************
543                 // e.g., ops that return numeric values
544             } else if (isStrongSanitation(opName)) {
545                 if (trimUntainted && trimAllowed) {
546                     automaton = FSAAutomaton.makeString("");
547                 } else {
548                     automaton = FSAAutomaton.makeAnyString();
549                 }
550     
551                 // EVIL FUNCTIONS ***************************************
552                 // take care: if you define evil functions, you must adjust
553                 // the treatment of SCC nodes in decorate()
554     
555                 // MULTI-OR-DEPENDENCY **********************************
556             } else if (isMultiDependencyOperation(opName, multiList)) {
557                 automaton = this.multiDependencyAutoSanit(successors, deco, multiList, false);
558             } else if (isInverseMultiDependencyOperation(opName, multiList)) {
559                 automaton = this.multiDependencyAutoSanit(successors, deco, multiList, true);
560     
561                 // CATCH-ALL ********************************************
562             } else {
563                 System.out.println("Unmodeled builtin function (SQL-Sanit): " + opName);
564     
565                 // conservative decision for operations that have not been
566                 // modeled yet: .*
567                 automaton = FSAAutomaton.makeAnyString();
568             }
569     
570             return automaton;
571         }

573         /**
574          * If trimUntainted == false: always returns .*
575          * else:
576          * - if all successors are empty: returns empty
577          * - else: returns .*
578          *
579          * @param successors
580          * @param deco
581          * @param indices
582          * @param inverse
583          *
584          * @return
585          */
586         private FSAAutomaton multiDependencyAutoSanit(
587             List<AbstractNode> successors, Map<AbstractNode, FSAAutomaton> deco, List<Integer> indices, boolean inverse
588         ) {
589             if (!trimUntainted) {
590                 return FSAAutomaton.makeAnyString();
591             }
592     
593             Set<Integer> indexSet = new HashSet<>(indices);
594     
595             int count = -1;
596             for (AbstractNode succ : successors) {
597                 count++;
598     
599                 // check if there is a dependency on this successor
600                 if (inverse) {
601                     if (indexSet.contains(count)) {
602                         continue;
603                     }
604                 } else {
605                     if (!indexSet.contains(count)) {
606                         continue;
607                     }
608                 }
609     
610                 FSAAutomaton succAuto = deco.get(succ);
611                 if (succAuto == null) {
612                     throw new RuntimeException("SNH");
613                 }
614                 if (!succAuto.isEmpty()) {
615                     return FSAAutomaton.makeAnyString();
616                 }
617             }
618     
619             return FSAAutomaton.makeString("");
620         }

631         /**
632          * Checks if the given node is a custom sanitization node.
633          *
634          * @param node
635          *
636          * @return
637          */
638         public static boolean isCustomSanit(AbstractNode node) {
639             if (node instanceof NormalNode) {
640                 return false;
641             } else if (node instanceof UninitializedNode) {
642                 return false;
643             } else if (node instanceof BuiltinFunctionNode) {
644                 // check if this operation could be used for custom sanitization
645                 BuiltinFunctionNode builtinFunctionNode = (BuiltinFunctionNode) node;
646                 if (builtinFunctionNode.isBuiltin()) {
647                     AbstractCfgNode cfgNode = builtinFunctionNode.getCfgNode();
648                     if (cfgNode instanceof CallBuiltinFunction) {
649                         CallBuiltinFunction callBuiltin = (CallBuiltinFunction) cfgNode;
650                         String funcName = callBuiltin.getFunctionName();
651     
652                         // here is the list of custom sanitization functions
653                         return funcName.equals("ereg_replace") ||
654                             funcName.equals("preg_replace") ||
655                             funcName.equals("str_replace");
656                     } else {
657                         return false;
658                     }
659                 } else {
660                     return false;
661                 }
662             } else if (node instanceof CompleteGraphNode) {
663                 throw new RuntimeException("SNH");
664             } else {
665                 throw new RuntimeException("SNH");
666             }
667         }

669         public boolean isIneffective(AbstractNode customSanit, Map<AbstractNode, FSAAutomaton> deco) {
670             FSAAutomaton auto = deco.get(customSanit);
671             if (auto == null) {
672                 // no decoration for this node: be conservative
673                 return true;
674             }
675     
676             // intersect!
677             FSAAutomaton intersection = auto.intersect(this.undesiredAutomaton);
678             return !intersection.isEmpty();
679         }

681         /**
682          * Locates custom sanitization nodes in the given dependency graph and returns them.
683          *
684          * @param dependencyGraph
685          *
686          * @return
687          */
688         public static List<AbstractNode> findCustomSanit(DependencyGraph dependencyGraph) {
689             List<AbstractNode> customSanitationNOdes = new LinkedList<>();
690             for (AbstractNode node : dependencyGraph.getNodes()) {
691                 if (isCustomSanit(node)) {
692                     customSanitationNOdes.add(node);
693                 }
694             }
695     
696             return customSanitationNOdes;
697         }

699         /**
700          * Take care: if trimAllowed == false, no need to call this method.
701          *
702          * @param node
703          *
704          * @return
705          */
706         private TrimInfo checkTrim(AbstractNode node) {
707             // start with default triminfo: everything can be trimmed
708             TrimInfo retMe = new TrimInfo();
709     
710             // handle cases where trimming is not allowed
711             if (node instanceof BuiltinFunctionNode) {
712                 BuiltinFunctionNode builtinFunctionNode = (BuiltinFunctionNode) node;
713                 if (builtinFunctionNode.isBuiltin()) {
714                     String opName = builtinFunctionNode.getName();
715                     if (opName.equals("preg_replace") ||
716                         opName.equals("ereg_replace") ||
717                         opName.equals("str_replace")) {
718                         retMe.addNoTrim(0);
719                     }
720                 }
721             }
722     
723             return retMe;
724         }

726         /**
727          * Helper class for exchanging information on whether to allow trimming.
728          */
729         private class TrimInfo {
730             /** these indices must be trimmed */
731             private List<Integer> trim;
732             /** these indices must not be trimmed */
733             private List<Integer> noTrim;
734             /** what to do with all remaining indices */
735             private boolean defaultTrim;
736     
737             TrimInfo() {
738                 this.defaultTrim = true;
739                 this.trim = new LinkedList<>();
740                 this.noTrim = new LinkedList<>();
741             }
742     
743             void setDefaultTrim(boolean defaultTrim) {
744                 this.defaultTrim = defaultTrim;
745             }
746     
747             void addNoTrim(int i) {
748                 this.noTrim.add(i);
749             }
750     
751             boolean mayTrim(int i) {
752                 return trim.contains(i) || !noTrim.contains(i) && defaultTrim;
753             }
754         }
