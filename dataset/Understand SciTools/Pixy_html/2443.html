1     package at.ac.tuwien.infosys.www.pixy.analysis.dependency;
2     
3     import at.ac.tuwien.infosys.www.pixy.analysis.GenericRepository;
4     import at.ac.tuwien.infosys.www.pixy.analysis.Recyclable;
5     import at.ac.tuwien.infosys.www.pixy.conversion.cfgnodes.AbstractCfgNode;
6     
7     /**
8      * Corresponds to one dependency label, consisting of
9      * <source descriptor (e.g., variable or function name)>, <location (cfg node)>.
10      *
11      * @author Nenad Jovanovic <enji@seclab.tuwien.ac.at>
12      */
13     public class DependencyLabel implements Recyclable {
14         public static GenericRepository<DependencyLabel> repos =
15             new GenericRepository<>();
16     
17         // special, parameterized label
18         public static final DependencyLabel UNINIT =
19             new DependencyLabel(null);
20     
21         private AbstractCfgNode cfgNode;
22     
23     //  ********************************************************************************
24     //  CONSTRUCTORS *******************************************************************
25     //  ********************************************************************************
26     
27     //  ********************************************************************************
28     
29         private DependencyLabel(AbstractCfgNode cfgNode) {
30             this.cfgNode = cfgNode;
31         }
32     
33     //  ********************************************************************************
34     
35         public static DependencyLabel create(AbstractCfgNode cfgNode) {
36             DependencyLabel ret = new DependencyLabel(cfgNode);
37             ret = repos.recycle(ret);
38             return ret;
39         }
40     
41     //  ********************************************************************************
42     //  OTHER **************************************************************************
43     //  ********************************************************************************
44     
45     //  ********************************************************************************
46     
47         public AbstractCfgNode getCfgNode() {
48             return this.cfgNode;
49         }
50     
51     //  ********************************************************************************
52     
53         public String toString() {
54             if (this == UNINIT) {
55                 return " <uninit> ";
56             }
57             StringBuilder buf = new StringBuilder();
58             buf.append(" (");
59             if (this.cfgNode == null) {
60                 buf.append("none");
61             } else {
62                 buf.append(this.cfgNode.getOriginalLineNumber());
63             }
64             buf.append(", ");
65             buf.append(this.cfgNode.toString());
66             buf.append(") ");
67             return buf.toString();
68         }
69     
70     //  ********************************************************************************
71     
72         public boolean structureEquals(Object compX) {
73             if (compX == this) {
74                 return true;
75             }
76             if (!(compX instanceof DependencyLabel)) {
77                 return false;
78             }
79             DependencyLabel comp = (DependencyLabel) compX;
80             if (this.cfgNode == null) {
81                 return this == comp;
82             }
83             return this.cfgNode.equals(comp.cfgNode);
84         }
85     
86     //  ********************************************************************************
87     
88         public int structureHashCode() {
89             int hashCode = 17;
90             hashCode = 37 * hashCode + this.cfgNode.hashCode();
91             return hashCode;
92         }
93     }
