12     public class InterproceduralAnalysisInformation extends AbstractAnalysisInformation {
13         public InterproceduralAnalysisInformation() {
14             super();
15         }
16     
17         // folds all analysis nodes (using recycling) and clears the phi maps
18         // (=> saves memory)
19         public void foldRecycledAndClean(AbstractInterproceduralAnalysis analysis) {
20             for (AbstractAnalysisNode analysisNode1 : this.map.values()) {
21                 AbstractInterproceduralAnalysisNode analysisNode = (AbstractInterproceduralAnalysisNode) analysisNode1;
22                 AbstractLatticeElement foldedValue = analysisNode.computeFoldedValue();
23                 foldedValue = analysis.recycle(foldedValue);
24                 analysisNode.setFoldedValue(foldedValue);
25                 analysisNode.clearPhiMap();
26             }
27         }
28     
29         // note that not all cfg nodes have an associated analysis node:
30         // - nodes inside basic blocks
31         // - nodes inside function default cfgs
32         // for such nodes, you should query the enclosing basic block, or the
33         // entry node of the function default cfg; use the appropriate "get"
34         // method of CfgNode to retrieve these nodes
35         public AbstractInterproceduralAnalysisNode getAnalysisNode(AbstractCfgNode cfgNode) {
36             return (AbstractInterproceduralAnalysisNode) this.map.get(cfgNode);
37         }
38     
39         public AbstractTransferFunction getTransferFunction(AbstractCfgNode cfgNode) {
40             AbstractAnalysisNode analysisNode = this.getAnalysisNode(cfgNode);
41             return analysisNode.getTransferFunction();
42         }
