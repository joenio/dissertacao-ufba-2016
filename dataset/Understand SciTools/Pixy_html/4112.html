13     public class MustAliases {
14         // contains MustAliasGroup's
15         private Set<MustAliasGroup> groups;

21         // creates empty must alias information
22         public MustAliases() {
23             this.groups = new HashSet<>();
24         }
25     
26         // clones the given object
27         public MustAliases(MustAliases cloneMe) {
28             this.groups = new HashSet<>();
29             Set<MustAliasGroup> origGroup = cloneMe.getGroups();
30             for (MustAliasGroup group : origGroup) {
31                 this.groups.add(new MustAliasGroup(group));
32             }
33         }
34     
35     //  ********************************************************************************
36     //  GET ****************************************************************************
37     //  ********************************************************************************
38     
39         public Set<MustAliasGroup> getGroups() {
40             return this.groups;
41         }
42     
43         // returns a set of all variables that occur in the contained groups
44         public Set<Variable> getVariables() {
45             Set<Variable> variables = new HashSet<>();
46             for (MustAliasGroup group : this.groups) {
47                 variables.addAll(group.getVariables());
48             }
49             return variables;
50         }
51     
52         // returns the MustAliasGroup that contains the given
53         // variable, or NULL if there is no such explicit group
54         public MustAliasGroup getMustAliasGroup(Variable x) {
55             for (MustAliasGroup group : this.groups) {
56                 if (group.contains(x)) {
57                     return group;
58                 }
59             }
60     
61             return null;
62         }
63     
64         // returns all global variables that are must-aliases of the given variable
65         // (a set of Variables)
66         public Set<Variable> getGlobalAliases(Variable var) {
67             Set<Variable> retMe = new HashSet<>();
68             MustAliasGroup group = this.getMustAliasGroup(var);
69             if (group != null) {
70                 retMe.addAll(group.getGlobals());
71             }
72             return retMe;
73         }
74     
75         // returns true if the given variables are must-aliases, and false otherwise
76         public boolean isMustAlias(Variable x, Variable y) {
77     
78             MustAliasGroup groupX = this.getMustAliasGroup(x);
79     
80             // if x's group is implicit, x isn't a must-alias of anyone
81             return groupX != null && groupX.contains(y);
82         }

88         // merges the must-alias-groups that the given variables belong to (also
89         // considering implicit one-element groups)
90         public void merge(Variable x, Variable y) {
91     
92             MustAliasGroup groupX = this.getMustAliasGroup(x);
93             MustAliasGroup groupY = this.getMustAliasGroup(y);
94     
95             if (groupX == null) {
96                 if (groupY == null) {
97                     // both variables only have implicit groups;
98                     // add a new two-element group containing the variables
99                     this.add(new MustAliasGroup(x, y));
100                 } else {
101                     // add x to y's existing group
102                     this.addToGroup(x, groupY);
103                 }
104             } else {
105                 if (groupY == null) {
106                     // add y to x's existing group
107                     this.addToGroup(y, groupX);
108                 } else {
109                     // both groups exist and must be merged
110                     MustAliasGroup mergedGroup = new MustAliasGroup(groupX, groupY);
111                     this.groups.remove(groupX);
112                     this.groups.remove(groupY);
113                     this.groups.add(mergedGroup);
114                 }
115             }
116         }
117     
118         public void removeLocals() {
119     
120             List<MustAliasGroup> keepUs = new LinkedList<>();
121             for (MustAliasGroup group : this.groups) {
122                 group.removeLocals();
123     
124                 if (group.size() > 1) {
125                     keepUs.add(group);
126                 }
127             }
128             this.groups = new HashSet<>(keepUs);
129         }
130     
131         public void removeGlobals() {
132     
133             List<MustAliasGroup> keepUs = new LinkedList<>();
134             for (MustAliasGroup group : this.groups) {
135                 group.removeGlobals();
136     
137                 if (group.size() > 1) {
138                     keepUs.add(group);
139                 }
140             }
141             this.groups = new HashSet<>(keepUs);
142         }
143     
144         public void removeVariables(SymbolTable symTab) {
145     
146             List<MustAliasGroup> keepUs = new LinkedList<>();
147             for (MustAliasGroup group : this.groups) {
148                 group.removeVariables(symTab);
149     
150                 if (group.size() > 1) {
151                     keepUs.add(group);
152                 }
153             }
154             this.groups = new HashSet<>(keepUs);
155         }
156     
157         // removes the given variable from its group (if there
158         // is such a group); if this leads to a group having only
159         // one member, this group is removed completely
160         public void remove(Variable var) {
161             MustAliasGroup group = this.getMustAliasGroup(var);
162             if (group != null) {
163                 MustAliasGroup groupCopy = new MustAliasGroup(group);
164                 groupCopy.remove(var);
165                 if (groupCopy.size() < 2) {
166                     this.groups.remove(group);
167                 } else {
168                     group.remove(var);
169                 }
170             }
171         }
172     
173         // adds "addMe" to the group of "host"; if the host has no
174         // group (i.e., it is contained in an implicit one-element group),
175         // a new two-element group with addMe and host as members is created
176         public void addToGroup(Variable addMe, Variable host) {
177             MustAliasGroup hostGroup = this.getMustAliasGroup(host);
178     
179             if (hostGroup == null) {
180                 hostGroup = new MustAliasGroup(addMe, host);
181                 this.groups.add(hostGroup);
182             } else {
183                 hostGroup.add(addMe);
184             }
185         }
186     
187         // adds "addMe" to the given (non-null) group
188         public void addToGroup(Variable addMe, MustAliasGroup hostGroup) {
189             hostGroup.add(addMe);
190         }
191     
192         // adds the groups of "source" to this.groups
193         // (without deep copy);
194         // the caller has to make sure that they are disjoint
195         public void add(MustAliases source) {
196             Set<MustAliasGroup> sourceGroups = source.getGroups();
197             this.groups.addAll(sourceGroups);
198         }
199     
200         public void add(MustAliasGroup group) {
201             this.groups.add(group);
202         }
203     
204         public void replace(Map<Variable, Variable> replacements) {
205             for (MustAliasGroup group : this.groups) {
206                 group.replace(replacements);
207             }
208         }

212             Set<MustAliasGroup> compGroupSet = comp.getGroups();
