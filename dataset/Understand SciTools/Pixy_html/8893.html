26         // input elements (incoming and target) are not modified,
27         // output element is newly allocated
28         public AbstractLatticeElement lub(
29             AbstractLatticeElement incomingElementX,
30             AbstractLatticeElement targetElementX) {
31     
32             // if the incoming element is the bottom element: return the other element
33             if (incomingElementX == this.bottom) {
34                 if (targetElementX != this.bottom) {
35                     return targetElementX.cloneMe();
36                 } else {
37                     return this.bottom;
38                 }
39             }
40     
41             // if the target element is the bottom element: return the other element
42             if (targetElementX == this.bottom) {
43                 // return new TaintLatticeElement((TaintLatticeElement) incomingElementX);
44                 return incomingElementX.cloneMe();
45             }
46     
47             // if one of the elements is the top element: return the top element;
48             // not necessary here: we will never encounter the top element
49     
50             // class cast
51             DependencyLatticeElement incomingElement = (DependencyLatticeElement) incomingElementX;
52             DependencyLatticeElement targetElement = (DependencyLatticeElement) targetElementX;
53     
54             // initialize the resulting lattice element as clone of the target
55             // lattice element (we don't want to modify the target lattice element
56             // given by the caller of this method);
57             // EFF: note that we must not modify the incoming element either: while
58             // most transfer functions generate a new lattice element object which
59             // could be reused, the ID transfer function simply passes on the
60             // reference; if you reuse here, the ID transfer function must return
61             // a new element (not clear which method is more efficient)
62             DependencyLatticeElement resultElement = new DependencyLatticeElement(targetElement);
63     
64             // lub the incoming element over the clone of the target element
65             resultElement.lub(incomingElement);
66     
67             return resultElement;
68         }
