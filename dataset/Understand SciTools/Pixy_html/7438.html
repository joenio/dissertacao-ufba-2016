14     public class MayAliasPair {
15         // a pair of Variable's;
16         // EFF: it might be faster if you implement this explicitly with two variables
17         // instead of using a hash set
18         private Set<Variable> pair;

23     
24         public MayAliasPair(Variable a, Variable b) {
25             this.pair = new HashSet<>();
26             this.pair.add(a);
27             this.pair.add(b);
28         }
29     
30         // clones the given pair
31         public MayAliasPair(MayAliasPair orig) {
32             this.pair = new HashSet<>();
33             for (Variable variable : orig.getPair()) {
34                 this.pair.add(variable);
35             }
36         }
37     
38     //  ********************************************************************************
39     //  GET ****************************************************************************
40     //  ********************************************************************************
41     
42         public Set<Variable> getPair() {
43             return this.pair;
44         }

46         // if the given variable is contained in this group and the other member
47         // of the group is a global variable, this global variable is returned;
48         // null otherwise
49         public Variable getGlobalMayAlias(Variable var) {
50             Iterator<Variable> iter = this.pair.iterator();
51             Variable firstElement = iter.next();
52             Variable secondElement = iter.next();
53     
54             if (firstElement == var) {
55                 if (secondElement.isGlobal()) {
56                     return secondElement;
57                 } else {
58                     return null;
59                 }
60             } else if (secondElement == var) {
61                 if (firstElement.isGlobal()) {
62                     return firstElement;
63                 } else {
64                     return null;
65                 }
66             } else {
67                 return null;
68             }
69         }
70     
71         // if the given variable is contained in this group and the other member
72         // of the group is a local variable, this local variable is returned;
73         // null otherwise
74         public Variable getLocalMayAlias(Variable var) {
75             Iterator<Variable> iter = this.pair.iterator();
76             Variable firstElement = iter.next();
77             Variable secondElement = iter.next();
78     
79             if (firstElement == var) {
80                 if (secondElement.isLocal()) {
81                     return secondElement;
82                 } else {
83                     return null;
84                 }
85             } else if (secondElement == var) {
86                 if (firstElement.isLocal()) {
87                     return firstElement;
88                 } else {
89                     return null;
90                 }
91             } else {
92                 return null;
93             }
94         }
95     
96         // if the given variable is contained in this pair, the other variable
97         // from the pair is returned; null otherwise
98         public Variable getMayAlias(Variable var) {
99             Iterator<Variable> iter = this.pair.iterator();
100             Variable firstElement = iter.next();
101             Variable secondElement = iter.next();
102     
103             if (firstElement == var) {
104                 return secondElement;
105             } else if (secondElement == var) {
106                 return firstElement;
107             } else {
108                 return null;
109             }
110         }
111     
112         // if this pair contains one local and one global variable, an array containing
113         // these variables (in this order) is returned; null otherwise
114         public Variable[] getLocalGlobal() {
115             Iterator<Variable> iter = this.pair.iterator();
116             Variable firstElement = iter.next();
117             Variable secondElement = iter.next();
118     
119             if (firstElement.isLocal()) {
120                 if (secondElement.isGlobal()) {
121                     Variable[] retMe = {firstElement, secondElement};
122                     return retMe;
123                 } else {
124                     return null;
125                 }
126             } else if (firstElement.isGlobal()) {
127                 if (secondElement.isLocal()) {
128                     Variable[] retMe = {secondElement, firstElement};
129                     return retMe;
130                 } else {
131                     return null;
132                 }
133             } else {
134                 throw new RuntimeException("SNH");
135             }
136         }

142         // returns true if at least one of the contained variables is a local
143         // variable, and false otherwise
144         public boolean containsLocals() {
145             for (Variable var : this.pair) {
146                 if (var.isLocal()) {
147                     return true;

149             }
150             return false;
151         }

153         // returns true if at least one of the contained variables is a local
154         // variable, and false otherwise
155         public boolean containsGlobals() {
156             for (Variable var : this.pair) {
157                 if (var.isGlobal()) {
158                     return true;
159                 }
160             }
161             return false;
162         }

164         // returns true if at least one of the contained variables belongs to
165         // the given symbol table, and false otherwise
166         public boolean containsVariables(SymbolTable symTab) {
167             for (Variable var : this.pair) {
168                 if (var.belongsTo(symTab)) {
169                     return true;
170                 }
171             }
172             return false;
173         }

175         // replaces "replaceMe" by "replaceBy"; expects "replaceMe" to
176         // really appear in this pair
177         public void replaceBy(Variable replaceMe, Variable replaceBy) {
178             if (!this.pair.remove(replaceMe)) {
179                 throw new RuntimeException("SNH");
180             }
181             this.pair.add(replaceBy);
182         }
183     
184         // expects a map Variable -> Variable (replaceMe -> replaceBy)
185         public void replace(Map<Variable, Variable> replacements) {
186             Set<Variable> newPair = new HashSet<>(this.pair);
187             for (Variable var : this.pair) {
188                 Variable replaceBy = replacements.get(var);
189                 if (replaceBy != null) {
190                     newPair.remove(var);
191                     newPair.add(replaceBy);
192                 }
193             }
194             this.pair = newPair;
195         }
196     
197         public boolean contains(Variable var) {
198             return this.pair.contains(var);
199         }
