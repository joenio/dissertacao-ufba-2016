51         // checks if the given node (inside the given function) is a sensitive sink;
52         // adds an appropriate sink object to the given list if it is a sink
53         protected void checkForSink(AbstractCfgNode cfgNodeX, TacFunction traversedFunction,
54                                     List<Sink> sinks) {
55     
56             if (cfgNodeX instanceof CallBuiltinFunction) {
57     
58                 // builtin function sinks
59     
60                 CallBuiltinFunction cfgNode = (CallBuiltinFunction) cfgNodeX;
61                 String functionName = cfgNode.getFunctionName();
62     
63                 checkForSinkHelper(functionName, cfgNode, cfgNode.getParamList(), traversedFunction, sinks);
64             } else if (cfgNodeX instanceof CallPreparation) {
65     
66                 CallPreparation cfgNode = (CallPreparation) cfgNodeX;
67                 String functionName = cfgNode.getFunctionNamePlace().toString();
68     
69                 // user-defined custom sinks
70     
71                 checkForSinkHelper(functionName, cfgNode, cfgNode.getParamList(), traversedFunction, sinks);
72             } else {
73                 // not a sink
74             }
75         }
76     
77     //  ********************************************************************************
78     
79         private void checkForSinkHelper(String functionName, AbstractCfgNode cfgNode,
80                                         List<TacActualParameter> paramList, TacFunction traversedFunction, List<Sink> sinks) {
81     
82             if (this.vulnerabilityAnalysisInformation.getSinks().containsKey(functionName)) {
83                 Sink sink = new Sink(cfgNode, traversedFunction);
84                 for (Integer param : this.vulnerabilityAnalysisInformation.getSinks().get(functionName)) {
85                     if (paramList.size() > param) {
86                         sink.addSensitivePlace(paramList.get(param).getPlace());
87                         // add this sink to the list of sensitive sinks
88                         sinks.add(sink);
89                     }
90                 }
91             } else {
92                 // not a sink
93             }
94         }
