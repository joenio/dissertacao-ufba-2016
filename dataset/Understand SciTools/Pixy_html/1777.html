41         // restriction that superclass constructors have to be called first;
42         protected void initGeneral(TacFunction function) {
43     
44             // first-time initialization
45             if (this.functions == null) {
46                 this.functions = new LinkedList<>();
47                 // initialize carrier lattice
48                 this.initLattice();
49             }
50     
51             this.functions.add(function);
52     
53             // initialize worklist
54             this.workList = new IntraproceduralWorklist();
55             this.workList.add(function.getControlFlowGraph().getHead());
56     
57             // initialize analysis nodes
58             this.analysisInfo = new IntraproceduralAnalysisInformation();
59             this.genericAnalysisInformation = analysisInfo;
60     
61             // assign transfer functions to analysis nodes
62             this.traverseCfg(function.getControlFlowGraph(), function);
63             // this.asfsafsaf: initTransferFunctions

65             // initialize inValue for start node

107     // analyze *************************************************************************
108     
109         // this method applies the worklist algorithm
110         public void analyze() {
111     
112             // for each element in the worklist...
113             while (this.workList.hasNext()) {
114     
115                 // remove the element from the worklist
116                 AbstractCfgNode node = this.workList.removeNext();
117     
118                 // get incoming value at node n
119                 IntraproceduralAnalysisNode analysisNode = this.analysisInfo.getAnalysisNode(node);
120                 AbstractLatticeElement inValue = analysisNode.getInValue();
121                 if (inValue == null) {
122                     throw new RuntimeException("SNH");
123                 }
124     
125                 try {
126     
127                     // apply transfer function to incoming value
128                     AbstractLatticeElement outValue;
129                     outValue = this.analysisInfo.getAnalysisNode(node).transfer(inValue);
130     
131                     // for each outgoing edge...
132                     CfgEdge[] outEdges = node.getOutEdges();
133                     for (CfgEdge outEdge : outEdges) {
134                         if (outEdge != null) {
135     
136                             // determine the successor
137                             AbstractCfgNode succ = outEdge.getDestination();
138     
139                             // propagate the result of applying the transfer function
140                             // to the successor
141                             propagate(outValue, succ);
142                         }
143                     }
144                 } catch (RuntimeException ex) {
145                     System.out.println("File:" + node.getFileName() + ", Line: " + node.getOriginalLineNumber());
146                     throw ex;
147                 }
148             }
149     
150             // worklist algorithm finished!
151         }
152     
153     //  propagate ***********************************************************************
154     
155         // helper method for analyze();
156         // propagates a value to the target node
157         void propagate(AbstractLatticeElement value, AbstractCfgNode target) {
158     
159             // analysis information for the target node
160             IntraproceduralAnalysisNode analysisNode = this.analysisInfo.getAnalysisNode(target);
161     
162             if (analysisNode == null) {
163                 System.out.println(Dumper.makeCfgNodeName(target));
164                 throw new RuntimeException("SNH: " + target.getClass());
165             }
166     
167             // determine the target's old invalue
168             AbstractLatticeElement oldInValue = analysisNode.getInValue();
169             if (oldInValue == null) {
170                 // initial value of this analysis
171                 oldInValue = this.initialValue;
172             }
173     
174             // speedup: if incoming value and target value are exactly the same
175             // object, then the result certainly can't change
176             if (value == oldInValue) {
177                 return;
178             }
179     
180             // the new invalue is computed as usual (with lub)
181             AbstractLatticeElement newInValue = this.lattice.lub(value, oldInValue);
182     
183             // if the invalue changed...
184             if (!oldInValue.equals(newInValue)) {
185     
186                 // update analysis information
187                 analysisNode.setInValue(newInValue);
188     
189                 // add this node to the worklist
190                 this.workList.add(target);
191             }
192         }
