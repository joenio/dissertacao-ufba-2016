141         /**
142          * Returns the automaton representation of the given dependency graph.
143          *
144          * This is done by decorating the nodes of the graph with automata bottom-up, and returning the automaton that
145          * eventually decorates the root.
146          *
147          * Beware: This also eliminates cycles!
148          *
149          * @param dependencyGraph
150          *
151          * @return
152          */
153         private Automaton toAutomaton(DependencyGraph dependencyGraph) {
154             dependencyGraph.eliminateCycles();
155             AbstractNode root = dependencyGraph.getRootNode();
156             Map<AbstractNode, Automaton> deco = new HashMap<>();
157             Set<AbstractNode> visited = new HashSet<>();
158             this.decorate(root, deco, visited, dependencyGraph);
159             Automaton rootDeco = deco.get(root).clone();
160     
161             // BEWARE: minimization can lead to an automaton that is less human-readable
162             //rootDeco.minimize();
163             return rootDeco;
164         }

166         /**
167          * Decorates the given node (and all its successors) with an automaton.
168          *
169          * @param node
170          * @param deco
171          * @param visited
172          * @param dependencyGraph
173          */
174         private void decorate(
175             AbstractNode node, Map<AbstractNode, Automaton> deco, Set<AbstractNode> visited, DependencyGraph dependencyGraph
176         ) {
177             visited.add(node);
178     
179             // if this node has successors, decorate them first (if not done yet)
180             List<AbstractNode> successors = dependencyGraph.getSuccessors(node);
181             if (successors != null && !successors.isEmpty()) {
182                 for (AbstractNode succ : successors) {
183                     if (!visited.contains(succ) && deco.get(succ) == null) {
184                         decorate(succ, deco, visited, dependencyGraph);
185                     }
186                 }
187             }
188     
189             // now that all successors are decorated, we can decorate this node
190             Automaton auto = null;
191             if (node instanceof NormalNode) {
192                 NormalNode normalNode = (NormalNode) node;
193                 if (successors == null || successors.isEmpty()) {
194                     // this should be a string leaf node
195                     AbstractTacPlace place = normalNode.getPlace();
196                     if (place.isLiteral()) {
197                         auto = Automaton.makeString(place.toString());
198                     } else {
199                         // this case should not happen any longer (now that
200                         // we have "uninit" nodes, see below)
201                         System.err.println(normalNode.getCfgNode().getLoc());
202                         throw new RuntimeException("SNH: " + place);
203                     }
204                 } else {
205                     // this is an interior node, not a leaf node;
206                     // the automaton for this node is the union of all the
207                     // successor automatas
208                     for (AbstractNode succ : successors) {
209                         Automaton succAuto = deco.get(succ);
210                         if (auto == null) {
211                             auto = succAuto;  // cloning not necessary here
212                         } else {
213                             auto = auto.union(succAuto);
214                         }
215                     }
216                 }
217             } else if (node instanceof BuiltinFunctionNode) {
218                 auto = this.makeAutomatonForOperationNode((BuiltinFunctionNode) node, deco, dependencyGraph);
219             } else if (node instanceof CompleteGraphNode) {
220                 // conservative decision for SCCs
221                 auto = Automaton.makeAnyString(Transition.Taint.Directly);
222             } else if (node instanceof UninitializedNode) {
223                 // retrieve predecessor
224                 Set<AbstractNode> predecessors = dependencyGraph.getPredecessors(node);
225                 if (predecessors.size() != 1) {
226                     throw new RuntimeException("SNH");
227                 }
228                 AbstractNode pre = predecessors.iterator().next();
229     
230                 if (pre instanceof NormalNode) {
231                     NormalNode preNormal = (NormalNode) pre;
232     
233                     switch (this.getInitialTaintForPlace(preNormal.getPlace())) {
234                         case ALWAYS:
235                         case IF_REGISTER_GLOBALS:
236                             auto = Automaton.makeAnyString(Transition.Taint.Directly);
237                             break;
238                         case NEVER:
239                             auto = Automaton.makeAnyString(Transition.Taint.Untainted);
240                             break;
241                         default:
242                             throw new RuntimeException("SNH");
243                     }
244                 } else if (pre instanceof BuiltinFunctionNode) {
245                     throw new RuntimeException("not yet");
246                 } else if (pre instanceof CompleteGraphNode) {
247                     // conservative decision for SCCs
248                     auto = Automaton.makeAnyString(Transition.Taint.Directly);
249                 } else {
250                     throw new RuntimeException("SNH");
251                 }
252             } else {
253                 throw new RuntimeException("SNH");
254             }
255     
256             if (auto == null) {
257                 throw new RuntimeException("SNH");
258             }
259     
260             deco.put(node, auto);
261         }

263         /**
264          * Returns an automaton for the given operation node.
265          *
266          * @param node
267          * @param deco
268          * @param dependencyGraph
269          *
270          * @return
271          */
272         private Automaton makeAutomatonForOperationNode(
273             BuiltinFunctionNode node, Map<AbstractNode, Automaton> deco, DependencyGraph dependencyGraph)
274         {
275             List<AbstractNode> successors = dependencyGraph.getSuccessors(node);
276             if (successors == null) {
277                 successors = new LinkedList<>();
278             }
279     
280             Automaton retMe = null;
281     
282             String opName = node.getName();
283             if (opName.equals(".")) {
284                 // CONCAT
285                 for (AbstractNode successor : successors) {
286                     Automaton successorAutomaton = deco.get(successor);
287                     if (retMe == null) {
288                         retMe = successorAutomaton;
289                     } else {
290                         retMe = retMe.concatenate(successorAutomaton);
291                     }
292                 }
293             } else {
294                 // conservative decision for operations that have not been
295                 // modeled yet: .*
296                 retMe = Automaton.makeAnyString(Transition.Taint.Directly);
297             }
298     
299             return retMe;
300         }

302         /**
303          * Checks if the given node (inside the given function) is a sensitive sink.
304          *
305          * Adds an appropriate sink object to the given list if it is a sink.
306          *
307          * @param cfgNodeX
308          * @param traversedFunction
309          * @param sinks
310          */
311         protected void checkForSink(AbstractCfgNode cfgNodeX, TacFunction traversedFunction, List<Sink> sinks) {
312             if (cfgNodeX instanceof CallBuiltinFunction) {
313                 // builtin function sinks
314                 CallBuiltinFunction cfgNode = (CallBuiltinFunction) cfgNodeX;
315                 String functionName = cfgNode.getFunctionName();
316     
317                 checkForSinkHelper(functionName, cfgNode, cfgNode.getParamList(), traversedFunction, sinks);
318             } else if (cfgNodeX instanceof CallPreparation) {
319                 // user-defined custom sinks
320                 CallPreparation cfgNode = (CallPreparation) cfgNodeX;
321                 String functionName = cfgNode.getFunctionNamePlace().toString();
322     
323                 checkForSinkHelper(functionName, cfgNode, cfgNode.getParamList(), traversedFunction, sinks);
324             }
325         }

327         private void checkForSinkHelper(
328             String functionName, AbstractCfgNode cfgNode, List<TacActualParameter> paramList, TacFunction traversedFunction,
329             List<Sink> sinks
330         ) {
331             if (this.vulnerabilityAnalysisInformation.getSinks().containsKey(functionName)) {
332                 Sink sink = new Sink(cfgNode, traversedFunction);
333                 for (Integer param : this.vulnerabilityAnalysisInformation.getSinks().get(functionName)) {
334                     if (paramList.size() > param) {
335                         sink.addSensitivePlace(paramList.get(param).getPlace());
336                         // add this sink to the list of sensitive sinks
337                         sinks.add(sink);
338                     }
339                 }
340             }
341         }
