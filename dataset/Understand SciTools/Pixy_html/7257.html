14     /**
15      * Transfer function for reference assignment nodes.
16      *
17      * @author Nenad Jovanovic <enji@seclab.tuwien.ac.at>
18      */
19     public class AssignReference extends AbstractTransferFunction {
20         private Variable left;
21         private Variable right;
22         private boolean supported;
23     
24     // *********************************************************************************
25     // CONSTRUCTORS ********************************************************************
26     // *********************************************************************************
27     
28         // mustAliases, mayAliases: of setMe
29         public AssignReference(AbstractTacPlace left, AbstractTacPlace right) {
30     
31             this.left = (Variable) left;    // must be a variable
32             this.right = (Variable) right;  // must be a variable
33     
34             // check for unsupported features
35             this.supported = AliasAnalysis.isSupported(this.left, this.right, false, -1);
36         }
37     
38     // *********************************************************************************
39     // OTHER ***************************************************************************
40     // *********************************************************************************
41     
42         public AbstractLatticeElement transfer(AbstractLatticeElement inX) {
43     
44             // if this reference assignment is not supported by our alias analysis,
45             // we simply ignore it
46             if (!supported) {
47                 return inX;
48             }
49     
50             LiteralLatticeElement in = (LiteralLatticeElement) inX;
51             LiteralLatticeElement out = new LiteralLatticeElement(in);
52     
53             // "left =& right" means that left is redirected to right;
54             // for the literal mapping, this means that nothing changes
55             // except that left receives the literal of right;
56             // we achieve this through the following actions:
57     
58             Set<Variable> mustAliases = new HashSet<>();
59             mustAliases.add(left);
60             Set<Variable> mayAliases = Collections.emptySet();
61     
62             // let the lattice element handle the details
63             out.assignSimple(left, right, mustAliases, mayAliases);
64     
65             return out;
66         }
67     }
