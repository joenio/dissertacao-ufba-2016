10     /**
11      * Transfer function for simple assignment nodes.
12      *
13      * @author Nenad Jovanovic <enji@seclab.tuwien.ac.at>
14      */
15     public class Unset extends AbstractTransferFunction {
16         private Variable operand;
17         private AliasAnalysis aliasAnalysis;
18         private boolean supported;
19     
20     // *********************************************************************************
21     // CONSTRUCTORS ********************************************************************
22     // *********************************************************************************
23     
24         public Unset(AbstractTacPlace operand, AliasAnalysis aliasAnalysis) {
25     
26             // only variables can be unset
27             if (!operand.isVariable()) {
28                 throw new RuntimeException("Trying to unset a non-variable.");
29             }
30     
31             this.operand = operand.getVariable();
32             this.aliasAnalysis = aliasAnalysis;
33             this.supported = AliasAnalysis.isSupported(this.operand);
34         }
35     
36     // *********************************************************************************
37     // OTHER ***************************************************************************
38     // *********************************************************************************
39     
40         public AbstractLatticeElement transfer(AbstractLatticeElement inX) {
41     
42             AliasLatticeElement in = (AliasLatticeElement) inX;
43     
44             // if the operand is not supported for reference statements:
45             // nothing to do, since we don't support such aliases
46             if (!this.supported) {
47                 return in;
48             }
49     
50             AliasLatticeElement out = new AliasLatticeElement(in);
51     
52             // perform unset operation on "out"
53             out.unset(this.operand);
54     
55             // recycle
56             out = (AliasLatticeElement) this.aliasAnalysis.recycle(out);
57     
58             return out;
59         }
60     }
