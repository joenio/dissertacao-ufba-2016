30     package at.ac.tuwien.infosys.www.pixy.automaton;
31     
32     /**
33      * Pair of states.
34      *
35      * @author Anders M&oslash;ller &lt;<a href="mailto:amoeller@brics.dk">amoeller@brics.dk</a>&gt;
36      */
37     public class StatePair {
38         State s;
39         State s1;
40         State s2;
41     
42         StatePair(State s, State s1, State s2) {
43             this.s = s;
44             this.s1 = s1;
45             this.s2 = s2;
46         }
47     
48         /**
49          * Constructs new state pair.
50          *
51          * @param s1 first state
52          * @param s2 second state
53          */
54         public StatePair(State s1, State s2) {
55             this.s1 = s1;
56             this.s2 = s2;
57         }
58     
59         /**
60          * Checks for equality.
61          *
62          * @param obj object to compare with
63          *
64          * @return true if <tt>obj</tt> represents the same pair of states as this pair
65          */
66         @Override
67         public boolean equals(Object obj) {
68             if (obj instanceof StatePair) {
69                 StatePair p = (StatePair) obj;
70                 return p.s1 == s1 && p.s2 == s2;
71             } else
72                 return false;
73         }
74     
75         /**
76          * Returns hash code.
77          *
78          * @return hash code
79          */
80         @Override
81         public int hashCode() {
82             return s1.hashCode() + s2.hashCode();
83         }
84     }
