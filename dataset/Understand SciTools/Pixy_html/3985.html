9     class IncludeNode {

11         private String canonicalPath;
12     
13         IncludeNode(File file) {
14             this.file = file;
15             try {
16                 this.canonicalPath = file.getCanonicalPath();
17             } catch (IOException e) {
18                 throw new RuntimeException(e.getMessage());
19             }
20         }
21     
22         String getCanonicalPath() {
23             return this.canonicalPath;
24         }
25     
26         public int hashCode() {
27             return this.canonicalPath.hashCode();
28         }
29     
30         public boolean equals(Object obj) {
31             // the "equals" method of File is stupid:
32             // only compares with File.getPath() (not canonical
33             // and not even absolute), so we have to do this
34             // ourselves
35             if (!(obj instanceof IncludeNode)) {
36                 return false;
37             }
38             IncludeNode comp = (IncludeNode) obj;
39     
40             return this.canonicalPath.equals(comp.getCanonicalPath());
41         }
42     
43         public String toString() {
44             return this.canonicalPath;
45         }
46     }
