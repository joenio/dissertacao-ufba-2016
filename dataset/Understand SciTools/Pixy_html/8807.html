51         protected AbstractTransferFunction makeBasicBlockTf(BasicBlock basicBlock, TacFunction traversedFunction) {
52             // we can override the general method from Analysis with this, because
53             // analysis information must not change inside basic blocks
54             // (all nodes inside a basic block should have an ID transfer function,
55             // so we can use this shortcut)
56             return TransferFunctionId.INSTANCE;
57         }
58     
59         protected AbstractTransferFunction includeStart(AbstractCfgNode cfgNodeX) {
60             return new Add(cfgNodeX, this);
61         }
62     
63         protected AbstractTransferFunction includeEnd(AbstractCfgNode cfgNodeX) {
64             return includeStart(cfgNodeX);
65         }
66     
67     //  ********************************************************************************
68     //  GET ****************************************************************************
69     //  ********************************************************************************
70     
71         // computes and returns the include chain list (consisting of cfg nodes)
72         // for the given cfg node
73         public LinkedList<AbstractCfgNode> getIncludeChain(AbstractCfgNode cfgNode) {
74             InclusionDominatorLatticeElement latElem = (InclusionDominatorLatticeElement) this.getAnalysisNode(cfgNode).getInValue();
75             List<AbstractCfgNode> dominators = latElem.getDominators();
76     
77             // * input: a list of (dominating) cfg nodes, both includeEnd and
78             //   includeStart
79             // * output: a chain of unclosed includeStart nodes
80             LinkedList<AbstractCfgNode> chain = new LinkedList<>();
81             for (AbstractCfgNode dominator : dominators) {
82                 if (dominator instanceof IncludeStart) {
83                     chain.add(dominator);
84                 } else if (dominator instanceof IncludeEnd) {
85                     IncludeEnd incEnd = (IncludeEnd) dominator;
86                     if (incEnd.isPeer(chain.getLast())) {
87                         chain.removeLast();
88                     } else {
89                         throw new RuntimeException("SNH");
90                     }
91                 } else {
92                     throw new RuntimeException("SNH");
93                 }
94             }
95             return chain;
96         }
