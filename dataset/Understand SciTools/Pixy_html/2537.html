50     
51         public List<Variable> getVariables() {
52             List<Variable> retMe = new LinkedList<>();
53             for (TacActualParameter param : this.paramList) {
54                 AbstractTacPlace paramPlace = param.getPlace();
55                 if (paramPlace instanceof Variable) {
56                     retMe.add((Variable) paramPlace);
57                 } else {
58                     retMe.add(null);
59                 }
60             }
61             return retMe;
62         }
