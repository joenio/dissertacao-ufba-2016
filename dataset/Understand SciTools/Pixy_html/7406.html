13     /**
14      * This is a helper class to prevent code redundancy and confusion.
15      *
16      * It is used in the Checker class.
17      *
18      * @author Nenad Jovanovic <enji@seclab.tuwien.ac.at>
19      */
20     public class GenericTaintAnalysis {
21         private List<AbstractVulnerabilityAnalysis> abstractVulnerabilityAnalyses;
22     
23         public DependencyAnalysis dependencyAnalysis;
24     
25         private GenericTaintAnalysis() {
26             this.abstractVulnerabilityAnalyses = new LinkedList<>();
27         }
28     
29         private void addDepClient(AbstractVulnerabilityAnalysis dependencyClient) {
30             this.abstractVulnerabilityAnalyses.add(dependencyClient);
31         }
32     
33         /**
34          * Returns null if the given taintString is illegal.
35          *
36          * @param tac
37          * @param enclosingAnalysis
38          * @param checker
39          * @param workList
40          * @param globalsModificationAnalysis
41          *
42          * @return
43          */
44         static GenericTaintAnalysis createAnalysis(
45             TacConverter tac, AbstractAnalysisType enclosingAnalysis, Checker checker, InterproceduralWorklist workList,
46             GlobalsModificationAnalysis globalsModificationAnalysis
47         ) {
48             GenericTaintAnalysis genericTaintAnalysis = new GenericTaintAnalysis();
49     
50             genericTaintAnalysis.dependencyAnalysis = new DependencyAnalysis(
51                 tac, checker.aliasAnalysis, checker.literalAnalysis, enclosingAnalysis, workList, globalsModificationAnalysis
52             );
53     
54             try {
55                 // each of the VulnerabilityAnalysis will get the dependencyAnalysis as parameter
56                 Class<?>[] argumentsClass = new Class<?>[] {
57                     Class.forName("at.ac.tuwien.infosys.www.pixy.analysis.dependency.DependencyAnalysis")
58                 };
59                 Object[] arguments = new Object[]{genericTaintAnalysis.dependencyAnalysis};
60     
61                 // for each requested VulnerabilityAnalysis ...
62                 for (VulnerabilityAnalysisInformation analysisInformation : MyOptions.getVulnerabilityAnalyses()) {
63                     if (!analysisInformation.performMe()) {
64                         continue;
65                     }
66                     Class<?> clientDefinition = Class.forName(analysisInformation.getClassName());
67                     Constructor<?> constructor = clientDefinition.getConstructor(argumentsClass);
68                     AbstractVulnerabilityAnalysis dependencyClient = (AbstractVulnerabilityAnalysis) constructor.newInstance(arguments);
69                     genericTaintAnalysis.addDepClient(dependencyClient);
70                 }
71             } catch (Exception e) {
72                 throw new RuntimeException(e);
73             }
74     
75             return genericTaintAnalysis;
76         }
77     
78         void analyze() {
79             this.dependencyAnalysis.analyze();
80     
81             // check for unreachable code
82             this.dependencyAnalysis.checkReachability();
83         }
84     
85         /**
86          * Detects vulnerabilities and returns a list with the line numbers of the detected vulnerabilities.
87          *
88          * @return the line numbers of the detected vulnerabilities
89          */
90         List<Integer> detectVulnerabilities() {
91             List<Integer> lineNumbersOfVulnerabilities = new LinkedList<>();
92             for (AbstractVulnerabilityAnalysis dependencyClient : this.abstractVulnerabilityAnalyses) {
93                 lineNumbersOfVulnerabilities.addAll(dependencyClient.detectVulnerabilities());
94             }
95     
96             return lineNumbersOfVulnerabilities;
97         }
98     
99         List<AbstractVulnerabilityAnalysis> getAbstractVulnerabilityAnalyses() {
100             return this.abstractVulnerabilityAnalyses;
101         }
