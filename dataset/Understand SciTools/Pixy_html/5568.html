11     /**
12      * Transfer function for function entries.
13      *
14      * @author Nenad Jovanovic <enji@seclab.tuwien.ac.at>
15      */
16     public class FunctionEntry extends AbstractTransferFunction {
17         private TacFunction function;
18     
19     // *********************************************************************************
20     // CONSTRUCTORS ********************************************************************
21     // *********************************************************************************
22     
23         public FunctionEntry(TacFunction function) {
24             this.function = function;
25         }
26     
27     // *********************************************************************************
28     // OTHER ***************************************************************************
29     // *********************************************************************************
30     
31         public AbstractLatticeElement transfer(AbstractLatticeElement inX) {
32     
33             LiteralLatticeElement in = (LiteralLatticeElement) inX;
34             LiteralLatticeElement out = new LiteralLatticeElement(in);
35     
36             // initialize g-shadows
37     
38             Map<Variable, Variable> globals2GShadows = this.function.getSymbolTable().getGlobals2GShadows();
39     
40             for (Map.Entry<Variable, Variable> entry : globals2GShadows.entrySet()) {
41                 Variable global = entry.getKey();
42                 Variable gShadow = entry.getValue();
43     
44                 // note: the TacConverter already took care that arrays and array elements
45                 // don't get shadow variables, so you don't have to check this here again
46     
47                 // initialize shadow to the literal of its original
48                 out.setShadow(gShadow, global);
49             }
50     
51             // initialize f-shadows
52     
53             Map<Variable, Variable> formals2FShadows = this.function.getSymbolTable().getFormals2FShadows();
54     
55             for (Map.Entry<Variable, Variable> entry : formals2FShadows.entrySet()) {
56                 Variable formal = entry.getKey();
57                 Variable fShadow = entry.getValue();
58     
59                 // initialize
60                 out.setShadow(fShadow, formal);
61             }
62     
63             return out;
64         }
65     }
