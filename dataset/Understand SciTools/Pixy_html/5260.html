28         public SymbolTable(String name, boolean isSuperSymTab) {
29             if (name == null) {
30                 throw new RuntimeException("SNH");
31             }
32             this.variables = new LinkedHashMap<>();
33             this.name = name;
34             this.isSuperSymTab = isSuperSymTab;
35             this.isMain = false;
36             this.globals2GShadows = null;
37             this.formals2FShadows = null;
38         }
39     
40         // use this constructor for non-supersymtabs
41         public SymbolTable(String name) {
42             this(name, false);
43             if (name.equals(InternalStrings.mainFunctionName)) {
44                 this.isMain = true;
45             }
46             this.globals2GShadows = new HashMap<>();
47             this.formals2FShadows = new HashMap<>();
48         }

127         // adds the variables of the given symbol table, changing their symbolTable
128         // member variable;
129         // WARNING: the current implementation of this method doesn't care about
130         // handling duplicates: it simply overwrites the mappings in this table;
131         // note that this doesn't affect the references to the overwritten variables
132         // from other places: in this case, they point to a variable that is no longer
133         // part of the symbol table! [but probably, this doesn't matter, since
134         // we don't use the == operator in connection with variables]
135         // LATER: for tackling this issue, you could use a separate variable
136         // repository for each ProgramConverter
137         // WARNING: the given symbol table must be trashed afterwards, since
138         // its contents are modified! if you need the original symbol table
139         // after this operation, you have to use a method that copies the original
140         // variables and adds the modified clones to the current symbol table
141         void addAll(SymbolTable table) {
142     
143             // the Variable objects' member field pointing at the enclosing symbol
144             // table has to be modified;
145             // this means that the following tempting line is not enough:
146             // this.variables.putAll(table.getVariables());
147     
148             // we can iterate over the keys since they are identical to the values
149             for (Variable variable : table.getVariables().keySet()) {
150                 variable.setSymbolTable(this);
151                 // adjust "local" and "global" attributes
152                 if (this.isMain) {
153                     variable.setIsGlobal();
154                 } else {
155                     variable.setIsLocal();
156                 }
157                 this.variables.put(variable, variable);
158             }
159         }
