66         public AbstractLatticeElement transfer(AbstractLatticeElement inX, AbstractContext context) {
67     
68             // lattice element entering the call prep node under the current
69             // context
70             LiteralLatticeElement origInfo =
71                 (LiteralLatticeElement) this.analysisNodeAtCallPrep.getPhiValue(context);
72     
73             if (origInfo == null) {
74                 throw new RuntimeException("SNH");
75             }
76     
77             // lattice element coming in from the callee (= base for interprocedural info);
78             // still contains the callee's locals
79             LiteralLatticeElement calleeIn = (LiteralLatticeElement) inX;
80     
81             // start only with default mappings
82             LiteralLatticeElement outInfo = new LiteralLatticeElement();
83     
84             // contains variables that have been tagged as visited
85             Set<Variable> visitedVars = new HashSet<>();
86     
87             // copy mappings of "global-like" places from calleeIn to outInfo
88             // ("global-like": globals, superglobals, and constants)
89             outInfo.copyGlobalLike(calleeIn);
90     
91             // LOCAL VARIABLES *************
92     
93             // no need to do this if the caller is main:
94             // its local variables are global variables
95             if (this.caller.isMain()) {
96                 this.handleReturnValue(calleeIn, outInfo);
97                 return outInfo;
98             }
99     
100             // initialize local variables with the mappings at call-time
101             outInfo.copyLocals(origInfo);
102     
103             // MUST WITH GLOBALS
104     
105             // for all local variables of the calling function
106             for (Variable localCallerVar : localCallerVars) {
107                 // an arbitrary global must-alias of this local
108                 Variable globalMustAlias = this.aliasAnalysis.getGlobalMustAlias(localCallerVar, this.prepNode);
109                 if (globalMustAlias == null) {
110                     continue;
111                 }
112     
113                 // the shadow of this global must-alias
114                 Variable globalMustAliasShadow = this.callee.getSymbolTable().getGShadow(globalMustAlias);
115     
116                 // set & mark
117                 if (globalMustAliasShadow == null) {
118                     System.out.println("call: " + this.caller.getName() + " -> " + this.callee.getName());
119                     throw new RuntimeException("no shadow for: " + globalMustAlias);
120                 }
121                 outInfo.setLocal(localCallerVar, calleeIn.getLiteral(globalMustAliasShadow));
122                 visitedVars.add(localCallerVar);
123             }
124     
125             // MUST WITH FORMALS
126     
127             // for each call-by-reference parameter pair
128             for (List<Variable> paramPair : this.cbrParams) {
129                 Iterator<Variable> paramPairIter = paramPair.iterator();
130                 Variable actualVar = paramPairIter.next();
131                 Variable formalVar = paramPairIter.next();
132     
133                 // local must-aliases of the actual parameter (including trivial ones,
134                 // so this set contains at least one element)
135                 Set<Variable> localMustAliases = this.aliasAnalysis.getLocalMustAliases(actualVar, this.prepNode);
136     
137                 for (Variable localMustAlias : localMustAliases) {
138                     // no need to handle visited variables again
139                     if (visitedVars.contains(localMustAlias)) {
140                         continue;
141                     }
142     
143                     // the formal parameter's f-shadow
144                     Variable fShadow = this.callee.getSymbolTable().getFShadow(formalVar);
145     
146                     // set & mark
147                     outInfo.setLocal(localMustAlias, calleeIn.getLiteral(fShadow));
148                     visitedVars.add(localMustAlias);
149                 }
150             }
151     
152             // MAY WITH GLOBALS
153     
154             // for each local variable that was not visited yet
155             for (Variable localCallerVar : localCallerVars) {
156                 if (visitedVars.contains(localCallerVar)) {
157                     continue;
158                 }
159     
160                 // global may-aliases of this variable (at the time of call)
161                 Set<Variable> globalMayAliases = this.aliasAnalysis.getGlobalMayAliases(localCallerVar, this.prepNode);
162     
163                 // if there are no such aliases: nothing to do
164                 if (globalMayAliases.isEmpty()) {
165                     continue;
166                 }
167     
168                 // initialize this variable's literal with its original literal
169                 Literal computedLit = origInfo.getLiteral(localCallerVar);
170     
171                 // for all these global may-aliases...
172                 for (Variable globalMayAlias : globalMayAliases) {
173                     // its g-shadow
174                     Variable globalMayAliasShadow = this.callee.getSymbolTable().getGShadow(globalMayAlias);
175     
176                     // the shadow's literal (from flowback-info)
177                     Literal shadowLit = calleeIn.getLiteral(globalMayAliasShadow);
178     
179                     // lub
180                     computedLit = LiteralLatticeElement.lub(computedLit, shadowLit);
181                 }
182     
183                 // set the local's literal to the computed literal in outInfo
184                 outInfo.setLocal(localCallerVar, computedLit);
185     
186                 // DON'T mark it as visited
187             }
188     
189             // MAY WITH FORMALS
190     
191             // for each call-by-reference parameter pair
192             for (List<Variable> paramPair : this.cbrParams) {
193                 Iterator<Variable> paramPairIter = paramPair.iterator();
194                 Variable actualVar = paramPairIter.next();
195                 Variable formalVar = paramPairIter.next();
196     
197                 // local may-aliases of the actual parameter (at call-time)
198                 Set<Variable> localMayAliases = this.aliasAnalysis.getLocalMayAliases(actualVar, this.prepNode);
199     
200                 // for each such may-alias that was not visited yet
201                 for (Variable localMayAlias : localMayAliases) {
202                     if (visitedVars.contains(localMayAlias)) {
203                         continue;
204                     }
205     
206                     // the current literal of the local may-alias in output-info
207                     Literal localLit = outInfo.getLiteral(localMayAlias);
208     
209                     // the formal parameter's f-shadow
210                     Variable fShadow = this.callee.getSymbolTable().getFShadow(formalVar);
211     
212                     // the shadow's literal (from flowback-info)
213                     Literal shadowLit = calleeIn.getLiteral(fShadow);
214     
215                     // lub & set
216                     Literal newLit = LiteralLatticeElement.lub(localLit, shadowLit);
217                     outInfo.setLocal(localMayAlias, newLit);
218                 }
219             }
220     
221             this.handleReturnValue(calleeIn, outInfo);
222     
223             return outInfo;
224         }
225     
226         private void handleReturnValue(LiteralLatticeElement calleeIn, LiteralLatticeElement outInfo) {
227     
228             // literal of the return variable at the end of the called function
229             Literal retLit = calleeIn.getLiteral(this.retNode.getRetVar());
230     
231             // assign this literal to the return node's temporary
232             // and clear the return variable afterwards
233             outInfo.handleReturnValue(this.retNode.getTempVar(), retLit, this.retNode.getRetVar());
234         }
