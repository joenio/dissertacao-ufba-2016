229         // returns the automaton representation of the given dependency graph;
230         // is done by decorating the nodes of the graph with automata bottom-up,
231         // and returning the automaton that eventually decorates the root;
232         // BEWARE: this also eliminates cycles!
233         Automaton toAutomaton(DependencyGraph dependencyGraph, DependencyGraph origDependencyGraph) {
234             dependencyGraph.eliminateCycles();
235             AbstractNode root = dependencyGraph.getRootNode();
236             Map<AbstractNode, Automaton> deco = new HashMap<>();
237             Set<AbstractNode> visited = new HashSet<>();
238             this.decorate(root, deco, visited, dependencyGraph, origDependencyGraph);
239             Automaton rootDeco = deco.get(root).clone();
240     
241             return rootDeco;
242         }

246         // decorates the given node (and all its successors) with an automaton
247         private void decorate(
248             AbstractNode node, Map<AbstractNode, Automaton> deco,
249             Set<AbstractNode> visited, DependencyGraph dependencyGraph, DependencyGraph origDependencyGraph
250         ) {
251             visited.add(node);
252     
253             // if this node has successors, decorate them first (if not done yet)
254             List<AbstractNode> successors = dependencyGraph.getSuccessors(node);
255             if (successors != null && !successors.isEmpty()) {
256                 for (AbstractNode succ : successors) {
257                     if (!visited.contains(succ) && deco.get(succ) == null) {
258                         decorate(succ, deco, visited, dependencyGraph, origDependencyGraph);
259                     }
260                 }
261             }
262     
263             // now that all successors are decorated, we can decorate this node
264             Automaton auto = null;
265             if (node instanceof NormalNode) {
266                 NormalNode normalNode = (NormalNode) node;
267                 if (successors == null || successors.isEmpty()) {
268                     // this should be a string leaf node
269                     AbstractTacPlace place = normalNode.getPlace();
270                     if (place.isLiteral()) {
271                         auto = Automaton.makeString(place.toString());
272                     } else {
273                         // this case should not happen any longer (now that
274                         // we have "uninit" nodes, see below)
275                         throw new RuntimeException("SNH: " + place + ", " + normalNode.getCfgNode().getFileName() + "," +
276                             normalNode.getCfgNode().getOriginalLineNumber());
277                     }
278                 } else {
279                     // this is an interior node, not a leaf node;
280                     // the automaton for this node is the union of all the
281                     // successor automatas
282                     for (AbstractNode succ : successors) {
283                         if (succ == node) {
284                             // a simple loop, can be ignored
285                             continue;
286                         }
287                         Automaton succAuto = deco.get(succ);
288                         if (succAuto == null) {
289                             throw new RuntimeException("SNH");
290                         }
291                         if (auto == null) {
292                             auto = succAuto;  // cloning not necessary here
293                         } else {
294                             auto = auto.union(succAuto);
295                         }
296                     }
297                 }
298             } else if (node instanceof BuiltinFunctionNode) {
299                 auto = this.makeAutoForOp((BuiltinFunctionNode) node, deco, dependencyGraph);
300             } else if (node instanceof CompleteGraphNode) {
301                 // for SCC nodes, we generate a coarse string approximation (.* automaton);
302                 // the taint value depends on the taint value of the successors:
303                 // if any of the successors is tainted in any way, we make the resulting
304                 // automaton tainted as well
305     
306                 /*
307                  * this approach works under the assumption that the SCC contains
308                  * no "evil" functions (functions that always return a tainted value),
309                  * and that the direct successors of the SCC are not <uninit> nodes
310                  * (see there, below);
311                  * it is primarily based on the fact that SCCs can never contain
312                  * <uninit> nodes, because <uninit> nodes are always leaf nodes in the dependency
313                  * graph (since they have no successors);
314                  * under the above assumptions and observations, it is valid to
315                  * say that the taint value of an SCC node solely depends on
316                  * the taint values of its successors, and that is exactly what we
317                  * do here
318                  *
319                  */
320     
321                 Transition.Taint taint = Transition.Taint.Untainted;
322                 for (AbstractNode successor : successors) {
323                     if (successor == node) {
324                         // a simple loop, should be part of the SCC
325                         throw new RuntimeException("SNH");
326                     }
327                     Automaton succAuto = deco.get(successor);
328                     if (succAuto == null) {
329                         throw new RuntimeException("SNH");
330                     }
331                     if (succAuto.hasTaintedTransitions()) {
332                         taint = Transition.Taint.Directly;
333                         break;
334                     }
335                 }
336     
337                 auto = Automaton.makeAnyString(taint);
338             } else if (node instanceof UninitializedNode) {
339                 // retrieve predecessor
340                 Set<AbstractNode> preds = dependencyGraph.getPredecessors(node);
341                 if (preds.size() != 1) {
342                     throw new RuntimeException("SNH");
343                 }
344                 AbstractNode pre = preds.iterator().next();
345     
346                 if (pre instanceof NormalNode) {
347                     NormalNode preNormal = (NormalNode) pre;
348                     switch (this.getInitialTaintForPlace(preNormal.getPlace())) {
349                         case ALWAYS:
350                         case IF_REGISTER_GLOBALS:
351                             auto = Automaton.makeAnyString(Transition.Taint.Directly);
352                             break;
353                         case NEVER:
354                             auto = Automaton.makeAnyString(Transition.Taint.Untainted);
355                             break;
356                         default:
357                             throw new RuntimeException("SNH");
358                     }
359                 } else if (pre instanceof CompleteGraphNode) {
360                     // this case can really happen (e.g.: dcpportal: advertiser.php, forums.php);
361     
362                     // take a look at the "real" predecessors (i.e., take a look "into"
363                     // the SCC node): if there is exactly one predecessor, namely a
364                     // NormalNode, and if the contained place is initially untainted,
365                     // there is no danger from here; else: we will have to set it to tainted
366                     Set<AbstractNode> origPreds = origDependencyGraph.getPredecessors(node);
367                     if (origPreds.size() == 1) {
368                         AbstractNode origPre = origPreds.iterator().next();
369                         if (origPre instanceof NormalNode) {
370                             NormalNode origPreNormal = (NormalNode) origPre;
371     
372                             switch (this.getInitialTaintForPlace(origPreNormal.getPlace())) {
373                                 case ALWAYS:
374                                 case IF_REGISTER_GLOBALS:
375                                     auto = Automaton.makeAnyString(Transition.Taint.Directly);
376                                     break;
377                                 case NEVER:
378                                     auto = Automaton.makeAnyString(Transition.Taint.Untainted);
379                                     break;
380                                 default:
381                                     throw new RuntimeException("SNH");
382                             }
383                         } else {
384                             auto = Automaton.makeAnyString(Transition.Taint.Directly);
385                         }
386                     } else {
387                         // conservative decision for this SCC
388                         auto = Automaton.makeAnyString(Transition.Taint.Directly);
389                     }
390                 } else {
391                     throw new RuntimeException("SNH: " + pre.getClass());
392                 }
393             } else {
394                 throw new RuntimeException("SNH");
395             }
396     
397             if (auto == null) {
398                 throw new RuntimeException("SNH");
399             }
400     
401             deco.put(node, auto);
402         }

404         /**
405          * Returns an automaton for the given operation node.
406          *
407          * @param node
408          * @param deco
409          * @param dependencyGraph
410          *
411          * @return
412          */
413         private Automaton makeAutoForOp(
414             BuiltinFunctionNode node, Map<AbstractNode, Automaton> deco, DependencyGraph dependencyGraph
415         ) {
416             List<AbstractNode> successors = dependencyGraph.getSuccessors(node);
417             if (successors == null) {
418                 successors = new LinkedList<>();
419             }
420     
421             Automaton retMe = null;
422     
423             String opName = node.getName();
424     
425             List<Integer> multiList = new LinkedList<>();
426     
427             if (!node.isBuiltin()) {
428                 // call to function or method for which no definition
429                 // could be found
430     
431                 AbstractCfgNode cfgNodeX = node.getCfgNode();
432                 if (cfgNodeX instanceof CallUnknownFunction) {
433                     CallUnknownFunction cfgNode = (CallUnknownFunction) cfgNodeX;
434                     if (cfgNode.isMethod()) {
435                         retMe = Automaton.makeAnyString(Transition.Taint.Untainted);
436                     } else {
437                         retMe = Automaton.makeAnyString(Transition.Taint.Directly);
438                     }
439                 } else {
440                     throw new RuntimeException("SNH");
441                 }
442             } else if (opName.equals(".")) {
443                 // CONCAT
444                 for (AbstractNode succ : successors) {
445                     Automaton succAuto = deco.get(succ);
446                     if (retMe == null) {
447                         retMe = succAuto;
448                     } else {
449                         retMe = retMe.concatenate(succAuto);
450                     }
451                 }
452     
453                 // WEAK SANITIZATION FUNCTIONS *******************************
454                 // ops that perform sanitization, but which are insufficient
455                 // in cases where the output is not enclosed by quotes in an SQL query
456             } else if (isWeakSanitation(opName, multiList)) {
457                 retMe = Automaton.makeAnyString(Transition.Taint.Indirectly);
458     
459                 // STRONG SANITIZATION FUNCTIONS *******************************
460                 // e.g., ops that return numeric values
461             } else if (isStrongSanitation(opName)) {
462                 retMe = Automaton.makeAnyString(Transition.Taint.Untainted);
463     
464                 // EVIL FUNCTIONS ***************************************
465                 // take care: if you define evil functions, you must adjust
466                 // the treatment of SCC nodes in decorate()
467     
468                 // MULTI-OR-DEPENDENCY **********************************
469             } else if (useTransducers && opName.equals("str_replace")) {
470                 if (successors.size() < 3) {
471                     throw new RuntimeException("SNH");
472                 }
473                 Automaton searchAuto = deco.get(successors.get(0));
474                 Automaton replaceAuto = deco.get(successors.get(1));
475                 Automaton subjectAuto = deco.get(successors.get(2));
476     
477                 // search and replace have to be finite strings;
478                 // extract them
479                 boolean supported = true;
480                 String searchString = null;
481                 String replaceString = null;
482                 if (searchAuto.isFinite() && replaceAuto.isFinite()) {
483                     Set<String> searchSet = searchAuto.getFiniteStrings();
484                     Set<String> replaceSet = replaceAuto.getFiniteStrings();
485                     if (searchSet.size() != 1 || replaceSet.size() != 1) {
486                         supported = false;
487                     } else {
488                         searchString = searchSet.iterator().next();
489                         replaceString = replaceSet.iterator().next();
490                     }
491                 } else {
492                     supported = false;
493                 }
494                 if (!supported) {
495                     throw new RuntimeException("not supported yet");
496                 }
497     
498                 Automaton transduced = new MyTransductions().str_replace(searchString, replaceString, subjectAuto);
499                 return transduced;
500             } else if (isMultiDependencyOperation(opName, multiList)) {
501                 Transition.Taint taint = this.multiDependencyAuto(successors, deco, multiList, false);
502                 retMe = Automaton.makeAnyString(taint);
503             } else if (isInverseMultiDependencyOperation(opName, multiList)) {
504                 Transition.Taint taint = this.multiDependencyAuto(successors, deco, multiList, true);
505                 retMe = Automaton.makeAnyString(taint);
506     
507                 // CATCH-ALL ********************************************
508             } else {
509                 System.out.println("Unmodeled builtin function (SQL): " + opName);
510     
511                 // conservative decision for operations that have not been
512                 // modeled yet: .*
513                 retMe = Automaton.makeAnyString(Transition.Taint.Directly);
514             }
515     
516             return retMe;
517         }

519         /**
520          * Checks if the given node (inside the given function) is a sensitive sink.
521          *
522          * Adds an appropriate sink object to the given list if it is a sink.
523          *
524          * @param cfgNodeX
525          * @param traversedFunction
526          * @param sinks
527          */
528         protected void checkForSink(AbstractCfgNode cfgNodeX, TacFunction traversedFunction, List<Sink> sinks) {
529             if (cfgNodeX instanceof CallBuiltinFunction) {
530                 // builtin function sinks
531                 CallBuiltinFunction cfgNode = (CallBuiltinFunction) cfgNodeX;
532                 String functionName = cfgNode.getFunctionName();
533     
534                 checkForSinkHelper(functionName, cfgNode, cfgNode.getParamList(), traversedFunction, sinks);
535             } else if (cfgNodeX instanceof CallPreparation) {
536                 CallPreparation cfgNode = (CallPreparation) cfgNodeX;
537                 String functionName = cfgNode.getFunctionNamePlace().toString();
538     
539                 // user-defined custom sinks
540     
541                 checkForSinkHelper(functionName, cfgNode, cfgNode.getParamList(), traversedFunction, sinks);
542             }
543         }

545         private void checkForSinkHelper(
546             String functionName, AbstractCfgNode cfgNode,
547             List<TacActualParameter> paramList, TacFunction traversedFunction, List<Sink> sinks
548         ) {
549             if (this.vulnerabilityAnalysisInformation.getSinks().containsKey(functionName)) {
550                 Sink sink = new Sink(cfgNode, traversedFunction);
551                 for (Integer param : this.vulnerabilityAnalysisInformation.getSinks().get(functionName)) {
552                     if (paramList.size() > param) {
553                         sink.addSensitivePlace(paramList.get(param).getPlace());
554                         // add this sink to the list of sensitive sinks
555                         sinks.add(sink);
556                     }
557                 }
558             }
559         }

561         private Transition.Taint multiDependencyAuto(
562             List<AbstractNode> successors, Map<AbstractNode, Automaton> deco, List<Integer> indices, boolean inverse
563         ) {
564             boolean indirectly = false;
565             Set<Integer> indexSet = new HashSet<>(indices);
566     
567             int count = -1;
568             for (AbstractNode successor : successors) {
569                 count++;
570     
571                 // check if there is a dependency on this successor
572                 if (inverse) {
573                     if (indexSet.contains(count)) {
574                         continue;
575                     }
576                 } else {
577                     if (!indexSet.contains(count)) {
578                         continue;
579                     }
580                 }
581     
582                 Automaton successorAutomaton = deco.get(successor);
583                 if (successorAutomaton == null) {
584                     throw new RuntimeException("SNH");
585                 }
586                 if (successorAutomaton.hasDirectlyTaintedTransitions()) {
587                     return Transition.Taint.Directly;
588                 }
589                 if (successorAutomaton.hasIndirectlyTaintedTransitions()) {
590                     indirectly = true;
591                 }
592             }
593     
594             if (indirectly) {
595                 return Transition.Taint.Indirectly;
596             } else {
597                 // no tainted successors have been found
598                 return Transition.Taint.Untainted;
599             }
600         }
