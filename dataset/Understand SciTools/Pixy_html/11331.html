71         // returns true if the given varname is $_SERVER[...]
72         // or $HTTP_SERVER_VARS[...], with ... == one of the above indices
73         public static boolean isHarmlessServerVar(String varName) {
74     
75             // extract index name
76             String index;
77             if (varName.startsWith("$_SERVER[") && varName.endsWith("]")) {
78                 index = varName.substring(9, varName.length() - 1);
79             } else if (varName.startsWith("$HTTP_SERVER_VARS[") && varName.endsWith("]")) {
80                 index = varName.substring(18, varName.length() - 1);
81             } else {
82                 return false;
83             }
84     
85             // compare index name
86             return harmlessServerIndices.contains(index);
87         }

102         /**
103          * Reads models for builtin PHP functions.
104          *
105          * @param dci
106          *
107          * @return
108          */
109         private static FunctionModels readModelFile(VulnerabilityAnalysisInformation dci) {
110             Set<String> f_evil = new HashSet<>();
111             Map<String, Set<Integer>> f_multi = new HashMap<>();
112             Map<String, Set<Integer>> f_invMulti = new HashMap<>();
113             Set<String> f_strongSanit = new HashSet<>();
114             Map<String, Set<Integer>> f_weakSanit = new HashMap<>();
115     
116             String strongSanitMarker = "0";
117             String weakSanitMarker = "1";
118             String multiMarker = "2";
119             String invMultiMarker = "3";
120             String evilMarker = "4";
121     
122             // read file into properties object
123             String modelFileName = MyOptions.pixyHome + "/" + MyOptions.configurationDirectory + "/model_" + dci.getName() + ".txt";
124             File modelFile = new File(modelFileName);
125             Properties sinkProperties = new Properties();
126             try {
127                 FileInputStream in = new FileInputStream(modelFile);
128                 sinkProperties.load(in);
129                 in.close();
130             } catch (FileNotFoundException e) {
131                 Utils.bail("Error: Can't find configuration file: " + modelFileName);
132             } catch (IOException e) {
133                 Utils.bail("Error: I/O exception while reading configuration file:" + modelFileName,
134                     e.getMessage());
135             }
136     
137             Class<?> tacOps;
138             try {
139                 tacOps = Class.forName("at.ac.tuwien.infosys.www.pixy.conversion.TacOperators");
140             } catch (ClassNotFoundException e1) {
141                 throw new RuntimeException("SNH");
142             }
143     
144             // convert properties...
145             for (Map.Entry<Object, Object> propsEntry : sinkProperties.entrySet()) {
146                 String functionName = ((String) propsEntry.getKey()).trim();
147                 String functionList = (String) propsEntry.getValue();
148     
149                 if (!BuiltinFunctions.isBuiltinFunction(functionName)) {
150                     if (functionName.startsWith("op(") && functionName.endsWith(")")) {
151                         functionName = functionName.substring(3, functionName.length() - 1);
152                         // convert the operator name to its symbol
153                         try {
154                             Field field = tacOps.getDeclaredField(functionName);
155                             functionName = TacOperators.opToName(field.getInt(null));
156                         } catch (NoSuchFieldException e) {
157                             Utils.bail("Error: Non-builtin function in config file: " + functionName);
158                             continue;
159                         } catch (IllegalAccessException e) {
160                             Utils.bail("Error: Non-builtin function in config file: " + functionName);
161                         }
162                     } else {
163                         Utils.bail("Error: Non-builtin function in config file: " + functionName);
164                     }
165                 }
166     
167                 StringTokenizer funcTokenizer = new StringTokenizer(functionList, ":");
168                 if (!funcTokenizer.hasMoreTokens()) {
169                     Utils.bail("Error: Missing type for builtin function model: " + functionName);
170                 }
171     
172                 String type = funcTokenizer.nextToken().trim();
173                 if (type.equals(strongSanitMarker)) {
174                     // strong sanitization
175                     f_strongSanit.add(functionName);
176                 } else if (type.equals(weakSanitMarker)) {
177                     // weak sanitization
178                     Set<Integer> params = new HashSet<>();
179                     while (funcTokenizer.hasMoreTokens()) {
180                         String param = funcTokenizer.nextToken().trim();
181                         try {
182                             params.add(Integer.valueOf(param));
183                         } catch (NumberFormatException e) {
184                             Utils.bail("Error: Illegal parameter for builtin function model: " + functionName);
185                         }
186                     }
187                     f_weakSanit.put(functionName, params);
188                 } else if (type.equals(multiMarker)) {
189                     // multi-dependency
190                     Set<Integer> params = new HashSet<>();
191                     while (funcTokenizer.hasMoreTokens()) {
192                         String param = funcTokenizer.nextToken().trim();
193                         try {
194                             params.add(Integer.valueOf(param));
195                         } catch (NumberFormatException e) {
196                             Utils.bail("Error: Illegal parameter for builtin function model: " + functionName);
197                         }
198                     }
199                     f_multi.put(functionName, params);
200                 } else if (type.equals(invMultiMarker)) {
201                     // inverse multi-dependency
202                     Set<Integer> params = new HashSet<>();
203                     while (funcTokenizer.hasMoreTokens()) {
204                         String param = funcTokenizer.nextToken().trim();
205                         try {
206                             params.add(Integer.valueOf(param));
207                         } catch (NumberFormatException e) {
208                             Utils.bail("Error: Illegal parameter for builtin function model: " + functionName);
209                         }
210                     }
211                     f_invMulti.put(functionName, params);
212                 } else if (type.equals(evilMarker)) {
213                     // evil functions
214                     f_evil.add(functionName);
215                 } else {
216                     Utils.bail("Error: Unknown type for builtin function model: " + functionName);
217                 }
218             }
219     
220             // add Pixy's suppression function
221             f_strongSanit.add(InternalStrings.suppression);
222     
223             return new FunctionModels(f_evil, f_multi, f_invMulti, f_strongSanit, f_weakSanit);
224         }

237         // reads the given sink file, fills the given map with the contained info,
238         // and returns the contained "sinkType" property (can also be null)
239         private static String readSinkFile(String sinkFileName,
240                                            Map<String, Set<Integer>> sinks) {
241     
242             // read file into properties object
243             File sinkFile = new File(sinkFileName);
244             Properties sinkProps = new Properties();
245             //System.out.println("Reading sink config file: " + sinkFileName);
246             try {
247                 FileInputStream in = new FileInputStream(sinkFile);
248                 sinkProps.load(in);
249                 in.close();
250             } catch (FileNotFoundException e) {
251                 System.out.println("Warning: Can't find sink configuration file: " + sinkFileName);
252             } catch (IOException e) {
253                 System.out.println("Warning: I/O exception while reading configuration file:" + sinkFileName);
254                 System.out.println(e.getMessage());
255             }
256     
257             // convert properties into the above map
258             String sinkType = null;
259             for (Map.Entry<Object, Object> propsEntry : sinkProps.entrySet()) {
260                 String functionName = (String) propsEntry.getKey();
261                 String params = (String) propsEntry.getValue();
262     
263                 // the entry with this name is special: it denotes the analysis type
264                 // for which the sinks shall be adjusted
265                 if (functionName.equals("sinkType")) {
266                     sinkType = params;
267                     continue;
268                 }
269     
270                 // convert parameter list for this sink
271                 StringTokenizer paramTokenizer = new StringTokenizer(params, ":");
272                 int numTokens = paramTokenizer.countTokens();
273                 Set<Integer> paramSet = new HashSet<>();
274                 while (paramTokenizer.hasMoreTokens()) {
275                     String param = paramTokenizer.nextToken();
276                     try {
277                         paramSet.add(Integer.parseInt(param));
278                     } catch (NumberFormatException e) {
279                         System.out.println("Warning: Illegal param argument for " + functionName + " in " + sinkFileName);
280                         System.out.println("param: " + param);
281                     }
282                 }
283                 if (numTokens == 0) {
284                     // e.g., for printf (XSS): all params are sensitive
285                     paramSet = null;
286                 }
287     
288                 sinks.put(functionName, paramSet);
289             }
290     
291             return sinkType;
292         }

305         public static void readCustomSinkFiles() {
306     
307             if (MyOptions.optionS != null) {
308     
309                 // "sink file name" tokenizer
310                 StringTokenizer sfnTokenizer = new StringTokenizer(MyOptions.optionS, ":");
311     
312                 // for each given sink file...
313                 while (sfnTokenizer.hasMoreTokens()) {
314                     String sinkFileName = sfnTokenizer.nextToken();
315     
316                     // fill this map with the information contained in the file
317                     Map<String, Set<Integer>> sinks = new HashMap<>();
318     
319                     String sinkType = readSinkFile(sinkFileName, sinks);
320     
321                     if (sinkType == null) {
322                         System.out.println("Missing sinkType in file " + sinkFileName);
323                     } else {
324                         VulnerabilityAnalysisInformation dci = name2Analysis.get(sinkType);
325                         if (dci == null) {
326                             System.out.println("Invalid sinkType in file " + sinkFileName);
327                             System.out.println("- " + sinkType);
328                         } else {
329                             dci.addSinks(sinks);
330                         }
331                     }
332                 }
333             }

350         // flags requested analysis
351         public static boolean setAnalyses(String taintStrings) {
352     
353             if (taintStrings == null) {
354                 // no analyses are requested, which is also OK
355                 return true;
356             }
357     
358             StringTokenizer st = new StringTokenizer(taintStrings, ":");
359             while (st.hasMoreTokens()) {
360     
361                 String taintString = st.nextToken();
362     
363                 VulnerabilityAnalysisInformation dci = name2Analysis.get(taintString);
364                 if (dci != null) {
365                     dci.setPerformMe(true);
366                 } else {
367                     System.out.println("Invalid analysis type: " + taintString + ".");
368                     System.out.println("Choose one of the following: ");
369                     StringBuilder b = new StringBuilder();
370                     for (String name : name2Analysis.keySet()) {
371                         b.append(name + ", ");
372                     }
373                     if (!name2Analysis.isEmpty()) {
374                         b.deleteCharAt(b.length() - 1);
375                         b.deleteCharAt(b.length() - 1);
376                     }
377                     System.out.println(b.toString());
378                     return false;
379                 }
380             }
381             return true;
382         }
