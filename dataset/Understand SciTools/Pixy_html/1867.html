15     public class LiteralLattice extends AbstractLattice {
16         public LiteralLattice(
17             List<AbstractTacPlace> places, ConstantsTable constantsTable, List<TacFunction> functions, SymbolTable superSymbolTable
18         ) {
19             LiteralLatticeElement.initDefault(places, constantsTable, functions, superSymbolTable);
20         }
21     
22         public AbstractLatticeElement lub(
23             AbstractLatticeElement incomingElementX,
24             AbstractLatticeElement targetElementX) {
25     
26             // if the incoming element is the bottom element:
27             // return (a clone of) the other element
28             if (incomingElementX == this.bottom) {
29                 if (targetElementX != this.bottom) {
30                     return targetElementX.cloneMe();
31                 } else {
32                     return this.bottom;
33                 }
34             }
35     
36             // if the target element is the bottom element:
37             // return (a clone of) the other element
38             if (targetElementX == this.bottom) {
39                 return incomingElementX.cloneMe();
40             }
41     
42             // if one of the elements is the top element: return the top element;
43             // not necessary here: we will never encounter the top element
44     
45             // class cast
46             LiteralLatticeElement incomingElement = (LiteralLatticeElement) incomingElementX;
47             LiteralLatticeElement targetElement = (LiteralLatticeElement) targetElementX;
48     
49             // initialize the resulting lattice element as clone of the target
50             // lattice element (we don't want to modify the target lattice element
51             // given by the caller of this method);
52             // EFF: note that we must not modify the incoming element either: while
53             // most transfer functions generate a new lattice element object which
54             // could be reused, the ID transfer function simply passes on the
55             // reference; if you reuse here, the ID transfer function must return
56             // a new element (not clear which method is more efficient)
57             LiteralLatticeElement resultElement = new LiteralLatticeElement(targetElement);
58     
59             // lub the incoming element over the clone of the target element
60             resultElement.lub(incomingElement);
61     
62             return resultElement;
63         }
