11     /**
12      * Transfer function for simple assignment nodes.
13      *
14      * @author Nenad Jovanovic <enji@seclab.tuwien.ac.at>
15      */
16     public class Assign extends AbstractTransferFunction {
17         private Variable left;
18         private Variable right;
19     
20         private boolean supported;
21         private AliasAnalysis aliasAnalysis;
22     
23     // *********************************************************************************
24     // CONSTRUCTORS ********************************************************************
25     // *********************************************************************************
26     
27         public Assign(AbstractTacPlace left, AbstractTacPlace right, AliasAnalysis aliasAnalysis, AbstractCfgNode cfgNode) {
28     
29             // both arguments are variables if the PHP input is correct
30             this.left = (Variable) left;
31             this.right = (Variable) right;
32     
33             this.aliasAnalysis = aliasAnalysis;
34     
35             // check for unsupported features
36             this.supported =
37                 AliasAnalysis.isSupported(this.left, this.right, true, cfgNode.getOriginalLineNumber());
38         }
39     
40     // *********************************************************************************
41     // OTHER ***************************************************************************
42     // *********************************************************************************
43     
44         public AbstractLatticeElement transfer(AbstractLatticeElement inX) {
45     
46             // ignore unsupported operations
47             if (!this.supported) {
48                 return inX;
49             }
50     
51             // ignore useless statements like "$a =& $a"
52             if (this.left == this.right) {
53                 return inX;
54             }
55     
56             AliasLatticeElement in = (AliasLatticeElement) inX;
57             AliasLatticeElement out = new AliasLatticeElement(in);
58     
59             // perform redirect operation on "out"
60             out.redirect(this.left, this.right);
61     
62             // recycle
63             out = (AliasLatticeElement) this.aliasAnalysis.recycle(out);
64     
65             return out;
66         }
67     }
