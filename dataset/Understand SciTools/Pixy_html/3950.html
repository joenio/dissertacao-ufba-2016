3     import at.ac.tuwien.infosys.www.pixy.conversion.AbstractTacPlace;
4     import at.ac.tuwien.infosys.www.pixy.conversion.TacFunction;
5     import at.ac.tuwien.infosys.www.pixy.conversion.cfgnodes.AbstractCfgNode;

7     import java.util.LinkedList;
8     import java.util.List;
9     
10     /**
11      * This class represents a sink.
12      *
13      * Note: This class has a natural ordering that is inconsistent with equals().
14      *
15      * @author Nenad Jovanovic <enji@seclab.tuwien.ac.at>
16      */
17     public class Sink implements Comparable<Sink> {
18         /**
19          * a list of sensitive places (i.e., places from which this sink uses data)
20          */
21         private List<AbstractTacPlace> sensitivePlaces;
22         private AbstractCfgNode cfgNode;
23         private int lineNumber = -1;
24     
25         /** function containing this sink */
26         private TacFunction function;
27     
28         public Sink(AbstractCfgNode cfgNode, TacFunction function) {
29             this.cfgNode = cfgNode;
30             this.sensitivePlaces = new LinkedList<>();
31             this.function = function;
32         }
33     
34         AbstractCfgNode getNode() {
35             return this.cfgNode;
36         }
37     
38         public int getLineNumber() {
39             if (this.lineNumber == -1) {
40                 this.lineNumber = this.cfgNode.getOriginalLineNumber();
41             }
42             return this.lineNumber;
43         }
44     
45         public void addSensitivePlace(AbstractTacPlace place) {
46             this.sensitivePlaces.add(place);
47         }
48     
49         /**
50          * Returns a list of SinkProblems for each sensitive place in this sink.
51          *
52          * @return
53          */
54         List<SinkProblem> getSinkProblems() {
55             List<SinkProblem> problems = new LinkedList<>();
56     
57             for (AbstractTacPlace sensitivePlace : this.sensitivePlaces) {
58                 /** list of call nodes that call the function containing the sink */
59                 List<AbstractCfgNode> calledBy = new LinkedList<>();
60     
61                 SinkProblem problem = new SinkProblem(sensitivePlace);
62                 problem.setCallList(calledBy);
63                 problems.add(problem);
64             }
65     
66             return problems;
67         }
68     
69         /**
70          * Comparison regarding line number.
71          *
72          * @param otherSink
73          *
74          * @return
75          */
76         public int compareTo(Sink otherSink) {
77             int myLineNo = this.getLineNumber();
78             int compLineNo = otherSink.getLineNumber();
79             if (myLineNo < compLineNo) {
80                 return -1;
81             } else if (myLineNo == compLineNo) {
82                 return 0;
83             } else {
84                 return 1;
85             }
86         }
87     }
