17         public static FSAAutomaton reg_replace(FSAAutomaton phpPatternAuto,
18                                                FSAAutomaton replaceAuto, FSAAutomaton subjectAuto, boolean preg,
19                                                AbstractCfgNode cfgNode) {
20     
21             // get a finite string from the pattern
22             List<String> finitePattern = phpPatternAuto.getFiniteString();
23             if (finitePattern == null) {
24                 // the pattern is not a finite string, not supported
25                 return FSAAutomaton.makeAnyString();
26             }
27             if (finitePattern.isEmpty()) {
28                 // the pattern is empty for some reason;
29                 // e.g., happens if the $pattern argument to preg_replace is an array
30                 // => be conservative
31                 return FSAAutomaton.makeAnyString();
32             }
33     
34             // convert the PHP pattern into a prolog regexp
35             boolean approximate = false;
36             FSAAutomaton prologPatternAuto = null;
37             try {
38                 prologPatternAuto = FSAAutomaton.convertPhpRegex(finitePattern, preg);
39             } catch (UnsupportedRegexException e) {
40                 // if the regex is not supported yet: just return .*
41                 System.err.println("unsupported regex:");
42                 System.err.println("- " + cfgNode.getLoc());
43                 approximate = true;
44             } catch (Exception e) {
45                 // if anything else goes wrong:
46                 // - also return .*
47                 // - but generate a different warning
48                 System.err.println("Exception during regex conversion");
49                 System.err.println("- " + cfgNode.getLoc());
50                 System.err.println(e.getMessage());
51                 e.printStackTrace();
52                 approximate = true;
53             }
54     
55             FSAAutomaton retMe;
56             if (approximate) {
57                 retMe = FSAAutomaton.makeAnyString();
58             } else {
59                 String patternFile = prologPatternAuto.toFile("temp1.auto");
60                 String replaceFile = replaceAuto.toFile("temp2.auto");
61                 String subjectFile = subjectAuto.toFile("temp3.auto");
62     
63                 String c = MyOptions.fsaHome + "/" +
64                     "fsa -aux " + mohri + " -r compose(file('" + subjectFile + "'),replace(file('" + patternFile + "'),file('" + replaceFile + "'))) ";
65                 String autoString = Utils.exec(c);
66                 retMe = new FSAAutomaton(autoString);
67     
68                 // projection to the output side (turning a transducer into a recognizer)
69                 retMe = retMe.projectOut();
70             }
71     
72             return retMe;
73         }

75         public static FSAAutomaton str_replace(FSAAutomaton searchAuto,
76                                                FSAAutomaton replaceAuto, FSAAutomaton subjectAuto, AbstractCfgNode cfgNode) {
77     
78             // current approximation:
79             // if the search automaton does not encode a finite string,
80             // we use .* for the replace automaton;
81             // alternatively (and more precisely), we could also use the following for
82             // the replace automaton:
83             // replace = union(replace, search)
84             // which covers the possibility that something that matches the search does
85             // not have to be replaced in every case
86             if (searchAuto.getFiniteString() == null) {
87                 System.out.println("Warning: search automaton is not finite!");
88                 System.out.println("- " + cfgNode.getLoc());
89                 replaceAuto = FSAAutomaton.makeAnyString();
90             }
91     
92             String searchFile = searchAuto.toFile("temp1.auto");
93             String replaceFile = replaceAuto.toFile("temp2.auto");
94             String subjectFile = subjectAuto.toFile("temp3.auto");
95     
96             String c = MyOptions.fsaHome + "/" +
97                 "fsa -aux " + mohri + " -r compose(file('" + subjectFile + "'),replace(file('" + searchFile + "'),file('" + replaceFile + "'))) ";
98     
99             String autoString = Utils.exec(c);
100     
101             FSAAutomaton retMe = new FSAAutomaton(autoString);
102     
103             // projection to the output side (turning a transducer into a recognizer)
104             retMe = retMe.projectOut();
105     
106             return retMe;
107         }

109         public static FSAAutomaton addslashes(FSAAutomaton subjectAuto, AbstractCfgNode cfgNode) {
110             // the easy way: addslashes is the same as applying str_replace
111             // several times:
112             // \ -> \\
113             FSAAutomaton searchAuto = FSAAutomaton.makeString("\\");
114             FSAAutomaton replaceAuto = FSAAutomaton.makeString("\\\\");
115             subjectAuto = str_replace(searchAuto, replaceAuto, subjectAuto, cfgNode);
116             // ' -> \'
117             searchAuto = FSAAutomaton.makeString("'");
118             replaceAuto = FSAAutomaton.makeString("\\'");
119             subjectAuto = str_replace(searchAuto, replaceAuto, subjectAuto, cfgNode);
120             // " -> \"
121             searchAuto = FSAAutomaton.makeString("\"");
122             replaceAuto = FSAAutomaton.makeString("\\\"");
123             subjectAuto = str_replace(searchAuto, replaceAuto, subjectAuto, cfgNode);
124     
125             return subjectAuto;
126         }
