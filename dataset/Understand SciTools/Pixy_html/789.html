16     /**
17      * This class represents a node in the control flow graph.
18      *
19      * A node generally can have several ingoing edges, but only one real outgoing edge.
20      *
21      * @author Nenad Jovanovic <enji@seclab.tuwien.ac.at>
22      */
23     public abstract class AbstractCfgNode {
24         /** the parse node (from the parse tree) to which this node refers */
25         protected final ParseNode parseNode;
26     
27         protected List<CfgEdge> inEdges = new ArrayList<>();
28         // index 0: for false edge (or normal edge)
29         // index 1: for true edge
30         protected CfgEdge[] outEdges = new CfgEdge[2];
31     
32         // number of this cfg node in reverse post-order (speeds up the analysis
33         // if used by the worklist); -1 if uninitialized
34         private int reversePostOrder = -1;
35     
36         // this can be one of the following:
37         // - the enclosing basic block, if there is one (BasicBlock)
38         // - a function's CfgEntry, if this cfg node is member of one of this
39         //   function's default param cfgs
40         // - null, if neither of the above applies
41         private AbstractCfgNode enclosingNode = null;
42     
43         // function that contains this cfgNode;
44         // note: you can't just set this in the constructor, since it
45         // might change during include file resolution
46         private TacFunction enclosingFunction = null;
47     
48         AbstractCfgNode() {
49             this(null);
50         }
51     
52         AbstractCfgNode(ParseNode parseNode) {
53             this.parseNode = parseNode;
54             this.outEdges[0] = null;
55             this.outEdges[1] = null;
56         }
57     
58         // returns
59         // - the enclosing basic block, if it is enclosed in one
60         // - the entry node of the function default cfg, if it is inside such a cfg
61         // - itself otherwise
62         public AbstractCfgNode getSpecial() {
63             AbstractCfgNode retMe;
64     
65             retMe = this.getEnclosingBasicBlock();
66             if (retMe != null) {
67                 return retMe;
68             }
69     
70             retMe = this.getDefaultParamEntry();
71             if (retMe != null) {
72                 return retMe;
73             }
74     
75             return this;
76         }
77     
78         // can return null!
79         public ParseNode getParseNode() {
80             return this.parseNode;
81         }
82     
83         public List<CfgEdge> getInEdges() {
84             return this.inEdges;
85         }
86     
87         public CfgEdge[] getOutEdges() {
88             return this.outEdges;
89         }
90     
91         public CfgEdge getOutEdge(int index) {
92             return this.outEdges[index];
93         }
94     
95         public AbstractCfgNode getSuccessor(int index) {
96             if (this.outEdges[index] != null) {
97                 return this.outEdges[index].getDestination();
98             } else {
99                 return null;
100             }
101         }
102     
103         public List<AbstractCfgNode> getSuccessors() {
104             List<AbstractCfgNode> successors = new LinkedList<>();
105             if (this.outEdges[0] != null) {
106                 successors.add(this.outEdges[0].getDestination());
107                 if (this.outEdges[1] != null) {
108                     successors.add(this.outEdges[1].getDestination());
109                 }
110             }
111     
112             return successors;
113         }
114     
115         // returns the unique predecessor if there is one;
116         // throws an exception otherwise
117         public AbstractCfgNode getPredecessor() {
118             List<AbstractCfgNode> predecessors = this.getPredecessors();
119             if (predecessors.size() != 1) {
120                 throw new RuntimeException("SNH: " + predecessors.size());
121             }
122             return predecessors.get(0);
123         }
124     
125         public List<AbstractCfgNode> getPredecessors() {
126             List<AbstractCfgNode> predecessors = new LinkedList<>();
127             for (CfgEdge inEdge : this.inEdges) {
128                 predecessors.add(inEdge.getSource());
129             }
130             return predecessors;
131         }
132     
133         public int getOriginalLineNumber() {
134             // in some cases, this method is currently not very useful because
135             // it returns "-2" (i.e., the line number of the epsilon node), especially
136             // for constructs such as $x = "hello $world";
137             // PhpParser needs to be improved to overcome this problem
138             if (this.parseNode != null) {
139                 return this.parseNode.getLinenoLeft();
140             } else {
141                 return -1;
142             }
143         }
144     
145         public String getFileName() {
146             if (this.parseNode != null) {
147                 return this.parseNode.getFileName();
148             } else {
149                 return "<file name unknown>";
150             }
151         }
152     
153         public String getLoc() {
154             if (!MyOptions.optionB && !MyOptions.optionW) {
155                 return this.getFileName() + ":" + this.getOriginalLineNumber();
156             } else {
157                 return Utils.basename(this.getFileName()) + ":" + this.getOriginalLineNumber();
158             }
159         }
160     
161         public TacFunction getEnclosingFunction() {
162             if (this.enclosingFunction == null) {
163                 System.out.println(this.getFileName());
164                 System.out.println(this.toString() + ", " + this.getOriginalLineNumber());
165                 throw new RuntimeException("SNH");
166             }
167     
168             return this.enclosingFunction;
169         }
170     
171         // returns a list of Variables referenced by this node; an empty list
172         // if there are none; can also contain null values (placeholders);
173         // targeted at the replacement of $GLOBALS, so you
174         // should take a look at the actual implementations before using it
175         // for something else
176         public abstract List<Variable> getVariables();
177     
178         public int getReversePostOrder() {
179             return this.reversePostOrder;
180         }
181     
182         // returns either null or the enclosing basic block
183         public BasicBlock getEnclosingBasicBlock() {
184             if (this.enclosingNode == null) {
185                 return null;
186             }
187             if (this.enclosingNode instanceof BasicBlock) {
188                 return (BasicBlock) this.enclosingNode;
189             } else {
190                 return null;
191             }
192         }
193     
194         // returns either null or the entry node of the corresponding function
195         // (if this node belongs to the default cfg of a function's formal parameter)
196         public CfgEntry getDefaultParamEntry() {
197             if (this.enclosingNode == null) {
198                 return null;
199             }
200             if (this.enclosingNode instanceof CfgEntry) {
201                 return (CfgEntry) this.enclosingNode;
202             } else {
203                 return null;
204             }
205         }
206     
207         /**
208          * Replaces the variable with the given index in the list returned by getVariables by the given replacement variable.
209          *
210          * @param index
211          * @param replacement
212          */
213         public abstract void replaceVariable(int index, Variable replacement);
214     
215         public void setOutEdge(int index, CfgEdge edge) {
216             this.outEdges[index] = edge;
217         }
218     
219         public void setReversePostOrder(int i) {
220             if (i == Integer.MAX_VALUE) {
221                 throw new RuntimeException("Integer Overflow");
222             }
223             this.reversePostOrder = i;
224         }
225     
226         public void setEnclosingBasicBlock(BasicBlock basicBlock) {
227             this.enclosingNode = basicBlock;
228         }
229     
230         public void setDefaultParamPrep(CfgEntry callPrep) {
231             this.enclosingNode = callPrep;
232         }
233     
234         public void setEnclosingFunction(TacFunction function) {
235             this.enclosingFunction = function;
236         }
237     
238         public void addInEdge(CfgEdge edge) {
239             this.inEdges.add(edge);
240         }
241     
242         // removes the edge coming in from the given predecessor
243         public void removeInEdge(AbstractCfgNode predecessor) {
244             for (Iterator<CfgEdge> iterator = this.inEdges.iterator(); iterator.hasNext(); ) {
245                 CfgEdge inEdge = iterator.next();
246                 if (inEdge.getSource() == predecessor) {
247                     iterator.remove();
248                 }
249             }
250         }
251     
252         public void clearInEdges() {
253             this.inEdges = new LinkedList<>();
254         }
255     
256         public void clearOutEdges() {
257             this.outEdges[0] = this.outEdges[1] = null;
258         }
259     
260         public String toString() {
261             return Dumper.makeCfgNodeName(this);
262         }
