46         public AbstractLatticeElement transfer(AbstractLatticeElement inX, AbstractContext context) {
47     
48             // lattice element entering the call prep node under the current
49             // context (= base for local-info)
50             AliasLatticeElement origInfo =
51                 (AliasLatticeElement) this.analysisNodeAtCallPrep.getPhiValue(context);
52             AliasLatticeElement localInfo = new AliasLatticeElement(origInfo);
53             localInfo.removeGlobals();
54     
55             // lattice element coming in from the callee (= base for interprocedural info);
56             // still contains the callee's locals in the current implementation, so we
57             // have to remove them first
58             AliasLatticeElement calleeIn = (AliasLatticeElement) inX;
59             AliasLatticeElement interInfo = new AliasLatticeElement(calleeIn);
60             interInfo.removeLocals();
61     
62             // start with empty alias information
63             AliasLatticeElement outInfo = new AliasLatticeElement();
64     
65             // add aliases between locals from local-info
66             outInfo.add(localInfo);
67     
68             // add aliases between globals from inter-info
69             outInfo.add(interInfo);
70     
71             // COMPUTE ALIASES BETWEEN LOCALS AND GLOBALS
72             // USING G-SHADOWS AND F-SHADOWS
73     
74             // contains groups from orig-info that have been tagged as visited
75             Set<MustAliasGroup> visitedGroups = new HashSet<>();
76     
77             // G-SHADOWS, MUST
78     
79             // for each must-alias-group in the orig-info
80             for (MustAliasGroup group : origInfo.getMustAliases().getGroups()) {
81                 // pick an arbitrary global from this group
82                 Variable someGlobal = group.getArbitraryGlobal();
83     
84                 // get all locals from this group
85                 Set<Variable> groupLocals = group.getLocals();
86     
87                 // retrieve the global's g-shadow
88                 Variable gShadow = this.callee.getSymbolTable().getGShadow(someGlobal);
89     
90                 // if it contains at least one global and at least one local variable
91                 if (someGlobal != null && !groupLocals.isEmpty()) {
92                     // pick an arbitrary local
93                     Variable someLocal = groupLocals.iterator().next();
94     
95                     // mark this group as visited
96                     visitedGroups.add(group);
97     
98                     // get (non-trivial) must-alias-group for this g-shadow (=> can also be null!)
99                     MustAliasGroup gShadowGroup = calleeIn.getMustAliasGroup(gShadow);
100     
101                     // if such a group exists...
102                     if (gShadowGroup != null) {
103     
104                         // pick an arbitrary global from this group
105                         Variable gShadowGlobalMustAlias = gShadowGroup.getArbitraryGlobal();
106     
107                         // if there is such a global must-alias of the g-shadow
108                         if (gShadowGlobalMustAlias != null) {
109     
110                             // in the output-info, merge the local's group with the group
111                             // that contains this global must-alias, considering implicit
112                             // one-element groups as well
113                             outInfo.merge(someLocal, gShadowGlobalMustAlias);
114                         }
115                     }
116                 }
117     
118                 // for each global may-alias of the g-shadow...
119                 // (note: it would be cleaner to move this loop into the above if-branch, since
120                 // this loop has no final effect if the above branch is not entered;
121                 // that's because of either:
122                 // - groupLocals is empty
123                 // - there is no global variable in this must-alias group, which leads to
124                 //   the iteration over an empty set here
125                 for (Variable globalMayAlias : calleeIn.getGlobalMayAliases(gShadow)) {
126                     outInfo.addMayAliasPairs(groupLocals, globalMayAlias);
127                 }
128             }
129     
130             // G-SHADOWS, MAY
131     
132             // for each may-alias-pair in the orig-info
133             for (MayAliasPair pair : origInfo.getMayAliases().getPairs()) {
134     
135                 Variable[] localGlobal = pair.getLocalGlobal();
136     
137                 // we are only interested in pairs that contain one local
138                 // and one global variable
139                 if (localGlobal == null) {
140                     continue;
141                 }
142     
143                 Variable gShadow = this.callee.getSymbolTable().getGShadow(localGlobal[1]);
144     
145                 // for all global aliases (must and may) of the g-shadow...
146                 for (Variable globalAlias : calleeIn.getGlobalAliases(gShadow)) {
147                     MayAliasPair addMePair = new MayAliasPair(globalAlias, localGlobal[0]);
148                     outInfo.add(addMePair);
149                 }
150             }
151     
152             // F-SHADOWS
153     
154             // for all cbr-params...
155             for (List<Variable> paramPair : this.cbrParams) {
156                 Variable actual = paramPair.get(0);
157     
158                 // we are only interested in *local* actual cbr-params
159                 if (!actual.isLocal()) {
160                     continue;
161                 }
162     
163                 Variable formal = paramPair.get(1);
164                 Variable fShadow = this.callee.getSymbolTable().getFShadow(formal);
165     
166                 // the f-shadow's global must-aliases
167                 Set<Variable> fShadowGlobalMustAliases = calleeIn.getGlobalMustAliases(fShadow);
168     
169                 // the f-shadow's global may-aliases
170                 Set<Variable> fShadowGlobalMayAliases = calleeIn.getGlobalMayAliases(fShadow);
171     
172                 // the actual param's (non-trivial) must-alias group (=> can also be null!)
173                 MustAliasGroup actualGroup = origInfo.getMustAliasGroup(actual);
174     
175                 // if this group is not marked as visited...
176                 if (!visitedGroups.contains(actualGroup)) {
177                     // differentiate between implicit (null) and explicit group;
178                     // LATER: maybe you should do this before coming to the if-branch
179                     Set<Variable> actualGroupLocals = new HashSet<>();
180                     if (actualGroup == null) {
181                         actualGroupLocals.add(actual);
182                     } else {
183                         // mark the group as visited
184                         visitedGroups.add(actualGroup);
185                         actualGroupLocals = actualGroup.getLocals();
186                     }
187     
188                     // if the f-shadow has at least one global must-alias...
189                     if (!fShadowGlobalMustAliases.isEmpty()) {
190                         // pick one
191                         Variable fShadowGlobalMustAlias = fShadowGlobalMustAliases.iterator().next();
192     
193                         // in the output-info, merge the actual's group with the group
194                         // that contains this global must-alias, considering implicit
195                         // one-element groups as well
196                         outInfo.merge(actual, fShadowGlobalMustAlias);
197                     }
198     
199                     // for each global may-alias of the f-shadow...
200                     for (Variable fShadowGlobalMayAlias : fShadowGlobalMayAliases) {
201                         outInfo.addMayAliasPairs(actualGroupLocals, fShadowGlobalMayAlias);
202                     }
203                 }
204     
205                 // for every local may-alias of the actual param...
206                 for (Variable actualLocalMayAlias : origInfo.getLocalMayAliases(actual)) {
207                     // for every global must-alias of the f-shadow...
208                     for (Variable fShadowGlobalMustAlias : fShadowGlobalMustAliases) {
209                         outInfo.add(new MayAliasPair(fShadowGlobalMustAlias, actualLocalMayAlias));
210                     }
211     
212                     // for every global may-alias of the f-shadow...
213                     for (Variable fShadowGlobalMayAlias : fShadowGlobalMayAliases) {
214                         outInfo.add(new MayAliasPair(fShadowGlobalMayAlias, actualLocalMayAlias));
215                     }
216                 }
217             }
218     
219             // FINAL
220     
221             // eliminate alias pairs that "conflict" with must-alias information
222             outInfo.removeConflictingPairs();
223     
224             // recycle
225             outInfo = (AliasLatticeElement) this.aliasAnalysis.recycle(outInfo);
226     
227             return outInfo;
228         }
