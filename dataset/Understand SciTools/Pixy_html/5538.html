12      * Transfer function for function entries.
13      *
14      * @author Nenad Jovanovic <enji@seclab.tuwien.ac.at>

16     public class FunctionEntry extends AbstractTransferFunction {
17         private TacFunction function;
18     
19     // *********************************************************************************
20     // CONSTRUCTORS ********************************************************************
21     // *********************************************************************************
22     
23         public FunctionEntry(TacFunction function) {
24             this.function = function;
25         }
26     
27     // *********************************************************************************
28     // OTHER ***************************************************************************
29     // *********************************************************************************
30     
31         public AbstractLatticeElement transfer(AbstractLatticeElement inX) {
32     
33             DependencyLatticeElement in = (DependencyLatticeElement) inX;
34             DependencyLatticeElement out = new DependencyLatticeElement(in);
35     
36             // initialize g-shadows
37             for (Map.Entry<Variable, Variable> entry : this.function.getSymbolTable().getGlobals2GShadows().entrySet()) {
38                 Variable global = entry.getKey();
39                 Variable gShadow = entry.getValue();
40     
41                 // note: the TacConverter already took care that arrays and array elements
42                 // don't get shadow variables, so you don't have to check this here again
43     
44                 // initialize shadow to the taint/label of its original
45                 out.setShadow(gShadow, global);
46             }
47     
48             // initialize f-shadows
49     
50             for (Map.Entry<Variable, Variable> entry : this.function.getSymbolTable().getFormals2FShadows().entrySet()) {
51                 Variable formal = entry.getKey();
52                 Variable fShadow = entry.getValue();
53     
54                 // initialize
55                 out.setShadow(fShadow, formal);
56             }
57     
58             return out;
59         }
60     }
