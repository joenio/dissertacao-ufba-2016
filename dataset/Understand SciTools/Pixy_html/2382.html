12     /**
13      * Computes for every function the set of global variables that this function (and its callees) may modify.
14      *
15      * It  does NOT consider aliases: This is particularly important for the use of the "global" keyword.
16      *
17      * If you want reasonable results, don't use this GlobalsModificationAnalysis together with a real alias analysis.
18      *
19      * @author Nenad Jovanovic <enji@seclab.tuwien.ac.at>
20      */
21     public class GlobalsModificationAnalysis {
22         // this is what we want to compute:
23         // a set of modified global-likes for each function
24         // (global variables, superglobals, and constants)
25         // NOTE: currently, we do NOT support constants for this
26         Map<TacFunction, Set<AbstractTacPlace>> func2Mod;
27     
28     //  ********************************************************************************
29     
30         public GlobalsModificationAnalysis(List<TacFunction> functions, CallGraph callGraph) {
31             this.analyze(functions, callGraph);
32         }
33     
34     //  ********************************************************************************

36         public Set<AbstractTacPlace> getMod(TacFunction function) {
37             return this.func2Mod.get(function);
38         }

42         private void analyze(List<TacFunction> functions, CallGraph callGraph) {
43     
44             this.func2Mod = new HashMap<>();
45     
46             // intraprocedural analysis
47     
48             // - for each function:
49             //   - make a simple pass over the function's cfg nodes
50             //     (order irrelevant => flow-insensitive!)
51             //   - result: for this function, a set of global variables that
52             //     can be modified inside this function;
53             //     ignore function calls at this stage
54             for (TacFunction function : functions) {
55                 Set<AbstractTacPlace> modSet = new HashSet<>();
56     
57                 for (AbstractCfgNode cfgNodeX : function.getControlFlowGraph().dfPreOrder()) {
58                     this.processNode(cfgNodeX, modSet);
59                 }
60     
61                 func2Mod.put(function, modSet);
62             }
63     
64             // interprocedural analysis
65             // - operates on the call graph
66             // - the worklist consists of functions
67             // - the worklist is initialized with all functions in postorder
68             // - while (worklist not empty):
69             //   - remove next worklist element: function f
70             //   - for all functions c that call this function f:
71             //     - compute the union u = mod(c) + mod(f)
72             //     - if u != mod(c) [faster: if u > mod(c)]
73             //       - set mod(c) = u
74             //       - add c to the worklist
75     
76             Map<TacFunction, Integer> postorder = callGraph.getPostOrder();
77     
78             // initialize worklist
79             SortedMap<Integer, TacFunction> worklist = new TreeMap<>();
80             for (Map.Entry<TacFunction, Integer> entry : postorder.entrySet()) {
81                 worklist.put(entry.getValue(), entry.getKey());
82             }
83     
84             // do the worklist algorithm...
85             while (!worklist.isEmpty()) {
86                 TacFunction f = worklist.remove(worklist.firstKey());
87                 Collection<CallGraphNode> callers = callGraph.getCallers(f);
88                 for (CallGraphNode callerNode : callers) {
89                     TacFunction caller = callerNode.getFunction();
90                     Set<AbstractTacPlace> modF = func2Mod.get(f);
91                     Set<AbstractTacPlace> modCaller = func2Mod.get(caller);
92                     int modCallerSize = modCaller.size();
93                     modCaller.addAll(modF);
94                     if (modCallerSize != modCaller.size()) {
95                         worklist.put(postorder.get(caller), caller);
96                     }
97                 }
98             }
99         }

103         // if the given cfg node has an effect on mod info, this method
104         // adjusts the given modSet accordingly (i.e., it adds variables to it)
105         private void processNode(AbstractCfgNode cfgNodeX, Set<AbstractTacPlace> modSet) {
106     
107             if (cfgNodeX instanceof BasicBlock) {
108     
109                 BasicBlock basicBlock = (BasicBlock) cfgNodeX;
110                 for (AbstractCfgNode cfgNode : basicBlock.getContainedNodes()) {
111                     processNode(cfgNode, modSet);
112                 }
113             } else if (cfgNodeX instanceof AssignSimple) {
114     
115                 AssignSimple cfgNode = (AssignSimple) cfgNodeX;
116                 Variable modVar = cfgNode.getLeft();
117                 if (modVar.isGlobal() || modVar.isSuperGlobal()) {
118                     this.modify(modVar, modSet);
119                 }
120             } else if (cfgNodeX instanceof AssignUnary) {
121     
122                 AssignUnary cfgNode = (AssignUnary) cfgNodeX;
123                 Variable modVar = cfgNode.getLeft();
124                 if (modVar.isGlobal() || modVar.isSuperGlobal()) {
125                     this.modify(modVar, modSet);
126                 }
127             } else if (cfgNodeX instanceof AssignBinary) {
128     
129                 AssignBinary cfgNode = (AssignBinary) cfgNodeX;
130                 Variable modVar = cfgNode.getLeft();
131                 if (modVar.isGlobal() || modVar.isSuperGlobal()) {
132                     this.modify(modVar, modSet);
133                 }
134             } else if (cfgNodeX instanceof AssignArray) {
135     
136                 AssignArray cfgNode = (AssignArray) cfgNodeX;
137                 Variable modVar = cfgNode.getLeft();
138                 if (modVar.isGlobal() || modVar.isSuperGlobal()) {
139                     this.modify(modVar, modSet);
140                 }
141             } else if (cfgNodeX instanceof AssignReference) {
142     
143                 AssignReference cfgNode = (AssignReference) cfgNodeX;
144                 Variable modVar = cfgNode.getLeft();
145                 if (modVar.isGlobal() || modVar.isSuperGlobal()) {
146                     this.modify(modVar, modSet);
147                 }
148             } else if (cfgNodeX instanceof Unset) {
149     
150                 Unset cfgNode = (Unset) cfgNodeX;
151                 Variable modVar = cfgNode.getOperand();
152                 if (modVar.isGlobal() || modVar.isSuperGlobal()) {
153                     this.modify(modVar, modSet);
154                 }
155             } else {
156                 // no change to mod-info for the remaining cfg nodes
157             }
158         }

162         private void modify(Variable modVar, Set<AbstractTacPlace> modSet) {
163             modSet.add(modVar);
164             if (modVar.isArray()) {
165                 // the whole array subtree is modified as well
166                 modSet.addAll(modVar.getElementsRecursive());
167             }
168             if (modVar.isArrayElement()) {
169                 // by marking the top array as modified,
170                 // we indirectly mark its array label as modified
171                 modSet.add(modVar.getTopEnclosingArray());
172             }
173         }
