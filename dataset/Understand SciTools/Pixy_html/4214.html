47         public Set<Type> getType(Variable var, AbstractCfgNode cfgNode) {
48             AbstractInterproceduralAnalysisNode ian = this.interproceduralAnalysisInformation.getAnalysisNode(cfgNode);
49             if (ian == null) {
50                 // this means that this cfg node was not assigned an analysis node
51                 // (should never happen)
52                 System.out.println(var);
53                 System.out.println(cfgNode);
54                 System.out.println(cfgNode.getLoc());
55                 throw new RuntimeException("SNH");
56             }
57             TypeLatticeElement elem = (TypeLatticeElement) ian.computeFoldedValue();
58             if (elem == null) {
59                 // this cfg node has no associated analysis info,
60                 // which means that it is unreachable
61                 // (e.g., because it is inside a function that is never called)
62                 return null;
63             }
64             return elem.getType(var);
65         }

139         protected AbstractTransferFunction callPrep(AbstractCfgNode cfgNodeX, TacFunction traversedFunction) {
140     
141             at.ac.tuwien.infosys.www.pixy.conversion.cfgnodes.CallPreparation cfgNode = (at.ac.tuwien.infosys.www.pixy.conversion.cfgnodes.CallPreparation) cfgNodeX;
142             TacFunction calledFunction = cfgNode.getCallee();
143             TacFunction callingFunction = traversedFunction;
144     
145             // call to an unknown function;
146             // should be prevented in practice (all functions should be
147             // modeled in the builtin functions file), but if
148             // it happens: assume that it doesn't do anything;
149             if (calledFunction == null) {
150     
151                 // how this works:
152                 // - propagate with ID transfer function to Call
153                 // - the analysis algorithm propagates from Call
154                 //   to CallReturn with ID transfer function
155                 // - CallReturn does the rest
156                 //System.out.println("unknown function: " + cfgNode.getFunctionNamePlace());
157                 return TransferFunctionId.INSTANCE;
158             }
159     
160             // extract actual and formal params
161             List<TacActualParameter> actualParams = cfgNode.getParamList();
162             List<TacFormalParameter> formalParams = calledFunction.getParams();
163     
164             // the transfer function to be assigned to this node
165             AbstractTransferFunction tf = null;
166     
167             if (actualParams.size() > formalParams.size()) {
168                 // more actual than formal params; either a bug or a varargs
169                 // occurrence;
170                 // note that cfgNode.getFunctionNamePlace() returns a different
171                 // result than function.getName() if "function" is
172                 // the unknown function
173                 throw new RuntimeException(
174                     "More actual than formal params for function " +
175                         cfgNode.getFunctionNamePlace().toString() + " on line " + cfgNode.getOriginalLineNumber());
176             } else {
177                 tf = new CallPreparation(actualParams, formalParams,
178                     callingFunction, calledFunction, this);
179             }
180     
181             return tf;
182         }

184         protected AbstractTransferFunction callRet(AbstractCfgNode cfgNodeX, TacFunction traversedFunction) {
185     
186             at.ac.tuwien.infosys.www.pixy.conversion.cfgnodes.CallReturn cfgNodeRet = (at.ac.tuwien.infosys.www.pixy.conversion.cfgnodes.CallReturn) cfgNodeX;
187             Call cfgNodeCall = cfgNodeRet.getCallNode();
188             at.ac.tuwien.infosys.www.pixy.conversion.cfgnodes.CallPreparation cfgNodePrep = cfgNodeRet.getCallPrepNode();
189     
190             TacFunction callingFunction = traversedFunction;
191             TacFunction calledFunction = cfgNodeCall.getCallee();
192     
193             // call to an unknown function;
194             // for explanations see above (handling CallPreparation)
195             AbstractTransferFunction tf;
196             if (calledFunction == null) {
197     
198                 tf = new CallReturnUnknown(cfgNodeRet);
199             } else {
200     
201                 tf = new CallReturn(
202                     this.interproceduralAnalysisInformation.getAnalysisNode(cfgNodePrep),
203                     callingFunction,
204                     calledFunction,
205                     cfgNodeCall);
206             }
207     
208             return tf;
