62         // returns the contents of the given file as string
63         private String readFile(String fileName) {
64             StringBuilder ret = new StringBuilder();
65             try {
66                 FileReader fr = new FileReader(fileName);
67                 int c;
68                 ret = new StringBuilder();
69                 while ((c = fr.read()) != -1) {
70                     ret.append((char) c);
71                 }
72             } catch (FileNotFoundException e) {
73                 Assert.fail("File not found: " + fileName);
74             } catch (IOException e) {
75                 Assert.fail(e.getMessage());
76             }
77             return ret.toString();
78         }

87         private void performTest(String testNum, int sinkNum, int graphNum,
88                                  boolean generate, boolean functional, int vulnNum) {
89             mySetUp("test" + testNum + ".php", functional);
90     
91             Assert.assertTrue("Sinks real: " + sinks.size() + ", expected: "
92                 + sinkNum, sinks.size() == sinkNum);
93     
94             // collect dependencyGraphs
95             List<DependencyGraph> dependencyGraphs = new LinkedList<>();
96             for (Sink sink : sinks) {
97                 dependencyGraphs.addAll(dependencyAnalysis.getDependencyGraphsForSink(sink));
98             }
99     
100             Assert.assertTrue("Graphs real: " + dependencyGraphs.size() + ", expected: "
101                 + graphNum, dependencyGraphs.size() == graphNum);
102     
103             int graphCount = 0;
104             int vulnCount = 0;
105             for (DependencyGraph dependencyGraph : dependencyGraphs) {
106     
107                 // check depgraph
108     
109                 graphCount++;
110                 String fileName = "test" + testNum + "_" + graphCount;
111                 if (generate) {
112                     dependencyGraph.dumpDotUnique(fileName, this.path);
113                 } else {
114                     String encountered = dependencyGraph.makeDotUnique(fileName);
115                     String expected = this.readFile(this.path + fileName + ".dot");
116                     Assert.assertEquals(expected, encountered);
117                 }
118     
119                 // check xssgraph
120     
121                 String xssFileName = "test" + testNum + "_" + graphCount + "_xss";
122                 DependencyGraph relevant = this.xssAnalysis.getRelevantSubgraph(dependencyGraph);
123                 Map<UninitializedNode, AbstractVulnerabilityAnalysis.InitialTaint> dangerousUninit = this.xssAnalysis.findDangerousUninitializedNodes(relevant);
124                 if (!dangerousUninit.isEmpty()) {
125                     vulnCount++;
126                     relevant.reduceWithLeaves(dangerousUninit.keySet());
127     
128                     if (generate) {
129                         relevant.dumpDotUnique(xssFileName, this.path);
130                     } else {
131                         String encountered = relevant.makeDotUnique(xssFileName);
132                         String expected = this.readFile(this.path + xssFileName + ".dot");
133                         Assert.assertEquals(expected, encountered);
134                     }
135                 }
136             }
137     
138             // check if all vulns were detected
139             Assert.assertEquals(vulnNum, vulnCount);
140     
141             if (generate) {
142                 // just to make sure that you don't accidentally forget
143                 // to switch generation off, and turn checking on
144                 Assert.fail("no check performed");
145             }
146         }
