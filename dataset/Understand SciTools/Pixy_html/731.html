15     /**
16      * An AnalysisNode holds analysis-specific information for a certain CFGNode.
17      *
18      * @author Nenad Jovanovic <enji@seclab.tuwien.ac.at>
19      */
20     public class FunctionalAnalysisNode extends AbstractInterproceduralAnalysisNode {
21         // mapping input AbstractLatticeElement -> Set of context LatticeElements;
22         // only needed for call nodes
23         Map<AbstractLatticeElement, Set<FunctionalContext>> reversePhi;
24     
25     // *********************************************************************************
26     // CONSTRUCTORS ********************************************************************
27     // *********************************************************************************
28     
29         public FunctionalAnalysisNode(AbstractCfgNode node, AbstractTransferFunction tf) {
30             super(tf);
31             // maintain reverse mapping for call nodes
32             if (node instanceof Call) {
33                 this.reversePhi = new HashMap<>();
34             } else {
35                 this.reversePhi = null;
36             }
37         }
38     
39     // *********************************************************************************
40     // GET *****************************************************************************
41     // *********************************************************************************
42     
43         // returns a set of contexts that are mapped to the given value
44         Set<FunctionalContext> getReversePhiContexts(AbstractLatticeElement value) {
45             return this.reversePhi.get(value);
46         }
47     
48     // *********************************************************************************
49     // SET *****************************************************************************
50     // *********************************************************************************
51     
52         // sets the PHI value for the given context
53         protected void setPhiValue(AbstractContext contextX, AbstractLatticeElement value) {
54     
55             FunctionalContext context = (FunctionalContext) contextX;
56     
57             super.setPhiValue(context, value);
58     
59             // maintain reverse mapping, if needed
60             if (this.reversePhi != null) {
61                 Set<FunctionalContext> contextSet = this.reversePhi.get(value);
62                 if (contextSet == null) {
63                     contextSet = new HashSet<>();
64                     this.reversePhi.put(value, contextSet);
65                 }
66                 contextSet.add(context);
67             }
68         }
69     }
