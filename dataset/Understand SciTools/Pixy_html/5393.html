8     /**
9      * This class represents a basic block that contains other CFG nodes.
10      *
11      * @author Nenad Jovanovic <enji@seclab.tuwien.ac.at>
12      */
13     public class BasicBlock extends AbstractCfgNode {
14         List<AbstractCfgNode> containedNodes;
15     
16         // associate parse node of the basic block's initial node
17         // with this basic block node (important for line number)
18         public BasicBlock(AbstractCfgNode initialNode) {
19             super(initialNode.getParseNode());
20             this.containedNodes = new LinkedList<>();
21             this.containedNodes.add(initialNode);
22         }
23     
24         // informs enclosed nodes that they are inside a basic block
25         public void informEnclosedNodes() {
26             for (AbstractCfgNode cfgNode : this.containedNodes) {
27                 cfgNode.setEnclosingBasicBlock(this);
28             }
29         }
30     
31         public void addNode(AbstractCfgNode cfgNode) {
32             this.containedNodes.add(cfgNode);
33         }
34     
35         public List<AbstractCfgNode> getContainedNodes() {
36             return this.containedNodes;
37         }
38     
39         public List<Variable> getVariables() {
40             List<Variable> variables = new LinkedList<>();
41             for (AbstractCfgNode node : this.containedNodes) {
42                 variables.addAll(node.getVariables());
43             }
44             return variables;
45         }
46     
47         public void replaceVariable(int index, Variable replacement) {
48             for (AbstractCfgNode node : this.containedNodes) {
49                 List<Variable> gotVariables = node.getVariables();
50                 if (gotVariables.size() > index) {
51                     node.replaceVariable(index, replacement);
52                     return;
53                 } else {
54                     index -= gotVariables.size();
55                 }
56             }
57             // if you reach this point, no variable was replaced
58             throw new RuntimeException("SNH");
59         }
60     
61         public String getFileName() {
62             if (!this.containedNodes.isEmpty()) {
63                 return this.containedNodes.get(0).getFileName();
64             } else {
65                 return super.getFileName();
66             }
67         }
68     
69         public int getOriginalLineNumber() {
70             if (!this.containedNodes.isEmpty()) {
71                 return this.containedNodes.get(0).getOriginalLineNumber();
72             } else {
73                 return super.getOriginalLineNumber();
74             }
75         }
76     }
