8     /**
9      * This interprocedural worklist uses a better order (interprocedural reverse post-order).
10      *
11      * @author Nenad Jovanovic <enji@seclab.tuwien.ac.at>
12      */
13     public final class InterproceduralWorklistBetter implements InterproceduralWorklist {
14         private InterproceduralWorklistOrder order;
15         private SortedMap<Integer, InterproceduralWorklistElement> sortedWorkList;
16     
17     // *********************************************************************************
18     // CONSTRUCTORS ********************************************************************
19     // *********************************************************************************
20     
21         public InterproceduralWorklistBetter(InterproceduralWorklistOrder order) {
22             this.order = order;
23             this.sortedWorkList = new TreeMap<>();
24         }
25     
26     // *********************************************************************************
27     // OTHER ***************************************************************************
28     // *********************************************************************************
29     
30         public void add(AbstractCfgNode cfgNode, AbstractContext context) {
31             // null contexts are not allowed
32             if (context == null) {
33                 throw new RuntimeException("SNH");
34             }
35             InterproceduralWorklistElement element = new InterproceduralWorklistElement(cfgNode, context);
36     
37             Integer rpo = this.order.getReversePostOrder(element);
38             if (rpo == null) {
39                 throw new RuntimeException("SNH");
40             }
41             this.sortedWorkList.put(rpo, element);
42         }
43     
44         public InterproceduralWorklistElement removeNext() {
45             Integer key = this.sortedWorkList.firstKey();
46             return this.sortedWorkList.remove(key);
47         }
48     
49         public boolean hasNext() {
50             return !this.sortedWorkList.isEmpty();
51         }
52     }
