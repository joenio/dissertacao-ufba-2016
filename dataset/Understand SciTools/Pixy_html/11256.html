44             List<File> retMe = new LinkedList<>();
45             File rootDir = new File(rootDirName);

50         private static void fileListHelper(File dir, List<File> retMe) {
51     
52             try {
53     
54                 List<File> dirList = new LinkedList<>();
55                 File[] fileList = dir.listFiles();
56                 for (File f : fileList) {
57                     if (f.isFile()) {
58                         retMe.add(f.getCanonicalFile());
59                     } else if (f.isDirectory()) {
60                         dirList.add(f);
61                     }
62                 }
63                 for (File d : dirList) {
64                     fileListHelper(d, retMe);
65                 }
66             } catch (IOException e) {
67                 throw new RuntimeException(e.getMessage());
68             }
69         }

81         // executes the given command and returns the output;
82         // error output is written to a logfile
83         public static String exec(String command) {
84             String execErrorLog = MyOptions.graphPath + "/exec-errors.txt";
85     
86             StringBuilder retMe = new StringBuilder();
87             try {
88                 Process p = Runtime.getRuntime().exec(command);
89     
90                 InputStream instream = p.getInputStream();      // standard output of prog
91                 InputStream errstream = p.getErrorStream();     // error output of prog
92                 p.waitFor();
93     
94                 // read output
95                 try {
96                     BufferedReader inreader = new BufferedReader(new InputStreamReader(instream));
97                     String line;
98                     while ((line = inreader.readLine()) != null) {
99                         retMe.append(line);
100                         retMe.append("\n");
101                     }
102                     inreader.close();
103                 } catch (IOException e) {
104                     throw new RuntimeException(e.getMessage());
105                 }
106     
107                 // read and print errors
108                 Writer outWriter = new FileWriter(execErrorLog, true);
109                 try {
110                     BufferedReader inreader = new BufferedReader(new InputStreamReader(errstream));
111                     String line;
112     
113                     while ((line = inreader.readLine()) != null) {
114                         outWriter.append(line + "\n");
115                     }
116                     inreader.close();
117                 } catch (IOException e) {
118                     throw new RuntimeException(e.getMessage());
119                 }
120     
121                 if (p.exitValue() != 0) {
122                     System.err.println("Exit value: " + p.exitValue());
123                     System.err.println("- " + command);
124                     new RuntimeException().printStackTrace();
125                 }
126                 outWriter.close();
127             } catch (IOException | InterruptedException e) {
128                 e.printStackTrace();
129                 throw new RuntimeException();
130             }
131     
132             return retMe.toString();
133         }
