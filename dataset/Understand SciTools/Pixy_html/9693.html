37     /**
38      * Finite-state automaton with regular expression operations.
39      * <p/>
40      * Automatons are represented using {@link State} and {@link Transition} objects.
41      * Implicitly, all states and transitions of an automaton are reachable from its initial state.
42      * If the states or transitions are manipulated manually, the {@link #restoreInvariant()}
43      * and {@link #setDeterministic(boolean)} methods should be used afterwards to restore
44      * certain representation invariants that are assumed by the built-in automaton operations.
45      *
46      * Class invariants:
47      *
48      * - An automaton is either represented explicitly (with State and Transition objects)
49      *   or with a singleton string in case the automaton accepts exactly one string.
50      * - Automatons are always reduced (see reduce())
51      *   and have no transitions to dead states (see removeDeadTransitions()).
52      * - If an automaton is nondeterministic, then isDeterministic() returns false (but
53      *   the converse is not required).
54      *
55      * @author Anders M&oslash;ller &lt;<a href="mailto:amoeller@brics.dk">amoeller@brics.dk</a>&gt;
56      */

138         // performs a simple data flow analysis that maps states to [open,closed]
139         // (with respect to "apostrophe areas"); if there are indirectly tainted
140         // transitions with a [closed] source state, it means that we have a
141         // problem => returns true; else: false
142         public boolean hasDangerousIndirectTaint() {
143     
144             // "data flow analysis information"
145             Map<State, Integer> dfi = new HashMap<>();
146             int open = 0;
147             int closed = 1;
148             dfi.put(this.initial, closed);
149     
150             // initialize worklist with start state
151             LinkedList<State> worklist = new LinkedList<>();
152             worklist.add(this.initial);
153     
154             while (worklist.size() > 0) {
155                 State source = worklist.removeFirst();
156     
157                 for (Transition t : source.transitions) {
158                     int sourceInfo = dfi.get(source);
159                     // the value that is to be propagated
160                     int propagate = -1;
161                     // if the current transition does not correspond to a certain
162                     // character: simply propagate source info
163                     if (t.min != t.max) {
164                         propagate = sourceInfo;
165                     }
166                     // if the current transition is labeled with one of these,
167                     // "flip" the info that is to be propagated
168                     if (t.min == '"' || t.min == '\'') {
169                         if (sourceInfo == open) {
170                             propagate = closed;
171                         } else if (sourceInfo == closed) {
172                             propagate = open;
173                         } else {
174                             throw new RuntimeException("SNH: " + sourceInfo);
175                         }
176                     } else {
177                         propagate = sourceInfo;
178                     }
179     
180                     // propagate to target state
181                     State target = t.getDest();
182                     Integer targetInfo = dfi.get(target);
183                     if (targetInfo == null) {
184                         // the target node has not been visited yet
185                         dfi.put(target, propagate);
186                         worklist.add(target);
187                     } else {
188                         int newInfo = this.lub(propagate, targetInfo);
189                         if (newInfo == targetInfo) {
190                             // no change, don't continue here
191                         } else {
192                             // update analysis info and worklist
193                             dfi.put(target, newInfo);
194                             worklist.add(target);
195                         }
196                     }
197                 }
198             }
199     
200             for (Map.Entry<State, Integer> entry : dfi.entrySet()) {
201                 State s = entry.getKey();
202                 Integer info = entry.getValue();
203                 if (info == open) {
204                     // no danger here
205                     continue;
206                 }
207                 for (Transition t : s.getTransitions()) {
208                     if (t.taint == Transition.Taint.Indirectly) {
209                         // detected indirect vulnerability!
210                         return true;
211                     }
212                 }
213             }
214             return false;
215         }

220                 return 1;
221             } else {
222                 return 0;
223             }

305         public Set<State> getStates() {
306             expandSingleton();
307             HashSet<State> visited = new HashSet<>();
308             LinkedList<State> worklist = new LinkedList<>();
309             worklist.add(initial);
310             visited.add(initial);
311             while (worklist.size() > 0) {
312                 State s = worklist.removeFirst();
313                 for (Transition t : s.transitions)
314                     if (!visited.contains(t.to)) {
315                         visited.add(t.to);
316                         worklist.add(t.to);
317                     }
318             }
319             return visited;

372         private void bfIteratorHelper(
373             LinkedList<State> queue, Set<State> visited,
374             Comparator<Transition> comp) {
375     
376             State node = queue.removeFirst();
377     
378             // handle successors
379             List<Transition> successors = new LinkedList<>(node.transitions);
380             Collections.sort(successors, comp);
381     
382             for (Transition succ : successors) {
383                 // for all successors that have not been visited yet...
384                 if (!visited.contains(succ.to)) {
385                     succ.to.number = visited.size();
386                     // add it to the queue
387                     queue.add(succ.to);
388                     // mark it as visited
389                     visited.add(succ.to);
390                 }
391             }
392     
393             // if the queue is non-empty: recurse
394             if (queue.size() > 0) {
395                 bfIteratorHelper(queue, visited, comp);
396             }
397         }

399         private class TransComparator<T>
400             implements Comparator<T> {
401     
402             public int compare(T o1, T o2) {
403                 if (!(o1 instanceof Transition) || !(o2 instanceof Transition)) {
404                     throw new RuntimeException("SNH");
405                 }
406                 Transition n1 = (Transition) o1;
407                 Transition n2 = (Transition) o2;
408                 if (n1.min != n2.min) {
409                     return (new Character(n1.min).compareTo(n2.min));
410                 } else {
411                     if (n1.max != n2.max) {
412                         return (new Character(n1.max).compareTo(n2.max));
413                     }
414                     // they have the same labels; try taint status...
415                     if (n1.taint != n2.taint) {
416                         if (n1.taint == Transition.Taint.Untainted) {
417                             return -1;
418                         } else if (n2.taint == Transition.Taint.Untainted) {
419                             return 1;
420                         } else if (n1.taint == Transition.Taint.Indirectly) {
421                             return -1;
422                         } else if (n2.taint == Transition.Taint.Indirectly) {
423                             return 1;
424                         } else {
425                             throw new RuntimeException("SNH");
426                         }
427                     } else {
428                         return 0;
429                     }
430                 }
431             }
432         }

448          * Checks whether there is a loop containing s. (This is sufficient since
449          * there are never transitions to dead states.)
450          */
451         boolean isFinite(State s, HashSet<State> path) {
452             path.add(s);
453             for (Transition t : s.transitions)
454                 if (path.contains(t.to) || !isFinite(t.to, path))
455                     return false;
456             path.remove(s);
457             return true;
458         }

460         /**
461          * Returns the strings that can be produced from s, returns false if more than
462          * <code>limit</code> strings are found. <code>limit</code>&lt;0 means "infinite".
463          */
464         boolean getFiniteStrings(State s, HashSet<State> pathstates, HashSet<String> strings, StringBuilder path, int limit) {
465             pathstates.add(s);
466             for (Transition t : s.transitions) {
467                 if (pathstates.contains(t.to))
468                     return false;
469                 for (int n = t.min; n <= t.max; n++) {
470                     path.append((char) n);
471                     if (t.to.accept) {
472                         strings.add(path.toString());
473                         if (limit >= 0 && strings.size() > limit)
474                             return false;
475                     }
476                     if (!getFiniteStrings(t.to, pathstates, strings, path, limit))
477                         return false;
478                     path.deleteCharAt(path.length() - 1);
479                 }
480             }
481             pathstates.remove(s);
482             return true;
483         }

485         /**
486          * Adds transitions to explicit crash state to ensure that transition function is total.
487          */
488         void totalize() {
489             State s = new State();
490             s.transitions.add(new Transition(Character.MIN_VALUE, Character.MAX_VALUE, s));
491             for (State p : getStates()) {
492                 int maxi = Character.MIN_VALUE;
493                 for (Transition t : p.getSortedTransitions(false)) {
494                     if (t.min > maxi)
495                         p.transitions.add(new Transition((char) maxi, (char) (t.min - 1), s));
496                     if (t.max + 1 > maxi)
497                         maxi = t.max + 1;
498                 }
499                 if (maxi <= Character.MAX_VALUE)
500                     p.transitions.add(new Transition((char) maxi, Character.MAX_VALUE, s));
501             }
502         }

504         /**
505          * Reduces this automaton.
506          * An automaton is "reduced" by combining overlapping and adjacent edge intervals with same destination.
507          */
508         public void reduce() {
509             if (isSingleton())
510                 return;
511             Set<State> states = getStates();
512             setStateNumbers(states);
513             for (State s : states) {
514                 List<Transition> st = s.getSortedTransitions(true);
515                 s.resetTransitions();
516                 State p = null;
517                 int min = -1, max = -1;
518                 for (Transition t : st) {
519                     if (p == t.to) {
520                         if (t.min <= max + 1) {
521                             if (t.max > max)
522                                 max = t.max;
523                         } else {
524                             if (p != null)
525                                 s.transitions.add(new Transition((char) min, (char) max, p));
526                             min = t.min;
527                             max = t.max;
528                         }
529                     } else {
530                         if (p != null)
531                             s.transitions.add(new Transition((char) min, (char) max, p));
532                         p = t.to;
533                         min = t.min;
534                         max = t.max;
535                     }
536                 }
537                 if (p != null)
538                     s.transitions.add(new Transition((char) min, (char) max, p));
539             }
540         }

542         /**
543          * Gets sorted array of all interval start points.
544          */
545         char[] getStartPoints() {
546             Set<Character> pointset = new HashSet<>();
547             for (State s : getStates()) {
548                 pointset.add(Character.MIN_VALUE);
549                 for (Transition t : s.transitions) {
550                     pointset.add(t.min);
551                     if (t.max < Character.MAX_VALUE)
552                         pointset.add((char) (t.max + 1));
553                 }
554             }
555             char[] points = new char[pointset.size()];
556             int n = 0;
557             for (Character m : pointset)
558                 points[n++] = m;
559             Arrays.sort(points);
560             return points;
561         }

563         Set<State> getLiveStates(Set<State> states) {
564             HashMap<State, Set<State>> map = new HashMap<>();
565             for (State s : states)
566                 map.put(s, new HashSet<State>());
567             for (State s : states)
568                 for (Transition t : s.transitions)
569                     map.get(t.to).add(s);
570             Set<State> live = new HashSet<>(getAcceptStates());
571             LinkedList<State> worklist = new LinkedList<>(live);
572             while (worklist.size() > 0) {
573                 State s = worklist.removeFirst();
574                 for (State p : map.get(s))
575                     if (!live.contains(p)) {
576                         live.add(p);
577                         worklist.add(p);
578                     }
579             }
580             return live;
581         }

583         /**
584          * Removes transitions to dead states and calls {@link #reduce()}
585          * (a state is "dead" if no accept state is reachable from it).
586          */
587         public void removeDeadTransitions() {
588             if (isSingleton())
589                 return;
590             Set<State> states = getStates();
591             Set<State> live = getLiveStates(states);
592             for (State s : states) {
593                 Set<Transition> st = s.transitions;
594                 s.resetTransitions();
595                 for (Transition t : st)
596                     if (live.contains(t.to))
597                         s.transitions.add(t);
598             }
599             reduce();
600         }

703             }
704             return s;

725          * Constructs sub-automaton corresponding to decimal numbers of value
726          * between x.substring(n) and y.substring(n) and of
727          * length x.substring(n).length() (which must be equal to y.substring(n).length()).
728          */
729         private static State between(String x, String y, int n, Collection<State> initials, boolean zeros) {
730             State s = new State();
731             if (x.length() == n)
732                 s.setAccept(true);
733             else {
734                 if (zeros)
735                     initials.add(s);
736                 char cx = x.charAt(n);
737                 char cy = y.charAt(n);
738                 if (cx == cy)
739                     s.addTransition(new Transition(cx, between(x, y, n + 1, initials, zeros && cx == '0')));
740                 else { // cx<cy
741                     s.addTransition(new Transition(cx, atLeast(x, n + 1, initials, zeros && cx == '0')));
742                     s.addTransition(new Transition(cy, atMost(y, n + 1)));
743                     if (cx + 1 < cy)
744                         s.addTransition(new Transition((char) (cx + 1), (char) (cy - 1), anyOfRightLength(x, n + 1)));
745                 }
746             }
747             return s;
748         }

802          * the given automata.
803          * <p/>
804          * Complexity: linear in total number of states.
805          */
806         static public Automaton concatenate(List<Automaton> l) {
807             if (l.isEmpty())
808                 return makeEmptyString();
809             boolean all_singleton = true;
810             for (Automaton a : l)
811                 if (!a.isSingleton()) {
812                     all_singleton = false;
813                     break;
814                 }
815             if (all_singleton) {
816                 StringBuilder b = new StringBuilder();
817                 for (Automaton a : l)
818                     b.append(a.singleton);
819                 return makeString(b.toString());
820             } else {

822                     if (a.isEmpty())
823                         return makeEmpty();

825                 Set<State> ac = b.getAcceptStates();
826                 boolean first = true;
827                 for (Automaton a : l)
828                     if (first)

832                             continue;

834                         Set<State> ns = aa.getAcceptStates();
835                         for (State s : ac) {

842                         }

844                     }

886         /**
887          * Returns new (deterministic) automaton that accepts the intersection of
888          * the languages of this and the given automaton. As a side-effect, both
889          * this and the given automaton are determinized, if not already
890          * deterministic.
891          * <p/>
892          * Complexity: quadratic in number of states (if already deterministic).
893          */
894         public Automaton intersection(Automaton a) {
895             if (isSingleton()) {
896                 if (a.run(singleton))
897                     return clone();
898                 else
899                     return makeEmpty();
900             }
901             if (a.isSingleton()) {
902                 if (run(a.singleton))
903                     return a.clone();
904                 else
905                     return makeEmpty();
906             }
907             determinize();
908             a.determinize();
909             if (a == this)
910                 return clone();
911             Transition[][] transitions1 = getSortedTransitions(getStates());
912             Transition[][] transitions2 = getSortedTransitions(a.getStates());
913             Automaton c = new Automaton();
914             LinkedList<StatePair> worklist = new LinkedList<>();
915             HashMap<StatePair, StatePair> newstates = new HashMap<>();
916             State s = new State();
917             c.initial = s;
918             StatePair p = new StatePair(s, initial, a.initial);
919             worklist.add(p);
920             newstates.put(p, p);
921             while (worklist.size() > 0) {
922                 p = worklist.removeFirst();
923                 p.s.accept = p.s1.accept && p.s2.accept;
924                 Transition[] t1 = transitions1[p.s1.number];
925                 Transition[] t2 = transitions2[p.s2.number];
926                 for (int n1 = 0, n2 = 0; n1 < t1.length && n2 < t2.length; ) {
927                     if (t1[n1].max < t2[n2].min)
928                         n1++;
929                     else if (t2[n2].max < t1[n1].min)
930                         n2++;
931                     else {
932                         StatePair q = new StatePair(t1[n1].to, t2[n2].to);
933                         StatePair r = newstates.get(q);
934                         if (r == null) {
935                             q.s = new State();
936                             worklist.add(q);
937                             newstates.put(q, q);
938                             r = q;
939                         }
940                         char min = t1[n1].min > t2[n2].min ? t1[n1].min : t2[n2].min;
941                         char max = t1[n1].max < t2[n2].max ? t1[n1].max : t2[n2].max;
942                         p.s.transitions.add(new Transition(min, max, r.s));
943                         if (t1[n1].max < t2[n2].max)
944                             n1++;
945                         else
946                             n2++;
947                     }
948                 }
949             }
950             c.deterministic = true;
951             c.removeDeadTransitions();
952             c.checkMinimizeAlways();
953             return c;
954         }

989         /**
990          * Determinizes this automaton using the given set of initial states.
991          */
992         private void determinize(Set<State> initialset) {
993             char[] points = getStartPoints();
994             // subset construction
995             Map<Set<State>, Set<State>> sets = new HashMap<>();
996             LinkedList<Set<State>> worklist = new LinkedList<>();
997             Map<Set<State>, State> newstate = new HashMap<>();
998             sets.put(initialset, initialset);
999             worklist.add(initialset);
1000             initial = new State();
1001             newstate.put(initialset, initial);
1002             while (worklist.size() > 0) {
1003                 Set<State> s = worklist.removeFirst();
1004                 State r = newstate.get(s);
1005                 for (State q : s)
1006                     if (q.accept) {
1007                         r.accept = true;
1008                         break;
1009                     }
1010                 for (int n = 0; n < points.length; n++) {
1011                     Set<State> p = new HashSet<>();
1012                     for (State q : s)
1013                         for (Transition t : q.transitions)
1014                             if (t.min <= points[n] && points[n] <= t.max)
1015                                 p.add(t.to);
1016                     if (!sets.containsKey(p)) {
1017                         sets.put(p, p);
1018                         worklist.add(p);
1019                         newstate.put(p, new State());
1020                     }
1021                     State q = newstate.get(p);
1022                     char min = points[n];
1023                     char max;
1024                     if (n + 1 < points.length)
1025                         max = (char) (points[n + 1] - 1);
1026                     else
1027                         max = Character.MAX_VALUE;
1028                     r.transitions.add(new Transition(min, max, q));
1029                 }
1030             }
1031             deterministic = true;
1032             removeDeadTransitions();
1033         }

1059         private boolean statesAgree(Transition[][] transitions, boolean[][] mark, int n1, int n2) {
1060             Transition[] t1 = transitions[n1];
1061             Transition[] t2 = transitions[n2];
1062             for (int k1 = 0, k2 = 0; k1 < t1.length && k2 < t2.length; ) {
1063                 if (t1[k1].max < t2[k2].min)
1064                     k1++;
1065                 else if (t2[k2].max < t1[k1].min)
1066                     k2++;
1067                 else {
1068                     int m1 = t1[k1].to.number;
1069                     int m2 = t2[k2].to.number;
1070                     if (m1 > m2) {
1071                         int t = m1;
1072                         m1 = m2;
1073                         m2 = t;
1074                     }
1075                     if (mark[m1][m2])
1076                         return false;
1077                     if (t1[k1].max < t2[k2].max)
1078                         k1++;
1079                     else
1080                         k2++;
1081                 }
1082             }
1083             return true;
1084         }

1087             Transition[] t1 = transitions[n1];
1088             Transition[] t2 = transitions[n2];
1089             for (int k1 = 0, k2 = 0; k1 < t1.length && k2 < t2.length; ) {
1090                 if (t1[k1].max < t2[k2].min)
1091                     k1++;
1092                 else if (t2[k2].max < t1[k1].min)
1093                     k2++;
1094                 else {
1095                     if (t1[k1].to != t2[k2].to) {
1096                         int m1 = t1[k1].to.number;
1097                         int m2 = t2[k2].to.number;
1098                         if (m1 > m2) {
1099                             int t = m1;
1100                             m1 = m2;
1101                             m2 = t;
1102                         }
1103                         if (triggers.get(m1).get(m2) == null)
1104                             triggers.get(m1).set(m2, new HashSet<IntPair>());
1105                         triggers.get(m1).get(m2).add(new IntPair(n1, n2));
1106                     }
1107                     if (t1[k1].max < t2[k2].max)
1108                         k1++;
1109                     else
1110                         k2++;
1111                 }
1112             }

1137         /**
1138          * Minimize using Huffman's algorithm.
1139          */
1140         private void minimizeHuffman() {
1141             determinize();
1142             totalize();
1143             Set<State> ss = getStates();
1144             Transition[][] transitions = new Transition[ss.size()][];
1145             State[] states = ss.toArray(new State[0]);
1146             boolean[][] mark = new boolean[states.length][states.length];
1147             ArrayList<ArrayList<HashSet<IntPair>>> triggers = new ArrayList<>();
1148             for (State state : states) {
1149                 ArrayList<HashSet<IntPair>> v = new ArrayList<>();
1150                 initialize(v, states.length);
1151                 triggers.add(v);
1152             }
1153             // initialize marks based on acceptance status and find transition arrays
1154             for (int n1 = 0; n1 < states.length; n1++) {
1155                 states[n1].number = n1;
1156                 transitions[n1] = states[n1].getSortedTransitionArray(false);
1157                 for (int n2 = n1 + 1; n2 < states.length; n2++)
1158                     if (states[n1].accept != states[n2].accept)
1159                         mark[n1][n2] = true;
1160             }
1161             // for all pairs, see if states agree
1162             for (int n1 = 0; n1 < states.length; n1++)
1163                 for (int n2 = n1 + 1; n2 < states.length; n2++)
1164                     if (!mark[n1][n2]) {
1165                         if (statesAgree(transitions, mark, n1, n2))
1166                             addTriggers(transitions, triggers, n1, n2);
1167                         else
1168                             markPair(mark, triggers, n1, n2);
1169                     }
1170             // assign equivalence class numbers to states
1171             int numclasses = 0;
1172             for (State state : states) state.number = -1;
1173             for (int n1 = 0; n1 < states.length; n1++)
1174                 if (states[n1].number == -1) {
1175                     states[n1].number = numclasses;
1176                     for (int n2 = n1 + 1; n2 < states.length; n2++)
1177                         if (!mark[n1][n2])
1178                             states[n2].number = numclasses;
1179                     numclasses++;
1180                 }
1181             // make a new state for each equivalence class
1182             State[] newstates = new State[numclasses];
1183             for (int n = 0; n < numclasses; n++)
1184                 newstates[n] = new State();
1185             // select a class representative for each class and find the new initial
1186             // state
1187             for (int n = 0; n < states.length; n++) {
1188                 newstates[states[n].number].number = n;
1189                 if (states[n] == initial)
1190                     initial = newstates[states[n].number];
1191             }
1192             // build transitions and set acceptance
1193             for (int n = 0; n < numclasses; n++) {
1194                 State s = newstates[n];
1195                 s.accept = states[s.number].accept;
1196                 for (Transition t : states[s.number].transitions)
1197                     s.transitions.add(new Transition(t.min, t.max, newstates[t.to.number]));
1198             }
1199             removeDeadTransitions();
1200         }

1212         /**
1213          * Minimize using Hopcroft's algorithm.
1214          */
1215         private void minimizeHopcroft() {
1216             determinize();
1217             Set<Transition> tr = initial.getTransitions();
1218             if (tr.size() == 1) {
1219                 Transition t = tr.iterator().next();
1220                 if (t.to == initial && t.min == Character.MIN_VALUE && t.max == Character.MAX_VALUE)
1221                     return;
1222             }
1223             totalize();
1224             // make arrays for numbered states and effective alphabet
1225             Set<State> ss = getStates();
1226             State[] states = new State[ss.size()];
1227             int number = 0;
1228             for (State q : ss) {
1229                 states[number] = q;
1230                 q.number = number++;
1231             }
1232             char[] sigma = getStartPoints();
1233             // initialize data structures
1234             ArrayList<ArrayList<LinkedList<State>>> reverse = new ArrayList<>();
1235             for (State state : states) {
1236                 ArrayList<LinkedList<State>> v = new ArrayList<>();
1237                 initialize(v, sigma.length);
1238                 reverse.add(v);
1239             }
1240             boolean[][] reverse_nonempty = new boolean[states.length][sigma.length];
1241             ArrayList<LinkedList<State>> partition = new ArrayList<>();
1242             initialize(partition, states.length);
1243             int[] block = new int[states.length];
1244             StateList[][] active = new StateList[states.length][sigma.length];
1245             StateListNode[][] active2 = new StateListNode[states.length][sigma.length];
1246             LinkedList<IntPair> pending = new LinkedList<>();
1247             boolean[][] pending2 = new boolean[sigma.length][states.length];
1248             ArrayList<State> split = new ArrayList<>();
1249             boolean[] split2 = new boolean[states.length];
1250             ArrayList<Integer> refine = new ArrayList<>();
1251             boolean[] refine2 = new boolean[states.length];
1252             ArrayList<ArrayList<State>> splitblock = new ArrayList<>();
1253             initialize(splitblock, states.length);
1254             for (int q = 0; q < states.length; q++) {
1255                 splitblock.set(q, new ArrayList<State>());
1256                 partition.set(q, new LinkedList<State>());
1257                 for (int a = 0; a < sigma.length; a++) {
1258                     reverse.get(q).set(a, new LinkedList<State>());
1259                     active[q][a] = new StateList();
1260                 }
1261             }
1262             // find initial partition and reverse edges
1263             for (State qq : states) {
1264                 int j;
1265                 if (qq.accept)
1266                     j = 0;
1267                 else
1268                     j = 1;
1269                 partition.get(j).add(qq);
1270                 block[qq.number] = j;
1271                 for (int a = 0; a < sigma.length; a++) {
1272                     char aa = sigma[a];
1273                     State p = qq.step(aa);
1274                     reverse.get(p.number).get(a).add(qq);
1275                     reverse_nonempty[p.number][a] = true;
1276                 }
1277             }
1278             // initialize active sets
1279             for (int j = 0; j <= 1; j++)
1280                 for (int a = 0; a < sigma.length; a++)
1281                     for (State qq : partition.get(j))
1282                         if (reverse_nonempty[qq.number][a])
1283                             active2[qq.number][a] = active[j][a].add(qq);
1284             // initialize pending
1285             for (int a = 0; a < sigma.length; a++) {
1286                 int a0 = active[0][a].size;
1287                 int a1 = active[1][a].size;
1288                 int j;
1289                 if (a0 <= a1)
1290                     j = 0;
1291                 else
1292                     j = 1;
1293                 pending.add(new IntPair(j, a));
1294                 pending2[a][j] = true;
1295             }
1296             // process pending until fixed point
1297             int k = 2;
1298             while (!pending.isEmpty()) {
1299                 IntPair ip = pending.removeFirst();
1300                 int p = ip.n1;
1301                 int a = ip.n2;
1302                 pending2[a][p] = false;
1303                 // find states that need to be split off their blocks
1304                 for (StateListNode m = active[p][a].first; m != null; m = m.next)
1305                     for (State s : reverse.get(m.q.number).get(a))
1306                         if (!split2[s.number]) {
1307                             split2[s.number] = true;
1308                             split.add(s);
1309                             int j = block[s.number];
1310                             splitblock.get(j).add(s);
1311                             if (!refine2[j]) {
1312                                 refine2[j] = true;
1313                                 refine.add(j);
1314                             }
1315                         }
1316                 // refine blocks
1317                 for (int j : refine) {
1318                     if (splitblock.get(j).size() < partition.get(j).size()) {
1319                         LinkedList<State> b1 = partition.get(j);
1320                         LinkedList<State> b2 = partition.get(k);
1321                         for (State s : splitblock.get(j)) {
1322                             b1.remove(s);
1323                             b2.add(s);
1324                             block[s.number] = k;
1325                             for (int c = 0; c < sigma.length; c++) {
1326                                 StateListNode sn = active2[s.number][c];
1327                                 if (sn != null && sn.sl == active[j][c]) {
1328                                     sn.remove();
1329                                     active2[s.number][c] = active[k][c].add(s);
1330                                 }
1331                             }
1332                         }
1333                         // update pending
1334                         for (int c = 0; c < sigma.length; c++) {
1335                             int aj = active[j][c].size;
1336                             int ak = active[k][c].size;
1337                             if (!pending2[c][j] && 0 < aj && aj <= ak) {
1338                                 pending2[c][j] = true;
1339                                 pending.add(new IntPair(j, c));
1340                             } else {
1341                                 pending2[c][k] = true;
1342                                 pending.add(new IntPair(k, c));
1343                             }
1344                         }
1345                         k++;
1346                     }
1347                     for (State s : splitblock.get(j))
1348                         split2[s.number] = false;
1349                     refine2[j] = false;
1350                     splitblock.get(j).clear();
1351                 }
1352                 split.clear();
1353                 refine.clear();
1354             }
1355             // make a new state for each equivalence class, set initial state
1356             State[] newstates = new State[k];
1357             for (int n = 0; n < newstates.length; n++) {
1358                 State s = new State();
1359                 newstates[n] = s;
1360                 for (State q : partition.get(n)) {
1361                     if (q == initial)
1362                         initial = s;
1363                     s.accept = q.accept;
1364                     s.number = q.number; // select representative
1365                     q.number = n;
1366                 }
1367             }
1368             // build transitions and set acceptance
1369             for (State s : newstates) {
1370                 s.accept = states[s.number].accept;
1371                 for (Transition t : states[s.number].transitions)
1372                     s.transitions.add(new Transition(t.min, t.max, newstates[t.to.number]));
1373             }
1374             removeDeadTransitions();
1375         }

1377         /**
1378          * Reverses the language of this (non-singleton) automaton while returning
1379          * the set of new initial states. (Used for
1380          * <code>minimizeBrzozowski()</code>)
1381          */
1382         private Set<State> reverse() {
1383             // reverse all edges
1384             HashMap<State, HashSet<Transition>> m = new HashMap<>();
1385             Set<State> states = getStates();
1386             Set<State> accept = getAcceptStates();
1387             for (State r : states) {
1388                 m.put(r, new HashSet<Transition>());
1389                 r.accept = false;
1390             }
1391             for (State r : states)
1392                 for (Transition t : r.getTransitions())
1393                     m.get(t.to).add(new Transition(t.min, t.max, r));
1394             for (State r : states)
1395                 r.transitions = m.get(r);
1396             // make new initial+final states
1397             initial.accept = true;
1398             initial = new State();
1399             for (State r : accept)
1400                 initial.addEpsilon(r); // ensures that all initial states are reachable
1401             deterministic = false;
1402             return accept;
1403         }

1405         /**
1406          * Adds epsilon transitions to this automaton.
1407          * This method adds extra character interval transitions that are equivalent to the given
1408          * set of epsilon transitions.
1409          *
1410          * @param pairs collection of {@link StatePair} objects representing pairs of source/destination states
1411          *              where epsilon transitions should be added
1412          */
1413         public void addEpsilons(Collection<StatePair> pairs) {
1414             expandSingleton();
1415             HashMap<State, HashSet<State>> forward = new HashMap<>();
1416             HashMap<State, HashSet<State>> back = new HashMap<>();
1417             for (StatePair p : pairs) {
1418                 HashSet<State> to = forward.get(p.s1);
1419                 if (to == null) {
1420                     to = new HashSet<>();
1421                     forward.put(p.s1, to);
1422                 }
1423                 to.add(p.s2);
1424                 HashSet<State> from = back.get(p.s2);
1425                 if (from == null) {
1426                     from = new HashSet<>();
1427                     back.put(p.s2, from);
1428                 }
1429                 from.add(p.s1);
1430             }
1431             // calculate epsilon closure
1432             LinkedList<StatePair> worklist = new LinkedList<>(pairs);
1433             HashSet<StatePair> workset = new HashSet<>(pairs);
1434             while (!worklist.isEmpty()) {
1435                 StatePair p = worklist.removeFirst();
1436                 workset.remove(p);
1437                 HashSet<State> to = forward.get(p.s2);
1438                 HashSet<State> from = back.get(p.s1);
1439                 if (to != null) {
1440                     for (State s : to) {
1441                         StatePair pp = new StatePair(p.s1, s);
1442                         if (!pairs.contains(pp)) {
1443                             pairs.add(pp);
1444                             forward.get(p.s1).add(s);
1445                             back.get(s).add(p.s1);
1446                             worklist.add(pp);
1447                             workset.add(pp);
1448                             if (from != null) {
1449                                 for (State q : from) {
1450                                     StatePair qq = new StatePair(q, p.s1);
1451                                     if (!workset.contains(qq)) {
1452                                         worklist.add(qq);
1453                                         workset.add(qq);
1454                                     }
1455                                 }
1456                             }
1457                         }
1458                     }
1459                 }
1460             }
1461             // add transitions
1462             for (StatePair p : pairs)
1463                 p.s1.addEpsilon(p.s2);
1464             deterministic = false;
1465             checkMinimizeAlways();
1466         }

1578         // analogous to getCommonPrefix, but with reverse and with bigger effort
1579         public String getCommonSuffix() {
1580             if (isSingleton())
1581                 return singleton;
1582             StringBuilder b = new StringBuilder();
1583             HashSet<State> visited = new HashSet<>();
1584     
1585             Set<State> acceptStates = getAcceptStates();
1586             if (acceptStates.size() != 1) {
1587                 return b.toString();
1588             }
1589     
1590             Map<State, Set<State>> reverse = this.getReverseTransitions(this.getStates());
1591     
1592             State s = acceptStates.iterator().next();
1593             boolean done;
1594             do {
1595                 done = true;
1596                 visited.add(s);
1597                 Set<State> fromSet = reverse.get(s);
1598     
1599                 // if there is exactly one successor
1600                 if (fromSet.size() == 1) {
1601                     State from = fromSet.iterator().next();
1602                     Character c = null;
1603                     // for all transitions between this successor and the current node...
1604                     for (Transition fromTrans : from.transitions) {
1605                         if (fromTrans.to.equals(s)) {
1606                             if (fromTrans.min == fromTrans.max && !visited.contains(from)) {
1607                                 if (c == null) {
1608                                     c = fromTrans.min;
1609                                 } else if (fromTrans.min == c) {
1610                                     // match, do nothing
1611                                 } else {
1612                                     // mismatch
1613                                     c = null;
1614                                     break;
1615                                 }
1616                             }
1617                         }
1618                     }
1619                     if (c != null) {
1620                         b.insert(0, c.charValue());
1621                         s = from;
1622                         done = false;
1623                     }
1624                 } else {
1625                     // there is more than one successor;
1626                     // try one last step
1627                     Character c = null;
1628                     for (State from : fromSet) {
1629     
1630                         for (Transition fromTrans : from.transitions) {
1631                             if (fromTrans.to.equals(s)) {
1632                                 if (fromTrans.min == fromTrans.max && !visited.contains(from)) {
1633                                     if (c == null) {
1634                                         c = fromTrans.min;
1635                                     } else if (fromTrans.min == c) {
1636                                         // match, do nothing
1637                                     } else {
1638                                         // mismatch
1639                                         c = null;
1640                                         break;
1641                                     }
1642                                 }
1643                             }
1644                         }
1645                         if (c == null) {
1646                             break;
1647                         }
1648                     }
1649                     if (c != null) {
1650                         b.insert(0, c.charValue());
1651                     }
1652                 }
1653             } while (!done);
1654     
1655             return b.toString();
