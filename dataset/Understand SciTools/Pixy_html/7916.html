65         /**
66          * Returns the contents of the given file as string.
67          *
68          * @param fileName
69          *
70          * @return
71          */
72         private String readFile(String fileName) {
73             StringBuilder fileContents = new StringBuilder();
74             try {
75                 FileReader fileReader = new FileReader(fileName);
76                 int character;
77                 fileContents = new StringBuilder();
78                 while ((character = fileReader.read()) != -1) {
79                     fileContents.append((char) character);
80                 }
81             } catch (FileNotFoundException e) {
82                 Assert.fail("File not found: " + fileName);
83             } catch (IOException e) {
84                 Assert.fail(e.getMessage());
85             }
86             return fileContents.toString();
87         }

121         private void performTest(
122             String testNumber, int expectedNumberOfSinks, int expectedNumberOfGraphs, boolean useFunctionalAnalysis,
123             int expectedNumberOfVulnerabilities
124         ) {
125             initialize("test" + testNumber + ".php", useFunctionalAnalysis);
126     
127             Assert.assertTrue(
128                 "Sinks real: " + sinks.size() + ", expected: " + expectedNumberOfSinks,
129                 sinks.size() == expectedNumberOfSinks
130             );
131     
132             List<DependencyGraph> dependencyGraphs = collectDependencyGraphs();
133             Assert.assertTrue(
134                 "Graphs real: " + dependencyGraphs.size() + ", expected: " + expectedNumberOfGraphs,
135                 dependencyGraphs.size() == expectedNumberOfGraphs
136             );
137     
138             int graphNumber = 0;
139             int actualNumberOfVulnerabilities = 0;
140     
141             for (DependencyGraph dependencyGraph : dependencyGraphs) {
142                 graphNumber++;
143                 checkDependencyGraph(testNumber, dependencyGraph, graphNumber);
144     
145                 if (checkXssGraph(testNumber, dependencyGraph, graphNumber)) {
146                     actualNumberOfVulnerabilities++;
147                 }
148             }
149     
150             Assert.assertEquals(expectedNumberOfVulnerabilities, actualNumberOfVulnerabilities);
151     
152             if (generateGraphs) {
153                 // just to make sure that you don't accidentally forget
154                 // to switch generation off, and turn checking on
155                 Assert.fail("no check performed");
156             }
157         }

159         private List<DependencyGraph> collectDependencyGraphs() {
160             List<DependencyGraph> dependencyGraphs = new LinkedList<>();
161             for (Sink sink : sinks) {
162                 dependencyGraphs.addAll(dependencyAnalysis.getDependencyGraphsForSink(sink));
163             }
164     
165             return dependencyGraphs;
166         }

168         private void checkDependencyGraph(String testNumber, DependencyGraph dependencyGraph, int graphNumber) {
169             String fileName = "test" + testNumber + "_" + graphNumber;
170             if (generateGraphs) {
171                 dependencyGraph.dumpDotUnique(fileName, this.testFilesPathWithTrailingSlash);
172             } else {
173                 String encountered = dependencyGraph.makeDotUnique(fileName);
174                 String expected = this.readFile(this.testFilesPathWithTrailingSlash + fileName + ".dot");
175                 Assert.assertEquals(expected, encountered);
176             }
177         }
178     
179         /**
180          * Checks the XSS graph.
181          *
182          *
183          * @param testNumber
184          * @param dependencyGraph
185          * @param actualNumberOfGraphs
186          *
187          * @return true if a vulnerability has been detected, false otherwise
188          */
189         private boolean checkXssGraph(
190             String testNumber, DependencyGraph dependencyGraph, int actualNumberOfGraphs
191         ) {
192             String xssFileName = "test" + testNumber + "_" + actualNumberOfGraphs + "_xss";
193             DependencyGraph relevant = this.xssAnalysis.getRelevantSubgraph(dependencyGraph);
194             Map<UninitializedNode, AbstractVulnerabilityAnalysis.InitialTaint> dangerousUninitializedNodes
195                 = this.xssAnalysis.findDangerousUninitializedNodes(relevant);
196     
197             boolean hasVulnerability = !dangerousUninitializedNodes.isEmpty();
198     
199             if (hasVulnerability) {
200                 relevant.reduceWithLeaves(dangerousUninitializedNodes.keySet());
201     
202                 if (generateGraphs) {
203                     relevant.dumpDotUnique(xssFileName, this.testFilesPathWithTrailingSlash);
204                 } else {
205                     String encountered = relevant.makeDotUnique(xssFileName);
206                     String expected = this.readFile(this.testFilesPathWithTrailingSlash + xssFileName + ".dot");
207                     Assert.assertEquals(expected, encountered);
208                 }
209             }
210     
211             return hasVulnerability;
212         }
