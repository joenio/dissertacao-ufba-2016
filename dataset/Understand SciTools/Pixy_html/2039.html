25     //  getPropagationContext ***********************************************************
26     
27         public AbstractContext getPropagationContext(Call callNode, AbstractContext context) {
28             // propagation context = incoming value at the call node under the
29             // current context
30             AbstractLatticeElement inValue = this.enclosedAnalysis.getInterproceduralAnalysisInformation().getAnalysisNode(callNode).getPhiValue(context);
31             return new FunctionalContext(inValue);
32         }
33     
34     //  getReverseTargets ***************************************************************
35     
36         public List<ReverseTarget> getReverseTargets(TacFunction exitedFunction, AbstractContext contextX) {
37     
38             //System.out.println("call to getReverseTarget!");
39     
40             List<ReverseTarget> retMe = new LinkedList<>();
41     
42             FunctionalContext context = (FunctionalContext) contextX;
43     
44             // for each call to this function...
45             for (Call callNode : exitedFunction.getCalledFrom()) {
46                 // find out possible contexts of the callee;
47                 // example: caller has two contexts c1 and c2; under both contexts,
48                 // the incoming value at the call node is e1; for the callee,
49                 // the context e1 is used for both caller contexts; hence, when
50                 // returning from the callee, we have to propagate the info back
51                 // to both caller contexts c1 and c2
52                 FunctionalAnalysisNode analysisNode = (FunctionalAnalysisNode)
53                     this.enclosedAnalysis.getInterproceduralAnalysisInformation().getAnalysisNode(callNode);
54                 if (analysisNode == null) {
55                     continue;
56                 }
57                 Set<? extends AbstractContext> calleeContexts = analysisNode.getReversePhiContexts(context.getLatticeElement());
58     
59                 // during this for loop, there is always at least one non-null set, i.e.
60                 // the following branch is entered at least once (for all exit nodes
61                 // except that of the _main function)
62                 if (calleeContexts != null) {
63                     // found matching contexts for this call node!
64                     ReverseTarget revTarget = new ReverseTarget(callNode, calleeContexts);
65                     retMe.add(revTarget);
66                 } else {
67                     System.out.println("no matching contexts found!");
68                 }
69             }
70     
71             return retMe;
72         }
