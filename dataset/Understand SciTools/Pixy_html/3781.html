3     import java.util.HashMap;
4     import java.util.LinkedList;
5     import java.util.List;
6     import java.util.Map;
7     
8     /**
9      * A generic repository of recyclable objects.
10      *
11      * Example usage: You have some *immutable* class that contains
12      * its own static repository of objects that have been instantiated
13      * from it so far. When a new object is requested from this class,
14      * it first checks its repository to see if there already is such
15      * an object. In this case, a reference to the already existing
16      * object is returned. Otherwise, a new object is created, added to
17      * the repository, and returned.
18      *
19      * Advantages: This saves memory (no redundant objects) and allows
20      * users of this class to perform quick comparisons with "==".
21      *
22      * Guide for how to design such a class:
23      * - implement the Recyclable interface
24      * - insert a private static GenericRepository<that class>
25      * - make the constructor private
26      * - use a static factory method instead
27      * - make sure that the private constructor is ONLY called by factory methods (and perhaps by static field initializers),
28      * and that these methods perform recycling by means of the repository
29      * - do not override its default equals and hashCode (would destroy the advantages mentioned above)
30      *
31      * @author Nenad Jovanovic <enji@seclab.tuwien.ac.at>
32      */
33     public class GenericRepository<E extends Recyclable> {
34         // structure hash code (Integer) -> List of Recyclable's
35         private Map<Integer, List<E>> repos;
36     
37         public GenericRepository() {
38             this.repos = new HashMap<>();
39         }
40     
41         // if the given element equals one from the repository: the
42         // repository element is returned; else: the element is
43         // entered into the repository and returned
44         public E recycle(E recycleMe) {
45     
46             if (recycleMe == null) {
47                 return recycleMe;
48             }
49     
50             Integer structureHashCode = recycleMe.structureHashCode();
51             List<E> candidates = this.repos.get(structureHashCode);
52             if (candidates == null) {
53                 // no candidates list: add recycleMe to the repos and return it
54                 List<E> recycleMeList = new LinkedList<>();
55                 recycleMeList.add(recycleMe);
56                 this.repos.put(structureHashCode, recycleMeList);
57                 return recycleMe;
58             }
59     
60             // search the candidates list
61             for (E candidate : candidates) {
62                 if (candidate.structureEquals(recycleMe)) {
63                     // recycling!
64                     return candidate;
65                 }
66             }
67     
68             // no candidate matches: add recycleMe to this list and return it
69             candidates.add(recycleMe);
70             return recycleMe;
71         }
72     }
