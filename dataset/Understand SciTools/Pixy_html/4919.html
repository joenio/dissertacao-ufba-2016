55         public Variable(String name, SymbolTable symbolTable) {
56             this.name = name;
57             this.symbolTable = symbolTable;
58             if (symbolTable.isMain()) {

103         // returns all elements recursively (i.e., the whole array tree, without
104         // the root)
105         public List<Variable> getElementsRecursive() {
106             List<Variable> retMe = new LinkedList<>();
107             Collection<Variable> directElements = this.elements.values();
108             retMe.addAll(directElements);
109             for (Variable directElement : directElements) {
110                 if (directElement.isArray()) {
111                     retMe.addAll(directElement.getElementsRecursive());
112                 }
113             }
114     
115             return retMe;
116         }

199         public boolean isArrayElementOf(Variable array) {
200             // if we don't even have an enclosing array: can't be true
201             if (this.enclosingArray == null) {
202                 return false;
203             }
204             if (this.enclosingArray.equals(array)) {
205                 // found it!
206                 return true;
207             }
208             // walk upwards recursively
209             return this.enclosingArray.isArrayElementOf(array);
210         }

230         void addElement(Variable variable) {
231             if (!variable.isArrayElement()) {
232                 throw new RuntimeException("SNH");
233             }
234             this.elements.put(variable.getIndex(), variable);
235             if (variable.getIndex() instanceof Literal) {
236                 this.literalElements.add(variable);
237             }
238         }

240         void setArrayElementAttributes(Variable enclosingArray, AbstractTacPlace index) {
241             this.isArrayElement = true;
242             this.indices = new LinkedList<>();
243             this.hasNonLiteralIndices = enclosingArray.hasNonLiteralIndices();
244             this.enclosingArray = enclosingArray;
245             if (enclosingArray.isArrayElement()) {
246                 this.topEnclosingArray = enclosingArray.getTopEnclosingArray();
247                 this.indices.addAll(enclosingArray.getIndices());
248             } else {
249                 this.topEnclosingArray = enclosingArray;
250             }
251             this.index = index;
252             this.indices.add(index);
253             if (!(index instanceof Literal)) {
254                 this.hasNonLiteralIndices = true;
255             }
256     
257             if (index.isVariable()) {
258                 ((Variable) index).addIndexFor(this);
259             }
260     
261             // if the enclosing array is a temporary, then this
262             // one also has to be a temporary
263             if (enclosingArray.isTemp()) {
264                 this.isTemp = true;
265             }
266     
267             // if the enclosing array is a global, then this
268             // one also has to be a global
269             if (enclosingArray.isGlobal()) {
270                 this.isGlobal = true;
271             }
272         }
