15     /**
16      * Transfer function for unset nodes.
17      *
18      * @author Nenad Jovanovic <enji@seclab.tuwien.ac.at>
19      */
20     public class Unset extends AbstractTransferFunction {
21         private Variable operand;
22         private boolean supported;
23     
24     // *********************************************************************************
25     // CONSTRUCTORS ********************************************************************
26     // *********************************************************************************
27     
28         public Unset(AbstractTacPlace operand) {
29     
30             // only variables can be unset
31             if (!operand.isVariable()) {
32                 throw new RuntimeException("Trying to unset a non-variable.");
33             }
34     
35             this.operand = (Variable) operand;
36             this.supported = AliasAnalysis.isSupported(this.operand);
37         }
38     
39     // *********************************************************************************
40     // OTHER ***************************************************************************
41     // *********************************************************************************
42     
43         public AbstractLatticeElement transfer(AbstractLatticeElement inX) {
44     
45             // if this statement is not supported by our alias analysis,
46             // we simply ignore it
47             if (!supported) {
48                 return inX;
49             }
50     
51             LiteralLatticeElement in = (LiteralLatticeElement) inX;
52             LiteralLatticeElement out = new LiteralLatticeElement(in);
53     
54             // unsetting a variable means setting it to null for literal analysis
55             Set<Variable> mustAliases = new HashSet<>();
56             mustAliases.add(operand);
57             Set<Variable> mayAliases = Collections.emptySet();
58             out.assignSimple(operand, Literal.NULL, mustAliases, mayAliases);
59     
60             return out;
61         }
62     }
