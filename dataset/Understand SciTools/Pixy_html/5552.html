10     public class CallString {
11         // a list of Call's; never longer than the k-size of the analysis
12         private LinkedList<Call> callNodeList;
13     
14         // creates the empty call string
15         public CallString() {
16             this.callNodeList = new LinkedList<>();
17         }
18     
19         // shall only be used by CallString.append
20         private CallString(LinkedList<Call> callNodeList) {
21             this.callNodeList = callNodeList;
22         }
23     
24         public CallString append(Call callNode, int kSize) {
25             LinkedList<Call> newList = new LinkedList<>(this.callNodeList);
26             newList.add(callNode);
27             if (newList.size() > kSize) {
28                 newList.remove(0);
29             }
30             return new CallString(newList);
31         }
32     
33         // returns the last (rightmost) call node
34         public Call getLast() {
35             return this.callNodeList.getLast();
36         }
37     
38         public LinkedList<Call> getCallNodeList() {
39             return this.callNodeList;
40         }
41     
42         public int hashCode() {
43             return this.callNodeList.hashCode();
44         }
45     
46         public boolean equals(Object obj) {
47     
48             if (obj == this) {
49                 return true;
50             }
51     
52             if (!(obj instanceof CallString)) {
53                 return false;
54             }
55             CallString comp = (CallString) obj;
56     
57             return this.callNodeList.equals(comp.getCallNodeList());
58         }
59     
60         public String dump() {
61             StringBuilder b = new StringBuilder();
62             for (Call callNode : this.callNodeList) {
63                 b.append(callNode.getFileName());
64                 b.append(":");
65                 b.append(callNode.getOriginalLineNumber());
66                 b.append("\n");
67             }
68             return b.toString();
69         }
