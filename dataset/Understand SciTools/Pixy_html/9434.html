65         // constructor for default element (to be called by "initDefault"):
66         // basically initializes everything to UNINIT;
67         // TODO: the code for this can probably be simplified dramatically ;
68         // also, the comments inside this function are partly obsolete
69         private DependencyLatticeElement(
70             List<AbstractTacPlace> places,
71             ConstantsTable constantsTable,
72             List<TacFunction> functions,
73             SymbolTable superSymbolTable,
74             Variable memberPlace) {
75     
76             // initialize base mapping for variables: UNINIT
77             // (note: array elements have no explicit array label: their label is
78             // that of their top enclosing array)
79             this.placeToDep = new HashMap<>();
80             this.arrayLabels = new HashMap<>();
81             for (AbstractTacPlace place : places) {
82                 if ((place instanceof Variable) &&
83                     place.getVariable().isArrayElement() &&
84                     place.getVariable().hasNonLiteralIndices()) {
85                     // if this is a non-literal array element:
86                     // don't add a mapping
87                 } else {
88                     this.placeToDep.put(place, DependencySet.UNINIT);
89                 }
90     
91                 // arrayLabels are not defined for variables that are array-elements
92                 if ((place instanceof Variable) && !place.getVariable().isArrayElement()) {
93                     this.arrayLabels.put((Variable) place, DependencySet.UNINIT);
94                 }
95             }
96     
97             // initialize function return values:
98             // if a function has no return statement, the return variable is not touched;
99             // since the real-world return value is then "NULL", the default mapping
100             // for return variables has to have the same properties as the default
101             // mapping for NULL (i.e., harmless);
102             for (AbstractTacPlace place : places) {
103                 if ((place instanceof Variable) && place.getVariable().isReturnVariable()) {
104                     this.placeToDep.put(place, DependencySet.UNINIT);
105                     this.arrayLabels.put((Variable) place, DependencySet.UNINIT);
106                 }
107             }
108     
109             // special member variable
110             this.placeToDep.put(memberPlace, DependencySet.UNINIT);
111             this.arrayLabels.put(memberPlace, DependencySet.UNINIT);
112     
113             // initialize constants
114             for (Constant constant : constantsTable.getConstants().values()) {
115                 this.placeToDep.put(constant, DependencySet.UNINIT);
116             }
117     
118             // initialize local function variables & parameters to HARMLESS;
119             // locals of main function == globals of the program:
120             // their initialization depends on the register_globals setting:
121             // true: UNINIT
122             // false: like other locals
123             for (TacFunction function : functions) {
124                 if (function.isMain()) {
125                     if (MyOptions.optionG) {
126                         // if register_globals is active, we don't change the conservative
127                         // base mapping (tainted/dirty) for the locals of the main function;
128                         // NOTE: this is also quite useless, as most of the rest of this function;
129                         // should be cleaned up some day...
130                         continue;
131                     }
132                 }
133                 SymbolTable symbolTable = function.getSymbolTable();
134                 for (Variable variable : symbolTable.getVariables().values()) {
135                     // array elements are handled along with their root
136                     if (variable.isArrayElement()) {
137                         continue;
138                     }
139                     // init the whole tree below this variable to untainted
140                     // (note that using setTree wouldn't work here)
141                     this.initTree(variable, DependencySet.UNINIT);
142                     this.arrayLabels.put(variable, DependencySet.UNINIT);
143                 }
144             }
145     
146             // untaint the whole $_SESSION array
147             Variable sess = superSymbolTable.getVariable("$_SESSION");
148             this.initTree(sess, DependencySet.UNINIT);
149             this.arrayLabels.put(sess, DependencySet.UNINIT);
150     
151             // untaint harmless superglobals;
152             // fill this list with harmless superglobals that must be set to
153             // harmless;
154             List<Variable> harmlessSuperGlobals = new LinkedList<>();
155     
156             // NOTE: WHEN ADDING A HARMLESS SUPERGLOBAL HERE, YOU SHOULD ALSO ADD IT TO
157             // TacConverter.addSuperGlobalElements()
158     
159             // case-sensitive, and that's OK
160             addHarmlessServerVar(harmlessSuperGlobals, superSymbolTable, "SERVER_NAME");
161             addHarmlessServerVar(harmlessSuperGlobals, superSymbolTable, "HTTP_HOST");
162             addHarmlessServerVar(harmlessSuperGlobals, superSymbolTable, "HTTP_ACCEPT_LANGUAGE");
163             addHarmlessServerVar(harmlessSuperGlobals, superSymbolTable, "SERVER_SOFTWARE");
164             addHarmlessServerVar(harmlessSuperGlobals, superSymbolTable, "PHP_AUTH_USER");
165             addHarmlessServerVar(harmlessSuperGlobals, superSymbolTable, "PHP_AUTH_PW");
166             addHarmlessServerVar(harmlessSuperGlobals, superSymbolTable, "PHP_AUTH_TYPE");
167             addHarmlessServerVar(harmlessSuperGlobals, superSymbolTable, "SCRIPT_NAME");
168             addHarmlessServerVar(harmlessSuperGlobals, superSymbolTable, "SCRIPT_FILENAME");
169             addHarmlessServerVar(harmlessSuperGlobals, superSymbolTable, "REQUEST_URI");
170             addHarmlessServerVar(harmlessSuperGlobals, superSymbolTable, "QUERY_STRING");
171             addHarmlessServerVar(harmlessSuperGlobals, superSymbolTable, "SCRIPT_URI");
172     
173             // same for 'argv'
174             Variable argv = superSymbolTable.getVariable("$_SERVER[argv]");
175             if (argv != null) {
176                 for (Variable argvElement : argv.getElements()) {
177                     harmlessSuperGlobals.add(argvElement);
178                 }
179             }
180             argv = superSymbolTable.getVariable("$HTTP_SERVER_VARS[argv]");
181             if (argv != null) {
182                 for (Variable argvElement : argv.getElements()) {
183                     harmlessSuperGlobals.add(argvElement);
184                 }
185             }
186     
187             // collection of harmless variables finished, now make them harmless
188             for (Variable harmlessSuperGlobal : harmlessSuperGlobals) {
189                 // if it is null, it hasn't been used in the php script
190                 if (harmlessSuperGlobal != null) {
191                     this.placeToDep.put(harmlessSuperGlobal, DependencySet.UNINIT);
192                 }
193             }
194         }

246         // returns the non-default dependency for this place if that mapping exists,
247         // or the default dependency otherwise
248         private DependencySet getDepFrom(
249             AbstractTacPlace place, Map<? extends AbstractTacPlace, DependencySet> readFrom) {
250     
251             if (place instanceof Literal) {
252                 throw new RuntimeException("SNH any longer");
253             }
254     
255             // if we encounter a non-literal array element:
256             // dependency depends solely on the array label of the enclosing array
257             if (place instanceof Variable) {
258                 Variable var = (Variable) place;
259                 if (var.isArrayElement() && var.hasNonLiteralIndices()) {
260                     return getArrayLabel(var.getTopEnclosingArray());
261                 }
262             }
263     
264             // return the non-default mapping (if there is one)
265             DependencySet nonDefaultDep = readFrom.get(place);
266             if (nonDefaultDep != null) {
267                 return nonDefaultDep;
268             }
269     
270             // return the default mapping if there is no non-default one
271             DependencySet defaultDep = getDefaultDep(place);
272             if (defaultDep == null) {
273                 throw new RuntimeException("SNH: " + place);
274             } else {
275                 return defaultDep;
276             }
277         }

343         private void setDep(AbstractTacPlace place, DependencySet dependencySet) {
344     
345             if (place instanceof Literal) {
346                 throw new RuntimeException("SNH");
347             }
348             if (place instanceof Variable && place.getVariable().isMember()) {
349                 // we don't want to modify the special member variable
350                 return;
351             }
352     
353             if (getDefaultDep(place).equals(dependencySet)) {
354                 // if the target dependency is the default, we simply remove the mapping
355                 this.placeToDep.remove(place);
356             } else {
357                 this.placeToDep.put(place, dependencySet);
358             }
359         }

363         private void lubDep(AbstractTacPlace place, DependencySet dependencySet) {
364     
365             if (place instanceof Literal) {
366                 throw new RuntimeException("SNH");
367             }
368             if (place instanceof Variable && place.getVariable().isMember()) {
369                 // we don't want to modify the special member variable
370                 return;
371             }
372     
373             DependencySet oldDep = this.getDep(place);
374             DependencySet resultDep = DependencySet.lub(oldDep, dependencySet);
375             if (getDefaultDep(place).equals(resultDep)) {
376                 // if the target dependency is the default, we simply remove the mapping
377                 this.placeToDep.remove(place);
378             } else {
379                 this.placeToDep.put(place, resultDep);
380             }
381         }

387             if (var.isArrayElement()) {

404     
405         private void lubArrayLabel(Variable var, DependencySet dependencySet) {
406             if (var.isMember()) {
407                 // we don't want to modify the special member variable
408                 return;
409             }
410     
411             DependencySet oldDep = this.getArrayLabel(var);
412             DependencySet resultDep = DependencySet.lub(dependencySet, oldDep);
413             if (resultDep.equals(getDefaultArrayLabel(var))) {
414                 // if the resulting dependency is equal to this variable's default
415                 // dependency, we simply have to remove the default mapping
416                 this.arrayLabels.remove(var);
417             } else {
418                 // replace the non-default mapping
419                 this.arrayLabels.put(var, resultDep);
420             }
421         }

454     //  lub ****************************************************************************
455     
456         // lubs the given lattice element over <<this>> lattice element
457         public void lub(AbstractLatticeElement foreignX) {
458     
459             DependencyLatticeElement foreign = (DependencyLatticeElement) foreignX;
460     
461             // DEPS ***
462     
463             // lub over my non-default mappings;
464             // this one is necessary because we must not modify a map while
465             // iterating over it
466             Map<AbstractTacPlace, DependencySet> newPlaceToDep =
467                 new HashMap<>(this.placeToDep);
468             for (Map.Entry<AbstractTacPlace, DependencySet> myEntry : this.placeToDep.entrySet()) {
469                 AbstractTacPlace myPlace = myEntry.getKey();
470                 DependencySet myDep = myEntry.getValue();
471                 DependencySet foreignDep = foreign.getDep(myPlace);
472                 newPlaceToDep.put(myPlace, DependencySet.lub(myDep, foreignDep));
473             }
474             this.placeToDep = newPlaceToDep;
475     
476             // lub the remaining non-default mappings of "foreign" over my
477             // default mappings
478             for (Map.Entry<AbstractTacPlace, DependencySet> foreignEntry : foreign.getPlaceToDep().entrySet()) {
479                 AbstractTacPlace foreignPlace = foreignEntry.getKey();
480                 DependencySet foreignDep = foreignEntry.getValue();
481     
482                 DependencySet myDep = this.getNonDefaultDep(foreignPlace);
483                 // make sure that we handle my default mappings now
484                 if (myDep == null) {
485                     myDep = getDefaultDep(foreignPlace);
486                     this.placeToDep.put(foreignPlace, DependencySet.lub(foreignDep, myDep));
487                 }
488             }
489     
490             // cleaning pass: remove defaults
491             for (Iterator<Map.Entry<AbstractTacPlace, DependencySet>> iter = this.placeToDep.entrySet().iterator(); iter.hasNext(); ) {
492                 Map.Entry<AbstractTacPlace, DependencySet> entry = iter.next();
493                 AbstractTacPlace place = entry.getKey();
494                 DependencySet dep = entry.getValue();
495                 if (getDefaultDep(place).equals(dep)) {
496                     iter.remove();
497                 }
498             }
499     
500             // ARRAY LABELS ***
501     
502             // lub over my non-default mappings
503             Map<Variable, DependencySet> newArrayLabels = new HashMap<>(this.arrayLabels);
504             for (Map.Entry<Variable, DependencySet> myEntry : this.arrayLabels.entrySet()) {
505                 Variable myVar = myEntry.getKey();
506                 DependencySet myArrayLabel = myEntry.getValue();
507                 DependencySet foreignArrayLabel = foreign.getArrayLabel(myVar);
508                 newArrayLabels.put(myVar, DependencySet.lub(myArrayLabel, foreignArrayLabel));
509             }
510             this.arrayLabels = newArrayLabels;
511     
512             // lub the remaining non-default mappings of "foreign" over my
513             // default mappings
514             for (Map.Entry<Variable, DependencySet> foreignEntry : foreign.getArrayLabels().entrySet()) {
515                 Variable foreignVar = foreignEntry.getKey();
516                 DependencySet foreignArrayLabel = foreignEntry.getValue();
517     
518                 // try non-default mapping:
519                 DependencySet myArrayLabel = getNonDefaultArrayLabel(foreignVar);
520                 if (myArrayLabel == null) {
521                     // fetch default mapping:
522                     myArrayLabel = getDefaultArrayLabel(foreignVar);
523                     this.arrayLabels.put(foreignVar, DependencySet.lub(myArrayLabel, foreignArrayLabel));
524                 }
525             }
526     
527             // cleaning pass: remove defaults
528             for (Iterator<Map.Entry<Variable, DependencySet>> iter = this.arrayLabels.entrySet().iterator(); iter.hasNext(); ) {
529                 Map.Entry<Variable, DependencySet> entry = iter.next();
530                 Variable var = entry.getKey();
531                 DependencySet arrayLabel = entry.getValue();
532                 if (getDefaultArrayLabel(var).equals(arrayLabel)) {
533                     iter.remove();
534                 }
535             }
536         }

547         // inits the deps for all literal array elements in the
548         // tree specified by the given root;

564         // mustAliases and mayAliases: of left; mustAliases always have to
565         // include left itself
566         public void assign(Variable left, Set<Variable> mustAliases, Set<Variable> mayAliases, AbstractCfgNode cfgNode) {
567             // dependency to be assigned to the left side
568             DependencySet dep = DependencySet.create(DependencyLabel.create(cfgNode));
569     
570             // case distinguisher for the left variable
571             int leftCase;
572     
573             // find out more about the left variable
574             if (!left.isArrayElement()) {
575                 // left: not an array element
576                 if (!left.isArray()) {
577                     // left: neither an array nor an array element
578                     leftCase = 1;
579                 } else {
580                     // left: an array that is not an array element
581                     leftCase = 2;
582                 }
583             } else {
584                 // left: array element
585                 if (!left.hasNonLiteralIndices()) {
586                     // left: literal array element
587                     leftCase = 3;
588                 } else {
589                     // left: non-literal array element
590                     leftCase = 4;
591                 }
592             }
593     
594             // take appropariate actions
595             switch (leftCase) {
596     
597                 // not an array element and not known as array ("normal variable")
598                 case 1: {
599                     // strong update for must-aliases (including left itself)
600                     for (Variable mustAlias : mustAliases) {
601                         this.setDep(mustAlias, dep);
602                         this.setArrayLabel(mustAlias, dep);
603                     }
604     
605                     // weak update for may-aliases
606                     for (Variable mayAlias : mayAliases) {
607                         this.lubDep(mayAlias, dep);
608                         this.lubArrayLabel(mayAlias, dep);
609                     }
610     
611                     break;
612                 }
613     
614                 // array, but not an array element
615                 case 2: {
616                     // set target.caFlag
617                     this.setArrayLabel(left, dep);
618     
619                     // no strong overlap here, but: set the whole subtree to resultDep
620                     this.setWholeTree(left, dep);
621                     break;
622                 }
623     
624                 // array element (and maybe an array) without non-literal indices
625                 case 3: {
626                     // lub target.root.caFlag
627                     this.lubArrayLabel(left.getTopEnclosingArray(), dep);
628     
629                     // no strong overlap here, but: set the whole subtree to resultDep
630                     this.setWholeTree(left, dep);
631                     break;
632                 }
633     
634                 // array element (and maybe an array) with non-literal indices
635                 case 4: {
636                     // lub target.root.caFlag
637                     this.lubArrayLabel(left.getTopEnclosingArray(), dep);
638     
639                     // "weak overlap for all MI variables of left"
640                     // here: lub the whole subtrees of all MI variables of left
641                     for (Variable miVar : this.getMiList(left)) {
642                         this.lubWholeTree(miVar, dep);
643                     }
644                     break;
645                 }
646     
647                 default:
648                     throw new RuntimeException("SNH");
649             }
650         }

678     
679         // returns a list of array elements that are maybe identical
680         // to the given array element; only array elements without
681         // non-literal indices are returned
682         List<Variable> getMiList(Variable var) {
683             if (!var.isArrayElement()) {
684                 throw new RuntimeException("SNH");
685             }
686             if (!var.hasNonLiteralIndices()) {
687                 throw new RuntimeException("SNH");
688             }
689     
690             // the list to be returned (contains Variables)
691             List<Variable> miList = new LinkedList<>();
692     
693             // root of the array tree, indices of the array element
694             Variable root = var.getTopEnclosingArray();
695             List<AbstractTacPlace> indices = var.getIndices();
696     
697             this.miRecurse(miList, root, new LinkedList<>(indices));
698             return miList;
699         }

701     // miRecurse ***********************************************************************
702     
703         // CAUTION: the indices list is modified inside this method, so you might
704         // want to pass a shallow copy instead of a reference to the list
705         private void miRecurse(List<Variable> miList, Variable root, List<AbstractTacPlace> indices) {
706             /*
707              * - separate head from the indices list
708              * - if this first index is literal:
709              *   - if the remaining indices list is empty:
710              *     add the target array element to the miList
711              *   - else: recurse with target array element and
712              *     the remaining list
713              * - else (first index is non-literal):
714              *   - if the remaining indices list is empty:
715              *     add all literal array elements to the miList
716              *   - else: recurse for all literal array elements and
717              *     <<a copy>> of the remaining indices list: otherwise,
718              *     the different branches would operate on the same
719              *     indices list
720              *
721              */
722     
723             // proceeding only makes sense if the considered root has known
724             // array elements
725             if (!root.isArray()) {
726                 return;
727             }
728     
729             AbstractTacPlace index = indices.remove(0);
730             if (index instanceof Literal) {
731     
732                 Variable target = root.getElement(index);
733                 // it is possible that the considered array doesn't have this index
734                 if (target != null) {
735                     if (indices.isEmpty()) {
736                         miList.add(target);
737                     } else {
738                         this.miRecurse(miList, target, indices);
739                     }
740                 }
741             } else {
742                 List<Variable> literalElements = root.getLiteralElements();
743                 if (indices.isEmpty()) {
744                     miList.addAll(literalElements);
745                 } else {
746                     for (Variable target : literalElements) {
747                         this.miRecurse(miList, target, new LinkedList<>(indices));
748                     }
749                 }
750             }
751         }

753     //  resetVariables *****************************************************************
754     
755         // resets all variables that belong to the given symbol table
756         // (by removing their non-default mapping)
757         public void resetVariables(SymbolTable symTab) {
758     
759             // reset deps
760             for (Iterator<Map.Entry<AbstractTacPlace, DependencySet>> iter = this.placeToDep.entrySet().iterator(); iter.hasNext(); ) {
761                 Map.Entry<AbstractTacPlace, DependencySet> entry = iter.next();
762                 AbstractTacPlace place = entry.getKey();
763     
764                 if (!(place instanceof Variable)) {
765                     // nothing to do for non-variables (i.e., constants)
766                     continue;
767                 }
768     
769                 Variable var = (Variable) place;
770                 if (var.belongsTo(symTab)) {
771                     iter.remove();
772                 }
773             }
774     
775             // reset array labels
776             for (Iterator<Map.Entry<Variable, DependencySet>> iter = this.arrayLabels.entrySet().iterator(); iter.hasNext(); ) {
777                 Map.Entry<Variable, DependencySet> entry = iter.next();
778                 AbstractTacPlace place = entry.getKey();
779     
780                 if (!(place != null)) {
781                     // nothing to do for non-variables (i.e., constants)
782                     // Note: Is this possible here at all as of the Map contents data types?
783                     continue;
784                 }
785     
786                 Variable var = (Variable) place;
787                 if (var.belongsTo(symTab)) {
788                     iter.remove();
789                 }
790             }
791         }

865     //  copyGlobalLike *****************************************************************
866     
867         // copies the non-default dependency/label mappings for "global-like" places
868         // from interIn (i.e., global variables, superglobal variables, and
869         // constants)
870         public void copyGlobalLike(DependencyLatticeElement interIn) {
871     
872             // dependency mappings
873             for (Map.Entry<AbstractTacPlace, DependencySet> entry : interIn.getPlaceToDep().entrySet()) {
874                 AbstractTacPlace origPlace = entry.getKey();
875                 DependencySet origDep = entry.getValue();
876     
877                 // decide whether to copy this place
878                 boolean copyMe = false;
879                 if (origPlace instanceof Constant) {
880                     copyMe = true;
881                 } else if (origPlace instanceof Variable) {
882                     Variable origVar = (Variable) origPlace;
883                     if (origVar.isGlobal() || origVar.isSuperGlobal()) {
884                         copyMe = true;
885                     }
886                 } else {
887                     throw new RuntimeException("SNH: " + origPlace);
888                 }
889     
890                 if (copyMe) {
891                     this.setDep(origPlace, origDep);
892                 }
893             }
894     
895             // array label mappings
896             for (Map.Entry<Variable, DependencySet> entry : interIn.getArrayLabels().entrySet()) {
897                 AbstractTacPlace origPlace = entry.getKey();
898                 DependencySet origArrayLabel = entry.getValue();
899     
900                 // decide whether to copy this place
901                 boolean copyMe = false;
902                 if (origPlace != null) {
903                     Variable origVar = (Variable) origPlace;
904                     if (origVar.isGlobal() || origVar.isSuperGlobal()) {
905                         copyMe = true;
906                     }
907                 } else {
908                     throw new RuntimeException("SNH: " + origPlace);
909                 }
910     
911                 if (copyMe) {
912                     this.setArrayLabel((Variable) origPlace, origArrayLabel);
913                 }
914             }
915         }

917     //  copyGlobalLike *****************************************************************
918     
919         // analogous to the one-parameter copyGlobalLike, but also takes into account
920         // MOD-info; currently, constants are not considered as global-like for this
921         public void copyGlobalLike(DependencyLatticeElement interIn, DependencyLatticeElement intraIn,
922                                    Set<AbstractTacPlace> calleeMod) {
923     
924             // dependency mappings
925             for (Map.Entry<AbstractTacPlace, DependencySet> entry : interIn.getPlaceToDep().entrySet()) {
926                 AbstractTacPlace interPlace = entry.getKey();
927                 DependencySet interDep = entry.getValue();
928     
929                 // decide whether to copy this place
930                 boolean copyMe = false;
931                 if (interPlace instanceof Constant) {
932                     copyMe = true;
933                 } else if (interPlace instanceof Variable) {
934                     Variable interVar = (Variable) interPlace;
935                     if (interVar.isGlobal() || interVar.isSuperGlobal()) {
936                         copyMe = true;
937                     }
938                 } else {
939                     throw new RuntimeException("SNH: " + interPlace);
940                 }
941     
942                 if (copyMe) {
943                     // ok, we have decided that this place shall be copied;
944                     // now we have to decide from where we want to copy it;
945                     // NOTE: here we exclude constants from special treatment
946                     if (interPlace instanceof Constant || calleeMod.contains(interPlace)) {
947                         this.setDep(interPlace, interDep);
948                     } else {
949                         this.setDep(interPlace, intraIn.getDep(interPlace));
950                     }
951                 }
952             }
953     
954             // array label mappings
955             for (Map.Entry<Variable, DependencySet> entry : interIn.getArrayLabels().entrySet()) {
956                 AbstractTacPlace interPlace = entry.getKey();
957                 DependencySet interArrayLabel = entry.getValue();
958     
959                 // decide whether to copy this place
960                 boolean copyMe = false;
961                 if (interPlace != null) {
962                     Variable interVar = (Variable) interPlace;
963                     if (interVar.isGlobal() || interVar.isSuperGlobal()) {
964                         copyMe = true;
965                     }
966                 } else {
967                     throw new RuntimeException("SNH: " + interPlace);
968                 }
969     
970                 if (copyMe) {
971                     if (calleeMod.contains(interPlace)) {
972                         this.setArrayLabel((Variable) interPlace, interArrayLabel);
973                     } else {
974                         this.setArrayLabel((Variable) interPlace, intraIn.getArrayLabel(interPlace));
975                     }
976                 }
977             }
978         }

980     //  copyMainTemporaries ****************************************************************
981     
982         // copies the dependency/label mappings for local temporaries of the main function
983         public void copyMainTemporaries(DependencyLatticeElement origElement) {
984     
985             // dependency mappings
986             for (Map.Entry<AbstractTacPlace, DependencySet> entry : origElement.getPlaceToDep().entrySet()) {
987                 AbstractTacPlace origPlace = entry.getKey();
988                 DependencySet origDep = entry.getValue();
989     
990                 // nothing to do for non-variables, non-main's, and non-temporaries
991                 if (!(origPlace instanceof Variable)) {
992                     continue;
993                 }
994                 Variable origVar = (Variable) origPlace;
995                 SymbolTable symTab = origVar.getSymbolTable();
996                 if (!symTab.isMain()) {
997                     continue;
998                 }
999                 if (!origVar.isTemp()) {
1000                     continue;
1001                 }
1002     
1003                 //System.out.println("setting dependency for " + origVar);
1004                 this.setDep(origVar, origDep);
1005             }
1006     
1007             // array label mappings
1008             for (Map.Entry<Variable, DependencySet> entry : origElement.getArrayLabels().entrySet()) {
1009                 AbstractTacPlace origPlace = entry.getKey();
1010                 DependencySet origArrayLabel = entry.getValue();
1011     
1012                 // nothing to do for non-variables, non-main's, and non-temporaries
1013                 if (!(origPlace != null)) {
1014                     continue;
1015                 }
1016                 Variable origVar = (Variable) origPlace;
1017                 SymbolTable symTab = origVar.getSymbolTable();
1018                 if (!symTab.isMain()) {
1019                     continue;
1020                 }
1021                 if (!origVar.isTemp()) {
1022                     continue;
1023                 }
1024     
1025                 this.setArrayLabel(origVar, origArrayLabel);
1026             }
1027         }
