17     public class CallPreparation extends AbstractTransferFunction {
18         private List<TacActualParameter> actualParams;
19         private List<TacFormalParameter> formalParams;
20         private TacFunction caller;
21         private TacFunction callee;
22         private TypeAnalysis typeAnalysis;
23

27     
28         public CallPreparation(
29             List<TacActualParameter> actualParams, List<TacFormalParameter> formalParams, TacFunction caller, TacFunction callee,
30             TypeAnalysis typeAnalysis
31         ) {
32             this.actualParams = actualParams;
33             this.formalParams = formalParams;
34             this.caller = caller;
35             this.callee = callee;
36             this.typeAnalysis = typeAnalysis;
37         }
38     
39     //  *********************************************************************************
40     //  OTHER ***************************************************************************
41     //  *********************************************************************************
42     
43         public AbstractLatticeElement transfer(AbstractLatticeElement inX) {
44     
45             TypeLatticeElement in = (TypeLatticeElement) inX;
46             TypeLatticeElement out = new TypeLatticeElement(in);
47     
48             // set formal params...
49     
50             // use a ListIterator for formals because we might need to step back (see below)
51             ListIterator<TacFormalParameter> formalIter = formalParams.listIterator();
52             Iterator<TacActualParameter> actualIter = actualParams.iterator();
53     
54             // for each formal parameter...
55             while (formalIter.hasNext()) {
56                 TacFormalParameter formalParam = formalIter.next();
57     
58                 if (actualIter.hasNext()) {
59                     // there is a corresponding actual parameter; advance iterator
60                     TacActualParameter actualParam = actualIter.next();
61     
62                     // set the formal
63                     out.assign(formalParam.getVariable(), actualParam.getPlace());
64                 } else {
65                     // there is no corresponding actual parameter, use default values
66                     // for the remaining formal parameters
67     
68                     // make one step back (so we can use a while loop)
69                     formalIter.previous();
70     
71                     while (formalIter.hasNext()) {
72                         formalParam = formalIter.next();
73     
74                         if (formalParam.hasDefault()) {
75                             ControlFlowGraph defaultControlFlowGraph = formalParam.getDefaultControlFlowGraph();
76     
77                             // default CFG's have no branches;
78                             // start at the CFG's head and apply all transfer functions
79                             AbstractCfgNode defaultNode = defaultControlFlowGraph.getHead();
80                             while (defaultNode != null) {
81                                 AbstractTransferFunction tf = this.typeAnalysis.getTransferFunction(defaultNode);
82                                 out = (TypeLatticeElement) tf.transfer(out);
83                                 defaultNode = defaultNode.getSuccessor(0);
84                             }
85                         } else {
86                             // missing actual parameter;
87                             // we have already generated a warning for this during conversion;
88                             // simply ignore it (=ok, is exactly what PHP does)
89                         }
90                     }
91                 }
92             }
93     
94             // reset all local variables that belong to the symbol table of the
95             // caller; shortcut: if the caller is main, we don't have to do
96             // this (since there are no real local variables in the main function)
97             SymbolTable callerSymTab = this.caller.getSymbolTable();
98             if (!callerSymTab.isMain()) {
99                 // only do this for non-recursive calls;
100                 // EFF: it might be better to reset everything except the formal params;
101                 // TODO: also think about correctness
102                 if (!(callee == caller)) {
103                     out.resetVariables(callerSymTab);
104                 }
105             } else {
106                 // for the main function, we can at least reset the temporary variables
107                 out.resetTemporaries(callerSymTab);
108             }
109     
110             return out;
111         }
