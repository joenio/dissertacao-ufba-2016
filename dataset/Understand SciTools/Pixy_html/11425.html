20     public class CallPreparation extends AbstractTransferFunction {
21         private List<List<Variable>> cbrParams;

30         public CallPreparation(TacFunction caller, AliasAnalysis aliasAnalysis, at.ac.tuwien.infosys.www.pixy.conversion.cfgnodes.CallPreparation cfgNode) {
31     
32             this.cfgNode = cfgNode;
33             this.cbrParams = cfgNode.getCbrParams();
34             this.caller = caller;
35             this.aliasAnalysis = aliasAnalysis;
36         }

41     
42         public AbstractLatticeElement transfer(AbstractLatticeElement inX) {
43     
44             AliasLatticeElement in = (AliasLatticeElement) inX;
45             AliasLatticeElement out = new AliasLatticeElement(in);
46     
47             // see alias analysis tutorial for an explanation how this works
48     
49             if (!this.cbrParams.isEmpty()) {
50     
51                 // note: alias analysis does not have to care about default params
52     
53                 // note: at this point, it should already be ensured that there
54                 // are at least as many formal params as actual params
55     
56                 SymbolTable placeHolderSymTab = new SymbolTable("_placeHolder");
57                 // placeholder -> real formal
58                 Map<Variable, Variable> replacements = new HashMap<>();
59     
60                 // for all cbr-params...
61                 for (List<Variable> pairList : this.cbrParams) {
62                     Iterator<Variable> pairListIterator = pairList.iterator();
63                     Variable actualVar = pairListIterator.next();
64                     Variable formalVar = pairListIterator.next();
65     
66                     Variable formalPlaceHolder = new Variable(formalVar.getName(), placeHolderSymTab);
67                     replacements.put(formalPlaceHolder, formalVar);
68     
69                     // add the formal's placeholder to the actual's must-alias-group
70                     out.addToGroup(formalPlaceHolder, actualVar);
71     
72                     // see the function in MayAliases.java for an explanation
73                     out.createAdjustedPairCopies(actualVar, formalPlaceHolder);
74                 }
75     
76                 // remove all local variables that belong to the symbol table of the
77                 // caller; shortcut: if the caller is main, we don't have to do
78                 // this (since there are no real local variables in the main function)
79                 SymbolTable callerSymTab = this.caller.getSymbolTable();
80                 if (!callerSymTab.isMain()) {
81                     out.removeVariables(callerSymTab);
82                 }
83     
84                 // replace the placeholders by the formals of the callee
85                 out.replace(replacements);
86             } else {
87                 // there are no cbr params; hence, we can simply remove
88                 // all local variables of the caller
89                 // (note: at this point, the only locals that can appear
90                 // in the analysis info are those of the caller, so this
91                 // reduces to removing all local variables)
92                 out.removeLocals();
93             }
94     
95             // recycle
96             out = (AliasLatticeElement) this.aliasAnalysis.recycle(out);
97     
98             return out;
99         }
