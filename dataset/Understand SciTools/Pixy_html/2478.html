1     package at.ac.tuwien.infosys.www.pixy.analysis.interprocedural;

3     import at.ac.tuwien.infosys.www.pixy.analysis.interprocedural.callstring.CallStringContext;

5     import java.util.HashMap;
6     import java.util.HashSet;
7     import java.util.Map;
8     import java.util.Set;
9     
10     /**
11      * Maps from position to position for a certain call node.
12      *
13      * @author Nenad Jovanovic <enji@seclab.tuwien.ac.at>
14      */
15     public class ConnectorFunction {
16         // CallStringContext -> CallStringContext
17         private Map<CallStringContext, CallStringContext> pos2pos;
18     
19         // reverse mapping of pos2pos:
20         // CallStringContext -> Set of CSContexts
21         private Map<CallStringContext, Set<CallStringContext>> reverse;
22     
23         // creates an empty connector function
24         public ConnectorFunction() {
25             this.pos2pos = new HashMap<>();
26             this.reverse = new HashMap<>();
27         }
28     
29         // adds the given mapping
30         public void add(int from, int to) {
31     
32             CallStringContext fromInt = new CallStringContext(from);
33             CallStringContext toInt = new CallStringContext(to);
34     
35             this.pos2pos.put(fromInt, toInt);
36     
37             // maintain reverse mapping
38     
39             Set<CallStringContext> reverseSet = this.reverse.get(toInt);
40             if (reverseSet == null) {
41                 // there was no such reverse mapping:
42                 // create it together with a new set
43                 reverseSet = new HashSet<>();
44                 reverseSet.add(fromInt);
45                 this.reverse.put(toInt, reverseSet);
46             } else {
47                 // add to already existing reverse mapping set
48                 reverseSet.add(fromInt);
49             }
50         }
51     
52         // applies this connector function to the given input value
53         public CallStringContext apply(int input) {
54             CallStringContext output = this.pos2pos.get(new CallStringContext(input));
55             return output;
56         }
57     
58         // reverse application: returns a set of inputs (CallStringContext's) for the given output
59         // (might be null if there is no such output)
60         public Set<CallStringContext> reverseApply(int output) {
61             return this.reverse.get(new CallStringContext(output));
62         }
63     
64         public String toString() {
65             if (this.pos2pos.isEmpty()) {
66                 return "<empty>";
67             }
68             StringBuilder myString = new StringBuilder();
69             for (Map.Entry<CallStringContext, CallStringContext> entry : this.pos2pos.entrySet()) {
70                 CallStringContext from = entry.getKey();
71                 CallStringContext to = entry.getValue();
72                 myString.append(from);
73                 myString.append(" -> ");
74                 myString.append(to);
75                 myString.append(System.getProperty("line.separator"));
76             }
77             return myString.substring(0, myString.length() - 1);
78         }
79     }
