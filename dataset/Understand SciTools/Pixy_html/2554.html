97         // takes the given invalue and applies transfer functions from the start of the
98         // basic block until the beginning of "untilHere" (i.e., the transfer function
99         // of this node is NOT applied);
100         // only required by DependencyGraph
101         public DependencyLatticeElement applyInsideBasicBlock(
102             BasicBlock basicBlock, AbstractCfgNode untilHere, DependencyLatticeElement invalue) {
103     
104             DependencyLatticeElement outValue = new DependencyLatticeElement(invalue);
105             List<AbstractCfgNode> containedNodes = basicBlock.getContainedNodes();
106             CompositeTransferFunction ctf = (CompositeTransferFunction) this.getTransferFunction(basicBlock);
107     
108             Iterator<AbstractCfgNode> nodesIter = containedNodes.iterator();
109             Iterator<AbstractTransferFunction> tfIter = ctf.iterator();
110     
111             while (nodesIter.hasNext() && tfIter.hasNext()) {
112                 AbstractCfgNode node = nodesIter.next();
113                 AbstractTransferFunction tf = tfIter.next();
114                 if (node == untilHere) {
115                     break;
116                 }
117                 outValue = (DependencyLatticeElement) tf.transfer(outValue);
118             }
119     
120             return outValue;
121         }

124     
125         // takes the given invalue and applies transfer functions from the start of the
126         // default cfg (given by "defaultNode") until the beginning of "untilHere"
127         // (i.e., the transfer function of this node is NOT applied);
128         // only required by DependencyGraph
129         public DependencyLatticeElement applyInsideDefaultCfg(AbstractCfgNode defaultNode,
130                                                        AbstractCfgNode untilHere, DependencyLatticeElement invalue) {
131             DependencyLatticeElement out = new DependencyLatticeElement(invalue);
132     
133             while (defaultNode != untilHere) {
134                 AbstractTransferFunction tf = this.getTransferFunction(defaultNode);
135                 out = (DependencyLatticeElement) tf.transfer(out);
136                 defaultNode = defaultNode.getSuccessor(0);
137             }
138     
139             return out;
140         }

482         // checks if the callgraph contains unreachable code (i.e., nodes that have not
483         // been associated with analysis information)
484         public void checkReachability() {
485     
486             if (!(this.analysisType instanceof CallStringAnalysis)) {
487                 // in this case, we do not have a callgraph, and can't check for
488                 // unreachable code
489                 System.out.println("Warning: Can't check for unreachable code");
490                 return;
491             }
492     
493             ConnectorComputation cc = ((CallStringAnalysis) this.analysisType).getConnectorComputation();
494             CallGraph callGraph = cc.getCallGraph();
495     
496             // for each function in the call graph...
497             for (TacFunction f : callGraph.getFunctions()) {
498                 AbstractCfgNode head = f.getControlFlowGraph().getHead();
499     
500                 // make a depth-first traversal
501     
502                 // the number of contexts for this function
503                 int numContexts = cc.getNumContexts(f);
504     
505                 // auxiliary stack and visited set
506                 LinkedList<AbstractCfgNode> stack = new LinkedList<>();
507                 Set<AbstractCfgNode> visited = new HashSet<>();
508     
509                 // visit head
510                 AbstractCfgNode current = head;
511                 visited.add(current);
512                 if (!this.isReachable(head, numContexts)) {
513                     this.warnUnreachable(current);
514                 } else {
515                     stack.add(current);
516                 }
517     
518                 // how it works:
519                 // while there is something on the stack:
520                 // - mark the top stack element as visited
521                 // - try to get an unvisited successor of this element
522                 // - if there is such a successor: push it on the stack and continue
523                 // - else: pop the stack
524                 while (!stack.isEmpty()) {
525     
526                     // inspect the top stack element
527                     current = stack.getLast();
528     
529                     // we will try to get an unvisited successor element
530                     AbstractCfgNode next = null;
531                     for (int i = 0; (i < 2) && (next == null); i++) {
532                         CfgEdge outEdge = current.getOutEdge(i);
533                         if (outEdge != null) {
534                             next = outEdge.getDestination();
535                             if (visited.contains(next)) {
536                                 // try another one
537                                 next = null;
538                             } else {
539                                 // found it!
540                             }
541                         }
542                     }
543     
544                     if (next == null) {
545                         // pop from stack
546                         stack.removeLast();
547                     } else {
548                         // visit next
549                         visited.add(next);
550     
551                         if (!this.isReachable(next, numContexts)) {
552                             this.warnUnreachable(next);
553                         } else {
554                             stack.add(next);
555                         }
556                     }
557                 }
558             }
559         }

561         private boolean isReachable(AbstractCfgNode cfgNode, int numContexts) {
562             Map<AbstractContext, AbstractLatticeElement> phi = this.interproceduralAnalysisInformation.getAnalysisNode(cfgNode).getPhi();
563             if (phi.size() == 0) {
564                 // there is not a single context for this node
565                 return false;
566             }
567             for (AbstractLatticeElement elem : phi.values()) {
568                 if (elem == null) {
569                     // a null lattice element?
570                     throw new RuntimeException("SNH");
571                     //return false;
572                 }
573             }
574             return true;
575         }
