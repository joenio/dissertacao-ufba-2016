17     public class MyTransducer extends AbstractAutomaton {
18         public MyTransducer(rationals.transductions.TransducerNivat a) {
19             super();
20     
21             // auxiliary map from foreign state to my cloned state
22             Map<rationals.State, MyState> map = new HashMap<>();
23     
24             // copy states, updating auxiliary map on the way
25             @SuppressWarnings("unchecked") Set<State> states = a.states();
26             for (State state : states) {
27                 map.put(state, this.addState(state.isInitial(), state.isTerminal()));
28             }
29     
30             // copy transitions (and transform TransducerRelations)
31             @SuppressWarnings("unchecked") Set<rationals.Transition> delta = a.delta();
32             for (rationals.Transition t : delta) {
33                 TransducerRelation origtr = (TransducerRelation) t.label();
34                 this.addTransition(new MyTransition(
35                     map.get(t.start()),
36                     new MyTransducerRelation(origtr.getIn(), origtr.getOut()),
37                     map.get(t.end())));
38             }
39         }

41         // you can use this if you already have the symbols
42         public void toFsmFile(String filename)
43             throws IOException {
44     
45             Writer writer = new FileWriter(filename + ".txt");
46     
47             Set<MyTransition> delta = this.getDelta();
48     
49             String eps = "EPS";
50     
51             // print transitions starting from the initial state
52             for (MyTransition t : this.getOutgoingTransitions(this.getInitial())) {
53                 writeTransition(t, writer, eps);
54             }
55     
56             // print remaining transitions
57             for (MyTransition t : delta) {
58                 if (t.getStart().isInitial()) {
59                     continue;
60                 }
61                 writeTransition(t, writer, eps);
62             }
63     
64             // print final states
65             for (MyState state : this.getTerminals()) {
66                 writer.write(state + "\n");
67             }
68     
69             writer.close();
70         }

72         private void writeTransition(MyTransition t, Writer writer, String eps)
73             throws IOException {
74     
75             MyTransducerRelation rel = (MyTransducerRelation) t.getLabel();
76             Object inLabel = rel.getIn();
77             String inString;
78             if (inLabel == null) {
79                 // epsilon transition
80                 inString = eps;
81             } else {
82                 inString = inLabel.toString();
83                 StringBuilder temp = new StringBuilder();
84                 Transition.appendCharString(inString.charAt(0), temp);
85                 inString = temp.toString();
86             }
87     
88             Object outLabel = rel.getOut();
89             String outString;
90             if (outLabel == null) {
91                 // epsilon transition
92                 outString = eps;
93             } else {
94                 outString = outLabel.toString();
95                 StringBuilder temp = new StringBuilder();
96                 Transition.appendCharString(outString.charAt(0), temp);
97                 outString = temp.toString();
98             }
99     
100             writer.write(
101                 t.getStart() + " " +
102                     t.getEnd() + " " +
103                     inString + " " +
104                     outString + " " +
105                     "\n");
106         }
