18     /**
19      * Vase class for interprocedural analyses (Sharir and Pnueli). Can be used for the functional and the call-string
20      * approach.
21      *
22      * The different approaches have to implement the following abstract methods:
23      * - getPropagationContext
24      * - getReverseTargets
25      *
26      * The concrete analyses derived from these approaches have to implement the remaining abstract methods:
27      * - initLattice
28      * - evalIf
29      * - override those transfer function generators that shall return transfer functions other than the ID transfer function
30      * - call initGeneral()
31      *
32      * @author Nenad Jovanovic <enji@seclab.tuwien.ac.at>
33      */

60     // initGeneral *********************************************************************
61     
62         // general initialization work; taken out of the constructor to bypass the
63         // restriction that superclass constructors have to be called first;
64         // the "functions" map has to map function name -> TacFunction object
65         protected void initGeneral(List<TacFunction> functions, TacFunction mainFunction,
66                                    AbstractAnalysisType analysisType, InterproceduralWorklist workList) {
67     
68             this.analysisType = analysisType;
69             this.analysisType.setAnalysis(this);
70             this.functions = functions;
71     
72             // determine ControlFlowGraph of main function: start analysis here
73             this.mainFunction = mainFunction;
74             ControlFlowGraph mainControlFlowGraph = this.mainFunction.getControlFlowGraph();
75             AbstractCfgNode mainHead = mainControlFlowGraph.getHead();
76     
77             // initialize carrier lattice
78             this.initLattice();
79     
80             // initialize main context
81             this.mainContext = this.analysisType.initContext(this);
82     
83             // initialize worklist
84             this.workList = workList;
85             this.workList.add(mainHead, this.mainContext);
86     
87             // initialize analysis nodes
88             this.interproceduralAnalysisInformation = new InterproceduralAnalysisInformation();
89             this.genericAnalysisInformation = interproceduralAnalysisInformation;
90             // assign transfer functions
91             this.initTransferFunctions();
92     
93             // initialize PHI map for start node
94             AbstractInterproceduralAnalysisNode startAnalysisNode = this.interproceduralAnalysisInformation.getAnalysisNode(mainHead);
95             startAnalysisNode.setPhiValue(this.mainContext, this.startValue);
96         }

98     //  initTransferFunctions ***********************************************************
99     
100         // controls the assignment of transfer functions to analysis nodes by calling
101         // traverseCfg()
102         void initTransferFunctions() {
103     
104             // handle default CFGs (for default parameters) first;
105             // for all functions...
106             for (TacFunction function : this.functions) {
107                 for (TacFormalParameter param : function.getParams()) {
108                     // if this param has a default value, it also has a small CFG;
109                     // traverse it as well...;
110                     // NOTE: default CFGs will not be associated with analysis information,
111                     // see transfer functions for CallPreparation; analogous to the
112                     // contents of basic blocks
113                     if (param.hasDefault()) {
114                         ControlFlowGraph defaultControlFlowGraph = param.getDefaultControlFlowGraph();
115                         this.traverseCfg(defaultControlFlowGraph, function);
116                     }
117                 }
118             }
119     
120             // now handle "normal" CFGs;
121             // for all functions...
122             for (TacFunction function : this.functions) {
123                 // extract and traverse CFG
124                 this.traverseCfg(function.getControlFlowGraph(), function);
125             }
126         }

193     //  analyze ************************************************************************
194     
195         // this method applies the worklist algorithm
196         public void analyze() {
197     
198             int steps = 0;
199     
200             // for each element in the worklist...
201             // (each worklist element is a pair of CFG node & context lattice element)
202             while (this.workList.hasNext()) {
203     
204                 steps++;
205                 if (steps % 10000 == 0) System.out.println("Steps so far: " + steps);
206     
207                 // remove the element from the worklist
208                 InterproceduralWorklistElement element = this.workList.removeNext();
209     
210                 // extract information from the element
211                 AbstractCfgNode node = element.getCfgNode();
212                 AbstractContext context = element.getContext();
213     
214                 // get incoming value at node n (you need to understand the PHI table :)
215                 AbstractInterproceduralAnalysisNode analysisNode = this.interproceduralAnalysisInformation.getAnalysisNode(node);
216                 AbstractLatticeElement inValue = analysisNode.getPhiValue(context);
217                 if (inValue == null) {
218                     throw new RuntimeException("SNH");
219                 }
220     
221                 try {
222     
223                     // distinguish between various types of CFG nodes
224                     if (node instanceof Call) {
225     
226                         Call callNode = (Call) node;
227     
228                         // get necessary function information (= called function)
229                         TacFunction function = callNode.getCallee();
230                         CallReturn callRet = (CallReturn) node.getOutEdge(0).getDestination();
231     
232                         if (function == null) {
233                             // callee could not be determined yet;
234                             // the search for a function summary doesn't make
235                             // sense; simply go on to the return node;
236                             // the concrete analysis is responsible for handling
237                             // calls to unknown functions in the transfer functions
238                             // for CallPrep and CallRet
239     
240                             // note: even though calls to unknown functions will be
241                             // replaced with a special cfg node at the end of tac conversion,
242                             // this case might still occur *during* tac conversion
243                             // (especially during include file resolution)
244     
245                             propagate(context, inValue, callRet);
246                             continue;
247                         }
248     
249                         ControlFlowGraph functionControlFlowGraph = function.getControlFlowGraph();
250     
251                         AbstractCfgNode exitNode = functionControlFlowGraph.getTail();
252                         // the tail of the function's CFG has to be an exit node
253                         if (!(exitNode instanceof CfgExit)) {
254                             throw new RuntimeException("SNH");
255                         }
256     
257                         AbstractContext propagationContext = this.getPropagationContext(callNode, context);
258     
259                         // look if the exit node's PHI map has an entry under the context
260                         // resulting from this call
261                         AbstractInterproceduralAnalysisNode exitAnalysisNode = this.interproceduralAnalysisInformation.getAnalysisNode(exitNode);
262                         if (exitAnalysisNode == null) {
263                             // this can only mean that there is no way to reach the
264                             // function's natural exit node, i.e. there is something like
265                             // die() on each path to the natural exit node; in this
266                             // case, we simply enter the function; this can lead to
267                             // redundant computations, but it is simpler than a
268                             // special, more efficient treatment of this rare case
269                             AbstractCfgNode entryNode = functionControlFlowGraph.getHead();
270                             propagate(propagationContext, inValue, entryNode);
271                             continue;
272                         }
273     
274                         AbstractLatticeElement exitInValue = exitAnalysisNode.getPhiValue(propagationContext);
275     
276                         if (this.useSummaries() && exitInValue != null) {
277     
278                             // previously computed function summary can be used;
279                             // determine successor node (unique) of this call node
280                             CfgEdge[] outEdges = callNode.getOutEdges();
281                             AbstractCfgNode succ = outEdges[0].getDestination();
282                             propagate(context, exitInValue, succ);
283                         } else {
284     
285                             // there is no function summary yet (or we don't want to
286                             // use summaries)
287     
288                             // necessary for call-string analyses
289                             // EFF: think about additional conditions to add here
290                             if ((this.analysisType instanceof CallStringAnalysis) && exitInValue != null) {
291                                 this.workList.add(exitNode, propagationContext);
292                             }
293     
294                             // there is no function summary yet (or we don't want to
295                             // use summaries), so compute it now by entering the function
296                             AbstractCfgNode entryNode = functionControlFlowGraph.getHead();
297                             propagate(propagationContext, inValue, entryNode);
298                         }
299     
300                         // calls to a builtin function are simply treated by invoking
301                         // the corresponding transfer function; covered by the catch-all below
302                         //} else if (node instanceof CallBuiltinFunction) {
303     
304                     } else if (node instanceof CfgExit) {
305     
306                         CfgExit exitNode = (CfgExit) node;
307     
308                         // the function to this exit node
309                         TacFunction function = exitNode.getEnclosingFunction();
310     
311                         // no need to proceed if this is the exit node of the
312                         // main function
313                         if (function == this.mainFunction) {
314                             continue;
315                         }
316     
317                         // the exit node gets a special treatment: pass incoming value
318                         // in a lazy manner
319                         // AbstractLatticeElement outValue = this.analysisInfo[node.getId()].transfer(inValue);
320                         AbstractLatticeElement outValue = inValue;
321     
322                         // get targets that we have to return to
323                         for (ReverseTarget reverseTarget : this.getReverseTargets(function, context)) {
324                             // extract target call node
325                             Call callNode = reverseTarget.getCallNode();
326     
327                             // determine successor node (unique) of the call node
328                             CfgEdge[] outEdges = callNode.getOutEdges();
329                             CallReturn callRetNode = (CallReturn) outEdges[0].getDestination();
330     
331                             // determine predecessor node (unique) of the call node
332                             CallPreparation callPrepNode = callRetNode.getCallPrepNode();
333     
334                             for (AbstractContext targetContext : reverseTarget.getContexts()) {
335                                 // if the incoming value at the callprep node is undefined, this means
336                                 // that the analysis hasn't made the call under this context
337                                 // (can happen for call-string analysis);
338                                 // => don't propagate
339                                 //if (this.analysisInfo[callPrepNode.getId()].getPhiValue(targetContext) == null) {
340                                 AbstractInterproceduralAnalysisNode callPrepANode = this.interproceduralAnalysisInformation.getAnalysisNode(callPrepNode);
341                                 if (callPrepANode.getPhiValue(targetContext) == null) {
342                                     // don't propagate
343                                 } else {
344                                     // propagate!
345                                     propagate(targetContext, outValue, callRetNode);
346                                 }
347                             }
348                         }
349                     } else if (node instanceof If) {
350     
351                         If ifNode = (If) node;
352     
353                         AbstractLatticeElement outValue = this.interproceduralAnalysisInformation.getAnalysisNode(node).transfer(inValue);
354                         CfgEdge[] outEdges = node.getOutEdges();
355     
356                         // try to evaluate the "if" condition
357                         Boolean eval = this.evalIf(ifNode, inValue);
358     
359                         if (eval == null) {
360                             // static evaluation of if condition failed, continue
361                             // analysis along both outgoing edges
362     
363                             propagate(context, outValue, outEdges[0].getDestination());
364                             propagate(context, outValue, outEdges[1].getDestination());
365                         } else if (eval == Boolean.TRUE) {
366                             // continue analysis along true edge
367                             propagate(context, outValue, outEdges[1].getDestination());
368                         } else {
369                             // continue analysis along false edge
370                             propagate(context, outValue, outEdges[0].getDestination());
371                         }
372                     } else if (node instanceof CallReturn) {
373     
374                         // a call return node is to be handled just as a normal node,
375                         // with the exception that it also needs to know about the
376                         // current context
377     
378                         // apply transfer function to incoming value
379                         AbstractInterproceduralAnalysisNode aNode = this.interproceduralAnalysisInformation.getAnalysisNode(node);
380                         AbstractLatticeElement outValue = aNode.transfer(inValue, context);
381     
382                         // for each outgoing edge...
383                         CfgEdge[] outEdges = node.getOutEdges();
384                         for (CfgEdge outEdge : outEdges) {
385                             if (outEdge != null) {
386     
387                                 // determine the successor
388                                 AbstractCfgNode succ = outEdge.getDestination();
389     
390                                 // propagate the result of applying the transfer function
391                                 // to the successor (under the current context)
392                                 propagate(context, outValue, succ);
393                             }
394                         }
395                     } else {
396     
397                         // apply transfer function to incoming value
398                         AbstractLatticeElement outValue;
399                         outValue = this.interproceduralAnalysisInformation.getAnalysisNode(node).transfer(inValue);
400     
401                         // for each outgoing edge...
402                         CfgEdge[] outEdges = node.getOutEdges();
403                         for (CfgEdge outEdge : outEdges) {
404                             if (outEdge != null) {
405     
406                                 // determine the successor
407                                 AbstractCfgNode succ = outEdge.getDestination();
408     
409                                 // propagate the result of applying the transfer function
410                                 // to the successor (under the current context)
411                                 propagate(context, outValue, succ);
412                             }
413                         }
414                     }
415                 } catch (RuntimeException ex) {
416                     System.out.println("File:" + node.getFileName() + ", Line: " + node.getOriginalLineNumber());
417                     throw ex;
418                 }
419             }
420     
421             if (!MyOptions.optionB && MyOptions.optionV) {
422                 System.out.println("Steps total: " + steps);
423             }
424             // worklist algorithm finished!
425         }

427     // propagate ***********************************************************************
428     
429         // helper method for analyze();
430         // propagates a value under the given context to the target node
431         void propagate(AbstractContext context, AbstractLatticeElement value, AbstractCfgNode target) {
432             // analysis information for the target node
433             AbstractInterproceduralAnalysisNode analysisNode = this.interproceduralAnalysisInformation.getAnalysisNode(target);
434     
435             if (analysisNode == null) {
436                 System.out.println(Dumper.makeCfgNodeName(target));
437                 throw new RuntimeException("SNH: " + target.getClass());
438             }
439     
440             // determine the target's old PHI value
441             AbstractLatticeElement oldPhiValue = analysisNode.getPhiValue(context);
442             if (oldPhiValue == null) {
443                 // initial value of this analysis
444                 oldPhiValue = this.initialValue;
445             }
446     
447             // speedup: if incoming value and target value are exactly the same
448             // object, then the result certainly can't change
449             if (value == oldPhiValue) {
450                 System.out.println("exact match!");
451                 return;
452             }
453     
454             // the new PHI value is computed as usual (with lub)
455             AbstractLatticeElement newPhiValue = this.lattice.lub(value, oldPhiValue);
456     
457             // if the PHI value changed...
458             if (!oldPhiValue.equals(newPhiValue)) {
459                 // update analysis information
460                 analysisNode.setPhiValue(context, newPhiValue);
461     
462                 // add this node (under the current context) to the worklist
463                 this.workList.add(target, context);
464             }
465         }
