5     package tacle.internal.rta.data;
6     
7     import java.util.HashSet;
8     import java.util.Set;
9     
10     import tacle.internal.rta.ResolvedCallSite;
11     
12     /**
13      * @author Mariana L Barca
14      * 
15      * Representation of methods from ASTs
16      */
17     public class ASTMethod implements IWorklistMethod {
18     
19     	private String key;
20     
21     	private IWorklistType declaringClass;
22     
23     	private String name;
24     
25     	private IWorklistType returnType;
26     
27     	private IWorklistType[] parameters;
28     
29     	private Set resolvedCallSites;
30     
31     	// constructor
32     	public ASTMethod(String k, String n, IWorklistType dC, IWorklistType rT,
33     			IWorklistType[] ps) {
34     		key = k;
35     		name = n;
36     		declaringClass = dC;
37     		returnType = rT;
38     		parameters = ps;
39     		resolvedCallSites = new HashSet();
40     
41     		location = -1;
42     	}
43     
44     	private int location;
45     
46     	private ASTType parent;
47     
48     	public ASTMethod(String n, ASTType p, int l) {
49     		name = n;
50     		parent = p;
51     		location = l;
52     
53     		key = "";
54     		resolvedCallSites = new HashSet();
55     	}
56     
57     	// getters
58     	public Set getResolvedCallSites() {
59     		return resolvedCallSites;
60     	}
61     
62     	public IWorklistType getDeclaringClass() {
63     		return declaringClass;
64     	}
65     
66     	public String getKey() {
67     		return key;
68     	}
69     
70     	public String getName() {
71     		return name;
72     	}
73     
74     	public IWorklistType[] getParameters() {
75     		return parameters;
76     	}
77     
78     	public IWorklistType getReturnType() {
79     		return returnType;
80     	}
81     
82     	public IWorklistType getParent() {
83     		return parent;
84     	}
85     	
86     	public int getLocation() {
87     		return location;
88     	}
89     	
90     	// setter
91     	public void addResolvedCallSite(ResolvedCallSite rcs) {
92     		resolvedCallSites.add(rcs);
93     	}
94     
95     	public void setKey(String k) {
96     		key = k;
97     	}
98     	
99     	public void setDeclaringClass(IWorklistType dc) {
100     		declaringClass = dc;
101     	}
102     	
103     	public void setReturnType(IWorklistType rt) {
104     		returnType = rt;
105     	}
106     	
107     	public void setParameters(IWorklistType[] p) {
108     		parameters = p;
109     	}
110     	
111     	// overriding methods from class Object
112     	public String toString() {
113     		String pNames = "";
114     		for (int i = 0; i < getParameters().length - 1; i++) {
115     			pNames += getParameters()[i] + ",";
116     		}
117     		if (getParameters().length > 0) {
118     			pNames += getParameters()[getParameters().length - 1];
119     		}
120     		return getReturnType().toString() + " " + getName() + "(" + pNames
121     				+ ") KEY:" + getKey();
122     	}
123     
124     	public boolean equals(Object other) {
125     		if (other instanceof ASTMethod) {
126     			ASTMethod astM = (ASTMethod) other;
127     			return this.key.equals(astM.key)
128     					|| ((this.parent != null) && astM.parent != null
129     							&& this.parent.equals(astM.parent)
130     							&& this.location != -1 && this.location == astM.location);
131     		}
132     		return false;
133     
134     	}
135     
136     	public int hashCode() {
137     		return this.name.hashCode();
138     	}
139     }
