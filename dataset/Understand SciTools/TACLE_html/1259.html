5     package tacle.core;
6     
7     import java.util.Set;
8     import org.eclipse.jdt.core.IMethod;
9     import org.eclipse.jdt.core.IType;
10     
11     /**
12      * Generic interface to a call graph construction analysis.
13      */
14     
15     public interface ICallGraphAnalysis {
16     
17     	/**
18     	 * Run the analysis, starting from the given start method (the main method
19     	 * for a whole-program analysis)
20     	 */
21     	void run(IMethod startMethod);
22     
23     	/**
24     	 * Returns a set of <code>IMethod</code> references, representing the
25     	 * methods reachable from the start method (the main method for a
26     	 * whole-program analysis). Not guarantted to be complete, due to the
27     	 * limitations of <code>IMethod</code>. For example, static initializers
28     	 * are not represented by IMethods, and any calls from them are not
29     	 * represented. Also some methods declared in the library do not have an
30     	 * IMethod representation and will not be included in the returned set.
31     	 * 
32     	 * The current version does not deal with reflective calls and throw
33     	 * clauses. See http://presto.cse.ohio-state.edu/tacle/index.html for more
34     	 * details on the current implementation.
35     	 */
36     	Set getReachableIMethods();
37     
38     	/**
39     	 * Returns a set of <code>IType</code> references, representing the class
40     	 * and interface types that are initialized in code reachable from the main
41     	 * method. Not guarantted to be complete, due to the limitations of
42     	 * <code>IType</code>. For example, some types declared in the library do
43     	 * not have corresponding IType representations and they will not be
44     	 * included in the returned result.
45     	 * 
46     	 * See http://presto.cse.ohio-state.edu/tacle/index.html for more details on
47     	 * the current implementation.
48     	 */
49     	Set getInitializedITypes();
50     
51     	/**
52     	 * Returns a set of <code>IMethod</code> references for all methods called
53     	 * by the given method. The given method should be reachable. Not guarantted
54     	 * to be complete, due to the limitations of and <code>IMethod</code>.
55     	 * For example, static initializers are not represented by IMethods, and any
56     	 * calls from them are not represented. Also some methods declared in the
57     	 * library have a corresponding IMethod representation and they will not be
58     	 * included in the returned result.
59     	 * 
60     	 * The current version does not deal with reflective calls and throw
61     	 * clauses. See http://presto.cse.ohio-state.edu/tacle/index.html for more
62     	 * details on the current implementation.
63     	 * 
64     	 * Current version works only with features of Java 1.4.
65     	 */
66     	Set getCalledIMethods(IMethod method);
67     
68     	/**
69     	 * Returns the set of call sites in the method. The given method should be
70     	 * reachable.
71     	 * 
72     	 * @see ICallSite
73     	 */
74     	Set getCallSites(IMethod method);
75     
76     	/**
77     	 * Reports true the given method was analyzed from source code, and false
78     	 * otherwise. The given method should be reachable.
79     	 */
80     	boolean analyzedFromSourceCode(IMethod method);
81     
82     	/**
83     	 * Looks for a call site in the given method such that the start position
84     	 * and the length of the source range match exactly the given values. This
85     	 * method can be used to obtain information about an AST node representing a
86     	 * call site: <blockquote>
87     	 * 
88     	 * <pre>
89     	 * 
90     	 *  
91     	 *   
92     	 *    
93     	 *     
94     	 *      
95     	 *       
96     	 *        
97     	 *         
98     	 *          
99     	 *           
100     	 *            
101     	 *             IMethod m = ...;
102     	 *             MethodInvocation n = ...; // node in m's AST
103     	 *             CallGraphAnalysis c = ...;
104     	 *             CallSite s = c.getCallSite(m, n.getStartPosition(), n.getLength());
105     	 *             if (s != null) // null will be returned if there is no match
106     	 *                System.out.println(&quot;Num callees: &quot; + s.getCalledIMethods().size());
107     	 *             
108     	 *            
109     	 *           
110     	 *          
111     	 *         
112     	 *        
113     	 *       
114     	 *      
115     	 *     
116     	 *    
117     	 *   
118     	 *  
119     	 * </pre>
120     	 * 
121     	 * </blockquote> The given method should be reachable.
122     	 */
123     	ICallSite getCallSite(IMethod method, int pos, int length);
124     
125     	/**
126     	 * Returns a set of <code>IMethod</code> references for all methods
127     	 * calling the given method. The given method should be reachable. Not
128     	 * guarantted to be complete, due to the limitations of and
129     	 * <code>IMethod</code>.
130     	 * 
131     	 * The current version does not deal with reflective calls and throw
132     	 * clauses. See http://presto.cse.ohio-state.edu/tacle/index.html for more
133     	 * details on the current implementation.
134     	 */
135     	public Set getCallingMethods(IMethod method);
136     
137     	/**
138     	 * Adds a method declared in the library to the list of reachable methods.
139     	 * The elements of the array in the parameter <code>method</code>
140     	 * represent, in this order: <br>- package name of class declaring the
141     	 * method <br>- complete name of class declaring the method (" <init>" for
142     	 * constructors) <br>- package name of return type (or "" if none) <br>-
143     	 * complete name of return type <br>- name of method <br>- package name of
144     	 * class of first parameter (or "" if none) <br>- complete name of type of
145     	 * first parameter <br>- ... <br>
146     	 * (list continues for the other parameters). Package names are provided
147     	 * separately in order to avoid confusions in finding the package name for
148     	 * inner types.
149     	 * 
150     	 * Examples of usage: <br>
151     	 * 
152     	 * <blockquote>
153     	 * 
154     	 * <pre>
155     	 * addLibraryreachableMethod(&quot;java.awt.color&quot;, &quot;java.awt.color.ICC_Profile&quot;, &quot;&quot;,
156     	 * 		&quot;void&quot;, &quot;setData&quot;, &quot;&quot;, &quot;int&quot;, &quot;&quot;, &quot;byte[]&quot;);
157     	 * 
158     	 * addLibraryreachableMethod(&quot;java.text&quot;, &quot;java.text.FieldPosition&quot;, &quot;&quot;, &quot;void&quot;,
159     	 * 		&quot;&lt;init&gt;&quot;, &quot;java.text&quot;, &quot;java.text.Format.Field&quot;);
160     	 * 
161     	 * addLibraryreachableMethod(&quot;java.security&quot;, &quot;java.security.KeyStore&quot;, &quot;&quot;,
162     	 * 		&quot;java.security.cert.Certificate[]&quot;, &quot;getCertificateChain&quot;, &quot;java.lang&quot;,
163     	 * 		&quot;java.lang.String&quot;);
164     	 * </pre>
165     	 * 
166     	 * </blockquote>
167     	 * 
168     	 * This method should be called before the method <code>run</code> is
169     	 * called in order to have effect in the analysis.
170     	 */
171     	public void addLibraryReachableMethod(String[] method);
172     
173     	/**
174     	 * Adds the method declared in user code <code>method</code> to the list
175     	 * of reachable methods. This method should be called before the method
176     	 * <code>run</code> is called in order to have effect in the analysis.
177     	 */
178     	public void addReachableMethod(IMethod method);
179     
180     	/**
181     	 * Adds the library declared type <code>typeName</code> to the list of
182     	 * instantiated classes, if it is not already there. <code>typeName</code>
183     	 * is the complete name of the type, including the package name if any.
184     	 * <code>packageName</code> is a substring of <code>typeName</code> (or
185     	 * "") and is provided in order to avoid confusions in finding the package
186     	 * name for inner classes.
187     	 * 
188     	 * Examples of usage: <blockquote>
189     	 * 
190     	 * <pre>
191     	 * addLibraryInstantiatedClass(&quot;java.util.zip&quot;, &quot;java.util.zip.GZIPInputStream&quot;);
192     	 * 
193     	 * addLibraryInstantiatedClass(&quot;&quot;, &quot;java.util.zip.GZIPInputStream[]&quot;);
194     	 * </pre>
195     	 * 
196     	 * </blockquote>
197     	 * 
198     	 * This method should be called before the method <code>run</code> is
199     	 * called in order to have effect in the analysis.
200     	 */
201     	public void addLibraryInstantiatedClass(String packageName, String typeName);
202     
203     	/**
204     	 * Adds the type declared in user code <code>type</code> to the list of
205     	 * instantiated classes, if it is not already there. This method should be
206     	 * called before the method <code>run</code> is called in order to have
207     	 * effect in the analysis.
208     	 */
209     	public void addInstantiatedClass(IType type);
210     }
