268     	public Set getCallSites(IMethod method) {
269     		if (!methodsPopulated) {
270     			populateMethods();
271     		}
272     		IWorklistMethod internalM = (IWorklistMethod) externalToInternalMethods
273     				.get(method);
274     		if (internalM == null) {
275     			return null;
276     		}
277     		Set result = new HashSet();
278     		String internalMKey = internalM.getKey();
279     		Set resolvedSites = internalM.getResolvedCallSites();
280     		for (Iterator it = resolvedSites.iterator(); it.hasNext();) {
281     			ResolvedCallSite rcs = (ResolvedCallSite) it.next();
282     			Set targetMethods = new HashSet();
283     			Set targetMethodKeys = new HashSet();
284     			CallSite cs = null;
285     			if (rcs.isVirtual()) {
286     				cs = new InstanceCallSite(method, internalMKey, rcs.getPos(),
287     						rcs.getLength(), targetMethods, targetMethodKeys);
288     			} else {
289     				cs = new CallSite(method, internalMKey, rcs.getPos(), rcs
290     						.getLength(), targetMethods, targetMethodKeys);
291     			}
292     			Set targets = rcs.resolvedTargets;
293     			for (Iterator it1 = targets.iterator(); it1.hasNext();) {
294     				IWorklistMethod tgt = (IWorklistMethod) it1.next();
295     				IMethod iMtgt = (IMethod) internalToExternalMethods.get(tgt);
296     				if (iMtgt == null) {
297     					System.out.println("IMethod not found for " + tgt);
298     				} else {
299     					String tgtKey = tgt.getKey();
300     					targetMethods.add(iMtgt);
301     					targetMethodKeys.add(tgtKey);
302     				}
303     			}
304     			result.add(cs);
305     		}
306     		return result;
307     	}

322     	public ICallSite getCallSite(IMethod method, int pos, int length) {
323     		if (!methodsPopulated) {
324     			populateMethods();
325     		}
326     		IWorklistMethod internalM = (IWorklistMethod) externalToInternalMethods
327     				.get(method);
328     		if (internalM == null) {
329     			return null;
330     		}
331     		String internalMKey = internalM.getKey();
332     		Set resolvedSites = internalM.getResolvedCallSites();
333     		Iterator it = resolvedSites.iterator();
334     		while (it.hasNext()) {
335     			ResolvedCallSite rcs = (ResolvedCallSite) it.next();
336     			if (rcs.getPos() == pos && rcs.getLength() == length) {
337     				Set targetMethods = new HashSet();
338     				Set targetMethodKeys = new HashSet();
339     				CallSite cs = null;
340     				if (rcs.isVirtual()) {
341     					cs = new InstanceCallSite(method, internalMKey, pos,
342     							length, targetMethods, targetMethodKeys);
343     				} else {
344     					cs = new CallSite(method, internalMKey, pos, length,
345     							targetMethods, targetMethodKeys);
346     				}
347     				Set targets = rcs.resolvedTargets;
348     				for (Iterator it1 = targets.iterator(); it1.hasNext();) {
349     					IWorklistMethod tgt = (IWorklistMethod) it1.next();
350     					IMethod iMtgt = (IMethod) internalToExternalMethods
351     							.get(tgt);
352     					String tgtKey = tgt.getKey();
353     					targetMethods.add(iMtgt);
354     					targetMethodKeys.add(tgtKey);
355     				}
356     				return cs;
357     			}
358     		}
359     		return null;
360     	}

395     	private IType getIType(IWorklistType type) {
396     		String typeName = type.getCompleteName();
397     		IType iType = null;
398     		// search for the corresponding IType
399     		// if it is anonymous, search differently
400     		if (type instanceof ASTType && /*
401     									    * ((ASTType)
402     									    * type).getSimpleName().equals("")
403     									    */
404     		((ASTType) type).getPosition() > 0) {
405     			iType = getAnonymousType((ASTType) type);
406     			//System.out.println(typeName + " is anonymous " + (iType ==
407     			// null));
408     			return iType;
409     		}
410     
411     		boolean found = false;
412     		for (Iterator it = javaProjects.iterator(); !found && it.hasNext();) {
413     			IJavaProject ijp = (IJavaProject) it.next();
414     			try {
415     				iType = ijp.findType(typeName);
416     				if (iType != null)
417     					found = true;
418     			} catch (JavaModelException jme) {
419     				System.out.println(jme.getMessage());
420     			}
421     		}
422     		// try a second search if not found
423     		if (iType == null) {
424     			iType = getSecondaryIType(type.getCompleteName(), type
425     					.getPackageName(), (type instanceof ASTType));
426     		}
427     		return iType;
428     	}

431     	// when the type is secondary type
432     	// or otherwise not available by regular find
433     	private IType getSecondaryIType(String completeName, String packageName,
434     			boolean isFromAST) {
435     		String[] possTypeName = getAllPossSimpleNames(completeName, packageName);
436     		IType result = null;
437     		for (Iterator it = javaProjects.iterator(); result == null
438     				&& it.hasNext();) {
439     			IJavaProject ijProj = (IJavaProject) it.next();
440     			try {
441     				IPackageFragment[] chl = ijProj.getPackageFragments();
442     				IPackageFragment pkg = null;
443     				for (int i = 0; i < chl.length; i++) {
444     					if (chl[i].getElementName().equals(packageName)) {
445     						pkg = chl[i];
446     						break;
447     					}
448     				}
449     				if (pkg != null) {
450     					ICompilationUnit[] units = pkg.getCompilationUnits();
451     					for (int i = 0; i < units.length; i++) {
452     						IType[] types = units[i].getAllTypes();
453     						for (int j = 0; j < types.length; j++) {
454     							for (int k = 0; k < possTypeName.length; k++) {
455     								if (types[j].getTypeQualifiedName().equals(
456     										possTypeName[k])) {
457     									result = types[j];
458     								}
459     							}
460     						}
461     					}
462     				}
463     			} catch (JavaModelException jme) {
464     				jme.printStackTrace();
465     			}
466     		}
467     		if (isFromAST) {
468     			//System.out.println("AST TYPE:" + completeName + " " +
469     			// result.getFullyQualifiedName());
470     			result = getInnerDeclaredType(result, completeName);
471     			return result;
472     		} else {
473     			// weird
474     			System.out.println("IType not found for " + completeName);
475     		}
476     		return result;
477     	}

480     	// other types or methods
481     	private IType getInnerDeclaredType(IJavaElement elem, String name) {
482     		if (elem == null)
483     			return null;
484     
485     		if (elem instanceof IType
486     				&& ((IType) elem).getFullyQualifiedName().equals(name)) {
487     			return (IType) elem;
488     		} else if (elem instanceof IType) {
489     			try {
490     				IJavaElement[] children = ((IType) elem).getChildren();
491     				for (int i = 0; i < children.length; i++) {
492     					IType result = getInnerDeclaredType(children[i], name);
493     					if (result != null)
494     						return result;
495     				}
496     			} catch (JavaModelException jme) {
497     				System.out.println(jme.getMessage());
498     			}
499     		} else if (elem instanceof IMethod) {
500     			try {
501     				IJavaElement[] children = ((IMethod) elem).getChildren();
502     				for (int i = 0; i < children.length; i++) {
503     					IType result = getInnerDeclaredType(children[i], name);
504     					if (result != null)
505     						return result;
506     				}
507     			} catch (JavaModelException jme) {
508     				System.out.println(jme.getMessage());
509     			}
510     		}
511     		return null;
512     	}

527     	// other types or methods
528     	private IType getInnerDeclaredType(IJavaElement elem, int pos) {
529     		//System.out.println("getInnerDeclaredType:" + elem.getElementName());
530     		if (elem == null)
531     			return null;
532     
533     		if (elem instanceof IType) {
534     			try {
535     				IType t = (IType) elem;
536     				if (t.getSourceRange().getOffset() == pos)
537     					return t;
538     				IJavaElement[] children = ((IType) elem).getChildren();
539     				for (int i = 0; i < children.length; i++) {
540     					IType result = getInnerDeclaredType(children[i], pos);
541     					if (result != null)
542     						return result;
543     				}
544     			} catch (JavaModelException jme) {
545     				System.out.println(jme.getMessage());
546     			}
547     		} else {
548     			IJavaElement[] children = null;
549     			try {
550     				if (elem instanceof IMethod) {
551     					children = ((IMethod) elem).getChildren();
552     				} else if (elem instanceof IField) {
553     					children = ((IField) elem).getChildren();
554     				} else if (elem instanceof IInitializer) {
555     					children = ((IInitializer) elem).getChildren();
556     				}
557     			} catch (JavaModelException jme) {
558     				System.out.println(jme.getMessage());
559     			}
560     			for (int i = 0; children != null && i < children.length; i++) {
561     				IType result = getInnerDeclaredType(children[i], pos);
562     				if (result != null)
563     					return result;
564     			}
565     		}
566     		return null;
567     	}

569     	// returns all possible simple names of this class type
570     	// to be used when the above method fails
571     	public String[] getAllPossSimpleNames(String completeName,
572     			String packageName) {
573     		String name = completeName;
574     		// remove package
575     		if (completeName.indexOf(packageName) == 0 && !packageName.equals("")) {
576     			name = completeName.substring(packageName.length() + 1);
577     		}
578     		String[] result = new String[name.split("\\$|\\.").length];
579     		for (int i = 0; i < result.length; i++) {
580     			if (i > 0) {
581     				result[i] = result[i - 1] + name.charAt(0);
582     				name = name.substring(1);
583     			} else {
584     				result[i] = "";
585     			}
586     			int index;
587     			int indexDot = name.indexOf(".");
588     			int indexDollar = name.indexOf("$");
589     			if (indexDot > -1)
590     				if (indexDollar > -1)
591     					index = (indexDot > indexDollar) ? indexDollar : indexDot;
592     				else
593     					index = indexDot;
594     			else if (indexDollar > -1)
595     				index = indexDollar;
596     			else
597     				index = name.length();
598     			result[i] += name.substring(0, index);
599     			name = name.substring(index);
600     		}
601     		return result;
602     	}

608     		// search for the corresponding IType
609     		IType declaringIType = getIType(declaringClass);
610     		// given the IType, search for the IMethod
611     
612     		if (declaringIType != null) {
613     			try {
614     				IMethod[] allMethods = declaringIType.getMethods();
615     				for (int i = 0; i < allMethods.length; i++) {
616     					IMethod crt = allMethods[i];
617     					if ((iwm.getName().equals(crt.getElementName()) || (iwm
618     							.getName().equals("<init>") && crt.isConstructor()))
619     							&& iwm.getParameters().length == crt
620     									.getNumberOfParameters()) {
621     						String[] params = crt.getParameterTypes();
622     						boolean found = true;
623     						for (int j = 0; found && j < params.length; j++) {
624     							String compareTo = iwm.getParameters()[j]
625     									.getCompleteName();
626     							//System.out.println("compareto:" + iwm);
627     							String compareToSign = Signature
628     									.createTypeSignature(compareTo, true);
629     							if (!params[j].equals(compareToSign)) {
630     								compareToSign = Signature.createTypeSignature(
631     										compareTo, false);
632     								if (!params[j].equals(compareToSign)) {
633     									compareToSign = Signature
634     											.createTypeSignature(Signature
635     													.getSimpleName(compareTo),
636     													false);
637     									if (!params[j].equals(compareToSign)) {
638     										if (iwm.getParameters()[j] instanceof ASTType) { // last
639     											// try
640     											String simpleName = ((ASTType) (iwm
641     													.getParameters()[j]))
642     													.getSimpleName();
643     											if (!simpleName.equals("")) {
644     												compareToSign = Signature
645     														.createTypeSignature(
646     																simpleName,
647     																false);
648     												if (!params[j]
649     														.equals(compareToSign)) {
650     													found = false;
651     												}
652     											} else {
653     												found = false;
654     											}
655     										} else {
656     											found = false;
657     										}
658     									}
659     								}
660     							}
661     						}
662     						if (found) {
663     							return crt;
664     						}
665     					}
666     				}
667     			} catch (JavaModelException jme) {
668     				System.out.println(jme.getMessage());
669     			}
670     		} else {
671     			System.out.println("Declaring class for " + iwm.getKey()
672     					+ " is null");
673     		}
674     		return result;
675     	}
