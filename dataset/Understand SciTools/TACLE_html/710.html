66     	// divides str into token and returns an array
67     	// that contains them
68     	private String[] parseTokens(String str, String delim) {
69     		StringTokenizer st = new StringTokenizer(str, " ");
70     		int count = st.countTokens();
71     		String[] values = new String[count];
72     		int i = 0;
73     		while (st.hasMoreTokens()) {
74     			values[i] = st.nextToken();
75     			i++;
76     		}
77     		return values;
78     	}

81     	public void loadAllPackages(Map packageIdByName, Map packageNameById) {
82     		String packagesFile = dataPath + fileSeparator + "packages";
83     		try {
84     			BufferedReader in = new BufferedReader(new FileReader(packagesFile));

99     	public void loadAllMethods(Integer packageId, Map methodInfos,
100     			Map methodInfosByClassId) {
101     		String methodsFile = dataPath + fileSeparator + packageId
102     				+ fileSeparator + "methods";
103     		try {
104     			BufferedReader in = new BufferedReader(new FileReader(methodsFile));
105     			String str = "";
106     			while ((str = in.readLine()) != null) {
107     				// format:
108     				// methodid methodname declpkgid declclsid returnpkgid
109     				// returntypeid paramno parampkgid
110     				// paramtypeid... modifiers
111     				String[] values = parseTokens(str, " ");
112     				MethodInfo mi = new MethodInfo();
113     				mi.setMethodId(Integer.parseInt(values[0]));
114     				mi.setName(values[1]);
115     				mi.setDeclaringPackageId(Integer.parseInt(values[2]));
116     				mi.setDeclaringClassId(Integer.parseInt(values[3]));
117     				mi.setReturnPackageId(Integer.parseInt(values[4]));
118     				mi.setReturnTypeId(Integer.parseInt(values[5]));
119     				int[][] param = new int[Integer.parseInt(values[6])][2];
120     				for (int i = 0; i < param.length; i++) {
121     					param[i][0] = Integer.parseInt(values[7 + 2 * i]);
122     					param[i][1] = Integer.parseInt(values[8 + 2 * i]);
123     				}
124     				mi.setParamTypeIds(param);
125     				mi.setModifiers(Integer.parseInt(values[7 + 2 * param.length]));
126     				methodInfos.put(Integer.valueOf(values[0]), mi);
127     
128     				// add to methodInfosByClassId: format:
129     				// Map methodInfosByClassId; //Integer->Map//String
130     				// methodName->Set of MethodInfos
131     				Integer classId = Integer.valueOf(values[3]);
132     				Map methodsByName = (Map) methodInfosByClassId.get(classId);
133     				if (methodsByName == null) {
134     					methodsByName = new HashMap();
135     					methodInfosByClassId.put(classId, methodsByName);
136     				}
137     				String methodName = values[1];
138     				Set methods = (Set) methodsByName.get(methodName);
139     				if (methods == null) {
140     					methods = new HashSet();
141     					methodsByName.put(methodName, methods);
142     				}
143     				methods.add(mi);
144     			}
145     			in.close();
146     		} catch (IOException ioe) {
147     			ioe.printStackTrace();
148     		}
149     	}

151     	// loads file classes from the specified package 
152     	public void loadAllClasses(Integer packageId, Map classInfos,
153     			Map classInfosByName) {
154     		String classesFile = dataPath + fileSeparator + packageId
155     				+ fileSeparator + "classes";
156     		try {
157     			BufferedReader in = new BufferedReader(new FileReader(classesFile));
158     			String str = "";
159     			while ((str = in.readLine()) != null) {
160     				// format:
161     				// classid name completename superclasspkgid superclassid
162     				// interfno
163     				// interfpkgid interfid... modifiers
164     				String[] values = parseTokens(str, " ");
165     				ClassInfo ci = new ClassInfo();
166     				ci.setClassId(Integer.parseInt(values[0]));
167     				ci.setPackageId(packageId.intValue());
168     				ci.setCompleteName(values[1]);
169     				ci.setSuperPackageId(Integer.parseInt(values[2]));
170     				ci.setSuperClassId(Integer.parseInt(values[3]));
171     				int[][] interf = new int[Integer.parseInt(values[4])][2];
172     				for (int i = 0; i < interf.length; i++) {
173     					interf[i][0] = Integer.parseInt(values[5 + 2 * i]);
174     					interf[i][1] = Integer.parseInt(values[6 + 2 * i]);
175     				}
176     				ci.setInterfaceIds(interf);
177     				ci.setModifiers(Integer.parseInt(values[2 * interf.length + 5]));
178     				classInfos.put(Integer.valueOf(values[0]), ci);
179     				classInfosByName.put(ci.getCompleteName(), ci);
180     			}
181     			in.close();
182     		} catch (IOException ioe) {
183     			ioe.printStackTrace();
184     		}
185     	}

256     	// format of Map methodInfos: // Integer -> MethodInfo
257     	public void loadAllMethodHasCallSite(Integer packageId, Map methodInfos) {
258     		String mthdHasCallSiteFile = dataPath + fileSeparator + packageId
259     				+ fileSeparator + "methodHasCallSite";
260     		try {
261     			BufferedReader in = new BufferedReader(new FileReader(
262     					mthdHasCallSiteFile));
263     			String str = "";
264     			while ((str = in.readLine()) != null) {
265     				// methodid packageid classid targetpackageid targetmethodid
266     				// virtual(1)
267     				String[] values = parseTokens(str, " ");
268     				CallSiteInfo csi = new CallSiteInfo();
269     				int methodId = Integer.parseInt(values[0]);
270     
271     				MethodInfo mi = InfoDictionary.v().getMethodInfoForId(methodId,
272     						packageId.intValue());
273     
274     				csi.setPackageId(packageId.intValue());
275     				csi.setMethodId(methodId);
276     
277     				int classPkgId = Integer.parseInt(values[1]);
278     				int classId = Integer.parseInt(values[2]);
279     				csi.setVarPackageId(classPkgId);
280     				csi.setVarClassId(classId);
281     
282     				int targetPackageId = Integer.parseInt(values[3]);
283     				int targetMethodId = Integer.parseInt(values[4]);
284     				csi.setTargetPackageId(targetPackageId);
285     				csi.setTargetMethodId(targetMethodId);
286     
287     				csi.setVirtual(values[5].equals("1"));
288     				
289     				mi.setCallSite(csi);
290     			}
291     			in.close();
292     		} catch (IOException ioe) {
293     			ioe.printStackTrace();
294     		}
