117     	// returns only the direct subtypes
118     	// to get all subtypes, call method below
119     	public Set getDirectSubtypes(IWorklistType type) {
120     		if (typeHierarchy.containsKey(type))
121     			return (Set) typeHierarchy.get(type);
122     		else
123     			return null;
124     	}
125     
126     	/**
127     	 * returns set of strings or ClassInfos representing all subclasses calls
128     	 * recursive method findAllSubtypes
129     	 */
130     	public Set getAllSubtypes(IWorklistType type) {
131     		Set result = new HashSet();
132     		result.add(type);
133     		findAllSubtypes(type, result);
134     		allTypeHierarchy.put(type, result);
135     		return result;
136     	}
137     
138     	// recursive
139     	private void findAllSubtypes(IWorklistType type, Set result) {
140     		if (typeHierarchy.containsKey(type)) {
141     			Set subclasses = (Set) typeHierarchy.get(type);
142     			result.addAll(subclasses);
143     			for (Iterator it = subclasses.iterator(); it.hasNext();) {
144     				findAllSubtypes((IWorklistType) it.next(), result);
145     			}
146     		}
147     	}
148     
149     	// parent, child could be String or ClassInfo
150     	// adds pair to type hierarchy
151     	public void addInHierarchy(IWorklistType parent, IWorklistType child) {
152     		// test, keep
153     		if (child instanceof ASTType
154     				&& child.getCompleteName().equals("java.lang.String")) {
155     			throw new RuntimeException("STOP");
156     		}
157     
158     		if (!typeHierarchy.containsKey(parent)
159     				|| typeHierarchy.get(parent) == null) {
160     			typeHierarchy.put(parent, new HashSet());
161     		}
162     		Set subCls = (Set) typeHierarchy.get(parent);
163     		subCls.add(child);
164     	}

166     	// used in testing, printsthe type hierarchy
167     	public void printHierarchy(PrintWriter pw) {
168     		for (Iterator it = typeHierarchy.keySet().iterator(); it.hasNext();) {
169     			IWorklistType key = (IWorklistType) it.next();
170     			if (pw == null) {
171     				System.out.println(key);
172     			} else {
173     				pw.println(key);
174     			}
175     			Set set = (Set) typeHierarchy.get(key);
176     			for (Iterator it1 = set.iterator(); it1.hasNext();) {
177     				IWorklistType subcls = (IWorklistType) it1.next();
178     				if (pw == null) {
179     					System.out.println(" -> " + subcls.getClass() + " "
180     							+ subcls);
181     				} else {
182     					pw.println(" -> " + subcls.getClass() + " " + subcls);
183     				}
184     			}
185     		}
186     	}
