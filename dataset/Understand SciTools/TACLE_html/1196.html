5     package tacle.internal.rta;
6     
7     import org.eclipse.jdt.core.dom.ClassInstanceCreation;
8     import org.eclipse.jdt.core.dom.ConstructorInvocation;
9     import org.eclipse.jdt.core.dom.FieldDeclaration;
10     import org.eclipse.jdt.core.dom.ITypeBinding;
11     import org.eclipse.jdt.core.dom.Initializer;
12     import org.eclipse.jdt.core.dom.MethodDeclaration;
13     import org.eclipse.jdt.core.dom.MethodInvocation;
14     import org.eclipse.jdt.core.dom.Modifier;
15     import org.eclipse.jdt.core.dom.SimpleName;
16     import org.eclipse.jdt.core.dom.TypeDeclaration;
17     import org.eclipse.jdt.core.dom.SuperConstructorInvocation;
18     import org.eclipse.jdt.core.dom.SuperMethodInvocation;
19     
20     import tacle.internal.rta.data.IWorklistType;
21     import tacle.internal.rta.data.IWorklistMethod;
22     import tacle.internal.rta.data.InfoDictionary;
23     
24     /**
25      * @author Mariana L Barca
26      * 
27      * filters execution of method visitor to apply only to aspects relevant to
28      * static initializers the whole compilation unit must be visited, unlike
29      * methodVisitor which is applied only to the AST of a method in case of a
30      * regular method (not constructor, not initializer)
31      */
32     public class StaticInitializerVisitor extends MethodVisitor {
33     
34     	// is in initializer code
35     	private boolean isInInitializer;
36     
37     	// visitor is inside the class we are looking for
38     	private boolean isInsideClass;
39     
40     	// the class for which we analyze the static initializer
41     	private IWorklistType cls;
42     
43     	// field relevant only after traversal
44     	private boolean hasInitializer;
45     
46     	public boolean hasStaticInitializer() {
47     		return hasInitializer;
48     	}
49     
50     	// need to know which constructor to analyze
51     	StaticInitializerVisitor(RTAMethodWorklist worklist, IWorklistType cls, IWorklistMethod method) {
52     		super(worklist, method);
53     		isInInitializer = false;
54     		isInsideClass = false;
55     		hasInitializer = false;
56     		this.cls = cls;
57     	}
58     
59     	public boolean visit(ClassInstanceCreation node) {
60     		if (isInInitializer) {
61     			//System.out.println("Visiting... " + node);
62     			super.visit(node);
63     		}
64     		return true;
65     	}
66     
67     	public boolean visit(ConstructorInvocation node) {
68     		if (isInInitializer) {
69     			//System.out.println("Visiting... " + node);
70     			super.visit(node);
71     		}
72     		return true;
73     	}
74     
75     	// visit only static field declarations
76     	public boolean visit(FieldDeclaration node) {
77     		if (isInsideClass && Modifier.isStatic(node.getModifiers())) {
78     			isInInitializer = true;
79     			hasInitializer = true;
80     			//System.out.println(node + " in " + cls);
81     		} else {
82     			isInInitializer = false;
83     		}
84     		return isInInitializer;
85     	}
86     
87     	// flags say if this is a static block
88     	public boolean visit(Initializer node) {
89     		if (isInsideClass && Modifier.isStatic(node.getModifiers())) {
90     			isInInitializer = true;
91     			hasInitializer = true;
92     			//System.out.println(node + " in " + cls);
93     		} else {
94     			isInInitializer = false;
95     		}
96     		return isInInitializer;
97     	}
98     
99     	public boolean visit(MethodDeclaration node) {
100     		if (isInInitializer) {
101     			super.visit(node);
102     		}
103     		return true;
104     	}
105     
106     	public boolean visit(MethodInvocation node) {
107     		if (isInInitializer) {
108     			//System.out.println("Visiting... " + node);
109     			super.visit(node);
110     		}
111     		return true;
112     	}
113     
114     	// this can be the name of a static field
115     	public boolean visit(SimpleName node) {
116     		if (isInInitializer) {
117     			super.visit(node);
118     		}
119     		return true;
120     	}
121     
122     	// no way this can be in an initializer
123     	// so we don't continue the traversal down the tree
124     	public boolean visit(SuperConstructorInvocation node) {
125     		return false;
126     	}
127     
128     	// no way this can be in an initializer
129     	// so we don't continue the traversal down the tree
130     	public boolean visit(SuperMethodInvocation node) {
131     		return false;
132     	}
133     
134     	// this is to remember the parent
135     	// in case super was not called explicitly and
136     	// we have to call it manually
137     	public boolean visit(TypeDeclaration node) {
138     		ITypeBinding itb = node.resolveBinding();
139     		IWorklistType temp = InfoDictionary.v().getTypeForBinding(itb);
140     		if (cls.equals(temp)) {
141     			isInsideClass = true;
142     		} else if (isInsideClass) {
143     			return false;
144     		}
145     		// go down other types because the type we are searching for
146     		// may be inner
147     		return true; 
148     	}
149     
150     	// set isInClass to false after desired class is visited
151     	public void endVisit(TypeDeclaration node) {
152     		ITypeBinding itb = node.resolveBinding();
153     		IWorklistType temp = InfoDictionary.v().getTypeForBinding(itb);
154     		if (cls.equals(temp)) {
155     			isInsideClass = false;
156     		}
157     	}
158     
159     }
