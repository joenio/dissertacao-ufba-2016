5     package tacle.internal.rta.data;

7     import java.util.Set;

10     
11     import tacle.core.ICallSite;
12

17     public class CallSite implements ICallSite {
18     
19     	private IMethod callingMethod;
20     
21     	private int startPos;
22     
23     	private int len;
24     
25     	protected Set calledMethods; // member: IMethod
26     
27     	// constructors
28     	private CallSite() {
29     	}
30     
31     	public CallSite(IMethod callingMethod, String callingMethodKey,
32     			int startPos, int len, Set calledMethods, Set calledMethodKeys) {
33     		this.callingMethod = callingMethod;
34     		this.startPos = startPos;
35     		this.len = len;
36     		this.calledMethods = calledMethods;
37     	}
38     
39     	// methods from interface
40     	public IMethod getMethod() {
41     		return callingMethod;
42     	}
43     
44     	public int getStartPosition() {
45     		return startPos;
46     	}
47     
48     	public int getLength() {
49     		return len;
50     	}
51     
52     	public Set getCalledIMethods() {
53     		return calledMethods;
54     	}
55     
56     	public String toString() {
57     		return startPos + " " + len;
58     	}
59     
60     	public boolean equals(Object obj) {
61     		if (obj instanceof CallSite) {
62     			CallSite cs = (CallSite) obj;
63     			return callingMethod.equals(cs.callingMethod)
64     					&& startPos == cs.startPos && len == cs.len;
65     		} else
66     			return false;
67     	}
68     	
69     	public int hashCode() {
70     		return callingMethod.hashCode() + startPos + len;
71     	}
72     }
