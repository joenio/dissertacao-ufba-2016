5     package tacle.core;
6     
7     import org.eclipse.jdt.core.IMethod;
8     import java.util.Set;
9     
10     /**
11      * Tacle-specific representation of a call site. Client code should not
12      * implement this interface: instances should be obtained through methods
13      * <code>getXYZ</code> from <code>CallGraphAnalysis</code>.
14      * <p>
15      * Eventually we may add subinterfaces <code>StaticCallSite</code> and
16      * possibly <code>ConstructorCallSite</code>. But for now, just use one
17      * subinterface <code>InstanceCallSite</code>.
18      */
19     
20     public interface ICallSite {
21     
22     	/**
23     	 * Returns a reference to the IMethod for the enclosing method. Could be
24     	 * <code>null</code> if no corresponding <code>IMethod</code>
25     	 * representation exists for the enclosing method.
26     	 */
27     	IMethod getMethod();
28     
29     	/**
30     	 * Start position in the source code, copied from the corresponding AST
31     	 * node. If the method does not come from source code (e.g., it is a method
32     	 * from a .class file), returns -1.
33     	 */
34     	int getStartPosition();
35     
36     	/**
37     	 * Length of the source segment, copied from the corresponding AST node. If
38     	 * the method does not come from source code (e.g., it is a method from a
39     	 * .class file), returns -1.
40     	 */
41     	int getLength();
42     
43     	/**
44     	 * Returns a set of <code>IMethod</code> references for all methods called
45     	 * by the call site. Not guaranteed to be complete, due to the limitations
46     	 * of <code>IMethod</code>. For example some methods declared in the
47     	 * library do not have a corresponding <code>IMethod</code> representation
48     	 * and will not be included in the returned result.
49     	 * 
50     	 * The current version does not deal with reflective calls and throw
51     	 * clauses. See http://presto.cse.ohio-state.edu/tacle/index.html for more
52     	 * details on the current implementation.
53     	 */
54     	Set getCalledIMethods();
55     
56     }
