75     	public boolean visit(MethodDeclaration node) {
76     		if (node.getStartPosition()==((ASTMethod)method).getLocation()) {
77     			isInMethod = true;
78     			IMethodBinding mb = node.resolveBinding();
79     			((ASTMethod)method).setKey(mb.getKey());
80     			IWorklistType[] params = new IWorklistType[mb.getParameterTypes().length];
81     			for (int j = 0; j < params.length; j++) {
82     				params[j] = InfoDictionary.v().getTypeForBinding(
83     						mb.getParameterTypes()[j]);
84     			}
85     			((ASTMethod)method).setParameters(params);
86     			IWorklistType returnType = InfoDictionary.v()
87     					.getTypeForBinding(mb.getReturnType());
88     			((ASTMethod)method).setReturnType(returnType);
89     			IWorklistType declCls = InfoDictionary.v()
90     					.getTypeForBinding(mb.getDeclaringClass());
91     			((ASTMethod)method).setDeclaringClass(declCls);
92     			worklist.addMethodAfterAnalysis(method);
93     		} 
94     		return isInMethod;
95     	}
