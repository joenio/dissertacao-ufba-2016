34     		
35     		completeName = itb.getQualifiedName();
36     		if (completeName.equals("")) {
37     			completeName = itb.getBinaryName();
38     		}
39     
40     		// test this?
41     		if (!itb.isTopLevel() && !itb.isPrimitive() && !itb.isArray())
42     			completeName = itb.getBinaryName();
43     
44     		if (itb.getPackage() == null)
45     			packageName = "";
46     		else {
47     			packageName = itb.getPackage().getName();
48     			if (packageName == null || packageName.equals("UNNAMED")) {
49     				packageName = "";
50     			}

113     	// returns all possible simple names of this class type
114     	// to be used when the above method fails
115     	public String[] getAllPossSimpleNames() {
116     		String name = completeName;
117     		// remove package
118     		if (completeName.indexOf(packageName) == 0 && !packageName.equals("")) {
119     			name = completeName.substring(packageName.length() + 1);
120     		}
121     		String[] result = new String[name.split("\\$|\\.").length];
122     		for (int i = 0; i < result.length; i++) {
123     			if (i > 0) {
124     				result[i] = result[i - 1] + name.charAt(0);
125     				name = name.substring(1);
126     			} else {
127     				result[i] = "";
128     			}
129     			int index;
130     			int indexDot = name.indexOf(".");
131     			int indexDollar = name.indexOf("$");
132     			if (indexDot > -1)
133     				if (indexDollar > -1)
134     					index = (indexDot > indexDollar) ? indexDollar : indexDot;
135     				else
136     					index = indexDot;
137     			else if (indexDollar > -1)
138     				index = indexDollar;
139     			else
140     				index = name.length();
141     			result[i] += name.substring(0, index);
142     			name = name.substring(index);
143     		}
144     		return result;
145     	}
