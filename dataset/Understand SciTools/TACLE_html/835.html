5     package tacle.core;
6     
7     import org.eclipse.jdt.core.IMethod;
8     import org.eclipse.jdt.core.IType;
9     import java.util.Set;
10     
11     /**
12      * Tacle-specific representation of a call site. Client code should not
13      * implement this interface: instances should be obtained through methods
14      * <code>getXYZ</code> from <code>CallGraphAnalysis</code>.
15      * 
16      * Eventually we may add subinterfaces <code>StaticCallSite</code> and
17      * possibly <code>ConstructorCallSite</code>. The current version provides
18      * only the subinterface <code>InstanceCallSite</code>.
19      */
20     
21     public interface IInstanceCallSite extends ICallSite {
22     
23     	/**
24     	 * Returns a set of <code>IType</code> references, representing the
25     	 * possible run-time receiver types. For each returned reference
26     	 * <code>x</code>, it is guaranteed that <blockquote>
27     	 * 
28     	 * <pre>
29     	 * getInitializedITypes().contains(x) == true
30     	 * </pre>
31     	 * 
32     	 * </blockquote> Not guaranteed to be complete, due to the limitations of
33     	 * <code>IType</code>. For example, some types may not have a
34     	 * corresponding <code>IType</code> representation and will not be
35     	 * included in the returned result.
36     	 * 
37     	 * See http://presto.cse.ohio-state.edu/tacle/index.html for more details on
38     	 * the current implementation.
39     	 */
40     	Set getReceiverITypes();
41     
42     	/**
43     	 * Returns an <code>IMethod</code> reference for the method called by the
44     	 * call site for the given run-time receiver type. If such a receiver type
45     	 * is not possible (i.e., it is not an element of
46     	 * <code>getReceiverITypes()</code>, returns <code>null</code>. Not
47     	 * guaranteed to be complete, due to the limitations of <code>IType</code>.
48     	 * For example, some types may not have a corresponding <code>IType</code>
49     	 * representation and will not be included in the returned result.
50     	 * 
51     	 * The current version does not deal with reflective calls and throw
52     	 * clauses. See http://presto.cse.ohio-state.edu/tacle/index.html for more
53     	 * details on the current implementation.
54     	 */
55     	IMethod getCalledIMethod(IType receiverType);
56     
57     }
