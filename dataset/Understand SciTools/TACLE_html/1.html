30     /**
31      * @author Mariana L Barca
32      * 
33      * to be used for the whole compilation unit of the class that declares main
34      */
35     
36     public class MainMethodVisitor extends MethodVisitor {
37     
38     	// true if visiting the main method or the static init or the class
39     	private boolean isInMainOrInit;
40     
41     	// the class for which we analyze main & the static initializer
42     	private IWorklistType cls;
43     
44     	private String mainName;
45     	
46     	private int mainLocation;
47     	
48     	// field relevant only after traversal
49     	private boolean hasInitializer;
50     
51     	private boolean isInsideClass;
52     
53     	public boolean hasStaticInitializer() {
54     		return hasInitializer;
55     	}
56     
57     	// constructor
58     	MainMethodVisitor(RTAMethodWorklist worklist, IWorklistType cls, String name, int location) {
59     		super(worklist, null);
60     		isInMainOrInit = false;
61     		hasInitializer = false;
62     		isInsideClass = false;
63     		this.cls = cls;
64     		this.mainName = name;
65     		this.mainLocation = location;
66     		
67     		// create a dummy main method
68     		this.method = new ASTMethod("", "STARTUP", null, null, null);
69     	}
70     
71     	//////////// visit methods /////////////
72     
73     	public boolean visit(ClassInstanceCreation node) {
74     		if (isInMainOrInit) {
75     			//System.out.println("Visiting... " + node);
76     			super.visit(node);
77     		}
78     		return true;
79     	}
80     
81     	public boolean visit(ConstructorInvocation node) {
82     		if (isInMainOrInit) {
83     			//System.out.println("Visiting... " + node);
84     			super.visit(node);
85     		}
86     		return true;
87     	}
88     
89     	// visit only static field declarations
90     	public boolean visit(FieldDeclaration node) {
91     		if (isInsideClass && Modifier.isStatic(node.getModifiers())) {
92     			isInMainOrInit = true;
93     			hasInitializer = true;
94     			//System.out.println(node + " in " + cls);
95     		} else {
96     			isInMainOrInit = false;
97     		}
98     		return isInMainOrInit;
99     	}
100     
101     	// flags say if this is a static block
102     	public boolean visit(Initializer node) {
103     		if (isInsideClass && Modifier.isStatic(node.getModifiers())) {
104     			isInMainOrInit = true;
105     			hasInitializer = true;
106     			//System.out.println(node + " in " + cls);
107     		} else {
108     			isInMainOrInit = false;
109     		}
110     		return isInMainOrInit;
111     	}
112     
113     	public boolean visit(MethodDeclaration node) {
114     		int position = node.getStartPosition();
115     		
116     		if (isInsideClass
117     				&& node.getName().getFullyQualifiedName().equals(mainName) &&
118     				position == mainLocation
119     				) {
120     			//System.out.println("Visiting... " + node);
121     			IMethodBinding mb = node.resolveBinding();
122     			
123     //			// test
124     //			if (mb == null)
125     //				System.out.println("MainMethodVisitor:MethodDeclaration");
126     
127     			ASTMethod m = InfoDictionary.v().getASTMethod(mb.getKey());
128     			if (m == null) {
129     				IWorklistType[] params = new IWorklistType[mb
130     						.getParameterTypes().length];
131     				for (int j = 0; j < params.length; j++) {
132     					params[j] = InfoDictionary.v().getTypeForBinding(
133     							mb.getParameterTypes()[j]);
134     				}
135     				IWorklistType declClass = InfoDictionary.v().getTypeForBinding(
136     						mb.getDeclaringClass());
137     				IWorklistType retType = InfoDictionary.v().getTypeForBinding(
138     						mb.getReturnType());
139     				m = new ASTMethod(mb.getKey(), mb.getName(), declClass,
140     						retType, params);
141     				InfoDictionary.v().setASTMethod(mb.getKey(), m);
142     			}
143     			if (m != null) {
144     				// we have to replace the fake main method
145     				ASTMethod fakeMain = (ASTMethod)this.method;
146     				this.method = m;
147     				
148     				// transfer the resolved call sites
149     				Set resolvedSites = fakeMain.getResolvedCallSites();
150     				for(Iterator it = resolvedSites.iterator(); it.hasNext(); ) {
151     					this.method.addResolvedCallSite((ResolvedCallSite)it.next());
152     				}
153     			}
154     			worklist.addMethodAfterAnalysis(m);
155     			isInMainOrInit = true;
156     		} else {
157     			isInMainOrInit = false;
158     		}
159     		return isInMainOrInit;
160     	}
161     
162     	public boolean visit(MethodInvocation node) {
163     		if (isInMainOrInit) {
164     			//System.out.println("Visiting... " + node);
165     			super.visit(node);
166     		}
167     		return true;
168     	}
169     
170     	// this can be the name of a static field
171     	public boolean visit(SimpleName node) {
172     		if (isInMainOrInit) {
173     			super.visit(node);
174     		}
175     		return true;
176     	}
177     
178     	public boolean visit(SuperConstructorInvocation node) {
179     		return false;
180     	}
181     
182     	public boolean visit(SuperMethodInvocation node) {
183     		return false;
184     	}
185     
186     	// this is to remember the parent
187     	// in case super was not called explicitly and
188     	// we have to call it manually
189     	public boolean visit(TypeDeclaration node) {
190     		ITypeBinding itb = node.resolveBinding();
191     		
192     //		// test
193     //		if (itb == null)
194     //			System.out.println("MainMethodVisitor:TypeDeclaration");
195     		
196     		IWorklistType temp = InfoDictionary.v().getTypeForBinding(itb);
197     		if (cls.equals(temp)) {
198     			isInsideClass = true;
199     		} else if (isInsideClass) {
200     			return false;
201     		}
202     		// go down other types because the type we are searching for
203     		// may be inner
204     		return true;
205     	}
206     
207     	// set isInClass to false after desired class is visited
208     	public void endVisit(TypeDeclaration node) {
209     		ITypeBinding itb = node.resolveBinding();
210     		
211     //		// test
212     //		if (itb == null)
213     //			System.out.println("MainMethodVisitor:TypeDeclaration");
214     
215     		IWorklistType temp = InfoDictionary.v().getTypeForBinding(itb);
216     		if (cls.equals(temp)) {
217     			isInsideClass = false;
218     		}
219     	}
