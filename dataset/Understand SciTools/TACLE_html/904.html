26     /**
27      * @author Mariana L Barca
28      * 
29      * visits reachable code in constructor and finalizer (if any) if a finalizer
30      * exists, it is reported by the method hasFinalizer() this should be used to
31      * add finalizers to the reachable methods AFTER they are visited (should not be
32      * added to worklist)
33      */
34     public class ConstructorVisitor extends MethodVisitor {
35     
36     	private boolean isInConstructor;
37     
38     	private boolean isInClass;
39     
40     	private IWorklistType[] params;
41     
42     	private IWorklistType cls;
43     
44     	private String parentDefContrKey;
45     
46     	private IWorklistType parent;
47     
48     	// fields that report meaningful information
49     	// only after a traversal
50     	private boolean hasCalledSuperThis;
51     
52     	private boolean hasFinalize;
53     
54     	private String finalizerKey;
55     
56     	// getters
57     	public String getFinalizerKey() {
58     		return finalizerKey;
59     	}
60     
61     	public IWorklistType getParent() {
62     		return parent;
63     	}
64     
65     	public String getParentDefaultConstrKey() {
66     		return parentDefContrKey;
67     	}
68     
69     	// to be called only after visit
70     	// needed to remember the call to other constructors (super or this)
71     	public boolean hasCalledAnotherConstructor() {
72     		return hasCalledSuperThis;
73     	}
74     
75     	// true if the class has a finalizer
76     	public boolean hasFinalizer() {
77     		return hasFinalize;
78     	}
79     
80     	// need to know which constructor to analyze
81     	ConstructorVisitor(RTAMethodWorklist worklist, IWorklistType cls,
82     			IWorklistMethod method) {
83     		super(worklist, method);
84     		isInConstructor = false;
85     		isInClass = false;
86     		hasCalledSuperThis = false;
87     		this.params = method.getParameters();
88     		this.cls = cls;
89     
90     		parentDefContrKey = null;
91     		parent = null;
92     	}
93     
94     	//// visiting methods
95     
96     	public boolean visit(ClassInstanceCreation node) {
97     		if (isInConstructor) {
98     			//System.out.println("Visiting... " + node);
99     			super.visit(node);
100     		}
101     		return true;
102     	}
103     
104     	public boolean visit(ConstructorInvocation node) {
105     		if (isInConstructor) {
106     			//System.out.println("Visiting... " + node);
107     			hasCalledSuperThis = true;
108     			super.visit(node);
109     		}
110     		return true;
111     	}
112     
113     	// visit only instance field declarations
114     	public boolean visit(FieldDeclaration node) {
115     		if (!Modifier.isStatic(node.getModifiers())) {
116     			isInConstructor = true;
117     		} else {
118     			isInConstructor = false;
119     		}
120     		return isInConstructor;
121     	}
122     
123     	// assumed that the flags say if this is a static block
124     	public boolean visit(Initializer node) {
125     		//System.out.println(node);
126     		if (isInClass && !Modifier.isStatic(node.getModifiers())) {
127     			isInConstructor = true;
128     		} else {
129     			isInConstructor = false;
130     		}
131     		return isInConstructor;
132     	}
133     
134     	public boolean visit(MethodDeclaration node) {
135     		IMethodBinding mb = node.resolveBinding();
136     		if (isInClass) {
137     			if (node.isConstructor()) {
138     				//System.out.println("Visiting... " + node);
139     				// check that this is the constructor we are looking for
140     				// and not the one of a different class declared in the same cu
141     				IWorklistType temp = InfoDictionary.v().getTypeForBinding(
142     						mb.getDeclaringClass());
143     				if (cls.equals(temp)
144     						&& params.length == mb.getParameterTypes().length) {
145     					isInConstructor = true;
146     					for (int j = 0; j < params.length; j++) {
147     						temp = InfoDictionary.v().getTypeForBinding(
148     								mb.getParameterTypes()[j]);
149     						if (!params[j].equals(temp))
150     							isInConstructor = false;
151     					}
152     				} else {
153     					isInConstructor = false;
154     				}
155     			} else {
156     				if (mb.getName().equals("finalize")) {
157     					hasFinalize = true;
158     					finalizerKey = mb.getKey();
159     				}
160     				isInConstructor = false;
161     			}
162     			return isInConstructor;
163     		} else {
164     			return true; // may be in surrounding type/method
165     		}
166     	}
167     
168     	public boolean visit(MethodInvocation node) {
169     		if (isInConstructor) {
170     			//System.out.println(method + "Visiting... " + node);
171     			super.visit(node);
172     		}
173     		return true;
174     	}
175     
176     	// this can be the name of a static field
177     	public boolean visit(SimpleName node) {
178     		if (isInConstructor) {
179     			super.visit(node);
180     		}
181     		return true;
182     	}
183     
184     	public boolean visit(StringLiteral node) {
185     		if (isInConstructor) {
186     			super.visit(node);
187     		}
188     		return true;
189     	}
190     
191     	public boolean visit(SuperConstructorInvocation node) {
192     		if (isInConstructor) {
193     			//System.out.println("Visiting... " + node);
194     			hasCalledSuperThis = true;
195     			super.visit(node);
196     		}
197     		return true;
198     	}
199     
200     	public boolean visit(SuperMethodInvocation node) {
201     		if (isInConstructor) {
202     			//System.out.println("Visiting... " + node);
203     			super.visit(node);
204     		}
205     		return true;
206     	}
207     
208     	// this is to remember the parent
209     	// in case super was not called explicitly and
210     	// we have to call it manually
211     	public boolean visit(TypeDeclaration node) {
212     		ITypeBinding itb = node.resolveBinding();
213     		IWorklistType temp = InfoDictionary.v().getTypeForBinding(itb);
214     		if (cls.equals(temp)) {
215     			isInClass = true;
216     			if (itb.getSuperclass() != null) {
217     				parent = InfoDictionary.v().getTypeForBinding(
218     						itb.getSuperclass());
219     				// search for the key of the default constructor
220     				IMethodBinding[] methods = itb.getSuperclass()
221     						.getDeclaredMethods();
222     				// which one is the default constr
223     				for (int i = 0; i < methods.length; i++) {
224     					if (methods[i].isDefaultConstructor()) {
225     						parentDefContrKey = methods[i].getKey();
226     					}
227     				}
228     			}
229     		}
230     		return true;
231     	}
232     
233     	// end visit
234     	
235     	// this is to remember the parent
236     	// in case super was not called explicitly and
237     	// we have to call it manually
238     	public void endVisit(TypeDeclaration node) {
239     		ITypeBinding itb = node.resolveBinding();
240     		IWorklistType temp = InfoDictionary.v().getTypeForBinding(itb);
241     		if (cls.equals(temp)) {
242     			isInClass = false;
243     		}
244     	}
