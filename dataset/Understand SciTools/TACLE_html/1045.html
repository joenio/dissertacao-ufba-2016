106     	// returns first IWorklistClass found for this type binding
107     	public IWorklistType getTypeForBinding(ITypeBinding tb) {
108     		IWorklistType result = null;
109     		String qName = tb.getQualifiedName();
110     
111     		// search in typesByName for a class already loaded
112     		if (!qName.equals("")) {
113     			result = (IWorklistType) typesByName.get(qName);
114     			if (result != null) {
115     				setTypeKey(result, tb.getKey());
116     				return result;
117     			}
118     		}
119     		String bName = tb.getBinaryName();
120     		// search in typesByName for a weird class already loaded
121     		if (!tb.isTopLevel() && !tb.isPrimitive() && !tb.isArray()) {
122     			if (!bName.equals("")) {
123     				result = (IWorklistType) typesByName.get(bName);
124     				if (result != null) {
125     					setTypeKey(result, tb.getKey());
126     					return result;
127     				}
128     			}
129     		}
130     		// search in the summary on disk
131     		if (tb.getPackage() != null) {
132     			int packageId = getPackageIdForName(tb.getPackage().getName());
133     			if (packageId > 0) {
134     				if (!qName.equals("")) {
135     					result = getClassInfoForName(qName, packageId);
136     					if (result != null) {
137     						return result;
138     					}
139     				}
140     				if (!tb.isTopLevel() && !tb.isPrimitive() && !tb.isArray()
141     						&& !bName.equals("")) {
142     					result = getClassInfoForName(bName, packageId);
143     					if (result != null) {
144     						return result;
145     					}
146     				}
147     			}
148     		}
149     		// it's probably a new user class
150     		result = new ASTType(tb, false);
151     		// for safety, add alternate names for weird classes
152     		if (!qName.equals("") && !bName.equals("") && !qName.equals(bName)) {
153     			typesByName.put(qName, result);
154     			typesByName.put(bName, result);
155     		} else { // normal class
156     			typesByName.put(result.getCompleteName(), result);
157     		}
158     		// add type keys, for the search of ASTs
159     		setTypeKey(result, tb.getKey());
160     		return result;
161     	}

174     	// type will be added artificially, if it doesn't exist
175     	public ASTType addOtherTypeForName(String packageName, String typeName,
176     				String simpleName, String parentName, int position) {
177     		ASTType t = getOtherTypeForName(typeName);
178     		if (t==null) {
179     			t = new ASTType(packageName, typeName, simpleName, false);
180     			t.setPosition(position, parentName);
181     			typesByName.put(typeName, t);
182     			// class key is unknown, will be added later
183     		}
184     		return t;

247     	public MethodInfo getMethodInfoByName(String name, IWorklistType declClass,
248     			IWorklistType[] paramTypes) {
249     		if (declClass instanceof ClassInfo) {
250     			// this method triggers the loading of declPkg if not loaded already
251     			ClassInfo declClassInfo = (ClassInfo) declClass;
252     			int declClassId = declClassInfo.getClassId();
253     			Map methodsByName = (Map) methodInfosByClassId.get(new Integer(
254     					declClassId));
255     			if (methodsByName != null) {
256     				Set methods = (Set) methodsByName.get(name);
257     				if (methods != null) {
258     					for (Iterator it = methods.iterator(); it.hasNext();) {
259     						MethodInfo mi = (MethodInfo) it.next();
260     						// check signature
261     						IWorklistType[] params = mi.getParameters();
262     						if (params.length == paramTypes.length) {
263     							boolean sameParams = true;
264     							for (int i = 0; i < params.length; i++) {
265     								
266     								// test
267     								if (paramTypes[i]==null) {
268     									System.out.println(name + " in " + declClass.getCompleteName());
269     								}
270     								if (!paramTypes[i].equals(params[i]))
271     									sameParams = false;
272     							}
273     							if (sameParams)
274     								return mi;
275     						}
276     					}
277     				}
278     			}
279     		}
280     		return null;
281     	}

322     	private void loadPackage(Integer packageId) {
323     		loadedPackages.add(packageId);
324     		// load classes
325     		DataReader.v().loadAllClasses(packageId, classInfos, typesByName);
326     		// load methods
327     		DataReader.v().loadAllMethods(packageId, methodInfos,
328     				methodInfosByClassId);
329     		// load init classes
330     		DataReader.v().loadAllMethodInitializesClass(packageId, methodInfos);
331     		// load inst classes
332     		DataReader.v().loadAllMethodInstantiatesClass(packageId, methodInfos);
333     		// load call sites
334     		DataReader.v().loadAllMethodHasCallSite(packageId, methodInfos);
335     	}
