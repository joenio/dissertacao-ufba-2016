5     package tacle.examples.dotgraph;
6     
7     import org.eclipse.jdt.core.IMethod;
8     import java.io.*;
9     import tacle.core.*;
10     import java.util.*;
11     
12     public class DotGraphGenerator {
13     	
14     	//flag that specifies the type of graph to generate
15     	//true = multigraph with an arbitrary number assigned to each callsite
16     	//	the edges are labled with the number
17     	//false = call graph with no information about the callsites DEFAULT
18     	private boolean detailedGraph = true;
19     	
20     	//analysis interface
21     	private ICallGraphAnalysis cga = null;
22     	
23     	//Simple function that opens a BufferedWriter to
24     	//the dot file and returns it, returns null and prints
25     	//error if it is unable to open the file
26     	private BufferedWriter openDotFile (String path){
27     		BufferedWriter out=null;
28     		
29     		try{
30     			out = new BufferedWriter(new FileWriter(path,false));
31     			out.write("digraph \"\" {");
32     			out.newLine();
33     		}
34     		catch(IOException e){
35     			System.out.println("Unable to open file "+path);
36     			e.printStackTrace();
37     		}
38     		
39     		return out;
40     	}
41     	
42     	//Closes the dot file
43     	private void closeDotFile(BufferedWriter out){
44     			
45     		try{
46     			out.write("}\n");
47     			out.close();
48     		}
49     		catch(IOException e){
50     			System.out.println("Unable to close dot file ");
51     			e.printStackTrace();
52     		}
53     		
54     	}
55     	
56     	//Function that writes to the dot file and does the 
57     	//appropriate Graphviz format 
58     	private void writeDotFile(BufferedWriter out,IMethod current, IMethod next, 
59     			String label){
60     		
61     		try{
62     			
63     			out.write(current.getElementName()+" -> "+next.getElementName()
64     					+"[label=\""+label+"\"]");
65     			out.newLine();
66     		}
67     		catch(IOException e){
68     			System.out.println("Unable write to dot file ");
69     			e.printStackTrace();
70     		}
71     		
72     	}
73     	
74     	
75     	//this method is the the only public method for this class
76     	//the parameters: 
77     	//IMethod mainMethod = the IMethod for the main method being analyzed
78     	//String outFilePath = full path to the desired dot file
79     	public boolean run(IMethod mainMethod, String outFilePath){
80     		//create the dot file do simple check to make sure it was created
81     		BufferedWriter out = openDotFile(outFilePath);
82     		if (out == null){
83     			return false;
84     		}
85     		//create and analysis factory
86     		AnalysisFactory anaFac =  new AnalysisFactory();
87     		//get the RTA implementation of ICallGraphAnalysis
88     		cga = anaFac.getAnalysis(AnalysisFactory.RTA);
89     		
90     		//use flag to determine which type of graph to generate
91     		try{
92     			if(detailedGraph){
93     				runDetailed(mainMethod,out);
94     				}else {
95     				runSimple(mainMethod,out);	
96     			}
97     		}finally{
98     			closeDotFile(out);
99     			anaFac = null;
100     			cga = null;
101     		}
102     		return true;
103     	}
104     	
105     	//create a multigraph call graph with the callsite info as edge labels
106     	//IMethod mainMethod = is the IMethod for the main method being analyzed
107     	//BufferedWriter out = the opened BufferedWriter to the dot file that will contain
108     	//	the call graph
109     	private void runDetailed(IMethod mainMethod, BufferedWriter out){
110     		//necessary if there is a loop in the call graph 
111     		Set alreadyAnalyzed = new HashSet();
112     		//run the analysis on the passed in main method
113     		cga.run(mainMethod);
114     		List workList = new ArrayList();
115     		
116     		//add main to the work list
117     		workList.add(mainMethod);
118     		alreadyAnalyzed.add(mainMethod);
119     		//arbitrary number for the method to be used in the edge label
120     		int mCount = 1;
121     		
122     		while(!workList.isEmpty()){
123     			//remove any method from the worklist
124     			IMethod current = (IMethod)workList.remove(0);
125     			//get the set of callsites in the current method
126     			Set callS = cga.getCallSites(current);
127     			//arbitrary number of the callsites
128     			int csCount = 1;
129     			if(callS != null){
130     				//iterate over all the callsites in the method
131     				for(Iterator i = callS.iterator(); i.hasNext();){
132     					ICallSite cs = (ICallSite)i.next();
133     					//Get the set of possible target methods for the current callsite
134     					Set csMethods = cs.getCalledIMethods();
135     					if(csMethods != null){
136     						//iterator over the possible target methods
137     						for(Iterator iM = csMethods.iterator(); iM.hasNext();){
138     							IMethod next = (IMethod)iM.next();	
139     							//Only add user methods to the worklist that have not 
140     							//already been analyzed
141     							if(cga.analyzedFromSourceCode(next) && !alreadyAnalyzed.contains(next)){
142     								workList.add(next);
143     								alreadyAnalyzed.add(next);
144     								
145     							}
146     							//only add an edge if the target method is user defined
147     							if(cga.analyzedFromSourceCode(next)){
148     								writeDotFile(out,current,next,mCount+"."+csCount);
149     							}
150     						}
151     					}
152     					csCount++;
153     				}
154     			}
155     			mCount++;
156     		}
157     	}
158     	
159     	
160     	//Create a simple call graph 
161     	//IMethod mainMethod = is the IMethod for the main method being analyzed
162     	//BufferedWriter out = the opened BufferedWriter to the dot file that will contain
163     	//	the call graph
164     	private void runSimple(IMethod mainMethod, BufferedWriter out){
165     		//necessary if there is a loop in the call graph 
166     		Set alreadyAnalyzed = new HashSet();
167     		
168     		//run the analysis
169     		cga.run(mainMethod);
170     		List workList = new ArrayList();
171     		//add the mainMethod to the work list
172     		workList.add(mainMethod);
173     		alreadyAnalyzed.add(mainMethod);
174     		while(!workList.isEmpty()){
175     			//remove any method from the work list
176     			IMethod current = (IMethod)workList.remove(0);
177     			//get the methods called from the current method
178     			//this will return both library and user methods or null if it is empty
179     			Set callM = cga.getCalledIMethods(current);
180     		
181     			if(callM != null){
182     				//iterate over all the methods called in the current method
183     				for(Iterator i = callM.iterator(); i.hasNext();){
184     					IMethod next = (IMethod)i.next();
185     					//Only add edged for user code and method that are not 
186     					//already anaylyzed 
187     					if(cga.analyzedFromSourceCode(next)&& !alreadyAnalyzed.contains(next)){
188     						//add new methods to the worklist
189     						workList.add(next);
190     						alreadyAnalyzed.add(next);
191     						//add the new edge to the call graph
192     						writeDotFile(out,current,next,"");
193     					}
194     				}
195     			}
196     		}
197     		
198     	
199     	}
