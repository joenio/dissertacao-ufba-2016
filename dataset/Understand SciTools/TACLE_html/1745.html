5     package tacle.internal.rta.data;
6     
7     /**
8      * @author Mariana L Barca
9      *  
10      */
11     public class CallSiteInfo implements IWorklistCallSite {
12     	// enclosing method
13     	private int packageId;
14     
15     	private int methodId;
16     
17     	private MethodInfo method;
18     
19     	// var type
20     	private int varClassId;
21     
22     	private int varPackageId;
23     
24     	private ClassInfo varClass;
25     
26     	// static target method
27     	private int targetPackageId;
28     
29     	private int targetMethodId;
30     
31     	private MethodInfo targetMethod;
32     
33     	// virtual info
34     	private boolean virtual;
35     
36     	// methods that populate info
37     	private void populateEnclosingMethod() {
38     		method = InfoDictionary.v().getMethodInfoForId(methodId, packageId);
39     	}
40     
41     	private void populateVarType() {
42     		varClass = InfoDictionary.v().getClassInfo(varClassId, varPackageId);
43     	}
44     
45     	private void populateTargetMethod() {
46     		targetMethod = InfoDictionary.v().getMethodInfoForId(targetMethodId,
47     				targetPackageId);
48     	}
49     
50     	// constructors
51     	public CallSiteInfo() {
52     	}
53     
54     	// getters & setters
55     	public MethodInfo getMethod() {
56     		if (method == null)
57     			this.populateEnclosingMethod();
58     		return method;
59     	}
60     
61     	public void setMethod(MethodInfo method) {
62     		this.method = method;
63     	}
64     
65     	public int getMethodId() {
66     		return methodId;
67     	}
68     
69     	public void setMethodId(int methodId) {
70     		this.methodId = methodId;
71     	}
72     
73     	public int getPackageId() {
74     		return packageId;
75     	}
76     
77     	public void setPackageId(int packageId) {
78     		this.packageId = packageId;
79     	}
80     
81     	public MethodInfo getTargetMethod() {
82     		if (targetMethod == null)
83     			this.populateTargetMethod();
84     		return targetMethod;
85     	}
86     
87     	public void setTargetMethod(MethodInfo targetMethod) {
88     		this.targetMethod = targetMethod;
89     	}
90     
91     	public int getTargetMethodId() {
92     		return targetMethodId;
93     	}
94     
95     	public void setTargetMethodId(int targetMethodId) {
96     		this.targetMethodId = targetMethodId;
97     	}
98     
99     	public int getTargetPackageId() {
100     		return targetPackageId;
101     	}
102     
103     	public void setTargetPackageId(int targetPackageId) {
104     		this.targetPackageId = targetPackageId;
105     	}
106     
107     	public ClassInfo getVarClass() {
108     		if (varClass == null)
109     			this.populateVarType();
110     		return varClass;
111     	}
112     
113     	public void setVarClass(ClassInfo varClass) {
114     		this.varClass = varClass;
115     	}
116     
117     	public int getVarClassId() {
118     		return varClassId;
119     	}
120     
121     	public void setVarClassId(int varClassId) {
122     		this.varClassId = varClassId;
123     	}
124     
125     	public int getVarPackageId() {
126     		return varPackageId;
127     	}
128     
129     	public void setVarPackageId(int varPackageId) {
130     		this.varPackageId = varPackageId;
131     	}
132     
133     	public boolean isVirtual() {
134     		return virtual;
135     	}
136     
137     	public void setVirtual(boolean virtual) {
138     		this.virtual = virtual;
139     	}
140     
141     	// overriding methods from class Object
142     	public boolean equals(Object o) {
143     		if (o instanceof CallSiteInfo) {
144     			CallSiteInfo csi = (CallSiteInfo) o;
145     			return (this.methodId == csi.methodId
146     					&& this.targetMethodId == csi.targetMethodId && this.varClassId == csi.varClassId);
147     		} else
148     			return false;
149     	}
150     
151     	public int hashCode() {
152     		return methodId + targetMethodId + varClassId;
153     	}
154     }
