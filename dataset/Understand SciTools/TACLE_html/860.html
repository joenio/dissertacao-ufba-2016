5     package tacle.internal.types;
6     
7     import java.util.HashMap;
8     import java.util.Map;
9     
10     import org.eclipse.jdt.core.IClassFile;
11     import org.eclipse.jdt.core.ICompilationUnit;
12     import org.eclipse.jdt.core.IMember;
13     import org.eclipse.jdt.core.JavaModelException;
14     import org.eclipse.jdt.core.dom.AST;
15     import org.eclipse.jdt.core.dom.ASTParser;
16     import org.eclipse.jdt.core.dom.CompilationUnit;
17     
18     /**
19      * @author Mariana L Barca
20      *
21      * TACLE implementation of IASTManager  
22      */
23     public class ASTManager implements IASTManager {
24     
25     	private final boolean resolveBindings = true;
26     
27     	// map of compilation units, used as a cache to avoid
28     	// the time spen tin parsing sources
29     	// member: String handle identifier of source file (usually class name) 
30     	// -> CompilationUnit
31     	private Map cUnits;
32     
33     	private boolean keepUnits;
34     
35     	// constructor
36     	public ASTManager(boolean keepUnits) {
37     		this.keepUnits = keepUnits;
38     		if (keepUnits)
39     			cUnits = new HashMap();
40     	}
41     
42     	// private, so it cannot be called
43     	private ASTManager() {
44     	}
45     	
46     	// dispatcher method, further calls the appropriate methods 
47     	// unit can be ICompilationUnit or IClassFile
48     	public CompilationUnit createCompilationUnit(Object unit)
49     			throws TacleTypesException {
50     		if (unit instanceof IClassFile) {
51     			IClassFile classFile = (IClassFile) unit;
52     			return createCompilationUnit(classFile);
53     		} else if (unit instanceof ICompilationUnit) {
54     			ICompilationUnit icu = (ICompilationUnit) unit;
55     			return createCompilationUnit(icu);
56     		} else
57     			return null;
58     	}
59     
60     	// parses the source file with an AST parser and returns the 
61     	// resulted CompilationUnit
62     	private CompilationUnit createCompilationUnit(ICompilationUnit icu)
63     			throws TacleTypesException {
64     		if (icu != null && icu.exists()) {
65     			CompilationUnit cu = null;
66     			if (keepUnits) {
67     				cu = (CompilationUnit) cUnits.get(icu.getHandleIdentifier());
68     				if (cu != null)
69     					return cu;
70     			}
71     			ASTParser parser = ASTParser.newParser(AST.JLS2);
72     			parser.setSource(icu);
73     			parser.setResolveBindings(resolveBindings);
74     			cu = (CompilationUnit) parser.createAST(null);
75     			if (keepUnits && cu != null) {
76     				cUnits.put(icu.getHandleIdentifier(), cu);
77     			}
78     			return cu;
79     		}
80     
81     		return null;
82     	}
83     
84     	// parses the source file associated to the class file
85     	// with an AST parser and returns the 
86     	// resulted CompilationUnit
87     	private CompilationUnit createCompilationUnit(IClassFile classFile)
88     			throws TacleTypesException {
89     		try {
90     			if (classFile != null && classFile.exists()
91     					&& classFile.getSource() != null) {
92     				CompilationUnit cu = null;
93     				if (keepUnits) {
94     					cu = (CompilationUnit) cUnits.get(classFile
95     							.getHandleIdentifier());
96     					if (cu != null)
97     						return cu;
98     				}
99     				ASTParser parser = ASTParser.newParser(AST.JLS2);
100     				parser.setSource(classFile);
101     				parser.setResolveBindings(resolveBindings);
102     				cu = (CompilationUnit) parser.createAST(null);
103     				if (keepUnits && cu != null) {
104     					cUnits.put(classFile.getHandleIdentifier(), cu);
105     				}
106     				return cu;
107     			} 
108     		} catch (JavaModelException e) {
109     			e.printStackTrace();
110     			throw new TacleTypesException(e.getMessage());
111     		}
112     
113     		return null;
114     	}
115     
116     	public Object getUnit(IMember member) {
117     		if (member.isBinary()) {
118     			IClassFile classFile = member.getClassFile();
119     			return classFile;
120     		} else {
121     			ICompilationUnit icu = member.getCompilationUnit();
122     			return icu;
123     		}
124     	}
125     
126     }
