41     	 * @see IActionDelegate#run(IAction)
42     	 */
43     	public void run(IAction action) {
44     
45     		final int[] x = new int[1];
46     
47     		Job j = new Job("TacleAnalysis") {
48     			protected IStatus run(IProgressMonitor monitor) {
49     				Object element = getSingleElement(selection);
50     				IMethod method = getSelectedMethod(element);
51     				ILock accessLock = Platform.getJobManager().newLock();
52     				try {
53     					accessLock.acquire();
54     					HelloTacle ht = new HelloTacle();
55     					x[0] = ht.run(method);
56     				} finally {
57     					accessLock.release();
58     				}
59     				accessLock.release();
60     				return new Status(IStatus.INFO,
61     						"edu.ohio-state.cse.presto.tacle", IStatus.OK,
62     						"There are " + x[0] + " reachable methods.", null);
63     			}
64     		};
65     
66     		System.out.println("Job created");
67     		IAction jobAction = new Action() {
68     			public void run() {
69     				Shell shell = new Shell();
70     				MessageDialog.openInformation(shell, "Tacle Plug-in",
71     						"There are " + x[0] + " reachable methods");
72     			}
73     		};
74     		j.setProperty(IProgressConstants.KEEP_PROPERTY, Boolean.TRUE);
75     		j.setProperty(IProgressConstants.ACTION_PROPERTY, jobAction);
76     		j.schedule();
77     		System.out.println("Job scheduled");
78     	}

103     	public static Object getSingleElement(ISelection s) {
104     		if (!(s instanceof IStructuredSelection))
105     			return null;
106     		IStructuredSelection selection = (IStructuredSelection) s;
107     		if (selection.size() != 1)
108     			return null;
109     		return selection.getFirstElement();
110     
111     	}

113     
114     	private IMethod getSelectedMethod(Object element) {
115     		IMethod method = null;
116     		if (element instanceof IMethod) {
117     			method = (IMethod) element;
118     		}
119     		if(method.getCompilationUnit() == null){
120     			throw new RuntimeException("The selected method, "+method.getElementName()+", doesn't have a compilation unit");		
121     		}
122     		return method;
123     	}
