42     /**
43      * Scan application classes for NonNull annotations.
44      *
45      * @author David Hovemeyer
46      * @author William Pugh
47      *
48      * @deprecated AnnotationDatabases are being phased out, since annotations are
49      *             now stored directly in the XClass/XMethod/XField objects.
50      *             Resolving nullness annotations will be handled through the
51      *             JSR-305 type qualifier code.
52      */
53     @Deprecated
54     public class BuildNonNullAnnotationDatabase extends AnnotationVisitor {
55         private static final boolean DEBUG = SystemProperties.getBoolean("fnd.debug.annotation");
56     
57         //    private static final String DEFAULT_ANNOTATION_ANNOTATION_CLASS = "DefaultAnnotation";
58     
59         @StaticConstant
60         private static final Map<String, AnnotationDatabase.Target> defaultKind = new HashMap<String, AnnotationDatabase.Target>();
61         static {
62             defaultKind.put("", AnnotationDatabase.Target.ANY);
63             defaultKind.put("ForParameters", AnnotationDatabase.Target.PARAMETER);
64             defaultKind.put("ForMethods", AnnotationDatabase.Target.METHOD);
65             defaultKind.put("ForFields", AnnotationDatabase.Target.FIELD);
66     
67         }
68     
69         private final NullnessAnnotationDatabase database;
70     
71         public BuildNonNullAnnotationDatabase(@CheckForNull NullnessAnnotationDatabase database) {
72             this.database = database;
73         }
74     
75         static String lastPortion(String className) {
76             int i = className.lastIndexOf('.');
77             if (i < 0) {
78                 return className;
79             }
80             return className.substring(i + 1);
81         }
82     
83         private void handleClassElementValue(ClassElementValue value, Target annotationTarget) {
84             NullnessAnnotation n = NullnessAnnotation.Parser.parse(value.getClassString());
85             if (n != null) {
86                 database.addDefaultAnnotation(annotationTarget, getDottedClassName(), n);
87             }
88     
89         }
90     
91         @Override
92         public void visitAnnotation(String annotationClass, Map<String, ElementValue> map, boolean runtimeVisible) {
93     
94             if (database == null) {
95                 return;
96             }
97     
98             NullnessAnnotation n = NullnessAnnotation.Parser.parse(annotationClass);
99             annotationClass = lastPortion(annotationClass);
100             if (n == null) {
101                 if (annotationClass.startsWith("DefaultAnnotation")) {
102                     annotationClass = annotationClass.substring("DefaultAnnotation".length());
103     
104                     Target annotationTarget = defaultKind.get(annotationClass);
105                     if (annotationTarget != Target.METHOD) {
106                         return;
107                     }
108     
109                     ElementValue v = map.get("value");
110                     if (v instanceof ClassElementValue) {
111                         handleClassElementValue((ClassElementValue) v, annotationTarget);
112                     } else if (v instanceof ArrayElementValue) {
113                         for (ElementValue v2 : ((ArrayElementValue) v).getElementValuesArray()) {
114                             if (v2 instanceof ClassElementValue) {
115                                 handleClassElementValue((ClassElementValue) v2, annotationTarget);
116                             }
117                         }
118                     }
119     
120                     return;
121                 }
122     
123             } else if (visitingMethod()) {
124                 database.addDirectAnnotation(XFactory.createXMethod(this), n);
125             } else if (visitingField()) {
126                 database.addDirectAnnotation(XFactory.createXField(this), n);
127             }
128     
129         }
130     
131         @Override
132         public void visitSyntheticParameterAnnotation(int p, boolean runtimeVisible) {
133             if (database == null) {
134                 return;
135             }
136     
137             XMethod xmethod = XFactory.createXMethod(this);
138     
139             XMethodParameter xparameter = new XMethodParameter(xmethod, p);
140     
141             database.addDirectAnnotation(xparameter, NullnessAnnotation.UNKNOWN_NULLNESS);
142     
143         }
144     
145         @Override
146         public void visitParameterAnnotation(int p, String annotationClass, Map<String, ElementValue> map, boolean runtimeVisible) {
147             if (database == null) {
148                 return;
149             }
150     
151             NullnessAnnotation n = NullnessAnnotation.Parser.parse(annotationClass);
152             annotationClass = lastPortion(annotationClass);
153             if (n == null) {
154                 return;
155             }
156     
157             XMethod xmethod = XFactory.createXMethod(this);
158             if (DEBUG) {
159                 System.out.println("Parameter " + p + " @" + annotationClass.substring(annotationClass.lastIndexOf('/') + 1) + " in "
160                         + xmethod.toString());
161             }
162             XMethodParameter xparameter = new XMethodParameter(xmethod, p);
163     
164             database.addDirectAnnotation(xparameter, n);
165     
166         }
167     
168     }
