20     package edu.umd.cs.findbugs;
21     
22     import java.util.Iterator;
23     
24     import org.apache.bcel.classfile.JavaClass;
25     import org.apache.bcel.classfile.Method;
26     import org.apache.bcel.generic.MethodGen;
27     
28     import edu.umd.cs.findbugs.ba.CFGBuilderException;
29     import edu.umd.cs.findbugs.ba.ClassContext;
30     import edu.umd.cs.findbugs.ba.DataflowAnalysisException;
31     import edu.umd.cs.findbugs.ba.bcp.ByteCodePattern;
32     import edu.umd.cs.findbugs.ba.bcp.ByteCodePatternMatch;
33     import edu.umd.cs.findbugs.ba.bcp.PatternElementMatch;
34     import edu.umd.cs.findbugs.ba.bcp.PatternMatcher;

36     /**
37      * A base class for bug detectors that are based on a ByteCodePattern.
38      * ByteCodePatterns provide an easy way to detect patterns of bytecode
39      * instructions, taking into account control flow and uses of fields and values.
40      *
41      * @see ByteCodePattern
42      */
43     public abstract class ByteCodePatternDetector implements Detector {
44         private static final boolean DEBUG = SystemProperties.getBoolean("bcpd.debug");
45     
46         private static final String METHOD = SystemProperties.getProperty("bcpd.method");
47     
48         protected abstract BugReporter getBugReporter();
49     
50         @Override
51         public void visitClassContext(ClassContext classContext) {
52             try {
53                 ByteCodePattern pattern = getPattern();
54                 JavaClass jclass = classContext.getJavaClass();
55                 Method[] methodList = jclass.getMethods();
56     
57                 for (Method method : methodList) {
58                     if (method.isAbstract() || method.isNative()) {
59                         continue;
60                     }
61     
62                     if (METHOD != null && !method.getName().equals(METHOD)) {
63                         continue;
64                     }
65     
66                     if (DEBUG) {
67                         System.out.print("=====================================================================\n" + "Method "
68                                 + jclass.getClassName() + "." + method.getName() + "\n"
69                                 + "=====================================================================\n");
70                     }
71     
72                     if (!prescreen(method, classContext)) {
73                         continue;
74                     }
75     
76                     MethodGen methodGen = classContext.getMethodGen(method);
77                     if (methodGen == null) {
78                         continue;
79                     }
80     
81                     PatternMatcher matcher = new PatternMatcher(pattern, classContext, method);
82                     matcher.execute();
83     
84                     Iterator<ByteCodePatternMatch> j = matcher.byteCodePatternMatchIterator();
85                     while (j.hasNext()) {
86                         ByteCodePatternMatch match = j.next();
87     
88                         if (DEBUG) {
89                             System.out.println("Pattern match:");
90                             Iterator<PatternElementMatch> pemIter = match.patternElementMatchIterator();
91                             while (pemIter.hasNext()) {
92                                 PatternElementMatch pem = pemIter.next();
93                                 System.out.println("\t" + pem.toString());
94                             }
95                         }
96     
97                         reportMatch(classContext, method, match);
98                     }
99                 }
100             } catch (DataflowAnalysisException e) {
101                 getBugReporter().logError(getDetectorName() + " caught exception", e);
102             } catch (CFGBuilderException e) {
103                 getBugReporter().logError(getDetectorName() + " caught exception", e);
104             }
105         }
106     
107         private String getDetectorName() {
108             String className = this.getClass().getName();
109             int lastDot = className.lastIndexOf('.');
110             if (lastDot >= 0) {
111                 className = className.substring(lastDot + 1);
112             }
113             return className;
114         }
115     
116         @Override
117         public void report() {
118         }
119     
120         /**
121          * Get the ByteCodePattern for this detector.
122          */
123         public abstract ByteCodePattern getPattern();
124     
125         /**
126          * Prescreen a method. It is a valid, but dumb, implementation simply to
127          * return true unconditionally. A better implementation is to call
128          * ClassContext.getBytecodeSet() to check whether the method actually
129          * contains the bytecode instructions that the pattern will look for. The
130          * theory is that checking the bytecode set is very fast, while building the
131          * MethodGen, CFG, ValueNumberAnalysis, etc. objects required to match
132          * ByteCodePatterns is slow, and the bytecode pattern matching algorithm is
133          * also not particularly fast.
134          * <p/>
135          * <p>
136          * As a datapoint, prescreening speeds up the BCPDoubleCheck detector <b>by
137          * a factor of 5</b> with no loss of generality and only a dozen or so extra
138          * lines of code.
139          *
140          * @param method
141          *            the method
142          * @param classContext
143          *            the ClassContext for the method
144          * @return true if the method should be analyzed for instances of the
145          *         ByteCodePattern
146          */
147         public abstract boolean prescreen(Method method, ClassContext classContext);
148     
149         /**
150          * Called to report an instance of the ByteCodePattern.
151          *
152          * @param classContext
153          *            the ClassContext for the analyzed class
154          * @param method
155          *            the method to instance appears in
156          * @param match
157          *            the ByteCodePatternMatch object representing the match of the
158          *            ByteCodePattern against actual instructions in the method
159          */
160         public abstract void reportMatch(ClassContext classContext, Method method, ByteCodePatternMatch match)
161                 throws CFGBuilderException, DataflowAnalysisException;
162     }
