20     package edu.umd.cs.findbugs.util;
21     
22     import junit.framework.TestCase;
23     
24     public class StringsTest extends TestCase {
25     
26         public static String[] escapedStrings = {
27             // mixed entities/unicode escape sequences
28             "a b c 1 2 3 &amp; &lt; &gt; &quot; &apos; \\u0005 \\u0013 &#955; \\\\u0007",
29             // *even* series of prefixed slashes + \\u -> do escape
30             "a b c \\\\\\u0005",
31             // *odd* series of prefixed slashes + \\u -> don't escape
32             "a b c \\\\\\\\u0005",
33             // mixed even/odd prefixed slashes
34             "a b c \\\\\\u0005 \\\\\\\\u0013",
35             // make sure slashes work on their own (no double un/escaping)
36             "\\\\\\",
37             // make sure that normal characters are handled correctly if they
38             // appear after escapes
39             "a b c 1 2 3 &amp; &lt; &gt; &quot; &apos; \\u0005 \\u0013 &#955; \\\\u0007 a b c 1 2 3",
40             // escaping a null string should be safe
41             null,
42             // an empty string should be safe too
43             "", };
44     
45         public static String[] unescapedStrings = { "a b c 1 2 3 & < > \" ' \u0005 \u0013 \u03BB \\\\u0007", "a b c \\\\\u0005",
46             "a b c \\\\\\\\u0005", "a b c \\\\\u0005 \\\\\\\\u0013", "\\\\\\",
47             "a b c 1 2 3 & < > \" ' \u0005 \u0013 \u03BB \\\\u0007 a b c 1 2 3", null, "", };
48     
49         public void testEscapeXml() {
50             assert (escapedStrings.length == unescapedStrings.length);
51             for (int i = 0; i < unescapedStrings.length; i++) {
52                 if (unescapedStrings[i] == null) {
53                     assert (Strings.escapeXml(unescapedStrings[i]) == null);
54                 } else {
55                     assert (Strings.escapeXml(unescapedStrings[i]).compareTo(escapedStrings[i]) == 0);
56                 }
57             }
58         }
59     
60         public void testUnescapeXml() {
61             assert (escapedStrings.length == unescapedStrings.length);
62             for (int i = 0; i < escapedStrings.length; i++) {
63                 if (escapedStrings[i] == null) {
64                     assert (Strings.unescapeXml(escapedStrings[i]) == null);
65                 } else {
66                     assert (Strings.unescapeXml(escapedStrings[i]).compareTo(unescapedStrings[i]) == 0);
67                 }
68             }
69         }
70     
71         public void checkEscapeLFCRBackSlash(String expected, String argument) {
72             assertEquals(argument, expected, Strings.escapeLFCRBackSlash(argument));
73     
74         }
75     
76         public void testEscapeLFCRBackSlash() {
77     
78             checkEscapeLFCRBackSlash("abc", "abc");
79             checkEscapeLFCRBackSlash("\\n", "\n");
80             checkEscapeLFCRBackSlash("\\r", "\r");
81             checkEscapeLFCRBackSlash("\\\\a\\r", "\\a\r");
82         }
83     
84     }
