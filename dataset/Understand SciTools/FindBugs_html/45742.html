20     package edu.umd.cs.findbugs.detect;
21     
22     import java.util.HashSet;
23     
24     import org.apache.bcel.Constants;
25     import org.apache.bcel.classfile.Code;
26     import org.apache.bcel.classfile.Field;
27     import org.apache.bcel.classfile.JavaClass;
28     
29     import edu.umd.cs.findbugs.BugInstance;
30     import edu.umd.cs.findbugs.BugReporter;
31     import edu.umd.cs.findbugs.OpcodeStack;
32     import edu.umd.cs.findbugs.Priorities;
33     import edu.umd.cs.findbugs.ba.AnalysisContext;
34     import edu.umd.cs.findbugs.ba.NullnessAnnotation;
35     import edu.umd.cs.findbugs.ba.XFactory;
36     import edu.umd.cs.findbugs.ba.XField;
37     import edu.umd.cs.findbugs.bcel.OpcodeStackDetector;
38     
39     public class InitializeNonnullFieldsInConstructor extends OpcodeStackDetector {
40     
41         final BugReporter bugReporter;
42     
43         final HashSet<XField> initializedFields = new HashSet<XField>();
44     
45         final HashSet<XField> nonnullFields = new HashSet<XField>();
46     
47         final HashSet<XField> nonnullStaticFields = new HashSet<XField>();
48     
49         public InitializeNonnullFieldsInConstructor(BugReporter bugReporter) {
50             this.bugReporter = bugReporter;
51         }
52     
53         @Override
54         public void setupVisitorForClass(JavaClass obj) {
55             super.setupVisitorForClass(obj);
56             nonnullFields.clear();
57         }
58     
59         @Override
60         public void visitAfter(JavaClass obj) {
61             super.visitAfter(obj);
62             nonnullFields.clear();
63             nonnullStaticFields.clear();
64     
65         }
66     
67         @Override
68         public void visit(Field obj) {
69             super.visit(obj);
70             XField f = XFactory.createXField(this);
71             if (checkForInitialization(f) && !f.isSynthetic()) {
72                 if (f.isStatic()) {
73                     nonnullStaticFields.add(f);
74                 } else {
75                     nonnullFields.add(f);
76                 }
77             }
78         }
79     
80         public boolean checkForInitialization(XField f) {
81             if (!f.isReferenceType() || f.isFinal()) {
82                 return false;
83             }
84             NullnessAnnotation annotation = AnalysisContext.currentAnalysisContext().getNullnessAnnotationDatabase()
85                     .getResolvedAnnotation(f, false);
86             boolean isNonnull = annotation == NullnessAnnotation.NONNULL;
87             return isNonnull;
88         }
89     
90         @Override
91         public void visit(Code code) {
92             boolean interesting = "<init>".equals(getMethodName()) || "<clinit>".equals(getMethodName());
93             if (!interesting) {
94                 return;
95             }
96     
97             secondaryConstructor = false;
98             HashSet<XField> needToInitialize = getMethod().isStatic() ? nonnullStaticFields : nonnullFields;
99             if (needToInitialize.isEmpty()) {
100                 return;
101             }
102             // initialize any variables we want to initialize for the method
103             super.visit(code); // make callbacks to sawOpcode for all opcodes
104             if (!secondaryConstructor && !initializedFields.containsAll(needToInitialize)) {
105                 int priority = Priorities.NORMAL_PRIORITY;
106                 if (needToInitialize.size() - initializedFields.size() == 1 && needToInitialize.size() > 1) {
107                     priority = Priorities.HIGH_PRIORITY;
108                 }
109     
110                 for (XField f : needToInitialize) {
111                     if (initializedFields.contains(f)) {
112                         continue;
113                     }
114     
115                     BugInstance b = new BugInstance(this, "NP_NONNULL_FIELD_NOT_INITIALIZED_IN_CONSTRUCTOR", priority)
116                     .addClassAndMethod(this).addField(f);
117                     bugReporter.reportBug(b);
118                 }
119     
120             }
121             initializedFields.clear();
122     
123         }
124     
125         boolean secondaryConstructor;
126     
127         @Override
128         public void sawOpcode(int seen) {
129     
130             if (secondaryConstructor) {
131                 return;
132             }
133     
134             switch (seen) {
135             case Constants.INVOKESPECIAL:
136                 if (!getMethod().isStatic() && "<init>".equals(getNameConstantOperand()) && isSelfOperation()) {
137                     OpcodeStack.Item invokedOn = stack.getItemMethodInvokedOn(this);
138                     if (invokedOn.isInitialParameter() && invokedOn.getRegisterNumber() == 0) {
139                         secondaryConstructor = true;
140                     }
141                     break;
142                 }
143                 break;
144             case Constants.PUTFIELD:
145                 if (getMethod().isStatic()) {
146                     return;
147                 }
148                 OpcodeStack.Item left = stack.getStackItem(1);
149                 if (left.isInitialParameter() && left.getRegisterNumber() == 0 && isSelfOperation()) {
150                     XField f = getXFieldOperand();
151                     if (f == null) {
152                         break;
153                     }
154                     if (checkForInitialization(f)) {
155                         initializedFields.add(f);
156                     }
157                 }
158                 break;
159             case Constants.PUTSTATIC:
160                 if (!getMethod().isStatic()) {
161                     break;
162                 }
163     
164                 if (isSelfOperation()) {
165                     XField f = getXFieldOperand();
166                     if (f == null) {
167                         break;
168                     }
169     
170                     if (checkForInitialization(f)) {
171                         initializedFields.add(f);
172                     }
173                 }
174                 break;
175             default:
176                 break;
177     
178             }
179     
180         }
181     
182         public boolean isSelfOperation() {
183             return getClassConstantOperand().equals(getClassName());
184         }
185     
186     }
