40     /**
41      * Analysis factory which creates instances of ObligationDataflow.
42      *
43      * @author David Hovemeyer
44      */
45     public class ObligationDataflowFactory extends AnalysisFactory<ObligationDataflow> {
46     
47         private static final boolean DEBUG_PRINTCFG = SystemProperties.getBoolean("oa.printcfg");
48     
49         public ObligationDataflowFactory() {
50             super("Obligation dataflow", ObligationDataflow.class);
51         }
52     
53         @Override
54         public ObligationDataflow analyze(IAnalysisCache analysisCache, MethodDescriptor methodDescriptor)
55                 throws CheckedAnalysisException {
56             CFG cfg = analysisCache.getMethodAnalysis(CFG.class, methodDescriptor);
57             DepthFirstSearch dfs = analysisCache.getMethodAnalysis(DepthFirstSearch.class, methodDescriptor);
58             XMethod xmethod = XFactory.createXMethod(methodDescriptor);
59             ConstantPoolGen cpg = analysisCache.getClassAnalysis(ConstantPoolGen.class, methodDescriptor.getClassDescriptor());
60     
61             ObligationPolicyDatabase database = analysisCache.getDatabase(ObligationPolicyDatabase.class);
62     
63             TypeDataflow typeDataflow = analysisCache.getMethodAnalysis(TypeDataflow.class, methodDescriptor);
64             IsNullValueDataflow invDataflow = analysisCache.getMethodAnalysis(IsNullValueDataflow.class, methodDescriptor);
65     
66             ObligationFactory factory = database.getFactory();
67     
68             ObligationAnalysis analysis = new ObligationAnalysis(dfs, xmethod, cpg, factory, database, typeDataflow, invDataflow,
69                     analysisCache.getErrorLogger());
70             ObligationDataflow dataflow = new ObligationDataflow(cfg, analysis);
71     
72             Profiler profiler = analysisCache.getProfiler();
73             profiler.start(analysis.getClass());
74             try {
75                 dataflow.execute();
76             } finally {
77                 profiler.end(analysis.getClass());
78             }
79     
80             if (DEBUG_PRINTCFG) {
81                 System.out.println("Dataflow CFG:");
82                 DataflowCFGPrinter.printCFG(dataflow, System.out);
83             }
84     
85             return dataflow;
86         }
87     }
