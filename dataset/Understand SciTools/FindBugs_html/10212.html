22     package edu.umd.cs.findbugs.graph;
23     
24     import java.util.Iterator;
25     import java.util.Set;
26     import java.util.TreeSet;
27     
28     /**
29      * Algorithm to merge a set of vertices into a single vertex. Note that the
30      * graph is modified as part of this process.
31      */
32     public class MergeVertices<GraphType extends Graph<EdgeType, VertexType>, EdgeType extends GraphEdge<EdgeType, VertexType>, VertexType extends GraphVertex<VertexType>> {
33     
34         /**
35          * Constructor.
36          */
37         public MergeVertices() {
38         }
39     
40         /**
41          * Merge the specified set of vertices into a single vertex.
42          *
43          * @param vertexSet
44          *            the set of vertices to be merged
45          * @param g
46          *            the graph to be modified
47          * @param combinator
48          *            object used to combine vertices
49          * @param toolkit
50          *            GraphToolkit used to copy auxiliary information for edges
51          */
52         public void mergeVertices(Set<VertexType> vertexSet, GraphType g, VertexCombinator<VertexType> combinator,
53                 GraphToolkit<GraphType, EdgeType, VertexType> toolkit) {
54     
55             // Special case: if the vertex set contains a single vertex
56             // or is empty, there is nothing to do
57             if (vertexSet.size() <= 1) {
58                 return;
59             }
60     
61             // Get all vertices to which we have outgoing edges
62             // or from which we have incoming edges, since they'll need
63             // to be fixed
64             TreeSet<EdgeType> edgeSet = new TreeSet<EdgeType>();
65             for (Iterator<EdgeType> i = g.edgeIterator(); i.hasNext();) {
66                 EdgeType e = i.next();
67                 if (vertexSet.contains(e.getSource()) || vertexSet.contains(e.getTarget())) {
68                     edgeSet.add(e);
69                 }
70             }
71     
72             // Combine all of the vertices into a single composite vertex
73             VertexType compositeVertex = combinator.combineVertices(vertexSet);
74     
75             // For each original edge into or out of the vertex set,
76             // create an equivalent edge referencing the composite
77             // vertex
78             for (EdgeType e : edgeSet) {
79                 VertexType source = vertexSet.contains(e.getSource()) ? compositeVertex : e.getSource();
80                 VertexType target = vertexSet.contains(e.getTarget()) ? compositeVertex : e.getTarget();
81     
82                 // if (source != compositeVertex && target != compositeVertex)
83                 // System.out.println("BIG OOPS!");
84     
85                 // Don't create a self edge for the composite vertex
86                 // unless one of the vertices in the vertex set
87                 // had a self edge
88                 if (source == compositeVertex && target == compositeVertex && e.getSource() != e.getTarget()) {
89                     continue;
90                 }
91     
92                 // Don't create duplicate edges.
93                 if (g.lookupEdge(source, target) != null) {
94                     continue;
95                 }
96     
97                 EdgeType compositeEdge = g.createEdge(source, target);
98                 // FIXME: we really should have an EdgeCombinator here
99                 toolkit.copyEdge(e, compositeEdge);
100             }
101     
102             // Remove all of the vertices in the vertex set; this will
103             // automatically remove the edges into and out of those
104             // vertices
105             for (VertexType aVertexSet : vertexSet) {
106                 g.removeVertex(aVertexSet);
107             }
108     
109         }
110     
111     }
