20     package edu.umd.cs.findbugs.gui2;
21     
22     import java.io.File;
23     
24     import javax.swing.filechooser.FileFilter;
25     
26     public final class FindBugsFilterFileFilter extends FileFilter {
27     
28         public static final FindBugsFilterFileFilter INSTANCE = new FindBugsFilterFileFilter();
29     
30         @Override
31         public boolean accept(File arg0) {
32             return arg0.getName().endsWith(".xml") || arg0.isDirectory();
33         }
34     
35         @Override
36         public String getDescription() {
37             return "FindBugs Filter Files (*.xml)";
38         }
39     
40     }
