41     /**
42      * @author Tagir Valeev
43      */
44     public class CovariantArrayAssignment extends OpcodeStackDetector {
45         private final BugAccumulator accumulator;
46     
47         public CovariantArrayAssignment(BugReporter bugReporter) {
48             accumulator = new BugAccumulator(bugReporter);
49         }
50     
51         @Override
52         public void visit(Code obj) {
53             super.visit(obj);
54             accumulator.reportAccumulatedBugs();
55         }
56     
57         /**
58          * @param superClass
59          * @param subClass
60          * @return true if superClass is abstract or interface and all known non-abstract implementations
61          * are derived from given subClass
62          */
63         private static boolean allImplementationsDerivedFromSubclass(@SlashedClassName String superClass, @SlashedClassName String subClass) {
64             ClassDescriptor superDescriptor = DescriptorFactory.createClassDescriptor(superClass);
65             XClass xClass = AnalysisContext.currentXFactory().getXClass(superDescriptor);
66             if(xClass == null || (!xClass.isInterface() && !xClass.isAbstract())) {
67                 return false;
68             }
69             try {
70                 ClassDescriptor wantedDescriptor = DescriptorFactory.createClassDescriptor(subClass);
71                 Subtypes2 subtypes2 = AnalysisContext.currentAnalysisContext().getSubtypes2();
72                 for (ClassDescriptor subDescriptor : subtypes2.getSubtypes(superDescriptor)) {
73                     if (subDescriptor.equals(superDescriptor) || subDescriptor.equals(wantedDescriptor)) {
74                         continue;
75                     }
76                     XClass xSubClass = AnalysisContext.currentXFactory().getXClass(subDescriptor);
77                     if (xSubClass == null
78                             || (!xSubClass.isAbstract() && !xSubClass.isInterface() && !subtypes2.isSubtype(subDescriptor,
79                                     wantedDescriptor))) {
80                         return false;
81                     }
82                 }
83                 return true;
84             } catch (ClassNotFoundException e) {
85                 // unresolved class
86             }
87             return false;
88         }

90         @Override
91         public void sawOpcode(int seen) {
92             if ((isRegisterStore() && !isRegisterLoad()) || seen == PUTFIELD || seen == PUTSTATIC || seen == ARETURN) {
93                 Item valueItem = getStack().getStackItem(0);
94                 if(!valueItem.isNull() && valueItem.isNewlyAllocated() && valueItem.getSignature().startsWith("[L")
95                         && !((Integer)0).equals(valueItem.getConstant())) {
96                     String valueClass = valueItem.getSignature().substring(2, valueItem.getSignature().length()-1);
97                     String arraySignature = null;
98                     int priority = LOW_PRIORITY;
99                     String pattern = null;
100                     FieldDescriptor field = null;
101                     if(seen == PUTFIELD || seen == PUTSTATIC) {
102                         arraySignature = getSigConstantOperand();
103                         pattern = "CAA_COVARIANT_ARRAY_FIELD";
104                         field = getFieldDescriptorOperand();
105                         if(field instanceof XField) {
106                             XField xField = (XField)field;
107                             if((xField.isPublic() || xField.isProtected())) {
108                                 XClass xClass = AnalysisContext.currentXFactory().getXClass(xField.getClassDescriptor());
109                                 if(xClass != null && xClass.isPublic()) {
110                                     priority = NORMAL_PRIORITY;
111                                 }
112                             }
113                         }
114                     } else if(seen == ARETURN) {
115                         if(getXMethod().bridgeFrom() == null) {
116                             pattern = "CAA_COVARIANT_ARRAY_RETURN";
117                             arraySignature = new SignatureParser(getMethodSig()).getReturnTypeSignature();
118                             if (!arraySignature.equals("[Ljava/lang/Object;")
119                                     && (getXMethod().isPublic() || getXMethod().isProtected()) && getXClass().isPublic()) {
120                                 priority = NORMAL_PRIORITY;
121                             }
122                         }
123                     } else {
124                         LocalVariableTable lvt = getMethod().getLocalVariableTable();
125                         if(lvt != null) {
126                             LocalVariable localVariable = lvt.getLocalVariable(getRegisterOperand(), getNextPC());
127                             if(localVariable != null) {
128                                 pattern = "CAA_COVARIANT_ARRAY_LOCAL";
129                                 arraySignature = localVariable.getSignature();
130                             }
131                         }
132                     }
133                     if(arraySignature != null && arraySignature.startsWith("[L")) {
134                         String arrayClass = arraySignature.substring(2, arraySignature.length()-1);
135                         if(!valueClass.equals(arrayClass)) {
136                             if(priority == NORMAL_PRIORITY && allImplementationsDerivedFromSubclass(arrayClass, valueClass)) {
137                                 priority = LOW_PRIORITY;
138                             }
139                             BugInstance bug = new BugInstance(this, pattern, priority).addClassAndMethod(this)
140                                     .addFoundAndExpectedType(valueItem.getSignature(), arraySignature)
141                                     .addSourceLine(this).addValueSource(valueItem, this);
142                             if(field != null) {
143                                 bug.addField(field);
144                             }
145                             accumulator.accumulateBug(bug, this);
146                         }
147                     }
148                 }
149             }
150     
151             if (seen == AASTORE) {
152                 Item valueItem = getStack().getStackItem(0);
153                 if(!valueItem.isNull()) {
154                     Item arrayItem = getStack().getStackItem(2);
155                     String arraySignature = arrayItem.getSignature();
156                     String valueSignature = valueItem.getSignature();
157                     // if valueSignature is "Ljava/lang/Object;" then OpcodeStack probably could not define actual type at all: skip this case
158                     if(arraySignature.startsWith("[L") && valueSignature.startsWith("L") && !valueSignature.equals("Ljava/lang/Object;")) {
159                         String arrayClass = arraySignature.substring(2, arraySignature.length()-1);
160                         String valueClass = valueSignature.substring(1, valueSignature.length()-1);
161                         try {
162                             ClassDescriptor valueClassDescriptor = DescriptorFactory.createClassDescriptor(valueClass);
163                             ClassDescriptor arrayClassDescriptor = DescriptorFactory.createClassDescriptor(arrayClass);
164                             if (!AnalysisContext.currentAnalysisContext().getSubtypes2()
165                                     .isSubtype(valueClassDescriptor, arrayClassDescriptor)) {
166                                 int priority = HIGH_PRIORITY;   // in this case we may be pretty sure that if this line is executed ArrayStoreException will happen
167                                 if(AnalysisContext.currentAnalysisContext().getSubtypes2().isSubtype(arrayClassDescriptor, valueClassDescriptor)) {
168                                     priority = NORMAL_PRIORITY;
169                                     if(allImplementationsDerivedFromSubclass(valueClass, arrayClass)) {
170                                         // Every implementation of valueClass also extends arrayClass
171                                         // In this case ArrayStoreException will never occur in current project
172                                         // So it's enough that we reported a bug when this array was created
173                                         priority = IGNORE_PRIORITY;
174                                     }
175                                 }
176                                 BugInstance bug = new BugInstance(this, "CAA_COVARIANT_ARRAY_ELEMENT_STORE", priority).addClassAndMethod(this)
177                                         .addFoundAndExpectedType(valueSignature, 'L'+arrayClass+';')
178                                         .addSourceLine(this)
179                                         .addValueSource(valueItem, this)
180                                         .addValueSource(arrayItem, this);
181                                 accumulator.accumulateBug(bug, this);
182                             }
183                         } catch (ClassNotFoundException e) {
184                             // Probably class was not supplied to the analysis: assume that everything is correct
185                         }
186                     }
187                 }
188             }
189         }
190     }
