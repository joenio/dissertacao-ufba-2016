57         static class CSRCommandLine extends CommandLine {
58     
59             final CSPoptions options;
60     
61             public CSRCommandLine(CSPoptions options) {
62                 this.options = options;
63                 addOption("-cloud", "id", "id of the cloud to use");
64                 addOption("-recent", "hours", "maximum age in hours for an issue to be recent");
65                 addOption("-cloudSummary", "file", "write a cloud summary to thie file");
66             }
67     
68             /*
69              * (non-Javadoc)
70              *
71              * @see
72              * edu.umd.cs.findbugs.config.CommandLine#handleOption(java.lang.String,
73              * java.lang.String)
74              */
75             @Override
76             protected void handleOption(String option, String optionExtraPart) throws IOException {
77                 throw new IllegalArgumentException("Unknown option : " + option);
78     
79             }
80     
81             /*
82              * (non-Javadoc)
83              *
84              * @see
85              * edu.umd.cs.findbugs.config.CommandLine#handleOptionWithArgument(java
86              * .lang.String, java.lang.String)
87              */
88             @Override
89             protected void handleOptionWithArgument(String option, String argument) throws IOException {
90                 if ("-cloud".equals(option)) {
91                     options.cloudId = argument;
92                 } else if ("-recent".equals(option)) {
93                     options.ageInHours = Integer.parseInt(argument);
94                 } else if ("-cloudSummary".equals(option)) {
95                     options.cloudSummary = argument;
96                 } else {
97                     throw new IllegalArgumentException("Unknown option : " + option);
98                 }
99             }
100         }

102         public static void main(String[] argv) throws Exception {
103     
104             FindBugs.setNoAnalysis();
105             final CSPoptions options = new CSPoptions();
106             final CSRCommandLine commandLine = new CSRCommandLine(options);
107             int argCount = commandLine.parse(argv, 0, 1, "Usage: " + CloudSyncAndReport.class.getName()
108                     + " [options] [<results1> <results2> ... <resultsn>] ");
109     
110             if (argCount < argv.length) {
111                 options.analysisFile = argv[argCount];
112             }
113     
114             CloudSyncAndReport csr = new CloudSyncAndReport(options);
115             csr.load();
116     
117             csr.sync();
118             PrintWriter out = UTF8.printWriter(System.out);
119             csr.report(out);
120             out.flush();
121             csr.shutdown();
122             out.close();
123         }

125         final CSPoptions options;
126     
127         final SortedBugCollection bugCollection = new SortedBugCollection();
128     
129         /**
130          * @param options
131          */
132         public CloudSyncAndReport(CSPoptions options) {
133             this.options = options;
134     
135         }
136     
137         public void load() throws IOException, DocumentException {
138             if (options.analysisFile == null) {
139                 bugCollection.readXML(UTF8.bufferedReader(System.in));
140             } else {
141                 bugCollection.readXML(options.analysisFile);
142             }
143             if (options.cloudId != null && !options.cloudId.equals(bugCollection.getProject().getCloudId())) {
144                 bugCollection.getProject().setCloudId(options.cloudId);
145                 bugCollection.reinitializeCloud();
146             }
147         }
148     
149         public void sync() {
150             Cloud cloud = bugCollection.getCloud();
151             cloud.initiateCommunication();
152             cloud.waitUntilIssueDataDownloaded();
153         }
154     
155         static class Stats {
156             int total, recent;
157     
158         }
159     
160         public void report(PrintWriter out) {
161             TreeMap<BugRankCategory, Stats> stats = new TreeMap<BugRankCategory, Stats>();
162             ProjectStats projectStats = bugCollection.getProjectStats();
163             Collection<BugInstance> bugs = bugCollection.getCollection();
164             Cloud cloud = bugCollection.getCloud();
165             cloud.setMode(Cloud.Mode.COMMUNAL);
166     
167             out.printf("Cloud sync and summary report for %s%n", bugCollection.getProject().getProjectName());
168     
169             out.printf("Code dated %s%n", new Date(bugCollection.getTimestamp()));
170             out.printf("Code analyzed %s%n", new Date(bugCollection.getAnalysisTimestamp()));
171     
172             out.printf("%7d total classes%n", projectStats.getNumClasses());
173             out.printf("%7d total issues%n", bugs.size());
174             long recentTimestamp = System.currentTimeMillis() - options.ageInHours * 3600 * 1000L;
175             int allRecentIssues = 0;
176     
177             for (BugInstance b : bugs) {
178                 Stats s = stats.get(BugRankCategory.getRank(b.getBugRank()));
179                 if (s == null) {
180                     s = new Stats();
181                     stats.put(BugRankCategory.getRank(b.getBugRank()), s);
182                 }
183                 s.total++;
184                 long firstSeen = cloud.getFirstSeen(b);
185                 if (firstSeen > recentTimestamp) {
186                     s.recent++;
187                     allRecentIssues++;
188                 }
189     
190             }
191             out.printf("%7d recent issues%n", allRecentIssues);
192     
193             if (options.cloudSummary != null && cloud.supportsCloudSummaries()) {
194                 try {
195                     PrintWriter cs = UserTextFile.printWriter(options.cloudSummary);
196                     cs.printf("%6s %6s %s%n", "recent", "total", "Rank category");
197                     for (Entry<BugRankCategory, Stats> e : stats.entrySet()) {
198                         Stats s = e.getValue();
199                         if (s.total > 0) {
200                             cs.printf("%6d %6d %s%n", s.recent, s.total, e.getKey());
201                         }
202                     }
203                     cs.println();
204                     cloud.printCloudSummary(cs, bugs, null);
205                     cs.close();
206                 } catch (Exception e) {
207                     out.println("Error writing cloud summary to " + options.cloudSummary);
208                     e.printStackTrace(out);
209                 }
210     
211             }
212     
213         }
214     
215         public void shutdown() {
216             Cloud cloud = bugCollection.getCloud();
217             cloud.shutdown();
218         }
