173         private boolean prescreen(ClassContext classContext, Method method) {
174             return true;
175         }
176     
177         @Override
178         public void visitClassContext(ClassContext classContext) {
179             JavaClass javaClass = classContext.getJavaClass();
180     
181             Method[] methodList = javaClass.getMethods();
182     
183             for (Method method : methodList) {
184                 MethodGen methodGen = classContext.getMethodGen(method);
185                 if (methodGen == null) {
186                     continue;
187                 }
188     
189                 if (!prescreen(classContext, method)) {
190                     continue;
191                 }
192     
193                 try {
194                     analyzeMethod(classContext, method);
195                 } catch (DataflowAnalysisException e) {
196                     bugReporter.logError("Error analyzing " + method.toString(), e);
197                 } catch (CFGBuilderException e) {
198                     bugReporter.logError("Error analyzing " + method.toString(), e);
199                 }
200             }
201         }

203         private void analyzeMethod(ClassContext classContext, Method method) throws DataflowAnalysisException, CFGBuilderException {
204             if (BCELUtil.isSynthetic(method) || (method.getAccessFlags() & Constants.ACC_BRIDGE) == Constants.ACC_BRIDGE) {
205                 return;
206             }
207     
208             if (DEBUG) {
209                 System.out.println("    Analyzing method " + classContext.getJavaClass().getClassName() + "." + method.getName());
210             }
211     
212             JavaClass javaClass = classContext.getJavaClass();
213             BitSet linesMentionedMultipleTimes = classContext.linesMentionedMultipleTimes(method);
214             BugAccumulator accumulator = new BugAccumulator(bugReporter);
215             Dataflow<BitSet, LiveLocalStoreAnalysis> llsaDataflow = classContext.getLiveLocalStoreDataflow(method);
216     
217             int numLocals = method.getCode().getMaxLocals();
218             int[] localStoreCount = new int[numLocals];
219             int[] localLoadCount = new int[numLocals];
220             int[] localIncrementCount = new int[numLocals];
221             MethodGen methodGen = classContext.getMethodGen(method);
222             CFG cfg = classContext.getCFG(method);
223             if (cfg.isFlagSet(CFG.FOUND_INEXACT_UNCONDITIONAL_THROWERS)) {
224                 return;
225             }
226             BitSet liveStoreSetAtEntry = llsaDataflow.getAnalysis().getResultFact(cfg.getEntry());
227             BitSet complainedAbout = new BitSet();
228             TypeDataflow typeDataflow = classContext.getTypeDataflow(method);
229     
230             // Get number of locals that are parameters.
231             int localsThatAreParameters = PreorderVisitor.getNumberArguments(method.getSignature());
232             if (!method.isStatic()) {
233                 localsThatAreParameters++;
234             }
235     
236             // Scan method to determine number of loads, stores, and increments
237             // of local variables.
238             countLocalStoresLoadsAndIncrements(localStoreCount, localLoadCount, localIncrementCount, cfg);
239             for (int i = 0; i < localsThatAreParameters; i++) {
240                 localStoreCount[i]++;
241             }
242     
243             // For each source line, keep track of # times
244             // the line was a live store. This can eliminate false positives
245             // due to inlining of finally blocks.
246             BitSet liveStoreSourceLineSet = new BitSet();
247     
248             // Scan method for
249             // - dead stores
250             // - stores to parameters that are dead upon entry to the method
251             for (Iterator<Location> i = cfg.locationIterator(); i.hasNext();) {
252                 Location location = i.next();
253     
254                 BugInstance pendingBugReportAboutOverwrittenParameter = null;
255                 try {
256                     WarningPropertySet<WarningProperty> propertySet = new WarningPropertySet<WarningProperty>();
257                     // Skip any instruction which is not a store
258                     if (!isStore(location)) {
259                         continue;
260                     }
261     
262                     // Heuristic: exception handler blocks often contain
263                     // dead stores generated by the compiler.
264                     if (location.getBasicBlock().isExceptionHandler()) {
265                         propertySet.addProperty(DeadLocalStoreProperty.EXCEPTION_HANDLER);
266                     }
267                     InstructionHandle handle = location.getHandle();
268                     int pc = handle.getPosition();
269                     IndexedInstruction ins = (IndexedInstruction) location.getHandle().getInstruction();
270     
271                     int local = ins.getIndex();
272     
273                     // Get live stores at this instruction.
274                     // Note that the analysis also computes which stores were
275                     // killed by a subsequent unconditional store.
276                     BitSet liveStoreSet = llsaDataflow.getAnalysis().getFactAtLocation(location);
277     
278                     // Is store alive?
279                     boolean storeLive = llsaDataflow.getAnalysis().isStoreAlive(liveStoreSet, local);
280     
281                     LocalVariableAnnotation lvAnnotation = LocalVariableAnnotation.getLocalVariableAnnotation(method, location, ins);
282     
283                     String sourceFileName = javaClass.getSourceFileName();
284                     if ("?".equals(lvAnnotation.getName())) {
285                         if (sourceFileName.endsWith(".groovy")) {
286                             continue;
287                         }
288                         if (method.getCode().getLocalVariableTable() != null) {
289                             continue;
290                         }
291                     }
292     
293                     SourceLineAnnotation sourceLineAnnotation = SourceLineAnnotation.fromVisitedInstruction(classContext, methodGen,
294                             sourceFileName, location.getHandle());
295     
296                     if (DEBUG) {
297                         System.out.println("    Store at " + sourceLineAnnotation.getStartLine() + "@"
298                                 + location.getHandle().getPosition() + " is " + (storeLive ? "live" : "dead"));
299                         System.out.println("Previous is: " + location.getHandle().getPrev());
300                     }
301     
302                     // Note source lines of live stores.
303                     if (storeLive && sourceLineAnnotation.getStartLine() > 0) {
304                         liveStoreSourceLineSet.set(sourceLineAnnotation.getStartLine());
305                     }
306     
307                     String lvName = lvAnnotation.getName();
308                     if (lvName.charAt(0) == '$' || lvName.charAt(0) == '_') {
309                         propertySet.addProperty(DeadLocalStoreProperty.SYNTHETIC_NAME);
310                     }
311                     if (EXCLUDED_LOCALS.contains(lvName)) {
312                         continue;
313                     }
314                     propertySet.setProperty(DeadLocalStoreProperty.LOCAL_NAME, lvName);
315     
316                     boolean isParameter = local < localsThatAreParameters;
317                     if (isParameter) {
318                         propertySet.addProperty(DeadLocalStoreProperty.IS_PARAMETER);
319                     }
320     
321                     Field shadowedField = null;
322     
323                     for (Field f : javaClass.getFields()) {
324                         if (f.getName().equals(lvName) && f.isStatic() == method.isStatic()) {
325                             shadowedField = f;
326                             propertySet.addProperty(DeadLocalStoreProperty.SHADOWS_FIELD);
327                             break;
328                         }
329                     }
330     
331                     // Is this a store to a parameter which was dead on entry to the
332                     // method?
333                     boolean parameterThatIsDeadAtEntry = isParameter
334                             && !llsaDataflow.getAnalysis().isStoreAlive(liveStoreSetAtEntry, local);
335                     if (parameterThatIsDeadAtEntry && !complainedAbout.get(local)) {
336     
337                         int priority = storeLive ? LOW_PRIORITY : NORMAL_PRIORITY;
338                         if (shadowedField != null) {
339                             priority--;
340                         }
341                         pendingBugReportAboutOverwrittenParameter = new BugInstance(this, "IP_PARAMETER_IS_DEAD_BUT_OVERWRITTEN",
342                                 priority).addClassAndMethod(methodGen, sourceFileName).add(lvAnnotation);
343     
344                         if (shadowedField != null) {
345                             pendingBugReportAboutOverwrittenParameter.addField(
346                                     FieldAnnotation.fromBCELField(classContext.getJavaClass(), shadowedField)).describe(
347                                             FieldAnnotation.DID_YOU_MEAN_ROLE);
348                         }
349     
350                         pendingBugReportAboutOverwrittenParameter.addSourceLine(classContext, methodGen, sourceFileName,
351                                 location.getHandle());
352                         complainedAbout.set(local);
353                     }
354     
355                     if (storeLive) {
356                         continue;
357                     }
358     
359                     TypeFrame typeFrame = typeDataflow.getAnalysis().getFactAtLocation(location);
360                     Type typeOfValue = null;
361                     if (typeFrame.isValid() && typeFrame.getStackDepth() > 0) {
362                         typeOfValue = typeFrame.getTopValue();
363                     }
364     
365                     boolean storeOfNull = false;
366                     InstructionHandle prevInsHandle = location.getHandle().getPrev();
367                     if (prevInsHandle != null) {
368                         Instruction prevIns = prevInsHandle.getInstruction();
369                         boolean foundDeadClassInitialization = false;
370                         String initializationOf = null;
371                         if (prevIns instanceof ConstantPushInstruction) {
372                             continue; // not an interesting dead store
373                         } else if (prevIns instanceof GETSTATIC) {
374                             GETSTATIC getStatic = (GETSTATIC) prevIns;
375                             ConstantPoolGen cpg = methodGen.getConstantPool();
376                             foundDeadClassInitialization = getStatic.getFieldName(cpg).startsWith("class$")
377                                     && "Ljava/lang/Class;".equals(getStatic.getSignature(cpg));
378                             for (Iterator<Location> j = cfg.locationIterator(); j.hasNext();) {
379                                 Location location2 = j.next();
380                                 if (location2.getHandle().getPosition() + 15 == location.getHandle().getPosition()) {
381                                     Instruction instruction2 = location2.getHandle().getInstruction();
382                                     if (instruction2 instanceof LDC) {
383                                         Object value = ((LDC) instruction2).getValue(methodGen.getConstantPool());
384                                         if (value instanceof String) {
385                                             String n = (String) value;
386                                             if (n.length() > 0) {
387                                                 initializationOf = ClassName.toSignature(n);
388                                             }
389                                         }
390                                     }
391                                 }
392                             }
393     
394                         } else if (prevIns instanceof LDC) {
395                             LDC ldc = (LDC) prevIns;
396                             Type t = ldc.getType(methodGen.getConstantPool());
397                             if ("Ljava/lang/Class;".equals(t.getSignature())) {
398                                 Object value = ldc.getValue(methodGen.getConstantPool());
399                                 if (value instanceof ConstantClass) {
400                                     ConstantClass v = (ConstantClass) value;
401                                     initializationOf = ClassName.toSignature(v.getBytes(javaClass.getConstantPool()));
402                                     foundDeadClassInitialization = true;
403                                 } else if (value instanceof ObjectType) {
404                                     ObjectType v = (ObjectType) value;
405                                     initializationOf = ClassName.toSignature(v.getClassName());
406                                     foundDeadClassInitialization = true;
407                                 } else {
408                                     AnalysisContext.logError("LDC loaded " + value + "at " + location.getHandle().getPosition() + " in " + classContext.getFullyQualifiedMethodName(method));
409                                 }
410     
411                             }
412                             else {
413                                 continue; // not an interesting DLS
414                             }
415     
416                         } else if (prevIns instanceof DUP2) {
417                             // Check for the case where, due to the bytecode
418                             // compiler, a long is needlessly stored just
419                             // after we've DUP2'ed the stack and just
420                             // before we return
421                             Instruction cur = location.getHandle().getInstruction();
422                             Instruction nxt = location.getHandle().getNext().getInstruction();
423                             if (cur instanceof LSTORE && nxt instanceof LRETURN) {
424                                 continue; // not an interesting DLS
425                             }
426                         }
427                         if (foundDeadClassInitialization) {
428                             if ("org.apache.axis.client.Stub".equals(classContext.getJavaClass().getSuperclassName())) {
429                                 continue;
430                             }
431                             BugInstance bugInstance = new BugInstance(this, "DLS_DEAD_STORE_OF_CLASS_LITERAL",
432                                     Priorities.NORMAL_PRIORITY).addClassAndMethod(methodGen, sourceFileName).add(lvAnnotation)
433                                     .addType(initializationOf);
434                             accumulator.accumulateBug(bugInstance, sourceLineAnnotation);
435                             continue;
436                         }
437     
438                         if (prevIns instanceof LDC || prevIns instanceof ConstantPushInstruction) {
439                             propertySet.addProperty(DeadLocalStoreProperty.STORE_OF_CONSTANT);
440                         } else if (prevIns instanceof ACONST_NULL) {
441                             storeOfNull = true;
442                             propertySet.addProperty(DeadLocalStoreProperty.STORE_OF_NULL);
443                         }
444                     }
445     
446                     if (typeOfValue instanceof BasicType || Type.STRING.equals(typeOfValue)) {
447                         propertySet.addProperty(DeadLocalStoreProperty.BASE_VALUE);
448                     }
449     
450                     // Ignore assignments that were killed by a subsequent
451                     // assignment.
452                     boolean killedBySubsequentStore = llsaDataflow.getAnalysis().killedByStore(liveStoreSet, local);
453                     if (killedBySubsequentStore) {
454                         if (propertySet.containsProperty(DeadLocalStoreProperty.STORE_OF_NULL)
455                                 || propertySet.containsProperty(DeadLocalStoreProperty.STORE_OF_CONSTANT)) {
456                             continue;
457                         }
458                         propertySet.addProperty(DeadLocalStoreProperty.KILLED_BY_SUBSEQUENT_STORE);
459                     }
460     
461                     // Ignore dead assignments of null and 0.
462                     // These often indicate defensive programming.
463                     InstructionHandle prev = location.getBasicBlock().getPredecessorOf(location.getHandle());
464                     //                int prevOpCode = -1;
465     
466                     if (prev != null) {
467                         if (defensiveConstantValueOpcodes.get(prev.getInstruction().getOpcode())) {
468                             propertySet.addProperty(DeadLocalStoreProperty.DEFENSIVE_CONSTANT_OPCODE);
469                             //                        prevOpCode = prev.getInstruction().getOpcode();
470                         }
471     
472                         if (prev.getInstruction() instanceof GETFIELD) {
473                             InstructionHandle prev2 = prev.getPrev();
474     
475                             if (prev2 != null && prev2.getInstruction() instanceof ALOAD) {
476                                 propertySet.addProperty(DeadLocalStoreProperty.CACHING_VALUE);
477                             }
478                         }
479                         if (prev.getInstruction() instanceof LoadInstruction) {
480                             propertySet.addProperty(DeadLocalStoreProperty.COPY_VALUE);
481                         }
482                         if (prev.getInstruction() instanceof InvokeInstruction) {
483                             propertySet.addProperty(DeadLocalStoreProperty.METHOD_RESULT);
484                         }
485                     }
486                     boolean deadObjectStore = false;
487                     if (ins instanceof IINC) {
488                         // special handling of IINC
489     
490                         if ("main".equals(method.getName()) && method.isStatic()
491                                 && "([Ljava/lang/String;)V".equals(method.getSignature())) {
492                             propertySet.addProperty(DeadLocalStoreProperty.DEAD_INCREMENT_IN_MAIN);
493                         }
494     
495                         InstructionHandle next = location.getHandle().getNext();
496                         if (next != null && next.getInstruction() instanceof IRETURN) {
497                             propertySet.addProperty(DeadLocalStoreProperty.DEAD_INCREMENT_IN_RETURN);
498                         } else {
499                             propertySet.addProperty(DeadLocalStoreProperty.DEAD_INCREMENT);
500                         }
501                         if (localIncrementCount[local] == 1) {
502                             propertySet.addProperty(DeadLocalStoreProperty.SINGLE_DEAD_INCREMENT);
503                         } else {
504                             propertySet.removeProperty(DeadLocalStoreProperty.IS_PARAMETER);
505                         }
506     
507                     } else if (ins instanceof ASTORE && prev != null) {
508                         // Look for objects created but never used
509     
510                         Instruction prevIns = prev.getInstruction();
511                         if ((prevIns instanceof INVOKESPECIAL && "<init>".equals(((INVOKESPECIAL) prevIns).getMethodName(methodGen.getConstantPool())))
512                                 || prevIns instanceof ANEWARRAY
513                                 || prevIns instanceof NEWARRAY
514                                 || prevIns instanceof MULTIANEWARRAY) {
515                             deadObjectStore = true;
516                         } else if (prevIns instanceof DUP) {
517                             propertySet.addProperty(DeadLocalStoreProperty.DUP_THEN_STORE);
518                         }
519                     }
520                     if (deadObjectStore) {
521                         propertySet.addProperty(DeadLocalStoreProperty.DEAD_OBJECT_STORE);
522                     } else if (!killedBySubsequentStore && localStoreCount[local] == 2 && localLoadCount[local] > 0) {
523                         // TODO: why is this significant?
524     
525                         propertySet.addProperty(DeadLocalStoreProperty.TWO_STORES_MULTIPLE_LOADS);
526     
527                     } else if (!parameterThatIsDeadAtEntry && localStoreCount[local] == 1 && localLoadCount[local] == 0
528                             && propertySet.containsProperty(DeadLocalStoreProperty.DEFENSIVE_CONSTANT_OPCODE)) {
529                         // might be final local constant
530                         propertySet.addProperty(DeadLocalStoreProperty.SINGLE_STORE);
531     
532                     } else if (!parameterThatIsDeadAtEntry && !propertySet.containsProperty(DeadLocalStoreProperty.SHADOWS_FIELD)
533                             && localLoadCount[local] == 0) {
534                         // TODO: why is this significant?
535                         propertySet.addProperty(DeadLocalStoreProperty.NO_LOADS);
536                     }
537                     if (!storeOfNull && typeOfValue != null
538                             && !propertySet.containsProperty(DeadLocalStoreProperty.EXCEPTION_HANDLER)) {
539                         String signatureOfValue = typeOfValue.getSignature();
540                         if ((signatureOfValue.startsWith("Ljava/sql/") || signatureOfValue.startsWith("Ljavax/sql/"))
541                                 && !signatureOfValue.endsWith("Exception")) {
542                             propertySet.addProperty(DeadLocalStoreProperty.STORE_OF_DATABASE_VALUE);
543                         }
544                     }
545     
546                     if (parameterThatIsDeadAtEntry) {
547                         propertySet.addProperty(DeadLocalStoreProperty.PARAM_DEAD_ON_ENTRY);
548                         if (pendingBugReportAboutOverwrittenParameter != null) {
549                             pendingBugReportAboutOverwrittenParameter.setPriority(Priorities.HIGH_PRIORITY);
550                         }
551                     }
552     
553                     if (localStoreCount[local] > 3) {
554                         propertySet.addProperty(DeadLocalStoreProperty.MANY_STORES);
555                     }
556                     int occurrences = cfg.getLocationsContainingInstructionWithOffset(pc).size();
557                     if (occurrences > 2 || sourceLineAnnotation.getStartLine() > 0
558                             && linesMentionedMultipleTimes.get(sourceLineAnnotation.getStartLine())) {
559                         propertySet.addProperty(DeadLocalStoreProperty.CLONED_STORE);
560                     }
561                     String sourceFile = javaClass.getSourceFileName();
562                     if (Subtypes2.isJSP(javaClass)) {
563                         propertySet.addProperty(DeadLocalStoreProperty.IN_JSP_PAGE);
564                     } else if (BCELUtil.isSynthetic(javaClass) || sourceFile != null && !sourceFile.endsWith(".java")) {
565                         if (sourceFile != null && sourceFile.endsWith(".gxp") && (lvName.startsWith("gxp$") || lvName.startsWith("gxp_"))) {
566                             continue;
567                         }
568                         propertySet.addProperty(DeadLocalStoreProperty.NOT_JAVA);
569                     }
570     
571                     // Report the warning
572                     String bugPattern;
573                     if (storeOfNull) {
574                         bugPattern = "DLS_DEAD_LOCAL_STORE_OF_NULL";
575                     } else if (shadowedField != null) {
576                         bugPattern = "DLS_DEAD_LOCAL_STORE_SHADOWS_FIELD";
577                     } else if (propertySet.containsProperty(DeadLocalStoreProperty.DEAD_INCREMENT_IN_RETURN)) {
578                         bugPattern = "DLS_DEAD_LOCAL_INCREMENT_IN_RETURN";
579                     } else {
580                         bugPattern = "DLS_DEAD_LOCAL_STORE";
581                     }
582                     BugInstance bugInstance = new BugInstance(this, bugPattern, NORMAL_PRIORITY).addClassAndMethod(methodGen,
583                             sourceFileName).add(lvAnnotation);
584     
585                     if (shadowedField != null) {
586                         bugInstance.addField(FieldAnnotation.fromBCELField(classContext.getJavaClass(), shadowedField)).describe(
587                                 FieldAnnotation.DID_YOU_MEAN_ROLE);
588                     }
589     
590                     // If in relaxed reporting mode, encode heuristic
591                     // information.
592                     if (FindBugsAnalysisFeatures.isRelaxedMode()) {
593                         // Add general-purpose warning properties
594                         WarningPropertyUtil.addPropertiesForDataMining(propertySet, classContext, method, location);
595                     }
596                     // Turn all warning properties into BugProperties
597                     propertySet.decorateBugInstance(bugInstance);
598                     if (DEBUG) {
599                         System.out.println(sourceFileName + " : " + methodGen.getName());
600                         System.out.println("priority: " + bugInstance.getPriority());
601                         System.out.println("Reporting " + bugInstance);
602                         System.out.println(propertySet);
603                     }
604                     accumulator.accumulateBug(bugInstance, sourceLineAnnotation);
605     
606                 } finally {
607                     if (pendingBugReportAboutOverwrittenParameter != null) {
608                         bugReporter.reportBug(pendingBugReportAboutOverwrittenParameter);
609                     }
610                 }
611             }
612     
613             suppressWarningsIfOneLiveStoreOnLine(accumulator, liveStoreSourceLineSet);
614     
615             accumulator.reportAccumulatedBugs();
616         }

618         /**
619          * If feature is enabled, suppress warnings where there is at least one live
620          * store on the line where the warning would be reported.
621          *
622          * @param accumulator
623          *            BugAccumulator containing warnings for method
624          * @param liveStoreSourceLineSet
625          *            bitset of lines where at least one live store was seen
626          */
627         private void suppressWarningsIfOneLiveStoreOnLine(BugAccumulator accumulator, BitSet liveStoreSourceLineSet) {
628             if (!SUPPRESS_IF_AT_LEAST_ONE_LIVE_STORE_ON_LINE) {
629                 return;
630             }
631     
632             // Eliminate any accumulated warnings for instructions
633             // that (due to inlining) *can* be live stores.
634             entryLoop: for (Iterator<? extends BugInstance> i = accumulator.uniqueBugs().iterator(); i.hasNext();) {
635     
636                 for (SourceLineAnnotation annotation : accumulator.locations(i.next())) {
637                     if (liveStoreSourceLineSet.get(annotation.getStartLine())) {
638                         // This instruction can be a live store; don't report
639                         // it as a warning.
640                         i.remove();
641                         continue entryLoop;
642                     }
643                 }
644             }
645         }

650          *
651          * @param localStoreCount
652          *            counts of local stores (indexed by local)
653          * @param localLoadCount
654          *            counts of local loads (indexed by local)
655          * @param localIncrementCount
656          *            counts of local increments (indexed by local)
657          * @param cfg
658          *            control flow graph (CFG) of method
659          */
660         private void countLocalStoresLoadsAndIncrements(int[] localStoreCount, int[] localLoadCount, int[] localIncrementCount,
661                 CFG cfg) {
662             for (Iterator<Location> i = cfg.locationIterator(); i.hasNext();) {
663                 Location location = i.next();
664     
665                 if (location.getBasicBlock().isExceptionHandler()) {
666                     continue;
667                 }
668     
669                 boolean isStore = isStore(location);
670                 boolean isLoad = isLoad(location);
671                 if (!isStore && !isLoad) {
672                     continue;
673                 }
674     
675                 IndexedInstruction ins = (IndexedInstruction) location.getHandle().getInstruction();
676                 int local = ins.getIndex();
677                 if (ins instanceof IINC) {
678                     localStoreCount[local]++;
679                     localLoadCount[local]++;
680                     localIncrementCount[local]++;
681                 } else if (isStore) {
682                     localStoreCount[local]++;
683                 } else {
684                     localLoadCount[local]++;
685                 }
686             }
687         }
