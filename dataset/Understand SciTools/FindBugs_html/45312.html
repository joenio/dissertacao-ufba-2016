44     /*
45      * Our TreeModel.  Once upon a time it was a simple model, that queried data, its BugSet, for what to show under each branch
46      * Then it got more and more complicated, no one knows why it still seems to work... or why it doesn't if it in fact doesn't.
47      *
48      * Here's a tip, Dont even attempt to deal with suppressions or filtering and their related tree model events without the API
49      * for TreeModelEvents open.  And read it three times first.  Ignore the fact that its inconsistent for sending events about the root, just pick one of the things it says and go with it
50      *
51      * Heres the order things MUST be done when dealing with suppressions, filtering, unsuppressions... unfiltering... all that fun stuff
52      *
53      * Inserts:
54      * Update model
55      * Get Path
56      * Make Event
57      * Send Event
58      * ResetData
59      *
60      * Removes:
61      * Get Path
62      * Make Event
63      * Update Model
64      * Send Event
65      * ResetData
66      *
67      * Restructure:
68      * Update Model
69      * Get Path
70      * Make Event
71      * Send Event
72      * resetData? hmmm
73      *
74      * These may or may not be the orders of events used in suppressBug, unsuppressBug, branchOperations and so forth
75      * if they seem to work anyway, I wouldn't touch them.
76      *
77      * changeSet() is what to do when the data set is completely different (loaded a new collection, reran the analysis what have you)
78      * changeSet calls rebuild(), which does a very tricky thing, where it makes a new model, and a new JTree, and swaps them in in place of this one, as well as
79      * turning off user input in hopefully every place it needs to be turned off
80      *
81      */

174             int childCount = getChildCount(o);
175             if (index < 0 || index >= childCount) {
176                 if (SystemProperties.ASSERTIONS_ENABLED) {
177                     System.out.printf("Unable to get child %d of %d from %s:%s%n" , index, childCount, o.getClass().getSimpleName(), o);

186         private @Nonnull Object getChild(BugAspects a, int index) {
187     
188             int treeLevels = st.getOrderBeforeDivider().size();
189             int queryDepth = a.size();
190             assert queryDepth <= treeLevels;
191     
192             if (treeLevels == 0 && a.size() == 0) {
193                 BugLeafNode bugLeafNode = bugSet.get(index);
194                 assert bugLeafNode != null;
195                 return bugLeafNode;
196             }
197             if (SystemProperties.ASSERTIONS_ENABLED) {
198                 for (int i = 0; i < queryDepth; i++) {
199                     Sortables treeSortable = st.getOrderBeforeDivider().get(i);
200                     Sortables querySortable = a.get(i).key;
201                     assert treeSortable.equals(querySortable) : treeSortable + " vs " + querySortable;
202                 }
203             }
204     
205             if (queryDepth < treeLevels) {
206                 BugAspects child = a.addToNew(enumsThatExist(a).get(index));
207                 child.setCount(bugSet.query(child).size());
208                 return child;
209             } else {
210                 BugLeafNode bugLeafNode = bugSet.query(a).get(index);
211                 assert bugLeafNode != null;
212                 return bugLeafNode;
213             }
214         }

216         @Override
217         public int getChildCount(Object o) {
218     
219             if (!(o instanceof BugAspects)) {
220                 return 0;
221             }
222     
223             BugAspects a = (BugAspects) o;
224     
225             if (st.getOrderBeforeDivider().size() == 0 && a.size() == 0) {
226                 // the root
227                 // and we
228                 // aren't
229                 // sorting
230                 // by
231                 // anything
232                 return bugSet.size();
233             }
234     
235             if ((a.size() == 0) || (a.last().key != st.getOrderBeforeDivider().get(st.getOrderBeforeDivider().size() - 1))) {
236                 return enumsThatExist(a).size();
237             } else {
238                 return bugSet.query(a).size();
239             }
240         }

243          * This contract has been changed to return a HashList of Stringpair, our
244          * own data structure in which finding the index of an object in the list is

248         private @Nonnull List<SortableValue> enumsThatExist(BugAspects a) {
249             List<Sortables> orderBeforeDivider = st.getOrderBeforeDivider();
250             if (orderBeforeDivider.size() == 0) {
251                 List<SortableValue> result = Collections.emptyList();
252                 assert false;
253                 return result;
254             }
255     
256             Sortables key;
257             if (a.size() == 0) {
258                 key = orderBeforeDivider.get(0);
259             } else {
260                 Sortables lastKey = a.last().key;
261                 int index = orderBeforeDivider.indexOf(lastKey);
262                 if (index + 1 < orderBeforeDivider.size()) {
263                     key = orderBeforeDivider.get(index + 1);
264                 } else {
265                     key = lastKey;
266                 }
267             }
268     
269             String[] all = key.getAll(bugSet.query(a));
270             ArrayList<SortableValue> result = new ArrayList<SortableValue>(all.length);
271             for (String i : all) {
272                 result.add(new SortableValue(key, i));
273             }
274             return result;
275     
276         }

348         public void rebuild() {
349             if (TRACE) {
350                 System.out.println("rebuilding bug tree model");
351             }
352     
353             NewFilterFromBug.closeAll();
354     
355             // If this thread is not interrupting a previous thread, set the paths
356             // to be opened when the new tree is complete
357             // If the thread is interrupting another thread, dont do this, because
358             // you dont have the tree with the correct paths selected
359     
360             // As of now, it should be impossible to interrupt a rebuilding thread,
361             // in another version this may change, so this if statement check is
362             // left in, even though it should always be true.
363             if (rebuildingThread == null) {
364                 setOldSelectedBugs();
365             }
366     
367             Debug.println("Please Wait called right before starting rebuild thread");
368             mainFrame.acquireDisplayWait();
369             rebuildingThread =  edu.umd.cs.findbugs.util.Util.runInDameonThread(new Runnable() {
370                 BugTreeModel newModel;
371     
372                 @Override
373                 public void run() {
374                     try {
375                         newModel = new BugTreeModel(BugTreeModel.this);
376                         newModel.listeners = listeners;
377                         newModel.resetData();
378                         newModel.bugSet.sortList();
379                     } finally {
380                         rebuildingThread = null;
381                         SwingUtilities.invokeLater(new Runnable() {
382                             @Override
383                             public void run() {
384                                 if (newModel != null) {
385                                     JTree newTree = new JTree(newModel);
386                                     newModel.tree = newTree;
387                                     mainFrame.mainFrameTree.newTree(newTree, newModel);
388                                     mainFrame.releaseDisplayWait();
389                                 }
390                                 getOffListenerList();
391                             }
392                         });
393                     }
394                 }
395             }, "Rebuilding thread");
396     
397         }

399         public void crawl(final ArrayList<BugAspects> path, final int depth) {
400             for (int i = 0; i < getChildCount(path.get(path.size() - 1)); i++) {
401                 if (depth > 0) {
402                     ArrayList<BugAspects> newPath = new ArrayList<BugAspects>(path);
403                     newPath.add((BugAspects) getChild(path.get(path.size() - 1), i));
404                     crawl(newPath, depth - 1);
405                 } else {
406                     for (TreeModelListener l : listeners) {
407                         l.treeStructureChanged(new TreeModelEvent(this, path.toArray()));
408                     }
409                 }
410             }
411         }

434          * Recursively traverses the tree, opens all nodes matching any bug in the
435          * list, then creates the full paths to the bugs that are selected This
436          * keeps whatever bugs were selected selected when sorting DEPRECATED--Too
437          * Slow, use openPreviouslySelected

440         public void crawlToOpen(TreePath path, ArrayList<BugLeafNode> bugLeafNodes, ArrayList<TreePath> treePaths) {
441             for (int i = 0; i < getChildCount(path.getLastPathComponent()); i++) {
442                 if (!isLeaf(getChild(path.getLastPathComponent(), i))) {
443                     for (BugLeafNode p : bugLeafNodes) {
444                         if (p.matches((BugAspects) getChild(path.getLastPathComponent(), i))) {
445                             tree.expandPath(path);
446                             crawlToOpen(path.pathByAddingChild(getChild(path.getLastPathComponent(), i)), bugLeafNodes, treePaths);
447                             break;
448                         }
449                     }
450                 } else {
451                     for (BugLeafNode b : bugLeafNodes) {
452                         if (getChild(path.getLastPathComponent(), i).equals(b)) {
453                             tree.expandPath(path);
454                             treePaths.add(path.pathByAddingChild(getChild(path.getLastPathComponent(), i)));
455                         }
456                     }
457                 }
458             }
459         }

509         public TreePath getPathToBug(BugInstance b) {
510             // ArrayList<Sortables>
511             // order=MainFrame.getInstance().getSorter().getOrder();
512             List<Sortables> order = st.getOrderBeforeDivider();
513             // Create an array of BugAspects of lengths from one to the full
514             // BugAspect list of the bugInstance
515             BugAspects[] toBug = new BugAspects[order.size()];
516             for (int i = 0; i < order.size(); i++) {
517                 toBug[i] = new BugAspects();
518             }
519     
520             for (int x = 0; x < order.size(); x++) {
521                 for (int y = 0; y <= x; y++) {
522                     Sortables s = order.get(y);
523                     toBug[x].add(new SortableValue(s, s.getFrom(b)));
524                 }
525             }
526             // Add this array as elements of the path
527             TreePath pathToBug = new TreePath(root);
528             for (int x = 0; x < order.size(); x++) {
529                 int index = getIndexOfChild(pathToBug.getLastPathComponent(), toBug[x]);
530     
531                 if (index == -1) {
532                     if (MainFrame.GUI2_DEBUG)
533                     {
534                         System.err.println("Node does not exist in the tree");// For
535                     }
536                     // example,
537                     // not
538                     // a
539                     // bug
540                     // bugs
541                     // are
542                     // filtered,
543                     // they
544                     // set
545                     // a
546                     // bug
547                     // to
548                     // be
549                     // not
550                     // a
551                     // bug
552                     // it
553                     // filters
554                     // out
555                     return null;
556                 }
557     
558                 pathToBug = pathToBug.pathByAddingChild(getChild(pathToBug.getLastPathComponent(), index));
559             }
560             // Using a hashlist to store bugs in BugSet will make getIndexOfChild
561             // Waaaaaay faster, thus making this O(1) (avg case)
562             int index = getIndexOfChild(pathToBug.getLastPathComponent(), new BugLeafNode(b));
563             if (index == -1) {
564                 return null;
565             }
566             pathToBug = pathToBug.pathByAddingChild(getChild(pathToBug.getLastPathComponent(), index));
567             return pathToBug;
568     
569         }

674         private TreeModelEvent branchOperations(ArrayList<String> stringsToBranch, TreeModification whatToDo)
675                 throws BranchOperationException {
676             TreeModelEvent event = null;
677     
678             if (whatToDo == TreeModification.REMOVE) {
679                 Debug.println("Removing a branch......");
680             } else if (whatToDo == TreeModification.INSERT) {
681                 Debug.println("Inserting a branch......");
682             } else if (whatToDo == TreeModification.REMOVERESTRUCTURE) {
683                 Debug.println("Restructuring from branch to remove......");
684             } else if (whatToDo == TreeModification.INSERTRESTRUCTURE) {
685                 Debug.println("Restructuring from branch to insert......");
686             }
687             Debug.println(stringsToBranch);
688     
689             if (whatToDo == TreeModification.INSERT || whatToDo == TreeModification.INSERTRESTRUCTURE) {
690                 resetData();
691             }
692             // ArrayList<Sortables>
693             // order=MainFrame.getInstance().getSorter().getOrder();
694             List<Sortables> order = st.getOrderBeforeDivider();
695             // Create an array of BugAspects of lengths from one to the full
696             // BugAspect list of the bugInstance
697             BugAspects[] toBug = new BugAspects[stringsToBranch.size()];
698             for (int x = 0; x < stringsToBranch.size(); x++) {
699                 toBug[x] = new BugAspects();
700     
701                 for (int y = 0; y <= x; y++) {
702                     Sortables s = order.get(y);
703                     toBug[x].add(new SortableValue(s, stringsToBranch.get(y)));
704                 }
705             }
706     
707             // Add this array as elements of the path
708             TreePath pathToBranch = new TreePath(root);
709             for (int x = 0; x < stringsToBranch.size(); x++) {
710                 BugAspects child = toBug[x];
711                 BugAspects parent = (BugAspects) pathToBranch.getLastPathComponent();
712                 if (getIndexOfChild(parent, child) != -1) {
713                     pathToBranch = pathToBranch.pathByAddingChild(child);
714                 } else {
715                     Debug.println(parent + " does not contain " + child);
716                     throw new BranchOperationException("Branch has been filtered out by another filter.");
717                     // break;
718                 }
719             }
720             if (pathToBranch.getParentPath() != null) {
721                 while (getChildCount(pathToBranch.getParentPath().getLastPathComponent()) == 1) {
722                     if (pathToBranch.getParentPath().getLastPathComponent().equals(root)) {
723                         break;
724                     }
725                     pathToBranch = pathToBranch.getParentPath();
726                 }
727             }
728             Debug.println(pathToBranch);
729     
730             if (whatToDo == TreeModification.INSERT) {
731                 event = new TreeModelEvent(this, pathToBranch.getParentPath(), new int[] { getIndexOfChild(pathToBranch
732                         .getParentPath().getLastPathComponent(), pathToBranch.getLastPathComponent()) },
733                         new Object[] { pathToBranch.getLastPathComponent() });
734             } else if (whatToDo == TreeModification.INSERTRESTRUCTURE) {
735                 event = new TreeModelEvent(this, pathToBranch);
736             }
737     
738             if (whatToDo == TreeModification.REMOVE) {
739                 event = new TreeModelEvent(this, pathToBranch.getParentPath(), new int[] { getIndexOfChild(pathToBranch
740                         .getParentPath().getLastPathComponent(), pathToBranch.getLastPathComponent()) },
741                         new Object[] { pathToBranch.getLastPathComponent() });
742     
743             } else if (whatToDo == TreeModification.REMOVERESTRUCTURE) {
744                 event = new TreeModelEvent(this, pathToBranch);
745             }
746     
747             if (whatToDo == TreeModification.REMOVE || whatToDo == TreeModification.REMOVERESTRUCTURE) {
748                 resetData();
749             }
750     
751             return event;
752         }

754         void sendEvent(TreeModelEvent event, TreeModification whatToDo) {
755             Debug.println("Sending An Event!");
756             if (event == null) {
757                 throw new IllegalStateException("Dont throw null events.");
758             }
759             resetData();
760             for (TreeModelListener l : listeners) {
761                 if (whatToDo == TreeModification.REMOVE) {
762                     l.treeNodesRemoved(event);
763                 } else if (whatToDo == TreeModification.INSERT) {
764                     l.treeNodesInserted(event);
765                     l.treeStructureChanged(new TreeModelEvent(this, new TreePath(event.getPath()).pathByAddingChild(event
766                             .getChildren()[0])));
767                 } else if (whatToDo == TreeModification.INSERTRESTRUCTURE || whatToDo == TreeModification.REMOVERESTRUCTURE) {
768                     l.treeStructureChanged(event);
769                 }
770             }
771     
772             root.setCount(bugSet.size());
773             TreePath changedPath = new TreePath(root);
774             treeNodeChanged(changedPath);
775             changedPath = new TreePath(event.getPath());
776             while (changedPath.getParentPath() != null) {
777                 treeNodeChanged(changedPath);
778                 changedPath = changedPath.getParentPath();
779             }
780         }
