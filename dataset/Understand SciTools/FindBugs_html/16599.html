34         final static String HASH_SEED = SystemProperties.getProperty("hashSeed", "");
35     
36         public static String hashData(String in) {
37             MessageDigest md;
38             try {
39                 md = MessageDigest.getInstance("SHA-1");
40                 byte[] hash = md.digest((HASH_SEED + in).getBytes("UTF-8"));
41                 return String.format("%040x", new BigInteger(1, hash));
42             } catch (RuntimeException e) {
43                 throw e;
44             } catch (Exception e) {
45                 throw new RuntimeException(e);
46             }
47         }
48     
49         public static String hashFilename(String fileName) {
50             int lastDot = fileName.lastIndexOf('.');
51             if (lastDot >= 0) {
52                 String suffix = fileName.substring(lastDot);
53                 return hashData(fileName.substring(0, lastDot)) + suffix;
54             }
55             return hashData(fileName);
56         }
57     
58         public static String hashClass(@DottedClassName String className) {
59             if (className.startsWith("java")) {
60                 return className;
61             }
62             return "obfuscated.H" + hashData(className);
63         }
64     
65         public static String hashSignature(String signature) {
66             char firstChar = signature.charAt(0);
67             switch (firstChar) {
68             case '[':
69                 return '[' + hashSignature(signature.substring(1));
70             case 'V':
71             case 'Z':
72     
73             case 'B':
74             case 'S':
75             case 'C':
76             case 'I':
77             case 'J':
78             case 'D':
79             case 'F':
80                 if (signature.length() == 1) {
81                     return signature;
82                 }
83                 throw new IllegalArgumentException("bad signature: " + signature);
84             case 'L':
85                 if (!signature.endsWith(";")) {
86                     throw new IllegalArgumentException("bad signature: " + signature);
87                 }
88                 return hashFieldSignature(signature);
89             default:
90                 throw new IllegalArgumentException("bad signature: " + signature);
91             }
92         }
93     
94         public static String hashFieldSignature(String signature) {
95             signature = signature.substring(1, signature.length() - 1);
96             if (!signature.startsWith("java")) {
97                 signature = "obfuscated/H" + hashData(signature);
98             }
99             return "L" + signature + ";";
100         }
101     
102         public static String hashMethodSignature(String signature) {
103             SignatureParser parser = new SignatureParser(signature);
104             StringBuilder buf = new StringBuilder("(");
105             for (Iterator<String> i = parser.parameterSignatureIterator(); i.hasNext();) {
106                 String param = i.next();
107                 buf.append(hashSignature(param));
108             }
109             buf.append(")");
110             buf.append(hashSignature(parser.getReturnTypeSignature()));
111             return buf.toString();
112     
113         }

115         static MethodAnnotation obfuscate(MethodAnnotation m) {
116             String className = m.getClassName();
117             if (className.startsWith("java")) {
118                 return m;
119             }
120     
121             String methodName = m.getMethodName();
122             String methodSignature = m.getMethodSignature();
123     
124             if ("hashCode".equals(methodName) && "()I".equals(methodSignature) || "equals".equals(methodName)
125                     && "(Ljava/lang/Object;)Z".equals(methodSignature) || "compareTo".equals(methodName)
126                     && "(Ljava/lang/Object;)I".equals(methodSignature) || "<init>".equals(methodName)
127                     || "<clinit>".equals(methodName)) {
128                 // don't need to obfuscate method name
129             } else {
130                 methodName = hashData(methodName);
131             }
132     
133             MethodAnnotation result = new MethodAnnotation(hashClass(className), methodName, hashMethodSignature(methodSignature),
134                     m.isStatic());
135             result.setDescription(m.getDescription());
136             return result;
137         }

139         static FieldAnnotation obfuscate(FieldAnnotation m) {
140             FieldAnnotation result = new FieldAnnotation(hashClass(m.getClassName()), hashData(m.getFieldName()),
141                     hashSignature(m.getFieldSignature()), m.isStatic());
142             result.setDescription(m.getDescription());
143             return result;
144     
145         }

175         static SourceLineAnnotation obfuscate(SourceLineAnnotation m) {
176             SourceLineAnnotation result = new SourceLineAnnotation(hashClass(m.getClassName()), hashFilename(m.getSourceFile()),
177                     m.getStartLine(), m.getEndLine(), m.getStartBytecode(), m.getEndBytecode());
178             result.setDescription(m.getDescription());
179             return result;

184     
185             LocalVariableAnnotation result = new LocalVariableAnnotation(hashData(m.getName()), m.getRegister(), m.getPC());
186             result.setDescription(m.getDescription());
187             return result;

191         public static BugInstance obfuscate(BugInstance b) {
192             final BugInstance result = new BugInstance(b.getType(), b.getPriority());
193             BugAnnotationVisitor visitor = new BugAnnotationVisitor() {
194     
195                 @Override
196                 public void visitTypeAnnotation(TypeAnnotation typeAnnotation) {
197                     result.add(obfuscate(typeAnnotation));
198     
199                 }
200     
201                 @Override
202                 public void visitStringAnnotation(StringAnnotation stringAnnotation) {
203                     result.add(obfuscate(stringAnnotation));
204     
205                 }
206     
207                 @Override
208                 public void visitSourceLineAnnotation(SourceLineAnnotation sourceLineAnnotation) {
209                     result.add(obfuscate(sourceLineAnnotation));
210     
211                 }
212     
213                 @Override
214                 public void visitMethodAnnotation(MethodAnnotation methodAnnotation) {
215                     result.add(obfuscate(methodAnnotation));
216     
217                 }
218     
219                 @Override
220                 public void visitLocalVariableAnnotation(LocalVariableAnnotation fieldAnnotation) {
221                     result.add(obfuscate(fieldAnnotation));
222     
223                 }
224     
225                 @Override
226                 public void visitIntAnnotation(IntAnnotation fieldAnnotation) {
227                     result.add(obfuscate(fieldAnnotation));
228     
229                 }
230     
231                 @Override
232                 public void visitFieldAnnotation(FieldAnnotation fieldAnnotation) {
233                     result.add(obfuscate(fieldAnnotation));
234     
235                 }
236     
237                 @Override
238                 public void visitClassAnnotation(ClassAnnotation classAnnotation) {
239                     result.add(obfuscate(classAnnotation));
240     
241                 }
242             };
243             for (BugAnnotation a : b.getAnnotations()) {
244                 a.accept(visitor);
245             }
246             result.setOldInstanceHash(hashData(b.getInstanceHash()));
247             result.setHistory(b);
248             return result;
249         }
