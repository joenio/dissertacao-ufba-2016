46         private void compute() {
47             if (defined == null) {
48                 // System.out.println("Computing");
49                 defined = new HashSet<String>();
50     
51                 Subtypes2 subtypes2 = AnalysisContext.currentAnalysisContext().getSubtypes2();
52                 Collection<XClass> allClasses = subtypes2.getXClassCollection();
53     
54                 IAnalysisCache analysisCache = Global.getAnalysisCache();
55     
56                 for (XClass c : allClasses) {
57                     try {
58                         JavaClass jclass = analysisCache.getClassAnalysis(JavaClass.class, c.getClassDescriptor());
59                         addAllDefinitions(jclass);
60                     } catch (MissingClassException e) {
61                         bugReporter.reportMissingClass(e.getClassDescriptor());
62                     } catch (CheckedAnalysisException e) {
63                         bugReporter.logError("Could not find class " + c.getClassDescriptor().toDottedClassName(), e);
64                     }
65                 }
66                 // System.out.println("Done Computing: " +
67                 // defined.contains("edu.umd.cs.findbugs.ba.IsNullValueAnalysis.UNKNOWN_VALUES_ARE_NSP : Z"));
68             }
69         }

81         public void addAllDefinitions(JavaClass obj) {
82             String className2 = obj.getClassName();
83     
84             defined.add(className2);
85             for (Method m : obj.getMethods()) {
86                 if (!m.isPrivate()) {
87                     String name = getMemberName(obj, className2, m.getNameIndex(), m.getSignatureIndex());
88                     defined.add(name);
89                 }
90             }
91             for (Field f : obj.getFields()) {
92                 if (!f.isPrivate()) {
93                     String name = getMemberName(obj, className2, f.getNameIndex(), f.getSignatureIndex());
94                     defined.add(name);
95                 }
96             }
97         }

99         private String getClassName(JavaClass c, int classIndex) {
100             String name = c.getConstantPool().getConstantString(classIndex, CONSTANT_Class);
101             return ClassName.extractClassName(name).replace('/', '.');
102         }
103     
104         private String getMemberName(JavaClass c, String className, int memberNameIndex, int signatureIndex) {
105             return className + "." + ((ConstantUtf8) c.getConstantPool().getConstant(memberNameIndex, CONSTANT_Utf8)).getBytes()
106                     + " : " + ((ConstantUtf8) c.getConstantPool().getConstant(signatureIndex, CONSTANT_Utf8)).getBytes();
107         }
108     
109         private String getMemberName(String className, String memberName, String signature) {
110             return className.replace('/', '.') + "." + memberName + " : " + signature;
111         }
112     
113         private boolean find(JavaClass target, String name, String signature) throws ClassNotFoundException {
114             if (target == null) {
115                 return false;
116             }
117             String ref = getMemberName(target.getClassName(), name, signature);
118             if (defined.contains(ref)) {
119                 return true;
120             }
121             if (find(target.getSuperClass(), name, signature)) {
122                 return true;
123             }
124             for (JavaClass i : target.getInterfaces()) {
125                 if (find(i, name, signature)) {
126                     return true;
127                 }
128             }
129             return false;
130         }

132         @Override
133         public void visit(JavaClass obj) {
134             compute();
135             ConstantPool cp = obj.getConstantPool();
136             Constant[] constants = cp.getConstantPool();
137             checkConstant: for (int i = 0; i < constants.length; i++) {
138                 Constant co = constants[i];
139                 if (co instanceof ConstantDouble || co instanceof ConstantLong) {
140                     i++;
141                 }
142                 if (co instanceof ConstantClass) {
143                     String ref = getClassName(obj, i);
144                     if ((ref.startsWith("java") || ref.startsWith("org.w3c.dom")) && !defined.contains(ref)) {
145                         bugReporter.reportBug(new BugInstance(this, "VR_UNRESOLVABLE_REFERENCE", NORMAL_PRIORITY).addClass(obj)
146                                 .addString(ref));
147                     }
148     
149                 } else if (co instanceof ConstantFieldref) {
150                     // do nothing until we handle static fields defined in
151                     // interfaces
152                 } else if (co instanceof ConstantCP) {
153                     ConstantCP co2 = (ConstantCP) co;
154                     String className = getClassName(obj, co2.getClassIndex());
155     
156                     // System.out.println("checking " + ref);
157                     if (className.equals(obj.getClassName()) || !defined.contains(className)) {
158                         // System.out.println("Skipping check of " + ref);
159                         continue checkConstant;
160                     }
161                     ConstantNameAndType nt = (ConstantNameAndType) cp.getConstant(co2.getNameAndTypeIndex());
162                     String name = ((ConstantUtf8) obj.getConstantPool().getConstant(nt.getNameIndex(), CONSTANT_Utf8)).getBytes();
163                     String signature = ((ConstantUtf8) obj.getConstantPool().getConstant(nt.getSignatureIndex(), CONSTANT_Utf8))
164                             .getBytes();
165     
166                     try {
167                         JavaClass target = Repository.lookupClass(className);
168                         if (!find(target, name, signature)) {
169                             bugReporter.reportBug(new BugInstance(this, "VR_UNRESOLVABLE_REFERENCE", NORMAL_PRIORITY).addClass(obj)
170                                     .addString(getMemberName(target.getClassName(), name, signature)));
171                         }
172     
173                     } catch (ClassNotFoundException e) {
174                         bugReporter.reportMissingClass(e);
175                     }
176                 }
177     
178             }
179         }
