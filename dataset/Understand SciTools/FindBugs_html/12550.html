20     package edu.umd.cs.findbugs.filter;
21     
22     import java.io.IOException;
23     
24     import edu.umd.cs.findbugs.BugAnnotation;
25     import edu.umd.cs.findbugs.BugInstance;
26     import edu.umd.cs.findbugs.MethodAnnotation;
27     import edu.umd.cs.findbugs.ba.SignatureConverter;
28     import edu.umd.cs.findbugs.xml.XMLAttributeList;
29     import edu.umd.cs.findbugs.xml.XMLOutput;
30     
31     public class MethodMatcher extends MemberMatcher implements Matcher {
32     
33         public MethodMatcher(String name) {
34             super(name);
35         }
36     
37         public MethodMatcher(String name, String role) {
38             super(name, null, role);
39         }
40     
41         public MethodMatcher(String name, String params, String returns) {
42             super(name, SignatureUtil.createMethodSignature(params, returns));
43         }
44     
45         public MethodMatcher(String name, String params, String returns, String role) {
46             super(name, SignatureUtil.createMethodSignature(params, returns), role);
47         }
48     
49         @Override
50         public boolean match(BugInstance bugInstance) {
51     
52             MethodAnnotation methodAnnotation = null;
53             if (role == null || "".equals(role)) {
54                 methodAnnotation = bugInstance.getPrimaryMethod();
55             } else {
56                 for (BugAnnotation a : bugInstance.getAnnotations()) {
57                     if (a instanceof MethodAnnotation && role.equals(a.getDescription())) {
58                         methodAnnotation = (MethodAnnotation) a;
59                         break;
60                     }
61                 }
62             }
63             if (methodAnnotation == null) {
64                 return false;
65             }
66             if (!name.match(methodAnnotation.getMethodName())) {
67                 return false;
68             }
69             if (signature != null && !signature.match(methodAnnotation.getMethodSignature())) {
70                 return false;
71             }
72             return true;
73         }
74     
75         @Override
76         public String toString() {
77             return "Method(" + super.toString() + ")";
78         }
79     
80         @Override
81         public void writeXML(XMLOutput xmlOutput, boolean disabled) throws IOException {
82             XMLAttributeList attributes = new XMLAttributeList().addAttribute("name", name.getSpec());
83             if (signature != null && signature.getSpec() != null) {
84                 StringBuilder paramsBuilder = new StringBuilder();
85                 SignatureConverter converter = new SignatureConverter(signature.getSpec());
86                 converter.skip();
87                 while (converter.getFirst() != ')') {
88                     if (paramsBuilder.length() > 1) {
89                         paramsBuilder.append(", ");
90                     }
91                     paramsBuilder.append(converter.parseNext());
92                 }
93                 converter.skip();
94     
95                 String params = paramsBuilder.toString();
96                 String returns = converter.parseNext();
97                 attributes.addAttribute("params", params);
98                 attributes.addAttribute("returns", returns);
99             }
100             attributes.addOptionalAttribute("role", role);
101             if (disabled) {
102                 attributes.addAttribute("disabled", "true");
103             }
104             xmlOutput.openCloseTag("Method", attributes);
105         }
106     }
