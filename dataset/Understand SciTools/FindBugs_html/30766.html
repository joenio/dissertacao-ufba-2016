28     import javax.swing.text.AttributeSet;
29     import javax.swing.text.BadLocationException;
30     import javax.swing.text.EditorKit;
31     import javax.swing.text.Element;
32     import javax.swing.text.html.HTMLDocument;
33     import javax.swing.text.html.HTMLEditorKit;
34     import javax.swing.text.html.HTMLWriter;
35     
36     public class HTML {
37     
38         private static final class HTMLtoPlainTextWriter2 extends HTMLWriter {
39     
40             boolean inPre = false;
41     
42             boolean startingParagraph = false;
43     
44             public HTMLtoPlainTextWriter2(Writer w, HTMLDocument doc) {
45                 super(w, doc);
46                 setLineLength(80);
47                 setCanWrapLines(true);
48             }
49     
50             @Override
51             protected void startTag(Element elem) throws IOException {
52                 String name = elem.getName();
53                 startingParagraph = true;
54                 if ("ul".equals(name)) {
55                     super.incrIndent();
56                     write("  ");
57                 } else if ("pre".equals(name)) {
58                     inPre = true;
59                 } else if ("li".equals(name)) {
60                     super.incrIndent();
61                     write("* ");
62                 } /*else if (name.equals("p")) {
63     
64                 }*/
65             }
66     
67             @Override
68             protected void writeEmbeddedTags(AttributeSet attr) throws IOException {
69     
70             }
71     
72             @Override
73             protected void endTag(Element elem) throws IOException {
74                 String name = elem.getName();
75                 if ("p".equals(name)) {
76                     writeLineSeparator();
77                     indent();
78                 } else if ("pre".equals(name)) {
79                     inPre = false;
80                 } else if ("ul".equals(name)) {
81                     super.decrIndent();
82                     writeLineSeparator();
83                     indent();
84                 } else if ("li".equals(name)) {
85                     super.decrIndent();
86                     writeLineSeparator();
87                     indent();
88                 }
89             }
90     
91             @Override
92             protected void incrIndent() {
93             }
94     
95             @Override
96             protected void decrIndent() {
97             }
98     
99             @Override
100             protected void emptyTag(Element elem) throws IOException, BadLocationException {
101                 if ("content".equals(elem.getName())) {
102                     super.emptyTag(elem);
103                 }
104             }
105     
106             @Override
107             protected void text(Element elem) throws IOException, BadLocationException {
108                 String contentStr = getText(elem);
109                 if (!inPre) {
110                     contentStr = contentStr.replaceAll("\\s+", " ");
111     
112                     if (startingParagraph) {
113                         while (contentStr.length() > 0 && contentStr.charAt(0) == ' ') {
114                             contentStr = contentStr.substring(1);
115                         }
116                     }
117     
118                     startingParagraph = false;
119                 }
120                 if (contentStr.length() > 0) {
121     
122                     setCanWrapLines(!inPre);
123                     write(contentStr);
124                 }
125             }
126         }
127     
128         private HTML() {
129         }
130     
131         public static void convertHtmlToText(Reader reader, Writer writer) throws IOException, BadLocationException {
132     
133             EditorKit kit = new HTMLEditorKit();
134             HTMLDocument doc = new HTMLDocument();
135             kit.read(reader, doc, 0);
136     
137             HTMLtoPlainTextWriter2 x = new HTMLtoPlainTextWriter2(writer, doc);
138             x.write();
139             writer.close();
140     
141         }
142     
143         public static String convertHtmlSnippetToText(String htmlSnippet) throws IOException, BadLocationException {
144             StringWriter writer = new StringWriter();
145             StringReader reader = new StringReader("<HTML><BODY>" + htmlSnippet + "</BODY></HTML>");
146             convertHtmlToText(reader, writer);
147             return writer.toString();
148         }
