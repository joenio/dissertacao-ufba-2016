39         public void showUpdateDialog(Collection<UpdateChecker.PluginUpdate> updates, boolean force) {
40             List<UpdateChecker.PluginUpdate> sortedUpdates = new ArrayList<UpdateChecker.PluginUpdate>();
41             UpdateChecker.PluginUpdate core = sortUpdates(updates, sortedUpdates);
42     
43             if (DetectorFactoryCollection.instance().getUpdateChecker()
44                     .updatesHaveBeenSeenBefore(sortedUpdates) && !force) {
45                 return;
46             }
47     
48             String headline;
49             if (core != null && updates.size() >= 2) {
50                 headline = "FindBugs and some plugins have updates";
51             } else if (updates.isEmpty()) {
52                 headline = "FindBugs and all plugins are up to date!";
53             } else if (core == null) {
54                 headline = "Some FindBugs plugins have updates";
55             } else {
56                 headline = null;
57             }
58     
59             final JPanel comp = new JPanel(new GridBagLayout());
60             GridBagConstraints gbc = new GridBagConstraints();
61             gbc.insets = new Insets(5, 5, 5, 5);
62             gbc.gridwidth = 3;
63             gbc.fill = GridBagConstraints.BOTH;
64             if (headline != null) {
65                 JLabel headlineLabel = new JLabel(headline);
66                 headlineLabel.setFont(headlineLabel.getFont().deriveFont(Font.BOLD, 24));
67                 comp.add(headlineLabel, gbc);
68             }
69             if (!updates.isEmpty()) {
70                 int i = 1;
71                 for (final UpdateChecker.PluginUpdate update : sortedUpdates) {
72                     gbc.gridy = ++i;
73                     gbc.gridx = 1;
74                     gbc.fill = GridBagConstraints.BOTH;
75                     gbc.gridwidth = 1;
76                     gbc.weightx = 1;
77                     JLabel label = createPluginLabel(update);
78                     comp.add(label, gbc);
79                     gbc.weightx = 0;
80                     gbc.gridx = 2;
81                     String url = update.getUrl();
82                     if (url != null && url.length() > 0) {
83                         JButton button = createPluginUpdateButton(comp, update);
84                         comp.add(button, gbc);
85                     }
86                     String msg = update.getMessage();
87                     if (msg != null && msg.length() > 0) {
88                         gbc.gridx = 1;
89                         gbc.gridwidth = 3;
90                         gbc.weightx = 1;
91                         gbc.fill = GridBagConstraints.BOTH;
92                         gbc.gridy = ++i;
93                         JTextPane msgpane = createMessagePane(msg);
94                         comp.add(msgpane, gbc);
95                     }
96                 }
97             }
98             JOptionPane.showMessageDialog(null, comp, "Software Updates", JOptionPane.INFORMATION_MESSAGE);
99         }

109         private JLabel createPluginLabel(UpdateChecker.PluginUpdate update) {
110             String name;
111             if (update.getPlugin().isCorePlugin()) {
112                 name = "FindBugs";
113             } else {
114                 name = update.getPlugin().getShortDescription();
115             }
116             JLabel label = new JLabel(MessageFormat.format(
117                     "<html><b>{0} {2}</b> is available<br><i><small>(currently installed: {1})",
118                     name, update.getPlugin().getVersion(), update.getVersion()));
119             label.setFont(label.getFont().deriveFont(Font.PLAIN, label.getFont().getSize() + 4));
120             return label;
121         }

130         private JButton createPluginUpdateButton(final JPanel comp, final UpdateChecker.PluginUpdate update) {
131             JButton button = new JButton("<html><u><font color=#0000ff>More info...");
132             button.setBorderPainted(false);
133             button.setOpaque(false);
134             button.setContentAreaFilled(false);
135             button.setBackground(comp.getBackground());
136             button.setToolTipText(update.getUrl());
137             button.setCursor(Cursor.getPredefinedCursor(Cursor.HAND_CURSOR));
138             button.addActionListener(new ActionListener() {
139                 @Override
140                 public void actionPerformed(ActionEvent e) {
141                     boolean failed;
142                     String url = update.getUrl();
143                     try {
144                         failed = url == null || !LaunchBrowser.showDocument(new URL(url));
145                     } catch (MalformedURLException e1) {
146                         failed = true;
147                     }
148                     if (failed) {
149                         JOptionPane.showMessageDialog(comp, "Could not open URL " + url);
150                     }
151                 }
152             });
153             return button;
154         }

156         private UpdateChecker.PluginUpdate sortUpdates(Collection<UpdateChecker.PluginUpdate> updates,
157                 List<UpdateChecker.PluginUpdate> sorted) {
158             UpdateChecker.PluginUpdate core = null;
159             for (UpdateChecker.PluginUpdate update : updates) {
160                 if (update.getPlugin().isCorePlugin()) {
161                     core = update;
162                 } else {
163                     sorted.add(update);
164                 }
165             }
166             // sort by name
167             Collections.sort(sorted, new Comparator<UpdateChecker.PluginUpdate>() {
168                 @Override
169                 public int compare(UpdateChecker.PluginUpdate o1, UpdateChecker.PluginUpdate o2) {
170                     return o1.getPlugin().getShortDescription().compareTo(o2.getPlugin().getShortDescription());
171                 }
172             });
173             // place core plugin first, if present
174             if (core != null) {
175                 sorted.add(0, core);
176             }
177             return core;
178         }

180         private class MyPluginUpdateListener implements PluginUpdateListener {
181             @Override
182             public void pluginUpdateCheckComplete(final Collection<UpdateChecker.PluginUpdate> updates, final boolean force) {
183                 if (updates.isEmpty() && !force) {
184                     return;
185                 }
186     
187                 if (force) {
188                     showUpdateDialogInSwingThread(updates, force);
189                 } else {
190                     // wait 5 seconds before showing dialog
191                     edu.umd.cs.findbugs.util.Util.runInDameonThread(new Runnable() {
192                         @Override
193                         public void run() {
194                             try {
195                                 Thread.sleep(SOFTWARE_UPDATE_DIALOG_DELAY_MS);
196                                 showUpdateDialogInSwingThread(updates, force);
197                             } catch (InterruptedException e) {
198                                 LOGGER.log(Level.FINE, "Software update dialog thread interrupted", e);
199                             }
200                         }
201                     }, "Software Update Dialog");
202                 }
203             }
204     
205             private void showUpdateDialogInSwingThread(final Collection<UpdateChecker.PluginUpdate> updates, final boolean force) {
206                 SwingUtilities.invokeLater(new Runnable() {
207                     @Override
208                     public void run() {
209                         showUpdateDialog(updates, force);
210                     }
211                 });
212             }
213         }
