20     package edu.umd.cs.findbugs.detect;
21     
22     import java.util.HashSet;
23     import java.util.Set;
24     
25     import org.apache.bcel.classfile.Code;
26     
27     import edu.umd.cs.findbugs.BugInstance;
28     import edu.umd.cs.findbugs.BugReporter;
29     import edu.umd.cs.findbugs.LocalVariableAnnotation;
30     import edu.umd.cs.findbugs.OpcodeStack;
31     import edu.umd.cs.findbugs.OpcodeStack.Item;
32     import edu.umd.cs.findbugs.Priorities;
33     import edu.umd.cs.findbugs.StatelessDetector;
34     import edu.umd.cs.findbugs.SystemProperties;
35     import edu.umd.cs.findbugs.ba.XField;
36     import edu.umd.cs.findbugs.bcel.OpcodeStackDetector;
37     
38     public class FindFieldSelfAssignment extends OpcodeStackDetector implements StatelessDetector {
39         private final BugReporter bugReporter;
40     
41         private static final boolean DEBUG = SystemProperties.getBoolean("fsa.debug");
42         int state;
43     
44         public FindFieldSelfAssignment(BugReporter bugReporter) {
45             this.bugReporter = bugReporter;
46         }
47     
48         @Override
49         public void visit(Code obj) {
50             state = 0;
51             lastMethodCall = -1;
52     
53             if (DEBUG) {
54                 System.out.println(getXMethod());
55             }
56             super.visit(obj);
57             possibleOverwrite = null;
58             if (DEBUG) {
59                 System.out.println();
60             }
61             initializedFields.clear();
62         }
63     
64         int register;
65     
66         int lastMethodCall;
67         Set<String> initializedFields = new HashSet<String>();
68     
69         XField possibleOverwrite;
70         @Override
71         public void sawOpcode(int seen) {
72     
73             if (DEBUG) {
74                 System.out.printf("%5d %12s %s%n", getPC(), OPCODE_NAMES[seen],stack);
75             }
76             if (seen == PUTFIELD) {
77                 OpcodeStack.Item top = stack.getStackItem(0);
78                 OpcodeStack.Item next = stack.getStackItem(1);
79     
80                 if (possibleOverwrite != null && possibleOverwrite.equals(getXFieldOperand())) {
81                     bugReporter.reportBug(new BugInstance(this, "SA_FIELD_SELF_ASSIGNMENT", Priorities.HIGH_PRIORITY).addClassAndMethod(this)
82                             .addReferencedField(this).addSourceLine(this));
83     
84                 }
85                 possibleOverwrite = null;
86     
87                 if (stack.getStackDepth() >= 4 && getNextOpcode() == PUTFIELD) {
88                     OpcodeStack.Item third = stack.getStackItem(2);
89                     OpcodeStack.Item fourth = stack.getStackItem(3);
90                     XField f2 = third.getXField();
91                     int registerNumber2 = fourth.getRegisterNumber();
92                     if (f2 != null && f2.equals(getXFieldOperand()) && registerNumber2 >= 0
93                             && registerNumber2 == third.getFieldLoadedFromRegister()
94                             && !third.sameValue(top) && (third.getPC() == -1 || third.getPC() > lastMethodCall)) {
95                         possibleOverwrite = f2;
96                     }
97                 }
98     
99                 XField f = top.getXField();
100                 int registerNumber = next.getRegisterNumber();
101                 if (f != null && f.equals(getXFieldOperand()) && registerNumber >= 0
102                         && registerNumber == top.getFieldLoadedFromRegister() && (top.getPC() == -1 || top.getPC() > lastMethodCall)) {
103                     int priority = NORMAL_PRIORITY;
104     
105                     LocalVariableAnnotation possibleMatch = LocalVariableAnnotation.findMatchingIgnoredParameter(getClassContext(),
106                             getMethod(), getNameConstantOperand(), getSigConstantOperand());
107                     if (possibleMatch != null) {
108                         priority--;
109                     } else {
110                         possibleMatch = LocalVariableAnnotation.findUniqueBestMatchingParameter(getClassContext(), getMethod(),
111                                 getNameConstantOperand(), getSigConstantOperand());
112                     }
113                     if (possibleMatch == null) {
114                         String signature = stack.getLVValue(registerNumber).getSignature();
115                         for (int i = 0; i < stack.getNumLocalValues(); i++) {
116                             if (i != register) {
117                                 Item lvValue = stack.getLVValue(i);
118                                 if (lvValue != null && lvValue.getSignature().equals(signature)) {
119                                     priority--;
120                                     break;
121                                 }
122                             }
123                         }
124                     }
125     
126                     bugReporter.reportBug(new BugInstance(this, "SA_FIELD_SELF_ASSIGNMENT", priority).addClassAndMethod(this)
127                             .addReferencedField(this).addOptionalAnnotation(possibleMatch).addSourceLine(this));
128     
129                 }
130             } else {
131                 possibleOverwrite = null;
132             }
133             if (isMethodCall()) {
134                 lastMethodCall = getPC();
135             }
136             switch (state) {
137             case 0:
138                 if (seen == DUP) {
139                     state = 6;
140                 }
141                 break;
142             case 6:
143                 if (isRegisterStore()) {
144                     state = 7;
145                     register = getRegisterOperand();
146                 } else {
147                     state = 0;
148                 }
149                 break;
150             case 7:
151                 if (isRegisterStore() && register == getRegisterOperand()) {
152                     bugReporter.reportBug(new BugInstance(this, "SA_LOCAL_DOUBLE_ASSIGNMENT", NORMAL_PRIORITY)
153                     .addClassAndMethod(this)
154                     .add(LocalVariableAnnotation.getLocalVariableAnnotation(getMethod(), register, getPC(), getPC() - 1))
155                     .addSourceLine(this));
156                 }
157                 state = 0;
158                 break;
159             default:
160                 break;
161             }
162     
163         }
164     
165     }
