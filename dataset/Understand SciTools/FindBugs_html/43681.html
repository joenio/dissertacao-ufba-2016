20     package edu.umd.cs.findbugs.ba.bcp;
21     
22     import org.apache.bcel.generic.ConstantPoolGen;
23     import org.apache.bcel.generic.Instruction;
24     import org.apache.bcel.generic.InstructionHandle;
25     import org.apache.bcel.generic.NEW;
26     
27     import edu.umd.cs.findbugs.ba.DataflowAnalysisException;
28     import edu.umd.cs.findbugs.ba.vna.ValueNumberFrame;
29     
30     /**
31      * A PatternElement which matches NEW instructions and binds the result to a
32      * variable.
33      *
34      * @author David Hovemeyer
35      * @see PatternElement
36      */
37     public class New extends OneVariableInstruction {
38         /**
39          * Constructor.
40          *
41          * @param resultVarName
42          *            name of the result of the NEW instruction
43          */
44         public New(String resultVarName) {
45             super(resultVarName);
46         }
47     
48         @Override
49         public MatchResult match(InstructionHandle handle, ConstantPoolGen cpg, ValueNumberFrame before, ValueNumberFrame after,
50                 BindingSet bindingSet) throws DataflowAnalysisException {
51     
52             Instruction ins = handle.getInstruction();
53             if (!(ins instanceof NEW)) {
54                 return null;
55             }
56     
57             LocalVariable result = new LocalVariable(after.getTopValue());
58             return addOrCheckDefinition(result, bindingSet);
59         }
60     }
