20     package edu.umd.cs.findbugs.filter;
21     
22     import javax.annotation.Nonnull;
23     
24     /**
25      * @author pugh
26      */
27     public class VersionMatcher {
28     
29         protected final long version;
30     
31         protected final RelationalOp relOp;
32     
33         @Override
34         public int hashCode() {
35             return (int) version + relOp.hashCode();
36         }
37     
38         @Override
39         public boolean equals(Object o) {
40             if (o == null || this.getClass() != o.getClass()) {
41                 return false;
42             }
43             VersionMatcher m = (VersionMatcher) o;
44             return version == m.version && relOp.equals(m.relOp);
45         }
46     
47         public VersionMatcher(long version, @Nonnull RelationalOp relOp) {
48             if (relOp == null) {
49                 throw new NullPointerException("relOp must be nonnull");
50             }
51             this.version = version;
52             this.relOp = relOp;
53         }
54     
55     }
