20     package edu.umd.cs.findbugs;
21     
22     /**
23      * Format the message for a BugInstance. This class works in much the same way
24      * as <code>java.text.MessageFormat</code>; however, each placeholder may have
25      * an optional "key" which specifies how the object at that position should be
26      * formatted.
27      * <p/>
28      * <p>
29      * Example:
30      *
31      * <pre>
32      * new FindBugsMessageFormat(&quot;BUG: {1} does something bad to field {2.fullField}&quot;)
33      * </pre>
34      *
35      * In this example, the method annotation at position 1 is formatted using the
36      * empty (default) key. The field annotation at position 2 is formatted using
37      * the "fullField" key, which uses the long format for the field rather than the
38      * usual "class.fieldname" format.
39      *
40      * @author David Hovemeyer
41      * @see BugInstance
42      */
43     public class FindBugsMessageFormat {
44         private final String pattern;
45     
46         /**
47          * Constructor.
48          *
49          * @param pattern
50          *            the pattern for the message
51          */
52         public FindBugsMessageFormat(String pattern) {
53             this.pattern = pattern;
54         }
55     
56         public String format(BugAnnotation[] args, ClassAnnotation primaryClass) {
57             return format(args, primaryClass, false);
58         }
59     
60         /**
61          * Format the message using the given array of BugAnnotations as arguments
62          * to bind to the placeholders in the pattern string.
63          *
64          * @param args
65          *            the BugAnnotations used as arguments
66          * @param primaryClass
67          *            TODO
68          * @return the formatted message
69          */
70         public String format(BugAnnotation[] args, ClassAnnotation primaryClass, boolean abridgedMessages) {
71             String pat = pattern;
72             StringBuilder result = new StringBuilder();
73     
74             while (pat.length() > 0) {
75                 int subst = pat.indexOf('{');
76                 if (subst < 0) {
77                     result.append(pat);
78                     break;
79                 }
80     
81                 result.append(pat.substring(0, subst));
82                 pat = pat.substring(subst + 1);
83     
84                 int end = pat.indexOf('}');
85                 if (end < 0) {
86                     throw new IllegalStateException("unmatched { in " + pat);
87                 }
88     
89                 String substPat = pat.substring(0, end);
90     
91                 int dot = substPat.indexOf('.');
92                 String key = "";
93                 if (dot >= 0) {
94                     key = substPat.substring(dot + 1);
95                     substPat = substPat.substring(0, dot);
96                 } else if (abridgedMessages && primaryClass != null) {
97                     key = "givenClass";
98                 }
99     
100                 int fieldNum;
101                 try {
102                     fieldNum = Integer.parseInt(substPat);
103                 } catch (NumberFormatException e) {
104                     throw new IllegalArgumentException("Bad integer value " + substPat + " in " + pattern);
105                 }
106     
107                 // System.out.println("fn: " + fieldNum);
108                 if (fieldNum < 0) {
109                     result.append("?<?" + fieldNum + "/" + args.length + "???");
110                 } else if (fieldNum >= args.length) {
111                     result.append("?>?" + fieldNum + "/" + args.length + "???");
112                 } else {
113                     BugAnnotation field = args[fieldNum];
114                     String formatted = "";
115                     try {
116                         formatted = field.format(key, primaryClass);
117                     } catch (IllegalArgumentException iae) {
118                         if (SystemProperties.ASSERTIONS_ENABLED) {
119     
120                             throw new IllegalArgumentException("Problem processing " + pattern + " format " + substPat + " for "
121                                     + field.getClass().getSimpleName(), iae);
122                         }
123                         // unknown key -- not unprecedented when reading xml
124                         // generated by older versions of findbugs
125                         formatted = "\u00BF" + fieldNum + ".(key=" + key + ")?"; // "\u00BF"
126                         // is
127                         // inverted
128                         // question
129                         // mark
130                         // System.err.println(iae.getMessage()+" in FindBugsMessageFormat");
131                         // // FIXME: log this error better
132                     }
133                     result.append(formatted);
134                 }
135     
136                 pat = pat.substring(end + 1);
137             }
138     
139             return result.toString();
140         }
141     }
