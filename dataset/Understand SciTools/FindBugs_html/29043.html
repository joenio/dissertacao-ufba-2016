50         public static void attachAcceleratorKey(JMenuItem item, int keystroke, int additionalMask) {
51             // As far as I know, Mac is the only platform on which it is normal
52             // practice to use accelerator masks such as Shift and Alt, so
53             // if we're not running on Mac, just ignore them
54             if (!MainFrame.MAC_OS_X && additionalMask != 0) {
55                 return;
56             }
57     
58             item.setAccelerator(KeyStroke.getKeyStroke(keystroke, Toolkit.getDefaultToolkit().getMenuShortcutKeyMask()
59                     | additionalMask));
60         }
61     
62         public static void attachAcceleratorKeyNoCtrl(JMenuItem item, int keyEvent) {
63             item.setAccelerator(KeyStroke.getKeyStroke(keyEvent, 0));
