41     /**
42      * Factory for Obligation and ObligationSet objects to be used in an instance of
43      * ObligationAnalysis.
44      */
45     public class ObligationFactory {
46         private final Map<String, Obligation> classNameToObligationMap;
47     
48         private final Set<String> slashedClassNames = new HashSet<String>();
49     
50         // // XXX: this is just for debugging.
51         // static ObligationFactory lastInstance;
52     
53         @SuppressFBWarnings("ST_WRITE_TO_STATIC_FROM_INSTANCE_METHOD")
54         public ObligationFactory() {
55             this.classNameToObligationMap = new HashMap<String, Obligation>();
56             // lastInstance = this;
57         }
58     
59         public int getMaxObligationTypes() {
60             return classNameToObligationMap.size();
61         }
62     
63         public boolean signatureInvolvesObligations(String sig) {
64             sig = sig.replaceAll("java/io/File", "java/io/");
65             for (String c : slashedClassNames) {
66                 if (sig.indexOf(c) >= 0) {
67                     return true;
68                 }
69             }
70             return false;
71         }
72     
73         /**
74          * Determine whether class named by given ClassDescriptor is an Obligation
75          * type.
76          *
77          * @param classDescriptor
78          *            a class
79          * @return true if the class is an Obligation type, false otherwise
80          */
81         public boolean isObligationType(ClassDescriptor classDescriptor) {
82             try {
83                 return getObligationByType(BCELUtil.getObjectTypeInstance(classDescriptor.toDottedClassName())) != null;
84             } catch (ClassNotFoundException e) {
85                 Global.getAnalysisCache().getErrorLogger().reportMissingClass(e);
86                 return false;
87             }
88         }
89     
90         /**
91          * Get an Iterator over known Obligation types.
92          *
93          * @return Iterator over known Obligation types
94          */
95         public Iterator<Obligation> obligationIterator() {
96             return classNameToObligationMap.values().iterator();
97         }
98     
99         /**
100          * Look up an Obligation by type. This returns the first Obligation that is
101          * a supertype of the type given (meaning that the given type could be an
102          * instance of the returned Obligation).
103          *
104          * @param type
105          *            a type
106          * @return an Obligation that is a supertype of the given type, or null if
107          *         there is no such Obligation
108          * @throws ClassNotFoundException
109          */
110         public @CheckForNull
111         Obligation getObligationByType(ObjectType type) throws ClassNotFoundException {
112             for (Iterator<Obligation> i = obligationIterator(); i.hasNext();) {
113                 Obligation obligation = i.next();
114                 if (Hierarchy.isSubtype(type, obligation.getType())) {
115                     return obligation;
116                 }
117             }
118             return null;
119         }
120     
121         /**
122          * Look up an Obligation by type. This returns the first Obligation that is
123          * a supertype of the type given (meaning that the given type could be an
124          * instance of the returned Obligation).
125          *
126          * @param classDescriptor
127          *            a ClassDescriptor naming a class type
128          * @return an Obligation that is a supertype of the given type, or null if
129          *         there is no such Obligation
130          */
131         public @CheckForNull
132         Obligation getObligationByType(ClassDescriptor classDescriptor) {
133             try {
134                 return getObligationByType(BCELUtil.getObjectTypeInstance(classDescriptor.toDottedClassName()));
135             } catch (ClassNotFoundException e) {
136                 Global.getAnalysisCache().getErrorLogger().reportMissingClass(e);
137                 return null;
138             }
139         }
140     
141         /**
142          * Get array of Obligation types corresponding to the parameters of the
143          * given method.
144          *
145          * @param xmethod
146          *            a method
147          * @return array of Obligation types for each of the method's parameters; a
148          *         null element means the corresponding parameter is not an
149          *         Obligation type
150          */
151         public Obligation[] getParameterObligationTypes(XMethod xmethod) {
152             Type[] paramTypes = Type.getArgumentTypes(xmethod.getSignature());
153             Obligation[] result = new Obligation[paramTypes.length];
154             for (int i = 0; i < paramTypes.length; i++) {
155                 if (!(paramTypes[i] instanceof ObjectType)) {
156                     continue;
157                 }
158                 try {
159                     result[i] = getObligationByType((ObjectType) paramTypes[i]);
160                 } catch (ClassNotFoundException e) {
161                     Global.getAnalysisCache().getErrorLogger().reportMissingClass(e);
162                 }
163             }
164             return result;
165         }
166     
167         public Obligation addObligation(@DottedClassName String className) {
168             int nextId = classNameToObligationMap.size();
169             slashedClassNames.add(className.replace('.', '/'));
170             Obligation obligation = new Obligation(className, nextId);
171             if (classNameToObligationMap.put(className, obligation) != null) {
172                 throw new IllegalStateException("Obligation " + className + " added multiple times");
173             }
174             return obligation;
175         }
176     
177         public Obligation getObligationById(int id) {
178             for (Obligation obligation : classNameToObligationMap.values()) {
179                 if (obligation.getId() == id) {
180                     return obligation;
181                 }
182             }
183             return null;
184         }
185     
186         public Obligation getObligationByName(@DottedClassName String className) {
187             return classNameToObligationMap.get(className);
188         }
189     
190         public ObligationSet createObligationSet() {
191             return new ObligationSet(/* getMaxObligationTypes(), */this);
192         }
193     }
