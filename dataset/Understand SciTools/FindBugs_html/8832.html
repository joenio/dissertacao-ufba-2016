49         final Set<String> abstractClasses = new HashSet<String>();

53         final Set<String> classesScanned = new HashSet<String>();
54     
55         final Set<XField> fieldsOfNativeClasses = new HashSet<XField>();
56     
57         final Set<XField> reflectiveFields = new HashSet<XField>();
58     
59         final Set<XField> fieldsOfSerializableOrNativeClassed = new HashSet<XField>();
60     
61         final  Set<XField> staticFieldsReadInThisMethod = new HashSet<XField>();
62

64     
65         final Set<XField> myFields = new TreeSet<XField>();

67         final  Set<XField> writtenFields = new HashSet<XField>();

91     
92         public boolean isContainerField(XField f) {
93             return containerFields.contains(f);
94         }
95     
96         public void strongEvidenceForIntendedSerialization(ClassDescriptor c) {
97             toldStrongEvidenceForIntendedSerialization.add(c);
98         }
99     
100         public boolean existsStrongEvidenceForIntendedSerialization(ClassDescriptor c) {
101             return toldStrongEvidenceForIntendedSerialization.contains(c);
102         }
103     
104         public boolean isWrittenOutsideOfInitialization(XField f) {
105             return writtenOutsideOfInitializationFields.contains(f);
106         }
107     
108         public boolean isReflexive(XField f) {
109             return reflectiveFields.contains(f);
110         }
111     
112         public Set<? extends XField> getReadFields() {
113             return readFields;
114         }
115     
116         public Set<? extends XField> getWrittenFields() {
117             return writtenFields;
118         }
119     
120         public boolean isWrittenInConstructor(XField f) {
121             return writtenInConstructorFields.contains(f);
122         }
123     
124         public boolean isWrittenDuringInitialization(XField f) {
125             return writtenInInitializationFields.contains(f);
126         }
