20     package edu.umd.cs.findbugs.ba;
21     
22     import java.io.FileNotFoundException;
23     import java.io.IOException;
24     import java.io.InputStream;
25     import java.util.zip.ZipEntry;
26     import java.util.zip.ZipFile;
27     
28     /**
29      * A source file data source for source files residing in Zip or Jar archives.
30      */
31     public class ZipSourceFileDataSource implements SourceFileDataSource {
32         private final ZipFile zipFile;
33     
34         private final String entryName;
35     
36         private final ZipEntry zipEntry;
37     
38         public ZipSourceFileDataSource(ZipFile zipFile, String entryName) {
39             this.zipFile = zipFile;
40             this.entryName = entryName;
41             this.zipEntry = zipFile.getEntry(entryName);
42         }
43     
44         @Override
45         public InputStream open() throws IOException {
46             if (zipEntry == null) {
47                 throw new FileNotFoundException("No zip entry for " + entryName);
48             }
49             return zipFile.getInputStream(zipEntry);
50         }
51     
52         @Override
53         public String getFullFileName() {
54             return entryName;
55         }
56     
57         /* (non-Javadoc)
58          * @see edu.umd.cs.findbugs.ba.SourceFileDataSource#getLastModified()
59          */
60         @Override
61         public long getLastModified() {
62             long time = zipEntry.getTime();
63             if (time < 0) {
64                 return 0;
65             }
66             return time;
67         }
68     }
