20     package edu.umd.cs.findbugs.ba;
21     
22     import java.util.ArrayList;
23     import java.util.Arrays;
24     import java.util.Comparator;
25     import java.util.Iterator;
26     
27     /**
28      * Abstract base class for BlockOrder variants. It allows the subclass to
29      * specify just a Comparator for BasicBlocks, and handles the work of doing the
30      * sorting and providing Iterators.
31      *
32      * @see BlockOrder
33      */
34     public abstract class AbstractBlockOrder implements BlockOrder {
35         private final ArrayList<BasicBlock> blockList;
36         private final Comparator<BasicBlock> comparator;
37     
38         public AbstractBlockOrder(CFG cfg, Comparator<BasicBlock> comparator) {
39             this.comparator = comparator;
40             // Put the blocks in an array
41             int numBlocks = cfg.getNumBasicBlocks(), count = 0;
42             BasicBlock[] blocks = new BasicBlock[numBlocks];
43             for (Iterator<BasicBlock> i = cfg.blockIterator(); i.hasNext();) {
44                 blocks[count++] = i.next();
45             }
46             assert count == numBlocks;
47     
48             // Sort the blocks according to the comparator
49             Arrays.sort(blocks, comparator);
50     
51             // Put the ordered blocks into an array list
52             blockList = new ArrayList<BasicBlock>(numBlocks);
53             for (int i = 0; i < numBlocks; ++i) {
54                 blockList.add(blocks[i]);
55             }
56         }
57     
58         @Override
59         public Iterator<BasicBlock> blockIterator() {
60             return blockList.iterator();
61         }
62     
63         @Override
64         public int compare(BasicBlock b1, BasicBlock b2) {
65             return comparator.compare(b1, b2);
66         }
67     }
