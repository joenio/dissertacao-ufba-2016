100         private DetectorFactoryCollection(boolean loadCore, boolean forceLoad,
101                 @Nonnull Collection<Plugin> pluginsToLoad,
102                 @Nonnull Collection<Plugin> enabledPlugins) {
103             if(loadCore) {
104                 loadCorePlugin();
105             }
106             for(Plugin plugin : pluginsToLoad) {
107                 if (forceLoad || plugin.isGloballyEnabled() && !plugin.isCorePlugin()) {
108                     loadPlugin(plugin);
109                     if(!enabledPlugins.contains(plugin)) {
110                         enabledPlugins.add(plugin);
111                     }
112                 }
113             }
114             setGlobalOptions();
115             updateChecker = new UpdateChecker(this);
116             pluginsToUpdate = combine(enabledPlugins);
117             // There are too many places where the detector factory collection can be created,
118             // and in many cases it has nothing to do with update checks, like Plugin#addCustomPlugin(URI).
119     
120             // Line below commented to allow custom plugins be loaded/registered BEFORE we
121             // start to update something. The reason is that custom plugins
122             // can disable update globally OR just will be NOT INCLUDED in the update check!
123             // updateChecker.checkForUpdates(pluginsToUpdate, false);
124         }

134         private Collection<Plugin> combine(Collection<Plugin> enabled) {
135             List<Plugin> result = new ArrayList<Plugin>(enabled);
136             if (corePlugin != null && !result.contains(corePlugin)) {
137                 result.add(corePlugin);
138             }
139             return result;
140         }

169         private void setGlobalOptions() {
170             globalOptions.clear();
171             globalOptionsSetter.clear();
172     
173             for(Plugin p : plugins()) {
174                 if (p.isGloballyEnabled()) {
175                     for(Map.Entry<String, String> e : p.getMyGlobalOptions().entrySet()) {
176                         String key = e.getKey();
177                         String value = e.getValue();
178                         String oldValue = globalOptions.get(key);
179     
180                         if (oldValue != null) {
181                             if (oldValue.equals(value)) {
182                                 continue;
183                             }
184                             Plugin oldP = globalOptionsSetter.get(key);
185                             throw new RuntimeException(
186                                     "Incompatible global options for " + key + "; conflict between " + oldP.getPluginId() + " and " + p.getPluginId());
187                         }
188                         globalOptions.put(key, value);
189                         globalOptionsSetter.put(key, p);
190                     }
191                 }
192             }
193         }

255         public boolean isDisabledByDefault(String bugPatternOrCode) {
256             @CheckForNull BugPattern pattern = lookupBugPattern(bugPatternOrCode);
257             if (pattern != null) {
258                 for(DetectorFactory fac : factoryList) {
259                     if (fac.isDefaultEnabled()  && fac.getReportedBugPatterns().contains(pattern)) {
260                         return false;
261                     }
262                 }
263                 return true;
264             }
265             @CheckForNull BugCode code = lookupBugCode(bugPatternOrCode);
266             if (code != null) {
267                 for(DetectorFactory fac : factoryList) {
268                     if (fac.isDefaultEnabled()) {
269                         for(BugPattern p : fac.getReportedBugPatterns()) {
270                             if (p.getBugCode().equals(code)) {
271                                 return false;
272                             }
273                         }
274                     }
275                 }
276                 return true;
277             }
278     
279             return false;
280         }

306         /**
307          * Register a DetectorFactory.
308          */
309         void registerDetector(DetectorFactory factory) {
310             if (FindBugs.DEBUG) {
311                 System.out.println("Registering detector: " + factory.getFullName());
312             }
313             String detectorName = factory.getShortName();
314             if(!factoryList.contains(factory)) {
315                 factoryList.add(factory);
316             } else {
317                 LOGGER.log(Level.WARNING, "Trying to add already registered factory: " + factory +
318                         ", " + factory.getPlugin());
319             }
320             factoriesByName.put(detectorName, factory);
321             factoriesByDetectorClassName.put(factory.getFullName(), factory);
322         }
323     
324         void unRegisterDetector(DetectorFactory factory) {
325             if (FindBugs.DEBUG) {
326                 System.out.println("Unregistering detector: " + factory.getFullName());
327             }
328             String detectorName = factory.getShortName();
329             factoryList.remove(factory);
330             factoriesByName.remove(detectorName);
331             factoriesByDetectorClassName.remove(factory.getFullName());
332         }

335         /**
336          * See if the location of ${findbugs.home} can be inferred from the location
337          * of findbugs.jar in the classpath.
338          *
339          * @return inferred ${findbugs.home}, or null if we can't figure it out
340          */
341         private static String inferFindBugsHome() {
342             Pattern[] findbugsJarNames = { Pattern.compile("findbugs\\.jar$"), };
343     
344             for (Pattern jarNamePattern : findbugsJarNames) {
345                 String findbugsJarCodeBase = ClassPathUtil.findCodeBaseInClassPath(jarNamePattern,
346                         SystemProperties.getProperty("java.class.path"));
347                 if (findbugsJarCodeBase != null) {
348                     File findbugsJar = new File(findbugsJarCodeBase);
349                     File libDir = findbugsJar.getParentFile();
350                     if ("lib".equals(libDir.getName())) {
351                         String fbHome = libDir.getParent();
352                         FindBugs.setHome(fbHome);
353                         return fbHome;
354                     }
355                 }
356             }
357             String classFilePath = FindBugs.class.getName().replaceAll("\\.", "/") + ".class";
358             URL resource = FindBugs.class.getClassLoader().getResource(classFilePath);
359             if (resource != null && "file".equals(resource.getProtocol())) {
360                 try {
361                     String classfile = URLDecoder.decode(resource.getPath(), Charset.defaultCharset().name());
362                     Matcher m = Pattern.compile("(.*)/.*?/edu/umd.*").matcher(classfile);
363                     if (m.matches()) {
364                         String home = m.group(1);
365                         if (new File(home + "/etc/findbugs.xml").exists()) {
366                             FindBugs.setHome(home);
367                             return home;
368                         }
369                     }
370                 } catch (UnsupportedEncodingException e) {
371                 }
372             }
373             return null;
374     
375         }

377         public static String getFindBugsHome() {
378     
379             String homeDir = FindBugs.getHome();
380     
381             if (homeDir == null) {
382                 // Attempt to infer findbugs.home from the observed
383                 // location of findbugs.jar.
384                 homeDir = inferFindBugsHome();
385             }
386             return homeDir;

409         void loadPlugin(Plugin plugin)  {
410     
411             if (FindBugs.DEBUG) {
412                 System.out.println("Loading " + plugin.getPluginId());
413             }
414             pluginByIdMap.put(plugin.getPluginId(), plugin);
415     
416             setGlobalOptions();
417     
418             // Register all of the detectors that this plugin contains
419             for (DetectorFactory factory : plugin.getDetectorFactories()) {
420                 registerDetector(factory);
421             }
422     
423             for (BugCategory bugCategory : plugin.getBugCategories()) {
424                 registerBugCategory(bugCategory);
425             }
426     
427             // Register the BugPatterns
428             for (BugPattern bugPattern : plugin.getBugPatterns()) {
429                 registerBugPattern(bugPattern);
430             }
431     
432             // Register the BugCodes
433             for (BugCode bugCode : plugin.getBugCodes()) {
434                 registerBugCode(bugCode);
435             }
436             for (CloudPlugin cloud : plugin.getCloudPlugins()) {
437                 registerCloud(cloud);
438             }
439         }
440     
441         void unLoadPlugin(Plugin plugin)  {
442             pluginByIdMap.remove(plugin.getPluginId());
443     
444             setGlobalOptions();
445     
446             for (DetectorFactory factory : plugin.getDetectorFactories()) {
447                 unRegisterDetector(factory);
448             }
449             for (BugCategory bugCategory : plugin.getBugCategories()) {
450                 unRegisterBugCategory(bugCategory);
451             }
452             for (BugPattern bugPattern : plugin.getBugPatterns()) {
453                 unRegisterBugPattern(bugPattern);
454             }
455             for (BugCode bugCode : plugin.getBugCodes()) {
456                 unRegisterBugCode(bugCode);
457             }
458             for (CloudPlugin cloud : plugin.getCloudPlugins()) {
459                 unRegisterCloud(cloud);
460             }
461         }

463         @Override
464         @SuppressWarnings({"ConstantConditions"})
465         public void pluginUpdateCheckComplete(List<UpdateChecker.PluginUpdate> newUpdates, boolean force) {
466             this.updates = newUpdates;
467             this.updatesForced = force;
468             for (PluginUpdateListener listener : pluginUpdateListeners) {
469                 try {
470                     listener.pluginUpdateCheckComplete(newUpdates, force);
471                 } catch (Throwable e) {
472                     LOGGER.log(Level.INFO, "Error during update check callback", e);
473                 }
474             }
475         }
476     
477         public void addPluginUpdateListener(PluginUpdateListener listener) {
478             pluginUpdateListeners.add(listener);
479             if (updates != null) {
480                 listener.pluginUpdateCheckComplete(updates, updatesForced);
481             } else if(!updateChecker.updateChecksGloballyDisabled()){
482                 checkForUpdates(false);
483             }
484         }

486         public FutureValue<Collection<UpdateChecker.PluginUpdate>> getUpdates() {
487             final FutureValue<Collection<UpdateChecker.PluginUpdate>> results = new FutureValue<Collection<UpdateChecker.PluginUpdate>>();
488             addPluginUpdateListener(new PluginUpdateListener() {
489                 @Override
490                 public void pluginUpdateCheckComplete(Collection<UpdateChecker.PluginUpdate> u, boolean force) {
491                     results.set(u);
492                 }
493             });
494             return results;
495         }
