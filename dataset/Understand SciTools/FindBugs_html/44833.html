92                 if(seen == INVOKESTATIC && getClassConstantOperand().equals("java/lang/Math") && (getMethodDescriptorOperand().getName().equals("max")
93                         || getMethodDescriptorOperand().getName().equals("min"))) {
94                     Object const1 = stack.getStackItem(0).getConstant();
95                     Object const2 = stack.getStackItem(1).getConstant();
96                     Number n = null;
97                     if(const1 != null ^ const2 != null) {
98                         n = (const1 instanceof Number) ? (Number)const1 : (Number)const2;
99                         if(getMethodDescriptorOperand().getName().equals("min")) {
100                             upperBound = n;
101                         } else {
102                             lowerBound = n;
103                         }
104                     } else {
105                         upperBound = lowerBound = null;
106                     }
107                     XMethod rvo1 = stack.getStackItem(0).getReturnValueOf();
108                     XMethod rvo2 = stack.getStackItem(1).getReturnValueOf();
109                     if(rvo1 != null ^ rvo2 != null) {
110                         XMethod rvo = rvo1 == null ? rvo2 : rvo1;
111                         if (lowerBound instanceof Comparable && upperBound != null && upperBound.getClass() == lowerBound.getClass()
112                                 && rvo.getClassDescriptor().getClassName().equals("java/lang/Math")
113                                 && (rvo.getName().equals("max") || rvo.getName().equals("min"))) {
114                             @SuppressWarnings("unchecked")
115                             int result = ((Comparable<Number>)lowerBound).compareTo(upperBound);
116                             if(result > 0) {
117                                 accumulator.accumulateBug(
118                                         new BugInstance("DM_INVALID_MIN_MAX", HIGH_PRIORITY).addClassAndMethod(DumbMethods.this)
119                                         .addString(String.valueOf(n)),
120                                         DumbMethods.this);

221             private void checkRange(Item item, Object minValue, Object maxValue, String pattern) {
222                 if(!(item.getConstant() instanceof Number)) {
223                     return;
224                 }
225                 int value = ((Number)item.getConstant()).intValue();
226                 int intMin = Integer.MIN_VALUE;
227                 int intMax = Integer.MAX_VALUE;
228                 if(minValue instanceof Number) {
229                     intMin = ((Number)minValue).intValue();
230                 }
231                 if(maxValue instanceof Number) {
232                     intMax = ((Number)maxValue).intValue();
233                 } else if(maxValue instanceof String) {
234                     intMax = ((String)maxValue).length()-1;
235                 } else if (maxValue instanceof OpcodeStack.Item){
236                     OpcodeStack.Item maxItem = (OpcodeStack.Item ) maxValue;
237                     if (maxItem.getSignature().charAt(0) == '[' && maxItem.getConstant() instanceof Integer) {
238                         intMax = ((Integer)maxItem.getConstant())-1;
239     
240                     }
241                 }
242     
243                 if(value < intMin || value > intMax) {
244                     BugInstance bug = new BugInstance(pattern, NORMAL_PRIORITY ).addClassAndMethod(DumbMethods.this).addSourceLine(DumbMethods.this)
245                             .addInt(value).describe(IntAnnotation.INT_VALUE);
246     
247                     if (intMin <= intMax) {
248                         if (value < intMin) {
249                             bug.addInt(intMin).describe(IntAnnotation.INT_MIN_VALUE);
250                         }
251                         if (value > intMax) {
252                             bug.addInt(intMax) .describe(IntAnnotation.INT_MAX_VALUE);
253                         }
254                     }
255     
256     
257                     if (isMethodCall()) {
258                         bug.addCalledMethod(DumbMethods.this);
259                     }
260     
261     
262     
263                     accumulator.accumulateBug(bug, DumbMethods.this);
264                 }
265             }

268             @Override
269             public void sawOpcode(int seen) {
270                 // System.out.printf("%4d %s%n", getPC(), OPCODE_NAMES[seen]);
271                 switch(seen) {
272                 case IALOAD:
273                 case AALOAD:
274                 case SALOAD:
275                 case CALOAD:
276                 case BALOAD:
277                 case LALOAD:
278                 case DALOAD:
279                 case FALOAD: {
280                     checkRange(stack.getStackItem(0), 0, stack.getStackItem(1), "RANGE_ARRAY_INDEX");
281                     break;
282                 }
283                 case IASTORE:
284                 case AASTORE:
285                 case SASTORE:
286                 case CASTORE:
287                 case BASTORE:
288                 case LASTORE:
289                 case DASTORE:
290                 case FASTORE: {
291     
292                     checkRange(stack.getStackItem(1), 0,  stack.getStackItem(2), "RANGE_ARRAY_INDEX");
293                     break;
294                 }
295                 case INVOKESTATIC: {
296                     MethodDescriptor m = getMethodDescriptorOperand();
297                     if(m.getSlashedClassName().equals("java/lang/System") && m.getName().equals("arraycopy")) {
298                         // void arraycopy(Object src, int srcPos, Object dest, int destPos, int length)
299                         Item length = stack.getStackItem(0);
300                         Object constantLength = length.getConstant();
301                         //                    if (constantLength instanceof Number && constantLength.equals(0)) {
302                         //                        break;
303                         //                    }
304                         Item srcPos = stack.getStackItem(3);
305                         Item src = stack.getStackItem(4);
306                         checkRange(srcPos, 0, src, "RANGE_ARRAY_OFFSET");
307                         Item dest =  stack.getStackItem(2);
308                         Item destPos = stack.getStackItem(1);
309                         checkRange(destPos, 0, dest, "RANGE_ARRAY_OFFSET");
310     
311                         if(constantLength instanceof Number) {
312                             int length1 = Integer.MAX_VALUE;
313                             if(src.getConstant() instanceof Integer) {
314                                 length1 = (int) src.getConstant();
315                             }
316                             if(srcPos.getConstant() instanceof Integer) {
317                                 length1 -= (int) srcPos.getConstant();
318                             }
319                             int length2 = Integer.MAX_VALUE;
320                             if(dest.getConstant() instanceof Integer) {
321                                 length2 = (int) stack.getStackItem(2).getConstant();
322                             }
323                             if(destPos.getConstant() instanceof Integer) {
324                                 length2 -= (int) stack.getStackItem(1).getConstant();
325                             }
326                             checkRange(length, 0, Math.min(length1, length2), "RANGE_ARRAY_LENGTH");
327                         }
328                     }
329                     break;
330                 }
331                 case INVOKEVIRTUAL:
332                 case INVOKESPECIAL: {
333                     MethodDescriptor m = getMethodDescriptorOperand();
334                     if(m.getSlashedClassName().equals("java/lang/String")) {
335                         if((m.getName().equals("charAt") || m.getName().equals("codePointAt")) && m.getSignature().startsWith("(I)")) {
336                             checkRange(stack.getStackItem(0), 0, stack.getStackItem(1).getConstant(), "RANGE_STRING_INDEX");
337                         }
338                         if(m.getName().equals("substring") || m.getName().equals("subSequence")) {
339                             int nArgs = getNumberArguments(m.getSignature());
340                             Item thisArg = stack.getStackItem(nArgs);
341                             Item firstArg = stack.getStackItem(nArgs-1);
342                             Object thisConstantValue = thisArg.getConstant();
343                             int maxLength = thisConstantValue instanceof String ? ((String)thisConstantValue).length() : Integer.MAX_VALUE;
344                             checkRange(firstArg, 0,maxLength, "RANGE_STRING_INDEX");
345                             if(nArgs == 2) {
346                                 Item secondArg = stack.getStackItem(0);
347                                 checkRange(secondArg, firstArg.getConstant() == null ? 0 : firstArg.getConstant(),
348                                         maxLength,
349                                         "RANGE_STRING_INDEX");
350                             }
351                         }
352                     }
353                     if ((m.getSignature().startsWith("([BII)") || m.getSignature().startsWith("([CII)") || m.getSignature().startsWith("([III)"))
354                             && (((m.getName().equals("write") || m.getName().equals("read")) && m.getSlashedClassName().startsWith(
355                                     "java/io/")) || (m.getName().equals("<init>") && m.getSlashedClassName().equals("java/lang/String")))) {
356                         Item arrayArg = stack.getStackItem(2);
357                         Item offsetArg = stack.getStackItem(1);
358                         Item lengthArg = stack.getStackItem(0);
359                         int length = Integer.MAX_VALUE;
360                         if(arrayArg.getConstant() instanceof Integer) {
361                             length = (int) arrayArg.getConstant();
362                         }
363                         if(offsetArg.getConstant() instanceof Integer) {
364                             checkRange(offsetArg, 0, saturatingIncrement(length), "RANGE_ARRAY_OFFSET");
365                             length -= (int) offsetArg.getConstant();
366                         }
367                         checkRange(lengthArg, 0, saturatingIncrement(length), "RANGE_ARRAY_LENGTH");
368                     }
369                     break;
370                 }
371                 default:
372                     break;
373                 }
374             }
375         }

394             @Override
395             public void sawOpcode(int seen) {
396                 boolean foundVacuousComparison = false;
397                 if (seen == IF_ICMPGT || seen == IF_ICMPLE) {
398                     OpcodeStack.Item rhs = stack.getStackItem(0);
399                     Object rhsConstant = rhs.getConstant();
400                     if (rhsConstant instanceof Integer && ((Integer) rhsConstant).intValue() == Integer.MAX_VALUE) {
401                         foundVacuousComparison = true;
402                     }
403                     OpcodeStack.Item lhs = stack.getStackItem(1);
404                     Object lhsConstant = lhs.getConstant();
405                     if (lhsConstant instanceof Integer && ((Integer) lhsConstant).intValue() == Integer.MIN_VALUE) {
406                         foundVacuousComparison = true;
407                     }
408     
409                 }
410                 if (seen == IF_ICMPLT || seen == IF_ICMPGE) {
411                     OpcodeStack.Item rhs = stack.getStackItem(0);
412                     Object rhsConstant = rhs.getConstant();
413                     if (rhsConstant instanceof Integer && ((Integer) rhsConstant).intValue() == Integer.MIN_VALUE) {
414                         foundVacuousComparison = true;
415                     }
416                     OpcodeStack.Item lhs = stack.getStackItem(1);
417                     Object lhsConstant = lhs.getConstant();
418                     if (lhsConstant instanceof Integer && ((Integer) lhsConstant).intValue() == Integer.MAX_VALUE) {
419                         foundVacuousComparison = true;
420                     }
421     
422                 }
423                 if (foundVacuousComparison) {
424                     accumulator.accumulateBug(new BugInstance(DumbMethods.this, "INT_VACUOUS_COMPARISON", getBranchOffset() < 0 ? HIGH_PRIORITY
