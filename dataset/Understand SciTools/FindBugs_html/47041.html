99         static class AbnormalAnalysisResult {
100             final CheckedAnalysisException checkedAnalysisException;
101     
102             final RuntimeException runtimeException;
103     
104             final boolean isNull;
105     
106             AbnormalAnalysisResult(CheckedAnalysisException checkedAnalysisException) {
107                 this.checkedAnalysisException = checkedAnalysisException;
108                 this.runtimeException = null;
109                 isNull = false;
110             }
111     
112             AbnormalAnalysisResult(RuntimeException runtimeException) {
113                 this.runtimeException = runtimeException;
114                 this.checkedAnalysisException = null;
115                 isNull = false;
116             }
117     
118             AbnormalAnalysisResult() {
119                 this.isNull = true;
120                 this.checkedAnalysisException = null;
121                 this.runtimeException = null;
122             }
123     
124             public Object returnOrThrow() throws CheckedAnalysisException {
125                 if (isNull) {
126                     return null;
127                 } else if (runtimeException != null) {
128                     // runtimeException.fillInStackTrace();
129                     throw runtimeException;
130                 } else if (checkedAnalysisException != null) {
131                     // checkedAnalysisException.fillInStackTrace();
132                     throw checkedAnalysisException;
133                 }
134     
135                 throw new IllegalStateException("It has to be something");
136             }
137         }

172         @Override
173         public void purgeAllMethodAnalysis() {
174             // System.out.println("ZZZ : purging all method analyses");
175     
176             try {
177                 Map<ClassDescriptor, ClassContext> map = getAllClassAnalysis(ClassContext.class);
178                 Collection<?> allClassContexts = map.values();
179                 for (Object c : allClassContexts) {
180                     if (c instanceof ClassContext) {
181                         ((ClassContext) c).purgeAllMethodAnalyses();
182                     }
183                 }
184             } catch (ClassCastException e) {
185                 AnalysisContext.logError("Unable to purge method analysis", e);
186             }
187         }

222         /**
223          * Adds the data for given analysis type from given map to the cache
224          * @param analysisClass non null analysis type
225          * @param map non null, pre-filled map with analysis data for given type
226          */
227         public <E> void reuseClassAnalysis(Class<E> analysisClass, Map<ClassDescriptor, Object> map) {
228             Map<ClassDescriptor, Object> myMap = classAnalysisMap.get(analysisClass);
229             if (myMap != null) {
230                 myMap.putAll(map);
231             } else {
232                 myMap = createMap(classAnalysisEngineMap, analysisClass);
233                 myMap.putAll(map);
234                 classAnalysisMap.put(analysisClass, myMap);
235             }
236         }

238         @Override
239         @SuppressWarnings("unchecked")
240         public <E> E getClassAnalysis(Class<E> analysisClass, @Nonnull ClassDescriptor classDescriptor) throws CheckedAnalysisException {
241             requireNonNull(classDescriptor, "classDescriptor is null");
242             // Get the descriptor->result map for this analysis class,
243             // creating if necessary
244             Map<ClassDescriptor, Object> descriptorMap = findOrCreateDescriptorMap(classAnalysisMap,
245                     classAnalysisEngineMap,
246                     analysisClass);
247     
248             // See if there is a cached result in the descriptor map
249             Object analysisResult = descriptorMap.get(classDescriptor);
250             if (analysisResult == null) {
251                 // No cached result - compute (or recompute)
252     
253                 IAnalysisEngine<ClassDescriptor, E> engine = (IAnalysisEngine<ClassDescriptor, E>) classAnalysisEngineMap
254                         .get(analysisClass);
255                 if (engine == null) {
256                     throw new IllegalArgumentException("No analysis engine registered to produce " + analysisClass.getName());
257                 }
258                 Profiler profiler = getProfiler();
259                 // Perform the analysis
260                 try {
261                     profiler.start(engine.getClass());
262                     analysisResult = engine.analyze(this, classDescriptor);
263     
264                     // If engine returned null, we need to construct
265                     // an AbnormalAnalysisResult object to record that fact.
266                     // Otherwise we will try to recompute the value in
267                     // the future.
268                     if (analysisResult == null) {
269                         analysisResult = NULL_ANALYSIS_RESULT;
270                     }
271                 } catch (CheckedAnalysisException e) {
272                     // Exception - make note
273                     // Andrei: e.getStackTrace() cannot be null, but getter clones
274                     // the stack...
275                     // if (e.getStackTrace() == null)
276                     // e.fillInStackTrace();
277                     analysisResult = new AbnormalAnalysisResult(e);
278                 } catch (RuntimeException e) {
279                     // Exception - make note
280                     // Andrei: e.getStackTrace() cannot be null, but getter clones
281                     // the stack...
282                     // if (e.getStackTrace() == null)
283                     // e.fillInStackTrace();
284                     analysisResult = new AbnormalAnalysisResult(e);
285                 } finally {
286                     profiler.end(engine.getClass());
287                 }
288     
289                 // Save the result
290                 descriptorMap.put(classDescriptor, analysisResult);
291             }
292     
293             // Abnormal analysis result?
294             if (analysisResult instanceof AbnormalAnalysisResult) {
295                 return checkedCast(analysisClass, ((AbnormalAnalysisResult) analysisResult).returnOrThrow());
296             }
297     
298             return checkedCast(analysisClass, analysisResult);
299         }

301         @Override
302         public <E> E probeClassAnalysis(Class<E> analysisClass, @Nonnull ClassDescriptor classDescriptor) {
303             Map<ClassDescriptor, Object> descriptorMap = classAnalysisMap.get(analysisClass);
304             if (descriptorMap == null) {
305                 return null;
306             }
307             return checkedCast(analysisClass, descriptorMap.get(classDescriptor));
308         }

314         @Override
315         public <E> E getMethodAnalysis(Class<E> analysisClass, @Nonnull MethodDescriptor methodDescriptor) throws CheckedAnalysisException {
316             requireNonNull(methodDescriptor, "methodDescriptor is null");
317             ClassContext classContext = getClassAnalysis(ClassContext.class, methodDescriptor.getClassDescriptor());
318             Object object = classContext.getMethodAnalysis(analysisClass, methodDescriptor);
319     
320             if (object == null) {
321                 try {
322                     object = analyzeMethod(classContext, analysisClass, methodDescriptor);
323                     if (object == null) {
324                         object = NULL_ANALYSIS_RESULT;
325                     }
326                 } catch (RuntimeException e) {
327                     object = new AbnormalAnalysisResult(e);
328                 } catch (CheckedAnalysisException e) {
329                     object = new AbnormalAnalysisResult(e);
330                 }
331     
332                 classContext.putMethodAnalysis(analysisClass, methodDescriptor, object);
333     
334             }
335             if (Debug.VERIFY_INTEGRITY && object == null) {
336                 throw new IllegalStateException("AnalysisFactory failed to produce a result object");
337             }
338     
339             if (object instanceof AbnormalAnalysisResult) {
340                 return checkedCast(analysisClass, ((AbnormalAnalysisResult) object).returnOrThrow());
341             }
342     
343             return checkedCast(analysisClass, object);
344         }

346         /**
347          * Analyze a method.
348          *
349          * @param classContext
350          *            ClassContext storing method analysis objects for method's
351          *            class
352          * @param analysisClass
353          *            class the method analysis object should belong to
354          * @param methodDescriptor
355          *            method descriptor identifying the method to analyze
356          * @return the computed analysis object for the method
357          * @throws CheckedAnalysisException
358          */
359         @SuppressWarnings("unchecked")
360         private <E> E analyzeMethod(ClassContext classContext, Class<E> analysisClass, MethodDescriptor methodDescriptor)
361                 throws CheckedAnalysisException {
362             IMethodAnalysisEngine<E> engine = (IMethodAnalysisEngine<E>) methodAnalysisEngineMap.get(analysisClass);
363             if (engine == null) {
364                 throw new IllegalArgumentException("No analysis engine registered to produce " + analysisClass.getName());
365             }
366             Profiler profiler = getProfiler();
367             profiler.start(engine.getClass());
368             try {
369                 return engine.analyze(this, methodDescriptor);
370             } finally {
371                 profiler.end(engine.getClass());
372             }
373         }

375         @Override
376         public <E> void eagerlyPutMethodAnalysis(Class<E> analysisClass, @Nonnull MethodDescriptor methodDescriptor, E analysisObject) {
377             try {
378                 ClassContext classContext = getClassAnalysis(ClassContext.class, methodDescriptor.getClassDescriptor());
379                 assert analysisClass.isInstance(analysisObject);
380                 classContext.putMethodAnalysis(analysisClass, methodDescriptor, analysisObject);
381             } catch (CheckedAnalysisException e) {
382                 IllegalStateException ise = new IllegalStateException("Unexpected exception adding method analysis to cache");
383                 ise.initCause(e);
384                 throw ise;
385             }
386     
387         }

402         /**
403          * Find or create a descriptor to analysis object map.
404          *
405          * @param <DescriptorType>
406          *            type of descriptor used as the map's key type (ClassDescriptor
407          *            or MethodDescriptor)
408          * @param analysisClassToDescriptorMapMap
409          *            analysis class to descriptor map map
410          * @param engineMap
411          *            analysis class to analysis engine map
412          * @param analysisClass
413          *            the analysis map
414          * @return the descriptor to analysis object map
415          */
416         private static <DescriptorType> Map<DescriptorType, Object> findOrCreateDescriptorMap(
417                 final Map<Class<?>, Map<DescriptorType, Object>> analysisClassToDescriptorMapMap,
418                 final Map<Class<?>, ? extends IAnalysisEngine<DescriptorType, ?>> engineMap,
419                         final Class<?> analysisClass) {
420             Map<DescriptorType, Object> descriptorMap = analysisClassToDescriptorMapMap.get(analysisClass);
421             if (descriptorMap == null) {
422                 descriptorMap = createMap(engineMap, analysisClass);
423                 analysisClassToDescriptorMapMap.put(analysisClass, descriptorMap);
424             }
425             return descriptorMap;
426         }
427     
428         private static <DescriptorType> Map<DescriptorType, Object> createMap(
429                 final Map<Class<?>, ? extends IAnalysisEngine<DescriptorType, ?>> engineMap,
430                         final Class<?> analysisClass) {
431             Map<DescriptorType, Object> descriptorMap;
432             // Create a MapCache that allows the analysis engine to
433             // decide that analysis results should be retained indefinitely.
434             IAnalysisEngine<DescriptorType, ?> engine = engineMap.get(analysisClass);
435             if (analysisClass.equals(JavaClass.class)) {
436                 descriptorMap = new MapCache<DescriptorType, Object>(MAX_JAVACLASS_RESULTS_TO_CACHE);
437             } else if (analysisClass.equals(FBClassReader.class)) {
438                 descriptorMap = new MapCache<DescriptorType, Object>(MAX_FBCLASSREADER_RESULTS_TO_CACHE);
439             } else if (analysisClass.equals(ConstantPoolGen.class)) {
440                 descriptorMap = new MapCache<DescriptorType, Object>(MAX_CONSTANT_POOL_GEN_RESULTS_TO_CACHE);
441             } else if (analysisClass.equals(ClassContext.class)) {
442                 descriptorMap = new MapCache<DescriptorType, Object>(10);
443             } else if (engine instanceof IClassAnalysisEngine && ((IClassAnalysisEngine<?>) engine).canRecompute()) {
444                 descriptorMap = new MapCache<DescriptorType, Object>(MAX_CLASS_RESULTS_TO_CACHE);
445             } else {
446                 descriptorMap = new HashMap<DescriptorType, Object>();
447             }
448             return descriptorMap;
449         }

474         public <E> E getDatabase(Class<E> databaseClass, boolean optional) {
475             Object database = databaseMap.get(databaseClass);
476     
477             if (database == null) {
478                 try {
479                     // Find the database factory
480                     IDatabaseFactory<?> databaseFactory = databaseFactoryMap.get(databaseClass);
481                     if (databaseFactory == null) {
482                         if (optional) {
483                             return null;
484                         }
485                         throw new IllegalArgumentException("No database factory registered for " + databaseClass.getName());
486                     }
487     
488                     // Create the database
489                     database = databaseFactory.createDatabase();
490                 } catch (CheckedAnalysisException e) {
491                     // Error - record the analysis error
492                     database = new AbnormalAnalysisResult(e);
493                 }
494                 // FIXME: should catch and re-throw RuntimeExceptions?
495                 databaseMap.put(databaseClass, database);
496             }
497     
498             if (database instanceof AbnormalAnalysisResult) {
499                 throw new UncheckedAnalysisException("Error instantiating " + databaseClass.getName() + " database",
500                         ((AbnormalAnalysisResult) database).checkedAnalysisException);
501             }
502             return databaseClass.cast(database);
503         }
