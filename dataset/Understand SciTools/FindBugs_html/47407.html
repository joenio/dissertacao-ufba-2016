20     package edu.umd.cs.findbugs.ba.bcp;
21     
22     import edu.umd.cs.findbugs.ba.vna.ValueNumber;
23     
24     public class LocalVariable implements Variable {
25         private final ValueNumber valueNumber;
26     
27         public LocalVariable(ValueNumber valueNumber) {
28             this.valueNumber = valueNumber;
29         }
30     
31         @Override
32         public boolean sameAs(Variable other) {
33             if (!(other instanceof LocalVariable)) {
34                 return false;
35             }
36             LocalVariable otherLocal = (LocalVariable) other;
37             return valueNumber.equals(otherLocal.valueNumber);
38         }
39     
40         @Override
41         public String toString() {
42             return valueNumber.toString();
43         }
44     }
