20     package edu.umd.cs.findbugs.detect;
21     
22     import org.apache.bcel.classfile.Code;
23     import org.apache.bcel.classfile.JavaClass;
24     
25     import edu.umd.cs.findbugs.BugAccumulator;
26     import edu.umd.cs.findbugs.BugInstance;
27     import edu.umd.cs.findbugs.BugReporter;
28     import edu.umd.cs.findbugs.BytecodeScanningDetector;
29     import edu.umd.cs.findbugs.ba.ch.Subtypes2;
30     import edu.umd.cs.findbugs.internalAnnotations.DottedClassName;
31     
32     /**
33      * @author pugh
34      */
35     public class DoInsideDoPrivileged extends BytecodeScanningDetector {
36         BugAccumulator bugAccumulator;
37     
38         public DoInsideDoPrivileged(BugReporter bugReporter) {
39             this.bugAccumulator = new BugAccumulator(bugReporter);
40         }
41     
42         boolean isDoPrivileged = false;
43     
44         @Override
45         public void visit(JavaClass obj) {
46     
47             isDoPrivileged = Subtypes2.instanceOf(getDottedClassName(), "java.security.PrivilegedAction")
48                     || Subtypes2.instanceOf(getDottedClassName(), "java.security.PrivilegedExceptionAction");
49         }
50     
51         @Override
52         public void visit(Code obj) {
53             if (isDoPrivileged && "run".equals(getMethodName())) {
54                 return;
55             }
56             if (getMethod().isPrivate()) {
57                 return;
58             }
59             if (DumbMethods.isTestMethod(getMethod())) {
60                 return;
61             }
62             super.visit(obj);
63             bugAccumulator.reportAccumulatedBugs();
64         }
65     
66         @Override
67         public void sawOpcode(int seen) {
68             if (seen == INVOKEVIRTUAL && "setAccessible".equals(getNameConstantOperand())) {
69                 @DottedClassName
70                 String className = getDottedClassConstantOperand();
71                 if ("java.lang.reflect.Field".equals(className) || "java.lang.reflect.Method".equals(className)) {
72                     bugAccumulator.accumulateBug(
73                             new BugInstance(this, "DP_DO_INSIDE_DO_PRIVILEGED", LOW_PRIORITY).addClassAndMethod(this)
74                             .addCalledMethod(this), this);
75                 }
76     
77             }
78             if (seen == NEW) {
79                 @DottedClassName
80                 String classOfConstructedClass = getDottedClassConstantOperand();
81                 if (Subtypes2.instanceOf(classOfConstructedClass, "java.lang.ClassLoader")
82                         && !("main".equals(getMethodName()) && "([Ljava/lang/String;)V".equals(getMethodSig()) && getMethod()
83                                 .isStatic())) {
84                     bugAccumulator.accumulateBug(new BugInstance(this, "DP_CREATE_CLASSLOADER_INSIDE_DO_PRIVILEGED", NORMAL_PRIORITY)
85                     .addClassAndMethod(this).addClass(classOfConstructedClass), this);
86                 }
87             }
88     
89         }
90     
91     }
