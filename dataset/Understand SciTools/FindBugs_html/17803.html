19     package edu.umd.cs.findbugs;
20     
21     import java.io.IOException;
22     
23     import edu.umd.cs.findbugs.xml.XMLAttributeList;
24     import edu.umd.cs.findbugs.xml.XMLOutput;
25     
26     /**
27      * Utility methods for BugAnnotation classes.
28      *
29      * @author David Hovemeyer
30      */
31     public abstract class BugAnnotationUtil {
32         /**
33          * Write a BugAnnotation as XML.
34          *
35          * @param xmlOutput
36          *            the XMLOutput
37          * @param elementName
38          *            name of element for BugAnnotation
39          * @param annotation
40          *            the BugAnnotation
41          * @param attributeList
42          *            the XML attribute list
43          * @param addMessages
44          *            true if descriptive messages should be added
45          * @throws IOException
46          */
47         public static void writeXML(XMLOutput xmlOutput, String elementName, BugAnnotation annotation,
48                 XMLAttributeList attributeList, boolean addMessages) throws IOException {
49     
50             SourceLineAnnotation src = null;
51             if (annotation instanceof BugAnnotationWithSourceLines) {
52                 src = ((BugAnnotationWithSourceLines) annotation).getSourceLines();
53             }
54     
55             if (addMessages || src != null) {
56                 xmlOutput.openTag(elementName, attributeList);
57                 if (src != null) {
58                     src.writeXML(xmlOutput, addMessages, false);
59                 }
60                 if (addMessages) {
61                     xmlOutput.openTag(BugAnnotation.MESSAGE_TAG);
62                     xmlOutput.writeText(annotation.toString());
63                     xmlOutput.closeTag(BugAnnotation.MESSAGE_TAG);
64                 }
65                 xmlOutput.closeTag(elementName);
66             } else {
67                 xmlOutput.openCloseTag(elementName, attributeList);
68             }
69     
70         }
71     }
