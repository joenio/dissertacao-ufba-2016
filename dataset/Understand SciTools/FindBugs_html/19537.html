20     package edu.umd.cs.findbugs.util;
21     
22     /**
23      * @author pwilliam
24      */
25     public class EditDistance {
26         /**
27          *
28          */
29         private static final int INSERT_OR_DELETE_COST = 2;
30     
31         private static int minimum(int a, int b, int c) {
32             if (a > b) {
33                 return Math.min(b, c);
34             }
35             return Math.min(a, c);
36         }
37     
38         private static int distance(char a, char b) {
39             if (a == b) {
40                 return 0;
41             }
42             if (Character.toLowerCase(a) == Character.toLowerCase(b)) {
43                 return 1;
44             }
45             return 2;
46         }
47     
48         public static double editDistanceRatio(String str1, String str2) {
49             double distance = editDistance(str1, str2);
50             int maxDistance = INSERT_OR_DELETE_COST * Math.max(str1.length(), str2.length());
51             return Math.min(distance / maxDistance, distance / 4);
52         }
53     
54         public static int editDistance(String str1, String str2) {
55             int n1 = str1.length();
56             int n2 = str2.length();
57             int diff = Math.abs(n1 - n2);
58             if (diff > 6) {
59                 return INSERT_OR_DELETE_COST * Math.max(n1, n2);
60             }
61             return editDistance1(str1, str2);
62     
63         }
64     
65         public static int editDistance0(String str1, String str2) {
66             int n1 = str1.length();
67             int n2 = str2.length();
68     
69             int[][] distance = new int[n1 + 1][];
70     
71             for (int i = 0; i <= n1; i++) {
72                 distance[i] = new int[n2 + 1];
73                 distance[i][0] = INSERT_OR_DELETE_COST * i;
74             }
75             for (int j = 1; j <= n2; j++) {
76                 distance[0][j] = INSERT_OR_DELETE_COST * j;
77             }
78     
79             for (int i = 1; i <= n1; i++) {
80                 for (int j = 1; j <= n2; j++) {
81                     distance[i][j] = minimum(distance[i - 1][j] + INSERT_OR_DELETE_COST, distance[i][j - 1] + INSERT_OR_DELETE_COST,
82                             distance[i - 1][j - 1] + distance(str1.charAt(i - 1), str2.charAt(j - 1)));
83                 }
84             }
85     
86             return distance[n1][n2];
87         }
88     
89         public static int editDistance1(String str1, String str2) {
90             int n1 = str1.length();
91             int n2 = str2.length();
92     
93             int[] distance = new int[n2 + 1];
94             int[] oldDistance = new int[n2 + 1];
95     
96             for (int j = 1; j <= n2; j++) {
97                 oldDistance[j] = INSERT_OR_DELETE_COST * j;
98             }
99     
100             for (int i = 1; i <= n1; i++) {
101                 distance[0] = INSERT_OR_DELETE_COST * i;
102                 for (int j = 1; j <= n2; j++) {
103                     distance[j] = minimum(oldDistance[j] + INSERT_OR_DELETE_COST, distance[j - 1] + INSERT_OR_DELETE_COST,
104                             oldDistance[j - 1] + distance(str1.charAt(i - 1), str2.charAt(j - 1)));
105                 }
106                 int[] tmp = oldDistance;
107                 oldDistance = distance;
108                 distance = tmp;
109             }
110     
111             int result = oldDistance[n2];
112             assert result == editDistance0(str1, str2);
113             return result;
114         }
