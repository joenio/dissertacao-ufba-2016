19     package edu.umd.cs.findbugs.ba.obl;
20     
21     import java.util.Arrays;
22     import java.util.Collection;
23     
24     import org.apache.bcel.generic.ReferenceType;
25     
26     import edu.umd.cs.findbugs.ba.XMethod;
27     import edu.umd.cs.findbugs.classfile.MethodDescriptor;
28     import edu.umd.cs.findbugs.util.ExactStringMatcher;
29     import edu.umd.cs.findbugs.util.StringMatcher;
30     import edu.umd.cs.findbugs.util.SubtypeTypeMatcher;
31     import edu.umd.cs.findbugs.util.TypeMatcher;
32     
33     /**
34      * An ObligationPolicyDatabaseEntry which creates or deletes an obligation based
35      * on a call to a specified method.
36      *
37      * @author David Hovemeyer
38      */
39     public class MatchMethodEntry implements ObligationPolicyDatabaseEntry {
40     
41         private final TypeMatcher receiverType;
42     
43         private final StringMatcher methodName;
44     
45         private final StringMatcher signature;
46     
47         private final boolean isStatic;
48     
49         private final ObligationPolicyDatabaseActionType action;
50     
51         private final Obligation[] obligations;
52     
53         private final ObligationPolicyDatabaseEntryType entryType;
54     
55         @Override
56         public Collection<Obligation> getAllObligations() {
57             return Arrays.asList(obligations);
58         }
59     
60         /**
61          * Constructor. Creates an entry which matches the given XMethod.
62          *
63          * @param xmethod
64          *            an XMethod
65          * @param action
66          *            ActionType (ADD or DEL, depending on whether obligation is
67          *            added or deleted)
68          * @param entryType
69          *            entry type
70          * @param obligations
71          *            Obligation to be added or deleted
72          */
73         public MatchMethodEntry(XMethod xmethod, ObligationPolicyDatabaseActionType action,
74                 ObligationPolicyDatabaseEntryType entryType, Obligation... obligations) {
75             this(new SubtypeTypeMatcher(xmethod.getClassDescriptor()), new ExactStringMatcher(xmethod.getName()),
76                     new ExactStringMatcher(xmethod.getSignature()), xmethod.isStatic(), action, entryType, obligations);
77         }
78     
79         public MatchMethodEntry(MethodDescriptor method, ObligationPolicyDatabaseActionType action,
80                 ObligationPolicyDatabaseEntryType entryType, Obligation... obligations) {
81             this(new SubtypeTypeMatcher(method.getClassDescriptor()), new ExactStringMatcher(method.getName()),
82                     new ExactStringMatcher(method.getSignature()), method.isStatic(), action, entryType, obligations);
83         }
84     
85         /**
86          * Constructor.
87          *
88          * @param receiverType
89          *            TypeMatcher to match the receiver type (or class containing
90          *            static method)
91          * @param methodName
92          *            StringMatcher to match name of called method
93          * @param signature
94          *            StringMatcher to match signature of called method
95          * @param isStatic
96          *            true if matched method must be static, false otherwise
97          * @param action
98          *            ActionType (ADD or DEL, depending on whether obligation is
99          *            added or deleted)
100          * @param entryType
101          *            entry type
102          * @param obligations
103          *            Obligation to be added or deleted
104          */
105         public MatchMethodEntry(TypeMatcher receiverType, StringMatcher methodName, StringMatcher signature, boolean isStatic,
106                 ObligationPolicyDatabaseActionType action, ObligationPolicyDatabaseEntryType entryType, Obligation... obligations) {
107             this.receiverType = receiverType;
108             this.methodName = methodName;
109             this.signature = signature;
110             this.isStatic = isStatic;
111             this.action = action;
112             this.obligations = obligations;
113             this.entryType = entryType;
114         }
115     
116         @Override
117         public ObligationPolicyDatabaseEntryType getEntryType() {
118             return entryType;
119         }
120     
121         @Override
122         public boolean getActions(ReferenceType receiverType, String methodName, String signature, boolean isStatic,
123                 Collection<ObligationPolicyDatabaseAction> actionList) {
124             if (this.methodName.matches(methodName) && this.signature.matches(signature) && this.isStatic == isStatic
125                     && this.receiverType.matches(receiverType)) {
126                 for (Obligation o : obligations) {
127                     actionList.add(new ObligationPolicyDatabaseAction(action, o));
128                 }
129                 return true;
130             }
131             return false;
132         }
133     
134         @Override
135         public String toString() {
136             return "(" + receiverType + "," + methodName + "," + signature + "," + isStatic + "," + action + ","
137                     + Arrays.asList(obligations) + "," + entryType + ")";
138         }
139     }
