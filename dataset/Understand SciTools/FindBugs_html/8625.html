198         public BugCollection mergeCollections(BugCollection origCollection, BugCollection newCollection, boolean copyDeadBugs,
199                 boolean incrementalAnalysis) {
200     
201             for (BugInstance b : newCollection) {
202                 if (b.isDead()) {
203                     throw new IllegalArgumentException("Can't merge bug collections if the newer collection contains dead bugs: " + b);
204                 }
205             }
206     
207             mapFromNewToOldBug.clear();
208     
209             matchedOldBugs.clear();
210             BugCollection resultCollection = newCollection.createEmptyCollectionWithMetadata();
211             // Previous sequence number
212             long lastSequence = origCollection.getSequenceNumber();
213             // The AppVersion history is retained from the orig collection,
214             // adding an entry for the sequence/timestamp of the current state
215             // of the orig collection.
216             resultCollection.clearAppVersions();
217             for (Iterator<AppVersion> i = origCollection.appVersionIterator(); i.hasNext();) {
218                 AppVersion appVersion = i.next();
219                 resultCollection.addAppVersion((AppVersion) appVersion.clone());
220             }
221             AppVersion origCollectionVersion = origCollection.getCurrentAppVersion();
222             AppVersion origCollectionVersionClone = new AppVersion(lastSequence);
223             origCollectionVersionClone.setTimestamp(origCollectionVersion.getTimestamp());
224             origCollectionVersionClone.setReleaseName(origCollectionVersion.getReleaseName());
225             origCollectionVersionClone.setNumClasses(origCollection.getProjectStats().getNumClasses());
226             origCollectionVersionClone.setCodeSize(origCollection.getProjectStats().getCodeSize());
227     
228             resultCollection.addAppVersion(origCollectionVersionClone);
229     
230             // We assign a sequence number to the new collection as one greater than
231             // the original collection.
232             long currentSequence = origCollection.getSequenceNumber() + 1;
233             resultCollection.setSequenceNumber(currentSequence);
234     
235             //        int oldBugs = 0;
236             matchBugs(origCollection, newCollection);
237     
238             if (sloppyMatch) {
239                 matchBugs(new SloppyBugComparator(), origCollection, newCollection);
240             }
241     
242             //        int newlyDeadBugs = 0;
243             //        int persistantBugs = 0;
244             //        int addedBugs = 0;
245             //        int addedInNewCode = 0;
246             //        int deadBugInDeadCode = 0;
247     
248             HashSet<String> analyzedSourceFiles = sourceFilesInCollection(newCollection);
249             // Copy unmatched bugs
250             if (copyDeadBugs || incrementalAnalysis) {
251                 for (BugInstance bug : origCollection.getCollection()) {
252                     if (!matchedOldBugs.containsKey(bug)) {
253                         if (bug.isDead()) {
254                             //                        oldBugs++;
255                             BugInstance newBug = (BugInstance) bug.clone();
256                             resultCollection.add(newBug, false);
257                         } else {
258                             //                        newlyDeadBugs++;
259     
260                             BugInstance newBug = (BugInstance) bug.clone();
261     
262                             ClassAnnotation classBugFoundIn = bug.getPrimaryClass();
263                             String className = classBugFoundIn.getClassName();
264                             String sourceFile = classBugFoundIn.getSourceFileName();
265                             boolean fixed = sourceFile != null && analyzedSourceFiles.contains(sourceFile)
266                                     || newCollection.getProjectStats().getClassStats(className) != null;
267                             if (fixed) {
268                                 if (!copyDeadBugs) {
269                                     continue;
270                                 }
271                                 newBug.setRemovedByChangeOfPersistingClass(true);
272                                 newBug.setLastVersion(lastSequence);
273                             } else {
274                                 //                            deadBugInDeadCode++;
275                                 if (!incrementalAnalysis) {
276                                     newBug.setLastVersion(lastSequence);
277                                 }
278                             }
279     
280                             if (newBug.isDead() && newBug.getFirstVersion() > newBug.getLastVersion()) {
281                                 throw new IllegalStateException("Illegal Version range: " + newBug.getFirstVersion() + ".."
282                                         + newBug.getLastVersion());
283                             }
284                             resultCollection.add(newBug, false);
285                         }
286                     }
287                 }
288             }
289             // Copy matched bugs
290             for (BugInstance bug : newCollection.getCollection()) {
291                 BugInstance newBug = (BugInstance) bug.clone();
292                 if (mapFromNewToOldBug.containsKey(bug)) {
293                     BugInstance origWarning = mapFromNewToOldBug.get(bug);
294     
295                     mergeBugHistory(origWarning, newBug);
296                     // handle getAnnotationText()/setAnnotationText() and
297                     // designation key
298                     BugDesignation designation = newBug.getUserDesignation();
299                     if (designation != null) {
300                         designation.merge(origWarning.getUserDesignation());
301                     }
302                     else {
303                         newBug.setUserDesignation(origWarning.getUserDesignation()); // clone??
304                     }
305     
306                     //                persistantBugs++;
307                 } else {
308                     newBug.setFirstVersion(lastSequence + 1);
309                     //                addedBugs++;
310     
311                     ClassAnnotation classBugFoundIn = bug.getPrimaryClass();
312     
313                     String className = classBugFoundIn.getClassName();
314                     if (origCollection.getProjectStats().getClassStats(className) != null) {
315                         newBug.setIntroducedByChangeOfExistingClass(true);
316                         // System.out.println("added bug to existing code " +
317                         // newBug.getUniqueId() + " : " + newBug.getAbbrev() + " in
318                         // " + classBugFoundIn);
319                     } else {
320                         //                    addedInNewCode++;
321                     }
322                 }
323                 if (newBug.isDead()) {
324                     throw new IllegalStateException("Illegal Version range: " + newBug.getFirstVersion() + ".."
325                             + newBug.getLastVersion());
326                 }
327                 int oldSize = resultCollection.getCollection().size();
328                 resultCollection.add(newBug, false);
329                 int newSize = resultCollection.getCollection().size();
330                 if (newSize != oldSize + 1) {
331                     System.out.println("Failed to add bug" + newBug.getMessage());
332                 }
333             }
334             /*
335             if (false && verbose) {
336                 System.out.println(origCollection.getCollection().size() + " orig bugs, " + newCollection.getCollection().size()
337                         + " new bugs");
338                 System.out.println("Bugs: " + oldBugs + " old, " + deadBugInDeadCode + " in removed code, "
339                         + (newlyDeadBugs - deadBugInDeadCode) + " died, " + persistantBugs + " persist, " + addedInNewCode
340                         + " in new code, " + (addedBugs - addedInNewCode) + " added");
341                 System.out.println(resultCollection.getCollection().size() + " resulting bugs");
342             }
343              */
344             return resultCollection;
345     
346         }

374         private void matchBugs(BugCollection origCollection, BugCollection newCollection) {
375             matchBugs(SortedBugCollection.BugInstanceComparator.instance, origCollection, newCollection);
376     
377             mapFromNewToOldBug.clear();
378             matchedOldBugs.clear();
379     
380             matchBugs(versionInsensitiveBugComparator, origCollection, newCollection);
381             matchBugs(versionInsensitiveBugComparator, origCollection, newCollection, MatchOldBugs.IF_CLASS_NOT_SEEN_UNTIL_NOW);
382             if (doMatchFixedBugs) {
383                 matchBugs(versionInsensitiveBugComparator, origCollection, newCollection, MatchOldBugs.ALWAYS);
384             }
385     
386             if (!preciseMatch) {
387                 matchBugs(fuzzyBugPatternMatcher, origCollection, newCollection);
388             }
389     
390             if (!noPackageMoves) {
391                 VersionInsensitiveBugComparator movedBugComparator = new VersionInsensitiveBugComparator();
392                 MovedClassMap movedClassMap = new MovedClassMap(origCollection, newCollection).execute();
393                 if (!movedClassMap.isEmpty()) {
394                     movedBugComparator.setClassNameRewriter(movedClassMap);
395                     movedBugComparator.setComparePriorities(precisePriorityMatch);
396                     matchBugs(movedBugComparator, origCollection, newCollection);
397                     if (!preciseMatch) {
398                         movedBugComparator.setExactBugPatternMatch(false);
399                         matchBugs(movedBugComparator, origCollection, newCollection);
400                     }
401                 }
402                 /*
403                 if (false) {
404                     System.out.println("Matched old bugs: " + matchedOldBugs.size());
405                 }
406                  */
407     
408             }
409         }

423         public void doit(String[] args) throws IOException, DocumentException {
424     
425             DetectorFactoryCollection.instance();
426             UpdateCommandLine commandLine = new UpdateCommandLine();
427             int argCount = commandLine.parse(args, 1, Integer.MAX_VALUE, USAGE);
428     
429             if (commandLine.outputFilename == null) {
430                 verbose = false;
431             }
432             if (mostRecent > 0) {
433                 argCount = Math.max(argCount, args.length - mostRecent);
434             }
435     
436             String[] firstPathParts = getFilePathParts(args[argCount]);
437             int commonPrefix = firstPathParts.length;
438             for (int i = argCount + 1; i <= (args.length - 1); i++) {
439     
440                 commonPrefix = Math.min(commonPrefix, lengthCommonPrefix(firstPathParts, getFilePathParts(args[i])));
441             }
442     
443             String origFilename = args[argCount++];
444             BugCollection origCollection;
445             origCollection = new SortedBugCollection();
446             if (verbose) {
447                 System.out.println("Starting with " + origFilename);
448             }
449     
450             while (true) {
451                 try {
452                     while (true) {
453                         File f = new File(origFilename);
454                         if (f.length() > 0) {
455                             break;
456                         }
457                         if (verbose) {
458                             System.out.println("Empty input file: " + f);
459                         }
460                         origFilename = args[argCount++];
461                     }
462                     origCollection.readXML(origFilename);
463                     break;
464                 } catch (Exception e) {
465                     if (verbose) {
466                         System.out.println("Error reading " + origFilename);
467                         e.printStackTrace(System.out);
468                     }
469                     origFilename = args[argCount++];
470                 }
471             }
472     
473             if (commandLine.overrideRevisionNames || origCollection.getReleaseName() == null
474                     || origCollection.getReleaseName().length() == 0) {
475     
476                 if (commonPrefix >= firstPathParts.length) {
477                     // This should only happen if either
478                     //
479                     // (1) there is only one input file, or
480                     // (2) all of the input files have the same name
481                     //
482                     // In either case, make the release name the same
483                     // as the file part of the input file(s).
484                     commonPrefix = firstPathParts.length - 1;
485                 }
486     
487                 origCollection.setReleaseName(firstPathParts[commonPrefix]);
488                 if (useAnalysisTimes) {
489                     origCollection.setTimestamp(origCollection.getAnalysisTimestamp());
490                 }
491             }
492     
493             for (BugInstance bug : origCollection.getCollection()) {
494                 if (bug.getLastVersion() >= 0 && bug.getFirstVersion() > bug.getLastVersion()) {
495                     throw new IllegalStateException("Illegal Version range: " + bug.getFirstVersion() + ".." + bug.getLastVersion());
496                 }
497             }
498     
499             discardUnwantedBugs(origCollection);
500     
501             while (argCount <= (args.length - 1)) {
502     
503                 BugCollection newCollection = new SortedBugCollection();
504     
505                 String newFilename = args[argCount++];
506                 if (verbose) {
507                     System.out.println("Merging " + newFilename);
508                 }
509                 try {
510                     File f = new File(newFilename);
511                     if (f.length() == 0) {
512                         if (verbose) {
513                             System.out.println("Empty input file: " + f);
514                         }
515                         continue;
516                     }
517                     newCollection.readXML(newFilename);
518     
519                     if (commandLine.overrideRevisionNames || newCollection.getReleaseName() == null
520                             || newCollection.getReleaseName().length() == 0) {
521                         newCollection.setReleaseName(getFilePathParts(newFilename)[commonPrefix]);
522                     }
523                     if (useAnalysisTimes) {
524                         newCollection.setTimestamp(newCollection.getAnalysisTimestamp());
525                     }
526                     discardUnwantedBugs(newCollection);
527     
528                     origCollection = mergeCollections(origCollection, newCollection, true, false);
529                 } catch (IOException e) {
530                     IOException e2 = new IOException("Error parsing " + newFilename);
531                     e2.initCause(e);
532                     if (verbose) {
533                         e2.printStackTrace();
534                     }
535                     throw e2;
536                 } catch (DocumentException e) {
537                     DocumentException e2 = new DocumentException("Error parsing " + newFilename);
538                     e2.initCause(e);
539                     if (verbose) {
540                         e2.printStackTrace();
541                     }
542                     throw e2;
543                 }
544             }
545             /*
546             if (false) {
547                 for (Iterator<BugInstance> i = origCollection.iterator(); i.hasNext();) {
548                     if (!resurrected.contains(i.next().getInstanceKey())) {
549                         i.remove();
550                     }
551                 }
552             }
553              */
554             origCollection.setWithMessages(commandLine.withMessages);
555             if (commandLine.outputFilename != null) {
556                 if (verbose) {
557                     System.out.println("Writing " + commandLine.outputFilename);
558                 }
559                 origCollection.writeXML(commandLine.outputFilename);
560             } else {
561                 origCollection.writeXML(System.out);
562             }
563     
564         }

566         private static int lengthCommonPrefix(String[] string, String[] string2) {
567             int maxLength = Math.min(string.length, string2.length);
568             for (int result = 0; result < maxLength; result++) {
569                 if (!string[result].equals(string2[result])) {
570                     return result;
571                 }
572             }
573             return maxLength;
574         }

583         enum MatchOldBugs {
584             IF_LIVE, IF_CLASS_NOT_SEEN_UNTIL_NOW, ALWAYS;
585             boolean match(BugInstance b) {
586                 switch (this) {
587                 case ALWAYS:
588                     return true;
589                 case IF_CLASS_NOT_SEEN_UNTIL_NOW:
590                     return !b.isDead() || b.isRemovedByChangeOfPersistingClass();
591                 case IF_LIVE:
592                     return !b.isDead();
593                 }
594                 throw new IllegalStateException();
595             }
596         }

598         private void matchBugs(Comparator<BugInstance> bugInstanceComparator, BugCollection origCollection,
599                 BugCollection newCollection) {
600             matchBugs(bugInstanceComparator, origCollection, newCollection, MatchOldBugs.IF_LIVE);
601     
602         }

604         private void matchBugs(Comparator<BugInstance> bugInstanceComparator, BugCollection origCollection,
605                 BugCollection newCollection, MatchOldBugs matchOld) {
606     
607             TreeMap<BugInstance, LinkedList<BugInstance>> set = new TreeMap<BugInstance, LinkedList<BugInstance>>(
608                     bugInstanceComparator);
609             //        int oldBugs = 0;
610             //        int newBugs = 0;
611             //        int matchedBugs = 0;
612             for (BugInstance bug : origCollection.getCollection()) {
613                 if (!matchedOldBugs.containsKey(bug)) {
614                     if (matchOld.match(bug)) {
615                         //                    oldBugs++;
616                         LinkedList<BugInstance> q = set.get(bug);
617                         if (q == null) {
618                             q = new LinkedList<BugInstance>();
619                             set.put(bug, q);
620                         }
621                         q.add(bug);
622                     }
623     
624                 }
625             }
626             long newVersion = origCollection.getCurrentAppVersion().getSequenceNumber() + 1;
627             for (BugInstance bug : newCollection.getCollection()) {
628                 if (!mapFromNewToOldBug.containsKey(bug)) {
629                     //                newBugs++;
630                     LinkedList<BugInstance> q = set.get(bug);
631                     if (q == null) {
632                         continue;
633                     }
634                     for (Iterator<BugInstance> i = q.iterator(); i.hasNext();) {
635                         BugInstance matchedBug = i.next();
636     
637                         if (matchedBug.isDead()) {
638                             if (noResurrections || matchedBug.isRemovedByChangeOfPersistingClass()
639                                     && newVersion - matchedBug.getLastVersion() > maxResurrection) {
640                                 continue;
641                             }
642                             resurrected.add(bug.getInstanceKey());
643                             // System.out.println("in version " +
644                             // newCollection.getReleaseName());
645                             // System.out.println("  resurrected " +
646                             // bug.getMessageWithoutPrefix());
647                         }
648     
649                         //                    matchedBugs++;
650                         mapFromNewToOldBug.put(bug, matchedBug);
651                         matchedOldBugs.put(matchedBug, null);
652                         i.remove();
653                         if (q.isEmpty()) {
654                             set.remove(bug);
655                         }
656                         break;
657                     }
658                 }
659             }
660         }
