48     /**
49      * Find relevant type qualifiers needing to be checked for a given method.
50      *
51      * @author William Pugh
52      */
53     public class Analysis {
54         private static final boolean DEBUG = SystemProperties.getBoolean("ctq.debug.analysis");
55     
56         /**
57          * This system property enables additional work to try to detect all
58          * *effective* type qualifiers (direct, inherited, and default) applied to
59          * methods and called methods.
60          *
61          * This step uses an interprocedural call graph.
62          */
63         public static final boolean FIND_EFFECTIVE_RELEVANT_QUALIFIERS = true; // SystemProperties.getBoolean("ctq.findeffective");
64     
65         public static final boolean DEBUG_FIND_EFFECTIVE_RELEVANT_QUALIFIERS = FIND_EFFECTIVE_RELEVANT_QUALIFIERS
66                 && SystemProperties.getBoolean("ctq.findeffective.debug");
67     
68         /**
69          * Find relevant type qualifiers needing to be checked for a given method.
70          *
71          * @param methodDescriptor
72          *            a method
73          * @return Collection of relevant type qualifiers needing to be checked
74          * @throws CheckedAnalysisException
75          */
76         public static Collection<TypeQualifierValue<?>> getRelevantTypeQualifiers(MethodDescriptor methodDescriptor, CFG cfg)
77                 throws CheckedAnalysisException {
78     
79             final HashSet<TypeQualifierValue<?>> result = new HashSet<TypeQualifierValue<?>>();
80     
81             XMethod xmethod = XFactory.createXMethod(methodDescriptor);
82     
83             if (FIND_EFFECTIVE_RELEVANT_QUALIFIERS) {
84                 if (DEBUG_FIND_EFFECTIVE_RELEVANT_QUALIFIERS) {
85                     System.out.println("**** Finding effective type qualifiers for " + xmethod);
86                 }
87     
88                 //
89                 // This will take care of methods using fields annotated with
90                 // a type qualifier.
91                 //
92                 getDirectlyRelevantTypeQualifiers(xmethod, result);
93     
94                 // For all known type qualifiers, find the effective (direct,
95                 // inherited,
96                 // or default) type qualifier annotations
97                 // on the method and all methods directly called by the method.
98                 //
99     
100                 addEffectiveRelevantQualifiers(result, xmethod);
101     
102                 IAnalysisCache analysisCache = Global.getAnalysisCache();
103                 ConstantPoolGen cpg = analysisCache.getClassAnalysis(ConstantPoolGen.class, methodDescriptor.getClassDescriptor());
104                 for (Iterator<Location> i = cfg.locationIterator(); i.hasNext();) {
105                     Location location = i.next();
106                     Instruction ins = location.getHandle().getInstruction();
107                     if (ins instanceof InvokeInstruction) {
108                         XMethod called = XFactory.createXMethod((InvokeInstruction) ins, cpg);
109                         addEffectiveRelevantQualifiers(result, called);
110                     }
111     
112                     if (DEBUG_FIND_EFFECTIVE_RELEVANT_QUALIFIERS) {
113                         System.out.println("===> result: " + result);
114                     }
115                 }
116     
117                 //
118                 // XXX: this code can go away eventually
119                 //
120     
121                 if (!methodDescriptor.isStatic()) {
122                     // Instance method - must consider type qualifiers inherited
123                     // from superclasses
124     
125                     SupertypeTraversalVisitor visitor = new OverriddenMethodsVisitor(xmethod) {
126                         /*
127                          * (non-Javadoc)
128                          *
129                          * @see edu.umd.cs.findbugs.ba.ch.OverriddenMethodsVisitor#
130                          * visitOverriddenMethod(edu.umd.cs.findbugs.ba.XMethod)
131                          */
132     
133                         @Override
134                         protected boolean visitOverriddenMethod(XMethod xmethod) {
135                             getDirectlyRelevantTypeQualifiers(xmethod, result);
136                             return true;
137                         }
138                     };
139     
140                     try {
141                         AnalysisContext.currentAnalysisContext().getSubtypes2()
142                         .traverseSupertypesDepthFirst(xmethod.getClassDescriptor(), visitor);
143                     } catch (ClassNotFoundException e) {
144                         AnalysisContext.currentAnalysisContext().getLookupFailureCallback().reportMissingClass(e);
145                         return Collections.<TypeQualifierValue<?>> emptySet();
146                     } catch (UncheckedAnalysisException e) {
147                         AnalysisContext.currentAnalysisContext().getLookupFailureCallback()
148                         .logError("Error getting relevant type qualifiers for " + xmethod.toString(), e);
149                         return Collections.<TypeQualifierValue<?>> emptySet();
150                     }
151                 }
152             }
153     
154             return result;
155     
156         }
157     
158         private static void addEffectiveRelevantQualifiers(HashSet<TypeQualifierValue<?>> result, XMethod xmethod) {
159             if (DEBUG_FIND_EFFECTIVE_RELEVANT_QUALIFIERS) {
160                 System.out.println("  Finding effective qualifiers for " + xmethod);
161             }
162     
163             for (TypeQualifierValue<?> tqv : TypeQualifierValue.getAllKnownTypeQualifiers()) {
164                 if (DEBUG_FIND_EFFECTIVE_RELEVANT_QUALIFIERS) {
165                     System.out.print("    " + tqv + "...");
166                 }
167     
168                 TypeQualifierAnnotation tqa;
169                 boolean add = false;
170     
171                 tqa = TypeQualifierApplications.getEffectiveTypeQualifierAnnotation(xmethod, tqv);
172                 if (tqa != null) {
173                     add = true;
174                 }
175     
176                 if (!add) {
177                     int numParams = xmethod.getNumParams();
178                     for (int i = 0; i < numParams; i++) {
179                         tqa = TypeQualifierApplications.getEffectiveTypeQualifierAnnotation(xmethod, i, tqv);
180                         if (tqa != null) {
181                             add = true;
182                             break;
183                         }
184                     }
185                 }
186     
187                 if (add) {
188                     result.add(tqv);
189                 }
190     
191                 if (DEBUG_FIND_EFFECTIVE_RELEVANT_QUALIFIERS) {
192                     System.out.println(add ? "YES" : "NO");
193                 }
194             }
195         }

201         private static void getDirectlyRelevantTypeQualifiers(XMethod xmethod, HashSet<TypeQualifierValue<?>> result) {
202             result.addAll(AnalysisContext.currentAnalysisContext().getDirectlyRelevantTypeQualifiersDatabase()
203                     .getDirectlyRelevantTypeQualifiers(xmethod.getMethodDescriptor()));
204     
205         }
206     
207         /**
208          * @param result
209          * @param m
210          */
211         public static void addKnownTypeQualifiersForParameters(HashSet<? super TypeQualifierValue<?>> result, XMethod m) {
212             int numParameters = new SignatureParser(m.getSignature()).getNumParameters();
213             for (int p = 0; p < numParameters; p++) {
214                 addKnownTypeQualifiers(result, TypeQualifierApplications.getApplicableApplications(m, p));
215             }
216         }
217     
218         /**
219          * @param result
220          * @param applicableApplications
221          */
222         public static void addKnownTypeQualifiers(HashSet<? super TypeQualifierValue<?>> result,
223                 Collection<TypeQualifierAnnotation> applicableApplications) {
224             for (TypeQualifierAnnotation t : applicableApplications) {
225                 if (t.when != When.UNKNOWN) {
226                     result.add(t.typeQualifier);
227                 }
228             }
229         }
