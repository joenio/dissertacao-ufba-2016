125         @After
126         public void checkForUnexpectedBugs() {
127             List<BugInstance> unexpectedBugs = new ArrayList<BugInstance>();
128             for (BugInstance bug : bugReporter.getBugCollection()) {
129                 if (isUnexpectedBug(bug) && bug.getPriority() == Priorities.HIGH_PRIORITY) {
130                     unexpectedBugs.add(bug);
131                     System.out.println(bug.getMessageWithPriorityTypeAbbreviation());
132                     System.out.println("  " + bug.getPrimarySourceLineAnnotation());
133                 }
134             }
135     
136             if (!unexpectedBugs.isEmpty()) {
137                 Assert.fail("Unexpected bugs (" + unexpectedBugs.size() + "):" + getBugsLocations(unexpectedBugs));
138             }
139         }
140     
141         /**
142          * Returns a printable String concatenating bug locations.
143          */
144         private static String getBugsLocations(List<BugInstance> unexpectedBugs) {
145             StringBuilder message = new StringBuilder();
146             for (BugInstance bugInstance : unexpectedBugs) {
147                 message.append("\n");
148                 if (bugInstance.getBugPattern().getType().equals(FB_MISSING_EXPECTED_WARNING)) {
149                     message.append("missing ");
150                 } else {
151                     message.append("unexpected ");
152                 }
153                 StringAnnotation pattern = (StringAnnotation) bugInstance.getAnnotations().get(2);
154                 message.append(pattern.getValue());
155                 message.append(" ");
156                 message.append(bugInstance.getPrimarySourceLineAnnotation());
157             }
158             return message.toString();
159         }
160     
161         /**
162          * Returns if a bug instance is unexpected for this test.
163          */
164         private static boolean isUnexpectedBug(BugInstance bug) {
165             return FB_MISSING_EXPECTED_WARNING.equals(bug.getType()) || FB_UNEXPECTED_WARNING.equals(bug.getType());
166         }

168         /**
169          * Loads the default detectors from findbugs.jar, to isolate the test from
170          * others that use fake plugins.
171          */
172         private static void loadFindbugsPlugin() {
173             DetectorFactoryCollection dfc = new DetectorFactoryCollection();
174             DetectorFactoryCollection.resetInstance(dfc);
175         }
176     
177         /**
178          * Sets up a FB engine to run on the 'findbugsTestCases' project. It enables
179          * all the available detectors and reports all the bug categories. Uses a
180          * low priority threshold.
181          */
182         private void setUpEngine(String... analyzeMe) throws IOException {
183             this.engine = new FindBugs2();
184             Project project = new Project();
185             project.setProjectName("findbugsTestCases");
186             this.engine.setProject(project);
187     
188             DetectorFactoryCollection detectorFactoryCollection = DetectorFactoryCollection.instance();
189             engine.setDetectorFactoryCollection(detectorFactoryCollection);
190     
191             bugReporter = new BugCollectionBugReporter(project);
192             bugReporter.setPriorityThreshold(Priorities.LOW_PRIORITY);
193             bugReporter.setRankThreshold(BugRanker.VISIBLE_RANK_MAX);
194     
195             engine.setBugReporter(this.bugReporter);
196             UserPreferences preferences = UserPreferences.createDefaultUserPreferences();
197             for (String factory : enabledDetectors) {
198                 DetectorFactory detFactory = detectorFactoryCollection.getFactory(factory);
199                 preferences.enableDetector(detFactory, true);
200             }
201             preferences.getFilterSettings().clearAllCategories();
202             this.engine.setUserPreferences(preferences);
203     
204             for (String s : analyzeMe) {
205                 project.addFile(getFindbugsTestCasesFile(s).getPath());
206                 if(s.indexOf("Eclipse") >= 0){
207                     // TODO see testAllRegressionFilesEcj() comments
208                     engine.setNoClassOk(true);
209                 }
210             }
211     
212             project.addAuxClasspathEntry("lib/junit.jar");
213             File lib = getFindbugsTestCasesFile("lib");
214             for(File f : lib.listFiles()) {
215                 String path = f.getPath();
216                 if (f.canRead() && path.endsWith(".jar")) {
217                     project.addAuxClasspathEntry(path);
218                 }
219             }
220     
221         }
