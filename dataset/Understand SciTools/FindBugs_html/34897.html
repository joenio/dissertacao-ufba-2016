47     /**
48      * Match a ByteCodePattern against the code of a method, represented by a CFG.
49      * Produces some number of ByteCodePatternMatch objects, which indicate how the
50      * pattern matched the bytecode instructions in the method.
51      * <p/>
52      * <p>
53      * This code is a hack and should probably be rewritten.
54      *
55      * @author David Hovemeyer
56      * @see ByteCodePattern
57      */

81         /**
82          * Constructor.
83          *
84          * @param pattern
85          *            the ByteCodePattern to look for examples of
86          * @param classContext
87          *            ClassContext for the class to analyze
88          * @param method
89          *            the Method to analyze
90          */
91         public PatternMatcher(ByteCodePattern pattern, ClassContext classContext, Method method) throws CFGBuilderException,
92         DataflowAnalysisException {
93             this.pattern = pattern;
94             this.cfg = classContext.getCFG(method);
95             this.cpg = classContext.getConstantPoolGen();
96             this.dfs = classContext.getDepthFirstSearch(method);
97             this.vnaDataflow = classContext.getValueNumberDataflow(method);
98             this.domAnalysis = classContext.getNonExceptionDominatorsAnalysis(method);
99             this.workList = new LinkedList<BasicBlock>();
100             this.visitedBlockMap = new IdentityHashMap<BasicBlock, BasicBlock>();
101             this.resultList = new LinkedList<ByteCodePatternMatch>();
102         }
103     
104         /**
105          * Search for examples of the ByteCodePattern.
106          *
107          * @return this object
108          * @throws DataflowAnalysisException
109          *             if the ValueNumberAnalysis did not produce useful values for
110          *             the method
111          */
112         public PatternMatcher execute() throws DataflowAnalysisException {
113             workList.addLast(cfg.getEntry());
114     
115             while (!workList.isEmpty()) {
116                 BasicBlock basicBlock = workList.removeLast();
117                 visitedBlockMap.put(basicBlock, basicBlock);
118     
119                 // Scan instructions of basic block for possible matches
120                 BasicBlock.InstructionIterator i = basicBlock.instructionIterator();
121                 while (i.hasNext()) {
122                     attemptMatch(basicBlock, i.duplicate());
123                     i.next();
124                 }
125     
126                 // Add successors of the basic block (which haven't been visited
127                 // already)
128                 Iterator<BasicBlock> succIterator = cfg.successorIterator(basicBlock);
129                 while (succIterator.hasNext()) {
130                     BasicBlock succ = succIterator.next();
131                     if (visitedBlockMap.get(succ) == null) {
132                         workList.addLast(succ);
133                     }
134                 }
135             }
136     
137             return this;
138         }

141          * Return an Iterator over the ByteCodePatternMatch objects representing
142          * successful matches of the ByteCodePattern.
143          */
144         public Iterator<ByteCodePatternMatch> byteCodePatternMatchIterator() {
145             return resultList.iterator();

152          *            the basic block
153          * @param instructionIterator

155          *            instruction to be matched

166          * Object representing the current state of the matching algorithm. Provides
167          * convenient methods to implement the various steps of the algorithm.
168          */
169         private class State {

171     
172             private final BasicBlock.InstructionIterator instructionIterator;

193             /**
194              * Constructor. Builds the start state.
195              *
196              * @param basicBlock
197              *            the initial basic block

201              *            the first PatternElement of the pattern

207             /**
208              * Constructor.
209              */
210             public State(@Nullable State parent, BasicBlock basicBlock, BasicBlock.InstructionIterator instructionIterator,
211                     PatternElement patternElement, int matchCount, @Nullable PatternElementMatch currentMatch,
212                     @Nullable BindingSet bindingSet, boolean canFork) {
213                 this.basicBlock = basicBlock;
214                 this.instructionIterator = instructionIterator;
215                 this.patternElement = patternElement;
216                 this.matchCount = matchCount;
217                 this.currentMatch = currentMatch;
218                 this.bindingSet = bindingSet;
219                 this.canFork = canFork;
220                 this.parentPath = (parent != null) ? parent.path : -1;
221                 this.path = nextPath++;

261             /**
262              * Get a ByteCodePatternMatch representing the complete match.
263              */
264             public ByteCodePatternMatch getResult() {
265                 if (!isComplete()) {
266                     throw new IllegalStateException("match not complete!");
267                 }
268                 return new ByteCodePatternMatch(bindingSet, currentMatch);

271             /**
272              * Try to produce a new state that will finish matching the current
273              * element and start matching the next element. Returns null if the
274              * current element is not complete.
275              */
276             public State advanceToNextElement() {
277                 if (!canFork || matchCount < patternElement.minOccur()) {
278                     // Current element is not complete, or we already
279                     // forked at this point
280                     return null;
281                 }
282     
283                 // Create state to advance to matching next pattern element
284                 // at current basic block and instruction.
285                 State advance = new State(this, basicBlock, instructionIterator.duplicate(), patternElement.getNext(), 0,
286                         currentMatch, bindingSet, true);
287     
288                 // Now that this state has forked from this element
289                 // at this instruction, it must not do so again.
290                 this.canFork = false;
291     
292                 return advance;
293             }

295             /**
296              * Determine if the current pattern element can continue to match
297              * instructions.
298              */
299             public boolean currentElementCanContinue() {
300                 return matchCount < patternElement.maxOccur();
301             }

303             /**
304              * Determine if there are more instructions in the same basic block.
305              */
306             public boolean moreInstructionsInBasicBlock() {
307                 return instructionIterator.hasNext();
308             }

310             /**
311              * Match current pattern element with next instruction in basic block.
312              * Returns MatchResult if match succeeds, null otherwise.
313              */
314             public MatchResult matchNextInBasicBlock() throws DataflowAnalysisException {
315                 if (!moreInstructionsInBasicBlock()) {
316                     throw new IllegalStateException("At end of BB!");
317                 }
318     
319                 // Move to location of next instruction to be matched
320                 Location location = new Location(instructionIterator.next(), basicBlock);
321                 return matchLocation(location);
322             }

324             /**
325              * Determine if it is possible to continue matching in a successor basic
326              * block.
327              */
328             public boolean canAdvanceToNextBasicBlock() {
329                 return currentMatch == null || currentMatch.allowTrailingEdges();

333              * Get most recently matched instruction.

342             /**
343              * Return a new State for continuing the overall pattern match in a
344              * successor basic block.
345              *
346              * @param edge
347              *            the Edge leading to the successor basic block
348              * @param matchResult
349              *            a MatchResult representing the match of the last
350              *            instruction in the predecessor block; null if none
351              */
352             public State advanceToSuccessor(Edge edge, MatchResult matchResult) {
353                 // If we have just matched an instruction, then we allow the
354                 // matching PatternElement to choose which edges are acceptable.
355                 // This allows PatternElements to select particular control edges;
356                 // for example, only continue the pattern on the true branch
357                 // of an "if" comparison.
358                 if (matchResult != null && !matchResult.getPatternElement().acceptBranch(edge, getLastMatchedInstruction())) {
359                     return null;
360                 }
361     
362                 return new State(this, edge.getTarget(), edge.getTarget().instructionIterator(), patternElement, matchCount,
363                         currentMatch, bindingSet, canFork);
364             }

366             /**
367              * Determine if we need to look for a dominated instruction at this
368              * point in the search.
369              */
370             public boolean lookForDominatedInstruction() {
371                 return patternElement.getDominatedBy() != null && matchCount == 0;
372             }
373     
374             /**
375              * Return Iterator over states representing dominated instructions that
376              * continue the match.
377              */
378             public Iterable<State> dominatedInstructionStateIterable() throws DataflowAnalysisException {
379                 if (!lookForDominatedInstruction()) {
380                     throw new IllegalStateException();
381                 }
382                 LinkedList<State> stateList = new LinkedList<State>();
383     
384                 State dup = this.duplicate();
385     
386                 if (currentMatch != null) {
387                     // Find the referenced instruction.
388                     PatternElementMatch dominator = currentMatch.getFirstLabeledMatch(patternElement.getDominatedBy());
389                     BasicBlock domBlock = dominator.getBasicBlock();
390                     InstructionHandle domInstruction = dominator.getMatchedInstructionInstructionHandle();
391                     // Find all basic blocks dominated by the dominator block.
392                     for (Iterator<BasicBlock> i = cfg.blockIterator(); i.hasNext();) {
393                         BasicBlock block = i.next();
394                         boolean includeInstructions = block != domBlock;
395     
396                         BitSet dominators = domAnalysis.getResultFact(block);
397                         if (block == domBlock || dominators.get(domBlock.getLabel())) {
398                             // This block is dominated by the dominator block.
399                             // Each instruction in the block which matches the
400                             // current pattern
401                             // element is a new state continuing the match.
402                             for (Iterator<InstructionHandle> j = block.instructionIterator(); j.hasNext();) {
403                                 InstructionHandle next = j.next();
404                                 if (includeInstructions) {
405                                     MatchResult matchResult = dup.matchLocation(new Location(next, block));
406                                     if (matchResult != null) {
407                                         stateList.add(dup);
408                                         dup = this.duplicate();
409                                     }
410                                 } else if (next.equals(domInstruction)) {
411                                     includeInstructions = true;
412                                 }
413                             }
414                         }
415                     }
416                 }
417     
418                 return stateList;
419             }

421             private MatchResult matchLocation(Location location) throws DataflowAnalysisException {
422                 // Get the ValueNumberFrames before and after the instruction
423                 ValueNumberFrame before = vnaDataflow.getFactAtLocation(location);
424                 ValueNumberFrame after = vnaDataflow.getFactAfterLocation(location);
425     
426                 // Try to match the instruction against the pattern element.
427                 boolean debug = DEBUG && (!(patternElement instanceof Wild) || SHOW_WILD);
428                 if (debug) {
429     
430                     debug((parentPath >= 0 ? parentPath + "->" : "") + path + ": Match " + patternElement + " against "
431                             + location.getHandle() + " " + (bindingSet != null ? bindingSet.toString() : "[]") + "...");
432                 }
433                 MatchResult matchResult = patternElement.match(location.getHandle(), cpg, before, after, bindingSet);
434                 if (debug) {
435                     debug("\t" + ((matchResult != null) ? " ==> MATCH" : " ==> NOT A MATCH"));
436                 }
437                 if (matchResult != null) {
438                     // Successful match!
439                     // Update state to reflect that the match has occurred.
440                     ++matchCount;
441                     canFork = true;
442                     currentMatch = new PatternElementMatch(matchResult.getPatternElement(), location.getHandle(),
443                             location.getBasicBlock(), matchCount, currentMatch);
444                     bindingSet = matchResult.getBindingSet();
445                 }
446                 return matchResult;
447             }
448         }

450         private void debug(String s) {
451             if (!DEBUG) {
452                 throw new IllegalStateException("Only call if DEBUG is true");
453             }
454             System.out.print("                                            ".substring(0, depth));
455             System.out.println(s);
456         }

458         /**
459          * Match a sequence of pattern elements, starting at the given one. The
460          * InstructionIterator should generally be positioned just before the next
461          * instruction to be matched. However, it may be positioned at the end of a
462          * basic block, in which case nothing will happen except that we will try to
463          * continue the match in the non-backedge successors of the basic block.
464          */
465     
466         int depth = 0;
467     
468         private void work(State state) throws DataflowAnalysisException {
469             depth++;
470             try {
471                 // Have we reached the end of the pattern?
472                 if (state.isComplete()) {
473                     // This is a complete match.
474                     if (DEBUG) {
475                         debug("FINISHED A MATCH!");
476                     }
477                     resultList.add(state.getResult());
478                     return;
479                 }
480     
481                 if (DEBUG) {
482                     debug("Matching " + state.getPatternElement() + " against " + state.currentMatch);
483                 }
484                 // If we've reached the minimum number of occurrences for this
485                 // pattern element, we can advance to the next pattern element
486                 // without trying
487                 // to match this instruction again. We make sure that we only
488                 // advance to
489                 // the next element once for this matchCount.
490                 State advance = state.advanceToNextElement();
491                 if (advance != null) {
492                     work(advance);
493                 }
494     
495                 // If we've reached the maximum number of occurrences for this
496                 // pattern element, then we can't continue.
497                 if (!state.currentElementCanContinue()) {
498                     return;
499                 }
500     
501                 MatchResult matchResult = null;
502     
503                 // Are we looking for an instruction dominated by an earlier
504                 // matched instruction?
505                 if (state.lookForDominatedInstruction()) {
506                     Iterable<State> dominatedInstructions = state.dominatedInstructionStateIterable();
507                     for (State s : dominatedInstructions) {
508                         if (DEBUG) {
509                             debug("trying " + s);
510                         }
511                         work(s);
512                     }
513                     return;
514                 }
515     
516                 // Is there another instruction in this basic block?
517                 if (state.moreInstructionsInBasicBlock()) {
518                     // Try to match it.
519                     matchResult = state.matchNextInBasicBlock();
520                     if (matchResult == null) {
521                         return;
522                     }
523                 }
524     
525                 // Continue the match at each successor instruction,
526                 // using the same PatternElement.
527                 if (state.moreInstructionsInBasicBlock()) {
528                     // Easy case; continue matching in the same basic block.
529                     work(state);
530                 } else if (state.canAdvanceToNextBasicBlock()) {
531                     // We've reached the end of the basic block.
532                     // Try to advance to the successors of this basic block,
533                     // ignoring loop backedges.
534                     Iterator<Edge> i = cfg.outgoingEdgeIterator(state.getBasicBlock());
535                     BitSet visitedSuccessorSet = new BitSet();
536                     while (i.hasNext()) {
537                         Edge edge = i.next();
538                         if (dfs.getDFSEdgeType(edge) == BACK_EDGE) {
539                             continue;
540                         }
541     
542                         BasicBlock destBlock = edge.getTarget();
543                         int destId = destBlock.getLabel();
544     
545                         // CFGs can have duplicate edges
546                         if (visitedSuccessorSet.get(destId)) {
547                             continue;
548                         }
549                         visitedSuccessorSet.set(destId, true);
550     
551                         // See if we can continue matching in the successor basic
552                         // block.
553                         State succState = state.advanceToSuccessor(edge, matchResult);
554                         if (succState != null) {
555                             work(succState);
556                         }
557                     }
558                 }
559             } finally {
560                 depth--;
561             }
562     
563         }
