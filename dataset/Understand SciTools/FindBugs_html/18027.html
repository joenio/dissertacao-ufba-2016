24     import java.util.List;
25     import java.util.Map;

27     import javax.annotation.CheckForNull;
28     
29     import org.apache.bcel.classfile.Code;
30     
31     import edu.umd.cs.findbugs.BugAccumulator;
32     import edu.umd.cs.findbugs.BugInstance;
33     import edu.umd.cs.findbugs.BugReporter;
34     import edu.umd.cs.findbugs.ba.ClassContext;
35     import edu.umd.cs.findbugs.bcel.OpcodeStackDetector;

37     import edu.umd.cs.findbugs.internalAnnotations.SlashedClassName;

39     /**
40      * Detector to find calls to Number constructors with base type argument in Java
41      * 5 or newer bytecode.
42      *
43      * Using <code>new Integer(int)</code> is guaranteed to always result in a new
44      * object whereas <code>Integer.valueOf(int)</code> allows caching of values to
45      * be done by the javac, JVM class library or JIT.
46      *
47      * Currently only the JVM class library seems to do caching in the range of -128
48      * to 127. There does not seem to be any caching for float and double which is
49      * why those are reported as low priority.
50      *
51      * All invokes of Number constructor with a constant argument are flagged as
52      * high priority and invokes with unknwon value are normal priority.
53      *
54      * @author Mikko Tiihonen
55      */
56     public class NumberConstructor extends OpcodeStackDetector {
57     
58         static class Pair {
59             final MethodDescriptor boxingMethod;
60             final MethodDescriptor parsingMethod;
61             public Pair(MethodDescriptor boxingMethod, MethodDescriptor parsingMethod) {
62                 this.boxingMethod = boxingMethod;
63                 this.parsingMethod = parsingMethod;
64             }
65         }
66         private final Map<String, Pair> boxClasses = new HashMap<String, Pair>();
67     
68         private final List<MethodDescriptor> methods = new ArrayList<>();
69     
70         private final BugAccumulator bugAccumulator;
71     
72         /**
73          * Constructs a NC detector given the reporter to report bugs on
74          *
75          * @param bugReporter
76          *            the sync of bug reports
77          */
78         public NumberConstructor(BugReporter bugReporter) {
79             this.bugAccumulator = new BugAccumulator(bugReporter);
80             handle("java/lang/Byte", false, "(B)");
81             handle("java/lang/Character", false, "(C)");
82             handle("java/lang/Short", false, "(S)");
83             handle("java/lang/Integer", false, "(I)");
84             handle("java/lang/Long", false, "(J)");
85             handle("java/lang/Float", true, "(F)");
86             handle("java/lang/Double", true, "(D)");
87     
88         }
89     
90         private void handle(@SlashedClassName String className, boolean isFloatingPoint, String sig) {
91             MethodDescriptor boxingMethod = new MethodDescriptor(className, "valueOf", sig + "L" + className +";", true);
92             MethodDescriptor parsingMethod = new MethodDescriptor(className, "valueOf", "(Ljava/lang/String;)" + "L" + className +";", true);
93             boxClasses.put(className, new Pair(boxingMethod, parsingMethod));
94             methods.add(new MethodDescriptor(className, "<init>", "(Ljava/lang/String;)V"));
95             methods.add(new MethodDescriptor(className, "<init>", sig+"V"));
96         }
97     
98         /**
99          * The detector is only meaningful for Java5 class libraries.
100          *
101          * @param classContext
102          *            the context object that holds the JavaClass parsed
103          */
104         @Override
105         public void visitClassContext(ClassContext classContext) {
106             int majorVersion = classContext.getJavaClass().getMajor();
107             if (majorVersion >= MAJOR_1_5 && hasInterestingMethod(classContext.getJavaClass().getConstantPool(), methods)) {
108                 super.visitClassContext(classContext);
109             }
110         }
111     
112         @Override
113         public void visit(Code obj) {
114             super.visit(obj);
115             bugAccumulator.reportAccumulatedBugs();
116         }
117     
118         private boolean matchArguments(String sig1, String sig2) {
119             int lastParen = sig1.indexOf(')');
120             String args = sig1.substring(0, lastParen+1);
121             return sig2.startsWith(args);
122         }
123     
124         private @CheckForNull MethodDescriptor getShouldCall() {
125             String cls = getClassConstantOperand();
126             Pair pair =  boxClasses.get(cls);
127             if (pair == null) {
128                 return null;
129             }
130             MethodDescriptor shouldCall;
131             if (getSigConstantOperand().startsWith("(Ljava/lang/String;)")) {
132                 shouldCall = pair.parsingMethod;
133             } else {
134                 shouldCall = pair.boxingMethod;
135             }
136     
137             if (shouldCall == null) {
138                 return null;
139             }
140     
141             if (matchArguments(getSigConstantOperand(), shouldCall.getSignature())) {
142                 return shouldCall;
143             }
144     
145             return null;
146         }
147         @Override
148         public void sawOpcode(int seen) {
149             // only acts on constructor invoke
150             if (seen != INVOKESPECIAL) {
151                 return;
152             }
153     
154             if (!"<init>".equals(getNameConstantOperand())) {
155                 return;
156             }
157             @SlashedClassName String cls = getClassConstantOperand();
158             MethodDescriptor shouldCall = getShouldCall();
159             if (shouldCall == null) {
160                 return;
161             }
162     
163             int prio;
164             String type;
165             if ("java/lang/Float".equals(cls) || "java/lang/Double".equals(cls)) {
166                 prio = LOW_PRIORITY;
167                 type = "DM_FP_NUMBER_CTOR";
168             } else {
169                 prio = NORMAL_PRIORITY;
170                 Object constantValue = stack.getStackItem(0).getConstant();
171                 if (constantValue instanceof Number) {
172                     long value = ((Number) constantValue).longValue();
173                     if (value < -128 || value > 127) {
174                         prio = LOW_PRIORITY;
175                     }
176                 }
177                 type = "DM_NUMBER_CTOR";
178             }
179     
180             BugInstance bug = new BugInstance(this, type, prio).addClass(this).addMethod(this).addCalledMethod(this)
181                     .addMethod(shouldCall).describe("SHOULD_CALL");
182             bugAccumulator.accumulateBug(bug, this);
183         }
184     }
