20     package edu.umd.cs.findbugs.classfile.impl;
21     
22     import java.io.BufferedOutputStream;
23     import java.io.File;
24     import java.io.FileOutputStream;
25     import java.io.IOException;
26     import java.io.InputStream;
27     import java.io.OutputStream;
28     
29     import edu.umd.cs.findbugs.ba.AnalysisContext;
30     import edu.umd.cs.findbugs.classfile.ICodeBase;
31     import edu.umd.cs.findbugs.classfile.ICodeBaseEntry;
32     import edu.umd.cs.findbugs.classfile.ICodeBaseIterator;
33     import edu.umd.cs.findbugs.classfile.ResourceNotFoundException;
34     import edu.umd.cs.findbugs.io.IO;
35     
36     /**
37      * A scannable code base class for a zip (or Jar) file nested inside some other
38      * codebase. These are handled by extracting the nested zip/jar file to a
39      * temporary file, and delegating to an internal ZipFileCodeBase that reads from
40      * the temporary file.
41      *
42      * @author David Hovemeyer
43      */
44     public class NestedZipFileCodeBase extends AbstractScannableCodeBase {
45         private final ICodeBase parentCodeBase;
46     
47         private final String resourceName;
48     
49         private File tempFile;
50     
51         private AbstractScannableCodeBase delegateCodeBase;
52     
53         /**
54          * Constructor.
55          *
56          * @param codeBaseLocator
57          *            the codebase locator for this codebase
58          */
59         public NestedZipFileCodeBase(NestedZipFileCodeBaseLocator codeBaseLocator) throws ResourceNotFoundException, IOException {
60             super(codeBaseLocator);
61             this.parentCodeBase = codeBaseLocator.getParentCodeBase();
62             this.resourceName = codeBaseLocator.getResourceName();
63     
64             InputStream inputStream = null;
65             OutputStream outputStream = null;
66             try {
67                 // Create a temp file
68                 this.tempFile = File.createTempFile("findbugs", ".zip");
69                 tempFile.deleteOnExit(); // just in case we crash before the
70                 // codebase is closed
71     
72                 // Copy nested zipfile to the temporary file
73                 // FIXME: potentially long blocking operation - should be
74                 // interruptible
75                 ICodeBaseEntry resource = parentCodeBase.lookupResource(resourceName);
76                 if (resource == null) {
77                     throw new ResourceNotFoundException(resourceName);
78                 }
79                 inputStream = resource.openResource();
80                 outputStream = new BufferedOutputStream(new FileOutputStream(tempFile));
81                 IO.copy(inputStream, outputStream);
82                 outputStream.flush();
83     
84                 // Create the delegate to read from the temporary file
85                 delegateCodeBase = ZipCodeBaseFactory.makeZipCodeBase(codeBaseLocator, tempFile);
86             } finally {
87                 if (inputStream != null) {
88                     IO.close(inputStream);
89                 }
90     
91                 if (outputStream != null) {
92                     IO.close(outputStream);
93                 }
94             }
95         }
96     
97         /*
98          * (non-Javadoc)
99          *
100          * @see edu.umd.cs.findbugs.classfile.IScannableCodeBase#iterator()
101          */
102         @Override
103         public ICodeBaseIterator iterator() throws InterruptedException {
104             return new DelegatingCodeBaseIterator(this, delegateCodeBase);
105         }
106     
107         /*
108          * (non-Javadoc)
109          *
110          * @see
111          * edu.umd.cs.findbugs.classfile.ICodeBase#lookupResource(java.lang.String)
112          */
113         @Override
114         public ICodeBaseEntry lookupResource(String resourceName) {
115             ICodeBaseEntry delegateCodeBaseEntry = delegateCodeBase.lookupResource(resourceName);
116             if (delegateCodeBaseEntry == null) {
117                 return null;
118             }
119             return new DelegatingCodeBaseEntry(this, delegateCodeBaseEntry);
120         }
121     
122         /*
123          * (non-Javadoc)
124          *
125          * @see edu.umd.cs.findbugs.classfile.ICodeBase#getPathName()
126          */
127         @Override
128         public String getPathName() {
129             return null;
130         }
131     
132         /*
133          * (non-Javadoc)
134          *
135          * @see edu.umd.cs.findbugs.classfile.ICodeBase#close()
136          */
137         @Override
138         public void close() {
139             delegateCodeBase.close();
140             if (!tempFile.delete()) {
141                 AnalysisContext.logError("Could not delete " + tempFile);
142             }
143         }
144     }
