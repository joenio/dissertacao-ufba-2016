22     package edu.umd.cs.findbugs.graph;
23     
24     import java.util.ArrayList;
25     import java.util.Iterator;
26     import java.util.Set;
27     import java.util.TreeSet;
28     
29     /**
30      * Algorithm to find strongly connected components in a graph. Based on
31      * algorithm in Cormen et. al., <cite>Introduction to Algorithms</cite>, p. 489.
32      */
33     public class StronglyConnectedComponents<GraphType extends Graph<EdgeType, VertexType>, EdgeType extends GraphEdge<EdgeType, VertexType>, VertexType extends GraphVertex<VertexType>> {
34     
35         private final ArrayList<SearchTree<VertexType>> m_stronglyConnectedSearchTreeList;
36     
37         private VertexChooser<VertexType> m_vertexChooser;
38     
39         /**
40          * Constructor.
41          */
42         public StronglyConnectedComponents() {
43             m_stronglyConnectedSearchTreeList = new ArrayList<SearchTree<VertexType>>();
44             m_vertexChooser = null;
45         }
46     
47         /**
48          * Specify a VertexChooser object to restrict which vertices are considered.
49          * This is useful if you only want to find strongly connected components
50          * among a particular category of vertices.
51          */
52         public void setVertexChooser(VertexChooser<VertexType> vertexChooser) {
53             m_vertexChooser = vertexChooser;
54         }
55     
56         /**
57          * Find the strongly connected components in given graph.
58          *
59          * @param g
60          *            the graph
61          * @param toolkit
62          *            a GraphToolkit, used to create temporary graphs used by the
63          *            algorithm
64          */
65         public void findStronglyConnectedComponents(GraphType g, GraphToolkit<GraphType, EdgeType, VertexType> toolkit) {
66     
67             // Perform the initial depth first search
68             DepthFirstSearch<GraphType, EdgeType, VertexType> initialDFS = new DepthFirstSearch<GraphType, EdgeType, VertexType>(g);
69             if (m_vertexChooser != null) {
70                 initialDFS.setVertexChooser(m_vertexChooser);
71             }
72             initialDFS.search();
73     
74             // Create a transposed graph
75             Transpose<GraphType, EdgeType, VertexType> t = new Transpose<GraphType, EdgeType, VertexType>();
76             GraphType transpose = t.transpose(g, toolkit);
77     
78             // Create a set of vertices in the transposed graph,
79             // in descending order of finish time in the initial
80             // depth first search.
81             VisitationTimeComparator<VertexType> comparator = new VisitationTimeComparator<VertexType>(
82                     initialDFS.getFinishTimeList(), VisitationTimeComparator.DESCENDING);
83             Set<VertexType> descendingByFinishTimeSet = new TreeSet<VertexType>(comparator);
84             Iterator<VertexType> i = transpose.vertexIterator();
85             while (i.hasNext()) {
86                 descendingByFinishTimeSet.add(i.next());
87             }
88     
89             // Create a SearchTreeBuilder for transposed DFS
90             SearchTreeBuilder<VertexType> searchTreeBuilder = new SearchTreeBuilder<VertexType>();
91     
92             // Now perform a DFS on the transpose, choosing the vertices
93             // to visit in the main loop by descending finish time
94             final Iterator<VertexType> vertexIter = descendingByFinishTimeSet.iterator();
95             DepthFirstSearch<GraphType, EdgeType, VertexType> transposeDFS = new DepthFirstSearch<GraphType, EdgeType, VertexType>(
96                     transpose) {
97                 @Override
98                 protected VertexType getNextSearchTreeRoot() {
99                     while (vertexIter.hasNext()) {
100                         VertexType vertex = vertexIter.next();
101                         if (visitMe(vertex)) {
102                             return vertex;
103                         }
104                     }
105                     return null;
106                 }
107             };
108             if (m_vertexChooser != null) {
109                 transposeDFS.setVertexChooser(m_vertexChooser);
110             }
111             transposeDFS.setSearchTreeCallback(searchTreeBuilder);
112             transposeDFS.search();
113     
114             // The search tree roots of the second DFS represent the
115             // strongly connected components. Note that we call copySearchTree()
116             // to make the returned search trees relative to the original
117             // graph, not the transposed graph (which would be very confusing).
118             Iterator<SearchTree<VertexType>> j = searchTreeBuilder.searchTreeIterator();
119             while (j.hasNext()) {
120                 m_stronglyConnectedSearchTreeList.add(copySearchTree(j.next(), t));
121             }
122         }
123     
124         /**
125          * Make a copy of given search tree (in the transposed graph) using vertices
126          * of the original graph.
127          *
128          * @param tree
129          *            a search tree in the transposed graph
130          * @param t
131          *            the Transpose object which performed the transposition of the
132          *            original graph
133          */
134         private SearchTree<VertexType> copySearchTree(SearchTree<VertexType> tree, Transpose<GraphType, EdgeType, VertexType> t) {
135             // Copy this node
136             SearchTree<VertexType> copy = new SearchTree<VertexType>(t.getOriginalGraphVertex(tree.getVertex()));
137     
138             // Copy children
139             Iterator<SearchTree<VertexType>> i = tree.childIterator();
140             while (i.hasNext()) {
141                 SearchTree<VertexType> child = i.next();
142                 copy.addChild(copySearchTree(child, t));
143             }
144     
145             return copy;
146         }
147     
148         /**
149          * Returns an iterator over the search trees containing the vertices of each
150          * strongly connected component.
151          *
152          * @return an Iterator over a sequence of SearchTree objects
153          */
154         public Iterator<SearchTree<VertexType>> searchTreeIterator() {
155             return m_stronglyConnectedSearchTreeList.iterator();
156         }
157     
158         /**
159          * Iterator for iterating over sets of vertices in strongly connected
160          * components.
161          */
162         private class SCCSetIterator implements Iterator<Set<VertexType>> {
163             private final Iterator<SearchTree<VertexType>> m_searchTreeIterator;
164     
165             public SCCSetIterator() {
166                 m_searchTreeIterator = searchTreeIterator();
167             }
168     
169             @Override
170             public boolean hasNext() {
171                 return m_searchTreeIterator.hasNext();
172             }
173     
174             @Override
175             public Set<VertexType> next() {
176                 SearchTree<VertexType> tree = m_searchTreeIterator.next();
177                 TreeSet<VertexType> set = new TreeSet<VertexType>();
178                 tree.addVerticesToSet(set);
179                 return set;
180             }
181     
182             @Override
183             public void remove() {
184                 throw new UnsupportedOperationException();
185             }
186         }
187     
188         /**
189          * Returns an iterator over the sets of vertices of each strongly connected
190          * component.
191          *
192          * @return an Iterator over a sequence of Set objects
193          */
194         public Iterator<Set<VertexType>> setIterator() {
195             return new SCCSetIterator();
196         }
197     
198     }
