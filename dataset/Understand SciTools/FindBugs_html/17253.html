48         public static XClass getXClass(ClassDescriptor c) throws CheckedAnalysisException {

52         public static XClass getXClass(@SlashedClassName String className) throws CheckedAnalysisException {

56         public static XClass findSuperImplementor(XClass clazz, String name, String signature, boolean isStatic,
57                 BugReporter bugReporter) {
58     
59             try {
60                 return findSuperImplementor(clazz, name, signature, isStatic);
61             } catch (MissingClassException e) {
62                 bugReporter.reportMissingClass(e.getClassDescriptor());
63                 return clazz;
64             } catch (CheckedAnalysisException e) {
65                 bugReporter.logError("Error finding " + clazz + "." + name + signature, e);
66                 return clazz;
67             }
68     
69         }

85         public static XClass findSuperImplementor(XClass clazz, String name, String signature, boolean isStatic)
86                 throws CheckedAnalysisException {
87     
88             ClassDescriptor superclassDescriptor = clazz.getSuperclassDescriptor();
89             if (superclassDescriptor == null) {
90                 return clazz;
91             }
92             return findImplementor(getXClass(superclassDescriptor), name, signature, isStatic);
93         }
94     
95         public static XClass findImplementor(XClass clazz, String name, String signature, boolean isStatic)
96                 throws CheckedAnalysisException {
97             XMethod m = clazz.findMethod(name, signature, isStatic);
98             if (m != null) {
99                 return clazz;
100             }
101             return findSuperImplementor(clazz, name, signature, isStatic);
102         }

104         public static @CheckForNull
105         JavaClass findSuperDefiner(JavaClass clazz, String name, String signature, BugReporter bugReporter) {
106             try {
107                 JavaClass c = clazz;
108                 while (true) {
109                     c = c.getSuperClass();
110                     if (c == null) {
111                         return null;
112                     }
113                     Method m = findImplementation(c, name, signature);
114                     if (m != null) {
115                         return c;
116                     }
117                 }
118             } catch (ClassNotFoundException e) {
119                 bugReporter.reportMissingClass(e);
120                 return null;
121             }
122         }

166         public static @DottedClassName
167         String findSuperImplementor(@DottedClassName String clazz, String name, String signature, BugReporter bugReporter) {
168             try {
169                 JavaClass c = findImplementor(Repository.getSuperClasses(clazz), name, signature);
170                 return (c != null) ? c.getClassName() : clazz;
171             } catch (ClassNotFoundException e) {
172                 bugReporter.reportMissingClass(e);
173                 return clazz;
174             }
175         }
176     
177         public static @CheckForNull
178         JavaClass findImplementor(JavaClass[] clazz, String name, String signature) {
179     
180             for (JavaClass aClazz : clazz) {
181                 Method m = findImplementation(aClazz, name, signature);
182                 if (m != null && !m.isAbstract()) {
183                     return aClazz;
184                 }
185     
186             }
187             return null;
188         }
189     
190         public static Method findImplementation(JavaClass clazz, String name, String signature) {
191             Method[] m = clazz.getMethods();
192             for (Method aM : m) {
193                 if (aM.getName().equals(name) && aM.getSignature().equals(signature) && !aM.isPrivate() && !aM.isStatic()) {
194                     return aM;
195                 }
196             }
197             return null;
198         }
