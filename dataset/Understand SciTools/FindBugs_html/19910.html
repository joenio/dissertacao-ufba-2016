48     public class CountClassVersions {
49     
50         public static List<String> readFromStandardInput() throws IOException {
51             return readFrom(UserTextFile.reader(System.in));
52         }
53     
54         public static List<String> readFrom(Reader r) throws IOException {
55             BufferedReader in = new BufferedReader(r);
56             List<String> lst = new LinkedList<String>();
57             while (true) {
58                 String s = in.readLine();
59                 if (s == null) {
60                     return lst;
61                 }
62                 lst.add(s);
63             }
64         }
65     
66         static class CountClassVersionsCommandLine extends CommandLine {
67             public String prefix = "";
68     
69             public String inputFileList;
70     
71             long maxAge = Long.MIN_VALUE;
72     
73             CountClassVersionsCommandLine() {
74                 addOption("-maxAge", "days", "maximum age in days (ignore jar files older than this");
75                 addOption("-inputFileList", "filename", "text file containing names of jar files");
76     
77                 addOption("-prefix", "class name prefix", "prefix of class names that should be analyzed e.g., edu.umd.cs.)");
78             }
79     
80             @Override
81             protected void handleOption(String option, String optionExtraPart) throws IOException {
82                 throw new IllegalArgumentException("Unknown option : " + option);
83             }
84     
85             @Override
86             protected void handleOptionWithArgument(String option, String argument) throws IOException {
87                 if ("-prefix".equals(option)) {
88                     prefix = argument;
89                 } else if ("-inputFileList".equals(option)) {
90                     inputFileList = argument;
91                 } else if ("-maxAge".equals(option)) {
92                     maxAge = System.currentTimeMillis() - (24 * 60 * 60 * 1000L) * Integer.parseInt(argument);
93                 } else {
94                     throw new IllegalArgumentException("Unknown option : " + option);
95                 }
96             }
97     
98         }
99     
100         public static void main(String args[]) throws Exception {
101             FindBugs.setNoAnalysis();
102             CountClassVersionsCommandLine commandLine = new CountClassVersionsCommandLine();
103             int argCount = commandLine.parse(args, 0, Integer.MAX_VALUE, "Usage: " + CountClassVersions.class.getName()
104                     + " [options] [<jarFile>+] ");
105     
106             List<String> fileList;
107     
108             if (commandLine.inputFileList != null) {
109                 fileList = readFrom(UTF8.fileReader(commandLine.inputFileList));
110             } else if (argCount == args.length) {
111                 fileList = readFromStandardInput();
112             } else {
113                 fileList = Arrays.asList(args).subList(argCount, args.length - 1);
114             }
115             byte buffer[] = new byte[8192];
116             MessageDigest digest = Util.getMD5Digest();
117             DualKeyHashMap<String, String, String> map = new DualKeyHashMap<String, String, String>();
118     
119             for (String fInName : fileList) {
120                 File f = new File(fInName);
121                 if (f.lastModified() < commandLine.maxAge) {
122                     System.err.println("Skipping " + fInName + ", too old (" + new Date(f.lastModified()) + ")");
123                     continue;
124                 }
125                 System.err.println("Opening " + f);
126     
127                 try (ZipFile zipInputFile = new ZipFile(f)){
128                     for (Enumeration<? extends ZipEntry> e = zipInputFile.entries(); e.hasMoreElements();) {
129                         ZipEntry ze = e.nextElement();
130     
131                         if (ze == null) {
132                             break;
133                         }
134                         if (ze.isDirectory()) {
135                             continue;
136                         }
137     
138                         String name = ze.getName();
139                         if (!name.endsWith(".class")) {
140                             continue;
141                         }
142                         if (!name.replace('/', '.').startsWith(commandLine.prefix)) {
143                             continue;
144                         }
145     
146                         InputStream zipIn = zipInputFile.getInputStream(ze);
147     
148                         while (true) {
149                             int bytesRead = zipIn.read(buffer);
150                             if (bytesRead < 0) {
151                                 break;
152                             }
153                             digest.update(buffer, 0, bytesRead);
154     
155                         }
156                         String hash = new BigInteger(1, digest.digest()).toString(16);
157                         map.put(name, hash, fInName);
158                     }
159                 } catch (IOException e) {
160                     e.printStackTrace();
161                     continue;
162                 }
163             }
164             for (String s : map.keySet()) {
165                 Map<String, String> values = map.get(s);
166                 if (values.size() > 1) {
167                     System.out.println(values.size() + "\t" + s + "\t" + values.values());
168                 }
169     
170             }
171         }
