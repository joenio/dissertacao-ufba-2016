143         private int adjustPriority(int factor, int priority) {
144             if (factor <= 4) {
145                 return LOW_PRIORITY + 2;
146             }
147             if (factor <= 10000) {
148                 return priority + 1;
149             }
150             if (factor <= 60 * 60 * 1000) {
151                 return priority;
152             }
153             return priority - 1;
154         }

156         private int adjustMultiplier(Object constant, int mul) {
157             if (!(constant instanceof Integer)) {
158                 return mul;
159             }
160             return Math.abs(((Integer) constant).intValue()) * mul;

183                 }

188                 bugReporter.reportBug(pendingUnreachableBranch);
189                 pendingUnreachableBranch = null;
190             }

192     
193             if (seen == INVOKESPECIAL && "<init>".equals(getNameConstantOperand()) && "(Ljava/util/Collection;)V".equals(getSigConstantOperand())
194                     && getClassConstantOperand().contains("Set")
195                     || (seen == INVOKEVIRTUAL || seen == INVOKEINTERFACE) && "addAll".equals(getNameConstantOperand()) && "(Ljava/util/Collection;)Z".equals(getSigConstantOperand())) {
196                 OpcodeStack.Item top = stack.getStackItem(0);
197                 XMethod returnValueOf = top.getReturnValueOf();
198                 if (returnValueOf != null && "entrySet".equals(returnValueOf.getName())) {
199                     String name = returnValueOf.getClassName();
200                     int priority = Priorities.LOW_PRIORITY;
201                     if ("java.util.Map".equals(name)) {
202                         priority = Priorities.NORMAL_PRIORITY;
203                     } else if (name.equals(EnumMap.class.getName())
204                             || name.equals(IdentityHashMap.class.getName())) {
205                         priority = Priorities.HIGH_PRIORITY;
206                     }
207                     bugReporter.reportBug(new BugInstance(this, "DMI_ENTRY_SETS_MAY_REUSE_ENTRY_OBJECTS", priority)
208                     .addClassAndMethod(this).addCalledMethod(returnValueOf).addCalledMethod(this).addValueSource(top, this).addSourceLine(this));
209                 }

211     
212             }
213     
214             if (seen == INVOKEVIRTUAL && "hashCode".equals(getNameConstantOperand()) && "()I".equals(getSigConstantOperand())
215                     && stack.getStackDepth() > 0) {
216                 OpcodeStack.Item item0 = stack.getStackItem(0);
217                 if (item0.getSignature().charAt(0) == '[') {
218                     bugReporter.reportBug(new BugInstance(this, "DMI_INVOKING_HASHCODE_ON_ARRAY", NORMAL_PRIORITY)
219                     .addClassAndMethod(this).addValueSource(item0, this).addSourceLine(this));
220                 }
221             }
222             if (seen != RETURN && isReturn(seen) && isRegisterStore(getPrevOpcode(1))) {
223     
224                 int priority = Priorities.NORMAL_PRIORITY;
225                 if (getMethodSig().endsWith(")Z")) {
226                     priority = Priorities.HIGH_PRIORITY;
227                 } else {
228                     if (getMethodSig().endsWith(")Ljava/lang/String;")) {
229                         priority = Priorities.LOW_PRIORITY;
230                     }
231                     if (getPC() == getCode().getCode().length - 1) {
232                         priority++;
233                     }
234                 }
235                 bugReporter.reportBug(new BugInstance(this, "DLS_DEAD_LOCAL_STORE_IN_RETURN", priority).addClassAndMethod(this)
236                         .addSourceLine(this));
237             }
238             // System.out.println(getPC() + " " + OPCODE_NAMES[seen] + " " +
239             // ternaryConversionState);
240             if (seen == IMUL) {
241                 if (imul_distance != 1) {
242                     resetIMulCastLong();
243                 }
244                 imul_distance = 0;
245                 if (stack.getStackDepth() > 1) {
246                     OpcodeStack.Item item0 = stack.getStackItem(0);
247                     OpcodeStack.Item item1 = stack.getStackItem(1);
248                     imul_constant = adjustMultiplier(item0.getConstant(), imul_constant);
249                     imul_constant = adjustMultiplier(item1.getConstant(), imul_constant);
250     
251                     if (item0.isInitialParameter() || item1.isInitialParameter()) {
252                         imul_operand_is_parameter = true;
253                     }
254                 }
255             } else {
256                 imul_distance++;
257             }
258     
259             if (prevOpCode == IMUL && seen == I2L) {
260                 int priority = adjustPriority(imul_constant, NORMAL_PRIORITY);
261                 if (priority >= LOW_PRIORITY && imul_constant != 1000 && imul_constant != 60 && imul_operand_is_parameter) {
262                     priority = NORMAL_PRIORITY;
263                 }
264                 if (priority <= best_priority_for_ICAST_INTEGER_MULTIPLY_CAST_TO_LONG) {
265                     best_priority_for_ICAST_INTEGER_MULTIPLY_CAST_TO_LONG = priority;
266                     bugAccumulator.accumulateBug(
267                             new BugInstance(this, "ICAST_INTEGER_MULTIPLY_CAST_TO_LONG", priority).addClassAndMethod(this), this);
268                 }
269             }
270     
271             if ("<clinit>".equals(getMethodName()) && (seen == PUTSTATIC || seen == GETSTATIC || seen == INVOKESTATIC)) {
272                 String clazz = getClassConstantOperand();
273                 if (!clazz.equals(getClassName())) {
274                     try {
275                         JavaClass targetClass = Repository.lookupClass(clazz);
276                         if (Repository.instanceOf(targetClass, getThisClass())) {
277                             int priority = NORMAL_PRIORITY;
278                             if (seen == GETSTATIC) {
279                                 priority--;
280                             }
281                             if (!targetClass.isPublic()) {
282                                 priority++;
283                             }
284                             bugAccumulator.accumulateBug(new BugInstance(this, "IC_SUPERCLASS_USES_SUBCLASS_DURING_INITIALIZATION",
285                                     priority).addClassAndMethod(this).addClass(getClassConstantOperand()), this);
286     
287                         }
288                     } catch (ClassNotFoundException e) {
289                         // ignore it
290                     }
291     
292                 }
293             }
294             /*
295             if (false && (seen == INVOKEVIRTUAL) && getNameConstantOperand().equals("equals")
296                     && getSigConstantOperand().equals("(Ljava/lang/Object;)Z") && stack.getStackDepth() > 1) {
297                 OpcodeStack.Item item0 = stack.getStackItem(0);
298                 OpcodeStack.Item item1 = stack.getStackItem(1);
299     
300                 if (item0.isArray() || item1.isArray()) {
301                     bugAccumulator.accumulateBug(
302                             new BugInstance(this, "EC_BAD_ARRAY_COMPARE", NORMAL_PRIORITY).addClassAndMethod(this), this);
303                 }
304             }
305              */

307             if (seen >= IALOAD && seen <= SALOAD || seen >= IASTORE && seen <= SASTORE) {
308                 Item index = stack.getStackItem(0);
309                 if (index.getSpecialKind() == Item.AVERAGE_COMPUTED_USING_DIVISION) {
310                     SourceLineAnnotation where;
311                     if (index.getPC() >= 0) {
312                         where = SourceLineAnnotation.fromVisitedInstruction(this, index.getPC());
313                     } else {
314                         where = SourceLineAnnotation.fromVisitedInstruction(this);
315                     }
316                     bugAccumulator.accumulateBug(
317                             new BugInstance(this, "IM_AVERAGE_COMPUTATION_COULD_OVERFLOW", NORMAL_PRIORITY).addClassAndMethod(this),
318                             where);
319                 }
320

323             if ((seen == IFEQ || seen == IFNE) && getPrevOpcode(1) == IMUL
324                     && (getPrevOpcode(2) == SIPUSH || getPrevOpcode(2) == BIPUSH) && getPrevOpcode(3) == IREM) {
325                 bugAccumulator.accumulateBug(
326                         new BugInstance(this, "IM_MULTIPLYING_RESULT_OF_IREM", LOW_PRIORITY).addClassAndMethod(this), this);
327             }

329             if (seen == I2S && getPrevOpcode(1) == IUSHR && !shiftOfNonnegativeValue
330                     && (!constantArgumentToShift || valueOfConstantArgumentToShift % 16 != 0) || seen == I2B
331                     && getPrevOpcode(1) == IUSHR && !shiftOfNonnegativeValue
332                     && (!constantArgumentToShift || valueOfConstantArgumentToShift % 8 != 0)) {
333                 bugAccumulator.accumulateBug(
334                         new BugInstance(this, "ICAST_QUESTIONABLE_UNSIGNED_RIGHT_SHIFT", NORMAL_PRIORITY).addClassAndMethod(this),
335                         this);
336             }
337     
338     
339             if (seen == IADD && (getNextOpcode() == ISHL || getNextOpcode() == LSHL) && stack.getStackDepth() >=3) {
340                 OpcodeStack.Item l = stack.getStackItem(2);
341                 OpcodeStack.Item v = stack.getStackItem(1);
342                 Object constantValue = v.getConstant();
343                 // Ignore 1 << (const + var) as it's usually intended
344                 if (constantValue instanceof Integer && !Integer.valueOf(1).equals(l.getConstant())) {
345                     int c = ((Integer) constantValue).intValue();
346                     int priority = LOW_PRIORITY;
347                     // If (foo << 32 + var) encountered, then ((foo << 32) + var) is absolutely meaningless,
348                     // but (foo << (32 + var)) can be meaningful for negative var values
349                     if (c < 32 || (c < 64 && getNextOpcode() == LSHL)) {
350                         if (c == 8) {
351                             priority--;
352                         }
353                         if (getPrevOpcode(1) == IAND) {
354                             priority--;
355                         }
356                         if (getMethodName().equals("hashCode") && getMethodSig().equals("()I")
357                                 && (getCode().getCode()[getNextPC() + 1] & 0xFF) == IRETURN) {
358                             // commonly observed error is hashCode body like "return foo << 16 + bar;"
359                             priority = HIGH_PRIORITY;
360                         }
361                         bugAccumulator.accumulateBug(new BugInstance(this, "BSHIFT_WRONG_ADD_PRIORITY", priority)
362                         .addClassAndMethod(this)
363                         .addInt(c).describe(IntAnnotation.INT_SHIFT)
364                         .addValueSource(stack.getStackItem(2), this)
365                         .addValueSource(stack.getStackItem(0), this)
366                         , this);
367                     }
368                 }
369     
370             }
371             constantArgumentToShift = false;
372             shiftOfNonnegativeValue = false;
373             if ((seen == IUSHR || seen == ISHR || seen == ISHL)) {
374                 if (stack.getStackDepth() <= 1) {
375                     // don't understand; lie so other detectors won't get concerned
376                     constantArgumentToShift = true;
377                     valueOfConstantArgumentToShift = 8;
378                 } else {
379                     Object rightHandSide = stack.getStackItem(0).getConstant();
380     
381                     Object leftHandSide = stack.getStackItem(1).getConstant();
382                     shiftOfNonnegativeValue = stack.getStackItem(1).isNonNegative();
383                     if (rightHandSide instanceof Integer) {
384                         constantArgumentToShift = true;
385                         valueOfConstantArgumentToShift = ((Integer) rightHandSide);
386                         if (valueOfConstantArgumentToShift < 0 || valueOfConstantArgumentToShift >= 32) {
387                             bugAccumulator.accumulateBug(new BugInstance(this, "ICAST_BAD_SHIFT_AMOUNT",
388                                     valueOfConstantArgumentToShift < 0 ? LOW_PRIORITY : (valueOfConstantArgumentToShift == 32
389                                     && "hashCode".equals(getMethodName()) ? NORMAL_PRIORITY : HIGH_PRIORITY))
390                             .addClassAndMethod(this).addInt(valueOfConstantArgumentToShift).describe(IntAnnotation.INT_SHIFT)
391                             .addValueSource(stack.getStackItem(1), this), this);
392                         }
393                     }
394                     if (leftHandSide != null && leftHandSide instanceof Integer && ((Integer) leftHandSide) > 0) {
395                         // boring; lie so other detectors won't get concerned
396                         constantArgumentToShift = true;
397                         valueOfConstantArgumentToShift = 8;
398                     }
399     
400                 }
401             }
402     
403             if (seen == INVOKEVIRTUAL && stack.getStackDepth() > 0
404                     && ("java/util/Date".equals(getClassConstantOperand()) || "java/sql/Date".equals(getClassConstantOperand()))
405                     && "setMonth".equals(getNameConstantOperand()) && "(I)V".equals(getSigConstantOperand())) {
406                 OpcodeStack.Item item = stack.getStackItem(0);
407                 Object o = item.getConstant();
408                 if (o != null && o instanceof Integer) {
409                     int v = (Integer) o;
410                     if (v < 0 || v > 11) {
411                         bugReporter.reportBug(new BugInstance(this, "DMI_BAD_MONTH", HIGH_PRIORITY).addClassAndMethod(this).addInt(v)
412                                 .describe(IntAnnotation.INT_VALUE).addCalledMethod(this).addSourceLine(this));
413                     }
414                 }
415             }
416     
417             if (seen == INVOKEVIRTUAL && stack.getStackDepth() > 1 && "java/util/Calendar".equals(getClassConstantOperand())
418                     && "set".equals(getNameConstantOperand())
419     
420                     || seen == INVOKESPECIAL && stack.getStackDepth() > 1
421                     && "java/util/GregorianCalendar".equals(getClassConstantOperand()) && "<init>".equals(getNameConstantOperand())
422     
423                     ) {
424                 String sig = getSigConstantOperand();
425                 if (sig.startsWith("(III")) {
426                     int pos = sig.length() - 5;
427                     OpcodeStack.Item item = stack.getStackItem(pos);
428                     Object o = item.getConstant();
429                     if (o != null && o instanceof Integer) {
430                         int v = (Integer) o;
431                         if (v < 0 || v > 11) {
432                             bugReporter.reportBug(new BugInstance(this, "DMI_BAD_MONTH", NORMAL_PRIORITY).addClassAndMethod(this)
433                                     .addInt(v).describe(IntAnnotation.INT_VALUE).addCalledMethod(this).addSourceLine(this));
434                         }
435                     }
436                 }
437             }
438     
439             if (isRegisterStore() && (seen == ISTORE || seen == ISTORE_0 || seen == ISTORE_1 || seen == ISTORE_2 || seen == ISTORE_3)
440                     && getRegisterOperand() == prevOpcodeIncrementedRegister) {
441                 bugAccumulator.accumulateBug(
442                         new BugInstance(this, "DLS_OVERWRITTEN_INCREMENT", HIGH_PRIORITY).addClassAndMethod(this), this);
443     
444             }
445             if (seen == IINC) {
446                 prevOpcodeIncrementedRegister = getRegisterOperand();
447             } else {
448                 prevOpcodeIncrementedRegister = -1;
449             }
450     
451             // Java Puzzlers, Chapter 2, puzzle 1
452             // Look for ICONST_2 IREM ICONST_1 IF_ICMPNE L1
453     
454             switch (badlyComputingOddState) {
455             case 0:
456                 if (seen == ICONST_2) {
457                     badlyComputingOddState++;
458                 }
459                 break;
460             case 1:
461                 if (seen == IREM) {
462                     OpcodeStack.Item item = stack.getStackItem(1);
463                     if (!item.isNonNegative() && item.getSpecialKind() != OpcodeStack.Item.MATH_ABS) {
464                         badlyComputingOddState++;
465                     } else {
466                         badlyComputingOddState = 0;
467                     }
468                 } else {
469                     badlyComputingOddState = 0;
470                 }
471                 break;
472             case 2:
473                 if (seen == ICONST_1) {
474                     badlyComputingOddState++;
475                 } else {
476                     badlyComputingOddState = 0;
477                 }
478                 break;
479             case 3:
480                 if (seen == IF_ICMPEQ || seen == IF_ICMPNE) {
481                     bugAccumulator.accumulateBug(
482                             new BugInstance(this, "IM_BAD_CHECK_FOR_ODD", NORMAL_PRIORITY).addClassAndMethod(this), this);
483                 }
484                 badlyComputingOddState = 0;
485                 break;
486             default:
487                 break;
488             }
489     
490             // Java Puzzlers, chapter 3, puzzle 12
491             if (seen == INVOKEVIRTUAL
492                     && stack.getStackDepth() > 0
493                     && ("toString".equals(getNameConstantOperand()) && "()Ljava/lang/String;".equals(getSigConstantOperand())
494                             || "append".equals(getNameConstantOperand())
495                             && "(Ljava/lang/Object;)Ljava/lang/StringBuilder;".equals(getSigConstantOperand())
496                             && "java/lang/StringBuilder".equals(getClassConstantOperand())
497                             || "append".equals(getNameConstantOperand())
498                             && "(Ljava/lang/Object;)Ljava/lang/StringBuffer;".equals(getSigConstantOperand())
499                             && "java/lang/StringBuffer".equals(getClassConstantOperand()) || ("print".equals(getNameConstantOperand()) || "println".equals(getNameConstantOperand()))
500                             && "(Ljava/lang/Object;)V".equals(getSigConstantOperand()))) {
501                 OpcodeStack.Item item = stack.getStackItem(0);
502                 String signature = item.getSignature();
503                 if (signature != null && signature.startsWith("[")) {
504                     boolean debuggingContext = "[Ljava/lang/StackTraceElement;".equals(signature);
505     
506                     if (!debuggingContext) {
507                         for (CodeException e : getCode().getExceptionTable()) {
508                             if (e.getHandlerPC() <= getPC() && e.getHandlerPC() + 30 >= getPC()) {
509                                 debuggingContext = true;
510                             }
511                         }
512     
513                         for (int i = 1; !debuggingContext && i < stack.getStackDepth(); i++) {
514                             OpcodeStack.Item e = stack.getStackItem(i);
515     
516                             if (e.getSignature().indexOf("Logger") >= 0 || e.getSignature().indexOf("Exception") >= 0) {
517                                 debuggingContext = true;
518                             }
519     
520                             XField f = e.getXField();
521                             if (f != null && (SYSTEM_ERR.equals(f.getFieldDescriptor()) || SYSTEM_OUT.equals(f.getFieldDescriptor()))) {
522                                 debuggingContext = true;
523                             }
524                         }
525                     }
526                     //                String name = null;
527                     int reg = item.getRegisterNumber();
528                     Collection<BugAnnotation> as = new ArrayList<BugAnnotation>();
529                     XField field = item.getXField();
530                     FieldAnnotation fieldAnnotation = null;
531                     if (field != null) {
532                         fieldAnnotation = FieldAnnotation.fromXField(field);
533                         fieldAnnotation.setDescription(FieldAnnotation.LOADED_FROM_ROLE);
534                     }
535     
536                     if (reg != -1) {
537                         LocalVariableAnnotation lva = LocalVariableAnnotation.getLocalVariableAnnotation(getMethod(), reg, getPC(),
538                                 getPC() - 1);
539                         if (lva.isNamed()) {
540                             as.add(lva);
541                             if (fieldAnnotation != null) {
542                                 as.add(fieldAnnotation);
543                             }
544                         } else {
545                             if (fieldAnnotation != null) {
546                                 as.add(fieldAnnotation);
547                             }
548                             as.add(lva);
549                         }
550                     } else if (fieldAnnotation != null) {
551                         as.add(fieldAnnotation);
552                     } else {
553                         XMethod m = item.getReturnValueOf();
554                         if (m != null) {
555                             MethodAnnotation methodAnnotation = MethodAnnotation.fromXMethod(m);
556                             methodAnnotation.setDescription(MethodAnnotation.METHOD_RETURN_VALUE_OF);
557                             as.add(methodAnnotation);
558                         }
559                     }
560                     int priority = debuggingContext ? NORMAL_PRIORITY : HIGH_PRIORITY;
561                     if (!as.isEmpty()) {
562                         bugAccumulator.accumulateBug(new BugInstance(this, "DMI_INVOKING_TOSTRING_ON_ARRAY", priority)
563                         .addClassAndMethod(this).addAnnotations(as), this);
564                     } else {
565                         bugAccumulator.accumulateBug(
566                                 new BugInstance(this, "DMI_INVOKING_TOSTRING_ON_ANONYMOUS_ARRAY", priority).addClassAndMethod(this),
567                                 this);
568                     }
569                 }
570             }
571     
572             if (isTigerOrHigher) {
573                 if (previousMethodInvocation != null && prevOpCode == INVOKEVIRTUAL && seen == INVOKESTATIC) {
574                     String classNameForPreviousMethod = previousMethodInvocation.getClassName();
575                     String classNameForThisMethod = getClassConstantOperand();
576                     if (classNameForPreviousMethod.startsWith("java.lang.")
577                             && classNameForPreviousMethod.equals(classNameForThisMethod.replace('/', '.'))
578                             && previousMethodInvocation.getName().endsWith("Value")
579                             && previousMethodInvocation.getSignature().length() == 3
580                             && "valueOf".equals(getNameConstantOperand())
581                             && getSigConstantOperand().charAt(1) == previousMethodInvocation.getSignature().charAt(2)) {
582                         bugAccumulator.accumulateBug(
583                                 new BugInstance(this, "BX_UNBOXING_IMMEDIATELY_REBOXED", NORMAL_PRIORITY).addClassAndMethod(this)
584                                 .addCalledMethod(this),
585                                 this);
586                     }
587     
588                 }
589     
590                 if (previousMethodInvocation != null && prevOpCode == INVOKESPECIAL && seen == INVOKEVIRTUAL) {
591                     String classNameForPreviousMethod = previousMethodInvocation.getClassName();
592                     String classNameForThisMethod = getClassConstantOperand();
593                     if (classNameForPreviousMethod.startsWith("java.lang.")
594                             && classNameForPreviousMethod.equals(classNameForThisMethod.replace('/', '.'))
595                             && getNameConstantOperand().endsWith("Value") && getSigConstantOperand().length() == 3) {
596                         if (getSigConstantOperand().charAt(2) == previousMethodInvocation.getSignature().charAt(1)) {
597                             bugAccumulator.accumulateBug(
598                                     new BugInstance(this, "BX_BOXING_IMMEDIATELY_UNBOXED", NORMAL_PRIORITY).addClassAndMethod(this),
599                                     this);
600                         } else {
601                             bugAccumulator.accumulateBug(new BugInstance(this, "BX_BOXING_IMMEDIATELY_UNBOXED_TO_PERFORM_COERCION",
602                                     NORMAL_PRIORITY).addClassAndMethod(this), this);
603                         }
604     
605                         ternaryConversionState = 1;
606                     } else {
607                         ternaryConversionState = 0;
608                     }
609     
610                 } else if (seen == INVOKEVIRTUAL) {
611                     if (getClassConstantOperand().startsWith("java/lang") && getNameConstantOperand().endsWith("Value")
612                             && getSigConstantOperand().length() == 3) {
613                         ternaryConversionState = 1;
614                     } else {
615                         ternaryConversionState = 0;
616                     }
617                 } else if (ternaryConversionState == 1) {
618                     if (I2L < seen && seen <= I2S) {
619                         ternaryConversionState = 2;
620                     } else {
621                         ternaryConversionState = 0;
622                     }
623                 } else if (ternaryConversionState == 2) {
624                     ternaryConversionState = 0;
625                     if (seen == GOTO) {
626                         bugReporter.reportBug(new BugInstance(this, "BX_UNBOXED_AND_COERCED_FOR_TERNARY_OPERATOR", NORMAL_PRIORITY)
627                         .addClassAndMethod(this).addSourceLine(this));
628                     }
629                 } else {
630                     ternaryConversionState = 0;
631                 }
632             }
633     
634             AssertInvokedFromRun: if (seen == INVOKESTATIC) {
635                 if ((getNameConstantOperand().startsWith("assert") || getNameConstantOperand().startsWith("fail"))
636                         && "run".equals(getMethodName()) && implementsRunnable(getThisClass())) {
637                     int size1 = Util.getSizeOfSurroundingTryBlock(getConstantPool(), getMethod().getCode(), "java/lang/Throwable",
638                             getPC());
639                     int size2 = Util.getSizeOfSurroundingTryBlock(getConstantPool(), getMethod().getCode(), "java/lang/Error",
640                             getPC());
641                     int size3 = Util.getSizeOfSurroundingTryBlock(getConstantPool(), getMethod().getCode(),
642                             "java/lang/AssertionFailureError", getPC());
643                     int size = Math.min(Math.min(size1, size2), size3);
644                     if (size == Integer.MAX_VALUE) {
645                         String dottedClassName = getClassConstantOperand().replace('/', '.');
646                         if (!dottedClassName.startsWith("junit")) {
647                             try {
648                                 JavaClass targetClass = AnalysisContext.currentAnalysisContext().lookupClass(dottedClassName);
649                                 if (!targetClass.getSuperclassName().startsWith("junit")) {
650                                     break AssertInvokedFromRun;
651                                 }
652                             } catch (ClassNotFoundException e) {
653                                 AnalysisContext.reportMissingClass(e);
654                                 break AssertInvokedFromRun;
655                             }
656                         }
657     
658                         bugAccumulator.accumulateBug(new BugInstance(this, "IJU_ASSERT_METHOD_INVOKED_FROM_RUN_METHOD",
659                                 extendsThread(getThisClass()) ? NORMAL_PRIORITY : LOW_PRIORITY).addClassAndMethod(this), this);
660     
661                     }
662     
663                 }
664             }
665             if (seen == INVOKESPECIAL && getClassConstantOperand().startsWith("java/lang/")
666                     && "<init>".equals(getNameConstantOperand()) && getSigConstantOperand().length() == 4) {
667                 previousMethodInvocation = XFactory.createReferencedXMethod(this);
668             } else if (seen == INVOKESTATIC && getClassConstantOperand().startsWith("java/lang/")
669                     && "valueOf".equals(getNameConstantOperand()) && getSigConstantOperand().length() == 4) {
670                 previousMethodInvocation = XFactory.createReferencedXMethod(this);
671             } else if (seen == INVOKEVIRTUAL && getClassConstantOperand().startsWith("java/lang/")
672                     && getNameConstantOperand().endsWith("Value")
673                     && getSigConstantOperand().length() == 3) {
674                 previousMethodInvocation = XFactory.createReferencedXMethod(this);
675             } else {
676                 previousMethodInvocation = null;
677             }
678     
679             if (testingEnabled && seen == IAND || seen == LAND) {
680                 OpcodeStack.Item rhs = stack.getStackItem(0);
681                 OpcodeStack.Item lhs = stack.getStackItem(1);
682                 Object constant = rhs.getConstant();
683                 OpcodeStack.Item value = lhs;
684                 if (constant == null) {
685                     constant = lhs.getConstant();
686                     value = rhs;
687                 }
688                 if (constant instanceof Number && (seen == LAND || value.getSpecialKind() == OpcodeStack.Item.RESULT_OF_L2I)) {
689                     long constantValue = ((Number) constant).longValue();
690                     if ((constantValue == 0xEFFFFFFFL || constantValue == 0xEFFFFFFFFFFFFFFFL || seen == IAND
691                             && constantValue == 0xEFFFFFFF)) {
692                         bugAccumulator.accumulateBug(new BugInstance(this, "TESTING", seen == LAND ? HIGH_PRIORITY : NORMAL_PRIORITY)
693                         .addClassAndMethod(this).addString("Possible failed attempt to mask lower 31 bits of an int")
694                         .addValueSource(value, this), this);
695                     }
696     
697                 }
698             }
699     
700             if (seen == INEG) {
701                 OpcodeStack.Item top = stack.getStackItem(0);
702                 XMethod m = top.getReturnValueOf();
703                 if (m != null) {
704                     if ("compareTo".equals(m.getName()) || "compare".equals(m.getName())) {
705                         bugAccumulator.accumulateBug(new BugInstance(this, "RV_NEGATING_RESULT_OF_COMPARETO", NORMAL_PRIORITY)
706                         .addClassAndMethod(this)
707                         .addCalledMethod(m).addValueSource(top, this), this);
708                     }
709                 }
710     
711             }
712             prevOpCode = seen;
713     
714         }
715     
716         boolean implementsRunnable(JavaClass obj) {
717             return Subtypes2.instanceOf(obj, "java.lang.Runnable");
718         }
719     
720         boolean extendsThread(JavaClass obj) {
721             return Subtypes2.instanceOf(obj, "java.lang.Thread");
722         }
