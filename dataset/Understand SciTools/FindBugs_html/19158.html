20     package edu.umd.cs.findbugs.classfile.engine.bcel;
21     
22     import edu.umd.cs.findbugs.ba.BytecodeScanner;
23     import edu.umd.cs.findbugs.ba.MethodBytecodeSet;
24     
25     public class UnpackedBytecodeCallback implements BytecodeScanner.Callback {
26         private final MethodBytecodeSet bytecodeSet;
27     
28         private final short[] offsetToOpcodeMap;
29     
30         public UnpackedBytecodeCallback(int codeSize) {
31             this.bytecodeSet = new MethodBytecodeSet();
32             this.offsetToOpcodeMap = new short[codeSize];
33         }
34     
35         /*
36          * (non-Javadoc)
37          *
38          * @see
39          * edu.umd.cs.findbugs.ba.BytecodeScanner.Callback#handleInstruction(int,
40          * int)
41          */
42         @Override
43         public void handleInstruction(int opcode, int index) {
44             bytecodeSet.set(opcode);
45             offsetToOpcodeMap[index] = (short) opcode;
46         }
47     
48         public UnpackedCode getUnpackedCode() {
49             return new UnpackedCode(bytecodeSet, offsetToOpcodeMap);
50         }
51     }
