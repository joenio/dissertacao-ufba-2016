92         @Override
93         public void visit(Code obj) {
94             if (DEBUG) {
95                 System.out.printf("%nVisiting %s%n", getMethodDescriptor());
96             }
97             reachable = false;
98             lastPC = 0;
99             biggestJumpTarget = -1;
100             found.clear();
101             switchHdlr = new SwitchHandler();
102             clearAllDeadStores();
103             deadStore = null;
104             priority = NORMAL_PRIORITY;
105             fallthroughDistance = 1000;
106             enumType = null;
107             super.visit(obj);
108             enumType = null;
109             if (!found.isEmpty()) {
110                 if (found.size() >= 4 && priority == NORMAL_PRIORITY) {
111                     priority = LOW_PRIORITY;
112                 }
113                 for (SourceLineAnnotation s : found) {
114                     bugAccumulator.accumulateBug(new BugInstance(this, "SF_SWITCH_FALLTHROUGH", priority).addClassAndMethod(this), s);
115                 }
116             }
117     
118             bugAccumulator.reportAccumulatedBugs();
119         }

121         private void foundSwitchNoDefault(SourceLineAnnotation s) {
122             LineNumberTable table = getCode().getLineNumberTable();
123     
124             if (table != null) {
125                 int startLine = s.getStartLine();
126                 int prev = Integer.MIN_VALUE;
127                 for (LineNumber ln : table.getLineNumberTable()) {
128                     int thisLineNumber = ln.getLineNumber();
129                     if (thisLineNumber < startLine && thisLineNumber > prev && ln.getStartPC() < s.getStartBytecode()) {
130                         prev = thisLineNumber;
131                     }
132                 }
133                 int diff = startLine - prev;
134                 if (diff > 5) {
135                     return;
136                 }
137     
138                 bugAccumulator.accumulateBug(new BugInstance(this, "SF_SWITCH_NO_DEFAULT", NORMAL_PRIORITY).addClassAndMethod(this),
139                         s);
140             }
141     
142         }

146         @Override
147         public void sawOpcode(int seen) {
148             boolean isDefaultOffset = switchHdlr.getDefaultOffset() == getPC();
149             boolean isCaseOffset = switchHdlr.isOnSwitchOffset(this);
150     
151             if (DEBUG) {
152                 if (seen == GOTO) {
153                     System.out.printf("%4d: goto %-7d %s %s %s %d%n", getPC(), getBranchTarget(), reachable, isCaseOffset,
154                             isDefaultOffset, switchHdlr.stackSize());
155                 } else {
156                     System.out.printf("%4d: %-12s %s %s %s %d%n", getPC(), OPCODE_NAMES[seen], reachable, isCaseOffset,
157                             isDefaultOffset, switchHdlr.stackSize());
158                 }
159             }
160     
161             if (reachable && (isDefaultOffset || isCaseOffset)) {
162                 if (DEBUG) {
163                     System.out.println("Fallthrough at : " + getPC() + ": " + OPCODE_NAMES[seen]);
164                 }
165                 fallthroughDistance = 0;
166                 potentiallyDeadStoresFromBeforeFallthrough = (BitSet) potentiallyDeadStores.clone();
167                 potentiallyDeadFieldsFromBeforeFallthrough = new HashSet<XField>(potentiallyDeadFields);
168                 if (!hasFallThruComment(lastPC + 1, getPC() - 1)) {
169                     if (!isDefaultOffset) {
170                         SourceLineAnnotation sourceLineAnnotation = SourceLineAnnotation.fromVisitedInstructionRange(
171                                 getClassContext(), this, lastPC, getPC());
172                         found.add(sourceLineAnnotation);
173                     } else if ( getPC() >= biggestJumpTarget) {
174                         SourceLineAnnotation sourceLineAnnotation = switchHdlr.getCurrentSwitchStatement(this);
175                         if (DEBUG) {
176                             System.out.printf("Found fallthrough to default offset at %d (BJT is %d)%n", getPC(), biggestJumpTarget);
177                         }
178     
179                         foundSwitchNoDefault(sourceLineAnnotation);
180                     }
181                 }
182     
183             }
184     
185             if (isBranch(seen) || isSwitch(seen) || seen == GOTO || seen == ARETURN || seen == IRETURN || seen == RETURN
186                     || seen == LRETURN || seen == DRETURN || seen == FRETURN) {
187                 clearAllDeadStores();
188             }
189     
190             if (seen == GETFIELD && stack.getStackDepth() > 0) {
191                 OpcodeStack.Item top = stack.getStackItem(0);
192                 if (top.getRegisterNumber() == 0) {
193                     potentiallyDeadFields.remove(getXFieldOperand());
194                 }
195             }
196     
197             else if (seen == PUTFIELD && stack.getStackDepth() >= 2) {
198                 OpcodeStack.Item obj = stack.getStackItem(1);
199                 if (obj.getRegisterNumber() == 0) {
200                     XField f = getXFieldOperand();
201                     if (potentiallyDeadFields.contains(f) && potentiallyDeadFieldsFromBeforeFallthrough.contains(f)) {
202                         // killed store
203                         priority = HIGH_PRIORITY;
204                         bugAccumulator.accumulateBug(new BugInstance(this, "SF_DEAD_STORE_DUE_TO_SWITCH_FALLTHROUGH", priority)
205                         .addClassAndMethod(this).addField(f), this);
206     
207                     }
208                     potentiallyDeadFields.add(f);
209                 }
210             }
211     
212             if (seen == ATHROW) {
213                 int sz = edu.umd.cs.findbugs.visitclass.Util.getSizeOfSurroundingTryBlock(getMethod(), (String) null, getPC());
214                 if (sz == Integer.MAX_VALUE) {
215     
216                     BitSet dead = new BitSet();
217                     dead.or(potentiallyDeadStores);
218                     dead.and(potentiallyDeadStoresFromBeforeFallthrough);
219                     if (dead.cardinality() > 0) {
220                         int register = dead.nextSetBit(0);
221                         priority = HIGH_PRIORITY;
222                         deadStore = LocalVariableAnnotation.getLocalVariableAnnotation(getMethod(), register, getPC() - 1, getPC());
223                         bugAccumulator.accumulateBug(new BugInstance(this, "SF_DEAD_STORE_DUE_TO_SWITCH_FALLTHROUGH_TO_THROW",
224                                 priority).addClassAndMethod(this).add(deadStore), this);
225                     }
226                 }
227                 clearAllDeadStores();
228             }
229     
230             if (isRegisterLoad()) {
231                 potentiallyDeadStores.clear(getRegisterOperand());
232             } else if (isRegisterStore() && !atCatchBlock()) {
233                 int register = getRegisterOperand();
234                 if (potentiallyDeadStores.get(register) && (potentiallyDeadStoresFromBeforeFallthrough.get(register))) {
235                     // killed store
236                     priority = HIGH_PRIORITY;
237                     deadStore = LocalVariableAnnotation.getLocalVariableAnnotation(getMethod(), register, getPC() - 1, getPC());
238                     bugAccumulator.accumulateBug(new BugInstance(this, "SF_DEAD_STORE_DUE_TO_SWITCH_FALLTHROUGH", priority)
239                     .addClassAndMethod(this).add(deadStore), this);
240     
241                 }
242                 potentiallyDeadStores.set(register);
243             }
244     
245     
246             if (seen == INVOKEVIRTUAL && "ordinal".equals(getNameConstantOperand()) && "()I".equals(getSigConstantOperand())) {
247                 XClass c = getXClassOperand();
248                 if (c != null) {
249                     ClassDescriptor superclassDescriptor = c.getSuperclassDescriptor();
250                     if (superclassDescriptor != null && "java/lang/Enum".equals(superclassDescriptor.getClassName())) {
251                         enumType = c;
252                     }
253                     if (DEBUG) {
254                         System.out.println("Saw " + enumType + ".ordinal()");
255                     }
256                 }
257             } else if (seen != TABLESWITCH && seen != LOOKUPSWITCH && seen != IALOAD) {
258                 enumType = null;
259             }
260     
261             switch (seen) {
262             case TABLESWITCH:
263             case LOOKUPSWITCH:
264                 if (justSawHashcode)
265                 {
266                     break; // javac compiled switch statement
267                 }
268                 reachable = false;
269                 biggestJumpTarget = -1;
270                 switchHdlr.enterSwitch(this, enumType);
271                 if (DEBUG) {
272                     System.out.printf("  entered switch, default is %d%n", switchHdlr.getDefaultOffset());
273                 }
274                 break;
275     
276             case GOTO_W:
277             case GOTO:
278                 if (biggestJumpTarget < getBranchTarget()) {
279                     biggestJumpTarget = getBranchTarget();
280                     if (DEBUG) {
281                         System.out.printf("  Setting BJT to %d%n", biggestJumpTarget);
282                     }
283                 }
284     
285                 reachable = false;
286                 break;
287     
288             case ATHROW:
289             case RETURN:
290             case ARETURN:
291             case IRETURN:
292             case LRETURN:
293             case DRETURN:
294             case FRETURN:
295                 reachable = false;
296                 break;
297     
298             case INVOKESTATIC:
299                 reachable = !("exit".equals(getNameConstantOperand()) && "java/lang/System".equals(getClassConstantOperand()));
300                 break;
301     
302             default:
303                 reachable = true;
304             }
305     
306             justSawHashcode =   seen == INVOKEVIRTUAL && "hashCode".equals(getNameConstantOperand()) && "()I".equals(getSigConstantOperand());
307             lastPC = getPC();
308             fallthroughDistance++;
309         }

311         boolean justSawHashcode;

313          *
314          */
315         private void clearAllDeadStores() {
316             potentiallyDeadStores.clear();
317             potentiallyDeadStoresFromBeforeFallthrough.clear();
318             potentiallyDeadFields.clear();
319             potentiallyDeadFieldsFromBeforeFallthrough.clear();
320         }

322         private boolean hasFallThruComment(int startPC, int endPC) {
323             if (LOOK_IN_SOURCE_FOR_FALLTHRU_COMMENT) {
324                 BufferedReader r = null;
325                 try {
326                     SourceLineAnnotation srcLine = SourceLineAnnotation.fromVisitedInstructionRange(this, lastPC, getPC());
327                     SourceFinder sourceFinder = AnalysisContext.currentAnalysisContext().getSourceFinder();
328                     SourceFile sourceFile = sourceFinder.findSourceFile(srcLine.getPackageName(), srcLine.getSourceFile());
329     
330                     int startLine = srcLine.getStartLine();
331                     int numLines = srcLine.getEndLine() - startLine - 1;
332                     if (numLines <= 0) {
333                         return false;
334                     }
335                     r = UTF8.bufferedReader(sourceFile.getInputStream());
336                     for (int i = 0; i < startLine; i++) {
337                         String line = r.readLine();
338                         if (line == null) {
339                             return false;
340                         }
341                     }
342                     for (int i = 0; i < numLines; i++) {
343                         String line = r.readLine();
344                         if (line == null) {
345                             return false;
346                         }
347                         line = line.toLowerCase();
348                         if (line.indexOf("fall") >= 0 || line.indexOf("nobreak") >= 0) {
349                             return true;
350                         }
351                     }
352                 } catch (IOException ioe) {
353                     // Problems with source file, mean report the bug
354                 } finally {
355                     try {
356                         if (r != null) {
357                             r.close();
358                         }
359                     } catch (IOException ioe) {
360                     }
361                 }
362             }
363             return false;
364         }
