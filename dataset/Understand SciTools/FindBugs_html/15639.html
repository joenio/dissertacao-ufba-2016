20     package edu.umd.cs.findbugs.classfile;
21     
22     /**
23      * Iterator over the resources in an IScannableCodeBase. Note that some of the
24      * methods can throw InterruptedException. This occurs when the analysis is
25      * canceled by interrupting the analysis thread.
26      *
27      * <p>
28      * Note that the close() method must be called when done with an
29      * ICodeBaseIterator object.
30      * </p>
31      *
32      * @author David Hovemeyer
33      */
34     public interface ICodeBaseIterator {
35         /**
36          * Return true if there is another resource to be scanned, false otherwise.
37          *
38          * @return true if there is another resource to be scanned, false otherwise
39          */
40         public boolean hasNext() throws InterruptedException;
41     
42         /**
43          * Get the ICodeBaseEntry representing the next resource in the code base.
44          *
45          * @return the ICodeBaseEntry representing the next resource in the code
46          *         base
47          * @throws InterruptedException
48          */
49         public ICodeBaseEntry next() throws InterruptedException;
50     }
