43     public class WrongMapIterator extends BytecodeScanningDetector implements StatelessDetector {
44         private static final Set<MethodDescriptor> methods = Collections.singleton(new MethodDescriptor("", "keySet", "()Ljava/util/Set;"));
45     
46         static enum LoadedVariableState {
47             NOTHING, LOCAL, FIELD
48         }
49     
50         final LoadedVariable NONE = new LoadedVariable(LoadedVariableState.NOTHING, 0, null);
51     
52         final class LoadedVariable {
53             private final LoadedVariableState lvState;
54             private final int num;
55             private final FieldDescriptor fd;
56     
57             private LoadedVariable(LoadedVariableState state, int num, FieldDescriptor fd) {
58                 this.lvState = state;
59                 this.num = num;
60                 this.fd = fd;
61             }
62     
63             public boolean none() {
64                 return lvState == LoadedVariableState.NOTHING;
65             }
66     
67             public boolean isRegister(int register) {
68                 return lvState == LoadedVariableState.LOCAL && num == register;
69             }
70     
71             public LoadedVariable seen(int opcode) {
72                 if(isRegisterLoad() && !isRegisterStore()) {
73                     return new LoadedVariable(LoadedVariableState.LOCAL, getRegisterOperand(), null);
74                 }
75                 switch(opcode) {
76                 case GETSTATIC:
77                     return new LoadedVariable(LoadedVariableState.FIELD, 0, getFieldDescriptorOperand());
78                 case GETFIELD:
79                     if(lvState == LoadedVariableState.LOCAL && num == 0) {
80                         // Ignore fields from other classes
81                         return new LoadedVariable(LoadedVariableState.FIELD, 0, getFieldDescriptorOperand());
82                     }
83                     return NONE;
84                 default:
85                     return NONE;
86                 }
87             }
88     
89             public boolean same(LoadedVariable other) {
90                 if(other.lvState != lvState) {
91                     return false;
92                 }
93                 if(lvState == LoadedVariableState.LOCAL && num != other.num) {
94                     return false;
95                 }
96                 if ((lvState == LoadedVariableState.FIELD) && !fd.equals(other.fd)) {
97                     return false;
98                 }
99                 return true;
100             }
101     
102             public BugInstance annotate(BugInstance bug) {
103                 if(lvState == LoadedVariableState.FIELD) {
104                     bug.addField(fd);
105                 }
106                 return bug;
107             }
108         }
109     
110         private final BugAccumulator bugAccumulator;
111     
112         private static final int NOT_FOUND = -2;
113         private static final int IN_STACK = -1;
114     
115         private LoadedVariable loadedVariable = NONE;
116     
117         private LoadedVariable mapVariable = NONE;
118     
119         private int keySetRegister;
120     
121         private int iteratorRegister;
122     
123         private int keyRegister;
124     
125         private boolean mapAndKeyLoaded;
126     
127         public WrongMapIterator(BugReporter bugReporter) {
128             this.bugAccumulator = new BugAccumulator(bugReporter);
129         }
130     
131         @Override
132         public void visitClassContext(ClassContext classContext) {
133             if(hasInterestingMethod(classContext.getJavaClass().getConstantPool(), methods)) {
134                 super.visitClassContext(classContext);
135             }
136         }
137     
138         @Override
139         public void visit(Method obj) {
140             reset();
141         }
142     
143         private void reset() {
144             loadedVariable = NONE;
145             mapVariable = NONE;
146             mapAndKeyLoaded = false;
147             keySetRegister = NOT_FOUND;
148             iteratorRegister = NOT_FOUND;
149             keyRegister = NOT_FOUND;
150         }
151     
152         @Override
153         public void visit(Code code) {
154             super.visit(code);
155             bugAccumulator.reportAccumulatedBugs();
156         }
157     
158         /**
159          * Determine from the class descriptor for a variable whether that variable
160          * implements java.util.Map.
161          *
162          * @param d
163          *            class descriptor for variable we want to check implements Map
164          * @return true iff the descriptor corresponds to an implementor of Map
165          */
166         private static boolean implementsMap(ClassDescriptor d) {
167             while (d != null) {
168                 try {
169                     // True if variable is itself declared as a Map
170                     if ("java.util.Map".equals(d.getDottedClassName())) {
171                         return true;
172                     }
173                     XClass classNameAndInfo = Global.getAnalysisCache().getClassAnalysis(XClass.class, d);
174                     ClassDescriptor is[] = classNameAndInfo.getInterfaceDescriptorList();
175                     d = classNameAndInfo.getSuperclassDescriptor();
176                     for (ClassDescriptor i : is) {
177                         if ("java.util.Map".equals(i.getDottedClassName())) {
178                             return true;
179                         }
180                     }
181                 } catch (CheckedAnalysisException e) {
182                     d = null;
183                 }
184             }
185             return false;
186         }
187     
188         private int handleStore(int storeRegister, int current) {
189             if(storeRegister == current) {
190                 return NOT_FOUND;
191             }
192             if(current == IN_STACK) {
193                 return storeRegister;
194             }
195             return current;
196         }
197     
198         private void handleStore(int register) {
199             keySetRegister = handleStore(register, keySetRegister);
200             iteratorRegister = handleStore(register, iteratorRegister);
201             keyRegister = handleStore(register, keyRegister);
202         }
203     
204         private void removedFromStack(boolean includeKey) {
205             if(keySetRegister == IN_STACK) {
206                 keySetRegister = NOT_FOUND;
207             }
208             if(iteratorRegister == IN_STACK) {
209                 iteratorRegister = NOT_FOUND;
210             }
211             if(keyRegister == IN_STACK && includeKey) {
212                 keyRegister = NOT_FOUND;
213             }
214         }
215     
216         @Override
217         public void sawOpcode(int seen) {
218             boolean loadedPreserved = false;
219             if(isRegisterStore() && !isRegisterLoad()) {
220                 handleStore(getRegisterOperand());
221             } else {
222                 switch (seen) {
223                 case INVOKEINTERFACE:
224                 case INVOKEVIRTUAL:
225                     if (!loadedVariable.none() &&
226                             "keySet".equals(getNameConstantOperand()) && "()Ljava/util/Set;".equals(getSigConstantOperand())
227                             // Following check solves sourceforge bug 1830576
228                             && implementsMap(getClassDescriptorOperand())) {
229                         mapVariable = loadedVariable;
230                         removedFromStack(true);
231                         keySetRegister = IN_STACK;
232                     } else if ((keySetRegister == IN_STACK || loadedVariable.isRegister(keySetRegister))
233                             && "iterator".equals(getNameConstantOperand()) && "()Ljava/util/Iterator;".equals(getSigConstantOperand())) {
234                         removedFromStack(true);
235                         iteratorRegister = IN_STACK;
236                     } else if ((iteratorRegister == IN_STACK || loadedVariable.isRegister(iteratorRegister))
237                             && "next".equals(getNameConstantOperand())
238                             && "()Ljava/lang/Object;".equals(getSigConstantOperand())) {
239                         removedFromStack(true);
240                         keyRegister = IN_STACK;
241                     } else if (mapAndKeyLoaded && "get".equals(getNameConstantOperand())
242                             && "(Ljava/lang/Object;)Ljava/lang/Object;".equals(getSigConstantOperand())) {
243                         MethodAnnotation ma = MethodAnnotation.fromVisitedMethod(this);
244                         bugAccumulator.accumulateBug(mapVariable
245                                 .annotate(new BugInstance(this, "WMI_WRONG_MAP_ITERATOR", NORMAL_PRIORITY).addClass(this).addMethod(ma)),
246                                 this);
247                         reset();
248                     } else if(("intValue".equals(getNameConstantOperand()) && "java/lang/Integer".equals(getClassConstantOperand())) ||
249                             ("longValue".equals(getNameConstantOperand()) && "java/lang/Long".equals(getClassConstantOperand())) ||
250                             ("doubleValue".equals(getNameConstantOperand()) && "java/lang/Double".equals(getClassConstantOperand())) ||
251                             ("floatValue".equals(getNameConstantOperand()) && "java/lang/Float".equals(getClassConstantOperand()))) {
252                         removedFromStack(false);
253                     } else {
254                         removedFromStack(true);
255                     }
256                     break;
257                 case INVOKESTATIC:
258                     if ("valueOf".equals(getNameConstantOperand())
259                             && ("java/lang/Integer".equals(getClassConstantOperand())
260                                     || "java/lang/Long".equals(getClassConstantOperand())
261                                     || "java/lang/Double".equals(getClassConstantOperand()) || "java/lang/Float"
262                                     .equals(getClassConstantOperand()))) {
263                         loadedPreserved = true;
264                     }
265                     removedFromStack(true);
266                     break;
267                 case CHECKCAST:
268                     removedFromStack(false);
269                     break;
270                 default:
271                     removedFromStack(true);
272                 }
273             }
274             if(!loadedPreserved) {
275                 boolean mapLoaded = !loadedVariable.none() && loadedVariable.same(mapVariable);
276                 loadedVariable = loadedVariable.seen(seen);
277                 mapAndKeyLoaded = mapLoaded && loadedVariable.isRegister(keyRegister);
278             }
279         }
