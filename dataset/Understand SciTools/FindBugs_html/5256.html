88         @Override
89         public void parse(ClassNameAndSuperclassInfo.Builder builder) throws InvalidClassFileFormatException {
90             try {
91                 int magic = in.readInt();
92                 if (magic != 0xcafebabe) {
93                     throw new InvalidClassFileFormatException("Classfile header isn't 0xCAFEBABE", expectedClassDescriptor,
94                             codeBaseEntry);
95                 }
96                 int major_version = in.readUnsignedShort();
97                 int minor_version = in.readUnsignedShort();
98                 int constant_pool_count = in.readUnsignedShort();
99     
100                 constantPool = new Constant[constant_pool_count];
101                 for (int i = 1; i < constantPool.length; i++) {
102                     constantPool[i] = readConstant();
103                     if (constantPool[i].tag == IClassConstants.CONSTANT_Double
104                             || constantPool[i].tag == IClassConstants.CONSTANT_Long) {
105                         // Double and Long constants take up two constant pool
106                         // entries
107                         ++i;
108                     }
109                 }
110     
111                 int access_flags = in.readUnsignedShort();
112     
113                 int this_class = in.readUnsignedShort();
114                 ClassDescriptor thisClassDescriptor = getClassDescriptor(this_class);
115     
116                 int super_class = in.readUnsignedShort();
117                 ClassDescriptor superClassDescriptor = getClassDescriptor(super_class);
118     
119                 int interfaces_count = in.readUnsignedShort();
120                 if (interfaces_count < 0) {
121                     throw new InvalidClassFileFormatException(expectedClassDescriptor, codeBaseEntry);
122                 }
123                 ClassDescriptor[] interfaceDescriptorList;
124                 if (interfaces_count == 0) {
125                     interfaceDescriptorList = ClassDescriptor.EMPTY_ARRAY;
126                 } else {
127                     interfaceDescriptorList = new ClassDescriptor[interfaces_count];
128                     for (int i = 0; i < interfaceDescriptorList.length; i++) {
129                         interfaceDescriptorList[i] = getClassDescriptor(in.readUnsignedShort());
130                     }
131                 }
132                 // Extract all references to other classes,
133                 // both CONSTANT_Class entries and also referenced method
134                 // signatures.
135                 Collection<ClassDescriptor> referencedClassDescriptorList = extractReferencedClasses();
136     
137                 builder.setClassDescriptor(thisClassDescriptor);
138                 builder.setSuperclassDescriptor(superClassDescriptor);
139                 builder.setInterfaceDescriptorList(interfaceDescriptorList);
140                 builder.setCodeBaseEntry(codeBaseEntry);
141                 builder.setAccessFlags(access_flags);
142                 builder.setReferencedClassDescriptors(referencedClassDescriptorList);
143                 builder.setClassfileVersion(major_version, minor_version);
144             } catch (IOException e) {
145                 throw new InvalidClassFileFormatException(expectedClassDescriptor, codeBaseEntry, e);
146             }
147         }

154         /**
155          * Extract references to other classes.
156          *
157          * @return array of ClassDescriptors of referenced classes
158          * @throws InvalidClassFileFormatException
159          */
160         private Collection<ClassDescriptor> extractReferencedClasses() throws InvalidClassFileFormatException {
161             Set<ClassDescriptor> referencedClassSet = new HashSet<ClassDescriptor>();
162             for (Constant constant : constantPool) {
163                 if (constant == null) {
164                     continue;
165                 }
166                 if (constant.tag == IClassConstants.CONSTANT_Class) {
167                     @SlashedClassName
168                     String className = getUtf8String((Integer) constant.data[0]);
169                     if (className.indexOf('[') >= 0) {
170                         extractReferencedClassesFromSignature(referencedClassSet, className);
171                     } else if (ClassName.isValidClassName(className)) {
172                         referencedClassSet.add(DescriptorFactory.instance().getClassDescriptor(className));
173                     }
174                 } else if (constant.tag == IClassConstants.CONSTANT_Methodref || constant.tag == IClassConstants.CONSTANT_Fieldref
175                         || constant.tag == IClassConstants.CONSTANT_InterfaceMethodref) {
176                     // Get the target class name
177                     String className = getClassName((Integer) constant.data[0]);
178                     extractReferencedClassesFromSignature(referencedClassSet, className);
179     
180                     // Parse signature to extract class names
181                     String signature = getSignatureFromNameAndType((Integer) constant.data[1]);
182                     extractReferencedClassesFromSignature(referencedClassSet, signature);
183                 }
184             }
185             return referencedClassSet;
186         }
187     
188         public static void extractReferencedClassesFromSignature(Set<ClassDescriptor> referencedClassSet, String signature) {
189             while (signature.length() > 0) {
190                 int start = signature.indexOf('L');
191                 if (start < 0) {
192                     break;
193                 }
194                 int end = signature.indexOf(';', start);
195                 if (end < 0) {
196                     break;
197                 }
198                 @SlashedClassName
199                 String className = signature.substring(start + 1, end);
200                 if (ClassName.isValidClassName(className)) {
201                     referencedClassSet.add(DescriptorFactory.instance().getClassDescriptor(className));
202                 }
203                 signature = signature.substring(end + 1);
204             }
205         }

207         // 8: UTF-8 string
208         // I: int
209         // F: float
210         // L: long
211         // D: double
212         // i: 2-byte constant pool index
213         private static final String[] CONSTANT_FORMAT_MAP = { null,
214             "8", // 1:CONSTANT_Utf8
215             null, // 2:
216             "I", // 3: CONSTANT_Integer
217             "F", // 4: CONSTANT_Float
218             "L", // 5: CONSTANT_Long
219             "D", // 6: CONSTANT_Double
220             "i", // 7: CONSTANT_Class
221             "i", // 8: CONSTANT_String
222             "ii", // 9: CONSTANT_Fieldref
223             "ii", // 10: CONSTANT_Methodref
224             "ii", // 11: CONSTANT_InterfaceMethodref
225             "ii", // 12: CONSTANT_NameAndType
226             null, // 13:
227             null, // 14:
228             "bi", // 15: CONSTANT_MethodHandle
229             "i", // 16: CONSTANT_MethodType
230             null, // 17:
231             "ii", // 18: CONSTANT_InvokeDynamic

235         /**
236          * Read a constant from the constant pool. Return null for
237          *
238          * @return a StaticConstant
239          * @throws InvalidClassFileFormatException
240          * @throws IOException
241          */
242         private Constant readConstant() throws InvalidClassFileFormatException, IOException {
243             int tag = in.readUnsignedByte();
244             if (tag < 0 || tag >= CONSTANT_FORMAT_MAP.length) {
245                 throw new InvalidClassFileFormatException(expectedClassDescriptor, codeBaseEntry);
246             }
247             String format = CONSTANT_FORMAT_MAP[tag];
248             if (format == null) {
249                 throw new InvalidClassFileFormatException(expectedClassDescriptor, codeBaseEntry);
250             }
251     
252             Object[] data = new Object[format.length()];
253             for (int i = 0; i < format.length(); i++) {
254                 char spec = format.charAt(i);
255                 switch (spec) {
256                 case '8':
257                     data[i] = in.readUTF();
258                     break;
259                 case 'I':
260                     data[i] = in.readInt();
261                     break;
262                 case 'F':
263                     data[i] = Float.valueOf(in.readFloat());
264                     break;
265                 case 'L':
266                     data[i] = in.readLong();
267                     break;
268                 case 'D':
269                     data[i] = Double.valueOf(in.readDouble());
270                     break;
271                 case 'i':
272                     data[i] = in.readUnsignedShort();
273                     break;
274                 case 'b':
275                     data[i] = in.readUnsignedByte();
276                     break;
277                 default:
278                     throw new IllegalStateException();
279                 }
280             }
281     
282             return new Constant(tag, data);
283         }

285         /**
286          * Get a class name from a CONSTANT_Class. Note that this may be an array
287          * (e.g., "[Ljava/lang/String;").
288          *
289          * @param index
290          *            index of the constant
291          * @return the class name
292          * @throws InvalidClassFileFormatException
293          */
294         private @SlashedClassName
295         String getClassName(int index) throws InvalidClassFileFormatException {
296             if (index == 0) {
297                 return null;
298             }
299     
300             checkConstantPoolIndex(index);
301             Constant constant = constantPool[index];
302             checkConstantTag(constant, IClassConstants.CONSTANT_Class);
303     
304             int refIndex = ((Integer) constant.data[0]).intValue();
305             String stringValue = getUtf8String(refIndex);
306     
307             return stringValue;
308         }

310         /**
311          * Get the ClassDescriptor of a class referenced in the constant pool.
312          *
313          * @param index
314          *            index of the referenced class in the constant pool
315          * @return the ClassDescriptor of the referenced class
316          * @throws InvalidClassFileFormatException
317          */
318         private ClassDescriptor getClassDescriptor(int index) throws InvalidClassFileFormatException {
319             @SlashedClassName
320             String className = getClassName(index);
321             return className != null ? DescriptorFactory.instance().getClassDescriptor(className) : null;
322         }

324         /**
325          * Get the UTF-8 string constant at given constant pool index.
326          *
327          * @param refIndex
328          *            the constant pool index
329          * @return the String at that index
330          * @throws InvalidClassFileFormatException
331          */
332         private String getUtf8String(int refIndex) throws InvalidClassFileFormatException {
333             checkConstantPoolIndex(refIndex);
334             Constant refConstant = constantPool[refIndex];
335             checkConstantTag(refConstant, IClassConstants.CONSTANT_Utf8);
336             return (String) refConstant.data[0];
337         }

339         /**
340          * Check that a constant pool index is valid.
341          *
342          * @param index
343          *            the index to check
344          * @throws InvalidClassFileFormatException
345          *             if the index is not valid
346          */
347         private void checkConstantPoolIndex(int index) throws InvalidClassFileFormatException {
348             if (index < 0 || index >= constantPool.length || constantPool[index] == null) {
349                 throw new InvalidClassFileFormatException(expectedClassDescriptor, codeBaseEntry);
350             }
351         }

354          * Check that a constant has the expected tag.
355          *
356          * @param constant
357          *            the constant to check
358          * @param expectedTag
359          *            the expected constant tag
360          * @throws InvalidClassFileFormatException
361          *             if the constant's tag does not match the expected tag
362          */
363         private void checkConstantTag(Constant constant, int expectedTag) throws InvalidClassFileFormatException {
364             if (constant.tag != expectedTag) {
365                 throw new InvalidClassFileFormatException(expectedClassDescriptor, codeBaseEntry);
366             }
367         }
