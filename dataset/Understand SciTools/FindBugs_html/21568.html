20     package edu.umd.cs.findbugs.classfile;
21     
22     import edu.umd.cs.findbugs.internalAnnotations.SlashedClassName;
23     
24     /**
25      * Common superclass for FieldDescriptor and MethodDescriptor.
26      *
27      * @author David Hovemeyer
28      */
29     public abstract class FieldOrMethodDescriptor implements FieldOrMethodName {
30     
31         private final @SlashedClassName
32         String slashedClassName;
33     
34         private final String name;
35     
36         private final String signature;
37     
38         private final boolean isStatic;
39     
40         private int cachedHashCode;
41     
42         private final int nameSigHashCode;
43     
44         public FieldOrMethodDescriptor(@SlashedClassName String slashedClassName, String name, String signature, boolean isStatic) {
45             assert slashedClassName.indexOf('.') == -1 : "class name not in VM format: " + slashedClassName;
46     
47             this.slashedClassName = DescriptorFactory.canonicalizeString(slashedClassName);
48             this.name = DescriptorFactory.canonicalizeString(name);
49             this.signature = DescriptorFactory.canonicalizeString(signature);
50             this.isStatic = isStatic;
51             this.nameSigHashCode = getNameSigHashCode(this.name, this.signature);
52         }
53     
54         public static int getNameSigHashCode(String name, String signature) {
55             return name.hashCode() * 3119 + signature.hashCode() * 131;
56         }
57     
58         public int getNameSigHashCode() {
59             return nameSigHashCode;
60         }
61     
62         /**
63          *
64          *
65          * @return Returns the class name
66          */
67         public @SlashedClassName
68         String getSlashedClassName() {
69             return slashedClassName;
70         }
71     
72         /**
73          * @return a ClassDescriptor for the method's class
74          */
75         @Override
76         public ClassDescriptor getClassDescriptor() {
77             return DescriptorFactory.createClassDescriptor(slashedClassName);
78         }
79     
80         /**
81          * @return Returns the method name
82          */
83         @Override
84         public String getName() {
85             return name;
86         }
87     
88         /**
89          * @return Returns the method signature
90          */
91         @Override
92         public String getSignature() {
93             return signature;
94         }
95     
96         /**
97          * @return Returns true if method is static, false if not
98          */
99         @Override
100         public boolean isStatic() {
101             return isStatic;
102         }
103     
104         protected int compareTo(FieldOrMethodName o) {
105             int cmp;
106             cmp = this.getClassDescriptor().compareTo(o.getClassDescriptor());
107             if (cmp != 0) {
108                 return cmp;
109             }
110             cmp = this.name.compareTo(o.getName());
111             if (cmp != 0) {
112                 return cmp;
113             }
114             cmp = this.signature.compareTo(o.getSignature());
115             if (cmp != 0) {
116                 return cmp;
117             }
118             return (this.isStatic ? 1 : 0) - (o.isStatic() ? 1 : 0);
119         }
120     
121     
122         protected boolean haveEqualFields(FieldOrMethodDescriptor other) {
123             return this.isStatic == other.isStatic && this.slashedClassName.equals(other.slashedClassName)
124                     && this.name.equals(other.name) && this.signature.equals(other.signature);
125         }
126     
127         /*
128          * (non-Javadoc)
129          *
130          * @see java.lang.Object#hashCode()
131          */
132         @Override
133         public final int hashCode() {
134             if (cachedHashCode == 0) {
135                 cachedHashCode = slashedClassName.hashCode() * 7919 + nameSigHashCode + (isStatic ? 1 : 0);
136             }
137             return cachedHashCode;
138         }
139     
140         /*
141          * (non-Javadoc)
142          *
143          * @see java.lang.Object#toString()
144          */
145         @Override
146         public String toString() {
147             return (isStatic ? "static " : "") + getClassDescriptor().getDottedClassName() + "." + name + signature;
148         }
149     
150         public static int compareTo(FieldOrMethodDescriptor thas, FieldOrMethodDescriptor that) {
151             int result = thas.slashedClassName.compareTo(that.slashedClassName);
152             if (result != 0) {
153                 return result;
154             }
155             result = thas.name.compareTo(that.name);
156             if (result != 0) {
157                 return result;
158             }
159             result = thas.signature.compareTo(that.signature);
160             if (result != 0) {
161                 return result;
162             }
163             result = (thas.isStatic ? 1 : 0) - (that.isStatic ? 1 : 0);
164             return result;
165         }
166     }
