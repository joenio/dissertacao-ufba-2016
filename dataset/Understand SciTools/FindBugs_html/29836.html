20     package edu.umd.cs.findbugs.detect;
21     
22     import java.util.HashSet;
23     
24     import edu.umd.cs.findbugs.BugReporter;
25     import edu.umd.cs.findbugs.BytecodeScanningDetector;
26     import edu.umd.cs.findbugs.NonReportingDetector;
27     import edu.umd.cs.findbugs.ba.AnalysisContext;
28     import edu.umd.cs.findbugs.ba.XFactory;
29     import edu.umd.cs.findbugs.ba.XField;
30     import edu.umd.cs.findbugs.ba.ch.Subtypes2;
31     import edu.umd.cs.findbugs.classfile.ClassDescriptor;
32     
33     /**
34      * Detector to find private methods that are never called.
35      */
36     public class CalledMethods extends BytecodeScanningDetector implements NonReportingDetector {
37         boolean emptyArrayOnTOS;
38     
39         HashSet<XField> emptyArray = new HashSet<XField>();
40     
41         HashSet<XField> nonEmptyArray = new HashSet<XField>();
42     
43         XFactory xFactory = AnalysisContext.currentXFactory();
44     
45         public CalledMethods(BugReporter bugReporter) {
46     
47         }
48     
49         @Override
50         public void sawOpcode(int seen) {
51     
52             if ((seen == PUTFIELD || seen == PUTSTATIC)) {
53                 XField f = getXFieldOperand();
54                 if (f != null) {
55                     if (f.isFinal() || !f.isProtected() && !f.isPublic()) {
56                         if (emptyArrayOnTOS) {
57                             emptyArray.add(f);
58                         } else {
59                             nonEmptyArray.add(f);
60                         }
61                     }
62                 }
63     
64             }
65             emptyArrayOnTOS = (seen == ANEWARRAY || seen == NEWARRAY || seen == MULTIANEWARRAY && getIntConstant() == 1)
66                     && getPrevOpcode(1) == ICONST_0;
67     
68             if (seen == GETSTATIC || seen == GETFIELD) {
69                 XField f = getXFieldOperand();
70                 if (emptyArray.contains(f) && !nonEmptyArray.contains(f) && f.isFinal()) {
71                     emptyArrayOnTOS = true;
72                 }
73             }
74             switch (seen) {
75             case INVOKEVIRTUAL:
76             case INVOKESPECIAL:
77             case INVOKESTATIC:
78             case INVOKEINTERFACE:
79                 ClassDescriptor c = getClassDescriptorOperand();
80                 Subtypes2 subtypes2 = AnalysisContext.currentAnalysisContext().getSubtypes2();
81                 if (subtypes2.isApplicationClass(c)) {
82                     xFactory.addCalledMethod(getMethodDescriptorOperand());
83                 }
84     
85                 break;
86             default:
87                 break;
88             }
89         }
90     
91         @Override
92         public void report() {
93             emptyArray.removeAll(nonEmptyArray);
94             for (XField f : emptyArray) {
95                 xFactory.addEmptyArrayField(f);
96             }
97             emptyArray.clear();
98         }
99     
100     }
