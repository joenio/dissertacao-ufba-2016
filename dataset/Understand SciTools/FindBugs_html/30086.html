45     /**
46      * Mark methodref constant pool entries of methods that are likely to implement
47      * assertions. This is useful for pruning likely false paths.
48      *
49      * @author David Hovemeyer
50      */
51     public class AssertionMethods implements Constants {
52     
53         private static final boolean DEBUG = SystemProperties.getBoolean("assertionmethods.debug");
54     
55         /**
56          * Bitset of methodref constant pool indexes referring to likely assertion
57          * methods.
58          */
59         private final BitSet assertionMethodRefSet;
60     
61         private static class UserAssertionMethod {
62             private final String className;
63     
64             private final String methodName;
65     
66             public UserAssertionMethod(String className, String methodName) {
67                 this.className = className;
68                 this.methodName = methodName;
69             }
70     
71             public String getClassName() {
72                 return className;
73             }
74     
75             public String getMethodName() {
76                 return methodName;
77             }
78         }
79     
80         @edu.umd.cs.findbugs.internalAnnotations.StaticConstant
81         private static final List<UserAssertionMethod> userAssertionMethodList = new ArrayList<UserAssertionMethod>();
82     
83         static {
84             String userProperty = SystemProperties.getProperty("findbugs.assertionmethods");
85             if (userProperty != null) {
86                 StringTokenizer tok = new StringTokenizer(userProperty, ",");
87                 while (tok.hasMoreTokens()) {
88                     String fullyQualifiedName = tok.nextToken();
89                     int lastDot = fullyQualifiedName.lastIndexOf('.');
90                     if (lastDot < 0) {
91                         continue;
92                     }
93                     String className = fullyQualifiedName.substring(0, lastDot);
94                     String methodName = fullyQualifiedName.substring(lastDot + 1);
95                     userAssertionMethodList.add(new UserAssertionMethod(className, methodName));
96                 }
97             }
98         }
99     
100         /**
101          * Constructor.
102          *
103          * @param jclass
104          *            the JavaClass containing the methodrefs
105          */
106         public AssertionMethods(JavaClass jclass) {
107             this.assertionMethodRefSet = new BitSet();
108             init(jclass);
109         }
110     
111         private void init(JavaClass jclass) {
112             ConstantPool cp = jclass.getConstantPool();
113             int numConstants = cp.getLength();
114             for (int i = 0; i < numConstants; ++i) {
115                 try {
116                     Constant c = cp.getConstant(i);
117                     if (c instanceof ConstantMethodref) {
118                         ConstantMethodref cmr = (ConstantMethodref) c;
119                         ConstantNameAndType cnat = (ConstantNameAndType) cp.getConstant(cmr.getNameAndTypeIndex(),
120                                 CONSTANT_NameAndType);
121                         String methodName = ((ConstantUtf8) cp.getConstant(cnat.getNameIndex(), CONSTANT_Utf8)).getBytes();
122                         String className = cp.getConstantString(cmr.getClassIndex(), CONSTANT_Class).replace('/', '.');
123                         String methodSig = ((ConstantUtf8) cp.getConstant(cnat.getSignatureIndex(), CONSTANT_Utf8)).getBytes();
124     
125                         String classNameLC = className.toLowerCase();
126                         String methodNameLC = methodName.toLowerCase();
127     
128                         boolean voidReturnType = methodSig.endsWith(")V");
129                         boolean boolReturnType = methodSig.endsWith(")Z");
130     
131     
132     
133                         if (DEBUG) {
134                             System.out.print("Is " + className + "." + methodName + " assertion method: " + voidReturnType);
135                         }
136     
137                         if (isUserAssertionMethod(className, methodName)
138                                 || className.endsWith("Assert")
139                                 && methodName.startsWith("is")
140                                 || (voidReturnType || boolReturnType)
141                                 && (classNameLC.indexOf("assert") >= 0 || methodNameLC.startsWith("throw")
142                                 || methodName.startsWith("affirm") || methodName.startsWith("panic")
143                                 || "logTerminal".equals(methodName) || methodName.startsWith("logAndThrow")
144                                 || "insist".equals(methodNameLC) || "usage".equals(methodNameLC)
145                                 || "exit".equals(methodNameLC) || methodNameLC.startsWith("fail")
146                                 || methodNameLC.startsWith("fatal") || methodNameLC.indexOf("assert") >= 0
147                                 || methodNameLC.indexOf("legal") >= 0 || methodNameLC.indexOf("error") >= 0
148                                 || methodNameLC.indexOf("abort") >= 0
149                                 // || methodNameLC.indexOf("check") >= 0
150                                 || methodNameLC.indexOf("failed") >= 0) || "addOrThrowException".equals(methodName)) {
151                             assertionMethodRefSet.set(i);
152                             if (DEBUG) {
153                                 System.out.println("==> YES");
154                             }
155                         } else {
156                             if (DEBUG) {
157                                 System.out.println("==> NO");
158                             }
159                         }
160                     }
161                 } catch (ClassFormatException e) {
162                     // FIXME: should report
163                 }
164             }
165         }
166     
167         private static boolean isUserAssertionMethod(String className, String methodName) {
168             for (UserAssertionMethod uam : userAssertionMethodList) {
169                 if (className.equals(uam.getClassName()) && methodName.equals(uam.getMethodName())) {
170                     return true;
171                 }
172             }
173             return false;
174         }
175     
176         public boolean isAssertionHandle(InstructionHandle handle, ConstantPoolGen cpg) {
177             Instruction ins = handle.getInstruction();
178             if (isAssertionInstruction(ins, cpg)) {
179                 return true;
180             }
181     
182             if (ins instanceof SIPUSH) {
183                 int v = ((SIPUSH) ins).getValue().intValue();
184                 if (v == 500) {
185                     Instruction next = handle.getNext().getInstruction();
186                     if (next instanceof INVOKEINTERFACE) {
187                         INVOKEINTERFACE iInterface = (INVOKEINTERFACE) next;
188                         String className = iInterface.getClassName(cpg);
189                         String fieldName = iInterface.getMethodName(cpg);
190                         if ("javax.servlet.http.HttpServletResponse".equals(className) && "setStatus".equals(fieldName)) {
191                             return true;
192                         }
193     
194                     }
195                 }
196             }
197             return false;
198         }
199     
200         /**
201          * Does the given instruction refer to a likely assertion method?
202          *
203          * @param ins
204          *            the instruction
205          * @return true if the instruction likely refers to an assertion, false if
206          *         not
207          */
208     
209         public boolean isAssertionInstruction(Instruction ins, ConstantPoolGen cpg) {
210     
211             if (ins instanceof InvokeInstruction) {
212                 return isAssertionCall((InvokeInstruction) ins);
213             }
214             if (ins instanceof GETSTATIC) {
215                 GETSTATIC getStatic = (GETSTATIC) ins;
216                 String className = getStatic.getClassName(cpg);
217                 String fieldName = getStatic.getFieldName(cpg);
218                 if ("java.util.logging.Level".equals(className) && "SEVERE".equals(fieldName)) {
219                     return true;
220                 }
221                 if ("org.apache.log4j.Level".equals(className) && ("ERROR".equals(fieldName) || "FATAL".equals(fieldName))) {
222                     return true;
223                 }
224                 return false;
225     
226             }
227             return false;
228         }
229     
230         public boolean isAssertionCall(InvokeInstruction inv) {
231     
232             boolean isAssertionMethod = assertionMethodRefSet.get(inv.getIndex());
233     
234             return isAssertionMethod;
235         }
236     }
