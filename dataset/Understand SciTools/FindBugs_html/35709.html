44     /**
45      * A plan for executing Detectors on an application. Automatically assigns
46      * Detectors to passes and orders Detectors within each pass based on ordering
47      * constraints specified in the plugin descriptor(s).
48      *
49      * @author David Hovemeyer
50      */
51     public class ExecutionPlan {

53         public static final boolean DEBUG = SystemProperties.getBoolean("findbugs.execplan.debug");

69         /**
70          * Constructor. Creates an empty plan.
71          */
72         public ExecutionPlan() {
73             this.pluginList = new LinkedList<Plugin>();
74             this.factoryChooser = new DetectorFactoryChooser() {
75                 @Override
76                 public boolean choose(DetectorFactory factory) {
77                     return true;
78                 }
79     
80                 @Override
81                 public void enable(DetectorFactory factory) {
82                     // OK...
83                 }
84             };
85             this.passList = new LinkedList<AnalysisPass>();
86             this.factoryMap = new HashMap<String, DetectorFactory>();
87             this.interPassConstraintList = new LinkedList<DetectorOrderingConstraint>();
88             this.intraPassConstraintList = new LinkedList<DetectorOrderingConstraint>();
89             this.assignedToPassSet = new HashSet<DetectorFactory>();
90         }

92         public void dispose() {
93             pluginList.clear();
94             factoryChooser = null;
95             passList.clear();
96             factoryMap.clear();
97             interPassConstraintList.clear();
98             intraPassConstraintList.clear();
99             assignedToPassSet.clear();

119         /**
120          * Add a Plugin whose Detectors should be added to the execution plan.
121          */
122         public void addPlugin(Plugin plugin) throws OrderingConstraintException {
123             if (DEBUG) {
124                 System.out.println("Adding plugin " + plugin.getPluginId() + " to execution plan");
125             }
126     
127             pluginList.add(plugin);
128     
129             // Add ordering constraints
130             copyTo(plugin.interPassConstraintIterator(), interPassConstraintList);
131             copyTo(plugin.intraPassConstraintIterator(), intraPassConstraintList);
132     
133             // Add detector factories
134             for (DetectorFactory factory : plugin.getDetectorFactories()) {
135                 if (DEBUG) {
136                     System.out.println("  Detector factory " + factory.getShortName());
137                 }
138                 if (factoryMap.put(factory.getFullName(), factory) != null) {
139                     throw new OrderingConstraintException("Detector " + factory.getFullName() + " is defined by more than one plugin");
140                 }
141             }
142         }

144         /**
145          * Build the execution plan. Using the ordering constraints specified in the
146          * plugin descriptor(s), assigns Detectors to passes and orders the
147          * Detectors within those passes.
148          */
149         public void build() throws OrderingConstraintException {
150     
151             for (DetectorFactory detectorFactory : factoryMap.values()) {
152                 detectorFactory.setEnabledButNonReporting(false);
153             }
154     
155             ArrayList<DetectorOrderingConstraint> allConstraints = new ArrayList<DetectorOrderingConstraint>(
156                     interPassConstraintList.size() + intraPassConstraintList.size());
157             allConstraints.addAll(interPassConstraintList);
158             allConstraints.addAll(intraPassConstraintList);
159     
160             Map<String, DetectorNode> nodeMapAll = new HashMap<String, DetectorNode>();
161             ConstraintGraph allPassConstraintGraph = buildConstraintGraph(nodeMapAll,
162                     new HashSet<DetectorFactory>(factoryMap.values()), allConstraints);
163             boolean change;
164             do {
165                 change = false;
166                 for (Iterator<DetectorNode> i = allPassConstraintGraph.vertexIterator(); i.hasNext();) {
167                     DetectorNode end = i.next();
168                     if (factoryChooser.choose(end.getFactory())) {
169                         for (Iterator<ConstraintEdge> j = allPassConstraintGraph.incomingEdgeIterator(end); j.hasNext();) {
170                             ConstraintEdge edge = j.next();
171                             DetectorNode start = edge.getSource();
172                             DetectorFactory startFactory = start.getFactory();
173                             //
174                             // Note that we only enable an otherwise-disabled
175                             // detector
176                             // if it was the earlier detector in a single-source
177                             // constraint.
178                             //
179                             if (!factoryChooser.choose(startFactory) && edge.isSingleSource()) {
180                                 factoryChooser.enable(startFactory);
181                                 change = true;
182                                 if (DEBUG || FindBugs2.DEBUG) {
183                                     System.out.println("Dependences force enabling of " + startFactory.getFullName());
184                                 }
185                             }
186     
187                         }
188                     }
189     
190                 }
191             } while (change);
192     
193             for (Iterator<Map.Entry<String, DetectorFactory>> i = factoryMap.entrySet().iterator(); i.hasNext();) {
194                 Map.Entry<String, DetectorFactory> e = i.next();
195                 if (!factoryChooser.choose(e.getValue())) {
196                     i.remove();
197                 }
198             }
199     
200             // Build inter-pass constraint graph
201             Map<String, DetectorNode> nodeMap = new HashMap<String, DetectorNode>();
202             ConstraintGraph interPassConstraintGraph = buildConstraintGraph(nodeMap,
203                     new HashSet<DetectorFactory>(factoryMap.values()), interPassConstraintList);
204             if (DEBUG) {
205                 System.out.println(interPassConstraintGraph.getNumVertices() + " nodes in inter-pass constraint graph");
206             }
207     
208             // Build list of analysis passes.
209             // This will assign all detectors referenced in inter- or intra-pass
210             // ordering constraints to passes. Detectors with any ordering
211             // constraint will be left unassigned.
212             buildPassList(interPassConstraintGraph);
213     
214             // Sort each pass by intra-pass ordering constraints.
215             // This may assign some previously unassigned detectors to passes.
216             for (AnalysisPass pass : passList) {
217                 sortPass(intraPassConstraintList, factoryMap, pass);
218             }
219     
220             // If there are any unassigned detectors remaining,
221             // add them to the final pass.
222             if (factoryMap.size() > assignedToPassSet.size()) {
223                 AnalysisPass lastPass;
224                 if (passList.isEmpty()) {
225                     lastPass = new AnalysisPass();
226                     addPass(lastPass);
227                 } else {
228                     lastPass = passList.getLast();
229                 }
230     
231                 Set<DetectorFactory> unassignedSet = getUnassignedSet();
232                 for (DetectorFactory factory : unassignedSet) {
233                     assignToPass(factory, lastPass);
234                 }
235                 appendDetectorsToPass(unassignedSet, lastPass);
236             }
237             if (DEBUG) {
238                 print();
239             }
240         }

258         private static <T> void copyTo(Iterator<T> iter, Collection<T> dest) {
259             while (iter.hasNext()) {
260                 dest.add(iter.next());
261             }

264         /**
265          * Build a constraint graph. This represents ordering constraints between
266          * Detectors. A topological sort of the constraint graph will yield a
267          * correct ordering of the detectors (which may mean either passes or an
268          * ordering within a single pass, depending on whether the constraints are
269          * inter-pass or intra-pass).
270          *
271          * @param nodeMap
272          *            map to be populated with detector class names to constraint
273          *            graph nodes for those detectors
274          * @param factorySet
275          *            build the graph using these DetectorFactories as nodes
276          * @param constraintList
277          *            List of ordering constraints
278          * @return the ConstraintGraph
279          */
280         private ConstraintGraph buildConstraintGraph(Map<String, DetectorNode> nodeMap, Set<DetectorFactory> factorySet,
281                 List<DetectorOrderingConstraint> constraintList) {
282     
283             ConstraintGraph result = new ConstraintGraph();
284     
285             for (DetectorOrderingConstraint constraint : constraintList) {
286                 Set<DetectorNode> earlierSet = addOrCreateDetectorNodes(constraint.getEarlier(), nodeMap, factorySet, result);
287                 Set<DetectorNode> laterSet = addOrCreateDetectorNodes(constraint.getLater(), nodeMap, factorySet, result);
288     
289                 createConstraintEdges(result, earlierSet, laterSet, constraint);
290             }
291     
292             return result;
293         }
294     
295         private Set<DetectorFactory> selectDetectors(DetectorFactorySelector selector, Set<DetectorFactory> candidateSet) {
296             Set<DetectorFactory> result = new HashSet<DetectorFactory>();
297             for (DetectorFactory factory : candidateSet) {
298                 if (selector.selectFactory(factory)) {
299                     result.add(factory);
300                 }
301             }
302             return result;
303         }
304     
305         private Set<DetectorNode> addOrCreateDetectorNodes(DetectorFactorySelector selector, Map<String, DetectorNode> nodeMap,
306                 Set<DetectorFactory> factorySet, ConstraintGraph constraintGraph) {
307             HashSet<DetectorNode> result = new HashSet<DetectorNode>();
308     
309             Set<DetectorFactory> chosenSet = selectDetectors(selector, factorySet);
310     
311             for (DetectorFactory factory : chosenSet) {
312                 DetectorNode node = addOrCreateDetectorNode(factory, nodeMap, constraintGraph);
313                 result.add(node);
314             }
315     
316             return result;
317         }
318     
319         private DetectorNode addOrCreateDetectorNode(DetectorFactory factory, Map<String, DetectorNode> nodeMap,
320                 ConstraintGraph constraintGraph) {
321             DetectorNode node = nodeMap.get(factory.getFullName());
322             if (node == null) {
323                 node = new DetectorNode(factory);
324                 nodeMap.put(factory.getFullName(), node);
325                 constraintGraph.addVertex(node);
326             }
327             return node;
328         }

330         private void createConstraintEdges(ConstraintGraph result, Set<DetectorNode> earlierSet, Set<DetectorNode> laterSet,
331                 DetectorOrderingConstraint constraint)  {
332     
333             // It is perfectly fine for a constraint to produce no edges
334             // if any detector it specifies is not enabled.
335             if (earlierSet.isEmpty() || laterSet.isEmpty()) {
336                 return;
337             }
338     
339             for (DetectorNode earlier : earlierSet) {
340                 for (DetectorNode later : laterSet) {
341                     ConstraintEdge edge = result.createEdge(earlier, later);
342                     edge.setConstraint(constraint);
343                 }
344             }
345         }

347         private void buildPassList(ConstraintGraph constraintGraph) throws OrderingConstraintException {
348     
349             int passCount = 0;
350             while (constraintGraph.getNumVertices() > 0) {
351                 List<DetectorNode> inDegreeZeroList = new LinkedList<DetectorNode>();
352                 // Get all of the detectors nodes with in-degree 0.
353                 // These have no unsatisfied prerequisites, and thus can
354                 // be chosen for the current pass.
355                 for (Iterator<DetectorNode> i = constraintGraph.vertexIterator(); i.hasNext();) {
356                     DetectorNode node = i.next();
357                     if (constraintGraph.getNumIncomingEdges(node) == 0) {
358                         inDegreeZeroList.add(node);
359                     } else if (DEBUG ) {
360                         System.out.println("Can't schedule " + node.getFactory().getShortName());
361                         Iterator<ConstraintEdge> incomingEdgeIterator = constraintGraph.incomingEdgeIterator(node);
362                         while (incomingEdgeIterator.hasNext()) {
363                             ConstraintEdge edge = incomingEdgeIterator.next();
364                             System.out.println("  requires " + edge.getSource().getFactory().getShortName());
365     
366                         }
367                     }
368     
369                 }
370     
371                 if (inDegreeZeroList.isEmpty()) {
372                     throw new OrderingConstraintException("Cycle in inter-pass ordering constraints");
373                 }
374     
375                 // Remove all of the chosen detectors from the constraint graph.
376                 for (DetectorNode node : inDegreeZeroList) {
377                     constraintGraph.removeVertex(node);
378                 }
379     
380                 // Create analysis pass and add detector factories.
381                 // Note that this just makes the detectors members of the pass:
382                 // it doesn't assign them a position in the pass.
383                 AnalysisPass pass = new AnalysisPass();
384                 addPass(pass);
385                 passCount++;
386                 for (DetectorNode node : inDegreeZeroList) {
387                     assignToPass(node.getFactory(), pass);
388                 }
389     
390             }
391         }

393         private void addPass(AnalysisPass pass) {
394             if (DEBUG) {
395                 System.out.println("Adding pass " + passList.size());
396             }
397             passList.add(pass);
398         }

400         private void sortPass(List<DetectorOrderingConstraint> constraintList, Map<String, DetectorFactory> factoryMap,
401                 AnalysisPass pass) throws OrderingConstraintException {
402     
403             // Build set of all (initial) detectors in pass
404             Set<DetectorFactory> detectorSet = new HashSet<DetectorFactory>(pass.getMembers());
405             if (DEBUG) {
406                 System.out.println(detectorSet.size() + " detectors currently in this pass");
407             }
408     
409             // Build list of ordering constraints in this pass only
410             List<DetectorOrderingConstraint> passConstraintList = new LinkedList<DetectorOrderingConstraint>();
411             for (DetectorOrderingConstraint constraint : constraintList) {
412                 // Does this constraint specify any detectors in this pass?
413                 // If so, add it to the pass constraints
414                 if (selectDetectors(constraint.getEarlier(), detectorSet).size() > 0
415                         || selectDetectors(constraint.getLater(), detectorSet).size() > 0) {
416                     passConstraintList.add(constraint);
417                 }
418             }
419             if (DEBUG) {
420                 System.out.println(passConstraintList.size() + " constraints are applicable for this pass");
421             }
422     
423             // Build set of all detectors available to be added to this pass
424             HashSet<DetectorFactory> availableSet = new HashSet<DetectorFactory>();
425             availableSet.addAll(detectorSet);
426             availableSet.addAll(getUnassignedSet());
427     
428             // Build intra-pass constraint graph
429             Map<String, DetectorNode> nodeMap = new HashMap<String, DetectorNode>();
430             ConstraintGraph constraintGraph = buildConstraintGraph(nodeMap, availableSet, passConstraintList);
431             if (DEBUG) {
432                 System.out.println("Pass constraint graph:");
433                 dumpGraph(constraintGraph);
434             }
435     
436             // See if any detectors were brought into the pass by an intrapass
437             // ordering constraint.
438             // Assign them to the pass officially.
439             for (DetectorNode node : nodeMap.values()) {
440                 if (!pass.contains(node.getFactory())) {
441                     assignToPass(node.getFactory(), pass);
442                 }
443             }
444     
445             // Perform DFS, check for cycles
446             DepthFirstSearch<ConstraintGraph, ConstraintEdge, DetectorNode> dfs = new DepthFirstSearch<ConstraintGraph, ConstraintEdge, DetectorNode>(
447                     constraintGraph);
448             dfs.search();
449             if (dfs.containsCycle()) {
450                 throw new OrderingConstraintException("Cycle in intra-pass ordering constraints!");
451             }
452     
453             // Do a topological sort to put the detectors in the pass
454             // in the right order.
455             for (Iterator<DetectorNode> i = dfs.topologicalSortIterator(); i.hasNext();) {
456                 DetectorNode node = i.next();
457                 appendToPass(node.getFactory(), pass);
458             }
459     
460             // Add any detectors not explicitly involved in intra-pass ordering
461             // constraints
462             // to the end of the pass.
463             appendDetectorsToPass(pass.getUnpositionedMembers(), pass);
464         }

466         private Set<DetectorFactory> getUnassignedSet() {
467             Set<DetectorFactory> unassignedSet = new HashSet<DetectorFactory>();
468             unassignedSet.addAll(factoryMap.values());
469             unassignedSet.removeAll(assignedToPassSet);
470             return unassignedSet;
471         }

473         /**
474          * Make a DetectorFactory a member of an AnalysisPass.
475          */
476         private void assignToPass(DetectorFactory factory, AnalysisPass pass) {
477             pass.addToPass(factory);
478             assignedToPassSet.add(factory);
479         }
480     
481         /**
482          * Append a DetectorFactory to the end position in an AnalysisPass. The
483          * DetectorFactory must be a member of the pass.
484          */
485         private void appendToPass(DetectorFactory factory, AnalysisPass pass) {
486             pass.append(factory);
487         }
488     
489         private void appendDetectorsToPass(Collection<DetectorFactory> detectorSet, AnalysisPass pass)
490         {
491             DetectorFactory[] unassignedList = detectorSet.toArray(new DetectorFactory[detectorSet.size()]);
492             Arrays.sort(unassignedList, new Comparator<DetectorFactory>() {
493                 @Override
494                 public int compare(DetectorFactory a, DetectorFactory b) {
495                     // Sort first by plugin id...
496                     int cmp = a.getPlugin().getPluginId().compareTo(b.getPlugin().getPluginId());
497                     if (cmp != 0) {
498                         return cmp;
499                     }
500                     // Then by order specified in plugin descriptor
501                     return a.getPositionSpecifiedInPluginDescriptor() - b.getPositionSpecifiedInPluginDescriptor();
502                 }
503             });
504             for (DetectorFactory factory : unassignedList) {
505                 appendToPass(factory, pass);
506             }
507         }

509         private void print() {
510             System.out.println("\nExecution plan:");
511             int passCount = 0;
512             for (Iterator<AnalysisPass> i = passList.iterator(); i.hasNext(); ++passCount) {
513                 System.out.println("Pass " + passCount);
514                 AnalysisPass pass = i.next();
515                 for (Iterator<DetectorFactory> j = pass.iterator(); j.hasNext();) {
516                     DetectorFactory factory = j.next();
517                     System.out.println("  " + factory.getShortName());
518                 }
519             }
520             System.out.println();
521         }

523         private void dumpGraph(ConstraintGraph graph) {
524             for (Iterator<ConstraintEdge> i = graph.edgeIterator(); i.hasNext();) {
525                 ConstraintEdge edge = i.next();
526                 System.out.println(edge.getSource().getFactory().getShortName() + " ==> "
527                         + edge.getTarget().getFactory().getShortName());
528             }
529         }

531         public static void main(String[] argv) throws Exception {
532             DetectorFactoryCollection detectorFactoryCollection = DetectorFactoryCollection.instance();
533     
534             ExecutionPlan execPlan = new ExecutionPlan();
535     
536             for (String pluginId : argv) {
537                 Plugin plugin = detectorFactoryCollection.getPluginById(pluginId);
538                 if (plugin != null) {
539                     execPlan.addPlugin(plugin);
540                 }
541             }
542     
543             execPlan.build();
544     
545             System.out.println(execPlan.getNumPasses() + " passes in plan");
546             execPlan.print();
547         }
