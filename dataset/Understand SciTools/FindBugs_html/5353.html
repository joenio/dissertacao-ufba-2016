214         /*
215          * (non-Javadoc)
216          *
217          * @see
218          * edu.umd.cs.findbugs.classfile.IClassPathBuilder#build(edu.umd.cs.findbugs
219          * .classfile.IClassPath,
220          * edu.umd.cs.findbugs.classfile.IClassPathBuilderProgress)
221          */
222         @Override
223         public void build(IClassPath classPath, IClassPathBuilderProgress progress) throws CheckedAnalysisException, IOException,
224         InterruptedException {
225             // Discover all directly and indirectly referenced codebases
226             processWorkList(classPath, projectWorkList, progress);
227     
228             // If not already located, try to locate any additional codebases
229             // containing classes required for analysis.
230             if (!discoveredCodeBaseList.isEmpty()) {
231                 locateCodebasesRequiredForAnalysis(classPath, progress);
232             }
233     
234             // Add all discovered codebases to the classpath
235             for (DiscoveredCodeBase discoveredCodeBase : discoveredCodeBaseList) {
236                 classPath.addCodeBase(discoveredCodeBase.getCodeBase());
237             }
238     
239             Set<ClassDescriptor> appClassSet = new HashSet<ClassDescriptor>();
240     
241             // Build collection of all application classes.
242             // Also, add resource name -> codebase entry mappings for application
243             // classes.
244             for (DiscoveredCodeBase discoveredCodeBase : discoveredCodeBaseList) {
245                 if (!discoveredCodeBase.getCodeBase().isApplicationCodeBase()) {
246                     continue;
247                 }
248     
249                 codeBaseEntryLoop: for (ICodeBaseIterator i = discoveredCodeBase.iterator(); i.hasNext();) {
250                     ICodeBaseEntry entry = i.next();
251                     if (!DescriptorFactory.isClassResource(entry.getResourceName())) {
252                         continue;
253                     }
254     
255                     ClassDescriptor classDescriptor = entry.getClassDescriptor();
256                     if (classDescriptor == null) {
257                         throw new IllegalStateException();
258                     }
259     
260                     if (appClassSet.contains(classDescriptor)) {
261                         // An earlier entry takes precedence over this class
262                         continue codeBaseEntryLoop;
263                     }
264                     appClassSet.add(classDescriptor);
265                     appClassList.add(classDescriptor);
266     
267                     classPath.mapResourceNameToCodeBaseEntry(entry.getResourceName(), entry);
268                 }
269             }
270     
271             if (DEBUG) {
272                 System.out.println("Classpath:");
273                 dumpCodeBaseList(classPath.appCodeBaseIterator(), "Application codebases");
274                 dumpCodeBaseList(classPath.auxCodeBaseIterator(), "Auxiliary codebases");
275             }
276         }

278         /**
279          * Make an effort to find the codebases containing any files required for
280          * analysis.
281          */
282         private void locateCodebasesRequiredForAnalysis(IClassPath classPath, IClassPathBuilderProgress progress)
283                 throws InterruptedException, IOException, ResourceNotFoundException {
284             boolean foundJavaLangObject = false;
285             boolean foundFindBugsAnnotations = false;
286             boolean foundJSR305Annotations = false;
287     
288             for (DiscoveredCodeBase discoveredCodeBase : discoveredCodeBaseList) {
289                 if (!foundJavaLangObject) {
290                     foundJavaLangObject = probeCodeBaseForResource(discoveredCodeBase, "java/lang/Object.class");
291                 }
292                 if (!foundFindBugsAnnotations) {
293                     foundFindBugsAnnotations = probeCodeBaseForResource(discoveredCodeBase,
294                             "edu/umd/cs/findbugs/annotations/Nonnull.class");
295                 }
296                 if (!foundJSR305Annotations) {
297                     foundJSR305Annotations = probeCodeBaseForResource(discoveredCodeBase, "javax/annotation/meta/TypeQualifier.class");
298                     if (DEBUG) {
299                         System.out.println("foundJSR305Annotations: " + foundJSR305Annotations);
300                     }
301                 }
302             }
303     
304             if (!foundJavaLangObject) {
305                 processWorkList(classPath, buildSystemCodebaseList(), progress);
306             }
307     
308             // If we're running findbugs-full.jar, IT contains the contents
309             // of jsr305.jar and annotations.jar. So, add it to the classpath.
310             if (runningFindBugsFullJar()) {
311                 processWorkList(classPath, buildFindBugsFullJarCodebaseList(), progress);
312                 return;
313             }
314     
315             // Not running findbugs-full.jar: try to find jsr305.jar and
316             // annotations.jar.
317     
318             if (!foundFindBugsAnnotations) {
319                 processWorkList(classPath, buildFindBugsAnnotationCodebaseList(), progress);
320             }
321             if (!foundJSR305Annotations) {
322                 processWorkList(classPath, buildJSR305AnnotationsCodebaseList(), progress);
323             }
324         }

372         private LinkedList<WorkListItem> buildSystemCodebaseList() {
373             // This method is based on the
374             // org.apache.bcel.util.ClassPath.getClassPath()
375             // method.
376     
377             LinkedList<WorkListItem> workList = new LinkedList<WorkListItem>();
378     
379             String bootClassPath = SystemProperties.getProperty("sun.boot.class.path");
380             // Seed worklist with system codebases.
381             // addWorkListItemsForClasspath(workList,
382             // SystemProperties.getProperty("java.class.path"));
383             addWorkListItemsForClasspath(workList, bootClassPath);
384             String extPath = SystemProperties.getProperty("java.ext.dirs");
385             if (extPath != null) {
386                 StringTokenizer st = new StringTokenizer(extPath, File.pathSeparator);
387                 while (st.hasMoreTokens()) {
388                     String extDir = st.nextToken();
389                     addWorkListItemsForExtDir(workList, extDir);
390                 }
391             }
392     
393             return workList;
394         }

412         private LinkedList<WorkListItem> createFindBugsLibWorkList(String jarFileName) {
413             LinkedList<WorkListItem> workList = new LinkedList<WorkListItem>();
414     
415             boolean found = false;
416     
417             String findbugsHome = FindBugs.getHome();
418             if (findbugsHome != null) {
419                 //
420                 // If the findbugs.home property is set,
421                 // we should be able to find the jar file in
422                 // the lib subdirectory.
423                 //
424                 File base = new File(findbugsHome);
425                 File loc1 = new File(new File(base, "lib"), jarFileName);
426                 File loc2 = new File(base, jarFileName);
427                 File loc = null;
428                 if (loc1.exists()) {
429                     loc = loc1;
430                 } else if (loc2.exists()) {
431                     loc = loc2;
432                 }
433                 if (loc != null) {
434                     found = true;
435                     ICodeBaseLocator codeBaseLocator = classFactory.createFilesystemCodeBaseLocator(loc.getPath());
436                     workList.add(new WorkListItem(codeBaseLocator, false, ICodeBase.Discovered.IN_SYSTEM_CLASSPATH));
437                 }
438             }
439     
440             if (!found) {
441                 if (DEBUG) {
442                     System.out.println("Looking for " + jarFileName + " on classpath...");
443                 }
444                 //
445                 // See if the required jar file is available on the class path.
446                 //
447                 String javaClassPath = SystemProperties.getProperty("java.class.path");
448                 StringTokenizer t = new StringTokenizer(javaClassPath, File.pathSeparator);
449                 while (t.hasMoreTokens()) {
450                     String entry = t.nextToken();
451                     if (DEBUG) {
452                         System.out.print("  Checking " + entry + "...");
453                     }
454     
455                     if (matchesJarFile(entry, jarFileName)) {
456                         found = true;
457                     } else if (matchesJarFile(entry, "findbugs.jar")) {
458                         // See if the searched-for jar file can be found
459                         // alongside findbugs.jar.
460                         File findbugsJar = new File(entry);
461                         File loc = new File(findbugsJar.getParent() + File.separator + jarFileName);
462                         if (DEBUG) {
463                             System.out.print(" [findbugs.jar, checking " + loc.getPath() + "] ");
464                         }
465                         if (loc.exists()) {
466                             entry = loc.getPath();
467                             found = true;
468                         }
469                     }
470     
471                     if (DEBUG) {
472                         System.out.println(found ? "FOUND" : "no");
473                     }
474                     if (found) {
475                         ICodeBaseLocator codeBaseLocator = classFactory.createFilesystemCodeBaseLocator(entry);
476                         workList.add(new WorkListItem(codeBaseLocator, false, ICodeBase.Discovered.IN_SYSTEM_CLASSPATH));
477                         break;
478                     }
479     
480                 }
481             }
482     
483             return workList;
484         }

486         private boolean matchesJarFile(String entry, String jarFileName) {
487             return entry.equals(jarFileName) || entry.endsWith(File.separator + jarFileName) || entry.endsWith("/" + jarFileName);

494          *            the worklist

496          *            a system classpath

506                 if (DEBUG) {

508                 }
509                 addToWorkList(workList, new WorkListItem(classFactory.createFilesystemCodeBaseLocator(entry), false,
510                         ICodeBase.Discovered.IN_SYSTEM_CLASSPATH));

515          * Add worklist items from given extensions directory.
516          *
517          * @param workList
518          *            the worklist
519          * @param extDir
520          *            an extensions directory
521          */
522         private void addWorkListItemsForExtDir(LinkedList<WorkListItem> workList, String extDir) {
523             File dir = new File(extDir);
524             File[] fileList = dir.listFiles(new FileFilter() {
525                 /*
526                  * (non-Javadoc)
527                  *
528                  * @see java.io.FileFilter#accept(java.io.File)
529                  */
530                 @Override
531                 public boolean accept(File pathname) {
532                     String path = pathname.getPath();
533                     boolean isArchive = Archive.isArchiveFileName(path);
534                     return isArchive;
535                 }
536             });
537             if (fileList == null) {
538                 return;
539             }
540     
541             for (File archive : fileList) {
542                 addToWorkList(workList, new WorkListItem(classFactory.createFilesystemCodeBaseLocator(archive.getPath()), false,
543                         ICodeBase.Discovered.IN_SYSTEM_CLASSPATH));
544             }

547         /**
548          * Process classpath worklist items. We will attempt to find all nested
549          * archives and Class-Path entries specified in Jar manifests. This should
550          * give us as good an idea as possible of all of the classes available (and
551          * which are part of the application).
552          *
553          * @param workList
554          *            the worklist to process
555          * @param progress
556          *            IClassPathBuilderProgress callback
557          * @throws InterruptedException
558          * @throws IOException
559          * @throws ResourceNotFoundException
560          */
561         private void processWorkList(IClassPath classPath, LinkedList<WorkListItem> workList, IClassPathBuilderProgress progress)
562                 throws InterruptedException, IOException, ResourceNotFoundException {
563             // Build the classpath, scanning codebases for nested archives
564             // and referenced codebases.
565             while (!workList.isEmpty()) {
566                 WorkListItem item = workList.removeFirst();
567                 if (item.getHowDiscovered() == ICodeBase.Discovered.SPECIFIED) {
568                     progress.startArchive(item.toString());
569                 }
570                 if (DEBUG) {
571                     System.out.println("Working: " + item.getCodeBaseLocator());
572                 }
573     
574                 DiscoveredCodeBase discoveredCodeBase;
575     
576                 // See if we have encountered this codebase before
577                 discoveredCodeBase = discoveredCodeBaseMap.get(item.getCodeBaseLocator().toString());
578                 if (discoveredCodeBase != null) {
579                     // If the codebase is not an app codebase and
580                     // the worklist item says that it is an app codebase,
581                     // change it. Otherwise, we have nothing to do.
582                     if (!discoveredCodeBase.getCodeBase().isApplicationCodeBase() && item.isAppCodeBase()) {
583                         discoveredCodeBase.getCodeBase().setApplicationCodeBase(true);
584                     }
585     
586                     continue;
587                 }
588     
589                 // Detect .java files, which are probably human error
590                 if (item.getCodeBaseLocator() instanceof FilesystemCodeBaseLocator) {
591                     FilesystemCodeBaseLocator l = (FilesystemCodeBaseLocator) item.getCodeBaseLocator();
592                     if (l.getPathName().endsWith(".java")) {
593                         if (DEBUG){
594                             System.err.println("Ignoring .java file \"" + l.getPathName() + "\" specified in classpath or auxclasspath");
595                         }
596                         continue;
597                     }
598                 }
599     
600                 // If we are working on an application codebase,
601                 // then failing to open/scan it is a fatal error.
602                 // We issue warnings about problems with aux codebases,
603                 // but continue anyway.
604     
605                 try {
606                     // Open the codebase and add it to the classpath
607                     discoveredCodeBase = new DiscoveredCodeBase(item.getCodeBaseLocator().openCodeBase());
608                     discoveredCodeBase.getCodeBase().setApplicationCodeBase(item.isAppCodeBase());
609                     discoveredCodeBase.getCodeBase().setHowDiscovered(item.getHowDiscovered());
610     
611                     // Note that this codebase has been visited
612                     discoveredCodeBaseMap.put(item.getCodeBaseLocator().toString(), discoveredCodeBase);
613                     discoveredCodeBaseList.addLast(discoveredCodeBase);
614     
615                     // If it is a scannable codebase, check it for nested archives.
616                     // In addition, if it is an application codebase then
617                     // make a list of application classes.
618                     if (discoveredCodeBase.getCodeBase() instanceof IScannableCodeBase
619                             && ( discoveredCodeBase.codeBase.isApplicationCodeBase()
620                                     || item.getHowDiscovered() == ICodeBase.Discovered.SPECIFIED)
621                             ) {
622                         scanCodebase(classPath, workList, discoveredCodeBase);
623                     }
624     
625                     // Check for a Jar manifest for additional aux classpath
626                     // entries.
627                     scanJarManifestForClassPathEntries(workList, discoveredCodeBase.getCodeBase());
628                 } catch (IOException e) {
629                     if (item.isAppCodeBase() || item.getHowDiscovered() == ICodeBase.Discovered.SPECIFIED) {
630                         if (e instanceof FileNotFoundException) {
631                             errorLogger.logError("File not found: " + item.getCodeBaseLocator());
632                         } else {
633                             errorLogger.logError("Cannot open codebase " + item.getCodeBaseLocator(), e);
634                         }
635                     }
636                 } catch (ResourceNotFoundException e) {
637                     if (item.getHowDiscovered() == ICodeBase.Discovered.SPECIFIED) {
638                         errorLogger.logError("Cannot open codebase " + item.getCodeBaseLocator(), e);
639                     }
640                 }
641     
642                 if (item.getHowDiscovered() == ICodeBase.Discovered.SPECIFIED) {
643                     progress.finishArchive();
644                 }
645             }
646         }

648         /**
649          * Scan given codebase in order to
650          * <ul>
651          * <li>check the codebase for nested archives (adding any found to the
652          * worklist)
653          * <li>build a list of class resources found in the codebase
654          * </ul>
655          *
656          * @param workList
657          *            the worklist
658          * @param discoveredCodeBase
659          *            the codebase to scan
660          * @throws InterruptedException
661          */
662         private void scanCodebase(IClassPath classPath, LinkedList<WorkListItem> workList, DiscoveredCodeBase discoveredCodeBase)
663                 throws InterruptedException {
664             if (DEBUG) {
665                 System.out.println("Scanning " + discoveredCodeBase.getCodeBase().getCodeBaseLocator());
666             }
667     
668             IScannableCodeBase codeBase = (IScannableCodeBase) discoveredCodeBase.getCodeBase();
669     
670             ICodeBaseIterator i = codeBase.iterator();
671             while (i.hasNext()) {
672                 ICodeBaseEntry entry = i.next();
673                 if (VERBOSE) {
674                     System.out.println("Entry: " + entry.getResourceName());
675                 }
676     
677                 if (!NO_PARSE_CLASS_NAMES && codeBase.isApplicationCodeBase()
678                         && DescriptorFactory.isClassResource(entry.getResourceName()) && !(entry instanceof SingleFileCodeBaseEntry)) {
679                     parseClassName(entry);
680                 }
681     
682                 // Note the resource exists in this codebase
683                 discoveredCodeBase.addCodeBaseEntry(entry);
684     
685                 // If resource is a nested archive, add it to the worklist
686                 if (scanNestedArchives && (codeBase.isApplicationCodeBase() || codeBase instanceof DirectoryCodeBase)
687                         && Archive.isLibraryFileName(entry.getResourceName())) {
688                     if (VERBOSE) {
689                         System.out.println("Entry is an library!");
690                     }
691                     ICodeBaseLocator nestedArchiveLocator = classFactory.createNestedArchiveCodeBaseLocator(codeBase,
692                             entry.getResourceName());
693                     addToWorkList(workList,
694                             new WorkListItem(nestedArchiveLocator, codeBase.isApplicationCodeBase(), ICodeBase.Discovered.NESTED));
695                 }
696             }
697         }

699         /**
700          * Attempt to parse data of given resource in order to divine the real name
701          * of the class contained in the resource.
702          *
703          * @param entry
704          *            the resource
705          */
706         private void parseClassName(ICodeBaseEntry entry) {
707             DataInputStream in = null;
708             try {
709                 InputStream resourceIn = entry.openResource();
710                 if (resourceIn == null) {
711                     throw new NullPointerException("Got null resource");
712                 }
713                 in = new DataInputStream(resourceIn);
714                 ClassParserInterface parser = new ClassParser(in, null, entry);
715                 ClassNameAndSuperclassInfo.Builder builder = new ClassNameAndSuperclassInfo.Builder();
716                 parser.parse(builder);
717     
718                 String trueResourceName = builder.build().getClassDescriptor().toResourceName();
719                 if (!trueResourceName.equals(entry.getResourceName())) {
720                     entry.overrideResourceName(trueResourceName);
721                 }
722             } catch (IOException e) {
723                 errorLogger.logError("Invalid class resource " + entry.getResourceName() + " in " + entry, e);
724             } catch (InvalidClassFileFormatException e) {
725                 errorLogger.logError("Invalid class resource " + entry.getResourceName() + " in " + entry, e);
726             } finally {
727                 IO.close(in);
728             }
729         }

731         /**
732          * Check a codebase for a Jar manifest to examine for Class-Path entries.
733          *
734          * @param workList
735          *            the worklist
736          * @param codeBase
737          *            the codebase for examine for a Jar manifest
738          * @throws IOException
739          */
740         private void scanJarManifestForClassPathEntries(LinkedList<WorkListItem> workList, ICodeBase codeBase) throws IOException {
741             // See if this codebase has a jar manifest
742             ICodeBaseEntry manifestEntry = codeBase.lookupResource("META-INF/MANIFEST.MF");
743             if (manifestEntry == null) {
744                 // Do nothing - no Jar manifest found
745                 return;
746             }
747     
748             // Try to read the manifest
749             InputStream in = null;
750             try {
751                 in = manifestEntry.openResource();
752                 Manifest manifest = new Manifest(in);
753     
754                 Attributes mainAttrs = manifest.getMainAttributes();
755                 String classPath = mainAttrs.getValue("Class-Path");
756                 if (classPath != null) {
757                     String[] pathList = classPath.split("\\s+");
758     
759                     for (String path : pathList) {
760                         // Create a codebase locator for the classpath entry
761                         // relative to the codebase in which we discovered the Jar
762                         // manifest
763                         ICodeBaseLocator relativeCodeBaseLocator = codeBase.getCodeBaseLocator().createRelativeCodeBaseLocator(path);
764     
765                         // Codebases found in Class-Path entries are always
766                         // added to the aux classpath, not the application.
767                         addToWorkList(workList, new WorkListItem(relativeCodeBaseLocator, false, ICodeBase.Discovered.IN_JAR_MANIFEST));
768                     }
769                 }
770             } finally {
771                 if (in != null) {
772                     IO.close(in);
773                 }
774             }
775         }

777         /**
778          * Add a worklist item to the worklist. This method maintains the invariant
779          * that all of the worklist items representing application codebases appear
780          * <em>before</em> all of the worklist items representing auxiliary
781          * codebases.
782          *
783          * @param projectWorkList
784          *            the worklist
785          * @param itemToAdd
786          *            the worklist item to add
787          */
788         private void addToWorkList(LinkedList<WorkListItem> workList, WorkListItem itemToAdd) {
789             if (DEBUG) {
790                 new RuntimeException("Adding work list item " + itemToAdd).printStackTrace(System.out);
791             }
792             if (!itemToAdd.isAppCodeBase()) {
793                 // Auxiliary codebases are always added at the end
794                 workList.addLast(itemToAdd);
795                 return;
796             }
797     
798             // Adding an application codebase: position a ListIterator
799             // just before first auxiliary codebase (or at the end of the list
800             // if there are no auxiliary codebases)
801             ListIterator<WorkListItem> i = workList.listIterator();
802             while (i.hasNext()) {
803                 WorkListItem listItem = i.next();
804                 if (!listItem.isAppCodeBase()) {
805                     i.previous();
806                     break;
807                 }
808             }
809     
810             // Add the codebase to the worklist
811             i.add(itemToAdd);
812         }
