46     /**
47      * Factory for producing ForwardTypeQualifierDataflow objects for various kinds
48      * of type qualifiers.
49      *
50      * @author David Hovemeyer
51      */
52     public class ForwardTypeQualifierDataflowFactory extends
53     TypeQualifierDataflowFactory<ForwardTypeQualifierDataflowAnalysis, ForwardTypeQualifierDataflow> {
54     
55         /**
56          * Constructor.
57          *
58          * @param methodDescriptor
59          *            MethodDescriptor of method being analyzed
60          */
61         public ForwardTypeQualifierDataflowFactory(MethodDescriptor methodDescriptor) {
62             super(methodDescriptor);
63         }
64     
65     
66         @Override
67         protected ForwardTypeQualifierDataflow getDataflow(DepthFirstSearch dfs, XMethod xmethod, CFG cfg,
68                 ValueNumberDataflow vnaDataflow, ConstantPoolGen cpg, IAnalysisCache analysisCache,
69                 MethodDescriptor methodDescriptor, TypeQualifierValue<?> typeQualifierValue) throws DataflowAnalysisException {
70             ForwardTypeQualifierDataflowAnalysis analysis = new ForwardTypeQualifierDataflowAnalysis(dfs, xmethod, cfg, vnaDataflow,
71                     cpg, typeQualifierValue);
72             analysis.registerSourceSinkLocations();
73     
74             ForwardTypeQualifierDataflow dataflow = new ForwardTypeQualifierDataflow(cfg, analysis);
75             dataflow.execute();
76             if (ClassContext.DUMP_DATAFLOW_ANALYSIS) {
77                 dataflow.dumpDataflow(analysis);
78             }
79             return dataflow;
80         }
81     
82         @Override
83         protected void populateDatabase(ForwardTypeQualifierDataflow dataflow, ValueNumberDataflow vnaDataflow, XMethod xmethod,
84                 TypeQualifierValue<?> tqv) throws CheckedAnalysisException {
85             assert TypeQualifierDatabase.USE_DATABASE;
86     
87             if (xmethod.getSignature().endsWith(")V")) {
88                 return;
89             }
90     
91             // If there is no effective type qualifier annotation
92             // on the method's return value, and we computed an
93             // "interesting" (ALWAYS or NEVER) value for the return
94             // value, add it to the database.
95     
96             TypeQualifierAnnotation tqa = TypeQualifierApplications.getEffectiveTypeQualifierAnnotation(xmethod, tqv);
97             if (tqa == null) {
98                 // Find the meet of the flow values at all instructions
99                 // which return a value.
100                 FlowValue effectiveFlowValue = null;
101     
102                 CFG cfg = dataflow.getCFG();
103                 Iterator<Location> i = cfg.locationIterator();
104                 while (i.hasNext()) {
105                     Location loc = i.next();
106                     InstructionHandle handle = loc.getHandle();
107                     Instruction ins = handle.getInstruction();
108                     if (ins instanceof ReturnInstruction && !(ins instanceof RETURN)) {
109                         ValueNumberFrame vnaFrame = vnaDataflow.getFactAtLocation(loc);
110                         ValueNumber topVN = vnaFrame.getTopValue();
111     
112                         TypeQualifierValueSet flowSet = dataflow.getFactAtLocation(loc);
113                         FlowValue topFlowValue = flowSet.getValue(topVN);
114     
115                         if (topFlowValue != null) {
116                             if (TypeQualifierDatabase.DEBUG) {
117                                 System.out.println("at pc " + handle.getPosition() + " of " + xmethod + ", return value for " + tqv
118                                         + " is " + topFlowValue);
119                             }
120                             if (effectiveFlowValue == null) {
121                                 effectiveFlowValue = topFlowValue;
122                             } else {
123                                 effectiveFlowValue = FlowValue.meet(effectiveFlowValue, topFlowValue);
124                             }
125                         }
126                     }
127                 }
128     
129                 if (effectiveFlowValue == FlowValue.ALWAYS || effectiveFlowValue == FlowValue.NEVER) {
130                     TypeQualifierDatabase tqdb = Global.getAnalysisCache().getDatabase(TypeQualifierDatabase.class);
131                     if (TypeQualifierDatabase.DEBUG) {
132                         System.out.println("inferring return value for " + xmethod + " of" + tqv + " : " + effectiveFlowValue);
133                     }
134                     tqa = TypeQualifierAnnotation.getValue(tqv, effectiveFlowValue == FlowValue.ALWAYS ? When.ALWAYS : When.NEVER);
135                     tqdb.setReturnValue(xmethod.getMethodDescriptor(), tqv, tqa);
136                 }
137             }
138         }
139     }
