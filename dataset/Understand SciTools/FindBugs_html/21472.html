104         public JavaSourceDocument(String title, Reader in, SourceFile theSource) throws IOException {
105             doc = new DefaultStyledDocument();
106             this.title = title;
107             this.sourceFile = theSource;
108             Debug.println("Created JavaSourceDocument for " + title);
109             try {
110                 dek.read(in, doc, 0);
111             } catch (BadLocationException e) {
112                 throw new RuntimeException(e);
113             }
114             in.close();
115             doc.putProperty(Document.TitleProperty, title);
116             //        root = doc.getDefaultRootElement();
117             Toolkit toolkit = Toolkit.getDefaultToolkit();
118             FontMetrics fontMetrics = toolkit.getFontMetrics(sourceFont);
119             TabStop[] tabs = new TabStop[50];
120             float width = fontMetrics.stringWidth(" ");
121     
122             int tabSize = GUISaveState.getInstance().getTabSize();
123             for (int i = 0; i < tabs.length; i++) {
124                 tabs[i] = new TabStop(width * (tabSize + tabSize * i));
125             }
126             TAB_SET = new TabSet(tabs);
127             StyleConstants.setTabSet(commentAttributes, TAB_SET);
128             StyleConstants.setTabSet(javadocAttributes, TAB_SET);
129     
130             StyleConstants.setTabSet(quotesAttributes, TAB_SET);
131     
132             StyleConstants.setTabSet(keywordsAttributes, TAB_SET);
133     
134             StyleConstants.setTabSet(commentAttributes, TAB_SET);
135     
136             StyleConstants.setTabSet(whiteAttributes, TAB_SET);
137             StyleConstants.setFontFamily(whiteAttributes, sourceFont.getFamily());
138             StyleConstants.setFontSize(whiteAttributes, sourceFont.getSize());
139             StyleConstants.setLeftIndent(whiteAttributes, NumberedParagraphView.NUMBERS_WIDTH);
140     
141             doc.setParagraphAttributes(0, doc.getLength(), whiteAttributes, true);
142             JavaScanner parser = new JavaScanner(new DocumentCharacterIterator(doc));
143             while (parser.next() != JavaScanner.EOF) {
144                 int kind = parser.getKind();
145                 switch (kind) {
146                 case JavaScanner.COMMENT:
147                     doc.setCharacterAttributes(parser.getStartPosition(), parser.getLength(), commentAttributes, true);
148                     break;
149     
150                 case JavaScanner.KEYWORD:
151                     doc.setCharacterAttributes(parser.getStartPosition(), parser.getLength(), keywordsAttributes, true);
152                     break;
153     
154                 case JavaScanner.JAVADOC:
155                     doc.setCharacterAttributes(parser.getStartPosition(), parser.getLength(), javadocAttributes, true);
156                     break;
157     
158                 case JavaScanner.QUOTE:
159                     doc.setCharacterAttributes(parser.getStartPosition(), parser.getLength(), quotesAttributes, true);
160                     break;
161     
162                 default:
163                     break;
164                 }
165     
166             }
167     
168         }
