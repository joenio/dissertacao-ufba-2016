20     package edu.umd.cs.findbugs;
21     
22     import java.util.HashSet;
23     import java.util.Iterator;
24     import java.util.Set;
25     
26     /**
27      * Search for bug instances whose text annotations contain one of a set of
28      * keywords.
29      */
30     public abstract class QueryBugAnnotations {
31         // Bug's text annotation must contain one of the key
32         // words in order to match
33         private final HashSet<String> keywordSet = new HashSet<String>();
34     
35         /**
36          * Add a keyword to the query. A BugInstance's text annotation must contain
37          * at least one keyword in order to match the query.
38          *
39          * @param keyword
40          *            the keyword
41          */
42         public void addKeyword(String keyword) {
43             keywordSet.add(keyword);
44         }
45     
46         /**
47          * Scan bug instances contained in given file, reporting those whose text
48          * annotations contain at least one of the keywords in the query.
49          *
50          * @param filename
51          *            an XML file containing bug instances
52          */
53         public void scan(String filename) throws Exception {
54             BugCollection bugCollection = new SortedBugCollection();
55             bugCollection.readXML(filename);
56             scan(bugCollection, filename);
57         }
58     
59         /**
60          * Scan bug instances contained in given bug collection, reporting those
61          * whose text annotations contain at least one of the keywords in the query.
62          *
63          * @param bugCollection
64          *            the bug collection
65          * @param filename
66          *            the XML file from which the bug collection was read
67          */
68         public void scan(BugCollection bugCollection, String filename) throws Exception {
69             Iterator<BugInstance> i = bugCollection.iterator();
70             while (i.hasNext()) {
71                 BugInstance bugInstance = i.next();
72     
73                 Set<String> contents = bugInstance.getTextAnnotationWords();
74                 for (String aKeywordSet : keywordSet) {
75                     if (contents.contains(aKeywordSet)) {
76                         match(bugInstance, filename);
77                         break;
78                     }
79                 }
80             }
81         }
82     
83         /**
84          * Called when a bug instance contains a query keyword.
85          *
86          * @param bugInstance
87          *            the bug instance containing the keyword
88          * @param filename
89          *            name of the file containing the bug instance
90          */
91         protected abstract void match(BugInstance bugInstance, String filename) throws Exception;
92     }
