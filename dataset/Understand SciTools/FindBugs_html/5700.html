20     package edu.umd.cs.findbugs.ba;
21     
22     import javax.annotation.CheckForNull;
23     
24     import edu.umd.cs.findbugs.internalAnnotations.DottedClassName;
25     
26     /**
27      * Interface for querying nullness annotations on methods, fields, and
28      * parameters.
29      *
30      * @author David Hovemeyer
31      */
32     public interface INullnessAnnotationDatabase {
33     
34         /**
35          * Determine whether given parameter must be non-null.
36          *
37          * @param m
38          *            a method
39          * @param param
40          *            parameter (0 == first parameter)
41          * @return true if the parameter must be non-null, false otherwise
42          */
43         public abstract boolean parameterMustBeNonNull(XMethod m, int param);
44     
45         /**
46          * Get a resolved NullnessAnnotation on given XMethod, XField, or
47          * XMethodParameter.
48          *
49          * @param o
50          *            an XMethod, XField, or XMethodParameter
51          * @param getMinimal
52          *            TODO: what does this mean?
53          * @return resolved NullnessAnnotation
54          */
55         @CheckForNull
56         public abstract NullnessAnnotation getResolvedAnnotation(final Object o, boolean getMinimal);
57     
58         /**
59          * Load "built-in" annotations that might not be evident from the
60          * analyzed/referenced code.
61          */
62         public void loadAuxiliaryAnnotations();
63     
64         // /**
65         // * Add a default nullness annotation to the database.
66         // *
67         // * @param cName class name (FIXME: is this right?)
68         // * @param annotation the default NullnessAnnotation
69         // */
70         // public void addDefaultMethodAnnotation(String cName, NullnessAnnotation
71         // annotation);
72     
73         /**
74          * Add a field annotation to the database.
75          *
76          * @param cName
77          *            dotted class name
78          * @param mName
79          *            field name
80          * @param mSig
81          *            field signature
82          * @param isStatic
83          *            true if field is static, false otherwise
84          * @param annotation
85          *            NullnessAnnotation to add
86          */
87         public void addFieldAnnotation(@DottedClassName String cName, String mName, String mSig, boolean isStatic,
88                 NullnessAnnotation annotation);
89     
90         /**
91          * Add a method annotation to the database.
92          *
93          * @param cName
94          *            dotted class name
95          * @param mName
96          *            method name
97          * @param mSig
98          *            method signature
99          * @param isStatic
100          *            true if method is static, false otherwise
101          * @param annotation
102          *            NullnessAnnotation to add
103          */
104         public void addMethodAnnotation(@DottedClassName String cName, String mName, String mSig, boolean isStatic,
105                 NullnessAnnotation annotation);
106     
107         /**
108          * Add a method parameter annotation to the database.
109          *
110          * @param cName
111          *            dotted class name
112          * @param mName
113          *            method name
114          * @param mSig
115          *            method signature
116          * @param isStatic
117          *            true if method is static, false otherwise
118          * @param param
119          *            parameter (0 == first parameter)
120          * @param annotation
121          *            the NullnessAnnotation to add
122          */
123         public void addMethodParameterAnnotation(@DottedClassName String cName, String mName, String mSig, boolean isStatic,
124                 int param, NullnessAnnotation annotation);
125     
126         /**
127          * Add a default annotation to the database.
128          *
129          * @param target
130          *            one of AnnotationDatabase.METHOD, AnnotationDatabase.FIELD,
131          *            AnnotationDatabase.PARAMETER, or AnnotationDatabase.ANY
132          * @param c
133          *            dotted class name of class default annotation pertains to
134          * @param n
135          *            the default NullnessAnnotation
136          */
137         public void addDefaultAnnotation(AnnotationDatabase.Target target, @DottedClassName String c, NullnessAnnotation n);
138     }
