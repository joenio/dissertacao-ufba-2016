40     /**
41      * Class to handle Strings annotated with embedded mnemonics
42      *
43      * Note: Since the human interface guidelines for Mac OS X say never to use
44      * mnemonics, this class behaves as if no mnemonics are set when run on Mac OS
45      * X.
46      */
47     public class AnnotatedString {
48     
49         private static final boolean MAC_OS_X = SystemProperties.getProperty("os.name").toLowerCase().startsWith("mac os x");
50     
51         private final String myAnnotatedString;
52     
53         public AnnotatedString(String s) {
54             myAnnotatedString = s;
55         }
56     
57         /*
58          * Return the string minus any annotation
59          */
60         @Override
61         public String toString() {
62             if (MAC_OS_X) {
63                 // Support annotations like "File(&F)"
64                 if (myAnnotatedString.matches("[^&]+\\(&\\p{Alnum}\\)")) {
65                     int endIndex = myAnnotatedString.length() - "(&X)".length();
66     
67                     return myAnnotatedString.substring(0, endIndex);
68                 }
69     
70                 // Support annotations like "File(&F)..."
71                 if (myAnnotatedString.matches("[^&]+\\(&\\p{Alnum}\\)\\.\\.\\.")) {
72                     int startIndex = myAnnotatedString.length() - "(&X)...".length();
73                     int endIndex = startIndex + "(&X)".length();
74     
75                     return new StringBuilder(myAnnotatedString).delete(startIndex, endIndex).toString();
76                 }
77             }
78             return myAnnotatedString.replaceFirst("&", "");
79         }
80     
81         /**
82          * Return the appropriate mnemonic character for this string. If no mnemonic
83          * should be displayed, KeyEvent.VK_UNDEFINED is returned.
84          *
85          * @return the Mnemonic character, or VK_UNDEFINED if no mnemonic should be
86          *         set
87          */
88         public int getMnemonic() {
89             int mnemonic = KeyEvent.VK_UNDEFINED;
90             if (!MAC_OS_X) {
91                 int index = getMnemonicIndex();
92                 if ((index >= 0) && ((index + 1) < myAnnotatedString.length())) {
93                     mnemonic = Character.toUpperCase(myAnnotatedString.charAt(index + 1));
94                 }
95             }
96             return mnemonic;
97         }
98     
99         /**
100          * @return the index in the plain string at which the mnemonic should be
101          *         displayed, or -1 if no mnemonic should be set
102          */
103         public int getMnemonicIndex() {
104             int index = -1;
105             if (!MAC_OS_X) {
106                 index = myAnnotatedString.indexOf('&');
107                 if (index + 1 >= myAnnotatedString.length()) {
108                     index = -1;
109                 }
110             }
111             return index;
112         }

114         /*
115          * Some test code
116          */
117         private static void addButton(JFrame frame, String s) {
118             AnnotatedString as = new AnnotatedString(s);
119             JButton button = new JButton(as.toString());
120             button.setMnemonic(as.getMnemonic());
121             button.setDisplayedMnemonicIndex(as.getMnemonicIndex());
122             frame.getContentPane().add(button);
123     
124             System.out.println("\"" + s + "\" \"" + as + "\" '" + as.getMnemonic() + "' " + as.getMnemonicIndex());
125         }

127         public static void main(String[] args) {
128             // Some basic tests
129     
130             final JFrame frame = new JFrame();
131             frame.getContentPane().setLayout(new FlowLayout());
132     
133             addButton(frame, "&File");
134             addButton(frame, "S&ave As...");
135             addButton(frame, "Save &As...");
136             addButton(frame, "Fo&o");
137             addButton(frame, "Foo");
138     
139             // Error cases - make sure we handle them sensibly
140             addButton(frame, "");
141             addButton(frame, "&");
142             addButton(frame, "Foo&");
143             addButton(frame, "Cat & Dog");
144             addButton(frame, "Cat && Dog");
145     
146             SwingUtilities.invokeLater(new Runnable() {
147                 @Override
148                 public void run() {
149                     frame.pack();
150                     frame.setVisible(true);
151                 }
152             });
153     
154         }
155     
156         /**
157          * Localise the given AbstractButton, setting the text and optionally
158          * mnemonic Note that AbstractButton includes menus and menu items.
159          *
160          * @param button
161          *            The button to localise
162          * @param key
163          *            The key to look up in resource bundle
164          * @param defaultString
165          *            default String to use if key not found
166          * @param setMnemonic
167          *            whether or not to set the mnemonic. According to Sun's
168          *            guidelines, default/cancel buttons should not have mnemonics
169          *            but instead should use Return/Escape
170          */
171         public static void localiseButton(AbstractButton button, String key, String defaultString, boolean setMnemonic) {
172             AnnotatedString as = new AnnotatedString(L10N.getLocalString(key, defaultString));
173             button.setText(as.toString());
174             int mnemonic;
175             if (setMnemonic && (mnemonic = as.getMnemonic()) != KeyEvent.VK_UNDEFINED) {
176                 button.setMnemonic(mnemonic);
177                 button.setDisplayedMnemonicIndex(as.getMnemonicIndex());
178             }
179         }
