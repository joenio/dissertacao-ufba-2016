22     import java.util.Iterator;

24     import javax.annotation.CheckForNull;

26     /**
27      * A convenient base class for dataflow analysis classes which use Frames as
28      * values.
29      *
30      * @author David Hovemeyer
31      * @see Frame
32      * @see DataflowAnalysis
33      */
34     public abstract class FrameDataflowAnalysis<ValueType, FrameType extends Frame<ValueType>> extends
35     ForwardDataflowAnalysis<FrameType> {
36         public FrameDataflowAnalysis(DepthFirstSearch dfs) {
37             super(dfs);
38         }
39     
40         @Override
41         public void copy(FrameType source, FrameType dest) {
42             dest.copyFrom(source);
43         }
44     
45         @Override
46         public void makeFactTop(FrameType fact) {
47             fact.setTop();
48         }
49     
50         @Override
51         public boolean isTop(FrameType fact) {
52             return fact.isTop();
53         }
54     
55         @Override
56         public boolean same(FrameType fact1, FrameType fact2) {
57             return fact1.sameAs(fact2);
58         }
59     
60         /**
61          * Get the dataflow fact representing the point just before given Location.
62          * Note "before" is meant in the logical sense, so for backward analyses,
63          * before means after the location in the control flow sense.
64          *
65          * @return the fact at the point just before the location
66          */
67     
68         public FrameType getFactAtPC(CFG cfg, int pc) throws DataflowAnalysisException {
69             FrameType result = createFact();
70             makeFactTop(result);
71     
72             for (Location l : cfg.locations()) {
73                 if (l.getHandle().getPosition() == pc) {
74                     FrameType fact = getFactAtLocation(l);
75                     if (isFactValid(fact)) {
76                         mergeInto(fact, result);
77                     }
78                 }
79             }
80             return result;
81         }
82     
83         /**
84          * Get the dataflow fact representing the point just before given Location.
85          * Note "before" is meant in the logical sense, so for backward analyses,
86          * before means after the location in the control flow sense.
87          *
88          * @return the fact at the point just before the location
89          */
90     
91         public FrameType getFactBeforeExceptionCheck(CFG cfg, int pc) throws DataflowAnalysisException {
92             FrameType result = createFact();
93             makeFactTop(result);
94     
95             for (Location loc : cfg.getLocationsContainingInstructionWithOffset(pc)) {
96                 BasicBlock b = loc.getBasicBlock();
97                 BasicBlock b2 = null;
98                 if (b.getFirstInstruction() != null && b.getFirstInstruction().getPosition() == pc) {
99                     b2 = cfg.getPredecessorWithEdgeType(b, EdgeTypes.FALL_THROUGH_EDGE);
100                 }
101                 if(b2 != null && b2.isExceptionThrower()) {
102                     for (Iterator<Edge> i = cfg.incomingEdgeIterator(b2); i.hasNext();) {
103                         Edge e = i.next();
104                         FrameType fact = getFactOnEdge(e);
105                         if (isFactValid(fact)) {
106                             mergeInto(fact, result);
107                         }
108                     }
109                 } else {
110                     FrameType fact = getFactAtLocation(loc);
111                     if (isFactValid(fact)) {
112                         mergeInto(fact, result);
113                     }
114                 }
115             }
116             return result;
117         }

121             return fact.isValid();
122         }

134         /**
135          * Create a modifiable copy of a frame. This is useful for meetInto(), if
136          * the frame needs to be modified in a path-sensitive fashion. A typical
137          * usage pattern is:
138          * <p/>
139          *
140          * <pre>
141          * FrameType copy = null;
142          * if (someCondition()) {
143          *     copy = modifyFrame(fact, copy);
144          *     // modify copy
145          * }
146          * if (someOtherCondition()) {
147          *     copy = modifyFrame(fact, copy);
148          *     // modify copy
149          * }
150          * if (copy != null)
151          *     fact = copy;
152          * <p/>
153          * mergeInto(fact, result);
154          * </pre>
155          * <p/>
156          * The advantage of using modifyFrame() is that new code can be added before
157          * or after other places where the frame is modified, and the code will
158          * remain correct.
159          *
160          * @param orig
161          *            the original frame
162          * @param copy
163          *            the modifiable copy (returned by a previous call to
164          *            modifyFrame()), or null if this is the first time
165          *            modifyFrame() is being called
166          * @return a modifiable copy of fact
167          */
168         final protected FrameType modifyFrame(FrameType orig, @CheckForNull FrameType copy) {
169             if (copy == null) {
170                 copy = createFact();
171                 copy.copyFrom(orig);
172             }
173             return copy;
174         }
175     
176         /**
177          * Merge one frame into another.
178          *
179          * @param other
180          *            the frame to merge with the result
181          * @param result
182          *            the result frame, which is modified to be the merge of the two
183          *            frames
184          */
185         protected void mergeInto(FrameType other, FrameType result) throws DataflowAnalysisException {
186             // Handle if result Frame or the other Frame is the special "TOP" value.
187             if (result.isTop()) {
188                 // Result is the identity element, so copy the other Frame
189                 result.copyFrom(other);
190                 return;
191             } else if (other.isTop()) {
192                 // Other Frame is the identity element, so result stays the same
193                 return;
194             }
195     
196             // Handle if result Frame or the other Frame is the special "BOTTOM"
197             // value.
198             if (result.isBottom()) {
199                 // Result is the bottom element, so it stays that way
200                 return;
201             } else if (other.isBottom()) {
202                 // Other Frame is the bottom element, so result becomes the bottom
203                 // element too
204                 result.setBottom();
205                 return;
206             }
207     
208             // If the number of slots in the Frames differs,
209             // then the result is the special "BOTTOM" value.
210             if (result.getNumSlots() != other.getNumSlots()) {
211                 result.setBottom();
212                 return;
213             }
214     
215             // Usual case: ordinary Frames consisting of the same number of values.
216             // Merge each value in the two slot lists element-wise.
217             for (int i = 0; i < result.getNumSlots(); ++i) {
218                 mergeValues(other, result, i);
219             }
220         }
221     
222         /**
223          * Merge the values contained in a given slot of two Frames.
224          *
225          * @param otherFrame
226          *            a Frame
227          * @param resultFrame
228          *            a Frame which will contain the resulting merged value
229          * @param slot
230          *            a slot in both frames
231          * @throws DataflowAnalysisException
232          */
233         protected abstract void mergeValues(FrameType otherFrame, FrameType resultFrame, int slot) throws DataflowAnalysisException;
234     }
