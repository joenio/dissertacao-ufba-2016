35     /**
36      * Analysis to determine where particular values are locked in a method. The
37      * dataflow values are maps of value numbers to the number of times those values
38      * are locked.
39      *
40      * @author David Hovemeyer
41      * @see ValueNumberAnalysis
42      */
43     public class LockAnalysis extends ForwardDataflowAnalysis<LockSet> {
44         private static final boolean DEBUG = SystemProperties.getBoolean("la.debug");
45     
46         private final MethodGen methodGen;
47     
48         private final ValueNumberDataflow vnaDataflow;
49     
50         private final ValueNumberAnalysis vna;
51     
52         private final boolean isSynchronized;
53     
54         private final boolean isStatic;
55     
56         public LockAnalysis(MethodGen methodGen, ValueNumberDataflow vnaDataflow, DepthFirstSearch dfs) {
57             super(dfs);
58             this.methodGen = methodGen;
59             this.vnaDataflow = vnaDataflow;
60             this.vna = vnaDataflow.getAnalysis();
61             this.isSynchronized = methodGen.isSynchronized();
62             this.isStatic = methodGen.isStatic();
63             if (DEBUG) {
64                 System.out.println("Analyzing Locks in " + methodGen.getClassName() + "." + methodGen.getName());
65             }
66         }
67     
68         @Override
69         public LockSet createFact() {
70             return new LockSet();
71         }
72     
73         @Override
74         public void copy(LockSet source, LockSet dest) {
75             dest.copyFrom(source);
76         }
77     
78         @Override
79         public void initEntryFact(LockSet result) {
80             result.clear();
81             result.setDefaultLockCount(0);
82     
83             if (isSynchronized && !isStatic) {
84                 ValueNumber thisValue = vna.getThisValue();
85                 result.setLockCount(thisValue.getNumber(), 1);
86             } else if (isSynchronized && isStatic) {
87                 ValueNumber thisValue = vna.getClassObjectValue(methodGen.getClassName());
88                 result.setLockCount(thisValue.getNumber(), 1);
89             }
90         }
91     
92         @Override
93         public void makeFactTop(LockSet fact) {
94             fact.clear();
95             fact.setDefaultLockCount(LockSet.TOP);
96         }
97     
98         @Override
99         public boolean isTop(LockSet fact) {
100             return fact.isTop();
101         }
102     
103         @Override
104         public boolean same(LockSet fact1, LockSet fact2) {
105             return fact1.sameAs(fact2);
106         }
107     
108         @Override
109         public void meetInto(LockSet fact, Edge edge, LockSet result) throws DataflowAnalysisException {
110             result.meetWith(fact);
111         }
112     
113         @Override
114         public void transferInstruction(InstructionHandle handle, BasicBlock basicBlock, LockSet fact)
115                 throws DataflowAnalysisException {
116     
117             Instruction ins = handle.getInstruction();
118             short opcode = ins.getOpcode();
119             if (opcode == Constants.MONITORENTER || opcode == Constants.MONITOREXIT) {
120                 ValueNumberFrame frame = vnaDataflow.getFactAtLocation(new Location(handle, basicBlock));
121     
122                 modifyLock(frame, fact, opcode == Constants.MONITORENTER ? 1 : -1);
123     
124             } else if (opcode == Constants.INVOKEVIRTUAL || opcode == Constants.INVOKEINTERFACE) {
125     
126                 InvokeInstruction inv = (InvokeInstruction) ins;
127                 String name = inv.getMethodName(methodGen.getConstantPool());
128                 String sig = inv.getSignature(methodGen.getConstantPool());
129                 ValueNumberFrame frame = vnaDataflow.getFactAtLocation(new Location(handle, basicBlock));
130     
131                 if ("()V".equals(sig) && ("lock".equals(name) || "lockInterruptibly".equals(name))) {
132                     modifyLock(frame, fact, 1);
133                 } else if ("()V".equals(sig) && ("unlock".equals(name))) {
134                     modifyLock(frame, fact, -1);
135                 }
136     
137             } else if ((ins instanceof ReturnInstruction) && isSynchronized && !isStatic) {
138     
139                 lockOp(fact, vna.getThisValue().getNumber(), -1);
140             }
141         }
142     
143         private void modifyLock(ValueNumberFrame frame, LockSet fact, int delta) throws DataflowAnalysisException {
144             if (frame.isValid()) {
145                 int lockNumber = frame.getTopValue().getNumber();
146                 lockOp(fact, lockNumber, delta);
147             }
148         }
149     
150         private void lockOp(LockSet fact, int lockNumber, int delta) {
151             int value = fact.getLockCount(lockNumber);
152             if (value < 0) {
153                 return;
154             }
155             value += delta;
156             if (value < 0) {
157                 value = LockSet.BOTTOM;
158             }
159             if (DEBUG) {
160                 System.out.println("Setting " + lockNumber + " to " + value + " in " + methodGen.getClassName() + "."
161                         + methodGen.getName());
162             }
163             fact.setLockCount(lockNumber, value);
164         }
165     
166         @Override
167         public boolean isFactValid(LockSet fact) {
168             return true;
169         }
