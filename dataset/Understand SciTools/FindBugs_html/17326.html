83         enum StackFrameType {
84             SAME_FRAME, SAME_LOCALS_1_STACK_ITEM_FRAME, CHOP_FRAME, APPEND_FRAME, FULL_FRAME;
85     
86             static StackFrameType get(int frame_type) {
87                 if (frame_type >= Constants.SAME_FRAME && frame_type <= Constants.SAME_FRAME_MAX) {
88                     return SAME_FRAME;
89                 } else if (frame_type >= Constants.SAME_LOCALS_1_STACK_ITEM_FRAME
90                         && frame_type <= Constants.SAME_LOCALS_1_STACK_ITEM_FRAME_MAX) {
91                     return SAME_LOCALS_1_STACK_ITEM_FRAME;
92                 } else if (frame_type == Constants.SAME_LOCALS_1_STACK_ITEM_FRAME_EXTENDED) {
93                     return SAME_LOCALS_1_STACK_ITEM_FRAME;
94                 } else if (frame_type >= Constants.CHOP_FRAME && frame_type <= Constants.CHOP_FRAME_MAX) {
95                     return CHOP_FRAME;
96                 } else if (frame_type == Constants.SAME_FRAME_EXTENDED) {
97                     return SAME_FRAME;
98                 } else if (frame_type >= Constants.APPEND_FRAME && frame_type <= Constants.APPEND_FRAME_MAX) {
99                     return APPEND_FRAME;
100                 } else if (frame_type == Constants.FULL_FRAME) {
101                     return FULL_FRAME;
102                 } else {
103                     /* Can't happen */
104                     throw new ClassFormatException("Invalid frame type : " + frame_type);
105                 }
106             }
107         }

109         static @CheckForNull StackMapTable getStackMapTable(Code code) {
110             for(Attribute a : code.getAttributes()) {
111                 if (a instanceof StackMapTable) {
112                     return (StackMapTable) a;
113                 }
114             }
115             return null;
116         }

118         static  List<Item>  getInitialLocals(MethodDescriptor descriptor) {
119             List<Item> locals = new ArrayList<Item>();
120             Type[] argTypes = Type.getArgumentTypes(descriptor.getSignature());
121             int reg = 0;
122             if (!descriptor.isStatic()) {
123                 Item it = Item.typeOnly("L" + descriptor.getSlashedClassName() + ";");
124                 locals.add(it);
125                 reg += it.getSize();
126             }
127             for (Type argType : argTypes) {
128                 Item it = Item.typeOnly(argType.getSignature());
129                 locals.add(it);
130                 reg += it.getSize();
131                 if (it.usesTwoSlots()) {
132                     locals.add(null);
133                 }
134             }
135             return locals;
136         }

138         static final @CheckForNull Field frame_type_field;
139         static {
140             Field f;
141             try {
142                 f = AccessController.doPrivileged(new PrivilegedAction<Field>() {
143     
144                     @Override
145                     public Field run() {
146                         Class<StackMapTableEntry> c = StackMapTableEntry.class;
147                         Field result;
148                         try {
149                             result = c.getDeclaredField("frame_type");
150                             result.setAccessible(true);
151                             return result;
152                         } catch (NoSuchFieldException e) {
153                             throw new AssertionError("frame_type field doesn't exist");
154                         } catch (SecurityException e) {
155                             return null;
156                         }
157     
158                     }
159     
160                 });
161             } catch (Exception e) {
162                 AnalysisContext.logError("Unable to create frame_type accessor",e );
163                 f = null;
164             }
165             if (DEBUG) {
166                 System.out.println("Frame type field is null:" + (f == null));
167             }
168             frame_type_field = f;
169         }

171         static int getFrameType(StackMapTableEntry e) {
172             if (frame_type_field == null) {
173                 return -1;
174             }
175             try {
176                 return (Integer) frame_type_field.get(e);
177             } catch (IllegalArgumentException e1) {
178                 return -1;
179             } catch (IllegalAccessException e1) {
180                 return -1;
181             }
182         }
183         static  private  @CheckForNull JumpInfoFromStackMap getFromStackMap(IAnalysisCache analysisCache, MethodDescriptor descriptor)  {
184             if (frame_type_field == null) {
185                 return null;
186             }
187     
188             Method method;
189             try {
190                 method = analysisCache.getMethodAnalysis(Method.class, descriptor);
191             } catch (CheckedAnalysisException e1) {
192                 analysisCache.getErrorLogger().logError("Unable to get method for " + descriptor, e1);
193                 return null;
194             }
195     
196             Code code = method.getCode();
197             if (code == null) {
198                 return null;
199             }
200             StackMapTable stackMapTable = getStackMapTable(code);
201             if (stackMapTable == null) {
202                 return null;
203             }
204             Map<Integer, List<Item>> jumpEntries = new HashMap<Integer, List<Item>>();
205     
206             Map<Integer, List<Item>> jumpStackEntries = new HashMap<Integer, List<Item>>();
207     
208             List<Item> locals = getInitialLocals(descriptor);
209             List<Item> stack = new ArrayList<Item>();
210             BitSet jumpEntryLocations = new BitSet();
211             if (DEBUG) {
212                 System.out.println(descriptor);
213                 System.out.println(locals);
214             }
215             int pc = 0;
216             for(StackMapTableEntry e : stackMapTable.getStackMapTable()) {
217                 pc += e.getByteCodeOffsetDelta();
218                 int rawFrameType = getFrameType(e);
219                 StackFrameType stackFrameType = StackFrameType.get(rawFrameType);
220                 switch (stackFrameType) {
221                 case SAME_FRAME:
222                     stack.clear();
223                     break;
224                 case SAME_LOCALS_1_STACK_ITEM_FRAME:
225                     stack.clear();
226                     addStack(stack, e.getTypesOfStackItems());
227                     break;
228                 case CHOP_FRAME :
229                     stack.clear();
230                     int n = Constants.CHOP_FRAME_MAX+1-rawFrameType;
231                     for(int i = 0; i < n; i++) {
232                         Item it = locals.remove(locals.size()-1);
233                         if (it == null) {
234                             it = locals.remove(locals.size()-1);
235                             assert it.usesTwoSlots();
236                         }
237                     }
238                     break;
239     
240                 case APPEND_FRAME:
241     
242                     stack.clear();
243                     addLocals(locals, e.getTypesOfLocals());
244     
245                     break;
246                 case FULL_FRAME:
247                     stack.clear();
248                     locals.clear();
249                     addLocals(locals, e.getTypesOfLocals());
250                     addStack(stack, e.getTypesOfStackItems());
251                     break;
252     
253                 }
254                 if (DEBUG) {
255                     System.out.printf("%4d %2d %2d  %12s %s%n",
256     
257                             pc,   e.getNumberOfLocals(), e.getNumberOfStackItems(), stackFrameType, e);
258                     System.out.printf("     %s :: %s%n", stack, locals);
259                 }
260                 if (pc > 0) {
261                     jumpEntries.put(pc, new ArrayList<Item>(locals));
262                     if (!stack.isEmpty()) {
263                         jumpStackEntries.put(pc, new ArrayList<Item>(stack));
264                     }
265                     jumpEntryLocations.set(pc);
266                 }
267                 pc++;
268             }
269             if (DEBUG) {
270                 System.out.println("\n");
271             }
272             return new JumpInfoFromStackMap(jumpEntries, jumpStackEntries, jumpEntryLocations);
273     
274         }

276         static  private Item getItem(StackMapType t) {
277     
278             switch (t.getType()) {
279     
280             case Constants.ITEM_Double:
281                 return Item.typeOnly("D");
282             case Constants.ITEM_Float:
283                 return  Item.typeOnly("F");
284             case Constants.ITEM_Integer:
285                 return  Item.typeOnly("I");
286             case Constants.ITEM_Long:
287                 return  Item.typeOnly("J");
288             case Constants.ITEM_Bogus:
289             case Constants.ITEM_NewObject:
290                 return Item.typeOnly("Ljava/lang/Object;");
291             case Constants.ITEM_Null:
292                 Item it = new Item();
293                 it.setSpecialKind(Item.TYPE_ONLY);
294                 return it;
295             case Constants.ITEM_InitObject:
296                 return Item.typeOnly("Ljava/lang/Object;");
297             case Constants.ITEM_Object:
298                 int index = t.getIndex();
299                 ConstantClass c = (ConstantClass) t.getConstantPool().getConstant(index);
300                 String name = c.getBytes(t.getConstantPool());
301                 if (name.charAt(0) != '[') {
302                     name = "L" + name + ";";
303                 }
304                 return Item.typeOnly(name);
305             default:
306                 throw new IllegalArgumentException("Bad item type: " + t.getType());
307     
308             }
309         }
310         static private void addLocals(List<Item> lst, StackMapType[] typesOfStackItems) {
311             for(StackMapType t : typesOfStackItems) {
312                 Item item = getItem(t);
313                 lst.add(item);
314                 if (item.usesTwoSlots()) {
315                     lst.add(null);
316                 }
317             }
318     
319         }
320         static private void addStack(List<Item> lst, StackMapType[] typesOfStackItems) {
321             for(StackMapType t : typesOfStackItems) {
322                 Item item = getItem(t);
323                 lst.add(item);
324             }
325     
326         }
