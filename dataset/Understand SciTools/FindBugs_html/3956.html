44     /**
45      * Abstract implementation of a Detector to find methods where a particular kind
46      * of created resource is not cleaned up or closed properly. Subclasses should
47      * override the abstract methods to determine what kinds of resources are
48      * tracked by the detector.
49      *
50      * @author David Hovemeyer
51      */
52     public abstract class ResourceTrackingDetector<Resource, ResourceTrackerType extends ResourceTracker<Resource>> implements
53     Detector {
54     
55         private static final boolean DEBUG = SystemProperties.getBoolean("rtd.debug");
56     
57         private static final String DEBUG_METHOD_NAME = SystemProperties.getProperty("rtd.method");
58     
59         protected BugAccumulator bugAccumulator;
60     
61         protected BugReporter bugReporter;
62     
63         public ResourceTrackingDetector(BugReporter bugReporter) {
64             this.bugReporter = bugReporter;
65             this.bugAccumulator = new BugAccumulator(bugReporter);
66         }
67     
68         public abstract boolean prescreen(ClassContext classContext, Method method, boolean mightClose);
69     
70         public abstract ResourceTrackerType getResourceTracker(ClassContext classContext, Method method)
71                 throws DataflowAnalysisException, CFGBuilderException;
72     
73         public abstract void inspectResult(ClassContext classContext, MethodGen methodGen, CFG cfg,
74                 Dataflow<ResourceValueFrame, ResourceValueAnalysis<Resource>> dataflow, Resource resource);
75     
76         @Override
77         public void visitClassContext(ClassContext classContext) {
78     
79             final JavaClass jclass = classContext.getJavaClass();
80             Method[] methodList = jclass.getMethods();
81             for (Method method : methodList) {
82                 if (method.isAbstract() || method.isNative()) {
83                     continue;
84                 }
85     
86                 MethodGen methodGen = classContext.getMethodGen(method);
87                 if (methodGen == null) {
88                     continue;
89                 }
90     
91                 if (DEBUG_METHOD_NAME != null && !DEBUG_METHOD_NAME.equals(method.getName())) {
92                     continue;
93                 }
94     
95                 if (DEBUG) {
96                     System.out.println("----------------------------------------------------------------------");
97                     System.out.println("Analyzing " + SignatureConverter.convertMethodSignature(methodGen));
98                     System.out.println("----------------------------------------------------------------------");
99                 }
100     
101                 try {
102                     ResourceTrackerType resourceTracker = getResourceTracker(classContext, method);
103                     boolean mightClose = mightCloseResource(classContext, method, resourceTracker);
104     
105                     if (!prescreen(classContext, method, mightClose)) {
106                         continue;
107                     }
108     
109                     ResourceCollection<Resource> resourceCollection = buildResourceCollection(classContext, method, resourceTracker);
110                     if (resourceCollection.isEmpty()) {
111                         continue;
112                     }
113     
114                     analyzeMethod(classContext, method, resourceTracker, resourceCollection);
115                 } catch (CFGBuilderException e) {
116                     bugReporter.logError("Error analyzing method " + method.toString(), e);
117                 } catch (DataflowAnalysisException e) {
118                     bugReporter.logError("Error analyzing method " + method.toString(), e);
119                 }
120                 bugAccumulator.reportAccumulatedBugs();
121             }
122     
123         }
124     
125         private ResourceCollection<Resource> buildResourceCollection(ClassContext classContext, Method method,
126                 ResourceTrackerType resourceTracker) throws CFGBuilderException, DataflowAnalysisException {
127     
128             ResourceCollection<Resource> resourceCollection = new ResourceCollection<Resource>();
129     
130             CFG cfg = classContext.getCFG(method);
131             ConstantPoolGen cpg = classContext.getConstantPoolGen();
132     
133             for (Iterator<Location> i = cfg.locationIterator(); i.hasNext();) {
134                 Location location = i.next();
135                 Resource resource = resourceTracker.isResourceCreation(location.getBasicBlock(), location.getHandle(), cpg);
136                 if (resource != null) {
137                     resourceCollection.addCreatedResource(location, resource);
138                 }
139             }
140     
141             return resourceCollection;
142         }
143     
144         private boolean mightCloseResource(ClassContext classContext, Method method, ResourceTrackerType resourceTracker)
145                 throws CFGBuilderException, DataflowAnalysisException {
146     
147             CFG cfg = classContext.getCFG(method);
148             ConstantPoolGen cpg = classContext.getConstantPoolGen();
149     
150             for (Iterator<Location> i = cfg.locationIterator(); i.hasNext();) {
151                 Location location = i.next();
152                 if (resourceTracker.mightCloseResource(location.getBasicBlock(), location.getHandle(), cpg)) {
153                     return true;
154                 }
155     
156             }
157     
158             return false;
159         }
160     
161         public void analyzeMethod(ClassContext classContext, Method method, ResourceTrackerType resourceTracker,
162                 ResourceCollection<Resource> resourceCollection) throws CFGBuilderException, DataflowAnalysisException {
163     
164             MethodGen methodGen = classContext.getMethodGen(method);
165             if (methodGen == null) {
166                 return;
167             }
168             try {
169                 CFG cfg = classContext.getCFG(method);
170                 DepthFirstSearch dfs = classContext.getDepthFirstSearch(method);
171     
172                 if (DEBUG) {
173                     System.out.println(SignatureConverter.convertMethodSignature(methodGen));
174                 }
175     
176                 for (Iterator<Resource> i = resourceCollection.resourceIterator(); i.hasNext();) {
177                     Resource resource = i.next();
178     
179                     ResourceValueAnalysis<Resource> analysis = new ResourceValueAnalysis<Resource>(methodGen, cfg, dfs,
180                             resourceTracker, resource);
181                     Dataflow<ResourceValueFrame, ResourceValueAnalysis<Resource>> dataflow = new Dataflow<ResourceValueFrame, ResourceValueAnalysis<Resource>>(
182                             cfg, analysis);
183     
184                     Profiler profiler = Global.getAnalysisCache().getProfiler();
185                     profiler.start(resourceTracker.getClass());
186                     try {
187                         dataflow.execute();
188                     } finally {
189                         profiler.end(resourceTracker.getClass());
190                     }
191                     inspectResult(classContext, methodGen, cfg, dataflow, resource);
192                 }
193             } catch (RuntimeException e) {
194                 AnalysisContext.logError("Exception while analyzing " + methodGen.getClassName() + "." + methodGen.getName() + ":"
195                         + methodGen.getSignature(), e);
196             }
197         }
198     
199         @Override
200         public void report() {
201         }
202     
203     }
