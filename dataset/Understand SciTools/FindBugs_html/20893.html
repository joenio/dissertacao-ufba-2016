20     package edu.umd.cs.findbugs.ba.interproc;
21     
22     import java.io.IOException;
23     import java.io.Writer;
24     
25     import org.apache.bcel.Constants;
26     
27     import edu.umd.cs.findbugs.ba.XFactory;
28     import edu.umd.cs.findbugs.ba.XField;
29     import edu.umd.cs.findbugs.classfile.DescriptorFactory;
30     import edu.umd.cs.findbugs.classfile.FieldDescriptor;
31     import edu.umd.cs.findbugs.util.ClassName;
32     
33     /**
34      * Interprocedural field property database.
35      *
36      * @author David Hovemeyer
37      */
38     public abstract class FieldPropertyDatabase<Property> extends PropertyDatabase<FieldDescriptor, Property> {
39     
40         /*
41          * (non-Javadoc)
42          *
43          * @see
44          * edu.umd.cs.findbugs.ba.interproc.PropertyDatabase#parseKey(java.lang.
45          * String)
46          */
47         @Override
48         protected FieldDescriptor parseKey(String s) throws PropertyDatabaseFormatException {
49             String[] tuple = s.split(",");
50             if (tuple.length != 4) {
51                 throw new PropertyDatabaseFormatException("Invalid field tuple: " + s);
52             }
53     
54             String className = XFactory.canonicalizeString(tuple[0]);
55             String fieldName = XFactory.canonicalizeString(tuple[1]);
56             String signature = XFactory.canonicalizeString(tuple[2]);
57             int accessFlags;
58             try {
59                 accessFlags = Integer.parseInt(tuple[3]);
60             } catch (NumberFormatException e) {
61                 throw new PropertyDatabaseFormatException("Invalid field access flags: " + tuple[3]);
62             }
63     
64             return DescriptorFactory.instance().getFieldDescriptor(ClassName.toSlashedClassName(className), fieldName, signature,
65                     (accessFlags & Constants.ACC_STATIC) != 0);
66         }
67     
68         /*
69          * (non-Javadoc)
70          *
71          * @see
72          * edu.umd.cs.findbugs.ba.interproc.PropertyDatabase#writeKey(java.io.Writer
73          * , KeyType)
74          */
75     
76         @Override
77         protected void writeKey(Writer writer, FieldDescriptor key) throws IOException {
78             writer.write(key.getClassDescriptor().getDottedClassName());
79             writer.write(",");
80             writer.write(key.getName());
81             writer.write(",");
82             writer.write(key.getSignature());
83             writer.write(",");
84             XField xField = XFactory.createXField(key);
85             int flags = xField.getAccessFlags() & 0xf;
86             writer.write(String.valueOf(flags));
87         }
88     
89     }
