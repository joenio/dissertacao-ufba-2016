41     
42     public class InconsistentAnnotations implements Detector, UseAnnotationDatabase {
43     
44         public final TypeQualifierValue<?> nonnullTypeQualifierValue;
45     
46         final BugReporter reporter;
47     
48         public InconsistentAnnotations(BugReporter reporter) {
49             ClassDescriptor nonnullClassDesc = DescriptorFactory.createClassDescriptor(javax.annotation.Nonnull.class);
50             this.nonnullTypeQualifierValue = TypeQualifierValue.getValue(nonnullClassDesc, null);
51             this.reporter = reporter;
52         }
53     
54         @Override
55         public void visitClassContext(ClassContext classContext) {
56     
57             JavaClass jclass = classContext.getJavaClass();
58     
59             for (Method method : jclass.getMethods()) {
60                 XMethod xmethod = XFactory.createXMethod(classContext.getJavaClass(), method);
61                 ParameterProperty nonnullParameters = AnalysisContext.currentAnalysisContext().getUnconditionalDerefParamDatabase()
62                         .getProperty(xmethod.getMethodDescriptor());
63                 if (nonnullParameters != null) {
64                     for (int p : nonnullParameters.iterable()) {
65                         TypeQualifierAnnotation directTypeQualifierAnnotation = TypeQualifierApplications
66                                 .getDirectTypeQualifierAnnotation(xmethod, p, nonnullTypeQualifierValue);
67                         if (directTypeQualifierAnnotation != null && directTypeQualifierAnnotation.when == When.UNKNOWN) {
68                             //
69                             // The LocalVariableAnnotation is constructed using the
70                             // local variable
71                             // number of the parameter, not the parameter number.
72                             //
73                             int paramLocal = xmethod.isStatic() ? p : p + 1;
74     
75                             reporter.reportBug(new BugInstance(this, "NP_PARAMETER_MUST_BE_NONNULL_BUT_MARKED_AS_NULLABLE",
76                                     NORMAL_PRIORITY).addClassAndMethod(jclass, method).add(
77                                             LocalVariableAnnotation.getParameterLocalVariableAnnotation(method, paramLocal)));
78     
79                         }
80     
81                     }
82                 }
83             }
84     
85         }
86     
87         @Override
88         public void report() {
89         }
90     }
