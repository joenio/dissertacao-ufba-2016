20     package edu.umd.cs.findbugs.bcel.generic;
21     
22     import org.apache.bcel.generic.ConstantPoolGen;
23     import org.apache.bcel.generic.ConversionInstruction;
24     import org.apache.bcel.generic.Type;
25     import org.apache.bcel.generic.Visitor;
26     
27     /** A synthetic instruction that converts a the nullness of a reference to a boolean value.
28      *
29      */
30     public abstract class NullnessConversationInstruction extends ConversionInstruction {
31     
32         protected NullnessConversationInstruction(short opcode) {
33             super(opcode);
34         }
35     
36         @Override
37         public Type getType( ConstantPoolGen cp ) {
38             return Type.BOOLEAN;
39         }
40     
41         @Override
42         public int consumeStack( ConstantPoolGen cpg ) {
43             return 1;
44         }
45         @Override
46         public int produceStack( ConstantPoolGen cpg ) {
47             return 1;
48         }
49     
50         @Override
51         public int hashCode() {
52             return this.getClass().hashCode();
53         }
54         /**
55          * Call corresponding visitor method(s). The order is:
56          * Call visitor methods of implemented interfaces first, then
57          * call methods according to the class hierarchy in descending order,
58          * i.e., the most specific visitXXX() call comes last.
59          *
60          * @param v Visitor object
61          */
62         @Override
63         public void accept( Visitor v ) {
64             v.visitTypedInstruction(this);
65             v.visitStackProducer(this);
66             v.visitStackConsumer(this);
67             v.visitConversionInstruction(this);
68         }
69     }
