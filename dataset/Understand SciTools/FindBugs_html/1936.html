42     /**
43      * Abstract factory class for creating analysis objects.
44      */
45     public abstract class AnalysisFactory<Analysis> implements IMethodAnalysisEngine<Analysis> {
46         private final String analysisName;
47     
48         private final Class<Analysis> analysisClass;
49     
50         /**
51          * Constructor.
52          *
53          * @param analysisName
54          *            name of the analysis factory: for diagnostics/debugging
55          */
56         public AnalysisFactory(String analysisName, Class<Analysis> analysisClass) {
57             this.analysisName = analysisName;
58             this.analysisClass = analysisClass;
59         }
60     
61         @Override
62         public String toString() {
63             return analysisName + " : " + analysisClass.getName();
64         }
65     
66         /*
67          * ----------------------------------------------------------------------
68          * IAnalysisEngine methods
69          * ----------------------------------------------------------------------
70          */
71     
72         /*
73          * (non-Javadoc)
74          *
75          * @see
76          * edu.umd.cs.findbugs.classfile.IAnalysisEngine#registerWith(edu.umd.cs
77          * .findbugs.classfile.IAnalysisCache)
78          */
79         @Override
80         public void registerWith(IAnalysisCache analysisCache) {
81             analysisCache.registerMethodAnalysisEngine(analysisClass, this);
82         }
83     
84         private static final Object NULL_ANALYSIS_RESULT = new Object();
85     
86         /*
87          * ----------------------------------------------------------------------
88          * Helper methods to get required analysis objects.
89          * ----------------------------------------------------------------------
90          */
91     
92         protected CFG getCFG(IAnalysisCache analysisCache, MethodDescriptor methodDescriptor) throws CheckedAnalysisException {
93             return analysisCache.getMethodAnalysis(CFG.class, methodDescriptor);
94         }
95     
96         protected DepthFirstSearch getDepthFirstSearch(IAnalysisCache analysisCache, MethodDescriptor methodDescriptor)
97                 throws CheckedAnalysisException {
98             return analysisCache.getMethodAnalysis(DepthFirstSearch.class, methodDescriptor);
99         }
100     
101         protected ConstantPoolGen getConstantPoolGen(IAnalysisCache analysisCache, ClassDescriptor classDescriptor)
102                 throws CheckedAnalysisException {
103             return analysisCache.getClassAnalysis(ConstantPoolGen.class, classDescriptor);
104         }
105     
106         protected MethodGen getMethodGen(IAnalysisCache analysisCache, MethodDescriptor methodDescriptor)
107                 throws CheckedAnalysisException {
108             return analysisCache.getMethodAnalysis(MethodGen.class, methodDescriptor);
109         }
110     
111         protected CompactLocationNumbering getCompactLocationNumbering(IAnalysisCache analysisCache, MethodDescriptor methodDescriptor)
112                 throws CheckedAnalysisException {
113             return analysisCache.getMethodAnalysis(CompactLocationNumbering.class, methodDescriptor);
114         }
115     
116         protected ValueNumberDataflow getValueNumberDataflow(IAnalysisCache analysisCache, MethodDescriptor methodDescriptor)
117                 throws CheckedAnalysisException {
118             return analysisCache.getMethodAnalysis(ValueNumberDataflow.class, methodDescriptor);
119         }
120     
121         protected AssertionMethods getAssertionMethods(IAnalysisCache analysisCache, ClassDescriptor classDescriptor)
122                 throws CheckedAnalysisException {
123             return analysisCache.getClassAnalysis(AssertionMethods.class, classDescriptor);
124         }
125     
126         protected JavaClass getJavaClass(IAnalysisCache analysisCache, ClassDescriptor classDescriptor)
127                 throws CheckedAnalysisException {
128             return analysisCache.getClassAnalysis(JavaClass.class, classDescriptor);
129         }
130     
131         protected Method getMethod(IAnalysisCache analysisCache, MethodDescriptor methodDescriptor) throws CheckedAnalysisException {
132             return analysisCache.getMethodAnalysis(Method.class, methodDescriptor);
133         }
134     
135         protected ReverseDepthFirstSearch getReverseDepthFirstSearch(IAnalysisCache analysisCache, MethodDescriptor methodDescriptor)
136                 throws CheckedAnalysisException {
137             return analysisCache.getMethodAnalysis(ReverseDepthFirstSearch.class, methodDescriptor);
138         }
139     
140         protected ExceptionSetFactory getExceptionSetFactory(IAnalysisCache analysisCache, MethodDescriptor methodDescriptor)
141                 throws CheckedAnalysisException {
142             return analysisCache.getMethodAnalysis(ExceptionSetFactory.class, methodDescriptor);
143         }
144     
145         protected IsNullValueDataflow getIsNullValueDataflow(IAnalysisCache analysisCache, MethodDescriptor methodDescriptor)
