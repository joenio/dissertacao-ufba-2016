49     public class FunctionsThatMightBeMistakenForProcedures extends OpcodeStackDetector implements FirstPassDetector {
50     
51         final BugReporter bugReporter;
52     
53         private final boolean testingEnabled;
54     
55         final static boolean REPORT_INFERRED_METHODS = SystemProperties.getBoolean("mrc.inferred.report");
56     
57         public FunctionsThatMightBeMistakenForProcedures(BugReporter bugReporter) {
58             this.bugReporter = bugReporter;
59             setVisitMethodsInCallOrder(true);
60             testingEnabled = SystemProperties.getBoolean("report_TESTING_pattern_in_standard_detectors");
61         }
62     
63         boolean isInnerClass, hasNonFinalFields;
64     
65         @Override
66         public void visit(JavaClass obj) {
67             isInnerClass = false;
68             hasNonFinalFields = false;
69     
70         }
71     
72         @Override
73         public void visit(Field obj) {
74             if ("this$0".equals(obj.getName())) {
75                 isInnerClass = true;
76             }
77             if (!obj.isFinal() && !obj.isStatic() && !BCELUtil.isSynthetic(obj)) {
78                 hasNonFinalFields = true;
79             }
80         }
81     
82         @Override
83         public void visitAfter(JavaClass obj) {
84             okToIgnore.clear();
85             doNotIgnore.clear();
86             doNotIgnoreHigh.clear();
87             methodsSeen.clear();
88     
89         }
90     
91         HashSet<XMethod> okToIgnore = new HashSet<XMethod>();
92     
93         HashSet<XMethod> methodsSeen = new HashSet<XMethod>();
94     
95         HashSet<XMethod> doNotIgnore = new HashSet<XMethod>();
96     
97         HashSet<XMethod> doNotIgnoreHigh = new HashSet<XMethod>();
98     
99         int returnSelf, returnOther, returnNew, returnUnknown;
100     
101         int updates;
102     
103         @CheckForNull BugInstance inferredMethod;
104     
105         @Override
106         public void visit(Code code) {
107     
108             methodsSeen.add(getXMethod());
109             String signature = getMethodSig();
110             SignatureParser parser = new SignatureParser(signature);
111     
112             String returnType = parser.getReturnTypeSignature();
113             @SlashedClassName
114             String r = ClassName.fromFieldSignature(returnType);
115             if (r == null || !r.equals(getClassName())) {
116                 return;
117             }
118             // System.out.println("Checking " + getFullyQualifiedMethodName());
119             boolean funky = false;
120             for (int i = 0; i < parser.getNumParameters(); i++) {
121                 String p = ClassName.fromFieldSignature(parser.getParameter(i));
122                 if (getClassName().equals(p)) {
123                     funky = true;
124                 }
125     
126             }
127             //
128             XMethod m = getXMethod();
129             String sourceSig = m.getSourceSignature();
130             if (sourceSig != null) {
131                 GenericSignatureParser sig = new GenericSignatureParser(sourceSig);
132                 String genericReturnValue = sig.getReturnTypeSignature();
133                 Type t = GenericUtilities.getType(genericReturnValue);
134                 if (t instanceof GenericObjectType) {
135                     funky = true;
136                 }
137     
138                 //            if (false) {
139                 //                XClass c = getXClass();
140                 //                String classSourceSig = c.getSourceSignature();
141                 //                if (!genericReturnValue.equals(classSourceSig))
142                 //                    return;
143                 //            }
144             }
145     
146             //         System.out.println("Investigating " + getFullyQualifiedMethodName());
147             returnSelf = returnOther = updates = returnNew = returnUnknown = 0;
148     
149             if (testingEnabled && REPORT_INFERRED_METHODS
150                     && AnalysisContext.currentAnalysisContext().isApplicationClass(getThisClass())) {
151                 inferredMethod = new BugInstance("TESTING", NORMAL_PRIORITY).addClassAndMethod(this);
152             } else {
153                 inferredMethod = null;
154             }
155             super.visit(code); // make callbacks to sawOpcode for all opcodes
156             //         System.out.printf("  %3d %3d %3d %3d%n", returnSelf, updates, returnOther, returnNew);
157     
158             if (returnSelf > 0 && returnOther == 0) {
159                 okToIgnore.add(m);
160             } else if (funky) {
161                 okToIgnore.add(m);
162             } else if (returnOther > 0 && returnOther >= returnSelf && returnNew > 0 && returnNew >= returnOther - 1) {
163     
164                 int priority = HIGH_PRIORITY;
165                 if (returnSelf > 0 || updates > 0) {
166                     priority++;
167                 }
168                 if (returnUnknown > 0) {
169                     priority++;
170                 }
171                 if (returnNew > 0 && priority > NORMAL_PRIORITY) {
172                     priority = NORMAL_PRIORITY;
173                 }
174                 if (updates > 0) {
175                     priority = LOW_PRIORITY;
176                 }
177                 if (priority <= HIGH_PRIORITY) {
178                     doNotIgnoreHigh.add(m);
179                 }
180                 if (priority <= NORMAL_PRIORITY) {
181                     // System.out.printf("  adding %d %s%n", priority,
182                     // MethodAnnotation.fromVisitedMethod(this).getSourceLines());
183                     doNotIgnore.add(m);
184                     if (!m.isStatic()) {
185                         XFactory xFactory = AnalysisContext.currentXFactory();
186                         xFactory.addFunctionThatMightBeMistakenForProcedures(getMethodDescriptor());
187                         if (inferredMethod != null) {
188                             inferredMethod.setPriority(priority);
189                             inferredMethod.addString(String.format("%3d %3d %5d %3d", returnOther, returnSelf, returnNew, updates));
190                             bugReporter.reportBug(inferredMethod);
191                         }
192                     }
193                 }
194     
195     
196                 inferredMethod = null;
197     
198             }
199     
200         }
201     
202         @Override
203         public void sawOpcode(int seen) {
204             switch (seen) {
205             case INVOKEVIRTUAL:
206             case INVOKESPECIAL: {
207                 if (getMethod().isStatic() || !hasNonFinalFields) {
208                     break;
209                 }
210     
211                 String name = getNameConstantOperand();
212                 String sig = getSigConstantOperand();
213                 if ((name.startsWith("set") || name.startsWith("update")) || sig.endsWith(")V")) {
214                     Item invokedOn = stack.getItemMethodInvokedOn(this);
215                     if (invokedOn.isInitialParameter() && invokedOn.getRegisterNumber() == 0) {
216                         updates++;
217                     }
218                     if (inferredMethod != null) {
219                         inferredMethod.addCalledMethod(this);
220                     }
221                 }
222                 break;
223             }
224     
225             case ARETURN: {
226                 OpcodeStack.Item rv = stack.getStackItem(0);
227                 if (rv.isNull()) {
228                     break;
229                 }
230                 if (rv.isInitialParameter()) {
231                     returnSelf++;
232                     break;
233                 }
234     
235                 XMethod xMethod = rv.getReturnValueOf();
236                 if (xMethod == null) {
237                     returnSelf++;
238                     break;
239                 }
240                 if (inferredMethod != null) {
241                     inferredMethod.addCalledMethod(xMethod);
242                 }
243                 if (okToIgnore.contains(xMethod) ) {
244                     returnSelf++;
245                     break;
246                 }
247                 if ("<init>".equals(xMethod.getName())) {
248                     String sig = xMethod.getSignature();
249                     // returning a newly constructed value
250                     boolean voidConstructor;
251                     if (!isInnerClass) {
252                         voidConstructor = "()V".equals(sig);
253                     } else {
254                         SignatureParser parser = new SignatureParser(sig);
255                         voidConstructor = parser.getNumParameters()  <= 1;
256                     }
257                     if (voidConstructor) {
258                         returnSelf++;
259                     } else {
260                         returnOther++;
261                         returnNew++;
262                     }
263                     break;
264     
265                 }
266     
267     
268                 if (xMethod.isAbstract() && !xMethod.getClassDescriptor().equals(getClassDescriptor())) {
269                     returnUnknown++;
270                     break;
271                 }
272                 if ("<init>".equals(xMethod.getName()) || doNotIgnoreHigh.contains(xMethod)) {
273                     returnOther++;
274                     // System.out.println("  calls " + xMethod);
275                     // System.out.println("  at " +
276                     // MethodAnnotation.fromXMethod(xMethod).getSourceLines());
277                     if ("<init>".equals(xMethod.getName()) || doNotIgnore.contains(xMethod)) {
278                         returnNew++;
279                     }
280                 } else if (doNotIgnore.contains(xMethod)) {
281                     returnOther++;
282                     // System.out.println("  calls " + xMethod);
283                     // System.out.println("  at " +
284                     // MethodAnnotation.fromXMethod(xMethod).getSourceLines());
285     
286                 } else {
287                     returnUnknown++;
288                 }
289     
290             }
291             break;
292             case PUTFIELD: {
293     
294                 OpcodeStack.Item rv = stack.getStackItem(1);
295                 if (rv.getRegisterNumber() == 0 && rv.isInitialParameter()) {
296                     if (inferredMethod != null) {
297                         inferredMethod.addReferencedField(this);
298                     }
299                     updates++;
300     
301                 }
302             }
303             break;
304             default:
305                 break;
306             }
307         }
308     }
