85     /**
86      * tracks the types and numbers of objects that are currently on the operand
87      * stack throughout the execution of method. To use, a detector should
88      * instantiate one for each method, and call
89      * <p>

91      * </p>

93      * the stack and see what the types of objects are on the stack, including
94      * constant values if they were pushed. The types described are of course, only

509             public static Item merge(Item i1, Item i2) {
510                 if (i1 == null) {
511                     return i2;
512                 }
513                 if (i2 == null) {
514                     return i1;
515                 }
516                 if (i1.equals(i2)) {
517                     return i1;
518                 }
519                 Item m = new Item();
520                 if (i1.getSpecialKind() == TYPE_ONLY && i2.getSpecialKind() != TYPE_ONLY) {
521                     return i2;
522                 } else if (i2.getSpecialKind() == TYPE_ONLY && i1.getSpecialKind() != TYPE_ONLY) {
523                     return i1;
524                 }
525                 m.flags = i1.flags & i2.flags;
526                 m.setCouldBeZero(i1.isCouldBeZero() || i2.isCouldBeZero());
527                 if (i1.pc == i2.pc) {
528                     m.pc = i1.pc;
529                 }
530                 if (Util.nullSafeEquals(i1.signature, i2.signature)) {
531                     m.signature = i1.signature;
532                 } else if (i1.isNull()) {
533                     m.signature = i2.signature;
534                 } else if (i2.isNull()) {
535                     m.signature = i1.signature;
536                 }
537                 if (Util.nullSafeEquals(i1.constValue, i2.constValue)) {
538                     m.constValue = i1.constValue;
539                 }
540                 if (Util.nullSafeEquals(i1.source, i2.source)) {
541                     m.source = i1.source;
542                 } else if ("".equals(i1.constValue)) {
543                     m.source = i2.source;
544                 } else if ("".equals(i2.constValue)) {
545                     m.source = i1.source;
546                 }
547     
548                 if (Util.nullSafeEquals(i1.userValue, i2.userValue)) {
549                     m.userValue = i1.userValue;
550                 }
551     
552                 if (i1.registerNumber == i2.registerNumber) {
553                     m.registerNumber = i1.registerNumber;
554                 }
555                 if (i1.fieldLoadedFromRegister == i2.fieldLoadedFromRegister) {
556                     m.fieldLoadedFromRegister = i1.fieldLoadedFromRegister;
557                 }
558     
559                 if (i1.specialKind == SERVLET_REQUEST_TAINTED) {
560                     m.specialKind = SERVLET_REQUEST_TAINTED;
561                     m.injection = i1.injection;
562                 } else if (i2.specialKind == SERVLET_REQUEST_TAINTED) {
563                     m.specialKind = SERVLET_REQUEST_TAINTED;
564                     m.injection = i2.injection;
565                 } else if (i1.specialKind == i2.specialKind) {
566                     m.specialKind = i1.specialKind;
567                 } else if (i1.specialKind == NASTY_FLOAT_MATH || i2.specialKind == NASTY_FLOAT_MATH) {
568                     m.specialKind = NASTY_FLOAT_MATH;
569                 } else if (i1.specialKind == FLOAT_MATH || i2.specialKind == FLOAT_MATH) {
570                     m.specialKind = FLOAT_MATH;
571                 }
572                 if (DEBUG) {
573                     System.out.println("Merge " + i1 + " and " + i2 + " gives " + m);
574                 }
575                 return m;
576             }

1103         public void mergeJumps(DismantleBytecode dbc) {
1104             if (!needToMerge) {
1105                 return;
1106             }
1107             needToMerge = false;
1108             if (dbc.getPC() == zeroOneComing) {
1109                 pop();
1110                 top = false;
1111                 OpcodeStack.Item item = new Item("I");
1112                 if (oneMeansNull) {
1113                     item.setSpecialKind(Item.NONZERO_MEANS_NULL);
1114                 } else {
1115                     item.setSpecialKind(Item.ZERO_MEANS_NULL);
1116                 }
1117                 item.setPC(dbc.getPC() - 8);
1118                 item.setCouldBeZero(true);
1119     
1120                 push(item);
1121     
1122                 zeroOneComing = -1;
1123                 if (DEBUG) {
1124                     System.out.println("Updated to " + this);
1125                 }
1126                 return;
1127             }
1128     
1129             boolean stackUpdated = false;
1130             if (!isTop() && (convertJumpToOneZeroState == 3 || convertJumpToZeroOneState == 3)) {
1131                 pop();
1132                 Item topItem = new Item("I");
1133                 topItem.setCouldBeZero(true);
1134                 push(topItem);
1135                 convertJumpToOneZeroState = convertJumpToZeroOneState = 0;
1136                 stackUpdated = true;
1137             }
1138     
1139             List<Item> jumpEntry = null;
1140             if (jumpEntryLocations.get(dbc.getPC())) {
1141                 jumpEntry = jumpEntries.get(Integer.valueOf(dbc.getPC()));
1142             }
1143             boolean wasReachOnlyByBranch = isReachOnlyByBranch();
1144             if (jumpEntry != null) {
1145                 setReachOnlyByBranch(false);
1146                 List<Item> jumpStackEntry = jumpStackEntries.get(Integer.valueOf(dbc.getPC()));
1147     
1148                 if (DEBUG2) {
1149                     if (wasReachOnlyByBranch) {
1150                         System.out.println("Reached by branch at " + dbc.getPC() + " with " + jumpEntry);
1151                         if (jumpStackEntry != null) {
1152                             System.out.println(" and stack " + jumpStackEntry);
1153                         }
1154                     } else if (!jumpEntry.equals(lvValues)
1155                             || jumpStackEntry != null && !jumpStackEntry.equals(stack)) {
1156     
1157                         System.out.println("Merging at " + dbc.getPC() + " with " + jumpEntry);
1158                         if (jumpStackEntry != null) {
1159                             System.out.println(" and stack " + jumpStackEntry);
1160                         }
1161     
1162                     }
1163     
1164                 }
1165                 if (isTop()) {
1166                     lvValues = new ArrayList<Item>(jumpEntry);
1167                     if (jumpStackEntry != null) {
1168                         stack = new ArrayList<Item>(jumpStackEntry);
1169                     } else {
1170                         stack.clear();
1171                     }
1172                     setTop(false);
1173                     return;
1174                 }
1175                 if (isReachOnlyByBranch()) {
1176                     setTop(false);
1177                     lvValues = new ArrayList<Item>(jumpEntry);
1178                     if (!stackUpdated) {
1179                         if (jumpStackEntry != null) {
1180                             stack = new ArrayList<Item>(jumpStackEntry);
1181                         } else {
1182                             stack.clear();
1183                         }
1184                     }
1185     
1186                 } else {
1187                     setTop(false);
1188                     mergeLists(lvValues, jumpEntry, false);
1189                     if (!stackUpdated && jumpStackEntry != null) {
1190                         mergeLists(stack, jumpStackEntry, false);
1191                     }
1192                 }
1193                 if (DEBUG) {
1194                     System.out.println(" merged lvValues " + lvValues);
1195                 }
1196             } else if (isReachOnlyByBranch() && !stackUpdated) {
1197                 stack.clear();
1198     
1199                 Item item = null;
1200                 for (CodeException e : dbc.getCode().getExceptionTable()) {
1201                     if (e.getHandlerPC() == dbc.getPC()) {
1202                         Item newItem = new Item(getExceptionSig(dbc, e));
1203                         if (item == null) {
1204                             item = newItem;
1205                         } else {
1206                             item = Item.merge(item, newItem);
1207                         }
1208     
1209     
1210                     }
1211                 }
1212     
1213                 if (item != null) {
1214                     push(item);
1215                     setReachOnlyByBranch(false);
1216                     setTop(false);
1217                 } else {
1218     
1219                     setTop(true);
1220                 }
1221     
1222     
1223             }
