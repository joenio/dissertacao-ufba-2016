20     package edu.umd.cs.findbugs.anttask;
21     
22     import java.io.File;
23     import java.util.ArrayList;
24     
25     import org.apache.tools.ant.BuildException;
26     import org.apache.tools.ant.types.FileSet;
27     
28     /**
29      * An ant task that is wraps the behavior of the UnionResults executable into an
30      * ant task.
31      *
32      * <taskdef name="UnionBugs2" classname="edu.umd.cs.findbugs.anttask.UnionBugs2"
33      * classpath="...">
34      *
35      * <UnionBugs2 to="${basedir}/findbugs.xml" > <fileset dir="plugins"> <include
36      * name="*_findbugs_partial.xml" /> </fileset> </UnionBugs>
37      *
38      *
39      * @ant.task category="utility"
40      *
41      */
42     
43     public class UnionBugs2 extends AbstractFindBugsTask {
44     
45         private String to;
46     
47         private final ArrayList<FileSet> fileSets = new ArrayList<FileSet>();
48     
49         public UnionBugs2() {
50             super("edu.umd.cs.findbugs.workflow.UnionResults");
51             setFailOnError(true);
52         }
53     
54         /**
55          * The fileset containing all the findbugs xml files that need to be merged
56          *
57          * @param arg
58          */
59         public void addFileset(FileSet arg) {
60             this.fileSets.add(arg);
61         }
62     
63         /**
64          * The File everything should get merged into
65          *
66          * @param arg
67          */
68         public void setTo(String arg) {
69             this.to = arg;
70         }
71     
72         @Override
73         protected void checkParameters() {
74             super.checkParameters();
75     
76             if (to == null) {
77                 throw new BuildException("to attribute is required", getLocation());
78             }
79     
80             if (fileSets.size() < 1) {
81                 throw new BuildException("fileset is required");
82             }
83         }
84     
85     
86         @Override
87         protected void beforeExecuteJavaProcess() {
88             log("unioning bugs...");
89         }
90     
91         @Override
92         protected void configureFindbugsEngine() {
93             addArg("-withMessages");
94             addArg("-output");
95             addArg(to);
96             for (FileSet s : fileSets) {
97                 File fromDir = s.getDir(getProject());
98                 for (String file : s.getDirectoryScanner(getProject()).getIncludedFiles()) {
99                     addArg(new File(fromDir, file).toString());
100                 }
101             }
102     
103         }
104     
105     }
