19     package edu.umd.cs.findbugs.detect;
20     
21     import java.util.Arrays;
22     import java.util.List;
23     
24     import org.apache.bcel.classfile.Code;
25     import org.apache.bcel.classfile.JavaClass;
26     import org.apache.bcel.classfile.Signature;
27     
28     import edu.umd.cs.findbugs.BugAccumulator;
29     import edu.umd.cs.findbugs.BugInstance;
30     import edu.umd.cs.findbugs.BugReporter;
31     import edu.umd.cs.findbugs.OpcodeStack;
32     import edu.umd.cs.findbugs.ba.ClassContext;
33     import edu.umd.cs.findbugs.bcel.OpcodeStackDetector;
34     import edu.umd.cs.findbugs.classfile.MethodDescriptor;
35     
36     /**
37      * equals and hashCode are blocking methods on URL's. Warn about invoking equals
38      * or hashCode on them, or defining Set or Maps with them as keys.
39      */
40     public class URLProblems extends OpcodeStackDetector {
41     
42         private static final MethodDescriptor URL_EQUALS = new MethodDescriptor("java/net/URL", "equals", "(Ljava/lang/Object;)Z");
43         private static final MethodDescriptor URL_HASHCODE = new MethodDescriptor("java/net/URL", "hashCode", "()I");
44     
45         final static String[] BAD_SIGNATURES = { "Hashtable<Ljava/net/URL", "Map<Ljava/net/URL", "Set<Ljava/net/URL" };
46     
47         // Must be sorted
48         private static final String[] HASHSET_KEY_METHODS = {"add", "contains", "remove"};
49         private static final String[] HASHMAP_KEY_METHODS = {"containsKey", "get", "remove"};
50         private static final String[] HASHMAP_TWO_ARG_KEY_METHODS = {"put"};
51     
52         private static final List<MethodDescriptor> methods = Arrays.asList(URL_EQUALS, URL_HASHCODE,
53                 new MethodDescriptor("", "add", "(Ljava/lang/Object;)Z"),
54                 new MethodDescriptor("", "contains", "(Ljava/lang/Object;)Z"),
55                 new MethodDescriptor("", "remove", "(Ljava/lang/Object;)Z"),
56                 new MethodDescriptor("", "containsKey", "(Ljava/lang/Object;)Z"),
57                 new MethodDescriptor("", "get", "(Ljava/lang/Object;)Ljava/lang/Object;"),
58                 new MethodDescriptor("", "remove", "(Ljava/lang/Object;)Ljava/lang/Object;"),
59                 new MethodDescriptor("", "put", "(Ljava/lang/Object;Ljava/lang/Object;)Ljava/lang/Object;"));
60     
61         final private BugReporter bugReporter;
62     
63         final private BugAccumulator accumulator;
64     
65         private boolean hasInterestingMethodCalls;
66     
67         public URLProblems(BugReporter bugReporter) {
68             this.bugReporter = bugReporter;
69             this.accumulator = new BugAccumulator(bugReporter);
70         }
71     
72         @Override
73         public void visitClassContext(ClassContext classContext) {
74             this.hasInterestingMethodCalls = hasInterestingMethod(classContext.getJavaClass().getConstantPool(), methods);
75             super.visitClassContext(classContext);
76         }
77     
78         @Override
79         public void visitAfter(JavaClass obj) {
80             accumulator.reportAccumulatedBugs();
81         }
82     
83         @Override
84         public void visit(Code obj) {
85             if(this.hasInterestingMethodCalls) {
86                 super.visit(obj);
87             }
88         }
89     
90         @Override
91         public void visit(Signature obj) {
92             String sig = obj.getSignature();
93             for (String s : BAD_SIGNATURES) {
94                 if (sig.indexOf(s) >= 0) {
95                     if (visitingField()) {
96                         bugReporter.reportBug(new BugInstance(this, "DMI_COLLECTION_OF_URLS", HIGH_PRIORITY).addClass(this)
97                                 .addVisitedField(this));
98                     } else if (visitingMethod()) {
99                         bugReporter.reportBug(new BugInstance(this, "DMI_COLLECTION_OF_URLS", HIGH_PRIORITY).addClassAndMethod(this));
100                     } else {
101                         bugReporter.reportBug(new BugInstance(this, "DMI_COLLECTION_OF_URLS", HIGH_PRIORITY).addClass(this).addClass(
102                                 this));
103                     }
104                 }
105             }
106         }
107     
108         void check(String className, String[] methodNames, int target, int url) {
109             if (Arrays.binarySearch(methodNames, getNameConstantOperand()) < 0) {
110                 return;
111             }
112             if (stack.getStackDepth() <= target) {
113                 return;
114             }
115             OpcodeStack.Item targetItem = stack.getStackItem(target);
116             OpcodeStack.Item urlItem = stack.getStackItem(url);
117             if (!"Ljava/net/URL;".equals(urlItem.getSignature())) {
118                 return;
119             }
120             if (!targetItem.getSignature().equals(className)) {
121                 return;
122             }
123             accumulator.accumulateBug(new BugInstance(this, "DMI_COLLECTION_OF_URLS", HIGH_PRIORITY).addClassAndMethod(this)
124                     .addCalledMethod(this), this);
125         }
126     
127         @Override
128         public void sawOpcode(int seen) {
129     
130             if (seen == INVOKEVIRTUAL || seen == INVOKEINTERFACE) {
131                 check("Ljava/util/HashSet;", HASHSET_KEY_METHODS, 1, 0);
132                 check("Ljava/util/HashMap;", HASHMAP_KEY_METHODS, 1, 0);
133                 check("Ljava/util/HashMap;", HASHMAP_TWO_ARG_KEY_METHODS, 2, 1);
134             }
135     
136             if (seen == INVOKEVIRTUAL && (getMethodDescriptorOperand().equals(URL_EQUALS)
137                     || getMethodDescriptorOperand().equals(URL_HASHCODE))) {
138                 accumulator.accumulateBug(
139                         new BugInstance(this, "DMI_BLOCKING_METHODS_ON_URL", HIGH_PRIORITY).addClassAndMethod(this)
140                         .addCalledMethod(this), this);
141             }
142         }
143     }
