20     package edu.umd.cs.findbugs.detect;
21     
22     import org.apache.bcel.classfile.Code;
23     
24     import edu.umd.cs.findbugs.BugAccumulator;
25     import edu.umd.cs.findbugs.BugInstance;
26     import edu.umd.cs.findbugs.BugReporter;
27     import edu.umd.cs.findbugs.BytecodeScanningDetector;
28     
29     public class BadUseOfReturnValue extends BytecodeScanningDetector {
30     
31         BugAccumulator bugAccumulator;
32     
33         public BadUseOfReturnValue(BugReporter bugReporter) {
34             this.bugAccumulator = new BugAccumulator(bugReporter);
35         }
36     
37         boolean readLineOnTOS = false;
38     
39         boolean stringIndexOfOnTOS = false;
40     
41         @Override
42         public void visit(Code obj) {
43             stringIndexOfOnTOS = false;
44             readLineOnTOS = false;
45             super.visit(obj);
46             bugAccumulator.reportAccumulatedBugs();
47         }
48     
49         @Override
50         public void sawOpcode(int seen) {
51             if (seen == INVOKEVIRTUAL && "indexOf".equals(getNameConstantOperand())
52                     && "java/lang/String".equals(getClassConstantOperand())
53                     && "(Ljava/lang/String;)I".equals(getSigConstantOperand())) {
54                 stringIndexOfOnTOS = true;
55             } else if (stringIndexOfOnTOS) {
56                 if (seen == IFLE || seen == IFGT) {
57                     bugAccumulator.accumulateBug(
58                             new BugInstance(this, "RV_CHECK_FOR_POSITIVE_INDEXOF", LOW_PRIORITY).addClassAndMethod(this), this);
59                 }
60                 stringIndexOfOnTOS = false;
61             }
62     
63             if (seen == INVOKEVIRTUAL && "readLine".equals(getNameConstantOperand())
64                     && "()Ljava/lang/String;".equals(getSigConstantOperand()) && getClassConstantOperand().startsWith("java/io")
65                     && !"java/io/LineNumberReader".equals(getClassConstantOperand())) {
66                 readLineOnTOS = true;
67             } else if (readLineOnTOS) {
68                 if (seen == IFNULL || seen == IFNONNULL) {
69                     bugAccumulator.accumulateBug(
70                             new BugInstance(this, "RV_DONT_JUST_NULL_CHECK_READLINE", NORMAL_PRIORITY).addClassAndMethod(this), this);
71                 }
72     
73                 readLineOnTOS = false;
74             }
75         }
76     
77     }
