41     public class SelfMethodCalls {
42     
43         static private boolean interestingSignature(String signature) {
44             return !"()V".equals(signature);
45         }
46     
47         public static <T> MultiMap<T, T> getSelfCalls(final ClassDescriptor classDescriptor, final Map<String, T> methods) {
48             final MultiMap<T, T> map = new MultiMap<T, T>(HashSet.class);
49     
50             FBClassReader reader;
51             try {
52                 reader = Global.getAnalysisCache().getClassAnalysis(FBClassReader.class, classDescriptor);
53             } catch (CheckedAnalysisException e) {
54                 AnalysisContext.logError("Error finding self method calls for " + classDescriptor, e);
55                 return map;
56             }
57             reader.accept(new ClassVisitor(FindBugsASM.ASM_VERSION) {
58     
59                 @Override
60                 public MethodVisitor visitMethod(final int access, final String name, final String desc, String signature,
61                         String[] exceptions) {
62                     return new MethodVisitor(FindBugsASM.ASM_VERSION) {
63                         @Override
64                         public void visitMethodInsn(int opcode, String owner, String name2, String desc2, boolean itf) {
65                             if (owner.equals(classDescriptor.getClassName()) && interestingSignature(desc2)) {
66                                 T from = methods.get(name + desc + ((access & Opcodes.ACC_STATIC) != 0));
67                                 T to = methods.get(name2 + desc2 + (opcode == Opcodes.INVOKESTATIC));
68                                 map.add(from, to);
69                             }
70     
71                         }
72     
73                     };
74                 }
75     
76             }, ClassReader.SKIP_DEBUG | ClassReader.SKIP_FRAMES);
77             return map;
78         }
