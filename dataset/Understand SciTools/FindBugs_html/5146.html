41         @Override
42         public void visitClassContext(ClassContext classContext) {
43             Method[] methodList = classContext.getJavaClass().getMethods();
44     
45             for (Method method : methodList) {
46                 if (method.getCode() == null) {
47                     continue;
48                 }
49     
50                 try {
51                     analyzeMethod(classContext, method);
52                 } catch (MethodUnprofitableException mue) {
53                     if (SystemProperties.getBoolean("unprofitable.debug")) {
54                         // don't
55                         // report
56                         bugReporter.logError("skipping unprofitable method in " + getClass().getName());
57                     }
58                 } catch (CFGBuilderException e) {
59                     bugReporter.logError("Detector " + this.getClass().getName() + " caught exception", e);
60                 } catch (DataflowAnalysisException e) {
61                     bugReporter.logError("Detector " + this.getClass().getName() + " caught exception", e);
62                 }
63             }
64         }

66         static boolean booleanComparisonMethod(String methodName) {
67             return "equals".equals(methodName) ||"endsWith".equals(methodName) || "startsWith".equals(methodName)
68                     || "contains".equals(methodName) || "equalsIgnoreCase".equals(methodName);
69         }

71         static boolean comparatorMethod(String methodName) {
72             return  "compareTo".equals(methodName) || "compareToIgnoreCase".equals(methodName);
73         }
74         private void analyzeMethod(ClassContext classContext, Method method) throws CFGBuilderException, DataflowAnalysisException {
75             CFG cfg = classContext.getCFG(method);
76             ValueNumberDataflow valueNumberDataflow = classContext.getValueNumberDataflow(method);
77             ConstantPoolGen cpg = classContext.getConstantPoolGen();
78             MethodGen methodGen = classContext.getMethodGen(method);
79             String sourceFile = classContext.getJavaClass().getSourceFileName();
80     
81             for (Iterator<Location> i = cfg.locationIterator(); i.hasNext();) {
82                 Location location = i.next();
83     
84                 Instruction ins = location.getHandle().getInstruction();
85                 switch (ins.getOpcode()) {
86                 case INVOKEVIRTUAL:
87                 case INVOKEINTERFACE:
88                     InvokeInstruction iins = (InvokeInstruction) ins;
89                     String invoking = iins.getName(cpg);
90                     if ( comparatorMethod(invoking) || booleanComparisonMethod(invoking) ) {
91                         if (methodGen.getName().toLowerCase().indexOf("test") >= 0) {
92                             break;
93                         }
94                         if (methodGen.getClassName().toLowerCase().indexOf("test") >= 0) {
95                             break;
96                         }
97                         if (classContext.getJavaClass().getSuperclassName().toLowerCase().indexOf("test") >= 0) {
98                             break;
99                         }
100                         if (location.getHandle().getNext().getInstruction().getOpcode() == POP) {
101                             break;
102                         }
103                         String sig = iins.getSignature(cpg);
104     
105                         SignatureParser parser = new SignatureParser(sig);
106                         if (parser.getNumParameters() == 1
107                                 && ( booleanComparisonMethod(invoking)  && sig.endsWith(";)Z") || comparatorMethod(invoking) && sig.endsWith(";)I"))) {
108                             checkForSelfOperation(classContext, location, valueNumberDataflow, "COMPARISON", method, methodGen,
109                                     sourceFile);
110                         }
111     
112                     }
113                     break;
114     
115                 case LOR:
116                 case LAND:
117                 case LXOR:
118                 case LSUB:
119                 case IOR:
120                 case IAND:
121                 case IXOR:
122                 case ISUB:
123                     checkForSelfOperation(classContext, location, valueNumberDataflow, "COMPUTATION", method, methodGen, sourceFile);
124                     break;
125                 case FCMPG:
126                 case DCMPG:
127                 case DCMPL:
128                 case FCMPL:
129                     break;
130                 case LCMP:
131                 case IF_ACMPEQ:
132                 case IF_ACMPNE:
133                 case IF_ICMPNE:
134                 case IF_ICMPEQ:
135                 case IF_ICMPGT:
136                 case IF_ICMPLE:
137                 case IF_ICMPLT:
138                 case IF_ICMPGE:
139                     checkForSelfOperation(classContext, location, valueNumberDataflow, "COMPARISON", method, methodGen, sourceFile);
140                     break;
141                 default:
142                     break;
143                 }
144     
145             }
146         }

149          * @param classContext
150          * @param location
151          * @param method
152          * @param methodGen
153          * @param sourceFile
154          * @throws DataflowAnalysisException
155          */
156         private void checkForSelfOperation(ClassContext classContext, Location location, ValueNumberDataflow valueNumberDataflow,
157                 String op, Method method, MethodGen methodGen, String sourceFile) throws DataflowAnalysisException {
158             ValueNumberFrame frame = valueNumberDataflow.getFactAtLocation(location);
159             if (!frame.isValid()) {
160                 return;
161             }
162             Instruction ins = location.getHandle().getInstruction();
163             int opcode = ins.getOpcode();
164             int offset = 1;
165             if (opcode == LCMP || opcode == LXOR || opcode == LAND || opcode == LOR || opcode == LSUB) {
166                 offset = 2;
167             }
168             ValueNumber v0 = frame.getStackValue(0);
169             ValueNumber v1 = frame.getStackValue(offset);
170             if (!v1.equals(v0)) {
171                 return;
172             }
173             if (v0.hasFlag(ValueNumber.CONSTANT_CLASS_OBJECT) || v0.hasFlag(ValueNumber.CONSTANT_VALUE)) {
174                 return;
175             }
176     
177             int priority = HIGH_PRIORITY;
178             if (opcode == ISUB || opcode == LSUB || opcode == INVOKEINTERFACE || opcode == INVOKEVIRTUAL) {
179                 priority = NORMAL_PRIORITY;
180             }
181             XField field = ValueNumberSourceInfo.findXFieldFromValueNumber(method, location, v0, frame);
182             BugAnnotation annotation;
183             String prefix;
184             if (field != null) {
185                 if (field.isVolatile()) {
186                     return;
187                 }
188                 if (true) {
189                     return; // don't report these; too many false positives
190                 }
191                 //            annotation = FieldAnnotation.fromXField(field);
192                 //            prefix = "SA_FIELD_SELF_";
193             } else {
194                 annotation = ValueNumberSourceInfo.findLocalAnnotationFromValueNumber(method, location, v0, frame);
195                 prefix = "SA_LOCAL_SELF_";
196                 if (opcode == ISUB) {
197                     return; // only report this if simple detector reports it
198                 }
199             }
200             if (annotation == null) {
201                 return;
202             }
203             SourceLineAnnotation sourceLine = SourceLineAnnotation.fromVisitedInstruction(classContext, methodGen, sourceFile,
204                     location.getHandle());
205             int line = sourceLine.getStartLine();
206             BitSet occursMultipleTimes = classContext.linesMentionedMultipleTimes(method);
207             if (line > 0 && occursMultipleTimes.get(line)) {
208                 return;
209             }
210             BugInstance bug = new BugInstance(this, prefix + op, priority).addClassAndMethod(methodGen, sourceFile);
211             if (ins instanceof InvokeInstruction) {
212                 bug.addCalledMethod(classContext.getConstantPoolGen(), (InvokeInstruction) ins);
213             }
214     
215             bug.add(annotation)
216             .addSourceLine(classContext, methodGen, sourceFile, location.getHandle());
217             bugReporter.reportBug(bug);
218         }
