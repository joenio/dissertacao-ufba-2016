20     package edu.umd.cs.findbugs.ba;
21     
22     import org.apache.bcel.Constants;
23     import org.apache.bcel.generic.ConstantPoolGen;
24     import org.apache.bcel.generic.Instruction;
25     import org.apache.bcel.generic.InstructionHandle;
26     
27     /**
28      * A really simple forward dataflow analysis to find the depth of the Java
29      * operand stack. This is more of a proof of concept for the dataflow analysis
30      * framework than anything useful.
31      *
32      * @see Dataflow
33      * @see DataflowAnalysis
34      */
35     public class StackDepthAnalysis extends ForwardDataflowAnalysis<StackDepth> {
36         public static final int TOP = -1;
37     
38         public static final int BOTTOM = -2;
39     
40         private final ConstantPoolGen cpg;
41     
42         /**
43          * Constructor.
44          *
45          * @param cpg
46          *            the ConstantPoolGen of the method whose CFG we're performing
47          *            the analysis on
48          * @param dfs
49          *            DepthFirstSearch of the method's CFG
50          */
51         public StackDepthAnalysis(ConstantPoolGen cpg, DepthFirstSearch dfs) {
52             super(dfs);
53             this.cpg = cpg;
54         }
55     
56         @Override
57         public StackDepth createFact() {
58             return new StackDepth(TOP);
59         }
60     
61         @Override
62         public void makeFactTop(StackDepth fact) {
63             fact.setDepth(TOP);
64         }
65     
66         @Override
67         public boolean isTop(StackDepth fact) {
68             return fact.getDepth() == TOP;
69         }
70     
71         @Override
72         public boolean isFactValid(StackDepth fact) {
73             int depth = fact.getDepth();
74             return depth != TOP && depth != BOTTOM;
75         }
76     
77         @Override
78         public void copy(StackDepth source, StackDepth dest) {
79             dest.setDepth(source.getDepth());
80         }
81     
82         @Override
83         public void initEntryFact(StackDepth entryFact) {
84             entryFact.setDepth(0); // stack depth == 0 at entry to CFG
85         }
86     
87         @Override
88         public boolean same(StackDepth fact1, StackDepth fact2) {
89             return fact1.getDepth() == fact2.getDepth();
90         }
91     
92         @Override
93         public void transferInstruction(InstructionHandle handle, BasicBlock basicBlock, StackDepth fact)
94                 throws DataflowAnalysisException {
95             Instruction ins = handle.getInstruction();
96             int produced = ins.produceStack(cpg);
97             int consumed = ins.consumeStack(cpg);
98             if (produced == Constants.UNPREDICTABLE || consumed == Constants.UNPREDICTABLE) {
99                 throw new IllegalStateException("Unpredictable stack delta for instruction: " + handle);
100             }
101             int depth = fact.getDepth();
102             depth += (produced - consumed);
103             if (depth < 0) {
104                 fact.setDepth(BOTTOM);
105             } else {
106                 fact.setDepth(depth);
107             }
108         }
109     
110         @Override
111         public void meetInto(StackDepth fact, Edge edge, StackDepth result) {
112             int a = fact.getDepth();
113             int b = result.getDepth();
114             int combined;
115     
116             if (a == TOP) {
117                 combined = b;
118             } else if (b == TOP) {
119                 combined = a;
120             } else if (a == BOTTOM || b == BOTTOM || a != b) {
121                 combined = BOTTOM;
122             } else {
123                 combined = a;
124             }
125     
126             result.setDepth(combined);
127         }
