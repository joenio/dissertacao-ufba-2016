37             TreeSet<FindBugsMain> cmds = new TreeSet<FindBugsMain>();

42             }
43             for(FindBugsMain m : cmds) {
44                 System.out.printf("fb %-12s %-12s %s%n", m.cmd, m.kind, m.description);
45             }

56             showGeneralOptions();
57     
58         }
