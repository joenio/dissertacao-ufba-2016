20     package edu.umd.cs.findbugs.ba.jsr305;
21     
22     import java.lang.annotation.Annotation;
23     import java.security.Permission;
24     
25     import javax.annotation.CheckForNull;
26     import javax.annotation.meta.TypeQualifierValidator;
27     import javax.annotation.meta.When;
28     
29     final class ValidationSecurityManager extends SecurityManager {
30     
31     
32         static final ValidationSecurityManager INSTANCE = new ValidationSecurityManager();
33         final static ValidatorClassLoader VALIDATOR_LOADER = ValidatorClassLoader.INSTANCE;
34     
35     
36         static {
37             if (TypeQualifierValue.DEBUG_CLASSLOADING) {
38                 new RuntimeException("Creating ValidationSecurityManager #").printStackTrace();
39             }
40     
41         }
42         public static <A extends Annotation> When sandboxedValidation(A proxy, TypeQualifierValidator<A> v, @CheckForNull
43                 Object constantValue) {
44             if (performingValidation.get()) {
45                 throw new IllegalStateException("recursive validation");
46             }
47     
48             try {
49                 performingValidation.set(Boolean.TRUE);
50                 if (TypeQualifierValue.DEBUG_CLASSLOADING) {
51                     System.out.println("Performing validation in thread " + Thread.currentThread().getName());
52                 }
53                 try {
54                     When result = v.forConstantValue(proxy, constantValue);
55                     if (!performingValidation.get()) {
56                         throw new IllegalStateException("performingValidation not set when validation completes");
57                     }
58                     return result;
59                 } catch (ClassCastException e) {
60                     Class<? extends Annotation> c = proxy.getClass();
61                     System.out.println(c.getName() + " extends " + c.getSuperclass().getName());
62                     for(Class<?> i : c.getInterfaces()) {
63                         System.out.println("  " + i.getName());
64                     }
65                     throw e;
66                 }
67     
68             } finally {
69                 performingValidation.set(Boolean.FALSE);
70                 if (TypeQualifierValue.DEBUG_CLASSLOADING) {
71                     System.out.println("Validation finished in thread " + Thread.currentThread().getName());
72                 }
73     
74             }
75         }
76     
77     
78         @Override
79         public void checkPermission(Permission perm) {
80             if (TypeQualifierValue.DEBUG_CLASSLOADING) {
81                 System.out.println("Checking for " + perm + " permission in thread " + Thread.currentThread().getName());
82             }
83             if (performingValidation.get() && inValidation()) {
84                 SecurityException e = new SecurityException("No permissions granted while performing JSR-305 validation");
85                 if (TypeQualifierValue.DEBUG_CLASSLOADING) {
86                     e.printStackTrace(System.out);
87                 }
88                 throw e;
89             }
90     
91         }
92     
93         @Override
94         public void checkPermission(Permission perm, Object context) {
95             if (TypeQualifierValue.DEBUG_CLASSLOADING) {
96                 System.out.println("Checking for " + perm + " permission with content in thread " + Thread.currentThread().getName());
97             }
98     
99             if (performingValidation.get() && inValidation()) {
100                 SecurityException e = new SecurityException("No permissions granted while performing JSR-305 validation");
101                 if (TypeQualifierValue.DEBUG_CLASSLOADING) {
102                     e.printStackTrace(System.out);
103                 }
104                 throw e;
105             }
106         }
107     
108         private ValidationSecurityManager() { }
109     
110         private boolean inValidation() {
111             for (Class<?> c : getClassContext()) {
112                 if (TypeQualifierValidator.class.isAssignableFrom(c)
113                         || c.getClassLoader() == VALIDATOR_LOADER) {
114                     return true;
115                 }
116             }
117             return false;
118         }
119     
120         private static final ThreadLocal<Boolean> performingValidation = new ThreadLocal<Boolean>() {
121             @Override
122             protected Boolean initialValue() {
123                 return Boolean.FALSE;
124             }
125     
126         };
127     
128     
129     }
