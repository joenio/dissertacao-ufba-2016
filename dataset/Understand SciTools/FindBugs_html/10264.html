20     package edu.umd.cs.findbugs.graph;
21     
22     import java.util.HashMap;
23     import java.util.Iterator;
24     import java.util.LinkedList;
25     
26     /**
27      * A search tree callback implementation that builds a list of SearchTrees
28      * recording a graph search.
29      *
30      * @see SearchTreeCallback
31      * @author David Hovemeyer
32      */
33     public class SearchTreeBuilder<VertexType extends GraphVertex<VertexType>> implements SearchTreeCallback<VertexType> {
34     
35         private final HashMap<VertexType, SearchTree<VertexType>> searchTreeMap = new HashMap<VertexType, SearchTree<VertexType>>();
36     
37         private final LinkedList<SearchTree<VertexType>> searchTreeList = new LinkedList<SearchTree<VertexType>>();
38     
39         @Override
40         public void startSearchTree(VertexType vertex) {
41             searchTreeList.add(createSearchTree(vertex));
42         }
43     
44         @Override
45         public void addToSearchTree(VertexType parent, VertexType child) {
46             SearchTree<VertexType> parentTree = searchTreeMap.get(parent);
47             if (parentTree == null) {
48                 throw new IllegalStateException();
49             }
50             SearchTree<VertexType> childTree = createSearchTree(child);
51             parentTree.addChild(childTree);
52         }
53     
54         /**
55          * Get an Iterator over the recorded SearchTrees.
56          */
57         public Iterator<SearchTree<VertexType>> searchTreeIterator() {
58             return searchTreeList.iterator();
59         }
60     
61         private SearchTree<VertexType> createSearchTree(VertexType vertex) {
62             SearchTree<VertexType> searchTree = new SearchTree<VertexType>(vertex);
63             searchTreeMap.put(vertex, searchTree);
64             return searchTree;
65         }
66     }
