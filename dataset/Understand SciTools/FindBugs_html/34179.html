20     package edu.umd.cs.findbugs;
21     
22     import javax.annotation.Nonnull;
23     
24     import edu.umd.cs.findbugs.util.HTML;
25     
26     /**
27      * A BugPattern object collects all of the metadata for a particular species of
28      * BugInstance. Specifically, it stores the human-readable text for displaying a
29      * bug instance. BugPatterns derive from the BugPattern elements in the
30      * "findbugs.xml" and "messages.xml" found in a FindBugs plugin.
31      *
32      * @author David Hovemeyer
33      * @see BugInstance
34      */
35     public class BugPattern implements Comparable<BugPattern> {
36         final private String type;
37     
38         final private String abbrev;
39     
40         final private String category;
41     
42         final private boolean experimental;
43     
44         final private String shortDescription;
45     
46         final private String longDescription;
47     
48         final private String detailText;
49     
50         final private String url;
51     
52         final int cweid;
53     
54         int priorityAdjustment;
55     
56         private boolean deprecated;
57     
58         /**
59          * Constructor.
60          *
61          * @param type
62          *            the type (species) of BugInstance
63          * @param abbrev
64          *            the abbreviation or "bug code"; see {@link BugCode}
65          * @param category
66          *            the category
67          * @param experimental
68          *            true if the bug pattern is experimental
69          * @param shortDescription
70          *            short one-line description of the bug species
71          * @param longDescription
72          *            longer one-line description; may contain placeholders for use
73          *            by {@link FindBugsMessageFormat} to format BugAnnotations
74          * @param detailText
75          *            HTML text containing a full description of the bug species
76          * @param bugsUrl
77          *            URL of web-page containing bug descriptions or null if there's no such page.
78          */
79         public BugPattern(String type, String abbrev, String category, boolean experimental, String shortDescription,
80                 String longDescription, String detailText, String bugsUrl, int cweid) {
81     
82             this.type = type;
83             this.abbrev = abbrev.intern();
84             this.category = category.intern();
85             this.experimental = experimental;
86             this.shortDescription = shortDescription;
87             this.longDescription = longDescription;
88             this.detailText = detailText;
89             this.cweid = cweid;
90             this.url = bugsUrl;
91         }
92     
93         static final BugPattern REALLY_UNKNOWN = new BugPattern("REALLY_UNKNOWN", "TEST", "CORRECTNESS", false,
94                 "Unknown warning; core bug patterns not found", "Unknown warning BUG_PATTERN in {1}; core bug patterns not found",
95                 "<p>A warning was recorded, but findbugs can't find the description of this bug pattern "
96                         + "and so can't describe it. This should occur only in cases of a bug in FindBugs or its configuration, "
97                         + "or perhaps if an analysis was generated using a plugin, but that plugin is not currently loaded. "
98                         + "</p>", null, 0);
99     
100         /**
101          * Get the BugPattern
102          */
103         public String getType() {
104             return type;
105         }
106     
107         /**
108          * Get the BugKind
109          */
110         public String getAbbrev() {
111             return abbrev;
112         }
113     
114         /**
115          * Get the BugCode
116          */
117         public @Nonnull BugCode getBugCode() {
118             return DetectorFactoryCollection.instance().getBugCode(abbrev);
119     
120         }
121         /**
122          * Get the BugCategory
123          */
124         public String getCategory() {
125             return category;
126         }
127     
128         public int getPriorityAdjustment() {
129             return priorityAdjustment;
130         }
131     
132         public void adjustPriority(int change) {
133             priorityAdjustment += change;
134         }
135     
136         public String getCategoryAbbrev() {
137             String categoryAbbrev = null;
138             BugCategory bcat = DetectorFactoryCollection.instance().getBugCategory(getCategory());
139             if (bcat != null) {
140                 categoryAbbrev = bcat.getAbbrev();
141             }
142             if (categoryAbbrev == null) {
143                 categoryAbbrev = TextUIBugReporter.OTHER_CATEGORY_ABBREV;
144             }
145             return categoryAbbrev;
146         }
147     
148         /**
149          * Is the bug pattern experimental?
150          */
151         public boolean isExperimental() {
152             return experimental;
153         }
154     
155         /**
156          * Get the short description.
157          */
158         public String getShortDescription() {
159             return shortDescription;
160         }
161     
162         /**
163          * Get the long description.
164          */
165         public String getLongDescription() {
166             return longDescription;
167         }
168     
169         /**
170          * Get the HTML detail text describing the bug.
171          */
172         public String getDetailText() {
173             return detailText;
174         }
175     
176         /**
177          * Get the Plain text describing the bug.
178          */
179         public String getDetailPlainText() {
180             try {
181                 return HTML.convertHtmlSnippetToText(detailText);
182             } catch (Exception e) {
183                 return detailText;
184     
185             }
186         }
187     
188         /**
189          * Get the detail text as a complete HTML document.
190          */
191         public String getDetailHTML() {
192             return getDetailHTML(getDetailText());
193         }
194     
195         public String getDetailHTML(String detailText) {
196     
197             StringBuilder buf = new StringBuilder();
198             buf.append("<!DOCTYPE HTML\">\n");
199             buf.append("<HTML><HEAD><TITLE>");
200             buf.append(getShortDescription());
201             buf.append("</TITLE></HEAD><BODY><H1>");
202             buf.append(wrapInDescriptionLink(getShortDescription()));
203             buf.append("</H1>\n");
204             buf.append(detailText);
205             buf.append("</BODY></HTML>\n");
206             return buf.toString();
207     
208         }
209     
210         public String wrapInDescriptionLink(String text) {
211             if(url == null) {
212                 return text;
213             }
214             return "<a href=\"" + url + "#" + type + "\">" + text + "</a>";
215         }
216     
217         @Override
218         public int compareTo(BugPattern other) {
219             return type.compareTo(other.type);
220         }
221     
222         @Override
223         public int hashCode() {
224             return type.hashCode();
225         }
226     
227         @Override
228         public boolean equals(Object o) {
229             if (!(o instanceof BugPattern)) {
230                 return false;
231             }
232             BugPattern other = (BugPattern) o;
233             return type.equals(other.type);
234         }
235     
236         /**
237          * @return Returns the cweid.
238          */
239         public int getCWEid() {
240             return cweid;
241         }
242     
243         @Override
244         public String toString() {
245             return "BugPattern[" + type + "]";
246         }
247     
248         /**
249          * The method is not public to disallow modification of this attribute
250          *
251          * @param deprecated
252          *            true if this bug pattern is not used anymore and exists for
253          *            backward compatibility only
254          */
255         /* package protected */void setDeprecated(boolean deprecated) {
256             this.deprecated = deprecated;
257         }
258     
259         /**
260          * @return true if this bug pattern is not used anymore and exists for
261          *         backward compatibility only
262          */
263         public boolean isDeprecated() {
264             return deprecated;
265         }
266     }
