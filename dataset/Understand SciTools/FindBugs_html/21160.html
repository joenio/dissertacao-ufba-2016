290          * file a ClassNotFoundException with the lookupFailureCallback

292          * @see #getLookupFailureCallback()

294         static public void reportMissingClass(ClassNotFoundException e) {
295             requireNonNull(e, "argument is null");
296             String missing = AbstractBugReporter.getMissingClassName(e);
297             if (skipReportingMissingClass(missing)) {
298                 return;
299             }
300             if (!analyzingApplicationClass()) {
301                 return;
302             }
303     
304             RepositoryLookupFailureCallback lookupFailureCallback = getCurrentLookupFailureCallback();
305             if (lookupFailureCallback != null) {
306                 lookupFailureCallback.reportMissingClass(e);
307             }
308         }

315         static public boolean analyzingApplicationClass() {
316             AnalysisContext context = AnalysisContext.currentAnalysisContext();
317             if (context == null) {
318                 return false;
319             }
320             ClassDescriptor clazz = context.getClassBeingAnalyzed();
321             if (clazz == null) {
322                 return false;
323             }
324             return context.isApplicationClass(clazz);
325         }

332         static public void reportMissingClass(ClassDescriptor c) {
333             requireNonNull(c, "argument is null");
334             if (!analyzingApplicationClass()) {
335                 return;
336             }
337             String missing = c.getDottedClassName();
338             if (missing.length() == 1) {
339                 System.out.println(c);
340             }
341             if (skipReportingMissingClass(missing)) {
342                 return;
343             }
344             RepositoryLookupFailureCallback lookupFailureCallback = getCurrentLookupFailureCallback();
345             if (lookupFailureCallback != null) {
346                 lookupFailureCallback.reportMissingClass(c);
347             }
348         }

350         /**
351          * Report an error
352          */
353         static public void logError(String msg, Exception e) {
354             AnalysisContext currentAnalysisContext2 = currentAnalysisContext();
355             if (currentAnalysisContext2 == null) {
356                 if (e instanceof NoSuchBugPattern) {
357                     return;
358                 }
359                 /*
360                 if (false && SystemProperties.ASSERTIONS_ENABLED) {
361                     AssertionError e2 = new AssertionError("Exception logged with no analysis context");
362                     e2.initCause(e);
363                     throw e2;
364                 }
365                  */
366                 e.printStackTrace(System.err);
367                 return;
368             }
369             if (e instanceof MissingClassException) {
370                 reportMissingClass(((MissingClassException) e).getClassNotFoundException());
371                 return;
372             }
373             if (e instanceof edu.umd.cs.findbugs.classfile.MissingClassException) {
374                 reportMissingClass(((edu.umd.cs.findbugs.classfile.MissingClassException) e).toClassNotFoundException());
375                 return;
376             }
377     
378             RepositoryLookupFailureCallback lookupFailureCallback = currentAnalysisContext2.getLookupFailureCallback();
379             if (lookupFailureCallback != null) {
380                 lookupFailureCallback.logError(msg, e);
381             }
382         }

479         public boolean isTooBig(ClassDescriptor desc) {
480             IAnalysisCache analysisCache = Global.getAnalysisCache();
481     
482             try {
483                 ClassContext classContext = analysisCache.getClassAnalysis(ClassContext.class, desc);
484                 ClassData classData = analysisCache.getClassAnalysis(ClassData.class, desc);
485                 if (classData.getData().length > 1000000) {
486                     return true;
487                 }
488                 try {
489                     JavaClass javaClass = classContext.getJavaClass();
490                     if (javaClass.getMethods().length > 1000) {
491                         return true;
492                     }
493                 } catch (RuntimeException e) {
494                     AnalysisContext.logError("Error parsing class " + desc
495                             + " from " + classData.getCodeBaseEntry().getCodeBase(), e);
496                     return true;
497                 }
498     
499             } catch (RuntimeException e) {
500                 AnalysisContext.logError("Error getting class data for " + desc, e);
501                 return true;
502             } catch (CheckedAnalysisException e) {
503                 AnalysisContext.logError("Could not get class context for "  + desc, e);
504                 return true;
505             }
506             return false;
507         }

523         /**
524          * This is equivalent to Repository.lookupClass() or this.lookupClass(),
525          * except it uses the original Repository instead of the current one.
526          *
527          * This can be important because URLClassPathRepository objects are closed
528          * after an analysis, so JavaClass objects obtained from them are no good on
529          * subsequent runs.
530          *
531          * @param className
532          *            the name of the class
533          * @return the JavaClass representing the class
534          * @throws ClassNotFoundException
535          */
536         public static JavaClass lookupSystemClass(@Nonnull String className) throws ClassNotFoundException {
537             // TODO: eventually we should move to our own thread-safe repository
538             // implementation
539             requireNonNull (className, "className is null");
540             if (originalRepository == null) {
541                 throw new IllegalStateException("originalRepository is null");
542             }
543     
544             JavaClass clazz = originalRepository.findClass(className);
545             return (clazz == null ? originalRepository.loadClass(className) : clazz);
546         }

548         /**
549          * Lookup a class's source file
550          *
551          * @param dottedClassName
552          *            the name of the class
553          * @return the source file for the class, or
554          *         {@link SourceLineAnnotation#UNKNOWN_SOURCE_FILE} if unable to
555          *         determine
556          */
557         public final String lookupSourceFile(@Nonnull @DottedClassName String dottedClassName) {
558             requireNonNull(dottedClassName, "className is null");
559             try {
560                 XClass xClass = Global.getAnalysisCache().getClassAnalysis(XClass.class,
561                         DescriptorFactory.createClassDescriptorFromDottedClassName(dottedClassName));
562                 String name = xClass.getSource();
563                 if (name == null) {
564                     return SourceLineAnnotation.UNKNOWN_SOURCE_FILE;
565                 }
566                 return name;
567             } catch (CheckedAnalysisException e) {
568                 return SourceLineAnnotation.UNKNOWN_SOURCE_FILE;
569             }
570         }

665         /**
666          * Load an interprocedural property database.
667          *
668          * @param <DatabaseType>
669          *            actual type of the database
670          * @param <KeyType>
671          *            type of key (e.g., method or field)
672          * @param <Property>
673          *            type of properties stored in the database
674          * @param database
675          *            the empty database object
676          * @param fileName
677          *            file to load database from
678          * @param description
679          *            description of the database (for diagnostics)
680          * @return the database object, or null if the database couldn't be loaded
681          */
682         public <DatabaseType extends PropertyDatabase<KeyType, Property>, KeyType extends FieldOrMethodDescriptor, Property> DatabaseType loadPropertyDatabase(
683                 DatabaseType database, String fileName, String description) {
684             try {
685                 File dbFile = new File(getDatabaseInputDir(), fileName);
686                 if (DEBUG) {
687                     System.out.println("Loading " + description + " from " + dbFile.getPath() + "...");
688                 }
689     
690                 database.readFromFile(dbFile.getPath());
691                 return database;
692             } catch (IOException e) {
693                 getLookupFailureCallback().logError("Error loading " + description, e);
694             } catch (PropertyDatabaseFormatException e) {
695                 getLookupFailureCallback().logError("Invalid " + description, e);
696             }
697             return null;
698         }

709          * @param database
710          *            the empty database object
711          * @param resourceName
712          *            name of resource to load the database from
713          * @param description
714          *            description of the database (for diagnostics)
715          * @return the database object, or null if the database couldn't be loaded
716          */
717         public <DatabaseType extends PropertyDatabase<KeyType, Property>, KeyType extends FieldOrMethodDescriptor, Property> DatabaseType loadPropertyDatabaseFromResource(
718                 DatabaseType database, String resourceName, String description) {
719             try {
720                 if (DEBUG) {
721                     System.out.println("Loading default " + description + " from " + resourceName + " @ "
722                             + database.getClass().getResource(resourceName) + " ... ");
723                 }
724                 try(InputStream in = database.getClass().getResourceAsStream(resourceName)){
725                     if (in == null) {
726                         AnalysisContext.logError("Unable to load " + description + " from resource " + resourceName);
727                     } else {
728                         database.read(in);
729                     }
730                 }
731                 return database;
732             } catch (IOException e) {
733                 getLookupFailureCallback().logError("Error loading " + description, e);
734             } catch (PropertyDatabaseFormatException e) {
735                 getLookupFailureCallback().logError("Invalid " + description, e);
736             }
737             return null;
738         }

929         /**
930          * Lookup a class.
931          * <em>Use this method instead of Repository.lookupClass().</em>
932          *
933          * @param className
934          *            the name of the class
935          * @return the JavaClass representing the class
936          * @throws ClassNotFoundException
937          *             (but not really)
938          */
939         public JavaClass lookupClass(@Nonnull @DottedClassName String className) throws ClassNotFoundException {
940             try {
941                 if (className.length() == 0) {
942                     throw new IllegalArgumentException("Class name is empty");
943                 }
944                 if (!ClassName.isValidClassName(className)) {
945                     throw new ClassNotFoundException("Invalid class name: " + className);
946                 }
947                 return Global.getAnalysisCache().getClassAnalysis(JavaClass.class,
948                         DescriptorFactory.instance().getClassDescriptor(ClassName.toSlashedClassName(className)));
949             } catch (CheckedAnalysisException e) {
950                 throw new ClassNotFoundException("Class not found: " + className, e);
951             }
952         }

958         public void setAppClassList(List<ClassDescriptor> appClassCollection) {
959             // FIXME: we really should drive the progress callback here
960             HashSet<ClassDescriptor> appSet = new HashSet<ClassDescriptor>(appClassCollection);
961     
962             Collection<ClassDescriptor> allClassDescriptors = new ArrayList<ClassDescriptor>(DescriptorFactory.instance()
963                     .getAllClassDescriptors());
964             for (ClassDescriptor appClass : allClassDescriptors) {
965                 try {
966                     XClass xclass = currentXFactory().getXClass(appClass);
967     
968                     if (xclass == null) {
969                         continue;
970                     }
971     
972                     // Add the application class to the database
973                     if (appSet.contains(appClass)) {
974                         getSubtypes2().addApplicationClass(xclass);
975                     } else if (xclass instanceof ClassInfo) {
976                         getSubtypes2().addClass(xclass);
977                     }
978     
979                 } catch (Exception e) {
980                     AnalysisContext.logError("Unable to get XClass for " + appClass, e);
981                 }
982             }
983     
984             if (true && Subtypes2.DEBUG) {
985                 System.out.println(getSubtypes2().getGraph().getNumVertices() + " vertices in inheritance graph");
986             }
987         }
