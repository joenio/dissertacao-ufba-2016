213         public void acquireDisplayWait() {
214             synchronized (waitLock) {
215                 waitCount++;
216                 if (GUI2_DEBUG) {
217                     System.err.println("acquiring display wait, count " + waitCount);
218                     Thread.dumpStack();
219                 }
220                 if (waitCount == 1) {
221                     mainFrameTree.showCard(BugCard.WAITCARD, new Cursor(Cursor.WAIT_CURSOR), this);
222                 }
223             }
224         }
225     
226         volatile Exception previousDecrementToZero;
227     
228         public void releaseDisplayWait() {
229             synchronized (waitLock) {
230                 if (waitCount <= 0) {
231                     if (previousDecrementToZero != null) {
232                         throw new IllegalStateException("Can't decrease wait count; already zero", previousDecrementToZero);
233                     } else {
234                         throw new IllegalStateException("Can't decrease wait count; never incremented");
235                     }
236                 }
237                 waitCount--;
238                 if (GUI2_DEBUG) {
239                     System.err.println("releasing display wait, count " + waitCount);
240                     Thread.dumpStack();
241                 }
242                 if (waitCount == 0) {
243                     mainFrameTree.showCard(BugCard.TREECARD, new Cursor(Cursor.DEFAULT_CURSOR), this);
244                     previousDecrementToZero = new Exception("Previously decremented at");
245                 }
246             }
247         }

343         @SwingThread
344         void updateStatusBar() {
345             int countFilteredBugs = BugSet.countFilteredBugs();
346             String msg = "";
347             if (countFilteredBugs == 1) {
348                 msg = "  1 " + edu.umd.cs.findbugs.L10N.getLocalString("statusbar.bug_hidden", "bug hidden (see view menu)");
349             } else if (countFilteredBugs > 1) {
350                 msg = "  " + countFilteredBugs + " "
351                         + edu.umd.cs.findbugs.L10N.getLocalString("statusbar.bugs_hidden", "bugs hidden (see view menu)");
352             }
353             msg = updateCloudSigninStatus(msg);
354             if (errorMsg != null && errorMsg.length() > 0) {
355                 msg = join(msg, errorMsg);
356             }
357     
358             mainFrameTree.setWaitStatusLabelText(msg); // should not be the URL
359             if (msg.length() == 0) {
360                 msg = "http://findbugs.sourceforge.net";
361             }
362             statusBarLabel.setText(msg);
363         }

376         /**
377          * This method is called when the application is closing. This is either by
378          * the exit menuItem or by clicking on the window's system menu.
379          */
380         void callOnClose() {
381             if (!canNavigateAway()) {
382                 return;
383             }
384     
385             if (projectChanged && !SystemProperties.getBoolean("findbugs.skipSaveChangesWarning")) {
386                 Object[] options = {
387                         L10N.getLocalString("dlg.save_btn", "Save"),
388                         L10N.getLocalString("dlg.dontsave_btn", "Don't Save"),
389                         L10N.getLocalString("dlg.cancel_btn", "Cancel"),
390                 };
391                 int value = JOptionPane.showOptionDialog(this, getActionWithoutSavingMsg("closing"),
392                         edu.umd.cs.findbugs.L10N.getLocalString("msg.confirm_save_txt", "Do you want to save?"),
393                         JOptionPane.YES_NO_CANCEL_OPTION, JOptionPane.QUESTION_MESSAGE, null, options, options[0]);
394     
395                 if (value == 2 || value == JOptionPane.CLOSED_OPTION) {
396                     return;
397                 } else if (value == 0) {
398     
399                     if (saveFile == null) {
400                         if (!mainFrameLoadSaveHelper.saveAs()) {
401                             return;
402                         }
403                     } else {
404                         mainFrameLoadSaveHelper.save();
405                     }
406                 }
407             }
408     
409             GUISaveState guiSaveState = GUISaveState.getInstance();
410             guiLayout.saveState();
411             guiSaveState.setFrameBounds(getBounds());
412             guiSaveState.setExtendedWindowState(getExtendedState());
413             guiSaveState.save();
414             if (this.bugCollection != null) {
415                 Cloud cloud = this.bugCollection.getCloud();
416                 cloud.shutdown();
417             }
418             System.exit(0);
419         }

475         @SwingThread
476         private void setProjectAndBugCollection(@CheckForNull Project project, @CheckForNull BugCollection bugCollection) {
477             if (GUI2_DEBUG) {
478                 if (bugCollection == null) {
479                     System.out.println("Setting bug collection to null");
480                 } else {
481                     System.out.println("Setting bug collection; contains " + bugCollection.getCollection().size() + " bugs");
482                 }
483     
484             }
485             if (bugCollection != null && bugCollection.getProject() != project) {
486                 Project p2 = bugCollection.getProject();
487                 throw new IllegalArgumentException(String.format("project %x and bug collection %x don't match",
488                         System.identityHashCode(project), System.identityHashCode(p2)));
489             }
490             acquireDisplayWait();
491             try {
492     
493                 if (this.bugCollection != bugCollection && this.bugCollection != null) {
494                     Cloud plugin = this.bugCollection.getCloud();
495                     plugin.removeListener(userAnnotationListener);
496                     plugin.removeStatusListener(cloudStatusListener);
497                     plugin.shutdown();
498                 }
499                 // setRebuilding(false);
500                 setProject(project);
501                 this.bugCollection = bugCollection;
502                 BugLoader.addDeadBugMatcher(bugCollection);
503     
504                 comments.updateBugCollection();
505                 displayer.clearCache();
506                 if (bugCollection != null) {
507                     Cloud plugin = bugCollection.getCloud();
508                     plugin.addListener(userAnnotationListener);
509                     plugin.addStatusListener(cloudStatusListener);
510                 }
511                 mainFrameTree.updateBugTree();
512                 setProjectChanged(false);
513                 Runnable runnable = new Runnable() {
514                     @Override
515                     public void run() {
516                         PreferencesFrame.getInstance().updateFilterPanel();
517                         mainFrameMenu.getReconfigMenuItem().setEnabled(true);
518                         comments.refresh();
519                         mainFrameMenu.setViewMenu();
520                         newProject();
521                         clearSourcePane();
522                         clearSummaryTab();
523     
524                         /*
525                          * This is here due to a threading issue. It can only be
526                          * called after curProject has been changed. Since this
527                          * method is called by both open methods it is put here.
528                          */
529                         updateTitle();
530                     }
531                 };
532                 if (SwingUtilities.isEventDispatchThread()) {
533                     runnable.run();
534                 } else {
535                     SwingUtilities.invokeLater(runnable);
536                 }
537             } finally {
538                 releaseDisplayWait();
539             }
540         }

652         void syncBugInformation() {
653             boolean prevProjectChanged = projectChanged;
654             if (mainFrameTree.getCurrentSelectedBugLeaf() != null) {
655                 BugInstance bug = mainFrameTree.getCurrentSelectedBugLeaf().getBug();
656                 displayer.displaySource(bug, bug.getPrimarySourceLineAnnotation());
657                 updateDesignationDisplay();
658                 comments.updateCommentsFromLeafInformation(mainFrameTree.getCurrentSelectedBugLeaf());
659                 updateSummaryTab(mainFrameTree.getCurrentSelectedBugLeaf());
660             } else if (currentSelectedBugAspects != null) {
661                 updateDesignationDisplay();
662                 comments.updateCommentsFromNonLeafInformation(currentSelectedBugAspects);
663                 displayer.displaySource(null, null);
664                 clearSummaryTab();
665             } else {
666                 displayer.displaySource(null, null);
667                 clearSummaryTab();
668             }
669             setProjectChanged(prevProjectChanged);
670         }

746         public void updateTitle() {
747             Project project = getProject();
748             String name = project.getProjectName();
749             if ((name == null || "".equals(name.trim())) && saveFile != null) {
750                 name = saveFile.getAbsolutePath();
751             }
752             if (name == null) {
753                 name = "";//Project.UNNAMED_PROJECT;
754             }
755             String oldTitle = this.getTitle();
756             String newTitle = TITLE_START_TXT + ("".equals(name.trim()) ? "" : " - " + name);
757             if (oldTitle.equals(newTitle)) {
758                 return;
759             }
760             this.setTitle(newTitle);
761         }

773         public void selectPackagePrefixByProject() {
774             TreeSet<String> projects = new TreeSet<String>(String.CASE_INSENSITIVE_ORDER);
775             Multiset<String> count = new Multiset<String>();
776             int total = 0;
777             for (BugInstance b : getBugCollection().getCollection()) {
778                 if (shouldDisplayIssueIgnoringPackagePrefixes(b)) {
779                     TreeSet<String> projectsForThisBug = projectPackagePrefixes.getProjects(b.getPrimaryClass().getClassName());
780                     projects.addAll(projectsForThisBug);
781                     count.addAll(projectsForThisBug);
782                     total++;
783                 }
784             }
785             if (projects.size() == 0) {
786                 JOptionPane.showMessageDialog(this, "No issues in current view");
787                 return;
788             }
789             ArrayList<ProjectSelector> selectors = new ArrayList<ProjectSelector>(projects.size() + 1);
790             ProjectSelector everything = new ProjectSelector("all projects", "", total);
791             selectors.add(everything);
792             for (String projectName : projects) {
793                 ProjectPackagePrefixes.PrefixFilter filter = projectPackagePrefixes.getFilter(projectName);
794                 selectors.add(new ProjectSelector(projectName, filter.toString(), count.getCount(projectName)));
795             }
796             ProjectSelector choice = (ProjectSelector) JOptionPane.showInputDialog(null,
797                     "Choose a project to set appropriate package prefix(es)", "Select package prefixes by package",
798                     JOptionPane.QUESTION_MESSAGE, null, selectors.toArray(), everything);
799             if (choice == null) {
800                 return;
801             }
802     
803             mainFrameTree.setFieldForPackagesToDisplayText(choice.filter);
804             viewFilter.setPackagesToDisplay(choice.filter);
805             resetViewCache();
806     
807         }

809         private static String join(String s1, String s2) {
810             if (s1 == null || s1.length() == 0) {
811                 return s2;
812             }
813             if (s2 == null || s2.length() == 0) {
814                 return s1;
815             }
816             return s1 + "; " + s2;
817         }

863             int targetLineNum = -1;
864             String targetString = sourceSearchTextField.getText();
865             switch (type) {
866             case 0:
867                 targetLineNum = displayer.find(targetString);
868                 break;
869             case 1:

876                 break;
877             }
878             if (targetLineNum != -1) {
879                 displayer.foundItem(targetLineNum);

1100         private class ShownBugsIterator implements Iterator<BugInstance> {
1101             Iterator<BugInstance> base = getBugCollection().getCollection().iterator();
1102     
1103             boolean nextKnown;
1104     
1105             BugInstance next;
1106     
1107             @Override
1108             public boolean hasNext() {
1109                 if (!nextKnown) {
1110                     nextKnown = true;
1111                     while (base.hasNext()) {
1112                         next = base.next();
1113                         if (shouldDisplayIssue(next)) {
1114                             return true;
1115                         }
1116                     }
1117                     next = null;
1118                     return false;
1119                 }
1120                 return next != null;
1121             }
1122     
1123             @Override
1124             public BugInstance next() {
1125                 if (!hasNext()) {
1126                     throw new NoSuchElementException();
1127                 }
1128                 BugInstance result = next;
1129                 next = null;
1130                 nextKnown = false;
1131                 return result;
1132             }
1133     
1134             @Override
1135             public void remove() {
1136                 throw new UnsupportedOperationException();
1137             }
1138         }
