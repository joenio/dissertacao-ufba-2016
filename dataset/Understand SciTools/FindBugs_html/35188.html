22

24     
25     import edu.umd.cs.findbugs.BugInstance;
26     import edu.umd.cs.findbugs.BugReporter;
27     import edu.umd.cs.findbugs.IntAnnotation;
28     import edu.umd.cs.findbugs.OpcodeStack;
29     import edu.umd.cs.findbugs.StringAnnotation;
30     import edu.umd.cs.findbugs.TypeAnnotation;
31     import edu.umd.cs.findbugs.ba.AnalysisContext;
32     import edu.umd.cs.findbugs.ba.Hierarchy;
33     import edu.umd.cs.findbugs.ba.XMethod;
34     import edu.umd.cs.findbugs.bcel.OpcodeStackDetector;
35     import edu.umd.cs.findbugs.classfile.ClassDescriptor;
36     import edu.umd.cs.findbugs.classfile.DescriptorFactory;
37     import edu.umd.cs.findbugs.formatStringChecker.ExtraFormatArgumentsException;
38     import edu.umd.cs.findbugs.formatStringChecker.Formatter;
39     import edu.umd.cs.findbugs.formatStringChecker.FormatterNumberFormatException;
40     import edu.umd.cs.findbugs.formatStringChecker.IllegalFormatConversionException;
41     import edu.umd.cs.findbugs.formatStringChecker.MissingFormatArgumentException;
42     
43     public class FormatStringChecker extends OpcodeStackDetector {
44     
45         final BugReporter bugReporter;
46     
47         public FormatStringChecker(BugReporter bugReporter) {
48             this.bugReporter = bugReporter;
49         }
50     
51         enum FormatState {
52             NONE, READY_FOR_FORMAT, EXPECTING_ASSIGNMENT
53         }
54     
55         FormatState state;
56     
57         String formatString;
58     
59         int stackDepth;
60     
61         OpcodeStack.Item arguments[];
62     
63         @Override
64         public void visit(Code code) {
65             state = FormatState.NONE;
66             super.visit(code); // make callbacks to sawOpcode for all opcodes
67             arguments = null;
68         }
69     
70         /*
71          * (non-Javadoc)
72          *
73          * @see edu.umd.cs.findbugs.bcel.OpcodeStackDetector#sawOpcode(int)
74          */
75         @Override
76         public void sawOpcode(int seen) {
77             // System.out.println(getPC() + " " + OPCODE_NAMES[seen] + " " + state);
78     
79             if (stack.getStackDepth() < stackDepth) {
80                 state = FormatState.NONE;
81                 stackDepth = 0;
82                 arguments = null;
83             }
84             if (seen == ANEWARRAY && stack.getStackDepth() >= 2) {
85                 Object size = stack.getStackItem(0).getConstant();
86                 Object formatStr = stack.getStackItem(1).getConstant();
87                 if (size instanceof Integer && formatStr instanceof String) {
88                     arguments = new OpcodeStack.Item[(Integer) size];
89                     this.formatString = (String) formatStr;
90                     state = FormatState.READY_FOR_FORMAT;
91                     stackDepth = stack.getStackDepth();
92                 }
93             } else if (state == FormatState.READY_FOR_FORMAT && seen == DUP) {
94                 state = FormatState.EXPECTING_ASSIGNMENT;
95             } else if (state == FormatState.EXPECTING_ASSIGNMENT && stack.getStackDepth() == stackDepth + 3 && seen == AASTORE) {
96                 Object pos = stack.getStackItem(1).getConstant();
97                 OpcodeStack.Item value = stack.getStackItem(0);
98                 if (pos instanceof Integer) {
99                     int index = (Integer) pos;
100                     if (index >= 0 && index < arguments.length) {
101                         arguments[index] = value;
102                         state = FormatState.READY_FOR_FORMAT;
103                     } else {
104                         state = FormatState.NONE;
105                     }
106                 } else {
107                     state = FormatState.NONE;
108                 }
109             } else if (state == FormatState.READY_FOR_FORMAT
110                     && (seen == INVOKESPECIAL || seen == INVOKEVIRTUAL || seen == INVOKESTATIC || seen == INVOKEINTERFACE)
111                     && stack.getStackDepth() == stackDepth) {
112     
113                 String cl = getClassConstantOperand();
114                 String nm = getNameConstantOperand();
115                 String sig = getSigConstantOperand();
116                 XMethod m = getXMethodOperand();
117                 if ((m == null || m.isVarArgs())
118                         && sig.indexOf("Ljava/lang/String;[Ljava/lang/Object;)") >= 0
119                         && ("java/util/Formatter".equals(cl) && "format".equals(nm) || "java/lang/String".equals(cl)
120                                 && "format".equals(nm) || "java/io/PrintStream".equals(cl) && "format".equals(nm)
121                                 || "java/io/PrintStream".equals(cl) && "printf".equals(nm) || cl.endsWith("Writer")
122                                 && "format".equals(nm) || cl.endsWith("Writer") && "printf".equals(nm)) || cl.endsWith("Logger")
123                                 && nm.endsWith("fmt")) {
124     
125                     if (formatString.indexOf('\n') >= 0) {
126                         bugReporter.reportBug(new BugInstance(this, "VA_FORMAT_STRING_USES_NEWLINE", NORMAL_PRIORITY)
127                         .addClassAndMethod(this).addCalledMethod(this).addString(formatString)
128                         .describe(StringAnnotation.FORMAT_STRING_ROLE).addSourceLine(this));
129                     }
130                     try {
131                         String[] signatures = new String[arguments.length];
132                         for (int i = 0; i < signatures.length; i++) {
133                             signatures[i] = arguments[i].getSignature();
134                         }
135                         Formatter.check(formatString, signatures);
136     
137                     } catch (IllegalFormatConversionException e) {
138     
139                         if (e.getConversion() == 'b') {
140                             bugReporter.reportBug(new BugInstance(this, "VA_FORMAT_STRING_BAD_CONVERSION_TO_BOOLEAN", HIGH_PRIORITY)
141                             .addClassAndMethod(this).addCalledMethod(this).addType(e.getArgumentSignature())
142                             .describe(TypeAnnotation.FOUND_ROLE).addString(e.getFormatSpecifier())
143                             .describe(StringAnnotation.FORMAT_SPECIFIER_ROLE).addString(formatString)
144                             .describe(StringAnnotation.FORMAT_STRING_ROLE)
145                             .addValueSource(arguments[e.getArgIndex()], getMethod(), getPC()).addSourceLine(this));
146                         } else if (e.getArgumentSignature().charAt(0) == '[' && e.getConversion() == 's') {
147                             bugReporter.reportBug(new BugInstance(this, "VA_FORMAT_STRING_BAD_CONVERSION_FROM_ARRAY", HIGH_PRIORITY)
148                             .addClassAndMethod(this).addCalledMethod(this).addType(e.getArgumentSignature())
149                             .describe(TypeAnnotation.FOUND_ROLE).addString(e.getFormatSpecifier())
150                             .describe(StringAnnotation.FORMAT_SPECIFIER_ROLE).addString(formatString)
151                             .describe(StringAnnotation.FORMAT_STRING_ROLE)
152                             .addValueSource(arguments[e.getArgIndex()], getMethod(), getPC()).addSourceLine(this));
153                         } else {
154                             String aSig = e.getArgumentSignature();
155                             char conversion = e.getConversion();
156                             if ((conversion == 't' || conversion == 'T') && aSig.charAt(0) == 'L') {
157                                 ClassDescriptor argDescriptor = DescriptorFactory.createClassDescriptorFromFieldSignature(aSig);
158                                 assert argDescriptor != null : "sig started with L, should get descriptor";
159                                 String arg = argDescriptor.toDottedClassName();
160                                 try {
161                                     if (arg.equals("java.time.LocalDate")
162                                             || Hierarchy.isSubtype(arg,  java.util.Date.class.getName())
163                                             || Hierarchy.isSubtype(arg,  java.util.Calendar.class.getName())
164                                             || Hierarchy.isSubtype(arg,  "java.time.temporal.TemporalAccessor")) {
165                                         return;
166                                     }
167                                 } catch (ClassNotFoundException e1) {
168                                     AnalysisContext.reportMissingClass(e1);
169                                 }
170     
171                             }
172                             bugReporter.reportBug(new BugInstance(this, "VA_FORMAT_STRING_BAD_CONVERSION", HIGH_PRIORITY)
173                             .addClassAndMethod(this).addCalledMethod(this).addType(aSig)
174                             .describe(TypeAnnotation.FOUND_ROLE).addString(e.getFormatSpecifier())
175                             .describe(StringAnnotation.FORMAT_SPECIFIER_ROLE).addString(formatString)
176                             .describe(StringAnnotation.FORMAT_STRING_ROLE)
177                             .addValueSource(arguments[e.getArgIndex()], getMethod(), getPC()).addSourceLine(this));
178                         }
179                     } catch (IllegalArgumentException e) {
180                         bugReporter.reportBug(new BugInstance(this, "VA_FORMAT_STRING_ILLEGAL", HIGH_PRIORITY)
181                         .addClassAndMethod(this).addCalledMethod(this).addString(formatString)
182                         .describe(StringAnnotation.FORMAT_STRING_ROLE).addSourceLine(this));
183                     } catch (MissingFormatArgumentException e) {
184     
185                         if (e.pos < 0) {
186                             bugReporter.reportBug(new BugInstance(this, "VA_FORMAT_STRING_NO_PREVIOUS_ARGUMENT", HIGH_PRIORITY)
187                             .addClassAndMethod(this).addCalledMethod(this).addString(e.formatSpecifier)
188                             .describe(StringAnnotation.FORMAT_SPECIFIER_ROLE).addString(formatString)
189                             .describe(StringAnnotation.FORMAT_STRING_ROLE).addSourceLine(this));
190                         } else {
191                             bugReporter.reportBug(new BugInstance(this, "VA_FORMAT_STRING_MISSING_ARGUMENT", HIGH_PRIORITY)
192                             .addClassAndMethod(this).addCalledMethod(this).addString(e.formatSpecifier)
193                             .describe(StringAnnotation.FORMAT_SPECIFIER_ROLE).addString(formatString)
194                             .describe(StringAnnotation.FORMAT_STRING_ROLE).addInt(e.pos + 1)
195                             .describe(IntAnnotation.INT_EXPECTED_ARGUMENTS).addInt(arguments.length)
196                             .describe(IntAnnotation.INT_ACTUAL_ARGUMENTS).addSourceLine(this));
197                         }
198     
199                     } catch (ExtraFormatArgumentsException e) {
200                         int priority = NORMAL_PRIORITY;
201                         String pattern = "VA_FORMAT_STRING_EXTRA_ARGUMENTS_PASSED";
202                         if (e.used == 0) {
203                             priority = HIGH_PRIORITY;
204                             if (formatString.indexOf("{0") >= 0 || formatString.indexOf("{1") >= 0) {
205                                 pattern = "VA_FORMAT_STRING_EXPECTED_MESSAGE_FORMAT_SUPPLIED";
206                             }
207                         }
208     
209                         bugReporter.reportBug(new BugInstance(this, pattern, priority).addClassAndMethod(this).addCalledMethod(this)
210                                 .addString(formatString).describe(StringAnnotation.FORMAT_STRING_ROLE).addInt(e.used)
211                                 .describe(IntAnnotation.INT_EXPECTED_ARGUMENTS).addInt(e.provided)
212                                 .describe(IntAnnotation.INT_ACTUAL_ARGUMENTS).addSourceLine(this));
213                     } catch (FormatterNumberFormatException e) {
214                         bugReporter.reportBug(new BugInstance(this, "VA_FORMAT_STRING_ILLEGAL", NORMAL_PRIORITY)
215                         .addClassAndMethod(this).addCalledMethod(this).addString(formatString)
216                         .describe(StringAnnotation.FORMAT_STRING_ROLE)
217                         .addString("Can't use " + e.getTxt() + " for " + e.getKind()).describe(StringAnnotation.STRING_MESSAGE)
218                         .addSourceLine(this));
219                     }
220                 }
221     
222             }
223         }
224     
225     }
