37     class NumberedParagraphView extends ParagraphView {
38         public final static int NUMBERS_WIDTH = (int) Driver.getFontSize() * 3 + 9;
39     
40         HighlightInformation highlight;
41     
42         public NumberedParagraphView(Element e, HighlightInformation highlight) {
43             super(e);
44             this.highlight = highlight;
45         }
46     
47         // protected void setInsets(short top, short left, short bottom,
48         // short right) {super.setInsets
49         // (top,(short)(left+NUMBERS_WIDTH),
50         // bottom,right);
51         // }
52         @Override
53         public void paint(Graphics g, Shape allocation) {
54             Rectangle r = (allocation instanceof Rectangle) ? (Rectangle) allocation : allocation.getBounds();
55     
56             Color oldColor = g.getColor();
57             Integer lineNumber = getLineNumber();
58             Color highlightColor = highlight.getHighlight(lineNumber);
59             if (highlightColor != null) {
60                 g.setColor(highlightColor);
61                 g.fillRect(r.x, r.y, r.width, r.height);
62                 g.setColor(oldColor);
63             }
64             // r.x += NUMBERS_WIDTH;
65     
66             super.paint(g, r);
67     
68             FontMetrics metrics = g.getFontMetrics();
69     
70             g.setColor(Color.GRAY);
71             String lineNumberString = lineNumber.toString();
72             int width = metrics.stringWidth(lineNumberString);
73             int numberX = r.x - width - 9 + NUMBERS_WIDTH;
74             int numberY = r.y + metrics.getAscent();
75     
76             g.drawString(lineNumberString, numberX, numberY);
77             g.setColor(oldColor);
78             // System.out.println("Drawing line for " + lineNumber + " @ " + numberX
79             // +"," + numberY);
80             // r.x -= NUMBERS_WIDTH;
81     
82         }

84         public int getPreviousLineCount0() {
85             int lineCount = 0;
86             View parent = this.getParent();
87             int count = parent.getViewCount();
88             for (int i = 0; i < count; i++) {
89                 if (parent.getView(i) == this) {
90                     break;
91                 } else {
92                     lineCount += parent.getView(i).getViewCount();
93                 }
94             }
95             return lineCount;
96         }
97     
98         static WeakHashMap<Element, Integer> elementLineNumberCache = new WeakHashMap<Element, Integer>();
99     
100         public Integer getLineNumber() {
101     
102             Element element = this.getElement();
103             Integer result = elementLineNumberCache.get(element);
104             if (result != null) {
105                 return result;
106             }
107             Element parent = element.getParentElement();
108             int count = parent.getElementCount();
109             for (int i = 0; i < count; i++) {
110                 elementLineNumberCache.put(parent.getElement(i), i + 1);
111             }
112             result = elementLineNumberCache.get(element);
113             if (result != null) {
114                 return result;
115             }
116             return -1;
117     
118         }
