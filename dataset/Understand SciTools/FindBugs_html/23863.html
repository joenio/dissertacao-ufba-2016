19     package edu.umd.cs.findbugs.props;
20     
21     /**
22      * Enum representing how a particular warning property is expected to affect its
23      * likelihood of being serious, benign, or a false positive.
24      *
25      * @author David Hovemeyer
26      */
27     public class PriorityAdjustment {
28         private final String value;
29     
30         private PriorityAdjustment(String value) {
31             this.value = value;
32         }
33     
34         @Override
35         public String toString() {
36             return value;
37         }
38     
39         /** No adjustment to the priority. */
40         public static final PriorityAdjustment NO_ADJUSTMENT = new PriorityAdjustment("NO_ADJUSTMENT");
41     
42         /** Raise the priority. */
43         public static final PriorityAdjustment RAISE_PRIORITY = new PriorityAdjustment("RAISE_PRIORITY");
44     
45         /** Raise the priority. */
46         public static final PriorityAdjustment RAISE_PRIORITY_TO_AT_LEAST_NORMAL = new PriorityAdjustment(
47                 "RAISE_PRIORITY_TO_AT_LEAST_NORMAL");
48     
49         /** lower the priority. */
50         public static final PriorityAdjustment LOWER_PRIORITY_TO_AT_MOST_NORMAL = new PriorityAdjustment(
51                 "LOWER_PRIORITY_TO_AT_MOST_NORMAL");
52     
53         /** Raise the priority. */
54         public static final PriorityAdjustment RAISE_PRIORITY_TO_HIGH = new PriorityAdjustment("RAISE_PRIORITY_TO_HIGH");
55     
56         /** Priority is at most low. */
57         public static final PriorityAdjustment AT_MOST_LOW = new PriorityAdjustment("AT_MOST_LOW");
58     
59         /** Priority is at most medium. */
60         public static final PriorityAdjustment AT_MOST_MEDIUM = new PriorityAdjustment("AT_MOST_MEDIUM");
61     
62         /** Pegged high */
63         public static final PriorityAdjustment PEGGED_HIGH = new PriorityAdjustment("PEGGED_HIGH");
64     
65         /** Lower the priority. */
66         public static final PriorityAdjustment LOWER_PRIORITY = new PriorityAdjustment("LOWER_PRIORITY");
67     
68         /** Lower the priority a little */
69         public static final PriorityAdjustment A_LITTLE_BIT_LOWER_PRIORITY = new PriorityAdjustment("A_LITTLE_BIT_LOWER_PRIORITY");
70     
71         /** Raise the priority a little */
72         public static final PriorityAdjustment A_LITTLE_BIT_HIGHER_PRIORITY = new PriorityAdjustment("A_LITTLE_BIT_HIGHER_PRIORITY");
73     
74         /** Warning is likely to be a false positive. */
75         public static final PriorityAdjustment FALSE_POSITIVE = new PriorityAdjustment("FALSE_POSITIVE");
76     }
