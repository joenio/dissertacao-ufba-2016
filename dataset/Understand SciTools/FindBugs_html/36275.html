81         private static class ValueInfo {
82             Location created;
83             String var;
84             int origValue;
85             boolean hasObjectOnlyCall;
86             boolean escaped;
87             boolean used;
88             boolean derivedEscaped;
89             public BitSet origValues;
90             public BitSet derivedValues = new BitSet();
91             Type type;
92     
93             public ValueInfo(int origValue, Location location, Type type) {
94                 this.created = location;
95                 this.origValue = origValue;
96                 this.type = type;
97             }
98     
99             @Override
100             public String toString() {
101                 return "[" + (escaped ? "E" : "-") + (hasObjectOnlyCall ? "O" : "-") + (used ? "U" : "-")
102                         + (derivedEscaped ? "D" : "-") + "] " + (var == null ? "" : var + " ") + type + " " + created;
103             }
104         }

106         private class UselessValuesContext {
107             ValueNumberAnalysis vna;
108             TypeAnalysis ta;
109             CFG cfg;
110             int count;
111             Map<Integer, ValueInfo> observedValues = new HashMap<>();
112             ConstantPoolGen cpg;
113             Map<Integer, Set<ValueInfo>> values;
114             ValueNumber thisValue;
115             ClassContext classContext;
116             Method method;
117     
118             UselessValuesContext(ClassContext classContext, Method method) throws CheckedAnalysisException {
119                 this.classContext = classContext;
120                 this.method = method;
121                 cfg = classContext.getCFG(method);
122                 cpg = cfg.getMethodGen().getConstantPool();
123                 ta = classContext.getTypeDataflow(method).getAnalysis();
124                 vna = classContext.getValueNumberDataflow(method).getAnalysis();
125             }
126     
127             void initObservedValues() throws DataflowAnalysisException {
128                 for(Iterator<Location> iterator = cfg.locationIterator(); iterator.hasNext(); ) {
129                     Location location = iterator.next();
130                     Instruction instruction = location.getHandle().getInstruction();
131                     if(instruction instanceof ANEWARRAY || instruction instanceof NEWARRAY || instruction instanceof MULTIANEWARRAY) {
132                         int number = vna.getFactAfterLocation(location).getTopValue().getNumber();
133                         TypeFrame typeFrame = ta.getFactAfterLocation(location);
134                         if(typeFrame.isValid()) {
135                             Type type = typeFrame.getTopValue();
136                             observedValues.put(number, new ValueInfo(number, location, type));
137                         }
138                     } else if(instruction instanceof INVOKESPECIAL) {
139                         InvokeInstruction inv = (InvokeInstruction) instruction;
140                         if (inv.getMethodName(cpg).equals("<init>")
141                                 && noSideEffectMethods.hasNoSideEffect(new MethodDescriptor(inv, cpg))) {
142                             int number = vna.getFactAtLocation(location).getStackValue(inv.consumeStack(cpg)-1).getNumber();
143                             TypeFrame typeFrame = ta.getFactAtLocation(location);
144                             if(typeFrame.isValid()) {
145                                 Type type = typeFrame.getStackValue(inv.consumeStack(cpg)-1);
146                                 observedValues.put(number, new ValueInfo(number, location, type));
147                             }
148                         }
149                     }
150                 }
151                 thisValue = vna.getThisValue();
152                 if(thisValue != null) {
153                     observedValues.remove(thisValue.getNumber());
154                 }
155                 count = observedValues.size();
156             }
157     
158             void enhanceViaMergeTree() {
159                 values = new HashMap<>();
160                 for (Entry<Integer, ValueInfo> entry : observedValues.entrySet()) {
161                     BitSet outputSet = vna.getMergeTree().getTransitiveOutputSet(entry.getKey());
162                     outputSet.set(entry.getKey());
163                     entry.getValue().origValues = outputSet;
164                     for (int i = outputSet.nextSetBit(0); i >= 0; i = outputSet.nextSetBit(i+1)) {
165                         Set<ValueInfo> list = values.get(i);
166                         if(list == null) {
167                             list = new HashSet<>();
168                             values.put(i, list);
169                         }
170                         list.add(entry.getValue());
171                     }
172                 }
173             }
174     
175             boolean setEscape(Set<ValueInfo> vals) {
176                 boolean result = false;
177                 for(ValueInfo vi : vals) {
178                     result |= !vi.escaped;
179                     vi.escaped = true;
180                     count--;
181                 }
182                 return result;
183             }
184     
185             boolean setDerivedEscape(Set<ValueInfo> vals, ValueNumber vn) {
186                 boolean result = false;
187                 for(ValueInfo vi : vals) {
188                     if(vi.origValues.get(vn.getNumber())) {
189                         result |= !vi.derivedEscaped;
190                         vi.derivedEscaped = true;
191                     }
192                 }
193                 return result;
194             }
195     
196             boolean setUsed(Set<ValueInfo> vals) {
197                 boolean result = false;
198                 for(ValueInfo vi : vals) {
199                     result |= !vi.used;
200                     vi.used = true;
201                 }
202                 return result;
203             }
204     
205             boolean setObjectOnly(Set<ValueInfo> vals, ValueNumber vn) {
206                 boolean result = false;
207                 for(ValueInfo vi : vals) {
208                     if(vi.origValues.get(vn.getNumber()) || (!vi.derivedEscaped && vi.derivedValues.get(vn.getNumber()))) {
209                         result |= !vi.hasObjectOnlyCall;
210                         vi.hasObjectOnlyCall = true;
211                     } else {
212                         result |= !vi.escaped;
213                         vi.escaped = true;
214                         count--;
215                     }
216                 }
217                 return result;
218             }
219     
220             boolean propagateValues(Set<ValueInfo> vals, ValueNumber origNumber, ValueNumber vn) {
221                 int number = vn.getNumber();
222                 if(vals.size() == 1 && vals.iterator().next().origValue == number) {
223                     return false;
224                 }
225                 boolean result = setUsed(vals);
226                 if(origNumber != null) {
227                     for(ValueInfo vi : vals) {
228                         if(vi.origValues.get(origNumber.getNumber()) && !vi.derivedValues.get(number)) {
229                             vi.derivedValues.set(number);
230                             result = true;
231                         }
232                     }
233                 }
234                 Set<ValueInfo> list = values.get(number);
235                 if(list == null) {
236                     list = new HashSet<>();
237                     values.put(number, list);
238                 }
239                 result |= list.addAll(vals);
240                 BitSet outputSet = vna.getMergeTree().getTransitiveOutputSet(number);
241                 for (int i = outputSet.nextSetBit(0); i >= 0; i = outputSet.nextSetBit(i+1)) {
242                     list = values.get(i);
243                     if(list == null) {
244                         list = new HashSet<>();
245                         values.put(i, list);
246                     }
247                     result |= list.addAll(vals);
248                 }
249                 return result;
250             }
251     
252             boolean propagateToReturnValue(Set<ValueInfo> vals, ValueNumber vn, GenLocation location, MethodDescriptor m)
253                     throws DataflowAnalysisException {
254                 for(ValueInfo vi : vals) {
255                     if(vi.type.getSignature().startsWith("[") && vi.hasObjectOnlyCall && vi.var == null && vn.getNumber() == vi.origValue) {
256                         // Ignore initialized arrays passed to methods
257                         vi.escaped = true;
258                         count--;
259                     }
260                 }
261                 if (Type.getReturnType(m.getSignature()) == Type.VOID || location instanceof ExceptionLocation) {
262                     return false;
263                 }
264                 InstructionHandle nextHandle = location.getHandle().getNext();
265                 if (nextHandle == null || (nextHandle.getInstruction() instanceof POP || nextHandle.getInstruction() instanceof POP2)) {
266                     return false;
267                 }
268                 return propagateValues(vals, null, location.frameAfter().getTopValue());
269             }
270     
271             boolean isEmpty() {
272                 return count == 0;
273             }
274     
275             Iterator<GenLocation> genIterator() {
276                 return new Iterator<FindUselessObjects.GenLocation>() {
277                     Iterator<Location> locIterator = cfg.locationIterator();
278                     Iterator<BasicBlock> blockIterator = cfg.blockIterator();
279                     GenLocation next = advance();
280     
281                     private GenLocation advance() {
282                         if(locIterator.hasNext()) {
283                             return new RegularLocation(ta, vna, locIterator.next());
284                         }
285                         while(blockIterator.hasNext()) {
286                             BasicBlock block = blockIterator.next();
287                             if(block.isExceptionThrower() && cfg.getOutgoingEdgeWithType(block, EdgeTypes.FALL_THROUGH_EDGE) == null) {
288                                 return new ExceptionLocation(ta, vna, block);
289                             }
290                         }
291                         return null;
292                     }
293     
294                     @Override
295                     public boolean hasNext() {
296                         return next != null;
297                     }
298     
299                     @Override
300                     public GenLocation next() {
301                         if (!hasNext()) {
302                             throw new NoSuchElementException();
303                         }
304                         GenLocation cur = next;
305                         next = advance();
306                         return cur;
307                     }
308     
309                     @Override
310                     public void remove() {
311                         throw new UnsupportedOperationException();
312                     }
313                 };
314             }
315     
316             boolean escaped(ValueNumber vn) {
317                 Set<ValueInfo> vals = values.get(vn.getNumber());
318                 if(vals == null) {
319                     return true;
320                 }
321                 for(ValueInfo vi : vals) {
322                     if(vi.escaped) {
323                         return true;
324                     }
325                 }
326                 return false;
327             }
328     
329             Set<ValueInfo> getLiveVals(ValueNumber vn) {
330                 Set<ValueInfo> vals = this.values.get(vn.getNumber());
331                 if(vals == null) {
332                     return null;
333                 }
334                 if(vals.size() == 1) {
335                     return vals.iterator().next().escaped ? null : vals;
336                 }
337                 Set<ValueInfo> result = new HashSet<>();
338                 for(ValueInfo vi : vals) {
339                     if(!vi.escaped) {
340                         result.add(vi);
341                     }
342                 }
343                 return result.isEmpty() ? null : result;
344             }
345     
346             void report() {
347                 for(ValueInfo vi : observedValues.values()) {
348                     if(!vi.escaped) {
349                         if(vi.hasObjectOnlyCall && vi.used && vi.var == null) {
350                             continue;
351                         }
352                         if(vi.hasObjectOnlyCall || (vi.used && vi.var != null)) {
353                             BugInstance bug = new BugInstance(vi.var == null ? "UC_USELESS_OBJECT_STACK" : "UC_USELESS_OBJECT",
354                                     NORMAL_PRIORITY).addClassAndMethod(classContext.getJavaClass(), method);
355                             if(vi.var != null) {
356                                 bug.add(new StringAnnotation(vi.var));
357                             }
358                             reporter.reportBug(bug.addType(vi.type).addSourceLine(classContext, method, vi.created));
359                         }
360                     }
361                 }
362             }
363         }

446         public FindUselessObjects(BugReporter reporter) {
447             this.reporter = reporter;
448             this.noSideEffectMethods = Global.getAnalysisCache().getDatabase(NoSideEffectMethodsDatabase.class);
449         }

451         @Override
452         public void visitClassContext(ClassContext classContext) {
453             for(Method method : classContext.getMethodsInCallOrder()) {
454                 if(method.isAbstract() || method.isNative()) {
455                     continue;
456                 }
457                 try {
458                     analyzeMethod(classContext, method);
459                 } catch (CheckedAnalysisException e) {
460                     reporter.logError("Error analyzing "+method+" (class: "+classContext.getJavaClass().getClassName()+")", e);
461                 }
462             }
463         }

465         private void analyzeMethod(ClassContext classContext, Method method) throws CheckedAnalysisException {
466             LocalVariableTable lvt = method.getLocalVariableTable();
467             UselessValuesContext context = new UselessValuesContext(classContext, method);
468             context.initObservedValues();
469             if(context.isEmpty()) {
470                 return;
471             }
472             context.enhanceViaMergeTree();
473             boolean changed;
474             do {
475                 changed = false;
476                 for(Iterator<GenLocation> iterator = context.genIterator(); iterator.hasNext() && !context.isEmpty(); ) {
477                     GenLocation location = iterator.next();
478                     Instruction inst = location.getHandle().getInstruction();
479                     ValueNumberFrame before = location.frameBefore();
480                     if (!before.isValid()) {
481                         continue;
482                     }
483                     if(inst instanceof IINC) {
484                         int index = ((IINC)inst).getIndex();
485                         Set<ValueInfo> vals = context.getLiveVals(before.getValue(index));
486                         if(vals != null) {
487                             changed |= context.propagateValues(vals, null, location.frameAfter().getValue(index));
488                         }
489                         continue;
490                     }
491                     int nconsumed = inst.consumeStack(context.cpg);
492                     if(nconsumed > 0) {
493                         ValueNumber[] vns = new ValueNumber[nconsumed];
494                         before.getTopStackWords(vns);
495                         for(int i=0; i<nconsumed; i++) {
496                             ValueNumber vn = vns[i];
497                             Set<ValueInfo> vals = context.getLiveVals(vn);
498                             if(vals != null) {
499                                 switch(inst.getOpcode()) {
500                                 case ASTORE:
501                                 case ASTORE_0:
502                                 case ASTORE_1:
503                                 case ASTORE_2:
504                                 case ASTORE_3:
505                                     for(ValueInfo vi : vals) {
506                                         if(vi.var == null && vi.origValue == vn.getNumber()) {
507                                             int index = ((StoreInstruction)inst).getIndex();
508                                             LocalVariable lv = lvt == null ? null : lvt.getLocalVariable(index, location.getHandle().getNext().getPosition());
509                                             vi.var = lv == null ? "var$"+index : lv.getName();
510                                             vi.hasObjectOnlyCall = false;
511                                             changed = true;
512                                         }
513                                     }
514                                     break;
515                                 case POP:
516                                 case POP2:
517                                 case DUP:
518                                 case DUP2:
519                                 case DUP_X1:
520                                 case DUP2_X1:
521                                 case ISTORE:
522                                 case ISTORE_0:
523                                 case ISTORE_1:
524                                 case ISTORE_2:
525                                 case ISTORE_3:
526                                 case LSTORE:
527                                 case LSTORE_0:
528                                 case LSTORE_1:
529                                 case LSTORE_2:
530                                 case LSTORE_3:
531                                 case FSTORE:
532                                 case FSTORE_0:
533                                 case FSTORE_1:
534                                 case FSTORE_2:
535                                 case FSTORE_3:
536                                 case DSTORE:
537                                 case DSTORE_0:
538                                 case DSTORE_1:
539                                 case DSTORE_2:
540                                 case DSTORE_3:
541                                 case SWAP:
542                                 case IMPDEP1:
543                                 case IMPDEP2:
544                                 case CHECKCAST:
545                                 case MONITORENTER:
546                                     break;
547                                 case IADD:
548                                 case LADD:
549                                 case FADD:
550                                 case DADD:
551                                 case ISUB:
552                                 case LSUB:
553                                 case FSUB:
554                                 case DSUB:
555                                 case IMUL:
556                                 case DMUL:
557                                 case LMUL:
558                                 case FMUL:
559                                 case IDIV:
560                                 case DDIV:
561                                 case LDIV:
562                                 case FDIV:
563                                 case INEG:
564                                 case LNEG:
565                                 case FNEG:
566                                 case DNEG:
567                                 case IREM:
568                                 case LREM:
569                                 case FREM:
570                                 case DREM:
571                                 case ISHL:
572                                 case LSHL:
573                                 case ISHR:
574                                 case LSHR:
575                                 case IUSHR:
576                                 case LUSHR:
577                                 case IAND:
578                                 case LAND:
579                                 case IOR:
580                                 case LOR:
581                                 case IXOR:
582                                 case LXOR:
583                                 case I2L:
584                                 case I2F:
585                                 case I2D:
586                                 case L2I:
587                                 case L2F:
588                                 case L2D:
589                                 case F2I:
590                                 case F2L:
591                                 case F2D:
592                                 case D2I:
593                                 case D2L:
594                                 case D2F:
595                                 case I2B:
596                                 case I2C:
597                                 case I2S:
598                                 case LCMP:
599                                 case FCMPL:
600                                 case FCMPG:
601                                 case DCMPL:
602                                 case DCMPG:
603                                 case ARRAYLENGTH:
604                                     changed |= context.propagateValues(vals, null, location.frameAfter().getTopValue());
605                                     break;
606                                 case GETFIELD:
607                                 case AALOAD:
608                                 case DALOAD:
609                                 case BALOAD:
610                                 case CALOAD:
611                                 case LALOAD:
612                                 case SALOAD:
613                                 case IALOAD:
614                                     changed |= context.propagateValues(vals, vn, location.frameAfter().getTopValue());
615                                     break;
616                                 case AASTORE:
617                                 case DASTORE:
618                                 case BASTORE:
619                                 case CASTORE:
620                                 case LASTORE:
621                                 case SASTORE:
622                                 case IASTORE:
623                                 case PUTFIELD:
624                                     if(i == 0) {
625                                         ValueNumber value = vns[vns.length-1];
626                                         if(!value.hasFlag(ValueNumber.CONSTANT_VALUE) && !value.hasFlag(ValueNumber.CONSTANT_CLASS_OBJECT) &&
627                                                 !context.observedValues.containsKey(value.getNumber())) {
628                                             changed |= context.setDerivedEscape(vals, vn);
629                                         }
630                                         changed |= context.setObjectOnly(vals, vn);
631                                     } else {
632                                         if(context.escaped(vns[0])) {
633                                             changed |= context.setEscape(vals);
634                                         } else {
635                                             changed |= context.propagateValues(vals, null, vns[0]);
636                                         }
637                                     }
638                                     break;
639                                 case INVOKESTATIC:
640                                 case INVOKESPECIAL:
641                                 case INVOKEINTERFACE:
642                                 case INVOKEVIRTUAL:
643                                     MethodDescriptor m = new MethodDescriptor((InvokeInstruction) inst, context.cpg);
644                                     XMethod xMethod = null;
645                                     try {
646                                         Type type = location.typeFrameBefore().getStackValue(nconsumed-1);
647                                         xMethod = Global
648                                                 .getAnalysisCache()
649                                                 .getClassAnalysis(XClass.class,
650                                                         DescriptorFactory.createClassDescriptorFromSignature(type.getSignature()))
651                                                         .findMatchingMethod(m);
652                                     } catch (CheckedAnalysisException e) {
653                                         // ignore
654                                     }
655                                     if(xMethod != null) {
656                                         m = xMethod.getMethodDescriptor();
657                                     }
658                                     MethodSideEffectStatus status = noSideEffectMethods.status(m);
659                                     if(status == MethodSideEffectStatus.NSE || status == MethodSideEffectStatus.SE_CLINIT) {
660                                         if(m.getName().equals("<init>")) {
661                                             if(vns[0].equals(context.thisValue)) {
662                                                 changed |= context.setEscape(vals);
663                                             } else {
664                                                 changed |= context.propagateValues(vals, null, vns[0]);
665                                             }
666                                         } else {
667                                             changed |= context.propagateToReturnValue(vals, vn, location, m);
668                                         }
669                                         break;
670                                     }
671                                     if(status == MethodSideEffectStatus.OBJ) {
672                                         if(i == 0) {
673                                             changed |= context.setDerivedEscape(vals, vn);
674                                             changed |= context.propagateToReturnValue(vals, vn, location, m);
675                                             changed |= context.setObjectOnly(vals, vn);
676                                             break;
677                                         } else {

691                             }
692                         }
693                     }
