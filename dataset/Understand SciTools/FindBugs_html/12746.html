41     /**
42      * Find occurrences of collection.toArray( new Foo[0] ); This causes another
43      * memory allocation through reflection Much better to do collection.toArray(
44      * new Foo[collection.size()] );
45      *
46      * @author Dave Brosius
47      */
48     public class InefficientToArray extends BytecodeScanningDetector implements StatelessDetector {
49         private static final boolean DEBUG = SystemProperties.getBoolean("ita.debug");
50     
51         private static final List<MethodDescriptor> methods = Collections.singletonList(new MethodDescriptor("", "toArray",
52                 "([Ljava/lang/Object;)[Ljava/lang/Object;"));
53     
54         static final int SEEN_NOTHING = 0;
55     
56         static final int SEEN_ICONST_0 = 1;
57     
58         static final int SEEN_ANEWARRAY = 2;
59     
60         private final static JavaClass collectionClass;
61     
62         private final BugReporter bugReporter;
63     
64         private final BugAccumulator bugAccumulator;
65     
66         private int state = SEEN_NOTHING;
67     
68         static {
69             JavaClass tmp = null;
70             try {
71                 tmp = AnalysisContext.lookupSystemClass("java.util.Collection");
72             } catch (ClassNotFoundException cnfe) {
73                 AnalysisContext.reportMissingClass(cnfe);
74             }
75             collectionClass = tmp;
76         }
77     
78         public InefficientToArray(BugReporter bugReporter) {
79             this.bugReporter = bugReporter;
80             this.bugAccumulator = new BugAccumulator(bugReporter);
81         }
82     
83         @Override
84         public void visitClassContext(ClassContext classContext) {
85             if (collectionClass != null && hasInterestingMethod(classContext.getJavaClass().getConstantPool(), methods)) {
86                 classContext.getJavaClass().accept(this);
87             }
88         }
89     
90         @Override
91         public void visit(Method obj) {
92             if (DEBUG) {
93                 System.out.println("------------------- Analyzing " + obj.getName() + " ----------------");
94             }
95             state = SEEN_NOTHING;
96             super.visit(obj);
97         }
98     
99         @Override
100         public void visit(Code obj) {
101             super.visit(obj);
102             bugAccumulator.reportAccumulatedBugs();
103     
104         }
105     
106         @Override
107         public void sawOpcode(int seen) {
108             if (DEBUG) {
109                 System.out.println("State: " + state + "  Opcode: " + OPCODE_NAMES[seen]);
110             }
111     
112             switch (state) {
113             case SEEN_NOTHING:
114                 if (seen == ICONST_0) {
115                     state = SEEN_ICONST_0;
116                 }
117                 break;
118     
119             case SEEN_ICONST_0:
120                 if (seen == ANEWARRAY) {
121                     state = SEEN_ANEWARRAY;
122                 } else {
123                     state = SEEN_NOTHING;
124                 }
125                 break;
126     
127             case SEEN_ANEWARRAY:
128                 if (((seen == INVOKEVIRTUAL) || (seen == INVOKEINTERFACE)) && ("toArray".equals(getNameConstantOperand()))
129                         && ("([Ljava/lang/Object;)[Ljava/lang/Object;".equals(getSigConstantOperand()))) {
130                     try {
131                         String clsName = getDottedClassConstantOperand();
132                         JavaClass cls = Repository.lookupClass(clsName);
133                         if (cls.implementationOf(collectionClass)) {
134                             bugAccumulator.accumulateBug(
135                                     new BugInstance(this, "ITA_INEFFICIENT_TO_ARRAY", LOW_PRIORITY).addClassAndMethod(this), this);
136                         }
137     
138                     } catch (ClassNotFoundException cnfe) {
139                         bugReporter.reportMissingClass(cnfe);
140                     }
141                 }
142                 state = SEEN_NOTHING;
143                 break;
144     
145             default:
146                 state = SEEN_NOTHING;
147                 break;
148             }
149         }
150     }
