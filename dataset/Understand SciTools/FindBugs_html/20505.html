20     package edu.umd.cs.findbugs.detect;
21     
22     import java.util.HashSet;
23     import java.util.Set;
24     
25     import org.apache.bcel.classfile.CodeException;
26     import org.apache.bcel.classfile.ExceptionTable;
27     
28     import edu.umd.cs.findbugs.BugInstance;
29     import edu.umd.cs.findbugs.BugReporter;
30     import edu.umd.cs.findbugs.Detector;
31     import edu.umd.cs.findbugs.SystemProperties;
32     import edu.umd.cs.findbugs.ba.ClassContext;
33     import edu.umd.cs.findbugs.visitclass.PreorderVisitor;
34     
35     public class DontCatchIllegalMonitorStateException extends PreorderVisitor implements Detector {
36     
37         private static final boolean DEBUG = SystemProperties.getBoolean("dcimse.debug");
38     
39         BugReporter bugReporter;
40     
41         Set<String> msgs = null;
42     
43         ClassContext classContext;
44     
45         public DontCatchIllegalMonitorStateException(BugReporter bugReporter) {
46             this.bugReporter = bugReporter;
47             if (DEBUG) {
48                 msgs = new HashSet<String>();
49             }
50         }
51     
52         @Override
53         public void visit(ExceptionTable obj) {
54             if (DEBUG) {
55                 String names[] = obj.getExceptionNames();
56                 for (String name : names) {
57                     if ("java.lang.Exception".equals(name) || "java.lang.Throwable".equals(name)) {
58                         System.out.println(name + " thrown by " + getFullyQualifiedMethodName());
59                     }
60                 }
61             }
62         }
63     
64         @Override
65         public void visit(CodeException obj) {
66             int type = obj.getCatchType();
67             if (type == 0) {
68                 return;
69             }
70             String name = getConstantPool().constantToString(getConstantPool().getConstant(type));
71             if (DEBUG) {
72                 String msg = "Catching " + name + " in " + getFullyQualifiedMethodName();
73                 if (msgs.add(msg)) {
74                     System.out.println(msg);
75                 }
76             }
77             if ("java.lang.IllegalMonitorStateException".equals(name)) {
78                 bugReporter.reportBug(new BugInstance(this, "IMSE_DONT_CATCH_IMSE", HIGH_PRIORITY).addClassAndMethod(this)
79                         .addSourceLine(this.classContext, this, obj.getHandlerPC()));
80             }
81     
82         }
83     
84         @Override
85         public void visitClassContext(ClassContext classContext) {
86             this.classContext = classContext;
87             classContext.getJavaClass().accept(this);
88         }
89     
90         @Override
91         public void report() {
92         }
93     }
