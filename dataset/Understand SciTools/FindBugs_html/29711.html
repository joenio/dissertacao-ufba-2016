22     import java.util.Set;
23     import java.util.SortedSet;
24     
25     import javax.annotation.CheckForNull;
26     import javax.annotation.Nonnull;
27     
28     import edu.umd.cs.findbugs.BugAnnotation;
29     import edu.umd.cs.findbugs.ba.Location;
30     import edu.umd.cs.findbugs.ba.vna.ValueNumber;
31     import edu.umd.cs.findbugs.ba.vna.ValueNumberDataflow;
32     import edu.umd.cs.findbugs.ba.vna.ValueNumberFrame;
33     
34     /**
35      * Callback interface for collecting null pointer derefs and redundant null
36      * comparisons.
37      *
38      * @see edu.umd.cs.findbugs.ba.npe.NullDerefAndRedundantComparisonFinder
39      * @author David Hovemeyer
40      */
41     public interface NullDerefAndRedundantComparisonCollector {
42         /**
43          * Subclasses should override this method to capture locations where a null
44          * pointer is dereferenced.
45          *
46          * @param location
47          *            the Location of the null dereference
48          * @param valueNumber
49          *            the ValueNumber of the possibly-null value
50          * @param refValue
51          *            the kind of possibly-null value dereferenced
52          * @param vnaFrame
53          *            The ValueNumber Frame at the point where the dereference
54          *            occurred
55          * @deprecated Use
56          *             {@link #foundNullDeref(Location,ValueNumber,IsNullValue,ValueNumberFrame,boolean)}
57          *             instead
58          */
59         @Deprecated
60         public void foundNullDeref(Location location, ValueNumber valueNumber, IsNullValue refValue, ValueNumberFrame vnaFrame);
61     
62         /**
63          * Subclasses should override this method to capture locations where a null
64          * pointer is dereferenced.
65          *
66          * @param location
67          *            the Location of the null dereference
68          * @param valueNumber
69          *            the ValueNumber of the possibly-null value
70          * @param refValue
71          *            the kind of possibly-null value dereferenced
72          * @param vnaFrame
73          *            The ValueNumber Frame at the point where the dereference
74          *            occurred
75          * @param isConsistent
76          *            true if the refValue is identical at all clones of the same
77          *            instruction
78          */
79         public void foundNullDeref(Location location, ValueNumber valueNumber, IsNullValue refValue, ValueNumberFrame vnaFrame,
80                 boolean isConsistent);
81     
82         /**
83          * Subclasses should override this method to capture locations where a
84          * redundant null comparison is performed.
85          *
86          * @param location
87          *            the Location of the redundant null check
88          * @param redundantBranch
89          *            the RedundantBranch
90          */
91         public void foundRedundantNullCheck(Location location, RedundantBranch redundantBranch);
92     
93         /**
94          * Subclasses should override this method to capture values assigned null
95          * (or that become null through a comparison and branch) that are guaranteed
96          * to reach a dereference (ignoring implicit exception paths).
97          *
98          * @param assignedNullLocationSet
99          *            set of locations where the value becomes null
100          * @param derefLocationSet
101          *            set of locations where dereferences occur
102          * @param doomedLocations
103          *            locations at which the value is doomed
104          * @param vna
105          *            ValueNumberDataflow
106          * @param refValue
107          *            the null value
108          * @param variableAnnotation
109          *            TODO
110          * @param deref
111          *            TODO
112          * @param npeIfStatementCovered
113          *            true if doom location is a statement
114          */
115         public void foundGuaranteedNullDeref(@Nonnull Set<Location> assignedNullLocationSet, @Nonnull Set<Location> derefLocationSet,
116                 SortedSet<Location> doomedLocations, ValueNumberDataflow vna, ValueNumber refValue,
117                 @CheckForNull BugAnnotation variableAnnotation, NullValueUnconditionalDeref deref, boolean npeIfStatementCovered);
118     }
