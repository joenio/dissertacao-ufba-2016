74     /*
75      * Look for lazy initialization of fields which
76      * are not volatile.  This is quite similar to checking for
77      * double checked locking, except that there is no lock.
78      *
79      * @author David Hovemeyer
80      */
81     
82     public final class LazyInit extends ByteCodePatternDetector implements StatelessDetector {
83         private final BugReporter bugReporter;
84     
85         private static final boolean DEBUG = SystemProperties.getBoolean("lazyinit.debug");
86     
87         /**
88          * The pattern to look for.
89          */
90         private static ByteCodePattern pattern = new ByteCodePattern();
91     
92         static {
93             pattern.add(new Load("f", "val").label("start")).add(new IfNull("val").label("test"))
94             .add(new Wild(1, 1).label("createObject").dominatedBy("test"))
95             .add(new Store("f", pattern.dummyVariable()).label("end").dominatedBy("createObject"));
96         }
97     
98         public LazyInit(BugReporter bugReporter) {
99             this.bugReporter = bugReporter;
100         }

103         public Object clone() {
104             try {
105                 return super.clone();
106             } catch (CloneNotSupportedException e) {
107                 throw new AssertionError(e);
108             }
109         }

111         @Override
112         public BugReporter getBugReporter() {
113             return bugReporter;
114         }

116         BitSet reported = new BitSet();
117     
118         @Override
119         public ByteCodePattern getPattern() {
120             return pattern;
121         }

123         @Override
124         public boolean prescreen(Method method, ClassContext classContext) {
125             if ("<clinit>".equals(method.getName())) {
126                 return false;
127             }
128     
129             Code code = method.getCode();
130             if (code.getCode().length > 5000) {
131                 return false;
132             }
133     
134             BitSet bytecodeSet = classContext.getBytecodeSet(method);
135             if (bytecodeSet == null) {
136                 return false;
137             }
138     
139             // The pattern requires a get/put pair accessing the same field.
140             boolean hasGetStatic = bytecodeSet.get(Constants.GETSTATIC);
141             boolean hasPutStatic = bytecodeSet.get(Constants.PUTSTATIC);
142             if (!hasGetStatic || !hasPutStatic) {
143                 return false;
144             }
145     
146             // If the method is synchronized, then we'll assume that
147             // things are properly synchronized
148             if (method.isSynchronized()) {
149                 return false;
150             }
151     
152             reported.clear();
153             return true;
154         }
155     
156         @Override
157         public void reportMatch(ClassContext classContext, Method method, ByteCodePatternMatch match) throws CFGBuilderException,
158         DataflowAnalysisException {
159             JavaClass javaClass = classContext.getJavaClass();
160             MethodGen methodGen = classContext.getMethodGen(method);
161             CFG cfg = classContext.getCFG(method);
162     
163     
164             // Get the variable referenced in the pattern instance.
165             BindingSet bindingSet = match.getBindingSet();
166             Binding binding = bindingSet.lookup("f");
167     
168             // Look up the field as an XField.
169             // If it is volatile, then the instance is not a bug.
170             FieldVariable field = (FieldVariable) binding.getVariable();
171             XField xfield = Hierarchy.findXField(field.getClassName(), field.getFieldName(), field.getFieldSig(),
172                     field.isStatic());
173             if (!xfield.isResolved()) {
174                 return;
175             }
176     
177             // XXX: for now, ignore lazy initialization of instance fields
178             if (!xfield.isStatic()) {
179                 return;
180             }
181     
182             // Definitely ignore synthetic class$ fields
183             if (xfield.getName().startsWith("class$") || xfield.getName().startsWith("array$")) {
184                 if (DEBUG) {
185                     System.out.println("Ignoring field " + xfield.getName());
186                 }
187                 return;
188             }
189     
190             // Ignore non-reference fields
191             String signature = xfield.getSignature();
192             if (!signature.startsWith("[") && !signature.startsWith("L")) {
193                 if (DEBUG) {
194                     System.out.println("Ignoring non-reference field " + xfield.getName());
195                 }
196                 return;
197             }
198     
199             // Strings are (mostly) safe to pass by data race in 1.5
200             if ("Ljava/lang/String;".equals(signature)) {
201                 return;
202             }
203     
204             // GUI types should not be accessed from multiple threads
205     
206             if (signature.charAt(0) == 'L') {
207                 ClassDescriptor fieldType = DescriptorFactory.createClassDescriptorFromFieldSignature(signature);
208     
209                 while (fieldType != null) {
210                     XClass fieldClass;
211                     try {
212                         fieldClass = Global.getAnalysisCache().getClassAnalysis(XClass.class, fieldType);
213                     } catch (CheckedAnalysisException e) {
214                         break;
215                     }
216     
217                     String name = fieldClass.getClassDescriptor().getClassName();
218                     if (name.startsWith("java/awt") || name.startsWith("javax/swing")) {
219                         return;
220                     }
221                     if ("java/lang/Object".equals(name)) {
222                         break;
223                     }
224                     fieldType = fieldClass.getSuperclassDescriptor();
225                 }
226             }
227     
228             // Get locations matching the beginning of the object creation,
229             // and the final field store.
230             PatternElementMatch createBegin = match.getFirstLabeledMatch("createObject");
231             PatternElementMatch store = match.getFirstLabeledMatch("end");
232             PatternElementMatch test = match.getFirstLabeledMatch("test");
233             InstructionHandle testInstructionHandle = test.getMatchedInstructionInstructionHandle();
234             if (reported.get(testInstructionHandle.getPosition())) {
235                 return;
236             }
237     
238             // Get all blocks
239             //
240             // (1) dominated by the wildcard instruction matching
241             // the beginning of the instructions creating the object, and
242             // (2) postdominated by the field store
243             //
244             // Exception edges are not considered in computing
245             // dominators/postdominators.
246             // We will consider this to be all of the code that creates
247             // the object.
248             DominatorsAnalysis domAnalysis = classContext.getNonExceptionDominatorsAnalysis(method);
249             PostDominatorsAnalysis postDomAnalysis = classContext.getNonExceptionPostDominatorsAnalysis(method);
250             BitSet extent = domAnalysis.getAllDominatedBy(createBegin.getBasicBlock());
251             BitSet postDom = postDomAnalysis.getAllDominatedBy(store.getBasicBlock());
252             // System.out.println("Extent: " + extent);
253             if (DEBUG) {
254                 System.out.println("test  dominates: " + extent);
255                 System.out.println("Field store postdominates " + postDom);
256             }
257             extent.and(postDom);
258             if (DEBUG) {
259                 System.out.println("extent: " + extent);
260             }
261             // Check all instructions in the object creation extent
262             //
263             // (1) to determine the common lock set, and
264             // (2) to check for NEW and Invoke instructions that might create an
265             // object
266             //
267             // We ignore matches where a lock is held consistently,
268             // or if the extent does not appear to create a new object.
269             LockDataflow lockDataflow = classContext.getLockDataflow(method);
270             LockSet lockSet = null;
271             boolean sawNEW = false, sawINVOKE = false;
272             for (BasicBlock block : cfg.getBlocks(extent)) {
273                 for (Iterator<InstructionHandle> j = block.instructionIterator(); j.hasNext();) {
274                     InstructionHandle handle = j.next();
275                     if (handle.equals(store.getMatchedInstructionInstructionHandle())) {
276                         break;
277                     }
278                     Location location = new Location(handle, block);
279     
280                     // Keep track of whether we saw any instructions
281                     // that might actually have created a new object.
282                     Instruction ins = handle.getInstruction();
283                     if (DEBUG) {
284                         System.out.println(location);
285                     }
286                     if (ins instanceof AllocationInstruction) {
287                         sawNEW = true;
288                     } else if (ins instanceof InvokeInstruction) {
289                         if (ins instanceof INVOKESTATIC
290                                 && ((INVOKESTATIC) ins).getMethodName(classContext.getConstantPoolGen()).startsWith("new")) {
291                             sawNEW = true;
292                         }
293                         sawINVOKE = true;
294                     }
295     
296                     // Compute lock set intersection for all matched
297                     // instructions.
298                     LockSet insLockSet = lockDataflow.getFactAtLocation(location);
299                     if (lockSet == null) {
300                         lockSet = new LockSet();
301                         lockSet.copyFrom(insLockSet);
302                     } else {
303                         lockSet.intersectWith(insLockSet);
304                     }
305                 }
306             }
307     
308             if (!(sawNEW || sawINVOKE)) {
309                 return;
310             }
311             if (lockSet == null) {
312                 throw new IllegalStateException("lock set is null");
313             }
314             if (!lockSet.isEmpty()) {
315                 return;
316             }
317     
318             boolean sawGetStaticAfterPutStatic = false;
319             check: if (signature.startsWith("[") || signature.startsWith("L")) {
320     
321                 BitSet postStore = domAnalysis.getAllDominatedBy(store.getBasicBlock());
322                 for (BasicBlock block : cfg.getBlocks(postStore)) {
323                     for (Iterator<InstructionHandle> j = block.instructionIterator(); j.hasNext();) {
324                         InstructionHandle handle = j.next();
325     
326                         InstructionHandle nextHandle = handle.getNext();
327                         Instruction ins = handle.getInstruction();
328     
329                         if (ins instanceof GETSTATIC && potentialInitialization(nextHandle)) {
330                             XField field2 = XFactory.createXField((FieldInstruction) ins, methodGen.getConstantPool());
331                             if (xfield.equals(field2)) {
332                                 sawGetStaticAfterPutStatic = true;
333                                 break check;
334                             }
335                         }
336                     }
337                 }
338             }
339     
340             // Compute the priority:
341             // - ignore lazy initialization of instance fields
342             // - when it's done in a public method, emit a high priority warning
343             // - protected or default access method, emit a medium priority
344             // warning
345             // - otherwise, low priority
346     
347             if (!sawGetStaticAfterPutStatic && xfield.isVolatile()) {
348                 return;
349             }
350             int priority = LOW_PRIORITY;
351             boolean isDefaultAccess = (method.getAccessFlags() & (Constants.ACC_PUBLIC | Constants.ACC_PRIVATE | Constants.ACC_PROTECTED)) == 0;
352             if (method.isPublic()) {
353                 priority = NORMAL_PRIORITY;
354             } else if (method.isProtected() || isDefaultAccess) {
355                 priority = NORMAL_PRIORITY;
356             }
357             if (signature.startsWith("[") || signature.startsWith("Ljava/util/")) {
358                 priority--;
359             }
360             if (!sawNEW) {
361                 priority++;
362             }
363             if (!sawGetStaticAfterPutStatic && priority < LOW_PRIORITY) {
364                 priority = LOW_PRIORITY;
365             }
366             if (classContext.getXClass().usesConcurrency()) {
367                 priority--;
368             }
369             // Report the bug.
370             InstructionHandle start = match.getLabeledInstruction("start");
371             InstructionHandle end = match.getLabeledInstruction("end");
372             String sourceFile = javaClass.getSourceFileName();
373             bugReporter.reportBug(new BugInstance(this, sawGetStaticAfterPutStatic ? "LI_LAZY_INIT_UPDATE_STATIC"
374                     : "LI_LAZY_INIT_STATIC", priority).addClassAndMethod(methodGen, sourceFile).addField(xfield)
375                     .describe("FIELD_ON").addSourceLine(classContext, methodGen, sourceFile, start, end));
376             reported.set(testInstructionHandle.getPosition());
377     
378         }
379     
380         private boolean potentialInitialization(InstructionHandle nextHandle) {
381             if (nextHandle == null) {
382                 return true;
383             }
384             Instruction instruction = nextHandle.getInstruction();
385             if (instruction instanceof ReturnInstruction) {
386                 return false;
387             }
388             if (instruction instanceof IfInstruction) {
389                 return false;
390             }
391             return true;
392         }
