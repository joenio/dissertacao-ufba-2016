20     package edu.umd.cs.findbugs.detect;
21     
22     import java.util.Set;
23     
24     import org.apache.bcel.classfile.Constant;
25     import org.apache.bcel.classfile.ConstantClass;
26     import org.apache.bcel.classfile.ConstantString;
27     import org.apache.bcel.classfile.JavaClass;
28     import org.apache.bcel.classfile.Method;
29     
30     import edu.umd.cs.findbugs.BugInstance;
31     import edu.umd.cs.findbugs.BugReporter;
32     import edu.umd.cs.findbugs.BytecodeScanningDetector;
33     import edu.umd.cs.findbugs.StatelessDetector;
34     import edu.umd.cs.findbugs.ba.AnalysisContext;
35     import edu.umd.cs.findbugs.ba.ch.Subtypes2;
36     import edu.umd.cs.findbugs.classfile.ClassDescriptor;
37     
38     public class InheritanceUnsafeGetResource extends BytecodeScanningDetector implements StatelessDetector {
39     
40         private final BugReporter bugReporter;
41     
42         private boolean classIsFinal;
43     
44         // private boolean methodIsVisibleToOtherPackages;
45         private boolean classIsVisibleToOtherPackages;
46     
47         // private boolean methodIsFinal;
48         private boolean methodIsStatic;
49     
50         int state = 0;
51     
52         int sawGetClass;
53     
54         boolean reportedForThisClass;
55     
56         String stringConstant;
57     
58         int prevOpcode;
59     
60         public InheritanceUnsafeGetResource(BugReporter bugReporter) {
61             this.bugReporter = bugReporter;
62         }
63     
64         @Override
65         public void visit(JavaClass obj) {
66             classIsFinal = obj.isFinal();
67             reportedForThisClass = false;
68             classIsVisibleToOtherPackages = obj.isPublic() || obj.isProtected();
69         }
70     
71         @Override
72         public void visit(Method obj) {
73             methodIsStatic = obj.isStatic();
74             state = 0;
75             sawGetClass = -100;
76         }
77     
78         @Override
79         public void sawOpcode(int seen) {
80             if (reportedForThisClass) {
81                 return;
82             }
83     
84             switch (seen) {
85             case LDC:
86                 Constant constantValue = getConstantRefOperand();
87                 if (constantValue instanceof ConstantClass) {
88                     sawGetClass = -100;
89                 } else if (constantValue instanceof ConstantString) {
90                     stringConstant = ((ConstantString) constantValue).getBytes(getConstantPool());
91                 }
92                 break;
93     
94             case ALOAD_0:
95                 state = 1;
96                 break;
97             case INVOKEVIRTUAL:
98                 if ("java/lang/Class".equals(getClassConstantOperand())
99                         && ("getResource".equals(getNameConstantOperand()) || "getResourceAsStream".equals(getNameConstantOperand()))
100                         && sawGetClass + 10 >= getPC()) {
101                     int priority = NORMAL_PRIORITY;
102                     if (prevOpcode == LDC && stringConstant != null && stringConstant.length() > 0 && stringConstant.charAt(0) == '/') {
103                         priority = LOW_PRIORITY;
104                     } else {
105                         priority = adjustPriority(priority);
106                     }
107                     bugReporter.reportBug(new BugInstance(this, "UI_INHERITANCE_UNSAFE_GETRESOURCE", priority)
108                     .addClassAndMethod(this).addSourceLine(this));
109                     reportedForThisClass = true;
110     
111                 } else if (state == 1 && !methodIsStatic && !classIsFinal && classIsVisibleToOtherPackages
112                         && "getClass".equals(getNameConstantOperand()) && "()Ljava/lang/Class;".equals(getSigConstantOperand())) {
113                     sawGetClass = getPC();
114                 }
115                 state = 0;
116                 break;
117             default:
118                 state = 0;
119                 break;
120             }
121             if (seen != LDC) {
122                 stringConstant = null;
123             }
124             prevOpcode = seen;
125     
126         }
127     
128         /**
129          * Adjust the priority of a warning about to be reported.
130          *
131          * @param priority
132          *            initial priority
133          * @return adjusted priority
134          */
135         private int adjustPriority(int priority) {
136     
137             try {
138                 Subtypes2 subtypes2 = AnalysisContext.currentAnalysisContext().getSubtypes2();
139     
140                 if (!subtypes2.hasSubtypes(getClassDescriptor())) {
141                     priority++;
142                 } else {
143                     Set<ClassDescriptor> mySubtypes = subtypes2.getSubtypes(getClassDescriptor());
144     
145                     String myPackagename = getThisClass().getPackageName();
146     
147                     for (ClassDescriptor c : mySubtypes) {
148                         if (c.equals(getClassDescriptor())) {
149                             continue;
150                         }
151                         if (!c.getPackageName().equals(myPackagename)) {
152                             priority--;
153                             break;
154                         }
155                     }
156                 }
157             } catch (ClassNotFoundException e) {
158                 bugReporter.reportMissingClass(e);
159             }
160             return priority;
161         }
