20     package edu.umd.cs.findbugs;
21     
22     import java.io.ByteArrayInputStream;
23     import java.io.IOException;
24     import java.io.InputStream;
25     import java.net.URL;
26     import java.util.Properties;
27     import java.util.regex.Matcher;
28     import java.util.regex.Pattern;
29     
30     import edu.umd.cs.findbugs.ba.AnalysisContext;
31     import edu.umd.cs.findbugs.io.IO;
32     
33     /**
34      * @author pugh
35      */
36     public class PropertyBundle {
37     
38         private final Properties properties;
39     
40         class Rewriter {
41             private final String urlRewritePatternString = getOSDependentProperty("findbugs.urlRewritePattern");
42     
43             private final Pattern urlRewritePattern;
44     
45             private final String urlRewriteFormat = getOSDependentProperty("findbugs.urlRewriteFormat");
46     
47             Rewriter() {
48                 Pattern p = null;
49                 if (urlRewritePatternString != null && urlRewriteFormat != null) {
50                     try {
51                         p = Pattern.compile(urlRewritePatternString);
52                     } catch (Throwable e) {
53                         assert true;
54                     }
55                 }
56     
57                 urlRewritePattern = p;
58             }
59         }
60     
61         volatile Rewriter rewriter;
62     
63         Rewriter getRewriter() {
64             if (rewriter == null) {
65                 synchronized (this) {
66                     if (rewriter == null) {
67                         rewriter = new Rewriter();
68                     }
69                 }
70             }
71             return rewriter;
72         }
73     
74         public PropertyBundle() {
75             properties = new Properties();
76         }
77     
78         public PropertyBundle(Properties properties) {
79             this.properties = (Properties) properties.clone();
80         }
81     
82         public PropertyBundle copy() {
83             return new PropertyBundle(properties);
84         }
85     
86         public Properties getProperties() {
87             return properties;
88         }
89     
90         public void loadPropertiesFromString(String contents) {
91             if (contents == null) {
92                 return;
93             }
94             InputStream in = null;
95             try {
96                 in = new ByteArrayInputStream(contents.getBytes("ISO-8859-1"));
97                 properties.load(in);
98             } catch (IOException e) {
99                 AnalysisContext.logError("Unable to load properties from " + contents, e);
100             } finally {
101                 IO.close(in);
102             }
103         }
104     
105         public void loadPropertiesFromURL(URL url) {
106     
107             if (url == null) {
108                 return;
109             }
110             InputStream in = null;
111             try {
112                 in = url.openStream();
113                 properties.load(in);
114             } catch (IOException e) {
115                 AnalysisContext.logError("Unable to load properties from " + url, e);
116             } finally {
117                 IO.close(in);
118             }
119         }
120     
121         public void loadProperties(Properties properties) {
122             this.properties.putAll(properties);
123         }
124     
125         /**
126          * Get boolean property, returning false if a security manager prevents us
127          * from accessing system properties
128          *
129          * @return true if the property exists and is set to true
130          */
131         public boolean getBoolean(String name) {
132             return getBoolean(name, false);
133         }
134     
135         public boolean getBoolean(String name, boolean defaultValue) {
136             boolean result = defaultValue;
137             try {
138                 String value = getProperty(name);
139                 if (value == null) {
140                     return defaultValue;
141                 }
142                 result = toBoolean(value);
143             } catch (IllegalArgumentException e) {
144             } catch (NullPointerException e) {
145             }
146             return result;
147         }
148     
149         private boolean toBoolean(String name) {
150             return ((name != null) && "true".equalsIgnoreCase(name));
151         }
152     
153         /**
154          * @param name
155          *            property name
156          * @param defaultValue
157          *            default value
158          * @return the int value (or defaultValue if the property does not exist)
159          */
160         public int getInt(String name, int defaultValue) {
161             try {
162                 String value = getProperty(name);
163                 if (value != null) {
164                     return Integer.decode(value);
165                 }
166             } catch (Exception e) {
167                 assert true;
168             }
169             return defaultValue;
170         }
171     
172         /**
173          * @param name
174          *            property name
175          * @return string value (or null if the property does not exist)
176          */
177         public String getOSDependentProperty(String name) {
178             String osDependentName = name + SystemProperties.OS_NAME;
179             String value = getProperty(osDependentName);
180             if (value != null) {
181                 return value;
182             }
183             return getProperty(name);
184         }
185     
186         /**
187          * @param name
188          *            property name
189          * @return string value (or null if the property does not exist)
190          */
191         public String getProperty(String name) {
192             try {
193                 String value = SystemProperties.getProperty(name);
194                 if (value != null) {
195                     return value;
196                 }
197                 return properties.getProperty(name);
198             } catch (Exception e) {
199                 return null;
200             }
201     
202         }
203     
204         public void setProperty(String name, String value) {
205             properties.setProperty(name, value);
206         }
207     
208         @Override
209         public String toString() {
210             return properties.toString();
211         }
212     
213         /**
214          * @param name
215          *            property name
216          * @param defaultValue
217          *            default value
218          * @return string value (or defaultValue if the property does not exist)
219          */
220         public String getProperty(String name, String defaultValue) {
221             String value = getProperty(name);
222             if (value != null) {
223                 return value;
224             }
225             return defaultValue;
226         }
227     
228         public String rewriteURLAccordingToProperties(String u) {
229             if (getRewriter().urlRewritePattern == null || getRewriter().urlRewriteFormat == null) {
230                 return u;
231             }
232             Matcher m = getRewriter().urlRewritePattern.matcher(u);
233             if (!m.matches()) {
234                 return u;
235             }
236             String result = String.format(getRewriter().urlRewriteFormat, m.group(1));
237             return result;
238         }

240     }
