272             public static long getVersionNum(BugCollection collection, String val,
273                     boolean roundToLaterVersion) {
274                 if (val == null) {
275                     return -1;
276                 }
277                 Map<String, AppVersion> versions = new HashMap<String, AppVersion>();
278                 SortedMap<Long, AppVersion> timeStamps = new TreeMap<Long, AppVersion>();
279     
280                 for (Iterator<AppVersion> i = collection.appVersionIterator(); i.hasNext();) {
281                     AppVersion v = i.next();
282                     versions.put(v.getReleaseName(), v);
283                     timeStamps.put(v.getTimestamp(), v);
284                 }
285                 // add current version to the maps
286                 AppVersion v = collection.getCurrentAppVersion();
287                 versions.put(v.getReleaseName(), v);
288                 timeStamps.put(v.getTimestamp(), v);
289     
290                 return getVersionNum(versions,  timeStamps,  val,
291                         roundToLaterVersion,  v.getSequenceNumber());
292             }

294             public static long getVersionNum(Map<String, AppVersion> versions, SortedMap<Long, AppVersion> timeStamps, String val,
295                     boolean roundToLaterVersion, long currentSeqNum) {
296                 if (val == null) {
297                     return -1;
298                 }
299                 long numVersions = currentSeqNum + 1;
300     
301                 if ("last".equals(val) || "lastVersion".equals(val)) {
302                     return numVersions - 1;
303                 }
304     
305                 AppVersion v = versions.get(val);
306                 if (v != null) {
307                     return v.getSequenceNumber();
308                 }
309                 try {
310                     long time = 0;
311                     if (val.endsWith("daysAgo")) {
312                         time = System.currentTimeMillis() - MILLISECONDS_PER_DAY
313                                 * Integer.parseInt(val.substring(0, val.length() - 7));
314                     } else {
315                         time = Date.parse(val);
316                     }
317                     return getAppropriateSeq(timeStamps, time, roundToLaterVersion);
318                 } catch (Exception e) {
319                     try {
320                         long version = Long.parseLong(val);
321                         if (version < 0) {
322                             version = numVersions + version;
323                         }
324                         return version;
325                     } catch (NumberFormatException e1) {
326                         throw new IllegalArgumentException("Could not interpret version specification of '" + val + "'");
327                     }
328                 }
329             }
330     
331             // timeStamps contains 0 10 20 30
332             // if roundToLater == true, ..0 = 0, 1..10 = 1, 11..20 = 2, 21..30 = 3,
333             // 31.. = Long.MAX
334             // if roundToLater == false, ..-1 = Long.MIN, 0..9 = 0, 10..19 = 1,
335             // 20..29 = 2, 30..39 = 3, 40 .. = 4
336             static private long getAppropriateSeq(SortedMap<Long, AppVersion> timeStamps, long when, boolean roundToLaterVersion) {
337                 if (roundToLaterVersion) {
338                     SortedMap<Long, AppVersion> geq = timeStamps.tailMap(when);
339                     if (geq.isEmpty()) {
340                         return Long.MAX_VALUE;
341                     }
342                     return geq.get(geq.firstKey()).getSequenceNumber();
343                 } else {
344                     SortedMap<Long, AppVersion> leq = timeStamps.headMap(when);
345                     if (leq.isEmpty()) {
346                         return Long.MIN_VALUE;
347                     }
348                     return leq.get(leq.lastKey()).getSequenceNumber();
349                 }
350             }

357             void adjustFilter(Project project, BugCollection collection) {
358                 suppressionFilter = project.getSuppressionFilter();
359     
360                 if (maxAgeSpecified) {
361                     minFirstSeen = collection.getAnalysisTimestamp() - maxAge * MILLISECONDS_PER_DAY;
362                 }
363                 first = getVersionNum(collection, firstAsString, true);
364                 maybeMutated = getVersionNum(collection, maybeMutatedAsString, true);
365                 last = getVersionNum(collection, lastAsString, true);
366                 before = getVersionNum(collection, beforeAsString, true);
367                 after = getVersionNum(collection, afterAsString, false);
368                 present = getVersionNum(collection, presentAsString, true);
369                 absent = getVersionNum(collection, absentAsString, true);
370     
371                 if (sloppyUniqueSpecified) {
372                     uniqueSloppy = new TreeSet<BugInstance>(new SloppyBugComparator());
373                 }
374     
375                 long fixed = getVersionNum(collection, fixedAsString, true);
376                 if (fixed >= 0)
377                 {
378                     last = fixed - 1; // fixed means last on previous sequence (ok
379                     // if -1)
380                 }
381             }

391             boolean evaluate(BugCollection collection, BugInstance bug) {
392     
393                 for (Matcher m : includeFilter) {
394                     if (!m.match(bug)) {
395                         return false;
396                     }
397                 }
398                 for (Matcher m : excludeFilter) {
399                     if (m.match(bug)) {
400                         return false;
401                     }
402                 }
403                 if (excludedInstanceHashes.contains(bug.getInstanceHash())) {
404                     return false;
405                 }
406                 if (annotation != null && bug.getAnnotationText().indexOf(annotation) == -1) {
407                     return false;
408                 }
409                 if (bug.getPriority() > priority) {
410                     return false;
411                 }
412                 if (firstAsString != null && bug.getFirstVersion() != first) {
413                     return false;
414                 }
415                 if (afterAsString != null && bug.getFirstVersion() <= after) {
416                     return false;
417                 }
418                 if (beforeAsString != null && bug.getFirstVersion() >= before) {
419                     return false;
420                 }
421                 if (hashesFromFile != null && !hashesFromFile.contains(bug.getInstanceHash())) {
422                     return false;
423                 }
424                 long lastSeen = bug.getLastVersion();
425                 if (lastSeen < 0) {
426                     lastSeen = collection.getSequenceNumber();
427                 }
428                 long thisDuration = lastSeen - bug.getFirstVersion();
429                 if (duration > 0 && thisDuration > duration) {
430                     return false;
431                 }
432                 if ((lastAsString != null || fixedAsString != null) && (last < 0 || bug.getLastVersion() != last)) {
433                     return false;
434                 }
435                 if (presentAsString != null && !bugLiveAt(bug, present)) {
436                     return false;
437                 }
438                 if (absentAsString != null && bugLiveAt(bug, absent)) {
439                     return false;
440                 }
441     
442                 if (hasFieldSpecified && (hasField != (bug.getPrimaryField() != null))) {
443                     return false;
444                 }
445                 if (hasLocalSpecified && (hasLocal != (bug.getPrimaryLocalVariableAnnotation() != null))) {
446                     return false;
447                 }
448     
449                 if (maxRank < Integer.MAX_VALUE && BugRanker.findRank(bug) > maxRank) {
450                     return false;
451                 }
452     
453                 if (activeSpecified && active == bug.isDead()) {
454                     return false;
455                 }
456                 if (removedByChangeSpecified && bug.isRemovedByChangeOfPersistingClass() != removedByChange) {
457                     return false;
458                 }
459                 if (introducedByChangeSpecified && bug.isIntroducedByChangeOfExistingClass() != introducedByChange) {
460                     return false;
461                 }
462                 if (newCodeSpecified && newCode != (!bug.isIntroducedByChangeOfExistingClass() && bug.getFirstVersion() != 0)) {
463                     return false;
464                 }
465                 if (removedCodeSpecified && removedCode != (!bug.isRemovedByChangeOfPersistingClass() && bug.isDead())) {
466                     return false;
467                 }
468     
469                 if (bugPattern != null && !bugPattern.matcher(bug.getType()).find()) {
470                     return false;
471                 }
472                 if (classPattern != null && !classPattern.matcher(bug.getPrimaryClass().getClassName()).find()) {
473                     return false;
474                 }
475                 if (callsPattern != null) {
476                     MethodAnnotation m = bug.getAnnotationWithRole(MethodAnnotation.class, MethodAnnotation.METHOD_CALLED);
477                     if (m == null) {
478                         return false;
479                     }
480                     if (!callsPattern.matcher(m.getClassName()).find()  && !callsPattern.matcher(m.getMethodName()).find()) {
481                         return false;
482                     }
483                 }
484     
485                 if (maybeMutatedAsString != null && !(atMutationPoint(bug) && mutationPoints.contains(getBugLocation(bug)))) {
486                     return false;
487                 }
488     
489                 BugPattern thisBugPattern = bug.getBugPattern();
490                 if (!categoryKey.isEmpty() && thisBugPattern != null && !categoryKey.contains(thisBugPattern.getCategory())) {
491                     return false;
492                 }
493                 if (!designationKey.isEmpty() && !designationKey.contains(bug.getUserDesignationKey())) {
494                     return false;
495                 }
496     
497                 if (hashChangedSpecified) {
498                     if (bug.isInstanceHashConsistent() == hashChanged) {
499                         return false;
500                     }
501                 }
502                 if (applySuppressionSpecified && applySuppression && suppressionFilter.match(bug)) {
503                     return false;
504                 }
505                 SourceLineAnnotation primarySourceLineAnnotation = bug.getPrimarySourceLineAnnotation();
506     
507                 if (knownSourceSpecified) {
508                     if (primarySourceLineAnnotation.isUnknown() == knownSource) {
509                         return false;
510                     }
511                 }
512                 if (withSourceSpecified) {
513                     if (sourceSearcher.findSource(primarySourceLineAnnotation) != withSource) {
514                         return false;
515                     }
516                 }
517     
518                 Cloud cloud = collection.getCloud();
519                 if (maxAgeSpecified) {
520                     long firstSeen = cloud.getFirstSeen(bug);
521                     if (!cloud.isInCloud(bug)) {
522                         return false;
523                     }
524                     if (firstSeen < minFirstSeen) {
525                         return false;
526                     }
527                 }
528     
529                 if (notAProblemSpecified && notAProblem != (cloud.getConsensusDesignation(bug).score() < 0)) {
530                     return false;
531                 }
532                 if (shouldFixSpecified && shouldFix != (cloud.getConsensusDesignation(bug).score() > 0)) {
533                     return false;
534                 }
535     
536                 if (sloppyUniqueSpecified) {
537                     boolean unique = uniqueSloppy.add(bug);
538                     if (unique != sloppyUnique) {
539                         return false;
540                     }
541                 }
542     
543                 return true;
544             }

546             private void addDesignationKey(String argument) {
547                 I18N i18n = I18N.instance();
548     
549                 for (String x : argument.split("[,|]")) {
550                     for (String designationKey : i18n.getUserDesignationKeys()) {
551                         if (designationKey.equals(x) || i18n.getUserDesignation(designationKey).equals(x)) {
552                             this.designationKey.add(designationKey);
553                             break;
554                         }
555     
556                     }
557                 }
558             }
559     
560             private void addCategoryKey(String argument) {
561                 DetectorFactoryCollection i18n = DetectorFactoryCollection.instance();
562     
563                 for (String x : argument.split("[,|]")) {
564                     for (BugCategory category : i18n.getBugCategoryObjects()) {
565                         if (category.getAbbrev().equals(x) || category.getCategory().equals(x)) {
566                             this.categoryKey.add(category.getCategory());
567                             break;
568                         }
569                     }
570                 }
571             }

573             private boolean bugLiveAt(BugInstance bug, long now) {
574                 if (now < bug.getFirstVersion()) {
575                     return false;
576                 }
577                 if (bug.isDead() && bug.getLastVersion() < now) {
578                     return false;
579                 }
580                 return true;
581             }

606             @Override
607             protected void handleOptionWithArgument(String option, String argument) throws IOException {
608     
609                 if ("-priority".equals(option) || "-confidence".equals(option)) {
610                     priority = parsePriority(argument);
611                 }
612     
613                 else if ("-maxRank".equals(option)) {
614                     maxRank = Integer.parseInt(argument);
615                 } else if ("-first".equals(option)) {
616                     firstAsString = argument;
617                 } else if ("-maybeMutated".equals(option)) {
618                     maybeMutatedAsString = argument;
619                 } else if ("-last".equals(option)) {
620                     lastAsString = argument;
621                 } else if ("-trimToVersion".equals(option)) {
622                     trimToVersionAsString = argument;
623                 } else if ("-maxDuration".equals(option)) {
624                     duration = Integer.parseInt(argument);
625                 } else if ("-fixed".equals(option)) {
626                     fixedAsString = argument;
627                 } else if ("-after".equals(option)) {
628                     afterAsString = argument;
629                 } else if ("-before".equals(option)) {
630                     beforeAsString = argument;
631                 } else if ("-present".equals(option)) {
632                     presentAsString = argument;
633                 } else if ("-absent".equals(option)) {
634                     absentAsString = argument;
635                 } else if ("-category".equals(option)) {
636                     addCategoryKey(argument);
637                 } else if ("-designation".equals(option)) {
638                     addDesignationKey(argument);
639                 } else if ("-class".equals(option)) {
640                     classPattern = Pattern.compile(argument.replace(',', '|'));
641                 } else if ("-calls".equals(option)) {
642                     callsPattern = Pattern.compile(argument.replace(',', '|'));
643                 } else if ("-bugPattern".equals(option)) {
644                     bugPattern = Pattern.compile(argument);
645                 } else if ("-annotation".equals(option)) {
646                     annotation = argument;
647                 } else if ("-excludeBugs".equals(option)) {
648                     try {
649                         ExcludingHashesBugReporter.addToExcludedInstanceHashes(excludedInstanceHashes, argument);
650                     } catch (DocumentException e) {
651                         throw new IllegalArgumentException("Error processing include file: " + argument, e);
652                     }
653                 } else if ("-include".equals(option)) {
654                     try {
655                         includeFilter.add(new edu.umd.cs.findbugs.filter.Filter(argument));
656                     } catch (FilterException e) {
657                         throw new IllegalArgumentException("Error processing include file: " + argument, e);
658                     }
659                 } else if ("-exclude".equals(option)) {
660                     try {
661                         excludeFilter.add(new edu.umd.cs.findbugs.filter.Filter(argument));
662                     } catch (FilterException e) {
663                         throw new IllegalArgumentException("Error processing include file: " + argument, e);
664                     }
665                 } else if ("-maxAge".equals(option)) {
666                     maxAge = Integer.parseInt(argument);
667                     maxAgeSpecified = true;
668                 } else if ("-hashes".equals(option)) {
669                     hashesFromFile = new HashSet<String>();
670                     BufferedReader in = null;
671                     try {
672                         in = new BufferedReader(UTF8.fileReader(argument));
673                         while (true) {
674                             String h = in.readLine();
675                             if (h == null) {
676                                 break;
677                             }
678                             hashesFromFile.add(h);
679                         }
680                     } catch (IOException e) {
681                         throw new RuntimeException("Error reading hashes from " + argument, e);
682                     } finally {
683                         Util.closeSilently(in);
684                     }
685                 } else {
686                     throw new IllegalArgumentException("can't handle command line argument of " + option);
687                 }
688     
689             }

691             HashSet<String> mutationPoints;
692     
693             /**
694              * Do any prep work needed to perform bug filtering
695              *
696              * @param origCollection
697              */
698             public void getReady(SortedBugCollection origCollection) {
699                 if (maybeMutatedAsString != null) {
700                     HashSet<String> addedIssues = new HashSet<String>();
701                     HashSet<String> removedIssues = new HashSet<String>();
702                     for (BugInstance b : origCollection) {
703                         if (b.getFirstVersion() == maybeMutated) {
704                             addedIssues.add(getBugLocation(b));
705                         } else if (b.getLastVersion() == maybeMutated - 1) {
706                             removedIssues.add(getBugLocation(b));
707                         }
708                     }
709                     addedIssues.remove(null);
710                     addedIssues.retainAll(removedIssues);
711                     mutationPoints = addedIssues;
712                 }
713     
714             }
715     
716             /**
717              * @param b
718              * @return
719              */
720             private boolean atMutationPoint(BugInstance b) {
721                 return b.getFirstVersion() == maybeMutated || b.getLastVersion() == maybeMutated - 1;
722             }
723     
724             /**
725              * @param b
726              * @return
727              */
728             private String getBugLocation(BugInstance b) {
729                 String point;
730                 MethodAnnotation m = b.getPrimaryMethod();
731                 FieldAnnotation f = b.getPrimaryField();
732                 if (m != null) {
733                     point = m.toString();
734                 } else if (f != null) {
735                     point = f.toString();
736                 } else {
737                     point = null;
738                 }
739                 return point;
740             }

744         public static int parsePriority(String argument) {
745             int i = " HMLE".indexOf(argument);
746             if (i == -1) {
747                 i = " 1234".indexOf(argument);
748             }
749             if (i == -1) {
750                 throw new IllegalArgumentException("Bad priority: " + argument);
751             }
752             return i;
753         }

756     
757     
758         public static void main(String[] args) throws Exception {
759             FindBugs.setNoAnalysis();
760             DetectorFactoryCollection.instance();
761     
762             final FilterCommandLine commandLine = new FilterCommandLine();
763     
764             int argCount = commandLine.parse(args, 0, 2, "Usage: " + Filter.class.getName()
765                     + " [options] [<orig results> [<new results]] ");
766             SortedBugCollection origCollection = new SortedBugCollection();
767     
768             if (argCount == args.length) {
769                 origCollection.readXML(System.in);
770             } else {
771                 origCollection.readXML(args[argCount++]);
772             }
773             boolean verbose = argCount < args.length;
774             SortedBugCollection resultCollection = origCollection.createEmptyCollectionWithMetadata();
775             Project project = resultCollection.getProject();
776             int passed = 0;
777             int dropped = 0;
778             resultCollection.setWithMessages(commandLine.withMessages);
779             if (commandLine.hashChangedSpecified) {
780                 origCollection.computeBugHashes();
781             }
782             commandLine.adjustFilter(project, resultCollection);
783             ProjectStats projectStats = resultCollection.getProjectStats();
784             projectStats.clearBugCounts();
785             if (commandLine.classPattern != null) {
786                 projectStats.purgeClassesThatDontMatch(commandLine.classPattern);
787             }
788             sourceSearcher = new SourceSearcher(project);
789     
790             long trimToVersion = -1;
791             if (commandLine.trimToVersionAsString != null) {
792                 Map<String, AppVersion> versions = new HashMap<String, AppVersion>();
793                 SortedMap<Long, AppVersion> timeStamps = new TreeMap<Long, AppVersion>();
794     
795                 for (Iterator<AppVersion> i = origCollection.appVersionIterator(); i.hasNext();) {
796                     AppVersion v = i.next();
797                     versions.put(v.getReleaseName(), v);
798                     timeStamps.put(v.getTimestamp(), v);
799                 }
800                 // add current version to the maps
801                 AppVersion v = resultCollection.getCurrentAppVersion();
802                 versions.put(v.getReleaseName(), v);
803                 timeStamps.put(v.getTimestamp(), v);
804     
805                 trimToVersion = edu.umd.cs.findbugs.workflow.Filter.FilterCommandLine.getVersionNum(versions, timeStamps,
806                         commandLine.trimToVersionAsString, true, v.getSequenceNumber());
807                 if (trimToVersion < origCollection.getSequenceNumber()) {
808                     String name = resultCollection.getAppVersionFromSequenceNumber(trimToVersion).getReleaseName();
809                     long timestamp = resultCollection.getAppVersionFromSequenceNumber(trimToVersion).getTimestamp();
810                     resultCollection.setReleaseName(name);
811                     resultCollection.setTimestamp(timestamp);
812                     resultCollection.trimAppVersions(trimToVersion);
813                 }
814     
815             }
816     
817             if (commandLine.maxAgeSpecified || commandLine.notAProblemSpecified || commandLine.shouldFixSpecified) {
818     
819                 Cloud cloud = origCollection.getCloud();
820                 SigninState signinState = cloud.getSigninState();
821                 if (!signinState.canDownload()) {
822                     disconnect(verbose, commandLine, resultCollection,  cloud.getCloudName() + " state is " + signinState
823                             + "; ignoring filtering options that require cloud access");
824     
825                 } else if (!cloud.waitUntilIssueDataDownloaded(20, TimeUnit.SECONDS)) {
826                     if (verbose) {
827                         System.out.println("Waiting for cloud information required for filtering");
828                     }
829                     if (!cloud.waitUntilIssueDataDownloaded(60, TimeUnit.SECONDS)) {
830                         disconnect(verbose, commandLine, resultCollection,
831                                 "Unable to connect to cloud; ignoring filtering options that require cloud access");
832                     }
833                 }
834             }
835     
836             commandLine.getReady(origCollection);
837     
838             for (BugInstance bug : origCollection.getCollection()) {
839                 if (commandLine.accept(origCollection, bug)) {
840                     if (trimToVersion >= 0) {
841                         if (bug.getFirstVersion() > trimToVersion) {
842                             dropped++;
843                             continue;
844                         } else if (bug.getLastVersion() >= trimToVersion) {
845                             bug.setLastVersion(-1);
846                             bug.setRemovedByChangeOfPersistingClass(false);
847                         }
848                     }
849                     resultCollection.add(bug, false);
850                     passed++;
851                 } else {
852                     dropped++;
853                 }
854             }
855     
856             if (commandLine.purgeHistorySpecified && commandLine.purgeHistory) {
857                 resultCollection.clearAppVersions();
858                 for (BugInstance bug : resultCollection.getCollection()) {
859                     bug.clearHistory();
860                 }
861     
862     
863             }
864             if (verbose) {
865                 System.out.println(passed + " warnings passed through, " + dropped + " warnings dropped");
866             }
867             if (commandLine.withSourceSpecified && commandLine.withSource && !commandLine.dontUpdateStats
868                     && projectStats.hasClassStats()) {
869                 for (PackageStats stats : projectStats.getPackageStats()) {
870                     Iterator<ClassStats> i = stats.getClassStats().iterator();
871                     while (i.hasNext()) {
872                         String className = i.next().getName();
873                         if (sourceSearcher.sourceNotFound.contains(className) || !sourceSearcher.sourceFound.contains(className)
874                                 && !sourceSearcher.findSource(SourceLineAnnotation.createReallyUnknown(className))) {
875                             i.remove();
876                         }
877                     }
878                 }
879     
880             }
881             projectStats.recomputeFromComponents();
882             if (argCount == args.length) {
883                 assert !verbose;
884                 resultCollection.writeXML(System.out);
885             } else {
886                 resultCollection.writeXML(args[argCount++]);
887     
888             }
889     
890         }

894         private static void disconnect(boolean verbose, final FilterCommandLine commandLine, SortedBugCollection resultCollection,
895                 String msg) {
896             if (verbose) {
897                 System.out.println(msg);
898             }
899             resultCollection.addError(msg);
900             commandLine.maxAgeSpecified = commandLine.notAProblemSpecified = commandLine.shouldFixSpecified = false;
901         }
