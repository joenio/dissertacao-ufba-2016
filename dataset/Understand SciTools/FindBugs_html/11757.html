20     package edu.umd.cs.findbugs.util;
21     
22     import java.util.HashMap;
23     import java.util.Map;
24     import java.util.Set;
25     
26     /**
27      * @author pugh
28      */
29     public abstract class MergeMap<K, V> {
30     
31         public static class MinMap<K, V extends Comparable<? super V>> extends MergeMap<K, V> {
32             @Override
33             protected V mergeValues(V oldValue, V newValue) {
34     
35                 if (oldValue.compareTo(newValue) > 0) {
36                     return newValue;
37                 }
38                 return oldValue;
39             }
40         }
41     
42         public static class MaxMap<K, V extends Comparable<? super V>> extends MergeMap<K, V> {
43     
44             @Override
45             protected V mergeValues(V oldValue, V newValue) {
46     
47                 if (oldValue.compareTo(newValue) < 0) {
48                     return newValue;
49                 }
50                 return oldValue;
51             }
52         }
53     
54         final Map<K, V> map;
55     
56         protected abstract V mergeValues(V oldValue, V newValue);
57     
58         public MergeMap() {
59             map = new HashMap<K, V>();
60         }
61     
62         public MergeMap(Map<K, V> map) {
63             this.map = map;
64         }
65     
66         public V put(K k, V v) {
67             V currentValue = map.get(k);
68             if (currentValue == null) {
69                 map.put(k, v);
70                 return v;
71             }
72             V result = mergeValues(currentValue, v);
73             if (currentValue != result) {
74                 map.put(k, v);
75             }
76     
77             return result;
78         }
79     
80         public V get(K k) {
81             return map.get(k);
82         }
83     
84         public boolean containsKey(K k) {
85             return map.containsKey(k);
86         }
87     
88         public Set<Map.Entry<K, V>> entrySet() {
89             return map.entrySet();
90         }
91     
92         public static void main(String args[]) {
93     
94             MergeMap<String, Integer> m = new MaxMap<String, Integer>();
95     
96             m.put("a", 1);
97             m.put("a", 2);
98             m.put("b", 2);
99             m.put("b", 1);
100             System.out.println(m.entrySet());
101     
102         }
103     
104     }
