31     /**
32      * Filter out bugs which fail (match) all filters. This is what happens when you
33      * filter out a branch.
34      */
35     @Deprecated
36     public class StackedFilterMatcher extends FilterMatcher {
37         private static final long serialVersionUID = 3958267780332359162L;
38     
39         private final FilterMatcher[] filters;
40     
41         @Override
42         Sortables getFilterBy() {
43             throw new UnsupportedOperationException("Stacked filter matchers do not filter out a single Sortables, use getFilters()");
44         }
45     
46         @Override
47         String getValue() {
48             throw new UnsupportedOperationException(
49                     "Stacked filter matchers do not filter out a single Sortables's value, use getFilters and getValue individually on returned filters.");
50         }
51     
52         public StackedFilterMatcher(FilterMatcher... filters) {
53             super(null, null);
54             this.filters = filters;
55         }
56     
57         // If only FilterMatcher's setActive were as simple as this one... not.
58         // See BugTreeModel's long ranting comment about filtering to see the reason
59         // for all this
60         // All this does is not force the tree to rebuild when you turn filters for
61         // branches on and off
62         @Override
63         public void setActive(boolean active) {
64             TreeModelEvent event = null;
65             BugTreeModel.TreeModification whatToDo;
66     
67             if (active != this.active) {
68                 if (active == false) {
69                     this.active = active;
70                 }
71     
72                 StackedFilterMatcher theSame = this;
73                 FilterMatcher[] filtersInStack = theSame.getFilters();
74                 ArrayList<Sortables> order = MainFrame.getInstance().getSorter().getOrder();
75                 int sizeToCheck = filtersInStack.length;
76                 if (order.contains(Sortables.DIVIDER)) {
77                     if (order.indexOf(Sortables.DIVIDER) < filtersInStack.length) {
78                         sizeToCheck++;
79                     }
80                 }
81                 List<Sortables> sortablesToCheck = order.subList(0, Math.min(sizeToCheck, order.size()));
82                 Debug.println("Size to check" + sizeToCheck + " checking list" + sortablesToCheck);
83                 Debug.println("checking filters");
84                 ArrayList<String> almostPath = new ArrayList<String>();
85                 ArrayList<Sortables> almostPathSortables = new ArrayList<Sortables>();
86                 for (int x = 0; x < sortablesToCheck.size(); x++) {
87                     Sortables s = sortablesToCheck.get(x);
88                     for (FilterMatcher fm : filtersInStack) {
89                         if (s.equals(fm.getFilterBy())) {
90                             almostPath.add(fm.getValue());
91                             almostPathSortables.add(fm.getFilterBy());
92                         }
93                     }
94                 }
95                 ArrayList<String> finalPath = new ArrayList<String>();
96                 for (int x = 0; x < almostPath.size(); x++) {
97                     Sortables s = almostPathSortables.get(x);
98                     if (MainFrame.getInstance().getSorter().getOrderBeforeDivider().contains(s)) {
99                         finalPath.add(almostPath.get(x));
100                     }
101                 }
102                 try {
103                     if (finalPath.size() == filtersInStack.length) {
104                         if (active == true) {
105                             event = (MainFrame.getInstance().getBugTreeModel()).removeBranch(finalPath);
106                             whatToDo = BugTreeModel.TreeModification.REMOVE;
107                         } else {
108                             event = (MainFrame.getInstance().getBugTreeModel()).insertBranch(finalPath);
109                             whatToDo = BugTreeModel.TreeModification.INSERT;
110                         }
111                     } else {
112                         event = (MainFrame.getInstance().getBugTreeModel()).restructureBranch(finalPath, active);// if
113                         // active
114                         // is
115                         // true,
116                         // this
117                         // removes,
118                         // if
119                         // active
120                         // if
121                         // false,
122                         // it
123                         // inserts
124                         if (active) {
125                             whatToDo = BugTreeModel.TreeModification.REMOVERESTRUCTURE;
126                         } else {
127                             whatToDo = BugTreeModel.TreeModification.INSERTRESTRUCTURE;
128                         }
129                     }
130     
131                     if (active == true) {
132                         this.active = active;
133                     }
134                     (MainFrame.getInstance().getBugTreeModel()).sendEvent(event, whatToDo);
135                 } catch (BranchOperationException e) {
136                     // Another filter already filters out the branch this filter
137                     // would filter out, set active, but dont send any tree model
138                     // events.
139                     this.active = active;
140                 }
141             }
142     
143         }
144     
145         public FilterMatcher[] getFilters() {
146             return filters;
147         }
148     
149         @Override
150         public boolean match(BugInstance bugInstance) {
151             if (!isActive()) {
152                 return true;
153             }
154     
155             for (FilterMatcher i : filters) {
156                 if (i.match(bugInstance)) {
157                     return true;
158                 }
159             }
160     
161             return false;
162         }
163     
164         @Override
165         public String toString() {
166             // return "StackedFilterMatcher: " + Arrays.toString(filters);
167             StringBuilder result = new StringBuilder();
168             for (int i = 0; i < filters.length - 1; i++) {
169                 result.append(filters[i].toString() + (i == filters.length - 2 ? " " : ", "));
170             }
171             if (filters.length > 1) {
172                 result.append("and ");
173             }
174             if (filters.length > 0) {
175                 result.append(filters[filters.length - 1]);
176             }
177             return result.toString();
178         }
179     
180         @Override
181         public boolean equals(Object o) {
182             if (o == null || !(o instanceof StackedFilterMatcher)) {
183                 return false;
184             }
185     
186             FilterMatcher[] mine = new FilterMatcher[filters.length];
187             System.arraycopy(this.filters, 0, mine, 0, mine.length);
188             Arrays.sort(mine);
189     
190             FilterMatcher[] others = new FilterMatcher[((StackedFilterMatcher) o).filters.length];
191             System.arraycopy(((StackedFilterMatcher) o).filters, 0, others, 0, others.length);
192             Arrays.sort(others);
193     
194             return (Arrays.equals(mine, others));
195         }
196     
197         @Override
198         public int hashCode() {
199             int hash = 0;
200             for (FilterMatcher f : filters) {
201                 hash += f.hashCode();
202             }
203             return hash;
204         }
205     
206         public static void main(String[] args) {
207             System.out.println(new StackedFilterMatcher(new FilterMatcher[0]).equals(new StackedFilterMatcher(new FilterMatcher[0])));
208         }
209     }
