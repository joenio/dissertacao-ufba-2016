20     package edu.umd.cs.findbugs.classfile;
21     
22     /**
23      * Interface for a basic code base in which we can look up resources but not
24      * necessarily scan for the list of all resources.
25      *
26      * @author David Hovemeyer
27      */
28     public interface ICodeBase {
29     
30         enum Discovered {
31     
32             /** Codebase was explicitly specified. */
33             SPECIFIED,
34     
35             /** Codebase was discovered as a nested archive in another codebase. */
36             NESTED,
37     
38             /**
39              * Codebase was referenced in the Class-Path attribute of a Jar manifest
40              * of another codebase.
41              */
42             IN_JAR_MANIFEST,
43     
44             /** Codebase was discovered in the system classpath. */
45             IN_SYSTEM_CLASSPATH,
46         }
47         /**
48          * Get the codebase locator describing the location of this codebase.
49          *
50          * @return the ICodeBaseLocator
51          */
52         public ICodeBaseLocator getCodeBaseLocator();
53     
54         /**
55          * Look up a resource in this code base.
56          *
57          * @param resourceName
58          *            name of the resource to look up
59          * @return ICodeBaseEntry representing the resource or null if the resource
60          *         cannot be found in this code base
61          */
62         public ICodeBaseEntry lookupResource(String resourceName);
63     
64         /**
65          * Designate this code base as an application codebase.
66          *
67          * @param isAppCodeBase
68          *            true if this is an application codebase, false if not
69          */
70         public void setApplicationCodeBase(boolean isAppCodeBase);
71     
72         /**
73          * Return whether or not this codebase is an application codebase.
74          *
75          * @return true if this is an application codebase, false if not
76          */
77         public boolean isApplicationCodeBase();
78     
79         /**
80          * Set how this codebase was discovered.
81          *
82          * @param howDiscovered
83          *            one of the constants SPECIFIED, NESTED, IN_JAR_MANIFEST, or
84          *            IN_SYSTEM_CLASSPATH
85          */
86         public void setHowDiscovered(ICodeBase.Discovered howDiscovered);
87     
88         /**
89          * Return how this codebase was discovered.
90          *
91          * @return one of the constants SPECIFIED, NESTED, IN_JAR_MANIFEST, or
92          *         IN_SYSTEM_CLASSPATH
93          */
94         public ICodeBase.Discovered getHowDiscovered();
95     
96         /**
97          * Return whether or not this code base contains any source files.
98          *
99          * @return true if the code base contains source file(s), false if it does
100          *         not contain source files
101          */
102         public boolean containsSourceFiles() throws InterruptedException;
103     
104         /**
105          * Get the filesystem pathname of this codebase.
106          *
107          * @return the filesystem pathname of this codebase, or null if this
108          *         codebase is not accessible via the filesystem
109          */
110         public String getPathName();
111     
112         /**
113          * Set timestamp indicating the most recent time when any of the files in
114          * the codebase were modified.
115          *
116          * @param lastModifiedTime
117          *            timestamp when any codebase files were most-recently modified
118          */
119         public void setLastModifiedTime(long lastModifiedTime);
120     
121         /**
122          * Get timestamp indicating the most recent time when any of the files in
123          * the codebase were modified. This information is only likely to be
124          * accurate if an ICodeBaseIterator has been used to scan the resources in
125          * the codebase (scannable codebases only, obviously).
126          *
127          * @return timestamp when any codebase files were most-recently modified, -1
128          *         if unknown
129          */
130         public long getLastModifiedTime();
131     
132         /**
133          * This method should be called when done using the code base.
134          */
135         public void close();
136     }
