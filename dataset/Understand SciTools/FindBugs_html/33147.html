46     /**
47      *
48      * Lets you choose your new filter, shouldn't let you choose filters that
49      * wouldn't filter anything out including filters that you already have
50      *
51      */
52     @SuppressWarnings("serial")
53     public class NewFilterFrame extends FBDialog {
54     
55         private final JList<String> list = new JList<>();
56     
57         private static NewFilterFrame instance = null;
58     
59         public static void open() {
60             if (instance == null) {
61                 instance = new NewFilterFrame();
62                 instance.setVisible(true);
63                 instance.toFront();
64             }
65         }
66     
67         public static void close() {
68             if (instance != null) {
69                 instance.dispose();
70                 instance = null;
71             }
72         }
73     
74         private NewFilterFrame() {
75             super(PreferencesFrame.getInstance());
76             setContentPane(new JPanel() {
77                 @Override
78                 public Insets getInsets() {
79                     return new Insets(3, 3, 3, 3);
80                 }
81             });
82             setLayout(new BorderLayout());
83     
84             JPanel north = new JPanel();
85             north.setLayout(new BoxLayout(north, BoxLayout.X_AXIS));
86             north.add(Box.createHorizontalStrut(3));
87             north.add(new JLabel(edu.umd.cs.findbugs.L10N.getLocalString("dlg.filter_bugs_lbl", "Filter out all bugs whose") + " "));
88     
89             // Argh divider
90             Sortables[] sortables = MainFrame.getInstance().getAvailableSortables();
91             Sortables[] valuesWithoutDivider = new Sortables[sortables.length - 1];
92             int index = 0;
93     
94             for (int x = 0; x < sortables.length; x++) {
95                 if (sortables[x] != Sortables.DIVIDER) {
96                     valuesWithoutDivider[index] = sortables[x];
97                     index++;
98                 }
99             }
100     
101             final JComboBox<Sortables> comboBox = new JComboBox<>(valuesWithoutDivider);
102             comboBox.setRenderer(new ListCellRenderer<Sortables>() {
103                 final Color SELECTED_BACKGROUND = new Color(183, 184, 204);
104     
105                 @Override
106                 public Component getListCellRendererComponent(JList<? extends Sortables> list, Sortables value, int index, boolean isSelected,
107                         boolean cellHasFocus) {
108                     JLabel result = new JLabel();
109                     result.setFont(result.getFont().deriveFont(Driver.getFontSize()));
110                     result.setOpaque(true);
111                     result.setText(value.toString().toLowerCase());
112                     if (isSelected) {
113                         result.setBackground(SELECTED_BACKGROUND);
114                     }
115                     return result;
116                 }
117             });
118             comboBox.addActionListener(new ActionListener() {
119                 @Override
120                 public void actionPerformed(ActionEvent evt) {
121                     Sortables filterBy = (Sortables) comboBox.getSelectedItem();
122                     String[] rawValues = filterBy.getAllSorted();
123                     String[] listData = new String[rawValues.length];
124                     for (int i = 0; i < listData.length; i++) {
125                         listData[i] = filterBy.formatValue(rawValues[i]);
126                     }
127                     list.setListData(listData);
128                 }
129             });
130             comboBox.validate();
131             north.add(comboBox);
132             north.add(new JLabel(" " + edu.umd.cs.findbugs.L10N.getLocalString("dlg.is", "is") + " "));
133             String[] filterModes = { edu.umd.cs.findbugs.L10N.getLocalString("mode.equal_to", "equal to"),
134                     edu.umd.cs.findbugs.L10N.getLocalString("mode.at_or_after", "at or after"),
135                     edu.umd.cs.findbugs.L10N.getLocalString("mode.at_or_before", "at or before") };
136             final JComboBox<String> filterModeComboBox = new JComboBox<>(filterModes);
137             north.add(filterModeComboBox);
138             north.add(new JLabel(":"));
139             north.add(Box.createHorizontalGlue());
140             JPanel south = new JPanel();
141             JButton okButton = new JButton(edu.umd.cs.findbugs.L10N.getLocalString("dlg.ok_btn", "OK"));
142             okButton.addActionListener(new ActionListener() {
143                 @Override
144                 public void actionPerformed(ActionEvent evt) {
145                     Sortables key = (Sortables) comboBox.getSelectedItem();
146                     String[] values = key.getAllSorted();
147     
148                     ArrayList<SortableValue> filterMe = new ArrayList<SortableValue>();
149                     for (int i : list.getSelectedIndices()) {
150                         filterMe.add(new BugAspects.SortableValue(key, values[i]));
151                     }
152                     try {
153                         MainFrame.getInstance().getProject().getSuppressionFilter().addChild(FilterFactory.makeOrMatcher(filterMe));
154                     } catch (RuntimeException e) {
155                         MainFrame.getInstance().showMessageDialog("Unable to create filter: " + e.getMessage());
156                         close();
157                         return;
158                     }
159                     FilterActivity.notifyListeners(FilterListener.Action.FILTERING, null);
160                     PreferencesFrame.getInstance().updateFilterPanel();
161                     MainFrame.getInstance().setProjectChanged(true);
162                     close();
163                 }
164             });
165             JButton cancelButton = new JButton(edu.umd.cs.findbugs.L10N.getLocalString("dlg.cancel_btn", "Cancel"));
166             cancelButton.addActionListener(new ActionListener() {
167                 @Override
168                 public void actionPerformed(ActionEvent evt) {
169                     close();
170                 }
171             });
172             GuiUtil.addOkAndCancelButtons(south, okButton, cancelButton);
173     
174             list.addMouseListener(new MouseAdapter() {
175                 @Override
176                 public void mouseClicked(MouseEvent evt) {
177                     if (evt.getClickCount() == 2) {
178                         // Dupe from OK button's ActionListener
179                         Sortables key = (Sortables) comboBox.getSelectedItem();
180                         String[] values = key.getAllSorted();
181                         // for (int i : list.getSelectedIndices())
182                         // {
183                         // FilterMatcher fm=new FilterMatcher(key,values[i]);
184                         // if
185                         // (!ProjectSettings.getInstance().getAllMatchers().contains(fm))
186                         // ProjectSettings.getInstance().addFilter(fm);
187                         // }
188                         FilterMatcher[] newFilters = new FilterMatcher[list.getSelectedIndices().length];
189                         for (int i = 0; i < newFilters.length; i++) {
190                             newFilters[i] = new FilterMatcher(key, values[list.getSelectedIndices()[i]]);
191                         }
192                         ProjectSettings.getInstance().addFilters(newFilters);
193                         PreferencesFrame.getInstance().updateFilterPanel();
194                         close();
195                     }
196                 }
197             });
198     
199             add(north, BorderLayout.NORTH);
200             add(Box.createHorizontalStrut(2), BorderLayout.WEST);
201             add(new JScrollPane(list), BorderLayout.CENTER);
202             add(Box.createHorizontalStrut(2), BorderLayout.EAST);
203             add(south, BorderLayout.SOUTH);
204     
205             // Populate the box with initial values
206             comboBox.getActionListeners()[0].actionPerformed(null);
207     
208             setDefaultCloseOperation(DO_NOTHING_ON_CLOSE);
209             addWindowListener(new WindowAdapter() {
210                 @Override
211                 public void windowClosing(WindowEvent arg0) {
212                     close();
213                 }
214             });
215     
216             setTitle(edu.umd.cs.findbugs.L10N.getLocalString("dlg.create_new_filter_ttl", "Create New Filter"));
217             pack();
218         }
219     
220         public static void main(String[] args) {
221             new NewFilterFrame().setDefaultCloseOperation(EXIT_ON_CLOSE);
222         }
223     }
