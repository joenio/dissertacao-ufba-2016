20     package edu.umd.cs.findbugs.ba;
21     
22     import org.apache.bcel.generic.InstructionHandle;
23     import org.apache.bcel.generic.MethodGen;
24     
25     /**
26      * Map BCEL InstructionHandles to some kind of value type.
27      *
28      * @author David Hovemeyer
29      */
30     public class InstructionHandleMap<ValueType> {
31         private final Object[] map;
32     
33         public InstructionHandleMap(MethodGen methodGen) {
34             map = new Object[methodGen.getMethod().getCode().getLength()];
35         }
36     
37         @SuppressWarnings("unchecked")
38         public ValueType get(InstructionHandle handle) {
39             return (ValueType) map[handle.getPosition()];
40         }
41     
42         public ValueType put(InstructionHandle handle, ValueType value) {
43             ValueType old = get(handle);
44             map[handle.getPosition()] = value;
45             return old;
46         }
47     }
