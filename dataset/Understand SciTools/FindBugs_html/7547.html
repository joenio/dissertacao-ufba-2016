20     package edu.umd.cs.findbugs.detect;
21     
22     import java.util.Collection;
23     import java.util.HashSet;
24     import java.util.Iterator;
25     import java.util.LinkedList;
26     import java.util.Map;
27     import java.util.Set;
28     
29     import org.apache.bcel.classfile.Code;
30     import org.apache.bcel.classfile.ElementValue;
31     import org.apache.bcel.classfile.Field;
32     import org.apache.bcel.classfile.JavaClass;
33     import org.apache.bcel.classfile.Method;
34     
35     import edu.umd.cs.findbugs.BugInstance;
36     import edu.umd.cs.findbugs.BugReporter;
37     import edu.umd.cs.findbugs.BytecodeScanningDetector;
38     import edu.umd.cs.findbugs.FieldAnnotation;
39     import edu.umd.cs.findbugs.LocalVariableAnnotation;
40     import edu.umd.cs.findbugs.StatelessDetector;
41     import edu.umd.cs.findbugs.ba.AnalysisContext;
42     import edu.umd.cs.findbugs.ba.FieldSummary;
43     import edu.umd.cs.findbugs.ba.XFactory;
44     import edu.umd.cs.findbugs.ba.XField;
45     
46     public class FindUninitializedGet extends BytecodeScanningDetector implements StatelessDetector {
47         Set<FieldAnnotation> initializedFields = new HashSet<FieldAnnotation>();
48     
49         Set<FieldAnnotation> declaredFields = new HashSet<FieldAnnotation>();
50     
51         Set<FieldAnnotation> containerFields = new HashSet<FieldAnnotation>();
52     
53         Collection<BugInstance> pendingBugs = new LinkedList<BugInstance>();
54     
55         BugInstance uninitializedFieldReadAndCheckedForNonnull;
56     
57         boolean inConstructor;
58     
59         boolean thisOnTOS = false;
60     
61         private final BugReporter bugReporter;
62     
63         public FindUninitializedGet(BugReporter bugReporter) {
64             this.bugReporter = bugReporter;
65         }
66     
67         @Override
68         public void visit(JavaClass obj) {
69             pendingBugs.clear();
70             declaredFields.clear();
71             containerFields.clear();
72             super.visit(obj);
73         }
74     
75         @Override
76         public void visit(Field obj) {
77             super.visit(obj);
78             FieldAnnotation f = FieldAnnotation.fromVisitedField(this);
79             declaredFields.add(f);
80     
81         }
82     
83         @Override
84         public void visitAnnotation(String annotationClass, Map<String, ElementValue> map, boolean runtimeVisible) {
85             if (!visitingField()) {
86                 return;
87             }
88             if (UnreadFields.isInjectionAttribute(annotationClass)) {
89                 containerFields.add(FieldAnnotation.fromVisitedField(this));
90             }
91     
92         }
93     
94         @Override
95         public void visit(Method obj) {
96             super.visit(obj);
97             initializedFields.clear();
98     
99             thisOnTOS = false;
100             inConstructor = "<init>".equals(getMethodName()) && getMethodSig().indexOf(getClassName()) == -1;
101     
102         }
103     
104         @Override
105         public void visit(Code obj) {
106             if (!inConstructor) {
107                 return;
108             }
109             uninitializedFieldReadAndCheckedForNonnull = null;
110             super.visit(obj);
111             for (BugInstance bug : pendingBugs) {
112                 bugReporter.reportBug(bug);
113             }
114             pendingBugs.clear();
115         }
116     
117         @Override
118         public void sawBranchTo(int target) {
119             Iterator<BugInstance> i = pendingBugs.iterator();
120             while (i.hasNext()) {
121                 BugInstance bug = i.next();
122                 if (bug.getPrimarySourceLineAnnotation().getStartBytecode() >= target) {
123                     i.remove();
124                 }
125             }
126         }
127     
128         @Override
129         public void sawOpcode(int seen) {
130             if (!inConstructor) {
131                 return;
132             }
133             if (uninitializedFieldReadAndCheckedForNonnull != null) {
134                 if (seen == NEW && getClassConstantOperand().endsWith("Exception")) {
135                     uninitializedFieldReadAndCheckedForNonnull.raisePriority();
136                 }
137                 uninitializedFieldReadAndCheckedForNonnull = null;
138             }
139     
140             if (seen == ALOAD_0) {
141                 thisOnTOS = true;
142                 return;
143             }
144     
145             if (seen == PUTFIELD && getClassConstantOperand().equals(getClassName())) {
146                 initializedFields.add(FieldAnnotation.fromReferencedField(this));
147             } else if (thisOnTOS && seen == GETFIELD && getClassConstantOperand().equals(getClassName())) {
148                 UnreadFieldsData unreadFields = AnalysisContext.currentAnalysisContext().getUnreadFieldsData();
149                 XField xField = XFactory.createReferencedXField(this);
150                 FieldAnnotation f = FieldAnnotation.fromReferencedField(this);
151                 int nextOpcode = 0xff & codeBytes[getPC() + 3];
152                 if (nextOpcode != POP && !initializedFields.contains(f) && declaredFields.contains(f) && !containerFields.contains(f)
153                         && !unreadFields.isContainerField(xField)) {
154                     // System.out.println("Next opcode: " +
155                     // OPCODE_NAMES[nextOpcode]);
156                     LocalVariableAnnotation possibleTarget = LocalVariableAnnotation.findMatchingIgnoredParameter(getClassContext(),
157                             getMethod(), getNameConstantOperand(), xField.getSignature());
158                     if (possibleTarget == null) {
159                         possibleTarget = LocalVariableAnnotation.findUniqueBestMatchingParameter(getClassContext(), getMethod(),
160                                 getNameConstantOperand(), getSigConstantOperand());
161                     }
162                     int priority = unreadFields.getReadFields().contains(xField) ? NORMAL_PRIORITY : LOW_PRIORITY;
163                     boolean priorityLoweredBecauseOfIfNonnullTest = false;
164                     if (possibleTarget != null) {
165                         priority--;
166                     } else {
167                         FieldSummary fieldSummary = AnalysisContext.currentAnalysisContext().getFieldSummary();
168                         if (fieldSummary.callsOverriddenMethodsFromSuperConstructor(getClassDescriptor())) {
169                             priority++;
170                         } else if (nextOpcode == IFNONNULL) {
171                             priority++;
172                             priorityLoweredBecauseOfIfNonnullTest = true;
173                         }
174                     }
175     
176                     BugInstance bug = new BugInstance(this, "UR_UNINIT_READ", priority).addClassAndMethod(this).addField(f)
177                             .addOptionalAnnotation(possibleTarget).addSourceLine(this);
178                     pendingBugs.add(bug);
179                     if (priorityLoweredBecauseOfIfNonnullTest) {
180                         uninitializedFieldReadAndCheckedForNonnull = bug;
181                     }
182                     initializedFields.add(f);
183                 }
184             } else if ((seen == INVOKESPECIAL && !("<init>".equals(getNameConstantOperand()) && !getClassConstantOperand().equals(
185                     getClassName())))
186                     || (seen == INVOKESTATIC && "doPrivileged".equals(getNameConstantOperand()) && "java/security/AccessController".equals(getClassConstantOperand()))
187                             || (seen == INVOKEVIRTUAL && getClassConstantOperand().equals(getClassName()))
188                             || (seen == INVOKEVIRTUAL && "start".equals(getNameConstantOperand()))) {
189     
190                 inConstructor = false;
191             }
192     
193             thisOnTOS = false;
194         }
195     }
