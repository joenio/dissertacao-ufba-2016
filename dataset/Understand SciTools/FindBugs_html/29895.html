20     package edu.umd.cs.findbugs.filter;
21     
22     import java.io.IOException;
23     import java.util.Iterator;
24     
25     import edu.umd.cs.findbugs.BugInstance;
26     import edu.umd.cs.findbugs.xml.XMLOutput;
27     
28     public class OrMatcher extends CompoundMatcher {
29     
30         @Override
31         public boolean match(BugInstance bugInstance) {
32             Iterator<Matcher> i = childIterator();
33             while (i.hasNext()) {
34                 Matcher child = i.next();
35                 if (child.match(bugInstance)) {
36                     return true;
37                 }
38             }
39             return false;
40         }
41     
42         @Override
43         public void writeXML(XMLOutput xmlOutput, boolean disabled) throws IOException {
44             if (numberChildren() == 1) {
45                 childIterator().next().writeXML(xmlOutput, false);
46                 return;
47             }
48             xmlOutput.startTag("Or");
49             if (disabled) {
50                 xmlOutput.addAttribute("disabled", "true");
51             }
52             xmlOutput.stopTag(false);
53             writeChildrenXML(xmlOutput);
54             xmlOutput.closeTag("Or");
55         }
56     
57         @Override
58         public String toString() {
59             if (numberChildren() == 1) {
60                 return super.toString();
61             }
62             return "Or(" + super.toString() + ")";
63         }
64     
65     }
