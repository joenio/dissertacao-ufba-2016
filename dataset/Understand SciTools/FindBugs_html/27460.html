134         public static LocalVariableAnnotation getLocalVariableAnnotation(Method method, Location location, IndexedInstruction ins) {
135             int local = ins.getIndex();
136             InstructionHandle handle = location.getHandle();
137             int position1 = handle.getNext().getPosition();
138             int position2 = handle.getPosition();
139             return getLocalVariableAnnotation(method, local, position1, position2);
140         }

142         public static LocalVariableAnnotation getLocalVariableAnnotation(Method method, int local, int position1, int position2) {
143     
144             LocalVariableTable localVariableTable = method.getLocalVariableTable();
145             String localName = "?";
146             if (localVariableTable != null) {
147                 LocalVariable lv1 = localVariableTable.getLocalVariable(local, position1);
148                 if (lv1 == null) {
149                     lv1 = localVariableTable.getLocalVariable(local, position2);
150                     position1 = position2;
151                 }
152                 if (lv1 != null) {
153                     localName = lv1.getName();
154                 } else {
155                     for (LocalVariable lv : localVariableTable.getLocalVariableTable()) {
156                         if (lv.getIndex() == local) {
157                             if (!"?".equals(localName) && !localName.equals(lv.getName())) {
158                                 // not a single consistent name
159                                 localName = "?";
160                                 break;
161                             }
162                             localName = lv.getName();
163                         }
164                     }
165                 }
166             }
167             LineNumberTable lineNumbers = method.getLineNumberTable();
168             if (lineNumbers == null) {
169                 return new LocalVariableAnnotation(localName, local, position1);
170             }
171             int line = lineNumbers.getSourceLine(position1);
172             return new LocalVariableAnnotation(localName, local, position1, line);
173         }

175         /**
176          * Get a local variable annotation describing a parameter.
177          *
178          * @param method
179          *            a Method
180          * @param local
181          *            the local variable containing the parameter
182          * @return LocalVariableAnnotation describing the parameter
183          */
184         public static LocalVariableAnnotation getParameterLocalVariableAnnotation(Method method, int local) {
185             LocalVariableAnnotation lva = getLocalVariableAnnotation(method, local, 0, 0);
186             if (lva.isNamed()) {
187                 lva.setDescription(LocalVariableAnnotation.PARAMETER_NAMED_ROLE);
188             } else {
189                 lva.setDescription(LocalVariableAnnotation.PARAMETER_ROLE);
190             }
191             return lva;
192         }

208         @Override
209         public String format(String key, ClassAnnotation primaryClass) {
210             // System.out.println("format: " + key + " reg: " + register + " name: "
211             // + value);
212             if ("hash".equals(key)) {
213                 if (register < 0) {
214                     return "??";
215                 }
216                 return name;
217             }
218             if (register < 0) {
219                 return "?";
220             }
221             if ("register".equals(key)) {
222                 return String.valueOf(register);
223             } else if ("pc".equals(key)) {
224                 return String.valueOf(pc);
225             } else if ("name".equals(key) || "givenClass".equals(key)) {
226                 return name;
227             } else if (!"?".equals(name)) {
228                 return name;
229             }
230             return "$L" + register;
231         }

343         public static @CheckForNull
344         LocalVariableAnnotation findMatchingIgnoredParameter(ClassContext classContext, Method method, String name, String signature) {
345             try {
346                 Dataflow<BitSet, LiveLocalStoreAnalysis> llsaDataflow = classContext.getLiveLocalStoreDataflow(method);
347                 CFG cfg;
348     
349                 cfg = classContext.getCFG(method);
350                 LocalVariableAnnotation match = null;
351                 int lowestCost = Integer.MAX_VALUE;
352                 BitSet liveStoreSetAtEntry = llsaDataflow.getAnalysis().getResultFact(cfg.getEntry());
353                 int localsThatAreParameters = PreorderVisitor.getNumberArguments(method.getSignature());
354                 int startIndex = 0;
355                 if (!method.isStatic()) {
356                     startIndex = 1;
357                 }
358                 SignatureParser parser = new SignatureParser(method.getSignature());
359                 Iterator<String> signatureIterator = parser.parameterSignatureIterator();
360                 for (int i = startIndex; i < localsThatAreParameters + startIndex; i++) {
361                     String sig = signatureIterator.next();
362                     if (!liveStoreSetAtEntry.get(i) && signature.equals(sig)) {
363                         // parameter isn't live and signatures match
364                         LocalVariableAnnotation potentialMatch = LocalVariableAnnotation.getLocalVariableAnnotation(method, i, 0, 0);
365                         potentialMatch.setDescription(DID_YOU_MEAN_ROLE);
366                         if (!potentialMatch.isNamed()) {
367                             return potentialMatch;
368                         }
369                         int distance = EditDistance.editDistance(name, potentialMatch.getName());
370                         if (distance < lowestCost) {
371                             match = potentialMatch;
372                             match.setDescription(DID_YOU_MEAN_ROLE);
373                             lowestCost = distance;
374                         } else if (distance == lowestCost) {
375                             // not unique best match
376                             match = null;
377                         }
378     
379                     }
380                 }
381                 return match;
382             } catch (DataflowAnalysisException e) {
383                 AnalysisContext.logError("", e);
384             } catch (CFGBuilderException e) {
385                 AnalysisContext.logError("", e);
386             }
387             return null;
388         }

390         public static @CheckForNull
391         LocalVariableAnnotation findUniqueBestMatchingParameter(ClassContext classContext, Method method, String name,
392                 String signature) {
393             LocalVariableAnnotation match = null;
394             int localsThatAreParameters = PreorderVisitor.getNumberArguments(method.getSignature());
395             int startIndex = 0;
396             if (!method.isStatic()) {
397                 startIndex = 1;
398             }
399             SignatureParser parser = new SignatureParser(method.getSignature());
400             Iterator<String> signatureIterator = parser.parameterSignatureIterator();
401             int lowestCost = Integer.MAX_VALUE;
402             for (int i = startIndex; i < localsThatAreParameters + startIndex; i++) {
403                 String sig = signatureIterator.next();
404                 if (signature.equals(sig)) {
405                     LocalVariableAnnotation potentialMatch = LocalVariableAnnotation.getLocalVariableAnnotation(method, i, 0, 0);
406                     if (!potentialMatch.isNamed()) {
407                         continue;
408                     }
409                     int distance = EditDistance.editDistance(name, potentialMatch.getName());
410                     if (distance < lowestCost) {
411                         match = potentialMatch;
412                         match.setDescription(DID_YOU_MEAN_ROLE);
413                         lowestCost = distance;
414                     } else if (distance == lowestCost) {
415                         // not unique best match
416                         match = null;
417                     }
418                     // signatures match
419                 }
420             }
421             if (lowestCost < 5) {
422                 return match;
423             }
424             return null;
425         }
