20     package edu.umd.cs.findbugs.ba;
21     
22     import java.util.Comparator;
23     
24     /**
25      * A BlockOrder for visiting the blocks of a CFG in the reverse of the order in
26      * which they are finished in a depth first search. This is the most efficient
27      * visitation order for forward dataflow analyses.
28      *
29      * @see BlockOrder
30      * @see DepthFirstSearch
31      * @see CFG
32      * @see BasicBlock
33      */
34     public class ReversePostOrder extends AbstractBlockOrder {
35         /**
36          * A Comparator to order the blocks in the reverse of the order in which
37          * they would be finished by a depth first search.
38          */
39         private static class ReversePostfixComparator implements Comparator<BasicBlock> {
40             private final DepthFirstSearch dfs;
41     
42             public ReversePostfixComparator(DepthFirstSearch dfs) {
43                 this.dfs = dfs;
44             }
45     
46             @Override
47             public int compare(BasicBlock aa, BasicBlock bb) {
48                 return dfs.getFinishTime(bb) - dfs.getFinishTime(aa);
49             }
50         }
51     
52         /**
53          * Constructor.
54          *
55          * @param cfg
56          *            the CFG for the method
57          * @param dfs
58          *            the DepthFirstSearch on the method
59          */
60         public ReversePostOrder(CFG cfg, DepthFirstSearch dfs) {
61             super(cfg, new ReversePostfixComparator(dfs));
62         }
63     }
