31     /**
32      * Dataflow analysis to find constant values.
33      *
34      * @see edu.umd.cs.findbugs.ba.constant.Constant
35      * @author David Hovemeyer
36      */
37     public class ConstantAnalysis extends FrameDataflowAnalysis<Constant, ConstantFrame> {
38         private final MethodGen methodGen;
39     
40         private final ConstantFrameModelingVisitor visitor;
41     
42         public ConstantAnalysis(MethodGen methodGen, DepthFirstSearch dfs) {
43             super(dfs);
44             this.methodGen = methodGen;
45             this.visitor = new ConstantFrameModelingVisitor(methodGen.getConstantPool());
46         }
47     
48         @Override
49         public ConstantFrame createFact() {
50             return new ConstantFrame(methodGen.getMaxLocals());
51         }
52     
53         @Override
54         public void initEntryFact(ConstantFrame frame) {
55             frame.setValid();
56             frame.clearStack();
57             int numSlots = frame.getNumSlots();
58             for (int i = 0; i < numSlots; ++i) {
59                 frame.setValue(i, Constant.NOT_CONSTANT);
60             }
61         }
62     
63         @Override
64         public void transferInstruction(InstructionHandle handle, BasicBlock basicBlock, ConstantFrame frame)
65                 throws DataflowAnalysisException {
66             visitor.setFrameAndLocation(frame, new Location(handle, basicBlock));
67             visitor.analyzeInstruction(handle.getInstruction());
68         }
69     
70         @Override
71         public void meetInto(ConstantFrame fact, Edge edge, ConstantFrame result) throws DataflowAnalysisException {
72     
73             if (fact.isValid()) {
74                 ConstantFrame tmpFact = null;
75     
76                 if (edge.isExceptionEdge()) {
77                     tmpFact = modifyFrame(fact, null);
78                     tmpFact.clearStack();
79                     tmpFact.pushValue(Constant.NOT_CONSTANT);
80                 }
81     
82                 if (tmpFact != null) {
83                     fact = tmpFact;
84                 }
85             }
86     
87             mergeInto(fact, result);
88         }
89     
90         @Override
91         protected void mergeValues(ConstantFrame otherFrame, ConstantFrame resultFrame, int slot) throws DataflowAnalysisException {
92             Constant value = Constant.merge(resultFrame.getValue(slot), otherFrame.getValue(slot));
93             resultFrame.setValue(slot, value);
94         }
