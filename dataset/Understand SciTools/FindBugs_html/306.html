126         @Override
127         public void visitClass(ClassDescriptor classDescriptor) throws CheckedAnalysisException {
128             IAnalysisCache analysisCache = Global.getAnalysisCache();
129     
130             ObligationFactory factory = database.getFactory();
131     
132             JavaClass jclass = analysisCache.getClassAnalysis(JavaClass.class, classDescriptor);
133             for (Constant c : jclass.getConstantPool().getConstantPool()) {
134                 if (c instanceof ConstantNameAndType) {
135                     ConstantNameAndType cnt = (ConstantNameAndType) c;
136                     String signature = cnt.getSignature(jclass.getConstantPool());
137                     if (factory.signatureInvolvesObligations(signature)) {
138                         super.visitClass(classDescriptor);
139                         return;
140                     }
141                 } else if (c instanceof ConstantClass) {
142                     String className = ((ConstantClass) c).getBytes(jclass.getConstantPool());
143                     if (factory.signatureInvolvesObligations(className)) {
144                         super.visitClass(classDescriptor);
145                         return;
146                     }
147                 }
148             }
149             if (DEBUG) {
150                 System.out.println(classDescriptor + " isn't interesting for obligation analysis");
151             }
152         }

166             Obligation consumed, produced;

169                 this.consumed = consumed;
170                 this.produced = produced;

173             /**
174              * Determine whether the state has "balanced" obligation counts for the
175              * consumed and produced Obligation types.
176              *
177              * @param state
178              *            a State
179              * @return true if the obligation counts are balanced, false otherwise
180              */
181             private boolean balanced(State state) {
182                 int consumedCount = state.getObligationSet().getCount(consumed.getId());
183                 int producedCount = state.getObligationSet().getCount(produced.getId());
184                 return (consumedCount + producedCount == 0) && (consumedCount == 1 || producedCount == 1);
185             }
186     
187             private boolean matches(Obligation possiblyLeakedObligation) {
188                 return consumed.equals(possiblyLeakedObligation) || produced.equals(possiblyLeakedObligation);
189             }
190     
191             @Override
192             public String toString() {
193                 return consumed + " -> " + produced;
194             }
195         }

224             public void analyzeMethod() throws CheckedAnalysisException {
225                 if (DEBUG_METHOD != null && !methodDescriptor.getName().equals(DEBUG_METHOD)) {
226                     return;
227                 }
228     
229                 if (DEBUG) {
230                     System.out.println("*** Analyzing method " + methodDescriptor);
231                 }
232     
233                 xmethod = XFactory.createXMethod(methodDescriptor);
234                 analysisCache = Global.getAnalysisCache();
235     
236                 //
237                 // Execute the obligation dataflow analysis
238                 //
239                 try {
240                     dataflow = analysisCache.getMethodAnalysis(ObligationDataflow.class, methodDescriptor);
241                 } catch (ObligationAcquiredOrReleasedInLoopException e) {
242                     // It is not possible to analyze this method.
243                     if (DEBUG) {
244                         System.out.println("FindUnsatisifedObligation: " + methodDescriptor + ": " + e.getMessage());
245                     }
246                     return;
247                 }
248     
249                 //
250                 // Additional analyses
251                 // needed these to apply the false-positive
252                 // suppression heuristics.
253                 //
254                 cpg = analysisCache.getClassAnalysis(ConstantPoolGen.class, methodDescriptor.getClassDescriptor());
255                 typeDataflow = analysisCache.getMethodAnalysis(TypeDataflow.class, methodDescriptor);
256                 subtypes2 = Global.getAnalysisCache().getDatabase(Subtypes2.class);
257     
258                 //
259                 // Main loop: looking at the StateSet at the exit block of the CFG,
260                 // see if there are any states with nonempty obligation sets.
261                 //
262                 Map<Obligation, State> leakedObligationMap = new HashMap<Obligation, State>();
263                 StateSet factAtExit = dataflow.getResultFact(cfg.getExit());
264                 for (Iterator<State> i = factAtExit.stateIterator(); i.hasNext();) {
265                     State state = i.next();
266                     checkStateForLeakedObligations(state, leakedObligationMap);
267                 }
268     
269                 //
270                 // Report a separate BugInstance for each Obligation,State pair.
271                 // (Two different obligations may be leaked in the same state.)
272                 //
273                 for (Map.Entry<Obligation, State> entry : leakedObligationMap.entrySet()) {
274                     Obligation obligation = entry.getKey();
275                     State state = entry.getValue();
276                     reportWarning(obligation, state, factAtExit);
277                 }
278                 // TODO: closing of nonexistent resources
279     
280             }

282             private void checkStateForLeakedObligations(State state, Map<Obligation, State> leakedObligationMap)
283                     throws IllegalStateException {
284                 if (DEBUG) {
285                     Path path = state.getPath();
286                     if (path.getLength() > 0 && path.getBlockIdAt(path.getLength() - 1) != cfg.getExit().getLabel()) {
287                         throw new IllegalStateException("path " + path + " at cfg exit has no label for exit block");
288                     }
289                 }
290     
291                 for (int id = 0; id < database.getFactory().getMaxObligationTypes(); ++id) {
292                     Obligation obligation = database.getFactory().getObligationById(id);
293                     // If the raw count produced by the analysis
294                     // for this obligation type is 0,
295                     // assume everything is ok on this state's path.
296                     int rawLeakCount = state.getObligationSet().getCount(id);
297                     if (rawLeakCount == 0) {
298                         continue;
299                     }
300     
301                     // Apply the false-positive suppression heuristics
302                     int leakCount = getAdjustedLeakCount(state, id);
303     
304     
305                     if (leakCount > 0) {
306                         leakedObligationMap.put(obligation, state);
307                     }
308                     // TODO: if the leak count is less than 0, then a nonexistent
309                     // resource was closed
310                 }
311             }

313             private void reportWarning(Obligation obligation, State state, StateSet factAtExit) {
314                 String className = obligation.getClassName();
315     
316                 if (methodDescriptor.isStatic() && "main".equals(methodDescriptor.getName())
317                         && "([Ljava/lang/String;)V".equals(methodDescriptor.getSignature())
318                         && (className.contains("InputStream") || className.contains("Reader") || factAtExit.isOnExceptionPath())) {
319                     // Don't report unclosed input streams and readers in main()
320                     // methods
321                     return;
322                 }
323     
324                 if (methodDescriptor.getName().equals("<init>")) {
325                     try {
326     
327                         if (subtypes2.isSubtype(methodDescriptor.getClassDescriptor(), DescriptorFactory.createClassDescriptorFromDottedClassName(obligation.getClassName()))) {
328                             return;
329                         }
330     
331                     } catch (Exception e) {
332                         AnalysisContext.logError("huh", e);
333                     }
334                 }
335                 String bugPattern = factAtExit.isOnExceptionPath() ? "OBL_UNSATISFIED_OBLIGATION_EXCEPTION_EDGE" : "OBL_UNSATISFIED_OBLIGATION";
336                 BugInstance bugInstance = new BugInstance(FindUnsatisfiedObligation.this, bugPattern,
337                         NORMAL_PRIORITY).addClassAndMethod(methodDescriptor).addClass(className).describe("CLASS_REFTYPE");
338     
339                 // Report how many instances of the obligation are remaining
340                 bugInstance.addInt(state.getObligationSet().getCount(obligation.getId())).describe(
341                         IntAnnotation.INT_OBLIGATIONS_REMAINING);
342     
343                 // Add source line information
344                 annotateWarningWithSourceLineInformation(state, obligation, bugInstance);
345     
346                 if (REPORT_OBLIGATION_SET) {
347                     bugInstance.addString(state.getObligationSet().toString()).describe(StringAnnotation.REMAINING_OBLIGATIONS_ROLE);
348                 }
349     
350                 bugReporter.reportBug(bugInstance);
351             }
352     
353             private void annotateWarningWithSourceLineInformation(State state, Obligation obligation, BugInstance bugInstance) {
354                 // The reportPath() method currently does all reporting
355                 // of source line information.
356                 if (REPORT_PATH) {
357                     reportPath(bugInstance, obligation, state);
358                 }
359             }

361             /**
362              * Helper class to apply the false-positive suppression heuristics along
363              * a Path where an obligation leak might have occurred.
364              */
365             private class PostProcessingPathVisitor implements PathVisitor {
366                 Obligation possiblyLeakedObligation;
367     
368                 State state;
369     
370                 int adjustedLeakCount;
371     
372                 BasicBlock curBlock;
373     
374                 boolean couldNotAnalyze;
375     
376                 List<PossibleObligationTransfer> transferList;
377     
378                 public PostProcessingPathVisitor(Obligation possiblyLeakedObligation/*
379                  * ,
380                  * int
381                  * initialLeakCount
382                  */, State state) {
383                     this.possiblyLeakedObligation = possiblyLeakedObligation;
384                     this.state = state;
385                     this.adjustedLeakCount = state.getObligationSet().getCount(possiblyLeakedObligation.getId());
386                     if (COMPUTE_TRANSFERS) {
387                         this.transferList = new LinkedList<PossibleObligationTransfer>();
388                     }
389                 }
390     
391                 public int getAdjustedLeakCount() {
392                     return adjustedLeakCount;
393                 }
394     
395                 public boolean couldNotAnalyze() {
396                     return couldNotAnalyze;
397                 }
398     
399                 @Override
400                 public void visitBasicBlock(BasicBlock basicBlock) {
401                     curBlock = basicBlock;
402     
403                     if (COMPUTE_TRANSFERS && basicBlock == cfg.getExit()) {
404                         // We're at the CFG exit.
405     
406                         if (adjustedLeakCount == 1) {
407                             applyPossibleObligationTransfers();
408                         }
409                     }
410                 }

412                 @Override
413                 public void visitInstructionHandle(InstructionHandle handle) {
414                     try {
415                         Instruction ins = handle.getInstruction();
416                         short opcode = ins.getOpcode();
417                         if (DEBUG) {
418                             System.out.printf("%3d %s%n", handle.getPosition(),Constants.OPCODE_NAMES[opcode]);
419                         }
420     
421                         if (opcode == Constants.PUTFIELD || opcode == Constants.PUTSTATIC || opcode == Constants.ARETURN) {
422                             //
423                             // A value is being assigned to a field or returned from
424                             // the method.
425                             //
426                             Location loc = new Location(handle, curBlock);
427                             TypeFrame typeFrame = typeDataflow.getFactAtLocation(loc);
428                             if (!typeFrame.isValid()) {
429                                 // dead code?
430                                 couldNotAnalyze = true;
431                             }
432                             Type tosType = typeFrame.getTopValue();
433                             if (tosType instanceof ObjectType
434                                     && isPossibleInstanceOfObligationType(subtypes2, (ObjectType) tosType,
435                                             possiblyLeakedObligation.getType())) {
436                                 // Remove one obligation of this type
437                                 adjustedLeakCount--;
438                                 if (DEBUG) {
439                                     System.out.println("removing obligation to close " + tosType + " at " + handle.getPosition());
440                                 }
441                             }
442                         }
443     
444                         if (COMPUTE_TRANSFERS && ins instanceof InvokeInstruction) {
445                             checkForPossibleObligationTransfer((InvokeInstruction) ins, handle);
446                         }
447                     } catch (ClassNotFoundException e) {
448                         bugReporter.reportMissingClass(e);
449                         couldNotAnalyze = true;
450                     } catch (DataflowAnalysisException e) {
451                         couldNotAnalyze = true;
452                     }
453                 }

455                 private void applyPossibleObligationTransfers() {
456                     //
457                     // See if we recorded any possible obligation transfers
458                     // that might have created a "wrapper" object.
459                     // In many cases, it is correct to close either
460                     // the "wrapped" or "wrapper" object.
461                     // So, if we see a possible transfer, and we see
462                     // a +1/-1 obligation count for the pair
463                     // (consumed and produced obligation types),
464                     // rather than 0/0,
465                     // then we will assume that which resource was closed
466                     // (wrapper or wrapped) was the opposite of what
467                     // we expected.
468                     //
469                     for (PossibleObligationTransfer transfer : transferList) {
470                         if (DEBUG_FP) {
471                             System.out.println("Checking possible transfer " + transfer + "...");
472                         }
473     
474                         boolean matches = transfer.matches(possiblyLeakedObligation);
475     
476                         if (DEBUG_FP) {
477                             System.out.println("  matches: " + possiblyLeakedObligation);
478                         }
479     
480                         if (matches) {
481                             boolean balanced = transfer.balanced(state);
482                             if (DEBUG_FP) {
483                                 System.out.println("  balanced: " + balanced + " in " + state.getObligationSet());
484                             }
485                             if (balanced) {
486                                 if (DEBUG_FP) {
487                                     System.out.println("  Suppressing path because " + "a transfer appears to result in balanced "
488                                             + "outstanding obligations");
489                                 }
490     
491                                 adjustedLeakCount = 0;
492                                 break;
493                             }
494                         }
495                     }
496                 }

498                 private void checkForPossibleObligationTransfer(InvokeInstruction inv, InstructionHandle handle)
499                         throws ClassNotFoundException {
500                     //
501                     // We will assume that a method invocation might transfer
502                     // an obligation from one type to another if
503                     // 1. either
504                     // - it's a constructor where the constructed
505                     // type and exactly one param type
506                     // are obligation types, or
507                     // - it's a method where the return type and
508                     // exactly one param type are obligation types
509                     // 2. at least one instance of the resource "consumed"
510                     // by the transfer exists at the point of the transfer.
511                     // E.g., if we see a transfer of InputStream->Reader,
512                     // there must be an instance of InputStream at
513                     // the transfer point.
514                     //
515     
516                     if (DEBUG_FP) {
517                         System.out.println("Checking " + handle + " as possible obligation transfer...:");
518                     }
519     
520                     // Find the State which is a prefix of the error state
521                     // at the location of this (possible) transfer.
522                     State transferState = getTransferState(handle);
523                     if (transferState == null) {
524                         if (DEBUG_FP) {
525                             System.out.println("No transfer state???");
526                         }
527                         return;
528                     }
529     
530                     String methodName = inv.getMethodName(cpg);
531                     Type producedType = "<init>".equals(methodName) ? inv.getReferenceType(cpg) : inv.getReturnType(cpg);
532     
533                     if (DEBUG_FP && !(producedType instanceof ObjectType)) {
534                         System.out.println("Produced type " + producedType + " not an ObjectType");
535                     }
536     
537                     if (producedType instanceof ObjectType) {
538                         Obligation produced = database.getFactory().getObligationByType((ObjectType) producedType);
539     
540                         if (DEBUG_FP && produced == null) {
541                             System.out.println("Produced type  " + producedType + " not an obligation type");
542                         }
543     
544                         if (produced != null) {
545                             XMethod calledMethod = XFactory.createXMethod(inv, cpg);
546                             Obligation[] params = database.getFactory().getParameterObligationTypes(calledMethod);
547     
548                             for (int i = 0; i < params.length; i++) {
549                                 Obligation consumed = params[i];
550     
551                                 if (DEBUG_FP && consumed == null) {
552                                     System.out.println("Param " + i + " not an obligation type");
553                                 }
554     
555                                 if (DEBUG_FP && consumed != null && consumed.equals(produced)) {
556                                     System.out.println("Consumed type is the same as produced type");
557                                 }
558     
559                                 if (consumed != null && !consumed.equals(produced)) {
560                                     // See if an instance of the consumed obligation
561                                     // type
562                                     // exists here.
563                                     if (transferState.getObligationSet().getCount(consumed.getId()) > 0) {
564                                         transferList.add(new PossibleObligationTransfer(consumed, produced));
565                                         if (DEBUG_FP) {
566                                             System.out.println("===> Possible transfer of " + consumed + " to " + produced + " at "
567                                                     + handle);
568                                         }
569                                     } else if (DEBUG_FP) {
570                                         System.out.println(handle + " not a transfer " + "of " + consumed + "->" + produced
571                                                 + " because no instances of " + consumed);
572                                         System.out.println("I see " + transferState.getObligationSet());
573                                     }
574                                 }
575                             }
576                         }
577                     }
578                 }

581                 public void visitEdge(Edge edge) {
582                     if (DEBUG_FP) {
583                         System.out.println("visit edge " + edge);
584                     }

587                 private State getTransferState(InstructionHandle handle) {
588                     StateSet stateSet;
589                     try {
590                         stateSet = dataflow.getFactAtLocation(new Location(handle, curBlock));
591                     } catch (DataflowAnalysisException e) {
592                         bugReporter.logError("Error checking obligation state at " + handle, e);
593                         return null;
594                     }
595     
596                     List<State> prefixes = stateSet.getPrefixStates(state.getPath());
597                     if (prefixes.size() != 1) {
598                         // Could this happen?
599                         if (DEBUG_FP) {
600                             System.out.println("at " + handle + " in " + xmethod + " found " + prefixes.size()
601                                     + " states which are prefixes of error state");
602                         }
603                         return null;
604                     }
605     
606                     return prefixes.get(0);
607                 }
608             }

610             /**
611              * Get the adjusted leak count for the given State and obligation type.
612              * Use heuristics to account for:
613              * <ul>
614              * <li>null checks (count the number of times the supposedly leaked
615              * obligation is compared to null, and subtract those from the leak
616              * count)</li>
617              * <li>field assignments (count number of times obligation type is
618              * assigned to a field, and subtract those from the leak count)</li>
619              * <li>return statements (if an instance of the obligation type is
620              * returned from the method, subtract one from leak count)</li>
621              * </ul>
622              *
623              * @return the adjusted leak count (positive if leaked obligation,
624              *         negative if attempt to release an un-acquired obligation)
625              */
626             private int getAdjustedLeakCount(State state, int obligationId) {
627     
628                 final Obligation obligation = database.getFactory().getObligationById(obligationId);
629                 Path path = state.getPath();
630                 PostProcessingPathVisitor visitor = new PostProcessingPathVisitor(obligation, state);
631                 path.acceptVisitor(cfg, visitor);
632     
633                 if (visitor.couldNotAnalyze()) {
634                     return 0;
635                 } else {
636                     return visitor.getAdjustedLeakCount();
637                 }
638             }

640             private boolean isPossibleInstanceOfObligationType(Subtypes2 subtypes2, ObjectType type, ObjectType obligationType)
641                     throws ClassNotFoundException {
642                 //
643                 // If we're tracking, e.g., InputStream obligations,
644                 // and we see a FileInputStream reference being assigned
645                 // to a field (or returned from a method),
646                 // then the false-positive supressions heuristic should apply.
647                 //
648     
649                 return subtypes2.isSubtype(type, obligationType);
650             }

652             private void reportPath(final BugInstance bugInstance, final Obligation obligation, final State state) {
653     
654                 Path path = state.getPath();
655     
656                 // This PathVisitor will traverse the Path and add appropriate
657                 // SourceLineAnnotations to the BugInstance.
658                 PathVisitor visitor = new PathVisitor() {
659                     boolean sawFirstCreation;
660     
661                     SourceLineAnnotation lastSourceLine;// = creationSourceLine;
662     
663                     BasicBlock curBlock;
664     
665                     @Override
666                     public void visitBasicBlock(BasicBlock basicBlock) {
667                         curBlock = basicBlock;
668     
669                         // See if the initial instance of the leaked resource
670                         // is in the entry fact due to a @WillClose annotation.
671                         if (curBlock == cfg.getEntry()) {
672                             // Get the entry fact - it should have precisely one
673                             // state
674                             StateSet entryFact = dataflow.getResultFact(curBlock);
675                             Iterator<State> i = entryFact.stateIterator();
676                             if (i.hasNext()) {
677                                 State entryState = i.next();
678                                 if (entryState.getObligationSet().getCount(obligation.getId()) > 0) {
679                                     lastSourceLine = SourceLineAnnotation.forFirstLineOfMethod(methodDescriptor);
680                                     lastSourceLine
681                                     .setDescription(SourceLineAnnotation.ROLE_OBLIGATION_CREATED_BY_WILLCLOSE_PARAMETER);
682                                     bugInstance.add(lastSourceLine);
683                                     sawFirstCreation = true;
684     
685                                     if (REPORT_PATH_DEBUG) {
686                                         System.out.println("  " + obligation + " created by @WillClose parameter at "
687                                                 + lastSourceLine);
688                                     }
689                                 }
690                             }
691                         }
692                     }
693     
694                     @Override
695                     public void visitInstructionHandle(InstructionHandle handle) {
696                         boolean isCreation = (dataflow.getAnalysis().getActionCache().addsObligation(curBlock, handle, obligation));
697     
698                         if (!sawFirstCreation && !isCreation) {
699                             return;
700                         }
701     
702                         SourceLineAnnotation sourceLine = SourceLineAnnotation.fromVisitedInstruction(methodDescriptor, new Location(
703                                 handle, curBlock));
704                         sourceLine.setDescription(isCreation ? SourceLineAnnotation.ROLE_OBLIGATION_CREATED
705                                 : SourceLineAnnotation.ROLE_PATH_CONTINUES);
706     
707                         boolean isInteresting = (sourceLine.getStartLine() > 0)
708                                 && (lastSourceLine == null || isCreation || sourceLine.getStartLine() != lastSourceLine.getStartLine());
709     
710                         if (REPORT_PATH_DEBUG) {
711                             System.out.println("  " + handle.getPosition() + " --> " + sourceLine + (isInteresting ? " **" : ""));
712                         }
713                         if (isInteresting) {
714                             bugInstance.add(sourceLine);
715                             lastSourceLine = sourceLine;
716                             if (isCreation) {
717                                 sawFirstCreation = true;
718                             }
719                         }
720                     }
721     
722                     @Override
723                     public void visitEdge(Edge edge) {
724                         if (REPORT_PATH_DEBUG) {
725                             System.out.println("Edge of type " + Edge.edgeTypeToString(edge.getType()) + " to "
726                                     + edge.getTarget().getLabel());
727                             if (edge.getTarget().getFirstInstruction() != null) {
728                                 System.out.println("  First instruction in target: " + edge.getTarget().getFirstInstruction());
729                             }
730                             if (edge.getTarget().isExceptionThrower()) {
731                                 System.out.println("  exception thrower for " + edge.getTarget().getExceptionThrower());
732                             }
733                             if (edge.isExceptionEdge()) {
734                                 System.out.println("  exceptions thrown: " + typeDataflow.getEdgeExceptionSet(edge));
735                             }
736                         }
737                     }
738                 };
739     
740                 // Visit the Path
741                 path.acceptVisitor(cfg, visitor);
742             }
743         }
