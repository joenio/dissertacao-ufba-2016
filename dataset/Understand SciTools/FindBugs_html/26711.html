339         private void applyToBugs(final BugAction bugAction) {
340             Executor executor = backgroundExecutor;
341     
342             final AtomicInteger shownErrorMessages = new AtomicInteger(0);
343             for (final BugInstance bug : getSelectedBugs()) {
344                 executor.execute(new Runnable() {
345                     @Override
346                     public void run() {
347                         if (shownErrorMessages.get() > 5) {
348                             // 5 errors? let's just stop trying.
349                             return;
350                         }
351                         try {
352                             bugAction.execute(bug);
353                         } catch (Throwable e) {
354                             if (shownErrorMessages.addAndGet(1) > 5) {
355                                 return;
356                             }
357                             JOptionPane.showMessageDialog(CloudCommentsPane.this,
358                                     "Error while submitting cloud reviews:\n"
359                                             + e.getClass().getSimpleName() + ": " + e.getMessage(),
360                                             "Review Submission Error", JOptionPane.ERROR_MESSAGE);
361                         }
362                     }
363                 });
364             }
365         }

367         protected void signInOrOutClicked() {
368             if (_bugCollection != null) {
369                 final Cloud cloud = _bugCollection.getCloud();
370                 if ("edu.umd.cs.findbugs.cloud.doNothingCloud".equals(cloud.getPlugin().getId())) {
371                     changeClicked();
372                 }
373                 SigninState state = cloud.getSigninState();
374                 if (state == SigninState.SIGNED_IN) {
375                     backgroundExecutor.execute(new Runnable() {
376                         @Override
377                         public void run() {
378                             cloud.signOut();
379                             refresh();
380                         }
381                     });
382                     refresh();
383                 } else if (state.couldSignIn()) {
384                     backgroundExecutor.execute(new Runnable() {
385                         @Override
386                         public void run() {
387                             try {
388                                 cloud.signIn();
389                             } catch (Exception e) {
390                                 _bugCollection
391                                 .getProject()
392                                 .getGuiCallback()
393                                 .showMessageDialog(
394                                         "The FindBugs Cloud could not be contacted at this time.\n\n"
395                                                 + Util.getNetworkErrorMessage(e));
396                             }
397                             refresh();
398                         }
399                     });
400                     refresh();
401     
402                 }
403             }

406         protected void commentBoxClicked() {
407             if (commentWasChanged()) {
408                 return;
409             }
410             setCanAddComments(false, true);
411             CommentInfo commentInfo = new CommentInfo().invoke();
412             boolean sameText = commentInfo.isSameText();
413             String txt = commentInfo.getTxt();
414             if (!sameText) {
415                 txt = "";
416             }
417             if (txt == null || txt.trim().length() == 0) {
418                 txt = "";
419             }
420             resetCommentBoxFont();
421             boolean sameTextInBox = commentBox.getText().equals(txt);
422             setCommentText(txt);
423             int start = commentBox.getSelectionStart();
424             int end = commentBox.getSelectionEnd();
425             if (!commentBox.hasFocus() && (!sameTextInBox || start != 0 || end != txt.length())) {
426                 commentBox.setSelectionStart(0);
427                 commentBox.setSelectionEnd(txt.length());
428             }
429             updateSaveButton();
430         }

457         public void setDesignation(final String designationKey) {
458     
459             //        List<BugInstance> selectedBugs = getSelectedBugs();
460             //        if (selectedBugs.size() > 1)
461             //            if (!confirmAnnotation(selectedBugs))
462             //                return;
463             final AtomicBoolean stop = new AtomicBoolean(false);
464             applyToBugs(new BugAction() {
465                 @Override
466                 public void execute(BugInstance bug) {
467                     if (stop.get()) {
468                         return;
469                     }
470                     String oldValue = bug.getUserDesignationKey();
471                     String key = designationKey;
472                     if (key.equals(oldValue)) {
473                         return;
474                     }
475                     Cloud plugin = _bugCollection != null ? _bugCollection.getCloud() : null;
476                     if (plugin != null && "I_WILL_FIX".equals(key) && plugin.supportsClaims()) {
477                         String claimedBy = plugin.claimedBy(bug);
478                         if (claimedBy != null && !plugin.getUser().equals(claimedBy)) {
479                             int result = JOptionPane.showConfirmDialog(null,
480                                     bug.getMessage() + "\n"
481                                             + claimedBy + " has already said they will fix this issue\n"
482                                             + "Do you want to also be listed as fixing this issue?\n"
483                                             + "If so, please coordinate with " + claimedBy,
484                                             "Issue already claimed", JOptionPane.YES_NO_CANCEL_OPTION);
485                             if (result == JOptionPane.CANCEL_OPTION) {
486                                 stop.set(true);
487                                 return;
488                             }
489                             if (result != JOptionPane.YES_OPTION) {
490                                 key = "MUST_FIX";
491                             }
492                         }
493                     }
494                     changeDesignationOfBugRightNow(bug, key);
495                     refresh();
496                 }
497             });
498         }

501         private void submitComment(List<BugInstance> selectedBugs) {
502             String comment = commentBox.getText();
503             if (isDefaultComment(comment)) {
504                 comment = "";
505             }
506             //        if (selectedBugs.size() > 1)

570         protected void changeClicked() {
571             final List<CloudPlugin> plugins = new ArrayList<CloudPlugin>();
572             final List<String> descriptions = new ArrayList<String>();
573             List<CloudPlugin> clouds = new ArrayList<CloudPlugin>(DetectorFactoryCollection.instance().getRegisteredClouds().values());
574             Collections.sort(clouds, new Comparator<CloudPlugin>() {
575                 @Override
576                 public int compare(CloudPlugin o1, CloudPlugin o2) {
577                     return o1.getDescription().compareToIgnoreCase(o2.getDescription());
578                 }
579             });
580             for (final CloudPlugin plugin : clouds) {
581                 final boolean disabled = isDisabled(plugin);
582                 if (!disabled && !plugin.isHidden()) {
583                     descriptions.add(plugin.getDescription());
584                     plugins.add(plugin);
585                 }
586             }
587             showCloudChooser(plugins, descriptions);
588         }

594         protected void changeCloud(String newCloudId) {
595             final String oldCloudId = _bugCollection.getCloud().getPlugin().getId();
596             if (!oldCloudId.equals(newCloudId)) {
597                 _bugCollection.getProject().setCloudId(newCloudId);
598                 MainFrame.getInstance().setProjectChanged(true);
599                 backgroundExecutor.execute(new Runnable() {
600                     @Override
601                     public void run() {
602                         _bugCollection.reinitializeCloud();
603                         Cloud cloud = _bugCollection.getCloud();
604                         if (cloud != null) {
605                             cloud.waitUntilIssueDataDownloaded();
606                         }
607                         updateCloudListeners(_bugCollection);
608                         refresh();
609                     }
610                 });
611                 refresh();
612             }
613         }

670         private void updateCloudListeners(BugCollection newBugCollection) {
671             final Cloud newCloud = newBugCollection == null ? null : newBugCollection.getCloud();
672             if (_bugCollection != null) {
673                 //noinspection ObjectEquality
674                 if (lastCloud != newCloud) {
675                     if (lastCloud != null) {
676                         lastCloud.removeStatusListener(_cloudStatusListener);
677                     }
678                 }
679             }
680             if (lastCloud != newCloud && newCloud != null) {
681                 lastCloud = newCloud;
682                 newCloud.addStatusListener(_cloudStatusListener);
683             }
684         }
