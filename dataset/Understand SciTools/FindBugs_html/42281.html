182         /**
183          * Execute the analysis. For obscure reasons, CheckedAnalysisExceptions are
184          * re-thrown as IOExceptions. However, these can only happen during the
185          * setup phase where we scan codebases for classes.
186          *
187          * @throws IOException
188          * @throws InterruptedException
189          */
190         @Override
191         public void execute() throws IOException, InterruptedException {
192     
193             if (FindBugs.isNoAnalysis()) {
194                 throw new UnsupportedOperationException("This FindBugs invocation was started without analysis capabilities");
195             }
196     
197             Profiler profiler = bugReporter.getProjectStats().getProfiler();
198     
199             try {
200                 try {
201                     // Get the class factory for creating classpath/codebase/etc.
202                     classFactory = ClassFactory.instance();
203     
204                     // The class path object
205                     createClassPath();
206     
207                     progress.reportNumberOfArchives(project.getFileCount() + project.getNumAuxClasspathEntries());
208                     profiler.start(this.getClass());
209     
210                     // The analysis cache object
211                     createAnalysisCache();
212     
213                     // Create BCEL compatibility layer
214                     createAnalysisContext(project, appClassList, analysisOptions.sourceInfoFileName);
215     
216                     // Discover all codebases in classpath and
217                     // enumerate all classes (application and non-application)
218                     buildClassPath();
219     
220     
221                     // Build set of classes referenced by application classes
222                     buildReferencedClassSet();
223     
224                     // Create BCEL compatibility layer
225                     setAppClassList(appClassList);
226     
227                     // Configure the BugCollection (if we are generating one)
228                     FindBugs.configureBugCollection(this);
229     
230                     // Enable/disabled relaxed reporting mode
231                     FindBugsAnalysisFeatures.setRelaxedMode(analysisOptions.relaxedReportingMode);
232                     FindBugsDisplayFeatures.setAbridgedMessages(analysisOptions.abridgedMessages);
233     
234                     // Configure training databases
235                     FindBugs.configureTrainingDatabases(this);
236     
237                     // Configure analysis features
238                     configureAnalysisFeatures();
239     
240                     // Create the execution plan (which passes/detectors to execute)
241                     createExecutionPlan();
242     
243                     for (Plugin p : detectorFactoryCollection.plugins()) {
244                         for (ComponentPlugin<BugReporterDecorator> brp
245                                 : p.getComponentPlugins(BugReporterDecorator.class)) {
246                             if (brp.isEnabledByDefault() && !brp.isNamed(explicitlyDisabledBugReporterDecorators)
247                                     || brp.isNamed(explicitlyEnabledBugReporterDecorators)) {
248                                 bugReporter = BugReporterDecorator.construct(brp, bugReporter);
249                             }
250                         }
251                     }
252                     if (!classScreener.vacuous()) {
253                         bugReporter = new DelegatingBugReporter(bugReporter) {
254     
255                             @Override
256                             public void reportBug(@Nonnull BugInstance bugInstance) {
257                                 String className = bugInstance.getPrimaryClass().getClassName();
258                                 String resourceName = className.replace('.', '/') + ".class";
259                                 if (classScreener.matches(resourceName)) {
260                                     this.getDelegate().reportBug(bugInstance);
261                                 }
262                             }
263                         };
264                     }
265     
266                     if (executionPlan.isActive(NoteSuppressedWarnings.class)) {
267                         SuppressionMatcher m = AnalysisContext.currentAnalysisContext().getSuppressionMatcher();
268                         bugReporter = new FilterBugReporter(bugReporter, m, false);
269                     }
270     
271                     if (appClassList.size() == 0) {
272                         Map<String, ICodeBaseEntry> codebase = classPath.getApplicationCodebaseEntries();
273                         if (analysisOptions.noClassOk) {
274                             System.err.println("No classfiles specified; output will have no warnings");
275                         } else if  (codebase.isEmpty()) {
276                             throw new IOException("No files to analyze could be opened");
277                         } else {
278                             throw new NoClassesFoundToAnalyzeException(classPath);
279                         }
280                     }
281     
282                     // Analyze the application
283                     analyzeApplication();
284                 } catch (CheckedAnalysisException e) {
285                     IOException ioe = new IOException("IOException while scanning codebases");
286                     ioe.initCause(e);
287                     throw ioe;
288                 } catch (OutOfMemoryError e) {
289                     System.err.println("Out of memory");
290                     System.err.println("Total memory: " + Runtime.getRuntime().maxMemory() / 1000000 + "M");
291                     System.err.println(" free memory: " + Runtime.getRuntime().freeMemory() / 1000000 + "M");
292     
293                     for (String s : project.getFileList()) {
294                         System.err.println("Analyzed: " + s);
295                     }
296                     for (String s : project.getAuxClasspathEntryList()) {
297                         System.err.println("     Aux: " + s);
298                     }
299                     throw e;
300                 } finally {
301                     clearCaches();
302                     profiler.end(this.getClass());
303                     profiler.report();
304                 }
305             } catch (IOException e) {
306                 bugReporter.reportQueuedErrors();
307                 throw e;
308             }
309         }

478         protected void configureFilters(UserPreferences userPreferences) {
479             IllegalArgumentException deferredError = null;
480             Set<Entry<String, Boolean>> excludeBugFiles = userPreferences.getExcludeBugsFiles().entrySet();
481             for (Entry<String, Boolean> entry : excludeBugFiles) {
482                 if (entry.getValue() == null || !entry.getValue()) {
483                     continue;
484                 }
485                 try {
486                     excludeBaselineBugs(entry.getKey());
487                 } catch (Exception e) {
488                     String message = "Unable to read filter: " + entry.getKey() + " : " + e.getMessage();
489                     if (getBugReporter() != null) {
490                         getBugReporter().logError(message, e);
491                     } else if (deferredError == null){
492                         deferredError = new IllegalArgumentException(message, e);
493                     }
494                 }
495             }
496             Set<Entry<String, Boolean>> includeFilterFiles = userPreferences.getIncludeFilterFiles().entrySet();
497             for (Entry<String, Boolean> entry : includeFilterFiles) {
498                 if (entry.getValue() == null || !entry.getValue()) {
499                     continue;
500                 }
501                 try {
502                     addFilter(entry.getKey(), true);
503                 } catch (Exception e) {
504                     String message = "Unable to read filter: " + entry.getKey() + " : " + e.getMessage();
505                     if (getBugReporter() != null) {
506                         getBugReporter().logError(message, e);
507                     } else if (deferredError == null){
508                         deferredError = new IllegalArgumentException(message, e);
509                     }
510                 }
511             }
512             Set<Entry<String, Boolean>> excludeFilterFiles = userPreferences.getExcludeFilterFiles().entrySet();
513     
514             for (Entry<String, Boolean> entry : excludeFilterFiles) {
515                 Boolean value = entry.getValue();
516                 if (value == null || !value) {
517                     continue;
518                 }
519                 String excludeFilterFile = entry.getKey();
520                 try {
521                     addFilter(excludeFilterFile, false);
522                 } catch (Exception e) {
523                     String message = "Unable to read filter: " + excludeFilterFile + " : " + e.getMessage();
524                     if (getBugReporter() != null) {
525                         getBugReporter().logError(message, e);
526                     } else if (deferredError == null){
527                         deferredError = new IllegalArgumentException(message, e);
528                     }
529                 }
530             }
531             if (deferredError != null) {
532                 throw deferredError;
533             }
534         }

578         /**
579          * Create the analysis cache object and register it for current execution thread.
580          * <p>
581          * This method is protected to allow clients override it and possibly reuse
582          * some previous analysis data (for Eclipse interactive re-build)
583          *
584          * @throws IOException
585          *             if error occurs registering analysis engines in a plugin
586          */
587         protected IAnalysisCache createAnalysisCache() throws IOException {
588             IAnalysisCache analysisCache = ClassFactory.instance().createAnalysisCache(classPath, bugReporter);
589     
590             // Register the "built-in" analysis engines
591             registerBuiltInAnalysisEngines(analysisCache);
592     
593             // Register analysis engines in plugins
594             registerPluginAnalysisEngines(detectorFactoryCollection, analysisCache);
595     
596             // Install the DetectorFactoryCollection as a database
597             analysisCache.eagerlyPutDatabase(DetectorFactoryCollection.class, detectorFactoryCollection);
598     
599             Global.setAnalysisCacheForCurrentThread(analysisCache);
600             return analysisCache;
601         }

614         /**
615          * Register all of the analysis engines defined in the plugins contained in
616          * a DetectorFactoryCollection with an IAnalysisCache.
617          *
618          * @param detectorFactoryCollection
619          *            a DetectorFactoryCollection
620          * @param analysisCache
621          *            an IAnalysisCache
622          * @throws IOException
623          */
624         public static void registerPluginAnalysisEngines(DetectorFactoryCollection detectorFactoryCollection,
625                 IAnalysisCache analysisCache) throws IOException {
626             for (Iterator<Plugin> i = detectorFactoryCollection.pluginIterator(); i.hasNext();) {
627                 Plugin plugin = i.next();
628     
629                 Class<? extends IAnalysisEngineRegistrar> engineRegistrarClass = plugin.getEngineRegistrarClass();
630                 if (engineRegistrarClass != null) {
631                     try {
632                         IAnalysisEngineRegistrar engineRegistrar = engineRegistrarClass.newInstance();
633                         engineRegistrar.registerAnalysisEngines(analysisCache);
634                     } catch (InstantiationException e) {
635                         IOException ioe = new IOException("Could not create analysis engine registrar for plugin "
636                                 + plugin.getPluginId());
637                         ioe.initCause(e);
638                         throw ioe;
639                     } catch (IllegalAccessException e) {
640                         IOException ioe = new IOException("Could not create analysis engine registrar for plugin "
641                                 + plugin.getPluginId());
642                         ioe.initCause(e);
643                         throw ioe;
644                     }
645                 }
646             }
647         }

649         /**
650          * Build the classpath from project codebases and system codebases.
651          *
652          * @throws InterruptedException
653          *             if the analysis thread is interrupted
654          * @throws IOException
655          *             if an I/O error occurs
656          * @throws CheckedAnalysisException
657          */
658         private void buildClassPath() throws InterruptedException, IOException, CheckedAnalysisException {
659             IClassPathBuilder builder = classFactory.createClassPathBuilder(bugReporter);
660     
661             {
662                 HashSet<String> seen = new HashSet<String>();
663                 for (String path : project.getFileArray()) {
664                     if (seen.add(path)) {
665                         builder.addCodeBase(classFactory.createFilesystemCodeBaseLocator(path), true);
666                     }
667                 }
668                 for (String path : project.getAuxClasspathEntryList()) {
669                     if (seen.add(path)) {
670                         builder.addCodeBase(classFactory.createFilesystemCodeBaseLocator(path), false);
671                     }
672                 }
673             }
674     
675             builder.scanNestedArchives(analysisOptions.scanNestedArchives);
676     
677             builder.build(classPath, progress);
678     
679             appClassList = builder.getAppClassList();
680     
681             if (PROGRESS) {
682                 System.out.println(appClassList.size() + " classes scanned");
683             }
684     
685             // If any of the application codebases contain source code,
686             // add them to the source path.
687             // Also, use the last modified time of application codebases
688             // to set the project timestamp.
689             for (Iterator<? extends ICodeBase> i = classPath.appCodeBaseIterator(); i.hasNext();) {
690                 ICodeBase appCodeBase = i.next();
691     
692                 if (appCodeBase.containsSourceFiles()) {
693                     String pathName = appCodeBase.getPathName();
694                     if (pathName != null) {
695                         project.addSourceDir(pathName);
696                     }
697                 }
698     
699                 project.addTimestamp(appCodeBase.getLastModifiedTime());
700             }

716             Set<ClassDescriptor> appClassSet = new HashSet<ClassDescriptor>(appClassList);
717     
718             Set<ClassDescriptor> badAppClassSet = new HashSet<ClassDescriptor>();
719             HashSet<ClassDescriptor> knownDescriptors = new HashSet<ClassDescriptor>(DescriptorFactory.instance()
720                     .getAllClassDescriptors());
721             int count = 0;
722             Set<ClassDescriptor> addedToWorkList = new HashSet<ClassDescriptor>(appClassList);
723     
724             // add fields
725             //noinspection ConstantIfStatement
726             /*
727             if (false)

731                         for (XField f : classNameAndInfo.getXFields()) {
732                             String sig = f.getSignature();
733                             ClassDescriptor d = DescriptorFactory.createClassDescriptorFromFieldSignature(sig);
734                             if (d != null && addedToWorkList.add(d))
735                                 workList.addLast(d);

737                     } catch (RuntimeException e) {

756     
757                 if (seen.contains(classDesc)) {
758                     continue;
759                 }
760                 seen.add(classDesc);
761     
762                 if (!knownDescriptors.contains(classDesc)) {
763                     count++;
764                     if (PROGRESS && count % 5000 == 0) {
765                         System.out.println("Adding referenced class " + classDesc);
766                     }
767                 }
768     
769                 referencedPackageSet.add(classDesc.getPackageName());
770     
771                 // Get list of referenced classes and add them to set.
772                 // Add superclasses and superinterfaces to worklist.
773                 try {
774                     XClass classNameAndInfo = Global.getAnalysisCache().getClassAnalysis(XClass.class, classDesc);
775     
776                     ClassDescriptor superclassDescriptor = classNameAndInfo.getSuperclassDescriptor();
777                     if (superclassDescriptor != null && addedToWorkList.add(superclassDescriptor)) {
778                         workList.addLast(superclassDescriptor);
779                     }
780     
781                     for (ClassDescriptor ifaceDesc : classNameAndInfo.getInterfaceDescriptorList()) {
782                         if (addedToWorkList.add(ifaceDesc)) {
783                             workList.addLast(ifaceDesc);
784                         }
785                     }
786     
787                     ClassDescriptor enclosingClass = classNameAndInfo.getImmediateEnclosingClass();
788                     if (enclosingClass != null && addedToWorkList.add(enclosingClass)) {
789                         workList.addLast(enclosingClass);
790                     }
791     
792                 } catch (RuntimeException e) {
793                     bugReporter.logError("Error scanning " + classDesc + " for referenced classes", e);
794                     if (appClassSet.contains(classDesc)) {
795                         badAppClassSet.add(classDesc);
796                     }
797                 } catch (MissingClassException e) {
798                     // Just log it as a missing class
799                     bugReporter.reportMissingClass(e.getClassDescriptor());
800                     if (appClassSet.contains(classDesc)) {
801                         badAppClassSet.add(classDesc);
802                     }
803                 } catch (CheckedAnalysisException e) {
804                     // Failed to scan a referenced class --- just log the error and
805                     // continue
806                     bugReporter.logError("Error scanning " + classDesc + " for referenced classes", e);
807                     if (appClassSet.contains(classDesc)) {
808                         badAppClassSet.add(classDesc);
809                     }
810                 }
811             }
812             // Delete any application classes that could not be read
813             appClassList.removeAll(badAppClassSet);
814             DescriptorFactory.instance().purge(badAppClassSet);
815     
816             for (ClassDescriptor d : DescriptorFactory.instance().getAllClassDescriptors()) {
817                 referencedPackageSet.add(d.getPackageName());
818             }
819             referencedClassSet = new ArrayList<ClassDescriptor>(DescriptorFactory.instance().getAllClassDescriptors());
820     
821             // Based on referenced packages, add any resolvable package-info classes
822             // to the set of referenced classes.
823             if (PROGRESS) {
824                 referencedPackageSet.remove("");
825                 System.out.println("Added " + count + " referenced classes");
826                 System.out.println("Total of " + referencedPackageSet.size() + " packages");
827                 for (ClassDescriptor d : referencedClassSet) {
828                     System.out.println("  " + d);
829                 }
830             }
831         }

891         /**
892          * Create an execution plan.
893          *
894          * @throws OrderingConstraintException
895          *             if the detector ordering constraints are inconsistent
896          */
897         private void createExecutionPlan() throws OrderingConstraintException {
898             executionPlan = new ExecutionPlan();
899     
900             // Use user preferences to decide which detectors are enabled.
901             DetectorFactoryChooser detectorFactoryChooser = new DetectorFactoryChooser() {
902                 HashSet<DetectorFactory> forcedEnabled = new HashSet<DetectorFactory>();
903     
904                 @Override
905                 public boolean choose(DetectorFactory factory) {
906                     boolean result = FindBugs.isDetectorEnabled(FindBugs2.this, factory, rankThreshold) || forcedEnabled.contains(factory);
907                     if (ExecutionPlan.DEBUG) {
908                         System.out.printf("  %6s %s %n", result, factory.getShortName());
909                     }
910                     return result;
911                 }
912     
913                 @Override
914                 public void enable(DetectorFactory factory) {
915                     forcedEnabled.add(factory);
916                     factory.setEnabledButNonReporting(true);
917                 }
918     
919             };
920             executionPlan.setDetectorFactoryChooser(detectorFactoryChooser);
921     
922             if (ExecutionPlan.DEBUG) {
923                 System.out.println("rank threshold is " + rankThreshold);
924             }
925             // Add plugins
926             for (Iterator<Plugin> i = detectorFactoryCollection.pluginIterator(); i.hasNext();) {
927                 Plugin plugin = i.next();
928                 if (DEBUG) {
929                     System.out.println("Adding plugin " + plugin.getPluginId() + " to execution plan");
930                 }
931                 executionPlan.addPlugin(plugin);
932             }
933     
934             // Build the execution plan
935             executionPlan.build();
936     
937             // Stash the ExecutionPlan in the AnalysisCache.
938             Global.getAnalysisCache().eagerlyPutDatabase(ExecutionPlan.class, executionPlan);
939     
940             if (PROGRESS) {
941                 System.out.println(executionPlan.getNumPasses() + " passes in execution plan");
942             }
943         }

945         /**
946          * Analyze the classes in the application codebase.
947          */
948         private void analyzeApplication() throws InterruptedException {
949             int passCount = 0;
950             Profiler profiler = bugReporter.getProjectStats().getProfiler();
951             profiler.start(this.getClass());
952             AnalysisContext.currentXFactory().canonicalizeAll();
953             try {
954                 boolean multiplePasses = executionPlan.getNumPasses() > 1;
955                 if (executionPlan.getNumPasses() == 0) {
956                     throw new AssertionError("no analysis passes");
957                 }
958                 int[] classesPerPass = new int[executionPlan.getNumPasses()];
959                 classesPerPass[0] = referencedClassSet.size();
960                 for (int i = 0; i < classesPerPass.length; i++) {
961                     classesPerPass[i] = i == 0 ? referencedClassSet.size() : appClassList.size();
962                 }
963                 progress.predictPassCount(classesPerPass);
964                 XFactory factory = AnalysisContext.currentXFactory();
965                 Collection<ClassDescriptor> badClasses = new LinkedList<ClassDescriptor>();
966                 for (ClassDescriptor desc : referencedClassSet) {
967                     try {
968                         XClass info = Global.getAnalysisCache().getClassAnalysis(XClass.class, desc);
969                         factory.intern(info);
970                     } catch (CheckedAnalysisException e) {
971                         AnalysisContext.logError("Couldn't get class info for " + desc, e);
972                         badClasses.add(desc);
973                     } catch (RuntimeException e) {
974                         AnalysisContext.logError("Couldn't get class info for " + desc, e);
975                         badClasses.add(desc);
976                     }
977                 }
978                 if (!badClasses.isEmpty()) {
979                     referencedClassSet = new LinkedHashSet<ClassDescriptor>(referencedClassSet);
980                     referencedClassSet.removeAll(badClasses);
981                 }
982     
983                 long startTime = System.currentTimeMillis();
984                 bugReporter.getProjectStats().setReferencedClasses(referencedClassSet.size());
985                 for (Iterator<AnalysisPass> passIterator = executionPlan.passIterator(); passIterator.hasNext();) {
986                     AnalysisPass pass = passIterator.next();
987                     yourkitController.advanceGeneration("Pass " + passCount);
988                     // The first pass is generally a non-reporting pass which
989                     // gathers information about referenced classes.
990                     boolean isNonReportingFirstPass = multiplePasses && passCount == 0;
991     
992                     // Instantiate the detectors
993                     Detector2[] detectorList = pass.instantiateDetector2sInPass(bugReporter);
994     
995                     // If there are multiple passes, then on the first pass,
996                     // we apply detectors to all classes referenced by the
997                     // application classes.
998                     // On subsequent passes, we apply detector only to application
999                     // classes.
1000                     Collection<ClassDescriptor> classCollection = (isNonReportingFirstPass) ? referencedClassSet : appClassList;
1001                     AnalysisContext.currentXFactory().canonicalizeAll();
1002                     if (PROGRESS || LIST_ORDER) {
1003                         System.out.printf("%6d : Pass %d: %d classes%n", (System.currentTimeMillis() - startTime)/1000, passCount,  classCollection.size());
1004                         if (DEBUG) {
1005                             XFactory.profile();
1006                         }
1007                     }
1008                     if (!isNonReportingFirstPass) {
1009                         OutEdges<ClassDescriptor> outEdges = new OutEdges<ClassDescriptor>() {
1010     
1011                             @Override
1012                             public Collection<ClassDescriptor> getOutEdges(ClassDescriptor e) {
1013                                 try {
1014                                     XClass classNameAndInfo = Global.getAnalysisCache().getClassAnalysis(XClass.class, e);
1015                                     return classNameAndInfo.getCalledClassDescriptors();
1016                                 } catch (CheckedAnalysisException e2) {
1017                                     AnalysisContext.logError("error while analyzing " + e.getClassName(), e2);
1018                                     return Collections.emptyList();
1019     
1020                                 }
1021                             }
1022                         };
1023     
1024                         classCollection = sortByCallGraph(classCollection, outEdges);
1025                     }
1026                     if (LIST_ORDER) {
1027                         System.out.println("Analysis order:");
1028                         for (ClassDescriptor c : classCollection) {
1029                             System.out.println("  " + c);
1030                         }
1031                     }
1032                     AnalysisContext currentAnalysisContext = AnalysisContext.currentAnalysisContext();
1033                     currentAnalysisContext.updateDatabases(passCount);
1034     
1035                     progress.startAnalysis(classCollection.size());
1036                     int count = 0;
1037                     Global.getAnalysisCache().purgeAllMethodAnalysis();
1038                     Global.getAnalysisCache().purgeClassAnalysis(FBClassReader.class);
1039                     for (ClassDescriptor classDescriptor : classCollection) {
1040                         long classStartNanoTime = 0;
1041                         if (PROGRESS) {
1042                             classStartNanoTime = System.nanoTime();
1043                             System.out.printf("%6d %d/%d  %d/%d %s%n", (System.currentTimeMillis() - startTime)/1000,
1044                                     passCount, executionPlan.getNumPasses(), count,
1045                                     classCollection.size(), classDescriptor);
1046                         }
1047                         count++;
1048                         if (!isNonReportingFirstPass && count % 1000 == 0) {
1049                             yourkitController.advanceGeneration(String.format("Pass %d.%02d", passCount, count/1000));
1050                         }
1051     
1052     
1053                         // Check to see if class is excluded by the class screener.
1054                         // In general, we do not want to screen classes from the
1055                         // first pass, even if they would otherwise be excluded.
1056                         if ((SCREEN_FIRST_PASS_CLASSES || !isNonReportingFirstPass)
1057                                 && !classScreener.matches(classDescriptor.toResourceName())) {
1058                             if (DEBUG) {
1059                                 System.out.println("*** Excluded by class screener");
1060                             }
1061                             continue;
1062                         }
1063                         boolean isHuge = currentAnalysisContext.isTooBig(classDescriptor);
1064                         if (isHuge && currentAnalysisContext.isApplicationClass(classDescriptor)) {
1065                             bugReporter.reportBug(new BugInstance("SKIPPED_CLASS_TOO_BIG", Priorities.NORMAL_PRIORITY)
1066                             .addClass(classDescriptor));
1067                         }
1068                         currentClassName = ClassName.toDottedClassName(classDescriptor.getClassName());
1069                         notifyClassObservers(classDescriptor);
1070                         profiler.startContext(currentClassName);
1071                         currentAnalysisContext.setClassBeingAnalyzed(classDescriptor);
1072     
1073                         try {
1074                             for (Detector2 detector : detectorList) {
1075                                 if (Thread.interrupted()) {
1076                                     throw new InterruptedException();
1077                                 }
1078                                 if (isHuge && !FirstPassDetector.class.isAssignableFrom(detector.getClass())) {
1079                                     continue;
1080                                 }
1081                                 if (DEBUG) {
1082                                     System.out.println("Applying " + detector.getDetectorClassName() + " to " + classDescriptor);
1083                                     // System.out.println("foo: " +
1084                                     // NonReportingDetector.class.isAssignableFrom(detector.getClass())
1085                                     // + ", bar: " + detector.getClass().getName());
1086                                 }
1087                                 try {
1088                                     profiler.start(detector.getClass());
1089                                     detector.visitClass(classDescriptor);
1090                                 } catch (ClassFormatException e) {
1091                                     logRecoverableException(classDescriptor, detector, e);
1092                                 } catch (MissingClassException e) {
1093                                     Global.getAnalysisCache().getErrorLogger().reportMissingClass(e.getClassDescriptor());
1094                                 } catch (CheckedAnalysisException e) {
1095                                     logRecoverableException(classDescriptor, detector, e);
1096                                 } catch (RuntimeException e) {
1097                                     logRecoverableException(classDescriptor, detector, e);
1098                                 } finally {
1099                                     profiler.end(detector.getClass());
1100                                 }
1101                             }
1102                         } finally {
1103     
1104                             progress.finishClass();
1105                             profiler.endContext(currentClassName);
1106                             currentAnalysisContext.clearClassBeingAnalyzed();
1107                             if (PROGRESS) {
1108                                 long usecs = (System.nanoTime() - classStartNanoTime)/1000;
1109                                 if (usecs > 15000) {
1110                                     int classSize = currentAnalysisContext.getClassSize(classDescriptor);
1111                                     long speed = usecs /classSize;
1112                                     if (speed > 15) {
1113                                         System.out.printf("  %6d usecs/byte  %6d msec  %6d bytes  %d pass %s%n", speed, usecs/1000, classSize, passCount,
1114                                                 classDescriptor);
1115                                     }
1116                                 }
1117     
1118                             }
1119                         }
1120                     }
1121     
1122                     if (!passIterator.hasNext()) {
1123                         yourkitController.captureMemorySnapshot();
1124                     }
1125                     // Call finishPass on each detector
1126                     for (Detector2 detector : detectorList) {
1127                         detector.finishPass();
1128                     }
1129     
1130                     progress.finishPerClassAnalysis();
1131     
1132                     passCount++;
1133                 }
1134     
1135     
1136             } finally {
1137     
1138                 bugReporter.finish();
1139                 bugReporter.reportQueuedErrors();
1140                 profiler.end(this.getClass());
1141                 if (PROGRESS) {
1142                     System.out.println("Analysis completed");
1143                 }
1144             }
1145     
1146         }
