20     package edu.umd.cs.findbugs.model;
21     
22     import java.util.Iterator;
23     
24     import edu.umd.cs.findbugs.FieldAnnotation;
25     import edu.umd.cs.findbugs.MethodAnnotation;
26     import edu.umd.cs.findbugs.ba.SignatureParser;
27     
28     /**
29      * Utility methods for using a ClassNameRewriter.
30      *
31      * @author David Hovemeyer
32      */
33     public abstract class ClassNameRewriterUtil {
34     
35         /**
36          * Rewrite a method signature.
37          *
38          * @param classNameRewriter
39          *            a ClassNameRewriter
40          * @param methodSignature
41          *            a method signature
42          * @return the rewritten method signature
43          */
44         public static String rewriteMethodSignature(ClassNameRewriter classNameRewriter, String methodSignature) {
45             if (classNameRewriter != IdentityClassNameRewriter.instance()) {
46                 SignatureParser parser = new SignatureParser(methodSignature);
47     
48                 StringBuilder buf = new StringBuilder();
49     
50                 buf.append('(');
51                 for (Iterator<String> i = parser.parameterSignatureIterator(); i.hasNext();) {
52                     buf.append(rewriteSignature(classNameRewriter, i.next()));
53                 }
54     
55                 buf.append(')');
56                 buf.append(rewriteSignature(classNameRewriter, parser.getReturnTypeSignature()));
57     
58                 methodSignature = buf.toString();
59             }
60     
61             return methodSignature;
62         }
63     
64         /**
65          * Rewrite a signature.
66          *
67          * @param classNameRewriter
68          *            a ClassNameRewriter
69          * @param signature
70          *            a signature (parameter, return type, or field)
71          * @return rewritten signature with class name updated if required
72          */
73         public static String rewriteSignature(ClassNameRewriter classNameRewriter, String signature) {
74             if (classNameRewriter != IdentityClassNameRewriter.instance() && signature.startsWith("L")) {
75     
76                 String className = signature.substring(1, signature.length() - 1).replace('/', '.');
77                 className = classNameRewriter.rewriteClassName(className);
78     
79                 signature = "L" + className.replace('.', '/') + ";";
80             }
81     
82             return signature;
83         }
84     
85         /**
86          * Rewrite a MethodAnnotation to update the class name, and any class names
87          * mentioned in the method signature.
88          *
89          * @param classNameRewriter
90          *            a ClassNameRewriter
91          * @param annotation
92          *            a MethodAnnotation
93          * @return the possibly-rewritten MethodAnnotation
94          */
95         public static MethodAnnotation convertMethodAnnotation(ClassNameRewriter classNameRewriter, MethodAnnotation annotation) {
96     
97             if (classNameRewriter != IdentityClassNameRewriter.instance()) {
98                 annotation = new MethodAnnotation(classNameRewriter.rewriteClassName(annotation.getClassName()),
99                         annotation.getMethodName(), rewriteMethodSignature(classNameRewriter, annotation.getMethodSignature()),
100                         annotation.isStatic());
101             }
102             return annotation;
103         }
104     
105         /**
106          * Rewrite a FieldAnnotation to update the class name and field signature,
107          * if needed.
108          *
109          * @param classNameRewriter
110          *            a ClassNameRewriter
111          * @param annotation
112          *            a FieldAnnotation
113          * @return the possibly-rewritten FieldAnnotation
114          */
115         public static FieldAnnotation convertFieldAnnotation(ClassNameRewriter classNameRewriter, FieldAnnotation annotation) {
116     
117             if (classNameRewriter != IdentityClassNameRewriter.instance()) {
118                 annotation = new FieldAnnotation(classNameRewriter.rewriteClassName(annotation.getClassName()),
119                         annotation.getFieldName(), rewriteSignature(classNameRewriter, annotation.getFieldSignature()),
120                         annotation.isStatic());
121             }
122             return annotation;
123         }
124     
125     }
