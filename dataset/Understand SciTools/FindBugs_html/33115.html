20     package edu.umd.cs.findbugs.ba;
21     
22     import java.util.Map;
23     import java.util.TreeMap;
24     
25     import org.apache.bcel.classfile.JavaClass;
26     import org.apache.bcel.classfile.Method;
27     
28     import edu.umd.cs.findbugs.OpcodeStack;
29     import edu.umd.cs.findbugs.bcel.OpcodeStackDetector;
30     
31     /**
32      * @author pwilliam
33      */
34     public class PutfieldScanner {
35     
36         public static Map<Integer, OpcodeStack.Item> getPutfieldsFor(JavaClass theClass, Method method, XField field) {
37             Scanner scanner = new Scanner(theClass, method, field);
38     
39             scanner.execute();
40             return scanner.putfields;
41     
42         }
43     
44         static class Scanner extends OpcodeStackDetector {
45     
46             Map<Integer, OpcodeStack.Item> putfields = new TreeMap<Integer, OpcodeStack.Item>();
47     
48             public Scanner(JavaClass theClass, Method targetMethod, XField target) {
49                 this.theClass = theClass;
50                 this.targetMethod = targetMethod;
51                 this.targetField = target;
52             }
53     
54             final JavaClass theClass;
55     
56             final Method targetMethod;
57     
58             final XField targetField;
59     
60             @Override
61             public void sawOpcode(int seen) {
62                 if (seen != PUTFIELD) {
63                     return;
64                 }
65                 XField xFieldOperand = getXFieldOperand();
66                 if (xFieldOperand != null && xFieldOperand.equals(targetField) && stack.getStackItem(1).getRegisterNumber() == 0) {
67                     putfields.put(getPC(), new OpcodeStack.Item(stack.getStackItem(0)));
68                 }
69     
70             }
71     
72             @Override
73             public void visitJavaClass(JavaClass obj) {
74                 setupVisitorForClass(obj);
75                 getConstantPool().accept(this);
76                 doVisitMethod(targetMethod);
77             }
78     
79             public void execute() {
80                 theClass.accept(this);
81             }
82         }
83     
84     }
