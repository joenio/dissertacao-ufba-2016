19     package edu.umd.cs.findbugs.classfile;
20     
21     import java.util.Map;
22     
23     import javax.annotation.CheckForNull;
24     import javax.annotation.Nonnull;
25     
26     import edu.umd.cs.findbugs.log.Profiler;
27     
28     /**
29      * The analysis cache performs analyses on classes and methods and caches the
30      * results.
31      *
32      * @author David Hovemeyer
33      */
34     public interface IAnalysisCache {
35     
36         /**
37          * Register the given class analysis engine as producing the analysis result
38          * type whose Class is given.
39          *
40          * @param <E>
41          *            analysis result type
42          * @param analysisResultType
43          *            analysis result type Class object
44          * @param classAnalysisEngine
45          *            the class analysis engine to register
46          */
47         public <E> void registerClassAnalysisEngine(Class<E> analysisResultType, IClassAnalysisEngine<E> classAnalysisEngine);
48     
49         /**
50          * Register the given method analysis engine as producing the analysis
51          * result type whose Class is given.
52          *
53          * @param <E>
54          *            analysis result type
55          * @param analysisResultType
56          *            analysis result type Class object
57          * @param methodAnalysisEngine
58          *            the method analysis engine to register
59          */
60         public <E> void registerMethodAnalysisEngine(Class<E> analysisResultType, IMethodAnalysisEngine<E> methodAnalysisEngine);
61     
62         /**
63          * Get an analysis of the given class.
64          *
65          * @param <E>
66          *            the type of the analysis (e.g., FoobarAnalysis)
67          * @param analysisClass
68          *            the analysis class object (e.g., FoobarAnalysis.class)
69          * @param classDescriptor
70          *            the descriptor of the class to analyze
71          * @return the analysis object (e.g., instance of FoobarAnalysis for the
72          *         class)
73          * @throws CheckedAnalysisException
74          *             if an error occurs performing the analysis
75          */
76         public <E> E getClassAnalysis(Class<E> analysisClass, @Nonnull ClassDescriptor classDescriptor)
77                 throws CheckedAnalysisException;
78     
79         /**
80          * See if the cache contains a cached class analysis result for given class
81          * descriptor.
82          *
83          * @param analysisClass
84          *            analysis result class
85          * @param classDescriptor
86          *            the class descriptor
87          * @return a cached analysis result, or null if there is no cached analysis
88          *         result
89          */
90         public <E> E probeClassAnalysis(Class<E> analysisClass, @Nonnull ClassDescriptor classDescriptor);
91     
92         /**
93          * Get an analysis of the given method.
94          *
95          * @param <E>
96          *            the type of the analysis (e.g., FoobarAnalysis)
97          * @param analysisClass
98          *            the analysis class object (e.g., FoobarAnalysis.class)
99          * @param methodDescriptor
100          *            the descriptor of the method to analyze
101          * @return the analysis object (e.g., instance of FoobarAnalysis for the
102          *         method)
103          * @throws CheckedAnalysisException
104          *             if an error occurs performing the analysis
105          */
106         public <E> E getMethodAnalysis(Class<E> analysisClass, @Nonnull MethodDescriptor methodDescriptor)
107                 throws CheckedAnalysisException;
108     
109         /**
110          * Eagerly put a method analysis object in the cache. This can be necessary
111          * if an method analysis engine invokes other analysis engines that might
112          * recursively require the analysis being produced.
113          *
114          * @param <E>
115          *            the type of the analysis (e.g., FoobarAnalysis)
116          * @param analysisClass
117          *            the analysis class object (e.g., FoobarAnalysis.class)
118          * @param methodDescriptor
119          *            the descriptor of the method to analyze
120          * @param analysisObject
121          */
122         public <E> void eagerlyPutMethodAnalysis(Class<E> analysisClass, @Nonnull MethodDescriptor methodDescriptor,
123                 E analysisObject);
124     
125         /**
126          * Purge all analysis results for given method. This can be called when a
127          * CFG is pruned and we want to compute more accurate analysis results on
128          * the new CFG.
129          *
130          * @param methodDescriptor
131          *            method whose analysis results should be purged
132          */
133         public void purgeMethodAnalyses(@Nonnull MethodDescriptor methodDescriptor);
134     
135         /**
136          * Purge all method analysis results for all methods.
137          */
138         public void purgeAllMethodAnalysis();
139     
140         /**
141          * Purge all class analysis results of a particular kind
142          */
143         public void purgeClassAnalysis(Class<?> analysisClass);
144     
145         /**
146          * Register a database factory.
147          *
148          * @param <E>
149          *            type of database
150          * @param databaseClass
151          *            Class of database
152          * @param databaseFactory
153          *            the database factory
154          */
155         public <E> void registerDatabaseFactory(Class<E> databaseClass, IDatabaseFactory<E> databaseFactory);
156     
157         /**
158          * Get a database.
159          *
160          * <em>Note</em>: an unchecked analysis exception will be thrown if the
161          * database cannot be instantiated. Since instantiation of most kinds of
162          * databases simply involves creating an object (and not opening a file or
163          * other failure-prone operation), throwing a CheckedAnalysisException
164          * creates too great of an exception-handling burden on analyses and
165          * detectors which use databases.
166          *
167          * @param <E>
168          *            type of database
169          * @param databaseClass
170          *            Class of database
171          * @return the database (which is created by a database factory if required)
172          */
173         public <E> E getDatabase(Class<E> databaseClass);
174     
175         public @CheckForNull <E>  E getOptionalDatabase(Class<E> databaseClass);
176     
177         /**
178          * Eagerly install a database. This avoids the need to register a database
179          * factory.
180          *
181          * @param <E>
182          *            type of database
183          * @param databaseClass
184          *            Class of database
185          * @param database
186          *            database object
187          */
188         public <E> void eagerlyPutDatabase(Class<E> databaseClass, E database);
189     
190         /**
191          * Get the classpath from which classes are loaded.
192          *
193          * @return the classpath
194          */
195         public IClassPath getClassPath();
196     
197         /**
198          * Get the error logger.
199          *
200          * @return the error logger
201          */
202         public IErrorLogger getErrorLogger();
203     
204         /**
205          * Get map of analysis-local objects.
206          */
207         public Map<?, ?> getAnalysisLocals();
208     
209         /**
210          * Get the analysis profiler instance, never null
211          */
212         public Profiler getProfiler();
213     }
