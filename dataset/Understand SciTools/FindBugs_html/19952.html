20     package edu.umd.cs.findbugs;

22     import java.io.PrintStream;
23     import java.io.PrintWriter;
24     import java.util.Iterator;

26     import edu.umd.cs.findbugs.charsets.UTF8;

28     /**
29      * Base class for BugReporters which provides convenient formatting and
30      * reporting of warnings and analysis errors.
31      *
32      * <p>
33      * "TextUIBugReporter" is a bit of a misnomer, since this class is useful in
34      * GUIs, too.
35      * </p>
36      *
37      * @author David Hovemeyer
38      */
39     public abstract class TextUIBugReporter extends AbstractBugReporter {
40         private boolean reportStackTrace;
41     
42         private boolean useLongBugCodes = false;
43     
44         private boolean showRank = false;
45     
46         private boolean reportHistory = false;
47     
48         private boolean reportUserDesignations = false;
49     
50         private boolean applySuppressions = false;
51     
52         static final String OTHER_CATEGORY_ABBREV = "X";
53     
54         protected PrintWriter outputStream = UTF8.printWriter(System.out, true);
55     
56         public TextUIBugReporter() {
57             reportStackTrace = true;
58         }
59     
60         /**
61          * Set the PrintStream to write bug output to.
62          *
63          * @param outputStream
64          *            the PrintStream to write bug output to
65          */
66         public void setOutputStream(PrintStream outputStream) {
67             this.outputStream = UTF8.printWriter(outputStream, true);
68         }
69     
70         public void setWriter(PrintWriter writer) {
71             this.outputStream = writer;
72         }
73     
74         /**
75          * Set whether or not stack traces should be reported in error output.
76          *
77          * @param reportStackTrace
78          *            true if stack traces should be reported, false if not
79          */
80         public void setReportStackTrace(boolean reportStackTrace) {
81             this.reportStackTrace = reportStackTrace;
82         }
83     
84         /**
85          * Print bug in one-line format.
86          *
87          * @param bugInstance
88          *            the bug to print
89          */
90         protected void printBug(BugInstance bugInstance) {
91             if (showRank) {
92                 int rank = BugRanker.findRank(bugInstance);
93                 outputStream.printf("%2d ", rank);
94             }
95             switch (bugInstance.getPriority()) {
96             case Priorities.EXP_PRIORITY:
97                 outputStream.print("E ");
98                 break;
99             case Priorities.LOW_PRIORITY:
100                 outputStream.print("L ");
101                 break;
102             case Priorities.NORMAL_PRIORITY:
103                 outputStream.print("M ");
104                 break;
105             case Priorities.HIGH_PRIORITY:
106                 outputStream.print("H ");
107                 break;
108             default:
109                 assert false;
110             }
111     
112             BugPattern pattern = bugInstance.getBugPattern();
113             if (pattern != null) {
114                 String categoryAbbrev = null;
115                 BugCategory bcat = DetectorFactoryCollection.instance().getBugCategory(pattern.getCategory());
116                 if (bcat != null) {
117                     categoryAbbrev = bcat.getAbbrev();
118                 }
119                 if (categoryAbbrev == null) {
120                     categoryAbbrev = OTHER_CATEGORY_ABBREV;
121                 }
122                 outputStream.print(categoryAbbrev);
123                 outputStream.print(" ");
124             }
125     
126             if (useLongBugCodes) {
127                 outputStream.print(bugInstance.getType());
128                 outputStream.print(" ");
129             }
130             if (reportUserDesignations) {
131                 outputStream.print(bugInstance.getUserDesignationKey());
132                 outputStream.print(" ");
133             }
134     
135             if (reportHistory) {
136                 long first = bugInstance.getFirstVersion();
137                 long last = bugInstance.getLastVersion();
138                 outputStream.print(first);
139                 outputStream.print(" ");
140                 outputStream.print(last);
141                 outputStream.print(" ");
142             }
143             SourceLineAnnotation line = bugInstance.getPrimarySourceLineAnnotation();
144             outputStream.println(bugInstance.getMessage().replace('\n', ' ') + "  " + line.toString());
145         }
146     
147         private boolean analysisErrors;
148     
149         private boolean missingClasses;
150     
151         @Override
152         public void reportQueuedErrors() {
153             boolean errors = analysisErrors || missingClasses || getQueuedErrors().size() > 0;
154             analysisErrors = missingClasses = false;
155             super.reportQueuedErrors();
156             if (errors) {
157                 emitLine("");
158             }
159         }
160     
161         @Override
162         public void reportAnalysisError(AnalysisError error) {
163             if (!analysisErrors) {
164                 emitLine("The following errors occurred during analysis:");
165                 analysisErrors = true;
166             }
167             emitLine("\t" + error.getMessage());
168             if (error.getExceptionMessage() != null) {
169                 emitLine("\t\t" + error.getExceptionMessage());
170                 if (reportStackTrace) {
171                     String[] stackTrace = error.getStackTrace();
172                     if (stackTrace != null) {
173                         for (String aStackTrace : stackTrace) {
174                             emitLine("\t\t\tAt " + aStackTrace);
175                         }
176                     }
177                 }
178             }
179         }
180     
181         @Override
182         public void reportMissingClass(String message) {
183             if (!missingClasses) {
184                 emitLine("The following classes needed for analysis were missing:");
185                 missingClasses = true;
186             }
187             emitLine("\t" + message);
188         }
189     
190         /**
191          * Emit one line of the error message report. By default, error messages are
192          * printed to System.err. Subclasses may override.
193          *
194          * @param line
195          *            one line of the error report
196          */
197         protected void emitLine(String line) {
198             line = line.replaceAll("\t", "  ");
199             System.err.println(line);
200         }
201     
202         public boolean getUseLongBugCodes() {
203             return useLongBugCodes;
204         }
205     
206         public void setReportHistory(boolean reportHistory) {
207             this.reportHistory = reportHistory;
208         }
209     
210         public void setUseLongBugCodes(boolean useLongBugCodes) {
211             this.useLongBugCodes = useLongBugCodes;
212         }
213     
214         public void setShowRank(boolean showRank) {
215             this.showRank = showRank;
216         }
217     
218         public void setApplySuppressions(boolean applySuppressions) {
219             this.applySuppressions = applySuppressions;
220         }
221     
222         public void setReportUserDesignations(boolean reportUserDesignations) {
223             this.reportUserDesignations = reportUserDesignations;
224         }
225     
226         /*
227          * (non-Javadoc)
228          *
229          * @see edu.umd.cs.findbugs.BugReporter#getRealBugReporter()
230          */
231         public BugReporter getRealBugReporter() {
232             return this;
233         }
234     
235         /**
236          * For debugging: check a BugInstance to make sure it is valid.
237          *
238          * @param bugInstance
239          *            the BugInstance to check
240          */
241         protected void checkBugInstance(BugInstance bugInstance) {
242             for (Iterator<BugAnnotation> i = bugInstance.annotationIterator(); i.hasNext();) {
243                 BugAnnotation bugAnnotation = i.next();
244                 if (bugAnnotation instanceof PackageMemberAnnotation) {
245                     PackageMemberAnnotation pkgMember = (PackageMemberAnnotation) bugAnnotation;
246                     if (pkgMember.getSourceLines() == null) {
247                         throw new IllegalStateException("Package member " + pkgMember + " reported without source lines!");
248                     }
249                 }
250             }
251         }
252     
253         public boolean isApplySuppressions() {
254             return applySuppressions;
255         }
256     
257     }
