36             assertEquals("0x7fffffff", IntAnnotation.getShortInteger(0x7fffffff));
37             assertEquals("15", IntAnnotation.getShortInteger(15));
38             assertEquals("255", IntAnnotation.getShortInteger(255));
39             assertEquals("-1", IntAnnotation.getShortInteger(-1));
40             assertEquals("-2", IntAnnotation.getShortInteger(-2));
41             assertEquals( "0xffffffff",  IntAnnotation.getShortInteger(0xffffffffL));
42             assertEquals("0x1ffffffff", IntAnnotation.getShortInteger(0x1ffffffffL));
43             assertEquals("0xfffffffff", IntAnnotation.getShortInteger(0xfffffffffL));
