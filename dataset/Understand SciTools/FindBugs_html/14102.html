20     package edu.umd.cs.findbugs.xml;
21     
22     import java.io.IOException;
23     
24     import edu.umd.cs.findbugs.annotations.CleanupObligation;
25     import edu.umd.cs.findbugs.annotations.DischargesObligation;
26     
27     /**
28      * Interface to generate an XML document in some form. E.g., writing it to a
29      * stream, generating SAX events, etc.
30      *
31      * @author David Hovemeyer
32      */
33     @CleanupObligation
34     public interface XMLOutput {
35         /**
36          * Begin the XML document.
37          */
38         public void beginDocument() throws IOException;
39     
40         /**
41          * Open a tag with given name.
42          *
43          * @param tagName
44          *            the tag name
45          */
46         public void openTag(String tagName) throws IOException;
47     
48         /**
49          * Open a tag with given name and given attributes.
50          *
51          * @param tagName
52          *            the tag name
53          * @param attributeList
54          *            the attributes
55          */
56         public void openTag(String tagName, XMLAttributeList attributeList) throws IOException;
57     
58         /**
59          * Start a tag, with the intention of adding attributes. Must be followed by
60          * stopTag after zero or more addAttribute calls.
61          *
62          * @param tagName
63          *            the tag name
64          */
65         public void startTag(String tagName) throws IOException;
66     
67         /**
68          * Add an attribute to a started tag. Must follow a call to startTag.
69          *
70          * @param name
71          *            the attribute name.
72          * @param value
73          *            the attribute value, unescaped.
74          */
75         public void addAttribute(String name, String value) throws IOException;
76     
77         /**
78          * End a started tag. Must follow a call to startTag.
79          *
80          * @param close
81          *            true if the element has no content.
82          */
83         public void stopTag(boolean close) throws IOException;
84     
85         /**
86          * Open and close tag with given name.
87          *
88          * @param tagName
89          *            the tag name
90          */
91         public void openCloseTag(String tagName) throws IOException;
92     
93         /**
94          * Open and close tag with given name and given attributes.
95          *
96          * @param tagName
97          *            the tag name
98          * @param attributeList
99          *            the attributes
100          */
101         public void openCloseTag(String tagName, XMLAttributeList attributeList) throws IOException;
102     
103         /**
104          * Close tag with given name.
105          *
106          * @param tagName
107          *            the tag name
108          */
109         public void closeTag(String tagName) throws IOException;
110     
111         /**
112          * Write text to the XML document. XML metacharacters are automatically
113          * escaped.
114          *
115          * @param text
116          *            the text to write
117          */
118         public void writeText(String text) throws IOException;
119     
120         /**
121          * Write a CDATA section to the XML document. The characters are not escaped
122          * in any way.
123          *
124          * @param cdata
125          *            the character data to write
126          */
127         public void writeCDATA(String cdata) throws IOException;
128     
129         /**
130          * Finish writing XML output, closing any underlying resources (such as
131          * output streams). A call to this method should always be made, even if one
132          * of the XML-generation methods throws an exception. Therefore, a call to
133          * this method should be performed in a finally block.
134          */
135         @DischargesObligation
136         public void finish() throws IOException;
137     }
