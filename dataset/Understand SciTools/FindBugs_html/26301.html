42     /**
43      * Factory for creating ClassDescriptors, MethodDescriptors, and
44      * FieldDescriptors.
45      *
46      * @author David Hovemeyer
47      */
48     public class DescriptorFactory {
49         private static ThreadLocal<DescriptorFactory> instanceThreadLocal = new ThreadLocal<DescriptorFactory>() {
50             @Override
51             protected DescriptorFactory initialValue() {
52                 return new DescriptorFactory();
53             }
54         };
55     
56         private final Map<String, ClassDescriptor> classDescriptorMap;
57     
58         private final Map<String, ClassDescriptor> dottedClassDescriptorMap;
59     
60         private final Map<MethodDescriptor, MethodDescriptor> methodDescriptorMap;
61     
62         private final Map<FieldDescriptor, FieldDescriptor> fieldDescriptorMap;
63     
64         private DescriptorFactory() {
65             this.classDescriptorMap = new HashMap<String, ClassDescriptor>();
66             this.dottedClassDescriptorMap = new HashMap<String, ClassDescriptor>();
67             this.methodDescriptorMap = new HashMap<MethodDescriptor, MethodDescriptor>();
68             this.fieldDescriptorMap = new HashMap<FieldDescriptor, FieldDescriptor>();
69         }
70     
71         private final MapCache<String, String> stringCache = new MapCache<String, String>(10000);
72     
73         public static String canonicalizeString(@CheckForNull String s) {
74             if (s == null) {
75                 return s;
76             }
77             DescriptorFactory df =  instanceThreadLocal.get();
78             String cached = df.stringCache.get(s);
79             if (cached != null) {
80                 return cached;
81             }
82             df.stringCache.put(s, s);
83             return s;
84         }

103         public void purge(Collection<ClassDescriptor> unusable) {
104             for (ClassDescriptor c : unusable) {
105                 classDescriptorMap.remove(c.getClassName());
106                 dottedClassDescriptorMap.remove(c.getClassName().replace('/', '.'));
107             }
108         }

115         /**
116          * Get a ClassDescriptor for a class name in VM (slashed) format.
117          *
118          * @param className
119          *            a class name in VM (slashed) format
120          * @return ClassDescriptor for that class
121          */
122         public @Nonnull
123         ClassDescriptor getClassDescriptor(@SlashedClassName String className) {
124             assert className.indexOf('.') == -1;
125             className = canonicalizeString(className);
126             ClassDescriptor classDescriptor = classDescriptorMap.get(className);
127             if (classDescriptor == null) {
128                 classDescriptor = new ClassDescriptor(className);
129                 classDescriptorMap.put(className, classDescriptor);
130             }
131             return classDescriptor;
132         }
133     
134         /**
135          * Get a ClassDescriptor for a class name in dotted format.
136          *
137          * @param dottedClassName
138          *            a class name in dotted format
139          * @return ClassDescriptor for that class
140          */
141         public ClassDescriptor getClassDescriptorForDottedClassName(@DottedClassName String dottedClassName) {
142             assert dottedClassName != null;
143             ClassDescriptor classDescriptor = dottedClassDescriptorMap.get(dottedClassName);
144             if (classDescriptor == null) {
145                 classDescriptor = getClassDescriptor(dottedClassName.replace('.', '/'));
146                 dottedClassDescriptorMap.put(dottedClassName, classDescriptor);
147             }
148             return classDescriptor;
149         }

153                     method.isStatic());
154         }

156         /**
157          * Get a MethodDescriptor.
158          *
159          * @param className
160          *            name of the class containing the method, in VM format (e.g.,
161          *            "java/lang/String")
162          * @param name
163          *            name of the method
164          * @param signature
165          *            signature of the method
166          * @param isStatic
167          *            true if method is static, false otherwise
168          * @return MethodDescriptor
169          */
170         public MethodDescriptor getMethodDescriptor(@SlashedClassName String className, String name, String signature,
171                 boolean isStatic) {
172             if (className == null) {
173                 throw new NullPointerException("className must be nonnull");
174             }
175             MethodDescriptor methodDescriptor = new MethodDescriptor(className, name, signature, isStatic);
176             MethodDescriptor existing = methodDescriptorMap.get(methodDescriptor);
177             if (existing == null) {
178                 methodDescriptorMap.put(methodDescriptor, methodDescriptor);
179                 existing = methodDescriptor;
180             }
181             return existing;
182         }

184         public void profile() {
185             int total = 0;
186             int keys = 0;
187             int values = 0;
188             int bad = 0;
189             for (Map.Entry<MethodDescriptor, MethodDescriptor> e : methodDescriptorMap.entrySet()) {
190                 total++;
191                 if (e.getKey() instanceof MethodInfo) {
192                     keys++;
193                 }
194                 if (e.getValue() instanceof MethodInfo) {
195                     values++;
196                 }
197             }
198             System.out.printf("Descriptor factory: %d/%d/%d%n", keys, values, total);
199     
200         }

203             MethodDescriptor existing = methodDescriptorMap.get(m);

272         /**
273          * Create a class descriptor from a resource name.
274          *
275          * @param resourceName
276          *            the resource name
277          * @return the class descriptor
278          */
279         public static ClassDescriptor createClassDescriptorFromResourceName(String resourceName) {
280             if (!isClassResource(resourceName)) {
281                 throw new IllegalArgumentException("Resource " + resourceName + " is not a class");
282             }
283             return createClassDescriptor(resourceName.substring(0, resourceName.length() - 6));
284         }
285     
286         /**
287          * Create a class descriptor from a field signature
288          *
289          */
290         public static @CheckForNull
291         ClassDescriptor createClassDescriptorFromFieldSignature(String signature) {
292             int start = signature.indexOf('L');
293             if (start < 0) {
294                 return null;
295             }
296             int end = signature.indexOf(';', start);
297             if (end < 0) {
298                 return null;
299             }
300             return createClassDescriptor(signature.substring(start + 1, end));
301         }
302     
303         /**
304          * Determine whether or not the given resource name refers to a class.
305          *
306          * @param resourceName
307          *            the resource name
308          * @return true if the resource is a class, false otherwise
309          */
310         public static boolean isClassResource(String resourceName) {
311             // This could be more sophisticated.
312             return resourceName.endsWith(".class");
313         }
314     
315         public static ClassDescriptor createClassDescriptorFromSignature(String signature) {
316             int length = signature.length();
317             if (length == 0) {
318                 throw new IllegalArgumentException("Empty signature");
319             }
320             if (signature.charAt(0) == 'L' && signature.endsWith(";")) {
321                 signature = signature.substring(1, signature.length() - 1);
322             }
323             return createClassDescriptor(signature);
324         }
325     
326         public static ClassDescriptor createClassOrObjectDescriptorFromSignature(String signature) {
327             if (signature.charAt(0) == '[') {
328                 return createClassDescriptor("java/lang/Object");
329             }
330             return createClassDescriptorFromSignature(signature);
331         }

341         public static ClassDescriptor[] createClassDescriptor(String[] classNames) {
342             if (classNames.length == 0) {
343                 return ClassDescriptor.EMPTY_ARRAY;
344             }
345             ClassDescriptor[] result = new ClassDescriptor[classNames.length];
346             for (int i = 0; i < classNames.length; i++) {
347                 result[i] = createClassDescriptor(classNames[i]);
348             }
349             return result;
350         }
351     
352         public static ClassDescriptor createClassDescriptorFromDottedClassName(String dottedClassName) {
353             return createClassDescriptor(dottedClassName.replace('.', '/'));
354         }
