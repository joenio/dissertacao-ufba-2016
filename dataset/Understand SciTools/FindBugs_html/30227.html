20     package edu.umd.cs.findbugs.ba;
21     
22     import java.util.IdentityHashMap;
23     
24     import org.apache.bcel.classfile.LineNumber;
25     import org.apache.bcel.classfile.LineNumberTable;
26     import org.apache.bcel.generic.InstructionHandle;
27     import org.apache.bcel.generic.MethodGen;
28     
29     import edu.umd.cs.findbugs.SystemProperties;
30     
31     /**
32      * Summarize line numbers (and other source information) for a method.
33      */
34     public class LineNumberMap {
35         /**
36          * Set this property to true to get debug print statements.
37          */
38         private static final boolean DEBUG = SystemProperties.getBoolean("lnm.debug");
39     
40         /**
41          * When this is true, the workaround for the bug in BCEL 5.0's
42          * LineNumberTable class is disabled.
43          */
44         private static final boolean LINE_NUMBER_BUG = SystemProperties.getBoolean("lineNumberBug");
45     
46         private final MethodGen methodGen;
47     
48         private final IdentityHashMap<InstructionHandle, LineNumber> lineNumberMap;
49     
50         private boolean hasLineNumbers;
51     
52         /**
53          * Constructor.
54          *
55          * @param methodGen
56          *            the method to summarize line numbers for
57          */
58         public LineNumberMap(MethodGen methodGen) {
59             this.methodGen = methodGen;
60             lineNumberMap = new IdentityHashMap<InstructionHandle, LineNumber>();
61             hasLineNumbers = false;
62         }
63     
64         /**
65          * Build the line number information. Should be called before any other
66          * methods.
67          */
68         public void build() {
69             int numGood = 0, numBytecodes = 0;
70     
71             if (DEBUG) {
72                 System.out.println("Method: " + methodGen.getName() + " - " + methodGen.getSignature() + "in class "
73                         + methodGen.getClassName());
74             }
75     
76             // Associate line number information with each InstructionHandle
77             LineNumberTable table = methodGen.getLineNumberTable(methodGen.getConstantPool());
78     
79             if (table != null && table.getTableLength() > 0) {
80                 checkTable(table);
81                 InstructionHandle handle = methodGen.getInstructionList().getStart();
82                 while (handle != null) {
83                     int bytecodeOffset = handle.getPosition();
84                     if (bytecodeOffset < 0) {
85                         throw new IllegalStateException("Bad bytecode offset: " + bytecodeOffset);
86                     }
87                     if (DEBUG) {
88                         System.out.println("Looking for source line for bytecode offset " + bytecodeOffset);
89                     }
90                     int sourceLine;
91                     try {
92                         sourceLine = table.getSourceLine(bytecodeOffset);
93                     } catch (ArrayIndexOutOfBoundsException e) {
94                         if (LINE_NUMBER_BUG) {
95                             throw e;
96                         } else {
97                             sourceLine = -1;
98                         }
99                     }
100                     if (sourceLine >= 0) {
101                         ++numGood;
102                     }
103                     lineNumberMap.put(handle, new LineNumber(bytecodeOffset, sourceLine));
104                     handle = handle.getNext();
105                     ++numBytecodes;
106                 }
107                 hasLineNumbers = true;
108     
109                 if (DEBUG) {
110                     System.out.println("\t" + numGood + "/" + numBytecodes + " had valid line numbers");
111                 }
112             }
113         }
114     
115         private void checkTable(LineNumberTable table) {
116             if (DEBUG) {
117                 System.out.println("line number table has length " + table.getTableLength());
118             }
119             LineNumber[] entries = table.getLineNumberTable();
120             int lastBytecode = -1;
121             for (int i = 0; i < entries.length; ++i) {
122                 LineNumber ln = entries[i];
123                 if (DEBUG) {
124                     System.out.println("Entry " + i + ": pc=" + ln.getStartPC() + ", line=" + ln.getLineNumber());
125                 }
126                 int pc = ln.getStartPC();
127                 if (pc <= lastBytecode) {
128                     throw new IllegalStateException("LineNumberTable is not sorted");
129                 }
130             }
131         }
132     
133         /**
134          * Does this method have line number information?
135          */
136         public boolean hasLineNumbers() {
137             return hasLineNumbers;
138         }
139     
140         /**
141          * Find the line number information for instruction whose handle is given.
142          *
143          * @param handle
144          *            the InstructionHandle
145          * @return the LineNumber object containing bytecode offset and source line
146          *         number
147          */
148         public LineNumber lookupLineNumber(InstructionHandle handle) {
149             return lineNumberMap.get(handle);
150         }
151     }
