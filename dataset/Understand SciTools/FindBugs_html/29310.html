46     
47     public class ReadOfInstanceFieldInMethodInvokedByConstructorInSuperclass extends OpcodeStackDetector {
48     
49         final BugAccumulator accumulator;
50     
51         public ReadOfInstanceFieldInMethodInvokedByConstructorInSuperclass(BugReporter bugReporter) {
52             this.accumulator = new BugAccumulator(bugReporter);
53         }
54     
55         Set<XField> initializedFields, nullCheckedFields;
56     
57         @Override
58         public void visit(Code obj) {
59             if (getMethod().isStatic()) {
60                 return;
61             }
62             initializedFields = new HashSet<XField>();
63             nullCheckedFields = new HashSet<XField>();
64             super.visit(obj);
65             accumulator.reportAccumulatedBugs();
66         }
67     
68         @Override
69         public void sawOpcode(int opcode) {
70             if (opcode == PUTFIELD) {
71                 XField f = getXFieldOperand();
72                 OpcodeStack.Item item = stack.getStackItem(1);
73                 if (item.getRegisterNumber() != 0) {
74                     return;
75                 }
76                 initializedFields.add(f);
77                 return;
78             }
79             if (opcode != GETFIELD) {
80                 return;
81             }
82             OpcodeStack.Item item = stack.getStackItem(0);
83             if (item.getRegisterNumber() != 0) {
84                 return;
85             }
86             XField f = getXFieldOperand();
87     
88             if (f == null || !f.getClassDescriptor().equals(getClassDescriptor())) {
89                 return;
90             }
91             if (f.isSynthetic() || f.getName().startsWith("this$")) {
92                 return;
93             }
94             if (initializedFields.contains(f)) {
95                 return;
96             }
97             FieldSummary fieldSummary = AnalysisContext.currentAnalysisContext().getFieldSummary();
98     
99             ClassDescriptor superClassDescriptor = DescriptorFactory.createClassDescriptor(getSuperclassName());
100             Set<ProgramPoint> calledFrom = fieldSummary.getCalledFromSuperConstructor(superClassDescriptor, getXMethod());
101             if (calledFrom.isEmpty()) {
102                 return;
103             }
104             UnreadFieldsData unreadFields = AnalysisContext.currentAnalysisContext().getUnreadFieldsData();
105     
106             int priority;
107             if (!unreadFields.isWrittenInConstructor(f)) {
108                 return;
109             }
110     
111             if (f.isFinal()) {
112                 priority = HIGH_PRIORITY;
113             } else if (unreadFields.isWrittenDuringInitialization(f) || unreadFields.isWrittenOutsideOfInitialization(f)) {
114                 priority = NORMAL_PRIORITY;
115             } else {
116                 priority = HIGH_PRIORITY;
117             }
118     
119             int nextOpcode = getNextOpcode();
120             if (nullCheckedFields.contains(f) || nextOpcode == IFNULL || nextOpcode == IFNONNULL || nextOpcode == IFEQ
121                     || nextOpcode == IFNE) {
122                 priority++;
123                 nullCheckedFields.add(f);
124             }
125     
126             for (ProgramPoint p : calledFrom) {
127                 XMethod upcall = getConstructorThatCallsSuperConstructor(p.method);
128                 if (upcall == null) {
129                     continue;
130                 }
131                 Method upcallMethod = null;
132                 for (Method m : getThisClass().getMethods()) {
133                     if (m.getName().equals(upcall.getName()) && m.getSignature().equals(upcall.getSignature())) {
134                         upcallMethod = m;
135                         break;
136                     }
137                 }
138                 if (upcallMethod == null) {
139                     continue;
140                 }
141                 Map<Integer, OpcodeStack.Item> putfieldsAt = PutfieldScanner.getPutfieldsFor(getThisClass(), upcallMethod, f);
142                 if (putfieldsAt.isEmpty()) {
143                     continue;
144                 }
145                 Map.Entry<Integer, OpcodeStack.Item> e = putfieldsAt.entrySet().iterator().next();
146                 int pc = e.getKey();
147                 OpcodeStack.Item value = e.getValue();
148                 if (value.isNull() || value.hasConstantValue(0)) {
149                     priority++;
150                 }
151     
152                 SourceLineAnnotation fieldSetAt = SourceLineAnnotation.fromVisitedInstruction(getThisClass(), upcallMethod, pc);
153     
154                 BugInstance bug = new BugInstance(this, "UR_UNINIT_READ_CALLED_FROM_SUPER_CONSTRUCTOR", priority).addClassAndMethod(
155                         this).addField(f);
156                 bug.addMethod(p.method).describe(MethodAnnotation.METHOD_SUPERCLASS_CONSTRUCTOR)
157                 .addSourceLine(p.getSourceLineAnnotation()).describe(SourceLineAnnotation.ROLE_CALLED_FROM_SUPERCLASS_AT)
158                 .addMethod(upcall).describe(MethodAnnotation.METHOD_CONSTRUCTOR).add(fieldSetAt)
159                 .describe(SourceLineAnnotation.ROLE_FIELD_SET_TOO_LATE_AT);
160     
161                 accumulator.accumulateBug(bug, this);
162             }
163     
164         }
165     
166         private @CheckForNull
167         XMethod getConstructorThatCallsSuperConstructor(XMethod superConstructor) {
168             FieldSummary fieldSummary = AnalysisContext.currentAnalysisContext().getFieldSummary();
169     
170             XMethod lookfor = "()V".equals(superConstructor.getSignature()) ? null : superConstructor;
171             for (XMethod m : getXClass().getXMethods()) {
172                 if ("<init>".equals(m.getName())) {
173                     if (fieldSummary.getSuperCall(m) == lookfor) {
174                         return m;
175                     }
176                 }
177             }
178             return null;
179         }
