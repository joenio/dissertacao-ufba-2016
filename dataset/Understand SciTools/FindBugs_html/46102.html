198             public void visitLdcInsn(Object cst) {
199                 if (cst.equals("Stub!")) {

207             @Override
208             public void visitInsn(int opcode) {
209                 switch (opcode) {
210                 case Constants.MONITORENTER:
211                     mBuilder.setUsesConcurrency();
212                     break;
213                 case Constants.ARETURN:
214                 case Constants.IRETURN:
215                 case Constants.LRETURN:
216                 case Constants.DRETURN:
217                 case Constants.FRETURN:
218                     if (identityState == IdentityMethodState.LOADED_PARAMETER) {
219                         mBuilder.setIsIdentity();
220                     }
221                     sawReturn = true;
222                     break;
223                 case Constants.RETURN:
224                     sawReturn = true;
225                     break;
226                 case Constants.ATHROW:
227                     if (stubState == StubState.INITIALIZE_RUNTIME) {
228                         sawStubThrow = true;
229                     } else if (justSawInitializationOfUnsupportedOperationException) {
230                         sawUnsupportedThrow = true;
231                     } else {
232                         sawNormalThrow = true;
233                     }
234                     break;
235                 default:
236                     break;
237                 }
238     
239                 resetState();
240             }
241     
242             public void resetState() {
243                 stubState = StubState.INITIAL;
244             }

246             @Override
247             public void visitSomeInsn() {
248                 identityState = IdentityMethodState.NOT;
249                 parameterLoadState = ParameterLoadState.OTHER;
250                 resetState();

253             @Override
254             public void visitVarInsn(int opcode, int var) {
255     
256                 boolean match = false;
257                 if (parameterLoadState == ParameterLoadState.OTHER && !isStatic() && var == 0) {
258                     parameterLoadState = ParameterLoadState.LOADED_THIS;
259     
260                     match = true;
261                 }
262                 else if (parameterLoadState == ParameterLoadState.LOADED_THIS  && var > 0){
263                     parameterLoadState = ParameterLoadState.LOADED_THIS_AND_PARAMETER;
264                     parameterForLoadState = var;
265                     match = true;
266                 }
267     
268                 if (identityState == IdentityMethodState.INITIAL) {
269                     match = true;
270                     if (var > 0 || isStatic()) {
271                         identityState = IdentityMethodState.LOADED_PARAMETER;
272                     } else {
273                         identityState = IdentityMethodState.NOT;
274                     }
275     
276                 }
277                 if (!match) {
278                     visitSomeInsn();
279                 }
280             }

282             @Override
283             public void visitFieldInsn(int opcode,
284                     String owner,
285                     String name,
286                     String desc) {
287                 if (opcode == Opcodes.PUTFIELD && parameterLoadState == ParameterLoadState.LOADED_THIS_AND_PARAMETER
288                         && owner.equals(slashedClassName) && name.startsWith("this$")) {
289                     mBuilder.setVariableIsSynthetic(parameterForLoadState);
290                 }
291                 fieldInstructionCount++;
292     
293                 if (isAccessMethod && this.accessOwner == null) {
294                     this.accessOwner = owner;
295                     this.accessName = name;
296                     this.accessDesc = desc;
297                     this.accessIsStatic = opcode == Opcodes.GETSTATIC || opcode == Opcodes.PUTSTATIC;
298                     this.accessForField = true;
299                 }
300                 visitSomeInsn();
301             }

316             @Override
317             public void visitMethodInsn(int opcode, String owner, String name, String desc, boolean itf) {
318                 identityState = IdentityMethodState.NOT;
319                 methodCallCount++;
320                 if (isAccessMethod && this.accessOwner == null) {
321                     this.accessOwner = owner;
322                     this.accessName = name;
323                     this.accessDesc = desc;
324                     this.accessIsStatic = opcode == Opcodes.INVOKESTATIC;
325                     this.accessForField = false;
326                 }
327                 if (stubState == StubState.LOADED_STUB && opcode == Opcodes.INVOKESPECIAL
328                         && "java/lang/RuntimeException".equals(owner) && "<init>".equals(name)) {
329                     stubState = StubState.INITIALIZE_RUNTIME;
330                 } else {
331                     stubState = StubState.INITIAL;
332                 }
333                 if (owner.startsWith("java/util/concurrent")) {
334                     mBuilder.setUsesConcurrency();
335                 }
336                 if (opcode == Opcodes.INVOKEINTERFACE) {
337                     return;
338                 }
339     
340                 if (owner.charAt(0) == '[' && owner.charAt(owner.length() - 1) != ';') {
341                     // primitive array
342                     return;
343                 }
344                 if (opcode == Opcodes.INVOKESTATIC && "java/lang/System".equals(owner) && "exit".equals(name)
345                         && !sawReturn) {
346                     sawSystemExit = true;
347                 }
348                 justSawInitializationOfUnsupportedOperationException = opcode == Opcodes.INVOKESPECIAL
349                         && "java/lang/UnsupportedOperationException".equals(owner) && "<init>".equals(name);
350     
351                 if (isBridge && bridgedMethodSignature == null) {
352                     switch (opcode) {
353                     case Opcodes.INVOKEVIRTUAL:
354                     case Opcodes.INVOKESPECIAL:
355                     case Opcodes.INVOKESTATIC:
356                     case Opcodes.INVOKEINTERFACE:
357                         if (desc != null && name.equals(methodName)) {
358                             bridgedMethodSignature = desc;
359                         }
360                         break;
361                     default:
362                         break;
363                     }
364                 }
365     
366                 // System.out.println("Call from " +
367                 // ClassParserUsingASM.this.slashedClassName +
368                 // " to " + owner + " : " + desc);
369                 if (desc == null || desc.indexOf('[') == -1 && desc.indexOf('L') == -1) {
370                     return;
371                 }
372                 if (ClassParserUsingASM.this.slashedClassName.equals(owner)) {
373                     return;
374                 }
375                 ClassDescriptor classDescriptor = DescriptorFactory.instance().getClassDescriptor(owner);
376                 calledClassSet.add(classDescriptor);
377     
378             }

380             private void sawBranchTo(Label label) {
381                 sawBranch = true;

421             @Override
422             public void visitEnd() {
423                 labelsSeen.clear();
424                 if (isAccessMethod && accessOwner != null) {
425                     if (!accessForField && methodCallCount == 1) {
426                         mBuilder.setAccessMethodForMethod(accessOwner, accessName, accessDesc, accessIsStatic);
427                     } else if(accessForField && fieldInstructionCount == 1) {
428                         boolean isSetter = methodDesc.endsWith(")V");
429                         int numArg = new SignatureParser(methodDesc).getNumParameters();
430                         int expected = 0;
431                         if (!accessIsStatic) {
432                             expected++;
433                         }
434                         if (isSetter) {
435                             expected++;
436                         }
437                         boolean OK;
438                         if (isSetter) {
439                             OK = methodDesc.substring(1).startsWith(ClassName.toSignature(accessOwner) + accessDesc);
440                         } else {
441                             OK = methodDesc.substring(1).startsWith(ClassName.toSignature(accessOwner));
442                         }
443                         if (numArg == expected && OK) {
444                             mBuilder.setAccessMethodForField(accessOwner, accessName, accessDesc, accessIsStatic);
445                         }
446                     }
447                 }
448                 if (sawBackBranch) {
449                     mBuilder.setHasBackBranch();
450                 }
451                 boolean sawThrow = sawNormalThrow | sawUnsupportedThrow | sawStubThrow;
452                 if (sawThrow && !sawReturn || sawSystemExit && !sawBranch) {
453     
454                     mBuilder.setIsUnconditionalThrower();
455                     if (!sawReturn && !sawNormalThrow) {
456                         if (sawUnsupportedThrow) {
457                             mBuilder.setUnsupported();
458                         }
459                         if (sawStubThrow) {
460                             mBuilder.addAccessFlags(Constants.ACC_SYNTHETIC);
461                             mBuilder.setIsStub();
462     
463                         }
464                     }
465                     // else
466                     // System.out.println(slashedClassName+"."+methodName+methodDesc
467                     // + " is thrower");
468                 }
469                 mBuilder.setNumberMethodCalls(methodCallCount);
470                 MethodInfo methodInfo = mBuilder.build();
471                 Builder classBuilder = (ClassInfo.Builder) cBuilder;
472                 if (isBridge && bridgedMethodSignature != null && !bridgedMethodSignature.equals(methodDesc)) {
473                     classBuilder.addBridgeMethodDescriptor(methodInfo, bridgedMethodSignature);
474                 } else {
475                     classBuilder.addMethodDescriptor(methodInfo);
476                 }
477     
478                 if (methodInfo.usesConcurrency()) {
479                     classBuilder.setUsesConcurrency();
480                 }
481                 if (methodInfo.isStub()) {
482                     classBuilder.setHasStubs();
483                 }
484             }

512         @Override
513         public void parse(final ClassNameAndSuperclassInfo.Builder cBuilder) throws InvalidClassFileFormatException {
514     
515             cBuilder.setCodeBaseEntry(codeBaseEntry);
516     
517             final TreeSet<ClassDescriptor> calledClassSet = new TreeSet<ClassDescriptor>();
518     
519             classReader.accept(new ClassVisitor(FindBugsASM.ASM_VERSION) {
520     
521                 //            boolean isInnerClass = false;
522     
523                 @Override
524                 public void visit(int version, int access, String name, String signature, String superName, String[] interfaces) {
525                     ClassParserUsingASM.this.slashedClassName = name;
526                     cBuilder.setClassfileVersion(version >>> 16, version & 0xffff);
527                     cBuilder.setAccessFlags(access);
528                     cBuilder.setClassDescriptor(DescriptorFactory.createClassDescriptor(name));
529                     cBuilder.setInterfaceDescriptorList(DescriptorFactory.createClassDescriptor(interfaces));
530                     if (superName != null) {
531                         cBuilder.setSuperclassDescriptor(DescriptorFactory.createClassDescriptor(superName));
532                     }
533                     if (cBuilder instanceof ClassInfo.Builder) {
534                         ((ClassInfo.Builder) cBuilder).setSourceSignature(signature);
535                     }
536                 }
537     
538                 @Override
539                 public org.objectweb.asm.AnnotationVisitor visitAnnotation(String desc, boolean isVisible) {
540                     if (cBuilder instanceof ClassInfo.Builder) {
541                         AnnotationValue value = new AnnotationValue(desc);
542                         ((ClassInfo.Builder) cBuilder).addAnnotation(desc, value);
543                         return value.getAnnotationVisitor();
544                     }
545                     return null;
546                 }
547     
548                 @Override
549                 public void visitAttribute(Attribute arg0) {
550                     //
551                 }
552     
553                 @Override
554                 public void visitEnd() {
555                     //
556                 }
557     
558                 @Override
559                 public FieldVisitor visitField(int access, String name, String desc, String signature, Object value) {
560                     //                if (name.equals("this$0"))
561                     //                    isInnerClass = true;
562     
563                     if (desc == null) {
564                         throw new NullPointerException("Description cannot be null");
565                     }
566                     if (cBuilder instanceof ClassInfo.Builder) {
567                         final ClassInfo.Builder cBuilder2 = (ClassInfo.Builder) cBuilder;
568                         if ((access & Opcodes.ACC_VOLATILE) != 0 || desc.contains("util/concurrent")) {
569                             cBuilder2.setUsesConcurrency();
570                         }
571                         final FieldInfo.Builder fBuilder = new FieldInfo.Builder(slashedClassName, name, desc, access);
572                         fBuilder.setSourceSignature(signature);
573                         return new AbstractFieldAnnotationVisitor() {
574     
575                             @Override
576                             public org.objectweb.asm.AnnotationVisitor visitAnnotation(final String desc, boolean visible) {
577                                 AnnotationValue value = new AnnotationValue(desc);
578                                 fBuilder.addAnnotation(desc, value);
579                                 return value.getAnnotationVisitor();
580                             }
581     
582                             @Override
583                             public void visitEnd() {
584                                 cBuilder2.addFieldDescriptor(fBuilder.build());
585     
586                             }
587     
588                         };
589     
590                     }
591                     return null;
592                 }
593     
594                 @Override
595                 public void visitInnerClass(String name, String outerName, String innerName, int access) {
596                     if (name.equals(slashedClassName) && outerName != null) {
597                         if (cBuilder instanceof ClassInfo.Builder) {
598                             ClassDescriptor outerClassDescriptor = DescriptorFactory.createClassDescriptor(outerName);
599                             ((ClassInfo.Builder) cBuilder).setImmediateEnclosingClass(outerClassDescriptor);
600                             ((ClassInfo.Builder) cBuilder).setAccessFlags(access);
601                         }
602     
603                     }
604     
605                 }
606     
607                 @Override
608                 public MethodVisitor visitMethod(final int access, final String methodName, final String methodDesc,
609                         String signature, String[] exceptions) {
610                     if (cBuilder instanceof ClassInfo.Builder) {
611                         final MethodInfo.Builder mBuilder = new MethodInfo.Builder(slashedClassName, methodName, methodDesc, access);
612                         mBuilder.setSourceSignature(signature);
613                         mBuilder.setThrownExceptions(exceptions);
614                         if ((access & Opcodes.ACC_SYNCHRONIZED) != 0) {
615                             mBuilder.setUsesConcurrency();
616                         }
617     
618                         return new ClassParserMethodVisitor(calledClassSet, mBuilder, methodName, access, methodDesc, cBuilder);
619     
620                     }
621                     return null;
622                 }
623     
624                 @Override
625                 public void visitOuterClass(String owner, String name, String desc) {
626     
627                 }
628     
629                 @Override
630                 public void visitSource(String arg0, String arg1) {
631                     if (cBuilder instanceof ClassInfo.Builder) {
632                         ((ClassInfo.Builder) cBuilder).setSource(arg0);
633                     }
634     
635                 }
636             }, ClassReader.SKIP_FRAMES);
637             HashSet<ClassDescriptor> referencedClassSet = new HashSet<ClassDescriptor>();
638     
639             // collect class references
640     
641             int constantPoolCount = classReader.readUnsignedShort(8);
642             int offset = 10;
643             char[] buf = new char[1024];
644             // System.out.println("constant pool count: " + constantPoolCount);
645             for (int count = 1; count < constantPoolCount; count++) {
646                 int tag = classReader.readByte(offset);
647     
648                 int size;
649                 switch (tag) {
650                 case Constants.CONSTANT_Methodref:
651                 case Constants.CONSTANT_InterfaceMethodref:
652                 case Constants.CONSTANT_Fieldref:
653                 case Constants.CONSTANT_Integer:
654                 case Constants.CONSTANT_Float:
655                 case Constants.CONSTANT_NameAndType:
656                 case Constants.CONSTANT_InvokeDynamic:
657                     size = 5;
658                     break;
659                 case Constants.CONSTANT_Long:
660                 case Constants.CONSTANT_Double:
661                     size = 9;
662                     count++;
663                     break;
664                 case Constants.CONSTANT_Utf8:
665                     size = 3 + classReader.readUnsignedShort(offset + 1);
666                     break;
667                 case Constants.CONSTANT_Class:
668                     @SlashedClassName
669                     String className = classReader.readUTF8(offset + 1, buf);
670                     if (className.indexOf('[') >= 0) {
671                         ClassParser.extractReferencedClassesFromSignature(referencedClassSet, className);
672                     } else if (ClassName.isValidClassName(className)) {
673                         ClassDescriptor classDescriptor = DescriptorFactory.instance().getClassDescriptor(className);
674                         referencedClassSet.add(classDescriptor);
675                     }
676                     size = 3;
677                     break;
678                 case Constants.CONSTANT_String:
679                 case Constants.CONSTANT_MethodType:
680                     size = 3;
681                     break;
682                 case Constants.CONSTANT_MethodHandle:
683                     size = 4;
684                     break;
685                 default:
686                     throw new IllegalStateException("Unexpected tag of " + tag + " at offset " + offset + " while parsing "
687                             + slashedClassName + " from " + codeBaseEntry);
688                 }
689                 // System.out.println(count + "@" + offset + " : [" + tag
690                 // +"] size="+size);
691                 offset += size;
692             }
693             cBuilder.setCalledClassDescriptors(calledClassSet);
694             cBuilder.setReferencedClassDescriptors(referencedClassSet);
695         }

699             parse((ClassNameAndSuperclassInfo.Builder) builder);
700     
701         }
