68         public static UsagesRequiringNonNullValues getAnalysis(ClassContext classContext, Method method) {
69             XMethod thisMethod = classContext.getXClass().findMethod(method.getName(), method.getSignature(), method.isStatic());
70             if (DEBUG) {
71                 System.out.println(thisMethod);
72             }
73             UsagesRequiringNonNullValues derefs = new UsagesRequiringNonNullValues();
74             try {
75     
76                 CFG cfg = classContext.getCFG(method);
77     
78                 ValueNumberDataflow vna = classContext.getValueNumberDataflow(method);
79                 TypeDataflow typeDataflow = classContext.getTypeDataflow(method);
80                 INullnessAnnotationDatabase db = AnalysisContext.currentAnalysisContext().getNullnessAnnotationDatabase();
81     
82                 ParameterNullnessPropertyDatabase unconditionalDerefParamDatabase = AnalysisContext.currentAnalysisContext()
83                         .getUnconditionalDerefParamDatabase();
84                 Iterator<BasicBlock> bbIter = cfg.blockIterator();
85                 ConstantPoolGen cpg = classContext.getConstantPoolGen();
86                 ValueNumber valueNumberForThis = null;
87                 if (!method.isStatic()) {
88                     ValueNumberFrame frameAtEntry = vna.getStartFact(cfg.getEntry());
89                     valueNumberForThis = frameAtEntry.getValue(0);
90                 }
91     
92                 NullnessAnnotation methodAnnotation = getMethodNullnessAnnotation(classContext, method);
93     
94                 while (bbIter.hasNext()) {
95                     BasicBlock basicBlock = bbIter.next();
96     
97                     if (basicBlock.isNullCheck()) {
98                         InstructionHandle exceptionThrowerHandle = basicBlock.getExceptionThrower();
99                         Instruction exceptionThrower = exceptionThrowerHandle.getInstruction();
100                         ValueNumberFrame vnaFrame = vna.getStartFact(basicBlock);
101                         if (!vnaFrame.isValid()) {
102                             continue;
103                         }
104                         ValueNumber valueNumber = vnaFrame.getInstance(exceptionThrower, cpg);
105     
106                         Location location = new Location(exceptionThrowerHandle, basicBlock);
107                         if (valueNumberForThis != valueNumber) {
108                             derefs.add(location, valueNumber, PointerUsageRequiringNonNullValue.getPointerDereference());
109                         }
110     
111                     }
112                 }
113     
114                 for (Iterator<Location> i = cfg.locationIterator(); i.hasNext();) {
115                     Location location = i.next();
116                     InstructionHandle handle = location.getHandle();
117                     Instruction ins = handle.getInstruction();
118                     ValueNumberFrame valueNumberFrame = vna.getFactAtLocation(location);
119                     TypeFrame typeFrame = typeDataflow.getFactAtLocation(location);
120                     if (ins instanceof InvokeInstruction) {
121                         InvokeInstruction inv = (InvokeInstruction) ins;
122                         XMethod m = XFactory.createXMethod(inv, cpg);
123                         SignatureParser sigParser = new SignatureParser(m.getSignature());
124                         int numParams = sigParser.getNumParameters();
125     
126                         // Check nonnull annotations
127     
128                         for (int j = 0; j < numParams; j++) {
129                             if (db.parameterMustBeNonNull(m, j)) {
130                                 int slot = sigParser.getSlotsFromTopOfStackForParameter(j);
131                                 ValueNumber valueNumber = valueNumberFrame.getStackValue(slot);
132                                 if (valueNumberForThis != valueNumber) {
133                                     derefs.add(location, valueNumber,
134                                             PointerUsageRequiringNonNullValue.getPassedAsNonNullParameter(m, j));
135                                 }
136                             }
137                         }
138     
139                         // Check actual targets
140                         try {
141                             Set<JavaClassAndMethod> targetMethodSet = Hierarchy.resolveMethodCallTargets(inv, typeFrame, cpg);
142                             BitSet unconditionallyDereferencedNullArgSet = null;
143                             for (JavaClassAndMethod targetMethod : targetMethodSet) {
144     
145                                 ParameterProperty property = unconditionalDerefParamDatabase.getProperty(targetMethod
146                                         .toMethodDescriptor());
147                                 if (property == null) {
148                                     unconditionallyDereferencedNullArgSet = null;
149                                     break;
150                                 }
151                                 BitSet foo = property.getAsBitSet();
152                                 if (unconditionallyDereferencedNullArgSet == null) {
153                                     unconditionallyDereferencedNullArgSet = foo;
154                                 } else {
155                                     unconditionallyDereferencedNullArgSet.and(foo);
156                                 }
157                                 if (unconditionallyDereferencedNullArgSet.isEmpty()) {
158                                     break;
159                                 }
160                             }
161     
162                             if (unconditionallyDereferencedNullArgSet != null && !unconditionallyDereferencedNullArgSet.isEmpty()
163                                     && valueNumberFrame.isValid()) {
164                                 for (int j = unconditionallyDereferencedNullArgSet.nextSetBit(0); j >= 0; j = unconditionallyDereferencedNullArgSet
165                                         .nextSetBit(j + 1)) {
166                                     int slot = sigParser.getSlotsFromTopOfStackForParameter(j);
167                                     ValueNumber valueNumber = valueNumberFrame.getStackValue(slot);
168                                     if (valueNumberForThis != valueNumber) {
169                                         derefs.add(location, valueNumber,
170                                                 PointerUsageRequiringNonNullValue.getPassedAsNonNullParameter(m, j));
171                                     }
172                                 }
173                             }
174     
175                         } catch (ClassNotFoundException e) {
176                             AnalysisContext.reportMissingClass(e);
177                         }
178     
179                     } else if (ins instanceof IFNONNULL && UnconditionalValueDerefAnalysis.isNullCheck(handle, cpg)) {
180                         ValueNumber valueNumber = valueNumberFrame.getTopValue();
181                         derefs.add(location, valueNumber, PointerUsageRequiringNonNullValue.getPointerNullChecked());
182                     } else if (ins instanceof ARETURN && methodAnnotation == NullnessAnnotation.NONNULL) {
183                         ValueNumber valueNumber = valueNumberFrame.getTopValue();
184                         if (valueNumberForThis != valueNumber) {
185                             derefs.add(location, valueNumber,
186                                     PointerUsageRequiringNonNullValue.getReturnFromNonNullMethod(thisMethod));
187                         }
188     
189                     } else if (ins instanceof PUTFIELD || ins instanceof PUTSTATIC) {
190                         FieldInstruction inf = (FieldInstruction) ins;
191                         XField field = XFactory.createXField(inf, cpg);
192                         NullnessAnnotation annotation = AnalysisContext.currentAnalysisContext().getNullnessAnnotationDatabase()
193                                 .getResolvedAnnotation(field, false);
194                         if (annotation == NullnessAnnotation.NONNULL) {
195                             ValueNumber valueNumber = valueNumberFrame.getTopValue();
196                             if (valueNumberForThis != valueNumber) {
197                                 derefs.add(location, valueNumber, PointerUsageRequiringNonNullValue.getStoredIntoNonNullField(field));
198                             }
199                         }
200     
201                     }
202                 }
203     
204             } catch (CFGBuilderException e) {
205                 AnalysisContext.logError("Error generating derefs for " + thisMethod, e);
206             } catch (DataflowAnalysisException e) {
207                 AnalysisContext.logError("Error generating derefs for " + thisMethod, e);
208             }
209             return derefs;
210         }

212         public static NullnessAnnotation getMethodNullnessAnnotation(ClassContext classContext, Method method) {
213     
214             if (method.getSignature().indexOf(")L") >= 0 || method.getSignature().indexOf(")[") >= 0) {
215     
216                 XMethod m = XFactory.createXMethod(classContext.getJavaClass(), method);
217                 return AnalysisContext.currentAnalysisContext().getNullnessAnnotationDatabase().getResolvedAnnotation(m, false);
218             }
219             return NullnessAnnotation.UNKNOWN_NULLNESS;
220         }
