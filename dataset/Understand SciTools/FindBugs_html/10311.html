284         MethodInfo(@SlashedClassName String className, String methodName, String methodSignature, String methodSourceSignature,
285                 int accessFlags, boolean isUnconditionalThrower, boolean isUnsupported, boolean usesConcurrency,
286                 boolean hasBackBranch, boolean isStub, boolean isIdentity,
287                 boolean usesInvokeDynamic, int methodCallCount, @CheckForNull String[] exceptions,
288                 @CheckForNull MethodDescriptor accessMethodForMethod,
289                 @CheckForNull FieldDescriptor accessMethodForField,
290                 Map<ClassDescriptor, AnnotationValue> methodAnnotations, Map<Integer, Map<ClassDescriptor, AnnotationValue>> methodParameterAnnotations, long variableIsSynthetic) {
291             super(className, methodName, methodSignature, (accessFlags & Constants.ACC_STATIC) != 0);
292             this.accessFlags = accessFlags;
293             this.exceptions = exceptions;
294             if (exceptions != null) {
295                 for (int i = 0; i < exceptions.length; i++) {
296                     exceptions[i] = DescriptorFactory.canonicalizeString(exceptions[i]);
297                 }
298             }
299             this.methodSourceSignature = DescriptorFactory.canonicalizeString(methodSourceSignature);
300             this.methodAnnotations = Util.immutableMap(methodAnnotations);
301             this.methodParameterAnnotations = Util.immutableMap(methodParameterAnnotations);
302             if (isUnconditionalThrower) {
303                 getUnconditionalthrowers().put(this, null);
304             }
305             if (isUnsupported) {
306                 getUnconditionalthrowers().put(this, null);
307             }
308             if (accessMethodForMethod != null) {
309                 getAccessmethodformethod().put(this, accessMethodForMethod);
310             }
311             if (accessMethodForField!= null) {
312                 getAccessmethodforfield().put(this, accessMethodForField);
313             }
314             if (isIdentity) {
315                 getIdentitymethods().put(this, null);
316             }
317             if (usesInvokeDynamic) {
318                 getInvokeDynamicMethods().put(this, null);
319             }
320     
321             this.usesConcurrency = usesConcurrency;
322             this.hasBackBranch = hasBackBranch;
323             this.isStub = isStub;
324             this.methodCallCount = methodCallCount;
325             this.variableIsSynthetic = variableIsSynthetic;
326         }

432         @Override
433         public int compareTo(ComparableMethod rhs) {
434             if (rhs instanceof MethodDescriptor) {
435                 return FieldOrMethodDescriptor.compareTo(this, (MethodDescriptor) rhs);
436             }
437     
438             if (rhs instanceof XMethod) {
439                 return XFactory.compare((XMethod) this, (XMethod) rhs);
440             }
441     
442             throw new ClassCastException("Can't compare a " + this.getClass().getName() + " to a " + rhs.getClass().getName());
443         }

560     
561             Collection<AnnotationValue> result =  methodAnnotations.values();

571         /**
572          * Destructively add an annotation. We do this for "built-in" annotations
573          * that might not be directly evident in the code. It's not a great idea in
574          * general, but we can get away with it as long as it's done early enough
575          * (i.e., before anyone asks what annotations this method has.)
576          *
577          * @param annotationValue
578          *            an AnnotationValue representing a method annotation
579          */
580         @Override
581         public void addAnnotation(AnnotationValue annotationValue) {
582             HashMap<ClassDescriptor, AnnotationValue> updatedAnnotations = new HashMap<ClassDescriptor, AnnotationValue>(
583                     methodAnnotations);
584             updatedAnnotations.put(annotationValue.getAnnotationClass(), annotationValue);
585             methodAnnotations = updatedAnnotations;
586             TypeQualifierApplications.updateAnnotations(this);
587         }
588     
589         /**
590          * Destructively add a parameter annotation.
591          *
592          * @param param
593          *            parameter (0 == first parameter)
594          * @param annotationValue
595          *            an AnnotationValue representing a parameter annotation
596          */
597         @Override
598         public void addParameterAnnotation(int param, AnnotationValue annotationValue) {
599             HashMap<Integer, Map<ClassDescriptor, AnnotationValue>> updatedAnnotations = new HashMap<Integer, Map<ClassDescriptor, AnnotationValue>>(
600                     methodParameterAnnotations);
601             Map<ClassDescriptor, AnnotationValue> paramMap = updatedAnnotations.get(param);
602             if (paramMap == null) {
603                 paramMap = new HashMap<ClassDescriptor, AnnotationValue>();
604                 updatedAnnotations.put(param, paramMap);
605             }
606             paramMap.put(annotationValue.getAnnotationClass(), annotationValue);
607     
608             methodParameterAnnotations = updatedAnnotations;
609             TypeQualifierApplications.updateAnnotations(this);
610         }
