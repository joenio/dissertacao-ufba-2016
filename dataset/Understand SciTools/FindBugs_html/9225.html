20     package edu.umd.cs.findbugs.util;
21     
22     import java.util.HashMap;
23     import java.util.Map;
24     
25     /**
26      * @author pugh
27      */
28     public class TripleKeyHashMap<K1, K2, K3, V> {
29         Map<K1, DualKeyHashMap<K2, K3, V>> map = new HashMap<K1, DualKeyHashMap<K2, K3, V>>();
30     
31         public V get(K1 k1, K2 k2, K3 k3) {
32             DualKeyHashMap<K2, K3, V> m = map.get(k1);
33             if (m == null) {
34                 return null;
35             }
36             return m.get(k2, k3);
37         }
38     
39         public V put(K1 k1, K2 k2, K3 k3, V v) {
40             DualKeyHashMap<K2, K3, V> m = map.get(k1);
41             if (m == null) {
42                 m = new DualKeyHashMap<K2, K3, V>();
43                 map.put(k1, m);
44             }
45             return m.put(k2, k3, v);
46         }
47     
48     }
