56         public FindNullDerefsInvolvingNonShortCircuitEvaluation(BugReporter bugReporter) {
57             this.bugReporter = bugReporter;

60         @Override
61         public void visit(Code code) {
62             boolean interesting = true;
63             if (interesting) {
64                 // initialize any variables we want to initialize for the method
65                 super.visit(code); // make callbacks to sawOpcode for all opcodes
66             }
67         }
68     
69         @Override
70         public void sawOpcode(int seen) {
71             if (seen == IAND || seen == IOR) {
72     
73                 int nextOpcode = getCodeByte(getPC() + 1);
74                 if (nextOpcode == IFEQ || nextOpcode == IFNE) {
75                     OpcodeStack.Item left = stack.getStackItem(1);
76                     OpcodeStack.Item right = stack.getStackItem(0);
77                     checkForNullForcingABranch(seen, nextOpcode, left);
78                     checkForNullForcingABranch(seen, nextOpcode, right);
79                 }
80     
81             }
82         }
83     
84         private void checkForNullForcingABranch(int seen, int nextOpcode, OpcodeStack.Item item) {
85             if (nullGuaranteesBranch(seen, item)) {
86                 // null guarantees a branch
87                 boolean nullGuaranteesZero = seen == IAND;
88                 boolean nullGuaranteesBranch = nullGuaranteesZero ^ (nextOpcode == IFNE);
89                 if (DEBUG) {
90                     System.out.println(item.getPC() + " null guarantees " + nullGuaranteesBranch + " branch");
91                 }
92                 try {
93                     CFG cfg = getClassContext().getCFG(getMethod());
94                     Location produced = findLocation(cfg, item.getPC());
95                     Location branch = findLocation(cfg, getPC() + 1);
96                     if (produced == null || branch == null) {
97                         return;
98                     }
99     
100                     IfInstruction branchInstruction = (IfInstruction) branch.getHandle().getInstruction();
101     
102                     IsNullValueDataflow isNullValueDataflow = getClassContext().getIsNullValueDataflow(getMethod());
103                     ValueNumberDataflow valueNumberDataflow = getClassContext().getValueNumberDataflow(getMethod());
104                     UnconditionalValueDerefDataflow unconditionalValueDerefDataflow = getClassContext()
105                             .getUnconditionalValueDerefDataflow(getMethod());
106                     ValueNumberFrame valueNumberFact = valueNumberDataflow.getFactAtLocation(produced);
107                     IsNullValueFrame isNullFact = isNullValueDataflow.getFactAtLocation(produced);
108                     ValueNumber value = valueNumberFact.getTopValue();
109                     if (isNullFact.getTopValue().isDefinitelyNotNull()) {
110                         return;
111                     }
112                     if (DEBUG) {
113                         System.out.println("Produced: " + produced);
114                         System.out.println(valueNumberFact);
115                         System.out.println(isNullFact);
116                         System.out.println("value: " + value);
117                         System.out.println("branch: " + branch);
118                         System.out.println("instruction: " + branchInstruction);
119                         System.out.println("target: " + branchInstruction.getTarget());
120                         System.out.println("next: " + branch.getHandle().getNext());
121                     }
122                     Location guaranteed = findLocation(cfg, nullGuaranteesBranch ? branchInstruction.getTarget() : branch.getHandle()
123                             .getNext());
124                     if (guaranteed == null) {
125                         return;
126                     }
127     
128                     UnconditionalValueDerefSet unconditionalDeref = unconditionalValueDerefDataflow.getFactAtLocation(guaranteed);
129                     if (DEBUG) {
130                         System.out.println("Guaranteed on null: " + guaranteed);
131                         System.out.println(unconditionalDeref);
132                     }
133     
134                     if (unconditionalDeref.isUnconditionallyDereferenced(value)) {
135                         SourceLineAnnotation tested = SourceLineAnnotation.fromVisitedInstruction(getClassContext(), getMethod(),
136                                 produced);
137                         BugAnnotation variableAnnotation = ValueNumberSourceInfo.findAnnotationFromValueNumber(getMethod(), produced,
138                                 value, valueNumberFact, "VALUE_OF");
139                         Set<Location> unconditionalDerefLocationSet = unconditionalDeref.getUnconditionalDerefLocationSet(value);
140     
141                         BugInstance bug;
142                         if (unconditionalDerefLocationSet.size() > 1) {
143                             bug = new BugInstance(this, "NP_GUARANTEED_DEREF", NORMAL_PRIORITY).addClassAndMethod(this);
144                             bug.addOptionalAnnotation(variableAnnotation);
145                             bug.addSourceLine(tested).describe("SOURCE_LINE_KNOWN_NULL");
146                             for (Location dereferenced : unconditionalDerefLocationSet) {
147                                 bug.addSourceLine(getClassContext(), getMethod(), dereferenced).describe("SOURCE_LINE_DEREF");
148                             }
149     
150                         } else {
151                             bug = new BugInstance(this, "NP_NULL_ON_SOME_PATH", NORMAL_PRIORITY).addClassAndMethod(this);
152                             bug.addOptionalAnnotation(variableAnnotation);
153                             for (Location dereferenced : unconditionalDerefLocationSet) {
154                                 bug.addSourceLine(getClassContext(), getMethod(), dereferenced).describe("SOURCE_LINE_DEREF");
155                             }
156     
157                             bug.addSourceLine(tested).describe("SOURCE_LINE_KNOWN_NULL");
158     
159                         }
160     
161                         bugReporter.reportBug(bug);
162                     }
163     
164                 } catch (DataflowAnalysisException e) {
165                     bugReporter.logError("Error getting analysis for " + getFullyQualifiedMethodName(), e);
166                 } catch (CFGBuilderException e) {
167                     bugReporter.logError("Error getting analysis for " + getFullyQualifiedMethodName(), e);
168                 }
169     
170             }
171         }

173         @CheckForNull
174         Location findLocation(CFG cfg, int pc) {
175             for (Iterator<Location> i = cfg.locationIterator(); i.hasNext();) {
176                 Location loc = i.next();
177                 if (loc.getHandle().getPosition() == pc) {
178                     return loc;
179                 }
180             }
181             return null;
182         }
183     
184         @CheckForNull
185         Location findLocation(CFG cfg, InstructionHandle handle) {
186             for (Iterator<Location> i = cfg.locationIterator(); i.hasNext();) {
187                 Location loc = i.next();
188                 if (loc.getHandle() == handle) {
189                     return loc;
190                 }
191             }
192             return null;
193         }
194     
195         private boolean nullGuaranteesBranch(int seen, OpcodeStack.Item item) {
196             return item.getSpecialKind() == OpcodeStack.Item.ZERO_MEANS_NULL && seen == IAND
197                     || item.getSpecialKind() == OpcodeStack.Item.NONZERO_MEANS_NULL && seen == IOR;
198         }
