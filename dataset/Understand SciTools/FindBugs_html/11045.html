20     package edu.umd.cs.findbugs.detect;
21     
22     import edu.umd.cs.findbugs.BugInstance;
23     import edu.umd.cs.findbugs.BugReporter;
24     import edu.umd.cs.findbugs.OpcodeStack;
25     import edu.umd.cs.findbugs.OpcodeStack.Item;
26     import edu.umd.cs.findbugs.ba.AnalysisContext;
27     import edu.umd.cs.findbugs.ba.ClassContext;
28     import edu.umd.cs.findbugs.ba.interproc.ParameterProperty;
29     import edu.umd.cs.findbugs.bcel.OpcodeStackDetector;
30     import edu.umd.cs.findbugs.visitclass.PreorderVisitor;
31     
32     public class IntCast2LongAsInstant extends OpcodeStackDetector {
33     
34         final BugReporter bugReporter;
35     
36         int lastConstantForSIPUSH;
37     
38         TrainLongInstantfParams.LongInstantParameterDatabase database = new TrainLongInstantfParams.LongInstantParameterDatabase();
39     
40         public IntCast2LongAsInstant(BugReporter bugReporter) {
41             this.bugReporter = bugReporter;
42             AnalysisContext.currentAnalysisContext().loadPropertyDatabaseFromResource(database, "longInstant.db",
43                     "long instant database");
44         }
45     
46         @Override
47         public void sawOpcode(int seen) {
48             if (seen == SIPUSH) {
49                 lastConstantForSIPUSH = getIntConstant();
50             }
51             if (seen == INVOKEINTERFACE || seen == INVOKEVIRTUAL || seen == INVOKESPECIAL || seen == INVOKESTATIC) {
52                 String signature = getSigConstantOperand();
53     
54                 int numberArguments = PreorderVisitor.getNumberArguments(signature);
55     
56                 for (int i = 0; i < numberArguments; i++) {
57                     Item item = stack.getStackItem(numberArguments - 1 - i);
58                     if (item.getSpecialKind() == OpcodeStack.Item.RESULT_OF_I2L) {
59                         ParameterProperty property = database.getProperty(getMethodDescriptorOperand());
60                         if (property != null && property.hasProperty(i)) {
61                             int priority = NORMAL_PRIORITY;
62     
63                             if (getPrevOpcode(1) == I2L && getPrevOpcode(2) == IMUL && getPrevOpcode(3) == SIPUSH
64                                     && lastConstantForSIPUSH == 1000) {
65                                 priority = HIGH_PRIORITY;
66     
67                             } else if (getPrevOpcode(1) == I2L && getPrevOpcode(2) == IMUL && getPrevOpcode(4) == SIPUSH
68                                     && lastConstantForSIPUSH == 1000) {
69                                 priority = HIGH_PRIORITY;
70                             }
71                             BugInstance bug = new BugInstance(this, "ICAST_INT_2_LONG_AS_INSTANT", priority).addClassAndMethod(this)
72                                     .addCalledMethod(this).addValueSource(item, this).addSourceLine(this);
73                             bugReporter.reportBug(bug);
74                         }
75     
76                     }
77                 }
78     
79             }
80         }
81     
82         /*
83          * (non-Javadoc)
84          *
85          * @see edu.umd.cs.findbugs.Detector#report()
86          */
87         @Override
88         public void report() {
89     
90         }
91     
92         @Override
93         public void visitClassContext(ClassContext classContext) {
94             classContext.getJavaClass().accept(this);
95         }
96     
97     }
