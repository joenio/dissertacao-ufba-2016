20     package edu.umd.cs.findbugs;
21     
22     import java.util.Set;
23     
24     
25     /**
26      * @author pugh
27      */
28     public class ComponentPlugin<T> {
29     
30         public ComponentPlugin(Plugin plugin, String id, ClassLoader classLoader,
31                 Class<? extends T> componentClass,
32                 PropertyBundle properties, boolean enabledByDefault,
33                 String description, String details) {
34             this.plugin = plugin;
35             this.id = id;
36             int i = id.lastIndexOf('.');
37             this.shortId = id.substring(i + 1);
38     
39             this.classLoader = classLoader;
40             this.componentClass = componentClass;
41             this.properties = properties;
42             this.enabledByDefault = enabledByDefault;
43             this.description = description;
44             this.details = details;
45     
46         }
47         protected final Plugin plugin;
48     
49         public String getId() {
50             return id;
51         }
52     
53         public ClassLoader getClassLoader() {
54             return classLoader;
55         }
56     
57         public boolean isEnabledByDefault() {
58             return enabledByDefault;
59         }
60     
61         public PropertyBundle getProperties() {
62             return properties;
63         }
64     
65         public String getDescription() {
66             return description;
67         }
68     
69         public String getDetails() {
70             return details;
71         }
72     
73         @Override
74         public String toString() {
75             return getDescription();
76         }
77     
78         public Plugin getPlugin() {
79             return plugin;
80         }
81     
82         public boolean isNamed(Set<String> names) {
83             return names.contains(id) || names.contains(shortId);
84         }
85     
86         protected final String id;
87         protected final String shortId;
88         protected final ClassLoader classLoader;
89         protected final PropertyBundle properties;
90         protected final String description;
91         protected final String details;
92         protected final boolean enabledByDefault;
93     
94         public Class<? extends T> getComponentClass() {
95             if (!isAvailable()) {
96                 if (FindBugs.isNoAnalysis()) {
97                     throw new IllegalStateException("No analysis set; no component class loaded for " + getPlugin());
98                 }
99                 throw new IllegalStateException("No component class for " + getPlugin());
100             }
101             return componentClass;
102         }
103     
104         public boolean isAvailable() {
105             return componentClass != null;
106         }
107     
108         final Class<? extends T> componentClass;
109     }
