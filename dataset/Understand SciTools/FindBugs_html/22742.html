19     package edu.umd.cs.findbugs.ba;
20     
21     import java.util.Collection;
22     
23     import javax.annotation.CheckForNull;
24     
25     import edu.umd.cs.findbugs.classfile.ClassDescriptor;
26     import edu.umd.cs.findbugs.classfile.FieldDescriptor;
27     import edu.umd.cs.findbugs.classfile.MethodDescriptor;
28     import edu.umd.cs.findbugs.classfile.analysis.AnnotatedObject;
29     import edu.umd.cs.findbugs.classfile.analysis.AnnotationValue;
30     
31     /**
32      * An XMethod represents symbolic information about a particular method.
33      *
34      * <p>
35      * If the resolved() method returns true, then any information queried from this
36      * object can be assumed to be accurate. If the resolved() method returns false,
37      * then FindBugs can't find the method and any information other than
38      * name/signature/etc. cannot be trusted.
39      * </p>
40      *
41      * @author David Hovemeyer
42      * @author Bill Pugh
43      */
44     public interface XMethod extends ClassMember, AnnotatedObject, ComparableMethod{
45         public boolean isNative();
46     
47         public boolean isAbstract();
48     
49         public boolean isSynchronized();
50     
51         public int getNumParams();
52     
53         /**
54          * @return the MethodDescriptor identifying this object
55          */
56         public MethodDescriptor getMethodDescriptor();
57     
58         /**
59          *
60          * @return the exceptions this method is declared to throw
61          */
62         public String[] getThrownExceptions();
63     
64         /**
65          * @return does this method unconditionally throw an exception?
66          */
67         public boolean isUnconditionalThrower();
68     
69         /**
70          * @return does the method directly make use of
71          *         concurrency/threads/synchronization?
72          */
73         public boolean usesConcurrency();
74     
75         /**
76          * @return is the method a synthetic stub method?
77          */
78         public boolean isStub();
79     
80         /** Is this an identity method */
81         public boolean isIdentity();
82     
83     
84         /**
85          * If nonnull, then this method is a synthetic method that overrides a
86          * method in a superclass. This method simply forwards the call to the
87          * method it bridges to, which is a method with an identical name but possibly co-variant arguments and
88          * return values.
89          */
90         public @CheckForNull
91         XMethod bridgeTo();
92     
93         public @CheckForNull
94         XMethod bridgeFrom();
95     
96         public @CheckForNull
97         MethodDescriptor getAccessMethodForMethod();
98     
99     
100         public @CheckForNull
101         FieldDescriptor getAccessMethodForField();
102     
103     
104         public
105         XMethod resolveAccessMethodForMethod();
106     
107     
108         /**
109          * @return does this method unconditionally throw an
110          *         UnsupportedOperationException?
111          */
112         public boolean isUnsupported();
113     
114         /**
115          * @return is this a var args method?
116          */
117         public boolean isVarArgs();
118     
119         /**
120          * @return true if method's return type is a reference type, false otherwise
121          */
122         public boolean isReturnTypeReferenceType();
123     
124         /**
125          * Is this a bridge method?
126          */
127         public boolean isBridge();
128     
129         /**
130          * Get ClassDescriptors (annotation classes) of annotations applied directly
131          * to this method's parameters.
132          *
133          * @param param
134          *            parameter number (0 for first parameter)
135          * @return ClassDescriptors of annotations applied directly to this method's
136          *         parameters
137          */
138         public Collection<ClassDescriptor> getParameterAnnotationDescriptors(int param);
139     
140         /**
141          * Get the AnnotationValue of annotation applied directly to given
142          * parameter.
143          *
144          * @param param
145          *            parameter number (0 for first parameter)
146          * @param desc
147          *            ClassDescriptor of the annotation class
148          * @return AnnotationValue annotating the parameter, or null if parameter is
149          *         not annotated with this kind of annotation
150          */
151         public @CheckForNull
152         AnnotationValue getParameterAnnotation(int param, ClassDescriptor desc);
153     
154         /**
155          * Get collection of all AnnotationValues applied directly to given
156          * parameter.
157          *
158          * @param param
159          *            parameter number (0 for first parameter)
160          * @return Collection of all AnnotationValues applied directly to given
161          *         parameter
162          */
163         public Collection<AnnotationValue> getParameterAnnotations(int param);
164     
165         public boolean hasParameterAnnotations();
166         /**
167          * Get ClassDescriptors (annotation classes) of annotations applied directly
168          * to this method.
169          *
170          * @return ClassDescriptors of annotations applied directly to this method
171          */
172         @Override
173         public Collection<ClassDescriptor> getAnnotationDescriptors();
174     
175         /**
176          * Get the AnnotationValue of annotation applied directly to the method.
177          *
178          * @param desc
179          *            ClassDescriptor of the annotation class
180          * @return AnnotationValue annotating the method, or null if method is not
181          *         annotated with this kind of annotation
182          */
183         @Override
184         public AnnotationValue getAnnotation(ClassDescriptor desc);
185     
186         /**
187          * Get collection of all AnnotationValues applied directly to the method.
188          *
189          * @return Collection of all AnnotationValues applied directly to the method
190          */
191         @Override
192         public Collection<AnnotationValue> getAnnotations();
193     
194         /**
195          * Destructively add a parameter annotation.
196          *
197          * @param param
198          *            parameter (0 == first parameter)
199          * @param annotationValue
200          *            an AnnotationValue representing a parameter annotation
201          */
202         public void addParameterAnnotation(int param, AnnotationValue annotationValue);
203     
204         /** Is the variable synthetic? */
205         public boolean isVariableSynthetic(int param);
206         /**
207          * Destructively add an annotation. We do this for "built-in" annotations
208          * that might not be directly evident in the code. It's not a great idea in
209          * general, but we can get away with it as long as it's done early enough
210          * (i.e., before anyone asks what annotations this method has.)
211          *
212          * @param annotationValue
213          *            an AnnotationValue representing a method annotation
214          */
215         public void addAnnotation(AnnotationValue annotationValue);
216     
217         boolean usesInvokeDynamic();
218     
219     
220     
221     }
