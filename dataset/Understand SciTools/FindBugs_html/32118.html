32     /**
33      * A dataflow fact used in ObligationAnalysis. It is a set of State objects,
34      * plus the additional capability to represent top and bottom elements.
35      *
36      * <p>
37      * Invariant: no StateSet may contain more than one State with the same
38      * ObligationSet.
39      * </p>
40      *
41      * <p>
42      * See Weimer and Necula, <a href="http://doi.acm.org/10.1145/1028976.1029011"
43      * >Finding and preventing run-time error handling mistakes</a>, OOPSLA 2004.
44      * </p>
45      *
46      * @author David Hovemeyer
47      */
48     public class StateSet {
49         private boolean isTop;
50     
51         private boolean isBottom;
52     
53         private boolean onExceptionPath;
54     
55         private Map<ObligationSet, State> stateMap;
56     
57         public boolean isEmpty() {
58             return stateMap.isEmpty();
59         }
60     
61         private final ObligationFactory factory;
62     
63         public StateSet(ObligationFactory factory) {
64             this.isTop = this.isBottom = false;
65             this.stateMap = new HashMap<ObligationSet, State>();
66             this.factory = factory;
67         }
68     
69         public void setTop() {
70             this.isTop = true;
71             this.isBottom = false;
72             this.onExceptionPath = false;
73             this.stateMap.clear();
74         }
75     
76         public boolean isTop() {
77             return isTop;
78         }
79     
80         public void setBottom() {
81             this.isBottom = true;
82             this.isTop = false;
83         }
84     
85         public boolean isBottom() {
86             return this.isBottom;
87         }
88     
89         public boolean isValid() {
90             return !this.isTop && !this.isBottom;
91         }
92     
93         public boolean isOnExceptionPath() {
94             return onExceptionPath;
95         }
96     
97         public void setOnExceptionPath(boolean onExceptionPath) {
98             this.onExceptionPath = onExceptionPath;
99         }
100     
101         public void clear() {
102             this.isTop = this.isBottom = this.onExceptionPath = false;
103             stateMap.clear();
104         }
105     
106         /**
107          * Return an Iterator over the States in the StateSet.
108          *
109          * @return an Iterator over the States in the StateSet
110          */
111         public Iterator<State> stateIterator() {
112             return stateMap.values().iterator();
113         }
114     
115         /**
116          * Get Set of all ObligationsSets in this StateSet.
117          *
118          * @return Set of all ObligationsSets in this StateSet
119          */
120         public Set<ObligationSet> getAllObligationSets() {
121             return Collections.unmodifiableSet(stateMap.keySet());
122         }
123     
124         /**
125          * Get the State which has the given ObligationSet. Returns null if there is
126          * no such state.
127          *
128          * @param obligationSet
129          *            we want to get the State with this ObligationSet
130          * @return the State with the given ObligationSet, or null if there is no
131          *         such State
132          */
133         public State getStateWithObligationSet(ObligationSet obligationSet) {
134             return stateMap.get(obligationSet);
135         }

152         /**
153          * Make this StateSet an exact copy of the given StateSet.
154          *
155          * @param other
156          *            a StateSet; this StateSet will be made identical to it
157          */
158         public void copyFrom(StateSet other) {
159             this.isTop = other.isTop;
160             this.isBottom = other.isBottom;
161             this.onExceptionPath = other.onExceptionPath;
162             this.stateMap.clear();
163             for (State state : other.stateMap.values()) {
164                 State dup = state.duplicate();
165                 this.stateMap.put(dup.getObligationSet(), dup);
166             }
167         }
168     
169         /**
170          * Return an exact deep copy of this StateSet.
171          *
172          * @return an exact deep copy of this StateSet
173          */
174         public StateSet duplicate() {
175             StateSet dup = new StateSet(factory);
176             dup.copyFrom(this);
177             return dup;
178         }
179     
180         /**
181          * Add an obligation to every State in the StateSet.
182          *
183          * @param obligation
184          *            the obligation to add
185          * @param basicBlockId
186          *            the id of the basic block (path component) adding the
187          *            obligation
188          */
189         public void addObligation(final Obligation obligation, int basicBlockId) throws ObligationAcquiredOrReleasedInLoopException {
190             Map<ObligationSet, State> updatedStateMap = new HashMap<ObligationSet, State>();
191             if (stateMap.isEmpty()) {
192                 State s = new State(factory);
193                 s.getObligationSet().add(obligation);
194                 updatedStateMap.put(s.getObligationSet(), s);
195             } else {
196                 for (State state : stateMap.values()) {
197                     checkCircularity(state, obligation, basicBlockId);
198                     state.getObligationSet().add(obligation);
199                     updatedStateMap.put(state.getObligationSet(), state);
200     
201                 }
202             }
203             replaceMap(updatedStateMap);
204         }
205     
206         /**
207          * Remove an Obligation from every State in the StateSet.
208          *
209          * @param obligation
210          *            the obligation to remove
211          * @param basicBlockId
212          *            the id of the basic block (path component) removing the
213          *            obligation
214          * @throws ObligationAcquiredOrReleasedInLoopException
215          */
216         public void deleteObligation(final Obligation obligation, int basicBlockId)
217                 throws ObligationAcquiredOrReleasedInLoopException {
218             Map<ObligationSet, State> updatedStateMap = new HashMap<ObligationSet, State>();
219             for (Iterator<State> i = stateIterator(); i.hasNext();) {
220                 State state = i.next();
221                 checkCircularity(state, obligation, basicBlockId);
222                 ObligationSet obligationSet = state.getObligationSet();
223                 obligationSet.remove(obligation);
224                 if (!obligationSet.isEmpty()) {
225                     updatedStateMap.put(obligationSet, state);
226                 }
227             }
228             replaceMap(updatedStateMap);
229         }
230     
231         /**
232          * Bail out of the analysis is an obligation is acquired or released in a
233          * loop.
234          *
235          * @param state
236          *            a State to which an obligation is being added or removed
237          * @param obligation
238          *            the Obligation being added or removed
239          * @param basicBlockId
240          *            the id of the BasicBlock adding or removing the obligation
241          */
242         private void checkCircularity(State state, Obligation obligation, int basicBlockId)
243                 throws ObligationAcquiredOrReleasedInLoopException {
244             if (state.getPath().hasComponent(basicBlockId)) {
245                 throw new ObligationAcquiredOrReleasedInLoopException(obligation);
246             }
247         }
248     
249         /**
250          * Replace the map of ObligationSets to States with the given one.
251          *
252          * @param stateMap
253          *            enw map of ObligationSets to States
254          */
255         public void replaceMap(Map<ObligationSet, State> stateMap) {
256             this.stateMap = stateMap;
257         }
258     
259         /**
260          * Get all States that have Paths which are prefixes of the given Path.
261          *
262          * @param path
263          *            a Path
264          * @return Collection of States that have Paths which are prefixes of the
265          *         given Path
266          */
267         public List<State> getPrefixStates(Path path) {
268             List<State> result = new LinkedList<State>();
269             for (State state : stateMap.values()) {
270                 if (state.getPath().isPrefixOf(path)) {
271                     result.add(state);
272                 }
273             }
274             return result;
275         }

277         @Override
278         public boolean equals(Object o) {
279             if (o == null || o.getClass() != this.getClass()) {
280                 return false;
281             }
282             StateSet other = (StateSet) o;
283             return this.isTop == other.isTop && this.isBottom == other.isBottom
284                     && this.onExceptionPath == other.onExceptionPath && this.stateMap.equals(other.stateMap);
285         }
286     
287         @Override
288         public int hashCode() {
289             throw new UnsupportedOperationException();
290         }
291     
292         @Override
293         public String toString() {
294             if (isTop) {
295                 return "TOP";
296             } else if (isBottom) {
297                 return "BOTTOM";
298             } else {
299                 StringBuilder buf = new StringBuilder();
300                 buf.append(stateMap);
301     
302                 if (onExceptionPath) {
303                     buf.append(" On exception path");
304                 }
305                 return buf.toString();
306             }
307         }
308     
309         /**
310          * Return a newly allocated Map of ObligationSet to State that may be passed
311          * to applyToAllStatesAndUpdateMap().
312          */
313         public Map<ObligationSet, State> createEmptyMap() {
314             return new HashMap<ObligationSet, State>();
315         }
316     }
