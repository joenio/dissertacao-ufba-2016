38     
39     import edu.umd.cs.findbugs.L10N;

41     /**
42      * @author pugh
43      */
44     public class SplitLayout implements FindBugsLayoutManager {
45     
46         final MainFrame frame;
47     
48         JLabel sourceTitle;
49     
50         JSplitPane topLeftSPane;
51     
52         JSplitPane topSPane;
53     
54         JSplitPane summarySPane;
55     
56         JSplitPane mainSPane;
57     
58         JButton viewSource = new JButton("View in browser");
59     
60         /**
61          * @param frame
62          */
63         public SplitLayout(MainFrame frame) {
64             this.frame = frame;
65         }
66     
67         /*
68          * (non-Javadoc)
69          *
70          * @see edu.umd.cs.findbugs.gui2.FindBugsLayoutManager#createWindowMenu()
71          */
72         @Override
73         public JMenu createWindowMenu() {
74             return null;
75         }
76     
77         @Override
78         public void resetCommentsInputPane() {
79             if (topLeftSPane != null) {
80                 int position = topLeftSPane.getDividerLocation();
81                 topLeftSPane.setRightComponent(frame.createCommentsInputPanel());
82                 topLeftSPane.setDividerLocation(position);
83             }
84         }
85     
86         /*
87          * (non-Javadoc)
88          *
89          * @see edu.umd.cs.findbugs.gui2.FindBugsLayoutManager#initialize()
90          */
91         @Override
92         public void initialize() {
93     
94             Font buttonFont = viewSource.getFont();
95             viewSource.setFont(buttonFont.deriveFont(buttonFont.getSize() / 2));
96             viewSource.setPreferredSize(new Dimension(150, 15));
97             viewSource.setEnabled(false);
98     
99             topLeftSPane = new JSplitPane(JSplitPane.VERTICAL_SPLIT, frame.mainFrameTree.bugListPanel(),
100                     frame.createCommentsInputPanel());
101             topLeftSPane.setOneTouchExpandable(true);
102             topLeftSPane.setContinuousLayout(true);
103             topLeftSPane.setDividerLocation(GUISaveState.getInstance().getSplitTreeComments());
104             removeSplitPaneBorders(topLeftSPane);
105     
106             JPanel sourceTitlePanel = new JPanel();
107             sourceTitlePanel.setLayout(new BorderLayout());
108     
109             JPanel sourcePanel = new JPanel();
110             BorderLayout sourcePanelLayout = new BorderLayout();
111             sourcePanelLayout.setHgap(3);
112             sourcePanelLayout.setVgap(3);
113             sourcePanel.setLayout(sourcePanelLayout);
114             sourceTitle = new JLabel();
115             sourceTitle.setText(L10N.getLocalString("txt.source_listing", ""));
116     
117             sourceTitlePanel.setBorder(new EmptyBorder(3, 3, 3, 3));
118             sourceTitlePanel.add(viewSource, BorderLayout.EAST);
119             sourceTitlePanel.add(sourceTitle, BorderLayout.CENTER);
120     
121             sourcePanel.setBorder(new LineBorder(Color.GRAY));
122             sourcePanel.add(sourceTitlePanel, BorderLayout.NORTH);
123             sourcePanel.add(frame.createSourceCodePanel(), BorderLayout.CENTER);
124             sourcePanel.add(frame.createSourceSearchPanel(), BorderLayout.SOUTH);
125             topSPane = new JSplitPane(JSplitPane.HORIZONTAL_SPLIT, topLeftSPane, sourcePanel);
126             topSPane.setOneTouchExpandable(true);
127             topSPane.setContinuousLayout(true);
128             topSPane.setDividerLocation(GUISaveState.getInstance().getSplitTop());
129             removeSplitPaneBorders(topSPane);
130     
131             summarySPane = frame.summaryTab();
132             mainSPane = new JSplitPane(JSplitPane.VERTICAL_SPLIT, topSPane, summarySPane);
133             mainSPane.setOneTouchExpandable(true);
134             mainSPane.setContinuousLayout(true);
135             mainSPane.setDividerLocation(GUISaveState.getInstance().getSplitMain());
136             removeSplitPaneBorders(mainSPane);
137     
138             frame.setLayout(new BorderLayout());
139             frame.add(mainSPane, BorderLayout.CENTER);
140             frame.add(frame.statusBar(), BorderLayout.SOUTH);
141     
142         }
143     
144         private void removeSplitPaneBorders(JSplitPane pane) {
145             pane.setUI(new BasicSplitPaneUI() {
146                 @Override
147                 public BasicSplitPaneDivider createDefaultDivider() {
148                     return new BasicSplitPaneDivider(this) {
149                         @Override
150                         public void setBorder(Border b) {
151                         }
152                     };
153                 }
154             });
155             pane.setBorder(new EmptyBorder(3, 3, 3, 3));
156         }
157

159          * (non-Javadoc)
160          *
161          * @see edu.umd.cs.findbugs.gui2.FindBugsLayoutManager#makeCommentsVisible()
162          */
163         @Override
164         public void makeCommentsVisible() {
165     
166         }
167

172          */
173         @Override
174         public void makeSourceVisible() {
175     
176         }
177     
178         /*
179          * (non-Javadoc)
180          *
181          * @see edu.umd.cs.findbugs.gui2.FindBugsLayoutManager#saveState()
182          */
183         @Override
184         public void saveState() {
185             GUISaveState.getInstance().setSplitTreeComments(topLeftSPane.getDividerLocation());
186             GUISaveState.getInstance().setSplitTop(topSPane.getDividerLocation());
187             GUISaveState.getInstance().setSplitSummary(summarySPane.getDividerLocation());
188             GUISaveState.getInstance().setSplitMain(mainSPane.getDividerLocation());
189         }
190     
191         /*
192          * (non-Javadoc)
193          *
194          * @see
195          * edu.umd.cs.findbugs.gui2.FindBugsLayoutManager#setSourceTitle(java.lang
196          * .String)
197          */
198         @Override
199         public void setSourceTitle(String title) {
200             sourceTitle.setText(title);
201     
202         }
203     
204         /*
205          * (non-Javadoc)
206          *
207          * @see
208          * edu.umd.cs.findbugs.gui2.FindBugsLayoutManager#getSourceTitleComponent()
209          */
210         @Override
211         public JComponent getSourceViewComponent() {
212             return viewSource;
213         }
