20     package edu.umd.cs.findbugs.detect;
21     
22     import java.io.ObjectInputStream;
23     import java.io.ObjectOutputStream;
24     import java.io.Serializable;
25     
26     import org.apache.bcel.classfile.JavaClass;
27     
28     import edu.umd.cs.findbugs.BugReporter;
29     import edu.umd.cs.findbugs.NonReportingDetector;
30     import edu.umd.cs.findbugs.OpcodeStack;
31     import edu.umd.cs.findbugs.ba.AnalysisContext;
32     import edu.umd.cs.findbugs.ba.XClass;
33     import edu.umd.cs.findbugs.ba.XFactory;
34     import edu.umd.cs.findbugs.ba.XMethod;
35     import edu.umd.cs.findbugs.ba.ch.Subtypes2;
36     import edu.umd.cs.findbugs.bcel.OpcodeStackDetector;
37     import edu.umd.cs.findbugs.classfile.CheckedAnalysisException;
38     import edu.umd.cs.findbugs.classfile.ClassDescriptor;
39     import edu.umd.cs.findbugs.classfile.DescriptorFactory;
40     import edu.umd.cs.findbugs.classfile.Global;
41     
42     public class ExplicitSerialization extends OpcodeStackDetector implements NonReportingDetector {
43     
44         final static XMethod writeObject = XFactory.createXMethod("java.io.ObjectOutputStream", "writeObject", "(Ljava/lang/Object;)V", false);
45     
46         final static XMethod readObject = XFactory.createXMethod("java.io.ObjectInputStream", "readObject", "()Ljava/lang/Object;", false);
47     
48         final static ClassDescriptor ObjectOutputStream = DescriptorFactory.createClassDescriptor(ObjectOutputStream.class);
49         final static ClassDescriptor ObjectInputStream = DescriptorFactory.createClassDescriptor(ObjectInputStream.class);
50     
51         final UnreadFieldsData unreadFields;
52     
53         final BugReporter bugReporter;
54     
55         public ExplicitSerialization(BugReporter bugReporter) {
56             AnalysisContext context = AnalysisContext.currentAnalysisContext();
57             unreadFields = context.getUnreadFieldsData();
58             this.bugReporter = bugReporter;
59         }
60     
61         @Override
62         public boolean shouldVisit(JavaClass obj) {
63             XClass xClass = getXClass();
64             return xClass.getCalledClassDescriptors().contains(ObjectOutputStream)
65                     || xClass.getCalledClassDescriptors().contains(ObjectInputStream);
66         }
67     
68         @Override
69         public void sawOpcode(int seen) {
70             if (seen == INVOKEVIRTUAL && writeObject.equals(getXMethodOperand())) {
71                 OpcodeStack.Item top = stack.getStackItem(0);
72                 String signature = top.getSignature();
73                 while (signature.charAt(0) == '[') {
74                     signature = signature.substring(1);
75                 }
76                 ClassDescriptor c = DescriptorFactory.createClassDescriptorFromFieldSignature(signature);
77                 if (c == null || !Subtypes2.instanceOf(c, Serializable.class)) {
78                     return;
79                 }
80     
81                 try {
82                     XClass xClass = Global.getAnalysisCache().getClassAnalysis(XClass.class, c);
83                     if (xClass.isInterface()) {
84                         return;
85                     }
86                     if (xClass.isSynthetic()) {
87                         return;
88                     }
89                     if (xClass.isAbstract()) {
90                         return;
91                     }
92                     unreadFields.strongEvidenceForIntendedSerialization(c);
93                 } catch (CheckedAnalysisException e) {
94                     bugReporter.logError("Error looking up xClass of " + c, e);
95                 }
96     
97             }
98             if (seen == CHECKCAST) {
99                 OpcodeStack.Item top = stack.getStackItem(0);
100                 if (readObject.equals(top.getReturnValueOf())) {
101                     ClassDescriptor c = getClassDescriptorOperand();
102                     if (!Subtypes2.instanceOf(c, Serializable.class)) {
103                         return;
104                     }
105     
106                     try {
107                         XClass xClass = Global.getAnalysisCache().getClassAnalysis(XClass.class, c);
108                         if (xClass.isInterface()) {
109                             return;
110                         }
111                         if (xClass.isSynthetic()) {
112                             return;
113                         }
114                         if (xClass.isAbstract()) {
115                             return;
116                         }
117                         unreadFields.strongEvidenceForIntendedSerialization(c);
118                     } catch (CheckedAnalysisException e) {
119                         bugReporter.logError("Error looking up xClass of " + c, e);
120                     }
121                 }
122             }
123     
124         }
125     }
