240         private void previousButtonActionPerformed(java.awt.event.ActionEvent evt) {// GEN-FIRST:event_previousButtonActionPerformed
241             if (step > MIN_STEP) {
242                 setStep(step - 1);
243             }

245             foundModel.removeAllElements();
246         }// GEN-LAST:event_previousButtonActionPerformed

248         private void nextButtonActionPerformed(java.awt.event.ActionEvent evt) {// GEN-FIRST:event_nextButtonActionPerformed
249     
250             discover = new Thread() {
251                 @Override
252                 public void run() {
253                     IErrorLogger errorLogger = new IErrorLogger() {
254                         @Override
255                         public void reportMissingClass(ClassNotFoundException ex) {
256                             String className = ClassNotFoundExceptionParser.getMissingClassName(ex);
257                             if (className != null) {
258                                 logError("Missing class: " + className);
259                             } else {
260                                 logError("Missing class: " + ex);
261                             }
262                         }
263     
264                         @Override
265                         public void reportMissingClass(ClassDescriptor classDescriptor) {
266                             logError("Missing class: " + classDescriptor.toDottedClassName());
267                         }
268     
269                         @Override
270                         public void logError(String message) {
271                             System.err.println("Error: " + message);
272                         }
273     
274                         @Override
275                         public void logError(String message, Throwable e) {
276                             logError(message + ": " + e.getMessage());
277                         }
278     
279                         @Override
280                         public void reportSkippedAnalysis(MethodDescriptor method) {
281                             logError("Skipped analysis of method " + method.toString());
282                         }
283                     };
284     
285                     DiscoverSourceDirectories.Progress progress = new DiscoverSourceDirectories.Progress() {
286     
287                         @Override
288                         public void startRecursiveDirectorySearch() {
289                             progressModel.addElement("Scanning directories...");
290                         }
291     
292                         @Override
293                         public void doneRecursiveDirectorySearch() {
294                         }
295     
296                         @Override
297                         public void startScanningArchives(int numArchivesToScan) {
298                             progressModel.addElement("Scanning " + numArchivesToScan + " archives..");
299                         }
300     
301                         @Override
302                         public void doneScanningArchives() {
303                         }
304     
305                         @Override
306                         public void startScanningClasses(int numClassesToScan) {
307                             progressModel.addElement("Scanning " + numClassesToScan + " classes...");
308                         }
309     
310                         @Override
311                         public void finishClass() {
312                         }
313     
314                         @Override
315                         public void doneScanningClasses() {
316                         }
317     
318                         @Override
319                         public void finishArchive() {
320                         }
321     
322                         @Override
323                         public void startArchive(String name) {
324                         }
325                     };
326                     DiscoverSourceDirectories discoverSourceDirectories = new DiscoverSourceDirectories();
327                     discoverSourceDirectories.setProject(project);
328                     discoverSourceDirectories.setRootSourceDirectory(sourceRootBox.getText());
329                     discoverSourceDirectories.setErrorLogger(errorLogger);
330                     discoverSourceDirectories.setProgress(progress);
331     
332                     try {
333                         discoverSourceDirectories.execute();
334                     } catch (CheckedAnalysisException e1) {
335                         // TODO Auto-generated catch block
336                         e1.printStackTrace();
337                     } catch (IOException e1) {
338                         // TODO Auto-generated catch block
339                         e1.printStackTrace();
340                     } catch (InterruptedException e1) {
341                         // TODO Auto-generated catch block
342                         e1.printStackTrace();
343                     }
344     
345                     jList1.setModel(foundModel);
346     
347                     for (String srcDir : discoverSourceDirectories.getDiscoveredSourceDirectoryList()) {
348                         foundModel.addElement(srcDir);
349                     }
350     
351                     if (step < MAX_STEP) {
352                         setStep(step + 1);
353                     }
354                 }
355             };
356             discover.start();
357         }// GEN-LAST:event_nextButtonActionPerformed

359         private void finshButtonActionPerformed(java.awt.event.ActionEvent evt) {// GEN-FIRST:event_finshButtonActionPerformed
360             if (parentGUI != null) {
361                 parentGUI.setSourceDirecs(foundModel);
362             }
363             if (discover != null && discover.isAlive()) {
364                 discover.stop();
365             }
366             dispose();
367         }// GEN-LAST:event_finshButtonActionPerformed

445         private void setStep(int step) {
446             if (step < MIN_STEP || step > MAX_STEP) {
447                 throw new IllegalArgumentException("Invalid step " + step);
448             }
449             this.step = step;
450             previousButton.setEnabled(step != MIN_STEP);
451             nextButton.setEnabled(step != MAX_STEP && !"".equals(sourceRootBox.getText()));
452     
453             CardLayout cards = (CardLayout) contentPanel.getLayout();
454             cards.show(contentPanel, "card" + step);
455     
456         }
