42     /**
43      * This detector is just a test harness to test a dataflow analysis class
44      * specified by the dataflow.classname property.
45      *
46      * @author David Hovemeyer
47      */
48     public class TestDataflowAnalysis<Fact,AnalysisType extends DataflowAnalysis<Fact>> implements Detector2, NonReportingDetector {
49     
50         private final String dataflowClassName;
51     
52         private final String methodName;
53     
54         private Class<? extends Dataflow<Fact,AnalysisType>> dataflowClass;
55     
56         private boolean initialized;
57     
58         public TestDataflowAnalysis(BugReporter bugReporter) {
59             dataflowClassName = SystemProperties.getProperty("dataflow.classname");
60             methodName = SystemProperties.getProperty("dataflow.method");
61         }
62     
63         /*
64          * (non-Javadoc)
65          *
66          * @see edu.umd.cs.findbugs.Detector2#finishPass()
67          */
68         @Override
69         public void finishPass() {
70         }
71     
72         /*
73          * (non-Javadoc)
74          *
75          * @see edu.umd.cs.findbugs.Detector2#getDetectorClassName()
76          */
77         @Override
78         public String getDetectorClassName() {
79             return getClass().getName();
80         }
81     
82         /*
83          * (non-Javadoc)
84          *
85          * @see
86          * edu.umd.cs.findbugs.Detector2#visitClass(edu.umd.cs.findbugs.classfile
87          * .ClassDescriptor)
88          */
89         @Override
90         @SuppressWarnings("unchecked")
91         public void visitClass(ClassDescriptor classDescriptor) throws CheckedAnalysisException {
92             if (dataflowClassName == null) {
93                 return;
94             }
95     
96             if (!initialized) {
97                 initialize();
98             }
99     
100             if (dataflowClass == null) {
101                 return;
102             }
103     
104             IAnalysisCache analysisCache = Global.getAnalysisCache();
105     
106             XClass classInfo = analysisCache.getClassAnalysis(XClass.class, classDescriptor);
107     
108             // Test dataflow analysis on each method]
109             for (XMethod xMethod : classInfo.getXMethods()) {
110                 if (methodName != null && !methodName.equals(xMethod.getName())) {
111                     continue;
112                 }
113                 MethodDescriptor methodDescriptor = xMethod.getMethodDescriptor();
114     
115                 System.out.println("-----------------------------------------------------------------");
116                 System.out.println("Method: " + SignatureConverter.convertMethodSignature(methodDescriptor));
117                 System.out.println("-----------------------------------------------------------------");
118     
119                 // Create and execute the dataflow analysis
120                 Dataflow<Fact,AnalysisType> dataflow = analysisCache.getMethodAnalysis(dataflowClass, methodDescriptor);
121     
122                 System.out.println("Dataflow finished after " + dataflow.getNumIterations());
123     
124                 if (SystemProperties.getBoolean("dataflow.printcfg")) {
125                     DataflowCFGPrinter<Fact,AnalysisType> cfgPrinter
126                     = new DataflowCFGPrinter<Fact,AnalysisType>(dataflow);
127                     cfgPrinter.print(System.out);
128                 }
129     
130             }
131         }
132     
133         private void initialize()  {
134             initialized = true;
135     
136             IAnalysisCache analysisCache = Global.getAnalysisCache();
137     
138             Class<? extends Dataflow<Fact,AnalysisType>> cls = null;
139     
140             // First, try loading the dataflow class from the general findBugs code.
141             try {
142                 Class<?> c = getClass().getClassLoader().loadClass(dataflowClassName);
143                 cls = asDataflowClass(c);
144     
145             } catch (ClassNotFoundException e) {
146                 assert true;
147             }
148     
149             if (cls == null) {
150                 // Find the dataflow class from the plugin in which it was loaded
151     
152                 DetectorFactoryCollection detectorFactoryCollection = analysisCache.getDatabase(DetectorFactoryCollection.class);
153                 for (Iterator<Plugin> i = detectorFactoryCollection.pluginIterator(); i.hasNext();) {
154                     Plugin plugin = i.next();
155     
156                     try {
157                         cls = asDataflowClass(plugin.getClassLoader().loadClass(dataflowClassName));
158                         break;
159                     } catch (ClassNotFoundException e) {
160                         assert true;
161                     }
162     
163                 }
164             }
165     
166             if (cls == null) {
167                 analysisCache.getErrorLogger().logError("TestDataflowAnalysis: could not load class " + dataflowClassName);
168                 return;
169             }
170     
171     
172             dataflowClass = cls;
173         }
174     
175     
176         @SuppressWarnings("unchecked")
177         private Class<? extends Dataflow<Fact, AnalysisType>> asDataflowClass(Class<?> c) {
178             return (Class<? extends Dataflow<Fact, AnalysisType>>) c.asSubclass(Dataflow.class);
179         }
180     
181     }
