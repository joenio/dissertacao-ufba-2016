38     /**
39      * Bug rankers are used to compute a bug rank for each bug instance. Bug ranks
40      * 1-20 are for bugs that are visible to users. Bug rank 1 is more the most
41      * relevant/scary bugs. A bug rank greater than 20 is for issues that should not
42      * be shown to users.
43      *
44      *
45      * The following bug rankers may exist:
46      * <ul>
47      * <li>core bug ranker (loaded from etc/bugrank.txt)
48      * <li>a bug ranker for each plugin (loaded from <plugin>/etc/bugrank.txt)
49      * <li>A global adjustment ranker (loaded from plugins/adjustBugrank.txt)
50      * </ul>
51      *
52      * A bug ranker is comprised of a list of bug patterns, bug kinds and bug
53      * categories. For each, either an absolute or relative bug rank is provided. A
54      * relative rank is one preceeded by a + or -.
55      *
56      * For core bug detectors, the bug ranker search order is:
57      * <ul>
58      * <li>global adjustment bug ranker
59      * <li>core bug ranker
60      * </ul>
61      *
62      * For third party plugins, the bug ranker search order is:
63      * <ul>
64      * <li>global adjustment bug ranker
65      * <li>plugin adjustment bug ranker
66      * <li>core bug ranker
67      * </ul>
68      *
69      * The overall search order is
70      * <ul>
71      * <li>Bug patterns, in search order across bug rankers
72      * <li>Bug kinds, in search order across bug rankers
73      * <li>Bug categories, in search order across bug rankers
74      * </ul>
75      *
76      * Search stops at the first absolute bug rank found, and the result is the sum
77      * of all of relative bug ranks plus the final absolute bug rank. Since all bug
78      * categories are defined by the core bug ranker, we should always find an
79      * absolute bug rank.
80      *
81      * @see BugRankCategory
82      * @see Priorities
83      * @see edu.umd.cs.findbugs.annotations.Confidence
84      *
85      * @author Bill Pugh
86      */
87     public class BugRanker {
88         /** Maximum value for user visible ranks (least relevant) */
89         public static final int VISIBLE_RANK_MAX = 20;
90         /** Minimum value for user visible ranks (most relevant) */
91         public static final int VISIBLE_RANK_MIN = 1;
92     
93         static final boolean PLUGIN_DEBUG = Boolean.getBoolean("bugranker.plugin.debug");

95         static class Scorer {
96             private final HashMap<String, Integer> adjustment = new HashMap<String, Integer>();
97     
98             private final HashSet<String> isRelative = new HashSet<String>();
99     
100             int get(String key) {
101                 Integer v = adjustment.get(key);
102                 if (v == null) {
103                     return 0;
104                 }
105                 return v;
106             }
107     
108             boolean isRelative(String key) {
109                 return !adjustment.containsKey(key) || isRelative.contains(key);
110             }
111     
112             void storeAdjustment(String key, String value) {
113                 for (String k : key.split(",")) {
114                     char firstChar = value.charAt(0);
115                     if (firstChar == '+') {
116                         value = value.substring(1);
117                     }
118     
119                     int v = Integer.parseInt(value);
120                     adjustment.put(k, v);
121                     if (firstChar == '+' || firstChar == '-') {
122                         isRelative.add(k);
123                     }
124     
125     
126                 }
127             }
128         }

130         /**
131          * @param u
132          *            may be null. In this case, a default value will be used for
133          *            all bugs
134          * @throws IOException
135          */
136         BugRanker(@CheckForNull URL u) throws IOException {
137             if (u == null) {
138                 return;
139             }
140             BufferedReader in = UTF8.bufferedReader(u.openStream());
141             try {
142                 while (true) {
143                     String s = in.readLine();
144                     if (s == null) {
145                         break;
146                     }
147     
148                     s = s.trim();
149                     if (s.length() == 0) {
150                         continue;
151                     }
152     
153                     String parts[] = s.split(" ");
154                     if (parts.length != 3) {
155                         AnalysisContext.logError("Can't parse bug rank line: '" + s + "'. "
156                                 + "Valid line must contain 3 parts separated by spaces.");
157                         continue;
158                     }
159                     String rank = parts[0];
160                     String kind = parts[1];
161                     String what = parts[2];
162                     if ("BugPattern".equals(kind)) {
163                         bugPatterns.storeAdjustment(what, rank);
164                     } else if ("BugKind".equals(kind)) {
165                         bugKinds.storeAdjustment(what, rank);
166                     } else if ("Category".equals(kind)) {
167                         bugCategories.storeAdjustment(what, rank);
168                     } else {
169                         AnalysisContext.logError("Can't parse rank kind from line: '" + s + "'. "
170                                 + "Valid kind must be either 'BugPattern', 'BugKind' or 'Category'.");
171                     }
172                 }
173             } finally {
174                 Util.closeSilently(in);
175             }
176         }

191         private static int priorityAdjustment(int priority) {
192             switch (priority) {
193             case Priorities.HIGH_PRIORITY:
194                 return 0;
195             case Priorities.NORMAL_PRIORITY:
196                 return 2;
197             case Priorities.LOW_PRIORITY:
198                 return 5;
199             default:
200                 return 10;
201             }
202         }
203     
204         private static int adjustRank(int patternRank, int priority) {
205             int priorityAdjustment = priorityAdjustment(priority);
206             if (patternRank > VISIBLE_RANK_MAX) {
207                 return patternRank + priorityAdjustment;
208             }
209             return Math.max(VISIBLE_RANK_MIN, Math.min(patternRank + priorityAdjustment, VISIBLE_RANK_MAX));
210         }
211     
212         private static int rankBugPattern(BugPattern bugPattern, BugRanker... rankers) {
213             String type = bugPattern.getType();
214             int rank = 0;
215             for (BugRanker b : rankers) {
216                 if (b != null) {
217                     rank += b.bugPatterns.get(type);
218                     if (!b.bugPatterns.isRelative(type)) {
219                         return rank;
220                     }
221                 }
222             }
223             String kind = bugPattern.getAbbrev();
224             for (BugRanker b : rankers) {
225                 if (b != null) {
226                     rank += b.bugKinds.get(kind);
227                     if (!b.bugKinds.isRelative(kind)) {
228                         return rank;
229                     }
230                 }
231             }
232             String category = bugPattern.getCategory();
233             for (BugRanker b : rankers) {
234                 if (b != null) {
235                     rank += b.bugCategories.get(category);
236                     if (!b.bugCategories.isRelative(category)) {
237                         return rank;
238                     }
239                 }
240             }
241             return rank;
242         }

247             Plugin corePlugin = PluginLoader.getCorePluginLoader().getPlugin();

251         public static int findRank(BugInstance bug) {
252             int patternRank = findRank(bug.getBugPattern(), bug.getDetectorFactory());
253             return adjustRank(patternRank, bug.getPriority());
254         }

256         public static int findRank(BugPattern bugPattern, int priority) {
257             int patternRank = findRank(bugPattern, null);
258             return adjustRank(patternRank, priority);
259         }

262         private static AnalysisLocal<HashMap<BugPattern, Integer>> rankForBugPattern
263         = new AnalysisLocal<HashMap<BugPattern, Integer>>() {
264             @Override
265             protected HashMap<BugPattern, Integer> initialValue() {
266                 return new HashMap<BugPattern, Integer>();
267             }
268         };

271         public static int findRank(BugPattern pattern, @CheckForNull DetectorFactory detectorFactory) {
272             boolean haveCache = Global.getAnalysisCache() != null;
273             if (haveCache) {
274                 Integer cachedResult = rankForBugPattern.get().get(pattern);
275                 if (cachedResult != null) {
276                     return cachedResult;
277                 }
278             }
279     
280             int rank;
281             if (detectorFactory == null) {
282                 rank = findRankUnknownPlugin(pattern);
283             } else {
284                 Plugin plugin = detectorFactory.getPlugin();
285                 BugRanker pluginRanker = plugin.getBugRanker();
286                 BugRanker coreRanker = getCoreRanker();
287     
288                 if (pluginRanker == coreRanker) {
289                     rank = rankBugPattern(pattern, coreRanker);
290                 } else {
291                     rank = rankBugPattern(pattern, pluginRanker, coreRanker);
292                 }
293             }
294             if (haveCache) {
295                 rankForBugPattern.get().put(pattern, rank);
296             }
297             return rank;
298         }
299     
300         private static int findRankUnknownPlugin(BugPattern pattern) {
301     
302             List<BugRanker> rankers = new ArrayList<BugRanker>();
303             pluginLoop: for (Plugin plugin : Plugin.getAllPlugins()) {
304                 if (plugin.isCorePlugin()) {
305                     continue;
306                 }
307                 if (false) {
308                     rankers.add(plugin.getBugRanker());
309                     continue pluginLoop;
310                 }
311                 for (DetectorFactory df : plugin.getDetectorFactories()) {
312     
313                     if (df.getReportedBugPatterns().contains(pattern)) {
314                         if (PLUGIN_DEBUG) {
315                             System.out.println("Bug rank match " + plugin + " " + df + " for " + pattern);
316                         }
317                         rankers.add(plugin.getBugRanker());
318                         continue pluginLoop;
319                     }
320                 }
321                 if (PLUGIN_DEBUG) {
322                     System.out.println("plugin " + plugin + " doesn't match " + pattern);
323                 }
324     
325             }
326             rankers.add(getCoreRanker());
327     
328             return rankBugPattern(pattern, rankers.toArray(new BugRanker[rankers.size()]));
329         }

331         public static void trimToMaxRank(BugCollection origCollection, int maxRank) {
332             for (Iterator<BugInstance> i = origCollection.getCollection().iterator(); i.hasNext();) {
333                 BugInstance b = i.next();
334                 if (BugRanker.findRank(b) > maxRank) {
335                     i.remove();
336                 }
337     
338             }
339         }
