20     package edu.umd.cs.findbugs.util;
21     
22     import java.lang.reflect.InvocationTargetException;
23     import java.lang.reflect.Method;
24     import java.net.MalformedURLException;
25     import java.net.URL;
26     
27     import javax.annotation.CheckForNull;
28     import javax.swing.JOptionPane;
29     
30     
31     /**
32      * @author pugh
33      */
34     public class JavaWebStart {
35     
36         static final @CheckForNull
37         Method jnlpShowDocumentMethod;
38     
39         static final @CheckForNull
40         Method jnlpGetCodeBaseMethod;
41     
42         static final Object jnlpBasicService; // will not be null if
43         // jnlpShowMethod!=null
44     
45         static {
46             // attempt to set the JNLP BasicService object and its showDocument(URL)
47             // method
48             Method showMethod = null;
49             Method getCodeBase = null;
50     
51             Object showObject = null;
52             try {
53                 Class<?> serviceManagerClass = Class.forName("javax.jnlp.ServiceManager");
54                 Method lookupMethod = serviceManagerClass.getMethod("lookup", new Class[] { String.class });
55                 showObject = lookupMethod.invoke(null, new Object[] { "javax.jnlp.BasicService" });
56                 showMethod = showObject.getClass().getMethod("showDocument", new Class[] { URL.class });
57                 getCodeBase = showObject.getClass().getMethod("getCodeBase", new Class[] {});
58     
59             } catch (ClassNotFoundException e) {
60                 assert true;
61             } catch (NoSuchMethodException e) {
62                 assert true;
63             } catch (IllegalAccessException e) {
64                 assert true;
65             } catch (InvocationTargetException e) {
66                 assert true;
67             }
68             jnlpShowDocumentMethod = showMethod;
69             jnlpGetCodeBaseMethod = getCodeBase;
70             jnlpBasicService = showObject;
71     
72         }
73     
74         public static boolean isRunningViaJavaWebstart() {
75             return JavaWebStart.jnlpBasicService != null;
76         }
77     
78         public static URL resolveRelativeToJnlpCodebase(String s) throws MalformedURLException {
79             if (JavaWebStart.jnlpGetCodeBaseMethod != null) {
80                 try {
81                     URL base = (URL) JavaWebStart.jnlpGetCodeBaseMethod.invoke(JavaWebStart.jnlpBasicService);
82                     if (base != null) {
83                         return new URL(base, s);
84                     }
85                 } catch (RuntimeException e) {
86                     assert true;
87                 } catch (Exception e) {
88                     assert true;
89                 }
90             }
91             return new URL(s);
92         }
93     
94         static Boolean viaWebStart(URL url) throws IllegalArgumentException, IllegalAccessException, InvocationTargetException {
95             if (JavaWebStart.jnlpShowDocumentMethod == null) {
96                 throw new UnsupportedOperationException("Launch via web start not available");
97             }
98             return (Boolean) JavaWebStart.jnlpShowDocumentMethod.invoke(JavaWebStart.jnlpBasicService, url);
99         }
100     
101         static boolean showViaWebStart(URL url) {
102             if (JavaWebStart.jnlpShowDocumentMethod != null) {
103                 try {
104                     if (LaunchBrowser.DEBUG) {
105                         JOptionPane.showMessageDialog(null, "Trying browse via webstart");
106                     }
107     
108                     Boolean b = viaWebStart(url);
109                     boolean success = b != null && b.booleanValue();
110     
111                     if (LaunchBrowser.DEBUG) {
112                         JOptionPane.showMessageDialog(null, " browse via webstart: " + success);
113                     }
114                     return success;
115     
116                 } catch (InvocationTargetException ite) {
117                     assert true;
118                 } catch (IllegalAccessException iae) {
119                     assert true;
120                 }
121             }
122             return false;
123         }
