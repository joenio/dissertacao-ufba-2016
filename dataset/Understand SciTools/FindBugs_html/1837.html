22     /**
23      * <p>
24      * Finds invocations of JDK methods that rely on the default platform encoding.
25      * </p>
26      * <p>
27      * If a Java application assumes that the default platform encoding is
28      * acceptable, the app's behaviour will vary from platform to platform. In
29      * particular, conversions between byte[] and java.lang.String (in either
30      * direction) may yield inconsistent results. To ensure Java code is portable,
31      * the desired encoding should be specified explicitly wherever such a
32      * conversion takes place.
33      * </p>
34      * <p>
35      * This FindBugs pattern detects invocations of Java Class Library methods and
36      * constructors that are known to use the default platform encoding.
37      * </p>
38      *
39      * @author Robin Fernandes
40      */
41     public class DefaultEncodingDetector extends OpcodeStackDetector {

143         public DefaultEncodingDetector(BugReporter bugReporter) {
144             this.bugAccumulator = new BugAccumulator(bugReporter);
145             this.defaultEncodingAnnotationDatabase = new DefaultEncodingAnnotationDatabase();
146         }
147     
148         @Override
149         public boolean shouldVisit(JavaClass obj) {
150             Set<ClassDescriptor> called = getXClass().getCalledClassDescriptors();
151             for(ClassDescriptor c : defaultEncodingAnnotationDatabase.classes) {
152                 if (called.contains(c)) {
153                     return true;
154                 }
155             }
156     
157             return false;
158         }
159         @Override
160         public void visit(Code code) {
161             super.visit(code); // make callbacks to sawOpcode for all opcodes
162             bugAccumulator.reportAccumulatedBugs();
163     
164         }
165         @Override
166         public void sawOpcode(int seen) {
167             switch (seen) {
168             case INVOKEVIRTUAL:
169             case INVOKESPECIAL:
170             case INVOKESTATIC:
171                 XMethod callSeen = XFactory.createXMethod(MethodAnnotation.fromCalledMethod(this));
172                 DefaultEncodingAnnotation annotation = defaultEncodingAnnotationDatabase.getDirectAnnotation(callSeen);
173                 if (annotation != null) {
174                     bugAccumulator.accumulateBug(new BugInstance(this, "DM_DEFAULT_ENCODING", HIGH_PRIORITY).addClassAndMethod(this)
175                             .addCalledMethod(this), this);
176                 }
177                 break;
178             default:
179                 break;
180             }
181         }
