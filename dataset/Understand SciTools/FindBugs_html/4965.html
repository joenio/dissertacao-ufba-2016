229     
230             checker.checkForUpdates(Arrays.asList(plugin), true);
231             latch.await();
232         }

234         private String convertGlobToRE(String pattern) {
235             StringBuilder builder = new StringBuilder();
236             boolean first = true;
237             for (String blah : pattern.split("\\*")) {
238                 if (first) {
239                     first = false;
240                 } else {
241                     builder.append(".*");
242                 }
243                 builder.append(Pattern.quote(blah.replaceAll("'", "\"")));
244             }
245             return builder.toString();
246         }
