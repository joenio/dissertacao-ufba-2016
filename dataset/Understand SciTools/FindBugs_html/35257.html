22     package edu.umd.cs.findbugs.filter;
23     
24     import java.io.IOException;
25     
26     import edu.umd.cs.findbugs.BugAnnotation;
27     import edu.umd.cs.findbugs.BugInstance;
28     import edu.umd.cs.findbugs.TypeAnnotation;
29     import edu.umd.cs.findbugs.xml.XMLAttributeList;
30     import edu.umd.cs.findbugs.xml.XMLOutput;
31     
32     public class TypeMatcher implements Matcher {
33     
34         private final NameMatch descriptor;
35     
36         private final String role;
37     
38         private final String typeParameters;
39     
40         @Override
41         public String toString() {
42             return "Type(descriptor=\"" + descriptor.getValue() + "\")";
43         }
44     
45         public TypeMatcher(String descriptor, String role, String typeParameters) {
46             this.descriptor = new NameMatch(descriptor);
47             this.role = role;
48             this.typeParameters = typeParameters;
49         }
50     
51         @Override
52         public boolean match(BugInstance bugInstance) {
53             TypeAnnotation typeAnnotation = bugInstance.getPrimaryType();
54             if (role != null && !"".equals(role)) {
55                 for (BugAnnotation a : bugInstance.getAnnotations()) {
56                     if (a instanceof TypeAnnotation && role.equals(a.getDescription())) {
57                         typeAnnotation = (TypeAnnotation) a;
58                         break;
59                     }
60                 }
61             }
62             if(typeAnnotation == null){
63                 return false;
64             }
65             String typeDesctiptor = typeAnnotation.getTypeDescriptor();
66             if(!descriptor.match(typeDesctiptor)){
67                 return false;
68             }
69             if (typeParameters != null && !typeParameters.equals(typeAnnotation.getTypeParameters())) {
70                 return false;
71             }
72             return true;
73         }
74     
75         @Override
76         public void writeXML(XMLOutput xmlOutput, boolean disabled) throws IOException {
77             XMLAttributeList attributes = new XMLAttributeList().addAttribute("descriptor", descriptor.getSpec());
78             if (disabled) {
79                 attributes.addAttribute("disabled", "true");
80             }
81             attributes.addOptionalAttribute("typeParameters", typeParameters);
82             attributes.addOptionalAttribute("role", role);
83             xmlOutput.openCloseTag("Type", attributes);
84         }
85     }
