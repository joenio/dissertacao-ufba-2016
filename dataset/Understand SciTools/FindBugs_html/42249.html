74         /*
75          * (non-Javadoc)
76          *
77          * @see edu.umd.cs.findbugs.classfile.impl.IClassFactory#
78          * createFilesystemCodeBaseLocator(java.lang.String)
79          */
80         @Override
81         public ICodeBaseLocator createFilesystemCodeBaseLocator(String pathName) {
82             // Attempt to canonicalize the pathname.
83             // It's not fatal if we can't.
84             try {
85                 pathName = new File(pathName).getCanonicalPath();
86             } catch (IOException e) {
87                 // Ignore
88             }
89     
90             return new FilesystemCodeBaseLocator(pathName);
91         }
92     
93         /*
94          * (non-Javadoc)
95          *
96          * @see edu.umd.cs.findbugs.classfile.IClassFactory#
97          * createNestedArchiveCodeBaseLocator

99          */
100         @Override
101         public ICodeBaseLocator createNestedArchiveCodeBaseLocator(ICodeBase parentCodeBase, String path) {
102             return new NestedZipFileCodeBaseLocator(parentCodeBase, path);
103         }
104     
105         static IScannableCodeBase createFilesystemCodeBase(FilesystemCodeBaseLocator codeBaseLocator) throws IOException {
106             String fileName = codeBaseLocator.getPathName();
107     
108             File file = new File(fileName);
109     
110             if (!file.exists()) {
111                 throw new FileNotFoundException("File " + file.getAbsolutePath() + " doesn't exist");
112             } else if (!file.canRead()) {
113                 throw new IOException("File " + file.getAbsolutePath() + " not readable");
114             } else if (file.isDirectory()) {
115                 return new DirectoryCodeBase(codeBaseLocator, file);
116             } else if (!file.isFile()) {
117                 throw new IOException("File " + file.getAbsolutePath() + " is not a normal file");
118             } else if (fileName.endsWith(".class")) {
119                 return new SingleFileCodeBase(codeBaseLocator, fileName);
120             } else {
121                 return ZipCodeBaseFactory.makeZipCodeBase(codeBaseLocator, file);
122             }
123         }
124     
125         static IScannableCodeBase createNestedZipFileCodeBase(NestedZipFileCodeBaseLocator codeBaseLocator)
126                 throws ResourceNotFoundException, IOException {
127             return new NestedZipFileCodeBase(codeBaseLocator);
128         }
