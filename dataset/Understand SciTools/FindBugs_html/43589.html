58             NONE, EMPTY_ARRAY
59         }

61         private static final Map<String, Map<String, AllowedParameter>> MUTABLE_COLLECTION_METHODS = new HashMap<>();
62         static {
63             MUTABLE_COLLECTION_METHODS.put("java/util/Arrays", Collections.singletonMap("asList", AllowedParameter.EMPTY_ARRAY));
64             Map<String, AllowedParameter> listsMap = new HashMap<>();
65             listsMap.put("newArrayList", AllowedParameter.NONE);
66             listsMap.put("newLinkedList", AllowedParameter.NONE);
67             MUTABLE_COLLECTION_METHODS.put("com/google/common/collect/Lists", listsMap);
68             Map<String, AllowedParameter> setsMap = new HashMap<>();
69             setsMap.put("newHashSet", AllowedParameter.NONE);
70             setsMap.put("newTreeSet", AllowedParameter.NONE);
71             MUTABLE_COLLECTION_METHODS.put("com/google/common/collect/Sets", setsMap);
72         }

74         static String extractPackage(String c) {
75             int i = c.lastIndexOf('/');
76             if (i < 0) {
77                 return "";
78             }
79             return c.substring(0, i);
80         }

82         static boolean mutableSignature(String sig) {
83             return sig.equals("Ljava/util/Hashtable;") || sig.equals("Ljava/util/Date;") ||
84                     sig.equals("Ljava/sql/Date;") ||
85                     sig.equals("Ljava/sql/Timestamp;") ||
86                     sig.charAt(0) == '[';
87         }

123         /**
124          * Eclipse uses reflection to initialize NLS message bundles. Classes which
125          * using this mechanism are usualy extending org.eclipse.osgi.util.NLS class
126          * and contains lots of public static String fields which are used as
127          * message constants. Unfortunately these fields cannot be final, so FB
128          * reports tons of warnings for such Eclipse classes.
129          */
130         private boolean isEclipseNLS;

136         @Override
137         public void visit(JavaClass obj) {
138             super.visit(obj);
139             int flags = obj.getAccessFlags();
140             publicClass = (flags & ACC_PUBLIC) != 0 && !getDottedClassName().startsWith("sun.");
141     
142             packageName = extractPackage(getClassName());
143             isEclipseNLS = "org.eclipse.osgi.util.NLS".equals(obj.getSuperclassName());
144         }

146         @Override
147         public void visit(Method obj) {
148             zeroOnTOS = false;
149             // System.out.println(methodName);
150             inStaticInitializer = getMethodName().equals("<clinit>");
151         }

153         @Override
154         public void visit(Code obj) {
155             writtenInMethod.clear();
156             writtenTwiceInMethod.clear();
157             super.visit(obj);
158             if (inStaticInitializer) {
159                 needsRefactoringToBeFinal.addAll(writtenTwiceInMethod);
160             }
161             writtenInMethod.clear();
162             writtenTwiceInMethod.clear();
163         }

165         @Override
166         public void sawOpcode(int seen) {
167             // System.out.println("saw\t" + OPCODE_NAMES[seen] + "\t" + zeroOnTOS);
168             switch (seen) {
169             case GETSTATIC:
170             case PUTSTATIC:
171     
172                 XField xField = getXFieldOperand();
173                 if (xField == null) {
174                     break;
175                 }
176                 if (!interesting(xField)) {
177                     break;
178                 }
179     
180                 boolean samePackage = packageName.equals(extractPackage(xField.getFieldDescriptor().getSlashedClassName()));
181                 boolean initOnly = seen == GETSTATIC || getClassName().equals(getClassConstantOperand()) && inStaticInitializer;
182                 boolean safeValue = seen == GETSTATIC || emptyArrayOnTOS
183                         || AnalysisContext.currentXFactory().isEmptyArrayField(xField) || !mutableSignature(getSigConstantOperand());
184     
185                 if (seen == GETSTATIC) {
186                     readAnywhere.add(xField);
187                 }
188                 if (seen == PUTSTATIC) {
189                     if (xField.isFinal() && mutableCollectionJustCreated) {
190                         mutableCollection.add(xField);
191                     }
192                     if (!writtenInMethod.add(xField)) {
193                         writtenTwiceInMethod.add(xField);
194                     }
195                 }
196     
197                 if (!samePackage) {
198                     outsidePackage.add(xField);
199                 }
200     
201                 if (!initOnly) {
202                     notFinal.add(xField);
203                 }
204     
205                 if (!safeValue) {
206                     unsafeValue.add(xField);
207                 }
208     
209                 // Remove inStaticInitializer check to report all source lines of
210                 // first use
211                 // doing so, however adds quite a bit of memory bloat.
212                 if (inStaticInitializer && !firstFieldUse.containsKey(xField)) {
213                     SourceLineAnnotation sla = SourceLineAnnotation.fromVisitedInstruction(this);
214                     firstFieldUse.put(xField, sla);
215                 }
216                 break;
217             case ANEWARRAY:
218             case NEWARRAY:
219                 if (zeroOnTOS) {
220                     emptyArrayOnTOS = true;
221                 }
222                 zeroOnTOS = false;
223                 return;
224             case ICONST_0:
225                 zeroOnTOS = true;
226                 emptyArrayOnTOS = false;
227                 return;
228             case INVOKESPECIAL:
229                 if (inStaticInitializer && "<init>".equals(getMethodDescriptorOperand().getName())) {
230                     ClassDescriptor classDescriptor = getClassDescriptorOperand();
231                     if (MUTABLE_COLLECTION_CLASSES.contains(classDescriptor.getClassName())) {
232                         mutableCollectionJustCreated = true;
233                         return;
234                     }
235                     try {
236                         /* Check whether it's statically initialized anonymous class like this:
237                          * public static final Map map = new HashMap() {{put("a", "b");}}
238                          * We do not check whether all modification methods are overridden or not for simplicity:
239                          * Skip if there's at least one method is present
240                          */
241                         XClass xClass = classDescriptor.getXClass();
242                         ClassDescriptor superclassDescriptor = xClass.getSuperclassDescriptor();
243                         if (superclassDescriptor != null
244                                 && MUTABLE_COLLECTION_CLASSES.contains(superclassDescriptor.getClassName())) {
245                             mutableCollectionJustCreated = true;
246                             for (XMethod xMethod : xClass.getXMethods()) {
247                                 if (xMethod != null && !"<init>".equals(xMethod.getName()) && !"<clinit>".equals(xMethod.getName())) {
248                                     mutableCollectionJustCreated = false;
249                                     break;
250                                 }
251                             }
252                             return;
253                         }
254                     } catch (CheckedAnalysisException e) {
255                         // ignore
256                     }
257                 }
258                 break;
259             case INVOKESTATIC:
260                 if (inStaticInitializer) {
261                     Map<String, AllowedParameter> methods = MUTABLE_COLLECTION_METHODS.get(getMethodDescriptorOperand()
262                             .getSlashedClassName());
263                     if (methods != null) {
264                         String name = getMethodDescriptorOperand().getName();
265                         AllowedParameter allowedParameter = methods.get(name);
266                         if (allowedParameter == AllowedParameter.NONE
267                                 || (allowedParameter == AllowedParameter.EMPTY_ARRAY && !emptyArrayOnTOS)) {
268                             mutableCollectionJustCreated = true;
269                             return;
270                         }
271                     }
272                 }
273                 break;
274             }
275             zeroOnTOS = false;
276             emptyArrayOnTOS = false;
277             mutableCollectionJustCreated = false;
278         }

280         private boolean isCollection(String signature) {
281             if (signature.startsWith("L") && signature.endsWith(";")) {
282                 String fieldClass = signature.substring(1, signature.length() - 1);
283                 return COLLECTION_SUPERCLASSES.contains(fieldClass) || MUTABLE_COLLECTION_CLASSES.contains(fieldClass);
284             }
285             return false;
286         }
287     
288         private boolean interesting(XField f) {
289             if (!f.isPublic() && !f.isProtected()) {
290                 return false;
291             }
292             if (!f.isStatic() || f.isSynthetic() || f.isVolatile()) {
293                 return false;
294             }
295             if (!f.isFinal()) {
296                 return true;
297             }
298             boolean isArray = f.getSignature().charAt(0) == '[';
299             if (!(isArray || isCollection(f.getSignature()))) {
300                 return false;
301             }
302             return true;
303     
304         }

306         @Override
307         public void visit(Field obj) {
308             super.visit(obj);
309             int flags = obj.getAccessFlags();
310             boolean isStatic = (flags & ACC_STATIC) != 0;
311             if (!isStatic) {
312                 return;
313             }
314             boolean isVolatile = (flags & ACC_VOLATILE) != 0;
315             if (isVolatile) {
316                 return;
317             }
318             boolean isFinal = (flags & ACC_FINAL) != 0;
319             boolean isPublic = publicClass && (flags & ACC_PUBLIC) != 0;
320             boolean isProtected = publicClass && (flags & ACC_PROTECTED) != 0;
321             if (!isPublic && !isProtected) {
322                 return;
323             }
324     
325             boolean isArray = getFieldSig().charAt(0) == '[';
326     
327             if (isFinal && !(isArray || isCollection(getFieldSig()))) {
328                 return;
329             }
330             if (isEclipseNLS && getFieldSig().equals("Ljava/lang/String;")) {
331                 return;
332             }
333     
334             seen.add(getXField());
335         }

337         @Override
338         public void report() {
339             /*
340              * for(Iterator i = unsafeValue.iterator(); i.hasNext(); ) {
341              * System.out.println("Unsafe: " + i.next()); }
342              */
343             for (XField f : seen) {
344                 boolean isFinal = f.isFinal();
345                 String className = f.getClassName();
346                 String fieldSig = f.getSignature();
347                 String fieldName = f.getName();
348                 boolean couldBeFinal = !isFinal && !notFinal.contains(f);
349                 //            boolean isPublic = f.isPublic();
350                 boolean couldBePackage = !outsidePackage.contains(f);
351                 boolean isMutableCollection = mutableCollection.contains(f);
352                 boolean movedOutofInterface = false;
353     
354                 try {
355                     XClass xClass = Global.getAnalysisCache().getClassAnalysis(XClass.class, f.getClassDescriptor());
356                     movedOutofInterface = couldBePackage && xClass.isInterface();
357                 } catch (CheckedAnalysisException e) {
358                     assert true;
359                 }
360                 boolean isHashtable = fieldSig.equals("Ljava/util/Hashtable;");
361                 boolean isArray = fieldSig.charAt(0) == '[' && unsafeValue.contains(f);
362                 boolean isReadAnywhere = readAnywhere.contains(f);
363                 //            if (false) {
364                 //                System.out.println(className + "." + fieldName + " : " + fieldSig + "\t" + isHashtable + "\t" + isArray);
365                 //            }
366     
367                 String bugType;
368                 int priority = NORMAL_PRIORITY;
369                 if (isFinal && !isHashtable && !isArray && !isMutableCollection) {
370                     continue;
371                 } else if (movedOutofInterface) {
372                     bugType = "MS_OOI_PKGPROTECT";
373                 } else if (couldBePackage && couldBeFinal && (isHashtable || isArray)) {
374                     bugType = "MS_FINAL_PKGPROTECT";
375                 } else if (couldBeFinal && !isHashtable && !isArray) {
376                     bugType = "MS_SHOULD_BE_FINAL";
377                     if (needsRefactoringToBeFinal.contains(f)) {
378                         bugType = "MS_SHOULD_BE_REFACTORED_TO_BE_FINAL";
379                     }
380                     if (fieldName.equals(fieldName.toUpperCase()) || fieldSig.charAt(0) == 'L') {
381                         priority = HIGH_PRIORITY;
382                     }
383                 } else if (couldBePackage) {
384                     bugType = isMutableCollection ? "MS_MUTABLE_COLLECTION_PKGPROTECT" : "MS_PKGPROTECT";
385                 } else if (isHashtable) {
386                     bugType = "MS_MUTABLE_HASHTABLE";
387                     if (!isFinal) {
388                         priority = HIGH_PRIORITY;
389                     }
390                 } else if (isArray) {
391                     bugType = "MS_MUTABLE_ARRAY";
392                     if (fieldSig.indexOf('L') >= 0 || !isFinal) {
393                         priority = HIGH_PRIORITY;
394                     }
395                 } else if (isMutableCollection) {
396                     bugType = "MS_MUTABLE_COLLECTION";
397                     priority = HIGH_PRIORITY;
398                 } else if (!isFinal) {
399                     bugType = "MS_CANNOT_BE_FINAL";
400                 } else {
401                     throw new IllegalStateException("impossible");
402                 }
403                 if (!isReadAnywhere) {
404                     priority = LOW_PRIORITY;
405                 }
406     
407                 BugInstance bug = new BugInstance(this, bugType, priority).addClass(className).addField(f);
408                 SourceLineAnnotation firstPC = firstFieldUse.get(f);
409                 if (firstPC != null) {
410                     bug.addSourceLine(firstPC);
411                 }
412                 bugReporter.reportBug(bug);
413     
414             }
415         }
