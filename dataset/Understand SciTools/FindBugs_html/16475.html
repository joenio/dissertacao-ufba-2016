20     package edu.umd.cs.findbugs.ba.bcp;
21     
22     /**
23      * The result of matching a single PatternElement against a single instruction.
24      * Specifies the PatternElement and the (possibly updated) BindingSet. The
25      * reason we need this class is because some kinds of PatternElements, such as
26      * MatchAny, may use it to indicate that a child PatternElement was the one that
27      * actually matched the instruction.
28      *
29      * @author David Hovemeyer
30      * @see PatternElement
31      * @see BindingSet
32      */
33     public class MatchResult {
34         private final PatternElement patternElement;
35     
36         private final BindingSet bindingSet;
37     
38         /**
39          * Constructor.
40          *
41          * @param patternElement
42          *            the PatternElement that matched the instruction
43          * @param bindingSet
44          *            the possibly updated BindingSet
45          */
46         public MatchResult(PatternElement patternElement, BindingSet bindingSet) {
47             this.patternElement = patternElement;
48             this.bindingSet = bindingSet;
49         }
50     
51         /**
52          * Get the PatternElement.
53          */
54         public PatternElement getPatternElement() {
55             return patternElement;
56         }
57     
58         /**
59          * Get the BindingSet.
60          */
61         public BindingSet getBindingSet() {
62             return bindingSet;
63         }
64     
65     }
