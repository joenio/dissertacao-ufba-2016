46     public class InefficientInitializationInsideLoop extends OpcodeStackDetector {
47         private static final MethodDescriptor NODELIST_GET_LENGTH = new MethodDescriptor("org/w3c/dom/NodeList", "getLength", "()I");
48         private static final MethodDescriptor PATTERN_COMPILE = new MethodDescriptor("java/util/regex/Pattern", "compile", "(Ljava/lang/String;)Ljava/util/regex/Pattern;", true);
49         private static final MethodDescriptor PATTERN_COMPILE_2 = new MethodDescriptor("java/util/regex/Pattern", "compile", "(Ljava/lang/String;I)Ljava/util/regex/Pattern;", true);
50         private static final MethodDescriptor PATTERN_MATCHES = new MethodDescriptor("java/util/regex/Pattern", "matches", "(Ljava/lang/String;Ljava/lang/CharSequence;)Z", true);
51         private static final MethodDescriptor STRING_REPLACEALL = new MethodDescriptor("java/lang/String", "replaceAll", "(Ljava/lang/String;Ljava/lang/String;)Ljava/lang/String;");
52         private static final MethodDescriptor STRING_REPLACEFIRST = new MethodDescriptor("java/lang/String", "replaceFirst", "(Ljava/lang/String;Ljava/lang/String;)Ljava/lang/String;");
53         private static final MethodDescriptor STRING_MATCHES = new MethodDescriptor("java/lang/String", "matches", "(Ljava/lang/String;)Z");
54         private static final MethodDescriptor STRING_SPLIT = new MethodDescriptor("java/lang/String", "split", "(Ljava/lang/String;)[Ljava/lang/String;");
55         private static final MethodDescriptor STRING_SPLIT_2 = new MethodDescriptor("java/lang/String", "split", "(Ljava/lang/String;I)[Ljava/lang/String;");
56     
57         private static final Set<MethodDescriptor> implicitPatternMethods = new HashSet<>(Arrays.asList(PATTERN_MATCHES,
58                 STRING_MATCHES, STRING_REPLACEALL, STRING_REPLACEFIRST, STRING_SPLIT, STRING_SPLIT_2));
59     
60         private static final List<MethodDescriptor> methods = new ArrayList<>();
61     
62         static {
63             methods.add(NODELIST_GET_LENGTH);
64             methods.add(PATTERN_COMPILE);
65             methods.add(PATTERN_COMPILE_2);
66             methods.addAll(implicitPatternMethods);
67         }
68     
69         private SortedMap<Integer, BugInstance> matched;
70     
71         private SortedMap<Integer, Integer> conditions;
72     
73         private SortedMap<Integer, Integer> sources;
74     
75         private final BugReporter bugReporter;
76     
77         public InefficientInitializationInsideLoop(BugReporter bugReporter) {
78             this.bugReporter = bugReporter;
79         }

81         @Override
82         public void visitClassContext(ClassContext classContext) {
83             if (hasInterestingMethod(classContext.getJavaClass().getConstantPool(), methods)
84                     || hasInterestingClass(classContext.getJavaClass().getConstantPool(),
85                             Collections.singleton("java/sql/Connection"))) {
86                 super.visitClassContext(classContext);
87             }
88         }
89     
90         @Override
91         public void visitMethod(Method obj) {
92             matched = new TreeMap<>();
93             conditions = new TreeMap<>();
94             sources = new TreeMap<>();
95             super.visitMethod(obj);
96         }

98         /**
99          * Since JDK 1.7 there's a special branch in String.split which works very fast for one-character pattern
100          * We do not report a bug if this case takes place
101          * (in fact precompilation will make split much slower since this fast path doesn't use regexp engine at all)
102          * @param regex regex to test whether it's suitable for the fast path
103          * @return true if fast path is possible
104          */
105         private boolean isFastPath(String regex) {
106             char ch;
107             return (((regex.length() == 1 && ".$|()[{^?*+\\".indexOf(ch = regex.charAt(0)) == -1) || (regex.length() == 2
108                     && regex.charAt(0) == '\\' && (((ch = regex.charAt(1)) - '0') | ('9' - ch)) < 0 && ((ch - 'a') | ('z' - ch)) < 0 && ((ch - 'A') | ('Z' - ch)) < 0)) && (ch < Character.MIN_HIGH_SURROGATE || ch > Character.MAX_LOW_SURROGATE));
109         }
110     
111         @Override
112         public void sawOpcode(int seen) {
113             if (seen == INVOKEINTERFACE && getClassConstantOperand().equals("java/sql/Connection")
114                     && getMethodDescriptorOperand().getName().equals("prepareStatement") && hasConstantArguments()) {
115                 matched.put(getPC(), new BugInstance(this, "IIL_PREPARE_STATEMENT_IN_LOOP", NORMAL_PRIORITY).addClassAndMethod(this)
116                         .addSourceLine(this, getPC()).addCalledMethod(this));
117             } else if (seen == INVOKEINTERFACE && getMethodDescriptorOperand().equals(NODELIST_GET_LENGTH)) {
118                 Item item = getStack().getStackItem(0);
119                 XMethod returnValueOf = item.getReturnValueOf();
120                 if(returnValueOf != null && returnValueOf.getClassName().startsWith("org.w3c.dom.") && returnValueOf.getName().startsWith("getElementsByTagName")) {
121                     matched.put(getPC(),
122                             new BugInstance(this, "IIL_ELEMENTS_GET_LENGTH_IN_LOOP", NORMAL_PRIORITY).addClassAndMethod(this)
123                             .addSourceLine(this, getPC()).addCalledMethod(this));
124                     sources.put(getPC(), item.getPC());
125                 }
126             } else if (seen == INVOKESTATIC
127                     && (getMethodDescriptorOperand().equals(PATTERN_COMPILE) || getMethodDescriptorOperand()
128                             .equals(PATTERN_COMPILE_2)) && hasConstantArguments()) {
129                 String regex = getFirstArgument();
130                 matched.put(getPC(), new BugInstance(this, "IIL_PATTERN_COMPILE_IN_LOOP", NORMAL_PRIORITY).addClassAndMethod(this)
131                         .addSourceLine(this, getPC()).addCalledMethod(this).addString(regex).describe(StringAnnotation.REGEX_ROLE));
132             } else if ((seen == INVOKESTATIC || seen == INVOKEVIRTUAL) && implicitPatternMethods.contains(getMethodDescriptorOperand())) {
133                 String regex = getFirstArgument();
134                 if (regex != null && !(getNameConstantOperand().equals("split") && isFastPath(regex))) {
135                     BugInstance bug = new BugInstance(this, "IIL_PATTERN_COMPILE_IN_LOOP_INDIRECT", LOW_PRIORITY)
136                     .addClassAndMethod(this).addSourceLine(this, getPC()).addCalledMethod(this).addString(regex)
137                     .describe(StringAnnotation.REGEX_ROLE);
138                     matched.put(getPC(), bug);
139                 }
140             } else if (isBranch(seen) && getBranchOffset() > 0) {
141                 conditions.put(getPC(), getBranchTarget());
142             } else if (!matched.isEmpty() && isBranch(seen) && getBranchOffset() < 0) {
143                 for (Entry<Integer, BugInstance> entry : matched.tailMap(getBranchTarget()).entrySet()) {
144                     Integer source = sources.get(entry.getKey());
145                     if(source != null && (source > getBranchTarget() && source < getPC())) {
146                         // Object was created in the same loop: ignore
147                         return;
148                     }
149                     for (int target : conditions.subMap(getBranchTarget(), entry.getKey()).values()) {
150                         if (target > entry.getKey() && target < getPC()) {
151                             return;
152                         }
153                     }
154                     bugReporter.reportBug(entry.getValue());
155                 }
156             }
157         }
158     
159         /**
160          * @return first argument of the called method if it's a constant
161          */
162         private String getFirstArgument() {
163             Object value = getStack().getStackItem(getNumberArguments(getMethodDescriptorOperand().getSignature()) - 1)
164                     .getConstant();
165             return value == null ? null : value.toString();
166         }
167     
168         /**
169          * @return true if only constants are passed to the called method
170          */
171         private boolean hasConstantArguments() {
172             int nArgs = getNumberArguments(getMethodDescriptorOperand().getSignature());
173             for (int i = 0; i < nArgs; i++) {
174                 if (getStack().getStackItem(i).getConstant() == null) {
175                     return false;
176                 }
177             }
178             return true;
179         }
