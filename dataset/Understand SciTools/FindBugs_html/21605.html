35     /**
36      * Analysis engine to produce MethodGen objects for analyzed methods.
37      *
38      * @author David Hovemeyer
39      * @author Bill Pugh
40      */
41     public class MethodGenFactory extends AnalysisFactory<MethodGen> {
42         /**
43          * Constructor.
44          */
45         public MethodGenFactory() {
46             super("MethodGen construction", MethodGen.class);
47         }
48     
49         /*
50          * (non-Javadoc)
51          *
52          * @see
53          * edu.umd.cs.findbugs.classfile.IAnalysisEngine#analyze(edu.umd.cs.findbugs
54          * .classfile.IAnalysisCache, java.lang.Object)
55          */
56         @Override
57         public MethodGen analyze(IAnalysisCache analysisCache, MethodDescriptor descriptor) throws CheckedAnalysisException {
58             Method method = getMethod(analysisCache, descriptor);
59     
60             if (method.getCode() == null) {
61                 return null;
62             }
63             XMethod xmethod =  XFactory.createXMethod(descriptor);
64             if (xmethod.usesInvokeDynamic() && false) {
65                 AnalysisContext.currentAnalysisContext().analysisSkippedDueToInvokeDynamic(xmethod);
66                 return null;
67             }
68     
69             try {
70                 AnalysisContext analysisContext = AnalysisContext.currentAnalysisContext();
71                 JavaClass jclass = getJavaClass(analysisCache, descriptor.getClassDescriptor());
72                 ConstantPoolGen cpg = getConstantPoolGen(analysisCache, descriptor.getClassDescriptor());
73     
74                 String methodName = method.getName();
75                 int codeLength = method.getCode().getCode().length;
76                 String superclassName = jclass.getSuperclassName();
77                 if (codeLength > 6000 && "<clinit>".equals(methodName) && "java.lang.Enum".equals(superclassName)) {
78                     analysisContext.getLookupFailureCallback().reportSkippedAnalysis(
79                             new JavaClassAndMethod(jclass, method).toMethodDescriptor());
80                     return null;
81                 }
82                 if (analysisContext.getBoolProperty(AnalysisFeatures.SKIP_HUGE_METHODS)) {
83                     if (codeLength > 6000 || ("<clinit>".equals(methodName) || "getContents".equals(methodName)) && codeLength > 2000) {
84                         analysisContext.getLookupFailureCallback().reportSkippedAnalysis(
85                                 new JavaClassAndMethod(jclass, method).toMethodDescriptor());
86                         return null;
87                     }
88                 }
89     
90                 return new MethodGen(method, jclass.getClassName(), cpg);
91             } catch (Exception e) {
92                 AnalysisContext.logError("Error constructing methodGen", e);
93                 return null;
94             }
95         }
96     }
