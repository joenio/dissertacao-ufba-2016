20     package edu.umd.cs.findbugs.workflow;
21     
22     import java.io.IOException;
23     import java.io.InputStream;
24     import java.util.HashSet;
25     
26     import edu.umd.cs.findbugs.Project;
27     import edu.umd.cs.findbugs.SourceLineAnnotation;
28     import edu.umd.cs.findbugs.ba.SourceFinder;
29     
30     /**
31      * @author pugh
32      */
33     public class SourceSearcher {
34         final HashSet<String> sourceFound = new HashSet<String>();
35     
36         final HashSet<String> sourceNotFound = new HashSet<String>();
37     
38         private final SourceFinder sourceFinder;
39     
40         public SourceSearcher(Project project) {
41             sourceFinder = new SourceFinder(project);
42         }
43     
44         public boolean findSource(SourceLineAnnotation srcLine) {
45             if (srcLine == null) {
46                 return false;
47             }
48             String cName = srcLine.getClassName();
49             if (sourceFound.contains(cName)) {
50                 return true;
51             }
52             if (sourceNotFound.contains(cName)) {
53                 return false;
54             }
55     
56             boolean result = sourceFinder.hasSourceFile(srcLine);
57             return result;
58         }
59     
60         public boolean findSource0(SourceLineAnnotation srcLine) {
61             if (srcLine == null) {
62                 return false;
63             }
64             String cName = srcLine.getClassName();
65             if (sourceFound.contains(cName)) {
66                 return true;
67             }
68             if (sourceNotFound.contains(cName)) {
69                 return false;
70             }
71     
72             try {
73                 InputStream in = sourceFinder.openSource(srcLine);
74                 in.close();
75                 sourceFound.add(cName);
76                 return true;
77             } catch (IOException e1) {
78                 sourceNotFound.add(cName);
79                 return false;
80             }
81         }
82     
83     }
