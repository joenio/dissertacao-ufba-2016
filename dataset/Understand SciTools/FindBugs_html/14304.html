20     package edu.umd.cs.findbugs.ba.jsr305;
21     
22     import java.util.HashMap;
23     import java.util.Map;
24     
25     import edu.umd.cs.findbugs.SystemProperties;
26     import edu.umd.cs.findbugs.classfile.MethodDescriptor;
27     import edu.umd.cs.findbugs.util.DualKeyHashMap;
28     
29     /**
30      * Store computed type qualifiers for method parameters and return values. This
31      * allows interprocedural checking of type qualifiers.
32      *
33      * @author David Hovemeyer
34      */
35     public class TypeQualifierDatabase {
36         /**
37          * If true, populate and use interprocedural database.
38          */
39         public static final boolean USE_DATABASE = SystemProperties.getBoolean("ctq.usedatabase", true);
40     
41         public static final boolean DEBUG = SystemProperties.getBoolean("ctq.db.debug");
42     
43         private final Map<MethodDescriptor, Map<TypeQualifierValue<?>, TypeQualifierAnnotation>> returnValueMap;
44     
45         private final DualKeyHashMap<MethodDescriptor, Integer, Map<TypeQualifierValue<?>, TypeQualifierAnnotation>> parameterMap;
46     
47         /**
48          * Constructor.
49          */
50         public TypeQualifierDatabase() {
51             this.returnValueMap = new HashMap<MethodDescriptor, Map<TypeQualifierValue<?>, TypeQualifierAnnotation>>();
52             this.parameterMap = new DualKeyHashMap<MethodDescriptor, Integer, Map<TypeQualifierValue<?>, TypeQualifierAnnotation>>();
53         }
54     
55         /**
56          * Set a TypeQualifierAnnotation on a method return value.
57          *
58          * @param methodDesc
59          *            the method
60          * @param tqv
61          *            the type qualifier
62          * @param tqa
63          *            the type qualifier annotation
64          */
65         public void setReturnValue(MethodDescriptor methodDesc, TypeQualifierValue<?> tqv, TypeQualifierAnnotation tqa) {
66             Map<TypeQualifierValue<?>, TypeQualifierAnnotation> map = returnValueMap.get(methodDesc);
67             if (map == null) {
68                 map = new HashMap<TypeQualifierValue<?>, TypeQualifierAnnotation>();
69                 returnValueMap.put(methodDesc, map);
70             }
71             map.put(tqv, tqa);
72     
73             if (DEBUG) {
74                 System.out.println("tqdb: " + methodDesc + " for " + tqv + " ==> " + tqa);
75             }
76         }
77     
78         /**
79          * Get the TypeQualifierAnnotation on a method return value.
80          *
81          * @param methodDesc
82          *            the method
83          * @param tqv
84          *            the type qualifier
85          * @return the type qualifier annotation on the method return value, or null
86          *         if no (interesting) type qualifier annotation was computed for
87          *         this method
88          */
89         public TypeQualifierAnnotation getReturnValue(MethodDescriptor methodDesc, TypeQualifierValue<?> tqv) {
90             //
91             // TODO: handling of overridden methods?
92             //
93             Map<TypeQualifierValue<?>, TypeQualifierAnnotation> map = returnValueMap.get(methodDesc);
94             if (map == null) {
95                 return null;
96             }
97             return map.get(tqv);
98         }
99     
100         /**
101          * Set a TypeQualifierAnnotation on a method parameter.
102          *
103          * @param methodDesc
104          *            the method
105          * @param param
106          *            the parameter (0 == first parameter)
107          * @param tqv
108          *            the type qualifier
109          * @param tqa
110          *            the type qualifier annotation
111          */
112         public void setParameter(MethodDescriptor methodDesc, int param, TypeQualifierValue<?> tqv, TypeQualifierAnnotation tqa) {
113             Map<TypeQualifierValue<?>, TypeQualifierAnnotation> map = parameterMap.get(methodDesc, param);
114             if (map == null) {
115                 map = new HashMap<TypeQualifierValue<?>, TypeQualifierAnnotation>();
116                 parameterMap.put(methodDesc, param, map);
117             }
118             map.put(tqv, tqa);
119     
120             if (DEBUG) {
121                 System.out.println("tqdb: " + methodDesc + " parameter " + param + " for " + tqv + " ==> " + tqa);
122             }
123         }
124     
125         /**
126          * Get the TypeQualifierAnnotation on a parameter.
127          *
128          * @param methodDesc
129          *            the method
130          * @param param
131          *            the parameter (0 == first parameter)
132          * @param tqv
133          *            the type qualifier
134          * @return the type qualifier annotation on the method return value, or null
135          *         if no (interesting) type qualifier annotation was computed for
136          *         this method
137          */
138         public TypeQualifierAnnotation getParameter(MethodDescriptor methodDesc, int param, TypeQualifierValue<?> tqv) {
139             //
140             // TODO: handling of overridden methods?
141             //
142             Map<TypeQualifierValue<?>, TypeQualifierAnnotation> map = parameterMap.get(methodDesc, param);
143             if (map == null) {
144                 return null;
145             }
146             return map.get(tqv);
147         }
148     }
