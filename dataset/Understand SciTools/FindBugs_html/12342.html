19     package edu.umd.cs.findbugs.ba.interproc;
20     
21     import java.util.BitSet;
22     import java.util.Iterator;
23     
24     /**
25      * Method property recording which parameters are have some property
26      * (originally, which were required to be nonnull, now made more generic)
27      *
28      * @author David Hovemeyer
29      */
30     public class ParameterProperty {
31         /**
32          * Maximum number of parameters that can be represented by a
33          * ParameterProperty.
34          */
35         public static final int MAX_PARAMS = 32;
36     
37         private int bits;
38     
39         /**
40          * Constructor. Parameters are all assumed not to be non-null.
41          */
42         public ParameterProperty() {
43             this.bits = 0;
44         }
45     
46         /**
47          * Constructor. Parameters are all assumed not to be non-null.
48          */
49         public ParameterProperty(int bits) {
50             this.bits = bits;
51         }
52     
53         /**
54          * Get the non-null param bitset.
55          *
56          * @return the non-null param bitset
57          */
58         public int getParamsWithProperty() {
59             return bits;
60         }
61     
62         public Iterable<Integer> iterable() {
63             return new Iterable<Integer>() {
64     
65                 @Override
66                 public Iterator<Integer> iterator() {
67                     return new Iterator<Integer>() {
68                         int nextInt = 0;
69                         {
70                             advanceNextInt();
71                         }
72     
73                         private void advanceNextInt() {
74                             while (!hasProperty(nextInt) && nextInt < 32) {
75                                 nextInt++;
76                             }
77                             if (nextInt >= 32) {
78                                 nextInt = -1;
79                             }
80                         }
81     
82                         @Override
83                         public boolean hasNext() {
84                             return nextInt >= 0;
85                         }
86     
87                         @Override
88                         public Integer next() {
89                             int result = nextInt;
90                             nextInt++;
91                             advanceNextInt();
92                             return result;
93                         }
94     
95                         @Override
96                         public void remove() {
97                             throw new UnsupportedOperationException();
98     
99                         }
100                     };
101                 }
102     
103             };
104         }
105     
106         /**
107          * Set the non-null param bitset.
108          *
109          * @param nonNullParamSet
110          *            the non-null param bitset
111          */
112         public void setParamsWithProperty(int nonNullParamSet) {
113             this.bits = nonNullParamSet;
114         }
115     
116         /**
117          * Set the non-null param set from given BitSet.
118          *
119          * @param nonNullSet
120          *            BitSet indicating which parameters are non-null
121          */
122         public void setParamsWithProperty(BitSet nonNullSet) {
123             for (int i = 0; i < 32; ++i) {
124                 setParamWithProperty(i, nonNullSet.get(i));
125             }
126         }
127     
128         /**
129          * Set whether or not a parameter might be non-null.
130          *
131          * @param param
132          *            the parameter index
133          * @param hasProperty
134          *            true if the parameter might be non-null, false otherwise
135          */
136         public void setParamWithProperty(int param, boolean hasProperty) {
137             if (param < 0 || param > 31) {
138                 return;
139             }
140             if (hasProperty) {
141                 bits |= (1 << param);
142             } else {
143                 bits &= ~(1 << param);
144             }
145         }
146     
147         /**
148          * Return whether or not a parameter might be non-null.
149          *
150          * @param param
151          *            the parameter index
152          * @return true if the parameter might be non-null, false otherwise
153          */
154         public boolean hasProperty(int param) {
155             if (param < 0 || param > 31) {
156                 return false;
157             } else {
158                 return (bits & (1 << param)) != 0;
159             }
160         }
161     
162         /**
163          * Given a bitset of null arguments passed to the method represented by this
164          * property, return a bitset indicating which null arguments correspond to
165          * an non-null param.
166          *
167          * @param nullArgSet
168          *            bitset of null arguments
169          * @return bitset intersecting null arguments and non-null params
170          */
171         public BitSet getMatchingParameters(BitSet nullArgSet) {
172             BitSet result = new BitSet();
173             for (int i = 0; i < 32; ++i) {
174                 result.set(i, nullArgSet.get(i) && hasProperty(i));
175             }
176             return result;
177         }
178     
179         public BitSet getAsBitSet() {
180             BitSet result = new BitSet();
181             if (isEmpty()) {
182                 return result;
183             }
184             for (int i = 0; i < 32; ++i) {
185                 result.set(i, hasProperty(i));
186             }
187             return result;
188         }
189     
190         /**
191          * Return whether or not the set of non-null parameters is empty.
192          *
193          * @return true if the set is empty, false if it contains at least one
194          *         parameter
195          */
196         public boolean isEmpty() {
197             return bits == 0;
198         }
199     
200         @Override
201         public String toString() {
202             StringBuilder buf = new StringBuilder();
203     
204             buf.append('{');
205             for (int i = 0; i < 32; ++i) {
206                 if (hasProperty(i)) {
207                     if (buf.length() > 1) {
208                         buf.append(',');
209                     }
210                     buf.append(i);
211                 }
212             }
213             buf.append('}');
214     
215             return buf.toString();
216         }
217     
218         /**
219          * Intersect this set with the given set. Useful for summarizing the
220          * properties of multiple methods.
221          *
222          * @param targetDerefParamSet
223          *            another set
224          */
225         public void intersectWith(ParameterProperty targetDerefParamSet) {
226             bits &= targetDerefParamSet.bits;
227         }
228     
229         /**
230          * Make this object the same as the given one.
231          *
232          * @param other
233          *            another ParameterNullnessProperty
234          */
235         public void copyFrom(ParameterProperty other) {
236             this.bits = other.bits;
237         }
238     }
