29     import edu.umd.cs.findbugs.classfile.ICodeBaseEntry;
30     import edu.umd.cs.findbugs.classfile.MissingClassException;

33     import edu.umd.cs.findbugs.classfile.analysis.ClassData;

35     import edu.umd.cs.findbugs.io.IO;
36     
37     /**
38      * Analysis engine to produce the data (bytes) of a class.
39      *
40      * @author David Hovemeyer
41      */
42     public class ClassDataAnalysisEngine extends RecomputableClassAnalysisEngine<ClassData> {
43     
44         @Override
45         public ClassData analyze(IAnalysisCache analysisCache, ClassDescriptor descriptor) throws CheckedAnalysisException {
46     
47             // Compute the resource name
48             String resourceName = descriptor.toResourceName();
49     
50             // Look up the codebase entry for the resource
51             ICodeBaseEntry codeBaseEntry;
52             try {
53                 codeBaseEntry = analysisCache.getClassPath().lookupResource(resourceName);
54             } catch (ResourceNotFoundException e) {
55                 // Allow loading javax.annotation's from our own classpath - in case we analyze projects
56                 // using 3rd party nullness annotations (which do not have jsr305 classes on classpath)
57                 if(resourceName.startsWith("javax/annotation/")){
58                     codeBaseEntry = new VirtualCodeBaseEntry(descriptor);
59                 } else {
60                     throw new MissingClassException(descriptor, e);
61                 }
62             }
63     
64             byte[] data;
65             if (codeBaseEntry instanceof ZipInputStreamCodeBaseEntry) {
66                 data = ((ZipInputStreamCodeBaseEntry) codeBaseEntry).getBytes();
67             } else {
68                 try {
69                     // Create a ByteArrayOutputStream to capture the class data
70                     int length = codeBaseEntry.getNumBytes();
71                     InputStream in = codeBaseEntry.openResource();
72                     if (length >= 0) {
73                         data = IO.readAll(in, length);
74                     } else {
75                         data = IO.readAll(in);
76                     }
77     
78                 } catch (IOException e) {
79                     throw new MissingClassException(descriptor, e);
80                 }
81             }
82             return new ClassData(descriptor, codeBaseEntry, data);
83         }
84     
85         @Override
86         public void registerWith(IAnalysisCache analysisCache) {
87             analysisCache.registerClassAnalysisEngine(ClassData.class, this);
88         }
89     
90         private static class VirtualCodeBaseEntry implements ICodeBaseEntry {
91     
92             private final ClassDescriptor descriptor;
93     
94             public VirtualCodeBaseEntry(ClassDescriptor descriptor) {
95                 this.descriptor = descriptor;
96             }
97     
98             @Override
99             public String getResourceName() {
100                 return descriptor.toResourceName();
101             }
102     
103             @Override
104             public int getNumBytes() {
105                 return -1;
106             }
107     
108             @Override
109             public InputStream openResource() throws IOException {
110                 InputStream stream = getClass().getClassLoader().getResourceAsStream(descriptor.toResourceName());
111                 if(stream == null){
112                     throw new IOException("Can not load '" + descriptor.toResourceName() + "' from FindBugs classpath.");
113                 }
114                 return stream;
115             }
116     
117             @Override
118             public ICodeBase getCodeBase() {
119                 return null;
120             }
121     
122             @Override
123             public ClassDescriptor getClassDescriptor() {
124                 return descriptor;
125             }
126     
127             @Override
128             public void overrideResourceName(String resourceName) {
129                 //
130             }
131     
132         }
133     }
