36     /**
37      * Compute the union of two sets of bug results, preserving annotations.
38      */
39     public class UnionResults {
40     
41         static class UnionResultsCommandLine extends CommandLine {
42             public String outputFile;
43     
44             boolean withMessages;
45     
46             UnionResultsCommandLine() {
47                 addSwitch("-withMessages", "Generated XML should contain msgs for external processing");
48                 addOption("-output", "outputFile", "File in which to store combined results");
49             }
50     
51             /*
52              * (non-Javadoc)
53              *
54              * @see
55              * edu.umd.cs.findbugs.config.CommandLine#handleOption(java.lang.String,
56              * java.lang.String)
57              */
58             @Override
59             protected void handleOption(String option, String optionExtraPart) throws IOException {
60                 if ("-withMessages".equals(option)) {
61                     withMessages = true;
62                 } else {
63                     throw new IllegalArgumentException("Unknown option : " + option);
64                 }
65             }
66     
67             /*
68              * (non-Javadoc)
69              *
70              * @see
71              * edu.umd.cs.findbugs.config.CommandLine#handleOptionWithArgument(java
72              * .lang.String, java.lang.String)
73              */
74             @Override
75             protected void handleOptionWithArgument(String option, String argument) throws IOException {
76                 if ("-output".equals(option)) {
77                     outputFile = argument;
78                 } else {
79                     throw new IllegalArgumentException("Unknown option : " + option);
80                 }
81             }
82     
83         }
84     
85         static {
86             DetectorFactoryCollection.instance(); // as a side effect, loads
87             // detector plugins
88         }
89     
90         static public SortedBugCollection union(SortedBugCollection origCollection, SortedBugCollection newCollection) {
91             SortedBugCollection result = origCollection.duplicate();
92             merge(null, result, newCollection);
93             return result;
94         }
95         static public void merge(HashSet<String> hashes, SortedBugCollection into, SortedBugCollection from) {
96     
97             for (BugInstance bugInstance : from.getCollection()) {
98                 if (hashes == null || hashes.add(bugInstance.getInstanceHash())) {
99                     into.add(bugInstance);
100                 }
101             }
102             ProjectStats stats = into.getProjectStats();
103             ProjectStats stats2 = from.getProjectStats();
104             stats.addStats(stats2);
105     
106             Project project = into.getProject();
107             Project project2 = from.getProject();
108             project.add(project2);
109     
110             for(AnalysisError error : from.getErrors()) {
111                 into.addError(error);
112             }
113     
114             return;
115         }
116     
117         public static void main(String[] argv) throws IOException {
118     
119             FindBugs.setNoAnalysis();
120             final UnionResultsCommandLine commandLine = new UnionResultsCommandLine();
121     
122             int argCount = commandLine.parse(argv, 2, Integer.MAX_VALUE, "Usage: " + UnionResults.class.getName()
123                     + " [options] [<results1> <results2> ... <resultsn>] ");
124     
125             SortedBugCollection results = null;
126             HashSet<String> hashes = new HashSet<String>();
127     
128             for (int i = argCount; i < argv.length; i++) {
129                 try {
130                     SortedBugCollection more = new SortedBugCollection();
131     
132                     more.readXML(argv[i]);
133                     if (results == null) {
134                         results = more.createEmptyCollectionWithMetadata();
135                     }
136     
137                     merge(hashes, results, more);
138     
139                 } catch (IOException e) {
140                     System.err.println("Trouble reading/parsing " + argv[i]);
141                 } catch (DocumentException e) {
142                     System.err.println("Trouble reading/parsing " + argv[i]);
143                 }
144             }
145     
146             if (results == null) {
147                 System.err.println("No files successfully read");
148                 System.exit(1);
149                 return;
150             }
151             results.setWithMessages(commandLine.withMessages);
152             if (commandLine.outputFile == null) {
153                 results.writeXML(System.out);
154             } else {
155                 results.writeXML(commandLine.outputFile);
156             }
157         }
158     
159     }
