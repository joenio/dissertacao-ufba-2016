92         /*
93          * (non-Javadoc)
94          *
95          * @see
96          * edu.umd.cs.findbugs.ba.AbstractFrameModelingVisitor#analyzeInstruction
97          * (org.apache.bcel.generic.Instruction)
98          */
99         @Override
100         public void analyzeInstruction(Instruction ins) throws DataflowAnalysisException {
101             if (!getFrame().isValid()) {
102                 return;
103             }
104             slotContainingNewNullValue = -1;
105             super.analyzeInstruction(ins);
106             if (!getFrame().isValid()) {
107                 return;
108             }
109     
110             if (!NO_ASSERT_HACK) {
111                 if (assertionMethods.isAssertionHandle(getLocation().getHandle(), cpg)) {
112                     IsNullValueFrame frame = getFrame();
113                     for (int i = 0; i < frame.getNumSlots(); ++i) {
114                         IsNullValue value = frame.getValue(i);
115                         if (value.isDefinitelyNull() || value.isNullOnSomePath()) {
116                             frame.setValue(i, IsNullValue.nonReportingNotNullValue());
117                         }
118                     }
119                     for (Map.Entry<ValueNumber, IsNullValue> e : frame.getKnownValueMapEntrySet()) {
120                         IsNullValue value = e.getValue();
121                         if (value.isDefinitelyNull() || value.isNullOnSomePath()) {
122                             e.setValue(IsNullValue.nonReportingNotNullValue());
123                         }
124     
125                     }
126                 }
127             }
128     
129         }

144         // Overrides of specific instruction visitor methods.
145         // ACONST_NULL obviously produces a value that is DEFINITELY NULL.
146         // LDC produces values that are NOT NULL.
147         // NEW produces values that are NOT NULL.
148     
149         // Note that all instructions that have an implicit null
150         // check (field access, invoke, etc.) are handled in IsNullValueAnalysis,
151         // because handling them relies on control flow (the existence of
152         // an ETB and exception edge prior to the block containing the
153         // instruction with the null check.)
154     
155         // Note that we don't override IFNULL and IFNONNULL.
156         // Those are handled in the analysis itself, because we need
157         // to produce different values in each of the control successors.
158     
159         private void produce(IsNullValue value) {
160             IsNullValueFrame frame = getFrame();
161             frame.pushValue(value);
162             newValueOnTOS();
163         }
164     
165         private void produce2(IsNullValue value) {
166             IsNullValueFrame frame = getFrame();
167             frame.pushValue(value);
168             frame.pushValue(value);
169         }

171         /**
172          * Handle method invocations. Generally, we want to get rid of null
173          * information following a call to a likely exception thrower or assertion.
174          */
175         private void handleInvoke(InvokeInstruction obj) {
176             Type returnType = obj.getReturnType(getCPG());
177     
178             Location location = getLocation();
179     
180             if (trackValueNumbers) {
181                 try {
182                     ValueNumberFrame vnaFrame = vnaDataflow.getFactAtLocation(location);
183                     Set<ValueNumber> nonnullParameters = UnconditionalValueDerefAnalysis.checkAllNonNullParams(location, vnaFrame,
184                             cpg, null, null, typeDataflow);
185     
186                     if (!nonnullParameters.isEmpty()) {
187                         IsNullValue kaboom = IsNullValue.noKaboomNonNullValue(location);
188                         IsNullValueFrame frame = getFrame();
189                         for (ValueNumber vn : nonnullParameters) {
190                             IsNullValue knownValue = frame.getKnownValue(vn);
191                             if (knownValue != null && !knownValue.isDefinitelyNotNull()) {
192                                 if (knownValue.isDefinitelyNull()) {
193                                     frame.setTop();
194                                     return;
195                                 }
196                                 frame.setKnownValue(vn, kaboom);
197                             }
198                             for (int i = 0; i < vnaFrame.getNumSlots(); i++) {
199                                 IsNullValue value = frame.getValue(i);
200                                 if (vnaFrame.getValue(i).equals(vn) && !value.isDefinitelyNotNull()) {
201                                     frame.setValue(i, kaboom);
202                                     if (value.isDefinitelyNull()) {
203                                         frame.setTop();
204                                         return;
205                                     }
206                                 }
207                             }
208                         }
209     
210                     }
211                 } catch (DataflowAnalysisException e) {
212                     AnalysisContext.logError("Error looking up nonnull parameters for invoked method", e);
213                 }
214             }
215             // Determine if we are going to model the return value of this call.
216             boolean modelCallReturnValue = MODEL_NONNULL_RETURN && returnType instanceof ReferenceType;
217     
218             if (!modelCallReturnValue) {
219                 // Normal case: Assume returned values are non-reporting non-null.
220                 handleNormalInstruction(obj);
221             } else {
222                 // Special case: some special value is pushed on the stack for the
223                 // return value
224                 IsNullValue result = null;
225                 TypeFrame typeFrame;
226                 try {
227                     typeFrame = typeDataflow.getFactAtLocation(location);
228     
229                     Set<XMethod> targetSet = Hierarchy2.resolveMethodCallTargets(obj, typeFrame, cpg);
230     
231                     if (targetSet.isEmpty()) {
232                         XMethod calledMethod = XFactory.createXMethod(obj, getCPG());
233                         result = getReturnValueNullness(calledMethod);
234                     } else {
235                         for (XMethod calledMethod : targetSet) {
236                             IsNullValue pushValue = getReturnValueNullness(calledMethod);
237                             if (result == null) {
238                                 result = pushValue;
239                             } else {
240                                 result = IsNullValue.merge(result, pushValue);
241                             }
242                         }
243                     }
244                 } catch (DataflowAnalysisException e) {
245                     result = IsNullValue.nonReportingNotNullValue();
246                 } catch (ClassNotFoundException e) {
247                     result = IsNullValue.nonReportingNotNullValue();
248                 }
249                 modelInstruction(obj, getNumWordsConsumed(obj), getNumWordsProduced(obj), result);
250                 newValueOnTOS();
251             }
252     
253         }

255         public IsNullValue getReturnValueNullness(XMethod calledMethod) {
256             IsNullValue pushValue;
257             if (IsNullValueAnalysis.DEBUG) {
258                 System.out.println("Check " + calledMethod + " for null return...");
259             }
260             NullnessAnnotation annotation = AnalysisContext.currentAnalysisContext().getNullnessAnnotationDatabase()
261                     .getResolvedAnnotation(calledMethod, false);
262             Boolean alwaysNonNull = AnalysisContext.currentAnalysisContext().getReturnValueNullnessPropertyDatabase()
263                     .getProperty(calledMethod.getMethodDescriptor());
264             if (annotation == NullnessAnnotation.CHECK_FOR_NULL) {
265                 if (IsNullValueAnalysis.DEBUG) {
266                     System.out.println("Null value returned from " + calledMethod);
267                 }
268                 pushValue = IsNullValue.nullOnSimplePathValue().markInformationAsComingFromReturnValueOfMethod(
269                         calledMethod);
270             } else if (annotation == NullnessAnnotation.NULLABLE) {
271                 pushValue = IsNullValue.nonReportingNotNullValue();
272             } else if (annotation == NullnessAnnotation.NONNULL
273                     || (alwaysNonNull != null && alwaysNonNull.booleanValue())) {
274                 // Method is declared NOT to return null
275                 if (IsNullValueAnalysis.DEBUG) {
276                     System.out.println("NonNull value return from " + calledMethod);
277                 }
278                 pushValue = IsNullValue.nonNullValue().markInformationAsComingFromReturnValueOfMethod(calledMethod);
279     
280             } else {
281                 pushValue = IsNullValue.nonReportingNotNullValue();
282             }
283             return pushValue;
284         }

286         /**
287          * Hook indicating that a new (possibly-null) value is on the top of the
288          * stack.
289          */
290         private void newValueOnTOS() {
291             IsNullValueFrame frame = getFrame();
292             if (frame.getStackDepth() < 1) {
293                 return;
294             }
295             int tosSlot = frame.getNumSlots() - 1;
296             IsNullValue tos = frame.getValue(tosSlot);
297             if (tos.isDefinitelyNull()) {
298                 slotContainingNewNullValue = tosSlot;
299             }
300             if (trackValueNumbers) {
301                 try {
302                     ValueNumberFrame vnaFrameAfter = vnaDataflow.getFactAfterLocation(getLocation());
303                     if (vnaFrameAfter.isValid()) {
304                         ValueNumber tosVN = vnaFrameAfter.getTopValue();
305                         getFrame().setKnownValue(tosVN, tos);
306                     }
307                 } catch (DataflowAnalysisException e) {
308                     AnalysisContext.logError("error", e);
309                 }
310             }
311         }

313         @Override
314         public void visitPUTFIELD(PUTFIELD obj) {
315             if (getNumWordsConsumed(obj) != 2) {
316                 super.visitPUTFIELD(obj);
317                 return;
318             }
319     
320             IsNullValue nullValueStored = null;
321             try {
322                 nullValueStored = getFrame().getTopValue();
323             } catch (DataflowAnalysisException e1) {
324                 AnalysisContext.logError("Oops", e1);
325             }
326             super.visitPUTFIELD(obj);
327             XField field = XFactory.createXField(obj, cpg);
328             if (nullValueStored != null && ValueNumberAnalysisFeatures.REDUNDANT_LOAD_ELIMINATION) {
329                 try {
330                     ValueNumberFrame vnaFrameBefore = vnaDataflow.getFactAtLocation(getLocation());
331                     ValueNumber refValue = vnaFrameBefore.getStackValue(1);
332                     AvailableLoad load = new AvailableLoad(refValue, field);
333                     ValueNumberFrame vnaFrameAfter = vnaDataflow.getFactAfterLocation(getLocation());
334                     ValueNumber[] newValueNumbersForField = vnaFrameAfter.getAvailableLoad(load);
335                     if (newValueNumbersForField != null && trackValueNumbers) {
336                         for (ValueNumber v : newValueNumbersForField) {
337                             getFrame().setKnownValue(v, nullValueStored);
338                         }
339                     }
340                 } catch (DataflowAnalysisException e) {
341                     AnalysisContext.logError("Oops", e);
342                 }
343             }
344         }

346         @Override
347         public void visitGETFIELD(GETFIELD obj) {
348             if (getNumWordsProduced(obj) != 1) {
349                 super.visitGETFIELD(obj);
350                 return;
351             }
352     
353             if (checkForKnownValue(obj)) {
354                 return;
355             }
356     
357             XField field = XFactory.createXField(obj, cpg);
358     
359             NullnessAnnotation annotation = AnalysisContext.currentAnalysisContext().getNullnessAnnotationDatabase()
360                     .getResolvedAnnotation(field, false);
361             if (annotation == NullnessAnnotation.NONNULL) {
362                 modelNormalInstruction(obj, getNumWordsConsumed(obj), 0);
363                 produce(IsNullValue.nonNullValue());
364             } else if (annotation == NullnessAnnotation.CHECK_FOR_NULL) {
365                 modelNormalInstruction(obj, getNumWordsConsumed(obj), 0);
366                 produce(IsNullValue.nullOnSimplePathValue().markInformationAsComingFromFieldValue(field));
367             } else {
368     
369                 super.visitGETFIELD(obj);
370             }
371     
372         }

374         /*
375          * (non-Javadoc)
376          *
377          * @see
378          * edu.umd.cs.findbugs.ba.AbstractFrameModelingVisitor#visitGETSTATIC(org
379          * .apache.bcel.generic.GETSTATIC)
380          */
381         @Override
382         public void visitGETSTATIC(GETSTATIC obj) {
383             if (getNumWordsProduced(obj) != 1) {
384                 super.visitGETSTATIC(obj);
385                 return;
386             }
387     
388             if (checkForKnownValue(obj)) {
389                 return;
390             }
391             XField field = XFactory.createXField(obj, cpg);
392             if (field.isFinal()) {
393                 Item summary = AnalysisContext.currentAnalysisContext().getFieldSummary().getSummary(field);
394                 if (summary.isNull()) {
395                     produce(IsNullValue.nullValue());
396                     return;
397                 }
398             }
399             if ("java.util.logging.Level".equals(field.getClassName()) && "SEVERE".equals(field.getName())
400                     || "org.apache.log4j.Level".equals(field.getClassName())
401                     && ("ERROR".equals(field.getName()) || "FATAL".equals(field.getName()))) {
402                 getFrame().toExceptionValues();
403             }
404     
405             if (field.getName().startsWith("class$")) {
406                 produce(IsNullValue.nonNullValue());
407                 return;
408             }
409             NullnessAnnotation annotation = AnalysisContext.currentAnalysisContext().getNullnessAnnotationDatabase()
410                     .getResolvedAnnotation(field, false);
411             if (annotation == NullnessAnnotation.NONNULL) {
412                 modelNormalInstruction(obj, getNumWordsConsumed(obj), 0);
413                 produce(IsNullValue.nonNullValue());
414             } else if (annotation == NullnessAnnotation.CHECK_FOR_NULL) {
415                 modelNormalInstruction(obj, getNumWordsConsumed(obj), 0);
416                 produce(IsNullValue.nullOnSimplePathValue().markInformationAsComingFromFieldValue(field));
417             } else {
418     
419                 super.visitGETSTATIC(obj);
420             }
421         }

423         /**
424          * Check given Instruction to see if it produces a known value. If so, model
425          * the instruction and return true. Otherwise, do nothing and return false.
426          * Should only be used for instructions that produce a single value on the
427          * top of the stack.
428          *
429          * @param obj
430          *            the Instruction the instruction
431          * @return true if the instruction produced a known value and was modeled,
432          *         false otherwise
433          */
434         private boolean checkForKnownValue(Instruction obj) {
435             if (trackValueNumbers) {
436                 try {
437                     // See if the value number loaded here is a known value
438                     ValueNumberFrame vnaFrameAfter = vnaDataflow.getFactAfterLocation(getLocation());
439                     if (vnaFrameAfter.isValid()) {
440                         ValueNumber tosVN = vnaFrameAfter.getTopValue();
441                         IsNullValue knownValue = getFrame().getKnownValue(tosVN);
442                         if (knownValue != null) {
443                             // System.out.println("Produce known value!");
444                             // The value produced by this instruction is known.
445                             // Push the known value.
446                             modelNormalInstruction(obj, getNumWordsConsumed(obj), 0);
447                             produce(knownValue);
448                             return true;
449                         }
450                     }
451                 } catch (DataflowAnalysisException e) {
452                     // Ignore...
453                 }
454             }
455             return false;
456         }
