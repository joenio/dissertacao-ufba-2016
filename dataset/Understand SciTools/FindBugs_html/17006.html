20     package edu.umd.cs.findbugs.sourceViewer;
21     
22     import java.text.CharacterIterator;
23     
24     import javax.swing.text.BadLocationException;
25     import javax.swing.text.Document;
26     import javax.swing.text.Segment;
27     
28     /**
29      * A CharacterIterator over a Document. Only a partial implementation.
30      */
31     public class DocumentCharacterIterator implements CharacterIterator {
32     
33         private final Document doc;
34     
35         private final Segment text;
36     
37         /** Position of iterator in document. */
38         private int docPos = 0;
39     
40         /** Index of end of current segment in document. */
41         private int segmentEnd;
42     
43         DocumentCharacterIterator(Document doc) {
44             this.doc = doc;
45             text = new Segment();
46             text.setPartialReturn(true);
47     
48             try {
49                 doc.getText(0, doc.getLength(), text);
50             } catch (BadLocationException e) {
51                 throw new RuntimeException(e);
52             }
53             segmentEnd = text.count;
54         }
55     
56         @Override
57         public Object clone() {
58             throw new UnsupportedOperationException();
59         }
60     
61         @Override
62         public char current() {
63             return text.current();
64         }
65     
66         @Override
67         public char first() {
68             throw new UnsupportedOperationException();
69         }
70     
71         @Override
72         public int getBeginIndex() {
73             throw new UnsupportedOperationException();
74         }
75     
76         @Override
77         public int getEndIndex() {
78             throw new UnsupportedOperationException();
79         }
80     
81         @Override
82         public int getIndex() {
83             return docPos;
84         }
85     
86         @Override
87         public char last() {
88             throw new UnsupportedOperationException();
89         }
90     
91         /**
92          * Increments the iterator's index by one and returns the character at the
93          * new index.
94          *
95          * @return the character at the new position, or DONE if the new position is
96          *         off the end
97          */
98         @Override
99         public char next() {
100             ++docPos;
101             if (docPos < segmentEnd || segmentEnd >= doc.getLength()) {
102                 return text.next();
103             }
104             try {
105                 doc.getText(segmentEnd, doc.getLength() - segmentEnd, text);
106             } catch (BadLocationException e) {
107                 throw new RuntimeException(e);
108             }
109             segmentEnd += text.count;
110             return text.current();
111         }
112     
113         @Override
114         public char previous() {
115             throw new UnsupportedOperationException();
116         }
117     
118         @Override
119         public char setIndex(int position) {
120             throw new UnsupportedOperationException();
121         }
122     
123     }
