42         @Override
43         public void visitClassContext(ClassContext classContext) {
44             Method[] methodList = classContext.getJavaClass().getMethods();
45     
46             for (Method method : methodList) {
47                 if (method.getCode() == null) {
48                     continue;
49                 }
50     
51                 try {
52                     analyzeMethod(classContext, method);
53                 } catch (CFGBuilderException e) {
54                     bugReporter.logError("Detector " + this.getClass().getName() + " caught exception", e);
55                 } catch (DataflowAnalysisException e) {
56                     // bugReporter.logError("Detector " + this.getClass().getName()
57                     // + " caught exception", e);
58                 }
59             }
60         }

62         private void analyzeMethod(ClassContext classContext, Method method) throws CFGBuilderException, DataflowAnalysisException {
63             MethodGen methodGen = classContext.getMethodGen(method);
64             if (methodGen == null) {
65                 return;
66             }
67             BitSet bytecodeSet = classContext.getBytecodeSet(method);
68             if (bytecodeSet == null) {
69                 return;
70             }
71             // We don't adequately model instanceof interfaces yet
72             if (bytecodeSet.get(Constants.INSTANCEOF) || bytecodeSet.get(Constants.CHECKCAST)) {
73                 return;
74             }
75             CFG cfg = classContext.getCFG(method);
76             TypeDataflow typeDataflow = classContext.getTypeDataflow(method);
77             ConstantPoolGen cpg = classContext.getConstantPoolGen();
78     
79             String sourceFile = classContext.getJavaClass().getSourceFileName();
80             if (DEBUG) {
81                 String methodName = methodGen.getClassName() + "." + methodGen.getName();
82                 System.out.println("Checking " + methodName);
83             }
84     
85             for (Iterator<Location> i = cfg.locationIterator(); i.hasNext();) {
86                 Location location = i.next();
87                 InstructionHandle handle = location.getHandle();
88                 int pc = handle.getPosition();
89                 Instruction ins = handle.getInstruction();
90     
91                 if (!(ins instanceof InvokeInstruction)) {
92                     continue;
93                 }
94     
95                 InvokeInstruction invoke = (InvokeInstruction) ins;
96                 String mName = invoke.getMethodName(cpg);
97                 if (!"writeObject".equals(mName)) {
98                     continue;
99                 }
100                 String cName = invoke.getClassName(cpg);
101                 if (!"java.io.ObjectOutput".equals(cName) && !"java.io.ObjectOutputStream".equals(cName)) {
102                     continue;
103                 }
104     
105                 TypeFrame frame = typeDataflow.getFactAtLocation(location);
106                 if (!frame.isValid()) {
107                     // This basic block is probably dead
108                     continue;
109                 }
110                 Type operandType = frame.getTopValue();
111     
112                 if (operandType.equals(TopType.instance())) {
113                     // unreachable
114                     continue;
115                 }
116                 if (!(operandType instanceof ReferenceType)) {
117                     // Shouldn't happen - illegal bytecode
118                     continue;
119                 }
120                 ReferenceType refType = (ReferenceType) operandType;
121     
122                 if (refType.equals(NullType.instance())) {
123                     continue;
124                 }
125     
126                 try {
127     
128                     double isSerializable = DeepSubtypeAnalysis.isDeepSerializable(refType);
129     
130                     if (isSerializable >= 0.9) {
131                         continue;
132                     }
133     
134                     ReferenceType problem = DeepSubtypeAnalysis.getLeastSerializableTypeComponent(refType);
135     
136                     double isRemote = DeepSubtypeAnalysis.isDeepRemote(refType);
137                     if (isRemote >= 0.9) {
138                         continue;
139                     }
140                     if (isSerializable < isRemote) {
141                         isSerializable = isRemote;
142                     }
143     
144     
145                     SourceLineAnnotation sourceLineAnnotation = SourceLineAnnotation.fromVisitedInstruction(classContext,
146                             methodGen, sourceFile, handle);
147     
148                     bugReporter.reportBug(new BugInstance(this, "DMI_NONSERIALIZABLE_OBJECT_WRITTEN",
149                             isSerializable < 0.15 ? HIGH_PRIORITY : isSerializable > 0.5 ? LOW_PRIORITY : NORMAL_PRIORITY)
150                     .addClassAndMethod(methodGen, sourceFile).addType(problem).describe(TypeAnnotation.FOUND_ROLE)
151                     .addSourceLine(sourceLineAnnotation));
152     
153                 } catch (ClassNotFoundException e) {
154                     // ignore
155                 }
156             }
157         }
