20     package edu.umd.cs.findbugs.classfile;
21     
22     /**
23      * Interface for objects that log various kinds of analysis errors.
24      *
25      * @author David Hovemeyer
26      */
27     public interface IErrorLogger {
28         /**
29          * Called to report a class lookup failure.
30          *
31          * @param ex
32          *            a ClassNotFoundException resulting from the class lookup
33          *            failure
34          */
35         public void reportMissingClass(ClassNotFoundException ex);
36     
37         /**
38          * Called to report a class lookup failure.
39          *
40          * @param classDescriptor
41          *            ClassDescriptor of a missing class
42          */
43         public void reportMissingClass(ClassDescriptor classDescriptor);
44     
45         /**
46          * Log an error that occurs while performing analysis.
47          *
48          * @param message
49          *            the error message
50          */
51         public void logError(String message);
52     
53         /**
54          * Log an error that occurs while performing analysis.
55          *
56          * @param message
57          *            the error message
58          * @param e
59          *            the exception which is the underlying cause of the error
60          */
61         public void logError(String message, Throwable e);
62     
63         /**
64          * Report that we skipped some analysis of a method
65          *
66          * @param method
67          *            the method we skipped
68          */
69         public void reportSkippedAnalysis(MethodDescriptor method);
70     }
