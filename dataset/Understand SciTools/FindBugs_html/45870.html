70         public static byte[] readAll(@WillClose InputStream in) throws IOException {
71             try {
72                 ByteArrayOutputStream byteSink = new ByteArrayOutputStream();
73                 copy(in, byteSink);
74                 return byteSink.toByteArray();
75             } finally {
76                 close(in);
77             }
78         }

80         static byte[] copyOf(byte[] original, int newLength) {
81             byte[] copy = new byte[newLength];
82             System.arraycopy(original, 0, copy, 0,
83                     Math.min(original.length, newLength));
84             return copy;
85         }

88         public static byte[] readAll(@WillClose InputStream in, int size) throws IOException {
89             try {
90                 if (size == 0) {
91                     throw new IllegalArgumentException();
92                 }
93                 byte[] result = new byte[size];
94                 int pos = 0;
95                 while (true) {
96                     int sz;
97                     while ((sz = in.read(result, pos, size - pos)) > 0) {
98                         pos += sz;
99                     }
100     
101                     if (pos < size) {
102                         return copyOf(result, pos);
103                     }
104                     int nextByte = in.read();
105                     if (nextByte == -1) {
106                         return result;
107                     }
108                     size = size * 2 + 500;
109                     result = copyOf(result, size);
110                     result[pos++] = (byte) nextByte;
111                 }
112             } finally {
113                 close(in);
114             }
115         }

132         public static long copy(@WillNotClose InputStream in, @WillNotClose OutputStream out, long maxBytes)
133     
134                 throws IOException {
135             long total = 0;
136     
137             int sz = 0;
138     
139             byte buf[] = myByteBuf.get();
140     
141             while (maxBytes > 0 && (sz = in.read(buf, 0, (int) Math.min(maxBytes, buf.length))) > 0) {
142                 total += sz;
143                 maxBytes -= sz;
144                 out.write(buf, 0, sz);
145             }
146     
147             return total;
148         }

221         /**
222          * Provide a skip fully method. Either skips the requested number of bytes
223          * or throws an IOException;
224          *
225          * @param in
226          *            The input stream on which to perform the skip
227          * @param bytes
228          *            Number of bytes to skip
229          * @throws EOFException
230          *             if we reach EOF and still need to skip more bytes
231          * @throws IOException
232          *             if in.skip throws an IOException
233          */
234         public static void skipFully(InputStream in, long bytes) throws IOException {
235             if (bytes < 0) {
236                 throw new IllegalArgumentException("Can't skip " + bytes + " bytes");
237             }
238             long remaining = bytes;
239             while (remaining > 0) {
240                 long skipped = in.skip(remaining);
241                 if (skipped <= 0) {
242                     throw new EOFException("Reached EOF while trying to skip a total of " + bytes);
243                 }
244                 remaining -= skipped;
245             }
246     
247         }

249         public static boolean verifyURL(URL u) {
250             if (u == null) {
251                 return false;
252             }
253     
254             InputStream i = null;
255     
256             try {
257                 URLConnection uc = u.openConnection();
258                 i = uc.getInputStream();
259                 int firstByte = i.read();
260                 i.close();
261                 return firstByte >= 0;
262             }
263     
264             catch (Exception e) {
265                 return false;
266             } finally {
267                 Util.closeSilently(i);
268             }
269     
270         }
