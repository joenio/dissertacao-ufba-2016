46         protected AbstractClassMember(@DottedClassName String className, String name, String signature, int accessFlags) {
47             if (className.indexOf('.') >= 0) {
48                 // className = className.replace('.','/');
49                 //            dottedCountClass++;
50             } else if (className.indexOf('/') >= 0) {
51                 assert false;
52                 //            slashCountClass++;
53                 className = className.replace('/', '.');
54             }
55     
56             if (signature.indexOf('.') >= 0) {
57                 assert false;
58                 signature = signature.replace('.', '/');
59                 //            dottedCountSignature++;
60             }
61             // else if (signature.indexOf('/') >= 0) {
62             //     slashCountSignature++;
63             // }
64             this.className = DescriptorFactory.canonicalizeString(className);
65             this.name = DescriptorFactory.canonicalizeString(name);
66             this.signature = DescriptorFactory.canonicalizeString(signature);
67             this.accessFlags = accessFlags;
68         }
69     
70         @Override
71         @DottedClassName
72         public String getClassName() {
73             return className;
74         }
75     
76         @Override
77         public ClassDescriptor getClassDescriptor() {
78             return DescriptorFactory.instance().getClassDescriptorForDottedClassName(className);
79         }
80     
81         @Override
82         public String getName() {
83             return name;
84         }
85     
86         @Override
87         @DottedClassName
88         public String getPackageName() {
89             int lastDot = className.lastIndexOf('.');
90             if (lastDot == -1) {
91                 return className;
92             }
93             return className.substring(0, lastDot);
94         }
95     
96         @Override
97         public String getSignature() {
98             return signature;
99         }
100     
101         public boolean isReferenceType() {
102             return signature.startsWith("L") || signature.startsWith("[");
103         }
104     
105         @Override
106         public int getAccessFlags() {
107             return accessFlags;
108         }
109     
110         @Override
111         public boolean isStatic() {
112             return (accessFlags & Constants.ACC_STATIC) != 0;
113         }
114     
115         @Override
116         public boolean isFinal() {
117             return (accessFlags & Constants.ACC_FINAL) != 0;
118         }
119     
120         @Override
121         public boolean isPublic() {
122             return (accessFlags & Constants.ACC_PUBLIC) != 0;
123         }
124     
125         @Override
126         public boolean isProtected() {
127             return (accessFlags & Constants.ACC_PROTECTED) != 0;
128         }
129     
130         @Override
131         public boolean isPrivate() {
132             return (accessFlags & Constants.ACC_PRIVATE) != 0;
133         }

173         void markAsResolved() {
174             resolved = true;
175         }
176     
177         @Override
178         public int hashCode() {
179             if (cachedHashCode == 0) {
180                 cachedHashCode = className.hashCode() ^ name.hashCode() ^ signature.hashCode();
181             }
182             return cachedHashCode;
183         }
184     
185         @Override
186         public boolean equals(Object o) {
187             if (o == null || this.getClass() != o.getClass()) {
188                 return false;
189             }
190             AbstractClassMember other = (AbstractClassMember) o;
191             return className.equals(other.className) && name.equals(other.name) && signature.equals(other.signature);
192         }
