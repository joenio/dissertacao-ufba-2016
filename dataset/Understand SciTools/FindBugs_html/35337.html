20     package edu.umd.cs.findbugs.ba;
21     
22     import java.io.ByteArrayInputStream;
23     import java.io.ByteArrayOutputStream;
24     import java.io.IOException;
25     import java.io.InputStream;
26     
27     /**
28      * Cached data for a source file. Contains a map of line numbers to byte
29      * offsets, for quick searching of source lines.
30      *
31      * @author David Hovemeyer
32      * @see SourceFinder
33      */
34     public class SourceFile {
35         private static int intValueOf(byte b) {
36             return b & 0xff;
37         }
38     
39         /**
40          * Helper object to build map of line number to byte offset for a source
41          * file.
42          */
43         private static class LineNumberMapBuilder {
44             private final SourceFile sourceFile;
45     
46             private int offset;
47     
48             private int lastSeen;
49     
50             public LineNumberMapBuilder(SourceFile sourceFile) {
51                 this.sourceFile = sourceFile;
52                 this.offset = 0;
53                 this.lastSeen = -1;
54             }
55     
56             public void addData(byte[] data, int len) {
57                 for (int i = 0; i < len; ++i) {
58                     int ch = intValueOf(data[i]);
59                     // if (ch < 0) throw new IllegalStateException();
60                     add(ch);
61                 }
62             }
63     
64             public void eof() {
65                 add(-1);
66             }
67     
68             private void add(int ch) {
69                 switch (ch) {
70                 case '\n':
71                     sourceFile.addLineOffset(offset + 1);
72                     break;
73                 case '\r':
74                     // Need to see next character to know if it's a
75                     // line terminator.
76                     break;
77                 default:
78                     if (lastSeen == '\r') {
79                         // We consider a bare CR to be an end of line
80                         // if it is not followed by a new line.
81                         // Mac OS has historically used a bare CR as
82                         // its line terminator.
83                         sourceFile.addLineOffset(offset);
84                     }
85                 }
86     
87                 lastSeen = ch;
88                 ++offset;
89             }
90         }
91     
92         private static final int DEFAULT_SIZE = 100;
93     
94         private final SourceFileDataSource dataSource;
95     
96         private byte[] data;
97     
98         private int[] lineNumberMap;
99     
100         private int numLines;
101     
102         /**
103          * Constructor.
104          *
105          * @param dataSource
106          *            the SourceFileDataSource object which will provide the data of
107          *            the source file
108          */
109         public SourceFile(SourceFileDataSource dataSource) {
110             this.dataSource = dataSource;
111             this.lineNumberMap = new int[DEFAULT_SIZE];
112             this.numLines = 0;
113         }
114     
115         /**
116          * Get the full path name of the source file (with directory).
117          */
118         public String getFullFileName() {
119             return dataSource.getFullFileName();
120         }
121     
122         /**
123          * Get an InputStream on data.
124          *
125          * @return an InputStream on the data in the source file, starting from
126          *         given offset
127          */
128         public InputStream getInputStream() throws IOException {
129             loadFileData();
130             return new ByteArrayInputStream(data);
131         }
132     
133         /**
134          * Get an InputStream on data starting at given offset.
135          *
136          * @param offset
137          *            the start offset
138          * @return an InputStream on the data in the source file, starting at the
139          *         given offset
140          */
141         public InputStream getInputStreamFromOffset(int offset) throws IOException {
142             loadFileData();
143             return new ByteArrayInputStream(data, offset, data.length - offset);
144         }
145     
146         /**
147          * Add a source line byte offset. This method should be called for each line
148          * in the source file, in order.
149          *
150          * @param offset
151          *            the byte offset of the next source line
152          */
153         public void addLineOffset(int offset) {
154             if (numLines >= lineNumberMap.length) {
155                 // Grow the line number map.
156                 int capacity = lineNumberMap.length * 2;
157                 int[] newLineNumberMap = new int[capacity];
158                 System.arraycopy(lineNumberMap, 0, newLineNumberMap, 0, lineNumberMap.length);
159                 lineNumberMap = newLineNumberMap;
160             }
161     
162             lineNumberMap[numLines++] = offset;
163         }
164     
165         /**
166          * Get the byte offset in the data for a source line. Note that lines are
167          * considered to be zero-index, so the first line in the file is numbered
168          * zero.
169          *
170          * @param line
171          *            the line number
172          * @return the byte offset in the file's data for the line, or -1 if the
173          *         line is not valid
174          */
175         public int getLineOffset(int line) {
176             try {
177                 loadFileData();
178             } catch (IOException e) {
179                 System.err.println("SourceFile.getLineOffset: " + e.getMessage());
180                 return -1;
181             }
182             if (line < 0 || line >= numLines) {
183                 return -1;
184             }
185             return lineNumberMap[line];
186         }
187     
188         private synchronized void loadFileData() throws IOException {
189             if (data != null) {
190                 return;
191             }
192     
193             InputStream in = null;
194     
195             try {
196                 in = dataSource.open();
197                 ByteArrayOutputStream out = new ByteArrayOutputStream();
198     
199                 addLineOffset(0); // Line 0 starts at offset 0
200                 LineNumberMapBuilder mapBuilder = new LineNumberMapBuilder(this);
201     
202                 // Copy all of the data from the file into the byte array output
203                 // stream
204                 byte[] buf = new byte[1024];
205                 int n;
206                 while ((n = in.read(buf)) >= 0) {
207                     mapBuilder.addData(buf, n);
208                     out.write(buf, 0, n);
209                 }
210                 mapBuilder.eof();
211     
212                 setData(out.toByteArray());
213             } finally {
214                 if (in != null) {
215                     in.close();
216                 }
217             }
218     
219         }
220     
221         /**
222          * Set the source file data.
223          *
224          * @param data
225          *            the data
226          */
227         private void setData(byte[] data) {
228             this.data = data;
229         }
230     
231         public long getLastModified() {
232             return dataSource.getLastModified();
233         }
234     }
