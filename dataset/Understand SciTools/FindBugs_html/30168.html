19     package edu.umd.cs.findbugs.filter;
20     
21     import java.io.IOException;
22     
23     import edu.umd.cs.findbugs.BugInstance;
24     import edu.umd.cs.findbugs.xml.XMLAttributeList;
25     import edu.umd.cs.findbugs.xml.XMLOutput;
26     
27     /**
28      * Matcher to select BugInstances with a particular first version.
29      */
30     public class FirstVersionMatcher extends VersionMatcher implements Matcher {
31         public FirstVersionMatcher(String versionAsString, String relOpAsString) {
32             this(Long.parseLong(versionAsString), RelationalOp.byName(relOpAsString));
33         }
34     
35         public FirstVersionMatcher(String versionAsString, RelationalOp relOp) {
36             this(Long.parseLong(versionAsString), relOp);
37         }
38     
39         public FirstVersionMatcher(long version, RelationalOp relOp) {
40             super(version, relOp);
41         }
42     
43         @Override
44         public String toString() {
45             return "FirstVersion(version" + relOp + version + ")";
46         }
47     
48         @Override
49         public boolean match(BugInstance bugInstance) {
50             return relOp.check(bugInstance.getFirstVersion(), version);
51         }
52     
53         @Override
54         public void writeXML(XMLOutput xmlOutput, boolean disabled) throws IOException {
55             XMLAttributeList attributes = new XMLAttributeList().addAttribute("value", Long.toString(version)).addAttribute("relOp",
56                     relOp.getName());
57             if (disabled) {
58                 attributes.addAttribute("disabled", "true");
59             }
60             xmlOutput.openCloseTag("FirstVersion", attributes);
61         }
62     }
