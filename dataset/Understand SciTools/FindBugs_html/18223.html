20     package edu.umd.cs.findbugs.tools.html;
21     
22     import java.io.IOException;
23     import java.util.Collection;
24     import java.util.HashSet;
25     import java.util.Iterator;
26     
27     import edu.umd.cs.findbugs.BugPattern;
28     import edu.umd.cs.findbugs.DetectorFactory;
29     import edu.umd.cs.findbugs.DetectorFactoryCollection;
30     
31     public abstract class PrintBugDescriptions {
32         public void print() throws IOException {
33             // Ensure bug patterns are loaded
34             DetectorFactoryCollection factories = DetectorFactoryCollection.instance();
35     
36             // Find all bug patterns reported by at least one non-disabled detector.
37             Collection<BugPattern> enabledPatternSet = new HashSet<BugPattern>();
38             for (Iterator<DetectorFactory> i = factories.factoryIterator(); i.hasNext();) {
39                 DetectorFactory factory = i.next();
40                 if (isEnabled(factory)) {
41                     enabledPatternSet.addAll(factory.getReportedBugPatterns());
42                 }
43             }
44     
45             prologue();
46     
47             Iterator<BugPattern> i = DetectorFactoryCollection.instance().bugPatternIterator();
48             while (i.hasNext()) {
49                 BugPattern bugPattern = i.next();
50                 if (!enabledPatternSet.contains(bugPattern)) {
51                     continue;
52                 }
53                 emit(bugPattern);
54             }
55     
56             epilogue();
57         }
58     
59         protected boolean isEnabled(DetectorFactory factory) {
60             return factory.isDefaultEnabled();
61         }
62     
63         protected abstract void prologue() throws IOException;
64     
65         protected abstract void emit(BugPattern bugPattern) throws IOException;
66     
67         protected abstract void epilogue() throws IOException;
68     }
