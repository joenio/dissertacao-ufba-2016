20     package edu.umd.cs.findbugs.util;
21     
22     import junit.framework.TestCase;
23     
24     /**
25      * @author pugh
26      */
27     public class ClassNameTest extends TestCase {
28     
29         public void testExtractPackagePrefix() {
30             assertEquals("", ClassName.extractPackagePrefix("org.apache.ant.subpkg.sub2", 0));
31             assertEquals("org", ClassName.extractPackagePrefix("org", 1));
32             assertEquals("org.apache.ant", ClassName.extractPackagePrefix("org.apache.ant.subpkg.sub2", 3));
33             assertEquals("org.apache.ant.subpkg.sub2", ClassName.extractPackagePrefix("org.apache.ant.subpkg.sub2", 5));
34             assertEquals("org.apache.ant.subpkg.sub2", ClassName.extractPackagePrefix("org.apache.ant.subpkg.sub2", 6));
35         }
36     
37         public void testExtractClassName() {
38             assertEquals("java/lang/Integer", ClassName.extractClassName("Ljava/lang/Integer;"));
39             assertEquals("java/lang/Integer", ClassName.extractClassName("[Ljava/lang/Integer;"));
40             assertEquals("java/lang/Integer", ClassName.extractClassName("[[Ljava/lang/Integer;"));
41             assertEquals("java/lang/Integer", ClassName.extractClassName("[[[Ljava/lang/Integer;"));
42             assertEquals("java/lang/Integer", ClassName.extractClassName("java/lang/Integer"));
43         }
44     
45         public void testGetPrimitiveType() {
46             assertEquals("I", ClassName.getPrimitiveType("java/lang/Integer"));
47             assertEquals("F", ClassName.getPrimitiveType("java/lang/Float"));
48             assertEquals("D", ClassName.getPrimitiveType("java/lang/Double"));
49             assertEquals("J", ClassName.getPrimitiveType("java/lang/Long"));
50             assertEquals("B", ClassName.getPrimitiveType("java/lang/Byte"));
51             assertEquals("C", ClassName.getPrimitiveType("java/lang/Character"));
52             assertEquals("S", ClassName.getPrimitiveType("java/lang/Short"));
53             assertEquals("Z", ClassName.getPrimitiveType("java/lang/Boolean"));
54             assertNull(ClassName.getPrimitiveType("java/lang/String"));
55             assertNull(ClassName.getPrimitiveType("java/util/HashMap"));
56         }
57     
58         public void testExtractClassNameBad() {
59             try {
60                 ClassName.extractClassName("L[Ljava/lang/Integer;");
61                 fail();
62             } catch (IllegalArgumentException e) {
63                 assert true;
64             }
65     
66         }
67     
68     }
