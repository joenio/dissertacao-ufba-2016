7     public class WriteOnceProperties extends Properties {
8     
9         private static final long serialVersionUID = 1L;
10         private final Map<String, PropertyReadAt> propertReadAt = new HashMap<String, PropertyReadAt>();
11     
12         static class PropertyReadAt extends Exception {
13             private static final long serialVersionUID = 1L;
14         }
15     
16         private WriteOnceProperties(Properties initialValue) {
17             super.putAll(initialValue);
18         }
19     
20         @Override
21         public final synchronized int hashCode() {
22             return super.hashCode();
23         }
24     
25         /* overridden to avoid EQ_DOESNT_OVERRIDE_EQUALS */
26         @Override
27         public final synchronized boolean equals(Object o) {
28             return super.equals(o);
29         }
30     
31         @Override
32         public synchronized String getProperty(String key) {
33             String result = super.getProperty(key);
34             if (result != null && result.length() > 0 && !propertReadAt.containsKey(key)) {
35                 propertReadAt.put(key, new PropertyReadAt());
36             }
37             return result;
38         }
39     
40         @Override
41         public synchronized String getProperty(String key, String defaultValue) {
42             String result = super.getProperty(key, defaultValue);
43             if (result != null && result.length() > 0 && !propertReadAt.containsKey(key)) {
44                 propertReadAt.put(key, new PropertyReadAt());
45             }
46             return result;
47         }
48     
49         @Override
50         public synchronized Object setProperty(String key, String value) {
51             if (propertReadAt.containsKey(key) && !value.equals(super.getProperty(key))) {
52                 IllegalStateException e = new IllegalStateException("Changing property '" + key + "' to '" + value
53                         + "' after it has already been read as '" + super.getProperty(key) + "'");
54                 e.initCause(propertReadAt.get(key));
55                 throw e;
56             }
57             return super.setProperty(key, value);
58         }
59     
60         public static void makeSystemPropertiesWriteOnce() {
61             Properties properties = System.getProperties();
62             if (properties instanceof WriteOnceProperties) {
63                 return;
64             }
65             System.setProperties(new WriteOnceProperties(properties));
66         }
