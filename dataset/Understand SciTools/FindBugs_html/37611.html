20     package edu.umd.cs.findbugs.detect;
21     
22     import java.util.Set;
23     
24     import org.apache.bcel.classfile.Code;
25     
26     import edu.umd.cs.findbugs.BugInstance;
27     import edu.umd.cs.findbugs.BugReporter;
28     import edu.umd.cs.findbugs.ClassAnnotation;
29     import edu.umd.cs.findbugs.ba.AnalysisContext;
30     import edu.umd.cs.findbugs.ba.ch.Subtypes2;
31     import edu.umd.cs.findbugs.bcel.OpcodeStackDetector;
32     import edu.umd.cs.findbugs.classfile.ClassDescriptor;
33     
34     public class SynchronizeOnClassLiteralNotGetClass extends OpcodeStackDetector {
35     
36         BugReporter bugReporter;
37     
38         public SynchronizeOnClassLiteralNotGetClass(BugReporter bugReporter) {
39             this.bugReporter = bugReporter;
40         }
41     
42         @Override
43         public void visit(Code code) {
44             boolean interesting = !getMethod().isStatic() && !getThisClass().isFinal();
45     
46             if (interesting) {
47                 // initialize any variables we want to initialize for the method
48                 pendingBug = null;
49                 super.visit(code); // make callbacks to sawOpcode for all opcodes
50                 assert (pendingBug == null);
51             }
52         }
53     
54         /*
55          * Looking for ALOAD 0 INVOKEVIRTUAL
56          * java/lang/Object.getClass()Ljava/lang/Class; DUP ASTORE 1 MONITORENTER
57          */
58         int state = 0;
59     
60         boolean seenPutStatic, seenGetStatic;
61     
62         BugInstance pendingBug;
63     
64         @Override
65         public void sawOpcode(int seen) {
66             if (pendingBug != null) {
67                 if (seen == PUTSTATIC) {
68                     String classConstantOperand = getClassConstantOperand();
69                     String thisClassName = getThisClass().getClassName().replace('.', '/');
70                     if (classConstantOperand.equals(thisClassName)) {
71                         seenPutStatic = true;
72                     }
73                 } else if (seen == GETSTATIC) {
74                     String classConstantOperand = getClassConstantOperand();
75                     String thisClassName = getThisClass().getClassName().replace('.', '/');
76                     if (classConstantOperand.equals(thisClassName)) {
77                         seenGetStatic = true;
78                     }
79                 } else if (seen == MONITOREXIT) {
80                     int priority = LOW_PRIORITY;
81                     if (seenPutStatic || seenGetStatic) {
82                         priority--;
83                     }
84                     try {
85                         Subtypes2 subtypes2 = AnalysisContext.currentAnalysisContext().getSubtypes2();
86                         Set<ClassDescriptor> directSubtypes = subtypes2.getDirectSubtypes(getClassDescriptor());
87                         if (!directSubtypes.isEmpty()) {
88                             for (ClassDescriptor sub : directSubtypes) {
89                                 pendingBug.addClass(sub).describe(ClassAnnotation.SUBCLASS_ROLE);
90                             }
91                             priority--;
92                         }
93                     } catch (ClassNotFoundException e) {
94                         bugReporter.reportMissingClass(e);
95                     }
96                     pendingBug.setPriority(priority);
97                     bugReporter.reportBug(pendingBug);
98                     pendingBug = null;
99                 }
100                 return;
101             }
102             switch (state) {
103             case 0:
104                 if (seen == ALOAD_0) {
105                     state = 1;
106                 }
107                 break;
108             case 1:
109                 if (seen == INVOKEVIRTUAL && "getClass".equals(getNameConstantOperand())
110                 && "()Ljava/lang/Class;".equals(getSigConstantOperand())) {
111                     state = 2;
112                 } else {
113                     state = 0;
114                 }
115                 break;
116             case 2:
117                 if (seen == DUP) {
118                     state = 3;
119                 } else {
120                     state = 0;
121                 }
122                 break;
123             case 3:
124                 if (isRegisterStore()) {
125                     state = 4;
126                 } else {
127                     state = 0;
128                 }
129                 break;
130             case 4:
131                 if (seen == MONITORENTER) {
132                     pendingBug = new BugInstance(this, "WL_USING_GETCLASS_RATHER_THAN_CLASS_LITERAL", NORMAL_PRIORITY)
133                     .addClassAndMethod(this).addSourceLine(this);
134                 }
135                 state = 0;
136                 seenGetStatic = seenPutStatic = false;
137                 break;
138             default:
139                 break;
140             }
141         }
142     
143     }
