35     /**
36      * Mine historical information from a BugCollection. The BugCollection should be
37      * built using UpdateBugCollection to record the history of analyzing all
38      * versions over time.
39      *
40      * @author David Hovemeyer
41      * @author William Pugh
42      */
43     public class RebornIssues {
44         BugCollection bugCollection;
45     
46         public RebornIssues() {
47         }
48     
49         public RebornIssues(BugCollection bugCollection) {
50             this.bugCollection = bugCollection;
51         }
52     
53         public void setBugCollection(BugCollection bugCollection) {
54             this.bugCollection = bugCollection;
55         }
56     
57         public RebornIssues execute() {
58     
59             Map<String, List<BugInstance>> map = new HashMap<String, List<BugInstance>>();
60             for (BugInstance b : bugCollection.getCollection()) {
61                 if (b.getFirstVersion() != 0 || b.getLastVersion() != -1) {
62                     List<BugInstance> lst = map.get(b.getInstanceHash());
63                     if (lst == null) {
64                         lst = new LinkedList<BugInstance>();
65                         map.put(b.getInstanceHash(), lst);
66                     }
67                     lst.add(b);
68                 }
69             }
70             for (List<BugInstance> lst : map.values()) {
71                 if (lst.size() > 1) {
72                     TreeSet<Long> removalTimes = new TreeSet<Long>();
73                     TreeSet<Long> additionTimes = new TreeSet<Long>();
74     
75                     String bugPattern = "XXX";
76                     for (BugInstance b : lst) {
77                         bugPattern = b.getBugPattern().getType();
78                         if (b.getFirstVersion() > 0) {
79                             additionTimes.add(b.getFirstVersion());
80                         }
81                         if (b.getLastVersion() != -1) {
82                             removalTimes.add(b.getLastVersion());
83                         }
84                     }
85                     Iterator<Long> aI = additionTimes.iterator();
86                     if (!aI.hasNext()) {
87                         continue;
88                     }
89                     long a = aI.next();
90                     loop: for (Long removed : removalTimes) {
91                         while (a <= removed) {
92                             if (!aI.hasNext()) {
93                                 break loop;
94                             }
95                             a = aI.next();
96                         }
97                         System.out.printf("%5d %5d %s%n", removed, a, bugPattern);
98                     }
99     
100                 }
101             }
102             return this;
103         }

118         public static void main(String[] args) throws Exception {
119             FindBugs.setNoAnalysis();
120             DetectorFactoryCollection.instance(); // load plugins
121     
122             RebornIssues reborn = new RebornIssues();
123             CommandLine commandLine = new CommandLine();
124             int argCount = commandLine.parse(args, 0, 2, "Usage: " + RebornIssues.class.getName()
125                     + " [options] [<xml results> [<history]] ");
126     
127             SortedBugCollection bugCollection = new SortedBugCollection();
128             if (argCount < args.length) {
129                 bugCollection.readXML(args[argCount++]);
130             } else {
131                 bugCollection.readXML(System.in);
132             }
133             reborn.setBugCollection(bugCollection);
134             reborn.execute();
