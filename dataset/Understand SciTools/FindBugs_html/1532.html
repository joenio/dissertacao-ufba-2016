20     package edu.umd.cs.findbugs.detect;
21     
22     import java.util.regex.Pattern;
23     
24     import org.apache.bcel.classfile.Code;
25     
26     import edu.umd.cs.findbugs.BugInstance;
27     import edu.umd.cs.findbugs.BugReporter;
28     import edu.umd.cs.findbugs.BytecodeScanningDetector;
29     import edu.umd.cs.findbugs.StatelessDetector;
30     import edu.umd.cs.findbugs.TypeAnnotation;
31     
32     /* Look for sequences of the form:
33      *   ICONST_1
34      ANEWARRAY int[]
35      DUP
36      ICONST_0
37      ALOAD 1: a
38      AASTORE
39      INVOKESTATIC Arrays.asList(Object[]) : List
40      */
41     
42     public class VarArgsProblems extends BytecodeScanningDetector implements StatelessDetector {
43     
44         private final BugReporter bugReporter;
45     
46         private int state;
47     
48         public VarArgsProblems(BugReporter bugReporter) {
49             this.bugReporter = bugReporter;
50         }
51     
52         @Override
53         public void visitCode(Code obj) {
54             state = SEEN_NOTHING;
55             super.visitCode(obj);
56         }
57     
58         public static final int SEEN_NOTHING = 0;
59     
60         public static final int SEEN_ICONST_1 = 1;
61     
62         public static final int SEEN_ANEWARRAY = 2;
63     
64         public static final int SEEN_DUP = 3;
65     
66         public static final int SEEN_ICONST_0 = 4;
67     
68         public static final int SEEN_ALOAD = 5;
69     
70         public static final int SEEN_AASTORE = 6;
71     
72         public static final int SEEN_GOTO = 7;
73     
74         Pattern primitiveArray = Pattern.compile("\\[[IJDFSCB]");
75     
76         String primitiveArraySig;
77     
78         @Override
79         public void sawOpcode(int seen) {
80             // System.out.println("State:" + state);
81             if (seen == GOTO && getBranchOffset() == 4) {
82                 state = SEEN_GOTO;
83             } else {
84                 switch (state) {
85                 case SEEN_NOTHING:
86                     if ((seen == ICONST_1)) {
87                         state = SEEN_ICONST_1;
88                     }
89                     break;
90     
91                 case SEEN_ICONST_1:
92                     if (seen == ANEWARRAY && primitiveArray.matcher(getClassConstantOperand()).matches()) {
93                         // System.out.println("Allocation of array of type " +
94                         // getClassConstantOperand());
95                         primitiveArraySig = getClassConstantOperand();
96                         state = SEEN_ANEWARRAY;
97                     } else {
98                         state = SEEN_NOTHING;
99                     }
100                     break;
101     
102                 case SEEN_ANEWARRAY:
103                     if (seen == DUP) {
104                         state = SEEN_DUP;
105                     } else {
106                         state = SEEN_NOTHING;
107                     }
108                     break;
109                 case SEEN_DUP:
110                     if (seen == ICONST_0) {
111                         state = SEEN_ICONST_0;
112                     } else {
113                         state = SEEN_NOTHING;
114                     }
115                     break;
116                 case SEEN_ICONST_0:
117                     if (((seen >= ALOAD_0) && (seen < ALOAD_3)) || (seen == ALOAD)) {
118                         state = SEEN_ALOAD;
119                     } else {
120                         state = SEEN_NOTHING;
121                     }
122                     break;
123     
124                 case SEEN_ALOAD:
125                     if (seen == AASTORE) {
126                         state = SEEN_AASTORE;
127                     } else {
128                         state = SEEN_NOTHING;
129                     }
130                     break;
131     
132                 case SEEN_AASTORE:
133                     if (seen == INVOKESTATIC || seen == INVOKEINTERFACE || seen == INVOKESPECIAL || seen == INVOKEVIRTUAL) {
134                         // System.out.println(getClassConstantOperand());
135                         // System.out.println(getNameConstantOperand());
136                         // System.out.println(getSigConstantOperand());
137                         if (getSigConstantOperand().indexOf("Ljava/lang/Object;)") == -1) {
138                             break;
139                         }
140                         int priority = NORMAL_PRIORITY;
141                         if ("asList".equals(getNameConstantOperand()) && "java/util/Arrays".equals(getClassConstantOperand())) {
142                             priority = HIGH_PRIORITY;
143                         }
144                         bugReporter.reportBug(new BugInstance(this, "VA_PRIMITIVE_ARRAY_PASSED_TO_OBJECT_VARARG", priority)
145                         .addClassAndMethod(this).addType(primitiveArraySig).describe(TypeAnnotation.FOUND_ROLE)
146                         .addCalledMethod(this).addSourceLine(this));
147                     }
148                     state = SEEN_NOTHING;
149                     break;
150     
151                 case SEEN_GOTO:
152                     state = SEEN_NOTHING;
153                     break;
154                 default:
155                     throw new IllegalStateException("State " + state + " not expected");
156     
157                 }
158             }
159         }
160     }
