20     package edu.umd.cs.findbugs.util;
21     
22     import java.util.Collections;
23     import java.util.HashMap;
24     import java.util.Map;
25     import java.util.Set;
26     
27     /**
28      * @author William Pugh
29      */
30     public class DualKeyHashMap<K1, K2, V> {
31         Map<K1, Map<K2, V>> map = new HashMap<K1, Map<K2, V>>();
32     
33         public V get(K1 k1, K2 k2) {
34             Map<K2, V> m = map.get(k1);
35             if (m == null) {
36                 return null;
37             }
38             return m.get(k2);
39         }
40     
41         public boolean containsKey(K1 k1, K2 k2) {
42             Map<K2, V> m = map.get(k1);
43             if (m == null) {
44                 return false;
45             }
46             return m.containsKey(k2);
47         }
48     
49         public Map<K2, V> get(K1 k1) {
50             Map<K2, V> m = map.get(k1);
51             if (m == null) {
52                 return Collections.<K2, V> emptyMap();
53             }
54             return m;
55         }
56     
57         public Set<K1> keySet() {
58             return map.keySet();
59         }
60     
61         public V put(K1 k1, K2 k2, V v) {
62             Map<K2, V> m = map.get(k1);
63             if (m == null) {
64                 m = Collections.singletonMap(k2, v);
65                 map.put(k1, m);
66                 return null;
67             } else if (m instanceof HashMap) {
68                 return m.put(k2, v);
69             } else {
70                 m = Util.makeSmallHashMap(m);
71                 map.put(k1, m);
72                 return m.put(k2, v);
73     
74             }
75     
76         }
77     
78     }
