24     /**
25      * A multiset of obligations that must be cleaned up by error-handling code.
26      *
27      * <p>
28      * See Weimer and Necula, <a href="http://doi.acm.org/10.1145/1028976.1029011"
29      * >Finding and preventing run-time error handling mistakes</a>, OOPSLA 2004.
30      * </p>
31      *
32      * @author David Hovemeyer
33      */
34     public class ObligationSet {
35         private static final int INVALID_HASH_CODE = -1;
36     
37         private final short[] countList;
38     
39         // private final short[] whereCreated;
40         private final ObligationFactory factory;
41     
42         private int cachedHashCode;
43     
44         public ObligationSet(/* int maxObligationTypes, */ObligationFactory factory) {
45             this.countList = new short[factory.getMaxObligationTypes()];
46             // this.whereCreated = new short[factory.getMaxObligationTypes()];
47             this.factory = factory;
48             invalidate();
49         }
50     
51     
52         public boolean isEmpty() {
53             for(short s : countList) {
54                 if (s > 0) {
55                     return false;
56                 }
57             }
58             return true;
59         }
60         public void add(Obligation obligation) {
61             invalidate();
62             countList[obligation.getId()]++;
63         }
64     
65         public void remove(Obligation obligation) {
66             invalidate();
67             int count = countList[obligation.getId()];
68             if (count > 0)
69             {
70                 countList[obligation.getId()]--; // = (short)(count - 1);
71             }
72         }
73     
74         public int getCount(int id) {
75             return countList[id];
76         }

112         @Override
113         public boolean equals(Object o) {
114             if (o == null || o.getClass() != this.getClass()) {
115                 return false;
116             }
117     
118             ObligationSet other = (ObligationSet) o;
119     
120             if (!Arrays.equals(this.countList, other.countList)
121                     /* || !Arrays.equals(this.whereCreated, other.whereCreated) */) {
122                 return false;
123             }
124     
125             return true;
126         }
127     
128         /*
129          * NOTE: this string is incorporated into a StringAnnotation when reporting
130          * OBL_ warnings, so the output needs to be user-friendly.
131          */
132         @Override
133         public String toString() {
134             StringBuilder buf = new StringBuilder();
135             buf.append("{");
136             int count = 0;
137             for (int i = 0; i < countList.length; ++i) {
138                 if (countList[i] == 0) {
139                     continue;
140                 }
141                 if (count > 0) {
142                     buf.append(",");
143                 }
144                 buf.append(factory.getObligationById(i).toString());
145                 buf.append(" x ");
146                 buf.append(countList[i]);
147                 ++count;
148             }
149             buf.append("}");
150             // buf.append("@" + System.identityHashCode(this));
151             return buf.toString();
152         }
153     
154         public void copyFrom(ObligationSet other) {
155             System.arraycopy(other.countList, 0, this.countList, 0, other.countList.length);
156             // System.arraycopy(other.whereCreated, 0, this.whereCreated, 0,
157             // other.whereCreated.length);
158             invalidate();
159         }
160     
161         public ObligationSet duplicate() {
162             ObligationSet dup = new ObligationSet(/* countList.length, */factory);
163             dup.copyFrom(this);
164             return dup;
165         }
166     
167         @Override
168         public int hashCode() {
169             if (cachedHashCode == INVALID_HASH_CODE) {
170                 int value = 0;
171                 for (int i = 0; i < countList.length; ++i) {
172                     value += (13 * i * (countList[i]/* + whereCreated[i] */));
173                 }
174                 cachedHashCode = value;
175             }
176             return cachedHashCode;
177         }
178     
179         private void invalidate() {
180             this.cachedHashCode = INVALID_HASH_CODE;
181         }
182     }
