39     /**
40      * A PatternElement to match a method invocation. Currently, we don't allow
41      * variables in this element (for arguments and return value). This would be a
42      * good thing to add. We also don't distinguish between invokevirtual,
43      * invokeinterface, and invokespecial.
44      * <p/>
45      * <p>
46      * Invoke objects match by class name, method name, method signature, and
47      * <em>mode</em>.
48      * <p/>
49      * <p>
50      * Names and signatures may be matched in several ways:
51      * <ol>
52      * <li>By an exact match. This is the default behavior.
53      * <li>By a regular expression. If the string provided to the Invoke constructor
54      * begins with a "/" character, the rest of the string is treated as a regular
55      * expression.
56      * <li>As a subclass match. This only applies to class name matches. If the
57      * first character of a class name string is "+", then the rest of the string is
58      * treated as the name of a base class. Any subclass or subinterface of the
59      * named type will be accepted.
60      * </ol>
61      * <p/>
62      * <p>
63      * The <em>mode</em> specifies what kind of invocations in the Invoke element
64      * matches. It is specified as the bitwise combination of the following values:
65      * <ol>
66      * <li> <code>INSTANCE</code>, which matches ordinary instance method invocations
67      * <li> <code>STATIC</code>, which matches static method invocations
68      * <li> <code>CONSTRUCTOR</code>, which matches object constructor invocations
69      * </ol>
70      * The special mode <code>ORDINARY_METHOD</code> is equivalent to
71      * <code>INSTANCE|STATIC</code>. The special mode <code>ANY</code> is equivalent
72      * to <code>INSTANCE|STATIC|CONSTRUCTOR</code>.
73      *
74      * @author David Hovemeyer
75      * @see PatternElement
76      */
77     public class Invoke extends PatternElement {
78     
79         /**
80          * Match ordinary (non-constructor) instance invocations.
81          */
82         public static final int INSTANCE = 1;
83     
84         /**
85          * Match static invocations.
86          */
87         public static final int STATIC = 2;
88     
89         /**
90          * Match object constructor invocations.
91          */
92         public static final int CONSTRUCTOR = 4;
93     
94         /**
95          * Match ordinary methods (everything except constructors).
96          */
97         public static final int ORDINARY_METHOD = INSTANCE | STATIC;
98     
99         /**
100          * Match both static and instance invocations.
101          */
102         public static final int ANY = INSTANCE | STATIC | CONSTRUCTOR;
103     
104         private interface StringMatcher {
105             public boolean match(String s);
106         }
107     
108         private static class ExactStringMatcher implements StringMatcher {
109             private final String value;
110     
111             public ExactStringMatcher(String value) {
112                 this.value = value;
113             }
114     
115             @Override
116             public boolean match(String s) {
117                 return s.equals(value);
118             }
119         }
120     
121         private static class RegexpStringMatcher implements StringMatcher {
122             private final Pattern pattern;
123     
124             public RegexpStringMatcher(String re) {
125                 pattern = Pattern.compile(re);
126             }
127     
128             @Override
129             public boolean match(String s) {
130                 return pattern.matcher(s).matches();
131             }
132         }
133     
134         private static class SubclassMatcher implements StringMatcher {
135             private final String className;
136     
137             public SubclassMatcher(String className) {
138                 this.className = className;
139             }
140     
141             @Override
142             public boolean match(String s) {
143                 try {
144                     return Hierarchy.isSubtype(s, className);
145                 } catch (ClassNotFoundException e) {
146                     AnalysisContext.reportMissingClass(e);
147                     return false;
148                 }
149             }
150         }
151     
152         private final StringMatcher classNameMatcher;
153     
154         private final StringMatcher methodNameMatcher;
155     
156         private final StringMatcher methodSigMatcher;
157     
158         private final int mode;
159     
160         /**
161          * Constructor.
162          *
163          * @param className
164          *            the class name of the method; may be specified exactly, as a
165          *            regexp, or as a subtype match
166          * @param methodName
167          *            the name of the method; may be specified exactly or as a
168          *            regexp
169          * @param methodSig
170          *            the signature of the method; may be specified exactly or as a
171          *            regexp
172          * @param mode
173          *            the mode of invocation
174          */
175         public Invoke(String className, String methodName, String methodSig, int mode,
176                 @Nullable RepositoryLookupFailureCallback lookupFailureCallback) {
177             this.classNameMatcher = createClassMatcher(className);
178             this.methodNameMatcher = createMatcher(methodName);
179             this.methodSigMatcher = createMatcher(methodSig);
180             this.mode = mode;
181         }
182     
183         private StringMatcher createClassMatcher(String s) {
184             return s.startsWith("+") ? new SubclassMatcher(s.substring(1)) : createMatcher(s);
185         }
186     
187         private StringMatcher createMatcher(String s) {
188             return s.startsWith("/") ? (StringMatcher) new RegexpStringMatcher(s.substring(1))
189             : (StringMatcher) new ExactStringMatcher(s);
190         }
191     
192         @Override
193         public MatchResult match(InstructionHandle handle, ConstantPoolGen cpg, ValueNumberFrame before, ValueNumberFrame after,
194                 BindingSet bindingSet) throws DataflowAnalysisException {
195     
196             // See if the instruction is an InvokeInstruction
197             Instruction ins = handle.getInstruction();
198             if (!(ins instanceof InvokeInstruction)) {
199                 return null;
200             }
201             InvokeInstruction inv = (InvokeInstruction) ins;
202     
203             String methodName = inv.getMethodName(cpg);
204             boolean isStatic = inv.getOpcode() == Constants.INVOKESTATIC;
205             boolean isCtor = "<init>".equals(methodName);
206     
207             int actualMode = 0;
208     
209             if (isStatic) {
210                 actualMode |= STATIC;
211             }
212             if (isCtor) {
213                 actualMode |= CONSTRUCTOR;
214             }
215             if (!isStatic && !isCtor) {
216                 actualMode |= INSTANCE;
217             }
218     
219             // Intersection of actual and desired modes must be nonempty.
220             if ((actualMode & mode) == 0) {
221                 return null;
222             }
223     
224             // Check class name, method name, and method signature.
225             if (!methodNameMatcher.match(methodName) || !methodSigMatcher.match(inv.getSignature(cpg))
226                     || !classNameMatcher.match(inv.getClassName(cpg))) {
227                 return null;
228             }
229     
230             // It's a match!
231             return new MatchResult(this, bindingSet);
232     
233         }
234     
235         @Override
236         public boolean acceptBranch(Edge edge, InstructionHandle source) {
237             return true;
238         }
239     
240         @Override
241         public int minOccur() {
242             return 1;
243         }
244     
245         @Override
246         public int maxOccur() {
247             return 1;
248         }
249     }
