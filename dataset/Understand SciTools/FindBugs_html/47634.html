20     package edu.umd.cs.findbugs.ba;
21     
22     import java.io.PrintStream;
23     
24     import org.apache.bcel.generic.InstructionHandle;
25     
26     /**
27      * CFGPrinter class which prints dataflow values at each basic block and
28      * instruction.
29      */
30     public class DataflowCFGPrinter<Fact, AnalysisType extends DataflowAnalysis<Fact>> extends CFGPrinter {
31         private final Dataflow<Fact, AnalysisType> dataflow;
32     
33         /**
34          * Constructor.
35          *
36          * @param dataflow
37          *            the Dataflow object whose values should be used to annotate
38          *            the printed CFG
39          */
40         public DataflowCFGPrinter(Dataflow<Fact, AnalysisType> dataflow) {
41             super(dataflow.getCFG());
42             this.dataflow = dataflow;
43     
44             setIsForwards(dataflow.getAnalysis().isForwards());
45         }
46     
47         /*
48          * (non-Javadoc)
49          *
50          * @see
51          * edu.umd.cs.findbugs.ba.CFGPrinter#edgeAnnotate(edu.umd.cs.findbugs.ba
52          * .Edge)
53          */
54         @Override
55         public String edgeAnnotate(Edge edge) {
56             String edgeAnnotation = "";
57             try {
58                 edgeAnnotation = " " + dataflow.getAnalysis().factToString(dataflow.getAnalysis().getFactOnEdge(edge));
59             } catch (Throwable e) {
60                 // ignore
61             }
62             return edgeAnnotation;
63         }
64     
65         @Override
66         public String blockStartAnnotate(BasicBlock bb) {
67             boolean flip = isForwards() != dataflow.getAnalysis().isForwards();
68             Fact fact = flip ? dataflow.getResultFact(bb) : dataflow.getStartFact(bb);
69     
70             return " " + dataflow.getAnalysis().factToString(fact);
71         }
72     
73         @Override
74         public String blockAnnotate(BasicBlock bb) {
75             boolean flip = isForwards() != dataflow.getAnalysis().isForwards();
76             Fact fact = flip ? dataflow.getStartFact(bb) : dataflow.getResultFact(bb);
77     
78             return " " + dataflow.getAnalysis().factToString(fact);
79         }
80     
81         @Override
82         public String instructionAnnotate(InstructionHandle handle, BasicBlock bb) {
83             try {
84                 boolean flip = isForwards() != dataflow.getAnalysis().isForwards();
85     
86                 Location loc = new Location(handle, bb);
87     
88                 Fact fact = flip ? dataflow.getAnalysis().getFactAfterLocation(loc) : dataflow.getAnalysis().getFactAtLocation(loc);
89                 return " " + dataflow.getAnalysis().factToString(fact);
90             } catch (DataflowAnalysisException e) {
91                 throw new IllegalStateException("Caught exception: " + e.toString());
92             }
93         }
94     
95         /**
96          * Print CFG annotated with results from given dataflow analysis.
97          *
98          * @param <Fact>
99          *            Dataflow fact type
100          * @param <AnalysisType>
101          *            Dataflow analysis type
102          * @param dataflow
103          *            dataflow driver
104          * @param out
105          *            PrintStream to use
106          */
107         public static <Fact, AnalysisType extends BasicAbstractDataflowAnalysis<Fact>> void printCFG(
108                 Dataflow<Fact, AnalysisType> dataflow, PrintStream out) {
109             DataflowCFGPrinter<Fact, AnalysisType> printer = new DataflowCFGPrinter<Fact, AnalysisType>(dataflow);
110             printer.print(out);
111         }
112     
113     }
