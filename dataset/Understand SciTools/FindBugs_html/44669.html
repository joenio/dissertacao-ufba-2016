22     package edu.umd.cs.findbugs.graph;
23     
24     import java.util.Iterator;
25     
26     /**
27      * Graph interface; defines the operations used to access and manipulate a
28      * graph.
29      */
30     public interface Graph<EdgeType extends GraphEdge<EdgeType, VertexType>, VertexType extends GraphVertex<VertexType>> {
31     
32         /**
33          * Get number of edges in the graph.
34          */
35         public int getNumEdges();
36     
37         /**
38          * Get number of vertices in the graph.
39          */
40         public int getNumVertices();
41     
42         /**
43          * Get Iterator over all edges in the graph.
44          */
45         public Iterator<EdgeType> edgeIterator();
46     
47         /**
48          * Get Iterator over all vertices in the graph.
49          */
50         public Iterator<VertexType> vertexIterator();
51     
52         /**
53          * Add given vertex to the graph. The vertex should not be part of any other
54          * graph.
55          *
56          * @param v
57          *            the vertex to add
58          */
59         public void addVertex(VertexType v);
60     
61         /**
62          * Determine if the graph contains the given vertex.
63          *
64          * @param v
65          *            the vertex
66          * @return true if the vertex is part of the graph, false if not
67          */
68         public boolean containsVertex(VertexType v);
69     
70         /**
71          * Add a new edge to the graph. Duplicate edges (with same source and target
72          * vertices) are allowed.
73          *
74          * @param source
75          *            the source vertex
76          * @param target
77          *            the target vertex
78          * @return the new edge
79          */
80         public EdgeType createEdge(VertexType source, VertexType target);
81     
82         /**
83          * Look up an edge by source and target vertex. If multiple edges with same
84          * source and target vertex exist, one is selected arbitrarily.
85          *
86          * @param source
87          *            the source vertex
88          * @param target
89          *            the target vertex
90          * @return a matching edge, or null if there is no matching edge
91          */
92         public EdgeType lookupEdge(VertexType source, VertexType target);
93     
94         /**
95          * Get the number of numeric (integer) labels that have been assigned to
96          * vertices in the graph. All vertices in the graph are guaranteed to have
97          * labels in the range 0..n, where n is the value returned by this method.
98          */
99         public int getNumVertexLabels();
100     
101         /**
102          * Reset number of (integer) labels. This might be necessary if an algorithm
103          * has assigned new labels to a graph's vertices.
104          */
105         public void setNumVertexLabels(int numLabels);
106     
107         /**
108          * Get the number of numeric labels that have been assigned to edges.
109          */
110         public int getNumEdgeLabels();
111     
112         /**
113          * Reset the number of edge labels.
114          */
115         public void setNumEdgeLabels(int numLabels);
116     
117         /**
118          * Remove given edge from the graph.
119          */
120         public void removeEdge(EdgeType e);
121     
122         /**
123          * Remove given vertex from the graph. Note that all edges referencing the
124          * vertex will be removed.
125          */
126         public void removeVertex(VertexType v);
127     
128         /**
129          * Get an Iterator over outgoing edges from given vertex.
130          *
131          * @param source
132          *            the source vertex
133          * @return an Iterator over outgoing edges
134          */
135         public Iterator<EdgeType> outgoingEdgeIterator(VertexType source);
136     
137         /**
138          * Get an Iterator over incoming edges to a given vertex.
139          *
140          * @param target
141          *            the target vertex
142          * @return an Iterator over incoming edges
143          */
144         public Iterator<EdgeType> incomingEdgeIterator(VertexType target);
145     
146         /**
147          * Get number of edges going into given vertex.
148          *
149          * @param vertex
150          *            the vertex
151          * @return number of edges going into the vertex
152          */
153         public int getNumIncomingEdges(VertexType vertex);
154     
155         /**
156          * Get number of edges going out of given vertex.
157          *
158          * @param vertex
159          *            the vertex
160          * @return number of edges going out of the vertex
161          */
162         public int getNumOutgoingEdges(VertexType vertex);
163     
164         /**
165          * Get an iterator over the successors of this vertex; i.e., the targets of
166          * the vertex's outgoing edges.
167          *
168          * @param source
169          *            the source vertex
170          * @return an Iterator over the successors of the vertex
171          */
172         public Iterator<VertexType> successorIterator(VertexType source);
173     
174         /**
175          * Get an iterator over the predecessors of this vertex; i.e., the sources
176          * of the vertex's incoming edges.
177          *
178          * @param target
179          *            the target vertex
180          * @return an Iterator over the predecessors of the vertex
181          */
182         public Iterator<VertexType> predecessorIterator(VertexType target);
183     
184     }
