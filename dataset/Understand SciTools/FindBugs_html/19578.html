18     package edu.umd.cs.findbugs.workflow;
19     
20     import java.io.IOException;
21     import java.util.Map;
22     import java.util.TreeMap;
23     
24     import org.dom4j.DocumentException;
25     
26     import edu.umd.cs.findbugs.BugCollection;
27     import edu.umd.cs.findbugs.BugInstance;
28     import edu.umd.cs.findbugs.DetectorFactoryCollection;
29     import edu.umd.cs.findbugs.PackageStats;
30     import edu.umd.cs.findbugs.SortedBugCollection;
31     import edu.umd.cs.findbugs.util.ClassName;
32     
33     /**
34      * Java main application to compute update a historical bug collection with
35      * results from another build/analysis.
36      *
37      * @author William Pugh
38      */
39     
40     public class CountByPackagePrefix {
41     
42         /**
43          *
44          */
45         private static final String USAGE = "Usage: <cmd> " + " <prefixLength> [<bugs.xml>]";
46     
47         public static void main(String[] args) throws IOException, DocumentException {
48     
49             DetectorFactoryCollection.instance();
50             if (args.length != 1 && args.length != 2) {
51                 System.out.println(USAGE);
52                 return;
53             }
54     
55             int prefixLength = Integer.parseInt(args[0]);
56             BugCollection origCollection = new SortedBugCollection();
57             if (args.length == 1) {
58                 origCollection.readXML(System.in);
59             } else {
60                 origCollection.readXML(args[1]);
61             }
62             Map<String, Integer> map = new TreeMap<String, Integer>();
63             Map<String, Integer> ncss = new TreeMap<String, Integer>();
64     
65             for (BugInstance b : origCollection.getCollection()) {
66                 String prefix = ClassName.extractPackagePrefix(b.getPrimaryClass().getPackageName(), prefixLength);
67                 Integer v = map.get(prefix);
68                 if (v == null) {
69                     map.put(prefix, 1);
70                 } else {
71                     map.put(prefix, v + 1);
72                 }
73             }
74             for (PackageStats ps : origCollection.getProjectStats().getPackageStats()) {
75                 String prefix = ClassName.extractPackagePrefix(ps.getPackageName(), prefixLength);
76     
77                 Integer v = ncss.get(prefix);
78                 if (v == null) {
79                     ncss.put(prefix, ps.size());
80                 } else {
81                     ncss.put(prefix, v + ps.size());
82                 }
83     
84             }
85             for (Map.Entry<String, Integer> e : map.entrySet()) {
86                 String prefix = e.getKey();
87                 int warnings = e.getValue();
88                 if (warnings == 0) {
89                     continue;
90                 }
91                 Integer v = ncss.get(prefix);
92                 if (v == null || v.intValue() == 0) {
93                     v = 1;
94                 }
95     
96                 int density = warnings * 1000000 / v;
97                 if (warnings < 3 || v < 2000) {
98                     System.out.printf("%4s %4d %4d %s%n", " ", warnings, v / 1000, prefix);
99                 } else {
100                     System.out.printf("%4d %4d %4d %s%n", density, warnings, v / 1000, prefix);
101                 }
102             }
103     
104         }
105     }
