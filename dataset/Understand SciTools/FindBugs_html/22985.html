82         public ValueNumberAnalysis(MethodGen methodGen, DepthFirstSearch dfs, LoadedFieldSet loadedFieldSet,
83                 RepositoryLookupFailureCallback lookupFailureCallback) {
84     
85             super(dfs);
86             this.methodGen = methodGen;
87             this.factory = new ValueNumberFactory();
88             ValueNumberCache cache = new ValueNumberCache();
89             this.visitor = new ValueNumberFrameModelingVisitor(methodGen, factory, cache, loadedFieldSet, lookupFailureCallback);
90     
91             int numLocals = methodGen.getMaxLocals();
92             this.entryLocalValueList = new ValueNumber[numLocals];
93             for (int i = 0; i < numLocals; ++i) {
94                 this.entryLocalValueList[i] = factory.createFreshValue();
95             }
96     
97             this.exceptionHandlerValueNumberMap = new IdentityHashMap<BasicBlock, ValueNumber>();
98     
99             // For non-static methods, keep track of which value represents the
100             // "this" reference
101             if (!methodGen.isStatic()) {
102                 this.thisValue = entryLocalValueList[0];
103             }
104     
105             this.factAtLocationMap = new HashMap<Location, ValueNumberFrame>();
106             this.factAfterLocationMap = new HashMap<Location, ValueNumberFrame>();
107             if (DEBUG) {
108                 System.out.println("VNA Analysis " + methodGen.getClassName() + "." + methodGen.getName() + " : "
109                         + methodGen.getSignature());
110             }
111     
112         }

142         /**
143          * Get the value number assigned to the given local variable upon entry to
144          * the method.
145          *
146          * @param local
147          *            local variable number
148          * @return ValueNumber assigned to the local variable
149          */
150         public ValueNumber getEntryValue(int local) {
151             return entryLocalValueList[local];
152         }
153     
154         /**
155          * Get the value number assigned to the given parameter upon entry to the
156          * method.
157          *
158          * @param param
159          *            a parameter (0 == first parameter)
160          * @return the ValueNumber assigned to that parameter
161          */
162         public ValueNumber getEntryValueForParameter(int param) {
163     
164             SignatureParser sigParser = new SignatureParser(methodGen.getSignature());
165             int p = 0;
166             int slotOffset =  methodGen.isStatic() ? 0 : 1;
167     
168             for ( String paramSig : sigParser.parameterSignatures()) {
169                 if (p == param) {
170                     return getEntryValue(slotOffset);
171                 }
172                 param++;
173                 slotOffset += SignatureParser.getNumSlotsForType(paramSig);
174             }
175     
176             throw new IllegalStateException();
177         }

184         @Override
185         public void initEntryFact(ValueNumberFrame result) {
186             // Change the frame from TOP to something valid.
187             result.setValid();
188     
189             // At entry to the method, each local has (as far as we know) a unique
190             // value.
191             int numSlots = result.getNumSlots();
192             for (int i = 0; i < numSlots; ++i) {
193                 result.setValue(i, entryLocalValueList[i]);
194             }
195         }

197         @Override
198         public void transfer(BasicBlock basicBlock, InstructionHandle end, ValueNumberFrame start, ValueNumberFrame result)
199                 throws DataflowAnalysisException {
200             if(basicBlock.isExceptionThrower() && isFactValid(start)) {
201                 /* If exceptionThrower is invoke instruction then it's possible that
202                  * it was partially executed before an exception occurred
203                  * So we have to kill available loads when control is transferred to the catch block
204                  */
205                 InstructionHandle handle = basicBlock.getExceptionThrower();
206                 Instruction inst = handle.getInstruction();
207                 if(inst instanceof InvokeInstruction || inst instanceof INVOKEDYNAMIC) {
208                     copy(start, result);
209                     visitor.setFrameAndLocation(result, new Location(handle, basicBlock));
210                     visitor.setHandle(handle);
211                     visitor.visitInvokeOnException(inst);
212                     return;
213                 }
214             }
215             super.transfer(basicBlock, end, start, result);
216         }

218         @Override
219         public void transferInstruction(InstructionHandle handle, BasicBlock basicBlock, ValueNumberFrame fact)
220                 throws DataflowAnalysisException {
221     
222             Location location = new Location(handle, basicBlock);
223     
224             ValueNumberFrame atLocation = getFactAtLocation(location);
225             copy(fact, atLocation);
226     
227             visitor.setFrameAndLocation(fact, location);
228             visitor.setHandle(handle);
229             visitor.analyzeInstruction(handle.getInstruction());
230     
231             ValueNumberFrame afterLocation = getFactAfterLocation(location);
232             copy(fact, afterLocation);
233         }

235         @Override
236         public void meetInto(ValueNumberFrame fact, Edge edge, ValueNumberFrame result) throws DataflowAnalysisException {
237             if (edge.getTarget().isExceptionHandler() && fact.isValid()) {
238                 // Special case: when merging predecessor facts for entry to
239                 // an exception handler, we clear the stack and push a
240                 // single entry for the exception object. That way, the locals
241                 // can still be merged.
242     
243                 // Get the value number for the exception
244                 BasicBlock handlerBlock = edge.getTarget();
245                 ValueNumber exceptionValueNumber = getExceptionValueNumber(handlerBlock);
246     
247                 // Set up the stack frame
248                 ValueNumberFrame tmpFact = createFact();
249                 tmpFact.copyFrom(fact);
250                 tmpFact.clearStack();
251                 tmpFact.pushValue(exceptionValueNumber);
252                 fact = tmpFact;
253             }
254     
255             mergeInto(fact, result);
256         }

264         @Override
265         protected void mergeValues(ValueNumberFrame otherFrame, ValueNumberFrame resultFrame, int slot)
266                 throws DataflowAnalysisException {
267             ValueNumber value = mergeValues(resultFrame, slot, resultFrame.getValue(slot), otherFrame.getValue(slot));
268             resultFrame.setValue(slot, value);
269         }

271         private ValueNumber mergeValues(ValueNumberFrame frame, int slot, ValueNumber mine, ValueNumber other) {
272     
273             // Merging slot values:
274             // - Merging identical values results in no change
275             // - If the values are different, and the value in the result
276             // frame is not the result of a previous result, a fresh value
277             // is allocated.
278             // - If the value in the result frame is the result of a
279             // previous merge, IT STAYS THE SAME.
280             //
281             // The "one merge" rule means that merged values are essentially like
282             // phi nodes. They combine some number of other values.
283     
284             // I believe (but haven't proved) that this technique is a dumb way
285             // of computing SSA.
286     
287             if (mine != frame.getValue(slot)) {
288                 throw new IllegalStateException();
289             }
290     
291             if (mine.equals(other)) {
292                 return mine;
293             }
294     
295             ValueNumber mergedValue = frame.getMergedValue(slot);
296             if (mergedValue == null) {
297                 mergedValue = factory.createFreshValue(ValueNumber.mergeFlags(mine.getFlags(), other.getFlags()) | ValueNumber.PHI_NODE);
298                 frame.setMergedValue(slot, mergedValue);
299     
300             }
301     
302             if (mergeTree != null) {
303                 mergeTree.mapInputToOutput(mine, mergedValue);
304                 mergeTree.mapInputToOutput(other, mergedValue);
305             }
306     
307             return mergedValue;
308         }

310         @Override
311         public ValueNumberFrame getFactAtLocation(Location location) {
312             ValueNumberFrame fact = factAtLocationMap.get(location);
313             if (fact == null) {
314                 fact = createFact();
315                 makeFactTop(fact);
316                 factAtLocationMap.put(location, fact);
317             }
318             return fact;
319         }

322         @Override
323         public ValueNumberFrame getFactAfterLocation(Location location) {
324             if (TRACE) {
325                 System.out.println("getting fact after " + location);
326             }
327             ValueNumberFrame fact = factAfterLocationMap.get(location);
328             if (fact == null) {
329                 if (TRACE) {
330                     System.out
331                     .println("Initialized fact after " + location + " @ "
332                             + Integer.toHexString(System.identityHashCode(location)) + " in "
333                             + Integer.toHexString(System.identityHashCode(this)) + " : "
334                             + factAfterLocationMap.containsKey(location));
335                 }
336     
337                 fact = createFact();
338                 makeFactTop(fact);
339                 factAfterLocationMap.put(location, fact);
340             }
341             return fact;
342         }

481         private ValueNumber getExceptionValueNumber(BasicBlock handlerBlock) {
482             ValueNumber valueNumber = exceptionHandlerValueNumberMap.get(handlerBlock);
483             if (valueNumber == null) {
484                 valueNumber = factory.createFreshValue();
485                 exceptionHandlerValueNumberMap.put(handlerBlock, valueNumber);
486             }
487             return valueNumber;
488         }
