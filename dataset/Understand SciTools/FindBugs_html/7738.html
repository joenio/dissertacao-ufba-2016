20     package edu.umd.cs.findbugs;
21     
22     import java.awt.GraphicsEnvironment;
23     import java.lang.reflect.Method;
24     import java.util.HashMap;
25     import java.util.Map;
26     
27     import edu.umd.cs.findbugs.internalAnnotations.StaticConstant;
28     
29     /**
30      * Class to launch the appropriate textUI or GUI. This class is the Main-Class
31      * in the findbugs.jar manifest, and is responsible for running an appropriate
32      * main() method.
33      *
34      * @author Bill Pugh
35      * @author David Hovemeyer
36      */
37     public class LaunchAppropriateUI {
38         /**
39          * UI code for the text (command line) UI.
40          */
41         public static final int TEXTUI = 0;
42     
43         /**
44          * UI code for the old Swing GUI. This is deprecated now.
45          */
46         public static final int GUI1 = 1;
47     
48         /**
49          * UI code for the new Swing GUI.
50          */
51         public static final int GUI2 = 2;
52     
53         /**
54          * UI code for displaying command line help.
55          */
56         public static final int SHOW_HELP = 1000;
57     
58         /**
59          * UI code for displaying command line version information.
60          */
61         public static final int SHOW_VERSION = 1001;
62     
63         /**
64          * Map of UI name strings to integer UI codes.
65          */
66         @StaticConstant
67         public static final Map<String, Integer> uiNameToCodeMap;
68         static {
69             uiNameToCodeMap = new HashMap<String, Integer>();
70             uiNameToCodeMap.put("textui", TEXTUI);
71             uiNameToCodeMap.put("gui", GUI2);
72             uiNameToCodeMap.put("gui1", GUI1);
73             uiNameToCodeMap.put("gui2", GUI2);
74             uiNameToCodeMap.put("help", SHOW_HELP);
75             uiNameToCodeMap.put("version", SHOW_VERSION);
76         }
77     
78         // Fields
79         /** Command line arguments. */
80         private String[] args;
81     
82         /**
83          * Constructor.
84          *
85          * @param args
86          *            command line arguments
87          */
88         public LaunchAppropriateUI(String[] args) {
89             this.args = args;
90         }
91     
92         /**
93          * Launch the appropriate UI.
94          *
95          * @throws java.lang.Exception
96          */
97         public void launch() throws Exception {
98             // Sanity-check the loaded BCEL classes
99             if (!CheckBcel.check()) {
100                 System.exit(1);
101             }
102     
103             int launchProperty = getLaunchProperty();
104     
105             if (GraphicsEnvironment.isHeadless() || launchProperty == TEXTUI) {
106                 FindBugs2.main(args);
107             } else if (launchProperty == SHOW_HELP) {
108                 ShowHelp.main(args);
109             } else if (launchProperty == SHOW_VERSION) {
110                 Version.main(new String[] { "-release" });
111             } else {
112                 Class<?> launchClass = Class.forName("edu.umd.cs.findbugs.gui2.Driver");
113                 Method mainMethod = launchClass.getMethod("main", args.getClass());
114                 mainMethod.invoke(null, (Object) args);
115             }
116     
117         }
118     
119         /**
120          * Find out what UI should be launched.
121          *
122          * <p>
123          * First, we check the command line arguments to see if the first argument
124          * specifies the UI (e.g., "-textui", "-gui", etc.)
125          *
126          * <p>
127          * If the first command line argument does not specify the UI, then we check
128          * the <code>findbugs.launchUI</code> system property to one of the
129          * following values:
130          *
131          * <ul>
132          * <li>-Dfindbugs.launchUI=textui for textui,</li>
133          * <li>-Dfindbugs.launchUI=gui1 for the original swing gui,</li>
134          * <li>-Dfindbugs.launchUI=gui2 for the new swing gui,</li>
135          * <li>-Dfindbugs.launchUI=version for the ShowVersion main() method, or</li>
136          * <li>-Dfindbugs.launchUI=help for the ShowHelp main() method.</li>
137          * </ul>
138          *
139          * Any other value (or the absence of any value) will not change the default
140          * behavior, which is to launch the newer "gui2" on systems that support it.
141          *
142          * @return an integer UI code: TEXTUI, GUI1, GUI2, SHOW_VERSION, SHOW_HELP,
143          *         or possibly another user-set int value
144          */
145         private int getLaunchProperty() {
146             // See if the first command line argument specifies the UI.
147             if (args.length > 0) {
148                 String firstArg = args[0];
149                 if (firstArg.startsWith("-")) {
150                     String uiName = firstArg.substring(1);
151                     if (uiNameToCodeMap.containsKey(uiName)) {
152                         // Strip the first argument from the command line arguments.
153                         String[] modifiedArgs = new String[args.length - 1];
154                         System.arraycopy(args, 1, modifiedArgs, 0, args.length - 1);
155                         args = modifiedArgs;
156     
157                         return uiNameToCodeMap.get(uiName);
158                     }
159                 }
160             }
161     
162             // Check findbugs.launchUI property.
163             // "gui2" is the default if not otherwise specified.
164             String s = System.getProperty("findbugs.launchUI");
165     
166             if (s == null) {
167                 for (String a : args) {
168                     if ("-output".equals(a) || "-xml".equals(a) || a.endsWith(".class") || a.endsWith(".jar")) {
169                         return TEXTUI;
170                     }
171                 }
172                 s = "gui2";
173             }
174     
175             // See if the property value is one of the human-readable
176             // UI names.
177             if (uiNameToCodeMap.containsKey(s)) {
178                 return uiNameToCodeMap.get(s);
179             }
180     
181             // Fall back: try to parse it as an integer.
182             try {
183                 return Integer.parseInt(s);
184             } catch (NumberFormatException nfe) {
185                 return GUI2;
186             }
187         }
188     
189         /**
190          * main() method. This is where execution of FindBugs (any UI) starts.
191          *
192          * @param args
193          *            command line arguments
194          * @throws java.lang.Exception
195          */
196         public static void main(String args[]) throws Exception {
197             LaunchAppropriateUI launcher = new LaunchAppropriateUI(args);
198             launcher.launch();
199         }
200     }
