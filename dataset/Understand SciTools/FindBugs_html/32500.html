40     public class EqualsOperandShouldHaveClassCompatibleWithThis extends OpcodeStackDetector implements FirstPassDetector {
41     
42         final BugReporter bugReporter;
43     
44         final BugAccumulator bugAccumulator;
45     
46         final ClassSummary classSummary = new ClassSummary();
47     
48         public EqualsOperandShouldHaveClassCompatibleWithThis(BugReporter bugReporter) {
49             this.bugReporter = bugReporter;
50             this.bugAccumulator = new BugAccumulator(bugReporter);
51             AnalysisContext context = AnalysisContext.currentAnalysisContext();
52             context.setClassSummary(classSummary);
53         }
54     
55         @Override
56         public void visit(Code obj) {
57             if ("equals".equals(getMethodName()) && "(Ljava/lang/Object;)Z".equals(getMethodSig())) {
58                 super.visit(obj);
59                 if (AnalysisContext.currentAnalysisContext().isApplicationClass(getThisClass())) {
60                     bugAccumulator.reportAccumulatedBugs();
61                 }
62                 bugAccumulator.clearBugs();
63             }
64     
65         }
66     
67         /*
68          * (non-Javadoc)
69          *
70          * @see edu.umd.cs.findbugs.bcel.OpcodeStackDetector#sawOpcode(int)
71          */
72         @Override
73         public void sawOpcode(int seen) {
74             if (seen == INVOKEVIRTUAL) {
75                 if ("equals".equals(getNameConstantOperand()) && "(Ljava/lang/Object;)Z".equals(getSigConstantOperand())) {
76                     OpcodeStack.Item item = stack.getStackItem(1);
77                     ClassDescriptor c = DescriptorFactory.createClassDescriptorFromSignature(item.getSignature());
78                     check(c);
79     
80                 } else if ("java/lang/Class".equals(getClassConstantOperand())
81                         && ("isInstance".equals(getNameConstantOperand()) || "cast".equals(getNameConstantOperand()))) {
82                     OpcodeStack.Item item = stack.getStackItem(1);
83                     if ("Ljava/lang/Class;".equals(item.getSignature())) {
84                         Object value = item.getConstant();
85                         if (value instanceof String) {
86                             ClassDescriptor c = DescriptorFactory.createClassDescriptor((String) value);
87                             check(c);
88                         }
89                     }
90     
91                 }
92     
93             } else if (seen == INSTANCEOF || seen == CHECKCAST) {
94                 check(getClassDescriptorOperand());
95             }
96     
97         }
98     
99         /**
100          *
101          */
102         private void check(ClassDescriptor c) {
103             OpcodeStack.Item item = stack.getStackItem(0);
104             if (item.isInitialParameter() && item.getRegisterNumber() == 1) {
105                 ClassDescriptor thisClassDescriptor = getClassDescriptor();
106                 if (c.equals(thisClassDescriptor)) {
107                     return;
108                 }
109                 Subtypes2 subtypes2 = AnalysisContext.currentAnalysisContext().getSubtypes2();
110                 try {
111                     if (!c.isArray() && (subtypes2.isSubtype(c, thisClassDescriptor) || subtypes2.isSubtype(thisClassDescriptor, c))) {
112                         return;
113                     }
114     
115                     Type thisType = Type.getType(thisClassDescriptor.getSignature());
116                     Type cType = Type.getType(c.getSignature());
117                     IncompatibleTypes check = IncompatibleTypes.getPriorityForAssumingCompatible(thisType, cType, false);
118                     int priority = check.getPriority();
119                     if ("java/lang/Object".equals(getSuperclassName()) && ClassName.isAnonymous(getClassName())) {
120                         priority++;
121                     }
122                     bugAccumulator.accumulateBug(new BugInstance(this, "EQ_CHECK_FOR_OPERAND_NOT_COMPATIBLE_WITH_THIS", priority)
123                     .addClassAndMethod(this).addType(c).describe(TypeAnnotation.FOUND_ROLE), this);
124                     classSummary.checksForEqualTo(thisClassDescriptor, c);
125     
126                 } catch (ClassNotFoundException e) {
127                     bugReporter.reportMissingClass(e);
128                 }
129     
130             }
131         }
132     
133     }
