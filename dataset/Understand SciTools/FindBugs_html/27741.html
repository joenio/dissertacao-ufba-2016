20     package edu.umd.cs.findbugs.detect;
21     
22     import java.util.ArrayList;
23     import java.util.Iterator;
24     import java.util.List;
25     
26     import org.apache.bcel.classfile.Code;
27     
28     import edu.umd.cs.findbugs.BugAccumulator;
29     import edu.umd.cs.findbugs.BugInstance;
30     import edu.umd.cs.findbugs.BugReporter;
31     import edu.umd.cs.findbugs.MethodAnnotation;
32     import edu.umd.cs.findbugs.OpcodeStack;
33     import edu.umd.cs.findbugs.OpcodeStack.Item;
34     import edu.umd.cs.findbugs.ba.ClassContext;
35     import edu.umd.cs.findbugs.ba.ch.Subtypes2;
36     import edu.umd.cs.findbugs.bcel.OpcodeStackDetector;
37     import edu.umd.cs.findbugs.classfile.MethodDescriptor;
38     
39     /**
40      * @author Tagir Valeev
41      */
42     public class FindComparatorProblems extends OpcodeStackDetector {
43         private static final MethodDescriptor FLOAT_DESCRIPTOR = new MethodDescriptor("java/lang/Float", "compare", "(FF)I", true);
44         private static final MethodDescriptor DOUBLE_DESCRIPTOR = new MethodDescriptor("java/lang/Double", "compare", "(DD)I", true);
45     
46         private boolean isComparator;
47         private int lastEmptyStackPC;
48         private List<int[]> twoDoublesInStack;
49         private final BugAccumulator accumulator;
50     
51         public FindComparatorProblems(BugReporter reporter) {
52             this.accumulator = new BugAccumulator(reporter);
53         }
54     
55         @Override
56         public void visitClassContext(ClassContext classContext) {
57             boolean comparator = Subtypes2.instanceOf(classContext.getClassDescriptor(), "java.util.Comparator");
58             boolean comparable = Subtypes2.instanceOf(classContext.getClassDescriptor(), "java.lang.Comparable");
59             isComparator = comparator;
60             if (comparator || comparable) {
61                 super.visitClassContext(classContext);
62             }
63         }
64     
65         @Override
66         public boolean shouldVisitCode(Code obj) {
67             return !getMethodDescriptor().isStatic()
68                     && ((isComparator && getMethodName().equals("compare") && getMethodSig().endsWith(")I")) || ((getMethodName()
69                             .equals("compareTo") && getMethodSig().equals("(L"+getClassName()+";)I"))));
70         }
71     
72         @Override
73         public void visit(Code obj) {
74             this.twoDoublesInStack = new ArrayList<>();
75             this.lastEmptyStackPC = 0;
76             super.visit(obj);
77             this.accumulator.reportAccumulatedBugs();
78         }
79     
80         @Override
81         public void sawOpcode(int seen) {
82             if(getStack().getStackDepth() == 0) {
83                 this.lastEmptyStackPC = getPC();
84             }
85             if((seen == DCMPG || seen == DCMPL || seen == FCMPL || seen == FCMPG) && getStack().getStackDepth() == 2) {
86                 int[] startEnd = new int[] {this.lastEmptyStackPC, getPC()};
87                 for(Iterator<int[]> iterator = twoDoublesInStack.iterator(); iterator.hasNext(); ) {
88                     int[] oldStartEnd = iterator.next();
89                     if(codeEquals(oldStartEnd, startEnd)) {
90                         Item item1 = getStack().getStackItem(0);
91                         Item item2 = getStack().getStackItem(1);
92                         accumulator.accumulateBug(
93                                 new BugInstance("CO_COMPARETO_INCORRECT_FLOATING", NORMAL_PRIORITY).addClassAndMethod(this)
94                                 .addType(item1.getSignature())
95                                 .addMethod(item1.getSignature().equals("D")?DOUBLE_DESCRIPTOR:FLOAT_DESCRIPTOR).describe(MethodAnnotation.SHOULD_CALL)
96                                 .addValueSource(item1, this)
97                                 .addValueSource(item2, this), this);
98                         iterator.remove();
99                         return;
100                     }
101                 }
102                 twoDoublesInStack.add(startEnd);
103             }
104             if (seen == IRETURN) {
105                 OpcodeStack.Item top = stack.getStackItem(0);
106                 Object o = top.getConstant();
107                 if (o instanceof Integer && ((Integer)o).intValue() == Integer.MIN_VALUE) {
108                     accumulator.accumulateBug(
109                             new BugInstance(this, "CO_COMPARETO_RESULTS_MIN_VALUE", NORMAL_PRIORITY).addClassAndMethod(this), this);
110                 }
111             }
112         }
113     
114         /**
115          * @param oldStartEnd - int[] {oldStart, oldEnd}
116          * @param startEnd - int[] {start, end}
117          * @return true if code slices are the same
118          */
119         private boolean codeEquals(int[] oldStartEnd, int[] startEnd) {
120             int oldStart = oldStartEnd[0];
121             int oldEnd = oldStartEnd[1];
122             int start = startEnd[0];
123             int end = startEnd[1];
124             if(end-start != oldEnd - oldStart) {
125                 return false;
126             }
127             for(int i=start; i<end; i++) {
128                 if(getCodeByte(i) != getCodeByte(i-start+oldStart)) {
129                     return false;
130                 }
131             }
132             return true;
133         }
134     }
