22     import static org.junit.Assert.*;
23     
24     import java.io.File;
25     import java.util.concurrent.atomic.AtomicBoolean;
26     
27     import javax.annotation.Nonnull;
28     import javax.annotation.meta.TypeQualifierValidator;
29     import javax.annotation.meta.When;
30     
31     import org.junit.Test;
32     
33     import edu.umd.cs.findbugs.internalAnnotations.SlashedClassName;
34     
35     /**
36      * @author pugh
37      */
38     public class ValidationSecurityManagerTest {
39     
40     
41         private static final SlashedClassName ANNOTATION = AnnotationTemplate.class.getAnnotation(SlashedClassName.class);
42         static class BadValidator implements TypeQualifierValidator<SlashedClassName> {
43     
44             @Override
45             public @Nonnull
46             When forConstantValue(@Nonnull
47                     SlashedClassName annotation, Object value) {
48                 Thread  t = new Thread() {
49                     @Override
50                     public void run() {
51                         System.out.println("bang");
52                     }
53                 };
54                 t.start();
55                 return When.NEVER;
56             }
57     
58         }
59     
60         public When test(TypeQualifierValidator<SlashedClassName> validator) {
61             return ValidationSecurityManager.sandboxedValidation(ANNOTATION, validator, "java/lang/String");
62     
63         }
64     
65         public void havePermissions() throws InterruptedException {
66             final AtomicBoolean b = new AtomicBoolean(false);
67             Thread t = new Thread() {
68                 @Override
69                 public void run() {
70                     b.set(true);
71                 }
72             };
73             t.start();
74             t.join();
75             assertEquals(true, b.get());
76             for (File f :  File.listRoots()) {
77                 f.listFiles();
78             }
79         }
80     
81         @SlashedClassName static class AnnotationTemplate {}
82         @Test
83         public void test() throws InterruptedException {
84             ValidationSecurityManager sm = ValidationSecurityManager.INSTANCE;
85             SecurityManager old = System.getSecurityManager();
86             System.setSecurityManager(sm);
87             TypeQualifierValidator<SlashedClassName> goodValidator = new SlashedClassName.Checker();
88             havePermissions();
89             assertEquals(When.ALWAYS, test(goodValidator));
90             try {
91                 test(new BadValidator());
92                 fail("Should have thrown SecurityException");
93             } catch (SecurityException e) {
94                 assert true;
95             }
96             havePermissions();
97             System.setSecurityManager(old);
98             havePermissions();
99     
100     
101         }
