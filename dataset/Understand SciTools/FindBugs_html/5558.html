108         static {
109             String suffix;
110             if (RELEASE_CANDIDATE > 0) {
111                 suffix = "rc" + RELEASE_CANDIDATE;
112             } else if (PREVIEW > 0) {
113                 suffix = "preview" + PREVIEW;
114             } else {
115                 suffix = "dev-" + COMPUTED_ECLIPSE_DATE;
116                 if (!"Unknown".equals(GIT_REVISION)) {
117                     suffix += "-" + GIT_REVISION;
118                 }
119             }
120             RELEASE_SUFFIX_WORD = suffix;

138         private static final String COMPUTED_ECLIPSE_UI_VERSION = RELEASE_BASE + "." + COMPUTED_ECLIPSE_DATE;
139     
140         static {
141             Class<Version> c = Version.class;
142             URL u = c.getResource(c.getSimpleName() + ".class");
143             boolean fromFile = "file".equals(u.getProtocol());
144             InputStream in = null;
145             String release = null;
146             String date = null;
147             String plugin_release_date = null;
148             if (!fromFile) {
149                 try {
150                     Properties versionProperties = new Properties();
151                     in = Version.class.getResourceAsStream("version.properties");
152                     if (in != null)  {
153                         versionProperties.load(in);
154                         release = (String) versionProperties.get("release.number");
155                         date = (String) versionProperties.get("release.date");
156                         plugin_release_date =  (String) versionProperties.get("plugin.release.date");
157                     }
158                 } catch (Exception e) {
159                     assert true; // ignore
160                 } finally {
161                     Util.closeSilently(in);
162                 }
163             }
164             if (release == null) {
165                 release = COMPUTED_RELEASE;
166             }
167             if (date == null) {
168                 date = COMPUTED_DATE;
169             }
170             if (plugin_release_date == null) {
171                 plugin_release_date = COMPUTED_PLUGIN_RELEASE_DATE;
172             }
173     
174             RELEASE = release;
175             DATE = date;
176             CORE_PLUGIN_RELEASE_DATE = plugin_release_date;
177             Date parsedDate;
178             try {
179                 SimpleDateFormat fmt = new SimpleDateFormat(UpdateChecker.PLUGIN_RELEASE_DATE_FMT, Locale.ENGLISH);
180     
181                 parsedDate = fmt.parse(CORE_PLUGIN_RELEASE_DATE);
182             } catch (ParseException e) {
183                 if (SystemProperties.ASSERTIONS_ENABLED) {
184                     e.printStackTrace();
185                 }
186                 parsedDate = null;
187             }
188             releaseDate = parsedDate;
189         }

220         public static void main(String[] argv) throws InterruptedException {
221     
222             if (!IS_DEVELOPMENT && RELEASE_CANDIDATE != 0) {
223                 throw new IllegalStateException("Non developmental version, but is release candidate " + RELEASE_CANDIDATE);
224             }
225             if (argv.length == 0) {
226                 printVersion(false);
227                 return;
228             }
229     
230             String arg = argv[0];
231     
232             if ("-release".equals(arg)) {
233                 System.out.println(RELEASE);
234             } else if ("-date".equals(arg)) {
235                 System.out.println(DATE);
236             } else if ("-props".equals(arg)) {
237                 System.out.println("release.base=" + RELEASE_BASE);
238                 System.out.println("release.number=" + COMPUTED_RELEASE);
239                 System.out.println("release.date=" + COMPUTED_DATE);
240                 System.out.println("plugin.release.date=" + COMPUTED_PLUGIN_RELEASE_DATE);
241                 System.out.println("eclipse.ui.version=" + COMPUTED_ECLIPSE_UI_VERSION);
242                 System.out.println("findbugs.website=" + WEBSITE);
243                 System.out.println("findbugs.downloads.website=" + DOWNLOADS_WEBSITE);
244                 System.out.println("findbugs.git.revision=" + GIT_REVISION);
245             } else if ("-plugins".equals(arg)) {
246                 DetectorFactoryCollection.instance();
247                 for(Plugin p : Plugin.getAllPlugins()) {
248                     System.out.println("Plugin: " + p.getPluginId());
249                     System.out.println("  description: " + p.getShortDescription());
250                     System.out.println("     provider: " + p.getProvider());
251                     String version = p.getVersion();
252                     if (version != null && version.length() > 0) {
253                         System.out.println("      version: " + version);
254                     }
255                     String website = p.getWebsite();
256                     if (website != null && website.length() > 0) {
257                         System.out.println("      website: " + website);
258                     }
259                     System.out.println();
260                 }
261             } else if ("-configuration".equals(arg)){
262                 printVersion(true);
263             } else {
264     
265                 usage();
266                 System.exit(1);
267             }
268         }

285         /**
286          * @param justPrintConfiguration
287          * @throws InterruptedException
288          */
289         public static void printVersion(boolean justPrintConfiguration) throws InterruptedException {
290             System.out.println("FindBugs " + Version.COMPUTED_RELEASE);
291             if (justPrintConfiguration) {
292                 for (Plugin plugin : Plugin.getAllPlugins()) {
293                     System.out.printf("Plugin %s, version %s, loaded from %s%n", plugin.getPluginId(), plugin.getVersion(),
294                             plugin.getPluginLoader().getURL());
295                     if (plugin.isCorePlugin()) {
296                         System.out.println("  is core plugin");
297                     }
298                     if (plugin.isInitialPlugin()) {
299                         System.out.println("  is initial plugin");
300                     }
301                     if (plugin.isEnabledByDefault()) {
302                         System.out.println("  is enabled by default");
303                     }
304                     if (plugin.isGloballyEnabled()) {
305                         System.out.println("  is globally enabled");
306                     }
307                     Plugin parent = plugin.getParentPlugin();
308                     if (parent != null) {
309                         System.out.println("  has parent plugin " + parent.getPluginId());
310                     }
311     
312                     for (CloudPlugin cloudPlugin : plugin.getCloudPlugins()) {
313                         System.out.printf("  cloud %s%n", cloudPlugin.getId());
314                         System.out.printf("     %s%n", cloudPlugin.getDescription());
315                     }
316                     for (DetectorFactory factory : plugin.getDetectorFactories()) {
317                         System.out.printf("  detector %s%n", factory.getShortName());
318                     }
319                     System.out.println();
320                 }
321                 printPluginUpdates(true, 10);
322             } else {
323                 printPluginUpdates(false, 3);
324             }
325         }

327         private static void printPluginUpdates(boolean verbose, int secondsToWait) throws InterruptedException {
328             DetectorFactoryCollection dfc = DetectorFactoryCollection.instance();
329     
330             if (dfc.getUpdateChecker().updateChecksGloballyDisabled()) {
331                 if (verbose) {
332                     System.out.println();
333                     System.out.print("Update checking globally disabled");
334                 }
335                 return;
336             }
337             if (verbose) {
338                 System.out.println();
339                 System.out.print("Checking for plugin updates...");
340             }
341             FutureValue<Collection<UpdateChecker.PluginUpdate>>
342             updateHolder  = dfc.getUpdates();
343     
344             try {
345                 Collection<UpdateChecker.PluginUpdate> updates = updateHolder.get(secondsToWait, TimeUnit.SECONDS);
346                 if (updates.isEmpty()) {
347                     if (verbose) {
348                         System.out.println("none!");
349                     }
350                 } else {
351                     System.out.println();
352                     for (UpdateChecker.PluginUpdate update : updates) {
353                         System.out.println(update);
354                         System.out.println();
355     
356                     }
357                 }
358             } catch (TimeoutException e) {
359                 if (verbose) {
360                     System.out.println("Timeout while trying to get updates");
361                 }
362             }
363     
364         }
