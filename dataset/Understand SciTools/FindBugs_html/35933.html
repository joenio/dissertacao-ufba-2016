91         @Nonnull
92         private JavaSourceDocument getDocument(SourceLineAnnotation source) {
93             try {
94                 SourceFile sourceFile = frame.getProject().getSourceFinder().findSourceFile(source);
95                 String fullFileName = sourceFile.getFullFileName();
96                 SoftReference<JavaSourceDocument> resultReference = map.get(fullFileName);
97                 JavaSourceDocument result = null;
98                 if (resultReference != null) {
99                     result = resultReference.get();
100                 }
101                 if (result != null) {
102                     return result;
103                 }
104                 try {
105                     InputStream in = sourceFile.getInputStream();
106                     result = new JavaSourceDocument(source.getClassName(), SourceCharset.bufferedReader(in), sourceFile);
107                 } catch (Exception e) {
108                     result = JavaSourceDocument.UNKNOWNSOURCE;
109                     Debug.println(e); // e.printStackTrace();
110                 }
111                 map.put(fullFileName, new SoftReference<JavaSourceDocument>(result));
112                 return result;
113             } catch (Exception e) {
114                 Debug.println(e); // e.printStackTrace();
115                 return JavaSourceDocument.UNKNOWNSOURCE;
116     
117             }
118         }

120         @Override
121         public void run() {
122             while (true) {
123                 DisplayMe display;
124                 try {
125                     display = queue.take();
126                 } catch (InterruptedException e1) {
127                     assert false;
128                     Debug.println(e1);
129                     continue;
130                 }
131                 BugInstance myBug = display.bug;
132                 SourceLineAnnotation mySourceLine = display.source;
133     
134                 if (myBug == null || mySourceLine == null) {
135                     frame.clearSourcePane();
136                     continue;
137                 }
138     
139                 try {
140                     JavaSourceDocument src = getDocument(mySourceLine);
141                     this.myDocument = src;
142                     src.getHighlightInformation().clear();
143                     String primaryKind = mySourceLine.getDescription();
144                     // Display myBug and mySourceLine
145                     for (Iterator<BugAnnotation> i = myBug.annotationIterator(); i.hasNext();) {
146                         BugAnnotation annotation = i.next();
147                         if (annotation instanceof SourceLineAnnotation) {
148                             SourceLineAnnotation sourceAnnotation = (SourceLineAnnotation) annotation;
149                             if (sourceAnnotation == mySourceLine) {
150                                 continue;
151                             }
152                             if (sourceAnnotation.getDescription().equals(primaryKind)) {
153                                 highlight(src, sourceAnnotation, MAIN_HIGHLIGHT_MORE);
154                             } else {
155                                 highlight(src, sourceAnnotation, ALTERNATIVE_HIGHLIGHT);
156                             }
157                         }
158                     }
159                     highlight(src, mySourceLine, MAIN_HIGHLIGHT);
160                     javax.swing.SwingUtilities.invokeLater(new DisplayBug(src, myBug, mySourceLine));
161                 } catch (Exception e) {
162                     Debug.println(e); // e.printStackTrace();
163                 }
164             }
165         }

168         private final class DisplayBug implements Runnable {
169     
170             private final SourceLineAnnotation mySourceLine;
171             private final JavaSourceDocument src;
172             private final BugInstance myBug;
173     
174     
175             private DisplayBug(JavaSourceDocument src, BugInstance myBug, SourceLineAnnotation mySourceLine) {
176                 this.mySourceLine = mySourceLine;
177                 this.src = src;
178                 this.myBug = myBug;
179             }
180     
181             @Override
182             public void run() {
183                 frame.getSourceCodeTextPane().setEditorKit(src.getEditorKit());
184                 StyledDocument document = src.getDocument();
185                 frame.getSourceCodeTextPane().setDocument(document);
186                 String sourceFile = mySourceLine.getSourceFile();
187                 if (sourceFile == null || "<Unknown>".equals(sourceFile)) {
188                     sourceFile = mySourceLine.getSimpleClassName();
189                 }
190                 int startLine = mySourceLine.getStartLine();
191                 int endLine = mySourceLine.getEndLine();
192                 frame.setSourceTab(sourceFile + " in " + mySourceLine.getPackageName(), myBug);
193     
194                 int originLine = (startLine + endLine) / 2;
195                 LinkedList<Integer> otherLines = new LinkedList<Integer>();
196                 // show(frame.getSourceCodeTextPane(), document,
197                 // thisSource);
198                 for (Iterator<BugAnnotation> i = myBug.annotationIterator(); i.hasNext();) {
199                     BugAnnotation annotation = i.next();
200                     if (annotation instanceof SourceLineAnnotation) {
201                         SourceLineAnnotation sourceAnnotation = (SourceLineAnnotation) annotation;
202                         if (sourceAnnotation != mySourceLine) {
203                             // show(frame.getSourceCodeTextPane(),
204                             // document, sourceAnnotation);
205                             int otherLine = sourceAnnotation.getStartLine();
206                             if (otherLine > originLine) {
207                                 otherLine = sourceAnnotation.getEndLine();
208                             }
209                             otherLines.add(otherLine);
210                         }
211                     }
212                 }
213     
214                 if (startLine >= 0 && endLine >= 0) {
215                     frame.getSourceCodeTextPane().scrollLinesToVisible(startLine, endLine, otherLines);
216                 }
217             }
218         }

220         /**
221          * @param src
222          * @param sourceAnnotation
223          */
224         private void highlight(JavaSourceDocument src, SourceLineAnnotation sourceAnnotation, Color color) {
225     
226             int startLine = sourceAnnotation.getStartLine();
227             if (startLine == -1) {
228                 return;
229             }
230             String sourceFile = sourceAnnotation.getSourcePath();
231             String sourceFile2 = src.getSourceFile().getFullFileName();
232             if (!java.io.File.separator.equals(String.valueOf(SourceLineAnnotation.CANONICAL_PACKAGE_SEPARATOR))) {
233                 sourceFile2 = sourceFile2.replace(java.io.File.separatorChar, SourceLineAnnotation.CANONICAL_PACKAGE_SEPARATOR);
234             }
235             if (!sourceFile2.endsWith(sourceFile)) {
236                 return;
237             }
238             src.getHighlightInformation().setHighlight(startLine, sourceAnnotation.getEndLine(), color);
239         }

248         private int search(JavaSourceDocument document, String target, int start, Boolean backwards) {
249             if (document == null) {
250                 return -1;
251             }
252     
253             String docContent = null;
254             try {
255                 StyledDocument document2 = document.getDocument();
256                 if (document2 == null) {
257                     return -1;
258                 }
259                 docContent = document2.getText(0, document2.getLength());
260             } catch (BadLocationException ble) {
261                 System.out.println("Bad location exception");
262             } catch (NullPointerException npe) {
263                 return -1;
264             }
265             if (docContent == null) {
266                 return -1;
267             }
268             int targetLen = target.length();
269             int sourceLen = docContent.length();
270             if (targetLen > sourceLen) {
271                 return -1;
272             } else if (backwards) {
273                 for (int i = start; i >= 0; i--) {
274                     if (docContent.substring(i, i + targetLen).equals(target)) {
275                         return i;
276                     }
277                 }
278                 for (int i = (sourceLen - targetLen); i > start; i--) {
279                     if (docContent.substring(i, i + targetLen).equals(target)) {
280                         return i;
281                     }
282                 }
283                 return -1;
284             } else {
285                 for (int i = start; i <= (sourceLen - targetLen); i++) {
286                     if (docContent.substring(i, i + targetLen).equals(target)) {
287                         return i;
288                     }
289                 }
290                 for (int i = 0; i < start; i++) {
291                     if (docContent.substring(i, i + targetLen).equals(target)) {
292                         return i;
293                     }
294                 }
295                 return -1;
296             }
297         }

299         private int charToLineNum(int charNum) {
300             if (charNum == -1) {
301                 return -1;
302             }
303             try {
304                 for (int i = 1; true; i++) {
305                     if (frame.getSourceCodeTextPane().getLineOffset(i) > charNum) {
306                         return i - 1;
307                     } else if (frame.getSourceCodeTextPane().getLineOffset(i) == -1) {
308                         return -1;
309                     }
310                 }
311             } catch (BadLocationException ble) {
312                 return -1;
313             }
314         }
