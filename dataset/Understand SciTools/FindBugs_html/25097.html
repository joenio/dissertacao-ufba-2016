115         public ClassContext(JavaClass jclass, AnalysisContext analysisContext) {
116             this.jclass = jclass;

119             try {

127         public Map<MethodDescriptor, Object> getObjectMap(Class<?> analysisClass) {
128             Map<MethodDescriptor, Object> objectMap = methodAnalysisObjectMap.get(analysisClass);
129             if (objectMap == null) {
130                 if (analysisClass == ValueNumberDataflow.class) {
131                     objectMap = new MapCache<MethodDescriptor, Object>(300);
132                 } else if (Dataflow.class.isAssignableFrom(analysisClass)) {
133                     objectMap = new MapCache<MethodDescriptor, Object>(500);
134                 } else {
135                     objectMap = new HashMap<MethodDescriptor, Object>();
136                 }
137                 methodAnalysisObjectMap.put(analysisClass, objectMap);
138             }
139             return objectMap;
140         }

143          * Store a method analysis object. Note that the cached analysis object
144          * could be a special value (indicating null or an exception).
145          *
146          * @param analysisClass

148          * @param methodDescriptor

153         public void putMethodAnalysis(Class<?> analysisClass, MethodDescriptor methodDescriptor, Object object) {
154             if (object == null) {

179         /**
180          * Purge all CFG-based method analyses for given method.
181          *
182          * @param methodDescriptor
183          *            method descriptor identifying method to purge
184          */
185         public void purgeMethodAnalyses(MethodDescriptor methodDescriptor) {
186             Set<Map.Entry<Class<?>, Map<MethodDescriptor, Object>>> entrySet = methodAnalysisObjectMap.entrySet();
187             for (Iterator<Map.Entry<Class<?>, Map<MethodDescriptor, Object>>> i = entrySet.iterator(); i.hasNext();) {
188                 Map.Entry<Class<?>, Map<MethodDescriptor, Object>> entry = i.next();
189     
190                 Class<?> cls = entry.getKey();
191     
192                 // FIXME: hack
193                 if (!DataflowAnalysis.class.isAssignableFrom(cls) && !Dataflow.class.isAssignableFrom(cls)) {
194                     // There is really no need to purge analysis results
195                     // that aren't CFG-based.
196                     // Currently, only dataflow analyses need
197                     // to be purged.
198                     continue;
199                 }
200     
201                 entry.getValue().remove(methodDescriptor);
202             }
203         }

242         public @Nonnull
243         List<Method> getMethodsInCallOrder() {
244             Map<XMethod, Method> map = new HashMap<XMethod, Method>();
245             for (Method m : getJavaClass().getMethods()) {
246                 XMethod xMethod = classInfo.findMethod(m.getName(), m.getSignature(), m.isStatic());
247                 map.put(xMethod, m);
248             }
249             List<? extends XMethod> xmethodsInCallOrder = classInfo.getXMethodsInCallOrder();
250             List<Method> methodsInCallOrder = new ArrayList<Method>(xmethodsInCallOrder.size());
251             for (XMethod x : xmethodsInCallOrder) {
252                 Method m = map.get(x);
253                 if (m != null) {
254                     methodsInCallOrder.add(m);
255                 }
256             }
257             return methodsInCallOrder;
258         }

417         /**
418          * Get a BitSet representing the bytecodes that are used in the given
419          * method. This is useful for prescreening a method for the existence of
420          * particular instructions. Because this step doesn't require building a
421          * MethodGen, it is very fast and memory-efficient. It may allow a Detector
422          * to avoid some very expensive analysis, which is a Big Win for the user.
423          *
424          * @param method
425          *            the method
426          * @return the BitSet containing the opcodes which appear in the method, or
427          *         null if the method has no code
428          */
429         @CheckForNull
430         static public BitSet getBytecodeSet(JavaClass clazz, Method method) {
431     
432             XMethod xmethod = XFactory.createXMethod(clazz, method);
433             if (cachedBitsets().containsKey(xmethod)) {
434                 return cachedBitsets().get(xmethod);
435             }
436             Code code = method.getCode();
437             if (code == null) {
438                 return null;
439             }
440     
441             byte[] instructionList = code.getCode();
442     
443             // Create callback
444             UnpackedBytecodeCallback callback = new UnpackedBytecodeCallback(instructionList.length);
445     
446             // Scan the method.
447             BytecodeScanner scanner = new BytecodeScanner();
448             scanner.scan(instructionList, callback);
449     
450             UnpackedCode unpackedCode = callback.getUnpackedCode();
451             BitSet result = null;
452             if (unpackedCode != null) {
453                 result = unpackedCode.getBytecodeSet();
454             }
455             cachedBitsets().put(xmethod, result);
456             return result;
457         }

461         }

463         @Nonnull
464         static public Set<Integer> getLoopExitBranches(Method method, MethodGen methodGen) {
465     
466             XMethod xmethod = XFactory.createXMethod(methodGen);
467             if (cachedLoopExits().containsKey(xmethod)) {
468                 Set<Integer> result = cachedLoopExits().get(xmethod);
469                 if (result == null) {
470                     AnalysisContext.logError("Null cachedLoopExits for " + xmethod, new NullPointerException());
471                     assert false;
472                     return Collections.<Integer> emptySet();
473                 }
474                 return result;
475             }
476             Code code = method.getCode();
477             if (code == null) {
478                 assert false;
479                 return Collections.<Integer> emptySet();
480             }
481     
482             byte[] instructionList = code.getCode();
483     
484             Set<Integer> result = new HashSet<Integer>();
485             for (int i = 0; i < instructionList.length; i++) {
486                 if (checkForBranchExit(instructionList, i)) {
487                     result.add(i);
488                 }
489             }
490             if (result.size() == 0) {
491                 result = Collections.<Integer> emptySet();
492             }
493     
494             cachedLoopExits().put(xmethod, result);
495             return result;
496         }

500         }

502         static short getBranchOffset(byte[] codeBytes, int pos) {
503             int branchByte1 = 0xff & codeBytes[pos];
504             int branchByte2 = 0xff & codeBytes[pos + 1];
505             int branchOffset = (short) (branchByte1 << 8 | branchByte2);
506             return (short) branchOffset;
507     
508         }
509     
510         static boolean checkForBranchExit(byte[] codeBytes, int pos) {
511             if (pos < 0 || pos + 2 >= codeBytes.length) {
512                 return false;
513             }
514             switch (0xff & codeBytes[pos]) {
515             case Constants.IF_ACMPEQ:
516             case Constants.IF_ACMPNE:
517             case Constants.IF_ICMPEQ:
518             case Constants.IF_ICMPGE:
519             case Constants.IF_ICMPGT:
520             case Constants.IF_ICMPLE:
521             case Constants.IF_ICMPLT:
522             case Constants.IF_ICMPNE:
523                 break;
524             default:
525                 return false;
526             }
527             int branchTarget = pos + getBranchOffset(codeBytes, pos + 1);
528             if (branchTarget - 3 < pos || branchTarget >= codeBytes.length) {
529                 return false;
530             }
531             if ((codeBytes[branchTarget - 3] & 0xff) != Constants.GOTO) {
532                 return false;
533             }
534             int backBranchTarget = branchTarget + getBranchOffset(codeBytes, branchTarget - 2);
535             if (backBranchTarget <= pos && backBranchTarget + 12 >= pos) {
536                 return true;
537             }
538             return false;
539         }

766         public BitSet linesMentionedMultipleTimes(Method method) {
767             IAnalysisCache analysisCache = Global.getAnalysisCache();
768             XMethod xMethod = XFactory.createXMethod(jclass, method);
769             JumpInfo jumpInfo = null;
770             try {
771                 jumpInfo = analysisCache.getMethodAnalysis(JumpInfo.class, xMethod.getMethodDescriptor());
772             } catch (CheckedAnalysisException e) {
773                 AnalysisContext.logError("Error getting jump information", e);
774             }
775             BitSet lineMentionedMultipleTimes = new BitSet();
776             BitSet pcInFinallyBlock = new BitSet();
777     
778             Code code = method.getCode();
779             if (code == null) {
780                 return lineMentionedMultipleTimes;
781             }
782             CodeException[] exceptionTable = code.getExceptionTable();
783             if (exceptionTable == null || exceptionTable.length == 0) {
784                 return lineMentionedMultipleTimes;
785             }
786             int firstHandler = Integer.MAX_VALUE;
787             for (CodeException e : exceptionTable) {
788                 if (e.getCatchType() == 0) {
789                     int pc = e.getHandlerPC();
790                     firstHandler = Math.min(firstHandler, pc);
791                     if (jumpInfo != null) {
792                         int end = jumpInfo.getNextJump(pc + 1);
793                         if (end >= pc) {
794                             pcInFinallyBlock.set(pc, end);
795                         }
796                     }
797                 }
798             }
799             BitSet foundOnce = new BitSet();
800             BitSet afterHandler = new BitSet();
801             LineNumberTable lineNumberTable = method.getLineNumberTable();
802             int lineNum = -1;
803             int prevStartPc = -1;
804             if (lineNumberTable != null) {
805                 for (LineNumber line : lineNumberTable.getLineNumberTable()) {
806                     int newLine = line.getLineNumber();
807                     if (newLine == lineNum || newLine == -1) {
808                         continue;
809                     }
810                     if (prevStartPc >= 0) {
811                         int nextPcInFinallyBlock = pcInFinallyBlock.nextSetBit(prevStartPc);
812                         if (nextPcInFinallyBlock < line.getStartPC()) {
813                             lineMentionedMultipleTimes.set(lineNum);
814                         }
815                     }
816                     if (line.getStartPC() >= firstHandler) {
817                         afterHandler.set(lineNum);
818                     }
819     
820                     lineNum = newLine;
821                     prevStartPc = line.getStartPC();
822                     if (foundOnce.get(lineNum)) {
823                         lineMentionedMultipleTimes.set(lineNum);
824                     } else {
825                         foundOnce.set(lineNum);
826                     }
827                 }
828             }
829             lineMentionedMultipleTimes.and(afterHandler);
830             return lineMentionedMultipleTimes;
831         }

902                 @CheckForNull UnconditionalValueDerefDataflow dataflow, @CheckForNull TypeDataflow typeDataflow)
903                         throws DataflowAnalysisException {
904             System.out.println("\n\n{ UnconditionalValueDerefAnalysis analysis for " + method.getName());
905             TreeSet<Location> tree = new TreeSet<Location>();
906     
907             for (Iterator<Location> locs = cfg.locationIterator(); locs.hasNext();) {
908                 Location loc = locs.next();
909                 tree.add(loc);
910             }
911             for (Location loc : tree) {
912                 System.out.println();
913                 if (dataflow != null) {
914                     System.out.println("\n Pre: " + dataflow.getFactAfterLocation(loc));
915                 }
916                 System.out.println("Vna: " + vnd.getFactAtLocation(loc));
917                 System.out.println("inv: " + inv.getFactAtLocation(loc));
918                 if (typeDataflow != null) {
919                     System.out.println("type: " + typeDataflow.getFactAtLocation(loc));
920                 }
921                 System.out.println("Location: " + loc);
922                 if (dataflow != null) {
923                     System.out.println("Post: " + dataflow.getFactAtLocation(loc));
924                 }
925                 System.out.println("Vna: " + vnd.getFactAfterLocation(loc));
926                 System.out.println("inv: " + inv.getFactAfterLocation(loc));
927                 if (typeDataflow != null) {
928                     System.out.println("type: " + typeDataflow.getFactAfterLocation(loc));
929                 }
930             }
931             System.out.println("}\n\n");
932         }

998         private <Analysis> Analysis getMethodAnalysis(Class<Analysis> analysisClass, Method method) throws DataflowAnalysisException,
999         CFGBuilderException {
1000             try {
1001                 MethodDescriptor methodDescriptor = BCELUtil.getMethodDescriptor(jclass, method);
1002                 return Global.getAnalysisCache().getMethodAnalysis(analysisClass, methodDescriptor);
1003             } catch (DataflowAnalysisException e) {
1004                 throw e;
1005             } catch (CFGBuilderException e) {
1006                 throw e;
1007             } catch (CheckedAnalysisException e) {
1008                 Throwable cause = e.getCause();
1009                 if (cause instanceof CFGBuilderException) {
1010                     throw (CFGBuilderException) cause;
1011                 }
1012                 String message = "Should not happen: bad CAE: " + e.getClass().getName() + " for " + analysisClass.getName() + " of " + method;
1013                 IllegalStateException ise = new IllegalStateException(message);
1014                 ise.initCause(e);
1015                 throw ise;
1016             }
1017         }
