87         private static class LockFrameModelingVisitor extends ResourceValueFrameModelingVisitor {

96             public LockFrameModelingVisitor(ConstantPoolGen cpg, LockResourceTracker resourceTracker, Lock lock,
97                     ValueNumberDataflow vnaDataflow, IsNullValueDataflow isNullDataflow) {
98                 super(cpg);
99                 this.resourceTracker = resourceTracker;
100                 this.lock = lock;
101                 this.vnaDataflow = vnaDataflow;
102                 // this.isNullDataflow = isNullDataflow;
103             }

105             @Override
106             public void transferInstruction(InstructionHandle handle, BasicBlock basicBlock) throws DataflowAnalysisException {
107                 final Instruction ins = handle.getInstruction();
108                 final ConstantPoolGen cpg = getCPG();
109                 final ResourceValueFrame frame = getFrame();
110     
111                 int status = -1;
112     
113                 if (DEBUG) {
114                     System.out.println("PC : " + handle.getPosition() + " " + ins);
115                 }
116                 if (DEBUG && ins instanceof InvokeInstruction) {
117                     InvokeInstruction iins = (InvokeInstruction) ins;
118                     System.out.println("  " + ins.toString(cpg.getConstantPool()));
119                 }
120                 if (DEBUG) {
121                     System.out.println("resource frame before instruction: " + frame.toString());
122                 }
123     
124                 // Is a lock acquired or released by this instruction?
125                 Location creationPoint = lock.getLocation();
126                 if (handle == creationPoint.getHandle() && basicBlock == creationPoint.getBasicBlock()) {
127                     status = ResourceValueFrame.OPEN;
128                     if (DEBUG) {
129                         System.out.println("OPEN");
130                     }
131                 } else if (resourceTracker.isResourceClose(basicBlock, handle, cpg, lock, frame)) {
132                     status = ResourceValueFrame.CLOSED;
133                     if (DEBUG) {
134                         System.out.println("CLOSE");
135                     }
136                 }
137     
138                 // Model use of instance values in frame slots
139                 analyzeInstruction(ins);
140     
141                 final int updatedNumSlots = frame.getNumSlots();
142     
143                 // Mark any appearances of the lock value in the ResourceValueFrame.
144                 ValueNumberFrame vnaFrame = vnaDataflow.getFactAfterLocation(new Location(handle, basicBlock));
145                 if (DEBUG) {
146                     System.out.println("vna frame after instruction: " + vnaFrame.toString());
147                     System.out.println("Lock value number: " + lock.getLockValue());
148                     if (lock.getLockValue().hasFlag(ValueNumber.RETURN_VALUE)) {
149                         System.out.println("is return value");
150                     }
151                 }
152     
153                 for (int i = 0; i < updatedNumSlots; ++i) {
154                     if (DEBUG) {
155                         System.out.println("Slot " + i);
156                         System.out.println("  Lock value number: " + vnaFrame.getValue(i));
157                         if (vnaFrame.getValue(i).hasFlag(ValueNumber.RETURN_VALUE)) {
158                             System.out.println("  is return value");
159                         }
160                     }
161                     if (vnaFrame.fuzzyMatch(lock.getLockValue(), vnaFrame.getValue(i))) {
162                         if (DEBUG) {
163                             System.out.println("Saw lock value!");
164                         }
165                         frame.setValue(i, ResourceValue.instance());
166                     }
167                 }
168     
169                 // If needed, update frame status
170                 if (status != -1) {
171                     frame.setStatus(status);
172                 }
173                 if (DEBUG) {
174                     System.out.println("resource frame after instruction: " + frame.toString());
175                 }
176     
177             }

185         class LockResourceTracker implements ResourceTracker<Lock> {
186             private final RepositoryLookupFailureCallback lookupFailureCallback;
187     
188             private final CFG cfg;
189     
190             private final ValueNumberDataflow vnaDataflow;
191     
192             private final IsNullValueDataflow isNullDataflow;
193     
194             public LockResourceTracker(RepositoryLookupFailureCallback lookupFailureCallback, CFG cfg,
195                     ValueNumberDataflow vnaDataflow, IsNullValueDataflow isNullDataflow) {
196                 this.lookupFailureCallback = lookupFailureCallback;
197                 this.cfg = cfg;
198                 this.vnaDataflow = vnaDataflow;
199                 this.isNullDataflow = isNullDataflow;
200             }
201     
202             @Override
203             public Lock isResourceCreation(BasicBlock basicBlock, InstructionHandle handle, ConstantPoolGen cpg)
204                     throws DataflowAnalysisException {
205     
206                 InvokeInstruction inv = toInvokeInstruction(handle.getInstruction());
207                 if (inv == null) {
208                     return null;
209                 }
210     
211                 String className = inv.getClassName(cpg);
212                 String methodName = inv.getName(cpg);
213                 String methodSig = inv.getSignature(cpg);
214     
215                 try {
216                     if ("lock".equals(methodName) && "()V".equals(methodSig)
217                             && Hierarchy.isSubtype(className, "java.util.concurrent.locks.Lock")) {
218     
219                         Location location = new Location(handle, basicBlock);
220                         ValueNumberFrame frame = vnaDataflow.getFactAtLocation(location);
221                         ValueNumber lockValue = frame.getTopValue();
222                         if (DEBUG) {
223                             System.out.println("Lock value is " + lockValue.getNumber() + ", frame=" + frame.toString());
224                         }
225                         if (DEBUG) {
226                             ++numAcquires;
227                         }
228                         return new Lock(location, className, lockValue);
229                     }
230                 } catch (ClassNotFoundException e) {
231                     lookupFailureCallback.reportMissingClass(e);
232                 }
233                 return null;
234             }
235     
236             @Override
237             public boolean mightCloseResource(BasicBlock basicBlock, InstructionHandle handle, ConstantPoolGen cpg)
238                     throws DataflowAnalysisException {
239                 InvokeInstruction inv = toInvokeInstruction(handle.getInstruction());
240                 if (inv == null) {
241                     return false;
242                 }
243     
244                 String className = inv.getClassName(cpg);
245                 String methodName = inv.getName(cpg);
246                 String methodSig = inv.getSignature(cpg);
247     
248                 try {
249                     if ("unlock".equals(methodName) && "()V".equals(methodSig)
250                             && Hierarchy.isSubtype(className, "java.util.concurrent.locks.Lock")) {
251     
252                         return true;
253                     }
254                 } catch (ClassNotFoundException e) {
255                     lookupFailureCallback.reportMissingClass(e);
256                 }
257     
258                 return false;
259             }
260     
261             @Override
262             public boolean isResourceClose(BasicBlock basicBlock, InstructionHandle handle, ConstantPoolGen cpg, Lock resource,
263                     ResourceValueFrame frame) throws DataflowAnalysisException {
264     
265                 if (!mightCloseResource(basicBlock, handle, cpg)) {
266                     return false;
267                 }
268                 ResourceValue topValue = frame.getTopValue();
269                 return topValue.isInstance();
270     
271             }

285             @Override
286             public boolean ignoreExceptionEdge(Edge edge, Lock resource, ConstantPoolGen cpg) {
287     
288                 try {
289                     Location location = cfg.getExceptionThrowerLocation(edge);
290                     if (DEBUG) {
291                         System.out.println("Exception thrower location: " + location);
292                     }
293                     Instruction ins = location.getHandle().getInstruction();
294     
295                     if (ins instanceof GETFIELD) {
296                         GETFIELD insGetfield = (GETFIELD) ins;
297                         String fieldName = insGetfield.getFieldName(cpg);
298                         if (DEBUG) {
299                             System.out.println("Inspecting GETFIELD of " + fieldName + " at " + location);
300                         }
301                         // Ignore exceptions from getfield instructions where the
302                         // object reference is known not to be null
303                         if ("lock".equals(fieldName)) {
304                             return true;
305                         }
306                         IsNullValueFrame frame = isNullDataflow.getFactAtLocation(location);
307                         if (!frame.isValid()) {
308                             return false;
309                         }
310                         IsNullValue receiver = frame.getInstance(ins, cpg);
311                         boolean notNull = receiver.isDefinitelyNotNull();
312                         if (DEBUG && notNull) {
313                             System.out.println("Ignoring exception from non-null GETFIELD");
314                         }
315                         return notNull;
316                     } else if (ins instanceof InvokeInstruction) {
317                         InvokeInstruction iins = (InvokeInstruction) ins;
318                         String methodName = iins.getMethodName(cpg);
319                         // System.out.println("Method " + methodName);
320                         if (methodName.startsWith("access$")) {
321                             return true;
322                         }
323                         if ("readLock".equals(methodName) || "writeLock".equals(methodName)) {
324                             return true;
325                         }
326                         if ("lock".equals(methodName) || "unlock".equals(methodName)) {
327                             return true;
328                         }
329                     }
330                     if (DEBUG) {
331                         System.out.println("FOUND Exception thrower at: " + location);
332                     }
333                 } catch (DataflowAnalysisException e) {
334                     AnalysisContext.logError("Error while looking for exception edge", e);
335                 }
336     
337                 return false;
338             }

344                 return false;

347             private InvokeInstruction toInvokeInstruction(Instruction ins) {
348                 short opcode = ins.getOpcode();
349                 if (opcode != Constants.INVOKEVIRTUAL && opcode != Constants.INVOKEINTERFACE) {
350                     return null;
351                 }
352                 return (InvokeInstruction) ins;
353             }

367          * (non-Javadoc)
368          *
369          * @see
370          * edu.umd.cs.findbugs.Detector#visitClassContext(edu.umd.cs.findbugs.ba
371          * .ClassContext)
372          */
373         @Override
374         public void visitClassContext(ClassContext classContext) {
375             JavaClass jclass = classContext.getJavaClass();
376     
377             // We can ignore classes that were compiled for anything
378             // less than JDK 1.5. This should avoid lots of unnecessary work
379             // when analyzing code for older VM targets.
380             if (BCELUtil.preTiger(jclass)) {
381                 return;
382             }
383     
384             boolean sawUtilConcurrentLocks = false;
385             for (Constant c : jclass.getConstantPool().getConstantPool()) {
386                 if (c instanceof ConstantMethodref) {
387                     ConstantMethodref m = (ConstantMethodref) c;
388                     ConstantClass cl = (ConstantClass) jclass.getConstantPool().getConstant(m.getClassIndex());
389                     ConstantUtf8 name = (ConstantUtf8) jclass.getConstantPool().getConstant(cl.getNameIndex());
390                     String nameAsString = name.getBytes();
391                     if (nameAsString.startsWith("java/util/concurrent/locks")) {
392                         sawUtilConcurrentLocks = true;
393                     }
394     
395                 }
396             }
397             if (sawUtilConcurrentLocks) {
398                 super.visitClassContext(classContext);
399             }
400         }

402         @Override
403         public boolean prescreen(ClassContext classContext, Method method, boolean mightClose) {
404             if (!mightClose) {
405                 return false;
406             }
407             BitSet bytecodeSet = classContext.getBytecodeSet(method);
408             if (bytecodeSet == null) {
409                 return false;
410             }
411     
412             MethodGen methodGen = classContext.getMethodGen(method);
413     
414             return methodGen != null && methodGen.getName().toLowerCase().indexOf("lock") == -1
415                     && (bytecodeSet.get(Constants.INVOKEVIRTUAL) || bytecodeSet.get(Constants.INVOKEINTERFACE));
416         }

425         @Override
426         public void inspectResult(ClassContext classContext, MethodGen methodGen, CFG cfg,
427                 Dataflow<ResourceValueFrame, ResourceValueAnalysis<Lock>> dataflow, Lock resource) {
428     
429             JavaClass javaClass = classContext.getJavaClass();
430     
431             ResourceValueFrame exitFrame = dataflow.getResultFact(cfg.getExit());
432             if (DEBUG) {
433                 System.out.println("Resource value at exit: " + exitFrame);
434             }
435             int exitStatus = exitFrame.getStatus();
436     
437             if (exitStatus == ResourceValueFrame.OPEN || exitStatus == ResourceValueFrame.OPEN_ON_EXCEPTION_PATH) {
438                 String bugType;
439                 int priority;
440                 if (exitStatus == ResourceValueFrame.OPEN) {
441                     bugType = "UL_UNRELEASED_LOCK";
442                     priority = HIGH_PRIORITY;
443                 } else {
444                     bugType = "UL_UNRELEASED_LOCK_EXCEPTION_PATH";
445                     priority = NORMAL_PRIORITY;
446                 }
447     
448                 String sourceFile = javaClass.getSourceFileName();
449                 Location location = resource.getLocation();
450                 InstructionHandle handle = location.getHandle();
451                 InstructionHandle nextInstruction = handle.getNext();
452                 if (nextInstruction.getInstruction() instanceof RETURN)
453                 {
454                     return; // don't report as error; intentional
455                 }
456                 bugAccumulator.accumulateBug(new BugInstance(this, bugType, priority).addClassAndMethod(methodGen, sourceFile),
457                         SourceLineAnnotation.fromVisitedInstruction(classContext, methodGen, sourceFile, handle));
458             }
459         }
