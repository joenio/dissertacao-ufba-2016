20     package edu.umd.cs.findbugs.classfile;
21     
22     import java.io.IOException;
23     import java.io.InputStream;
24     
25     /**
26      * Object representing a resource in a code base.
27      *
28      * @author David Hovemeyer
29      */
30     public interface ICodeBaseEntry {
31         /**
32          * Get the name of the resource.
33          *
34          * @return the name of the resource
35          */
36         public String getResourceName();
37     
38         /**
39          * Get the number of bytes in the resource. Returns &lt;0 if the number of
40          * bytes is not known.
41          *
42          * @return number of bytes in the resource, or &lt;0 if not known.
43          */
44         public int getNumBytes();
45     
46         /**
47          * Open an input stream reading from the resource.
48          *
49          * @return InputStream reading from the resource.
50          * @throws IOException
51          *             if an error occurs reading from the resource
52          */
53         public InputStream openResource() throws IOException;
54     
55         /**
56          * Get the codebase this codebase entry belongs to.
57          *
58          * @return the codebase this codebase entry belongs to
59          */
60         public ICodeBase getCodeBase();
61     
62         /**
63          * Return the ClassDescriptor of the class resource accessed by this
64          * codebase entry. Do not call this method unless
65          * ClassDescriptor.isClassResource() returns true. This method may require
66          * the class data to be loaded in order to determine the class.
67          *
68          * @return ClassDescriptor of this entry
69          * @throws ResourceNotFoundException, InvalidClassFileFormatException
70          *             if the codebase entry does not reference a valid classfile
71          * @throws IllegalArgumentException
72          *             if the codebase entry's filename is definitely not a
73          *             classfile
74          */
75         public ClassDescriptor getClassDescriptor() throws ResourceNotFoundException, InvalidClassFileFormatException;
76     
77         /**
78          * Override the resource name of this codebase entry.
79          *
80          * @param resourceName
81          *            the new resource name
82          */
83         public void overrideResourceName(String resourceName);
84     }
