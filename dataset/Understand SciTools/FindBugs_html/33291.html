19     package edu.umd.cs.findbugs;
20     
21     import java.io.IOException;
22     import java.util.Iterator;
23     
24     import org.dom4j.DocumentException;
25     
26     @Deprecated
27     public class NewResults {
28         private final SortedBugCollection origCollection;
29     
30         private final SortedBugCollection newCollection;
31     
32         public NewResults(String origFilename, String newFilename) throws IOException, DocumentException {
33             this(new SortedBugCollection(), new SortedBugCollection());
34             origCollection.readXML(origFilename);
35             newCollection.readXML(newFilename);
36         }
37     
38         public NewResults(SortedBugCollection origCollection, SortedBugCollection newCollection) {
39             this.origCollection = origCollection;
40             this.newCollection = newCollection;
41         }
42     
43         public SortedBugCollection execute() {
44             SortedBugCollection result = new SortedBugCollection();
45     
46             for (Iterator<BugInstance> i = newCollection.iterator(); i.hasNext();) {
47                 BugInstance bugInstance = i.next();
48     
49                 if (!origCollection.contains(bugInstance)) {
50                     result.add(bugInstance);
51                 }
52             }
53     
54             return result;
55         }
56     
57         public static void main(String[] argv) throws Exception {
58             if (argv.length != 3) {
59                 System.err.println("Usage: " + NewResults.class.getName() + " <orig results> <new results> <output file>");
60                 System.exit(1);
61             }
62     
63             String origFilename = argv[0];
64             String newFilename = argv[1];
65             String outputFilename = argv[2];
66     
67             NewResults op = new NewResults(origFilename, newFilename);
68     
69             SortedBugCollection result = op.execute();
70     
71             result.writeXML(outputFilename);
72         }
73     }
