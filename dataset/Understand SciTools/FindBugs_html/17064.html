20     package edu.umd.cs.findbugs.detect;
21     
22     import org.apache.bcel.classfile.Code;
23     import org.apache.bcel.classfile.JavaClass;
24     import org.apache.bcel.classfile.Method;
25     
26     import edu.umd.cs.findbugs.BugInstance;
27     import edu.umd.cs.findbugs.BugReporter;
28     import edu.umd.cs.findbugs.BytecodeScanningDetector;
29     import edu.umd.cs.findbugs.StatelessDetector;
30     import edu.umd.cs.findbugs.ba.ClassContext;
31     
32     /**
33      * finds public classes that use 'this' as a semaphore, which can cause
34      * conflicts if clients of this class use an instance of this class as their own
35      * synchronization point. Frankly, Just calling synchronized on this, or
36      * defining synchronized methods is bad, but since that is so prevalent, don't
37      * warn on that.
38      */
39     public class PublicSemaphores extends BytecodeScanningDetector implements StatelessDetector {
40         private static final int SEEN_NOTHING = 0;
41     
42         private static final int SEEN_ALOAD_0 = 1;
43     
44         private final BugReporter bugReporter;
45     
46         private int state;
47     
48         private boolean alreadyReported;
49     
50         public PublicSemaphores(BugReporter bugReporter) {
51             this.bugReporter = bugReporter;
52         }
53     
54         @Override
55         public void visitClassContext(ClassContext classContext) {
56             JavaClass cls = classContext.getJavaClass();
57             if ((!cls.isPublic()) || (cls.getClassName().indexOf('$') >= 0)) {
58                 return;
59             }
60     
61             alreadyReported = false;
62             super.visitClassContext(classContext);
63         }
64     
65         @Override
66         public void visit(Code obj) {
67             Method m = getMethod();
68             if (m.isStatic() || alreadyReported) {
69                 return;
70             }
71     
72             state = SEEN_NOTHING;
73             super.visit(obj);
74         }
75     
76         @Override
77         public void sawOpcode(int seen) {
78             if (alreadyReported) {
79                 return;
80             }
81     
82             switch (state) {
83             case SEEN_NOTHING:
84                 if (seen == ALOAD_0) {
85                     state = SEEN_ALOAD_0;
86                 }
87                 break;
88     
89             case SEEN_ALOAD_0:
90                 if ((seen == INVOKEVIRTUAL) && "java/lang/Object".equals(getClassConstantOperand())) {
91                     String methodName = getNameConstantOperand();
92                     if ("wait".equals(methodName) || "notify".equals(methodName) || "notifyAll".equals(methodName)) {
93                         bugReporter.reportBug(new BugInstance(this, "PS_PUBLIC_SEMAPHORES", NORMAL_PRIORITY).addClassAndMethod(this)
94                                 .addSourceLine(this));
95                         alreadyReported = true;
96                     }
97                 }
98                 state = SEEN_NOTHING;
99                 break;
100             default:
101                 break;
102             }
103     
104         }
105     
106     }
