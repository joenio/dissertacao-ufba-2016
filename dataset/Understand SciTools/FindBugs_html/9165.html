18     package edu.umd.cs.findbugs.workflow;
19     
20     import java.io.IOException;
21     import java.io.PrintWriter;
22     import java.util.Date;
23     
24     import javax.annotation.CheckForNull;
25     
26     import org.dom4j.DocumentException;
27     
28     import edu.umd.cs.findbugs.AppVersion;
29     import edu.umd.cs.findbugs.DetectorFactoryCollection;
30     import edu.umd.cs.findbugs.FindBugs;
31     import edu.umd.cs.findbugs.Priorities;
32     import edu.umd.cs.findbugs.ProjectStats;
33     import edu.umd.cs.findbugs.SortedBugCollection;
34     import edu.umd.cs.findbugs.charsets.UTF8;
35     import edu.umd.cs.findbugs.config.CommandLine;
36     
37     /**
38      * Java main application to compute update a historical bug collection with
39      * results from another build/analysis.
40      *
41      * @author William Pugh
42      */
43     
44     public class ListBugDatabaseInfo {
45     
46         private static final String USAGE = "Usage: " + ListBugDatabaseInfo.class.getName()
47                 + " [options] data1File data2File data3File ... ";
48     
49         static class ListBugDatabaseInfoCommandLine extends CommandLine {
50     
51             boolean formatDates = false;
52     
53             ListBugDatabaseInfoCommandLine() {
54                 addSwitch("-formatDates", "render dates in textual form");
55             }
56     
57             @Override
58             public void handleOption(String option, String optionalExtraPart) {
59                 if ("-formatDates".equals(option)) {
60                     formatDates = true;
61                 } else {
62                     throw new IllegalArgumentException("unknown option: " + option);
63                 }
64             }
65     
66             @Override
67             public void handleOptionWithArgument(String option, String argument) {
68     
69                 throw new IllegalArgumentException("unknown option: " + option);
70             }
71         }
72     
73         public static void main(String[] args) throws IOException, DocumentException {
74             FindBugs.setNoAnalysis();
75             DetectorFactoryCollection.instance();
76             ListBugDatabaseInfoCommandLine commandLine = new ListBugDatabaseInfoCommandLine();
77             int argCount = commandLine.parse(args, 0, Integer.MAX_VALUE, USAGE);
78     
79             PrintWriter out = UTF8.printWriter(System.out, true);
80             if (argCount == args.length) {
81                 listVersion(out, null, commandLine.formatDates);
82             } else {
83                 out.println("version\ttime\tclasses\tNCSS\terrors\ttotal\thigh\tmedium\tlow\tfile");
84                 while (argCount < args.length) {
85                     String fileName = args[argCount++];
86                     listVersion(out, fileName, commandLine.formatDates);
87                 }
88             }
89             out.close();
90         }
91     
92         private static void listVersion(PrintWriter out, @CheckForNull String fileName, boolean formatDates) throws IOException,
93         DocumentException {
94             SortedBugCollection origCollection;
95             origCollection = new SortedBugCollection();
96     
97             if (fileName == null) {
98                 origCollection.readXML(System.in);
99             } else {
100                 origCollection.readXML(fileName);
101             }
102             AppVersion appVersion = origCollection.getCurrentAppVersion();
103             ProjectStats stats = origCollection.getProjectStats();
104             out.print(appVersion.getReleaseName());
105             out.print('\t');
106             if (formatDates) {
107                 out.print("\"" + new Date(appVersion.getTimestamp()) + "\"");
108             } else {
109                 out.print(appVersion.getTimestamp());
110             }
111             out.print('\t');
112     
113             out.print(appVersion.getNumClasses());
114             out.print('\t');
115             out.print(appVersion.getCodeSize());
116             out.print('\t');
117             out.print(origCollection.getErrors().size());
118             out.print('\t');
119             out.print(stats.getTotalBugs());
120             out.print('\t');
121             out.print(stats.getBugsOfPriority(Priorities.HIGH_PRIORITY));
122             out.print('\t');
123             out.print(stats.getBugsOfPriority(Priorities.NORMAL_PRIORITY));
124             out.print('\t');
125             out.print(stats.getBugsOfPriority(Priorities.LOW_PRIORITY));
126             if (fileName != null) {
127                 out.print('\t');
128                 out.print(fileName);
129             }
130     
131             out.println();
132         }
133     
134     }
