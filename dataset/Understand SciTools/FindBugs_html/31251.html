55         public BugFilingCommentHelper(Cloud cloud) {
56             this.cloud = cloud;
57             PropertyBundle properties = cloud.getPlugin().getProperties();
58             BUG_NOTE = properties.getProperty("findbugs.bugnote");
59             POSTMORTEM_NOTE = properties.getProperty("findbugs.postmortem.note");
60             POSTMORTEM_RANK = properties.getInt("findbugs.postmortem.maxRank", 4);
61         }

72         @SuppressWarnings("boxing")
73         public String getBugReportSourceCode(BugInstance b) {
74             StringWriter stringWriter = new StringWriter();
75             PrintWriter out = new PrintWriter(stringWriter);
76             ClassAnnotation primaryClass = b.getPrimaryClass();
77     
78             int firstLine = Integer.MAX_VALUE;
79             int lastLine = Integer.MIN_VALUE;
80             for (BugAnnotation a : b.getAnnotations()) {
81                 if (a instanceof SourceLineAnnotation) {
82                     SourceLineAnnotation s = (SourceLineAnnotation) a;
83                     if (s.getClassName().equals(primaryClass.getClassName()) && s.getStartLine() > 0) {
84                         firstLine = Math.min(firstLine, s.getStartLine());
85                         lastLine = Math.max(lastLine, s.getEndLine());
86     
87                     }
88     
89                 }
90             }
91     
92             SourceLineAnnotation primarySource = primaryClass.getSourceLines();
93             if (primarySource.isSourceFileKnown() && firstLine >= 1 && firstLine <= lastLine && lastLine - firstLine < 50) {
94                 BufferedReader in = null;
95                 try {
96                     Project project = cloud.getBugCollection().getProject();
97                     SourceFile sourceFile = project.getSourceFinder().findSourceFile(primarySource);
98                     in = UTF8.bufferedReader(sourceFile.getInputStream());
99                     int lineNumber = 1;
100                     String commonWhiteSpace = null;
101                     List<SourceLine> source = new ArrayList<SourceLine>();
102                     while (lineNumber <= lastLine + 4) {
103                         String txt = in.readLine();
104                         if (txt == null) {
105                             break;
106                         }
107                         if (lineNumber >= firstLine - 4) {
108                             String trimmed = txt.trim();
109                             if (trimmed.length() == 0) {
110                                 if (lineNumber > lastLine) {
111                                     break;
112                                 }
113                                 txt = trimmed;
114     
115                             }
116                             source.add(new SourceLine(lineNumber, txt));
117                             commonWhiteSpace = commonLeadingWhitespace(commonWhiteSpace, txt);
118                         }
119                         lineNumber++;
120                     }
121                     if (commonWhiteSpace == null) {
122                         commonWhiteSpace = "";
123                     }
124                     out.println("\nRelevant source code:");
125                     for (SourceLine s : source) {
126                         if (s.text.length() == 0) {
127                             out.printf("%5d: %n", s.line);
128                         } else {
129                             out.printf("%5d:   %s%n", s.line, s.text.substring(commonWhiteSpace.length()));
130                         }
131                     }
132     
133                     out.println();
134                 } catch (IOException e) {
135                     assert true;
136                 } finally {
137                     Util.closeSilently(in);
138                 }
139                 out.close();
140                 return stringWriter.toString();
141     
142             }
143             return "";
144         }

146         public String getBugReportHead(BugInstance b) {
147             StringWriter stringWriter = new StringWriter();
148             PrintWriter out = new PrintWriter(stringWriter);
149             out.println("Bug report generated from FindBugs");
150             out.println(b.getMessageWithoutPrefix());
151             out.println();
152             ClassAnnotation primaryClass = b.getPrimaryClass();
153     
154             for (BugAnnotation a : b.getAnnotations()) {
155                 if (a == primaryClass) {
156                     out.println(a);
157                 } else {
158                     out.println("  " + a.toString(primaryClass));
159                 }
160             }
161             if (cloud.supportsSourceLinks()) {
162                 URL link = cloud.getSourceLink(b);
163     
164                 if (link != null) {
165                     out.println();
166                     out.println(cloud.getSourceLinkToolTip(b) + ": " + link);
167                     out.println();
168                 }
169             }
170     
171             if (BUG_NOTE != null) {
172                 out.println(BUG_NOTE);
173                 if (POSTMORTEM_NOTE != null && BugRanker.findRank(b) <= POSTMORTEM_RANK
174                         && cloud.getConsensusDesignation(b).score() >= 0) {
175     
176                     out.println(POSTMORTEM_NOTE);
177                 }
178                 out.println();
179             }
180     
181             Collection<String> projects = cloud.getProjects(primaryClass.getClassName());
182             if (projects != null && !projects.isEmpty()) {
183                 String projectList = projects.toString();
184                 projectList = projectList.substring(1, projectList.length() - 1);
185                 out.println("Possibly part of: " + projectList);
186                 out.println();
187             }
188             out.close();
189             return stringWriter.toString();
190         }

192         public String getBugPatternExplanation(BugInstance b) {
193             String detailPlainText = b.getBugPattern().getDetailPlainText();
194             return "Bug pattern explanation:\n" + detailPlainText + "\n\n";
195         }
196     
197         public String getLineTerminatedUserEvaluation(BugInstance b) {
198             UserDesignation designation = cloud.getUserDesignation(b);
199     
200             String result;
201             if (designation != UserDesignation.UNCLASSIFIED) {
202                 result = "Classified as: " + designation.toString() + "\n";
203             } else {
204                 result = "";
205             }
206             String eval = cloud.getUserEvaluation(b).trim();
207             if (eval.length() > 0) {
208                 result = result + eval + "\n";
209             }
210             return result;
211         }
212     
213         public String getBugReportTail(BugInstance b) {
214             return "\nFindBugs issue identifier (do not modify or remove): " + b.getInstanceHash();
215         }

220         private String commonLeadingWhitespace(String soFar, String txt) {
221             if (txt.length() == 0) {
222                 return soFar;
223             }
224             if (soFar == null) {
225                 return txt;
226             }
227             soFar = Util.commonPrefix(soFar, txt);
228             for (int i = 0; i < soFar.length(); i++) {
229                 if (!Character.isWhitespace(soFar.charAt(i))) {
230                     return soFar.substring(0, i);
231                 }
232             }
233             return soFar;
234         }
