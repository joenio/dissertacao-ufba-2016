34     /**
35      * This is where it all begins run with -f int to set font size run with -clear
36      * to clear recent projects menu, or any other issues with program not starting
37      * properly due to something being corrupted (or just faulty) in backend store
38      * for GUISaveState.
39      *
40      */
41     public class Driver {
42         /** Force start time to be computed */
43         private static long START_TIME = StartTime.START_TIME;
44     
45         private static final String USAGE = Driver.class.getName() + " [options] [project or analysis results file]";
46     
47         private static GUI2CommandLine commandLine = new GUI2CommandLine();
48     
49         private static SplashFrame splash;
50     
51         public static void main(String[] args) throws Exception {
52             try {
53     
54                 String name = "FindBugs GUI";
55                 if (JavaWebStart.isRunningViaJavaWebstart()) {
56                     name = "FindBugs webstart GUI";
57                 }
58                 Version.registerApplication(name, Version.RELEASE);
59     
60                 if (SystemProperties.getProperty("os.name").startsWith("Mac")) {
61                     System.setProperty("apple.laf.useScreenMenuBar", "true");
62                     System.setProperty("com.apple.mrj.application.apple.menu.about.name", "FindBugs");
63                     Debug.println("Mac OS detected");
64                 }
65                 splash = new SplashFrame();
66                 splash.setVisible(true);
67     
68                 int numParsed = commandLine.parse(args, 0, 1, USAGE);
69     
70                 //
71                 // See if an argument filename was specified after the parsed
72                 // options/switches.
73                 //
74                 if (numParsed < args.length) {
75                     String arg = args[numParsed];
76                     String argLowerCase = arg.toLowerCase(Locale.ENGLISH);
77                     if (argLowerCase.endsWith(".fbp") || argLowerCase.endsWith(".fb")) {
78                         // Project file specified
79                         commandLine.loadProject(arg);
80                     } else if (argLowerCase.endsWith(".xml") || argLowerCase.endsWith(".xml.gz") || argLowerCase.endsWith(".fba")) {
81                         // Saved analysis results specified
82                         commandLine.setSaveFile(new File(arg));
83                     } else {
84                         System.out.println("Unknown argument: " + arg);
85                         commandLine.printUsage(System.out);
86                         System.exit(1);
87                     }
88                 }
89     
90                 if (commandLine.getDocking()) {
91                     // make sure docking runtime support is available
92                     try {
93                         Class.forName("net.infonode.docking.DockingWindow");
94                         Class.forName("edu.umd.cs.findbugs.gui2.DockLayout");
95                     } catch (Exception e) {
96                         commandLine.setDocking(false);
97                     }
98                 }
99     
100                 try {
101                     GUISaveState.loadInstance();
102                 } catch (RuntimeException e) {
103                     GUISaveState.clear();
104                     e.printStackTrace();
105                 }
106     
107                 GUISaveState guiSavedPreferences = GUISaveState.getInstance();
108                 if (commandLine.isFontSizeSpecified()) {
109                     guiSavedPreferences.setFontSize(commandLine.getFontSize());
110                 }
111     
112                 // System.setProperty("findbugs.home",".."+File.separator+"findbugs");
113     
114                 enablePlugins(guiSavedPreferences.getEnabledPlugins(), true);
115                 enablePlugins(guiSavedPreferences.getDisabledPlugins(), false);
116     
117                 // The bug with serializable idiom detection has been fixed on the
118                 // findbugs end.
119                 // DetectorFactory
120                 // serializableIdiomDetector=DetectorFactoryCollection.instance().getFactory("SerializableIdiom");
121                 // System.out.println(serializableIdiomDetector.getFullName());
122                 // UserPreferences.getUserPreferences().enableDetector(serializableIdiomDetector,false);
123     
124                 FindBugsLayoutManagerFactory factory;
125     
126                 if (isDocking()) {
127                     factory = new FindBugsLayoutManagerFactory("edu.umd.cs.findbugs.gui2.DockLayout");
128                 } else {
129                     factory = new FindBugsLayoutManagerFactory(SplitLayout.class.getName());
130                 }
131                 MainFrame.makeInstance(factory);
132     
133     
134                 splash.setVisible(false);
135                 splash.dispose();
136             } catch (Throwable t) {
137                 JOptionPane.showMessageDialog(null, t.toString(), "Fatal Error during FindBugs startup", JOptionPane.ERROR_MESSAGE);
138                 t.printStackTrace(System.err);
139                 System.exit(1);
140             }
141         }
142     
143         private static void enablePlugins(Iterable<String> plugins, boolean enabled) {
144             for (String pid : plugins) {
145                 Plugin plugin = Plugin.getByPluginId(pid);
146                 if (plugin != null) {
147                     if (!enabled && plugin.cannotDisable()) {
148                         JOptionPane.showMessageDialog(null,
149                                 "Cannot disable plugin: " + plugin.getPluginId() + "\n" + plugin.getShortDescription(),
150                                 "Cannot disable plugin", JOptionPane.ERROR_MESSAGE);
151                     } else {
152                         plugin.setGloballyEnabled(enabled);
153                     }
154                 }
155             }
156         }
157     
158         public static void removeSplashScreen() {
159             if (splash == null) {
160                 return;
161             }
162             splash.setVisible(false);
163             splash.dispose();
164     
165             if (commandLine.getSaveFile() != null) {
166                 MainFrame.getInstance().openAnalysis(commandLine.getSaveFile(), SaveType.XML_ANALYSIS);
167             } else if (commandLine.isProjectLoadedFromFile()) {
168                 MainFrame.getInstance().setProject(commandLine.getProject());
169                 MainFrame.getInstance().newProject();
170     
171                 MainFrame.getInstance().redoAnalysis();
172             }
173         }

185         }

188             return commandLine.getSettingList();
189         }
