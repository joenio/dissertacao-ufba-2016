149         /**
150          * Determine whether redundant load elimination should be performed for the
151          * heap location referenced by the current instruction.
152          *
153          * @return true if we should do redundant load elimination for the current
154          *         instruction, false if not
155          */
156         private boolean doRedundantLoadElimination() {
157             if (!REDUNDANT_LOAD_ELIMINATION) {
158                 return false;
159             }
160     
161             XField xfield = loadedFieldSet.getField(handle);
162             if (xfield == null) {
163                 return false;
164             }
165     
166             if(xfield.getSignature().equals("D") || xfield.getSignature().equals("J")) {
167                 // TODO: support two-slot fields
168                 return false;
169             }
170     
171             // Don't do redundant load elimination for fields that
172             // are loaded in only one place.
173             /*
174             if (false && loadedFieldSet.getLoadStoreCount(xfield).getLoadCount() <= 1){
175                 return false;
176             }
177              */
178             return true;
179         }
180     
181         /**
182          * Determine whether forward substitution should be performed for the heap
183          * location referenced by the current instruction.
184          *
185          * @return true if we should do forward substitution for the current
186          *         instruction, false if not
187          */
188         private boolean doForwardSubstitution() {
189             if (!REDUNDANT_LOAD_ELIMINATION) {
190                 return false;
191             }
192     
193             XField xfield = loadedFieldSet.getField(handle);
194             if (xfield == null) {
195                 return false;
196             }
197     
198             if(xfield.getSignature().equals("D") || xfield.getSignature().equals("J")) {
199                 return false;
200             }
201     
202             // Don't do forward substitution for fields that
203             // are never read.
204             if (!loadedFieldSet.isLoaded(xfield)) {
205                 return false;
206             }
207     
208             return true;
209         }

211         private void checkConsumedAndProducedValues(Instruction ins, ValueNumber[] consumedValueList, ValueNumber[] producedValueList) {
212             int numConsumed = ins.consumeStack(getCPG());
213             int numProduced = ins.produceStack(getCPG());
214     
215             if (numConsumed == Constants.UNPREDICTABLE) {
216                 throw new InvalidBytecodeException("Unpredictable stack consumption for " + ins);
217             }
218             if (numProduced == Constants.UNPREDICTABLE) {
219                 throw new InvalidBytecodeException("Unpredictable stack production for " + ins);
220             }
221     
222             if (consumedValueList.length != numConsumed) {
223                 throw new IllegalStateException("Wrong number of values consumed for " + ins + ": expected " + numConsumed + ", got "
224                         + consumedValueList.length);
225             }
226     
227             if (producedValueList.length != numProduced) {
228                 throw new IllegalStateException("Wrong number of values produced for " + ins + ": expected " + numProduced + ", got "
229                         + producedValueList.length);
230             }
231         }

233         /**
234          * This is the default instruction modeling method.
235          */
236         @Override
237         public void modelNormalInstruction(Instruction ins, int numWordsConsumed, int numWordsProduced) {
238     
239             int flags = 0;
240             if (ins instanceof InvokeInstruction) {
241                 flags = ValueNumber.RETURN_VALUE;
242             } else if (ins instanceof ArrayInstruction) {
243                 flags = ValueNumber.ARRAY_VALUE;
244             } else if (ins instanceof ConstantPushInstruction) {
245                 flags = ValueNumber.CONSTANT_VALUE;
246             }
247     
248             // Get the input operands to this instruction.
249             ValueNumber[] inputValueList = popInputValues(numWordsConsumed);
250     
251             // See if we have the output operands in the cache.
252             // If not, push default (fresh) values for the output,
253             // and add them to the cache.
254             ValueNumber[] outputValueList = getOutputValues(inputValueList, numWordsProduced, flags);
255     
256             if (VERIFY_INTEGRITY) {
257                 checkConsumedAndProducedValues(ins, inputValueList, outputValueList);
258             }
259     
260             // Push output operands on stack.
261             pushOutputValues(outputValueList);
262         }

264         @Override
265         public void visitGETFIELD(GETFIELD obj) {
266             XField xfield = Hierarchy.findXField(obj, getCPG());
267             if (xfield != null) {
268                 if (xfield.isVolatile()) {
269                     getFrame().killAllLoads();
270                 }
271     
272                 if (doRedundantLoadElimination()) {
273                     loadInstanceField(xfield, obj);
274                     return;
275                 }
276             }
277     
278             handleNormalInstruction(obj);
279         }

281         @Override
282         public void visitPUTFIELD(PUTFIELD obj) {
283             if (doForwardSubstitution()) {
284                 XField xfield = Hierarchy.findXField(obj, getCPG());
285                 if (xfield != null) {
286                     storeInstanceField(xfield, obj, false);
287                     return;
288                 }
289     
290             }
291             handleNormalInstruction(obj);
292         }

294         @Override
295         public void visitGETSTATIC(GETSTATIC obj) {
296             ConstantPoolGen cpg = getCPG();
297     
298             String fieldName = obj.getName(cpg);
299             String fieldSig = obj.getSignature(cpg);
300             ValueNumberFrame frame = getFrame();
301     
302             if (RLE_DEBUG) {
303                 System.out.println("GETSTATIC of " + fieldName + " : " + fieldSig);
304             }
305             // Is this an access of a Class object?
306             if (fieldName.startsWith("class$") && "Ljava/lang/Class;".equals(fieldSig)) {
307                 String className = fieldName.substring("class$".length()).replace('$', '.');
308                 if (RLE_DEBUG) {
309                     System.out.println("[found load of class object " + className + "]");
310                 }
311                 ValueNumber value = factory.getClassObjectValue(className);
312                 frame.pushValue(value);
313                 return;
314             }
315     
316             XField xfield = Hierarchy.findXField(obj, getCPG());
317             if (xfield != null) {
318                 if (xfield.isVolatile()) {
319                     getFrame().killAllLoads();
320                 }
321                 if (doRedundantLoadElimination()) {
322                     loadStaticField(xfield, obj);
323                     return;
324                 }
325     
326             }
327     
328             handleNormalInstruction(obj);
329         }

345         @Override
346         public void visitINVOKESTATIC(INVOKESTATIC obj) {
347             if (REDUNDANT_LOAD_ELIMINATION) {
348                 ConstantPoolGen cpg = getCPG();
349                 String targetClassName = obj.getClassName(cpg);
350                 String methodName = obj.getName(cpg);
351                 String methodSig = obj.getSignature(cpg);
352     
353                 if (("forName".equals(methodName) && "java.lang.Class".equals(targetClassName) || "class$".equals(methodName))
354                         && "(Ljava/lang/String;)Ljava/lang/Class;".equals(methodSig)) {
355                     // Access of a Class object
356                     ValueNumberFrame frame = getFrame();
357                     try {
358                         ValueNumber arg = frame.getTopValue();
359                         String className = stringConstantMap.get(arg);
360                         if (className != null) {
361                             frame.popValue();
362                             if (RLE_DEBUG) {
363                                 System.out.println("[found access of class object " + className + "]");
364                             }
365                             frame.pushValue(factory.getClassObjectValue(className));
366                             return;
367                         }
368                     } catch (DataflowAnalysisException e) {
369                         throw new InvalidBytecodeException("stack underflow", methodGen, handle, e);
370                     }
371                 } else if (Hierarchy.isInnerClassAccess(obj, cpg)) {
372                     // Possible access of field via an inner-class access method
373                     XField xfield = loadedFieldSet.getField(handle);
374                     if (xfield != null) {
375                         if (loadedFieldSet.instructionIsLoad(handle)) {
376                             // Load via inner-class accessor
377                             if (doRedundantLoadElimination()) {
378                                 if (xfield.isStatic()) {
379                                     loadStaticField(xfield, obj);
380                                 } else {
381                                     loadInstanceField(xfield, obj);
382                                 }
383                                 return;
384                             }
385                         } else {
386                             // Store via inner-class accessor
387                             if (doForwardSubstitution()) {
388                                 // Some inner class access store methods
389                                 // return the value stored.
390                                 boolean pushValue = !methodSig.endsWith(")V");
391     
392                                 if (xfield.isStatic()) {
393                                     storeStaticField(xfield, obj, pushValue);
394                                 } else {
395                                     storeInstanceField(xfield, obj, pushValue);
396                                 }
397     
398                                 return;
399                             }
400                         }
401                     }
402                 }
403             }
404             handleNormalInstruction(obj);
405         }

407         private void killLoadsOfObjectsPassed(INVOKEDYNAMIC ins) {
408             try {
409     
410                 int passed = getNumWordsConsumed(ins);
411                 ValueNumber[] arguments = allocateValueNumberArray(passed);
412                 getFrame().getTopStackWords(arguments);
413                 for (ValueNumber v : arguments) {
414                     getFrame().killAllLoadsOf(v);
415                 }
416     
417             } catch (DataflowAnalysisException e) {
418                 AnalysisContext.logError("Error in killLoadsOfObjectsPassed", e);
419             }
420         }
421     
422         private void killLoadsOfObjectsPassed(InvokeInstruction ins) {
423             try {
424     
425                 XMethod called = Hierarchy2.findExactMethod(ins, methodGen.getConstantPool(), Hierarchy.ANY_METHOD);
426                 if (called != null ) {
427                     NoSideEffectMethodsDatabase nse = Global.getAnalysisCache().getOptionalDatabase(NoSideEffectMethodsDatabase.class);
428                     if(nse != null && !nse.is(called.getMethodDescriptor(), MethodSideEffectStatus.SE, MethodSideEffectStatus.OBJ)) {
429                         return;
430                     }
431                 }
432                 FieldSummary fieldSummary = AnalysisContext.currentAnalysisContext().getFieldSummary();
433                 Set<XField> touched = fieldSummary.getFieldsWritten(called);
434                 if (!touched.isEmpty()) {
435                     getFrame().killLoadsOf(touched);
436                 }
437     
438     
439                 int passed = getNumWordsConsumed(ins);
440                 ValueNumber[] arguments = allocateValueNumberArray(passed);
441                 getFrame().killLoadsWithSimilarName(ins.getClassName(cpg), ins.getMethodName(cpg));
442                 getFrame().getTopStackWords(arguments);
443                 for (ValueNumber v : arguments) {
444                     getFrame().killAllLoadsOf(v);
445                 }
446     
447                 // Too many false-positives for primitives without transitive FieldSummary analysis,
448                 // so currently we simply kill any writable primitive on any method call
449                 // TODO: implement transitive FieldSummary
450                 getFrame().killAllLoads(true);
451     
452             } catch (DataflowAnalysisException e) {
453                 AnalysisContext.logError("Error in killLoadsOfObjectsPassed", e);
454             }
455         }

457         @Override
458         public void visitMONITORENTER(MONITORENTER obj) {
459             // Don't know what this sync invocation is doing.
460             // Kill all loads.
461             ValueNumber topValue = null;
462             try {
463                 topValue = getFrame().getStackValue(0);
464             } catch (DataflowAnalysisException e) {
465                 AnalysisContext.logError("error handling monitor enter in value numbering", e);
466             }
467             getFrame().killAllLoadsExceptFor(topValue);
468             handleNormalInstruction(obj);
469         }

471         public void visitInvokeOnException(Instruction obj) {
472             if(!REDUNDANT_LOAD_ELIMINATION || !getFrame().hasAvailableLoads()) {
473                 return;
474             }
475             if(obj instanceof INVOKEDYNAMIC) {
476                 killLoadsOfObjectsPassed((INVOKEDYNAMIC) obj);
477                 return;
478             }
479             InvokeInstruction inv = (InvokeInstruction) obj;
480             if ((inv instanceof INVOKEINTERFACE || inv instanceof INVOKEVIRTUAL)
481                     && inv.getMethodName(cpg).toLowerCase().indexOf("lock") >= 0) {
482                 // Don't know what this method invocation is doing.
483                 // Kill all loads.
484                 getFrame().killAllLoads();
485                 return;
486             }
487             if(inv instanceof INVOKEVIRTUAL && "cast".equals(inv.getMethodName(cpg)) && "java.lang.Class".equals(inv.getClassName(cpg))) {
488                 // No-op
489                 return;
490             }
491             if(inv instanceof INVOKESTATIC) {
492                 String methodName = inv.getName(cpg);
493                 if (("forName".equals(methodName) && "java.lang.Class".equals(inv.getClassName(cpg)) || "class$".equals(methodName))
494                         && "(Ljava/lang/String;)Ljava/lang/Class;".equals(inv.getSignature(cpg))
495                         || (Hierarchy.isInnerClassAccess((INVOKESTATIC) inv, cpg) && loadedFieldSet.getField(handle) != null)) {
496                     return;
497                 }
498             }
499     
500             killLoadsOfObjectsPassed(inv);
501             if(inv instanceof INVOKESTATIC) {
502                 getFrame().killAllLoadsOf(null);
503             }
504         }

506         @Override
507         public void visitINVOKEVIRTUAL(INVOKEVIRTUAL obj) {
508     
509             if ("cast".equals(obj.getMethodName(cpg)) && "java.lang.Class".equals(obj.getClassName(cpg))) {
510                 // treat as no-op
511                 try {
512                     ValueNumberFrame frame = getFrame();
513                     ValueNumber resultType = frame.popValue();
514                     frame.popValue();
515                     frame.pushValue(resultType);
516                 } catch (DataflowAnalysisException e) {
517                     AnalysisContext.logError("oops", e);
518                 }
519                 return;
520             }
521             handleNormalInstruction(obj);
522         }

524         @Override
525         public void visitACONST_NULL(ACONST_NULL obj) {
526             // Get the input operands to this instruction.
527             ValueNumber[] inputValueList = popInputValues(0);
528     
529             // See if we have the output operands in the cache.
530             // If not, push default (fresh) values for the output,
531             // and add them to the cache.
532             ValueNumber[] outputValueList = getOutputValues(inputValueList, 1, ValueNumber.CONSTANT_VALUE);
533     
534             if (VERIFY_INTEGRITY) {
535                 checkConsumedAndProducedValues(obj, inputValueList, outputValueList);

542         @Override
543         public void visitLDC(LDC obj) {
544             Object constantValue = obj.getValue(cpg);
545             ValueNumber value;
546             if (constantValue instanceof ConstantClass) {
547                 ConstantClass constantClass = (ConstantClass) constantValue;
548                 String className = constantClass.getBytes(cpg.getConstantPool());
549                 value = factory.getClassObjectValue(className);
550             } else if (constantValue instanceof ObjectType) {
551                 ObjectType objectType = (ObjectType) constantValue;
552                 String className = objectType.getClassName();
553                 value = factory.getClassObjectValue(className);
554             } else {
555                 value = constantValueMap.get(constantValue);
556                 if (value == null) {
557                     value = factory.createFreshValue(ValueNumber.CONSTANT_VALUE);
558                     constantValueMap.put(constantValue, value);
559     
560                     // Keep track of String constants
561     
562                     if (constantValue instanceof String) {
563                         stringConstantMap.put(value, (String) constantValue);
564                     }
565                 }
566             }
567             getFrame().pushValue(value);
568         }

570         @Override
571         public void visitIINC(IINC obj) {
572             if (obj.getIncrement() == 0) {
573                 // A no-op.
574                 return;
575             }
576     
577             // IINC is a special case because its input and output are not on the
578             // stack.
579             // However, we still want to use the value number cache to ensure that
580             // this operation is modeled consistently. (If we do nothing, we miss
581             // the fact that the referenced local is modified.)
582     
583             int local = obj.getIndex();
584     
585             ValueNumber[] input = new ValueNumber[] { getFrame().getValue(local) };
586             ValueNumberCache.Entry entry = new ValueNumberCache.Entry(handle, input);
587             ValueNumber[] output = cache.lookupOutputValues(entry);
588             if (output == null) {
589                 output = new ValueNumber[] { factory.createFreshValue() };
590                 cache.addOutputValues(entry, output);
591             }
592     
593             getFrame().setValue(local, output[0]);
594         }

607         /**
608          * Pop the input values for the given instruction from the current frame.
609          */
610         private ValueNumber[] popInputValues(int numWordsConsumed) {
611             ValueNumberFrame frame = getFrame();
612             ValueNumber[] inputValueList = allocateValueNumberArray(numWordsConsumed);
613     
614             // Pop off the input operands.
615             try {
616                 frame.getTopStackWords(inputValueList);
617                 while (numWordsConsumed-- > 0) {
618                     frame.popValue();
619                 }
620             } catch (DataflowAnalysisException e) {
621                 throw new InvalidBytecodeException("Error getting input operands", e);
622             }
623     
624             return inputValueList;
625         }
626     
627         /**
628          * Push given output values onto the current frame.
629          */
630         private void pushOutputValues(ValueNumber[] outputValueList) {
631             ValueNumberFrame frame = getFrame();
632             for (ValueNumber aOutputValueList : outputValueList) {
633                 frame.pushValue(aOutputValueList);
634             }
635         }
636     
637         /**
638          * Get output values for current instruction from the ValueNumberCache.
639          */
640         private ValueNumber[] getOutputValues(ValueNumber[] inputValueList, int numWordsProduced) {
641             return getOutputValues(inputValueList, numWordsProduced, 0);
642         }
643     
644         private ValueNumber[] getOutputValues(ValueNumber[] inputValueList, int numWordsProduced, int flags) {
645             ValueNumberCache.Entry entry = new ValueNumberCache.Entry(handle, inputValueList);
646             ValueNumber[] outputValueList = cache.lookupOutputValues(entry);
647             if (outputValueList == null) {
648                 outputValueList = allocateValueNumberArray(numWordsProduced);
649                 for (int i = 0; i < numWordsProduced; ++i) {
650                     ValueNumber freshValue = factory.createFreshValue(flags);
651                     outputValueList[i] = freshValue;
652                 }
653                 /*
654                 if (false && RLE_DEBUG) {
655                     System.out.println("<<cache fill for " + handle.getPosition() + ": " + vlts(inputValueList) + " ==> "
656                             + vlts(outputValueList) + ">>");
657                 }
658                  */
659                 cache.addOutputValues(entry, outputValueList);
660             } /* else if (false && RLE_DEBUG) {
661                 System.out.println("<<cache hit for " + handle.getPosition() + ": " + vlts(inputValueList) + " ==> "
662                         + vlts(outputValueList) + ">>");
663             } */
664             return outputValueList;
665         }

667         /**
668          * Creates a new empty array (if needed) with given size.
669          *
670          * @param size
671          *            array size
672          * @return if size is zero, returns {@link #EMPTY_INPUT_VALUE_LIST}
673          */
674         private static ValueNumber[] allocateValueNumberArray(int size) {
675             if (size == 0) {
676                 return EMPTY_INPUT_VALUE_LIST;
677             }
678             return new ValueNumber[size];
679         }

681         private static String vlts(ValueNumber[] vl) {
682             StringBuilder buf = new StringBuilder();
683             for (ValueNumber aVl : vl) {
684                 if (buf.length() > 0) {
685                     buf.append(',');
686                 }
687                 buf.append(aVl.getNumber());
688             }
689             return buf.toString();
690         }

692         /**
693          * Load an instance field.
694          *
695          * @param instanceField
696          *            the field
697          * @param obj
698          *            the Instruction loading the field
699          */
700         private void loadInstanceField(XField instanceField, Instruction obj) {
701             if (RLE_DEBUG) {
702                 System.out.println("[loadInstanceField for field " + instanceField + " in instruction " + handle);
703             }
704     
705             ValueNumberFrame frame = getFrame();
706     
707             try {
708                 ValueNumber reference = frame.popValue();
709     
710                 AvailableLoad availableLoad = new AvailableLoad(reference, instanceField);
711                 if (RLE_DEBUG) {
712                     System.out.println("[getfield of " + availableLoad + "]");
713                 }
714                 ValueNumber[] loadedValue = frame.getAvailableLoad(availableLoad);
715     
716                 if (loadedValue == null) {
717                     // Get (or create) the cached result for this instruction
718                     ValueNumber[] inputValueList = new ValueNumber[] { reference };
719                     loadedValue = getOutputValues(inputValueList, getNumWordsProduced(obj));
720     
721                     // Make the load available
722                     frame.addAvailableLoad(availableLoad, loadedValue);
723                     if (RLE_DEBUG) {
724                         System.out.println("[Making load available " + availableLoad + " <- " + vlts(loadedValue) + "]");
725                     }
726                 } else {
727                     // Found an available load!
728                     if (RLE_DEBUG) {
729                         System.out.println("[Found available load " + availableLoad + " <- " + vlts(loadedValue) + "]");
730                     }
731                 }
732     
733                 pushOutputValues(loadedValue);
734     
735                 if (VERIFY_INTEGRITY) {
736                     checkConsumedAndProducedValues(obj, new ValueNumber[] { reference }, loadedValue);
737                 }
738             } catch (DataflowAnalysisException e) {
739                 throw new InvalidBytecodeException("Error loading from instance field", e);
740             }
741         }

743         /**
744          * Load a static field.
745          *
746          * @param staticField
747          *            the field
748          * @param obj
749          *            the Instruction loading the field
750          */
751         private void loadStaticField(XField staticField, Instruction obj) {
752             if (RLE_DEBUG) {
753                 System.out.println("[loadStaticField for field " + staticField + " in instruction " + handle);
754             }
755     
756             ValueNumberFrame frame = getFrame();
757     
758             AvailableLoad availableLoad = new AvailableLoad(staticField);
759             ValueNumber[] loadedValue = frame.getAvailableLoad(availableLoad);
760     
761             if (loadedValue == null) {
762                 // Make the load available
763                 int numWordsProduced = getNumWordsProduced(obj);
764                 loadedValue = getOutputValues(EMPTY_INPUT_VALUE_LIST, numWordsProduced);
765     
766                 frame.addAvailableLoad(availableLoad, loadedValue);
767     
768                 if (RLE_DEBUG) {
769                     System.out.println("[making load of " + staticField + " available]");
770                 }
771             } else {
772                 if (RLE_DEBUG) {
773                     System.out.println("[found available load of " + staticField + "]");
774                 }
775             }
776     
777             if (VERIFY_INTEGRITY) {
778                 checkConsumedAndProducedValues(obj, EMPTY_INPUT_VALUE_LIST, loadedValue);
779             }
780     
781             pushOutputValues(loadedValue);
782         }

784         /**
785          * Store an instance field.
786          *
787          * @param instanceField
788          *            the field
789          * @param obj
790          *            the instruction which stores the field
791          * @param pushStoredValue
792          *            push the stored value onto the stack (because we are modeling
793          *            an inner-class field access method)
794          */
795         private void storeInstanceField(XField instanceField, Instruction obj, boolean pushStoredValue) {
796             if (RLE_DEBUG) {
797                 System.out.println("[storeInstanceField for field " + instanceField + " in instruction " + handle);
798             }
799     
800             ValueNumberFrame frame = getFrame();
801     
802             int numWordsConsumed = getNumWordsConsumed(obj);
803             /*
804              * System.out.println("Instruction is " + handle);
805              * System.out.println("numWordsConsumed="+numWordsConsumed);
806              */
807             ValueNumber[] inputValueList = popInputValues(numWordsConsumed);
808             ValueNumber reference = inputValueList[0];
809             ValueNumber[] storedValue = allocateValueNumberArray(inputValueList.length - 1);
810             System.arraycopy(inputValueList, 1, storedValue, 0, inputValueList.length - 1);
811     
812             if (pushStoredValue) {
813                 pushOutputValues(storedValue);
814             }
815     
816             // Kill all previous loads of the same field,
817             // in case there is aliasing we don't know about
818             frame.killLoadsOfField(instanceField);
819     
820             // Forward substitution
821             frame.addAvailableLoad(new AvailableLoad(reference, instanceField), storedValue);
822     
823             if (RLE_DEBUG) {
824                 System.out.println("[making store of " + instanceField + " available]");
825             }
826     
827             if (VERIFY_INTEGRITY) {
828                 /*
829                  * System.out.println("pushStoredValue="+pushStoredValue);
830                  */
831                 checkConsumedAndProducedValues(obj, inputValueList, pushStoredValue ? storedValue : EMPTY_INPUT_VALUE_LIST);
832             }
833         }

835         /**
836          * Store a static field.
837          *
838          * @param staticField
839          *            the static field
840          * @param obj
841          *            the instruction which stores the field
842          * @param pushStoredValue
843          *            push the stored value onto the stack (because we are modeling
844          *            an inner-class field access method)
845          */
846         private void storeStaticField(XField staticField, Instruction obj, boolean pushStoredValue) {
847             if (RLE_DEBUG) {
848                 System.out.println("[storeStaticField for field " + staticField + " in instruction " + handle);
849             }
850     
851             ValueNumberFrame frame = getFrame();
852     
853             AvailableLoad availableLoad = new AvailableLoad(staticField);
854     
855             int numWordsConsumed = getNumWordsConsumed(obj);
856             ValueNumber[] inputValueList = popInputValues(numWordsConsumed);
857     
858             if (pushStoredValue) {
859                 pushOutputValues(inputValueList);
860             }
861     
862             // Kill loads of this field
863             frame.killLoadsOfField(staticField);
864     
865             // Make load available
866             frame.addAvailableLoad(availableLoad, inputValueList);
867     
868             if (RLE_DEBUG) {
869                 System.out.println("[making store of " + staticField + " available]");
870             }
871     
872             if (VERIFY_INTEGRITY) {
873                 checkConsumedAndProducedValues(obj, inputValueList, pushStoredValue ? inputValueList : EMPTY_INPUT_VALUE_LIST);
874             }
875         }
