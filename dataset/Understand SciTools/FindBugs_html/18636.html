20     package edu.umd.cs.findbugs;
21     
22     /**
23      * A BugCode is an abbreviation that is shared among some number of BugPatterns.
24      * For example, the code "HE" is shared by all of the BugPatterns that represent
25      * hashcode/equals violations.
26      *
27      * @author David Hovemeyer
28      * @see BugPattern
29      */
30     public class BugCode implements Comparable<BugCode> {
31         private final String abbrev;
32     
33         private final int cweid;
34     
35         private final String description;
36     
37         /**
38          * Constructor.
39          *
40          * @param abbrev
41          *            the abbreviation for the bug code
42          * @param description
43          *            a short textual description of the class of bug pattern
44          *            represented by this bug code
45          */
46         public BugCode(String abbrev, String description) {
47             this.abbrev = abbrev;
48             this.description = description;
49             this.cweid = 0;
50         }
51     
52         public BugCode(String abbrev, String description, int cweid) {
53             this.abbrev = abbrev;
54             this.description = description;
55             this.cweid = cweid;
56         }
57     
58         /**
59          * Get the abbreviation for this bug code.
60          */
61         public String getAbbrev() {
62             return abbrev;
63         }
64     
65         /**
66          * Get the short textual description of the bug code.
67          */
68         public String getDescription() {
69             return description;
70         }
71     
72         /**
73          * Get the abbreviation fo this bug code.
74          */
75         @Override
76         public String toString() {
77             return "BugCode[" + abbrev + "]";
78         }
79     
80         /**
81          * @return Returns the cweid.
82          */
83         public int getCWEid() {
84             return cweid;
85         }
86     
87         @Override
88         public int compareTo(BugCode o) {
89             return abbrev.compareTo(o.abbrev);
90         }
91     
92         @Override
93         public boolean equals(Object o) {
94             if (this == o) {
95                 return true;
96             }
97             if (!(o instanceof BugCode)) {
98                 return false;
99             }
100             BugCode other = (BugCode) o;
101             return abbrev.equals(other.abbrev);
102         }
103     
104         @Override
105         public int hashCode() {
106             return abbrev.hashCode();
107         }
108     }
