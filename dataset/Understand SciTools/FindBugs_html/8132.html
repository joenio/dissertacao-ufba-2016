20     package edu.umd.cs.findbugs.anttask;
21     
22     import java.io.File;
23     import java.io.FileInputStream;
24     import java.io.FileOutputStream;
25     import java.io.IOException;
26     import java.nio.channels.FileChannel;
27     import java.util.ArrayList;
28     import java.util.List;
29     
30     import org.apache.tools.ant.BuildException;
31     import org.apache.tools.ant.Task;
32     import org.apache.tools.ant.types.FileSet;
33     
34     import edu.umd.cs.findbugs.workflow.UnionResults;
35     
36     /**
37      * An ant task that is wraps the behavior of the UnionResults executable into an
38      * ant task.
39      *
40      * <taskdef name="UnionBugs" classname="edu.umd.cs.findbugs.anttask.UnionBugs"
41      * classpath="...">
42      *
43      * <UnionBugs to="${basedir}/findbugs.xml" > <fileset dir="plugins"> <include
44      * name="*_findbugs_partial.xml" /> </fileset> </UnionBugs>
45      *
46      * @author Peter Franza <a href="mailto:pfranza@gmail.com">pfranza@gmail.com</a>
47      * @version 1.0
48      *
49      * @ant.task category="utility"
50      *
51      */
52     @Deprecated
53     public class UnionBugs extends Task {
54     
55         private final List<FileSet> fileSets = new ArrayList<FileSet>();
56     
57         private String into;
58     
59         /**
60          * The fileset containing all the findbugs xml files that need to be merged
61          *
62          * @param arg
63          */
64         public void addFileset(FileSet arg) {
65             fileSets.add(arg);
66         }
67     
68         /**
69          * The File everything should get merged into
70          *
71          * @param file
72          */
73         public void setTo(String file) {
74             into = file;
75         }
76     
77         @Override
78         public void execute() throws BuildException {
79             super.execute();
80     
81             try {
82     
83                 List<File> fileList = createListOfAllFilesToMerge();
84     
85                 // If there is nothing to merge, don't
86                 if (fileList.isEmpty()) {
87                     return;
88                 }
89     
90                 // if the merge target doesn't exist yet, then the first file
91                 // becomes
92                 // the merge target
93                 File to = new File(into);
94                 if (!to.exists()) {
95                     File from = fileList.remove(0);
96                     copyFile(from, to);
97                 }
98     
99                 // If there was only one file, and it has been copied as
100                 // the merge target then you're done
101                 if (fileList.isEmpty()) {
102                     return;
103                 }
104     
105                 UnionResults.main(createCommandArgumentsArray(fileList));
106             } catch (Exception e) {
107                 throw new BuildException(e);
108             }
109         }
110     
111         private List<File> createListOfAllFilesToMerge() {
112             List<File> fileList = new ArrayList<File>();
113             for (FileSet s : fileSets) {
114                 File fromDir = s.getDir(getProject());
115                 for (String file : s.getDirectoryScanner(getProject()).getIncludedFiles()) {
116                     fileList.add(new File(fromDir, file));
117                 }
118             }
119             return fileList;
120         }
121     
122         private String[] createCommandArgumentsArray(List<File> fileList) {
123             List<String> parts = new ArrayList<String>();
124             parts.add("-withMessages");
125             parts.add("-output");
126             parts.add(into);
127     
128             parts.add(into);
129     
130             for (File f : fileList) {
131                 parts.add(f.getAbsolutePath());
132             }
133     
134             String[] args = parts.toArray(new String[parts.size()]);
135             return args;
136         }
137     
138         /**
139          * Copy a File
140          *
141          * @param in
142          *            to Copy From
143          * @param out
144          *            to Copy To
145          * @throws IOException
146          */
147         private static void copyFile(File in, File out) throws IOException {
148             try (FileInputStream inStream = new FileInputStream(in);
149                     FileOutputStream outStream = new FileOutputStream(out);) {
150                 FileChannel inChannel = inStream.getChannel();
151                 inChannel.transferTo(0, inChannel.size(), outStream.getChannel());
152             }
153         }
154     
155     }
