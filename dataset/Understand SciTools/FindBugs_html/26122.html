44     /**
45      * Build database of methods that return values guaranteed to be nonnull
46      *
47      */
48     public class BuildNonnullReturnDatabase {
49         public static final boolean VERBOSE_DEBUG = SystemProperties.getBoolean("fnd.debug.nullarg.verbose");
50     
51         private static final boolean DEBUG = SystemProperties.getBoolean("fnd.debug.nullarg") || VERBOSE_DEBUG;
52     
53         public void visitClassContext(ClassContext classContext) {
54             boolean fullAnalysis = AnalysisContext.currentAnalysisContext().getBoolProperty(
55                     FindBugsAnalysisFeatures.INTERPROCEDURAL_ANALYSIS_OF_REFERENCED_CLASSES);
56             if (!fullAnalysis && !AnalysisContext.currentAnalysisContext()./*
57              * getSubtypes
58              * ().
59              */isApplicationClass(classContext.getJavaClass())) {
60                 return;
61             }
62             if (VERBOSE_DEBUG) {
63                 System.out.println("Visiting class " + classContext.getJavaClass().getClassName());
64             }
65     
66             for (Method m : classContext.getMethodsInCallOrder()) {
67                 considerMethod(classContext, m);
68             }
69         }
70     
71         private void considerMethod(ClassContext classContext, Method method) {
72             if ((method.getReturnType() instanceof ReferenceType) && classContext.getMethodGen(method) != null) {
73                 if (VERBOSE_DEBUG) {
74                     System.out.println("Check " + method);
75                 }
76                 analyzeMethod(classContext, method);
77             }
78         }
79     
80         protected int returnsReference;
81     
82         protected int returnsNonNull;
83     
84         private void analyzeMethod(ClassContext classContext, Method method) {
85             returnsReference++;
86             try {
87                 CFG cfg = classContext.getCFG(method);
88     
89                 IsNullValueDataflow inv = classContext.getIsNullValueDataflow(method);
90                 boolean guaranteedNonNull = true;
91                 for (Iterator<Location> i = cfg.locationIterator(); i.hasNext();) {
92                     Location location = i.next();
93                     InstructionHandle handle = location.getHandle();
94                     Instruction ins = handle.getInstruction();
95     
96                     if (!(ins instanceof ARETURN)) {
97                         continue;
98                     }
99                     IsNullValueFrame frame = inv.getFactAtLocation(location);
100                     if (!frame.isValid()) {
101                         continue;
102                     }
103                     IsNullValue value = frame.getTopValue();
104                     if (!value.isDefinitelyNotNull()) {
105                         guaranteedNonNull = false;
106                         break;
107                     }
108     
109                 }
110     
111                 XMethod xmethod = XFactory.createXMethod(classContext.getJavaClass(), method);
112                 if (guaranteedNonNull) {
113                     returnsNonNull++;
114                     AnalysisContext.currentAnalysisContext().getReturnValueNullnessPropertyDatabase()
115                     .setProperty(xmethod.getMethodDescriptor(), guaranteedNonNull);
116                     if (DEBUG) {
117                         System.out.println("Unconditional deref: " + xmethod + "=" + guaranteedNonNull);
118                     }
119     
120                 }
121     
122             } catch (CFGBuilderException e) {
123                 XMethod xmethod = XFactory.createXMethod(classContext.getJavaClass(), method);
124     
125                 AnalysisContext.currentAnalysisContext().getLookupFailureCallback()
126                 .logError("Error analyzing " + xmethod + " for unconditional deref training", e);
127             } catch (DataflowAnalysisException e) {
128                 XMethod xmethod = XFactory.createXMethod(classContext.getJavaClass(), method);
129                 AnalysisContext.currentAnalysisContext().getLookupFailureCallback()
130                 .logError("Error analyzing " + xmethod + " for unconditional deref training", e);
131             }
132         }
