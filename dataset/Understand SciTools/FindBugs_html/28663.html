73         @Override
74         public void visitMethod(Method method) {
75             try {
76                 if (method.getCode() == null) {
77                     return;
78                 }
79     
80                 CFG cfg = classContext.getCFG(method);
81     
82                 Iterator<BasicBlock> bbi = cfg.blockIterator();
83                 while (bbi.hasNext()) {
84                     BasicBlock bb = bbi.next();
85     
86                     int numOutgoing = cfg.getNumOutgoingEdges(bb);
87                     if (numOutgoing == 2) {
88                         findIfElseDuplicates(cfg, method, bb);
89                     } else if (numOutgoing > 2) {
90                         findSwitchDuplicates(cfg, method, bb);
91                     }
92                 }
93             } catch (MethodUnprofitableException mue) {
94                 if (SystemProperties.getBoolean("unprofitable.debug")) {
95                     // don't
96                     // report
97                     bugReporter.logError("skipping unprofitable method in " + getClass().getName());
98                 }
99             } catch (Exception e) {
100                 bugReporter.logError("Failure examining basic blocks in Duplicate Branches detector", e);
101             }
102             if (pendingBugs.size() <= 2) {
103                 for (BugInstance b : pendingBugs) {
104                     bugReporter.reportBug(b);
105                 }
106             }
107             pendingBugs.clear();
108     
109         }

111         private void findIfElseDuplicates(CFG cfg, Method method, BasicBlock bb) {
112             BasicBlock thenBB = null, elseBB = null;
113     
114             Iterator<Edge> iei = cfg.outgoingEdgeIterator(bb);
115             while (iei.hasNext()) {
116                 Edge e = iei.next();
117                 if (e.getType() == EdgeTypes.IFCMP_EDGE) {
118                     elseBB = e.getTarget();
119                 } else if (e.getType() == EdgeTypes.FALL_THROUGH_EDGE) {
120                     thenBB = e.getTarget();
121                 }
122             }
123     
124             if ((thenBB == null) || (elseBB == null)) {
125                 return;
126             }
127             InstructionHandle thenStartHandle = getDeepFirstInstruction(cfg, thenBB);
128             InstructionHandle elseStartHandle = getDeepFirstInstruction(cfg, elseBB);
129             if ((thenStartHandle == null) || (elseStartHandle == null)) {
130                 return;
131             }
132     
133             int thenStartPos = thenStartHandle.getPosition();
134             int elseStartPos = elseStartHandle.getPosition();
135     
136             InstructionHandle thenFinishIns = findThenFinish(cfg, thenBB, elseStartPos);
137             int thenFinishPos = thenFinishIns.getPosition();
138     
139             if (!(thenFinishIns.getInstruction() instanceof GotoInstruction)) {
140                 return;
141             }
142     
143             InstructionHandle elseFinishHandle = ((GotoInstruction) thenFinishIns.getInstruction()).getTarget();
144             int elseFinishPos = elseFinishHandle.getPosition();
145     
146             if (thenFinishPos >= elseStartPos) {
147                 return;
148             }
149     
150             if ((thenFinishPos - thenStartPos) != (elseFinishPos - elseStartPos)) {
151                 return;
152             }
153     
154             if (thenFinishPos <= thenStartPos) {
155                 return;
156             }
157     
158             byte[] thenBytes = getCodeBytes(method, thenStartPos, thenFinishPos);
159             byte[] elseBytes = getCodeBytes(method, elseStartPos, elseFinishPos);
160     
161             if (!Arrays.equals(thenBytes, elseBytes)) {
162                 return;
163             }
164     
165             // adjust elseFinishPos to be inclusive (for source line attribution)
166             InstructionHandle elseLastIns = elseFinishHandle.getPrev();
167             if (elseLastIns != null) {
168                 elseFinishPos = elseLastIns.getPosition();
169             }
170     
171             pendingBugs.add(new BugInstance(this, "DB_DUPLICATE_BRANCHES", NORMAL_PRIORITY)
172             .addClassAndMethod(classContext.getJavaClass(), method)
173             .addSourceLineRange(classContext, this, thenStartPos, thenFinishPos)
174             .addSourceLineRange(classContext, this, elseStartPos, elseFinishPos));
175         }

177         /**
178          * Like bb.getFirstInstruction() except that if null is returned it will
179          * follow the FALL_THROUGH_EDGE (if any)
180          */
181         private static InstructionHandle getDeepFirstInstruction(CFG cfg, BasicBlock bb) {
182             InstructionHandle ih = bb.getFirstInstruction();
183             if (ih != null) {
184                 return ih;
185             }
186             Iterator<Edge> iei = cfg.outgoingEdgeIterator(bb);
187             while (iei.hasNext()) {
188                 Edge e = iei.next();
189                 String edgeString = e.toString();
190                 if (EdgeTypes.FALL_THROUGH_EDGE == e.getType()) {
191                     return getDeepFirstInstruction(cfg, e.getTarget());
192                 }
193             }
194             return null;
195         }

197         private void findSwitchDuplicates(CFG cfg, Method method, BasicBlock bb) {
198     
199             int[] switchPos = new int[cfg.getNumOutgoingEdges(bb) + 1];
200             HashMap<Integer, InstructionHandle> prevHandle = new HashMap<Integer, InstructionHandle>();
201     
202             Iterator<Edge> iei = cfg.outgoingEdgeIterator(bb);
203             int idx = 0;
204     
205             while (iei.hasNext()) {
206                 Edge e = iei.next();
207                 int eType = e.getType();
208                 if (eType == EdgeTypes.SWITCH_EDGE || eType == EdgeTypes.SWITCH_DEFAULT_EDGE) {
209                     BasicBlock target = e.getTarget();
210                     InstructionHandle firstIns = getDeepFirstInstruction(cfg, target);
211                     if (firstIns == null)
212                     {
213                         continue; // give up on this edge
214                     }
215                     int firstInsPosition = firstIns.getPosition();
216                     switchPos[idx++] = firstInsPosition;
217                     InstructionHandle prevIns = firstIns.getPrev(); // prev in
218                     // bytecode, not
219                     // flow
220                     if (prevIns != null) {
221                         prevHandle.put(firstInsPosition, prevIns);
222                     }
223                 } else {
224                     // hmm, this must not be a switch statement, so give up
225                     return;
226                 }
227             }
228     
229             if (idx < 2) {
230                 return;
231             }
232     
233             Arrays.sort(switchPos, 0, idx); // sort the 'idx' switch positions
234     
235             // compute end position of final case (ok if set to 0 or <=
236             // switchPos[idx-1])
237             switchPos[idx] = getFinalTarget(cfg, switchPos[idx - 1], prevHandle.values());
238     
239             HashMap<BigInteger, Collection<Integer>> map = new HashMap<BigInteger, Collection<Integer>>();
240             for (int i = 0; i < idx; i++) {
241                 if (switchPos[i] + 7 >= switchPos[i + 1])
242                 {
243                     continue; // ignore small switch clauses
244                 }
245     
246                 int endPos = switchPos[i + 1];
247                 InstructionHandle last = prevHandle.get(switchPos[i + 1]);
248                 if (last == null) {
249                     // should be default case -- leave endPos as is
250                 } else if (last.getInstruction() instanceof GotoInstruction) {
251                     endPos = last.getPosition(); // don't store the goto
252                 } else if (last.getInstruction() instanceof ReturnInstruction) {
253                     // leave endPos as is (store the return instruction)
254                     // } else if (last.getInstruction() instanceof ATHROW) {
255                     // // leave endPos as is (store the throw instruction)
256                     // Don't do this since many cases may throw "not implemented".
257                 } else {
258                     if (i + 2 < idx)
259                     {
260                         continue; // falls through to next case, so don't store it
261                     }
262                     // at all
263                     if (i + 1 < idx && switchPos[idx] != switchPos[idx - 1])
264                     {
265                         continue; // also falls through unless switch has no default
266                         // case
267                     }
268                 }
269     
270                 BigInteger clauseAsInt = getCodeBytesAsBigInt(method, switchPos, i, endPos);
271                 updateMap(map, i, clauseAsInt);
272     
273             }
274             for (Collection<Integer> clauses : map.values()) {
275                 if (clauses.size() > 1) {
276                     BugInstance bug = new BugInstance(this, "DB_DUPLICATE_SWITCH_CLAUSES", LOW_PRIORITY).addClassAndMethod(
277                             classContext.getJavaClass(), method);
278                     for (int i : clauses)
279                     {
280                         bug.addSourceLineRange(this.classContext, this, switchPos[i], switchPos[i + 1] - 1); // not
281                     }
282                     // endPos,
283                     // but
284                     // that's
285                     // ok
286                     pendingBugs.add(bug);
287                 }
288             }
289         }

291         private void updateMap(HashMap<BigInteger, Collection<Integer>> map, int i, BigInteger clauseAsInt) {
292             Collection<Integer> values = map.get(clauseAsInt);
293     
294             if (values == null) {
295                 values = new LinkedList<Integer>();
296                 map.put(clauseAsInt, values);
297             }
298             values.add(i); // index into the sorted array
299         }

301         private BigInteger getCodeBytesAsBigInt(Method method, int[] switchPos, int i, int endPos) {
302             byte[] clause = getCodeBytes(method, switchPos[i], endPos);
303     
304             BigInteger clauseAsInt;
305             if (clause.length == 0) {
306                 clauseAsInt = BigInteger.ZERO;
307             } else {
308                 clauseAsInt = new BigInteger(clause);
309             }
310             return clauseAsInt;
311         }

313         /**
314          * determine the end position (exclusive) of the final case by looking at
315          * the gotos at the ends of the other cases
316          */
317         private static int getFinalTarget(CFG cfg, int myPos, Collection<InstructionHandle> prevs) {
318             int maxGoto = 0;
319             BasicBlock myBB = null;
320             // note: InstructionHandle doesn't override equals(), so use
321             // prevs.contains() with caution.
322             Iterator<BasicBlock> bbi = cfg.blockIterator();
323             while (bbi.hasNext()) {
324                 BasicBlock bb = bbi.next();
325                 InstructionHandle last = bb.getLastInstruction(); // may be null
326                 if (prevs.contains(last)) { // danger will robinson
327                     Iterator<Edge> iei = cfg.outgoingEdgeIterator(bb);
328                     while (iei.hasNext()) {
329                         Edge e = iei.next();
330                         int eType = e.getType();
331                         String aab = e.toString();
332                         if (eType == EdgeTypes.GOTO_EDGE) {
333                             BasicBlock target = e.getTarget();
334                             InstructionHandle targetFirst = getDeepFirstInstruction(cfg, target);
335                             if (targetFirst != null) {
336                                 int targetPos = targetFirst.getPosition();
337                                 if (targetPos > maxGoto) {
338                                     maxGoto = targetPos;
339                                 }
340                             }
341                         }
342                     }
343                 } else if (last != null && myPos == bb.getFirstInstruction().getPosition()) {
344                     // note: getFirstInstruction() may return null, but won't if
345                     // last!=null.
346                     myBB = bb; // used in case (c) below
347                 }
348             }
349             /*
350              * ok, there are three cases: a) maxGoto == myPos: There is no default
351              * case within the switch. b) maxGoto > myPos: maxGoto is the end
352              * (exclusive) of the default case c) maxGoto < myPos: all the cases do
353              * something like return or throw, so who knows if there is a default
354              * case (and it's length)?
355              */
356             if (maxGoto < myPos && myBB != null) {
357                 /*
358                  * We're in case (c), so let's guess that the end of the basic block
359                  * is the end of the default case (if it exists). This may give
360                  * false negatives (if the default case has branches, for example)
361                  * but shouldn't give false negatives (because if it matches one of
362                  * the cases, then it has also matched that case's return/throw).
363                  */
364                 InstructionHandle last = myBB.getLastInstruction();
365                 if (last != null) {
366                     // note: last.getNext() may return null, so do it this way
367                     return last.getPosition() + last.getInstruction().getLength();
368                 }
369             }
370             return maxGoto;
371         }

373         private byte[] getCodeBytes(Method m, int start, int end) {
374             byte[] code = m.getCode().getCode();
375             byte[] bytes = new byte[end - start];
376             System.arraycopy(code, start, bytes, 0, end - start);
377     
378             try {
379                 ByteSequence sequence = new ByteSequence(code);
380                 while ((sequence.available() > 0) && (sequence.getIndex() < start)) {
381                     Instruction.readInstruction(sequence);
382                 }
383     
384                 int pos;
385                 while (sequence.available() > 0 && ((pos = sequence.getIndex()) < end)) {
386                     Instruction ins = Instruction.readInstruction(sequence);
387                     if ((ins instanceof BranchInstruction) && !(ins instanceof TABLESWITCH) && !(ins instanceof LOOKUPSWITCH)) {
388                         BranchInstruction bi = (BranchInstruction) ins;
389                         int offset = bi.getIndex();
390                         int target = offset + pos;
391                         if (target >= end) { // or target < start ??
392                             byte hiByte = (byte) ((target >> 8) & 0x000000FF);
393                             byte loByte = (byte) (target & 0x000000FF);
394                             bytes[pos + bi.getLength() - 2 - start] = hiByte;
395                             bytes[pos + bi.getLength() - 1 - start] = loByte;
396                         }
397                     }
398                 }
399             } catch (IOException ioe) {
400             }
401     
402             return bytes;
403         }

405         private InstructionHandle findThenFinish(CFG cfg, BasicBlock thenBB, int elsePos) {
406             InstructionHandle inst = thenBB.getFirstInstruction();
407             while (inst == null) {
408                 Iterator<Edge> ie = cfg.outgoingEdgeIterator(thenBB);
409                 while (ie.hasNext()) {
410                     Edge e = ie.next();
411                     if (e.getType() == EdgeTypes.FALL_THROUGH_EDGE) {
412                         thenBB = e.getTarget();
413                         break;
414                     }
415                 }
416                 inst = thenBB.getFirstInstruction();
417             }
418     
419             InstructionHandle lastIns = inst;
420             while (inst.getPosition() < elsePos) {
421                 lastIns = inst;
422                 inst = inst.getNext();
423             }
424     
425             return lastIns;
426         }
