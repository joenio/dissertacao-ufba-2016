29         protected UnresolvedXMethod(MethodDescriptor m) {
30             super(m.getClassDescriptor().getDottedClassName(), m.getName(), m.getSignature(),
31                     m.isStatic() ? Constants.ACC_STATIC : 0);
32             if (XFactory.DEBUG_UNRESOLVED) {
33                 System.out.println("Unresolved xmethod: " + this);
34             }
35         }

56             if (o instanceof XMethod) {
57                 return XFactory.compare((XMethod) this, (XMethod) o);
58             }

177         Map<Integer, Map<ClassDescriptor, AnnotationValue>> methodParameterAnnotations = Collections.emptyMap();
178     
179         /*
180          * (non-Javadoc)
181          *
182          * @see edu.umd.cs.findbugs.ba.XMethod#addParameterAnnotation(int,
183          * edu.umd.cs.findbugs.classfile.analysis.AnnotationValue)
184          */
185         @Override
186         public void addParameterAnnotation(int param, AnnotationValue annotationValue) {
187             HashMap<Integer, Map<ClassDescriptor, AnnotationValue>> updatedAnnotations = new HashMap<Integer, Map<ClassDescriptor, AnnotationValue>>(
188                     methodParameterAnnotations);
189             Map<ClassDescriptor, AnnotationValue> paramMap = updatedAnnotations.get(param);
190             if (paramMap == null) {
191                 paramMap = new HashMap<ClassDescriptor, AnnotationValue>();
192                 updatedAnnotations.put(param, paramMap);
193             }
194             paramMap.put(annotationValue.getAnnotationClass(), annotationValue);
195     
196             methodParameterAnnotations = updatedAnnotations;
197     
198         }
199     
200         @Override
201         public Collection<ClassDescriptor> getParameterAnnotationDescriptors(int param) {
202             Map<ClassDescriptor, AnnotationValue> map = methodParameterAnnotations.get(param);
203             if (map == null) {
204                 return Collections.<ClassDescriptor> emptySet();
205             }
206             return map.keySet();
207         }
