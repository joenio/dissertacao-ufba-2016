34     public class Noise extends OpcodeStackDetector {

36         class HashQueue {
37             HashQueue(int size) throws NoSuchAlgorithmException {
38                 md = MessageDigest.getInstance("SHA");
39                 this.size = size;
40                 this.data = new byte[size];
41             }
42     
43             MessageDigest md;
44     
45             final int size;
46     
47             int next = 0; // 0 <= next < size
48     
49             final byte[] data;
50     
51             // data is next..size-1, 0..next-1
52             public void push(byte b) {
53                 data[next++] = b;
54                 if (next == size) {
55                     next = 0;
56                 }
57             }
58     
59             public void reset() {
60                 next = 0;
61                 for (int i = 0; i < size; i++) {
62                     data[i] = 0;
63                 }
64             }
65     
66             public void push(String s) {
67                 for (byte b : UTF8.getBytes(s)) {
68                     push(b);
69                 }
70             }
71     
72             public void pushHash(Object x) {
73                 push(x.hashCode());
74             }
75     
76             public void push(int x) {
77                 push((byte) (x));
78                 push((byte) (x >> 8));
79                 push((byte) (x >> 16));
80                 push((byte) (x >> 24));
81             }
82     
83             public int getHash() {
84                 md.update(primer);
85                 md.update(data, next, size - next);
86                 md.update(data, 0, next);
87                 byte[] hash = md.digest();
88                 int result = (hash[0] & 0xff) | (hash[1] & 0xff) << 8 | (hash[2] & 0xff) << 16 | (hash[3] & 0x7f) << 24;
89                 return result;
90             }
91     
92             public int getPriority() {
93                 int hash = getHash();
94     
95                 if ((hash & 0x1ff0) == 0) {
96                     hash = hash & 0xf;
97                     if (hash < 1) {
98                         return Priorities.HIGH_PRIORITY;
99                     } else if (hash < 1 + 2) {
100                         return Priorities.NORMAL_PRIORITY;
101                     } else if (hash < 1 + 2 + 4) {
102                         return Priorities.LOW_PRIORITY;
103                     } else {
104                         return Priorities.IGNORE_PRIORITY;
105                     }
106                 } else {
107                     return Priorities.IGNORE_PRIORITY + 1;
108                 }
109             }
110         }

114         final BugAccumulator accumulator;

116         final HashQueue hq;

120         public Noise(BugReporter bugReporter) throws NoSuchAlgorithmException {
121             //        this.bugReporter = bugReporter;
122             this.accumulator = new BugAccumulator(bugReporter);
123             hq = new HashQueue(24);
124         }
125     
126         @Override
127         public void visit(Code code) {
128             primer = UTF8.getBytes(getFullyQualifiedMethodName());
129     
130             super.visit(code); // make callbacks to sawOpcode for all opcodes
131             accumulator.reportAccumulatedBugs();
132         }
133     
134         @Override
135         public void sawInt(int i) {
136             hq.push(i);
137         }
138     
139         @Override
140         public void sawLong(long x) {
141             hq.push((int) (x >> 0));
142             hq.push((int) (x >> 32));
143         }
144     
145         @Override
146         public void sawString(String s) {
147             hq.pushHash(s);
148         }
149     
150         @Override
151         public void sawClass() {
152             hq.push(getClassConstantOperand());
153         }
154     
155         @Override
156         public void sawOpcode(int seen) {
157             int priority;
158             switch (seen) {
159             case INVOKEINTERFACE:
160             case INVOKEVIRTUAL:
161             case INVOKESPECIAL:
162             case INVOKESTATIC:
163                 hq.pushHash(getClassConstantOperand());
164                 if (getNameConstantOperand().indexOf('$') == -1) {
165                     hq.pushHash(getNameConstantOperand());
166                 }
167                 hq.pushHash(getSigConstantOperand());
168     
169                 priority = hq.getPriority();
170                 if (priority <= Priorities.LOW_PRIORITY) {
171                     accumulator.accumulateBug(new BugInstance(this, "NOISE_METHOD_CALL", priority).addClassAndMethod(this)
172                             .addCalledMethod(this), this);
173                 }
174                 break;
175             case GETFIELD:
176             case PUTFIELD:
177             case GETSTATIC:
178             case PUTSTATIC:
179                 hq.pushHash(getClassConstantOperand());
180                 if (getNameConstantOperand().indexOf('$') == -1) {
181                     hq.pushHash(getNameConstantOperand());
182                 }
183                 hq.pushHash(getSigConstantOperand());
184                 priority = hq.getPriority();
185                 if (priority <= Priorities.LOW_PRIORITY) {
186                     accumulator.accumulateBug(new BugInstance(this, "NOISE_FIELD_REFERENCE", priority).addClassAndMethod(this)
187                             .addReferencedField(this), this);
188                 }
189                 break;
190             case CHECKCAST:
191             case INSTANCEOF:
192             case NEW:
193                 hq.pushHash(getClassConstantOperand());
194                 break;
195             case IFEQ:
196             case IFNE:
197             case IFNONNULL:
198             case IFNULL:
199             case IF_ICMPEQ:
200             case IF_ICMPNE:
201             case IF_ICMPLE:
202             case IF_ICMPGE:
203             case IF_ICMPGT:
204             case IF_ICMPLT:
205             case IF_ACMPEQ:
206             case IF_ACMPNE:
207             case RETURN:
208             case ARETURN:
209             case IRETURN:
210             case MONITORENTER:
211             case MONITOREXIT:
212             case IINC:
213             case NEWARRAY:
214             case TABLESWITCH:
215             case LOOKUPSWITCH:
216             case LCMP:
217             case INEG:
218             case IADD:
219             case IMUL:
220             case ISUB:
221             case IDIV:
222             case IREM:
223             case IXOR:
224             case ISHL:
225             case ISHR:
226             case IUSHR:
227             case IAND:
228             case IOR:
229             case LAND:
230             case LOR:
231             case LADD:
232             case LMUL:
233             case LSUB:
234             case LDIV:
235             case LSHL:
236             case LSHR:
237             case LUSHR:
238             case AALOAD:
239             case AASTORE:
240             case IALOAD:
241             case IASTORE:
242             case BALOAD:
243             case BASTORE:
244                 hq.push(seen);
245                 priority = hq.getPriority();
246                 if (priority <= Priorities.LOW_PRIORITY) {
247                     accumulator.accumulateBug(
248                             new BugInstance(this, "NOISE_OPERATION", priority).addClassAndMethod(this).addString(OPCODE_NAMES[seen]),
249                             this);
250                 }
251                 break;
252             default:
253                 break;
254             }
255         }
