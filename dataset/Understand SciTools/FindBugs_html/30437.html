22     package edu.umd.cs.findbugs.filter;
23     
24     import static org.junit.Assert.*;
25     
26     import java.io.ByteArrayOutputStream;
27     import java.io.IOException;
28     
29     import org.apache.tools.ant.filters.StringInputStream;
30     import org.junit.Before;
31     import org.junit.Test;
32     
33     import edu.umd.cs.findbugs.BugInstance;
34     import edu.umd.cs.findbugs.ClassAnnotation;
35     import edu.umd.cs.findbugs.SourceLineAnnotation;
36     import edu.umd.cs.findbugs.xml.OutputStreamXMLOutput;
37     import edu.umd.cs.findbugs.xml.XMLOutput;
38     
39     public class SourceMatcherTest {
40     
41         private BugInstance bug;
42         private String fileName;
43     
44         @Before
45         public void setup() {
46             bug = new BugInstance("UUF_UNUSED_FIELD", 0);
47             fileName = "bla.groovy";
48         }
49     
50         @Test
51         public void writeXML() throws Exception {
52             SourceMatcher sm = new SourceMatcher(fileName);
53     
54             String xmlOutput = writeXMLAndGetStringOutput(sm, false);
55             assertEquals("<Source name=\"" + fileName + "\"/>", xmlOutput);
56     
57             sm = new SourceMatcher(fileName);
58             xmlOutput = writeXMLAndGetStringOutput(sm, true);
59             assertEquals("<Source name=\"" + fileName + "\" disabled=\"true\"/>", xmlOutput);
60         }
61     
62         @Test
63         public void readXML() throws Exception {
64             SourceMatcher sm = new SourceMatcher(fileName);
65     
66             String xml = writeXMLAndGetStringOutput(sm, false);
67             xml = "<?xml version=\"1.0\" encoding=\"UTF-8\"?>"
68                     + "\n<FindBugsFilter>"
69                     + "\n<Match>"
70                     + "\n" + xml
71                     + "\n</Match>"
72                     + "\n</FindBugsFilter>\n";
73     
74             Filter filter = new Filter(new StringInputStream(xml));
75     
76             assertFalse(filter.match(bug));
77     
78             bug.addClass("bla", fileName);
79             assertTrue(filter.match(bug));
80         }
81     
82     
83         @Test
84         public void match() throws Exception {
85             SourceMatcher sm = new SourceMatcher(fileName);
86     
87             // no source set: test incomplete data
88             assertFalse(sm.match(bug));
89     
90             bug.addClass("bla", null);
91             assertFalse(sm.match(bug));
92     
93             ClassAnnotation primaryClass = bug.getPrimaryClass();
94             primaryClass.setSourceLines(SourceLineAnnotation.createUnknown("bla", ""));
95             assertFalse(sm.match(bug));
96     
97             // set right source file
98             primaryClass.setSourceLines(SourceLineAnnotation.createUnknown("bla", fileName));
99     
100             // exact match
101             assertTrue(sm.match(bug));
102     
103             // regexp first part
104             sm = new SourceMatcher("~bla.*");
105             assertTrue(sm.match(bug));
106     
107             sm = new SourceMatcher("~blup.*");
108             assertFalse(sm.match(bug));
109     
110             // regexp second part
111             sm = new SourceMatcher("~.*\\.groovy");
112             assertTrue(sm.match(bug));
113     
114             sm = new SourceMatcher("~.*\\.java");
115             assertFalse(sm.match(bug));
116         }
117     
118         private String writeXMLAndGetStringOutput(SourceMatcher matcher, boolean disabled) throws IOException {
119             ByteArrayOutputStream outputStream = new ByteArrayOutputStream();
120             XMLOutput xmlOutput = new OutputStreamXMLOutput(outputStream);
121     
122             matcher.writeXML(xmlOutput, disabled);
123             xmlOutput.finish();
124     
125             return outputStream.toString().trim();
126         }

128     
129     }
