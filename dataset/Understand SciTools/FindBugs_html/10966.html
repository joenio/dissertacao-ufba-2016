20     package edu.umd.cs.findbugs;
21     
22     import java.io.BufferedOutputStream;
23     import java.io.FileOutputStream;
24     import java.util.HashSet;
25     import java.util.Iterator;
26     import java.util.Set;
27     
28     import org.dom4j.Document;
29     import org.dom4j.Element;
30     import org.dom4j.io.OutputFormat;
31     import org.dom4j.io.XMLWriter;
32     
33     import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
34     import edu.umd.cs.findbugs.xml.XMLUtil;
35     
36     /**
37      * Add human-readable messages to a dom4j tree containing FindBugs XML output.
38      * This transformation makes it easier to generate reports (such as HTML) from
39      * the XML.
40      *
41      * @see BugCollection
42      * @author David Hovemeyer
43      */
44     public class AddMessages {
45         private final BugCollection bugCollection;
46     
47         private final Document document;
48     
49         /**
50          * Constructor.
51          *
52          * @param bugCollection
53          *            the BugCollection the dom4j was generated from
54          * @param document
55          *            the dom4j tree
56          */
57         public AddMessages(BugCollection bugCollection, Document document) {
58             this.bugCollection = bugCollection;
59             this.document = document;
60         }
61     
62         /**
63          * Add messages to the dom4j tree.
64          */
65         public void execute() {
66             Iterator<?> elementIter = XMLUtil.selectNodes(document, "/BugCollection/BugInstance").iterator();
67             Iterator<BugInstance> bugInstanceIter = bugCollection.iterator();
68     
69             Set<String> bugTypeSet = new HashSet<String>();
70             Set<String> bugCategorySet = new HashSet<String>();
71             Set<String> bugCodeSet = new HashSet<String>();
72     
73             // Add short and long descriptions to BugInstance elements.
74             // We rely on the Document and the BugCollection storing
75             // the bug instances in the same order.
76             while (elementIter.hasNext() && bugInstanceIter.hasNext()) {
77                 Element element = (Element) elementIter.next();
78                 BugInstance bugInstance = bugInstanceIter.next();
79     
80                 String bugType = bugInstance.getType();
81                 bugTypeSet.add(bugType);
82     
83                 BugPattern bugPattern = bugInstance.getBugPattern();
84     
85                 bugCategorySet.add(bugPattern.getCategory());
86                 bugCodeSet.add(bugPattern.getAbbrev());
87     
88                 element.addElement("ShortMessage").addText(bugPattern.getShortDescription());
89                 element.addElement("LongMessage").addText(bugInstance.getMessage());
90     
91                 // Add pre-formatted display strings in "Message"
92                 // elements for all bug annotations.
93                 Iterator<?> annElementIter = element.elements().iterator();
94                 Iterator<BugAnnotation> annIter = bugInstance.annotationIterator();
95                 while (annElementIter.hasNext() && annIter.hasNext()) {
96                     Element annElement = (Element) annElementIter.next();
97                     BugAnnotation ann = annIter.next();
98                     annElement.addElement("Message").addText(ann.toString());
99                 }
100             }
101     
102             // Add BugPattern elements for each referenced bug types.
103             addBugCategories(bugCategorySet);
104             addBugPatterns(bugTypeSet);
105             addBugCodes(bugCodeSet);
106         }
107     
108         /**
109          * Add BugCategory elements.
110          *
111          * @param bugCategorySet
112          *            all bug categories referenced in the BugCollection
113          */
114         private void addBugCategories(Set<String> bugCategorySet) {
115             Element root = document.getRootElement();
116             for (String category : bugCategorySet) {
117                 Element element = root.addElement("BugCategory");
118                 element.addAttribute("category", category);
119                 Element description = element.addElement("Description");
120                 description.setText(I18N.instance().getBugCategoryDescription(category));
121     
122                 BugCategory bc = DetectorFactoryCollection.instance().getBugCategory(category);
123                 if (bc != null) { // shouldn't be null
124                     String s = bc.getAbbrev();
125                     if (s != null) {
126                         Element abbrev = element.addElement("Abbreviation");
127                         abbrev.setText(s);
128                     }
129                     s = bc.getDetailText();
130                     if (s != null) {
131                         Element details = element.addElement("Details");
132                         details.setText(s);
133                     }
134                 }
135             }
136         }
137     
138         /**
139          * Add BugCode elements.
140          *
141          * @param bugCodeSet
142          *            all bug codes (abbrevs) referenced in the BugCollection
143          */
144         private void addBugCodes(Set<String> bugCodeSet) {
145             Element root = document.getRootElement();
146             for (String bugCode : bugCodeSet) {
147                 Element element = root.addElement("BugCode");
148                 element.addAttribute("abbrev", bugCode);
149                 Element description = element.addElement("Description");
150                 description.setText(I18N.instance().getBugTypeDescription(bugCode));
151             }
152         }
153     
154         private void addBugPatterns(Set<String> bugTypeSet) {
155             Element root = document.getRootElement();
156             for (String bugType : bugTypeSet) {
157                 BugPattern bugPattern = DetectorFactoryCollection.instance().lookupBugPattern(bugType);
158                 if (bugPattern == null) {
159                     continue;
160                 }
161                 Element details = root.addElement("BugPattern");
162                 details.addAttribute("type", bugType).addAttribute("abbrev", bugPattern.getAbbrev())
163                 .addAttribute("category", bugPattern.getCategory());
164                 details.addElement("ShortDescription").addText(bugPattern.getShortDescription());
165                 details.addElement("Details").addCDATA(bugPattern.getDetailText());
166             }
167         }
168     
169         @SuppressFBWarnings("DM_EXIT")
170         public static void main(String[] args) throws Exception {
171             if (args.length != 2) {
172                 System.err.println("Usage: " + AddMessages.class.getName() + " <input collection> <output collection>");
173                 System.exit(1);
174             }
175     
176             // Load plugins, in order to get message files
177             DetectorFactoryCollection.instance();
178     
179             String inputFile = args[0];
180             String outputFile = args[1];
181             Project project = new Project();
182     
183             SortedBugCollection inputCollection = new SortedBugCollection(project);
184             inputCollection.readXML(inputFile);
185     
186             Document document = inputCollection.toDocument();
187     
188             AddMessages addMessages = new AddMessages(inputCollection, document);
189             addMessages.execute();
190     
191             XMLWriter writer = new XMLWriter(new BufferedOutputStream(new FileOutputStream(outputFile)),
192                     OutputFormat.createPrettyPrint());
193             writer.write(document);
194             writer.close();
195         }
196     }
