20     package edu.umd.cs.findbugs.detect;
21     
22     import java.util.HashSet;
23     
24     import org.apache.bcel.classfile.AnnotationEntry;
25     import org.apache.bcel.classfile.Constant;
26     import org.apache.bcel.classfile.ConstantCP;
27     import org.apache.bcel.classfile.ConstantMethodHandle;
28     import org.apache.bcel.classfile.ConstantNameAndType;
29     import org.apache.bcel.classfile.ConstantPool;
30     import org.apache.bcel.classfile.ConstantUtf8;
31     import org.apache.bcel.classfile.Method;
32     
33     import edu.umd.cs.findbugs.BugInstance;
34     import edu.umd.cs.findbugs.BugReporter;
35     import edu.umd.cs.findbugs.BytecodeScanningDetector;
36     import edu.umd.cs.findbugs.MethodAnnotation;
37     import edu.umd.cs.findbugs.StatelessDetector;
38     import edu.umd.cs.findbugs.ba.ClassContext;
39     import edu.umd.cs.findbugs.util.ClassName;
40     
41     /**
42      * Detector to find private methods that are never called.
43      */
44     public class FindUncalledPrivateMethods extends BytecodeScanningDetector implements StatelessDetector {
45         private final BugReporter bugReporter;
46     
47         private String className;
48     
49         private HashSet<MethodAnnotation> definedPrivateMethods, calledMethods;
50     
51         private HashSet<String> calledMethodNames;
52     
53         public FindUncalledPrivateMethods(BugReporter bugReporter) {
54             this.bugReporter = bugReporter;
55         }
56     
57         @Override
58         public void visitMethod(Method obj) {
59             if (!obj.isPrivate() || obj.isSynthetic()) {
60                 return;
61             }
62             super.visitMethod(obj);
63             String methodName = getMethodName();
64             if (!"writeReplace".equals(methodName) && !"readResolve".equals(methodName)
65                     && !"readObject".equals(methodName) && !"readObjectNoData".equals(methodName)
66                     && !"writeObject".equals(methodName)
67                     && methodName.indexOf("debug") == -1 && methodName.indexOf("Debug") == -1
68                     && methodName.indexOf("trace") == -1 && methodName.indexOf("Trace") == -1
69                     && !"<init>".equals(methodName) && !"<clinit>".equals(methodName)) {
70                 for(AnnotationEntry a : obj.getAnnotationEntries()) {
71                     String typeName =  a.getAnnotationType();
72                     if ("Ljavax/annotation/PostConstruct;".equals(typeName)
73                             || "Ljavax/annotation/PreDestroy;".equals(typeName)) {
74                         return;
75                     }
76                 }
77                 definedPrivateMethods.add(MethodAnnotation.fromVisitedMethod(this));
78             }
79         }
80     
81         @Override
82         public void sawOpcode(int seen) {
83             switch (seen) {
84             case INVOKEVIRTUAL:
85             case INVOKESPECIAL:
86             case INVOKESTATIC:
87                 if (getDottedClassConstantOperand().equals(className)) {
88                     String className = getDottedClassConstantOperand();
89                     MethodAnnotation called = new MethodAnnotation(className, getNameConstantOperand(), getSigConstantOperand(),
90                             seen == INVOKESTATIC);
91                     calledMethods.add(called);
92                     calledMethodNames.add(getNameConstantOperand().toLowerCase());
93                 }
94                 break;
95             default:
96                 break;
97             }
98         }
99     
100         @Override
101         public void visitClassContext(ClassContext classContext) {
102             definedPrivateMethods = new HashSet<MethodAnnotation>();
103             calledMethods = new HashSet<MethodAnnotation>();
104             calledMethodNames = new HashSet<String>();
105             className = classContext.getJavaClass().getClassName();
106             String[] parts = className.split("[$+.]");
107             String simpleClassName = parts[parts.length - 1];
108     
109             ConstantPool cp = classContext.getJavaClass().getConstantPool();
110             for(Constant constant : cp.getConstantPool()) {
111                 if(constant instanceof ConstantMethodHandle) {
112                     int kind = ((ConstantMethodHandle) constant).getReferenceKind();
113                     if(kind >= 5 && kind <= 9) {
114                         Constant ref = cp.getConstant(((ConstantMethodHandle)constant).getReferenceIndex());
115                         if(ref instanceof ConstantCP) {
116                             String className = cp.getConstantString(((ConstantCP) ref).getClassIndex(), CONSTANT_Class);
117                             ConstantNameAndType nameAndType = (ConstantNameAndType) cp.getConstant(((ConstantCP) ref).getNameAndTypeIndex());
118                             String name = ((ConstantUtf8)cp.getConstant(nameAndType.getNameIndex())).getBytes();
119                             String signature = ((ConstantUtf8)cp.getConstant(nameAndType.getSignatureIndex())).getBytes();
120                             MethodAnnotation called = new MethodAnnotation(ClassName.toDottedClassName(className), name, signature, kind==6 /* invokestatic */);
121                             calledMethods.add(called);
122                             calledMethodNames.add(name.toLowerCase());
123                         }
124                     }
125                 }
126             }
127     
128             super.visitClassContext(classContext);
129     
130             definedPrivateMethods.removeAll(calledMethods);
131     
132             for (MethodAnnotation m : definedPrivateMethods) {
133                 // System.out.println("Checking " + m);
134                 int priority = LOW_PRIORITY;
135                 String methodName = m.getMethodName();
136                 if (methodName.equals(simpleClassName) && "()V".equals(m.getMethodSignature())) {
137                     continue;
138                 }
139                 if (methodName.length() > 1 && calledMethodNames.contains(methodName.toLowerCase())) {
140                     priority = NORMAL_PRIORITY;
141                 }
142                 BugInstance bugInstance = new BugInstance(this, "UPM_UNCALLED_PRIVATE_METHOD", priority).addClass(this).addMethod(m);
143                 bugReporter.reportBug(bugInstance);
144             }
145     
146             definedPrivateMethods = null;
147             calledMethods = null;
148         }
149     }
