20     package edu.umd.cs.findbugs.gui2;
21     
22     import java.awt.BorderLayout;
23     import java.awt.event.ActionEvent;
24     import java.awt.event.ActionListener;
25     import java.util.ArrayList;
26     import java.util.List;
27     
28     import javax.swing.JButton;
29     import javax.swing.JLabel;
30     import javax.swing.JPanel;
31     
32     import edu.umd.cs.findbugs.filter.Matcher;
33     
34     
35     /**
36      * Allows you to make a new Filter by right clicking (control clicking) on a bug
37      * in the tree
38      */
39     @SuppressWarnings("serial")
40     public class NewFilterFromBug extends FBDialog {
41     
42         private static final List<NewFilterFromBug> listOfAllFrames = new ArrayList<NewFilterFromBug>();
43     
44         public NewFilterFromBug(final FilterFromBugPicker filterFromBugPicker, final ApplyNewFilter applyNewFilter) {
45             this.setModal(true);
46             listOfAllFrames.add(this);
47             setLayout(new BorderLayout());
48     
49             add(new JLabel("Filter out all bugs whose..."), BorderLayout.NORTH);
50     
51             JPanel center = filterFromBugPicker.pickerPanel();
52             add(center, BorderLayout.CENTER);
53     
54             JPanel south = new JPanel();
55             JButton okButton = new JButton(edu.umd.cs.findbugs.L10N.getLocalString("dlg.ok_btn", "OK"));
56     
57             okButton.addActionListener(new ActionListener() {
58                 @Override
59                 public void actionPerformed(ActionEvent evt) {
60                     Matcher matcherFromSelection = filterFromBugPicker.makeMatcherFromSelection();
61                     applyNewFilter.fromMatcher(matcherFromSelection);
62                     closeDialog();
63                 }
64             });
65             JButton cancelButton = new JButton(edu.umd.cs.findbugs.L10N.getLocalString("dlg.cancel_btn", "Cancel"));
66             cancelButton.addActionListener(new ActionListener() {
67                 @Override
68                 public void actionPerformed(ActionEvent evt) {
69                     closeDialog();
70                 }
71             });
72             GuiUtil.addOkAndCancelButtons(south, okButton, cancelButton);
73             add(south, BorderLayout.SOUTH);
74     
75             pack();
76             setVisible(true);
77         }
78     
79         private final void closeDialog() {
80             NewFilterFromBug.this.dispose();
81         }
82     
83         static void closeAll() {
84             for (NewFilterFromBug frame : listOfAllFrames) {
85                 frame.dispose();
86             }
87             listOfAllFrames.clear();
88         }
89     }
