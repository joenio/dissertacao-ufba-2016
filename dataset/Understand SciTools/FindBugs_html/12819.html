20     package edu.umd.cs.findbugs.gui2;
21     
22     import java.awt.Component;
23     import java.awt.Container;
24     import java.awt.Dialog;
25     import java.awt.Frame;
26     
27     import javax.swing.JDialog;
28     
29     /**
30      * All Dialogs are FBDialogs so font size will work.
31      *
32      * @author Kristin
33      *
34      */
35     @SuppressWarnings("serial")
36     public class FBDialog extends JDialog {
37     
38         public FBDialog() {
39             super(MainFrame.getInstance());
40         }
41     
42         public FBDialog(Frame f) {
43             super(f);
44         }
45     
46         public FBDialog(Dialog d) {
47             super(d);
48         }
49     
50         /**
51          * Sets size of font
52          *
53          * @param size
54          */
55         protected void setFontSize(float size) {
56             setFont(this.getFont().deriveFont(size));
57     
58             setFontSizeHelper(this.getComponents(), size);
59         }
60     
61         /*
62          * Helps above method, runs through all components recursively.
63          */
64         protected void setFontSizeHelper(Component[] comps, float size) {
65             if (comps.length <= 0) {
66                 return;
67             }
68     
69             for (Component comp : comps) {
70                 comp.setFont(comp.getFont().deriveFont(size));
71                 if (comp instanceof Container) {
72                     setFontSizeHelper(((Container) comp).getComponents(), size);
73                 }
74             }
75         }
76     
77         @Override
78         public void addNotify() {
79             super.addNotify();
80     
81             setFontSize(Driver.getFontSize());
82         }
83     
84     }
