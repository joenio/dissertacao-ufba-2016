148         public void setCloudId(@Nullable String cloudId) {
149             if (cloudId != null && cloudId.indexOf('.') == -1) {
150                 Map<String, CloudPlugin> registeredClouds = DetectorFactoryCollection.instance().getRegisteredClouds();
151                 String check = "." + cloudId;
152                 int count = 0;
153                 String result = cloudId;
154                 for(String name : registeredClouds.keySet()) {
155                     if (name.endsWith(check)) {
156                         count++;
157                         result = name;
158                     }
159                 }
160                 if (count == 1) {
161                     cloudId = result;
162                 }
163             }
164             this.cloudId = cloudId;
165         }

510              * Add a worklist item.
511              *
512              * @param item
513              *            the WorkListItem representing a zip/jar file to be
514              *            examined

548         /**
549          * Return the list of implicit classpath entries. The implicit classpath is
550          * computed from the closure of the set of jar files that are referenced by
551          * the <code>"Class-Path"</code> attribute of the manifest of the any jar
552          * file that is part of this project or by the <code>"Class-Path"</code>
553          * attribute of any directly or indirectly referenced jar. The referenced
554          * jar files that exist are the list of implicit classpath entries.
555          *
556          * @deprecated FindBugs2 and ClassPathBuilder take care of this
557          *             automatically
558          */
559         @Deprecated
560         public List<String> getImplicitClasspathEntryList() {
561             final LinkedList<String> implicitClasspath = new LinkedList<String>();
562             WorkList workList = new WorkList();
563     
564             // Prime the worklist by adding the zip/jar files
565             // in the project.
566             for (String fileName : analysisTargets) {
567                 try {
568                     URL url = workList.createURL(fileName);
569                     WorkListItem item = new WorkListItem(url);
570                     workList.add(item);
571                 } catch (MalformedURLException ignore) {
572                     // Ignore
573                 }
574             }
575     
576             // Scan recursively.
577             while (!workList.isEmpty()) {
578                 WorkListItem item = workList.getNextItem();
579                 processComponentJar(item.getURL(), workList, implicitClasspath);
580             }
581     
582             return implicitClasspath;
583         }

585         /**
586          * Examine the manifest of a single zip/jar file for implicit classapth
587          * entries.
588          *
589          * @param jarFileURL
590          *            URL of the zip/jar file
591          * @param workList
592          *            worklist of zip/jar files to examine
593          * @param implicitClasspath
594          *            list of implicit classpath entries found
595          */
596         private void processComponentJar(URL jarFileURL, WorkList workList, List<String> implicitClasspath) {
597     
598             if (DEBUG) {
599                 System.out.println("Processing " + jarFileURL.toString());
600             }
601     
602             if (!jarFileURL.toString().endsWith(".zip") && !jarFileURL.toString().endsWith(".jar")) {
603                 return;
604             }
605     
606             try {
607                 URL manifestURL = new URL("jar:" + jarFileURL.toString() + "!/META-INF/MANIFEST.MF");
608     
609                 InputStream in = null;
610                 try {
611                     in = manifestURL.openStream();
612                     Manifest manifest = new Manifest(in);
613     
614                     Attributes mainAttrs = manifest.getMainAttributes();
615                     String classPath = mainAttrs.getValue("Class-Path");
616                     if (classPath != null) {
617                         String[] fileList = classPath.split("\\s+");
618     
619                         for (String jarFile : fileList) {
620                             URL referencedURL = workList.createRelativeURL(jarFileURL, jarFile);
621                             if (workList.add(new WorkListItem(referencedURL))) {
622                                 implicitClasspath.add(referencedURL.toString());
623                                 if (DEBUG) {
624                                     System.out.println("Implicit jar: " + referencedURL.toString());
625                                 }
626                             }
627                         }
628                     }
629                 } finally {
630                     if (in != null) {
631                         in.close();
632                     }
633                 }
634             } catch (IOException ignore) {
635                 // Ignore
636             }
637         }

704         public static Project readXML(File f) throws IOException,  SAXException {
705             InputStream in = new BufferedInputStream(new FileInputStream(f));
706             Project project = new Project();
707             try {
708                 String tag = Util.getXMLType(in);
709                 SAXBugCollectionHandler handler;
710                 if ("Project".equals(tag)) {
711                     handler = new SAXBugCollectionHandler(project, f);
712                 } else if ("BugCollection".equals(tag)) {
713                     SortedBugCollection bugs = new SortedBugCollection(project);
714                     handler = new SAXBugCollectionHandler(bugs, f);
715                 } else {
716                     throw new IOException("Can't load a project from a " + tag + " file");
717                 }
718     
719                 XMLReader xr = XMLReaderFactory.createXMLReader();
720     
721                 xr.setContentHandler(handler);
722                 xr.setErrorHandler(handler);
723     
724                 Reader reader = Util.getReader(in);
725     
726                 xr.parse(new InputSource(reader));
727             } finally {
728                 in.close();
729             }
730     
731             // Presumably, project is now up-to-date
732             project.setModified(false);
733     
734             return project;
735         }

836         public void writeXML(XMLOutput xmlOutput, @CheckForNull File destination, @CheckForNull BugCollection bugCollection)
837                 throws IOException {
838             {
839                 XMLAttributeList attributeList = new XMLAttributeList();
840     
841                 if (getProjectName() != null) {
842                     attributeList = attributeList.addAttribute(PROJECTNAME_ATTRIBUTE_NAME, getProjectName());
843                 }
844                 xmlOutput.openTag(BugCollection.PROJECT_ELEMENT_NAME, attributeList);
845             }
846     
847             if (destination != null) {
848                 String base = destination.getParent();
849                 writeElementList(xmlOutput, JAR_ELEMENT_NAME, convertToRelative(analysisTargets, base));
850                 writeElementList(xmlOutput, AUX_CLASSPATH_ENTRY_ELEMENT_NAME, convertToRelative(auxClasspathEntryList, base));
851                 writeElementList(xmlOutput, SRC_DIR_ELEMENT_NAME, convertToRelative(srcDirList, base));
852                 List<String> cwdStrings = new ArrayList<String>();
853                 for (File file : currentWorkingDirectoryList) {
854                     cwdStrings.add(file.getPath());
855                 }
856                 XMLOutputUtil.writeElementList(xmlOutput, WRK_DIR_ELEMENT_NAME, convertToRelative(cwdStrings, base));
857             } else {
858                 // TODO to allow relative paths: refactor the code which uses null
859                 // file arguments
860                 writeElementList(xmlOutput, JAR_ELEMENT_NAME, analysisTargets);
861                 writeElementList(xmlOutput, AUX_CLASSPATH_ENTRY_ELEMENT_NAME, auxClasspathEntryList);
862                 writeElementList(xmlOutput, SRC_DIR_ELEMENT_NAME, srcDirList);
863                 XMLOutputUtil.writeFileList(xmlOutput, WRK_DIR_ELEMENT_NAME, currentWorkingDirectoryList);
864             }
865     
866             if (!suppressionFilter.isEmpty()) {
867                 xmlOutput.openTag("SuppressionFilter");
868                 suppressionFilter.writeBodyAsXML(xmlOutput);
869                 xmlOutput.closeTag("SuppressionFilter");
870             }
871     
872             for(Map.Entry<String, Boolean> e : enabledPlugins.entrySet()) {
873                 String pluginId = e.getKey();
874                 Boolean enabled = e.getValue();
875                 Plugin plugin = Plugin.getByPluginId(pluginId);
876                 if (plugin == null || enabled == null) {
877                     continue;
878                 }
879                 XMLAttributeList pluginAttributeList = new XMLAttributeList();
880                 pluginAttributeList.addAttribute(PLUGIN_ID_ATTRIBUTE_NAME, plugin.getPluginId());
881                 pluginAttributeList.addAttribute(PLUGIN_STATUS_ELEMENT_NAME, enabled.toString());
882                 xmlOutput.openCloseTag(PLUGIN_ELEMENT_NAME, pluginAttributeList);
883             }
884             CloudPlugin cloudPlugin = bugCollection == null ? null : CloudFactory.getCloudPlugin(bugCollection);
885             if (cloudPlugin != null) {
886                 String id = cloudPlugin.getId();
887                 if (id == null) {
888                     id = cloudId;
889                 }
890                 xmlOutput.startTag(CLOUD_ELEMENT_NAME);
891                 xmlOutput.addAttribute(CLOUD_ID_ATTRIBUTE_NAME, id);
892                 boolean onlineCloud = cloudPlugin.isOnline();
893                 xmlOutput.addAttribute("online", Boolean.toString(onlineCloud));
894                 String url = cloudPlugin.getProperties().getProperty("cloud.detailsUrl");
895                 if (url != null) {
896                     xmlOutput.addAttribute("detailsUrl", url);
897                 }
898                 xmlOutput.stopTag(false);
899                 for (Map.Entry<?,?> e : cloudProperties.entrySet()) {
900                     xmlOutput.startTag(CLOUD_PROPERTY_ELEMENT_NAME);
901                     xmlOutput.addAttribute("key", e.getKey().toString());
902                     xmlOutput.stopTag(false);
903                     Object value = e.getValue();
904                     xmlOutput.writeText(value.toString());
905                     xmlOutput.closeTag(CLOUD_PROPERTY_ELEMENT_NAME);
906                 }
907                 xmlOutput.closeTag(CLOUD_ELEMENT_NAME);
908             }
909             xmlOutput.closeTag(BugCollection.PROJECT_ELEMENT_NAME);
910         }

927         /**
928          * Converts a full path to a relative path if possible
929          *
930          * @param srcFile
931          *            path to convert
932          * @return the converted filename
933          */
934         private String convertToRelative(String srcFile, String base) {
935             String slash = SystemProperties.getProperty("file.separator");
936     
937             if (FILE_IGNORE_CASE) {
938                 srcFile = srcFile.toLowerCase();
939                 base = base.toLowerCase();
940             }
941     
942             if (base.equals(srcFile)) {
943                 return ".";
944             }
945     
946             if (!base.endsWith(slash)) {
947                 base = base + slash;
948             }
949     
950             if (base.length() <= srcFile.length()) {
951                 String root = srcFile.substring(0, base.length());
952                 if (root.equals(base)) {
953                     // Strip off the base directory, make relative
954                     return "." + SystemProperties.getProperty("file.separator") + srcFile.substring(base.length());
955                 }
956             }
957     
958             // See if we can build a relative path above the base using .. notation
959             int slashPos = srcFile.indexOf(slash);
960             int branchPoint;
961             if (slashPos >= 0) {
962                 String subPath = srcFile.substring(0, slashPos);
963                 if ((subPath.length() == 0) || base.startsWith(subPath)) {
964                     branchPoint = slashPos + 1;
965                     slashPos = srcFile.indexOf(slash, branchPoint);
966                     while (slashPos >= 0) {
967                         subPath = srcFile.substring(0, slashPos);
968                         if (base.startsWith(subPath)) {
969                             branchPoint = slashPos + 1;
970                         } else {
971                             break;
972                         }
973                         slashPos = srcFile.indexOf(slash, branchPoint);
974                     }
975     
976                     int slashCount = 0;
977                     slashPos = base.indexOf(slash, branchPoint);
978                     while (slashPos >= 0) {
979                         slashCount++;
980                         slashPos = base.indexOf(slash, slashPos + 1);
981                     }
982     
983                     StringBuilder path = new StringBuilder();
984                     String upDir = ".." + slash;
985                     for (int i = 0; i < slashCount; i++) {
986                         path.append(upDir);
987                     }
988                     path.append(srcFile.substring(branchPoint));
989                     return path.toString();
990                 }
991             }
992             return srcFile;
993         }

1031         /**
1032          * Make the given filename absolute relative to the current working
1033          * directory candidates.
1034          *
1035          * If the given filename exists in more than one of the working directories,
1036          * a list of these existing absolute paths is returned.
1037          *
1038          * The returned list is guaranteed to be non-empty. The returned paths might
1039          * exist or not exist and might be relative or absolute.
1040          *
1041          * @return A list of at least one candidate path for the given filename.
1042          */
1043         private List<String> makeAbsoluteCwdCandidates(String fileName) {
1044             List<String> candidates = new ArrayList<String>();
1045     
1046             boolean hasProtocol = (URLClassPath.getURLProtocol(fileName) != null);
1047             if (hasProtocol) {
1048                 candidates.add(fileName);
1049                 return candidates;
1050             }
1051     
1052             if (new File(fileName).isAbsolute()) {
1053                 candidates.add(fileName);
1054                 return candidates;
1055             }
1056     
1057             for (File currentWorkingDirectory : currentWorkingDirectoryList) {
1058                 File relativeToCurrent = new File(currentWorkingDirectory, fileName);
1059                 if (relativeToCurrent.exists()) {
1060                     candidates.add(relativeToCurrent.toString());
1061                 }
1062             }
1063     
1064             if (candidates.isEmpty()) {
1065                 candidates.add(fileName);
1066             }
1067     
1068             return candidates;
1069         }

1151             List<String> result = new ArrayList<String>();
1152             for (String s : srcDirList) {
1153                 boolean hasProtocol = (URLClassPath.getURLProtocol(s) != null);
1154                 if (hasProtocol) {
1155                     result.add(s);
1156                     continue;
1157                 }
1158                 File f = new File(s);

1160                     if (f.canRead()) {
1161                         result.add(s);
1162                     }
1163                     continue;
1164                 }
1165                 for (File d : currentWorkingDirectoryList) {
1166                     if (d.canRead() && d.isDirectory()) {
1167                         File a = new File(d, s);
1168                         if (a.canRead()) {
1169                             result.add(a.getAbsolutePath());
1170                         }
1171     
1172                     }
1173                 }

1175             }
