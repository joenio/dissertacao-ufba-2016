28     import edu.umd.cs.findbugs.BugAccumulator;

37     /**
38      * We found a problem with the new OpenJDK that everyone is now using to compile
39      * and run java code. In particular, the java.util.logging.Logger behavior has
40      * changed. Instead of using strong references, it now uses weak references
41      * internally. That's a reasonable change, but unfortunately some code relies on
42      * the old behavior - when changing logger configuration, it simply drops the
43      * logger reference. That means that the garbage collector is free to reclaim
44      * that memory, which means that the logger configuration is lost.
45      */
46     public class LostLoggerDueToWeakReference extends OpcodeStackDetector {
47         private static final List<MethodDescriptor> methods = Arrays.asList(
48                 new MethodDescriptor("java/util/logging/Logger", "getLogger", "(Ljava/lang/String;)Ljava/util/logging/Logger;", true),
49                 new MethodDescriptor("java/util/logging/Logger", "getLogger", "(Ljava/lang/String;Ljava/lang/String;)Ljava/util/logging/Logger;", true));
50     
51         //    final BugReporter bugReporter;
52     
53         final BugAccumulator bugAccumulator;
54     
55         final HashSet<String> namesOfSetterMethods = new HashSet<String>();
56     
57         public LostLoggerDueToWeakReference(BugReporter bugReporter) {
58             //        this.bugReporter = bugReporter;
59             this.bugAccumulator = new BugAccumulator(bugReporter);
60             namesOfSetterMethods.add("addHandler");
61             namesOfSetterMethods.add("setUseParentHandlers");
62             namesOfSetterMethods.add("setLevel");
63             namesOfSetterMethods.add("setFilter");
64         }
65     
66         @Override
67         public void visitClassContext(ClassContext classContext) {
68             if(hasInterestingMethod(classContext.getJavaClass().getConstantPool(), methods)) {
69                 super.visitClassContext(classContext);
70             }
71         }
72     
73         @Override
74         public void visit(Code code) {
75             if (getMethodSig().indexOf("Logger") == -1) {
76                 sawGetLogger = -1;
77                 loggerEscaped = loggerImported = false;
78                 super.visit(code); // make callbacks to sawOpcode for all opcodes
79                 /*
80                 if (false) {
81                     System.out.println(getFullyQualifiedMethodName());
82                     System.out.printf("%d %s %s\n", sawGetLogger, loggerEscaped, loggerImported);
83                 }
84                  */
85                 if (sawGetLogger >= 0 && !loggerEscaped && !loggerImported) {
86                     bugAccumulator.reportAccumulatedBugs();
87                 } else {
88                     bugAccumulator.clearBugs();
89                 }
90             }
91         }
92     
93         int sawGetLogger;
94     
95         boolean loggerEscaped;
96     
97         boolean loggerImported;
98     
99         @Override
100         public void sawOpcode(int seen) {
101             if (loggerEscaped || loggerImported) {
102                 return;
103             }
104             switch (seen) {
105             case INVOKESTATIC:
106                 if ("java/util/logging/Logger".equals(getClassConstantOperand()) && "getLogger".equals(getNameConstantOperand())) {
107                     OpcodeStack.Item item = stack.getStackItem(0);
108                     if (!"".equals(item.getConstant())) {
109                         sawGetLogger = getPC();
110                     }
111                     break;
112                 }
113                 checkForImport();
114                 break;
115             case INVOKEVIRTUAL:
116                 if ("java/util/logging/Logger".equals(getClassConstantOperand())
117                         && namesOfSetterMethods.contains(getNameConstantOperand())) {
118                     int priority = HIGH_PRIORITY;
119                     if (getMethod().isStatic() && "main".equals(getMethodName()) && "([Ljava/lang/String;)V".equals(getMethodSig())) {
120                         priority = NORMAL_PRIORITY;
121                     }
122     
123                     OpcodeStack.Item item = stack.getItemMethodInvokedOn(this);
124                     BugInstance bug = new BugInstance(this, "LG_LOST_LOGGER_DUE_TO_WEAK_REFERENCE", priority)
125                     .addClassAndMethod(this).addValueSource(item, this);
126                     bugAccumulator.accumulateBug(bug, this);
127                     break;
128                 }
129                 checkForImport();
130                 checkForMethodExportImport();
131                 break;
132     
133             case INVOKEINTERFACE:
134             case INVOKESPECIAL:
135                 checkForImport();
136                 checkForMethodExportImport();
137                 break;
138     
139             case CHECKCAST:
140                 String sig = getClassConstantOperand();
141                 if (sig.indexOf("Logger") >= 0) {
142                     loggerImported = true;
143                 }
144                 break;
145     
146             case GETFIELD:
147             case GETSTATIC:
148                 checkForImport();
149                 break;
150             case PUTFIELD:
151             case PUTSTATIC:
152                 checkForFieldEscape();
153                 break;
154             default:
155                 break;
156             }
157     
158         }
159     
160         private void checkForImport() {
161             if (getSigConstantOperand().endsWith("Logger;")) {
162                 loggerImported = true;
163             }
164         }
165     
166         private void checkForMethodExportImport() {
167             int numArguments = PreorderVisitor.getNumberArguments(getSigConstantOperand());
168             for (int i = 0; i < numArguments; i++) {
169                 OpcodeStack.Item item = stack.getStackItem(i);
170                 if (item.getSignature().endsWith("Logger;")) {
171                     loggerEscaped = true;
172                 }
173             }
174             String sig = getSigConstantOperand();
175             int pos = sig.indexOf(')');
176             int loggerPos = sig.indexOf("Logger");
177             if (0 <= loggerPos && loggerPos < pos) {
178                 loggerEscaped = true;
179             }
180         }
181     
182         private void checkForFieldEscape() {
183             String sig = getSigConstantOperand();
184             if (sig.indexOf("Logger") >= 0) {
185                 loggerEscaped = true;
186             }
187             OpcodeStack.Item item = stack.getStackItem(0);
188             if (item.getSignature().endsWith("Logger;")) {
189                 loggerEscaped = true;
190             }
191     
192         }

194     }
