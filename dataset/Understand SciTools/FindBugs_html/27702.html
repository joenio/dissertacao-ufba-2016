20     package edu.umd.cs.findbugs.ba;
21     
22     import java.util.BitSet;
23     import java.util.Iterator;
24     
25     import javax.annotation.CheckForNull;
26     
27     import org.apache.bcel.generic.InstructionHandle;
28     
29     
30     /**
31      * A dataflow analysis to compute dominator relationships between basic blocks.
32      * Use the {@link #getResultFact} method to get the dominator set for a given
33      * basic block. The dominator sets are represented using the
34      * {@link java.util.BitSet} class, with the individual bits corresponding to the
35      * IDs of basic blocks.
36      * <p/>
37      * <p>
38      * Subclasses extend this class to compute either dominators or postdominators.
39      * <p/>
40      * <p>
41      * An EdgeChooser may be specified to select which edges to take into account.
42      * For example, exception edges could be ignored.
43      * </p>
44      *
45      * @author David Hovemeyer
46      * @see DataflowAnalysis
47      * @see CFG
48      * @see BasicBlock
49      */
50     public abstract class AbstractDominatorsAnalysis extends BasicAbstractDataflowAnalysis<BitSet> {
51         private final CFG cfg;
52     
53         private final EdgeChooser edgeChooser;
54     
55         /**
56          * Constructor.
57          *
58          * @param cfg
59          *            the CFG to compute dominator relationships for
60          * @param ignoreExceptionEdges
61          *            true if exception edges should be ignored
62          */
63         public AbstractDominatorsAnalysis(CFG cfg, final boolean ignoreExceptionEdges) {
64             this(cfg, new EdgeChooser() {
65                 @Override
66                 public boolean choose(Edge edge) {
67                     if (ignoreExceptionEdges && edge.isExceptionEdge()) {
68                         return false;
69                     } else {
70                         return true;
71                     }
72                 }
73             });
74         }
75     
76         /**
77          * Constructor.
78          *
79          * @param cfg
80          *            the CFG to compute dominator relationships for
81          * @param edgeChooser
82          *            EdgeChooser to choose which Edges to consider significant
83          */
84         public AbstractDominatorsAnalysis(CFG cfg, EdgeChooser edgeChooser) {
85             this.cfg = cfg;
86             this.edgeChooser = edgeChooser;
87         }
88     
89         @Override
90         public BitSet createFact() {
91             return new BitSet();
92         }
93     
94         @Override
95         public void copy(BitSet source, BitSet dest) {
96             dest.clear();
97             dest.or(source);
98         }
99     
100         @Override
101         public void initEntryFact(BitSet result) {
102             // No blocks dominate the entry block
103             result.clear();
104         }
105     
106         @Override
107         public boolean isTop(BitSet fact) {
108             // We represent TOP as a bitset with an illegal bit set
109             return fact.get(cfg.getNumBasicBlocks());
110         }
111     
112         @Override
113         public void makeFactTop(BitSet fact) {
114             // We represent TOP as a bitset with an illegal bit set
115             fact.set(cfg.getNumBasicBlocks());
116         }
117     
118         @Override
119         public boolean same(BitSet fact1, BitSet fact2) {
120             return fact1.equals(fact2);
121         }
122     
123         @Override
124         public void transfer(BasicBlock basicBlock, @CheckForNull InstructionHandle end, BitSet start, BitSet result)
125                 throws DataflowAnalysisException {
126             // Start with intersection of dominators of predecessors
127             copy(start, result);
128     
129             if (!isTop(result)) {
130                 // Every block dominates itself
131                 result.set(basicBlock.getLabel());
132             }
133         }
134     
135         @Override
136         public void meetInto(BitSet fact, Edge edge, BitSet result) throws DataflowAnalysisException {
137             if (!edgeChooser.choose(edge)) {
138                 return;
139             }
140     
141             if (isTop(fact)) {
142                 return;
143             } else if (isTop(result)) {
144                 copy(fact, result);
145             } else {
146                 // Meet is intersection
147                 result.and(fact);
148             }
149         }
150     
151         /**
152          * Get a bitset containing the unique IDs of all blocks which dominate (or
153          * postdominate) the given block.
154          *
155          * @param block
156          *            a BasicBlock
157          * @return BitSet of the unique IDs of all blocks that dominate (or
158          *         postdominate) the BasicBlock
159          */
160         public BitSet getAllDominatorsOf(BasicBlock block) {
161             return getResultFact(block);
162         }
163     
164         /**
165          * Get a bitset containing the unique IDs of all blocks in CFG dominated (or
166          * postdominated, depending on how the analysis was done) by given block.
167          *
168          * @param dominator
169          *            we want to get all blocks dominated (or postdominated) by this
170          *            block
171          * @return BitSet of the ids of all blocks dominated by the given block
172          */
173         public BitSet getAllDominatedBy(BasicBlock dominator) {
174             BitSet allDominated = new BitSet();
175             for (Iterator<BasicBlock> i = cfg.blockIterator(); i.hasNext();) {
176                 BasicBlock block = i.next();
177                 BitSet dominators = getResultFact(block);
178                 if (dominators.get(dominator.getLabel())) {
179                     allDominated.set(block.getLabel());
180                 }
181             }
182             return allDominated;
183         }
184     
185     }
