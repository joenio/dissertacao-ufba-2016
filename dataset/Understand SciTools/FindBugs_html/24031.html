46         /**
47          * Determine the outer class of obj.
48          *
49          * @param obj
50          * @return JavaClass for outer class, or null if obj is not an outer class
51          * @throws ClassNotFoundException
52          */
53     
54         @CheckForNull
55         public static JavaClass getOuterClass(JavaClass obj) throws ClassNotFoundException {
56             for (Attribute a : obj.getAttributes()) {
57                 if (a instanceof InnerClasses) {
58                     for (InnerClass ic : ((InnerClasses) a).getInnerClasses()) {
59                         if (obj.getClassNameIndex() == ic.getInnerClassIndex()) {
60                             // System.out.println("Outer class is " +
61                             // ic.getOuterClassIndex());
62                             ConstantClass oc = (ConstantClass) obj.getConstantPool().getConstant(ic.getOuterClassIndex());
63                             String ocName = oc.getBytes(obj.getConstantPool());
64                             return Repository.lookupClass(ocName);
65                         }
66                     }
67                 }
68             }
69             return null;
70         }

72         public static int getSizeOfSurroundingTryBlock(@CheckForNull Method method, Class<? extends Throwable> exceptionClass, int pc) {
73             if (method == null) {
74                 return Integer.MAX_VALUE;
75             }
76     
77             return getSizeOfSurroundingTryBlock(method, ClassName.toSlashedClassName(exceptionClass), pc);
78         }
79         public static int getSizeOfSurroundingTryBlock(@CheckForNull Method method, @CheckForNull String vmNameOfExceptionClass, int pc) {
80             if (method == null) {
81                 return Integer.MAX_VALUE;
82             }
83             return getSizeOfSurroundingTryBlock(method.getConstantPool(), method.getCode(), vmNameOfExceptionClass, pc);
84         }

86         public static @CheckForNull
87         CodeException getSurroundingTryBlock(ConstantPool constantPool, Code code, @CheckForNull String vmNameOfExceptionClass, int pc) {
88             int size = Integer.MAX_VALUE;
89             if (code.getExceptionTable() == null) {
90                 return null;
91             }
92             CodeException result = null;
93             for (CodeException catchBlock : code.getExceptionTable()) {
94                 if (vmNameOfExceptionClass != null) {
95                     Constant catchType = constantPool.getConstant(catchBlock.getCatchType());
96                     if (catchType == null && !vmNameOfExceptionClass.isEmpty() ||  catchType instanceof ConstantClass
97                             && !((ConstantClass) catchType).getBytes(constantPool).equals(vmNameOfExceptionClass)) {
98                         continue;
99                     }
100                 }
101                 int startPC = catchBlock.getStartPC();
102                 int endPC = catchBlock.getEndPC();
103                 if (pc >= startPC && pc <= endPC) {
104                     int thisSize = endPC - startPC;
105                     if (size > thisSize) {
106                         size = thisSize;
107                         result = catchBlock;
108                     }
109                 }
110             }
111             return result;
112         }

114         public static int getSizeOfSurroundingTryBlock(ConstantPool constantPool, Code code,
115                 @CheckForNull @SlashedClassName String vmNameOfExceptionClass, int pc) {
116             int size = Integer.MAX_VALUE;
117             int tightStartPC = 0;
118             int tightEndPC = Integer.MAX_VALUE;
119             if (code.getExceptionTable() == null) {
120                 return size;
121             }
122             for (CodeException catchBlock : code.getExceptionTable()) {
123                 if (vmNameOfExceptionClass != null) {
124                     Constant catchType = constantPool.getConstant(catchBlock.getCatchType());
125                     if (catchType == null && !vmNameOfExceptionClass.isEmpty() ||  catchType instanceof ConstantClass
126                             && !((ConstantClass) catchType).getBytes(constantPool).equals(vmNameOfExceptionClass)) {
127                         continue;
128                     }
129                 }
130                 int startPC = catchBlock.getStartPC();
131                 int endPC = catchBlock.getEndPC();
132                 if (pc >= startPC && pc <= endPC) {
133                     int thisSize = endPC - startPC;
134                     if (size > thisSize) {
135                         size = thisSize;
136                         tightStartPC = startPC;
137                         tightEndPC = endPC;
138                     }
139                 }
140             }
141             if (size == Integer.MAX_VALUE) {
142                 return size;
143             }
144     
145             // try to guestimate number of lines that correspond
146             size = (size + 7) / 8;
147             LineNumberTable lineNumberTable = code.getLineNumberTable();
148             if (lineNumberTable == null) {
149                 return size;
150             }
151     
152             int count = 0;
153             for (LineNumber line : lineNumberTable.getLineNumberTable()) {
154                 if (line.getStartPC() > tightEndPC) {
155                     break;
156                 }
157                 if (line.getStartPC() >= tightStartPC) {
158                     count++;
159                 }
160             }
161             return count;
162     
163         }
