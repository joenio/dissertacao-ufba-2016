45     /**
46      * Java main application to compute update a historical bug collection with
47      * results from another build/analysis.
48      *
49      * @author William Pugh
50      */
51     
52     public class CopyBuggySource {
53         enum SrcKind {
54             DIR, ZIP, Z0P_GZ;
55             static SrcKind get(File f) {
56                 if (f.exists() && f.isDirectory() && f.canWrite()) {
57                     return DIR;
58                 }
59                 if (!f.exists()) {
60                     if (f.getName().endsWith(".zip")) {
61                         return ZIP;
62                     }
63                     if (f.getName().endsWith(".z0p.gz")) {
64                         return Z0P_GZ;
65                     }
66                 }
67                 throw new IllegalArgumentException("Invalid src destination: " + f);
68             }
69     
70         }
71     
72         /**
73          *
74          */
75         private static final String USAGE = "Usage: <cmd> " + "  <bugs.xml> <destinationSrc>";
76     
77     
78         public static void main(String[] args) throws Exception {
79             FindBugs.setNoAnalysis();
80             DetectorFactoryCollection.instance();
81             if (args.length != 2) {
82                 System.out.println(USAGE);
83                 return;
84             }
85             new CopyBuggySource(args).execute();
86         }
87     
88         SortedBugCollection origCollection;
89         File src;
90         SrcKind kind;
91         ZipOutputStream zOut;
92         byte buf[] = new byte[4096];
93         Project project;
94         SourceFinder sourceFinder;
95         HashSet<String> copied = new HashSet<String>();
96         HashSet<String> couldNotFind = new HashSet<String>();
97         HashSet<String> couldNotCreate = new HashSet<String>();
98         int copyCount = 0;
99         File dstFile;
100     
101         public CopyBuggySource(String[] args) throws Exception {
102             origCollection = new SortedBugCollection();
103             origCollection.readXML(args[0]);
104             project = origCollection.getProject();
105     
106             sourceFinder = new SourceFinder(project);
107             src = new File(args[1]);
108             kind = SrcKind.get(src);
109     
110             switch (kind) {
111             case DIR:
112                 break;
113             case ZIP:
114                 zOut = new ZipOutputStream(new FileOutputStream(src));
115                 break;
116             case Z0P_GZ:
117                 zOut = new ZipOutputStream(new DeflaterOutputStream(new FileOutputStream(src)));
118                 zOut.setLevel(0);
119                 break;
120             }
121     
122         }
123     
124         @SuppressFBWarnings("RV_RETURN_VALUE_IGNORED_BAD_PRACTICE")
125         private void copySourceFile(String fullName, SourceFile sourceFile) {
126             InputStream in = null;
127             OutputStream out = null;
128             try {
129                 long lastModified = sourceFile.getLastModified();
130                 in = sourceFile.getInputStream();
131                 out = getOutputStream(fullName, 0);
132                 if (out == null) {
133                     return;
134                 }
135                 while (true) {
136                     int sz = in.read(buf);
137                     if (sz < 0) {
138                         break;
139                     }
140                     out.write(buf, 0, sz);
141                 }
142                 if (dstFile != null) {
143                     dstFile.setLastModified(lastModified);
144                 }
145                 System.out.println("Copied " + fullName);
146                 copyCount++;
147             } catch (IOException e) {
148                 if (couldNotFind.add(fullName)) {
149                     System.out.println("Problem copying " + fullName);
150                     e.printStackTrace(System.out);
151                 }
152             } finally {
153                 close(in);
154                 close(out);
155             }
156         }
157     
158         private void copySourceForAnnotation(BugAnnotation ann) {
159             SourceLineAnnotation sourceAnnotation;
160             if (ann instanceof BugAnnotationWithSourceLines) {
161                 sourceAnnotation = ((BugAnnotationWithSourceLines) ann).getSourceLines();
162             } else if (ann instanceof SourceLineAnnotation) {
163                 sourceAnnotation = (SourceLineAnnotation) ann;
164             } else {
165                 return;
166             }
167             if (sourceAnnotation == null) {
168                 return;
169             }
170             if (sourceAnnotation.isUnknown()) {
171                 return;
172             }
173     
174             String fullName = SourceFinder.getPlatformName(sourceAnnotation);
175     
176             SourceFile sourceFile;
177             try {
178                 sourceFile = sourceFinder.findSourceFile(sourceAnnotation);
179             } catch (IOException e) {
180     
181                 if (couldNotFind.add(fullName)) {
182                     System.out.println("Did not find " + fullName);
183                 }
184                 return;
185             }
186     
187             if (copied.add(fullName)) {
188                 copySourceFile(fullName, sourceFile);
189             }
190         }
191     
192         public void execute() throws IOException {
193     
194             for (BugInstance bug : origCollection.getCollection()) {
195                 for (Iterator<BugAnnotation> i = bug.annotationIterator(); i.hasNext();) {
196                     copySourceForAnnotation(i.next());
197                 }
198             }
199             if (zOut != null) {
200                 zOut.close();
201             }
202     
203             System.out.printf("All done. %d files not found, %d files copied%n", couldNotFind.size(), copyCount);
204         }
205     
206         private @CheckForNull
207         OutputStream getOutputStream(String fullName, long lastModifiedTime) throws IOException {
208             if (kind == SrcKind.DIR) {
209                 dstFile = new File(src, fullName);
210     
211                 if (dstFile.exists()) {
212                     System.out.println(dstFile + " already exists");
213                     return null;
214                 }
215     
216                 File parent = dstFile.getParentFile();
217                 OutputStream out = null;
218     
219                 if (!parent.mkdirs() && !parent.isDirectory()) {
220                     String path = parent.getPath();
221                     if (couldNotCreate.add(path)) {
222                         System.out.println("Can't create directory for " + path);
223                     }
224                     return null;
225                 }
226     
227                 return new FileOutputStream(dstFile);
228             } else {
229                 ZipEntry e = new ZipEntry(fullName);
230                 e.setTime(lastModifiedTime);
231                 zOut.putNextEntry(e);
232                 return zOut;
233             }
234     
235         }
236     
237         public static void close(InputStream in) {
238             try {
239                 if (in != null) {
240                     in.close();
241                 }
242             } catch (IOException e) {
243             }
244     
245         }
246     
247         public static void close(OutputStream out) {
248             try {
249                 if (out instanceof ZipOutputStream) {
250                     ((ZipOutputStream) out).closeEntry();
251                 } else if (out != null) {
252                     out.close();
253                 }
254             } catch (IOException e) {
255             }
256     
257         }
