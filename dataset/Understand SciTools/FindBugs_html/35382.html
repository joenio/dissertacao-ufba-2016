144         public void execute() {
145             Profiler profiler = Global.getAnalysisCache().getProfiler();
146             profiler.start(this.getClass());
147             try {
148                 // Do the null-value analysis
149                 this.invDataflow = classContext.getIsNullValueDataflow(method);
150                 this.vnaDataflow = classContext.getValueNumberDataflow(method);
151                 if (findGuaranteedDerefs) {
152                     if (DEBUG_DEREFS) {
153                         System.out.println("Checking for guaranteed derefs in "
154                                 + classContext.getClassDescriptor().getDottedClassName() + "." + method.getName()
155                                 + method.getSignature());
156                     }
157                     this.uvdDataflow = classContext.getUnconditionalValueDerefDataflow(method);
158                 }
159     
160                 // Check method and report potential null derefs and
161                 // redundant null comparisons.
162                 examineBasicBlocks();
163                 if (findGuaranteedDerefs) {
164                     examineNullValues();
165                 }
166                 examineRedundantBranches();
167             } catch (MissingClassException e) {
168                 AnalysisContext.reportMissingClass(e.getClassNotFoundException());
169             } catch (CheckedAnalysisException e) {
170                 AnalysisContext.logError("Error while checking guaranteed derefs in "
171                         + classContext.getClassDescriptor().getDottedClassName() + "." + method.getName() + method.getSignature(), e);
172             } finally {
173                 profiler.end(this.getClass());
174             }

178         /**
179          * Examine basic blocks for null checks and potentially-redundant null
180          * comparisons.
181          *
182          * @throws DataflowAnalysisException
183          * @throws CFGBuilderException
184          */
185         private void examineBasicBlocks() throws DataflowAnalysisException, CFGBuilderException {
186             // Look for null check blocks where the reference being checked
187             // is definitely null, or null on some path
188             Iterator<BasicBlock> bbIter = invDataflow.getCFG().blockIterator();
189             while (bbIter.hasNext()) {
190                 BasicBlock basicBlock = bbIter.next();
191     
192                 if (basicBlock.isNullCheck()) {
193                     analyzeNullCheck(invDataflow, basicBlock);
194                 } else if (!basicBlock.isEmpty()) {
195                     // Look for all reference comparisons where
196                     // - both values compared are definitely null, or
197                     // - one value is definitely null and one is definitely not null
198                     // These cases are not null dereferences,
199                     // but they are quite likely to indicate an error, so while
200                     // we've got
201                     // information about null values, we may as well report them.
202                     InstructionHandle lastHandle = basicBlock.getLastInstruction();
203                     Instruction last = lastHandle.getInstruction();
204                     switch (last.getOpcode()) {
205                     case Constants.IF_ACMPEQ:
206                     case Constants.IF_ACMPNE:
207                         analyzeRefComparisonBranch(basicBlock, lastHandle);
208                         break;
209                     case Constants.IFNULL:
210                     case Constants.IFNONNULL:
211                         analyzeIfNullBranch(basicBlock, lastHandle);
212                         break;
213                     default:
214                         break;
215                     }
216                 }
217             }
218         }

220         /**
221          * Examine null values. Report any that are guaranteed to be dereferenced on
222          * non-implicit-exception paths.
223          *
224          * @throws CFGBuilderException
225          * @throws DataflowAnalysisException
226          */
227         private void examineNullValues() throws CFGBuilderException, DataflowAnalysisException {
228             Set<LocationWhereValueBecomesNull> locationWhereValueBecomesNullSet = invDataflow.getAnalysis()
229                     .getLocationWhereValueBecomesNullSet();
230             if (DEBUG_DEREFS) {
231                 System.out.println("----------------------- examineNullValues " + locationWhereValueBecomesNullSet.size());
232             }
233     
234             Map<ValueNumber, SortedSet<Location>> bugStatementLocationMap = new HashMap<ValueNumber, SortedSet<Location>>();
235             // Inspect the method for locations where a null value is guaranteed to
236             // be dereferenced. Add the dereference locations
237             Map<ValueNumber, NullValueUnconditionalDeref> nullValueGuaranteedDerefMap = new HashMap<ValueNumber, NullValueUnconditionalDeref>();
238     
239             // Check every location
240             CFG cfg = classContext.getCFG(method);
241             for (Iterator<Location> i = cfg.locationIterator(); i.hasNext();) {
242                 Location location = i.next();
243     
244                 if (DEBUG_DEREFS) {
245                     System.out.println("At location " + location);
246                 }
247     
248                 checkForUnconditionallyDereferencedNullValues(location, bugStatementLocationMap, nullValueGuaranteedDerefMap,
249                         vnaDataflow.getFactAtLocation(location), invDataflow.getFactAtLocation(location),
250                         uvdDataflow.getFactAfterLocation(location), false);
251             }
252             HashSet<ValueNumber> npeIfStatementCovered = new HashSet<ValueNumber>(nullValueGuaranteedDerefMap.keySet());
253             Map<ValueNumber, SortedSet<Location>> bugEdgeLocationMap = new HashMap<ValueNumber, SortedSet<Location>>();
254     
255             checkEdges(cfg, nullValueGuaranteedDerefMap, bugEdgeLocationMap);
256     
257             Map<ValueNumber, SortedSet<Location>> bugLocationMap = bugEdgeLocationMap;
258             bugLocationMap.putAll(bugStatementLocationMap);
259     
260             // For each value number that is null somewhere in the
261             // method, collect the set of locations where it becomes null.
262             // FIXME: we may see some locations that are not guaranteed to be
263             // dereferenced (how to fix this?)
264             Map<ValueNumber, Set<Location>> nullValueAssignmentMap = findNullAssignments(locationWhereValueBecomesNullSet);
265     
266             reportBugs(nullValueGuaranteedDerefMap, npeIfStatementCovered, bugLocationMap, nullValueAssignmentMap);
267         }

269         public Map<ValueNumber, Set<Location>> findNullAssignments(Set<LocationWhereValueBecomesNull> locationWhereValueBecomesNullSet) {
270             Map<ValueNumber, Set<Location>> nullValueAssignmentMap = new HashMap<ValueNumber, Set<Location>>();
271             for (LocationWhereValueBecomesNull lwvbn : locationWhereValueBecomesNullSet) {
272                 if (DEBUG_DEREFS) {
273                     System.out.println("OOO " + lwvbn);
274                 }
275                 Set<Location> locationSet = nullValueAssignmentMap.get(lwvbn.getValueNumber());
276                 if (locationSet == null) {
277                     locationSet = new HashSet<Location>(4);
278                     nullValueAssignmentMap.put(lwvbn.getValueNumber(), locationSet);
279                 }
280                 locationSet.add(lwvbn.getLocation());
281                 if (DEBUG_DEREFS) {
282                     System.out.println(lwvbn.getValueNumber() + " becomes null at " + lwvbn.getLocation());
283                 }
284             }
285             return nullValueAssignmentMap;
286         }

288         public void reportBugs(Map<ValueNumber, NullValueUnconditionalDeref> nullValueGuaranteedDerefMap,
289                 HashSet<ValueNumber> npeIfStatementCovered, Map<ValueNumber, SortedSet<Location>> bugLocationMap,
290                 Map<ValueNumber, Set<Location>> nullValueAssignmentMap) throws CFGBuilderException, DataflowAnalysisException {
291             // Report
292             for (Map.Entry<ValueNumber, NullValueUnconditionalDeref> e : nullValueGuaranteedDerefMap.entrySet()) {
293                 ValueNumber valueNumber = e.getKey();
294                 Set<Location> derefLocationSet = e.getValue().getDerefLocationSet();
295                 Set<Location> assignedNullLocationSet = nullValueAssignmentMap.get(valueNumber);
296                 if (assignedNullLocationSet == null) {
297                     if (DEBUG_DEREFS) {
298                         String where = classContext.getJavaClass().getClassName() + "." + method.getName() + ":"
299                                 + method.getSignature();
300                         System.out.println("Problem at " + where);
301                         System.out.println("Value number " + valueNumber);
302                         for (Location loc : derefLocationSet) {
303                             System.out.println("Dereference at " + loc);
304                         }
305                     }
306                     // TODO: figure out why this is failing
307                     //                if (false) {
308                     //                    assert false : "No assigned NullLocationSet for " + valueNumber + " in " + nullValueAssignmentMap.keySet()
309                     //                            + " while analyzing " + classContext.getJavaClass().getClassName() + "." + method.getName();
310                     //                }
311                     assignedNullLocationSet = Collections.<Location> emptySet();
312                 }
313                 SortedSet<Location> knownNullAndDoomedAt = bugLocationMap.get(valueNumber);
314     
315                 BugAnnotation variableAnnotation = null;
316                 try {
317                     for (Location loc : derefLocationSet) {
318                         variableAnnotation = ValueNumberSourceInfo.findAnnotationFromValueNumber(method, loc, valueNumber,
319                                 vnaDataflow.getFactAtLocation(loc), "VALUE_OF");
320                         if (variableAnnotation != null) {
321                             break;
322                         }
323                     }
324                     if (variableAnnotation == null) {
325                         for (Location loc : knownNullAndDoomedAt) {
326                             variableAnnotation = ValueNumberSourceInfo.findAnnotationFromValueNumber(method, loc, valueNumber,
327                                     vnaDataflow.getFactAtLocation(loc), "VALUE_OF");
328                             if (variableAnnotation != null) {
329                                 break;
330                             }
331                         }
332                     }
333                     if (variableAnnotation == null) {
334                         for (Location loc : assignedNullLocationSet) {
335                             variableAnnotation = ValueNumberSourceInfo.findAnnotationFromValueNumber(method, loc, valueNumber,
336                                     vnaDataflow.getFactAtLocation(loc), "VALUE_OF");
337                             if (variableAnnotation != null) {
338                                 break;
339                             }
340                         }
341                     }
342     
343                 } catch (DataflowAnalysisException e2) {
344                 }
345                 if (variableAnnotation == null) {
346                     variableAnnotation = new LocalVariableAnnotation("?", -1, derefLocationSet.iterator().next().getHandle()
347                             .getPosition());
348                 }
349     
350                 if (PRUNE_GUARANTEED_DEREFERENCES) {
351                     PostDominatorsAnalysis postDomAnalysis = classContext.getNonExceptionPostDominatorsAnalysis(method);
352     
353                     removeStrictlyPostDominatedLocations(derefLocationSet, postDomAnalysis);
354     
355                     removeStrictlyPostDominatedLocations(assignedNullLocationSet, postDomAnalysis);
356                     DominatorsAnalysis domAnalysis = classContext.getNonExceptionDominatorsAnalysis(method);
357                     removeStrictlyDominatedLocations(knownNullAndDoomedAt, domAnalysis);
358     
359                 }
360     
361                 collector.foundGuaranteedNullDeref(assignedNullLocationSet, derefLocationSet, knownNullAndDoomedAt, vnaDataflow,
362                         valueNumber, variableAnnotation, e.getValue(), npeIfStatementCovered.contains(valueNumber));
363             }
364         }

366         public void checkEdges(CFG cfg, Map<ValueNumber, NullValueUnconditionalDeref> nullValueGuaranteedDerefMap,
367                 Map<ValueNumber, SortedSet<Location>> bugEdgeLocationMap) throws DataflowAnalysisException {
368             // Check every non-exception control edge
369             for (Iterator<Edge> i = cfg.edgeIterator(); i.hasNext();) {
370                 Edge edge = i.next();
371     
372                 UnconditionalValueDerefSet uvdFact = uvdDataflow.getFactOnEdge(edge);
373     
374                 if (uvdFact.isEmpty()) {
375                     continue;
376                 }
377     
378                 if (edge.isExceptionEdge()) {
379                     if (DEBUG_DEREFS) {
380                         System.out.println("On exception edge " + edge.formatAsString(false));
381                     }
382     
383                     // continue;
384                 }
385     
386                 if (DEBUG_DEREFS) {
387                     System.out.println("On edge " + edge.formatAsString(false));
388                 }
389     
390                 BasicBlock source = edge.getSource();
391                 ValueNumberFrame vnaFact = vnaDataflow.getResultFact(source);
392                 IsNullValueFrame invFact = invDataflow.getFactAtMidEdge(edge);
393     
394                 Location location = null;
395                 if (edge.isExceptionEdge()) {
396                     BasicBlock b = cfg.getSuccessorWithEdgeType(source, EdgeTypes.FALL_THROUGH_EDGE);
397                     if (b != null) {
398                         location = new Location(source.getExceptionThrower(), b);
399                     }
400                 } else {
401                     location = Location.getLastLocation(source);
402                 }
403     
404                 if (location != null) {
405                     Instruction in = location.getHandle().getInstruction();
406                     if (assertionMethods.isAssertionInstruction(in, classContext.getConstantPoolGen())) {
407                         if (DEBUG_DEREFS) {
408                             System.out.println("Skipping because it is an assertion method ");
409                         }
410                         continue;
411                     }
412     
413                     checkForUnconditionallyDereferencedNullValues(location, bugEdgeLocationMap, nullValueGuaranteedDerefMap, vnaFact,
414                             invFact, uvdFact, true);
415                 }
416             }
417         }

419         private void removeStrictlyPostDominatedLocations(Set<Location> locations, PostDominatorsAnalysis postDomAnalysis) {
420             BitSet strictlyDominated = new BitSet();
421             for (Location loc : locations) {
422                 BitSet allDominatedBy = postDomAnalysis.getAllDominatedBy(loc.getBasicBlock());
423                 allDominatedBy.clear(loc.getBasicBlock().getLabel());
424                 strictlyDominated.or(allDominatedBy);
425             }
426             LinkedList<Location> locations2 = new LinkedList<Location>(locations);
427     
428             for (Iterator<Location> i = locations.iterator(); i.hasNext();) {
429                 Location loc = i.next();
430                 if (strictlyDominated.get(loc.getBasicBlock().getLabel())) {
431                     i.remove();
432                     continue;
433                 }
434                 for (Location loc2 : locations2) {
435                     if (loc.getBasicBlock().equals(loc2.getBasicBlock())
436                             && loc.getHandle().getPosition() > loc2.getHandle().getPosition()) {
437                         i.remove();
438                         break;
439                     }
440                 }
441             }
442         }

583                 }
584             }
585         }

587         /**
588          * Note the locations where a known-null value is unconditionally
589          * dereferenced.
590          *
591          * @param thisLocation
592          * @param bugLocations
593          * @param nullValueGuaranteedDerefMap
594          *            map of null values to sets of Locations where they are derefed
595          * @param derefSet
596          *            set of values known to be unconditionally dereferenced
597          * @param isNullValue
598          *            the null value
599          * @param valueNumber
600          *            the value number of the null value
601          */
602         private void noteUnconditionallyDereferencedNullValue(Location thisLocation,
603                 Map<ValueNumber, SortedSet<Location>> bugLocations,
604                 Map<ValueNumber, NullValueUnconditionalDeref> nullValueGuaranteedDerefMap, UnconditionalValueDerefSet derefSet,
605                 IsNullValue isNullValue, ValueNumber valueNumber) {
606             if (DEBUG) {
607                 System.out.println("%%% HIT for value number " + valueNumber + " @ " + thisLocation);
608             }
609             Set<Location> unconditionalDerefLocationSet = derefSet.getUnconditionalDerefLocationSet(valueNumber);
610             if (unconditionalDerefLocationSet.isEmpty()) {
611                 AnalysisContext.logError("empty set of unconditionally dereferenced locations at "
612                         + thisLocation.getHandle().getPosition() + " in " + classContext.getClassDescriptor() + "."
613                         + method.getName() + method.getSignature());
614                 return;
615             }
616     
617             // OK, we have a null value that is unconditionally
618             // derferenced. Make a note of the locations where it
619             // will be dereferenced.
620             NullValueUnconditionalDeref thisNullValueDeref = nullValueGuaranteedDerefMap.get(valueNumber);
621             if (thisNullValueDeref == null) {
622                 thisNullValueDeref = new NullValueUnconditionalDeref();
623                 nullValueGuaranteedDerefMap.put(valueNumber, thisNullValueDeref);
624             }
625             thisNullValueDeref.add(isNullValue, unconditionalDerefLocationSet);
626     
627             if (thisLocation != null) {
628                 SortedSet<Location> locationsForThisBug = bugLocations.get(valueNumber);
629     
630                 if (locationsForThisBug == null) {
631                     locationsForThisBug = new TreeSet<Location>();
632                     bugLocations.put(valueNumber, locationsForThisBug);
633                 }
634                 locationsForThisBug.add(thisLocation);
635             }
636         }

638         /**
639          * Examine redundant branches.
640          */
641         private void examineRedundantBranches() {
642             for (RedundantBranch redundantBranch : redundantBranchList) {
643                 if (DEBUG) {
644                     System.out.println("Redundant branch: " + redundantBranch);
645                 }
646                 int lineNumber = redundantBranch.lineNumber;
647     
648                 // The source to bytecode compiler may sometimes duplicate blocks of
649                 // code along different control paths. So, to report the bug,
650                 // we check to ensure that the branch is REALLY determined each
651                 // place it is duplicated, and that it is determined in the same
652                 // way.
653     
654                 boolean confused = undeterminedBranchSet.get(lineNumber)
655                         || (definitelySameBranchSet.get(lineNumber) && definitelyDifferentBranchSet.get(lineNumber));
656     
657                 // confused if there is JSR confusion or multiple null checks with
658                 // different results on the same line
659     
660                 boolean reportIt = true;
661     
662                 if (lineMentionedMultipleTimes.get(lineNumber) && confused) {
663                     reportIt = false;
664                 } else if (redundantBranch.location.getBasicBlock().isInJSRSubroutine() /*
665                  * occurs
666                  * in
667                  * a
668                  * JSR
669                  */
670                         && confused) {
671                     reportIt = false;
672                 } else {
673                     int pc = redundantBranch.location.getHandle().getPosition();
674                     for (CodeException e : method.getCode().getExceptionTable()) {
675                         if (e.getCatchType() == 0 && e.getStartPC() != e.getHandlerPC() && e.getEndPC() <= pc
676                                 && pc <= e.getEndPC() + 5) {
677                             reportIt = false;
678                         }
679                     }
680                 }
681     
682                 if (reportIt) {
683                     collector.foundRedundantNullCheck(redundantBranch.location, redundantBranch);
684                 }
685             }
686         }

688         private void analyzeRefComparisonBranch(BasicBlock basicBlock, InstructionHandle lastHandle) throws DataflowAnalysisException {
689     
690             Location location = new Location(lastHandle, basicBlock);
691     
692             IsNullValueFrame frame = invDataflow.getFactAtLocation(location);
693             if (!frame.isValid()) {
694                 // Probably dead code due to pruning infeasible exception edges.
695                 return;
696             }
697             if (frame.getStackDepth() < 2) {
698                 throw new DataflowAnalysisException("Stack underflow at " + lastHandle);
699             }
700     
701             // Find the line number.
702             int lineNumber = getLineNumber(method, lastHandle);
703             if (lineNumber < 0) {
704                 return;
705             }
706     
707             int numSlots = frame.getNumSlots();
708             IsNullValue top = frame.getValue(numSlots - 1);
709             IsNullValue topNext = frame.getValue(numSlots - 2);
710     
711             boolean definitelySame = top.isDefinitelyNull() && topNext.isDefinitelyNull();
712             boolean definitelyDifferent = (top.isDefinitelyNull() && topNext.isDefinitelyNotNull())
713                     || (top.isDefinitelyNotNull() && topNext.isDefinitelyNull());
714     
715             if (definitelySame || definitelyDifferent) {
716                 if (definitelySame) {
717                     if (DEBUG) {
718                         System.out.println("Line " + lineNumber + " always same");
719                     }
720                     definitelySameBranchSet.set(lineNumber);
721                 }
722                 if (definitelyDifferent) {
723                     if (DEBUG) {
724                         System.out.println("Line " + lineNumber + " always different");
725                     }
726                     definitelyDifferentBranchSet.set(lineNumber);
727                 }
728     
729                 RedundantBranch redundantBranch = new RedundantBranch(location, lineNumber, top, topNext);
730     
731                 // Figure out which control edge is made infeasible by the redundant
732                 // comparison
733                 boolean wantSame = (lastHandle.getInstruction().getOpcode() == Constants.IF_ACMPEQ);
734                 int infeasibleEdgeType = (wantSame == definitelySame) ? EdgeTypes.FALL_THROUGH_EDGE : EdgeTypes.IFCMP_EDGE;
735                 Edge infeasibleEdge = invDataflow.getCFG().getOutgoingEdgeWithType(basicBlock, infeasibleEdgeType);
736                 redundantBranch.setInfeasibleEdge(infeasibleEdge);
737     
738                 if (DEBUG) {
739                     System.out.println("Adding redundant branch: " + redundantBranch);
740                 }
741                 redundantBranchList.add(redundantBranch);
742             } else {
743                 if (DEBUG) {
744                     System.out.println("Line " + lineNumber + " undetermined");
745                 }
746                 undeterminedBranchSet.set(lineNumber);
747             }
748         }

750         /** This is called for both IFNULL and IFNONNULL instructions. */
751         private void analyzeIfNullBranch(BasicBlock basicBlock, InstructionHandle lastHandle) throws DataflowAnalysisException {
752     
753             Location location = new Location(lastHandle, basicBlock);
754     
755             IsNullValueFrame frame = invDataflow.getFactAtLocation(location);
756     
757             if (!frame.isValid()) {
758                 // This is probably dead code due to an infeasible exception edge.
759                 return;
760             }
761     
762             IsNullValue top = frame.getTopValue();
763     
764             // Find the line number.
765             int lineNumber = getLineNumber(method, lastHandle);
766             if (lineNumber < 0) {
767                 return;
768             }
769     
770             if (!(top.isDefinitelyNull() || top.isDefinitelyNotNull())) {
771                 if (DEBUG) {
772                     System.out.println("Line " + lineNumber + " undetermined");
773                 }
774                 undeterminedBranchSet.set(lineNumber);
775                 return;
776             }
777     
778             // Figure out if the branch is always taken
779             // or always not taken.
780             short opcode = lastHandle.getInstruction().getOpcode();
781             boolean definitelySame = top.isDefinitelyNull();
782             if (opcode != Constants.IFNULL) {
783                 definitelySame = !definitelySame;
784             }
785     
786             if (definitelySame) {
787                 if (DEBUG) {
788                     System.out.println("Line " + lineNumber + " always same");
789                 }
790                 definitelySameBranchSet.set(lineNumber);
791             } else {
792                 if (DEBUG) {
793                     System.out.println("Line " + lineNumber + " always different");
794                 }
795                 definitelyDifferentBranchSet.set(lineNumber);
796             }
797     
798             RedundantBranch redundantBranch = new RedundantBranch(location, lineNumber, top);
799     
800             // Determine which control edge is made infeasible by the redundant
801             // comparison
802             boolean wantNull = (opcode == Constants.IFNULL);
803             int infeasibleEdgeType = (wantNull == top.isDefinitelyNull()) ? EdgeTypes.FALL_THROUGH_EDGE : EdgeTypes.IFCMP_EDGE;
804             Edge infeasibleEdge = invDataflow.getCFG().getOutgoingEdgeWithType(basicBlock, infeasibleEdgeType);
805             redundantBranch.setInfeasibleEdge(infeasibleEdge);
806     
807             if (DEBUG) {
808                 System.out.println("Adding redundant branch: " + redundantBranch);
809             }
810             redundantBranchList.add(redundantBranch);
811         }

813         private void analyzeNullCheck(IsNullValueDataflow invDataflow, BasicBlock basicBlock) throws DataflowAnalysisException,
814         CFGBuilderException {
815             // Look for null checks where the value checked is definitely
816             // null or null on some path.
817     
818             InstructionHandle exceptionThrowerHandle = basicBlock.getExceptionThrower();
819             Instruction exceptionThrower = exceptionThrowerHandle.getInstruction();
820     
821             // Get the stack values at entry to the null check.
822             IsNullValueFrame frame = invDataflow.getStartFact(basicBlock);
823             if (!frame.isValid()) {
824                 return;
825             }
826     
827             // Could the reference be null?
828             IsNullValue refValue = frame.getInstance(exceptionThrower, classContext.getConstantPoolGen());
829             if (DEBUG) {
830                 System.out.println("For basic block " + basicBlock + " value is " + refValue);
831             }
832             if (refValue.isDefinitelyNotNull())
833             {
834                 return;
835                 //        if (false && !refValue.mightBeNull())
836                 //            return;
837             }
838     
839             if (!refValue.isDefinitelyNull()) {
840                 return;
841             }
842             // Get the value number
843             ValueNumberFrame vnaFrame = classContext.getValueNumberDataflow(method).getStartFact(basicBlock);
844             if (!vnaFrame.isValid()) {
845                 return;
846             }
847             ValueNumber valueNumber = vnaFrame.getInstance(exceptionThrower, classContext.getConstantPoolGen());
848             Location location = new Location(exceptionThrowerHandle, basicBlock);
849             if (DEBUG) {
850                 System.out.println("Warning: VN " + valueNumber + " invf: " + frame + " @ " + location);
851             }
852     
853             boolean isConsistent = true;
854             Iterator<BasicBlock> bbIter = invDataflow.getCFG().blockIterator();
855             LineNumberTable table = method.getLineNumberTable();
856             int position = exceptionThrowerHandle.getPosition();
857             int line = table == null ? 0 : table.getSourceLine(position);
858             while (bbIter.hasNext()) {
859                 BasicBlock bb = bbIter.next();
860     
861                 if (!bb.isNullCheck()) {
862                     continue;
863                 }
864     
865                 InstructionHandle eth = bb.getExceptionThrower();
866                 if (eth == exceptionThrowerHandle) {
867                     continue;
868                 }
869                 if (eth.getInstruction().getOpcode() != exceptionThrower.getOpcode()) {
870                     continue;
871                 }
872     
873                 int ePosition = eth.getPosition();
874                 if (ePosition == position || table != null && line == table.getSourceLine(ePosition)) {
875     
876                     IsNullValueFrame frame2 = invDataflow.getStartFact(bb);
877                     if (!frame2.isValid()) {
878                         continue;
879                     }
880                     // apparent clone
881                     IsNullValue rv = frame2.getInstance(eth.getInstruction(), classContext.getConstantPoolGen());
882                     if (!rv.equals(refValue)) {
883                         isConsistent = false;
884                     }
885                 }
886     
887             }
888             // Issue a warning
889             collector.foundNullDeref(location, valueNumber, refValue, vnaFrame, isConsistent);
890         }
