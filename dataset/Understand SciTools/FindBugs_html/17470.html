20     package edu.umd.cs.findbugs.detect;
21     
22     import org.apache.bcel.classfile.Method;
23     import org.apache.bcel.generic.Type;
24     
25     import edu.umd.cs.findbugs.BugInstance;
26     import edu.umd.cs.findbugs.BugReporter;
27     import edu.umd.cs.findbugs.OpcodeStack;
28     import edu.umd.cs.findbugs.StatelessDetector;
29     import edu.umd.cs.findbugs.SystemProperties;
30     import edu.umd.cs.findbugs.ba.XFactory;
31     import edu.umd.cs.findbugs.ba.XMethod;
32     import edu.umd.cs.findbugs.bcel.OpcodeStackDetector;
33     
34     public class InfiniteRecursiveLoop extends OpcodeStackDetector implements StatelessDetector {
35     
36         private final BugReporter bugReporter;
37     
38         private boolean seenTransferOfControl;
39     
40         private boolean seenReturn;
41     
42         //    private boolean seenThrow;
43     
44         private boolean seenStateChange;
45     
46         private int largestBranchTarget;
47     
48         private final static boolean DEBUG = SystemProperties.getBoolean("irl.debug");
49     
50         public InfiniteRecursiveLoop(BugReporter bugReporter) {
51             this.bugReporter = bugReporter;
52         }
53     
54         @Override
55         public void visit(Method obj) {
56             seenTransferOfControl = false;
57             seenStateChange = false;
58             seenReturn = false;
59             //        seenThrow = false;
60             largestBranchTarget = -1;
61     
62             if (DEBUG) {
63                 System.out.println();
64                 System.out.println(" --- " + getFullyQualifiedMethodName());
65                 System.out.println();
66             }
67         }
68     
69         @Override
70         public void sawBranchTo(int target) {
71             if (target == getNextPC()) {
72                 return;
73             }
74             if (largestBranchTarget < target) {
75                 largestBranchTarget = target;
76             }
77             seenTransferOfControl = true;
78         }
79     
80         /**
81          * Signal an infinite loop if either: we see a call to the same method with
82          * the same parameters, or we see a call to the same (dynamically dispatched
83          * method), and there has been no transfer of control.
84          */
85         @Override
86         public void sawOpcode(int seen) {
87             if (seenReturn && seenTransferOfControl && seenStateChange) {
88                 return;
89             }
90     
91             if (DEBUG) {
92                 System.out.println(stack);
93                 System.out.println(getPC() + " : " + OPCODE_NAMES[seen]);
94             }
95     
96             if ((seen == INVOKEVIRTUAL || seen == INVOKEINTERFACE) && "add".equals(getNameConstantOperand())
97                     && "(Ljava/lang/Object;)Z".equals(getSigConstantOperand()) && stack.getStackDepth() >= 2) {
98                 OpcodeStack.Item it0 = stack.getStackItem(0);
99                 int r0 = it0.getRegisterNumber();
100                 OpcodeStack.Item it1 = stack.getStackItem(1);
101                 int r1 = it1.getRegisterNumber();
102                 if (r0 == r1 && r0 > 0) {
103                     bugReporter.reportBug(new BugInstance(this, "IL_CONTAINER_ADDED_TO_ITSELF", NORMAL_PRIORITY).addClassAndMethod(
104                             this).addSourceLine(this));
105                 }
106             }
107     
108             if ((seen == INVOKEVIRTUAL || seen == INVOKESPECIAL || seen == INVOKEINTERFACE || seen == INVOKESTATIC)
109                     && getNameConstantOperand().equals(getMethodName())
110                     && getSigConstantOperand().equals(getMethodSig())
111                     && (seen == INVOKESTATIC) == getMethod().isStatic()
112                     && (seen == INVOKESPECIAL) == (getMethod().isPrivate() && !getMethod().isStatic() || "<init>".equals(getMethodName()))) {
113                 Type arguments[] = getMethod().getArgumentTypes();
114                 // stack.getStackDepth() >= parameters
115                 int parameters = arguments.length;
116                 if (!getMethod().isStatic()) {
117                     parameters++;
118                 }
119                 XMethod xMethod = XFactory.createReferencedXMethod(this);
120                 if (DEBUG) {
121                     System.out.println("IL: Checking...");
122                     System.out.println(xMethod);
123                     System.out.println("vs. " + getClassName() + "." + getMethodName() + " : " + getMethodSig());
124     
125                 }
126                 if (xMethod.getClassName().replace('.', '/').equals(getClassName()) || seen == INVOKEINTERFACE) {
127                     // Invocation of same method
128                     // Now need to see if parameters are the same
129                     int firstParameter = 0;
130                     if ("<init>".equals(getMethodName())) {
131                         firstParameter = 1;
132                     }
133     
134                     // match1 should be true if it is any call to the exact same
135                     // method
136                     // and no state has been change.
137                     // if match1 is true, the method may not always perform
138                     // a recursive infinite loop, but this particular method call is
139                     // an infinite
140                     // recursive loop
141     
142                     boolean match1 = !seenStateChange;
143                     for (int i = firstParameter; match1 && i < parameters; i++) {
144                         OpcodeStack.Item it = stack.getStackItem(parameters - 1 - i);
145                         if (!it.isInitialParameter() || it.getRegisterNumber() != i) {
146                             match1 = false;
147                         }
148                     }
149     
150                     boolean sameMethod = seen == INVOKESTATIC || "<init>".equals(getNameConstantOperand());
151                     if (!sameMethod) {
152                         // Have to check if first parmeter is the same
153                         // know there must be a this argument
154                         if (DEBUG) {
155                             System.out.println("Stack is " + stack);
156                         }
157                         OpcodeStack.Item p = stack.getStackItem(parameters - 1);
158                         if (DEBUG) {
159                             System.out.println("parameters = " + parameters + ", Item is " + p);
160                         }
161                         String sig = p.getSignature();
162                         sameMethod = p.isInitialParameter() && p.getRegisterNumber() == 0 && sig.equals("L" + getClassName() + ";");
163     
164                     }
165     
166                     // match2 and match3 are two different ways of seeing if the
167                     // call site
168                     // postdominates the method entry.
169     
170                     // technically, we use (!seenTransferOfControl || !seenReturn &&
171                     // largestBranchTarget < getPC())
172                     // as a check that the call site postdominates the method entry.
173                     // If those are true,
174                     // and sameMethod is true, we have a guaranteed IL.
175     
176                     boolean match2 = sameMethod && !seenTransferOfControl;
177                     boolean match3 = sameMethod && !seenReturn && largestBranchTarget < getPC();
178                     if (match1 || match2 || match3) {
179                         if (DEBUG) {
180                             System.out.println("IL: " + sameMethod + " " + match1 + " " + match2 + " " + match3);
181                         }
182                         //                    int priority = HIGH_PRIORITY;
183                         //                    if (!match1 && !match2 && seenThrow)
184                         //                        priority = NORMAL_PRIORITY;
185                         //                    if (seen == INVOKEINTERFACE)
186                         //                        priority = NORMAL_PRIORITY;
187                         bugReporter.reportBug(new BugInstance(this, "IL_INFINITE_RECURSIVE_LOOP", HIGH_PRIORITY).addClassAndMethod(
188                                 this).addSourceLine(this));
189                     }
190                 }
191             }
192     
193             switch (seen) {
194             case ARETURN:
195             case IRETURN:
196             case LRETURN:
197             case RETURN:
198             case DRETURN:
199             case FRETURN:
200                 seenReturn = true;
201                 seenTransferOfControl = true;
202                 break;
203             case ATHROW:
204                 //            seenThrow = true;
205                 seenTransferOfControl = true;
206                 break;
207             case PUTSTATIC:
208             case PUTFIELD:
209             case IASTORE:
210             case AASTORE:
211             case DASTORE:
212             case FASTORE:
213             case LASTORE:
214             case SASTORE:
215             case CASTORE:
216             case BASTORE:
217                 seenStateChange = true;
218                 break;
219             case INVOKEVIRTUAL:
220             case INVOKESPECIAL:
221             case INVOKEINTERFACE:
222             case INVOKESTATIC:
223                 if ("print".equals(getNameConstantOperand()) || "println".equals(getNameConstantOperand())
224                         || "log".equals(getNameConstantOperand()) || "toString".equals(getNameConstantOperand())) {
225                     break;
226                 }
227                 seenStateChange = true;
228                 break;
229             default:
230                 break;
231             }
232         }
233     
234     }
