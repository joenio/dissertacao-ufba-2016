20     package edu.umd.cs.findbugs.anttask;
21     
22     import java.io.File;
23     
24     import org.apache.tools.ant.BuildException;
25     
26     /**
27      * Ant task to invoke the MineBugHistory program in the workflow package
28      *
29      * @author David Hovemeyer
30      * @author Ben Langmead
31      */
32     public class MineBugHistoryTask extends AbstractFindBugsTask {
33     
34         private File outputFile;
35     
36         private String formatDates;
37     
38         private String noTabs;
39     
40         private String summary;
41     
42         private DataFile inputFile;
43     
44         public MineBugHistoryTask() {
45             super("edu.umd.cs.findbugs.workflow.MineBugHistory");
46             setFailOnError(true);
47         }
48     
49         public DataFile createDataFile() {
50             if (inputFile != null) {
51                 throw new BuildException("only one dataFile element is allowed", getLocation());
52             }
53             inputFile = new DataFile();
54             return inputFile;
55         }
56     
57         public void setOutput(File output) {
58             this.outputFile = output;
59         }
60     
61         public void setInput(String input) {
62             this.inputFile = new DataFile();
63             this.inputFile.name = input;
64         }
65     
66         public void setFormatDates(String arg) {
67             this.formatDates = arg;
68         }
69     
70         public void setNoTabs(String arg) {
71             this.noTabs = arg;
72         }
73     
74         public void setSummary(String arg) {
75             this.summary = arg;
76         }
77     
78         private void checkBoolean(String attrVal, String attrName) {
79             if (attrVal == null) {
80                 return;
81             }
82             attrVal = attrVal.toLowerCase();
83             if (!"true".equals(attrVal) && !"false".equals(attrVal)) {
84                 throw new BuildException("attribute " + attrName + " requires boolean value", getLocation());
85             }
86         }
87     
88         /*
89          * (non-Javadoc)
90          *
91          * @see edu.umd.cs.findbugs.anttask.AbstractFindBugsTask#checkParameters()
92          */
93         @Override
94         protected void checkParameters() {
95             super.checkParameters();
96     
97             if (inputFile == null) {
98                 throw new BuildException("inputFile element is required");
99             }
100     
101             checkBoolean(formatDates, "formatDates");
102             checkBoolean(noTabs, "noTabs");
103             checkBoolean(summary, "summary");
104         }
105     
106         public void addBoolOption(String option, String value) {
107             if (value != null) {
108                 addArg(option + ":" + value);
109             }
110         }
111     
112         /*
113          * (non-Javadoc)
114          *
115          * @see
116          * edu.umd.cs.findbugs.anttask.AbstractFindBugsTask#configureFindbugsEngine
117          * ()
118          */
119         @Override
120         protected void configureFindbugsEngine() {
121             addBoolOption("-formatDates", formatDates);
122             addBoolOption("-noTabs", noTabs);
123             addBoolOption("-summary", summary);
124             addArg(inputFile.getName());
125             if (outputFile != null) {
126                 // Don't use .getName() because it discards path
127                 addArg(outputFile.getAbsolutePath());
128             }
129         }
130     
131         /*
132          * (non-Javadoc)
133          *
134          * @see
135          * edu.umd.cs.findbugs.anttask.AbstractFindBugsTask#beforeExecuteJavaProcess
136          * ()
137          */
138         @Override
139         protected void beforeExecuteJavaProcess() {
140             log("running mineBugHistory...");
141         }
142     
143         /*
144          * (non-Javadoc)
145          *
146          * @see
147          * edu.umd.cs.findbugs.anttask.AbstractFindBugsTask#afterExecuteJavaProcess
148          * (int)
149          */
150         @Override
151         protected void afterExecuteJavaProcess(int rc) {
152             if (rc != 0) {
153                 throw new BuildException("execution of " + getTaskName() + " failed");
154             }
155         }
156     
157     }
