48     public final class FindTwoLockWait implements Detector, StatelessDetector {
49     
50         private final BugReporter bugReporter;
51     
52         private JavaClass javaClass;
53     
54         private final Collection<BugInstance> possibleWaitBugs = new LinkedList<BugInstance>();
55     
56         private final Collection<SourceLineAnnotation> possibleNotifyLocations = new LinkedList<SourceLineAnnotation>();
57     
58         public FindTwoLockWait(BugReporter bugReporter) {
59             this.bugReporter = bugReporter;
60         }
61     
62         @Override
63         public Object clone() {
64             try {
65                 return super.clone();
66             } catch (CloneNotSupportedException e) {
67                 throw new AssertionError(e);
68             }
69         }
70     
71         @Override
72         public void visitClassContext(ClassContext classContext) {
73             javaClass = classContext.getJavaClass();
74             possibleWaitBugs.clear();
75             possibleNotifyLocations.clear();
76             Method[] methodList = javaClass.getMethods();
77             for (Method method : methodList) {
78                 MethodGen methodGen = classContext.getMethodGen(method);
79                 if (methodGen == null) {
80                     continue;
81                 }
82     
83                 if (!preScreen(methodGen)) {
84                     continue;
85                 }
86     
87                 try {
88                     analyzeMethod(classContext, method);
89                 } catch (DataflowAnalysisException e) {
90                     // bugReporter.logError("Error analyzing " + method.toString(),
91                     // e);
92                 } catch (CFGBuilderException e) {
93                     bugReporter.logError("Error analyzing " + method.toString(), e);
94                 }
95             }
96             if (!possibleNotifyLocations.isEmpty()) {
97                 for (BugInstance bug : possibleWaitBugs) {
98                     for (SourceLineAnnotation notifyLine : possibleNotifyLocations) {
99                         bug.addSourceLine(notifyLine).describe("SOURCE_NOTIFICATION_DEADLOCK");
100                     }
101                     bugReporter.reportBug(bug);
102                 }
103             }
104         }
105     
106         private void analyzeMethod(ClassContext classContext, Method method) throws CFGBuilderException, DataflowAnalysisException {
107     
108             MethodGen methodGen = classContext.getMethodGen(method);
109             CFG cfg = classContext.getCFG(method);
110             LockDataflow dataflow = classContext.getLockDataflow(method);
111     
112             for (Iterator<Location> j = cfg.locationIterator(); j.hasNext();) {
113                 Location location = j.next();
114                 visitLocation(classContext, location, methodGen, dataflow);
115             }
116         }
117     
118         public boolean preScreen(MethodGen mg) {
119             ConstantPoolGen cpg = mg.getConstantPool();
120     
121             int lockCount = mg.isSynchronized() ? 1 : 0;
122             boolean sawWaitOrNotify = false;
123     
124             InstructionHandle handle = mg.getInstructionList().getStart();
125             while (handle != null && !(lockCount >= 2 && sawWaitOrNotify)) {
126                 Instruction ins = handle.getInstruction();
127                 if (ins instanceof MONITORENTER) {
128                     ++lockCount;
129                 } else if (ins instanceof INVOKEVIRTUAL) {
130                     INVOKEVIRTUAL inv = (INVOKEVIRTUAL) ins;
131                     String methodName = inv.getMethodName(cpg);
132                     if ("wait".equals(methodName) || methodName.startsWith("notify")) {
133                         sawWaitOrNotify = true;
134                     }
135                 }
136     
137                 handle = handle.getNext();
138             }
139     
140             return lockCount >= 2 && sawWaitOrNotify;
141         }
142     
143         public void visitLocation(ClassContext classContext, Location location, MethodGen methodGen, LockDataflow dataflow)
144                 throws DataflowAnalysisException {
145             ConstantPoolGen cpg = methodGen.getConstantPool();
146     
147             if (Hierarchy.isMonitorWait(location.getHandle().getInstruction(), cpg)) {
148                 int count = dataflow.getFactAtLocation(location).getNumLockedObjects();
149                 if (count > 1) {
150                     // A wait with multiple locks held?
151                     String sourceFile = javaClass.getSourceFileName();
152                     possibleWaitBugs.add(new BugInstance(this, "TLW_TWO_LOCK_WAIT", HIGH_PRIORITY).addClassAndMethod(methodGen,
153                             sourceFile).addSourceLine(classContext, methodGen, sourceFile, location.getHandle()));
154                 }
155             }
156             if (Hierarchy.isMonitorNotify(location.getHandle().getInstruction(), cpg)) {
157                 int count = dataflow.getFactAtLocation(location).getNumLockedObjects();
158                 if (count > 1) {
159                     // A notify with multiple locks held?
160                     String sourceFile = javaClass.getSourceFileName();
161                     possibleNotifyLocations.add(SourceLineAnnotation.fromVisitedInstruction(classContext, methodGen, sourceFile,
162                             location.getHandle()));
163                 }
164             }
165         }
166     
167         @Override
168         public void report() {
169         }
170     }
