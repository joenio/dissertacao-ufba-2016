36     /**
37      * A simple class to parse method signatures that include generic information.
38      * <p>
39      *
40      * Modified from edu.umd.cs.findbugs.ba.SignatureParser
41      *
42      * @author Nat Ayewah
43      */
44     public class GenericSignatureParser {
45         private class ParameterSignatureIterator implements Iterator<String> {
46             private int index = 1;
47     
48             @Override
49             public boolean hasNext() {
50                 return index < signature.length() && signature.charAt(index) != ')' && signature.charAt(index) != '^';
51             }
52     
53             @Override
54             public String next() {
55                 if (!hasNext()) {
56                     throw new NoSuchElementException();
57                 }
58                 StringBuilder result = new StringBuilder();
59                 boolean done;
60                 do {
61                     done = true;
62                     char ch = signature.charAt(index);
63                     switch (ch) {
64                     case 'B':
65                     case 'C':
66                     case 'D':
67                     case 'F':
68                     case 'I':
69                     case 'J':
70                     case 'S':
71                     case 'Z':
72                     case '*': // wildcard
73                         result.append(signature.charAt(index));
74                         ++index;
75                         break;
76     
77                     case 'L':
78                     case 'T':
79                         String tmp = "";
80                         int startsemi = index;
81                         int leftCount = 0;
82                         int i = startsemi + 1;
83                         loop: while (true) {
84                             char c = signature.charAt(i);
85                             switch (c) {
86                             case ';':
87                                 if (leftCount == 0) {
88                                     break loop;
89                                 }
90                                 break;
91                             case '<':
92                                 leftCount++;
93                                 break;
94                             case '>':
95                                 leftCount--;
96                                 break;
97                             }
98                             i++;
99     
100                         }
101                         String foo = signature.substring(startsemi, i + 1);
102                         result.append(foo);
103                         index = i + 1;
104                         break;
105     
106                     case '[':
107                     case '+':
108                     case '-':
109                         result.append(signature.charAt(index));
110                         ++index;
111                         done = false;
112                         break;
113     
114                     case ')':
115                     case '^':
116                         throw new NoSuchElementException("Should have already thrown NoSuchElementException");
117                     case 'V':
118                     default:
119                         throw new IllegalStateException("Invalid method signature: '" + signature + "' : "
120                                 + signature.substring(index) + " " + result);
121                     }
122                 } while (!done);
123                 return result.toString();
124             }
125     
126             @Override
127             public void remove() {
128                 throw new UnsupportedOperationException();
129             }
130         }
131     
132         private final String signature;
133     
134         /**
135          * Parses a generic method signature of the form:
136          * <code>(argument_signature)return_type_signature</code>
137          *
138          * @param signature
139          *            the method signature to be parsed
140          */
141         public GenericSignatureParser(String signature) {
142             // XXX not currently handling Type parameters for class, interface or
143             // method definitions
144             int s = signature.indexOf('(');
145             String sig = signature;
146             if (s > 0) {
147                 sig = sig.substring(s);
148             } else if (s < 0 || sig.indexOf(':') >= 0 || sig.startsWith("(V)")) {
149                 throw new IllegalArgumentException("Bad method signature: " + signature);
150             }
151             this.signature = sig;
152         }
153     
154         /**
155          * Get an Iterator over signatures of the method parameters.
156          *
157          * @return Iterator which returns the parameter type signatures in order
158          */
159         public Iterator<String> parameterSignatureIterator() {
160             return new ParameterSignatureIterator();
161         }
162     
163         /**
164          * Get the method return type signature.
165          *
166          * @return the method return type signature
167          */
168         public String getReturnTypeSignature() {
169             int endOfParams = signature.lastIndexOf(')');
170             if (endOfParams < 0) {
171                 throw new IllegalArgumentException("Bad method signature: " + signature);
172             }
173             return signature.substring(endOfParams + 1);
174         }
175     
176         /**
177          * Get the number of parameters in the signature.
178          *
179          * @return the number of parameters
180          */
181         public int getNumParameters() {
182             int count = 0;
183             for (Iterator<String> i = parameterSignatureIterator(); i.hasNext();) {
184                 i.next();
185                 ++count;
186             }
187             return count;
188         }
189     
190         /**
191          * Get the number of parameters passed to method invocation.
192          *
193          * @param inv
194          * @param cpg
195          * @return int number of parameters
196          */
197         public static int getNumParametersForInvocation(InvokeInstruction inv, ConstantPoolGen cpg) {
198             GenericSignatureParser sigParser = new GenericSignatureParser(inv.getSignature(cpg));
199             return sigParser.getNumParameters();
200         }
201     
202         /**
203          * @param target
204          *            the method whose signature is to be parsed
205          * @return an iterator over the parameters of the generic signature of
206          *         method. Returns null if the generic signature cannot be parsed
207          */
208         public static @CheckForNull
209         Iterator<String> getGenericSignatureIterator(Method target) {
210             try {
211                 GenericSignatureParser parser = null;
212                 String genericSignature = null;
213                 for (Attribute a : target.getAttributes()) {
214                     if (a instanceof Signature) {
215     
216                         Signature sig = (Signature) a;
217                         if (genericSignature != null) {
218                             if (!genericSignature.equals(sig.getSignature())) {
219                                 if (false) {
220                                     System.out.println("Inconsistent signatures: ");
221                                     System.out.println(genericSignature);
222                                     System.out.println(sig.getSignature());
223                                 }
224                                 return null; // we've seen two inconsistent
225                                 // signatures
226                             }
227                             continue;
228                         }
229     
230                         genericSignature = sig.getSignature();
231                         if (compareSignatures(target.getSignature(), genericSignature)) {
232                             parser = new GenericSignatureParser(genericSignature);
233                         }
234                     }
235                 }
236                 Iterator<String> iter = parser == null ? null : parser.parameterSignatureIterator();
237                 return iter;
238             } catch (RuntimeException e) {
239             } // degrade gracefully
240             return null;
241         }
242     
243         /**
244          * Compare a plain method signature to the a generic method Signature and
245          * return true if they match
246          */
247         public static boolean compareSignatures(String plainSignature, String genericSignature) {
248             GenericSignatureParser plainParser = new GenericSignatureParser(plainSignature);
249             GenericSignatureParser genericParser = new GenericSignatureParser(genericSignature);
250     
251             if (plainParser.getNumParameters() != genericParser.getNumParameters()) {
252                 return false;
253             }
254     
255             return true;
256         }
257     
258         public static void main(String[] args) {
259             if (args.length != 1) {
260                 System.err.println("Usage: " + GenericSignatureParser.class.getName() + " '<method signature>'");
261                 System.exit(1);
262             }
263             GenericSignatureParser parser = new GenericSignatureParser(args[0]);
264             for (Iterator<String> i = parser.parameterSignatureIterator(); i.hasNext();) {
265                 String s = i.next();
266                 System.out.println(s);
267                 Type t = GenericUtilities.getType(s);
268                 System.out.println("-~- " + t);
269                 if (t instanceof ObjectType) {
270                     System.out.println("-~- " + ((ObjectType) t).toString());
271                 }
272                 if (t != null) {
273                     System.out.println("-~- " + t.getClass());
274                 }
275             }
276             System.out.println(parser.getNumParameters() + " parameter(s)");
277     
278         }
