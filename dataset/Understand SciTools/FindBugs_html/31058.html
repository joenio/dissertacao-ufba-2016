20     package edu.umd.cs.findbugs.workflow;
21     
22     import java.util.HashSet;
23     import java.util.Map;
24     import java.util.TreeMap;
25     
26     import edu.umd.cs.findbugs.BugCollection;
27     import edu.umd.cs.findbugs.DetectorFactoryCollection;
28     import edu.umd.cs.findbugs.FindBugs;
29     import edu.umd.cs.findbugs.PackageStats;
30     import edu.umd.cs.findbugs.Priorities;
31     import edu.umd.cs.findbugs.SortedBugCollection;
32     import edu.umd.cs.findbugs.util.Bag;
33     
34     public class TreemapVisualization {
35     
36         HashSet<String> buggyPackages = new HashSet<String>();
37     
38         HashSet<String> interiorPackages = new HashSet<String>();
39     
40         Bag<String> goodCodeSize = new Bag<String>(new TreeMap<String, Integer>());
41     
42         Bag<String> goodCodeCount = new Bag<String>(new TreeMap<String, Integer>());
43     
44         public void addInteriorPackages(String packageName) {
45             String p = superpackage(packageName);
46             if (p.length() > 0) {
47                 interiorPackages.add(p);
48                 addInteriorPackages(p);
49             }
50         }
51     
52         private static String superpackage(String packageName) {
53             int i = packageName.lastIndexOf('.');
54             if (i == -1) {
55                 return "";
56             }
57             String p = packageName.substring(0, i);
58             return p;
59         }
60     
61         public boolean isInteriorPackage(String packageName) {
62             return interiorPackages.contains(packageName);
63         }
64     
65         public void cleanCode(String packageName, int loc, int classes) {
66             String superpackage = superpackage(packageName);
67             if (buggyPackages.contains(superpackage) || interiorPackages.contains(superpackage) || superpackage.length() == 0) {
68                 goodCodeCount.add(packageName, classes);
69                 goodCodeSize.add(packageName, loc);
70                 if (superpackage.length() > 0) {
71                     interiorPackages.add(superpackage);
72                 }
73     
74             } else {
75                 cleanCode(superpackage, loc, classes);
76             }
77         }
78     
79         public void generateTreeMap(BugCollection bugCollection) {
80             for (PackageStats p : bugCollection.getProjectStats().getPackageStats()) {
81                 if (p.getTotalBugs() > 0) {
82                     buggyPackages.add(p.getPackageName());
83                     addInteriorPackages(p.getPackageName());
84     
85                 }
86             }
87     
88             for (PackageStats p : bugCollection.getProjectStats().getPackageStats()) {
89                 if (p.getTotalBugs() == 0) {
90                     cleanCode(p.getPackageName(), p.size(), p.getClassStats().size());
91                 }
92             }
93             System.out.println("LOC\tTypes\tH\tHM\tDensity");
94             System.out.println("INTEGER\tINTEGER\tINTEGER\tINTEGER\tFLOAT");
95             for (PackageStats p : bugCollection.getProjectStats().getPackageStats()) {
96                 if (p.getTotalBugs() > 0) {
97                     int high = p.getBugsAtPriority(Priorities.HIGH_PRIORITY);
98                     int normal = p.getBugsAtPriority(Priorities.NORMAL_PRIORITY);
99                     System.out.printf("%d\t%d\t%d\t%d\t%g\t\t%s", p.size(), p.getClassStats().size(), high, high + normal,
100                             (high + normal) * 1000.0 / p.size(), p.getPackageName().substring(11).replace('.', '\t'));
101                     if (isInteriorPackage(p.getPackageName())) {
102                         System.out.print("\t*");
103                     }
104                     System.out.println();
105                 }
106             }
107             for (Map.Entry<String, Integer> e : goodCodeSize.entrySet()) {
108                 System.out.printf("%d\t%d\t%d\t%d\t%g\t\t%s%n", e.getValue(), goodCodeCount.getCount(e.getKey()), 0, 0, 0.0, e
109                         .getKey().substring(11).replace('.', '\t'));
110     
111             }
112     
113         }
114     
115         public static void main(String[] args) throws Exception {
116             FindBugs.setNoAnalysis();
117             DetectorFactoryCollection.instance(); // load plugins
118     
119             SortedBugCollection bugCollection = new SortedBugCollection();
120             int argCount = 0;
121             if (argCount < args.length) {
122                 bugCollection.readXML(args[argCount++]);
123             } else {
124                 bugCollection.readXML(System.in);
125             }
126     
127             new TreemapVisualization().generateTreeMap(bugCollection);
128     
129         }
130     }
