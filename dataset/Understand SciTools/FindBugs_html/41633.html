20     package edu.umd.cs.findbugs.ba.obl;
21     
22     import edu.umd.cs.findbugs.ba.Path;
23     
24     /**
25      * Error-handling obligation analysis state. This is a set of obligations and a
26      * program path on which they are outstanding (not cleaned up).
27      *
28      * <p>
29      * See Weimer and Necula, <a href="http://doi.acm.org/10.1145/1028976.1029011"
30      * >Finding and preventing run-time error handling mistakes</a>, OOPSLA 2004.
31      * </p>
32      *
33      * @author David Hovemeyer
34      */
35     public class State {
36         private ObligationSet obligationSet;
37     
38         private Path path;
39     
40         private State() {
41         }
42     
43         public State(/* int maxObligationTypes, */ObligationFactory factory) {
44             this.obligationSet = new ObligationSet(/* maxObligationTypes, */factory);
45             this.path = new Path();
46         }
47     
48         /**
49          * @return Returns the obligationSet.
50          */
51         public ObligationSet getObligationSet() {
52             return obligationSet;
53         }
54     
55         /**
56          * @return Returns the path.
57          */
58         public Path getPath() {
59             return path;
60         }
61     
62         public State duplicate() {
63             State dup = new State();
64             dup.obligationSet = this.obligationSet.duplicate();
65             dup.path = this.path.duplicate();
66     
67             return dup;
68         }
69     
70         @Override
71         public boolean equals(Object o) {
72             if (o == null || o.getClass() != this.getClass()) {
73                 return false;
74             }
75             State other = (State) o;
76             return this.obligationSet.equals(other.obligationSet) || this.path.equals(other.path);
77         }
78     
79         @Override
80         public int hashCode() {
81             return obligationSet.hashCode() + (1009 * path.hashCode());
82         }
83     
84         @Override
85         public String toString() {
86             StringBuilder buf = new StringBuilder();
87             buf.append("[");
88             buf.append(obligationSet.toString());
89             buf.append(",");
90             buf.append(path.toString());
91             buf.append("]");
92             return buf.toString();
93         }
94     }
