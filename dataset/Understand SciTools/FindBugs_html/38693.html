21     
22     import java.util.ArrayList;
23     import java.util.Collection;
24     import java.util.Collections;
25     import java.util.HashSet;
26     import java.util.Set;
27     
28     import javax.annotation.Nonnull;
29     
30     import org.objectweb.asm.Opcodes;
31     
32     import edu.umd.cs.findbugs.classfile.ClassDescriptor;
33     import edu.umd.cs.findbugs.classfile.IClassConstants;
34     import edu.umd.cs.findbugs.classfile.ICodeBaseEntry;
35     
36     /**
37      * Represents the class name, superclass name, and interface list of a class.
38      *
39      * @author David Hovemeyer
40      */
41     public class ClassNameAndSuperclassInfo extends ClassDescriptor {
42         private final ClassDescriptor superclassDescriptor;
43     
44         private final ClassDescriptor[] interfaceDescriptorList;
45     
46         private final ICodeBaseEntry codeBaseEntry;
47     
48         private final int accessFlags;
49     
50         private final Set<ClassDescriptor> calledClassDescriptors;
51     
52         private final int majorVersion, minorVersion;
53     
54         public static class Builder {
55             ClassDescriptor classDescriptor;
56     
57             ClassDescriptor superclassDescriptor;
58     
59             ClassDescriptor[] interfaceDescriptorList;
60     
61             ICodeBaseEntry codeBaseEntry;
62     
63             int accessFlags;
64     
65             int majorVersion, minorVersion;
66     
67             Collection<ClassDescriptor> referencedClassDescriptorList;
68     
69             Set<ClassDescriptor> calledClassDescriptors = Collections.<ClassDescriptor> emptySet();
70     
71             public ClassNameAndSuperclassInfo build() {
72                 return new ClassNameAndSuperclassInfo(classDescriptor, superclassDescriptor, interfaceDescriptorList, codeBaseEntry,
73                         accessFlags, referencedClassDescriptorList, calledClassDescriptors, majorVersion, minorVersion);
74             }
75     
76             /**
77              * @param accessFlags
78              *            The accessFlags to set.
79              */
80             public void setAccessFlags(int accessFlags) {
81                 this.accessFlags = accessFlags;
82             }
83     
84             /**
85              * @param classDescriptor
86              *            The classDescriptor to set.
87              */
88             public void setClassDescriptor(ClassDescriptor classDescriptor) {
89                 this.classDescriptor = classDescriptor;
90             }
91     
92             /**
93              * @param codeBaseEntry
94              *            The codeBaseEntry to set.
95              */
96             public void setCodeBaseEntry(ICodeBaseEntry codeBaseEntry) {
97                 this.codeBaseEntry = codeBaseEntry;
98             }
99     
100             /**
101              * @param interfaceDescriptorList
102              *            The interfaceDescriptorList to set.
103              */
104             public void setInterfaceDescriptorList(ClassDescriptor[] interfaceDescriptorList) {
105                 this.interfaceDescriptorList = interfaceDescriptorList;
106             }
107     
108             /**
109              * @param superclassDescriptor
110              *            The superclassDescriptor to set.
111              */
112             public void setSuperclassDescriptor(ClassDescriptor superclassDescriptor) {
113                 this.superclassDescriptor = superclassDescriptor;
114             }
115     
116             public void setClassfileVersion(int majorVersion, int minorVersion) {
117                 this.majorVersion = majorVersion;
118                 this.minorVersion = minorVersion;
119             }
120     
121             /**
122              * @param referencedClassDescriptorList
123              *            The referencedClassDescriptorList to set.
124              */
125             public void setReferencedClassDescriptors(Collection<ClassDescriptor> referencedClassDescriptorList) {
126                 if (referencedClassDescriptorList.size() == 0) {
127                     this.referencedClassDescriptorList = Collections.emptyList();
128                 } else {
129                     this.referencedClassDescriptorList = new ArrayList<ClassDescriptor>(referencedClassDescriptorList);
130                 }
131             }
132     
133             public void setCalledClassDescriptors(Collection<ClassDescriptor> calledClassDescriptorList) {
134                 if (calledClassDescriptorList.size() == 0) {
135                     this.calledClassDescriptors = Collections.emptySet();
136                 } else {
137                     this.calledClassDescriptors = new HashSet<ClassDescriptor>(calledClassDescriptorList);
138                 }
139             }
140         }
141     
142     
143         ClassNameAndSuperclassInfo(ClassDescriptor classDescriptor, ClassDescriptor superclassDescriptor,
144                 ClassDescriptor[] interfaceDescriptorList, ICodeBaseEntry codeBaseEntry, int accessFlags,
145                 /* TODO: We aren't doing anything with this */
146                 Collection<ClassDescriptor> referencedClassDescriptorList,
147                 @Nonnull Set<ClassDescriptor> calledClassDescriptors, int majorVersion, int minorVersion) {
148             super(classDescriptor.getClassName());
149             this.superclassDescriptor = superclassDescriptor;
150             this.interfaceDescriptorList = interfaceDescriptorList;
151             this.codeBaseEntry = codeBaseEntry;
152             this.accessFlags = accessFlags;
153             if (calledClassDescriptors == null) {
154                 throw new NullPointerException("calledClassDescriptors must not be null");
155             }
156             this.calledClassDescriptors = calledClassDescriptors;
157             this.majorVersion = majorVersion;
158             this.minorVersion = minorVersion;
159     
160         }
161     
162         /**
163          * @return Returns the accessFlags.
164          */
165         public int getAccessFlags() {
166             return accessFlags;
167         }
168     
169         /**
170          * @return Returns the majorVersion.
171          */
172         public int getMajorVersion() {
173             return majorVersion;
174         }
175     
176         /**
177          * @return Returns the minorVersion.
178          */
179         public int getMinorVersion() {
180             return minorVersion;
181         }
182     
183         /**
184          * @return Returns the classDescriptor.
185          */
186         public ClassDescriptor getClassDescriptor() {
187             return this;
188         }
189     
190         /**
191          * @return Returns the codeBaseEntry.
192          */
193         public ICodeBaseEntry getCodeBaseEntry() {
194             return codeBaseEntry;
195         }
196     
197         /**
198          * @return Returns the interfaceDescriptorList.
199          */
200         public ClassDescriptor[] getInterfaceDescriptorList() {
201             return interfaceDescriptorList;
202         }
203     
204         /**
205          * @return Returns the called class descriptors.
206          */
207         public Set<ClassDescriptor> getCalledClassDescriptors() {
208             return calledClassDescriptors;
209         }
210     
211         /**
212          * @return Returns the superclassDescriptor.
213          */
214         public ClassDescriptor getSuperclassDescriptor() {
215             return superclassDescriptor;
216         }
217     
218         private boolean isFlagSet(int flag) {
219             return (getAccessFlags() & flag) != 0;
220         }
221     
222         public boolean isFinal() {
223             return isFlagSet(IClassConstants.ACC_FINAL);
224         }
225     
226         public boolean isPrivate() {
227             return isFlagSet(IClassConstants.ACC_PRIVATE);
228         }
229     
230         public boolean isProtected() {
231             return isFlagSet(IClassConstants.ACC_PROTECTED);
232         }
233     
234         public boolean isPublic() {
235             return isFlagSet(IClassConstants.ACC_PUBLIC);
236         }
237     
238         public boolean isStatic() {
239             return isFlagSet(IClassConstants.ACC_STATIC);
240         }
241     
242         public boolean isInterface() {
243             return isFlagSet(IClassConstants.ACC_INTERFACE);
244         }
245     
246         public boolean isAbstract() {
247             return isFlagSet(IClassConstants.ACC_ABSTRACT);
248         }
249     
250         public boolean isAnnotation() {
251             return isFlagSet(IClassConstants.ACC_ANNOTATION);
252         }
253     
254         public boolean isSynthetic() {
255             return isFlagSet(IClassConstants.ACC_SYNTHETIC);
256         }
257     
258         public boolean isDeprecated() {
259             return isFlagSet(Opcodes.ACC_DEPRECATED);
260         }

262     }
