20     package edu.umd.cs.findbugs.util;

22     import javax.annotation.CheckForNull;
23     import javax.annotation.meta.When;
24     
25     import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
26     import edu.umd.cs.findbugs.classfile.DescriptorFactory;
27     import edu.umd.cs.findbugs.internalAnnotations.DottedClassName;

30     /**
31      * Utility methods for working with class names.
32      *
33      * @author David Hovemeyer
34      */
35     public abstract class ClassName {
36     
37         public static boolean isMathClass(@SlashedClassName String className) {
38             return "java/lang/Math".equals(className) || "java/lang/StrictMath".equals(className);
39         }
40     
41         public static @DottedClassName String assertIsDotted(@DottedClassName String className) {
42             assert className.indexOf('/') == -1 : "Not dotted: " + className;
43             return className;
44         }
45         public static @SlashedClassName String assertIsSlashed(@SlashedClassName String className) {
46             assert className.indexOf('.') == -1 : "Not slashed: " + className;
47             return className;
48         }
49         public static String toSignature(@SlashedClassName String className) {
50             if (className.length() == 0) {
51                 throw new IllegalArgumentException("classname can't be empty");
52             }
53             if (className.charAt(0) == '[' || className.endsWith(";")) {
54                 return className;
55             }
56             return "L" + className + ";";
57         }
58     
59         public static @CheckForNull String getPrimitiveType(@SlashedClassName String cls) {
60             if (!cls.startsWith("java/lang/")) {
61                 return null;
62             }
63             cls = cls.substring(10);
64             if ("Integer".equals(cls)) {
65                 return "I";
66             }
67             if ("Float".equals(cls)) {
68                 return "F";
69             }
70             if ("Double".equals(cls)) {
71                 return "D";
72             }
73             if ("Long".equals(cls)) {
74                 return "J";
75             }
76             if ("Byte".equals(cls)) {
77                 return "B";
78             }
79             if ("Character".equals(cls)) {
80                 return "C";
81             }
82             if ("Short".equals(cls)) {
83                 return "S";
84             }
85             if ("Boolean".equals(cls)) {
86                 return "Z";
87             }
88             return null;
89         }
90     
91         /**
92          * Converts from signature to slashed class name
93          * (e.g., from Ljava/lang/String; to java/lang/String).
94          * Returns null if it is the signature for an array or
95          * primitive type.
96          */
97         public static @CheckForNull
98         @SlashedClassName
99         String fromFieldSignature(String signature) {
100             if (signature.charAt(0) != 'L') {
101                 return null;
102             }
103             return signature.substring(1, signature.length() - 1);
104         }
105     
106         /**
107          * Convert class name to slashed format. If the class name is already in
108          * slashed format, it is returned unmodified.
109          *
110          * @param className
111          *            a class name
112          * @return the same class name in slashed format
113          */
114         @SlashedClassName
115         @SuppressFBWarnings("TQ_EXPLICIT_UNKNOWN_SOURCE_VALUE_REACHES_ALWAYS_SINK")
116         public static String toSlashedClassName(@SlashedClassName(when = When.UNKNOWN) String className) {
117             if (className.indexOf('.') >= 0) {
118                 return DescriptorFactory.canonicalizeString(className.replace('.', '/'));
119             }
120             return className;
121         }
122     
123         /**
124          * Convert class name to dotted format. If the class name is already in
125          * dotted format, it is returned unmodified.
126          *
127          * @param className
128          *            a class name
129          * @return the same class name in dotted format
130          */
131         @DottedClassName
132         @SuppressFBWarnings("TQ_EXPLICIT_UNKNOWN_SOURCE_VALUE_REACHES_NEVER_SINK")
133         public static String toDottedClassName(@SlashedClassName(when = When.UNKNOWN) String className) {
134             if (className.indexOf('/') >= 0) {
135                 return DescriptorFactory.canonicalizeString(className.replace('/', '.'));
136             }
137             return className;
138         }
139     
140         /**
141          * extract the package name from a dotted class name. Package names are
142          * always in dotted format.
143          *
144          * @param className
145          *            a dotted class name
146          * @return the name of the package containing the class
147          */
148         public static @DottedClassName
149         String extractPackageName(@DottedClassName String className) {
150             int i = className.lastIndexOf('.');
151             if (i < 0) {
152                 return "";
153             }
154             return className.substring(0, i);
155         }
156     
157         public static String extractSimpleName(@DottedClassName String className) {
158             int i = className.lastIndexOf('.');
159             if (i > 0) {
160                 className = className.substring(i + 1);
161             }
162             // to be consistent with the Class.getSimpleName(),
163             // simple class name does not! contain the enclosing class name
164             i = className.lastIndexOf('$');
165             if (i > 0) {
166                 className = className.substring(i + 1);
167             }
168             // can be empty!
169             return className;
170         }
171     
172         /**
173          * Return whether or not the given class name is valid.
174          *
175          * @param className
176          *            a possible class name
177          * @return true if it's a valid class name, false otherwise
178          */
179         public static boolean isValidClassName(String className) {
180             // FIXME: should use a regex
181     
182             if (className.indexOf('(') >= 0) {
183                 return false;
184             }
185             return true;
186         }
187     
188         /**
189          * Does a class name appear to designate an anonymous class? Only the name
190          * is analyzed. No classes are loaded or looked up.
191          *
192          * @param className
193          *            class name, slashed or dotted, fully qualified or unqualified
194          * @return true if className is the name of an anonymous class
195          */
196         public static boolean isAnonymous(String className) {
197             int i = className.lastIndexOf('$');
198             if (i >= 0 && i + 1 < className.length()) {
199                 return Character.isDigit(className.charAt(i + 1));
200             }
201             return false;
202         }
203     
204         /**
205          * Extract a slashed classname from a JVM classname or signature.
206          *
207          * @param originalName
208          *            JVM classname or signature
209          * @return a slashed classname
210          */
211         public static @SlashedClassName
212         String extractClassName(String originalName) {
213             String name = originalName;
214             if (name.charAt(0) != '[' && name.charAt(name.length() - 1) != ';') {
215                 return name;
216             }
217             while (name.charAt(0) == '[') {
218                 name = name.substring(1);
219             }
220             if (name.charAt(0) == 'L' && name.charAt(name.length() - 1) == ';') {
221                 name = name.substring(1, name.length() - 1);
222             }
223             if (name.charAt(0) == '[') {
224                 throw new IllegalArgumentException("Bad class name: " + originalName);
225             }
226             return name;
227         }
228     
229         public static String extractPackagePrefix(String packageName, int count) {
230             int dotsSeen = 0;
231             int prefixLength = 0;
232             while (dotsSeen < count) {
233                 int p = packageName.indexOf('.', prefixLength);
234                 if (p < 0) {
235                     return packageName;
236                 }
237                 prefixLength = p + 1;
238                 dotsSeen++;
239             }
240             if (prefixLength == 0) {
241                 return "";
242             }
243             return packageName.substring(0, prefixLength - 1);
244         }
245     
246         public static boolean matchedPrefixes(String[] classSearchStrings, @DottedClassName String className) {
247             String[] pp = classSearchStrings;
248             if (pp == null || pp.length == 0) {
249                 return true;
250             }
251     
252             for (String p : pp) {
253                 if (p.length() > 0 && className.indexOf(p) >= 0) {
254                     return true;
255                 }
256             }
257     
258             return false;
259     
260         }
261     
262     
263         public static @SlashedClassName String toSlashedClassName(Class<?> class1) {
264             return toSlashedClassName(class1.getName());
265         }
266     }
