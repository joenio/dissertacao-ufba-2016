60             if (storedException != null) {

97         private static boolean isPrimitiveComponentClass(String refSig) {
98             int c = 0;
99             while (c < refSig.length() && refSig.charAt(c) == '[') {
100                 c++;
101             }
102     
103             // If the string is now empty, then we evidently have
104             // an invalid type signature. We'll return "true",
105             // which in turn will cause isDeepSerializable() to return
106             // 1.0, hopefully avoiding any warnings from being generated
107             // by whatever detector is calling us.
108             return c >= refSig.length() || refSig.charAt(c) != 'L';
109         }

111         public static String getComponentClass(String refSig) {
112             while (refSig.charAt(0) == '[') {
113                 refSig = refSig.substring(1);
114             }
115     
116             // TODO: This method now returns primitive type signatures, is this ok?
117             if (refSig.charAt(0) == 'L') {
118                 return refSig.substring(1, refSig.length() - 1).replace('/', '.');
119             }
120             return refSig;
121         }

123         public static double isDeepSerializable(JavaClass x) throws ClassNotFoundException {
124             if (storedException != null) {
125                 throw storedException;
126             }
127     
128             double result = deepInstanceOf(x, serializable);
129             if (result >= 0.9) {
130                 return result;
131             }
132             result = Math.max(result, deepInstanceOf(x, collection));
133             if (result >= 0.9) {
134                 return result;
135             }
136             result = Math.max(result, deepInstanceOf(x, map));
137             if (result >= 0.9) {
138                 return result;
139             }
140             result = Math.max(result, 0.5 * deepInstanceOf(x, comparator));
141             if (result >= 0.9) {
142                 return result;
143             }
144             return result;
145         }

164         /**
165          * Given two JavaClasses, try to estimate the probability that an reference
166          * of type x is also an instance of type y. Will return 0 only if it is
167          * impossible and 1 only if it is guaranteed.
168          *
169          * @param x
170          *            Known type of object
171          * @param y
172          *            Type queried about
173          * @return 0 - 1 value indicating probability
174          */
175         public static double deepInstanceOf(JavaClass x, JavaClass y) throws ClassNotFoundException {
176     
177             if (x.equals(y)) {
178                 return 1.0;
179             }
180             if ("java.lang.Object".equals(y.getClassName())) {
181                 return 1.0;
182             }
183             Subtypes2 subtypes2 = AnalysisContext.currentAnalysisContext().getSubtypes2();
184             ClassDescriptor xDesc = DescriptorFactory.createClassDescriptor(x);
185             ClassDescriptor yDesc = DescriptorFactory.createClassDescriptor(y);
186     
187             boolean xIsSubtypeOfY = Repository.instanceOf(x, y);
188             if (xIsSubtypeOfY) {
189                 return 1.0;
190             }
191             boolean yIsSubtypeOfX = Repository.instanceOf(y, x);
192             if (!yIsSubtypeOfX) {
193                 if (x.isFinal() || y.isFinal()) {
194                     return 0.0;
195                 }
196                 if (!x.isInterface() && !y.isInterface()) {
197                     return 0.0;
198                 }
199             }
200     
201             Set<ClassDescriptor> transitiveCommonSubtypes = subtypes2.getTransitiveCommonSubtypes(xDesc, yDesc);
202     
203             if (transitiveCommonSubtypes.isEmpty()) {
204                 for (ClassDescriptor c : subtypes2.getSubtypes(xDesc)) {
205                     XClass cx;
206                     try {
207                         cx = Global.getAnalysisCache().getClassAnalysis(XClass.class, c);
208                     } catch (CheckedAnalysisException e) {
209                         continue;
210                     }
211                     if (!cx.isAbstract() && !cx.isInterface()) {
212                         if (x.isAbstract() || x.isInterface()) {
213                             return 0.2;
214                         }
215                         return 0.1;
216                     }
217                 }
218                 return 0.3;
219             }
220     
221             // exist classes that are both X and Y
222             Set<ClassDescriptor> xButNotY = new HashSet<ClassDescriptor>(subtypes2.getSubtypes(xDesc));
223             xButNotY.removeAll(transitiveCommonSubtypes);
224             for (ClassDescriptor c : xButNotY) {
225     
226                 try {
227                     XClass cx = Global.getAnalysisCache().getClassAnalysis(XClass.class, c);
228                     if (!cx.isAbstract() && !cx.isInterface()) {
229                         return 0.7;
230                     }
231                 } catch (CheckedAnalysisException e) {
232                     continue;
233                 }
234     
235             }
236             return 0.99;
237         }
