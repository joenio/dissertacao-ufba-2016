45         public static class PrefixFilter {
46             final String[] parts;
47     
48             PrefixFilter(String prefixes) {
49                 prefixes = prefixes.replace('/', '.').trim();
50                 if (prefixes.length() == 0) {
51                     parts = new String[0];
52                 } else {
53                     parts = prefixes.split("[ ,:]+");
54                 }
55             }
56     
57             boolean matches(@DottedClassName String className) {
58                 if (parts.length == 0) {
59                     return true;
60                 }
61                 for (String p : parts) {
62                     if (p.length() > 0 && className.startsWith(p)) {
63                         return true;
64                     }
65                 }
66                 return false;
67             }
68     
69             @Override
70             public String toString() {
71                 String result = Arrays.asList(parts).toString();
72                 return result.substring(1, result.length() - 1);
73             }
74     
75         }

91         public void countBug(BugInstance b) {
92             String packageName = b.getPrimaryClass().getPackageName();
93     
94             countPackageMember(packageName);
95         }
96     
97         /**
98          * @param packageName
99          */
100         public void countPackageMember(String packageName) {
101             totalCount++;
102             TreeSet<String> results = getProjects(packageName);
103             incrementCount(count, results);
104             incrementCount(rawPackageCount, packageName);
105             if (results.size() == 0) {
106                 incrementCount(missingProjectCount, packageName);
107             }
108         }
109     
110         public PrefixFilter getFilter(String projectName) {
111             return map.get(projectName);
112         }
113     
114         public TreeSet<String> getProjects(@DottedClassName String className) {
115             TreeSet<String> results = new TreeSet<String>();
116             for (Map.Entry<String, PrefixFilter> e : map.entrySet()) {
117                 if (e.getValue().matches(className)) {
118                     results.add(e.getKey());
119                 }
120             }
121             return results;
122         }
123     
124         static <T> void incrementCount(Map<T, Integer> counter, T t) {
125             incrementCount(counter, t, 1);
126         }
127     
128         static <T> void incrementCount(Map<T, Integer> counter, T t, int valueToAdd) {
129             Integer v = counter.get(t);
130             if (v == null) {
131                 counter.put(t, valueToAdd);
132             } else {
133                 counter.put(t, v + valueToAdd);
134             }
135         }

137         static final Pattern FORBIDDEN_PACKAGE_PREFIXES = Pattern.compile(SystemProperties.getProperty(
138                 "findbugs.forbiddenPackagePrefixes", " none ").replace(',', '|'));
139     
140         public void report() {
141             System.out.println("# of items counted: " + totalCount);
142             System.out.println("# of projects: " + size());
143             System.out.println("By package: ");
144     
145             for (Map.Entry<String, Integer> e : rawPackageCount.entrySet()) {
146                 String packageName = e.getKey();
147                 if (e.getValue() > 5) {
148                     System.out.printf("%5d %s%n", e.getValue(), packageName);
149                 }
150             }
151             System.out.println("Count by project");
152     
153             for (Map.Entry<Set<String>, Integer> e : count.entrySet()) {
154                 Set<String> projects = e.getKey();
155                 if (e.getValue() > 5) {
156                     System.out.printf("%5d %s%n", e.getValue(), projects);
157                 }
158             }
159             System.out.println("Count by package for items not associated with a project");
160     
161             Set<String> packages = missingProjectCount.keySet();
162             for (int count = 0; count < 3; count++) {
163                 HashSet<String> extraSuperPackages = new HashSet<String>();
164     
165                 for (String p1 : packages) {
166                     int num = missingProjectCount.get(p1);
167                     if (num < 3) {
168                         int x = p1.lastIndexOf('.');
169                         String p2 = p1.substring(0, x);
170                         if (FORBIDDEN_PACKAGE_PREFIXES.matcher(p2).matches()) {
171                             continue;
172                         }
173     
174                         extraSuperPackages.add(p2);
175                     }
176                 }
177                 for (String p1 : extraSuperPackages) {
178                     missingProjectCount.put(p1, 0);
179                 }
180     
181                 for (Iterator<String> i = packages.iterator(); i.hasNext();) {
182                     String p1 = i.next();
183                     int num = missingProjectCount.get(p1);
184     
185                     for (String p2 : packages) {
186                         if (p2.length() < p1.length() && p1.startsWith(p2)) {
187                             // p1 is a subpackage of p2
188                             // System.out.printf("%s is a subpackage of %s\n", p1,
189                             // p2);
190                             i.remove();
191                             incrementCount(missingProjectCount, p2, num);
192                             break;
193                         }
194                     }
195     
196                 }
197             }
198     
199             System.out.println("Count of missing files in packages not associated with a project");
200             for (Map.Entry<String, Integer> e : missingProjectCount.entrySet()) {
201                 if (e.getValue() > 5) {
202                     System.out.printf("%5d %s%n", e.getValue(), e.getKey());
203                 }
204             }
205         }

207         public ProjectPackagePrefixes() {
208             URL u = DetectorFactoryCollection.getCoreResource("projectPaths.properties");
209             if (u != null) {
210     
211                 BufferedReader in = null;
212                 try {
213                     in = UTF8.bufferedReader(u.openStream());
214                     while (true) {
215                         String s = in.readLine();
216                         if (s == null) {
217                             break;
218                         }
219                         String[] parts = s.split("=");
220                         if (parts.length == 2 && !map.containsKey(parts[0])) {
221                             map.put(parts[0], new PrefixFilter(parts[1]));
222                         }
223                     }
224                 } catch (IOException e1) {
225     
226                     AnalysisContext.logError("Error loading projects paths", e1);
227                 } finally {
228                     Util.closeSilently(in);
229                 }
230     
231             }
232     
233         }
