45     public class InitializationChain extends BytecodeScanningDetector {
46         Set<String> requires = new TreeSet<String>();
47     
48         Map<String, Set<String>> classRequires = new TreeMap<String, Set<String>>();
49     
50     
51     
52         private final BugReporter bugReporter;
53     
54         private final Map<XMethod, Set<XField>> staticFieldsRead = new HashMap<XMethod, Set<XField>>();
55         private final Set<XField> staticFieldsReadInAnyConstructor = new HashSet<XField>();
56         private Set<XField> fieldsReadInThisConstructor = new HashSet<XField>();
57     
58         private final Set<XMethod> constructorsInvokedInStaticInitializer = new HashSet<XMethod>();
59         private final List<InvocationInfo> invocationInfo = new ArrayList<InvocationInfo>();
60         private final Set<XField> warningGiven = new HashSet<XField>();
61     
62         private InvocationInfo lastInvocation;
63     
64         static class InvocationInfo {
65             public InvocationInfo(XMethod constructor, int pc) {
66                 this.constructor = constructor;
67                 this.pc = pc;
68             }
69             XMethod constructor;
70             int pc;
71             XField field;
72         }
73     
74         private static final boolean DEBUG = SystemProperties.getBoolean("ic.debug");
75     
76         public InitializationChain(BugReporter bugReporter) {
77             this.bugReporter = bugReporter;
78         }
79     
80         @Override
81         protected Iterable<Method> getMethodVisitOrder(JavaClass obj) {
82             ArrayList<Method> visitOrder = new ArrayList<Method>();
83             Method staticInitializer = null;
84             for(Method m : obj.getMethods()) {
85                 String name = m.getName();
86                 if ("<clinit>".equals(name)) {
87                     staticInitializer = m;
88                 } else if ("<init>".equals(name)) {
89                     visitOrder.add(m);
90                 }
91     
92             }
93             if (staticInitializer != null) {
94                 visitOrder.add(staticInitializer);
95             }
96             return visitOrder;
97         }
98     
99     
100         @Override
101         public void visit(Code obj) {
102             fieldsReadInThisConstructor  = new HashSet<XField>();
103             super.visit(obj);
104             staticFieldsRead.put(getXMethod(), fieldsReadInThisConstructor);
105             requires.remove(getDottedClassName());
106             if ("java.lang.System".equals(getDottedClassName())) {
107                 requires.add("java.io.FileInputStream");
108                 requires.add("java.io.FileOutputStream");
109                 requires.add("java.io.BufferedInputStream");
110                 requires.add("java.io.BufferedOutputStream");
111                 requires.add("java.io.PrintStream");
112             }
113             if (!requires.isEmpty()) {
114                 classRequires.put(getDottedClassName(), requires);
115                 requires = new TreeSet<String>();
116             }
117         }
118     
119         @Override
120         public void visitAfter(JavaClass obj) {
121     
122             staticFieldsRead.clear();
123     
124             staticFieldsReadInAnyConstructor.clear();
125             fieldsReadInThisConstructor.clear();
126     
127             constructorsInvokedInStaticInitializer.clear();
128             invocationInfo.clear();
129             lastInvocation = null;
130     
131         }
132     
133         @Override
134         public void sawOpcode(int seen) {
135             InvocationInfo prev = lastInvocation;
136             lastInvocation = null;
137             if ("<init>".equals(getMethodName())) {
138                 if (seen == GETSTATIC && getClassConstantOperand().equals(getClassName())) {
139                     staticFieldsReadInAnyConstructor.add(getXFieldOperand());
140                     fieldsReadInThisConstructor.add(getXFieldOperand());
141                 }
142                 return;
143             }
144     
145             if (seen == INVOKESPECIAL && "<init>".equals(getNameConstantOperand()) &&  getClassConstantOperand().equals(getClassName())) {
146     
147                 XMethod m = getXMethodOperand();
148                 Set<XField> read = staticFieldsRead.get(m);
149                 if (constructorsInvokedInStaticInitializer.add(m) && read != null && !read.isEmpty()) {
150                     lastInvocation = new InvocationInfo(m, getPC());
151                     invocationInfo.add(lastInvocation);
152     
153                 }
154     
155             }
156             if (seen == PUTSTATIC && getClassConstantOperand().equals(getClassName())) {
157                 XField f = getXFieldOperand();
158                 if (prev != null) {
159                     prev.field = f;
160                 }
161                 if (staticFieldsReadInAnyConstructor.contains(f) && !warningGiven.contains(f)) {
162                     for(InvocationInfo i : invocationInfo) {
163                         Set<XField> fields = staticFieldsRead.get(i.constructor);
164                         if (fields != null && fields.contains(f)) {
165                             warningGiven.add(f);
166                             BugInstance bug = new BugInstance(this, "SI_INSTANCE_BEFORE_FINALS_ASSIGNED", NORMAL_PRIORITY).addClassAndMethod(this);
167                             if (i.field != null) {
168                                 bug.addField(i.field).describe(FieldAnnotation.STORED_ROLE);
169                             }
170                             bug.addMethod(i.constructor).describe(MethodAnnotation.METHOD_CONSTRUCTOR);
171                             bug.addReferencedField(this).describe(FieldAnnotation.VALUE_OF_ROLE).addSourceLine(this, i.pc);
172                             bugReporter.reportBug(bug);
173                             break;
174     
175                         }
176                     }
177                 }
178     
179             } else if (seen == PUTSTATIC || seen == GETSTATIC || seen == INVOKESTATIC || seen == NEW) {
180                 if (getPC() + 6 < codeBytes.length) {
181                     requires.add(getDottedClassConstantOperand());
182                 }
183             }
184         }
185     
186         public void compute() {
187             Set<String> allClasses = classRequires.keySet();
188             Set<String> emptyClasses = new TreeSet<String>();
189             for (String c : allClasses) {
190                 Set<String> needs = classRequires.get(c);
191                 needs.retainAll(allClasses);
192                 Set<String> extra = new TreeSet<String>();
193                 for (String need : needs) {
194                     extra.addAll(classRequires.get(need));
195                 }
196                 needs.addAll(extra);
197                 needs.retainAll(allClasses);
198                 classRequires.put(c, needs);
199                 if (needs.isEmpty()) {
200                     emptyClasses.add(c);
201                 }
202             }
203             for (String c : emptyClasses) {
204                 classRequires.remove(c);
205             }
206         }
207     
208         @Override
209         public void report() {
210     
211             if (DEBUG) {
212                 System.out.println("Finishing computation");
213             }
214             compute();
215             compute();
216             compute();
217             compute();
218             compute();
219             compute();
220             compute();
221             compute();
222     
223             for (Entry<String, Set<String>> entry : classRequires.entrySet()) {
224                 String c = entry.getKey();
225                 if (DEBUG) {
226                     System.out.println("Class " + c + " requires:");
227                 }
228                 for (String needs : entry.getValue()) {
229                     if (DEBUG) {
230                         System.out.println("  " + needs);
231                     }
232                     Set<String> s = classRequires.get(needs);
233                     if (s != null && s.contains(c) && c.compareTo(needs) < 0) {
234                         bugReporter.reportBug(new BugInstance(this, "IC_INIT_CIRCULARITY", NORMAL_PRIORITY).addClass(c).addClass(
235                                 needs));
236                     }
237                 }
238             }
239         }
