168         private static boolean check(InstructionHandle h, int[] opcodes) {
169             for (int opcode : opcodes) {
170                 if (h == null) {
171                     return false;
172                 }
173                 short opcode2 = h.getInstruction().getOpcode();
174                 if (opcode == Constants.LDC) {
175                     switch (opcode2) {
176                     case Constants.LDC:
177                     case Constants.ALOAD:
178                     case Constants.ALOAD_0:
179                     case Constants.ALOAD_1:
180                     case Constants.ALOAD_2:
181                     case Constants.ALOAD_3:
182                         break;
183                     default:
184                         return false;
185                     }
186                 } else if (opcode2 != opcode) {
187                     return false;
188                 }
189                 h = h.getNext();
190             }
191             return true;
192         }

194         public static boolean isNullCheck(InstructionHandle h, ConstantPoolGen cpg) {
195             if (!(h.getInstruction() instanceof IFNONNULL)) {
196                 return false;
197             }
198             h = h.getNext();
199             final Instruction newInstruction = h.getInstruction();
200             if (!(newInstruction instanceof NEW)) {
201                 return false;
202             }
203             final ObjectType loadClassType = ((NEW) newInstruction).getLoadClassType(cpg);
204             if (!"java.lang.NullPointerException".equals(loadClassType.getClassName())) {
205                 return false;
206             }
207             h = h.getNext();
208             return check(h, NULLCHECK1) || check(h, NULLCHECK2);
209     
210         }

212         private void handleNullCheck(Location location, ValueNumberFrame vnaFrame, UnconditionalValueDerefSet fact)
213                 throws DataflowAnalysisException {
214             if (reportPotentialDereference(location, invDataflow.getFactAtLocation(location))) {
215                 ValueNumber vn = vnaFrame.getTopValue();
216                 fact.addDeref(vn, location);
217             }

220         public static boolean reportPotentialDereference(Location location, IsNullValueFrame invFrame)
221                 throws DataflowAnalysisException {
222             if (!invFrame.isValid()) {
223                 return false;
224             }
225             IsNullValue value = invFrame.getTopValue();
226             if (value.isDefinitelyNotNull()) {
227                 return false;
228             }
229             if (value.isDefinitelyNull()) {
230                 return false;
231             }
232             return true;
233         }

235         @Override
236         public void transferInstruction(InstructionHandle handle, BasicBlock basicBlock, UnconditionalValueDerefSet fact)
237                 throws DataflowAnalysisException {
238     
239             Instruction instruction = handle.getInstruction();
240             if (fact.isTop()) {
241                 return;
242             }
243             Location location = new Location(handle, basicBlock);
244     
245             // If this is a call to an assertion method,
246             // change the dataflow value to be TOP.
247             // We don't want to report future derefs that would
248             // be guaranteed only if the assertion methods
249             // returns normally.
250             // TODO: at some point, evaluate whether we should revisit this
251             if (isAssertion(handle) // || handle.getInstruction() instanceof ATHROW
252                     ) {
253                 if (DEBUG) {
254                     System.out.println("MAKING BOTTOM0 AT: " + location);
255                 }
256                 fact.clear();
257                 return;
258             }
259     
260             // Get value number frame
261             ValueNumberFrame vnaFrame = vnaDataflow.getFactAtLocation(location);
262             if (!vnaFrame.isValid()) {
263                 if (DEBUG) {
264                     System.out.println("MAKING TOP1 AT: " + location);
265                 }
266                 // Probably dead code.
267                 // Assume this location can't be reached.
268                 makeFactTop(fact);
269                 return;
270             }
271             if (isNullCheck(handle, methodGen.getConstantPool())) {
272                 handleNullCheck(location, vnaFrame, fact);
273             }
274     
275             // Check for calls to a method that unconditionally dereferences
276             // a parameter. Mark any such arguments as derefs.
277             if (CHECK_CALLS && instruction instanceof InvokeInstruction) {
278                 checkUnconditionalDerefDatabase(location, vnaFrame, fact);
279             }
280     
281             // If this is a method call instruction,
282             // check to see if any of the parameters are @NonNull,
283             // and treat them as dereferences.
284             if (CHECK_ANNOTATIONS && instruction instanceof InvokeInstruction) {
285                 checkNonNullParams(location, vnaFrame, fact);
286             }
287     
288             if (CHECK_ANNOTATIONS && instruction instanceof ARETURN) {
289                 XMethod thisMethod = XFactory.createXMethod(methodGen);
290                 checkNonNullReturnValue(thisMethod, location, vnaFrame, fact);
291             }
292     
293             if (CHECK_ANNOTATIONS && (instruction instanceof PUTFIELD || instruction instanceof PUTSTATIC)) {
294                 checkNonNullPutField(location, vnaFrame, fact);
295             }
296     
297             // Check to see if an instance value is dereferenced here
298             checkInstance(location, vnaFrame, fact);
299     
300             /*
301             if (false) {
302                 fact.cleanDerefSet(location, vnaFrame);
303             }*/
304     
305             if (DEBUG && fact.isTop()) {
306                 System.out.println("MAKING TOP2 At: " + location);
307             }
308     
309         }

311         /**
312          * Check method call at given location to see if it unconditionally
313          * dereferences a parameter. Mark any such arguments as derefs.
314          *
315          * @param location
316          *            the Location of the method call
317          * @param vnaFrame
318          *            ValueNumberFrame at the Location
319          * @param fact
320          *            the dataflow value to modify
321          * @throws DataflowAnalysisException
322          */
323         private void checkUnconditionalDerefDatabase(Location location, ValueNumberFrame vnaFrame, UnconditionalValueDerefSet fact)
324                 throws DataflowAnalysisException {
325             ConstantPoolGen constantPool = methodGen.getConstantPool();
326     
327             for (ValueNumber vn : checkUnconditionalDerefDatabase(location, vnaFrame, constantPool,
328                     invDataflow.getFactAtLocation(location), typeDataflow)) {
329                 fact.addDeref(vn, location);
330             }
331         }

333         public static Set<ValueNumber> checkUnconditionalDerefDatabase(Location location, ValueNumberFrame vnaFrame,
334                 ConstantPoolGen constantPool, @CheckForNull IsNullValueFrame invFrame, TypeDataflow typeDataflow)
335                         throws DataflowAnalysisException {
336             if (invFrame != null && !invFrame.isValid()) {
337                 return Collections.emptySet();
338             }
339     
340             InvokeInstruction inv = (InvokeInstruction) location.getHandle().getInstruction();
341     
342             SignatureParser sigParser = new SignatureParser(inv.getSignature(constantPool));
343             int numParams = sigParser.getNumParameters();
344             if (numParams == 0 || !sigParser.hasReferenceParameters()) {
345                 return Collections.emptySet();
346             }
347             ParameterNullnessPropertyDatabase database = AnalysisContext.currentAnalysisContext()
348                     .getUnconditionalDerefParamDatabase();
349             if (database == null) {
350                 if (DEBUG_CHECK_CALLS) {
351                     System.out.println("no database!");
352                 }
353                 return Collections.emptySet();
354             }
355     
356             TypeFrame typeFrame = typeDataflow.getFactAtLocation(location);
357             if (!typeFrame.isValid()) {
358                 if (DEBUG_CHECK_CALLS) {
359                     System.out.println("invalid type frame!");
360                 }
361                 return Collections.emptySet();
362             }
363     
364             try {
365                 Set<XMethod> targetSet = Hierarchy2.resolveMethodCallTargets(inv, typeFrame, constantPool);
366     
367                 if (targetSet.isEmpty()) {
368                     return Collections.emptySet();
369                 }
370     
371                 if (DEBUG_CHECK_CALLS) {
372                     System.out.println("target set size: " + targetSet.size());
373                 }
374                 // Compute the intersection of all properties
375                 ParameterProperty derefParamSet = null;
376                 for (XMethod target : targetSet) {
377                     if (target.isStub()) {
378                         continue;
379                     }
380                     if (DEBUG_CHECK_CALLS) {
381                         System.out.print("Checking: " + target + ": ");
382                     }
383     
384                     ParameterProperty targetDerefParamSet = database.getProperty(target.getMethodDescriptor());
385                     if (targetDerefParamSet == null) {
386                         // Hmm...no information for this target.
387                         // assume it doesn't dereference anything
388                         if (DEBUG_CHECK_CALLS) {
389                             System.out.println("==> no information, assume no guaranteed dereferences");
390                         }
391                         return Collections.emptySet();
392                     }
393     
394                     if (DEBUG_CHECK_CALLS) {
395                         System.out.println("==> " + targetDerefParamSet);
396                     }
397                     if (derefParamSet == null) {
398                         derefParamSet = new ParameterProperty();
399                         derefParamSet.copyFrom(targetDerefParamSet);
400                     } else {
401                         derefParamSet.intersectWith(targetDerefParamSet);
402                     }
403                 }
404     
405                 if (derefParamSet == null || derefParamSet.isEmpty()) {
406                     if (DEBUG) {
407                         System.out.println("** Nothing");
408                     }
409                     return Collections.emptySet();
410                 }
411                 if (DEBUG_CHECK_CALLS) {
412                     System.out.println("** Summary of call @ " + location.getHandle().getPosition() + ": " + derefParamSet);
413                 }
414     
415                 HashSet<ValueNumber> requiredToBeNonnull = new HashSet<ValueNumber>();
416                 for (int i = 0; i < numParams; i++) {
417                     if (!derefParamSet.hasProperty(i)) {
418                         continue;
419                     }
420                     int argSlot = vnaFrame.getStackLocation(sigParser.getSlotsFromTopOfStackForParameter(i));
421                     if (invFrame != null && !reportDereference(invFrame, argSlot)) {
422                         continue;
423                     }
424                     if (DEBUG_CHECK_CALLS) {
425                         System.out.println("  dereference @ " + location.getHandle().getPosition() + " of parameter " + i);
426                     }
427     
428                     requiredToBeNonnull.add(vnaFrame.getValue(argSlot));
429                 }
430                 return requiredToBeNonnull;
431     
432             } catch (ClassNotFoundException e) {
433                 AnalysisContext.reportMissingClass(e);
434             }
435             return Collections.emptySet();
436         }

467          * If this is a putfield or putstatic instruction, check to see if the field

498                     fact.addDeref(vn, location);
499                 }

525         public static Set<ValueNumber> checkAllNonNullParams(Location location, ValueNumberFrame vnaFrame,
526                 ConstantPoolGen constantPool, @CheckForNull Method method, @CheckForNull IsNullValueDataflow invDataflow,
527                 TypeDataflow typeDataflow) throws DataflowAnalysisException {
528             IsNullValueFrame invFrame = null;
529             if (invDataflow != null) {
530                 invFrame = invDataflow.getFactAtLocation(location);
531             }
532             Set<ValueNumber> result1 = checkNonNullParams(location, vnaFrame, constantPool, method, invFrame);
533             Set<ValueNumber> result2 = checkUnconditionalDerefDatabase(location, vnaFrame, constantPool, invFrame, typeDataflow);
534             if (result1.isEmpty()) {
535                 return result2;
536             }
537             if (result2.isEmpty()) {
538                 return result1;
539             }
540             result1.addAll(result2);
541             return result1;
542         }

544         public static Set<ValueNumber> checkNonNullParams(Location location, ValueNumberFrame vnaFrame, ConstantPoolGen constantPool,
545                 @CheckForNull Method method, @CheckForNull IsNullValueFrame invFrame) throws DataflowAnalysisException {
546     
547             if (invFrame != null && !invFrame.isValid()) {
548                 return Collections.emptySet();
549             }
550             INullnessAnnotationDatabase database = AnalysisContext.currentAnalysisContext().getNullnessAnnotationDatabase();
551     
552             InvokeInstruction inv = (InvokeInstruction) location.getHandle().getInstruction();
553             XMethod called = XFactory.createXMethod(inv, constantPool);
554             SignatureParser sigParser = new SignatureParser(called.getSignature());
555             int numParams = sigParser.getNumParameters();
556     
557             Set<ValueNumber> result = new HashSet<ValueNumber>();
558             Iterator<String> parameterIterator = sigParser.parameterSignatureIterator();
559             for (int i = 0; i < numParams; i++) {
560                 String parameterSignature = parameterIterator.next();
561                 char firstChar = parameterSignature.charAt(0);
562                 if (firstChar != 'L' && firstChar != '[') {
563                     continue;
564                 }
565                 int offset = sigParser.getSlotsFromTopOfStackForParameter(i);
566                 if (invFrame != null) {
567                     int slot = invFrame.getStackLocation(offset);
568                     if (!reportDereference(invFrame, slot)) {
569                         continue;
570                     }
571                 }
572                 if (database.parameterMustBeNonNull(called, i)) {
573                     int catchSizeNPE = Util.getSizeOfSurroundingTryBlock(method, "java/lang/NullPointerException", location
574                             .getHandle().getPosition());
575                     int catchSizeNFE = Util.getSizeOfSurroundingTryBlock(method, "java/lang/NumberFormatException", location
576                             .getHandle().getPosition());
577                     if (catchSizeNPE == Integer.MAX_VALUE
578                             && (!"java.lang.Integer".equals(called.getClassName()) || catchSizeNFE == Integer.MAX_VALUE)) {
579                         // Get the corresponding value number
580                         ValueNumber vn = vnaFrame.getArgument(inv, constantPool, i, sigParser);
581                         result.add(vn);
582                     }
583                 }
584             }
585             return result;
586         }

588         /**
589          * Check to see if the instruction has a null check associated with it, and
590          * if so, add a dereference.
591          *
592          * @param location
593          *            the Location of the instruction
594          * @param vnaFrame
595          *            ValueNumberFrame at the Location of the instruction
596          * @param fact
597          *            the dataflow value to modify
598          * @throws DataflowAnalysisException
599          */
600         private void checkInstance(Location location, ValueNumberFrame vnaFrame, UnconditionalValueDerefSet fact)
601                 throws DataflowAnalysisException {
602             // See if this instruction has a null check.
603             // If it does, the fall through predecessor will be
604             // identify itself as the null check.
605             if (!location.isFirstInstructionInBasicBlock()) {
606                 return;
607             }
608             if (invDataflow == null) {
609                 return;
610             }
611             BasicBlock fallThroughPredecessor = cfg.getPredecessorWithEdgeType(location.getBasicBlock(), EdgeTypes.FALL_THROUGH_EDGE);
612             if (fallThroughPredecessor == null || !fallThroughPredecessor.isNullCheck()) {
613                 return;
614             }
615     
616             // Get the null-checked value
617             ValueNumber vn = vnaFrame.getInstance(location.getHandle().getInstruction(), methodGen.getConstantPool());
618     
619             // Ignore dereferences of this
620             if (!methodGen.isStatic()) {
621                 ValueNumber v = vnaFrame.getValue(0);
622                 if (v.equals(vn)) {
623                     return;
624                 }
625             }
626             if (vn.hasFlag(ValueNumber.CONSTANT_CLASS_OBJECT)) {
627                 return;
628             }
629     
630             IsNullValueFrame startFact = null;
631     
632             startFact = invDataflow.getStartFact(fallThroughPredecessor);
633     
634             if (!startFact.isValid()) {
635                 return;
636             }
637     
638             int slot = startFact.getInstanceSlot(location.getHandle().getInstruction(), methodGen.getConstantPool());
639             if (!reportDereference(startFact, slot)) {
640                 return;
641             }
642             if (DEBUG) {
643                 System.out.println("FOUND GUARANTEED DEREFERENCE");
644                 System.out.println("Load: " + vnaFrame.getLoad(vn));
645                 System.out.println("Pred: " + fallThroughPredecessor);
646                 System.out.println("startFact: " + startFact);
647                 System.out.println("Location: " + location);
648                 System.out.println("Value number frame: " + vnaFrame);
649                 System.out.println("Dereferenced valueNumber: " + vn);
650                 System.out.println("invDataflow: " + startFact);
651                 System.out.println("IGNORE_DEREF_OF_NCP: " + IGNORE_DEREF_OF_NCP);
652             }
653             // Mark the value number as being dereferenced at this location
654             fact.addDeref(vn, location);
655         }

657         private static boolean reportDereference(IsNullValueFrame invFrameAtNullCheck, int instance) {
658             return reportDereference(invFrameAtNullCheck.getValue(instance));
659         }
660     
661         private static boolean reportDereference(IsNullValue value) {
662             if (value.isDefinitelyNotNull()) {
663                 return false;
664             }
665             if (value.isDefinitelyNull()) {
666                 return false;
667             }
668             if (IGNORE_DEREF_OF_NCP && value.isNullOnComplicatedPath()) {
669                 return false;
670             }
671             return true;
672         }

725         public void meetInto(UnconditionalValueDerefSet fact, Edge edge, UnconditionalValueDerefSet result, boolean onlyEdge) {
726             if (isExceptionEdge(edge) && !onlyEdge) {
727                 if (DEBUG) {
728                     System.out.println("Skipping exception edge");
729                 }
730                 return;
731             }
732     
733             ValueNumber knownNonnullOnBranch = null;
734             // Edge transfer function
735             if (isFactValid(fact)) {
736                 fact = propagateDerefSetsToMergeInputValues(fact, edge);
737                 if (invDataflow != null) {
738                     knownNonnullOnBranch = findValueKnownNonnullOnBranch(fact, edge);
739                     if (knownNonnullOnBranch != null) {
740                         fact = duplicateFact(fact);
741                         fact.clearDerefSet(knownNonnullOnBranch);
742                     }
743                 }
744             }
745             boolean isBackEdge = edge.isBackwardInBytecode();
746             Set<Integer> loopExitBranches = ClassContext.getLoopExitBranches(method, methodGen);
747             assert loopExitBranches != null;
748             boolean sourceIsTopOfLoop = edge.sourceIsTopOfLoop(loopExitBranches);
749             if (sourceIsTopOfLoop && edge.getType() == EdgeTypes.FALL_THROUGH_EDGE) {
750                 isBackEdge = true;
751             }
752             /*
753             if (false && (edge.getType() == EdgeTypes.IFCMP_EDGE || sourceIsTopOfLoop)) {
754                 System.out.println("Meet into " + edge);
755                 System.out.println("  foo2: " + sourceIsTopOfLoop);
756                 System.out.println("  getType: " + edge.getType());
757                 System.out.println("  Backedge according to bytecode: " + isBackEdge);
758                 System.out.println("  Fact hashCode: " + System.identityHashCode(result));
759                 System.out.println("  Initial fact: " + result);
760                 System.out.println("  Edge fact: " + fact);
761             }
762              */
763             if (result.isTop() || fact.isBottom()) {
764                 // Make result identical to other fact
765                 copy(fact, result);
766                 if (ASSUME_NONZERO_TRIP_LOOPS && isBackEdge && !fact.isTop()) {
767                     result.resultsFromBackEdge = true;
768                 }
769             } else if (ASSUME_NONZERO_TRIP_LOOPS && isBackEdge && !fact.isTop()) {
770                 result.unionWith(fact, vnaDataflow.getAnalysis().getFactory());
771                 result.resultsFromBackEdge = true;
772                 if (DEBUG) {
773                     System.out.println("\n Forcing union of " + System.identityHashCode(result) + " due to backedge info");
774                     System.out.println("  result: " + result);
775                 }
776     
777             } else if (result.isBottom() || fact.isTop()) {
778                 // No change in result fact
779             } else {
780                 // Dataflow merge
781                 // (intersection of unconditional deref values)
782                 if (ASSUME_NONZERO_TRIP_LOOPS && result.resultsFromBackEdge) {
783                     result.backEdgeUpdateCount++;
784                     if (result.backEdgeUpdateCount < 10) {
785                         if (DEBUG) {
786                             System.out.println("\n Union update of " + System.identityHashCode(result) + " due to backedge info");
787                         }
788                         result.unionWith(fact, vnaDataflow.getAnalysis().getFactory());
789                         return;
790                     }
791                 }
792                 result.mergeWith(fact, knownNonnullOnBranch, vnaDataflow.getAnalysis().getFactory());
793                 if (DEBUG) {
794                     System.out.println("  updated: " + System.identityHashCode(result));
795                     System.out.println("  result: " + result);
796     
797                 }
798             }
799             if (DEBUG && isBackEdge && edge.getType() == EdgeTypes.IFCMP_EDGE) {
800                 System.out.println("  result: " + result);
801             }
802         }

804         /**
805          * Find out if any VNs in the source block contribute to unconditionally
806          * dereferenced VNs in the target block. If so, the VN in the source block
807          * is also unconditionally dereferenced, and we must propagate the target
808          * VN's dereferences.
809          *
810          * @param fact
811          *            a dataflow value
812          * @param edge
813          *            edge to check for merge input values
814          * @return possibly-modified dataflow value
815          */
816         private UnconditionalValueDerefSet propagateDerefSetsToMergeInputValues(UnconditionalValueDerefSet fact, Edge edge) {
817     
818             ValueNumberFrame blockValueNumberFrame = vnaDataflow.getResultFact(edge.getSource());
819             ValueNumberFrame targetValueNumberFrame = vnaDataflow.getStartFact(edge.getTarget());
820     
821             UnconditionalValueDerefSet originalFact = fact;
822             fact = duplicateFact(fact);
823     
824             if (blockValueNumberFrame.isValid() && targetValueNumberFrame.isValid()) {
825                 int slots = 0;
826                 if (targetValueNumberFrame.getNumSlots() == blockValueNumberFrame.getNumSlots()) {
827                     slots = targetValueNumberFrame.getNumSlots();
828                 } else if (targetValueNumberFrame.getNumLocals() == blockValueNumberFrame.getNumLocals()) {
829                     slots = targetValueNumberFrame.getNumLocals();
830                 }
831     
832                 if (slots > 0) {
833                     if (DEBUG) {
834                         System.out.println("** Valid VNA frames for " + edge);
835                         System.out.println("** Block : " + blockValueNumberFrame);
836                         System.out.println("** Target: " + targetValueNumberFrame);
837                     }
838     
839                     for (int i = 0; i < slots; i++) {
840                         ValueNumber blockVN = blockValueNumberFrame.getValue(i);
841                         ValueNumber targetVN = targetValueNumberFrame.getValue(i);
842                         if (blockVN.equals(targetVN)) {
843                             continue;
844                         }
845                         fact.clearDerefSet(blockVN);
846                         if (originalFact.isUnconditionallyDereferenced(targetVN)) {
847                             fact.setDerefSet(blockVN, originalFact.getUnconditionalDerefLocationSet(targetVN));
848                         }
849     
850                     } // for all slots
851     
852                     for (ValueNumber blockVN : blockValueNumberFrame.valueNumbersForLoads()) {
853                         AvailableLoad load = blockValueNumberFrame.getLoad(blockVN);
854                         if (load == null) {
855                             continue;
856                         }
857                         ValueNumber[] targetVNs = targetValueNumberFrame.getAvailableLoad(load);
858                         if (targetVNs != null) {
859                             for (ValueNumber targetVN : targetVNs) {
860                                 if (targetVN.hasFlag(ValueNumber.PHI_NODE) && fact.isUnconditionallyDereferenced(targetVN)
861                                         && !fact.isUnconditionallyDereferenced(blockVN)) {
862                                     // Block VN is also dereferenced
863                                     // unconditionally.
864                                     AvailableLoad targetLoad = targetValueNumberFrame.getLoad(targetVN);
865                                     if (!load.equals(targetLoad)) {
866                                         continue;
867                                     }
868                                     if (DEBUG) {
869                                         System.out.println("** Copy vn derefs for " + load + " from " + targetVN + " --> " + blockVN);
870                                         System.out.println("** block phi for " + System.identityHashCode(blockValueNumberFrame)
871                                                 + " is " + blockValueNumberFrame.phiNodeForLoads);
872                                         System.out.println("** target phi for " + System.identityHashCode(targetValueNumberFrame)
873                                                 + " is " + targetValueNumberFrame.phiNodeForLoads);
874                                     }
875                                     fact.setDerefSet(blockVN, fact.getUnconditionalDerefLocationSet(targetVN));
876     
877                                 }
878                             }
879                         }
880                     }
881     
882                 }
883             }
884             if (DEBUG) {
885                 System.out.println("Target VNF: " + targetValueNumberFrame);
886                 System.out.println("Block VNF: " + blockValueNumberFrame);
887                 System.out.println("fact: " + fact);
888             }
889             fact.cleanDerefSet(null, blockValueNumberFrame);
890             return fact;
891         }

907         /**
908          * Clear deref sets of values if this edge is the non-null branch of an if
909          * comparison.
910          *
911          * @param fact
912          *            a datflow fact
913          * @param edge
914          *            edge to check
915          * @return possibly-modified dataflow fact
916          */
917         private @CheckForNull
918         ValueNumber findValueKnownNonnullOnBranch(UnconditionalValueDerefSet fact, Edge edge) {
919     
920             IsNullValueFrame invFrame = invDataflow.getResultFact(edge.getSource());
921             if (!invFrame.isValid()) {
922                 return null;
923             }
924             IsNullConditionDecision decision = invFrame.getDecision();
925             if (decision == null) {
926                 return null;
927             }
928     
929             IsNullValue inv = decision.getDecision(edge.getType());
930             if (inv == null || !inv.isDefinitelyNotNull()) {
931                 return null;
932             }
933             ValueNumber value = decision.getValue();
934             if (DEBUG) {
935                 System.out.println("Value number " + value + " is known nonnull on " + edge);
936             }
937     
938             return value;
939         }

941         /**
942          * Determine whether dataflow should be propagated on given edge.
943          *
944          * @param edge
945          *            the edge
946          * @return true if dataflow should be propagated on the edge, false
947          *         otherwise
948          */
949         private boolean isExceptionEdge(Edge edge) {
950             boolean isExceptionEdge = edge.isExceptionEdge();
951             if (isExceptionEdge) {
952                 if (DEBUG) {
953                     System.out.println("NOT Ignoring " + edge);
954                 }
955                 return true; // false
956             }
957             if (edge.getType() != EdgeTypes.FALL_THROUGH_EDGE) {
958                 return false;
959             }
960             InstructionHandle h = edge.getSource().getLastInstruction();
961             if (h != null && h.getInstruction() instanceof IFNONNULL && isNullCheck(h, methodGen.getConstantPool())) {
962                 return true;
963             }
964     
965             return false;
966     
967         }
