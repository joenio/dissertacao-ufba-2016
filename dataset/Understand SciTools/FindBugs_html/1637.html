58         @Override
59         public void visitJavaClass(JavaClass obj) {
60             try {
61                 obj.getSuperClass();
62             } catch (ClassNotFoundException e) {
63                 AnalysisContext.reportMissingClass(e);
64                 return;
65             }
66     
67             String superclassName2 = getSuperclassName();
68             boolean weird = "java.lang.Object".equals(superclassName2) && obj.getInterfaceIndices().length == 0;
69             boolean hasAnonymousName = ClassName.isAnonymous(obj.getClassName());
70             boolean isAnonymousInnerClass = hasAnonymousName && !weird;
71             if (isAnonymousInnerClass) {
72                 super.visitJavaClass(obj);
73             }
74         }

76         boolean definedInThisClassOrSuper(JavaClass clazz, String method) throws ClassNotFoundException {
77             if (clazz == null) {
78                 return false;
79             }
80             // System.out.println("Checking to see if " + method + " is defined in "
81             // + clazz.getClassName());
82             for (Method m : clazz.getMethods()) {
83                 String key = m.getName() + ":" + m.getSignature();
84                 if (!m.isStatic() && method.equals(key)) {
85                     return true;
86                 }
87             }
88     
89             return definedInSuperClassOrInterface(clazz, method);
90     
91         }

93         @Override
94         public void sawOpcode(int seen) {
95             if (seen == INVOKESPECIAL) {
96                 XMethod m = getXMethodOperand();
97                 if (m == null) {
98                     return;
99                 }
100                 XClass c = getXClass();
101                 int nameDistance = EditDistance.editDistance(m.getName(), getMethodName());
102                 if (nameDistance < 4 && c.findMatchingMethod(m.getMethodDescriptor()) == null && !m.isFinal()) {
103                     potentialSuperCall = m;
104                 }
105             }
106         }

108         boolean definedInSuperClassOrInterface(JavaClass clazz, String method) throws ClassNotFoundException {
109             if (clazz == null) {
110                 return false;
111             }
112             JavaClass superClass = clazz.getSuperClass();
113             if (superClass == null) {
114                 return false;
115             }
116             try {
117                 XClass xClass = Global.getAnalysisCache().getClassAnalysis(XClass.class,
118                         DescriptorFactory.createClassDescriptorFromDottedClassName(superClass.getClassName()));
119                 if (xClass.hasStubs()) {
120                     return true;
121                 }
122             } catch (CheckedAnalysisException e) {
123                 return true;
124             }
125     
126             if (definedInThisClassOrSuper(superClass, method)) {
127                 return true;
128             }
129             for (JavaClass i : clazz.getInterfaces()) {
130                 if (definedInThisClassOrSuper(i, method)) {
131                     return true;
132                 }
133             }
134             return false;
135         }

137         Set<String> definedInClass(JavaClass clazz) {
138             HashSet<String> result = new HashSet<String>();
139             for (Method m : clazz.getMethods()) {
140                 if (!skip(m)) {
141                     result.add(m.getName() + m.getSignature());
142                 }
143             }
144             return result;
145         }
146     
147         private boolean skip(Method obj) {
148             if (BCELUtil.isSynthetic(obj)) {
149                 return true;
150             }
151             if (obj.isPrivate()) {
152                 return true;
153             }
154             if (obj.isAbstract()) {
155                 return true;
156             }
157     
158             String methodName = obj.getName();
159             String sig = obj.getSignature();
160             if ("<init>".equals(methodName)) {
161                 return true;
162             }
163             if ("<clinit>".equals(methodName)) {
164                 return true;
165             }
166             if ("()Ljava/lang/Object;".equals(sig) && ("readResolve".equals(methodName) || "writeReplace".equals(methodName))) {
167                 return true;
168             }
169             if (methodName.startsWith("access$")) {
170                 return true;
171             }
172             if (methodName.length() < 2 || methodName.indexOf('$') >= 0) {
173                 return true;
174             }
175             XMethod m = getXMethod();
176             for (ClassDescriptor c : m.getAnnotationDescriptors()) {
177                 if (c.getClassName().indexOf("inject") >= 0) {
178                     return true;
179                 }
180             }
181             return false;
182         }

184         BugInstance pendingBug;
185     
186         @Override
187         public void doVisitMethod(Method obj) {
188             super.doVisitMethod(obj);
189             if (pendingBug != null) {
190                 if (potentialSuperCall == null) {
191                     String role = ClassAnnotation.SUPERCLASS_ROLE;
192     
193                     @DottedClassName String superclassName =  ClassName.toDottedClassName(getSuperclassName());
194                     if ("java.lang.Object".equals(superclassName)) {
195     
196                         try {
197                             JavaClass interfaces[] = getThisClass().getInterfaces();
198                             if (interfaces.length == 1) {
199                                 superclassName = interfaces[0].getClassName();
200                                 role = ClassAnnotation.IMPLEMENTED_INTERFACE_ROLE;
201                             }
202                         } catch (ClassNotFoundException e) {
203                             AnalysisContext.reportMissingClass(e);
204                         }
205                     }
206                     pendingBug.addClass(superclassName).describe(role);
207                     try {
208                         XClass from = Global.getAnalysisCache().getClassAnalysis(XClass.class,
209                                 DescriptorFactory.createClassDescriptorFromDottedClassName(superclassName));
210                         XMethod  potentialMatch = null;
211                         for(XMethod m : from.getXMethods()) {
212                             if (!m.isStatic() && !m.isPrivate() && m.getName().toLowerCase().equals(obj.getName().toLowerCase())) {
213                                 if (potentialMatch == null) {
214                                     potentialMatch = m;
215                                 } else {
216                                     // multiple matches; ignore all
217                                     potentialMatch = null;
218                                     break;
219                                 }
220                             }
221                         }
222                         if (potentialMatch != null) {
223                             pendingBug.addMethod(potentialMatch)
224                             .describe(MethodAnnotation.METHOD_DID_YOU_MEAN_TO_OVERRIDE);
225                         }
226     
227                     } catch (CheckedAnalysisException e) {
228                         AnalysisContext.logError("Error: ", e);
229                     }
230                 } else {
231                     pendingBug.setPriority(pendingBug.getPriority() - 1);
232                     pendingBug.addMethod(potentialSuperCall).describe(MethodAnnotation.METHOD_DID_YOU_MEAN_TO_OVERRIDE);
233                 }
234                 bugReporter.reportBug(pendingBug);
235                 pendingBug = null;
236                 potentialSuperCall = null;
237             }
238     
239         }

241         @Override
242         public void visit(Code obj) {
243             if (pendingBug != null) {
244                 super.visit(obj);
245             }
246         }

248         @Override
249         public void visit(Method obj) {
250             try {
251     
252                 if (skip(obj)) {
253                     return;
254                 }
255     
256                 JavaClass clazz = getThisClass();
257                 XMethod xmethod = XFactory.createXMethod(clazz, obj);
258                 XFactory factory = AnalysisContext.currentXFactory();
259                 String key = obj.getName() + ":" + obj.getSignature();
260                 if (!factory.isCalled(xmethod) && (obj.isStatic() || !definedInSuperClassOrInterface(clazz, key))) {
261                     int priority = NORMAL_PRIORITY;
262                     JavaClass superClass = clazz.getSuperClass();
263                     String superClassName = superClass.getClassName();
264                     if ("java.lang.Object".equals(superClassName)) {
265                         priority = NORMAL_PRIORITY;
266     
267                     } else if (definedInClass(superClass).containsAll(definedInClass(clazz))) {
268                         priority = NORMAL_PRIORITY;
269                     } else {
270                         priority = HIGH_PRIORITY;
271                     }
272                     Code code = null;
273                     for (Attribute a : obj.getAttributes()) {
274                         if (a instanceof Code) {
275                             code = (Code) a;
276                             break;
277                         }
278                     }
279                     if (code != null && code.getLength() == 1) {
280                         priority++;
281                     }
282     
283                     pendingBug = new BugInstance(this, "UMAC_UNCALLABLE_METHOD_OF_ANONYMOUS_CLASS", priority).addClassAndMethod(this);
284                     potentialSuperCall = null;
285                 }
286     
287             } catch (ClassNotFoundException e) {
288                 bugReporter.reportMissingClass(e);
289             }
290     
291         }
