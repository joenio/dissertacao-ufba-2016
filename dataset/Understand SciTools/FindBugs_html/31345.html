24     import org.apache.bcel.classfile.Code;
25     import org.apache.bcel.classfile.JavaClass;

36     import edu.umd.cs.findbugs.classfile.CheckedAnalysisException;

38     import edu.umd.cs.findbugs.classfile.Global;

40     public class InvalidJUnitTest extends BytecodeScanningDetector {
41     
42         private static final int SEEN_NOTHING = 0;
43     
44         private static final int SEEN_ALOAD_0 = 1;
45     
46         private final BugReporter bugReporter;
47     
48         private int state;
49     
50         public InvalidJUnitTest(BugReporter bugReporter) {
51             this.bugReporter = bugReporter;
52         }
53     
54         boolean directChildOfTestCase;
55     
56         @Override
57         public void visitClassContext(ClassContext classContext) {
58             if (!enabled()) {
59                 return;
60             }
61     
62             JavaClass jClass = classContext.getJavaClass();
63             XClass xClass = classContext.getXClass();
64     
65             try {
66     
67                 if (!isJunit3TestCase(xClass)) {
68                     return;
69                 }
70                 if ((jClass.getAccessFlags() & ACC_ABSTRACT) == 0) {
71                     if (!hasTestMethods(jClass)) {
72                         bugReporter.reportBug(new BugInstance(this, "IJU_NO_TESTS", LOW_PRIORITY).addClass(jClass));
73                     }
74                 }
75                 directChildOfTestCase = "junit.framework.TestCase".equals(jClass.getSuperclassName());
76                 jClass.accept(this);
77             } catch (ClassNotFoundException cnfe) {
78                 bugReporter.reportMissingClass(cnfe);
79             }
80     
81         }
82     
83         private boolean isJunit3TestCase(XClass jClass) throws ClassNotFoundException {
84             ClassDescriptor sDesc = jClass.getSuperclassDescriptor();
85             if (sDesc == null) {
86                 return false;
87             }
88             String sName = sDesc.getClassName();
89             if (sName.equals("junit/framework/TestCase")) {
90                 return true;
91             }
92             if (sName.equals("java/lang/Object")) {
93                 return false;
94             }
95     
96             try {
97                 XClass sClass = Global.getAnalysisCache().getClassAnalysis(XClass.class, sDesc);
98                 if (sClass == null) {
99                     return false;
100                 }
101                 return isJunit3TestCase(sClass);
102             } catch (CheckedAnalysisException e) {
103                 return false;
104             }
105     
106         }
107     
108         private boolean hasTestMethods(JavaClass jClass) {
109             boolean foundTest = false;
110             Method[] methods = jClass.getMethods();
111             for (Method m : methods) {
112                 if (m.isPublic() && m.getName().startsWith("test") && m.getSignature().equals("()V")) {
113                     return true;
114                 }
115                 if (m.getName().startsWith("runTest") && m.getSignature().endsWith("()V")) {
116                     return true;
117                 }
118             }
119             if (hasSuite(methods)) {
120                 return true;
121             }
122     
123             try {
124                 JavaClass sClass = jClass.getSuperClass();
125                 if (sClass != null) {
126                     return hasTestMethods(sClass);
127                 }
128             } catch (ClassNotFoundException e) {
129                 AnalysisContext.reportMissingClass(e);
130             }
131     
132             return false;
133         }
134     
135         /** is there a JUnit3TestSuite */
136         private boolean hasSuite(Method[] methods) {
137             for (Method m : methods) {
138                 if (m.getName().equals("suite") && m.isPublic() && m.isStatic()
139                         // && m.getReturnType().equals(junit.framework.Test.class)
140                         // && m.getArgumentTypes().length == 0
141                         && m.getSignature().equals("()Ljunit/framework/Test;")) {
142                     return true;
143                 }
144             }
145             return false;
146         }
147     
148         /**
149          * Check whether or not this detector should be enabled. The detector is
150          * disabled if the TestCase class cannot be found (meaning we don't have
151          * junit.jar on the aux classpath).
152          *
153          * @return true if it should be enabled, false if not
154          */
155         private boolean enabled() {
156             return true;
157         }
158     
159         @Override
160         public void visit(Method obj) {
161             if (getMethodName().equals("suite") && !obj.isStatic()) {
162                 bugReporter.reportBug(new BugInstance(this, "IJU_SUITE_NOT_STATIC", NORMAL_PRIORITY).addClassAndMethod(this));
163             }
164     
165             if (getMethodName().equals("suite") && obj.getSignature().startsWith("()") && obj.isStatic()) {
166                 if ((!obj.getSignature().equals("()Ljunit/framework/Test;") && !obj.getSignature().equals(
167                         "()Ljunit/framework/TestSuite;"))
168                         || !obj.isPublic()) {
169                     bugReporter.reportBug(new BugInstance(this, "IJU_BAD_SUITE_METHOD", NORMAL_PRIORITY).addClassAndMethod(this));
170                 }
171     
172             }
173         }
174     
175         private boolean sawSuperCall;
176     
177         @Override
178         public void visit(Code obj) {
179             if (!directChildOfTestCase && (getMethodName().equals("setUp") || getMethodName().equals("tearDown"))
180                     && !getMethod().isPrivate() && getMethodSig().equals("()V")) {
181                 sawSuperCall = false;
182                 super.visit(obj);
183                 if (sawSuperCall) {
184                     return;
185                 }
186                 JavaClass we = Lookup.findSuperImplementor(getThisClass(), getMethodName(), "()V", bugReporter);
187                 if (we != null && !we.getClassName().equals("junit.framework.TestCase")) {
188                     // OK, got a bug
189                     int offset = 0;
190                     if (getMethodName().equals("tearDown")) {
191                         offset = obj.getCode().length - 1;
192                     }
193                     Method superMethod = Lookup.findImplementation(we, getMethodName(), "()V");
194                     Code superCode = superMethod.getCode();
195                     if (superCode != null && superCode.getCode().length > 3) {
196                         bugReporter.reportBug(new BugInstance(this, getMethodName().equals("setUp") ? "IJU_SETUP_NO_SUPER"
197                                 : "IJU_TEARDOWN_NO_SUPER", NORMAL_PRIORITY).addClassAndMethod(this).addMethod(we, superMethod)
198                                 .describe(MethodAnnotation.METHOD_OVERRIDDEN).addSourceLine(this, offset));
199                     }
200                 }
201             }
202         }
203     
204         @Override
205         public void sawOpcode(int seen) {
206             switch (state) {
207             case SEEN_NOTHING:
208                 if (seen == ALOAD_0) {
209                     state = SEEN_ALOAD_0;
210                 }
211                 break;
212     
213             case SEEN_ALOAD_0:
214                 if ((seen == INVOKESPECIAL) && (getNameConstantOperand().equals(getMethodName()))
215                         && (getSigConstantOperand().equals("()V"))) {
216                     sawSuperCall = true;
217                 }
218                 state = SEEN_NOTHING;
219                 break;
220             default:
221                 state = SEEN_NOTHING;
222             }
223         }
224     }
