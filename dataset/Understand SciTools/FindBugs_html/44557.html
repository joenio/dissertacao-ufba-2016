38     public class TypeQualifierValueSet {
39         // States
40         enum State {
41             VALID, TOP, BOTTOM
42         }

44         private final Map<ValueNumber, FlowValue> valueMap;
45     
46         private final Map<ValueNumber, Set<SourceSinkInfo>> whereAlways;
47     
48         private final Map<ValueNumber, Set<SourceSinkInfo>> whereNever;

50         private State state = State.VALID;

52         final boolean  isStrict;

54         public TypeQualifierValueSet(TypeQualifierValue<?> typeQualifierValue) {
55             this.valueMap = new HashMap<ValueNumber, FlowValue>(3);
56             this.whereAlways = new HashMap<ValueNumber, Set<SourceSinkInfo>>(3);
57             this.whereNever = new HashMap<ValueNumber, Set<SourceSinkInfo>>(3);
58             this.state = State.TOP;
59             isStrict = typeQualifierValue.isStrictQualifier();
60         }

62         public void modelSourceSink(SourceSinkInfo sourceSinkInfo) {
63             assert sourceSinkInfo != null;
64             ValueNumber vn = sourceSinkInfo.getValueNumber();
65             FlowValue flowValue = FlowValue.flowValueFromWhen(sourceSinkInfo.getWhen());
66     
67             setValue(vn, flowValue);
68             if (true) {
69                 switch (flowValue) {
70                 case ALWAYS:
71                     addSourceSinkInfo(whereAlways, vn, sourceSinkInfo);
72                     break;
73                 case NEVER:
74                     addSourceSinkInfo(whereNever, vn, sourceSinkInfo);
75                     break;
76                 default:
77                     break;
78                 }
79             }/* else {
80     
81                 if (flowValue.isYes()) {
82                     addSourceSinkInfo(whereAlways, vn, sourceSinkInfo);
83                 }
84     
85                 if (flowValue.isNo()) {
86                     addSourceSinkInfo(whereNever, vn, sourceSinkInfo);
87                 }
88             }*/
89         }

91         private void setValue(ValueNumber vn, FlowValue flowValue) {
92             if (flowValue == FlowValue.TOP) {
93                 pruneValue(vn);
94             } else {
95                 valueMap.put(vn, flowValue);
96             }
97         }

99         static <K, V> void copyMapValue(Map<K, V> map, K from, K to) {
100             if (!map.containsKey(from)) {
101                 return;
102             }
103             map.put(to, map.get(from));
104         }
105     
106         void copyInfo(ValueNumber from, ValueNumber to) {
107             if (state == State.TOP) {
108                 return;
109             }
110             copyMapValue(valueMap, from, to);
111             copyMapValue(whereAlways, from, to);
112             copyMapValue(whereNever, from, to);
113         }
114     
115         private static void addSourceSinkInfo(Map<ValueNumber, Set<SourceSinkInfo>> sourceSinkInfoSetMap, ValueNumber vn,
116                 SourceSinkInfo sourceSinkInfo) {
117             Set<SourceSinkInfo> sourceSinkInfoSet = sourceSinkInfoSetMap.get(vn);
118             if (sourceSinkInfoSet == null) {
119                 sourceSinkInfoSet = new HashSet<SourceSinkInfo>(3);
120                 sourceSinkInfoSetMap.put(vn, sourceSinkInfoSet);
121             }
122             sourceSinkInfoSet.add(sourceSinkInfo);
123         }
124     
125         public void pruneValue(ValueNumber vn) {
126             assert isValid();
127             valueMap.remove(vn);
128             whereAlways.remove(vn);
129             whereNever.remove(vn);
130         }

132         public Set<? extends SourceSinkInfo> getWhere(ValueNumber vn) {
133             FlowValue f = getValue(vn);
134             switch (f) {
135             case ALWAYS:  return getSourceSinkInfoSet(whereAlways, vn);
136             case NEVER:  return getSourceSinkInfoSet(whereNever, vn);
137             }
138             return Collections.emptySet();
139     
140         }
141         public Set<? extends SourceSinkInfo> getWhereAlways(ValueNumber vn) {
142             return getSourceSinkInfoSet(whereAlways, vn);
143         }
144     
145         public Set<? extends SourceSinkInfo> getWhereNever(ValueNumber vn) {
146             return getSourceSinkInfoSet(whereNever, vn);
147         }

149         private static Set<? extends SourceSinkInfo> getSourceSinkInfoSet(Map<ValueNumber, Set<SourceSinkInfo>> sourceSinkInfoSetMap,
150                 ValueNumber vn) {
151             Set<SourceSinkInfo> sourceSinkInfoSet = sourceSinkInfoSetMap.get(vn);
152             if (sourceSinkInfoSet == null || sourceSinkInfoSet.isEmpty()) {
153                 return Collections.emptySet();
154             }
155     
156             return sourceSinkInfoSet;
157         }
158         private static Set<SourceSinkInfo> getOrCreateSourceSinkInfoSet(Map<ValueNumber, Set<SourceSinkInfo>> sourceSinkInfoSetMap,
159                 ValueNumber vn) {
160             Set<SourceSinkInfo> sourceSinkInfoSet = sourceSinkInfoSetMap.get(vn);
161             if (sourceSinkInfoSet == null) {
162                 sourceSinkInfoSet = new HashSet<SourceSinkInfo>(3);
163                 sourceSinkInfoSetMap.put(vn, sourceSinkInfoSet);
164             }
165             return sourceSinkInfoSet;
166         }

171             FlowValue result = valueMap.get(vn);
172             return result != null ? result : FlowValue.TOP;
173         }

179         public boolean isValid() {
180             return state == State.VALID;
181         }
182     
183         public void makeValid() {
184             /*
185              * this.state = State.VALID; this.valueMap.clear();
186              * this.whereAlways.clear(); this.whereNever.clear();
187              */
188             reset(State.VALID);
189         }
190     
191         public void makeSameAs(TypeQualifierValueSet source) {
192             /*
193              * this.state = source.state; this.valueMap.clear();
194              */
195             reset(source.state);
196             this.valueMap.putAll(source.valueMap);
197             copySourceSinkInfoSetMap(this.whereAlways, source.whereAlways);
198             copySourceSinkInfoSetMap(this.whereNever, source.whereNever);
199         }
200     
201         private void copySourceSinkInfoSetMap(Map<ValueNumber, Set<SourceSinkInfo>> dest, Map<ValueNumber, Set<SourceSinkInfo>> source) {
202             dest.clear();
203     
204             for (Map.Entry<ValueNumber, Set<SourceSinkInfo>> entry : source.entrySet()) {
205                 HashSet<SourceSinkInfo> copy = new HashSet<SourceSinkInfo>(entry.getValue());
206                 dest.put(entry.getKey(), copy);
207             }
208         }

232         private void reset(State state) {
233             valueMap.clear();
234             whereAlways.clear();
235             whereNever.clear();
236             this.state = state;
237         }

239         public void propagateAcrossPhiNode(ValueNumber fromVN, ValueNumber toVN) {
240             assert isValid();
241     
242             setValue(toVN, getValue(fromVN));
243     
244             // Propagate source/sink information
245             transferSourceSinkInfoSet(whereAlways, fromVN, toVN);
246             transferSourceSinkInfoSet(whereNever, fromVN, toVN);
247     
248             // Remove all information about the "from" value
249             valueMap.remove(fromVN);
250             whereAlways.remove(fromVN);
251             whereNever.remove(fromVN);
252         }
253     
254         private static void transferSourceSinkInfoSet(Map<ValueNumber, Set<SourceSinkInfo>> sourceSinkInfoSetMap, ValueNumber fromVN,
255                 ValueNumber toVN) {
256             Set<? extends SourceSinkInfo> locSet = getSourceSinkInfoSet(sourceSinkInfoSetMap, fromVN);
257     
258             for (SourceSinkInfo loc : locSet) {
259                 addSourceSinkInfo(sourceSinkInfoSetMap, toVN, loc);
260             }
261         }

263         public void mergeWith(TypeQualifierValueSet fact) throws DataflowAnalysisException {
264             if (!isValid() || !fact.isValid()) {
265                 throw new DataflowAnalysisException("merging an invalid TypeQualifierValueSet");
266             }
267     
268             Set<ValueNumber> interesting = new HashSet<ValueNumber>();
269             interesting.addAll(this.valueMap.keySet());
270             interesting.addAll(fact.valueMap.keySet());
271     
272             for (ValueNumber vn : interesting) {
273                 setValue(vn, FlowValue.meet(this.getValue(vn), fact.getValue(vn)));
274                 mergeSourceSinkInfoSets(this.whereAlways, fact.whereAlways, vn);
275                 mergeSourceSinkInfoSets(this.whereNever, fact.whereNever, vn);
276             }
277         }
278     
279         private void mergeSourceSinkInfoSets(Map<ValueNumber, Set<SourceSinkInfo>> sourceSinkInfoSetMapToUpdate,
280                 Map<ValueNumber, Set<SourceSinkInfo>> otherSourceSinkInfoSetMap, ValueNumber vn) {
281             if (!otherSourceSinkInfoSetMap.containsKey(vn)) {
282                 return;
283             }
284             Set<? extends SourceSinkInfo> otherInfo = getSourceSinkInfoSet(otherSourceSinkInfoSetMap, vn);
285             if (!otherInfo.isEmpty()) {
286                 Set<SourceSinkInfo> sourceSinkInfoSetToUpdate = getOrCreateSourceSinkInfoSet(sourceSinkInfoSetMapToUpdate, vn);
287                 sourceSinkInfoSetToUpdate.addAll(otherInfo);
288             }
289         }

297         public boolean equals(Object obj) {
298             if (obj == null || obj.getClass() != this.getClass()) {
299                 return false;
300             }
301             TypeQualifierValueSet other = (TypeQualifierValueSet) obj;
302             if (this.isValid() && other.isValid()) {
303                 return this.valueMap.equals(other.valueMap);
304             } else {
305                 return this.state == other.state;
306             }

319         @Override
320         public String toString() {
321             if (state != State.VALID) {
322                 return state.toString();
323             }
324     
325             TreeSet<ValueNumber> interesting = new TreeSet<ValueNumber>();
326             interesting.addAll(valueMap.keySet());
327     
328             StringBuilder buf = new StringBuilder();
329     
330             buf.append("{");
331             //        boolean first = true;
332     
333             for (ValueNumber vn : interesting) {
334                 FlowValue value = getValue(vn);
335                 if (value == FlowValue.TOP ||  /* !isStrict && */ value == FlowValue.UNKNOWN) {
336                     continue;
337                 }
338                 if (buf.length() > 1) {
339                     buf.append(", ");
340                 }
341                 buf.append(valueNumberToString(vn, value));
342             }
343     
344             buf.append("}");
345     
346             return buf.toString();
347         }

349         public String valueNumberToString(ValueNumber vn ) {
350             return valueNumberToString(vn, getValue(vn));
351         }
352         public String valueNumberToString(ValueNumber vn, FlowValue value ) {
353             StringBuilder buf = new StringBuilder();
354     
355             buf.append(vn.getNumber());
356             buf.append("->");
357     
358             buf.append(value);
359             if (value != FlowValue.TOP) {
360                 Set<? extends SourceSinkInfo> always = getSourceSinkInfoSet(whereAlways, vn);
361                 Set<? extends SourceSinkInfo> never = getSourceSinkInfoSet(whereNever, vn);
362                 if (value != FlowValue.UNKNOWN || !always.equals(never)) {
363                     buf.append("[");
364                     if (!always.isEmpty()) {
365                         appendSourceSinkInfos(buf, "YES=", always);
366                     }
367                     if (!always.isEmpty() && !never.isEmpty()) {
368                         buf.append(",");
369                     }
370                     if (!never.isEmpty()) {
371                         appendSourceSinkInfos(buf, "NO=", never);
372                     }
373                     buf.append("]");
374                 }
375             }
376     
377             return buf.toString();
378         }
379     
380         private static void appendSourceSinkInfos(StringBuilder buf, String key, Set<? extends SourceSinkInfo> sourceSinkInfoSet) {
381             TreeSet<SourceSinkInfo> sortedLocSet = new TreeSet<SourceSinkInfo>();
382             sortedLocSet.addAll(sourceSinkInfoSet);
383             boolean first = true;
384             buf.append(key);
385             buf.append("(");
386             for (SourceSinkInfo loc : sortedLocSet) {
387                 if (first) {
388                     first = false;
389                 } else {
390                     buf.append(",");
391                 }
392                 buf.append(loc.getLocation().toCompactString());
393             }
394             buf.append(")");
395         }
