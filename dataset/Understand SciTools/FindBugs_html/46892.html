136         private static class InMemorySourceRepository implements SourceRepository {
137     
138             Map<String, byte[]> contents = new HashMap<String, byte[]>();
139             Map<String, Long> lastModified = new HashMap<String, Long>();
140     
141             InMemorySourceRepository(@WillClose ZipInputStream in) throws IOException {
142                 try {
143                     while (true) {
144     
145                         ZipEntry e = in.getNextEntry();
146                         if (e == null) {
147                             break;
148                         }
149                         if (!e.isDirectory()) {
150                             String name = e.getName();
151                             long size = e.getSize();
152     
153                             if (size > Integer.MAX_VALUE) {
154                                 throw new IOException(name + " is too big at " + size + " bytes");
155                             }
156                             ByteArrayOutputStream out;
157                             if (size <= 0) {
158                                 out = new ByteArrayOutputStream();
159                             } else {
160                                 out = new ByteArrayOutputStream((int) size);
161                             }
162                             GZIPOutputStream gOut = new GZIPOutputStream(out);
163                             IO.copy(in, gOut);
164                             gOut.close();
165                             byte data[] = out.toByteArray();
166                             contents.put(name, data);
167                             lastModified.put(name, e.getTime());
168                         }
169                         in.closeEntry();
170                     }
171                 } finally {
172                     Util.closeSilently(in);
173                 }
174     
175             }
176     
177             @Override
178             public boolean contains(String fileName) {
179                 return contents.containsKey(fileName);
180             }
181     
182             @Override
183             public SourceFileDataSource getDataSource(final String fileName) {
184                 return new SourceFileDataSource() {
185     
186                     @Override
187                     public String getFullFileName() {
188                         return fileName;
189                     }
190     
191                     @Override
192                     public InputStream open() throws IOException {
193                         return new GZIPInputStream(new ByteArrayInputStream(contents.get(fileName)));
194                     }
195     
196                     @Override
197                     public long getLastModified() {
198                         Long when = lastModified.get(fileName);
199                         if (when == null || when < 0) {
200                             return 0;
201                         }
202                         return when;
203                     }
204                 };
205             }
206     
207             @Override
208             public boolean isPlatformDependent() {
209                 return false;
210             }
211         }

213         SourceRepository makeInMemorySourceRepository(final String url) {
214             final BlockingSourceRepository r = new BlockingSourceRepository();
215             Util.runInDameonThread(new Runnable() {
216     
217                 @Override
218                 public void run() {
219                     InputStream in = null;
220                     try {
221     
222                         URLConnection connection = new URL(url).openConnection();
223                         in = connection.getInputStream();
224                         if (getProject().isGuiAvaliable()) {
225                             in = getProject().getGuiCallback().getProgressMonitorInputStream(in, connection.getContentLength(),
226                                     "Downloading project source code...");
227                         }
228     
229                         if (url.endsWith(".z0p.gz")) {
230                             in = new GZIPInputStream(in);
231                         }
232     
233                         r.setBase(new InMemorySourceRepository(new ZipInputStream(in)));
234     
235                     } catch (IOException e) {
236                         if (getProject().isGuiAvaliable()) {
237                             getProject().getGuiCallback().setErrorMessage("Unable to load " + url + "; " + e.getMessage());
238                         }
239                         AnalysisContext.logError("Unable to load " + url, e);
240                         Util.closeSilently(in);
241                     }
242                 }
243             }, "Source loading thread");
244             return r;
245         }

247         SourceRepository makeJarURLConnectionSourceRepository(final String url) throws MalformedURLException, IOException {
248             final File file = File.createTempFile("jar_cache", null);
249             file.deleteOnExit();
250             final BlockingSourceRepository r = new BlockingSourceRepository();
251             Util.runInDameonThread(new Runnable() {
252     
253                 @Override
254                 public void run() {
255                     InputStream in = null;
256                     OutputStream out = null;
257                     try {
258                         URLConnection connection = new URL(url).openConnection();
259                         if (getProject().isGuiAvaliable()) {
260                             int size = connection.getContentLength();
261                             in = getProject().getGuiCallback().getProgressMonitorInputStream(connection.getInputStream(), size,
262                                     "Loading source via url");
263                         } else {
264                             in = connection.getInputStream();
265                         }
266                         out = new FileOutputStream(file);
267                         IO.copy(in, out);
268                         r.setBase(new ZipSourceRepository(new ZipFile(file)));
269                     } catch (IOException e) {
270                         assert true;
271                     } finally {
272                         Util.closeSilently(in);
273                         Util.closeSilently(out);
274                     }
275                 }
276             }, "Source loading thread");
277             return r;
278         }

280         static class BlockingSourceRepository implements SourceRepository {
281             SourceRepository base;
282     
283             final CountDownLatch ready = new CountDownLatch(1);
284     
285             public BlockingSourceRepository() {
286             }
287     
288             public boolean isReady() {
289                 return ready.getCount() == 0;
290             }
291     
292             public void setBase(SourceRepository base) {
293                 this.base = base;
294                 ready.countDown();
295             }
296     
297             private void await() {
298                 try {
299                     ready.await();
300                 } catch (InterruptedException e) {
301                     throw new IllegalStateException("Unexpected interrupt", e);
302                 }
303             }
304     
305             @Override
306             public boolean contains(String fileName) {
307                 await();
308                 return base.contains(fileName);
309             }
310     
311             @Override
312             public SourceFileDataSource getDataSource(String fileName) {
313                 await();
314                 return base.getDataSource(fileName);
315             }
316     
317             @Override
318             public boolean isPlatformDependent() {
319                 await();
320                 return base.isPlatformDependent();
321             }
322     
323         }

380         /**
381          * Set the list of source directories.
382          */
383         void setSourceBaseList(Iterable<String> sourceBaseList) {
384             for (String repos : sourceBaseList) {
385                 if (repos.endsWith(".zip") || repos.endsWith(".jar") || repos.endsWith(".z0p.gz")) {
386                     // Zip or jar archive
387                     try {
388                         if (repos.startsWith("http:") || repos.startsWith("https:") || repos.startsWith("file:")) {
389                             String url = SystemProperties.rewriteURLAccordingToProperties(repos);
390                             repositoryList.add(makeInMemorySourceRepository(url));
391                         } else {
392                             repositoryList.add(new ZipSourceRepository(new ZipFile(repos)));
393                         }
394                     } catch (IOException e) {
395                         // Ignored - we won't use this archive
396                         AnalysisContext.logError("Unable to load " + repos, e);
397                     }
398                 } else {
399                     File dir = new File(repos);
400                     if (dir.canRead() && dir.isDirectory()) {
401                         repositoryList.add(new DirectorySourceRepository(repos));
402                     } else {
403                         AnalysisContext.logError("Unable to load " + repos);
404     
405                     }
406                 }
407             }
408         }

436         /**
437          * Open a source file in given package.
438          *
439          * @param packageName
440          *            the name of the package containing the class whose source file
441          *            is given
442          * @param fileName
443          *            the unqualified name of the source file
444          * @return the source file
445          * @throws IOException
446          *             if a matching source file cannot be found
447          */
448         public SourceFile findSourceFile(String packageName, String fileName) throws IOException {
449             // On windows the fileName specification is different between a file in
450             // a directory tree, and a
451             // file in a zip file. In a directory tree the separator used is '\',
452             // while in a zip it's '/'
453             // Therefore for each repository figure out what kind it is and use the
454             // appropriate separator.
455     
456             // In all practicality, this code could just use the hardcoded '/' char,
457             // as windows can open
458             // files with this separator, but to allow for the mythical 'other'
459             // platform that uses an
460             // alternate separator, make a distinction
461     
462             String platformName = getPlatformName(packageName, fileName);
463             String canonicalName = getCanonicalName(packageName, fileName);
464     
465             // Is the file in the cache already? Always cache it with the canonical
466             // name
467             SourceFile sourceFile = cache.get(canonicalName);
468             if (sourceFile != null) {
469                 return sourceFile;
470             }
471     
472             // Find this source file, add its data to the cache
473             if (DEBUG) {
474                 System.out.println("Trying " + fileName + " in package " + packageName + "...");
475             }
476             // Query each element of the source path to find the requested source
477             // file
478             for (SourceRepository repos : repositoryList) {
479                 if (repos instanceof BlockingSourceRepository && !((BlockingSourceRepository) repos).isReady()) {
480                     continue;
481                 }
482                 fileName = repos.isPlatformDependent() ? platformName : canonicalName;
483                 if (DEBUG) {
484                     System.out.println("Looking in " + repos + " for " + fileName);
485                 }
486                 if (repos.contains(fileName)) {
487                     // Found it
488                     sourceFile = new SourceFile(repos.getDataSource(fileName));
489                     cache.put(canonicalName, sourceFile); // always cache with
490                     // canonicalName
491                     return sourceFile;
492                 }
493             }
494     
495             throw new FileNotFoundException("Can't find source file " + fileName);
496         }

499             String platformName = packageName.replace('.', File.separatorChar) + (packageName.length() > 0 ? File.separator : "")

502         }

517         public static String getOrGuessSourceFile(SourceLineAnnotation source)  {
518             if (source.isSourceFileKnown()) {
519                 return source.getSourceFile();
520             }
521             String baseClassName = source.getClassName();
522             int i = baseClassName.lastIndexOf('.');
523             baseClassName = baseClassName.substring(i + 1);
524             int j = baseClassName.indexOf('$');
525             if (j >= 0) {
526                 baseClassName = baseClassName.substring(0, j);
527             }
528             return baseClassName + ".java";
529         }

535         public boolean hasSourceFile(String packageName, String fileName) {
536             // On windows the fileName specification is different between a file in
537             // a directory tree, and a
538             // file in a zip file. In a directory tree the separator used is '\',
539             // while in a zip it's '/'
540             // Therefore for each repository figure out what kind it is and use the
541             // appropriate separator.
542     
543             // In all practicality, this code could just use the hardcoded '/' char,
544             // as windows can open
545             // files with this separator, but to allow for the mythical 'other'
546             // platform that uses an
547             // alternate separator, make a distinction
548     
549             // Create a fully qualified source filename using the package name for
550             // both directories and zips
551             String platformName = getPlatformName(packageName, fileName);
552             String canonicalName = getCanonicalName(packageName, fileName);
553     
554             // Is the file in the cache already? Always cache it with the canonical
555             // name
556             SourceFile sourceFile = cache.get(canonicalName);
557             if (sourceFile != null) {
558                 return true;
559             }
560     
561             // Find this source file, add its data to the cache
562             if (DEBUG) {
563                 System.out.println("Trying " + fileName + " in package " + packageName + "...");
564             }
565             // Query each element of the source path to find the requested source
566             // file
567             for (SourceRepository repos : repositoryList) {
568                 if (repos instanceof BlockingSourceRepository && !((BlockingSourceRepository) repos).isReady()) {
569                     continue;
570                 }
571                 fileName = repos.isPlatformDependent() ? platformName : canonicalName;
572                 if (DEBUG) {
573                     System.out.println("Looking in " + repos + " for " + fileName);
574                 }
575                 if (repos.contains(fileName)) {
576                     return true;
577                 }
578             }
579     
580             return false;
581         }
