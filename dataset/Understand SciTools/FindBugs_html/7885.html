20     package edu.umd.cs.findbugs.util;
21     
22     import java.io.IOException;
23     import java.lang.reflect.InvocationTargetException;
24     import java.lang.reflect.Method;
25     import java.net.URI;
26     import java.net.URISyntaxException;
27     import java.net.URL;
28     import java.util.regex.Pattern;
29     
30     import javax.swing.JOptionPane;
31     
32     import edu.umd.cs.findbugs.SystemProperties;
33     
34     /**
35      *
36      */
37     public class LaunchBrowser {
38     
39         private static Pattern validExec = Pattern.compile("[a-zA-Z0-9-_/]+");
40     
41         public static final String execCommand = SystemProperties.getProperty("findbugs.execCommand");
42     
43         public static final boolean launchViaExec = execCommand != null && validExec.matcher(execCommand).matches()
44                 && "Linux".equals(SystemProperties.getProperty("os.name"));
45     
46         private static Object desktopObject;
47     
48         private static Method desktopBrowseMethod;
49     
50         private static boolean launchViaExecFailed = false;
51     
52         static boolean DEBUG = SystemProperties.getBoolean("findbugs.launchBrowser.debug");
53     
54         static final Exception desktopException;
55     
56         static {
57             Exception toStore = null;
58             try {
59                 Class<?> desktopClass = Class.forName("java.awt.Desktop");
60                 desktopObject = desktopClass.getMethod("getDesktop").invoke(null);
61                 desktopBrowseMethod = desktopClass.getMethod("browse", URI.class);
62             } catch (Exception e) {
63                 toStore = e;
64             }
65             desktopException = toStore;
66         }
67     
68         static boolean desktopFeasible() {
69             return desktopObject != null && desktopBrowseMethod != null;
70         }
71     
72         static boolean webstartFeasible() {
73             return JavaWebStart.jnlpShowDocumentMethod != null && JavaWebStart.jnlpBasicService != null;
74         }
75     
76         static boolean showDocumentViaDesktop(URL u) {
77     
78             if (desktopObject != null && desktopBrowseMethod != null) {
79                 try {
80                     if (DEBUG) {
81                         JOptionPane.showMessageDialog(null, "Trying desktop browse");
82                     }
83                     viaDesktop(u.toURI());
84                     if (DEBUG) {
85                         JOptionPane.showMessageDialog(null, "desktop browse succeeded");
86                     }
87                     return true;
88                 } catch (InvocationTargetException ite) {
89                     assert true;
90                 } catch (IllegalAccessException iae) {
91                     assert true;
92                 } catch (IllegalArgumentException e) {
93                     assert true;
94                 } catch (URISyntaxException e) {
95                     assert true;
96                 }
97             }
98             return false;
99         }
100     
101         static void viaDesktop(URI u) throws IllegalAccessException, InvocationTargetException {
102             if (desktopBrowseMethod == null) {
103                 throw new UnsupportedOperationException("Launch via desktop not available");
104             }
105             desktopBrowseMethod.invoke(desktopObject, u);
106         }
107     
108         static boolean showDocumentViaExec(URL url) {
109             if (launchViaExec && !launchViaExecFailed) {
110                 if (DEBUG) {
111                     JOptionPane.showMessageDialog(null, "Trying exec browse");
112                 }
113     
114                 try {
115                     Process p = launchViaExec(url);
116                     Thread.sleep(90);
117     
118                     int exitValue = p.exitValue();
119                     if (exitValue != 0) {
120                         launchViaExecFailed = true;
121                         if (DEBUG) {
122                             JOptionPane.showMessageDialog(null, "exec browse launch failed with exit code " + exitValue);
123                         }
124                         return false;
125                     }
126                     if (DEBUG) {
127                         JOptionPane.showMessageDialog(null, "exec browse succeeded");
128                     }
129                     return true;
130                 } catch (IllegalThreadStateException e) {
131                     if (DEBUG) {
132                         JOptionPane.showMessageDialog(null, "exec browse succeeded but not done");
133                     }
134                     return true;
135                 } catch (Exception e) {
136                     if (DEBUG) {
137                         JOptionPane.showMessageDialog(null, "exec browse failed" + e.getMessage());
138                     }
139                     launchViaExecFailed = true;
140                 }
141             }
142             return false;
143     
144         }
145     
146         static Process launchViaExec(URL url) throws IOException {
147             ProcessBuilder builder = new ProcessBuilder(execCommand, url.toString());
148             Process p = builder.start();
149             return p;
150         }
151     
152         /**
153          * attempt to show the given URL. will first attempt via the JNLP api, then
154          * will try showViaExec().
155          *
156          * @param url
157          *            the URL
158          * @return true on success
159          */
160         public static boolean showDocument(URL url) {
161     
162             if (showDocumentViaDesktop(url)) {
163                 return true;
164             }
165             if (showDocumentViaExec(url)) {
166                 return true;
167             }
168             if (JavaWebStart.showViaWebStart(url)) {
169                 return true;
170             }
171     
172             return false;
173     
174         }
