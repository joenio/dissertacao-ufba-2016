39     /**
40      * Helper methods to find out information about the source of the value
41      * represented by a given ValueNumber.
42      *
43      * @author Bill Pugh
44      * @author David Hovemeyer
45      */
46     public abstract class ValueNumberSourceInfo {
47     
48         /**
49          * @param method
50          * @param location
51          * @param valueNumber
52          * @param vnaFrame
53          * @param partialRole
54          *            TODO
55          * @return the annotation
56          */
57         public static @CheckForNull
58         BugAnnotation findAnnotationFromValueNumber(Method method, Location location, ValueNumber valueNumber,
59                 ValueNumberFrame vnaFrame, @CheckForNull String partialRole) {
60             if (location.getHandle().getInstruction() instanceof ACONST_NULL) {
61                 StringAnnotation nullConstant = new StringAnnotation("null");
62                 nullConstant.setDescription(StringAnnotation.STRING_NONSTRING_CONSTANT_ROLE);
63                 return nullConstant;
64             }
65             LocalVariableAnnotation ann = ValueNumberSourceInfo.findLocalAnnotationFromValueNumber(method, location, valueNumber,
66                     vnaFrame);
67             if (ann != null && partialRole != null) {
68                 ann.setDescription("LOCAL_VARIABLE_" + partialRole);
69             }
70     
71             if (ann != null && ann.isSignificant()) {
72                 return ann;
73             }
74             FieldAnnotation field = ValueNumberSourceInfo.findFieldAnnotationFromValueNumber(method, location, valueNumber, vnaFrame);
75             if (field != null) {
76                 if (partialRole != null) {
77                     field.setDescription("FIELD_" + partialRole);
78                 }
79     
80                 return field;
81             }
82             if (ann != null) {
83                 return ann;
84             }
85             return null;
86         }
87     
88         /**
89          * @param method
90          * @param location
91          * @param valueNumber
92          * @param vnaFrame
93          * @param partialRole
94          *            TODO
95          * @return the annotation
96          */
97         public static @Nonnull
98         BugAnnotation findRequiredAnnotationFromValueNumber(Method method, Location location, ValueNumber valueNumber,
99                 ValueNumberFrame vnaFrame, @CheckForNull String partialRole) {
100             BugAnnotation result = findAnnotationFromValueNumber(method, location, valueNumber, vnaFrame, partialRole);
101             if (result != null) {
102                 return result;
103             }
104             return new LocalVariableAnnotation("?", -1, location.getHandle().getPosition());
105         }
106     
107         public static LocalVariableAnnotation findLocalAnnotationFromValueNumber(Method method, Location location,
108                 ValueNumber valueNumber, ValueNumberFrame vnaFrame) {
109     
110             if (vnaFrame == null || vnaFrame.isBottom() || vnaFrame.isTop()) {
111                 return null;
112             }
113     
114             LocalVariableAnnotation localAnnotation = null;
115             for (int i = 0; i < vnaFrame.getNumLocals(); i++) {
116                 if (valueNumber.equals(vnaFrame.getValue(i))) {
117                     InstructionHandle handle = location.getHandle();
118                     InstructionHandle prev = handle.getPrev();
119                     if (prev == null) {
120                         continue;
121                     }
122                     int position1 = prev.getPosition();
123                     int position2 = handle.getPosition();
124                     localAnnotation = LocalVariableAnnotation.getLocalVariableAnnotation(method, i, position1, position2);
125                     if (localAnnotation != null) {
126                         return localAnnotation;
127                     }
128                 }
129             }
130             return null;
131         }
132     
133         public static FieldAnnotation findFieldAnnotationFromValueNumber(Method method, Location location, ValueNumber valueNumber,
134                 ValueNumberFrame vnaFrame) {
135             XField field = ValueNumberSourceInfo.findXFieldFromValueNumber(method, location, valueNumber, vnaFrame);
136             if (field == null) {
137                 return null;
138             }
139             return FieldAnnotation.fromXField(field);
140         }
141     
142         public static XField findXFieldFromValueNumber(Method method, Location location, ValueNumber valueNumber,
143                 ValueNumberFrame vnaFrame) {
144             if (vnaFrame == null || vnaFrame.isBottom() || vnaFrame.isTop()) {
145                 return null;
146             }
147     
148             AvailableLoad load = vnaFrame.getLoad(valueNumber);
149             if (load != null) {
150                 return load.getField();
151             }
152             return null;
153         }
154     
155         /**
156          * @param classContext
157          * @param method
158          * @param location
159          * @param stackPos
160          * @throws DataflowAnalysisException
161          * @throws CFGBuilderException
162          */
163         static @CheckForNull
164         public BugAnnotation getFromValueNumber(ClassContext classContext, Method method, Location location, int stackPos)
165                 throws DataflowAnalysisException, CFGBuilderException {
166             ValueNumberFrame vnaFrame = classContext.getValueNumberDataflow(method).getFactAtLocation(location);
167             if (!vnaFrame.isValid()) {
168                 return null;
169             }
170             ValueNumber valueNumber = vnaFrame.getStackValue(stackPos);
171             if (valueNumber.hasFlag(ValueNumber.CONSTANT_CLASS_OBJECT)) {
172                 return null;
173             }
174             BugAnnotation variableAnnotation = findAnnotationFromValueNumber(method, location, valueNumber, vnaFrame, "VALUE_OF");
175     
176             return variableAnnotation;
177     
178         }
