31     /**
32      * Scan classes for @NonNull, @PossiblyNull and @CheckForNull annotations, and
33      * convey them to FindNullDeref.
34      *
35      * @deprecated AnnotationDatabases are being phased out, since annotations are
36      *             now stored directly in the XClass/XMethod/XField objects.
37      *             Resolving nullness annotations will be handled through the
38      *             JSR-305 type qualifier code.
39      */
40     @Deprecated
41     public class NoteNonNullAnnotations extends BuildNonNullAnnotationDatabase implements Detector, NonReportingDetector {
42     
43         public NoteNonNullAnnotations(BugReporter bugReporter) {
44             super(getDatabase());
45         }
46     
47         private static NullnessAnnotationDatabase getDatabase() {
48             return null;
49         }
50     
51         @Override
52         public void visitClassContext(ClassContext classContext) {
53     
54             JavaClass javaClass = classContext.getJavaClass();
55             if (!BCELUtil.preTiger(javaClass)) {
56                 javaClass.accept(this);
57             }
58         }
59     
60         @Override
61         public void report() {
62         }
63     }
