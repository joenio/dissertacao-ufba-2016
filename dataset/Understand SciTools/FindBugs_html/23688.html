20     package edu.umd.cs.findbugs.ba.npe;
21     
22     import edu.umd.cs.findbugs.ba.JavaClassAndMethod;
23     import edu.umd.cs.findbugs.ba.interproc.ParameterProperty;
24     
25     public class NonNullSpecification {
26         private final JavaClassAndMethod classAndMethod;
27     
28         private final ParameterProperty nonNullProperty;
29     
30         private final ParameterProperty possiblyNullProperty;
31     
32         public NonNullSpecification(JavaClassAndMethod classAndMethod, ParameterProperty nonParamProperty,
33                 ParameterProperty possiblyNullProperty) {
34             this.classAndMethod = classAndMethod;
35             this.nonNullProperty = nonParamProperty;
36             this.possiblyNullProperty = possiblyNullProperty;
37         }
38     
39         public JavaClassAndMethod getClassAndMethod() {
40             return classAndMethod;
41         }
42     
43         public ParameterProperty getNonNullProperty() {
44             return nonNullProperty;
45         }
46     
47         public ParameterProperty getCheckForNullProperty() {
48             return possiblyNullProperty;
49         }
50     
51         @Override
52         public String toString() {
53             StringBuilder buf = new StringBuilder();
54             buf.append(classAndMethod);
55             buf.append(":");
56             if (!nonNullProperty.isEmpty()) {
57                 buf.append(" nonull=");
58                 buf.append(nonNullProperty);
59             }
60             if (!possiblyNullProperty.isEmpty()) {
61                 buf.append(" possiblynull=");
62                 buf.append(possiblyNullProperty);
63             }
64             return buf.toString();
65         }
66     }
