20     package edu.umd.cs.findbugs.gui2;
21     
22     import java.util.ArrayList;
23     import java.util.Iterator;
24     
25     import edu.umd.cs.findbugs.filter.Matcher;
26     
27     /**
28      * These are the branches in our tree, each branch forms a complete query that
29      * could be sent to the main bugset to return all the bugs it contains For
30      * example, a single bugAspects could be <priority,high> or it could be
31      * <priority,high>, <designation,must fix>,<class,fishpond>,<package,default>
32      *
33      * In this implementation, <priority,high>,<designation,unclassified> is
34      * different from <designation,unclassified>,<priority,high>. (I'm not talking
35      * about the fact we use the .equals from ArrayList, I'm talking about what a
36      * query would return, though both are true) For a speed boost, this class could
37      * be rewritten to make these equal, BugSet could be rewritten to cache full
38      * queries off the main BugSet, (instead of caching each part of the query
39      * separately in the BugSets created) and resetData could be rewritten to work
40      * more like Swing's validate, only clearing data if the data is wrong. This
41      * would save time after changing certain aspects of the tree. Just an idea, I
42      * wouldn't suggest it unless its absolutely necessary. -Dan
43      *
44      *
45      * @author All of us
46      */
47     public class BugAspects implements Iterable<BugAspects.SortableValue> {
48         private static final long serialVersionUID = -5503915081879996968L;
49     
50         private int count = -1;
51     
52         private ArrayList<BugAspects.SortableValue> lst = new ArrayList<BugAspects.SortableValue>();
53     
54         public SortableValue last() {
55             return lst.get(lst.size() - 1);
56         }
57     
58         public int size() {
59             return lst.size();
60         }
61     
62         public SortableValue get(int i) {
63             return lst.get(i);
64         }
65     
66         @Override
67         public String toString() {
68             if (lst.isEmpty()) {
69                 return edu.umd.cs.findbugs.L10N.getLocalString("tree.bugs", "Bugs") + " (" + count + ")";
70             } else {
71                 if (count == -1) {
72                     return last().value;
73                 } else {
74                     return last().key.formatValue(last().value) + " (" + count + ")";
75                 }
76             }
77         }
78     
79         /**
80          * This is how the numbers after the branches contain the number of bugs in
81          * them, even if they aren't the final branch
82          *
83          * @param count
84          */
85         public void setCount(int count) {
86             this.count = count;
87         }
88     
89         public int getCount() {
90             return count;
91         }
92     
93         public BugAspects() {
94             super();
95         }
96     
97         public BugAspects(BugAspects a) {
98             lst = new ArrayList<SortableValue>(a.lst);
99             count = a.count;
100         }
101     
102         public void add(SortableValue sp) {
103             lst.add(sp);
104         }
105     
106         public BugAspects addToNew(SortableValue sp) {
107             BugAspects result = new BugAspects(this);
108             result.lst.add(sp);
109             return result;
110         }
111     
112         public Matcher getMatcher() {
113             return FilterFactory.makeAndMatcher(lst);
114         }
115     
116         public StackedFilterMatcher getStackedFilterMatcher() {
117             FilterMatcher[] filters = new FilterMatcher[lst.size()];
118             for (int i = 0; i < filters.length; i++) {
119                 filters[i] = new FilterMatcher(lst.get(i));
120             }
121             StackedFilterMatcher sfm = new StackedFilterMatcher(filters);
122             return sfm;
123         }
124     
125         public BugSet getMatchingBugs(BugSet theSet) {
126             return theSet.getBugsMatchingFilter(this.getStackedFilterMatcher());
127         }
128     
129         static class SortableValue {
130             final public Sortables key;
131     
132             final public String value;
133     
134             public SortableValue(Sortables key, String value) {
135                 this.key = key;
136                 this.value = value;
137             }
138     
139             @Override
140             public int hashCode() {
141                 return key.hashCode() + value.hashCode();
142             }
143     
144             @Override
145             public boolean equals(Object that) {
146                 if (!(that instanceof SortableValue)) {
147                     return false;
148                 }
149                 SortableValue thatStringPair = ((SortableValue) that);
150                 return this.key.equals(thatStringPair.key) && this.value.equals(thatStringPair.value);
151             }
152     
153             @Override
154             public String toString() {
155                 return key + ":" + value;
156             }
157         }
158     
159         @Override
160         public Iterator<SortableValue> iterator() {
161             return lst.iterator();
162         }
163     }
