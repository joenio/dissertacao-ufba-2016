20     package edu.umd.cs.findbugs.detect;
21     
22     import java.util.HashSet;
23     import java.util.Set;
24     
25     import org.apache.bcel.Repository;
26     import org.apache.bcel.classfile.JavaClass;
27     
28     import edu.umd.cs.findbugs.BugInstance;
29     import edu.umd.cs.findbugs.BugReporter;
30     import edu.umd.cs.findbugs.BytecodeScanningDetector;
31     import edu.umd.cs.findbugs.ba.ClassContext;
32     import edu.umd.cs.findbugs.internalAnnotations.StaticConstant;
33     
34     public class XMLFactoryBypass extends BytecodeScanningDetector {
35         private final BugReporter bugReporter;
36     
37         @StaticConstant
38         private static final Set<String> xmlInterfaces = new HashSet<String>() {
39             static final long serialVersionUID = -9117982073509840017L;
40             {
41                 add("javax.xml.parsers.DocumentBuilder");
42                 add("org.w3c.dom.Document");
43                 add("javax.xml.parsers.SAXParser");
44                 add("org.xml.sax.XMLReader");
45                 add("org.xml.sax.XMLFilter");
46                 add("javax.xml.transform.Transformer");
47                 add("org.w3c.dom.Attr");
48                 add("org.w3c.dom.CDATASection");
49                 add("org.w3c.dom.Comment");
50                 add("org.w3c.dom.Element");
51                 add("org.w3c.dom.Text");
52             }
53         };
54     
55         private final Set<String> rejectedXMLClasses = new HashSet<String>();
56     
57         private JavaClass curClass;
58     
59         public XMLFactoryBypass(BugReporter bugReporter) {
60             this.bugReporter = bugReporter;
61         }
62     
63         @Override
64         public void visitClassContext(ClassContext classContext) {
65             curClass = classContext.getJavaClass();
66             super.visitClassContext(classContext);
67         }
68     
69         @Override
70         public void sawOpcode(int seen) {
71             try {
72                 if (seen == INVOKESPECIAL) {
73                     String newClsName = getClassConstantOperand();
74                     if (rejectedXMLClasses.contains(newClsName)) {
75                         return;
76                     }
77                     rejectedXMLClasses.add(newClsName);
78     
79                     if (newClsName.startsWith("java/") || newClsName.startsWith("javax/")) {
80                         return;
81                     }
82     
83                     if (newClsName.endsWith("Adapter")) {
84                         return;
85                     }
86     
87                     if (!"<init>".equals(getNameConstantOperand())) {
88                         return;
89                     }
90     
91                     String invokerClsName = this.getClassName();
92                     if (samePackageBase(invokerClsName, newClsName)) {
93                         return;
94                     }
95     
96                     JavaClass newCls = Repository.lookupClass(getDottedClassConstantOperand());
97     
98                     JavaClass superCls = curClass.getSuperClass();
99                     if (superCls.getClassName().equals(newClsName.replace('/', '.'))) {
100                         return;
101                     }
102     
103                     JavaClass[] infs = newCls.getAllInterfaces();
104                     for (JavaClass inf : infs) {
105                         if (xmlInterfaces.contains(inf.getClassName())) {
106                             bugReporter.reportBug(new BugInstance(this, "XFB_XML_FACTORY_BYPASS", LOW_PRIORITY).addClassAndMethod(
107                                     this).addSourceLine(this));
108                             rejectedXMLClasses.remove(newClsName);
109                         }
110                     }
111                 }
112             } catch (ClassNotFoundException cnfe) {
113                 bugReporter.reportMissingClass(cnfe);
114             }
115         }
116     
117         public boolean samePackageBase(String invokerClsName, String newClsName) {
118             String[] invokerParts = invokerClsName.split("/");
119             String[] newClsParts = newClsName.split("/");
120     
121             if (newClsParts.length < 3) {
122                 return false;
123             }
124             if (invokerParts.length < 3) {
125                 return false;
126             }
127     
128             if (!invokerParts[0].equals(newClsParts[0])) {
129                 return false;
130             }
131     
132             return invokerParts[1].equals(newClsParts[1]);
133         }
134     }
