14     public class IDivResultCastToDouble extends BytecodeScanningDetector {
15         private static final boolean DEBUG = SystemProperties.getBoolean("idcd.debug");
16     
17         //    private final BugReporter bugReporter;
18     
19         private final BugAccumulator bugAccumulator;
20     
21         private int prevOpCode;
22     
23         public IDivResultCastToDouble(BugReporter bugReporter) {
24             //        this.bugReporter = bugReporter;
25             this.bugAccumulator = new BugAccumulator(bugReporter);
26         }
27     
28         @Override
29         public void visit(Method obj) {
30             if (DEBUG) {
31                 System.out.println("Visiting " + obj);
32             }
33         }
34     
35         @Override
36         public void visit(Code obj) {
37             super.visit(obj);
38             bugAccumulator.reportAccumulatedBugs();
39         }
40     
41         SourceLineAnnotation pendingIdivCastToDivBugLocation = null;
42     
43         @Override
44         public void sawOpcode(int seen) {
45     
46             if (DEBUG) {
47                 System.out.println("Saw opcode " + OPCODE_NAMES[seen] + " " + pendingIdivCastToDivBugLocation);
48             }
49     
50             if ((prevOpCode == I2D || prevOpCode == L2D) && seen == INVOKESTATIC && ClassName.isMathClass(getClassConstantOperand())
51                     && "ceil".equals(getNameConstantOperand())) {
52                 bugAccumulator
53                 .accumulateBug(new BugInstance(this, "ICAST_INT_CAST_TO_DOUBLE_PASSED_TO_CEIL", HIGH_PRIORITY)
54                 .addClassAndMethod(this), this);
55                 pendingIdivCastToDivBugLocation = null;
56             } else if ((prevOpCode == I2F || prevOpCode == L2F) && seen == INVOKESTATIC
57                     && ClassName.isMathClass(getClassConstantOperand()) && "round".equals(getNameConstantOperand())) {
58                 bugAccumulator.accumulateBug(
59                         new BugInstance(this, "ICAST_INT_CAST_TO_FLOAT_PASSED_TO_ROUND", NORMAL_PRIORITY).addClassAndMethod(this),
60                         this);
61                 pendingIdivCastToDivBugLocation = null;
62             } else if (pendingIdivCastToDivBugLocation != null) {
63                 bugAccumulator.accumulateBug(
64                         new BugInstance(this, "ICAST_IDIV_CAST_TO_DOUBLE", NORMAL_PRIORITY).addClassAndMethod(this),
65                         pendingIdivCastToDivBugLocation);
66                 pendingIdivCastToDivBugLocation = null;
67             }
68     
69             if (prevOpCode == IDIV && (seen == I2D || seen == I2F) || prevOpCode == LDIV && (seen == L2D || seen == L2F)) {
70                 pendingIdivCastToDivBugLocation = SourceLineAnnotation.fromVisitedInstruction(this);
71             }
72             prevOpCode = seen;
73         }
74     }
