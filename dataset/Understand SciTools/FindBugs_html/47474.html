20     package edu.umd.cs.findbugs.ba;
21     
22     import java.util.Iterator;
23     import java.util.LinkedList;
24     import java.util.List;
25     
26     import edu.umd.cs.findbugs.SystemProperties;
27     
28     /**
29      * Object to enumerate (some subset of) the simple paths in a CFG. A simple path
30      * is a path from entry to exit, ignoring backedges and unhandled exceptions.
31      * <p/>
32      * <p>
33      * FIXME: instead of storing the simple paths, should invoke a callback as each
34      * simple path is produced. That would save memory.
35      *
36      * @author David Hovemeyer
37      * @see CFG
38      */
39     public class SimplePathEnumerator implements EdgeTypes, DFSEdgeTypes {
40         private final CFG cfg;
41     
42         private final DepthFirstSearch dfs;
43     
44         private final int maxPaths;
45     
46         private final int maxWork;
47     
48         private int work;
49     
50         private final List<List<Edge>> pathList;
51     
52         private static final boolean DEBUG = SystemProperties.getBoolean("spe.debug");
53     
54         /**
55          * Default number of steps to be performed in path enumeration.
56          */
57         public static final int DEFAULT_MAX_WORK = 200000;
58     
59         /**
60          * Constructor.
61          *
62          * @param cfg
63          *            the control flow graph to enumerate simple paths of
64          * @param maxPaths
65          *            maximum number of simple paths to find
66          * @param maxWork
67          *            maximum number of steps to be performed in the path
68          *            enumeration (to handle exponential blowup of search space)
69          */
70         public SimplePathEnumerator(CFG cfg, int maxPaths, int maxWork) {
71             this.cfg = cfg;
72             this.dfs = new DepthFirstSearch(cfg);
73             dfs.search();
74             this.maxPaths = maxPaths;
75             this.maxWork = maxWork;
76             this.work = 0;
77             this.pathList = new LinkedList<List<Edge>>();
78         }
79     
80         /**
81          * Constructor; max work is set to DEFAULT_MAX_WORK.
82          *
83          * @param cfg
84          *            the control flow graph to enumerate simple paths of
85          * @param maxPaths
86          *            maximum number of simple paths to find
87          */
88         public SimplePathEnumerator(CFG cfg, int maxPaths) {
89             this(cfg, maxPaths, DEFAULT_MAX_WORK);
90         }
91     
92         /**
93          * Enumerate the simple paths.
94          *
95          * @return this object
96          */
97         public SimplePathEnumerator enumerate() {
98             Iterator<Edge> entryOut = cfg.outgoingEdgeIterator(cfg.getEntry());
99             if (!entryOut.hasNext()) {
100                 throw new IllegalStateException();
101             }
102             Edge entryEdge = entryOut.next();
103     
104             LinkedList<Edge> init = new LinkedList<Edge>();
105             init.add(entryEdge);
106     
107             work(init);
108             if (DEBUG && work == maxWork) {
109                 System.out.println("**** Reached max work! ****");
110             }
111     
112             return this;
113         }
114     
115         /**
116          * Iterate over simple paths.
117          */
118         public Iterator<List<Edge>> iterator() {
119             return pathList.iterator();
120         }
121     
122         private void work(LinkedList<Edge> partialPath) {
123             if (pathList.size() == maxPaths) {
124                 return;
125             }
126     
127             Edge last = partialPath.getLast();
128     
129             // Is this a complete path?
130             if (last.getTarget() == cfg.getExit()) {
131                 pathList.add(new LinkedList<Edge>(partialPath));
132                 return;
133             }
134     
135             // Look for non-backedge, non-unhandled-exception outgoing edges, and
136             // recur.
137             Iterator<Edge> i = cfg.outgoingEdgeIterator(last.getTarget());
138             while (i.hasNext()) {
139                 Edge outEdge = i.next();
140     
141                 // Ignore back edges and unhandled exception edges
142                 if (dfs.getDFSEdgeType(outEdge) == BACK_EDGE || outEdge.getType() == UNHANDLED_EXCEPTION_EDGE) {
143                     continue;
144                 }
145     
146                 // Add the edge to the current partial path, and recur
147                 partialPath.add(outEdge);
148                 work(partialPath);
149                 partialPath.removeLast();
150     
151                 // Have we done the maximum amount of work?
152                 if (work == maxWork) {
153                     return;
154                 }
155                 ++work;
156     
157                 // Did we reach the maximum number of simple paths?
158                 if (pathList.size() == maxPaths) {
159                     return;
160                 }
161             }
162         }
163     }
