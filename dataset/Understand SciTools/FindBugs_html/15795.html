382         private static class RefComparisonTypeFrameModelingVisitor extends TypeFrameModelingVisitor {
383             private final RepositoryLookupFailureCallback lookupFailureCallback;
384     
385             private boolean sawStringIntern;
386     
387             public RefComparisonTypeFrameModelingVisitor(ConstantPoolGen cpg, TypeMerger typeMerger,
388                     RepositoryLookupFailureCallback lookupFailureCallback) {
389                 super(cpg, typeMerger);
390                 this.lookupFailureCallback = lookupFailureCallback;
391                 this.sawStringIntern = false;
392             }
393     
394             public boolean sawStringIntern() {
395                 return sawStringIntern;
396             }
397     
398             // Override handlers for bytecodes that may return String objects
399             // known to be dynamic or static.
400     
401             @Override
402             public void visitINVOKESTATIC(INVOKESTATIC obj) {
403                 if (returnsString(obj)) {
404                     consumeStack(obj);
405     
406                     String className = obj.getClassName(getCPG());
407                     if ("java.lang.String".equals(className)) {
408                         pushValue(dynamicStringTypeInstance);
409                     } else {
410                         pushReturnType(obj);
411                     }
412                 } else {
413                     super.visitINVOKESTATIC(obj);
414                 }
415             }
416     
417             @Override
418             public void visitINVOKESPECIAL(INVOKESPECIAL obj) {
419                 if (returnsString(obj)) {
420                     handleInstanceMethod(obj);
421                 } else {
422                     super.visitINVOKESPECIAL(obj);
423                 }
424             }
425     
426             @Override
427             public void visitINVOKEINTERFACE(INVOKEINTERFACE obj) {
428                 if (returnsString(obj)) {
429                     handleInstanceMethod(obj);
430                 } else {
431                     super.visitINVOKEINTERFACE(obj);
432                 }
433     
434             }
435     
436             @Override
437             public void visitINVOKEVIRTUAL(INVOKEVIRTUAL obj) {
438                 if (returnsString(obj)) {
439                     handleInstanceMethod(obj);
440                 } else {
441                     super.visitINVOKEVIRTUAL(obj);
442                 }
443             }
444     
445             private boolean returnsString(InvokeInstruction inv) {
446                 String methodSig = inv.getSignature(getCPG());
447                 return methodSig.endsWith(")Ljava/lang/String;");
448             }
449     
450             private void handleInstanceMethod(InvokeInstruction obj) {
451     
452                 assert returnsString(obj);
453                 consumeStack(obj);
454                 String className = obj.getClassName(getCPG());
455                 String methodName = obj.getName(getCPG());
456                 // System.out.println(className + "." + methodName);
457     
458                 if ("intern".equals(methodName) && "java.lang.String".equals(className)) {
459                     sawStringIntern = true;
460                     pushValue(staticStringTypeInstance);
461                 } else if ("toString".equals(methodName) || "java.lang.String".equals(className)) {
462                     pushValue(dynamicStringTypeInstance);
463                     // System.out.println("  dynamic");
464                 } else {
465                     pushReturnType(obj);
466                 }
467     
468             }
469     
470             @Override
471             public void visitLDC(LDC obj) {
472                 Type type = obj.getType(getCPG());
473                 if (isString(type)) {
474                     Object value = obj.getValue(getCPG());
475                     if (value instanceof String && ((String)value).length() == 0) {
476                         pushValue( emptyStringTypeInstance);
477                     } else {
478                         pushValue( staticStringTypeInstance);
479                     }
480                 } else {
481                     pushValue(type);
482                 }
483             }
484     
485             @Override
486             public void visitLDC2_W(LDC2_W obj) {
487                 Type type = obj.getType(getCPG());
488                 pushValue(isString(type) ? staticStringTypeInstance : type);
489             }
490     
491             private boolean isString(Type type) {
492                 return STRING_SIGNATURE.equals(type.getSignature());
493             }
494     
495             @Override
496             public void visitGETSTATIC(GETSTATIC obj) {
497                 Type type = obj.getType(getCPG());
498                 XField xf = XFactory.createXField(obj, cpg);
499                 if (xf.isFinal()) {
500                     FieldSummary fieldSummary = AnalysisContext.currentAnalysisContext().getFieldSummary();
501                     Item summary = fieldSummary.getSummary(xf);
502                     if (summary.isNull()) {
503                         pushValue(TypeFrame.getNullType());
504                         return;
505                     }
506     
507                     String slashedClassName = ClassName.fromFieldSignature(type.getSignature());
508                     if (slashedClassName != null) {
509                         String dottedClassName = ClassName.toDottedClassName(slashedClassName);
510                         if (DEFAULT_SUSPICIOUS_SET.contains(dottedClassName)) {
511                             type = new FinalConstant(dottedClassName, xf);
512                             consumeStack(obj);
513                             pushValue(type);
514                             return;
515                         }
516                     }
517     
518                 }
519                 if (STRING_SIGNATURE.equals(type.getSignature())) {
520                     handleLoad(obj);
521                 } else {
522                     super.visitGETSTATIC(obj);
523                 }
524             }
525     
526             @Override
527             public void visitGETFIELD(GETFIELD obj) {
528                 Type type = obj.getType(getCPG());
529                 if (STRING_SIGNATURE.equals(type.getSignature())) {
530                     handleLoad(obj);
531                 } else {
532                     XField xf = XFactory.createXField(obj, cpg);
533                     if (xf.isFinal()) {
534                         FieldSummary fieldSummary = AnalysisContext.currentAnalysisContext().getFieldSummary();
535                         Item summary = fieldSummary.getSummary(xf);
536                         if (summary.isNull()) {
537                             consumeStack(obj);
538                             pushValue(TypeFrame.getNullType());
539                             return;
540                         }
541     
542                         String slashedClassName = ClassName.fromFieldSignature(type.getSignature());
543                         if (slashedClassName != null) {
544                             String dottedClassName = ClassName.toDottedClassName(slashedClassName);
545                             if (DEFAULT_SUSPICIOUS_SET.contains(dottedClassName)) {
546                                 type = new FinalConstant(dottedClassName, xf);
547                                 consumeStack(obj);
548                                 pushValue(type);
549                                 return;
550                             }
551                         }
552                     }
553                     super.visitGETFIELD(obj);
554                 }
555             }
556     
557             private void handleLoad(FieldInstruction obj) {
558                 consumeStack(obj);
559     
560                 Type type = obj.getType(getCPG());
561                 if (!STRING_SIGNATURE.equals(type.getSignature())) {
562                     throw new IllegalArgumentException("type is not String: " + type);
563                 }
564                 try {
565                     String className = obj.getClassName(getCPG());
566                     String fieldName = obj.getName(getCPG());
567                     Field field = Hierarchy.findField(className, fieldName);
568     
569                     if (field != null) {
570                         // If the field is final, we'll assume that the String value
571                         // is static.
572                         if (field.isFinal() && field.isFinal()) {
573                             pushValue(staticStringTypeInstance);
574                         } else {
575                             pushValue(type);
576                         }
577     
578                         return;
579                     }
580                 } catch (ClassNotFoundException ex) {
581                     lookupFailureCallback.reportMissingClass(ex);
582                 }
583     
584                 pushValue(type);
585             }
586         }

588         /**
589          * Type merger to use the extended String types.
590          */
591         private static class RefComparisonTypeMerger extends StandardTypeMerger {
592             public RefComparisonTypeMerger(RepositoryLookupFailureCallback lookupFailureCallback,
593                     ExceptionSetFactory exceptionSetFactory) {
594                 super(lookupFailureCallback, exceptionSetFactory);
595             }
596     
597             @Override
598             protected boolean isReferenceType(byte type) {
599                 return super.isReferenceType(type) || type == T_STATIC_STRING || type == T_DYNAMIC_STRING;
600             }
601     
602             @Override
603             protected ReferenceType mergeReferenceTypes(ReferenceType aRef, ReferenceType bRef) throws DataflowAnalysisException {
604                 byte aType = aRef.getType();
605                 byte bType = bRef.getType();
606     
607                 if (isExtendedStringType(aType) || isExtendedStringType(bType)) {
608                     // If both types are the same extended String type,
609                     // then the same type is returned. Otherwise, extended
610                     // types are downgraded to plain java.lang.String,
611                     // and a standard merge is applied.
612                     if (aType == bType) {
613                         return aRef;
614                     }
615     
616                     if (isExtendedStringType(aType)) {
617                         aRef = Type.STRING;
618                     }
619                     if (isExtendedStringType(bType)) {
620                         bRef = Type.STRING;
621                     }
622                 }
623     
624                 return super.mergeReferenceTypes(aRef, bRef);
625             }
626     
627             private boolean isExtendedStringType(byte type) {
628                 return type == T_DYNAMIC_STRING || type == T_STATIC_STRING || type == T_PARAMETER_STRING;
629             }
630         }

654         public FindRefComparison(BugReporter bugReporter) {
655             this.bugReporter = bugReporter;
656             this.bugAccumulator = new BugAccumulator(bugReporter);
657             this.suspiciousSet = new HashSet<String>(DEFAULT_SUSPICIOUS_SET);
658     
659             // Check frc.suspicious system property for additional suspicious types
660             // to check
661             String extraSuspiciousTypes = SystemProperties.getProperty("frc.suspicious");
662             if (extraSuspiciousTypes != null) {
663                 StringTokenizer tok = new StringTokenizer(extraSuspiciousTypes, ",");
664                 while (tok.hasMoreTokens()) {
665                     suspiciousSet.add(tok.nextToken());
666                 }
667             }
668             testingEnabled = SystemProperties.getBoolean("report_TESTING_pattern_in_standard_detectors");
669         }
670     
671         @Override
672         public void visitClassContext(ClassContext classContext) {
673             this.classContext = classContext;
674     
675             JavaClass jclass = classContext.getJavaClass();
676             Method[] methodList = jclass.getMethods();
677     
678             for (Method method : methodList) {
679                 MethodGen methodGen = classContext.getMethodGen(method);
680                 if (methodGen == null) {
681                     continue;
682                 }
683     
684                 // Prescreening - must have IF_ACMPEQ, IF_ACMPNE,
685                 // or an invocation of an instance method
686                 BitSet bytecodeSet = classContext.getBytecodeSet(method);
687                 if (bytecodeSet == null || !bytecodeSet.intersects(prescreenSet)) {
688                     continue;
689                 }
690     
691                 if (DEBUG) {
692                     System.out.println("FindRefComparison: analyzing " + SignatureConverter.convertMethodSignature(methodGen));
693                 }
694     
695                 try {
696                     analyzeMethod(classContext, method);
697                 } catch (CFGBuilderException e) {
698                     bugReporter.logError("Error analyzing " + method.toString(), e);
699                 } catch (DataflowAnalysisException e) {
700                     // bugReporter.logError("Error analyzing " + method.toString(),
701                     // e);
702                 }
703                 bugAccumulator.reportAccumulatedBugs();
704             }

733         private void analyzeMethod(ClassContext classContext, final Method method) throws CFGBuilderException,
734         DataflowAnalysisException {
735     
736             MethodGen methodGen = classContext.getMethodGen(method);
737             if (methodGen == null) {
738                 return;
739             }
740     
741             JavaClass jclass = classContext.getJavaClass();
742             ConstantPoolGen cpg = classContext.getConstantPoolGen();
743     
744             // Enqueue all of the potential violations we find in the method.
745             // Normally we'll only report the first highest-priority warning,
746             // but if in relaxed mode or if REPORT_ALL_REF_COMPARISONS is set,
747             // then we'll report everything.
748             LinkedList<WarningWithProperties> refComparisonList = new LinkedList<WarningWithProperties>();
749             LinkedList<WarningWithProperties> stringComparisonList = new LinkedList<WarningWithProperties>();
750     
751     
752             comparedForEqualityInThisMethod = new HashMap<String,Integer>();
753             CFG cfg = classContext.getCFG(method);
754             DepthFirstSearch dfs = classContext.getDepthFirstSearch(method);
755             ExceptionSetFactory exceptionSetFactory = classContext.getExceptionSetFactory(method);
756     
757             // Perform type analysis using our special type merger
758             // (which handles String types specially, keeping track of
759             // which ones appear to be dynamically created)
760             RefComparisonTypeMerger typeMerger = new RefComparisonTypeMerger(bugReporter, exceptionSetFactory);
761             RefComparisonTypeFrameModelingVisitor visitor = new RefComparisonTypeFrameModelingVisitor(methodGen.getConstantPool(),
762                     typeMerger, bugReporter);
763             TypeAnalysis typeAnalysis = new SpecialTypeAnalysis(method, methodGen, cfg, dfs, typeMerger, visitor, bugReporter,
764                     exceptionSetFactory);
765             TypeDataflow typeDataflow = new TypeDataflow(cfg, typeAnalysis);
766             Profiler profiler = Global.getAnalysisCache().getProfiler();
767             profiler.start(SpecialTypeAnalysis.class);
768             try {
769                 typeDataflow.execute();
770             } finally {
771                 profiler.end(SpecialTypeAnalysis.class);
772             }
773     
774             // Inspect Locations in the method for suspicious ref comparisons and
775             // calls to equals()
776             for (Iterator<Location> i = cfg.locationIterator(); i.hasNext();) {
777                 Location location = i.next();
778     
779                 inspectLocation(jclass, cpg, method, methodGen, refComparisonList, stringComparisonList, visitor, typeDataflow,
780                         location);
781             }
782     
783             if (stringComparisonList.isEmpty() && refComparisonList.isEmpty()) {
784                 return;
785             }
786             // Add method-wide properties to BugInstances
787             final boolean likelyTestcase = TestCaseDetector.likelyTestCase(XFactory.createXMethod(jclass, method));
788     
789             decorateWarnings(stringComparisonList, new WarningDecorator() {
790                 @Override
791                 public void decorate(WarningWithProperties warn) {
792                     if (mightBeLaterCheckedUsingEquals(warn)) {
793                         warn.propertySet.addProperty(RefComparisonWarningProperty.SAW_CALL_TO_EQUALS);
794                     }
795     
796                     if (likelyTestcase) {
797                         warn.propertySet.addProperty(RefComparisonWarningProperty.COMPARE_IN_TEST_CASE);
798                     }
799                     /*
800                     if (false && !(method.isPublic() || method.isProtected())) {
801                         warn.propertySet.addProperty(RefComparisonWarningProperty.PRIVATE_METHOD);
802                     }
803                      */
804                 }
805             });
806             decorateWarnings(refComparisonList, new WarningDecorator() {
807                 @Override
808                 public void decorate(WarningWithProperties warn) {
809                     if (likelyTestcase) {
810                         warn.propertySet.addProperty(RefComparisonWarningProperty.COMPARE_IN_TEST_CASE);
811                     }
812     
813                     if (mightBeLaterCheckedUsingEquals(warn)) {
814                         warn.propertySet.addProperty(RefComparisonWarningProperty.SAW_CALL_TO_EQUALS);
815                     }
816                 }
817             });
818     
819             // Report violations
820             boolean relaxed = FindBugsAnalysisFeatures.isRelaxedMode();
821             reportBest(classContext, method, stringComparisonList, relaxed);
822             reportBest(classContext, method, refComparisonList, relaxed);
823         }
824     
825         boolean mightBeLaterCheckedUsingEquals(WarningWithProperties warning) {
826             for (BugAnnotation a : warning.instance.getAnnotations()) {
827                 if (a instanceof TypeAnnotation) {
828                     String signature = ((TypeAnnotation) a).getTypeDescriptor();
829                     Integer pc = comparedForEqualityInThisMethod.get(signature);
830                     if (pc != null && pc > warning.location.getHandle().getPosition()) {
831                         return true;
832                     }
833                 }
834             }
835             return false;
836         }

838         private void inspectLocation(JavaClass jclass, ConstantPoolGen cpg, Method method, MethodGen methodGen,
839                 LinkedList<WarningWithProperties> refComparisonList, LinkedList<WarningWithProperties> stringComparisonList,
840                 RefComparisonTypeFrameModelingVisitor visitor, TypeDataflow typeDataflow, Location location)
841                         throws DataflowAnalysisException {
842             Instruction ins = location.getHandle().getInstruction();
843             short opcode = ins.getOpcode();
844             if (opcode == Constants.IF_ACMPEQ || opcode == Constants.IF_ACMPNE) {
845                 checkRefComparison(location, jclass, method, methodGen, visitor, typeDataflow, stringComparisonList,
846                         refComparisonList);
847             } else if (ins instanceof InvokeInstruction) {
848                 InvokeInstruction inv = (InvokeInstruction) ins;
849                 boolean isStatic = inv instanceof INVOKESTATIC;
850                 @DottedClassName String className = inv.getClassName(cpg);
851                 String methodName = inv.getMethodName(cpg);
852                 String methodSig = inv.getSignature(cpg);
853                 if ( "assertSame".equals(methodName) && "(Ljava/lang/Object;Ljava/lang/Object;)V".equals(methodSig)) {
854                     checkRefComparison(location, jclass, method, methodGen, visitor, typeDataflow, stringComparisonList,
855                             refComparisonList);
856                 } else if ( "assertFalse".equals(methodName) && "(Z)V".equals(methodSig)) {
857                     SourceLineAnnotation lastLocation = bugAccumulator.getLastBugLocation();
858                     InstructionHandle prevHandle = location.getHandle().getPrev();
859                     if (lastLocation != null && prevHandle != null && lastLocation.getEndBytecode() == prevHandle.getPosition()){
860                         bugAccumulator.forgetLastBug();
861                         if (DEBUG) {
862                             System.out.println("Forgetting last bug due to call to " + className +"." + methodName);
863                         }
864                     }
865     
866                 } else {
867                     boolean equalsMethod = !isStatic && "equals".equals(methodName) && "(Ljava/lang/Object;)Z".equals(methodSig)
868                             || isStatic &&  "assertEquals".equals(methodName)
869                             && "(Ljava/lang/Object;Ljava/lang/Object;)V".equals(methodSig)
870                             || isStatic && "equal".equals(methodName) && "(Ljava/lang/Object;Ljava/lang/Object;)Z".equals(methodSig)
871                             && "com.google.common.base.Objects".equals(className)
872                             || isStatic && "equals".equals(methodName) && "(Ljava/lang/Object;Ljava/lang/Object;)Z".equals(methodSig)
873                             && "java.util.Objects".equals(className);
874     
875                     if (equalsMethod) {
876                         checkEqualsComparison(location, jclass, method, methodGen, cpg, typeDataflow);
877                     }
878                 }
879             }
880     
881         }

883         private void decorateWarnings(LinkedList<WarningWithProperties> stringComparisonList, WarningDecorator warningDecorator) {
884             for (WarningWithProperties warn : stringComparisonList) {
885                 warningDecorator.decorate(warn);
886                 warn.propertySet.decorateBugInstance(warn.instance);
887             }

890         private void reportBest(ClassContext classContext, Method method, LinkedList<WarningWithProperties> warningList,
891                 boolean relaxed) {
892             boolean reportAll = relaxed || REPORT_ALL_REF_COMPARISONS;
893     
894             int bestPriority = Integer.MAX_VALUE;
895             for (WarningWithProperties warn : warningList) {
896                 int priority = warn.instance.getPriority();
897                 if (bestPriority > priority) {
898                     bestPriority = priority;
899                 }
900     
901                 if (reportAll) {
902                     if (relaxed) {
903                         // Add general warning properties
904                         WarningPropertyUtil.addPropertiesForDataMining(warn.propertySet, classContext, method, warn.location);
905     
906                         // Convert warning properties to bug properties
907                         warn.propertySet.decorateBugInstance(warn.instance);
908                     }
909                     bugAccumulator.accumulateBug(warn.instance, warn.sourceLine);
910                 }
911     
912             }
913             if (!reportAll) {
914                 for (WarningWithProperties warn : warningList) {
915                     int priority = warn.instance.getPriority();
916                     if (priority <= bestPriority) {
917                         bugAccumulator.accumulateBug(warn.instance, warn.sourceLine);
918                     }
919                 }
920             }
921         }

923         private void checkRefComparison(Location location, JavaClass jclass, Method method, MethodGen methodGen,
924                 RefComparisonTypeFrameModelingVisitor visitor, TypeDataflow typeDataflow,
925                 List<WarningWithProperties> stringComparisonList, List<WarningWithProperties> refComparisonList)
926                         throws DataflowAnalysisException {
927     
928             InstructionHandle handle = location.getHandle();
929     
930             TypeFrame frame = typeDataflow.getFactAtLocation(location);
931             if (frame.getStackDepth() < 2) {
932                 throw new DataflowAnalysisException("Stack underflow", methodGen, handle);
933             }
934     
935             int numSlots = frame.getNumSlots();
936             Type lhsType = frame.getValue(numSlots - 2);
937             Type rhsType = frame.getValue(numSlots - 1);
938     
939             if (lhsType instanceof NullType || rhsType instanceof NullType) {
940                 return;
941             }
942             if (lhsType instanceof ReferenceType && rhsType instanceof ReferenceType) {
943                 IncompatibleTypes result = IncompatibleTypes.getPriorityForAssumingCompatible(lhsType, rhsType, true);
944                 if (result != IncompatibleTypes.SEEMS_OK && result != IncompatibleTypes.UNCHECKED) {
945                     String sourceFile = jclass.getSourceFileName();
946     
947                     boolean isAssertSame = handle.getInstruction() instanceof INVOKESTATIC;
948                     if (isAssertSame) {
949                         if(testingEnabled) {
950                             bugAccumulator.accumulateBug(
951                                     new BugInstance(this, "TESTING", result.getPriority())
952                                     .addClassAndMethod(methodGen, sourceFile)
953                                     .addString("Calling assertSame with two distinct objects")
954                                     .addFoundAndExpectedType(rhsType, lhsType)
955                                     .addSomeSourceForTopTwoStackValues(classContext, method, location),
956                                     SourceLineAnnotation.fromVisitedInstruction(classContext, methodGen, sourceFile, handle));
957                         }
958                     } else {
959                         bugAccumulator.accumulateBug(
960                                 new BugInstance(this, "EC_UNRELATED_TYPES_USING_POINTER_EQUALITY", result.getPriority())
961                                 .addClassAndMethod(methodGen, sourceFile).addFoundAndExpectedType(rhsType, lhsType)
962                                 .addSomeSourceForTopTwoStackValues(classContext, method, location),
963                                 SourceLineAnnotation.fromVisitedInstruction(classContext, methodGen, sourceFile, handle));
964                     }
965                     return;
966                 }
967                 if (lhsType.equals(Type.OBJECT) && rhsType.equals(Type.OBJECT)) {
968                     return;
969                 }
970                 String lhs = SignatureConverter.convert(lhsType.getSignature());
971                 String rhs = SignatureConverter.convert(rhsType.getSignature());
972     
973                 if ("java.lang.String".equals(lhs) || "java.lang.String".equals(rhs)) {
974                     handleStringComparison(jclass, method, methodGen, visitor, stringComparisonList, location, lhsType, rhsType);
975                 } else if (suspiciousSet.contains(lhs)) {
976                     handleSuspiciousRefComparison(jclass, method, methodGen, refComparisonList, location, lhs,
977                             (ReferenceType) lhsType, (ReferenceType) rhsType);
978                 } else if (suspiciousSet.contains(rhs)) {
979                     handleSuspiciousRefComparison(jclass, method, methodGen, refComparisonList, location, rhs,
980                             (ReferenceType) lhsType, (ReferenceType) rhsType);
981                 }
982             }
983         }

985         private void handleStringComparison(JavaClass jclass, Method method, MethodGen methodGen,
986                 RefComparisonTypeFrameModelingVisitor visitor, List<WarningWithProperties> stringComparisonList, Location location,
987                 Type lhsType, Type rhsType) {
988             if (DEBUG) {
989                 System.out.println("String/String comparison at " + location.getHandle());
990             }
991     
992             // Compute the priority:
993             // - two static strings => do not report
994             // - dynamic string and anything => high
995             // - static string and unknown => medium
996             // - all other cases => low
997             // System.out.println("Compare " + lhsType + " == " + rhsType);
998             byte type1 = lhsType.getType();
999             byte type2 = rhsType.getType();
1000     
1001             String bugPattern = "ES_COMPARING_STRINGS_WITH_EQ";
1002             // T1 T2 result
1003             // S S no-op
1004             // D ? high
1005             // ? D high
1006             // S ? normal
1007             // ? S normal
1008             WarningPropertySet<WarningProperty> propertySet = new WarningPropertySet<WarningProperty>();
1009             if (type1 == T_STATIC_STRING && type2 == T_STATIC_STRING) {
1010                 propertySet.addProperty(RefComparisonWarningProperty.COMPARE_STATIC_STRINGS);
1011             } else if (type1 == T_DYNAMIC_STRING || type2 == T_DYNAMIC_STRING) {
1012                 propertySet.addProperty(RefComparisonWarningProperty.DYNAMIC_AND_UNKNOWN);
1013             } else if (type2 == T_PARAMETER_STRING || type1 == T_PARAMETER_STRING) {
1014                 bugPattern = "ES_COMPARING_PARAMETER_STRING_WITH_EQ";
1015                 if (methodGen.isPublic() || methodGen.isProtected()) {
1016                     propertySet.addProperty(RefComparisonWarningProperty.STRING_PARAMETER_IN_PUBLIC_METHOD);
1017                 } else {
1018                     propertySet.addProperty(RefComparisonWarningProperty.STRING_PARAMETER);
1019                 }
1020             } else if (type1 == T_STATIC_STRING || type2 == T_STATIC_STRING) {
1021                 if (lhsType instanceof EmptyStringType || rhsType instanceof EmptyStringType) {
1022                     propertySet.addProperty(RefComparisonWarningProperty.EMPTY_AND_UNKNOWN);
1023                 } else {
1024                     propertySet.addProperty(RefComparisonWarningProperty.STATIC_AND_UNKNOWN);
1025                 }
1026             } else if (visitor.sawStringIntern()) {
1027                 propertySet.addProperty(RefComparisonWarningProperty.SAW_INTERN);
1028             }
1029     
1030             String sourceFile = jclass.getSourceFileName();
1031             BugInstance instance = new BugInstance(this, bugPattern, BASE_ES_PRIORITY).addClassAndMethod(methodGen, sourceFile)
1032                     .addType("Ljava/lang/String;").describe(TypeAnnotation.FOUND_ROLE).addSomeSourceForTopTwoStackValues(classContext, method, location);
1033             SourceLineAnnotation sourceLineAnnotation = SourceLineAnnotation.fromVisitedInstruction(classContext, methodGen,
1034                     sourceFile, location.getHandle());
1035     
1036             WarningWithProperties warn = new WarningWithProperties(instance, propertySet, sourceLineAnnotation, location);
1037             stringComparisonList.add(warn);
1038         }

1040         private void handleSuspiciousRefComparison(JavaClass jclass, Method method, MethodGen methodGen,
1041                 List<WarningWithProperties> refComparisonList, Location location, String lhs, ReferenceType lhsType,
1042                 ReferenceType rhsType) {
1043             XField xf = null;
1044             if (lhsType instanceof FinalConstant) {
1045                 xf = ((FinalConstant) lhsType).getXField();
1046             } else if (rhsType instanceof FinalConstant) {
1047                 xf = ((FinalConstant) rhsType).getXField();
1048             }
1049             String sourceFile = jclass.getSourceFileName();
1050             String bugPattern = "RC_REF_COMPARISON";
1051             int priority = Priorities.HIGH_PRIORITY;
1052             if ("java.lang.Boolean".equals(lhs)) {
1053                 bugPattern = "RC_REF_COMPARISON_BAD_PRACTICE_BOOLEAN";
1054                 priority = Priorities.NORMAL_PRIORITY;
1055             } else if (xf != null && xf.isStatic() && xf.isFinal()) {
1056                 bugPattern = "RC_REF_COMPARISON_BAD_PRACTICE";
1057                 if (xf.isPublic() || !methodGen.isPublic()) {
1058                     priority = Priorities.NORMAL_PRIORITY;
1059                 }
1060             }
1061             BugInstance instance = new BugInstance(this, bugPattern, priority).addClassAndMethod(methodGen, sourceFile)
1062                     .addType("L" + lhs.replace('.', '/') + ";").describe(TypeAnnotation.FOUND_ROLE);
1063             if (xf != null) {
1064                 instance.addField(xf).describe(FieldAnnotation.LOADED_FROM_ROLE);
1065             } else {
1066                 instance.addSomeSourceForTopTwoStackValues(classContext, method, location);
1067             }
1068             SourceLineAnnotation sourceLineAnnotation = SourceLineAnnotation.fromVisitedInstruction(classContext, methodGen,
1069                     sourceFile, location.getHandle());
1070     
1071             refComparisonList.add(new WarningWithProperties(instance, new WarningPropertySet<WarningProperty>(),
1072                     sourceLineAnnotation, location));
1073         }

1075         private Map<String, Integer> comparedForEqualityInThisMethod;
1076     
1077         void addEqualsCheck(String type, int pc) {
1078             Integer oldPC = comparedForEqualityInThisMethod.get(type);
1079             if (oldPC == null || pc < oldPC) {
1080                 comparedForEqualityInThisMethod.put(type, pc);
1081             }
1082         }

1084         private void checkEqualsComparison(Location location, JavaClass jclass, Method method, MethodGen methodGen,
1085                 ConstantPoolGen cpg, TypeDataflow typeDataflow) throws DataflowAnalysisException {
1086     
1087             InstructionHandle handle = location.getHandle();
1088             InstructionHandle next = handle.getNext();
1089             if (next != null && next.getInstruction() instanceof INVOKESTATIC) {
1090                 INVOKESTATIC is = (INVOKESTATIC) next.getInstruction();
1091                 if ("assertFalse".equals(is.getMethodName(cpg))) {
1092                     return;
1093                 }
1094             }
1095             String sourceFile = jclass.getSourceFileName();
1096     
1097             TypeFrame frame = typeDataflow.getFactAtLocation(location);
1098             if (frame.getStackDepth() < 2) {
1099                 throw new DataflowAnalysisException("Stack underflow", methodGen, handle);
1100             }
1101     
1102             int numSlots = frame.getNumSlots();
1103             Type lhsType_ = frame.getValue(numSlots - 2);
1104             Type rhsType_ = frame.getValue(numSlots - 1);
1105     
1106             // Ignore top and bottom values
1107             if (lhsType_.getType() == T_TOP || lhsType_.getType() == T_BOTTOM || rhsType_.getType() == T_TOP
1108                     || rhsType_.getType() == T_BOTTOM) {
1109                 return;
1110             }
1111             InvokeInstruction inv = (InvokeInstruction) handle.getInstruction();
1112             MethodAnnotation calledMethodAnnotation = getMethodCalledAnnotation(cpg, inv);
1113             boolean looksLikeTestCase = TestCaseDetector.likelyTestCase(XFactory.createXMethod(methodGen));
1114             int priorityModifier = 0;
1115             if (looksLikeTestCase) {
1116                 priorityModifier = 1;
1117             }
1118     
1119             if (rhsType_.getType() == T_NULL) {
1120                 // A literal null value was passed directly to equals().
1121                 if (!looksLikeTestCase) {
1122     
1123                     try {
1124                         IsNullValueDataflow isNullDataflow = classContext.getIsNullValueDataflow(method);
1125                         IsNullValueFrame isNullFrame = isNullDataflow.getFactAtLocation(location);
1126                         BugAnnotation a = BugInstance.getSourceForTopStackValue(classContext, method, location);
1127                         int priority = NORMAL_PRIORITY;
1128                         if (a instanceof FieldAnnotation && ((FieldAnnotation) a).isStatic()) {
1129                             priority = LOW_PRIORITY;
1130                         }
1131                         if (isNullFrame.isValid() && isNullFrame.getTopValue().isDefinitelyNull()) {
1132                             String type = "EC_NULL_ARG";
1133                             if (calledMethodAnnotation != null && calledMethodAnnotation.isStatic()){
1134                                 type = "DMI_DOH";
1135                                 priority = LOW_PRIORITY;
1136                             }
1137                             BugInstance bug = new BugInstance(this, type, priority + priorityModifier).addClassAndMethod(methodGen, sourceFile)
1138                                     .addOptionalAnnotation(calledMethodAnnotation);
1139                             if ("DMI_DOH".equals(type)) {
1140                                 bug.addString("Use \"== null\" to check for a value being null");
1141                             }
1142                             bugAccumulator.accumulateBug(
1143                                     bug,
1144                                     SourceLineAnnotation.fromVisitedInstruction(this.classContext, methodGen, sourceFile,
1145                                             location.getHandle()));
1146                         }
1147                     } catch (CFGBuilderException e) {
1148                         AnalysisContext.logError("Error getting null value analysis", e);
1149                     }
1150     
1151                 }
1152                 return;
1153             } else if (lhsType_.getType() == T_NULL) {
1154                 // Hmm...in this case, equals() is being invoked on
1155                 // a literal null value. This is really the
1156                 // purview of FindNullDeref. So, we'll just do nothing.
1157                 return;
1158             } else if (!(lhsType_ instanceof ReferenceType) || !(rhsType_ instanceof ReferenceType)) {
1159                 bugReporter.logError("equals() used to compare non-object type(s) " + lhsType_ + " and " + rhsType_ + " in "
1160                         + SignatureConverter.convertMethodSignature(methodGen) + " at " + location.getHandle());
1161                 return;
1162             }
1163             IncompatibleTypes result = IncompatibleTypes.getPriorityForAssumingCompatible(lhsType_, rhsType_);
1164     
1165             if (lhsType_ instanceof ArrayType && rhsType_ instanceof ArrayType) {
1166                 String pattern = "EC_BAD_ARRAY_COMPARE";
1167                 IncompatibleTypes result2 = IncompatibleTypes.getPriorityForAssumingCompatible(lhsType_, rhsType_, true);
1168                 if (result2.getPriority() <= Priorities.NORMAL_PRIORITY) {
1169                     pattern = "EC_INCOMPATIBLE_ARRAY_COMPARE";
1170                 } else if (calledMethodAnnotation != null && "org.testng.Assert".equals(calledMethodAnnotation.getClassName())) {
1171                     return;
1172                 }
1173                 bugAccumulator.accumulateBug(new BugInstance(this, pattern, NORMAL_PRIORITY).addClassAndMethod(methodGen, sourceFile)
1174                         .addFoundAndExpectedType(rhsType_, lhsType_)
1175                         .addSomeSourceForTopTwoStackValues(classContext, method, location)
1176                         .addOptionalAnnotation(calledMethodAnnotation, MethodAnnotation.METHOD_CALLED),
1177                         SourceLineAnnotation.fromVisitedInstruction(this.classContext, methodGen, sourceFile, location.getHandle()));
1178                 return;
1179             }
1180     
1181             if (result.getPriority() >= Priorities.LOW_PRIORITY) {
1182                 addEqualsCheck(lhsType_.getSignature(), handle.getPosition());
1183                 addEqualsCheck(rhsType_.getSignature(), handle.getPosition());
1184             }
1185     
1186             if (result == IncompatibleTypes.SEEMS_OK) {
1187                 return;
1188             }
1189     
1190     
1191             if (result.getPriority() > Priorities.LOW_PRIORITY) {
1192                 return;
1193             }
1194     
1195             if (result == IncompatibleTypes.ARRAY_AND_NON_ARRAY || result == IncompatibleTypes.ARRAY_AND_OBJECT) {
1196                 String lhsSig = lhsType_.getSignature();
1197                 String rhsSig = rhsType_.getSignature();
1198                 boolean allOk = checkForWeirdEquals(lhsSig, rhsSig, new HashSet<XMethod>());
1199                 if (allOk) {
1200                     priorityModifier += 2;
1201                 }
1202                 bugAccumulator.accumulateBug(new BugInstance(this, "EC_ARRAY_AND_NONARRAY", result.getPriority() + priorityModifier)
1203                 .addClassAndMethod(methodGen, sourceFile).addFoundAndExpectedType(rhsType_, lhsType_)
1204                 .addSomeSourceForTopTwoStackValues(classContext, method, location)
1205                 .addOptionalAnnotation(calledMethodAnnotation, MethodAnnotation.METHOD_CALLED),
1206                 SourceLineAnnotation.fromVisitedInstruction(this.classContext, methodGen, sourceFile, location.getHandle()));
1207             } else if (result == IncompatibleTypes.INCOMPATIBLE_CLASSES) {
1208                 String lhsSig = lhsType_.getSignature();
1209                 String rhsSig = rhsType_.getSignature();
1210                 boolean core = lhsSig.startsWith("Ljava") && rhsSig.startsWith("Ljava");
1211                 if (core) {
1212                     looksLikeTestCase = false;
1213                     priorityModifier = 0;
1214                 }
1215                 if (true) {
1216                     Set<XMethod> targets = new HashSet<XMethod>();
1217                     boolean allOk = checkForWeirdEquals(lhsSig, rhsSig, targets);
1218                     if (allOk) {
1219                         priorityModifier += 2;
1220                     }
1221     
1222                     int priority = result.getPriority() + priorityModifier;
1223                     bugAccumulator.accumulateBug(
1224                             new BugInstance(this, "EC_UNRELATED_TYPES", priority)
1225                             .addClassAndMethod(methodGen, sourceFile).addFoundAndExpectedType(rhsType_, lhsType_)
1226                             .addSomeSourceForTopTwoStackValues(classContext, method, location).addEqualsMethodUsed(targets)
1227                             .addOptionalAnnotation(calledMethodAnnotation, MethodAnnotation.METHOD_CALLED),
1228                             SourceLineAnnotation.fromVisitedInstruction(this.classContext, methodGen, sourceFile,
1229                                     location.getHandle()));
1230                 }
1231             } else if (result == IncompatibleTypes.UNRELATED_CLASS_AND_INTERFACE
1232                     || result == IncompatibleTypes.UNRELATED_FINAL_CLASS_AND_INTERFACE) {
1233                 bugAccumulator.accumulateBug(
1234                         new BugInstance(this, "EC_UNRELATED_CLASS_AND_INTERFACE", result.getPriority() + priorityModifier)
1235                         .addClassAndMethod(methodGen, sourceFile).addFoundAndExpectedType(rhsType_, lhsType_)
1236                         .addSomeSourceForTopTwoStackValues(classContext, method, location)
1237                         .addEqualsMethodUsed(DescriptorFactory.createClassDescriptorFromSignature(lhsType_.getSignature()))
1238                         .addOptionalAnnotation(calledMethodAnnotation, MethodAnnotation.METHOD_CALLED),
1239                         SourceLineAnnotation.fromVisitedInstruction(this.classContext, methodGen, sourceFile, location.getHandle()));
1240             } else if (result == IncompatibleTypes.UNRELATED_INTERFACES) {
1241                 bugAccumulator.accumulateBug(
1242                         new BugInstance(this, "EC_UNRELATED_INTERFACES", result.getPriority() + priorityModifier)
1243                         .addClassAndMethod(methodGen, sourceFile).addFoundAndExpectedType(rhsType_, lhsType_)
1244                         .addSomeSourceForTopTwoStackValues(classContext, method, location)
1245                         .addEqualsMethodUsed(DescriptorFactory.createClassDescriptorFromSignature(lhsType_.getSignature()))
1246                         .addOptionalAnnotation(calledMethodAnnotation, MethodAnnotation.METHOD_CALLED),
1247                         SourceLineAnnotation.fromVisitedInstruction(this.classContext, methodGen, sourceFile, location.getHandle()));
1248             } else if (result != IncompatibleTypes.UNCHECKED && result.getPriority() <= Priorities.LOW_PRIORITY) {
1249                 bugAccumulator.accumulateBug(new BugInstance(this, "EC_UNRELATED_TYPES", result.getPriority() + priorityModifier)
1250                 .addClassAndMethod(methodGen, sourceFile).addFoundAndExpectedType(rhsType_, lhsType_)
1251                 .addSomeSourceForTopTwoStackValues(classContext, method, location)
1252                 .addOptionalAnnotation(calledMethodAnnotation, MethodAnnotation.METHOD_CALLED),
1253                 SourceLineAnnotation.fromVisitedInstruction(this.classContext, methodGen, sourceFile, location.getHandle()));
1254             }

1274         private boolean checkForWeirdEquals(String lhsSig, String rhsSig, Set<XMethod> targets) {
1275             boolean allOk = false;
1276             try {
1277                 ClassSummary classSummary = AnalysisContext.currentAnalysisContext().getClassSummary();
1278     
1279                 ClassDescriptor expectedClassDescriptor = DescriptorFactory.createClassDescriptorFromSignature(lhsSig);
1280                 ClassDescriptor actualClassDescriptor = DescriptorFactory.createClassDescriptorFromSignature(rhsSig);
1281     
1282                 targets.addAll(Hierarchy2.resolveVirtualMethodCallTargets(expectedClassDescriptor, "equals", "(Ljava/lang/Object;)Z",
1283                         false, false));
1284                 allOk = targets.size() > 0;
1285                 for (XMethod m2 : targets) {
1286                     if (!classSummary.mightBeEqualTo(m2.getClassDescriptor(), actualClassDescriptor)) {
1287                         allOk = false;
1288                     }
1289                 }
1290     
1291             } catch (ClassNotFoundException e) {
1292                 AnalysisContext.reportMissingClass(e);
1293             }
1294             return allOk;
1295         }
