123         public static class LongRangeSet implements Iterable<LongRangeSet> {
124             private final SortedMap<Long, Long> map = new TreeMap<>();
125             private final TypeLongRange range;
126     
127             public LongRangeSet(String type) {
128                 TypeLongRange range = typeRanges.get(type);
129                 if(range == null) {
130                     throw new IllegalArgumentException("Type is not supported: " + type);
131                 }
132                 map.put(range.min, range.max);
133                 this.range = range;
134             }
135     
136             private LongRangeSet(TypeLongRange range, long from, long to) {
137                 this.range = range;
138                 if(from < range.min) {
139                     from = range.min;
140                 }
141                 if(to > range.max) {
142                     to = range.max;
143                 }
144                 if (from <= to) {
145                     map.put(from, to);
146                 }
147             }
148     
149             private LongRangeSet(TypeLongRange range) {
150                 this.range = range;
151             }
152     
153             public LongRangeSet gt(long value) {
154                 splitGreater(value);
155                 if(value == Long.MAX_VALUE) {
156                     return new LongRangeSet(range);
157                 }
158                 return new LongRangeSet(range, value + 1, range.max);
159             }
160     
161             public LongRangeSet ge(long value) {
162                 splitGreater(value-1);
163                 return new LongRangeSet(range, value, range.max);
164             }
165     
166             public LongRangeSet lt(long value) {
167                 splitGreater(value-1);
168                 if(value == Long.MIN_VALUE) {
169                     return new LongRangeSet(range);
170                 }
171                 return new LongRangeSet(range, range.min, value - 1);
172             }
173     
174             public LongRangeSet le(long value) {
175                 splitGreater(value);
176                 return new LongRangeSet(range, range.min, value);
177             }
178     
179             public LongRangeSet eq(long value) {
180                 splitGreater(value);
181                 splitGreater(value-1);
182                 return new LongRangeSet(range, value, value);
183             }
184     
185             public LongRangeSet ne(long value) {
186                 splitGreater(value);
187                 splitGreater(value-1);
188                 LongRangeSet rangeSet = lt(value);
189                 if (value < range.max) {
190                     rangeSet.map.put(value + 1, range.max);
191                 }
192                 return rangeSet;
193             }
194     
195             public void addBordersTo(Set<Long> borders) {
196                 range.addBordersTo(borders);
197             }
198     
199             public LongRangeSet empty() {
200                 return new LongRangeSet(range);
201             }
202     
203             public boolean intersects(LongRangeSet other) {
204                 for (Entry<Long, Long> entry : map.entrySet()) {
205                     SortedMap<Long, Long> subMap = entry.getValue() == Long.MAX_VALUE ? other.map.tailMap(entry.getKey()) : other.map
206                             .subMap(entry.getKey(), entry.getValue() + 1);
207                     if (!subMap.isEmpty()) {
208                         return true;
209                     }
210                     SortedMap<Long, Long> headMap = other.map.headMap(entry.getKey());
211                     if (!headMap.isEmpty() && headMap.get(headMap.lastKey()) >= entry.getKey()) {
212                         return true;
213                     }
214                 }
215                 return false;
216             }
217     
218             public void splitGreater(long number) {
219                 Long lNumber = number;
220                 if(number == Long.MAX_VALUE) {
221                     return;
222                 }
223                 Long nextNumber = number + 1;
224                 SortedMap<Long, Long> headMap = map.headMap(nextNumber);
225                 if (headMap.isEmpty()) {
226                     return;
227                 }
228                 Long lastKey = headMap.lastKey();
229                 Long lastValue = headMap.get(lastKey);
230                 if (number >= lastValue) {
231                     return;
232                 }
233                 map.put(lastKey, lNumber);
234                 map.put(nextNumber, lastValue);
235             }
236     
237             public String getSignature() {
238                 return range.signature;
239             }
240     
241             public boolean isEmpty() {
242                 return map.isEmpty();
243             }
244     
245             public boolean isFull() {
246                 if(map.size() != 1) {
247                     return false;
248                 }
249                 Long min = map.firstKey();
250                 Long max = map.get(min);
251                 return min <= range.min && max >= range.max;
252             }
253     
254             @Override
255             public String toString() {
256                 StringBuilder sb = new StringBuilder();
257                 for (Entry<Long, Long> entry : map.entrySet()) {
258                     if (sb.length() > 0) {
259                         sb.append("+");
260                     }
261                     if (entry.getKey().equals(entry.getValue())) {
262                         sb.append("{").append(entry.getKey()).append("}");
263                     } else {
264                         sb.append("[").append(entry.getKey()).append(", ").append(entry.getValue()).append("]");
265                     }
266                 }
267                 return sb.toString();
268             }

272                 final Iterator<Entry<Long, Long>> iterator = map.entrySet().iterator();
273                 return new Iterator<ValueRangeAnalysisFactory.LongRangeSet>() {
274                     @Override

277                     }

283                     }

285                     @Override
286                     public void remove() {
287                         throw new UnsupportedOperationException();
288                     }
289                 };
290             }

292             private void add(Long start, Long end) {
293                 SortedMap<Long, Long> headMap;
294                 if (end < Long.MAX_VALUE) {
295                     headMap = map.headMap(end + 1);
296                     Long tailEnd = map.remove(end + 1);
297                     if (tailEnd != null) {
298                         end = tailEnd;
299                     }
300                     if (!headMap.isEmpty()) {
301                         tailEnd = headMap.get(headMap.lastKey());
302                         if (tailEnd > end) {
303                             end = tailEnd;
304                         }
305                     }
306                 }
307                 headMap = map.headMap(start);
308                 if (!headMap.isEmpty()) {
309                     Long headStart = headMap.lastKey();
310                     Long headEnd = map.get(headStart);
311                     if (headEnd >= start - 1) {
312                         map.remove(headStart);
313                         start = headStart;
314                     }
315                 }
316                 map.subMap(start, end).clear();
317                 map.remove(end);
318                 map.put(start, end);
319             }
320     
321             public LongRangeSet add(LongRangeSet rangeSet) {
322                 for(Entry<Long, Long> entry : rangeSet.map.entrySet()) {
323                     add(entry.getKey(), entry.getValue());
324                 }
325                 return this;
326             }
327     
328             public boolean same(LongRangeSet rangeSet) {
329                 return map.equals(rangeSet.map);
330             }

352             private String fixCondition(String condition) {
353                 if (condition.equals("!= true")) {
354                     return "== false";
355                 }
356                 if (condition.equals("!= false")) {
357                     return "== true";
358                 }
359                 return condition;

401             public Condition extractCondition(BackIterator iterator) throws DataflowAnalysisException {
402                 Instruction comparisonInstruction = iterator.next().getInstruction();
403                 if (!(comparisonInstruction instanceof IfInstruction)) {
404                     return null;
405                 }
406                 short cmpOpcode = comparisonInstruction.getOpcode();
407                 int nargs = ((IfInstruction) comparisonInstruction).consumeStack(null);
408                 if (nargs == 2) {
409                     return extractTwoArgCondition(iterator, cmpOpcode, "I");
410                 } else if (nargs == 1) {
411                     Object val = extractValue(iterator, "I");
412                     if(val instanceof Value) {
413                         return new Condition(cmpOpcode, (Value) val, 0);
414                     } else if(val instanceof LCMP) {
415                         return extractTwoArgCondition(iterator, cmpOpcode, "J");
416                     }
417                 }
418                 return null;
419             }
420     
421             private Object extractValue(BackIterator iterator, String defSignature) throws DataflowAnalysisException {
422                 if(!iterator.hasNext()) {
423                     return null;
424                 }
425                 BasicBlock block = iterator.block;
426                 InstructionHandle ih = iterator.next();
427                 Instruction inst = ih.getInstruction();
428                 if (inst instanceof ConstantPushInstruction) {
429                     return ((ConstantPushInstruction) inst).getValue();
430                 }
431                 if (inst instanceof CPInstruction && inst instanceof PushInstruction) {
432                     Constant constant = cp.getConstant(((CPInstruction) inst).getIndex());
433                     if (constant instanceof ConstantObject) {
434                         Object value = ((ConstantObject) constant).getConstantValue(cp);
435                         if (value instanceof Number) {
436                             return value;
437                         }
438                     }
439                     return inst;
440                 }
441                 if(inst instanceof ARRAYLENGTH) {
442                     Object valueObj = extractValue(iterator, defSignature);
443                     if(valueObj instanceof Value) {
444                         Value value = (Value)valueObj;
445                         return new Value(value.name+".length", value.vn, "I");
446                     }
447                     return null;
448                 }
449                 if(inst instanceof GETFIELD) {
450                     Object valueObj = extractValue(iterator, defSignature);
451                     if(valueObj instanceof Value) {
452                         Value value = (Value)valueObj;
453                         ConstantCP desc = (ConstantCP)cp.getConstant(((GETFIELD)inst).getIndex());
454                         ConstantNameAndType nameAndType = (ConstantNameAndType) cp.getConstant(desc.getNameAndTypeIndex());
455                         String name = ((ConstantUtf8)cp.getConstant(nameAndType.getNameIndex())).getBytes();
456                         String signature = ((ConstantUtf8)cp.getConstant(nameAndType.getSignatureIndex())).getBytes();
457                         return new Value(value.name+"."+name, vnaDataflow.getFactAfterLocation(new Location(ih, block)).getStackValue(0), signature);
458                     }
459                     return null;
460                 }
461                 if(inst instanceof INVOKEVIRTUAL) {
462                     ConstantCP desc = (ConstantCP)cp.getConstant(((INVOKEVIRTUAL)inst).getIndex());
463                     ConstantNameAndType nameAndType = (ConstantNameAndType) cp.getConstant(desc.getNameAndTypeIndex());
464                     String className = cp.getConstantString(desc.getClassIndex(), CONSTANT_Class);
465                     String name = ((ConstantUtf8)cp.getConstant(nameAndType.getNameIndex())).getBytes();
466                     String signature = ((ConstantUtf8)cp.getConstant(nameAndType.getSignatureIndex())).getBytes();
467                     if(className.equals("java/lang/Integer") && name.equals("intValue") && signature.equals("()I") ||
468                             className.equals("java/lang/Long") && name.equals("longValue") && signature.equals("()J") ||
469                             className.equals("java/lang/Short") && name.equals("shortValue") && signature.equals("()S") ||
470                             className.equals("java/lang/Byte") && name.equals("byteValue") && signature.equals("()B") ||
471                             className.equals("java/lang/Boolean") && name.equals("booleanValue") && signature.equals("()Z") ||
472                             className.equals("java/lang/Character") && name.equals("charValue") && signature.equals("()C")) {
473                         Object valueObj = extractValue(iterator, defSignature);
474                         if(valueObj instanceof Value) {
475                             Value value = (Value)valueObj;
476                             return new Value(value.name, value.vn, String.valueOf(signature.charAt(signature.length()-1)));
477                         }
478                     }
479                     if(className.equals("java/lang/String") && name.equals("length") && signature.equals("()I")) {
480                         Object valueObj = extractValue(iterator, defSignature);
481                         if(valueObj instanceof Value) {
482                             Value value = (Value)valueObj;
483                             return new Value(value.name+".length()", value.vn, "I");
484                         }
485                     }
486                     return null;
487                 }
488                 if(inst instanceof LoadInstruction) {
489                     int index = ((LoadInstruction)inst).getIndex();
490                     LocalVariable lv = lvTable == null ? null : lvTable.getLocalVariable(index, ih.getPosition());
491                     String name, signature;
492                     if(lv == null) {
493                         name = "local$"+index;
494                         if(types.containsKey(index)) {
495                             signature = types.get(index).signature;
496                             name = types.get(index).name;
497                         } else {
498                             signature = defSignature;
499                         }
500                     } else {
501                         name = lv.getName();
502                         signature = lv.getSignature();
503                     }
504                     return new Value(name, vnaDataflow.getFactAfterLocation(new Location(ih, block)).getStackValue(0), signature);
505                 }
506                 return inst;
507             }

509             /**
510              * @param opcode
511              * @return opcode which returns the same result when arguments are placed in opposite order
512              */
513             private static short revertOpcode(short opcode) {
514                 switch (opcode) {
515                 case IF_ICMPGE:
516                     return IF_ICMPLE;
517                 case IF_ICMPGT:
518                     return IF_ICMPLT;
519                 case IF_ICMPLE:
520                     return IF_ICMPGE;
521                 case IF_ICMPLT:
522                     return IF_ICMPGT;
523                 case IFLE:
524                     return IFGE;
525                 case IFGE:
526                     return IFLE;
527                 case IFGT:
528                     return IFLT;
529                 case IFLT:
530                     return IFGT;
531                 default:
532                     return opcode;
533                 }
534             }

536             private Condition extractTwoArgCondition(BackIterator iterator, short cmpOpcode, String signature) throws DataflowAnalysisException {
537                 Object val2 = extractValue(iterator, signature);
538                 if(val2 instanceof Instruction) {
539                     return null;
540                 }
541                 Object val1 = extractValue(iterator, signature);
542                 if(val1 instanceof Instruction) {
543                     return null;
544                 }
545                 if (!(val1 instanceof Value) && !(val2 instanceof Value)) {
546                     return null;
547                 }
548                 if (!(val1 instanceof Value)) {
549                     Object tmp = val1;
550                     val1 = val2;
551                     val2 = tmp;
552                     cmpOpcode = revertOpcode(cmpOpcode);
553                 }
554                 if(!(val2 instanceof Number)) {
555                     return null;
556                 }
557                 return new Condition(cmpOpcode, (Value)val1, (Number)val2);
558             }

648         @Override
649         public ValueRangeAnalysis analyze(IAnalysisCache analysisCache, MethodDescriptor descriptor) throws CheckedAnalysisException {
650             XMethod xMethod = XFactory.createXMethod(descriptor);
651             if (xMethod.isNative() || xMethod.isSynthetic() || xMethod.isAbstract()) {
652                 return null;
653             }
654             CFG cfg;
655             try {
656                 cfg = analysisCache.getMethodAnalysis(CFG.class, descriptor);
657             } catch (MethodUnprofitableException e) {
658                 return null;
659             }
660             if (cfg == null) {
661                 return null;
662             }
663             ClassContext classContext = analysisCache.getClassAnalysis(ClassContext.class, descriptor.getClassDescriptor());
664             Method method = analysisCache.getMethodAnalysis(Method.class, descriptor);
665             Context context = new Context(cfg.getMethodGen().getConstantPool().getConstantPool(), method.getCode().getLocalVariableTable(),
666                     getParameterTypes(descriptor), classContext.getValueNumberDataflow(method));
667             Map<ValueNumber, VariableData> analyzedArguments = new HashMap<>();
668             Map<Edge, Branch> allEdges = new IdentityHashMap<>();
669             for (Iterator<Edge> edgeIterator = cfg.edgeIterator(); edgeIterator.hasNext();) {
670                 Edge edge = edgeIterator.next();
671                 if (edge.getType() == EdgeTypes.IFCMP_EDGE) {
672                     BasicBlock source = edge.getSource();
673                     Condition condition = context.extractCondition(new BackIterator(cfg, source));
674                     if(condition == null) {
675                         continue;
676                     }
677                     ValueNumber valueNumber = condition.value.vn;
678                     String varName = condition.value.name;
679                     VariableData data = analyzedArguments.get(valueNumber);
680                     if (data == null) {
681                         try {
682                             data = new VariableData(condition.value.signature);
683                         } catch (IllegalArgumentException e) {
684                             continue;
685                         }
686                         analyzedArguments.put(valueNumber, data);
687                     }
688                     Number number = condition.number;
689                     String numberStr = convertNumber(data.splitSet.getSignature(), number);
690                     Branch branch = null;
691                     switch (condition.opcode) {
692                     case IF_ICMPGT:
693                     case IFGT:
694                         branch = new Branch(varName, "> " + numberStr, "<= " + numberStr, data.splitSet.gt(number.longValue()), number);
695                         break;
696                     case IF_ICMPLE:
697                     case IFLE:
698                         branch = new Branch(varName, "<= " + numberStr, "> " + numberStr, data.splitSet.le(number.longValue()), number);
699                         break;
700                     case IF_ICMPGE:
701                     case IFGE:
702                         branch = new Branch(varName, ">= " + numberStr, "< " + numberStr, data.splitSet.ge(number.longValue()), number);
703                         break;
704                     case IF_ICMPLT:
705                     case IFLT:
706                         branch = new Branch(varName, "< " + numberStr, ">= " + numberStr, data.splitSet.lt(number.longValue()), number);
707                         break;
708                     case IF_ICMPEQ:
709                     case IFEQ:
710                         branch = new Branch(varName, "== " + numberStr, "!= " + numberStr, data.splitSet.eq(number.longValue()), number);
711                         break;
712                     case IF_ICMPNE:
713                     case IFNE:
714                         branch = new Branch(varName, "!= " + numberStr, "== " + numberStr, data.splitSet.ne(number.longValue()), number);
715                         break;
716                     default:
717                         break;
718                     }
719                     if(branch != null) {
720                         data.addBranch(edge, branch);
721                         allEdges.put(edge, branch);
722                     }
723                 }
724             }
725             FinallyDuplicatesInfo fi = null;
726             List<RedundantCondition> redundantConditions = new ArrayList<>();
727             for (VariableData data : analyzedArguments.values()) {
728                 for (LongRangeSet subRange : data.splitSet) {
729                     BitSet reachedBlocks = new BitSet();
730                     walkCFG(cfg, cfg.getEntry(), subRange, data.edges, reachedBlocks, new HashSet<Long>());
731                     data.reachableBlocks.or(reachedBlocks);
732                 }
733             }
734             for (VariableData data : analyzedArguments.values()) {
735                 for (Entry<Edge, Branch> entry : data.edges.entrySet()) {
736                     Branch branch = entry.getValue();
737                     Edge edge = entry.getKey();
738                     if (branch.trueReachedSet.isEmpty() ^ branch.falseReachedSet.isEmpty()) {
739                         if(fi == null) {
740                             fi = analysisCache.getMethodAnalysis(FinallyDuplicatesInfo.class, descriptor);
741                         }
742                         List<Edge> duplicates = fi.getDuplicates(cfg, edge);
743                         if(!duplicates.isEmpty()) {
744                             boolean trueValue = !branch.trueReachedSet.isEmpty();
745                             boolean falseValue = !branch.falseReachedSet.isEmpty();
746                             for(Edge dup : duplicates) {
747                                 Branch dupBranch = allEdges.get(dup);
748                                 if(dupBranch != null) {
749                                     trueValue |= !dupBranch.trueReachedSet.isEmpty();
750                                     falseValue |= !dupBranch.falseReachedSet.isEmpty();
751                                 }
752                             }
753                             if(trueValue && falseValue) {
754                                 continue;
755                             }
756                         }
757                         BasicBlock trueTarget = edge.getTarget();
758                         BasicBlock falseTarget = cfg.getSuccessorWithEdgeType(edge.getSource(), EdgeTypes.FALL_THROUGH_EDGE);
759                         String condition;
760                         BasicBlock deadTarget, aliveTarget;
761                         if(branch.trueReachedSet.isEmpty()) {
762                             condition = branch.varName + " " + branch.falseCondition;
763                             deadTarget = trueTarget;
764                             aliveTarget = falseTarget;
765                         } else {
766                             condition = branch.varName + " " + branch.trueCondition;
767                             deadTarget = falseTarget;
768                             aliveTarget = trueTarget;
769                         }
770                         redundantConditions.add(new RedundantCondition(Location.getLastLocation(edge.getSource()), condition,
771                                 !data.reachableBlocks.get(deadTarget.getLabel()), getLocation(deadTarget), getLocation(aliveTarget),
772                                 branch.trueSet.getSignature(), branch.trueSet.isEmpty() || branch.trueSet.isFull(),
773                                 branch.number, branch.numbers.contains(branch.number.longValue())));
774                     }
775                 }
776             }
777             if (!redundantConditions.isEmpty()) {
778                 BitSet assertionBlocks = new BitSet();
779                 MethodGen methodGen = cfg.getMethodGen();
780                 Iterator<InstructionHandle> iterator = methodGen.getInstructionList().iterator();
781                 while(iterator.hasNext()) {
782                     InstructionHandle ih = iterator.next();
783                     if(ih.getInstruction() instanceof GETSTATIC) {
784                         Instruction next = ih.getNext().getInstruction();
785                         if(next instanceof IFNE) {
786                             GETSTATIC getStatic = (GETSTATIC)ih.getInstruction();
787                             if ("$assertionsDisabled".equals(getStatic.getFieldName(methodGen.getConstantPool()))
788                                     && "Z".equals(getStatic.getSignature(methodGen.getConstantPool()))) {
789                                 int end = ((IFNE)next).getTarget().getPosition();
790                                 assertionBlocks.set(ih.getNext().getPosition(), end);
791                             }
792                         }
793                     }
794                 }
795                 if(!assertionBlocks.isEmpty()) {
796                     List<RedundantCondition> filtered = new ArrayList<>();
797                     for(RedundantCondition condition : redundantConditions) {
798                         if(!(assertionBlocks.get(condition.getLocation().getHandle().getPosition()))) {
799                             // TODO: do not filter out failed asserts
800                             filtered.add(condition);
801                         }
802                     }
803                     redundantConditions = filtered;
804                 }
805                 Collections.sort(redundantConditions, new Comparator<RedundantCondition>() {
806                     @Override
807                     public int compare(RedundantCondition o1, RedundantCondition o2) {
808                         return o1.location.compareTo(o2.location);
809                     }
810                 });
811                 return new ValueRangeAnalysis(redundantConditions);
812             }
813             return null;
814         }

816         private static Location getLocation(BasicBlock block) {
817             InstructionHandle handle = block.getFirstInstruction();
818             if(handle == null) {
819                 handle = block.getExceptionThrower();
820             }
821             return handle == null ? null : new Location(handle, block);
822         }

824         private static String convertNumber(String signature, Number number) {
825             long val = number.longValue();
826             switch (signature) {
827             case "Z":
828                 if (val == 0) {
829                     return "false";
830                 }
831                 return "true";
832             case "C":
833                 if (val == '\n') {
834                     return "'\\n'";
835                 }
836                 if (val == '\r') {
837                     return "'\\r'";
838                 }
839                 if (val == '\b') {
840                     return "'\\b'";
841                 }
842                 if (val == '\t') {
843                     return "'\\t'";
844                 }
845                 if (val == '\'') {
846                     return "'\\''";
847                 }
848                 if (val == '\\') {
849                     return "'\\\\'";
850                 }
851                 if (val >= 32 && val < 128) {
852                     return "'" + ((char) val) + "'";
853                 }
854                 return convertNumber(val);
855             case "I":
856                 if(val >= 32 && val < 128) {
857                     return val+" ('" + ((char) val) + "')";
858                 }
859                 return convertNumber(val);
860             default:
861                 return convertNumber(val);
862             }
863         }
864     
865         private static String convertNumber(long val) {
866             if(val == Long.MIN_VALUE) {
867                 return "Long.MIN_VALUE";
868             }
869             if(val == Long.MAX_VALUE) {
870                 return "Long.MAX_VALUE";
871             }
872             String suffix = "";
873             if (val > Integer.MAX_VALUE || val < Integer.MIN_VALUE) {
874                 suffix = "L";
875             }
876             if (val > 128) {
877                 return val + suffix + " (0x" + Long.toHexString(val) + suffix + ")";
878             }
879             return val + suffix;
880         }

882         private static Map<Integer, Value> getParameterTypes(MethodDescriptor descriptor) {
883             Type[] argumentTypes = Type.getArgumentTypes(descriptor.getSignature());
884             int j = 0;
885             Map<Integer, Value> result = new HashMap<>();
886             if(!descriptor.isStatic()) {
887                 result.put(j++, new Value("this", null, "L"+descriptor.getSlashedClassName()+";"));
888             }
889             for (int i = 0; i < argumentTypes.length; i++) {
890                 result.put(j, new Value("arg"+i, null, argumentTypes[i].getSignature()));
891                 j += argumentTypes[i].getSize();
892             }
893             return result;
894         }

896         private static void walkCFG(CFG cfg, BasicBlock basicBlock, LongRangeSet subRange, Map<Edge, Branch> edges, BitSet reachedBlocks, Set<Long> numbers) {
897             reachedBlocks.set(basicBlock.getLabel());
898             for (Iterator<Edge> iterator = cfg.outgoingEdgeIterator(basicBlock); iterator.hasNext();) {
899                 Edge edge = iterator.next();
900                 Branch branch = edges.get(edge);
901                 if (branch != null) {
902                     branch.numbers.addAll(numbers);
903                     numbers = new HashSet<>(numbers);
904                     numbers.add(branch.number.longValue());
905                     if (branch.trueSet.intersects(subRange)) {
906                         branch.trueReachedSet.add(subRange);
907                     } else {
908                         branch.falseReachedSet.add(subRange);
909                         continue;
910                     }
911                 }
912                 BasicBlock target = edge.getTarget();
913                 if (!reachedBlocks.get(target.getLabel())) {
914                     walkCFG(cfg, target, subRange, edges, reachedBlocks, numbers);
915                 }
916                 if (branch != null) {
917                     break;
918                 }
919             }
920         }

922         private static class BackIterator implements Iterator<InstructionHandle> {
923             private BasicBlock block;
924             private InstructionHandle next;
925             private final CFG cfg;
926     
927             public BackIterator(CFG cfg, BasicBlock block) {
928                 this.block = block;
929                 this.cfg = cfg;
930                 this.next = block.getLastInstruction();
931             }
932     
933             @Override
934             public boolean hasNext() {
935                 return next != null;
936             }
937     
938             @Override
939             public InstructionHandle next() {
940                 if (!hasNext()) {
941                     throw new NoSuchElementException();
942                 }
943                 InstructionHandle result = next;
944                 if(result == block.getFirstInstruction()) {
945                     do {
946                         Iterator<Edge> edgeIterator = cfg.incomingEdgeIterator(block);
947                         if(!edgeIterator.hasNext()) {
948                             break;
949                         }
950                         Edge edge = edgeIterator.next();
951                         if(!edgeIterator.hasNext() && edge.getType() == EdgeTypes.FALL_THROUGH_EDGE) {
952                             block = edge.getSource();
953                         } else {
954                             break;
955                         }
956                     } while(block.isExceptionThrower());
957                 }
958                 next = (block.isExceptionThrower() || result == block.getFirstInstruction()) ? null : next.getPrev();
959                 return result;
960             }
961     
962             @Override
963             public void remove() {
964                 throw new UnsupportedOperationException();
965             }
966         }
967     
968         @Override
969         public void registerWith(IAnalysisCache analysisCache) {
970             analysisCache.registerMethodAnalysisEngine(ValueRangeAnalysis.class, this);
971         }
