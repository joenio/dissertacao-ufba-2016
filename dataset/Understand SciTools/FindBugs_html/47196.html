42     public class RepeatedConditionals extends OpcodeStackDetector {
43         BugReporter bugReporter;
44     
45         private final NoSideEffectMethodsDatabase noSideEffectMethods;
46     
47         public RepeatedConditionals(BugReporter bugReporter) {
48             this.bugReporter = bugReporter;
49             this.noSideEffectMethods = Global.getAnalysisCache().getDatabase(NoSideEffectMethodsDatabase.class);
50             reset();
51         }
52     
53         @Override
54         public void visit(Code code) {
55             boolean interesting = true;
56             if (interesting) {
57                 // initialize any variables we want to initialize for the method
58                 super.visit(code); // make callbacks to sawOpcode for all opcodes
59                 reset();
60             }
61         }

63         /*
64          * (non-Javadoc)
65          *
66          * @see edu.umd.cs.findbugs.bcel.OpcodeStackDetector#sawOpcode(int)
67          */
68         int oldPC;
69     
70         LinkedList<Integer> emptyStackLocations = new LinkedList<Integer>();
71     
72         LinkedList<Integer> prevOpcodeLocations = new LinkedList<Integer>();
73     
74         Map<Integer, Integer> branchTargets = new HashMap<Integer, Integer>();
75     
76         @Override
77         public void sawBranchTo(int pc) {
78             branchTargets.put(getPC(), pc);
79         }
80     
81         @Override
82         public void sawOpcode(int seen) {
83             if (hasSideEffect(seen)) {
84                 reset();
85             } else if (stack.getStackDepth() == 0) {
86                 if (emptyStackLocations.size() > 1) {
87                     for(int n=1; n<=emptyStackLocations.size()/2; n++) {
88                         int first = emptyStackLocations.get(emptyStackLocations.size() - 2*n);
89                         int second = emptyStackLocations.get(emptyStackLocations.size() - n);
90                         int third = getPC();
91                         if (third - second == second - first) {
92                             int endOfFirstSegment = prevOpcodeLocations.get(emptyStackLocations.size() - n);
93                             int endOfSecondSegment = oldPC;
94                             int opcodeAtEndOfFirst = getCodeByte(endOfFirstSegment);
95                             int opcodeAtEndOfSecond = getCodeByte(endOfSecondSegment);
96     
97                             if (!isBranch(opcodeAtEndOfFirst) || !isBranch(opcodeAtEndOfSecond)) {
98                                 continue;
99                             }
100                             if (opcodeAtEndOfFirst == Opcodes.GOTO || opcodeAtEndOfSecond == Opcodes.GOTO) {
101                                 continue;
102                             }
103                             if (opcodeAtEndOfFirst != opcodeAtEndOfSecond
104                                     && !areOppositeBranches(opcodeAtEndOfFirst, opcodeAtEndOfSecond)) {
105                                 continue;
106                             }
107     
108                             if (first == endOfFirstSegment) {
109                                 continue;
110                             }
111                             Integer firstTarget = branchTargets.get(endOfFirstSegment);
112                             Integer secondTarget = branchTargets.get(endOfSecondSegment);
113                             if (firstTarget == null || secondTarget == null) {
114                                 continue;
115                             }
116                             if (firstTarget >= second && firstTarget <= endOfSecondSegment) {
117                                 // first jumps inside second
118                                 continue;
119                             }
120                             boolean identicalCheck = firstTarget.equals(secondTarget) && opcodeAtEndOfFirst == opcodeAtEndOfSecond
121                                     || (firstTarget.intValue() == getPC() && opcodeAtEndOfFirst != opcodeAtEndOfSecond);
122                             if(!compareCode(first, endOfFirstSegment, second, endOfSecondSegment, !identicalCheck)) {
123                                 continue;
124                             }
125                             SourceLineAnnotation firstSourceLine = SourceLineAnnotation.fromVisitedInstructionRange(getClassContext(),
126                                     this, first, endOfFirstSegment - 1);
127                             SourceLineAnnotation secondSourceLine = SourceLineAnnotation.fromVisitedInstructionRange(getClassContext(),
128                                     this, second, endOfSecondSegment - 1);
129     
130                             int priority = HIGH_PRIORITY;
131                             if (firstSourceLine.getStartLine() == -1 || firstSourceLine.getStartLine() != secondSourceLine.getEndLine()) {
132                                 priority++;
133                             }
134                             if (stack.isJumpTarget(second)) {
135                                 priority++;
136                             }
137                             if (!identicalCheck) {
138                                 // opposite checks
139                                 priority += 2;
140                             }
141     
142                             BugInstance bug = new BugInstance(this, "RpC_REPEATED_CONDITIONAL_TEST", priority).addClassAndMethod(this)
143                                     .add(firstSourceLine).add(secondSourceLine);
144                             bugReporter.reportBug(bug);
145                         }
146                     }
147                 }
148                 emptyStackLocations.add(getPC());
149                 prevOpcodeLocations.add(oldPC);
150     
151             }
152             oldPC = getPC();
153         }
154     
155         private boolean compareCode(int first, int endOfFirstSegment, int second,
156                 int endOfSecondSegment, boolean oppositeChecks) {
157             if(endOfFirstSegment-first != endOfSecondSegment-second) {
158                 return false;
159             }
160             MethodGen methodGen = null;
161             try {
162                 methodGen = Global.getAnalysisCache().getMethodAnalysis(MethodGen.class, getMethodDescriptor());
163             } catch (CheckedAnalysisException e) {
164                 // Ignore
165             }
166             if(methodGen == null) {
167                 // MethodGen is absent for some reason: fallback to byte-to-byte comparison
168                 byte[] code = getCode().getCode();
169                 for (int i = first; i < endOfFirstSegment; i++) {
170                     if (code[i] != code[i - first + second]) {
171                         return false;
172                     }
173                 }
174                 return true;
175             }
176             InstructionHandle firstHandle = methodGen.getInstructionList().findHandle(first);
177             InstructionHandle secondHandle = methodGen.getInstructionList().findHandle(second);
178             while(true) {
179                 if(firstHandle == null || secondHandle == null) {
180                     return false;
181                 }
182                 if(firstHandle.getPosition() >= endOfFirstSegment) {
183                     return secondHandle.getPosition() >= endOfSecondSegment;
184                 }
185                 if(secondHandle.getPosition() >= endOfSecondSegment) {
186                     return firstHandle.getPosition() >= endOfFirstSegment;
187                 }
188                 Instruction firstInstruction = firstHandle.getInstruction();
189                 Instruction secondInstruction = secondHandle.getInstruction();
190                 if(firstInstruction instanceof BranchInstruction && secondInstruction instanceof BranchInstruction) {
191                     int firstOpcode = firstInstruction.getOpcode();
192                     int secondOpcode = secondInstruction.getOpcode();
193                     if(firstOpcode != secondOpcode) {
194                         return false;
195                     }
196                     int firstTarget = ((BranchInstruction)firstInstruction).getTarget().getPosition();
197                     int secondTarget = ((BranchInstruction)secondInstruction).getTarget().getPosition();
198                     if(firstTarget == second) {
199                         if(oppositeChecks || secondTarget <= endOfSecondSegment) {
200                             return false;
201                         }
202                     } else {
203                         if(!((firstTarget >= first && firstTarget <= endOfFirstSegment && firstTarget - first == secondTarget - second)
204                                 || firstTarget == secondTarget)) {
205                             return false;
206                         }
207                     }
208                 } else {
209                     if(!firstInstruction.equals(secondInstruction)) {
210                         return false;
211                     }
212                 }
213                 firstHandle = firstHandle.getNext();
214                 secondHandle = secondHandle.getNext();
215             }
216         }
217     
218         private boolean hasSideEffect(int seen) {
219             if(seen == INVOKEVIRTUAL || seen == INVOKESPECIAL || seen == INVOKEINTERFACE || seen == INVOKESTATIC) {
220                 return noSideEffectMethods.is(getMethodDescriptorOperand(), MethodSideEffectStatus.SE, MethodSideEffectStatus.OBJ);
221             }
222             return isRegisterStore() || isReturn(seen) || isSwitch(seen) || seen == INVOKEDYNAMIC || seen == PUTFIELD
223                     || seen == PUTSTATIC;
224         }
225     
226         private void reset() {
227             emptyStackLocations.clear();
228             prevOpcodeLocations.clear();
229             branchTargets.clear();
230             oldPC = -1;
231         }
