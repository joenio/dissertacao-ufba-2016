20     package edu.umd.cs.findbugs;
21     
22     import org.apache.bcel.generic.InstructionHandle;
23     
24     import edu.umd.cs.findbugs.ba.Edge;
25     
26     /**
27      * A scanner for implementing a state machine over a sequence of instructions
28      * and control edges.
29      */
30     public interface InstructionScanner {
31         /**
32          * Traverse an edge.
33          */
34         public void traverseEdge(Edge edge);
35     
36         /**
37          * Traverse an instruction.
38          */
39         public void scanInstruction(InstructionHandle handle);
40     
41         /**
42          * Return true if this scanner has completed, false otherwise.
43          */
44         public boolean isDone();
45     }
