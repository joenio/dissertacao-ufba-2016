61     /**
62      * Build database of unconditionally dereferenced parameters.
63      *
64      * @author David Hovemeyer
65      */
66     public abstract class BuildUnconditionalParamDerefDatabase implements Detector {
67         public static final boolean VERBOSE_DEBUG = SystemProperties.getBoolean("fnd.debug.nullarg.verbose");
68     
69         private static final boolean DEBUG = SystemProperties.getBoolean("fnd.debug.nullarg") || VERBOSE_DEBUG;
70     
71         public final TypeQualifierValue<javax.annotation.Nonnull> nonnullTypeQualifierValue;
72     
73         abstract protected void reportBug(BugInstance bug);
74     
75         public BuildUnconditionalParamDerefDatabase() {
76             this.nonnullTypeQualifierValue =  TypeQualifierValue.getValue(javax.annotation.Nonnull.class, null);
77         }
78     
79         @Override
80         public void visitClassContext(ClassContext classContext) {
81             boolean fullAnalysis = AnalysisContext.currentAnalysisContext().getBoolProperty(
82                     FindBugsAnalysisFeatures.INTERPROCEDURAL_ANALYSIS_OF_REFERENCED_CLASSES);
83             if (!fullAnalysis && !AnalysisContext.currentAnalysisContext().isApplicationClass(classContext.getJavaClass())) {
84                 return;
85             }
86             if (VERBOSE_DEBUG) {
87                 System.out.println("Visiting class " + classContext.getJavaClass().getClassName());
88             }
89     
90             for (Method m : classContext.getMethodsInCallOrder()) {
91                 considerMethod(classContext, m);
92             }
93         }
94     
95         private void considerMethod(ClassContext classContext, Method method) {
96             boolean hasReferenceParameters = false;
97             for (Type argument : method.getArgumentTypes()) {
98                 if (argument instanceof ReferenceType) {
99                     hasReferenceParameters = true;
100                 }
101             }
102     
103             if (hasReferenceParameters && classContext.getMethodGen(method) != null) {
104                 if (VERBOSE_DEBUG) {
105                     System.out.println("Check " + method);
106                 }
107                 analyzeMethod(classContext, method);
108             }
109         }
110     
111         private void analyzeMethod(ClassContext classContext, Method method) {
112             JavaClass jclass = classContext.getJavaClass();
113             XMethod xmethod = XFactory.createXMethod(jclass, method);
114             try {
115                 CFG cfg = classContext.getCFG(method);
116     
117                 ValueNumberDataflow vnaDataflow = classContext.getValueNumberDataflow(method);
118                 UnconditionalValueDerefDataflow dataflow = classContext.getUnconditionalValueDerefDataflow(method);
119     
120                 SignatureParser parser = new SignatureParser(method.getSignature());
121                 int paramLocalOffset = method.isStatic() ? 0 : 1;
122     
123                 // Build BitSet of params that are unconditionally dereferenced
124                 BitSet unconditionalDerefSet = new BitSet();
125                 UnconditionalValueDerefSet entryFact = dataflow.getResultFact(cfg.getEntry());
126                 Iterator<String> paramIterator = parser.parameterSignatureIterator();
127                 int i = 0;
128                 while (paramIterator.hasNext()) {
129                     String paramSig = paramIterator.next();
130     
131                     ValueNumber paramVN = vnaDataflow.getAnalysis().getEntryValue(paramLocalOffset);
132     
133                     handleParameter: if (entryFact.isUnconditionallyDereferenced(paramVN)) {
134                         TypeQualifierAnnotation directTypeQualifierAnnotation = TypeQualifierApplications
135                                 .getDirectTypeQualifierAnnotation(xmethod, i, nonnullTypeQualifierValue);
136                         TypeQualifierAnnotation typeQualifierAnnotation = TypeQualifierApplications
137                                 .getEffectiveTypeQualifierAnnotation(xmethod, i, nonnullTypeQualifierValue);
138                         boolean implicitNullCheckForEquals = false;
139                         if (directTypeQualifierAnnotation == null && "equals".equals(method.getName())
140                                 && "(Ljava/lang/Object;)Z".equals(method.getSignature()) && !method.isStatic()) {
141                             implicitNullCheckForEquals = true;
142                             Code code = method.getCode();
143                             ConstantPool cp = jclass.getConstantPool();
144                             byte codeBytes[] = code.getCode();
145                             for (CodeException e : code.getExceptionTable()) {
146                                 ConstantClass cl = (ConstantClass) cp.getConstant(e.getCatchType());
147                                 int endPC = e.getEndPC();
148                                 int startPC = e.getStartPC();
149                                 int handlerPC = e.getHandlerPC();
150                                 if (startPC == 0 && endPC + 1 == handlerPC && handlerPC == codeBytes.length - 3
151                                         && (codeBytes[handlerPC + 1] & 0xff) == Constants.ICONST_0
152                                         && (codeBytes[handlerPC + 2] & 0xff) == Constants.IRETURN
153                                         && FindNullDeref.catchTypesForNull.contains(cl.getBytes(cp))) {
154                                     // equals method body contained in try clause
155                                     return;
156                                 }
157                             }
158                             typeQualifierAnnotation = TypeQualifierAnnotation.getValue(nonnullTypeQualifierValue, When.MAYBE);
159                         }
160     
161                         if (typeQualifierAnnotation != null && typeQualifierAnnotation.when == When.ALWAYS) {
162                             unconditionalDerefSet.set(i);
163                         } else if (isCaught(classContext, method, entryFact, paramVN)) {
164                             // ignore
165                         } else if (typeQualifierAnnotation == null) {
166                             unconditionalDerefSet.set(i);
167                         } else {
168                             int paramLocal = xmethod.isStatic() ? i : i + 1;
169                             int priority = Priorities.NORMAL_PRIORITY;
170                             if (typeQualifierAnnotation.when != When.UNKNOWN) {
171                                 priority--;
172                             }
173                             if (xmethod.isStatic() || xmethod.isFinal() || xmethod.isPrivate() || "<init>".equals(xmethod.getName())
174                                     || jclass.isFinal()) {
175                                 priority--;
176                             }
177                             if (directTypeQualifierAnnotation == null) {
178                                 priority++;
179                             }
180                             String bugPattern = implicitNullCheckForEquals ? "NP_EQUALS_SHOULD_HANDLE_NULL_ARGUMENT"
181                                     : "NP_PARAMETER_MUST_BE_NONNULL_BUT_MARKED_AS_NULLABLE";
182                             reportBug(new BugInstance(this, bugPattern, priority).addClassAndMethod(jclass, method).add(
183                                     LocalVariableAnnotation.getParameterLocalVariableAnnotation(method, paramLocal)));
184                         }
185                     }
186                     i++;
187                     if ("D".equals(paramSig) || "J".equals(paramSig)) {
188                         paramLocalOffset += 2;
189                     } else {
190                         paramLocalOffset += 1;
191                     }
192                 }
193     
194                 // No need to add properties if there are no unconditionally
195                 // dereferenced params
196                 if (unconditionalDerefSet.isEmpty()) {
197                     if (VERBOSE_DEBUG) {
198                         System.out.println("\tResult is empty");
199                     }
200                     return;
201                 }
202     
203                 if (VERBOSE_DEBUG) {
204                     ClassContext.dumpDataflowInformation(method, cfg, vnaDataflow, classContext.getIsNullValueDataflow(method),
205                             dataflow, classContext.getTypeDataflow(method));
206                 }
207                 ParameterProperty property = new ParameterProperty();
208                 property.setParamsWithProperty(unconditionalDerefSet);
209     
210                 AnalysisContext.currentAnalysisContext().getUnconditionalDerefParamDatabase()
211                 .setProperty(xmethod.getMethodDescriptor(), property);
212                 if (DEBUG) {
213                     System.out.println("Unconditional deref: " + xmethod + "=" + property);
214                 }
215             } catch (CheckedAnalysisException e) {
216                 AnalysisContext.currentAnalysisContext().getLookupFailureCallback()
217                 .logError("Error analyzing " + xmethod + " for unconditional deref training", e);
218             }
219         }
220     
221         public boolean isCaught(ClassContext classContext, Method method, UnconditionalValueDerefSet entryFact, ValueNumber paramVN) {
222             boolean caught = true;
223     
224             Set<Location> dereferenceSites
225             = entryFact.getDerefLocationSet(paramVN);
226             if (dereferenceSites != null && !dereferenceSites.isEmpty()) {
227                 ConstantPool cp = classContext.getJavaClass().getConstantPool();
228     
229                 for(Location loc : dereferenceSites) {
230                     if (!FindNullDeref.catchesNull(cp, method.getCode(), loc)) {
231                         caught = false;
232                     }
233                 }
234     
235             }
236             return caught;
237         }
