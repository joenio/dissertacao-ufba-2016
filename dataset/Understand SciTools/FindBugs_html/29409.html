20     package edu.umd.cs.findbugs;
21     
22     import java.util.regex.Matcher;
23     import java.util.regex.Pattern;
24     
25     /**
26      * Support for finding out what version of Java we're running on.
27      */
28     public class JavaVersion {
29         private static final Pattern PATTERN = Pattern.compile("^(\\d+)\\.(\\d+)(\\..*)?$");
30     
31         private final int major;
32     
33         private final int minor;
34     
35         private final String rest;
36     
37         /**
38          * StaticConstant for the Java version we're currently running on.
39          */
40         private static JavaVersion runtimeVersion;
41     
42         static {
43             try {
44                 runtimeVersion = new JavaVersion(SystemProperties.getProperty("java.version"));
45             } catch (JavaVersionException e) {
46                 System.err.println("Warning: Unknown version of Java");
47                 // Assume Java 1.0.
48                 runtimeVersion = new JavaVersion(1, 0);
49                 e.printStackTrace();
50             }
51         }
52     
53         /**
54          * StaticConstant for Java 1.5 (Tiger).
55          */
56         public static final JavaVersion JAVA_1_5 = new JavaVersion(1, 5);
57     
58         /**
59          * Constructor.
60          *
61          * @param versionString
62          *            a version string, as returned from the
63          *            <code>java.version</code> system property: e.g., "1.4.2_04"
64          */
65         public JavaVersion(String versionString) throws JavaVersionException {
66             Matcher matcher = PATTERN.matcher(versionString);
67             if (!matcher.matches()) {
68                 throw new JavaVersionException("Could not parse Java version string: " + versionString);
69             }
70             try {
71                 major = Integer.parseInt(matcher.group(1));
72                 minor = Integer.parseInt(matcher.group(2));
73                 if (matcher.group(3) != null) {
74                     rest = matcher.group(3);
75                 } else {
76                     rest = "";
77                 }
78             } catch (NumberFormatException e) {
79                 throw new JavaVersionException("Could not parse Java Version string: " + versionString, e);
80             }
81         }
82     
83         /**
84          * Constructor.
85          *
86          * @param major
87          *            major version
88          * @param minor
89          *            minor version
90          */
91         public JavaVersion(int major, int minor) {
92             this.major = major;
93             this.minor = minor;
94             this.rest = "";
95         }
96     
97         /**
98          * Get the major version number.
99          */
100         public int getMajor() {
101             return major;
102         }
103     
104         /**
105          * Get the minor version number.
106          */
107         public int getMinor() {
108             return minor;
109         }
110     
111         /**
112          * Get the rest of the version string after the major and minor numbers.
113          */
114         public String getRest() {
115             return rest;
116         }
117     
118         @Override
119         public String toString() {
120             StringBuilder buf = new StringBuilder();
121             buf.append(major);
122             buf.append('.');
123             buf.append(minor);
124             if (rest != null) {
125                 buf.append(rest);
126             }
127     
128             return buf.toString();
129         }
130     
131         /**
132          * Get the version of Java that we are currently running under.
133          */
134         public static JavaVersion getRuntimeVersion() {
135             return runtimeVersion;
136         }
137     
138         /**
139          * Return whether the Java version represented by this object is at least as
140          * recent as the one given.
141          *
142          * @param other
143          *            another JavaVersion
144          * @return true if this Java version is at least as recent as the one given
145          */
146         public boolean isSameOrNewerThan(JavaVersion other) {
147             return this.major > other.major || (this.major == other.major && this.minor >= other.minor);
148         }
149     }
