21     package edu.umd.cs.findbugs.detect;
22     
23     import java.util.HashMap;
24     import java.util.Map;
25     import java.util.Map.Entry;
26     
27     import org.apache.bcel.classfile.Method;
28     
29     import edu.umd.cs.findbugs.BugInstance;
30     import edu.umd.cs.findbugs.BugReporter;
31     import edu.umd.cs.findbugs.BytecodeScanningDetector;
32     import edu.umd.cs.findbugs.StatelessDetector;
33     import edu.umd.cs.findbugs.SystemProperties;
34     import edu.umd.cs.findbugs.visitclass.DismantleBytecode;
35     
36     /**
37      * Find occurrences of using the String "+" or "+=" operators within a loop.
38      * This is much less efficient than creating a dedicated StringBuffer object
39      * outside the loop, and then appending to it.
40      *
41      * @author Dave Brosius
42      * @author William Pugh
43      */
44     public class StringConcatenation extends BytecodeScanningDetector implements StatelessDetector {
45         private static final boolean DEBUG = SystemProperties.getBoolean("sbsc.debug");
46     
47         static final int SEEN_NOTHING = 0;
48     
49         static final int SEEN_NEW = 1;
50     
51         static final int SEEN_APPEND1 = 2;
52     
53         static final int SEEN_APPEND2 = 3;
54     
55         static final int CONSTRUCTED_STRING_ON_STACK = 4;
56     
57         static final int POSSIBLE_CASE = 5;
58     
59         private final BugReporter bugReporter;
60     
61         private boolean reportedThisMethod;
62     
63         private int registerOnStack = -1;
64     
65         private int stringSource = -1;
66     
67         private int createPC = -1;
68     
69         private int state = SEEN_NOTHING;
70     
71         public StringConcatenation(BugReporter bugReporter) {
72             this.bugReporter = bugReporter;
73         }
74     
75         // Keep track of which registers where clobbered at which PC, on
76         // a per-method basis
77         private Map<Integer, Integer> clobberedRegisters = new HashMap<Integer, Integer>();
78     
79         @Override
80         public void visit(Method obj) {
81             if (DEBUG) {
82                 System.out.println("------------------- Analyzing " + obj.getName() + " ----------------");
83             }
84             reset();
85             clobberedRegisters = new HashMap<Integer, Integer>();
86             reportedThisMethod = false;
87             super.visit(obj);
88         }
89     
90         private void reset() {
91             state = SEEN_NOTHING;
92             createPC = -1;
93             registerOnStack = -1;
94             stringSource = -1;
95     
96             // For debugging: print what call to reset() is being invoked.
97             // This helps figure out why the detector is failing to
98             // recognize a particular idiom.
99             if (DEBUG) {
100                 System.out.println("Reset from: " + new Throwable().getStackTrace()[1]);
101             }
102         }
103     
104         private boolean storeIntoRegister(int seen, int reg) {
105             switch (seen) {
106             case ASTORE_0:
107                 return reg == 0;
108             case ASTORE_1:
109                 return reg == 1;
110             case ASTORE_2:
111                 return reg == 2;
112             case ASTORE_3:
113                 return reg == 3;
114             case ASTORE:
115                 return reg == getRegisterOperand();
116             default:
117                 return false;
118             }
119         }
120     
121         @Override
122         public void sawOpcode(int seen) {
123             if (reportedThisMethod) {
124                 return;
125             }
126             int oldState = state;
127             if (DEBUG) {
128                 System.out.print("Opcode: ");
129                 printOpCode(seen);
130             }
131     
132             // Keep track of registers that are clobbered and at what PC,
133             // not including stores due to string concatenations
134             int storeTo = -1;
135             switch (seen) {
136             case ASTORE_0:
137                 storeTo = 0;
138                 break;
139             case ASTORE_1:
140                 storeTo = 1;
141                 break;
142             case ASTORE_2:
143                 storeTo = 2;
144                 break;
145             case ASTORE_3:
146                 storeTo = 3;
147                 break;
148             case ASTORE:
149                 storeTo = getRegisterOperand();
150                 break;
151             default:
152                 break;
153             }
154             if (storeTo >= 0 && state != CONSTRUCTED_STRING_ON_STACK) {
155                 clobberedRegisters.put(storeTo, getPC());
156             }
157     
158             switch (state) {
159             case SEEN_NOTHING:
160                 if ((seen == NEW) && getClassConstantOperand().startsWith("java/lang/StringBu")) {
161                     state = SEEN_NEW;
162                     createPC = getPC();
163                 }
164                 break;
165     
166             case SEEN_NEW:
167                 if (seen == INVOKESPECIAL && "<init>".equals(getNameConstantOperand())
168                 && "(Ljava/lang/String;)V".equals(getSigConstantOperand())
169                 && getClassConstantOperand().startsWith("java/lang/StringBu") && registerOnStack >= 0) {
170                     state = SEEN_APPEND1;
171                     stringSource = registerOnStack;
172                 } else if (seen == INVOKEVIRTUAL && "append".equals(getNameConstantOperand())
173                         && getClassConstantOperand().startsWith("java/lang/StringBu")) {
174                     if (DEBUG) {
175                         System.out.println("Saw string being appended from register " + registerOnStack);
176                     }
177                     if (getSigConstantOperand().startsWith("(Ljava/lang/String;)") && registerOnStack >= 0) {
178                         if (DEBUG) {
179                             System.out.println("Saw string being appended, source = " + registerOnStack);
180                         }
181                         state = SEEN_APPEND1;
182                         stringSource = registerOnStack;
183                     } else {
184                         reset();
185                     }
186                 }
187                 break;
188             case SEEN_APPEND1:
189                 if (storeIntoRegister(seen, stringSource)) {
190                     reset();
191                 } else if (seen == INVOKEVIRTUAL && "append".equals(getNameConstantOperand())
192                         && getClassConstantOperand().startsWith("java/lang/StringBu")) {
193                     state = SEEN_APPEND2;
194                 }
195                 break;
196     
197             case SEEN_APPEND2:
198                 if (storeIntoRegister(seen, stringSource)) {
199                     reset();
200                 } else if (seen == INVOKEVIRTUAL && "toString".equals(getNameConstantOperand())
201                         && getClassConstantOperand().startsWith("java/lang/StringBu")) {
202                     state = CONSTRUCTED_STRING_ON_STACK;
203                 }
204                 break;
205     
206             case CONSTRUCTED_STRING_ON_STACK:
207                 if (storeIntoRegister(seen, stringSource)) {
208                     state = POSSIBLE_CASE;
209                 } else {
210                     reset();
211                 }
212                 break;
213     
214             case POSSIBLE_CASE:
215                 // Note: the bottom of a loop is not necessarily a goto;
216                 // one sourceforge bug (Bug1811106) pointed out that for
217                 // do/while loops, it may be a if_icmpge. I generalized
218                 // it to any branch.
219                 if (DismantleBytecode.isBranch(seen) && (getPC() - getBranchTarget()) < 300 && getBranchTarget() <= createPC) {
220     
221                     // Next check: was the destination register clobbered
222                     // elsewhere in this loop?
223                     boolean clobberedInLoop = false;
224                     for (Entry<Integer, Integer> entry : clobberedRegisters.entrySet()) {
225                         int reg = entry.getKey();
226                         if (reg != stringSource) {
227                             continue;
228                         }
229                         int pc = entry.getValue();
230                         if (pc >= getBranchTarget()) {
231                             clobberedInLoop = true;
232                             break;
233                         }
234                     }
235                     if (clobberedInLoop) {
236                         reset();
237                         break;
238                     }
239     
240                     bugReporter.reportBug(new BugInstance(this, "SBSC_USE_STRINGBUFFER_CONCATENATION", NORMAL_PRIORITY)
241                     .addClassAndMethod(this).addSourceLine(this, createPC));
242                     // System.out.println("SBSC spread: " + (getPC() -
243                     // getBranchTarget()));
244                     reset();
245                     reportedThisMethod = true;
246                 } else if ((seen == NEW) && getClassConstantOperand().startsWith("java/lang/StringBu")) {
247                     state = SEEN_NEW;
248                     createPC = getPC();
249                 } else {
250                     if (DEBUG && DismantleBytecode.isBranch(seen)) {
251                         System.out.println("Rejecting branch:");
252                         System.out.println("  spread: " + (getPC() - getBranchTarget()));
253                         System.out.println("  getBranchTarget(): " + getBranchTarget());
254                         System.out.println("  createPC: " + createPC);
255                     }
256                 }
257                 break;
258             default:
259                 break;
260             }
261     
262             if (seen == INVOKESTATIC && "valueOf".equals(getNameConstantOperand())
263                     && "java/lang/String".equals(getClassConstantOperand())
264                     && "(Ljava/lang/Object;)Ljava/lang/String;".equals(getSigConstantOperand())) {
265                 // leave registerOnStack unchanged
266             } else {
267                 registerOnStack = -1;
268                 switch (seen) {
269                 case ALOAD_0:
270                     registerOnStack = 0;
271                     break;
272                 case ALOAD_1:
273                     registerOnStack = 1;
274                     break;
275                 case ALOAD_2:
276                     registerOnStack = 2;
277                     break;
278                 case ALOAD_3:
279                     registerOnStack = 3;
280                     break;
281                 case ALOAD:
282                     registerOnStack = getRegisterOperand();
283                     break;
284                 default:
285                     break;
286                 }
287             }
288             if (DEBUG && state != oldState) {
289                 System.out.println("At PC " + getPC() + " changing from state " + oldState + " to state " + state + ", regOnStack = "
290                         + registerOnStack);
291             }
292         }
293     }
