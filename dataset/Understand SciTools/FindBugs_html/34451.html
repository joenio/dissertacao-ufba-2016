20     package edu.umd.cs.findbugs.ba;
21     
22     import java.util.BitSet;
23     
24     /**
25      * Dataflow value representing the current nesting of catch and finally blocks.
26      * We assume that any catch block with a non-empty catch type is a user catch
27      * block, and any catch block with an empty catch type (i.e., catch all
28      * exceptions) is a finally block. This assumption isn't quite accurate, but it
29      * seems to be a reasonable first approximation.
30      *
31      * <p>
32      * If valid (isValid() returns true), a BlockType value is a stack of elements,
33      * which are either CATCH or FINALLY values. Call getDepth() to get the current
34      * nesting depth. Call get(int <i>n</i>) to get the <i>n</i>th stack item. Call
35      * getTopValue() to get the current top of the stack.
36      * </p>
37      *
38      * <p>
39      * If invalid (isValid() returns false), a BlockType value is either <i>top</i>
40      * or <i>bottom</i>. These are the special values at the top and bottom of the
41      * dataflow lattice.
42      * </p>
43      *
44      * <p>
45      * The dataflow lattice is effectively finite-height because real Java methods
46      * are guaranteed to have a finite catch and finally block nesting level.
47      * </p>
48      *
49      * @see BlockTypeAnalysis
50      * @author David Hovemeyer
51      */
52     public class BlockType extends BitSet {
53         /**
54          *
55          */
56         private static final long serialVersionUID = 1L;
57     
58         public static final boolean CATCH = false;
59     
60         public static final boolean FINALLY = true;
61     
62         private boolean isValid;
63     
64         private boolean isTop;
65     
66         private int depth;
67     
68         /**
69          * Constructor. Should only be called by BlockTypeAnalysis.
70          */
71         BlockType() {
72         }
73     
74         @Override
75         public int hashCode() {
76             final int prime = 31;
77             int result = super.hashCode();
78             result = prime * result + depth;
79             result = prime * result + (isTop ? 1231 : 1237);
80             result = prime * result + (isValid ? 1231 : 1237);
81             return result;
82         }
83     
84         @Override
85         public boolean equals(Object obj) {
86             if (this == obj) {
87                 return true;
88             }
89             if (!super.equals(obj)) {
90                 return false;
91             }
92             if (!(obj instanceof BlockType)) {
93                 return false;
94             }
95             final BlockType other = (BlockType) obj;
96             if (depth != other.depth) {
97                 return false;
98             }
99             if (isTop != other.isTop) {
100                 return false;
101             }
102             if (isValid != other.isValid) {
103                 return false;
104             }
105             return true;
106         }
107     
108         /**
109          * Return whether or not this value is valid, meaning it contains a valid
110          * representation of the nesting of catch and finally blocks.
111          */
112         public boolean isValid() {
113             return isValid;
114         }
115     
116         /**
117          * Get the current nesting depth. The value must be valid.
118          */
119         public int getDepth() {
120             if (!isValid) {
121                 throw new IllegalStateException();
122             }
123             return depth;
124         }
125     
126         /**
127          * Get the top value on the catch and finally block nesting stack.
128          */
129         public boolean getTopValue() {
130             if (depth == 0) {
131                 throw new IllegalStateException();
132             }
133             return get(depth - 1);
134         }
135     
136         /**
137          * Return whether or not this value represents "normal" control-flow. Normal
138          * control flow are all blocks outside any catch or finally block.
139          */
140         public boolean isNormal() {
141             if (!isValid) {
142                 throw new IllegalStateException();
143             }
144             return getDepth() == 0;
145         }
146     
147         /**
148          * Make this value represent "normal" control flow.
149          */
150         public void setNormal() {
151             this.isValid = true;
152             this.depth = 0;
153         }
154     
155         /**
156          * Return whether or not this is the special "top" dataflow value.
157          */
158         public boolean isTop() {
159             return !isValid && isTop;
160         }
161     
162         /**
163          * Make this the special "top" dataflow value.
164          */
165         public void setTop() {
166             this.isValid = false;
167             this.isTop = true;
168         }
169     
170         /**
171          * Return whether or not this is the special "bottom" dataflow value.
172          */
173         public boolean isBottom() {
174             return !isValid && !isTop;
175         }
176     
177         /**
178          * Make this the special "bottom" dataflow value.
179          */
180         public void setBottom() {
181             this.isValid = false;
182             this.isTop = false;
183         }
184     
185         /**
186          * Make this object an exact duplicate of given object.
187          *
188          * @param other
189          *            the other BlockType object
190          */
191         public void copyFrom(BlockType other) {
192             this.isValid = other.isValid;
193             this.isTop = other.isTop;
194             if (isValid) {
195                 this.depth = other.depth;
196                 this.clear();
197                 this.or(other);
198             }
199         }
200     
201         /**
202          * Return whether or not this object is identical to the one given.
203          *
204          * @param other
205          *            the other BlockType object
206          * @return true if this object is identical to the one given, false
207          *         otherwise
208          */
209         public boolean sameAs(BlockType other) {
210             if (!this.isValid) {
211                 return !other.isValid && (this.isTop == other.isTop);
212             } else {
213                 if (!other.isValid) {
214                     return false;
215                 } else {
216                     // Both facts are valid
217                     if (this.depth != other.depth) {
218                         return false;
219                     }
220     
221                     // Compare bits
222                     for (int i = 0; i < this.depth; ++i) {
223                         if (this.get(i) != other.get(i)) {
224                             return false;
225                         }
226                     }
227     
228                     return true;
229                 }
230             }
231         }
232     
233         /**
234          * Merge other dataflow value into this value.
235          *
236          * @param other
237          *            the other BlockType value
238          */
239         public void mergeWith(BlockType other) {
240             if (this.isTop() || other.isBottom()) {
241                 copyFrom(other);
242             } else if (isValid()) {
243                 // To merge, we take the common prefix
244                 int pfxLen = Math.min(this.depth, other.depth);
245                 int commonLen;
246                 for (commonLen = 0; commonLen < pfxLen; ++commonLen) {
247                     if (this.get(commonLen) != other.get(commonLen)) {
248                         break;
249                     }
250                 }
251                 this.depth = commonLen;
252             }
253         }
254     
255         /**
256          * Enter a catch block.
257          */
258         public void pushCatch() {
259             push(CATCH);
260         }
261     
262         /**
263          * Enter a finally block.
264          */
265         public void pushFinally() {
266             push(FINALLY);
267         }
268     
269         @Override
270         public String toString() {
271             if (isTop()) {
272                 return "<top>";
273             } else if (isBottom()) {
274                 return "<bottom>";
275             } else {
276                 StringBuilder buf = new StringBuilder();
277                 buf.append("N");
278                 for (int i = 0; i < depth; ++i) {
279                     buf.append(get(i) == CATCH ? "C" : "F");
280                 }
281                 return buf.toString();
282             }
283         }
284     
285         private void push(boolean value) {
286             set(depth, value);
287             ++depth;
288         }
289     }
