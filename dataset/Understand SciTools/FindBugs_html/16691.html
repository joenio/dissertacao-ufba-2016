162         private String memoized(String s) {
163             if (s == null) {
164                 return s;
165             }
166             String result = cache.get(s);
167             if (result != null) {
168                 return result;
169             }
170             cache.put(s, s);
171             return s;
172         }

176         @Override
177         public void startElement(String uri, String name, String qName, Attributes attributes) throws SAXException {
178             // URI should always be empty.
179             // So, qName is the name of the element.
180     
181             if (discardedElement(qName)) {
182                 nestingOfIgnoredElements++;
183             } else if (nestingOfIgnoredElements > 0) {
184                 // ignore it
185             } else {
186                 // We should be parsing the outer BugCollection element.
187                 if (elementStack.isEmpty() && !qName.equals(topLevelName)) {
188                     throw new SAXException("Invalid top-level element (expected " + topLevelName + ", saw " + qName + ")");
189                 }
190     
191                 if (BUG_COLLECTION.equals(qName)) {
192                     BugCollection bugCollection = this.bugCollection;
193                     assert bugCollection != null;
194                     // Read and set the sequence number.
195                     String version = getOptionalAttribute(attributes, "version");
196                     if (bugCollection instanceof SortedBugCollection) {
197                         bugCollection.setAnalysisVersion(version);
198                     }
199     
200                     // Read and set the sequence number.
201                     String sequence = getOptionalAttribute(attributes, "sequence");
202                     long seqval = parseLong(sequence, 0L);
203                     bugCollection.setSequenceNumber(seqval);
204     
205                     // Read and set timestamp.
206                     String timestamp = getOptionalAttribute(attributes, "timestamp");
207                     long tsval = parseLong(timestamp, -1L);
208                     bugCollection.setTimestamp(tsval);
209                     // Read and set timestamp.
210                     String analysisTimestamp = getOptionalAttribute(attributes, "analysisTimestamp");
211                     if (analysisTimestamp != null) {
212                         bugCollection.setAnalysisTimestamp(parseLong(analysisTimestamp, -1L));
213                     }
214                     String analysisVersion = getOptionalAttribute(attributes, "version");
215                     if (analysisVersion != null) {
216                         bugCollection.setAnalysisVersion(analysisVersion);
217                     }
218     
219                     // Set release name, if present.
220                     String releaseName = getOptionalAttribute(attributes, "release");
221                     bugCollection.setReleaseName((releaseName != null) ? releaseName : "");
222                 } else if (isTopLevelFilter(qName)) {
223                     if (project != null) {
224                         filter = new Filter();
225                         project.setSuppressionFilter(filter);
226                     }
227                     matcherStack.clear();
228                     pushCompoundMatcher(filter);
229                 } else if (PROJECT.equals(qName)) {
230                     Project project = this.project;
231                     assert project != null;
232                     // Project element
233                     String projectName = getOptionalAttribute(attributes, Project.PROJECTNAME_ATTRIBUTE_NAME);
234                     if (projectName != null) {
235                         project.setProjectName(projectName);
236                     }
237                 } else {
238                     String outerElement = elementStack.get(elementStack.size() - 1);
239                     if (BUG_COLLECTION.equals(outerElement)) {
240     
241                         // Parsing a top-level element of the BugCollection
242                         if ("BugInstance".equals(qName)) {
243                             // BugInstance element - get required type and priority
244                             // attributes
245                             String type = getRequiredAttribute(attributes, "type", qName);
246                             String priority = getRequiredAttribute(attributes, "priority", qName);
247     
248                             try {
249                                 int prio = Integer.parseInt(priority);
250                                 bugInstance = new BugInstance(type, prio);
251                             } catch (NumberFormatException e) {
252                                 throw new SAXException("BugInstance with invalid priority value \"" + priority + "\"", e);
253                             }
254     
255                             String firstVersion = getOptionalAttribute(attributes, "first");
256                             if (firstVersion != null) {
257                                 bugInstance.setFirstVersion(Long.parseLong(firstVersion));
258                             }
259                             String lastVersion = getOptionalAttribute(attributes, "last");
260                             if (lastVersion != null) {
261                                 bugInstance.setLastVersion(Long.parseLong(lastVersion));
262                             }
263     
264                             if (bugInstance.isDead() && bugInstance.getFirstVersion() > bugInstance.getLastVersion()) {
265                                 throw new IllegalStateException("huh");
266                             }
267     
268                             String introducedByChange = getOptionalAttribute(attributes, "introducedByChange");
269                             if (introducedByChange != null) {
270                                 bugInstance.setIntroducedByChangeOfExistingClass(Boolean.parseBoolean(introducedByChange));
271                             }
272                             String removedByChange = getOptionalAttribute(attributes, "removedByChange");
273                             if (removedByChange != null) {
274                                 bugInstance.setRemovedByChangeOfPersistingClass(Boolean.parseBoolean(removedByChange));
275                             }
276                             String oldInstanceHash = getOptionalAttribute(attributes, "instanceHash");
277                             if (oldInstanceHash == null) {
278                                 oldInstanceHash = getOptionalAttribute(attributes, "oldInstanceHash");
279                             }
280                             if (oldInstanceHash != null) {
281                                 bugInstance.setOldInstanceHash(oldInstanceHash);
282                             }
283     
284                             String firstSeen = getOptionalAttribute(attributes, "firstSeen");
285                             if (firstSeen != null) {
286                                 try {
287                                     bugInstance.getXmlProps().setFirstSeen(BugInstance.firstSeenXMLFormat().parse(firstSeen));
288                                 } catch (ParseException e) {
289                                     LOGGER.warning("Could not parse first seen entry: " + firstSeen);
290                                     // ignore
291                                 }
292                             }
293     
294                             String isInCloud = getOptionalAttribute(attributes, "isInCloud");
295                             if (isInCloud != null) {
296                                 bugInstance.getXmlProps().setIsInCloud(Boolean.parseBoolean(isInCloud));
297                             }
298     
299                             String reviewCount = getOptionalAttribute(attributes, "reviews");
300                             if (reviewCount != null) {
301                                 bugInstance.getXmlProps().setReviewCount(Integer.parseInt(reviewCount));
302                             }
303     
304                             String consensus = getOptionalAttribute(attributes, "consensus");
305                             if (consensus != null) {
306                                 bugInstance.getXmlProps().setConsensus(consensus);
307                             }
308     
309                         } else if ("FindBugsSummary".equals(qName)) {
310                             BugCollection bugCollection = this.bugCollection;
311                             assert bugCollection != null;
312                             String timestamp = getRequiredAttribute(attributes, "timestamp", qName);
313                             String vmVersion = getOptionalAttribute(attributes, "vm_version");
314                             String totalClasses = getOptionalAttribute(attributes, "total_classes");
315                             if (totalClasses != null && totalClasses.length() > 0) {
316                                 bugCollection.getProjectStats().setTotalClasses(Integer.parseInt(totalClasses));
317                             }
318     
319                             String totalSize = getOptionalAttribute(attributes, "total_size");
320                             if (totalSize != null && totalSize.length() > 0) {
321                                 bugCollection.getProjectStats().setTotalSize(Integer.parseInt(totalSize));
322                             }
323     
324                             String referencedClasses = getOptionalAttribute(attributes, "referenced_classes");
325                             if (referencedClasses != null && referencedClasses.length() > 0) {
326                                 bugCollection.getProjectStats().setReferencedClasses(Integer.parseInt(referencedClasses));
327                             }
328                             bugCollection.getProjectStats().setVMVersion(vmVersion);
329                             try {
330                                 bugCollection.getProjectStats().setTimestamp(timestamp);
331                             } catch (java.text.ParseException e) {
332                                 throw new SAXException("Unparseable sequence number: '" + timestamp + "'", e);
333                             }
334                         }
335                     } else if ("BugInstance".equals(outerElement)) {
336                         parseBugInstanceContents(qName, attributes);
337                     } else if ("Method".equals(outerElement) || "Field".equals(outerElement) || "Class".equals(outerElement)
338                             || "Type".equals(outerElement)) {
339                         if ("SourceLine".equals(qName)) {
340                             // package member elements can contain nested SourceLine
341                             // elements.
342                             bugAnnotationWithSourceLines.setSourceLines(createSourceLineAnnotation(qName, attributes));
343                         }
344                     } else if (BugCollection.ERRORS_ELEMENT_NAME.equals(outerElement)) {
345                         if (BugCollection.ANALYSIS_ERROR_ELEMENT_NAME.equals(qName) || BugCollection.ERROR_ELEMENT_NAME.equals(qName)) {
346                             analysisError = new AnalysisError("Unknown error");
347                             stackTrace.clear();
348                         }
349                     } else if ("FindBugsSummary".equals(outerElement) && "PackageStats".equals(qName)) {
350                         BugCollection bugCollection = this.bugCollection;
351                         assert bugCollection != null;
352                         String packageName = getRequiredAttribute(attributes, "package", qName);
353                         int numClasses = Integer.parseInt(getRequiredAttribute(attributes, "total_types", qName));
354                         int size = Integer.parseInt(getRequiredAttribute(attributes, "total_size", qName));
355                         bugCollection.getProjectStats().putPackageStats(packageName, numClasses, size);
356     
357                     } else if ("PackageStats".equals(outerElement)) {
358                         BugCollection bugCollection = this.bugCollection;
359                         assert bugCollection != null;
360                         if ("ClassStats".equals(qName)) {
361                             String className = getRequiredAttribute(attributes, "class", qName);
362                             Boolean isInterface = Boolean.valueOf(getRequiredAttribute(attributes, "interface", qName));
363                             int size = Integer.parseInt(getRequiredAttribute(attributes, "size", qName));
364                             String sourceFile = getOptionalAttribute(attributes, "sourceFile");
365                             bugCollection.getProjectStats().addClass(className, sourceFile, isInterface, size, false);
366                         }
367     
368                     } else if (isTopLevelFilter(outerElement) || isCompoundElementTag(outerElement)) {
369                         parseMatcher(qName, attributes);
370                     } else if ("ClassFeatures".equals(outerElement)) {
371                         if (ClassFeatureSet.ELEMENT_NAME.equals(qName)) {
372                             String className = getRequiredAttribute(attributes, "class", qName);
373                             classFeatureSet = new ClassFeatureSet();
374                             classFeatureSet.setClassName(className);
375                         }
376                     } else if (ClassFeatureSet.ELEMENT_NAME.equals(outerElement)) {
377                         if (ClassFeatureSet.FEATURE_ELEMENT_NAME.equals(qName)) {
378                             String value = getRequiredAttribute(attributes, "value", qName);
379                             classFeatureSet.addFeature(value);
380                         }
381                     } else if (BugCollection.HISTORY_ELEMENT_NAME.equals(outerElement)) {
382                         if (AppVersion.ELEMENT_NAME.equals(qName)) {
383                             BugCollection bugCollection = this.bugCollection;
384                             assert bugCollection != null;
385     
386                             try {
387                                 String sequence = getRequiredAttribute(attributes, "sequence", qName);
388                                 String timestamp = getOptionalAttribute(attributes, "timestamp");
389                                 String releaseName = getOptionalAttribute(attributes, "release");
390                                 String codeSize = getOptionalAttribute(attributes, "codeSize");
391                                 String numClasses = getOptionalAttribute(attributes, "numClasses");
392                                 AppVersion appVersion = new AppVersion(Long.parseLong(sequence));
393                                 if (timestamp != null) {
394                                     appVersion.setTimestamp(Long.parseLong(timestamp));
395                                 }
396                                 if (releaseName != null) {
397                                     appVersion.setReleaseName(releaseName);
398                                 }
399                                 if (codeSize != null) {
400                                     appVersion.setCodeSize(Integer.parseInt(codeSize));
401                                 }
402                                 if (numClasses != null) {
403                                     appVersion.setNumClasses(Integer.parseInt(numClasses));
404                                 }
405     
406                                 bugCollection.addAppVersion(appVersion);
407                             } catch (NumberFormatException e) {
408                                 throw new SAXException("Invalid AppVersion element", e);
409                             }
410                         }
411                     } else if (BugCollection.PROJECT_ELEMENT_NAME.equals(outerElement)) {
412                         Project project = this.project;
413                         assert project != null;
414                         if (Project.CLOUD_ELEMENT_NAME.equals(qName)) {
415                             String cloudId = getRequiredAttribute(attributes, Project.CLOUD_ID_ATTRIBUTE_NAME, qName);
416                             project.setCloudId(cloudId);
417                             if(bugCollection != null){
418                                 Map<String,String> map = new HashMap<String, String>();
419                                 for (int i = 0; i < attributes.getLength(); i++) {
420                                     map.put(attributes.getLocalName(i), attributes.getValue(i));
421                                 }
422                                 bugCollection.setXmlCloudDetails(Collections.unmodifiableMap(map));
423                             }
424                         } else if (Project.PLUGIN_ELEMENT_NAME.equals(qName)) {
425                             String pluginId = getRequiredAttribute(attributes, Project.PLUGIN_ID_ATTRIBUTE_NAME, qName);
426                             Boolean enabled = Boolean.valueOf(getRequiredAttribute(attributes, Project.PLUGIN_STATUS_ELEMENT_NAME, qName));
427                             project.setPluginStatusTrinary(pluginId, enabled);
428                         }
429     
430                     } else if (Project.CLOUD_ELEMENT_NAME.equals(outerElement)) {
431                         if (Project.CLOUD_PROPERTY_ELEMENT_NAME.equals(qName)) {
432                             cloudPropertyKey = getRequiredAttribute(attributes, "key", qName);
433                         }
434     
435                     }
436                 }
437             }
438     
439             textBuffer.delete(0, textBuffer.length());
440             elementStack.add(qName);
441         }

451         private void addMatcher(Matcher m) {
452             if (m == null) {
453                 throw new IllegalArgumentException("matcher must not be null");
454             }
455     
456             CompoundMatcher peek = matcherStack.peek();
457             if (peek == null) {
458                 throw new NullPointerException("Top of stack is null");
459             }
460             peek.addChild(m);
461             if (nextMatchedIsDisabled) {
462                 if (peek instanceof Filter) {
463                     ((Filter) peek).disable(m);
464                 } else {
465                     assert false;
466                 }
467                 nextMatchedIsDisabled = false;
468             }
469         }
470     
471         private void pushCompoundMatcherAsChild(CompoundMatcher m) {
472             addMatcher(m);
473             pushCompoundMatcher(m);
474         }
475     
476         private void pushCompoundMatcher(CompoundMatcher m) {
477             if (m == null) {
478                 throw new IllegalArgumentException("matcher must not be null");
479             }
480             matcherStack.push(m);
481         }

483         boolean nextMatchedIsDisabled;
484         private final Set<String> outerElementTags = unmodifiableSet(new HashSet<String>(asList("And", "Match", "Or", "Not")));
485     
486         private void parseMatcher(String qName, Attributes attributes) throws SAXException {
487             if (DEBUG) {
488                 System.out.println(elementStack + " " + qName + " " + matcherStack);
489             }
490             String disabled = getOptionalAttribute(attributes, "disabled");
491             nextMatchedIsDisabled = "true".equals(disabled);
492             if ("Bug".equals(qName)) {
493                 addMatcher(new BugMatcher(getOptionalAttribute(attributes, "code"), getOptionalAttribute(attributes, "pattern"),
494                         getOptionalAttribute(attributes, "category")));
495             } else if ("Class".equals(qName)) {
496                 String role = getOptionalAttribute(attributes, "role");
497                 addMatcher(new ClassMatcher(getRequiredAttribute(attributes, "name", qName), role));
498             } else if ("Type".equals(qName)) {
499                 String role = getOptionalAttribute(attributes, "role");
500                 String typeParameters = getOptionalAttribute(attributes, "typeParameters");
501                 addMatcher(new TypeMatcher(getRequiredAttribute(attributes, "descriptor", qName), role, typeParameters));
502             } else if ("FirstVersion".equals(qName)) {
503                 addMatcher(new FirstVersionMatcher(getRequiredAttribute(attributes, "value", qName), getRequiredAttribute(attributes,
504                         "relOp", qName)));
505             } else if ("LastVersion".equals(qName)) {
506                 addMatcher(new LastVersionMatcher(getRequiredAttribute(attributes, "value", qName), getRequiredAttribute(attributes,
507                         "relOp", qName)));
508             } else if ("Designation".equals(qName)) {
509                 addMatcher(new DesignationMatcher(getRequiredAttribute(attributes, "designation", qName)));
510             } else if ("BugCode".equals(qName)) {
511                 addMatcher(new BugMatcher(getRequiredAttribute(attributes, "name", qName), "", ""));
512             } else if ("Local".equals(qName)) {
513                 addMatcher(new LocalMatcher(getRequiredAttribute(attributes, "name", qName)));
514             } else if ("BugPattern".equals(qName)) {
515                 addMatcher(new BugMatcher("", getRequiredAttribute(attributes, "name", qName), ""));
516             } else if ("Priority".equals(qName)) {
517                 addMatcher(new PriorityMatcher(getRequiredAttribute(attributes, "value", qName)));
518             } else if ("Confidence".equals(qName)) {
519                 addMatcher(new ConfidenceMatcher(getRequiredAttribute(attributes, "value", qName)));
520             } else if ("Rank".equals(qName)) {
521                 addMatcher(new RankMatcher(getRequiredAttribute(attributes, "value", qName)));
522             } else if ("Package".equals(qName)) {
523                 String pName = getRequiredAttribute(attributes, "name", qName);
524                 pName = pName.startsWith("~") ? pName : "~" + pName.replace(".", "\\.");
525                 addMatcher(new ClassMatcher(pName + "\\.[^.]+"));
526             } else if ("Method".equals(qName)) {
527                 String name = getOptionalAttribute(attributes, "name");
528                 String params = getOptionalAttribute(attributes, "params");
529                 String returns = getOptionalAttribute(attributes, "returns");
530                 String role = getOptionalAttribute(attributes, "role");
531                 addMatcher(new MethodMatcher(name, params, returns, role));
532             } else if ("Field".equals(qName)) {
533                 String name = getOptionalAttribute(attributes, "name");
534                 String type = getOptionalAttribute(attributes, "type");
535                 String role = getOptionalAttribute(attributes, "role");
536                 addMatcher(new FieldMatcher(name, type, role));
537             } else if ("Or".equals(qName)) {
538                 CompoundMatcher matcher = new OrMatcher();
539                 pushCompoundMatcherAsChild(matcher);
540             } else if ("And".equals(qName) || "Match".equals(qName)) {
541                 AndMatcher matcher = new AndMatcher();
542                 pushCompoundMatcherAsChild(matcher);
543                 if ("Match".equals(qName)) {
544                     String classregex = getOptionalAttribute(attributes, "classregex");
545                     String classMatch = getOptionalAttribute(attributes, "class");
546     
547                     if (classregex != null) {
548                         addMatcher(new ClassMatcher("~" + classregex));
549                     } else if (classMatch != null) {
550                         addMatcher(new ClassMatcher(classMatch));
551                     }
552                 }
553             } else if("Not".equals(qName)) {
554                 NotMatcher matcher = new NotMatcher();
555                 pushCompoundMatcherAsChild(matcher);
556             } else if ("Source".equals(qName)) {
557                 addMatcher(new SourceMatcher(getRequiredAttribute(attributes, "name", qName)));
558             }
559             nextMatchedIsDisabled = false;
560         }

562         private void parseBugInstanceContents(String qName, Attributes attributes) throws SAXException {
563             // Parsing an attribute or property of a BugInstance
564             BugAnnotation bugAnnotation = null;
565             if ("Class".equals(qName)) {
566                 String className = getRequiredAttribute(attributes, "classname", qName);
567                 bugAnnotation = bugAnnotationWithSourceLines = new ClassAnnotation(className);
568             } else if ("Type".equals(qName)) {
569                 String typeDescriptor = getRequiredAttribute(attributes, "descriptor", qName);
570                 TypeAnnotation typeAnnotation;
571                 bugAnnotation = bugAnnotationWithSourceLines = typeAnnotation = new TypeAnnotation(typeDescriptor);
572                 String typeParameters = getOptionalAttribute(attributes, "typeParameters");
573                 if (typeParameters != null) {
574                     typeAnnotation.setTypeParameters(Strings.unescapeXml(typeParameters));
575                 }
576     
577             } else if ("Method".equals(qName) || "Field".equals(qName)) {
578                 String classname = getRequiredAttribute(attributes, "classname", qName);
579                 String fieldOrMethodName = getRequiredAttribute(attributes, "name", qName);
580                 String signature = getRequiredAttribute(attributes, "signature", qName);
581                 if ("Method".equals(qName)) {
582                     String isStatic = getOptionalAttribute(attributes, "isStatic");
583                     if (isStatic == null) {
584                         isStatic = "false"; // Hack for old data
585                     }
586     
587                     bugAnnotation = bugAnnotationWithSourceLines = new MethodAnnotation(classname, fieldOrMethodName, signature,
588                             Boolean.valueOf(isStatic));
589     
590                 } else {
591                     String isStatic = getRequiredAttribute(attributes, "isStatic", qName);
592                     String sourceSignature = getOptionalAttribute(attributes, "sourceSignature");
593                     bugAnnotation = bugAnnotationWithSourceLines = new FieldAnnotation(classname, fieldOrMethodName, signature,
594                             sourceSignature, Boolean.valueOf(isStatic));
595                 }
596     
597             } else if ("SourceLine".equals(qName)) {
598                 SourceLineAnnotation sourceAnnotation = createSourceLineAnnotation(qName, attributes);
599                 if (!sourceAnnotation.isSynthetic()) {
600                     bugAnnotation = sourceAnnotation;
601                 }
602             } else if ("Int".equals(qName)) {
603                 try {
604                     String value = getRequiredAttribute(attributes, "value", qName);
605                     bugAnnotation = new IntAnnotation(Integer.parseInt(value));
606                 } catch (NumberFormatException e) {
607                     throw new SAXException("Bad integer value in Int");
608                 }
609             } else if ("String".equals(qName)) {
610                 String value = getRequiredAttribute(attributes, "value", qName);
611                 bugAnnotation = StringAnnotation.fromXMLEscapedString(value);
612             } else if ("LocalVariable".equals(qName)) {
613                 try {
614                     String varName = getRequiredAttribute(attributes, "name", qName);
615                     int register = Integer.parseInt(getRequiredAttribute(attributes, "register", qName));
616                     int pc = Integer.parseInt(getRequiredAttribute(attributes, "pc", qName));
617                     bugAnnotation = new LocalVariableAnnotation(varName, register, pc);
618                 } catch (NumberFormatException e) {
619                     throw new SAXException("Invalid integer value in attribute of LocalVariable element");
620                 }
621             } else if ("Property".equals(qName)) {
622                 // A BugProperty.
623                 String propName = getRequiredAttribute(attributes, "name", qName);
624                 String propValue = getRequiredAttribute(attributes, "value", qName);
625                 bugInstance.setProperty(propName, propValue);
626             } else if ("UserAnnotation".equals(qName)) {
627                 // ignore AnnotationText for now; will handle in endElement
628                 String s = getOptionalAttribute(attributes, "designation"); // optional
629                 if (s != null) {
630                     bugInstance.setUserDesignationKey(s, null);
631                 }
632                 s = getOptionalAttribute(attributes, "user"); // optional
633                 if (s != null) {
634                     bugInstance.setUser(s);
635                 }
636                 s = getOptionalAttribute(attributes, "timestamp"); // optional
637                 if (s != null) {
638                     try {
639                         long timestamp = Long.parseLong(s);
640                         bugInstance.setUserAnnotationTimestamp(timestamp);
641                     } catch (NumberFormatException nfe) {
642                         // ok to contine -- just won't set a timestamp for the user
643                         // designation.
644                         // but is there anyplace to report this?
645                     }
646                 }
647                 s = getOptionalAttribute(attributes, "needsSync"); // optional
648                 if (s == null || "false".equals(s)) {
649                     bugInstance.setUserAnnotationDirty(false);
650                 }
651     
652             } else {
653                 throw new SAXException("Unknown bug annotation named " + qName);
654             }
655     
656             if (bugAnnotation != null) {
657                 setAnnotationRole(attributes, bugAnnotation);
658                 bugInstance.add(bugAnnotation);
659             }
660         }

663             long value;
664             try {
665                 value = (s != null) ? Long.parseLong(s) : defaultValue;
666             } catch (NumberFormatException e) {
667                 value = defaultValue;
668             }
669             return value;
670         }

701         private SourceLineAnnotation createSourceLineAnnotation(String qName, Attributes attributes) throws SAXException {
702             String classname = getRequiredAttribute(attributes, "classname", qName);
703             String sourceFile = getOptionalAttribute(attributes, "sourcefile");
704             if (sourceFile == null) {
705                 sourceFile = SourceLineAnnotation.UNKNOWN_SOURCE_FILE;
706             }
707             String startLine = getOptionalAttribute(attributes, "start"); // "start"/"end"
708             // are now
709             // optional
710             String endLine = getOptionalAttribute(attributes, "end"); // (were too
711             // many "-1"s
712             // in the xml)
713             String startBytecode = getOptionalAttribute(attributes, "startBytecode");
714             String endBytecode = getOptionalAttribute(attributes, "endBytecode");
715             String synthetic = getOptionalAttribute(attributes, "synthetic");
716     
717             try {
718                 int sl = startLine != null ? Integer.parseInt(startLine) : -1;
719                 int el = endLine != null ? Integer.parseInt(endLine) : -1;
720                 int sb = startBytecode != null ? Integer.parseInt(startBytecode) : -1;
721                 int eb = endBytecode != null ? Integer.parseInt(endBytecode) : -1;
722     
723                 SourceLineAnnotation s = new SourceLineAnnotation(classname, sourceFile, sl, el, sb, eb);
724                 if ("true".equals(synthetic)) {
725                     s.setSynthetic(true);
726                 }
727                 return s;
728             } catch (NumberFormatException e) {
729                 throw new SAXException("Bad integer value in SourceLine element", e);
730             }
731         }

733         @Override
734         public void endElement(String uri, String name, String qName) throws SAXException {
735             // URI should always be empty.
736             // So, qName is the name of the element.
737     
738             if (discardedElement(qName)) {
739                 nestingOfIgnoredElements--;
740             } else if (nestingOfIgnoredElements > 0) {
741                 // ignore it
742             } else if ("Project".equals(qName)) {
743                 // noop
744             } else if (elementStack.size() > 1) {
745                 String outerElement = elementStack.get(elementStack.size() - 2);
746     
747                 if (isTopLevelFilter(qName) || isCompoundElementTag(qName)) {
748                     if (DEBUG) {
749                         System.out.println("  ending " + elementStack + " " + qName + " " + matcherStack);
750                     }
751     
752                     matcherStack.pop();
753                 } else if (BUG_COLLECTION.equals(outerElement)) {
754                     BugCollection bugCollection = this.bugCollection;
755                     assert bugCollection != null;
756                     if ("BugInstance".equals(qName)) {
757                         bugCollection.add(bugInstance, false);
758                     }
759                 } else if (PROJECT.equals(outerElement)) {
760                     Project project = this.project;
761                     assert project != null;
762                     if ("Jar".equals(qName)) {
763                         project.addFile(makeAbsolute(getTextContents()));
764                     } else if ("SrcDir".equals(qName)) {
765                         project.addSourceDir(makeAbsolute(getTextContents()));
766                     } else if ("AuxClasspathEntry".equals(qName)) {
767                         project.addAuxClasspathEntry(makeAbsolute(getTextContents()));
768                     }
769     
770     
771     
772                 } else if (Project.CLOUD_ELEMENT_NAME.equals(outerElement) && Project.CLOUD_PROPERTY_ELEMENT_NAME.equals(qName)) {
773                     Project project = this.project;
774                     assert project != null;
775                     assert cloudPropertyKey != null;
776                     project.getCloudProperties().setProperty(cloudPropertyKey, getTextContents());
777                     cloudPropertyKey = null;
778                 } else if ("BugInstance".equals(outerElement)) {
779                     if ("UserAnnotation".equals(qName)) {
780                         bugInstance.setAnnotationText(getTextContents(), null);
781                     }
782                 } else if (BugCollection.ERRORS_ELEMENT_NAME.equals(outerElement)) {
783                     BugCollection bugCollection = this.bugCollection;
784                     assert bugCollection != null;
785                     if (BugCollection.ANALYSIS_ERROR_ELEMENT_NAME.equals(qName)) {
786                         analysisError.setMessage(getTextContents());
787                         bugCollection.addError(analysisError);
788                     } else if (BugCollection.ERROR_ELEMENT_NAME.equals(qName)) {
789                         if (stackTrace.size() > 0) {
790                             analysisError.setStackTrace(stackTrace.toArray(new String[stackTrace.size()]));
791                         }
792                         bugCollection.addError(analysisError);
793                     } else if (BugCollection.MISSING_CLASS_ELEMENT_NAME.equals(qName)) {
794                         bugCollection.addMissingClass(getTextContents());
795                     }
796     
797                 } else if (BugCollection.ERROR_ELEMENT_NAME.equals(outerElement)) {
798                     if (BugCollection.ERROR_MESSAGE_ELEMENT_NAME.equals(qName)) {
799                         analysisError.setMessage(getTextContents());
800                     } else if (BugCollection.ERROR_EXCEPTION_ELEMENT_NAME.equals(qName)) {
801                         analysisError.setExceptionMessage(getTextContents());
802                     } else if (BugCollection.ERROR_STACK_TRACE_ELEMENT_NAME.equals(qName)) {
803                         stackTrace.add(getTextContents());
804                     }
805                 } else if ("ClassFeatures".equals(outerElement)) {
806                     if (ClassFeatureSet.ELEMENT_NAME.equals(qName)) {
807                         BugCollection bugCollection = this.bugCollection;
808                         assert bugCollection != null;
809     
810                         bugCollection.setClassFeatureSet(classFeatureSet);
811                         classFeatureSet = null;
812                     }
813                 }
814             }
815     
816             elementStack.remove(elementStack.size() - 1);
817         }

819         private String makeAbsolute(String possiblyRelativePath) {
820             if (possiblyRelativePath.contains("://") || possiblyRelativePath.startsWith("http:")
821                     || possiblyRelativePath.startsWith("https:") || possiblyRelativePath.startsWith("file:")) {
822                 return possiblyRelativePath;
823             }
824             if (base == null) {
825                 return possiblyRelativePath;
826             }
827             if (new File(possiblyRelativePath).isAbsolute()) {
828                 return possiblyRelativePath;
829             }
830     
831             return new File(base.getParentFile(), possiblyRelativePath).getAbsolutePath();
832         }

839         private String getRequiredAttribute(Attributes attributes, String attrName, String elementName) throws SAXException {
