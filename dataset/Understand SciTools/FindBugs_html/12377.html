58     /**
59      * A cache for looking up the collection of ObligationPolicyDatabaseActions
60      * associated with a given InstructionHandle. Avoids the need for repeated
61      * (slow) lookups.
62      *
63      * @author David Hovemeyer
64      */
65     public class InstructionActionCache {
66         private static final boolean DEBUG_LOOKUP = SystemProperties.getBoolean("oa.debug.lookup");
67     
68         private final ObligationPolicyDatabase database;
69     
70         private final Map<InstructionHandle, Collection<ObligationPolicyDatabaseAction>> actionCache;
71     
72         private final XMethod xmethod;
73         private final TypeDataflow typeDataflow;
74         private final ConstantPoolGen cpg;
75     
76     
77     
78         public InstructionActionCache(ObligationPolicyDatabase database, XMethod xmethod, ConstantPoolGen cpg, TypeDataflow typeDataflow) {
79             this.database = database;
80             this.actionCache = new HashMap<InstructionHandle, Collection<ObligationPolicyDatabaseAction>>();
81             this.xmethod = xmethod;
82             this.cpg = cpg;
83             this.typeDataflow = typeDataflow;
84         }
85     
86         static final ClassDescriptor WILL_CLOSE = DescriptorFactory.createClassDescriptor(WillClose.class);
87         public Collection<ObligationPolicyDatabaseAction> getActions(BasicBlock block, InstructionHandle handle) {
88             Collection<ObligationPolicyDatabaseAction> actionList = actionCache.get(handle);
89             if (actionList == null) {
90                 Instruction ins = handle.getInstruction();
91                 actionList = Collections.emptyList();
92                 if (ins instanceof InvokeInstruction) {
93     
94                     InvokeInstruction inv = (InvokeInstruction) ins;
95                     XMethod invokedMethod = XFactory.createXMethod(inv, cpg);
96                     String signature = invokedMethod.getSignature();
97                     String methodName = invokedMethod.getName();
98     
99                     if (DEBUG_LOOKUP) {
100                         System.out.println("Looking up actions for call to " + invokedMethod);
101                     }
102     
103     
104                     if (invokedMethod.getAnnotationDescriptors().contains(WILL_CLOSE) && methodName.startsWith("close") && signature.endsWith(")V")) {
105                         actionList = Collections.singletonList(ObligationPolicyDatabaseAction.CLEAR);
106                     } else if (signature.indexOf(';') >= -1) {
107                         ReferenceType receiverType = inv.getReferenceType(cpg);
108     
109                         boolean isStatic = inv.getOpcode() == Constants.INVOKESTATIC;
110                         actionList = new LinkedList<ObligationPolicyDatabaseAction>();
111     
112                         database.getActions(receiverType, methodName, signature, isStatic, actionList);
113     
114                         if (actionList.isEmpty()) {
115     
116                             try {
117                                 TypeFrame factAtLocation = null;
118                                 SignatureParser sigParser = new SignatureParser(signature);
119                                 //                        int startIndex = 0;
120                                 //                        if (!xmethod.isStatic())
121                                 //                            startIndex = 1;
122                                 Iterator<String> signatureIterator = sigParser.parameterSignatureIterator();
123                                 int parameters = sigParser.getNumParameters();
124                                 for (int i = 0; i < parameters; i++) {
125                                     String sig = signatureIterator.next();
126                                     Collection<ClassDescriptor> annotations = invokedMethod.getParameterAnnotationDescriptors(i);
127                                     if (annotations.contains(WILL_CLOSE) || "Ljava/io/Closeable;".equals(sig) || methodName.startsWith("close")) {
128                                         // closing this value
129                                         if (factAtLocation == null) {
130                                             factAtLocation = typeDataflow.getFactAtLocation( new Location(handle, block));
131                                         }
132     
133                                         Type argumentType = factAtLocation.getArgument(inv, cpg, i, sigParser);
134                                         if (argumentType instanceof ObjectType) {
135                                             Obligation obligation = database.getFactory().getObligationByType((ObjectType) argumentType);
136                                             if (obligation != null) {
137                                                 actionList.add(new ObligationPolicyDatabaseAction(ObligationPolicyDatabaseActionType.DEL, obligation));
138                                             }
139     
140                                         }
141     
142                                     }
143                                 }
144     
145                             } catch (CheckedAnalysisException e) {
146                                 AnalysisContext.logError("Error checking " + invokedMethod, e);
147                             } catch (ClassNotFoundException e) {
148                                 AnalysisContext.reportMissingClass(e);
149                             } finally { }
150     
151     
152                         }
153                         if (DEBUG_LOOKUP && !actionList.isEmpty()) {
154                             System.out.println("  At " + handle + ": " + actionList);
155                         }
156                     }
157                 } else if (ins instanceof PUTFIELD || ins instanceof PUTSTATIC || ins instanceof ARETURN) {
158                     Location loc = new Location(handle, block);
159                     try {
160                         TypeFrame typeFrame = typeDataflow.getFactAtLocation(loc);
161                         if (typeFrame.isValid()) {
162                             Type tosType = typeFrame.getTopValue();
163                             if (tosType instanceof ObjectType) {
164                                 ObligationFactory factory = database.getFactory();
165                                 Obligation obligation = factory.getObligationByType((ObjectType) tosType);
166                                 if (obligation != null) {
167                                     if ("java.sql.ResultSet".equals(obligation.getClassName())) {
168                                         ObjectType sType = ObjectTypeFactory.getInstance(java.sql.Statement.class);
169                                         Obligation sObligation = factory.getObligationByType(sType);
170                                         actionList = Arrays.asList(
171                                                 new ObligationPolicyDatabaseAction(ObligationPolicyDatabaseActionType.DEL, obligation),
172                                                 new ObligationPolicyDatabaseAction(ObligationPolicyDatabaseActionType.DEL, sObligation));
173                                     } else {
174                                         actionList = Collections.singleton(new ObligationPolicyDatabaseAction(ObligationPolicyDatabaseActionType.DEL,
175                                                 obligation));
176                                     }
177     
178                                 }
179                             }
180                         }
181                     } catch (ClassNotFoundException e) {
182                         AnalysisContext.reportMissingClass(e);
183                     } catch (Exception e) {
184                         AnalysisContext.logError("Error in checking obligation analysis for " + xmethod, e);
185                     }
186     
187                 }
188     
189                 actionCache.put(handle, actionList);
190             }
191     
192             return actionList;
193         }
194     
195         public boolean addsObligation(BasicBlock block,InstructionHandle handle,  Obligation obligation) {
196             return hasAction(block, handle, obligation, ObligationPolicyDatabaseActionType.ADD);
197         }
198     
199         public boolean deletesObligation(BasicBlock block,InstructionHandle handle, Obligation obligation) {
200             return hasAction(block, handle, obligation, ObligationPolicyDatabaseActionType.DEL);
201         }
202     
203         private boolean hasAction(BasicBlock block, InstructionHandle handle,Obligation obligation,
204                 ObligationPolicyDatabaseActionType actionType) {
205             Collection<ObligationPolicyDatabaseAction> actionList = getActions(block, handle);
206             for (ObligationPolicyDatabaseAction action : actionList) {
207                 if (action.getActionType() == actionType && action.getObligation().equals(obligation)) {
208                     return true;
209                 }
210             }
211             return false;
212         }
