20     package edu.umd.cs.findbugs.detect;
21     
22     import java.util.HashSet;
23     import java.util.Set;
24     
25     import org.apache.bcel.classfile.Code;
26     
27     import edu.umd.cs.findbugs.BugInstance;
28     import edu.umd.cs.findbugs.BugReporter;
29     import edu.umd.cs.findbugs.BytecodeScanningDetector;
30     import edu.umd.cs.findbugs.Priorities;
31     import edu.umd.cs.findbugs.ba.AnalysisContext;
32     import edu.umd.cs.findbugs.ba.ClassContext;
33     import edu.umd.cs.findbugs.ba.XField;
34     import edu.umd.cs.findbugs.ba.ch.Subtypes2;
35     
36     public class VolatileUsage extends BytecodeScanningDetector {
37         enum IncrementState {
38             START, GETFIELD, LOADCONSTANT, ADD
39         }
40     
41         private final BugReporter bugReporter;
42     
43         public VolatileUsage(BugReporter bugReporter) {
44             this.bugReporter = bugReporter;
45         }
46     
47         @Override
48         public void visitClassContext(ClassContext classContext) {
49             classContext.getJavaClass().accept(this);
50         }
51     
52         Set<XField> initializationWrites = new HashSet<XField>();
53     
54         Set<XField> otherWrites = new HashSet<XField>();
55     
56         IncrementState state = IncrementState.START;
57     
58         XField incrementField;
59     
60         @Override
61         public void visit(Code obj) {
62             resetIncrementState();
63             super.visit(obj);
64         }
65     
66         @Override
67         public void sawOpcode(int seen) {
68             switch (state) {
69             case START:
70                 if (seen == GETFIELD) {
71                     XField f = getXFieldOperand();
72                     if (isVolatile(f)) {
73                         incrementField = f;
74                         state = IncrementState.GETFIELD;
75                     }
76                 }
77                 break;
78             case GETFIELD:
79                 if (seen == ICONST_1 || seen == LCONST_1 || seen == ICONST_M1) {
80                     state = IncrementState.LOADCONSTANT;
81                 } else {
82                     resetIncrementState();
83                 }
84     
85                 break;
86             case LOADCONSTANT:
87                 if (seen == IADD || seen == ISUB || seen == LADD || seen == LSUB) {
88                     state = IncrementState.ADD;
89                 } else {
90                     resetIncrementState();
91                 }
92                 break;
93             case ADD:
94                 if (seen == PUTFIELD && incrementField.equals(getXFieldOperand())) {
95                     bugReporter.reportBug(new BugInstance(this, "VO_VOLATILE_INCREMENT",
96                             "J".equals(incrementField.getSignature()) ? Priorities.HIGH_PRIORITY : Priorities.NORMAL_PRIORITY)
97                     .addClassAndMethod(this).addField(incrementField).addSourceLine(this));
98                 }
99                 resetIncrementState();
100                 break;
101             }
102             switch (seen) {
103             case PUTSTATIC: {
104                 XField f = getXFieldOperand();
105                 if (!isVolatileArray(f)) {
106                     return;
107                 }
108                 if ("<clinit>".equals(getMethodName())) {
109                     initializationWrites.add(f);
110                 } else {
111                     otherWrites.add(f);
112                 }
113                 break;
114             }
115             case PUTFIELD: {
116                 XField f = getXFieldOperand();
117                 if (!isVolatileArray(f)) {
118                     return;
119                 }
120     
121                 if ("<init>".equals(getMethodName())) {
122                     initializationWrites.add(f);
123                 } else {
124                     otherWrites.add(f);
125                 }
126                 break;
127             }
128             default:
129                 break;
130             }
131         }
132     
133         /**
134          *
135          */
136         private void resetIncrementState() {
137             state = IncrementState.START;
138             incrementField = null;
139         }
140     
141         @Override
142         public void report() {
143             Subtypes2 subtypes2 = AnalysisContext.currentAnalysisContext().getSubtypes2();
144     
145             for (XField f : AnalysisContext.currentXFactory().allFields()) {
146                 if (isVolatileArray(f) && subtypes2.isApplicationClass(f.getClassDescriptor())) {
147                     int priority = LOW_PRIORITY;
148                     if (initializationWrites.contains(f) && !otherWrites.contains(f)) {
149                         priority = NORMAL_PRIORITY;
150                     }
151                     bugReporter.reportBug(new BugInstance(this, "VO_VOLATILE_REFERENCE_TO_ARRAY", priority).addClass(
152                             f.getClassDescriptor()).addField(f));
153                 }
154             }
155         }
156     
157         private boolean isVolatile(XField f) {
158             return f != null && f.isVolatile();
159         }
160     
161         private boolean isVolatileArray(XField f) {
162             return isVolatile(f) && f.getSignature().charAt(0) == '[';
163         }
164     }
