20     package edu.umd.cs.findbugs.tools.html;
21     
22     import java.io.IOException;
23     import java.io.OutputStream;
24     import java.io.PrintStream;
25     import java.io.Serializable;
26     import java.util.Comparator;
27     import java.util.Set;
28     import java.util.TreeSet;
29     
30     import edu.umd.cs.findbugs.BugPattern;
31     import edu.umd.cs.findbugs.DetectorFactory;
32     import edu.umd.cs.findbugs.I18N;
33     
34     public class PrettyPrintBugDescriptions extends PlainPrintBugDescriptions {
35         private final Set<BugPattern> bugPatternSet;
36     
37         private String headerText;
38     
39         private String beginBodyText;
40     
41         private String prologueText;
42     
43         private String endBodyText;
44     
45         private boolean unabridged;
46     
47         private static final String[] TABLE_COLORS = new String[] { "#eeeeee", "#ffffff" };
48     
49         private static class BugPatternComparator implements Comparator<BugPattern>, Serializable {
50             @Override
51             public int compare(BugPattern a, BugPattern b) {
52                 int cmp = a.getCategory().compareTo(b.getCategory());
53                 if (cmp != 0) {
54                     return cmp;
55                 }
56                 cmp = a.getAbbrev().compareTo(b.getAbbrev());
57                 if (cmp != 0) {
58                     return cmp;
59                 }
60                 return a.getType().compareTo(b.getType());
61             }
62         }
63     
64         public PrettyPrintBugDescriptions(String docTitle, OutputStream out) {
65             super(docTitle, out);
66             this.bugPatternSet = new TreeSet<BugPattern>(new BugPatternComparator());
67             this.headerText = this.beginBodyText = this.prologueText = this.endBodyText = "";
68         }
69     
70         public void setHeaderText(String headerText) {
71             this.headerText = headerText;
72         }
73     
74         public void setBeginBodyText(String beginBodyText) {
75             this.beginBodyText = beginBodyText;
76         }
77     
78         public void setPrologueText(String prologueText) {
79             this.prologueText = prologueText;
80         }
81     
82         public void setEndBodyText(String endBodyText) {
83             this.endBodyText = endBodyText;
84         }
85     
86         @Override
87         protected void prologue() throws IOException {
88             super.prologue();
89             PrintStream out = getPrintStream();
90             out.println(prologueText);
91         }
92     
93         @Override
94         protected void emit(BugPattern bugPattern) throws IOException {
95             bugPatternSet.add(bugPattern);
96         }
97     
98         @Override
99         protected void epilogue() throws IOException {
100             emitSummaryTable();
101             emitBugDescriptions();
102             super.epilogue();
103         }
104     
105         @Override
106         protected void header() throws IOException {
107             PrintStream out = getPrintStream();
108             out.println(headerText);
109         }
110     
111         /** Extra stuff printed at the beginning of the &lt;body&gt; element. */
112         @Override
113         protected void beginBody() throws IOException {
114             PrintStream out = getPrintStream();
115             out.println(beginBodyText);
116         }
117     
118         /** Extra stuff printed at the end of the &lt;body&gt; element. */
119         @Override
120         protected void endBody() throws IOException {
121             PrintStream out = getPrintStream();
122             out.println(endBodyText);
123         }
124     
125         private void emitSummaryTable() {
126             PrintStream out = getPrintStream();
127     
128             out.println("<h2>Summary</h2>");
129     
130             out.println("<table width=\"100%\">");
131     
132             out.println("<tr bgcolor=\"#b9b9fe\"><th>Description</th><th>Category</th></tr>");
133     
134             ColorAlternator colorAlternator = new ColorAlternator(TABLE_COLORS);
135     
136             for (BugPattern bugPattern : bugPatternSet) {
137                 out.print("<tr bgcolor=\"" + colorAlternator.nextColor() + "\">");
138                 out.print("<td><a href=\"#" + bugPattern.getType() + "\">" + bugPattern.getAbbrev() + ": "
139                         + bugPattern.getShortDescription() + "</a></td>");
140                 out.println("<td>" + I18N.instance().getBugCategoryDescription(bugPattern.getCategory()) + "</td></tr>");
141             }
142     
143             out.println("</table>");
144         }
145     
146         private void emitBugDescriptions() {
147             PrintStream out = getPrintStream();
148     
149             out.println("<h2>Descriptions</h2>");
150     
151             for (BugPattern bugPattern : bugPatternSet) {
152                 out.println("<h3><a name=\"" + bugPattern.getType() + "\">" + bugPattern.getAbbrev() + ": "
153                         + bugPattern.getShortDescription() + " (" + bugPattern.getType() + ")" + "</a></h3>");
154                 out.println(bugPattern.getDetailText());
155             }
156         }
157     
158         @Override
159         protected boolean isEnabled(DetectorFactory factory) {
160             return unabridged || super.isEnabled(factory);
161         }
162     
163         public static void main(String[] args) throws Exception {
164             int argCount = 0;
165             boolean unabridged = false;
166     
167             if (argCount < args.length && "-unabridged".equals(args[argCount])) {
168                 ++argCount;
169                 // Unabridged mode: emit all warnings reported by at least one
170                 // detector, even for disabled detectors.
171                 unabridged = true;
172             }
173     
174             if (Boolean.getBoolean("findbugs.bugdesc.unabridged")) {
175                 unabridged = true;
176             }
177     
178             String docTitle = "FindBugs Bug Descriptions";
179             if (argCount < args.length) {
180                 docTitle = args[argCount++];
181             }
182             PrettyPrintBugDescriptions pp = new PrettyPrintBugDescriptions(docTitle, System.out);
183     
184             if (argCount < args.length) {
185                 pp.setHeaderText(args[argCount++]);
186             }
187             if (argCount < args.length) {
188                 pp.setBeginBodyText(args[argCount++]);
189             }
190             if (argCount < args.length) {
191                 pp.setPrologueText(args[argCount++]);
192             }
193             if (argCount < args.length) {
194                 pp.setEndBodyText(args[argCount++]);
195             }
196     
197             if (unabridged) {
198                 pp.unabridged = true;
199             }
200     
201             pp.print();
202         }
203     }
