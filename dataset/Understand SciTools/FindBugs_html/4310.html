111         /*
112          * (non-Javadoc)
113          *
114          * @see
115          * edu.umd.cs.findbugs.bcel.CFGDetector#visitClass(edu.umd.cs.findbugs.classfile
116          * .ClassDescriptor)
117          */
118         @Override
119         public void visitClass(ClassDescriptor classDescriptor) throws CheckedAnalysisException {
120     
121             if (!checked) {
122                 checked = true;
123                 Collection<TypeQualifierValue<?>> allKnownTypeQualifiers = TypeQualifierValue.getAllKnownTypeQualifiers();
124                 int size = allKnownTypeQualifiers.size();
125                 if (size == 1) {
126                     TypeQualifierValue<?> value = Util.first(allKnownTypeQualifiers);
127                     if (!value.typeQualifier.getClassName().equals(NONNULL_ANNOTATION)) {
128                         shouldRunAnalysis = true;
129                     }
130     
131                 } else if (size > 1) {
132                     shouldRunAnalysis = true;
133                 }
134             }
135     
136             if (shouldRunAnalysis) {
137                 super.visitClass(classDescriptor);
138             }
139         }

141         /*
142          * (non-Javadoc)
143          *
144          * @see
145          * edu.umd.cs.findbugs.bcel.CFGDetector#visitMethodCFG(edu.umd.cs.findbugs
146          * .classfile.MethodDescriptor, edu.umd.cs.findbugs.ba.CFG)
147          */
148         @Override
149         protected void visitMethodCFG(MethodDescriptor methodDescriptor, CFG cfg) throws CheckedAnalysisException {
150             if (METHOD != null && !methodDescriptor.getName().equals(METHOD)) {
151                 return;
152             }
153     
154             if (methodDescriptor.getName().startsWith("access$")) {
155                 return;
156             }
157     
158             XMethod xMethod = XFactory.createXMethod(methodDescriptor);
159             if (DEBUG) {
160                 System.out.println("CheckTypeQualifiers: checking " + methodDescriptor.toString());
161             }
162     
163             Collection<TypeQualifierValue<?>> relevantQualifiers = Analysis.getRelevantTypeQualifiers(methodDescriptor, cfg);
164             for(Iterator<TypeQualifierValue<?>> i = relevantQualifiers.iterator(); i.hasNext();  ) {
165                 if (i.next().getTypeQualifierClassDescriptor().getClassName().equals(NONNULL_ANNOTATION)) {
166                     i.remove();
167                 }
168             }
169             if (relevantQualifiers.isEmpty()) {
170                 return;
171             }
172             if (DEBUG) {
173                 System.out.println("  Relevant type qualifiers are " + relevantQualifiers);
174             }
175             IAnalysisCache analysisCache = Global.getAnalysisCache();
176             ForwardTypeQualifierDataflowFactory forwardDataflowFactory = analysisCache.getMethodAnalysis(
177                     ForwardTypeQualifierDataflowFactory.class, methodDescriptor);
178             BackwardTypeQualifierDataflowFactory backwardDataflowFactory = analysisCache.getMethodAnalysis(
179                     BackwardTypeQualifierDataflowFactory.class, methodDescriptor);
180             ValueNumberDataflow vnaDataflow = analysisCache.getMethodAnalysis(ValueNumberDataflow.class, methodDescriptor);
181     
182             for (TypeQualifierValue<?> typeQualifierValue : relevantQualifiers) {
183     
184                 try {
185                     checkQualifier(xMethod, cfg, typeQualifierValue, forwardDataflowFactory, backwardDataflowFactory,
186                             vnaDataflow);
187                 } catch (MissingClassException e) {
188                     AnalysisContext.reportMissingClass(e);
189                 } catch (CheckedAnalysisException e) {
190                     bugReporter.logError("Exception checking type qualifier " + typeQualifierValue.toString() + " on method "
191                             + methodDescriptor.toString(), e);
192                 }
193             }
194         }

217                 ForwardTypeQualifierDataflowFactory forwardDataflowFactory,
218                 BackwardTypeQualifierDataflowFactory backwardDataflowFactory, ValueNumberDataflow vnaDataflow)

220     
221             if (DEBUG) {
222                 System.out.println("----------------------------------------------------------------------");

224                         + xmethod.toString());
225                 if (typeQualifierValue.isStrictQualifier()) {
226                     System.out.println("  Strict type qualifier");
227                 }
228                 System.out.println("----------------------------------------------------------------------");
229             }
230     
231             if (DEBUG_DATAFLOW) {
232                 System.out.println("********* Valuenumber analysis *********");
233                 DataflowCFGPrinter<ValueNumberFrame, ValueNumberAnalysis> p = new DataflowCFGPrinter<ValueNumberFrame, ValueNumberAnalysis>(vnaDataflow);
234                 p.print(System.out);
235             }
236     
237             ForwardTypeQualifierDataflow forwardDataflow = forwardDataflowFactory.getDataflow(typeQualifierValue);
238     
239             if (DEBUG_DATAFLOW && (DEBUG_DATAFLOW_MODE.startsWith("forward") || "both".equals(DEBUG_DATAFLOW_MODE))) {
240                 System.out.println("********* Forwards analysis *********");
241                 DataflowCFGPrinter<TypeQualifierValueSet, ForwardTypeQualifierDataflowAnalysis> p = new DataflowCFGPrinter<TypeQualifierValueSet, ForwardTypeQualifierDataflowAnalysis>(
242                         forwardDataflow);
243                 p.print(System.out);
244             }
245     
246             BackwardTypeQualifierDataflow backwardDataflow = backwardDataflowFactory.getDataflow(typeQualifierValue);
247     
248             if (DEBUG_DATAFLOW && (DEBUG_DATAFLOW_MODE.startsWith("backward") || "both".equals(DEBUG_DATAFLOW_MODE))) {
249                 System.out.println("********* Backwards analysis *********");
250                 DataflowCFGPrinter<TypeQualifierValueSet, BackwardTypeQualifierDataflowAnalysis> p = new DataflowCFGPrinter<TypeQualifierValueSet, BackwardTypeQualifierDataflowAnalysis>(
251                         backwardDataflow);
252                 p.print(System.out);
253             }
254     
255             checkDataflow(xmethod, cfg, typeQualifierValue, vnaDataflow, forwardDataflow, backwardDataflow);
256             checkValueSources(xmethod, cfg, typeQualifierValue, vnaDataflow, forwardDataflow, backwardDataflow);
257         }

259         private void checkDataflow(XMethod xmethod, CFG cfg, TypeQualifierValue<?> typeQualifierValue,
260                 ValueNumberDataflow vnaDataflow, ForwardTypeQualifierDataflow forwardDataflow,
261                 BackwardTypeQualifierDataflow backwardDataflow) throws DataflowAnalysisException, CheckedAnalysisException {
262             for (Iterator<Location> i = cfg.locationIterator(); i.hasNext();) {
263                 Location loc = i.next();
264     
265                 TypeQualifierValueSet forwardsFact = forwardDataflow.getFactAtLocation(loc);
266                 TypeQualifierValueSet backwardsFact = backwardDataflow.getFactAfterLocation(loc);
267     
268                 if (!forwardsFact.isValid() || !backwardsFact.isValid()) {
269                     continue;
270                 }
271     
272                 if (DEBUG) {
273                     checkLocation = "location " + loc.toCompactString();
274                 }
275                 checkForConflictingValues(xmethod, cfg, typeQualifierValue, forwardsFact, backwardsFact, loc,
276                         loc, vnaDataflow.getFactAtLocation(loc));
277                 checkForEqualityTest(xmethod, cfg, typeQualifierValue, forwardsFact, loc, vnaDataflow.getFactAtLocation(loc));
278             }
279     
280             for (Iterator<Edge> i = cfg.edgeIterator(); i.hasNext();) {
281                 Edge edge = i.next();
282     
283     
284                 // NOTE: when checking forwards and backwards values on an edge,
285                 // we don't want to apply BOTH edge transfer functions,
286                 // since the purpose of the edge transfer function is to
287                 // propagate information across phi nodes (effectively
288                 // copying information about one value to another).
289                 // Due to pruning of backwards values when a conflict is detected,
290                 // we need to check backwards values as "early" as possible,
291                 // meaning that we want to check at the edge target
292                 // (before the backwards edge transfer function has pruned
293                 // the backwards value.)
294                 TypeQualifierValueSet forwardFact = forwardDataflow.getFactOnEdge(edge);
295                 TypeQualifierValueSet backwardFact = backwardDataflow.getResultFact(edge.getTarget());
296     
297                 // The edge target location is where we can check
298                 // for conflicting flow values.
299                 Location edgeTargetLocation = getEdgeTargetLocation(cfg, edge);
300                 ValueNumberFrame vnaFrame = (edgeTargetLocation != null) ? vnaDataflow.getFactAtLocation(edgeTargetLocation) : null;
301     
302                 // What location do we want to report to the user
303                 // as where the conflict occurs?
304                 // The edge source location is generally better,
305                 // but edge target location is ok as a fallback.
306                 Location locationToReport;
307                 if (edge.getSource().getLastInstruction() != null) {
308                     locationToReport = getEdgeSourceLocation(cfg, edge);
309                 } else {
310                     locationToReport = edgeTargetLocation;
311                 }
312     
313                 checkForConflictingValues(xmethod, cfg, typeQualifierValue, forwardFact, backwardFact,
314                         locationToReport, edgeTargetLocation, vnaFrame);
315     
316             }
317         }

320         private void checkForEqualityTest(XMethod  xmethod, CFG cfg, TypeQualifierValue<?> typeQualifierValue,
321                 TypeQualifierValueSet forwardsFact, Location loc, ValueNumberFrame factAtLocation) throws DataflowAnalysisException {
322             InstructionHandle handle = loc.getHandle();
323             Instruction ins = handle.getInstruction();
324             boolean isTest = false;
325     
326             ConstantPoolGen cpg = cfg.getMethodGen().getConstantPool();
327             if (ins instanceof IfInstruction && ins.consumeStack(cpg) == 2) {
328                 isTest = true;
329             }  else if (ins instanceof InvokeInstruction && ins.consumeStack(cpg) == 2) {
330                 InvokeInstruction invoke = (InvokeInstruction) ins;
331                 isTest = "equals".equals(invoke.getMethodName(cpg)) &&"(Ljava/lang/Object;)Z".equals(invoke.getSignature(cpg)) ;
332             }
333             if (isTest) {
334                 ValueNumber top = factAtLocation.getStackValue(0);
335                 if (top.hasFlag(ValueNumber.CONSTANT_VALUE)) {
336                     return;
337                 }
338                 ValueNumber next = factAtLocation.getStackValue(1);
339                 if (next.hasFlag(ValueNumber.CONSTANT_VALUE)) {
340                     return;
341                 }
342                 FlowValue topTQ = forwardsFact.getValue(top);
343                 FlowValue nextTQ = forwardsFact.getValue(next);
344                 if (DEBUG) {
345                     System.out.println("Comparing values at " + loc.toCompactString());
346                     System.out.println(" Comparing " + topTQ + " and " + nextTQ);
347                 }
348                 if (topTQ.equals(nextTQ)) {
349                     return;
350                 }
351                 if (FlowValue.valuesConflict(typeQualifierValue.isStrictQualifier() && !xmethod.isIdentity(), topTQ, nextTQ)) {
352                     BugInstance warning = new BugInstance(this,"TQ_COMPARING_VALUES_WITH_INCOMPATIBLE_TYPE_QUALIFIERS", HIGH_PRIORITY).addClassAndMethod(xmethod);
353                     annotateWarningWithTypeQualifier(warning, typeQualifierValue);
354                     for(SourceSinkInfo s : forwardsFact.getWhere(top)) {
355                         annotateWarningWithSourceSinkInfo(warning, xmethod, top, s);
356                     }
357                     for(SourceSinkInfo s : forwardsFact.getWhere(next)) {
358                         annotateWarningWithSourceSinkInfo(warning, xmethod, next, s);
359                     }
360                     SourceLineAnnotation observedLocation = SourceLineAnnotation.fromVisitedInstruction(xmethod.getMethodDescriptor(),
361                             loc);
362                     warning.add(observedLocation);
363                     warning.addSomeSourceForTopTwoStackValues(classContext, method, loc);
364                     bugReporter.reportBug(warning);
365     
366                 }
367     
368             }
369     
370     
371         }

373         private void checkValueSources(XMethod xMethod, CFG cfg, TypeQualifierValue<?> typeQualifierValue,
374                 ValueNumberDataflow vnaDataflow, ForwardTypeQualifierDataflow forwardDataflow,
375                 BackwardTypeQualifierDataflow backwardDataflow) throws DataflowAnalysisException, CheckedAnalysisException {
376     
377             // Check to see if any backwards ALWAYS or NEVER values
378             // reach incompatible sources.
379     
380             for (Iterator<Location> i = cfg.locationIterator(); i.hasNext();) {
381                 Location location = i.next();
382     
383                 Set<SourceSinkInfo> sourceSet = forwardDataflow.getAnalysis().getSourceSinkInfoSet(location);
384     
385                 for (SourceSinkInfo source : sourceSet) {
386                     ValueNumber vn = source.getValueNumber();
387                     TypeQualifierValueSet backwardsFact = backwardDataflow.getFactAtLocation(location);
388                     FlowValue backwardsFlowValue = backwardsFact.getValue(vn);
389     
390                     if (!(backwardsFlowValue == FlowValue.ALWAYS || backwardsFlowValue == FlowValue.NEVER)) {
391                         continue;
392                     }
393                     if (DEBUG) {
394                         System.out.println("Checking value source at " + location.toCompactString() + " for " + typeQualifierValue);
395                         System.out.println("  back=" + backwardsFact);
396                         System.out.println("  source=" + source);
397                     }
398     
399                     // Check to see if this warning has already been reported
400                     // because
401                     // the dataflow values conflict directly with each other.
402                     TypeQualifierValueSet forwardsFact = forwardDataflow.getFactAfterLocation(location);
403                     FlowValue forwardsFlowValue = forwardsFact.getValue(vn);
404                     if (FlowValue.valuesConflict(typeQualifierValue.isStrictQualifier() && !xMethod.isIdentity(), forwardsFlowValue, backwardsFlowValue)) {
405                         continue;
406                     }
407     
408                     if (FlowValue.backwardsValueConflictsWithSource(backwardsFlowValue, source, typeQualifierValue, xMethod.isIdentity())) {
409                         String bugType = (backwardsFlowValue == FlowValue.NEVER) ? "TQ_MAYBE_SOURCE_VALUE_REACHES_NEVER_SINK"
410                                 : "TQ_MAYBE_SOURCE_VALUE_REACHES_ALWAYS_SINK";
411     
412                         emitSourceWarning(bugType, xMethod, typeQualifierValue, backwardsFlowValue, backwardsFact, source,
413                                 vn, location);
414                     } else if (source.getWhen() == When.UNKNOWN && source.getType() == SourceSinkType.PARAMETER) {
415     
416                         int p = source.getParameter();
417                         TypeQualifierAnnotation directTypeQualifierAnnotation = TypeQualifierApplications
418                                 .getDirectTypeQualifierAnnotation(xMethod, p, typeQualifierValue);
419                         if (directTypeQualifierAnnotation != null && directTypeQualifierAnnotation.when == When.UNKNOWN) {
420                             String bugType = (backwardsFlowValue == FlowValue.NEVER) ? "TQ_EXPLICIT_UNKNOWN_SOURCE_VALUE_REACHES_NEVER_SINK"
421                                     : "TQ_EXPLICIT_UNKNOWN_SOURCE_VALUE_REACHES_ALWAYS_SINK";
422     
423                             emitSourceWarning(bugType, xMethod, typeQualifierValue, backwardsFlowValue, backwardsFact,
424                                     source, vn, location);
425                         }
426     
427                     } else if (source.getWhen() == When.UNKNOWN && typeQualifierValue.isStrictQualifier()) {
428                         emitDataflowWarning(xMethod, typeQualifierValue, forwardsFact, backwardsFact, vn, forwardsFlowValue, backwardsFlowValue,
429                                 location, null, vnaDataflow.getFactAtLocation(location));
430                     }
431                 }
432             }
433     
434         }

436         private Location getEdgeTargetLocation(CFG cfg, Edge edge) {
437             BasicBlock targetBlock = edge.getTarget();
438     
439             // Target block is nonempty?
440             if (targetBlock.getFirstInstruction() != null) {
441                 return new Location(targetBlock.getFirstInstruction(), targetBlock);
442             }
443     
444             // Target block is an ETB?
445             if (targetBlock.isExceptionThrower()) {
446                 BasicBlock fallThroughSuccessor = cfg.getSuccessorWithEdgeType(targetBlock, EdgeTypes.FALL_THROUGH_EDGE);
447                 if (fallThroughSuccessor == null) {
448                     // Fall through edge might have been pruned
449                     for (Iterator<Edge> i = cfg.removedEdgeIterator(); i.hasNext();) {
450                         Edge removedEdge = i.next();
451                         if (removedEdge.getSource() == targetBlock && removedEdge.getType() == EdgeTypes.FALL_THROUGH_EDGE) {
452                             fallThroughSuccessor = removedEdge.getTarget();
453                             break;
454                         }
455                     }
456                 }
457     
458                 if (fallThroughSuccessor != null && fallThroughSuccessor.getFirstInstruction() != null) {
459                     return new Location(fallThroughSuccessor.getFirstInstruction(), fallThroughSuccessor);
460                 }
461             }
462     
463             return null;
464         }

466         private Location getEdgeSourceLocation(CFG cfg, Edge edge) {
467             BasicBlock sourceBlock = edge.getSource();
468             return (sourceBlock.getLastInstruction() != null) ? new Location(sourceBlock.getLastInstruction(), sourceBlock) : null;
469         }

471         private void checkForConflictingValues(XMethod xMethod, CFG cfg,
472                 TypeQualifierValue<?> typeQualifierValue, TypeQualifierValueSet forwardsFact, TypeQualifierValueSet backwardsFact,
473                 Location locationToReport, Location locationWhereDoomedValueIsObserved, ValueNumberFrame vnaFrame) throws CheckedAnalysisException {
474             Set<ValueNumber> valueNumberSet = new HashSet<ValueNumber>();
475             valueNumberSet.addAll(forwardsFact.getValueNumbers());
476             valueNumberSet.addAll(backwardsFact.getValueNumbers());
477     
478             for (ValueNumber vn : valueNumberSet) {
479                 FlowValue forward = forwardsFact.getValue(vn);
480                 FlowValue backward = backwardsFact.getValue(vn);
481                 if (!FlowValue.valuesConflict(typeQualifierValue.isStrictQualifier() && !xMethod.isIdentity(), forward, backward)) {
482                     continue;
483                 }
484     
485                 if (DEBUG) {
486                     System.out.println("Check " + vn + ": forward=" + forward + ", backward=" + backward + " at " + checkLocation);
487                 }
488     
489                 emitDataflowWarning(xMethod, typeQualifierValue, forwardsFact, backwardsFact, vn, forward, backward,
490                         locationToReport, locationWhereDoomedValueIsObserved, vnaFrame);
491     
492             }
493         }

495         private void emitDataflowWarning(XMethod xMethod, TypeQualifierValue<?> typeQualifierValue,
496                 TypeQualifierValueSet forwardsFact, TypeQualifierValueSet backwardsFact, ValueNumber vn, FlowValue forward,
497                 FlowValue backward, Location locationToReport, @CheckForNull Location locationWhereDoomedValueIsObserved, ValueNumberFrame vnaFrame)
498                         throws CheckedAnalysisException {
499             String bugType;
500             if (typeQualifierValue.isStrictQualifier() && forward == FlowValue.UNKNOWN) {
501                 bugType = "TQ_UNKNOWN_VALUE_USED_WHERE_ALWAYS_STRICTLY_REQUIRED";
502             } else if (backward == FlowValue.NEVER) {
503                 bugType = "TQ_ALWAYS_VALUE_USED_WHERE_NEVER_REQUIRED";
504             } else {
505                 bugType = "TQ_NEVER_VALUE_USED_WHERE_ALWAYS_REQUIRED";
506             }
507     
508             // Issue warning
509             BugInstance warning = new BugInstance(this, bugType, Priorities.NORMAL_PRIORITY).addClassAndMethod(xMethod);
510             annotateWarningWithTypeQualifier(warning, typeQualifierValue);
511     
512             Set<? extends SourceSinkInfo> sourceSet = (forward == FlowValue.ALWAYS) ? forwardsFact.getWhereAlways(vn) : forwardsFact
513                     .getWhereNever(vn);
514             for (SourceSinkInfo source : sourceSet) {
515                 annotateWarningWithSourceSinkInfo(warning, xMethod, vn, source);
516             }
517             Set<? extends SourceSinkInfo> sinkSet = (backward == FlowValue.ALWAYS) ? backwardsFact.getWhereAlways(vn) : backwardsFact
518                     .getWhereNever(vn);
519     
520             Location sinkLocation = getSinkLocation(sinkSet);
521             if (sinkLocation == null) {
522                 AnalysisContext.logError("Unable to compute sink location for " + xMethod);
523                 return;
524             }
525     
526             // Hopefully we can find the conflicted value in a local variable
527             if (locationWhereDoomedValueIsObserved != null) {
528                 Method method = Global.getAnalysisCache().getMethodAnalysis(Method.class, xMethod.getMethodDescriptor());
529     
530                 LocalVariableAnnotation localVariable = ValueNumberSourceInfo.findLocalAnnotationFromValueNumber(method,
531                         locationWhereDoomedValueIsObserved, vn, vnaFrame);
532                 if (localVariable != null && !localVariable.equals(warning.getPrimaryLocalVariableAnnotation())) {
533                     localVariable.setDescription(localVariable.isSignificant() ? "LOCAL_VARIABLE_VALUE_DOOMED_NAMED"
534                             : "LOCAL_VARIABLE_VALUE_DOOMED");
535                     warning.add(localVariable);
536     
537                 }
538                 if (!sinkLocation.equals(locationToReport)) {
539                     // Report where we observed the value.
540                     // Note that for conflicts detected on control edges,
541                     // we REPORT the edge source location
542                     // rather than the target location, even though it is the
543                     // target location where the conflict is detected.
544                     // The only reason to use a different reporting location
545                     // is to produce a more informative report for the user,
546                     // since the edge source is where the branch is found.
547                     SourceLineAnnotation observedLocation = SourceLineAnnotation.fromVisitedInstruction(xMethod.getMethodDescriptor(),
548                             locationToReport);
549                     observedLocation.setDescription("SOURCE_LINE_VALUE_DOOMED");
550                     warning.add(observedLocation);
551                 }
552             }
553     
554             // Add value sinks
555             for (SourceSinkInfo sink : sinkSet) {
556                 annotateWarningWithSourceSinkInfo(warning, xMethod, vn, sink);
557             }
558     
559             bugReporter.reportBug(warning);
560         }
561     
562         private void emitSourceWarning(String bugType, XMethod xMethod, TypeQualifierValue<?> typeQualifierValue,
563                 FlowValue backwardsFlowValue, TypeQualifierValueSet backwardsFact, SourceSinkInfo source, ValueNumber vn,
564                 Location location) {
565     
566             BugInstance warning = new BugInstance(this, bugType, Priorities.NORMAL_PRIORITY).addClassAndMethod(xMethod);
567             annotateWarningWithTypeQualifier(warning, typeQualifierValue);
568     
569             annotateWarningWithSourceSinkInfo(warning, xMethod, vn, source);
570     
571             Set<? extends SourceSinkInfo> sinkSet = (backwardsFlowValue == FlowValue.NEVER) ? backwardsFact.getWhereNever(vn) : backwardsFact
572                     .getWhereAlways(vn);
573             for (SourceSinkInfo sink : sinkSet) {
574                 annotateWarningWithSourceSinkInfo(warning, xMethod, vn, sink);
575             }
576     
577             bugReporter.reportBug(warning);
578         }
579     
580         private void annotateWarningWithTypeQualifier(BugInstance warning, TypeQualifierValue<?> typeQualifierValue) {
581             if (TypeQualifierValue.hasMultipleVariants(typeQualifierValue)) {
582                 StringBuilder buf = new StringBuilder();
583                 buf.append("@");
584                 buf.append(typeQualifierValue.typeQualifier.getDottedClassName());
585     
586                 // When there are multiple variants, qualify the type
587                 // qualifier with the value indicating which variant.
588                 buf.append("(");
589                 buf.append(typeQualifierValue.value);
590                 buf.append(")");
591                 warning.addString(buf.toString()).describe(StringAnnotation.TYPE_QUALIFIER_ROLE);
592             } else {
593                 warning.addClass(typeQualifierValue.typeQualifier).describe(ClassAnnotation.TYPE_QUALIFIER_ROLE);
594             }
595     
596         }
597     
598         private void annotateWarningWithSourceSinkInfo(BugInstance warning, XMethod xMethod, ValueNumber vn,
599                 SourceSinkInfo sourceSinkInfo) {
600             MethodDescriptor methodDescriptor = xMethod.getMethodDescriptor();
601             switch (sourceSinkInfo.getType()) {
602             case PARAMETER:
603                 try {
604                     Method method = Global.getAnalysisCache().getMethodAnalysis(Method.class, methodDescriptor);
605                     LocalVariableAnnotation lva = LocalVariableAnnotation.getParameterLocalVariableAnnotation(method,
606                             sourceSinkInfo.getLocal());
607                     lva.setDescription(lva.isSignificant() ? LocalVariableAnnotation.PARAMETER_VALUE_SOURCE_NAMED_ROLE
608                             : LocalVariableAnnotation.PARAMETER_VALUE_SOURCE_ROLE);
609     
610                     warning.add(lva);
611                 } catch (CheckedAnalysisException e) {
612                     warning.addSourceLine(methodDescriptor, sourceSinkInfo.getLocation()).describe("SOURCE_LINE_VALUE_SOURCE");
613                 }
614                 break;
615     
616             case CONSTANT_VALUE:
617                 Object constantValue = sourceSinkInfo.getConstantValue();
618                 if (constantValue instanceof String) {
619                     warning.addString((String) constantValue).describe(StringAnnotation.STRING_CONSTANT_ROLE);
620                 } else if (constantValue instanceof Integer) {
621                     warning.addInt((Integer) constantValue).describe(IntAnnotation.INT_VALUE);
622                 } else if (constantValue == null) {
623                     warning.addString("null").describe(StringAnnotation.STRING_NONSTRING_CONSTANT_ROLE);
624                 } else {
625                     warning.addString(constantValue.toString()).describe(StringAnnotation.STRING_NONSTRING_CONSTANT_ROLE);
626                 }
627                 break;
628     
629             case RETURN_VALUE_OF_CALLED_METHOD:
630             case FIELD_LOAD:
631                 warning.addSourceLine(methodDescriptor, sourceSinkInfo.getLocation()).describe("SOURCE_LINE_VALUE_SOURCE");
632                 break;
633     
634             case ARGUMENT_TO_CALLED_METHOD:
635             case RETURN_VALUE:
636             case FIELD_STORE:
637                 warning.addSourceLine(methodDescriptor, sourceSinkInfo.getLocation());
638                 return;
639     
640             default:
641                 throw new IllegalStateException();
642             }
643         }
644     
645         private @CheckForNull
646         Location getSinkLocation(SourceSinkInfo sourceSinkInfo) {
647             switch (sourceSinkInfo.getType()) {
648     
649             case ARGUMENT_TO_CALLED_METHOD:
650             case RETURN_VALUE:
651             case FIELD_STORE:
652                 return sourceSinkInfo.getLocation();
653     
654             default:
655                 return null;
656             }
657         }
658     
659         private @CheckForNull
660         Location getSinkLocation(Iterable<? extends SourceSinkInfo> info) {
661             for (SourceSinkInfo s : info) {
662                 Location l = getSinkLocation(s);
663                 if (l != null) {
664                     return l;
665                 }
666     
667             }
668             return null;
669         }
