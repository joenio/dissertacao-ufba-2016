20     package edu.umd.cs.findbugs.ba.obl;
21     
22     import java.util.Collection;
23     import java.util.Collections;
24     import java.util.HashSet;
25     import java.util.LinkedList;
26     import java.util.List;
27     import java.util.Set;
28     
29     import org.apache.bcel.generic.ReferenceType;
30     
31     import edu.umd.cs.findbugs.SystemProperties;
32     import edu.umd.cs.findbugs.ba.XMethod;
33     
34     /**
35      * Policy database which defines which methods create and remove obligations.
36      *
37      * <p>
38      * See Weimer and Necula, <a href="http://doi.acm.org/10.1145/1028976.1029011"
39      * >Finding and preventing run-time error handling mistakes</a>, OOPSLA 2004.
40      * </p>
41      *
42      * @author David Hovemeyer
43      */
44     public class ObligationPolicyDatabase {
45         public static final boolean DEBUG = SystemProperties.getBoolean("oa.debug.db");
46     
47         private final ObligationFactory factory;
48     
49         private final LinkedList<ObligationPolicyDatabaseEntry> entryList;
50     
51         private final HashSet<Obligation> allObligations = new HashSet<Obligation>();
52     
53         private boolean strictChecking;
54     
55         public ObligationPolicyDatabase() {
56             this.factory = new ObligationFactory();
57             this.entryList = new LinkedList<ObligationPolicyDatabaseEntry>();
58     
59         }
60     
61         public ObligationFactory getFactory() {
62             return factory;
63         }
64     
65         public Set<Obligation> getAllObligations() {
66             return allObligations;
67         }
68     
69         public void addEntry(ObligationPolicyDatabaseEntry entry) {
70             if (DEBUG) {
71                 System.out.println("Adding entry " + entry);
72             }
73             allObligations.addAll(entry.getAllObligations());
74             entryList.add(entry);
75         }
76     
77         /**
78          * Add an appropriate policy database entry for parameters marked with the
79          * WillClose annotation.
80          *
81          * @param xmethod
82          *            a method
83          * @param obligation
84          *            the Obligation deleted by the method
85          * @param entryType
86          *            type of entry (STRONG or WEAK)
87          */
88         public ObligationPolicyDatabaseEntry addParameterDeletesObligationDatabaseEntry(XMethod xmethod, Obligation obligation,
89                 ObligationPolicyDatabaseEntryType entryType) {
90             // Add a policy database entry noting that this method
91             // will delete one instance of the obligation type.
92             ObligationPolicyDatabaseEntry entry = new MatchMethodEntry(xmethod, ObligationPolicyDatabaseActionType.DEL, entryType,
93                     obligation);
94             addEntry(entry);
95             return entry;
96         }
97     
98         public void setStrictChecking(boolean strictChecking) {
99             if (DEBUG) {
100                 System.out.println("Setting strict checking to " + strictChecking );
101             }
102             this.strictChecking = strictChecking;
103         }
104     
105         public boolean isStrictChecking() {
106             return strictChecking;
107         }
108     
109         public void getActions(ReferenceType receiverType, String methodName, String signature, boolean isStatic,
110                 Collection<ObligationPolicyDatabaseAction> actionList) {
111             if (DEBUG) {
112                 System.out.println("Lookup for " + receiverType + "," + methodName + "," + signature + "," + isStatic + ": ");
113             }
114             for (ObligationPolicyDatabaseEntry entry : entryList) {
115     
116                 boolean matched = entry.getActions(receiverType, methodName, signature, isStatic, actionList);
117     
118                 if (DEBUG) {
119                     if (matched)
120                     {
121                         System.out.println(" Entry " + entry + "  ==> MATCH");
122                         //                else
123                         //                    System.out.println("  ==> no match");
124                     }
125                 }
126             }
127             if (DEBUG) {
128                 System.out.println("  ** Resulting action list: " + actionList);
129             }
130         }
131     
132         public List<ObligationPolicyDatabaseEntry> getEntries() {
133             return Collections.unmodifiableList(entryList);
134         }
135     }
