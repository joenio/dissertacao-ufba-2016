24     
25     import edu.umd.cs.findbugs.BugInstance;

30     /**
31      * Why this wasn't just called Filter is still somewhat of a mystery.
32      * FilterMatchers are Filters, pass in a StringPair like Priority, High and all
33      * the high priority bugs disappear, Its that easy.
34      */
35     @Deprecated
36     public class FilterMatcher implements Matcher, Serializable, Comparable<FilterMatcher> {
37         enum FilterWhere {
38             FILTER_EXACTLY, FILTER_AT_OR_AFTER, FILTER_AT_OR_BEFORE, FILTER_ALL_BUT
39         }
40     
41         private static final long serialVersionUID = -4859486064351510016L;
42     
43         private final Sortables filterBy;
44     
45         private final String value;
46     
47         private final FilterWhere mode;
48     
49         protected boolean active;
50     
51         public FilterMatcher(SortableValue sp) {
52             this(sp.key, sp.value);
53         }
54     
55         Sortables getFilterBy() {
56             return filterBy;
57         }
58     
59         String getValue() {
60             return value;
61         }
62     
63         public FilterMatcher(Sortables filterBy, String value, FilterWhere mode) // 0
64         // =
65         // exactly;
66         // 1
67         // =
68         // at
69         // or
70         // after;
71         // 2
72         // =
73         // at
74         // or
75         // before;
76         // 3
77         // =
78         // not
79         // at
80         {
81             this.filterBy = filterBy;
82             this.value = value;
83             this.mode = mode;
84             this.active = true;
85         }
86     
87         public FilterMatcher(Sortables filterBy, String value) {
88             this.filterBy = filterBy;
89             this.value = value;
90             this.mode = FilterWhere.FILTER_EXACTLY;
91             this.active = true;
92         }
93     
94         public void setActive(boolean active) {
95             if (active != this.active) {
96                 this.active = active;
97                 if (active == true) {
98                     FilterActivity.notifyListeners(FilterListener.Action.FILTERING, null);
99                 } else {
100                     FilterActivity.notifyListeners(FilterListener.Action.UNFILTERING, null);
101                 }
102             }
103         }
104     
105         public boolean isActive() {
106             return active;
107         }
108     
109         @Override
110         public boolean match(BugInstance bugInstance) {
111             if (!active) {
112                 return true;
113             }
114     
115             SortableStringComparator ssc = filterBy.getComparator();
116             int compare = ssc.compare(filterBy.getFrom(bugInstance), value);
117             switch (mode) {
118             case FILTER_EXACTLY:
119                 return (compare != 0);
120             case FILTER_AT_OR_AFTER:
121                 return (compare < 0);
122             case FILTER_AT_OR_BEFORE:
123                 return (compare > 0);
124             case FILTER_ALL_BUT:
125                 return (compare == 0);
126             default:
127                 return true;
128             }
129         }
130     
131         @Override
132         public String toString() {
133             switch (mode) {
134             case FILTER_EXACTLY:
135                 return filterBy.toString() + " " + edu.umd.cs.findbugs.L10N.getLocalString("dlg.is", "is") + " "
136                 + edu.umd.cs.findbugs.L10N.getLocalString("mode.equal_to", "equal to") + " " + filterBy.formatValue(value);
137             case FILTER_AT_OR_AFTER:
138                 return filterBy.toString() + " " + edu.umd.cs.findbugs.L10N.getLocalString("dlg.is", "is") + " "
139                 + edu.umd.cs.findbugs.L10N.getLocalString("mode.at_or_after", "at or after") + " "
140                 + filterBy.formatValue(value);
141             case FILTER_AT_OR_BEFORE:
142                 return filterBy.toString() + " " + edu.umd.cs.findbugs.L10N.getLocalString("dlg.is", "is") + " "
143                 + edu.umd.cs.findbugs.L10N.getLocalString("mode.at_or_before", "at or before") + " "
144                 + filterBy.formatValue(value);
145             case FILTER_ALL_BUT:
146                 return filterBy.toString() + " " + edu.umd.cs.findbugs.L10N.getLocalString("dlg.is", "is") + " "
147                 + edu.umd.cs.findbugs.L10N.getLocalString("mode.not_equal_to", "not equal to") + " "
148                 + filterBy.formatValue(value);
149             default:
150                 throw new RuntimeException();
151             }
152         }
153     
154         @Override
155         public boolean equals(Object o) {
156             if (o == null) {
157                 return false;
158             }
159             if (getClass() != o.getClass()) {
160                 return false;
161             }
162     
163             if (filterBy.equals(((FilterMatcher) o).filterBy) && value.equals(((FilterMatcher) o).value)) {
164                 return true;
165             }
166             return false;
167         }
168     
169         @Override
170         public int hashCode() {
171             return value.hashCode() + filterBy.hashCode();
172         }
173     
174         @Override
175         public int compareTo(FilterMatcher that) {
176             if (this.filterBy != that.filterBy) {
177                 return (this.filterBy.ordinal() < that.filterBy.ordinal() ? -1 : 1);
178             }
179     
180             return this.value.compareTo(that.value);
181         }
182     
183         @Override
184         public void writeXML(XMLOutput xmlOutput, boolean disabled) throws IOException {
185             throw new UnsupportedOperationException();
186         }
187     }
