20     package edu.umd.cs.findbugs.gui2;
21     
22     import java.awt.Insets;
23     import java.util.HashMap;
24     import java.util.HashSet;
25     import java.util.List;
26     import java.util.Map;
27     
28     import javax.swing.BoxLayout;
29     import javax.swing.JCheckBox;
30     import javax.swing.JPanel;
31     import javax.swing.JSeparator;
32     import javax.swing.SwingConstants;
33     import javax.swing.border.EmptyBorder;
34     
35     import edu.umd.cs.findbugs.BugInstance;
36     import edu.umd.cs.findbugs.filter.Matcher;
37     import edu.umd.cs.findbugs.filter.NotMatcher;
38     
39     /**
40      * Creates a list of options on for filtering bugs based on the current bug selected.
41      *
42      * Gives the option to invert the created filter by wrapping it in a {@link NotMatcher}.
43      *
44      * @author Graham Allan (grundlefleck@gmail.com)
45      */
46     final class FilterFromBugPicker {
47     
48         private final HashMap<JCheckBox, Sortables> map = new HashMap<JCheckBox, Sortables>();
49         private final BugInstance bug;
50         private final List<Sortables> availableSortables;
51         private final JPanel pickerPanel;
52         private final JCheckBox notFilterCheck = new JCheckBox("Invert (i.e. filter bugs which do not match selected criteria).");
53     
54         public FilterFromBugPicker(BugInstance bug, List<Sortables> availableSortables) {
55             this.bug = bug;
56             this.availableSortables = availableSortables;
57             this.pickerPanel = buildPanel();
58         }
59     
60         private JPanel buildPanel() {
61             JPanel panel = new JPanel();
62             panel.setLayout(new BoxLayout(panel, BoxLayout.Y_AXIS));
63             panel.setBorder(new EmptyBorder(new Insets(6, 6, 6, 6)));
64     
65             addFilterLikeCheckboxes(panel);
66             addNotFilterOption(panel);
67     
68             return panel;
69         }
70     
71         private void addFilterLikeCheckboxes(JPanel center) {
72             for (Sortables sortable : availableSortables) {
73                 if (!FilterFactory.canFilter(sortable)) { continue; }
74     
75                 JCheckBox checkBox = new JCheckBox(sortable.toString() + " is " + sortable.formatValue(sortable.getFrom(bug)));
76     
77                 map.put(checkBox, sortable);
78                 center.add(checkBox);
79             }
80         }
81     
82         private void addNotFilterOption(JPanel center) {
83             center.add(new JSeparator(SwingConstants.HORIZONTAL));
84             center.add(notFilterCheck);
85         }
86     
87         public JPanel pickerPanel() {
88             return pickerPanel;
89         }
90     
91         public Matcher makeMatcherFromSelection() {
92             HashSet<Sortables> set = new HashSet<Sortables>();
93             for (Map.Entry<JCheckBox, Sortables> e : map.entrySet()) {
94                 if (e.getKey().isSelected()) {
95                     set.add(e.getValue());
96                 }
97             }
98             Matcher matcher = null;
99             if (!set.isEmpty()) {
100                 matcher = FilterFactory.makeMatcher(set, bug);
101     
102                 if(notFilterCheck.isSelected()) {
103                     matcher = FilterFactory.invertMatcher(matcher);
104                 }
105             }
106     
107             return matcher;
108         }
109     
110     }
