20     package edu.umd.cs.findbugs.ba.bcp;
21     
22     import org.apache.bcel.generic.ConstantPoolGen;
23     import org.apache.bcel.generic.FieldInstruction;
24     import org.apache.bcel.generic.Type;
25     
26     import edu.umd.cs.findbugs.ba.DataflowAnalysisException;
27     import edu.umd.cs.findbugs.ba.vna.ValueNumber;
28     import edu.umd.cs.findbugs.ba.vna.ValueNumberFrame;
29     
30     /**
31      * Base class for Load and Store PatternElements. Handles some of the grunt work
32      * of representing fields and extracting field values from the stack frame.
33      *
34      * @author David Hovemeyer
35      * @see Load
36      * @see Store
37      */
38     public abstract class FieldAccess extends SingleInstruction implements org.apache.bcel.Constants {
39         private final String fieldVarName;
40     
41         private final String valueVarName;
42     
43         /**
44          * Constructor.
45          *
46          * @param fieldVarName
47          *            name of the variable to bind to the field
48          * @param valueVarName
49          *            name of the variable to bind to the value store in or loaded
50          *            from the field
51          */
52         public FieldAccess(String fieldVarName, String valueVarName) {
53             this.fieldVarName = fieldVarName;
54             this.valueVarName = valueVarName;
55         }
56     
57         /**
58          * Check that the Variables determined for the field and the value
59          * loaded/stored are consistent with previous variable definitions.
60          *
61          * @param field
62          *            Variable representing the field
63          * @param value
64          *            Variable representing the value loaded/stored
65          * @param bindingSet
66          *            previous definitions
67          * @return a MatchResult containing an updated BindingSet if successful, or
68          *         null if unsucessful
69          */
70         protected MatchResult checkConsistent(Variable field, Variable value, BindingSet bindingSet) {
71             // Ensure that the field and value variables are consistent with
72             // previous definitions (if any)
73             bindingSet = addOrCheckDefinition(fieldVarName, field, bindingSet);
74             if (bindingSet == null) {
75                 return null;
76             }
77             bindingSet = addOrCheckDefinition(valueVarName, value, bindingSet);
78             if (bindingSet == null) {
79                 return null;
80             }
81             return new MatchResult(this, bindingSet);
82         }
83     
84         /**
85          * Return whether the given FieldInstruction accesses a long or double
86          * field.
87          *
88          * @param fieldIns
89          *            the FieldInstruction
90          * @param cpg
91          *            the ConstantPoolGen for the method
92          */
93         protected static boolean isLongOrDouble(FieldInstruction fieldIns, ConstantPoolGen cpg) {
94             Type type = fieldIns.getFieldType(cpg);
95             int code = type.getType();
96             return code == T_LONG || code == T_DOUBLE;
97         }
98     
99         /**
100          * Get a Variable representing the stack value which will either be stored
101          * into or loaded from a field.
102          *
103          * @param fieldIns
104          *            the FieldInstruction accessing the field
105          * @param cpg
106          *            the ConstantPoolGen for the method
107          * @param frame
108          *            the ValueNumberFrame containing the value to be stored or the
109          *            value loaded
110          */
111         protected static Variable snarfFieldValue(FieldInstruction fieldIns, ConstantPoolGen cpg, ValueNumberFrame frame)
112                 throws DataflowAnalysisException {
113     
114             if (isLongOrDouble(fieldIns, cpg)) {
115                 int numSlots = frame.getNumSlots();
116                 ValueNumber topValue = frame.getValue(numSlots - 1);
117                 ValueNumber nextValue = frame.getValue(numSlots - 2);
118                 return new LongOrDoubleLocalVariable(topValue, nextValue);
119             } else {
120                 return new LocalVariable(frame.getTopValue());
121             }
122     
123         }
124     }
