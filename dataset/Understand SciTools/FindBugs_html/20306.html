20     package edu.umd.cs.findbugs.ba;
21     
22     import org.apache.bcel.classfile.JavaClass;
23     import org.apache.bcel.classfile.Method;
24     
25     import edu.umd.cs.findbugs.OpcodeStack;
26     import edu.umd.cs.findbugs.SystemProperties;
27     import edu.umd.cs.findbugs.bcel.OpcodeStackDetector;
28     import edu.umd.cs.findbugs.internalAnnotations.DottedClassName;
29     
30     /**
31      * @author pwilliam
32      */
33     public class OpcodeStackScanner {
34     
35         static final boolean DEBUG = SystemProperties.getBoolean("oss.debug");
36     
37         public static class UnreachableCodeException extends RuntimeException {
38     
39             public UnreachableCodeException( @DottedClassName String className, String methodName, String methodSignature, int pc) {
40                 super("Didn't reach pc " + pc + " of " + className + "." + methodName + methodSignature);
41                 this.className = className;
42                 this.methodName = methodName;
43                 this.methodSignature = methodSignature;
44                 this.pc = pc;
45             }
46             @DottedClassName String className;
47             String methodName;
48             String methodSignature;
49             int pc;
50         }
51     
52         static class EarlyExitException extends RuntimeException {
53             final OpcodeStack stack;
54     
55             public EarlyExitException(OpcodeStack stack) {
56     
57                 this.stack = stack;
58             }
59         }
60     
61         public static OpcodeStack getStackAt(JavaClass theClass, Method method, int pc) {
62             Scanner scanner = new Scanner(theClass, method, pc);
63             try {
64                 scanner.execute();
65             } catch (EarlyExitException e) {
66                 return e.stack;
67             }
68             throw new UnreachableCodeException(theClass.getClassName(), method.getName(), method.getSignature(), pc);
69         }
70     
71         static class Scanner extends OpcodeStackDetector {
72     
73             Scanner(JavaClass theClass, Method targetMethod, int targetPC) {
74                 if(DEBUG) {
75                     System.out.println("Scanning " + theClass.getClassName() + "." + targetMethod.getName());
76                 }
77                 this.theClass = theClass;
78                 this.targetMethod = targetMethod;
79                 this.targetPC = targetPC;
80             }
81     
82             final JavaClass theClass;
83     
84             final Method targetMethod;
85     
86             final int targetPC;
87     
88             @Override
89             public void sawOpcode(int seen) {
90             }
91     
92             @Override
93             public void afterOpcode(int seen) {
94                 if(DEBUG) {
95                     System.out.printf("%3d: %8s %s%n", getPC(), OPCODE_NAMES[seen], getStack());
96                 }
97                 if (getPC() == targetPC) {
98                     throw new EarlyExitException(stack);
99                 }
100                 super.afterOpcode(seen);
101             }
102     
103             @Override
104             public void visitJavaClass(JavaClass obj) {
105                 setupVisitorForClass(obj);
106                 getConstantPool().accept(this);
107                 doVisitMethod(targetMethod);
108             }
109     
110             public void execute() {
111                 theClass.accept(this);
112             }
113         }
114     
115     }
