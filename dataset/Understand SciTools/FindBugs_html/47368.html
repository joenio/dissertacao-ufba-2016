20     package edu.umd.cs.findbugs;
21     
22     import edu.umd.cs.findbugs.model.ClassNameRewriter;
23     import edu.umd.cs.findbugs.model.ClassNameRewriterUtil;
24     import edu.umd.cs.findbugs.model.IdentityClassNameRewriter;
25     
26     /**
27      * Very sloppy bug comparator: if the warnings are of the same type, and in the
28      * same class/method/field, assume they are the same.
29      *
30      * @author David Hovemeyer
31      */
32     public class SloppyBugComparator implements WarningComparator {
33     
34         private static final boolean DEBUG = SystemProperties.getBoolean("sloppyComparator.debug");
35     
36         private ClassNameRewriter classNameRewriter = IdentityClassNameRewriter.instance();
37     
38         public SloppyBugComparator() {
39         }
40     
41         @Override
42         public void setClassNameRewriter(ClassNameRewriter classNameRewriter) {
43             this.classNameRewriter = classNameRewriter;
44         }
45     
46         private int compareNullElements(Object lhs, Object rhs) {
47             if (lhs == null && rhs == null) {
48                 return 0;
49             } else {
50                 return (lhs == null) ? -1 : 1;
51             }
52         }
53     
54         /**
55          * Compare class annotations.
56          *
57          * @param lhs
58          *            left hand class annotation
59          * @param rhs
60          *            right hand class annotation
61          * @return comparison of the class annotations
62          */
63         private int compareClassesAllowingNull(ClassAnnotation lhs, ClassAnnotation rhs) {
64             if (lhs == null || rhs == null) {
65                 return compareNullElements(lhs, rhs);
66             }
67     
68             String lhsClassName = classNameRewriter.rewriteClassName(lhs.getClassName());
69             String rhsClassName = classNameRewriter.rewriteClassName(rhs.getClassName());
70     
71             if (DEBUG) {
72                 System.err.println("Comparing " + lhsClassName + " and " + rhsClassName);
73             }
74     
75             int cmp = lhsClassName.compareTo(rhsClassName);
76             if (DEBUG) {
77                 System.err.println("\t==> " + cmp);
78             }
79             return cmp;
80         }
81     
82         private int compareMethodsAllowingNull(MethodAnnotation lhs, MethodAnnotation rhs) {
83             if (lhs == null || rhs == null) {
84                 return compareNullElements(lhs, rhs);
85             }
86             lhs = convertMethod(lhs);
87             rhs = convertMethod(rhs);
88             return lhs.compareTo(rhs);
89         }
90     
91         private int compareFieldsAllowingNull(FieldAnnotation lhs, FieldAnnotation rhs) {
92             if (lhs == null || rhs == null) {
93                 return compareNullElements(lhs, rhs);
94             }
95             lhs = convertField(lhs);
96             rhs = convertField(rhs);
97             if (DEBUG) {
98                 System.err.println("Compare fields: " + lhs + " and " + rhs);
99             }
100             return lhs.compareTo(rhs);
101         }
102     
103         private MethodAnnotation convertMethod(MethodAnnotation methodAnnotation) {
104             return ClassNameRewriterUtil.convertMethodAnnotation(classNameRewriter, methodAnnotation);
105         }
106     
107         private FieldAnnotation convertField(FieldAnnotation fieldAnnotation) {
108             return ClassNameRewriterUtil.convertFieldAnnotation(classNameRewriter, fieldAnnotation);
109         }
110     
111         @Override
112         public int compare(BugInstance lhs, BugInstance rhs) {
113     
114             int cmp;
115     
116             // Bug abbrevs must match
117             BugPattern lhsPattern = lhs.getBugPattern();
118             BugPattern rhsPattern = rhs.getBugPattern();
119             String lhsAbbrev, rhsAbbrev;
120             lhsAbbrev = lhsPattern.getAbbrev();
121             rhsAbbrev = rhsPattern.getAbbrev();
122             cmp = lhsAbbrev.compareTo(rhsAbbrev);
123             if (cmp != 0) {
124                 if (DEBUG) {
125                     System.err.println("bug abbrevs do not match");
126                 }
127                 return cmp;
128             }
129     
130             // Primary class must match
131             cmp = compareClassesAllowingNull(lhs.getPrimaryClass(), rhs.getPrimaryClass());
132             if (cmp != 0) {
133                 return cmp;
134             }
135     
136             boolean havePrimaryMethods = lhs.getPrimaryMethod() != null &&  rhs.getPrimaryMethod() != null;
137             // Primary method must match (if any)
138             cmp = compareMethodsAllowingNull(lhs.getPrimaryMethod(), rhs.getPrimaryMethod());
139             if (cmp != 0) {
140                 if (DEBUG) {
141                     System.err.println("primary methods do not match");
142                 }
143                 return cmp;
144             }
145     
146             if (!havePrimaryMethods) {
147                 // Primary field must match (if any)
148                 cmp = compareFieldsAllowingNull(lhs.getPrimaryField(), rhs.getPrimaryField());
149                 if (cmp != 0) {
150                     if (DEBUG) {
151                         System.err.println("primary fields do not match");
152                     }
153                     return cmp;
154                 }
155             }
156     
157             // Assume they're the same
158             return 0;
159         }
