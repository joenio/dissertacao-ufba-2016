131         public boolean initialize() throws IOException {
132             abstractCloudInitialized = true;
133             String modeString = getCloudProperty("votingmode");
134             Mode newMode = DEFAULT_VOTING_MODE;
135             if (modeString != null) {
136                 try {
137                     newMode = Mode.valueOf(modeString.toUpperCase());
138                 } catch (IllegalArgumentException e) {
139                     LOGGER.log(Level.WARNING, "No such voting mode " + modeString, e);
140                 }
141             }
142             setMode(newMode);
143     
144             String sp = properties.getProperty("findbugs.sourcelink.pattern");
145             String sf = properties.getProperty("findbugs.sourcelink.format");
146             String sfwl = properties.getProperty("findbugs.sourcelink.formatWithLine");
147     
148             String stt = properties.getProperty("findbugs.sourcelink.tooltip");
149             if (sp != null && sf != null) {
150                 try {
151                     this.sourceFileLinkPattern = Pattern.compile(sp);
152                     this.sourceFileLinkFormat = sf;
153                     this.sourceFileLinkToolTip = stt;
154                     this.sourceFileLinkFormatWithLine = sfwl;
155                 } catch (RuntimeException e) {
156                     LOGGER.log(Level.WARNING, "Could not compile pattern " + sp, e);
157                     if (THROW_EXCEPTION_IF_CANT_CONNECT) {
158                         throw e;
159                     }
160                 }
161             }
162             return true;
163         }

292         private String getSelectiveCloudReport(BugInstance b, Set<String> usersToExclude) {
293             if (!isInCloud(b)) {
294                 return notInCloudMsg(b);
295             }
296             initiateCommunication();
297             SimpleDateFormat format = new SimpleDateFormat("MM/dd, yyyy", Locale.ENGLISH);
298             StringBuilder builder = new StringBuilder();
299             long firstSeen = getFirstSeen(b);
300             builder.append(String.format("First seen %s%n", format.format(new Date(firstSeen))));
301             builder.append("\n");
302     
303             I18N i18n = I18N.instance();
304             boolean canSeeCommentsByOthers = canSeeCommentsByOthers(b);
305             if (canSeeCommentsByOthers && supportsBugLinks()) {
306                 BugFilingStatus bugLinkStatus = getBugLinkStatus(b);
307                 if (bugLinkStatus != null && bugLinkStatus.bugIsFiled()) {
308     
309                     builder.append("\nBug status is ").append(getBugStatus(b));
310                     if (getBugIsUnassigned(b)) {
311                         builder.append("\nBug is unassigned");
312                     }
313     
314                     builder.append("\n\n");
315                 }
316             }
317             String me = getUser();
318             for (BugDesignation d : getLatestDesignationFromEachUser(b)) {
319                 if (!usersToExclude.contains(d.getUser())
320                         && (me != null && me.equals(d.getUser()) || canSeeCommentsByOthers)) {
321                     builder.append(String.format("%s@ %s: %s%n", d.getUser() == null ? "" : d.getUser() + " ",
322                             format.format(new Date(d.getTimestamp())),
323                             i18n.getUserDesignation(d.getDesignationKey())));
324                     String annotationText = d.getAnnotationText();
325                     if (annotationText != null && annotationText.length() > 0) {
326                         builder.append(annotationText);
327                         builder.append("\n\n");
328                     }
329                 }
330             }
331             return builder.toString();
332         }

395         @Override
396         public UserDesignation getConsensusDesignation(BugInstance b) {
397             if (b == null) {
398                 throw new NullPointerException("null bug instance");
399             }
400             Multiset<UserDesignation> designations = new Multiset<UserDesignation>();
401             int count = 0;
402             int totalCount = 0;
403             double total = 0.0;
404             int isAProblem = 0;
405             int notAProblem = 0;
406             for (BugDesignation designation : getLatestDesignationFromEachUser(b)) {
407                 UserDesignation d = UserDesignation.valueOf(designation.getDesignationKey());
408                 if (d == UserDesignation.I_WILL_FIX) {
409                     d = UserDesignation.MUST_FIX;
410                 } else if (d == UserDesignation.UNCLASSIFIED) {
411                     continue;
412                 }
413                 switch (d) {
414                 case I_WILL_FIX:
415                 case MUST_FIX:
416                 case SHOULD_FIX:
417                     isAProblem++;
418                     break;
419                 case BAD_ANALYSIS:
420                 case NOT_A_BUG:
421                 case MOSTLY_HARMLESS:
422                 case OBSOLETE_CODE:
423                     notAProblem++;
424                     break;
425                 default:
426                     break;
427                 }
428                 designations.add(d);
429                 totalCount++;
430                 if (d.nonVoting()) {
431                     continue;
432                 }
433                 count++;
434                 total += d.score();
435             }
436             if (totalCount == 0) {
437                 return UserDesignation.UNCLASSIFIED;
438             }
439             UserDesignation mostCommonVotingDesignation = null;
440             UserDesignation mostCommonDesignation = null;
441     
442             for (Map.Entry<UserDesignation, Integer> e : designations.entriesInDecreasingFrequency()) {
443                 UserDesignation d = e.getKey();
444                 if (mostCommonVotingDesignation == null && !d.nonVoting()) {
445                     mostCommonVotingDesignation = d;
446                     if (e.getValue() > count / 2) {
447                         return d;
448                     }
449                 }
450                 if (mostCommonDesignation == null && d != UserDesignation.UNCLASSIFIED) {
451                     mostCommonDesignation = d;
452                     if (e.getValue() > count / 2) {
453                         return d;
454                     }
455                 }
456             }
457     
458             double score = total / count;
459             if (score >= UserDesignation.SHOULD_FIX.score() || isAProblem > notAProblem) {
460                 return UserDesignation.SHOULD_FIX;
461             }
462             if (score <= UserDesignation.NOT_A_BUG.score()) {
463                 return UserDesignation.NOT_A_BUG;
464             }
465             if (score <= UserDesignation.MOSTLY_HARMLESS.score() || notAProblem > isAProblem) {
466                 return UserDesignation.MOSTLY_HARMLESS;
467             }
468             return UserDesignation.NEEDS_STUDY;
469     
470         }

495         @Override
496         public double getClassificationVariance(BugInstance b) {
497     
498             int count = 0;
499             double total = 0.0;
500             double totalSquares = 0.0;
501             for (BugDesignation designation : getLatestDesignationFromEachUser(b)) {
502                 UserDesignation d = UserDesignation.valueOf(designation.getDesignationKey());
503                 if (d.nonVoting()) {
504                     continue;
505                 }
506                 count++;
507                 total += d.score();
508                 totalSquares += d.score() * d.score();
509             }
510             double average = total / count;
511             return totalSquares / count - average * average;
512     
513         }

540         @Override
541         @SuppressWarnings("boxing")
542         public void printCloudSummary(PrintWriter w, Iterable<BugInstance> bugs, String[] packagePrefixes) {
543     
544             Multiset<String> evaluations = new Multiset<String>();
545             Multiset<String> designations = new Multiset<String>();
546             Multiset<String> bugStatus = new Multiset<String>();
547     
548             int issuesWithThisManyReviews[] = new int[100];
549             I18N i18n = I18N.instance();
550     
551             int packageCount = 0;
552             int classCount = 0;
553             int ncss = 0;
554             ProjectStats projectStats = bugCollection.getProjectStats();
555             for (PackageStats ps : projectStats.getPackageStats()) {
556                 int num = ps.getNumClasses();
557                 if (ClassName.matchedPrefixes(packagePrefixes, ps.getPackageName()) && num > 0) {
558                     packageCount++;
559                     ncss += ps.size();
560                     classCount += num;
561                 }
562             }
563     
564             if (classCount == 0) {
565                 w.println("No classes were analyzed");
566                 return;
567             }
568             if (packagePrefixes != null && packagePrefixes.length > 0) {
569                 String lst = Arrays.asList(packagePrefixes).toString();
570                 w.println("Code analyzed in " + lst.substring(1, lst.length() - 1));
571             } else {
572                 w.println("Code analyzed");
573             }
574             w.printf("%,7d packages%n%,7d classes%n", packageCount, classCount);
575             if (ncss > 0) {
576                 w.printf("%,7d thousands of lines of non-commenting source statements%n", (ncss + 999) / 1000);
577             }
578             w.println();
579             int count = 0;
580             for (BugInstance bd : bugs) {
581     
582                 count++;
583                 HashSet<String> reviewers = new HashSet<String>();
584                 String status = supportsBugLinks() && getBugLinkStatus(bd).bugIsFiled() ? getBugStatus(bd) : null;
585                 if (status != null) {
586                     bugStatus.add(status);
587                 }
588     
589                 for (BugDesignation d : getLatestDesignationFromEachUser(bd)) {
590                     if (reviewers.add(d.getUser())) {
591                         evaluations.add(d.getUser());
592                         designations.add(i18n.getUserDesignation(d.getDesignationKey()));
593                     }
594                 }
595     
596                 int numReviews = Math.min(reviewers.size(), issuesWithThisManyReviews.length - 1);
597                 issuesWithThisManyReviews[numReviews]++;
598     
599             }
600             if (count == getBugCollection().getCollection().size()) {
601                 w.printf("Summary for %d issues%n%n", count);
602             } else {
603                 w.printf("Summary for %d issues that are in the current view%n%n", count);
604             }
605             if (evaluations.numKeys() == 0) {
606                 w.println("No reviews found");
607             } else {
608                 w.println("People who have performed the most reviews");
609                 printLeaderBoard(w, evaluations, 9, getUser(), true, "reviewer");
610                 w.println();
611                 w.println("Distribution of reviews");
612                 printLeaderBoard(w, designations, 100, " --- ", false, "designation");
613             }
614     
615             if (supportsBugLinks()) {
616                 if (bugStatus.numKeys() == 0) {
617                     w.println();
618                     w.println("No bugs filed");
619                 } else {
620                     w.println();
621                     w.println("Distribution of bug status");
622                     printLeaderBoard(w, bugStatus, 100, " --- ", false, "status of filed bug");
623                 }
624             }
625             w.println();
626             w.println("Distribution of number of reviews");
627             for (int i = 0; i < issuesWithThisManyReviews.length; i++) {
628                 if (issuesWithThisManyReviews[i] > 0) {
629                     w.printf("%4d  with %3d review", issuesWithThisManyReviews[i], i);
630                     if (i != 1) {
631                         w.print("s");
632                     }
633                     w.println();
634     
635                 }
636             }
637         }

639         @SuppressWarnings("boxing")
640         public static void printLeaderBoard2(PrintWriter w, Multiset<String> evaluations, int maxRows, String alwaysPrint,
641                 String format, String title) {
642             int row = 1;
643             int position = 0;
644             int previousScore = -1;
645             boolean foundAlwaysPrint = false;
646     
647             for (Map.Entry<String, Integer> e : evaluations.entriesInDecreasingFrequency()) {
648                 int num = e.getValue();
649                 if (num != previousScore) {
650                     position = row;
651                     previousScore = num;
652                 }
653                 String key = e.getKey();
654                 if (LEADERBOARD_BLACKLIST_PATTERN != null && LEADERBOARD_BLACKLIST_PATTERN.matcher(key).matches()) {
655                     continue;
656                 }
657     
658                 boolean shouldAlwaysPrint = key.equals(alwaysPrint);
659                 if (row <= maxRows || shouldAlwaysPrint) {
660                     w.printf(format, position, num, key);
661                 }
662     
663                 if (shouldAlwaysPrint) {
664                     foundAlwaysPrint = true;
665                 }
666                 row++;
667                 if (row >= maxRows) {
668                     if (alwaysPrint == null) {
669                         break;
670                     }
671                     if (foundAlwaysPrint) {
672                         w.printf("Total of %d %ss%n", evaluations.numKeys(), title);
673                         break;
674                     }
675                 }
676     
677             }
678         }

858         @Override
859         @SuppressWarnings("boxing")
860         public @CheckForNull URL getSourceLink(BugInstance b) {
861             if (sourceFileLinkPattern == null) {
862                 return null;
863             }
864     
865             SourceLineAnnotation src = b.getPrimarySourceLineAnnotation();
866             String fileName = src.getSourcePath();
867             int startLine = src.getStartLine();
868             int endLine = src.getEndLine();
869             java.util.regex.Matcher m = sourceFileLinkPattern.matcher(fileName);
870             boolean isMatch = m.matches();
871             if (isMatch) {
872                 try {
873                     URL link;
874                     if (startLine > 0) {
875                         link = new URL(String.format(sourceFileLinkFormatWithLine, m.group(1),
876                                 startLine, startLine - 10, endLine));
877                     } else {
878                         link = new URL(String.format(sourceFileLinkFormat, m.group(1)));
879                     }
880                     return link;
881                 } catch (Exception e) {
882                     AnalysisContext.logError("Error generating source link for " + src, e);
883                 }
884             }
885     
886             return null;
887     
888         }

943         public long getLocalFirstSeen(BugInstance b) {
944             long firstVersion = b.getFirstVersion();
945             AppVersion v = getBugCollection().getAppVersionFromSequenceNumber(firstVersion);
946             if (v == null) {
947                 return getBugCollection().getTimestamp();
948             }
949             long firstSeen = v.getTimestamp();
950             if (b.hasXmlProps()) {
951                 XmlProps props = b.getXmlProps();
952                 Date propsFirstSeen = props.getFirstSeen();
953                 if (propsFirstSeen != null && firstSeen > propsFirstSeen.getTime()) {
954                     firstSeen = propsFirstSeen.getTime();
955                 }
956             }
957             return firstSeen;
958         }
