99         @Override
100         public void visitClassContext(ClassContext classContext) {
101             JavaClass javaClass = classContext.getJavaClass();
102             Method[] methodList = javaClass.getMethods();
103     
104             for (Method method : methodList) {
105                 if (method.getCode() == null) {
106                     continue;
107                 }
108     
109                 try {
110                     analyzeMethod(classContext, method);
111                 } catch (MethodUnprofitableException e) {
112                     assert true; // move along; nothing to see
113                 } catch (CFGBuilderException e) {
114                     String msg = "Detector " + this.getClass().getName() + " caught exception while analyzing "
115                             + javaClass.getClassName() + "." + method.getName() + " : " + method.getSignature();
116                     bugReporter.logError(msg, e);
117                 } catch (DataflowAnalysisException e) {
118                     String msg = "Detector " + this.getClass().getName() + " caught exception while analyzing "
119                             + javaClass.getClassName() + "." + method.getName() + " : " + method.getSignature();
120                     bugReporter.logError(msg, e);
121                 }
122             }
123         }

569                             if (catchSize < 15) {
570                                 return;
571                             }
572                             if (catchSize < 25) {
573                                 priority++;
574                             }
575                             if (DEBUG) {
576                                 System.out.println(" priority i: " + priority);
577                             }
578     
579     
580                             if (priority < HIGH_PRIORITY) {
581                                 priority = HIGH_PRIORITY;
582                             }
583                             if (priority <= LOW_PRIORITY) {
584                                 String bug = "BC_UNCONFIRMED_CAST";
585                                 if (valueNumber.hasFlag(ValueNumber.RETURN_VALUE) || valueSource instanceof MethodAnnotation) {
586                                     bug = "BC_UNCONFIRMED_CAST_OF_RETURN_VALUE";
587                                 } else if (castToConcreteCollection) {
588                                     bug = "BC_BAD_CAST_TO_CONCRETE_COLLECTION";
589                                 } else if (castToAbstractCollection) {
590                                     bug = "BC_BAD_CAST_TO_ABSTRACT_COLLECTION";
591                                 }
592     
593                                 BugInstance bugInstance = new BugInstance(this, bug, priority)
594                                 .addClassAndMethod(methodGen, sourceFile).addFoundAndExpectedType(refType, castType)
595                                 .addOptionalAnnotation(valueSource);
596     
597                                 accumulator.accumulateBug(bugInstance, sourceLineAnnotation);
598                             }
599     
600                         }
601     
602                     }
603                 } catch (ClassNotFoundException e) {
604                     if (DEBUG) {
605                         e.printStackTrace(System.out);
606                     }
607                     if (isCast && "[Ljava/lang/Object;".equals(refSig) && source instanceof MethodAnnotation
608                             && "toArray".equals(((MethodAnnotation) source).getMethodName())
609                             && "()[Ljava/lang/Object;".equals(((MethodAnnotation) source).getMethodSignature())) {
610                         bugReporter.reportBug(new BugInstance(this,  "BC_IMPOSSIBLE_DOWNCAST_OF_TOARRAY", HIGH_PRIORITY)
611                         .addClassAndMethod(methodGen, sourceFile)
612                         .addFoundAndExpectedType(refType, castType).addOptionalUniqueAnnotations(valueSource, source)
613                         .addSourceLine(sourceLineAnnotation));
614                     }
615     
616     
617                 }
618             }
619             accumulator.reportAccumulatedBugs();
620         }
