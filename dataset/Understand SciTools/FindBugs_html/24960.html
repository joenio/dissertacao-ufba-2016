20     package edu.umd.cs.findbugs.ba.jsr305;
21     
22     import edu.umd.cs.findbugs.classfile.CheckedAnalysisException;
23     import edu.umd.cs.findbugs.internalAnnotations.DottedClassName;
24     
25     /**
26      * @author pugh
27      */
28     class ValidatorClassLoader extends ClassLoader {
29     
30         static {
31             if (TypeQualifierValue.DEBUG_CLASSLOADING) {
32                 new RuntimeException("Initialising ValidatorClassLoader").printStackTrace();
33             }
34     
35         }
36         final static ValidatorClassLoader INSTANCE = new ValidatorClassLoader();
37     
38         ValidatorClassLoader() {
39             super(ClassLoader.getSystemClassLoader().getParent());
40             if (TypeQualifierValue.DEBUG_CLASSLOADING) {
41                 new RuntimeException("Creating ValidatorClassLoader #").printStackTrace();
42             }
43         }
44     
45         @Override
46         protected Class<?> loadClass(String name, boolean resolve)
47                 throws ClassNotFoundException {
48     
49             if (TypeQualifierValue.DEBUG_CLASSLOADING) {
50                 if (resolve) {
51                     System.out.println("Loading and resolving class for " + name);
52                 } else {
53                     System.out.println("Loading class for " + name);
54                 }
55             }
56     
57             return super.loadClass(name, resolve);
58         }
59         @Override
60         public Class<?> findClass(@DottedClassName String name) throws ClassNotFoundException {
61             if (TypeQualifierValue.DEBUG_CLASSLOADING) {
62                 System.out.println("Looking for class data for " + name);
63             }
64     
65             if (name.startsWith("javax.annotation")) {
66                 return Class.forName(name);
67             }
68     
69             try {
70                 byte[] b = TypeQualifierValue.loadClassData(name);
71                 return findClass(name, b);
72             } catch (CheckedAnalysisException e) {
73                 if (TypeQualifierValue.DEBUG_CLASSLOADING) {
74                     e.printStackTrace();
75                 }
76                 return super.findClass(name);
77             } catch (RuntimeException e) {
78                 if (TypeQualifierValue.DEBUG_CLASSLOADING) {
79                     e.printStackTrace();
80                 }
81                 throw e;
82             }
83         }
84     
85     
86         private Class<?> findClass(@DottedClassName String name, byte [] b)  {
87             try {
88                 if (TypeQualifierValue.DEBUG_CLASSLOADING) {
89                     System.out.println("Loading " + b.length + " bytes for class " + name);
90                 }
91                 Class<?> result = defineClass(name, b, 0, b.length);
92                 super.resolveClass(result);
93                 if (TypeQualifierValue.DEBUG_CLASSLOADING) {
94                     System.out.println("defined class " + name);
95                 }
96                 return result;
97             } catch (RuntimeException e) {
98                 e.printStackTrace();
99                 throw e;
100             }
101     
102     
103         }
104     
105     
106     
107     }
