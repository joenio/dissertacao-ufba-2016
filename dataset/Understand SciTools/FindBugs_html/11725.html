48     /**
49      * Build a database of reference types stored into fields. This can be used in
50      * the future to improve the precision of type analysis when values are loaded
51      * from fields.
52      *
53      * @author David Hovemeyer
54      */
55     public class TrainFieldStoreTypes implements Detector, TrainingDetector {
56         private final BugReporter bugReporter;
57     
58         private final FieldStoreTypeDatabase database;
59     
60         public TrainFieldStoreTypes(BugReporter bugReporter) {
61             this.bugReporter = bugReporter;
62             this.database = new FieldStoreTypeDatabase();
63         }
64     
65         @Override
66         public void visitClassContext(ClassContext classContext) {
67             Method[] methodList = classContext.getJavaClass().getMethods();
68             for (Method method : methodList) {
69                 if (method.getCode() == null) {
70                     continue;
71                 }
72     
73                 try {
74                     analyzeMethod(classContext, method);
75                 } catch (CFGBuilderException e) {
76                     bugReporter.logError("Error compting field store types", e);
77                 } catch (DataflowAnalysisException e) {
78                     bugReporter.logError("Error compting field store types", e);
79                 }
80             }
81         }
82     
83         private void analyzeMethod(ClassContext classContext, Method method) throws CFGBuilderException, DataflowAnalysisException
84         {
85             CFG cfg = classContext.getCFG(method);
86             TypeDataflow typeDataflow = classContext.getTypeDataflow(method);
87             ConstantPoolGen cpg = classContext.getConstantPoolGen();
88     
89             for (Iterator<Location> i = cfg.locationIterator(); i.hasNext();) {
90                 Location location = i.next();
91                 Instruction ins = location.getHandle().getInstruction();
92                 short opcode = ins.getOpcode();
93     
94                 // Field store instruction?
95                 if (opcode != Constants.PUTFIELD && opcode != Constants.PUTSTATIC) {
96                     continue;
97                 }
98     
99                 // Check if field type is a reference type
100                 FieldInstruction fins = (FieldInstruction) ins;
101                 Type fieldType = fins.getType(cpg);
102                 if (!(fieldType instanceof ReferenceType)) {
103                     continue;
104                 }
105     
106                 // Find the exact field being stored into
107                 XField xfield = Hierarchy.findXField(fins, cpg);
108                 if (xfield == null) {
109                     continue;
110                 }
111     
112                 // Skip public and protected fields, since it is reasonable to
113                 // assume
114                 // we won't see every store to those fields
115                 if (xfield.isPublic() || xfield.isProtected()) {
116                     continue;
117                 }
118     
119                 // The top value on the stack is the one which will be stored
120                 // into the field
121                 TypeFrame frame = typeDataflow.getFactAtLocation(location);
122                 if (!frame.isValid()) {
123                     continue;
124                 }
125                 Type storeType = frame.getTopValue();
126                 if (!(storeType instanceof ReferenceType)) {
127                     continue;
128                 }
129     
130                 // Get or create the field store type set
131                 FieldStoreType property = database.getProperty(xfield.getFieldDescriptor());
132                 if (property == null) {
133                     property = new FieldStoreType();
134                     database.setProperty(xfield.getFieldDescriptor(), property);
135                 }
136     
137                 // Add the store type to the set
138                 property.addTypeSignature(storeType.getSignature());
139             }
140         }
141     
142         @Override
143         public void report() {
144             database.purgeBoringEntries();
145             AnalysisContext.currentAnalysisContext().storePropertyDatabase(database, FieldStoreTypeDatabase.DEFAULT_FILENAME,
146                     "store type database");
147         }
