67         enum RankFilter implements ViewFilterEnum {
68             SCARIEST(4, "Scariest"), SCARY(9, "Scary"), TROUBLING(14, "Troubling"), ALL(Integer.MAX_VALUE, "All bug ranks");
69             final int maxRank;
70     
71             final String displayName;
72     
73             private RankFilter(int maxRank, String displayName) {
74                 this.maxRank = maxRank;
75                 this.displayName = displayName;
76             }
77     
78             @Override
79             public boolean show(MainFrame mf, BugInstance b) {
80                 int rank = BugRanker.findRank(b);

94         enum OverallClassificationFilter implements ViewFilterEnum {
95             NOT_HARMLESS("Not classified as harmless") {
96                 @Override
97                 boolean show(Cloud cloud, BugInstance b) {
98                     double score = cloud.getClassificationScore(b);
99                     if (score <= UserDesignation.MOSTLY_HARMLESS.score()) {
100                         return false;
101                     }
102     
103                     score = cloud.getPortionObsoleteClassifications(b);
104                     if (score >= 0.5) {
105                         return false;
106                     }
107     
108                     return true;
109                 }
110             },
111             SHOULD_FIX("Overall classification is should fix") {
112                 @Override
113                 boolean show(Cloud cloud, BugInstance b) {
114                     double score = cloud.getClassificationScore(b);
115                     return score >= UserDesignation.SHOULD_FIX.score();
116                 }
117             },
118             DONT_FIX("Overall classification is don't fix") {
119                 @Override
120                 boolean show(Cloud cloud, BugInstance b) {
121                     double score = cloud.getClassificationScore(b);
122                     return score <= UserDesignation.MOSTLY_HARMLESS.score();
123                 }
124             },
125             OBSOLETE("Overall classification is obsolete code") {
126                 @Override
127                 boolean show(Cloud cloud, BugInstance b) {
128                     double score = cloud.getPortionObsoleteClassifications(b);
129                     return score >= 0.5;
130                 }
131             },
132             UNCERTAIN("Overall classification is uncertain") {
133                 @Override
134                 boolean show(Cloud cloud, BugInstance b) {
135                     if (SHOULD_FIX.show(cloud, b) || DONT_FIX.show(cloud, b) || OBSOLETE.show(cloud, b)) {
136                         return false;
137                     }
138                     if (cloud.getNumberReviewers(b) >= 2) {
139                         return true;
140                     }
141                     return false;
142                 }
143             },
144             HIGH_VARIANCE("Controversial") {
145                 @Override
146                 boolean show(Cloud cloud, BugInstance b) {
147                     double variance = cloud.getClassificationDisagreement(b);
148                     return variance > 0.26;
149                 }
150     
151             },
152             ALL("All issues") {
153     
154                 @Override
155                 public boolean show(MainFrame mf, BugInstance b) {
156                     return true;
157                 }
158     
159                 @Override
160                 boolean show(Cloud cloud, BugInstance b) {
161                     return true;
162                 }
163     
164             };
165             OverallClassificationFilter(String displayName) {
166                 this.displayName = displayName;
167             }

329         enum FirstSeenFilter implements ViewFilterEnum {
330             LAST_DAY(1, "Last day"), LAST_3_DAYS(3, "Last 3 days"), LAST_WEEK(7, "Last week"), LAST_MONTH(30, "Last month"), LAST_THREE_MONTHS(
331                     91, "Last 90 days"), LAST_YEAR(
332                             365, "Last year"), ALL(Integer.MAX_VALUE, "All time") {
333                 @Override
334                 public boolean show(MainFrame mf, BugInstance b) {
335                     return true;
336                 }
337     
338             };
339     
340             final int maxDays;
341     
342             final String displayName;
343     
344             private FirstSeenFilter(int days, String displayName) {
345                 this.maxDays = days;
346                 this.displayName = displayName;
347             }
348     
349             @Override
350             public boolean show(MainFrame mf, BugInstance b) {
351                 Cloud cloud = mf.getBugCollection().getCloud();
352                 if (!cloud.isInCloud(b)) {
353                     return false;
354                 }
355                 long firstSeen = cloud.getFirstSeen(b);
356                 long time = System.currentTimeMillis() - firstSeen;
357                 long days = TimeUnit.SECONDS.convert(time, TimeUnit.MILLISECONDS) / 3600 / 24;
358                 return days < this.maxDays;
359             }
360     
361             @Override
362             public String toString() {
363                 return displayName;
364             }
365         }

382         static final Pattern legalClassSearchString = Pattern.compile("[\\p{javaLowerCase}\\p{javaUpperCase}0-9.$/_]*");
383     
384         void setPackagesToDisplay(String value) {
385             value = value.replace('/', '.').trim();
386             if (value.length() == 0) {
387                 classSearchStrings = new String[0];
388             } else {
389                 String[] parts = value.split("[ ,:]+");
390                 for (String p : parts) {
391                     if (!legalClassSearchString.matcher(p).matches()) {
392                         throw new IllegalArgumentException("Classname filter must be legal Java identifier: " + p);
393                     }
394                 }
395     
396                 classSearchStrings = parts;
397             }
398             FilterActivity.notifyListeners(FilterListener.Action.FILTERING, null);
399         }

457         public boolean showIgnoringPackagePrefixes(BugInstance b) {
458             if (!firstSeen.show(mf, b)) {
459                 return false;
460             }
461             if (!rank.show(mf, b)) {
462                 return false;
463             }
464             if (!priority.show(mf, b)) {
465                 return false;
466             }
467             if (!eval.show(mf, b)) {
468                 return false;
469             }
470             if (!classificationFilter.show(mf, b)) {
471                 return false;
472             }
473             return true;
474         }

477          * @deprecated Use {@link ClassName#matchedPrefixes(String[],String)}
478          *             instead
479          */
480         @Deprecated
481         public static boolean matchedPrefixes(String[] classSearchStrings, @DottedClassName String className) {

485         public boolean show(BugInstance b) {
486     
487             String className = b.getPrimaryClass().getClassName();
488     
489             return ClassName.matchedPrefixes(classSearchStrings, className) && showIgnoringPackagePrefixes(b);
