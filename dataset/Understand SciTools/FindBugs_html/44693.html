20     package edu.umd.cs.findbugs.sourceViewer;
21     
22     import javax.swing.text.AbstractDocument;
23     import javax.swing.text.ComponentView;
24     import javax.swing.text.Element;
25     import javax.swing.text.IconView;
26     import javax.swing.text.LabelView;
27     import javax.swing.text.StyleConstants;
28     import javax.swing.text.View;
29     import javax.swing.text.ViewFactory;
30     
31     class NumberedViewFactory implements ViewFactory {
32         final HighlightInformation highlight;
33     
34         public NumberedViewFactory(HighlightInformation highlight) {
35             this.highlight = highlight;
36         }
37     
38         @Override
39         public View create(Element elem) {
40             String kind = elem.getName();
41             // System.out.println("Kind: " + kind);
42             if (kind != null) {
43                 if (AbstractDocument.ContentElementName.equals(kind)) {
44                     return new LabelView(elem);
45                 } else if (AbstractDocument.ParagraphElementName.equals(kind)) {
46                     return new NumberedParagraphView(elem, highlight);
47                 } else if (AbstractDocument.SectionElementName.equals(kind)) {
48                     return new NoWrapBoxView(elem, View.Y_AXIS);
49                 } else if (StyleConstants.ComponentElementName.equals(kind)) {
50                     return new ComponentView(elem);
51                 } else if (StyleConstants.IconElementName.equals(kind)) {
52                     return new IconView(elem);
53                 }
54             }
55             // default to text display
56             return new LabelView(elem);
57         }
58     }
