20     package edu.umd.cs.findbugs.workflow;
21     
22     import java.io.IOException;
23     
24     import edu.umd.cs.findbugs.BugCollection;
25     import edu.umd.cs.findbugs.BugInstance;
26     import edu.umd.cs.findbugs.FindBugs;
27     import edu.umd.cs.findbugs.Obfuscate;
28     import edu.umd.cs.findbugs.Project;
29     import edu.umd.cs.findbugs.ProjectPackagePrefixes;
30     import edu.umd.cs.findbugs.SortedBugCollection;
31     
32     public class ObfuscateBugs {
33         BugCollection bugCollection;
34     
35         public ObfuscateBugs() {
36         }
37     
38         public ObfuscateBugs(BugCollection bugCollection) {
39             this.bugCollection = bugCollection;
40         }
41     
42         public void setBugCollection(BugCollection bugCollection) {
43             this.bugCollection = bugCollection;
44         }
45     
46         public ObfuscateBugs execute() {
47             ProjectPackagePrefixes foo = new ProjectPackagePrefixes();
48     
49             for (BugInstance b : bugCollection.getCollection()) {
50                 foo.countBug(b);
51             }
52             foo.report();
53     
54             return this;
55         }
56     
57         static class CommandLine extends edu.umd.cs.findbugs.config.CommandLine {
58     
59             @Override
60             public void handleOption(String option, String optionalExtraPart) {
61                 throw new IllegalArgumentException("unknown option: " + option);
62             }
63     
64             /*
65              * (non-Javadoc)
66              *
67              * @see
68              * edu.umd.cs.findbugs.config.CommandLine#handleOptionWithArgument(java
69              * .lang.String, java.lang.String)
70              */
71             @Override
72             protected void handleOptionWithArgument(String option, String argument) throws IOException {
73                 throw new IllegalArgumentException("Unknown option : " + option);
74             }
75     
76         }
77     
78         public static void main(String[] args) throws Exception {
79             FindBugs.setNoAnalysis();
80             CommandLine commandLine = new CommandLine();
81     
82             int argCount = commandLine.parse(args, 0, 2, "Usage: " + ObfuscateBugs.class.getName() + " [options] [<xml results>] ");
83     
84             SortedBugCollection bugCollection = new SortedBugCollection();
85             if (argCount < args.length) {
86                 bugCollection.readXML(args[argCount++]);
87             } else {
88                 bugCollection.readXML(System.in);
89             }
90     
91             SortedBugCollection results = bugCollection.createEmptyCollectionWithMetadata();
92             Project project = results.getProject();
93             project.getSourceDirList().clear();
94             project.getFileList().clear();
95             project.getAuxClasspathEntryList().clear();
96     
97             results.getProjectStats().getPackageStats().clear();
98             results.clearMissingClasses();
99             results.clearErrors();
100     
101             for (BugInstance bug : bugCollection) {
102                 results.add(Obfuscate.obfuscate(bug), false);
103             }
104     
105             if (argCount == args.length) {
106                 results.writeXML(System.out);
107             } else {
108                 results.writeXML(args[argCount++]);
109     
110             }
111     
112         }
113     }
