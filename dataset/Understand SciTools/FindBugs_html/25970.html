20     package edu.umd.cs.findbugs.detect;
21     
22     import org.apache.bcel.Constants;
23     import org.apache.bcel.generic.ConstantPoolGen;
24     import org.apache.bcel.generic.Instruction;
25     import org.apache.bcel.generic.ObjectType;
26     
27     import edu.umd.cs.findbugs.ba.Hierarchy;
28     import edu.umd.cs.findbugs.ba.Location;
29     import edu.umd.cs.findbugs.ba.RepositoryLookupFailureCallback;
30     
31     /**
32      * StreamFactory for stream objects loaded from instance fields.
33      *
34      * @author David Hovemeyer
35      */
36     public class InstanceFieldLoadStreamFactory implements StreamFactory {
37         private final String streamBaseClass;
38     
39         private String bugPatternType;
40     
41         /**
42          * Constructor. By default, Streams created by this factory will not be
43          * marked as interesting. The setBugPatternType() method should be called to
44          * make the factory produce interesting streams.
45          *
46          * @param streamBaseClass
47          *            the base class of the streams produced by the factory
48          */
49         public InstanceFieldLoadStreamFactory(String streamBaseClass) {
50             this.streamBaseClass = streamBaseClass;
51         }
52     
53         /**
54          * Set the bug pattern type reported for unclosed streams loaded from this
55          * field. This makes the created streams "interesting".
56          *
57          * @param bugPatternType
58          *            the bug pattern type
59          */
60         public InstanceFieldLoadStreamFactory setBugPatternType(String bugPatternType) {
61             this.bugPatternType = bugPatternType;
62             return this;
63         }
64     
65         @Override
66         public Stream createStream(Location location, ObjectType type, ConstantPoolGen cpg,
67                 RepositoryLookupFailureCallback lookupFailureCallback) {
68     
69             Instruction ins = location.getHandle().getInstruction();
70             if (ins.getOpcode() != Constants.GETFIELD) {
71                 return null;
72             }
73     
74             String fieldClass = type.getClassName();
75             try {
76                 if (fieldClass.startsWith("[")) {
77                     return null;
78                 }
79                 if (!Hierarchy.isSubtype(fieldClass, streamBaseClass)) {
80                     return null;
81                 }
82     
83                 Stream stream = new Stream(location, fieldClass, streamBaseClass);
84                 stream.setIsOpenOnCreation(true);
85                 stream.setOpenLocation(location);
86                 if (bugPatternType != null) {
87                     stream.setInteresting(bugPatternType);
88                 }
89     
90                 // System.out.println("Instance field stream at " + location);
91                 return stream;
92             } catch (ClassNotFoundException e) {
93                 lookupFailureCallback.reportMissingClass(e);
94                 return null;
95             }
96         }
97     }
