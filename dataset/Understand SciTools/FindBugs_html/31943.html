30         /**
31          * Aligns the first <code>rows</code> * <code>cols</code> components of
32          * <code>parent</code> in a grid. Each component is as big as the maximum
33          * preferred width and height of the components. The parent is made just big
34          * enough to fit them all.
35          *
36          * @param rows
37          *            number of rows
38          * @param cols
39          *            number of columns
40          * @param initialX
41          *            x location to start the grid at
42          * @param initialY
43          *            y location to start the grid at
44          * @param xPad
45          *            x padding between cells
46          * @param yPad
47          *            y padding between cells
48          */
49         public static void makeGrid(Container parent, int rows, int cols, int initialX, int initialY, int xPad, int yPad) {
50             SpringLayout layout;
51             try {
52                 layout = (SpringLayout) parent.getLayout();
53             } catch (ClassCastException exc) {
54                 if (MainFrame.GUI2_DEBUG) {
55                     System.err.println("The first argument to makeGrid must use SpringLayout.");
56                 }
57                 return;
58             }
59     
60             Spring xPadSpring = Spring.constant(xPad);
61             Spring yPadSpring = Spring.constant(yPad);
62             Spring initialXSpring = Spring.constant(initialX);
63             Spring initialYSpring = Spring.constant(initialY);
64             int max = rows * cols;
65     
66             // Calculate Springs that are the max of the width/height so that all
67             // cells have the same size.
68             Spring maxWidthSpring = layout.getConstraints(parent.getComponent(0)).getWidth();
69             Spring maxHeightSpring = layout.getConstraints(parent.getComponent(0)).getWidth();
70             for (int i = 1; i < max; i++) {
71                 SpringLayout.Constraints cons = layout.getConstraints(parent.getComponent(i));
72     
73                 maxWidthSpring = Spring.max(maxWidthSpring, cons.getWidth());
74                 maxHeightSpring = Spring.max(maxHeightSpring, cons.getHeight());
75             }
76     
77             // Apply the new width/height Spring. This forces all the
78             // components to have the same size.
79             for (int i = 0; i < max; i++) {
80                 SpringLayout.Constraints cons = layout.getConstraints(parent.getComponent(i));
81     
82                 cons.setWidth(maxWidthSpring);
83                 cons.setHeight(maxHeightSpring);
84             }
85     
86             // Then adjust the x/y constraints of all the cells so that they
87             // are aligned in a grid.
88             SpringLayout.Constraints lastCons = null;
89             SpringLayout.Constraints lastRowCons = null;
90             for (int i = 0; i < max; i++) {
91                 SpringLayout.Constraints cons = layout.getConstraints(parent.getComponent(i));
92                 if (i % cols == 0) { // start of new row
93                     lastRowCons = lastCons;
94                     cons.setX(initialXSpring);
95                 } else { // x position depends on previous component
96                     assert lastCons != null;
97                     cons.setX(Spring.sum(lastCons.getConstraint(SpringLayout.EAST), xPadSpring));
98                 }
99     
100                 if (i / cols == 0) { // first row
101                     cons.setY(initialYSpring);
102                 } else { // y position depends on previous row
103                     assert lastRowCons != null;
104                     cons.setY(Spring.sum(lastRowCons.getConstraint(SpringLayout.SOUTH), yPadSpring));
105                 }
106                 lastCons = cons;
107             }
108     
109             assert lastCons != null;
110             // Set the parent's size.
111             SpringLayout.Constraints pCons = layout.getConstraints(parent);
112             pCons.setConstraint(SpringLayout.SOUTH, Spring.sum(Spring.constant(yPad), lastCons.getConstraint(SpringLayout.SOUTH)));
113             pCons.setConstraint(SpringLayout.EAST, Spring.sum(Spring.constant(xPad), lastCons.getConstraint(SpringLayout.EAST)));
114         }

116         /* Used by makeCompactGrid. */
117         private static SpringLayout.Constraints getConstraintsForCell(int row, int col, Container parent, int cols) {
118             SpringLayout layout = (SpringLayout) parent.getLayout();
119             Component c = parent.getComponent(row * cols + col);
120             return layout.getConstraints(c);
121         }

123         /**
124          * Aligns the first <code>rows</code> * <code>cols</code> components of
125          * <code>parent</code> in a grid. Each component in a column is as wide as
126          * the maximum preferred width of the components in that column; height is
127          * similarly determined for each row. The parent is made just big enough to
128          * fit them all.
129          *
130          * @param rows
131          *            number of rows
132          * @param cols
133          *            number of columns
134          * @param initialX
135          *            x location to start the grid at
136          * @param initialY
137          *            y location to start the grid at
138          * @param xPad
139          *            x padding between cells
140          * @param yPad
141          *            y padding between cells
142          */
143         public static void makeCompactGrid(Container parent, int rows, int cols, int initialX, int initialY, int xPad, int yPad) {
144             SpringLayout layout;
145             try {
146                 layout = (SpringLayout) parent.getLayout();
147             } catch (ClassCastException exc) {
148                 if (MainFrame.GUI2_DEBUG) {
149                     System.err.println("The first argument to makeCompactGrid must use SpringLayout.");
150                 }
151                 return;
152             }
153     
154             // Align all cells in each column and make them the same width.
155             Spring x = Spring.constant(initialX);
156             for (int c = 0; c < cols; c++) {
157                 Spring width = Spring.constant(0);
158                 for (int r = 0; r < rows; r++) {
159                     width = Spring.max(width, getConstraintsForCell(r, c, parent, cols).getWidth());
160                 }
161                 for (int r = 0; r < rows; r++) {
162                     SpringLayout.Constraints constraints = getConstraintsForCell(r, c, parent, cols);
163                     constraints.setX(x);
164                     constraints.setWidth(width);
165                 }
166                 x = Spring.sum(x, Spring.sum(width, Spring.constant(xPad)));
167             }
168     
169             // Align all cells in each row and make them the same height.
170             Spring y = Spring.constant(initialY);
171             for (int r = 0; r < rows; r++) {
172                 Spring height = Spring.constant(0);
173                 for (int c = 0; c < cols; c++) {
174                     height = Spring.max(height, getConstraintsForCell(r, c, parent, cols).getHeight());
175                 }
176                 for (int c = 0; c < cols; c++) {
177                     SpringLayout.Constraints constraints = getConstraintsForCell(r, c, parent, cols);
178                     constraints.setY(y);
179                     constraints.setHeight(height);
180                 }
181                 y = Spring.sum(y, Spring.sum(height, Spring.constant(yPad)));
182             }
183     
184             // Set the parent's size.
185             SpringLayout.Constraints pCons = layout.getConstraints(parent);
186             pCons.setConstraint(SpringLayout.SOUTH, y);
187             pCons.setConstraint(SpringLayout.EAST, x);
188         }
