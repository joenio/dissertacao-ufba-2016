20     package edu.umd.cs.findbugs.detect;
21     
22     import java.util.Arrays;
23     import java.util.List;
24     
25     import edu.umd.cs.findbugs.BugInstance;
26     import edu.umd.cs.findbugs.BugReporter;
27     import edu.umd.cs.findbugs.OpcodeStack;
28     import edu.umd.cs.findbugs.StringAnnotation;
29     import edu.umd.cs.findbugs.ba.ClassContext;
30     import edu.umd.cs.findbugs.bcel.OpcodeStackDetector;
31     import edu.umd.cs.findbugs.classfile.MethodDescriptor;
32     
33     /**
34      * Use whenever possible String.indexOf(int) instead of String.indexOf(String),
35      * or String.lastIndexOf(int) instead of String.lastIndexOf(String).
36      *
37      * @author Reto Merz
38      */
39     public class InefficientIndexOf extends OpcodeStackDetector {
40         private final BugReporter bugReporter;
41     
42         private static final List<MethodDescriptor> methods = Arrays.asList(
43                 new MethodDescriptor("java/lang/String", "indexOf", "(Ljava/lang/String;)I"),
44                 new MethodDescriptor("java/lang/String", "lastIndexOf", "(Ljava/lang/String;)I"),
45                 new MethodDescriptor("java/lang/String", "indexOf", "(Ljava/lang/String;I)I"),
46                 new MethodDescriptor("java/lang/String", "lastIndexOf", "(Ljava/lang/String;I)I")
47                 );
48     
49         public InefficientIndexOf(BugReporter bugReporter) {
50             this.bugReporter = bugReporter;
51         }
52     
53         @Override
54         public void visitClassContext(ClassContext classContext) {
55             if(hasInterestingMethod(classContext.getJavaClass().getConstantPool(), methods)) {
56                 super.visitClassContext(classContext);
57             }
58         }
59     
60         @Override
61         public void sawOpcode(int seen) {
62             if (seen == INVOKEVIRTUAL && stack.getStackDepth() > 0 && "java/lang/String".equals(getClassConstantOperand())) {
63     
64                 boolean lastIndexOf = "lastIndexOf".equals(getNameConstantOperand());
65                 if (lastIndexOf || "indexOf".equals(getNameConstantOperand())) {
66     
67                     int stackOff = -1;
68                     if ("(Ljava/lang/String;)I".equals(getSigConstantOperand())) { // sig: String
69                         stackOff = 0;
70                     } else if ("(Ljava/lang/String;I)I".equals(getSigConstantOperand())) { // sig: String, int
71                         stackOff = 1;
72                     }
73                     if (stackOff > -1) {
74                         OpcodeStack.Item item = stack.getStackItem(stackOff);
75                         Object o = item.getConstant();
76                         if (o != null && ((String) o).length() == 1) {
77                             bugReporter.reportBug(new BugInstance(this, lastIndexOf ? "IIO_INEFFICIENT_LAST_INDEX_OF" : "IIO_INEFFICIENT_INDEX_OF", LOW_PRIORITY).addClassAndMethod(this)
78                                     .describe(StringAnnotation.STRING_MESSAGE).addCalledMethod(this).addSourceLine(this));
79                         }
80                     }
81                 }
82             }
83         }
