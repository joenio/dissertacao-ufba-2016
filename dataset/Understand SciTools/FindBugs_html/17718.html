44         public FBFileChooser() {
45             addHiddenFileCheckBox();
46             assert java.awt.EventQueue.isDispatchThread();
47             this.setCurrentDirectory(GUISaveState.getInstance().getStarterDirectoryForLoadBugs());
48         }
49     
50         /**
51          * Sets size of font
52          *
53          * @param size
54          */
55         protected void setFontSize(float size) {
56             setFont(this.getFont().deriveFont(size));
57     
58             setFontSizeHelper(this.getComponents(), size);
59         }
60     
61         /*
62          * Helps above method, runs through all components recursively.
63          */
64         protected void setFontSizeHelper(Component[] comps, float size) {
65             if (comps.length <= 0) {
66                 return;
67             }
68     
69             for (Component comp : comps) {
70                 comp.setFont(comp.getFont().deriveFont(size));
71                 if (comp instanceof Container) {
72                     setFontSizeHelper(((Container) comp).getComponents(), size);
73                 }
74             }
75         }
76     
77         @Override
78         public void addNotify() {
79             super.addNotify();
80             setFontSize(Driver.getFontSize());
81     
82         }

84         private static void workAroundJFileChooserBug() {
85             // Travis McLeskey
86             // http://www.mcleskey.org/bugs.html
87             try {
88                 Object o = javax.swing.UIManager.getBorder("TableHeader.cellBorder");
89                 Method m = o.getClass().getMethod("setHorizontalShift", new Class[] { int.class });
90                 m.invoke(o, 0);
91             } catch (NoSuchMethodException e) {
92                 assert true;
93             } catch (InvocationTargetException e) {
94                 assert true;
95             } catch (IllegalAccessException e) {
96                 assert true;
97             }
98         }

102             assert java.awt.EventQueue.isDispatchThread();

104             if (SystemProperties.getProperty("os.name").startsWith("Mac")) {

139         private void addHiddenFileCheckBox() {
140             final JCheckBox showHiddenFileCheckBox = new JCheckBox("Show Hidden");
141             JPanel accessory = new JPanel();
142             accessory.setLayout(new FlowLayout());
143             accessory.add(showHiddenFileCheckBox);
144     
145             setAccessory(accessory);
146     
147             showHiddenFileCheckBox.addActionListener(new ActionListener() {
148                 @Override
149                 public void actionPerformed(ActionEvent ae) {
150                     setFileHidingEnabled(!showHiddenFileCheckBox.isSelected());
151                 }
152             });
153         }
