20     package edu.umd.cs.findbugs.detect;
21     
22     import org.apache.bcel.classfile.Code;
23     import org.apache.bcel.classfile.LineNumberTable;
24     
25     import edu.umd.cs.findbugs.BugInstance;
26     import edu.umd.cs.findbugs.BugReporter;
27     import edu.umd.cs.findbugs.OpcodeStack;
28     import edu.umd.cs.findbugs.StatelessDetector;
29     import edu.umd.cs.findbugs.bcel.OpcodeStackDetector;
30     
31     public class FindBadForLoop extends OpcodeStackDetector implements StatelessDetector {
32     
33         BugReporter bugReporter;
34     
35         public FindBadForLoop(BugReporter bugReporter) {
36             this.bugReporter = bugReporter;
37         }
38     
39         LineNumberTable lineNumbers;
40     
41         @Override
42         public void visit(Code obj) {
43             lastRegStore = -1;
44             lineNumbers = obj.getLineNumberTable();
45             super.visit(obj);
46         }
47     
48         int lastRegStore;
49     
50         @Override
51         public void sawOpcode(int seen) {
52             if (seen == ISTORE || seen == ISTORE_0 || seen == ISTORE_1 || seen == ISTORE_2 || seen == ISTORE_3) {
53                 lastRegStore = getRegisterOperand();
54             }
55             if (lineNumbers != null
56                     && stack.getStackDepth() >= 2
57                     && (seen == IF_ICMPGE || seen == IF_ICMPGT || seen == IF_ICMPLT || seen == IF_ICMPLE || seen == IF_ICMPNE || seen == IF_ICMPEQ)) {
58                 OpcodeStack.Item item0 = stack.getStackItem(0);
59                 OpcodeStack.Item item1 = stack.getStackItem(1);
60                 int r0 = item0.getRegisterNumber();
61                 int r1 = item1.getRegisterNumber();
62                 int rMin = Math.min(r0, r1);
63                 int rMax = Math.max(r0, r1);
64                 int branchTarget = getBranchTarget();
65                 if (rMin == -1 && rMax > 0 && rMax == lastRegStore && branchTarget - 6 > getPC()) {
66                     int beforeTarget = getCodeByte(branchTarget - 3);
67                     int beforeGoto = getCodeByte(branchTarget - 6);
68                     if (beforeTarget == GOTO && beforeGoto == IINC) {
69                         int offset1 = (byte) getCodeByte(branchTarget - 2);
70                         int offset2 = getCodeByte(branchTarget - 1);
71                         int offset = offset1 << 8 | offset2;
72                         int backTarget = branchTarget - 3 + offset;
73                         int reg = getCodeByte(branchTarget - 5);
74                         int testLineNumber = lineNumbers.getSourceLine(getPC());
75                         int incLineNumber = lineNumbers.getSourceLine(branchTarget - 6);
76                         int beforeIncLineNumber = lineNumbers.getSourceLine(branchTarget - 7);
77                         if (backTarget < getPC() && getPC() - 8 < backTarget && reg != rMax && incLineNumber < testLineNumber + 3
78                                 && beforeIncLineNumber > incLineNumber) {
79     
80                             bugReporter.reportBug(new BugInstance(this, "QF_QUESTIONABLE_FOR_LOOP", NORMAL_PRIORITY)
81                             .addClassAndMethod(this).addSourceLine(this));
82                         }
83                     }
84     
85                 }
86             }
87         }
88     
89     }
