20     package edu.umd.cs.findbugs;
21     
22     import java.util.ArrayList;
23     import java.util.List;
24     
25     import javax.annotation.CheckForNull;
26     
27     import org.apache.bcel.Constants;
28     
29     import edu.umd.cs.findbugs.ba.XClass;
30     import edu.umd.cs.findbugs.ba.XField;
31     import edu.umd.cs.findbugs.util.ClassName;
32     import edu.umd.cs.findbugs.visitclass.DismantleBytecode;
33     
34     public class SwitchHandler {
35         private final List<SwitchDetails> switchOffsetStack;
36     
37         public SwitchHandler() {
38             switchOffsetStack = new ArrayList<SwitchDetails>();
39         }
40     
41         public int stackSize() {
42             return switchOffsetStack.size();
43         }
44         int numEnumValues(@CheckForNull XClass c) {
45             if (c == null) {
46                 return -1;
47             }
48             int total = 0;
49             String enumSignature = ClassName.toSignature(c.getClassDescriptor().getClassName());
50             for(XField f : c.getXFields()) {
51                 if (f.getSignature().equals(enumSignature)
52                         && f.isPublic() && f.isFinal()) {
53                     total++;
54                 }
55             }
56             return total;
57         }
58     
59         public void enterSwitch(DismantleBytecode dbc, @CheckForNull XClass enumType) {
60     
61             assert dbc.getOpcode() == Constants.TABLESWITCH || dbc.getOpcode() == Constants.LOOKUPSWITCH;
62             int[] switchOffsets = dbc.getSwitchOffsets();
63             SwitchDetails details = new SwitchDetails(dbc.getPC(), switchOffsets, dbc.getDefaultSwitchOffset(), switchOffsets.length == numEnumValues(enumType));
64     
65     
66             int size = switchOffsetStack.size();
67             while (--size >= 0) {
68                 SwitchDetails existingDetail = switchOffsetStack.get(size);
69                 if (details.switchPC > (existingDetail.switchPC + existingDetail.swOffsets[existingDetail.swOffsets.length - 1])) {
70                     switchOffsetStack.remove(size);
71                 }
72             }
73             switchOffsetStack.add(details);
74         }
75     
76         public boolean isOnSwitchOffset(DismantleBytecode dbc) {
77             int pc = dbc.getPC();
78             if (pc == getDefaultOffset()) {
79                 return false;
80             }
81     
82             return (pc == getNextSwitchOffset(dbc));
83         }
84     
85         public int getNextSwitchOffset(DismantleBytecode dbc) {
86             int size = switchOffsetStack.size();
87             while (size > 0) {
88                 SwitchDetails details = switchOffsetStack.get(size - 1);
89     
90                 int nextSwitchOffset = details.getNextSwitchOffset(dbc.getPC());
91                 if (nextSwitchOffset >= 0) {
92                     return nextSwitchOffset;
93                 }
94     
95                 if (dbc.getPC() <= details.getDefaultOffset()) {
96                     return -1;
97                 }
98                 switchOffsetStack.remove(size - 1);
99                 size--;
100             }
101     
102             return -1;
103         }
104     
105         public int getDefaultOffset() {
106             int size = switchOffsetStack.size();
107             if (size == 0) {
108                 return -1;
109             }
110     
111             SwitchDetails details = switchOffsetStack.get(size - 1);
112             return details.getDefaultOffset();
113         }
114     
115         public SourceLineAnnotation getCurrentSwitchStatement(BytecodeScanningDetector detector) {
116             if (switchOffsetStack.isEmpty()) {
117                 throw new IllegalStateException("No current switch statement");
118             }
119             SwitchDetails details = switchOffsetStack.get(switchOffsetStack.size()-1);
120             return SourceLineAnnotation.fromVisitedInstructionRange(
121                     detector.getClassContext(), detector, details.switchPC, details.switchPC + details.maxOffset-1);
122         }
123         public static class SwitchDetails {
124             final int switchPC;
125     
126             final int[] swOffsets;
127     
128             final int defaultOffset;
129             final int maxOffset;
130     
131             int nextOffset;
132     
133             final boolean exhaustive;
134     
135             public SwitchDetails(int pc, int[] offsets, int defOffset, boolean exhaustive) {
136                 switchPC = pc;
137                 int uniqueOffsets = 0;
138                 int lastValue = -1;
139                 int maxOffset = defOffset;
140                 for (int offset : offsets) {
141                     if (maxOffset < offset) {
142                         maxOffset = offset;
143                     }
144                     if (offset == defOffset) {
145                         exhaustive = false;
146                     }
147                     if (offset != lastValue) {
148                         uniqueOffsets++;
149                         lastValue = offset;
150                     }
151                 }
152     
153                 this.maxOffset = maxOffset;
154                 swOffsets = new int[uniqueOffsets];
155                 int insertPos = 0;
156                 lastValue = -1;
157                 for (int offset1 : offsets) {
158                     if (offset1 != lastValue) {
159                         swOffsets[insertPos++] = offset1;
160                         lastValue = offset1;
161                     }
162                 }
163                 defaultOffset = defOffset;
164                 nextOffset = 0;
165                 this.exhaustive = exhaustive;
166             }
167     
168             public int getNextSwitchOffset(int currentPC) {
169                 while ((nextOffset < swOffsets.length) && (currentPC > (switchPC + swOffsets[nextOffset]))) {
170                     nextOffset++;
171                 }
172     
173                 if (nextOffset >= swOffsets.length) {
174                     return -1;
175                 }
176     
177                 return switchPC + swOffsets[nextOffset];
178             }
179     
180             public int getDefaultOffset() {
181                 if (exhaustive) {
182                     return Short.MIN_VALUE;
183                 }
184                 return switchPC + defaultOffset;
185             }
186         }
187     }
