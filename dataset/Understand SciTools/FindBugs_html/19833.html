56         @Override
57         public void visitClassContext(ClassContext classContext) {
58             JavaClass javaClass = classContext.getJavaClass();
59             boolean skip = false;
60             ConstantPool constantPool = javaClass.getConstantPool();
61             for(Constant c : constantPool.getConstantPool() ) {
62                 if (c instanceof ConstantMethodref || c instanceof ConstantInterfaceMethodref) {
63                     ConstantCP m = (ConstantCP) c;
64                     @DottedClassName String clazz = m.getClass(constantPool);
65                     ConstantNameAndType nt = (ConstantNameAndType) constantPool.getConstant(m.getNameAndTypeIndex(), Constants.CONSTANT_NameAndType);
66                     String name = nt.getName(constantPool);
67                     if ("setAttribute".equals(name) && "javax.servlet.http.HttpSession".equals(clazz) || ("writeObject".equals(name)
68                             && ("java.io.ObjectOutput".equals(clazz)
69                                     || "java.io.ObjectOutputStream".equals(clazz)))) {
70                         if (DEBUG) {
71                             System.out.println("Found call to " + clazz + "." + name);
72                         }
73     
74                         skip = false;
75                         break;
76                     }
77     
78                 }
79             }
80             if (skip) {
81                 return;
82             }
83             if (DEBUG) {
84                 System.out.println(this.getClass().getSimpleName() + " Checking " + javaClass.getClassName());
85             }
86             Method[] methodList = javaClass.getMethods();
87     
88             for (Method method : methodList) {
89                 if (method.getCode() == null) {
90                     continue;
91                 }
92     
93                 try {
94                     analyzeMethod(classContext, method);
95                 } catch (CFGBuilderException e) {
96                     bugReporter.logError("Detector " + this.getClass().getName() + " caught exception", e);
97                 } catch (DataflowAnalysisException e) {
98                     // bugReporter.logError("Detector " + this.getClass().getName()
99                     // + " caught exception", e);
100                 }
101                 bugAccumulator.reportAccumulatedBugs();
102             }
103         }

105         enum Use { STORE_INTO_HTTP_SESSION, PASSED_TO_WRITE_OBJECT, STORED_IN_SERIALZIED_FIELD }
106     
107         @CheckForNull Use getUse(ConstantPoolGen cpg, Instruction ins) {
108             if (ins instanceof InvokeInstruction) {
109                 InvokeInstruction invoke = (InvokeInstruction) ins;
110     
111                 String mName = invoke.getMethodName(cpg);
112                 String cName = invoke.getClassName(cpg);
113     
114                 if ("setAttribute".equals(mName) && "javax.servlet.http.HttpSession".equals(cName)) {
115                     return Use.STORE_INTO_HTTP_SESSION;
116                 }
117                 if ("writeObject".equals(mName)
118                         && ("java.io.ObjectOutput".equals(cName)
119                                 || "java.io.ObjectOutputStream".equals(cName))) {
120                     return Use.PASSED_TO_WRITE_OBJECT;
121                 }
122             }
123             return null;
124         }
125         private void analyzeMethod(ClassContext classContext, Method method) throws CFGBuilderException, DataflowAnalysisException {
126             MethodGen methodGen = classContext.getMethodGen(method);
127             if (methodGen == null) {
128                 return;
129             }
130             BitSet bytecodeSet = classContext.getBytecodeSet(method);
131             if (bytecodeSet == null) {
132                 return;
133             }
134             // We don't adequately model instanceof interfaces yet
135             if (bytecodeSet.get(Constants.INSTANCEOF) || bytecodeSet.get(Constants.CHECKCAST)) {
136                 return;
137             }
138             CFG cfg = classContext.getCFG(method);
139             TypeDataflow typeDataflow = classContext.getTypeDataflow(method);
140             ConstantPoolGen cpg = classContext.getConstantPoolGen();
141     
142             String sourceFile = classContext.getJavaClass().getSourceFileName();
143             if (DEBUG) {
144                 String methodName = methodGen.getClassName() + "." + methodGen.getName();
145                 System.out.println("Checking " + methodName);
146             }
147     
148             for (Iterator<Location> i = cfg.locationIterator(); i.hasNext();) {
149                 Location location = i.next();
150                 InstructionHandle handle = location.getHandle();
151                 Instruction ins = handle.getInstruction();
152     
153                 Use use = getUse(cpg, ins);
154                 if (use == null) {
155                     continue;
156                 }
157     
158                 TypeFrame frame = typeDataflow.getFactAtLocation(location);
159                 if (!frame.isValid()) {
160                     // This basic block is probably dead
161                     continue;
162                 }
163                 Type operandType = frame.getTopValue();
164     
165                 if (operandType.equals(TopType.instance())) {
166                     // unreachable
167                     continue;
168                 }
169                 if (!(operandType instanceof ReferenceType)) {
170                     // Shouldn't happen - illegal bytecode
171                     continue;
172                 }
173                 ReferenceType refType = (ReferenceType) operandType;
174     
175                 if (refType.equals(NullType.instance())) {
176                     continue;
177                 }
178     
179                 try {
180     
181                     double isSerializable = DeepSubtypeAnalysis.isDeepSerializable(refType);
182     
183                     if (isSerializable < 0.9) {
184                         SourceLineAnnotation sourceLineAnnotation = SourceLineAnnotation.fromVisitedInstruction(classContext,
185                                 methodGen, sourceFile, handle);
186                         ReferenceType problem = DeepSubtypeAnalysis.getLeastSerializableTypeComponent(refType);
187     
188                         String pattern;
189                         switch(use) {
190                         case PASSED_TO_WRITE_OBJECT:
191                             pattern = "DMI_NONSERIALIZABLE_OBJECT_WRITTEN";
192                             double isRemote = DeepSubtypeAnalysis.isDeepRemote(refType);
193                             if (isRemote >= 0.9) {
194                                 continue;
195                             }
196                             if (isSerializable < isRemote) {
197                                 isSerializable = isRemote;
198                             }
199                             break;
200                         case STORE_INTO_HTTP_SESSION:
201                             pattern = "J2EE_STORE_OF_NON_SERIALIZABLE_OBJECT_INTO_SESSION";
202                             break;
203                         default:
204                             throw new IllegalStateException();
205                         }
206     
207                         bugAccumulator.accumulateBug(new BugInstance(this, pattern,
208                                 isSerializable < 0.15 ? HIGH_PRIORITY : isSerializable > 0.5 ? LOW_PRIORITY : NORMAL_PRIORITY)
209                         .addClassAndMethod(methodGen, sourceFile).addType(problem).describe(TypeAnnotation.FOUND_ROLE),
210                         sourceLineAnnotation);
211     
212                     }
213                 } catch (ClassNotFoundException e) {
214                     // ignore
215                 }
216             }
217         }
