20     package edu.umd.cs.findbugs.io;
21     
22     import java.io.ByteArrayInputStream;
23     import java.io.IOException;
24     import java.util.Random;
25     
26     import org.junit.Assert;
27     import org.junit.Test;
28     
29     /**
30      * @author pugh
31      */
32     public class IOTest {
33     
34         Random r = new Random();
35     
36         private byte[] randomBytes(int size) {
37             byte [] result = new byte[size];
38             r.nextBytes(result);
39             return result;
40         }
41         @Test
42         public void testReadAllWithCorrectSize() throws IOException {
43     
44             for(int i = 10; i <= 10000; i *= 10) {
45                 byte [] input = randomBytes(i);
46                 byte [] output = IO.readAll(new ByteArrayInputStream(input), i);
47                 Assert.assertArrayEquals(input, output);
48             }
49         }
50         @Test
51         public void testReadAllWithSmallSize() throws IOException {
52     
53             for(int i = 10; i <= 10000; i *= 10) {
54                 byte [] input = randomBytes(i);
55                 byte [] output = IO.readAll(new ByteArrayInputStream(input), i-9);
56                 Assert.assertArrayEquals(input, output);
57             }
58         }
59         @Test
60         public void testReadAllWithLargeSize() throws IOException {
61     
62             for(int i = 10; i <= 10000; i *= 10) {
63                 byte [] input = randomBytes(i);
64                 byte [] output = IO.readAll(new ByteArrayInputStream(input), i+29);
65                 Assert.assertArrayEquals(input, output);
66             }
67         }
68         @Test
69         public void testReadAllWithoutSize() throws IOException {
70     
71             for(int i = 10; i <= 10000; i *= 10) {
72                 byte [] input = randomBytes(i);
73                 byte [] output = IO.readAll(new ByteArrayInputStream(input));
74                 Assert.assertArrayEquals(input, output);
75             }
76         }
77     }
