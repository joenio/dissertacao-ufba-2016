43     /**
44      * Register BCEL-framework analysis engines.
45      *
46      * <p>
47      * <b>NOTE</b>: the database factories will only work with
48      * AnalysisCacheToAnalysisContextAdapter, not with LegacyAnalysisContext.
49      * However, that's ok since the databases for BCEL-based analyses are only ever
50      * accessed through the AnalysisContext.
51      * </p>
52      *
53      * @author David Hovemeyer
54      */
55     public class EngineRegistrar implements IAnalysisEngineRegistrar {
56         private static final IClassAnalysisEngine<?>[] classAnalysisEngineList = { new ClassContextClassAnalysisEngine(),
57             new JavaClassAnalysisEngine(), new ConstantPoolGenFactory(),
58             // new AssignedFieldMapFactory(),
59             new AssertionMethodsFactory(), };
60     
61         private static final IMethodAnalysisEngine<?>[] methodAnalysisEngineList = { new MethodFactory(), new MethodGenFactory(),
62             new CFGFactory(), new UsagesRequiringNonNullValuesFactory(), new ValueNumberDataflowFactory(),
63             new IsNullValueDataflowFactory(), new TypeDataflowFactory(), new DepthFirstSearchFactory(),
64             new ReverseDepthFirstSearchFactory(), new UnpackedCodeFactory(), new LockDataflowFactory(), new LockCheckerFactory(),
65             new ReturnPathDataflowFactory(), new DominatorsAnalysisFactory(), new NonExceptionPostdominatorsAnalysisFactory(),
66             new NonImplicitExceptionPostDominatorsAnalysisFactory(), new ExceptionSetFactoryFactory(),
67             new ParameterSignatureListFactory(), new ConstantDataflowFactory(), new LoadDataflowFactory(),
68             new StoreDataflowFactory(), new LoadedFieldSetFactory(), new LiveLocalStoreDataflowFactory(),
69             new BlockTypeAnalysisFactory(), new CallListDataflowFactory(), new UnconditionalValueDerefDataflowFactory(),
70             new CompactLocationNumberingFactory(),  new ReturnPathTypeDataflowFactory(),
71             new ForwardTypeQualifierDataflowFactoryFactory(), new BackwardTypeQualifierDataflowFactoryFactory(),
72             new OpcodeStack.JumpInfoFactory(), new StackMapAnalysisFactory(), new ObligationDataflowFactory(),
73             new ValueRangeAnalysisFactory(), new FinallyDuplicatesInfoFactory()};
74     
75         private static final IDatabaseFactory<?>[] databaseFactoryList = {
76             // new ReflectionDatabaseFactory<Subtypes>(Subtypes.class),
77             new ReflectionDatabaseFactory<Subtypes2>(Subtypes2.class),
78             new ReflectionDatabaseFactory<InnerClassAccessMap>(InnerClassAccessMap.class),
79             new ReflectionDatabaseFactory<CheckReturnAnnotationDatabase>(CheckReturnAnnotationDatabase.class),
80             new ReflectionDatabaseFactory<AnnotationRetentionDatabase>(AnnotationRetentionDatabase.class),
81             new ReflectionDatabaseFactory<JCIPAnnotationDatabase>(JCIPAnnotationDatabase.class),
82             new ReflectionDatabaseFactory<SourceInfoMap>(SourceInfoMap.class),
83             new ReflectionDatabaseFactory<FieldStoreTypeDatabase>(FieldStoreTypeDatabase.class),
84             new ReflectionDatabaseFactory<ParameterNullnessPropertyDatabase>(ParameterNullnessPropertyDatabase.class),
85             new ReflectionDatabaseFactory<ReturnValueNullnessPropertyDatabase>(ReturnValueNullnessPropertyDatabase.class),
86             new ReflectionDatabaseFactory<DirectlyRelevantTypeQualifiersDatabase>(DirectlyRelevantTypeQualifiersDatabase.class),
87             new ReflectionDatabaseFactory<TypeQualifierDatabase>(TypeQualifierDatabase.class),
88             new ReflectionDatabaseFactory<MethodInfoDatabase>(MethodInfoDatabase.class),
89         };
90     
91         /*
92          * (non-Javadoc)
93          *
94          * @see edu.umd.cs.findbugs.classfile.IAnalysisEngineRegistrar#
95          * registerAnalysisEngines(edu.umd.cs.findbugs.classfile.IAnalysisCache)
96          */
97         @Override
98         public void registerAnalysisEngines(IAnalysisCache analysisCache) {
99             for (IClassAnalysisEngine<?> engine : classAnalysisEngineList) {
100                 engine.registerWith(analysisCache);
101             }
102     
103             for (IMethodAnalysisEngine<?> engine : methodAnalysisEngineList) {
104                 engine.registerWith(analysisCache);
105             }
106     
107             for (IDatabaseFactory<?> databaseFactory : databaseFactoryList) {
108                 databaseFactory.registerWith(analysisCache);
109             }
110     
111             // if (!Subtypes.DO_NOT_USE) {
112             // new
113             // ReflectionDatabaseFactory<Subtypes>(Subtypes.class).registerWith(analysisCache);
114             // }
115         }
116     
117     }
