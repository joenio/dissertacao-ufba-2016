20     package edu.umd.cs.findbugs.ba.vna;
21     
22     import edu.umd.cs.findbugs.ba.XField;
23     
24     /**
25      * An AvailableLoad indicates a field and (optionally) object reference for
26      * which a value is available. It is used to implement redundant load
27      * elimination and forward substitution in ValueNumberAnalysis. The idea is that
28      * programmers often reload fields unnecessarily when they "know" that the value
29      * will not change. In order to deduce the intended meaning of such code, our
30      * analyses need to figure out that such loads return the same value.
31      * <p/>
32      * <p>
33      * AvailableLoad objects may be used as keys in both hash and tree sets and
34      * maps.
35      *
36      * @author David Hovemeyer
37      * @see ValueNumberAnalysis
38      */
39     public class AvailableLoad implements Comparable<AvailableLoad> {
40         private final ValueNumber reference;
41     
42         private final XField field;
43     
44         /**
45          * Constructor from static field.
46          *
47          * @param staticField
48          *            the StaticField
49          */
50         public AvailableLoad(XField staticField) {
51             this.reference = null;
52             this.field = staticField;
53         }
54     
55         /**
56          * Constructor from object reference and instance field.
57          *
58          * @param reference
59          *            the ValueNumber of the object reference
60          * @param field
61          *            the InstanceField
62          */
63         public AvailableLoad(ValueNumber reference, XField field) {
64             if (reference == null) {
65                 throw new IllegalArgumentException();
66             }
67             this.reference = reference;
68             this.field = field;
69         }
70     
71         /**
72          * Get the ValueNumber of the object reference.
73          *
74          * @return the ValueNumber, or null if this is a an available static field
75          *         load
76          */
77         public ValueNumber getReference() {
78             return reference;
79         }
80     
81         public boolean matchesReference(ValueNumber v) {
82             if (v == reference) {
83                 return true;
84             }
85             if (reference == null) {
86                 return false;
87             }
88             return reference.equals(v);
89         }
90     
91         /**
92          * Get the field for which a load is available.
93          *
94          * @return the XField
95          */
96         public XField getField() {
97             return field;
98         }
99     
100         @Override
101         @SuppressWarnings("unchecked")
102         public int compareTo(AvailableLoad other) {
103             int cmp = field.compareTo(other.field);
104             if (cmp != 0) {
105                 return cmp;
106             } else if (reference == other.reference) {
107                 return 0;
108             } else if (reference == null) {
109                 return -1;
110             } else if (other.reference == null) {
111                 return 1;
112             } else {
113                 return reference.compareTo(other.reference);
114             }
115         }
116     
117         @Override
118         public int hashCode() {
119             return (reference == null ? 0 : reference.hashCode()) ^ field.hashCode();
120         }
121     
122         @Override
123         public boolean equals(Object o) {
124             if (o == null || this.getClass() != o.getClass()) {
125                 return false;
126             }
127             AvailableLoad other = (AvailableLoad) o;
128             return (reference == other.reference || (reference != null && other.reference != null && reference
129                     .equals(other.reference))) && field.equals(other.field);
130         }
131     
132         @Override
133         public String toString() {
134             return (reference == null ? "" : reference.getNumber() + ".") + field;
135         }
136     }
