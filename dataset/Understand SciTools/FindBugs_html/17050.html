20     package edu.umd.cs.findbugs.bugReporter;
21     
22     import java.lang.reflect.Constructor;
23     
24     import edu.umd.cs.findbugs.BugReporter;
25     import edu.umd.cs.findbugs.ComponentPlugin;
26     import edu.umd.cs.findbugs.DelegatingBugReporter;
27     
28     /**
29      * Abstract base class for bug reporters defined as plugins.
30      *
31      * @author pugh
32      */
33     public abstract class BugReporterDecorator extends DelegatingBugReporter {
34     
35         static public BugReporterDecorator construct(ComponentPlugin<BugReporterDecorator> plugin, BugReporter delegate) {
36     
37             Class<? extends BugReporterDecorator> pluginClass = plugin.getComponentClass();
38     
39             try {
40                 Constructor<? extends BugReporterDecorator> constructor = pluginClass.getConstructor(ComponentPlugin.class,
41                         BugReporter.class);
42                 return constructor.newInstance(plugin, delegate);
43             } catch (InstantiationException e) {
44                 throw new RuntimeException(e.getCause());
45     
46             } catch (Exception e) {
47                 throw new IllegalArgumentException("Unable to construct " + plugin.getId(), e);
48             }
49     
50         }
51     
52         final ComponentPlugin<BugReporterDecorator> plugin;
53     
54         public BugReporterDecorator(ComponentPlugin<BugReporterDecorator> plugin, BugReporter delegate) {
55             super(delegate);
56             this.plugin = plugin;
57     
58         }
59     
60     }
