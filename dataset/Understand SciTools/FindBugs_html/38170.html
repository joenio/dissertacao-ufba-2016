20     package edu.umd.cs.findbugs.ba.type;
21     
22     import java.util.Collection;
23     
24     import edu.umd.cs.findbugs.ba.CFG;
25     import edu.umd.cs.findbugs.ba.Dataflow;
26     import edu.umd.cs.findbugs.ba.DataflowAnalysisException;
27     import edu.umd.cs.findbugs.ba.Edge;
28     import edu.umd.cs.findbugs.ba.Location;
29     
30     public class TypeDataflow extends Dataflow<TypeFrame, TypeAnalysis> {
31         public static class LocationAndFactPair {
32             public final Location location;
33     
34             public final TypeFrame frame;
35     
36             LocationAndFactPair(Location location, TypeFrame frame) {
37                 this.location = location;
38                 this.frame = frame;
39             }
40         }
41     
42         public TypeDataflow(CFG cfg, TypeAnalysis analysis) {
43             super(cfg, analysis);
44         }
45     
46         public ExceptionSet getEdgeExceptionSet(Edge edge) {
47             return getAnalysis().getEdgeExceptionSet(edge);
48         }
49     
50         public LocationAndFactPair getLocationAndFactForInstruction(int pc) {
51             Collection<Location> locations = getCFG().getLocationsContainingInstructionWithOffset(pc);
52     
53             LocationAndFactPair result = null;
54     
55             // Return the first valid frame at any of the returned Locations
56             for (Location location : locations) {
57                 try {
58                     TypeFrame frame = getFactAtLocation(location);
59                     if (frame.isValid()) {
60                         result = new LocationAndFactPair(location, frame);
61                         break;
62                     }
63                 } catch (DataflowAnalysisException e) {
64                     // Ignore
65                 }
66             }
67     
68             return result;
69         }
70     }
