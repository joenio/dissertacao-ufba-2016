78         /**
79          * Performs an analysis and returns the BugSet created
80          *
81          * @param p
82          *            The Project to run the analysis on
83          * @param progressCallback
84          *            the progressCallBack is supposed to be supplied by analyzing
85          *            dialog, FindBugs supplies progress information while it runs
86          *            the analysis
87          * @return the bugs found
88          * @throws InterruptedException
89          * @throws IOException
90          */
91         public static BugCollection doAnalysis(@Nonnull Project p, FindBugsProgress progressCallback) throws IOException,
92         InterruptedException {
93             StringWriter stringWriter = new StringWriter();
94             BugCollectionBugReporter pcb = new BugCollectionBugReporter(p, new PrintWriter(stringWriter, true));
95             pcb.setPriorityThreshold(Priorities.LOW_PRIORITY);
96             IFindBugsEngine fb = createEngine(p, pcb);
97             fb.setUserPreferences(getUserPreferences());
98             fb.setProgressCallback(progressCallback);
99             fb.setProjectName(p.getProjectName());
100     
101             fb.execute();
102             String warnings = stringWriter.toString();
103             if (warnings.length() > 0) {
104                 JTextArea tp = new JTextArea(warnings);
105                 tp.setEditable(false);
106                 JScrollPane pane = new JScrollPane(tp);
107                 pane.setPreferredSize(new Dimension(600, 400));
108                 JOptionPane.showMessageDialog(MainFrame.getInstance(),
109                         pane, "Analysis errors",
110                         JOptionPane.WARNING_MESSAGE);
111             }
112     
113             return pcb.getBugCollection();
114         }

140         public static @CheckForNull
141         SortedBugCollection loadBugs(MainFrame mainFrame, Project project, File source) {
142             if (!source.isFile() || !source.canRead()) {
143                 JOptionPane.showMessageDialog(mainFrame, "Unable to read " + source);
144                 return null;
145             }
146             SortedBugCollection col = new SortedBugCollection(project);
147             try {
148                 col.readXML(source);
149                 initiateCommunication(col);
150                 if (col.hasDeadBugs()) {
151                     addDeadBugMatcher(col);
152                 }
153             } catch (Exception e) {
154                 e.printStackTrace();
155                 JOptionPane.showMessageDialog(mainFrame, "Could not read " + source + "; " + e.getMessage());
156             }
157             MainFrame.getInstance().setProjectAndBugCollectionInSwingThread(project, col);
158             return col;
159         }

164         }

166         public static @CheckForNull
167         SortedBugCollection loadBugs(MainFrame mainFrame, Project project, URL url) {
168     
169             SortedBugCollection col = new SortedBugCollection(project);
170             try {
171                 if (MainFrame.GUI2_DEBUG) {
172                     System.out.println("loading from: " + url);
173                     JOptionPane.showMessageDialog(mainFrame, "loading from: " + url);
174     
175                 }
176                 col.readXML(url);
177                 if (MainFrame.GUI2_DEBUG) {
178                     System.out.println("finished reading: " + url);
179                     JOptionPane.showMessageDialog(mainFrame, "loaded: " + url);
180                 }
181                 initiateCommunication(col);
182                 addDeadBugMatcher(col);
183     
184             } catch (Throwable e) {
185                 String msg = SystemProperties.getOSDependentProperty("findbugs.unableToLoadViaURL");
186                 if (msg == null) {
187                     msg = e.getMessage();
188                 } else {
189                     try {
190                         msg = String.format(msg, url);
191                     } catch (Exception e2) {
192                         msg = e.getMessage();
193                     }
194                 }
195                 JOptionPane.showMessageDialog(mainFrame, "Could not read " + url + "\n" + msg);
196                 if (SystemProperties.getBoolean("findbugs.failIfUnableToLoadViaURL")) {
197                     System.exit(1);
198                 }
199             }
200             MainFrame.getInstance().setProjectAndBugCollectionInSwingThread(project, col);
201             return col;
202         }

204         static void addDeadBugMatcher(BugCollection bugCollection) {
205             if (bugCollection == null  || !bugCollection.hasDeadBugs()) {
206                 return;
207             }
208     
209             Filter suppressionMatcher = bugCollection.getProject().getSuppressionFilter();
210             suppressionMatcher.softAdd(LastVersionMatcher.DEAD_BUG_MATCHER);
211         }

213         public static @CheckForNull
214         Project loadProject(MainFrame mainFrame, File f) {
215             try {
216                 Project project = Project.readXML(f);
217                 project.setGuiCallback(mainFrame.getGuiCallback());
218                 return project;
219             } catch (IOException e) {
220                 JOptionPane.showMessageDialog(mainFrame, "Could not read " + f + "; " + e.getMessage());
221             } catch (SAXException e) {
222                 JOptionPane.showMessageDialog(mainFrame, "Could not read  project from " + f + "; " + e.getMessage());
223             }
224             return null;
225         }

235          *
236          * Merges bug collection histories from xmls selected by the user. Right now

238          * once Makes use of FindBugs's mergeCollection method in the Update class
239          * of the workflow package

242          */
243         public static BugCollection combineBugHistories() {
244             try {
245                 FBFileChooser chooser = new FBFileChooser();
246                 chooser.setFileFilter(new FindBugsAnalysisFileFilter());
247                 // chooser.setCurrentDirectory(GUISaveState.getInstance().getStarterDirectoryForLoadBugs());
248                 // This is done by FBFileChooser.
249                 chooser.setMultiSelectionEnabled(true);
250                 chooser.setDialogTitle(edu.umd.cs.findbugs.L10N.getLocalString("dlg.choose_xmls_ttl", "Choose All XML's To Combine"));
251                 if (chooser.showOpenDialog(MainFrame.getInstance()) == JFileChooser.CANCEL_OPTION) {
252                     return null;
253                 }
254     
255                 SortedBugCollection conglomeration = new SortedBugCollection();
256                 conglomeration.readXML(chooser.getSelectedFiles()[0]);
257                 Update update = new Update();
258                 for (int x = 1; x < chooser.getSelectedFiles().length; x++) {
259                     File f = chooser.getSelectedFiles()[x];
260                     SortedBugCollection col = new SortedBugCollection();
261                     col.readXML(f);
262                     conglomeration = (SortedBugCollection) update.mergeCollections(conglomeration, col, false, false);// False
263                     // means
264                     // dont
265                     // show
266                     // dead
267                     // bugs
268                 }
269     
270                 return conglomeration;
271             } catch (IOException e) {
272                 Debug.println(e);
273                 return null;
274             } catch (DocumentException e) {
275                 Debug.println(e);
276                 return null;
277             }
278

290             requireNonNull(p, "null project");

292             RedoAnalysisCallback ac = new RedoAnalysisCallback();
293     
294             AnalyzingDialog.show(p, ac, true);
295     
296             if (ac.finished) {
297                 return ac.getBugCollection();
298             } else {
299                 return null;
300             }

304         /**
305          * Does what it says it does, hit apple r (control r on pc) and the analysis
306          * is redone using the current project
307          *
308          * @param p
309          * @return the bugs from the reanalysis, or null if canceled
310          */
311         public static @CheckForNull
312         BugCollection redoAnalysisKeepComments(@Nonnull Project p) {
313             requireNonNull(p, "null project");
314     
315             BugCollection current = MainFrame.getInstance().getBugCollection();
316     
317             Update update = new Update();
318     
319             RedoAnalysisCallback ac = new RedoAnalysisCallback();
320     
321             AnalyzingDialog.show(p, ac, true);
322     
323             if (!ac.finished) {
324                 return null;
325             }
326             if (current == null) {
327                 current =  ac.getBugCollection();
328             } else {
329                 current =  update.mergeCollections(current, ac.getBugCollection(), true, false);
330                 if (current.hasDeadBugs()) {
331                     addDeadBugMatcher(current);
332                 }
333             }
334             return current;
335     
336     
337         }

339         /** just used to know how the new analysis went */
340         private static class RedoAnalysisCallback implements AnalysisCallback {
341     
342             BugCollection getBugCollection() {
343                 return justAnalyzed;
344             }
345     
346             BugCollection justAnalyzed;
347     
348             volatile boolean finished;
349     
350             @Override
351             public void analysisFinished(BugCollection b) {
352                 justAnalyzed = b;
353                 finished = true;
354             }
355     
356             @Override
357             public void analysisInterrupted() {
358                 finished = false;
359             }
360         }
