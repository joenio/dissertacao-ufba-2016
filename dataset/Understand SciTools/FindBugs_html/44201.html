20     package edu.umd.cs.findbugs.detect;
21     
22     import static org.apache.bcel.Constants.*;
23     
24     import org.objectweb.asm.FieldVisitor;
25     import org.objectweb.asm.MethodVisitor;
26     import org.objectweb.asm.Opcodes;
27     
28     import edu.umd.cs.findbugs.BugInstance;
29     import edu.umd.cs.findbugs.BugReporter;
30     import edu.umd.cs.findbugs.MethodAnnotation;
31     import edu.umd.cs.findbugs.Priorities;
32     import edu.umd.cs.findbugs.asm.AbstractFBMethodVisitor;
33     import edu.umd.cs.findbugs.asm.ClassNodeDetector;
34     
35     /**
36      * Sample detector, using ASM
37      *
38      * @author David Hovemeyer
39      */
40     public class TestASM extends ClassNodeDetector {
41     
42         public TestASM(BugReporter bugReporter) {
43             super(bugReporter);
44         }
45     
46         @Override
47         public MethodVisitor visitMethod(final int access, final String name, final String desc, final String signature,
48                 final String[] exceptions) {
49             if (Character.isUpperCase(name.charAt(0))) {
50                 BugInstance bug0 = new BugInstance(this, "NM_METHOD_NAMING_CONVENTION", NORMAL_PRIORITY).addClass(this).addMethod(
51                         this.name, name, desc, access);
52                 bugReporter.reportBug(bug0);
53             }
54             return new AbstractFBMethodVisitor() {
55                 int prevOpcode;
56     
57                 int prevPC;
58     
59                 @Override
60                 public void visitInsn(int opcode) {
61                     prevOpcode = opcode;
62                     prevPC = getPC();
63                 }
64     
65                 @Override
66                 public void visitMethodInsn(int opcode, String owner, String invokedName, String invokedDesc, boolean itf) {
67                     if (prevPC + 1 == getPC() && prevOpcode == I2D && opcode == INVOKESTATIC && "java/lang/Math".equals(owner)
68                             && "ceil".equals(invokedName) && "(D)D".equals(invokedDesc)) {
69                         BugInstance bug0 = new BugInstance(TestASM.this, "ICAST_INT_CAST_TO_DOUBLE_PASSED_TO_CEIL", NORMAL_PRIORITY);
70                         MethodAnnotation methodAnnotation = MethodAnnotation.fromForeignMethod(TestASM.this.name, name, desc, access);
71                         bug0.addClass(TestASM.this).addMethod(methodAnnotation);
72                         bugReporter.reportBug(bug0);
73                     }
74                 }
75             };
76     
77         }
78     
79         @Override
80         public FieldVisitor visitField(int access, String name, String desc, String signature, Object value) {
81             if ((access & Opcodes.ACC_STATIC) != 0 && (access & Opcodes.ACC_FINAL) != 0 && (access & Opcodes.ACC_PUBLIC) != 0
82                     && !name.equals(name.toUpperCase())) {
83                 bugReporter.reportBug(new BugInstance(this, "NM_FIELD_NAMING_CONVENTION", Priorities.LOW_PRIORITY).addClass(this)
84                         .addField(this.name, name, desc, access));
85             }
86             return null;
87         }
88     
89     }
