19     package edu.umd.cs.findbugs;
20     
21     import java.io.InputStream;
22     import java.net.URL;
23     import java.util.List;
24     import java.util.concurrent.ExecutorService;
25     
26     import javax.swing.JComponent;
27     
28     import edu.umd.cs.findbugs.cloud.Cloud;
29     
30     /**
31      * Interface for any kind of GUI attached to the current FindBug analysis
32      *
33      * @author Andrei
34      */
35     public interface IGuiCallback {
36         /** If true, do not open windows or browsers */
37         boolean isHeadless();
38     
39         void showMessageDialog(String message);
40     
41         public final static int YES_OPTION = 0;
42     
43         public final static int NO_OPTION = 1;
44     
45         public final static int CANCEL_OPTION = 2;
46     
47         public static final int YES_NO_OPTION = 0;
48     
49         public static final int YES_NO_CANCEL_OPTION = 1;
50     
51         public static final int OK_CANCEL_OPTION = 2;
52     
53         void invokeInGUIThread(Runnable r);
54     
55         int showConfirmDialog(String message, String title, String ok, String cancel);
56     
57         String showQuestionDialog(String message, String title, String defaultValue);
58     
59         List<String> showForm(String message, String title, List<FormItem> labels);
60     
61         InputStream getProgressMonitorInputStream(InputStream in, int length, String msg);
62     
63         void setErrorMessage(String errorMsg);
64     
65         void displayNonmodelMessage(String title, String message);
66     
67         boolean showDocument(URL u);
68     
69         /**
70          * Called as soon as the cloud object is created, before it is initialized.
71          * Useful for adding status msg listener.
72          */
73         void registerCloud(Project project, BugCollection collection, Cloud cloud);
74     
75         void unregisterCloud(Project project, BugCollection collection, Cloud cloud);
76     
77         /**
78          * Use this executor to queue bug collection updates without interfering
79          * with the GUI. Runs on the AWT event thread.
80          */
81         ExecutorService getBugUpdateExecutor();
82     
83         void showMessageDialogAndWait(String message) throws InterruptedException;
84     
85         public class FormItem {
86             private final String label;
87     
88             private final String defaultValue;
89     
90             private boolean password = false;
91     
92             private final List<String> possibleValues;
93     
94             private JComponent field;
95     
96             private String currentValue;
97     
98             private List<FormItem> items;
99     
100             public FormItem(String label) {
101                 this(label, null, null);
102             }
103     
104             public FormItem(String label, String defaultValue) {
105                 this(label, defaultValue, null);
106             }
107     
108             public FormItem(String label, String defaultValue, List<String> possibleValues) {
109                 this.label = label;
110                 this.defaultValue = defaultValue;
111                 this.possibleValues = possibleValues;
112             }
113     
114             public FormItem password() {
115                 password = true;
116                 return this;
117             }
118     
119             public boolean isPassword() {
120                 return password;
121             }
122     
123             public String getLabel() {
124                 return label;
125             }
126     
127             public String getDefaultValue() {
128                 return defaultValue;
129             }
130     
131             public List<String> getPossibleValues() {
132                 return possibleValues;
133             }
134     
135             public JComponent getField() {
136                 return field;
137             }
138     
139             public void setField(JComponent field) {
140                 this.field = field;
141             }
142     
143             public void setItems(List<FormItem> items) {
144                 this.items = items;
145             }
146     
147             public List<FormItem> getItems() {
148                 return items;
149             }
150     
151             public void setCurrentValue(String currentValue) {
152                 this.currentValue = currentValue;
153             }
154     
155             public String getCurrentValue() {
156                 return currentValue;
157             }
158     
159             public void updated() {
160             }
161         }
162     }
