66             static class PatternMatcher {
67                 final Pattern[] pattern;
68     
69                 PatternMatcher(String arg) {
70                     String[] p = arg.split(",");
71                     this.pattern = new Pattern[p.length];
72                     for (int i = 0; i < p.length; i++) {
73                         pattern[i] = Pattern.compile(p[i]);
74                     }
75                 }
76     
77                 public boolean matches(String arg) {
78                     for (Pattern p : pattern) {
79                         if (p.matcher(arg).find()) {
80                             return true;
81                         }
82                     }
83     
84                     return false;
85                 }
86             }
87     
88             static class PrefixMatcher {
89                 final String[] prefixes;
90     
91                 PrefixMatcher(String arg) {
92                     this.prefixes = arg.split(",");
93                 }
94     
95                 PrefixMatcher() {
96                     this.prefixes = new String[0];
97                 }
98     
99                 public boolean matches(String arg) {
100                     for (String p : prefixes) {
101                         if (arg.startsWith(p)) {
102                             return true;
103                         }
104                     }
105     
106                     return false;
107                 }
108     
109                 public boolean matchesEverything() {
110                     for (String p : prefixes) {
111                         if (p.length() == 0) {
112                             return true;
113                         }
114                     }
115                     return false;
116                 }
117             }

311         private boolean exclude(String dottedName) {
312             if (!Character.isJavaIdentifierStart(dottedName.charAt(0))) {
313                 return true;
314             }
315             if (commandLine.excludePatterns != null && commandLine.excludePatterns.matches(dottedName)
316                     || commandLine.exclude.matches(dottedName)) {
317                 excluded.add(dottedName);
318                 return true;
319             }
320             return false;

323         boolean classFileFound;
324         public void execute() throws IOException {
325     
326             ArrayList<String> fileList = new ArrayList<String>();
327     
328             if (commandLine.inputFileList != null) {
329                 readFrom(fileList, UTF8.fileReader(commandLine.inputFileList));
330             } else if (argCount == args.length) {
331                 readFromStandardInput(fileList);
332             } else {
333                 fileList.addAll(Arrays.asList(args).subList(argCount, args.length));
334             }
335             ArrayList<String> auxFileList = new ArrayList<String>();
336             if (commandLine.auxFileList != null) {
337                 readFrom(auxFileList, UTF8.fileReader(commandLine.auxFileList));
338                 auxFileList.removeAll(fileList);
339             }
340     
341             List<File> inputZipFiles = new ArrayList<File>(fileList.size());
342             List<File> auxZipFiles = new ArrayList<File>(auxFileList.size());
343             for (String fInName : fileList) {
344                 final File f = new File(fInName);
345                 if (f.lastModified() < commandLine.maxAge) {
346                     System.err.println("Skipping " + fInName + ", too old (" + new Date(f.lastModified()) + ")");
347                     continue;
348                 }
349     
350                 int oldSize = copied.size();
351                 classFileFound = false;
352                 if (processZipEntries(f, new ZipElementHandler() {
353                     boolean checked = false;
354     
355                     @Override
356                     public void handle(ZipFile file, ZipEntry ze) throws IOException {
357                         if (commandLine.skip(ze)) {
358                             return;
359                         }
360                         String name = ze.getName();
361     
362                         String dottedName = name.replace('/', '.');
363                         if (exclude(dottedName)) {
364                             return;
365                         }
366     
367                         if (!checked) {
368                             checked = true;
369                             if (embeddedNameMismatch(file, ze)) {
370                                 System.out.println("Class name mismatch for " + name + " in " + file.getName());
371                                 throw new ClassFileNameMismatch();
372                             }
373                         }
374                         if (!commandLine.prefix.matches(dottedName)) {
375                             return;
376                         }
377                         classFileFound = true;
378                         long timestamp = ze.getTime();
379                         Long oldTimestamp = copied.get(name);
380                         if (oldTimestamp == null) {
381                             copied.put(name, timestamp);
382                             copyFrom.put(name, f);
383                             filesToAnalyze.add(name);
384                             numFilesToAnalyze++;
385                         } else if (!commandLine.ignoreTimestamps && oldTimestamp < timestamp) {
386                             System.out.printf("Found later version of %s; switching from %s to %s%n", name, copyFrom.get(name), f);
387                             copied.put(name, timestamp);
388                             copyFrom.put(name, f);
389                         }
390                     }
391     
392                     /**
393                      * @param dottedName
394                      * @return
395                      */
396     
397                 }) && oldSize < copied.size()) {
398                     inputZipFiles.add(f);
399                 } else if (classFileFound) {
400                     System.err.println("Skipping " + fInName  + ", no new classes found");
401                 } else {
402                     System.err.println("Skipping " + fInName  + ", no classes found");
403                 }
404             }
405             for (String fInName : auxFileList) {
406                 final File f = new File(fInName);
407                 if (f.lastModified() < commandLine.maxAge) {
408                     System.err.println("Skipping " + fInName + ", too old (" + new Date(f.lastModified()) + ")");
409                     continue;
410                 }
411                 int oldSize = copied.size();
412                 classFileFound = false;
413                 if (processZipEntries(f, new ZipElementHandler() {
414                     @Override
415                     public void handle(ZipFile file, ZipEntry ze) {
416                         if (commandLine.skip(ze)) {
417                             return;
418                         }
419     
420                         String name = ze.getName();
421                         String dottedName = name.replace('/', '.');
422                         if (!exclude(dottedName)) {
423                             classFileFound = true;
424                             long timestamp = ze.getTime();
425                             Long oldTimestamp = copied.get(name);
426                             if (oldTimestamp == null) {
427                                 copied.put(name, timestamp);
428                                 copyFrom.put(name, f);
429                             } else if (!commandLine.ignoreTimestamps && oldTimestamp < timestamp) {
430                                 copied.put(name, timestamp);
431                                 copyFrom.put(name, f);
432                             }
433                         }
434                     }
435                 }) && oldSize < copied.size()) {
436                     auxZipFiles.add(f);
437                 } else if (classFileFound) {
438                     System.err.println("Skipping aux file " + fInName  + ", no new classes found");
439                 } else {
440                     System.err.println("Skipping aux file" + fInName  + ", no classes found");
441                 }
442             }
443     
444             System.out.printf("    # Zip/jar files: %2d%n", inputZipFiles.size());
445             System.out.printf("# aux Zip/jar files: %2d%n", auxZipFiles.size());
446             System.out.printf("Unique class files: %6d%n", copied.size());
447             if (numFilesToAnalyze != copied.size()) {
448                 System.out.printf("  files to analyze: %6d%n", numFilesToAnalyze);
449             }
450     
451             if (!excluded.isEmpty()) {
452                 System.out.printf("   excluded  files: %6d%n", excluded.size());
453             }
454     
455             if (commandLine.onlyAnalyze) {
456                 return;
457             }
458     
459             if (numFilesToAnalyze < copied.size() || numFilesToAnalyze > commandLine.maxClasses) {
460                 auxilaryOut = createZipFile(getNextAuxilaryFileOutput());
461             }
462     
463             int count = Integer.MAX_VALUE;
464             String oldBaseClass = "x x";
465             String oldPackage = "x x";
466             for (String path : filesToAnalyze) {
467                 int lastSlash = path.lastIndexOf('/');
468                 String packageName = lastSlash <= 0 ? "" : path.substring(0, lastSlash - 1);
469                 int firstDollar = path.indexOf('$', lastSlash);
470                 String baseClass = firstDollar < 0 ? path : path.substring(0, firstDollar - 1);
471                 boolean switchOutput;
472                 if (count > commandLine.maxClasses) {
473                     switchOutput = true;
474                 } else if (count + 50 > commandLine.maxClasses && !baseClass.equals(oldBaseClass)) {
475                     switchOutput = true;
476                 } else if (count + 250 > commandLine.maxClasses && !packageName.equals(oldPackage)) {
477                     switchOutput = true;
478                 } else {
479                     switchOutput = false;
480                 }
481     
482                 if (switchOutput) {
483                     // advance
484                     String zipFileName = getNextAnalyzeFileOutput();
485                     analysisOutputFiles.put(path, createZipFile(zipFileName));
486                     System.out.printf("%s%n -> %s%n", path, zipFileName);
487                     count = 0;
488                 }
489                 count++;
490                 oldPackage = packageName;
491                 oldBaseClass = baseClass;
492             }
493     
494             for (File f : inputZipFiles) {
495                 System.err.println("Reading " + f);
496                 final File ff = f;
497                 processZipEntries(f, new ZipElementHandler() {
498     
499                     @Override
500                     public void handle(ZipFile zipInputFile, ZipEntry ze) throws IOException {
501                         if (commandLine.skip(ze)) {
502                             return;
503                         }
504     
505     
506                         String name = ze.getName();
507                         String dottedName = name.replace('/', '.');
508                         if (exclude(dottedName)) {
509                             return;
510                         }
511                         if (!ff.equals(copyFrom.get(name))) {
512                             return;
513                         }
514                         if (name.contains("DefaultProblem.class")) {
515                             System.out.printf("%40s %40s%n", name, ff);
516                         }
517     
518                         boolean writeToAnalyzeOut = false;
519                         boolean writeToAuxilaryOut = false;
520                         if (commandLine.prefix.matches(dottedName)) {
521                             writeToAnalyzeOut = true;
522                             if (numFilesToAnalyze > commandLine.maxClasses) {
523                                 writeToAuxilaryOut = true;
524                             }
525                         } else {
526                             writeToAuxilaryOut = auxilaryOut != null;
527                         }
528                         ZipOutputStream out = null;
529                         if (writeToAnalyzeOut) {
530                             out = getZipOutputFile(name);
531                             out.putNextEntry(newZipEntry(ze));
532                         }
533     
534                         if (writeToAuxilaryOut) {
535                             auxilaryClassCount++;
536                             if (auxilaryClassCount > 29999) {
537                                 auxilaryClassCount = 0;
538                                 advanceAuxilaryOut();
539                             }
540                             auxilaryOut.putNextEntry(newZipEntry(ze));
541                         }
542     
543                         copyEntry(zipInputFile, ze, writeToAnalyzeOut, out, writeToAuxilaryOut, auxilaryOut);
544                     }
545     
546                 });
547             }
548     
549             for (File f : auxZipFiles) {
550                 final File ff = f;
551                 System.err.println("Opening aux file " + f);
552                 processZipEntries(f, new ZipElementHandler() {
553     
554                     @Override
555                     public void handle(ZipFile zipInputFile, ZipEntry ze) throws IOException {
556                         if (commandLine.skip(ze)) {
557                             return;
558                         }
559     
560                         String name = ze.getName();
561                         String dottedName = name.replace('/', '.');
562     
563                         if (exclude(dottedName)) {
564                             return;
565                         }
566                         if (!ff.equals(copyFrom.get(name))) {
567                             return;
568                         }
569     
570                         auxilaryClassCount++;
571                         if (auxilaryClassCount > 29999) {
572                             auxilaryClassCount = 0;
573                             advanceAuxilaryOut();
574                         }
575                         auxilaryOut.putNextEntry(newZipEntry(ze));
576     
577                         copyEntry(zipInputFile, ze, false, null, true, auxilaryOut);
578                     }
579     
580                 });
581             }
582     
583             if (auxilaryOut != null) {
584                 auxilaryOut.close();
585             }
586             for (ZipOutputStream out : analysisOutputFiles.values()) {
587                 out.close();
588             }
589     
590             System.out.println("All done");
591         }

599         private boolean embeddedNameMismatch(ZipFile zipInputFile, ZipEntry ze) throws IOException {
600             InputStream zipIn = zipInputFile.getInputStream(ze);
601             String name = ze.getName();
602             JavaClass j = new ClassParser(zipIn, name).parse();
603             zipIn.close();
604             String className = j.getClassName();
605             String computedFileName = ClassName.toSlashedClassName(className)+".class";
606             if (name.charAt(0) == '1') {
607                 System.out.println(name);
608                 System.out.println("  " + className);
609             }
610             if (computedFileName.equals(name)) {
611                 return false;
612             }
613             System.out.println("In " + name + " found " + className);
614             return true;
615         }

617         private void copyEntry(ZipFile zipInputFile, ZipEntry ze, boolean writeToAnalyzeOut, ZipOutputStream analyzeOut1,
618                 boolean writeToAuxilaryOut, ZipOutputStream auxilaryOut1) throws IOException {
619             InputStream zipIn = zipInputFile.getInputStream(ze);
620     
621             while (true) {
622                 int bytesRead = zipIn.read(buffer);
623                 if (bytesRead < 0) {
624                     break;
625                 }
626                 if (writeToAnalyzeOut) {
627                     analyzeOut1.write(buffer, 0, bytesRead);
628                 }
629                 if (writeToAuxilaryOut) {
630                     auxilaryOut1.write(buffer, 0, bytesRead);
631                 }
632             }
633             if (writeToAnalyzeOut) {
634                 analyzeOut1.closeEntry();
635             }
636             if (writeToAuxilaryOut) {
637                 auxilaryOut1.closeEntry();
638             }
639             zipIn.close();
640         }

647         boolean processZipEntries(File f, ZipElementHandler handler) {
648             if (!f.exists()) {
649                 System.out.println("file not found: '" + f + "'");
650                 return false;
651             }
652             if (!f.canRead() || f.isDirectory()) {
653                 System.out.println("not readable: '" + f + "'");
654                 return false;
655             }
656             if (!f.canRead() || f.isDirectory()) {
657                 System.out.println("not readable: '" + f + "'");
658                 return false;
659             }
660             if (f.length() == 0) {
661                 System.out.println("empty zip file: '" + f + "'");
662                 return false;
663             }
664             ZipFile zipInputFile;
665             try {
666                 zipInputFile = new ZipFile(f);
667                 for (Enumeration<? extends ZipEntry> e = zipInputFile.entries(); e.hasMoreElements();) {
668                     ZipEntry ze = e.nextElement();
669                     if (!ze.isDirectory() && ze.getName().endsWith(".class") && ze.getSize() != 0) {
670                         handler.handle(zipInputFile, ze);
671                     }
672     
673                 }
674                 zipInputFile.close();
675             } catch (ClassFileNameMismatch e) {
676                 return false;
677             } catch (IOException e) {
678                 System.out.println("Error processing '" + f + "'");
679                 return false;
680             }
681             return true;
682     
683         }
