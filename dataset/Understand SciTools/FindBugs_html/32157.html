20     package edu.umd.cs.findbugs.ba.bcp;
21     
22     import java.util.ArrayList;
23     import java.util.Iterator;
24     import java.util.LinkedList;
25     
26     import org.apache.bcel.generic.InstructionHandle;
27     
28     public class ByteCodePatternMatch {
29         private final BindingSet bindingSet;
30     
31         private final PatternElementMatch lastElementMatch;
32     
33         private final LinkedList<PatternElementMatch> patternElementMatchList;
34     
35         @Override
36         public String toString() {
37             ArrayList<Integer> lst = new ArrayList<Integer>();
38             for (PatternElementMatch m : patternElementMatchList) {
39                 lst.add(m.getMatchedInstructionInstructionHandle().getPosition());
40             }
41             return lst.toString();
42         }
43     
44         public ByteCodePatternMatch(BindingSet bindingSet, PatternElementMatch lastElementMatch) {
45             this.bindingSet = bindingSet;
46             this.lastElementMatch = lastElementMatch;
47             this.patternElementMatchList = new LinkedList<PatternElementMatch>();
48     
49             // The PatternElementMatch objects are stored in reverse order.
50             // So, put them in a LinkedList to get them in the right order.
51             while (lastElementMatch != null) {
52                 patternElementMatchList.addFirst(lastElementMatch);
53                 lastElementMatch = lastElementMatch.getPrev();
54             }
55         }
56     
57         public BindingSet getBindingSet() {
58             return bindingSet;
59         }
60     
61         public Iterator<PatternElementMatch> patternElementMatchIterator() {
62             return patternElementMatchList.iterator();
63         }
64     
65         public InstructionHandle getLabeledInstruction(String label) {
66             return lastElementMatch != null ? lastElementMatch.getLabeledInstruction(label) : null;
67         }
68     
69         public PatternElementMatch getFirstLabeledMatch(String label) {
70             return lastElementMatch != null ? lastElementMatch.getFirstLabeledMatch(label) : null;
71         }
72     
73         public PatternElementMatch getLastLabeledMatch(String label) {
74             return lastElementMatch != null ? lastElementMatch.getLastLabeledMatch(label) : null;
75         }
76     }
