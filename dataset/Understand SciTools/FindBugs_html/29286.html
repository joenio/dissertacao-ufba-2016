6     package edu.umd.cs.findbugs.annotations;
7     
8     import java.lang.annotation.Retention;
9     import java.lang.annotation.RetentionPolicy;
10     
11     import edu.umd.cs.findbugs.BugRanker;
12     
13     /**
14      * Annotation indicating that a FindBugs warning is expected.
15      *
16      * See http://code.google.com/p/findbugs/wiki/FindbugsTestCases
17      *
18      * @author David Hovemeyer
19      */
20     @Retention(RetentionPolicy.CLASS)
21     public @interface ExpectWarning {
22         /**
23          * The value indicates the bug code (e.g., NP) or bug pattern (e.g.,
24          * IL_INFINITE_LOOP) of the expected warning. Can be a comma-separated list.
25          */
26         public String value();
27     
28         /** Want a warning at this priority or higher */
29         public Confidence confidence() default Confidence.LOW;
30     
31         /** Expect a warning at least this scary */
32         public int rank() default BugRanker.VISIBLE_RANK_MAX;
33     
34         /** Expect at least this many warnings */
35         public int num() default 1;
36     }
