20     package edu.umd.cs.findbugs.graph;
21     
22     /**
23      * GraphEdge implementation for use with AbstractGraph.
24      *
25      * @see GraphEdge
26      * @see AbstractGraph
27      * @see AbstractVertex
28      * @author David Hovemeyer
29      */
30     public class AbstractEdge<ActualEdgeType extends AbstractEdge<ActualEdgeType, VertexType>, VertexType extends AbstractVertex<ActualEdgeType, VertexType>>
31     implements GraphEdge<ActualEdgeType, VertexType> {
32     
33         private final VertexType source;
34     
35         private final VertexType target;
36     
37         private int label;
38     
39         private ActualEdgeType nextOutgoingEdge;
40     
41         private ActualEdgeType nextIncomingEdge;
42     
43         /**
44          * Constructor.
45          *
46          * @param source
47          *            the source vertex of the edge
48          * @param target
49          *            the target vertex of the edge
50          */
51         public AbstractEdge(VertexType source, VertexType target) {
52             this.source = source;
53             this.target = target;
54         }
55     
56         @Override
57         public VertexType getSource() {
58             return source;
59         }
60     
61         @Override
62         public VertexType getTarget() {
63             return target;
64         }
65     
66         @Override
67         public int getLabel() {
68             return label;
69         }
70     
71         @Override
72         public void setLabel(int label) {
73             this.label = label;
74         }
75     
76         @Override
77         public int hashCode() {
78             return source.hashCode() + target.hashCode() * 3;
79         }
80     
81         @Override
82         public boolean equals(Object o) {
83             if (!(o instanceof AbstractEdge)) {
84                 return false;
85             }
86             AbstractEdge<?,?> other = (AbstractEdge<?,?>) o;
87             return source.equals(other.source) && target.equals(other.target);
88         }
89     
90         @Override
91         public int compareTo(ActualEdgeType other) {
92             int cmp = source.compareTo(other.getSource());
93             if (cmp != 0) {
94                 return cmp;
95             }
96             return target.compareTo(other.getTarget());
97         }
98     
99         void setNextOutgoingEdge(ActualEdgeType edge) {
100             nextOutgoingEdge = edge;
101         }
102     
103         ActualEdgeType getNextOutgoingEdge() {
104             return nextOutgoingEdge;
105         }
106     
107         void setNextIncomingEdge(ActualEdgeType edge) {
108             nextIncomingEdge = edge;
109         }
110     
111         ActualEdgeType getNextIncomingEdge() {
112             return nextIncomingEdge;
113         }
114     
115     }
