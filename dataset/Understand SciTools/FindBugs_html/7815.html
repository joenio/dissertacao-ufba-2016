30     import edu.umd.cs.findbugs.ExitCodes;

32     /**
33      * FindBugsViewerTask.java -- Ant Task to launch the FindBugsFrame
34      *
35      * To use, create a new task that refrences the ant task (such as
36      * "findbugs-viewer"). Then call this task while passing in parameters to modify
37      * it's behaviour. It supports several options that are the same as the findbugs
38      * task:
39      *
40      * -projectFile -debug -jvmargs -home -classpath -pluginList -timeout
41      *
42      * It also adds some new options:
43      *
44      * -look: string name representing look and feel. Can be "native", "plastic" or
45      * "gtk" -loadbugs: file name of bug report to load
46      *
47      * The below is an example of how this could be done in an ant script:
48      *
49      * <taskdef name="findbugs" classname="edu.umd.cs.findbugs.anttask.FindBugsTask"
50      * classpath="C:\dev\cvs.sourceforge.net\findbugs\lib\findbugs-ant.jar" />
51      * <taskdef name="findbugs-viewer"
52      * classname="edu.umd.cs.findbugs.anttask.FindBugsViewerTask"
53      * classpath="C:\dev\cvs.sourceforge.net\findbugs\lib\findbugs-ant.jar" />
54      *
55      * <property name="findbugs.home" location="C:\dev\cvs.sourceforge.net\findbugs"
56      * /> <property name="findbugs.bugReport" location="bcel-fb.xml" />
57      *
58      * <target name="findbugs-viewer" depends="jar"> <findbugs-viewer
59      * home="${findbugs.home}" look="native" loadbugs="${findbugs.bugReport}"/>
60      * </target>
61      *
62      * Created on March 21, 2006, 12:57 PM
63      *
64      * @author Mark McKay, mark@kitfox.com
65      */
66     public class FindBugsViewerTask extends Task {
67     
68         private static final long DEFAULT_TIMEOUT = -1; // ten minutes
69     
70         // location to load bug report from
71         private boolean debug = false;
72     
73         private File projectFile = null;
74     
75         private File loadbugs = null;
76     
77         private long timeout = DEFAULT_TIMEOUT;
78     
79         private String jvmargs = "";
80     
81         private String look = "native";
82     
83         private File homeDir = null;
84     
85         private Path classpath = null;
86     
87         private Path pluginList = null;
88     
89         private Java findbugsEngine = null;
90     
91         /** Creates a new instance of FindBugsViewerTask */
92         public FindBugsViewerTask() {
93         }
94     
95         /**
96          * Sets the file that contains the XML output of a findbugs report.
97          *
98          * @param loadbugs
99          *            XML output from a findbugs session
100          */
101         public void setLoadbugs(File loadbugs) {
102             this.loadbugs = loadbugs;
103         }
104     
105         /**
106          * Set the project file
107          */
108         public void setProjectFile(File projectFile) {
109             this.projectFile = projectFile;
110         }
111     
112         /**
113          * Set the debug flag
114          */
115         public void setDebug(boolean flag) {
116             this.debug = flag;
117         }
118     
119         /**
120          * Set any specific jvm args
121          */
122         public void setJvmargs(String args) {
123             this.jvmargs = args;
124         }
125     
126         /**
127          * Set look. One of "native", "gtk" or "plastic"
128          */
129         public void setLook(String look) {
130             this.look = look;
131         }
132     
133         /**
134          * Set the home directory into which findbugs was installed
135          */
136         public void setHome(File homeDir) {
137             this.homeDir = homeDir;
138         }
139     
140         /**
141          * Path to use for classpath.
142          */
143         public Path createClasspath() {
144             if (classpath == null) {
145                 classpath = new Path(getProject());
146             }
147             return classpath.createPath();
148         }
149     
150         /**
151          * Adds a reference to a classpath defined elsewhere.
152          */
153         public void setClasspathRef(Reference r) {
154             createClasspath().setRefid(r);
155         }
156     
157         /**
158          * the plugin list to use.
159          */
160         public void setPluginList(Path src) {
161             if (pluginList == null) {
162                 pluginList = src;
163             } else {
164                 pluginList.append(src);
165             }
166         }
167     
168         /**
169          * Path to use for plugin list.
170          */
171         public Path createPluginList() {
172             if (pluginList == null) {
173                 pluginList = new Path(getProject());
174             }
175             return pluginList.createPath();
176         }
177     
178         /**
179          * Adds a reference to a plugin list defined elsewhere.
180          */
181         public void setPluginListRef(Reference r) {
182             createPluginList().setRefid(r);
183         }
184     
185         /**
186          * Set timeout in milliseconds.
187          *
188          * @param timeout
189          *            the timeout
190          */
191         public void setTimeout(long timeout) {
192             this.timeout = timeout;
193         }
194     
195         /**
196          * Add an argument to the JVM used to execute FindBugs.
197          *
198          * @param arg
199          *            the argument
200          */
201         private void addArg(String arg) {
202             findbugsEngine.createArg().setValue(arg);
203         }
204     
205         @Override
206         public void execute() throws BuildException {
207             findbugsEngine = (Java) getProject().createTask("java");
208     
209             findbugsEngine.setTaskName(getTaskName());
210             findbugsEngine.setFork(true);
211     
212             if (timeout > 0) {
213                 findbugsEngine.setTimeout(timeout);
214             }
215     
216             if (debug) {
217                 jvmargs = jvmargs + " -Dfindbugs.debug=true";
218             }
219             findbugsEngine.createJvmarg().setLine(jvmargs);
220     
221             if (homeDir != null) {
222                 // Use findbugs.home to locate findbugs.jar and the standard
223                 // plugins. This is the usual means of initialization.
224                 File findbugsLib = new File(homeDir, "lib");
225     
226                 File findbugsLibFindBugs = new File(findbugsLib, "findbugs.jar");
227                 File findBugsFindBugs = new File(homeDir, "findbugs.jar");
228                 // log("executing using home dir [" + homeDir + "]");
229                 if (findbugsLibFindBugs.exists()) {
230                     findbugsEngine.setClasspath(new Path(getProject(), findbugsLibFindBugs.getPath()));
231                 } else if (findBugsFindBugs.exists()) {
232                     findbugsEngine.setClasspath(new Path(getProject(), findBugsFindBugs.getPath()));
233                 } else {
234                     throw new IllegalArgumentException("Can't find findbugs.jar in " + homeDir);
235                 }
236     
237                 findbugsEngine.setClassname("edu.umd.cs.findbugs.LaunchAppropriateUI");
238                 findbugsEngine.createJvmarg().setValue("-Dfindbugs.home=" + homeDir.getPath());
239             } else {
240                 // Use an explicitly specified classpath and list of plugin Jars
241                 // to initialize. This is useful for other tools which may have
242                 // FindBugs installed using a non-standard directory layout.
243     
244                 findbugsEngine.setClasspath(classpath);
245                 findbugsEngine.setClassname("edu.umd.cs.findbugs.LaunchAppropriateUI");
246     
247                 addArg("-pluginList");
248                 addArg(pluginList.toString());
249             }
250     
251             if (projectFile != null) {
252                 addArg("-project");
253                 addArg(projectFile.getPath());
254             }
255     
256             if (loadbugs != null) {
257                 addArg("-loadbugs");
258                 addArg(loadbugs.getPath());
259             }
260     
261             if (look != null) {
262                 addArg("-look:" + look);
263                 // addArg("-look");
264                 // addArg(look);
265             }
266     
267             // findbugsEngine.setClassname("edu.umd.cs.findbugs.gui.FindBugsFrame");
268     
269             log("Launching FindBugs Viewer...");
270     
271             int rc = findbugsEngine.executeJava();
272     
273             if ((rc & ExitCodes.ERROR_FLAG) != 0) {
274                 throw new BuildException("Execution of findbugs failed.");
275             }
276             if ((rc & ExitCodes.MISSING_CLASS_FLAG) != 0) {
277                 log("Classes needed for analysis were missing");
278             }
279         }
280     
281     }
