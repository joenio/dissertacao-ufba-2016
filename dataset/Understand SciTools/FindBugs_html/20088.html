45         public static enum TypeCategory {
46     
47             /** A simple (non-generic ObjectType) */
48             PLAIN_OBJECT_TYPE {
49                 @Override
50                 public ReferenceType produce(GenericObjectType obj) {
51                     return obj;
52                 }
53     
54                 @Override
55                 public String asString(GenericObjectType obj) {
56                     // obj.getTypeCategory() does not return PLAIN_OBJECT_TYPE
57                     return GenericUtilities.getString(obj);
58                 }
59             },
60     
61             /** A array */
62             ARRAY_TYPE {
63                 @Override
64                 public ReferenceType produce(GenericObjectType obj) {
65                     return obj;
66                 }
67     
68                 @Override
69                 public String asString(GenericObjectType obj) {
70                     // obj.getTypeCategory() does not return ARRAY_TYPE
71                     return GenericUtilities.getString(obj);
72                 }
73             },
74     
75             /** A parameterized class e.g. <code>List&lt;String&gt;</code> */
76             PARAMETERIZED {
77                 @Override
78                 public ReferenceType produce(GenericObjectType obj) {
79                     return obj;
80                 }
81     
82                 @Override
83                 public String asString(GenericObjectType obj) {
84                     StringBuilder b = new StringBuilder(obj.toPlainString());
85                     b.append("<");
86                     boolean first = true;
87                     for (Type t : obj.parameters) {
88                         if (!first) {
89                             b.append(",");
90                         }
91                         first = false;
92                         b.append(GenericUtilities.getString(t));
93                     }
94                     b.append(">");
95                     return b.toString();
96                 }
97             },
98     
99             /**
100              * A simple type variable e.g. <code>E</code>. Underlying ObjectType is
101              * <code>java.lang.Object</code>
102              */
103             TYPE_VARIABLE {
104                 @Override
105                 public ReferenceType produce(GenericObjectType obj) {
106                     return Type.OBJECT;
107                 }
108     
109                 @Override
110                 public String asString(GenericObjectType obj) {
111                     return obj.variable;
112                 }
113             },
114     
115             /**
116              * A simple wildcard i.e. <code>?</code>. Underlying ObjectType is
117              * <code>java.lang.Object</code>
118              */
119             WILDCARD {
120                 @Override
121                 public ReferenceType produce(GenericObjectType obj) {
122                     return Type.OBJECT;
123                 }
124     
125                 @Override
126                 public String asString(GenericObjectType obj) {
127                     return "?";
128                 }
129             },
130     
131             /**
132              * A wildcard that extends another ObjectType e.g.
133              * <code>? extends Comparable</code>. Underlying ObjectType is
134              * <code>java.lang.Object</code>. The extended type can be an ObjectType
135              * or a GenericObjectType
136              */
137             WILDCARD_EXTENDS {
138                 @Override
139                 public ReferenceType produce(GenericObjectType obj) {
140                     return obj.extension;
141                 }
142     
143                 @Override
144                 public String asString(GenericObjectType obj) {
145                     Type extension = obj.extension;
146                     assert extension != null;
147                     return "? extends " + GenericUtilities.getString(extension);
148                 }
149             },
150     
151             /**
152              * A wildcard that is extended by another ObjectType e.g.
153              * <code>? super Comparable</code>. Underlying ObjectType is
154              * <code>java.lang.Object</code>. The super type can be an ObjectType or
155              * a GenericObjectType
156              */
157             WILDCARD_SUPER {
158                 @Override
159                 public ReferenceType produce(GenericObjectType obj) {
160                     return Type.OBJECT;
161                 }
162     
163                 @Override
164                 public String asString(GenericObjectType obj) {
165                     Type extension = obj.extension;
166                     assert extension != null;
167                     return "? super " + GenericUtilities.getString(extension);
168                 }
169             };

176                 Type obj = atype.getBasicType();

182         /**
183          * Get the TypeCategory that represents this Object
184          *
185          * @see GenericUtilities.TypeCategory
186          */
187         public static final TypeCategory getTypeCategory(Type type) {
188             if (type instanceof GenericObjectType) {
189                 return ((GenericObjectType) type).getTypeCategory();
190             }
191     
192             if (type instanceof ObjectType || type instanceof NullType) {
193                 return TypeCategory.PLAIN_OBJECT_TYPE;
194             }
195     
196             if (type instanceof ArrayType) {
197                 return TypeCategory.ARRAY_TYPE;
198             }
199     
200             throw new IllegalArgumentException("Not a reference type: " + type);
201         }

203         public static final boolean isPlainObject(Type type) {
204             return getTypeCategory(type) == TypeCategory.PLAIN_OBJECT_TYPE;
205         }

207         /**
208          * Get String representation of a Type including Generic information
209          */
210         public static final String getString(Type type) {
211             if (type instanceof GenericObjectType) {
212                 return ((GenericObjectType) type).toString(true);
213             } else if (type instanceof ArrayType) {
214                 return TypeCategory.asString((ArrayType) type);
215             } else {
216                 return type.toString();
217             }
218         }

220         static String stripAngleBrackets(String s) {
221             if (s.indexOf('<') == -1) {
222                 return s;
223             }
224             StringBuilder result = new StringBuilder(s.length());
225             int nesting = 0;
226             boolean seenLeftBracket = false;
227             for (int i = 0; i < s.length(); i++) {
228                 char c = s.charAt(i);
229                 if (c == '<') {
230                     nesting++;
231                     seenLeftBracket = true;
232                 } else if (c == '>') {
233                     nesting--;
234                 } else if (nesting == 0) {
235                     if (seenLeftBracket && c == '.') {
236                         result.append('$');
237                     } else {
238                         result.append(c);
239                     }
240                 }
241             }
242             return result.toString();
243         }

249         /**
250          * This method is analogous to <code>Type.getType(String)</code>, except
251          * that it also accepts signatures with generic information. e.g.
252          * <code>Ljava/util/ArrayList&lt;TT;&gt;;</code>
253          * <p>
254          *
255          * The signature should only contain one type. Use GenericSignatureParser to
256          * break up a signature with many types or call createTypes(String) to
257          * return a list of types
258          */
259         public static @CheckForNull
260         Type getType(String signature) {
261             try {
262                 // ensure signature only has one type
263                 if (new GenericSignatureParser("(" + signature + ")V").getNumParameters() != 1) {
264                     throw new IllegalArgumentException("the following signature does not " + "contain exactly one type: " + signature);
265                 }
266                 int index = 0;
267     
268                 if (signature.startsWith("L")) {
269                     index = lastMatchedLeftAngleBracket(signature);
270                     if (index < 0) {
271                         return Type.getType(stripAngleBrackets(signature));
272                     }
273     
274                     String typeParameters = signature.substring(index + 1, nextUnmatchedRightAngleBracket(signature, index + 1));
275                     List<ReferenceType> parameters = GenericUtilities.getTypeParameters(typeParameters);
276                     if (parameters == null) {
277                         return null;
278                     }
279                     String baseType = removeMatchedAngleBrackets(signature.substring(1, index)).replace('.', '$');
280                     return new GenericObjectType(baseType, parameters);
281     
282                 } else if (signature.startsWith("T")) {
283                     int i = signature.indexOf(';');
284                     if (i > 0) {
285                         String var = signature.substring(1, i);
286                         if (var.indexOf('<') == -1) {
287                             return new GenericObjectType(var);
288                         }
289                     }
290                     // can't handle type variables
291                     return null;
292     
293                 } else if (signature.startsWith("[")) {
294                     index++;
295                     while (signature.charAt(index) == '[') {
296                         index++;
297                     }
298                     Type componentType = getType(signature.substring(index));
299                     if (componentType == null) {
300                         return null;
301                     }
302                     return new ArrayType(componentType, index);
303     
304                 } else if (signature.startsWith("*")) {
305                     return new GenericObjectType("*");
306     
307                 } else if (signature.startsWith("+") || signature.startsWith("-")) {
308                     Type baseType = getType(signature.substring(1));
309                     if (baseType == null) {
310                         return null;
311                     }
312                     return new GenericObjectType(signature.substring(0, 1), (ReferenceType) baseType);
313     
314                 } else {
315                     // assert signature contains no generic information
316                     return Type.getType(signature);
317                 }
318             } catch (IllegalStateException e) {
319                 AnalysisContext.logError("Error parsing signature " + signature, e);
320                 return null;
321             }
322         }

324         public static ObjectType merge(@CheckForNull Type t1, ObjectType t2) {
325             if (t1 instanceof GenericObjectType) {
326                 return merge((GenericObjectType) t1, t2);
327             }
328             return t2;

331         public static Type merge(@CheckForNull GenericObjectType t1, Type t2) {
332             if (t1 == null) {
333                 return t2;
334             }
335             if (t2 instanceof ObjectType) {
336                 return merge(t1,(ObjectType) t2);
337             }
338             if (t2 instanceof NullType) {
339                 return t1;
340             }
341             return t2;
342         }

344         public static ObjectType merge(@CheckForNull GenericObjectType t1, ObjectType t2) {
345             if (t1 == null || t2 instanceof GenericObjectType) {
346                 return t2;
347             }
348             List<? extends ReferenceType> parameters = t1.getParameters();
349             if (parameters == null) {
350                 return t2;
351             }
352             return new GenericObjectType(t2.getClassName(), parameters);
353         }

355         public static String removeMatchedAngleBrackets(String s) {
356             int first = s.indexOf('<');
357             if (first < 0) {
358                 return s;
359             }
360             StringBuilder result = new StringBuilder(s.substring(0, first));
361             int pos = first;
362             int nesting = 0;
363             while (pos < s.length()) {
364                 char c = s.charAt(pos++);
365                 if (c == '<') {
366                     nesting++;
367                 } else if (c == '>') {
368                     nesting--;
369                 } else if (nesting == 0) {
370                     result.append(c);
371                 }
372             }
373             return result.toString();
374     
375         }
376     
377         public static int nextUnmatchedRightAngleBracket(String s, int startingAt) {
378             int nesting = 0;
379             int pos = startingAt;
380     
381             while (true) {
382                 if (pos < 0) {
383                     return -1;
384                 }
385                 char c = s.charAt(pos);
386                 if (c == '>') {
387                     if (nesting == 0) {
388                         return pos;
389                     }
390                     nesting--;
391                 } else if (c == '<') {
392                     nesting++;
393                 }
394                 pos++;
395             }
396         }
397     
398         public static int lastMatchedLeftAngleBracket(String s) {
399             int nesting = 0;
400             int pos = s.length() - 2;
401     
402             while (true) {
403                 if (pos < 0) {
404                     return -1;
405                 }
406                 char c = s.charAt(pos);
407                 if (c == '<') {
408                     nesting--;
409                     if (nesting == 0) {
410                         return pos;
411                     }
412                 } else if (c == '>') {
413                     nesting++;
414                 } else if (nesting == 0) {
415                     return -1;
416                 }
417                 pos--;
418             }
419         }

421         /**
422          * Parse a bytecode signature that has 1 or more (possibly generic) types
423          * and return a list of the Types.
424          *
425          * @param signature
426          *            bytecode signature e.g. e.g.
427          *            <code>Ljava/util/ArrayList&lt;Ljava/lang/String;&gt;;Ljava/util/ArrayList&lt;TT;&gt;;Ljava/util/ArrayList&lt;*&gt;;</code>
428          */
429         public static final @CheckForNull
430         List<ReferenceType> getTypeParameters(String signature) {
431             GenericSignatureParser parser = new GenericSignatureParser("(" + signature + ")V");
432             List<ReferenceType> types = new ArrayList<ReferenceType>();
433     
434             Iterator<String> iter = parser.parameterSignatureIterator();
435             while (iter.hasNext()) {
436                 String parameterString = iter.next();
437                 ReferenceType t = (ReferenceType) getType(parameterString);
438                 if (t == null) {
439                     return null;
440                 }
441                 types.add(t);
442             }
443             return types;
444         }

446         public static final List<String> split(String signature, boolean skipInitialAngleBracket) {
447             List<String> result = new ArrayList<String>();
448             if (signature.charAt(0) != '<') {
449                 skipInitialAngleBracket = false;
450             }
451             int depth = 0;
452             int start = 0;
453             for (int pos = start; pos < signature.length(); pos++) {
454                 switch (signature.charAt(pos)) {
455                 case '<':
456                     depth++;
457                     break;
458                 case '>':
459                     depth--;
460                     if (depth == 0 && skipInitialAngleBracket) {
461                         skipInitialAngleBracket = false;
462                         start = pos + 1;
463                     }
464                     break;
465                 case ';':
466                     if (depth > 0) {
467                         break;
468                     }
469                     String substring = signature.substring(start, pos + 1);
470                     result.add(substring);
471                     start = pos + 1;
472                     break;
473                 default:
474                     break;
475                 }
476             }
477             if (depth != 0) {
478                 throw new IllegalArgumentException("Unbalanced signature: " + signature);
479             }
480             return result;
481         }
