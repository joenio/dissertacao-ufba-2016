37     /**
38      * Generic class for representing a Java stack frame as a dataflow value. A
39      * frame consists of "slots", which represent the local variables and values on
40      * the Java operand stack. Slots 0 .. <code>getNumLocals() - 1</code> represent
41      * the local variables. Slots <code>getNumLocals()</code> ..
42      * <code>getNumSlots() - 1</code> represent the Java operand stack.
43      * <p/>
44      * <p>
45      * Frame is parametized by "ValueType", which is the type of value to be stored
46      * in the Frame's slots. This type must form a lattice, according to the
47      * abstract mergeValues() operation in the corresponding analysis class (which
48      * should be derived from FrameDataflowAnalysis). When a Frame is constructed,
49      * all of its slots will contain null. The analysis is responsible for
50      * initializing created Frames with default values at the appropriate time.
51      * Typically, only initEntryFact() will need to do this.
52      * <p/>
53      * <p>
54      * A Frame may have the special "TOP" value. Such frames serve as the identity
55      * element for the meet operation operation.
56      * <p/>
57      * <p>
58      * A Frame may have the special "BOTTOM" value. The result of merging any frame
59      * with BOTTOM is BOTTOM.
60      *
61      * @author David Hovemeyer
62      * @see FrameDataflowAnalysis
63      */
64     public abstract class Frame<ValueType> {

103         /**
104          * Constructor. This version of the constructor is for subclasses for which
105          * it is always safe to call getDefaultValue(), even when the object is not
106          * fully initialized.
107          *
108          * @param numLocals
109          *            number of local variable slots in the method
110          */
111         public Frame(int numLocals) {
112             this.numLocals = numLocals;
113             this.slotList = new ArrayList<ValueType>(numLocals + DEFAULT_STACK_CAPACITY);
114             for (int i = 0; i < numLocals; ++i) {
115                 slotList.add(null);
116             }
117         }

165             return !isTop() && !isBottom();

168         /**
169          * Push a value onto the Java operand stack.
170          *
171          * @param value
172          *            the ValueType to push
173          */
174         public void pushValue(ValueType value) {
175             if (VERIFY_INTEGRITY && value == null) {
176                 throw new IllegalArgumentException();
177             }
178             if (!isValid()) {
179                 throw new IllegalStateException("accessing top or bottom frame");
180             }
181             slotList.add(value);
182         }

184         /**
185          * Pop a value off of the Java operand stack.
186          *
187          * @return the value that was popped
188          * @throws DataflowAnalysisException
189          *             if the Java operand stack is empty
190          */
191         public ValueType popValue() throws DataflowAnalysisException {
192             if (!isValid()) {
193                 throw new DataflowAnalysisException("accessing top or bottom frame");
194             }
195             if (slotList.size() == numLocals) {
196                 throw new DataflowAnalysisException("operand stack empty");
197             }
198             return slotList.remove(slotList.size() - 1);
199         }
200     
201         /**
202          * Get the value on the top of the Java operand stack.
203          *
204          * @throws DataflowAnalysisException
205          *             if the Java operand stack is empty
206          */
207         public ValueType getTopValue() throws DataflowAnalysisException {
208             if (!isValid()) {
209                 throw new DataflowAnalysisException("accessing top or bottom frame");
210             }
211             assert slotList.size() >= numLocals;
212             if (slotList.size() == numLocals) {
213                 throw new DataflowAnalysisException("operand stack is empty");
214             }
215             return slotList.get(slotList.size() - 1);
216         }
217     
218         /**
219          * Get the values on the top of the Java operand stack. The top stack item
220          * is placed at the end of the array, so that to restore the values to the
221          * stack, you would push them in the order they appear in the array.
222          */
223         public void getTopStackWords(ValueType[] valueList) throws DataflowAnalysisException {
224             int stackDepth = getStackDepth();
225             if (valueList.length > stackDepth) {
226                 throw new DataflowAnalysisException("not enough values on stack");
227             }
228             int numSlots = slotList.size();
229             for (int i = numSlots - valueList.length, j = 0; i < numSlots; ++i, ++j) {
230                 valueList[j] = slotList.get(i);
231             }
232         }

234         /**
235          * Get a value on the operand stack.
236          *
237          * @param loc
238          *            the stack location, counting downwards from the top (location
239          *            0)
240          */
241         public ValueType getStackValue(int loc) throws DataflowAnalysisException {
242             if (!isValid()) {
243                 throw new DataflowAnalysisException("Accessing TOP or BOTTOM frame!");
244             }
245             int stackDepth = getStackDepth();
246             if (loc >= stackDepth) {
247                 throw new DataflowAnalysisException("not enough values on stack: access=" + loc + ", avail=" + stackDepth);
248             }
249             if (loc < 0) {
250                 throw new DataflowAnalysisException("can't get position " + loc + " of stack");
251             }
252             int pos = slotList.size() - (loc + 1);
253             return slotList.get(pos);
254         }

256         /**
257          * Get a the location in the frame of a value on the operand stack.
258          *
259          * @param loc
260          *            the stack location, counting downwards from the top (location
261          *            0)
262          */
263         public int getStackLocation(int loc) throws DataflowAnalysisException {
264             int stackDepth = getStackDepth();
265             if (loc >= stackDepth) {
266                 throw new DataflowAnalysisException("not enough values on stack: access=" + loc + ", avail=" + stackDepth);
267             }
268             return slotList.size() - (loc + 1);
269         }

286         /**
287          * Get the stack location (counting down from top of stack, starting at 0)
288          * containing the object instance referred to by given instruction. This
289          * relies on the observation that in instructions which use an object
290          * instance (such as getfield, invokevirtual, etc.), the object instance is
291          * the first operand used by the instruction.
292          *
293          * <p>
294          * The value returned may be passed to getStackValue(int).
295          * </p>
296          *
297          * @param ins
298          *            the Instruction
299          * @param cpg
300          *            the ConstantPoolGen for the method
301          * @return stack location (counting down from top of stack, starting at 0)
302          *         containing the object instance
303          * @throws DataflowAnalysisException
304          */
305         public int getInstanceStackLocation(Instruction ins, ConstantPoolGen cpg) throws DataflowAnalysisException {
306             int numConsumed = ins.consumeStack(cpg);
307             if (numConsumed == Constants.UNPREDICTABLE) {
308                 throw new DataflowAnalysisException("Unpredictable stack consumption in " + ins);
309             }
310             return numConsumed - 1;
311         }

313         /**
314          * Get the slot the object instance referred to by given instruction is
315          * located in.
316          *
317          * @param ins
318          *            the Instruction
319          * @param cpg
320          *            the ConstantPoolGen for the method
321          * @return stack slot the object instance is in
322          * @throws DataflowAnalysisException
323          */
324         public int getInstanceSlot(Instruction ins, ConstantPoolGen cpg) throws DataflowAnalysisException {
325             if (!isValid()) {
326                 throw new DataflowAnalysisException("Accessing invalid frame at " + ins);
327             }
328             int numConsumed = ins.consumeStack(cpg);
329             if (numConsumed == Constants.UNPREDICTABLE) {
330                 throw new DataflowAnalysisException("Unpredictable stack consumption in " + ins);
331             }
332             if (numConsumed > getStackDepth()) {
333                 throw new DataflowAnalysisException("Stack underflow " + ins);
334             }
335             return getNumSlots() - numConsumed;
336         }

434          * Get the <i>i</i>th operand used by given instruction.

436          * @param ins
437          *            the instruction, which must be a StackConsumer
438          * @param cpg
439          *            the ConstantPoolGen

445         public ValueType getOperand(StackConsumer ins, ConstantPoolGen cpg, int i) throws DataflowAnalysisException {
446             int numOperands = ins.consumeStack(cpg);
447             if (numOperands == Constants.UNPREDICTABLE) {

449             }

453         /**
454          * Get set of arguments passed to a method invocation which match given
455          * predicate.
456          *
457          * @param invokeInstruction
458          *            the InvokeInstruction
459          * @param cpg
460          *            the ConstantPoolGen
461          * @param chooser
462          *            predicate to choose which argument values should be in the
463          *            returned set
464          * @return BitSet specifying which arguments match the predicate, indexed by
465          *         argument number (starting from 0)
466          * @throws DataflowAnalysisException
467          */
468         public BitSet getArgumentSet(InvokeInstruction invokeInstruction, ConstantPoolGen cpg, DataflowValueChooser<ValueType> chooser)
469                 throws DataflowAnalysisException {
470             BitSet chosenArgSet = new BitSet();
471             SignatureParser sigParser = new SignatureParser(invokeInstruction.getSignature(cpg));
472     
473             for (int i = 0; i < sigParser.getNumParameters(); ++i) {
474                 ValueType value = getArgument(invokeInstruction, cpg, i, sigParser);
475                 if (chooser.choose(value)) {
476                     chosenArgSet.set(i);
477                 }
478             }
479     
480             return chosenArgSet;
481         }

483         /**
484          * Clear the Java operand stack. Only local variable slots will remain in
485          * the frame.
486          */
487         public void clearStack() {
488             if (!isValid()) {
489                 throw new IllegalStateException("accessing top or bottom frame");
490             }
491             assert slotList.size() >= numLocals;
492             if (slotList.size() > numLocals) {
493                 slotList.subList(numLocals, slotList.size()).clear();
494             }
495         }

518         public boolean contains(ValueType value) {
519             if (!isValid()) {
520                 throw new IllegalStateException("accessing top or bottom frame");
521             }
522             for (ValueType v : slotList) {
523                 if (v.equals(value)) {
524                     return true;
525                 }
526             }
527             return false;
528         }

562         /**
563          * Return true if this stack frame is the same as the one given as a
564          * parameter.
565          *
566          * @param other
567          *            the other Frame
568          * @return true if the frames are the same, false otherwise
569          */
570         public boolean sameAs(Frame<ValueType> other) {
571             if (isTop != other.isTop) {
572                 return false;
573             }
574     
575             if (isTop && other.isTop) {
576                 return true;
577             }
578     
579             if (isBottom != other.isBottom) {
580                 return false;
581             }
582     
583             if (isBottom && other.isBottom) {
584                 return true;
585             }
586     
587             if (getNumSlots() != other.getNumSlots()) {
588                 return false;
589             }
590     
591             for (int i = 0; i < getNumSlots(); ++i) {
592                 if (!getValue(i).equals(other.getValue(i))) {
593                     return false;
594                 }
595             }
596     
597             return true;
598         }

600         /**
601          * Make this Frame exactly the same as the one given as a parameter.
602          *
603          * @param other
604          *            the Frame to make this object the same as
605          */
606         public void copyFrom(Frame<ValueType> other) {
607             lastUpdateTimestamp = other.lastUpdateTimestamp;
608             if (true) {
609                 int size = slotList.size();
610                 if (size == other.slotList.size()) {
611                     for (int i = 0; i < size; i++) {
612                         slotList.set(i, other.slotList.get(i));
613                     }
614                 } else {
615                     slotList.clear();
616                     for (ValueType v : other.slotList) {
617                         slotList.add(v);
618                     }
619                 }
620             } else {
621                 slotList.clear();
622                 slotList.addAll(other.slotList);
623     
624             }
625             /*
626              * Andrei, 27.02.2008: "optimized" code below takes ~18% overall FB
627              * execution time, code above only 5% int size = slotList.size(); if
628              * (size == other.slotList.size()) { for (int i = 0; i < size; i++)
629              * slotList.set(i, other.slotList.get(i)); } else { slotList.clear();
630              * for (ValueType v : other.slotList) slotList.add(v); }
631              */
632             isTop = other.isTop;
633             isBottom = other.isBottom;
634         }

636         private static final boolean STACK_ONLY = SystemProperties.getBoolean("dataflow.stackonly");
637     
638         /**
639          * Convert to string.
640          */
641         @Override
642         public String toString() {
643             if (isTop()) {
644                 return "[TOP]";
645             }
646             if (isBottom()) {
647                 return "[BOTTOM]";
648             }
649             StringBuilder buf = new StringBuilder();
650             buf.append('[');
651             int numSlots = getNumSlots();
652             int start = STACK_ONLY ? getNumLocals() : 0;
653             for (int i = start; i < numSlots; ++i) {
654                 if (!STACK_ONLY && i == getNumLocals()) {
655                     // Use a "|" character to visually separate locals from
656                     // the operand stack.
657                     int last = buf.length() - 1;
658                     if (last >= 0) {
659                         if (buf.charAt(last) == ',') {
660                             buf.deleteCharAt(last);
661                         }
662                     }
663                     buf.append('|');
664                 }
665                 String value = valueToString(getValue(i));
666                 if (i == numSlots - 1 && value.endsWith(",")) {
667                     value = value.substring(0, value.length() - 1);
668                 }
669                 buf.append(value);
670                 // buf.append(' ');
671             }
672             buf.append(']');
673             return buf.toString();
674         }

676         /**
677          * Subclasses may override this if they want to do something special to
678          * convert Value objects to Strings. By default, we just call toString() on
679          * the values.
680          */
681         protected String valueToString(ValueType value) {
682             if (value == null) {
683                 return "null";
684             }
685             return value.toString();
686         }
