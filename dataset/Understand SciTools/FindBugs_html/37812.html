38     /**
39      * A TypeMerger which applies standard Java semantics when merging Types.
40      * Subclasses may override mergeReferenceTypes() in order to implement special
41      * typing rules for reference types.
42      *
43      * @author David Hovemeyer
44      * @see TypeMerger
45      */
46     public class StandardTypeMerger implements TypeMerger, Constants, ExtendedTypes {
47         private final RepositoryLookupFailureCallback lookupFailureCallback;
48     
49         private final ExceptionSetFactory exceptionSetFactory;
50     
51         private static final ObjectType OBJECT_TYPE = ObjectTypeFactory.getInstance("java.lang.Object");
52     
53         /**
54          * Constructor.
55          *
56          * @param lookupFailureCallback
57          *            object used to report Repository lookup failures
58          * @param exceptionSetFactory
59          *            factory for creating ExceptionSet objects
60          */
61         public StandardTypeMerger(RepositoryLookupFailureCallback lookupFailureCallback, ExceptionSetFactory exceptionSetFactory) {
62             this.lookupFailureCallback = lookupFailureCallback;
63             this.exceptionSetFactory = exceptionSetFactory;
64         }
65     
66         @Override
67         public Type mergeTypes(Type a, Type b) throws DataflowAnalysisException {
68             if (a == null) {
69                 return b;
70             }
71             if (b == null) {
72                 return a;
73             }
74             byte aType = a.getType(), bType = b.getType();
75     
76             if (aType == T_TOP) {
77                 return b;
78             } else if (bType == T_TOP) {
79                 return a;
80             } else if (aType == T_BOTTOM || bType == T_BOTTOM) {
81                 // is bottom
82                 return BottomType.instance();
83             } else if (isReferenceType(aType) && isReferenceType(bType)) { // Two
84                 // object
85                 // types!
86                 // Handle the Null type, which serves as a special "top"
87                 // value for reference types.
88                 if (aType == T_NULL) {
89                     return b;
90                 } else if (bType == T_NULL) {
91                     return a;
92                 }
93     
94                 ReferenceType aRef = (ReferenceType) a;
95                 ReferenceType bRef = (ReferenceType) b;
96                 return mergeReferenceTypes(aRef, bRef);
97             } else if (isReferenceType(aType) || isReferenceType(bType)) {
98                 // meet
99                 // non-object
100                 // is
101                 // bottom
102                 return BottomType.instance();
103             } else if (aType == bType) {
104                 return a;
105             } else if (isIntegerType(aType) && isIntegerType(bType)) {
106                 // integer types
107                 // - use T_INT
108                 return Type.INT;
109             } else {
110                 // Default - types are incompatible
111                 return BottomType.instance();
112             }
113         }

115         /**
116          * Determine if the given typecode refers to a reference type. This
117          * implementation just checks that the type code is T_OBJECT, T_ARRAY,
118          * T_NULL, or T_EXCEPTION. Subclasses should override this if they have
119          * defined new object types with different type codes.
120          */
121         protected boolean isReferenceType(byte type) {
122             return type == T_OBJECT || type == T_ARRAY || type == T_NULL || type == T_EXCEPTION;
123         }
124     
125         /**
126          * Determine if the given typecode refers to an Object type. Subclasses
127          * should override with any new object types.
128          */
129         protected boolean isObjectType(byte type) {
130             return type == T_OBJECT || type == T_EXCEPTION;
131         }
132     
133         /**
134          * Determine if the given typecode refers to an Integer type (other than
135          * long). This implementation checks that the type code is T_INT, T_BYTE,
136          * T_BOOLEAN, T_CHAR, or T_SHORT. Subclasses should override this if they
137          * have defined new integer types with different type codes.
138          */
139         protected boolean isIntegerType(byte type) {
140             return type == T_INT || type == T_BYTE || type == T_BOOLEAN || type == T_CHAR || type == T_SHORT;
141         }
142     
143         private static void updateExceptionSet(ExceptionSet exceptionSet, ObjectType type) {
144             if (type instanceof ExceptionObjectType) {
145                 exceptionSet.addAll(((ExceptionObjectType) type).getExceptionSet());
146             } else {
147                 exceptionSet.addExplicit(type);
148             }
149         }
150     
151         /**
152          * Default implementation of merging reference types. This just returns the
153          * first common superclass, which is compliant with the JVM Spec. Subclasses
154          * may override this method in order to implement extended type rules.
155          *
156          * @param aRef
157          *            a ReferenceType
158          * @param bRef
159          *            a ReferenceType
160          * @return the merged Type
161          */
162         protected ReferenceType mergeReferenceTypes(ReferenceType aRef, ReferenceType bRef) throws DataflowAnalysisException {
163             if (aRef.equals(bRef)) {
164                 return aRef;
165             }
166             byte aType = aRef.getType();
167             byte bType = bRef.getType();
168             try {
169                 // Special case: ExceptionObjectTypes.
170                 // We want to preserve the ExceptionSets associated,
171                 // in order to track the exact set of exceptions
172                 if (isObjectType(aType) && isObjectType(bType)
173                         && ((aType == T_EXCEPTION || isThrowable(aRef))  && (bType == T_EXCEPTION ||   isThrowable(bRef)))) {
174                     ExceptionSet union = exceptionSetFactory.createExceptionSet();
175                     if (aType == T_OBJECT && "Ljava/lang/Throwable;".equals(aRef.getSignature())) {
176                         return aRef;
177                     }
178                     if (bType == T_OBJECT && "Ljava/lang/Throwable;".equals(bRef.getSignature())) {
179                         return bRef;
180                     }
181     
182                     updateExceptionSet(union, (ObjectType) aRef);
183                     updateExceptionSet(union, (ObjectType) bRef);
184     
185                     Type t = ExceptionObjectType.fromExceptionSet(union);
186                     if (t instanceof ReferenceType) {
187                         return (ReferenceType) t;
188                     }
189                 }
190     
191                 if (aRef instanceof GenericObjectType && bRef instanceof GenericObjectType
192                         && aRef.getSignature().equals(bRef.getSignature())) {
193                     GenericObjectType aG = (GenericObjectType) aRef;
194                     GenericObjectType bG = (GenericObjectType) bRef;
195                     if (aG.getTypeCategory() == bG.getTypeCategory()) {
196                         switch (aG.getTypeCategory()) {
197                         case PARAMETERIZED:
198                             List<? extends ReferenceType> aP = aG.getParameters();
199                             List<? extends ReferenceType> bP = bG.getParameters();
200                             assert aP != null;
201                             assert bP != null;
202                             if (aP.size() != bP.size()) {
203                                 break;
204                             }
205                             ArrayList<ReferenceType> result = new ArrayList<ReferenceType>(aP.size());
206                             for (int i = 0; i < aP.size(); i++) {
207                                 result.add(mergeReferenceTypes(aP.get(i), bP.get(i)));
208                             }
209     
210                             GenericObjectType rOT = GenericUtilities.getType(aG.getClassName(), result);
211                             return rOT;
212     
213                         }
214     
215                     }
216     
217                 }
218                 if (aRef instanceof GenericObjectType) {
219                     aRef = ((GenericObjectType) aRef).getObjectType();
220                 }
221                 if (bRef instanceof GenericObjectType) {
222                     bRef = ((GenericObjectType) bRef).getObjectType();
223                 }
224     
225                 if (Subtypes2.ENABLE_SUBTYPES2_FOR_COMMON_SUPERCLASS_QUERIES) {
226                     return AnalysisContext.currentAnalysisContext().getSubtypes2().getFirstCommonSuperclass(aRef, bRef);
227                 } else {
228                     return aRef.getFirstCommonSuperclass(bRef);
229                 }
230             } catch (ClassNotFoundException e) {
231                 lookupFailureCallback.reportMissingClass(e);
232                 return Type.OBJECT;
233             }
234         }
235     
236         private boolean isThrowable(ReferenceType ref) /*
237          * throws
238          * ClassNotFoundException
239          */{
240             try {
241     
242                 Subtypes2 subtypes2 = AnalysisContext.currentAnalysisContext().getSubtypes2();
243                 return subtypes2.isSubtype(ref, Type.THROWABLE);
244     
245             } catch (ClassNotFoundException e) {
246                 // We'll just assume that it's not an exception type.
247                 lookupFailureCallback.reportMissingClass(e);
248                 return false;
249             }
250         }
