19     package edu.umd.cs.findbugs;
20     
21     import java.io.IOException;
22     
23     import org.apache.bcel.generic.Type;
24     
25     import edu.umd.cs.findbugs.ba.AnalysisContext;
26     import edu.umd.cs.findbugs.ba.SignatureConverter;
27     import edu.umd.cs.findbugs.ba.generic.GenericObjectType;
28     import edu.umd.cs.findbugs.ba.generic.GenericUtilities;
29     import edu.umd.cs.findbugs.xml.XMLAttributeList;
30     import edu.umd.cs.findbugs.xml.XMLOutput;
31     
32     /**
33      * Bug annotation class for java types. This is of lighter weight than
34      * ClassAnnotation, and can be used for things like array types.
35      *
36      * @see ClassAnnotation
37      */
38     public class TypeAnnotation extends BugAnnotationWithSourceLines {
39         private static final long serialVersionUID = 1L;
40     
41         public static final String DEFAULT_ROLE = "TYPE_DEFAULT";
42     
43         public static final String EXPECTED_ROLE = "TYPE_EXPECTED";
44     
45         public static final String FOUND_ROLE = "TYPE_FOUND";
46     
47         public static final String CLOSEIT_ROLE = "TYPE_CLOSEIT";
48     
49         public static final String UNHASHABLE_ROLE = "TYPE_UNHASHABLE";
50     
51         final private String descriptor; // jvm type descriptor, such as "[I"
52     
53         private String roleDescription;
54     
55         private String typeParameters;
56     
57         /**
58          * constructor.
59          *
60          * <p>
61          * For information on type descriptors, <br>
62          * see http://java.sun.com/docs/books/vmspec/2nd-edition/html/ClassFile.doc.
63          * html#14152 <br>
64          * or http://www.murrayc.com/learning/java/java_classfileformat.shtml#
65          * TypeDescriptors
66          *
67          * @param typeDescriptor
68          *            a jvm type descriptor, such as "[I"
69          */
70         public TypeAnnotation(String typeDescriptor) {
71             this(typeDescriptor, DEFAULT_ROLE);
72         }
73     
74         public TypeAnnotation(Type objectType) {
75             this(objectType, DEFAULT_ROLE);
76         }
77     
78         public TypeAnnotation(Type objectType, String roleDescription) {
79             this(objectType.getSignature(), roleDescription);
80             if (objectType instanceof GenericObjectType) {
81                 GenericObjectType genericObjectType = (GenericObjectType) objectType;
82                 if (genericObjectType.getTypeCategory() == GenericUtilities.TypeCategory.PARAMETERIZED) {
83                     typeParameters = genericObjectType.getGenericParametersAsString();
84                 }
85             }
86         }
87     
88         public TypeAnnotation(String typeDescriptor, String roleDescription) {
89             descriptor = typeDescriptor;
90             this.roleDescription = roleDescription;
91             if (descriptor.startsWith("L")) {
92                 String className = typeDescriptor.substring(1, typeDescriptor.length() - 1).replace('/', '.');
93                 AnalysisContext context = AnalysisContext.currentAnalysisContext();
94                 if (context != null) {
95                     this.sourceFileName = context.lookupSourceFile(className);
96                     this.sourceLines = ClassAnnotation.getSourceLinesForClass(className, sourceFileName);
97                 } else {
98                     this.sourceFileName = SourceLineAnnotation.UNKNOWN_SOURCE_FILE;
99                 }
100             }
101         }
102     
103         /**
104          * Get the type descriptor.
105          *
106          * @return the jvm type descriptor, such as "[I"
107          */
108         public String getTypeDescriptor() {
109             return descriptor;
110         }
111     
112         @Override
113         public void accept(BugAnnotationVisitor visitor) {
114             visitor.visitTypeAnnotation(this);
115         }
116     
117         @Override
118         public String format(String key, ClassAnnotation primaryClass) {
119             String name = new SignatureConverter(descriptor).parseNext().replace("java.lang.", "");
120             if ("givenClass".equals(key)) {
121                 name = PackageMemberAnnotation.shorten(primaryClass.getPackageName(), name);
122             } else if ("excludingPackage".equals(key)) {
123                 name = PackageMemberAnnotation.removePackage(name);
124             }
125     
126             if (typeParameters != null && !"hash".equals(key)) {
127                 name = name + typeParameters;
128             }
129             return name;
130         }
131     
132         @Override
133         public void setDescription(String roleDescription) {
134             this.roleDescription = roleDescription.intern();
135         }
136     
137         @Override
138         public String getDescription() {
139             return roleDescription;
140         }
141     
142         public void setTypeParameters(String typeParameters) {
143             this.typeParameters = typeParameters;
144         }
145     
146         public String getTypeParameters() {
147             return typeParameters;
148         }
149     
150         @Override
151         public int hashCode() {
152             return descriptor.hashCode();
153         }
154     
155         @Override
156         public boolean equals(Object o) {
157             if (!(o instanceof TypeAnnotation)) {
158                 return false;
159             }
160             return descriptor.equals(((TypeAnnotation) o).descriptor);
161         }
162     
163         @Override
164         public int compareTo(BugAnnotation o) {
165             if (!(o instanceof TypeAnnotation)) {
166                 // with any type of BugAnnotation
167                 return this.getClass().getName().compareTo(o.getClass().getName());
168             }
169             return descriptor.compareTo(((TypeAnnotation) o).descriptor);
170             // could try to determine equivalence with ClassAnnotation, but don't
171             // see how this would be useful
172         }
173     
174         @Override
175         public String toString() {
176             String pattern = I18N.instance().getAnnotationDescription(roleDescription);
177             FindBugsMessageFormat format = new FindBugsMessageFormat(pattern);
178             return format.format(new BugAnnotation[] { this }, null);
179         }
180     
181         /*
182          * ----------------------------------------------------------------------
183          * XML Conversion support
184          * ----------------------------------------------------------------------
185          */
186     
187         private static final String ELEMENT_NAME = "Type";
188     
189         @Override
190         public void writeXML(XMLOutput xmlOutput) throws IOException {
191             writeXML(xmlOutput, false, false);
192         }
193     
194         @Override
195         public void writeXML(XMLOutput xmlOutput, boolean addMessages, boolean isPrimary) throws IOException {
196             XMLAttributeList attributeList = new XMLAttributeList().addAttribute("descriptor", descriptor);
197     
198             String role = getDescription();
199             if (!DEFAULT_ROLE.equals(role)) {
200                 attributeList.addAttribute("role", role);
201             }
202             if (typeParameters != null) {
203                 attributeList.addAttribute("typeParameters", typeParameters);
204             }
205     
206             BugAnnotationUtil.writeXML(xmlOutput, ELEMENT_NAME, this, attributeList, addMessages);
207         }
208     
209         @Override
210         public boolean isSignificant() {
211             return true;
212         }
213     }
