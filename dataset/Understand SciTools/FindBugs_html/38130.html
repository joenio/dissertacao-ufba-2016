20     package edu.umd.cs.findbugs;
21     
22     import java.io.IOException;
23     import java.util.Set;
24     
25     import org.dom4j.DocumentException;
26     
27     import edu.umd.cs.findbugs.classfile.IClassObserver;
28     import edu.umd.cs.findbugs.config.AnalysisFeatureSetting;
29     import edu.umd.cs.findbugs.config.UserPreferences;
30     import edu.umd.cs.findbugs.filter.FilterException;
31     
32     /**
33      * Interface for a FindBugs engine class. An instance of this interface takes a
34      * project, user configuration options, orchestrates the analysis of the classes
35      * in the project, and reports the results to the configured BugReporter.
36      *
37      * @author David Hovemeyer
38      */
39     public interface IFindBugsEngine {
40     
41         /**
42          * Get the BugReporter.
43          *
44          * @return the BugReporter
45          */
46         public BugReporter getBugReporter();
47     
48         /**
49          * Set the BugReporter.
50          *
51          * @param bugReporter
52          *            The BugReporter to set
53          */
54         public void setBugReporter(BugReporter bugReporter);
55     
56         /**
57          * Set the Project.
58          *
59          * @param project
60          *            The Project to set
61          */
62         public void setProject(Project project);
63     
64         /**
65          * Get the Project.
66          *
67          * @return the Project
68          */
69         public Project getProject();
70     
71         /**
72          * Set the progress callback that will be used to keep track of the progress
73          * of the analysis.
74          *
75          * @param progressCallback
76          *            the progress callback
77          */
78         public void setProgressCallback(FindBugsProgress progressCallback);
79     
80         /**
81          * Set filter of bug instances to include or exclude.
82          *
83          * @param filterFileName
84          *            the name of the filter file
85          * @param include
86          *            true if the filter specifies bug instances to include, false
87          *            if it specifies bug instances to exclude
88          */
89         public void addFilter(String filterFileName, boolean include) throws IOException, FilterException;
90     
91         /**
92          * Provide baseline of bugs not to report
93          *
94          * @param baselineBugs
95          *            the name of the xml bug baseline file
96          * @throws DocumentException
97          */
98         public void excludeBaselineBugs(String baselineBugs) throws IOException, DocumentException;
99     
100         /**
101          * Set the UserPreferences representing which Detectors should be used. If
102          * UserPreferences are not set explicitly, the default set of Detectors will
103          * be used.
104          *
105          * @param userPreferences
106          *            the UserPreferences
107          */
108         public void setUserPreferences(UserPreferences userPreferences);
109     
110         /**
111          * Add an IClassObserver.
112          *
113          * @param classObserver
114          *            the IClassObserver
115          */
116         public void addClassObserver(IClassObserver classObserver);
117     
118         /**
119          * Set the ClassScreener. This object chooses which individual classes to
120          * analyze. By default, all classes are analyzed.
121          *
122          * @param classScreener
123          *            the ClassScreener to use
124          */
125         public void setClassScreener(IClassScreener classScreener);
126     
127         /**
128          * Set relaxed reporting mode.
129          *
130          * @param relaxedReportingMode
131          *            true if relaxed reporting mode should be enabled, false if not
132          */
133         public void setRelaxedReportingMode(boolean relaxedReportingMode);
134     
135         /**
136          * Set whether or not training output should be emitted.
137          *
138          * @param trainingOutputDir
139          *            directory to save training output in
140          */
141         public void enableTrainingOutput(String trainingOutputDir);
142     
143         /**
144          * Set whether or not training input should be used to make the analysis
145          * more precise.
146          *
147          * @param trainingInputDir
148          *            directory to load training input from
149          */
150         public void enableTrainingInput(String trainingInputDir);
151     
152         /**
153          * Set analysis feature settings.
154          *
155          * @param settingList
156          *            list of analysis feature settings
157          */
158         public void setAnalysisFeatureSettings(AnalysisFeatureSetting[] settingList);
159     
160         /**
161          * @return Returns the releaseName.
162          */
163         public String getReleaseName();
164     
165         /**
166          * @param releaseName
167          *            The releaseName to set.
168          */
169         public void setReleaseName(String releaseName);
170     
171         /**
172          * @return Returns the projectName.
173          */
174         public String getProjectName();
175     
176         /**
177          * @param projectName
178          *            The project name to set.
179          */
180         public void setProjectName(String projectName);
181     
182         /**
183          * Set the filename of the source info file containing line numbers for
184          * fields and classes.
185          *
186          * @param sourceInfoFile
187          *            the source info filename
188          */
189         public void setSourceInfoFile(String sourceInfoFile);
190     
191         /**
192          * Execute FindBugs on the Project. All bugs found are reported to the
193          * BugReporter object which was set when this object was constructed.
194          *
195          * @throws java.io.IOException
196          *             if an I/O exception occurs analyzing one of the files
197          * @throws InterruptedException
198          *             if the thread is interrupted while conducting the analysis
199          */
200         public void execute() throws java.io.IOException, InterruptedException;
201     
202         /**
203          * Get the name of the most recent class to be analyzed. This is useful for
204          * diagnosing an unexpected exception. Returns null if no class has been
205          * analyzed.
206          */
207         public String getCurrentClass();
208     
209         /**
210          * Get the number of bug instances that were reported during analysis.
211          */
212         public int getBugCount();
213     
214         /**
215          * Get the number of errors that occurred during analysis.
216          */
217         public int getErrorCount();
218     
219         /**
220          * Get the number of time missing classes were reported during analysis.
221          */
222         public int getMissingClassCount();
223     
224         /**
225          * Get the UserPreferences.
226          *
227          * @return the UserPreferences
228          */
229         public UserPreferences getUserPreferences();
230     
231         /**
232          * Return whether or not training output should be emitted after analysis
233          * completes.
234          *
235          * @return true if training output should be emitted, false if not
236          */
237         public boolean emitTrainingOutput();
238     
239         /**
240          * Get the training output directory.
241          *
242          * @return the training output directory
243          */
244         public String getTrainingOutputDir();
245     
246         /**
247          * Return whether or not we should make use of training data.
248          *
249          * @return true if training data should be used, false if not
250          */
251         public boolean useTrainingInput();
252     
253         /**
254          * Get the training input database directory.
255          *
256          * @return the training input database directory
257          */
258         public String getTrainingInputDir();
259     
260         /**
261          * Set whether or not nested archives should be scanned.
262          *
263          * @param scanNestedArchives
264          *            true if nested archives should be scanned, false if not
265          */
266         public void setScanNestedArchives(boolean scanNestedArchives);
267     
268         /**
269          * Set whether or not to generate an empty output file if there were no
270          * class files specified.
271          *
272          * @param noClassOk
273          *            true if FindBugs should generate empty output file
274          */
275         public void setNoClassOk(boolean noClassOk);
276     
277         /**
278          * Set the DetectorFactoryCollection from which plugins/detectors may be
279          * accessed.
280          *
281          * @param detectorFactoryCollection
282          *            the DetectorFactoryCollection
283          */
284         public void setDetectorFactoryCollection(DetectorFactoryCollection detectorFactoryCollection);
285     
286         /**
287          * @param xmlWithAbridgedMessages
288          */
289         public void setAbridgedMessages(boolean xmlWithAbridgedMessages);
290     
291         public void setMergeSimilarWarnings(boolean mergeSimilarWarnings);
292     
293         public void setApplySuppression(boolean applySuppression);
294     
295         public void finishSettings();
296     
297         void setRankThreshold(int rankThreshold);
298     
299         void setBugReporterDecorators(Set<String> explicitlyEnabled, Set<String> explicitlyDisabled);
300     
301     }
