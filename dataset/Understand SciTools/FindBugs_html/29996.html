19     package edu.umd.cs.findbugs.props;
20     
21     /**
22      * General warning properties. These are properties that could be attached to
23      * any warning to provide information which might be useful in determining
24      * whether or not the bug is a false positive, and/or the severity of the
25      * warning.
26      *
27      * @author David Hovemeyer
28      */
29     public class GeneralWarningProperty extends AbstractWarningProperty {
30         private GeneralWarningProperty(String name, PriorityAdjustment priorityAdjustment) {
31             super(name, priorityAdjustment);
32         }
33     
34         /**
35          * The type of the receiver object in a method call or instance field
36          * access.
37          */
38         public static final GeneralWarningProperty RECEIVER_OBJECT_TYPE = new GeneralWarningProperty("RECEIVER_OBJECT_TYPE",
39                 PriorityAdjustment.NO_ADJUSTMENT);
40     
41         /** Name of most recently called method. */
42         public static final GeneralWarningProperty CALLED_METHOD_1 = new GeneralWarningProperty("CALLED_METHOD_1",
43                 PriorityAdjustment.NO_ADJUSTMENT);
44     
45         /** Name of second-most recently called method. */
46         public static final GeneralWarningProperty CALLED_METHOD_2 = new GeneralWarningProperty("CALLED_METHOD_2",
47                 PriorityAdjustment.NO_ADJUSTMENT);
48     
49         /** Name of third-most recently called method. */
50         public static final GeneralWarningProperty CALLED_METHOD_3 = new GeneralWarningProperty("CALLED_METHOD_3",
51                 PriorityAdjustment.NO_ADJUSTMENT);
52     
53         /** Name of fourth-most recently called method. */
54         public static final GeneralWarningProperty CALLED_METHOD_4 = new GeneralWarningProperty("CALLED_METHOD_4",
55                 PriorityAdjustment.NO_ADJUSTMENT);
56     
57         /** Warning occurs on an exception control path. */
58         public static final GeneralWarningProperty ON_EXCEPTION_PATH = new GeneralWarningProperty("ON_EXCEPTION_PATH",
59                 PriorityAdjustment.NO_ADJUSTMENT);
60     
61         /** issue is in uncallable method */
62         public static final GeneralWarningProperty IN_UNCALLABLE_METHOD = new GeneralWarningProperty("IN_UNCALLABLE_METHOD",
63                 PriorityAdjustment.AT_MOST_LOW);
64     
65         public static final GeneralWarningProperty FALSE_POSITIVE = new GeneralWarningProperty("FALSE_POSITIVE",
66                 PriorityAdjustment.FALSE_POSITIVE);
67     
68         public static final GeneralWarningProperty NOISY_BUG = new GeneralWarningProperty("NOISY_BUG",
69                 PriorityAdjustment.AT_MOST_MEDIUM);
70     
71         public static final GeneralWarningProperty SILENT_BUG = new GeneralWarningProperty("SILENT_BUG",
72                 PriorityAdjustment.NO_ADJUSTMENT);
73     
74     }
