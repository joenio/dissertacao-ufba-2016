20     package edu.umd.cs.findbugs.detect;
21     
22     import java.util.HashSet;
23     import java.util.Set;
24     
25     import org.apache.bcel.classfile.Method;
26     
27     import edu.umd.cs.findbugs.BugInstance;
28     import edu.umd.cs.findbugs.BugReporter;
29     import edu.umd.cs.findbugs.FieldAnnotation;
30     import edu.umd.cs.findbugs.OpcodeStack.Item;
31     import edu.umd.cs.findbugs.ba.XField;
32     import edu.umd.cs.findbugs.bcel.OpcodeStackDetector;
33     import edu.umd.cs.findbugs.classfile.Global;
34     import edu.umd.cs.findbugs.detect.FindNoSideEffectMethods.MethodSideEffectStatus;
35     import edu.umd.cs.findbugs.detect.FindNoSideEffectMethods.NoSideEffectMethodsDatabase;
36     
37     public class FindDoubleCheck extends OpcodeStackDetector {
38         static final boolean DEBUG = false;
39     
40         int stage = 0;
41     
42         int startPC, endPC, assignPC;
43     
44         int count;
45     
46         boolean sawMonitorEnter;
47     
48         Set<FieldAnnotation> fields = new HashSet<FieldAnnotation>();
49     
50         Set<FieldAnnotation> twice = new HashSet<FieldAnnotation>();
51     
52         FieldAnnotation pendingFieldLoad;
53     
54         XField currentDoubleCheckField;
55     
56         int countSinceGetReference;
57     
58         int countSinceGetBoolean;
59     
60         private final BugReporter bugReporter;
61     
62         private final NoSideEffectMethodsDatabase nse;
63     
64         public FindDoubleCheck(BugReporter bugReporter) {
65             this.bugReporter = bugReporter;
66             this.nse = Global.getAnalysisCache().getDatabase(NoSideEffectMethodsDatabase.class);
67         }
68     
69         @Override
70         public void visit(Method obj) {
71             if (DEBUG) {
72                 System.out.println(getFullyQualifiedMethodName());
73             }
74             super.visit(obj);
75             fields.clear();
76             twice.clear();
77             stage = 0;
78             count = 0;
79             countSinceGetReference = 1000;
80             countSinceGetBoolean = 1000;
81             sawMonitorEnter = false;
82             pendingFieldLoad = null;
83             currentDoubleCheckField = null;
84         }
85     
86         @Override
87         public void sawOpcode(int seen) {
88             if (DEBUG) {
89                 System.out.println(getPC() + "\t" + OPCODE_NAMES[seen] + "\t" + stage + "\t" + count + "\t" + countSinceGetReference);
90             }
91     
92             if (seen == MONITORENTER) {
93                 sawMonitorEnter = true;
94             }
95             if (seen == GETFIELD || seen == GETSTATIC) {
96                 pendingFieldLoad = FieldAnnotation.fromReferencedField(this);
97                 if (DEBUG) {
98                     System.out.println("\t" + pendingFieldLoad);
99                 }
100                 String sig = getSigConstantOperand();
101                 if ("Z".equals(sig)) {
102                     countSinceGetBoolean = 0;
103                     countSinceGetReference++;
104                 } else if (sig.startsWith("L") || sig.startsWith("[")) {
105                     countSinceGetBoolean++;
106                     countSinceGetReference = 0;
107                 }
108             } else {
109                 countSinceGetReference++;
110             }
111             switch (stage) {
112             case 0:
113                 if (((seen == IFNULL || seen == IFNONNULL) && countSinceGetReference < 5)
114                         || ((seen == IFEQ || seen == IFNE) && countSinceGetBoolean < 5)) {
115                     int b = getBranchOffset();
116                     if (DEBUG) {
117                         System.out.println("branch offset is : " + b);
118                     }
119                     if (b > 0 && !(seen == IFNULL && b > 9) && !(seen == IFEQ && (b > 9 && b < 34))
120                             && !(seen == IFNE && (b > 9 && b < 34)) && (!sawMonitorEnter)) {
121                         fields.add(pendingFieldLoad);
122                         startPC = getPC();
123                         stage = 1;
124                     }
125                 }
126                 count = 0;
127                 break;
128             case 1:
129                 if (seen == MONITORENTER) {
130                     stage = 2;
131                     count = 0;
132                 } else if (((seen == IFNULL || seen == IFNONNULL) && countSinceGetReference < 5)
133                         || ((seen == IFEQ || seen == IFNE) && countSinceGetBoolean < 5)) {
134                     int b = getBranchOffset();
135                     if (b > 0 && (seen == IFNONNULL || b < 10)) {
136                         fields.add(pendingFieldLoad);
137                         startPC = getPC();
138                         count = 0;
139                     }
140                 } else {
141                     count++;
142                     if (count > 10) {
143                         stage = 0;
144                     }
145                 }
146                 break;
147             case 2:
148                 if (((seen == IFNULL || seen == IFNONNULL) && countSinceGetReference < 5)
149                         || ((seen == IFEQ || seen == IFNE) && countSinceGetBoolean < 5)) {
150                     if (getBranchOffset() >= 0 && fields.contains(pendingFieldLoad)) {
151                         endPC = getPC();
152                         stage++;
153                         twice.add(pendingFieldLoad);
154                         break;
155                     }
156                 }
157                 count++;
158                 if (count > 10) {
159                     stage = 0;
160                 }
161                 break;
162             case 3:
163                 if (seen == PUTFIELD || seen == PUTSTATIC) {
164                     FieldAnnotation f = FieldAnnotation.fromReferencedField(this);
165                     if (DEBUG) {
166                         System.out.println("\t" + f);
167                     }
168                     if (twice.contains(f) && !getNameConstantOperand().startsWith("class$")
169                             && !"Ljava/lang/String;".equals(getSigConstantOperand())) {
170                         XField declaration = getXFieldOperand();
171                         if (declaration == null || !declaration.isVolatile()) {
172                             bugReporter.reportBug(new BugInstance(this, "DC_DOUBLECHECK", NORMAL_PRIORITY).addClassAndMethod(this)
173                                     .addField(f).describe("FIELD_ON").addSourceLineRange(this, startPC, endPC));
174                         } else {
175                             if(declaration.isReferenceType()) {
176                                 currentDoubleCheckField = declaration;
177                                 assignPC = getPC();
178                             }
179                         }
180                         stage++;
181                     }
182                 }
183                 break;
184             case 4:
185                 if(currentDoubleCheckField != null) {
186                     switch(seen) {
187                     case MONITOREXIT:
188                         stage++;
189                         break;
190                     case INVOKEINTERFACE:
191                     case INVOKESPECIAL:
192                     case INVOKEVIRTUAL:
193                         if(nse.is(getMethodDescriptorOperand(), MethodSideEffectStatus.OBJ, MethodSideEffectStatus.SE)) {
194                             checkStackValue(getNumberArguments(getMethodDescriptorOperand().getSignature()));
195                         }
196                         break;
197                     case PUTFIELD:
198                         checkStackValue(1);
199                         break;
200                     case DASTORE:
201                     case FASTORE:
202                     case SASTORE:
203                     case LASTORE:
204                     case BASTORE:
205                     case CASTORE:
206                     case AASTORE:
207                     case IASTORE:
208                         checkStackValue(2);
209                         break;
210                     }
211                 }
212                 break;
213             default:
214                 break;
215             }
216         }
217     
218         private void checkStackValue(int arg) {
219             Item item = getStack().getStackItem(arg);
220             if(item.getXField() == currentDoubleCheckField) {
221                 bugReporter.reportBug(new BugInstance(this, "DC_PARTIALLY_CONSTRUCTED", NORMAL_PRIORITY).addClassAndMethod(this)
222                         .addField(currentDoubleCheckField).describe("FIELD_ON").addSourceLine(this).addSourceLine(this, assignPC)
223                         .describe("SOURCE_LINE_STORED"));
224                 stage++;
225             }
226         }
227     }
