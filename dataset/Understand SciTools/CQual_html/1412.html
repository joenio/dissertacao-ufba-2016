19     #ifndef TIMER_H
20     #define TIMER_H
21     
22     #include <sys/time.h>
23     #include "linkage.h"
24     
25     EXTERN_C_BEGIN
26     
27     #ifdef TIMER_USERTIME
28     typedef struct timer
29     {
30       unsigned int start;   /* When this timer was started */
31       unsigned int elapsed; /* Current elapsed time */
32     } timer;
33     
34     void reset_timer(timer *t);
35     #else
36     typedef struct timer
37     {
38       struct timeval start;   /* When this timer was started */
39       struct timeval elapsed; /* Current elapsed time */
40     } timer;
41     #endif
42     
43     /* #define ZERO_TIMER ((timer) {{0, 0}, {0, 0}}) */
44     
45     /* Begin timing */
46     void start_timer(timer *t);
47     
48     /* End timing, adding difference between start time and current time
49        to elapsed time */
50     void end_timer(timer *t);
51     
52     /* Convert elapsed time of t into a string */
53     char *timer_to_ascii(timer *t);
54     
55     EXTERN_C_END
56     
57     #endif
