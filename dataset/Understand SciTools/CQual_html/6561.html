19     #include <regions.h>
20     #include <assert.h>
21     #include <limits.h>
22     #include "array.h"
23     
24     struct array {
25       region sameregion r;
26       void *sameregion data;
27       size_t elemsize;
28       type_t elemtype;
29       size_t nelems, nalloc;
30     };
31     
32     struct array *new_array(region r, size_t initialsize,
33     			size_t typesize, type_t typeinfo)
34     {
35       struct array *a = ralloc(r, struct array);
36     
37       a->r = r;
38       a->data = typed_rarrayalloc(r, initialsize, typesize, typeinfo);
39       a->elemsize = typesize;
40       a->elemtype = typeinfo;
41       a->nelems = 0;
42       a->nalloc = initialsize;
43     
44       return a;
45     }
46     
47     void *array_extend(struct array *a, int by)
48     {
49       size_t oldelems = a->nelems;
50     
51       if (by < 0)
52         assert(-by <= a->nelems && by != INT_MIN);
53       else if (a->nelems + by > a->nalloc)
54         {
55           size_t newsize = a->nalloc * 2 + by;
56           void *newdata = typed_rarrayalloc(a->r, newsize, a->elemsize, a->elemtype);
57     
58           /* XXX: could work harder to support really large array sizes
59     	 (this code will fail for a->nalloc >= (max(size_t)-by)/2) */
60           assert(newsize > a->nalloc); /* die when we get really big */
61           typed_rarraycopy(newdata, a->data, a->nelems, a->elemsize, a->elemtype);
62           a->data = newdata;
63           a->nalloc = newsize;
64         }
65       a->nelems += by;
66     
67       return (char *)a->data + a->elemsize * oldelems;
68     }
69     
70     void array_reset(struct array *a)
71     {
72       a->nelems = 0;
73     }
74     
75     size_t array_length(struct array *a)
76     {
77       return a->nelems;
78     }
79     
80     void *array_data(struct array *a)
81     {
82       return a->data;
83     }
