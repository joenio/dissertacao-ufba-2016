126     data_declaration declare(environment b, data_declaration from,
127     			 bool ignore_shadow)
128     {
129       data_declaration dd = ralloc(parse_region, struct data_declaration);
130     
131       *dd = *from;
132     
133       if (dd->name)
134         {
135           /* In PCC-compatibility mode, extern decls of vars with no current decl
136     	 take effect at top level no matter where they are.  */
137           /* We don't do the GCC "type exists at global scope" check because
138     	 it's mostly meaningless.
139     	 XXX: review if I start freeing mem earlier */
140           if (flag_traditional && dd->isfilescoperef && !lookup_id(dd->name, FALSE))
141     	b = global_env;
142     
143           if (b != global_env)
144     	{
145     	  /* Here to install a non-global value.  */
146     	  data_declaration shadowed = lookup_id(dd->name, FALSE);
147     
148     	  /* Warn if shadowing an argument at the top level of the body.  */
149     	  if (shadowed && shadowed->islocal && !dd->isfilescoperef
150     	      /* This warning doesn't apply to the parms of a nested fcn.  */
151     	      && !b->parm_level
152     	      /* Check that this is one level down from the parms.  */
153     	      && b->parent->parm_level
154     	      /* Check that the decl exists in the parm level */
155     	      && env_lookup(b->parent->id_env, dd->name, TRUE))
156     	    {
157     	      if (shadowed->isparameter)
158     		pedwarn("declaration of `%s' shadows a parameter", dd->name);
159     	      else
160     		pedwarn("declaration of `%s' shadows a symbol from the parameter list",
161     			 dd->name);
162     	    }
163     	  /* Maybe warn if shadowing something else.  */
164     	  else if (warn_shadow && !ignore_shadow && !dd->isfilescoperef)
165     	    {
166     	      char *warnstring = 0;
167     
168     	      if (dd->isparameter
169     		  && b->parent->parm_level)
170     		/* Don't warn about the parm names in function declarator
171     		   within a function declarator.
172     		   It would be nice to avoid warning in any function
173     		   declarator in a declaration, as opposed to a definition,
174     		   but there is no way to tell it's not a definition.  */
175     		;
176     	      else if (shadowed && shadowed->isparameter)
177     		warnstring = "declaration of `%s' shadows a parameter";
178     	      else if (shadowed && shadowed->islocal)
179     		warnstring = "declaration of `%s' shadows previous local";
180     	      else if (shadowed)
181     		warnstring = "declaration of `%s' shadows global declaration";
182     
183     	      if (warnstring)
184     		(error_shadow ? error : warning)(warnstring, dd->name);
185     	    }
186     	}
187     
188           /* Add a limbo version of external identifiers to the global scope
189     	 if this is the first time this external symbol is seen */
190           if (dd->isfilescoperef && b != global_env && !lookup_global_id(dd->name))
191     	{
192     	  data_declaration limbodd = ralloc(parse_region,
193     					    struct data_declaration);
194     
195     	  *limbodd = *dd;
196     	  limbodd->islimbo = TRUE;
197     	  env_add(global_env->id_env, limbodd->name, limbodd);
198     	  assert(!dd->shadowed);
199     	  dd->shadowed = limbodd;
200     
201     	  if (dd->kind == decl_function && dd->ftype == function_implicit &&
202     	      mesg_implicit_function_declaration)
203     	    warning_or_error(mesg_implicit_function_declaration != 2,
204     			     "implicit declaration of function `%s'", dd->name);
205     	}
206           /* Warn of implicit decl following explicit local extern decl.
207     	 This is probably a program designed for traditional C.  */
208           else if (dd->kind == decl_function && dd->ftype == function_implicit &&
209     	       warn_traditional)
210     	warning("function `%s' was previously declared within a block",
211     		dd->name);
212     
213           if (warn_main && (b == global_env || dd->isfilescoperef) &&
214     	  dd->kind != decl_function && !strcmp(dd->name, "main"))
215     	warning_with_decl(dd->ast, "`main' is usually a function");
216     
217           /* Parameters are declared before the function_decl is created
218     	 (and may be declared in several places). The id's of parameters
219     	 are set in start_function */
220           /*
221           if (dd->kind == decl_variable && dd->islocal && !dd->isparameter)
222     	dd->id = b->fdecl->nlocals++;
223           */
224         }
225       assert(!dd->islimbo);
226       env_add(b->id_env, dd->name, dd);
227     
228       return dd;
229     }

234       const char *name = t->word1 ? t->word1->cstring.data : NULL;

236       tdecl->kind = t->kind;

304     /* Like shadow_tag, but warned is: 1 => we have done a pedwarn;
305        2 => we have done a warning, but no pedwarn. */
306     void shadow_tag_warned(type_element elements, int warned)
307     {
308       type_element elem;
309       int found_tag = 0;
310     
311       pending_invalid_xref = 0;
312     
313       scan_type_element (elem, elements)
314         {
315           ast_kind code = elem->kind;
316     
317           if (code == kind_struct_ref || code == kind_union_ref ||
318     	  code == kind_enum_ref)
319     	{
320     	  tag_ref tag = CAST(tag_ref, elem);
321     	  word name = tag->word1;
322     
323     	  found_tag++;
324     
325     	  if (name == 0)
326     	    {
327     	      if (warned != 1 && code != kind_enum_ref)
328     		/* Empty unnamed enum OK */
329     		{
330     		  pedwarn ("unnamed struct/union that defines no instances");
331     		  warned = 1;
332     		}
333     	    }
334     	  else
335     	    {
336     	      void *tagdecl = lookup_tag(tag, TRUE);
337     
338     	      if (tagdecl == 0)
339     		declare_tag(tag);
340     	      else
341     		pending_xref_error();
342     	    }
343     	}
344           else
345     	{
346     	  if (!warned && ! in_system_header)
347     	    {
348     	      warning("useless keyword or type name in empty declaration");
349     	      warned = 2;
350     	    }
351     	}
352         }
353     
354       if (found_tag > 1)
355         error("two types specified in one empty declaration");
356     
357       if (warned != 1)
358         {
359           if (found_tag == 0)
360     	pedwarn("empty declaration");
361         }
362     }

380     /* True if the parameters is just the parameter list '(void)' */
381     bool is_void_parms(declaration parms)
382     {
383       data_decl dd;
384       variable_decl vd;
385     
386       if (!parms || parms->next || !is_data_decl(parms))
387         return FALSE;
388     
389       dd = CAST(data_decl, parms);
390       vd = CAST(variable_decl, dd->decls);
391       assert(!vd->next);
392     
393       return !vd->declarator && dd->modifiers && !dd->modifiers->next &&
394         is_rid(dd->modifiers) && CAST(rid, dd->modifiers)->id == RID_VOID;
395     }

408     /* At end of parameter list, warn about any struct, union or enum tags
409        defined within.  Do so because these types cannot ever become complete.  */
410     static void parmlist_tags_warning(environment parm_env)
411     {
412       env_scanner scan_tags;
413       const char *tagname;
414       tag_declaration tagdecl;
415       static bool already = FALSE;
416     
417       env_scan(parm_env->tag_env, &scan_tags);
418     
419       while (env_next(&scan_tags, &tagname, (void **)&tagdecl))
420         {
421           int kind = tagdecl->kind;
422           const char *kindname = tagkind_name(kind);
423     
424           /* An anonymous union parm type is meaningful as a GNU extension.
425     	 So don't warn for that.  */
426           if (kind == kind_union_ref && !tagname && !pedantic)
427     	continue;
428     
429           if (tagname)
430     	warning("`%s %s' declared inside parameter list", kindname,
431     		tagname);
432           else
433     	warning("anonymous %s declared inside parameter list", kindname);
434     
435           if (!already)
436     	{
437     	  warning("its scope is only this definition or declaration,");
438     	  warning("which is probably not what you want.");
439     	  already = TRUE;
440     	}
441         }
442     }

447     typelist make_arg_types(bool definition, declaration parameters, bool *varargs,
448     			type_quals *varargs_quals,
449     			user_qual_list *varargs_user_quals)
450     {
451       declaration parameter;
452       typelist arg_types = new_typelist(parse_region);
453     
454       *varargs = FALSE;
455       *varargs_quals = no_qualifiers;
456       *varargs_user_quals = NULL;
457       if (!is_void_parms(parameters))
458         scan_declaration (parameter, parameters)
459           if (is_ellipsis_decl(parameter))
460     	{
461     	  *varargs = TRUE;
462     	  *varargs_quals = parse_qualifiers(parameter->loc,
463     				   CAST(ellipsis_decl, parameter)->qualifiers,
464     					    varargs_user_quals);
465     	}
466           else if (is_error_decl(parameter))
467     	{
468     	  /* Make an "accept everything" signature for arg lists with
469     	     error_decls */
470     	  *varargs = TRUE;
471     	  arg_types = new_typelist(parse_region);
472     	  typelist_append(arg_types, error_type);
473     
474     	  return arg_types;
475     	}
476           else
477     	{
478     	  data_decl dp = CAST(data_decl, parameter);
479     	  variable_decl vp = CAST(variable_decl, dp->decls);
480     
481     	  assert(!vp->next);
482     	  if (!vp->ddecl->name && definition)
483     	    error_with_decl(dp->decls, "parameter name omitted");
484     	  typelist_append(arg_types, vp->ddecl->type);
485     	}
486     
487       return arg_types;
488     }

507     static void check_duplicate_qualifiers(location l, type_quals new, type_quals old)
508     {
509       type_quals qual;
510     
511       for (qual = 1; qual < last_qualifier; qual <<= 1)
512         if (new & qual)
513           check_duplicate_qualifiers1(l, qual, old);
514     }

516     static type_quals parse_qualifiers(location l, type_element qlist,
517     				   user_qual_list *user_quals_out)
518     {
519       type_element qual;
520       type_quals tqs = no_qualifiers;
521       user_qual_list user_quals = NULL;
522     
523       /* Actually qlist is a list of qualifier */
524       scan_type_element (qual, qlist)
525         {
526           switch (qual->kind)
527     	{
528     	case kind_qualifier:
529     	  {
530     	    qualifier q = CAST(qualifier, qual);
531     	    check_duplicate_qualifiers1(q->loc, q->id, tqs);
532     	    tqs |= q->id;
533     	  }
534     	  break;
535     	case kind_user_qual:
536     	  user_quals = new_user_qual_list(CAST(user_qual, qual),
537     					  user_quals);
538     	  break;
539     	default: assert(0);
540     	}
541         }
542        *user_quals_out = user_quals;
543        return tqs;
544     }

546     void parse_declarator(type_element modifiers, declarator d, bool bitfield, 
547     		      int *oclass, bool *oinlinep, const char **oname,
548     		      type *ot, bool *owarn_defaulted_int,
549     		      function_declarator *ofunction_declarator)
550     {
551       location loc = d ? d->loc : modifiers->loc;
552       int specbits = 0, nclasses = 0;
553       type_quals specquals = no_qualifiers;
554       bool longlong = FALSE;
555       const char *printname;
556       type_element spec;
557       type t = NULL;
558       bool modified;
559       user_qual_list user_quals = NULL;
560     
561       *owarn_defaulted_int = FALSE;
562     
563       /* We get the name now so that we have a name for error messages */
564       *oname = declarator_name(d);
565       printname = *oname ? *oname : "type name";
566     
567       *oclass = 0;
568     
569       scan_type_element (spec, modifiers)
570         {
571           type newtype = NULL;
572     
573           switch (spec->kind)
574     	{
575     	case kind_rid:
576     	  {
577     	    rid rspec = CAST(rid, spec);
578     	    int id = rspec->id;
579     
580     	    switch (id)
581     	      {
582     	      case RID_INT: newtype = int_type; break;
583     	      case RID_CHAR: newtype = char_type; break;
584     	      case RID_FLOAT: newtype = float_type; break;
585     	      case RID_DOUBLE: newtype = double_type; break;
586     	      case RID_VOID: newtype = void_type; break;
587     		
588     	      case RID_AUTO: case RID_STATIC: case RID_EXTERN:
589     	      case RID_REGISTER: case RID_TYPEDEF:
590     		*oclass = id;
591     		nclasses++;
592     		break;
593     
594     	      case RID_LONG: /* long long detection */
595     		if (specbits & 1 << RID_LONG) 
596     		  {
597     		    if (longlong)
598     		      error_with_location(spec->loc,
599     					  "`long long long' is too long for GCC");
600     		    else
601     		      {
602     			if (pedantic && !in_system_header)
603     			  pedwarn_with_location(spec->loc,
604     						"ANSI C does not support `long long'");
605     			longlong = TRUE;
606     		      }
607     		    break;
608     		  }
609     		/* Fall through */
610     	      default:
611     		check_duplicate_rid(specbits, rspec);
612     		break;
613     	      }
614     	    specbits |= 1 << id;
615     	    break;
616     	  }
617     	case kind_qualifier:
618     	  {
619     	    qualifier q = CAST(qualifier, spec);
620     	    int id = q->id;
621     
622     	    check_duplicate_qualifiers1(loc, id, specquals);
623     	    specquals |= id;
624     	    break;
625     	  }
626     	case kind_typename:
627     	  newtype = CAST(typename, spec)->ddecl->type;
628     	  break;
629     	case kind_typeof_type:
630     	  newtype = CAST(typeof_type, spec)->asttype->type;
631     	  break;
632     	case kind_typeof_expr:
633     	  newtype = CAST(typeof_expr, spec)->arg1->type;
634     	  break;
635     	case kind_struct_ref: case kind_union_ref: case kind_enum_ref:
636     	  newtype = make_tagged_type(CAST(tag_ref, spec)->tdecl);
637     	  break;
638     	case kind_user_qual:
639     	  user_quals = new_user_qual_list(CAST(user_qual, spec), user_quals);
640     	  break;
641     	case kind_type_variable:
642     	  newtype = make_type_var(CAST(type_variable, spec)->cstring);
643     	  break;
644     	default: assert(0); break;
645     	}
646     
647           if (newtype)
648     	{
649     	  if (t)
650     	    error_with_location(spec->loc,
651     	       "two or more data types in declaration of `%s'", printname);
652     	  else
653     	    t = newtype;
654     	}
655         }
656     
657       /* Long double is a special combination.  */
658       if ((specbits & 1 << RID_LONG) && !longlong && (specbits & 1 << RID_DOUBLE))
659         {
660           specbits &= ~(1 << RID_LONG);
661           t = long_double_type;
662         }
663     
664       modified = !!(specbits & (1 << RID_LONG | 1 << RID_SHORT |
665     			    1 << RID_SIGNED | 1 << RID_UNSIGNED));
666     
667       /* No type at all: default to `int', or `double' (if complex specified
668          and no long/short/signed/unsigned) */
669       if (!t)
670         {
671           if (specbits & 1 << RID_COMPLEX)
672     	{
673     	  if (!modified)
674     	    {
675     	      specbits |= 1 << RID_DOUBLE;
676     	      t = double_type;
677     	    }
678     	  else
679     	    {
680     	      specbits |= 1 << RID_INT;
681     	      t = int_type;
682     	    }
683     	}
684           else
685     	{
686     	  /* Defer defaulted int warning to caller (msg depends on context) */
687     	  if (!modified)
688     	    *owarn_defaulted_int = TRUE;
689     
690     	  specbits |= 1 << RID_INT;
691     	  t = int_type;
692     	}
693         }
694     
695       if (nclasses > 1)
696         error_with_location(loc, "multiple storage classes in declaration of `%s'", printname);
697     
698       /* Now process the modifiers that were specified
699          and check for invalid combinations.  */
700     
701       /* Check all other uses of type modifiers.  */
702       if (modified)
703         {
704           int ok = 0;
705     
706           if ((specbits & 1 << RID_LONG) && (specbits & 1 << RID_SHORT))
707     	error_with_location(loc, "both long and short specified for `%s'", printname);
708           else if ((specbits & 1 << RID_SHORT) && !(specbits & 1 << RID_INT))
709     	{
710     	  static int already = 0;
711     
712     	  error_with_location(loc, "short invalid for `%s'", printname);
713     	  if (!already && !pedantic)
714     	    {
715     	      error_with_location(loc, "short is only valid with int");
716     	      already = 1;
717     	    }
718     	}
719           else if ((specbits & 1 << RID_LONG) && !(specbits & 1 << RID_INT))
720     	{
721     	  static int already = 0;
722     
723     	  error_with_location(loc, "long invalid for `%s'", printname);
724     	  if (!already && !pedantic)
725     	    {
726     	      error_with_location(loc, "long is only valid with int or double");
727     	      already = 1;
728     	    }
729     	}
730           else if ((specbits & 1 << RID_SIGNED) && (specbits & 1 << RID_UNSIGNED))
731     	error_with_location(loc, "both signed and unsigned specified for `%s'", printname);
732           else if (((specbits & 1 << RID_SIGNED) || (specbits & 1 << RID_UNSIGNED))
733     	       && !(specbits & (1 << RID_INT | 1 << RID_CHAR)))
734     	error_with_location(loc, "signed or unsigned invalid for `%s'", printname);
735           else
736     	ok = 1;
737     
738           /* Discard the type modifiers if they are invalid.  */
739           if (! ok)
740     	{
741     	  specbits &= ~(1 << RID_LONG | 1 << RID_SHORT
742     			| 1 << RID_UNSIGNED | 1 << RID_SIGNED);
743     	  longlong = 0;
744     	}
745         }
746     
747       if ((specbits & 1 << RID_COMPLEX) && !(type_integral(t) || type_floating(t)))
748         {
749           error_with_location(loc, "complex invalid for `%s'", printname);
750           specbits &= ~(1 << RID_COMPLEX);
751         }
752     
753       /* Decide whether an integer type is signed or not.
754          Optionally treat bitfields as signed by default.  */
755       if ((specbits & 1 << RID_UNSIGNED)
756           /* Traditionally, all bitfields are unsigned.  */
757           || (bitfield && flag_traditional
758     	  && (!explicit_flag_signed_bitfields || !flag_signed_bitfields))
759           || (bitfield && !flag_signed_bitfields
760     	  && ((specbits & 1 << RID_INT) || (specbits & 1 << RID_CHAR))
761     	  && !(specbits & 1 << RID_SIGNED)))
762         {
763           if (longlong)
764     	t = unsigned_long_long_type;
765           else if (specbits & 1 << RID_LONG)
766     	t = unsigned_long_type;
767           else if (specbits & 1 << RID_SHORT)
768     	t = unsigned_short_type;
769           else if (t == char_type)
770     	t = unsigned_char_type;
771           else
772     	t = unsigned_int_type;
773         }
774       else if ((specbits & 1 << RID_SIGNED) && (specbits & 1 << RID_CHAR))
775         t = signed_char_type;
776       else if (longlong)
777         t = long_long_type;
778       else if (specbits & 1 << RID_LONG)
779         t = long_type;
780       else if (specbits & 1 << RID_SHORT)
781         t = short_type;
782     
783       if (specbits & 1 << RID_COMPLEX)
784         t = make_complex_type(t);
785     
786       /* Check for qualifiers redundant with base type */
787       check_duplicate_qualifiers(loc, specquals, type_qualifiers(t));
788     
789       specquals |= type_qualifiers(t);
790       t = make_qualified_type(t, specquals, user_quals);
791     
792       *oinlinep = !!(specbits & 1 << RID_INLINE);
793       if (pedantic && type_function(t) && (type_const(t) || type_volatile(t)) &&
794           !in_system_header)
795         pedwarn_with_location(loc, "ANSI C forbids const or volatile function types");
796     
797       /* Now figure out the structure of the declarator proper.
798          Descend through it, creating more complex types, until we reach
799          the declared identifier (or NULL, in an abstract declarator).  */
800     
801       while (d && d->kind != kind_identifier_declarator)
802         {
803           switch (d->kind)
804     	{
805     	case kind_array_declarator:
806     	  {
807     	    array_declarator ad = CAST(array_declarator, d);
808     	    expression size = ad->arg1;
809     
810     	    d = ad->declarator;
811     
812     	    /* Check for some types that there cannot be arrays of.  */
813     	    if (type_void(t))
814     	      {
815     		error_with_location(d->loc,
816     				    "declaration of `%s' as array of voids", printname);
817     		t = error_type;
818     	      }
819     
820     	    if (type_function(t))
821     	      {
822     		error_with_location(d->loc,
823     				    "declaration of `%s' as array of functions", printname);
824     		t = error_type;
825     	      }
826     
827     	    if (size && is_error_expr(size))
828     	      t = error_type;
829     
830     	    if (t == error_type)
831     	      continue;
832     
833     	    if (size)
834     	      {
835     		if (!type_integer(size->type))
836     		  {
837     		    error_with_location(d->loc,
838     					"size of array `%s' has non-integer type", printname);
839     		    size = oneexpr;
840     		  }
841     
842     		if (pedantic && definite_zero(size))
843     		  pedwarn_with_location(d->loc,
844     					"ANSI C forbids zero-size array `%s'", printname);
845     
846     		if (size->cst && constant_integral(size->cst))
847     		  {
848     		    if (pedantic)
849     		      constant_overflow_warning(size->cst);
850     		    if (cval_intcompare(size->cst->cval, cval_zero) < 0)
851     		      {
852     			error_with_location(d->loc,
853     					    "size of array `%s' is negative", printname);
854     			size = oneexpr;
855     		      }
856     		  }
857     		else
858     		  {
859     		    if (!current_function_decl)
860     		      {
861     			if (size->cst)
862     			  error_with_location(d->loc, "type size can't be explicitly evaluated");
863     			else
864     			  error_with_location(d->loc, "variable-size type declared outside of any function");
865     		      }
866     		    else if (pedantic)
867     		      {
868     			if (size->cst)
869     			  pedwarn_with_location(d->loc, "ANSI C forbids array `%s' whose size can't be evaluated", printname);
870     			else
871     			  pedwarn_with_location(d->loc, "ANSI C forbids variable-size array `%s'", printname);
872     		      }
873     
874     
875     		    size->cst = NULL; /* Not usefully constant, don't confuse
876     					 the array type */
877     		  }
878     	      }
879     
880     	    /* Build the array type itself, then merge any constancy or
881     	       volatility  */
882     	    t = make_array_type(t, size);
883     	    break;
884     	  }
885     
886     	case kind_function_declarator:
887     	  {
888     	    function_declarator fd = CAST(function_declarator, d);
889     	    user_qual_list user_quals;
890     	    type_quals fnquals = parse_qualifiers(fd->loc, fd->qualifiers,
891     						  &user_quals);
892     
893     	    d = fd->declarator;
894     	    if (ofunction_declarator)
895     	      *ofunction_declarator = fd;
896     
897     	    /* Declaring a function type.
898     	       Make sure we have a valid type for the function to return.  */
899     	    if (t == error_type)
900     	      t = int_type;
901     
902     	    /* Warn about some types functions can't return.  */
903     	    if (type_function(t))
904     	      {
905     		error_with_location(d->loc,
906     				    "`%s' declared as function returning a function",
907     		      printname);
908     		t = int_type;
909     	      }
910     	    if (type_array(t))
911     	      {
912     		error_with_location(d->loc,
913     				    "`%s' declared as function returning an array",
914     		      printname);
915     		t = int_type;
916     	      }
917     
918     #ifndef TRADITIONAL_RETURN_FLOAT
919     	    /* Traditionally, declaring return type float means double.  */
920     	    if (flag_traditional && type_float(t))
921     	      t = qualify_type1(double_type, t);
922     #endif /* TRADITIONAL_RETURN_FLOAT */
923     
924     	    /* Construct the function type */
925     	    if (new_style(fd->parms))
926     	      {
927     		bool varargs;
928     		bool definition = ofunction_declarator && 
929     		  d && d->kind == kind_identifier_declarator;
930     		type_quals varargs_quals;
931     		user_qual_list varargs_user_quals;
932     		typelist argtypes = make_arg_types(definition, fd->parms,
933     						   &varargs, &varargs_quals,
934     						   &varargs_user_quals);
935     		t = make_function_type(t, argtypes, varargs,
936     				       varargs_quals, varargs_user_quals,
937     				       FALSE);
938     	      }
939     	    else  /* Old-style function */
940     	      t = make_function_type(t, NULL, FALSE, no_qualifiers, NULL,TRUE);
941     
942     	    t = make_qualified_type(t, fnquals, user_quals);
943     	    break;
944     	  }
945     
946     	case kind_pointer_declarator:
947     	  {
948     	    pointer_declarator pd = CAST(pointer_declarator, d);
949     	    type_quals quals;
950     	    user_qual_list user_quals;
951     
952     	    d = pd->declarator;
953     	    quals = parse_qualifiers(pd->loc, pd->qualifiers,
954     				     &user_quals);
955     	    t = make_qualified_type(make_pointer_type(t),
956     				    quals, user_quals);
957     	    break;
958     	  }
959     
960     	default: assert(0);
961     	}
962         }
963     
964       *ot = t;
965     }

967     declarator make_function_declarator(location l, declarator d, declaration parms, type_element quals)
968     {
969       declaration parm;
970     
971       environment penv = poplevel();
972       function_declarator fd =
973         new_function_declarator(parse_region, l, d, parms, quals, penv);
974     
975       if (new_style(parms) && !is_void_parms(parms))
976         scan_declaration (parm, parms)
977           if (!is_ellipsis_decl(parm) && !is_error_decl(parm))
978     	{
979     	  variable_decl vp = CAST(variable_decl, CAST(data_decl, parm)->decls);
980     
981     	  if (!vp->ddecl)
982     	    {
983     	      error_with_location(l, "parameter declared void");

985     	    }
986     	}
987     
988       parmlist_tags_warning(penv);
989     
990       return CAST(declarator, fd);

993     /* Return zero if the declaration NEWDECL is valid
994        when the declaration OLDDECL (assumed to be for the same name and kind
995        of declaration) has already been seen.
996        Otherwise return an error message format string with a %s
997        where the identifier should go.  */
998     
999     static char *redeclaration_error_message(data_declaration newdecl,
1000     					 data_declaration olddecl,
1001     					 bool newinitialised)
1002     {
1003       if (olddecl->islimbo)
1004         return 0;
1005     
1006       if (newdecl->kind == decl_typedef)
1007         {
1008           if (flag_traditional && type_compatible(newdecl->type, olddecl->type))
1009     	return 0;
1010           /* This gets a warning later */
1011           if (olddecl->in_system_header || newdecl->in_system_header)
1012     	return 0;
1013           return "redefinition of `%s'";
1014         }
1015       else if (newdecl->kind == decl_function)
1016         {
1017           /* Declarations of functions can insist on internal linkage
1018     	 but they can't be inconsistent with internal linkage,
1019     	 so there can be no error on that account.
1020     	 However defining the same name twice is no good.  */
1021           if (olddecl->definition && newdecl->definition
1022     	  /* However, defining once as extern inline and a second
1023     	     time in another way is ok.  */
1024     	  && !(olddecl->isexterninline && !newdecl->isexterninline))
1025     	return "redefinition of `%s'";
1026           return 0;
1027         }
1028       else if (newdecl->kind == decl_constant)
1029         return "redefinition of `%s'";
1030       else if (current_env == global_env)
1031         {
1032           /* Objects declared at top level:  */
1033           /* If at least one is a reference, it's ok.  */
1034           if (newdecl->isfilescoperef || olddecl->isfilescoperef)
1035     	return 0;
1036           /* Reject two definitions with initialisation.  */
1037           if (newinitialised && olddecl->initialiser)
1038     	return "redefinition of `%s'";
1039           /* Now we have two tentative defs, or one tentative and one real def.  */
1040           /* Insist that the linkage match.  */
1041           if (olddecl->isexternalscope != newdecl->isexternalscope)
1042     	return "conflicting declarations of `%s'";
1043           return 0;
1044         }
1045       else
1046         {
1047           /* Newdecl has block scope.  If olddecl has block scope also, then
1048     	 reject two definitions, and reject a definition together with an
1049     	 external reference.  Otherwise, it is OK, because newdecl must
1050     	 be an extern reference to olddecl.  */
1051           if (!(newdecl->isexternalscope && olddecl->isexternalscope))
1052     #if 0
1053     	Why check the context ?
1054     	  && DECL_CONTEXT (newdecl) == DECL_CONTEXT (olddecl)
1055     #endif
1056     	return "redeclaration of `%s'";
1057           return 0;
1058         }
1059     }

1061     /* Return TRUE if t1 looks like a modern declaration for malloc (&friends) and
1062        t2 looks like an oldstyle declaration thereof */
1063     static bool looks_like_malloc_redeclaration(type t1, type t2)
1064     {
1065       type t1return = type_function_return_type(t1);
1066       type t2return = type_function_return_type(t2);
1067     
1068       return
1069         type_function_oldstyle(t1) &&
1070         type_pointer(t1return) && type_pointer(t2return) &&
1071         type_void(type_points_to(t2return)) &&
1072         type_char(type_points_to(t1return)) &&
1073         self_promoting_args(t2);
1074     }

1076     void show_previous_decl(void (*message)(declaration d, const char *format, ...),
1077     			data_declaration olddecl)
1078     {
1079       if (olddecl->kind == decl_function && olddecl->ftype == function_implicit)
1080         message(olddecl->ast, "previous implicit declaration of `%s'", olddecl->name);
1081       else
1082         message(olddecl->ast, "previous declaration of `%s'", olddecl->name);
1083     }

1085     /* Handle when a new declaration NEWDECL
1086        has the same name as an old one OLDDECL
1087        in the same binding contour.
1088        Prints an error message if appropriate.
1089     
1090        If safely possible, alter OLDDECL to look like NEWDECL, and return 1.
1091        Otherwise, return 0.
1092     
1093        When DIFFERENT_BINDING_LEVEL is true, NEWDECL is an external declaration,
1094        and OLDDECL is in an outer binding level and should thus not be changed.  */
1095     
1096     static int duplicate_decls(data_declaration newdecl, data_declaration olddecl,
1097     			   bool different_binding_level, bool newinitialised)
1098     {
1099       type oldtype = olddecl->type;
1100       type newtype = newdecl->type;
1101       char *errmsg = 0;
1102       const char *name = olddecl->name;
1103       void (*previous_message)(declaration d, const char *format, ...) = NULL;
1104       bool types_match;
1105     
1106       assert(!(newdecl->kind == decl_function &&
1107     	   newdecl->ftype == function_implicit));
1108     
1109       /* New decl is completely inconsistent with the old one =>
1110          tell caller to replace the old one. This is an error,
1111          except if traditional and in different binding levels */
1112       if (newdecl->kind != olddecl->kind)
1113         {
1114           bool iswarning = 
1115     	(flag_traditional && different_binding_level) || olddecl->islimbo;
1116     
1117           warning_or_error(iswarning,
1118     		       "`%s' redeclared as different kind of symbol", name);
1119           show_previous_decl(iswarning ? warning_with_decl : error_with_decl, olddecl);
1120           newdecl->shadowed = olddecl;
1121           return 0;
1122         }
1123     
1124       if (newtype == error_type || oldtype == error_type)
1125         types_match = FALSE;
1126       else
1127         types_match = type_compatible_unqualified(newtype, oldtype);
1128     
1129       /* For real parm decl following a forward decl, or a declaration of an old
1130          style parameter (oldtype == void) return 1 so old decl will be reused. */
1131       if ((oldtype == void_type || types_match) && newdecl->isparameter &&
1132           !olddecl->isused)
1133         return 1;
1134     
1135       /* The new declaration is the same kind of object as the old one.
1136          The declarations may partially match.  Print warnings if they don't
1137          match enough.  Ultimately, copy most of the information from the new
1138          decl to the old one, and keep using the old one.  */
1139     
1140       if (flag_traditional && olddecl->kind == decl_function
1141           && olddecl->ftype == function_implicit)
1142         /* If -traditional, avoid error for redeclaring fcn
1143            after implicit decl.  */
1144         ;
1145       /* Permit char *foo () to match void *foo (...) if not pedantic,
1146          if one of them came from a system header file.  */
1147       else if (!types_match && olddecl->kind == decl_function
1148     	   && (olddecl->in_system_header || newdecl->in_system_header)
1149     	   && (looks_like_malloc_redeclaration(oldtype, newtype) ||
1150     	       looks_like_malloc_redeclaration(newtype, oldtype)))
1151         {
1152           if (pedantic)
1153     	pedwarn_with_decl(newdecl->ast, "conflicting types for `%s'", name);
1154           /* Make sure we keep void * as ret type, not char *.  */
1155           if (type_void(type_points_to(type_function_return_type(oldtype))))
1156     	newdecl->type = newtype = oldtype;
1157     
1158           /* Set IN_SYSTEM_HEADER, so that if we see another declaration
1159     	 we will come back here again.  */
1160           newdecl->in_system_header = TRUE;
1161         }
1162       else if (!types_match
1163     	   /* Permit char *foo (int, ...); followed by char *foo ();
1164     	      if not pedantic.  */
1165     	   && !(olddecl->kind == decl_function && !pedantic
1166     		&& type_function_oldstyle(newtype)
1167     		/* Return types must still match.  */
1168     		&& type_compatible(type_function_return_type(oldtype),
1169     				   type_function_return_type(newtype))))
1170         {
1171           void (*message)(const char *format, ...) =
1172     	olddecl->islimbo ? warning : error;
1173     
1174           previous_message =
1175     	olddecl->islimbo ? warning_with_decl : error_with_decl;
1176     
1177           message("conflicting types for `%s'", name);
1178           /* Check for function type mismatch
1179     	 involving an empty arglist vs a nonempty one.  */
1180           if (newdecl->kind == decl_function
1181     	  && type_compatible(type_function_return_type(oldtype),
1182     			     type_function_return_type(newtype))
1183     	  && ((type_function_oldstyle(oldtype) && !olddecl->definition)
1184     	      || (type_function_oldstyle(newtype) && !newdecl->definition)))
1185     	{
1186     	  /* Classify the problem further.  */
1187     	  if (type_function_varargs(newtype) || type_function_varargs(oldtype))
1188     	    {
1189     	      message("A parameter list with an ellipsis can't match");
1190     	      message("an empty parameter name list declaration.");
1191     	    }
1192     	  else
1193     	    {
1194     	      typelist_scanner scanargs;
1195     	      type t;
1196     	      typelist args = type_function_arguments(oldtype);
1197     
1198     	      if (!args)
1199     		args = type_function_arguments(newtype);
1200     
1201     	      typelist_scan(args, &scanargs);
1202     	      while ((t = typelist_next(&scanargs)))
1203     		if (!type_self_promoting(t))
1204     		  {
1205     		    message("An argument type that has a default promotion");
1206     		    message("can't match an empty parameter name list declaration.");
1207     		    break;
1208     		  }
1209     	    }
1210     	}
1211         }
1212       else
1213         {
1214           errmsg = redeclaration_error_message(newdecl, olddecl, newinitialised);
1215           if (errmsg)
1216     	{
1217     	  error_with_decl(newdecl->ast, errmsg, name);
1218     	  previous_message = error_with_decl;
1219     	}
1220           else if (newdecl->kind == decl_typedef &&
1221     	       (olddecl->in_system_header || newdecl->in_system_header))
1222     	{
1223     	  warning_with_decl(newdecl->ast, "redefinition of `%s'");
1224     	  previous_message = warning_with_decl;
1225     	}
1226           else if (olddecl->kind == decl_function
1227     	       && olddecl->oldstyle_args
1228     	       && !type_function_oldstyle(newtype))
1229     	{
1230     	  int nargs;
1231     	  typelist_scanner oldparms, newparms;
1232     	  /* Prototype decl follows defn w/o prototype.  */
1233     
1234     	  typelist_scan(olddecl->oldstyle_args, &oldparms);
1235     	  typelist_scan(type_function_arguments(newtype), &newparms);
1236     	  nargs = 1;
1237     	  errmsg = NULL;
1238     	  for (;;)
1239     	    {
1240     	      type oldparm = typelist_next(&oldparms);
1241     	      type newparm = typelist_next(&newparms);
1242     
1243     	      if (!oldparm && !newparm)
1244     		break;
1245     
1246     	      if (!oldparm || !newparm)
1247     		{
1248     		  errmsg = "prototype for `%s' follows and number of arguments";
1249     		  break;
1250     		}
1251     	      /* Type for passing arg must be consistent
1252     		 with that declared for the arg.  */
1253     	      if (!type_compatible(oldparm, newparm)
1254     		  /* If -traditional, allow `unsigned int' instead of `int'
1255     		     in the prototype.  */
1256     		  && !(flag_traditional
1257     		       && type_equal_unqualified(oldparm, int_type)
1258     		       && type_equal_unqualified(newparm, unsigned_int_type)))
1259     		{
1260     		  errmsg = "prototype for `%s' follows and argument %d";
1261     		  break;
1262     		}
1263     	      nargs++;
1264     	    }
1265     	  if (errmsg)
1266     	    {
1267     	      warning_or_error_with_decl(olddecl->islimbo, newdecl->ast,
1268     					 errmsg, name, nargs);
1269     	      warning_or_error_with_decl(olddecl->islimbo, olddecl->ast,
1270     			      "doesn't match non-prototype definition here");
1271     	    }
1272     	  else
1273     	    {
1274     	      warning_with_decl(newdecl->ast, "prototype for `%s' follows",
1275     				name);
1276     	      warning_with_decl(olddecl->ast, "non-prototype definition here");
1277     	    }
1278     	}
1279           /* Warn about mismatches in various flags. */
1280           else if (newdecl->kind == decl_function)
1281     	{
1282     	  /* Warn if function is now inline
1283     	     but was previously declared not inline and has been called. */
1284     	  if (!olddecl->isinline && newdecl->isinline)
1285     	    {
1286     	      if (olddecl->isused)
1287     		{
1288     		  warning("`%s' declared inline after being called", name);
1289     		  previous_message = warning_with_decl;
1290     		}
1291     	      if (olddecl->definition)
1292     		{
1293     		  warning("`%s' declared inline after its definition", name);
1294     		  previous_message = warning_with_decl;
1295     		}
1296     	    }
1297     
1298     	  /* Warn for static following external */
1299     	  if (newdecl->ftype == function_static &&
1300     	      olddecl->ftype == function_implicit)
1301     	    {
1302     	      pedwarn("`%s' was declared implicitly `extern' and later `static'", name);
1303     	      previous_message = pedwarn_with_decl;
1304     	    }
1305     	  else if (newdecl->ftype == function_static &&
1306     		   olddecl->ftype == function_normal)
1307     	    {
1308     	      pedwarn("static declaration for `%s' follows non-static", name);
1309     	      previous_message = pedwarn_with_decl;
1310     	    }
1311     	}
1312           else if (newdecl->kind == decl_variable)
1313     	{
1314     	  /* If pedantic, warn when static declaration follows a non-static
1315     	     declaration. */
1316     	  if (pedantic &&
1317     	      olddecl->isexternalscope && !newdecl->isexternalscope)
1318     	    {
1319     	      pedwarn("static declaration for `%s' follows non-static", name);
1320     	      previous_message = pedwarn_with_decl;
1321     	    }
1322     	  /* Warn when const declaration follows a non-const declaration */
1323     	  if (!type_const(oldtype) && type_const(newtype))
1324     	    warning("const declaration for `%s' follows non-const", name);
1325     	  /* These bits are logically part of the type, for variables. */
1326     	  else if (pedantic && 
1327     		   (type_const(oldtype) != type_const(newtype)
1328     		    || type_volatile(oldtype) != type_volatile(newtype)))
1329     	    {
1330     	      pedwarn("type qualifiers for `%s' conflict with previous decl",
1331     		      name);
1332     	      previous_message = pedwarn_with_decl;
1333     	    }
1334     	}
1335         }
1336     
1337       /* Optionally warn about more than one declaration for the same name.  */
1338       /* Let's try a different test than GCC */
1339       if (errmsg == 0 && warn_redundant_decls && !olddecl->islimbo &&
1340           !(olddecl->isfilescoperef && !newdecl->isfilescoperef))
1341         {
1342           warning_with_decl(newdecl->ast, "redundant redeclaration of `%s' in same scope", name);
1343           previous_message = warning_with_decl;
1344         }
1345     
1346       if (previous_message)
1347         show_previous_decl(previous_message, olddecl);
1348     
1349     
1350       /* Copy all the DECL_... slots specified in the new decl
1351          except for any that we copy here from the old type. */
1352     
1353       /* If either decl says `inline', this fn is inline,
1354          unless its definition was passed already.  */
1355       if (newdecl->isinline && !olddecl->definition)
1356         olddecl->isinline = TRUE;
1357       newdecl->isinline = olddecl->isinline;
1358     
1359       if (different_binding_level)
1360         {
1361           /* newdecl must be a reference to something at file scope */
1362           assert(newdecl->isfilescoperef && !newdecl->needsmemory);
1363           assert(!(newdecl->kind == decl_variable &&
1364     	       newdecl->vtype == variable_static));
1365           assert(!(newdecl->kind == decl_function &&
1366     	       (newdecl->ftype == function_implicit ||
1367     		newdecl->ftype == function_nested)));
1368     
1369           /* We copy some info over to the newdecl which will shadow olddecl */
1370           newdecl->shadowed = olddecl;
1371           newdecl->definition = olddecl->definition;
1372           newdecl->defined = olddecl->defined;
1373           newdecl->isinline = olddecl->isinline;
1374           newdecl->isexterninline = olddecl->isexterninline;
1375           newdecl->oldstyle_args = olddecl->oldstyle_args;
1376           if (olddecl->in_system_header)
1377     	newdecl->in_system_header = TRUE;
1378     
1379           newdecl->isexternalscope = olddecl->isexternalscope;
1380     
1381           /* We don't copy the type */
1382     
1383           return 0;
1384         }
1385     
1386       /* newdecl should be too new to have any oldstyle args yet */
1387       assert(!newdecl->oldstyle_args);
1388     
1389       /* Merge the data types specified in the two decls.  */
1390       if (types_match)
1391         olddecl->type = common_type(newtype, oldtype);
1392       else if (newtype != error_type)
1393         /* GCC keeps the old type. I think it makes more sense to keep the new
1394            one. And it means I can examine the decl in current_function_decl
1395            and find the type of the current function, not something random */
1396         olddecl->type = newtype;
1397       olddecl->islimbo = FALSE;
1398     
1399       if (newdecl->definition)
1400         {
1401           olddecl->definition = newdecl->definition;
1402           olddecl->defined = newdecl->defined;
1403         }
1404     
1405       olddecl->isexternalscope &= newdecl->isexternalscope;
1406       olddecl->isfilescoperef &= newdecl->isfilescoperef;
1407       olddecl->needsmemory |= newdecl->needsmemory;
1408       olddecl->noreturn |= newdecl->noreturn;
1409       olddecl->__init |= newdecl->__init;
1410       if (olddecl->alias == NULL)
1411         olddecl->alias = newdecl->alias;
1412       assert(newdecl->alias == NULL || !strcmp(olddecl->alias, newdecl->alias));
1413     
1414       /* For functions, static overrides non-static.  */
1415       if (newdecl->kind == decl_function)
1416         {
1417           if (olddecl->ftype != function_static)
1418     	olddecl->ftype = newdecl->ftype;
1419     
1420           /* Also set isexterninline correctly */
1421           if ((olddecl->definition && !olddecl->isexterninline) ||
1422     	  (newdecl->definition && !newdecl->isexterninline))
1423     	olddecl->isexterninline = FALSE;
1424           else if (olddecl->isexterninline || newdecl->isexterninline)
1425     	olddecl->isexterninline = TRUE;
1426           /* the last case is 2 non-inline externs, so isexterninline is correct */
1427         }
1428       else if (newdecl->kind == decl_variable)
1429         {
1430           /* static overrides extern (the combinations with register
1431     	 are errors anyway) */ 
1432           if (olddecl->vtype != variable_static)
1433     	olddecl->vtype = variable_static;
1434         }
1435     
1436       olddecl->in_system_header = newdecl->in_system_header =
1437         olddecl->in_system_header || newdecl->in_system_header;
1438     
1439       /* Point to the latest declaration */
1440       olddecl->ast = newdecl->ast;
1441     
1442       return 1;
1443     }

1548     void check_function(data_declaration dd, declaration fd, int class,
1549     		    bool inlinep, const char *name, type function_type,
1550     		    bool nested, bool isdeclaration, bool defaulted_int)
1551     {
1552       type return_type;
1553     
1554       if (defaulted_int && (warn_implicit_int || warn_return_type))
1555         warning("return-type defaults to `int'");
1556     
1557       return_type = type_function_return_type(function_type);
1558     
1559       /* XXX: Does this volatile/const stuff actually work with my imp ? */
1560       if (pedantic && type_void(return_type) &&
1561           (type_const(return_type) || type_volatile(return_type)) &&
1562           !in_system_header)
1563         pedwarn("ANSI C forbids const or volatile void function return type");
1564     
1565       if (type_volatile(function_type) && !type_void(return_type))
1566         warning("`noreturn' function returns non-void value");
1567     
1568       /* Record presence of `inline', if it is reasonable.  */
1569       if (inlinep && !strcmp(name, "main") && !nested)
1570         {
1571           warning("cannot inline function `main'");
1572           inlinep = FALSE;
1573         }
1574     
1575       /* Warn for unlikely, improbable, or stupid declarations of `main'. */
1576       if (warn_main && !strcmp("main", name) && !nested)
1577         {
1578           if (!type_equal_unqualified(return_type, int_type))
1579     	pedwarn("return type of `%s' is not `int'", name);
1580     
1581           /* Just being "bug"-compatible w/ GCC here */
1582           if (!type_function_oldstyle(function_type))
1583     	{
1584     	  typelist_scanner scanargs;
1585     	  type argtype;
1586     	  int argct = 0;
1587     
1588     	  typelist_scan(type_function_arguments(function_type), &scanargs);
1589     	  while ((argtype = typelist_next(&scanargs)))
1590     	    {
1591     	      ++argct;
1592     	      switch (argct)
1593     		{
1594     		case 1:
1595     		  if (!type_equal_unqualified(argtype, int_type))
1596     		    pedwarn("first argument of `%s' should be `int'", name);
1597     		  break;
1598     
1599     		case 2:
1600     		  if (!is_doublecharstar(argtype))
1601     		    pedwarn("second argument of `%s' should be `char **'",
1602     			    name);
1603     		  break;
1604     
1605     		case 3:
1606     		  if (!is_doublecharstar(argtype))
1607     		    pedwarn("third argument of `%s' should probably be `char **'",
1608     			    name);
1609     		  break;
1610     		}
1611     	    }
1612     
1613     	  /* It is intentional that this message does not mention the third
1614     	     argument, which is warned for only pedantically, because it's
1615     	     blessed by mention in an appendix of the standard. */
1616     	  if (argct > 0 && (argct < 2 || argct > 3))
1617     	    pedwarn("`%s' takes only zero or two arguments", name);
1618     
1619     	  if (argct == 3 && pedantic)
1620     	    pedwarn("third argument of `%s' is deprecated", name);
1621     
1622     	  if (class == RID_STATIC)
1623     	    pedwarn("`%s' is normally a non-static function", name);
1624     	}
1625         }
1626     
1627       init_data_declaration(dd, fd, name, function_type);
1628       dd->kind = decl_function;
1629       dd->isexternalscope = FALSE;
1630       if (nested)
1631         dd->ftype = function_nested;
1632       else if (class == RID_STATIC)
1633         dd->ftype = function_static;
1634       else
1635         {
1636           dd->ftype = function_normal;
1637           dd->isexternalscope = TRUE;
1638         }
1639       /* XXX: Should probably be FALSE for extern inline */
1640       dd->needsmemory = !isdeclaration;
1641       dd->isinline = inlinep;
1642       dd->isexterninline = inlinep && class == RID_EXTERN;
1643       dd->isfilescoperef = dd->isexterninline || isdeclaration;
1644     }

1759     /* Start definition of function 'elements d' with attributes attribs.
1760        nested is true for nested function definitions.
1761        Returns false in case of error.
1762        Sets current_function_decl to the declaration for this function */
1763     bool start_function(type_element elements, declarator d, attribute attribs,
1764     		    bool nested)
1765     {
1766       int class;
1767       bool inlinep;
1768       const char *name;
1769       type function_type;
1770       bool defaulted_int, old_decl_has_prototype, normal_function;
1771       data_declaration old_decl, ddecl;
1772       function_decl fdecl;
1773       function_declarator fdeclarator;
1774       struct data_declaration tempdecl;
1775       env_scanner scan;
1776       const char *id;
1777       void *idval;
1778     
1779       if (!nested)
1780         assert(current_env == global_env && current_function_decl == NULL);
1781     
1782       parse_declarator(elements, d, FALSE, &class, &inlinep, &name, &function_type,
1783     		   &defaulted_int, &fdeclarator);
1784     
1785       if (!type_function(function_type))
1786         return FALSE;
1787     
1788     
1789       /* We don't set current_function_decl yet so that error messages do not
1790          say "In function <thisfunctioname>" as we're not "in" the function yet */
1791       fdecl = new_function_decl(parse_region, d->loc, d, elements, attribs,
1792     			    NULL, NULL, current_function_decl, NULL);
1793       fdecl->declared_type = function_type;
1794       fdecl->undeclared_variables = new_env(parse_region, NULL);
1795       fdecl->current_loop = NULL;
1796     
1797       if (class == RID_AUTO)
1798         {
1799           if (pedantic || !nested)
1800     	pedwarn("function definition declared `auto'");
1801           class = 0;
1802         }
1803       else if (class == RID_REGISTER)
1804         {
1805           error("function definition declared `register'");
1806           class = 0;
1807         }
1808       else if (class == RID_TYPEDEF)
1809         {
1810           error("function definition declared `typedef'");
1811           class = 0;
1812         }
1813       else if (class == RID_EXTERN && nested)
1814         {
1815           error("nested function `%s' declared `extern'", name);
1816           class = 0;
1817         }
1818       else if ((class == RID_STATIC || inlinep) && nested)
1819         {
1820           if (pedantic)
1821     	pedwarn("invalid storage class for function `%s'", name);
1822           class = 0;
1823         }
1824     
1825       if (!type_void(type_function_return_type(function_type)) &&
1826           type_incomplete(type_function_return_type(function_type)))
1827         {
1828           error("return-type is an incomplete type");
1829           function_type = qualify_type1
1830     	(make_function_type(void_type,
1831     			    type_function_arguments(function_type),
1832     			    type_function_varargs(function_type),
1833     			    type_function_varargs_quals(function_type),
1834     			    type_function_varargs_user_quals(function_type),
1835     			    type_function_oldstyle(function_type)),
1836     	 function_type);
1837         }
1838     
1839       old_decl = lookup_id(name, TRUE);
1840       old_decl_has_prototype = old_decl && old_decl->kind == decl_function &&
1841         !type_function_oldstyle(old_decl->type);
1842     
1843       normal_function = !nested && class != RID_STATIC;
1844       /* Optionally warn of old-fashioned def with no previous prototype.  */
1845       if (warn_strict_prototypes
1846           && type_function_oldstyle(function_type)
1847           && !old_decl_has_prototype)
1848         warning("function declaration isn't a prototype");
1849       /* Optionally warn of any global def with no previous prototype.  */
1850       else if (warn_missing_prototypes
1851     	   && normal_function && !old_decl_has_prototype
1852     	   && strcmp("main", name))
1853         warning("no previous prototype for `%s'", name);
1854       /* Optionally warn of any def with no previous prototype
1855          if the function has already been used.  */
1856       else if (warn_missing_prototypes
1857     	   && old_decl && old_decl->ftype == function_implicit)
1858         warning("`%s' was used with no prototype before its definition", name);
1859       /* Optionally warn of any global def with no previous declaration.  */
1860       else if (warn_missing_declarations
1861     	   && normal_function && !old_decl && strcmp("main", name))
1862         warning("no previous declaration for `%s'", name);
1863       /* Optionally warn of any def with no previous declaration
1864          if the function has already been used.  */
1865       else if (warn_missing_declarations
1866     	   && old_decl && old_decl->ftype == function_implicit)
1867         warning("`%s' was used with no declaration before its definition", name);
1868     
1869       check_function(&tempdecl, CAST(declaration, fdecl), class,
1870     		 inlinep, name, function_type, nested, FALSE, defaulted_int);
1871       tempdecl.definition = tempdecl.ast;
1872       tempdecl.defined = TRUE;
1873     
1874       /* If return types match and old declaraton has a prototype and new
1875          declaration hasn't, borrow the prototype. We will check for errors
1876          in store_parm_decls. */
1877       if (old_decl_has_prototype && type_function_oldstyle(function_type) &&
1878           type_compatible(type_function_return_type(old_decl->type),
1879     		      type_function_return_type(function_type)))
1880         {
1881           function_type = qualify_type1
1882     	(make_function_type(type_function_return_type(function_type),
1883     			    type_function_arguments(old_decl->type),
1884     			    type_function_varargs(old_decl->type),
1885     			    type_function_varargs_quals(old_decl->type),
1886     			    type_function_varargs_user_quals(old_decl->type),
1887     			    FALSE), old_decl->type);
1888     
1889           tempdecl.type = function_type;
1890         }
1891     
1892       if (old_decl && duplicate_decls(&tempdecl, old_decl, FALSE, FALSE))
1893         ddecl = old_decl;
1894       else
1895         ddecl = declare(current_env, &tempdecl, FALSE);
1896     
1897       fdecl->base_labels = fdecl->scoped_labels =
1898         new_env(parse_region,
1899     	    current_function_decl ? current_function_decl->scoped_labels : NULL);
1900       fdecl->ddecl = ddecl;
1901       fdecl->fdeclarator = fdeclarator;
1902     
1903       current_env = fdeclarator->env;
1904       current_env->fdecl = current_function_decl = fdecl;
1905     
1906       /* Set id of parameters (done here rather than in declare because parameters
1907          of a given function may be declared several times) */
1908       env_scan(current_env->id_env, &scan);
1909       while (env_next(&scan, &id, &idval))
1910         {
1911           data_declaration iddecl = idval;
1912     
1913           if (iddecl->kind == decl_variable)
1914     	{
1915     	  assert(iddecl->isparameter);
1916     	  /*iddecl->id = current_function_decl->nlocals++;*/
1917     	}
1918         }
1919     
1920       handle_attributes(attribs, ddecl, NULL, NULL);
1921     
1922       return TRUE;
1923     }

1944        declarator, or from old_parms */
1945     void store_parm_decls(declaration old_parms)
1946     {
1947       if (!oldstyle_function(current_function_decl))
1948         {
1949           /* This case is when the function was defined with an ANSI prototype.
1950     	 The parms already have decls, so we need not do anything here
1951     	 except record them as in effect
1952     	 and complain if any redundant old-style parm decls were written.  */
1953           if (old_parms)
1954     	error_with_decl(CAST(declaration, current_function_decl),
1955     			"parm types given both in parmlist and separately");
1956         }
1957       else
1958         {
1959           oldidentifier_decl parm, parms;
1960     
1961           current_function_decl->old_parms = old_parms;
1962           /* Need to either:
1963     	 - compare arg types to previous prototype
1964     	 - or build pseudo-prototype for this function
1965           */
1966           parms = CAST(oldidentifier_decl,
1967     		   current_function_decl->fdeclarator->parms);
1968           scan_oldidentifier_decl (parm, parms)
1969     	/* If no declaration given, default to int.  */
1970     	if (parm->ddecl->type == void_type)
1971     	  {
1972     	    parm->ddecl->type = int_type;
1973     	    if (extra_warnings)
1974     	      warning_with_decl(CAST(declaration, parm),
1975     				"type of `%s' defaults to `int'",
1976     				parm->cstring.data);
1977     	  }
1978         }
1979     
1980       /* Declare __FUNCTION__ and __PRETTY_FUNCTION__ for this function.  */
1981       declare_function_name();
1982     }

2094     /* Start definition of variable 'elements d' with attributes
2095        extra_attributes and attributes, asm specification astmt.
2096        If initialised is true, the variable has an initialiser.

2100     		       bool initialised, attribute extra_attributes,
2101     		       attribute attributes)
2102     {

2104         new_variable_decl(parse_region, d->loc, d, extra_attributes, NULL,
2105     		      astmt, NULL);
2106       struct data_declaration tempdecl;
2107       data_declaration ddecl = NULL, old_decl;
2108     
2109       if (current_env->parm_level)
2110         {
2111           check_parameter(&tempdecl, elements, vd);
2112     
2113           if (type_void(tempdecl.type))
2114     	{
2115     	  error("parameter `%s' declared void", tempdecl.name);
2116     	  tempdecl.type = int_type;
2117     	}
2118     
2119           /* Update environment for old-style declarations only if function
2120     	 doesn't have a prototype. We will report an error message later
2121     	 for arguments specified for functions with prototypes. */
2122           if (oldstyle_function(current_function_decl))
2123     	{
2124     	  /* Traditionally, a parm declared float is actually a double.  */
2125     	  if (flag_traditional &&
2126     	      type_equal_unqualified(tempdecl.type, float_type))
2127     	    tempdecl.type = qualify_type1(double_type, tempdecl.type);
2128     
2129     	  old_decl = lookup_id(tempdecl.name, TRUE);
2130     
2131     	  if (old_decl && duplicate_decls(&tempdecl, old_decl, FALSE, FALSE))
2132     	    {

2134     		 of a forward parm decl.  */

2136     	      ddecl->type = tempdecl.type;
2137     	      ddecl->ast = CAST(declaration, vd);
2138     	      ddecl->isused = TRUE;
2139     	    }
2140     	  else
2141     	    {
2142     	      error("declaration for parameter `%s' but no such parameter",
2143     		    tempdecl.name);
2144     	      ddecl = declare(current_env, &tempdecl, FALSE);
2145     	    }
2146     	}
2147           else
2148     	/* Make a dummy decl to keep everyone happy */
2149     	ddecl = declare(current_env, &tempdecl, FALSE);
2150     
2151           if (initialised)
2152     	error("parameter `%s' is initialized",
2153     	      vd->ddecl->name ? vd->ddecl->name : "type name");
2154         }
2155       else
2156         {
2157           int class;
2158           bool inlinep;
2159           const char *name, *printname;
2160           bool defaulted_int;
2161           type var_type;
2162           bool different_binding_level = FALSE;
2163     
2164           parse_declarator(elements, d, FALSE, &class, &inlinep, &name, &var_type,
2165     		       &defaulted_int, NULL);
2166           vd->declared_type = var_type;
2167           printname = name ? name : "type name";
2168     
2169           if (warn_implicit_int && defaulted_int && !type_function(var_type))
2170     	warning("type defaults to `int' in declaration of `%s'", printname);
2171     
2172           init_data_declaration(&tempdecl, CAST(declaration, vd), name, var_type);
2173     
2174           /* The fun begins */
2175     
2176           /* `extern' with initialization is invalid if not at top level.  */
2177           if (class == RID_EXTERN && initialised)
2178     	{
2179     	  if (current_env == global_env)
2180     	    warning("`%s' initialized and declared `extern'", printname);
2181     	  else
2182     	    error("`%s' has both `extern' and initializer", printname);
2183     	}
2184     
2185           if (class == RID_AUTO && current_env == global_env)
2186     	{
2187     	  error("top-level declaration of `%s' specifies `auto'", printname);
2188     	  class = 0;
2189     	}
2190     
2191           if (class == RID_TYPEDEF)
2192     	{
2193     	  /* typedef foo = bar  means give foo the same type as bar.
2194     	     We haven't parsed bar yet, so `finish_decl' will fix that up.
2195     	     Any other case of an initialization in a TYPE_DECL is an error. */
2196     	  if (initialised && (pedantic || elements->next))
2197     	    error("typedef `%s' is initialized", printname);
2198     
2199     	  tempdecl.kind = decl_typedef;
2200     	  tempdecl.definition = tempdecl.ast;
2201     	  tempdecl.defined = TRUE;
2202     	  tempdecl.isexternalscope = FALSE;
2203     	  tempdecl.isfilescoperef = FALSE;
2204     	  tempdecl.needsmemory = FALSE;
2205     	}
2206           else if (type_function(var_type))
2207     	{
2208     	  bool nested = current_env != global_env && class == RID_AUTO;
2209     
2210     	  if (initialised)
2211     	    error("function `%s' is initialized like a variable",
2212     		  printname);
2213     
2214     	  if (class == RID_AUTO && pedantic)
2215     	    pedwarn("invalid storage class for function `%s'", name);
2216     	  if (class == RID_REGISTER)
2217     	    {
2218     	      error("invalid storage class for function `%s'", name);
2219     	      class = 0;
2220     	    }
2221     	  /* Function declaration not at top level.
2222     	     Storage classes other than `extern' are not allowed
2223     	     and `extern' makes no difference.  */
2224     	  if (current_env != global_env && pedantic
2225     	      && (class == RID_STATIC || class == RID_INLINE))
2226     	    pedwarn("invalid storage class for function `%s'", name);
2227     
2228     	  check_function(&tempdecl, CAST(declaration, vd), class, inlinep,
2229     			 name, var_type, nested, TRUE, defaulted_int);
2230     	}
2231           else
2232     	{
2233     	  int extern_ref = !initialised && class == RID_EXTERN;
2234     
2235     	  if (type_void(var_type) &&
2236     	      !(class == RID_EXTERN ||
2237     		(current_env == global_env &&
2238     		 !(class == RID_STATIC || class == RID_REGISTER))))
2239     	    {
2240     	      error("variable `%s' declared void", printname);
2241     	      var_type = int_type;
2242     	    }
2243     
2244     	  /* It's a variable.  */
2245     	  if (inlinep)
2246     	    pedwarn("variable `%s' declared `inline'", printname);
2247     #if 0
2248     	  /* Don't allow initializations for incomplete types
2249     	     except for arrays which might be completed by the initialization.  */
2250     	  if (TYPE_SIZE (TREE_TYPE (decl)) != 0)
2251     	    {
2252     	      /* A complete type is ok if size is fixed.  */
2253     
2254     	      if (TREE_CODE (TYPE_SIZE (TREE_TYPE (decl))) != INTEGER_CST
2255     		  || C_DECL_VARIABLE_SIZE (decl))
2256     		{
2257     		  error ("variable-sized object may not be initialized");
2258     		  initialised = 0;
2259     		}
2260     	    }
2261     	  else if (TREE_CODE (TREE_TYPE (decl)) != ARRAY_TYPE)
2262     	    {
2263     	      error ("variable `%s' has initializer but incomplete type",
2264     		     IDENTIFIER_POINTER (DECL_NAME (decl)));

2266     	    }

2270     		     IDENTIFIER_POINTER (DECL_NAME (decl)));

2272     	    }
2273     #endif
2274     	  tempdecl.kind = decl_variable;
2275     	  tempdecl.vtype =
2276     	    class == RID_REGISTER ? variable_register :
2277     	    class == RID_STATIC ? variable_static :
2278     	    variable_normal;
2279     	  tempdecl.isfilescoperef = extern_ref;
2280     	  if (!extern_ref)
2281     	    {
2282     	      tempdecl.definition = tempdecl.ast;
2283     	      tempdecl.defined = TRUE;
2284     	    }
2285     	  if (current_env == global_env)
2286     	    {
2287     	      tempdecl.isexternalscope =
2288     		class != RID_STATIC && class != RID_REGISTER;
2289     	      tempdecl.needsmemory = !extern_ref;
2290     	      tempdecl.islocal = FALSE;
2291     	    }
2292     	  else
2293     	    {
2294     	      tempdecl.isexternalscope = extern_ref;
2295     	      tempdecl.needsmemory = class == RID_STATIC;
2296     	      tempdecl.islocal = !(extern_ref || class == RID_STATIC);
2297     	    }
2298     	}
2299     
2300           if (warn_nested_externs && tempdecl.isfilescoperef &&
2301     	  current_env != global_env && !tempdecl.in_system_header)
2302     	warning("nested extern declaration of `%s'", printname);
2303     
2304           old_decl = lookup_id(name, TRUE);
2305           /* Check the global environment if declaring something with file
2306     	 scope */
2307           if (!old_decl && tempdecl.isfilescoperef)
2308     	{
2309     	  old_decl = lookup_global_id(name);
2310     	  /* global typedefs don't count */
2311     	  if (old_decl && old_decl->kind == decl_typedef)
2312     	    old_decl = NULL;
2313     	  if (old_decl)
2314     	    different_binding_level = TRUE;
2315     	}
2316     
2317           if (old_decl &&
2318     	  duplicate_decls(&tempdecl, old_decl, different_binding_level, initialised))
2319     	ddecl = old_decl;
2320           else
2321     	ddecl = declare(current_env, &tempdecl, FALSE);
2322         }
2323       assert(ddecl);
2324       vd->ddecl = ddecl;
2325     
2326       handle_attributes(extra_attributes, ddecl, NULL, NULL);
2327       handle_attributes(attributes, ddecl, NULL, NULL);
2328     
2329       return CAST(declaration, vd);
2330     }

2518     	  long long bitwidth;

2522     			   field->arg1 != NULL, &class, &inlinep, &name,
2523     			   &tmpft, &defaulted_int, NULL);
2524     	  field_type = tmpft;
2525     
2526     	  printname = name ? name : "(anonymous)";
2527     
2528     	  if (name)
2529     	    hasmembers = TRUE;
2530     
2531     	  if (!tdecl->fields_const)
2532     	    {
2533     	      base_field_type = type_base(field_type);
2534     	      if (type_const(base_field_type) ||
2535     		  ((type_struct(base_field_type) || type_union(base_field_type)) &&
2536     		   type_tag(base_field_type)->fields_const))
2537     		tdecl->fields_const = TRUE;
2538     	    }

2540     	  /* XXX: Surely we should do the same as for const here ? */
2541     	  if (type_volatile(field_type))
2542     	    tdecl->fields_volatile = TRUE;

2552     	      field_type = error_type;
2553     	    }
2554     	  else if (type_incomplete(field_type) && field->next) 
2555     	    {

2558     	    }

2576     	  handle_attributes(flist->attributes, NULL, fdecl, NULL);
2577     	  field_type = fdecl->type; /* attributes might change type */

2580     	  if (field->arg1)
2581     	    {

2584     	      if (!type_integer(field_type))

2586     	      else if (!(type_integer(field->arg1->type) && constant_integral(cwidth)))

2592     
2593     		  if (pedantic)
2594     		    constant_overflow_warning(cwidth);

2604     		  else if (width == 0 && name)
2605     		    error_with_location(floc, "zero width for bit-field `%s'", name);
2606     		  else
2607     		    bitwidth = width;
2608     		}
2609     
2610     #if 0
2611     	      if (DECL_INITIAL (x) && pedantic
2612     		  && TYPE_MAIN_VARIANT (TREE_TYPE (x)) != integer_type_node
2613     		  && TYPE_MAIN_VARIANT (TREE_TYPE (x)) != unsigned_type_node

2619     
2620     
2621     #if 0
2622     	  if (TREE_CODE (TREE_TYPE (x)) == ENUMERAL_TYPE
2623     	      && (width < min_precision (TYPE_MIN_VALUE (TREE_TYPE (x)),
2624     					 TREE_UNSIGNED (TREE_TYPE (x)))
2625     		  || width < min_precision (TYPE_MAX_VALUE (TREE_TYPE (x)),
2626     					    TREE_UNSIGNED (TREE_TYPE (x)))))
2627     	    warning_with_decl (x, "`%s' is narrower than values of its type");
2628     #endif
2629     
2630     #endif
2631     	    }
2632     
2633     	  /* ASSUME: for bitfields we're assuming that:
2634     	     - alignment of record is forced to be at least that of the base
2635     	       (except for width 0 which just forces alignment of the current offset
2636     	       with the base type's alignment)
2637     	     - bit fields cannot span more units of alignment of their type
2638     	       than the type itself
2639     	     This is gcc's behaviour when PCC_BITFIELD_TYPE_MATTERS is defined */
2640

2642     	  fdecl->offset_cc = size_cc || isunion;
2643     
2644     	  if (type_size_cc(field_type))
2645     	    fsize = type_size(field_type) * BITSPERBYTE;
2646     	  else
2647     	    {
2648     	      size_cc = FALSE;
2649     	      fsize = BITSPERBYTE; /* actual value is unimportant */
2650     	    }
2651     	  /* don't care about alignment if no size (type_incomplete(field_type)
2652     	     is true, so we got an error above */
2653     	  falign = (type_has_size(field_type) ? type_alignment(field_type) : 1)
2654     	    * BITSPERBYTE;
2655     
2656     	  if (bitwidth == 0)
2657     	    {
2658     	      offset = align_to(offset, falign);
2659     	      falign = 1; /* No structure alignment implications */
2660     	    }
2661     	  else if (bitwidth > 0)
2662     	    {
2663     	      if (((offset + bitwidth + falign - 1) / falign - offset / falign)
2664     		  > fsize / falign)
2665     		offset = align_to(offset, falign);
2666     	    }
2667     	  else /* regular field */
2668     	    {
2669     	      offset = align_to(offset, falign); 
2670     	      bitwidth = fsize;
2671     	    }
2672     	  fdecl->offset = offset;
2673     
2674     	  if (!isunion)
2675     	    {
2676     	      offset += bitwidth;
2677     	      size = offset;
2678     	    }
2679     	  else
2680     	    size = bitwidth > size ? bitwidth : size;
2681     
2682     	  alignment = lcm(alignment, falign); /* Note: GCC uses max. Must be hoping for powers of 2 */
2683     	}
2684         }
2685     
2686       if (pedantic && !hasmembers)
2687         pedwarn("%s has no %smembers",
2688     	    (s->kind == kind_union_ref ? "union" : "structure"),
2689     	    (fields ? "named " : ""));
2690     
2691       /* ASSUME: see previous assume */
2692       tdecl->size = align_to(size, alignment) / BITSPERBYTE;
2693       tdecl->alignment = alignment / BITSPERBYTE;
2694       tdecl->size_cc = size_cc;
2695     
2696       return t;
2697     }

2739     /* Finish definition of enum furnishing the names and attribs.
2740        Returns t */
2741     type_element finish_enum(type_element t, declaration names,
2742     			 attribute attribs)
2743     {
2744       tag_ref s = CAST(tag_ref, t);
2745       tag_declaration tdecl = s->tdecl;
2746       cval smallest, largest;
2747       bool enum_isunsigned;
2748       type type_smallest, type_largest, enum_reptype;
2749       enumerator v, values = CAST(enumerator, names);
2750     
2751       s->fields = names;
2752       s->attributes = attribs;
2753       handle_attributes(attribs, NULL, NULL, tdecl);
2754       tdecl->fields = 0;
2755       tdecl->defined = TRUE;
2756       tdecl->being_defined = FALSE;
2757     
2758       /* Pick a representation type for this enum. */
2759     
2760       /* First, find largest and smallest values defined in this enum. */
2761       if (!names)
2762         smallest = largest = cval_zero;
2763       else
2764         {
2765           smallest = largest = value_of_enumerator(values);
2766     
2767           scan_enumerator (v, CAST(enumerator, values->next))
2768     	{
2769     	  cval vv = value_of_enumerator(v);
2770     
2771     	  if (cval_intcompare(vv, largest) > 0)
2772     	    largest = vv;
2773     	  if (cval_intcompare(vv, smallest) < 0)
2774     	    smallest = vv;
2775     	}
2776         }
2777     
2778       /* Pick a type that will hold the smallest and largest values. */
2779       enum_isunsigned = cval_intcompare(smallest, cval_zero) >= 0;
2780       type_smallest = type_for_cval(smallest, enum_isunsigned);
2781       type_largest = type_for_cval(largest, enum_isunsigned);
2782       assert(type_smallest);
2783       if (!type_largest)
2784         {
2785           assert(!enum_isunsigned);
2786           warning("enumeration values exceed range of largest integer");
2787           type_largest = long_long_type;
2788         }
2789       if (type_size(type_smallest) > type_size(type_largest))
2790         enum_reptype = type_smallest;
2791       else
2792         enum_reptype = type_largest;
2793     
2794       /* int is the smallest possible type for an enum (except if the 
2795          enum has the packed attribute) */
2796       if (!tdecl->packed && type_size(enum_reptype) < type_size(int_type))
2797         enum_reptype = int_type;
2798     
2799       tdecl->reptype = enum_reptype;
2800       tdecl->size_cc = TRUE;
2801       tdecl->size = type_size(enum_reptype);
2802       tdecl->alignment = type_alignment(enum_reptype);
2803     
2804       /* Change type of all enum constants to enum_reptype */
2805       scan_enumerator (v, values)
2806         {
2807           known_cst val = v->ddecl->value;
2808     
2809           val->type = enum_reptype;
2810           val->cval = cval_cast(val->cval, enum_reptype);
2811         }
2812     
2813       return t;
2814     }

2914     /* Save the rest of the line in the directives list. Return '\n' */
2915     int save_directive(char *directive)
2916     {
2917       static char *directive_buffer = NULL;
2918       static unsigned buffer_length = 0;
2919       char *p;
2920       char *buffer_limit;
2921       int looking_for = 0;
2922       int char_escaped = 0;
2923     
2924       if (buffer_length == 0)
2925         {
2926           directive_buffer = (char *)xmalloc (128);
2927           buffer_length = 128;
2928         }
2929     
2930       buffer_limit = &directive_buffer[buffer_length];
2931     
2932       for (p = directive_buffer; ; )
2933         {
2934           int c;
2935     
2936           /* Make buffer bigger if it is full.  */
2937           if (p >= buffer_limit)
2938             {
2939     	  unsigned bytes_used = (p - directive_buffer);
2940     
2941     	  buffer_length *= 2;
2942     	  directive_buffer
2943     	    = (char *)xrealloc (directive_buffer, buffer_length);
2944     	  p = &directive_buffer[bytes_used];
2945     	  buffer_limit = &directive_buffer[buffer_length];
2946             }
2947     
2948           c = getc (finput);
2949     
2950           /* Discard initial whitespace.  */
2951           if ((c == ' ' || c == '\t') && p == directive_buffer)
2952     	continue;
2953     
2954           /* Detect the end of the directive.  */
2955           if (looking_for == 0
2956     	  && (c == '\n' || c == EOF))
2957     	{
2958               ungetc (c, finput);
2959     	  c = '\0';
2960     	}
2961     
2962           *p++ = c;
2963     
2964           if (c == 0)
2965     	break;
2966     
2967           /* Handle string and character constant syntax.  */
2968           if (looking_for)
2969     	{
2970     	  if (looking_for == c && !char_escaped)
2971     	    looking_for = 0;	/* Found terminator... stop looking.  */
2972     	}
2973           else
2974             if (c == '\'' || c == '"')
2975     	  looking_for = c;	/* Don't stop buffering until we see another
2976     				   another one of these (or an EOF).  */
2977     
2978           /* Handle backslash.  */
2979           char_escaped = (c == '\\' && ! char_escaped);
2980         }
2981       return '\n';
2982     }

3015     /* If statement list l1 ends with an unfinished label, attach l2 to that
3016        label. Otherwise attach l2 to the end of l1 */
3017     statement chain_with_labels(statement l1, statement l2)
3018     {
3019       node last, last_label;
3020     
3021       if (!l1) return l2;
3022       if (!l2) return l1;
3023     
3024       last_label = last = last_node(CAST(node, l1));
3025       /* There may be an unfinished sub-label due to 'a: b:' */
3026       while (last_label->kind == kind_labeled_stmt)
3027         {
3028           labeled_stmt ls = CAST(labeled_stmt, last_label);
3029     
3030           if (!ls->stmt) /* An unfinished labeled statement */
3031     	{
3032     	  ls->stmt = l2;
