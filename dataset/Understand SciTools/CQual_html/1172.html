23     bool oldstyle_function(function_decl fn)
24     {
25       return !fn->fdeclarator->parms || is_oldidentifier_decl(fn->fdeclarator->parms);
26     }

28     function_declarator get_fdeclarator(declarator d)
29     {
30       for (;;)
31         switch (d->kind)
32           {
33           case kind_function_declarator:
34     	return CAST(function_declarator, d);
35           case kind_identifier_declarator:
36     	return NULL;
37           case kind_array_declarator:
38     	d = CAST(array_declarator, d)->declarator;
39     	break;
40           case kind_pointer_declarator:
41     	d = CAST(pointer_declarator, d)->declarator;
42     	break;
43           default:
44     	assert(0);
45           }
46     }

48     data_declaration get_parameter(declaration d)
49     {
50       switch (d->kind)
51         {
52         case kind_ellipsis_decl:
53           return NULL;
54         case kind_oldidentifier_decl:
55           return CAST(oldidentifier_decl, d)->ddecl;
56         case kind_data_decl:
57           return CAST(variable_decl, CAST(data_decl, d)->decls)->ddecl;
58         default:
59           assert(0);
60           return NULL;
61         }
62     }

64     data_declaration base_identifier(data_declaration d)
65     {
66       while (d->isfilescoperef && d->shadowed)
67         d = d->shadowed;
68     
69       return d;
70     }

72     bool same_function(data_declaration d1, data_declaration d2)
73     {
74       d1 = base_identifier(d1);
75       d2 = base_identifier(d2);
76       return d1->kind == decl_function && d1 == d2;
77     }

79     bool call_to(data_declaration fnd, function_call fce)
80     {
81       return is_identifier(fce->arg1) &&
82         fnd && same_function(CAST(identifier, fce->arg1)->ddecl, fnd);
83     }

85     bool is_localvar(expression e)
86     {
87       data_declaration decl;
88     
89       if (!is_identifier(e))
90         return FALSE;
91     
92       decl = CAST(identifier, e)->ddecl;
93       return decl->kind == decl_variable && decl->islocal;
94     }

96     bool is_assignment(void *e)
97     {
98       int kind = ((node)e)->kind;
99     
100       /* XXX: Yuck */
101       return kind == kind_assign || kind == kind_plus_assign ||
102         kind == kind_minus_assign || kind == kind_times_assign ||
103         kind == kind_divide_assign || kind == kind_modulo_assign ||
104         kind == kind_lshift_assign || kind == kind_rshift_assign ||
105         kind == kind_bitand_assign || kind == kind_bitor_assign ||
106         kind == kind_bitxor_assign;
107     }

118     expression ignore_fields(expression e)
119     {
120       while (e->kind == kind_field_ref)
121         e = CAST(field_ref, e)->arg1;
122     
123       return e;

126     expression expression_of_stmt(compound_expr ce)
127     {
128       compound_stmt blk = CAST(compound_stmt, ce->stmt);
129       statement last_stmt = CAST(statement, last_node(CAST(node, blk->stmts)));
130     
131       if (last_stmt && is_expression_stmt(last_stmt))
132         {
133           expression_stmt es = CAST(expression_stmt, last_stmt);
134     
135           return es->arg1;
136         }
137       return NULL;
138     }

140     bool zero_expression(expression e)
141     {
142       /* a = 0 is not a constant expression but it's value is known to be 0,
143          and other similar cases */
144       for (;;)
145         {
146           if (is_assign(e))
147     	{
148     	  e = CAST(assign, e)->arg2;
149     	  continue;
150     	}
151           if (is_cast(e))
152     	{
153     	  e = CAST(cast, e)->arg1;
154     	  continue;
155     	}
156           if (is_comma(e))
157     	{
158     	  e = CAST(expression, last_node(CAST(node, CAST(comma, e)->arg1)));
159     	  continue;
160     	}
161           break;
162         }
163     
164       return definite_zero(e);
165     }

167     expression build_int_constant(region r, location loc, type t, largest_int c)
168     {
169       char cstbuf[64];
170       cstring csts;
171       lexical_cst cst;
172     
173       snprintf(cstbuf, sizeof cstbuf, "%lld", c);
174       csts.data = rstrdup(r, cstbuf);
175       csts.length = strlen(cstbuf);
176       cst = new_lexical_cst(r, loc, csts);
177       cst->type = t;
178       cst->cst = make_cst(make_cval_signed(c, t), t);
179     
180       return CAST(expression, cst);
181     }

204     expression build_identifier(region r, location loc, data_declaration id)
205     {
206       identifier e = new_identifier(r, loc, str2cstring(r, id->name), id);
207     
208       assert(id->kind == decl_variable || id->kind == decl_function);
209       e->type = id->type;
210       e->cst = fold_identifier(CAST(expression, e), id);
211     
212       return CAST(expression, e);
213     }

215     /* Convert an AST string into a char * */
216     char *string_to_charp(region r, string s)
217     {
218       expression e;
219       int len = 0;
220       char *str;
221     
222       scan_expression(e, s->strings)
223         if (e->kind == kind_string_cst)
224           len += CAST(string_cst, e)->cstring.length - 2; /* for begin/end "s */
225         else if (e->kind == kind_identifier)
226           len += CAST(identifier, e)->cstring.length - 2;
227         else
228           fail("Unexpected string component kind %x\n", e->kind);
229       str = rarrayalloc(r, len + 1, char);
230       str[0] = '\0';
231       scan_expression(e, s->strings)
232         if (e->kind == kind_string_cst)
233           strncat(str,
234     	      CAST(string_cst, e)->cstring.data + 1,
235     	      CAST(string_cst, e)->cstring.length - 2);
236         else if (e->kind == kind_identifier)
237           strncat(str,
238     	      CAST(identifier, e)->cstring.data + 1,
239     	      CAST(identifier, e)->cstring.length - 2);
240         else
241           fail("Unexpected string component kind %x\n", e->kind);
242       return str;
243     }
