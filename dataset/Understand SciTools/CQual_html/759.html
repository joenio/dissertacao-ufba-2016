19     #ifndef CSTRING_H
20     #define CSTRING_H
21     
22     #include "linkage.h"
23     #include "regions.h"
24     
25     EXTERN_C_BEGIN
26     
27     /* A C string is like a regular C string, but with a length. A null byte is
28        added AFTER the string for convenience */
29     typedef struct {
30       char *data;
31       int length;
32     } cstring;
33     
34     /* Make a new cstring with a copy of s, length l */
35     cstring make_cstring(region r, const char *s, int l);
36     
37     /* Make a new cstring with a copy of regular C string s */
38     cstring str2cstring(region r, const char *s);
39     
40     EXTERN_C_END
41     
42     #endif
