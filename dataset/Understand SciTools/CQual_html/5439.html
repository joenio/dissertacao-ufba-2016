19     #ifndef SET_H
20     #define SET_H
21     
22     #include "bool.h"
23     #include "linkage.h"
24     #include "regions.h"
25     
26     EXTERN_C_BEGIN
27     
28     typedef struct Set *set;
29     typedef set set_scanner;
30     
31     /* Fn for comparing set elements.  Should return 0 if e1 = e2, <0 if
32        e1 < e2, and >0 if e1 > e2 */
33     typedef int (*set_cmp_fn)(void *e1, void *e2);
34     
35     /* The empty set */
36     extern set empty_set;
37     
38     /* Return a copy of s in r */
39     set set_copy(region r, set s);
40     
41     /* Return TRUE iff s is empty */
42     bool set_empty(set s);
43     
44     /* Return TRUE if elt in s */
45     bool set_member(set_cmp_fn cmp, set s, void *elt);
46     
47     /* Return the number of elements in s */
48     int set_size(set s);
49     
50     /* Add elt to s. */
51     void set_insert(region r, set_cmp_fn cmp, bool bag, set *s, void *elt);
52     
53     /* Remove all copies of elt from s. */
54     void set_remove(set_cmp_fn cmp, bool bag, set *s, void *elt);
55     
56     /* Return TRUE if s1 <= s2 */
57     bool set_subset(set_cmp_fn cmp, set s1, set s2);
58     
59     /* Return the union of s1 and s2.  Destructive operation */
60     set set_union(set_cmp_fn cmp, bool bag, set s1, set s2);
61     
62     /* Return TRUE iff s is a singleton set */
63     bool set_single(set s);
64     
65     /* Sort s.  If you call set_sort just before using scan_set then
66        the set elements will be scanned in sorted order. */
67     void set_sort(set_cmp_fn cmp, set s);
68     
69     /* Remove duplicates from s.  Only makes sense if s is a bag. */
70     void set_remove_dups(set_cmp_fn cmp, bool bag, set s);
71     
72     /* Iterate over elements of s */
73     void set_scan(set s, set_scanner *ss);
74     void *set_next(set_scanner *ss);
75     
76     #define scan_set(var, scanner, set) \
77       for (set_scan(set, &scanner), var = set_next(&scanner); \
78            var; var = set_next(&scanner))
79     
80     EXTERN_C_END
81     
82     #endif
