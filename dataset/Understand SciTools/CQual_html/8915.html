1121     			num_to_read = YY_READ_BUF_SIZE;
1122     
1123     		/* Read in more data. */
1124     		YY_INPUT( (&yy_current_buffer->yy_ch_buf[number_to_move]),
1125     			yy_n_chars, num_to_read );
1126     
1127     		yy_current_buffer->yy_n_chars = yy_n_chars;
1128     		}
1129     
1130     	if ( yy_n_chars == 0 )
1131     		{
1132     		if ( number_to_move == YY_MORE_ADJ )
1133     			{
1134     			ret_val = EOB_ACT_END_OF_FILE;
1135     			yyrestart( yyin );
1136     			}
1137     
1138     		else
1139     			{
1140     			ret_val = EOB_ACT_LAST_MATCH;
1141     			yy_current_buffer->yy_buffer_status =
1142     				YY_BUFFER_EOF_PENDING;
1143     			}
1144     		}

1152     
1153     	yytext_ptr = &yy_current_buffer->yy_ch_buf[0];
1154     
1155     	return ret_val;
1156     	}

1269     #ifdef __cplusplus
1270     static int yyinput()
1271     #else
1272     static int input()
1273     #endif
1274     	{
1275     	int c;
1276     
1277     	*yy_c_buf_p = yy_hold_char;
1278     
1279     	if ( *yy_c_buf_p == YY_END_OF_BUFFER_CHAR )
1280     		{
1281     		/* yy_c_buf_p now points to the character we want to return.
1282     		 * If this occurs *before* the EOB characters, then it's a
1283     		 * valid NUL; if not, then we've hit the end of the buffer.
1284     		 */
1285     		if ( yy_c_buf_p < &yy_current_buffer->yy_ch_buf[yy_n_chars] )
1286     			/* This was really a NUL. */
1287     			*yy_c_buf_p = '\0';
1288     
1289     		else
1290     			{ /* need more input */
1291     			int offset = yy_c_buf_p - yytext_ptr;
1292     			++yy_c_buf_p;
1293     
1294     			switch ( yy_get_next_buffer() )
1295     				{
1296     				case EOB_ACT_LAST_MATCH:
1297     					/* This happens because yy_g_n_b()
1298     					 * sees that we've accumulated a
1299     					 * token and flags that we need to
1300     					 * try matching the token before
1301     					 * proceeding.  But for input(),
1302     					 * there's no matching to consider.
1303     					 * So convert the EOB_ACT_LAST_MATCH
1304     					 * to EOB_ACT_END_OF_FILE.
1305     					 */
1306     
1307     					/* Reset buffer status. */
1308     					yyrestart( yyin );
1309     
1310     					/* fall through */
1311     
1312     				case EOB_ACT_END_OF_FILE:
1313     					{
1314     					if ( yywrap() )
1315     						return EOF;
1316     
1317     					if ( ! yy_did_buffer_switch_on_eof )
1318     						YY_NEW_FILE;
1319     #ifdef __cplusplus
1320     					return yyinput();
1321     #else
1322     					return input();
1323     #endif
1324     					}
1325     
1326     				case EOB_ACT_CONTINUE_SCAN:
1327     					yy_c_buf_p = yytext_ptr + offset;
1328     					break;
1329     				}
1330     			}
1331     		}
1332     
1333     	c = *(unsigned char *) yy_c_buf_p;	/* cast for 8-bit char's */
1334     	*yy_c_buf_p = '\0';	/* preserve yytext */
1335     	yy_hold_char = *++yy_c_buf_p;
1336     
1337     
1338     	return c;
1339     	}
