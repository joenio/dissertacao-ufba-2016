129     void finish_analyze(void)
130     {
131       if (!flag_pam_mode && !flag_flow_sensitive)
132         /* This may be delayed until after second pass */
133         traverse_globals((traverse_global_fn) warn_if_dangerous_global, NULL);
134     
135       if (init_qual && noninit_qual)
136         {
137           dd_list_pos pos;
138     
139           dd_scan(pos, definitions)
140     	{
141     	  data_declaration root;
142     	  bool may_be_init;
143     	  bool declared_init;
144     	  
145     	  root = root_ddecl(DD_GET(data_declaration, pos));
146     	  may_be_init = !leq_qual(noninit_qual, qual_qtype(root->qtype));
147     	  declared_init = has_ub_qual(qual_qtype(root->qtype), init_qual);
148     
149     	  if (may_be_init && !declared_init && !root->isinline &&
150     	      !root->addresstaken)
151     	    report_qerror(location_declaration(root->ast), sev_info,
152     			  "Found undeclared %s %s\n",
153     			  qtype_function(root->qtype)
154     			      ? "__init function"
155     			      : "__initdata variable",
156     			  root->name);
157     	}
158         }
159     
160       check_nin(); /* Check the \rho\not\in\varepsilon constraints for
161                       restrict */
162     
163       state = state_finish;
164     }

200     /* Report an error at location l */
201     void vreport_qerror(location loc, severity sev, const char *format,
202     		    va_list args)
203     {
204       if (sev == sev_err || (sev == sev_warn && warnings_are_errors))
205         qerrors = TRUE;
206       if (flag_pam_mode)
207         {
208           if (current_function_decl)
209     	pam_add_error(root_ddecl(current_function_decl->ddecl)->name,
210     		      loc, sev, format, args);
211           else
212     	pam_add_error("top level", loc, sev, format, args);
213         }
214       else
215         {
216           fflush(NULL);
217           if (loc)
218     	fprintf(stderr, "%s:%ld ", loc->filename, loc->lineno);
219           vfprintf(stderr, format, args);
220           if (!growbuf_empty(qual_error_detail) ||
221     	  !growbuf_empty(qtype_error_detail))
222     	{
223     	  fprintf(stderr, "\n");
224     	  if (loc)
225     	    fprintf(stderr, "%s:%ld ", loc->filename, loc->lineno);
226     	  if (!growbuf_empty(qual_error_detail))
227     	    fprintf(stderr, "%s", growbuf_contents(qual_error_detail));
228     	  else
229     	    fprintf(stderr, "%s", growbuf_contents(qtype_error_detail));
230     	}
231           fprintf(stderr, "\n");
232         }
233       if (!growbuf_empty(qual_error_detail))
234         growbuf_reset(qual_error_detail);
235       if (!growbuf_empty(qtype_error_detail))
236         growbuf_reset(qtype_error_detail);
237     }

301     location location_declaration(declaration d)
302     {
303       switch(d->kind)
304         {
305         case kind_variable_decl:
306           {
307     	variable_decl vd = CAST(variable_decl, d);
308     	return location_declarator(vd->declarator);
309           }
310         case kind_oldidentifier_decl:
311           {
312     	return d->loc;
313           }
314         case kind_function_decl:
315           {
316     	function_decl fd = CAST(function_decl, d);
317     	return location_declarator(fd->declarator);
318           }
319         case kind_field_decl:
320           {
321     	field_decl fd = CAST(field_decl, d);
322     	if (!fd->declarator)
323     	  /* Anonymous field */
324     	  return fd->loc;
325     	else
326     	  return location_declarator(fd->declarator);
327           }
328         case kind_enumerator:
329           {
330     	enumerator e = CAST(enumerator, d);
331     	return e->loc;
332           }
333         case kind_implicit_decl:
334           {
335     	implicit_decl id = CAST(implicit_decl, d);
336     	return id->loc;
337           }
338         case kind_error_decl:
339           {
340     	return d->loc;
341           }
342         default:
343           fail_loc(d->loc,
344     	       "Unexpected decl kind 0x%x\n", d->kind);
345         }

539     /* Return TRUE if qt matches the signature s, false otherwise */
540     bool match_signature(enum sig_kind k, qtype qt, location loc)
541     {
542       if (qtype_function(qt))
543         {
544           qtype ret, arg1, arg2;
545           store in, out;
546           qtypelist_scanner s;
547           bool fmatch; /* Formals match */
548     
549           ret = return_qtype(qt);
550           in = store_in_qtype(qt);
551           out = store_out_qtype(qt);
552           qtypelist_scan(arg_qtypes(qt), &s);
553           arg1 = qtypelist_next(&s);
554           arg2 = qtypelist_next(&s);
555           if (qtypelist_next(&s) || qtype_varargs(qt) || qtype_oldstyle(qt))
556     	/* Too many arguments */
557     	return FALSE;
558           if (arg2 && (k == sig_deref || k == sig_addr || k == sig_unop))
559     	/* Too many arguments */
560     	return FALSE;
561     
562           switch (k)
563     	{
564     	case sig_deref:
565     	  /* Note last condition only true if arg1 is a variable */
566     	  /* XXX: Check, don't force qtypes to be equal! */
567     	  fmatch = (qtype_var(ret) && qtype_pointer(arg1) &&
568     		    !mkeq_qtype(loc, points_to_qtype(arg1), ret));
569     	  break;
570     	case sig_addr:
571     	  /* Note last condition only true if ret is a variable */
572     	  fmatch = (qtype_pointer(arg1) &&
573     		    qtype_var(points_to_qtype(arg1)) &&
574     		    qtype_pointer(ret) &&
575     		    !mkeq_qtype(loc,
576     				points_to_qtype(arg1),
577     				points_to_qtype(ret)));
578     	  break;
579     	case sig_unop:
580     	  fmatch = qtype_int(ret) && qtype_int(arg1);
581     	  break;
582     	case sig_binop:
583     	  fmatch = qtype_int(ret) && qtype_int(arg1) && qtype_int(arg2);
584     	  break;
585     	default:
586     	  fail("Unexpected signature kind %x\n", k);
587     	}
588           return fmatch;
589         }
590       else
591         return FALSE;
592     
593       /* Matches */
594       return TRUE;
595     }

597     /* Add qt and dqt to op, checking first that they match op's stated
598        signature kind.  Return TRUE iff an error occurs. */
599     bool add_op_signature(operator op, qtype qt, location loc)
600     {
601       /* If qt or dqt were generalized, instantiate them because we're
602          going to add some constraints and then generalize them again */
603       if (qtype_scheme(qt))
604         qt = instantiate_qtype(qt, loc);
605       if (match_signature(op->sig, qt, loc))
606         {
607           op->qt = generalize_qtype(qt);
608           return FALSE;
609         }
610       else
611         return TRUE;
612     }

614     /* Add the constraints implied by operator signature op to ret, arg1,
615        and arg2. */
616     bool add_op_constraints(operator op, einfo ret, einfo arg1, einfo arg2,
617     			location loc)
618     {
619       qtype op_inst;
620       qtypelist_scanner s;
621       bool result;
622     
623       result = FALSE;
624     
625       op_inst = instantiate_qtype(op->qt, NULL);
626       result = mkeq_qtype(loc, return_qtype(op_inst), ret.qt) || result;
627       qtypelist_scan(arg_qtypes(op_inst), &s);
628       result = mkeq_qtype(loc, qtypelist_next(&s), arg1.qt) || result;
629       if (arg2.qt)
630         result = mkeq_qtype(loc, qtypelist_next(&s), arg2.qt) || result;
631       else
632         assert(!qtypelist_next(&s));
633     
634       return result;
635     }

649     /* Initialize a data declaration's qtype fields.  If ddecl was already
650        declared in a prelude file, use the types from the prelude file.
651        Otherwise verify that ddecl can be unified with what has already
652        been declared.  Display an error if ddecl was already declared in
653        another file but with a different type. */
654     void add_global_ddecl_qtype(data_declaration ddecl, bool generalize)
655     {
656       data_declaration old_ddecl;
657       location loc;
658     
659       ddecl = root_ddecl(ddecl);
660       loc = location_declaration(ddecl->ast);
661       assert(ddecl->isexternalscope);
662     
663       if (ddecl->qtype)
664         return;
665     
666       if (hash_table_lookup(globals, (hash_key) ddecl->name,
667     			(hash_data *) &old_ddecl))
668         {
669           /* If ddecl was previously declared in a prelude file, that
670     	 declaration completely overrides this one */
671           if (old_ddecl->in_prelude && !ddecl->in_prelude)
672     	{
673     	  ddecl->qtype = old_ddecl->qtype;
674     	  ddecl->global = old_ddecl;
675     	  return;
676     	}
677     
678           /* Multiple declarations in prelude files are allowed; neither
679     	 overrides the other
680     	 assert(old_ddecl->in_prelude || !ddecl->in_prelude); */
681     
682           /* Otherwise, check whether the types match */
683           if (match_type(ddecl->type, old_ddecl->type))
684     	/* They did -- just reuse old qtype */
685     	ddecl->qtype = old_ddecl->qtype;
686           else
687     	{
688     	  /* The types differ in some way.  Create new qtypes for
689     	     ddecl and make sure they match the old declaration */
690     	  insist(add_ddecl_qtype(ddecl, generalize));
691     
692     	  if (qtype_scheme(old_ddecl->qtype) || qtype_scheme(ddecl->qtype))
693     	    {
694     	      report_qerror(loc, sev_warn,
695     			    "multiple polymorphic types for ``%s''"
696     			    " -- not checking for consistency", ddecl->name);
697     	    }
698     	  else if (unify_qtype(loc, old_ddecl->qtype, ddecl->qtype))
699     	    {
700     	      report_qerror(loc, sev_err, "conflicting types for ``%s''",
701     			    ddecl->name);
702     	      report_qerror(old_ddecl->ast->loc, sev_info,
703     			    "-- previous declaration of ``%s''", ddecl->name);
704     	    }
705     	}
706     
707           /* Special case: If you redeclare an oldstyle function with a
708     	 newstyle declaration, replcae the old ddecl with the new one */
709           if (qtype_function(ddecl->qtype) && !qtype_oldstyle(ddecl->qtype) &&
710     	  qtype_function(old_ddecl->qtype) && qtype_oldstyle(old_ddecl->qtype))
711     	{
712     	  insistnot(hash_table_insert(globals, (hash_key) ddecl->name, ddecl));
713     	  old_ddecl->global = ddecl;
714     	  old_ddecl = ddecl;
715     	}
716     
717           /* Check whether ddecl has been defined anywhere */
718           /* Don't do this check -- too annoying
719           if (old_ddecl->definition && ddecl->definition)
720     	report_qerror(loc, sev_warn,
721     		      "redefinition of extern ``%s''", ddecl->name);
722           else */
723     
724           /* Update flags */
725           old_ddecl->defined = ddecl->defined =
726     	(old_ddecl->defined || ddecl->defined);
727     
728           old_ddecl->isused = ddecl->isused =
729     	(old_ddecl->isused || ddecl->isused);
730     
731           old_ddecl->addresstaken = ddecl->addresstaken =
732     	(old_ddecl->addresstaken || ddecl->addresstaken);
733     
734           /* New declaration is shadowed by the old one. */
735           ddecl->global = old_ddecl;
736         }
737       else
738         {
739           insist(add_ddecl_qtype(ddecl, generalize));
740           if (ddecl->in_prelude && !ddecl->defined)
741     	ddecl->defined = TRUE;
742           insist(hash_table_insert(globals, (hash_key) ddecl->name, ddecl));
743         }
744     }

746     /* Add a qtype to a static ddecl.  We allow static declarations/definitions
747        in prelude files to supersede static declarations/definitions in files,
748        so we need to do a special check. */
749     void add_static_ddecl_qtype(data_declaration ddecl, bool generalize)
750     {
751       data_declaration old_ddecl;
752     
753       ddecl = root_ddecl(ddecl);
754       assert(static_ddecl(ddecl));
755       if (ddecl->qtype)
756         return;
757     
758       if (hash_table_lookup(globals, (hash_key) ddecl->name,
759     			(hash_data *) &old_ddecl))
760         {
761           if (old_ddecl->in_prelude && static_ddecl(old_ddecl) &&
762     	  !ddecl->in_prelude)
763     	{
764     	  ddecl->qtype = old_ddecl->qtype;
765     	  ddecl->global = old_ddecl;
766     	  return;
767     	}
768           if (ddecl->in_prelude)
769     	report_qerror(location_declaration(ddecl->ast), sev_err,
770     		      "ignoring static prelude declaration -- conflict with "
771     		      "non-static prelude declaration");
772         }
773       else if (ddecl->in_prelude)
774         {
775           add_ddecl_qtype(ddecl, generalize);
776           assert(ddecl->qtype);
777           ddecl->defined = TRUE;
778           insist(hash_table_insert(globals, (hash_key) ddecl->name, ddecl));
779         }
780     
781       /* Fall through case */
782       add_ddecl_qtype(ddecl, generalize);
783     }

785     /* Add a qtype to ddecl.  The logic from qtype.c should really be
786        moved here. */
787     void __add_ddecl_qtype(data_declaration ddecl, bool generalize)
788     {
789       ddecl = root_ddecl(ddecl);
790     
791       if (ddecl->qtype)
792         return;
793     
794       if (ddecl->isexternalscope)
795         add_global_ddecl_qtype(ddecl, generalize);
796       else if (static_ddecl(ddecl))
797         add_static_ddecl_qtype(ddecl, generalize);
798       else
799         /* You can declare things many times, so this may return either
800            true or false. */
801         add_ddecl_qtype(ddecl, generalize);
802     }

824     /* Here's an example of why we can't just pick one ddecl to stand for
825        all of them:
826     
827        file1: void foo(void) { extern char lower(); }
828        file2: char lower(char c) { return c; }
829     
830        If we analyze file1 before file2 and set lower's ddecl to be the
831        oldstyle function, we'll get messed up at the definition.  On the
832        other hand, if we change unify_qtype to merge extra parameters, we
833        can't handle varargs oldstyle functions. */

835     /* Apply f to one ddecl for each global */
836     void traverse_globals(void (*f)(data_declaration, void *),
837     				void *arg)
838     {
839       hash_table_scanner hts;
840       data_declaration ddecl;
841     
842       hash_table_scan(globals, &hts);
843       while (hash_table_next(&hts, NULL, (hash_data *) &ddecl))
844         {
845           assert(!ddecl->shadowed);
846           f(ddecl, arg);
847         }
848     }

856       /* The follow test doesn't work, because of the filtering w.r.t. effects
857          we do
858     	    (s ? nontriv_soln_qtype(ddecl->fs_qtype, NULL) : FALSE))));
859       */
860     }

907     /**************************************************************************
908      *                                                                        *
909      * Tree Traversal                                                         *
910      *                                                                        *
911      * The code that walks over the AST and generates the constraint          *
912      * graph, plus whatever info we'll need to do linking.  Think of this     *
913      * as a big pattern-match.                                                *
914      *                                                                        *
915      **************************************************************************/
916     
917     void analyze(declaration program)
918     {
919       declaration d;
920       dinfo next_info;
921       dd_list global_drinfolist;
922     
923       global_drinfolist = dd_new_list(parse_region);
924     
925       /* Set this fresh for each file - sleepy qual */
926       interrupt_status_qtype = NULL;
927     
928       unparse_start(stdout);
929       pam_add_file(program->loc->filename);
930       AST_set_parents(CAST(node, program));
931       scan_declaration(d, program)
932         {
933           /* Assume left-to-right order of evaluation */
934           next_info = analyze_declaration(d, global_env, global_drinfolist);
935           mkleq_effect(next_info.eff, global_effect);
936           mkleq_effect(next_info.alocs, global_env);
937           /* report_qerror(d->loc, sev_err,
938     		      "Allocation or initialization accesses globally-"
939     		      "restricted location");
940           */
941         }
942     }
943     
944     static dinfo analyze_declaration(declaration d, effect env, dd_list drinfolist)
945     {
946       switch (d->kind)
947         {
948         case kind_asm_decl:
949           {
950     	/*	report_qerror(d->loc, sev_warn,
951     		"warning: ignoring in-line assembly code");*/
952     	return mkdinfo(effect_empty, effect_empty);
953           };
954         case kind_data_decl:
955           {
956     	data_decl dd = CAST(data_decl, d);
957     	declaration decl;
958     	effect eff, local_env, alocs;
959     
960     	eff = effect_empty;
961     	local_env = env;
962     	alocs = effect_empty;
963     
964     	scan_declaration(decl, dd->decls) {
965     	  dinfo next;
966     
967     	  next = analyze_declaration(decl, local_env, drinfolist);
968     	  local_env = effect_union(local_env, next.alocs);
969     	  alocs = effect_union(alocs, next.alocs);
970     	  eff = effect_union(eff, next.eff);
971     	}
972     
973     	if (!dd->decls && dd->modifiers && is_tag_ref(dd->modifiers))
974     	  /* If we only declared a struct/union/enum, add types for
975     	     the fields */
976     	  analyze_tag_ref(CAST(tag_ref, dd->modifiers));
977     
978     	return mkdinfo(eff, alocs);
979           };
980         case kind_variable_decl:
981           {
982     	variable_decl vd = CAST(variable_decl, d);
983     	data_declaration root;
984     	effect eff, alocs;
985     	bool isglobal;
986     
987     	/* Ignore asm_stmt
988     	if (vd->asm_stmt)
989     	  fail_loc(vd->loc,
990     		   "Unimplemented: asm in variable_decl\n", 0);
991     	*/
992     
993     	root = root_ddecl(vd->ddecl);
994     
995     	if (is_op_name(root->name))
996     	  return analyze_operator_declaration(vd);
997     
998     	if (root->kind == decl_typedef)
999     	  {
1000     	    /* Skip typedefs -- except enums, since we have to give
1001     	       qtypes to those */
1002     	    if (type_enum(root->type))
1003     	      analyze_tag_ref(type_tag(root->type)->ast);
1004     	    return mkdinfo(effect_empty, effect_empty);
1005     	  }
1006     
1007     	__add_ddecl_qtype(root, TRUE);
1008     
1009     	eff = effect_empty;
1010     	alocs = effect_empty;
1011     	isglobal = (root->kind == decl_function ||
1012     		    root->isexternalscope ||
1013     		    root->vtype == variable_static);
1014     
1015     	/* Add the alocs from this qtype to the appropriate effects */
1016     	if (root->qtype)
1017     	  {
1018     	    if (isglobal)
1019     	      {
1020     		mkleq_effect(alocs_qtype(root->qtype), global_env);
1021     		mkleq_effect(defn_effect_qtype(root->qtype), global_effect);
1022     	      }
1023     	    else
1024     	      {
1025     		alocs = effect_union(alocs_qtype(root->qtype), alocs);
1026     		eff = effect_union(defn_effect_qtype(root->qtype), eff);
1027     	      }
1028     	  }
1029     
1030     	/* Handle restrict */
1031     	if (restrict_variable_decl(vd, isglobal))
1032     	  {
1033     	    if (isglobal)
1034     	      {
1035     		/* Forbid use of unrestricted loc anywhere */
1036     		mknin_effect(vd->loc, vd->rinfo->flocs, global_effect);
1037     
1038     		/* 2p:  Add effect of allocating the new loc */
1039     		if (flag_flow_sensitive)
1040     		  mkleq_effect(vd->rinfo->flocs_alloc, global_effect);
1041     	      }
1042     	    else
1043     	      {
1044     		/* Forbid escape */
1045     		mknin_effect(vd->loc, vd->rinfo->rlocs, env);
1046     
1047     		/* Add rloc to environment */
1048     		alocs = effect_union(vd->rinfo->rlocs, alocs); 
1049     
1050     		/* 2p:  Add effect of allocating the new loc */
1051     		if (flag_flow_sensitive)
1052     		  eff = effect_union(vd->rinfo->rlocs_alloc, eff);
1053     	      }
1054     	  }
1055     	
1056     	/* If readonly is on global, forbit write everywhere */
1057     	if (isglobal)
1058     	  enforce_var_decl_readonly(vd, global_effect);
1059     
1060     	if (vd->arg1)
1061     	  {
1062     	    qtype init_qt;
1063     	    sinfo arg1;
1064     
1065     	    /* If we are initializing a global restricted loc, then
1066     	       root->qtype contains the restricted loc */
1067     	    /* XXX: This doesn't change anything if we only have
1068     	       top-level restricts */
1069     	    if (vd->rinfo && vd->rinfo->isglobal)
1070     	      init_qt = vd->rinfo->fqtype;
1071     	    else
1072     	      init_qt = root->qtype;
1073     
1074     	    if (!qtype_array(init_qt))
1075     	      init_qt = points_to_qtype(init_qt);
1076     
1077     	    arg1 = analyze_init(vd->loc, init_qt, vd->arg1, env, drinfolist);
1078     	    vd->arg1_eff = arg1.eff;
1079     
1080     	    if (isglobal)
1081     	      /* enforce_restricts is never called for globals */
1082     	      eff = effect_union(eff, arg1.eff);
1083     	  }
1084     
1085     	/* If we restricted this loc, replace its qtype with its
1086     	   restricted qtype, which we'll see from now on */
1087     	if (vd->rinfo)
1088     	  root->qtype = vd->rinfo->rqtype;
1089     
1090     	/* If attribute alias, unify aliased qtype */
1091     	if (root->alias != NULL)
1092     	  {
1093     	    data_declaration alias_ddecl;
1094     
1095     	    if (hash_table_lookup(globals, (hash_key) root->alias, (hash_data *) &alias_ddecl))
1096     	      {
1097     		alias_ddecl = root_ddecl(alias_ddecl);
1098     		unify_qtype(vd->loc, alias_ddecl->qtype, root->qtype);
1099     	      }
1100     	  }
1101     
1102     	/* Set interrupt status qtype - sleepy qual */
1103     	if (have_sleepy_quals && 
1104     	    !strcmp(root->name, "fake_interrupt_status"))
1105     	  {
1106     	    if (interrupt_status_qtype != NULL)
1107     	      assert(interrupt_status_qtype == root->qtype);
1108     	    else
1109     	      interrupt_status_qtype = root->qtype;
1110     	  }
1111     
1112     	/* Add PAM markup */
1113     	if ((flag_pam_mode || flag_print_results) &&
1114     	    root->kind != decl_typedef)
1115     	  add_declarator_overlay(vd->declarator, root->qtype);
1116     
1117     	/*
1118     	printf("Effect of declaring %s: ", root->name);
1119     	print_effect(printf, eff);
1120     	printf("\n");
1121     	*/
1122     	return mkdinfo(eff, alocs);
1123           }
1124           break;
1125         case kind_function_decl:
1126           {
1127     	function_decl fd = CAST(function_decl, d), old_function_decl;
1128     	data_declaration root;
1129     	qtype ret_qtype;
1130     	effect param_rtype_alocs; /* alocs in parameters */
1131     	effect alocs, local_env, eff;
1132     
1133     	root = root_ddecl(fd->ddecl);
1134     	dd_add_last(parse_region, definitions, root);
1135     
1136     	__add_ddecl_qtype(root, TRUE);
1137     	if (root->global && root->global->in_prelude)
1138     	  /* Ignore definition of something from a prelude file */
1139     	  return mkdinfo(effect_empty, effect_empty);
1140     
1141     	if (qtype_scheme(root->qtype))
1142     	  {
1143     	    /* If you change this, also change the case in flow.c for
1144     	       ignoring the definition. */
1145     	    report_qerror(fd->loc, sev_warn,
1146     			  "Ignoring definition of polymorphic function");
1147     	    return mkdinfo(effect_empty, effect_empty);
1148     	  }
1149     
1150     	alocs = alocs_qtype(root->qtype);
1151     
1152     	if (flag_pam_mode || flag_print_results)
1153     	  add_declarator_overlay(fd->declarator, root->qtype);
1154     
1155     	old_function_decl = current_function_decl;
1156     	current_function_decl = fd;
1157     	ret_qtype = return_qtype(root->qtype);
1158     	mkleq_effect(effect_qtype(root->qtype), global_effect);
1159     	/*  report_qerror(fd->loc, sev_err,
1160     	    "Function accesses globally-restricted location"); */
1161     
1162     	/* 2p: Add effect of allocation the function to the global effect */
1163     	mkleq_effect(defn_effect_qtype(root->qtype), global_effect);
1164     
1165     	local_env = env;
1166     	eff = effect_empty;
1167     	param_rtype_alocs = effect_empty;
1168     
1169     	/*** Scan argument list ***/
1170     
1171     	{
1172     	  declaration arg;
1173     	  qtypelist_scanner old_qtypes;
1174     	  int i;
1175     
1176     	  i = 1;
1177     	  qtypelist_scan(arg_qtypes(root->qtype), &old_qtypes);
1178     
1179     	  if (!is_void_parms(fd->fdeclarator->parms))
1180     	    scan_declaration(arg, fd->fdeclarator->parms)
1181     	      /* Ignore ...'s */
1182     	      if (!is_ellipsis_decl(arg))
1183     		{
1184     		  data_declaration root_vd;
1185     		  variable_decl argvd;
1186     		  oldidentifier_decl oid;
1187     		  qtype old_qtype, arg_qt;
1188     		  
1189     		  /* Construct type of parameter */
1190     		  if (arg->kind == kind_data_decl)
1191     		    {
1192     		      data_decl argd = CAST(data_decl, arg);
1193     
1194     		      argvd = CAST(variable_decl, argd->decls);
1195     		      assert(!argvd->next);    /* Only var_decl per data_decl.
1196     						   multi var_decls are for
1197     						   things like int a, b. */
1198     		      oid = NULL;
1199     		      root_vd = root_ddecl(argvd->ddecl);
1200     		    }
1201     		  else
1202     		    {
1203     		      oid = CAST(oldidentifier_decl, arg);
1204     		      argvd = NULL;
1205     		      root_vd = root_ddecl(oid->ddecl);
1206     		    }
1207     
1208     		  __add_ddecl_qtype(root_vd, FALSE);
1209     		  assert(root_vd->qtype);
1210     	
1211     		  /* Match up parameter type with function type */
1212     		  old_qtype = qtypelist_next(&old_qtypes);
1213     		  arg_qt = points_to_qtype(root_vd->qtype);
1214     
1215     		  if (old_qtype && unify_qtype(arg->loc, old_qtype, arg_qt))
1216     		    report_qerror(arg->loc, sev_err,
1217     				  "type of parameter %d doesn't match "
1218     				  "previous uses", i);
1219     		  else if (!old_qtype && arg->kind == kind_data_decl)
1220     		    report_qerror(arg->loc, sev_err,
1221     				  "parameter %d doesn't appear in "
1222     				  "previous uses", i);
1223     
1224     		  /* NOTE:  No checking done for oldstyle identifiers!
1225     		     This matches gcc's beavhior. */
1226     
1227     		  if (old_qtype)
1228     		    param_rtype_alocs = effect_union(param_rtype_alocs,
1229     						     alocs_qtype(old_qtype));
1230     
1231     		  /* 2p:  Add the effect of allocating space for parameter */
1232     		  eff = effect_union(eff, defn_effect_qtype(root_vd->qtype));
1233     
1234     		  /* Assume the function touches all its parameters */
1235     		  /* Don't assume this for now */
1236     		  /*eff = effect_union(eff, alocs_qtype(root_vd->qtype));*/
1237     
1238     		  local_env = effect_union(local_env,
1239     					   alocs_qtype(root_vd->qtype));
1240     
1241     		  if (arg->kind == kind_data_decl &&
1242     		      restrict_variable_decl(argvd, FALSE))
1243     		    {
1244     		      mknin_effect(argvd->loc, argvd->rinfo->rlocs, local_env);
1245     		      local_env = effect_union(argvd->rinfo->rlocs, local_env);
1246     
1247     		      /* 2p:  Add the location from restrict_variable_decl */
1248     		      if (flag_flow_sensitive)
1249     			eff = effect_union(eff, argvd->rinfo->rlocs_alloc);
1250     		      root_vd->qtype = argvd->rinfo->rqtype;
1251     		    }
1252     
1253     		  if (flag_pam_mode || flag_print_results)
1254     		    {
1255     		      if (arg->kind == kind_data_decl)
1256     			add_declarator_overlay(argvd->declarator,
1257     					       root_vd->qtype);
1258     		      else
1259     			pam_add_overlay_file(oid->loc,
1260     					     oid->cstring.data,
1261     					     root_vd->qtype);
1262     		    }
1263     		  
1264     		  i++;
1265     		}
1266     	  if (qtypelist_next(&old_qtypes))
1267     	    report_qerror(fd->loc, sev_err,
1268     			  "inconsistent number of arguments for %s",
1269     			  root->name);
1270     	}
1271     
1272     	/*** Evaluate body ***/
1273     
1274     	{
1275     	  sinfo body;
1276     
1277     	  assert(is_compound_stmt(fd->stmt));
1278     	  body = analyze_statement(fd->stmt, local_env, drinfolist);
1279     	  eff = effect_union(eff,
1280     			     enforce_restricts(fd->fdeclarator->parms,
1281     					       body.eff,
1282     					       ret_qtype));
1283     	  enforce_readonly(fd->fdeclarator->parms, eff);
1284     	}
1285     
1286     	/* Apply (Down) once more to filter out locations neither in
1287     	   parameters (param_rtype_alocs) or return value (added by
1288     	   filter_effects) */
1289     	fd->scope_env = effect_union(param_rtype_alocs,
1290     				     filter_effects(NULL, env));
1291     	mark_effect_interesting(fd->scope_env);
1292     	eff = effect_inter(eff, fd->scope_env);
1293     	mkleq_effect(eff, effect_qtype(root->qtype));
1294     
1295     	current_function_decl = old_function_decl;
1296     	return mkdinfo(effect_empty, alocs);
1297           };
1298         case kind_extension_decl:
1299           {
1300     	extension_decl ed = CAST(extension_decl, d);
1301     	return analyze_declaration(ed->decl, env, drinfolist); 
1302     	/* Ignore __extension__ */
1303           };
1304         default:
1305           fail_loc(d->loc,
1306     	       "Unexpected decl kind 0x%x\n", d->kind);
1307         }
1308     }

1310     static dinfo analyze_operator_declaration(variable_decl vd)
1311     {
1312       data_declaration root;
1313       operator op;
1314     
1315       root = root_ddecl(vd->ddecl);
1316       assert(is_op_name(root->name));
1317       if (!root->isexternalscope)
1318         {
1319           report_qerror(vd->loc, sev_err,
1320     		    "Operator not declared at top level");
1321           return mkdinfo(effect_empty, effect_empty);
1322         }
1323     
1324       add_global_ddecl_qtype(root, TRUE);
1325     
1326       op = find_op_name(root->name);
1327       if (op && op->qt)
1328         report_qerror(vd->loc, sev_err,
1329     		  "redefinition of operator signature");
1330       else if (op)
1331         {
1332           /* It's an operator definition */
1333           if (add_op_signature(op, root->qtype, vd->loc))
1334     	report_qerror(vd->loc, sev_err,
1335     		      "definition of operator with incorrect signature");
1336           
1337           /* Reset qtype to fully constrained qtype */
1338           root->qtype = op->qt;      
1339         }
1340       else
1341         report_qerror(vd->loc, sev_err,
1342     		  "definition of non-existent operator signature");
1343       
1344       /* Add PAM markup */
1345       if ((flag_pam_mode || flag_print_results) &&
1346           root->kind != decl_typedef)
1347         add_declarator_overlay(vd->declarator, root->qtype);
1348       
1349       return mkdinfo(effect_empty, effect_empty);
1350     }

1415     /* XXX Init */
1416     /* Analyze the initialization of lhs_qtype and lhs_dqtype (r-types)
1417        with rhs, in initial store in. */
1418     static sinfo analyze_init(location loc, qtype lhs_qtype, expression rhs,
1419     			  effect env, dd_list drinfolist)
1420     {
1421       if (rhs->kind == kind_string && (qtype_string(lhs_qtype)))
1422         {
1423           /* Special case for strings.  The rhs is a char ptr, but the
1424     	 quals should flow into the array quals, anyhow. */
1425     
1426           einfo arg1;
1427           assert(qtype_string(lhs_qtype));
1428           /* assert(!rhs->next); not always true -- init arrays of strings */
1429           arg1 = analyze_expression(rhs, rpos, env, drinfolist);
1430           /* Hack:  rhs is a string, so no side effects. */
1431           if (init_string(rhs->loc, lhs_qtype, arg1.qt))
1432     	report_qerror(rhs->loc, sev_err, "incompatible initializer");
1433           
1434           return mksinfo(arg1.eff);
1435         }
1436       else if (rhs->kind == kind_init_list && (qtype_array(lhs_qtype)))
1437         {
1438           /* Initialize the array contents qtype with each rhs */
1439           init_list il = CAST(init_list, rhs);
1440           qtype lhs_contents_qtype;
1441           expression e;
1442           sinfo init;
1443           effect eff;
1444     
1445           assert(qtype_array(lhs_qtype));
1446     
1447           lhs_contents_qtype = array_of_qtype(lhs_qtype);
1448           eff = effect_empty;
1449     
1450           scan_expression(e, il->args)
1451     	if (e->kind == kind_init_index)
1452     	  {
1453     	    /* Handle this case here -- only arrays can be initialized
1454     	       this way. */
1455     	    init_index ii = CAST(init_index, e);
1456     	    einfo arg1, arg2;
1457     
1458     	    arg1 = analyze_expression(ii->arg1, rpos, env, drinfolist);
1459     	    eff = effect_union(eff, arg1.eff);
1460     	    if (ii->arg2)
1461     	      {
1462     		arg2 = analyze_expression(ii->arg2, rpos, env, drinfolist);
1463     		eff = effect_union(eff, arg2.eff);
1464     	      }
1465     	    init = analyze_init(loc, lhs_contents_qtype, ii->init_expr, env,
1466     				drinfolist);
1467     	    eff = effect_union(eff, init.eff);
1468     	  }
1469     	else
1470     	  {
1471     	    init = analyze_init(loc, lhs_contents_qtype, e, env, drinfolist);
1472     	    eff = effect_union(eff, init.eff);
1473     	  }
1474           return mksinfo(eff);
1475          }
1476        else if (rhs->kind == kind_init_list && (qtype_struct(lhs_qtype) ||
1477     					    qtype_union(lhs_qtype)))
1478          {
1479            /* Initialize each of the field qtypes with each rhs */
1480            init_list il = CAST(init_list, rhs);
1481            field_scanner fs;
1482            qtype fqt;
1483            expression e;
1484            effect eff;
1485     
1486            fqt = NULL;
1487            field_scan(lhs_qtype, &fs);
1488            eff = effect_empty;
1489     
1490            scan_expression(e, il->args)
1491     	 {
1492     	   expression init_expr;
1493     	   sinfo init;
1494     	   
1495     	   if (e->kind == kind_init_field)
1496     	     {
1497     	       /* Handle specific field initialization here -- only
1498     		  structures/unions can be initialized this way.
1499     		  Adding a label changes where we start initializing
1500     		  the struct/union. */
1501     	       init_field ifd = CAST(init_field, e);
1502     	       field_scan_at(lhs_qtype, ifd->word1->cstring, &fs);
1503     	       if (!fs)
1504     		 {
1505     		   report_qerror(loc, sev_err,
1506     			      "unknown field `%s' in aggregate initialization",
1507     				 ifd->word1->cstring);
1508     		   break;
1509     		 }
1510     	       init_expr = ifd->init_expr;
1511     	     }
1512     	   else
1513     	     init_expr = e;
1514     
1515     	   fqt = field_next(&fs);
1516     	   if (!fqt)
1517     	     {
1518     	       report_qerror(loc, sev_err,
1519     			  "aggregate initailized with wrong number of fields");
1520     	       break;
1521     	     }
1522     
1523     	   /* Unless we're initializing an array, get the contents of the
1524     	      ref.  Note we can't be initializing a function -- not C++. */
1525     	   if (!qtype_array(fqt))
1526     	     fqt = points_to_qtype(fqt);
1527     	   
1528     	   init = analyze_init(loc, fqt, init_expr, env, drinfolist);
1529     	   eff = effect_union(eff, init.eff);
1530     	 }
1531           /* OK to initialize aggregate with too few fields
1532     	 err = (field_next(&fs) || err); */
1533            return mksinfo(eff);
1534          }
1535       else
1536         {
1537           /* Default case.  Note we don't do a substitute_store, because variables
1538     	 can't be used before they're initialized. */
1539           einfo arg1;
1540     
1541           arg1 = analyze_expression(rhs, rpos, env, drinfolist);
1542           
1543           if (mkleq_assign_convert(loc, arg1.qt, lhs_qtype))
1544     	report_qerror(loc, sev_err,
1545     		      "incompatible right-hand side in initialization");
1546           return mksinfo(arg1.eff);
1547         }
1548     }

1550     static sinfo analyze_statement(statement s, effect env, dd_list drinfolist)
1551     {
1552       switch (s->kind)
1553         {
1554         case kind_asm_stmt:
1555           /*      report_qerror(s->loc, sev_warn,
1556     	      "warning: ignoring in-line assembly code");*/
1557           return mksinfo(effect_empty);
1558           break;
1559         case kind_compound_stmt:
1560           {
1561     	compound_stmt cs = CAST(compound_stmt, s);
1562     	declaration decl;
1563     	statement stmt;
1564     	effect eff, local_env;
1565     	dd_list new_drinfolist, old_drinfolist;
1566     
1567     	old_drinfolist = drinfolist;
1568     	new_drinfolist = dd_copy(parse_region, drinfolist);
1569     
1570     	cs->enclosing_cs = cur_cs;
1571     	cur_cs = cs;
1572     
1573     	eff = effect_empty;
1574     	local_env = env;
1575     
1576     	/* Analyze the declarations. */
1577     	scan_declaration(decl, cs->decls)
1578     	  {
1579     	    dinfo next;
1580     
1581     	    assert(decl->kind != kind_asm_decl); /*asm_decl only at toplevel */
1582     	    next = analyze_declaration(decl, local_env, new_drinfolist);
1583     	    eff = effect_union(eff, next.eff);
1584     	    local_env = effect_union(local_env, next.alocs);
1585     	  }
1586     
1587     	cs->scope_env = local_env;
1588     	mark_effect_interesting(cs->scope_env);
1589     
1590     	/* Analyze the body, gathering up the effects. */
1591     	scan_statement(stmt, cs->stmts)
1592     	  {
1593     	    sinfo next = analyze_statement(stmt, local_env, new_drinfolist);
1594     	    eff = effect_union(eff, next.eff);
1595     	  }
1596     
1597     	/* Add constraints from effects */
1598     	eff = enforce_restricts(cs->decls, eff, NULL);
1599     	enforce_readonly(cs->decls, eff);
1600     
1601     	/* Handle deep restricts */
1602     
1603     	/* Filter out non-escaping effects */
1604     	cs->filter_env = filter_effects(NULL, env);
1605     	mark_effect_interesting(cs->filter_env);
1606     	eff = effect_inter(eff, cs->filter_env);
1607     
1608     	/*
1609     	printf("Filter line %ld is", cs->loc.lineno);
1610     	print_effect(printf, eff);
1611     	printf("\n");
1612     	*/
1613     
1614     	cur_cs = cs->enclosing_cs;
1615     
1616     	return mksinfo(eff);
1617           };
1618           break;
1619         case kind_if_stmt:
1620           {
1621     	if_stmt is = CAST(if_stmt, s);
1622     	einfo guard;
1623     	sinfo then;
1624     	effect eff;
1625     
1626     	guard = analyze_expression(is->condition, rpos, env, drinfolist);
1627     	then = analyze_statement(is->stmt1, env, drinfolist);
1628     	eff = effect_union(guard.eff, then.eff);
1629     
1630     	if (is->stmt2)
1631     	  {
1632     	    sinfo els;
1633     
1634     	    els = analyze_statement(is->stmt2, env, drinfolist);
1635     	    eff = effect_union(eff, els.eff);
1636     	  }
1637     
1638     	return mksinfo(eff);
1639           };
1640           break;
1641         case kind_labeled_stmt:
1642           {
1643     	labeled_stmt ls = CAST(labeled_stmt, s);
1644     
1645     	ls->label->enclosing_cs = cur_cs;
1646     
1647     	return analyze_statement(ls->stmt, env, drinfolist);
1648           };
1649           break;
1650         case kind_expression_stmt:
1651           {
1652     	expression_stmt es = CAST(expression_stmt, s);
1653     	einfo ei;
1654     
1655     	ei = analyze_expression(es->arg1, rpos, env, drinfolist);
1656     	return mksinfo(ei.eff);
1657           };
1658           break;
1659         case kind_while_stmt:
1660           {
1661     	while_stmt ws = CAST(while_stmt, s);
1662     	einfo guard;
1663     	sinfo body;
1664     	effect eff;
1665     
1666     	ws->enclosing_cs = cur_cs;
1667     
1668     	guard = analyze_expression(ws->condition, rpos, env, drinfolist);
1669     	body = analyze_statement(ws->stmt, env, drinfolist);
1670     	eff = effect_union(guard.eff, body.eff);
1671     	return mksinfo(eff);
1672           };
1673           break;
1674         case kind_dowhile_stmt:
1675           {
1676     	dowhile_stmt dws = CAST(dowhile_stmt, s);
1677     	sinfo body;
1678     	effect eff;
1679     
1680     	dws->enclosing_cs = cur_cs;
1681     
1682     	body = analyze_statement(dws->stmt, env, drinfolist);
1683     	eff = body.eff;
1684     
1685     	if (!definite_zero(dws->condition))
1686     	  {
1687     	    /* Catch do { x } while(0); case -- used in macro expansions */
1688     	    einfo guard;
1689     
1690     	    guard = analyze_expression(dws->condition, rpos, env, drinfolist);
1691     	    eff = effect_union(eff, guard.eff);
1692     	  }
1693     	return mksinfo(eff);
1694           };
1695           break;
1696         case kind_switch_stmt:
1697           {
1698     	switch_stmt ss = CAST(switch_stmt, s);
1699     	einfo guard;
1700     	sinfo body;
1701     	effect eff;
1702     
1703     	ss->enclosing_cs = cur_cs;
1704     
1705     	guard = analyze_expression(ss->condition, rpos, env, drinfolist);
1706     	body = analyze_statement(ss->stmt, env, drinfolist);
1707     	eff = effect_union(guard.eff, body.eff);
1708     
1709     	return mksinfo(eff);
1710           };
1711           break;
1712         case kind_for_stmt:
1713           {
1714     	for_stmt fs = CAST(for_stmt, s);
1715     	sinfo body;
1716     	effect eff;
1717     
1718     	fs->enclosing_cs = cur_cs;
1719     
1720     	eff = effect_empty;
1721     
1722     	if (fs->arg1)
1723     	  {
1724     	    einfo arg1;
1725     
1726     	    arg1 = analyze_expression(fs->arg1, rpos, env, drinfolist);
1727     	    eff = effect_union(eff, arg1.eff);
1728     	  }
1729     
1730     	if (fs->arg2)
1731     	  {
1732     	    einfo arg2;
1733     
1734     	    arg2 = analyze_expression(fs->arg2, rpos, env, drinfolist);
1735     	    eff = effect_union(eff, arg2.eff);
1736     	  }
1737     
1738     	body = analyze_statement(fs->stmt, env, drinfolist);
1739     	eff = effect_union(eff, body.eff);
1740     
1741     	if (fs->arg3)
1742     	  {
1743     	    einfo arg3;
1744     
1745     	    arg3 = analyze_expression(fs->arg3, rpos, env, drinfolist);
1746     	    eff = effect_union(eff, arg3.eff);
1747     	  }
1748     
1749     	return mksinfo(eff);
1750           };
1751           break;
1752         case kind_return_stmt:
1753           {
1754     	return_stmt rs = CAST(return_stmt, s);
1755     	qtype ret_qtype;
1756     	effect eff;
1757     
1758     	ret_qtype = cur_return_qtype();
1759     
1760     	eff = effect_empty;
1761     	if (rs->arg1)
1762     	  {
1763     	    einfo arg1;
1764     
1765     	    arg1 = analyze_expression(rs->arg1, rpos, env, drinfolist);
1766     	    eff = effect_union(eff, arg1.eff);
1767     	    /* ret_qtype is an r-type, so no deref */
1768     	    if (mkleq_assign_convert(rs->loc, arg1.qt, ret_qtype))
1769     	      report_qerror(rs->loc, sev_err,
1770     			    "return type incompatible with function type");
1771     	  }
1772     	/* Don't raise an error here -- will already be flagged by
1773     	   parser if warn_return_type is enabled
1774     	else
1775     	  {
1776     	    if (warn_return_type && !qtype_void(ret_qtype))
1777     	      report_qerror(rs->loc, sev_err,
1778     		    "`return' with no value, in function returning non-void");
1779     	  }
1780     	*/
1781     
1782     	return mksinfo(eff);
1783           };
1784           break;
1785         case kind_computed_goto_stmt:
1786           {
1787     	computed_goto_stmt cgs = CAST(computed_goto_stmt, s);
1788     	einfo arg1;
1789     
1790     	/* stmt is goto *arg1 */
1791     	arg1 = analyze_expression(cgs->arg1, rpos, env, drinfolist);
1792     	return mksinfo(arg1.eff);
1793           };
1794           break;
1795         case kind_break_stmt:
1796           {
1797     	break_stmt bs = CAST(break_stmt, s);
1798     	bs->enclosing_cs = cur_cs;
1799     	
1800     	return mksinfo(effect_empty);
1801           }
1802         case kind_continue_stmt:
1803           {
1804     	continue_stmt cs = CAST(continue_stmt, s);
1805     	cs->enclosing_cs = cur_cs;
1806     	
1807     	return mksinfo(effect_empty);
1808           }
1809         case kind_goto_stmt:
1810           {
1811     	goto_stmt gs = CAST(goto_stmt, s);
1812     	gs->enclosing_cs = cur_cs;
1813     
1814     	return mksinfo(effect_empty);
1815           }
1816         case kind_empty_stmt:
1817           return mksinfo(effect_empty);
1818         case kind_change_type_stmt:
1819           {
1820     	change_type_stmt ct = CAST(change_type_stmt, s);
1821     	qtype new_qt;
1822     	einfo arg1;
1823     	aloc al;
1824     	
1825     	arg1 = analyze_expression(ct->arg1, lpos, env, drinfolist);
1826     	qtype_mklhs_nonconst(ct->loc, arg1.qt);
1827     	
1828     	new_qt = type_to_qtype(ct->asttype->type, "__change", ct->loc);
1829     	assert(!ct->qtype);
1830     	ct->qtype = new_qt;
1831     	al = aloc_qtype(arg1.qt);
1832     	
1833     	mark_aloc_interesting(al);
1834     	
1835     	assert(qtype_pointer(arg1.qt));
1836     	if (mkleq_qtype(ct->loc, points_to_qtype(arg1.qt), new_qt))
1837     	  report_qerror(ct->loc, sev_err,
1838     			"new type incompatible with previous type");
1839     	return mksinfo(effect_union(arg1.eff, effect_wr(al)));
1840           }
1841           break;
1842         case kind_assert_type_stmt:
1843           {
1844     	assert_type_stmt at = CAST(assert_type_stmt, s);
1845     	einfo arg1;
1846     	type t;
1847     	qtype assert_qt;
1848     	
1849     	/*arg1 = analyze_expression(at->arg1, rpos);*/
1850     	arg1 = analyze_expression(at->arg1, rpos, env, drinfolist);
1851     	t = at->asttype->type;
1852     	assert_qt = type_to_qtype(t, "assert", at->asttype->loc);
1853     	assert(!at->qtype);
1854     	at->qtype = assert_qt;
1855     	
1856     	if (mkleq_qtype(at->loc, arg1.qt, assert_qt))
1857     	  report_qerror(at->loc, sev_err,
1858     			"unsatisfiable assertion");
1859     	
1860     	/* Pretend we have the effect of arg1, so that in 3rd pass
1861     	   we'll be able to retrieve arg1 from the store, if necessary */
1862     	return mksinfo(arg1.eff);
1863           }
1864           break;
1865         case kind_deep_restrict_stmt:
1866           {
1867     	deep_restrict_stmt dr = CAST(deep_restrict_stmt, s);
1868     	drinfo dri;
1869     	aloc old_aloc, new_aloc;
1870     	einfo arg1;
1871     	sinfo stmt;
1872     	effect eff = effect_empty;
1873     
1874     	arg1 = analyze_expression(dr->arg1, lpos, env, drinfolist);
1875     	assert(qtype_pointer(arg1.qt));
1876     	old_aloc = aloc_qtype(arg1.qt);
1877     	new_aloc = make_aloc(rstrcat(parse_region,name_aloc(old_aloc),
1878     				     "_r"), points_to_qtype(arg1.qt));
1879     
1880     	mark_aloc_interesting(old_aloc);
1881     	mark_aloc_interesting(new_aloc);
1882     
1883     	dri = ralloc(parse_region, struct Drinfo);
1884     	dri->qtype = arg1.qt;
1885     	dri->rqtype = mkqtype_pointer(qual_qtype(arg1.qt), 
1886     				      points_to_qtype(arg1.qt), new_aloc);
1887     	dri->effect = arg1.eff;
1888     	dri->expression = dr->arg1;
1889     	dri->location = dr->loc;
1890     	dr->drinfo = dri;
1891     	    
1892     	/* Constraints for restrict */
1893     	mknin_effect_qtype(dri->location, effect_rwr(new_aloc),
1894     			   points_to_qtype(arg1.qt));
1895     	mknin_effect(dri->location, effect_single(new_aloc), env);
1896     	env = effect_union(effect_single(new_aloc), env);
1897     	if (flag_flow_sensitive)
1898     	  eff = effect_union(effect_alloc(new_aloc), eff);
1899     
1900     	/* Constraints for referential transparency */
1901     	forall_aloc_mknin_effect(dri->location, eff_wr, dri->effect);
1902     	forall_aloc_mknin_effect(dri->location, eff_alloc, dri->effect);
1903     
1904     	dd_add_first(parse_region, drinfolist, dri);
1905     
1906     	/* Analyze the stmt */
1907     	stmt = analyze_statement(dr->stmt, env, drinfolist);
1908     	
1909     	/* Constraints for restrict */
1910     	mknin_effect(dri->location, effect_rwr(aloc_qtype(dri->qtype)), 
1911     		     stmt.eff);
1912     
1913     	/* Constraints for referential transparency */
1914     	forall_aloc_in_effect_mknin_effect(dri->location,
1915     					   eff_r, dri->effect, eff_wr, 
1916     					   stmt.eff);
1917     
1918     	eff = effect_union(eff, stmt.eff);
1919     	eff = effect_union(eff, effect_rwr(aloc_qtype(dri->qtype)));
1920     
1921     	return mksinfo(eff);
1922           }
1923         default:
1924           fail_loc(s->loc, "Unexpected statement kind 0x%x\n", s->kind);
1925           break;
1926         }
1927     }

1929     static einfo analyze_expression(expression e, context context, effect env,
1930     				dd_list drinfolist)
1931     {
1932       einfo result;
1933       dd_list_pos cur;
1934     
1935       /*  printf("Analyzing ");
1936           prt_expression(e, 0);
1937           printf(" in %s\n", context_tostring(context));*/
1938     
1939       /* deep restrict check */
1940       dd_scan(cur, drinfolist)
1941         {
1942           drinfo dri = DD_GET(drinfo, cur);
1943           if (equal_expressions(e, dri->expression))
1944     	{
1945     	  /* if r context, then must dereference once. */
1946     	  if (context == rpos)
1947     	    {
1948     	      assert(qtype_pointer(dri->rqtype));
1949     	      e->drinfo = dri;
1950     	      e->qtype = points_to_qtype(dri->rqtype);
1951     	      return mkeinfo(e->qtype,
1952     			     effect_r(aloc_qtype(dri->rqtype)),
1953     			     FALSE);
1954     	    }
1955     	  else
1956     	    {
1957     	      e->drinfo = dri;
1958     	      e->qtype = dri->rqtype;
1959     	      return mkeinfo(e->qtype, effect_empty, FALSE);
1960     	    }
1961     	}
1962         }
1963       e->drinfo = NULL;
1964     
1965       switch(e->kind) {
1966       case kind_comma:
1967         {
1968           comma c = CAST(comma, e);
1969           expression e2;
1970           effect eff;
1971     
1972           eff = effect_empty;
1973           scan_expression (e2, c->arg1)
1974     	{
1975     	  result = analyze_expression(e2, e2->next ? rpos : context, env,
1976     				      drinfolist);
1977     	  eff = effect_union(eff, result.eff);
1978     	}
1979           if (!result.qt)
1980     	fail_loc(c->loc, "Empty comma expression\n", 0);
1981           /* XXX: cparser has weird rule here, but I don't think it's right:
1982     	 if (qtype_array(result))
1983     	   result = default_conversion_qtype(result); */
1984           result = mkeinfo(result.qt, eff, FALSE);
1985         };
1986         break;
1987       case kind_sizeof_type:
1988         {
1989           qtype qt;
1990     
1991           assert(context == rpos);
1992           qt = mkqtype_size_t(make_qvar("sizeof_type", e->loc, FALSE));
1993           result = mkeinfo(qt, effect_empty, FALSE);
1994         }
1995         break;
1996       case kind_alignof_type:
1997         {
1998           qtype qt;
1999     
2000           assert(context == rpos);
2001           qt = mkqtype_size_t(make_qvar("alignof_type", e->loc, FALSE));
2002           result = mkeinfo(qt, effect_empty, FALSE);
2003         }
2004         break;
2005       case kind_label_address:
2006         {
2007           qtype points_to, qt;
2008           aloc loc;
2009     
2010           assert(context == rpos);
2011           points_to = mkqtype_void(make_qvar("&&_p", e->loc,FALSE));
2012           loc = make_aloc("&&", points_to);
2013           mkleq_effect(effect_single(loc), global_env);
2014     
2015           qt = mkqtype_pointer(make_qvar("&&", e->loc, FALSE),
2016     			   points_to,
2017     			   loc);
2018     
2019           /* Add && location to global effect. */
2020           mkleq_effect(effect_alloc(aloc_qtype(qt)), global_effect);
2021     
2022           result = mkeinfo(qt, effect_empty, FALSE);
2023         }
2024         break;
2025       case kind_cast:
2026         {
2027           cast c = CAST(cast, e);
2028           einfo arg1;
2029           type cast_t;
2030           qtype qt;
2031           effect eff;
2032           
2033           arg1 = analyze_expression(c->arg1, context, env, drinfolist);
2034           eff = arg1.eff;
2035           cast_t = c->asttype->type;
2036           if (context == lpos || context == apos)
2037     	{
2038     	  type pt;
2039     
2040     	  assert(!type_function(cast_t));
2041     	  assert(!type_array(cast_t));
2042     
2043     	  pt = make_pointer_type(cast_t);
2044     	  qt = type_to_qtype(pt, "cast", c->loc);
2045     	}
2046           else
2047     	qt = type_to_qtype(cast_t, "cast", c->loc);
2048     
2049           /* ad-hoc malloc handling in action */
2050           if (arg1.ismalloc)
2051     	eff = effect_union(defn_effect_qtype(qt), eff);
2052     
2053           if (flag_casts_preserve &&
2054     	  !type_void(cast_t) && /* cast to void is special -- it means,
2055     				   ``ignore this result'' */
2056     	  !type_contains_user_quals(cast_t))
2057     	/* If cast_t doesn't have explicit user-defined qualifiers, */
2058     	if (mkeq_qtype_cast(c->loc, qt, arg1.qt))
2059     	  report_qerror(c->loc, sev_err,
2060     			"Incompatible qualifiers at cast");
2061     
2062           result = mkeinfo(qt, eff, FALSE);
2063         };
2064         break;
2065       case kind_cast_list:
2066         {
2067           /* XXX Fix! */
2068           cast_list cl = CAST(cast_list, e);
2069           qtype qt;
2070           sinfo init;
2071     
2072           assert(context == rpos);
2073           qt = type_to_qtype(cl->type, "castlist", cl->loc);
2074           assert(!cl->astqtype);
2075           cl->astqtype = qt;
2076           init = analyze_init(cl->loc, qt, cl->init_expr, env, drinfolist);
2077     
2078           qt = default_conversion_qtype(qt);
2079           result = mkeinfo(qt, init.eff, FALSE);
2080         };
2081         break;
2082       case kind_conditional:
2083         {
2084           conditional c = CAST(conditional, e);
2085           einfo cond, arg1, arg2;
2086           effect eff;
2087           qtype qt;
2088     
2089           cond = analyze_expression(c->condition, rpos, env, drinfolist);
2090           eff = cond.eff;
2091           if (c->arg1)
2092     	{
2093     	  arg1 = analyze_expression(c->arg1, context, env, drinfolist);
2094     	  eff = effect_union(eff, arg1.eff);
2095     	}
2096           else
2097     	/* gcc extension:  if you omit the middle op, evaluates to
2098     	   guard when guard is true, and guard is not reevaluated */
2099     	arg1 = cond;
2100           arg2 = analyze_expression(c->arg2, context, env, drinfolist);
2101           eff = effect_union(eff, arg2.eff);
2102     
2103           if (context == rpos)
2104     	qt = type_to_qtype(c->type, "lub", c->loc);
2105           else
2106     	qt = type_to_qtype(make_pointer_type(c->type), "lub", c->loc);
2107     
2108           if (qtype_void(arg1.qt))
2109     	mkleq_assign_convert(c->loc, arg1.qt, qt);
2110           else if (qtype_void(arg2.qt))
2111     	mkleq_assign_convert(c->loc, arg2.qt, qt);
2112           else if (mkleq_assign_convert(c->loc, arg1.qt, qt) ||
2113     	       mkleq_assign_convert(c->loc, arg2.qt, qt))
2114     	report_qerror(c->loc, sev_err,
2115     		      "incompatible branches of conditional");
2116           result = mkeinfo(qt, eff, FALSE);
2117         };
2118         break;
2119       case kind_identifier:
2120         {
2121           identifier id = CAST(identifier, e);
2122           data_declaration root;
2123           qtype qt;
2124     
2125           root = root_ddecl(id->ddecl);
2126           if (is_op_name(root->name))
2127     	report_qerror(id->loc, sev_err,
2128     		      "use of built-in operator name");
2129           mark_used(root); /* We need this because we're marking the *root* */
2130     
2131           if (id->parent->kind != kind_function_call ||
2132     	  CAST(function_call, id->parent)->arg1 != e)
2133     	/* Mark if not in the callee position of a function call */
2134     	mark_address_taken(root);
2135     
2136           qt = root->qtype;
2137     
2138           /*printf("Aloc of %s is ",id->cstring.data);
2139             print_aloc(printf,aloc_qtype(qt));
2140             printf("\n");*/
2141     
2142           if (!qt)
2143     	{
2144     	  /* Only functions can be implicitly declared.  But there are also
2145     	     some special gcc identifiers, notably __FUNCTION__ and
2146     	     __PRETTY_FUNCTION__, that also don't need to be declared. */
2147     	  if ((type_function(id->ddecl->type) &&
2148     	       id->ddecl->ftype == function_implicit) ||
2149     	      !strcmp(id->ddecl->name, "__FUNCTION__") ||
2150     	      !strcmp(id->ddecl->name, "__PRETTY_FUNCTION__") ||
2151     	      !strcmp(id->ddecl->name, "=va_arg"))
2152     	    {
2153     	      __add_ddecl_qtype(root, TRUE);
2154     	      qt = root->qtype;
2155     
2156     	      /* 2p: These are gcc artifacts.  Add the effect of their
2157     		 allocation to the global effect. */
2158     	      mkleq_effect(defn_effect_qtype(qt), global_effect);
2159     	    }
2160     	  else
2161     	    fail_loc(id->loc, "Use before declaration\n", 0);
2162     	}
2163     
2164           if (qtype_scheme(qt))
2165     	qt = instantiate_qtype(qt, id->loc);
2166           if (flag_pam_mode || flag_print_results)
2167     	pam_add_overlay_file(id->loc, root->name, qt);
2168           if (root->kind == decl_constant)
2169     	{
2170     	  /* Constants are always rtypes */
2171     	  if (context != rpos)
2172     	    {
2173     	      /* The front end should catch this */
2174     	      report_qerror(id->loc, sev_err, "invalid lvalue");
2175     	      /*qt = mkqtype_pointer(make_qvar("__err", id->loc, FALSE),
2176     		qt);*/
2177     	      qt = error_qtype;
2178     	    }
2179     	  result = mkeinfo(qt, effect_empty, FALSE);
2180     	}
2181           else
2182     	{
2183     	  result = put_id_in_context(id->loc,
2184     				     mkeinfo(qt, effect_empty, FALSE),
2185     				     context);
2186     	}
2187         };
2188         break;
2189       case kind_compound_expr:
2190         {
2191           compound_expr ce = CAST(compound_expr, e);
2192           compound_stmt cs = CAST(compound_stmt, ce->stmt);
2193           statement cur_stmt;
2194           declaration d;
2195           qtype qt;
2196           effect eff, local_env;
2197           dd_list new_drinfolist, old_drinfolist;
2198     
2199           old_drinfolist = drinfolist;
2200           new_drinfolist = dd_copy(parse_region, drinfolist);
2201     
2202     
2203     
2204           cs->enclosing_cs = cur_cs;
2205           cur_cs = cs;
2206     
2207           if (cs->id_labels)
2208     	fail_loc(cs->loc, "Unimplemented: id_labels\n", 0);
2209     
2210           eff = effect_empty;
2211           local_env = env;
2212     
2213           /* Analyze the declarations in the block */
2214           scan_declaration(d, cs->decls)
2215     	{
2216     	  dinfo next;
2217     
2218     	  assert(d->kind != kind_asm_decl); /*asm_decl only at toplevel */
2219     	  next = analyze_declaration(d, local_env, new_drinfolist);
2220     	  eff = effect_union(eff, next.eff);
2221     	  local_env = effect_union(local_env, next.alocs);
2222     	}
2223     
2224           cs->scope_env = local_env;
2225           mark_effect_interesting(cs->scope_env);
2226     
2227           /* Analyze the statements in the block.  Analyze all but the
2228              last one. */
2229           cur_stmt = cs->stmts;
2230           while (cur_stmt && cur_stmt->next)
2231     	{
2232     	  sinfo next = analyze_statement(cur_stmt, local_env, new_drinfolist);
2233     	  eff = effect_union(eff, next.eff);
2234     	  cur_stmt = CAST(statement, cur_stmt->next);
2235     	}
2236     
2237           /* Now analyze the last statement (if there is one), and
2238              compute the type of the expression. */
2239           if (cur_stmt && is_expression_stmt(cur_stmt))
2240     	{
2241     	  einfo next;
2242     
2243     	  next = analyze_expression(CAST(expression_stmt, cur_stmt)->arg1,
2244     				    context,
2245     				    local_env, new_drinfolist);
2246     
2247     	  eff = effect_union(eff, next.eff);
2248     	  qt = next.qt;
2249     	}
2250           else
2251     	{
2252     	  /* Type is void */
2253     	  qt = mkqtype_void(make_qvar("compound", ce->loc, FALSE));
2254     	  if (cur_stmt)
2255     	    {
2256     	      sinfo next = analyze_statement(cur_stmt, local_env, 
2257     					     new_drinfolist);
2258     	      eff = effect_union(eff, next.eff);
2259     	    }
2260     	}
2261     
2262           /* Add constraints from effects */
2263           eff = enforce_restricts(cs->decls, eff, qt);
2264           enforce_readonly(cs->decls, eff);
2265     
2266           /* Filter out non-escaping effects */
2267           ce->filter_env = filter_effects(qt, env);
2268           mark_effect_interesting(ce->filter_env);
2269           eff = effect_inter(eff, ce->filter_env);
2270     
2271           cur_cs = cs->enclosing_cs;
2272     
2273           result = mkeinfo(qt, eff, FALSE);
2274         };
2275         break;
2276       case kind_function_call:
2277         {
2278           function_call fc = CAST(function_call, e);
2279           einfo f_info, magic_info;
2280           qtype fqt;
2281           qtypelist args;
2282           expression arg;
2283           qtypelist_scanner args_qs;
2284           int i;
2285           effect eff;
2286     
2287           /* Extract the function type of the expr in fun posn.  Note
2288              that arg1 must be a pointer to a function.  From C9X:
2289     
2290     	 [#1]  The  expression  that  denotes  the called function
2291     	 shall have  type  pointer  to  function  returning  void  or
2292     	 returning an object type other than an array type. */
2293     
2294           assert(context == rpos);
2295           if (fc->va_arg_call)
2296     	{
2297     	  qtype va_arg_qtype;
2298     
2299     	  /* Magic call to __builtin_va_arg(args, fc->va_arg_call) */
2300     	  va_arg_qtype = type_to_qtype(fc->va_arg_call->type,
2301     				       "_va_arg",
2302     				       fc->loc);
2303     
2304     	  result = mkeinfo(va_arg_qtype, effect_empty, FALSE);
2305     	  break;
2306     	}
2307     
2308           f_info = analyze_expression(fc->arg1, rpos, env, drinfolist);
2309           fqt = points_to_qtype(f_info.qt);
2310           args = arg_qtypes(fqt);
2311     
2312           magic_info = analyze_magic_function_call(fc, fqt, env, drinfolist);
2313           if (magic_info.qt)
2314     	{
2315     	  magic_info.eff = effect_union(f_info.eff, magic_info.eff);
2316     	  result = magic_info;
2317     	  break;
2318     	}
2319     
2320           eff = effect_union(f_info.eff, effect_qtype(fqt));
2321     
2322           /* Analyze each argument, making a constrant with the
2323     	 corresponding formal paramter */
2324           qtypelist_scan(args, &args_qs);
2325           i = 1;
2326           scan_expression(arg, fc->args)
2327     	{
2328     	  einfo arg_info;
2329     	  qtype formal_qt;
2330     	  bool warned_oldstyle;
2331     
2332     	  arg_info = analyze_expression(arg, rpos, env, drinfolist);
2333     	  eff = effect_union(eff, arg_info.eff);
2334     	  formal_qt = qtypelist_next(&args_qs);
2335     	  warned_oldstyle = FALSE;
2336     
2337     	  if (!formal_qt)
2338     	    {
2339     	      if (!qtype_varargs(fqt) && !qtype_oldstyle(fqt))
2340     		report_qerror(arg->loc, sev_err,
2341     			      "too many actual arguments for function "
2342     			      "call.\n");
2343     	      else if (qtype_oldstyle(fqt) && !warned_oldstyle)
2344     		{
2345     		  /* Warn about calls to oldstyle functions, but
2346     		     not __builtins */
2347     		  if (fc->arg1->kind == kind_identifier)
2348     		    {
2349     		      identifier fid = CAST(identifier, fc->arg1);
2350     		      if (strstr(fid->cstring.data, "__builtin") !=
2351     			  fid->cstring.data)
2352     			report_qerror(fc->loc, sev_warn,
2353     				      "warning: ignoring argument types "
2354     				      "in call to oldstyle function");
2355     		    }
2356     		  warned_oldstyle = TRUE;
2357     		}
2358     
2359     	      /* If func is varargs, then only add constraints if func
2360     	         has a varargs qualifier. */
2361     	      if (qtype_varargs(fqt) && vqual_qtype(fqt))
2362     		{
2363     		  qual vq;
2364     
2365     		  vq = vqual_qtype(fqt);
2366     		  if (varargs_constrain_quals_qtype(arg->loc,
2367     						    arg_info.qt, vq))
2368     		    report_qerror(arg->loc, sev_err,
2369     				  "type of actual argument %d doesn't match "
2370     				  "varargs qualifier", i);
2371     		}
2372     	    }
2373     	  else
2374     	    if (mkleq_assign_convert(arg->loc, arg_info.qt, formal_qt))
2375     	      report_qerror(arg->loc, sev_err,
2376     			    "type of actual argument %d doesn't match "
2377     			    "type of formal", i);
2378     	  i++;
2379     	}
2380           if (qtypelist_next(&args_qs))
2381     	fail_loc(e->loc, "Not enough actual arguments.\n", 0);
2382     
2383           result = mkeinfo(return_qtype(fqt), eff, FALSE);
2384         };
2385         break;
2386       case kind_array_ref:
2387         {
2388           array_ref ar = CAST(array_ref, e);
2389           expression array, plus, star_plus;
2390     
2391           if (type_array(ar->arg1->type))
2392     	array = ar->arg1;
2393           else
2394     	array = ar->arg2;
2395     
2396           array->lvalue = TRUE; /* XXX: Hack to fix problem
2397     				  w/default_conversion */
2398           array->cst = NULL; /* XXX: Hack to fix problem w/default_conversion */
2399     
2400           plus = make_binary(ar->loc, kind_plus, ar->arg1, ar->arg2);
2401           star_plus = make_dereference(ar->loc, plus);
2402           assert(!ar->alt);
2403           ar->alt = star_plus;
2404     
2405           result = analyze_expression(star_plus, context, env, drinfolist);
2406         };
2407         break;
2408       case kind_field_ref:
2409         {
2410           /* We unify the qualifiers on the structure and the qualifiers
2411     	 on the field.  For now.  See previous versions for other
2412     	 approaches. */
2413           field_ref fr = CAST(field_ref, e);
2414           einfo arg1;
2415           qtype field_qt;
2416           effect eff;
2417     
2418           arg1 = analyze_expression(fr->arg1, context, env, drinfolist);
2419           eff = arg1.eff;
2420     
2421           /*** Compute field qtypes ***/
2422     
2423           if (qtype_error(arg1.qt))
2424     	field_qt = error_qtype;
2425           else
2426     	{
2427     	  qtype qt, ag_qt;
2428     
2429     	  qt = arg1.qt;
2430     	  if (context == lpos || context == apos)

2545     static einfo analyze_unary_expression(unary e, context context, effect env,
2546     				      dd_list drinfolist)
2547     {
2548       operator op;
2549     
2550       op = find_op_kind(e->kind);  /* Look up the signature of this operator */
2551     
2552       switch (e->kind)
2553         {
2554         case kind_dereference:
2555           {
2556     	/* Analyze the subexpression in an r-context; this will return
2557     	   an l-type, so either return that if we're in an l-context or
2558     	   deref it if we're in an r-context.  Function are special --
2559     	   *f is allowed even if f is a function and not a function
2560     	   pointer type. */
2561     	einfo arg1, result;
2562     	effect eff;
2563     
2564     	arg1 = analyze_expression(e->arg1, rpos, env, drinfolist);
2565     	if (!qtype_pointer(arg1.qt))
2566     	  {
2567     	    report_qerror(e->loc, sev_err,
2568     			  "dereference of non-pointer type (probably a "
2569     			  "type incompatibility across files)");
2570     	    return mkeinfo(error_qtype, effect_empty, FALSE);
2571     	  }
2572     	assert(qtype_pointer(arg1.qt));
2573     	eff = effect_union(arg1.eff, effect_r(aloc_qtype(arg1.qt)));
2574     	if (context == lpos || context == apos)
2575     	  /* *e in an l-context silently converts a (pointer) r-type
2576     	     to an l-type */
2577     	  result = mkeinfo(arg1.qt, eff, FALSE);
2578     	else if (context == rpos)
2579     	  {
2580     	    if (qtype_function(arg1.qt))
2581     	      /*return arg1_qtype;*/
2582     	      fail_loc(e->loc, "Unexpected function in rpos", 0);
2583     	    else
2584     	     result = mkeinfo(default_conversion_qtype(points_to_qtype(arg1.qt)),
2585     			      eff, FALSE);
2586     	  }
2587     	else
2588     	  fail_loc(e->loc, "Unexpected context %d\n", context);
2589     
2590     	if (op)
2591     	  {
2592     	    einfo result_rpos;
2593     
2594     	    if (context == rpos)
2595     	      result_rpos = result;
2596     	    else
2597     	      result_rpos = mkeinfo(default_conversion_qtype(points_to_qtype(arg1.qt)),
2598     				    eff, FALSE);
2599     
2600     	    if (add_op_constraints(op, result_rpos, arg1, dummy_einfo,
2601     				   e->loc))
2602     	      report_qerror(e->loc, sev_err,
2603     			    "Operands incompatible with operator definition");
2604     
2605     	  }
2606     
2607     	return result;
2608           }
2609           break;
2610         case kind_address_of:
2611           {
2612     	/* &e can only appear in an r-context.  We analyze the
2613     	   subexpression in an l-context (thus the subexpression type
2614     	   with have a ptr at the outside) and silently convert the
2615     	   l-type to an r-type.  On the other hand, taking the address
2616     	   of a function type is a no-op. */
2617     	if (context != rpos)
2618     	  fail_loc(e->loc, "Unexpected & in non rpos\n", 0);
2619     	if (type_function(e->arg1->type))
2620     	  return analyze_expression(e->arg1, context, env, drinfolist);
2621     	else
2622     	  return analyze_expression(e->arg1, apos, env, drinfolist);
2623           }
2624           break;
2625         case kind_extension_expr:
2626           {
2627     	return analyze_expression(e->arg1, context, env, drinfolist);
2628           }
2629           break;
2630         case kind_sizeof_expr:
2631           {
2632     	qtype qt;
2633     
2634     	assert(context == rpos);
2635     	/*analyze_expression(e->arg1, rpos); (arg1 not evaluated) */
2636     	qt = mkqtype_size_t(make_qvar("sizeof", e->loc, FALSE));
2637     	return mkeinfo(qt, effect_empty, FALSE);
2638           }
2639           break;
2640         case kind_alignof_expr:
2641           {
2642     	qtype qt;
2643     
2644     	assert(context == rpos);
2645     	/*analyze_expression(e->arg1, rpos); (arg1 not evaluated) */
2646     	qt = mkqtype_size_t(make_qvar("alignof", e->loc, FALSE));
2647     	return mkeinfo(qt, effect_empty, FALSE);
2648           }
2649           break;
2650         case kind_realpart:
2651         case kind_imagpart:
2652           {
2653     	einfo arg1;
2654     
2655     	assert(context == rpos);
2656     	arg1 = analyze_expression(e->arg1, rpos, env, drinfolist);
2657     	assert(qtype_complex(arg1.qt));
2658     	return mkeinfo(complex_base_qtype(arg1.qt), arg1.eff, FALSE);
2659           }
2660           break;
2661         case kind_unary_minus:
2662         case kind_unary_plus:
2663         case kind_conjugate:
2664         case kind_bitnot:
2665           {
2666     	/* These operations don't change types.  Note we assume
2667     	   that compatibility with these ops has already been
2668     	   tested. */
2669     	assert(context == rpos);
2670     	return analyze_expression(e->arg1, rpos, env, drinfolist);
2671           }
2672           break;
2673         case kind_not:
2674           {
2675     	einfo arg1;
2676     	qtype qt;
2677     
2678     	assert(context == rpos);
2679     	arg1 = analyze_expression(e->arg1, rpos, env, drinfolist);
2680     	qt = mkqtype_bool(make_qvar("!", e->loc, FALSE));
2681     	return mkeinfo(qt, arg1.eff, FALSE);
2682           }
2683           break;
2684         case kind_preincrement:
2685         case kind_postincrement:
2686         case kind_predecrement:
2687         case kind_postdecrement:
2688           {
2689     	einfo arg1;
2690     	effect eff;
2691     
2692     	assert(context == rpos);
2693     	arg1 = analyze_expression(e->arg1, lpos, env, drinfolist);
2694     	qtype_mklhs_nonconst(e->loc, arg1.qt);
2695     	eff = effect_union(arg1.eff, effect_wr(aloc_qtype(arg1.qt)));
2696     	return mkeinfo(points_to_qtype(arg1.qt),
2697     		       eff, FALSE);
2698           }
2699           break;
2700         default:
2701           fail_loc(e->loc, "Unexpected unary op kind 0x%x\n", e->kind);
2702         }
2703     }

2721     static einfo analyze_binary_expression(binary e, context context, effect env,
2722     				       dd_list drinfolist)
2723     {
2724       einfo arg1, arg2;
2725       operator op;
2726       effect eff;
2727     
2728       assert(context == rpos);
2729       op = find_op_kind(e->kind);  /* Look up the signature of this operator */
2730       if (isassignment(e))
2731         arg1 = analyze_expression(e->arg1, lpos, env, drinfolist);
2732       else
2733         arg1 = analyze_expression(e->arg1, rpos, env, drinfolist);
2734       arg2 = analyze_expression(e->arg2, rpos, env, drinfolist);
2735       eff = effect_union(arg1.eff, arg2.eff);
2736     
2737       if (qtype_error(arg1.qt) || qtype_error(arg2.qt))
2738         return mkeinfo(error_qtype, eff, FALSE);
2739     
2740       switch (e->kind)
2741         {
2742         case kind_plus:
2743           {
2744     	qtype qt;
2745     
2746     	if (qtype_pointer(arg1.qt) && qtype_pointer(arg2.qt))
2747     	  fail_loc(e->loc, "Unexpected pointer addition\n", 0);
2748     
2749     	if (qtype_pointer(arg1.qt) && !qtype_pointer(arg2.qt))
2750     	  qt = arg1.qt;
2751     	else if (!qtype_pointer(arg1.qt) && qtype_pointer(arg2.qt))
2752     	  qt = arg2.qt;
2753     	else
2754     	  {
2755     	    /* No lub here -- when would lub ever make sense? */
2756     	    if (mkeq_qtype(e->loc, arg1.qt, arg2.qt))
2757     	      report_qerror(e->loc, sev_err,
2758     			    "incompatible operands of ``%s''",
2759     			    binary_op_name(e->kind));
2760     	    qt = arg1.qt;
2761     	  }
2762     
2763     	return mkeinfo(qt, eff, FALSE);
2764           }
2765           break;
2766         case kind_minus:
2767           {
2768     	qtype qt;
2769     
2770     	if (qtype_pointer(arg1.qt) && qtype_pointer(arg2.qt))
2771     	  qt = mkqtype_ptrdiff_t(make_qvar("__ptr diff", e->loc, FALSE));
2772     	else if (qtype_pointer(arg1.qt) && !qtype_pointer(arg2.qt))
2773     	  qt = arg1.qt;
2774     	else if (!qtype_pointer(arg1.qt) && qtype_pointer(arg2.qt))
2775     	  qt = arg2.qt;
2776     	else
2777     	  {
2778     	    /* No lub here -- when would lub ever make sense? */
2779     	    if (mkeq_qtype(e->loc, arg1.qt, arg2.qt))
2780     	      report_qerror(e->loc, sev_err,
2781     			    "incompatible operands of ``%s''",
2782     			    binary_op_name(e->kind));
2783     	    qt = arg1.qt;
2784     	  }
2785     
2786     	return mkeinfo(qt, eff, FALSE);
2787           }
2788           break;
2789         case kind_modulo:
2790         case kind_lshift:
2791         case kind_rshift:
2792           {
2793     	return mkeinfo(arg1.qt, eff, FALSE);
2794           }
2795         case kind_modulo_assign:
2796         case kind_lshift_assign:
2797         case kind_rshift_assign:
2798           {
2799     	qtype_mklhs_nonconst(e->loc, arg1.qt);
2800     	return mkeinfo(points_to_qtype(arg1.qt),
2801     		       effect_union(eff, effect_wr(aloc_qtype(arg1.qt))), 
2802     		       FALSE);
2803           }
2804         case kind_times:
2805         case kind_divide:
2806         case kind_bitand:
2807         case kind_bitor:
2808         case kind_bitxor:
2809           {
2810     	/* Got rid of lub -- when would lub make sense? */
2811     	if (mkeq_qtype(e->loc, arg1.qt, arg2.qt))
2812     	  report_qerror(e->loc, sev_err,
2813     			"incompatible operands of ``%s''",
2814     			binary_op_name(e->kind));
2815     	return mkeinfo(arg1.qt, eff, FALSE);
2816           }
2817           break;
2818         case kind_leq:
2819         case kind_geq:
2820         case kind_lt:
2821         case kind_gt:
2822         case kind_eq:
2823         case kind_ne:
2824         case kind_andand:
2825         case kind_oror:
2826           {
2827     	qtype qt;
2828     	einfo result;
2829     
2830     	qt = mkqtype_bool(make_qvar("__boolop", e->loc, FALSE));
2831     
2832     	/* XXX: Ick! */
2833     	if ((e->kind == kind_eq || e->kind == kind_ne) &&
2834     	    ((file_pointer_qtype(arg1.qt) && definite_null(e->arg2)) ||
2835     	     (definite_null(e->arg1) && file_pointer_qtype(arg2.qt))))
2836     	  /* We'll do an assign in the flow-sensitive pass, so add an
2837     	     effect here */
2838     	  {
2839     	    aloc al;
2840     
2841     	    if (file_pointer_qtype(arg1.qt))
2842     	      al = aloc_qtype(arg1.qt);
2843     	    else
2844     	      al = aloc_qtype(arg2.qt);
2845     
2846     	    mark_aloc_interesting(al);
2847     	    eff = effect_union(eff, effect_wr(al));
2848     	  }
2849     
2850     	result = mkeinfo(qt, eff, FALSE);
2851     	if (op && add_op_constraints(op, result, arg1, arg2, e->loc))
2852     	  report_qerror(e->loc, sev_err,
2853     			"Operands incompatible with operator definition");
2854     	return result;
2855           }
2856         case kind_plus_assign:
2857         case kind_minus_assign:
2858           {
2859     	qtype arg1_pt;
2860     
2861     	arg1_pt = points_to_qtype(arg1.qt);
2862     
2863     	if ((qtype_pointer(arg1_pt) && qtype_pointer(arg2.qt)))
2864     	  fail_loc(e->loc, "Unexpected pointer addition\n", 0);
2865     
2866     	if (qtype_pointer(arg1_pt) && !qtype_pointer(arg2.qt))
2867     	  {
2868     	    /* In p += count, no interaction between p and count's
2869     	       qualifiers */
2870     	    qtype_mklhs_nonconst(e->loc, arg1.qt);
2871     	    return mkeinfo(arg1_pt,
2872     			   effect_union(eff, effect_wr(aloc_qtype(arg1.qt))),
2873     			   FALSE);
2874     	  }
2875           }
2876           /* else fall through to next case */
2877         case kind_times_assign:
2878         case kind_divide_assign:
2879         case kind_bitand_assign:
2880         case kind_bitor_assign:
2881         case kind_bitxor_assign:
2882         case kind_assign:
2883           {
2884     	if (arg2.ismalloc)
2885     	  eff = effect_union(defn_effect_qtype(points_to_qtype(arg1.qt)), eff);
2886     
2887     	qtype_mklhs_nonconst(e->loc, arg1.qt);
2888     	mark_aloc_interesting(aloc_qtype(arg1.qt));
2889     	if (mkleq_assign_convert(e->loc, arg2.qt, points_to_qtype(arg1.qt)))
2890     	  report_qerror(e->loc, sev_err,
2891     			"incompatible types in assignment");
2892     
2893     	return mkeinfo(points_to_qtype(arg1.qt),
2894     		       effect_union(eff, effect_wr(aloc_qtype(arg1.qt))),
2895     		       FALSE);
2896           }
2897         default:
2898           fail_loc(e->loc, "Unexpected binary op kind 0x%x\n", e->kind);
2899         }
2900     }

2902     /* Given the type of an identifier, convert it to the right type in
2903        the current context. */
2904     einfo put_id_in_context(location loc, einfo ei, context c)
2905     {
2906       qtype result_qt;
2907       effect result_eff;
2908     
2909       result_eff = ei.eff;
2910       if (qtype_error(ei.qt))
2911         result_qt = error_qtype;
2912       else if (qtype_array(ei.qt)) {
2913         /* Arrays are always in r-positions, and when they're used,
2914            they're actually pointers.  Well, not when they're used in
2915            sizeof and as the target of &, but you can't take & of an
2916            array, and in the other places we don't care (see C9X).
2917            EXCEPT you can take the address of an array, so that
2918            is considered an apos.  Yuck. */
2919         if (c == rpos)
2920           result_qt = default_conversion_qtype(ei.qt);
2921         else if (c == apos)
2922           result_qt = mkqtype_pointer(make_fresh_qvar("q", loc),
2923     				  ei.qt,
2924     				  array_aloc_qtype(ei.qt));
2925         else
2926           fail("Unexpected context %d\n", c);
2927       }
2928       else if (qtype_function(ei.qt)) {
2929         assert(c == rpos);
2930         /* Function are promoted to function pointers.  Hack: If we
2931            are in an & context, then still return a pointer.  The &
2932            only changes an l-type to an r-type by doing nothing, thus
2933            it will not change the fun ptr type! */
2934         result_qt = default_conversion_qtype(ei.qt);
2935       }
2936       else if (c == lpos || c == apos)
2937         result_qt = ei.qt;
2938       else if (c == rpos)
2939         {
2940           if (qtype_pointer(ei.qt))
2941     	{
2942     	  result_qt = points_to_qtype(ei.qt);
2943     	  result_eff = effect_union(result_eff, /* Implicit deref */
2944     				    effect_r(aloc_qtype(ei.qt)));
2945     	}
2946           else
2947     	fail("Unexpected qtype\n");
2948     
2949         }
2950       else fail_loc(loc, "Unexpected context %d\n", c);
2951     
2952       return mkeinfo(result_qt, result_eff, FALSE);
2953     }

3011     bool mkNonConst_aggregate(location loc, qtype ag_qtype)
3012     {
3013       field_scanner fields;
3014       qtype field_qtype, contents_qtype;
3015       bool result = FALSE;
3016       
3017       if (!nonconst_qual)
3018         return FALSE;
3019     
3020       field_scan(ag_qtype, &fields);
3021       while ((field_qtype = field_next(&fields)))
3022         {
3023           /* const *does* propagate through arrays.  This makes sense
3024       	 because arrays nested in structs are part of the same
3025       	 struct object, whereas pointers point to other objects.
3026     	 
3027     	 So if we have an array(t) type, we move through the type
3028       	 until t is not an array. */
3029           if (qtype_array(field_qtype))
3030       	{
3031       	  contents_qtype = array_of_qtype(field_qtype);
3032       	  while (qtype_array(contents_qtype))
3033       	    {
3034       	      field_qtype = contents_qtype;
3035       	      contents_qtype = array_of_qtype(field_qtype);
3036       	    }
3037       	}
3038           else
3039       	contents_qtype = points_to_qtype(field_qtype);
3040           
3041           /* Now we make field_qtype (the most deeply nested array qtype,
3042       	 or the ptr qtype) non-const, and then if the contents is
3043       	 a struct/union we recurse. */
3044           result = mkNonConst_qual(loc, qual_qtype(field_qtype)) || result;
3045           if (qtype_aggregate(contents_qtype))
3046       	result = mkNonConst_aggregate(loc, contents_qtype) || result;
3047         }
3048       return result;
3049     }

3160        list of variable declarations), verify that none of the conditions
3161        imposed by restricts in dls are violated.  qt can be NULL,
3162        indicating the code has type void. */
3163     effect enforce_var_decl_restrict(variable_decl vd, effect eff, qtype qt)
3164     {
3165       if (vd && vd->rinfo)
3166         {
3167           mknin_effect(vd->loc, vd->rinfo->flocs, eff);
3168           /* report_qerror(vd->loc, sev_err,
3169     		      "restricted location accessed within scope of "
3170     		      "declaration"); */
3171     
3172           if (qt != NULL && !(eq_effect(vd->rinfo->rlocs, effect_empty)))
3173     	mknin_effect_qtype(vd->loc, vd->rinfo->rlocs, qt);
3174          
3175           /* report_qerror(vd->loc, sev_err,
3176     	 "restricted location escapes scope of declaration"); */
3177     
3178           /* We could also reset vd's qtype back to its unrestricted qtype,
3179     	 but what's the point? */
3180     
3181           eff = effect_union(eff, vd->rinfo->flocs);
3182         }
3183     
3184       eff = effect_union(eff, vd->arg1_eff);
3185       return eff;
