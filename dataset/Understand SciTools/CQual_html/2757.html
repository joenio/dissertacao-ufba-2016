19     #include "parser.h"
20     #include "input.h"
21     
22     /* Source file current line is coming from.  */
23     char *input_filename;
24     
25     /* Top-level source file.  */
26     char *main_input_filename;
27     
28     /* Line number in current source file.  */
29     int lineno;
30     
31     /* Column number in current source file. */
32     int filepos;
33     
34     /* Stream for reading from input file.  */
35     FILE *finput;
36     
37     /* Stack of currently pending input files.
38        The line member is not accurate for the innermost file on the stack.  */
39     struct file_stack *input_file_stack;
40     
41     /* Incremented on each change to input_file_stack.  */
42     int input_file_stack_tick;
43     
44     void set_input(FILE *f, char *filename)
45     {
46       input_file_stack = xmalloc(sizeof *input_file_stack);
47       input_file_stack->name = main_input_filename = input_filename = filename;
48       input_file_stack->line = lineno = 0;
49       input_file_stack->filepos = filepos = -1;
50       input_file_stack->indent_level = 0;
51       input_file_stack->next = NULL;
52       finput = f;
53     }
