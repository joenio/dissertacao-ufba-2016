201     /* Print out a string, adding escapes where necessary */
202     int pam_print_escaped_string(const char *s)
203     {
204       int count = 0;
205     
206       pam_putc('\"');
207       while (*s)
208       {
209         switch (*s)
210         {
211         case '\\': pam_printf("\\\\"); break;
212         case '\"': pam_printf("\\\""); break;
213         case '\n': pam_printf("\\n"); break;
214         default:
215           if (isprint(*s))
216     	pam_putc(*s);
217           else
218     	fail("Error: 0x%x not printable\n", *s);
219         }
220         s++;
221         count++;
222       }
223       pam_putc('\"');
224       return count;
225     }

241     typedef enum { pam_clear_window, pam_change_window, pam_set_lower_window,
242     	       pam_set_upper_window, pam_close_window, pam_split_window
243     } pam_op_kind;
244     
245     void pam_window_op(pam_op_kind op, const char *wname, const char *kind)
246     {
247       pam_putc('(');
248       switch (op)
249         {
250         case pam_clear_window: pam_printf("pam-clear-window"); break;
251         case pam_change_window: pam_printf("pam-change-window"); break;
252         case pam_set_lower_window: pam_printf("pam-set-window-down"); break;
253         case pam_set_upper_window: pam_printf("pam-set-window-up"); break;
254         case pam_close_window: pam_printf("pam-close-window"); break;
255         case pam_split_window: pam_printf("pam-split-window"); break;
256         default:
257           fail("Unexpected window op %d\n", op);
258         }
259       pam_printf(" (");
260       pam_print_escaped_string(wname);
261       pam_putc(' ');
262       pam_print_escaped_string(kind);
263       pam_printf("))\n");
264     }

485     /**************************************************************************
486      *                                                                        *
487      * PAM environments                                                       *
488      *                                                                        *
489      * Functions for remembering what to do when the user clicks on stuff.    *
490      *                                                                        *

493     #define pam_any_click ""
494     
495     /* Add a dispatch to the click dispatch table for buffer bname on
496        click cname.  If the click was already there, override it
497        with a new click. */
498     void pam_add_buffer_click(const char *bname, const char *cname,
499     			  click_func func, void *arg)
500     {
501       pam_click_map clicks;
502       pam_click c;
503     
504       if (!pam_buffer_map_lookup(buffer_clicks, bname, &clicks))
505         {
506           clicks = make_pam_click_map(pam_region, 37);
507           insist(pam_buffer_map_insert(buffer_clicks, bname, clicks));
508         }
509     
510       c = ralloc(pam_region, struct Click);
511       c->func = func;
512       c->arg = arg;
513     
514       pam_click_map_insert(clicks, cname, c);
515     }
516     
517     /* Remove a dispatch from the click dispatch table for buffer bname on
518        click cname.  Aborts if click wasn't already there. */
519     void pam_remove_buffer_click(const char *bname, const char *cname)
520     {
521       pam_click_map clicks;
522     
523       insist(pam_buffer_map_lookup(buffer_clicks, bname, &clicks));
524       insist(pam_click_map_remove(clicks, cname));
525     }
526     
527     /* Look up the click for a particular buffer.  Return NULL if
528        no click exists. */
529     pam_click pam_get_buffer_click(const char *bname, const char *cname)
530     {
531       pam_click_map clicks;
532       pam_click c;
533     
534       if (!pam_buffer_map_lookup(buffer_clicks, bname, &clicks))
535         return NULL;
536       if (!pam_click_map_lookup(clicks, cname, &c) &&
537           !pam_click_map_lookup(clicks, pam_any_click, &c))
538         return NULL;
539       return c;
540     }

878     static qual last_printed = NULL;
879     static int num_pos = 0, num_neg = 0, num_eq = 0;
880     
881     void pam_print_const_qual(qual q, void *arg)
882     {
883       assert(constant_qual(q));
884       pam_add_text(" ");
885       pam_add_qual_link(q, "");
886       switch (sign_qual(q))
887         {
888         case sign_pos: num_pos++; break;
889         case sign_neg: num_neg++; break;
890         case sign_eq: num_eq++; break;
891         default: fail("Unexpected qualifier sign %x\n", sign_qual(q));
892         }
893     }

895     void pam_print_edge(qual left, qual right, location loc, void *arg)
896     {
897       const char *s, *arrow;
898       static int left_size;
899       qual to_print;
900     
901       if (!last_printed)
902         {
903           qual init_qual;
904     
905           if (constant_qual(left))
906     	{
907     	  assert(!constant_qual(right));
908     	  init_qual = left;
909     	  to_print = right;
910     	  arrow = "<=";
911     	}
912           else
913     	{
914     	  assert(!constant_qual(left));
915     	  init_qual = right;
916     	  to_print = left;
917     	  arrow = ">=";
918     	}
919     
920           s = unique_name_qual(init_qual);
921           left_size = pam_add_qual_link(init_qual, s);
922           if (flag_ugly)
923     	left_size += pam_add_text(" %s ", s);
924           else
925     	left_size += pam_add_text(" ");
926         }
927       else
928         {
929           pam_add_text("%*s", left_size, "");
930     
931           if (eq_qual(last_printed, left))
932     	{
933     	  to_print = right;
934     	  arrow = "<=";
935     	}
936           else
937     	{
938     	  assert(eq_qual(last_printed, right));
939     	  to_print = left;
940     	  arrow = ">=";
941     	}
942         }
943     
944       if (loc)
945         {
946           const char *name;
947     
948           name = get_next_link();
949           pam_add_link(name, pam_color_8, arrow);
950           pam_add_buffer_click(pam_type_buffer,
951     			   name,
952     			   (click_func) pam_jump_file,
953     			   loc);
954         }
955       else
956         pam_add_text(arrow);
957       pam_add_text(" ");
958       s = unique_name_qual(to_print);
959       pam_add_qual_link(to_print, s);
960       if (flag_ugly)
961         pam_add_text(" %s\n", s);
962       else
963         pam_add_text("\n");
964       last_printed = to_print;
965     }

968     void pam_print_path_to_qual(qual c, qual q)
969     {
970       last_printed = NULL;
971       traverse_shortest_path_edges(q, c, pam_print_edge, NULL);
972     }

974     void pam_qual_mouse_2_click_func(const char *file, const char *name, void *arg)
975     {
976       void *addr;
977       qual q;
978     
979       assert(!arg);
980       if (sscanf(name, "%p", &addr) == 0)
981         {
982           pam_message("Invalid click on ``%s'' in buffer ``%s''", name, file);
983           return;
984         }
985       q = (qual) addr;
986       /*  print_qual_graph(q, "qual.dot", TRUE);*/
987       pam_close_buffer(pam_type_buffer);
988       pam_set_lower_buffer(pam_type_buffer);
989       pam_change_buffer(pam_type_buffer);
990       pam_clear_buffer(pam_type_buffer);
991       pam_blobs_begin();
992       pam_markup_buffer(pam_type_buffer);
993     
994       /* Reset our state */
995       num_pos = 0;
996       num_neg = 0;
997       num_eq = 0;
998     
999       /* Print the qualifier bounds */
1000       if (flag_ugly)
1001         pam_add_text("%s %s: ", name_qual(q), unique_name_qual(q));
1002       else
1003         pam_add_text("%s: ", name_qual(q));
1004       scan_qual_bounds(q, pam_print_const_qual, NULL);
1005       pam_add_text("\n\n");
1006     
1007       /* Print the path(s) to the bounds */
1008       /*pam_remove_buffer_clicks(pam_type_buffer);*/
1009       if (num_pos + num_neg + num_eq > 0)
1010         {
1011           if (num_neg <= 1 && num_pos <= 1 && num_eq == 0)
1012     	{
1013     	  /* If we've got at most one positive and one negative qualifier,
1014     	     do the standard tainting-style edge printing */
1015     	  last_printed = NULL;
1016     	  traverse_shortest_path_edges(q, NULL, pam_print_edge, NULL);
1017     	}
1018           else
1019     	/* Otherwise, print paths for each qualifier separately */
1020     	scan_qual_bounds(q, (qual_traverse_fn) pam_print_path_to_qual, q);
1021         }
1022       else
1023         pam_add_text("No qualifiers\n");
1024     
1025       pam_blobs_end();
1026     }

1068     /* Load in file and mark it up. */
1069     void pam_load_file(const char *click_name, const char *file_containing_link,
1070     		   const char *name, char *filename)
1071     {
1072       pam_file f;
1073       pam_overlay_map_scanner oms;
1074       pam_overlay_map_scanner_sorted omss;
1075       location loc;
1076       pam_overlay overlay;
1077       int numOverlays = 0;
1078     
1079       pam_close_file(filename);
1080       pam_set_upper_file(filename);
1081       pam_change_file(filename);
1082     
1083       if (!pam_file_map_lookup(pam_files, filename, &f))
1084         {
1085           pam_message("Asked to add overlays to unknown file ``%s''", filename);
1086           return;
1087         }
1088     
1089       if (f->loaded)
1090         /* Assume already loaded info */
1091         return;
1092     
1093       pam_message("Adding overlays to ``%s''", filename);
1094     
1095       pam_overlays_begin();
1096       pam_markup_file(filename);
1097       if (flag_pam_html)
1098         pam_overlay_map_scan_sorted(f->overlays, location_cmp, &omss);
1099       else
1100         pam_overlay_map_scan(f->overlays, &oms);
1101     
1102       while (flag_pam_html ?
1103     	 pam_overlay_map_next_sorted(&omss, &loc, &overlay) :
1104     	 pam_overlay_map_next(&oms, &loc, &overlay))
1105         {
1106           const char *color = NULL;
1107           int name_length = overlay->name ? strlen(overlay->name) : 1;
1108     
1109           if (overlay->kind & overlay_qtype)
1110     	color = combine_colors_pam(color,
1111     				   color_qtype(overlay->qtype, NULL));
1112     
1113           if (overlay->kind & overlay_fs_qtype)
1114     	color = combine_colors_pam(color,
1115     				   color_qtype(overlay->fs_qtype,
1116     					       overlay->store));
1117     
1118           if (!color && (overlay->kind & (overlay_qtype | overlay_fs_qtype)))
1119     	color = pam_color_noqual;
1120     
1121           if (overlay->kind & overlay_lin)
1122     	{
1123     	  if (lin_from_store(overlay->store, overlay->aloc) == 2)
1124     	    color = pam_color_6;
1125     	  else
1126     	    color = pam_color_7;
1127     	}
1128     
1129           if (color != NULL)
1130     	{
1131     	  if (overlay->kind & overlay_anchor)
1132     	    pam_add_overlink_anchor(ptr_to_ascii(overlay),
1133     				    ptr_to_ascii(overlay),
1134     				    color,
1135     				    loc->filepos + 1,
1136     				    loc->filepos + 1 + name_length);
1137     	  else
1138     	    pam_add_overlink(ptr_to_ascii(overlay),
1139     			     color,
1140     			     loc->filepos + 1, /* emacs off by one */
1141     			     loc->filepos + 1 + name_length);
1142     	}
1143           else
1144     	pam_add_anchor(ptr_to_ascii(overlay),
1145     		       loc->filepos + 1,
1146     		       loc->filepos + 1 + name_length);
1147     
1148           numOverlays++;
1149           if (numOverlays > 1000)
1150     	{
1151     	  pam_overlays_end();
1152     	  pam_overlays_begin();
1153     	  pam_markup_file(filename);
1154     	  numOverlays = 0;
1155     	}
1156         }
1157       pam_overlays_end();
1158       pam_message("done.");
1159       f->loaded = TRUE;
1160     }

1232        exist. */
1233     void pam_add_overlay_file(location loc, const char *idname, qtype qt)
1234     {
1235       pam_file f;

1239         {

1277         /* don't do this in standard PAM mode */
1278         return;
1279     
1280       if (!pam_file_map_lookup(pam_files, loc->filename, &f))
1281         {
1282           pam_add_file(loc->filename);
1283           insist(pam_file_map_lookup(pam_files, loc->filename, &f));
1284         }
1285     
1286       /*  printf("Adding overlay %s to %s\n", idname, files->filename);*/
1287       if (pam_overlay_map_lookup(f->overlays, loc, &overlay))
1288         {
1289           overlay->kind |= overlay_anchor;
1290         }
1291       else
1292         {
1293           overlay = ralloc(pam_region, struct Overlay);
1294           overlay->kind = overlay_anchor;
1295           overlay->name = NULL;
1296           overlay->qtype = NULL;
1297           overlay->fs_qtype = NULL;
1298           overlay->store = NULL;
1299           overlay->aloc = NULL;
1300     
1301           insist(pam_overlay_map_insert(f->overlays, loc, overlay));
1302         }
1303     }

1400         {

1406     	{

1417     	  fail("Unexpected severity %d\n", e->sev);
1418     	}
1419     
1420           if (e->loc)
1421     	{
1422     	  /* If there's a location for this error, add link */
1423     	  s = rsprintf(scratch_region, "%s:%ld", e->loc->filename,
1424     		       e->loc->lineno);

1429           if (e->detail)
1430     	pam_add_text("\n%s", e->detail);
1431           pam_add_text("\n\n");
1432         }
1433       deleteregion(scratch_region);
1434     }

1536     /* Reads at most size-1 characters from the standard input, storing
1537        the result in buf.  Stops reading characters after detecting an \n
1538        or an EOF (not included in buf), or when the buffer is full.
1539        Always null-terminates buf.  Returns the number of characters
1540        stored in the buffer, including \0. */
1541     int get_line(char *buf, int size)
1542     {
1543       int n = 0;
1544       /*  static FILE *foo = 0;*/
1545     
1546       assert(size >= 1);   /* Must be space for \0 */
1547       while (n < size - 1) /* Leave space for terminating \0 */
1548         {
1549           char c = pam_getc();
1550           if (c == EOF || c == '\n')
1551     	break;
1552           buf[n++] = c;
1553         }
1554       buf[n++] = '\0';
1555       /* if (!foo)
1556            foo = fopen("clicks", "w");
1557          fprintf(foo, "%s\n", buf);
1558          fflush(foo);*/
1559       return n;
1560     }

1562     /* Enter PAM mode, and loop on PAM events */
1563     void enter_pam_mode(void)
1564     {
1565       /* Disable buffering on stdout to make cygwin happy */
1566     #ifdef SETVBUF_REVERSED
1567       setvbuf(stdout, _IONBF, NULL, 0);
1568     #else
1569       setvbuf(stdout, NULL, _IONBF, 0);
1570     #endif
1571       pam_install_generic_buffer_clicks();
1572     
1573       /* Set up the results buffer */
1574       pam_printf("\n\n///\n");  /* Mysterious message to indicate analysis is
1575     			       over */
1576       pam_close_buffer(pam_results_buffer);
1577       pam_change_buffer(pam_results_buffer);
1578       pam_create_initial_window();
1579       if (qerrors)
1580         pam_message("Qualifier inference failed.");
1581       else
1582         pam_message("Qualifier inference successful.");
1583     
1584       /* Go into a loop, accepting commands from PAM */
1585       while (1)
1586         {
1587           static char buf[BUF_MAX];
1588           int bufSize;
1589           char *cmd;
1590     
1591           pam_waiting(); /* Tell PAM we're waiting for input */
1592           bufSize = 0; /* Number of characters actually in buffer */
1593           cmd = buf + bufSize;
1594           if ((bufSize += get_line(cmd, BUF_MAX-bufSize)) == BUF_MAX)
1595     	fail("PAM sent a command that was too big\n");
1596           if (!strcmp(buf, pam_click_msg) || !strcmp(buf, pam_shift_click_msg))
1597     	{
1598     	  char *name, *file, *type;
1599     
1600     	  /* Read name */
1601     	  name = buf + bufSize;;
1602     	  if ((bufSize += get_line(name, BUF_MAX-bufSize)) == BUF_MAX)
1603     	    fail("PAM sent a command that was too big\n");
1604     
1605     	  /* Read file */
1606     	  file = buf + bufSize;
1607     	  if ((bufSize += get_line(file, BUF_MAX-bufSize)) == BUF_MAX)
1608     	    fail("PAM sent a command that was too big\n");
1609     
1610     	  /* Read type */
1611     	  type = buf + bufSize;
1612     	  if ((bufSize += get_line(type, BUF_MAX-bufSize)) == BUF_MAX)
1613     	    fail("PAM sent a command that was too big\n");
1614     
1615     	  /* pam_message("Click! name=%s, file=%s, type=%s", name, file,
1616     	     type); */
1617     	  pam_click_dispatch(cmd, name, file, type);
1618     	}
1619           else if (!strcmp(buf, "quit"))
1620     	{
1621     	  pam_message("Analysis program exiting.");
1622     	  break;
1623     	}
1624           else if (!strcmp(buf, "file"))
1625     	fail("Unimplemented: PAM file\n");
1626           else
1627     	pam_message("Unrecognized command ``%s'' returned from PAM", buf);
1628         }
1629     }
