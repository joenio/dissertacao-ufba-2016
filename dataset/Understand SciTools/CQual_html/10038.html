19     /* Routines to find syntactic equivalences of two expressions modulo
20      easily recognizable semantic equivalenes */
21

31     bool equal_unary_expressions(unary e1, unary e2);
32     bool equal_binary_expressions(binary e1, binary e2);
33     
34     bool equal_expressions(expression e1, expression e2)
35     {
36       if (e1->kind != e2->kind)
37         return FALSE;
38     
39       switch (e1->kind) {
40       case kind_cast:
41         {
42           /* Ignore cast syntax */
43           cast c1 = CAST(cast, e1);
44           cast c2 = CAST(cast, e2);
45           return equal_expressions(c1->arg1, c2->arg1);
46         }
47         break;
48       case kind_conditional:
49         {
50           conditional c1 = CAST(conditional, e1);
51           conditional c2 = CAST(conditional, e2);
52           
53           if ((c1->arg1 && !c2->arg1) || (!c1->arg1 && c2->arg1))
54     	return FALSE;
55           if (!equal_expressions(c1->condition, c2->condition))
56     	return FALSE;
57           if (c1->arg1)
58     	{
59     	  assert(c2->arg1);
60     	  if (!equal_expressions(c1->arg1, c2->arg1))
61     	    return FALSE;
62     	}
63           return equal_expressions(c1->arg2, c2->arg2);
64         }
65         break;
66       case kind_identifier:
67         {
68           identifier id1 = CAST(identifier, e1);
69           identifier id2 = CAST(identifier, e2);
70           return root_ddecl(id1->ddecl) == root_ddecl(id2->ddecl);
71         }
72         break;
73       case kind_function_call:
74         {
75           function_call fc1 = CAST(function_call, e1);
76           function_call fc2 = CAST(function_call, e2);
77           expression arg1, arg2;
78     
79           if (!equal_expressions(fc1->arg1, fc2->arg1)) 
80     	return FALSE;
81     
82           arg2 = fc2->args;
83           scan_expression(arg1, fc1->args)
84     	{
85     	  if (!arg2 || !equal_expressions(arg1, arg2))
86     	    return FALSE;
87     	  
88     	  arg2 = CAST(expression, arg2->next);
89     	}
90           return arg2 ? FALSE : TRUE;
91         }
92       case kind_field_ref:
93         {
94           field_ref fr1 = CAST(field_ref, e1);
95           field_ref fr2 = CAST(field_ref, e2);
96     
97           return equal_expressions(fr1->arg1, fr2->arg1) &&
98     	!strcmp(fr1->cstring.data, fr2->cstring.data);
99         }
100       case kind_array_ref:
101         {
102           array_ref ar1 = CAST(array_ref, e1);
103           array_ref ar2 = CAST(array_ref, e2);
104           
105           return equal_expressions(ar1->arg1, ar2->arg1) &&
106     	equal_expressions(ar1->arg2, ar2->arg2);
107         }
108       case kind_lexical_cst:
109         {
110           lexical_cst lc1 = CAST(lexical_cst, e1);
111           lexical_cst lc2 = CAST(lexical_cst, e2);
112     
113           return !strcmp(lc1->cstring.data, lc2->cstring.data);
114         }
115       default:
116         if (is_unary(e1) && is_unary(e2))
117           return equal_unary_expressions(CAST(unary, e1), CAST(unary, e2));
118         else if (is_binary(e1) && is_binary(e2))
119           return equal_binary_expressions(CAST(binary, e1), CAST(binary, e2));
120         else
121           return FALSE;
122       }
123     }
124     
125     bool equal_unary_expressions(unary e1, unary e2)
126     {
127       if (e1->kind != e2->kind)
128         return FALSE;  
129     
130       return equal_expressions(e1->arg1, e2->arg1);
131     }
132     
133     bool equal_binary_expressions(binary e1, binary e2)
134     {
135       if (e1->kind != e2->kind)
136         return FALSE;
137     
138       return equal_expressions(e1->arg1, e2->arg1) &&
139         equal_expressions(e1->arg2, e2->arg2);      
140     }
