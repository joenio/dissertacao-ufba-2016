39     void alloc_block(region r, struct allocator *a, struct ablock *blk,
40     		 void **p1, int s1, int a1, void **p2, int s2, int a2,
41     		 size_t blksize, int needsclear)
42     {
43       struct page *newp;
44       char *mem1, *mem2;
45       
46       mem1 = PALIGN(blk->allocfrom, a1);
47       mem2 = PALIGN(mem1 + s1, a2);
48     
49       /* Can't use last byte of page (pointers to the byte after an object are
50          valid) */
51       if (mem2 + s2 >= blk->base + blksize)
52         {
53           if (blksize == RPAGESIZE)
54     	{
55     	  newp = alloc_single_page(a->pages);
56     	  a->pages = newp;
57     	  blk->allocfrom = (char *)newp + offsetof(struct page, previous);
58     	  set_region(newp, 1, r);
59     	}
60           else
61     	{
62     	  newp = alloc_pages(blksize >> RPAGELOG, a->bigpages);
63     	  a->bigpages = newp;
64     	  blk->allocfrom = (char *)newp + offsetof(struct page, previous);
65     	  set_region(newp, blksize >> RPAGELOG, r);
66     	}
67           blk->base = (char *)newp;
68     
69           if (needsclear)
70     	preclear(blk->allocfrom, blksize - (blk->allocfrom - (char *)newp));
71           mem1 = PALIGN(blk->allocfrom, a1);
72           mem2 = PALIGN(mem1 + s1, a2);
73         }
74     
75       ASSERT_INUSE(blk->base, r);
76       blk->allocfrom = mem2 + s2;
77     
78       *p1 = mem1;
79       *p2 = mem2;
80     }

82     void qalloc(region r, struct allocator *a, void **p1, int s1, int a1,
83     	    void **p2, int s2, int a2, int needsclear)
84     {
85       struct page *p;
86       char *mem;
87       int npages;
88       int n = ALIGN(s1, a2) + s2; /* Yes, this is correct (see alloc_block) */
89     
90       if (n <= RPAGESIZE / K)
91         {
92           alloc_block(r, a, &a->page, p1, s1, a1, p2, s2, a2, RPAGESIZE,
93     		  needsclear);
94           return;
95         }
96       if (n <= RPAGESIZE)
97         {
98           alloc_block(r, a, &a->superpage, p1, s1, a1, p2, s2, a2,
99     		  K * RPAGESIZE, needsclear);
100           return;
101         }
102       if (n <= RPAGESIZE * K)
103         {
104           alloc_block(r, a, &a->hyperpage, p1, s1, a1, p2, s2, a2,
105     		  K * K * RPAGESIZE, needsclear);
106           return;
107         }
108     
109       npages = (n + ALIGN(offsetof(struct page, previous), a1) + RPAGESIZE - 1)
110         >> RPAGELOG;
111       p = alloc_pages(npages, a->bigpages);
112       a->bigpages = p;
113       set_region(p, npages, r);
114     
115       mem = (char *)p + offsetof(struct page, previous);
116       *p1 = PALIGN(mem, a1);
117       *p2 = PALIGN((char *) *p1 + s1, a2);
118       if (needsclear)
119         preclear(*p2, s2);
120     }

122     void free_all_pages(region r, struct allocator *a)
123     /* Assumes freepages_lock held */
124     {
125       struct page *p, *next;
126     
127       for (p = a->pages; p; p = next)
128         {
129           next = p->next;
130           free_single_page(r, p);
131         }
132       for (p = a->bigpages; p; p = next)
133         {
134           next = p->next;
135           free_pages(r, p);
136         }
137     }
