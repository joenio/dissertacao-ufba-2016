181     /* Given a data declaration, assign it a qualified type if it doesn't
182        have one already.  Return true iff it didn't already have a
183        qualified type.  If generalize is true, make a polymorphic type

196     /* Turn qt into a type scheme, generalizing over all qualifier

200     /* Make a copy of qt, making fresh qualifiers everywhere, but also

205     /* Return TRUE iff qt1 and qt2 have the same shape.  If match_quals is
206        not NULL, calls match_quals on every pair of corresponding
207        qualifiers (q1 from qt1, q2 from qt2), with extra arg. */
208     bool match_qtype(qtype qt1, qtype qt2,
209     		 void (*match_quals)(qual q1, qual q2, void *arg), void *arg);
210     
211     /* Return TRUE iff t1 and t2 match exactly, i.e., if converting t1 and
212        t2 to qtypes would yield exactly the same qtype */
213     bool match_type(type t1, type t2);

215     /**************************************************************************
216      *                                                                        *
217      * Constraint generation                                                  *
218      *                                                                        *
219      * These functions return FALSE if adding the constraint was              *
220      * successful (there was no error) and TRUE if adding the constraint      *
221      * was unsuccessful (there was an inconsistency).                         *
222      *                                                                        *
223      **************************************************************************/
224     
225     /* qt1 <= qt2 */
226     bool mkleq_qtype(location loc, qtype qt1, qtype qt2);
227     
228     /* qt1 = qt2 */
229     bool mkeq_qtype(location loc, qtype qt1, qtype qt2);
230     
231     /* qt1 = qt2, plus cannot distinguish qt1 and qt2 afterwards */
232     bool unify_qtype(location loc, qtype qt1, qtype qt2);
233     
234     /* Equate the qualifiers on types qt1 and qt2.  For the levels on
235        which qt1 and qt2 have the same shape, the qualifiers are equated
236        level by level.  After that point, all qualifiers on the remains
237        levels are equated. */
238     bool mkeq_qtype_cast(location loc, qtype qt1, qtype qt2);
239     
240     /* Make fresh qtype lub and have qt1 <= lub, qt2 <= lub. */
241     bool lub_qtype(location loc, qtype qt1, qtype qt2, qtype *lub);
242     
243     /* Return TRUE iff left and right are the same.  Does not generate
244        constraints. */
245     bool eq_qtype(qtype qt1, qtype qt2);
246     
247     /* Add constraints bewteen all qualifiers of qt and the varargs
248        qualifier q.  For each qualifier q' of qt, a copy of q is made
249        sharing q's constraints, and then the copy is equated to q'.
250        Returns TRUE if an error occurs. */
251     bool varargs_constrain_quals_qtype(location loc, qtype qt, qual q);
252     
253     /* Add constraint e \not\in qt */
254     void mknin_effect_qtype(location loc, effect e, qtype qt);
