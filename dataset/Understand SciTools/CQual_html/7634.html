23     int location_hash(location loc)
24     {
25       int h;
26     
27       h = 0;
28       h = loc->lineno;
29       h = 33*h + 720 + loc->filepos;
30       h = 33*h + 720 + string_hash(loc->filename);
31       return h;
32     }
33     
34     bool location_eq(location loc1, location loc2)
35     {
36       return (loc1->lineno == loc2->lineno &&
37     	  loc1->filepos == loc2->filepos &&
38     	  loc1->in_system_header == loc2->in_system_header &&
39     	  !strcmp(loc1->filename, loc2->filename));
40     }
41     
42     int location_cmp(location loc1, location loc2)
43     {
44       int result;
45     
46       result = strcmp(loc1->filename, loc2->filename);
47       if (!result)
48         result = loc1->filepos - loc2->filepos;
49       return result;
50     }
