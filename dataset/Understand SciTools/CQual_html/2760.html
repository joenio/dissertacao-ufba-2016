47     /* Return TRUE if TTL and TTR are pointers to types that are equivalent,
48        ignoring their qualifiers.  */
49     static bool compatible_pointer_targets(type ttl, type ttr, bool pedantic)
50     {
51       int val;
52     
53       val = type_compatible_unqualified(ttl, ttr);
54     
55       if (val == 2 && pedantic)
56         pedwarn("types are not quite compatible");
57       return val != 0;
58     }

60     static bool compatible_pointer_types(type tl, type tr)
61     {
62       return compatible_pointer_targets(type_points_to(tl), type_points_to(tr),
63     				    pedantic);
64     }

66     static void warn_for_assignment(const char *msg, const char *opname,
67     				const char *function, int argnum)
68     {
69       static char argstring[] = "passing arg %d of `%s'";
70       static char argnofun[] =  "passing arg %d";
71     
72       if (opname == 0)
73         {
74           char *tmpname;
75     
76           if (function)
77     	{
78     	  /* Function name is known; supply it.  */
79     	  tmpname = (char *)alloca(strlen(function) + sizeof(argstring) + 25 /*%d*/ + 1);
80     	  sprintf(tmpname, argstring, argnum, function);
81     	}
82           else
83     	{
84     	  /* Function name unknown (call through ptr); just give arg number.  */
85     	  tmpname = (char *)alloca(sizeof(argnofun) + 25 /*%d*/ + 1);
86     	  sprintf(tmpname, argnofun, argnum);
87     	}
88           opname = tmpname;
89         }
90       pedwarn(msg, opname);
91     }

93     static void incomplete_type_error(expression e, type t)
94     {
95       /* Avoid duplicate error message.  */
96       if (t == error_type)
97         return;
98     
99       if (e && is_identifier(e))
100         error("`%s' has an incomplete type", CAST(identifier, e)->cstring.data);
101       else
102         {
103           while (type_array(t) && type_array_size(t))
104     	t = type_array_of(t);
105     
106           if (type_struct(t))
107     	error("invalid use of undefined type `struct %s'", type_tag(t)->name);
108           else if (type_union(t))
109     	error("invalid use of undefined type `union %s'", type_tag(t)->name);
110           else if (type_enum(t))
111     	error("invalid use of undefined type `enum %s'", type_tag(t)->name);
112           else if (type_void(t))
113     	error("invalid use of void expression");
114           else if (type_array(t))
115     	error("invalid use of array with unspecified bounds");
116           else
117     	assert(0);
118           /* XXX: Missing special message for typedef's */
119         }
120     }

123     {
124       if (!type_incomplete(etype))
125         return e->type;

128     
129       return error_type;

132     type default_conversion(expression e)
133     {
134       type from = e->type;
135     
136       if (type_enum(from))
137         from = type_tag(from)->reptype;
138     
139       if (type_smallerthanint(from))
140         {
141           /* Traditionally, unsignedness is preserved in default promotions. */
142           if (flag_traditional && type_unsigned(from))
143     	return unsigned_int_type;
144           else
145     	return int_type;
146         }
147     
148       if (flag_traditional && !flag_allow_single_precision && type_float(from))
149         return double_type;
150     
151       if (type_void(from))
152         {
153           error("void value not ignored as it ought to be");
154           return error_type;
155         }
156     
157       if (type_function(from))
158         {
159           assert(!e->cst);
160           e->cst = e->static_address;
161           return make_pointer_type(from);
162         }
163     
164       if (type_array(from))
165         {
166           if (!e->lvalue)
167     	{
168     	  error("invalid use of non-lvalue array");
169     	  return error_type;
170     	}
171           assert(!e->cst);
172           e->cst = e->static_address;
173           /* It's being used as a pointer, so is not an lvalue */
174           e->lvalue = FALSE;
175           return make_pointer_type(type_array_of(from));
176         }
177     
178       return from;
179     }

189     static void readonly_warning(expression e, char *context)
190     {
191       char buf[80];
192     
193       strcpy(buf, context);
194     
195       if (is_field_ref(e))
196         {
197           field_ref field = CAST(field_ref, e);
198     
199           if (type_readonly(field->arg1->type))
200     	readonly_warning(field->arg1, context);
201           else
202     	{
203     	  strcat(buf, " of read-only member `%s'");
204     	  pedwarn(buf, field->cstring.data);
205     	}
206         }
207       else if (is_identifier(e))
208         {
209           strcat(buf, " of read-only variable `%s'");
210           pedwarn(buf, CAST(identifier, e)->cstring.data);
211         }
212       else
213         pedwarn ("%s of read-only location", buf);
214     }

217     {
218       if (!e->lvalue || type_array(e->type))

228     bool check_conversion(type to, type from)
229     {
230       if (type_equal_unqualified(to, from))
231         return TRUE;
232     
233       if (to == error_type || from == error_type)
234         return FALSE;
235     
236       if (type_void(from))
237         {
238           error("void value not ignored as it ought to be");
239           return FALSE;
240         }
241     
242       if (type_void(to))
243         return TRUE;
244     
245       if (type_integer(to))
246         {
247           if (!type_scalar(from))
248     	{
249     	  error("aggregate value used where an integer was expected");
250     	  return FALSE;
251     	}
252         }
253       else if (type_pointer(to))
254         {
255           if (!(type_integer(from) || type_pointer(from)))
256     	{
257     	  error("cannot convert to a pointer type");
258     	  return FALSE;
259     	}
260         }
261       else if (type_floating(to))
262         {
263           if (type_pointer(from))
264     	{
265     	  error("pointer value used where a floating point value was expected");
266     	  return FALSE;
267     	}
268           else if (!type_arithmetic(from))
269     	{
270     	  error("aggregate value used where a float was expected");
271     	  return FALSE;
272     	}
273         }
274       else if (type_complex(to))
275         {
276           if (type_pointer(from))
277     	{
278     	  error("pointer value used where a complex was expected");
279     	  return FALSE;
280     	}
281           else if (!type_arithmetic(from))
282     	{
283     	  error("aggregate value used where a complex was expected");
284     	  return FALSE;
285     	}
286         }
287       else
288         {
289           error("conversion to non-scalar type requested");
290           return FALSE;
291         }
292       return TRUE;
293     }

295     static bool assignable_pointer_targets(type tt1, type tt2, bool pedantic)
296     {
297       return type_void(tt1) || type_void(tt2)
298         || compatible_pointer_targets(tt1, tt2, pedantic);
299     }

301     static void ptrconversion_warnings(type ttl, type ttr, expression rhs,
302     				   const char *context,
303     				   const char *funname, int parmnum,
304     				   bool pedantic)
305     {
306       if (pedantic
307           && ((type_void(ttl) && type_function(ttr)) ||
308     	  (type_function(ttl) && type_void(ttr) &&
309     	   !(rhs && definite_null(rhs)))))
310         warn_for_assignment("ANSI forbids %s between function pointer and `void *'",
311     			context, funname, parmnum);
312     
313       /* Const and volatile mean something different for function
314          types, so the usual warnings are not appropriate.  */
315       else if (type_function(ttl) && type_function(ttr))
316         {
317           /* Because const and volatile on functions are
318     	 restrictions that say the function will not do
319     	 certain things, it is okay to use a const or volatile
320     	 function where an ordinary one is wanted, but not
321     	 vice-versa.  */
322           if (type_const(ttl) && !type_const(ttr))
323     	warn_for_assignment("%s makes `const *' function pointer from non-const",
324     			    context, funname, parmnum);
325           if (type_volatile(ttl) && !type_volatile(ttr))
326     	warn_for_assignment("%s makes `volatile *' function pointer from non-volatile",
327     			    context, funname, parmnum);
328         }
329       else if (!type_function(ttl) && !type_function(ttr))
330         {
331           if (!type_const(ttl) && type_const(ttr))
332     	warn_for_assignment("%s discards `const' from pointer target type",
333     			    context, funname, parmnum);
334           if (!type_volatile(ttl) && type_volatile(ttr))
335     	warn_for_assignment("%s discards `volatile' from pointer target type",
336     			    context, funname, parmnum);
337     
338           /* If this is not a case of ignoring a mismatch in signedness,
339     	 no warning.  */
340           if (!assignable_pointer_targets(ttl, ttr, FALSE) && pedantic)
341     	warn_for_assignment("pointer targets in %s differ in signedness",
342     			    context, funname, parmnum);
343         }
344     }

346     /* Return TRUE if no error and lhstype and rhstype are not error_type */
347     bool check_assignment(type lhstype, type rhstype, expression rhs,
348     		      const char *context, data_declaration fundecl,
349     		      const char *funname, int parmnum)
350     {
351       bool zerorhs = rhs && definite_zero(rhs);
352     
353       if (lhstype == error_type || rhstype == error_type)
354         return FALSE;
355     
356       if (type_void(rhstype))
357         {
358           error("void value not ignored as it ought to be");
359           return FALSE;
360         }
361     
362       if (type_equal_unqualified(lhstype, rhstype))
363         return TRUE;
364     
365       if (type_arithmetic(lhstype) && type_arithmetic(rhstype))
366         {
367           if (rhs)
368     	constant_overflow_warning(rhs->cst);
369           return check_conversion(lhstype, rhstype);
370         }
371       if (parmnum && (type_qualifiers(lhstype) & transparent_qualifier))
372         {
373           /* See if we can match any field of lhstype */
374           tag_declaration tag = type_tag(lhstype);
375           field_declaration fields, marginal_field = NULL;
376     
377           /* I blame gcc for this horrible mess (and it's minor inconsistencies
378     	 with the regular rules) */
379           /* pedantic warnings are skipped in here because we're already
380     	 issuing a warning for the use of this construct */
381           for (fields = tag->fieldlist; fields; fields = fields->next)
382     	{
383     	  type ft = fields->type;
384     
385     	  if (type_compatible(ft, rhstype))
386     	    break;
387     
388     	  if (!type_pointer(ft))
389     	    continue;
390     
391     	  if (type_pointer(rhstype))
392     	    {
393     	      type ttl = type_points_to(ft), ttr = type_points_to(rhstype);
394     	      bool goodmatch = assignable_pointer_targets(ttl, ttr, FALSE);
395     
396     	      /* Any non-function converts to a [const][volatile] void *
397     		 and vice versa; otherwise, targets must be the same.
398     		 Meanwhile, the lhs target must have all the qualifiers of
399     		 the rhs.  */
400     	      if (goodmatch)
401     		{
402     		  /* If this type won't generate any warnings, use it.  */
403     		  if ((type_function(ttr) && type_function(ttl))
404     		      ? ((!type_const(ttl) | type_const(ttr))
405     			 & (!type_volatile(ttl) | type_volatile(ttr)))
406     		      : ((type_const(ttl) | !type_const(ttr))
407     			 & (type_volatile(ttl) | !type_volatile(ttr))))
408     		    break;
409     
410     		  /* Keep looking for a better type, but remember this one.  */
411     		  if (!marginal_field)
412     		    marginal_field = fields;
413     		}
414     	    }
415     
416     	  /* Can convert integer zero to any pointer type.  */
417     	  /* Note that this allows passing *any* null pointer (gcc bug?) */
418     	  if (zerorhs)
419     	    break;
420     	}
421     
422           if (fields || marginal_field)
423     	{
424     	  if (!fields)
425     	    {
426     	      /* We have only a marginally acceptable member type;
427     		 it needs a warning.  */
428     	      type ttl = type_points_to(marginal_field->type),
429     		ttr = type_points_to(rhstype);
430     
431     	      ptrconversion_warnings(ttl, ttr, rhs, context, funname, parmnum,
432     				     FALSE);
433     	    }
434     	  
435     	  if (pedantic && !(fundecl && fundecl->in_system_header))
436     	    pedwarn("ANSI C prohibits argument conversion to union type");
437     
438     	  return TRUE;
439     	}
440         }
441     
442       if (type_pointer(lhstype) && type_pointer(rhstype))
443         {
444           type ttl = type_points_to(lhstype), ttr = type_points_to(rhstype);
445           bool goodmatch = assignable_pointer_targets(ttl, ttr, pedantic);
446     
447           /* Any non-function converts to a [const][volatile] void *
448     	 and vice versa; otherwise, targets must be the same.
449     	 Meanwhile, the lhs target must have all the qualifiers of the rhs.  */
450           if (goodmatch || (type_equal_unqualified(make_unsigned_type(ttl),
451     					       make_unsigned_type(ttr))))
452     	ptrconversion_warnings(ttl, ttr, rhs, context, funname, parmnum,
453     			       pedantic);
454           else
455     	warn_for_assignment("%s from incompatible pointer type",
456     			    context, funname, parmnum);
457     
458           return check_conversion(lhstype, rhstype);
459         }
460       /* enum = ptr and ptr = enum counts as an error, so use type_integral */
461       else if (type_pointer(lhstype) && type_integral(rhstype))
462         {
463           if (!zerorhs)
464     	warn_for_assignment("%s makes pointer from integer without a cast",
465     			    context, funname, parmnum);
466           return check_conversion(lhstype, rhstype);
467         }
468       else if (type_integral(lhstype) && type_pointer(rhstype))
469         {
470           warn_for_assignment("%s makes integer from pointer without a cast",
471     			  context, funname, parmnum);
472           return check_conversion(lhstype, rhstype);
473         }
474     
475       if (!context)
476         if (funname)
477           error("incompatible type for argument %d of `%s'", parmnum, funname);
478         else
479           error("incompatible type for argument %d of indirect function call",
480     	    parmnum);
481       else
482         error("incompatible types in %s", context);
483     
484       return FALSE;
485     }

594     expression make_address_of(location loc, expression e)
595     {
596       expression result = CAST(expression, new_address_of(parse_region, loc, e));
597     
598       result->type = error_type;
599     
600       if (e->type == error_type)
601         ;
602       else if (e->bitfield)
603         error("attempt to take address of a bit-field structure member");
604       else
605         {
606           if (e->isregister)
607     	pedwarn("address of a register variable requested");
608     
609           if (!(type_function(e->type) || e->lvalue))
610     	error("invalid lvalue in unary `&'");
611     
612           result->type = make_pointer_type(e->type);
613           result->cst = e->static_address;
614         }
615       return result;
616     }

618     expression make_unary(location loc, int unop, expression e)
619     {
620       switch (unop)
621         {
622         case kind_address_of:
623           return make_address_of(loc, e);
624         case kind_preincrement:
625           return make_preincrement(loc, e);
626         case kind_predecrement:
627           return make_predecrement(loc, e);
628         default:
629           {
630     	expression result = CAST(expression, newkind_unary(parse_region, unop, loc, e));
631     	type etype = default_conversion(e);
632     	const char *errstring = NULL;
633     
634     	if (etype == error_type)
635     	  result->type = error_type;
636     	else
637     	  {
638     	    switch (unop)
639     	      {
640     	      case kind_unary_plus:
641     		if (!type_arithmetic(etype))
642     		  errstring = "wrong type argument to unary plus";
643     		break;
644     	      case kind_unary_minus:
645     		if (!type_arithmetic(etype))
646     		  errstring = "wrong type argument to unary minus";
647     		break;
648     	      case kind_bitnot:
649     		if (type_complex(etype))
650     		  result->kind = kind_conjugate;
651     		else if (!type_integer(etype))
652     		  errstring = "wrong type argument to bit-complement";
653     		break;
654     	      case kind_not:
655     		if (!type_scalar(etype))
656     		  errstring = "wrong type argument to unary exclamation mark";
657     		else
658     		  etype = int_type;
659     		break;
660     	      case kind_realpart: case kind_imagpart:
661     		if (!type_arithmetic(etype))
662     		  if (unop == kind_realpart)
663     		    errstring = "wrong type argument to __real__";
664     		  else
665     		    errstring = "wrong type argument to __imag__";
666     		else
667     		  etype = type_complex(etype) ? make_base_type(etype) : etype;
668     	      default:
669     		assert(0);
670     	      }
671     	    if (errstring)
672     	      {
673     		error(errstring);
674     		result->type = error_type;
675     	      }
676     	    else
677     	      {
678     		result->type = etype;
679     		result->cst = fold_unary(result);
680     	      }
681     	  }
682     	return result;
683           }
684         }
685     }

687     expression make_label_address(location loc, id_label label)
688     {
689       expression result = CAST(expression, new_label_address(parse_region, loc, label));
690     
691       use_label(label);
692     
693       result->type = ptr_void_type;
694       result->cst = fold_label_address(result);
695     
696       if (pedantic)
697         pedwarn("ANSI C forbids `&&'");
698

755     expression make_cast(location loc, asttype t, expression e)
756     {
757       expression result = CAST(expression, new_cast(parse_region, loc, e, t));
758       type castto = t->type;
759       
760       if (castto == error_type || type_void(castto))
761         ; /* Do nothing */
762       else if (type_array(castto))
763         {
764           error("cast specifies array type");
765           castto = error_type;
766         }
767       else if (type_function(castto))
768         {
769           error("cast specifies function type");
770           castto = error_type;
771         }
772       else if (type_equal_unqualified(castto, e->type))
773         {
774           if (pedantic && type_aggregate(castto))
775     	pedwarn("ANSI C forbids casting nonscalar to the same type");
776         }
777       else
778         {
779           type etype = e->type;
780     
781           /* Convert functions and arrays to pointers,
782     	 but don't convert any other types.  */
783           if (type_function(etype) || type_array(etype))
784     	etype = default_conversion(e);
785     
786           if (type_union(castto))
787     	{
788     	  tag_declaration utag = type_tag(castto);
789     	  field_declaration ufield;
790     
791     	  /* Look for etype as a field of the union */
792     	  for (ufield = utag->fieldlist; ufield; ufield = ufield->next)
793     	    if (ufield->name && type_equal_unqualified(ufield->type, etype))
794     	      {
795     		if (pedantic)
796     		  pedwarn("ANSI C forbids casts to union type");
797     		break;
798     	      }
799     	  if (!ufield)
800     	    error("cast to union type from type not present in union");
801     	}
802           else 
803     	{
804     	  /* Optionally warn about potentially worrisome casts.  */
805     
806     	  if (warn_cast_qual && type_pointer(etype) && type_pointer(castto))
807     	    {
808     	      type ep = type_points_to(etype), cp = type_points_to(castto);
809     
810     	      if (type_volatile(ep) && !type_volatile(cp))
811     		pedwarn("cast discards `volatile' from pointer target type");
812     	      if (type_const(ep) && !type_const(cp))
813     		pedwarn("cast discards `const' from pointer target type");
814     	    }
815     
816     	  /* This warning is weird */
817     	  if (warn_bad_function_cast && is_function_call(e) &&
818     	      !type_equal_unqualified(castto, etype))
819     	    warning ("cast does not match function type");
820     
821     #if 0
822     	  /* Warn about possible alignment problems.  */
823     	  if (STRICT_ALIGNMENT && warn_cast_align
824     	      && TREE_CODE (type) == POINTER_TYPE
825     	      && TREE_CODE (otype) == POINTER_TYPE
826     	      && TREE_CODE (TREE_TYPE (otype)) != VOID_TYPE
827     	      && TREE_CODE (TREE_TYPE (otype)) != FUNCTION_TYPE
828     	      /* Don't warn about opaque types, where the actual alignment
829     		 restriction is unknown.  */
830     	      && !((TREE_CODE (TREE_TYPE (otype)) == UNION_TYPE
831     		    || TREE_CODE (TREE_TYPE (otype)) == RECORD_TYPE)
832     		   && TYPE_MODE (TREE_TYPE (otype)) == VOIDmode)
833     	      && TYPE_ALIGN (TREE_TYPE (type)) > TYPE_ALIGN (TREE_TYPE (otype)))
834     	    warning ("cast increases required alignment of target type");
835     
836     	  if (TREE_CODE (type) == INTEGER_TYPE
837     	      && TREE_CODE (otype) == POINTER_TYPE
838     	      && TYPE_PRECISION (type) != TYPE_PRECISION (otype)
839     	      && !TREE_CONSTANT (value))
840     	    warning ("cast from pointer to integer of different size");
841     
842     	  if (TREE_CODE (type) == POINTER_TYPE
843     	      && TREE_CODE (otype) == INTEGER_TYPE
844     	      && TYPE_PRECISION (type) != TYPE_PRECISION (otype)
845     #if 0
846     	      /* Don't warn about converting 0 to pointer,
847     		 provided the 0 was explicit--not cast or made by folding.  */
848     	      && !(TREE_CODE (value) == INTEGER_CST && integer_zerop (value))
849     #endif
850     	      /* Don't warn about converting any constant.  */
851     	      && !TREE_CONSTANT (value))
852     	    warning ("cast to pointer from integer of different size");
853     #endif
854     
855     	  check_conversion(castto, etype);
856     	}
857         }
858     
859       result->lvalue = !pedantic && e->lvalue;
860       result->isregister = e->isregister;
861       result->bitfield = e->bitfield;
862       result->static_address = e->static_address;
863       result->type = castto;
864       result->cst = fold_cast(result);
865     
866       return result;
867     }

900     type check_binary(int binop, expression e1, expression e2)
901     {
902       type t1 = default_conversion(e1), t2 = default_conversion(e2);
903       type rtype = NULL;
904       bool common = FALSE;
905     
906       /* XXX: Misc warnings (see build_binary_op) */
907       if (t1 == error_type || t2 == error_type)
908         rtype = error_type;
909       else switch(binop)
910         {
911         case kind_plus:
912           if (type_pointer(t1) && type_integer(t2))
913     	rtype = pointer_int_sum(t1, t2);
914           else if (type_pointer(t2) && type_integer(t1))
915     	rtype = pointer_int_sum(t2, t1);
916           else
917     	common = TRUE;
918           break;
919     
920         case kind_minus: 
921           if (type_pointer(t1) && type_integer(t2))
922     	rtype = pointer_int_sum(t1, t2);
923           else if (type_pointer(t1) && type_pointer(t2) &&
924     	       compatible_pointer_types(t1, t2))
925     	rtype = ptrdiff_t_type;
926           else
927     	common = TRUE;
928           break;
929     
930         case kind_plus_assign: case kind_minus_assign:
931           if (type_pointer(t1) && type_integer(t2))
932     	rtype = pointer_int_sum(t1, t2);
933           else
934     	common = TRUE;
935           break;
936     
937         case kind_times: case kind_divide:
938         case kind_times_assign: case kind_divide_assign:
939           common = TRUE;
940           break;
941     
942         case kind_modulo: case kind_bitand: case kind_bitor: case kind_bitxor:
943         case kind_lshift: case kind_rshift:
944         case kind_modulo_assign: case kind_bitand_assign: case kind_bitor_assign:
945         case kind_bitxor_assign: case kind_lshift_assign: case kind_rshift_assign:
946           if (type_integer(t1) && type_integer(t2))
947     	rtype = common_type(t1, t2);
948           break;
949     
950         case kind_leq: case kind_geq: case kind_lt: case kind_gt:
951           rtype = int_type; /* Default to assuming success */
952           if (type_real(t1) && type_real(t2))
953     	;
954           else if (type_pointer(t1) && type_pointer(t2))
955     	{
956     	  if (compatible_pointer_types(t1, t2))
957     	    {
958     	      /* XXX: how can this happen ? */
959     	      if (type_incomplete(t1) != type_incomplete(t2))
960     		pedwarn("comparison of complete and incomplete pointers");
961     	      else if (pedantic && type_function(type_points_to(t1)))
962     		pedwarn("ANSI C forbids ordered comparisons of pointers to functions");
963     	    }
964     	  else
965     	    pedwarn("comparison of distinct pointer types lacks a cast");
966     	}
967           /* XXX: Use of definite_zero may lead to extra warnings when !extra_warnings */
968           else if ((type_pointer(t1) && definite_zero(e2)) ||
969     	       (type_pointer(t2) && definite_zero(e1)))
970     	{
971     	  if (pedantic || extra_warnings)
972     	    pedwarn("ordered comparison of pointer with integer zero");
973     	}
974           else if ((type_pointer(t1) && type_integer(t2)) ||
975     	       (type_pointer(t2) && type_integer(t1)))
976     	{
977     	  if (!flag_traditional)
978     	    pedwarn("comparison between pointer and integer");
979     	}
980           else
981     	rtype = NULL; /* Force error */
982           break;
983     
984         case kind_eq: case kind_ne:
985           rtype = int_type; /* Default to assuming success */
986           if (type_arithmetic(t1) && type_arithmetic(t2))
987     	;
988           else if (type_pointer(t1) && type_pointer(t2))
989     	{
990     	  if (!compatible_pointer_types(t1, t2) &&
991     	      !valid_compare(t1, t2, e1) &&
992     	      !valid_compare(t2, t1, e2))
993     	    pedwarn("comparison of distinct pointer types lacks a cast");
994     	}
995           else if ((type_pointer(t1) && definite_null(e2)) ||
996     	       (type_pointer(t2) && definite_null(e1)))
997     	;
998           else if ((type_pointer(t1) && type_integer(t2)) ||
999     	       (type_pointer(t2) && type_integer(t1)))
1000     	{
1001     	  if (!flag_traditional)
1002     	    pedwarn("comparison between pointer and integer");
1003     	}
1004           else
1005     	rtype = NULL; /* Force error */
1006           break;
1007     
1008         case kind_andand: case kind_oror:
1009           if (type_scalar(t1) && type_scalar(t2))
1010     	rtype = int_type;
1011           break;
1012     
1013         default: assert(0); break;
1014         }
1015     
1016       if (common && type_arithmetic(t1) && type_arithmetic(t2))
1017         rtype = common_type(t1, t2);
1018     
1019       if (!rtype)
1020         {
1021           error("invalid operands to binary %s", binary_op_name(binop));
1022           rtype = error_type;
1023         }
1024     
1025       return rtype;
1026     }

1070     expression make_conditional(location loc, expression cond,
1071     			    expression true, expression false)
1072     {
1073       expression result =
1074         CAST(expression, new_conditional(parse_region, loc, cond, true, false));
1075       type ctype, ttype, ftype, rtype = NULL;
1076       bool truelvalue = true ? true->lvalue : FALSE;
1077     
1078       ctype = default_conversion(cond);
1079     
1080       if (!true)
1081         {
1082           true = cond;
1083           truelvalue = FALSE; /* Not an lvalue in gcc ! */
1084         }
1085     
1086       if (type_void(true->type))
1087         ttype = true->type;
1088       else
1089         ttype = default_conversion(true);
1090       
1091       if (type_void(false->type))
1092         ftype = false->type;
1093       else
1094         ftype = default_conversion(false);
1095     
1096       if (ctype == error_type || ttype == error_type || ftype == error_type)
1097         rtype = error_type;
1098       else if (type_equal(ttype, ftype))
1099         rtype = ttype;
1100       else if (type_equal_unqualified(ttype, ftype))
1101         rtype = make_qualified_type(ttype, no_qualifiers, NULL);
1102       else if (type_real(ttype) && type_real(ftype))
1103         /* This should probably be type_arithmetic. See complex3.c/C9X */
1104         rtype = common_type(ttype, ftype);
1105       else if (type_void(ttype) || type_void(ftype))
1106         {
1107           if (pedantic && (!type_void(ttype) || !type_void(ftype)))
1108     	pedwarn("ANSI C forbids conditional expr with only one void side");
1109           rtype = void_type;
1110         }
1111       else if (type_pointer(ttype) && type_pointer(ftype))
1112         {
1113           type tpointsto = type_points_to(ttype), fpointsto = type_points_to(ftype);
1114     
1115           if (compatible_pointer_types(ttype, ftype))
1116     	rtype = common_type(tpointsto, fpointsto);
1117           else if (definite_null(true) && type_void(tpointsto))
1118     	rtype = fpointsto;
1119           else if (definite_null(false) && type_void(fpointsto))
1120     	rtype = tpointsto;
1121           else if (voidstar_conditional(tpointsto, fpointsto))
1122     	rtype = tpointsto; /* void * result */
1123           else if (voidstar_conditional(fpointsto, tpointsto))
1124     	rtype = fpointsto; /* void * result */
1125           else
1126     	{
1127     	  pedwarn("pointer type mismatch in conditional expression");
1128     	  /* Slight difference from GCC: I qualify the result type with
1129     	     the appropriate qualifiers */
1130     	  rtype = void_type;
1131     	}
1132     
1133           /* Qualifiers depend on both types */
1134           rtype = make_pointer_type(qualify_type2(rtype, tpointsto, fpointsto));
1135         }
1136       else if (pointerint_conditional(ttype, ftype, false))
1137         rtype = ttype;
1138       else if (pointerint_conditional(ftype, ttype, true))
1139         rtype = ftype;
1140       else if (flag_cond_mismatch)
1141         rtype = void_type;
1142       else
1143         {
1144           error("type mismatch in conditional expression");
1145           rtype = error_type;
1146         }
1147       
1148       /* Qualifiers depend on both types */
1149       if (rtype != error_type)
1150         rtype = qualify_type2(rtype, ttype, ftype);
1151     
1152       result->type = rtype;
1153       result->lvalue = !pedantic && truelvalue && false->lvalue;
1154       result->isregister = true->isregister || false->isregister;
1155       result->bitfield = true->bitfield || false->bitfield;
1156       result->cst = fold_conditional(result);
1157     
1158       return result;
1159     }

1162     {
1163       expression result = CAST(expression, newkind_binary(parse_region, binop,
1164     						      loc, e1, e2));
1165       type t1 = require_complete_type(e1, e1->type), t2;
1166     
1167       result->type = error_type;
1168       if (t1 != error_type && e2->type != error_type)
1169         {
1170           expression rhs;

1173     	{

1175     	  rhs = e2;

1291     		     since without a prototype it would be `double'.  */
1292     		  else if (type_float(parmtype) && type_floating(argtype))
1293     		    warn_for_assignment ("%s as `float' rather than `double' due to prototype",
1294     					NULL, name, parmnum);

1299     		      type type1 = default_conversion(arglist);

1304     			;
1305     		      /* XXX: else messy stuff that cannot easily be done w/o constant
1306     			 folding and type size info */
1307     		    }
1308     #endif
1309     		}
1310     	      check_assignment(parmtype, default_conversion_for_assignment(arglist),
1311     			       arglist, NULL, fundecl, name, parmnum);
1312     	    }
1313     	  parmnum++;
1314     	  arglist = CAST(expression, arglist->next);
1315     	}
1316           if (parmtype)
1317     	{
1318     	  if (name)
1319     	    error("too few arguments to function `%s'", name);
1320     	  else
1321     	    error("too few arguments to function");
1322     	}
1323           else if (arglist && !type_function_varargs(fntype))
1324     	{
1325     	  if (name)
1326     	    error("too many arguments to function `%s'", name);
1327     	  else
1328     	    error("too many arguments to function");
1329     	}
1330         }
1331     
1332       /* Checks for arguments with no corresponding argument type */
1333       while (arglist)
1334         {
1335           require_complete_type(arglist, default_conversion(arglist));
1336           arglist = CAST(expression, arglist->next);
1337         }
1338     }

1340     expression make_function_call(location loc, expression fn, expression arglist)
1341     {
1342       expression result = CAST(expression, new_function_call(parse_region, loc, fn, arglist, NULL));
1343       type fntype = default_conversion(fn), rettype;
1344     
1345       result->type = error_type;
1346       if (fntype == error_type)
1347         ;
1348       else if (!(type_pointer(fntype) && type_function(type_points_to(fntype))))
1349         error("called object is not a function");
1350       else
1351         {
1352           char *funname = NULL;
1353           data_declaration fundecl = NULL;
1354     
1355           if (is_identifier(fn))
1356     	{
1357     	  identifier fnid = CAST(identifier, fn);
1358     
1359     	  if (fnid->ddecl->kind == decl_function)
1360     	    {
1361     	      funname = fnid->cstring.data;
1362     	      fundecl = fnid->ddecl;
1363     	    }
1364     	}
1365     
1366           fntype = type_points_to(fntype);
1367     
1368           check_arguments(fntype, arglist, fundecl, funname);
1369     
1370           rettype = type_function_return_type(fntype);
1371           result->type = rettype;
1372           if (!type_void(rettype))
1373     	result->type = require_complete_type(result, rettype);
1374        }
1375     
1376       return result;
1377     }

1380     {
1381       expression va_arg_id = build_identifier(parse_region, loc, builtin_va_arg_decl);
1382       expression result = CAST(expression, new_function_call(parse_region, loc, va_arg_id, arg, type));
1383     
1384       if (!type_equal_unqualified(arg->type, builtin_va_list_type))
1385         error("first argument to `va_arg' not of type `va_list'");
1386     
1387       if (!type_self_promoting(type->type))
1388         {
1389           static bool gave_help;
1390

1392           if (!gave_help)

1395     	  error("(so you should pass `int', `unsigned' or `double' to `va_arg')");
1396     	}
1397         }
1398       result->type = type->type;
1399     
1400       return result;

1403     expression make_array_ref(location loc, expression array, expression index)
1404     {
1405       expression result = CAST(expression, new_array_ref(parse_region, loc, array, index));
1406       type atype, itype = default_conversion(index);
1407     
1408       if (warn_char_subscripts && type_char(index->type))
1409         warning("subscript has type `char'");
1410     
1411       if (type_array(array->type) && !array->lvalue)
1412         {
1413           /* Some special GCC extensions */
1414           /* XXX: Ignoring the weird register stuff, going for a simple version
1415     	 which seems essentially identical for our purposes */
1416           if (pedantic)
1417     	pedwarn("ANSI C forbids subscripting non-lvalue array");
1418           atype = make_pointer_type(type_array_of(array->type));
1419     
1420            /* this should not be possible (non-lvalue arrays come from
1421     	  array fields of non-lvalue struct expressions) */
1422           assert(!array->static_address);
1423         }
1424       else
1425         atype = default_conversion(array);
1426     
1427       /* Put the integer in ITYPE to simplify error checking.  */
1428       if (type_integer(atype))
1429         {
1430           type temp = atype;
1431           atype = itype; 
1432           itype = temp;
1433         }
1434     
1435       if (!type_pointer(atype) || type_function(type_points_to(atype)))
1436         {
1437           error("subscripted value is neither array nor pointer");
1438           result->type = error_type;
1439           result->lvalue = TRUE;
1440         }
1441       else
1442         {
1443           check_dereference(result, atype, "array indexing");
1444           result->static_address = fold_binary(atype, result);
1445         }
1446     
1447       if (!type_integer(itype))
1448         error("array subscript is not an integer");
1449     
1450       return result;
1451     }

1453     expression make_field_ref(location loc, expression object, cstring field,
1454     			  location cstring_loc)
1455     {
1456       type otype = object->type;
1457       expression result = CAST(expression, new_field_ref(parse_region, loc, object, field, cstring_loc));
1458     
1459       result->type = error_type;
1460     
1461       if (type_aggregate(otype))
1462         {
1463           tag_declaration tag = type_tag(otype);
1464     
1465           if (!tag->defined)
1466     	incomplete_type_error(NULL, otype);
1467           else
1468     	{
1469     	  field_declaration fdecl = env_lookup(tag->fields, field.data, FALSE);
1470     
1471     	  if (!fdecl)
1472     	    error(type_struct(otype) ? "structure has no member named `%s'"
1473     		  : "union has no member named `%s'", field.data);
1474     	  else
1475     	    {
1476     	      result->type = qualify_type2(fdecl->type, fdecl->type, object->type);
1477     	      result->bitfield = fdecl->bitwidth >= 0;
1478     	      result->static_address = foldaddress_field_ref(object->static_address, fdecl);
1479     
1480     	    }
1481     	}
1482         }
1483       else if (otype != error_type)
1484         error("request for member `%s' in something not a structure or union",
1485     	  field.data);
1486     
1487       result->lvalue = object->lvalue;
1488     
1489       return result;
1490     }

1492     static expression increment(unary result, char *name)
1493     {
1494       expression e = result->arg1;
1495       type etype = e->type;
1496     
1497       result->type = error_type;
1498     
1499       if (!type_scalar(etype))
1500         error("wrong type argument to %s", name);
1501       else 
1502         {
1503           if (type_incomplete(etype))
1504     	error("%s of pointer to unknown structure or union", name);
1505           else if (type_pointer(etype) && (pedantic || warn_pointer_arith) &&
1506     	       (type_void(type_points_to(etype)) ||
1507     		type_function(type_points_to(etype))))
1508     	pedwarn("wrong type argument to %s", name);
1509     
1510           if (check_writable_lvalue(e, name))
1511     	result->type = etype;
1512         }
1513       return CAST(expression, result);
1514     }

1540     static size_t extract_strings(expression string_components,
1541     			      wchar_t *into, bool *wide)
1542     {
1543       size_t total_length = 0;
1544       expression astring;
1545       
1546       *wide = FALSE;
1547       scan_expression (astring, string_components)
1548         {
1549           const wchar_t *chars;
1550           size_t length;
1551     
1552           if (!type_equal(type_array_of(astring->type), char_type))
1553     	*wide = TRUE;
1554     
1555           if (is_identifier(astring))
1556     	{
1557     	  data_declaration dd = CAST(identifier, astring)->ddecl;
1558     
1559     	  chars = dd->chars;
1560     	  length = dd->chars_length;
1561     	}
1562           else
1563     	{
1564     	  string_cst s = CAST(string_cst, astring);
1565     
1566     	  chars = s->chars;
1567     	  length = s->length;
1568     	}
1569           if (into)
1570     	{
1571     	  memcpy(into, chars, length * sizeof(wchar_t));
1572     	  into += length;
1573     	}
1574     
1575           total_length += length;
1576         }
1577     
1578       return total_length;
1579     }
1580     
1581     string make_string(location loc, expression string_components)
1582     {
1583       string s = new_string(parse_region, loc, string_components, NULL);
1584       size_t total_length = 0;
1585       bool wide;
1586     
1587       total_length = extract_strings(string_components, NULL, &wide);
1588       s->ddecl = declare_string(NULL, wide, total_length);
1589       s->type = s->ddecl->type;
1590       extract_strings(string_components, (wchar_t *)s->ddecl->chars, &wide);
1591       s->static_address = foldaddress_string(s);
1592       s->lvalue = TRUE;
1593     
1594       return s;
1595     }
