161     /* Find equivalence class representative of al */
162     static aloc ecr_aloc(aloc al)
163     {
164       if (al->kind == aloc_link)
165         {
166           aloc ecr = al, cur, temp;
167           
168           /* Find root */
169           while (ecr->kind == aloc_link)
170     	ecr = ecr->u.link;
171           
172           /* Compress path */
173           cur = al;
174           while (cur->u.link != ecr)
175     	{
176     	  temp = cur->u.link;
177     	  cur->u.link = ecr;
178     	  cur = temp;
179     	}
180           return ecr;
181         }
182       else
183         return al;
184     }

209     /* Add constraint that aloc al1 = al2 */
210     void unify_aloc(location loc, aloc al1, aloc al2)
211     {
212       if (!restrict_qual && !flag_flow_sensitive)
213         return;
214     
215     #ifdef DEBUG
216       print_aloc(printf, al1);
217       printf(" %p == ", ecr_aloc(al1));
218       print_aloc(printf, al2);
219       printf(" %p\n", ecr_aloc(al2));
220     #endif
221     
222       al1 = ecr_aloc(al1);
223       al2 = ecr_aloc(al2);
224     
225       assert(al1->kind == aloc_var && al2->kind == aloc_var);
226       /* How should this go? */
227       /*  assert(!al1->u.var.fs_points_to && !al2->u.var.fs_points_to);*/
228       if (al1 != al2)
229         {
230           aloc new_ecr, new_link;
231     
232           if (al1->u.var.num_equiv <= al2->u.var.num_equiv)
233     	{
234     	  new_ecr = al2;
235     	  new_link = al1;
236     	}
237           else
238     	{
239     	  new_ecr = al1;
240     	  new_link = al2;
241     	}
242     
243           if (new_ecr->u.var.effects && new_link->u.var.effects)
244     	unify_constrs(new_ecr->u.var.effects, new_link->u.var.effects);
245           else if (new_link->u.var.effects)
246     	new_ecr->u.var.effects = new_link->u.var.effects;
247     
248           if (flag_flow_sensitive &&
249     	  !match_qtype(new_ecr->u.var.points_to, new_link->u.var.points_to,
250     		       NULL, NULL))
251     	{
252     	  report_qerror(loc, sev_err,
253     			"inconsistent location type for flow-sensitive pass");
254     	  new_ecr->u.var.points_to = error_qtype;
255     	}
256           new_ecr->u.var.num_equiv += new_link->u.var.num_equiv;
257           new_ecr->u.var.interesting =
258     	new_ecr->u.var.interesting || new_link->u.var.interesting;
259           new_link->kind = aloc_link;      
260           new_link->u.link = new_ecr;
261           /* Don't do anything for `points_to', since we only care about the
262              shape, anyhow, and the flow-insensitive analysis will make sure
263              they match. */
264     #ifdef DEBUG
265       printf("    result points to qtype ");
266       print_qtype_raw(printf, ecr_aloc(new_ecr)->u.var.points_to, NULL);
267       printf("\n");
268     #endif
269        }
270     }

326     struct Effect
327     {
328       enum { eff_constr, eff_union, eff_inter, eff_var, eff_link } kind;
329       int visited;
330       union
331       {
332         struct {
333           eff_kind kind;
334           aloc base;      /* This effect is kind(base) */
335           effect_set ub;  /* Upper bounds.  Variables only. */
336         } constr;
337         struct {
338           effect e1;
339           effect e2;
340         } u;              /* e1 \cup e2 */
341         struct {
342           effect e1;
343           effect e2;
344           effect_set ub;
345           int visited1, visited2;
346         } inter;          /* e1 \cap e2 */
347         struct {
348           const char *name;
349           effect_set lb, ub;     /* lb contains variables and constants,
350     			        ub contains variables and intersections */
351           int num_equiv;
352           bool interesting;      /* true iff we should fill in reach_cache */
353           aloc_int_map reach_cache; /* Cache of reachability computation result */
354     #ifdef STATS
355           int lb_size;      /* Size of lower-bound set (if any) */
356           int lowlink;      /* dfs number of root in scc */
357     #endif
358         } var;
359         effect link;
360       } u;
361     };

363     /* Find the equivalence class representative of e */
364     static effect ecr_effect(effect e)
365     {
366       if (e == NULL)
367         return e;
368       
369       if (e->kind == eff_link)
370         {
371           effect ecr = e, cur, temp;
372           
373           /* Find root */
374           while (ecr && ecr->kind == eff_link)
375     	ecr = ecr->u.link;
376           
377           /* Compress path */
378           cur = e;
379           while (cur->u.link != ecr)
380     	{
381     	  temp = cur->u.link;
382     	  cur->u.link = ecr;
383     	  cur = temp;
384     	}
385           return ecr;
386         }
387       else
388         return e;
389     }

416     /* The effect k(al) */
417     effect effect_constr(aloc al, eff_kind k)
418     {
419       int i;
420       effect *effects;
421     
422       assert(state == state_init);
423       if (!restrict_qual && !flag_flow_sensitive)
424         return NULL;
425     
426       al = ecr_aloc(al);
427       assert(al->kind == aloc_var);
428       assert(0 <= k && k <= eff_last);
429       if (!al->u.var.effects)
430         {
431           al->u.var.effects = rarrayalloc(effect_region, eff_last + 1, effect);
432           for (i = 0; i <= eff_last; i++)
433     	al->u.var.effects[i] = NULL;
434         }
435     
436       effects = al->u.var.effects;
437     
438       if (!effects[k])
439         {
440           effect e;
441     
442           e = ralloc(effect_region, struct Effect);
443           e->kind = eff_constr;
444           e->visited = 0;
445           e->u.constr.kind = k;
446           e->u.constr.base = al;
447           e->u.constr.ub = empty_effect_set();
448           effects[k] = e;
449     #ifdef STATS
450           dd_add_last(effect_region, effect_list, e);
451     #endif
452         }
453     
454       return effects[k];
455     }

605     /* The effect e1 \cap e2.  e2 cannot be the empty effect. */
606     effect effect_inter(effect e1, effect e2)
607     {
608       assert(state == state_init);
609       if (!restrict_qual && !flag_flow_sensitive)
610         return NULL;
611     
612       e1 = ecr_effect(e1);
613       e2 = ecr_effect(e2);
614     
615       assert(e2 != NULL);
616     
617       if (e1 == NULL)
618         return effect_empty;
619       else
620         {
621           effect e;
622     
623           assert(e2 != NULL);
624     
625           /* Get rid of any unions in e1 and e2 */
626           if (e1->kind == eff_union)
627     	{
628     	  effect temp;
629     
630     	  temp = effect_fresh();
631     	  mkleq_effect(e1, temp);
632     	  e1 = temp;
633     	}
634           if (e2->kind == eff_union)
635     	{
636     	  effect temp;
637     
638     	  temp = effect_fresh();
639     	  mkleq_effect(e2, temp);
640     	  e2 = temp;
641     	}
642     
643           e = ralloc(effect_region, struct Effect);
644           e->kind = eff_inter;
645           e->u.inter.e1 = e1;
646           e->u.inter.e2 = e2;
647           e->u.inter.ub = empty_effect_set();
648           e->u.inter.visited1 = 0;
649           e->u.inter.visited2 = 0;
650           e->visited = 0;
651     
652           add_to_effect_ub(e1, e);
653           add_to_effect_ub(e2, e);
654     
655     #ifdef STATS
656           dd_add_last(effect_region, effect_list, e);
657     #endif
658           return e;
659         }
660     }

677     /* Add constraint that effect set e1 is a subset of effect set e2. */
678     void internal_mkleq_effect(effect e1, effect e2)
679     {
680       e1 = ecr_effect(e1);
681       e2 = ecr_effect(e2);
682     
683       assert(e2->kind == eff_var);
684     
685       if (e1 == NULL)
686         return;
687       else
688         switch (e1->kind)
689           {
690           case eff_union:
691     	internal_mkleq_effect(e1->u.u.e1, e2);
692     	internal_mkleq_effect(e1->u.u.e2, e2);
693     	break;
694           case eff_constr:
695     	effect_set_insert(effect_region, &e1->u.constr.ub, e2);
696     	effect_set_insert(effect_region, &e2->u.var.lb, e1);
697     	break;
698           case eff_var:
699     	effect_set_insert(effect_region, &e1->u.var.ub, e2);
700     	effect_set_insert(effect_region, &e2->u.var.lb, e1);
701     	break;
702           case eff_inter:
703     	effect_set_insert(effect_region, &e1->u.inter.ub, e2);
704     	effect_set_insert(effect_region, &e2->u.var.lb, e1);
705     	break;
706           default:
707     	fail("Unexpected effect kind %x\n", e1->kind);
708           }
709     }

739       printf("\n");

742       internal_mkleq_effect(e1, e2);
743       internal_mkleq_effect(e2, e1);

746     /* Add constraint that effect sets e1 and e2 are equal. */
747     void unify_effect(effect e1, effect e2)
748     {
749       effect new_ecr, new_link;
750     
751       if (!restrict_qual && !flag_flow_sensitive)
752         return;
753     
754     #ifdef DEBUG
755       print_effect(printf, e1);
756       printf(" == ");
757       print_effect(printf, e2);
758       printf("\n");
759     #endif
760     
761       e1 = ecr_effect(e1);
762       e2 = ecr_effect(e2);
763     
764       if (e1 == e2)
765         return;
766     
767       assert(e1->kind == eff_var && e2->kind == eff_var);
768       assert(state == state_init);
769       assert(!e1->u.var.reach_cache && !e2->u.var.reach_cache);
770     
771       mkeq_effect(e1, e2);
772     
773       if (e1->u.var.num_equiv <= e2->u.var.num_equiv)
774         {
775           new_ecr = e2;
776           new_link = e1;
777         }
778       else
779         {
780           new_ecr = e1;
781           new_link = e2;
782         }
783     
784       new_ecr->u.var.num_equiv += new_link->u.var.num_equiv;
785       new_ecr->u.var.lb = effect_set_union(new_ecr->u.var.lb,
786     				       new_link->u.var.lb);
787       new_ecr->u.var.ub = effect_set_union(new_ecr->u.var.ub,
788     				       new_link->u.var.ub);
789       new_ecr->u.var.interesting =
790         new_ecr->u.var.interesting || new_link->u.var.interesting;
791       new_link->kind = eff_link;
792       new_link->u.link = new_ecr;
793     }

796     static void unify_constrs(effect *c1, effect *c2)
797     {
798       int i;
799     
800       for (i = 0; i <= eff_last; i++)
801         {
802           effect e1, e2;
803     
804           e1 = ecr_effect(c1[i]);
805           e2 = ecr_effect(c2[i]);
806           if (!e1)
807     	c1[i] = e2;
808           else if (!e2)
809     	c2[i] = e1;
810           else
811     	{
812     	  assert(e1->kind == eff_constr && e2->kind == eff_constr &&
813     		 e1->u.constr.kind == e2->u.constr.kind);
814     	  if (e1 != e2)
815     	    {
816     	      e1->u.constr.ub = effect_set_union(e1->u.constr.ub,
817     						 e2->u.constr.ub);
818     	      e2->kind = eff_link;
819     	      e2->u.link = e1;
820     	    }
821     	}
822         }
823     }

947     /* Return true if there is a path (respecting intersections) from e1
948        to e2.  e2 must be a variable. */
949     static bool reachable_forward(effect e1, effect e2) {
950       effect_set_scanner ss;
951       effect b;
952     
953       e1 = ecr_effect(e1);
954       e2 = ecr_effect(e2);
955     
956       assert(e2->kind == eff_var);
957       /*
958       printf("  ");
959       print_effect(printf, e1);
960       printf("\n");
961       */
962       if (e1 == e2)
963         return TRUE;
964       else
965         switch (e1->kind)
966           {
967           case eff_constr:
968     	{
969     	  assert(e1->visited == 0);
970     	  e1->visited = 1;
971     	  scan_effect_set(b, ss, e1->u.constr.ub)
972     	    if (reachable_forward(b, e2))
973     	      return TRUE;
974     	}
975     	break;
976           case eff_inter:
977     	{
978     	  if (e1->visited == 0)
979     	    e1->visited = 1;
980     	  else if (e1->visited == 1)
981     	    {
982     	      e1->visited = 2;
983     	      scan_effect_set(b, ss, e1->u.inter.ub)
984     		if (reachable_forward(b, e2))
985     		  return TRUE;
986     	    }
987     	}
988     	break;
989           case eff_var:
990     	{
991     	  if (!e1->visited)
992     	    {
993     	      e1->visited = 1;
994     	      scan_effect_set(b, ss, e1->u.var.ub)
995     		if (reachable_forward(b, e2))
996     		  return TRUE;
997     	    }
998     	}
999     	break;
1000           default:
1001     	fail("Unexpected effect kind %x\n", e1->kind);
1002           }
1003       return FALSE;
1004     }

1006     /* Set visited = 0 for everything reachable forwards from e.  Also
1007        fills in e's reach_cache (if any) */
1008     static void clean_reachable_forward(aloc al, effect e) {
1009       effect_set_scanner ss;
1010       effect b;
1011     
1012       e = ecr_effect(e);
1013     
1014       if (e->visited == 0 &&
1015           (e->kind != eff_inter || (e->u.inter.visited1 == 0 &&
1016     				e->u.inter.visited2 == 0)))
1017         return;
1018     
1019       switch (e->kind)
1020         {
1021         case eff_constr:
1022           e->visited = 0;
1023           scan_effect_set(b, ss, e->u.constr.ub)
1024     	clean_reachable_forward(al, b);
1025           break;
1026         case eff_inter:
1027           e->visited = 0;
1028           e->u.inter.visited1 = 0;
1029           e->u.inter.visited2 = 0;
1030           scan_effect_set(b, ss, e->u.inter.ub)
1031     	clean_reachable_forward(al, b);
1032           break;
1033         case eff_var:
1034           if (e->u.var.interesting)
1035     	{
1036     	  if (!e->u.var.reach_cache)
1037     	    e->u.var.reach_cache = make_aloc_int_map(effect_region, 17);
1038     	  aloc_int_map_insert(e->u.var.reach_cache, al, e->visited);
1039     	}
1040           e->visited = 0; /* After we insert in the cache */
1041           scan_effect_set(b, ss, e->u.var.ub)
1042     	clean_reachable_forward(al, b);
1043           break;
1044         default:
1045           fail("Unexpected effect kind %x\n", e->kind);
1046         }
1047     }

1049     /* Mark ub and everything reachable from ub with mark.  lb is where we
1050        came from on the traversal. */
1051     static void mark_reachable_forward(effect lb, effect ub, int mark)
1052     {
1053       effect_set_scanner ss;
1054       effect b;
1055     
1056       /*
1057       printf("  reaches ");
1058       print_effect(printf, ub);
1059       printf(" as %x\n", mark);
1060       */
1061       ub = ecr_effect(ub);
1062       switch(ub->kind)
1063         {
1064         case eff_constr:
1065           {
1066     	assert(!ub->visited);
1067     	ub->visited = mark;
1068     	scan_effect_set(b, ss, ub->u.constr.ub)
1069     	  mark_reachable_forward(ub, b, mark);
1070           }
1071           break;
1072         case eff_inter:
1073           {
1074     	int old_mark = ub->visited;
1075     
1076     	if (eq_effect(lb, ub->u.inter.e1))
1077     	  ub->u.inter.visited1 |= mark;
1078     	if (eq_effect(lb, ub->u.inter.e2))
1079     	  ub->u.inter.visited2 |= mark;
1080     
1081     	ub->visited = ub->u.inter.visited1 & ub->u.inter.visited2;
1082     	if (ub->visited != old_mark)
1083     	  scan_effect_set(b, ss, ub->u.var.ub)
1084     	    mark_reachable_forward(ub, b, ub->visited);
1085           }
1086           break;
1087         case eff_var:
1088           {
1089     	int old_mark = ub->visited;
1090     	ub->visited |= mark;
1091     	if (ub->visited != old_mark)
1092     	  scan_effect_set(b, ss, ub->u.var.ub)
1093     	    mark_reachable_forward(ub, b, mark);
1094           }
1095           break;
1096         default:
1097           fail("Unexpected effect kind %x\n", ub->kind);
1098         }
1099

1102     /* Fill in the reach caches for all variables and intersections */
1103     static void compute_reachable_forward(aloc al)
1104     {
1105       /*
1106       printf("Reachability of ");
1107       print_unique_aloc(printf, al);
1108       printf("\n");
1109       */
1110       al = ecr_aloc(al);
1111       assert(al->kind == aloc_var);
1112       assert(!al->u.var.reach_computed);
1113       if (al->u.var.effects)
1114         {
1115           effect any, rwr, r, wr, alloc;
1116     
1117           any = al->u.var.effects[eff_any];
1118           rwr = al->u.var.effects[eff_rwr];
1119           r = al->u.var.effects[eff_r];
1120           wr = al->u.var.effects[eff_wr];
1121           alloc = al->u.var.effects[eff_alloc];
1122           if (any)
1123     	mark_reachable_forward(effect_empty, any, MARK_ANY);
1124           if (rwr)
1125     	mark_reachable_forward(effect_empty, rwr, MARK_RWR);
1126           if (r)
1127     	mark_reachable_forward(effect_empty, r, MARK_R);
1128           if (wr)
1129     	mark_reachable_forward(effect_empty, wr, MARK_WR);
1130           if (alloc)
1131     	mark_reachable_forward(effect_empty, alloc, MARK_ALLOC);
1132           if (alloc)
1133     	clean_reachable_forward(al, alloc);
1134           if (rwr)
1135     	clean_reachable_forward(al, rwr);
1136           if (r)
1137     	clean_reachable_forward(al, r);
1138           if (wr)
1139     	clean_reachable_forward(al, wr);
1140           if (any)
1141     	clean_reachable_forward(al, any);
1142         }
1143       al->u.var.reach_computed = TRUE;
1144     }

1146     /* Return the way by which al reaches e in the least solution.
1147        Possible return values are 0 (unreachable), MARK_ANY, MARK_RWR,
1148        MARK_R, MARK_WR, and MARK_ALLOC. */
1149     static int aloc_in_effect(aloc al, effect e)
1150     {
1151       int mark;
1152     
1153       state = state_reach;
1154       if (!restrict_qual && !flag_flow_sensitive)
1155         return FALSE;
1156     
1157       al = ecr_aloc(al);
1158       assert(al->kind == aloc_var);
1159       e = ecr_effect(e);
1160       mark = 0; /* Assume not reachable */
1161       switch (e->kind)
1162         {
1163         case eff_constr:
1164           if (e->u.constr.kind == eff_any)
1165     	mark = MARK_ANY;
1166           else if (e->u.constr.kind == eff_rwr)
1167     	mark = MARK_RWR;
1168           else if (e->u.constr.kind == eff_r)
1169     	mark = MARK_R;
1170           else if (e->u.constr.kind == eff_wr)
1171     	mark = MARK_WR;
1172           else if (e->u.constr.kind == eff_alloc)
1173     	mark = MARK_ALLOC;
1174           else
1175     	fail("Unexpected constr kind %x\n", e->u.constr.kind);
1176           break;
1177         case eff_union:
1178           mark = aloc_in_effect(al, e->u.u.e1) | aloc_in_effect(al, e->u.u.e2);
1179           break;
1180         case eff_inter:
1181           mark = (aloc_in_effect(al, e->u.inter.e1) &
1182     	      aloc_in_effect(al, e->u.inter.e2));
1183           break;
1184         case eff_var:
1185           assert(e->u.var.interesting); /* can only check interesting effects */
1186           if (!al->u.var.reach_computed)
1187     	compute_reachable_forward(al);
1188           if (e->u.var.reach_cache)
1189     	/* If lookup finds nothing, mark is 0 */
1190     	aloc_int_map_lookup(e->u.var.reach_cache, al, &mark);
1191           break;
1192         default:
1193           fail("Unexpected effect kind %x\n", e->kind);
1194         }
1195       return mark;
1196     }

1226     /* Call check_nin_aloc_effect on all the constraints generated by
1227        mknin_aloc_effect and mknin_effect.  Generates error messages with
1228        calls to report_qerror. */
1229     /* Also checks forall constraints */
1230     void check_nin(void)
1231     {
1232       dd_list_pos cur;
1233     
1234       /* forall constraints */
1235       /* This ends up solving reachability for all aloc. */
1236       if (dd_length(forall_constraints) > 0 ||
1237           dd_length(forall_cond_constraints) > 0) {
1238         dd_scan(cur, aloc_list)
1239           {
1240     	aloc al;
1241     	dd_list_pos curconst;
1242     	al = ecr_aloc(DD_GET(aloc, cur));
1243           
1244     	assert(al->kind == aloc_var);
1245     	if (!al->u.var.reach_computed)
1246     	  compute_reachable_forward(al);
1247           
1248     	dd_scan(curconst, forall_constraints)
1249     	  {
1250     	    forall_nin_constraint c = DD_GET(forall_nin_constraint, curconst);
1251      
1252     	    if (aloc_in_effect(al, c->eff) & get_mask(c->kind))
1253     	      report_qerror(c->loc, sev_err, 
1254     			    "Invalid use of restrict - side effect");
1255     	  }
1256     
1257     	dd_scan(curconst, forall_cond_constraints)
1258     	  {
1259     	    forall_cond_nin_constraint c = DD_GET(forall_cond_nin_constraint, 
1260     						  curconst);
1261     	  
1262     	    if ((aloc_in_effect(al, c->eff1) & get_mask(c->kind1)) &&
1263     		(aloc_in_effect(al, c->eff2) & get_mask(c->kind2)))
1264     	      report_qerror(c->loc, sev_err, 
1265     			    "Invalid use of restrict - readonly written");
1266     	  }
1267           }
1268       }
1269     
1270       /* The rest */
1271       dd_scan(cur, constraints)
1272         {
1273           nin_constraint c = DD_GET(nin_constraint, cur);
1274     #ifdef DEBUG
1275           print_aloc(printf, c->aloc);
1276           printf(" !<= ");
1277           print_effect(printf, c->eff);
1278           printf("\n");
1279     #endif
1280           if (aloc_in_effect(c->aloc, c->eff) & get_mask(c->kind))
1281     	{
1282     #ifdef DEBUG
1283     	  printf("Invalid constraint ");
1284     	  print_aloc(printf, c->aloc);
1285     	  printf(" !<= ");
1286     	  print_effect(printf, c->eff);
1287     	  printf("\n");
1288     #endif
1289     	  report_qerror(c->loc, sev_err,
1290     			"Invalid use of restrict");
1291     	}
1292         }
1293     }

1472     static store ecr_store(store s) 
1473     {
1474       if (!s) return s;
1475     
1476       if (s->kind == store_link)
1477         {
1478           store ecr = s, cur, temp;
1479           
1480           /* Find root */
1481           while (ecr->kind == store_link)
1482     	ecr = ecr->u.link;
1483           
1484           /* Compress path */
1485           cur = s;
1486           while (cur->u.link != ecr)
1487     	{
1488     	  temp = cur->u.link;
1489     	  cur->u.link = ecr;
1490     	  cur = temp;
1491     	}
1492           return ecr;
1493         }
1494       else
1495         return s;
1496     }

1693     /* Add constraint left <= right */
1694     void mkleq_store(location loc, const char *err_msg, store left, store right)
1695     {
1696       bool err;
1697       cell_map_scanner cms;
1698       cell right_cell, left_cell;
1699       aloc al;
1700       region scratch_region;
1701     
1702       assert(state == state_init || state == state_reach);
1703     
1704     #ifdef DEBUGST
1705       print_store(printf, left);
1706       printf(" ");
1707       print_store_cells(printf, print_qual, left, TRUE);
1708       printf(" <= ");
1709       print_store(printf, right);
1710       printf(" ");
1711       print_store_cells(printf, print_qual, right, TRUE);
1712       printf("\n");
1713     #endif
1714       right = ecr_store(right);
1715       left = ecr_store(left);
1716       assert(right->kind == store_var);
1717       
1718       if (right == left)
1719         return;
1720     
1721       /* Add constraints between elements of left and elements of right */
1722       err = FALSE;
1723     
1724       /* Make constraints between all elements in right to left */
1725       cell_map_scan(right->u.var.cells, &cms);
1726       while (cell_map_next(&cms, &al, &right_cell))
1727         if (right_cell->propagated_backward)
1728           {
1729     	store left_store;
1730     	
1731     	left_cell = cell_from_store(left, al, &left_store, FALSE);
1732     	if (left_cell)
1733     	  {
1734     	    /* Could be NULL if lb is a filter store */
1735     
1736     	    /* Ensure al is in left */
1737     	    /* propagate_store_cell_backward(left_store, al); */
1738     
1739     	    err = mkleq_qtype(loc, left_cell->qtype, right_cell->qtype) || err;
1740     	  }
1741           }
1742     
1743       /* Find all the elements in left that aren't in right, add them to right,
1744          and make constraints */
1745       scratch_region = newregion();
1746       cell_map_scan(store_to_cells(scratch_region, left), &cms);
1747       while (cell_map_next(&cms, &al, &left_cell))
1748         if (left_cell->propagated_forward)
1749           {
1750     	/* propagate_store_cell_forward(right, al); *//* Add al to right */
1751     	/* insist(cell_map_lookup(right->u.var.cells, al, &right_cell)); */
1752     	
1753     	if (!cell_map_lookup(right->u.var.cells, al, &right_cell))
1754     	  {
1755     	    qtype qt;
1756     	    
1757     	    qt = qtype_to_fs_qtype(right->loc, al->u.var.points_to);
1758     	    insistnot(store_aloc_qtype(qt, right, al));
1759     	    right_cell = mkcell(store_region, qt, FALSE, FALSE);
1760     	    insist(cell_map_insert(right->u.var.cells, al, right_cell));
1761     	  }
1762     	err = mkleq_qtype(loc, left_cell->qtype, right_cell->qtype) || err;
1763           }
1764       
1765       deleteregion(scratch_region);
1766     
1767       if (err)
1768         report_qerror(loc, sev_err, "%s", err_msg);
1769       store_edge_set_insert(store_region, &right->u.var.lb,
1770     			mkstore_edge(store_region, loc, err_msg, left));
1771       store_edge_set_insert(store_region, &left->ub,
1772     			mkstore_edge(store_region, loc, err_msg, right));
1773     }

1798     /* Add constraint left == right, and can't distinguish left from right
1799        afterward. */
1800     void unify_store(location loc, const char *err_msg, store s1, store s2)
1801     {
1802       store new_ecr, new_link;
1803       cell_map_scanner cms;
1804       aloc al;
1805       cell link_cell, ecr_cell;
1806     
1807     #ifdef DEBUGST
1808       print_store(printf, s1);
1809       printf(" == ");
1810       print_store(printf, s2);
1811       printf("\n");
1812     #endif
1813       s1 = ecr_store(s1);
1814       s2 = ecr_store(s2);
1815       assert(s1->kind == store_var && s2->kind == store_var);
1816     
1817       if (s1 == s2)
1818         return;
1819     
1820       if (s1->u.var.num_equiv <= s2->u.var.num_equiv)
1821         {
1822           new_ecr = s2;
1823           new_link = s1;
1824         }
1825       else
1826         {
1827           new_ecr = s1;
1828           new_link = s2;
1829         }
1830     
1831       /* Add constraints between propagating elements of left and propagating
1832          elements of right */
1833       mkleq_store(loc, err_msg, new_link, new_ecr);
1834       mkleq_store(loc, err_msg, new_ecr, new_link);
1835     
1836       /* Add non-propagating elements of new_link to new_ecr.  Note that
1837          because these are non-propagating, we don't need to generate any
1838          constraints */
1839       cell_map_scan(new_link->u.var.cells, &cms);
1840       while (cell_map_next(&cms, &al, &link_cell))
1841         if (!link_cell->propagated_forward && !link_cell->propagated_backward)
1842           {
1843     	if (cell_map_lookup(new_ecr->u.var.cells, al, &ecr_cell))
1844     	  {
1845     	    assert(ecr_cell->propagated_forward == link_cell->propagated_forward);
1846     	    assert(ecr_cell->propagated_backward == link_cell->propagated_backward);
1847     	    insistnot(unify_qtype(loc, link_cell->qtype, ecr_cell->qtype));
1848     	  }
1849     	else
1850     	  insist(cell_map_insert(new_ecr->u.var.cells, al, link_cell));
1851           }
1852     
1853     
1854       new_ecr->u.var.lb = store_edge_set_union(new_ecr->u.var.lb,
1855     					   new_link->u.var.lb);
1856       new_ecr->u.var.num_equiv += new_link->u.var.num_equiv;
1857       new_ecr->ub = store_edge_set_union(new_ecr->ub, new_link->ub);
1858       new_link->kind = store_link;
1859       new_link->u.link = new_ecr;;
1860     }

1925     /* subroutine for compute_lin_scc */
1926     static int compute_lin_scc_sub(aloc al, store parent, store child);
1927     
1928     /* computes lin at al for all stores reachable from s */
1929     static int compute_lin_scc(aloc al, store s)
1930     {
1931       s->lowlink = s->visited = ++scc_store_number;
1932       scc_store_stack_push(s);
1933     
1934     #ifdef DEBUG
1935       printf("computing lin of ");
1936       print_aloc(printf, al);
1937       printf(" at ");
1938       print_store(printf, s);
1939       printf("\n");
1940     #endif
1941     
1942       switch (s->kind)
1943         {
1944         case store_var:
1945           {
1946     	store_edge_set_scanner sess;
1947     	store_edge b;
1948     	int lin, max;
1949     	
1950     	max = 0;
1951     	scan_store_edge_set(b, sess, s->u.var.lb)
1952     	  {
1953     	    b->store = ecr_store(b->store);
1954     	    lin = compute_lin_scc_sub(al, s, b->store);
1955     
1956     	    max = MAX(max, lin);
1957     	  }
1958     	s->lin = max;
1959           }
1960           break;
1961         case store_filter:
1962           {
1963     	effect e;
1964     	    
1965     	e = ecr_effect(s->u.filter.e);
1966     	if (e && aloc_in_effect(al, e))
1967     	  s->lin = compute_lin_scc_sub(al, s, ecr_store(s->u.filter.s));
1968     	else
1969     	  s->lin = 0;
1970           }
1971           break;
1972         case store_ref:
1973           {
1974     	s->lin = compute_lin_scc_sub(al, s, ecr_store(s->u.ref.s));
1975     	
1976     	if (eq_aloc(s->u.ref.al, al))
1977     	  s->lin = MIN(s->lin + 1, 2);
1978           }
1979           break;
1980         case store_ow:
1981           {
1982     	effect e;
1983     	    
1984     	e = ecr_effect(s->u.ow.e);
1985     	if (e && (aloc_in_effect(al, e) & MARK_ALLOC))
1986     	  s->lin = compute_lin_scc_sub(al, s, ecr_store(s->u.ow.s1));
1987     	else
1988     	  s->lin = compute_lin_scc_sub(al, s, ecr_store(s->u.ow.s2));
1989           }
1990           break;
1991         case store_assign:
1992           {
1993     	s->lin = compute_lin_scc_sub(al, s, ecr_store(s->u.assign.s));
1994           }
1995           break;
1996         default:
1997           fail("Unexpected store kind %d\n", s->kind);
1998         }
1999     
2000       if (s->visited == s->lowlink)
2001         {
2002           bool ref = FALSE;
2003           store c;
2004           int i, max, count = 0;
2005     
2006           /* look for a ref node in the scc found */
2007           max = 0;
2008           for (i = scc_store_top - 1; i >= 0; i--)
2009     	{
2010     	  c = scc_store_stack[i];
2011     	  if (c->kind == store_ref && eq_aloc(c->u.ref.al, al))
2012     	    ref = TRUE;
2013     	  max = MAX(max, c->lin);
2014     	  count++;
2015     	  if (c->visited <= s->visited)
2016     	    break;
2017     	}
2018           
2019           /* increasing cycle found */
2020           if (ref && count > 1)
2021     	max = 2;
2022     
2023           do {
2024     	c = scc_store_stack_pop();
2025     	c->lin = max;
2026     #ifdef DEBUG

2271     			 s->u.assign.cell->qtype,

2520           else
2521     	propagate_to_ub = eq_store(lb, ub->u.ow.s2);
2522           break;

2524           if (!eq_aloc(al, ub->u.assign.al))
2525     	/* assign stops forward propagation of al */

2540     	result = propagate_cell_forward(loc, ub, ub_ub->store, al, qt)
2541     	|| result;
2542         }

3106       assert(e->kind == eff_var && !e->visited);
3107       number = ++scc_number;

3109       dd_add_first(scc_region, scc_stack, e);

3111       /* Process all out-edges of e */

3113         {

3116     	{

3124     	    }

3134     	     happen to take, so do nothing */
