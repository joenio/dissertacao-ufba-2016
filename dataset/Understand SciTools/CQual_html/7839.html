213     	fclose(graph_file);
214           graph_file = fopen("quals.dot", "w");
215           fprintf(graph_file, "digraph G {\n");
216         }
217       state = state_init;

292     /* Construct the bitset for q and initailize the bitset hash table
293        with entries for q's bitset */
294     static void construct_qual_bitset(qual q)
295     {
296       qual_set_scanner qs;
297       qual p;
298       bitset b, up, down;
299       bounds bounds;
300     
301       assert(q->kind == qk_constant);
302       if (nonprop_qual(q))
303         return;
304     
305       /* Construct bitset for q */
306     
307       /* Assume initially q consistent with all qualifiers */
308       b = bitset_new(quals_region, max_index);
309       up = bitset_new(quals_region, max_index);
310       down = bitset_new(quals_region, max_index);
311       bitset_insert_all(b);
312       bitset_insert_all(up);
313       bitset_insert_all(down);
314     
315       /* Remove any qualifiers that are inconsistent with q.  Note that
316          these can only be qualifiers in q's partial order. */
317       scan_qual_set(p, qs, q->u.elt.po->qualifiers)
318         {
319           if (!eq_qual(q, p))
320     	insist(bitset_remove(b, p->u.elt.index));
321           if (!leq_qual(q, p))
322     	insist(bitset_remove(up, p->u.elt.index));
323           if (!leq_qual(p, q))
324     	insist(bitset_remove(down, p->u.elt.index));
325         }
326     
327       /*
328       printf("bitset for %s: ", q->u.elt.name);
329       bitset_print_pretty(b);
330       printf("\n");
331       */
332     
333       /* Add computed bounds to hash table */
334       bounds = ralloc(quals_region, struct Bounds);
335       bounds->bitset = b;
336       bounds->up = up;
337       bounds->down = down;
338       q->bounds = bounds;
339       insist(bitset_bounds_map_insert(bitset_hash_table, b, bounds));
340     
341       /*
342       printf("upward closure of %s: ", q->u.elt.name);
343       bitset_print_pretty(up);

345       bitset_print_pretty(down);
346       printf("\n");

348     }

350     /* Return bounds set for b.  b may be safely deallocated after calling
351        this function. */
352     static bounds find_bounds(bitset b)
353     {
354       bounds bounds;
355     
356       if (!bitset_bounds_map_lookup(bitset_hash_table, b, &bounds))
357         {
358           /* Not in table, so make a new entry. */
359           qual_set_scanner qs;
360           qual p;
361     
362           bounds = ralloc(quals_region, struct Bounds);
363           bounds->bitset = bitset_copy(quals_region, b);
364           bounds->up = bitset_new(quals_region, max_index);
365           bounds->down = bitset_new(quals_region, max_index);
366     
367           scan_qual_set(p, qs, all_quals)
368     	if (bitset_member(b, p->u.elt.index))
369     	  {
370     	    /* Add any upper/lower bounds of p */
371     	    qual_set_scanner qs2;
372     	    qual p2;
373     
374     	    scan_qual_set(p2, qs2, p->u.elt.po->qualifiers)
375     	      {
376     		if (leq_qual(p, p2))
377     		  bitset_insert(bounds->up, p2->u.elt.index);
378     		if (leq_qual(p2, p))
379     		  bitset_insert(bounds->down, p2->u.elt.index);
380     	      }
381     	  }
382           insist(bitset_bounds_map_insert(bitset_hash_table, bounds->bitset,
383     				      bounds));
384     
385           /*
386           printf("constructed new bitset ");
387           bitset_print(bounds->bitset);
388           printf("\nupward closure: ");
389           bitset_print(bounds->up);
390           printf("\ndownward closure: ");
391           bitset_print(bounds->down);
392           printf("\n");
393           */
394         }
395       return bounds;
396     }

442     /* Make a new (constant) qualifier */
443     qual add_qual(const char *name)
444     {
445       /* qual_set_insert does pointer comparisons, not strcmps, so we need
446          to do a special search by name first. */
447       qual new_qual;
448     
449       assert(state != state_pos_defined);
450       if (!current_po)
451         fail("add_qual called without previous begin_po_qual\n");
452       if ((new_qual = find_qual(name)))
453         {
454           if (new_qual->u.elt.po != current_po)
455     	fail("Qualifier %s in two different partial orders\n",
456     	     new_qual->u.elt.name);
457           return new_qual;
458         }
459       /* We didn't find the qualifier */
460       /*  printf("Adding qualifier %s\n", name);*/
461       new_qual = ralloc(quals_region, struct type_qualifier);
462       new_qual->kind = qk_constant;
463       new_qual->mark = FALSE;
464       new_qual->u.elt.name = rstrdup(quals_region, name);
465       new_qual->u.elt.color = NULL;
466       new_qual->u.elt.level = level_value;
467       new_qual->u.elt.sign = sign_pos;
468       new_qual->u.elt.lbc = empty_qual_set();
469       new_qual->u.elt.ubc = empty_qual_set();
470       new_qual->u.elt.po = current_po;
471     #ifdef BITSET
472       if (!current_po->nonprop)
473         {
474           new_qual->u.elt.index = max_index++;
475           if (qual_set_empty(current_po->qualifiers))
476     	current_po->first = new_qual->u.elt.index;
477           current_po->last = new_qual->u.elt.index;
478         }
479     #endif
480       qual_set_insert(quals_region, &all_quals, new_qual);
481       qual_set_insert(quals_region, &current_po->qualifiers, new_qual);
482       if (!strcmp(name, "const"))
483         const_qual = new_qual;
484       else if (!strcmp(name, "$nonconst"))
485         nonconst_qual = new_qual;
486       else if (!strcmp(name, "volatile"))
487         volatile_qual = new_qual;
488       else if (!strcmp(name, "restrict"))
489         restrict_qual = new_qual;
490       else if (!strcmp(name, "noreturn"))
491         noreturn_qual = new_qual;
492       else if (!strcmp(name, "$init"))
493         init_qual = new_qual;
494       else if (!strcmp(name, "$noninit"))
495         noninit_qual = new_qual;
496       return new_qual;
497     }

499     /* Add assumption left < right */
500     void add_qual_lt(qual left, qual right)
501     {
502       assert(left->kind == qk_constant);
503       assert(right->kind == qk_constant);
504       if (!qual_set_member(left->u.elt.ubc, right))
505         {
506           qual_set_scanner qs;
507           qual q;
508     
509           /*
510           printf("Adding assumption ");
511           print_qual_raw(printf, left);
512           printf(" < ");
513           print_qual_raw(printf, right);
514           printf("\n");
515           */
516     
517           qual_set_insert(quals_region, &left->u.elt.ubc, right);
518           qual_set_insert(quals_region, &right->u.elt.lbc, left);
519     
520           /* Add transitively-implied assumptions */
521           scan_qual_set(q, qs, right->u.elt.ubc)
522     	add_qual_lt(left, q);
523     
524           scan_qual_set(q, qs, left->u.elt.lbc)
525     	add_qual_lt(q, right);
526         }
527     }

638     /* Associate (s, al) with q; return TRUE if this triggers a propagation */
639     bool store_aloc_qual(qual q, store s, aloc al)
640     {
641       q = ecr_qual(q);
642       assert(q->kind == qk_variable);
643       assert(!q->u.var.store && !q->u.var.aloc);
644       q->u.var.store = s;
645       q->u.var.aloc = al;
646       if (q->u.var.lbc)
647         propagate_store_cell_forward(s, al);
648       if (q->u.var.ubc)
649         propagate_store_cell_backward(s, al);
650     
651       if (!q->u.var.lbc && !q->u.var.ubc)
652         return FALSE;
653       else
654         return TRUE;
655     }

657     /* Return the ECR for this qualifier */
658     static qual ecr_qual(qual q)
659     {
660       switch (q->kind)
661         {
662         case qk_constant:
663         case qk_variable:
664           return q;
665         case qk_link:
666           {
667     	qual ecr = q, cur, temp;
668     	
669     	/* Find root */
670     	while (ecr->kind == qk_link)
671     	  ecr = ecr->u.link;
672     	
673     	/* Compress path */
674     	cur = q;
675     	while (cur->u.link != ecr)
676     	  {
677     	    temp = cur->u.link;
678     	    cur->u.link = ecr;
679     	    cur = temp;
680     	  }
681     	return ecr;
682           }
683           break;
684         default:
685           fail("Unexpected qual kind %x\n", q->kind);
686         }
687     }

950        iff an error occurred. */
951     bool mkleq_qual(location loc, qual left, qual right)
952     {
953       bool result;
954     
955       result = FALSE;
956       if (flag_print_quals_graph)
957         {
958           /*
959           fprintf(graph_file, "%p %p\n", left, right);
960           */
961           fprintf(graph_file, "\"");
962           print_qual_raw(print_graph_file, left);
963           if (flag_ugly)
964     	fprintf(graph_file, " %p\" -> \"", left);
965           else
966     	fprintf(graph_file, "\"->\"");
967           print_qual_raw(print_graph_file, right);
968           if (flag_ugly)
969     	fprintf(graph_file, " %p\"\n", right);
970           else
971     	fprintf(graph_file, "\"\n");
972         }
973     #ifdef DEBUG
974       print_qual_raw(printf, left);
975       printf(" %p <= ", left);
976       print_qual_raw(printf, right);
977       printf(" %p\n", right);
978     #endif
979       left = ecr_qual(left);
980       right = ecr_qual(right);
981       if (left == right)
982         /* Reflexivity */
983         return FALSE;
984     
985       if (left->kind == qk_constant && right->kind == qk_constant)
986         result = !leq_qual(left, right) || result;
987       else
988         {
989           if (left->kind == qk_variable)
990     	{
991     	  qual_edge qe;
992     
993     	  qe = mkqual_edge(quals_region, loc, right);
994     	  /*printf("insert in %s: %p\n", name_qual(left), qe);*/
995     	  qual_edge_set_insert(quals_region, &left->u.var.ub, qe);
996     	  if (right->bounds) /* Right may be non-prop qual constant */
997     	    result = update_var_bounds_left(left, right->bounds) || result;
998     	}
999     
1000           if (right->kind == qk_variable)
1001     	{
1002     	  qual_edge qe;
1003     
1004     	  qe = mkqual_edge(quals_region, loc, left);
1005     	  /*printf("insert in %s: %p\n", name_qual(right), qe);*/
1006     	  qual_edge_set_insert(quals_region, &right->u.var.lb, qe);
1007     	  if (left->bounds)  /* Left may be non-prop qual constant */
1008     	    result = update_var_bounds_right(right, left->bounds) || result;
1009     	}
1010         }
1011     
1012       if (result)
1013         {
1014           fflush(NULL);
1015           printf("Unsatisfiable ");
1016           print_qual_raw(printf, left);
1017           printf(" <= ");
1018           print_qual_raw(printf, right);
1019           printf("\n");
1020         }
1021     
1022       if (result)
1023         increment_error_counts(left, right);
1024     
1025       return result;
1026     }

1350     bool unify_qual(location loc, qual left, qual right)
1351     {
1352       bool result;
1353     
1354       left = ecr_qual(left);
1355       right = ecr_qual(right);
1356       if (left == right) /* Short circuit to prevent loops in ecr_qual */
1357         return FALSE;
1358     #ifdef DEBUG
1359       print_qual_raw(printf, left);
1360       printf(" %p = ", left);
1361       print_qual_raw(printf, right);
1362       printf(" %p\n", right);
1363     #endif
1364     
1365       result = mkeq_qual(loc, left, right);
1366       if (left->kind == qk_variable && right->kind == qk_variable)
1367         {
1368           qual new_ecr, new_link;
1369     
1370           if (right->u.var.preferred ||
1371     	  (left->u.var.num_equiv <= right->u.var.num_equiv))
1372     	{
1373     	  new_ecr = right;
1374     	  new_link = left;
1375     	}
1376           else
1377     	{
1378     	  new_ecr = left;
1379     	  new_link = right;
1380     	}
1381     
1382     #ifdef BITSET
1383           assert(new_ecr->bounds == new_link->bounds);
1384     #endif
1385           new_ecr->u.var.num_equiv += new_link->u.var.num_equiv;
1386           new_ecr->u.var.num_errors += new_link->u.var.num_errors;
1387           new_ecr->u.var.lb = qual_edge_set_union(new_ecr->u.var.lb,
1388     					      new_link->u.var.lb);
1389           new_ecr->u.var.ub = qual_edge_set_union(new_ecr->u.var.ub,
1390     					      new_link->u.var.ub);
1391           new_ecr->u.var.cond_set = cond_set_union(new_ecr->u.var.cond_set,
1392     					       new_link->u.var.cond_set);
1393           new_link->kind = qk_link;
1394           new_link->u.link = new_ecr;
1395         }
1396       return result;
1397     }

1500     /* Adds the conditional constraint l1<=r1 ==> l2<=r2.  This constraint
1501        is allowed only if at least one of {l1,r1} is a constant. */
1502     bool cond_mkleq_qual(location loc, qual l1, qual r1, qual l2, qual r2)
1503     {
1504       assert(constant_qual(l1) || constant_qual(r1));
1505     #ifdef DEBUG
1506       print_qual_raw(printf, l1);
1507       printf(" %p <= ", l1);
1508       print_qual_raw(printf, r1);
1509       printf(" %p  ==>  ", r1);
1510       print_qual_raw(printf, l2);
1511       printf(" %p <= ", l2);
1512       print_qual_raw(printf, r2);
1513       printf(" %p\n", r2);
1514     #endif
1515     
1516       l1 = ecr_qual(l1);
1517       r1 = ecr_qual(r1);
1518       l2 = ecr_qual(l2);
1519       r2 = ecr_qual(r2);
1520       if (leq_qual(l1, r1))
1521         /* Condition is true */
1522         return mkleq_qual(loc, l2, r2);
1523       else if (constant_qual(l1) && constant_qual(r1))
1524         /* Conditional will never be true -- ignore constraint */
1525         return FALSE;
1526       else
1527         {
1528           /* Condition is not true yet; add constraint to pending set */
1529           qual v, c;
1530           cond_dir dir;
1531     
1532           if (constant_qual(l1))
1533     	{ c = l1; v = ecr_qual(r1); dir = c_leq_v; }
1534           else
1535     	{ c = r1; v = ecr_qual(l1); dir = v_leq_c; }
1536           assert (constant_qual(c) && variable_qual(v));
1537           cond_set_insert(quals_region, &v->u.var.cond_set,
1538     		      mkcond(dir, c, l2, r2, loc));
1539           return FALSE;
1540         }
1541     }

1543     /* If there are any conditional constraint depending on q, trigger
1544        any newly satisfied ones */
1545     static bool cond_set_trigger(qual q)
1546     {
1547       bool result;
1548       cond_set_scanner css;
1549       cond cond;
1550     
1551       result = FALSE;
1552       q = ecr_qual(q);
1553       assert(q->kind == qk_variable);
1554       scan_cond_set(cond, css, q->u.var.cond_set)
1555         if (!cond->triggered)
1556           {
1557     	switch (cond->dir)
1558     	  {
1559     	  case c_leq_v:
1560     	    cond->triggered = leq_qual(cond->c, q);
1561     	    break;
1562     	  case v_leq_c:
1563     	    cond->triggered = leq_qual(q, cond->c);
1564     	    break;
1565     	  default:
1566     	    fail("Unexpected condition dir %x\n", cond->dir);
1567     	  }
1568     	if (cond->triggered)
1569     	  result = mkleq_qual(cond->loc, cond->left, cond->right) || result;
1570           }
1571     
1572       return result;
1573     }

1581     /* Apply f to all possible values of q.  f is applied to NULL in
1582        the error case. */
1583     void scan_qual_bounds(qual q, qual_traverse_fn f, void *arg)
1584     {
1585     #ifdef BITSET
1586       qual_set_scanner qs;
1587       qual p;
1588       po_set_scanner ss;
1589       po_info po;
1590     
1591       q = ecr_qual(q);
1592       assert(q->kind == qk_variable);
1593       scan_po_set(po, ss, all_pos)
1594         {
1595           if (!po->nonprop &&
1596     	  bitset_full_range(q->bounds->bitset, po->first, po->last))
1597     	/* Don't do anything if all quals in the partial order
1598     	   are possible */
1599     	;
1600           else if (!po->nonprop &&
1601     	       bitset_empty_range(q->bounds->bitset, po->first, po->last))
1602     	/* If none of the quals in the p.o. are possible, it's
1603     	   an error */
1604     	/* XXX: Hack to make look like lattice case */
1605     	scan_qual_set(p, qs, po->qualifiers)
1606     	  f(p, arg);
1607           else if (!po->nonprop)
1608     	{
1609     	  scan_qual_set(p, qs, po->qualifiers)
1610     	    if (bitset_member(q->bounds->bitset, p->u.elt.index))
1611     	      f(p, arg);
1612     	}
1613           else
1614     	{
1615     	  assert(po->nonprop);
1616     	  scan_qual_set(p, qs, po->qualifiers)
1617     	    if (has_qual(q, p))
1618     	      f(p, arg);
1619     	}
1620         }
1621     
1622     #else /* #ifdef BITSET */
1623       qual_set_scanner qs;
1624       qual p;
1625       qual_edge_set_scanner qess;
1626       qual_edge qe;
1627     
1628       q = ecr_qual(q);
1629       assert(q->kind == qk_variable);
1630       scan_qual_set(p, qs, q->u.var.lbc)
1631         if ((sign_qual(p) == sign_pos) || (sign_qual(p) == sign_eq))
1632           f(p, arg);
1633     
1634       scan_qual_set(p, qs, q->u.var.ubc)
1635         if (sign_qual(p) == sign_neg)    /* eq handled above */
1636           f(p, arg);
1637     
1638       /* Handle non-propagating qualifiers */
1639       scan_qual_edge_set(qe, qess, q->u.var.lb)
1640         if (constant_qual(qe->qual) &&
1641     	nonprop_qual(qe->qual) &&
1642     	((sign_qual(qe->qual) == sign_pos) ||
1643     	 (sign_qual(qe->qual) == sign_eq)))
1644           f(qe->qual, arg);
1645     
1646       scan_qual_edge_set(qe, qess, q->u.var.ub)
1647         if (constant_qual(qe->qual) &&
1648     	nonprop_qual(qe->qual) &&
1649     	sign_qual(qe->qual) == sign_neg) /* eq handled above */
1650           f(qe->qual, arg);
1651     #endif
1652     }

1812     /* Called for each bound of the qualifier we're interested in.  arg is
1813        a pointer to the current dir. */
1814     static void interesting_scan_dir_fn(qual q, void *arg)
1815     {
1816       dir *d = (dir *) arg;
1817     
1818       switch (sign_qual(q))
1819         {
1820         case sign_pos:
1821           *d |= dir_backward;
1822           break;
1823         case sign_neg:
1824           *d |= dir_forward;
1825           break;
1826         case sign_eq:
1827           *d |= dir_forward | dir_backward;
1828           break;
1829         default:
1830           fail("Unexpected sign %x\n", sign_qual(q));
1831         }
1832     }

1834     /* Given a qualifier q, return in which directions to look for interesting
1835        explicit qualifiers.  If c is given, only looks for directions to c. */
1836     static dir interesting_scan_dir(qual q, qual c)
1837     {
1838       dir result;
1839     
1840       result = dir_none;
1841       if (!c)
1842         /* Look for any interesting qualifier */
1843         scan_qual_bounds(q, interesting_scan_dir_fn, &result);
1844       else
1845         {
1846           /* See if q has c has a bound */
1847           if (!nonprop_qual(c))
1848     	{
1849     	  switch (sign_qual(c))
1850     	    {
1851     	    case sign_pos:
1852     	      if (leq_qual(c, q))
1853     		result = dir_backward;
1854     	      break;
1855     	    case sign_neg:
1856     	      if (leq_qual(q, c))
1857     		result = dir_forward;
1858     	      break;
1859     	    case sign_eq:
1860     	      if (leq_qual(c, q)) /* Should be eq_qual, but that's something
1861     				     else */
1862     		result = dir_backward | dir_forward;
1863     	      break;
1864     	    default:
1865     	      fail("Unexpected sign %x\n", sign_qual(c));
1866     	    }
1867     	}
1868           else
1869     	{
1870     	  /* Handle non-propagating qualifiers */
1871     	  qual_edge_set_scanner qess;
1872     	  qual_edge qe;
1873     
1874     	  switch (sign_qual(c))
1875     	    {
1876     	    case sign_pos:
1877     	      scan_qual_edge_set(qe, qess, q->u.var.lb)
1878     		if (leq_qual(c, qe->qual))
1879     		  result = dir_backward;
1880     	      break;
1881     	    case sign_neg:
1882     	      scan_qual_edge_set(qe, qess, q->u.var.ub)
1883     		if (leq_qual(qe->qual, c))
1884     		  result = dir_forward;
1885     	      break;
1886     	    case sign_eq:
1887     	      scan_qual_edge_set(qe, qess, q->u.var.lb)
1888     		if (leq_qual(c, qe->qual))
1889     		  result = dir_backward;
1890     	      break;
1891     	    default:
1892     	      fail("Unexpected sign %x\n", sign_qual(c));
1893     	    }
1894     	}
1895         }
1896       return result;
1897     }

1899     /* Erase all the marks marked as TRUE reachable via TRUE marked nodes
1900        from q. */

1902     {
1903       qual_edge qe;
1904       qual_edge_set_scanner qess;
1905     
1906       q = ecr_qual(q);
1907       if (!q->mark)
1908         return;
1909       q->mark = FALSE;
1910     
1911       if (q->kind == qk_variable)
1912         {
1913           scan_qual_edge_set(qe, qess, q->u.var.lb)
1914     	traverse_clear_marks(qe->qual);
1915           
1916           scan_qual_edge_set(qe, qess, q->u.var.ub)
1917     	traverse_clear_marks(qe->qual);
1918         }
1919       else
1920         assert(q->kind == qk_constant);

1923     /* Traverse the constraint graph starting from qualifier variable q,
1924        looking for constant qualifier c in direction d.  If such a path
1925        exists, set the parent pointers and return c.  Otherwise return
1926        NULL.  If c is NULL, look for any path to a constant qualifier, and
1927        return that qualifier. */
1928     qual find_shortest_path(region r, qual q, qual c, dir d)
1929     {
1930       dd_list queue;
1931       qual path_end;
1932     
1933       q = ecr_qual(q);
1934       assert(q->kind == qk_variable && (!c || constant_qual(c)));
1935       queue = dd_new_list(r);
1936       dd_add_last(r, queue, q);
1937       q->parent = NULL;                /* Mark as root */
1938       path_end = NULL;
1939     
1940       while (!dd_is_empty(queue))
1941         {
1942           qual q;
1943     
1944           q = DD_GET(qual, dd_first(queue));
1945           dd_remove(dd_first(queue));
1946           q = ecr_qual(q);
1947           if (q->mark)
1948     	continue;
1949     
1950           q->mark = TRUE;
1951     
1952           if (constant_qual(q))
1953     	{
1954     	  if (q == c || !c)
1955     	    {
1956     	      path_end = q;
1957     	      break;
1958     	    }
1959     	  else
1960     	    continue;
1961     	}
1962           else
1963     	{
1964     	  qual_edge_set_scanner qess;
1965     	  qual_edge qe;
1966     	  dir dir_to_look;
1967     
1968     	  assert(q->kind == qk_variable);
1969     	  dir_to_look = interesting_scan_dir(q, c);
1970     	  if (d & dir_to_look & dir_backward)
1971     	    {
1972     	      scan_qual_edge_set(qe, qess, q->u.var.lb)
1973     		{
1974     		  qual b;
1975     		  location loc;
1976     
1977     		  b = ecr_qual(qe->qual);
1978     		  loc = qe->loc;
1979     		  if (!b->mark)
1980     		    {
1981     		      /* Dir of edge is opposite of dir of traversal */
1982     		      b->parent = mktraverse_edge(r, q, loc, dir_forward);
1983     		      dd_add_last(r, queue, b);
1984     		    }
1985     		}
1986     	    }

1988     	  if (d & dir_to_look & dir_forward)
1989     	    {
1990     	      scan_qual_edge_set(qe, qess, q->u.var.ub)
1991     		{
1992     		  qual b;
1993     		  location loc;
1994     		      
1995     		  b = ecr_qual(qe->qual);
1996     		  loc = qe->loc;
1997     		  if (!b->mark)
1998     		    {
1999     		      /* Dir of edge is opposite of dir of traversal */
2000     		      b->parent = mktraverse_edge(r, q, loc, dir_backward);
2001     		      dd_add_last(r, queue, b);
2002     		    }
2003     		}
2004     	    }
2005     	}
2006         }

2047     /* Call f(q1, q2, arg) for every edge q1->q2 on a shortest
2048        unidirectional path from qualifier variable q to constant qualifier
2049        c.  If c == NULL, finds shortest path to any constant qualifier.
2050        If there are paths in both directions, traverses both. */
2051     void traverse_shortest_path_edges(qual q, qual c, edge_traverse_fn f,
2052     				  void *arg)
2053     {
2054       region scratch_region;
2055     
2056       scratch_region = newregion();
2057     
2058       if (!c)
2059         {
2060           qual left_end, right_end;
2061     
2062           if ((left_end = find_shortest_path(scratch_region, q, c, dir_backward)))
2063     	traverse_parent_edges_forward(left_end, f, arg);
2064           
2065           if ((right_end = find_shortest_path(scratch_region, q, c,
2066     					  dir_forward)) &&
2067     	  (!left_end || !eq_qual(right_end, left_end)))
2068     	traverse_parent_edges_reverse(right_end, f, arg);
2069         }
2070       else
2071         {
2072           qual end;
2073           dir d;
2074     
2075           d = interesting_scan_dir(q, c);
2076           assert(d != dir_none);
2077     
2078           if (d & (dir_forward | dir_backward))
2079     	{
2080     	  /* Could look in both directions.  Try finding a
2081     	     unidirectional path first, since they're easier to understand */
2082     	  if ((end = find_shortest_path(scratch_region, q, c, dir_backward)))
2083     	    ;
2084     	  else if ((end = find_shortest_path(scratch_region, q, c,
2085     					     dir_forward)))
2086     	    ;
2087     	  else
2088     	    end = find_shortest_path(scratch_region, q, c, d);
2089     
2090     	  if (end)
2091     	    traverse_parent_edges_forward(end, f, arg);
2092     	}
2093           else
2094     	{
2095     	  /* Look only in one direction */
2096     	  if ((end = find_shortest_path(scratch_region, q, c, d)))
2097     	    traverse_parent_edges_forward(end, f, arg);
2098     	}
2099         }
2100     
2101       deleteregion(scratch_region);
2102     }

2383     /* Called when q's error count has changed. */
2384     static void update_error_count(qual q)
2385     {
2386       int i;
2387     
2388       q = ecr_qual(q);
2389       assert(q->kind == qk_variable);
2390       if (q->u.var.num_errors == 0)
2391         return;
2392     
2393       /* See if q is in hotspots list */
2394       for (i = 0; i < num_hotspots && hotspots[i]; i++)
2395         if (eq_qual(hotspots[i], q))
2396           break;
2397     
2398       if (i == num_hotspots)
2399         {
2400           /* List is full, but q is not in the list */
2401           int j;
2402     
2403           /* Find smallest element with greater count than q */
2404           for (j = num_hotspots - 1; j >= 0; j--)
2405     	if (error_count_qual(hotspots[j]) > error_count_qual(q))
2406     	  break;
2407     
2408           if (j != num_hotspots)
2409     	{
2410     	  /* Move q to position j */
2411     	  int k;
2412     
2413     	  for (k = num_hotspots - 1; k > j; k--)
2414     	    hotspots[k] = hotspots[k-1];
2415     	  hotspots[j] = q;
2416     	}
2417         }
2418       else if (hotspots[i] != NULL && i != 0)
2419         {
2420           /* In the list at position i, and it may need to be moved up
2421              the list. */
2422           int j;
2423     
2424           for (j = i; j > 0; j--)
2425     	if (error_count_qual(hotspots[j-1]) < error_count_qual(hotspots[j]))
2426     	  {
2427     	    qual tmp;
2428     	    
2429     	    tmp = hotspots[j-1];
2430     	    hotspots[j-1] = q;
2431     	    hotspots[j] = tmp;
2432     	  }
2433         }
2434       else
2435         /* Empty space on the hotspots list -- just add */
2436         hotspots[i] = q;
2437     }
