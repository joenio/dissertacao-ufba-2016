25     struct type
26     {
27       enum { tk_primitive, tk_complex, tk_tagged, tk_error, tk_void,
28     	 tk_pointer, tk_function, tk_array, tk_var } kind;
29       type_quals qualifiers;
30       user_qual_list user_quals;
31     
32       /* size is not used for aggregate types
33          (as the values may be discovered after the type is created)
34          or for arrays (as they may be arrays of aggregates)
35          alignment is used for aggregates and arrays only if it is non-zero
36          (indicating an alignment overridden by an attribute)
37       */
38       size_t size, alignment; 
39     
40       union {
41         /* tk_primtive and tk_complex.
42            The order reflects promotion order (for an arbitrary machine,
43            see common_primitive_type) */
44         enum { /* The elements of this enum must be ordered as follows:
45     	      - all integral types before tp_first_floating
46     	      - do not change the floating type order
47     	      - the integral types are ordered by "rank" (see c9x std) and
48     	        unsignedness (unsigned > signed) (common_primitive_type
49     		relies on this order)
50     	      - The tp_[u]int<n> types must be before tp_char (for the 
51     	        assert in common_primitive_type). These types are only
52     		used when the corresponding size is not available amongst
53     		short/int/long/long long
54     	        If this frontend followed c9x, these types would always exist
55     	        and be special integer types distinct from the regular ones
56     	        (see the rank stuff), but we're following gcc so they
57     	        aren't. If this changes, common_primitive_type, 
58     		default_conversion and type_default_conversion need revising.
59     	   */
60                tp_error,
61     	   tp_int2, tp_uint2, tp_int4, tp_uint4, tp_int8, tp_uint8,
62                tp_char, 
63     	   tp_signed_char, tp_unsigned_char,
64     	   tp_short, tp_unsigned_short,
65     	   tp_int, tp_unsigned_int,
66     	   tp_long, tp_unsigned_long,
67     	   tp_long_long, tp_unsigned_long_long,
68     
69     	   
70     	   tp_first_floating,
71     	   tp_float = tp_first_floating, tp_double, tp_long_double,
72     	   tp_last
73         } primitive;
74     
75         /* tk_tagged */
76         tag_declaration tag;
77     
78         /* tk_pointer */
79         type pointsto;
80     
81         /* tk_function */
82         struct {
83           type returns;
84           typelist argtypes;
85           type_quals varargs_quals;          /* allow quals on varargs for cqual */
86           user_qual_list varargs_user_quals;
87           bool varargs;
88           bool oldstyle;
89         } fn;
90     
91         /* tk_array */
92         struct {
93           type arrayof;
94           expression size;
95           /* If size is a constant, it is guaranteed >= 0 */
96         } array;
97         cstring var;
98       } u;
99     };

154     /* Return the type t with it's qualifiers set to tq (old qualifiers are 
155        ignored). This is illegal for function types. For arrays, the qualifiers
156        get pushed down to the base type. */
157     type make_qualified_type(type t, type_quals qualifiers,
158     			 user_qual_list user_quals)
159     {
160       /* Push const or volatile down to base type */
161       if (t->kind == tk_array)
162         return make_array_type(make_qualified_type(t->u.array.arrayof, qualifiers,
163     					       user_quals),
164     			   t->u.array.size);
165       else
166         {
167           type nt = copy_type(t);
168     
169           nt->qualifiers = qualifiers;
170           /* nt is a fresh copy, so we can just chain the lists together */
171           nt->user_quals = user_quals;
172           return nt;
173         }
174     }

176     /* Return type 'pointer to t' (unqualified) */
177     type make_pointer_type(type t)
178     {
179       type nt = new_type(tk_pointer);
180       nt->u.pointsto = t;

182       /* ASSUME: all pointers are the same */
183       nt->size = sizeof(void *); 
184       nt->alignment = __alignof__(void *);

186       return nt;

228     /* Make the single instance of pk, with specified size and alignment
229        (Note that we may make copies if this single instance for different alignments)
230        Requires: must be called at most once for each pk, from types_init
231     */
232     static type make_primitive(int pk, int size, int alignment)
233     {
234       type nt = new_type(tk_primitive), ct;
235     
236       nt->u.primitive = pk;
237       nt->size = size;
238       nt->alignment = alignment;
239       primitive_types[pk] = nt;
240     
241       ct = new_type(tk_complex);
242       ct->u.primitive = pk;
243       ct->size = size * 2;
244       ct->alignment = alignment; /* ASSUME: alignof(complex t) == alignof(t) */
245       complex_types[pk] = nt;
246     
247       return nt;
248     }

250     static type lookup_primitive(int default_kind, int size, int alignment,
251     			     bool isunsigned)
252     {
253       int i;
254     
255       for (i = tp_signed_char; i < tp_first_floating; i++)
256         if (primitive_types[i]->size == size && type_unsigned(primitive_types[i]) == isunsigned)
257           return primitive_types[i];
258     
259       return make_primitive(default_kind, size, alignment);
260     }

277     /* Return TRUE if a textually identifier qualifier appears in l */
278     bool member_user_qual_list(user_qual_list l, char *q)
279     {
280       for (; l; l = l->next)
281         if (!strcmp(l->qual->cstring.data, q))
282           return TRUE;
283       return FALSE;
284     }

286     user_qual_list union_user_qual_list(user_qual_list left,
287     				    user_qual_list right)
288     {
289       user_qual_list result = NULL;
290     
291       for (; left; left = left->next)
292         {
293           if (!member_user_qual_list(result, left->qual->cstring.data))
294     	result = new_user_qual_list(left->qual, result);
295         }
296       for (; right; right = right->next)
297         {
298           if (!member_user_qual_list(result, right->qual->cstring.data))
299     	result = new_user_qual_list(right->qual, result);
300         }
301       return result;
302     }

305     /* Return the integral type of size 'size', unsigned if 'isunsigned' is true */
306     type type_for_size(int size, bool isunsigned)
307     {
308       type t = lookup_primitive(tp_error, size, 0, isunsigned);
309       assert(t->u.primitive != tp_error);
310       return t;
311     }

313     type type_for_cval(cval c, bool isunsigned)
314     {
315       int i;
316     
317       for (i = tp_signed_char; i < tp_first_floating; i++)
318         if (type_unsigned(primitive_types[i]) == isunsigned &&
319     	cval_inrange(c, primitive_types[i]))
320           return primitive_types[i];
321     
322       return NULL;
323     }

325     void init_types(void)
326     {
327       types_region = newregion();
328     
329       float_type = make_primitive(tp_float, sizeof(float), __alignof__(float));
330       double_type = make_primitive(tp_double, sizeof(double), __alignof__(double));
331       long_double_type = make_primitive
332         (tp_long_double, sizeof(long double), __alignof__(long double));
333       short_type = make_primitive(tp_short, sizeof(short), __alignof__(short));
334       unsigned_short_type = make_primitive
335         (tp_unsigned_short, sizeof(unsigned short), __alignof__(unsigned short));
336       int_type = make_primitive(tp_int, sizeof(int), __alignof__(int));
337       unsigned_int_type = make_primitive
338         (tp_unsigned_int, sizeof(unsigned int), __alignof__(unsigned int));
339       long_type = make_primitive(tp_long, sizeof(long), __alignof__(long));
340       unsigned_long_type = make_primitive
341         (tp_unsigned_long, sizeof(unsigned long), __alignof__(unsigned long));
342     
343       long_long_type = make_primitive
344         (tp_long_long, sizeof(long long), __alignof__(long long));
345       unsigned_long_long_type = make_primitive
346         (tp_unsigned_long_long, sizeof(unsigned long long), __alignof__(unsigned long long));
347       signed_char_type = make_primitive
348         (tp_signed_char, sizeof(signed char), __alignof__(signed char));
349       unsigned_char_type = make_primitive
350         (tp_unsigned_char, sizeof(unsigned char), __alignof__(unsigned char));
351       char_type = make_primitive(tp_char, sizeof(char), __alignof__(char));
352     
353      {
354         /* GCC is broken (could be a broken design issue ;-)), so need the
355            typedefs (rather than using the types directly in the calls
356            to alignof) */
357         typedef int __attribute__ ((mode(__HI__))) myint2;
358         typedef int __attribute__ ((mode(__SI__))) myint4;
359         typedef int __attribute__ ((mode(__DI__))) myint8;
360         int2_type = lookup_primitive(tp_int2, 2, __alignof__(myint2), FALSE);
361         uint2_type = lookup_primitive(tp_uint2, 2, __alignof__(myint2), TRUE);
362         int4_type = lookup_primitive(tp_int4, 4, __alignof__(myint4), FALSE);
363         uint4_type = lookup_primitive(tp_uint4, 4, __alignof__(myint4), TRUE);
364         int8_type = lookup_primitive(tp_int8, 8, __alignof__(myint8), FALSE);
365         uint8_type = lookup_primitive(tp_uint8, 8, __alignof__(myint8), TRUE);
366       }
367     
368       char_array_type = make_array_type(char_type, NULL);
369       error_type = new_type(tk_error);
370       error_type->size = error_type->alignment = 1;
371       void_type = new_type(tk_void);
372       void_type->size = void_type->alignment = 1;
373       ptr_void_type = make_pointer_type(void_type);
374     
375       /* SAME: wchar_t, size_t, ptrdiff_t, pointer size */
376       wchar_type = type_for_size(sizeof(wchar_t), (wchar_t)-1 > 0);
377       wchar_array_type = make_array_type(wchar_type, NULL);
378       size_t_type = type_for_size(sizeof(size_t), TRUE);
379       ptrdiff_t_type = type_for_size(sizeof(ptrdiff_t), FALSE);
380       intptr_type = type_for_size(sizeof(void *), TRUE);
381     }

383     struct typelist
384     {
385       region sameregion r;
386       struct typelist_element *sameregion first;
387     };
388     
389     struct typelist_element
390     {
391       struct typelist_element *sameregion next;
392       type t;
393     };
394     
395     typelist new_typelist(region r)
396     {
397       typelist tl = ralloc(r, struct typelist);
398       tl->r = r;
399       tl->first = NULL;
400       return tl;
401     }
402     
403     void typelist_append(typelist tl, type t)
404     {
405       struct typelist_element *nte = ralloc(tl->r, struct typelist_element);
406       struct typelist_element *sameregion *last;
407     
408       nte->t = t;
409       last = &tl->first;
410       while (*last)
411         last = &(*last)->next;
412       *last = nte;
413     }
414     
415     bool empty_typelist(typelist tl)
416     {
417       return tl->first == NULL;
418     }
419     
420     void typelist_scan(typelist tl, typelist_scanner *scanner)
421     {
422       *scanner = tl->first;
423     }
424     
425     type typelist_next(typelist_scanner *scanner)
426     {
427       type t;
428     
429       if (!*scanner)
430         return NULL;
431       t = (*scanner)->t;
432       *scanner = (*scanner)->next;
433       return t;
434     }

720     /* Return TRUE if two type lists are equal */
721     static bool type_lists_equal(typelist al1, typelist al2)
722     {
723       struct typelist_element *args1 = al1->first, *args2 = al2->first;
724     
725       for (;;)
726         {
727           if (args1 == 0 && args2 == 0)
728     	return TRUE;
729           /* If one list is shorter than the other,
730     	 they fail to match.  */
731           if (args1 == 0 || args2 == 0)
732     	return FALSE;
733     
734           if (!type_equal(args1->t, args2->t))
735     	return FALSE;
736     
737           args1 = args1->next;
738           args2 = args2->next;
739         }
740     }

742     /* Return TRUE if two function types F1 and F2 are equal. */
743     bool function_equal(type t1, type t2)
744     {
745       /* Return types must match */
746       if (!type_equal(t1->u.fn.returns, t2->u.fn.returns))
747         return FALSE;
748     
749       if (!t1->u.fn.oldstyle && !t2->u.fn.oldstyle)
750         return t1->u.fn.varargs == t2->u.fn.varargs &&
751           t1->u.fn.varargs_quals == t2->u.fn.varargs_quals &&
752           type_lists_equal(t1->u.fn.argtypes, t2->u.fn.argtypes);
753       else
754         return t1->u.fn.oldstyle == t2->u.fn.oldstyle;
755     }

757     bool type_equal(type t1, type t2)
758     {
759       return t1->qualifiers == t2->qualifiers && type_equal_unqualified(t1, t2);
760     }

762     bool type_equal_unqualified(type t1, type t2)
763     {
764       if (t1 == error_type || t2 == error_type)
765         return TRUE;
766     
767       /* Short-circuit easy case */
768       if (t1 == t2)
769         return TRUE;
770     
771       /* Different classes of types can't be compatible.  */
772       if (t1->kind != t2->kind)
773         return FALSE;
774     
775       switch (t1->kind)
776         {
777         case tk_primitive: case tk_complex:
778           return t1->u.primitive == t2->u.primitive;
779     
780         case tk_void:
781           return TRUE;
782     
783         case tk_tagged:
784           return t1->u.tag == t2->u.tag;
785     
786         case tk_pointer:
787           return type_equal(t1->u.pointsto, t2->u.pointsto);
788     
789         case tk_function:
790           return function_equal(t1, t2);
791     
792         case tk_array: {
793           known_cst s1 = t1->u.array.size ? t1->u.array.size->cst : NULL;
794           known_cst s2 = t2->u.array.size ? t2->u.array.size->cst : NULL;
795     
796           return type_equal(t1->u.array.arrayof, t2->u.array.arrayof) &&
797     	(!s1 || !s2 || cval_intcompare(s1->cval, s2->cval) == 0);
798         }
799         default: assert(0); return FALSE;
800         }
801     }

803     /* Return TRUE if T is not affected by default promotions.  */
804     bool type_self_promoting(type t)
805     {
806       if (t->kind != tk_primitive)
807         return TRUE;
808     
809       /* Could also use array. Hmm. */
810       switch (t->u.primitive)
811         {
812         case tp_float: case tp_char: case tp_unsigned_char: case tp_signed_char:
813         case tp_short: case tp_unsigned_short:
814           return FALSE;
815         default:
816           return TRUE;
817         }
818     }

820     /* Return TRUE if function type FNTYPE specifies a fixed number of parameters
821        and none of their types is affected by default promotions.
822        TRUE for oldstyle functions */
823     bool self_promoting_args(type fntype)
824     {
825       struct typelist_element *parms;
826     
827       if (type_function_varargs(fntype))
828         return FALSE;
829     
830       if (type_function_oldstyle(fntype))
831         return TRUE;
832     
833       for (parms = type_function_arguments(fntype)->first; parms;
834            parms = parms->next)
835         if (!type_self_promoting(parms->t))
836           return FALSE;
837     
838       return TRUE;
839     }

841     /* Allow  wait (union {union wait *u; int *i})
842        and  wait (union wait *)  to be compatible.  */
843     static bool weird_parameter_match(type t1, type t2)
844     {
845       tag_declaration t1decl;
846     
847       if (type_union(t1)
848           && (!(t1decl = type_tag(t1))->name || t1decl->transparent_union)
849           && type_size_cc(t1) && type_size_cc(t2)
850           && type_size(t1) == type_size(t2))
851         {
852           field_declaration field;
853     
854           for (field = t1decl->fieldlist; field; field = field->next)
855     	if (type_compatible(field->type, t2))
856     	  return TRUE;
857         }
858       return FALSE;
859     }

861     static type weird_common_parameter(type t1, type t2)
862     {
863       /* Given  wait (union {union wait *u; int *i} *)
864          and  wait (union wait *),
865          prefer  union wait *  as type of parm.  */
866       if (weird_parameter_match(t1, t2))
867         {
868           if (pedantic)
869     	pedwarn("function types not truly compatible in ANSI C");
870           return t2;
871         }
872       return NULL;
873     }

875     /* Check two lists of types for compatibility,
876        returning 0 for incompatible, 1 for compatible,
877        or 2 for compatible with warning.  */
878     static int type_lists_compatible(typelist al1, typelist al2)
879     {
880       /* 1 if no need for warning yet, 2 if warning cause has been seen.  */
881       int val = 1;
882       int newval = 0;
883       struct typelist_element *args1 = al1->first, *args2 = al2->first;
884     
885       while (1)
886         {
887           if (args1 == 0 && args2 == 0)
888     	return val;
889           /* If one list is shorter than the other,
890     	 they fail to match.  */
891           if (args1 == 0 || args2 == 0)
892     	return 0;
893     
894           if (!(newval = type_compatible_unqualified(args1->t, args2->t)))
895     	{
896     	  if (!weird_parameter_match(args1->t, args2->t) &&
897     	      !weird_parameter_match(args2->t, args1->t))
898     	    return 0;
899     	}
900     
901           /* type_compatible said ok, but record if it said to warn.  */
902           if (newval > val)
903     	val = newval;
904     
905           args1 = args1->next;
906           args2 = args2->next;
907         }
908     }

910     /* Return 1 if two function types F1 and F2 are compatible.
911        If either type specifies no argument types,
912        the other must specify a fixed number of self-promoting arg types.
913        Otherwise, if one type specifies only the number of arguments, 
914        the other must specify that number of self-promoting arg types.
915        Otherwise, the argument types must match.  */
916     int function_compatible(type t1, type t2)
917     {
918       typelist args1, args2;
919     
920       /* 1 if no need for warning yet, 2 if warning cause has been seen.  */
921       int val = 1;
922     
923       /* Return types must match */
924       if (!type_compatible(t1->u.fn.returns, t2->u.fn.returns))
925         return 0;
926     
927       args1 = t1->u.fn.argtypes;
928       args2 = t2->u.fn.argtypes;
929     
930       /* An unspecified parmlist matches any specified parmlist
931          whose argument types don't need default promotions.  */
932     
933       if (t1->u.fn.oldstyle)
934         {
935           if (args2 && !self_promoting_args(t2))
936     	return 0;
937     #if 0
938           /* If one of these types comes from a non-prototype fn definition,
939     	 compare that with the other type's arglist.
940     	 If they don't match, ask for a warning (but no error).  */
941           if (TYPE_ACTUAL_ARG_TYPES (f1)
942     	  && 1 != type_lists_compatible(args2, TYPE_ACTUAL_ARG_TYPES (f1)))
943     	val = 2;
944     #endif
945           return val;
946         }
947       if (t2->u.fn.oldstyle)
948         {
949           if (args1 && !self_promoting_args(t1))
950     	return 0;
951     #if 0
952           if (TYPE_ACTUAL_ARG_TYPES (f2)
953     	  && 1 != type_lists_compatible(args1, TYPE_ACTUAL_ARG_TYPES (f2)))
954     	val = 2;
955     #endif
956           return val;
957         }
958     
959       if (t1->u.fn.varargs != t2->u.fn.varargs)
960         return 0;
961     
962       /* Both types have argument lists: compare them and propagate results.  */
963       return type_lists_compatible(args1, args2);
964     }

966     bool type_compatible_unqualified(type t1, type t2)
967     {
968       if (t1 == error_type || t2 == error_type)
969         return 1;
970     
971       /* Short-circuit easy case */
972       if (t1 == t2)
973         return 1;
974     
975       /* Different classes of types can't be compatible.  */
976       if (t1->kind != t2->kind)
977         return 0;
978     
979       switch (t1->kind)
980         {
981         case tk_primitive: case tk_complex:
982           return t1->u.primitive == t2->u.primitive;
983     
984         case tk_void:
985           return 1;
986     
987         case tk_tagged:
988           return t1->u.tag == t2->u.tag;
989     
990         case tk_pointer:
991           return type_compatible(t1->u.pointsto, t2->u.pointsto);
992     
993         case tk_function:
994           return function_compatible(t1, t2);
995     
996         case tk_array: {
997           known_cst s1 = t1->u.array.size ? t1->u.array.size->cst : NULL;
998           known_cst s2 = t2->u.array.size ? t2->u.array.size->cst : NULL;
999     
1000           return type_compatible(t1->u.array.arrayof, t2->u.array.arrayof) &&
1001     	(!s1 || !s2 || cval_intcompare(s1->cval, s2->cval) == 0);
1002         }
1003         default: assert(0); return 0;
1004         }
1005     }

1007     bool type_compatible(type t1, type t2)
1008     {
1009       /* Qualifiers must match.  */
1010       /* GCC appears to allow changes to restrict (see /usr/include/sys/stat.h
1011          and the decls/defs of stat/lstat in redhat linux 7) */
1012       /* XXX: investigate more. */
1013       if ((t1->qualifiers & ~restrict_qualifier) != (t2->qualifiers & ~restrict_qualifier))
1014         return 0;
1015       else
1016         return type_compatible_unqualified(t1, t2);
1017     }

1041     static int common_primitive_type(type t1, type t2)
1042     {
1043       int pk1 = t1->u.primitive, pk2 = t2->u.primitive;
1044       int pk = pk1 < pk2 ? pk2 : pk1;
1045     
1046       /* Note: the results of this function depend on the relative sizes of 
1047          char, short, int, long and long long. The results are only correct
1048          for a version of C that has types that match those set by types_init
1049          (which uses those from the C compiler that compiles this file)
1050       */
1051       if (pk1 < tp_first_floating && pk2 < tp_first_floating)
1052         {
1053           /* The simple cases */
1054           if (t1->size < t2->size)
1055     	return pk2;
1056           else if (t1->size > t2->size)
1057     	return pk1;
1058     
1059           /* The pain starts, see 6.3.1.8 of c9x */
1060           /* If the sizes are the same, then we can't have a tp_[u]int<n> or a 
1061     	 tp_char/short/int/long/etc pair (as we only have tp_[u]int<n> if there
1062     	 is no corresponding integer type of the same size. So we can compare rank
1063     	 by comparing pk1 and pk2 */
1064           assert(!((pk1 < tp_char && pk2 >= tp_char) || (pk1 >= tp_char && pk2 < tp_char)));
1065     
1066           /* the higher rank wins, and if either of the types is unsigned, the
1067     	 result is (thus unsigned short + int == unsigned int if
1068     	 sizeof(short) == sizeof(int) */
1069           if ((type_unsigned(t1) || type_unsigned(t2)) && !type_unsigned(primitive_types[pk]))
1070     	/* A bit inefficient, admittedly */
1071     	pk = make_unsigned_type(primitive_types[pk])->u.primitive;
1072     
1073           return pk;
1074         }
1075     
1076       /* Floating point types follow the order specified in the enum and win
1077          over all integral types */
1078       return pk;
1079     }
1080     
1081     /* Return the common type of two types.
1082     
1083        This is used in two cases:
1084         - when type_compatible(t1, t2) is true, to pick a merged type for
1085         declarations
1086         - to merge two types that are compatible in some expressions (e.g. ?:,
1087         arithmetic) */
1088     type common_type(type t1, type t2)
1089     {
1090       type rtype;
1091     
1092       /* Save time if the two types are the same.  */
1093       if (t1 == t2)
1094         return t1;
1095     
1096       /* If one type is nonsense, use the other.  */
1097       if (t1 == error_type)
1098         return t2;
1099       if (t2 == error_type)
1100         return t1;
1101     
1102       /* Special enum handling: if same enum, just merge qualifiers.  otherwise
1103          treat an enum type as the unsigned integer type of the same width.
1104          Note that gcc does not check for the same enum. Some weird behaviour...
1105          (see enum2.c)
1106       */
1107       if (type_enum(t1))
1108         {
1109           if (type_equal_unqualified(t1, t2))
1110     	return qualify_type2(t1, t1, t2);
1111           t1 = qualify_type1(type_for_size(type_size(t1), TRUE), t1);
1112         }
1113       if (type_enum(t2))
1114         t2 = qualify_type1(type_for_size(type_size(t1), TRUE), t2);
1115     
1116       /* If one type is complex, form the common type of the non-complex
1117          components, then make that complex. */
1118       if (type_complex(t1) || type_complex(t2))
1119         {
1120           int pk = common_primitive_type(t1, t2);
1121           assert((t1->kind == tk_primitive || t1->kind == tk_complex) && 
1122     	     (t2->kind == tk_primitive || t2->kind == tk_complex));
1123           rtype = complex_types[pk];
1124         }
1125       else
1126         switch (t1->kind)
1127           {
1128           case tk_primitive:
1129     	{
1130     	  int pk = common_primitive_type(t1, t2);
1131     	  assert(t2->kind == tk_primitive);
1132     	  rtype = primitive_types[pk];
1133     	  break;
1134     	}
1135     
1136           case tk_void: case tk_tagged:
1137     	rtype = t1;
1138     	break;
1139     
1140           case tk_pointer:
1141     	rtype = make_pointer_type(common_type(t1->u.pointsto, t2->u.pointsto));
1142     	break;
1143     
1144           case tk_array:
1145     	{
1146     	  /* Merge the element types, and have a size if either arg has one.  */
1147     	  type element_type =
1148     	    common_type(t1->u.array.arrayof, t2->u.array.arrayof);
1149     	  expression size1 = t1->u.array.size, size2 = t2->u.array.size;
1150     
1151     	  rtype = make_array_type(element_type, size1 ? size1 : size2);
1152     	  break;
1153     	}
1154     
1155           case tk_function:
1156     	/* Function types: prefer the one that specified arg types.
1157     	   If both do, merge the arg types.  Also merge the return types.  */
1158     	{
1159     	  type valtype = common_type(t1->u.fn.returns, t2->u.fn.returns);
1160     	  typelist args;
1161     	  bool oldstyle, varargs;
1162     	  type_quals varargs_quals;
1163     	  user_qual_list varargs_user_quals;
1164     
1165     	  if (t1->u.fn.oldstyle && t2->u.fn.oldstyle)
1166     	    {
1167     	      args = NULL;
1168     	      oldstyle = TRUE;
1169     	      varargs = FALSE;
1170     	      varargs_quals = no_qualifiers;
1171     	      varargs_user_quals = NULL;
1172     	    }
1173     	  else if (t1->u.fn.oldstyle)
1174     	    {
1175     	      args = t2->u.fn.argtypes;
1176     	      oldstyle = FALSE;
1177     	      varargs = t2->u.fn.varargs;
1178     	      varargs_quals = t2->u.fn.varargs_quals;
1179     	      varargs_user_quals = t2->u.fn.varargs_user_quals;
1180     	    }
1181     	  else if (t2->u.fn.oldstyle)
1182     	    {
1183     	      args = t1->u.fn.argtypes;
1184     	      oldstyle = FALSE;
1185     	      varargs = t1->u.fn.varargs;
1186     	      varargs_quals = t1->u.fn.varargs_quals;
1187     	      varargs_user_quals = t1->u.fn.varargs_user_quals;
1188     	    }
1189     	  else
1190     	    {
1191     	      /* If both args specify argument types, we must merge the two
1192     		 lists, argument by argument.  */
1193     	      struct typelist_element *args1 = t1->u.fn.argtypes->first;
1194     	      struct typelist_element *args2 = t2->u.fn.argtypes->first;
1195     	      type argtype;
1196     
1197     	      oldstyle = FALSE;
1198     	      varargs = t1->u.fn.varargs;
1199     	      args = new_typelist(t1->u.fn.argtypes->r);
1200     	      varargs_quals = t1->u.fn.varargs_quals | t2->u.fn.varargs_quals;
1201     	      if (t2->u.fn.varargs_user_quals)
1202     		varargs_user_quals = t2->u.fn.varargs_user_quals;
1203     	      else
1204     		varargs_user_quals = NULL;
1205     	      if (t1->u.fn.varargs_user_quals)
1206     		varargs_user_quals =
1207     		  union_user_qual_list(t1->u.fn.varargs_user_quals,
1208     				       varargs_user_quals);
1209     
1210     	      while (args1)
1211     		{
1212     		  argtype = weird_common_parameter(args1->t, args2->t);
1213     		  if (!argtype)
1214     		    argtype = weird_common_parameter(args2->t, args1->t);
1215     		  if (!argtype)
1216     		    argtype = common_type(args1->t, args2->t);
1217     		  typelist_append(args, argtype);
1218     
1219     		  args1 = args1->next;
1220     		  args2 = args2->next;
1221     		}
1222     
1223     	    }
1224     	  rtype = make_function_type(valtype, args, varargs, varargs_quals,
1225     				     varargs_user_quals, oldstyle);
1226     	  break;
1227     	}
1228     
1229           default:
1230     	assert(0); return NULL;
1231           }
1232     
1233       return qualify_type2(rtype, t1, t2);
1234     }

1301     }

1303     static type_element uql2ast(region r, location loc, user_qual_list uql,
1304     			    type_element rest)
1305     {
1306       if (uql)
1307         {

1309     			      new_user_qual(r, loc, uql->qual->cstring));
1310           ast->next = CAST(node, uql2ast(r, loc, uql->next, rest));
1311           return ast;
1312         }
1313       else
1314         return rest;
1315     }

1317     static type_element qualifiers2ast(region r, location loc, type_quals quals,
1318     				   user_qual_list uql, type_element rest)
1319     {
1320       if (quals & volatile_qualifier)
1321         rest = qualifier2ast(r, loc, volatile_qualifier, rest);
1322       if (quals & const_qualifier)
1323         rest = qualifier2ast(r, loc, const_qualifier, rest);
1324       rest = uql2ast(r, loc, uql, rest);
1325       return rest;
1326     }

1328     static type_element primitive2ast(region r, location loc, int primitive,
1329     				  type_element rest)
1330     {
1331       bool isunsigned = FALSE;
1332       int keyword;
1333     
1334       switch (primitive)
1335         {
1336         case tp_unsigned_char:
1337           isunsigned = TRUE;
1338         case tp_char:
1339           keyword = RID_CHAR;
1340           break;
1341         case tp_signed_char:
1342           return rid2ast(r, loc, RID_SIGNED, rid2ast(r, loc, RID_CHAR, rest));
1343         case tp_unsigned_short:
1344           isunsigned = TRUE;
1345         case tp_short:
1346           keyword = RID_SHORT;
1347           break;
1348         case tp_unsigned_int:
1349           isunsigned = TRUE;
1350         case tp_int:
1351           keyword = RID_INT;
1352           break;
1353         case tp_unsigned_long:
1354           isunsigned = TRUE;
1355         case tp_long:
1356           keyword = RID_LONG;
1357           break;
1358         case tp_unsigned_long_long:
1359           isunsigned = TRUE;
1360         case tp_long_long:
1361           keyword = RID_LONG;
1362           rest = rid2ast(r, loc, RID_LONG, rest);
1363           break;
1364         case tp_float:
1365           keyword = RID_FLOAT;
1366           break;
1367         case tp_double:
1368           keyword = RID_DOUBLE;
1369           break;
1370         case tp_long_double:
1371           keyword = RID_DOUBLE;
1372           rest = rid2ast(r, loc, RID_LONG, rest);
1373           break;
1374         default:
1375           assert(0);
1376           keyword = RID_INT; break;
1377         }
1378     
1379       rest = rid2ast(r, loc, keyword, rest);
1380       if (isunsigned)
1381         rest = rid2ast(r, loc, RID_UNSIGNED, rest);
1382     
1383       return rest;
1384     }

1386     #define UNNAMED_STRUCT_PREFIX "__unnamed"
1387     
1388     void name_tag(tag_declaration tag)
1389     {
1390       /* Name unnamed structs, unions or enums */
1391       if (!tag->name)
1392         {
1393           static long nextid = 4242;
1394           char tagname[sizeof(UNNAMED_STRUCT_PREFIX) + 20];
1395     
1396           sprintf(tagname, UNNAMED_STRUCT_PREFIX "%ld", nextid++);
1397           tag->ast->word1 = new_word(parse_region, dummy_location,
1398     				 str2cstring(parse_region, tagname));
1399           tag->name = tag->ast->word1->cstring.data;
1400         }
1401     }

1403     static type_element tag2ast(region r, location loc, tag_declaration tag,
1404     			    type_element rest)
1405     {
1406       tag_ref tr;
1407     
1408       name_tag(tag);
1409       tr = newkind_tag_ref(r, tag->kind, loc, 
1410     		       new_word(r, loc, tag->ast->word1->cstring),
1411     		       NULL, NULL, FALSE);
1412     
1413       tr->tdecl = tag;
1414       tr->next = CAST(node, rest);
1415     
1416       /* creating a type naming a tag that shadows another is tricky as
1417          the placement of the type affects its meaning. 
1418     
1419          The current use of type2ast is for declaring temporaries. These are
1420          placed at the start of the closest enclosing block. This is correct as
1421          long as the temporary does not rely on a tag declared in the same
1422          block that shadows a tag in an enclosing scope (as the temporary would
1423          then erroneously refer to the enclosing tag). The simplest check which
1424          detects this situation is any use of a shadowed tag (which I am
1425          assuming are very rare anyway).
1426     
1427          The correct solution is that if the type of temporary x refers to tags
1428          t1...tn declared in the current block, then x should be placed just
1429          before the declaration of the latest of ti, the tag in t1...tn which
1430          was declared latest, and the declaration of x should be preceded by
1431          'struct/union/enum ti;'. This is somewhat painful.
1432     
1433          To avoid any problems, compile with error_shadow = warn_shadow = 1
1434          (this is what RC does) */
1435     
1436       assert(!tag->shadowed);
1437     
1438       return CAST(type_element, tr);
1439     }

1441     static declaration parameter2ast(region r, location loc, type t)
1442     {
1443       variable_decl vd;
1444       data_decl dd;
1445       declarator tdeclarator;
1446       type_element tmodifiers;
1447     
1448       /* Build AST for the declaration */
1449       type2ast(r, loc, t, NULL, &tdeclarator, &tmodifiers);
1450       vd = new_variable_decl(r, loc, tdeclarator, NULL, NULL, NULL, NULL);
1451       vd->declared_type = t;
1452       dd = new_data_decl(r, loc, tmodifiers, NULL, CAST(declaration, vd));
1453     
1454       return CAST(declaration, dd);
1455     }

1457     /* Build AST nodes such that "MODIFIERS D" represents the declaration of
1458        "T INSIDE", at location loc, allocating in region r */
1459     void type2ast(region r, location loc, type t, declarator inside,
1460     	      declarator *d, type_element *modifiers)
1461     {
1462       /* XXX: De-recursify */
1463       type_element qualifiers = qualifiers2ast(r, loc, t->qualifiers,
1464     					   t->user_quals, NULL);
1465     
1466       switch (t->kind)
1467         {
1468         case tk_primitive:
1469           *modifiers = primitive2ast(r, loc, t->u.primitive, qualifiers);
1470           *d = inside;
1471           break;
1472         case tk_complex:
1473           *modifiers =
1474     	rid2ast(r, loc, RID_COMPLEX, 
1475     	primitive2ast(r, loc, t->u.primitive, qualifiers));
1476           *d = inside;
1477           break;
1478         case tk_tagged:
1479           *modifiers = tag2ast(r, loc, t->u.tag, qualifiers);
1480           *d = inside;
1481           break;
1482         case tk_void:
1483           *modifiers = rid2ast(r, loc, RID_VOID, qualifiers);
1484           *d = inside;
1485           break;
1486         case tk_pointer:
1487           inside = CAST(declarator,
1488     		    new_pointer_declarator(r, loc, inside, qualifiers));
1489           type2ast(r, loc, t->u.pointsto, inside, d, modifiers);
1490           break;
1491         case tk_array:
1492           assert(qualifiers == NULL);
1493           inside = CAST(declarator,
1494     		    new_array_declarator(r, loc, inside, t->u.array.size));
1495           type2ast(r, loc, t->u.array.arrayof, inside, d, modifiers);
1496           break;
1497         case tk_function: {
1498           declaration parms;
1499     
1500           /* XXX: doesn't rebuild fn qualifiers. Are we generating C here
1501     	 or not ? */
1502           /* XXX: Should build environment for parameters */
1503           if (t->u.fn.oldstyle)
1504     	parms = NULL;
1505           else if (empty_typelist(t->u.fn.argtypes))
1506     	parms = parameter2ast(r, loc, void_type);
1507           else
1508     	{
1509     	  struct typelist_element *args = t->u.fn.argtypes->first;
1510     	  declaration *lastparm = &parms;
1511     
1512     	  while (args)
1513     	    {
1514     	      *lastparm = parameter2ast(r, loc, args->t);
1515     	      lastparm = (declaration *)&(*lastparm)->next;
1516     	      args = args->next;
1517     	    }
1518     	  if (t->u.fn.varargs)
1519     	    {
1520     	      type_element vquals;
1521     
1522     	      vquals = qualifiers2ast(r, loc, t->u.fn.varargs_quals,
1523     				      t->u.fn.varargs_user_quals, NULL);
1524     	      *lastparm = CAST(declaration,
1525     			       new_ellipsis_decl(r, loc, vquals));
1526     	      lastparm = (declaration *)&(*lastparm)->next;
1527     	    }
1528     	  *lastparm = NULL;
1529     	}
1530           inside = CAST(declarator,
1531     		    new_function_declarator(r, loc, inside, parms, NULL, NULL));
1532           type2ast(r, loc, t->u.fn.returns, inside, d, modifiers);
1533           break;
1534         }
1535         default: assert(0); break;
1536         }
1537     }

1539     bool type_contains_pointers(type t)
1540     {
1541       field_declaration field;
1542     
1543       if (type_pointer(t))
1544         return TRUE;
1545     
1546       if (type_array(t))
1547         return type_contains_pointers(type_array_of(t));
1548     
1549       if (!type_aggregate(t))
1550         return FALSE;
1551     
1552       for (field = type_tag(t)->fieldlist; field; field = field->next)
1553         if (type_contains_pointers(field->type))
1554           return TRUE;

1580     bool type_contains_quals(type t)
1581     {
1582       if (!empty_user_qual_list(type_user_quals(t)) ||
1583           type_const(t) || type_volatile(t))
1584         return TRUE;
1585     
1586       if (type_pointer(t))
1587         return type_contains_quals(type_points_to(t));
1588     
1589       if (type_array(t))
1590         return type_contains_quals(type_array_of(t));
1591     
1592       if (type_function(t))
1593         {
1594           typelist_scanner ts;
1595           type arg;
1596     
1597           if (type_contains_quals(type_function_return_type(t)))
1598     	return TRUE;

1600     	{

1603     	    if (type_contains_quals(arg))
1604     	      return TRUE;
1605     	}
1606         }
1607     
1608       if (type_aggregate(t))
1609         /* XXX Fill in */
1610         ;

1612       return FALSE;

1649     type type_default_conversion(type from)
1650     {
1651       if (type_enum(from))
1652         from = type_tag(from)->reptype;
1653     
1654       if (type_smallerthanint(from))
1655         {
1656           /* Traditionally, unsignedness is preserved in default promotions. */
1657           if (flag_traditional && type_unsigned(from))
1658     	return unsigned_int_type;
1659           else
1660     	return int_type;
1661         }
1662       /* Note: if we had a type of same size as int, but of lesser rank, we should be
1663          returning one of int/unsigned int here, but we don't support that kind of
1664          type */
1665     
1666       if (flag_traditional && !flag_allow_single_precision && type_float(from))
1667         return double_type;
1668     
1669       if (type_function(from))
1670         return make_pointer_type(from);
1671     
1672       if (type_array(from))
1673         return make_pointer_type(type_array_of(from));
1674     
1675       return from;
1676     }
