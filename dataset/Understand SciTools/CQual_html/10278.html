26     /**************************************************************************
27      *                                                                        *
28      * A set represented as a bunch of bits                                   *
29      *                                                                        *
30      * A bit at position i is true if the ith element is in the set, false    *
31      * otherwise.                                                             *
32      *                                                                        *
33      **************************************************************************/
34     
35     #define BITS_PER_BYTE 8
36     #define BITS_PER_UINT (sizeof(unsigned int)*BITS_PER_BYTE)
37     #define UINT_MAX 0xffffffff
38     
39     /* Compute a/b, rounded up to the nearest integer.  Only works when a
40        and b are positive integers. */
41     static inline unsigned int div_ceil(unsigned int a, unsigned int b)
42     {
43       return ((a/b) + (a%b ? 1 : 0));
44     }
45     
46     /* Compute which unsigned int elt would be in in a bitset of nbits
47        bits.  Includes skip over length byte. */
48     static inline unsigned int compute_offset(bitset b, unsigned int elt)
49     {
50       unsigned int i;
51     
52       assert(elt < b[0]);
53       for (i = 0; elt > BITS_PER_UINT; i++, elt = elt - BITS_PER_UINT);
54       return i + 1;
55     }
56     
57     /* Construct a bitmask for element elt */
58     static inline unsigned int compute_mask(bitset b, unsigned int elt)
59     {
60       unsigned int a, d, c;
61     
62       assert(elt < b[0]);
63       a = BITS_PER_UINT;
64       d = elt % BITS_PER_UINT;
65       c = 1 << d;
66       return 1 << (elt % BITS_PER_UINT);
67     }
68     
69     /* Compute the number of elements of b.bits */
70     static inline unsigned int bitset_length(bitset b)
71     {
72       return div_ceil(b[0], BITS_PER_UINT);
73     }
74     
75     /* Return the number of unsigned int's to allocate for a bitset of
76        size nbits. */
77     unsigned int sizeof_bitset(unsigned int nbits)
78     {
79       return div_ceil(nbits, BITS_PER_UINT) + 1;
80     }
81     
82     /* Allocate a bit set of size nbits in region r. */
83     bitset bitset_new(region r, unsigned int nbits)
84     {
85       unsigned int num_uints;
86       bitset new_set;
87     
88       num_uints = sizeof_bitset(nbits);
89       assert(nbits >= 0);
90       new_set = rarrayalloc(r, num_uints, unsigned int);
91       new_set[0] = nbits;
92       return new_set;
93     }
94     
95     /* Make a copy of b */
96     bitset bitset_copy(region r, bitset b)
97     {
98       unsigned int num_uints;
99       bitset new_set;
100     
101       num_uints = bitset_length(b);
102       new_set = rarrayalloc(r, num_uints + 1, unsigned int);
103       rarraycopy(new_set, b, num_uints + 1, unsigned int);
104       return new_set;
105     }
106     
107     /* Replace bitset b1 by b2.  b1 and b2 must be the same size. */
108     void bitset_assign(bitset b1, bitset b2)
109     {
110       unsigned int num_uints;
111       int i;
112     
113       assert(bitset_length(b1) == bitset_length(b2));
114       num_uints = bitset_length(b1);
115       for (i = 0; i < num_uints; i++)
116         b1[i+1] = b2[i+1];
117     }
118     
119     /* Return TRUE iff bitset b is empty */
120     bool bitset_empty(bitset b)
121     {
122       int num_uints;
123       int i;
124     
125       num_uints = bitset_length(b);
126       for (i = 0; i < num_uints; i++)
127         if (b[i+1])
128           return FALSE;
129       return TRUE;
130     }
131     
132     /* Return TRUE iff elements of b from first to last, inclusive, are
133        all empty */
134     bool bitset_empty_range(bitset b, unsigned int first, unsigned int last)
135     {
136       int i;
137     
138       for (i = first; i <= last; i++)
139         if (bitset_member(b, i))
140           return FALSE;
141       return TRUE;
142     }
143     
144     /* Return TRUE iff elements of b from first to last, inclusive, are
145        all present */
146     bool bitset_full_range(bitset b, unsigned int first, unsigned int last)
147     {
148       int i;
149     
150       for (i = first; i <= last; i++)
151         if (!bitset_member(b, i))
152           return FALSE;
153       return TRUE;
154     }
155     
156     /* Add all possible elements to b. */
157     void bitset_insert_all(bitset b)
158     {
159       int num_ints;
160       int i;
161     
162       num_ints = bitset_length(b);
163       for (i = 0; i < num_ints; i++)
164         b[i+1] = UINT_MAX;
165     }
166     
167     /* Add elt to bitset b of nbits bits.  Returns TRUE if elt was not
168        already in the set. */
169     bool bitset_insert(bitset b, unsigned int elt)
170     {
171       int offset = compute_offset(b, elt);
172       int mask = compute_mask(b, elt);
173     
174       if (b[offset] & mask)
175         return FALSE;
176       else
177         {
178           b[offset] = b[offset] | mask;
179           return TRUE;
180         }
181     }
182     
183     /* Remove elt from bitset b.  Returns TRUE if elt was in the set. */
184     bool bitset_remove(bitset b, unsigned int elt)
185     {
186       int offset = compute_offset(b, elt);
187       int mask = compute_mask(b, elt);
188     
189       if (b[offset] & mask)
190         {
191           b[offset] = b[offset] & (~mask);
192           return TRUE;
193         }
194       else
195         return FALSE;
196     }
197     
198     /* Return TRUE iff elt is in b */
199     bool bitset_member(bitset b, unsigned int elt)
200     {
201       int offset = compute_offset(b, elt);
202       int mask = compute_mask(b, elt);
203     
204       if (b[offset] & mask)
205         return TRUE; /* Prevent truncating conversions to bool */
206       return FALSE;
207     }
208     
209     /* Replace b1 by b1 & b2.  Destructively updates b1.  b2 is unchanged.
210        Returns TRUE iff b1 was changed. */
211     bool bitset_intersect(bitset b1, const bitset b2)
212     {
213       int num_uints, i;
214       bool changed;
215     
216       assert(bitset_length(b1) == bitset_length(b2));
217       num_uints = bitset_length(b1);
218       changed = FALSE;
219       for (i = 0; i < num_uints; i++)
220         {
221           if (b1[i+1] != (b1[i+1] & b2[i+1]))
222     	{
223     	  b1[i+1] = (b1[i+1] & b2[i+1]);
224     	  changed = TRUE;
225     	}
226         }
227       return changed;
228     }

230     /* Print b to stdout */
231     void bitset_print(bitset b)
232     {
233       int num_uints, i;
234     
235       num_uints = bitset_length(b);
236       for (i = 0; i < num_uints; i++)
237         printf("%08x", b[i+1]);
238     }

240     /* Return a hash for b */
241     int bitset_hash(bitset b)
242     {
243       int result, num_uints, i;
244     
245       result = b[0];
246       num_uints = bitset_length(b);
247       for (i = 0; i < num_uints; i++)
248         result = result*77 + ((int) b[i+1]);
249       return result;
250     }
251     
252     /* Return TRUE iff b1 and b2 are equal */
253     bool bitset_eq(bitset b1, bitset b2)
254     {
255       int num_uints, i;
256     
257       assert(bitset_length(b1) == bitset_length(b2));
258       num_uints = bitset_length(b1);
259       for (i = 0; i < num_uints; i++)
260         if (b1[i+1] != b2[i+1])
261           return FALSE;
262       return TRUE;
263     }
