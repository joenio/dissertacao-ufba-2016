88     static void clear(void *start, __rcintptr size)
89     {
90       long *clear, *clearend;
91     
92       clear = (long *)start;
93       clearend = (long *)((char *)start + size);
94       do *clear++ = 0;
95       while (clear < clearend) ;
96     }

109     static void nochildren(region r)
110     {
111       if (r->children)
112         abort();
113     }

115     static void unlink_region(region r)
116     {
117       region *scan;
118     
119       scan = &r->parent->children;
120       while (*scan != r)
121         scan = &(*scan)->sibling;
122       *scan = (*scan)->sibling;
123     }
124     
125     static void link_region(region r, region parent)
126     {
127       r->sibling = parent->children;
128       r->parent = parent;
129       parent->children = r;
130     }

132     static int rstart;
133     
134     void initregion(region r)
135     {
136       char *first =
137         (char *)r - rstart - offsetof(struct page, previous);
138     
139       /* Start using page with region header as a pointer-containing page */
140       r->normal.page.base = first;
141       r->normal.page.allocfrom = (char *)(r + 1);
142     
143       /* Guarantee failure for all other blocks */
144       r->normal.superpage.allocfrom = (char *)(K * RPAGESIZE + 1);
145       r->normal.hyperpage.allocfrom = (char *)(K * K * RPAGESIZE + 1);
146     
147       /* Remember that r owns this page. */
148       r->normal.pages = (struct page *)first;
149       set_region(r->normal.pages, 1, r);
150     }

152     region newregion(void)
153     {

155     }

157     region newsubregion(region parent)
158     {
159       char *first;
160       region r;
161     
162       first = (char *)alloc_single_page(NULL);
163       preclear(first + offsetof(struct page, pagecount), RPAGESIZE - offsetof(struct page, pagecount));
164     
165       /* stagger regions across cache lines a bit */
166       rstart += 64;
167       if (rstart > RPAGESIZE / K) rstart = 0;
168       r = (region)(first + rstart + offsetof(struct page, previous));
169       postclear(r, sizeof *r);
170       initregion(r);
171     
172       link_region(r, parent);
173     
174       return r;
175     }

220     static char *internal_rstrextend(region r, const char *old, size_t newsize,
221     				 int needsclear)
222     {
223       /* For now we don't attempt to extend the old storage area */
224       void *newmem, *hdr;
225       unsigned long *oldhdr, oldsize;
226     
227       qalloc(r, &r->normal, &hdr, sizeof(unsigned long), ALIGNMENT_LONG,
228     	 &newmem, newsize, RALIGNMENT, 0);
229     
230       /* If we don't do this we can't find the header: */
231       hdr = (char *)newmem - sizeof(unsigned long);
232     
233       *(unsigned long *)hdr = newsize;
234     
235       if (old)
236         {
237           oldhdr = (unsigned long *)(old - ALIGNMENT_LONG);
238           oldsize = *oldhdr;
239     
240           if (oldsize > newsize)
241     	oldsize = newsize;
242           else if (needsclear)
243     	clear((char *) newmem + oldsize, newsize - oldsize);
244           memcpy(newmem, old, oldsize);
245         }
246       else if (needsclear)
247         clear(newmem, newsize);
248     
249       return newmem;
250     }

335     static void printref(void *x)
336     {
337       if (x >= (void *)__rcregionmap && x < (void *)&__rcregionmap[MAXPAGE])
338         return;
339     
340     #ifdef RCPAIRS
341       if (x >= (void *)__rcregions && x < (void *)&__rcregions[MAXREGIONS])
342         return;
343     
344     #endif
345     
346       fprintf(out, "info symbol 0x%p\n", x);
347     }
348     
349     void findrefs(region r, void *from, void *to)
350     {
351       char *f;
352     
353       if (!out)
354         out = fopen("/dev/tty", "w");
355     
356       for (f = PALIGN(from, PTRALIGNMENT); f < (char *)to; f += PTRALIGNMENT)
357         if (regionof(*(void **)f) == r)
358           printref(f);
359     
360       fflush(out);
361     }

372     void findrrefs(region r, region from)
373     {
374       struct page *p;
375     
376       for (p = from->normal.pages; p; p = p->next)
377         findrefs(r, (char *)&p->previous, (char *)p + RPAGESIZE);
378     
379       for (p = r->normal.bigpages; p; p = p->next)
380         findrefs(r, (char *)&p->previous, (char *)p + p->pagecount * RPAGESIZE);
381     }
