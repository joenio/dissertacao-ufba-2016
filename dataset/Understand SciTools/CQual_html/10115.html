19     #include "parser.h"
20     #include "env.h"
21     #include "hash.h"
22     
23     struct env 
24     {
25       env parent;
26       region sameregion r;
27       hash_table entries;
28     };
29     
30     /* Create a new, empty environment with ancestor 'parent'.
31        The environment, and any future contents, are allocated in r */
32     env new_env(region r, env parent)
33     {
34       env e = ralloc(r, struct env);
35     
36       e->parent = parent;
37       e->r = r;
38       e->entries = NULL;
39     
40       return e;
41     }
42     
43     /* Return parent environment of e */
44     env env_parent(env e)
45     {
46       return e->parent;
47     }
48     
49     /* Return region of e */
50     region env_region(env e)
51     {
52       return e->r;
53     }
54     
55     /* Find entry s in in environment e. If not found, check ancestors
56        except if this_level_only is true.
57        Returns entry's value if s is found, NULL otherwise */
58     void *env_lookup(env e, const char *s, bool this_level_only)
59     {
60       for (;;)
61         {
62           void *value;
63     
64           if (e->entries && hash_table_lookup(e->entries, (void *) s, &value))
65     	return value;
66           if (this_level_only || !e->parent) 
67     	return NULL;
68           e = e->parent;
69         }
70     }
71     
72     /* Add an entry for s, with value 'value' to environment e.
73        Behaviour is undefined if e already contains an entry for s.
74        Does not copy s. */
75     void env_add(env e, const char *s, void *value)
76     {
77       if (e->entries == NULL)
78         e->entries = make_string_hash_table(e->r, 10);
79       hash_table_insert(e->entries, (void *) s, value);
80     }
81     
82     void env_scan(env e, env_scanner *scanner)
83     {
84       if (e->entries)
85         hash_table_scan(e->entries, scanner);
86       else
87         scanner->ht = NULL;
88     }
89     
90     bool env_next(env_scanner *scanner, const char **name, void **value)
91     {
92       if (!scanner->ht)
93         return FALSE;
94     
95       return hash_table_next(scanner, (void **) name, value);
96     }
