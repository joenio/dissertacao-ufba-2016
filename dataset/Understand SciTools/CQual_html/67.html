57     /**************************************************************************
58      *                                                                        *
59      * Effects                                                                *
60      *                                                                        *
61      **************************************************************************/
62     
63     /* The \emptyset of effects */
64     extern effect effect_empty;
65     
66     /* returns aloc from effect */
67     aloc aloc_effect(effect e);
68     
69     /* mark this aloc as interesting (optimization help for store) */
70     void mark_aloc_interesting(aloc al);
71     
72     /* The effect {al} */
73     effect effect_single(aloc al);
74     
75     /* The effect alloc(al), which is different than al itself */
76     effect effect_alloc(aloc al);
77     
78     /* The effect rwr(al) */
79     effect effect_rwr(aloc al);
80     
81     /* The effect r(al) */
82     effect effect_r(aloc al);
83     
84     /* The effect wr(al) */
85     effect effect_wr(aloc al);
86     
87     /* The effect k(al) */
88     effect effect_constr(aloc al, eff_kind k);
89     
90     /* Create a fresh effect variable */
91     effect effect_var(const char *name);
92     
93     /* Create a fresh effect variable with a unique name */
94     effect effect_fresh(void);
95     
96     /* The effect e1 \cup e2 */
97     effect effect_union(effect e1, effect e2);
98     
99     /* The effect e1 \cap e2.  e2 cannot be the empty effect. */
100     effect effect_inter(effect e1, effect e2);
101     
102     /* Mark this effect as interesting (optimization for caching
103        reachability queries) */
104     void mark_effect_interesting(effect e);
105     
106     /* Return FALSE if e1 may not be equal to e2.  Does not generate a
107        constraint. */
108     bool eq_effect(effect e1, effect e2);
109     
110     /* Add constraint that effect set e1 is a subset of effect set e2. */
111     void mkleq_effect(effect e1, effect e2);
112     
113     /* Add constraint that effect set e1 is equal to the effect set e2. */
114     void mkeq_effect(effect e1, effect e2);
115     
116     /* Add constraint that effect sets e1 and e2 are equal and
117        indistinguishable.  e1 and e2 must be effect variables. */
118     void unify_effect(effect e1, effect e2);
119     
120     /* Remember the constraint al !<= e2.  The constraint will be checked
121        when check_nin is called. */
122     void mknin_aloc_effect(location loc, aloc al, eff_kind kind, effect e);
123     
124     /* Remember the constraint e1 !<= e2.  The constraint will be checked
125       when check_nin is called.  e1 must not contain any effect variables.  */
126     void mknin_effect(location loc, effect e1, effect e2);
127     
128     /* Forall rho.kind(rho) \notin e */
129     void forall_aloc_mknin_effect(location loc, eff_kind kind, effect e);
130     
131     /* Forall rho s.t. kind(rho) \in e1.kind(rho) \notin e2 */
132     void forall_aloc_in_effect_mknin_effect(location loc, 
133     					eff_kind kind1, effect e1, 
134     					eff_kind kind2, effect e2);
135     
136     /* Test whether the constraint al \not\in e is satisfiable, i.e.,
137        whether anything in al is in e, which must be an effect variable.
138        Return TRUE if the constraint is unsatisfiable, FALSE if the
139        constraint is satisfiable. */
140     bool check_nin_aloc_effect(aloc al, effect e);
141     
142     /* Call check_nin_aloc_effect on all the constraints generated by
143        mknin_aloc_effect and mknin_effect.  Generates error messages with
144        calls to report_qerror. */
145     void check_nin(void);
146     
147     int print_effect(printf_func pf, effect e);
148     
149     /**************************************************************************
150      *                                                                        *
151      * Stores                                                                 *
152      *                                                                        *
153      **************************************************************************/
154     
155     /* Return TRUE iff s1 and s2 are the same store */
156     bool eq_store(store s1, store s2);
157     
158     /* Make a fresh store variable */
159     store make_store_var(location loc, const char *name);
160     
161     /* Make the store s|_e, s restricted to e */
162     store make_store_filter(location loc, store s, effect e);
163     
164     /* Make the store ref(al, s) = s + {al : fresh} */
165     store make_store_ref(location loc, store s, aloc al);
166     
167     /* Make the store ref(al1, ref(al2, ..., ref(aln, s))) for ali in e.  e
168        must be a singleton effect or a union */
169     store make_store_ref_effect(location loc, store s, effect e);
170     
171     /* Make the store store_ow(s1, s2, e) = s1|_e + s2|_{!e} */
172     store make_store_ow(location loc, store s1, store s2, effect e);
173     
174     /* Make the store assign(s, al, qt) = (s - al) + {al : qt}, plus
175        the constraint that if al is \omega, then qt = s(al) .  If strong
176        is TRUE then this must be a strong update. */
177     store make_store_assign(location loc, const char *err_msg, store s, aloc al,
178     			qtype qt, bool strong);
179     
180     /* Add constraint left <= right */
181     void mkleq_store(location loc, const char *err_msg, store left, store right);
182     
183     /* Add constraint left = right */
184     void mkeq_store(location loc, const char *err_msg, store s1, store s2);
185     
186     /* Return a fresh store s such that s1 <= s and s2 <= s */
187     store lub_store(location loc, const char *err_msg, store s1, store s2);
188     
189     /* Add constraint left == right, and can't distinguish left from right
190        afterward. */
191     void unify_store(location loc, const char *err_msg, store s1, store s2);
192     
193     /* Mark the al cell of s as propagating, and propagate it to
194        neighboring stores */
195     void propagate_store_cell_forward(store s, aloc al);
196     void propagate_store_cell_backward(store s, aloc al);
197     
198     /* Compute linearities */
199     void compute_lins(void);
200     
201     /* Return the linearity of al in s, either 0, 1, or 2 (= omega).
202       compute_lins() must be called first, and this function only works on
203       assign stores */
204     int lin_from_store(store s, aloc al);
205     
206     /* Return al's qtype in s */
207     qtype qtype_from_store(store s, aloc al);
208     
209     int print_store(printf_func pf, store s);
210     int print_store_cells(printf_func pf, pr_qual_fn pr_qual, store s,
211                           bool print_alocs);
212     int print_store_raw(printf_func pf, store s);
