146     static void output_line_directive(location l, bool include_filename)
147     {
148       startline_noindent();
149       if (include_filename)
150         /* The filename doesn't need any quoting as the quotes were not stripped on
151            input */
152         outputln("#line %lu \"%s\"%s", l->lineno, l->filename,
153     	     l->in_system_header ? " 3" : "");
154       else
155         outputln("#line %lu", l->lineno);
156     }

158     static void set_location(location l)
159     {
160       /* Ignore dummy locations */
161       if (l->filename == dummy_location->filename)
162         return;
163     
164       if ((l->filename != output_loc->filename &&
165            strcmp(l->filename, output_loc->filename)) ||
166           l->in_system_header != output_loc->in_system_header)
167         output_line_directive(l, TRUE);
168       else if (output_loc->lineno != l->lineno)
169         {
170           /* Just send some newlines for small changes */
171           if (output_loc->lineno < l->lineno &&
172     	  output_loc->lineno + 10 >= l->lineno)
173     	{
174     	  while (output_loc->lineno != l->lineno)
175     	    newline();
176     	}
177           else
178     	output_line_directive(l, FALSE);
179         }
180     
181       *output_loc = *l;
182     }

184     location output_location(void)
185     {
186       return output_loc;
187     }

279     {
280       tempregion = newregion();

410     void prt_simple_declarator(declarator d, bool not_star)
411     {
412       if (d)
413         switch (d->kind)
414           {
415           case kind_function_declarator:
416     	prt_simple_declarator(CAST(function_declarator, d)->declarator, TRUE);
417     	prt_parameters(CAST(function_declarator, d)->parms);
418     	break;
419           case kind_array_declarator:
420     	{
421     	  array_declarator ad = CAST(array_declarator, d);
422     
423     	  prt_simple_declarator(ad->declarator, TRUE);
424     	  if (!ad->arg1)
425     	    output("[]");
426     	  else
427     	    {
428     	      set_location(ad->arg1->loc);
429     	      output("[");
430     	      prt_expression(ad->arg1, P_TOP);
431     	      output("]");
432     	    }
433     	  break;
434     	}
435           case kind_pointer_declarator:
436     	{
437     	  pointer_declarator pd = CAST(pointer_declarator, d);
438     
439     	  if (pd->qualifiers)
440     	    set_location(pd->qualifiers->loc);
441     	  if (not_star)
442     	    output("(");
443     	  output("*");
444     	  prt_type_elements(pd->qualifiers);
445     	  prt_simple_declarator(pd->declarator, FALSE);
446     	  if (not_star)
447     	    output(")");
448     	  break;
449     	}
450           case kind_identifier_declarator:
451     	set_location(d->loc);
452     	output_cstring(CAST(identifier_declarator, d)->cstring);
453     	break;
454     
455           default: assert(0); break;
456           }
457     }

546     void prt_tag_ref(tag_ref tr)
547     {
548       set_location(tr->loc);
549       switch (tr->kind)
550         {
551         case kind_struct_ref: output("struct "); break;
552         case kind_union_ref: output("union "); break;
553         case kind_enum_ref: output("enum "); break;
554         default: assert(0);
555         }
556     
557       if (tr->word1)
558         prt_word(tr->word1);
559       if (tr->defined)
560         {
561           if (tr->kind == kind_enum_ref)
562     	prt_enumerators(tr->fields);
563           else
564     	prt_fields(tr->fields);
565         }
566       if (tr->attributes)
567         {
568           output(" ");
569           prt_type_elements(CAST(type_element, tr->attributes));
570         }
571     }

718     #define PRTEXPR(type, x) case kind_ ## type: prt_ ## type(CAST(type, (x)), context_priority); return
719     
720     /* Context priorities are that of the containing operator, starting at 0
721        for , going up to 14 for ->, . See the symbolic P_XX constants 
722        P_TOP (-1) is used for contexts with no priority restrictions. */
723     void prt_expression(expression e, int context_priority)
724     {
725       switch (e->kind) 
726         {
727           PRTEXPR(comma, e);
728           PRTEXPR(sizeof_type, e);
729           PRTEXPR(alignof_type, e);
730           PRTEXPR(label_address, e);
731           PRTEXPR(cast, e);
732           PRTEXPR(cast_list, e);
733           PRTEXPR(conditional, e);
734           PRTEXPR(identifier, e);
735           PRTEXPR(compound_expr, e);
736           PRTEXPR(function_call, e);
737           PRTEXPR(array_ref, e);

896     void prt_unary(unary e, int context_priority)
897     {
898       char *op = NULL, *postop = NULL;
899       int pri = 0;
900     
901       /* Yuck. Evil hack because gcc is broken (breaks the a[i] === *(a+i)
902          rule when a is a non-lvalue array). So we undo our earlier rewrite
903          (from fix.c) of a[i] as *(a+i). Note that gcc doesn't allow i[a] in
904          this case (bozos at work?) */
905       if (is_dereference(e) && is_plus(e->arg1))
906         {
907           plus derefed = CAST(plus, e->arg1);
908     
909           if (type_array(derefed->arg1->type))
910     	{
911     	  prt_binary_ref(derefed, context_priority);
912     	  return;
913     	}
914         }
915       switch (e->kind)
916         {
917         case kind_dereference: op = "*"; break;
918         case kind_extension_expr: op = "__extension__ "; break;
919           /* Higher priority for sizeof/alignof expr because we must
920     	 add parens around sizeof cast_expr (e.g. sizeof((char)x), not
921     	 sizeof (char)x */
922         case kind_sizeof_expr: op = "sizeof "; pri = P_CALL; break;
923         case kind_alignof_expr: op = "__alignof__ "; pri = P_CALL; break;
924         case kind_realpart: op = "__real__ "; break;
925         case kind_imagpart: op = "__imag__ "; break;
926         case kind_address_of: op = "&"; break;
927         case kind_unary_minus: op = "-"; break;
928         case kind_unary_plus: op = "+"; break;
929         case kind_preincrement: op = "++"; break;
930         case kind_predecrement: op = "--"; break;
931         case kind_postincrement: postop = "++"; break;
932         case kind_postdecrement: postop = "--"; break;
933         case kind_conjugate: case kind_bitnot: op = "~"; break;
934         case kind_not: op = "!"; break;
935         default: assert(0); return;
936         }
937       OPEN(P_CAST);
938       set_location(e->loc);
939       if (op)
940         {
941           output(op);
942           if (is_unary(e->arg1))
943     	output(" "); /* Catch weirdness such as - - x */
944           if (!pri)
945     	pri = P_CAST;
946         }
947       prt_expression(e->arg1, pri ? pri : P_CALL);
948       if (postop)
949         output(postop);
950       CLOSE(P_CAST);
951     }

953     const char *binary_op_name(ast_kind kind)
954     {
955       switch (kind)
956         {
957         case kind_plus: return "+"; 
958         case kind_minus: return "-"; 
959         case kind_times: return "*"; 
960         case kind_divide: return "/"; 
961         case kind_modulo: return "%"; 
962         case kind_lshift: return "<<"; 
963         case kind_rshift: return ">>"; 
964         case kind_leq: return "<="; 
965         case kind_geq: return ">="; 
966         case kind_lt: return "<"; 
967         case kind_gt: return ">"; 
968         case kind_eq: return "=="; 
969         case kind_ne: return "!="; 
970         case kind_bitand: return "&"; 
971         case kind_bitor: return "|"; 
972         case kind_bitxor: return "^"; 
973         case kind_andand: return "&&"; 
974         case kind_oror: return "||"; 
975         case kind_assign: return "="; 
976         case kind_plus_assign: return "+="; 
977         case kind_minus_assign: return "-="; 
978         case kind_times_assign: return "*="; 
979         case kind_divide_assign: return "/="; 
980         case kind_modulo_assign: return "%="; 
981         case kind_lshift_assign: return "<<="; 
982         case kind_rshift_assign: return ">>="; 
983         case kind_bitand_assign: return "&="; 
984         case kind_bitor_assign: return "|="; 
985         case kind_bitxor_assign: return "^="; 
986         default: assert(0); return "<bad>";
987         }
988     }

990     void prt_binary(binary e, int context_priority)
991     {
992       int pri, lpri, rpri;
993       const char *op = binary_op_name(e->kind);
994       bool perturb = FALSE;
995     
996       switch (e->kind)
997         {
998         case kind_times: case kind_divide: case kind_modulo:
999           lpri = P_TIMES; pri = P_TIMES; rpri = P_CAST; break;
1000         case kind_plus: case kind_minus:
1001           lpri = P_PLUS; pri = P_PLUS; rpri = P_TIMES; break;
1002         case kind_lshift: case kind_rshift:
1003           pri = P_SHIFT;
1004           if (CONSERVATIVE_PARENS)
1005     	lpri = rpri = P_TIMES;
1006           else
1007     	{
1008     	  lpri = P_SHIFT; rpri = P_PLUS; 
1009     	}
1010           break;
1011         case kind_leq: case kind_geq: case kind_lt: case kind_gt:
1012           lpri = P_REL; pri = P_REL; rpri = P_SHIFT; break;
1013         case kind_eq: case kind_ne:
1014           lpri = P_EQUALS; pri = P_EQUALS; rpri = P_REL; break;
1015         case kind_bitand:
1016           pri = P_BITAND;
1017           if (CONSERVATIVE_PARENS)
1018     	lpri = rpri = P_TIMES;
1019           else
1020     	{
1021     	  lpri = P_BITAND; rpri = P_EQUALS; 
1022     	}
1023           break;
1024         case kind_bitxor:
1025           pri = P_BITXOR;
1026           if (CONSERVATIVE_PARENS)
1027     	lpri = rpri = P_TIMES;
1028           else
1029     	{
1030     	  lpri = P_BITXOR; rpri = P_BITAND;
1031     	}
1032           break;
1033         case kind_bitor:
1034           pri = P_BITOR;
1035           if (CONSERVATIVE_PARENS)
1036     	lpri = rpri = P_TIMES;
1037           else
1038     	{
1039     	  lpri = P_BITOR; rpri = P_BITXOR;
1040     	}
1041           break;
1042         case kind_andand:
1043           lpri = P_AND; pri = P_AND; rpri = P_BITOR; break;
1044         case kind_oror:
1045           pri = P_OR;
1046           if (CONSERVATIVE_PARENS)
1047     	lpri = rpri = P_BITOR;
1048           else
1049     	{
1050     	  lpri = P_OR; rpri = P_AND; 
1051     	}
1052           break;
1053         case kind_assign: case kind_plus_assign: case kind_minus_assign: 
1054         case kind_times_assign: case kind_divide_assign: case kind_modulo_assign:
1055         case kind_lshift_assign: case kind_rshift_assign: case kind_bitand_assign:
1056         case kind_bitor_assign: case kind_bitxor_assign:
1057           lpri = P_CAST; pri = P_ASSIGN; rpri = P_ASSIGN; break;
1058         default: assert(0); return;
1059         }
1060       OPEN(pri);
1061       prt_expression(e->arg1, lpri);
1062       set_location(e->loc);
1063       output(" %s ", op);
1064       prt_expression(e->arg2, rpri);
1065       if (perturb)
1066         output(")");
1067       CLOSE(pri);
1068     }

1166     {
1167       statement s1;
1168     
1169       set_location(s->loc);
1170       outputln("{");
1171       indent();
1172       if (s->id_labels)

1177           scan_id_label (l, s->id_labels)

1196     }
