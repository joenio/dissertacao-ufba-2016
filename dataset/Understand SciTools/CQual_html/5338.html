24     node ast_chain(node l1, node l2)
25     {
26       if (!l1) return l2;
27       last_node(l1)->next = l2;
28       return l1;
29     }
30     
31     node last_node(node n)
32     {
33       if (!n) return NULL;
34       while (n->next) n = n->next;
35       return n;
36     }

38     int chain_length(node n)
39     {
40       int l = 0;
41     
42       while (n) 
43         {
44           n = n->next;
45           l++;
46         }
47       return l;
48     }

50     node ast_reverse(node l)
51     {
52       node last = NULL, next;
53     
54       for (;;)
55         {
56           if (!l)
57     	return last;
58           next = l->next;
59           l->next = last;
60           last = l;
61           l = next;
62         }
63     }

65     void insert_before(node sameregion *list, node before, node n)
66     {
67       while (*list != before)
68         list = &(*list)->next;
69       *list = n;
70       n->next = before;
71     }

133     {
134       node n = *nptr;
135     
136       if (parent)
137         AST_set_parent(nptr, parent);
138     
139       switch (n->kind)
140         {
141     /* Template:
142         case kind_foo: {

145           AST_set_parent_list(&x->field1, n);
146           AST_set_parent_list(&x->field2, n);

152           assert(0);

176     {
177       int i;
178     
179       for (i = 0; i < by; i++)
180         putchar(' ');
181     }

183     static void AST_print1(int indent, node n)
184     {
185       pindent(indent);
186       indent += 1;
187     
188       /* Special cases */
189       switch (n->kind)
190         {
191         case kind_identifier: {
192           identifier x = CAST(identifier, n);
193     
194           printf("identifier %s\n", x->ddecl->name);
195           return;
196         }
197         case kind_lexical_cst: {
198           lexical_cst x = CAST(lexical_cst, n);
199     
200           printf("lexical_cst %s", x->cstring.data);
201           return;
202         }
203         case kind_string_cst: {
204           string_cst x = CAST(string_cst, n);
205     
206           printf("string_cst %s\n", x->cstring.data);
207           return;
208         }
209         case kind_id_label: {
210           id_label x = CAST(id_label, n);
211     
212           printf("id_label %s\n", x->cstring.data);
213           return;
214         }
215         case kind_identifier_declarator: {
216           identifier_declarator x = CAST(identifier_declarator, n);
217     
218           printf("identifier_declarator %s\n", x->cstring.data);
219           return;
220         }
221         case kind_word: {
222           word x = CAST(word, n);
223     
224           printf("word %s\n", x->cstring.data);
225           return;
226         }
227         case kind_rid: {
228           rid x = CAST(rid, n);
229     
230           printf("rid %s\n", rid_name(x));
231           return;
232         }
233         default:
234           break;
235         }
236     
237       switch (n->kind)
238         {
239     /* Template:
240         case kind_foo: {
241           foo x = CAST(foo, n);
242     
243           puts("foo\n");
244           pindent(indent); puts("field1:\n"); AST_print_list(indent, x->field1);
245           pindent(indent); puts("field2:\n"); AST_print_list(indent, x->field2);
246           break;
247         }
248     */
249     #include "AST_print.c"
250         default:
251           assert(0);
252         }
253     }
254     
255     static void AST_print_list(int indent, void *vn)
256     {
257       node n = CAST(node, vn);
258     
259       while (n)
260         {
261           AST_print1(indent, n);
262           n = n->next;
263         }
264     }
