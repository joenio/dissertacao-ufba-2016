27     /* Types representing declarations */

29     typedef struct field_declaration {
30       struct field_declaration *next; /* Next field in struct/union */
31       const char *name;		/* May be NULL for bitfields (if NULL, bitwidth == 0) */
32       type type;
33       field_decl ast;
34       int bitwidth;			/* for bitfields, -1 otherwise */
35       size_t offset;		/* in bits, not bytes */
36       bool offset_cc:1;		/* TRUE if offset is a compile-time
37     				   constant (can be false because of
38     				   variable-size arrays) */
39       bool packed:1;		/* if packed attribute specified */
40     } *field_declaration;
41     
42     /* A struct, union or enum */
43     typedef struct tag_declaration {
44       int kind; /* One of kind_{struct/union/enum}_ref */
45       const char *name; /* NULL for anonynous struct/union/enum */
46       type reptype; /* The type used to represent an enum, NULL for struct
47     		   and unions */
48       /* fields and fieldlist are only defined for structs/unions */
49       env fields;
50       field_declaration fieldlist;
51       tag_ref ast;
52       struct tag_declaration *shadowed; /* Any struct with the same tag
53                                            defined in enclosing scope */
54       bool defined:1;
55       bool being_defined:1;
56       bool fields_const:1;
57       bool fields_volatile:1;
58       bool transparent_union:1;	/* transparent_union attribute is present */
59       bool size_cc:1;		/* TRUE if size is a compile-time
60     				   constant (can be false because of
61     				   variable-size arrays) */
62       bool packed:1;		/* if packed attribute specified */
63     
64       size_t size, alignment;
65     
66       /* Type qualifier specific fields */
67       qtype qtype;                  /* qtype */
68     } *tag_declaration;

70     typedef enum { decl_variable, decl_constant, decl_function,
71     	       decl_typedef, decl_error, decl_magic_string } data_kind;
72     
73     typedef struct data_declaration {
74       data_kind kind;
75       const char *name;
76       type type;
77     
78       struct data_declaration *shadowed; /* For extern's shadowing globals
79                                             in inner scopes */
80       struct data_declaration *global;   /* For externs the first
81     					data_declaration for this name */
82       declaration definition;  /* Pointer to actual definition, if any */
83       declaration ast;         /* Last declaration */
84       expression initialiser;  /* NULL if none */
85     
86       bool islimbo:1;          /* TRUE if comes from an extern declaration
87     		              in an inner scope (also true for implicit
88     		              function declarations) */
89     
90     
91     /* (TREE_PUBLIC)
92        variables: (means: name is to be accessible from outside this module)
93        in global scope: non-static, non-register
94        in local scope: DECL_EXTERNAL
95        functions: non-static, non-nested fns */
96       bool isexternalscope:1;
97     
98       /* (DECL_EXTERNAL)
99          variables: !initialised, `extern' specified
100          functions: undefined non-nested fns (so true for static functions!)
101          also true for extern inline functions (which aren't "really" defined) */
102       bool isfilescoperef:1;
103     
104       /* (TREE_STATIC)
105          variables (means: alllocate static storage)
106          in global scope: !DECL_EXTERNAL
107          in local scope: static
108          functions (means: has been defined) */
109       bool needsmemory:1;
110     
111       /* isused is TRUE if declaration used. For parameters, there is a special
112          use during parameter list declaration to support forward parameters:
113            - a duplicate parameter declaration is allowed if isused is FALSE
114              once a duplicate is seen, isused is set to TRUE
115            - parameters are created with isused == TRUE
116            - after the forward parameters are seen, they have their isused field
117              set to FALSE */
118       bool isused:1;
119       bool in_system_header:1;
120     
121       bool in_prelude:1;            /* Declared in a prelude file -- override
122     				   later declarations */
123       bool defined:1;               /* TRUE if defined somewhere, in any
124     				   file */
125       bool isallocated:1;           /* TRUE if 2p allocated it */
126       bool addresstaken:1;          /* TRUE if this function's address is
127     			           taken in a non-function call
128     			           context */
129       bool __init:1;                /* Declared __init or __initdata */
130     
131       /* For functions */
132       enum { function_implicit, function_normal, function_static, function_nested }
133         ftype:2;
134       bool isinline:1;
135       bool isexterninline:1;
136       bool noreturn:1;        /* Noreturn attribute specified */
137       typelist oldstyle_args; /* Type of arguments from old-style declaration */
138       char * alias;           /* Alias attribute */
139     
140       /* For variables */
141       enum { variable_register, variable_static, variable_normal } vtype:2;
142       bool islocal:1;     /* belongs to local scope; could be
143     			 auto, static, or register */
144       bool isparameter:1; /* implies islocal */
145     
146       /* For constants */
147       known_cst value;
148     
149       /* For magic_strings */
150       const wchar_t *chars;
151       size_t chars_length;
152     
153       qtype qtype;                  /* Qualified type */
154       qtype fs_qtype;               /* Flow-sensitive qualified type */
155     
156       alocreftype alref;            /* Pointsto */
157     } *data_declaration;

159     typedef struct label_declaration {

161       bool explicitly_declared:1;
162       bool used:1;
163       id_label firstuse; /* Never NULL */
164       id_label definition; /* NULL until actually defined */
165       function_decl containing_function;
