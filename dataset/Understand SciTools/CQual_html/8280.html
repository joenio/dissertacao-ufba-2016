19     #include <assert.h>
20     #include <stdio.h>
21     #include <string.h>
22     #include "buffer.h"
23     
24     /* Invariant: buffer always null-terminated */
25     struct growbuf
26     {
27       region r;
28       unsigned int maxsize, cursize;
29       char *buffer;
30     };
31     
32     /* Make a new buffer with initial size */
33     growbuf growbuf_new(region r, int size)
34     {
35       growbuf b = ralloc(r, struct growbuf);
36     
37       assert(size > 0);
38       b->r = r;
39       b->maxsize = size; /* Force some growth! */
40       b->cursize = 1;
41       b->buffer = rstralloc(r, size);
42       b->buffer[0] = '\0';
43       return b;
44     }
45     
46     /* Empty a buffer */
47     void growbuf_reset(growbuf b)
48     {
49       assert(b->maxsize > 0);
50       b->cursize = 1;
51       b->buffer[0] = '\0';
52     }
53     
54     /* Print to a buffer */
55     int gprintf(growbuf b, const char *fmt, ...)
56     {
57       va_list args;
58       va_start(args, fmt);
59       return gvprintf(b, fmt, args);
60     }
61     
62     /* Print to a buffer */
63     int gvprintf(growbuf b, const char *fmt, va_list args)
64     {
65       int nchars;
66     
67       if (!fmt) /* Bug (?)/feature of vnsprintf -- printing \0 returns -1,
68     	       goes into infinite loop. */
69         return 0;
70       while (1)
71         {
72           char *bufStart;
73           int sizeLeft;
74     
75           bufStart = b->buffer + b->cursize - 1; /* chop trailing \0 */
76           sizeLeft = b->maxsize - b->cursize + 1; /* +1 size we're chooping
77     						 the trailing \0 */
78           assert(*bufStart == '\0');
79           nchars = vsnprintf(bufStart, sizeLeft, fmt, args);
80           if (nchars > -1 && nchars < sizeLeft)
81     	{
82     	  b->cursize += nchars; /* nchars doesn't include \0,
83     				   but we overwrote our \0 */
84     	  break;
85     	}
86           else
87     	{
88     	  /* How much room do we need?  In the new glibc, nchars
89     	     tells us how much (not including the trailing null).
90     	     So we need the current size, -1 since we'll remove the null,
91     	     plus the new size, plus 1 for the new null. */
92     	  int newSize = (nchars > -1) ? b->cursize - 1 + nchars + 1
93     	                              : b->maxsize * 2;
94     	  char *newBuf;
95     
96     	  /* fprintf(stderr, "Reallocating buffer, newSize=%d\n", newSize); */
97     	  newBuf = rstralloc(b->r, newSize);
98     	  memcpy(newBuf, b->buffer, b->cursize);
99     	  newBuf[b->cursize-1] = '\0';  /* vsnprintf has printed something! */
100     	  b->buffer = newBuf;
101     	  b->maxsize = newSize;
102     	  /* b->cursize unchanged */
103     	}
104         }
105       return nchars;
106     }
107     
108     /* Get the contents of a buffer */
109     char *growbuf_contents(growbuf b)
110     {
111       return b->buffer;
112     }
113     
114     bool growbuf_empty(growbuf b)
115     {
116       return b->cursize == 1; /* Buffer always null terminated */
117     }
