473     static void add_program_files(const char *file_list, dd_list files)
474     {
475       FILE *list = fopen(file_list, "r");
476       
477       if(!list)
478         {
479           error("unable to open %s", file_list);
480           return;
481         }
482     
483       /* each line in */
484       while(!ferror(list) && !feof(list))
485         {
486           char *newline = NULL;
487           char name[PATH_MAX] = {0};
488           
489           if(!fgets(name, PATH_MAX, list))
490     	break;
491     
492           /* if there is a \n, replace it with \0 */
493           newline = strchr(name, (int)'\n');
494           if (newline)
495     	*newline = '\0';
496     
497           /* add the file to the files list */
498           dd_add_last(parse_region, files, rstrdup(parse_region, name));
499         }
500     }

502     /* Compile an entire file of output from cpp, named NAME.
503        Write a file of assembly output and various debugging dumps.  */
504     static void compile_file(char *name)
505     {
506       FILE *ifile;
507     
508     #if !USE_CPPLIB
509       /* Open input file.  */
510     
511       if (name == 0 || !strcmp (name, "-"))
512         {
513           ifile = stdin;
514           name = "stdin";
515         }
516       else
517         ifile = fopen (name, "r");
518       if (ifile == 0)
519         pfatal_with_name (name);
520     
521     #ifdef IO_BUFFER_SIZE
522       setvbuf(ifile, (char *) xmalloc (IO_BUFFER_SIZE), _IOFBF, IO_BUFFER_SIZE);
523     #endif
524     #endif /* !USE_CPPLIB */
525     
526       set_nomem_handler(outofmemory);
527     
528       set_input(ifile, name);
529       init_lex(); /* needs to go before init_semantics, since inits
530     		 dummy_location */
531       init_semantics();
532     
533       if (yyparse () != 0)
534         {
535           if (errorcount == 0)
536     	fprintf (stderr, "Errors detected in input file (your bison.simple is out of date)");
537         }
538     
539       fclose (ifile);
540     
541       if (errorcount == 0)
542         {
543           /*if (flag_parse_only)
544     	{
545     	  fprintf(stderr, "\nDONE. Unparsing............\n\n");
546     	  unparse(stdout, the_program);
547     	}
548           else
549           exit(exec_cc1(the_program));*/
550     
551           if (the_program)
552     	{
553     	  parsed_file pf;
554     
555     	  pf = ralloc(parse_region, struct parsed_file);
556     	  pf->name = name;
557     	  pf->program = the_program;
558     	  dd_add_last(parse_region, parsed_files, pf);
559     
560     	  analyze(the_program);
561     	}
562         }
563     }

566     void c_decode_option(char *p)
567     {
568       if (!strcmp (p, "-ftraditional") || !strcmp (p, "-traditional"))
569         {
570           flag_traditional = 1;
571         }
572       else if (!strcmp (p, "-fallow-single-precision"))
573         flag_allow_single_precision = 1;
574       else if (!strcmp (p, "-fnotraditional") || !strcmp (p, "-fno-traditional"))
575         {
576           flag_traditional = 0;
577         }
578       else if (!strcmp (p, "-fdollars-in-identifiers"))
579         dollars_in_ident = 1;
580       else if (!strcmp (p, "-fno-dollars-in-identifiers"))
581         dollars_in_ident = 0;
582       else if (!strcmp (p, "-fsigned-char"))
583         flag_signed_char = 1;
584       else if (!strcmp (p, "-funsigned-char"))
585         flag_signed_char = 0;
586       else if (!strcmp (p, "-fno-signed-char"))
587         flag_signed_char = 0;
588       else if (!strcmp (p, "-fno-unsigned-char"))
589         flag_signed_char = 1;
590       else if (!strcmp (p, "-fsigned-bitfields")
591     	   || !strcmp (p, "-fno-unsigned-bitfields"))
592         {
593           flag_signed_bitfields = 1;
594           explicit_flag_signed_bitfields = 1;
595         }
596       else if (!strcmp (p, "-funsigned-bitfields")
597     	   || !strcmp (p, "-fno-signed-bitfields"))
598         {
599           flag_signed_bitfields = 0;
600           explicit_flag_signed_bitfields = 1;
601         }
602       else if (!strcmp (p, "-fshort-enums"))
603         flag_short_enums = 1;
604       else if (!strcmp (p, "-fno-short-enums"))
605         flag_short_enums = 0;
606       else if (!strcmp (p, "-fcond-mismatch"))
607         flag_cond_mismatch = 1;
608       else if (!strcmp (p, "-fno-cond-mismatch"))
609         flag_cond_mismatch = 0;
610       else if (!strcmp (p, "-fasm"))
611         flag_no_asm = 0;
612       else if (!strcmp (p, "-fno-asm"))
613         flag_no_asm = 1;
614       else if (!strcmp (p, "-ansi"))
615         flag_no_asm = 1;
616       else if (!strcmp (p, "-Werror-implicit-function-declaration"))

623         warn_implicit_int = 1;

628           warn_implicit_int = 1;

631         }
632       else if (!strcmp (p, "-Wno-implicit"))
633         warn_implicit_int = 0, mesg_implicit_function_declaration = 0;
634       else if (!strcmp (p, "-Wwrite-strings"))
635         warn_write_strings = 1;
636       else if (!strcmp (p, "-Wno-write-strings"))
637         warn_write_strings = 0;
638       else if (!strcmp (p, "-Wcast-qual"))
639         warn_cast_qual = 1;
640       else if (!strcmp (p, "-Wno-cast-qual"))
641         warn_cast_qual = 0;
642       else if (!strcmp (p, "-Wbad-function-cast"))
643         warn_bad_function_cast = 1;
644       else if (!strcmp (p, "-Wno-bad-function-cast"))
645         warn_bad_function_cast = 0;
646       else if (!strcmp (p, "-Wpointer-arith"))
647         warn_pointer_arith = 1;
648       else if (!strcmp (p, "-Wno-pointer-arith"))
649         warn_pointer_arith = 0;
650       else if (!strcmp (p, "-Wstrict-prototypes"))
651         warn_strict_prototypes = 1;
652       else if (!strcmp (p, "-Wno-strict-prototypes"))
653         warn_strict_prototypes = 0;
654       else if (!strcmp (p, "-Wmissing-prototypes"))
655         warn_missing_prototypes = 1;
656       else if (!strcmp (p, "-Wno-missing-prototypes"))
657         warn_missing_prototypes = 0;
658       else if (!strcmp (p, "-Wmissing-declarations"))
659         warn_missing_declarations = 1;
660       else if (!strcmp (p, "-Wno-missing-declarations"))
661         warn_missing_declarations = 0;
662       else if (!strcmp (p, "-Wredundant-decls"))
663         warn_redundant_decls = 1;
664       else if (!strcmp (p, "-Wno-redundant-decls"))
665         warn_redundant_decls = 0;
666       else if (!strcmp (p, "-Wnested-externs"))
667         warn_nested_externs = 1;
668       else if (!strcmp (p, "-Wno-nested-externs"))
669         warn_nested_externs = 0;
670       else if (!strcmp (p, "-Wtraditional"))
671         warn_traditional = 1;
672       else if (!strcmp (p, "-Wno-traditional"))
673         warn_traditional = 0;
674       else if (!strcmp (p, "-Wformat"))
675         warn_format = 1;
676       else if (!strcmp (p, "-Wno-format"))
677         warn_format = 0;
678       else if (!strcmp (p, "-Wchar-subscripts"))
679         warn_char_subscripts = 1;
680       else if (!strcmp (p, "-Wno-char-subscripts"))
681         warn_char_subscripts = 0;
682       else if (!strcmp (p, "-Wconversion"))
683         warn_conversion = 1;
684       else if (!strcmp (p, "-Wno-conversion"))
685         warn_conversion = 0;
686       else if (!strcmp (p, "-Wparentheses"))
687         warn_parentheses = 1;
688       else if (!strcmp (p, "-Wno-parentheses"))
689         warn_parentheses = 0;
690       else if (!strcmp (p, "-Wreturn-type"))
691         warn_return_type = 1;
692       else if (!strcmp (p, "-Wno-return-type"))
693         warn_return_type = 0;
694       else if (!strcmp (p, "-Wmissing-braces"))
695         warn_missing_braces = 1;
696       else if (!strcmp (p, "-Wno-missing-braces"))
697         warn_missing_braces = 0;
698       else if (!strcmp (p, "-Wmain"))
699         warn_main = 1;
700       else if (!strcmp (p, "-Wno-main"))
701         warn_main = 0;
702       else if (!strcmp (p, "-Wsign-compare"))
703         warn_sign_compare = 1;
704       else if (!strcmp (p, "-Wno-sign-compare"))
705         warn_sign_compare = 0;
706       else if (!strcmp (p, "-Wmultichar"))
707         warn_multichar = 1;
708       else if (!strcmp (p, "-Wno-multichar"))
709         warn_multichar = 0;
710       else if (!strcmp (p, "-Wall"))
711         {
712           /* We save the value of warn_uninitialized, since if they put
713     	 -Wuninitialized on the command line, we need to generate a
714     	 warning about not using it without also specifying -O.  */
715           if (warn_uninitialized != 1)
716     	warn_uninitialized = 2;
717           warn_implicit_int = 1;
718           mesg_implicit_function_declaration = 1;
719           warn_return_type = 1;
720           warn_unused = 1;
721           warn_switch = 1;
722           warn_format = 1;
723           warn_char_subscripts = 1;
724           warn_parentheses = 1;
725           warn_missing_braces = 1;
726           /* We set this to 2 here, but 1 in -Wmain, so -ffreestanding can turn
727     	 it off only if it's not explicit.  */
728           warn_main = 2;
729         }
730     }
731     
732     
733     /* Entry point of cc1/c++.  Decode command args, then call compile_file.
734        Exit code is 35 if can't open files, 34 if fatal error,
735        33 if had nonfatal errors, else success.  */
736     int main(int argc, char **argv)
737     {
738       register int i;
739       dd_list files, preludes;
740       dd_list_pos cur;
741       int version_flag = 0;
742       char *p;
743       char *config_file = NULL;
744     
745     #ifdef TIMER_USERTIME
746       reset_timer(&total_time);
747     #endif
748       start_timer(&total_time);
749       region_init();
750       parse_region = newregion();
751       in_prelude = FALSE;
752       num_hotspots = 0;
753       parsed_files = dd_new_list(parse_region);
754     
755       copy_argc = 0;
756       copy_argv = xmalloc((argc + 1) * sizeof(*copy_argv));
757       files = dd_new_list(parse_region);
758       preludes = dd_new_list(parse_region);
759     
760       p = argv[0] + strlen (argv[0]);
761       while (p != argv[0] && p[-1] != '/'
762     #ifdef DIR_SEPARATOR
763     	 && p[-1] != DIR_SEPARATOR
764     #endif
765     	 )
766         --p;
767       progname = p;
768     
769     #ifdef SIGPIPE
770       signal (SIGPIPE, pipe_closed);
771     #endif
772     
773       copy_argv[0] = argv[0];
774       copy_argc = 1;
775       for (i = 1; i < argc; i++)
776         {
777           int j;
778           bool copy_arg = TRUE;
779     
780           /* If this is a language-specific option,
781     	 decode it in a language-specific way.  */
782           for (j = 0; lang_options[j] != 0; j++)
783     	if (!strncmp (argv[i], lang_options[j],
784     		      strlen (lang_options[j])))
785     	  break;
786           if (lang_options[j] != 0)
787     	/* If the option is valid for *some* language,
788     	   treat it as valid even if this language doesn't understand it.  */
789     	c_decode_option(argv[i]);
790           else if (argv[i][0] == '-' && argv[i][1] != 0)
791     	{
792     	  register char *str = argv[i] + 1;
793     	  if (str[0] == 'Y')
794     	    str++;
795     
796     	  if (!strcmp (str, "dumpbase"))
797     	    copy_argv[copy_argc++] = argv[i++];
798     	  else if (str[0] == 'f')
799     	    {
800     	      register char *p = &str[1];
801     	      int found = 0;
802     
803     	      /* Some kind of -f option.
804     		 P's value is the option sans `-f'.
805     		 Search for it in the table of options.  */
806     
807     	      for (j = 0;
808     		   !found && j < sizeof (f_options) / sizeof (f_options[0]);
809     		   j++)
810     		{
811     		  if (!strcmp (p, f_options[j].string))
812     		    {
813     		      *f_options[j].variable = f_options[j].on_value;
814     		      /* A goto here would be cleaner,
815     			 but breaks the vax pcc.  */
816     		      found = 1;
817     		    }
818     		  if (p[0] == 'n' && p[1] == 'o' && p[2] == '-'
819     		      && ! strcmp (p+3, f_options[j].string))
820     		    {
821     		      *f_options[j].variable = ! f_options[j].on_value;
822     		      found = 1;
823     		    }
824     		}
825     	    }
826     	  else if (!strcmp (str, "pedantic"))
827     	    pedantic = 1;
828     	  else if (!strcmp (str, "pedantic-errors"))
829     	    flag_pedantic_errors = pedantic = 1;
830     	  else if (!strcmp (str, "quiet"))
831     	    quiet_flag = 1;
832     	  else if (!strcmp (str, "version"))
833     	    version_flag = 1;
834     	  else if (!strcmp (str, "w"))
835     	    inhibit_warnings = 1;
836     	  else if (!strcmp (str, "W"))
837     	    {
838     	      extra_warnings = 1;
839     	      /* We save the value of warn_uninitialized, since if they put
840     		 -Wuninitialized on the command line, we need to generate a
841     		 warning about not using it without also specifying -O.  */
842     	      if (warn_uninitialized != 1)
843     		warn_uninitialized = 2;
844     	    }
845     	  else if (str[0] == 'W')
846     	    {
847     	      register char *p = &str[1];
848     	      int found = 0;
849     
850     	      /* Some kind of -W option.
851     		 P's value is the option sans `-W'.
852     		 Search for it in the table of options.  */
853     
854     	      for (j = 0;
855     		   !found && j < sizeof (W_options) / sizeof (W_options[0]);
856     		   j++)
857     		{
858     		  if (!strcmp (p, W_options[j].string))
859     		    {
860     		      *W_options[j].variable = W_options[j].on_value;
861     		      /* A goto here would be cleaner,
862     			 but breaks the vax pcc.  */
863     		      found = 1;
864     		    }
865     		  if (p[0] == 'n' && p[1] == 'o' && p[2] == '-'
866     		      && ! strcmp (p+3, W_options[j].string))
867     		    {
868     		      *W_options[j].variable = ! W_options[j].on_value;
869     		      found = 1;
870     		    }
871     		}
872     
873     	      if (found)
874     		;
875     	      else if (!strncmp (p, "id-clash-", 9))
876     		{
877     		  char *endp = p + 9;
878     
879     		  while (*endp)
880     		    {
881     		      if (*endp >= '0' && *endp <= '9')
882     			endp++;
883     		      else
884     			{
885     			  error ("Invalid option `%s'", argv[i]);
886     			  goto id_clash_lose;
887     			}
888     		    }
889     		  warn_id_clash = 1;
890     		  id_clash_len = atoi (str + 10);
891     		id_clash_lose: ;
892     		}
893     	      else if (!strncmp (p, "larger-than-", 12))
894     		{
895     		  char *endp = p + 12;
896     
897     		  while (*endp)
898     		    {
899     		      if (*endp >= '0' && *endp <= '9')
900     			endp++;
901     		      else
902     			{
903     			  error ("Invalid option `%s'", argv[i]);
904     			  goto larger_than_lose;
905     			}
906     		    }
907     		  warn_larger_than = 1;
908     		  larger_than_size = atoi (str + 13);
909     		larger_than_lose: ;
910     		}
911     	    }
912     	  else if (!strcmp (str, "o"))
913     	    copy_argv[copy_argc++] = argv[i++];
914     	  else if (str[0] == 'G')
915     	    {
916     	      if (str[1] == '\0')
917     		copy_argv[copy_argc++] = argv[i++];
918     	    }
919     	  else if (!strncmp (str, "aux-info", 8))
920     	    {
921     	      if (str[8] == '\0')
922     		copy_argv[copy_argc++] = argv[i++];
923     	    }
924     	  else if (!strcmp(str, "config"))
925     	    {
926     	      if (i < argc - 1)
927     		{
928     		  i++;
929     		  config_file = strdup(argv[i]);
930     		}
931     	      else
932     		error ("Missing -config file");
933     	    }
934     	  else if (!strcmp(str, "prelude"))
935     	    {
936     	      if (i < argc - 1)
937     		{
938     		  i++;
939     		  dd_add_last(parse_region, preludes,
940     			      rstrdup(parse_region, argv[i]));
941     		}
942     	      else
943     		error("Missing -prelude file");
944     	    }
945     	  else if (!strcmp(str, "hotspots"))
946     	    {
947     	      if (i < argc - 1)
948     		{
949     		  i++;
950     		  num_hotspots = atoi(argv[i]);
951     		  if (num_hotspots < 0)
952     		    error("Negative value for -hotspots count");
953     		}
954     	      else
955     		error("Missing -hotspots count");
956     	    }
957     	  else if (!strcmp( str, "program-files"))
958     	    {
959     	      if (i < argc - 1)
960     		{
961     		  i++;
962     		  add_program_files(argv[i], files);
963     		}
964     	      else
965     		error("Missing -program-files file");
966     	    }
967     	}
968           else if (argv[i][0] == '+')
969     	;
970           else
971     	{
972     	  /* Allow wildcards, because PAM won't expand files */
973     	  glob_t globbuf;
974     	  char **cur;
975     
976     	  if (glob(argv[i], 0, NULL, &globbuf))
977     	    {
978     	      /* glob returned non-zero error status; abort */
979     	      fprintf(stderr, "%s: file not found\n", argv[i]);
980     	      exit(FATAL_EXIT_CODE);
981     	    }
982     	  else
983     	    for (cur = globbuf.gl_pathv; *cur; cur++)
984     	      {
985     		/* Assume anything called prelude.i is a prelude file */
986     		if ( strlen(*cur) >= 9 &&
987                         !strncmp("prelude.i", *cur + strlen(*cur) - 9, 9))
988     		  dd_add_last(parse_region, preludes,
989     			      rstrdup(parse_region, *cur));
990     		else
991     		  dd_add_last(parse_region, files,
992     			      rstrdup(parse_region, *cur));
993     	      }
994     	  copy_arg = FALSE;
995     	}
996     
997           if (copy_arg)
998     	copy_argv[copy_argc++] = argv[i];
999         }
1000       copy_argv[copy_argc] = NULL;
1001     
1002       if (flag_casts_preserve && flag_flow_sensitive) {
1003         fprintf(stderr, "-fcasts-preserve currently not allowed with "
1004     	    "-fflow_sensitive");
1005         exit(FATAL_EXIT_CODE);
1006       }
1007     
1008       /* Now analyze *all* of the files.  First, initialize all appropriate
1009          data structures. */
1010       init_types();
1011       cval_init();
1012       init_effects();
1013       init_qtype();
1014       init_quals();
1015       init_qerror();
1016     
1017       init_store();
1018     
1019       if (config_file)
1020         load_config_file_quals(config_file);
1021     
1022       /* Add const so that we can do -fconst-subtyping no matter what */
1023       if (!const_qual)
1024         {
1025           begin_po_qual();
1026           const_qual = add_qual("const");
1027           add_level_qual(const_qual, level_ref);
1028           set_po_nonprop();
1029           end_po_qual();
1030         }
1031     
1032       /* Add volatile so we can handle noreturn functions */
1033       if (!volatile_qual)
1034         {
1035           begin_po_qual();
1036           volatile_qual = add_qual("volatile");
1037           add_level_qual(volatile_qual, level_ref);
1038           add_sign_qual(volatile_qual, sign_eq);
1039           set_po_nonprop();
1040           end_po_qual();
1041         }
1042     
1043       /* Add noreturn, for non-returning functions */
1044       if (!noreturn_qual)
1045         {
1046           begin_po_qual();
1047           noreturn_qual = add_qual("noreturn");
1048           add_level_qual(noreturn_qual, level_value);
1049           add_sign_qual(noreturn_qual, sign_eq);
1050           set_po_nonprop();
1051           end_po_qual();
1052         }
1053     
1054       end_define_pos(); /* Allow cqual to run with no qualifiers */
1055     
1056       init_pam();
1057       init_analyze();
1058       found_fs_qual = FALSE; /* Force reset, since init_analyze may
1059     			    look up some quals */
1060     
1061       /* Now analyze the prelude files */
1062       in_prelude = TRUE;
1063       dd_scan(cur, preludes)
1064         {
1065           char *file;
1066           
1067           file = DD_GET(char *, cur);
1068           fprintf(stderr, "Analyzing prelude %s\n", file);
1069           compile_file(file);
1070         }
1071       in_prelude = FALSE;
1072       
1073       /* Finally, analyze the source files */
1074       if (dd_is_empty(files))
1075         compile_file(0); /* If no files given, compile stdin */
1076       else
1077         dd_scan(cur, files)
1078         {
1079           char *file;
1080           
1081           file = DD_GET(char *, cur);
1082           fprintf(stderr, "Analyzing %s\n", file);
1083           compile_file(file);
1084         }
1085     
1086       if (errorcount)
1087         exit (FATAL_EXIT_CODE);
1088       finish_analyze();
1089     
1090       if (flag_flow_sensitive && (flag_force_flow_sensitive || found_fs_qual))
1091         {
1092           dd_list_pos cur;
1093     
1094           init_flow_sensitive();
1095           dd_scan(cur, parsed_files)
1096     	{
1097     	  parsed_file pf;
1098     
1099     	  pf = DD_GET(parsed_file, cur);
1100     	  fprintf(stderr, "Flow-sensitive analysis of %s\n", pf->name);
1101     	  analyze_flow_sensitive(pf->program);
1102     	}
1103           finish_flow_sensitive();
1104         }
1105     
1106       finish_quals();
1107     
1108       end_timer(&total_time);
1109       if (flag_statistics)
1110         memusage = get_memusage();
1111     
1112       if (flag_print_results)
1113         print_results();
1114     
1115       if (flag_pam_mode)
1116         enter_pam_mode();
1117     
1118       if (flag_statistics)
1119         {
1120           fflush(NULL);
1121           fprintf(stderr, "VMSize: %ldK\n", memusage / 1024);
1122           fprintf(stderr, "Elapsed time: %ss\n", timer_to_ascii(&total_time));
1123           fprintf(stderr, "Analysis complete\n");
1124         }
1125     
1126       fflush(NULL);
1127       exit (SUCCESS_EXIT_CODE);
1128       return 0;
1129     }
