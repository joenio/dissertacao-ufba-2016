19     #include <regions.h>
20     #include "dd_list.h"

22     /* This is a hack: the list header holds the two pseudo-elements for the
23        beginning & end of a list: the `beginning' pseudo element has:
24          next = first element
25          previous = NULL
26          data = xxx
27        the `end' pseudo element has:
28          next = NULL
29          previous = last element
30          data = xxx
31     
32        if l is a dd_list, the `beginning' element is (dd_list_pos *)l
33        and the `end' is (dd_list_pos *)&l->null
34     
35        l is initialised to make all this work ...
36        Source: AmigaOS Exec kernel
37     */
38     struct dd_list
39     {
40       struct dd_list_pos *first;
41       struct dd_list_pos *null;
42       struct dd_list_pos *last;
43     };
44     
45     static dd_list_pos new_hdr(region r, void *data, dd_list_pos previous, dd_list_pos next)
46     /* Returns: A new list element containing `data', with previous & next
47         initialised
48     */
49     {
50       dd_list_pos new = ralloc(r, struct dd_list_pos);
51     
52       new->data = data;
53       new->previous = previous;
54       new->next = next;
55     
56       return new;
57     }
58     
59     dd_list dd_new_list(region r)
60     /* Returns: A new empty list
61     */
62     {
63       dd_list new = ralloc(r, struct dd_list);
64     
65       /* Initialise `beginning' and `end' elements (see comment at top) */
66       new->first = (dd_list_pos)&new->null;
67       new->null = NULL;
68       new->last = (dd_list_pos)new;
69     
70       return new;
71     }
72     
73     void dd_add_first(region r, dd_list l, void *data)
74     /* Effects: Adds a new element containing `data' to the beginning of l.
75        Modifies: l
76     */
77     {
78       dd_insert_after(r, (dd_list_pos)l, data);
79     }
80     
81     void dd_add_last(region r, dd_list l, void *data)
82     {
83       dd_insert_before(r, (dd_list_pos)&l->null, data);
84     }
85     
86     void dd_insert_before(region r, dd_list_pos where, void *data)
87     /* Effects: Adds a new element containg `data' after element `where'.
88        Modifies: the list containing `where'
89     */
90     {
91       dd_list_pos new = new_hdr(r, data, where->previous, where);
92     
93       where->previous->next = new;
94       where->previous = new;
95     }
96     
97     void dd_insert_after(region r, dd_list_pos where, void *data)
98     /* Effects: Adds a new element containg `data' before element `where'.
99        Modifies: the list containing `where'
100     */
101     {
102       dd_list_pos new = new_hdr(r, data, where, where->next);
103     
104       where->next->previous = new;
105       where->next = new;
106     }
107     
108     void dd_remove(dd_list_pos what)
109     /* Effects: Removes element `what' from its list.
110          No operations on what are valid after the call to dd_remove.
111        Modifies: the list containing `what'.
112     */
113     {
114       what->previous->next = what->next;
115       what->next->previous = what->previous;
116     
117       what->next = what->previous = NULL;
118       what->data = NULL;
119     }
120     
121     dd_list_pos dd_first(dd_list l)
122     /* Returns: The first element of list l
123     */
124     {
125       return l->first;
126     }
127     
128     dd_list_pos dd_last(dd_list l)
129     /* Returns: The last element of list l
130     */
131     {
132       return l->last;
133     }
134     
135     unsigned long dd_length(dd_list l)
136     /* Returns: length of list l
137     */
138     {
139       dd_list_pos scan;
140       unsigned long len = 0;
141     
142       dd_scan (scan, l) len++;
143     
144       return len;
145     }
146     
147     void dd_append(dd_list l1, dd_list l2)
148     /* Effects: Appends list l2 to the end of list l1.
149          List l2 is destroyed
150        Modifies: l1, l2
151     */
152     {
153       l1->last->next = l2->first;
154       l2->first->previous = l1->last;
155       l2->last->next = (dd_list_pos)&l1->null;
156       l1->last = l2->last;
157     
158       l2->first = l2->last = NULL;
159     }
160     
161     dd_list dd_copy(region r, dd_list l)
162     /* Returns: A new list with the same elements as l
163     */
164     {
165       dd_list new = dd_new_list(r);
166       dd_list_pos scan;
167       
168       dd_scan (scan, l) dd_add_last(r, new, scan->data);
169     
170       return new;
171     }
172     
173     void dd_free_list(dd_list l, void (*delete)(dd_list_pos p))
174     /* Effects: Destroys list l and all its elements.
175          If delete is not NULL, calls it before deleting each element.
176     */
177     {
178       dd_list_pos scan, next;
179     
180       scan = dd_first(l);
181       while (!dd_is_end(scan))
182         {
183           next = dd_next(scan);
184           if (delete) delete(next);
185     
186           scan->next = scan->previous = NULL;
187           scan->data = NULL;
188     
189           scan = next;
190         }
191       l->first = l->last = NULL;
192     }
193     
194     dd_list_pos dd_find(dd_list l, void *find)
195     /* Returns: The element of l whose data is 'find', or NULL if it isn't found
196     */
197     {
198       dd_list_pos scan;
199     
200       dd_scan (scan, l)
201         if (DD_GET(void *, scan) == find) return scan;
202     
203       return NULL;
204     }
