19     /* Constants come in three kinds:
20        - "unknown": value is completely unknown (constant_unknown)
21        - "address": value is the address of some global symbol and an offset 
22                     (constant_address)
23        - "value": known value (integer, floating point)
24     */

71     /* Print a warning if a constant expression had overflow in folding.
72        Invoke this function on every expression that the language
73        requires to be a constant expression.
74        Note the ANSI C standard says it is erroneous for a
75        constant expression to overflow.  */
76     void constant_overflow_warning(known_cst c);

78     /*bool constant_address(known_cst c);*/
79     #define constant_address(c) cval_isaddress((c)->cval)
80     /* Returns: TRUE if c is an address constant
81      */
82     
83     /*bool constant_unknown(known_cst c);*/
84     #define constant_unknown(c) cval_isunknown((c)->cval)
85     /* Returns: TRUE if c is a constant whose value is not computable 
86        at load time (these behave for most, but not all, purposes as 
87        non-constant expressions)
88      */
89     
90     /*bool constant_integral(known_cst c);*/
91     #define constant_integral(c) cval_isinteger((c)->cval)
92     /* Returns: TRUE if c is an integer constant (signed or unsigned)
93     */
94     
95     /*bool constant_float(known_cst c);*/
96     #define constant_float(c) (cval_isfloating((c)->cval))
97     /* Returns: TRUE if c is a floating-point constant
98     */
99     
100     /*largest_uint constant_uint_value(known_cst c);*/
101     #define constant_uint_value(c) cval_uint_value((c)->cval)
102     /* Returns: Value of c as an unsigned integer.
103        Requires: see cval_uint_value
104     */
105     
106     /*largest_int constant_sint_value(known_cst c);*/
107     #define constant_sint_value(c) cval_sint_value((c)->cval)
108     /* Returns: Value of c as an unsigned integer.
109        Requires: see cval_sint_value
110     */
111     
112     /*long double constant_float_value(known_cst c);*/
113     #define constant_float_value(c) cval_float_value((c)->cval)
114     /* Returns: Value of c as an unsigned integer.
115        Requires: see cval_float_value
116     */
117     
118     /*bool constant_knownbool(known_cst c);*/
119     /* Returns: TRUE if the truth-value of c can be determined (use 
120        constant_boolvalue to get that value)
121     */
122     #define constant_knownbool(c) cval_knownbool((c)->cval)
123     
124     /*bool cval_boolvalue(cval c);*/
125     /* Returns: TRUE if c is a non-zero constant
126        Requires: cval_knownbool(c)
127      */
128     #define constant_boolvalue(c) cval_boolvalue((c)->cval)
