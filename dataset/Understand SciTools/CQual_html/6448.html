19     #ifndef ENV_H
20     #define ENV_H
21     
22     #include "linkage.h"
23     #include "env.h"
24     #include "hash.h"
25     
26     EXTERN_C_BEGIN
27     
28     typedef struct env *env;
29     
30     /* Create a new, empty environment with ancestor 'parent'.
31        The environment, and any future contents, are allocated in r */
32     env new_env(region r, env parent);
33     
34     /* An environment can be deleted by deleting its region */
35     
36     /* Return parent environment of e */
37     env env_parent(env e);
38     
39     /* Return region of e */
40     region env_region(env e);
41     
42     /* Find entry s in in environment e. If not found, check ancestors
43        except if this_level_only is true.
44        Returns entry's value if s is found, NULL otherwise */
45     void *env_lookup(env e, const char *s, bool this_level_only);
46     
47     /* Add an entry for s, with value 'value' to environment e.
48        If e already contains an entry for s, then the old entry is
49        hidden (but will still be found by env_scan).
50        s can be NULL (this allows registering of untagged structs/etc
51        in environments). Such entries cannot be found by env_lookup.
52        Does not copy s. */
53     void env_add(env e, const char *s, void *value);
54     
55     /* Scanning */
56     typedef hash_table_scanner env_scanner;
57     void env_scan(env e, env_scanner *scanner);
58     bool env_next(env_scanner *scanner, const char **name, void **value);
59     
60     EXTERN_C_END
61     
62     #endif
