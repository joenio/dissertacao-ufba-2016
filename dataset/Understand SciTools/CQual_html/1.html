19     #ifndef BITSET_H
20     #define BITSET_H
21     
22     #include <stdlib.h>
23     #include "linkage.h"
24     #include "bool.h"
25     #include "regions.h"
26     
27     EXTERN_C_BEGIN
28     
29     typedef unsigned int *bitset; /* An array of ints.  First int is number
30     				 of bits. */
31     
32     /* Return the number of unsigned int's to allocate for a bitset of
33        size nbits. */
34     unsigned int sizeof_bitset(unsigned int nbits);
35     
36     /* Allocate a bit set in region r. */
37     bitset bitset_new(region r, unsigned int nbits);
38     
39     #define alloca_bitset(nbits) ((unsigned int *) alloca(sizeof_bitset(nbits)*sizeof(unsigned int)))
40     
41     /* Make a copy of b */
42     bitset bitset_copy(region r, bitset b);
43     
44     /* Replace bitset b1 by b2.  b1 and b2 must be the same size. */
45     void bitset_assign(bitset b1, bitset b2);
46     
47     /* Return TRUE iff bitset b is empty */
48     bool bitset_empty(bitset b);
49     
50     /* Return TRUE iff elements of b from first to last, inclusive, are
51        all absent */
52     bool bitset_empty_range(bitset b, unsigned int first, unsigned int last);
53     
54     /* Return TRUE iff elements of b from first to last, inclusive, are
55        all present */
56     bool bitset_full_range(bitset b, unsigned int first, unsigned int last);
57     
58     /* Add all possible elements to b. */
59     void bitset_insert_all(bitset b);
60     
61     /* Add elt to bitset b.  Returns TRUE if elt was not already in the set. */
62     bool bitset_insert(bitset b, unsigned int elt);
63     
64     /* Remove elt from bitset b.  Returns TRUE if elt was in the set. */
65     bool bitset_remove(bitset b, unsigned int elt);
66     
67     /* Return TRUE iff elt is in b */
68     bool bitset_member(bitset b, unsigned int elt);
69     
70     /* Replace b1 by b1 & b2.  Destructively updates b1.  b2 is unchanged.
71        Returns TRUE iff b1 was changed. */
72     bool bitset_intersect(bitset b1, const bitset b2);
73     
74     /* Print b to stdout */
75     void bitset_print(bitset b);
76     
77     /* Return a hash for b */
78     int bitset_hash(bitset b);
79     
80     /* Return TRUE iff b1 and b2 are equal */
81     bool bitset_eq(bitset b1, bitset b2);
82     
83     EXTERN_C_END
84     
85     #endif
