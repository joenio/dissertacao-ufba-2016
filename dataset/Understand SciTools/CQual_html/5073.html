40     known_cst make_unknown_cst(type t)
41     {
42       known_cst c = new_known_cst(parse_region, dummy_location, cval_unknown);
43       c->type = t;
44       return c;

47     known_cst make_cst(cval c, type t)
48     {
49       known_cst kc;
50     
51       if (cval_istop(c))
52         return NULL;
53     
54       kc = new_known_cst(parse_region, dummy_location, cval_cast(c, t));
55       kc->type = t;
56       return kc;
57     }

75     static known_cst cast_constant(known_cst c, type to)
76     {
77       if (type_equal(c->type, to))
78         return c;
79     
80       return make_cst(cval_cast(c->cval, to), to);
81     }

115     /* +, -, ! and ~ only */
116     known_cst fold_unary(expression e)
117     {
118       unary u = CAST(unary, e);
119       known_cst arg = u->arg1->cst;
120       type t = u->type;
121     
122       if (arg)
123         {
124           switch (u->kind)
125     	{
126     	case kind_unary_plus:
127     	  /* Note that this allows +(int)&x to be a constant... 
128     	     This is consistent with gcc */
129     	  return cast_constant(arg, t); /* essentially a no-op */
130     
131     	case kind_unary_minus:
132     	  return make_cst(cval_negate(cval_cast(arg->cval, t)), t);
133     
134     	case kind_not:
135     	  return make_cst(cval_not(cval_cast(arg->cval, t)), t);
136     
137     	case kind_bitnot:
138     	  return make_cst(cval_bitnot(cval_cast(arg->cval, t)), t);
139     
140     	case kind_conjugate:
141     	  return make_cst(cval_conjugate(cval_cast(arg->cval, t)), t);
142     
143     	case kind_realpart:
144     	  if (!type_complex(u->arg1->type))
145     	    return cast_constant(arg, t);
146     	  else
147     	    return make_cst(cval_realpart(cval_cast(arg->cval, t)), t);
148     
149     	case kind_imagpart:
150     	  if (!type_complex(u->arg1->type))
151     	    return make_cst(cval_cast(cval_zero, t), t);
152     	  else
153     	    return make_cst(cval_imagpart(cval_cast(arg->cval, t)), t);
154     
155     	default:
156     	  assert(0);
157     	  break;
158     	}
159         }
160       return NULL;
161     }

164     static known_cst fold_sub(type restype, known_cst c1, known_cst c2)
165     {
166       type t1 = c1->type, t2 = c2->type, basetype, ct;
167       cval s = make_cval_unsigned(1, size_t_type), res;
168     
169       basetype = type_pointer(t1) ? t1 : type_pointer(t2) ? t2 : NULL;
170       if (basetype)
171         {
172           basetype = type_points_to(basetype);
173     
174           if (!type_size_cc(basetype))
175     	return NULL;
176     
177           s = make_cval_unsigned(type_size(basetype), size_t_type);
178           ct = intptr_type;
179         }
180       else
181         ct = restype;
182     
183       s = cval_cast(s, ct);
184       if (type_pointer(t1) && type_pointer(t2))
185         res = cval_divide(cval_sub(cval_cast(c1->cval, ct), cval_cast(c2->cval, ct)),
186     		      s);
187       else
188         res = cval_sub(cval_cast(c1->cval, ct),
189     		   cval_divide(cval_cast(c2->cval, ct), s));
190     
191       return make_cst(res, restype);
192     }

194     /* XXX: overflow */
195     known_cst fold_add(type restype, known_cst c1, known_cst c2)
196     {
197       type t1 = c1->type, t2 = c2->type, ct;
198       cval s = make_cval_unsigned(1, size_t_type);
199     
200       if (type_pointer(t2))
201         {
202           known_cst ctmp;
203           type ttmp;
204     
205           ctmp = c1; c1 = c2; c2 = ctmp;
206           ttmp = t1; t1 = t2; t2 = ttmp;
207         }
208     
209       if (type_pointer(t1))
210         {
211           type basetype = type_points_to(t1);
212     
213           if (!type_size_cc(basetype))
214     	return NULL;
215     
216           s = make_cval_unsigned(type_size(basetype), size_t_type);
217           ct = intptr_type;
218         }
219       else
220         ct = restype;
221     
222       return make_cst(cval_add(cval_cast(c1->cval, ct),
223     			   cval_times(cval_cast(c2->cval, ct), cval_cast(s, ct))),
224     		  restype);
225     }

227     known_cst fold_binary(type t, expression e)
228     {
229       binary b = CAST(binary, e);
230       known_cst c1 = b->arg1->cst, c2 = b->arg2->cst;
231       type t1 = type_default_conversion(b->arg1->type),
232         t2 = type_default_conversion(b->arg2->type);
233       
234       if (b->kind == kind_andand || b->kind == kind_oror)
235         {
236           bool c1val;
237     
238           if (c1 && constant_knownbool(c1) && 
239     	  (c1val = constant_boolvalue(c1), b->kind == kind_andand ? !c1val : c1val))
240     	return make_signed_cst(FALSE, t);
241           if (c1 && c2)
242     	{
243     	  if (constant_knownbool(c2))
244     	    return make_signed_cst(constant_boolvalue(c2), t);
245     	  else
246     	    return make_unknown_cst(t);
247     	}
248         }
249       else if (c1 && c2)
250         {
251           cval cv1 = c1->cval, cv2 = c2->cval;
252     
253           switch (b->kind)
254     	{
255     	case kind_plus: case kind_array_ref:
256     	  return fold_add(t, c1, c2);
257     
258     	case kind_minus:
259     	  return fold_sub(t, c1, c2);
260     
261     	case kind_times: case kind_divide: case kind_modulo:
262     	case kind_lshift: case kind_rshift:
263     	case kind_bitand: case kind_bitor: case kind_bitxor: {
264     	  cval res;
265     
266     	  cv1 = cval_cast(cv1, t);
267     	  cv2 = cval_cast(cv2, t);
268     
269     	  switch (b->kind) {
270     	  case kind_times: res = cval_times(cv1, cv2); break;
271     	  case kind_divide: res = cval_divide(cv1, cv2); break;
272     	  case kind_modulo: res = cval_modulo(cv1, cv2); break;
273     	  case kind_lshift: res = cval_lshift(cv1, cv2); break;
274     	  case kind_rshift: res = cval_rshift(cv1, cv2); break;
275     	  case kind_bitand: res = cval_bitand(cv1, cv2); break;
276     	  case kind_bitor: res = cval_bitor(cv1, cv2); break;
277     	  case kind_bitxor: res = cval_bitxor(cv1, cv2); break;
278     	  default: abort(); return NULL;
279     	  }
280     	  return make_cst(res, t);
281     	}
282     	case kind_eq: case kind_ne:
283     	case kind_leq: case kind_geq: case kind_lt: case kind_gt: {
284     	  cval res;
285     	  type ct;
286     
287     	  /* Pointers win. */
288     	  if (type_pointer(t1) || type_pointer(t2))
289     	    ct = intptr_type; 
290     	  else
291     	    ct = common_type(t1, t2);
292     
293     	  cv1 = cval_cast(cv1, ct);
294     	  cv2 = cval_cast(cv2, ct);
295     
296     	  switch (b->kind) {
297     	  case kind_eq: res = cval_eq(cv1, cv2); break;
298     	  case kind_ne: res = cval_ne(cv1, cv2); break;
299     	  case kind_leq: res = cval_leq(cv1, cv2); break;
300     	  case kind_geq: res = cval_geq(cv1, cv2); break;
301     	  case kind_lt: res = cval_lt(cv1, cv2); break;
302     	  case kind_gt: res = cval_gt(cv1, cv2); break;
303     	  default: abort(); return NULL;
304     	  }
305     	  return make_cst(res, t);
306     	}
307     	default:
308     	  assert(0); return NULL;
309     	}
310           }
311     
312       return NULL;
313     }

315     known_cst fold_conditional(expression e)
316     {
317       conditional c = CAST(conditional, e);
318       known_cst cond = c->condition->cst;
319     
320       if (cond && constant_knownbool(cond))
321         {
322           expression value =
323     	constant_boolvalue(cond) ? (c->arg1 ? c->arg1 : c->condition) : c->arg2;
324     
325           e->static_address = value->static_address;
326           if (value->cst)
327     	return cast_constant(value->cst, e->type);
328           else
329     	return NULL;
330         }
331       return NULL;
332     }

357     known_cst foldaddress_field_ref(known_cst object, field_declaration fdecl)
358     {
359       if (!object || !fdecl->offset_cc || fdecl->bitwidth >= 0)
360         return NULL;
361     
362       if (constant_unknown(object))
363         return make_unknown_cst(object->type);
364     
365       return fold_add(object->type, object,
366     		  make_unsigned_cst(fdecl->offset / BITSPERBYTE, size_t_type));
367     }

369     lexical_cst fold_lexical_real(type realtype, location loc, cstring tok)
370     {
371       lexical_cst c = new_lexical_cst(parse_region, loc, tok);
372       cval realvalue = make_cval_float(strtold(tok.data, NULL));
373     
374       if (type_complex(realtype))
375         realvalue = make_cval_complex(cval_cast(cval_zero, realtype), realvalue);
376       c->cst = make_cst(realvalue, realtype);
377       c->type = realtype;
378       return c;
379     }

381     lexical_cst fold_lexical_char(location loc, cstring tok,
382     			      bool wide_flag, int charvalue)
383     {
384       lexical_cst c = new_lexical_cst(parse_region, loc, tok);
385       type ctype = wide_flag ? wchar_type : int_type;
386     
387       c->type = ctype;
388       c->cst = make_cst(type_unsigned(ctype) ?
389     		    make_cval_unsigned(charvalue, ctype) :
390     		    make_cval_signed(charvalue, ctype),
391     		    ctype);
392       return c;
393     }

395     string_cst fold_lexical_string(location loc, cstring tok,
396     			       bool wide_flag, wchar_array stringvalue)
397     {
398       size_t length = wchar_array_length(stringvalue);
399       wchar_t *chars = rarrayalloc(parse_region, length, wchar_t);
400       string_cst c = new_string_cst(parse_region, loc, tok, chars, length);
401     
402       c->type = wide_flag ? wchar_array_type : char_array_type;
403       memcpy(chars, wchar_array_data(stringvalue), length * sizeof(wchar_t));
404     
405       /* We don't set c->cst as a C string constant is a sequence of string_cst,
406          not a single one. See make_string. */
407       return c;
408     }

410     lexical_cst fold_lexical_int(type itype, location loc, cstring tok,
411     			     bool iscomplex, largest_uint intvalue, bool overflow)
412     {
413       lexical_cst c = new_lexical_cst(parse_region, loc, tok);
414       cval cv;
415     
416       if (overflow)
417         {
418           warning_with_location(loc, "integer constant out of range");
419           itype = unsigned_long_long_type;
420         }
421       else
422         {
423           /* Do some range checks */
424           if (!uint_inrange(intvalue, itype))
425     	{
426     	  if (uint_inrange(intvalue, unsigned_int_type))
427     	    {
428     	      warning_with_location(loc, "decimal constant is so large that it is unsigned");
429     	      itype = unsigned_int_type;
430     	    }
431     	  /* These other cases cause no warnings */
432     	  else if (uint_inrange(intvalue, long_type))
433     	    itype = long_type;
434     	  else if (uint_inrange(intvalue, unsigned_long_type))
435     	    itype = unsigned_long_type;
436     	  else if (uint_inrange(intvalue, long_long_type))
437     	    itype = long_long_type;
438     	  else if (uint_inrange(intvalue, unsigned_long_long_type))
439     	    itype = unsigned_long_long_type;
440     	}
441         }
442     
443       cv = type_unsigned(itype) ? make_cval_unsigned(intvalue, itype) :
444         make_cval_signed(intvalue, itype);
445       if (iscomplex)
446         {
447           cv = make_cval_complex(cval_cast(cval_zero, itype), cv);
448           itype = make_complex_type(itype);
449         }
450     
451       c->type = itype;
452       c->cst = make_cst(cv, itype);
453     
454       return c;
455     }

458     bool definite_null(expression e)
459     {
460       return
461         (type_integer(e->type) ||
462          (type_pointer(e->type) && type_void(type_points_to(e->type)))) &&
463         definite_zero(e);
464     }

466     bool definite_zero(expression e)
467     {
468       return e->cst && is_zero_constant(e->cst);
469     }

471     bool definite_one(expression e)
472     {
473       return e->cst && is_one_constant(e->cst);

476     bool is_one_constant(known_cst c)
477     {
478       return cval_knownbool(c->cval) && constant_boolvalue(c);

481     bool is_zero_constant(known_cst c)
482     {
483       return cval_knownbool(c->cval) && !constant_boolvalue(c);
