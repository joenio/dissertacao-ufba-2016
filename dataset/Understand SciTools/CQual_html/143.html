19     #ifndef HASH_H
20     #define HASH_H
21     
22     #include "bool.h"
23     #include "linkage.h"
24     
25     EXTERN_C_BEGIN
26     
27     typedef void *hash_key;
28     typedef void *hash_data;
29     
30     typedef int (*hash_fn)(hash_key k); /* Function to hash a key */
31     typedef bool (*keyeq_fn)(hash_key k1, hash_key k2);
32                                              /* Function returning true iff
33     					     k1 and k2 are equal */
34     typedef void (*hash_apply_fn)(hash_key k, hash_data d, void *arg);
35                                               /* Function applied to
36     					     elts in the hash table */
37     typedef hash_data (*hash_map_fn)(hash_key k, hash_data d, void *arg);
38                                               /* Function mapped to
39     					     elts in the hash table */
40     
41     typedef struct Hash_table *hash_table;
42     
43     /* Make a new hash table, with size buckets initially.  Hash table
44        elements are allocated in region rhash. */
45     hash_table make_hash_table(region rhash, int size, hash_fn hash,
46     			   keyeq_fn cmp);
47     
48     /* Make a hash table for strings. */
49     hash_table make_string_hash_table(region rhash, int size);
50     
51     /* Zero out ht.  Doesn't reclaim bucket space. */
52     void hash_table_reset(hash_table ht);
53     
54     /* Return the number of entries in ht */
55     int hash_table_size(hash_table ht);
56     
57     /* Lookup k in ht.  If d is not NULL, returns corresponding data in *d.
58        Function result is TRUE if the k was in ht, false otherwise. */
59     bool hash_table_lookup(hash_table ht, hash_key k, hash_data *d);
60     
61     /* Add k:d to ht.  If k was already in ht, replace old entry by k:d.
62        Rehash if necessary.  Returns TRUE if k was not already in ht. */
63     bool hash_table_insert(hash_table ht, hash_key k, hash_data d);
64     
65     /* Remove mapping for k in ht.  Returns TRUE if k was in ht. */
66     bool hash_table_remove(hash_table ht, hash_key k);
67     
68     /* Return a copy of ht, allocated in rhash */
69     hash_table hash_table_copy(region rhash, hash_table ht);
70     
71     /* Map f to all elements on ht, creating a new hash table */
72     hash_table hash_table_map(region r, hash_table ht, hash_map_fn f, void *arg);
73     
74     typedef struct bucket *bucket;
75     typedef struct
76     {
77       hash_table ht;
78       int i;
79       bucket cur;
80     } hash_table_scanner; /* Opaque type!  Do not modify fields. */
81     
82     /* Begin scanning ht */
83     void hash_table_scan(hash_table ht, hash_table_scanner *);
84     
85     /* Get next elt in table, storing the elt in *k and *d if k and d are
86        non-NULL, respectively.  Returns TRUE if there is a next elt, FALSE
87        otherwise. */
88     bool hash_table_next(hash_table_scanner *, hash_key *k, hash_data *d);
89     
90     /* Total order on hash table keys, only uesd for hash_table_scan_sorted */
91     typedef int (*keycmp_fn)(hash_key k1, hash_key k2);
92     
93     struct sorted_entry
94     {
95       hash_key k;
96       hash_data d;
97     };
98     
99     typedef struct
100     {
101       region r;
102       int i;
103       int size;
104       struct sorted_entry *entries;
105     } hash_table_scanner_sorted;
106     
107     /* Begin scanning ht in sorted order according to f */
108     void hash_table_scan_sorted(hash_table ht, keycmp_fn f,
109     			    hash_table_scanner_sorted *htss);
110     
111     /* Just like hash_table_next, but scans in sorted order */
112     bool hash_table_next_sorted(hash_table_scanner_sorted *htss, hash_key *k,
113     			    hash_data *d);
114     
115     
116     EXTERN_C_END
117     
118     #endif
