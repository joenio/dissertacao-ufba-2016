19     #ifndef ARRAY_H
20     #define ARRAY_H
21     
22     /* A region-based growable array type */
23     
24     struct array;
25     
26     struct array *new_array(region r, size_t initialsize,
27     			size_t typesize, type_t typeinfo);
28     void *array_extend(struct array *a, int by);
29     void array_reset(struct array *a);
30     size_t array_length(struct array *a);
31     void *array_data(struct array *a);
32     
33     
34     #define DECLARE_ARRAY(name, type) \
35     typedef struct name ## _a *name; \
36     name new_ ## name(region r, size_t initialsize); \
37     type *name ## _extend(name a, int by); \
38     void name ## _reset(name a); \
39     size_t name ## _length(name a); \
40     type *name ## _data(name a);
41     
42     #define DEFINE_ARRAY(name, type) \
43     name new_ ## name(region r, size_t initialsize) \
44     { \
45       return (name)new_array(r, initialsize, sizeof(type), rctypeof(type)); \
46     } \
47     type *name ## _extend(name a, int by) \
48     { \
49       return array_extend((struct array *)a, by); \
50     } \
51     void name ## _reset(name a) \
52     { \
53       return array_reset((struct array *)a); \
54     } \
55     size_t name ## _length(name a) \
56     { \
57       return array_length((struct array *)a); \
58     } \
59     type *name ## _data(name a) \
60     { \
61       return array_data((struct array *)a); \
62     }
