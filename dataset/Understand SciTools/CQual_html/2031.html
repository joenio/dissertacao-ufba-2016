19     #ifndef CVAL_H
20     #define CVAL_H
21     
22     /* Note: legal ops on complex are:
23        add, sub, times, divide, realpart, imagpart, conjugate
24     */
25     typedef struct {
26       enum { cval_variable, cval_unk, cval_address,
27     	 cval_float, cval_float_complex,
28     	 cval_uint, cval_uint_complex,
29     	 cval_sint, cval_sint_complex } kind;
30       /* Could be a union */
31       long double d, d_i; /* for cval_float */
32       struct data_declaration *ddecl; /* for cval_address */
33       struct label_declaration *ldecl;  /* for cval_address */
34       largest_int si, si_i;
35       largest_uint ui, ui_i;
36       size_t isize;
37     } cval;
38     
39     extern cval cval_top; /* The non-constant value */
40     extern cval cval_unknown; /* The unknown value */
41     extern cval cval_zero; /* A zero value. Use cval_cast to make the desired 
42     			  kind of constant */
43     
44     void cval_init(void);
45     
46     cval make_cval_signed(largest_int i, type t);
47     cval make_cval_unsigned(largest_uint i, type t);
48     cval make_cval_float(long double d);
49     cval make_cval_complex(cval r, cval i);
50     cval make_cval_address(data_declaration ddecl, label_declaration ldecl,
51     		       largest_int offset);
52     
53     /*bool cval_isunknown(cval c);*/
54     #define cval_isunknown(c) ((c).kind == cval_unk)
55     /* Return: TRUE if c is an unknown constant */
56     
57     /*bool cval_istop(cval c);*/
58     #define cval_istop(c) ((c).kind == cval_variable)
59     /* Return: TRUE if c is not a constant */
60     
61     /*bool cval_isaddress(cval c);*/
62     #define cval_isaddress(c) ((c).kind == cval_address)
63     /* Return: TRUE if c is not a constant */
64     
65     bool cval_isinteger(cval c);
66     /* Return: TRUE if c is an integer constant */
67     
68     bool cval_iscomplex(cval c);
69     /* Return: TRUE if c is a complex constant */
70     
71     bool cval_knownbool(cval c);
72     /* Returns: TRUE if the truth-value of c can be determined (use cval_boolvalue
73        to get that value)
74     */
75     
76     bool cval_boolvalue(cval c);
77     /* Returns: TRUE if c is a non-zero constant
78        Requires: cval_knownbool(c)
79      */
80     
81     bool cval_knownvalue(cval c);
82     /* Returns: TRUE if the value of c can be determined (this is false for
83        address constants, while cval_knownbool is true for address constants
84        with offset 0)
85     */
86     
87     largest_uint cval_uint_value(cval c);
88     /* Returns: The value of c as an unsigned int
89        Requires: cval_knownvalue(c) && !cval_iscomplex(c)
90     */
91     
92     largest_int cval_sint_value(cval c);
93     /* Returns: The value of c as a signed int
94        Requires: cval_knownvalue(c) && !cval_iscomplex(c)
95     */
96     
97     long double cval_float_value(cval c);
98     /* Returns: The value of c as a long double
99        Requires: cval_knownvalue(c) && !cval_iscomplex(c)
100     */
101     
102     bool cval_isone(cval c);
103     /* Returns: TRUE if c is 1 (FALSE for unknown constants)
104        Requires: c not be cval_top
105      */
106     
107     /* All of these functions will return cval_top if the result is not a
108        constant expression */
109     
110     cval cval_cast(cval c, type to); /* Cast c to type to */
111     cval cval_not(cval c); /* !c */
112     cval cval_negate(cval c); /* -c */
113     cval cval_bitnot(cval c); /* ~c */
114     cval cval_conjugate(cval c); /* ~c */
115     cval cval_realpart(cval c); /* __real__ c */
116     cval cval_imagpart(cval c); /* __imag__ c */
117     
118     /* The binary operators require that both arguments have been cast to a common
119        type. */
120     cval cval_add(cval c1, cval c2);
121     cval cval_sub(cval c1, cval c2);
122     cval cval_times(cval c1, cval c2);
123     cval cval_divide(cval c1, cval c2);
124     cval cval_modulo(cval c1, cval c2);
125     cval cval_lshift(cval c1, cval c2);
126     cval cval_rshift(cval c1, cval c2);
127     cval cval_bitand(cval c1, cval c2);
128     cval cval_bitor(cval c1, cval c2);
129     cval cval_bitxor(cval c1, cval c2);
130     cval cval_eq(cval c1, cval c2);
131     /*cval cval_ne(cval c1, cval c2);*/
132     cval cval_leq(cval c1, cval c2);
133     /*cval cval_lt(cval c1, cval c2);*/
134     /*cval cval_geq(cval c1, cval c2);*/
135     /*cval cval_gt(cval c1, cval c2);*/
136     #define cval_gt(c1, c2) (cval_not(cval_leq((c1), (c2))))
137     #define cval_lt(c1, c2) (cval_gt((c2), (c1)))
138     #define cval_geq(c1, c2) (cval_leq((c2), (c1)))
139     #define cval_ne(c1, c2) (cval_not(cval_eq((c1), (c2))))
140     
141     /* True if x fits in the range of type t */
142     bool uint_inrange(largest_uint x, type t);
143     bool sint_inrange(largest_int x, type t);
144     
145     bool cval_inrange(cval c, type t);
146     /* Requires: constant_integral(c)
147        Returns: TRUE if c is in range of type t. */
148     
149     largest_int cval_intcompare(cval c1, cval c2);
150     /* Requires: cval_isinteger(c1) && cval_isinteger(c2)
151        Returns: x, x<0 if c1 < c2, x = 0 if c1 = c2 and x > 0 if c1 > c2
152     */
153     
154     #endif
