19     #ifndef DD_LIST_H
20     #define DD_LIST_H
21     
22     /* 
23      * Doubly-linked list storing arbitrary pointers.
24      * The lists may homogeneous or heterogeneous at the discretion of
25      * the user (the list stores no type indications)
26      *
27      * dd_list is the type of all lists
28      * dd_list_pos is the type that represents a particular element of a list
29      */
30     
31     #include "linkage.h"
32     
33     EXTERN_C_BEGIN
34     
35     typedef struct dd_list *dd_list; /* A list */
36     typedef struct dd_list_pos	 /* A position in a list */
37     {
38       /* PRIVATE! Do not use the fields directly */
39       struct dd_list_pos *next;
40       struct dd_list_pos *previous;
41       void *data;
42     } *dd_list_pos;
43     
44     dd_list dd_new_list(region r);
45     /* Returns: A new empty list
46     */
47     
48     void dd_add_first(region r, dd_list l, void *data);
49     /* Effects: Adds a new element containing `data' to the beginning of l.
50        Modifies: l
51     */
52     void dd_add_last(region r, dd_list l, void *data);
53     /* Effects: Adds a new element containing `data' to the end of l.
54        Modifies: l
55     */
56     void dd_insert_before(region r, dd_list_pos where, void *data);
57     /* Effects: Adds a new element containg `data' after element `where'.
58        Modifies: the list containing `where'
59     */
60     void dd_insert_after(region r, dd_list_pos where, void *data);
61     /* Effects: Adds a new element containg `data' before element `where'.
62        Modifies: the list containing `where'
63     */
64     void dd_remove(dd_list_pos what);
65     /* Effects: Removes element `what' from its list.
66          No operations on what are valid after the call to dd_remove.
67        Modifies: the list containing `what'.
68     */
69     
70     dd_list_pos dd_first(dd_list l);
71     /* Returns: The first element of list l
72     */
73     dd_list_pos dd_last(dd_list l);
74     /* Returns: The last element of list l
75     */
76     #define dd_is_beginning(l) (!(l)->previous)
77     /* Returns: TRUE if l is the pseudo-element at the beginning of a list.
78        Note: dd_is_beginning(dd_previous(dd_first(l))) == TRUE
79     */
80     #define dd_is_end(l) (!(l)->next)
81     /* Returns: TRUE if l is the pseudo-element at the end of a list.
82        Note: dd_is_end(dd_next(dd_last(l))) == TRUE
83     */
84     #define dd_next(l) ((l)->next)
85     /* Returns: The element after l, or the pseudo-element indicating the
86          end of the list if none remain (see dd_is_end)
87     */
88     #define dd_previous(l) ((l)->previous)
89     /* Returns: The element before l, or the pseudo-element indicating the
90          beginning of the list if none remain (see dd_is_beginning)
91     */
92     
93     #define dd_is_empty(l) (dd_is_end(dd_first((l))))
94     /* Returns: TRUE if l is the empty list
95     */
96     
97     #define DD_GET(type, l) ((type)((l)->data))
98     /* Returns: The contents of element l, cast to type `type'
99          This operation is obviously invalid on the pseudo-elements at
100          the beginning and end of lists
101     */
102     #define DD_SET(l, to) ((l)->data = (to))
103     /* Effects: Sets the contents of element l to `to'
104          This operation is obviously invalid on the pseudo-elements at
105          the beginning and end of lists
106        Modifies: l
107     */
108     #define dd_scan(var, list) for (var = dd_first((list)); !dd_is_end(var); var = dd_next(var))
109     /* Effects: Iterates variable `scan' over the contents of the list.
110        Requires: list not be modified during iteration.
111        Example:
112          length = 0;
113          dd_scan (element, l)
114            length++;
115     */
116     
117     #define dd_scan_reverse(var, list) for (var = dd_last((list)); !dd_is_beginning(var); var = dd_previous(var))
118     /* Effects: Iterates variable `scan' over the contents of the list.
119        Requires: list not be modified during iteration.
120        Example:
121          length = 0;
122          dd_scan (element, l)
123            length++;
124     */
125     
126     unsigned long dd_length(dd_list l);
127     /* Returns: length of list l
128     */
129     
130     void dd_append(dd_list l1, dd_list l2);
131     /* Effects: Appends list l2 to the end of list l1.
132          List l2 is destroyed
133        Modifies: l1, l2
134     */
135     
136     dd_list dd_copy(region r, dd_list l);
137     /* Returns: A new list with the same elements as l
138     */
139     
140     void dd_free_list(dd_list l, void (*delete)(dd_list_pos p));
141     /* Effects: Destroys list l and all its elements.
142          If delete is not NULL, calls it before deleting each element.
143        Modifies: l
144     */
145     
146     dd_list_pos dd_find(dd_list l, void *find);
147     /* Returns: The element of l whose data is 'find', or NULL if it isn't found
148     */
149     
150     EXTERN_C_END
151     
152     #endif
