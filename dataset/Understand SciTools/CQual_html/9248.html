19     #include <assert.h>
20     #include <sys/types.h>
21     #include <unistd.h>
22     #include <stdio.h>
23     #include <stdlib.h>
24     #include <string.h>
25     #include "usage.h"
26     
27     #define PROC_LINE_LENGTH 4096
28     
29     /* Return the memory used by the current process, in bytes */
30     long get_memusage(void)
31     {
32       long vmsize;
33       pid_t pid;
34       char filename[64];
35       char buf[PROC_LINE_LENGTH];
36       FILE *stat_file;
37       int i, field;
38     
39       vmsize = 0;
40       pid = getpid();
41       snprintf(filename, 64, "/proc/%d/stat", pid);
42       filename[63] = '\0';
43       stat_file = fopen(filename, "r");
44       if (!stat_file)
45         {
46           perror(filename);
47           exit(EXIT_FAILURE);
48         }
49     
50       /* Read in /proc/<pid>/stat */
51       if (!fread(buf, sizeof(char), PROC_LINE_LENGTH, stat_file))
52         {
53           perror(filename);
54           exit(EXIT_FAILURE);
55         }
56     
57       /* Scan over fields; assume separated by spaces */
58       field = 0;
59       i = 0;
60       while (buf[i])
61         {
62           if (field == 22)
63     	{
64     	  /* memory size, in bytes */
65     	  for (; '0' <= buf[i] && buf[i] <= '9'; i++)
66     	    vmsize = 10*vmsize + buf[i] - '0';
67     
68     	}
69     
70           /* Skip ahead to next field */
71           while (buf[i] && buf[i] != ' ')
72     	i++;
73           i++; /* Skip over space */
74           field++;
75         }
76     
77       fclose(stat_file);
78       return vmsize;
79     }
