19     /* Region-based replacement for alloca */
20     #include <stdlib.h>
21     #include "regions.h"
22     
23     region alloca_region = NULL;
24     void *alloca(size_t t)
25     {
26       if (!alloca_region && t)
27         alloca_region = newregion();
28       else if (alloca_region && !t)
29         {
30           deleteregion(alloca_region);
31           alloc_region = newregion();
32         }
33     
34       if (t)
35         return typed_ralloc(alloca_region, t, 0);
36       else
37         return NULL;
38     }
