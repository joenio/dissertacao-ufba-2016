34     #ifndef REGIONS_H
35     #define REGIONS_H
36     
37     #include "../linkage.h"
38     
39     EXTERN_C_BEGIN
40     
41     #define RPAGELOG 13
42     
43     typedef struct Region *region;
44     
45     #include <stdlib.h>
46     
47     void region_init(void); /* You MUST call this function before using any
48     			   part of this library */
49     
50     region newregion(void);
51     region newsubregion(region parent);
52     
53     typedef int type_t;
54     #define rctypeof(type) 0
55     
56     /* Low-level alloc with dynamic type info */
57     void *typed_ralloc(region r, size_t size, type_t type);
58     void *typed_rarrayalloc(region r, size_t n, size_t size, type_t type);
59     void *typed_rarrayextend(region r, void *old, size_t n, size_t size, type_t type);
60     void typed_rarraycopy(void *to, void *from, size_t n, size_t size, type_t type);
61     
62     #define ralloc(r, type) typed_ralloc((r), sizeof(type), rctypeof(type))
63     #define rarrayalloc(r, n, type) typed_rarrayalloc((r), (n), sizeof(type), rctypeof(type))
64     #define rarrayextend(r, old, n, type) typed_rarrayextend((r), (old), (n), sizeof(type), rctypeof(type))
65     #define rarraycopy(to, from, n, type) typed_rarraycopy((to), (from), (n), sizeof(type), rctypeof(type))
66     
67     char *rstralloc(region r, size_t size);
68     char *rstralloc0(region r, size_t size);
69     char *rstrdup(region r, const char *s);
70     
71     /* rstrextend is used to extend an old string. The string MUST have been
72        initially allocated by a call to rstrextend with old == NULL (you cannot
73        initially allocate the string with rstralloc) */
74     char *rstrextend(region r, const char *old, size_t newsize);
75     char *rstrextend0(region r, const char *old, size_t newsize);
76     
77     void deleteregion(region r);
78     void deleteregion_ptr(region *r);
79     void deleteregion_array(int n, region *regions);
80     region regionof(void *ptr);
81     
82     typedef void (*nomem_handler)(void);
83     nomem_handler set_nomem_handler(nomem_handler newhandler);
84     
85     /* Debugging support */
86     void findrefs(region r, void *from, void *to);
87     void findgrefs(region r);
88     void findrrefs(region r, region from);
89     
90     #ifdef REGION_PROFILE
91     #include "profile.h"
92     #endif
93     
94     EXTERN_C_END
95     
96     #endif
