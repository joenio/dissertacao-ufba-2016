19     #include <assert.h>
20     #include <stdio.h>
21     #include "qerror.h"
22     
23     static region qerror_region = NULL;
24     
25     /* Set to TRUE if any qualifier errors happened (unsatisfiable
26        constraints) */
27     bool qerrors = FALSE;
28     
29     /* Detailed information about the last invalid constraint */
30     growbuf qual_error_detail = NULL; 
31     growbuf qtype_error_detail = NULL; 
32     
33     void init_qerror(void)
34     {
35       assert(!qerror_region);
36       qerror_region = newregion();
37       qual_error_detail = growbuf_new(qerror_region, 256);
38       qtype_error_detail = growbuf_new(qerror_region, 256);
39       assert(!qerrors);
40     }
41     
42     /* printf into qual_error_detail */
43     int qual_ed_printf(const char *fmt, ...)
44     {
45       va_list args;
46       va_start(args, fmt);
47       return gvprintf(qual_error_detail, fmt, args);
48     }
49     
50     /* printf into qtype_error_detail */
51     int qtype_ed_printf(const char *fmt, ...)
52     {
53       va_list args;
54       va_start(args, fmt);
55       return gvprintf(qtype_error_detail, fmt, args);
56     }
