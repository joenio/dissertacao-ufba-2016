114     void finish_flow_sensitive(void)
115     {
116       data_declaration main_ddecl;
117     
118       mkleq_store(NULL, "inconsistent flow-sensitive qualifiers",
119     	      global_store_ext, global_store);
120       main_ddecl = get_main_ddecl(); /* does root_ddecl */
121       if (main_ddecl && main_ddecl->definition)
122         {
123           qtype fqt;
124           store main_store_in;
125           effect main_effect;
126     
127           fqt = main_ddecl->qtype;
128           main_store_in = store_in_qtype(fqt);
129           main_effect = effect_qtype(fqt);
130           mkleq_store(location_declaration(main_ddecl->ast),
131     		  "inconsistent stores at main entry",
132     		  make_store_filter(NULL, toplvl_store, main_effect),
133     		  main_store_in);
134         }
135       else if (flag_driver)
136         {
137           bool found;
138           dd_list_pos cur;
139     
140           found = FALSE;
141           dd_scan(cur, function_defns)
142     	{
143     	  function_decl fd;
144     	  data_declaration ddecl;
145     
146     	  fd = DD_GET(function_decl, cur);
147     	  ddecl = root_ddecl(fd->ddecl);
148     	  if (is_externally_visible_fn(ddecl))
149     	    {
150     	      qtype fqt;
151     	      store store_in, store_out;
152     	      effect eff;
153     	      
154     	      found = TRUE;
155     	      fprintf(stderr, "Assuming %s is an interface function\n",
156     		      ddecl->name);
157     	      fqt = ddecl->qtype;
158     	      store_in = store_in_qtype(fqt);
159     	      store_out = store_out_qtype(fqt);
160     	      eff = effect_qtype(fqt);
161     
162     	      /* Add constraints global store <= f's store <= global store */
163     	      mkleq_store(location_declaration(ddecl->ast),
164     			  "inconsistent stores at interface function entry",
165     			  make_store_filter(NULL, toplvl_store, eff),
166     			  store_in);
167     	      mkleq_store(location_declaration(ddecl->ast),
168     			  "inconsistent stores at interface function exit",
169     			  make_store_ow(NULL, store_out, toplvl_store, eff),
170     			  toplvl_store);
171     	    }
172     	}
173           if (!found)
174     	report_qerror(NULL, sev_warn,
175     		      "No interface functions found\n");
176         }
177       else
178         /* Otherwise we'll report after third pass */
179         report_qerror(NULL, sev_warn,
180     		  "No main function -- initial environment discarded");
181     
182       compute_lins();
183     
184       if (!flag_pam_mode)
185         traverse_globals((traverse_global_fn) warn_if_dangerous_global,
186     		     global_store);
187     
188       if (flag_print_lin)
189         print_queue();
190     }

262     /* Initialize restricted location from non-restricted location.  This
263        is slightly messy because their alocs don't match, so we have to do
264        this manually. */
265     static bool init_restrict(variable_decl vd, store s)
266     {
267       bool result;
268       qtype fs_fqtype, fs_rqtype;
269     
270       assert(vd->rinfo);
271       result = FALSE;
272       fs_fqtype = vd->rinfo->fs_fqtype;
273       fs_rqtype = vd->rinfo->fs_rqtype;
274     
275       /* Top-level qualifier */
276       result = mkleq_qual(vd->loc, qual_qtype(fs_fqtype), qual_qtype(fs_rqtype))
277         || result;
278     
279       /* One level down */
280       fs_fqtype = points_to_fs_qtype(fs_fqtype, s);
281       fs_rqtype = points_to_fs_qtype(fs_rqtype, s);
282       result = mkleq_qual(vd->loc, qual_qtype(fs_fqtype), qual_qtype(fs_rqtype))
283         || result;
284     
285       /* Below this they must in fact match */
286       fs_fqtype = points_to_fs_qtype(fs_fqtype, s);
287       fs_rqtype = points_to_fs_qtype(fs_rqtype, s);
288       result = mkleq_qtype(vd->loc, fs_fqtype, fs_rqtype) || result;
289     
290       return result;
291     }

293     /* Initialize the fs_qtype field of ddecl */
294     static void add_ddecl_fs_qtype(data_declaration ddecl, store s)
295     {
296       location loc;
297     
298       if (!ddecl->qtype || ddecl->fs_qtype)
299         /* Do nothing if we're already assigned ddecl an fs_qtype */
300         return;
301     
302       ddecl = root_ddecl(ddecl);
303       loc = location_declaration(ddecl->ast);
304     
305       if (ddecl->global)
306         {
307           data_declaration global;
308     
309           global = ddecl->global;
310     
311           /* Doesn't shadow anything */
312           if (!global)
313     	{
314     	  ddecl->fs_qtype = qtype_to_fs_qtype_with_quals(loc,
315     							 ddecl->qtype,
316     							 global_store);
317     	  return;
318     	}
319     
320           /* Add fs_qtype to global if necessary */
321           if (global && !global->fs_qtype)
322     	  global->fs_qtype =
323     	    qtype_to_fs_qtype_with_quals(location_declaration(global->ast),
324     					 global->qtype,
325     					 global_store);
326     
327           /* Shadows a global from a prelude file */
328           if (global->in_prelude && !ddecl->in_prelude)
329     	{
330     	  ddecl->fs_qtype = global->fs_qtype;
331     	  return;
332     	}
333     
334           /* Shadows some other global */
335           if (match_type(ddecl->type, global->type))
336     	ddecl->fs_qtype = global->fs_qtype;
337           else
338     	{
339     	  ddecl->fs_qtype = qtype_to_fs_qtype_with_quals(loc,
340     							 ddecl->qtype,
341     							 global_store);
342     	  if (unify_qtype(loc, global->fs_qtype, ddecl->fs_qtype))
343     	    {
344     	      report_qerror(loc, sev_err, "conflicting types for ``%s''",
345     			    ddecl->name);
346     	      report_qerror(global->ast->loc, sev_info,
347     			    "-- previous declaration of ``%s''", ddecl->name);
348     	    }
349     	}
350         }
351       else
352         /* Local */
353         ddecl->fs_qtype = qtype_to_fs_qtype_with_quals(loc,
354     						   ddecl->qtype,
355     						   s);
356     }

364     void analyze_flow_sensitive(declaration program)
365     {
366       declaration d;
367       dinfo next_info;
368

370       scan_declaration(d, program)
371         {
372           /* Assume left-to-right order of evaluation */
373           next_info = analyze_declaration(d, toplvl_store);
374           toplvl_store = next_info.out;
375         }

378     /* Declaration */
379     static dinfo analyze_declaration(declaration d, store in)
380     {
381       switch (d->kind)
382         {
383         case kind_asm_decl:
384           return mkdinfo(in);
385         case kind_data_decl:
386           {
387     	data_decl dd = CAST(data_decl, d);
388     	declaration decl;
389     	store cur_store;
390     
391     	cur_store = in;
392     
393     	scan_declaration(decl, dd->decls) {
394     	  dinfo next;
395     
396     	  next = analyze_declaration(decl, cur_store);
397     	  cur_store = next.out;
398     	}
399     	return mkdinfo(cur_store);
400           }
401         case kind_variable_decl:
402           {
403     	variable_decl vd = CAST(variable_decl, d);
404     	store cur_store;
405     	data_declaration root;
406     
407     	/* Ignore asm_stmt
408     	   assert(!vd->asm_stmt); */
409     
410     	cur_store = in;
411     	root = root_ddecl(vd->ddecl);
412     
413     	if (vd->rinfo)
414     	  /* This var was restricted.  Restore its original qtype */
415     	  root->qtype = vd->rinfo->fqtype;
416     
417     	/* Allocate memory for any definitions */
418     	cur_store = allocate_ddecl(root, cur_store);
419     
420     	add_ddecl_fs_qtype(root, cur_store);
421     	
422     	if (vd->arg1)
423     	  {
424     	    qtype init_qt;
425     	    sinfo arg1;
426     
427     	    init_qt = root->fs_qtype;
428     	    if (!qtype_error(init_qt) && !qtype_array(init_qt))
429     	      init_qt = points_to_fs_qtype(init_qt, cur_store);
430     
431     	    /* XXX: enter blocks without going through init ?! */
432     	    arg1 = analyze_init(vd->loc, init_qt, vd->arg1, cur_store);
433     	    cur_store = arg1.out;
434     	  }
435     
436     	/* Handle restrict */
437     	if (vd->rinfo)
438     	  {
439     	    assert(!vd->rinfo->fs_rqtype && !vd->rinfo->fs_fqtype);
440     	    assert(qtype_pointer(vd->rinfo->rqtype));
441     
442     	    /* Add rtl location (hack to make consistent stores) and
443     	       restricted location to store */
444     	    cur_store = ref_qtype_store(vd->loc, vd->rinfo->rqtype, cur_store);
445     	    cur_store = make_store_ref_effect(vd->loc, cur_store,
446     					      vd->rinfo->rlocs);
447     
448     	    vd->rinfo->fs_fqtype = root->fs_qtype;
449     	    vd->rinfo->fs_rqtype =
450     	      qtype_to_fs_qtype_with_quals(vd->loc, vd->rinfo->rqtype,
451     					   cur_store);
452     
453     	    /* Initialize restricted location from non-restricted
454     	       location */
455     	    if (init_restrict(vd, cur_store))
456     	      report_qerror(vd->loc, sev_err,
457     			    "incompatible right-hand side in restricted "
458     			    "location initialization");
459     
460     	    /* Switch to restricted qtype, which we'll see from now
461     	       on */
462     	    root->fs_qtype = vd->rinfo->fs_rqtype;
463     	  }
464     
465     	if ((flag_pam_mode || flag_print_results) &&
466     	    root->kind != decl_typedef)
467     	  add_flow_sensitive_declarator_overlay(vd->declarator,
468     						root->fs_qtype,
469     						cur_store);
470     
471     	/* XXX: Operators */
472     
473     	return mkdinfo(cur_store);
474           }
475         case kind_function_decl:
476           {
477     	function_decl fd = CAST(function_decl, d), old_function_decl;
478     	data_declaration root;
479     	store cur_store;
480     
481     	dd_add_last(parse_region, function_defns, fd);
482     
483     	root = root_ddecl(fd->ddecl);
484     
485     	/* Ignore definition of polymorphic function, just like in
486     	   analyze.c. */
487     	if (qtype_scheme(root->qtype))
488     	  return mkdinfo(in);
489     
490     	add_ddecl_fs_qtype(root, in);
491     	if (root->global && root->global->in_prelude)
492     	  /* Ignore definition of something from a prelude file */
493     	  return mkdinfo(in);
494     
495     	/* Add location to global store */
496     	global_store_ext = allocate_qtype(location_declaration(root->ast),
497     					  root->qtype,
498     					  global_store_ext);
499     
500     	/* Add PAM overlay */
501     	if (flag_pam_mode || flag_print_results)
502     	  add_flow_sensitive_declarator_overlay(fd->declarator,
503     						root->fs_qtype,
504     						in);
505     
506     	old_function_decl = current_function_decl;
507     	current_function_decl = fd;
508     	if (qtype_error(root->fs_qtype))
509     	  cur_store = make_store_var(fd->loc, "error_store");
510     	else
511     	  cur_store = store_in_qtype(root->fs_qtype);
512     
513     	/*** Scan argument list ***/
514     
515     	{
516     	  declaration arg;
517     	  qtypelist_scanner old_qtypes;
518     	  int i;
519     
520     	  i = 1;
521     	  if (qtype_error(root->fs_qtype))
522     	    old_qtypes = NULL;
523     	  else
524     	    qtypelist_scan(arg_qtypes(root->fs_qtype), &old_qtypes);
525     
526     	  if (!is_void_parms(fd->fdeclarator->parms))
527     	    scan_declaration(arg, fd->fdeclarator->parms)
528     	      /* Ignore ...'s */
529     	      if (!is_ellipsis_decl(arg))
530     		{
531     		  data_declaration root_vd;
532     		  variable_decl argvd;
533     		  oldidentifier_decl oid;
534     		  qtype old_qtype, arg_qt;
535     		  
536     		  /* Construct type of parameter */
537     		  if (arg->kind == kind_data_decl)
538     		    {
539     		      data_decl argd = CAST(data_decl, arg);
540     
541     		      argvd = CAST(variable_decl, argd->decls);
542     		      assert(!argvd->next);    /* Only var_decl per data_decl.
543     						  multi var_decls are for
544     						  things like int a, b. */
545     		      root_vd = root_ddecl(argvd->ddecl);
546     		      oid = NULL;
547     		    }
548     		  else
549     		    {
550     		      oid = CAST(oldidentifier_decl, arg);
551     		      argvd = NULL;
552     		      root_vd = root_ddecl(oid->ddecl);
553     		    }
554     
555     		  /* Build flow sensitive qtype.  root_vd->qtype is
556     		     always a pointer qtype. */
557     		  assert(!root_vd->fs_qtype);
558     		  assert(qtype_pointer(root_vd->qtype));
559     		  if (argvd && argvd->rinfo)
560     		    /* This parameter was restricted.  Restore its
561                            original qtype */
562     		    root_vd->qtype = argvd->rinfo->fqtype;
563     
564     		  cur_store =
565     		    allocate_qtype(location_declaration(root_vd->ast),
566     				   root_vd->qtype,
567     				   cur_store);
568     
569     		  root_vd->fs_qtype =
570     		    qtype_to_fs_qtype_with_quals(arg->loc,
571     						 root_vd->qtype,
572     						 cur_store);
573     
574     		  /* Unify with qtype for parameter.  Unfortunately
575     		     we need to do this, because the previous unification
576     		     in analyze.c unified the qual and the ty, but *not*
577     		     the qtype itself */
578     		  arg_qt = points_to_fs_qtype(root_vd->fs_qtype, cur_store);
579     		  if (!qtype_error(root->fs_qtype))
580     		    {
581     		      old_qtype = qtypelist_next(&old_qtypes);
582     		      if (old_qtype &&
583     			  unify_qtype(arg->loc, old_qtype, arg_qt))
584     			report_qerror(arg->loc, sev_err,
585     				      "type of parameter %d doesn't match "
586     				      "previous uses",
587     				      i);
588     		      else if (!old_qtype && arg->kind == kind_data_decl)
589     			report_qerror(arg->loc, sev_err,
590     				      "parameter %d doesn't appear in "
591     				      "previous uses", i);
592     		    }
593     
594     		  /* Handle restrict */
595     		  if (argvd && argvd->rinfo)
596     		    {
597     		      assert(!argvd->rinfo->fs_rqtype &&
598     			     !argvd->rinfo->fs_fqtype);
599     		      assert(qtype_pointer(argvd->rinfo->rqtype));
600     
601     		      /* Add rtl location (hack to make consistent stores) and
602     			 restricted location to store */
603     		      cur_store = ref_qtype_store(arg->loc,
604     						  argvd->rinfo->rqtype,
605     						  cur_store);
606     		      cur_store = make_store_ref_effect(arg->loc,
607     							cur_store,
608     							argvd->rinfo->rlocs);
609     
610     		      argvd->rinfo->fs_fqtype = root_vd->fs_qtype;
611     		      argvd->rinfo->fs_rqtype =
612     			qtype_to_fs_qtype_with_quals(arg->loc,
613     						     argvd->rinfo->rqtype,
614     						     cur_store);
615     
616     		      /* Initialize restricted location from non-restricted
617     			 location */
618     		      if (init_restrict(argvd, cur_store))
619     			report_qerror(argvd->loc, sev_err,
620     				      "type of restricted parameter %d "
621     				      "doesn't match previous uses",
622     				      i);
623     
624     		      /* Switch to restricted qtype, which we'll see
625     			 from now on */
626     		      root_vd->fs_qtype = argvd->rinfo->fs_rqtype;
627     		    }
628     
629     		  i++;
630     
631     		  if (flag_pam_mode || flag_print_results)
632     		    {
633     		      if (arg->kind == kind_data_decl)
634     		       add_flow_sensitive_declarator_overlay(argvd->declarator,
635     							     root_vd->fs_qtype,
636     							     cur_store);
637     		      else
638     			pam_add_overlay_flow_sensitive(oid->loc,
639     						       root_vd->fs_qtype,
640     						       cur_store);
641     		    }
642     		}
643     	}
644     
645     	/*** Evaluate body ***/
646     
647     	{
648     	  sinfo body;
649     
650     	  assert(is_compound_stmt(fd->stmt));
651     	  body = analyze_statement(fd->stmt, cur_store);
652     	  cur_store = body.out;
653     
654     	  cur_store = merge_restricts(fd->fdeclarator->parms, body.out);
655     
656     	  cur_store = make_store_filter(fd->loc, cur_store, fd->scope_env);
657     	  if (!qtype_error(root->fs_qtype))
658     	    mkleq_store(fd->loc,
659     			"inconsitent stores at function exit",
660     			cur_store, store_out_qtype(root->fs_qtype));
661     	}
662     
663     	current_function_decl = old_function_decl;
664     	return mkdinfo(in);
665           }
666         case kind_extension_decl:
667           {
668     	dinfo result;
669     	extension_decl ed = CAST(extension_decl, d);
670             result =  analyze_declaration(ed->decl, in); /* Ignore __extnesion__ */
671     	return result;
672           }
673         default:
674           fail("Unexpected decl kind %x\n", d->kind);
675         }
676     }

678     static bool init_string(location loc, qtype lhs_qtype, qtype rhs_qtype,
679     			store in)
680     {
681       bool result = FALSE;
682       qtype lhs_contents_qtype, rhs_contents_qtype;
683     
684       lhs_contents_qtype = array_of_qtype(lhs_qtype);
685       rhs_contents_qtype = points_to_fs_qtype(rhs_qtype, in);
686       result = mkleq_qual(loc,
687     		      qual_qtype(rhs_contents_qtype),
688     		      qual_qtype(lhs_contents_qtype)) || result;
689       result = mkleq_qual(loc,
690     		      qual_qtype(rhs_qtype),
691     		      qual_qtype(lhs_qtype)) || result;
692       return result;
693     }

695     static sinfo analyze_init(location loc, qtype lhs_qtype, expression rhs,
696     			  store in)
697     {
698       if (rhs->kind == kind_string && qtype_string(lhs_qtype))
699         {
700           einfo arg1;
701           
702           arg1 = analyze_expression(rhs, rpos, in);
703           if (init_string(rhs->loc, lhs_qtype, arg1.qt, in))
704     	report_qerror(rhs->loc, sev_err, "incompatibile initializer");
705     
706           return mksinfo(arg1.out);
707         }
708     
709       if (rhs->kind == kind_init_list && (qtype_array(lhs_qtype)))
710         {
711           /* Initialize the array contents qtype with each rhs */
712           init_list il = CAST(init_list, rhs);
713           store cur_store;
714           expression e;
715           sinfo init;
716     
717           cur_store = in;
718     
719           scan_expression(e, il->args)
720     	if (e->kind == kind_init_index)
721     	  {
722     	    init_index ii = CAST(init_index, e);
723     	    einfo arg1, arg2;
724     
725     	    arg1 = analyze_expression(ii->arg1, rpos, cur_store);
726     	    cur_store = arg1.out;
727     	    if (ii->arg2)
728     	      {
729     		arg2 = analyze_expression(ii->arg2, rpos, cur_store);
730     		cur_store = arg2.out;
731     	      }
732     	    init = analyze_init(loc, array_of_qtype(lhs_qtype),
733     				ii->init_expr, cur_store);
734     	    cur_store = init.out;
735     	  }
736     	else
737     	  {
738     	    init = analyze_init(loc, array_of_qtype(lhs_qtype),
739     				e, cur_store);
740     	    cur_store = init.out;
741     	  }
742           return mksinfo(cur_store);
743          }
744     
745       if (rhs->kind == kind_init_list && (qtype_struct(lhs_qtype) ||
746     				      qtype_union(lhs_qtype)))
747          {
748            /* Initialize each of the field qtypes with each rhs */
749            init_list il = CAST(init_list, rhs);
750            field_scanner fs;
751            qtype fqt;
752            expression e;
753            store cur_store;
754     
755            fqt = NULL;
756            field_scan(lhs_qtype, &fs);
757            cur_store = in;
758     
759            scan_expression(e, il->args)
760     	 {
761     	   expression init_expr;
762     	   sinfo init;
763     	   
764     	   if (e->kind == kind_init_field)
765     	     {
766     	       /* Handle specific field initialization here -- only
767     		  structures/unions can be initialized this way.
768     		  Adding a label changes where we start initializing
769     		  the struct/union. */
770     	       init_field ifd = CAST(init_field, e);
771     	       field_scan_at(lhs_qtype, ifd->word1->cstring, &fs);
772     	       if (!fs)
773     		 /* Error -- but reported in first pass */
774     		 break;
775     	       init_expr = ifd->init_expr;
776     	     }
777     	   else
778     	     init_expr = e;
779     
780     	   fqt = field_next(&fs);
781     
782     	   if (!fqt)
783     	     /* Error -- but reported in first pass */
784     	     break;
785     
786     	   /* Unless we're initializing an array, get the contents of the
787     	      ref.  Note we can't be initializing a function -- not C++. */
788     	   if (!qtype_array(fqt))
789     	     fqt = points_to_fs_qtype(fqt, cur_store);
790     	   
791     	   init = analyze_init(loc, fqt, init_expr, cur_store);
792     	   cur_store = init.out;
793     	 }
794            return mksinfo(cur_store);
795          }
796     
797       /* Default case. */
798       {
799         einfo arg1;
800     
801         arg1 = analyze_expression(rhs, rpos, in);
802         if (mkleq_fs_assign_convert(loc, arg1.qt, lhs_qtype, arg1.out))
803           report_qerror(loc, sev_err,
804     		    "incompatible right-hand side in initialization");
805         
806         return mksinfo(arg1.out);
807       }
808     }
809     
810     static sinfo analyze_statement(statement s, store in)
811     {
812     #ifdef DEBUG
813       printf("Store before line %ld: ", s->loc->lineno);
814       print_store(printf, in);
815       printf("\n");
816     #endif
817       switch (s->kind)
818         {
819         case kind_asm_stmt:
820           /*      report_qerror(s->loc, sev_warn,
821     	      "warning: ignoring in-line assembly code");*/
822           return mksinfo(in);
823           break;
824         case kind_compound_stmt:
825           {
826     	compound_stmt cs = CAST(compound_stmt, s);	
827     	declaration decl;
828     	statement stmt;
829     	store cur_store, out;
830     	
831     	cur_store = in;
832     	/* cur_store = make_store_filter(in, cs->filter_env); */
833     
834     	/* Analyze the declarations. */
835     	scan_declaration(decl, cs->decls)
836     	  {
837     	    dinfo next;
838     
839     	    next = analyze_declaration(decl, cur_store);
840     	    cur_store = next.out;
841     	  }
842     
843     	/* Analyze the body, gathering up the effects. */
844     	scan_statement(stmt, cs->stmts)
845     	  {
846     	    sinfo next = analyze_statement(stmt, cur_store);
847     	    cur_store = next.out;
848     	  }
849     
850     	cur_store = merge_restricts(cs->decls, cur_store);
851     	out = make_store_filter(cs->loc, cur_store, cs->filter_env);
852     	/*
853     	cur_store = make_store_filter(cs->loc, cur_store, cs->filter_env);
854     	out = make_store_var(cs->loc, "compound_out");
855     	mkleq_store(cs->loc,
856     		    "inconsistent stores at compound statement exit",
857     		    cur_store, out);
858     	*/
859     
860     	return mksinfo(out);
861           };
862           break;
863         case kind_if_stmt:
864           {
865     	if_stmt is = CAST(if_stmt, s);
866     	einfo guard;
867     	sinfo then, els;
868     	store result;
869     
870     	guard = analyze_expression(is->condition, cpos, in);
871     	then = analyze_statement(is->stmt1, true_out_einfo(guard));
872     
873     	if (is->stmt2)
874     	  {
875     	    els = analyze_statement(is->stmt2, false_out_einfo(guard));
876     	    result = lub_store(is->loc,
877     			       "inconsistent then and else branch stores",
878     			       els.out, then.out);
879     	  }
880     	else
881     	  {
882     	    /* if (false) w/no else */
883     	    result = lub_store(is->loc,
884     			       "inconsistent then and fall-through stores",
885     			       then.out, false_out_einfo(guard));
886     	  }
887     
888     	return mksinfo(result);
889           };
890           break;
891         case kind_labeled_stmt:
892           {
893     	labeled_stmt ls = CAST(labeled_stmt, s);
894     	sinfo body;
895     
896     	if (ls->label->store_in)
897     	  mkleq_store(ls->loc,
898     		      "inconsistent stores at label entry",
899     		      in, ls->label->store_in);
900     	else
901     	  {
902     	    ls->label->store_in = make_store_var(ls->loc, "label");
903     	    mkleq_store(ls->loc,
904     			"inconsitent stores at label entry",
905     			in, ls->label->store_in);
906     	  }
907     	body = analyze_statement(ls->stmt, ls->label->store_in);
908     	return body;
909           };
910           break;
911         case kind_expression_stmt:
912           {
913     	expression_stmt es = CAST(expression_stmt, s);
914     	einfo ei;
915     
916     	ei = analyze_expression(es->arg1, rpos, in);
917     	return mksinfo(ei.out);
918           };
919           break;
920         case kind_while_stmt:
921           {
922     	while_stmt ws = CAST(while_stmt, s);
923     	einfo guard;
924     	sinfo body;
925     
926     	ws->break_dest = make_store_var(ws->loc, "ws_break");
927     	ws->continue_dest = make_store_var(ws->loc, "ws_continue");
928     	mkleq_store(ws->loc, "inconsitent stores at loop entry", in,
929     		    ws->continue_dest);
930     	guard = analyze_expression(ws->condition, rpos, ws->continue_dest);
931     
932     	body = analyze_statement(ws->stmt, guard.out);
933     
934     	/* 1+ times through loop */
935     	mkleq_store(ws->loc, "inconsistent stores at loop body exit",
936     		    body.out, ws->continue_dest);
937     
938     	if (!definite_one(ws->condition))
939     	  /* Non-infinite loop, so may exit from guard */
940     	  mkleq_store(ws->loc, "inconsistent stores at loop exit",
941     		      guard.out, ws->break_dest);
942     
943     	return mksinfo(ws->break_dest);
944           };
945           break;
946         case kind_dowhile_stmt:
947           {
948     	dowhile_stmt dws = CAST(dowhile_stmt, s);
949     	einfo guard;
950     	sinfo body;
951     	store before_body_store;
952     
953     	dws->break_dest = make_store_var(dws->loc, "dws_break");
954     	dws->continue_dest = make_store_var(dws->loc, "dws_continue");
955     	before_body_store = make_store_var(dws->loc, "dws_body");
956     	mkleq_store(dws->loc, "inconsistent stores at loop entry",
957     		    in, before_body_store);
958     	body = analyze_statement(dws->stmt, before_body_store);
959     	mkleq_store(dws->loc, "inconsistent stores at loop body entry",
960     		    body.out, dws->continue_dest);
961     
962     	if (definite_zero(dws->condition))
963     	  /* Catch do { x } while (0); case -- used in macro expansions */
964     	  mkleq_store(dws->loc, "inconsistent stores at loop exit",
965     		      dws->continue_dest, dws->break_dest);
966     	else if (definite_one(dws->condition))
967     	  {
968     	    /* 1 time through loop */
969     	    guard = analyze_expression(dws->condition, rpos,
970     				       dws->continue_dest);
971     	  
972     	    /* 2+ times through loop */
973     	    mkleq_store(dws->loc, "inconsistent stores at loop body entry",
974     			guard.out, before_body_store); 
975     	  }
976     	else
977     	  {
978     	    /* 1 time through loop */
979     	    guard = analyze_expression(dws->condition, rpos,
980     				       dws->continue_dest);
981     	  
982     	    /* 2+ times through loop */
983     	    mkleq_store(dws->loc, "inconsistent stores at loop body entry",
984     			guard.out, before_body_store);
985     
986     	    /* May exit after guard */
987     	    mkleq_store(dws->loc, "inconsistent stores at loop exit",
988     			guard.out, dws->break_dest);
989     	  }
990     
991     	return mksinfo(dws->break_dest);
992           };
993           break;
994         case kind_switch_stmt:
995           {
996     	switch_stmt ss = CAST(switch_stmt, s);
997     	einfo guard;
998     	sinfo body;
999     	label cases;
1000     	bool has_default;
1001     
1002     	ss->break_dest = make_store_var(ss->loc, "break_dest");
1003     
1004     	guard = analyze_expression(ss->condition, rpos, in);
1005     	has_default = FALSE;
1006     	for (cases = ss->next_label; cases; cases = cases->next_label)
1007     	  {
1008     	    switch (cases->kind)
1009     	      {
1010     	      case kind_default_label:
1011     		has_default = TRUE;
1012     	      case kind_case_label:
1013     		/* this is a jump, but target (case stmts) are
1014     		   guaranteed to be included in this scope.  So we
1015     		   don't need to apply Down */
1016      		cases->store_in = make_store_var(cases->loc, "ss_case");
1017     	 	mkleq_store(ss->loc, "inconsistent stores at case entry",
1018     			    guard.out, cases->store_in);
1019     		break;
1020      	      default:
1021     	 	fail("Unexpected label kind %x in switch\n", cases->kind);
1022     	      }
1023      	  }
1024     
1025     	/* Note:  no connection from guard to body.  Only can jump
1026     	   to a case. */
1027     	body = analyze_statement(ss->stmt,
1028     				 make_store_var(ss->loc, "ss_start"));
1029     	/* body may end without a break */
1030     	mkleq_store(ss->loc, "inconsistent stores at body exit",
1031     		    body.out, ss->break_dest); 
1032     
1033     	if (!has_default)
1034     	  /* There may be no matching case */
1035     	  mkleq_store(ss->loc, "inconsistent stores at body exit",
1036     		      guard.out, ss->break_dest);
1037     
1038     	return mksinfo(ss->break_dest);
1039           };
1040           break;
1041         case kind_for_stmt:
1042           {
1043     	for_stmt fs = CAST(for_stmt, s);
1044     	sinfo body;
1045     	store cur_store;
1046     
1047     	fs->break_dest = make_store_var(fs->loc, "fs_break");
1048     	fs->continue_dest = make_store_var(fs->loc, "fs_continue");
1049     
1050     	cur_store = in;
1051     
1052     	if (fs->arg1)
1053     	  {
1054     	    einfo arg1;
1055     	    arg1 = analyze_expression(fs->arg1, rpos, cur_store);
1056     	    cur_store = arg1.out;
1057     	  }
1058     
1059     	mkleq_store(fs->loc, "inconsistent stores at loop entry",
1060     		    cur_store, fs->continue_dest);
1061     	cur_store = fs->continue_dest;
1062     
1063     	if (fs->arg2)
1064     	  {
1065     	    einfo arg2;
1066     	    arg2 = analyze_expression(fs->arg2, rpos, cur_store);
1067     	    cur_store = arg2.out;
1068     
1069     	    /* Might exit loop */
1070     	    mkleq_store(fs->loc, "inconsistent stores at loop exit",
1071     			cur_store, fs->break_dest);
1072     	  }
1073     
1074     	body = analyze_statement(fs->stmt, cur_store);
1075     	cur_store = body.out;
1076     
1077     	if (fs->arg3)
1078     	  {
1079     	    einfo arg3;
1080     	    arg3 = analyze_expression(fs->arg3, rpos, cur_store);
1081     	    cur_store = arg3.out;
1082     	  }
1083     
1084     	mkleq_store(fs->loc, "inconsistent stores at loop entry",
1085     		    cur_store, fs->continue_dest);
1086     
1087     	return mksinfo(fs->break_dest);
1088           };
1089           break;
1090         case kind_return_stmt:
1091           {
1092     	return_stmt rs = CAST(return_stmt, s);
1093     	qtype fun_qtype;
1094     	store cur_store;
1095     
1096     	assert(current_function_decl);
1097     	fun_qtype = root_ddecl(current_function_decl->ddecl)->fs_qtype;
1098     
1099     	cur_store = in;
1100     	if (rs->arg1)
1101     	  {
1102     	    einfo arg1;
1103     
1104     	    arg1 = analyze_expression(rs->arg1, rpos, in);
1105     	    cur_store = arg1.out;
1106     	    if (!qtype_error(fun_qtype) &&
1107     		mkleq_fs_assign_convert(rs->loc, arg1.qt,
1108     					return_qtype(fun_qtype), arg1.out))
1109     	      report_qerror(rs->loc, sev_err,
1110     			    "return type incompatible with function type");
1111     	  }
1112     
1113     	cur_store = merge_restricts(current_function_decl->fdeclarator->parms,
1114     				    cur_store); 
1115     
1116     	/* XXX: Could get rid of this here and use a variable as the
1117     	   join point for returns. */
1118     	cur_store = make_store_filter(rs->loc,
1119     				      cur_store,
1120     				      current_function_decl->scope_env);
1121     	if (!qtype_error(fun_qtype))
1122     	  mkleq_store(rs->loc, "inconsistent stores at function exit",
1123     		      cur_store, store_out_qtype(fun_qtype));
1124     
1125     	return mksinfo(make_store_var(rs->loc, "return"));
1126           };
1127           break;
1128         case kind_computed_goto_stmt:
1129           {
1130     	/* o|-< UNSOUND */
1131     	computed_goto_stmt cgs = CAST(computed_goto_stmt, s);
1132     	einfo arg1;
1133     
1134     	arg1 = analyze_expression(cgs->arg1, rpos, in);/* stmt is goto *arg1 */
1135     	return mksinfo(make_store_var(cgs->loc, "computed_goto"));
1136           };
1137           break;
1138         case kind_break_stmt:
1139           {
1140     	break_stmt bs = CAST(break_stmt, s);
1141     	store cur_store;
1142     
1143     	cur_store = tfr_jumped_blocks(in, bs->enclosing_cs, 
1144     				      bs->parent_loop->enclosing_cs);
1145     
1146     	cur_store = make_store_filter(bs->loc, cur_store,
1147     				   bs->parent_loop->enclosing_cs->scope_env);
1148     
1149     	mkleq_store(bs->loc,
1150     		    (bs->parent_loop->kind == kind_switch_stmt ?
1151     		     "inconsistent stores at switch exit" :
1152     		     "inconsistent stores at loop exit"),
1153     		    cur_store, bs->parent_loop->break_dest);
1154     	return mksinfo(make_store_var(bs->loc, "break_stmt"));
1155           }
1156           break;
1157         case kind_continue_stmt:
1158           {
1159     	continue_stmt cs = CAST(continue_stmt, s);
1160     	store cur_store;
1161     
1162     	cur_store = tfr_jumped_blocks(in, cs->enclosing_cs, 
1163     				      cs->parent_loop->enclosing_cs);
1164     
1165     	cur_store = make_store_filter(cs->loc, in,
1166     				   cs->parent_loop->enclosing_cs->scope_env);
1167     
1168     	mkleq_store(cs->loc,
1169     		    (cs->parent_loop->kind == kind_switch_stmt ?
1170     		     "inconsistent stores at switch exit" :
1171     		     "inconsistent stores at loop exit"),
1172     		    cur_store, cs->parent_loop->continue_dest);
1173     	return mksinfo(make_store_var(cs->loc, "continue_stmt"));
1174           }
1175           break;
1176         case kind_goto_stmt:
1177           {
1178     	goto_stmt gs = CAST(goto_stmt, s);
1179     	label_declaration ldecl;
1180     	id_label target;
1181     	store cur_store;
1182     
1183     	ldecl = gs->id_label->ldecl;
1184     	target = ldecl->definition;
1185     	if (!target->store_in)
1186     	  target->store_in = make_store_var(target->loc, "label");
1187     	cur_store = tfr_jumped_blocks(in, gs->enclosing_cs,
1188     				      target->enclosing_cs);
1189     	cur_store = make_store_filter(gs->loc, cur_store,
1190     				      target->enclosing_cs->scope_env);
1191     	mkleq_store(gs->loc, "inconsistent stores at unconditional branch",
1192     		    cur_store, target->store_in);
1193     
1194     	return mksinfo(make_store_var(gs->loc, "goto_stmt"));
1195           }
1196           break;
1197         case kind_empty_stmt:
1198           return mksinfo(in);
1199         case kind_change_type_stmt:
1200           {
1201     	change_type_stmt ct = CAST(change_type_stmt, s);
1202     	einfo arg1;
1203     	aloc al;
1204     	store out;
1205     	qtype new_qt;
1206     
1207     	arg1 = analyze_expression(ct->arg1, lpos, in);
1208     	out = arg1.out;
1209     	al = aloc_qtype(arg1.qt);
1210     	new_qt = qtype_to_fs_qtype_with_quals(ct->loc, ct->qtype, out);
1211     	out = assign_flow_sensitive(ct->loc,
1212     				    "incompatible types in change_type",
1213     				    new_qt, arg1.qt, out, TRUE);
1214     
1215     	if (flag_pam_mode || flag_print_results)
1216     	  pam_add_overlay_lin(ct->loc, out, al);
1217     
1218     	if (flag_print_lin)
1219     	  queue_print_stmt((statement) s, out, al);
1220     
1221           return mksinfo(out);
1222           }
1223           break;
1224         case kind_assert_type_stmt:
1225           {
1226     	assert_type_stmt at = CAST(assert_type_stmt, s);
1227     	einfo arg1;
1228     	qtype assert_qt;
1229     	
1230     	arg1 = analyze_expression(at->arg1, rpos, in);
1231     	/* Get declared qtype; ignore any effect of arg1 */
1232     	assert_qt = qtype_to_fs_qtype_with_quals(at->loc, at->qtype, in);
1233     	if (mkleq_fs_qtype(at->loc, arg1.qt, assert_qt, in))
1234     	  report_qerror(at->loc, sev_err,
1235     			"unsatisfiable assert_type");
1236     	return mksinfo(arg1.out);
1237           }
1238           break;
1239         case kind_deep_restrict_stmt:
1240           {
1241     	deep_restrict_stmt dr = CAST(deep_restrict_stmt, s);
1242     	einfo arg1;
1243     	sinfo stmt;
1244     	store cur_store;
1245     	bool result;
1246     
1247     	assert(dr->drinfo);
1248     
1249     	arg1 = analyze_expression(dr->arg1, lpos, in);
1250     	cur_store = arg1.out;
1251     	dr->drinfo->fs_qtype = arg1.qt;
1252     
1253     	/* Allocated the deep restricted location */
1254     	cur_store = ref_qtype_store(dr->loc, dr->drinfo->rqtype, cur_store);
1255     
1256     	/* initialize the deep restrict type */
1257     	dr->drinfo->fs_rqtype = 
1258     	  qtype_to_fs_qtype_with_quals(dr->loc, dr->drinfo->rqtype, cur_store);
1259     	
1260     	result = mkleq_qual(dr->loc, qual_qtype(dr->drinfo->fs_qtype),
1261     			    qual_qtype(dr->drinfo->fs_rqtype));
1262     	assert(qtype_pointer_loc(dr->drinfo->fs_rqtype));
1263     	result = mkleq_qtype(dr->loc, 
1264     			     points_to_fs_qtype(dr->drinfo->fs_qtype, 
1265     						cur_store),
1266     			     points_to_fs_qtype(dr->drinfo->fs_rqtype,
1267     						cur_store));
1268     
1269     	if (result)
1270     	  report_qerror(dr->loc, sev_err,
1271     			"incompatible right-hand side in deeprestricted "
1272     			"location initialization");
1273     	
1274     	stmt = analyze_statement(dr->stmt, cur_store);
1275     	cur_store = stmt.out;
1276     	
1277     	/* Merge at the exit */
1278     	{
1279     	  qtype rqtype, fqtype;
1280     
1281     	  fqtype = dr->drinfo->fs_qtype;
1282     	  rqtype = points_to_fs_qtype(dr->drinfo->fs_rqtype, in);
1283           
1284     	  cur_store = assign_flow_sensitive(dr->loc,
1285     			   "incompatible types at deep-restrict scope end",
1286     				     rqtype, fqtype, cur_store, FALSE);
1287     	}
1288     	return mksinfo(cur_store);
1289           }
1290         break;
1291         default:
1292           fail("Unexpected statement kind %x\n", s->kind);
1293         }
1294     }

1296     static einfo analyze_expression(expression e, context context, store in)
1297     {
1298     #ifdef DEBUG
1299       printf("Store before ");
1300       prt_expression(e, 0);
1301       printf(": ");
1302       print_store(printf, in);
1303       printf("\n");
1304     #endif
1305       
1306       /* deep restrict */
1307       if (e->drinfo)
1308         {
1309           qtype qt;
1310     
1311           qt = e->drinfo->fs_rqtype;
1312           assert(qt);
1313           assert(qt && qtype_pointer_loc(qt));
1314     
1315           if (context == rpos)
1316     	return mkeinfo(points_to_fs_qtype(qt, in), in, NULL, NULL, FALSE);
1317           else 
1318     	return mkeinfo(qt, in, NULL, NULL, FALSE);
1319         }
1320     
1321       /*  printf("Analyzing ");
1322           prt_expression(e, 0);
1323           printf(" in %s\n", context_tostring(context));*/
1324       switch(e->kind) {
1325       case kind_comma:
1326         {
1327           comma c = CAST(comma, e);
1328           einfo last;
1329           expression e2;
1330     
1331           last = mkeinfo(NULL, in, NULL, NULL, FALSE);
1332           scan_expression (e2, c->arg1)
1333     	last = analyze_expression(e2, context, last.out);
1334           return last;
1335         };
1336         break;
1337       case kind_sizeof_type:
1338       case kind_alignof_type:
1339       case kind_known_cst:
1340       case kind_lexical_cst:
1341       case kind_sizeof_expr:
1342       case kind_alignof_expr:
1343         {
1344           qtype qt;
1345     
1346           qt = qtype_to_fs_qtype_with_quals(e->loc, e->qtype, in);
1347           return mkeinfo(qt, in, NULL, NULL, FALSE);
1348         }
1349         break;
1350       case kind_label_address:
1351       case kind_string:
1352         {
1353           qtype qt;
1354     
1355           qt = qtype_to_fs_qtype_with_quals(e->loc, e->qtype, in);
1356     
1357           global_store_ext = allocate_qtype(e->loc, e->qtype, global_store_ext);
1358           return mkeinfo(qt, in, NULL, NULL, FALSE);
1359         }
1360         break;
1361       case kind_cast:
1362         {
1363           cast c = CAST(cast, e);
1364           einfo arg1;
1365           type cast_t;
1366           qtype qt;
1367           store cur_store;
1368     
1369           arg1 = analyze_expression(c->arg1, context, in);
1370           cur_store = arg1.out;
1371           cast_t = c->asttype->type;
1372     
1373           qt = qtype_to_fs_qtype_with_quals(c->loc, e->qtype, in);
1374     
1375           /* ad-hoc malloc handling in action */
1376           if (arg1.ismalloc)
1377     	cur_store = allocate_qtype(c->loc, e->qtype, cur_store);
1378     
1379           /*
1380           if (flag_casts_preserve && !type_contains_quals(cast_t))
1381     	{
1382     	  if (mkeq_qtype_cast(c->loc, qt, arg1.qt))
1383     	    report_qerror(c->loc, sev_err,
1384     			  "Incompatible qualifiers at cast");
1385     	}
1386           */      
1387     
1388           if (flag_casts_warn && !arg1.ismalloc && !type_contains_quals(cast_t))
1389     	{
1390     	  mk_no_qual_qtype_fs(c->loc, arg1.qt, cur_store);
1391     	  mk_no_qual_qtype_fs(c->loc, qt, cur_store);
1392     	}
1393     
1394           return mkeinfo(qt, cur_store, NULL, NULL, FALSE);
1395         };
1396         break;
1397       case kind_cast_list:
1398         {
1399           cast_list cl = CAST(cast_list, e);
1400           qtype qt;
1401           sinfo init;
1402     
1403           qt = qtype_to_fs_qtype_with_quals(cl->loc, cl->astqtype, in);
1404           init = analyze_init(cl->loc, qt, cl->init_expr, in);
1405     
1406           qt = default_conversion_fs_qtype(qt);
1407           return mkeinfo(qt, init.out, NULL, NULL, FALSE);
1408         };
1409         break;
1410       case kind_conditional:
1411         {
1412           conditional c = CAST(conditional, e);
1413           einfo cond, arg1, arg2;
1414           store out;
1415           qtype fs_qt;
1416     
1417           /* XXX: cpos not handled yet, so changed to rpos */
1418           cond = analyze_expression(c->condition, rpos, in);
1419           if (c->arg1)
1420     	arg1 = analyze_expression(c->arg1,
1421     				  context == cpos ? rpos : context,
1422     				  cond.out);
1423           else
1424     	/* gcc extension:  if you omit the middle op, evaluates to
1425     	   guard when guard is true, and guard is not reevaluated */
1426     	arg1 = cond;
1427           arg2 = analyze_expression(c->arg2,
1428     				context == cpos ? rpos : context,
1429     				cond.out);
1430           out = lub_store(c->loc, "inconsistent ? and : stores",
1431     		      arg1.out, arg2.out);
1432     
1433           fs_qt = qtype_to_fs_qtype(c->loc, c->qtype);
1434           if (qtype_void(arg1.qt))
1435     	mkleq_fs_assign_convert(c->loc, arg1.qt, fs_qt, out);
1436           else if (qtype_void(arg2.qt))
1437     	mkleq_fs_assign_convert(c->loc, arg2.qt, fs_qt, out);
1438           else if (mkleq_fs_assign_convert(c->loc, arg1.qt, fs_qt, out) ||
1439     	       mkleq_fs_assign_convert(c->loc, arg2.qt, fs_qt, out))
1440     	report_qerror(c->loc, sev_err,
1441     		      "incompatible branches of conditional");
1442     
1443           return mkeinfo(fs_qt, out, NULL, NULL, FALSE);
1444         };
1445         break;
1446       case kind_identifier:
1447         {
1448           identifier id = CAST(identifier, e);
1449           data_declaration root;
1450           qtype qt;
1451     
1452           root = root_ddecl(id->ddecl);
1453     
1454           qt = root->fs_qtype;
1455           if (!qt)
1456     	{
1457     	  add_ddecl_fs_qtype(root, in);
1458     	  qt = root->fs_qtype;
1459     	}
1460           if ((type_function(id->ddecl->type) &&
1461               id->ddecl->ftype == function_implicit) ||
1462              !strcmp(id->ddecl->name, "__FUNCTION__") ||
1463              !strcmp(id->ddecl->name, "__PRETTY_FUNCTION__"))
1464             global_store_ext = allocate_qtype(id->loc, qt, global_store_ext);
1465     
1466           if (flag_pam_mode || flag_print_results)
1467     	pam_add_overlay_flow_sensitive(id->loc, qt, in);
1468     
1469           if (root->kind != decl_constant)
1470     	/* Put non-enums in context */
1471     	return put_id_in_fs_context(id->loc, qt, context, in);
1472           else
1473     	return mkeinfo(qt, in, NULL, NULL, FALSE);
1474         };
1475         break;
1476       case kind_compound_expr:
1477         {
1478           compound_expr ce = CAST(compound_expr, e);
1479           compound_stmt cs = CAST(compound_stmt, ce->stmt);
1480           statement cur_stmt;
1481           declaration d;
1482           qtype qt;
1483           store cur_store, out;
1484     
1485           assert(!cs->id_labels);
1486           
1487           cur_store = in;
1488           /* cur_store = make_store_filter(in, ce->filter_env); */
1489     
1490           /* Analyze the declarations in the block */
1491           scan_declaration(d, cs->decls)
1492     	{
1493     	  dinfo next;
1494     
1495     	  assert(d->kind != kind_asm_decl); /*asm_decl only at toplevel */
1496     	  next = analyze_declaration(d, cur_store);
1497     	  cur_store = next.out;
1498     	}
1499           /* Analyze the statements in the block.  Analyze all but the
1500              last one. */
1501           cur_stmt = cs->stmts;
1502           while (cur_stmt && cur_stmt->next)
1503     	{
1504     	  sinfo next = analyze_statement(cur_stmt, cur_store);
1505     	  cur_store = next.out;
1506     	  cur_stmt = CAST(statement, cur_stmt->next);
1507     	}
1508     
1509           /* Now analyze the last statement (if there is one), and
1510              compute the type of the expression. */
1511           if (cur_stmt && is_expression_stmt(cur_stmt))
1512     	{
1513     	  einfo next;
1514     	  next = analyze_expression(CAST(expression_stmt,cur_stmt)->arg1,
1515     				    context,
1516     				    cur_store);
1517     	  qt = next.qt;
1518     	  cur_store = next.out;
1519     	}
1520           else
1521     	{
1522     	  /* Type is void */
1523     	  qt = mkqtype_void(make_qvar("compound", ce->loc, FALSE));
1524     	  if (cur_stmt)
1525     	    {
1526     	      sinfo next = analyze_statement(cur_stmt, cur_store);
1527     	      cur_store = next.out;
1528     	    }
1529     	}
1530     
1531           cur_store = merge_restricts(cs->decls, cur_store);
1532           out = make_store_filter(ce->loc, cur_store, ce->filter_env);
1533           /*
1534           cur_store = make_store_filter(ce->loc, cur_store, ce->filter_env);
1535           out = make_store_var(ce->loc, "compound_out");
1536           mkleq_store(ce->loc,
1537     		  "inconsistent stores at compount expression exit",
1538     		  cur_store, out);
1539           */
1540     
1541           return mkeinfo(qt, out, NULL, NULL, FALSE);
1542         };
1543         break;
1544       case kind_function_call:
1545         {
1546           function_call fc = CAST(function_call, e);
1547           einfo f_info, magic_info;
1548           qtype fqt;
1549           qtypelist args;
1550           expression arg;
1551           qtypelist_scanner args_qs;
1552           int i;
1553           store cur_store, out;
1554     
1555           if (fc->va_arg_call)
1556     	{
1557     	  qtype va_arg_fs_qtype;
1558     
1559     	  /* Magic call to __builtin_va_arg(args, fc->va_arg_call) */
1560     	  va_arg_fs_qtype = qtype_to_fs_qtype(fc->loc, fc->qtype);
1561     	  return mkeinfo(va_arg_fs_qtype, in, NULL, NULL, FALSE);
1562     	}
1563     
1564           cur_store = in;
1565           f_info = analyze_expression(fc->arg1, rpos, cur_store);
1566           cur_store = f_info.out; /* Assume function evaluated first */
1567     
1568           fqt = NULL;
1569           if (!qtype_error(f_info.qt))
1570     	fqt = points_to_fs_qtype(f_info.qt, cur_store);
1571     
1572           if (qtype_error(f_info.qt) || qtype_error(fqt))
1573     	{
1574     	  /* Evaluate arguments */
1575     	  scan_expression(arg, fc->args)
1576     	    {
1577     	      einfo arg_info;
1578     
1579     	      arg_info = analyze_expression(arg, rpos, cur_store);
1580     	      cur_store = arg_info.out;
1581     	    }
1582     
1583     	  return mkeinfo(error_qtype, cur_store, NULL, NULL, FALSE);
1584     	}
1585     
1586     
1587           magic_info = analyze_magic_function_call(fc, fqt, cur_store);
1588           if (magic_info.out)
1589     	return magic_info;
1590     
1591           /* Analyze each argument, making a constraint with the
1592     	 corresponding formal paramter */
1593           args = arg_qtypes(fqt);
1594           qtypelist_scan(args, &args_qs);
1595           i = 1;
1596           scan_expression(arg, fc->args)
1597     	{
1598     	  einfo arg_info;
1599     	  qtype formal_qt;
1600     
1601     	  arg_info = analyze_expression(arg, rpos, cur_store);
1602     	  cur_store = arg_info.out; /* Assume left-to-right order
1603     				      of evaluation of arguments */
1604     	  formal_qt = qtypelist_next(&args_qs);
1605     
1606     	  if (formal_qt)
1607     	    {
1608     	      /* Flow arg qtype into function.  No flow-sensitivity
1609     		 here, because this is an initialization. */
1610     	      if (mkleq_fs_assign_convert(arg->loc, arg_info.qt, formal_qt,
1611     					  cur_store))
1612     		report_qerror(arg->loc, sev_err,
1613     			      "type of actual argument %d doesn't match "
1614     			      "type of formal", i);
1615     	    }
1616     	  
1617     	  i++;
1618     	}
1619     
1620           mkleq_store(fc->loc,
1621     		  "inconsistent stores at function entry",
1622     		  make_store_filter(fc->loc, cur_store, effect_qtype(fqt)),
1623     		  store_in_qtype(fqt));
1624           
1625           if (noreturn_qual &&
1626     	  has_qual_qtype(points_to_qtype(fc->arg1->qtype), noreturn_qual))
1627     	out = make_store_var(fc->loc, "noreturn");
1628           else
1629     	{
1630     	  /*
1631     	  out = make_store_ow(fc->loc, store_out_qtype(fqt), cur_store,
1632     			      effect_qtype(fqt));
1633     	  */
1634     	  cur_store = make_store_ow(fc->loc, store_out_qtype(fqt), cur_store,
1635     				    effect_qtype(fqt));
1636     	  out = make_store_var(fc->loc, "func_out");
1637     	  mkleq_store(fc->loc, "inconsistent stores at function return",
1638     		      cur_store, out);
1639     	}
1640     
1641           return mkeinfo(return_qtype(fqt), out, NULL, NULL, FALSE);
1642         };
1643         break;
1644       case kind_array_ref:
1645         {
1646           array_ref ar = CAST(array_ref, e);
1647     
1648           assert(ar->alt);
1649           return analyze_expression(ar->alt, context, in);
1650         };
1651         break;
1652       case kind_field_ref:
1653         {
1654           field_ref fr = CAST(field_ref, e);
1655           einfo arg1;
1656           qtype field_qt;
1657     
1658           arg1 = analyze_expression(fr->arg1, context, in);
1659           if (qtype_error(arg1.qt))
1660     	return arg1;
1661     
1662           /*** Compute field qtypes ***/
1663           
1664           {
1665     	qtype qt, ag_qt;
1666     
1667     	qt = arg1.qt;
1668     	if (context == lpos || context == apos)
1669     	  ag_qt = points_to_fs_qtype(qt, in);
1670     	else
1671     	  ag_qt = qt;
1672     
1673     	if (qtype_error(ag_qt))
1674     	  field_qt = error_qtype;
1675     	else
1676     	  {
1677     	    assert(qtype_aggregate(ag_qt));
1678     	    field_qt = field_qtype(ag_qt, fr->cstring);
1679     	  }
1680           }
1681     
1682           if (flag_pam_mode || flag_print_results)
1683     	pam_add_overlay_flow_sensitive(fr->cstring_loc, field_qt, arg1.out);
1684     
1685           return put_id_in_fs_context(fr->loc, field_qt, context, arg1.out);
1686         };
1687         break;
1688       case kind_init_list:
1689       case kind_init_index:
1690       case kind_init_field:
1691         fail("Unexpected expression kind %d\n", e->kind);
1692         break;
1693       default:
1694         if (is_unary(e))
1695           return analyze_unary_expression(CAST(unary, e), context, in);
1696         else if (is_binary(e))
1697           return analyze_binary_expression(CAST(binary, e), context, in);
1698         else
1699           fail("Unexpected expression kind %x\n", e->kind);
1700       }
1701     }

1703     static einfo analyze_unary_expression(unary e, context context, store in)
1704     {
1705       switch (e->kind)
1706         {
1707         case kind_dereference:
1708           {
1709     	/* Analyze the subexpression in an r-context; this will return
1710     	   an l-type, so either return that if we're in an l-context or
1711     	   deref it if we're in an r-context.  Function are special --
1712     	   *f is allowed even if f is a function and not a function
1713     	   pointer type. */
1714     	einfo arg1;
1715     
1716     	arg1 = analyze_expression(e->arg1, rpos, in);
1717     	if (qtype_error(arg1.qt))
1718     	  return arg1;
1719     	assert(qtype_pointer_loc(arg1.qt));
1720     	if (context == lpos || context == apos)
1721     	  /* *e in an l-context silently converts a (pointer) r-type
1722     	     to an l-type */
1723     	    return mkeinfo(arg1.qt, arg1.out, NULL, NULL, FALSE);
1724     	else if (context == rpos || context == cpos || context == apos)
1725     	  {
1726     	    qtype qt;
1727     
1728     	    assert(!qtype_function(arg1.qt));
1729     	    qt = points_to_fs_qtype(arg1.qt, arg1.out);
1730     	    qt = default_conversion_fs_qtype(qt);
1731     
1732     	    return mkeinfo(qt, arg1.out, NULL, NULL, FALSE);
1733     	  }
1734     	else
1735     	  fail("Unexpected context %d\n", context);
1736     
1737     	/* o|-< ignore op for now */
1738           }
1739         case kind_address_of:
1740           {
1741     	/* &e can only appear in an r-context.  We analyze the
1742     	   subexpression in an l-context (thus the subexpression type
1743     	   with have a ptr at the outside) and silently convert the
1744     	   l-type to an r-type.  On the other hand, taking the address
1745     	   of a function type is a no-op. */
1746     	if (type_function(e->arg1->type))
1747     	  return analyze_expression(e->arg1, context, in);
1748     	else
1749     	  return analyze_expression(e->arg1, apos, in);
1750           }
1751         case kind_extension_expr:
1752           return analyze_expression(e->arg1, context, in);
1753         case kind_realpart:
1754         case kind_imagpart:
1755           {
1756     	/* o|-< ??? */
1757     	einfo arg1;
1758     	qtype qt;
1759     
1760     	arg1 = analyze_expression(e->arg1, rpos, in);
1761     	assert(qtype_complex(arg1.qt));
1762     	qt = complex_base_qtype(arg1.qt);
1763     
1764     	return mkeinfo(qt, arg1.out, NULL, NULL, FALSE);
1765           }
1766         case kind_unary_minus:
1767         case kind_unary_plus:
1768         case kind_conjugate:
1769         case kind_bitnot:
1770           /* These operations don't change types.  Note we assume
1771     	 that compatibility with these ops has already been
1772     	 tested. */
1773           assert(context == rpos || context == cpos);
1774           return analyze_expression(e->arg1, context, in);
1775         case kind_not:
1776           {
1777     	/* not doesn't change types, but it swaps true/false branches */
1778     	einfo arg1;
1779     
1780     	arg1 = analyze_expression(e->arg1, context, in);
1781     	if (context == cpos)
1782     	  return mkeinfo(arg1.qt, arg1.out, arg1.false_out, arg1.true_out, FALSE);
1783     	else
1784     	  return arg1;
1785           }
1786         case kind_preincrement:
1787         case kind_postincrement:
1788         case kind_predecrement:
1789         case kind_postdecrement:
1790           {
1791     	einfo result;
1792     	result = analyze_expression(e->arg1, lpos, in);
1793     	result.qt = points_to_fs_qtype(result.qt, in);
1794     	if (context == cpos)
1795     	  {
1796     	    result.true_out = result.out;
1797     	    result.false_out = result.out;
1798     	    result.out = NULL;
1799     	  }
1800     	return result;
1801           }
1802         default:
1803           fail("Unexpected unary expression kind %x\n", e->kind);
1804         }
1805     }

1807     static einfo analyze_binary_expression(binary e, context context, store in)
1808     {
1809       assert(context == rpos || context == cpos);
1810       switch (e->kind)
1811         {
1812         case kind_plus:
1813           {
1814     	einfo arg1, arg2;
1815     	qtype qt;
1816     
1817     	arg1 = analyze_expression(e->arg1, rpos, in);
1818     	arg2 = analyze_expression(e->arg2, rpos, arg1.out);
1819     
1820     	if (qtype_pointer_loc(arg1.qt) && qtype_pointer_loc(arg2.qt))
1821     	  fail_loc(e->loc, "Unexpected pointer addition\n", 0);
1822     	else if (qtype_pointer_loc(arg1.qt) && !qtype_pointer_loc(arg2.qt))
1823     	  qt = arg1.qt;
1824     	else if (!qtype_pointer_loc(arg1.qt) && qtype_pointer_loc(arg2.qt))
1825     	  qt = arg2.qt;
1826     	else
1827     	  {
1828     	    /* No lub here -- when would lub ever make sense? */
1829     	    if (mkeq_qtype(e->loc, arg1.qt, arg2.qt))
1830     	      report_qerror(e->loc, sev_err,
1831     			    "incompatible operands of ``%s''",
1832     			    binary_op_name(e->kind));
1833     	    qt = arg1.qt;
1834     	  }
1835     	return mkeinfo(qt, arg2.out, NULL, NULL, FALSE);
1836           }
1837           break;
1838         case kind_minus:
1839           {
1840     	einfo arg1, arg2;
1841     	qtype qt;
1842     
1843     	arg1 = analyze_expression(e->arg1, rpos, in);
1844     	arg2 = analyze_expression(e->arg2, rpos, arg1.out);
1845     
1846     	if (qtype_pointer_loc(arg1.qt) && qtype_pointer_loc(arg2.qt))
1847     	  qt = qtype_to_fs_qtype(e->loc, e->qtype);
1848     	else if (qtype_pointer_loc(arg1.qt) && !qtype_pointer_loc(arg2.qt))
1849     	  qt = arg1.qt;
1850     	else if (!qtype_pointer_loc(arg1.qt) && qtype_pointer_loc(arg2.qt))
1851     	  qt = arg2.qt;
1852     	else
1853     	  {
1854     	    /* No lub here -- when would lub ever make sense? */
1855     	    if (mkeq_qtype(e->loc, arg1.qt, arg2.qt))
1856     	      report_qerror(e->loc, sev_err,
1857     			    "incompatible operands of ``%s''",
1858     			    binary_op_name(e->kind));
1859     	    qt = arg1.qt;
1860     	  }
1861     	return mkeinfo(qt, arg2.out, NULL, NULL, FALSE);
1862           }
1863           break;
1864         case kind_modulo:
1865         case kind_lshift:
1866         case kind_rshift:
1867           {
1868     	einfo arg1, arg2;
1869     
1870     	arg1 = analyze_expression(e->arg1, rpos, in);
1871     	arg2 = analyze_expression(e->arg2, rpos, arg1.out);
1872     
1873     	return mkeinfo(arg1.qt, arg2.out, NULL, NULL, FALSE);
1874           }
1875           break;
1876         case kind_times:
1877         case kind_divide:
1878         case kind_bitand:
1879         case kind_bitor:
1880         case kind_bitxor:
1881           {
1882     	einfo arg1, arg2;
1883     
1884     	arg1 = analyze_expression(e->arg1, rpos, in);
1885     	arg2 = analyze_expression(e->arg2, rpos, arg1.out);
1886     
1887     	/* Got rid of lub -- when would lub make sense? */
1888     	if (mkeq_qtype(e->loc, arg1.qt, arg2.qt))
1889     	  report_qerror(e->loc, sev_err,
1890     			"incompatible operands of ``%s''",
1891     			binary_op_name(e->kind));
1892     	return mkeinfo(arg1.qt, arg2.out, NULL, NULL, FALSE);
1893           }
1894           break;
1895         case kind_andand:
1896           {
1897     	einfo arg1, arg2;
1898     	qtype qt;
1899     
1900     	/* arg1 always evaluated */
1901     	arg1 = analyze_expression(e->arg1, cpos, in);
1902     
1903     	/* arg2 only evaluated if arg1 true, and it's evaluated in
1904     	   our context */
1905     	arg2 = analyze_expression(e->arg2, context, true_out_einfo(arg1));
1906     
1907     	qt = mkqtype_bool(make_qvar("__boolop", e->loc, FALSE));
1908     
1909     	if (context == cpos)
1910     	  return mkeinfo(qt, NULL,
1911     			 true_out_einfo(arg2),
1912     			 lub_store(e->loc,
1913     				   "inconsistent stores for false &&",
1914     				   false_out_einfo(arg1),
1915     				   false_out_einfo(arg2)),
1916     			 FALSE);
1917     	else
1918     	  return mkeinfo(qt,
1919     			 lub_store(e->loc,
1920     				   "inconsistent stores for &&",
1921     				   false_out_einfo(arg1),
1922     				   arg2.out),
1923     			 NULL, NULL, FALSE);
1924           }
1925           break;
1926         case kind_oror:
1927           {
1928     	einfo arg1, arg2;
1929     	qtype qt;
1930     
1931     	/* arg1 always evaluated */
1932     	arg1 = analyze_expression(e->arg1, cpos, in);
1933     
1934     	/* arg2 only evaluated if arg1 false, and it's evaluated in
1935     	   our context */
1936     	arg2 = analyze_expression(e->arg2, context, false_out_einfo(arg1));
1937     
1938     	qt = mkqtype_bool(make_qvar("__boolop", e->loc, FALSE));
1939     
1940     	if (context == cpos)
1941     	  return mkeinfo(qt, NULL,
1942     			 lub_store(e->loc,
1943     				   "inconsistent stores for true ||",
1944     				   true_out_einfo(arg1),
1945     				   true_out_einfo(arg2)),
1946     			 false_out_einfo(arg2),
1947     			 FALSE);
1948     	else
1949     	  return mkeinfo(qt,
1950     			 lub_store(e->loc,
1951     				   "inconsistent stores for ||",
1952     				   true_out_einfo(arg1),
1953     				   arg2.out),
1954     			 NULL, NULL, FALSE);
1955           }
1956           break;
1957         case kind_eq:
1958         case kind_ne:
1959           {
1960     	qtype qt;
1961     
1962     	qt = mkqtype_bool(make_qvar("__boolop", e->loc, FALSE));
1963     	if (context == cpos && have_stream_quals &&
1964     	    ((file_pointer_qtype(e->arg1->qtype) && definite_null(e->arg2)) ||
1965     	     (definite_null(e->arg1) && file_pointer_qtype(e->arg2->qtype))))
1966     	  {
1967     	    /* May be comparison of file pointer to NULL */
1968     	    einfo fp, null;
1969     
1970     	    /* XXX: Analyze null operand first; this changes the
1971     	       left-to-right order of evaluation */
1972     	    if (file_pointer_qtype(e->arg1->qtype))
1973     	      {
1974     		null = analyze_expression(e->arg2, rpos, in);
1975     		fp = analyze_expression(e->arg1, cpos, null.out);
1976     	      }
1977     	    else
1978     	      {
1979     		null = analyze_expression(e->arg1, rpos, in);
1980     		fp = analyze_expression(e->arg2, cpos, null.out);
1981     	      }
1982     
1983     	    if (e->kind == kind_eq)
1984     	      /* NULL on TRUE branch */
1985     	      return mkeinfo(qt, NULL, false_out_einfo(fp),
1986     			     true_out_einfo(fp), FALSE);
1987     	    else
1988     	      /* NULL on FALSE branch */
1989     	      return mkeinfo(qt, NULL, true_out_einfo(fp),
1990     			     false_out_einfo(fp), FALSE);
1991     	  }
1992     	else
1993     	  {
1994     	    /* Ordinary comparsion */
1995     	    einfo arg1, arg2;
1996     
1997     	    arg1 = analyze_expression(e->arg1, rpos, in);
1998     	    arg2 = analyze_expression(e->arg2, rpos, arg1.out);
1999     	    return mkeinfo(qt, arg2.out, NULL, NULL, FALSE);
2000     	  }
2001           }
2002           break;
2003         case kind_leq:
2004         case kind_geq:
2005         case kind_lt:
2006         case kind_gt:
2007           {
2008     	einfo arg1, arg2;
2009     
2010     	arg1 = analyze_expression(e->arg1, rpos, in);
2011     	arg2 = analyze_expression(e->arg2, rpos, arg1.out);
2012     	return mkeinfo(mkqtype_bool(make_qvar("__boolop", e->loc, FALSE)),
2013     		       arg2.out, NULL, NULL, FALSE);
2014           }
2015           break;
2016         case kind_modulo_assign:
2017         case kind_lshift_assign:
2018         case kind_rshift_assign:
2019           {
2020     	einfo arg1, arg2;
2021     	qtype qt;
2022     
2023     	arg1 = analyze_expression(e->arg1, lpos, in);
2024     	arg2 = analyze_expression(e->arg2, rpos, arg1.out);
2025     	if (!qtype_error(arg1.qt))
2026     	  qt = points_to_fs_qtype(arg1.qt, arg2.out);
2027     	else
2028     	  qt = error_qtype;
2029     	return mkeinfo(qt, arg2.out, NULL, NULL, FALSE);
2030           }
2031           break;
2032         case kind_plus_assign:
2033         case kind_minus_assign:
2034         case kind_times_assign:
2035         case kind_divide_assign:
2036         case kind_bitand_assign:
2037         case kind_bitor_assign:
2038         case kind_bitxor_assign:
2039         case kind_assign:
2040           {
2041     	einfo arg1, arg2;
2042     	qtype qt;
2043     	store out;
2044     
2045     	arg1 = analyze_expression(e->arg1, lpos, in);
2046     	arg2 = analyze_expression(e->arg2, rpos, arg1.out);
2047     	out = arg2.out;
2048     
2049     	if (arg2.ismalloc)
2050     	  {
2051     	    out = allocate_qtype(e->loc, points_to_qtype(e->arg1->qtype), out);
2052     	    if (flag_casts_warn)
2053     	      /* No interaction.  This prevents no-qual zone warning for
2054     		 x = malloc(...).  Also, this has the side effect of
2055     		 qualifier on magic_malloc being untouched, but it
2056     		 doesn't matter because magic_malloc is unique. */
2057     	      {
2058     		qt = points_to_fs_qtype(arg1.qt, out);
2059     		return mkeinfo(qt, out, NULL, NULL, FALSE);
2060     	      }
2061     	  }
2062     
2063     	if (!qtype_error(arg1.qt))
2064     	  {
2065     	    qt = points_to_fs_qtype(arg1.qt, out);
2066     	    if ((e->kind == kind_plus_assign ||
2067     		 e->kind == kind_minus_assign) &&
2068     		qtype_pointer_loc(qt) && !qtype_pointer_loc(arg2.qt))
2069     	      /* In p +/-= count, no interaction between p and count's
2070     		 qualifiers */
2071     	      return mkeinfo(qt, out, NULL, NULL, FALSE);
2072     
2073     	    out = assign_flow_sensitive(e->loc,
2074     					"incompatible types in assignment",
2075     					arg2.qt, arg1.qt, out, FALSE);
2076     
2077     	    if (flag_print_lin)
2078     	      queue_print_exp((expression) e, out, aloc_qtype(arg1.qt));
2079     	  }
2080     	else
2081     	  qt = error_qtype;
2082     
2083     	if (have_stream_quals && context == cpos &&
2084     	    file_pointer_fs_qtype(qt, out))
2085     	  return split_file_pointer_context(e->loc, qt, out);
2086     	else
2087     	  return mkeinfo(qt, out, NULL, NULL, FALSE);
2088           }
2089           break;
2090         default:
2091           fail_loc(e->loc, "Unexpected binary op kind 0x%x\n", e->kind);
2092         }
2093     }

2157     /* finds the smallest enclosing scope */
2158     static compound_stmt get_enclosing_scope(compound_stmt cs1, compound_stmt cs2)
2159     {
2160       compound_stmt temp;
2161     
2162       if (cs1 == cs2) return cs1; /* get the easy case out of the door */
2163       
2164       temp = cs1;
2165       while (temp != NULL)
2166         {
2167           temp->visited = TRUE;
2168           temp = temp->enclosing_cs;
2169         }
2170     
2171       temp = cs2;
2172       while (temp != NULL)
2173         {
2174           if (temp->visited == TRUE)
2175     	/* found */
2176     	{
2177     	  /* clean up */
2178     	  while (cs1 != NULL) 
2179     	    { cs1->visited = FALSE; cs1 = cs1->enclosing_cs; }
2180     	  while (cs2->visited) 
2181     	    { cs2->visited = FALSE; cs2 = cs2->enclosing_cs; }
2182     	  
2183     	  return temp;
2184     	}
2185           temp = temp->enclosing_cs;
2186         }
2187     
2188       /* toplvl match...this is ok since there should be no inter-procedural
2189          jumps */
2190       /* clean up */
2191       while (cs1 != NULL) 
2192         { cs1->visited = FALSE; cs1 = cs1->enclosing_cs; }
2193       while (cs2->visited) 
2194         { cs2->visited = FALSE; cs2 = cs2->enclosing_cs; }
2195     
2196       return NULL;
2197     }

2257     {

2260       enclosing_cs = get_enclosing_scope(from, to);

2420     /* If fc is a magic function, do a special analysis.  Otherwise return
2421        an einfo with NULL as out */
2422     static einfo analyze_magic_function_call(function_call fc, qtype fqt, store in)
2423     {
2424       if (fc->arg1->kind == kind_identifier)
2425         {
2426           identifier id = CAST(identifier, fc->arg1);
2427           if ((!strcmp(id->cstring.data, "fopen") ||
2428     	   !strcmp(id->cstring.data, "fdopen") ||
2429     	   !strcmp(id->cstring.data, "tmpfile") ||
2430     	   !strcmp(id->cstring.data, "safefopen") ||
2431     	   !strcmp(id->cstring.data, "popen") ||
2432     	   !strcmp(id->cstring.data, "my_popen"))
2433     	  && have_stream_quals)
2434     	{
2435     	  store cur_store;
2436     	  qual mode_qual;
2437     	  expression e;
2438     	  qtype ret_qtype;
2439     	  int num_args;
2440     
2441     	  cur_store = in;
2442     	  mode_qual = NULL;
2443     
2444     	  /* Process the function arguments for any side effects */
2445     	  num_args = 0;
2446     	  scan_expression(e, fc->args)
2447     	    {
2448     	      einfo arg_info = analyze_expression(e, rpos, cur_store);
2449     	      cur_store = arg_info.out;
2450     	      num_args++;
2451     	    }

2453     	  /* Extract the mode */

2608     static einfo split_file_pointer_context(location loc, qtype qt, store s)
2609     {
2610       aloc al;
2611       qtype old_qt, new_non_null_qt, new_null_qt;
2612       store non_null_store, null_store;
2613     
2614       assert(have_stream_quals && file_pointer_fs_qtype(qt, s));
2615       al = aloc_qtype(qt);
2616       old_qt = qtype_from_store(s, al);
2617       new_non_null_qt = copy_qtype(old_qt, loc);
2618       new_null_qt = copy_qtype(old_qt, loc);
2619     
2620       insistnot(cond_mkleq_qual(loc,
2621     			    read_unchecked_qual, qual_qtype(old_qt),
2622     			    read_qual, qual_qtype(new_non_null_qt)));
2623       insistnot(cond_mkleq_qual(loc,
2624     			    write_unchecked_qual, qual_qtype(old_qt),
2625     			    write_qual, qual_qtype(new_non_null_qt)));
2626       insistnot(mkleq_qual(loc, closed_qual,
2627     		       qual_qtype(new_null_qt)));
2628     
2629       non_null_store = 
2630         make_store_assign(loc,
2631     		      "strong update failure in fopen result test",
2632     		      s, al, new_non_null_qt, TRUE);
2633     
2634       null_store =
2635         make_store_assign(loc,
2636     		      "strong update failure in fopen result test",
2637     		      s, al, new_null_qt, TRUE);
2638     
2639       return mkeinfo(qt, NULL, non_null_store, null_store, FALSE);
