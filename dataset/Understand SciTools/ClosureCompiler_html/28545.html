142         /** List implementation for {@link JsArray}. */
143         private static class JsArrayList<T> extends AbstractList<T> {
144           final JsArray<T> array;
145           JsArrayList(JsArray<T> array) {
146             this.array = array;
147           }
148     
149           void checkBounds(int index) {
150             if (index < 0 || index >= array.getLength()) {
151               throw new IndexOutOfBoundsException();
152             }
153           }
154     
155           @Override public T get(int index) {
156             checkBounds(index);
157             return array.get(index);
158           }
159     
160           @Override public T set(int index, T elem) {
161             checkBounds(index);
162             T prev = array.get(index);
163             array.set(index, elem);
164             return prev;
165           }
166     
167           @Override public T remove(int index) {
168             checkBounds(index);
169             T prev = array.get(index);
170             array.splice(index, 1);
171             return prev;
172           }
173     
174           @Override public boolean add(T elem) {
175             array.push(elem);
176             return true;
177           }
178     
179           @Override public int size() {
180             return array.getLength();
181           }
182         }
183       }
