17     package com.google.javascript.jscomp.lint;
18     
19     import com.google.javascript.jscomp.CheckLevel;
20     import com.google.javascript.jscomp.Compiler;
21     import com.google.javascript.jscomp.CompilerOptions;
22     import com.google.javascript.jscomp.CompilerPass;
23     import com.google.javascript.jscomp.CompilerTestCase;
24     import com.google.javascript.jscomp.DiagnosticGroups;
25     
26     /**
27      * Test case for {@link CheckInterfaces}.
28      */
29     public final class CheckInterfacesTest extends CompilerTestCase {
30       @Override
31       public CompilerPass getProcessor(Compiler compiler) {
32         return new CheckInterfaces(compiler);
33       }
34     
35       @Override
36       protected CompilerOptions getOptions(CompilerOptions options) {
37         super.getOptions(options);
38         options.setWarningLevel(DiagnosticGroups.LINT_CHECKS, CheckLevel.WARNING);
39         return options;
40       }
41     
42       public void setUp() {
43         enableTypeCheck();
44       }
45     
46       public void testInterfaceArgs() throws Exception {
47         testSame("/** @interface */ function A(x) {}",
48             CheckInterfaces.INTERFACE_SHOULD_NOT_TAKE_ARGS);
49     
50         testSame("var ns = {};\n"
51             + "/** @interface */\n"
52             + "ns.SomeInterface = function(x) {};",
53             CheckInterfaces.INTERFACE_SHOULD_NOT_TAKE_ARGS);
54       }
55     
56       public void testInterfaceNotEmpty() {
57         testSame("/** @interface */ function A() { this.foo; }",
58             CheckInterfaces.INTERFACE_FUNCTION_NOT_EMPTY);
59     
60         testSame("var ns = {};\n"
61             + "/** @interface */\n"
62             + "ns.SomeInterface = function() { this.foo; };",
63             CheckInterfaces.INTERFACE_FUNCTION_NOT_EMPTY);
64       }
65     }
