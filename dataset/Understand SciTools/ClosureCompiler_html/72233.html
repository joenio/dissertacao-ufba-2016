31     /**
32      * Callback that gathers subexpressions that may have side effects
33      * and appends copies of those subexpressions to the replacements
34      * list.  In the case of branching subexpressions, it simplifies the
35      * subexpression before adding it to the replacement list.
36      *
37      */
38     class GatherSideEffectSubexpressionsCallback implements Callback {
39     
40       /**
41        * Used by GatherSideEffectSubexpressionsCallback to notify client
42        * code about side effect expressions that should be kept.
43        */
44       interface SideEffectAccumulator {
45     
46         /**
47          * Returns true if the "mixin" and "inherits" function calls
48          * should be treated as if they had side effects.
49          */
50         boolean classDefiningCallsHaveSideEffects();
51     
52         /**
53          * Adds subtree to the list of nodes that have side effects.
54          *
55          * @param original - root of the tree.
56          */
57         void keepSubTree(Node original);
58     
59         /**
60          * Simplifies a subtree whose root node is an AND or OR expression
61          * and adds the resulting subtree to the list of nodes that have
62          * side effects.
63          *
64          * @param original - root of the and/or expression.
65          */
66         void keepSimplifiedShortCircuitExpression(Node original);
67     
68         /**
69          * Simplifies a subtree whose root node is a HOOK expression
70          * and adds the resulting subtree to the list of nodes that have
71          * side effects.
72          *
73          * @param hook - root of the hook expression.
74          * @param thenHasSideEffects - then branch has side effects
75          * @param elseHasSideEffects - else branch has side effects
76          */
77         void keepSimplifiedHookExpression(Node hook,
78                                           boolean thenHasSideEffects,
79                                           boolean elseHasSideEffects);
80       }
81     
82       /**
83        * Populates the provided replacement list by appending copies of
84        * subtrees that have side effects.
85        *
86        * It is OK if this class tears up the original tree, because
87        * we're going to throw the tree out anyway.
88        */
89       static final class GetReplacementSideEffectSubexpressions
90           implements SideEffectAccumulator {
91         private final AbstractCompiler compiler;
92         private final List<Node> replacements;
93     
94         /**
95          * Creates the accumulator.
96          *
97          * @param compiler - the AbstractCompiler
98          * @param replacements - list to accumulate into
99          */
100         GetReplacementSideEffectSubexpressions(AbstractCompiler compiler,
101             List<Node> replacements) {
102           this.compiler = compiler;
103           this.replacements = replacements;
104         }
105     
106         @Override
107         public boolean classDefiningCallsHaveSideEffects() {
108           return true;
109         }
110     
111         @Override
112         public void keepSubTree(Node original) {
113           if (original.getParent() != null) {
114             original.detachFromParent();
115           }
116           replacements.add(original);
117         }
118     
119         @Override
120         public void keepSimplifiedShortCircuitExpression(Node original) {
121           Preconditions.checkArgument(
122               (original.isAnd()) || (original.isOr()),
123               "Expected: AND or OR, Got: %s",
124               original.getType());
125           Node left = original.getFirstChild();
126           Node right = left.getNext();
127           Node simplifiedRight = simplifyShortCircuitBranch(right);
128           original.detachChildren();
129           original.addChildToBack(left);
130           original.addChildToBack(simplifiedRight);
131           keepSubTree(original);
132         }
133     
134         @Override
135         public void keepSimplifiedHookExpression(Node hook,
136                                                  boolean thenHasSideEffects,
137                                                  boolean elseHasSideEffects) {
138           Preconditions.checkArgument(hook.isHook(), "Expected: HOOK, Got: %s", hook.getType());
139           Node condition = hook.getFirstChild();
140           Node thenBranch = condition.getNext();
141           Node elseBranch = thenBranch.getNext();
142           if (thenHasSideEffects && elseHasSideEffects) {
143             hook.detachChildren();
144             hook.addChildToBack(condition);
145             hook.addChildToBack(simplifyShortCircuitBranch(thenBranch));
146             hook.addChildToBack(simplifyShortCircuitBranch(elseBranch));
147             keepSubTree(hook);
148           } else if (thenHasSideEffects || elseHasSideEffects) {
149             Token type = thenHasSideEffects ? Token.AND : Token.OR;
150             Node body = thenHasSideEffects ? thenBranch : elseBranch;
151             Node simplified = new Node(
152                 type, condition.detachFromParent(),
153                 simplifyShortCircuitBranch(body))
154                 .useSourceInfoIfMissingFrom(hook);
155             keepSubTree(simplified);
156           } else {
157             throw new IllegalArgumentException(
158                 "keepSimplifiedHookExpression must keep at least 1 branch");
159           }
160         }
161     
162         private Node simplifyShortCircuitBranch(Node node) {
163           List<Node> parts = new ArrayList<>();
164           NodeTraversal.traverseEs6(
165               compiler, node,
166               new GatherSideEffectSubexpressionsCallback(
167                   compiler,
168                   new GetReplacementSideEffectSubexpressions(compiler, parts)));
169     
170           Node ret = null;
171           for (Node part : parts) {
172             if (ret != null) {
173               ret = IR.comma(ret, part).srcref(node);
174             } else {
175               ret = part;
176             }
177           }
178     
179           if (ret == null) {
180             throw new IllegalArgumentException(
181                 "expected at least one side effect subexpression in short "
182                 + "circuit branch.");
183           }
184     
185           return ret;
186         }
187       }

189       private static final Set<Token> FORBIDDEN_TYPES = ImmutableSet.of(
190           Token.BLOCK, Token.SCRIPT, Token.VAR, Token.EXPR_RESULT, Token.RETURN);
191       private final AbstractCompiler compiler;
192       private final SideEffectAccumulator accumulator;
193     
194       /**
195        * @param compiler - AbstractCompiler object
196        * @param accumulator - object that will accumulate roots of
197        *                      subtrees that have side effects.
198        */
199       GatherSideEffectSubexpressionsCallback(AbstractCompiler compiler,
200                                              SideEffectAccumulator accumulator) {
201         this.compiler = compiler;
202         this.accumulator = accumulator;
203       }
204     
205       /**
206        * Determines if a call defines a class inheritance or mixing
207        * relation, according to the current coding convention.
208        */
209       private boolean isClassDefiningCall(Node callNode) {
210         SubclassRelationship classes =
211             compiler.getCodingConvention().getClassesDefinedByCall(callNode);
212         return classes != null;
213       }
214     
215       /**
216        * Computes the list of subtrees whose root nodes have side effects.
217        *
218        * <p>If the current subtree's root has side effects this method should
219        * call accumulator.keepSubTree and return 'false' to add the
220        * subtree to the result list and avoid avoid traversing the nodes children.
221        *
222        * <p>Branching nodes whose then or else branch contain side effects
223        * must be simplified by doing a recursive traversal; this method
224        * should call the appropriate accumulator 'keepSimplified' method
225        * and return 'false' to stop the regular traversal.
226        */
227       @Override
228       public boolean shouldTraverse(
229           NodeTraversal traversal, Node node, Node parent) {
230         if (FORBIDDEN_TYPES.contains(node.getType()) ||
231             NodeUtil.isControlStructure(node)) {
232           throw new IllegalArgumentException(node.getType() + " nodes are not supported.");
233         }
234     
235         // Do not recurse into nested functions.
236         if (node.isFunction()) {
237           return false;
238         }
239     
240         // simplify and maybe keep hook expression.
241         if (node.isHook()) {
242           return processHook(node);
243         }
244     
245         // simplify and maybe keep AND/OR expression.
246         if ((node.isAnd()) || (node.isOr())) {
247           return processShortCircuitExpression(node);
248         }
249     
250         if (!NodeUtil.nodeTypeMayHaveSideEffects(node, compiler)) {
251           return true;
252         } else {
253     
254           // Node type suggests that the expression has side effects.
255     
256           if (node.isCall()) {
257             return processFunctionCall(node);
258           } else if (node.isNew()) {
259             return processConstructorCall(node);
260           } else {
261             accumulator.keepSubTree(node);
262             return false;
263           }
264         }
265       }
266     
267       /**
268        * Processes an AND or OR expression.
269        *
270        * @return true to continue traversal, false otherwise
271        */
272       boolean processShortCircuitExpression(Node node) {
273         Preconditions.checkArgument(
274             (node.isAnd()) || (node.isOr()), "Expected: AND or OR, Got: %s", node.getType());
275     
276         // keep whole expression if RHS of the branching expression
277         // contains a call.
278         Node left = node.getFirstChild();
279         Node right = left.getNext();
280         if (NodeUtil.mayHaveSideEffects(right, compiler)) {
281           accumulator.keepSimplifiedShortCircuitExpression(node);
282           return false;
283         } else {
284           return true;
285         }
286       }
287     
288       /**
289        * Processes a HOOK expression.
290        *
291        * @return true to continue traversal, false otherwise
292        */
293       boolean processHook(Node node) {
294         Preconditions.checkArgument(node.isHook(), "Expected: HOOK, Got: %s", node.getType());
295     
296         Node condition = node.getFirstChild();
297         Node ifBranch = condition.getNext();
298         Node elseBranch = ifBranch.getNext();
299         boolean thenHasSideEffects = NodeUtil.mayHaveSideEffects(
300             ifBranch, compiler);
301         boolean elseHasSideEffects = NodeUtil.mayHaveSideEffects(
302             elseBranch, compiler);
303         if (thenHasSideEffects || elseHasSideEffects) {
304           accumulator.keepSimplifiedHookExpression(
305               node, thenHasSideEffects, elseHasSideEffects);
306           return false;
307         } else {
308           return true;
309         }
310       }
311     
312       /**
313        * Processes a CALL expression.
314        *
315        * @return true to continue traversal, false otherwise
316        */
317       boolean processFunctionCall(Node node) {
318         Preconditions.checkArgument(node.isCall(), "Expected: CALL, Got: %s", node.getType());
319     
320         // Calls to functions that are known to be "pure" have no side
321         // effects.
322         Node functionName = node.getFirstChild();
323         if (functionName.isName() || functionName.isGetProp()) {
324           if (!accumulator.classDefiningCallsHaveSideEffects() &&
325               isClassDefiningCall(node)) {
326             return true;
327           }
328         }
329     
330         if (!NodeUtil.functionCallHasSideEffects(node)) {
331           return true;
332         }
333     
334         accumulator.keepSubTree(node);
335         return false;
336       }
337     
338       /**
339        * Processes a NEW expression.
340        *
341        * @return true to continue traversal, false otherwise
342        */
343       boolean processConstructorCall(Node node) {
344         Preconditions.checkArgument(node.isNew(), "Expected: NEW, Got: %s", node.getType());
345     
346         // Calls to constructors that are known to be "pure" have no
347         // side effects.
348         if (!NodeUtil.constructorCallHasSideEffects(node)) {
349           return true;
350         }
351     
352         accumulator.keepSubTree(node);
353         return false;
354       }
355     
356       @Override
357       public void visit(NodeTraversal traversal, Node node, Node parent) {}
358     }
