17     package com.google.javascript.jscomp;
18     
19     import com.google.gwt.core.client.GWT;
20     import com.google.gwt.resources.client.ClientBundle;
21     import com.google.gwt.resources.client.TextResource;
22     
23     import jsinterop.annotations.JsMethod;
24     
25     /**
26      * GWT compatible replacement for {@code ResourceLoader}
27      *
28      */
29     final class ResourceLoader {
30       static interface Libraries extends ClientBundle {
31         Libraries INSTANCE = GWT.create(Libraries.class);
32     
33         // This is a generated file containing all the text resources we want to package
34         // as a single JSON string mapping (relative) filename to file content strings.
35         @Source("resources.json")
36         TextResource resources();
37       }
38     
39       private static final JsObject RESOURCES = parse(Libraries.INSTANCE.resources().getText());
40     
41       static String loadTextResource(Class<?> clazz, String path) {
42         String content = get(RESOURCES, path);
43         if (content != null) {
44           return content;
45         }
46         throw new RuntimeException("Resource not found " + path);
47       }
48     
49       static boolean resourceExists(Class<?> clazz, String path) {
50         // TODO(sdh): this is supposed to be relative to the given class, but
51         // GWT can't handle that - probably better to remove the class argument
52         // and just require that paths be relative to c.g.javascript.jscomp.
53         return get(RESOURCES, path) != null;
54       }
55     
56       public static class JsObject {
57         public JsObject() {}
58       }
59     
60       private static native String get(JsObject obj, String key) /*-{
61         return obj[key];
62       }-*/;
63     
64       @JsMethod(namespace = "JSON")
65       private static native JsObject parse(String json);
66     }
