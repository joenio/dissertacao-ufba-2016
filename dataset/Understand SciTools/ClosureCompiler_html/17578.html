17     package com.google.javascript.jscomp.parsing.parser;
18     
19     import com.google.javascript.jscomp.parsing.parser.util.SourcePosition;
20     import com.google.javascript.jscomp.parsing.parser.util.SourceRange;
21     
22     import java.util.ArrayList;
23     import java.util.Arrays;
24     
25     /**
26      * Maps offsets into a source string into line/column positions.
27      *
28      * Immutable.
29      */
30     public class LineNumberTable {
31     
32       private final SourceFile sourceFile;
33       private final int[] lineStartOffsets;
34     
35       public LineNumberTable(SourceFile sourceFile) {
36         this.sourceFile = sourceFile;
37         this.lineStartOffsets = computeLineStartOffsets(sourceFile.contents);
38       }
39     
40       private static int[] computeLineStartOffsets(String source) {
41         // TODO(johnlenz): do this more efficiently.
42         ArrayList<Integer> lineStartOffsets = new ArrayList<>();
43         lineStartOffsets.add(0);
44         for (int index = 0; index < source.length(); index++) {
45           char ch = source.charAt(index);
46           if (isLineTerminator(ch)) {
47             if (index + 1 < source.length() && ch == '\r'
48                 && source.charAt(index + 1) == '\n') {
49               index++;
50             }
51             lineStartOffsets.add(index + 1);
52           }
53         }
54         lineStartOffsets.add(Integer.MAX_VALUE);
55         return toIntArray(lineStartOffsets);
56       }
57     
58       public static int[] toIntArray(ArrayList<Integer> integers) {
59           int[] ret = new int[integers.size()];
60           for (int i = 0; i < ret.length; i++) {
61               ret[i] = integers.get(i).intValue();
62           }
63           return ret;
64       }
65     
66       private static boolean isLineTerminator(char ch) {
67         switch (ch) {
68         case '\n': // Line Feed
69         case '\r':  // Carriage Return
70         case '\u2028':  // Line Separator
71         case '\u2029':  // Paragraph Separator
72           return true;
73         default:
74           return false;
75         }
76       }
77     
78       public SourcePosition getSourcePosition(int offset) {
79         int line = getLine(offset);
80         return new SourcePosition(sourceFile, offset, line, getColumn(line, offset));
81       }
82     
83       public int getLine(int offset) {
84         int index = Arrays.binarySearch(lineStartOffsets, offset);
85         // start of line
86         if (index >= 0) {
87           return index;
88         }
89         return -index - 2;
90       }
91     
92       public int offsetOfLine(int line) {
93         return lineStartOffsets[line];
94       }
95     
96       private int getColumn(int line, int offset) {
97         return offset - offsetOfLine(line);
98       }
99     
100       public SourceRange getSourceRange(int startOffset, int endOffset) {
101         return new SourceRange(getSourcePosition(startOffset), getSourcePosition(endOffset));
102       }
103     }
