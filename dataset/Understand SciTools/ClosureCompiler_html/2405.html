34     /**
35      * Replace known jQuery aliases and methods with standard
36      * conventions so that the compiler recognizes them. Expected
37      * replacements include:
38      *  - jQuery.fn -> jQuery.prototype
39      *  - jQuery.extend -> expanded into direct object assignments
40      *  - jQuery.expandedEach -> expand into direct assignments
41      *
42      * @author chadkillingsworth@missouristate.edu (Chad Killingsworth)
43      */
44     class ExpandJqueryAliases extends AbstractPostOrderCallback
45         implements CompilerPass {
46       private final AbstractCompiler compiler;
47       private final CodingConvention convention;
48       private static final Logger logger =
49           Logger.getLogger(ExpandJqueryAliases.class.getName());
50     
51       static final DiagnosticType JQUERY_UNABLE_TO_EXPAND_INVALID_LIT_ERROR =
52           DiagnosticType.warning("JSC_JQUERY_UNABLE_TO_EXPAND_INVALID_LIT",
53               "jQuery.expandedEach call cannot be expanded because the first " +
54               "argument must be an object literal or an array of strings " +
55               "literal.");
56     
57       static final DiagnosticType JQUERY_UNABLE_TO_EXPAND_INVALID_NAME =
58           DiagnosticType.error("JSC_JQUERY_UNABLE_TO_EXPAND_INVALID_NAME",
59               "jQuery.expandedEach expansion would result in an invalid property name.");
60     
61       static final DiagnosticType JQUERY_UNABLE_TO_EXPAND_INVALID_NAME_WITH_NAME =
62           DiagnosticType.error("JSC_JQUERY_UNABLE_TO_EXPAND_INVALID_NAME_WITH_NAME",
63               "jQuery.expandedEach expansion would result in the invalid " +
64               "property name \"{0}\".");
65     
66       static final DiagnosticType JQUERY_USELESS_EACH_EXPANSION =
67           DiagnosticType.warning("JSC_JQUERY_USELESS_EACH_EXPANSION",
68               "jQuery.expandedEach was not expanded as no valid property " +
69               "assignments were encountered. Consider using jQuery.each instead.");

71       private static final Set<String> JQUERY_EXTEND_NAMES = ImmutableSet.of(
72           "jQuery.extend", "jQuery.fn.extend", "jQuery.prototype.extend");
73     
74       private static final String JQUERY_EXPANDED_EACH_NAME =
75           "jQuery.expandedEach";

77       private final PeepholeOptimizationsPass peepholePasses;

79       ExpandJqueryAliases(AbstractCompiler compiler) {
80         this.compiler = compiler;
81         this.convention = compiler.getCodingConvention();
82     
83         // All of the "early" peephole optimizations.
84         // These passes should make the code easier to analyze.
85         // Passes, such as StatementFusion, are omitted for this reason.
86         final boolean late = false;
87         boolean useTypesForOptimization = compiler.getOptions().useTypesForOptimization;
88         this.peepholePasses = new PeepholeOptimizationsPass(compiler,
89             new PeepholeMinimizeConditions(late, useTypesForOptimization),
90             new PeepholeSubstituteAlternateSyntax(late),
91             new PeepholeReplaceKnownMethods(late),
92             new PeepholeRemoveDeadCode(),
93             new PeepholeFoldConstants(late, useTypesForOptimization),
94             new PeepholeCollectPropertyAssignments());
95       }

97       /**
98        * Check that Node n is a call to one of the jQuery.extend methods that we
99        * can expand. Valid calls are single argument calls where the first argument
100        * is an object literal or two argument calls where the first argument
101        * is a name and the second argument is an object literal.
102        */
103       public static boolean isJqueryExtendCall(Node n, String qname,
104           AbstractCompiler compiler) {
105         if (JQUERY_EXTEND_NAMES.contains(qname)) {
106           Node firstArgument = n.getNext();
107           if (firstArgument == null) {
108             return false;
109           }
110     
111           Node secondArgument = firstArgument.getNext();
112           if ((firstArgument.isObjectLit() && secondArgument == null) ||
113               (firstArgument.isName() || NodeUtil.isGet(firstArgument) &&
114               !NodeUtil.mayHaveSideEffects(firstArgument, compiler) &&
115               secondArgument != null && secondArgument.isObjectLit() &&
116               secondArgument.getNext() == null)) {
117             return true;
118           }
119         }
120         return false;
121       }
122     
123       public boolean isJqueryExpandedEachCall(Node call, String qName) {
124         Preconditions.checkArgument(call.isCall());
125         return call.getFirstChild() != null && JQUERY_EXPANDED_EACH_NAME.equals(qName);
126       }

128       @Override
129       public void visit(NodeTraversal t, Node n, Node parent) {
130         if (n.isGetProp() && convention.isPrototypeAlias(n)) {
131           maybeReplaceJqueryPrototypeAlias(n);
132     
133         } else if (n.isCall()) {
134           Node callTarget = n.getFirstChild();
135           String qName = callTarget.getQualifiedName();
136     
137           if (isJqueryExtendCall(callTarget, qName, this.compiler)) {
138             maybeExpandJqueryExtendCall(n);
139     
140           } else if (isJqueryExpandedEachCall(n, qName)) {
141             maybeExpandJqueryEachCall(t, n);
142           }
143         }
144       }

153       private void maybeReplaceJqueryPrototypeAlias(Node n) {
154         // Check to see if this is the assignment of the original alias.
155         // If so, leave it intact.
156         if (NodeUtil.isLValue(n)) {
157           Node maybeAssign = n.getParent();
158           while (!NodeUtil.isStatement(maybeAssign) && !maybeAssign.isAssign()) {
159             maybeAssign = maybeAssign.getParent();
160           }
161     
162           if (maybeAssign.isAssign()) {
163             maybeAssign = maybeAssign.getParent();
164             if (maybeAssign.isBlock() || maybeAssign.isScript() ||
165                 NodeUtil.isStatement(maybeAssign)) {
166               return;
167             }
168           }
169         }
170     
171         Node fn = n.getLastChild();
172         if (fn != null) {
173           n.replaceChild(fn, IR.string("prototype").srcref(fn));
174           compiler.reportCodeChange();
175         }
176       }

178       /**
179        * Expand jQuery.extend (and derivative) calls into direct object assignments
180        * Example: jQuery.extend(obj1, {prop1: val1, prop2: val2}) ->
181        *   obj1.prop1 = val1;
182        *   obj1.prop2 = val2;
183        */
184       private void maybeExpandJqueryExtendCall(Node n) {
185         Node callTarget = n.getFirstChild();
186         Node objectToExtend = callTarget.getNext(); // first argument
187         Node extendArg = objectToExtend.getNext(); // second argument
188         boolean ensureObjectDefined = true;
189     
190         if (extendArg == null) {
191           // Only one argument was specified, so extend jQuery namespace
192           extendArg = objectToExtend;
193           objectToExtend = callTarget.getFirstChild();
194           ensureObjectDefined = false;
195         } else if (objectToExtend.isGetProp() &&
196               (objectToExtend.getLastChild().getString().equals("prototype") ||
197               convention.isPrototypeAlias(objectToExtend))) {
198           ensureObjectDefined = false;
199         }
200     
201         // Check for an empty object literal
202         if (!extendArg.hasChildren()) {
203           return;
204         }
205     
206         // Since we are expanding jQuery.extend calls into multiple statements,
207         // encapsulate the new statements in a new block.
208         Node fncBlock = IR.block().srcref(n);
209     
210         if (ensureObjectDefined) {
211           Node assignVal = IR.or(objectToExtend.cloneTree(),
212               IR.objectlit().srcref(n)).srcref(n);
213           Node assign = IR.assign(objectToExtend.cloneTree(), assignVal).srcref(n);
214           fncBlock.addChildrenToFront(IR.exprResult(assign).srcref(n));
215         }
216     
217         while (extendArg.hasChildren()) {
218           Node currentProp = extendArg.removeFirstChild();
219           Node propValue;
220           if (currentProp.hasChildren()) {
221             propValue = currentProp.getLastChild().detachFromParent();
222           } else {
223             propValue = IR.name(currentProp.getString()).srcref(currentProp);
224           }
225     
226           Node newProp;
227           if (currentProp.isQuotedString()) {
228             newProp = IR.getelem(objectToExtend.cloneTree(),
229                 currentProp).srcref(currentProp);
230           } else if (currentProp.isComputedProp()) {
231             Node childOfcompProp = currentProp.removeFirstChild();
232             newProp = IR.getelem(objectToExtend.cloneTree(),
233                 childOfcompProp).srcref(currentProp);
234           } else {
235             currentProp.setType(Token.STRING);
236             newProp = IR.getprop(objectToExtend.cloneTree(),
237                 currentProp).srcref(currentProp);
238           }
239     
240           Node assignNode = IR.assign(newProp, propValue).srcref(currentProp);
241           fncBlock.addChildToBack(IR.exprResult(assignNode).srcref(currentProp));
242         }
243     
244         // Check to see if the return value is used. If not, replace the original
245         // call with new block. Otherwise, wrap the statements in an
246         // immediately-called anonymous function.
247         if (n.getParent().isExprResult()) {
248           Node parent = n.getParent();
249           parent.getParent().replaceChild(parent, fncBlock);
250         } else {
251           Node targetVal;
252           if ("jQuery.prototype".equals(objectToExtend.getQualifiedName())) {
253             // When extending the jQuery prototype, return the jQuery namespace.
254             // This is not commonly used.
255             targetVal = objectToExtend.removeFirstChild();
256           } else {
257             targetVal = objectToExtend.detachFromParent();
258           }
259           fncBlock.addChildToBack(IR.returnNode(targetVal).srcref(targetVal));
260     
261           Node fnc = IR.function(IR.name("").srcref(n),
262               IR.paramList().srcref(n),
263               fncBlock).srcref(n);
264     
265           // add an explicit "call" statement so that we can maintain
266           // the same reference for "this"
267           Node newCallTarget = IR.getprop(
268               fnc, IR.string("call").srcref(n)).srcref(n);
269           n.replaceChild(callTarget, newCallTarget);
270           n.putBooleanProp(Node.FREE_CALL, false);
271     
272           // remove any other pre-existing call arguments
273           while (newCallTarget.getNext() != null) {
274             n.removeChildAfter(newCallTarget);
275           }
276           n.addChildToBack(IR.thisNode().srcref(n));
277         }
278         compiler.reportCodeChange();
279       }

281       /**
282        * Expand a jQuery.expandedEach call
283        *
284        * Expanded jQuery.expandedEach calls will replace the GETELEM nodes of a
285        * property assignment with GETPROP nodes to allow for renaming.
286        */
287       private void maybeExpandJqueryEachCall(NodeTraversal t, Node n) {
288         Node objectToLoopOver = n.getSecondChild();
289     
290         if (objectToLoopOver == null) {
291           return;
292         }
293     
294         Node callbackFunction = objectToLoopOver.getNext();
295         if (callbackFunction == null || !callbackFunction.isFunction()) {
296           return;
297         }
298     
299         // Run the peephole optimizations on the first argument to handle
300         // cases like ("a " + "b").split(" ")
301         peepholePasses.process(null, n.getSecondChild());
302     
303         // Create a reference tree
304         Node nClone = n.cloneTree();
305     
306         objectToLoopOver = nClone.getSecondChild();
307     
308         // Check to see if the first argument is something we recognize and can
309         // expand.
310         if (!objectToLoopOver.isObjectLit() &&
311             !(objectToLoopOver.isArrayLit() &&
312             isArrayLitValidForExpansion(objectToLoopOver))) {
313           t.report(n, JQUERY_UNABLE_TO_EXPAND_INVALID_LIT_ERROR, (String) null);
314           return;
315         }
316     
317         // Find all references to the callback function arguments
318         List<Node> keyNodeReferences = new ArrayList<>();
319         List<Node> valueNodeReferences = new ArrayList<>();
320     
321         new NodeTraversal(
322             compiler,
323             new FindCallbackArgumentReferences(callbackFunction,
324                 keyNodeReferences, valueNodeReferences,
325                 objectToLoopOver.isArrayLit()))
326             .traverseInnerNode(
327                 NodeUtil.getFunctionBody(callbackFunction), callbackFunction, t.getScope());
328     
329         if (keyNodeReferences.isEmpty()) {
330          // We didn't do anything useful ...
331           t.report(n, JQUERY_USELESS_EACH_EXPANSION, (String) null);
332           return;
333         }
334     
335         Node fncBlock = tryExpandJqueryEachCall(t, nClone, callbackFunction,
336             keyNodeReferences, valueNodeReferences);
337     
338         if (fncBlock != null && fncBlock.hasChildren()) {
339             replaceOriginalJqueryEachCall(n, fncBlock);
340         } else {
341           // We didn't do anything useful ...
342           t.report(n, JQUERY_USELESS_EACH_EXPANSION, (String) null);
343         }
344       }

346       private Node tryExpandJqueryEachCall(NodeTraversal t, Node n,
347           Node callbackFunction, List<Node> keyNodes, List<Node> valueNodes) {
348     
349         Node callTarget = n.getFirstChild();
350         Node objectToLoopOver = callTarget.getNext();
351     
352         // New block to contain the expanded statements
353         Node fncBlock = IR.block().srcref(callTarget);
354     
355         boolean isValidExpansion = true;
356     
357         // Expand the jQuery.expandedEach call
358         Node key = objectToLoopOver.getFirstChild(), val = null;
359         for (int i = 0; key != null; key = key.getNext(), i++) {
360           if (key != null) {
361             if (objectToLoopOver.isArrayLit()) {
362               // Arrays have a value of their index number
363               val = IR.number(i).srcref(key);
364             } else {
365               val = key.getFirstChild();
366               if (val == null) {
367                 val = IR.name(key.getString());
368               }
369             }
370           }
371     
372           // Keep track of the replaced nodes so we can reset the tree
373           List<Node> newKeys = new ArrayList<>();
374           List<Node> newValues = new ArrayList<>();
375           List<Node> origGetElems = new ArrayList<>();
376           List<Node> newGetProps = new ArrayList<>();
377     
378           // Replace all of the key nodes with the prop name
379           for (int j = 0; j < keyNodes.size(); j++) {
380             if (key.isComputedProp()) {
381               t.report(key, JQUERY_UNABLE_TO_EXPAND_INVALID_NAME);
382               return null;
383             }
384             Node origNode = keyNodes.get(j);
385             Node ancestor = origNode.getParent();
386     
387             Node newNode = IR.string(key.getString()).srcref(key);
388             newKeys.add(newNode);
389             ancestor.replaceChild(origNode, newNode);
390     
391             // Walk up the tree to see if the key is used in a GETELEM
392             // assignment
393             while (ancestor != null && !NodeUtil.isStatement(ancestor) &&
394                 !ancestor.isGetElem()) {
395               ancestor = ancestor.getParent();
396             }
397     
398             // Convert GETELEM nodes to GETPROP nodes so that they can be
399             // renamed or removed.
400             if (ancestor != null && ancestor.isGetElem()) {
401     
402               Node propObject = ancestor;
403               while (propObject.isGetProp() || propObject.isGetElem()) {
404                 propObject = propObject.getFirstChild();
405               }
406     
407               Node ancestorClone = ancestor.cloneTree();
408               // Run the peephole passes to handle cases such as
409               // obj['lit' + key] = val;
410               peepholePasses.process(null, ancestorClone.getSecondChild());
411               Node prop = ancestorClone.getSecondChild();
412     
413               if (prop.isString() &&
414                 NodeUtil.isValidPropertyName(LanguageMode.ECMASCRIPT3, prop.getString())) {
415                 Node target = ancestorClone.getFirstChild();
416                 Node newGetProp = IR.getprop(target.detachFromParent(),
417                     prop.detachFromParent());
418                 newGetProps.add(newGetProp);
419                 origGetElems.add(ancestor);
420                 ancestor.getParent().replaceChild(ancestor, newGetProp);
421               } else {
422                 if (prop.isString() &&
423                     !NodeUtil.isValidPropertyName(LanguageMode.ECMASCRIPT3, prop.getString())) {
424                   t.report(n,
425                       JQUERY_UNABLE_TO_EXPAND_INVALID_NAME_WITH_NAME,
426                       prop.getString());
427                 }
428                 isValidExpansion = false;
429               }
430             }
431           }
432     
433           if (isValidExpansion) {
434             // Replace all of the value nodes with the prop value
435             for (int j = 0; val != null && j < valueNodes.size(); j++) {
436               Node origNode = valueNodes.get(j);
437               Node newNode = val.cloneTree();
438               newValues.add(newNode);
439               origNode.getParent().replaceChild(origNode, newNode);
440             }
441     
442             // Wrap the new tree in an anonymous function call
443             Node fnc = IR.function(IR.name("").srcref(key),
444                 IR.paramList().srcref(key),
445                 callbackFunction.getChildAtIndex(2).cloneTree()).srcref(key);
446             Node call = IR.call(fnc).srcref(key);
447             call.putBooleanProp(Node.FREE_CALL, true);
448             fncBlock.addChildToBack(IR.exprResult(call).srcref(call));
449           }
450     
451           // Reset the source tree
452           for (int j = 0; j < newGetProps.size(); j++) {
453             newGetProps.get(j).getParent().replaceChild(newGetProps.get(j),
454                 origGetElems.get(j));
455           }
456           for (int j = 0; j < newKeys.size(); j++) {
457             newKeys.get(j).getParent().replaceChild(newKeys.get(j),
458                 keyNodes.get(j));
459           }
460           for (int j = 0; j < newValues.size(); j++) {
461             newValues.get(j).getParent().replaceChild(newValues.get(j),
462                 valueNodes.get(j));
463           }
464     
465           if (!isValidExpansion) {
466             return null;
467           }
468         }
469         return fncBlock;
470       }

472       private void replaceOriginalJqueryEachCall(Node n, Node expandedBlock) {
473         // Check to see if the return value of the original jQuery.expandedEach
474         // call is used. If so, we need to wrap each loop expansion in an anonymous
475         // function and return the original objectToLoopOver.
476         if (n.getParent().isExprResult()) {
477           Node parent = n.getParent();
478           Node grandparent = parent.getParent();
479           Node insertAfter = parent;
480           while (expandedBlock.hasChildren()) {
481             Node child = expandedBlock.getFirstChild().detachFromParent();
482             grandparent.addChildAfter(child, insertAfter);
483             insertAfter = child;
484           }
485           grandparent.removeChild(parent);
486         } else {
487           // Return the original object
488           Node callTarget = n.getFirstChild();
489           Node objectToLoopOver = callTarget.getNext();
490     
491           objectToLoopOver.detachFromParent();
492           Node ret = IR.returnNode(objectToLoopOver).srcref(callTarget);
493           expandedBlock.addChildToBack(ret);
494     
495           // Wrap all of the expanded loop calls in a new anonymous function
496           Node fnc = IR.function(IR.name("").srcref(callTarget),
497               IR.paramList().srcref(callTarget),
498               expandedBlock);
499           n.replaceChild(callTarget, fnc);
500           n.putBooleanProp(Node.FREE_CALL, true);
501     
502           // remove any other pre-existing call arguments
503           while (fnc.getNext() != null) {
504             n.removeChildAfter(fnc);
505           }
506         }
507         compiler.reportCodeChange();
508       }

510       private boolean isArrayLitValidForExpansion(Node n) {
511         Iterator<Node> iter = n.children().iterator();
512         while (iter.hasNext()) {
513           Node child = iter.next();
514           if (!child.isString()) {
515             return false;
516           }
517         }
518         return true;
519       }

521       /**
522        * Given a jQuery.expandedEach callback function, traverse it and collect any
523        * references to its parameter names.
524        */
525       static class FindCallbackArgumentReferences extends AbstractPostOrderCallback
526           implements ScopedCallback {
527     
528         private final String keyName;
529         private final String valueName;
530         private Scope startingScope;
531         private List<Node> keyReferences;
532         private List<Node> valueReferences;
533     
534         FindCallbackArgumentReferences(Node functionRoot, List<Node> keyReferences,
535             List<Node> valueReferences, boolean useArrayMode) {
536           Preconditions.checkState(functionRoot.isFunction());
537     
538           String keyString = null, valueString = null;
539           Node callbackParams = NodeUtil.getFunctionParameters(functionRoot);
540           Node param = callbackParams.getFirstChild();
541           if (param != null) {
542             Preconditions.checkState(param.isName());
543             keyString = param.getString();
544     
545             param = param.getNext();
546             if (param != null) {
547               Preconditions.checkState(param.isName());
548               valueString = param.getString();
549             }
550           }
551     
552           this.keyName = keyString;
553           this.valueName = valueString;
554     
555           // For arrays, the keyString is the index number of the element.
556           // We're interested in the value of the element instead
557           if (useArrayMode) {
558             this.keyReferences = valueReferences;
559             this.valueReferences = keyReferences;
560           } else {
561             this.keyReferences = keyReferences;
562             this.valueReferences = valueReferences;
563           }
564     
565           this.startingScope = null;
566         }
567     
568         private boolean isShadowed(String name, Scope scope) {
569           Var nameVar = scope.getVar(name);
570           return nameVar != null &&
571               nameVar.getScope() != this.startingScope;
572         }
573     
574         @Override
575         public void visit(NodeTraversal t, Node n, Node parent) {
576           // In the top scope, "this" is a reference to "value"
577           boolean isThis = false;
578           if (t.getClosestHoistScope() == this.startingScope) {
579             isThis = n.isThis();
580           }
581     
582           if (isThis || n.isName() && !isShadowed(n.getString(), t.getScope())) {
583             String nodeValue = isThis ? null : n.getString();
584             if (!isThis && keyName != null && nodeValue.equals(keyName)) {
585               keyReferences.add(n);
586             } else if (isThis || (valueName != null &&
587                 nodeValue.equals(valueName))) {
588               valueReferences.add(n);
589             }
590           }
591         }
592     
593         /**
594          * As we enter each scope, make sure that the scope doesn't define
595          * a local variable with the same name as our original callback method
596          * parameter names.
597          */
598         @Override
599         public void enterScope(NodeTraversal t) {
600           if (this.startingScope == null) {
601             this.startingScope = t.getScope();
602           }
603         }
604     
605         @Override
606         public void exitScope(NodeTraversal t) { }
607       }
