29     /**
30      * Set the JSDocInfo on all types.
31      *
32      * Propagates JSDoc across the type graph, but not across the symbol graph.
33      * This means that if you have:
34      * <code>
35      * var x = new Foo();
36      * x.bar;
37      * </code>
38      * then the JSType attached to x.bar may get associated JSDoc, but the
39      * Node and Var will not.
40      *
41      * JSDoc is initially attached to AST Nodes at parse time.
42      * There are 3 ways that JSDoc get propagated across the type system.
43      * 1) Nominal types (e.g., constructors) may contain JSDocInfo for their
44      *    declaration.
45      * 2) Object types have a JSDocInfo slot for each property on that type.
46      * 3) Shape types (like structural functions) may have JSDocInfo.
47      *
48      * #1 and #2 should be self-explanatory, and non-controversial. #3 is
49      * a bit trickier. It means that if you have:
50      * <code>
51      * /** @param {number} x /
52      * Foo.prototype.bar = goog.abstractMethod;
53      * </code>
54      * the JSDocInfo will appear in two places in the type system: in the 'bar'
55      * slot of Foo.prototype, and on the function expression type created by
56      * this expression.
57      *
58      * @author nicksantos@google.com (Nick Santos)
59      */
60     class InferJSDocInfo extends AbstractPostOrderCallback
61         implements HotSwapCompilerPass {
62       private final AbstractCompiler compiler;
63     
64       InferJSDocInfo(AbstractCompiler compiler) {
65         this.compiler = compiler;
66       }
67     
68       @Override
69       public void process(Node externs, Node root) {
70         if (externs != null) {
71           NodeTraversal.traverseEs6(compiler, externs, this);
72         }
73         if (root != null) {
74           NodeTraversal.traverseEs6(compiler, root, this);
75         }
76       }
77     
78       @Override
79       public void hotSwapScript(Node root, Node originalRoot) {
80         Preconditions.checkNotNull(root);
81         Preconditions.checkState(root.isScript());
82         NodeTraversal.traverseEs6(compiler, root, this);
83       }
84     
85       @Override
86       public void visit(NodeTraversal t, Node n, Node parent) {
87         JSDocInfo docInfo;
88     
89         switch (n.getType()) {
90           // Infer JSDocInfo on types of all type declarations on variables.
91           case NAME:
92             if (parent == null) {
93               return;
94             }
95     
96             // Only allow JSDoc on VARs, function declarations, and assigns.
97             if (!parent.isVar() &&
98                 !NodeUtil.isFunctionDeclaration(parent) &&
99                 !(parent.isAssign() &&
100                   n == parent.getFirstChild())) {
101               return;
102             }
103     
104             // There are four places the doc info could live.
105             // 1) A FUNCTION node.
106             // /** ... */ function f() { ... }
107             // 2) An ASSIGN parent.
108             // /** ... */ x = function () { ... }
109             // 3) A NAME parent.
110             // var x, /** ... */ y = function() { ... }
111             // 4) A VAR grandparent.
112             // /** ... */ var x = function() { ... }
113             docInfo = n.getJSDocInfo();
114             if (docInfo == null &&
115                 !(parent.isVar() &&
116                     !parent.hasOneChild())) {
117               docInfo = parent.getJSDocInfo();
118             }
119     
120             // Try to find the type of the NAME.
121             JSType varType = n.getJSType();
122             if (varType == null && parent.isFunction()) {
123               varType = parent.getJSType();
124             }
125     
126             // If we have no type to attach JSDocInfo to, then there's nothing
127             // we can do.
128             if (varType == null || docInfo == null) {
129               return;
130             }
131     
132             // Dereference the type. If the result is not an object, or already
133             // has docs attached, then do nothing.
134             ObjectType objType = dereferenceToObject(varType);
135             if (objType == null || objType.getJSDocInfo() != null) {
136               return;
137             }
138     
139             attachJSDocInfoToNominalTypeOrShape(objType, docInfo, n.getString());
140             break;
141     
142           case STRING_KEY:
143           case GETTER_DEF:
144           case SETTER_DEF:
145             docInfo = n.getJSDocInfo();
146             if (docInfo == null) {
147               return;
148             }
149             ObjectType owningType = dereferenceToObject(parent.getJSType());
150             if (owningType != null) {
151               String propName = n.getString();
152               if (owningType.hasOwnProperty(propName)) {
153                 owningType.setPropertyJSDocInfo(propName, docInfo);
154               }
155             }
156             break;
157     
158           case GETPROP:
159             // Infer JSDocInfo on properties.
160             // There are two ways to write doc comments on a property.
161             //
162             // 1)
163             // /** @deprecated */
164             // obj.prop = ...
165             //
166             // 2)
167             // /** @deprecated */
168             // obj.prop;
169             if (parent.isExprResult() ||
170                 (parent.isAssign() &&
171                  parent.getFirstChild() == n)) {
172               docInfo = n.getJSDocInfo();
173               if (docInfo == null) {
174                 docInfo = parent.getJSDocInfo();
175               }
176               if (docInfo != null) {
177                 ObjectType lhsType =
178                     dereferenceToObject(n.getFirstChild().getJSType());
179                 if (lhsType != null) {
180                   // Put the JSDoc in the property slot, if there is one.
181                   String propName = n.getLastChild().getString();
182                   if (lhsType.hasOwnProperty(propName)) {
183                     lhsType.setPropertyJSDocInfo(propName, docInfo);
184                   }
185     
186                   // Put the JSDoc in any constructors or function shapes as well.
187                   ObjectType propType =
188                       dereferenceToObject(lhsType.getPropertyType(propName));
189                   if (propType != null) {
190                     attachJSDocInfoToNominalTypeOrShape(
191                         propType, docInfo, n.getQualifiedName());
192                   }
193                 }
194               }
195             }
196             break;
197         }
198       }
199     
200       /**
201        * Dereferences the given type to an object, or returns null.
202        */
203       private static ObjectType dereferenceToObject(JSType type) {
204         return ObjectType.cast(type == null ? null : type.dereference());
205       }
206     
207       /**
208        * Handle cases #1 and #3 in the class doc.
209        */
210       private static void attachJSDocInfoToNominalTypeOrShape(
211           ObjectType objType, JSDocInfo docInfo, @Nullable String qName) {
212         if (objType.isConstructor() ||
213             objType.isEnumType() ||
214             objType.isInterface()) {
215           // Named types.
216           if (objType.hasReferenceName() &&
217               objType.getReferenceName().equals(qName)) {
218             objType.setJSDocInfo(docInfo);
219     
220             if (objType.isConstructor() || objType.isInterface()) {
221               JSType.toMaybeFunctionType(objType).getInstanceType().setJSDocInfo(docInfo);
222             } else if (objType instanceof EnumType) {
223               ((EnumType) objType).getElementsType().setJSDocInfo(docInfo);
224             }
225           }
226         } else if (!objType.isNativeObjectType() && objType.isFunctionType()) {
227           // Structural functions.
228           objType.setJSDocInfo(docInfo);
229         }
230       }
231     }
