39     package com.google.javascript.rhino.jstype;
40     
41     import com.google.javascript.rhino.testing.BaseJSTypeTestCase;
42     
43     public class PrototypeObjectTypeTest extends BaseJSTypeTestCase {
44     
45       public void testToString() {
46         ObjectType type = registry.createAnonymousObjectType(null);
47         assertEquals("{}", type.toString());
48     
49         type.defineDeclaredProperty("foo", NUMBER_TYPE, null);
50         assertEquals("{foo: number}", type.toString());
51     
52         type.defineDeclaredProperty("bar", type, null);
53         assertEquals("{bar: {...}, foo: number}", type.toString());
54       }
55     }
