17     package com.google.javascript.jscomp.graph;
18     
19     import com.google.common.base.Preconditions;
20     import com.google.javascript.jscomp.graph.DiGraph.DiGraphEdge;
21     import com.google.javascript.jscomp.graph.DiGraph.DiGraphNode;
22     
23     import java.util.HashSet;
24     import java.util.LinkedHashSet;
25     import java.util.List;
26     import java.util.Set;
27     
28     /**
29      * A utility class for doing fixed-point computations. We traverse
30      * the edges over the given directed graph until the graph reaches
31      * a steady state.
32      *
33      * @author nicksantos@google.com (Nick Santos)
34      *
35      * @param <N> Value type that the graph node stores.
36      * @param <E> Value type that the graph edge stores.
37      */
38     public final class FixedPointGraphTraversal<N, E> {
39       // TODO(nicksantos): Generalize the algorithm for undirected graphs, if we
40       // need it.
41     
42       private final EdgeCallback<N, E> callback;
43     
44       public static final String NON_HALTING_ERROR_MSG =
45         "Fixed point computation not halting";
46     
47       /**
48        * Create a new traversal.
49        * @param callback A callback for updating the state of the graph each
50        *     time an edge is traversed.
51        */
52       public FixedPointGraphTraversal(EdgeCallback<N, E> callback) {
53         this.callback = callback;
54       }
55     
56       /**
57        * Helper method for creating new traversals.
58        */
59       public static <NODE, EDGE> FixedPointGraphTraversal<NODE, EDGE> newTraversal(
60           EdgeCallback<NODE, EDGE> callback) {
61         return new FixedPointGraphTraversal<>(callback);
62       }
63     
64       /**
65        * Compute a fixed point for the given graph.
66        * @param graph The graph to traverse.
67        */
68       public void computeFixedPoint(DiGraph<N, E> graph) {
69         Set<N> nodes = new HashSet<>();
70         for (DiGraphNode<N, E> node : graph.getDirectedGraphNodes()) {
71           nodes.add(node.getValue());
72         }
73         computeFixedPoint(graph, nodes);
74       }
75     
76       /**
77        * Compute a fixed point for the given graph, entering from the given node.
78        * @param graph The graph to traverse.
79        * @param entry The node to begin traversing from.
80        */
81       public void computeFixedPoint(DiGraph<N, E> graph, N entry) {
82         Set<N> entrySet = new HashSet<>();
83         entrySet.add(entry);
84         computeFixedPoint(graph, entrySet);
85       }
86     
87       /**
88        * Compute a fixed point for the given graph, entering from the given nodes.
89        * @param graph The graph to traverse.
90        * @param entrySet The nodes to begin traversing from.
91        */
92       public void computeFixedPoint(DiGraph<N, E> graph, Set<N> entrySet) {
93         int cycleCount = 0;
94         long nodeCount = graph.getNodes().size();
95     
96         // Choose a bail-out heuristically in case the computation
97         // doesn't converge.
98         long maxIterations = Math.max(nodeCount * nodeCount * nodeCount, 100);
99     
100         // Use a LinkedHashSet, so that the traversal is deterministic.
101         LinkedHashSet<DiGraphNode<N, E>> workSet =
102              new LinkedHashSet<>();
103         for (N n : entrySet) {
104           workSet.add(graph.getDirectedGraphNode(n));
105         }
106         for (; !workSet.isEmpty() && cycleCount < maxIterations; cycleCount++) {
107           // For every out edge in the workSet, traverse that edge. If that
108           // edge updates the state of the graph, then add the destination
109           // node to the resultSet, so that we can update all of its out edges
110           // on the next iteration.
111           DiGraphNode<N, E> source = workSet.iterator().next();
112           N sourceValue = source.getValue();
113     
114           workSet.remove(source);
115     
116           List<DiGraphEdge<N, E>> outEdges = source.getOutEdges();
117           for (DiGraphEdge<N, E> edge : outEdges) {
118             N destNode = edge.getDestination().getValue();
119             if (callback.traverseEdge(sourceValue, edge.getValue(), destNode)) {
120               workSet.add(edge.getDestination());
121             }
122           }
123         }
124     
125         Preconditions.checkState(cycleCount != maxIterations,
126             NON_HALTING_ERROR_MSG);
127       }
128     
129       /** Edge callback */
130       public static interface EdgeCallback<Node, Edge> {
131         /**
132          * Update the state of the destination node when the given edge
133          * is traversed. For the fixed-point computation to work, only the
134          * destination node may be modified. The source node and the edge must
135          * not be modified.
136          *
137          * @param source The start node.
138          * @param e The edge.
139          * @param destination The end node.
140          * @return Whether the state of the destination node changed.
141          */
142         boolean traverseEdge(Node source, Edge e, Node destination);
143       }
144     }
