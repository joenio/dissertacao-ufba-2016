40     package com.google.javascript.rhino.jstype;
41     
42     import com.google.common.collect.ImmutableSortedMap;
43     import com.google.javascript.rhino.Node;
44     
45     /**
46      * A builder for record types.
47      *
48      */
49     public class RecordTypeBuilder {
50       private boolean isEmpty = true;
51       private boolean isDeclared = true;
52       private final JSTypeRegistry registry;
53       private final ImmutableSortedMap.Builder<String, RecordProperty> properties =
54           ImmutableSortedMap.naturalOrder();
55     
56       public RecordTypeBuilder(JSTypeRegistry registry) {
57         this.registry = registry;
58       }
59     
60       /** See the comments on RecordType about synthetic types. */
61       void setSynthesized(boolean synthesized) {
62         isDeclared = !synthesized;
63       }
64     
65       /**
66        * Adds a property with the given name and type to the record type.
67        * If you add a property that has already been added, then {@link #build}
68        * will fail.
69        *
70        * @param name the name of the new property
71        * @param type the JSType of the new property
72        * @param propertyNode the node that holds this property definition
73        * @return The builder itself for chaining purposes.
74        */
75       public RecordTypeBuilder addProperty(String name, JSType type, Node
76           propertyNode) {
77         isEmpty = false;
78         properties.put(name, new RecordProperty(type, propertyNode));
79         return this;
80       }
81     
82       /**
83        * Creates a record. Fails if any duplicate property names were added.
84        * @return The record type.
85        */
86       public JSType build() {
87          // If we have an empty record, simply return the object type.
88         if (isEmpty) {
89            return registry.getNativeObjectType(JSTypeNative.OBJECT_TYPE);
90         }
91     
92         return new RecordType(registry, properties.build(), isDeclared);
93       }
94     
95       static class RecordProperty {
96         private final JSType type;
97         private final Node propertyNode;
98     
99         RecordProperty(JSType type, Node propertyNode) {
100           this.type = type;
101           this.propertyNode = propertyNode;
102         }
103     
104         public JSType getType() {
105           return type;
106         }
107     
108         public Node getPropertyNode() {
109           return propertyNode;
110         }
111       }
112     }
