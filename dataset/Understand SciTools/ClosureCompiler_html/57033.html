17     package com.google.javascript.jscomp.deps;
18     
19     import com.google.common.base.Function;
20     import com.google.common.base.Joiner;
21     import com.google.common.collect.ImmutableMap;
22     import com.google.common.collect.Iterables;
23     
24     import java.io.IOException;
25     import java.util.ArrayList;
26     import java.util.Collection;
27     import java.util.List;
28     import java.util.Map;
29     
30     /**
31      * A data structure for JS dependency information for a single .js file.
32      *
33      * @author agrieve@google.com (Andrew Grieve)
34      */
35     public interface DependencyInfo {
36     
37       /** Gets the unique name / path of this file. */
38       public String getName();
39     
40       /** Gets the path of this file relative to Closure's base.js file. */
41       public String getPathRelativeToClosureBase();
42     
43       /** Gets the symbols provided by this file. */
44       public Collection<String> getProvides();
45     
46       /** Gets the symbols required by this file. */
47       public Collection<String> getRequires();
48     
49       /** Gets the loading information for this file. */
50       public ImmutableMap<String, String> getLoadFlags();
51     
52       /** Whether the symbol is provided by a module */
53       public boolean isModule();
54     
55       /**
56        * Abstract base implementation that defines derived accessors such
57        * as {@link #isModule}.
58        */
59       public abstract class Base implements DependencyInfo {
60         @Override public boolean isModule() {
61           return "goog".equals(getLoadFlags().get("module"));
62         }
63       }
64     
65       /** Utility methods. */
66       class Util {
67         private Util() {}
68     
69         // TODO(sdh): This would be better as a defender method once Java 8 is allowed (b/28382956):
70         //     void DependencyInfo#writeAddDependency(Appendable);
71         /** Prints a goog.addDependency call for a single DependencyInfo. */
72         public static void writeAddDependency(Appendable out, DependencyInfo info) throws IOException {
73           out.append("goog.addDependency('")
74               .append(info.getPathRelativeToClosureBase())
75               .append("', ");
76           writeJsArray(out, info.getProvides());
77           out.append(", ");
78           writeJsArray(out, info.getRequires());
79           Map<String, String> loadFlags = info.getLoadFlags();
80           if (!loadFlags.isEmpty()) {
81             out.append(", ");
82             writeJsObject(out, loadFlags);
83           }
84           out.append(");\n");
85         }
86     
87         /** Prints a map as a JS object literal. */
88         private static void writeJsObject(Appendable out, Map<String, String> map) throws IOException {
89           List<String> entries = new ArrayList<>();
90           for (Map.Entry<String, String> entry : map.entrySet()) {
91             String key = entry.getKey().replace("'", "\\'");
92             String value = entry.getValue().replace("'", "\\'");
93             entries.add("'" + key + "': '" + value + "'");
94           }
95           out.append("{");
96           out.append(Joiner.on(", ").join(entries));
97           out.append("}");
98         }
99     
100         /** Prints a list of strings formatted as a JavaScript array of string literals. */
101         private static void writeJsArray(Appendable out, Collection<String> values) throws IOException {
102           Iterable<String> quoted =
103               Iterables.transform(
104                   values,
105                   new Function<String, String>() {
106                     @Override public String apply(String arg) {
107                       return "'" + arg.replace("'", "\\'") + "'";
108                     }
109                   });
110           out.append("[");
111           out.append(Joiner.on(", ").join(quoted));
112           out.append("]");
113         }
114       }
115     }
