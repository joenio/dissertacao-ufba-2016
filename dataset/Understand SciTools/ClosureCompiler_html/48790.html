37     /**
38      * A framework to help writing static program analysis. A subclass of
39      * this framework should specify how a single node changes the state
40      * of a program. This class finds a safe estimate (a fixed-point) for
41      * the whole program. The proven facts about the program will be
42      * annotated with
43      * {@link com.google.javascript.jscomp.graph.GraphNode#setAnnotation} to the
44      * given control flow graph's nodes in form of {@link LatticeElement}
45      * after calling {@link #analyze()}.
46      *
47      * <p>As a guideline, the following is a list of behaviors that any analysis
48      * can take:
49      * <ol>
50      * <li>Flow Direction: Is the analysis a forward or backward analysis?
51      * <li>Lattice Elements: How does the analysis represent the state of the
52      * program at any given point?
53      * <li>JOIN Operation: Given two incoming paths and a lattice state value, what
54      * can the compiler conclude at the join point?
55      * <li>Flow Equations: How does an instruction modify the state of program in
56      * terms of lattice values?
57      * <li>Initial Entry Value: What can the compiler assume at the beginning of the
58      * program?
59      * <li>Initial Estimate: What can the compiler assume at each point of the
60      * program? (What is the BOTTOM value of the lattice) By definition this lattice
61      * JOIN {@code x} for any {@code x} must also be {@code x}.
62      * </ol>
63      * To make these behaviors known to the framework, the following steps must be
64      * taken.
65      * <ol>
66      * <li>Flow Direction: Implement {@link #isForward()}.
67      * <li>Lattice Elements: Implement {@link LatticeElement}.
68      * <li>JOIN Operation: Implement
69      *    {@link JoinOp#apply}.
70      * <li>Flow Equations: Implement
71      * {@link #flowThrough(Object, LatticeElement)}.
72      * <li>Initial Entry Value: Implement {@link #createEntryLattice()}.
73      * <li>Initial Estimate: Implement {@link #createInitialEstimateLattice()}.
74      * </ol>
75      *
76      * <p>Upon execution of the {@link #analyze()} method, nodes of the input
77      * control flow graph will be annotated with a {@link FlowState} object that
78      * represents maximum fixed point solution. Any previous annotations at the
79      * nodes of the control flow graph will be lost.
80      *
81      *
82      * @param <N> The control flow graph's node value type.
83      * @param <L> Lattice element type.
84      */
85     abstract class DataFlowAnalysis<N, L extends LatticeElement> {
86     
87       private final ControlFlowGraph<N> cfg;
88       final JoinOp<L> joinOp;
89       protected final Set<DiGraphNode<N, Branch>> orderedWorkSet;

91       /*
92        * Feel free to increase this to a reasonable number if you are finding that
93        * more and more passes need more than 400000 steps before finding a
94        * fixed-point. If you just have a special case, consider calling
95        * {@link #analyse(int)} instead.
96        */
97       public static final int MAX_STEPS = 400000;

99       /**
100        * Constructs a data flow analysis.
101        *
102        * <p>Typical usage
103        * <pre>
104        * DataFlowAnalysis dfa = ...
105        * dfa.analyze();
106        * </pre>
107        *
108        * {@link #analyze()} annotates the result to the control flow graph by
109        * means of {@link DiGraphNode#setAnnotation} without any
110        * modification of the graph itself. Additional calls to {@link #analyze()}
111        * recomputes the analysis which can be useful if the control flow graph
112        * has been modified.
113        *
114        * @param targetCfg The control flow graph object that this object performs
115        *     on. Modification of the graph requires a separate call to
116        *     {@link #analyze()}.
117        *
118        * @see #analyze()
119        */
120       DataFlowAnalysis(ControlFlowGraph<N> targetCfg, JoinOp<L> joinOp) {
121         this.cfg = targetCfg;
122         this.joinOp = joinOp;
123         Comparator<DiGraphNode<N, Branch>> nodeComparator =
124           cfg.getOptionalNodeComparator(isForward());
125         if (nodeComparator != null) {
126           this.orderedWorkSet = new TreeSet<>(nodeComparator);
127         } else {
128           this.orderedWorkSet = new LinkedHashSet<>();
129         }
130       }
131     
132       /**
133        * Returns the control flow graph that this analysis was performed on.
134        * Modifications can be done on this graph, however, the only time that the
135        * annotations are correct is after {@link #analyze()} is called and before
136        * the graph has been modified.
137        */
138       final ControlFlowGraph<N> getCfg() {
139         return cfg;
140       }

142       /**
143        * Returns the lattice element at the exit point.
144        */
145       L getExitLatticeElement() {
146         DiGraphNode<N, Branch> node = getCfg().getImplicitReturn();
147         FlowState<L> state = node.getAnnotation();
148         return state.getIn();
149       }

173       /**
174        * Finds a fixed-point solution using at most {@link #MAX_STEPS}
175        * iterations.
176        *
177        * @see #analyze(int)
178        */
179       final void analyze() {
180         analyze(MAX_STEPS);
181       }

183       /**
184        * Finds a fixed-point solution. The function has the side effect of replacing
185        * the existing node annotations with the computed solutions using {@link
186        * com.google.javascript.jscomp.graph.GraphNode#setAnnotation(Annotation)}.
187        *
188        * <p>Initially, each node's input and output flow state contains the value
189        * given by {@link #createInitialEstimateLattice()} (with the exception of the
190        * entry node of the graph which takes on the {@link #createEntryLattice()}
191        * value. Each node will use the output state of its predecessor and compute a
192        * output state according to the instruction. At that time, any nodes that
193        * depends on the node's newly modified output value will need to recompute
194        * their output state again. Each step will perform a computation at one node
195        * until no extra computation will modify any existing output state anymore.
196        *
197        * @param maxSteps Max number of iterations before the method stops and throw
198        *        a {@link MaxIterationsExceededException}. This will prevent the
199        *        analysis from going into a infinite loop.
200        */
201       final void analyze(int maxSteps) {
202         initialize();
203         int step = 0;
204         while (!orderedWorkSet.isEmpty()) {
205           if (step > maxSteps) {
206             throw new MaxIterationsExceededException(
207               "Analysis did not terminate after " + maxSteps + " iterations");
208           }
209           DiGraphNode<N, Branch> curNode = orderedWorkSet.iterator().next();
210           orderedWorkSet.remove(curNode);
211           joinInputs(curNode);
212           if (flow(curNode)) {
213             // If there is a change in the current node, we want to grab the list
214             // of nodes that this node affects.
215             List<DiGraphNode<N, Branch>> nextNodes = isForward() ?
216                 cfg.getDirectedSuccNodes(curNode) :
217                 cfg.getDirectedPredNodes(curNode);
218             for (DiGraphNode<N, Branch> nextNode : nextNodes) {
219               if (nextNode != cfg.getImplicitReturn()) {
220                 orderedWorkSet.add(nextNode);
221               }
222             }
223           }
224           step++;
225         }
226         if (isForward()) {
227           joinInputs(getCfg().getImplicitReturn());
228         }
229       }

245       /**
246        * Initializes the work list and the control flow graph.
247        */
248       protected void initialize() {
249         // TODO(user): Calling clear doesn't deallocate the memory in a
250         // LinkedHashSet. Consider creating a new work set if we plan to repeatedly
251         // call analyze.
252         orderedWorkSet.clear();
253         for (DiGraphNode<N, Branch> node : cfg.getDirectedGraphNodes()) {
254           node.setAnnotation(new FlowState<>(createInitialEstimateLattice(),
255               createInitialEstimateLattice()));
256           if (node != cfg.getImplicitReturn()) {
257             orderedWorkSet.add(node);
258           }
259         }
260       }
261     
262       /**
263        * Performs a single flow through a node.
264        *
265        * @return {@code true} if the flow state differs from the previous state.
266        */
267       protected boolean flow(DiGraphNode<N, Branch> node) {
268         FlowState<L> state = node.getAnnotation();
269         if (isForward()) {
270           L outBefore = state.out;
271           state.out = flowThrough(node.getValue(), state.in);
272           return !outBefore.equals(state.out);
273         } else {
274           L inBefore = state.in;
275           state.in = flowThrough(node.getValue(), state.out);
276           return !inBefore.equals(state.in);
277         }
278       }

280       /**
281        * Computes the new flow state at a given node's entry by merging the
282        * output (input) lattice of the node's predecessor (successor).
283        *
284        * @param node Node to compute new join.
285        */
286       protected void joinInputs(DiGraphNode<N, Branch> node) {
287         FlowState<L> state = node.getAnnotation();
288         if (isForward()) {
289           if (cfg.getEntry() == node) {
290             state.setIn(createEntryLattice());
291           } else {
292             List<DiGraphNode<N, Branch>> inNodes = cfg.getDirectedPredNodes(node);
293             if (inNodes.size() == 1) {
294               FlowState<L> inNodeState = inNodes.get(0).getAnnotation();
295               state.setIn(inNodeState.getOut());
296             } else if (inNodes.size() > 1) {
297               List<L> values = new ArrayList<>(inNodes.size());
298               for (DiGraphNode<N, Branch> currentNode : inNodes) {
299                 FlowState<L> currentNodeState = currentNode.getAnnotation();
300                 values.add(currentNodeState.getOut());
301               }
302               state.setIn(joinOp.apply(values));
303             }
304           }
305         } else {
306           List<DiGraphNode<N, Branch>> inNodes = cfg.getDirectedSuccNodes(node);
307           if (inNodes.size() == 1) {
308             DiGraphNode<N, Branch> inNode = inNodes.get(0);
309             if (inNode == cfg.getImplicitReturn()) {
310               state.setOut(createEntryLattice());
311             } else {
312               FlowState<L> inNodeState = inNode.getAnnotation();
313               state.setOut(inNodeState.getIn());
314             }
315           } else if (inNodes.size() > 1) {
316             List<L> values = new ArrayList<>(inNodes.size());
317             for (DiGraphNode<N, Branch> currentNode : inNodes) {
318               FlowState<L> currentNodeState = currentNode.getAnnotation();
319               values.add(currentNodeState.getIn());
320             }
321             state.setOut(joinOp.apply(values));
322           }
323         }
324       }

327        * The in and out states of a node.
328        *
329        * @param <L> Input and output lattice element type.

331       static class FlowState<L extends LatticeElement> implements Annotation {
332         private L in;
333         private L out;
334     
335         /**
336          * Private constructor. No other classes should create new states.
337          *
338          * @param inState Input.
339          * @param outState Output.
340          */
341         private FlowState(L inState, L outState) {
342           Preconditions.checkNotNull(inState);
343           Preconditions.checkNotNull(outState);
344           this.in = inState;
345           this.out = outState;
346         }
347     
348         L getIn() {
349           return in;
350         }
351     
352         void setIn(L in) {
353           Preconditions.checkNotNull(in);
354           this.in = in;
355         }
356     
357         L getOut() {
358           return out;
359         }
360     
361         void setOut(L out) {
362           Preconditions.checkNotNull(out);
363           this.out = out;
364         }
365     
366         @Override
367         public String toString() {
368           return SimpleFormat.format("IN: %s OUT: %s", in, out);
369         }
370     
371         @Override
372         public boolean equals(Object o) {
373           if (o instanceof FlowState) {
374             FlowState<?> that = (FlowState<?>) o;
375             return that.in.equals(this.in)
376                 && that.out.equals(this.out);
377           }
378           return false;
379         }

381         @Override
382         public int hashCode() {
383           return Objects.hash(in, out);
384         }

399       abstract static class BranchedForwardDataFlowAnalysis
400           <N, L extends LatticeElement> extends DataFlowAnalysis<N, L> {
401     
402         @Override
403         protected void initialize() {
404           orderedWorkSet.clear();
405           for (DiGraphNode<N, Branch> node : getCfg().getDirectedGraphNodes()) {
406             int outEdgeCount = getCfg().getOutEdges(node.getValue()).size();
407             List<L> outLattices = new ArrayList<>();
408             for (int i = 0; i < outEdgeCount; i++) {
409               outLattices.add(createInitialEstimateLattice());
410             }
411             node.setAnnotation(new BranchedFlowState<>(
412                 createInitialEstimateLattice(), outLattices));
413             if (node != getCfg().getImplicitReturn()) {
414               orderedWorkSet.add(node);
415             }
416           }
417         }
418     
419         BranchedForwardDataFlowAnalysis(ControlFlowGraph<N> targetCfg,
420                                         JoinOp<L> joinOp) {
421           super(targetCfg, joinOp);
422         }
423     
424         /**
425          * Returns the lattice element at the exit point. Needs to be overridden
426          * because we use a BranchedFlowState instead of a FlowState; ugh.
427          */
428         @Override
429         L getExitLatticeElement() {
430           DiGraphNode<N, Branch> node = getCfg().getImplicitReturn();
431           BranchedFlowState<L> state = node.getAnnotation();
432           return state.getIn();
433         }
434     
435         @Override
436         final boolean isForward() {
437           return true;
438         }
439     
440         /**
441          * The branched flow function maps a single lattice to a list of output
442          * lattices.
443          *
444          * <p>Each outgoing edge of a node will have a corresponding output lattice
445          * in the ordered returned by
446          * {@link com.google.javascript.jscomp.graph.DiGraph#getOutEdges(Object)}
447          * in the returned list.
448          *
449          * @return A list of output values depending on the edge's branch type.
450          */
451         abstract List<L> branchedFlowThrough(N node, L input);
452     
453         @Override
454         protected final boolean flow(DiGraphNode<N, Branch> node) {
455           BranchedFlowState<L> state = node.getAnnotation();
456           List<L> outBefore = state.out;
457           state.out = branchedFlowThrough(node.getValue(), state.in);
458           Preconditions.checkState(outBefore.size() == state.out.size());
459           for (int i = 0; i < outBefore.size(); i++) {
460             if (!outBefore.get(i).equals(state.out.get(i))) {
461               return true;
462             }
463           }
464           return false;
465         }
466     
467         @Override
468         protected void joinInputs(DiGraphNode<N, Branch> node) {
469           BranchedFlowState<L> state = node.getAnnotation();
470           List<DiGraphNode<N, Branch>> predNodes =
471               getCfg().getDirectedPredNodes(node);
472           List<L> values = new ArrayList<>(predNodes.size());
473     
474           for (DiGraphNode<N, Branch> predNode : predNodes) {
475             BranchedFlowState<L> predNodeState = predNode.getAnnotation();
476     
477             L in = predNodeState.out.get(
478                 getCfg().getDirectedSuccNodes(predNode).indexOf(node));
479     
480             values.add(in);
481           }
482           if (getCfg().getEntry() == node) {
483             state.setIn(createEntryLattice());
484           } else if (!values.isEmpty()) {
485             state.setIn(joinOp.apply(values));
486           }
487         }
488       }

543       /**
544        * Compute set of escaped variables. When a variable is escaped in a
545        * dataflow analysis, it can be reference outside of the code that we are
546        * analyzing. A variable is escaped if any of the following is true:
547        *
548        * <p><ol>
549        * <li>It is defined as the exception name in CATCH clause so it became a
550        * variable local not to our definition of scope.</li>
551        * <li>Exported variables as they can be needed after the script terminates.
552        * </li>
553        * <li>Names of named functions because in JavaScript, <i>function foo(){}</i>
554        * does not kill <i>foo</i> in the dataflow.</li>
555        */
556       static void computeEscaped(final Scope jsScope, final Set<Var> escaped,
557           AbstractCompiler compiler) {
558         // TODO(user): Very good place to store this information somewhere.
559         AbstractPostOrderCallback finder = new AbstractPostOrderCallback() {
560           @Override
561           public void visit(NodeTraversal t, Node n, Node parent) {
562             if (jsScope == t.getScope() || !n.isName()
563                 || parent.isFunction()) {
564               return;
565             }
566             String name = n.getString();
567             Var var = t.getScope().getVar(name);
568             if (var != null && var.scope == jsScope) {
569               escaped.add(jsScope.getVar(name));
570             }
571           }
572         };
573     
574         NodeTraversal t = new NodeTraversal(compiler, finder);
575         t.traverseAtScope(jsScope);
576     
577         // 1: Remove the exception name in CATCH which technically isn't local to
578         //    begin with.
579         for (Var var : jsScope.getVarIterable()) {
580           if (var.getParentNode().isCatch() ||
581               compiler.getCodingConvention().isExported(var.getName())) {
582             escaped.add(var);
583           }
584         }
585       }
