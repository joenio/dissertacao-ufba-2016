17     package com.google.javascript.jscomp.newtypes;
18     
19     import com.google.common.base.Preconditions;
20     import com.google.common.collect.ImmutableSet;
21     import com.google.common.collect.Sets;
22     import com.google.javascript.rhino.JSTypeExpression;
23     
24     import java.util.Collection;
25     
26     /**
27      *
28      * @author blickly@google.com (Ben Lickly)
29      * @author dimvar@google.com (Dimitris Vardoulakis)
30      *
31      * Represents an enumerated type.
32      * Each enum declaration produces two types of interest:
33      * - We represent the object literal that defined the enum as an ObjectType.
34      * - We represent an element of the enum by using this class in JSType.
35      */
36     public final class EnumType extends Namespace implements TypeWithProperties {
37     
38       private enum State {
39         NOT_RESOLVED,
40         DURING_RESOLUTION,
41         RESOLVED
42       }
43     
44       private State state;
45       private JSTypeExpression typeExpr;
46       // The type that accompanies the enum declaration
47       private JSType declaredType;
48       // The type of the enum's properties, a subtype of the previous field.
49       private JSType enumPropType;
50       // All properties have the same type, so we only need a set, not a map.
51       private ImmutableSet<String> props;
52     
53       private EnumType(
54           String name, JSTypeExpression typeExpr, Collection<String> props) {
55         Preconditions.checkNotNull(typeExpr);
56         this.state = State.NOT_RESOLVED;
57         this.name = name;
58         // typeExpr is non-null iff the enum is not resolved
59         this.typeExpr = typeExpr;
60         this.props = ImmutableSet.copyOf(props);
61       }
62     
63       public static EnumType make(
64           String name, JSTypeExpression typeExpr, Collection<String> props) {
65         return new EnumType(name, typeExpr, props);
66       }
67     
68       public boolean isResolved() {
69         return this.state == State.RESOLVED;
70       }
71     
72       public JSType getEnumeratedType() {
73         Preconditions.checkState(this.state == State.RESOLVED);
74         return declaredType;
75       }
76     
77       public JSType getPropType() {
78         Preconditions.checkState(this.state == State.RESOLVED);
79         return enumPropType;
80       }
81     
82       // Returns null iff there is a type cycle
83       public JSTypeExpression getTypeExpr() {
84         Preconditions.checkState(this.state != State.RESOLVED);
85         if (this.state == State.DURING_RESOLUTION) {
86           return null;
87         }
88         this.state = State.DURING_RESOLUTION;
89         return typeExpr;
90       }
91     
92       public JSTypeExpression getTypeExprForErrorReporting() {
93         Preconditions.checkState(this.state == State.DURING_RESOLUTION);
94         return typeExpr;
95       }
96     
97       void resolveEnum(JSType t) {
98         Preconditions.checkNotNull(t);
99         if (this.state == State.RESOLVED) {
100           return;
101         }
102         Preconditions.checkState(this.state == State.DURING_RESOLUTION,
103             "Expected state DURING_RESOLUTION but found %s", this.state.toString());
104         this.state = State.RESOLVED;
105         typeExpr = null;
106         declaredType = t;
107         enumPropType = JSType.fromEnum(this);
108       }
109     
110       /**
111        * When defining an enum such as
112        *   /** @enum {number} * /
113        *   var X = { ONE: 1, TWO: 2 };
114        * the properties of the object literal are constant.
115        */
116       @Override
117       protected JSType computeJSType(JSTypes commonTypes) {
118         Preconditions.checkNotNull(enumPropType);
119         Preconditions.checkState(this.namespaceType == null);
120         PersistentMap<String, Property> propMap = PersistentMap.create();
121         for (String s : this.props) {
122           propMap = propMap.with(s,
123               Property.makeConstant(null, enumPropType, enumPropType));
124         }
125         return JSType.fromObjectType(ObjectType.makeObjectType(
126             null, propMap, null, this, false, ObjectKind.UNRESTRICTED));
127       }

142       }

145       public boolean hasProp(QualifiedName qname) {
146         return declaredType.hasProp(qname);
147       }
148     
149       @Override
150       public boolean hasConstantProp(QualifiedName qname) {
151         return declaredType.hasConstantProp(qname);
152       }

154       // Unlike hasProp, this method asks about the object literal in the enum
155       // definition, not about the declared type of the enum.
156       public boolean enumLiteralHasKey(String name) {
157         return props.contains(name);
158       }

160       static boolean hasScalar(ImmutableSet<EnumType> enums) {
161         for (EnumType e : enums) {
162           if (e.declaredType.hasScalar()) {
163             return true;
164           }
165         }
166         return false;
167       }
168     
169       static boolean hasNonScalar(ImmutableSet<EnumType> enums) {
170         for (EnumType e : enums) {
171           if (e.declaredType.hasNonScalar()) {
172             return true;
173           }
174         }
175         return false;
176       }

178       static ImmutableSet<EnumType> union(
179           ImmutableSet<EnumType> s1, ImmutableSet<EnumType> s2) {
180         if (s1.isEmpty()) {
181           return s2;
182         }
183         if (s2.isEmpty() || s1.equals(s2)) {
184           return s1;
185         }
186         return Sets.union(s1, s2).immutableCopy();
187       }
188     
189       // We normalize the type so that it doesn't contain both enum {T1} and T1.
190       static ImmutableSet<EnumType> normalizeForJoin(
191           ImmutableSet<EnumType> newEnums, JSType joinWithoutEnums) {
192         boolean recreateEnums = false;
193         for (EnumType e : newEnums) {
194           if (e.declaredType.isSubtypeOf(joinWithoutEnums, SubtypeCache.create())) {
195             recreateEnums = true;
196             break;
197           }
198         }
199         if (!recreateEnums) {
200           return newEnums;
201         }
202         ImmutableSet.Builder<EnumType> builder = ImmutableSet.builder();
203         for (EnumType e : newEnums) {
204           if (!e.declaredType.isSubtypeOf(joinWithoutEnums, SubtypeCache.create())) {
205             builder.add(e);
206           }
207         }
208         return builder.build();
209       }
210     
211       static boolean areSubtypes(JSType t1, JSType t2, SubtypeCache subSuperMap) {
212         ImmutableSet<EnumType> s1 = t1.getEnums();
213         if (s1 == null) {
214           return true;
215         }
216         ImmutableSet<EnumType> s2 = t2.getEnums();
217         for (EnumType e : s1) {
218           if (s2 != null && s2.contains(e)) {
219             continue;
220           }
221           if (!e.declaredType.isSubtypeOf(t2, subSuperMap)) {
222             return false;
223           }
224         }
225         return true;
226       }
