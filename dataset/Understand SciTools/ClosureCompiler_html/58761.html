29     /**
30      * ReplaceMessages replaces user-visible messages with alternatives.
31      * It uses Google specific JsMessageVisitor implementation.
32      *
33      * @author anatol@google.com (Anatol Pomazau)
34      */
35     @GwtIncompatible("JsMessage")
36     final class ReplaceMessages extends JsMessageVisitor {
37       private final MessageBundle bundle;
38       private final boolean strictReplacement;
39     
40       static final DiagnosticType BUNDLE_DOES_NOT_HAVE_THE_MESSAGE =
41           DiagnosticType.error("JSC_BUNDLE_DOES_NOT_HAVE_THE_MESSAGE",
42               "Message with id = {0} could not be found in replacement bundle");
43     
44       ReplaceMessages(AbstractCompiler compiler, MessageBundle bundle,
45           boolean checkDuplicatedMessages, JsMessage.Style style,
46           boolean strictReplacement) {
47     
48         super(compiler, checkDuplicatedMessages, style, bundle.idGenerator());
49     
50         this.bundle = bundle;
51         this.strictReplacement = strictReplacement;
52       }
53     
54       @Override
55       void processMessageFallback(
56           Node callNode, JsMessage message1, JsMessage message2) {
57         boolean isFirstMessageTranslated =
58             (bundle.getMessage(message1.getId()) != null);
59         boolean isSecondMessageTranslated =
60             (bundle.getMessage(message2.getId()) != null);
61         Node replacementNode =
62             isSecondMessageTranslated && !isFirstMessageTranslated ?
63             callNode.getChildAtIndex(2) : callNode.getSecondChild();
64         callNode.getParent().replaceChild(callNode,
65             replacementNode.detachFromParent());
66       }
67     
68       @Override
69       protected void processJsMessage(JsMessage message,
70           JsMessageDefinition definition) {
71     
72         // Get the replacement.
73         JsMessage replacement = bundle.getMessage(message.getId());
74         if (replacement == null) {
75           if (strictReplacement) {
76             compiler.report(JSError.make(
77                 definition.getMessageNode(), BUNDLE_DOES_NOT_HAVE_THE_MESSAGE,
78                 message.getId()));
79             // Fallback to the default message
80             return;
81           } else {
82             // In case if it is not a strict replacement we could leave original
83             // message.
84             replacement = message;
85           }
86         }
87     
88         // Replace the message.
89         Node newValue;
90         Node msgNode = definition.getMessageNode();
91         try {
92           newValue = getNewValueNode(replacement, msgNode);
93         } catch (MalformedException e) {
94           compiler.report(JSError.make(
95               e.getNode(), MESSAGE_TREE_MALFORMED, e.getMessage()));
96           newValue = msgNode;
97         }
98     
99         if (newValue != msgNode) {
100           newValue.useSourceInfoIfMissingFromForTree(msgNode);
101           msgNode.getParent().replaceChild(msgNode, newValue);
102           compiler.reportCodeChange();
103         }
104       }

106       /**
107        * Constructs a node representing a message's value, or, if possible, just
108        * modifies {@code origValueNode} so that it accurately represents the
109        * message's value.
110        *
111        * @param message  a message
112        * @param origValueNode  the message's original value node
113        * @return a Node that can replace {@code origValueNode}
114        *
115        * @throws MalformedException if the passed node's subtree structure is
116        *   not as expected
117        */
118       private Node getNewValueNode(JsMessage message, Node origValueNode)
119           throws MalformedException {
120         switch (origValueNode.getType()) {
121           case FUNCTION:
122             // The message is a function. Modify the function node.
123             updateFunctionNode(message, origValueNode);
124             return origValueNode;
125           case STRING:
126             // The message is a simple string. Modify the string node.
127             String newString = message.toString();
128             if (!origValueNode.getString().equals(newString)) {
129               origValueNode.setString(newString);
130               compiler.reportCodeChange();
131             }
132             return origValueNode;
133           case ADD:
134             // The message is a simple string. Create a string node.
135             return IR.string(message.toString());
136           case CALL:
137             // The message is a function call. Replace it with a string expression.
138             return replaceCallNode(message, origValueNode);
139           default:
140             throw new MalformedException(
141                 "Expected FUNCTION, STRING, or ADD node; found: " +
142                     origValueNode.getType(), origValueNode);
143         }
144       }

146       /**
147        * Updates the descendants of a FUNCTION node to represent a message's value.
148        * <p>
149        * The tree looks something like:
150        * <pre>
151        * function
152        *  |-- name
153        *  |-- lp
154        *  |   |-- name <arg1>
155        *  |    -- name <arg2>
156        *   -- block
157        *      |
158        *       --return
159        *           |
160        *            --add
161        *               |-- string foo
162        *                -- name <arg1>
163        * </pre>
164        *
165        * @param message  a message
166        * @param functionNode  the message's original FUNCTION value node
167        *
168        * @throws MalformedException if the passed node's subtree structure is
169        *         not as expected
170        */
171       private void updateFunctionNode(JsMessage message, Node functionNode)
172           throws MalformedException {
173         checkNode(functionNode, Token.FUNCTION);
174         Node nameNode = functionNode.getFirstChild();
175         checkNode(nameNode, Token.NAME);
176         Node argListNode = nameNode.getNext();
177         checkNode(argListNode, Token.PARAM_LIST);
178         Node oldBlockNode = argListNode.getNext();
179         checkNode(oldBlockNode, Token.BLOCK);
180     
181         Iterator<CharSequence> iterator = message.parts().iterator();
182         Node valueNode = iterator.hasNext()
183             ? constructAddOrStringNode(iterator, argListNode)
184             : IR.string("");
185         Node newBlockNode = IR.block(IR.returnNode(valueNode));
186     
187         // TODO(user): checkTreeEqual is overkill. I am in process of rewriting
188         // these functions.
189         if (newBlockNode.checkTreeEquals(oldBlockNode) != null) {
190           newBlockNode.useSourceInfoIfMissingFromForTree(oldBlockNode);
191           functionNode.replaceChild(oldBlockNode, newBlockNode);
192           compiler.reportCodeChange();
193         }
194       }

196       /**
197        * Creates a parse tree corresponding to the remaining message parts in
198        * an iteration. The result will contain only STRING nodes, NAME nodes
199        * (corresponding to placeholder references), and/or ADD nodes used to
200        * combine the other two types.
201        *
202        * @param partsIterator  an iterator over message parts
203        * @param argListNode  an LP node whose children are valid placeholder names
204        * @return the root of the constructed parse tree
205        *
206        * @throws MalformedException if {@code partsIterator} contains a
207        *   placeholder reference that does not correspond to a valid argument in
208        *   the arg list
209        */
210       private static Node constructAddOrStringNode(Iterator<CharSequence> partsIterator,
211                                                    Node argListNode)
212           throws MalformedException {
213         CharSequence part = partsIterator.next();
214         Node partNode = null;
215         if (part instanceof JsMessage.PlaceholderReference) {
216           JsMessage.PlaceholderReference phRef =
217               (JsMessage.PlaceholderReference) part;
218     
219           for (Node node : argListNode.children()) {
220             if (node.isName()) {
221               String arg = node.getString();
222     
223               // We ignore the case here because the transconsole only supports
224               // uppercase placeholder names, but function arguments in JavaScript
225               // code can have mixed case.
226               if (arg.equalsIgnoreCase(phRef.getName())) {
227                 partNode = IR.name(arg);
228               }
229             }
230           }
231     
232           if (partNode == null) {
233             throw new MalformedException(
234                 "Unrecognized message placeholder referenced: " + phRef.getName(),
235                 argListNode);
236           }
237         } else {
238           // The part is just a string literal.
239           partNode = IR.string(part.toString());
240         }
241     
242         if (partsIterator.hasNext()) {
243           return IR.add(partNode,
244                           constructAddOrStringNode(partsIterator, argListNode));
245         } else {
246           return partNode;
247         }
248       }

250       /**
251        * Replaces a CALL node with an inlined message value.
252        *  <p>
253        * The call tree looks something like:
254        * <pre>
255        * call
256        *  |-- getprop
257        *  |   |-- name 'goog'
258        *  |   +-- string 'getMsg'
259        *  |
260        *  |-- string 'Hi {$userName}! Welcome to {$product}.'
261        *  +-- objlit
262        *      |-- string 'userName'
263        *      |-- name 'someUserName'
264        *      |-- string 'product'
265        *      +-- call
266        *          +-- name 'getProductName'
267        * <pre>
268        * <p>
269        * For that example, we'd return:
270        * <pre>
271        * add
272        *  |-- string 'Hi '
273        *  +-- add
274        *      |-- name someUserName
275        *      +-- add
276        *          |-- string '! Welcome to '
277        *          +-- add
278        *              |-- call
279        *              |   +-- name 'getProductName'
280        *              +-- string '.'
281        * </pre>
282        * @param message  a message
283        * @param callNode  the message's original CALL value node
284        * @return a STRING node, or an ADD node that does string concatenation, if
285        *   the message has one or more placeholders
286        *
287        * @throws MalformedException if the passed node's subtree structure is
288        *   not as expected
289        */
290       private Node replaceCallNode(JsMessage message, Node callNode)
291           throws MalformedException {
292         checkNode(callNode, Token.CALL);
293         Node getPropNode = callNode.getFirstChild();
294         checkNode(getPropNode, Token.GETPROP);
295         Node stringExprNode = getPropNode.getNext();
296         checkStringExprNode(stringExprNode);
297         Node objLitNode = stringExprNode.getNext();
298     
299         // Build the replacement tree.
300         return constructStringExprNode(
301             message.parts().iterator(), objLitNode, callNode);
302       }
303     
304       /**
305        * Creates a parse tree corresponding to the remaining message parts in an
306        * iteration. The result consists of one or more STRING nodes, placeholder
307        * replacement value nodes (which can be arbitrary expressions), and ADD
308        * nodes.
309        *
310        * @param parts  an iterator over message parts
311        * @param objLitNode  an OBJLIT node mapping placeholder names to values
312        * @return the root of the constructed parse tree
313        *
314        * @throws MalformedException if {@code parts} contains a placeholder
315        *   reference that does not correspond to a valid placeholder name
316        */
317       private static Node constructStringExprNode(
318           Iterator<CharSequence> parts, Node objLitNode, Node refNode) throws MalformedException {
319         Preconditions.checkNotNull(refNode);
320     
321         CharSequence part = parts.next();
322         Node partNode = null;
323         if (part instanceof JsMessage.PlaceholderReference) {
324           JsMessage.PlaceholderReference phRef =
325               (JsMessage.PlaceholderReference) part;
326     
327           // The translated message is null
328           if (objLitNode == null) {
329             throw new MalformedException("Empty placeholder value map " +
330                 "for a translated message with placeholders.", refNode);
331           }
332     
333           for (Node key = objLitNode.getFirstChild(); key != null;
334                key = key.getNext()) {
335             if (key.getString().equals(phRef.getName())) {
336               Node valueNode = key.getFirstChild();
337               partNode = valueNode.cloneTree();
338             }
339           }
340     
341           if (partNode == null) {
342             throw new MalformedException(
343                 "Unrecognized message placeholder referenced: " + phRef.getName(),
344                 objLitNode);
345           }
346         } else {
347           // The part is just a string literal.
348           partNode = IR.string(part.toString());
349         }
350     
351         if (parts.hasNext()) {
352           return IR.add(partNode,
353               constructStringExprNode(parts, objLitNode, refNode));
354         } else {
355           return partNode;
356         }
357       }

359       /**
360        * Checks that a node is a valid string expression (either a string literal
361        * or a concatenation of string literals).
362        *
363        * @throws IllegalArgumentException if the node is null or the wrong type
364        */
365       private static void checkStringExprNode(@Nullable Node node) {
366         if (node == null) {
367           throw new IllegalArgumentException("Expected a string; found: null");
368         }
369         switch (node.getType()) {
370           case STRING:
371             break;
372           case ADD:
373             Node c = node.getFirstChild();
374             checkStringExprNode(c);
375             checkStringExprNode(c.getNext());
376             break;
377           default:
378             throw new IllegalArgumentException(
379                 "Expected a string; found: " + node.getType());
380         }
381       }
