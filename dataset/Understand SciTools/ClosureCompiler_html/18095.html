43     /**
44      * A MessageBundle that parses messages from an XML Translation Bundle (XTB)
45      * file.
46      *
47      * TODO(moz): Make this GWT compatible.
48      *
49      */
50     @GwtIncompatible("Currently not used in GWT version")
51     @SuppressWarnings("sunapi")
52     public final class XtbMessageBundle implements MessageBundle {
53       private static final SecureEntityResolver NOOP_RESOLVER
54           = new SecureEntityResolver();
55     
56       private final Map<String, JsMessage> messages;
57       private final JsMessage.IdGenerator idGenerator;
58     
59       public XtbMessageBundle(
60           InputStream xtb, @Nullable String projectId,
61           @SuppressWarnings("unused") boolean unused) {
62         this(xtb, projectId);
63       }
64     
65       /**
66        * Creates an instance and initializes it with the messages in an XTB file.
67        *
68        * @param xtb  the XTB file as a byte stream
69        * @param projectId  the translation console project id (i.e. name)
70        */
71       public XtbMessageBundle(InputStream xtb, @Nullable String projectId) {
72         Preconditions.checkState(!"".equals(projectId));
73         this.messages = new HashMap<>();
74         this.idGenerator = new GoogleJsMessageIdGenerator(projectId);
75     
76         try {
77           // Use a SAX parser for speed and less memory usage.
78           SAXParser parser = createSAXParser();
79           XMLReader reader = parser.getXMLReader();
80           Handler contentHandler = new Handler();
81           reader.setContentHandler(contentHandler);
82           reader.parse(new InputSource(xtb));
83         } catch (ParserConfigurationException | IOException | SAXException e) {
84           throw new RuntimeException(e);
85         }
86       }
87     
88       // Inlined from guava-internal.
89       private static SAXParser createSAXParser()
90           throws ParserConfigurationException, SAXException {
91         SAXParserFactory factory = SAXParserFactory.newInstance();
92         factory.setValidating(false);
93         factory.setXIncludeAware(false);
94         factory.setFeature(
95             "http://xml.org/sax/features/external-general-entities", false);
96         factory.setFeature(
97             "http://xml.org/sax/features/external-parameter-entities", false);
98         factory.setFeature(
99             "http://apache.org/xml/features/nonvalidating/load-external-dtd",
100             false);
101         factory.setFeature(XMLConstants.FEATURE_SECURE_PROCESSING, true);
102     
103         SAXParser parser = factory.newSAXParser();
104         XMLReader xmlReader = parser.getXMLReader();
105         xmlReader.setEntityResolver(NOOP_RESOLVER);
106         return parser;
107       }
108     
109       @Override
110       public JsMessage getMessage(String id) {
111         return messages.get(id);
112       }
113     
114       @Override
115       public JsMessage.IdGenerator idGenerator() {
116         return idGenerator;
117       }
118     
119       @Override
120       public Iterable<JsMessage> getAllMessages() {
121         return Iterables.unmodifiableIterable(messages.values());
122       }

124       /**
125        * A {@link ContentHandler} that creates a {@link JsMessage} for each message
126        * parsed from an XML Translation Bundle (XTB) file.
127        */
128       private class Handler implements ContentHandler {
129         private static final String BUNDLE_ELEM_NAME = "translationbundle";
130         private static final String LANG_ATT_NAME = "lang";
131     
132         private static final String TRANSLATION_ELEM_NAME = "translation";
133         private static final String MESSAGE_ID_ATT_NAME = "id";
134     
135         private static final String PLACEHOLDER_ELEM_NAME = "ph";
136         private static final String PLACEHOLDER_NAME_ATT_NAME = "name";
137     
138         String lang;
139         JsMessage.Builder msgBuilder;
140     
141         @Override
142         public void setDocumentLocator(Locator locator) {}
143     
144         @Override
145         public void startDocument() {}
146     
147         @Override
148         public void endDocument() {}
149     
150         @Override
151         public void startPrefixMapping(String prefix, String uri) {}
152     
153         @Override
154         public void endPrefixMapping(String prefix) {}
155     
156         @Override
157         public void startElement(String uri, String localName, String qName,
158                                  Attributes atts) {
159           if (BUNDLE_ELEM_NAME.equals(qName)) {
160             Preconditions.checkState(lang == null);
161             lang = atts.getValue(LANG_ATT_NAME);
162             Preconditions.checkState(lang != null && !lang.isEmpty());
163           } else if (TRANSLATION_ELEM_NAME.equals(qName)) {
164             Preconditions.checkState(msgBuilder == null);
165             String id = atts.getValue(MESSAGE_ID_ATT_NAME);
166             Preconditions.checkState(id != null && !id.isEmpty());
167             msgBuilder = new JsMessage.Builder(id);
168           } else if (PLACEHOLDER_ELEM_NAME.equals(qName)) {
169             Preconditions.checkState(msgBuilder != null);
170             String phRef = atts.getValue(PLACEHOLDER_NAME_ATT_NAME);
171             phRef = JsMessageVisitor.toLowerCamelCaseWithNumericSuffixes(phRef);
172             msgBuilder.appendPlaceholderReference(phRef);
173           }
174         }
175     
176         @Override
177         public void endElement(String uri, String localName, String qName) {
178           if (TRANSLATION_ELEM_NAME.equals(qName)) {
179             Preconditions.checkState(msgBuilder != null);
180             if (!msgBuilder.hasParts()) {
181               msgBuilder.appendStringPart("");
182             }
183             String key = msgBuilder.getKey();
184             messages.put(key, msgBuilder.build());
185             msgBuilder = null;
186           }
187         }
188     
189         @Override
190         public void characters(char ch[], int start, int length) {
191           if (msgBuilder != null) {
192             // Append a string literal to the message.
193             msgBuilder.appendStringPart(String.valueOf(ch, start, length));
194           }
195         }
196     
197         @Override
198         public void ignorableWhitespace(char ch[], int start, int length) {
199           if (msgBuilder != null) {
200             // Preserve whitespace in messages.
201             msgBuilder.appendStringPart(String.valueOf(ch, start, length));
202           }
203         }
204     
205         @Override
206         public void processingInstruction(String target, String data) {}
207     
208         @Override
209         public void skippedEntity(String name) {}
210       }

212       /**
213        * A secure EntityResolver that returns an empty string in response to
214        * any attempt to resolve an external entity. The class is used by our
215        * secure version of the internal saxon SAX parser.
216        */
217       private static final class SecureEntityResolver implements EntityResolver {
218     
219         @Override
220         public InputSource resolveEntity(String publicId, String systemId) {
221           return new InputSource(new StringReader(""));
222         }
223       }
