122       /**
123        * Infers all of a function's arguments if their types aren't declared.
124        */
125       private void inferArguments(TypedScope functionScope) {
126         Node functionNode = functionScope.getRootNode();
127         Node astParameters = functionNode.getSecondChild();
128         Node iifeArgumentNode = null;
129     
130         if (NodeUtil.isCallOrNewTarget(functionNode)) {
131           iifeArgumentNode = functionNode.getNext();
132         }
133     
134         FunctionType functionType =
135             JSType.toMaybeFunctionType(functionNode.getJSType());
136         if (functionType != null) {
137           Node parameterTypes = functionType.getParametersNode();
138           if (parameterTypes != null) {
139             Node parameterTypeNode = parameterTypes.getFirstChild();
140             for (Node astParameter : astParameters.children()) {
141               TypedVar var = functionScope.getVar(astParameter.getString());
142               Preconditions.checkNotNull(var);
143               if (var.isTypeInferred() &&
144                   var.getType() == unknownType) {
145                 JSType newType = null;
146     
147                 if (iifeArgumentNode != null) {
148                   newType = iifeArgumentNode.getJSType();
149                 } else if (parameterTypeNode != null) {
150                   newType = parameterTypeNode.getJSType();
151                 }
152     
153                 if (newType != null) {
154                   var.setType(newType);
155                   astParameter.setJSType(newType);
156                 }
157               }
158     
159               if (parameterTypeNode != null) {
160                 parameterTypeNode = parameterTypeNode.getNext();
161               }
162               if (iifeArgumentNode != null) {
163                 iifeArgumentNode = iifeArgumentNode.getNext();
164               }
165             }
166           }
167         }
168       }

180       @Override
181       FlowScope flowThrough(Node n, FlowScope input) {
182         // If we have not walked a path from <entry> to <n>, then we don't
183         // want to infer anything about this scope.
184         if (input == bottomScope) {
185           return input;
186         }
187     
188         FlowScope output = input.createChildFlowScope();
189         output = traverse(n, output);
190         return output;
191       }

193       @Override
194       @SuppressWarnings({"fallthrough", "incomplete-switch"})
195       List<FlowScope> branchedFlowThrough(Node source, FlowScope input) {
196         // NOTE(nicksantos): Right now, we just treat ON_EX edges like UNCOND
197         // edges. If we wanted to be perfect, we'd actually JOIN all the out
198         // lattices of this flow with the in lattice, and then make that the out
199         // lattice for the ON_EX edge. But it's probably too expensive to be
200         // worthwhile.
201         FlowScope output = flowThrough(source, input);
202         Node condition = null;
203         FlowScope conditionFlowScope = null;
204         BooleanOutcomePair conditionOutcomes = null;
205     
206         List<DiGraphEdge<Node, Branch>> branchEdges = getCfg().getOutEdges(source);
207         List<FlowScope> result = new ArrayList<>(branchEdges.size());
208         for (DiGraphEdge<Node, Branch> branchEdge : branchEdges) {
209           Branch branch = branchEdge.getValue();
210           FlowScope newScope = output;
211     
212           switch (branch) {
213             case ON_TRUE:
214               if (NodeUtil.isForIn(source)) {
215                 // item is assigned a property name, so its type should be string.
216                 Node item = source.getFirstChild();
217                 Node obj = item.getNext();
218     
219                 FlowScope informed = traverse(obj, output.createChildFlowScope());
220     
221                 if (item.isVar()) {
222                   item = item.getFirstChild();
223                 }
224                 if (item.isName()) {
225                   JSType iterKeyType = getNativeType(STRING_TYPE);
226                   ObjectType objType = getJSType(obj).dereference();
227                   JSType objIndexType = objType == null ?
228                       null : objType.getTemplateTypeMap().getResolvedTemplateType(
229                           registry.getObjectIndexKey());
230                   if (objIndexType != null && !objIndexType.isUnknownType()) {
231                     JSType narrowedKeyType =
232                         iterKeyType.getGreatestSubtype(objIndexType);
233                     if (!narrowedKeyType.isEmptyType()) {
234                       iterKeyType = narrowedKeyType;
235                     }
236                   }
237                   redeclareSimpleVar(informed, item, iterKeyType);
238                 }
239                 newScope = informed;
240                 break;
241               }
242     
243               // FALL THROUGH
244     
245             case ON_FALSE:
246               if (condition == null) {
247                 condition = NodeUtil.getConditionExpression(source);
248                 if (condition == null && source.isCase()) {
249                   condition = source;
250     
251                   // conditionFlowScope is cached from previous iterations
252                   // of the loop.
253                   if (conditionFlowScope == null) {
254                     conditionFlowScope = traverse(
255                         condition.getFirstChild(), output.createChildFlowScope());
256                   }
257                 }
258               }
259     
260               if (condition != null) {
261                 if (condition.isAnd() ||
262                     condition.isOr()) {
263                   // When handling the short-circuiting binary operators,
264                   // the outcome scope on true can be different than the outcome
265                   // scope on false.
266                   //
267                   // TODO(nicksantos): The "right" way to do this is to
268                   // carry the known outcome all the way through the
269                   // recursive traversal, so that we can construct a
270                   // different flow scope based on the outcome. However,
271                   // this would require a bunch of code and a bunch of
272                   // extra computation for an edge case. This seems to be
273                   // a "good enough" approximation.
274     
275                   // conditionOutcomes is cached from previous iterations
276                   // of the loop.
277                   if (conditionOutcomes == null) {
278                     conditionOutcomes = condition.isAnd() ?
279                         traverseAnd(condition, output.createChildFlowScope()) :
280                         traverseOr(condition, output.createChildFlowScope());
281                   }
282                   newScope =
283                       reverseInterpreter.getPreciserScopeKnowingConditionOutcome(
284                           condition,
285                           conditionOutcomes.getOutcomeFlowScope(
286                               condition.getType(), branch == Branch.ON_TRUE),
287                           branch == Branch.ON_TRUE);
288                 } else {
289                   // conditionFlowScope is cached from previous iterations
290                   // of the loop.
291                   if (conditionFlowScope == null) {
292                     conditionFlowScope =
293                         traverse(condition, output.createChildFlowScope());
294                   }
295                   newScope =
296                       reverseInterpreter.getPreciserScopeKnowingConditionOutcome(
297                           condition, conditionFlowScope, branch == Branch.ON_TRUE);
298                 }
299               }
300               break;
301           }
302     
303           result.add(newScope.optimize());
304         }
305         return result;
306       }

308       private FlowScope traverse(Node n, FlowScope scope) {
309         switch (n.getType()) {
310           case ASSIGN:
311             scope = traverseAssign(n, scope);
312             break;
313     
314           case NAME:
315             scope = traverseName(n, scope);
316             break;
317     
318           case GETPROP:
319             scope = traverseGetProp(n, scope);
320             break;
321     
322           case AND:
323             scope = traverseAnd(n, scope).getJoinedFlowScope()
324                 .createChildFlowScope();
325             break;
326     
327           case OR:
328             scope = traverseOr(n, scope).getJoinedFlowScope()
329                 .createChildFlowScope();
330             break;
331     
332           case HOOK:
333             scope = traverseHook(n, scope);
334             break;
335     
336           case OBJECTLIT:
337             scope = traverseObjectLiteral(n, scope);
338             break;
339     
340           case CALL:
341             scope = traverseCall(n, scope);
342             break;
343     
344           case NEW:
345             scope = traverseNew(n, scope);
346             break;
347     
348           case ASSIGN_ADD:
349           case ADD:
350             scope = traverseAdd(n, scope);
351             break;
352     
353           case POS:
354           case NEG:
355             scope = traverse(n.getFirstChild(), scope);  // Find types.
356             n.setJSType(getNativeType(NUMBER_TYPE));
357             break;
358     
359           case ARRAYLIT:
360             scope = traverseArrayLiteral(n, scope);
361             break;
362     
363           case THIS:
364             n.setJSType(scope.getTypeOfThis());
365             break;
366     
367           case ASSIGN_LSH:
368           case ASSIGN_RSH:
369           case LSH:
370           case RSH:
371           case ASSIGN_URSH:
372           case URSH:
373           case ASSIGN_DIV:
374           case ASSIGN_MOD:
375           case ASSIGN_BITAND:
376           case ASSIGN_BITXOR:
377           case ASSIGN_BITOR:
378           case ASSIGN_MUL:
379           case ASSIGN_SUB:
380           case DIV:
381           case MOD:
382           case BITAND:
383           case BITXOR:
384           case BITOR:
385           case MUL:
386           case SUB:
387           case DEC:
388           case INC:
389           case BITNOT:
390             scope = traverseChildren(n, scope);
391             n.setJSType(getNativeType(NUMBER_TYPE));
392             break;
393     
394           case PARAM_LIST:
395             scope = traverse(n.getFirstChild(), scope);
396             n.setJSType(getJSType(n.getFirstChild()));
397             break;
398     
399           case COMMA:
400             scope = traverseChildren(n, scope);
401             n.setJSType(getJSType(n.getLastChild()));
402             break;
403     
404           case TYPEOF:
405             scope = traverseChildren(n, scope);
406             n.setJSType(getNativeType(STRING_TYPE));
407             break;
408     
409           case DELPROP:
410           case LT:
411           case LE:
412           case GT:
413           case GE:
414           case NOT:
415           case EQ:
416           case NE:
417           case SHEQ:
418           case SHNE:
419           case INSTANCEOF:
420           case IN:
421             scope = traverseChildren(n, scope);
422             n.setJSType(getNativeType(BOOLEAN_TYPE));
423             break;
424     
425           case GETELEM:
426             scope = traverseGetElem(n, scope);
427             break;
428     
429           case EXPR_RESULT:
430             scope = traverseChildren(n, scope);
431             if (n.getFirstChild().isGetProp()) {
432               ensurePropertyDeclared(n.getFirstChild());
433             }
434             break;
435     
436           case SWITCH:
437             scope = traverse(n.getFirstChild(), scope);
438             break;
439     
440           case RETURN:
441             scope = traverseReturn(n, scope);
442             break;
443     
444           case VAR:
445           case THROW:
446             scope = traverseChildren(n, scope);
447             break;
448     
449           case CATCH:
450             scope = traverseCatch(n, scope);
451             break;
452     
453           case CAST:
454             scope = traverseChildren(n, scope);
455             JSDocInfo info = n.getJSDocInfo();
456             if (info != null && info.hasType()) {
457               n.setJSType(info.getType().evaluate(syntacticScope, registry));
458             }
459             break;
460         }
461     
462         return scope;
463       }

466        * Traverse a return value.

468       private FlowScope traverseReturn(Node n, FlowScope scope) {
469         scope = traverseChildren(n, scope);
470     
471         Node retValue = n.getFirstChild();
472         if (retValue != null) {
473           JSType type = functionScope.getRootNode().getJSType();
474           if (type != null) {
475             FunctionType fnType = type.toMaybeFunctionType();
476             if (fnType != null) {
477               inferPropertyTypesToMatchConstraint(
478                   retValue.getJSType(), fnType.getReturnType());
479             }
480           }
481         }
482         return scope;
483       }

485       /**
486        * Any value can be thrown, so it's really impossible to determine the type
487        * of a CATCH param. Treat it as the UNKNOWN type.
488        */
489       private FlowScope traverseCatch(Node catchNode, FlowScope scope) {
490         Node name = catchNode.getFirstChild();
491         JSType type;
492         // If the catch expression name was declared in the catch use that type,
493         // otherwise use "unknown".
494         JSDocInfo info = name.getJSDocInfo();
495         if (info != null && info.hasType()) {
496           type = info.getType().evaluate(syntacticScope, registry);
497         } else {
498           type = getNativeType(JSTypeNative.UNKNOWN_TYPE);
499         }
500         redeclareSimpleVar(scope, name, type);
501         name.setJSType(type);
502         return scope;
503       }

505       private FlowScope traverseAssign(Node n, FlowScope scope) {
506         Node left = n.getFirstChild();
507         Node right = n.getLastChild();
508         scope = traverseChildren(n, scope);
509     
510         JSType leftType = left.getJSType();
511         JSType rightType = getJSType(right);
512         n.setJSType(rightType);
513     
514         updateScopeForTypeChange(scope, left, leftType, rightType);
515         return scope;
516       }

518       /**
519        * Updates the scope according to the result of a type change, like
520        * an assignment or a type cast.
521        */
522       private void updateScopeForTypeChange(
523           FlowScope scope, Node left, JSType leftType, JSType resultType) {
524         Preconditions.checkNotNull(resultType);
525         switch (left.getType()) {
526           case NAME:
527             String varName = left.getString();
528             TypedVar var = syntacticScope.getVar(varName);
529             JSType varType = var == null ? null : var.getType();
530             boolean isVarDeclaration = left.hasChildren()
531                 && varType != null && !var.isTypeInferred();
532     
533             boolean isTypelessConstDecl =
534                 isVarDeclaration &&
535                 NodeUtil.isConstantDeclaration(
536                     compiler.getCodingConvention(),
537                     var.getJSDocInfo(), var.getNameNode()) &&
538                   !(var.getJSDocInfo() != null &&
539                     var.getJSDocInfo().hasType());
540     
541             // When looking at VAR initializers for declared VARs, we tend
542             // to use the declared type over the type it's being
543             // initialized to in the global scope.
544             //
545             // For example,
546             // /** @param {number} */ var f = goog.abstractMethod;
547             // it's obvious that the programmer wants you to use
548             // the declared function signature, not the inferred signature.
549             //
550             // Or,
551             // /** @type {Object.<string>} */ var x = {};
552             // the one-time anonymous object on the right side
553             // is as narrow as it can possibly be, but we need to make
554             // sure we back-infer the <string> element constraint on
555             // the left hand side, so we use the left hand side.
556     
557             boolean isVarTypeBetter = isVarDeclaration
558                 // Makes it easier to check for NPEs.
559                 && !resultType.isNullType() && !resultType.isVoidType()
560                 // Do not use the var type if the declaration looked like
561                 // /** @const */ var x = 3;
562                 // because this type was computed from the RHS
563                 && !isTypelessConstDecl;
564     
565             // TODO(nicksantos): This might be a better check once we have
566             // back-inference of object/array constraints.  It will probably
567             // introduce more type warnings.  It uses the result type iff it's
568             // strictly narrower than the declared var type.
569             //
570             //boolean isVarTypeBetter = isVarDeclaration &&
571             //    (varType.restrictByNotNullOrUndefined().isSubtype(resultType)
572             //     || !resultType.isSubtype(varType));
573     
574             if (isVarTypeBetter) {
575               redeclareSimpleVar(scope, left, varType);
576             } else {
577               redeclareSimpleVar(scope, left, resultType);
578             }
579             left.setJSType(resultType);
580     
581             if (var != null && var.isTypeInferred()) {
582               JSType oldType = var.getType();
583               var.setType(oldType == null ?
584                  resultType : oldType.getLeastSupertype(resultType));
585             } else if (isTypelessConstDecl) {
586               // /** @const */ var x = y;
587               // should be redeclared, so that the type of y
588               // gets propagated to inner scopes.
589               var.setType(resultType);
590             }
591             break;
592           case GETPROP:
593             String qualifiedName = left.getQualifiedName();
594             if (qualifiedName != null) {
595               boolean declaredSlotType = false;
596               JSType rawObjType = left.getFirstChild().getJSType();
597               if (rawObjType != null) {
598                 ObjectType objType = ObjectType.cast(
599                     rawObjType.restrictByNotNullOrUndefined());
600                 if (objType != null) {
601                   String propName = left.getLastChild().getString();
602                   declaredSlotType = objType.isPropertyTypeDeclared(propName);
603                 }
604               }
605               JSType safeLeftType = leftType == null ? unknownType : leftType;
606               scope.inferQualifiedSlot(left, qualifiedName, safeLeftType, resultType, declaredSlotType);
607             }
608     
609             left.setJSType(resultType);
610             ensurePropertyDefined(left, resultType);
611             break;
612         }
613       }

615       /**
616        * Defines a property if the property has not been defined yet.
617        */
618       private void ensurePropertyDefined(Node getprop, JSType rightType) {
619         String propName = getprop.getLastChild().getString();
620         Node obj = getprop.getFirstChild();
621         JSType nodeType = getJSType(obj);
622         ObjectType objectType = ObjectType.cast(
623             nodeType.restrictByNotNullOrUndefined());
624         boolean propCreationInConstructor = obj.isThis() &&
625             getJSType(syntacticScope.getRootNode()).isConstructor();
626     
627         if (objectType == null) {
628           registry.registerPropertyOnType(propName, nodeType);
629         } else {
630           if (nodeType.isStruct() && !objectType.hasProperty(propName)) {
631             // In general, we don't want to define a property on a struct object,
632             // b/c TypeCheck will later check for improper property creation on
633             // structs. There are two exceptions.
634             // 1) If it's a property created inside the constructor, on the newly
635             //    created instance, allow it.
636             // 2) If it's a prototype property, allow it. For example:
637             //    Foo.prototype.bar = baz;
638             //    where Foo.prototype is a struct and the assignment happens at the
639             //    top level and the constructor Foo is defined in the same file.
640             boolean staticPropCreation = false;
641             Node maybeAssignStm = getprop.getGrandparent();
642             if (syntacticScope.isGlobal() &&
643                 NodeUtil.isPrototypePropertyDeclaration(maybeAssignStm)) {
644               String propCreationFilename = maybeAssignStm.getSourceFileName();
645               Node ctor = objectType.getOwnerFunction().getSource();
646               if (ctor != null &&
647                   ctor.getSourceFileName().equals(propCreationFilename)) {
648                 staticPropCreation = true;
649               }
650             }
651             if (!propCreationInConstructor && !staticPropCreation) {
652               return; // Early return to avoid creating the property below.
653             }
654           }
655     
656           if (ensurePropertyDeclaredHelper(getprop, objectType)) {
657             return;
658           }
659     
660           if (!objectType.isPropertyTypeDeclared(propName)) {
661             // We do not want a "stray" assign to define an inferred property
662             // for every object of this type in the program. So we use a heuristic
663             // approach to determine whether to infer the property.
664             //
665             // 1) If the property is already defined, join it with the previously
666             //    inferred type.
667             // 2) If this isn't an instance object, define it.
668             // 3) If the property of an object is being assigned in the constructor,
669             //    define it.
670             // 4) If this is a stub, define it.
671             // 5) Otherwise, do not define the type, but declare it in the registry
672             //    so that we can use it for missing property checks.
673             if (objectType.hasProperty(propName) || !objectType.isInstanceType()) {
674               if ("prototype".equals(propName)) {
675                 objectType.defineDeclaredProperty(propName, rightType, getprop);
676               } else {
677                 objectType.defineInferredProperty(propName, rightType, getprop);
678               }
679             } else if (propCreationInConstructor) {

728       private FlowScope traverseName(Node n, FlowScope scope) {
729         String varName = n.getString();
730         Node value = n.getFirstChild();
731         JSType type = n.getJSType();
732         if (value != null) {
733           scope = traverse(value, scope);
734           updateScopeForTypeChange(scope, n, n.getJSType() /* could be null */,
735               getJSType(value));
736           return scope;
737         } else {
738           StaticTypedSlot<JSType> var = scope.getSlot(varName);
739           if (var != null) {
740             // There are two situations where we don't want to use type information
741             // from the scope, even if we have it.
742     
743             // 1) The var is escaped and assigned in an inner scope, e.g.,
744             // function f() { var x = 3; function g() { x = null } (x); }
745             boolean isInferred = var.isTypeInferred();
746             boolean unflowable = isInferred &&
747                 isUnflowable(syntacticScope.getVar(varName));
748     
749             // 2) We're reading type information from another scope for an
750             // inferred variable. That variable is assigned more than once,
751             // and we can't know which type we're getting.
752             //
753             // var t = null; function f() { (t); } doStuff(); t = {};
754             //
755             // Notice that this heuristic isn't perfect. For example, you might
756             // have:
757             //
758             // function f() { (t); } f(); var t = 3;
759             //
760             // In this case, we would infer the first reference to t as
761             // type {number}, even though it's undefined.
762             boolean nonLocalInferredSlot = false;
763             if (isInferred && syntacticScope.isLocal()) {
764               TypedVar maybeOuterVar = syntacticScope.getParent().getVar(varName);
765               if (var == maybeOuterVar &&
766                   !maybeOuterVar.isMarkedAssignedExactlyOnce()) {
767                 nonLocalInferredSlot = true;
768               }
769             }
770     
771             if (!unflowable && !nonLocalInferredSlot) {
772               type = var.getType();
773               if (type == null) {
774                 type = unknownType;
775               }
776             }
777           }
778         }
779         n.setJSType(type);
780         return scope;
781       }

790       private FlowScope traverseObjectLiteral(Node n, FlowScope scope) {
791         JSType type = n.getJSType();
792         Preconditions.checkNotNull(type);
793     
794         for (Node name = n.getFirstChild(); name != null; name = name.getNext()) {
795           scope = traverse(name.getFirstChild(), scope);
796         }
797     
798         // Object literals can be reflected on other types.
799         // See CodingConvention#getObjectLiteralCast and goog.reflect.object
800         // Ignore these types of literals.
801         ObjectType objectType = ObjectType.cast(type);
802         if (objectType == null
803             || n.getBooleanProp(Node.REFLECTED_OBJECT)
804             || objectType.isEnumType()) {
805           return scope;
806         }
807     
808         String qObjName = NodeUtil.getBestLValueName(
809             NodeUtil.getBestLValue(n));
810         for (Node name = n.getFirstChild(); name != null;
811              name = name.getNext()) {
812           String memberName = NodeUtil.getObjectLitKeyName(name);
813           if (memberName != null) {
814             JSType rawValueType =  name.getFirstChild().getJSType();
815             JSType valueType =
816                 TypeCheck.getObjectLitKeyTypeFromValueType(name, rawValueType);
817             if (valueType == null) {
818               valueType = unknownType;
819             }
820             objectType.defineInferredProperty(memberName, valueType, name);
821     
822             // Do normal flow inference if this is a direct property assignment.
823             if (qObjName != null && name.isStringKey()) {
824               String qKeyName = qObjName + "." + memberName;
825               TypedVar var = syntacticScope.getVar(qKeyName);
826               JSType oldType = var == null ? null : var.getType();
827               if (var != null && var.isTypeInferred()) {
828                 var.setType(oldType == null ?
829                     valueType : oldType.getLeastSupertype(oldType));
830               }
831     
832               scope.inferQualifiedSlot(name, qKeyName,
833                   oldType == null ? unknownType : oldType,
834                   valueType, false);
835             }
836           } else {
837             n.setJSType(unknownType);
838           }
839         }
840         return scope;
841       }

843       private FlowScope traverseAdd(Node n, FlowScope scope) {
844         Node left = n.getFirstChild();
845         Node right = left.getNext();
846         scope = traverseChildren(n, scope);
847     
848         JSType leftType = left.getJSType();
849         JSType rightType = right.getJSType();
850     
851         JSType type = unknownType;
852         if (leftType != null && rightType != null) {
853           boolean leftIsUnknown = leftType.isUnknownType();
854           boolean rightIsUnknown = rightType.isUnknownType();
855           if (leftIsUnknown && rightIsUnknown) {
856             type = unknownType;
857           } else if ((!leftIsUnknown && leftType.isString()) ||
858                      (!rightIsUnknown && rightType.isString())) {
859             type = getNativeType(STRING_TYPE);
860           } else if (leftIsUnknown || rightIsUnknown) {
861             type = unknownType;
862           } else if (isAddedAsNumber(leftType) && isAddedAsNumber(rightType)) {
863             type = getNativeType(NUMBER_TYPE);
864           } else {
865             type = registry.createUnionType(STRING_TYPE, NUMBER_TYPE);
866           }
867         }
868         n.setJSType(type);
869     
870         if (n.isAssignAdd()) {
871           updateScopeForTypeChange(scope, left, leftType, type);
872         }
873     
874         return scope;
875       }

882       private FlowScope traverseHook(Node n, FlowScope scope) {
883         Node condition = n.getFirstChild();
884         Node trueNode = condition.getNext();
885         Node falseNode = n.getLastChild();
886     
887         // verify the condition
888         scope = traverse(condition, scope);
889     
890         // reverse abstract interpret the condition to produce two new scopes
891         FlowScope trueScope = reverseInterpreter.
892             getPreciserScopeKnowingConditionOutcome(
893                 condition, scope, true);
894         FlowScope falseScope = reverseInterpreter.
895             getPreciserScopeKnowingConditionOutcome(
896                 condition, scope, false);
897     
898         // traverse the true node with the trueScope
899         traverse(trueNode, trueScope.createChildFlowScope());
900     
901         // traverse the false node with the falseScope
902         traverse(falseNode, falseScope.createChildFlowScope());
903     
904         // meet true and false nodes' types and assign
905         JSType trueType = trueNode.getJSType();
906         JSType falseType = falseNode.getJSType();
907         if (trueType != null && falseType != null) {
908           n.setJSType(trueType.getLeastSupertype(falseType));
909         } else {
910           n.setJSType(null);
911         }
912     
913         return scope.createChildFlowScope();
914       }

916       private FlowScope traverseCall(Node n, FlowScope scope) {
917         scope = traverseChildren(n, scope);
918     
919         Node left = n.getFirstChild();
920         JSType functionType = getJSType(left).restrictByNotNullOrUndefined();
921         if (functionType.isFunctionType()) {
922           FunctionType fnType = functionType.toMaybeFunctionType();
923           n.setJSType(fnType.getReturnType());
924           backwardsInferenceFromCallSite(n, fnType);
925         } else if (functionType.isEquivalentTo(
926             getNativeType(CHECKED_UNKNOWN_TYPE))) {
927           n.setJSType(getNativeType(CHECKED_UNKNOWN_TYPE));
928         }
929     
930         scope = tightenTypesAfterAssertions(scope, n);
931         return scope;
932       }

934       private FlowScope tightenTypesAfterAssertions(FlowScope scope,
935           Node callNode) {
936         Node left = callNode.getFirstChild();
937         Node firstParam = left.getNext();
938         AssertionFunctionSpec assertionFunctionSpec =
939             assertionFunctionsMap.get(left.getQualifiedName());
940         if (assertionFunctionSpec == null || firstParam == null) {
941           return scope;
942         }
943         Node assertedNode = assertionFunctionSpec.getAssertedParam(firstParam);
944         if (assertedNode == null) {
945           return scope;
946         }
947         JSType assertedType = assertionFunctionSpec.getAssertedOldType(
948             callNode, registry);
949         String assertedNodeName = assertedNode.getQualifiedName();
950     
951         JSType narrowed;
952         // Handle assertions that enforce expressions evaluate to true.
953         if (assertedType == null) {
954           // Handle arbitrary expressions within the assert.
955           scope = reverseInterpreter.getPreciserScopeKnowingConditionOutcome(
956               assertedNode, scope, true);
957           // Build the result of the assertExpression
958           narrowed = getJSType(assertedNode).restrictByNotNullOrUndefined();
959         } else {
960           // Handle assertions that enforce expressions are of a certain type.
961           JSType type = getJSType(assertedNode);
962           if (assertedType.isUnknownType() || type.isUnknownType()) {
963             narrowed = assertedType;
964           } else {
965             narrowed = type.getGreatestSubtype(assertedType);
966           }
967           if (assertedNodeName != null && type.differsFrom(narrowed)) {
968             scope = narrowScope(scope, assertedNode, narrowed);
969           }
970         }
971     
972         callNode.setJSType(narrowed);
973         return scope;
974       }

977         if (node.isThis()) {
978           // "this" references don't need to be modeled in the control flow graph.
979           return scope;

992       /**
993        * We only do forward type inference. We do not do full backwards
994        * type inference.
995        *
996        * In other words, if we have,
997        * <code>
998        * var x = f();
999        * g(x);
1000        * </code>
1001        * a forward type-inference engine would try to figure out the type
1002        * of "x" from the return type of "f". A backwards type-inference engine
1003        * would try to figure out the type of "x" from the parameter type of "g".
1004        *
1005        * However, there are a few special syntactic forms where we do some
1006        * some half-assed backwards type-inference, because programmers
1007        * expect it in this day and age. To take an example from Java,
1008        * <code>
1009        * List<String> x = Lists.newArrayList();
1010        * </code>
1011        * The Java compiler will be able to infer the generic type of the List
1012        * returned by newArrayList().
1013        *
1014        * In much the same way, we do some special-case backwards inference for
1015        * JS. Those cases are enumerated here.
1016        */
1017       private void backwardsInferenceFromCallSite(Node n, FunctionType fnType) {
1018         boolean updatedFnType = inferTemplatedTypesForCall(n, fnType);
1019         if (updatedFnType) {
1020           fnType = n.getFirstChild().getJSType().toMaybeFunctionType();
1021         }
1022         updateTypeOfParameters(n, fnType);
1023         updateBind(n);
1024       }

1026       /**
1027        * When "bind" is called on a function, we infer the type of the returned
1028        * "bound" function by looking at the number of parameters in the call site.
1029        * We also infer the "this" type of the target, if it's a function expression.
1030        */
1031       private void updateBind(Node n) {
1032         CodingConvention.Bind bind =
1033             compiler.getCodingConvention().describeFunctionBind(n, false, true);
1034         if (bind == null) {
1035           return;
1036         }
1037     
1038         Node target = bind.target;
1039         FunctionType callTargetFn = getJSType(target)
1040             .restrictByNotNullOrUndefined().toMaybeFunctionType();
1041         if (callTargetFn == null) {
1042           return;
1043         }
1044     
1045         if (bind.thisValue != null && target.isFunction()) {
1046           JSType thisType = getJSType(bind.thisValue);
1047           if (thisType.toObjectType() != null && !thisType.isUnknownType()
1048               && callTargetFn.getTypeOfThis().isUnknownType()) {
1049             callTargetFn = new FunctionBuilder(registry)
1050                 .copyFromOtherFunction(callTargetFn)
1051                 .withTypeOfThis(thisType.toObjectType())
1052                 .build();
1053             target.setJSType(callTargetFn);
1054           }
1055         }
1056     
1057         n.setJSType(
1058             callTargetFn.getBindReturnType(
1059                 // getBindReturnType expects the 'this' argument to be included.
1060                 bind.getBoundParameterCount() + 1));
1061       }

1063       /**
1064        * For functions with function parameters, type inference will set the type of
1065        * a function literal argument from the function parameter type.
1066        */
1067       private void updateTypeOfParameters(Node n, FunctionType fnType) {
1068         int i = 0;
1069         int childCount = n.getChildCount();
1070         for (Node iParameter : fnType.getParameters()) {
1071           if (i + 1 >= childCount) {
1072             // TypeCheck#visitParametersList will warn so we bail.
1073             return;
1074           }
1075     
1076           JSType iParameterType = getJSType(iParameter);
1077           Node iArgument = n.getChildAtIndex(i + 1);
1078           JSType iArgumentType = getJSType(iArgument);
1079           inferPropertyTypesToMatchConstraint(iArgumentType, iParameterType);
1080     
1081           // If the parameter to the call is a function expression, propagate the
1082           // function signature from the call site to the function node.
1083     
1084           // Filter out non-function types (such as null and undefined) as
1085           // we only care about FUNCTION subtypes here.
1086           FunctionType restrictedParameter = null;
1087           if (iParameterType.isUnionType()) {
1088             UnionType union = iParameterType.toMaybeUnionType();
1089             for (JSType alternative : union.getAlternates()) {
1090               if (alternative.isFunctionType()) {
1091                 // There is only one function type per union.
1092                 restrictedParameter = alternative.toMaybeFunctionType();
1093                 break;
1094               }
1095             }
1096           } else {
1097             restrictedParameter = iParameterType.toMaybeFunctionType();
1098           }
1099     
1100           if (restrictedParameter != null
1101               && iArgument.isFunction()
1102               && iArgumentType.isFunctionType()) {
1103             FunctionType argFnType = iArgumentType.toMaybeFunctionType();
1104             boolean declared = iArgument.getJSDocInfo() != null;
1105             iArgument.setJSType(
1106                 matchFunction(restrictedParameter, argFnType, declared));
1107           }
1108           i++;
1109         }
1110       }

1112       /**
1113        * Take the current function type, and try to match the expected function
1114        * type. This is a form of backwards-inference, like record-type constraint
1115        * matching.
1116        */
1117       private FunctionType matchFunction(
1118           FunctionType expectedType, FunctionType currentType, boolean declared) {
1119         if (declared) {
1120           // If the function was declared but it doesn't have a known "this"
1121           // but the expected type does, back fill it.
1122           if (currentType.getTypeOfThis().isUnknownType()
1123               && !expectedType.getTypeOfThis().isUnknownType()) {
1124             FunctionType replacement = new FunctionBuilder(registry)
1125                 .copyFromOtherFunction(currentType)
1126                 .withTypeOfThis(expectedType.getTypeOfThis())
1127                 .build();
1128              return replacement;
1129           }
1130         } else {
1131           // For now, we just make sure the current type has enough
1132           // arguments to match the expected type, and return the
1133           // expected type if it does.
1134           if (currentType.getMaxArguments() <= expectedType.getMaxArguments()) {
1135             return expectedType;
1136           }
1137         }
1138         return currentType;
1139       }

1141       private Map<TemplateType, JSType> inferTemplateTypesFromParameters(
1142           FunctionType fnType, Node call) {
1143         if (fnType.getTemplateTypeMap().getTemplateKeys().isEmpty()) {
1144           return Collections.emptyMap();
1145         }
1146     
1147         Map<TemplateType, JSType> resolvedTypes = Maps.newIdentityHashMap();
1148         Set<JSType> seenTypes = Sets.newIdentityHashSet();
1149     
1150         Node callTarget = call.getFirstChild();
1151         if (NodeUtil.isGet(callTarget)) {
1152           Node obj = callTarget.getFirstChild();
1153           maybeResolveTemplatedType(
1154               fnType.getTypeOfThis(),
1155               getJSType(obj),
1156               resolvedTypes,
1157               seenTypes);
1158         }
1159     
1160         if (call.hasMoreThanOneChild()) {
1161           maybeResolveTemplateTypeFromNodes(
1162               fnType.getParameters(),
1163               call.getSecondChild().siblings(),
1164               resolvedTypes,
1165               seenTypes);
1166         }
1167         return resolvedTypes;
1168       }

1170       private void maybeResolveTemplatedType(
1171           JSType paramType,
1172           JSType argType,
1173           Map<TemplateType, JSType> resolvedTypes, Set<JSType> seenTypes) {
1174         if (paramType.isTemplateType()) {
1175           // @param {T}
1176           resolvedTemplateType(
1177               resolvedTypes, paramType.toMaybeTemplateType(), argType);
1178         } else if (paramType.isUnionType()) {
1179           // @param {Array.<T>|NodeList|Arguments|{length:number}}
1180           UnionType unionType = paramType.toMaybeUnionType();
1181           for (JSType alernative : unionType.getAlternates()) {
1182             maybeResolveTemplatedType(alernative, argType, resolvedTypes, seenTypes);
1183           }
1184         } else if (paramType.isFunctionType()) {
1185           FunctionType paramFunctionType = paramType.toMaybeFunctionType();
1186           FunctionType argFunctionType = argType
1187               .restrictByNotNullOrUndefined()
1188               .collapseUnion()
1189               .toMaybeFunctionType();
1190           if (argFunctionType != null && argFunctionType.isSubtype(paramType)) {
1191             // infer from return type of the function type
1192             maybeResolveTemplatedType(
1193                 paramFunctionType.getTypeOfThis(),
1194                 argFunctionType.getTypeOfThis(), resolvedTypes, seenTypes);
1195             // infer from return type of the function type
1196             maybeResolveTemplatedType(
1197                 paramFunctionType.getReturnType(),
1198                 argFunctionType.getReturnType(), resolvedTypes, seenTypes);
1199             // infer from parameter types of the function type
1200             maybeResolveTemplateTypeFromNodes(
1201                 paramFunctionType.getParameters(),
1202                 argFunctionType.getParameters(), resolvedTypes, seenTypes);
1203           }
1204         } else if (paramType.isRecordType() && !paramType.isNominalType()) {
1205           // @param {{foo:T}}
1206           if (seenTypes.add(paramType)) {
1207             ObjectType paramRecordType = paramType.toObjectType();
1208             ObjectType argObjectType = argType.restrictByNotNullOrUndefined().toObjectType();
1209             if (argObjectType != null && !argObjectType.isUnknownType()
1210                 && !argObjectType.isEmptyType()) {
1211               Set<String> names = paramRecordType.getPropertyNames();
1212               for (String name : names) {
1213                 if (paramRecordType.hasOwnProperty(name) && argObjectType.hasProperty(name)) {
1214                   maybeResolveTemplatedType(paramRecordType.getPropertyType(name),
1215                       argObjectType.getPropertyType(name), resolvedTypes, seenTypes);
1216                 }
1217               }
1218             }
1219             seenTypes.remove(paramType);
1220           }
1221         } else if (paramType.isTemplatizedType()) {
1222           // @param {Array<T>}
1223           ObjectType referencedParamType = paramType
1224               .toMaybeTemplatizedType()
1225               .getReferencedType();
1226           JSType argObjectType = argType
1227               .restrictByNotNullOrUndefined()
1228               .collapseUnion();
1229     
1230     
1231           if (argObjectType.isSubtype(referencedParamType)) {
1232             // If the argument type is a subtype of the parameter type, resolve any
1233             // template types amongst their templatized types.
1234             TemplateTypeMap paramTypeMap = paramType.getTemplateTypeMap();
1235             TemplateTypeMap argTypeMap = argObjectType.getTemplateTypeMap();
1236             for (TemplateType key : paramTypeMap.getTemplateKeys()) {
1237               maybeResolveTemplatedType(
1238                   paramTypeMap.getResolvedTemplateType(key),
1239                   argTypeMap.getResolvedTemplateType(key),
1240                   resolvedTypes, seenTypes);
1241             }
1242           }
1243         }
1244       }

1247           Iterable<Node> declParams,
1248           Iterable<Node> callParams,
1249           Map<TemplateType, JSType> resolvedTypes, Set<JSType> seenTypes) {
1250         maybeResolveTemplateTypeFromNodes(
1251             declParams.iterator(), callParams.iterator(), resolvedTypes, seenTypes);

1254       private void maybeResolveTemplateTypeFromNodes(
1255           Iterator<Node> declParams,
1256           Iterator<Node> callParams,
1257           Map<TemplateType, JSType> resolvedTypes,
1258           Set<JSType> seenTypes) {
1259         while (declParams.hasNext() && callParams.hasNext()) {
1260           Node declParam = declParams.next();

1262               getJSType(declParam),
1263               getJSType(callParams.next()),
1264               resolvedTypes, seenTypes);
1265           if (declParam.isVarArgs()) {
1266             while (callParams.hasNext()) {
1267               maybeResolveTemplatedType(
1268                   getJSType(declParam),
1269                   getJSType(callParams.next()),
1270                   resolvedTypes, seenTypes);
1271             }
1272           }
1273         }

1316           Map<TemplateType, JSType> inferredTypes) {
1317         Map<String, JSType> typeVars = new LinkedHashMap<>();
1318         for (Entry<TemplateType, JSType> e : inferredTypes.entrySet()) {
1319           // Only add the template type that do not have a type transformation
1320           if (!e.getKey().isTypeTransformation()) {
1321             typeVars.put(e.getKey().getReferenceName(), e.getValue());
1322           }
1323         }
1324         return typeVars;
1325       }

1327       /**
1328        * This function will evaluate the type transformations associated to the
1329        * template types
1330        */
1331       private Map<TemplateType, JSType> evaluateTypeTransformations(
1332           ImmutableList<TemplateType> templateTypes,
1333           Map<TemplateType, JSType> inferredTypes) {
1334     
1335         Map<String, JSType> typeVars = null;
1336         Map<TemplateType, JSType> result = null;
1337         TypeTransformation ttlObj = null;
1338     
1339         for (TemplateType type : templateTypes) {
1340           if (type.isTypeTransformation()) {
1341             // Lazy initialization when the first type transformation is found
1342             if (ttlObj == null) {
1343               ttlObj = new TypeTransformation(compiler, syntacticScope);
1344               typeVars = buildTypeVariables(inferredTypes);
1345               result = new LinkedHashMap<>();
1346             }
1347             // Evaluate the type transformation expression using the current
1348             // known types for the template type variables
1349             JSType transformedType = ttlObj.eval(type.getTypeTransformation(),
1350                     ImmutableMap.<String, JSType>copyOf(typeVars));
1351             result.put(type, transformedType);
1352             // Add the transformed type to the type variables
1353             typeVars.put(type.getReferenceName(), transformedType);
1354           }
1355         }
1356         return result;
1357       }

1359       /**
1360        * For functions with function(this: T, ...) and T as parameters, type
1361        * inference will set the type of this on a function literal argument to the
1362        * the actual type of T.
1363        */
1364       private boolean inferTemplatedTypesForCall(
1365           Node n, FunctionType fnType) {
1366         final ImmutableList<TemplateType> keys = fnType.getTemplateTypeMap()
1367             .getTemplateKeys();
1368         if (keys.isEmpty()) {
1369           return false;
1370         }
1371     
1372         // Try to infer the template types
1373         Map<TemplateType, JSType> rawInferrence = inferTemplateTypesFromParameters(fnType, n);
1374         Map<TemplateType, JSType> inferred = Maps.newIdentityHashMap();
1375         for (TemplateType key : keys) {
1376           JSType type = rawInferrence.get(key);
1377           if (type == null) {
1378             type = unknownType;
1379           }
1380           inferred.put(key, type);
1381         }
1382     
1383         // Try to infer the template types using the type transformations
1384         Map<TemplateType, JSType> typeTransformations =
1385             evaluateTypeTransformations(keys, inferred);
1386         if (typeTransformations != null) {
1387           inferred.putAll(typeTransformations);
1388         }
1389     
1390         // Replace all template types. If we couldn't find a replacement, we
1391         // replace it with UNKNOWN.
1392         TemplateTypeReplacer replacer = new TemplateTypeReplacer(
1393             registry, inferred);
1394         Node callTarget = n.getFirstChild();
1395     
1396         FunctionType replacementFnType = fnType.visit(replacer)
1397             .toMaybeFunctionType();
1398         Preconditions.checkNotNull(replacementFnType);
1399     
1400         callTarget.setJSType(replacementFnType);
1401         n.setJSType(replacementFnType.getReturnType());
1402     
1403         return replacer.madeChanges;
1404       }

1406       private FlowScope traverseNew(Node n, FlowScope scope) {
1407         scope = traverseChildren(n, scope);
1408     
1409         Node constructor = n.getFirstChild();
1410         JSType constructorType = constructor.getJSType();
1411         JSType type = null;
1412         if (constructorType != null) {
1413           constructorType = constructorType.restrictByNotNullOrUndefined();
1414           if (constructorType.isUnknownType()) {
1415             type = unknownType;
1416           } else {
1417             FunctionType ct = constructorType.toMaybeFunctionType();
1418             if (ct == null && constructorType instanceof FunctionType) {
1419               // If constructorType is a NoObjectType, then toMaybeFunctionType will
1420               // return null. But NoObjectType implements the FunctionType
1421               // interface, precisely because it can validly construct objects.
1422               ct = (FunctionType) constructorType;
1423             }
1424             if (ct != null && ct.isConstructor()) {
1425               backwardsInferenceFromCallSite(n, ct);
1426     
1427               // If necessary, create a TemplatizedType wrapper around the instance
1428               // type, based on the types of the constructor parameters.
1429               ObjectType instanceType = ct.getInstanceType();
1430               Map<TemplateType, JSType> inferredTypes =
1431                   inferTemplateTypesFromParameters(ct, n);
1432               if (inferredTypes.isEmpty()) {
1433                 type = instanceType;
1434               } else {
1435                 type = registry.createTemplatizedType(instanceType, inferredTypes);
1436               }
1437             }
1438           }
1439         }
1440         n.setJSType(type);
1441         return scope;
1442       }

1514       private JSType getPropertyType(JSType objType, String propName,
1515           Node n, FlowScope scope) {
1516         // We often have a couple of different types to choose from for the
1517         // property. Ordered by accuracy, we have
1518         // 1) A locally inferred qualified name (which is in the FlowScope)
1519         // 2) A globally declared qualified name (which is in the FlowScope)
1520         // 3) A property on the owner type (which is on objType)
1521         // 4) A name in the type registry (as a last resort)
1522         JSType propertyType = null;
1523         boolean isLocallyInferred = false;
1524     
1525         // Scopes sometimes contain inferred type info about qualified names.
1526         String qualifiedName = n.getQualifiedName();
1527         StaticTypedSlot<JSType> var = scope.getSlot(qualifiedName);
1528         if (var != null) {
1529           JSType varType = var.getType();
1530           if (varType != null) {
1531             boolean isDeclared = !var.isTypeInferred();
1532             isLocallyInferred = (var != syntacticScope.getSlot(qualifiedName));
1533             if (isDeclared || isLocallyInferred) {
1534               propertyType = varType;
1535             }
1536           }
1537         }
1538     
1539         if (propertyType == null && objType != null) {
1540           JSType foundType = objType.findPropertyType(propName);
1541           if (foundType != null) {
1542             propertyType = foundType;
1543           }
1544         }
1545     
1546         if (propertyType != null && objType != null) {
1547           JSType restrictedObjType = objType.restrictByNotNullOrUndefined();
1548           if (!restrictedObjType.getTemplateTypeMap().isEmpty()
1549               && propertyType.hasAnyTemplateTypes()) {
1550             TemplateTypeMap typeMap = restrictedObjType.getTemplateTypeMap();
1551             TemplateTypeMapReplacer replacer = new TemplateTypeMapReplacer(
1552                 registry, typeMap);
1553             propertyType = propertyType.visit(replacer);
1554           }
1555         }
1556     
1557         if ((propertyType == null || propertyType.isUnknownType())
1558             && qualifiedName != null) {
1559           // If we find this node in the registry, then we can infer its type.
1560           ObjectType regType = ObjectType.cast(registry.getType(qualifiedName));
1561           if (regType != null) {
1562             propertyType = regType.getConstructor();
1563           }
1564         }
1565     
1566         if (propertyType == null) {
1567           return unknownType;
1568         } else if (propertyType.isEquivalentTo(unknownType) && isLocallyInferred) {
1569           // If the type has been checked in this scope,
1570           // then use CHECKED_UNKNOWN_TYPE instead to indicate that.
1571           return getNativeType(CHECKED_UNKNOWN_TYPE);
1572         } else {
1573           return propertyType;
1574         }
1575       }

1581       private BooleanOutcomePair traverseShortCircuitingBinOp(
1582           Node n, FlowScope scope) {
1583         Preconditions.checkArgument(n.isAnd() || n.isOr());
1584         boolean nIsAnd = n.isAnd();
1585         Node left = n.getFirstChild();
1586         Node right = n.getLastChild();
1587     
1588         // type the left node
1589         BooleanOutcomePair leftOutcome = traverseWithinShortCircuitingBinOp(
1590             left, scope.createChildFlowScope());
1591         JSType leftType = left.getJSType();
1592     
1593         // reverse abstract interpret the left node to produce the correct
1594         // scope in which to verify the right node
1595         FlowScope rightScope = reverseInterpreter.
1596             getPreciserScopeKnowingConditionOutcome(left,
1597                 leftOutcome.getOutcomeFlowScope(left.getType(), nIsAnd),
1598                 nIsAnd);
1599     
1600         // type the right node
1601         BooleanOutcomePair rightOutcome = traverseWithinShortCircuitingBinOp(
1602             right, rightScope.createChildFlowScope());
1603         JSType rightType = right.getJSType();
1604     
1605         JSType type;
1606         BooleanOutcomePair outcome;
1607         if (leftType != null && rightType != null) {
1608           leftType = leftType.getRestrictedTypeGivenToBooleanOutcome(!nIsAnd);
1609           if (leftOutcome.toBooleanOutcomes == BooleanLiteralSet.get(!nIsAnd)) {
1610             // Either n is && and lhs is false, or n is || and lhs is true.
1611             // Use the restricted left type; the right side never gets evaluated.
1612             type = leftType;
1613             outcome = leftOutcome;
1614           } else {
1615             // Use the join of the restricted left type knowing the outcome of the
1616             // ToBoolean predicate and of the right type.
1617             type = leftType.getLeastSupertype(rightType);
1618             outcome = new BooleanOutcomePair(
1619                 joinBooleanOutcomes(nIsAnd,
1620                     leftOutcome.toBooleanOutcomes, rightOutcome.toBooleanOutcomes),
1621                 joinBooleanOutcomes(nIsAnd,
1622                     leftOutcome.booleanValues, rightOutcome.booleanValues),
1623                 leftOutcome.getJoinedFlowScope(),
1624                 rightOutcome.getJoinedFlowScope());
1625           }
1626           // Exclude the boolean type if the literal set is empty because a boolean
1627           // can never actually be returned.
1628           if (outcome.booleanValues == BooleanLiteralSet.EMPTY &&
1629               getNativeType(BOOLEAN_TYPE).isSubtype(type)) {
1630             // Exclusion only makes sense for a union type.
1631             if (type.isUnionType()) {
1632               type = type.toMaybeUnionType().getRestrictedUnion(
1633                   getNativeType(BOOLEAN_TYPE));
1634             }
1635           }
1636         } else {
1637           type = null;
1638           outcome = new BooleanOutcomePair(
1639               BooleanLiteralSet.BOTH, BooleanLiteralSet.BOTH,
1640               leftOutcome.getJoinedFlowScope(),
1641               rightOutcome.getJoinedFlowScope());
1642         }
1643         n.setJSType(type);
1644         return outcome;
1645       }

1647       private BooleanOutcomePair traverseWithinShortCircuitingBinOp(
1648           Node n, FlowScope scope) {
1649         switch (n.getType()) {
1650           case AND:
1651             return traverseAnd(n, scope);
1652     
1653           case OR:
1654             return traverseOr(n, scope);
1655     
1656           default:
1657             scope = traverse(n, scope);
1658             return newBooleanOutcomePair(n.getJSType(), scope);
1659         }
1660       }

1662       private static BooleanLiteralSet joinBooleanOutcomes(
1663           boolean isAnd, BooleanLiteralSet left, BooleanLiteralSet right) {
1664         // A truthy value on the lhs of an {@code &&} can never make it to the
1665         // result. Same for a falsy value on the lhs of an {@code ||}.
1666         // Hence the intersection.
1667         return right.union(left.intersection(BooleanLiteralSet.get(!isAnd)));
1668       }

1670       /**
1671        * When traversing short-circuiting binary operations, we need to keep track
1672        * of two sets of boolean literals:
1673        * 1. {@code toBooleanOutcomes}: boolean literals as converted from any types,
1674        * 2. {@code booleanValues}: boolean literals from just boolean types.
1675        */
1676       private final class BooleanOutcomePair {
1677         final BooleanLiteralSet toBooleanOutcomes;
1678         final BooleanLiteralSet booleanValues;
1679     
1680         // The scope if only half of the expression executed, when applicable.
1681         final FlowScope leftScope;
1682     
1683         // The scope when the whole expression executed.
1684         final FlowScope rightScope;
1685     
1686         // The scope when we don't know how much of the expression is executed.
1687         FlowScope joinedScope = null;
1688     
1689         BooleanOutcomePair(
1690             BooleanLiteralSet toBooleanOutcomes, BooleanLiteralSet booleanValues,
1691             FlowScope leftScope, FlowScope rightScope) {
1692           this.toBooleanOutcomes = toBooleanOutcomes;
1693           this.booleanValues = booleanValues;
1694           this.leftScope = leftScope;
1695           this.rightScope = rightScope;
1696         }
1697     
1698         /**
1699          * Gets the safe estimated scope without knowing if all of the
1700          * subexpressions will be evaluated.
1701          */
1702         FlowScope getJoinedFlowScope() {
1703           if (joinedScope == null) {
1704             if (leftScope == rightScope) {
1705               joinedScope = rightScope;
1706             } else {
1707               joinedScope = join(leftScope, rightScope);
1708             }
1709           }
1710           return joinedScope;
1711         }
1712     
1713         /**
1714          * Gets the outcome scope if we do know the outcome of the entire
1715          * expression.
1716          */
1717         FlowScope getOutcomeFlowScope(Token nodeType, boolean outcome) {
1718           if (nodeType == Token.AND && outcome ||
1719               nodeType == Token.OR && !outcome) {
1720             // We know that the whole expression must have executed.
1721             return rightScope;
1722           } else {
1723             return getJoinedFlowScope();
1724           }
1725         }
1726       }

1728       private BooleanOutcomePair newBooleanOutcomePair(
1729           JSType jsType, FlowScope flowScope) {
1730         if (jsType == null) {
1731           return new BooleanOutcomePair(
1732               BooleanLiteralSet.BOTH, BooleanLiteralSet.BOTH, flowScope, flowScope);
1733         }
1734         return new BooleanOutcomePair(jsType.getPossibleToBooleanOutcomes(),
1735             registry.getNativeType(BOOLEAN_TYPE).isSubtype(jsType) ?
1736                 BooleanLiteralSet.BOTH : BooleanLiteralSet.EMPTY,
1737             flowScope, flowScope);
1738       }
