19     import com.google.common.collect.ImmutableList;
20     import com.google.javascript.jscomp.parsing.parser.util.SourceRange;
21     
22     public class ComprehensionTree extends ParseTree {
23       public static enum ComprehensionType {
24         ARRAY,
25         GENERATOR
26       };
27     
28       public final ComprehensionType type;
29     
30       // The "for of" and "if" trees.
31       public final ImmutableList<ParseTree> children;
32     
33       // The AssignmentExpression at the end of the ComprehensionTail.
34       public final ParseTree tailExpression;
35     
36       public ComprehensionTree(
37           SourceRange location,
38           ComprehensionType type,
39           ImmutableList<ParseTree> children,
40           ParseTree tailExpression) {
41         super(ParseTreeType.COMPREHENSION, location);
42         this.type = type;
43         this.children = children;
44         this.tailExpression = tailExpression;
45       }
46     }
