17     package com.google.javascript.jscomp.parsing.parser;
18     
19     import com.google.common.collect.ImmutableMap;
20     import com.google.common.collect.Maps;
21     
22     import java.util.EnumMap;
23     import java.util.Map;
24     
25     /**
26      * The JavaScript keywords.
27      */
28     public enum Keywords {
29     
30         // 7.6.1.1 Keywords
31         BREAK("break", TokenType.BREAK),
32         CASE("case", TokenType.CASE),
33         CATCH("catch", TokenType.CATCH),
34         CONTINUE("continue", TokenType.CONTINUE),
35         DEBUGGER("debugger", TokenType.DEBUGGER),
36         DEFAULT("default", TokenType.DEFAULT),
37         DELETE("delete", TokenType.DELETE),
38         DO("do", TokenType.DO),
39         ELSE("else", TokenType.ELSE),
40         FINALLY("finally", TokenType.FINALLY),
41         FOR("for", TokenType.FOR),
42         FUNCTION("function", TokenType.FUNCTION),
43         IF("if", TokenType.IF),
44         IN("in", TokenType.IN),
45         INSTANCEOF("instanceof", TokenType.INSTANCEOF),
46         NEW("new", TokenType.NEW),
47         RETURN("return", TokenType.RETURN),
48         SWITCH("switch", TokenType.SWITCH),
49         THIS("this", TokenType.THIS),
50         THROW("throw", TokenType.THROW),
51         TRY("try", TokenType.TRY),
52         TYPEOF("typeof", TokenType.TYPEOF),
53         VAR("var", TokenType.VAR),
54         VOID("void", TokenType.VOID),
55         WHILE("while", TokenType.WHILE),
56         WITH("with", TokenType.WITH),
57     
58         // 7.6.1.2 Future Reserved Words
59         CLASS("class", TokenType.CLASS),
60         CONST("const", TokenType.CONST),
61         ENUM("enum", TokenType.ENUM),
62         EXPORT("export", TokenType.EXPORT),
63         EXTENDS("extends", TokenType.EXTENDS),
64         IMPORT("import", TokenType.IMPORT),
65         SUPER("super", TokenType.SUPER),
66     
67         // Future Reserved Words in a strict context
68         IMPLEMENTS("implements", TokenType.IMPLEMENTS),
69         INTERFACE("interface", TokenType.INTERFACE),
70         LET("let", TokenType.LET),
71         PACKAGE("package", TokenType.PACKAGE),
72         PRIVATE("private", TokenType.PRIVATE),
73         PROTECTED("protected", TokenType.PROTECTED),
74         PUBLIC("public", TokenType.PUBLIC),
75         STATIC("static", TokenType.STATIC),
76         YIELD("yield", TokenType.YIELD),
77     
78         //7.8 Literals
79         NULL("null", TokenType.NULL),
80         TRUE("true", TokenType.TRUE),
81         FALSE("false", TokenType.FALSE),
82     
83         // TypeScript
84         DECLARE("declare", TokenType.DECLARE),
85         TYPE("type", TokenType.TYPE),
86         MODULE("module", TokenType.MODULE), // Only accepted as alias for namespaces.
87         NAMESPACE("namespace", TokenType.NAMESPACE);
88     
89       private static final Map<String, Keywords> KEYWORDS_BY_NAME;
90       private static final Map<TokenType, Keywords> KEYWORDS_BY_TYPE;
91     
92       static {
93         ImmutableMap.Builder<String, Keywords> keywordsByName = ImmutableMap.builder();
94         EnumMap<TokenType, Keywords> keywordsByType = new EnumMap<>(TokenType.class);
95         for (Keywords kw : Keywords.values()) {
96           keywordsByName.put(kw.value, kw);
97           keywordsByType.put(kw.type, kw);
98         }
99         KEYWORDS_BY_NAME = keywordsByName.build();
100         KEYWORDS_BY_TYPE = Maps.immutableEnumMap(keywordsByType);
101       }
102     
103       public final String value;
104       public final TokenType type;
105     
106       Keywords(String value, TokenType type) {
107         this.value = value;
108         this.type = type;
109       }
110     
111       @Override
112       public String toString() {
113         return value;
114       }
115     
116       public static boolean isKeyword(String value) {
117         return get(value) != null;
118       }
119     
120       public static boolean isKeyword(TokenType token) {
121         return get(token) != null;
122       }
123     
124       public static boolean isTypeScriptSpecificKeyword(String value) {
125         switch (value) {
126           case "declare":
127           case "type":
128           case "module":
129           case "namespace":
130             return true;
131           default:
132             return false;
133         }
134       }
135     
136       /**
137        * Returns true if {@code token} is a "future reserved word" which can
138        * be used as a variable identifier, but only in non-strict mode.
139        */
140       public static boolean isStrictKeyword(TokenType token) {
141         switch(token) {
142           case IMPLEMENTS:
143           case INTERFACE:
144           case LET:
145           case PACKAGE:
146           case PRIVATE:
147           case PROTECTED:
148           case PUBLIC:
149           case STATIC:
150           case YIELD:
151             return true;
152           default:
153             return false;
154         }
155       }
156     
157       public static TokenType getTokenType(String value) {
158         return KEYWORDS_BY_NAME.get(value).type;
159       }
160     
161       public static Keywords get(String value) {
162         return KEYWORDS_BY_NAME.get(value);
163       }
164     
165       public static Keywords get(TokenType token) {
166         return KEYWORDS_BY_TYPE.get(token);
167       }
168     }
