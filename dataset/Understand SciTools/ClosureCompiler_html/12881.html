213       /**
214        * Creates a scope with all types declared. Declares newly discovered types
215        * and type properties in the type registry.
216        */
217       @Override
218       public TypedScope createScope(Node root, Scope parent) {
219         Preconditions.checkArgument(parent == null || parent instanceof TypedScope);
220         TypedScope typedParent = (TypedScope) parent;
221         // Constructing the global scope is very different than constructing
222         // inner scopes, because only global scopes can contain named classes that
223         // show up in the type registry.
224         TypedScope newScope = null;
225         AbstractScopeBuilder scopeBuilder = null;
226         if (typedParent == null) {
227           JSType globalThis =
228               typeRegistry.getNativeObjectType(JSTypeNative.GLOBAL_THIS);
229     
230           // Mark the main root, the externs root, and the src root
231           // with the global this type.
232           root.setJSType(globalThis);
233           root.getFirstChild().setJSType(globalThis);
234           root.getLastChild().setJSType(globalThis);
235     
236           // Run a first-order analysis over the syntax tree.
237           (new FirstOrderFunctionAnalyzer(compiler, functionAnalysisResults))
238               .process(root.getFirstChild(), root.getLastChild());
239     
240           // Find all the classes in the global scope.
241           newScope = createInitialScope(root);
242     
243           GlobalScopeBuilder globalScopeBuilder = new GlobalScopeBuilder(newScope);
244           scopeBuilder = globalScopeBuilder;
245           NodeTraversal.traverseTyped(compiler, root, scopeBuilder);
246         } else {
247           newScope = new TypedScope(typedParent, root);
248           LocalScopeBuilder localScopeBuilder = new LocalScopeBuilder(newScope);
249           scopeBuilder = localScopeBuilder;
250           localScopeBuilder.build();
251         }
252     
253         scopeBuilder.resolveStubDeclarations();
254     
255         if (typedParent == null) {
256           codingConvention.defineDelegateProxyPrototypeProperties(
257               typeRegistry, newScope, delegateProxyPrototypes,
258               delegateCallingConventions);
259         }
260     
261         newScope.setTypeResolver(scopeBuilder);
262         return newScope;
263       }

265       /**
266        * Patches a given global scope by removing variables previously declared in
267        * a script and re-traversing a new version of that script.
268        *
269        * @param globalScope The global scope generated by {@code createScope}.
270        * @param scriptRoot The script that is modified.
271        */
272       void patchGlobalScope(TypedScope globalScope, Node scriptRoot) {
273         // Preconditions: This is supposed to be called only on (named) SCRIPT nodes
274         // and a global typed scope should have been generated already.
275         Preconditions.checkState(scriptRoot.isScript());
276         Preconditions.checkNotNull(globalScope);
277         Preconditions.checkState(globalScope.isGlobal());
278     
279         String scriptName = NodeUtil.getSourceName(scriptRoot);
280         Preconditions.checkNotNull(scriptName);
281         for (Node node : ImmutableList.copyOf(functionAnalysisResults.keySet())) {
282           if (scriptName.equals(NodeUtil.getSourceName(node))) {
283             functionAnalysisResults.remove(node);
284           }
285         }
286     
287         (new FirstOrderFunctionAnalyzer(
288             compiler, functionAnalysisResults)).process(null, scriptRoot);
289     
290         // TODO(bashir): Variable declaration is not the only side effect of last
291         // global scope generation but here we only wipe that part off!
292     
293         // Remove all variables that were previously declared in this scripts.
294         // First find all vars to remove then remove them because of iterator!
295         List<TypedVar> varsToRemove = new ArrayList<>();
296         for (TypedVar oldVar : globalScope.getVarIterable()) {
297           if (scriptName.equals(oldVar.getInputName())) {
298             varsToRemove.add(oldVar);
299           }
300         }
301         for (TypedVar var : varsToRemove) {
302           globalScope.undeclare(var);
303           globalScope.getTypeOfThis().toObjectType().removeProperty(var.getName());
304         }
305     
306         // Now re-traverse the given script.
307         GlobalScopeBuilder scopeBuilder = new GlobalScopeBuilder(globalScope);
308         NodeTraversal.traverseTyped(compiler, scriptRoot, scopeBuilder);
309       }

410       private abstract class AbstractScopeBuilder
411           implements NodeTraversal.Callback, TypedScope.TypeResolver {
412     
413         /**
414          * The scope that we're building.
415          */
416         final TypedScope scope;
417     
418         private final List<DeferredSetType> deferredSetTypes =
419              new ArrayList<>();
420     
421         /**
422          * Functions that we found in the global scope and not in externs.
423          */
424         private final List<Node> nonExternFunctions = new ArrayList<>();
425     
426         /**
427          * Object literals with a @lends annotation aren't analyzed until we
428          * reach the root of the statement they're defined in.
429          *
430          * This ensures that if there are any @lends annotations on the object
431          * literals, the type on the @lends annotation resolves correctly.
432          *
433          * For more information, see
434          * http://code.google.com/p/closure-compiler/issues/detail?id=314
435          */
436         private List<Node> lentObjectLiterals = null;
437     
438         /**
439          * Type-less stubs.
440          *
441          * If at the end of traversal, we still don't have types for these
442          * stubs, then we should declare UNKNOWN types.
443          */
444         private final List<StubDeclaration> stubDeclarations =
445              new ArrayList<>();
446     
447         /**
448          * The current source file that we're in.
449          */
450         private String sourceName = null;
451     
452         /**
453          * The InputId of the current node.
454          */
455         private InputId inputId;
456     
457         private AbstractScopeBuilder(TypedScope scope) {
458           this.scope = scope;
459         }
460     
461         void setDeferredType(Node node, JSType type) {
462           deferredSetTypes.add(new DeferredSetType(node, type));
463         }
464     
465         @Override
466         public void resolveTypes() {
467           // Resolve types and attach them to nodes.
468           for (DeferredSetType deferred : deferredSetTypes) {
469             deferred.resolve(scope);
470           }
471     
472           // Resolve types and attach them to scope slots.
473           for (TypedVar var : scope.getVarIterable()) {
474             var.resolveType(typeParsingErrorReporter);
475           }
476     
477           // Tell the type registry that any remaining types
478           // are unknown.
479           typeRegistry.resolveTypesInScope(scope);
480         }

482         @Override
483         public final boolean shouldTraverse(NodeTraversal t, Node n, Node parent) {
484           inputId = t.getInputId();
485           if (n.isFunction() ||
486               n.isScript()) {
487             Preconditions.checkNotNull(inputId);
488             sourceName = NodeUtil.getSourceName(n);
489           }
490     
491           // We do want to traverse the name of a named function, but we don't
492           // want to traverse the arguments or body.
493           boolean descend = parent == null || !parent.isFunction() ||
494               n == parent.getFirstChild() || parent == scope.getRootNode();
495     
496           if (descend) {
497             // Handle hoisted functions on pre-order traversal, so that they
498             // get hit before other things in the scope.
499             if (NodeUtil.isStatementParent(n)) {
500               for (Node child = n.getFirstChild();
501                    child != null;
502                    child = child.getNext()) {
503                 if (NodeUtil.isHoistedFunctionDeclaration(child)) {
504                   defineFunctionLiteral(child);
505                 }
506               }
507             }
508           }
509     
510           return descend;
511         }
512     
513         @Override
514         public void visit(NodeTraversal t, Node n, Node parent) {
515           inputId = t.getInputId();
516           attachLiteralTypes(n);
517     
518           switch (n.getType()) {
519             case CALL:
520               checkForClassDefiningCalls(n);
521               checkForCallingConventionDefiningCalls(n, delegateCallingConventions);
522               break;
523     
524             case FUNCTION:
525               if (t.getInput() == null || !t.getInput().isExtern()) {
526                 nonExternFunctions.add(n);
527               }
528     
529               // Hoisted functions are handled during pre-traversal.
530               if (!NodeUtil.isHoistedFunctionDeclaration(n)) {
531                 defineFunctionLiteral(n);
532               }
533               break;
534     
535             case ASSIGN:
536               // Handle initialization of properties.
537               Node firstChild = n.getFirstChild();
538               if (firstChild.isGetProp() &&
539                   firstChild.isQualifiedName()) {
540                 maybeDeclareQualifiedName(t, n.getJSDocInfo(),
541                     firstChild, n, firstChild.getNext());
542               }
543               break;
544     
545             case CATCH:
546               defineCatch(n);
547               break;
548     
549             case VAR:
550               defineVar(n);
551               break;
552     
553             case GETPROP:
554               // Handle stubbed properties.
555               if (parent.isExprResult() &&
556                   n.isQualifiedName()) {
557                 maybeDeclareQualifiedName(t, n.getJSDocInfo(), n, parent, null);
558               }
559               break;
560           }
561     
562           // Analyze any @lends object literals in this statement.
563           if (n.getParent() != null && NodeUtil.isStatement(n) &&
564               lentObjectLiterals != null) {
565             for (Node objLit : lentObjectLiterals) {
566               defineObjectLiteral(objLit);
567             }
568             lentObjectLiterals.clear();
569           }
570         }

572         private void attachLiteralTypes(Node n) {
573           switch (n.getType()) {
574             case NULL:
575               n.setJSType(getNativeType(NULL_TYPE));
576               break;
577     
578             case VOID:
579               n.setJSType(getNativeType(VOID_TYPE));
580               break;
581     
582             case STRING:
583               n.setJSType(getNativeType(STRING_TYPE));
584               break;
585     
586             case NUMBER:
587               n.setJSType(getNativeType(NUMBER_TYPE));
588               break;
589     
590             case TRUE:
591             case FALSE:
592               n.setJSType(getNativeType(BOOLEAN_TYPE));
593               break;
594     
595             case REGEXP:
596               n.setJSType(getNativeType(REGEXP_TYPE));
597               break;
598     
599             case OBJECTLIT:
600               JSDocInfo info = n.getJSDocInfo();
601               if (info != null &&
602                   info.getLendsName() != null) {
603                 if (lentObjectLiterals == null) {
604                   lentObjectLiterals = new ArrayList<>();
605                 }
606                 lentObjectLiterals.add(n);
607               } else {
608                 defineObjectLiteral(n);
609               }
610               break;
611     
612             // NOTE(johnlenz): If we ever support Array tuples,
613             // we will need to handle them here as we do object literals
614             // above.
615             case ARRAYLIT:
616               n.setJSType(getNativeType(ARRAY_TYPE));
617               break;
618           }
619         }

621         private void defineObjectLiteral(Node objectLit) {
622           // Handle the @lends annotation.
623           JSType type = null;
624           JSDocInfo info = objectLit.getJSDocInfo();
625           if (info != null && info.getLendsName() != null) {
626             String lendsName = info.getLendsName();
627             TypedVar lendsVar = scope.getVar(lendsName);
628             if (lendsVar == null) {
629               report(JSError.make(objectLit, UNKNOWN_LENDS, lendsName));
630             } else {
631               type = lendsVar.getType();
632               if (type == null) {
633                 type = unknownType;
634               }
635               if (!type.isSubtype(typeRegistry.getNativeType(OBJECT_TYPE))) {
636                 report(JSError.make(
637                     objectLit, LENDS_ON_NON_OBJECT, lendsName, type.toString()));
638                 type = null;
639               } else {
640                 objectLit.setJSType(type);
641               }
642             }
643           }
644     
645           info = NodeUtil.getBestJSDocInfo(objectLit);
646           Node lValue = NodeUtil.getBestLValue(objectLit);
647           String lValueName = NodeUtil.getBestLValueName(lValue);
648           boolean createdEnumType = false;
649           if (info != null && info.hasEnumParameterType()) {
650             type = createEnumTypeFromNodes(objectLit, lValueName, info);
651             createdEnumType = true;
652           }
653     
654           if (type == null) {
655             type = typeRegistry.createAnonymousObjectType(info);
656           }
657     
658           setDeferredType(objectLit, type);
659     
660           // If this is an enum, the properties were already taken care of above.
661           processObjectLitProperties(
662               objectLit, ObjectType.cast(objectLit.getJSType()), !createdEnumType);
663         }

665         /**
666          * Process an object literal and all the types on it.
667          * @param objLit The OBJECTLIT node.
668          * @param objLitType The type of the OBJECTLIT node. This might be a named
669          *     type, because of the lends annotation.
670          * @param declareOnOwner If true, declare properties on the objLitType as
671          *     well. If false, the caller should take care of this.
672          */
673         void processObjectLitProperties(
674             Node objLit, ObjectType objLitType,
675             boolean declareOnOwner) {
676           for (Node keyNode = objLit.getFirstChild(); keyNode != null;
677                keyNode = keyNode.getNext()) {
678             Node value = keyNode.getFirstChild();
679             String memberName = NodeUtil.getObjectLitKeyName(keyNode);
680             JSDocInfo info = keyNode.getJSDocInfo();
681             JSType valueType = getDeclaredType(info, keyNode, value);
682             JSType keyType =  objLitType.isEnumType() ?
683                 objLitType.toMaybeEnumType().getElementsType() :
684                 TypeCheck.getObjectLitKeyTypeFromValueType(keyNode, valueType);
685     
686             // Try to declare this property in the current scope if it
687             // has an authoritative name.
688             String qualifiedName = NodeUtil.getBestLValueName(keyNode);
689             if (qualifiedName != null) {
690               boolean inferred = keyType == null;
691               defineSlot(keyNode, objLit, qualifiedName, keyType, inferred);
692             } else if (keyType != null) {
693               setDeferredType(keyNode, keyType);
694             }
695     
696             if (keyType != null && objLitType != null && declareOnOwner) {
697               // Declare this property on its object literal.
698               objLitType.defineDeclaredProperty(memberName, keyType, keyNode);
699             }
700           }
701         }

704          * Returns the type specified in a JSDoc annotation near a GETPROP or NAME.
705          *
706          * Extracts type information from either the {@code @type} tag or from
707          * the {@code @return} and {@code @param} tags.
708          */
709         private JSType getDeclaredTypeInAnnotation(Node node, JSDocInfo info) {
710           JSType jsType = null;
711           if (info != null) {
712             if (info.hasType()) {
713     
714               ImmutableList<TemplateType> ownerTypeKeys = ImmutableList.of();
715               Node ownerNode = NodeUtil.getBestLValueOwner(node);
716               String ownerName = NodeUtil.getBestLValueName(ownerNode);
717               ObjectType ownerType = null;
718               if (ownerName != null) {

721                   ownerType = getPrototypeOwnerType(
722                       ObjectType.cast(ownerVar.getType()));
723                   if (ownerType != null) {
724                     ownerTypeKeys =
725                         ownerType.getTemplateTypeMap().getTemplateKeys();

727                 }
728               }

741               jsType = createFunctionTypeFromNodes(
742                   null, fnName, info, node);
743             }

852         /**
853          * Creates a new function type, based on the given nodes.
854          *
855          * This handles two cases that are semantically very different, but
856          * are not mutually exclusive:
857          * - A function literal that needs a type attached to it.
858          * - An assignment expression with function-type info in the JsDoc.
859          *
860          * All parameters are optional, and we will do the best we can to create
861          * a function type.
862          *
863          * This function will always create a function type, so only call it if
864          * you're sure that's what you want.
865          *
866          * @param rValue The function node.
867          * @param name the function's name
868          * @param info the {@link JSDocInfo} attached to the function definition
869          * @param lvalueNode The node where this function is being
870          *     assigned. For example, {@code A.prototype.foo = ...} would be used to
871          *     determine that this function is a method of A.prototype. May be
872          *     null to indicate that this is not being assigned to a qualified name.
873          */
874         private FunctionType createFunctionTypeFromNodes(
875             @Nullable Node rValue,
876             @Nullable String name,
877             @Nullable JSDocInfo info,
878             @Nullable Node lvalueNode) {
879           FunctionType functionType = null;
880           if (rValue != null && rValue.isQualifiedName() && scope.isGlobal()) {
881             TypedVar var = scope.getVar(rValue.getQualifiedName());
882             if (var != null && var.getType() != null &&
883                 var.getType().isFunctionType()) {
884               FunctionType aliasedType  = var.getType().toMaybeFunctionType();
885               if ((aliasedType.isConstructor() || aliasedType.isInterface())
886                   && !isGoogAbstractMethod(rValue)) {
887                 functionType = aliasedType;
888     
889                 // TODO(nick): Remove this. This should already be handled by
890                 // normal type resolution.
891                 if (name != null && scope.isGlobal()) {
892                   typeRegistry.declareType(name, functionType.getInstanceType());
893                 }
894               }
895             }
896           }
897     
898           if (functionType == null) {
899             Node errorRoot = rValue == null ? lvalueNode : rValue;
900             boolean isFnLiteral =
901                 rValue != null && rValue.isFunction();
902             Node fnRoot = isFnLiteral ? rValue : null;
903             Node parametersNode = isFnLiteral ?
904                 rValue.getSecondChild() : null;
905     
906             if (info != null && info.hasType()) {
907               JSType type = info.getType().evaluate(scope, typeRegistry);
908     
909               // Known to be not null since we have the FUNCTION token there.
910               type = type.restrictByNotNullOrUndefined();
911               if (type.isFunctionType()) {
912                 functionType = type.toMaybeFunctionType();
913                 functionType.setJSDocInfo(info);
914               }
915             }
916     
917             if (functionType == null) {
918               // Find the type of any overridden function.
919               Node ownerNode = NodeUtil.getBestLValueOwner(lvalueNode);
920               String ownerName = NodeUtil.getBestLValueName(ownerNode);
921               TypedVar ownerVar = null;
922               String propName = null;
923               ObjectType ownerType = null;
924               if (ownerName != null) {
925                 ownerVar = scope.getVar(ownerName);
926                 if (ownerVar != null) {
927                   ownerType = ObjectType.cast(ownerVar.getType());
928                 }
929                 if (name != null) {
930                   propName = name.substring(ownerName.length() + 1);
931                 }
932               }
933     
934               ObjectType prototypeOwner = getPrototypeOwnerType(ownerType);
935               TemplateTypeMap prototypeOwnerTypeMap = null;
936               if (prototypeOwner != null &&
937                   prototypeOwner.getTypeOfThis() != null) {
938                 prototypeOwnerTypeMap =
939                     prototypeOwner.getTypeOfThis().getTemplateTypeMap();
940               }
941     
942               FunctionType overriddenType = null;
943               if (ownerType != null && propName != null) {
944                 overriddenType = findOverriddenFunction(
945                     ownerType, propName, prototypeOwnerTypeMap);
946               }
947     
948               FunctionTypeBuilder builder =
949                   new FunctionTypeBuilder(name, compiler, errorRoot,
950                       scope)
951                   .setContents(getFunctionAnalysisResults(fnRoot))
952                   .inferFromOverriddenFunction(overriddenType, parametersNode)
953                   .inferTemplateTypeName(info, prototypeOwner)
954                   .inferInheritance(info);
955     
956               if (info == null || !info.hasReturnType()) {
957                 /**
958                  * when there is no {@code @return} annotation, look for inline
959                  * return type declaration
960                  */
961                 if (rValue != null && rValue.isFunction() &&
962                     rValue.getFirstChild() != null) {
963                   JSDocInfo nameDocInfo = rValue.getFirstChild().getJSDocInfo();
964                   builder.inferReturnType(nameDocInfo, true);
965                 }
966               } else {
967                 builder.inferReturnType(info, false);
968               }
969     
970               // Infer the context type.
971               boolean searchedForThisType = false;
972               if (ownerType != null && ownerType.isFunctionPrototypeType() &&
973                   ownerType.getOwnerFunction().hasInstanceType()) {
974                 builder.inferThisType(
975                     info, ownerType.getOwnerFunction().getInstanceType());
976                 searchedForThisType = true;
977               } else if (ownerNode != null && ownerNode.isThis()) {
978                 // If we have a 'this' node, use the scope type.
979                 builder.inferThisType(info, scope.getTypeOfThis());
980                 searchedForThisType = true;
981               }
982     
983               if (!searchedForThisType) {
984                 builder.inferThisType(info);
985               }
986     
987               functionType = builder
988                   .inferParameterTypes(parametersNode, info)
989                   .buildAndRegister();
990             }
991           }
992           // set structural interface matching flag
993           if (info != null && info.isInterface() && info.usesImplicitMatch()) {
994             functionType.setImplicitMatch(true);
995           }
996           // all done
997           return functionType;
998         }

1018         /**
1019          * Find the function that's being overridden on this type, if any.
1020          */
1021         private FunctionType findOverriddenFunction(
1022             ObjectType ownerType, String propName, TemplateTypeMap typeMap) {
1023           FunctionType result = null;
1024     
1025           // First, check to see if the property is implemented
1026           // on a superclass.
1027           JSType propType = ownerType.getPropertyType(propName);
1028           if (propType != null && propType.isFunctionType()) {
1029             result =  propType.toMaybeFunctionType();
1030           } else {
1031             // If it's not, then check to see if it's implemented
1032             // on an implemented interface.
1033             for (ObjectType iface :
1034                      ownerType.getCtorImplementedInterfaces()) {
1035               propType = iface.getPropertyType(propName);
1036               if (propType != null && propType.isFunctionType()) {
1037                 result = propType.toMaybeFunctionType();
1038                 break;
1039               }
1040             }
1041           }
1042     
1043           if (result != null && typeMap != null && !typeMap.isEmpty()) {
1044             result = result.visit(
1045                 new TemplateTypeMapReplacer(typeRegistry, typeMap))
1046                 .toMaybeFunctionType();
1047           }
1048     
1049           return result;
1050         }

1052         /**
1053          * Creates a new enum type, based on the given nodes.
1054          *
1055          * This handles two cases that are semantically very different, but
1056          * are not mutually exclusive:
1057          * - An object literal that needs an enum type attached to it.
1058          * - An assignment expression with an enum tag in the JsDoc.
1059          *
1060          * This function will always create an enum type, so only call it if
1061          * you're sure that's what you want.
1062          *
1063          * @param rValue The node of the enum.
1064          * @param name The enum's name
1065          * @param info The {@link JSDocInfo} attached to the enum definition.
1066          */
1067         private EnumType createEnumTypeFromNodes(Node rValue, String name, JSDocInfo info) {
1068           Preconditions.checkNotNull(info);
1069           Preconditions.checkState(info.hasEnumParameterType());
1070     
1071           EnumType enumType = null;
1072           if (rValue != null && rValue.isQualifiedName()) {
1073             // Handle an aliased enum.
1074             TypedVar var = scope.getVar(rValue.getQualifiedName());
1075             if (var != null && var.getType() instanceof EnumType) {
1076               enumType = (EnumType) var.getType();
1077             }
1078           }
1079     
1080           if (enumType == null) {
1081             JSType elementsType =
1082                 info.getEnumParameterType().evaluate(scope, typeRegistry);
1083             enumType = typeRegistry.createEnumType(name, rValue, elementsType);
1084     
1085             if (rValue != null && rValue.isObjectLit()) {
1086               // collect enum elements
1087               Node key = rValue.getFirstChild();
1088               while (key != null) {
1089                 String keyName = key.getString();
1090                 Preconditions.checkNotNull(keyName, "Invalid enum key: %s", key);
1091                 enumType.defineElement(keyName, key);
1092                 key = key.getNext();
1093               }
1094             }
1095           }
1096     
1097           if (name != null && scope.isGlobal()) {
1098             typeRegistry.declareType(name, enumType.getElementsType());
1099           }
1100     
1101           return enumType;
1102         }

1118          * variable's type of {@link JSTypeNative#UNKNOWN_TYPE} if its type is

1149         /**
1150          * Defines a symbol in the current scope.
1151          *
1152          * @param n the defining NAME or GETPROP or object literal key node.
1153          * @param parent the {@code n}'s parent.
1154          * @param variableName The name that this should be known by.
1155          * @param type the variable's type. It may be {@code null} if
1156          *     {@code inferred} is {@code true}.
1157          * @param inferred Whether the type is inferred or declared.
1158          */
1159         void defineSlot(Node n, Node parent, String variableName,
1160             JSType type, boolean inferred) {
1161           Preconditions.checkArgument(!variableName.isEmpty());
1162     
1163           boolean isGlobalVar = n.isName() && scope.isGlobal();
1164           boolean shouldDeclareOnGlobalThis =
1165               isGlobalVar &&
1166               (parent.isVar() ||
1167                parent.isFunction());
1168     
1169           // If n is a property, then we should really declare it in the
1170           // scope where the root object appears. This helps out people
1171           // who declare "global" names in an anonymous namespace.
1172           TypedScope scopeToDeclareIn = scope;
1173           if (n.isGetProp() && !scope.isGlobal() &&
1174               isQnameRootedInGlobalScope(n)) {
1175             TypedScope globalScope = scope.getGlobalScope();
1176     
1177             // don't try to declare in the global scope if there's
1178             // already a symbol there with this name.
1179             if (!globalScope.isDeclared(variableName, false)) {
1180               scopeToDeclareIn = scope.getGlobalScope();
1181             }
1182           }
1183     
1184           // The input may be null if we are working with a AST snippet. So read
1185           // the extern info from the node.
1186           TypedVar newVar = null;
1187     
1188           // declared in closest scope?
1189           CompilerInput input = compiler.getInput(inputId);
1190           if (scopeToDeclareIn.isDeclared(variableName, false)) {
1191             TypedVar oldVar = scopeToDeclareIn.getVar(variableName);
1192             newVar = validator.expectUndeclaredVariable(
1193                 sourceName, input, n, parent, oldVar, variableName, type);
1194           } else {
1195             if (type != null) {
1196               setDeferredType(n, type);
1197             }
1198     
1199             newVar =
1200               scopeToDeclareIn.declare(variableName, n, type, input, inferred);
1201     
1202             if (type instanceof EnumType) {
1203               Node initialValue = newVar.getInitialValue();
1204               boolean isValidValue = initialValue != null &&
1205                   (initialValue.isObjectLit() ||
1206                    initialValue.isQualifiedName());
1207               if (!isValidValue) {
1208                 report(JSError.make(n, ENUM_INITIALIZER));
1209               }
1210             }
1211           }
1212     
1213           // We need to do some additional work for constructors and interfaces.
1214           FunctionType fnType = JSType.toMaybeFunctionType(type);
1215           if (fnType != null &&
1216               // We don't want to look at empty function types.
1217               !type.isEmptyType()) {
1218     
1219             // We want to make sure that when we declare a new instance type
1220             // (with @constructor) that there's actually a ctor for it.
1221             // This doesn't apply to structural constructors (like
1222             // function(new:Array). Checking the constructed type against
1223             // the variable name is a sufficient check for this.
1224             if ((fnType.isConstructor() || fnType.isInterface()) &&
1225                 variableName.equals(fnType.getReferenceName())) {
1226               finishConstructorDefinition(n, variableName, fnType, scopeToDeclareIn,
1227                                           input, newVar);
1228             }
1229           }
1230     
1231           if (shouldDeclareOnGlobalThis) {
1232             ObjectType globalThis =
1233                 typeRegistry.getNativeObjectType(GLOBAL_THIS);
1234             if (inferred) {
1235               globalThis.defineInferredProperty(variableName,
1236                   type == null ?
1237                   getNativeType(JSTypeNative.NO_TYPE) :
1238                   type,
1239                   n);
1240             } else {
1241               globalThis.defineDeclaredProperty(variableName, type, n);
1242             }
1243           }
1244     
1245           if (isGlobalVar && "Window".equals(variableName)
1246               && type != null
1247               && type.isFunctionType()
1248               && type.isConstructor()) {
1249             FunctionType globalThisCtor =
1250                 typeRegistry.getNativeObjectType(GLOBAL_THIS).getConstructor();
1251             globalThisCtor.getInstanceType().clearCachedValues();
1252             globalThisCtor.getPrototype().clearCachedValues();
1253             globalThisCtor
1254                 .setPrototypeBasedOn((type.toMaybeFunctionType()).getInstanceType());
1255           }
1256         }

1258         private void finishConstructorDefinition(
1259             Node n, String variableName, FunctionType fnType,
1260             TypedScope scopeToDeclareIn, CompilerInput input, TypedVar newVar) {
1261           // Declare var.prototype in the scope chain.
1262           FunctionType superClassCtor = fnType.getSuperClassConstructor();
1263           Property prototypeSlot = fnType.getSlot("prototype");
1264     
1265           // When we declare the function prototype implicitly, we
1266           // want to make sure that the function and its prototype
1267           // are declared at the same node. We also want to make sure
1268           // that the if a symbol has both a TypedVar and a JSType, they have
1269           // the same node.
1270           //
1271           // This consistency is helpful to users of SymbolTable,
1272           // because everything gets declared at the same place.
1273           prototypeSlot.setNode(n);
1274     
1275           String prototypeName = variableName + ".prototype";
1276     
1277           // There are some rare cases where the prototype will already
1278           // be declared. See TypedScopeCreatorTest#testBogusPrototypeInit.
1279           // Fortunately, other warnings will complain if this happens.
1280           TypedVar prototypeVar = scopeToDeclareIn.getVar(prototypeName);
1281           if (prototypeVar != null && prototypeVar.scope == scopeToDeclareIn) {
1282             scopeToDeclareIn.undeclare(prototypeVar);
1283           }
1284     
1285           scopeToDeclareIn.declare(prototypeName,
1286               n, prototypeSlot.getType(), input,
1287               /* declared iff there's an explicit supertype */
1288               superClassCtor == null ||
1289               superClassCtor.getInstanceType().isEquivalentTo(
1290                   getNativeType(OBJECT_TYPE)));
1291     
1292           // Make sure the variable is initialized to something if
1293           // it constructs itself.
1294           if (newVar.getInitialValue() == null &&
1295               !n.isFromExterns()) {
1296             report(
1297                 JSError.make(n,
1298                     fnType.isConstructor() ?
1299                     CTOR_INITIALIZER : IFACE_INITIALIZER,
1300                     variableName));
1301           }
1302         }

1326         /**
1327          * Look for a type declaration on a property assignment
1328          * (in an ASSIGN or an object literal key).
1329          * @param info The doc info for this property.
1330          * @param lValue The l-value node.
1331          * @param rValue The node that {@code n} is being initialized to,
1332          *     or {@code null} if this is a stub declaration.
1333          */
1334         JSType getDeclaredType(JSDocInfo info, Node lValue,
1335             @Nullable Node rValue) {
1336           if (info != null && info.hasType()) {
1337             return getDeclaredTypeInAnnotation(lValue, info);
1338           } else if (rValue != null && rValue.isFunction() &&
1339               shouldUseFunctionLiteralType(
1340                   JSType.toMaybeFunctionType(rValue.getJSType()), info, lValue)) {
1341             return rValue.getJSType();
1342           } else if (info != null) {
1343             if (info.hasEnumParameterType()) {
1344               if (rValue != null && rValue.isObjectLit()) {
1345                 return rValue.getJSType();
1346               } else {
1347                 return createEnumTypeFromNodes(rValue, lValue.getQualifiedName(), info);
1348               }
1349             } else if (info.isConstructorOrInterface()) {
1350               return createFunctionTypeFromNodes(
1351                   rValue, lValue.getQualifiedName(), info, lValue);
1352             }
1353           }
1354     
1355           // Check if this is constant, and if it has a known type.
1356           if (NodeUtil.isConstantDeclaration(
1357                   compiler.getCodingConvention(), info, lValue)) {
1358             if (rValue != null) {
1359               JSType rValueType = getDeclaredRValueType(lValue, rValue);
1360               if (rValueType != null) {
1361                 return rValueType;
1362               }
1363             }
1364           }
1365     
1366           return getDeclaredTypeInAnnotation(lValue, info);
1367         }

1369         /**
1370          * Check for common idioms of a typed R-value assigned to a const L-value.
1371          *
1372          * Normally, we would only want this sort of propagation to happen under
1373          * type inference. But we want a declared const to be nameable in a type
1374          * annotation, so we need to figure out the type before we try to resolve
1375          * the annotation.
1376          */
1377         private JSType getDeclaredRValueType(Node lValue, Node rValue) {
1378           // If rValue has a type-cast, we use the type in the type-cast.
1379           JSDocInfo rValueInfo = rValue.getJSDocInfo();
1380           if (rValue.isCast() && rValueInfo != null && rValueInfo.hasType()) {
1381             return rValueInfo.getType().evaluate(scope, typeRegistry);
1382           }
1383     
1384           // Check if the type has already been computed during scope-creation.
1385           // This is mostly useful for literals like BOOLEAN, NUMBER, STRING, and
1386           // OBJECT_LITERAL
1387           JSType type = rValue.getJSType();
1388           if (type != null && !type.isUnknownType()) {
1389             return type;
1390           }
1391     
1392           // If rValue is a name, try looking it up in the current scope.
1393           if (rValue.isQualifiedName()) {
1394             return lookupQualifiedName(rValue);
1395           }
1396     
1397           // Check for simple invariant operations, such as "!x" or "+x" or "''+x"
1398           if (NodeUtil.isBooleanResult(rValue)) {
1399             return getNativeType(BOOLEAN_TYPE);
1400           }
1401     
1402           if (NodeUtil.isNumericResult(rValue)) {
1403             return getNativeType(NUMBER_TYPE);
1404           }
1405     
1406           if (NodeUtil.isStringResult(rValue)) {
1407             return getNativeType(STRING_TYPE);
1408           }
1409     
1410           if (rValue.isNew() && rValue.getFirstChild().isQualifiedName()) {
1411             JSType targetType = lookupQualifiedName(rValue.getFirstChild());
1412             if (targetType != null) {
1413               FunctionType fnType = targetType
1414                   .restrictByNotNullOrUndefined()
1415                   .toMaybeFunctionType();
1416               if (fnType != null && fnType.hasInstanceType()) {
1417                 return fnType.getInstanceType();
1418               }
1419             }
1420           }
1421     
1422           // Check for a very specific JS idiom:
1423           // var x = x || TYPE;
1424           // This is used by Closure's base namespace for esoteric
1425           // reasons, so we only really care about that case.
1426           if (rValue.isOr()) {
1427             Node firstClause = rValue.getFirstChild();
1428             Node secondClause = firstClause.getNext();
1429             boolean namesMatch = firstClause.isName()
1430                 && lValue.isName()
1431                 && firstClause.getString().equals(lValue.getString());
1432             if (namesMatch) {
1433               type = secondClause.getJSType();
1434               if (type != null && !type.isUnknownType()) {
1435                 return type;
1436               }
1437             }
1438           }
1439     
1440           return null;
1441         }

1443         private JSType lookupQualifiedName(Node n) {
1444           String name = n.getQualifiedName();
1445           TypedVar slot = scope.getVar(name);
1446           if (slot != null && !slot.isTypeInferred()) {
1447             JSType type = slot.getType();
1448             if (type != null && !type.isUnknownType()) {
1449               return type;
1450             }
1451           } else if (n.isGetProp()) {
1452             JSType type = lookupQualifiedName(n.getFirstChild());
1453             if (type != null && type.isRecordType()) {
1454               JSType propType = type.findPropertyType(
1455                  n.getLastChild().getString());
1456               return propType;
1457             }
1458           }
1459           return null;
1460         }

1471         /**
1472          * Look for class-defining calls.
1473          * Because JS has no 'native' syntax for defining classes,
1474          * this is often very coding-convention dependent and business-logic heavy.
1475          */
1476         private void checkForClassDefiningCalls(Node n) {
1477           SubclassRelationship relationship =
1478               codingConvention.getClassesDefinedByCall(n);
1479           if (relationship != null) {
1480             ObjectType superClass = TypeValidator.getInstanceOfCtor(
1481                 scope.getVar(relationship.superclassName));
1482             ObjectType subClass = TypeValidator.getInstanceOfCtor(
1483                 scope.getVar(relationship.subclassName));
1484             if (superClass != null && subClass != null) {
1485               // superCtor and subCtor might be structural constructors
1486               // (like {function(new:Object)}) so we need to resolve them back
1487               // to the original ctor objects.
1488               FunctionType superCtor = superClass.getConstructor();
1489               FunctionType subCtor = subClass.getConstructor();
1490               if (superCtor != null && subCtor != null) {
1491                 codingConvention.applySubclassRelationship(
1492                     superCtor, subCtor, relationship.type);
1493               }
1494             }
1495           }
1496     
1497           String singletonGetterClassName =
1498               codingConvention.getSingletonGetterClassName(n);
1499           if (singletonGetterClassName != null) {
1500             ObjectType objectType = ObjectType.cast(
1501                 typeRegistry.getType(singletonGetterClassName));
1502             if (objectType != null) {
1503               FunctionType functionType = objectType.getConstructor();
1504     
1505               if (functionType != null) {
1506                 FunctionType getterType = typeRegistry.createFunctionType(objectType);
1507                 codingConvention.applySingletonGetterOld(functionType, getterType, objectType);
1508               }
1509             }
1510           }
1511     
1512           DelegateRelationship delegateRelationship =
1513               codingConvention.getDelegateRelationship(n);
1514           if (delegateRelationship != null) {
1515             applyDelegateRelationship(delegateRelationship);
1516           }
1517     
1518           ObjectLiteralCast objectLiteralCast =
1519               codingConvention.getObjectLiteralCast(n);
1520           if (objectLiteralCast != null) {
1521             if (objectLiteralCast.diagnosticType == null) {
1522               ObjectType type = ObjectType.cast(
1523                   typeRegistry.getType(objectLiteralCast.typeName));
1524               if (type != null && type.getConstructor() != null) {
1525                 setDeferredType(objectLiteralCast.objectNode, type);
1526                 objectLiteralCast.objectNode.putBooleanProp(
1527                     Node.REFLECTED_OBJECT, true);
1528               } else {
1529                 report(JSError.make(n, CONSTRUCTOR_EXPECTED));
1530               }
1531             } else {
1532               report(JSError.make(n, objectLiteralCast.diagnosticType));
1533             }
1534           }
1535         }

1578         /**
1579          * Declare the symbol for a qualified name in the global scope.
1580          *
1581          * @param info The doc info for this property.
1582          * @param n A top-level GETPROP node (it should not be contained inside
1583          *     another GETPROP).
1584          * @param parent The parent of {@code n}.
1585          * @param rhsValue The node that {@code n} is being initialized to,
1586          *     or {@code null} if this is a stub declaration.
1587          */
1588         void maybeDeclareQualifiedName(NodeTraversal t, JSDocInfo info,
1589             Node n, Node parent, Node rhsValue) {
1590           Node ownerNode = n.getFirstChild();
1591           String ownerName = ownerNode.getQualifiedName();
1592           String qName = n.getQualifiedName();
1593           String propName = n.getLastChild().getString();
1594           Preconditions.checkArgument(qName != null && ownerName != null);
1595     
1596           // Precedence of type information on GETPROPs:
1597           // 1) @type annotation / @enum annotation
1598           // 2) ASSIGN to FUNCTION literal
1599           // 3) @param/@return annotation (with no function literal)
1600           // 4) ASSIGN to something marked @const
1601           // 5) ASSIGN to anything else
1602           //
1603           // 1, 3, and 4 are declarations, 5 is inferred, and 2 is a declaration iff
1604           // the function has JsDoc or has not been declared before.
1605           //
1606           // FUNCTION literals are special because TypedScopeCreator is very smart
1607           // about getting as much type information as possible for them.
1608     
1609           // Determining type for #1 + #2 + #3 + #4
1610           JSType valueType = getDeclaredType(info, n, rhsValue);
1611           if (valueType == null && rhsValue != null) {
1612             // Determining type for #5
1613             valueType = rhsValue.getJSType();
1614           }
1615     
1616           // Function prototypes are special.
1617           // It's a common JS idiom to do:
1618           // F.prototype = { ... };
1619           // So if F does not have an explicitly declared super type,
1620           // allow F.prototype to be redefined arbitrarily.
1621           if ("prototype".equals(propName)) {
1622             TypedVar qVar = scope.getVar(qName);
1623             if (qVar != null) {
1624               // If the programmer has declared that F inherits from Super,
1625               // and they assign F.prototype to an object literal,
1626               // then they are responsible for making sure that the object literal's
1627               // implicit prototype is set up appropriately. We just obey
1628               // the @extends tag.
1629               ObjectType qVarType = ObjectType.cast(qVar.getType());
1630               if (qVarType != null &&
1631                   rhsValue != null &&
1632                   rhsValue.isObjectLit()) {
1633                 typeRegistry.resetImplicitPrototype(
1634                     rhsValue.getJSType(), qVarType.getImplicitPrototype());
1635               } else if (!qVar.isTypeInferred()) {
1636                 // If the programmer has declared that F inherits from Super,
1637                 // and they assign F.prototype to some arbitrary expression,
1638                 // there's not much we can do. We just ignore the expression,
1639                 // and hope they've annotated their code in a way to tell us
1640                 // what props are going to be on that prototype.
1641                 return;
1642               }
1643     
1644               qVar.getScope().undeclare(qVar);
1645             }
1646           }
1647     
1648           if (valueType == null) {
1649             if (parent.isExprResult()) {
1650               stubDeclarations.add(new StubDeclaration(
1651                   n,
1652                   t.getInput() != null && t.getInput().isExtern(),
1653                   ownerName));
1654             }
1655     
1656             return;
1657           }
1658     
1659           boolean inferred = isQualifiedNameInferred(
1660               qName, n, info, rhsValue, valueType);
1661           if (!inferred) {
1662             ObjectType ownerType = getObjectSlot(ownerName);
1663             if (ownerType != null) {
1664               // Only declare this as an official property if it has not been
1665               // declared yet.
1666               boolean isExtern = t.getInput() != null && t.getInput().isExtern();
1667               if ((!ownerType.hasOwnProperty(propName) ||
1668                    ownerType.isPropertyTypeInferred(propName)) &&
1669                   ((isExtern && !ownerType.isNativeObjectType()) ||
1670                    !ownerType.isInstanceType())) {
1671                 // If the property is undeclared or inferred, declare it now.
1672                 ownerType.defineDeclaredProperty(propName, valueType, n);
1673               }
1674             }
1675     
1676             // If the property is already declared, the error will be
1677             // caught when we try to declare it in the current scope.
1678             defineSlot(n, parent, valueType, inferred);
1679           } else if (rhsValue != null && rhsValue.isTrue()) {
1680             // We declare these for delegate proxy method properties.
1681             ObjectType ownerType = getObjectSlot(ownerName);
1682             FunctionType ownerFnType = JSType.toMaybeFunctionType(ownerType);
1683             if (ownerFnType != null) {
1684               JSType ownerTypeOfThis = ownerFnType.getTypeOfThis();
1685               String delegateName = codingConvention.getDelegateSuperclassName();
1686               JSType delegateType = delegateName == null ?
1687                   null : typeRegistry.getType(delegateName);
1688               if (delegateType != null &&
1689                   ownerTypeOfThis.isSubtype(delegateType)) {
1690                 defineSlot(n, parent, getNativeType(BOOLEAN_TYPE), true);
1691               }
1692             }
1693           }
1694         }

1696         /**
1697          * Determines whether a qualified name is inferred.
1698          * NOTE(nicksantos): Determining whether a property is declared or not
1699          * is really really obnoxious.
1700          *
1701          * The problem is that there are two (equally valid) coding styles:
1702          *
1703          * (function() {
1704          *   /* The authoritative definition of goog.bar. /
1705          *   goog.bar = function() {};
1706          * })();
1707          *
1708          * function f() {
1709          *   goog.bar();
1710          *   /* Reset goog.bar to a no-op. /
1711          *   goog.bar = function() {};
1712          * }
1713          *
1714          * In a dynamic language with first-class functions, it's very difficult
1715          * to know which one the user intended without looking at lots of
1716          * contextual information (the second example demonstrates a small case
1717          * of this, but there are some really pathological cases as well).
1718          *
1719          * The current algorithm checks if either the declaration has
1720          * JsDoc type information, or @const with a known type,
1721          * or a function literal with a name we haven't seen before.
1722          */
1723         private boolean isQualifiedNameInferred(
1724             String qName, Node n, JSDocInfo info,
1725             Node rhsValue, JSType valueType) {
1726           if (valueType == null) {
1727             return true;
1728           }
1729     
1730           // Prototypes of constructors and interfaces are always declared.
1731           if (qName != null && qName.endsWith(".prototype")) {
1732             String className = qName.substring(0, qName.lastIndexOf(".prototype"));
1733             TypedVar slot = scope.getSlot(className);
1734             JSType classType = slot == null ? null : slot.getType();
1735             if (classType != null
1736                 && (classType.isConstructor() || classType.isInterface())) {
1737               return false;
1738             }
1739           }
1740     
1741           boolean inferred = true;
1742           if (info != null) {
1743             inferred = !(info.hasType()
1744                 || info.hasEnumParameterType()
1745                 || (NodeUtil.isConstantDeclaration(
1746                         compiler.getCodingConvention(), info, n)
1747                     && valueType != null
1748                     && !valueType.isUnknownType())
1749                 || FunctionTypeBuilder.isFunctionTypeDeclaration(info));
1750           }
1751     
1752           if (inferred && rhsValue != null && rhsValue.isFunction()) {
1753             if (info != null) {
1754               return false;
1755             } else if (!scope.isDeclared(qName, false) &&
1756                 n.isUnscopedQualifiedName()) {
1757     
1758               // Check if this is in a conditional block.
1759               // Functions assigned in conditional blocks are inferred.
1760               for (Node current = n.getParent();
1761                    !(current.isScript() || current.isFunction());
1762                    current = current.getParent()) {
1763                 if (NodeUtil.isControlStructure(current)) {
1764                   return true;
1765                 }
1766               }
1767     
1768               // Check if this is assigned in an inner scope.
1769               // Functions assigned in inner scopes are inferred.
1770               AstFunctionContents contents =
1771                   getFunctionAnalysisResults(scope.getRootNode());
1772               if (contents == null ||
1773                   !contents.getEscapedQualifiedNames().contains(qName)) {
1774                 return false;
1775               }
1776             }
1777           }
1778           return inferred;
1779         }

1887         /**
1888          * Handle typedefs.
1889          * @param candidate A qualified name node.
1890          * @param info JSDoc comments.
1891          */
1892         private void checkForTypedef(Node candidate, JSDocInfo info) {
1893           if (info == null || !info.hasTypedefType()) {
1894             return;
1895           }
1896     
1897           String typedef = candidate.getQualifiedName();
1898           if (typedef == null) {
1899             return;

1901     
1902           // TODO(nicksantos|user): This is a terrible, terrible hack
1903           // to bail out on recursive typedefs. We'll eventually need
1904           // to handle these properly.
1905           typeRegistry.declareType(typedef, unknownType);
1906     
1907           JSType realType = info.getTypedefType().evaluate(scope, typeRegistry);
1908           if (realType == null) {
1909             report(JSError.make(candidate, MALFORMED_TYPEDEF, typedef));
1910           }
1911     
1912           typeRegistry.overwriteDeclaredType(typedef, realType);
1913           if (candidate.isGetProp()) {
1914             defineSlot(candidate, candidate.getParent(),
1915                 getNativeType(NO_TYPE), false);
1916           }
1917         }
1918       } // end GlobalScopeBuilder

1920       /**
1921        * A shallow traversal of a local scope to find all arguments and
1922        * local variables.
1923        */
1924       private final class LocalScopeBuilder extends AbstractScopeBuilder {
1925         private final ObjectType thisTypeForProperties;
1926     
1927         /**
1928          * @param scope The scope that we're building.
1929          */
1930         private LocalScopeBuilder(TypedScope scope) {
1931           super(scope);
1932           thisTypeForProperties = getThisTypeForCollectingProperties();
1933         }
1934     
1935         /**
1936          * Traverse the scope root and build it.
1937          */
1938         void build() {
1939           NodeTraversal.traverseTyped(compiler, scope.getRootNode(), this);
1940     
1941           AstFunctionContents contents =
1942               getFunctionAnalysisResults(scope.getRootNode());
1943           if (contents != null) {
1944             for (String varName : contents.getEscapedVarNames()) {
1945               TypedVar v = scope.getVar(varName);
1946               Preconditions.checkState(v.getScope() == scope);
1947               v.markEscaped();
1948             }
1949     
1950             for (Multiset.Entry<String> entry :
1951                      contents.getAssignedNameCounts().entrySet()) {
1952               TypedVar v = scope.getVar(entry.getElement());
1953               Preconditions.checkState(v.getScope() == scope);
1954               if (entry.getCount() == 1) {
1955                 v.markAssignedExactlyOnce();
1956               }
1957             }
1958           }
1959         }
1960     
1961         /**
1962          * Visit a node in a local scope, and add any local variables or catch
1963          * parameters into the local symbol table.
1964          *
1965          * @param t The node traversal.
1966          * @param n The node being visited.
1967          * @param parent The parent of n
1968          */
1969         @Override public void visit(NodeTraversal t, Node n, Node parent) {
1970           if (n == scope.getRootNode()) {
1971             return;
1972           }
1973     
1974           if (n.isParamList() && parent == scope.getRootNode()) {
1975             handleFunctionInputs(parent);
1976             return;
1977           }
1978     
1979           // Gather the properties declared in the function,
1980           // if that function has a @this type that we can
1981           // build properties on.
1982           // TODO(nick): It's not clear to me why this is necessary;
1983           // it appears to be papering over bugs in the main analyzer.
1984           if (thisTypeForProperties != null && n.getParent().isExprResult()) {
1985             if (n.isAssign()) {
1986               maybeCollectMember(n.getFirstChild(), n, n.getLastChild());
1987             } else if (n.isGetProp()) {
1988               maybeCollectMember(n, n, null);
1989             }
1990           }
1991     
1992           super.visit(t, n, parent);
1993         }
1994     
1995         private ObjectType getThisTypeForCollectingProperties() {
1996           Node rootNode = scope.getRootNode();
1997           if (rootNode.isFromExterns()) return null;
1998     
1999           JSType type = rootNode.getJSType();
2000           if (type == null || !type.isFunctionType()) return null;
2001     
2002           FunctionType fnType = type.toMaybeFunctionType();
2003           JSType fnThisType = fnType.getTypeOfThis();
2004           return fnThisType.isUnknownType() ? null : fnThisType.toObjectType();
2005         }
2006     
2007         private void maybeCollectMember(Node member,
2008             Node nodeWithJsDocInfo, @Nullable Node value) {
2009           JSDocInfo info = nodeWithJsDocInfo.getJSDocInfo();
2010     
2011           // Do nothing if there is no JSDoc type info, or
2012           // if the node is not a member expression, or
2013           // if the member expression is not of the form: this.someProperty.
2014           if (info == null ||
2015               !member.isGetProp() ||
2016               !member.getFirstChild().isThis()) {
2017             return;
2018           }
2019     
2020           JSType jsType = getDeclaredType(info, member, value);
2021           Node name = member.getLastChild();
2022           if (jsType != null) {
2023             thisTypeForProperties.defineDeclaredProperty(
2024                 name.getString(),
2025                 jsType,
2026                 member);
2027           }
2028         }

2030         /** Handle bleeding functions and function parameters. */
2031         private void handleFunctionInputs(Node fnNode) {
2032           // Handle bleeding functions.
2033           Node fnNameNode = fnNode.getFirstChild();
2034           String fnName = fnNameNode.getString();
2035           if (!fnName.isEmpty()) {
2036             TypedVar fnVar = scope.getVar(fnName);
2037             if (fnVar == null ||
2038                 // Make sure we're not touching a native function. Native
2039                 // functions aren't bleeding, but may not have a declaration
2040                 // node.
2041                 (fnVar.getNameNode() != null &&
2042                     // Make sure that the function is actually bleeding by checking
2043                     // if has already been declared.
2044                     fnVar.getInitialValue() != fnNode)) {
2045               defineSlot(fnNameNode, fnNode, fnNode.getJSType(), false);
2046             }
2047           }
2048     
2049           declareArguments(fnNode);
2050         }
2051     
2052         /**
2053          * Declares all of a function's arguments.
2054          */
2055         private void declareArguments(Node functionNode) {
2056           Node astParameters = functionNode.getSecondChild();
2057           Node iifeArgumentNode = null;
2058     
2059           if (NodeUtil.isCallOrNewTarget(functionNode)) {
2060             iifeArgumentNode = functionNode.getNext();
2061           }
2062     
2063           FunctionType functionType =
2064               JSType.toMaybeFunctionType(functionNode.getJSType());
2065           if (functionType != null) {
2066             Node jsDocParameters = functionType.getParametersNode();
2067             if (jsDocParameters != null) {
2068               Node jsDocParameter = jsDocParameters.getFirstChild();
2069               for (Node astParameter : astParameters.children()) {
2070                 JSType paramType = jsDocParameter == null ?
2071                     unknownType : jsDocParameter.getJSType();
2072                 boolean inferred = paramType == null || paramType == unknownType;
2073     
2074                 if (iifeArgumentNode != null && inferred) {
2075                   String argumentName = iifeArgumentNode.getQualifiedName();
2076                   TypedVar argumentVar =
2077                       argumentName == null || scope.getParent() == null
2078                       ? null : scope.getParent().getVar(argumentName);
2079                   if (argumentVar != null && !argumentVar.isTypeInferred()) {
2080                     paramType = argumentVar.getType();
2081                   }
2082                 }
2083     
2084                 if (paramType == null) {
2085                   paramType = unknownType;
2086                 }
2087     
2088                 defineSlot(astParameter, functionNode, paramType, inferred);
2089     
2090                 if (jsDocParameter != null) {
2091                   jsDocParameter = jsDocParameter.getNext();
2092                 }
2093                 if (iifeArgumentNode != null) {
2094                   iifeArgumentNode = iifeArgumentNode.getNext();
2095                 }
2096               }
2097             }
2098           }
2099         } // end declareArguments

2103        * Does a first-order function analysis that just looks at simple things
2104        * like what variables are escaped, and whether 'this' is used.
2105        */
2106       private static class FirstOrderFunctionAnalyzer
2107           extends AbstractScopedCallback implements CompilerPass {
2108         private final AbstractCompiler compiler;
2109         private final Map<Node, AstFunctionContents> data;
2110     
2111         FirstOrderFunctionAnalyzer(
2112             AbstractCompiler compiler, Map<Node, AstFunctionContents> outParam) {
2113           this.compiler = compiler;
2114           this.data = outParam;
2115         }
2116     
2117         @Override public void process(Node externs, Node root) {
2118           if (externs == null) {
2119             NodeTraversal.traverseTyped(compiler, root, this);
2120           } else {
2121             NodeTraversal.traverseRootsTyped(compiler, this, externs, root);
2122           }
2123         }

2125         @Override public void enterScope(NodeTraversal t) {
2126           if (!t.inGlobalScope()) {
2127             Node n = t.getScopeRoot();
2128             data.put(n, new AstFunctionContents(n));
2129           }
2130         }
2131     
2132         @Override public void visit(NodeTraversal t, Node n, Node parent) {
2133           if (t.inGlobalScope()) {
2134             // The first-order function analyzer looks at two types of variables:
2135             //
2136             // 1) Local variables that are assigned in inner scopes ("escaped vars")
2137             //
2138             // 2) Local variables that are assigned more than once.
2139             //
2140             // We treat all global variables as escaped by default, so there's
2141             // no reason to do this extra computation for them.
2142             return;
2143           }
2144     
2145           if (n.isReturn() && n.getFirstChild() != null) {
2146             data.get(t.getScopeRoot()).recordNonEmptyReturn();
2147           }
2148     
2149           if (n.isName() && NodeUtil.isLValue(n) &&
2150               // Be careful of bleeding functions, which create variables
2151               // in the inner scope, not the scope where the name appears.
2152               !NodeUtil.isBleedingFunctionName(n)) {
2153             String name = n.getString();
2154             TypedScope scope = t.getTypedScope();
2155             TypedVar var = scope.getVar(name);
2156             if (var != null) {
2157               TypedScope ownerScope = var.getScope();
2158               if (ownerScope.isLocal()) {
2159                 data.get(ownerScope.getRootNode()).recordAssignedName(name);
2160               }
2161     
2162               if (scope != ownerScope && ownerScope.isLocal()) {
2163                 data.get(ownerScope.getRootNode()).recordEscapedVarName(name);
2164               }
2165             }
2166           } else if (n.isGetProp() && n.isUnscopedQualifiedName() &&
2167               NodeUtil.isLValue(n)) {
2168             String name = NodeUtil.getRootOfQualifiedName(n).getString();
2169             TypedScope scope = t.getTypedScope();
2170             TypedVar var = scope.getVar(name);
2171             if (var != null) {
2172               TypedScope ownerScope = var.getScope();
2173               if (scope != ownerScope && ownerScope.isLocal()) {
2174                 data.get(ownerScope.getRootNode())
2175                     .recordEscapedQualifiedName(n.getQualifiedName());
2176               }
2177             }
2178           }
2179         }
2180       }
