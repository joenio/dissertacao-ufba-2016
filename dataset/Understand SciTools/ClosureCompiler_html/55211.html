59       public void testBasicMismatch() throws Exception {
60         testSame("/** @param {number} x */ function f(x) {} f('a');",
61             TYPE_MISMATCH_WARNING);
62         assertMismatches(ImmutableList.of(fromNatives(STRING_TYPE, NUMBER_TYPE)));
63       }

65       public void testFunctionMismatch() throws Exception {
66         testSame(
67             "/** \n" +
68             " * @param {function(string): number} x \n" +
69             " * @return {function(boolean): string} \n" +
70             " */ function f(x) { return x; }",
71             TYPE_MISMATCH_WARNING);
72     
73         JSTypeRegistry registry = compiler.getTypeRegistry();
74         JSType string = registry.getNativeType(STRING_TYPE);
75         JSType bool = registry.getNativeType(BOOLEAN_TYPE);
76         JSType number = registry.getNativeType(NUMBER_TYPE);
77         JSType firstFunction = registry.createFunctionType(number, string);
78         JSType secondFunction = registry.createFunctionType(string, bool);
79     
80         assertMismatches(
81             ImmutableList.of(
82                 new TypeMismatch(firstFunction, secondFunction, null),
83                 fromNatives(STRING_TYPE, BOOLEAN_TYPE),
84                 fromNatives(NUMBER_TYPE, STRING_TYPE)));
85       }

108       public void testFunctionMismatchMediumLengthTypes() throws Exception {
109         testSame("",
110             LINE_JOINER.join(
111                 "/**",
112                 " * @param {{a: string, b: string, c: string, d: string, e: string}} x",
113                 " */",
114                 "function f(x) {}",
115                 "var y = {a:'',b:'',c:'',d:'',e:0};",
116                 "f(y);"),
117             TYPE_MISMATCH_WARNING,
118             LINE_JOINER.join(
119                 "actual parameter 1 of f does not match formal parameter",
120                 "found   : {",
121                 "  a: string,",
122                 "  b: string,",
123                 "  c: string,",
124                 "  d: string,",
125                 "  e: (number|string)",
126                 "}",
127                 "required: {",
128                 "  a: string,",
129                 "  b: string,",
130                 "  c: string,",
131                 "  d: string,",
132                 "  e: string",
133                 "}"));
134       }

136       /**
137        * Make sure the 'found' and 'required' strings are not identical when there is a mismatch.
138        * See https://code.google.com/p/closure-compiler/issues/detail?id=719.
139        */
140       public void testFunctionMismatchLongTypes() throws Exception {
141         testSame("",
142             LINE_JOINER.join(
143                 "/**",
144                 " * @param {{a: string, b: string, c: string, d: string, e: string,",
145                 " *          f: string, g: string, h: string, i: string, j: string, k: string}} x",
146                 " */",
147                 "function f(x) {}",
148                 "var y = {a:'',b:'',c:'',d:'',e:'',f:'',g:'',h:'',i:'',j:'',k:0};",
149                 "f(y);"),
150             TYPE_MISMATCH_WARNING,
151             LINE_JOINER.join(
152                 "actual parameter 1 of f does not match formal parameter",
153                 "found   : {a: string, b: string, c: string, d: string, e: string, f: string,"
154                   + " g: string, h: string, i: string, j: string, k: (number|string)}",
155                 "required: {a: string, b: string, c: string, d: string, e: string, f: string,"
156                   + " g: string, h: string, i: string, j: string, k: string}"));
157       }

185       public void testNullUndefined() {
186         testSame("/** @param {string} x */ function f(x) {}\n" +
187                  "f(/** @type {string|null|undefined} */ ('a'));",
188                  TYPE_MISMATCH_WARNING);
189         assertMismatches(Collections.<TypeMismatch>emptyList());
190       }
191     
192       public void testSubclass() {
193         testSame("/** @constructor */\n"  +
194                  "function Super() {}\n" +
195                  "/**\n" +
196                  " * @constructor\n" +
197                  " * @extends {Super}\n" +
198                  " */\n" +
199                  "function Sub() {}\n" +
200                  "/** @param {Sub} x */ function f(x) {}\n" +
201                  "f(/** @type {Super} */ (new Sub));",
202                  TYPE_MISMATCH_WARNING);
203         assertMismatches(Collections.<TypeMismatch>emptyList());
204       }

206       private TypeMismatch fromNatives(JSTypeNative a, JSTypeNative b) {
207         JSTypeRegistry registry = compiler.getTypeRegistry();
208         return new TypeMismatch(
209             registry.getNativeType(a), registry.getNativeType(b), null);
210       }
211     
212       private void assertMismatches(List<TypeMismatch> expected) {
213         List<TypeMismatch> actual = ImmutableList.copyOf(compiler.getTypeMismatches());
214         assertEquals(expected, actual);
215       }
