27     /**
28      * Checks that goog.module() is used correctly.
29      *
30      * Note that this file only does checks that can be done per-file. Whole program
31      * checks happen during goog.module rewriting, in {@link ClosureRewriteModule}.
32      */
33     public final class ClosureCheckModule implements Callback, HotSwapCompilerPass {
34       static final DiagnosticType AT_EXPORT_IN_GOOG_MODULE =
35           DiagnosticType.error(
36               "JSC_AT_EXPORT_IN_GOOG_MODULE",
37               "@export has no effect here");
38     
39       static final DiagnosticType AT_EXPORT_IN_NON_LEGACY_GOOG_MODULE =
40           DiagnosticType.disabled(
41               "JSC_AT_EXPORT_IN_GOOG_MODULE",
42               "@export is not allowed here in a non-legacy goog.module."
43               + " Consider using goog.exportSymbol instead.");
44     
45       static final DiagnosticType GOOG_MODULE_REFERENCES_THIS = DiagnosticType.error(
46           "JSC_GOOG_MODULE_REFERENCES_THIS",
47           "The body of a goog.module cannot reference 'this'.");
48     
49       static final DiagnosticType GOOG_MODULE_USES_THROW = DiagnosticType.error(
50           "JSC_GOOG_MODULE_USES_THROW",
51           "The body of a goog.module cannot use 'throw'.");
52     
53       static final DiagnosticType LET_GOOG_REQUIRE =
54           DiagnosticType.disabled(
55               "JSC_LET_GOOG_REQUIRE",
56               "Module imports must be constant. Please use 'const' instead of 'let'.");
57     
58       static final DiagnosticType MULTIPLE_MODULES_IN_FILE =
59           DiagnosticType.error(
60               "JSC_MULTIPLE_MODULES_IN_FILE",
61               "There should only be a single goog.module() statement per file.");
62     
63       static final DiagnosticType MODULE_AND_PROVIDES =
64           DiagnosticType.error(
65               "JSC_MODULE_AND_PROVIDES",
66               "A file using goog.module() may not also use goog.provide() statements.");
67     
68       static final DiagnosticType ONE_REQUIRE_PER_DECLARATION =
69           DiagnosticType.error(
70               "JSC_ONE_REQUIRE_PER_DECLARATION",
71               "There may only be one goog.require() per var/let/const declaration.");
72     
73       static final DiagnosticType EXPORT_NOT_A_MODULE_LEVEL_STATEMENT =
74           DiagnosticType.error(
75               "JSC_EXPORT_NOT_A_MODULE_LEVEL_STATEMENT",
76               "Exports must be a statement at the top-level of a module");
77     
78       static final DiagnosticType EXPORT_REPEATED_ERROR =
79           DiagnosticType.error(
80               "JSC_EXPORT_REPEATED_ERROR",
81               "Name cannot be exported multiple times. Previous export on line {0}.");
82     
83       static final DiagnosticType REFERENCE_TO_MODULE_GLOBAL_NAME =
84           DiagnosticType.error(
85               "JSC_REFERENCE_TO_MODULE_GLOBAL_NAME",
86               "References to the global name of a module are not allowed. Perhaps you meant exports?");
87     
88       static final DiagnosticType REFERENCE_TO_SHORT_IMPORT_BY_LONG_NAME =
89           DiagnosticType.disabled(
90               "JSC_REFERENCE_TO_SHORT_IMPORT_BY_LONG_NAME",
91               "Reference to fully qualified import name ''{0}''. Please use the short name instead.");
92     
93       static final DiagnosticType REFERENCE_TO_SHORT_IMPORT_BY_LONG_NAME_INCLUDING_SHORT_NAME =
94           DiagnosticType.disabled(
95               "JSC_REFERENCE_TO_SHORT_IMPORT_BY_LONG_NAME_INCLUDING_SHORT_NAME",
96               "Reference to fully qualified import name ''{0}''."
97                   + " Please use the short name ''{1}'' instead.");
98     
99       static final DiagnosticType REQUIRE_NOT_AT_TOP_LEVEL =
100           DiagnosticType.error(
101               "JSC_REQUIRE_NOT_AT_TOP_LEVEL",
102               "goog.require() must be called at file scope.");
103     
104       private final AbstractCompiler compiler;
105     
106       private String currentModuleName = null;
107       private Map<String, String> shortRequiredNamespaces = new HashMap<>();
108       private Node defaultExportNode = null;
109     
110       public ClosureCheckModule(AbstractCompiler compiler) {
111         this.compiler = compiler;
112       }
113     
114       @Override
115       public void process(Node externs, Node root) {
116         NodeTraversal.traverseEs6(compiler, root, this);
117       }
118     
119       @Override
120       public void hotSwapScript(Node scriptRoot, Node originalRoot) {
121         NodeTraversal.traverseEs6(compiler, scriptRoot, this);
122       }

124       @Override
125       public boolean shouldTraverse(NodeTraversal t, Node n, Node parent) {
126         if (n.isScript()) {
127           if (NodeUtil.isGoogModuleFile(n)) {
128             n.putBooleanProp(Node.GOOG_MODULE, true);
129             return true;
130           }
131           return false;
132         }
133         return true;
134       }
135     
136       @Override
137       public void visit(NodeTraversal t, Node n, Node parent) {
138         switch (n.getType()) {
139           case CALL:
140             Node callee = n.getFirstChild();
141             if (callee.matchesQualifiedName("goog.module")) {
142               if (currentModuleName == null) {
143                 currentModuleName = extractFirstArgumentName(n);
144               } else {
145                 t.report(n, MULTIPLE_MODULES_IN_FILE);
146               }
147             } else if (callee.matchesQualifiedName("goog.provide")) {
148               t.report(n, MODULE_AND_PROVIDES);
149             } else if (callee.matchesQualifiedName("goog.require")) {
150               checkRequireCall(t, n, parent);
151             }
152             break;
153           case ASSIGN: {
154             Node lhs = n.getFirstChild();
155             if (lhs.isQualifiedName()
156                 && NodeUtil.getRootOfQualifiedName(lhs).matchesQualifiedName("exports")) {
157               checkModuleExport(t, n, parent);
158             }
159             break;
160           }
161           case CLASS:
162           case FUNCTION:
163             if (!NodeUtil.isStatement(n)) {
164               break;
165             }
166             // fallthrough
167           case VAR:
168           case LET:
169           case CONST:
170             if (t.inGlobalHoistScope() && NodeUtil.getEnclosingClass(n) == null
171                 && NodeUtil.getEnclosingType(n, Token.OBJECTLIT) == null) {
172               JSDocInfo jsdoc = NodeUtil.getBestJSDocInfo(n);
173               if (jsdoc != null && jsdoc.isExport()) {
174                 t.report(n, AT_EXPORT_IN_GOOG_MODULE);
175               }
176             }
177             break;
178           case THIS:
179             if (t.inGlobalHoistScope()) {
180               t.report(n, GOOG_MODULE_REFERENCES_THIS);
181             }
182             break;
183           case THROW:
184             if (t.inGlobalHoistScope()) {
185               t.report(n, GOOG_MODULE_USES_THROW);
186             }
187             break;
188           case GETPROP:
189             if (currentModuleName != null && n.matchesQualifiedName(currentModuleName)) {
190               t.report(n, REFERENCE_TO_MODULE_GLOBAL_NAME);
191             } else if (shortRequiredNamespaces.containsKey(n.getQualifiedName())) {
192               String shortName = shortRequiredNamespaces.get(n.getQualifiedName());
193               if (shortName == null) {
194                 t.report(n, REFERENCE_TO_SHORT_IMPORT_BY_LONG_NAME, n.getQualifiedName());
195               } else {
196                 t.report(n, REFERENCE_TO_SHORT_IMPORT_BY_LONG_NAME_INCLUDING_SHORT_NAME,
197                     n.getQualifiedName(), shortName);
198               }
199             }
200             break;
201           case SCRIPT:
202             currentModuleName = null;
203             shortRequiredNamespaces.clear();
204             defaultExportNode = null;
205             break;
206         }
207       }

209       private void checkModuleExport(NodeTraversal t, Node n, Node parent) {
210         Preconditions.checkArgument(n.isAssign());
211         Node lhs = n.getFirstChild();
212         Preconditions.checkState(lhs.isQualifiedName());
213         Preconditions.checkState(NodeUtil.getRootOfQualifiedName(lhs).matchesQualifiedName("exports"));
214         if (lhs.isName()) {
215           if  (defaultExportNode != null) {
216             // Multiple exports
217             t.report(n, EXPORT_REPEATED_ERROR, String.valueOf(defaultExportNode.getLineno()));
218           } else if (!t.inGlobalScope() || !parent.isExprResult()) {
219             // Invalid export location.
220             t.report(n, EXPORT_NOT_A_MODULE_LEVEL_STATEMENT);
221           }
222           defaultExportNode = lhs;
223         }
224         if ((lhs.isName() || !NodeUtil.isPrototypeProperty(lhs))
225             && !NodeUtil.isLegacyGoogModuleFile(NodeUtil.getEnclosingScript(n))) {
226           JSDocInfo jsDoc = n.getJSDocInfo();
227           if (jsDoc != null && jsDoc.isExport()) {
228             t.report(n, AT_EXPORT_IN_NON_LEGACY_GOOG_MODULE);
229           }
230         }
231       }
232     
233       private String extractFirstArgumentName(Node callNode) {
234         Node firstArg = callNode.getSecondChild();
235         if (firstArg != null && firstArg.isString()) {
236           return firstArg.getString();
237         }
238         return null;
239       }
240     
241       private void checkRequireCall(NodeTraversal t, Node callNode, Node parent) {
242         Preconditions.checkState(callNode.isCall());
243         switch (parent.getType()) {
244           case EXPR_RESULT:
245             return;
246           case NAME:
247           case DESTRUCTURING_LHS:
248             checkShortGoogRequireCall(t, callNode, parent.getParent());
249             return;
250         }
251         t.report(callNode, REQUIRE_NOT_AT_TOP_LEVEL);
252       }
253     
254       private void checkShortGoogRequireCall(NodeTraversal t, Node callNode, Node declaration) {
255         if (declaration.isLet()) {
256           t.report(declaration, LET_GOOG_REQUIRE);
257         }
258         if (declaration.getChildCount() != 1) {
259           t.report(declaration, ONE_REQUIRE_PER_DECLARATION);
260         }
261         Node lhs = declaration.getFirstChild();
262         String shortName = lhs.isName() ? lhs.getString() : null;
263         shortRequiredNamespaces.put(extractFirstArgumentName(callNode), shortName);
264       }
