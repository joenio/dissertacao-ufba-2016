53       public DiagnosticGroups() {}
54     
55       private static final Map<String, DiagnosticGroup> groupsByName =
56            new HashMap<>();
57     
58       static DiagnosticGroup registerDeprecatedGroup(String name) {
59         return registerGroup(name, new DiagnosticGroup(name, UNUSED));
60       }
61     
62       static DiagnosticGroup registerGroup(String name,
63           DiagnosticGroup group) {
64         groupsByName.put(name, group);
65         return group;
66       }
67     
68       static DiagnosticGroup registerGroup(String name,
69           DiagnosticType ... types) {
70         DiagnosticGroup group = new DiagnosticGroup(name, types);
71         groupsByName.put(name, group);
72         return group;
73       }
74     
75       static DiagnosticGroup registerGroup(String name,
76           DiagnosticGroup ... groups) {
77         DiagnosticGroup group = new DiagnosticGroup(name, groups);
78         groupsByName.put(name, group);
79         return group;
80       }
81     
82       /** Get the registered diagnostic groups, indexed by name. */
83       public Map<String, DiagnosticGroup> getRegisteredGroups() {
84         return ImmutableMap.copyOf(groupsByName);
85       }
86     
87       /** Find the diagnostic group registered under the given name. */
88       public DiagnosticGroup forName(String name) {
89         return groupsByName.get(name);
90       }
