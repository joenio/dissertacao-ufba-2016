31     /**
32      * RenameLabels renames all the labels so that they have short names, to reduce
33      * code size and also to obfuscate the code.
34      *
35      * Label names have a unique namespace, so variable or function names clashes
36      * are not a concern, but keywords clashes are.
37      *
38      * Additionally, labels names are only within the statements include in the
39      * label and do not cross function boundaries. This means that it is possible to
40      * create one label name that is used for labels at any given depth of label
41      * nesting. Typically, the name "a" will be used for all top-level labels, "b"
42      * for the next nested label, and so on. For example:
43      *
44      * <code>
45      * function bar() {
46      *   a: {
47      *     b: {
48      *       foo();
49      *     }
50      *   }
51      *
52      *   a: {
53      *     b: break a;
54      *   }
55      * }
56      * </code>
57      *
58      * The general processes is as follows: process() is the entry point for the
59      * CompilerPass, and from there a standard "ScopedCallback" traversal is done,
60      * where "shouldTraverse" is called when descending the tree, and the "visit" is
61      * called in a depth first manner. The name for the label is selected during the
62      * decent in "shouldTraverse", and the references to the label name are renamed
63      * as they are encountered during the "visit". This means that if the label is
64      * unreferenced, it is known when the label node is visited, and, if so, can be
65      * safely removed.
66      *
67      * @author johnlenz@google.com (John Lenz)
68      */
69     final class RenameLabels implements CompilerPass {
70       private final AbstractCompiler compiler;
71       private final Supplier<String> nameSupplier;
72       private final boolean removeUnused;
73     
74       RenameLabels(final AbstractCompiler compiler) {
75         this(compiler, new DefaultNameSupplier(), true);
76       }
77     
78       RenameLabels(
79           AbstractCompiler compiler,
80           Supplier<String> supplier,
81           boolean removeUnused) {
82         this.compiler = compiler;
83         this.nameSupplier = supplier;
84         this.removeUnused = removeUnused;
85       }
86     
87       static class DefaultNameSupplier implements Supplier<String> {
88         // DefaultNameGenerator is used to create safe label names.
89         private final NameGenerator nameGenerator;
90     
91         private DefaultNameSupplier(final NameGenerator nameGen) {
92           this.nameGenerator = nameGen;
93         }
94     
95         private DefaultNameSupplier() {
96           this.nameGenerator = new DefaultNameGenerator(
97               new HashSet<String>(), "", null);
98         }
99     
100         @Override
101         public String get() {
102           return nameGenerator.generateNextName();
103         }
104       }
105     
106       /**
107        * Iterate through the nodes, renaming all the labels.
108        */
109       class ProcessLabels implements ScopedCallback {
110     
111         ProcessLabels() {
112           // Create a entry for global scope.
113           namespaceStack.push(new LabelNamespace());
114         }
115     
116         // A stack of labels namespaces. Labels in an outer scope aren't part of an
117         // inner scope, so a new namespace is created each time a scope is entered.
118         final Deque<LabelNamespace> namespaceStack = new ArrayDeque<>();
119     
120         // The list of generated names. Typically, the first name will be "a",
121         // the second "b", etc.
122         final ArrayList<String> names = new ArrayList<>();
123     
124     
125         @Override
126         public void enterScope(NodeTraversal nodeTraversal) {
127           // Start a new namespace for label names.
128           if (nodeTraversal.getScopeRoot().isFunction()) {
129             namespaceStack.push(new LabelNamespace());
130           }
131         }
132     
133         @Override
134         public void exitScope(NodeTraversal nodeTraversal) {
135           if (nodeTraversal.getScopeRoot().isFunction()) {
136             namespaceStack.pop();
137           }
138         }
139     
140         /**
141          * shouldTraverse is call when descending into the Node tree, so it is used
142          * here to build the context for label renames.
143          *
144          * {@inheritDoc}
145          */
146         @Override
147         public boolean shouldTraverse(NodeTraversal nodeTraversal, Node node,
148             Node parent) {
149           if (node.isLabel()) {
150             // Determine the new name for this label.
151             LabelNamespace current = namespaceStack.peek();
152             int currentDepth = current.renameMap.size() + 1;
153             String name = node.getFirstChild().getString();
154     
155             // Store the context for this label name.
156             LabelInfo li = new LabelInfo(currentDepth);
157             Preconditions.checkState(!current.renameMap.containsKey(name));
158             current.renameMap.put(name, li);
159     
160             // Create a new name, if needed, for this depth.
161             if (names.size() < currentDepth) {
162               names.add(nameSupplier.get());
163             }
164     
165             String newName = getNameForId(currentDepth);
166             compiler.addToDebugLog("label renamed: " + name + " => " + newName);
167           }
168     
169           return true;
170         }
171     
172         /**
173          * Delegate the actual processing of the node to visitLabel and
174          * visitBreakOrContinue.
175          *
176          * {@inheritDoc}
177          */
178         @Override
179         public void visit(NodeTraversal nodeTraversal, Node node, Node parent) {
180           switch (node.getType()) {
181             case LABEL:
182               visitLabel(node, parent);
183               break;
184     
185             case BREAK:
186             case CONTINUE:
187               visitBreakOrContinue(node);
188               break;
189           }
190         }
191     
192         /**
193          * Rename label references in breaks and continues.
194          * @param node The break or continue node.
195          */
196         private void visitBreakOrContinue(Node node) {
197           Node nameNode = node.getFirstChild();
198           if (nameNode != null) {
199             // This is a named break or continue;
200             String name = nameNode.getString();
201             Preconditions.checkState(!name.isEmpty());
202             LabelInfo li = getLabelInfo(name);
203             if (li != null) {
204               String newName = getNameForId(li.id);
205               // Mark the label as referenced so it isn't removed.
206               li.referenced = true;
207               if (!name.equals(newName)) {
208                 // Give it the short name.
209                 nameNode.setString(newName);
210                 compiler.reportCodeChange();
211               }
212             }
213           }
214         }
215     
216         /**
217          * Rename or remove labels.
218          * @param node  The label node.
219          * @param parent The parent of the label node.
220          */
221         private void visitLabel(Node node, Node parent) {
222           Node nameNode = node.getFirstChild();
223           Preconditions.checkState(nameNode != null);
224           String name = nameNode.getString();
225           LabelInfo li = getLabelInfo(name);
226           // This is a label...
227           if (li.referenced || !removeUnused) {
228             String newName = getNameForId(li.id);
229             if (!name.equals(newName)) {
230               // ... and it is used, give it the short name.
231               nameNode.setString(newName);
232               compiler.reportCodeChange();
233             }
234           } else {
235             // ... and it is not referenced, just remove it.
236             Node newChild = node.getLastChild();
237             node.removeChild(newChild);
238             parent.replaceChild(node, newChild);
239             if (newChild.isBlock()) {
240               NodeUtil.tryMergeBlock(newChild);
241             }
242             compiler.reportCodeChange();
243           }
244     
245           // Remove the label from the current stack of labels.
246           namespaceStack.peek().renameMap.remove(name);
247         }
248     
249         /**
250          * @param id The id, which is the depth of the label in the current context,
251          *        for which to get a short name.
252          * @return The short name of the identified label.
253          */
254         String getNameForId(int id) {
255           return names.get(id - 1);
256         }
257     
258         /**
259          * @param name The name to retrieve information about.
260          * @return The structure representing the name in the current context.
261          */
262         LabelInfo getLabelInfo(String name) {
263           return namespaceStack.peek().renameMap.get(name);
264         }
265       }
266     
267       @Override
268       public void process(Node externs, Node root) {
269         // Do variable reference counting.
270         NodeTraversal.traverseEs6(compiler, root, new ProcessLabels());
271       }
272     
273     
274       private static class LabelInfo {
275         boolean referenced = false;
276         final int id;
277     
278         LabelInfo(int id) {
279           this.id = id;
280         }
281       }
282     
283     
284       private static class LabelNamespace {
285         final Map<String, LabelInfo> renameMap = new HashMap<>();
286       }
