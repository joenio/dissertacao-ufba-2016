37     /**
38      * Converts ES6 code to valid ES5 code. This class does most of the transpilation, and
39      * https://github.com/google/closure-compiler/wiki/ECMAScript6 lists which ES6 features are
40      * supported. Other classes that start with "Es6" do other parts of the transpilation.
41      *
42      * <p>In most cases, the output is valid as ES3 (hence the class name) but in some cases, if
43      * the output language is set to ES5, we rely on ES5 features such as getters, setters,
44      * and Object.defineProperties.
45      *
46      * @author tbreisacher@google.com (Tyler Breisacher)
47      */
48     public final class Es6ToEs3Converter implements NodeTraversal.Callback, HotSwapCompilerPass {

109       /**
110        * Some nodes must be visited pre-order in order to rewrite the
111        * references to {@code this} correctly.
112        * Everything else is translated post-order in {@link #visit}.
113        */
114       @Override
115       public boolean shouldTraverse(NodeTraversal t, Node n, Node parent) {
116         switch (n.getType()) {
117           case REST:
118             visitRestParam(n, parent);
119             break;
120           case GETTER_DEF:
121           case SETTER_DEF:
122             if (compiler.getOptions().getLanguageOut() == LanguageMode.ECMASCRIPT3) {
123               cannotConvert(n, "ES5 getters/setters (consider using --language_out=ES5)");
124               return false;
125             }
126             break;
127           case NEW_TARGET:
128             cannotConvertYet(n, "new.target");
129             break;
130           case FUNCTION:
131             if (n.isAsyncFunction()) {
132               cannotConvertYet(n, "async function");
133             }
134             break;
135           default:
136             break;
137         }
138         return true;
139       }

141       @Override
142       public void visit(NodeTraversal t, Node n, Node parent) {
143         switch (n.getType()) {
144           case NAME:
145             if (!n.isFromExterns() && isGlobalSymbol(t, n)) {
146               initSymbolBefore(n);
147             }
148             break;
149           case GETPROP:
150             if (!n.isFromExterns()) {
151               visitGetprop(t, n);
152             }
153             break;
154           case OBJECTLIT:
155             visitObject(n);
156             break;
157           case MEMBER_FUNCTION_DEF:
158             if (parent.isObjectLit()) {
159               visitMemberFunctionDefInObjectLit(n, parent);
160             }
161             break;
162           case FOR_OF:
163             visitForOf(n, parent);
164             break;
165           case STRING_KEY:
166             visitStringKey(n);
167             break;
168           case CLASS:
169             visitClass(n, parent);
170             break;
171           case ARRAYLIT:
172           case NEW:
173           case CALL:
174             for (Node child : n.children()) {
175               if (child.isSpread()) {
176                 visitArrayLitOrCallWithSpread(n, parent);
177                 break;
178               }
179             }
180             break;
181           case TAGGED_TEMPLATELIT:
182             Es6TemplateLiterals.visitTaggedTemplateLiteral(t, n);
183             break;
184           case TEMPLATELIT:
185             if (!parent.isTaggedTemplateLit()) {
186               Es6TemplateLiterals.visitTemplateLiteral(t, n);
187             }
188             break;
189         }
190       }

192       /**
193        * @return Whether {@code n} is a reference to the global "Symbol" function.
194        */
195       private boolean isGlobalSymbol(NodeTraversal t, Node n) {
196         if (!n.matchesQualifiedName("Symbol")) {
197           return false;
198         }
199         Var var = t.getScope().getVar("Symbol");
200         return var == null || var.isGlobal();
201       }

203       /**
204        * Inserts a call to $jscomp.initSymbol() before {@code n}.
205        */
206       private void initSymbolBefore(Node n) {
207         compiler.ensureLibraryInjected("es6_runtime", false);
208         Node statement = NodeUtil.getEnclosingStatement(n);
209         Node initSymbol = IR.exprResult(IR.call(NodeUtil.newQName(compiler, "$jscomp.initSymbol")));
210         statement.getParent().addChildBefore(initSymbol.useSourceInfoFromForTree(statement), statement);
211         compiler.reportCodeChange();
212       }

214       // TODO(tbreisacher): Do this for all well-known symbols.
215       private void visitGetprop(NodeTraversal t, Node n) {
216         if (!n.matchesQualifiedName("Symbol.iterator")) {
217           return;
218         }
219         if (isGlobalSymbol(t, n.getFirstChild())) {
220           compiler.ensureLibraryInjected("es6_runtime", false);
221           Node statement = NodeUtil.getEnclosingStatement(n);
222           Node init = IR.exprResult(IR.call(NodeUtil.newQName(compiler, "$jscomp.initSymbolIterator")));
223           statement.getParent().addChildBefore(init.useSourceInfoFromForTree(statement), statement);
224           compiler.reportCodeChange();
225         }
226       }

228       /**
229        * Converts a member definition in an object literal to an ES3 key/value pair.
230        * Member definitions in classes are handled in {@link #visitClass}.
231        */
232       private void visitMemberFunctionDefInObjectLit(Node n, Node parent) {
233         String name = n.getString();
234         Node stringKey = IR.stringKey(name, n.getFirstChild().detachFromParent());
235         stringKey.setJSDocInfo(n.getJSDocInfo());
236         parent.replaceChild(n, stringKey);
237         compiler.reportCodeChange();
238       }

252       private void visitForOf(Node node, Node parent) {
253         Node variable = node.removeFirstChild();
254         Node iterable = node.removeFirstChild();
255         Node body = node.removeFirstChild();
256     
257         Node iterName = IR.name(ITER_BASE + compiler.getUniqueNameIdSupplier().get());
258         iterName.makeNonIndexable();
259         Node getNext = IR.call(IR.getprop(iterName.cloneTree(), IR.string("next")));
260         String variableName;
261         Token declType;
262         if (variable.isName()) {
263           declType = Token.NAME;
264           variableName = variable.getQualifiedName();
265         } else {
266           Preconditions.checkState(NodeUtil.isNameDeclaration(variable),
267               "Expected var, let, or const. Got %s", variable);
268           declType = variable.getType();
269           variableName = variable.getFirstChild().getQualifiedName();
270         }
271         Node iterResult = IR.name(ITER_RESULT + variableName);
272         iterResult.makeNonIndexable();
273     
274         Node init = IR.var(iterName.cloneTree(), makeIterator(compiler, iterable));
275         Node initIterResult = iterResult.cloneTree();
276         initIterResult.addChildToFront(getNext.cloneTree());
277         init.addChildToBack(initIterResult);
278     
279         Node cond = IR.not(IR.getprop(iterResult.cloneTree(), IR.string("done")));
280         Node incr = IR.assign(iterResult.cloneTree(), getNext.cloneTree());
281     
282         Node declarationOrAssign;
283         if (declType == Token.NAME) {
284           declarationOrAssign = IR.exprResult(IR.assign(
285               IR.name(variableName).useSourceInfoFrom(variable),
286               IR.getprop(iterResult.cloneTree(), IR.string("value"))));
287         } else {
288           declarationOrAssign = new Node(
289               declType,
290               IR.name(variableName).useSourceInfoFrom(variable.getFirstChild()));
291           declarationOrAssign.getFirstChild().addChildToBack(
292               IR.getprop(iterResult.cloneTree(), IR.string("value")));
293         }
294         body.addChildToFront(declarationOrAssign);
295     
296         Node newFor = IR.forNode(init, cond, incr, body);
297         newFor.useSourceInfoIfMissingFromForTree(node);
298         parent.replaceChild(node, newFor);
299         compiler.reportCodeChange();
300       }

302       private void checkClassReassignment(Node clazz) {
303         Node name = NodeUtil.getNameNode(clazz);
304         Node enclosingFunction = NodeUtil.getEnclosingFunction(clazz);
305         if (enclosingFunction == null) {
306           return;
307         }
308         CheckClassAssignments checkAssigns = new CheckClassAssignments(name);
309         NodeTraversal.traverseEs6(compiler, enclosingFunction, checkAssigns);
310       }

312       /**
313        * Processes a rest parameter
314        */
315       private void visitRestParam(Node restParam, Node paramList) {
316         Node functionBody = paramList.getLastSibling();
317         int restIndex = paramList.getIndexOfChild(restParam);
318         String paramName = restParam.getFirstChild().getString();
319     
320         Node nameNode = IR.name(paramName);
321         nameNode.setVarArgs(true);
322         nameNode.setJSDocInfo(restParam.getJSDocInfo());
323         paramList.replaceChild(restParam, nameNode);
324     
325         // Make sure rest parameters are typechecked
326         JSTypeExpression type = null;
327         JSDocInfo info = restParam.getJSDocInfo();
328         if (info != null) {
329           type = info.getType();
330         } else {
331           JSDocInfo functionInfo = NodeUtil.getBestJSDocInfo(paramList.getParent());
332           if (functionInfo != null) {
333             type = functionInfo.getParameterType(paramName);
334           }
335         }
336         if (type != null && type.getRoot().getType() != Token.ELLIPSIS) {
337           compiler.report(JSError.make(restParam, BAD_REST_PARAMETER_ANNOTATION));
338         }
339     
340         if (!functionBody.hasChildren()) {
341           // If function has no body, we are done!
342           compiler.reportCodeChange();
343           return;
344         }
345     
346         Node newBlock = IR.block().useSourceInfoFrom(functionBody);
347         Node name = IR.name(paramName);
348         Node let = IR.let(name, IR.name(REST_PARAMS))
349             .useSourceInfoIfMissingFromForTree(functionBody);
350         newBlock.addChildToFront(let);
351     
352         for (Node child : functionBody.children()) {
353           newBlock.addChildToBack(child.detachFromParent());
354         }
355     
356         if (type != null) {
357           Node arrayType = IR.string("Array");
358           Node typeNode = type.getRoot();
359           Node memberType =
360               typeNode.getType() == Token.ELLIPSIS
361                   ? typeNode.getFirstChild().cloneTree()
362                   : typeNode.cloneTree();
363           arrayType.addChildToFront(
364               new Node(Token.BLOCK, memberType).useSourceInfoIfMissingFrom(typeNode));
365           JSDocInfoBuilder builder = new JSDocInfoBuilder(false);
366           builder.recordType(
367               new JSTypeExpression(new Node(Token.BANG, arrayType), restParam.getSourceFileName()));
368           name.setJSDocInfo(builder.build());
369         }
370     
371         Node newArr = IR.var(IR.name(REST_PARAMS), IR.arraylit());
372         functionBody.addChildToFront(newArr.useSourceInfoIfMissingFromForTree(restParam));
373         Node init = IR.var(IR.name(REST_INDEX), IR.number(restIndex));
374         Node cond = IR.lt(IR.name(REST_INDEX), IR.getprop(IR.name("arguments"), IR.string("length")));
375         Node incr = IR.inc(IR.name(REST_INDEX), false);
376         Node body = IR.block(IR.exprResult(IR.assign(
377             IR.getelem(IR.name(REST_PARAMS), IR.sub(IR.name(REST_INDEX), IR.number(restIndex))),
378             IR.getelem(IR.name("arguments"), IR.name(REST_INDEX)))));
379         functionBody.addChildAfter(IR.forNode(init, cond, incr, body)
380             .useSourceInfoIfMissingFromForTree(restParam), newArr);
381         functionBody.addChildToBack(newBlock);
382         compiler.reportCodeChange();
383     
384         // For now, we are running transpilation before type-checking, so we'll
385         // need to make sure changes don't invalidate the JSDoc annotations.
386         // Therefore we keep the parameter list the same length and only initialize
387         // the values if they are set to undefined.
388       }

390       /**
391        * Processes array literals or calls containing spreads. Examples:
392        * [1, 2, ...x, 4, 5] => [].concat([1, 2], $jscomp.arrayFromIterable(x), [4, 5])
393        *
394        * f(...arr) => f.apply(null, [].concat($jscomp.arrayFromIterable(arr)))
395        *
396        * new F(...args) =>
397        *     new Function.prototype.bind.apply(F, [].concat($jscomp.arrayFromIterable(args)))
398        */
399       private void visitArrayLitOrCallWithSpread(Node node, Node parent) {
400         Preconditions.checkArgument(node.isCall() || node.isArrayLit() || node.isNew());
401         List<Node> groups = new ArrayList<>();
402         Node currGroup = null;
403         Node callee = node.isArrayLit() ? null : node.removeFirstChild();
404         Node currElement = node.removeFirstChild();
405         while (currElement != null) {
406           if (currElement.isSpread()) {
407             if (currGroup != null) {
408               groups.add(currGroup);
409               currGroup = null;
410             }
411             groups.add(arrayFromIterable(compiler, currElement.removeFirstChild()));
412           } else {
413             if (currGroup == null) {
414               currGroup = IR.arraylit();
415             }
416             currGroup.addChildToBack(currElement);
417           }
418           currElement = node.removeFirstChild();
419         }
420         if (currGroup != null) {
421           groups.add(currGroup);
422         }
423         Node result = null;
424         Node firstGroup = node.isNew() ? IR.arraylit(IR.nullNode()) : IR.arraylit();
425         Node joinedGroups =
426             IR.call(IR.getprop(firstGroup, IR.string("concat")), groups.toArray(new Node[0]));
427         if (node.isArrayLit()) {
428           result = joinedGroups;
429         } else if (node.isCall()) {
430           if (NodeUtil.mayHaveSideEffects(callee) && callee.isGetProp()) {
431             Node statement = node;
432             while (!NodeUtil.isStatement(statement)) {
433               statement = statement.getParent();
434             }
435             Node freshVar = IR.name(FRESH_SPREAD_VAR + compiler.getUniqueNameIdSupplier().get());
436             Node n = IR.var(freshVar.cloneTree());
437             n.useSourceInfoIfMissingFromForTree(statement);
438             statement.getParent().addChildBefore(n, statement);
439             callee.addChildToFront(IR.assign(freshVar.cloneTree(), callee.removeFirstChild()));
440             result = IR.call(
441                 IR.getprop(callee, IR.string("apply")),
442                 freshVar,
443                 joinedGroups);
444           } else {
445             Node context = callee.isGetProp() ? callee.getFirstChild().cloneTree() : IR.nullNode();
446             result = IR.call(IR.getprop(callee, IR.string("apply")), context, joinedGroups);
447           }
448         } else {
449           if (compiler.getOptions().getLanguageOut() == LanguageMode.ECMASCRIPT3) {
450             // TODO(tbreisacher): Support this in ES3 too by not relying on Function.bind.
451             cannotConvert(node, "\"...\" passed to a constructor (consider using --language_out=ES5)");
452           }
453           Node bindApply = NodeUtil.newQName(compiler,
454               "Function.prototype.bind.apply");
455           result = IR.newNode(IR.call(bindApply, callee, joinedGroups));
456         }
457         result.useSourceInfoIfMissingFromForTree(node);
458         parent.replaceChild(node, result);
459         compiler.reportCodeChange();
460       }

462       private void visitObject(Node obj) {
463         for (Node child : obj.children()) {
464           if (child.isComputedProp()) {
465             visitObjectWithComputedProperty(obj);
466             return;
467           }
468         }
469       }

471       private void visitObjectWithComputedProperty(Node obj) {
472         Preconditions.checkArgument(obj.isObjectLit());
473         List<Node> props = new ArrayList<>();
474         Node currElement = obj.getFirstChild();
475     
476         while (currElement != null) {
477           if (currElement.getBooleanProp(Node.COMPUTED_PROP_GETTER)
478               || currElement.getBooleanProp(Node.COMPUTED_PROP_SETTER)) {
479             cannotConvertYet(currElement, "computed getter/setter in an object literal");
480             return;
481           } else if (currElement.isGetterDef() || currElement.isSetterDef()) {
482             currElement = currElement.getNext();
483           } else {
484             Node nextNode = currElement.getNext();
485             obj.removeChild(currElement);
486             props.add(currElement);
487             currElement = nextNode;
488           }
489         }
490     
491         String objName = FRESH_COMP_PROP_VAR + compiler.getUniqueNameIdSupplier().get();
492     
493         props = Lists.reverse(props);
494         Node result = IR.name(objName);
495         for (Node propdef : props) {
496           if (propdef.isComputedProp()) {
497             Node propertyExpression = propdef.removeFirstChild();
498             Node value = propdef.removeFirstChild();
499             result = IR.comma(
500                 IR.assign(
501                     IR.getelem(
502                         IR.name(objName),
503                         propertyExpression),
504                     value),
505                 result);
506           } else {
507             if (!propdef.hasChildren()) {
508               Node name = IR.name(propdef.getString()).useSourceInfoIfMissingFrom(propdef);
509               propdef.addChildToBack(name);
510             }
511             Node val = propdef.removeFirstChild();
512             propdef.setType(Token.STRING);
513             Token type = propdef.isQuotedString() ? Token.GETELEM : Token.GETPROP;
514             Node access = new Node(type, IR.name(objName), propdef);
515             result = IR.comma(IR.assign(access, val), result);
516           }
517         }
518     
519         Node statement = obj;
520         while (!NodeUtil.isStatement(statement)) {
521           statement = statement.getParent();
522         }
523     
524         result.useSourceInfoIfMissingFromForTree(obj);
525         obj.getParent().replaceChild(obj, result);
526     
527         Node var = IR.var(IR.name(objName), obj);
528         var.useSourceInfoIfMissingFromForTree(statement);
529         statement.getParent().addChildBefore(var, statement);
530         compiler.reportCodeChange();
531       }

533       /**
534        * Classes are processed in 3 phases:
535        * <ol>
536        *   <li>The class name is extracted.
537        *   <li>Class members are processed and rewritten.
538        *   <li>The constructor is built.
539        * </ol>
540        */
541       private void visitClass(final Node classNode, final Node parent) {
542         checkClassReassignment(classNode);
543         // Collect Metadata
544         ClassDeclarationMetadata metadata = ClassDeclarationMetadata.create(classNode, parent);
545     
546         if (metadata == null || metadata.fullClassName == null) {
547           cannotConvert(parent, "Can only convert classes that are declarations or the right hand"
548               + " side of a simple assignment.");
549           return;
550         }
551         if (metadata.hasSuperClass() && !metadata.superClassNameNode.isQualifiedName()) {
552           compiler.report(JSError.make(metadata.superClassNameNode, DYNAMIC_EXTENDS_TYPE));
553           return;
554         }
555     
556         Preconditions.checkState(NodeUtil.isStatement(metadata.insertionPoint),
557             "insertion point must be a statement: %s", metadata.insertionPoint);
558     
559         Node constructor = null;
560         JSDocInfo ctorJSDocInfo = null;
561         // Process all members of the class
562         Node classMembers = classNode.getLastChild();
563         for (Node member : classMembers.children()) {
564           if ((member.isComputedProp()
565                   && (member.getBooleanProp(Node.COMPUTED_PROP_GETTER)
566                       || member.getBooleanProp(Node.COMPUTED_PROP_SETTER)))
567               || (member.isGetterDef() || member.isSetterDef())) {
568             visitComputedPropInClass(member, metadata);
569           } else if (member.isMemberFunctionDef() && member.getString().equals("constructor")) {
570             ctorJSDocInfo = member.getJSDocInfo();
571             constructor = member.getFirstChild().detachFromParent();
572             if (!metadata.anonymous) {
573               // Turns class Foo { constructor: function() {} } into function Foo() {},
574               // i.e. attaches the name to the ctor function.
575               constructor.replaceChild(
576                   constructor.getFirstChild(), metadata.classNameNode.cloneNode());
577             }
578           } else if (member.isEmpty()) {
579             // Do nothing.
580           } else {
581             Preconditions.checkState(member.isMemberFunctionDef() || member.isComputedProp(),
582                 "Unexpected class member:", member);
583             Preconditions.checkState(!member.getBooleanProp(Node.COMPUTED_PROP_VARIABLE),
584                 "Member variables should have been transpiled earlier:", member);
585             visitClassMember(member, metadata);
586           }
587         }
588     
589         if (metadata.definePropertiesObjForPrototype.hasChildren()) {
590           Node definePropsCall =
591               IR.exprResult(
592                   IR.call(
593                       NodeUtil.newQName(compiler, "Object.defineProperties"),
594                       NodeUtil.newQName(compiler, metadata.fullClassName + ".prototype"),
595                       metadata.definePropertiesObjForPrototype));
596           definePropsCall.useSourceInfoIfMissingFromForTree(classNode);
597           metadata.insertNodeAndAdvance(definePropsCall);
598     
599           visitObject(metadata.definePropertiesObjForPrototype);
600         }
601     
602         if (metadata.definePropertiesObjForClass.hasChildren()) {
603           Node definePropsCall =
604               IR.exprResult(
605                   IR.call(
606                       NodeUtil.newQName(compiler, "Object.defineProperties"),
607                       NodeUtil.newQName(compiler, metadata.fullClassName),
608                       metadata.definePropertiesObjForClass));
609           definePropsCall.useSourceInfoIfMissingFromForTree(classNode);
610           metadata.insertNodeAndAdvance(definePropsCall);
611     
612           visitObject(metadata.definePropertiesObjForClass);
613         }
614     
615     
616         Preconditions.checkNotNull(constructor);
617     
618         JSDocInfo classJSDoc = NodeUtil.getBestJSDocInfo(classNode);
619         JSDocInfoBuilder newInfo = JSDocInfoBuilder.maybeCopyFrom(classJSDoc);
620     
621         newInfo.recordConstructor();
622     
623         Node enclosingStatement = NodeUtil.getEnclosingStatement(classNode);
624         if (metadata.hasSuperClass()) {
625           String superClassString = metadata.superClassNameNode.getQualifiedName();
626           if (newInfo.isInterfaceRecorded()) {
627             newInfo.recordExtendedInterface(new JSTypeExpression(new Node(Token.BANG,
628                 IR.string(superClassString)),
629                 metadata.superClassNameNode.getSourceFileName()));
630           } else {
631             if (!classNode.isFromExterns()) {
632               Node inherits = IR.call(
633                   NodeUtil.newQName(compiler, INHERITS),
634                   NodeUtil.newQName(compiler, metadata.fullClassName),
635                   NodeUtil.newQName(compiler, superClassString));
636               Node inheritsCall = IR.exprResult(inherits);
637               compiler.ensureLibraryInjected("es6_runtime", false);
638     
639               inheritsCall.useSourceInfoIfMissingFromForTree(classNode);
640               enclosingStatement.getParent().addChildAfter(inheritsCall, enclosingStatement);
641             }
642             newInfo.recordBaseType(new JSTypeExpression(new Node(Token.BANG,
643                 IR.string(superClassString)),
644                 metadata.superClassNameNode.getSourceFileName()));
645           }
646         }
647     
648         addTypeDeclarations(metadata, enclosingStatement);
649     
650         updateClassJsDoc(ctorJSDocInfo, newInfo);
651     
652         if (NodeUtil.isStatement(classNode)) {
653           constructor.getFirstChild().setString("");
654           Node ctorVar = IR.let(metadata.classNameNode.cloneNode(), constructor);
655           ctorVar.useSourceInfoIfMissingFromForTree(classNode);
656           parent.replaceChild(classNode, ctorVar);
657         } else {
658           parent.replaceChild(classNode, constructor);
659         }
660     
661         if (NodeUtil.isStatement(constructor)) {
662           constructor.setJSDocInfo(newInfo.build());
663         } else if (parent.isName()) {
664           // The constructor function is the RHS of a var statement.
665           // Add the JSDoc to the VAR node.
666           Node var = parent.getParent();
667           var.setJSDocInfo(newInfo.build());
668         } else if (constructor.getParent().isName()) {
669           // Is a newly created VAR node.
670           Node var = constructor.getGrandparent();
671           var.setJSDocInfo(newInfo.build());
672         } else if (parent.isAssign()) {
673           // The constructor function is the RHS of an assignment.
674           // Add the JSDoc to the ASSIGN node.
675           parent.setJSDocInfo(newInfo.build());
676         } else {
677           throw new IllegalStateException("Unexpected parent node " + parent);
678         }
679     
680         compiler.reportCodeChange();
681       }

683       /**
684        * @param ctorInfo the JSDocInfo from the constructor method of the ES6 class.
685        * @param newInfo the JSDocInfo that will be added to the constructor function in the ES3 output
686        */
687       private void updateClassJsDoc(@Nullable JSDocInfo ctorInfo, JSDocInfoBuilder newInfo) {
688         // Classes are @struct by default.
689         if (!newInfo.isUnrestrictedRecorded() && !newInfo.isDictRecorded()
690             && !newInfo.isStructRecorded()) {
691           newInfo.recordStruct();
692         }
693     
694         if (ctorInfo != null) {
695           if (!ctorInfo.getSuppressions().isEmpty()) {
696             newInfo.recordSuppressions(ctorInfo.getSuppressions());
697           }
698     
699           for (String param : ctorInfo.getParameterNames()) {
700             newInfo.recordParameter(param, ctorInfo.getParameterType(param));
701             newInfo.recordParameterDescription(param, ctorInfo.getDescriptionForParameter(param));
702           }
703     
704           for (JSTypeExpression thrown : ctorInfo.getThrownTypes()) {
705             newInfo.recordThrowType(thrown);
706             newInfo.recordThrowDescription(thrown, ctorInfo.getThrowsDescriptionForType(thrown));
707           }
708     
709           JSDocInfo.Visibility visibility = ctorInfo.getVisibility();
710           if (visibility != null && visibility != JSDocInfo.Visibility.INHERITED) {
711             newInfo.recordVisibility(visibility);
712           }
713     
714           if (ctorInfo.isDeprecated()) {
715             newInfo.recordDeprecated();
716           }
717     
718           if (ctorInfo.getDeprecationReason() != null
719               && !newInfo.isDeprecationReasonRecorded()) {
720             newInfo.recordDeprecationReason(ctorInfo.getDeprecationReason());
721           }
722     
723           newInfo.mergePropertyBitfieldFrom(ctorInfo);
724     
725           for (String templateType : ctorInfo.getTemplateTypeNames()) {
726             newInfo.recordTemplateTypeName(templateType);
727           }
728         }
729       }

731       /**
732        * @param node A getter or setter node.
733        */
734       private JSTypeExpression getTypeFromGetterOrSetter(Node node) {
735         JSDocInfo info = node.getJSDocInfo();
736         if (info != null) {
737           boolean getter = node.isGetterDef() || node.getBooleanProp(Node.COMPUTED_PROP_GETTER);
738           if (getter && info.getReturnType() != null) {
739             return info.getReturnType();
740           } else {
741             Set<String> paramNames = info.getParameterNames();
742             if (paramNames.size() == 1) {
743               return info.getParameterType(Iterables.getOnlyElement(info.getParameterNames()));
744             }
745           }
746         }
747     
748         return new JSTypeExpression(new Node(Token.QMARK), node.getSourceFileName());
749       }

751       /**
752        * @param member A getter or setter, or a computed property that is a getter/setter.
753        */
754       private void addToDefinePropertiesObject(ClassDeclarationMetadata metadata, Node member) {
755         Node obj =
756             member.isStaticMember()
757                 ? metadata.definePropertiesObjForClass
758                 : metadata.definePropertiesObjForPrototype;
759         Node prop =
760             member.isComputedProp()
761                 ? NodeUtil.getFirstComputedPropMatchingKey(obj, member.getFirstChild())
762                 : NodeUtil.getFirstPropMatchingKey(obj, member.getString());
763         if (prop == null) {
764           prop =
765               IR.objectlit(
766                   IR.stringKey("configurable", IR.trueNode()),
767                   IR.stringKey("enumerable", IR.trueNode()));
768           if (member.isComputedProp()) {
769             obj.addChildToBack(IR.computedProp(member.getFirstChild().cloneTree(), prop));
770           } else {
771             obj.addChildToBack(IR.stringKey(member.getString(), prop));
772           }
773         }
774     
775         Node function = member.getLastChild();
776         JSDocInfoBuilder info = JSDocInfoBuilder.maybeCopyFrom(
777             NodeUtil.getBestJSDocInfo(function));
778     
779         info.recordThisType(new JSTypeExpression(new Node(
780             Token.BANG, IR.string(metadata.fullClassName)), member.getSourceFileName()));
781         Node stringKey =
782             IR.stringKey(
783                 (member.isGetterDef() || member.getBooleanProp(Node.COMPUTED_PROP_GETTER))
784                     ? "get"
785                     : "set",
786                 function.detachFromParent());
787         stringKey.setJSDocInfo(info.build());
788         prop.addChildToBack(stringKey);
789         prop.useSourceInfoIfMissingFromForTree(member);
790       }

792       private void visitComputedPropInClass(Node member, ClassDeclarationMetadata metadata) {
793         if (member.isComputedProp() && member.isStaticMember()) {
794           cannotConvertYet(member, "Static computed property");
795           return;
796         }
797         if (member.isComputedProp() && !member.getFirstChild().isQualifiedName()) {
798           cannotConvert(member.getFirstChild(), "Computed property with non-qualified-name key");
799           return;
800         }
801     
802         JSTypeExpression typeExpr = getTypeFromGetterOrSetter(member).clone();
803         addToDefinePropertiesObject(metadata, member);
804     
805         Map<String, JSDocInfo> membersToDeclare;
806         String memberName;
807         if (member.isComputedProp()) {
808           Preconditions.checkState(!member.isStaticMember());
809           membersToDeclare = metadata.prototypeComputedPropsToDeclare;
810           memberName = member.getFirstChild().getQualifiedName();
811         } else {
812           membersToDeclare = member.isStaticMember()
813               ? metadata.classMembersToDeclare
814               : metadata.prototypeMembersToDeclare;
815           memberName = member.getString();
816         }
817         JSDocInfo existingJSDoc = membersToDeclare.get(memberName);
818         JSTypeExpression existingType = existingJSDoc == null ? null : existingJSDoc.getType();
819         if (existingType != null && !existingType.equals(typeExpr)) {
820           compiler.report(JSError.make(member, CONFLICTING_GETTER_SETTER_TYPE, memberName));
821         } else {
822           JSDocInfoBuilder jsDoc = new JSDocInfoBuilder(false);
823           jsDoc.recordType(typeExpr);
824           if (member.getJSDocInfo() != null && member.getJSDocInfo().isExport()) {
825             jsDoc.recordExport();
826           }
827           if (member.isStaticMember() && !member.isComputedProp()) {
828             jsDoc.recordNoCollapse();
829           }
830           membersToDeclare.put(memberName, jsDoc.build());
831         }
832       }

834       /**
835        * Handles transpilation of a standard class member function. Getters, setters, and the
836        * constructor are not handled here.
837        */
838       private void visitClassMember(
839           Node member, ClassDeclarationMetadata metadata) {
840         Node qualifiedMemberAccess = getQualifiedMemberAccess(
841             member,
842             NodeUtil.newQName(compiler, metadata.fullClassName),
843             NodeUtil.newQName(compiler, metadata.fullClassName + ".prototype"));
844         Node method = member.getLastChild().detachFromParent();
845     
846         Node assign = IR.assign(qualifiedMemberAccess, method);
847         assign.useSourceInfoIfMissingFromForTree(member);
848     
849         JSDocInfo info = member.getJSDocInfo();
850         if (member.isStaticMember() && NodeUtil.referencesThis(assign.getLastChild())) {
851           JSDocInfoBuilder memberDoc = JSDocInfoBuilder.maybeCopyFrom(info);
852           memberDoc.recordThisType(
853               new JSTypeExpression(new Node(Token.BANG, new Node(Token.QMARK)),
854               member.getSourceFileName()));
855           info = memberDoc.build();
856         }
857         if (info != null) {
858           assign.setJSDocInfo(info);
859         }
860     
861         Node newNode = NodeUtil.newExpr(assign);
862         metadata.insertNodeAndAdvance(newNode);
863       }

865       /**
866        * Add declarations for properties that were defined with a getter and/or setter,
867        * so that the typechecker knows those properties exist on the class.
868        * This is a temporary solution. Eventually, the type checker should understand
869        * Object.defineProperties calls directly.
870        */
871       private void addTypeDeclarations(ClassDeclarationMetadata metadata, Node insertionPoint) {
872         for (Map.Entry<String, JSDocInfo> entry : metadata.prototypeMembersToDeclare.entrySet()) {
873           String declaredMember = entry.getKey();
874           Node declaration = IR.getprop(
875               NodeUtil.newQName(compiler, metadata.fullClassName + ".prototype"),
876               IR.string(declaredMember));
877           declaration.setJSDocInfo(entry.getValue());
878           declaration =
879               IR.exprResult(declaration).useSourceInfoIfMissingFromForTree(metadata.classNameNode);
880           insertionPoint.getParent().addChildAfter(declaration, insertionPoint);
881           insertionPoint = declaration;
882         }
883         for (Map.Entry<String, JSDocInfo> entry : metadata.classMembersToDeclare.entrySet()) {
884           String declaredMember = entry.getKey();
885           Node declaration = IR.getprop(
886               NodeUtil.newQName(compiler, metadata.fullClassName),
887               IR.string(declaredMember));
888           declaration.setJSDocInfo(entry.getValue());
889           declaration =
890               IR.exprResult(declaration).useSourceInfoIfMissingFromForTree(metadata.classNameNode);
891           insertionPoint.getParent().addChildAfter(declaration, insertionPoint);
892           insertionPoint = declaration;
893         }
894         for (Map.Entry<String, JSDocInfo> entry : metadata.prototypeComputedPropsToDeclare.entrySet()) {
895           String declaredMember = entry.getKey();
896           Node declaration = IR.getelem(
897               NodeUtil.newQName(compiler, metadata.fullClassName + ".prototype"),
898               NodeUtil.newQName(compiler, declaredMember));
899           declaration.setJSDocInfo(entry.getValue());
900           declaration =
901               IR.exprResult(declaration).useSourceInfoIfMissingFromForTree(metadata.classNameNode);
902           insertionPoint.getParent().addChildAfter(declaration, insertionPoint);
903           insertionPoint = declaration;
904         }
905       }

907       /**
908        * Constructs a Node that represents an access to the given class member, qualified by either the
909        * static or the instance access context, depending on whether the member is static.
910        *
911        * <p><b>WARNING:</b> {@code member} may be modified/destroyed by this method, do not use it
912        * afterwards.
913        */
914       private static Node getQualifiedMemberAccess(Node member,
915           Node staticAccess, Node instanceAccess) {
916         Node context = member.isStaticMember() ? staticAccess : instanceAccess;
917         context = context.cloneTree();
918         if (member.isComputedProp()) {
919           return IR.getelem(context, member.removeFirstChild());
920         } else {
921           Node methodName = member.getFirstChild().getFirstChild();
922           return IR.getprop(context, IR.string(member.getString()).useSourceInfoFrom(methodName));
923         }
924       }

980       /**
981        * Represents static metadata on a class declaration expression - i.e. the qualified name that a
982        * class declares (directly or by assignment), whether it's anonymous, and where transpiled code
983        * should be inserted (i.e. which object will hold the prototype after transpilation).
984        */
985       static class ClassDeclarationMetadata {
986         /** A statement node. Transpiled methods etc of the class are inserted after this node. */
987         private Node insertionPoint;
988     
989         /**
990          * An object literal node that will be used in a call to Object.defineProperties, to add getters
991          * and setters to the prototype.
992          */
993         private final Node definePropertiesObjForPrototype;
994     
995         /**
996          * An object literal node that will be used in a call to Object.defineProperties, to add getters
997          * and setters to the class.
998          */
999         private final Node definePropertiesObjForClass;
1000     
1001         // Normal declarations to be added to the prototype: Foo.prototype.bar
1002         private final Map<String, JSDocInfo> prototypeMembersToDeclare;
1003     
1004         // Computed property declarations to be added to the prototype: Foo.prototype[bar]
1005         private final Map<String, JSDocInfo> prototypeComputedPropsToDeclare;
1006     
1007         // Normal declarations to be added to the class: Foo.bar
1008         private final Map<String, JSDocInfo> classMembersToDeclare;
1009     
1010         /**
1011          * The fully qualified name of the class, which will be used in the output. May come from the
1012          * class itself or the LHS of an assignment.
1013          */
1014         final String fullClassName;
1015         /** Whether the constructor function in the output should be anonymous. */
1016         final boolean anonymous;
1017         final Node classNameNode;
1018         final Node superClassNameNode;
1019     
1020         private ClassDeclarationMetadata(Node insertionPoint, String fullClassName,
1021             boolean anonymous, Node classNameNode, Node superClassNameNode) {
1022           this.insertionPoint = insertionPoint;
1023           this.definePropertiesObjForClass = IR.objectlit();
1024           this.definePropertiesObjForPrototype = IR.objectlit();
1025           this.prototypeMembersToDeclare = new LinkedHashMap<>();
1026           this.prototypeComputedPropsToDeclare = new LinkedHashMap<>();
1027           this.classMembersToDeclare = new LinkedHashMap<>();
1028           this.fullClassName = fullClassName;
1029           this.anonymous = anonymous;
1030           this.classNameNode = classNameNode;
1031           this.superClassNameNode = superClassNameNode;
1032         }
1033     
1034         static ClassDeclarationMetadata create(Node classNode, Node parent) {
1035           Node classNameNode = classNode.getFirstChild();
1036           Node superClassNameNode = classNameNode.getNext();
1037     
1038           // If this is a class statement, or a class expression in a simple
1039           // assignment or var statement, convert it. In any other case, the
1040           // code is too dynamic, so return null.
1041           if (NodeUtil.isClassDeclaration(classNode)) {
1042             return new ClassDeclarationMetadata(classNode, classNameNode.getString(), false,
1043                 classNameNode, superClassNameNode);
1044           } else if (parent.isAssign() && parent.getParent().isExprResult()) {
1045             // Add members after the EXPR_RESULT node:
1046             // example.C = class {}; example.C.prototype.foo = function() {};
1047             String fullClassName = parent.getFirstChild().getQualifiedName();
1048             if (fullClassName == null) {
1049               return null;
1050             }
1051             return new ClassDeclarationMetadata(parent.getParent(), fullClassName, true, classNameNode,
1052                 superClassNameNode);
1053           } else if (parent.isName()) {
1054             // Add members after the 'var' statement.
1055             // var C = class {}; C.prototype.foo = function() {};
1056             return new ClassDeclarationMetadata(parent.getParent(), parent.getString(), true,
1057                 classNameNode, superClassNameNode);
1058           } else {
1059             // Cannot handle this class declaration.
1060             return null;
1061           }
1062         }
1063     
1064         void insertNodeAndAdvance(Node newNode) {
1065           insertionPoint.getParent().addChildAfter(newNode, insertionPoint);
1066           insertionPoint = newNode;
1067         }
1068     
1069         boolean hasSuperClass() {
1070           return !superClassNameNode.isEmpty();
1071         }
1072       }
