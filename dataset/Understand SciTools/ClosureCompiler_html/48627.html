28     /**
29      * Instruments functions for when functions first get called and defined.
30      *
31      * This pass be used to instrument code to:
32      *  1. Gather statistics from real users in the wild.
33      *  2. Incorporate utilization statistics into Selenium tests
34      *  3. Access utilization statistics from an app's debug UI
35      *
36      * By parametrizing the whole instrumentation process we expect to be
37      * able to support a wide variety of use cases and minimize the cost
38      * of developing new instrumentation schemes.
39      *
40      * TODO(user): This pass currently runs just before the variable and
41      * property renaming near the end of the optimization pass.  I think
42      * Mark put it there to minimize the difference between the code
43      * generated with/without instrumentation; instrumentation makes
44      * several optimization passes do less, for example inline functions.
45      *
46      * My opinion is that we want utilization/profiling information for
47      * all function.  This pass should run before most passes that modify
48      * the AST (exception being the localization pass, which makes
49      * assumptions about the structure of the AST).  We should move the
50      * pass up, list inlined functions or give clients the option to
51      * instrument before or after optimization.
52      *
53      */
54     class InstrumentFunctions implements CompilerPass {
55     
56       private final AbstractCompiler compiler;
57       private final FunctionNames functionNames;
58       private final String appNameStr;
59       private final String initCodeSource;
60       private final String definedFunctionName;
61       private final String reportFunctionName;
62       private final String reportFunctionExitName;
63       private final String appNameSetter;
64       private final List<String> declarationsToRemove;
65     
66       /**
67        * Creates an instrument functions compiler pass.
68        *
69        * @param compiler          The JSCompiler
70        * @param functionNames     Assigned function identifiers.
71        * @param template          Instrumentation template; for use during error reporting only.
72        * @param appNameStr        String to pass to appNameSetter.
73        */
74       InstrumentFunctions(AbstractCompiler compiler,
75                           FunctionNames functionNames,
76                           Instrumentation template,
77                           String appNameStr) {
78         this.compiler = compiler;
79         this.functionNames = functionNames;
80         this.appNameStr = appNameStr;
81     
82         StringBuilder initCodeSourceBuilder = new StringBuilder();
83         for (String line : template.getInitList()) {
84           initCodeSourceBuilder.append(line).append("\n");
85         }
86         this.initCodeSource = initCodeSourceBuilder.toString();
87     
88         this.definedFunctionName = template.getReportDefined();
89         this.reportFunctionName = template.getReportCall();
90         this.reportFunctionExitName = template.getReportExit();
91         this.appNameSetter = template.getAppNameSetter();
92     
93         this.declarationsToRemove = ImmutableList.copyOf(
94             template.getDeclarationToRemoveList());
95       }
96     
97       @Override
98       public void process(Node externs, Node root) {
99         Node initCode = null;
100         if (!initCodeSource.isEmpty()) {
101           Node initCodeRoot = compiler.parseSyntheticCode(
102               "template:init", initCodeSource);
103           if (initCodeRoot != null && initCodeRoot.getFirstChild() != null) {
104             initCode = initCodeRoot.removeChildren();
105           } else {
106             return;  // parse failure
107           }
108         }
109     
110         NodeTraversal.traverseEs6(compiler, root,
111                                new RemoveCallback(declarationsToRemove));
112         NodeTraversal.traverseEs6(compiler, root, new InstrumentCallback());
113     
114         if (!appNameSetter.isEmpty()) {
115           Node call = IR.call(
116               IR.name(appNameSetter),
117               IR.string(appNameStr));
118           call.putBooleanProp(Node.FREE_CALL, true);
119           Node expr = IR.exprResult(call);
120     
121           Node addingRoot = compiler.getNodeForCodeInsertion(null);
122           addingRoot.addChildrenToFront(expr.useSourceInfoIfMissingFromForTree(addingRoot));
123           compiler.reportCodeChange();
124         }
125     
126         if (initCode != null) {
127           Node addingRoot = compiler.getNodeForCodeInsertion(null);
128           addingRoot.addChildrenToFront(initCode);
129           compiler.reportCodeChange();
130         }
131       }

133       /**
134        * The application must refer to these variables to output them so the
135        * application must also declare these variables for the first
136        * {@link VarCheck} pass. These declarations must be removed before the
137        * second {@link VarCheck} pass. Otherwise, the second pass would warn about
138        * duplicate declarations.
139        */
140       private static class RemoveCallback extends AbstractPostOrderCallback {
141         private final List<String> removable;
142         RemoveCallback(List<String> removable) {
143           this.removable = removable;
144         }
145     
146         @Override
147         public void visit(NodeTraversal t, Node n, Node parent) {
148           if (NodeUtil.isVarDeclaration(n) && removable.contains(n.getString())) {
149             parent.removeChild(n);
150             if (!parent.hasChildren()) {
151               parent.getParent().removeChild(parent);
152             }
153           }
154         }
155       }

157       /**
158        * Traverse a function's body by instrument return sites by
159        * inserting calls to {@code reportFunctionExitName}.  If the
160        * function is missing an explicit return statement in some control
161        * path, this pass inserts a call to {@code reportFunctionExitName}
162        * as the last statement in the function's body.
163        *
164        * Example:
165        * Input:
166        * function f() {
167        *   if (pred) {
168        *     return a;
169        *   }
170        * }
171        *
172        * Template:
173        * reportFunctionExitName: "onExitFn"
174        *
175        * Output:
176        * function f() {
177        *   if (pred) {
178        *     return onExitFn(0, a, "f");
179        *   }
180        *   onExitFn(0, undefined, "f");
181        * }
182        *
183        **/
184       private class InstrumentReturns implements NodeTraversal.Callback {
185         private final int functionId;
186         private final String functionName;
187     
188         /**
189          * @param functionId Function identifier computed by FunctionNames;
190          *     used as first argument to {@code reportFunctionExitName}
191          *     {@code reportFunctionExitName} must be a 3 argument function that
192          *     returns it's second argument.
193          * @param functionName Function name.
194          */
195         InstrumentReturns(int functionId, String functionName) {
196           this.functionId = functionId;
197           this.functionName = functionName;
198         }
199     
200         /**
201          * @param function function with id == this.functionId
202          */
203         void process(Node function) {
204           Node body = function.getLastChild();
205           NodeTraversal.traverseEs6(compiler, body, this);
206     
207           if (!allPathsReturn(function)) {
208             Node call = newReportFunctionExitNode(function, null);
209             Node expr = IR.exprResult(call).useSourceInfoIfMissingFromForTree(function);
210             body.addChildToBack(expr);
211             compiler.reportCodeChange();
212           }
213         }
214     
215         @Override
216         public boolean shouldTraverse(NodeTraversal t, Node n, Node parent) {
217           return !n.isFunction();
218         }
219     
220         @Override
221         public void visit(NodeTraversal t, Node n, Node parent) {
222           if (!n.isReturn()) {
223             return;
224           }
225     
226           Node returnRhs = n.removeFirstChild();
227           Node call = newReportFunctionExitNode(n, returnRhs);
228           n.addChildToFront(call);
229           compiler.reportCodeChange();
230         }
231     
232         private Node newReportFunctionExitNode(Node infoNode, Node returnRhs) {
233           Node call = IR.call(
234               IR.name(reportFunctionExitName),
235               IR.number(functionId),
236               (returnRhs != null) ? returnRhs : IR.name("undefined"),
237               IR.string(functionName));
238           call.putBooleanProp(Node.FREE_CALL, true);
239           call.useSourceInfoFromForTree(infoNode);
240           return call;
241         }
242     
243         /**
244          * @return true if all paths from block must exit with an explicit return.
245          */
246         private boolean allPathsReturn(Node function) {
247           // Computes the control flow graph.
248           ControlFlowAnalysis cfa = new ControlFlowAnalysis(
249               compiler, false, false);
250           cfa.process(null, function);
251           ControlFlowGraph<Node> cfg = cfa.getCfg();
252     
253           Node returnPathsParent = cfg.getImplicitReturn().getValue();
254           for (DiGraphNode<Node, Branch> pred :
255             cfg.getDirectedPredNodes(returnPathsParent)) {
256             Node n = pred.getValue();
257             if (!n.isReturn()) {
258               return false;
259             }
260           }
261           return true;
262         }
263       }
264     
265       private class InstrumentCallback extends AbstractPostOrderCallback {
266         @Override
267         public void visit(NodeTraversal t, Node n, Node parent) {
268           if (!n.isFunction()) {
269             return;
270           }
271     
272           int id = functionNames.getFunctionId(n);
273           if (id < 0) {
274             // Function node was added during compilation; don't instrument.
275             return;
276           }
277     
278           String name = functionNames.getFunctionName(n);
279     
280           if (!reportFunctionName.isEmpty()) {
281             Node body = n.getLastChild();
282             Node call = IR.call(
283                 IR.name(reportFunctionName),
284                 IR.number(id),
285                 IR.string(name),
286                 IR.name("arguments"));
287             call.putBooleanProp(Node.FREE_CALL, true);
288             Node expr = IR.exprResult(call);
289             expr.useSourceInfoFromForTree(n);
290             body.addChildToFront(expr);
291             compiler.reportCodeChange();
292           }
293     
294           if (!reportFunctionExitName.isEmpty()) {
295             (new InstrumentReturns(id, name)).process(n);
296           }
297     
298           if (!definedFunctionName.isEmpty()) {
299             Node call = IR.call(
300                 IR.name(definedFunctionName),
301                 IR.number(id),
302                 IR.string(name));
303             call.putBooleanProp(Node.FREE_CALL, true);
304             call.useSourceInfoFromForTree(n);
305             Node expr = NodeUtil.newExpr(call);
306     
307             Node addingRoot = null;
308             if (NodeUtil.isFunctionDeclaration(n)) {
309               JSModule module = t.getModule();
310               addingRoot = compiler.getNodeForCodeInsertion(module);
311               addingRoot.addChildToFront(expr);
312             } else {
313               Node beforeChild = n;
314               for (Node ancestor : n.getAncestors()) {
315                 Token type = ancestor.getType();
316                 if (type == Token.BLOCK || type == Token.SCRIPT) {
317                   addingRoot = ancestor;
318                   break;
319                 }
320                 beforeChild = ancestor;
321               }
322               addingRoot.addChildBefore(expr, beforeChild);
323             }
324             compiler.reportCodeChange();
325           }
326         }
327       }
