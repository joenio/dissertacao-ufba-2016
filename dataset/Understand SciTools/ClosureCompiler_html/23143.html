46     package com.google.javascript.rhino;
47     
48     import com.google.common.annotations.GwtIncompatible;
49     import com.google.javascript.rhino.jstype.TernaryValue;
50     
51     /**
52      * Helper methods for parsing JavaScript.
53      * These methods use unsupported features in GWT's emulation of java.lang.Character.
54      *
55      * TODO(moz): Add a GWT-compatible version in the super-source directory.
56      */
57     @GwtIncompatible("Unsupported java.lang.Character fields")
58     public class TokenUtil {
59       static boolean isJSIdentifier(String s) {
60         int length = s.length();
61     
62         if (length == 0
63             || Character.isIdentifierIgnorable(s.charAt(0))
64             || !Character.isJavaIdentifierStart(s.charAt(0))) {
65           return false;
66         }
67     
68         for (int i = 1; i < length; i++) {
69           if (Character.isIdentifierIgnorable(s.charAt(i))
70               || !Character.isJavaIdentifierPart(s.charAt(i))) {
71             return false;
72           }
73         }
74         return true;
75       }
76     
77       /* As defined in ECMA.  jsscan.c uses C isspace() (which allows
78        * \v, I think.)  note that code in getChar() implicitly accepts
79        * '\r' == \u000D as well.
80        */
81       public static boolean isJSSpace(int c) {
82         if (c <= 127) {
83           return c == 0x20 || c == 0x9 || c == 0xC || c == 0xB;
84         } else {
85           return c == 0xA0
86               || Character.getType((char) c) == Character.SPACE_SEPARATOR;
87         }
88       }
89     
90       public static boolean isJSFormatChar(int c) {
91         return c > 127 && Character.getType(c) == Character.FORMAT;
92       }
93     
94       public static boolean isWhitespace(int c) {
95         return Character.isWhitespace(c);
96       }
97     
98       /**
99        * Copied from Rhino's ScriptRuntime
100        */
101       public static TernaryValue isStrWhiteSpaceChar(int c) {
102         switch (c) {
103           case '\u000B': // <VT>
104             return TernaryValue.UNKNOWN;  // IE says "no", ECMAScript says "yes"
105           case ' ': // <SP>
106           case '\n': // <LF>
107           case '\r': // <CR>
108           case '\t': // <TAB>
109           case '\u00A0': // <NBSP>
110           case '\u000C': // <FF>
111           case '\u2028': // <LS>
112           case '\u2029': // <PS>
113           case '\uFEFF': // <BOM>
114             return TernaryValue.TRUE;
115           default:
116             return (Character.getType(c) == Character.SPACE_SEPARATOR)
117                 ? TernaryValue.TRUE : TernaryValue.FALSE;
118         }
119       }
120     }
