17     package com.google.javascript.jscomp.deps;
18     
19     import com.google.javascript.jscomp.ErrorManager;
20     import com.google.javascript.jscomp.LoggerErrorManager;
21     
22     import java.util.ArrayList;
23     import java.util.Collection;
24     import java.util.HashSet;
25     import java.util.List;
26     import java.util.Set;
27     import java.util.logging.Logger;
28     
29     /**
30      * Class for resolving Closure dependencies.
31      *
32      * Given a valid deps.js file the dependency tree is parsed and stored in
33      * memory. The DependencyResolver can then be used to calculate the full list of
34      * transitive dependencies from: a block of code (
35      * {@link #getDependencies(String)}), or a list of symbols
36      * {@link #getDependencies(Collection)}.
37      *
38      */
39     public final class DefaultDependencyResolver implements DependencyResolver  {
40     
41       /** Filename for Closure's base.js file which is always added. */
42       static final String CLOSURE_BASE = "base.js";
43     
44       /** Provide for Closure's base.js. */
45       static final String CLOSURE_BASE_PROVIDE = "goog";
46     
47       /** Source files used to look up the dependencies. */
48       private final List<DependencyFile> depsFiles;
49     
50       /**
51        * Flag that determines if the resolver will strictly require all
52        * goog.requires.
53        * */
54       private final boolean strictRequires;
55     
56       /** Logger for DependencyResolver. */
57       private static Logger logger =
58           Logger.getLogger(DefaultDependencyResolver.class.getName());
59     
60       /**
61        * Creates a new dependency resolver.
62        * @param depsFiles List of deps file.
63        * @param strictRequires Determines if the resolver will through an exception
64        *     on a missing dependency.
65        */
66       public DefaultDependencyResolver(
67           List<DependencyFile> depsFiles, boolean strictRequires) {
68         this.depsFiles = depsFiles;
69         this.strictRequires = strictRequires;
70       }
71     
72       /** Gets a list of dependencies for the provided code. */
73       @Override
74       public List<String> getDependencies(String code)
75           throws ServiceException {
76         return getDependencies(parseRequires(code, true));
77       }
78     
79       /** Gets a list of dependencies for the provided list of symbols. */
80       @Override
81       public List<String> getDependencies(Collection<String> symbols)
82           throws ServiceException {
83         return getDependencies(symbols, new HashSet<String>());
84       }
85     
86       /**
87        * @param code The raw code to be parsed for requires.
88        * @param seen The set of already seen symbols.
89        * @param addClosureBaseFile Indicates whether the closure base file should be
90        *        added to the dependency list.
91        * @return A list of filenames for each of the dependencies for the provided
92        *         code.
93        * @throws ServiceException
94        */
95       @Override
96       public List<String> getDependencies(String code, Set<String> seen,
97           boolean addClosureBaseFile) throws ServiceException {
98         return getDependencies(parseRequires(code, addClosureBaseFile), seen);
99       }
100     
101       /**
102        * @param symbols A list of required symbols.
103        * @param seen The set of already seen symbols.
104        * @return A list of filenames for each of the required symbols.
105        * @throws ServiceException
106        */
107       @Override
108       public List<String> getDependencies(Collection<String> symbols,
109           Set<String> seen) throws ServiceException {
110         List<String> list = new ArrayList<>();
111         for (DependencyFile depsFile : depsFiles) {
112           depsFile.ensureUpToDate();
113         }
114     
115         for (String symbol : symbols) {
116           addDependency(symbol, seen, list);
117         }
118         return list;
119       }
120     
121       /**
122        * Adds all the transitive dependencies for a symbol to the provided list. The
123        * set is used to avoid adding dupes while keeping the correct order. NOTE:
124        * Use of a LinkedHashSet would require reversing the results to get correct
125        * dependency ordering.
126        */
127       private void addDependency(String symbol, Set<String> seen, List<String> list)
128           throws ServiceException {
129         DependencyInfo dependency = getDependencyInfo(symbol);
130         if (dependency == null) {
131           if (this.strictRequires) {
132             throw new ServiceException("Unknown require of " + symbol);
133           }
134         } else if (!seen.containsAll(dependency.getProvides())) {
135           seen.addAll(dependency.getProvides());
136           for (String require : dependency.getRequires()) {
137             addDependency(require, seen, list);
138           }
139           list.add(dependency.getPathRelativeToClosureBase());
140         }
141       }
142     
143       /**
144        * Parses a block of code for goog.require statements and extracts the
145        * required symbols.
146        */
147       private Collection<String> parseRequires(
148           String code, boolean addClosureBase) {
149         ErrorManager errorManager = new LoggerErrorManager(logger);
150         JsFileParser parser = new JsFileParser(errorManager);
151         DependencyInfo deps =
152             parser.parseFile("<unknown path>", "<unknown path>", code);
153         List<String> requires = new ArrayList<>();
154         if (addClosureBase) {
155           requires.add(CLOSURE_BASE_PROVIDE);
156         }
157         requires.addAll(deps.getRequires());
158         errorManager.generateReport();
159         return requires;
160       }
161     
162       /** Looks at each of the dependency files for dependency information. */
163       private DependencyInfo getDependencyInfo(String symbol) {
164         for (DependencyFile depsFile : depsFiles) {
165           DependencyInfo di = depsFile.getDependencyInfo(symbol);
166           if (di != null) {
167             return di;
168           }
169         }
170         return null;
171       }

173     }
