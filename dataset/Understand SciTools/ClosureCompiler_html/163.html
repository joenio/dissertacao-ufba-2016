16     package com.google.javascript.jscomp;
17     
18     import com.google.javascript.jscomp.CompilerOptions.LanguageMode;
19     import com.google.javascript.rhino.InputId;
20     import com.google.javascript.rhino.Node;
21     import com.google.javascript.rhino.Token;
22     
23     import junit.framework.TestCase;
24     
25     /** Base class for tests that exercise {@link CodePrinter}. */
26     public abstract class CodePrinterTestBase extends TestCase {
27       // If this is set, ignore parse warnings and only fail the test
28       // for parse errors.
29       protected boolean allowWarnings = false;
30       protected boolean trustedStrings = true;
31       protected boolean preserveTypeAnnotations = false;
32       protected LanguageMode languageMode = LanguageMode.ECMASCRIPT5;
33       protected Compiler lastCompiler = null;
34     
35       @Override public void setUp() {
36         allowWarnings = false;
37         preserveTypeAnnotations = false;
38         trustedStrings = true;
39         lastCompiler = null;
40         languageMode = LanguageMode.ECMASCRIPT5;
41       }
42     
43       Node parse(String js) {
44         return parse(js, false);
45       }
46     
47       Node parse(String js, boolean checkTypes) {
48         Compiler compiler = new Compiler();
49         lastCompiler = compiler;
50         CompilerOptions options = new CompilerOptions();
51         options.setTrustedStrings(trustedStrings);
52         options.preserveTypeAnnotations = preserveTypeAnnotations;
53     
54         // Allow getters and setters.
55         options.setLanguageIn(languageMode);
56         compiler.initOptions(options);
57         Node n = compiler.parseTestCode(js);
58     
59         if (checkTypes) {
60           DefaultPassConfig passConfig = new DefaultPassConfig(null);
61           CompilerPass typeResolver = passConfig.resolveTypes.create(compiler);
62           Node externs = new Node(Token.SCRIPT);
63           externs.setInputId(new InputId("externs"));
64           Node externAndJsRoot = new Node(Token.BLOCK, externs, n);
65           externAndJsRoot.setIsSyntheticBlock(true);
66           typeResolver.process(externs, n);
67           CompilerPass inferTypes = passConfig.inferTypes.create(compiler);
68           inferTypes.process(externs, n);
69         }
70     
71         checkUnexpectedErrorsOrWarnings(compiler, 0);
72         return n;
73       }
74     
75       private void checkUnexpectedErrorsOrWarnings(
76           Compiler compiler, int expected) {
77         int actual = compiler.getErrors().length;
78         if (!allowWarnings) {
79           actual += compiler.getWarnings().length;
80         }
81     
82         if (actual != expected) {
83           String msg = "";
84           for (JSError err : compiler.getErrors()) {
85             msg += "Error:" + err + "\n";
86           }
87           if (!allowWarnings) {
88             for (JSError err : compiler.getWarnings()) {
89               msg += "Warning:" + err + "\n";
90             }
91           }
92           assertEquals("Unexpected warnings or errors.\n " + msg, expected, actual);
93         }
94       }
95     
96       String parsePrint(String js, CompilerOptions options) {
97         return new CodePrinter.Builder(parse(js)).setCompilerOptions(options).build();
98       }
99     
100       abstract static class CompilerOptionBuilder {
101         abstract void setOptions(CompilerOptions options);
102       }
103     
104       CompilerOptions newCompilerOptions(CompilerOptionBuilder builder) {
105         CompilerOptions options = new CompilerOptions();
106         options.setTrustedStrings(trustedStrings);
107         options.preserveTypeAnnotations = preserveTypeAnnotations;
108         options.setLanguageOut(languageMode);
109         builder.setOptions(options);
110         return options;
111       }
112     
113       String printNode(Node n) {
114         CompilerOptions options = new CompilerOptions();
115         options.setLineLengthThreshold(CompilerOptions.DEFAULT_LINE_LENGTH_THRESHOLD);
116         options.setLanguageOut(languageMode);
117         return new CodePrinter.Builder(n).setCompilerOptions(options).build();
118       }
119     
120       void assertPrintNode(String expectedJs, Node ast) {
121         assertEquals(expectedJs, printNode(ast));
122       }
123     
124       protected void assertPrint(String js, String expected) {
125         parse(expected); // validate the expected string is valid JS
126         assertEquals(expected,
127             parsePrint(js, newCompilerOptions(new CompilerOptionBuilder() {
128               @Override void setOptions(CompilerOptions options) {
129                 options.setPrettyPrint(false);
130                 options.setLineLengthThreshold(CompilerOptions.DEFAULT_LINE_LENGTH_THRESHOLD);
131               }
132             })));
133       }
134     
135       protected void assertPrintSame(String js) {
136         assertPrint(js, js);
137       }
138     }
