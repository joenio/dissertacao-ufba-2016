17     package com.google.javascript.jscomp.parsing.parser.util;
18     
19     /**
20      * An error reporter that throws exceptions for parse errors. Used
21      * for speculative parses.
22      */
23     public class LookaheadErrorReporter extends ErrorReporter {
24       public static class ParseException extends RuntimeException {}
25     
26       @Override
27       public void reportError(SourcePosition location, String message) {
28         throw new ParseException();
29       }
30     
31       @Override
32       public void reportWarning(SourcePosition location, String message) {
33         // Don't report a failed parse for a warning.
34       }
35     }
