16     package com.google.javascript.jscomp.lint;

18     import com.google.common.base.Predicate;
19     import com.google.common.base.Predicates;
20     import com.google.javascript.jscomp.AbstractCompiler;
21     import com.google.javascript.jscomp.CheckPathsBetweenNodes;
22     import com.google.javascript.jscomp.ControlFlowGraph;
23     import com.google.javascript.jscomp.DiagnosticType;
24     import com.google.javascript.jscomp.HotSwapCompilerPass;
25     import com.google.javascript.jscomp.NodeTraversal;
26     import com.google.javascript.jscomp.NodeUtil;
27     import com.google.javascript.jscomp.graph.DiGraph.DiGraphEdge;
28     import com.google.javascript.rhino.FunctionTypeI;
29     import com.google.javascript.rhino.JSDocInfo;
30     import com.google.javascript.rhino.Node;
31     import com.google.javascript.rhino.Token;
32     import com.google.javascript.rhino.TypeI;

34     /**
35      * Checks when a function is annotated as returning {SomeType} (nullable)
36      * but actually always returns {!SomeType}, i.e. never returns null.
37      *
38      */
39     public final class CheckNullableReturn implements HotSwapCompilerPass, NodeTraversal.Callback {
40       final AbstractCompiler compiler;
41     
42       public static final DiagnosticType NULLABLE_RETURN =
43           DiagnosticType.warning(
44               "JSC_NULLABLE_RETURN",
45               "This function''s return type is nullable, but it always returns a "
46               + "non-null value. Consider making the return type non-nullable.");
47     
48       public static final DiagnosticType NULLABLE_RETURN_WITH_NAME =
49           DiagnosticType.warning(
50               "JSC_NULLABLE_RETURN_WITH_NAME",
51               "The return type of the function \"{0}\" is nullable, but it always "
52               + "returns a non-null value. Consider making the return type "
53               + "non-nullable.");
54     
55       private static final Predicate<Node> NULLABLE_RETURN_PREDICATE =
56           new Predicate<Node>() {
57         @Override
58         public boolean apply(Node input) {
59           // Check for null because the control flow graph's implicit return node is
60           // represented by null, so this value might be input.
61           if (input == null || !input.isReturn()) {
62             return false;
63           }
64           Node returnValue = input.getFirstChild();
65           return returnValue != null && isNullable(returnValue);
66         }
67       };
68     
69       public CheckNullableReturn(AbstractCompiler compiler) {
70         this.compiler = compiler;
71       }
72     
73       public static boolean hasReturnDeclaredNullable(Node n) {
74         return n.isBlock() && n.hasChildren() && isReturnTypeNullable(n.getParent())
75             && !hasSingleThrow(n);
76       }
77     
78       @Override
79       public void visit(NodeTraversal t, Node n, Node parent) {
80         // Do the checks when 'n' is the block node and 'parent' is the function
81         // node, so that getControlFlowGraph will return the graph inside
82         // the function, rather than the graph of the enclosing scope.
83         if (hasReturnDeclaredNullable(n)
84             && !canReturnNull(t.getControlFlowGraph())) {
85           String fnName = NodeUtil.getNearestFunctionName(parent);
86           if (fnName != null && !fnName.isEmpty()) {
87             compiler.report(t.makeError(parent, NULLABLE_RETURN_WITH_NAME, fnName));
88           } else {
89             compiler.report(t.makeError(parent, NULLABLE_RETURN));
90           }
91         }
92       }
93     
94       /**
95        * @return whether the blockNode contains only a single "throw" child node.
96        */
97       private static boolean hasSingleThrow(Node blockNode) {
98         if (blockNode.getChildCount() == 1
99             && blockNode.getFirstChild().getType() == Token.THROW) {
100           // Functions consisting of a single "throw FOO" can be actually abstract,
101           // so do not check their return type nullability.
102           return true;
103         }
104     
105         return false;
106       }
107     
108       /**
109        * @return True if n is a function node which is explicitly annotated
110        * as returning a nullable type, other than {?}.
111        */
112       private static boolean isReturnTypeNullable(Node n) {
113         if (n == null || !n.isFunction()) {
114           return false;
115         }
116         FunctionTypeI functionType = n.getTypeI().toMaybeFunctionType();
117         if (functionType == null) {
118           // If the JSDoc declares a non-function type on a function node, we still shouldn't crash.
119           return false;
120         }
121         TypeI returnType = functionType.getReturnType();
122         if (returnType == null
123             || returnType.isUnknownType() || !returnType.isNullable()) {
124           return false;
125         }
126         JSDocInfo info = NodeUtil.getBestJSDocInfo(n);
127         return info != null && info.hasReturnType();
128       }
129     
130       /**
131        * @return True if the given ControlFlowGraph could return null.
132        */
133       public static boolean canReturnNull(ControlFlowGraph<Node> graph) {
134         CheckPathsBetweenNodes<Node, ControlFlowGraph.Branch> test = new CheckPathsBetweenNodes<>(graph,
135             graph.getEntry(), graph.getImplicitReturn(), NULLABLE_RETURN_PREDICATE,
136             Predicates.<DiGraphEdge<Node, ControlFlowGraph.Branch>>alwaysTrue());
137     
138         return test.somePathsSatisfyPredicate();
139       }
140     
141       /**
142        * @return True if the node represents a nullable value. Essentially, this
143        *     is just n.getTypeI().isNullable(), but for purposes of this pass,
144        *     the expression {@code x || null} is considered nullable even if
145        *     x is always truthy. This often happens with expressions like
146        *     {@code arr[i] || null}: The compiler doesn't know that arr[i] can
147        *     be undefined.
148        */
149       private static boolean isNullable(Node n) {
150         return n.getTypeI().isNullable()
151             || (n.isOr() && n.getLastChild().isNull());
152       }
153     
154       @Override
155       public boolean shouldTraverse(NodeTraversal nodeTraversal, Node n, Node parent) {
156         return true;
157       }
158     
159       @Override
160       public void process(Node externs, Node root) {
161         NodeTraversal.traverseEs6(compiler, root, this);
162       }
163     
164       @Override
165       public void hotSwapScript(Node scriptRoot, Node originalRoot) {
166         NodeTraversal.traverseEs6(compiler, originalRoot, this);
167       }
168     }
