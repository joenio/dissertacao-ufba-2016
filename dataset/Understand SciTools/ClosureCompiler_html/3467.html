34       /**
35        * Switches all "this.$.foo" to "this.$['foo']".
36        */
37       static void switchDollarSignPropsToBrackets(Node objLit, final AbstractCompiler compiler) {
38         Preconditions.checkState(objLit.isObjectLit());
39         for (Node keyNode : objLit.children()) {
40           Node value = keyNode.getFirstChild();
41           if (value != null && value.isFunction()) {
42             NodeUtil.visitPostOrder(
43                 value.getLastChild(),
44                 new NodeUtil.Visitor() {
45                   @Override
46                   public void visit(Node n) {
47                     if (n.isString() && n.getString().equals("$") && n.getParent().isGetProp()
48                         && n.getGrandparent().isGetProp()) {
49                       Node dollarChildProp = n.getGrandparent();
50                       dollarChildProp.setType(Token.GETELEM);
51                       compiler.reportCodeChange();
52                     }
53                   }
54                 },
55                 Predicates.<Node>alwaysTrue());
56           }
57         }
58       }
59     
60       /**
61        * Makes sure that the keys for listeners and hostAttributes blocks are quoted to avoid renaming.
62        */
63       static void quoteListenerAndHostAttributeKeys(Node objLit) {
64         Preconditions.checkState(objLit.isObjectLit());
65         for (Node keyNode : objLit.children()) {
66           if (keyNode.isComputedProp()) {
67             continue;
68           }
69           if (!keyNode.getString().equals("listeners")
70               && !keyNode.getString().equals("hostAttributes")) {
71             continue;
72           }
73           for (Node keyToQuote : keyNode.getFirstChild().children()) {
74             keyToQuote.setQuotedString();
75           }
76         }
77       }

79       /**
80        * Extracts a list of {@link MemberDefinition}s for the {@code properties} block of the given
81        * descriptor Object literal.
82        */
83       static ImmutableList<MemberDefinition> extractProperties(Node descriptor) {
84         Node properties = NodeUtil.getFirstPropMatchingKey(descriptor, "properties");
85         if (properties == null) {
86           return ImmutableList.of();
87         }
88     
89         ImmutableList.Builder<MemberDefinition> members = ImmutableList.builder();
90         for (Node keyNode : properties.children()) {
91           members.add(new MemberDefinition(NodeUtil.getBestJSDocInfo(keyNode), keyNode,
92               keyNode.getFirstChild()));
93         }
94         return members.build();
95       }

97       /**
98        * Gets the JSTypeExpression for a given property using its "type" key.
99        * @see https://github.com/Polymer/polymer/blob/0.8-preview/PRIMER.md#configuring-properties
100        */
101       static JSTypeExpression getTypeFromProperty(
102           MemberDefinition property, AbstractCompiler compiler) {
103         if (property.info != null && property.info.hasType()) {
104           return property.info.getType();
105         }
106     
107         String typeString;
108         if (property.value.isObjectLit()) {
109           Node typeValue = NodeUtil.getFirstPropMatchingKey(property.value, "type");
110           if (typeValue == null || !typeValue.isName()) {
111             compiler.report(JSError.make(property.name, PolymerPassErrors.POLYMER_INVALID_PROPERTY));
112             return null;
113           }
114           typeString = typeValue.getString();
115         } else if (property.value.isName()) {
116           typeString = property.value.getString();
117         } else {
118           typeString = "";
119         }
120     
121         Node typeNode;
122         switch (typeString) {
123           case "Boolean":
124           case "String":
125           case "Number":
126             typeNode = IR.string(typeString.toLowerCase());
127             break;
128           case "Array":
129           case "Function":
130           case "Object":
131           case "Date":
132             typeNode = new Node(Token.BANG, IR.string(typeString));
133             break;
134           default:
135             compiler.report(JSError.make(property.name, PolymerPassErrors.POLYMER_INVALID_PROPERTY));
136             return null;
137         }
138     
139         return new JSTypeExpression(typeNode, PolymerPass.VIRTUAL_FILE);
140       }

142       /**
143        * @return The PolymerElement type string for a class definition.
144        */
145       public static String getPolymerElementType(final PolymerClassDefinition cls) {
146         String nativeElementName = cls.nativeBaseElement == null ? ""
147             : CaseFormat.LOWER_HYPHEN.to(CaseFormat.UPPER_CAMEL, cls.nativeBaseElement);
148         return SimpleFormat.format("Polymer%sElement", nativeElementName);
149       }
