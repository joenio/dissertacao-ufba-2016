40     package com.google.javascript.rhino.jstype;
41     
42     import com.google.common.base.Function;
43     import com.google.common.base.Joiner;
44     import com.google.common.base.Preconditions;
45     import com.google.common.collect.ImmutableList;
46     import com.google.common.collect.Lists;
47     
48     /**
49      * An object type with declared template types, such as
50      * {@code Array<string>}.
51      *
52      */
53     public final class TemplatizedType extends ProxyObjectType {
54       private static final long serialVersionUID = 1L;
55     
56       final ImmutableList<JSType> templateTypes;
57       final TemplateTypeMapReplacer replacer;
58     
59       TemplatizedType(
60           JSTypeRegistry registry, ObjectType objectType,
61           ImmutableList<JSType> templateTypes) {
62         super(registry, objectType, objectType.getTemplateTypeMap().addValues(
63             templateTypes));
64     
65         // Cache which template keys were filled, and what JSTypes they were filled
66         // with.
67         ImmutableList<TemplateType> filledTemplateKeys =
68             objectType.getTemplateTypeMap().getUnfilledTemplateKeys();
69         ImmutableList.Builder<JSType> builder = ImmutableList.builder();
70         for (TemplateType filledTemplateKey : filledTemplateKeys) {
71           builder.add(getTemplateTypeMap().getResolvedTemplateType(filledTemplateKey));
72         }
73         this.templateTypes = builder.build();
74     
75         replacer = new TemplateTypeMapReplacer(registry, getTemplateTypeMap());
76       }
77     
78       @Override
79       String toStringHelper(final boolean forAnnotations) {
80         String typeString = super.toStringHelper(forAnnotations);
81     
82         if (!templateTypes.isEmpty()) {
83           typeString += "<"
84               + Joiner.on(",").join(Lists.transform(templateTypes, new Function<JSType, String>() {
85                 @Override
86                 public String apply(JSType type) {
87                   return type.toStringHelper(forAnnotations);
88                 }
89               })) + ">";
90         }
91     
92         return typeString;
93       }
94     
95       @Override
96       public <T> T visit(Visitor<T> visitor) {
97         return visitor.caseTemplatizedType(this);
98       }
99     
100       @Override <T> T visit(RelationshipVisitor<T> visitor, JSType that) {
101         return visitor.caseTemplatizedType(this, that);
102       }
103     
104       @Override
105       public TemplatizedType toMaybeTemplatizedType() {
106         return this;
107       }
108     
109       @Override
110       public ImmutableList<JSType> getTemplateTypes() {
111         return templateTypes;
112       }
113     
114       @Override
115       public JSType getPropertyType(String propertyName) {
116         JSType result = super.getPropertyType(propertyName);
117         return result == null ? null : result.visit(replacer);
118       }
119     
120       @Override
121       public boolean isSubtype(JSType that) {
122         return isSubtype(that, ImplCache.create(), SubtypingMode.NORMAL);
123       }
124     
125       @Override
126       protected boolean isSubtype(JSType that,
127           ImplCache implicitImplCache, SubtypingMode subtypingMode) {
128         return isSubtypeHelper(this, that, implicitImplCache, subtypingMode);
129       }
130     
131       boolean wrapsSameRawType(JSType that) {
132         return that.isTemplatizedType() && this.getReferencedTypeInternal()
133             .isEquivalentTo(
134                 that.toMaybeTemplatizedType().getReferencedTypeInternal());
135       }
136     
137       /**
138        * Computes the greatest subtype of two related templatized types.
139        * @return The greatest subtype.
140        */
141       JSType getGreatestSubtypeHelper(JSType rawThat) {
142         Preconditions.checkNotNull(rawThat);
143     
144         if (!wrapsSameRawType(rawThat)) {
145           if (!rawThat.isTemplatizedType()) {
146             if (this.isSubtype(rawThat)) {
147               return this;
148             } else if (rawThat.isSubtype(this)) {
149               return filterNoResolvedType(rawThat);
150             }
151           }
152           if (this.isObject() && rawThat.isObject()) {
153             return this.getNativeType(JSTypeNative.NO_OBJECT_TYPE);
154           }
155           return this.getNativeType(JSTypeNative.NO_TYPE);
156         }
157     
158         TemplatizedType that = rawThat.toMaybeTemplatizedType();
159         Preconditions.checkNotNull(that);
160     
161         if (getTemplateTypeMap().checkEquivalenceHelper(
162             that.getTemplateTypeMap(), EquivalenceMethod.INVARIANT, SubtypingMode.NORMAL)) {
163           return this;
164         }
165     
166         // For types that have the same raw type but different type parameters,
167         // we simply create a type has a "unknown" type parameter.  This is
168         // equivalent to the raw type.
169         return getReferencedObjTypeInternal();
170       }
171     
172       @Override
173       public TemplateTypeMap getTemplateTypeMap() {
174         return templateTypeMap;
175       }
176     
177       @Override
178       public boolean hasAnyTemplateTypesInternal() {
179         return templateTypeMap.hasAnyTemplateTypesInternal();
180       }
181     
182       /**
183        * @return The referenced ObjectType wrapped by this TemplatizedType.
184        */
185       public ObjectType getReferencedType() {
186         return getReferencedObjTypeInternal();
187       }
188     }
