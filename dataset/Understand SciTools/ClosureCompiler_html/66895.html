35     /**
36      * Rewrites static inheritance to explicitly copy inherited properties from superclass to
37      * subclass so that the typechecker knows the subclass has those properties.
38      *
39      * <p>For example, {@link Es6ToEs3Converter} will convert
40      *
41      * <pre>
42      *   class Foo { static f() {} }
43      *   class Bar extends Foo {}
44      * </pre>
45      * to
46      *
47      * <pre>
48      *   function Foo() {}
49      *   Foo.f = function() {};
50      *   function Bar() {}
51      *   $jscomp.inherits(Bar, Foo);
52      * </pre>
53      *
54      * and then this class will convert that to
55      *
56      * <pre>
57      *   function Foo() {}
58      *   Foo.f = function() {};
59      *   function Bar() {}
60      *   $jscomp.inherits(Bar, Foo);
61      *   Bar.f = Foo.f;
62      * </pre>
63      *
64      * Additionally, there are getter and setter fields which are transpiled from:
65      * <pre>
66      *   class Foo { static get prop() { return 1; } }
67      *   class Bar extends Foo {}
68      * </pre>
69      * to:
70      * <pre>
71      *   var Foo = function() {};
72      *   Foo.prop; // stub declaration so that the type checker knows about prop
73      *   Object.defineProperties(Foo, {prop:{get:function() { return 1; }}});
74      *
75      *   var Bar = function() {};
76      *   $jscomp.inherits(Bar, Foo);
77      * </pre>
78      *
79      * The stub declaration of Foo.prop needs to be duplicated for Bar so that the type checker knows
80      * that Bar also has this property.  (ES5 clases don't have class-side inheritance).
81      * <pre>
82      *   var Bar = function() {};
83      *   Bar.prop;
84      *   $jscomp.inherits(Bar, Foo);
85      * </pre>
86      *
87      * <p>In order to gather the type checker declarations, this pass gathers all GETPROPs on
88      * a class.  In order to determine which of these are the stub declarations it filters them based
89      * on names discovered in Object.defineProperties.  Unfortunately, we cannot simply gather the
90      * defined properties because they don't have the type information (JSDoc).  The type information
91      * is stored on the stub declarations so we must gather both to transpile correctly.
92      * <p>
93      * TODO(tdeegan): In the future the type information for getter/setter properties could be stored
94      * in the defineProperies functions.  It would reduce the complexity of this pass significantly.
95      *
96      * @author mattloring@google.com (Matthew Loring)
97      * @author tdeegan@google.com (Thomas Deegan)
98      */
99     public final class Es6ToEs3ClassSideInheritance implements HotSwapCompilerPass {
100     
101       static final DiagnosticType DUPLICATE_CLASS = DiagnosticType.error(
102           "DUPLICATE_CLASS",
103           "Multiple classes cannot share the same name.");
104     
105       private final Set<String> duplicateClassNames = new HashSet<>();
106     
107       private static class JavascriptClass {
108         // All static members to the class including get set properties.
109         private Set<Node> staticMembers = new LinkedHashSet<>();
110         // Collect all the static field accesses to the class.
111         private Set<Node> staticFieldAccess = new LinkedHashSet<>();
112         // Collect all get set properties as defined by Object.defineProperties(...)
113         private Set<String> definedProperties = new LinkedHashSet<>();
114       }
115     
116       private final AbstractCompiler compiler;
117     
118       private final LinkedHashMap<String, JavascriptClass> classByAlias = new LinkedHashMap<>();
119     
120       public Es6ToEs3ClassSideInheritance(AbstractCompiler compiler) {
121         this.compiler = compiler;
122       }
123     
124       @Override
125       public void process(Node externs, Node root) {
126         FindStaticMembers findStaticMembers = new FindStaticMembers();
127         NodeTraversal.traverseEs6(compiler, externs, findStaticMembers);
128         NodeTraversal.traverseEs6(compiler, root, findStaticMembers);
129         processInherits(findStaticMembers.inheritsCalls);
130       }
131     
132       @Override
133       public void hotSwapScript(Node scriptRoot, Node originalRoot) {
134         FindStaticMembers findStaticMembers = new FindStaticMembers();
135         NodeTraversal.traverseEs6(compiler, scriptRoot, findStaticMembers);
136         processInherits(findStaticMembers.inheritsCalls);
137       }
138     
139       private void processInherits(List<Node> inheritsCalls) {
140         for (Node inheritsCall : inheritsCalls) {
141           Node superclassNameNode = inheritsCall.getLastChild();
142           String superclassQname = superclassNameNode.getQualifiedName();
143           Node subclassNameNode = inheritsCall.getChildBefore(superclassNameNode);
144           String subclassQname = subclassNameNode.getQualifiedName();
145           JavascriptClass superClass = classByAlias.get(superclassQname);
146           JavascriptClass subClass = classByAlias.get(subclassQname);
147           if (duplicateClassNames.contains(superclassQname)) {
148             compiler.report(JSError.make(inheritsCall, DUPLICATE_CLASS));
149             return;
150           }
151           if (superClass == null || subClass == null) {
152             continue;
153           }
154           copyStaticMembers(superClass, subClass, inheritsCall);
155           copyDeclarations(superClass, subClass, inheritsCall);
156         }
157       }
158     
159       /**
160        * When static get/set properties are transpiled, in addition to the Object.defineProperties, they
161        * are declared with stub GETPROP declarations so that the type checker understands that these
162        * properties exist on the class.
163        * When subclassing, we also need to declare these properties on the subclass so that the type
164        * checker knows they exist.
165        */
166       private void copyDeclarations(
167           JavascriptClass superClass, JavascriptClass subClass, Node inheritsCall) {
168         for (Node staticGetProp : superClass.staticFieldAccess) {
169           Preconditions.checkState(staticGetProp.isGetProp());
170           String memberName = staticGetProp.getLastChild().getString();
171           // We only copy declarations that have corresponding Object.defineProperties
172           if (!superClass.definedProperties.contains(memberName)) {
173             continue;
174           }
175           // If the subclass already declares the property no need to redeclare it.
176           if (isOverriden(subClass, memberName)) {
177             continue;
178           }
179           Node subclassNameNode = inheritsCall.getSecondChild();
180           Node getprop = IR.getprop(subclassNameNode.cloneTree(), IR.string(memberName));
181           JSDocInfoBuilder info = JSDocInfoBuilder.maybeCopyFrom(staticGetProp.getJSDocInfo());
182           JSTypeExpression unknown = new JSTypeExpression(new Node(Token.QMARK), "<synthetic>");
183           info.recordType(unknown); // In case there wasn't a type specified on the base class.
184           info.addSuppression("visibility");
185           getprop.setJSDocInfo(info.build());
186     
187           Node declaration = IR.exprResult(getprop);
188           declaration.useSourceInfoIfMissingFromForTree(inheritsCall);
189           Node parent = inheritsCall.getParent();
190           parent.getParent().addChildBefore(declaration, parent);
191           compiler.reportCodeChange();
192     
193           // Copy over field access so that subclasses of this subclass can also make the declarations
194           if (!subClass.definedProperties.contains(memberName)) {
195             subClass.staticFieldAccess.add(getprop);
196             subClass.definedProperties.add(memberName);
197           }
198         }
199       }

201       private void copyStaticMembers(
202           JavascriptClass superClass, JavascriptClass subClass, Node inheritsCall) {
203         for (Node staticMember : superClass.staticMembers) {
204           Preconditions.checkState(staticMember.isAssign(), staticMember);
205           String memberName = staticMember.getFirstChild().getLastChild().getString();
206           if (superClass.definedProperties.contains(memberName)) {
207             continue;
208           }
209           if (isOverriden(subClass, memberName)) {
210             continue;
211           }
212           JSDocInfoBuilder info = JSDocInfoBuilder.maybeCopyFrom(staticMember.getJSDocInfo());
213           Node function = staticMember.getLastChild();
214           Node sourceInfoNode = function;
215           if (function.isFunction()) {
216             sourceInfoNode = function.getFirstChild();
217             Node params = NodeUtil.getFunctionParameters(function);
218             Preconditions.checkState(params.isParamList(), params);
219             for (Node param : params.children()) {
220               if (param.getJSDocInfo() != null) {
221                 String name = param.getString();
222                 info.recordParameter(name, param.getJSDocInfo().getType());
223               }
224             }
225           }
226     
227           Node subclassNameNode = inheritsCall.getSecondChild();
228           Node superclassNameNode = subclassNameNode.getNext();
229           Node assign =
230               IR.assign(
231                   IR.getprop(subclassNameNode.cloneTree(), IR.string(memberName)),
232                   IR.getprop(superclassNameNode.cloneTree(), IR.string(memberName)));
233           info.addSuppression("visibility");
234           assign.setJSDocInfo(info.build());
235           Node exprResult = IR.exprResult(assign);
236           exprResult.useSourceInfoIfMissingFromForTree(sourceInfoNode);
237           Node inheritsExpressionResult = inheritsCall.getParent();
238           inheritsExpressionResult.getParent().addChildAfter(exprResult, inheritsExpressionResult);
239           compiler.reportCodeChange();
240     
241           // Add the static member to the subclass so that subclasses also copy this member.
242           subClass.staticMembers.add(assign);
243         }
244       }
245     
246       private boolean isOverriden(JavascriptClass subClass, String memberName) {
247         for (Node subclassMember : subClass.staticMembers) {
248           Preconditions.checkState(subclassMember.isAssign(), subclassMember);
249           if (subclassMember.getFirstChild().getLastChild().getString().equals(memberName)) {
250             // This subclass overrides the static method, so there is no need to copy the
251             // method from the base class.
252             return true;
253           }
254         }
255         if (subClass.definedProperties.contains(memberName)) {
256           return true;
257         }
258         return false;
259       }
260     
261       private boolean isReferenceToClass(NodeTraversal t, Node n) {
262         String className = n.getQualifiedName();
263         if (!classByAlias.containsKey(className)) {
264           return false;
265         }
266     
267         if (!n.isName()) {
268           return true;
269         }
270     
271         Var var = t.getScope().getVar(className);
272         return var == null || !var.isLocal();
273       }
274     
275       private class FindStaticMembers extends AbstractPostOrderCallback {
276         private final List<Node> inheritsCalls = new LinkedList<>();
277     
278         @Override
279         public void visit(NodeTraversal t, Node n, Node parent) {
280           switch (n.getType()) {
281             case CALL:
282               if (n.getFirstChild().matchesQualifiedName(Es6ToEs3Converter.INHERITS)) {
283                 inheritsCalls.add(n);
284               }
285               if (NodeUtil.isObjectDefinePropertiesDefinition(n)) {
286                 visitDefinedPropertiesCall(t, n);
287               }
288               break;
289             case VAR:
290               visitVar(n);
291               break;
292             case ASSIGN:
293               visitAssign(t, n);
294               break;
295             case GETPROP:
296               if (parent.isExprResult()) {
297                 visitGetProp(t, n);
298               }
299               break;
300             case FUNCTION:
301               visitFunctionClassDef(n);
302               break;
303           }
304         }
305     
306         private void visitDefinedPropertiesCall(NodeTraversal t, Node definePropertiesCall) {
307           Node object = definePropertiesCall.getSecondChild();
308           if (isReferenceToClass(t, object)) {
309             String className = object.getQualifiedName();
310             JavascriptClass c = classByAlias.get(className);
311             for (Node prop : NodeUtil.getObjectDefinedPropertiesKeys(definePropertiesCall)) {
312               c.definedProperties.add(prop.getString());
313             }
314           }
315         }
316     
317         private void visitFunctionClassDef(Node n) {
318           JSDocInfo classInfo = NodeUtil.getBestJSDocInfo(n);
319           if (classInfo != null && classInfo.isConstructor()) {
320             String name = NodeUtil.getName(n);
321             if (classByAlias.containsKey(name)) {
322               duplicateClassNames.add(name);
323             } else {
324               classByAlias.put(name, new JavascriptClass());
325             }
326           }
327         }
328     
329         private void setAlias(String original, String alias) {
330           Preconditions.checkArgument(classByAlias.containsKey(original));
331           classByAlias.put(alias, classByAlias.get(original));
332         }
333     
334         private void visitGetProp(NodeTraversal t, Node n) {
335           Node classNode = n.getFirstChild();
336           if (isReferenceToClass(t, classNode)) {
337             classByAlias.get(classNode.getQualifiedName()).staticFieldAccess.add(n);
338           }
339         }
340     
341         private void visitAssign(NodeTraversal t, Node n) {
342           // Alias for classes. We assume that the alias appears after the class declaration.
343           String existingClassQname = n.getLastChild().getQualifiedName();
344           if (existingClassQname != null && classByAlias.containsKey(existingClassQname)) {
345             String alias = n.getFirstChild().getQualifiedName();
346             if (alias != null) {
347               setAlias(existingClassQname, alias);
348             }
349           } else if (n.getFirstChild().isGetProp()) {
350             Node getProp = n.getFirstChild();
351             Node classNode = getProp.getFirstChild();
352             if (isReferenceToClass(t, classNode)) {
353               classByAlias.get(classNode.getQualifiedName()).staticMembers.add(n);
354             }
355           }
356         }
357     
358         private void visitVar(Node n) {
359           Node child = n.getFirstChild();
360           if (!child.hasChildren()) {
361             return;
362           }
363           String maybeOriginalName = child.getFirstChild().getQualifiedName();
364           if (classByAlias.containsKey(maybeOriginalName)) {
365             String maybeAlias = child.getQualifiedName();
366             if (maybeAlias != null) {
367               setAlias(maybeOriginalName, maybeAlias);
368             }
369           }
370         }
371       }
