36     /**
37      * Inlines functions that are divided into two types: "direct call node
38      * replacement" (aka "direct") and as a block of statements (aka block).
39      * Function that can be inlined "directly" functions consist of a single
40      * return statement, everything else is must be inlined as a "block". These
41      * functions must meet these general requirements:
42      * - it is not recursive
43      * - the function does not contain another function -- these may be
44      *   intentional to to limit the scope of closures.
45      * - function is called only once OR the size of the inline function is smaller
46      *   than the call itself.
47      * - the function name is not referenced in any other manner
48      *
49      * "directly" inlined functions must meet these additional requirements:
50      * - consists of a single return statement
51      *
52      * @author johnlenz@google.com (John Lenz)
53      */
54     class InlineFunctions implements CompilerPass {

101       FunctionState getOrCreateFunctionState(String fnName) {
102         FunctionState fs = fns.get(fnName);
103         if (fs == null) {
104           fs = new FunctionState();
105           fns.put(fnName, fs);
106         }
107         return fs;
108       }

110       @Override
111       public void process(Node externs, Node root) {
112         Preconditions.checkState(compiler.getLifeCycleStage().isNormalized());
113     
114         NodeTraversal.traverseEs6(compiler, root, new FindCandidateFunctions());
115         if (fns.isEmpty()) {
116           return;  // Nothing left to do.
117         }
118         NodeTraversal.traverseEs6(compiler, root,
119            new FindCandidatesReferences(fns, anonFns));
120         trimCandidatesNotMeetingMinimumRequirements();
121         if (fns.isEmpty()) {
122           return;  // Nothing left to do.
123         }
124     
125         // Store the set of function names eligible for inlining and use this to
126         // prevent function names from being moved into temporaries during
127         // expression decomposition. If this movement were allowed it would prevent
128         // the Inline callback from finding the function calls.
129         //
130         // This pass already assumes these are constants, so this is safe for anyone
131         // using function inlining.
132         //
133         Set<String> fnNames = new HashSet<>(fns.keySet());
134         injector.setKnownConstants(fnNames);
135     
136         trimCandidatesUsingOnCost();
137         if (fns.isEmpty()) {
138           return;  // Nothing left to do.
139         }
140         resolveInlineConflicts();
141         decomposeExpressions();
142         NodeTraversal.traverseEs6(compiler, root,
143             new CallVisitor(fns, anonFns, new Inline(injector)));
144     
145         removeInlinedFunctions();
146       }

148       private static boolean isAlwaysInlinable(Node fn) {
149         Preconditions.checkArgument(fn.isFunction());
150         Node body = NodeUtil.getFunctionBody(fn);
151         int numOfStmsInBody = body.getChildCount();
152         return numOfStmsInBody == 0
153             || numOfStmsInBody == 1 && body.getFirstChild().isReturn();
154       }
155     
156       private boolean targetSizeAfterInlineExceedsLimit(
157           NodeTraversal t, FunctionState fs) {
158         Node containingFunction = t.getEnclosingFunction();
159         // Always inline at the top level,
160         // unless maybeAddFunction has marked fs as not inlinable.
161         if (containingFunction == null) {
162           return false;
163         }
164         Node inlinedFun = fs.getFn().getFunctionNode();
165         if (isAlwaysInlinable(inlinedFun)) {
166           return false;
167         }
168         int inlinedFunSize = NodeUtil.countAstSizeUpToLimit(
169             NodeUtil.getFunctionBody(inlinedFun), maxSizeAfterInlining);
170         int targetFunSize = NodeUtil.countAstSizeUpToLimit(
171             containingFunction, maxSizeAfterInlining);
172         return inlinedFunSize + targetFunSize > maxSizeAfterInlining;
173       }

175       /**
176        * Find functions that might be inlined.
177        */
178       private class FindCandidateFunctions implements Callback {
179         private int callsSeen = 0;
180     
181         @Override
182         public boolean shouldTraverse(
183             NodeTraversal nodeTraversal, Node n, Node parent) {
184           // Don't traverse into function bodies
185           // if we aren't inlining local functions.
186           return inlineLocalFunctions || nodeTraversal.inGlobalHoistScope();
187         }
188     
189         @Override
190         public void visit(NodeTraversal t, Node n, Node parent) {
191           if ((t.inGlobalHoistScope() && inlineGlobalFunctions)
192               || (!t.inGlobalHoistScope() && inlineLocalFunctions)) {
193             findNamedFunctions(t, n, parent);
194     
195             findFunctionExpressions(t, n);
196           }
197         }
198     
199         public void findNamedFunctions(NodeTraversal t, Node n, Node parent) {
200           if (!NodeUtil.isStatement(n)) {
201             // There aren't any interesting functions here.
202             return;
203           }
204     
205           switch (n.getType()) {
206               // Functions expressions in the form of:
207               //   var fooFn = function(x) { return ... }
208             case VAR:
209               Preconditions.checkState(n.hasOneChild(), n);
210               Node nameNode = n.getFirstChild();
211               if (nameNode.isName()
212                   && nameNode.hasChildren()
213                   && nameNode.getFirstChild().isFunction()) {
214                 maybeAddFunction(new FunctionVar(n), t.getModule());
215               }
216               break;
217     
218             // Named functions
219             // function Foo(x) { return ... }
220             case FUNCTION:
221               Preconditions.checkState(NodeUtil.isStatementBlock(parent)
222                   || parent.isLabel());
223               if (!NodeUtil.isFunctionExpression(n)) {
224                 Function fn = new NamedFunction(n);
225                 maybeAddFunction(fn, t.getModule());
226               }
227               break;
228           }
229         }
230     
231         /**
232          * Find function expressions that are called directly in the form of
233          *   (function(a,b,...){...})(a,b,...)
234          * or
235          *   (function(a,b,...){...}).call(this,a,b, ...)
236          */
237         public void findFunctionExpressions(NodeTraversal t, Node n) {
238           switch (n.getType()) {
239             // Functions expressions in the form of:
240             //   (function(){})();
241             case CALL:
242               Node fnNode = null;
243               if (n.getFirstChild().isFunction()) {
244                 fnNode = n.getFirstChild();
245               } else if (NodeUtil.isFunctionObjectCall(n)) {
246                 Node fnIdentifingNode = n.getFirstFirstChild();
247                 if (fnIdentifingNode.isFunction()) {
248                   fnNode = fnIdentifingNode;
249                 }
250               }
251     
252               // If a interesting function was discovered, add it.
253               if (fnNode != null) {
254                 Function fn = new FunctionExpression(fnNode, callsSeen++);
255                 maybeAddFunction(fn, t.getModule());
256                 anonFns.put(fnNode, fn.getName());
257               }
258               break;
259           }
260         }
261       }

263       /**
264        * Updates the FunctionState object for the given function. Checks if the
265        * given function matches the criteria for an inlinable function.
266        */
267       private void maybeAddFunction(Function fn, JSModule module) {
268         String name = fn.getName();
269         FunctionState fs = getOrCreateFunctionState(name);
270     
271         // TODO(johnlenz): Maybe "smarten" FunctionState by adding this logic
272         // to it?
273     
274         // If the function has multiple definitions, don't inline it.
275         if (fs.hasExistingFunctionDefinition()) {
276           fs.setInline(false);
277           return;
278         }
279         Node fnNode = fn.getFunctionNode();
280         if (enforceMaxSizeAfterInlining
281             && !isAlwaysInlinable(fnNode)
282             && maxSizeAfterInlining
283             <= NodeUtil.countAstSizeUpToLimit(fnNode, maxSizeAfterInlining)) {
284           fs.setInline(false);
285           return;
286         }
287         // verify the function hasn't already been marked as "don't inline"
288         if (fs.canInline()) {
289           // store it for use when inlining.
290           fs.setFn(fn);
291           if (FunctionInjector.isDirectCallNodeReplacementPossible(
292               fn.getFunctionNode())) {
293             fs.inlineDirectly(true);
294           }
295     
296           // verify the function meets all the requirements.
297           // TODO(johnlenz): Minimum requirement checks are about 5% of the
298           // run-time cost of this pass.
299           if (!isCandidateFunction(fn)) {
300             // It doesn't meet the requirements.
301             fs.setInline(false);
302           }
303     
304           // Set the module and gather names that need temporaries.
305           if (fs.canInline()) {
306             fs.setModule(module);
307     
308             Set<String> namesToAlias =
309                 FunctionArgumentInjector.findModifiedParameters(fnNode);
310             if (!namesToAlias.isEmpty()) {
311               fs.inlineDirectly(false);
312               fs.setNamesToAlias(namesToAlias);
313             }
314     
315             Node block = NodeUtil.getFunctionBody(fnNode);
316             if (NodeUtil.referencesThis(block)) {
317               fs.setReferencesThis(true);
318             }
319     
320             if (NodeUtil.containsFunction(block)) {
321               fs.setHasInnerFunctions(true);
322               // If there are inner functions, we can inline into global scope
323               // if there are no local vars or named functions.
324               // TODO(johnlenz): this can be improved by looking at the possible
325               // values for locals.  If there are simple values, or constants
326               // we could still inline.
327               if (!assumeMinimumCapture && hasLocalNames(fnNode)) {
328                 fs.setInline(false);
329               }
330             }
331           }
332     
333           // Check if block inlining is allowed.
334           if (fs.canInline() && !fs.canInlineDirectly() && !blockFunctionInliningEnabled) {
335             fs.setInline(false);
336           }
337         }
338       }

340       /**
341        * @param fnNode The function to inspect.
342        * @return Whether the function has parameters, var, or function declarations.
343        */
344       private static boolean hasLocalNames(Node fnNode) {
345         Node block = NodeUtil.getFunctionBody(fnNode);
346         return NodeUtil.getFunctionParameters(fnNode).hasChildren()
347             || NodeUtil.has(
348                  block,
349                  new NodeUtil.MatchDeclaration(),
350                  new NodeUtil.MatchShallowStatement());
351       }

353       /**
354        * Checks if the given function matches the criteria for an inlinable
355        * function.
356        */
357       private boolean isCandidateFunction(Function fn) {
358         // Don't inline exported functions.
359         String fnName = fn.getName();
360         if (compiler.getCodingConvention().isExported(fnName)) {
361           // TODO(johnlenz): Should we allow internal references to be inlined?
362           // An exported name can be replaced externally, any inlined instance
363           // would not reflect this change.
364           // To allow inlining we need to be able to distinguish between exports
365           // that are used in a read-only fashion and those that can be replaced
366           // by external definitions.
367           return false;
368         }
369     
370         // Don't inline this special function
371         if (compiler.getCodingConvention().isPropertyRenameFunction(fnName)) {
372           return false;
373         }
374     
375         Node fnNode = fn.getFunctionNode();
376         return injector.doesFunctionMeetMinimumRequirements(fnName, fnNode);
377       }

387       /**
388        * Visit call sites for functions in functionMap.
389        */
390       private static class CallVisitor extends AbstractPostOrderCallback {
391     
392         protected CallVisitorCallback callback;
393         private Map<String, FunctionState> functionMap;
394         private Map<Node, String> anonFunctionMap;
395     
396         CallVisitor(Map<String, FunctionState> fns,
397                     Map<Node, String> anonFns,
398                     CallVisitorCallback callback) {
399           this.functionMap = fns;
400           this.anonFunctionMap = anonFns;
401           this.callback = callback;
402         }
403     
404         @Override
405         public void visit(NodeTraversal t, Node n, Node parent) {
406           switch (n.getType()) {
407             // Function calls
408             case CALL:
409               Node child = n.getFirstChild();
410               String name = null;
411               // NOTE: The normalization pass insures that local names do not
412               // collide with global names.
413               if (child.isName()) {
414                 name = child.getString();
415               } else if (child.isFunction()) {
416                 name = anonFunctionMap.get(child);
417               } else if (NodeUtil.isFunctionObjectCall(n)) {
418                 Preconditions.checkState(NodeUtil.isGet(child));
419                 Node fnIdentifingNode = child.getFirstChild();
420                 if (fnIdentifingNode.isName()) {
421                   name = fnIdentifingNode.getString();
422                 } else if (fnIdentifingNode.isFunction()) {
423                   name = anonFunctionMap.get(fnIdentifingNode);
424                 }
425               }
426     
427               if (name != null) {
428                 FunctionState fs = functionMap.get(name);
429     
430                 // Only visit call-sites for functions that can be inlined.
431                 if (fs != null) {
432                   callback.visitCallSite(t, n, fs);
433                 }
434               }
435               break;
436           }
437         }
438       }

440       /**
441        * @return Whether the name is used in a way that might be a candidate
442        *   for inlining.
443        */
444       static boolean isCandidateUsage(Node name) {
445         Node parent = name.getParent();
446         Preconditions.checkState(name.isName());
447         if (parent.isVar() || parent.isFunction()) {
448           // This is a declaration.  Duplicate declarations are handle during
449           // function candidate gathering.
450           return true;
451         }
452     
453         if (parent.isCall() && parent.getFirstChild() == name) {
454           // This is a normal reference to the function.
455           return true;
456         }
457     
458         // Check for a ".call" to the named function:
459         //   CALL
460         //     GETPROP/GETELEM
461         //       NAME
462         //       STRING == "call"
463         //     This-Value
464         //     Function-parameter-1
465         //     ...
466         if (NodeUtil.isGet(parent)
467              && name == parent.getFirstChild()
468              && name.getNext().isString()
469              && name.getNext().getString().equals("call")) {
470           Node grandparent = name.getAncestor(2);
471           if (grandparent.isCall()
472               && grandparent.getFirstChild() == parent) {
473             // Yep, a ".call".
474             return true;
475           }
476         }
477         return false;
478       }

480       /**
481        * Find references to functions that are inlinable.
482        */
483       private class FindCandidatesReferences
484           extends CallVisitor
485           implements CallVisitorCallback {
486         FindCandidatesReferences(
487             Map<String, FunctionState> fns,
488             Map<Node, String> anonFns) {
489           super(fns, anonFns, null);
490           this.callback = this;
491         }
492     
493         @Override
494         public void visit(NodeTraversal t, Node n, Node parent) {
495           super.visit(t, n, parent);
496           if (n.isName()) {
497             checkNameUsage(n, parent);
498           }
499         }
500     
501         @Override
502         public void visitCallSite(
503             NodeTraversal t, Node callNode, FunctionState fs) {
504           maybeAddReference(t, fs, callNode, t.getModule());
505         }
506     
507         void maybeAddReference(NodeTraversal t, FunctionState fs,
508             Node callNode, JSModule module) {
509           if (!fs.canInline()) {
510             return;
511           }
512     
513           InliningMode mode = fs.canInlineDirectly()
514                ? InliningMode.DIRECT : InliningMode.BLOCK;
515           boolean referenceAdded = maybeAddReferenceUsingMode(t, fs, callNode, module, mode);
516           if (!referenceAdded &&
517               mode == InliningMode.DIRECT && blockFunctionInliningEnabled) {
518             // This reference can not be directly inlined, see if
519             // block replacement inlining is possible.
520             mode = InliningMode.BLOCK;
521             referenceAdded = maybeAddReferenceUsingMode(
522                 t, fs, callNode, module, mode);
523           }
524     
525           if (!referenceAdded) {
526             // Don't try to remove a function if we can't inline all
527             // the references.
528             fs.setRemove(false);
529           }
530         }
531     
532         private boolean maybeAddReferenceUsingMode(
533             NodeTraversal t, FunctionState fs, Node callNode,
534             JSModule module, InliningMode mode) {
535     
536           // If many functions are inlined into the same function F in the same
537           // inlining round, then the size of F may exceed the max size.
538           // This could be avoided if we bail later, during the inlining phase, eg,
539           // in Inline#visitCallSite. However, that is not safe, because at that
540           // point expression decomposition has already run, and we want to
541           // decompose expressions only for the calls that are actually inlined.
542           if (enforceMaxSizeAfterInlining
543               && targetSizeAfterInlineExceedsLimit(t, fs)) {
544             return false;
545           }
546     
547           Reference candidate = new Reference(callNode, t.getScope(), module, mode);
548           CanInlineResult result = injector.canInlineReferenceToFunction(
549               candidate, fs.getFn().getFunctionNode(),
550               fs.getNamesToAlias(), fs.getReferencesThis(),
551               fs.hasInnerFunctions());
552           if (result != CanInlineResult.NO) {
553             // Yeah!
554             candidate.setRequiresDecomposition(
555                 result == CanInlineResult.AFTER_PREPARATION);
556             fs.addReference(candidate);
557             return true;
558           }
559     
560           return false;
561         }

563         /**
564          * Find functions that can be inlined.
565          */
566         private void checkNameUsage(Node n, Node parent) {
567           Preconditions.checkState(n.isName(), n);
568     
569           if (isCandidateUsage(n)) {
570             return;
571           }
572     
573           // Other refs to a function name remove its candidacy for inlining
574           String name = n.getString();
575           FunctionState fs = fns.get(name);
576           if (fs == null) {
577             return;
578           }
579     
580           // Unlike normal call/new parameters, references passed to
581           // JSCompiler_ObjectPropertyString are not aliases of a value, but
582           // a reference to the name itself, as such the value of the name is
583           // unknown and can not be inlined.
584           if (parent.isNew()) {
585             Node target = parent.getFirstChild();
586             if (target.isName() && target.getString().equals(
587                 SimpleDefinitionFinder.EXTERN_OBJECT_PROPERTY_STRING)) {
588               // This method is going to be replaced so don't inline it anywhere.
589               fs.setInline(false);
590             }
591           }
592     
593           // If the name is being assigned to it can not be inlined.
594           if (parent.isAssign() && parent.getFirstChild() == n) {
595             // e.g. bar = something; <== we can't inline "bar"
596             // so mark the function as uninlinable.
597             // TODO(johnlenz): Should we just remove it from fns here?
598             fs.setInline(false);
599           } else {
600             // e.g. var fn = bar; <== we can't inline "bar"
601             // As this reference can't be inlined mark the function as
602             // unremovable.
603             fs.setRemove(false);
604           }
605         }
606       }

608       /**
609        * Inline functions at the call sites.
610        */
611       private static class Inline implements CallVisitorCallback {
612         private final FunctionInjector injector;
613     
614         Inline(FunctionInjector injector) {
615           this.injector = injector;
616         }
617     
618         @Override
619         public void visitCallSite(
620             NodeTraversal t, Node callNode, FunctionState fs) {
621           Preconditions.checkState(fs.hasExistingFunctionDefinition());
622           if (fs.canInline()) {
623             Reference ref = fs.getReference(callNode);
624     
625             // There are two cases ref can be null: if the call site was introduced
626             // because it was part of a function that was inlined during this pass
627             // or if the call site was trimmed from the list of references because
628             // the function couldn't be inlined at this location.
629             if (ref != null) {
630               inlineFunction(t, ref, fs);
631               // Keep track of references that have been inlined so that
632               // we can verify that none have been missed.
633               ref.inlined = true;
634             }
635           }
636         }
637     
638         /**
639          * Inline a function into the call site.
640          */
641         private void inlineFunction(
642             NodeTraversal t, Reference ref, FunctionState fs) {
643           Function fn = fs.getFn();
644           String fnName = fn.getName();
645           Node fnNode = fs.getSafeFnNode();
646     
647           Node newExpr = injector.inline(ref, fnName, fnNode);
648           if (!newExpr.isEquivalentTo(ref.callNode)) {
649             t.getCompiler().reportChangeToEnclosingScope(newExpr);
650           }
651           t.getCompiler().addToDebugLog("Inlined function: " + fn.getName());
652         }
653       }

655       /**
656        * Remove entries that aren't a valid inline candidates, from the list of
657        * encountered names.
658        */
659       private void trimCandidatesNotMeetingMinimumRequirements() {
660        Iterator<Entry<String, FunctionState>> i;
661        for (i = fns.entrySet().iterator(); i.hasNext();) {
662          FunctionState fs = i.next().getValue();
663          if (!fs.hasExistingFunctionDefinition() || !fs.canInline()) {
664            i.remove();
665          }
666        }
667       }
668     
669       /**
670        * Remove entries from the list of candidates that can't be inlined.
671        */
672       private void trimCandidatesUsingOnCost() {
673         Iterator<Entry<String, FunctionState>> i;
674         for (i = fns.entrySet().iterator(); i.hasNext();) {
675           FunctionState fs = i.next().getValue();
676           if (fs.hasReferences()) {
677             // Only inline function if it decreases the code size.
678             boolean lowersCost = minimizeCost(fs);
679             if (!lowersCost) {
680               // It shouldn't be inlined; remove it from the list.
681               i.remove();
682             }
683           } else if (!fs.canRemove()) {
684             // Don't bother tracking functions without references that can't be
685             // removed.
686             i.remove();
687           }
688         }
689       }

691       /**
692        * Determines if the function is worth inlining and potentially
693        * trims references that increase the cost.
694        * @return Whether inlining the references lowers the overall cost.
695        */
696       private boolean minimizeCost(FunctionState fs) {
697         if (!inliningLowersCost(fs)) {
698           // Try again without Block inlining references
699           if (fs.hasBlockInliningReferences()) {
700             fs.setRemove(false);
701             fs.removeBlockInliningReferences();
702             if (!fs.hasReferences() || !inliningLowersCost(fs)) {
703               return false;
704             }
705           } else {
706             return false;
707           }
708         }
709         return true;
710       }
711     
712       /**
713        * @return Whether inlining the function reduces code size.
714        */
715       private boolean inliningLowersCost(FunctionState fs) {
716         return injector.inliningLowersCost(
717             fs.getModule(),
718             fs.getFn().getFunctionNode(),
719             fs.getReferences(),
720             fs.getNamesToAlias(),
721             fs.canRemove(),
722             fs.getReferencesThis());
723       }

726       /**
727        * Size base inlining calculations are thrown off when a function that is
728        * being inlined also contains calls to functions that are slated for
729        * inlining.
730        *
731        * Specifically, a clone of the FUNCTION node tree is used when the function
732        * is inlined. Calls in this new tree are not included in the list of function
733        * references so they won't be inlined (which is what we want). Here we mark
734        * those functions as non-removable (as they will have new references in the
735        * cloned node trees).
736        *
737        * This prevents a function that would only be inlined because it is
738        * referenced once from being inlined into multiple call sites because
739        * the calling function has been inlined in multiple locations or the
740        * function being removed while there are still references.
741        */
742       private void resolveInlineConflicts() {
743         for (FunctionState fs : fns.values()) {
744           resolveInlineConflictsForFunction(fs);
745         }
746       }
747     
748       /**
749        * @see #resolveInlineConflicts
750        */
751       private void resolveInlineConflictsForFunction(FunctionState fs) {
752         // Functions that aren't referenced don't cause conflicts.
753         if (!fs.hasReferences() || !fs.canInline()) {
754           return;
755         }
756     
757         Node fnNode = fs.getFn().getFunctionNode();
758         Set<String> names = findCalledFunctions(fnNode);
759         if (!names.isEmpty()) {
760           // Prevent the removal of the referenced functions.
761           for (String name : names) {
762             FunctionState fsCalled = fns.get(name);
763             if (fsCalled != null && fsCalled.canRemove()) {
764               fsCalled.setRemove(false);
765               // For functions that can no longer be removed, check if they should
766               // still be inlined.
767               if (!minimizeCost(fsCalled)) {
768                 // It can't be inlined remove it from the list.
769                 fsCalled.setInline(false);
770               }
771             }
772           }
773     
774           // Make a copy of the Node, so it isn't changed by other inlines.
775           fs.setSafeFnNode(fs.getFn().getFunctionNode().cloneTree());
776         }
777       }

779       /**
780        * This functions that may be called directly.
781        */
782       private Set<String> findCalledFunctions(Node node) {
783         Set<String> changed = new HashSet<>();
784         findCalledFunctions(NodeUtil.getFunctionBody(node), changed);
785         return changed;
786       }
787     
788       /**
789        * @see #findCalledFunctions(Node)
790        */
791       private static void findCalledFunctions(
792           Node node, Set<String> changed) {
793         Preconditions.checkArgument(changed != null);
794         // For each referenced function, add a new reference
795         if (node.isName() && isCandidateUsage(node)) {
796           changed.add(node.getString());
797         }
798     
799         for (Node c = node.getFirstChild(); c != null; c = c.getNext()) {
800           findCalledFunctions(c, changed);
801         }
802       }

804       /**
805        * For any call-site that needs it, prepare the call-site for inlining
806        * by rewriting the containing expression.
807        */
808       private void decomposeExpressions() {
809         for (FunctionState fs : fns.values()) {
810           if (fs.canInline()) {
811             for (Reference ref : fs.getReferences()) {
812               if (ref.requiresDecomposition) {
813                 injector.maybePrepareCall(ref);
814               }
815             }
816           }
817         }
818       }
819     
820       /**
821        * Removed inlined functions that no longer have any references.
822        */
823       void removeInlinedFunctions() {
824         for (FunctionState fs : fns.values()) {
825           if (fs.canRemove()) {
826             Function fn = fs.getFn();
827             Preconditions.checkState(fs.canInline());
828             Preconditions.checkState(fn != null);
829             verifyAllReferencesInlined(fs);
830             fn.remove();
831           }
832         }
833       }
834     
835       /**
836        * Sanity check to verify, that expression rewriting didn't
837        * make a call inaccessible.
838        */
839       void verifyAllReferencesInlined(FunctionState fs) {
840         for (Reference ref : fs.getReferences()) {
841           if (!ref.inlined) {
842             throw new IllegalStateException("Call site missed.\n call: "
843                 + ref.callNode.toStringTree() + "\n parent:  "
844                 + ref.callNode.getParent().toStringTree());
845           }
846         }
847       }

849       /**
850        * Use to track the decisions that have been made about a function.
851        */
852       private static class FunctionState {
853         private Function fn = null;
854         private Node safeFnNode = null;
855         private boolean inline = true;
856         private boolean remove = true;
857         private boolean inlineDirectly = false;
858         private boolean referencesThis = false;
859         private boolean hasInnerFunctions = false;
860         private Map<Node, Reference> references = null;
861         private JSModule module = null;
862         private Set<String> namesToAlias = null;
863     
864         boolean hasExistingFunctionDefinition() {
865           return (fn != null);
866         }
867     
868         public void setReferencesThis(boolean referencesThis) {
869           this.referencesThis = referencesThis;
870         }
871     
872         public boolean getReferencesThis() {
873           return this.referencesThis;
874         }
875     
876         public void setHasInnerFunctions(boolean hasInnerFunctions) {
877           this.hasInnerFunctions = hasInnerFunctions;
878         }
879     
880     
881         public boolean hasInnerFunctions() {
882           return hasInnerFunctions;
883         }
884     
885         void removeBlockInliningReferences() {
886           Iterator<Entry<Node, Reference>> i;
887           for (i = getReferencesInternal().entrySet().iterator(); i.hasNext();) {
888             Entry<Node, Reference> entry = i.next();
889             if (entry.getValue().mode == InliningMode.BLOCK) {
890               i.remove();
891             }
892           }
893         }
894     
895         public boolean hasBlockInliningReferences() {
896           for (Reference r : getReferencesInternal().values()) {
897             if (r.mode == InliningMode.BLOCK) {
898               return true;
899             }
900           }
901           return false;
902         }
903     
904         public Function getFn() {
905           return fn;
906         }
907     
908         public void setFn(Function fn) {
909           Preconditions.checkState(this.fn == null);
910           this.fn = fn;
911         }
912     
913         public Node getSafeFnNode() {
914           return (safeFnNode != null) ? safeFnNode : fn.getFunctionNode();
915         }
916     
917         public void setSafeFnNode(Node safeFnNode) {
918           this.safeFnNode = safeFnNode;
919         }
920     
921         public boolean canInline() {
922           return inline;
923         }
924     
925         public void setInline(boolean inline) {
926           this.inline = inline;
927           if (!inline) {
928             // No need to keep references to function that can't be inlined.
929             references = null;
930             // Don't remove functions that we aren't inlining.
931             remove = false;
932           }
933         }
934     
935         public boolean canRemove() {
936           return remove;
937         }
938     
939         public void setRemove(boolean remove) {
940           this.remove = remove;
941         }
942     
943         public boolean canInlineDirectly() {
944           return inlineDirectly;
945         }
946     
947         public void inlineDirectly(boolean directReplacement) {
948           this.inlineDirectly = directReplacement;
949         }

951         public boolean hasReferences() {
952           return (references != null && !references.isEmpty());
953         }

956           if (references == null) {
957             return Collections.emptyMap();

959           return references;
960         }

962         public void addReference(Reference ref) {
963           if (references == null) {
964             references = new LinkedHashMap<>();
965           }
966           references.put(ref.callNode, ref);
967         }

975         }

977         public Set<String> getNamesToAlias() {
978           if (namesToAlias == null) {
979             return Collections.emptySet();
980           }
981           return Collections.unmodifiableSet(namesToAlias);
982         }
983     
984         public void setNamesToAlias(Set<String> names) {
985           namesToAlias = names;
986         }

990         }

994         }
