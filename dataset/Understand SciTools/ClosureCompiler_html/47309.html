19     /** Unit tests for {@link ConvertToTypedInterface}. */
20     public final class ConvertToTypedInterfaceTest extends Es6CompilerTestCase {
21     
22       @Override
23       protected CompilerPass getProcessor(final Compiler compiler) {
24         return new ConvertToTypedInterface(compiler);

27       public void testInferAnnotatedTypeFromInferredType() {
28         enableTypeCheck();
29     
30         test("/** @const */ var x = 5;", "/** @const {number} */ var x;");
31     
32         test(
33             "/** @constructor */ function Foo() { /** @const */ this.x = 5; }",
34             "/** @constructor */ function Foo() {} \n /** @const {number} */ Foo.prototype.x;");
35       }
36     
37       public void testRemoveUselessStatements() {
38         test("34", "");
39         test("'str'", "");
40         test("({x:4})", "");
41         test("debugger;", "");
42         test("throw 'error';", "");
43         test("label: debugger;", "");
44       }
45     
46       public void testRemoveUnnecessaryBodies() {
47         test("function f(x,y) { /** @type {number} */ z = x + y; return z; }", "function f(x,y) {}");
48     
49         test(
50             "/** @return {number} */ function f(/** number */ x, /** number */ y) { return x + y; }",
51             "/** @return {number} */ function f(/** number */ x, /** number */ y) {}");
52     
53         testEs6(
54             "class Foo { method(/** string */ s) { return s.split(','); } }",
55             "class Foo { method(/** string */ s) {} }");
56       }

58       public void testRemoveCalls() {
59         test("alert('hello'); window.clearTimeout();", "");
60     
61         testSame("goog.provide('a.b.c');");
62     
63         testSame("goog.provide('a.b.c'); goog.require('x.y.z');");
64       }

66       public void testEnums() {
67         test(
68             "/** @enum {number} */ var E = { A: 1, B: 2, C: 3};",
69             "/** @enum {number} */ var E = { A: 0, B: 0, C: 0};");
70     
71         test(
72             "/** @enum {number} */ var E = { A: foo(), B: bar(), C: baz()};",
73             "/** @enum {number} */ var E = { A: 0, B: 0, C: 0};");
74     
75         // NOTE: This pattern typechecks when found in externs, but not for code.
76         // Since the goal of this pass is intended to be used as externs, this is acceptable.
77         test(
78             "/** @enum {string} */ var E = { A: 'hello', B: 'world'};",
79             "/** @enum {string} */ var E = { A: 0, B: 0};");
80     
81         test(
82             "/** @enum {Object} */ var E = { A: {b: 'c'}, D: {e: 'f'} };",
83             "/** @enum {Object} */ var E = { A: 0, D: 0};");
84     
85         test(
86             "var x = 7; /** @enum {number} */ var E = { A: x };",
87             "/** @type {*} */ var x; /** @enum {number} */ var E = { A: 0 };");
88       }

90       public void testTryCatch() {
91         test(
92             "try { /** @type {number} */ var n = foo(); } catch (e) { console.log(e); }",
93             "{ /** @type {number} */ var n; }");
94       }

96       public void testConstructorBodyWithThisDeclaration() {
97         test(
98             "/** @constructor */ function Foo() { /** @type {number} */ this.num = 5;}",
99             "/** @constructor */ function Foo() {} /** @type {number} */ Foo.prototype.num;");
100     
101         test(
102             "/** @constructor */ function Foo(b) { if (b) { /** @type {number} */ this.num = 5; } }",
103             "/** @constructor */ function Foo(b) {} /** @type {number} */ Foo.prototype.num;");
104     
105         testEs6(
106             "/** @constructor */ let Foo = function() { /** @type {number} */ this.num = 5;}",
107             "/** @constructor */ let Foo = function() {}; /** @type {number} */ Foo.prototype.num;");
108     
109         testEs6(
110             "class Foo { constructor() { /** @type {number} */ this.num = 5;} }",
111             "class Foo { constructor() {} } /** @type {number} */ Foo.prototype.num;");
112       }

139       public void testIfs() {
140         test(
141             "if (true) { var /** number */ x = 5; }",
142             "{/** @type {number} */ var x;}");
143     
144         test(
145             "if (true) { var /** number */ x = 5; } else { var /** string */ y = 'str'; }",
146             "{/** @type {number} */ var x;} {/** @type {string} */ var  y; }");
147     
148         test(
149             "if (true) { if (false) { var /** number */ x = 5; } }",
150             "{{/** @type {number} */ var x;}}");
151     
152         test(
153             "if (true) {} else { if (false) {} else { var /** number */ x = 5; } }",
154             "{}{{}{/** @type {number} */ var x;}}");
155       }

157       public void testLoops() {
158         test("while (true) { foo(); break; }", "{}");
159     
160         test("for (var i = 0; i < 10; i++) { var field = 88; }",
161             "/** @type {*} */ var i; {/** @type {*} */ var field;}");
162     
163         test(
164             "while (i++ < 10) { var /** number */ field = i; }",
165             "{ /** @type {number } */ var field; }");
166     
167         test(
168             "do { var /** number */ field = i; } while (i++ < 10);",
169             "{ /** @type {number } */ var field; }");
170     
171         test(
172             "for (var /** number */ i = 0; i < 10; i++) { var /** number */ field = i; }",
173             "/** @type {number} */ var i; { /** @type {number } */ var field; }");
174     
175         test(
176             "for (i = 0; i < 10; i++) { var /** number */ field = i; }",
177             "{ /** @type {number } */ var field; }");
178     
179         test(
180             "for (var i = 0; i < 10; i++) { var /** number */ field = i; }",
181             "/** @type {*} */ var i; { /** @type {number } */ var field; }");
182       }
