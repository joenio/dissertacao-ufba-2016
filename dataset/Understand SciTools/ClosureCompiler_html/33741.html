730       /**
731        * Returns the root script node produced from the compiled JS input.
732        */
733       private Node compileToScriptRoot(Compiler compiler) {
734         Node root = compiler.getRoot();
735         // The last child of the compiler root is a Block node, and the first child
736         // of that is the Script node.
737         return root.getLastChild().getFirstChild();
738       }

740       private Compiler getCompiler(String jsInput) {
741         Compiler compiler = new Compiler();
742         CompilerOptions options = RefactoringDriver.getCompilerOptions();
743         compiler.init(
744             ImmutableList.<SourceFile>of(), // Externs
745             ImmutableList.of(SourceFile.fromCode("test", jsInput)),
746             options);
747         compiler.parse();
748         return compiler;
749       }
