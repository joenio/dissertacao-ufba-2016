18     import static com.google.common.truth.Truth.assertThat;

20     import com.google.common.collect.ImmutableMap;

22     import com.google.javascript.jscomp.CommandLineRunner;
23     import com.google.javascript.jscomp.CompilerOptions;
24     import com.google.javascript.refactoring.ApplySuggestedFixes;
25     import com.google.javascript.refactoring.RefactoringDriver;
26     import com.google.javascript.refactoring.RefasterJsScanner;
27     import com.google.javascript.refactoring.SuggestedFix;
28     
29     import java.io.File;
30     import java.io.IOException;
31     import java.nio.charset.StandardCharsets;
32     import java.util.List;

34     /**
35      * Utilities for testing RefasterJs templates.
36      */
37     public final class RefasterJsTestUtils {
38     
39       /**
40        * Performs refactoring using a RefasterJs template and asserts that result is as expected.
41        *
42        * @param refasterJsTemplate path of the file or resource containing the RefasterJs
43        *    template to apply
44        * @param testDataPathPrefix path prefix of the directory from which input and
45        *    expected-output file will be read
46        * @param originalFile file name of the JavaScript source file to apply the refaster template to
47        * @param additionalSourceFiles list of additional source files to provide to the compiler
48        *    (e.g. dependencies)
49        * @param expectedFile the expected result of applying the specified template to
50        *    {@code originalFile}
51        * @throws IOException
52        */
53       public static void assertFileRefactoring(
54           String refasterJsTemplate,
55           String testDataPathPrefix,
56           String originalFile,
57           List<String> additionalSourceFiles,
58           String expectedFile)
59           throws IOException {
60         RefasterJsScanner scanner = new RefasterJsScanner();
61         scanner.loadRefasterJsTemplate(refasterJsTemplate);
62     
63         final String originalFilePath = testDataPathPrefix + File.separator + originalFile;
64         final String expectedFilePath = testDataPathPrefix + File.separator + expectedFile;
65     
66         RefactoringDriver.Builder driverBuilder =
67             new RefactoringDriver.Builder(scanner)
68                 .addExterns(CommandLineRunner.getBuiltinExterns(CompilerOptions.Environment.BROWSER));
69     
70         for (String additionalSource : additionalSourceFiles) {
71           driverBuilder.addInputsFromFile(testDataPathPrefix + File.separator + additionalSource);
72         }
73         RefactoringDriver driver = driverBuilder.addInputsFromFile(originalFilePath).build();
74     
75         List<SuggestedFix> fixes = driver.drive();
76         assertThat(driver.getCompiler().getErrors()).isEmpty();
77         assertThat(driver.getCompiler().getWarnings()).isEmpty();
78     
79         String newCode =
80             ApplySuggestedFixes.applySuggestedFixesToCode(
81                     fixes, ImmutableMap.of(originalFilePath, slurpFile(originalFilePath)))
82                 .get(originalFilePath);
83         assertThat(newCode).isEqualTo(slurpFile(expectedFilePath));
84       }
85     
86       private static String slurpFile(String originalFile) throws IOException {
87         return Files.toString(new File(originalFile), StandardCharsets.UTF_8);
88       }
89     }
