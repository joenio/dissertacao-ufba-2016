17     package com.google.javascript.jscomp;
18     
19     import com.google.common.base.Preconditions;
20     import com.google.common.collect.ImmutableList;
21     import com.google.common.collect.ImmutableMap;
22     import com.google.javascript.jscomp.deps.DependencyInfo;
23     import com.google.javascript.jscomp.deps.Es6SortedDependencies;
24     
25     import java.io.Serializable;
26     import java.util.ArrayDeque;
27     import java.util.ArrayList;
28     import java.util.Collections;
29     import java.util.HashSet;
30     import java.util.Iterator;
31     import java.util.List;
32     import java.util.Set;
33     
34     /**
35      * A JavaScript module has a unique name, consists of a list of compiler inputs,
36      * and can depend on other modules.
37      *
38      */
39     public final class JSModule implements DependencyInfo, Serializable {
40       private static final long serialVersionUID = 1;
41     
42       /** Module name */
43       private final String name;
44     
45       /** Source code inputs */
46       private final List<CompilerInput> inputs = new ArrayList<>();
47     
48       /** Modules that this module depends on */
49       private final List<JSModule> deps = new ArrayList<>();
50     
51       private int depth;
52       /**
53        * Creates an instance.
54        *
55        * @param name A unique name for the module
56        */
57       public JSModule(String name) {
58         this.name = name;
59         this.depth = -1;
60       }
61     
62       /** Gets the module name. */
63       @Override
64       public String getName() {
65         return name;
66       }
67     
68       @Override
69       public List<String> getProvides() {
70         return ImmutableList.of(name);
71       }
72     
73       @Override
74       public List<String> getRequires() {
75         ImmutableList.Builder<String> builder = ImmutableList.builder();
76         for (JSModule m : deps) {
77           builder.add(m.getName());
78         }
79         return builder.build();
80       }
81     
82       @Override
83       public String getPathRelativeToClosureBase() {
84         throw new UnsupportedOperationException();
85       }
86     
87       @Override
88       public ImmutableMap<String, String> getLoadFlags() {
89         throw new UnsupportedOperationException();
90       }
91     
92       @Override
93       public boolean isModule() {
94         // NOTE: The meaning of "module" has changed over time.  A "JsModule" is
95         // a collection of inputs that are loaded together. A "module" file,
96         // is a CommonJs module, ES6 module, goog.module or other file whose
97         // top level symbols are not in global scope.
98         throw new UnsupportedOperationException();
99       }
100     
101       /** Adds a source file input to this module. */
102       public void add(SourceFile file) {
103         add(new CompilerInput(file));
104       }
105     
106       /** Adds a source code input to this module. */
107       public void add(CompilerInput input) {
108         inputs.add(input);
109         input.setModule(this);
110       }
111     
112       /**
113        * Adds a source code input to this module. Call only if the input might
114        * already be associated with a module. Otherwise, use
115        * add(CompilerInput input).
116        */
117       void addAndOverrideModule(CompilerInput input) {
118         inputs.add(input);
119         input.overrideModule(this);
120       }
121     
122       /** Adds a source code input to this module directly after other. */
123       public void addAfter(CompilerInput input, CompilerInput other) {
124         Preconditions.checkState(inputs.contains(other));
125         inputs.add(inputs.indexOf(other), input);
126         input.setModule(this);
127       }
128     
129       /** Adds a dependency on another module. */
130       public void addDependency(JSModule dep) {
131         Preconditions.checkNotNull(dep);
132         Preconditions.checkState(dep != this, "Cannot add dependency on self", this);
133         deps.add(dep);
134       }
135     
136       /** Removes an input from this module. */
137       public void remove(CompilerInput input) {
138         input.setModule(null);
139         inputs.remove(input);
140       }
141     
142       /** Removes all of the inputs from this module. */
143       public void removeAll() {
144         for (CompilerInput input : inputs) {
145           input.setModule(null);
146         }
147         inputs.clear();
148       }
149     
150       /**
151        * Gets the list of modules that this module depends on.
152        *
153        * @return A list that may be empty but not null
154        */
155       public List<JSModule> getDependencies() {
156         return deps;
157       }
158     
159       /**
160        * Gets the names of the modules that this module depends on,
161        * sorted alphabetically.
162        */
163       List<String> getSortedDependencyNames() {
164         List<String> names = new ArrayList<>();
165         for (JSModule module : getDependencies()) {
166           names.add(module.getName());
167         }
168         Collections.sort(names);
169         return names;
170       }
171     
172       /**
173        * Returns the transitive closure of dependencies starting from the
174        * dependencies of this module.
175        */
176       public Set<JSModule> getAllDependencies() {
177         Set<JSModule> allDeps = new HashSet<>(deps);
178         ArrayDeque<JSModule> stack = new ArrayDeque<>(deps);
179     
180         while (!stack.isEmpty()) {
181           JSModule module = stack.pop();
182           for (JSModule dep : module.getDependencies()) {
183             if (allDeps.add(dep)) {
184               stack.push(dep);
185             }
186           }
187         }
188         return allDeps;
189       }
190     
191       /** Returns this module and all of its dependencies in one list. */
192       public Set<JSModule> getThisAndAllDependencies() {
193         Set<JSModule> deps = getAllDependencies();
194         deps.add(this);
195         return deps;
196       }
197     
198       /**
199        * Gets this module's list of source code inputs.
200        *
201        * @return A list that may be empty but not null
202        */
203       public List<CompilerInput> getInputs() {
204         return inputs;
205       }
206     
207       /** Returns the input with the given name or null if none. */
208       public CompilerInput getByName(String name) {
209         for (CompilerInput input : inputs) {
210           if (name.equals(input.getName())) {
211             return input;
212           }
213         }
214         return null;
215       }
216     
217       /**
218        * Removes any input with the given name. Returns whether any were removed.
219        */
220       public boolean removeByName(String name) {
221         boolean found = false;
222         Iterator<CompilerInput> iter = inputs.iterator();
223         while (iter.hasNext()) {
224           CompilerInput file = iter.next();
225           if (name.equals(file.getName())) {
226             iter.remove();
227             file.setModule(null);
228             found = true;
229           }
230         }
231         return found;
232       }
233     
234       /** Returns the module name (primarily for debugging). */
235       @Override
236       public String toString() {
237         return name;
238       }
239     
240       /**
241        * Removes any references to nodes of the AST.  This method is needed to
242        * allow the ASTs to be garbage collected if the modules are kept around.
243        */
244       public void clearAsts() {
245         for (CompilerInput input : inputs) {
246           input.clearAst();
247         }
248       }
249     
250       /**
251        * Puts the JS files into a topologically sorted order by their dependencies.
252        */
253       public void sortInputsByDeps(AbstractCompiler compiler) {
254         // Set the compiler, so that we can parse requires/provides and report
255         // errors properly.
256         for (CompilerInput input : inputs) {
257           input.setCompiler(compiler);
258         }
259     
260         // Sort the JSModule in this order.
261         List<CompilerInput> sortedList = new Es6SortedDependencies<>(inputs).getSortedList();
262         inputs.clear();
263         inputs.addAll(sortedList);
264       }
265     
266       /**
267        * @param dep the depth to set
268        */
269       public void setDepth(int dep) {
270         this.depth = dep;
271       }
272     
273       /**
274        * @return the depth
275        */
276       public int getDepth() {
277         return depth;
278       }
279     }
