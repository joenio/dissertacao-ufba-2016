36     /**
37      * Simple name-based definition gatherer that implements
38      * {@link DefinitionProvider}.
39      *
40      * It treats all variable writes as happening in the global scope and
41      * treats all objects as capable of having the same set of properties.
42      * The current implementation only handles definitions whose right
43      * hand side is an immutable value or function expression.  All
44      * complex definitions are treated as unknowns.
45      *
46      */
47     public class SimpleDefinitionFinder implements CompilerPass, DefinitionProvider {
48       public static final String EXTERN_OBJECT_PROPERTY_STRING =
49           "JSCompiler_ObjectPropertyString";
50     
51       private final AbstractCompiler compiler;
52       private final Map<Node, DefinitionSite> definitionSiteMap;
53       private final Multimap<String, Definition> nameDefinitionMultimap;
54       private final Multimap<String, UseSite> nameUseSiteMultimap;
55       // The same defFinder can be used by multiple passes, but its process method
56       // must be run only once
57       private boolean hasProcessBeenRun = false;
58     
59       public SimpleDefinitionFinder(AbstractCompiler compiler) {
60         this.compiler = compiler;
61         this.definitionSiteMap = new LinkedHashMap<>();
62         this.nameDefinitionMultimap = LinkedHashMultimap.create();
63         this.nameUseSiteMultimap = LinkedHashMultimap.create();
64       }
65     
66       /**
67        * Returns the collection of definition sites found during traversal.
68        *
69        * @return definition site collection.
70        */
71       public Collection<DefinitionSite> getDefinitionSites() {
72         return definitionSiteMap.values();
73       }
74     
75       private DefinitionSite getDefinitionAt(Node node) {
76         return definitionSiteMap.get(node);
77       }
78     
79       DefinitionSite getDefinitionForFunction(Node function) {
80         Preconditions.checkState(function.isFunction());
81         return getDefinitionAt(getNameNodeFromFunctionNode(function));
82       }
83     
84       @Override
85       public Collection<Definition> getDefinitionsReferencedAt(Node useSite) {
86         if (definitionSiteMap.containsKey(useSite)) {
87           return null;
88         }
89     
90         if (useSite.isGetProp()) {
91           String propName = useSite.getLastChild().getString();
92           if (propName.equals("apply") || propName.equals("call")) {
93             useSite = useSite.getFirstChild();
94           }
95         }
96     
97         String name = getSimplifiedName(useSite);
98         if (name != null) {
99           Collection<Definition> defs = nameDefinitionMultimap.get(name);
100           if (!defs.isEmpty()) {
101             return defs;
102           } else {
103             return null;
104           }
105         } else {
106           return null;
107         }
108       }
109     
110       @Override
111       public void process(Node externs, Node source) {
112         if (this.hasProcessBeenRun) {
113           return;
114         }
115         this.hasProcessBeenRun = true;
116         NodeTraversal.traverseEs6(
117             compiler, externs, new DefinitionGatheringCallback(true));
118         NodeTraversal.traverseEs6(
119             compiler, source, new DefinitionGatheringCallback(false));
120         NodeTraversal.traverseEs6(
121             compiler, source, new UseSiteGatheringCallback());
122       }

124       /**
125        * Returns a collection of use sites that may refer to provided
126        * definition.  Returns an empty collection if the definition is not
127        * used anywhere.
128        *
129        * @param definition Definition of interest.
130        * @return use site collection.
131        */
132       Collection<UseSite> getUseSites(Definition definition) {
133         String name = getSimplifiedName(definition.getLValue());
134         return nameUseSiteMultimap.get(name);
135       }

137       /**
138        * Extract a name from a node.  In the case of GETPROP nodes,
139        * replace the namespace or object expression with "this" for
140        * simplicity and correctness at the expense of inefficiencies due
141        * to higher chances of name collisions.
142        *
143        * TODO(user) revisit.  it would be helpful to at least use fully
144        * qualified names in the case of namespaces.  Might not matter as
145        * much if this pass runs after "collapsing properties".
146        */
147       private static String getSimplifiedName(Node node) {
148         if (node.isName()) {
149           String name = node.getString();
150           if (name != null && !name.isEmpty()) {
151             return name;
152           } else {
153             return null;
154           }
155         } else if (node.isGetProp()) {
156           return "this." + node.getLastChild().getString();
157         }
158         return null;
159       }

161       private class DefinitionGatheringCallback implements Callback {
162         private boolean inExterns;
163     
164         DefinitionGatheringCallback(boolean inExterns) {
165           this.inExterns = inExterns;
166         }
167     
168         @Override
169         public boolean shouldTraverse(NodeTraversal t, Node n, Node parent) {
170           if (inExterns) {
171             if (n.isFunction() && !n.getFirstChild().isName()) {
172               // No need to crawl functions in JSDoc
173               return false;
174             }
175             if (parent != null
176                 && parent.isFunction() && n != parent.getFirstChild()) {
177               // Arguments of external functions should not count as name
178               // definitions.  They are placeholder names for documentation
179               // purposes only which are not reachable from anywhere.
180               return false;
181             }
182           }
183           return true;
184         }
185     
186     
187         @Override
188         public void visit(NodeTraversal traversal, Node node, Node parent) {
189           if (inExterns && node.getJSDocInfo() != null) {
190             for (Node typeRoot : node.getJSDocInfo().getTypeNodes()) {
191               traversal.traverse(typeRoot);
192             }
193           }
194     
195           Definition def =
196               DefinitionsRemover.getDefinition(node, inExterns);
197           if (def != null) {
198             String name = getSimplifiedName(def.getLValue());
199             if (name != null) {
200               Node rValue = def.getRValue();
201               if ((rValue != null) &&
202                   !NodeUtil.isImmutableValue(rValue) &&
203                   !rValue.isFunction()) {
204     
205                 // Unhandled complex expression
206                 Definition unknownDef =
207                     new UnknownDefinition(def.getLValue(), inExterns);
208                 def = unknownDef;
209               }
210     
211               // TODO(johnlenz) : remove this stub dropping code if it becomes
212               // illegal to have untyped stubs in the externs definitions.
213               if (inExterns) {
214                 // We need special handling of untyped externs stubs here:
215                 //    the stub should be dropped if the name is provided elsewhere.
216     
217                 List<Definition> stubsToRemove = new ArrayList<>();
218     
219                 // If there is no qualified name for this, then there will be
220                 // no stubs to remove. This will happen if node is an object
221                 // literal key.
222                 if (node.isQualifiedName()) {
223                   for (Definition prevDef : nameDefinitionMultimap.get(name)) {
224                     if (prevDef instanceof ExternalNameOnlyDefinition
225                         && !jsdocContainsDeclarations(node)) {
226                       if (node.matchesQualifiedName(prevDef.getLValue())) {
227                         // Drop this stub, there is a real definition.
228                         stubsToRemove.add(prevDef);
229                       }
230                     }
231                   }
232     
233                   for (Definition prevDef : stubsToRemove) {
234                     nameDefinitionMultimap.remove(name, prevDef);
235                   }
236                 }
237               }
238     
239               nameDefinitionMultimap.put(name, def);
240               definitionSiteMap.put(node,
241                                     new DefinitionSite(node,
242                                                        def,
243                                                        traversal.getModule(),
244                                                        traversal.inGlobalScope(),
245                                                        inExterns));
246             }
247           }
248     
249           if (inExterns && (parent != null) && parent.isExprResult()) {
250             String name = getSimplifiedName(node);
251             if (name != null) {
252     
253               // TODO(johnlenz) : remove this code if it becomes illegal to have
254               // stubs in the externs definitions.
255     
256               // We need special handling of untyped externs stubs here:
257               //    the stub should be dropped if the name is provided elsewhere.
258               // We can't just drop the stub now as it needs to be used as the
259               //    externs definition if no other definition is provided.
260     
261               boolean dropStub = false;
262               if (!jsdocContainsDeclarations(node) && node.isQualifiedName()) {
263                 for (Definition prevDef : nameDefinitionMultimap.get(name)) {
264                   if (node.matchesQualifiedName(prevDef.getLValue())) {
265                     dropStub = true;
266                     break;
267                   }
268                 }
269               }
270     
271               if (!dropStub) {
272                 // Incomplete definition
273                 Definition definition = new ExternalNameOnlyDefinition(node);
274                 nameDefinitionMultimap.put(name, definition);
275                 definitionSiteMap.put(node,
276                                       new DefinitionSite(node,
277                                                          definition,
278                                                          traversal.getModule(),
279                                                          traversal.inGlobalScope(),
280                                                          inExterns));
281               }
282             }
283           }
284         }

286         /**
287          * @return Whether the node has a JSDoc that actually declares something.
288          */
289         private boolean jsdocContainsDeclarations(Node node) {
290           JSDocInfo info = node.getJSDocInfo();
291           return (info != null && info.containsDeclaration());
292         }

295       private class UseSiteGatheringCallback extends AbstractPostOrderCallback {
296         @Override
297         public void visit(NodeTraversal traversal, Node node, Node parent) {
298     
299           Collection<Definition> defs = getDefinitionsReferencedAt(node);
300           if (defs == null) {
301             return;
302           }
303     
304           Definition first = defs.iterator().next();
305     
306           String name = getSimplifiedName(first.getLValue());
307           Preconditions.checkNotNull(name);
308           nameUseSiteMultimap.put(
309               name,
310               new UseSite(node, traversal.getScope(), traversal.getModule()));
311         }
312       }

314       /**
315        * @param use A use site to check.
316        * @return Whether the use is a call or new.
317        */
318       static boolean isCallOrNewSite(UseSite use) {
319         Node call = use.node.getParent();
320         if (call == null) {
321           // The node has been removed from the AST.
322           return false;
323         }
324         // We need to make sure we're dealing with a call to the function we're
325         // optimizing. If the the first child of the parent is not the site, this
326         // is a nested call and it's a call to another function.
327         return NodeUtil.isCallOrNew(call) && call.getFirstChild() == use.node;
328       }

330       boolean canModifyDefinition(Definition definition) {
331         if (isExported(definition)) {
332           return false;
333         }
334     
335         // Don't modify unused definitions for two reasons:
336         // 1) It causes unnecessary churn
337         // 2) Other definitions might be used to reflect on this one using
338         //    goog.reflect.object (the check for definitions with uses is below).
339         Collection<UseSite> useSites = getUseSites(definition);
340         if (useSites.isEmpty()) {
341           return false;
342         }
343     
344         for (UseSite site : useSites) {
345           // This catches the case where an object literal in goog.reflect.object
346           // and a prototype method have the same property name.
347     
348           // NOTE(nicksantos): Maps and trogedit both do this by different
349           // mechanisms.
350     
351           Node nameNode = site.node;
352           Collection<Definition> singleSiteDefinitions =
353               getDefinitionsReferencedAt(nameNode);
354           if (singleSiteDefinitions.size() > 1) {
355             return false;
356           }
357     
358           Preconditions.checkState(!singleSiteDefinitions.isEmpty());
359           Preconditions.checkState(singleSiteDefinitions.contains(definition));
360         }
361     
362         return true;
363       }

365       /**
366        * @return Whether the definition is directly exported.
367        */
368       private boolean isExported(Definition definition) {
369         // Assume an exported method result is used.
370         Node lValue = definition.getLValue();
371         if (lValue == null) {
372           return true;
373         }
374     
375         String partialName;
376         if (lValue.isGetProp()) {
377           partialName = lValue.getLastChild().getString();
378         } else if (lValue.isName()) {
379           partialName = lValue.getString();
380         } else {
381           // GETELEM is assumed to be an export or other expression are unknown
382           // uses.
383           return true;
384         }
385     
386         CodingConvention codingConvention = compiler.getCodingConvention();
387         return codingConvention.isExported(partialName);
388       }

390       /**
391        * @return Whether the function is defined in a non-aliasing expression.
392        */
393       static boolean isSimpleFunctionDeclaration(Node fn) {
394         Node parent = fn.getParent();
395         Node grandparent = parent.getParent();
396     
397         // Simple definition finder doesn't provide useful results in some
398         // cases, specifically:
399         //  - functions with recursive definitions
400         //  - functions defined in object literals
401         //  - functions defined in array literals
402         // Here we defined a set of known function declaration that are 'ok'.
403     
404         // Some projects seem to actually define "JSCompiler_renameProperty"
405         // rather than simply having an extern definition.  Don't mess with it.
406         Node nameNode = SimpleDefinitionFinder.getNameNodeFromFunctionNode(fn);
407         if (nameNode != null
408             && nameNode.isName()) {
409           String name = nameNode.getString();
410           if (name.equals(NodeUtil.JSC_PROPERTY_NAME_FN)
411               || name.equals(EXTERN_OBJECT_PROPERTY_STRING)) {
412             return false;
413           }
414         }
415     
416         // example: function a(){};
417         if (NodeUtil.isFunctionDeclaration(fn)) {
418           return true;
419         }
420     
421         // example: a = function(){};
422         // example: var a = function(){};
423         return fn.getFirstChild().getString().isEmpty()
424             && (NodeUtil.isExprAssign(grandparent) || parent.isName());
425       }

427       /**
428        * @return the node defining the name for this function (if any).
429        */
430       static Node getNameNodeFromFunctionNode(Node function) {
431         Preconditions.checkState(function.isFunction());
432         if (NodeUtil.isFunctionDeclaration(function)) {
433           return function.getFirstChild();
434         } else {
435           Node parent = function.getParent();
436           if (NodeUtil.isVarDeclaration(parent)) {
437             return parent;
438           } else if (parent.isAssign()) {
439             return parent.getFirstChild();
440           } else if (NodeUtil.isObjectLitKey(parent)) {
441             return parent;
442           }
443         }
444         return null;
445       }

447       /**
448        * Traverse a node and its children and remove any references to from
449        * the structures.
450        */
451       void removeReferences(Node node) {
452         if (DefinitionsRemover.isDefinitionNode(node)) {
453           DefinitionSite defSite = definitionSiteMap.get(node);
454           if (defSite != null) {
455             Definition def = defSite.definition;
456             String name = getSimplifiedName(def.getLValue());
457             if (name != null) {
458               this.definitionSiteMap.remove(node);
459               this.nameDefinitionMultimap.remove(name, node);
460             }
461           }
462         } else {
463           Node useSite = node;
464           if (useSite.isGetProp()) {
465             String propName = useSite.getLastChild().getString();
466             if (propName.equals("apply") || propName.equals("call")) {
467               useSite = useSite.getFirstChild();
468             }
469           }
470           String name = getSimplifiedName(useSite);
471           if (name != null) {
472             this.nameUseSiteMultimap.remove(name, new UseSite(useSite, null, null));
473           }
474         }
475     
476         for (Node child : node.children()) {
477           removeReferences(child);
478         }
479       }
