16     package com.google.javascript.jscomp;
17     
18     import com.google.javascript.jscomp.NodeTraversal.AbstractShallowCallback;
19     import com.google.javascript.rhino.JSDocInfo;
20     import com.google.javascript.rhino.JSDocInfo.Visibility;
21     import com.google.javascript.rhino.Node;
22     
23     import java.util.HashMap;
24     import java.util.Map;
25     
26     /**
27      * Ensures that '@constructor X' has a 'goog.provide("X")' .
28      *
29      */
30     class CheckProvides implements HotSwapCompilerPass {
31       private final AbstractCompiler compiler;
32       private final CodingConvention codingConvention;
33     
34       static final DiagnosticType MISSING_PROVIDE_WARNING = DiagnosticType.warning(
35           "JSC_MISSING_PROVIDE",
36           "missing goog.provide(''{0}'')");
37     
38       CheckProvides(AbstractCompiler compiler) {
39         this.compiler = compiler;
40         this.codingConvention = compiler.getCodingConvention();
41       }
42     
43       @Override
44       public void process(Node externs, Node root) {
45         hotSwapScript(root, null);
46       }
47     
48       @Override
49       public void hotSwapScript(Node scriptRoot, Node originalRoot) {
50         CheckProvidesCallback callback =
51             new CheckProvidesCallback(codingConvention);
52         NodeTraversal.traverseEs6(compiler, scriptRoot, callback);
53       }
54     
55       private class CheckProvidesCallback extends AbstractShallowCallback {
56         private final Map<String, Node> provides = new HashMap<>();
57         private final Map<String, Node> ctors = new HashMap<>();
58         private final CodingConvention convention;
59         private boolean containsRequires = false;
60     
61         CheckProvidesCallback(CodingConvention convention){
62           this.convention = convention;
63         }
64     
65         @Override
66         public void visit(NodeTraversal t, Node n, Node parent) {
67           switch (n.getType()) {
68             case CALL:
69               String providedClassName =
70                 codingConvention.extractClassNameIfProvide(n, parent);
71               if (providedClassName != null) {
72                 provides.put(providedClassName, n);
73               }
74               if (!containsRequires && codingConvention.extractClassNameIfRequire(n, parent) != null) {
75                 containsRequires = true;
76               }
77               break;
78             case FUNCTION:
79               // Arrow function can't be constructors
80               if (!n.isArrowFunction()) {
81                 visitFunctionNode(n, parent);
82               }
83               break;
84             case CLASS:
85               visitClassNode(n);
86               break;
87             case SCRIPT:
88               visitScriptNode();
89           }
90         }
91     
92         private void visitFunctionNode(Node n, Node parent) {
93           // TODO(user): Use isPrivate method below to recognize all functions.
94           Node name = null;
95           JSDocInfo info = parent.getJSDocInfo();
96           if (info != null && info.isConstructor()) {
97             name = parent.getFirstChild();
98           } else {
99             // look to the child, maybe it's a named function
100             info = n.getJSDocInfo();
101             if (info != null && info.isConstructor()) {
102               name = n.getFirstChild();
103             }
104           }
105           if (name != null && name.isQualifiedName()) {
106             String qualifiedName = name.getQualifiedName();
107             if (!this.convention.isPrivate(qualifiedName)) {
108               Visibility visibility = info.getVisibility();
109               if (!visibility.equals(JSDocInfo.Visibility.PRIVATE)) {
110                 ctors.put(qualifiedName, name);
111               }
112             }
113           }
114         }
115     
116         private void visitClassNode(Node classNode) {
117           String name = NodeUtil.getName(classNode);
118           if (name != null && !isPrivate(classNode)) {
119             ctors.put(name, classNode);
120           }
121         }
122     
123         private boolean isPrivate(Node classOrFn) {
124           JSDocInfo info = NodeUtil.getBestJSDocInfo(classOrFn);
125           if (info != null && info.getVisibility().equals(JSDocInfo.Visibility.PRIVATE)) {
126             return true;
127           }
128           return compiler.getCodingConvention().isPrivate(NodeUtil.getName(classOrFn));
129         }
130     
131         private void visitScriptNode() {
132           for (Map.Entry<String, Node> ctorEntry : ctors.entrySet()) {
133             String ctorName = ctorEntry.getKey();
134             int index = -1;
135             boolean found = false;
136     
137             if (ctorName.startsWith("$jscomp.")
138                 || ClosureRewriteModule.isModuleContent(ctorName)
139                 || ClosureRewriteModule.isModuleExport(ctorName)) {
140               continue;
141             }
142     
143             do {
144               index = ctorName.indexOf('.', index + 1);
145               String provideKey = index == -1 ? ctorName : ctorName.substring(0, index);
146               if (provides.containsKey(provideKey)) {
147                 found = true;
148                 break;
149               }
150             } while (index != -1);
151     
152             if (!found && (containsRequires || !provides.isEmpty())) {
153               Node n = ctorEntry.getValue();
154               compiler.report(
155                   JSError.make(n, MISSING_PROVIDE_WARNING, ctorEntry.getKey()));
156             }
157           }
158           provides.clear();
159           ctors.clear();
160           containsRequires = false;
161         }
162       }
163     }
