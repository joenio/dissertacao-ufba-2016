28     /**
29      * Inlines methods that take no arguments and have only a return statement
30      * returning a property. Because it works on method names rather than type
31      * inference, a method with multiple definitions will be inlined if each
32      * definition is identical.
33      *
34      * <pre>
35      * A.prototype.foo = function() { return this.b; }
36      * B.prototype.foo = function() { return this.b; }
37      * </pre>
38      *
39      * will inline foo, but
40      *
41      * <pre>
42      * A.prototype.foo = function() { return this.b; }
43      * B.prototype.foo = function() { return this.c; }
44      * </pre>
45      *
46      * will not.
47      *
48      * Declarations are not removed because we do not find all possible
49      * call sites. For examples, calls of the form foo["bar"] are not
50      * detected.
51      *
52      * This pass is not on by default because it is not safe in simple mode.
53      * If the prototype method is mutated and we don't detect that, inlining it is
54      * unsafe.
55      * We enable it whenever function inlining is enabled.
56      *
57      */
58     class InlineSimpleMethods extends MethodCompilerPass {
59     
60       private static final Logger logger =
61           Logger.getLogger(InlineSimpleMethods.class.getName());
62     
63       InlineSimpleMethods(AbstractCompiler compiler) {
64         super(compiler);
65       }
66     
67       /**
68        * For each method call, see if it is a candidate for inlining.
69        * TODO(kushal): Cache the results of the checks
70        */
71       private class InlineTrivialAccessors extends InvocationsCallback {
72     
73         @Override
74         void visit(NodeTraversal t, Node callNode, Node parent, String callName) {
75           if (externMethods.contains(callName) ||
76               nonMethodProperties.contains(callName)) {
77             return;
78           }
79     
80           Collection<Node> definitions = methodDefinitions.get(callName);
81           if (definitions == null || definitions.isEmpty()) {
82             return;
83           }
84     
85           // Do check of arity, complexity, and consistency in what we think is
86           // the order from least to most complex
87           Node firstDefinition = definitions.iterator().next();
88     
89           // Check any multiple definitions
90           if (definitions.size() == 1 || allDefinitionsEquivalent(definitions)) {
91     
92             if (!argsMayHaveSideEffects(callNode)) {
93               // Verify this is a trivial return
94               Node returned = returnedExpression(firstDefinition);
95               if (returned != null) {
96                 if (isPropertyTree(returned)) {
97                   logger.fine("Inlining property accessor: " + callName);
98                   inlinePropertyReturn(parent, callNode, returned);
99                 } else if (NodeUtil.isLiteralValue(returned, false) &&
100                   !NodeUtil.mayHaveSideEffects(
101                       callNode.getFirstChild(), compiler)) {
102                   logger.fine("Inlining constant accessor: " + callName);
103                   inlineConstReturn(parent, callNode, returned);
104                 }
105               } else if (isEmptyMethod(firstDefinition) &&
106                   !NodeUtil.mayHaveSideEffects(
107                       callNode.getFirstChild(), compiler)) {
108                 logger.fine("Inlining empty method: " + callName);
109                 inlineEmptyMethod(parent, callNode);
110               }
111             }
112           } else {
113             logger.fine("Method '" + callName + "' has conflicting definitions.");
114           }
115         }
116       }
117     
118       @Override
119       Callback getActingCallback() {
120         return new InlineTrivialAccessors();
121       }
122     
123       /**
124        * Returns true if the provided node is a getprop for
125        * which the left child is this or a valid property tree
126        * and for which the right side is a string.
127        */
128       private static boolean isPropertyTree(Node expectedGetprop) {
129         if (!expectedGetprop.isGetProp()) {
130           return false;
131         }
132     
133         Node leftChild = expectedGetprop.getFirstChild();
134         if (!leftChild.isThis() &&
135             !isPropertyTree(leftChild)) {
136           return false;
137         }
138     
139         Node retVal = leftChild.getNext();
140         return NodeUtil.getStringValue(retVal) != null;
141       }
142     
143       /**
144        * Finds the occurrence of "this" in the provided property tree and replaces
145        * it with replacement
146        */
147       private static void replaceThis(Node expectedGetprop, Node replacement) {
148         Node leftChild = expectedGetprop.getFirstChild();
149         if (leftChild.isThis()) {
150           expectedGetprop.replaceChild(leftChild, replacement);
151         } else {
152           replaceThis(leftChild, replacement);
153         }
154       }
155     
156       /**
157        * Return the node that represents the expression returned
158        * by the method, given a FUNCTION node.
159        */
160       private static Node returnedExpression(Node fn) {
161         Node expectedBlock = getMethodBlock(fn);
162         if (!expectedBlock.hasOneChild()) {
163           return null;
164         }
165     
166         Node expectedReturn = expectedBlock.getFirstChild();
167         if (!expectedReturn.isReturn()) {
168           return null;
169         }
170     
171         if (!expectedReturn.hasOneChild()) {
172           return null;
173         }
174     
175         return expectedReturn.getLastChild();
176       }
177     
178     
179       /**
180        * Return whether the given FUNCTION node is an empty method definition.
181        *
182        * Must be private, or moved to NodeUtil.
183        */
184       private static boolean isEmptyMethod(Node fn) {
185         Node expectedBlock = getMethodBlock(fn);
186         return expectedBlock == null ?
187             false : NodeUtil.isEmptyBlock(expectedBlock);
188       }
189     
190       /**
191        * Return a BLOCK node if the given FUNCTION node is a valid method
192        * definition, null otherwise.
193        *
194        * Must be private, or moved to NodeUtil.
195        */
196       private static Node getMethodBlock(Node fn) {
197         if (fn.getChildCount() != 3) {
198           return null;
199         }
200     
201         Node expectedBlock = fn.getLastChild();
202         return  expectedBlock.isBlock() ?
203             expectedBlock : null;
204       }
205     
206       /**
207        * Given a set of method definitions, verify they are the same.
208        */
209       private boolean allDefinitionsEquivalent(
210           Collection<Node> definitions) {
211         List<Node> list = new ArrayList<>();
212         list.addAll(definitions);
213         Node node0 = list.get(0);
214         for (int i = 1; i < list.size(); i++) {
215           if (!compiler.areNodesEqualForInlining(list.get(i), node0)) {
216             return false;
217           }
218         }
219         return true;
220       }
221     
222       /**
223        * Replace the provided method call with the tree specified in returnedValue
224        *
225        * Parse tree of a call is
226        * name
227        *   call
228        *     getprop
229        *       obj
230        *       string
231        */
232       private void inlinePropertyReturn(Node parent, Node call,
233           Node returnedValue) {
234         Node getProp = returnedValue.cloneTree();
235         replaceThis(getProp, call.getFirstChild().removeFirstChild());
236         parent.replaceChild(call, getProp);
237         compiler.reportCodeChange();
238       }
239     
240       /**
241        * Replace the provided object and its method call with the tree specified
242        * in returnedValue. Should be called only if the object reference has
243        * no side effects.
244        */
245       private void inlineConstReturn(Node parent, Node call,
246           Node returnedValue) {
247         Node retValue = returnedValue.cloneTree();
248         parent.replaceChild(call, retValue);
249         compiler.reportCodeChange();
250       }
251     
252       /**
253        * Remove the provided object and its method call.
254        */
255       private void inlineEmptyMethod(Node parent, Node call) {
256         // If the return value of the method call is read,
257         // replace it with "void 0". Otherwise, remove the call entirely.
258         if (NodeUtil.isExprCall(parent)) {
259           parent.getParent().replaceChild(parent, IR.empty());
260         } else {
261           Node srcLocation = call;
262           parent.replaceChild(call, NodeUtil.newUndefinedNode(srcLocation));
263         }
264         compiler.reportCodeChange();
265       }
266     
267       /**
268        * Check whether the given method call's arguments have side effects.
269        * @param call The call node of a method invocation.
270        */
271       private boolean argsMayHaveSideEffects(Node call) {
272         for (Node currentChild = call.getSecondChild();
273              currentChild != null;
274              currentChild = currentChild.getNext()) {
275           if (NodeUtil.mayHaveSideEffects(currentChild, compiler)) {
276             return true;
277           }
278         }
279     
280         return false;
281       }
