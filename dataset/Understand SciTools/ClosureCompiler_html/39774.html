21     /**
22      * Helper methods for parsing JavaScript.
23      *
24      * @author moz@google.com (Michael Zhou)
25      */
26     public class TokenUtil {
27     
28       static native boolean isJSIdentifier(String s) /*-{
29         return /^[a-zA-Z_$][\w$]*$/.test(s); // TODO(moz): Support full range of valid characters.
30       }-*/;
31     
32       public static boolean isJSSpace(int c) {
33         if (c <= 127) {
34           return c == 0x20 || c == 0x9 || c == 0xC || c == 0xB;
35         } else {
36           return c == 0xA0; // TODO(moz): Correct this.
37         }
38       }
39     
40       public static boolean isJSFormatChar(int c) {
41         return c > 127; // TODO(moz): Correct this.
42       }
43     
44       public static native boolean isWhitespace(int c) /*-{
45         return /\s/.test(c);
46       }-*/;
47     
48       public static TernaryValue isStrWhiteSpaceChar(int c) {
49         switch (c) {
50           case '\u000B': // <VT>
51             return TernaryValue.UNKNOWN;  // Legacy IE says "no", ECMAScript says "yes"
52           case ' ': // <SP>
53           case '\n': // <LF>
54           case '\r': // <CR>
55           case '\t': // <TAB>
56           case '\u00A0': // <NBSP>
57           case '\u000C': // <FF>
58           case '\u2028': // <LS>
59           case '\u2029': // <PS>
60           case '\uFEFF': // <BOM>
61             return TernaryValue.TRUE;
62           default:
63             return TernaryValue.FALSE; // TODO(moz): Correct this.
64         }
65       }
66     }
