20     /**
21      * Ensures that the InlineVariables pass in constants-only mode
22      * is functionally equivalent to the old InlineVariablesConstants pass.
23      */
24     public final class InlineVariablesConstantsTest extends CompilerTestCase {
25     
26       private boolean inlineAllStrings = false;
27     
28       public InlineVariablesConstantsTest() {
29         enableNormalize();
30       }
31     
32       @Override
33       protected CompilerPass getProcessor(final Compiler compiler) {
34         return new InlineVariables(
35             compiler, InlineVariables.Mode.CONSTANTS_ONLY, inlineAllStrings);
36       }
37     
38       @Override
39       public void tearDown() {
40         inlineAllStrings = false;
41       }
42     
43       public void testInlineVariablesConstants() {
44         test("var ABC=2; var x = ABC;", "var x=2");
45         test("var AA = 'aa'; AA;", "'aa'");
46         test("var A_A=10; A_A + A_A;", "10+10");
47         test("var AA=1", "");
48         test("var AA; AA=1", "1");
49         test("var AA; if (false) AA=1; AA;", "if (false) 1; 1;");
50         testSame("var AA; if (false) AA=1; else AA=2; AA;");
51     
52         // Make sure that nothing explodes if there are undeclared variables.
53         testSame("var x = AA;");
54     
55         // Don't inline if it will make the output larger.
56         testSame("var AA = '1234567890'; foo(AA); foo(AA); foo(AA);");
57     
58         test("var AA = '123456789012345';AA;",
59              "'123456789012345'");
60       }
61     
62       public void testNoInlineArraysOrRegexps() {
63         testSame("var AA = [10,20]; AA[0]");
64         testSame("var AA = [10,20]; AA.push(1); AA[0]");
65         testSame("var AA = /x/; AA.test('1')");
66         testSame("/** @const */ var aa = /x/; aa.test('1')");
67       }
68     
69       public void testInlineVariablesConstantsJsDocStyle() {
70         test("/** @const */var abc=2; var x = abc;", "var x=2");
71         test("/** @const */var aa = 'aa'; aa;", "'aa'");
72         test("/** @const */var a_a=10; a_a + a_a;", "10+10");
73         test("/** @const */var aa=1;", "");
74         test("/** @const */var aa; aa=1;", "1");
75         testSame("/** @const */var aa;(function () {var y; aa=y})(); var z=aa");
76     
77         // Don't inline if it will make the output larger.
78         testSame("/** @const */var aa = '1234567890'; foo(aa); foo(aa); foo(aa);");
79     
80         test("/** @const */var aa = '123456789012345';aa;",
81              "'123456789012345'");
82       }
83     
84       public void testInlineConditionallyDefinedConstant1() {
85         // Note that inlining conditionally defined constants can change the
86         // run-time behavior of code (e.g. when y is true and x is false in the
87         // example below). We inline them anyway because if the code author didn't
88         // want one inlined, they could define it as a non-const variable instead.
89         test("if (x) var ABC = 2; if (y) f(ABC);",
90              "if (x); if (y) f(2);");
91       }
92     
93       public void testInlineConditionallyDefinedConstant2() {
94         test("if (x); else var ABC = 2; if (y) f(ABC);",
95              "if (x); else; if (y) f(2);");
96       }
97     
98       public void testInlineConditionallyDefinedConstant3() {
99         test("if (x) { var ABC = 2; } if (y) { f(ABC); }",
100              "if (x) {} if (y) { f(2); }");
101       }
102     
103       public void testInlineDefinedConstant() {
104         test(
105             "/**\n" +
106             " * @define {string}\n" +
107             " */\n" +
108             "var aa = '1234567890';\n" +
109             "foo(aa); foo(aa); foo(aa);",
110             "foo('1234567890');foo('1234567890');foo('1234567890')");
111     
112         test(
113             "/**\n" +
114             " * @define {string}\n" +
115             " */\n" +
116             "var ABC = '1234567890';\n" +
117             "foo(ABC); foo(ABC); foo(ABC);",
118             "foo('1234567890');foo('1234567890');foo('1234567890')");
119       }
120     
121       public void testInlineVariablesConstantsWithInlineAllStringsOn() {
122         inlineAllStrings = true;
123         test("var AA = '1234567890'; foo(AA); foo(AA); foo(AA);",
124              "foo('1234567890'); foo('1234567890'); foo('1234567890')");
125       }
126     
127       public void testNoInlineWithoutConstDeclaration() {
128         testSame("var abc = 2; var x = abc;");
129       }
