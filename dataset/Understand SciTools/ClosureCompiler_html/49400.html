17     package com.google.javascript.jscomp;
18     
19     
20     /**
21      * Tests for {@link CollapseAnonymousFunctions}
22      *
23      */
24     public final class CollapseAnonymousFunctionsTest extends CompilerTestCase {
25       public CollapseAnonymousFunctionsTest() {
26         this.enableNormalize();
27       }
28     
29       @Override
30       protected CompilerPass getProcessor(Compiler compiler) {
31         return new CollapseAnonymousFunctions(compiler);
32       }
33     
34       public void testGlobalScope() {
35         test("var f = function(){}", "function f(){}");
36       }
37     
38       public void testLocalScope1() {
39         test("function f(){ var x = function(){}; x() }",
40              "function f(){ function x(){} x() }");
41       }
42     
43       public void testLocalScope2() {
44         test("function f(){ var x = function(){}; return x }",
45              "function f(){ function x(){} return x }");
46       }
47     
48       public void testVarNotImmediatelyBelowScriptOrBlock1() {
49         testSame("if (x) var f = function(){}");
50       }
51     
52       public void testVarNotImmediatelyBelowScriptOrBlock2() {
53         testSame("var x = 1;" +
54                  "if (x == 1) {" +
55                  "  var f = function () { alert('b')}" +
56                  "} else {" +
57                  "  f = function() { alert('c')}" +
58                  "}" +
59                  "f();");
60       }
61     
62       public void testVarNotImmediatelyBelowScriptOrBlock3() {
63         testSame("var x = 1; if (x) {var f = function(){return x}; f(); x--;}");
64       }
65     
66       public void testMultipleVar() {
67         test("var f = function(){}; var g = f", "function f(){} var g = f");
68       }
69     
70       public void testMultipleVar2() {
71         test("var f = function(){}; var g = f; var h = function(){}",
72              "function f(){}var g = f;function h(){}");
73       }
74     
75       public void testBothScopes() {
76         test("var x = function() { var y = function(){} }",
77              "function x() { function y(){} }");
78       }
79     
80       public void testLocalScopeOnly1() {
81         test("if (x) var f = function(){ var g = function(){} }",
82              "if (x) var f = function(){ function g(){} }");
83       }
84     
85       public void testLocalScopeOnly2() {
86         test("if (x) var f = function(){ var g = function(){} };",
87              "if (x) var f = function(){ function g(){} }");
88       }
89     
90       public void testLocalScopeOnly3() {
91         test("if (x){ var f = function(){ var g = function(){} };}",
92              "if (x){ var f = function(){ function g(){} }}");
93       }
94     
95       public void testReturn() {
96         test("var f = function(x){return 2*x}; var g = f(2);",
97              "function f(x){return 2*x} var g = f(2)");
98       }
99     
100       public void testAlert() {
101         test("var x = 1; var f = function(){alert(x)};",
102              "var x = 1; function f(){alert(x)}");
103       }
104     
105       public void testRecursiveInternal1() {
106         testSame("var f = function foo() { foo() }");
107       }
108     
109       public void testRecursiveInternal2() {
110         testSame("var f = function foo() { function g(){foo()} g() }");
111       }
112     
113       public void testRecursiveExternal1() {
114         test("var f = function foo() { f() }",
115              "function f() { f() }");
116       }
117     
118       public void testRecursiveExternal2() {
119         test("var f = function foo() { function g(){f()} g() }",
120              "function f() { function g(){f()} g() }");
121       }
122     
123       public void testConstantFunction1() {
124         test("var FOO = function(){};FOO()",
125              "function FOO(){}FOO()");
126       }
127     
128       public void testInnerFunction1() {
129         test(
130             "function f() { " +
131             "  var x = 3; var y = function() { return 4; }; return x + y();" +
132             "}",
133             "function f() { " +
134             "  function y() { return 4; } var x = 3; return x + y();" +
135             "}");
136       }
