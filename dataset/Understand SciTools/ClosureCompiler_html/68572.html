25     /**
26      * Helper class for transpiling ES6 template literals.
27      *
28      * @author moz@google.com (Michael Zhou)
29      */
30     class Es6TemplateLiterals {
31       private static final String TEMPLATELIT_VAR = "$jscomp$templatelit$";
32     
33       /**
34        * Converts `${a} b ${c} d ${e}` to (a + " b " + c + " d " + e)
35        *
36        * @param n A TEMPLATELIT node that is not prefixed with a tag
37        */
38       static void visitTemplateLiteral(NodeTraversal t, Node n) {
39         int length = n.getChildCount();
40         if (length == 0) {
41           n.getParent().replaceChild(n, IR.string("\"\""));
42         } else {
43           Node first = n.removeFirstChild();
44           Preconditions.checkState(first.isString());
45           if (length == 1) {
46             n.getParent().replaceChild(n, first);
47           } else {
48             // Add the first string with the first substitution expression
49             Node add = IR.add(first, n.removeFirstChild().removeFirstChild());
50             // Process the rest of the template literal
51             for (int i = 2; i < length; i++) {
52               Node child = n.removeFirstChild();
53               if (child.isString()) {
54                 if (child.getString().isEmpty()) {
55                   continue;
56                 } else if (i == 2 && first.getString().isEmpty()) {
57                   // So that `${hello} world` gets translated into (hello + " world")
58                   // instead of ("" + hello + " world").
59                   add = add.getSecondChild().detachFromParent();
60                 }
61               }
62               add = IR.add(add, child.isString() ? child : child.removeFirstChild());
63             }
64             n.getParent().replaceChild(n, add.useSourceInfoIfMissingFromForTree(n));
65           }
66         }
67         t.getCompiler().reportCodeChange();
68       }

70       /**
71        * Converts tag`a\tb${bar}` to:
72        *   // A global (module) scoped variable
73        *   var $jscomp$templatelit$0 = ["a\tb"];   // cooked string array
74        *   $jscomp$templatelit$0.raw = ["a\\tb"];  // raw string array
75        *   ...
76        *   // A call to the tagging function
77        *   tag($jscomp$templatelit$0, bar);
78        *
79        *   See template_literal_test.js for more examples.
80        *
81        * @param n A TAGGED_TEMPLATELIT node
82        */
83       static void visitTaggedTemplateLiteral(NodeTraversal t, Node n) {
84         Node templateLit = n.getLastChild();
85         // Prepare the raw and cooked string arrays.
86         Node raw = createRawStringArray(templateLit);
87         Node cooked = createCookedStringArray(templateLit);
88     
89         // Specify the type of the first argument to be ITemplateArray.
90         JSTypeExpression nonNullSiteObject = new JSTypeExpression(
91             JsDocInfoParser.parseTypeString("!ITemplateArray"), "<Es6TemplateLiterals.java>");
92         JSDocInfoBuilder info = new JSDocInfoBuilder(false);
93         info.recordType(nonNullSiteObject);
94         Node siteObject = IR.cast(cooked, info.build());
95     
96         // Create a variable representing the template literal.
97         Node callsiteId = IR.name(
98             TEMPLATELIT_VAR + t.getCompiler().getUniqueNameIdSupplier().get());
99         Node var = IR.var(callsiteId, siteObject).useSourceInfoIfMissingFromForTree(n);
100         Node script = NodeUtil.getEnclosingScript(n);
101         script.addChildrenToFront(var);
102     
103         // Define the "raw" property on the introduced variable.
104         Node defineRaw = IR.exprResult(IR.assign(IR.getprop(
105             callsiteId.cloneNode(), IR.string("raw")), raw))
106                 .useSourceInfoIfMissingFromForTree(n);
107         script.addChildAfter(defineRaw, var);
108     
109         // Generate the call expression.
110         Node call = IR.call(n.removeFirstChild(), callsiteId.cloneNode());
111         for (Node child = templateLit.getFirstChild(); child != null; child = child.getNext()) {
112           if (!child.isString()) {
113             call.addChildToBack(child.removeFirstChild());
114           }
115         }
116         call.useSourceInfoIfMissingFromForTree(templateLit);
117         call.putBooleanProp(Node.FREE_CALL, !call.getFirstChild().isGetProp());
118         n.getParent().replaceChild(n, call);
119         t.getCompiler().reportCodeChange();
120       }

122       private static Node createRawStringArray(Node n) {
123         Node array = IR.arraylit();
124         for (Node child = n.getFirstChild(); child != null; child = child.getNext()) {
125           if (child.isString()) {
126             array.addChildToBack(IR.string((String) child.getProp(Node.RAW_STRING_VALUE)));
127           }
128         }
129         return array;
130       }
131     
132       private static Node createCookedStringArray(Node n) {
133         Node array = IR.arraylit();
134         for (Node child = n.getFirstChild(); child != null; child = child.getNext()) {
135           if (child.isString()) {
136             Node string = IR.string(cookString((String) child.getProp(Node.RAW_STRING_VALUE)));
137             array.addChildToBack(string);
138           }
139         }
140         return array;
141       }

143       /**
144        * Takes a raw string and returns a string that is suitable for the cooked
145        * value (the Template Value or TV as described in the specs). This involves
146        * removing line continuations.
147        */
148       private static String cookString(String s) {
149         StringBuilder sb = new StringBuilder();
150         for (int i = 0; i < s.length();) {
151           char c = s.charAt(i++);
152           if (c == '\\') {
153             char c2 = s.charAt(i++);
154             switch (c2) {
155               case 't':
156                 sb.append('\t');
157                 break;
158               case 'n':
159                 sb.append('\n');
160                 break;
161               case 'r':
162                 sb.append('\r');
163                 break;
164               case 'f':
165                 sb.append('\f');
166                 break;
167               case 'b':
168                 sb.append('\b');
169                 break;
170               case 'u':
171                 int unicodeValue = Integer.parseInt(s.substring(i, i + 4), 16);
172                 sb.append((char) unicodeValue);
173                 i += 4;
174                 break;
175               // Strip line continuation.
176               case '\n':
177               case '\u2028':
178               case '\u2029':
179                 break;
180               case '\r':
181                 // \ \r \n should be stripped as one
182                 if (s.charAt(i + 1) == '\n') {
183                   i++;
184                 }
185                 break;
186     
187               default:
188                 sb.append(c);
189                 sb.append(c2);
190             }
191           } else {
192             sb.append(c);
193           }
194         }
195         return sb.toString();
196       }
