37     /**
38      * Inline variables when possible. Using the information from
39      * {@link MaybeReachingVariableUse} and {@link MustBeReachingVariableDef},
40      * this pass attempts to inline a variable by placing the value at the
41      * definition where the variable is used. The basic requirements for inlining
42      * are the following:
43      *
44      * <ul>
45      * <li> There is exactly one reaching definition at the use of that variable
46      * </li>
47      * <li> There is exactly one use for that definition of the variable
48      * </li>
49      * </ul>
50      *
51      * <p>Other requirements can be found in {@link Candidate#canInline}. Currently
52      * this pass does not operate on the global scope due to compilation time.
53      *
54      */
55     class FlowSensitiveInlineVariables extends AbstractPostOrderCallback
56         implements CompilerPass, ScopedCallback {
57     
58       /**
59        * Implementation:
60        *
61        * This pass first perform a traversal to gather a list of Candidates that
62        * could be inlined using {@link GatherCandiates}.
63        *
64        * The second step involves verifying that each candidate is actually safe
65        * to inline with {@link Candidate#canInline(Scope)} and finally perform
66        * inlining using {@link Candidate#inlineVariable()}.
67        *
68        * The reason for the delayed evaluation of the candidates is because we
69        * need two separate dataflow result.
70        */
71       private final AbstractCompiler compiler;
72     
73       // These two pieces of data is persistent in the whole execution of enter
74       // scope.
75       private ControlFlowGraph<Node> cfg;
76       private Set<Candidate> candidates;
77       private MustBeReachingVariableDef reachingDef;
78       private MaybeReachingVariableUse reachingUses;
79     
80       private static final Predicate<Node> SIDE_EFFECT_PREDICATE =
81         new Predicate<Node>() {
82           @Override
83           public boolean apply(Node n) {
84             // When the node is null it means, we reached the implicit return
85             // where the function returns (possibly without an return statement)
86             if (n == null) {
87               return false;
88             }
89     
90             // TODO(user): We only care about calls to functions that
91             // passes one of the dependent variable to a non-side-effect free
92             // function.
93             if (n.isCall() && NodeUtil.functionCallHasSideEffects(n)) {
94               return true;
95             }
96     
97             if (n.isNew() && NodeUtil.constructorCallHasSideEffects(n)) {
98               return true;
99             }
100     
101             if (n.isDelProp()) {
102               return true;
103             }
104     
105             for (Node c = n.getFirstChild(); c != null; c = c.getNext()) {
106               if (!ControlFlowGraph.isEnteringNewCfgNode(c) && apply(c)) {
107                 return true;
108               }
109             }
110             return false;
111           }
112       };

114       public FlowSensitiveInlineVariables(AbstractCompiler compiler) {
115         this.compiler = compiler;
116       }

118       @Override
119       public void enterScope(NodeTraversal t) {
120     
121         if (t.inGlobalScope()) {
122           return; // Don't even brother. All global variables are likely escaped.
123         }
124     
125         if (LiveVariablesAnalysis.MAX_VARIABLES_TO_ANALYZE <
126             t.getScope().getVarCount()) {
127           return;
128         }
129     
130         // Compute the forward reaching definition.
131         ControlFlowAnalysis cfa = new ControlFlowAnalysis(compiler, false, true);
132         // Process the body of the function.
133         Preconditions.checkState(t.getScopeRoot().isFunction());
134         cfa.process(null, t.getScopeRoot());
135         cfg = cfa.getCfg();
136         reachingDef = new MustBeReachingVariableDef(cfg, t.getScope(), compiler);
137         reachingDef.analyze();
138         candidates = new LinkedHashSet<>();
139     
140         // Using the forward reaching definition search to find all the inline
141         // candidates
142         NodeTraversal.traverseEs6(compiler, t.getScopeRoot().getLastChild(), new GatherCandiates());
143     
144         // Compute the backward reaching use. The CFG can be reused.
145         reachingUses = new MaybeReachingVariableUse(cfg, t.getScope(), compiler);
146         reachingUses.analyze();
147         while (!candidates.isEmpty()) {
148           Candidate c = candidates.iterator().next();
149           if (c.canInline(t.getScope())) {
150             c.inlineVariable();
151             candidates.remove(c);
152     
153             // If candidate "c" has dependencies, then inlining it may have introduced new dependencies
154             // for our other inlining candidates. MustBeReachingVariableDef uses a dependency graph in
155             // its analysis. Generating a new dependency graph will need another CFG computation.
156             // Ideally we should iterate to a fixed point, but that can be costly. Therefore, we use
157             // a conservative heuristic here: For each candidate "other", we back off if its set of
158             // dependencies cannot contain all of "c"'s dependencies.
159             if (!c.defMetadata.depends.isEmpty()) {
160               for (Iterator<Candidate> it = candidates.iterator(); it.hasNext();) {
161                 Candidate other = it.next();
162                 if (other.defMetadata.depends.contains(t.getScope().getVar(c.varName))
163                     && !other.defMetadata.depends.containsAll(c.defMetadata.depends)) {
164                   it.remove();
165                 }
166               }
167             }
168           } else {
169             candidates.remove(c);
170           }
171         }
172       }

183       public void visit(NodeTraversal t, Node n, Node parent) {
184         // TODO(user): While the helpers do a subtree traversal on the AST, the
185         // compiler pass itself only traverse the AST to look for function
186         // declarations to perform dataflow analysis on. We could combine
187         // the traversal in DataFlowAnalysis's computeEscaped later to save some

191       /**
192        * Gathers a list of possible candidates for inlining based only on
193        * information from {@link MustBeReachingVariableDef}. The list will be stored
194        * in {@code candidates} and the validity of each inlining Candidate should
195        * be later verified with {@link Candidate#canInline(Scope)} when
196        * {@link MaybeReachingVariableUse} has been performed.
197        */
198       private class GatherCandiates extends AbstractShallowCallback {
199         @Override
200         public void visit(NodeTraversal t, Node n, Node parent) {
201           DiGraphNode<Node, Branch> graphNode = cfg.getDirectedGraphNode(n);
202           if (graphNode == null) {
203             // Not a CFG node.
204             return;
205           }
206           final Node cfgNode = n;
207           AbstractCfgNodeTraversalCallback gatherCb =
208               new AbstractCfgNodeTraversalCallback() {
209     
210             @Override
211             public void visit(NodeTraversal t, Node n, Node parent) {
212               if (n.isName()) {
213     
214                 // n.getParent() isn't null. This just the case where n is the root
215                 // node that gatherCb started at.
216                 if (parent == null) {
217                   return;
218                 }
219     
220                 // Make sure that the name node is purely a read.
221                 if ((NodeUtil.isAssignmentOp(parent) && parent.getFirstChild() == n)
222                     || parent.isVar() || parent.isInc() || parent.isDec() ||
223                     parent.isParamList() || parent.isCatch()) {
224                   return;
225                 }
226     
227                 String name = n.getString();
228                 if (compiler.getCodingConvention().isExported(name)) {
229                   return;
230                 }
231     
232                 Definition def = reachingDef.getDef(name, cfgNode);
233                 // TODO(nicksantos): We need to add some notion of @const outer
234                 // scope vars. We can inline those just fine.
235                 if (def != null &&
236                     !reachingDef.dependsOnOuterScopeVars(def)) {
237                   candidates.add(new Candidate(name, def, n, cfgNode));
238                 }
239               }
240             }
241           };
242     
243           NodeTraversal.traverseEs6(compiler, cfgNode, gatherCb);
244         }
245       }

247       /**
248        * Models the connection between a definition and a use of that definition.
249        */
250       private class Candidate {
251     
252         // Name of the variable.
253         private final String varName;
254     
255         // Nodes related to the definition.
256         private Node def;
257         private final Definition defMetadata;
258     
259         // Nodes related to the use.
260         private final Node use;
261         private final Node useCfgNode;
262     
263         // Number of uses of the variable within the current CFG node.
264         private int numUsesWithinCfgNode;
265     
266         Candidate(String varName, Definition defMetadata,
267             Node use, Node useCfgNode) {
268           Preconditions.checkArgument(use.isName());
269           this.varName = varName;
270           this.defMetadata = defMetadata;
271           this.use = use;
272           this.useCfgNode = useCfgNode;
273         }
274     
275         private Node getDefCfgNode() {
276           return defMetadata.node;
277         }
278     
279         private boolean canInline(final Scope scope) {
280           // Cannot inline a parameter.
281           if (getDefCfgNode().isFunction()) {
282             return false;
283           }
284     
285           getDefinition(getDefCfgNode());
286           getNumUseInUseCfgNode(useCfgNode);
287     
288           // Definition was not found.
289           if (def == null) {
290             return false;
291           }
292     
293           // Check that the assignment isn't used as a R-Value.
294           // TODO(user): Certain cases we can still inline.
295           if (def.isAssign() && !NodeUtil.isExprAssign(def.getParent())) {
296             return false;
297           }
298     
299           // The right of the definition has side effect:
300           // Example, for x:
301           // x = readProp(b), modifyProp(b); print(x);
302           if (checkRightOf(def, getDefCfgNode(), SIDE_EFFECT_PREDICATE)) {
303             return false;
304           }
305     
306           // Similar check as the above but this time, all the sub-expressions
307           // left of the use of the variable.
308           // x = readProp(b); modifyProp(b), print(x);
309           if (checkLeftOf(use, useCfgNode, SIDE_EFFECT_PREDICATE)) {
310             return false;
311           }
312     
313           // TODO(user): Side-effect is OK sometimes. As long as there are no
314           // side-effect function down all paths to the use. Once we have all the
315           // side-effect analysis tool.
316           if (NodeUtil.mayHaveSideEffects(def.getLastChild(), compiler)) {
317             return false;
318           }
319     
320           // TODO(user): We could inline all the uses if the expression is short.
321     
322           // Finally we have to make sure that there are no more than one use
323           // in the program and in the CFG node. Even when it is semantically
324           // correctly inlining twice increases code size.
325           if (numUsesWithinCfgNode != 1) {
326             return false;
327           }
328     
329           // Make sure that the name is not within a loop
330           if (NodeUtil.isWithinLoop(use)) {
331             return false;
332           }
333     
334     
335           Collection<Node> uses = reachingUses.getUses(varName, getDefCfgNode());
336     
337           if (uses.size() != 1) {
338             return false;
339           }
340     
341           // We give up inlining stuff with R-Value that has:
342           // 1) GETPROP, GETELEM,
343           // 2) anything that creates a new object.
344           // 3) a direct reference to a catch expression.
345           // Example:
346           // var x = a.b.c; j.c = 1; print(x);
347           // Inlining print(a.b.c) is not safe consider j and be alias to a.b.
348           // TODO(user): We could get more accuracy by looking more in-detail
349           // what j is and what x is trying to into to.
350           // TODO(johnlenz): rework catch expression handling when we
351           // have lexical scope support so catch expressions don't
352           // need to be special cased.
353           if (NodeUtil.has(def.getLastChild(),
354               new Predicate<Node>() {
355                   @Override
356                   public boolean apply(Node input) {
357                     switch (input.getType()) {
358                       case GETELEM:
359                       case GETPROP:
360                       case ARRAYLIT:
361                       case OBJECTLIT:
362                       case REGEXP:
363                       case NEW:
364                         return true;
365                       case NAME:
366                         Var var = scope.getOwnSlot(input.getString());
367                         if (var != null
368                             && var.getParentNode().isCatch()) {
369                           return true;
370                         }
371                     }
372                     return false;
373                   }
374               },
375               new Predicate<Node>() {
376                   @Override
377                   public boolean apply(Node input) {
378                     // Recurse if the node is not a function.
379                     return !input.isFunction();
380                   }
381               })) {
382             return false;
383           }
384     
385           // We can skip the side effect check along the paths of two nodes if
386           // they are just next to each other.
387           if (NodeUtil.isStatementBlock(getDefCfgNode().getParent()) &&
388               getDefCfgNode().getNext() != useCfgNode) {
389             // Similar side effect check as above but this time the side effect is
390             // else where along the path.
391             // x = readProp(b); while(modifyProp(b)) {}; print(x);
392             CheckPathsBetweenNodes<Node, ControlFlowGraph.Branch>
393               pathCheck = new CheckPathsBetweenNodes<>(
394                      cfg,
395                      cfg.getDirectedGraphNode(getDefCfgNode()),
396                      cfg.getDirectedGraphNode(useCfgNode),
397                      SIDE_EFFECT_PREDICATE,
398                      Predicates.
399                          <DiGraphEdge<Node, ControlFlowGraph.Branch>>alwaysTrue(),
400                      false);
401             if (pathCheck.somePathsSatisfyPredicate()) {
402               return false;
403             }
404           }
405     
406           return true;
407         }

409         /**
410          * Actual transformation.
411          */
412         private void inlineVariable() {
413           Node defParent = def.getParent();
414           Node useParent = use.getParent();
415           if (def.isAssign()) {
416             Node rhs = def.getLastChild();
417             rhs.detachFromParent();
418             // Oh yes! I have grandparent to remove this.
419             Preconditions.checkState(defParent.isExprResult());
420             while (defParent.getParent().isLabel()) {
421               defParent = defParent.getParent();
422             }
423             defParent.detachFromParent();
424             useParent.replaceChild(use, rhs);
425           } else if (defParent.isVar()) {
426             Node rhs = def.getLastChild();
427             def.removeChild(rhs);
428             useParent.replaceChild(use, rhs);
429           } else {
430             throw new IllegalStateException("No other definitions can be inlined.");
431           }
432           compiler.reportCodeChange();
433         }

435         /**
436          * Set the def node
437          *
438          * @param n A node that has a corresponding CFG node in the CFG.
439          */
440         private void getDefinition(Node n) {
441           AbstractCfgNodeTraversalCallback gatherCb =
442             new AbstractCfgNodeTraversalCallback() {
443     
444             @Override
445             public void visit(NodeTraversal t, Node n, Node parent) {
446               switch (n.getType()) {
447                 case NAME:
448                   if (n.getString().equals(varName) && n.hasChildren()) {
449                     def = n;
450                   }
451                   return;
452     
453                 case ASSIGN:
454                   Node lhs = n.getFirstChild();
455                   if (lhs.isName() && lhs.getString().equals(varName)) {
456                     def = n;
457                   }
458                   return;
459               }
460             }
461           };
462           NodeTraversal.traverseEs6(compiler, n, gatherCb);
463         }

465         /**
466          * Computes the number of uses of the variable varName and store it in
467          * numUseWithinUseCfgNode.
468          */
469         private void getNumUseInUseCfgNode(final Node cfgNode) {
470     
471           numUsesWithinCfgNode = 0;
472           AbstractCfgNodeTraversalCallback gatherCb =
473               new AbstractCfgNodeTraversalCallback() {
474     
475             @Override
476             public void visit(NodeTraversal t, Node n, Node parent) {
477               if (n.isName() && n.getString().equals(varName)) {
478                 // We make a special exception when the entire cfgNode is a chain
479                 // of assignments, since in that case the assignment statements
480                 // will happen after the inlining of the right hand side.
481                 // TODO(blickly): Make the SIDE_EFFECT_PREDICATE check more exact
482                 //   and remove this special case.
483                 if (parent.isAssign() && (parent.getFirstChild() == n)
484                     && isAssignChain(parent, cfgNode)) {
485                   // Don't count lhs of top-level assignment chain
486                   return;
487                 } else {
488                   numUsesWithinCfgNode++;
489                 }
490               }
491             }
492     
493             private boolean isAssignChain(Node child, Node ancestor) {
494               for (Node n = child; n != ancestor; n = n.getParent()) {
495                 if (!n.isAssign()) {
496                   return false;
497                 }
498               }
499               return true;
500             }
501           };
502     
503           NodeTraversal.traverseEs6(compiler, cfgNode, gatherCb);
504         }
505       }

507       /**
508        * Given an expression by its root and sub-expression n, return true if there
509        * the predicate is true for some expression on the right of n.
510        *
511        * Example:
512        *
513        * NotChecked(), NotChecked(), n, Checked(), Checked();
514        */
515       private static boolean checkRightOf(
516           Node n, Node expressionRoot, Predicate<Node> predicate) {
517         for (Node p = n; p != expressionRoot; p = p.getParent()) {
518           for (Node cur = p.getNext(); cur != null; cur = cur.getNext()) {
519             if (predicate.apply(cur)) {
520               return true;
521             }
522           }
523         }
524         return false;
525       }
526     
527       /**
528        * Given an expression by its root and sub-expression n, return true if there
529        * the predicate is true for some expression on the left of n.
530        *
531        * Example:
532        *
533        * Checked(), Checked(), n, NotChecked(), NotChecked();
534        */
535       private static boolean checkLeftOf(
536           Node n, Node expressionRoot, Predicate<Node> predicate) {
537         for (Node p = n; p != expressionRoot; p = p.getParent()) {
538           for (Node cur = p.getParent().getFirstChild(); cur != p;
539               cur = cur.getNext()) {
540             if (predicate.apply(cur)) {
541               return true;
542             }
543           }
544         }
545         return false;
546       }
