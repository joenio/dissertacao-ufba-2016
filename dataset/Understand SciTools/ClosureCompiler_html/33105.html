17     package com.google.javascript.jscomp;
18     
19     import com.google.common.base.Preconditions;
20     import com.google.common.collect.ImmutableList;
21     import com.google.javascript.jscomp.parsing.ParserRunner;
22     import com.google.javascript.jscomp.parsing.parser.FeatureSet;
23     import com.google.javascript.rhino.ErrorReporter;
24     import com.google.javascript.rhino.IR;
25     import com.google.javascript.rhino.InputId;
26     import com.google.javascript.rhino.Node;
27     
28     import java.io.IOException;
29     import java.util.ArrayList;
30     
31     /**
32      * Generates an AST for a JavaScript source file.
33      *
34      */
35     public class JsAst implements SourceAst {
36       private static final long serialVersionUID = 1L;
37     
38       private transient InputId inputId;
39       private transient SourceFile sourceFile;
40       private String fileName;
41       private Node root;
42       private FeatureSet features;
43     
44       public JsAst(SourceFile sourceFile) {
45         this.inputId = new InputId(sourceFile.getName());
46         this.sourceFile = sourceFile;
47         this.fileName = sourceFile.getName();
48       }
49     
50       @Override
51       public Node getAstRoot(AbstractCompiler compiler) {
52         if (root == null) {
53           parse(compiler);
54           root.setInputId(inputId);
55         }
56         return root;
57       }
58     
59       @Override
60       public void clearAst() {
61         root = null;
62         // While we're at it, clear out any saved text in the source file on
63         // the assumption that if we're dumping the parse tree, then we probably
64         // assume regenerating everything else is a smart idea also.
65         sourceFile.clearCachedSource();
66       }
67     
68       @Override
69       public InputId getInputId() {
70         return inputId;
71       }
72     
73       @Override
74       public SourceFile getSourceFile() {
75         return sourceFile;
76       }
77     
78       @Override
79       public void setSourceFile(SourceFile file) {
80         Preconditions.checkState(fileName.equals(file.getName()));
81         sourceFile = file;
82       }
83     
84       public FeatureSet getFeatures(AbstractCompiler compiler) {
85         getAstRoot(compiler); // parse if required
86         return features;
87       }
88     
89       public static class RhinoError {
90         public final String message;
91         public final String sourceName;
92         public final int line;
93         public final int lineOffset;
94     
95         public RhinoError(String message, String sourceName, int line, int lineOffset) {
96           this.message = message;
97           this.sourceName = sourceName;
98           this.line = line;
99           this.lineOffset = lineOffset;
100         }
101       }
102     
103       /** Simple class to share parse results between compilation jobs */
104       public static class ParseResult {
105         public final ImmutableList<RhinoError> errors;
106         public final ImmutableList<RhinoError> warnings;
107         ParseResult(ImmutableList<RhinoError> errors, ImmutableList<RhinoError> warnings) {
108           this.errors = errors;
109           this.warnings = warnings;
110         }
111       }
112     
113       private static class RecordingReporterProxy implements ErrorReporter {
114         final ArrayList<RhinoError> errors = new ArrayList<>();
115         final ArrayList<RhinoError> warnings = new ArrayList<>();
116         private ErrorReporter delegateReporter;
117     
118         RecordingReporterProxy(ErrorReporter delegateReporter) {
119           this.delegateReporter = delegateReporter;
120         }
121     
122         @Override
123         public void warning(String message, String sourceName, int line, int lineOffset) {
124           warnings.add(new RhinoError(message, sourceName, line, lineOffset));
125           delegateReporter.warning(message, sourceName, line, lineOffset);
126         }
127     
128         @Override
129         public void error(String message, String sourceName, int line, int lineOffset) {
130           errors.add(new RhinoError(message, sourceName, line, lineOffset));
131           delegateReporter.error(message, sourceName, line, lineOffset);
132         }
133       }
134     
135       private void parse(AbstractCompiler compiler) {
136         ErrorManager errorManager = compiler.getErrorManager();
137         int startErrorCount = errorManager.getErrorCount();
138     
139         RecordingReporterProxy reporter = new RecordingReporterProxy(
140             compiler.getDefaultErrorReporter());
141     
142         try {
143           ParserRunner.ParseResult result = ParserRunner.parse(
144               sourceFile,
145               sourceFile.getCode(),
146               compiler.getParserConfig(sourceFile.isExtern()
147                             ? AbstractCompiler.ConfigContext.EXTERNS
148                             : AbstractCompiler.ConfigContext.DEFAULT),
149               reporter);
150           root = result.ast;
151           features = result.features;
152     
153           if (compiler.getOptions().preservesDetailedSourceInfo()) {
154             compiler.addComments(sourceFile.getName(), result.comments);
155           }
156         } catch (IOException e) {
157           compiler.report(
158               JSError.make(AbstractCompiler.READ_ERROR, sourceFile.getName()));
159         }
160     
161         if (root == null
162             // Most passes try to report as many errors as possible,
163             // so there may already be errors. We only care if there were
164             // errors in the code we just parsed.
165             // Note: we use the ErrorManager here rather than the ErrorReporter as
166             // we don't want to fail if the error was excluded by a warning guard, conversely
167             // we do want to fail if a warning was promoted to an error.
168             || (errorManager.getErrorCount() > startErrorCount
169                 && !compiler.getOptions().canContinueAfterErrors())) {
170           // There was a parse error or IOException, so use a dummy block.
171     
172     
173           root = IR.script();
174         } else {
175           compiler.prepareAst(root);
176         }
177     
178         if (reporter.errors.size() > 0 || reporter.warnings.size() > 0) {
179           ParseResult result = new ParseResult(
180               ImmutableList.copyOf(reporter.errors),
181               ImmutableList.copyOf(reporter.warnings));
182           root.putProp(Node.PARSE_RESULTS, result);
183         }
184     
185         // Set the source name so that the compiler passes can track
186         // the source file and module.
187         root.setStaticSourceFile(sourceFile);
188       }
189     }
