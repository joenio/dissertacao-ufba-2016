103       @Test
104       public void testApplySuggestedFixes() throws Exception {
105         String code = "var someNode;";
106         Compiler compiler = getCompiler(code);
107         Node root = compileToScriptRoot(compiler);
108         List<SuggestedFix> fixes = ImmutableList.of(new SuggestedFix.Builder().delete(root).build());
109         Map<String, String> codeMap = ImmutableMap.of("test", code);
110         Map<String, String> newCodeMap = ApplySuggestedFixes.applySuggestedFixesToCode(

113         assertThat(newCodeMap).containsEntry("test", "");

116       @Test
117       public void testApplySuggestedFixes_insideJSDoc() throws Exception {
118         String code = "/** @type {Foo} */\nvar foo = new Foo()";
119         Compiler compiler = getCompiler(code);
120         Node root = compileToScriptRoot(compiler);
121         Node varNode = root.getFirstChild();
122         Node jsdocRoot =
123             Iterables.getOnlyElement(varNode.getJSDocInfo().getTypeNodes());
124         SuggestedFix fix = new SuggestedFix.Builder()
125             .insertBefore(jsdocRoot, "!")
126             .build();
127         List<SuggestedFix> fixes = ImmutableList.of(fix);
128         Map<String, String> codeMap = ImmutableMap.of("test", code);
129         Map<String, String> newCodeMap = ApplySuggestedFixes.applySuggestedFixesToCode(
130             fixes, codeMap);
131         assertThat(newCodeMap).hasSize(1);
132         assertThat(newCodeMap).containsEntry("test", "/** @type {!Foo} */\nvar foo = new Foo()");
133       }

137         String code = "/** @type {Array<Foo>} */\nvar arr = [new Foo()];";
138         Compiler compiler = getCompiler(code);
139         Node root = compileToScriptRoot(compiler);
140         Node varNode = root.getFirstChild();
141         Node jsdocRoot =
142             Iterables.getOnlyElement(varNode.getJSDocInfo().getTypeNodes());
143         SuggestedFix fix1 = new SuggestedFix.Builder()
144             .insertBefore(jsdocRoot, "!")
145             .build();
146         Node foo = jsdocRoot.getFirstFirstChild();
147         SuggestedFix fix2 = new SuggestedFix.Builder()
148             .insertBefore(foo, "!")
149             .build();
150         List<SuggestedFix> fixes = ImmutableList.of(fix1, fix2);
151         Map<String, String> codeMap = ImmutableMap.of("test", code);
152         Map<String, String> newCodeMap = ApplySuggestedFixes.applySuggestedFixesToCode(
153             fixes, codeMap);
154         assertThat(newCodeMap).hasSize(1);
155         assertThat(newCodeMap)

184       /**
185        * Returns the root script node produced from the compiled JS input.
186        */
187       private Node compileToScriptRoot(Compiler compiler) {
188         Node root = compiler.getRoot();
189         // The last child of the compiler root is a Block node, and the first child
190         // of that is the Script node.
191         return root.getLastChild().getFirstChild();
192       }
193     
194       private Compiler getCompiler(String jsInput) {
195         Compiler compiler = new Compiler();
196         CompilerOptions options = RefactoringDriver.getCompilerOptions();
197         compiler.init(
198             ImmutableList.<SourceFile>of(), // Externs
199             ImmutableList.of(SourceFile.fromCode("test", jsInput)),
200             options);
201         compiler.parse();
202         return compiler;
203       }
