127       private static CompilerOptions.LanguageMode parseLanguageMode(String value) {
128         switch (value) {
129           case "ECMASCRIPT6_STRICT":
130           case "ES6_STRICT":
131             return CompilerOptions.LanguageMode.ECMASCRIPT6_STRICT;
132           case "ECMASCRIPT6":
133           case "ES6":

135           case "ECMASCRIPT5_STRICT":
136           case "ES5_STRICT":
137             return CompilerOptions.LanguageMode.ECMASCRIPT5_STRICT;
138           case "ECMASCRIPT5":

143             return CompilerOptions.LanguageMode.ECMASCRIPT3;
144           default:

366       @Override
367       public void execute() {
368         if (this.outputFile == null) {
369           throw new BuildException("outputFile attribute must be set");
370         }
371     
372         Compiler.setLoggingLevel(Level.OFF);
373     
374         CompilerOptions options = createCompilerOptions();
375         Compiler compiler = createCompiler(options);
376     
377         List<SourceFile> externs = findExternFiles(options);
378         List<SourceFile> sources = findSourceFiles();
379     
380         if (isStale() || forceRecompile) {
381           log("Compiling " + sources.size() + " file(s) with " +
382               externs.size() + " extern(s)");
383     
384           Result result = compiler.compile(externs, sources, options);
385     
386           if (result.success) {
387             StringBuilder source = new StringBuilder(compiler.toSource());
388     
389             if (this.outputWrapperFile != null) {
390               try {
391                 this.outputWrapper = Files.toString(this.outputWrapperFile, UTF_8);
392               } catch (Exception e) {
393                 throw new BuildException("Invalid output_wrapper_file specified.");
394               }
395             }
396     
397             if (this.outputWrapper != null) {
398               int pos = -1;
399               pos = this.outputWrapper.indexOf(CommandLineRunner.OUTPUT_MARKER);
400               if (pos > -1) {
401                 String prefix = this.outputWrapper.substring(0, pos);
402                 source.insert(0, prefix);
403     
404                 // end of outputWrapper
405                 int suffixStart = pos + CommandLineRunner.OUTPUT_MARKER.length();
406                 String suffix = this.outputWrapper.substring(suffixStart);
407                 source.append(suffix);
408               } else {
409                 throw new BuildException("Invalid output_wrapper specified. " +
410                     "Missing '" + CommandLineRunner.OUTPUT_MARKER + "'.");
411               }
412             }
413     
414             if (result.sourceMap != null) {
415               flushSourceMap(result.sourceMap);
416               source.append(System.getProperty("line.separator"));
417               source.append("//# sourceMappingURL=" + sourceMapOutputFile.getName());
418             }
419             writeResult(source.toString());
420           } else {
421             throw new BuildException("Compilation failed.");
422           }
423         } else {
424           log("None of the files changed. Compilation skipped.");
425         }
426       }

438       private CompilerOptions createCompilerOptions() {
439         CompilerOptions options = new CompilerOptions();
440     
441         this.compilationLevel.setOptionsForCompilationLevel(options);
442         if (this.debugOptions) {
443           this.compilationLevel.setDebugOptionsForCompilationLevel(options);
444         }
445     
446         options.setEnvironment(this.environment);
447     
448         options.setPrettyPrint(this.prettyPrint);
449         options.setPrintInputDelimiter(this.printInputDelimiter);
450         options.setPreferSingleQuotes(this.preferSingleQuotes);
451         options.setGenerateExports(this.generateExports);
452     
453         options.setLanguageIn(this.languageIn);
454         options.setLanguageOut(this.languageOut);
455         options.setOutputCharset(this.outputEncoding);
456     
457         this.warningLevel.setOptionsForWarningLevel(options);
458         options.setManageClosureDependencies(manageDependencies);
459         convertEntryPointParameters(options);
460         options.setTrustedStrings(true);
461         options.setAngularPass(angularPass);
462     
463         if (replaceProperties) {
464           convertPropertiesMap(options);
465         }
466     
467         convertDefineParameters(options);
468     
469         for (Warning warning : warnings) {
470           CheckLevel level = warning.getLevel();
471           String groupName = warning.getGroup();
472           DiagnosticGroup group = new DiagnosticGroups().forName(groupName);
473           if (group == null) {
474             throw new BuildException(
475                 "Unrecognized 'warning' option value (" + groupName + ")");
476           }
477           options.setWarningLevel(group, level);
478         }
479     
480         if (!Strings.isNullOrEmpty(sourceMapFormat)) {
481           options.setSourceMapFormat(Format.valueOf(sourceMapFormat));
482         }
483     
484         if (!Strings.isNullOrEmpty(sourceMapLocationMapping)) {
485           String[] tokens = sourceMapLocationMapping.split("\\|", -1);
486           LocationMapping lm = new LocationMapping(tokens[0], tokens[1]);
487           options.setSourceMapLocationMappings(Arrays.asList(lm));
488         }
489     
490         if (sourceMapOutputFile != null) {
491           File parentFile = sourceMapOutputFile.getParentFile();
492           if (parentFile.mkdirs()) {
493             log("Created missing parent directory " + parentFile, Project.MSG_DEBUG);
494           }
495           options.setSourceMapOutputPath(parentFile.getAbsolutePath());
496         }
497         return options;
498       }

520       /**
521        * Converts {@code <define/>} nested elements into Compiler {@code @define}
522        * replacements. Note: unlike project properties, {@code <define/>} elements
523        * do not need to be named starting with the replacement prefix.
524        */
525       private void convertDefineParameters(CompilerOptions options) {
526         for (Parameter p : defineParams) {
527           String key = p.getName();
528           Object value = p.getValue();
529     
530           if (!setDefine(options, key, value)) {
531             log("Unexpected @define value for name=" + key + "; value=" + value);
532           }
533         }
534       }

536       /**
537        * Converts {@code <entrypoint/>} nested elements into Compiler entrypoint
538        * replacements.
539        */
540       private void convertEntryPointParameters(CompilerOptions options) {
541         List<String> entryPoints = new LinkedList<>();
542         for (Parameter p : entryPointParams) {
543           String key = p.getName();
544           entryPoints.add(key);
545         }
546         if (this.manageDependencies) {
547           options.setManageClosureDependencies(entryPoints);
548         }

551       /**
552        * Converts project properties beginning with the replacement prefix
553        * into Compiler {@code @define} replacements.
554        *
555        * @param options
556        */
557       private void convertPropertiesMap(CompilerOptions options) {
558         @SuppressWarnings("unchecked")
559         Map<String, Object> props = getProject().getProperties();
560         for (Map.Entry<String, Object> entry : props.entrySet()) {
561           String key = entry.getKey();
562           Object value = entry.getValue();
563     
564           if (key.startsWith(replacePropertiesPrefix)) {
565             key = key.substring(replacePropertiesPrefix.length());
566     
567             if (!setDefine(options, key, value)) {
568               log("Unexpected property value for key=" + key + "; value=" + value);
569             }
570           }
571         }
572       }

574       /**
575        * Maps Ant-style values (e.g., from Properties) into expected
576        * Closure {@code @define} literals
577        *
578        * @return True if the {@code @define} replacement succeeded, false if
579        *         the variable's value could not be mapped properly.
580        */
581       private boolean setDefine(CompilerOptions options,
582           String key, Object value) {
583         boolean success = false;
584     
585         if (value instanceof String) {
586           final boolean isTrue = "true".equals(value);
587           final boolean isFalse = "false".equals(value);
588     
589           if (isTrue || isFalse) {
590             options.setDefineToBooleanLiteral(key, isTrue);
591           } else {
592             try {
593               double dblTemp = Double.parseDouble((String) value);
594               options.setDefineToDoubleLiteral(key, dblTemp);
595             } catch (NumberFormatException nfe) {
596               // Not a number, assume string
597               options.setDefineToStringLiteral(key, (String) value);
598             }
599           }
600     
601           success = true;
602         } else if (value instanceof Boolean) {
603           options.setDefineToBooleanLiteral(key, (Boolean) value);
604           success = true;
605         } else if (value instanceof Integer) {
606           options.setDefineToNumberLiteral(key, (Integer) value);
607           success = true;
608         } else if (value instanceof Double) {
609           options.setDefineToDoubleLiteral(key, (Double) value);
610           success = true;
611         }
612     
613         return success;
614       }

626         List<SourceFile> files = new LinkedList<>();
627         files.addAll(getBuiltinExterns(options));

636       private List<SourceFile> findSourceFiles() {
637         List<SourceFile> files = new LinkedList<>();
638     
639         for (FileList list : this.sourceFileLists) {
640           files.addAll(findJavaScriptFiles(list));
641         }

643         for (Path list : this.sourcePaths) {
644           files.addAll(findJavaScriptFiles(list));
645         }

647         return files;

650       /**
651        * Translates an Ant resource collection into the file list format that
652        * the compiler expects.
653        */
654       private List<SourceFile> findJavaScriptFiles(ResourceCollection rc) {
655         List<SourceFile> files = new LinkedList<>();
656         Iterator<Resource> iter = rc.iterator();
657         while (iter.hasNext()) {
658           FileResource fr = (FileResource) iter.next();
659           // Construct path to file, relative to current working directory.
660           File file = Paths.get("")
661               .toAbsolutePath()
662               .relativize(fr.getFile().toPath())
663               .toFile();
664           files.add(SourceFile.fromFile(file, Charset.forName(encoding)));
665         }
666         return files;
667       }

702       /**
703        * Determine if compilation must actually happen, i.e. if any input file
704        * (extern or source) has changed after the outputFile was last modified.
705        *
706        * @return true if compilation should happen
707        */
708       private boolean isStale() {
709         long lastRun = outputFile.lastModified();
710         long sourcesLastModified = Math.max(
711             getLastModifiedTime(this.sourceFileLists),
712             getLastModifiedTime(this.sourcePaths));
713         long externsLastModified = getLastModifiedTime(this.externFileLists);
714     
715         return lastRun <= sourcesLastModified || lastRun <= externsLastModified;
716       }

718       /**
719        * Returns the most recent modified timestamp of the file collection.
720        *
721        * Note: this must be combined into one method to account for both
722        * Path and FileList erasure types.
723        *
724        * @param fileLists Collection of FileList or Path
725        * @return Most recent modified timestamp
726        */
727       private long getLastModifiedTime(List<?> fileLists) {
728         long lastModified = 0;
729     
730         for (Object entry : fileLists) {
731           if (entry instanceof FileList) {
732             FileList list = (FileList) entry;
733     
734             for (String fileName : list.getFiles(this.getProject())) {
735               File path = list.getDir(this.getProject());
736               File file = new File(path, fileName);
737               lastModified = Math.max(getLastModifiedTime(file), lastModified);
738             }
739           } else if (entry instanceof Path) {
740             Path path = (Path) entry;
741             for (String src : path.list()) {
742               File file = new File(src);
743               lastModified = Math.max(getLastModifiedTime(file), lastModified);
744             }
745           }
746         }
747     
748         return lastModified;
749       }
750     
751       /**
752        * Returns the last modified timestamp of the given File.
753        */
754       private static long getLastModifiedTime(File file) {
755         long fileLastModified = file.lastModified();
756         // If the file is absent, we don't know if it changed (maybe was deleted),
757         // so assume it has just changed.
758         if (fileLastModified == 0) {
759           fileLastModified = new Date().getTime();
760         }
761         return fileLastModified;
762       }
