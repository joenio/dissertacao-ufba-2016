338       /**
339        * When the CompilerOptions and its WarningsGuard overlap, reconcile
340        * any discrepencies.
341        */
342       protected void reconcileOptionsWithGuards() {
343         // DiagnosticGroups override the plain checkTypes option.
344         if (options.enables(DiagnosticGroups.CHECK_TYPES)) {
345           options.checkTypes = true;
346         } else if (options.disables(DiagnosticGroups.CHECK_TYPES)) {
347           options.checkTypes = false;
348         } else if (!options.checkTypes) {
349           // If DiagnosticGroups did not override the plain checkTypes
350           // option, and checkTypes is enabled, then turn off the
351           // parser type warnings.
352           options.setWarningLevel(
353               DiagnosticGroup.forType(
354                   RhinoErrorReporter.TYPE_PARSE_ERROR),
355               CheckLevel.OFF);
356         }
357         // With NTI, we still need OTI to run because the later passes that use
358         // types only understand OTI types at the moment.
359         // But we do not want to see the warnings from OTI.
360         if (options.getNewTypeInference() && options.getRunOTIAfterNTI()) {
361           options.checkTypes = true;
362           if (!options.reportOTIErrorsUnderNTI) {
363             options.setWarningLevel(
364                 DiagnosticGroups.OLD_CHECK_TYPES,
365                 CheckLevel.OFF);
366             options.setWarningLevel(
367                 DiagnosticGroups.OLD_REPORT_UNKNOWN_TYPES,
368                 CheckLevel.OFF);
369             options.setWarningLevel(
370                 FunctionTypeBuilder.ALL_DIAGNOSTICS,
371                 CheckLevel.OFF);
372           }
373           options.setWarningLevel(
374               DiagnosticGroup.forType(RhinoErrorReporter.TYPE_PARSE_ERROR),
375               CheckLevel.WARNING);
376         }
377     
378         if (options.checkGlobalThisLevel.isOn() &&
379             !options.disables(DiagnosticGroups.GLOBAL_THIS)) {
380           options.setWarningLevel(
381               DiagnosticGroups.GLOBAL_THIS,
382               options.checkGlobalThisLevel);
383         }
384     
385         if (options.getLanguageIn().isStrict()) {
386           options.setWarningLevel(
387               DiagnosticGroups.ES5_STRICT,
388               CheckLevel.ERROR);
389         }
390     
391         // All passes must run the variable check. This synthesizes
392         // variables later so that the compiler doesn't crash. It also
393         // checks the externs file for validity. If you don't want to warn
394         // about missing variable declarations, we shut that specific
395         // error off.
396         if (!options.checkSymbols &&
397             !options.enables(DiagnosticGroups.CHECK_VARIABLES)) {
398           options.setWarningLevel(
399               DiagnosticGroups.CHECK_VARIABLES, CheckLevel.OFF);
400         }
401       }

424       /**
425        * Initializes the instance state needed for a compile job if the sources
426        * are in modules.
427        */
428       public <T extends SourceFile> void initModules(
429           List<T> externs, List<JSModule> modules, CompilerOptions options) {
430         initOptions(options);
431     
432         checkFirstModule(modules);
433         fillEmptyModules(modules);
434     
435         this.externs = makeCompilerInput(externs, true);
436     
437         // Generate the module graph, and report any errors in the module
438         // specification as errors.
439         this.modules = modules;
440         if (modules.size() > 1) {
441           try {
442             this.moduleGraph = new JSModuleGraph(modules);
443           } catch (JSModuleGraph.ModuleDependenceException e) {
444             // problems with the module format.  Report as an error.  The
445             // message gives all details.
446             report(JSError.make(MODULE_DEPENDENCY_ERROR,
447                     e.getModule().getName(), e.getDependentModule().getName()));
448             return;
449           }
450         } else {
451           this.moduleGraph = null;
452         }
453     
454         this.inputs = getAllInputsFromModules(modules);
455         initBasedOnOptions();
456     
457         initInputsByIdMap();
458     
459         initAST();
460       }

542       /**
543        * Builds a single list of all module inputs. Verifies that it contains no
544        * duplicates.
545        */
546       private static List<CompilerInput> getAllInputsFromModules(
547           List<JSModule> modules) {
548         List<CompilerInput> inputs = new ArrayList<>();
549         Map<String, JSModule> inputMap = new HashMap<>();
550         for (JSModule module : modules) {
551           for (CompilerInput input : module.getInputs()) {
552             String inputName = input.getName();
553     
554             // NOTE(nicksantos): If an input is in more than one module,
555             // it will show up twice in the inputs list, and then we
556             // will get an error down the line.
557             inputs.add(input);
558             inputMap.put(inputName, module);
559           }
560         }
561         return inputs;
562       }

593             report(JSError.make(DUPLICATE_EXTERN_INPUT, input.getName()));

595         }
596         for (CompilerInput input : inputs) {
597           InputId id = input.getInputId();
598           CompilerInput previous = putCompilerInput(id, input);
599           if (previous != null) {
600             report(JSError.make(DUPLICATE_INPUT, input.getName()));
601           }

635             return getResult();
636           }

640           errorManager.generateReport();
641           stopTracer(t, "generateReport");
642         }

700       private void compileInternal() {
701         setProgress(0.0, null);
702         CompilerOptionsPreprocessor.preprocess(options);
703         parse();
704         // 15 percent of the work is assumed to be for parsing (based on some
705         // minimal analysis on big JS projects, of course this depends on options)
706         setProgress(0.15, "parse");
707         if (hasErrors()) {
708           return;
709         }
710     
711         if (!precheck()) {
712           return;
713         }
714     
715         if (options.skipNonTranspilationPasses) {
716           // i.e. whitespace-only mode, which will not work with goog.module without:
717           whitespaceOnlyPasses();
718         }
719     
720         if (!options.skipNonTranspilationPasses || options.lowerFromEs6()) {
721           check();
722           if (hasErrors()) {
723             return;
724           }
725     
726           if (!options.checksOnly && !options.shouldGenerateTypedExterns()) {
727             optimize();
728           }
729         }
730     
731         if (options.recordFunctionInformation) {
732           recordFunctionInformation();
733         }
734     
735         if (options.devMode == DevMode.START_AND_END) {
736           runSanityCheck();
737         }
738         setProgress(1.0, "recordFunctionInformation");
739     
740         if (tracker != null) {
741           tracker.outputTracerReport(outStream == null ? System.out : outStream);
742         }
743       }

801       public void check() {
802         runCustomPasses(CustomPassExecutionTime.BEFORE_CHECKS);
803     
804         // We are currently only interested in check-passes for progress reporting
805         // as it is used for IDEs, that's why the maximum progress is set to 1.0.
806         phaseOptimizer = new PhaseOptimizer(this, tracker,
807             new PhaseOptimizer.ProgressRange(getProgress(), 1.0));
808         if (options.devMode == DevMode.EVERY_PASS) {
809           phaseOptimizer.setSanityCheck(sanityCheck);
810         }
811         if (options.getCheckDeterminism()) {
812           phaseOptimizer.setPrintAstHashcodes(true);
813         }
814         phaseOptimizer.consume(getPassConfig().getChecks());
815         phaseOptimizer.process(externsRoot, jsRoot);
816         if (hasErrors()) {
817           return;
818         }
819     
820         if (options.getTweakProcessing().shouldStrip() ||
821             !options.stripTypes.isEmpty() ||
822             !options.stripNameSuffixes.isEmpty() ||
823             !options.stripTypePrefixes.isEmpty() ||
824             !options.stripNamePrefixes.isEmpty()) {
825           stripCode(options.stripTypes, options.stripNameSuffixes,
826               options.stripTypePrefixes, options.stripNamePrefixes);
827         }
828     
829         runCustomPasses(CustomPassExecutionTime.BEFORE_OPTIMIZATIONS);
830         phaseOptimizer = null;
831       }

1109       /**
1110        * Replace a source input dynamically. Intended for incremental
1111        * re-compilation.
1112        *
1113        * If the new source input doesn't parse, then keep the old input
1114        * in the AST and return false.
1115        *
1116        * @return Whether the new AST was attached successfully.
1117        */
1118       boolean replaceIncrementalSourceAst(JsAst ast) {
1119         CompilerInput oldInput = getInput(ast.getInputId());
1120         Preconditions.checkNotNull(oldInput, "No input to replace: %s", ast.getInputId().getIdName());
1121         Node newRoot = ast.getAstRoot(this);
1122         if (newRoot == null) {
1123           return false;
1124         }
1125     
1126         Node oldRoot = oldInput.getAstRoot(this);
1127         if (oldRoot != null) {
1128           oldRoot.getParent().replaceChild(oldRoot, newRoot);
1129         } else {
1130           getRoot().getLastChild().addChildToBack(newRoot);
1131         }
1132     
1133         CompilerInput newInput = new CompilerInput(ast);
1134         putCompilerInput(ast.getInputId(), newInput);
1135     
1136         JSModule module = oldInput.getModule();
1137         if (module != null) {
1138           module.addAfter(newInput, oldInput);
1139           module.remove(oldInput);
1140         }
1141     
1142         // Verify the input id is set properly.
1143         Preconditions.checkState(
1144             newInput.getInputId().equals(oldInput.getInputId()));
1145         InputId inputIdOnAst = newInput.getAstRoot(this).getInputId();
1146         Preconditions.checkState(newInput.getInputId().equals(inputIdOnAst));
1147     
1148         inputs.remove(oldInput);
1149         return true;
1150       }

1153        * Add a new source input dynamically. Intended for incremental compilation.
1154        * <p>
1155        * If the new source input doesn't parse, it will not be added, and a false
1156        * will be returned.
1157        *
1158        * @param ast the JS Source to add.
1159        * @return true if the source was added successfully, false otherwise.
1160        * @throws IllegalStateException if an input for this ast already exists.
1161        */
1162       boolean addNewSourceAst(JsAst ast) {
1163         CompilerInput oldInput = getInput(ast.getInputId());
1164         if (oldInput != null) {
1165           throw new IllegalStateException(
1166               "Input already exists: " + ast.getInputId().getIdName());
1167         }
1168         Node newRoot = ast.getAstRoot(this);
1169         if (newRoot == null) {
1170           return false;
1171         }
1172     
1173         getRoot().getLastChild().addChildToBack(newRoot);
1174     
1175         CompilerInput newInput = new CompilerInput(ast);
1176     
1177         // TODO(tylerg): handle this for multiple modules at some point.
1178         if (moduleGraph == null && !modules.isEmpty()) {
1179           // singleton module
1180           modules.get(0).add(newInput);
1181         }
1182     
1183         putCompilerInput(ast.getInputId(), newInput);
1184     
1185         return true;
1186       }

1349       /**
1350        * Parses the externs and main inputs.
1351        *
1352        * @return A synthetic root node whose two children are the externs root
1353        *     and the main root
1354        */
1355       Node parseInputs() {
1356         boolean devMode = options.devMode != DevMode.OFF;
1357     
1358         // If old roots exist (we are parsing a second time), detach each of the
1359         // individual file parse trees.
1360         externsRoot.detachChildren();
1361         jsRoot.detachChildren();
1362     
1363         if (options.tracer.isOn()) {
1364           tracker = new PerformanceTracker(jsRoot, options.tracer);
1365           addChangeHandler(tracker.getCodeChangeHandler());
1366         }
1367     
1368         Tracer tracer = newTracer(PARSING_PASS_NAME);
1369         beforePass(PARSING_PASS_NAME);
1370     
1371         try {
1372           // Parse externs sources.
1373           for (CompilerInput input : externs) {
1374             Node n = input.getAstRoot(this);
1375             if (hasErrors()) {
1376               return null;
1377             }
1378             externsRoot.addChildToBack(n);
1379           }
1380     
1381           if (options.lowerFromEs6()) {
1382             processEs6Modules();
1383           }
1384     
1385           // Modules inferred in ProcessCommonJS pass.
1386           if (options.transformAMDToCJSModules || options.processCommonJSModules) {
1387             processAMDAndCommonJSModules();
1388           }
1389     
1390           if (options.lowerFromEs6()
1391               || options.transformAMDToCJSModules
1392               || options.processCommonJSModules) {
1393     
1394             // Build a map of module identifiers for any input which provides no namespace.
1395             // These files could be imported modules which have no exports, but do have side effects.
1396             Map<String, CompilerInput> inputModuleIdentifiers = new HashMap<>();
1397             for (CompilerInput input : inputs) {
1398               if (input.getKnownProvides().isEmpty()) {
1399                 ModuleIdentifier modInfo =
1400                     ModuleIdentifier.forFile(input.getSourceFile().getOriginalPath());
1401     
1402                 inputModuleIdentifiers.put(modInfo.getClosureNamespace(), input);
1403               }
1404             }
1405     
1406             // Find out if any input attempted to import a module that had no exports.
1407             // In this case we must force module rewriting to occur on the imported file
1408             Map<String, CompilerInput> inputsToRewrite = new HashMap<>();
1409             for (CompilerInput input : inputs) {
1410               for (String require : input.getKnownRequires()) {
1411                 if (inputModuleIdentifiers.containsKey(require)
1412                     && !inputsToRewrite.containsKey(require)) {
1413                   inputsToRewrite.put(require, inputModuleIdentifiers.get(require));
1414                 }
1415               }
1416             }
1417     
1418             if (!inputsToRewrite.isEmpty()) {
1419               processEs6Modules(new ArrayList<>(inputsToRewrite.values()), true);
1420             }
1421           }
1422     
1423           orderInputs();
1424     
1425           // If in IDE mode, we ignore the error and keep going.
1426           if (hasErrors()) {
1427             return null;
1428           }
1429     
1430           // Build the AST.
1431           for (CompilerInput input : inputs) {
1432             Node n = input.getAstRoot(this);
1433             if (n == null) {
1434               continue;
1435             }
1436     
1437             if (devMode) {
1438               runSanityCheck();
1439               if (hasErrors()) {
1440                 return null;
1441               }
1442             }
1443     
1444             // TODO(johnlenz): we shouldn't need to check both isExternExportsEnabled and
1445             // externExportsPath.
1446             if (options.sourceMapOutputPath != null ||
1447                 options.nameReferenceReportPath != null ||
1448                 options.isExternExportsEnabled() ||
1449                 options.externExportsPath != null ||
1450                 !options.replaceStringsFunctionDescriptions.isEmpty()) {
1451     
1452               // Annotate the nodes in the tree with information from the
1453               // input file. This information is used to construct the SourceMap.
1454               SourceInformationAnnotator sia =
1455                   new SourceInformationAnnotator(
1456                       input.getName(), options.devMode != DevMode.OFF);
1457               NodeTraversal.traverseEs6(this, n, sia);
1458             }
1459     
1460             jsRoot.addChildToBack(n);
1461           }
1462     
1463           if (hasErrors()) {
1464             return null;
1465           }
1466           return externAndJsRoot;
1467         } finally {

1469           stopTracer(tracer, PARSING_PASS_NAME);
1470         }

1488       void orderInputs() {
1489         hoistExterns();
1490     
1491         // Check if the sources need to be re-ordered.
1492         boolean staleInputs = false;
1493         if (options.dependencyOptions.needsManagement()) {
1494           for (CompilerInput input : inputs) {
1495             // Forward-declare all the provided types, so that they
1496             // are not flagged even if they are dropped from the process.
1497             for (String provide : input.getProvides()) {
1498               getTypeRegistry().forwardDeclareType(provide);
1499             }
1500           }

1504                 (moduleGraph == null ? new JSModuleGraph(modules) : moduleGraph)
1505                 .manageDependencies(options.dependencyOptions, inputs);

1512                 MISSING_MODULE_ERROR, e.getMessage()));
1513           }
1514         }

1516         hoistNoCompileFiles();
1517     
1518         if (staleInputs) {
1519           repartitionInputs();
1520         }
1521       }

1523       /**
1524        * Hoists inputs with the @externs annotation into the externs list.
1525        */
1526       void hoistExterns() {
1527         boolean staleInputs = false;
1528         for (CompilerInput input : inputs) {
1529           if (options.dependencyOptions.needsManagement()) {
1530             // If we're doing scanning dependency info anyway, use that
1531             // information to skip sources that obviously aren't externs.
1532             if (!input.getProvides().isEmpty() || !input.getRequires().isEmpty()) {
1533               continue;
1534             }
1535           }
1536     
1537           Node n = input.getAstRoot(this);
1538     
1539           // Inputs can have a null AST on a parse error.
1540           if (n == null) {
1541             continue;
1542           }
1543     
1544           JSDocInfo info = n.getJSDocInfo();
1545           if (info != null && info.isExterns()) {
1546             // If the input file is explicitly marked as an externs file, then
1547             // assume the programmer made a mistake and throw it into
1548             // the externs pile anyways.
1549             externsRoot.addChildToBack(n);
1550             input.setIsExtern(true);
1551     
1552             input.getModule().remove(input);
1553     
1554             externs.add(input);
1555             staleInputs = true;
1556           }
1557         }
1558     
1559         if (staleInputs) {
1560           repartitionInputs();
1561         }
1562       }

1564       /**
1565        * Hoists inputs with the @nocompile annotation out of the inputs.
1566        */
1567       void hoistNoCompileFiles() {
1568         boolean staleInputs = false;
1569         for (CompilerInput input : inputs) {
1570           Node n = input.getAstRoot(this);
1571     
1572           // Inputs can have a null AST on a parse error.
1573           if (n == null) {

1577           JSDocInfo info = n.getJSDocInfo();
1578           if (info != null && info.isNoCompile()) {
1579             input.getModule().remove(input);
1580             staleInputs = true;
1581           }
1582         }
1583     
1584         if (staleInputs) {
1585           repartitionInputs();
1586         }
1587       }

1804       /**
1805        * Writes out JS code from a root node. If printing input delimiters, this
1806        * method will attach a comment to the start of the text indicating which
1807        * input the output derived from. If there were any preserve annotations
1808        * within the root's source, they will also be printed in a block comment
1809        * at the beginning of the output.
1810        */
1811       public void toSource(final CodeBuilder cb,
1812                            final int inputSeqNum,
1813                            final Node root) {
1814         runInCompilerThread(new Callable<Void>() {
1815           @Override
1816           public Void call() throws Exception {
1817             if (options.printInputDelimiter) {
1818               if ((cb.getLength() > 0) && !cb.endsWith("\n")) {
1819                 cb.append("\n");  // Make sure that the label starts on a new line
1820               }
1821               Preconditions.checkState(root.isScript());
1822     
1823               String delimiter = options.inputDelimiter;
1824     
1825               String inputName = root.getInputId().getIdName();
1826               String sourceName = root.getSourceFileName();
1827               Preconditions.checkState(sourceName != null);
1828               Preconditions.checkState(!sourceName.isEmpty());
1829     
1830               delimiter = delimiter
1831                   .replaceAll("%name%", Matcher.quoteReplacement(inputName))
1832                   .replaceAll("%num%", String.valueOf(inputSeqNum));
1833     
1834               cb.append(delimiter)
1835                 .append("\n");
1836             }
1837             if (root.getJSDocInfo() != null) {
1838               String license = root.getJSDocInfo().getLicense();
1839               if (license != null && cb.addLicense(license)) {
1840                 cb.append("/*\n")
1841                   .append(license)
1842                   .append("*/\n");
1843               }
1844             }
1845     
1846             // If there is a valid source map, then indicate to it that the current
1847             // root node's mappings are offset by the given string builder buffer.
1848             if (options.sourceMapOutputPath != null) {
1849               sourceMap.setStartingPosition(
1850                   cb.getLineIndex(), cb.getColumnIndex());
1851             }
1852     
1853             // if LanguageMode is strict, only print 'use strict'
1854             // for the first input file
1855             String code = toSource(root, sourceMap, inputSeqNum == 0);
1856             if (!code.isEmpty()) {
1857               cb.append(code);
1858     
1859               // In order to avoid parse ambiguity when files are concatenated
1860               // together, all files should end in a semi-colon. Do a quick
1861               // heuristic check if there's an obvious semi-colon already there.
1862               int length = code.length();
1863               char lastChar = code.charAt(length - 1);
1864               char secondLastChar = length >= 2 ?
1865                   code.charAt(length - 2) : '\0';
1866               boolean hasSemiColon = lastChar == ';' ||
1867                   (lastChar == '\n' && secondLastChar == ';');
1868               if (!hasSemiColon) {
1869                 cb.append(";");
1870               }
1871             }
1872             return null;
1873           }
1874         });
1875       }

1899       /**
1900        * Stores a buffer of text to which more can be appended.  This is just like a
1901        * StringBuilder except that we also track the number of lines.
1902        */
1903       public static class CodeBuilder {
1904         private final StringBuilder sb = new StringBuilder();
1905         private int lineCount = 0;
1906         private int colCount = 0;
1907         private final Set<String> uniqueLicenses = new HashSet<>();
1908     
1909         /** Removes all text, but leaves the line count unchanged. */
1910         void reset() {
1911           sb.setLength(0);
1912         }
1913     
1914         /** Appends the given string to the text buffer. */
1915         CodeBuilder append(String str) {
1916           sb.append(str);
1917     
1918           // Adjust the line and column information for the new text.
1919           int index = -1;
1920           int lastIndex = index;
1921           while ((index = str.indexOf('\n', index + 1)) >= 0) {
1922             ++lineCount;
1923             lastIndex = index;
1924           }
1925     
1926           if (lastIndex == -1) {
1927             // No new lines, append the new characters added.
1928             colCount += str.length();
1929           } else {
1930             colCount = str.length() - (lastIndex + 1);
1931           }
1932     
1933           return this;
1934         }
1935     
1936         /** Returns all text in the text buffer. */
1937         @Override
1938         public String toString() {
1939           return sb.toString();
1940         }
1941     
1942         /** Returns the length of the text buffer. */
1943         public int getLength() {
1944           return sb.length();
1945         }
1946     
1947         /** Returns the (zero-based) index of the last line in the text buffer. */
1948         int getLineIndex() {
1949           return lineCount;
1950         }
1951     
1952         /** Returns the (zero-based) index of the last column in the text buffer. */
1953         int getColumnIndex() {
1954           return colCount;
1955         }
1956     
1957         /** Determines whether the text ends with the given suffix. */
1958         boolean endsWith(String suffix) {
1959           return (sb.length() > suffix.length())
1960               && suffix.equals(sb.substring(sb.length() - suffix.length()));
1961         }
1962     
1963         /** Adds a license and returns whether it is unique (has yet to be encountered). */
1964         boolean addLicense(String license) {
1965           return uniqueLicenses.add(license);
1966         }
1967       }

1974         List<PassFactory> optimizations = getPassConfig().getOptimizations();
1975         if (optimizations.isEmpty()) {
1976           return;
1977         }
1978     
1979         // Ideally, this pass should be the first pass run, however:
1980         // 1) VariableReferenceCheck reports unexpected warnings if Normalize
1981         // is done first.

1983         // unmodified local names.
1984         normalize();

1997           phaseOptimizer.setPrintAstHashcodes(true);
1998         }
1999         phaseOptimizer.consume(optimizations);
2000         phaseOptimizer.process(externsRoot, jsRoot);
2001         phaseOptimizer = null;
2002       }

2616       @Override
2617       Node ensureLibraryInjected(String resourceName, boolean force) {
2618         boolean doNotInject =
2619             !force && (options.skipNonTranspilationPasses || options.preventLibraryInjection);
2620         if (injectedLibraries.containsKey(resourceName) || doNotInject) {
2621           return lastInjectedLibrary;
2622         }
2623     
2624         // Load/parse the code.
2625         String originalCode = ResourceLoader.loadTextResource(
2626             Compiler.class, "js/" + resourceName + ".js");
2627         Node ast = parseSyntheticCode(" [synthetic:" + resourceName + "] ", originalCode);
2628     
2629         // Look for string literals of the form 'require foo bar' or 'externs baz' or 'normalize'.
2630         // As we process each one, remove it from its parent.
2631         for (Node node = ast.getFirstChild();
2632              node != null && node.isExprResult() && node.getFirstChild().isString();
2633              node = ast.getFirstChild()) {
2634           String directive = node.getFirstChild().getString();
2635           List<String> words = Splitter.on(' ').splitToList(directive);
2636           switch (words.get(0)) {
2637             case "use":
2638               // 'use strict' is ignored (and deleted).
2639               break;
2640             case "require":
2641               // 'require lib1 lib2'; pulls in the named libraries before this one.
2642               for (String dependency : words.subList(1, words.size())) {
2643                 ensureLibraryInjected(dependency, force);
2644               }
2645               break;
2646             case "declare":
2647               // 'declare name1 name2'; adds the names to the externs (with no type information).
2648               // Note that we could simply add the entire externs library, but that leads to
2649               // potentially-surprising behavior when the externs that are present depend on
2650               // whether or not a polyfill is used.
2651               for (String extern : words.subList(1, words.size())) {
2652                 getSynthesizedExternsInputAtEnd()
2653                     .getAstRoot(this)
2654                     .addChildToBack(IR.var(IR.name(extern)));
2655               }
2656               break;
2657             default:
2658               throw new RuntimeException("Bad directive: " + directive);
2659           }
2660           ast.removeChild(node);
2661         }
2662     
2663         // If we've already started optimizations, then we need to normalize this.
2664         if (getLifeCycleStage().isNormalized()) {
2665           Normalize.normalizeSyntheticCode(this, ast, "jscomp_" + resourceName + "_");
2666         }
2667     
2668         // Insert the code immediately after the last-inserted runtime library.
2669         Node firstChild = ast.removeChildren();
2670         if (firstChild == null) {
2671           // Handle require-only libraries.
2672           return lastInjectedLibrary;
2673         }
2674         Node lastChild = firstChild.getLastSibling();
2675         Node parent = getNodeForCodeInsertion(null);
2676         if (lastInjectedLibrary == null) {
2677           parent.addChildrenToFront(firstChild);
2678         } else {
2679           parent.addChildrenAfter(firstChild, lastInjectedLibrary);
2680         }
2681         lastInjectedLibrary = lastChild;
2682         injectedLibraries.put(resourceName, lastChild);
2683     
2684         reportCodeChange();
2685         return lastChild;
2686       }
