40     package com.google.javascript.rhino;
41     
42     import java.util.List;
43     
44     /**
45      * @author blickly@google.com (Ben Lickly)
46      * @author dimvar@google.com (Dimitris Vardoulakis)
47      */
48     public interface FunctionTypeI extends TypeI {
49       /**
50        * Creates a new function type B based on the original function type A.
51        * Takes the receiver type (this type) of A and makes it the first
52        * argument of B; B has no receiver type.
53        * Used by the DevirtualizePrototypeMethods pass.
54        */
55       TypeI convertMethodToFunction();
56     
57       ObjectTypeI getInstanceType();
58     
59       TypeI getReturnType();
60     
61       String getReferenceName();
62     
63       /** Gets the AST Node where this function was defined. */
64       Node getSource();
65     
66       /**
67        * Returns a list of types that are subtypes of this type. This is only
68        * valid for constructor functions, and may be null. This allows a downward
69        * traversal of the subtype graph.
70        */
71       List<? extends FunctionTypeI> getSubTypes();
72     
73       /** Gets the type of {@code this} in this function. */
74       TypeI getTypeOfThis();
75     
76       boolean hasProperties();
77     
78       void setSource(Node n);
79     }
