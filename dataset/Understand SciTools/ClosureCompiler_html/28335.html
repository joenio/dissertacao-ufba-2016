17     package com.google.javascript.jscomp;
18     
19     import com.google.common.base.Preconditions;
20     import com.google.javascript.jscomp.JsAst.ParseResult;
21     import com.google.javascript.jscomp.JsAst.RhinoError;
22     import com.google.javascript.rhino.ErrorReporter;
23     import com.google.javascript.rhino.InputId;
24     import com.google.javascript.rhino.Node;
25     
26     /**
27      * An implementation of {@link SourceAst} that avoids re-creating the AST
28      * unless it was manually cleared.  This creates a single defensive copy of the
29      * AST; however, it is not safe for multiple compilations to use this
30      * simultaneously, as all compilations mutate this.  Since this class copies
31      * the tree, you instead should create a central RecoverableJsAst that does the
32      * caching across compilations, and create new RecoverableJsAst's that act as
33      * copying proxies around the original.
34      *
35      */
36     public class RecoverableJsAst implements SourceAst {
37     
38       private static final long serialVersionUID = 1L;
39     
40       // The AST copy that will be kept around.
41       private Node root = null;
42     
43       // This is the actual SourceAst this caching wrapper wraps around.
44       private final SourceAst realSource;
45     
46       private final boolean reportParseErrors;
47     
48       /**
49        * Wraps around an existing SourceAst that provides caching between
50        * compilations.
51        */
52       public RecoverableJsAst(SourceAst realSource, boolean reportParseErrors) {
53         Preconditions.checkNotNull(realSource);
54         this.realSource = realSource;
55         this.reportParseErrors = reportParseErrors;
56       }
57     
58       @Override
59       public synchronized Node getAstRoot(AbstractCompiler compiler) {
60         if (root == null) {
61           // The original source (generally SourceAst) might not be thread-safe;
62           // synchronize on it.
63           synchronized (realSource) {
64             root = realSource.getAstRoot(compiler).cloneTree(true);
65     
66             // Maybe replay parse error
67             JsAst.ParseResult result = (JsAst.ParseResult) root.getProp(Node.PARSE_RESULTS);
68             if (reportParseErrors && result != null) {
69               replay(compiler, result);
70             }
71           }
72         }
73         return root;
74       }
75     
76       private void replay(AbstractCompiler compiler, ParseResult result) {
77         ErrorReporter reporter = compiler.getDefaultErrorReporter();
78         for (RhinoError error : result.errors) {
79           reporter.error(error.message, error.sourceName, error.line, error.lineOffset);
80         }
81         for (RhinoError warning : result.warnings) {
82           reporter.warning(warning.message, warning.sourceName, warning.line, warning.lineOffset);
83         }
84       }
85     
86       @Override
87       public void clearAst() {
88         // Just do a shallow clear; don't re-parse the input.
89         this.root = null;
90       }
91     
92       @Override
93       public InputId getInputId() {
94         synchronized (realSource) {
95           return realSource.getInputId();
96         }
97       }
98     
99       @Override
100       public SourceFile getSourceFile() {
101         synchronized (realSource) {
102           return realSource.getSourceFile();
103         }
104       }
105     
106       @Override
107       public void setSourceFile(SourceFile file) {
108         // Explicitly forbid this operation through this interface; this
109         // RecoverableJsAst is a proxy view only.
110         throw new UnsupportedOperationException();
111       }
112     }
