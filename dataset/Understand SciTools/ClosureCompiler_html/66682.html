16     package com.google.javascript.jscomp;
17     
18     import com.google.common.collect.ImmutableSet;
19     import com.google.javascript.rhino.Node;
20     
21     import java.util.Collection;
22     
23     /**
24      * Disambiguate properties by file, when they are private by naming convention.
25      *
26      * This pass is unsafe. When some code doesn't respect the coding convention,
27      * the pass silently breaks the code.
28      * Projects who use this pass must also turn on CheckAccessControls for code
29      * using the coding convention (by default CheckAccessControls looks only at
30      * jsdoc annotations).
31      *
32      * If someone plans to make this pass non-experimental, or turn it on by
33      * default, they should modify CheckAccessControls to store violations, and use
34      * that list here to back-off renaming, like DisambiguateProperties does.
35      *
36      * Another option is that, in CompilerOptionsPreprocessor, if this pass is
37      * enabled, but the access controls are not, to throw an error.
38      */
39     class DisambiguatePrivateProperties
40        implements NodeTraversal.Callback, CompilerPass {
41     
42       private final AbstractCompiler compiler;
43       private final CodingConvention convention;
44       private final ImmutableSet<String> blacklist;
45       private String fileid;
46       private int id = 0;
47     
48       DisambiguatePrivateProperties(AbstractCompiler compiler) {
49         this.compiler = compiler;
50         this.convention = this.compiler.getCodingConvention();
51         Collection<String> indirect = convention.getIndirectlyDeclaredProperties();
52         blacklist = ImmutableSet.copyOf(indirect);
53       }
54     
55       @Override
56       public void process(Node externs, Node root) {
57         NodeTraversal.traverseEs6(compiler, root, this);
58       }
59     
60       @Override
61       public boolean shouldTraverse(NodeTraversal t, Node n, Node parent) {
62         if (n.isScript()) {
63           this.fileid = "$" + this.id++;
64         }
65         return true;
66       }
67     
68       @Override
69       public void visit(NodeTraversal t, Node n, Node parent) {
70         switch (n.getType()) {
71           case GETPROP:
72             maybeRename(n.getLastChild());
73             break;
74           case STRING_KEY:
75           case GETTER_DEF:
76           case SETTER_DEF:
77           case MEMBER_FUNCTION_DEF:
78             maybeRename(n);
79             break;
80         }
81       }
82     
83       private void maybeRename(Node n) {
84         String prop = n.getString();
85         if (!n.getBooleanProp(Node.QUOTED_PROP) && this.convention.isPrivate(prop)
86             && !blacklist.contains(prop)) {
87           n.setString(prop + fileid);
88           compiler.reportCodeChange();
89         }
90       }
91     }
