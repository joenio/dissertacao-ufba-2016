33       static final DiagnosticType PARSE_ERROR =
34           DiagnosticType.error("JSC_PARSE_ERROR", "Parse error. {0}");
35     
36       static final DiagnosticType TYPE_PARSE_ERROR =
37           DiagnosticType.warning("JSC_TYPE_PARSE_ERROR", "{0}");
38     
39       // This is separate from TYPE_PARSE_ERROR because there are many instances of this warning
40       // and it is unfeasible to fix them all right away.
41       static final DiagnosticType JSDOC_MISSING_BRACES_WARNING =
42           DiagnosticType.disabled("JSC_JSDOC_MISSING_BRACES_WARNING", "{0}");
43     
44       // This is separate from TYPE_PARSE_ERROR because there are many instances of this warning
45       // and it is unfeasible to fix them all right away.
46       static final DiagnosticType JSDOC_MISSING_TYPE_WARNING =
47           DiagnosticType.disabled("JSC_JSDOC_MISSING_TYPE_WARNING", "{0}");
48     
49       static final DiagnosticType TOO_MANY_TEMPLATE_PARAMS =
50           DiagnosticType.disabled("JSC_TOO_MANY_TEMPLATE_PARAMS", "{0}");
51     
52       // Special-cased errors, so that they can be configured via the
53       // warnings API.
54       static final DiagnosticType TRAILING_COMMA =
55           DiagnosticType.error("JSC_TRAILING_COMMA",
56               "Parse error. IE8 (and below) will parse trailing commas in " +
57               "array and object literals incorrectly. " +
58               "If you are targeting newer versions of JS, " +
59               "set the appropriate language_in option.");

62           DiagnosticType.error("JSC_DUPLICATE_PARAM", "Parse error. {0}");

65           DiagnosticType.warning("JSC_INVALID_PARAM", "Parse error. {0}");

71           DiagnosticType.warning("JSC_JSDOC_IN_BLOCK_COMMENT", "Parse error. {0}");

82               "Parse tree too deep.");

85           DiagnosticType.warning("INVALID_OCTAL_LITERAL",
86               "This style of octal literal is not supported in strict mode.");

89           DiagnosticType.error("ES6_FEATURE",

94           DiagnosticType.error("ES6_TYPED",
95               "{0}. Use --language_in=ECMASCRIPT6_TYPED " +

98       static final DiagnosticType MISPLACED_TYPE_SYNTAX =
99           DiagnosticType.error("MISPLACED_TYPE_SYNTAX",
100               "Can only have JSDoc or inline type annotations, not both");
101     
102       // A map of Rhino messages to their DiagnosticType.
103       private final Map<Pattern, DiagnosticType> typeMap;
104     
105       final AbstractCompiler compiler;
106     
107       /**
108        * For each message such as "Not a good use of {0}", replace the place
109        * holder {0} with a wild card that matches all possible strings.
110        * Also put the any non-place-holder in quotes for regex matching later.
111        */
112       private static Pattern replacePlaceHolders(String s) {
113         s = Pattern.quote(s);
114         return Pattern.compile(s.replaceAll("\\{\\d+\\}", "\\\\E.*\\\\Q"));
115       }
116     
117       private RhinoErrorReporter(AbstractCompiler compiler) {
118         this.compiler = compiler;
119         typeMap =
120             ImmutableMap.<Pattern, DiagnosticType>builder()
121                 // Trailing comma
122                 .put(
123                     replacePlaceHolders(
124                         "Trailing comma is not legal in an ECMA-262 object initializer"),
125                     TRAILING_COMMA)
126     
127                 // Duplicate parameter
128                 .put(replacePlaceHolders("Duplicate parameter name \"{0}\""), DUPLICATE_PARAM)
129     
130                 .put(Pattern.compile("^invalid param name.*"), INVALID_PARAM)
131     
132                 // Unknown @annotations.
133                 .put(
134                     replacePlaceHolders(SimpleErrorReporter.getMessage0("msg.bad.jsdoc.tag")),
135                     BAD_JSDOC_ANNOTATION)
136     
137                 .put(
138                     Pattern.compile("^" + Pattern.quote(
139                         "Non-JSDoc comment has annotations. "
140                             + "Did you mean to start it with '/**'?")),
141                     JSDOC_IN_BLOCK_COMMENT)
142     
143                 .put(
144                     Pattern.compile(
145                         "^Keywords and reserved words" + " are not allowed as unquoted property.*"),
146                     INVALID_ES3_PROP_NAME)
147     
148                 .put(Pattern.compile("^Too many template parameters"), TOO_MANY_TEMPLATE_PARAMS)
149     
150                 // Type annotation warnings.
151                 .put(
152                     replacePlaceHolders(
153                         "Bad type annotation. Type annotations should have curly braces."),
154                     JSDOC_MISSING_BRACES_WARNING)
155     
156                 .put(replacePlaceHolders("Missing type declaration."), JSDOC_MISSING_TYPE_WARNING)
157     
158                 // Type annotation errors.
159                 .put(
160                     Pattern.compile(
161                         "^Bad type annotation.*(?!Type annotations should have curly braces\\.)"),
162                     TYPE_PARSE_ERROR)
163     
164                 // Parse tree too deep.
165                 .put(replacePlaceHolders("Too deep recursion while parsing"), PARSE_TREE_TOO_DEEP)
166     
167                 // Octal literals
168                 .put(Pattern.compile("^Octal .*literal.*"), INVALID_OCTAL_LITERAL)
169     
170                 .put(
171                     Pattern.compile("^this language feature is only supported for ECMASCRIPT6 mode.*"),
172                     ES6_FEATURE)
173     
174                 .put(Pattern.compile("^type syntax is only supported in ES6 typed mode.*"), ES6_TYPED)
175     
176                 .put(Pattern.compile("^Can only have JSDoc or inline type.*"), MISPLACED_TYPE_SYNTAX)
177     
178                 .build();
179       }

181       public static ErrorReporter forOldRhino(AbstractCompiler compiler) {
182         return new OldRhinoErrorReporter(compiler);
183       }
184     
185       void warningAtLine(String message, String sourceName, int line,
186           int lineOffset) {
187         compiler.report(
188             makeError(message, sourceName, line, lineOffset, CheckLevel.WARNING));
189       }
190     
191       void errorAtLine(String message, String sourceName, int line,
192           int lineOffset) {
193         compiler.report(
194             makeError(message, sourceName, line, lineOffset, CheckLevel.ERROR));
195       }
196     
197       protected DiagnosticType mapError(String message) {
198         for (Entry<Pattern, DiagnosticType> entry : typeMap.entrySet()) {
199           if (entry.getKey().matcher(message).matches()) {
200             return entry.getValue();
201           }
202         }
203         return null;
204       }
205     
206       private JSError makeError(String message, String sourceName, int line,
207           int lineOffset, CheckLevel defaultLevel) {
208     
209         // Try to see if the message is one of the rhino errors we want to
210         // expose as DiagnosticType by matching it with the regex key.
211         DiagnosticType type = mapError(message);
212         if (type != null) {
213           return JSError.make(
214               sourceName, line, lineOffset, type, message);
215         }
216     
217         return JSError.make(sourceName, line, lineOffset, defaultLevel,
218             PARSE_ERROR, message);
219       }

222           implements ErrorReporter {
223     
224         private OldRhinoErrorReporter(AbstractCompiler compiler) {
225           super(compiler);
226         }

228         @Override

232         }
233     
234         @Override
235         public void warning(String message, String sourceName, int line,

237           super.warningAtLine(message, sourceName, line, lineOffset);
238         }
239       }
