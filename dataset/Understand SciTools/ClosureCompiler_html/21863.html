30     /**
31      * Compute the "liveness" of all local variables. A variable is "live" at a
32      * point of a program if the value it is currently holding might be read later.
33      * Otherwise, the variable is considered "dead" if we know for sure that it will
34      * no longer be read. Dead variables are candidates for dead assignment
35      * elimination and variable name sharing. The worst case safe assumption is to
36      * assume that all variables are live. In that case, we will have no opportunity
37      * for optimizations. This is especially the case within a TRY block when an
38      * assignment is not guaranteed to take place. We bail out by assuming that
39      * all variables are live.
40      * <p>
41      * Due to the possibility of inner functions and closures, certain "local"
42      * variables can escape the function. These variables will be considered as
43      * global and they can be retrieved with {@link #getEscapedLocals()}.
44      *
45      */
46     class LiveVariablesAnalysis extends
47         DataFlowAnalysis<Node, LiveVariablesAnalysis.LiveVariableLattice> {
48     
49       // 100 = ((# of original Power Rangers) ^
50       //        (# years of Warren Harding in office)) *
51       //       (# of Ninja Turtles)
52       static final int MAX_VARIABLES_TO_ANALYZE = 100;
53     
54       public static final String ARGUMENT_ARRAY_ALIAS = "arguments";
55     
56       private static class LiveVariableJoinOp
57           implements JoinOp<LiveVariableLattice> {
58         @Override
59         public LiveVariableLattice apply(List<LiveVariableLattice> in) {
60           LiveVariableLattice result = new LiveVariableLattice(in.get(0));
61           for (int i = 1; i < in.size(); i++) {
62             result.liveSet.or(in.get(i).liveSet);
63           }
64           return result;
65         }
66       }
67     
68       /**
69        * The lattice that stores the liveness of all local variables at a given
70        * point in the program. The whole lattice is the power set of all local
71        * variables and a variable is live if it is in the set.
72        */
73       static class LiveVariableLattice implements LatticeElement {
74         private final BitSet liveSet;
75     
76         /**
77          * @param numVars Number of all local variables.
78          */
79         private LiveVariableLattice(int numVars) {
80           this.liveSet = new BitSet(numVars);
81         }
82     
83         private LiveVariableLattice(LiveVariableLattice other) {
84           Preconditions.checkNotNull(other);
85           this.liveSet = (BitSet) other.liveSet.clone();
86         }
87     
88         @Override
89         public boolean equals(Object other) {
90           Preconditions.checkNotNull(other);
91           return (other instanceof LiveVariableLattice) &&
92               this.liveSet.equals(((LiveVariableLattice) other).liveSet);
93         }
94     
95         public boolean isLive(Var v) {
96           Preconditions.checkNotNull(v);
97           return liveSet.get(v.index);
98         }
99     
100         public boolean isLive(int index) {
101           return liveSet.get(index);
102         }
103     
104         @Override
105         public String toString() {
106           return liveSet.toString();
107         }
108     
109         @Override
110         public int hashCode() {
111           return liveSet.hashCode();
112         }
113       }

115       // The scope of the function that we are analyzing.
116       private final Scope jsScope;
117       private final Set<Var> escaped;
118     
119       LiveVariablesAnalysis(ControlFlowGraph<Node> cfg, Scope jsScope,
120           AbstractCompiler compiler) {
121         super(cfg, new LiveVariableJoinOp());
122         this.jsScope = jsScope;
123         this.escaped = new HashSet<>();
124         computeEscaped(jsScope, escaped, compiler);
125       }
126     
127       public Set<? extends Var> getEscapedLocals() {
128         return (Set<? extends Var>) escaped;
129       }
130     
131       public int getVarIndex(String var) {
132         return jsScope.getVar(var).index;
133       }
134     
135       @Override
136       boolean isForward() {
137         return false;
138       }
139     
140       @Override
141       LiveVariableLattice createEntryLattice() {
142         return new LiveVariableLattice(jsScope.getVarCount());
143       }
144     
145       @Override
146       LiveVariableLattice createInitialEstimateLattice() {
147         return new LiveVariableLattice(jsScope.getVarCount());
148       }
149     
150       @Override
151       LiveVariableLattice flowThrough(Node node, LiveVariableLattice input) {
152         final BitSet gen = new BitSet(input.liveSet.size());
153         final BitSet kill = new BitSet(input.liveSet.size());
154     
155         // Make kills conditional if the node can end abruptly by an exception.
156         boolean conditional = false;
157         List<DiGraphEdge<Node, Branch>> edgeList = getCfg().getOutEdges(node);
158         for (DiGraphEdge<Node, Branch> edge : edgeList) {
159           if (Branch.ON_EX.equals(edge.getValue())) {
160             conditional = true;
161           }
162         }
163         computeGenKill(node, gen, kill, conditional);
164         LiveVariableLattice result = new LiveVariableLattice(input);
165         // L_in = L_out - Kill + Gen
166         result.liveSet.andNot(kill);
167         result.liveSet.or(gen);
168         return result;
169       }
170     
171       /**
172        * Computes the GEN and KILL set.
173        *
174        * @param n Root node.
175        * @param gen Local variables that are live because of the instruction at
176        *        {@code n} will be added to this set.
177        * @param kill Local variables that are killed because of the instruction at
178        *        {@code n} will be added to this set.
179        * @param conditional {@code true} if any assignments encountered are
180        *        conditionally executed. These assignments might not kill a variable.
181        */
182       private void computeGenKill(Node n, BitSet gen, BitSet kill,
183           boolean conditional) {
184     
185         switch (n.getType()) {
186           case SCRIPT:
187           case BLOCK:
188           case FUNCTION:
189             return;
190     
191           case WHILE:
192           case DO:
193           case IF:
194             computeGenKill(NodeUtil.getConditionExpression(n), gen, kill,
195                 conditional);
196             return;
197     
198           case FOR:
199             if (!NodeUtil.isForIn(n)) {
200               computeGenKill(NodeUtil.getConditionExpression(n), gen, kill,
201                   conditional);
202             } else {
203               // for(x in y) {...}
204               Node lhs = n.getFirstChild();
205               if (lhs.isVar()) {
206                 // for(var x in y) {...}
207                 lhs = lhs.getLastChild();
208               }
209     
210               if (lhs.isName()) {
211                 addToSetIfLocal(lhs, kill);
212                 addToSetIfLocal(lhs, gen);
213               } else {
214                 computeGenKill(lhs, gen, kill, conditional);
215               }
216     
217               // rhs is executed only once so we don't go into it every loop.
218             }
219             return;
220     
221           case VAR:
222             for (Node c = n.getFirstChild(); c != null; c = c.getNext()) {
223               if (c.hasChildren()) {
224                 computeGenKill(c.getFirstChild(), gen, kill, conditional);
225                 if (!conditional) {
226                   addToSetIfLocal(c, kill);
227                 }
228               }
229             }
230             return;
231     
232           case AND:
233           case OR:
234             computeGenKill(n.getFirstChild(), gen, kill, conditional);
235             // May short circuit.
236             computeGenKill(n.getLastChild(), gen, kill, true);
237             return;
238     
239           case HOOK:
240             computeGenKill(n.getFirstChild(), gen, kill, conditional);
241             // Assume both sides are conditional.
242             computeGenKill(n.getSecondChild(), gen, kill, true);
243             computeGenKill(n.getLastChild(), gen, kill, true);
244             return;
245     
246           case NAME:
247             if (isArgumentsName(n)) {
248               markAllParametersEscaped();
249             } else {
250               addToSetIfLocal(n, gen);
251             }
252             return;
253     
254           default:
255             if (NodeUtil.isAssignmentOp(n) && n.getFirstChild().isName()) {
256               Node lhs = n.getFirstChild();
257               if (!conditional) {
258                 addToSetIfLocal(lhs, kill);
259               }
260               if (!n.isAssign()) {
261                 // assignments such as a += 1 reads a.
262                 addToSetIfLocal(lhs, gen);
263               }
264               computeGenKill(lhs.getNext(), gen, kill, conditional);
265             } else {
266               for (Node c = n.getFirstChild(); c != null; c = c.getNext()) {
267                 computeGenKill(c, gen, kill, conditional);
268               }
269             }
270             return;
271         }
272       }
273     
274       private void addToSetIfLocal(Node node, BitSet set) {
275         Preconditions.checkState(node.isName());
276         String name = node.getString();
277         if (!jsScope.isDeclared(name, false)) {
278           return;
279         }
280         Var var = jsScope.getVar(name);
281         if (!escaped.contains(var)) {
282           set.set(var.index);
283         }
284       }
285     
286       /**
287        * Give up computing liveness of formal parameter by putting all the parameter
288        * names in the escaped set.
289        */
290       void markAllParametersEscaped() {
291         Node lp = jsScope.getRootNode().getSecondChild();
292         for (Node arg = lp.getFirstChild(); arg != null; arg = arg.getNext()) {
293           escaped.add(jsScope.getVar(arg.getString()));
294         }
295       }
296     
297       private boolean isArgumentsName(Node n) {
298         return n.isName() && n.getString().equals(ARGUMENT_ARRAY_ALIAS)
299             && !jsScope.isDeclared(ARGUMENT_ARRAY_ALIAS, false);
300       }
