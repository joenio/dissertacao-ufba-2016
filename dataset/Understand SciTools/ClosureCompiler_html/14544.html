18     import com.google.common.base.Preconditions;
19     import com.google.common.collect.ImmutableList;
20     import com.google.common.collect.ImmutableSet;
21     import com.google.javascript.jscomp.GlobalNamespace.Name;
22     import com.google.javascript.jscomp.GlobalNamespace.Ref;
23     import com.google.javascript.jscomp.PolymerPass.MemberDefinition;
24     import com.google.javascript.rhino.JSDocInfo;
25     import com.google.javascript.rhino.Node;

27     import java.util.List;

29     /**
30      * Helpers to extract behaviors from Polymer element declarations.
31      */
32     final class PolymerBehaviorExtractor {
33     
34       private static final ImmutableSet<String> BEHAVIOR_NAMES_NOT_TO_COPY = ImmutableSet.of(
35             "created", "attached", "detached", "attributeChanged", "configure", "ready",
36             "properties", "listeners", "observers", "hostAttributes");
37     
38       private final AbstractCompiler compiler;
39       private final GlobalNamespace globalNames;
40     
41       PolymerBehaviorExtractor(AbstractCompiler compiler, GlobalNamespace globalNames) {
42         this.compiler = compiler;
43         this.globalNames = globalNames;
44       }
45     
46       /**
47        * Extracts all Behaviors from an array literal, recursively. Entries in the array can be
48        * object literals or array literals (of other behaviors). Behavior names must be
49        * global, fully qualified names.
50        * @see https://github.com/Polymer/polymer/blob/0.8-preview/PRIMER.md#behaviors
51        * @return A list of all {@code BehaviorDefinitions} in the array.
52        */
53       ImmutableList<BehaviorDefinition> extractBehaviors(Node behaviorArray) {
54         if (behaviorArray == null) {
55           return ImmutableList.of();
56         }
57     
58         if (!behaviorArray.isArrayLit()) {
59           compiler.report(
60               JSError.make(behaviorArray, PolymerPassErrors.POLYMER_INVALID_BEHAVIOR_ARRAY));
61           return ImmutableList.of();
62         }
63     
64         ImmutableList.Builder<BehaviorDefinition> behaviors = ImmutableList.builder();
65         for (Node behaviorName : behaviorArray.children()) {
66           if (behaviorName.isObjectLit()) {
67             PolymerPassStaticUtils.switchDollarSignPropsToBrackets(behaviorName, compiler);
68             PolymerPassStaticUtils.quoteListenerAndHostAttributeKeys(behaviorName);
69             behaviors.add(new BehaviorDefinition(
70                 PolymerPassStaticUtils.extractProperties(behaviorName),
71                 getBehaviorFunctionsToCopy(behaviorName),
72                 getNonPropertyMembersToCopy(behaviorName),
73                 !NodeUtil.isInFunction(behaviorName)));
74             continue;
75           }
76     
77           Name behaviorGlobalName = globalNames.getSlot(behaviorName.getQualifiedName());
78           boolean isGlobalDeclaration = true;
79           if (behaviorGlobalName == null) {
80             compiler.report(JSError.make(behaviorName, PolymerPassErrors.POLYMER_UNQUALIFIED_BEHAVIOR));
81             continue;
82           }
83     
84           Ref behaviorDeclaration = behaviorGlobalName.getDeclaration();
85     
86           // Use any set as a backup declaration, even if it's local.
87           if (behaviorDeclaration == null) {
88             List<Ref> behaviorRefs = behaviorGlobalName.getRefs();
89             for (Ref ref : behaviorRefs) {
90               if (ref.isSet()) {
91                 isGlobalDeclaration = false;
92                 behaviorDeclaration = ref;
93                 break;
94               }
95             }
96           }
97     
98           if (behaviorDeclaration == null) {
99             compiler.report(JSError.make(behaviorName, PolymerPassErrors.POLYMER_UNQUALIFIED_BEHAVIOR));
100             continue;
101           }
102     
103           Node behaviorDeclarationNode = behaviorDeclaration.getNode();
104           JSDocInfo behaviorInfo = NodeUtil.getBestJSDocInfo(behaviorDeclarationNode);
105           if (behaviorInfo == null || !behaviorInfo.isPolymerBehavior()) {
106             compiler.report(
107                 JSError.make(behaviorDeclarationNode, PolymerPassErrors.POLYMER_UNANNOTATED_BEHAVIOR));
108           }
109     
110           Node behaviorValue = NodeUtil.getRValueOfLValue(behaviorDeclarationNode);
111     
112           if (behaviorValue == null) {
113             compiler.report(JSError.make(behaviorName, PolymerPassErrors.POLYMER_UNQUALIFIED_BEHAVIOR));
114           } else if (behaviorValue.isArrayLit()) {
115             // Individual behaviors can also be arrays of behaviors. Parse them recursively.
116             behaviors.addAll(extractBehaviors(behaviorValue));
117           } else if (behaviorValue.isObjectLit()) {
118             PolymerPassStaticUtils.switchDollarSignPropsToBrackets(behaviorValue, compiler);
119             PolymerPassStaticUtils.quoteListenerAndHostAttributeKeys(behaviorValue);
120             behaviors.add(new BehaviorDefinition(
121                 PolymerPassStaticUtils.extractProperties(behaviorValue),
122                 getBehaviorFunctionsToCopy(behaviorValue),
123                 getNonPropertyMembersToCopy(behaviorValue),
124                 isGlobalDeclaration));
125           } else {
126             compiler.report(JSError.make(behaviorName, PolymerPassErrors.POLYMER_UNQUALIFIED_BEHAVIOR));
127           }
128         }
129     
130         return behaviors.build();
131       }
132     
133       /**
134        * @return A list of functions from a behavior which should be copied to the element prototype.
135        */
136       private static ImmutableList<MemberDefinition> getBehaviorFunctionsToCopy(Node behaviorObjLit) {
137         Preconditions.checkState(behaviorObjLit.isObjectLit());
138         ImmutableList.Builder<MemberDefinition> functionsToCopy = ImmutableList.builder();
139     
140         for (Node keyNode : behaviorObjLit.children()) {
141           boolean isFunctionDefinition = (keyNode.isStringKey() && keyNode.getFirstChild().isFunction())
142               || keyNode.isMemberFunctionDef();
143           if (isFunctionDefinition && !BEHAVIOR_NAMES_NOT_TO_COPY.contains(keyNode.getString())) {
144             functionsToCopy.add(new MemberDefinition(NodeUtil.getBestJSDocInfo(keyNode), keyNode,
145               keyNode.getFirstChild()));
146           }
147         }
148     
149         return functionsToCopy.build();
150       }
151     
152       /**
153        * @return A list of MemberDefinitions in a behavior which are not in the properties block, but
154        *     should still be copied to the element prototype.
155        */
156       private static ImmutableList<MemberDefinition> getNonPropertyMembersToCopy(Node behaviorObjLit) {
157         Preconditions.checkState(behaviorObjLit.isObjectLit());
158         ImmutableList.Builder<MemberDefinition> membersToCopy = ImmutableList.builder();
159     
160         for (Node keyNode : behaviorObjLit.children()) {
161           boolean isNonFunctionMember = keyNode.isGetterDef()
162               || (keyNode.isStringKey() && !keyNode.getFirstChild().isFunction());
163           if (isNonFunctionMember && !BEHAVIOR_NAMES_NOT_TO_COPY.contains(keyNode.getString())) {
164             membersToCopy.add(new MemberDefinition(NodeUtil.getBestJSDocInfo(keyNode), keyNode,
165               keyNode.getFirstChild()));
166           }
167         }
168     
169         return membersToCopy.build();
170       }
171     
172       /**
173        * Parsed definition of a Polymer Behavior. Includes members which should be copied to elements
174        * which use the behavior.
175        */
176       static final class BehaviorDefinition {
177         /**
178          * Properties declared in the behavior 'properties' block.
179          */
180         final List<MemberDefinition> props;
181     
182         /**
183          * Functions intended to be copied to elements which use this Behavior.
184          */
185         final List<MemberDefinition> functionsToCopy;
186     
187         /**
188          * Other members intended to be copied to elements which use this Behavior.
189          */
190         final List<MemberDefinition> nonPropertyMembersToCopy;
191     
192         /**
193          * Whether this Behvaior is declared in the global scope.
194          */
195         final boolean isGlobalDeclaration;
196     
197         BehaviorDefinition(
198             List<MemberDefinition> props, List<MemberDefinition> functionsToCopy,
199             List<MemberDefinition> nonPropertyMembersToCopy, boolean isGlobalDeclaration) {
200           this.props = props;
201           this.functionsToCopy = functionsToCopy;
202           this.nonPropertyMembersToCopy = nonPropertyMembersToCopy;
203           this.isGlobalDeclaration = isGlobalDeclaration;
204         }
205       }
206     }
