33     /**
34      * CodePrinter prints out JS code in either pretty format or compact format.
35      *
36      * @see CodeGenerator
37      */
38     public final class CodePrinter {
39       // There are two separate CodeConsumers, one for pretty-printing and
40       // another for compact printing.
41     
42       // There are two implementations because the CompactCodePrinter
43       // potentially has a very different implementation to the pretty
44       // version.
45     
46       private abstract static class MappedCodePrinter extends CodeConsumer {
47         private final Deque<Mapping> mappings;
48         private final List<Mapping> allMappings;
49         private final boolean createSrcMap;
50         private final SourceMap.DetailLevel sourceMapDetailLevel;
51         protected final StringBuilder code = new StringBuilder(1024);
52         protected final int lineLengthThreshold;
53         protected int lineLength = 0;
54         protected int lineIndex = 0;
55     
56         MappedCodePrinter(
57             int lineLengthThreshold,
58             boolean createSrcMap,
59             SourceMap.DetailLevel sourceMapDetailLevel) {
60           Preconditions.checkState(sourceMapDetailLevel != null);
61           this.lineLengthThreshold = lineLengthThreshold <= 0 ? Integer.MAX_VALUE :
62             lineLengthThreshold;
63           this.createSrcMap = createSrcMap;
64           this.sourceMapDetailLevel = sourceMapDetailLevel;
65           this.mappings = createSrcMap ? new ArrayDeque<Mapping>() : null;
66           this.allMappings = createSrcMap ? new ArrayList<Mapping>() : null;
67         }
68     
69         /**
70          * Maintains a mapping from a given node to the position
71          * in the source code at which its generated form was
72          * placed. This position is relative only to the current
73          * run of the CodeConsumer and will be normalized
74          * later on by the SourceMap.
75          *
76          * @see SourceMap
77          */
78         private static class Mapping {
79           Node node;
80           FilePosition start;
81           FilePosition end;
82         }
83     
84         /**
85          * Starts the source mapping for the given
86          * node at the current position.
87          */
88         @Override
89         void startSourceMapping(Node node) {
90           Preconditions.checkState(sourceMapDetailLevel != null);
91           Preconditions.checkState(node != null);
92           if (createSrcMap
93               && node.getSourceFileName() != null
94               && node.getLineno() > 0
95               && sourceMapDetailLevel.apply(node)) {
96             int line = getCurrentLineIndex();
97             int index = getCurrentCharIndex();
98             Preconditions.checkState(line >= 0);
99             Mapping mapping = new Mapping();
100             mapping.node = node;
101             mapping.start = new FilePosition(line, index);
102             mappings.push(mapping);
103             allMappings.add(mapping);
104           }
105         }
106     
107         /**
108          * Finishes the source mapping for the given
109          * node at the current position.
110          */
111         @Override
112         void endSourceMapping(Node node) {
113           if (createSrcMap && !mappings.isEmpty() && mappings.peek().node == node) {
114             Mapping mapping = mappings.pop();
115             int line = getCurrentLineIndex();
116             int index = getCurrentCharIndex();
117             Preconditions.checkState(line >= 0);
118             mapping.end = new FilePosition(line, index);
119           }
120         }
121     
122         /**
123          * Generates the source map from the given code consumer,
124          * appending the information it saved to the SourceMap
125          * object given.
126          */
127         void generateSourceMap(SourceMap map){
128           if (createSrcMap) {
129             for (Mapping mapping : allMappings) {
130               map.addMapping(mapping.node, mapping.start, mapping.end);
131             }
132           }
133         }
134     
135         /**
136          * Reports to the code consumer that the given line has been cut at the
137          * given position, i.e. a \n has been inserted there. Or that a cut has
138          * been undone, i.e. a previously inserted \n has been removed.
139          * All mappings in the source maps after that position will be renormalized
140          * as needed.
141          */
142         void reportLineCut(int lineIndex, int charIndex, boolean insertion) {
143           if (createSrcMap) {
144             for (Mapping mapping : allMappings) {
145               mapping.start = convertPosition(mapping.start, lineIndex, charIndex,
146                   insertion);
147     
148               if (mapping.end != null) {
149                 mapping.end = convertPosition(mapping.end, lineIndex, charIndex,
150                     insertion);
151               }
152             }
153           }
154         }
155     
156         /**
157          * Converts the given position by normalizing it against the insertion
158          * or removal of a newline at the given line and character position.
159          *
160          * @param position The existing position before the newline was inserted.
161          * @param lineIndex The index of the line at which the newline was inserted.
162          * @param characterPosition The position on the line at which the newline
163          *     was inserted.
164          * @param insertion True if a newline was inserted, false if a newline was
165          *     removed.
166          *
167          * @return The normalized position.
168          * @throws IllegalStateException if an attempt to reverse a line cut is
169          *     made on a previous line rather than the current line.
170          */
171         private static FilePosition convertPosition(FilePosition position, int lineIndex,
172                                                     int characterPosition, boolean insertion) {
173           int originalLine = position.getLine();
174           int originalChar = position.getColumn();
175           if (insertion) {
176             if (originalLine == lineIndex && originalChar >= characterPosition) {
177               // If the position falls on the line itself, then normalize it
178               // if it falls at or after the place the newline was inserted.
179               return new FilePosition(
180                   originalLine + 1, originalChar - characterPosition);
181             } else {
182               return position;
183             }
184           } else {
185             if (originalLine == lineIndex) {
186               return new FilePosition(
187                   originalLine - 1, originalChar + characterPosition);
188             } else if (originalLine > lineIndex) {
189                 // Not supported, can only undo a cut on the most recent line. To
190                 // do this on a previous lines would require reevaluating the cut
191                 // positions on all subsequent lines.
192                 throw new IllegalStateException(
193                     "Cannot undo line cut on a previous line.");
194             } else {
195               return position;
196             }
197           }
198         }
199     
200         public String getCode() {
201           return code.toString();
202         }
203     
204         @Override
205         char getLastChar() {
206           return (code.length() > 0) ? code.charAt(code.length() - 1) : '\0';
207         }
208     
209         protected final int getCurrentCharIndex() {
210           return lineLength;
211         }
212     
213         protected final int getCurrentLineIndex() {
214           return lineIndex;
215         }
216       }

218       static class PrettyCodePrinter
219           extends MappedCodePrinter {
220         // The number of characters after which we insert a line break in the code
221         static final String INDENT = "  ";
222     
223         private int indent = 0;
224     
225         /**
226          * @param lineLengthThreshold The length of a line after which we force
227          *                            a newline when possible.
228          * @param createSourceMap Whether to generate source map data.
229          * @param sourceMapDetailLevel A filter to control which nodes get mapped
230          *     into the source map.
231          */
232         private PrettyCodePrinter(
233             int lineLengthThreshold,
234             boolean createSourceMap,
235             SourceMap.DetailLevel sourceMapDetailLevel) {
236           super(lineLengthThreshold, createSourceMap, sourceMapDetailLevel);
237         }
238     
239         /**
240          * Appends a string to the code, keeping track of the current line length.
241          */
242         @Override
243         void append(String str) {
244           // For pretty printing: indent at the beginning of the line
245           if (lineLength == 0) {
246             for (int i = 0; i < indent; i++) {
247               code.append(INDENT);
248               lineLength += INDENT.length();
249             }
250           }
251           code.append(str);
252           lineLength += str.length();
253           // Correct lineIndex and lineLength if there were newlines in the string.
254           int newlines = CharMatcher.is('\n').countIn(str);
255           if (newlines > 0) {
256             lineIndex += newlines;
257             lineLength = str.length() - str.lastIndexOf('\n');
258           }
259         }
260     
261         /**
262          * Adds a newline to the code, resetting the line length and handling
263          * indenting for pretty printing.
264          */
265         @Override
266         void startNewLine() {
267           if (lineLength > 0) {
268             code.append('\n');
269             lineIndex++;
270             lineLength = 0;
271           }
272         }
273     
274         @Override
275         void maybeLineBreak() {
276           maybeCutLine();
277         }
278     
279         /**
280          * This may start a new line if the current line is longer than the line
281          * length threshold.
282          */
283         @Override
284         void maybeCutLine() {
285           if (lineLength > lineLengthThreshold) {
286             startNewLine();
287           }
288         }
289     
290         @Override
291         void endLine() {
292           startNewLine();
293         }
294     
295         @Override
296         void appendBlockStart() {
297           maybeInsertSpace();
298           append("{");
299           indent++;
300         }
301     
302         @Override
303         void appendBlockEnd() {
304           maybeEndStatement();
305           endLine();
306           indent--;
307           append("}");
308         }
309     
310         @Override
311         void listSeparator() {
312           add(", ");
313           maybeLineBreak();
314         }
315     
316         @Override
317         void endFunction(boolean statementContext) {
318           super.endFunction(statementContext);
319           if (statementContext) {
320             startNewLine();
321           }
322         }
323     
324         @Override
325         void beginCaseBody() {
326           super.beginCaseBody();
327           indent++;
328           endLine();
329         }
330     
331         @Override
332         void endCaseBody() {
333           super.endCaseBody();
334           indent--;
335           endStatement();
336         }
337     
338         @Override
339         void appendOp(String op, boolean binOp) {
340           if (getLastChar() != ' ' && binOp && op.charAt(0) != ',') {
341             append(" ");
342           }
343           append(op);
344           if (binOp) {
345             append(" ");
346           }
347         }
348     
349         /**
350          * If the body of a for loop or the then clause of an if statement has
351          * a single statement, should it be wrapped in a block?
352          * {@inheritDoc}
353          */
354         @Override
355         boolean shouldPreserveExtraBlocks() {
356           // When pretty-printing, always place the statement in its own block
357           // so it is printed on a separate line.  This allows breakpoints to be
358           // placed on the statement.
359           return true;
360         }
361     
362         @Override
363         void maybeInsertSpace() {
364           if (getLastChar() != ' ') {
365             add(" ");
366           }
367         }
368     
369         /**
370          * @return The TRY node for the specified CATCH node.
371          */
372         private static Node getTryForCatch(Node n) {
373           return n.getGrandparent();
374         }
375     
376         /**
377          * @return Whether the a line break should be added after the specified
378          * BLOCK.
379          */
380         @Override
381         boolean breakAfterBlockFor(Node n,  boolean isStatementContext) {
382           Preconditions.checkState(n.isBlock(), n);
383           Node parent = n.getParent();
384           if (parent != null) {
385             Token type = parent.getType();
386             switch (type) {
387               case DO:
388                 // Don't break before 'while' in DO-WHILE statements.
389                 return false;
390               case FUNCTION:
391                 // FUNCTIONs are handled separately, don't break here.
392                 return false;
393               case TRY:
394                 // Don't break before catch
395                 return n != parent.getFirstChild();
396               case CATCH:
397                 // Don't break before finally
398                 return !NodeUtil.hasFinally(getTryForCatch(parent));
399               case IF:
400                 // Don't break before else
401                 return n == parent.getLastChild();
402             }
403           }
404           return true;
405         }
406     
407         @Override
408         void endFile() {
409           maybeEndStatement();
410         }
411       }

414       static class CompactCodePrinter
415           extends MappedCodePrinter {
416     
417         // The CompactCodePrinter tries to emit just enough newlines to stop there
418         // being lines longer than the threshold.  Since the output is going to be
419         // gzipped, it makes sense to try to make the newlines appear in similar
420         // contexts so that gzip can encode them for 'free'.
421         //
422         // This version tries to break the lines at 'preferred' places, which are
423         // between the top-level forms.  This works because top-level forms tend to
424         // be more uniform than arbitrary legal contexts.  Better compression would
425         // probably require explicit modeling of the gzip algorithm.
426     
427         private final boolean lineBreak;
428         private final boolean preferLineBreakAtEndOfFile;
429         private int lineStartPosition = 0;
430         private int preferredBreakPosition = 0;
431         private int prevCutPosition = 0;
432         private int prevLineStartPosition = 0;
433     
434       /**
435        * @param lineBreak break the lines a bit more aggressively
436        * @param lineLengthThreshold The length of a line after which we force
437        *                            a newline when possible.
438        * @param createSrcMap Whether to gather source position
439        *                            mapping information when printing.
440        * @param sourceMapDetailLevel A filter to control which nodes get mapped into
441        *     the source map.
442        */
443         private CompactCodePrinter(boolean lineBreak,
444             boolean preferLineBreakAtEndOfFile, int lineLengthThreshold,
445             boolean createSrcMap, SourceMap.DetailLevel sourceMapDetailLevel) {
446           super(lineLengthThreshold, createSrcMap, sourceMapDetailLevel);
447           this.lineBreak = lineBreak;
448           this.preferLineBreakAtEndOfFile = preferLineBreakAtEndOfFile;
449         }
450     
451         /**
452          * Appends a string to the code, keeping track of the current line length.
453          */
454         @Override
455         void append(String str) {
456           code.append(str);
457           lineLength += str.length();
458           // Correct lineIndex and lineLength if there were newlines in the string.
459           int newlines = CharMatcher.is('\n').countIn(str);
460           if (newlines > 0) {
461             lineIndex += newlines;
462             lineLength = str.length() - str.lastIndexOf('\n');
463           }
464         }
465     
466         /**
467          * Adds a newline to the code, resetting the line length.
468          */
469         @Override
470         void startNewLine() {
471           if (lineLength > 0) {
472             prevCutPosition = code.length();
473             prevLineStartPosition = lineStartPosition;
474             code.append('\n');
475             lineLength = 0;
476             lineIndex++;
477             lineStartPosition = code.length();
478           }
479         }
480     
481         @Override
482         void maybeLineBreak() {
483           if (lineBreak) {
484             if (sawFunction) {
485               startNewLine();
486               sawFunction = false;
487             }
488           }
489     
490           // Since we are at a legal line break, can we upgrade the
491           // preferred break position?  We prefer to break after a
492           // semicolon rather than before it.
493           int len = code.length();
494           if (preferredBreakPosition == len - 1) {
495             char ch = code.charAt(len - 1);
496             if (ch == ';') {
497               preferredBreakPosition = len;
498             }
499           }
500           maybeCutLine();
501         }
502     
503         /**
504          * This may start a new line if the current line is longer than the line
505          * length threshold.
506          */
507         @Override
508         void maybeCutLine() {
509           if (lineLength > lineLengthThreshold) {
510             // Use the preferred position provided it will break the line.
511             if (preferredBreakPosition > lineStartPosition &&
512                 preferredBreakPosition < lineStartPosition + lineLength) {
513               int position = preferredBreakPosition;
514               code.insert(position, '\n');
515               prevCutPosition = position;
516               reportLineCut(lineIndex, position - lineStartPosition, true);
517               lineIndex++;
518               lineLength -= (position - lineStartPosition);
519               lineStartPosition = position + 1;
520             } else {
521               startNewLine();
522             }
523           }
524         }
525     
526         @Override
527         void notePreferredLineBreak() {
528           preferredBreakPosition = code.length();
529         }
530     
531         @Override
532         void endFile() {
533           super.endFile();
534           if (!preferLineBreakAtEndOfFile) {
535             return;
536           }
537           if (lineLength > lineLengthThreshold / 2) {
538             // Add an extra break at end of file.
539             append(";");
540             startNewLine();
541           } else if (prevCutPosition > 0) {
542             // Shift the previous break to end of file by replacing it with a
543             // <space> and adding a new break at end of file. Adding the space
544             // handles cases like instanceof\nfoo. (it would be nice to avoid this)
545             code.setCharAt(prevCutPosition, ' ');
546             lineStartPosition = prevLineStartPosition;
547             lineLength = code.length() - lineStartPosition;
548             reportLineCut(lineIndex, prevCutPosition + 1, false);
549             lineIndex--;
550             prevCutPosition = 0;
551             prevLineStartPosition = 0;
552             append(";");
553             startNewLine();
554           } else {
555             // A small file with no line breaks. We do nothing in this case to
556             // avoid excessive line breaks. It's not ideal if a lot of these pile
557             // up, but that is reasonably unlikely.
558           }
559         }
560     
561       }

676       /**
677        * Specifies a format for code generation.
678        */
679       public enum Format {
680         COMPACT,
681         PRETTY,
682         TYPED;
683     
684         static Format fromOptions(CompilerOptions options, boolean outputTypes, boolean prettyPrint) {
685           if (outputTypes) {
686             return Format.TYPED;
687           }
688           if (prettyPrint || options.getLanguageOut() == LanguageMode.ECMASCRIPT6_TYPED) {
689             return Format.PRETTY;
690           }
691           return Format.COMPACT;
692         }
693       }

695       /**
696        * Converts a tree to JS code
697        */
698       private static String toSource(Node root, Format outputFormat,
699           CompilerOptions options, SourceMap sourceMap, boolean tagAsStrict, boolean lineBreak,
700           CodeGeneratorFactory codeGeneratorFactory) {
701         Preconditions.checkState(options.sourceMapDetailLevel != null);
702     
703         boolean createSourceMap = (sourceMap != null);
704         MappedCodePrinter mcp =
705             outputFormat == Format.COMPACT
706             ? new CompactCodePrinter(
707                 lineBreak,
708                 options.preferLineBreakAtEndOfFile,
709                 options.lineLengthThreshold,
710                 createSourceMap,
711                 options.sourceMapDetailLevel)
712             : new PrettyCodePrinter(
713                 options.lineLengthThreshold,
714                 createSourceMap,
715                 options.sourceMapDetailLevel);
716         CodeGenerator cg = codeGeneratorFactory.getCodeGenerator(outputFormat, mcp);
717     
718         cg.maybeTagAsExterns();
719         if (tagAsStrict) {
720           cg.tagAsStrict();
721         }
722     
723         cg.add(root);
724         mcp.endFile();
725     
726         String code = mcp.getCode();
727     
728         if (createSourceMap) {
729           mcp.generateSourceMap(sourceMap);
730         }
731     
732         return code;
733       }
