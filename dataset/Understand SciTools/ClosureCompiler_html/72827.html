33     /**
34      * Rewrites "Polymer({})" calls into a form that is suitable for type checking and dead code
35      * elimination. Also ensures proper format and types.
36      *
37      * <p>Only works with Polymer version 0.8 and above.
38      *
39      * <p>Design and examples: https://github.com/google/closure-compiler/wiki/Polymer-Pass
40      *
41      * @author jlklein@google.com (Jeremy Klein)
42      */
43     final class PolymerPass extends AbstractPostOrderCallback implements HotSwapCompilerPass {
44     
45       static final String VIRTUAL_FILE = "<PolymerPass.java>";
46     
47       private final AbstractCompiler compiler;
48       private final Map<String, String> tagNameMap;
49     
50       private Node polymerElementExterns;
51       private Set<String> nativeExternsAdded;
52       private ImmutableList<Node> polymerElementProps;
53       private GlobalNamespace globalNames;
54     
55       PolymerPass(AbstractCompiler compiler) {
56         this.compiler = compiler;
57         tagNameMap = TagNameToType.getMap();
58         nativeExternsAdded = new HashSet<>();
59       }
60     
61       @Override
62       public void process(Node externs, Node root) {
63         PolymerPassFindExterns externsCallback = new PolymerPassFindExterns();
64         NodeTraversal.traverseEs6(compiler, externs, externsCallback);
65         polymerElementExterns = externsCallback.getPolymerElementExterns();
66         polymerElementProps = externsCallback.getPolymerElementProps();
67     
68         if (polymerElementExterns == null) {
69           compiler.report(JSError.make(externs, PolymerPassErrors.POLYMER_MISSING_EXTERNS));
70           return;
71         }
72     
73         globalNames = new GlobalNamespace(compiler, externs, root);
74     
75         hotSwapScript(root, null);
76       }
77     
78       @Override
79       public void hotSwapScript(Node scriptRoot, Node originalRoot) {
80         NodeTraversal.traverseEs6(compiler, scriptRoot, this);
81         PolymerPassSuppressBehaviors suppressBehaviorsCallback =
82             new PolymerPassSuppressBehaviors(compiler);
83         NodeTraversal.traverseEs6(compiler, scriptRoot, suppressBehaviorsCallback);
84       }
85     
86       @Override
87       public void visit(NodeTraversal traversal, Node node, Node parent) {
88         Preconditions.checkState(
89             polymerElementExterns != null && polymerElementProps != null && globalNames != null,
90             "Cannot call visit() before process()");
91         if (isPolymerCall(node)) {
92           rewriteClassDefinition(node, parent, traversal);
93         }
94       }
95     
96       private void rewriteClassDefinition(Node node, Node parent, NodeTraversal traversal) {
97         Node grandparent = parent.getParent();
98         if (grandparent.isConst()) {
99           compiler.report(JSError.make(node, PolymerPassErrors.POLYMER_INVALID_DECLARATION));
100           return;
101         }
102         PolymerClassDefinition def = PolymerClassDefinition.extractFromCallNode(
103             node, compiler, globalNames);
104         if (def != null) {
105           if (def.nativeBaseElement != null) {
106             appendPolymerElementExterns(def);
107           }
108           PolymerClassRewriter rewriter = new PolymerClassRewriter(compiler, polymerElementExterns);
109           if (NodeUtil.isNameDeclaration(grandparent) || parent.isAssign()) {
110             rewriter.rewritePolymerClass(grandparent, def, traversal.getScope().isGlobal());
111           } else {
112             rewriter.rewritePolymerClass(parent, def, traversal.getScope().isGlobal());
113           }
114         }
115       }
116     
117       /**
118        * Duplicates the PolymerElement externs with a different element base class if needed.
119        * For example, if the base class is HTMLInputElement, then a class PolymerInputElement will be
120        * added. If the element does not extend a native HTML element, this method is a no-op.
121        */
122       private void appendPolymerElementExterns(final PolymerClassDefinition def) {
123         if (!nativeExternsAdded.add(def.nativeBaseElement)) {
124           return;
125         }
126     
127         Node block = IR.block();
128     
129         Node baseExterns = polymerElementExterns.cloneTree();
130         String polymerElementType = PolymerPassStaticUtils.getPolymerElementType(def);
131         baseExterns.getFirstChild().setString(polymerElementType);
132     
133         String elementType = tagNameMap.get(def.nativeBaseElement);
134         JSTypeExpression elementBaseType =
135             new JSTypeExpression(new Node(Token.BANG, IR.string(elementType)), VIRTUAL_FILE);
136         JSDocInfoBuilder baseDocs = JSDocInfoBuilder.copyFrom(baseExterns.getJSDocInfo());
137         baseDocs.changeBaseType(elementBaseType);
138         baseExterns.setJSDocInfo(baseDocs.build());
139         block.addChildToBack(baseExterns);
140     
141         for (Node baseProp : polymerElementProps) {
142           Node newProp = baseProp.cloneTree();
143           Node newPropRootName =
144               NodeUtil.getRootOfQualifiedName(newProp.getFirstFirstChild());
145           newPropRootName.setString(polymerElementType);
146           block.addChildToBack(newProp);
147         }
148     
149         block.useSourceInfoIfMissingFromForTree(polymerElementExterns);
150     
151         Node parent = polymerElementExterns.getParent();
152         Node stmts = block.removeChildren();
153         parent.addChildrenAfter(stmts, polymerElementExterns);
154     
155         compiler.reportCodeChange();
156       }
157     
158       /**
159        * @return Whether the call represents a call to the Polymer function.
160        */
161       @VisibleForTesting
162       public static boolean isPolymerCall(Node value) {
163         return value != null && value.isCall() && value.getFirstChild().matchesQualifiedName("Polymer");
164       }
165     
166       /**
167        * Any member of a Polymer element or Behavior. These can be functions, properties, etc.
168        */
169       static class MemberDefinition {
170         /** Any {@link JSDocInfo} tied to this member. */
171         final JSDocInfo info;
172     
173         /** Name {@link Node} for the definition of this member. */
174         final Node name;
175     
176         /** Value {@link Node} (RHS) for the definition of this member. */
177         final Node value;
178     
179         MemberDefinition(JSDocInfo info, Node name, Node value) {
180           this.info = info;
181           this.name = name;
182           this.value = value;
183         }
184       }
185     }
