49       // Currently, TOP_OBJECT has two conflicting roles: the supertype of all
50       // object types, and the type of an empty object literal.
51       // In particular, its kind is UNRESTRICTED, which is confusing, because this
52       // kind is a subkind of STRUCT and DICT.

76       private ObjectType(NominalType nominalType,
77           PersistentMap<String, Property> props, FunctionType fn, Namespace ns,
78           boolean isLoose, ObjectKind objectKind) {
79         Preconditions.checkArgument(
80             fn == null || fn.isQmarkFunction() || fn.isLoose() == isLoose,
81             "isLoose: %s, fn: %s", isLoose, fn);
82         Preconditions.checkArgument(FunctionType.isInhabitable(fn));
83         Preconditions.checkArgument(fn == null || nominalType != null,
84               "Cannot create function %s without nominal type", fn);
85         if (ns != null && nominalType != null) {
86           String name = nominalType.getName();
87           Preconditions.checkArgument(name.equals("Object")
88               || name.equals("Function") || name.equals("Window"),
89               "Can't create namespace with nominal type %s", name);
90         }
91         if (nominalType != null) {
92           Preconditions.checkArgument(!nominalType.isClassy() || !isLoose,
93               "Cannot create loose objectType with nominal type %s", nominalType);
94           Preconditions.checkArgument(fn == null || nominalType.isFunction(),
95               "Cannot create objectType of nominal type %s with function (%s)",
96               nominalType, fn);
97         }
98         this.nominalType = nominalType;
99         this.props = isLoose ? loosenProps(props) : props;
100         this.fn = fn;
101         this.ns = ns;
102         this.isLoose = isLoose;
103         // Don't track @struct-ness/@dict-ness for loose objects
104         this.objectKind = isLoose ? ObjectKind.UNRESTRICTED : objectKind;
105       }

107       // Loose object types may have properties that are also loose objects, eg,
108       //   function f(obj) { obj.a.b.c = 123; }
109       // This function makes sure we mark these object properties as loose.
110       private static PersistentMap<String, Property> loosenProps(
111           PersistentMap<String, Property> props) {
112         PersistentMap<String, Property> newProps = props;
113         for (Map.Entry<String, Property> entry : props.entrySet()) {
114           JSType propType = entry.getValue().getType();
115           ObjectType objType = propType.getObjTypeIfSingletonObj();
116           if (objType != null
117               && !objType.getNominalType().isClassy() && !objType.isLoose()) {
118             newProps = newProps.with(
119                 entry.getKey(),
120                 Property.make(propType.withLoose(), null));
121           }
122         }
123         return newProps;
124       }

126       static ObjectType makeObjectType(NominalType nominalType,
127           PersistentMap<String, Property> props, FunctionType fn, Namespace ns,
128           boolean isLoose, ObjectKind ok) {
129         if (props == null) {
130           props = PersistentMap.create();
131         } else if (containsBottomProp(props) || !FunctionType.isInhabitable(fn)) {
132           return BOTTOM_OBJECT;
133         }
134         if (fn != null && !props.containsKey("prototype")
135             && (ns == null || ns.getNsProp("prototype") == null)) {
136           props = props.with("prototype", UNKNOWN_PROP);
137         }
138         return new ObjectType(nominalType, props, fn, ns, isLoose, ok);
139       }

150       /** Construct an object with the given declared properties. */
151       static ObjectType fromProperties(Map<String, Property> oldProps) {
152         PersistentMap<String, Property> newProps = PersistentMap.create();
153         for (Map.Entry<String, Property> entry : oldProps.entrySet()) {
154           Property prop = entry.getValue();
155           if (prop.getDeclaredType().isBottom()) {
156             return BOTTOM_OBJECT;
157           }
158           newProps = newProps.with(entry.getKey(), prop);
159         }
160         return new ObjectType(null, newProps, null, null, false, ObjectKind.UNRESTRICTED);
161       }

230       // Crude heuristic to decide whether a loose object is actually a scalar type
231       // and methods have been called on it.
232       // Does not apply to too-common properties such as toString (and for this
233       // reason it doesn't apply to booleans).
234       // Only uses property names; change it to use types if precision isn't
235       // satisfactory.
236       static JSType mayTurnLooseObjectToScalar(JSType t, JSTypes commonTypes) {
237         ObjectType obj = t.getObjTypeIfSingletonObj();
238         if (obj == null || !obj.isLoose() || obj.props.isEmpty() || obj.fn != null
239             || hasOnlyBuiltinProps(obj, TOP_OBJECT)
240             || hasOnlyBuiltinProps(
241                 obj, commonTypes.getArrayInstance().getObjTypeIfSingletonObj())) {
242           return t;
243         }
244         if (hasOnlyBuiltinProps(obj, commonTypes.getNumberInstanceObjType())) {
245           return JSType.NUMBER;
246         }
247         if (hasOnlyBuiltinProps(obj, commonTypes.getStringInstanceObjType())) {
248           return JSType.STRING;
249         }
250         return t;
251       }

253       // Trade-offs about property behavior on loose object types:
254       // We never mark properties as optional on loose objects. The reason is that
255       // we cannot know for sure when a property is optional or not.
256       // Eg, when we see an assignment to a loose obj
257       //   obj.p1 = 123;
258       // we cannot know if obj already has p1, or if this is a property creation.
259       // If the assignment is inside an IF branch, we should not say after the IF
260       // that p1 is optional. But as a consequence, this means that any property we
261       // see on a loose object might be optional. That's why we don't warn about
262       // possibly-inexistent properties on loose objects.
263       // Last, say we infer a loose object type with a property p1 for a formal
264       // parameter of a function f. If we pass a non-loose object to f that does not
265       // have a p1, we warn. This may create spurious warnings, if p1 is optional,
266       // but mostly it catches real bugs.
267     
268       private ObjectType withLoose() {
269         if (isLoose()
270             // Don't loosen nominal types
271             || this.nominalType != null && this.nominalType.isClassy()
272             // Don't loosen namespaces
273             || this.ns != null) {
274           return this;
275         }
276         FunctionType fn = this.fn == null ? null : this.fn.withLoose();
277         PersistentMap<String, Property> newProps = PersistentMap.create();
278         for (Map.Entry<String, Property> propsEntry : this.props.entrySet()) {
279           String pname = propsEntry.getKey();
280           Property prop = propsEntry.getValue();
281           // It's wrong to warn about a possibly absent property on loose objects.
282           newProps = newProps.with(pname, prop.withRequired());
283         }
284         // No need to call makeObjectType; we know that the new object is inhabitable.
285         return new ObjectType(this.nominalType, newProps, fn, null, true, this.objectKind);
286       }

288       ObjectType withFunction(FunctionType ft, NominalType fnNominal) {
289         Preconditions.checkState(this.isNamespace());
290         Preconditions.checkState(!ft.isLoose() || ft.isQmarkFunction());
291         ObjectType obj = makeObjectType(
292             fnNominal, this.props, ft, this.ns, false, this.objectKind);
293         this.ns.updateNamespaceType(JSType.fromObjectType(obj));
294         return obj;
295       }

306       // When type is null, this method removes the property.
307       // If the property is already declared, but isDeclared is false, be careful
308       // to not un-declare it.
309       // If the property is already constant, but isConstant is false, be careful
310       // to not un-const it.
311       private ObjectType withPropertyHelper(QualifiedName qname, JSType type,
312           boolean isDeclared, boolean isConstant) {
313         // TODO(dimvar): We do some short-circuiting based on the declared type,
314         // but maybe we can do more based also on the existing inferred type (?)
315         PersistentMap<String, Property> newProps = this.props;
316         if (qname.isIdentifier()) {
317           String pname = qname.getLeftmostName();
318           JSType declType = getDeclaredProp(qname);
319           if (type == null) {
320             type = declType;
321           }
322           if (declType != null) {
323             isDeclared = true;
324             if (hasConstantProp(qname)) {
325               isConstant = true;
326             }
327             if (type != null && !type.isSubtypeOf(declType, SubtypeCache.create())) {
328               // Can happen in inheritance-related type errors.
329               // Not sure what the best approach is.
330               // For now, just forget the inferred type.
331               type = declType;
332             }
333           }
334     
335           if (type == null && declType == null) {
336             newProps = newProps.without(pname);
337           } else if (!type.equals(declType)) {
338             if (isDeclared && declType == null) {
339               declType = type;
340             }
341             newProps = newProps.with(pname,
342                 isConstant ?
343                 Property.makeConstant(null, type, declType) :
344                 Property.make(type, isDeclared ? declType : null));
345           }
346         } else { // This has a nested object
347           String objName = qname.getLeftmostName();
348           QualifiedName objQname = new QualifiedName(objName);
349           if (!mayHaveProp(objQname)) {
350             Preconditions.checkState(type == null,
351                 "Trying to update property %s on type %s, but sub-property %s does"
352                 + " not exist", qname, this, objName);
353             return this;
354           }
355           QualifiedName innerProps = qname.getAllButLeftmost();
356           Property objProp = getLeftmostProp(objQname);
357           JSType inferred = type == null ?
358               objProp.getType().withoutProperty(innerProps) :
359               objProp.getType().withProperty(innerProps, type);
360           JSType declared = objProp.getDeclaredType();
361           if (!inferred.equals(declared)) {
362             newProps = newProps.with(objName, objProp.isOptional() ?
363                 Property.makeOptional(null, inferred, declared) :
364                 Property.make(inferred, declared));
365           }
366         }
367         // check for ref equality to avoid creating a new type
368         if (newProps == this.props) {
369           return this;
370         }
371         return makeObjectType(this.nominalType, newProps,
372             this.fn, this.ns, this.isLoose, this.objectKind);
373       }

416       private static PersistentMap<String, Property> meetPropsHelper(
417           boolean specializeProps1, NominalType resultNominalType,
418           PersistentMap<String, Property> props1,
419           PersistentMap<String, Property> props2) {
420         if (resultNominalType == null) {
421           // If props1 or props2 contains a property that also exists on Object,
422           // we must take the inherited property type into account.
423           resultNominalType = builtinObject;
424         }
425         PersistentMap<String, Property> newProps = props1;
426         for (Map.Entry<String, Property> propsEntry : props1.entrySet()) {
427           String pname = propsEntry.getKey();
428           Property otherProp = resultNominalType.getProp(pname);
429           if (otherProp != null) {
430             newProps = addOrRemoveProp(
431                 specializeProps1, newProps, pname, otherProp, propsEntry.getValue());
432             if (newProps == BOTTOM_MAP) {
433               return BOTTOM_MAP;
434             }
435           }
436         }
437         for (Map.Entry<String, Property> propsEntry : props2.entrySet()) {
438           String pname = propsEntry.getKey();
439           Property prop2 = propsEntry.getValue();
440           Property newProp;
441           if (!props1.containsKey(pname)) {
442             newProp = prop2;
443           } else {
444             Property prop1 = props1.get(pname);
445             if (prop1.equals(prop2)) {
446               continue;
447             }
448             newProp = specializeProps1 ?
449                 prop1.specialize(prop2) :
450                 Property.meet(prop1, prop2);
451           }
452           Property otherProp = resultNominalType.getProp(pname);
453           if (otherProp != null) {
454             newProps = addOrRemoveProp(specializeProps1, newProps, pname, otherProp, newProp);
455             if (newProps == BOTTOM_MAP) {
456               return BOTTOM_MAP;
457             }
458           } else {
459             if (newProp.getType().isBottom()) {
460               return BOTTOM_MAP;
461             }
462             newProps = newProps.with(pname, newProp);
463           }
464         }
465         return newProps;
466       }

468       private static PersistentMap<String, Property> addOrRemoveProp(
469           boolean specializeProps1, PersistentMap<String, Property> props,
470           String pname, Property nomProp, Property objProp) {
471         JSType nomPropType = nomProp.getType();
472         Property newProp = specializeProps1
473             ? nomProp.specialize(objProp)
474             : Property.meet(nomProp, objProp);
475         JSType newPropType = newProp.getType();
476         if (newPropType.isBottom()) {
477           return BOTTOM_MAP;
478         }
479         if (!newPropType.isUnknown()
480             && newPropType.isSubtypeOf(nomPropType, SubtypeCache.create())
481             && !newPropType.equals(nomPropType)) {
482           return props.with(pname, newProp);
483         }
484         return props.without(pname);
485       }

487       private static Property getProp(Map<String, Property> props, NominalType nom, String pname) {
488         if (props.containsKey(pname)) {
489           return props.get(pname);
490         } else if (nom != null) {
491           return nom.getProp(pname);
492         }
493         return null;
494       }

496       // This method needs the nominal types because otherwise a property may become
497       // optional by mistake after the join.
498       // joinPropsLoosely doesn't need that, because we don't create optional props
499       // on loose types.
500       private static PersistentMap<String, Property> joinProps(
501           Map<String, Property> props1, Map<String, Property> props2,
502           NominalType nom1, NominalType nom2) {
503         PersistentMap<String, Property> newProps = PersistentMap.create();
504         for (String pname : Sets.union(props1.keySet(), props2.keySet())) {
505           Property prop1 = getProp(props1, nom1, pname);
506           Property prop2 = getProp(props2, nom2, pname);
507           Property newProp = null;
508           if (prop1 == null) {
509             newProp = prop2.withOptional();
510           } else if (prop2 == null) {
511             newProp = prop1.withOptional();
512           } else {
513             newProp = Property.join(prop1, prop2);
514           }
515           newProps = newProps.with(pname, newProp);
516         }
517         return newProps;
518       }

520       private static PersistentMap<String, Property> joinPropsLoosely(
521           Map<String, Property> props1, Map<String, Property> props2) {
522         PersistentMap<String, Property> newProps = PersistentMap.create();
523         for (Map.Entry<String, Property> propsEntry : props1.entrySet()) {
524           String pname = propsEntry.getKey();
525           if (!props2.containsKey(pname)) {
526             newProps = newProps.with(pname, propsEntry.getValue().withRequired());
527           }
528           if (newProps == BOTTOM_MAP) {
529             return BOTTOM_MAP;
530           }
531         }
532         for (Map.Entry<String, Property> propsEntry : props2.entrySet()) {
533           String pname = propsEntry.getKey();
534           Property prop2 = propsEntry.getValue();
535           if (props1.containsKey(pname)) {
536             newProps = newProps.with(pname,
537                 Property.join(props1.get(pname), prop2).withRequired());
538           } else {
539             newProps = newProps.with(pname, prop2.withRequired());
540           }
541           if (newProps == BOTTOM_MAP) {
542             return BOTTOM_MAP;
543           }
544         }
545         return newProps;
546       }

554       static void whyNotUnionSubtypes(boolean keepLoosenessOfThis,
555           Set<ObjectType> objs1, Set<ObjectType> objs2, SubtypeCache subSuperMap,
556           MismatchInfo[] boxedInfo) {
557         Preconditions.checkArgument(boxedInfo.length == 1);

559             keepLoosenessOfThis, objs1, objs2, subSuperMap, boxedInfo);
560         Preconditions.checkState(!areSubtypes);

563       private static boolean isUnionSubtypeHelper(boolean keepLoosenessOfThis,
564           Set<ObjectType> objs1, Set<ObjectType> objs2, SubtypeCache subSuperMap,
565           MismatchInfo[] boxedInfo) {
566         for (ObjectType obj1 : objs1) {
567           boolean foundSupertype = false;
568           for (ObjectType obj2 : objs2) {
569             if (obj1.isSubtypeOfHelper(keepLoosenessOfThis, obj2, subSuperMap, null)) {
570               foundSupertype = true;
571               break;
572             }
573           }
574           if (!foundSupertype) {
575             if (boxedInfo != null) {
576               boxedInfo[0] =
577                   MismatchInfo.makeUnionTypeMismatch(JSType.fromObjectType(obj1));
578             }
579             return false;
580           }
581         }
582         return true;
583       }

597       /**
598        * Required properties are acceptable where an optional is required,
599        * but not vice versa.
600        * Optional properties create cycles in the type lattice, eg,
601        * { } \le { p: num= }  and also   { p: num= } \le { }.
602        */
603       private boolean isSubtypeOfHelper(boolean keepLoosenessOfThis,
604           ObjectType other, SubtypeCache subSuperMap, MismatchInfo[] boxedInfo) {
605         if (other == TOP_OBJECT) {
606           return true;
607         }
608     
609         if ((keepLoosenessOfThis && this.isLoose) || other.isLoose) {
610           return this.isLooseSubtypeOf(other, subSuperMap);
611         }
612     
613         NominalType thisNt = this.nominalType;
614         NominalType otherNt = other.nominalType;
615         boolean checkOnlyLocalProps = true;
616         if (otherNt != null && otherNt.isStructuralInterface()) {
617           if (otherNt.equals(subSuperMap.get(thisNt))) {
618             return true;
619           }
620           subSuperMap = subSuperMap.with(thisNt, otherNt);
621           if (thisNt == null || !thisNt.isNominalSubtypeOf(otherNt)) {
622             checkOnlyLocalProps = false;
623           }
624         } else if (otherNt != null && !otherNt.isStructuralInterface()
625             && (thisNt == null || !thisNt.isNominalSubtypeOf(otherNt))) {
626           return false;
627         }
628     
629         // If nominalType1 < nominalType2, we only need to check that the
630         // properties of other are in (obj1 or nominalType1)
631         Set<String> otherPropNames;
632         if (checkOnlyLocalProps) {
633           otherPropNames = other.props.keySet();
634         } else {
635           otherPropNames = otherNt.getAllPropsOfInterface();
636           if (otherPropNames == null) {
637             // Can't check structural interfaces for subtyping during GlobalTypeInfo
638             return false;
639           }
640         }
641         if (!arePropertiesSubtypes(other, otherPropNames, subSuperMap, boxedInfo)) {
642           return false;
643         }
644     
645         if (other.fn == null) {
646           return true;
647         } else if (this.fn == null) {
648           // Can only be executed if we have declared types for callable objects.
649           return false;
650         }
651         boolean areFunsSubtypes = this.fn.isSubtypeOf(other.fn, subSuperMap);
652         if (boxedInfo != null) {
653           FunctionType.whyNotSubtypeOf(this.fn, other.fn, subSuperMap, boxedInfo);
654         }
655         return areFunsSubtypes;
656       }

658       private boolean arePropertiesSubtypes(ObjectType other,
659           Set<String> otherPropNames, SubtypeCache subSuperMap,
660           MismatchInfo[] boxedInfo) {
661         for (String pname : otherPropNames) {
662           QualifiedName qname = new QualifiedName(pname);
663           if (!isPropertySubtype(
664               pname, this.getLeftmostProp(qname), other.getLeftmostProp(qname),
665               subSuperMap, boxedInfo)) {
666             return false;
667           }
668         }
669         if (other.ns != null) {
670           for (String pname : other.ns.getAllPropsOfNamespace()) {
671             if (!otherPropNames.contains(pname)) {
672               QualifiedName qname = new QualifiedName(pname);
673               if (!isPropertySubtype(
674                   pname, this.getLeftmostProp(qname), other.getLeftmostProp(qname),
675                   subSuperMap, boxedInfo)) {
676                 return false;
677               }
678             }
679           }
680         }
681         return true;
682       }

685           Property prop2, SubtypeCache subSuperMap, MismatchInfo[] boxedInfo) {
686         return boxedInfo != null
687             ? getPropMismatchInfo(pname, prop1, prop2, subSuperMap, boxedInfo)
688             : isPropertySubtypeHelper(prop1, prop2, subSuperMap);
689       }

691       private static boolean isPropertySubtypeHelper(
692           Property prop1, Property prop2, SubtypeCache subSuperMap) {
693         if (prop2.isOptional()) {
694           if (prop1 != null
695               && !prop1.getType().isSubtypeOf(prop2.getType(), subSuperMap)) {
696             return false;
697           }
698         } else {
699           if (prop1 == null || prop1.isOptional() ||
700               !prop1.getType().isSubtypeOf(prop2.getType(), subSuperMap)) {
701             return false;
702           }
703         }
704         return true;
705       }

707       // Like isPropertySubtypeHelper, but also provides mismatch information
708       private static boolean getPropMismatchInfo(String pname, Property prop1,
709           Property prop2, SubtypeCache subSuperMap, MismatchInfo[] boxedInfo) {
710         Preconditions.checkNotNull(pname);
711         if (prop2.isOptional()) {
712           if (prop1 != null
713               && !prop1.getType().isSubtypeOf(prop2.getType(), subSuperMap)) {
714             boxedInfo[0] = MismatchInfo.makePropTypeMismatch(
715                 pname, prop2.getType(), prop1.getType());
716             return false;
717           }
718         } else {
719           if (prop1 == null) {
720             boxedInfo[0] = MismatchInfo.makeMissingPropMismatch(pname);
721             return false;
722           } else if (prop1.isOptional()) {
723             boxedInfo[0] = MismatchInfo.makeMaybeMissingPropMismatch(pname);
724             return false;
725           } else if (!prop1.getType().isSubtypeOf(prop2.getType(), subSuperMap)) {
726             boxedInfo[0] = MismatchInfo.makePropTypeMismatch(
727                 pname, prop2.getType(), prop1.getType());
728             return false;
729           }
730         }
731         return true;
732       }

734       // We never infer properties as optional on loose objects,
735       // and we don't warn about possibly inexistent properties.
736       boolean isLooseSubtypeOf(ObjectType other, SubtypeCache subSuperMap) {
737         Preconditions.checkState(isLoose || other.isLoose);
738         if (other == TOP_OBJECT) {
739           return true;
740         }
741     
742         if (!isLoose) {
743           for (String pname : other.props.keySet()) {
744             QualifiedName qname = new QualifiedName(pname);
745             if (isStruct()) {
746               if (!mayHaveProp(qname)
747                   || !getProp(qname).isSubtypeOf(other.getProp(qname), subSuperMap)) {
748                 return false;
749               }
750             } else {
751               if (mayHaveProp(qname)
752                   && !getProp(qname).isSubtypeOf(other.getProp(qname), subSuperMap)) {
753                 return false;
754               }
755             }
756           }
757         } else { // this is loose, other may be loose
758           for (String pname : this.props.keySet()) {
759             QualifiedName qname = new QualifiedName(pname);
760             if (other.mayHaveProp(qname)
761                 && !getProp(qname).isSubtypeOf(other.getProp(qname), subSuperMap)) {
762               return false;
763             }
764           }
765         }
766     
767         if (other.fn == null) {
768           return this.fn == null || other.isLoose();
769         } else if (this.fn == null) {
770           return isLoose;
771         }
772         return fn.isLooseSubtypeOf(other.fn, subSuperMap);
773       }

775       ObjectType specialize(ObjectType other) {
776         Preconditions.checkState(
777             areRelatedNominalTypes(this.nominalType, other.nominalType));
778         if (this == TOP_OBJECT && other.objectKind.isUnrestricted()) {
779           return other;
780         }
781         if (this.ns != null) {
782           return specializeNamespace(other);
783         }
784         NominalType resultNomType =
785             NominalType.pickSubclass(this.nominalType, other.nominalType);
786         if (resultNomType != null && resultNomType.isClassy()) {
787           Preconditions.checkState(this.fn == null && other.fn == null);
788           PersistentMap<String, Property> newProps =
789               meetPropsHelper(true, resultNomType, this.props, other.props);
790           if (newProps == BOTTOM_MAP) {
791             return BOTTOM_OBJECT;
792           }
793           return new ObjectType(
794               resultNomType, newProps, null, this.ns, false, this.objectKind);
795         }
796         FunctionType thisFn = this.fn;
797         boolean isLoose = this.isLoose;
798         if (resultNomType != null && resultNomType.isFunction() && this.fn == null) {
799           thisFn = other.fn;
800           isLoose = other.fn.isLoose();
801         }
802         PersistentMap<String, Property> newProps =
803             meetPropsHelper(true, resultNomType, this.props, other.props);
804         if (newProps == BOTTOM_MAP) {
805           return BOTTOM_OBJECT;
806         }
807         FunctionType newFn = thisFn == null ? null : thisFn.specialize(other.fn);
808         if (!FunctionType.isInhabitable(newFn)) {
809           return BOTTOM_OBJECT;
810         }
811         return new ObjectType(
812             resultNomType, newProps, newFn, this.ns, isLoose, this.objectKind);
813       }

815       // If obj represents a type of the form {p1: p2: {... {p_n: A}}}
816       // then return the path p1,p2,...,p_n. Otherwise, return null.
817       private static QualifiedName getPropertyPath(ObjectType obj) {
818         if (obj.props.size() != 1) {
819           return null;
820         }
821         Map.Entry<String, Property> entry = obj.props.entrySet().iterator().next();
822         QualifiedName leftmostPname = new QualifiedName(entry.getKey());
823         ObjectType propAsObj = entry.getValue().getType().getObjTypeIfSingletonObj();
824     
825         if (propAsObj == null) {
826           return leftmostPname;
827         }
828         QualifiedName restPath = getPropertyPath(propAsObj);
829         if (restPath == null) {
830           return leftmostPname;
831         }
832         return QualifiedName.join(leftmostPname, restPath);
833       }

835       // Specializing namespace types is very expensive; not just the operation
836       // itself, but also the fact that you create a large type that you flow around
837       // later and many other expensive type operations happen on it.
838       // Therefore, we only specialize namespace types in a very specific case: to
839       // narrow down a mutable namespace field that has a union type, eg,
840       // if (goog.bar.baz !== null) { ... }
841       ObjectType specializeNamespace(ObjectType other) {
842         Preconditions.checkNotNull(this.ns);
843         if (this == other
844             || other.ns != null
845             || !other.getNominalType().equals(builtinObject)) {
846           return this;
847         }
848         QualifiedName propPath = getPropertyPath(other);
849         if (propPath == null) {
850           return this;
851         }
852         JSType otherPropType = other.getProp(propPath);
853         JSType thisPropType = mayHaveProp(propPath) ? getProp(propPath) : null;
854         JSType newPropType =
855             thisPropType == null ? null : thisPropType.specialize(otherPropType);
856         if (thisPropType != null
857             // Don't specialize for things like: if (goog.DEBUG) { ... }
858             && thisPropType.isUnion()
859             && !newPropType.isBottom()
860             && newPropType.isSubtypeOf(thisPropType)
861             && !thisPropType.isSubtypeOf(newPropType)) {
862           return withProperty(propPath, newPropType);
863         }
864         return this;
865       }

867       static ObjectType meet(ObjectType obj1, ObjectType obj2) {
868         Preconditions.checkState(
869             areRelatedNominalTypes(obj1.nominalType, obj2.nominalType));
870         if (obj1 == TOP_OBJECT) {
871           return obj2;
872         } else if (obj2 == TOP_OBJECT) {
873           return obj1;
874         }
875         NominalType resultNomType =
876             NominalType.pickSubclass(obj1.nominalType, obj2.nominalType);
877         FunctionType fn = FunctionType.meet(obj1.fn, obj2.fn);
878         if (!FunctionType.isInhabitable(fn)) {
879           return BOTTOM_OBJECT;
880         }
881         boolean isLoose = obj1.isLoose && obj2.isLoose || fn != null && fn.isLoose();
882         if (resultNomType != null && resultNomType.isFunction() && fn == null) {
883           fn = obj1.fn == null ? obj2.fn : obj1.fn;
884           isLoose = fn.isLoose();
885         }
886         PersistentMap<String, Property> props;
887         if (isLoose) {
888           props = joinPropsLoosely(obj1.props, obj2.props);
889         } else {
890           props = meetPropsHelper(false, resultNomType, obj1.props, obj2.props);
891         }
892         if (props == BOTTOM_MAP) {
893           return BOTTOM_OBJECT;
894         }
895         ObjectKind ok = ObjectKind.meet(obj1.objectKind, obj2.objectKind);
896         Namespace resultNs = Objects.equals(obj1.ns, obj2.ns) ? obj1.ns : null;
897         return new ObjectType(resultNomType, props, fn, resultNs, isLoose, ok);
898       }

900       private static ObjectType join(ObjectType obj1, ObjectType obj2) {
901         if (obj1 == TOP_OBJECT || obj2 == TOP_OBJECT) {
902           return TOP_OBJECT;
903         }
904         NominalType nom1 = obj1.nominalType;
905         NominalType nom2 = obj2.nominalType;
906         Preconditions.checkState(areRelatedNominalTypes(nom1, nom2));
907     
908         if (obj1.equals(obj2)) {
909           return obj1;
910         }
911         boolean isLoose = obj1.isLoose || obj2.isLoose;
912         FunctionType fn = FunctionType.join(obj1.fn, obj2.fn);
913         PersistentMap<String, Property> props;
914         if (isLoose) {
915           fn = fn == null ? null : fn.withLoose();
916           props = joinPropsLoosely(obj1.props, obj2.props);
917         } else {
918           props = joinProps(obj1.props, obj2.props, nom1, nom2);
919         }
920         NominalType nominal = NominalType.pickSuperclass(nom1, nom2);
921         // TODO(blickly): Split TOP_OBJECT from empty object and remove this case
922         if (nominal == null || !nominal.isFunction()) {
923           fn = null;
924         }
925         Namespace ns = Objects.equals(obj1.ns, obj2.ns) ? obj1.ns : null;
926         return makeObjectType(nominal, props, fn, ns, isLoose,
927             ObjectKind.join(obj1.objectKind, obj2.objectKind));
928       }

930       static ImmutableSet<ObjectType> joinSets(
931           ImmutableSet<ObjectType> objs1, ImmutableSet<ObjectType> objs2) {
932         if (objs1.isEmpty()) {
933           return objs2;
934         } else if (objs2.isEmpty()) {
935           return objs1;
936         }
937         ObjectType[] objs1Arr = objs1.toArray(new ObjectType[0]);
938         ObjectType[] keptFrom1 = Arrays.copyOf(objs1Arr, objs1Arr.length);
939         ImmutableSet.Builder<ObjectType> newObjs = ImmutableSet.builder();
940         for (ObjectType obj2 : objs2) {
941           boolean addedObj2 = false;
942           for (int i = 0; i < objs1Arr.length; i++) {
943             ObjectType obj1 = objs1Arr[i];
944             NominalType nominalType1 = obj1.nominalType;
945             NominalType nominalType2 = obj2.nominalType;
946             if (areRelatedNominalTypes(nominalType1, nominalType2)) {
947               if (nominalType2 == null && nominalType1 != null
948                   && !obj1.isSubtypeOf(obj2, SubtypeCache.create())
949                   || nominalType1 == null && nominalType2 != null
950                   && !obj2.isSubtypeOf(obj1, SubtypeCache.create())) {
951                 // Don't merge other classes with record types
952                 break;
953               }
954               keptFrom1[i] = null;
955               addedObj2 = true;
956               // obj1 and obj2 may be in a subtype relation.
957               // Even then, we want to join them because we don't want to forget
958               // any extra properties in the subtype object.
959               // TODO(dimvar): currently, a class and a @record that is a
960               // supertype can be in the same union. We must normalize like we do
961               // for other types, to maintain the invariant that the members of
962               // a union are not subtypes of each other.
963               newObjs.add(join(obj1, obj2));
964               break;
965             }
966           }
967           if (!addedObj2) {
968             newObjs.add(obj2);
969           }
970         }
971         for (ObjectType o : keptFrom1) {
972           if (o != null) {
973             newObjs.add(o);
974           }
975         }
976         return newObjs.build();
977       }

980         if (c1 == null || c2 == null) {
981           return true;
982         }
983         return c1.isNominalSubtypeOf(c2) || c2.isNominalSubtypeOf(c1);
984       }

986       // TODO(dimvar): handle greatest lower bound of interface types.
987       // If we do that, we need to normalize the output, otherwise it could contain
988       // two object types that are in a subtype relation, eg, see
989       // NewTypeInferenceES5OrLowerTest#testDifficultObjectSpecialization.
990       static ImmutableSet<ObjectType> meetSetsHelper(
991           boolean specializeObjs1,
992           Set<ObjectType> objs1, Set<ObjectType> objs2) {
993         ImmutableSet.Builder<ObjectType> newObjs = ImmutableSet.builder();
994         for (ObjectType obj2 : objs2) {
995           for (ObjectType obj1 : objs1) {
996             if (areRelatedNominalTypes(obj1.nominalType, obj2.nominalType)) {
997               ObjectType newObj;
998               if (specializeObjs1) {
999                 newObj = obj1.specialize(obj2);
1000                 if (newObj == null) {
1001                   continue;
1002                 }
1003               } else {
1004                 newObj = meet(obj1, obj2);
1005               }
1006               newObjs.add(newObj);
1007             }
1008           }
1009         }
1010         return newObjs.build();
1011       }

1093       /**
1094        * Unify the two types symmetrically, given that we have already instantiated
1095        * the type variables of interest in {@code t1} and {@code t2}, treating
1096        * JSType.UNKNOWN as a "hole" to be filled.
1097        * @return The unified type, or null if unification fails
1098        */
1099       static ObjectType unifyUnknowns(ObjectType t1, ObjectType t2) {
1100         // t1 and/or t2 may be loose in cases where there is a union of loose and
1101         // non-loose types, eg, !Array<!Function|string>|!Function.
1102         // Maybe the equality check below is too strict, we may end up checking for
1103         // subtyping, depending on how people use these in code.
1104         if (t1.isLoose()) {
1105           return t1.equals(t2) ? t1 : null;
1106         }
1107         if (t2.isLoose()) {
1108           return null;
1109         }
1110         if (!Objects.equals(t1.ns, t2.ns)) {
1111           return null;
1112         }
1113         NominalType nt1 = t1.nominalType;
1114         NominalType nt2 = t2.nominalType;
1115         NominalType nt;
1116         if (nt1 == null && nt2 == null) {
1117           nt = null;
1118         } else if (nt1 == null || nt2 == null) {
1119           return null;
1120         } else {
1121           nt = NominalType.unifyUnknowns(nt1, nt2);
1122           if (nt == null) {
1123             return null;
1124           }
1125         }
1126         FunctionType newFn = null;
1127         if (t1.fn != null || t2.fn != null) {
1128           newFn = FunctionType.unifyUnknowns(t1.fn, t2.fn);
1129           if (newFn == null) {
1130             return null;
1131           }
1132         }
1133         PersistentMap<String, Property> newProps = PersistentMap.create();
1134         for (String propName : t1.props.keySet()) {
1135           Property prop1 = t1.props.get(propName);
1136           Property prop2 = t2.props.get(propName);
1137           if (prop2 == null) {
1138             return null;
1139           }
1140           Property p = Property.unifyUnknowns(prop1, prop2);
1141           if (p == null) {
1142             return null;
1143           }
1144           newProps = newProps.with(propName, p);
1145         }
1146         return makeObjectType(nt, newProps, newFn, t1.ns, false,
1147             ObjectKind.join(t1.objectKind, t2.objectKind));
1148       }

1150       /**
1151        * Unify {@code this}, which may contain free type variables,
1152        * with {@code other}, a concrete type, modifying the supplied
1153        * {@code typeMultimap} to add any new template varaible type bindings.
1154        * @return Whether unification succeeded
1155        */
1156       boolean unifyWithSubtype(ObjectType other, List<String> typeParameters,
1157           Multimap<String, JSType> typeMultimap, SubtypeCache subSuperMap) {
1158         if (fn != null) {
1159           if (other.fn == null ||
1160               !fn.unifyWithSubtype(other.fn, typeParameters, typeMultimap, subSuperMap)) {
1161             return false;
1162           }
1163         }
1164         NominalType thisNt = this.nominalType;
1165         NominalType otherNt = other.nominalType;
1166         if (thisNt != null && otherNt != null) {
1167           if (thisNt.unifyWithSubtype(
1168               otherNt, typeParameters, typeMultimap, subSuperMap)) {
1169             return true;
1170           }
1171           if (thisNt.isClass()) {
1172             return false;
1173           }
1174           if (thisNt.isStructuralInterface()) {
1175             if (thisNt.equals(subSuperMap.get(otherNt))) {
1176               return true;
1177             }
1178             subSuperMap = subSuperMap.with(otherNt, thisNt);
1179           }
1180         }
1181         if (thisNt != null && !thisNt.isStructuralInterface() && otherNt == null) {
1182           return false;
1183         }
1184         Set<String> thisProps = thisNt != null && thisNt.isStructuralInterface()
1185             ? thisNt.getAllPropsOfInterface() : this.props.keySet();
1186         return unifyPropsWithSubtype(
1187             other, thisProps, typeParameters, typeMultimap, subSuperMap);
1188       }

1190       private boolean unifyPropsWithSubtype(ObjectType other,
1191           Set<String> thisProps, List<String> typeParameters,
1192           Multimap<String, JSType> typeMultimap, SubtypeCache subSuperMap) {
1193         for (String pname : thisProps) {

1199                 && !thisProp.getType().unifyWithSubtype(
1200                     otherProp.getType(), typeParameters, typeMultimap, subSuperMap)) {
1201               return false;

1203           } else if (otherProp == null || otherProp.isOptional()
1204               || !thisProp.getType().unifyWithSubtype(

1220               propsEntry.getValue().substituteGenerics(concreteTypes);
