152       /**
153        * Sets the prefix used for wrapping the generated source file before
154        * it is written. This ensures that the source map is adjusted for the
155        * change in character offsets.
156        *
157        * @param prefix The prefix that is added before the generated source code.
158        */
159       @Override
160       public void setWrapperPrefix(String prefix) {
161         // Determine the current line and character position.
162         int prefixLine = 0;
163         int prefixIndex = 0;
164     
165         for (int i = 0; i < prefix.length(); ++i) {
166           if (prefix.charAt(i) == '\n') {
167             prefixLine++;
168             prefixIndex = 0;
169           } else {
170             prefixIndex++;
171           }
172         }
173     
174         prefixPosition = new FilePosition(prefixLine, prefixIndex);
175       }

178        * Sets the source code that exists in the buffer for which the
179        * generated code is being generated. This ensures that the source map
180        * accurately reflects the fact that the source is being appended to
181        * an existing buffer and as such, does not start at line 0, position 0
182        * but rather some other line and position.
183        *
184        * @param offsetLine The index of the current line being printed.
185        * @param offsetIndex The column index of the current character being printed.
186        */
187       @Override
188       public void setStartingPosition(int offsetLine, int offsetIndex) {
189         Preconditions.checkState(offsetLine >= 0);
190         Preconditions.checkState(offsetIndex >= 0);
191         offsetPosition = new FilePosition(offsetLine, offsetIndex);
192       }

194       /**
195        * Adds a mapping for the given node.  Mappings must be added in order.
196        * @param startPosition The position on the starting line
197        * @param endPosition The position on the ending line.
198        */
199       @Override
200       public void addMapping(
201           String sourceName, @Nullable String symbolName,
202           FilePosition sourceStartPosition,
203           FilePosition startPosition, FilePosition endPosition) {
204     
205         // Don't bother if there is not sufficient information to be useful.
206         if (sourceName == null || sourceStartPosition.getLine() < 0) {
207           return;
208         }
209     
210         FilePosition adjustedStart = startPosition;
211         FilePosition adjustedEnd = endPosition;
212     
213         if (offsetPosition.getLine() != 0
214             || offsetPosition.getColumn() != 0) {
215           // If the mapping is found on the first line, we need to offset
216           // its character position by the number of characters found on
217           // the *last* line of the source file to which the code is
218           // being generated.
219           int offsetLine = offsetPosition.getLine();
220           int startOffsetPosition = offsetPosition.getColumn();
221           int endOffsetPosition = offsetPosition.getColumn();
222     
223           if (startPosition.getLine() > 0) {
224             startOffsetPosition = 0;
225           }
226     
227           if (endPosition.getLine() > 0) {
228             endOffsetPosition = 0;
229           }
230     
231           adjustedStart = new FilePosition(
232               startPosition.getLine() + offsetLine,
233               startPosition.getColumn() + startOffsetPosition);
234     
235           adjustedEnd = new FilePosition(
236               endPosition.getLine() + offsetLine,
237               endPosition.getColumn() + endOffsetPosition);
238         }
239     
240         // Create the new mapping.
241         Mapping mapping = new Mapping();
242         mapping.sourceFile = sourceName;
243         mapping.originalPosition = sourceStartPosition;
244         mapping.originalName = symbolName;
245         mapping.startPosition = adjustedStart;
246         mapping.endPosition = adjustedEnd;
247     
248         // Validate the mappings are in a proper order.
249         if (lastMapping != null) {
250           int lastLine = lastMapping.startPosition.getLine();
251           int lastColumn = lastMapping.startPosition.getColumn();
252           int nextLine = mapping.startPosition.getLine();
253           int nextColumn = mapping.startPosition.getColumn();
254           Preconditions.checkState(nextLine > lastLine
255               || (nextLine == lastLine && nextColumn >= lastColumn),
256               "Incorrect source mappings order, previous : (%s,%s)\n"
257               + "new : (%s,%s)",
258               lastLine, lastColumn, nextLine, nextColumn);
259         }
260     
261         lastMapping = mapping;
262         mappings.add(mapping);
263       }

278        * Merges current mapping with {@code mapSectionContents} considering the
279        * offset {@code (line, column)}. Any extension in the map section will be
280        * ignored.
281        *
282        * @param line The line offset
283        * @param column The column offset
284        * @param mapSectionContents The map section to be appended
285        * @throws SourceMapParseException

287       public void mergeMapSection(int line, int column, String mapSectionContents)
288           throws SourceMapParseException {
289         setStartingPosition(line, column);
290         SourceMapConsumerV3 section = new SourceMapConsumerV3();
291         section.parse(mapSectionContents);
292         section.visitMappings(new ConsumerEntryVisitor());
293       }

295       /**
296        * Works like {@link #mergeMapSection(int, int, String)}, except that
297        * extensions from the @{code mapSectionContents} are merged to the top level
298        * source map. For conflicts a {@code mergeAction} is performed.
299        *
300        * @param line The line offset
301        * @param column The column offset
302        * @param mapSectionContents The map section to be appended
303        * @param mergeAction The merge action for conflicting extensions
304        * @throws SourceMapParseException
305        */
306       public void mergeMapSection(int line, int column, String mapSectionContents,
307           ExtensionMergeAction mergeAction)
308           throws SourceMapParseException {
309         setStartingPosition(line, column);
310         SourceMapConsumerV3 section = new SourceMapConsumerV3();
311         section.parse(mapSectionContents);
312         section.visitMappings(new ConsumerEntryVisitor());
313         for (Entry<String, Object> entry : section.getExtensions().entrySet()) {
314            String extensionKey = entry.getKey();
315            if (extensions.containsKey(extensionKey)) {
316              extensions.put(extensionKey,
317                  mergeAction.merge(extensionKey,
318                                    extensions.get(extensionKey),
319                                    entry.getValue()));
320            } else {
321              extensions.put(extensionKey, entry.getValue());
322            }
323          }
324       }

326       /**
327        * Writes out the source map in the following format (line numbers are for
328        * reference only and are not part of the format):
329        *
330        * 1.  {
331        * 2.    version: 3,
332        * 3.    file: "out.js",
333        * 4.    lineCount: 2,
334        * 5.    sourceRoot: "",
335        * 6.    sources: ["foo.js", "bar.js"],
336        * 7.    names: ["src", "maps", "are", "fun"],
337        * 8.    mappings: "a;;abcde,abcd,a;"
338        * 9.    x_org_extension: value
339        * 10. }
340        *
341        * Line 1: The entire file is a single JSON object
342        * Line 2: File revision (always the first entry in the object)
343        * Line 3: The name of the file that this source map is associated with.
344        * Line 4: The number of lines represented in the source map.
345        * Line 5: An optional source root, useful for relocating source files on a
346        *     server or removing repeated prefix values in the "sources" entry.
347        * Line 6: A list of sources used by the "mappings" entry relative to the
348        *     sourceRoot.
349        * Line 7: A list of symbol names used by the "mapping" entry.  This list
350        *     may be incomplete.
351        * Line 8: The mappings field.
352        * Line 9: Any custom field (extension).
353        */
354       @Override
355       public void appendTo(Appendable out, String name) throws IOException {
356         int maxLine = prepMappings() + 1;
357     
358         // Add the header fields.
359         out.append("{\n");
360         appendFirstField(out, "version", "3");
361         appendField(out, "file", escapeString(name));
362         appendField(out, "lineCount", String.valueOf(maxLine));
363     
364         //optional source root
365         if (this.sourceRootPath != null && !this.sourceRootPath.isEmpty()) {
366           appendField(out, "sourceRoot", escapeString(this.sourceRootPath));
367         }
368     
369         // Add the mappings themselves.
370         appendFieldStart(out, "mappings");
371         // out.append("[");
372         (new LineMapper(out, maxLine)).appendLineMappings();
373     
374         // out.append("]");
375         appendFieldEnd(out);
376     
377         // Files names
378         appendFieldStart(out, "sources");
379         out.append("[");
380         addSourceNameMap(out);
381         out.append("]");
382         appendFieldEnd(out);
383     
384         // Files names
385         appendFieldStart(out, "names");
386         out.append("[");
387         addSymbolNameMap(out);
388         out.append("]");
389         appendFieldEnd(out);
390     
391         // Extensions, only if there is any
392         for (String key : this.extensions.keySet()) {
393           Object objValue = this.extensions.get(key);
394           String value = objValue.toString();
395           if (objValue instanceof String){
396             value = new Gson().toJson(value);
397           }
398           appendField(out, key, value);
399         }
400     
401         out.append("\n}\n");
402       }

534       /**
535        * Assigns sequential ids to used mappings, and returns the last line mapped.
536        */
537       private int prepMappings() throws IOException {
538         // Mark any unused mappings.
539         (new MappingTraversal()).traverse(new UsedMappingCheck());
540     
541         // Renumber used mappings and keep track of the last line.
542         int id = 0;
543         int maxLine = 0;
544         for (Mapping m : mappings) {
545           if (m.used) {
546             m.id = id++;
547             int endPositionLine = m.endPosition.getLine();
548             maxLine = Math.max(maxLine, endPositionLine);
549           }
550         }
551     
552         // Adjust for the prefix.
553         return maxLine + prefixPosition.getLine();
554       }

618       private interface MappingVisitor {
619         /**
620          * @param m The mapping for the current code segment. null if the segment
621          *     is unmapped.
622          * @param line The starting line for this code segment.
623          * @param col The starting column for this code segment.
624          * @param endLine The ending line
625          * @param endCol The ending column
626          * @throws IOException
627          */
628         void visit(Mapping m, int line, int col, int endLine, int endCol)
629             throws IOException;
630       }

632       /**
633        * Walk the mappings and visit each segment of the mappings, unmapped
634        * segments are visited with a null mapping, unused mapping are not visited.
635        */
636       private class MappingTraversal {
637         // The last line and column written
638         private int line;
639         private int col;
640     
641         MappingTraversal() {
642         }
643     
644         // Append the line mapping entries.
645         void traverse(MappingVisitor v) throws IOException {
646           // The mapping list is ordered as a pre-order traversal.  The mapping
647           // positions give us enough information to rebuild the stack and this
648           // allows the building of the source map in O(n) time.
649           Deque<Mapping> stack = new ArrayDeque<>();
650           for (Mapping m : mappings) {
651             // Find the closest ancestor of the current mapping:
652             // An overlapping mapping is an ancestor of the current mapping, any
653             // non-overlapping mappings are siblings (or cousins) and must be
654             // closed in the reverse order of when they encountered.
655             while (!stack.isEmpty() && !isOverlapped(stack.peek(), m)) {
656               Mapping previous = stack.pop();
657               maybeVisit(v, previous);
658             }
659     
660             // Any gaps between the current line position and the start of the
661             // current mapping belong to the parent.
662             Mapping parent = stack.peek();
663             maybeVisitParent(v, parent, m);
664     
665             stack.push(m);
666           }
667     
668           // There are no more children to be had, simply close the remaining
669           // mappings in the reverse order of when they encountered.
670           while (!stack.isEmpty()) {
671             Mapping m = stack.pop();
672             maybeVisit(v, m);
673           }
674         }

676         /**
677          * @return The line adjusted for the prefix position.
678          */
679         private int getAdjustedLine(FilePosition p) {
680           return p.getLine() + prefixPosition.getLine();
681         }
682     
683         /**
684          * @return The column adjusted for the prefix position.
685          */
686         private int getAdjustedCol(FilePosition p) {
687           int rawLine = p.getLine();
688           int rawCol = p.getColumn();
689           // Only the first line needs the character position adjusted.
690           return (rawLine != 0)
691               ? rawCol : rawCol + prefixPosition.getColumn();
692         }
693     
694         /**
695          * @return Whether m1 ends before m2 starts.
696          */
697         private boolean isOverlapped(Mapping m1, Mapping m2) {
698           // No need to use adjusted values here, relative positions are sufficient.
699           int l1 = m1.endPosition.getLine();
700           int l2 = m2.startPosition.getLine();
701           int c1 = m1.endPosition.getColumn();
702           int c2 = m2.startPosition.getColumn();
703     
704           return (l1 == l2 && c1 >= c2) || l1 > l2;
705         }
706     
707         /**
708          * Write any needed entries from the current position to the end of the
709          * provided mapping.
710          */
711         private void maybeVisit(MappingVisitor v, Mapping m) throws IOException {
712           int nextLine = getAdjustedLine(m.endPosition);
713           int nextCol = getAdjustedCol(m.endPosition);
714           // If this anything remaining in this mapping beyond the
715           // current line and column position, write it out now.
716           if (line < nextLine || (line == nextLine && col < nextCol)) {
717             visit(v, m, nextLine, nextCol);
718           }
719         }
720     
721         /**
722          * Write any needed entries to complete the provided mapping.
723          */
724         private void maybeVisitParent(MappingVisitor v, Mapping parent, Mapping m)
725             throws IOException {
726           int nextLine = getAdjustedLine(m.startPosition);
727           int nextCol = getAdjustedCol(m.startPosition);
728           // If the previous value is null, no mapping exists.
729           Preconditions.checkState(line < nextLine || col <= nextCol);
730           if (line < nextLine || (line == nextLine && col < nextCol)) {
731             visit(v, parent, nextLine, nextCol);
732           }
733         }
734     
735         /**
736          * Write any entries needed between the current position the next position
737          * and update the current position.
738          */
739         private void visit(MappingVisitor v, Mapping m,
740             int nextLine, int nextCol)
741             throws IOException {
742           Preconditions.checkState(line <= nextLine);
743           Preconditions.checkState(line < nextLine || col < nextCol);
744     
745           if (line == nextLine && col == nextCol) {
746             // Nothing to do.
747             throw new IllegalStateException();
748           }
749     
750           v.visit(m, line, col, nextLine, nextCol);
751     
752           line = nextLine;
753           col = nextCol;
754         }
755       }

757       /**
758        * Appends the index source map to the given buffer.
759        *
760        * @param out The stream to which the map will be appended.
761        * @param name The name of the generated source file that this source map
762        *   represents.
763        * @param sections An ordered list of map sections to include in the index.
764        * @throws IOException
765        */
766       @Override
767       public void appendIndexMapTo(
768           Appendable out, String name, List<SourceMapSection> sections)
769           throws IOException {
770         // Add the header fields.
771         out.append("{\n");
772         appendFirstField(out, "version", "3");
773         appendField(out, "file", escapeString(name));
774     
775         // Add the line character maps.
776         appendFieldStart(out, "sections");
777         out.append("[\n");
778         boolean first = true;
779         for (SourceMapSection section : sections) {
780           if (first) {
781             first = false;
782           } else {
783             out.append(",\n");
784           }
785           out.append("{\n");
786           appendFirstField(out, "offset",
787               offsetValue(section.getLine(), section.getColumn()));
788           if (section.getSectionType() == SourceMapSection.SectionType.URL) {
789             appendField(out, "url", escapeString(section.getSectionValue()));
790           } else if (section.getSectionType() == SourceMapSection.SectionType.MAP) {
791             appendField(out, "map", section.getSectionValue());
792           } else {
793             throw new IOException("Unexpected section type");
794           }
795           out.append("\n}");
796         }
797     
798         out.append("\n]");
799         appendFieldEnd(out);
800     
801         out.append("\n}\n");
802       }

804       private static CharSequence offsetValue(int line, int column) throws IOException {
805         StringBuilder out = new StringBuilder();
806         out.append("{\n");
807         appendFirstField(out, "line", String.valueOf(line));
808         appendField(out, "column", String.valueOf(column));
809         out.append("\n}");
810         return out;
811       }

813       private int getSourceId(String sourceName) {
814         if (sourceName != lastSourceFile) {
815           lastSourceFile = sourceName;
816           Integer index = sourceFileMap.get(sourceName);
817           if (index != null) {
818             lastSourceFileIndex = index;
819           } else {
820             lastSourceFileIndex = sourceFileMap.size();
821             sourceFileMap.put(sourceName, lastSourceFileIndex);
822           }
823         }
824         return lastSourceFileIndex;
825       }
826     
827       private int getNameId(String symbolName) {
828         int originalNameIndex;
829         Integer index = originalNameMap.get(symbolName);
830         if (index != null) {
831           originalNameIndex = index;
832         } else {
833           originalNameIndex = originalNameMap.size();
834           originalNameMap.put(symbolName, originalNameIndex);
835         }
836         return originalNameIndex;
837       }

839       private class LineMapper implements MappingVisitor {
840         // The destination.
841         private final Appendable out;
842         private final int maxLine; // TODO(johnlenz): This shouldn't be necessary to track.
843     
844         private int previousLine = -1;
845         private int previousColumn = 0;
846     
847         // Previous values used for storing relative ids.
848         private int previousSourceFileId;
849         private int previousSourceLine;
850         private int previousSourceColumn;
851         private int previousNameId;
852     
853         LineMapper(Appendable out, int maxLine) {
854           this.out = out;
855           this.maxLine = maxLine;
856         }
857     
858         /**
859          * As each segment is visited write out the appropriate line mapping.
860          */
861         @Override
862         public void visit(Mapping m, int line, int col, int nextLine, int nextCol)
863           throws IOException {
864           if (previousLine != line) {
865             previousColumn = 0;
866           }
867     
868           if (line != nextLine || col != nextCol) {
869             // TODO(johnlenz): For some reason, we have mappings beyond the max line.
870             // So far they're just null mappings and we can ignore them.
871             // (If they're non-null, we assert-fail.)
872             if (line < maxLine) {
873               if (previousLine == line) { // not the first entry for the line
874                 out.append(',');
875               }
876               writeEntry(m, col);
877               previousLine = line;
878               previousColumn = col;
879             } else {
880               Preconditions.checkState(m == null);
881             }
882           }
883     
884           for (int i = line; i <= nextLine && i < maxLine; i++) {
885             if (i == nextLine) {
886               break;
887             }
888     
889             closeLine(false);
890             openLine(false);
891           }
892         }
893     
894         /**
895          * Writes an entry for the given column (of the generated text) and
896          * associated mapping.
897          * The values are stored as relative to the last seen values for each
898          * field and encoded as Base64VLQs.
899          */
900         void writeEntry(Mapping m, int column) throws IOException {
901           // The relative generated column number
902           Base64VLQ.encode(out, column - previousColumn);
903           previousColumn = column;
904           if (m != null) {
905             // The relative source file id
906             int sourceId = getSourceId(m.sourceFile);
907             Base64VLQ.encode(out, sourceId - previousSourceFileId);
908             previousSourceFileId = sourceId;
909     
910             // The relative source file line and column
911             int srcline = m.originalPosition.getLine();
912             int srcColumn = m.originalPosition.getColumn();
913             Base64VLQ.encode(out, srcline - previousSourceLine);
914             previousSourceLine = srcline;
915     
916             Base64VLQ.encode(out, srcColumn - previousSourceColumn);
917             previousSourceColumn = srcColumn;
918     
919             if (m.originalName != null) {
920               // The relative id for the associated symbol name
921               int nameId = getNameId(m.originalName);
922               Base64VLQ.encode(out, (nameId - previousNameId));
923               previousNameId = nameId;
924             }
925           }
926         }
927     
928         // Append the line mapping entries.
929         void appendLineMappings() throws IOException {
930           // Start the first line.
931           openLine(true);
932     
933           (new MappingTraversal()).traverse(this);
934     
935           // And close the final line.
936           closeLine(true);
937         }
938     
939         /**
940          * Begin the entry for a new line.
941          */
942         private void openLine(boolean firstEntry) throws IOException {
943           if (firstEntry) {
944             out.append('\"');
945           }
946         }
947     
948         /**
949          * End the entry for a line.
950          */
951         private void closeLine(boolean finalEntry) throws IOException {
952           out.append(';');
953           if (finalEntry) {
954             out.append('\"');
955           }
956         }
957       }
