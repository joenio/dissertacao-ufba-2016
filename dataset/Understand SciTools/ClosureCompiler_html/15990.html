40       /**
41        * Returns a Node tree with the post-order traversal a b c d e f g h i j k l m
42        * and the in-order traversal m d a b c h e f g l i j k:
43        *
44        *                                   m
45        *                         ,---------|---------.
46        *                         d         h         l
47        *                      ,--|--.   ,--|--.   ,--|--.
48        *                      a  b  c   e  f  g   i  j  k
49        *
50        */
51       private static Node createPostOrderAlphabet() {
52         Node a = Node.newString("a");
53         Node b = Node.newString("b");
54         Node c = Node.newString("c");
55         Node d = Node.newString("d");
56         Node e = Node.newString("e");
57         Node f = Node.newString("f");
58         Node g = Node.newString("g");
59         Node h = Node.newString("h");
60         Node i = Node.newString("i");
61         Node j = Node.newString("j");
62         Node k = Node.newString("k");
63         Node l = Node.newString("l");
64         Node m = Node.newString("m");
65     
66         d.addChildToBack(a);
67         d.addChildToBack(b);
68         d.addChildToBack(c);
69     
70         h.addChildrenToBack(e);
71         h.addChildrenToBack(f);
72         h.addChildrenToBack(g);
73     
74         l.addChildToBack(i);
75         l.addChildToBack(j);
76         l.addChildToBack(k);
77     
78         m.addChildToBack(d);
79         m.addChildToBack(h);
80         m.addChildToBack(l);
81     
82         return m;
83       }

91       /**
92        * Concatenates contents of string nodes encountered in pre-order
93        * and post-order traversals. Abbreviates traversals by ignoring subtrees
94        * rooted with specified strings.
95        */
96       private static class ConcatTraversal implements Callback {
97         private StringBuilder visited = new StringBuilder();
98         private StringBuilder shouldTraversed = new StringBuilder();
99         private Set<String> ignoring = new HashSet<>();
100     
101         ConcatTraversal ignore(String s) {
102           ignoring.add(s);
103           return this;
104         }
105     
106         @Override
107         public void visit(NodeTraversal t, Node n, Node parent) {
108           assertEquals(Token.STRING, n.getType());
109           visited.append(n.getString());
110         }
111     
112         @Override
113         public boolean shouldTraverse(NodeTraversal t, Node n, Node parent) {
114           assertEquals(Token.STRING, n.getType());
115           shouldTraversed.append(n.getString());
116           return !ignoring.contains(n.getString());
117         }
118     
119         /** Returns strings concatenated during post-order traversal. */
120         String getVisited() {
121           return visited.toString();
122         }
123     
124         /** Returns strings concatenated during pre-order traversal. */
125         String getShouldTraversed() {
126           return shouldTraversed.toString();
127         }
128     
129         Collection<String> getIgnoring() {
130           return ignoring;
131         }
132       }
133     
134       /**
135        * Collection of data for a traversal test. Contains the traversal callback
136        * and the expected pre- and post-order traversal results.
137        */
138       private static class TestHelper {
139         private ConcatTraversal traversal;
140         private String expectedVisited;
141         private String shouldTraverseExpected;
142     
143         TestHelper(ConcatTraversal traversal, String expectedVisited,
144              String shouldTraverseExpected) {
145           this.traversal = traversal;
146           this.expectedVisited = expectedVisited;
147           this.shouldTraverseExpected = shouldTraverseExpected;
148         }
149     
150         ConcatTraversal getTraversal() {
151           return traversal;
152         }
153     
154         void checkResults() {
155           assertEquals(
156               "ConcatTraversal ignoring " + traversal.getIgnoring()
157               + " has unexpected visiting order",
158               expectedVisited, traversal.getVisited());
159     
160           assertEquals(
161               "ConcatTraversal ignoring " + traversal.getIgnoring()
162               + " has unexpected traversal order",
163               shouldTraverseExpected, traversal.getShouldTraversed());
164         }
165       }
166     
167       private static List<TestHelper> createStringTests() {
168         List<TestHelper> tests = new ArrayList<>();
169     
170         tests.add(new TestHelper(
171             new ConcatTraversal(), "abcdefghijklm", "mdabchefglijk"));
172     
173         tests.add(new TestHelper(
174             new ConcatTraversal().ignore("d"), "efghijklm", "mdhefglijk"));
175     
176         tests.add(new TestHelper(
177             new ConcatTraversal().ignore("f"), "abcdeghijklm", "mdabchefglijk"));
178     
179         tests.add(new TestHelper(new ConcatTraversal().ignore("m"), "", "m"));
180     
181         return tests;
182       }

184       public void testIndividualPasses() {
185         for (TestHelper test : createStringTests()) {
186           CombinedCompilerPass pass =
187               new CombinedCompilerPass(compiler, test.getTraversal());
188           pass.process(null, createPostOrderAlphabet());
189           test.checkResults();
190         }
191       }
192     
193       public void testCombinedPasses() {
194         List<TestHelper> tests  = createStringTests();
195         Callback[] callbacks = new Callback[tests.size()];
196         int i = 0;
197         for (TestHelper test : tests) {
198           callbacks[i++] = test.getTraversal();
199         }
200         CombinedCompilerPass pass =
201             new CombinedCompilerPass(compiler, callbacks);
202         pass.process(null, createPostOrderAlphabet());
203         for (TestHelper test : tests) {
204           test.checkResults();
205         }
206       }

208       /**
209        * Records the scopes visited during an AST traversal. Abbreviates traversals
210        * by ignoring subtrees rooted with specified NAME nodes.
211        */
212       private static class ScopeRecordingCallback implements ScopedCallback {
213     
214         Set<Node> visitedScopes = new HashSet<>();
215         Set<String> ignoring = new HashSet<>();
216     
217         void ignore(String name) {
218           ignoring.add(name);
219         }
220     
221         @Override
222         public void enterScope(NodeTraversal t) {
223           visitedScopes.add(t.getScopeRoot());
224         }
225     
226         @Override
227         public boolean shouldTraverse(NodeTraversal t, Node n, Node parent) {
228           return !n.isName() || !ignoring.contains(n.getString());
229         }
230     
231         Set<Node> getVisitedScopes() {
232           return visitedScopes;
233         }
234     
235         @Override
236         public void exitScope(NodeTraversal t) {
237         }
238     
239         @Override
240         public void visit(NodeTraversal t, Node n, Node parent) {
241         }
242     
243       }
244     
245       public void testScopes() {
246         Node root =
247             compiler.parseTestCode("var y = function() { var x = function() { };}");
248     
249         ScopeRecordingCallback c1 = new ScopeRecordingCallback();
250         c1.ignore("y");
251         ScopeRecordingCallback c2 = new ScopeRecordingCallback();
252         c2.ignore("x");
253         ScopeRecordingCallback c3 = new ScopeRecordingCallback();
254     
255         CombinedCompilerPass pass = new CombinedCompilerPass(compiler, c1, c2, c3);
256         pass.process(null, root);
257     
258         assertThat(c1.getVisitedScopes()).hasSize(1);
259         assertThat(c2.getVisitedScopes()).hasSize(3);
260         assertThat(c3.getVisitedScopes()).hasSize(5);
261       }
