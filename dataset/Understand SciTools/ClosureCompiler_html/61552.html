17     package com.google.javascript.jscomp;
18     
19     import com.google.common.base.Preconditions;
20     import com.google.javascript.rhino.Node;
21     import com.google.javascript.rhino.Token;
22     
23     /**
24      * Tries to fuse all the statements in a block into a one statement by using
25      * COMMAs.
26      *
27      * Because COMMAs has the lowest precedence, we never need to insert
28      * extra () around. Once we have only one statement in a block, we can then
29      * eliminate a pair of {}'s. Further more, we can also fold a single
30      * statement IF into && or create further opportunities for all the other
31      * goodies in {@link PeepholeMinimizeConditions}.
32      *
33      */
34     class StatementFusion extends AbstractPeepholeOptimization {
35       // TODO(user): We probably need to test this more. The current compiler
36       // assumes that there are more ;'s than ,'s in a real program. However,
37       // this assumption may be incorrect. We can probably do a quick traverse
38       // to check this assumption if that's necessary.
39       public static final boolean SHOULD_FAVOR_COMMA_OVER_SEMI_COLON = false;
40     
41       private final boolean favorsCommaOverSemiColon;
42     
43       public StatementFusion() {
44         this(SHOULD_FAVOR_COMMA_OVER_SEMI_COLON);
45       }
46     
47       public StatementFusion(boolean favorsCommaOverSemiColon) {
48         this.favorsCommaOverSemiColon = favorsCommaOverSemiColon;
49       }
50     
51       @Override
52       Node optimizeSubtree(Node n) {
53         // TODO(user): It is much cleaner to have two algorithms depending
54         // on favorsCommaOverSemiColon. If we decided the less aggressive one is
55         // no longer useful, delete it.
56         if (favorsCommaOverSemiColon) {
57           return tryFuseStatementsAggressively(n);
58         } else {
59           return tryFuseStatements(n);
60         }
61       }
62     
63       Node tryFuseStatements(Node n) {
64         if (!n.getParent().isFunction() && canFuseIntoOneStatement(n)) {
65           Node start = n.getFirstChild();
66           Node end = n.getLastChild();
67           Node result = fuseIntoOneStatement(n, start, end);
68           fuseExpressionIntoControlFlowStatement(result, n.getLastChild());
69           reportCodeChange();
70         }
71         return n;
72       }
73     
74       Node tryFuseStatementsAggressively(Node n) {
75         if (!NodeUtil.isStatementBlock(n)) {
76           return n;
77         }
78     
79         Node cur = n.getFirstChild();
80         while (cur != null) {
81           if (!cur.isExprResult()) {
82             cur = cur.getNext();
83             continue;
84           }
85           Node next = cur.getNext();
86           while (next != null && next.isExprResult()) {
87             next = next.getNext();
88           }
89           if (cur.getNext() != next) {
90             cur = fuseIntoOneStatement(n, cur, next);
91             reportCodeChange();
92           }
93           if (cur.isExprResult() &&
94               next != null && isFusableControlStatement(next)) {
95             fuseExpressionIntoControlFlowStatement(cur, next);
96             reportCodeChange();
97             next = next.getNext();
98           }
99           cur = next;
100         }
101     
102         return n;
103       }
104     
105       private boolean canFuseIntoOneStatement(Node block) {
106         // If we are favoring semi-colon, we shouldn't fuse script blocks.
107         if (!favorsCommaOverSemiColon && !block.isBlock()) {
108           return false;
109         }
110     
111         // Nothing to do here.
112         if (!block.hasChildren() || block.hasOneChild()) {
113           return false;
114         }
115     
116         Node last = block.getLastChild();
117     
118         for (Node c = block.getFirstChild(); c != null; c = c.getNext()) {
119           if (!c.isExprResult() && c != last) {
120             return false;
121           }
122         }
123     
124         return isFusableControlStatement(last);
125       }
126     
127       private boolean isFusableControlStatement(Node n) {
128         switch(n.getType()) {
129           case IF:
130           case THROW:
131           case SWITCH:
132           case EXPR_RESULT:
133             return true;
134           case RETURN:
135             // We don't want to add a new return value.
136             return n.hasChildren();
137           case FOR:
138             if (NodeUtil.isForIn(n)) {
139               // Avoid cases where we have for(var x = foo() in a) { ....
140               return !mayHaveSideEffects(n.getFirstChild());
141             } else {
142               // Avoid cases where we have for(var x;_;_) { ....
143               return !n.getFirstChild().isVar();
144             }
145           case LABEL:
146             return isFusableControlStatement(n.getLastChild());
147           case BLOCK:
148             return !n.isSyntheticBlock() &&
149                 isFusableControlStatement(n.getFirstChild());
150         }
151         return false;
152       }
153     
154       /**
155        * Given a block, fuse a list of statements with comma's.
156        *
157        * @param parent The parent that contains the statements.
158        * @param first The first statement to fuse (inclusive)
159        * @param last The last statement to fuse (exclusive)
160        * @return A single statement that contains all the fused statement as one.
161        */
162       private static Node fuseIntoOneStatement(Node parent, Node first, Node last) {
163         // Nothing to fuse if there is only one statement.
164         if (first.getNext() == last) {
165           return first;
166         }
167     
168         // Step one: Create a comma tree that contains all the statements.
169         Node commaTree = first.removeFirstChild();
170     
171         Node next = null;
172         for (Node cur = first.getNext(); cur != last; cur = next) {
173           commaTree = fuseExpressionIntoExpression(
174               commaTree, cur.removeFirstChild());
175           next = cur.getNext();
176           parent.removeChild(cur);
177         }
178     
179         // Step two: The last EXPR_RESULT will now hold the comma tree with all
180         // the fused statements.
181         first.addChildToBack(commaTree);
182         return first;
183       }
184     
185       private static void fuseExpressionIntoControlFlowStatement(
186           Node before, Node control) {
187         Preconditions.checkArgument(before.isExprResult(),
188             "before must be expression result");
189     
190         // Now we are just left with two statements. The comma tree of the first
191         // n - 1 statements (which can be used in an expression) and the last
192         // statement. We perform specific fusion based on the last statement's type.
193         switch(control.getType()) {
194           case IF:
195           case RETURN:
196           case THROW:
197           case SWITCH:
198           case EXPR_RESULT:
199             before.getParent().removeChild(before);
200             fuseExpressionIntoFirstChild(before.removeFirstChild(), control);
201             return;
202           case FOR:
203             before.getParent().removeChild(before);
204             if (NodeUtil.isForIn(control)) {
205               fuseExpressionIntoSecondChild(before.removeFirstChild(), control);
206             } else {
207               fuseExpressionIntoFirstChild(before.removeFirstChild(), control);
208             }
209             return;
210           case LABEL:
211             fuseExpressionIntoControlFlowStatement(before, control.getLastChild());
212             return;
213           case BLOCK:
214             fuseExpressionIntoControlFlowStatement(before, control.getFirstChild());
215             return;
216           default:
217             throw new IllegalStateException("Statement fusion missing.");
218         }
219       }

221       // exp1, exp1
222       static Node fuseExpressionIntoExpression(Node exp1, Node exp2) {
223         if (exp2.isEmpty()) {
224           return exp1;
225         }
226         Node comma = new Node(Token.COMMA, exp1);
227         comma.useSourceInfoIfMissingFrom(exp2);
228     
229         // We can just join the new comma expression with another comma but
230         // lets keep all the comma's in a straight line. That way we can use
231         // tree comparison.
232         if (exp2.isComma()) {
233           Node leftMostChild = exp2;
234           while (leftMostChild.isComma()) {
235             leftMostChild = leftMostChild.getFirstChild();
236           }
237           Node parent = leftMostChild.getParent();
238           comma.addChildToBack(leftMostChild.detachFromParent());
239           parent.addChildToFront(comma);
240           return exp2;
241         } else {
242           comma.addChildToBack(exp2);
243           return comma;
244         }
245       }
246     
247       protected static void fuseExpressionIntoFirstChild(Node exp, Node stmt) {
248         Node val = stmt.removeFirstChild();
249         Node comma = fuseExpressionIntoExpression(exp, val);
250         stmt.addChildToFront(comma);
251       }
252     
253       protected static void fuseExpressionIntoSecondChild(Node exp, Node stmt) {
254         Node val = stmt.removeChildAfter(stmt.getFirstChild());
255         Node comma = fuseExpressionIntoExpression(exp, val);
256         stmt.addChildAfter(comma, stmt.getFirstChild());
257       }
