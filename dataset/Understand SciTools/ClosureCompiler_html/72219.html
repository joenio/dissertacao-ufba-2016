17     package com.google.javascript.jscomp;
18     
19     import com.google.javascript.rhino.Node;
20     
21     /**
22      * A compiler pass that just reports an error.
23      *
24      */
25     
26     class ErrorPass implements CompilerPass {
27       private final AbstractCompiler compiler;
28       private final JSError error;
29     
30       ErrorPass(AbstractCompiler compiler, DiagnosticType error) {
31         this(compiler, JSError.make(error));
32       }
33     
34       ErrorPass(AbstractCompiler compiler, JSError error) {
35         this.compiler = compiler;
36         this.error = error;
37       }
38     
39       @Override
40       public void process(Node externs, Node root) {
41         compiler.report(error);
42       }
43     }
