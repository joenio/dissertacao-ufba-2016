16     package com.google.javascript.jscomp;
17     
18     import static com.google.common.base.Strings.isNullOrEmpty;
19     
20     import com.google.javascript.jscomp.parsing.parser.util.format.SimpleFormat;
21     
22     /**
23      * Checks for combinations of options that are incompatible, i.e. will produce
24      * incorrect code.
25      *
26      * This is run by Compiler#compileInternal, which is not run during unit tests.
27      * The catch is that it's run after Compiler#initOptions, so if for example
28      * you want to change the warningsGuard, you can't do it here.
29      *
30      * <p>Also, turns off options if the provided options don't make sense together.
31      *
32      * @author tbreisacher@google.com (Tyler Breisacher)
33      */
34     final class CompilerOptionsPreprocessor {
35     
36       static void preprocess(CompilerOptions options) {
37         if (options.checkMissingGetCssNameLevel.isOn()
38             && (isNullOrEmpty(options.checkMissingGetCssNameBlacklist))) {
39           throw new InvalidOptionsException(
40               "Cannot check use of goog.getCssName because of empty blacklist.");
41         }
42     
43         if (options.removeUnusedPrototypePropertiesInExterns
44             && !options.removeUnusedPrototypeProperties) {
45           throw new InvalidOptionsException(
46               "remove_unused_prototype_props_in_externs requires "
47               + "remove_unused_prototype_props to be turned on.");
48         }
49     
50         if (options.getLanguageOut().isEs6OrHigher()
51             && !options.skipNonTranspilationPasses && !options.skipTranspilationAndCrash) {
52           throw new InvalidOptionsException(
53               "ES6 is only supported for transpilation to a lower ECMAScript"
54               + " version. Set --language_out to ES3, ES5, or ES5_STRICT.");
55         }
56     
57         if (!options.inlineFunctions
58             && options.maxFunctionSizeAfterInlining
59             != CompilerOptions.UNLIMITED_FUN_SIZE_AFTER_INLINING) {
60           throw new InvalidOptionsException(
61               "max_function_size_after_inlining has no effect if inlining is"
62               + " disabled.");
63         }
64     
65         if (options.getNewTypeInference()) {
66           options.checkGlobalThisLevel = CheckLevel.OFF;
67         }
68     
69         if (options.jqueryPass && options.closurePass) {
70           throw new InvalidOptionsException(
71               "The jQuery pass and the Closure pass cannot both be enabled.");
72         }
73     
74         if (options.dartPass) {
75           if (!options.getLanguageOut().isEs5OrHigher()) {
76             throw new InvalidOptionsException("Dart requires --language_out=ES5 or higher.");
77           }
78           // --dart_pass does not support type-aware property renaming yet.
79           options.setAmbiguateProperties(false);
80           options.setDisambiguateProperties(false);
81         }
82     
83         if (options.removeUnusedPrototypePropertiesInExterns
84             && options.exportLocalPropertyDefinitions) {
85           throw new InvalidOptionsException(
86               "remove_unused_prototype_props_in_externs "
87               + "and export_local_property_definitions cannot be used together.");
88         }
89     
90       }
91     
92       /**
93        * Exception to indicate incompatible options in the CompilerOptions.
94        */
95       public static class InvalidOptionsException extends RuntimeException {
96         private InvalidOptionsException(String message, Object... args) {
97           super(SimpleFormat.format(message, args));
98         }
99       }
100     
101       // Don't instantiate.
102       private CompilerOptionsPreprocessor() {
103       }
104     }
