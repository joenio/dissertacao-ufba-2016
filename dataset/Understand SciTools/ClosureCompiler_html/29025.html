17     package com.google.javascript.jscomp;
18     
19     import com.google.common.base.Predicates;
20     import com.google.javascript.jscomp.NodeTraversal.AbstractPostOrderCallback;
21     import com.google.javascript.rhino.JSDocInfo;
22     import com.google.javascript.rhino.JSDocInfo.Visibility;
23     import com.google.javascript.rhino.Node;
24     
25     import java.util.LinkedHashMap;
26     import java.util.LinkedHashSet;
27     import java.util.Map;
28     import java.util.Set;
29     
30     /**
31      * Inline aliases created by exports of modules before type checking.
32      *
33      * The old type inference doesn't deal as well with aliased types as with unaliased ones,
34      * such as in extends clauses (@extends {alias}) and templated types (alias<T>).
35      * This pass inlines these aliases to make type checking's job easier.
36      *
37      * @author blickly@gmail.com (Ben Lickly)
38      */
39     final class InlineAliases implements CompilerPass {
40     
41       static final DiagnosticType ALIAS_CYCLE =
42           DiagnosticType.error("JSC_ALIAS_CYCLE", "Alias path contains a cycle: {0} to {1}");
43     
44       private final AbstractCompiler compiler;
45       private final Map<String, String> aliases = new LinkedHashMap<>();
46       private GlobalNamespace namespace;
47     
48       InlineAliases(AbstractCompiler compiler) {
49         this.compiler = compiler;
50       }
51     
52       @Override
53       public void process(Node externs, Node root) {
54         namespace = new GlobalNamespace(compiler, root);
55         NodeTraversal.traverseEs6(compiler, root, new AliasesCollector());
56         NodeTraversal.traverseEs6(compiler, root, new AliasesInliner());
57       }
58     
59       private class AliasesCollector extends AbstractPostOrderCallback {
60         @Override
61         public void visit(NodeTraversal t, Node n, Node parent) {
62           switch (n.getType()) {
63             case VAR:
64               if (n.getChildCount() == 1 && t.inGlobalScope()) {
65                 visitAliasDefinition(n.getFirstChild(), NodeUtil.getBestJSDocInfo(n.getFirstChild()));
66               }
67               break;
68             case ASSIGN:
69               if (parent != null && parent.isExprResult() && t.inGlobalScope()) {
70                 visitAliasDefinition(n.getFirstChild(), n.getJSDocInfo());
71               }
72               break;
73           }
74         }
75     
76         /**
77          * Maybe record that given lvalue is an alias of the qualified name on its rhs.
78          * Note that since we are doing a post-order traversal, any previous aliases contained in
79          * the rhs will have already been substituted by the time we record the new alias.
80          */
81         private void visitAliasDefinition(Node lhs, JSDocInfo info) {
82           if (info != null && info.hasConstAnnotation() && !info.hasTypeInformation()
83               && lhs.isQualifiedName()) {
84             Node rhs = NodeUtil.getRValueOfLValue(lhs);
85             if (rhs != null && rhs.isQualifiedName()) {
86               GlobalNamespace.Name lhsName = namespace.getOwnSlot(lhs.getQualifiedName());
87               GlobalNamespace.Name rhsName = namespace.getOwnSlot(rhs.getQualifiedName());
88               if (lhsName != null
89                   && lhsName.isInlinableGlobalAlias()
90                   && rhsName != null
91                   && rhsName.isInlinableGlobalAlias()
92                   && !isPrivate(rhsName.getDeclaration().getNode())) {
93                 aliases.put(lhs.getQualifiedName(), rhs.getQualifiedName());
94               }
95             }
96           }
97         }
98     
99         private boolean isPrivate(Node nameNode) {
100           if (nameNode.isQualifiedName()
101               && compiler.getCodingConvention().isPrivate(nameNode.getQualifiedName())) {
102             return true;
103           }
104           JSDocInfo info = NodeUtil.getBestJSDocInfo(nameNode);
105           return info != null && info.getVisibility().equals(Visibility.PRIVATE);
106         }
107       }
108     
109       private class AliasesInliner extends AbstractPostOrderCallback {
110         @Override
111         public void visit(NodeTraversal t, Node n, Node parent) {
112           switch (n.getType()) {
113             case NAME:
114             case GETPROP:
115               if (n.isQualifiedName() && aliases.containsKey(n.getQualifiedName())) {
116                 String leftmostName = NodeUtil.getRootOfQualifiedName(n).getString();
117                 Var v = t.getScope().getVar(leftmostName);
118                 if (v != null && v.isLocal()) {
119                   // Shadow of alias. Don't rewrite
120                   return;
121                 }
122                 if (NodeUtil.isVarOrSimpleAssignLhs(n, parent)) {
123                   // Alias definition. Don't rewrite
124                   return;
125                 }
126     
127                 Node newNode = NodeUtil.newQName(compiler, resolveAlias(n.getQualifiedName(), n))
128                     .copyInformationFromForTree(n);
129                 newNode.setLength(n.getLength());
130                 parent.replaceChild(n, newNode);
131                 compiler.reportCodeChange();
132               }
133               break;
134           }
135           maybeRewriteJsdoc(n.getJSDocInfo());
136         }
137     
138         /**
139          * Use the alias table to look up the resolved name of the given alias. If the result is also an
140          * alias repeat until the real name is resolved.
141          * @param n
142          */
143         private String resolveAlias(String name, Node n) {
144           Set<String> aliasPath = new LinkedHashSet<>();
145           while (aliases.containsKey(name)) {
146             if (!aliasPath.add(name)) {
147               compiler.report(JSError.make(n, ALIAS_CYCLE, aliasPath.toString(), name));
148     
149               // Cut the cycle so that it doesn't get reported more than once.
150               aliases.remove(name);
151               break;
152             }
153     
154             name = aliases.get(name);
155           }
156           return name;
157         }
158     
159         private void maybeRewriteJsdoc(JSDocInfo info) {
160           if (info == null) {
161             return;
162           }
163           for (Node typeNode : info.getTypeNodes()) {
164             NodeUtil.visitPreOrder(typeNode, fixJsdocTypeNodes, Predicates.<Node>alwaysTrue());
165           }
166         }
167     
168         private final NodeUtil.Visitor fixJsdocTypeNodes =
169             new NodeUtil.Visitor() {
170               @Override
171               public void visit(Node aliasReference) {
172                 if (!aliasReference.isString()) {
173                   return;
174                 }
175                 String fullTypeName = aliasReference.getString();
176                 int dotIndex = 0;
177                 do {
178                   dotIndex = fullTypeName.indexOf('.', dotIndex + 1);
179                   String aliasName =
180                       dotIndex == -1 ? fullTypeName : fullTypeName.substring(0, dotIndex);
181                   if (aliases.containsKey(aliasName)) {
182                     String replacement =
183                         resolveAlias(aliasName, aliasReference)
184                             + fullTypeName.substring(aliasName.length());
185                     aliasReference.setString(replacement);
186                     return;
187                   }
188                 } while (dotIndex != -1);
189               }
190             };
191       }
192     }
