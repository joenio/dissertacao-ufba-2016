17     package com.google.javascript.jscomp;
18     
19     import com.google.javascript.jscomp.NodeTraversal.AbstractPostOrderCallback;
20     import com.google.javascript.rhino.Node;
21     
22     /**
23      * Traversal callback that finds method invocations of the form
24      *
25      * <pre>
26      * call
27      *   getprop
28      *     ...
29      *     string
30      *   ...
31      * </pre>
32      *
33      * and invokes a method defined by subclasses for processing these invocations.
34      *
35      */
36     abstract class InvocationsCallback extends AbstractPostOrderCallback {
37     
38       @Override
39       public void visit(NodeTraversal t, Node n, Node parent) {
40         if (!n.isCall()) {
41           return;
42         }
43     
44         Node function = n.getFirstChild();
45     
46         if (!function.isGetProp()) {
47           return;
48         }
49     
50         Node nameNode = function.getSecondChild();
51     
52         // Don't care about numerical or variable indexes
53         if (!nameNode.isString()) {
54           return;
55         }
56     
57         visit(t, n, parent, nameNode.getString());
58       }
59     
60       /**
61        * Called for each callnode that is a method invocation.
62        *
63        * @param callNode node of type call
64        * @param parent parent of callNode
65        * @param callName name of method invoked by first child of call
66        */
67       abstract void visit(NodeTraversal t, Node callNode, Node parent,
68           String callName);
69     }
