19     import java.util.AbstractSet;
20     
21     /** A wrapper around a naive persistent set. */
22     abstract class PersistentSet<K> extends AbstractSet<K> {
23       private static final PersistentSet EMPTY = NaivePersistentSet.create();
24     
25       public abstract PersistentSet<K> with(K key);
26     
27       public abstract PersistentSet<K> without(K key);
28     
29       @SuppressWarnings("unchecked")
30       public static <K> PersistentSet<K> create()  {
31         return EMPTY;
32       }
33     }
