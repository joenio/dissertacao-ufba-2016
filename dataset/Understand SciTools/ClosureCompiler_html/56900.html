24       private static final String EXTERNS = LINE_JOINER.join(
25           "var window;",
26           "function alert(a) {}",
27           "var EXT = {};",
28           "EXT.ext;",
29           "var Object;",
30           "Object.defineProperties;",
31           "var foo");

33       public RemoveUnusedClassPropertiesTest() {
34         super(EXTERNS);
35         enableGatherExternProperties();
36       }

38       @Override
39       protected CompilerPass getProcessor(Compiler compiler) {
40         return new RemoveUnusedClassProperties(compiler, true);
41       }

43       public void testSimple1() {
44         // A property defined on "this" can be removed
45         test("this.a = 2", "2");
46         test("x = (this.a = 2)", "x = 2");
47         testSame("this.a = 2; x = this.a;");
48       }
49     
50       public void testSimple2() {
51         // A property defined on "this" can be removed, even when defined
52         // as part of an expression
53         test("this.a = 2, f()", "2, f()");
54         test("x = (this.a = 2, f())", "x = (2, f())");
55         test("x = (f(), this.a = 2)", "x = (f(), 2)");
56       }
57     
58       public void testSimple3() {
59         // A property defined on an object other than "this" can not be removed.
60         testSame("y.a = 2");
61         // and prevents the removal of the definition on 'this'.
62         testSame("y.a = 2; this.a = 2");
63         // Some use of the property "a" prevents the removal.
64         testSame("y.a = 2; this.a = 1; alert(x.a)");
65       }

67       public void testObjLit() {
68         // A property defined on an object other than "this" can not be removed.
69         testSame("({a:2})");
70         // and prevent the removal of the definition on 'this'.
71         testSame("({a:0}); this.a = 1;");
72         // Some use of the property "a" prevents the removal.
73         testSame("x = ({a:0}); this.a = 1; alert(x.a)");
74       }

88       public void testAssignOp1() {
89         // Properties defined using a compound assignment can be removed if the
90         // result of the assignment expression is not immediately used.
91         test("this.x += 2", "2");
92         testSame("x = (this.x += 2)");
93         testSame("this.x += 2; x = this.x;");
94         // But, of course, a later use prevents its removal.
95         testSame("this.x += 2; x.x;");
96       }

98       public void testAssignOp2() {
99         // Properties defined using a compound assignment can be removed if the
100         // result of the assignment expression is not immediately used.
101         test("this.a += 2, f()", "2, f()");
102         test("x = (this.a += 2, f())", "x = (2, f())");
103         testSame("x = (f(), this.a += 2)");
104       }

106       public void testInc1() {
107         // Increments and Decrements are handled similarly to compound assignments
108         // but need a placeholder value when replaced.
109         test("this.x++", "0");
110         testSame("x = (this.x++)");
111         testSame("this.x++; x = this.x;");
112     
113         test("--this.x", "0");
114         testSame("x = (--this.x)");
115         testSame("--this.x; x = this.x;");
116       }

118       public void testInc2() {
119         // Increments and Decrements are handled similarly to compound assignments
120         // but need a placeholder value when replaced.
121         test("this.a++, f()", "0, f()");
122         test("x = (this.a++, f())", "x = (0, f())");
123         testSame("x = (f(), this.a++)");
124     
125         test("--this.a, f()", "0, f()");
126         test("x = (--this.a, f())", "x = (0, f())");
127         testSame("x = (f(), --this.a)");
128       }

136       public void testForIn() {
137         // This is the basic assumption that this pass makes:
138         // it can remove properties even when the object is used in a FOR-IN loop
139         test("this.y = 1;for (var a in x) { alert(x[a]) }",
140              "1;for (var a in x) { alert(x[a]) }");
