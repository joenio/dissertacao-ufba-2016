40     package com.google.javascript.rhino;
41     
42     /**
43      * A common interface for types in the old type system and the new type system,
44      * so that the other passes need not know which type system they are using.
45      *
46      * @author blickly@google.com (Ben Lickly)
47      * @author dimvar@google.com (Dimitris Vardoulakis)
48      */
49     public interface TypeI {
50     
51       boolean isBottom();
52     
53       boolean isConstructor();
54     
55       boolean isEquivalentTo(TypeI type);
56     
57       boolean isFunctionType();
58     
59       boolean isInterface();
60     
61       boolean isSubtypeOf(TypeI type);
62     
63       boolean isUnknownType();
64     
65       boolean isNullable();
66     
67       /**
68        * If this is a union type, returns a union type that does not include
69        * the null or undefined type.
70        */
71       TypeI restrictByNotNullOrUndefined();
72     
73       /**
74        * Downcasts this to a FunctionTypeI, or returns null if this is not
75        * a function.
76        */
77       FunctionTypeI toMaybeFunctionType();
78     
79       /**
80        * If this type is a single object, downcast it to ObjectTypeI.
81        * If it is a non-object or a union of objects, return null.
82        */
83       ObjectTypeI toMaybeObjectType();
84     }
