27       }
28     
29       @Override
30       public int compareTo(URI o) {
31         return uriString.compareTo(o.toString());
32       }
33     
34       @Override
35       public int hashCode() {
36         return uriString.hashCode();
37       }
38     
39       @Override
40       public boolean equals(Object o) {
41         URI other = (URI) o;
42         return uriString.equals(other.toString());
43       }
44     
45       @Override
46       public String toString() {
47         return uriString;
48       }
49

51         return new URI(str);
52       }

56       }

60       }
