32     /**
33      * Checks variables to see if they are referenced before their declaration, or
34      * if they are redeclared in a way that is suspicious (i.e. not dictated by
35      * control structures). This is a more aggressive version of {@link VarCheck},
36      * but it lacks the cross-module checks.
37      *
38      * @author kushal@google.com (Kushal Dave)
39      */
40     class VariableReferenceCheck implements HotSwapCompilerPass {
41     
42       static final DiagnosticType EARLY_REFERENCE =
43           DiagnosticType.warning(
44               "JSC_REFERENCE_BEFORE_DECLARE", "Variable referenced before declaration: {0}");
45     
46       static final DiagnosticType REDECLARED_VARIABLE =
47           DiagnosticType.warning("JSC_REDECLARED_VARIABLE", "Redeclared variable: {0}");
48     
49       static final DiagnosticType AMBIGUOUS_FUNCTION_DECL =
50           DiagnosticType.error("AMBIGUOUS_FUNCTION_DECL", "Ambiguous use of a named function: {0}.");
51     
52       static final DiagnosticType EARLY_REFERENCE_ERROR =
53           DiagnosticType.error(
54               "JSC_REFERENCE_BEFORE_DECLARE_ERROR",
55               "Illegal variable reference before declaration: {0}");
56     
57       static final DiagnosticType REASSIGNED_CONSTANT =
58           DiagnosticType.error("JSC_REASSIGNED_CONSTANT", "Constant reassigned: {0}");
59     
60       static final DiagnosticType REDECLARED_VARIABLE_ERROR =
61           DiagnosticType.error("JSC_REDECLARED_VARIABLE_ERROR", "Illegal redeclared variable: {0}");
62     
63       static final DiagnosticType DECLARATION_NOT_DIRECTLY_IN_BLOCK =
64           DiagnosticType.error(
65               "JSC_DECLARATION_NOT_DIRECTLY_IN_BLOCK",
66               "Block-scoped declaration not directly within block: {0}");
67     
68       static final DiagnosticType UNUSED_LOCAL_ASSIGNMENT =
69           DiagnosticType.disabled(
70               "JSC_UNUSED_LOCAL_ASSIGNMENT", "Value assigned to local variable {0} is never read");
71     
72       private final AbstractCompiler compiler;
73     
74       // NOTE(nicksantos): It's a lot faster to use a shared Set that
75       // we clear after each method call, because the Set never gets too big.
76       private final Set<BasicBlock> blocksWithDeclarations = new HashSet<>();
77     
78       public VariableReferenceCheck(AbstractCompiler compiler) {
79         this.compiler = compiler;
80       }
81     
82       @Override
83       public void process(Node externs, Node root) {
84         ReferenceCollectingCallback callback =
85             new ReferenceCollectingCallback(compiler, new ReferenceCheckingBehavior());
86         callback.process(externs, root);
87       }
88     
89       @Override
90       public void hotSwapScript(Node scriptRoot, Node originalRoot) {
91         ReferenceCollectingCallback callback =
92             new ReferenceCollectingCallback(compiler, new ReferenceCheckingBehavior());
93         callback.hotSwapScript(scriptRoot, originalRoot);
94       }
95     
96       /**
97        * Behavior that checks variables for redeclaration or early references
98        * just after they go out of scope.
99        */
100       private class ReferenceCheckingBehavior implements Behavior {
101     
102         private Set<String> varsInFunctionBody;
103     
104         private ReferenceCheckingBehavior() {
105           varsInFunctionBody = new HashSet<>();
106         }
107     
108         @Override
109         public void afterExitScope(NodeTraversal t, ReferenceMap referenceMap) {
110           // TODO(bashir) In hot-swap version this means that for global scope we
111           // only go through all global variables accessed in the modified file not
112           // all global variables. This should be fixed.
113     
114           // Check all vars after finishing a scope
115           Scope scope = t.getScope();
116           if (scope.isFunctionBlockScope()) {
117             varsInFunctionBody.clear();
118             for (Var v : scope.getVarIterable()) {
119               varsInFunctionBody.add(v.name);
120             }
121           }
122           for (Var v : scope.getVarIterable()) {
123             ReferenceCollection referenceCollection = referenceMap.getReferences(v);
124             // TODO(moz): Figure out why this could be null
125             if (referenceCollection != null) {
126               if (scope.getRootNode().isFunction() && v.isDefaultParam()) {
127                 checkDefaultParam(v, scope, varsInFunctionBody);
128               }
129               if (scope.getRootNode().isFunction()) {
130                 checkShadowParam(v, scope, referenceCollection.references);
131               }
132               checkVar(v, referenceCollection.references);
133             }
134           }
135         }
136     
137         /**
138          * Do a shallow check since cases like:
139          *   function f(y = () => x, x = 5) { return y(); }
140          * is legal. We are going to miss cases like:
141          *   function f(y = (() => x)(), x = 5) { return y(); }
142          * but this should be rare.
143          */
144         private class ShallowReferenceCollector extends AbstractShallowCallback {
145           private final Set<Node> currParamReferences = new LinkedHashSet<>();
146     
147           @Override
148           public void visit(NodeTraversal t, Node n, Node parent) {
149             if (!NodeUtil.isReferenceName(n)) {
150               return;
151             }
152             currParamReferences.add(n);
153           }
154         }
155     
156         private void checkDefaultParam(Var param, Scope scope, Set<String> varsInFunctionBody) {
157           ShallowReferenceCollector check = new ShallowReferenceCollector();
158           NodeTraversal.traverseEs6(compiler, param.getParentNode().getSecondChild(), check);
159           for (Node ref : check.currParamReferences) {
160             String refName = ref.getString();
161             if (varsInFunctionBody.contains(refName) && !scope.isDeclared(refName, true)) {
162               compiler.report(JSError.make(ref, EARLY_REFERENCE_ERROR, refName));
163             }
164           }
165         }
166     
167         private void checkShadowParam(Var v, Scope functionScope, List<Reference> references) {
168           Var maybeParam = functionScope.getVar(v.getName());
169           if (maybeParam != null && maybeParam.isParam() && maybeParam.getScope() == functionScope) {
170             for (Reference r : references) {
171               if ((r.isVarDeclaration() || r.isHoistedFunction())
172                   && !r.getNode().equals(v.getNameNode())) {
173                 compiler.report(JSError.make(r.getNode(), REDECLARED_VARIABLE, v.name));
174               }
175             }
176           }
177         }
178     
179         /**
180          * If the variable is declared more than once in a basic block, generate a
181          * warning. Also check if a variable is used in a given scope before it is
182          * declared, which suggest a likely error. Relies on the fact that
183          * references is in parse-tree order.
184          */
185         private void checkVar(Var v, List<Reference> references) {
186           blocksWithDeclarations.clear();
187           boolean isDeclaredInScope = false;
188           boolean isUnhoistedNamedFunction = false;
189           boolean hasErrors = false;
190           boolean isRead = false;
191           Reference hoistedFn = null;
192           Reference unusedAssignment = null;
193     
194           // Look for hoisted functions.
195           for (Reference reference : references) {
196             if (reference.isHoistedFunction()) {
197               blocksWithDeclarations.add(reference.getBasicBlock());
198               isDeclaredInScope = true;
199               hoistedFn = reference;
200               break;
201             } else if (NodeUtil.isFunctionDeclaration(reference.getNode().getParent())) {
202               isUnhoistedNamedFunction = true;
203             }
204           }
205     
206           for (Reference reference : references) {
207             if (reference == hoistedFn) {
208               continue;
209             }
210             BasicBlock basicBlock = reference.getBasicBlock();
211             boolean isDeclaration = reference.isDeclaration();
212             Node referenceNode = reference.getNode();
213             boolean isAssignment = isDeclaration || reference.isLvalue();
214     
215             boolean allowDupe = VarCheck.hasDuplicateDeclarationSuppression(referenceNode, v);
216             boolean letConstShadowsVar =
217                 v.getParentNode().isVar()
218                     && (reference.isLetDeclaration() || reference.isConstDeclaration());
219             boolean isVarNodeSameAsReferenceNode = v.getNode() == reference.getNode();
220             // We disallow redeclaration of caught exception in ES6
221             boolean shadowCatchVar =
222                 isDeclaration
223                     && compiler.getLanguageMode().isEs6OrHigher()
224                     && v.getParentNode().isCatch()
225                     && !isVarNodeSameAsReferenceNode;
226             boolean shadowParam =
227                 isDeclaration
228                     && v.isParam()
229                     && NodeUtil.isBlockScopedDeclaration(referenceNode)
230                     && v.getScope() == reference.getScope().getParent();
231             boolean shadowDetected = false;
232             if (isDeclaration && !allowDupe) {
233               // Look through all the declarations we've found so far, and
234               // check if any of them are before this block.
235               for (BasicBlock declaredBlock : blocksWithDeclarations) {
236                 if (declaredBlock.provablyExecutesBefore(basicBlock)) {
237                   shadowDetected = true;
238                   DiagnosticType diagnosticType;
239                   Node warningNode = referenceNode;
240                   if (v.isLet()
241                       || v.isConst()
242                       || v.isClass()
243                       || letConstShadowsVar
244                       || shadowCatchVar
245                       || shadowParam) {
246                     // These cases are all hard errors that violate ES6 semantics
247                     diagnosticType = REDECLARED_VARIABLE_ERROR;
248                   } else if (reference.getNode().getParent().isCatch() || allowDupe) {
249                     return;
250                   } else {
251                     // These diagnostics are for valid, but suspicious, code, and are suppressible.
252                     // For vars defined in the global scope, give the same error as VarCheck
253                     diagnosticType =
254                         v.getScope().isGlobal()
255                             ? VarCheck.VAR_MULTIPLY_DECLARED_ERROR
256                             : REDECLARED_VARIABLE;
257                     // Since we skip hoisted functions, we would have the wrong warning node in cases
258                     // where the redeclaration is a function declaration. Check for that case.
259                     if (isVarNodeSameAsReferenceNode
260                         && hoistedFn != null
261                         && v.name.equals(hoistedFn.getNode().getString())) {
262                       warningNode = hoistedFn.getNode();
263                     }
264                   }
265                   compiler.report(
266                       JSError.make(
267                           warningNode,
268                           diagnosticType,
269                           v.name,
270                           v.input != null ? v.input.getName() : "??"));
271                   hasErrors = true;
272                   break;
273                 }
274               }
275             }
276     
277             if (!shadowDetected
278                 && isDeclaration
279                 && (letConstShadowsVar || shadowCatchVar)
280                 && v.getScope() == reference.getScope()) {
281               compiler.report(JSError.make(referenceNode, REDECLARED_VARIABLE_ERROR, v.name));
282             }
283     
284             if (isAssignment) {
285               Reference decl = references.get(0);
286               Node declNode = decl.getNode();
287               Node gp = declNode.getGrandparent();
288               boolean lhsOfForInLoop = NodeUtil.isForIn(gp) && gp.getFirstFirstChild() == declNode;
289     
290               if (decl.getScope().isLocal()
291                   && (decl.isVarDeclaration() || decl.isLetDeclaration() || decl.isConstDeclaration())
292                   && !decl.getNode().isFromExterns()
293                   && !lhsOfForInLoop) {
294                 unusedAssignment = reference;
295               }
296               if ((reference.getParent().isDec() || reference.getParent().isInc())
297                   && NodeUtil.isExpressionResultUsed(reference.getNode())) {
298                 isRead = true;
299               }
300             } else {
301               isRead = true;
302             }
303     
304             if (isUnhoistedNamedFunction && !isDeclaration && isDeclaredInScope) {
305               // Only allow an unhoisted named function to be used within the
306               // block it is declared.
307               for (BasicBlock declaredBlock : blocksWithDeclarations) {
308                 if (!declaredBlock.provablyExecutesBefore(basicBlock)) {
309                   compiler.report(JSError.make(referenceNode, AMBIGUOUS_FUNCTION_DECL, v.name));
310                   hasErrors = true;
311                   break;
312                 }
313               }
314             }
315     
316             boolean isUndeclaredReference = false;
317             if (!isDeclaration && !isDeclaredInScope) {
318               // Don't check the order of refer in externs files.
319               if (!referenceNode.isFromExterns()) {
320                 // Special case to deal with var goog = goog || {}. Note that
321                 // let x = x || {} is illegal, just like var y = x || {}; let x = y;
322                 Node grandparent = reference.getGrandparent();
323                 if ((v.isVar() && grandparent.isName() && grandparent.getString().equals(v.name))) {
324                   continue;
325                 }
326     
327                 // Only generate warnings if the scopes do not match in order
328                 // to deal with possible forward declarations and recursion
329                 // TODO(moz): Remove the bypass for "goog" once VariableReferenceCheck
330                 // is run after the Closure passes.
331                 if (reference.getScope() == v.scope && !v.getName().equals("goog")) {
332                   isUndeclaredReference = true;
333                   compiler.report(
334                       JSError.make(
335                           reference.getNode(),
336                           (v.isLet() || v.isConst() || v.isClass() || v.isParam())
337                               ? EARLY_REFERENCE_ERROR
338                               : EARLY_REFERENCE,
339                           v.name));
340                   hasErrors = true;
341                 }
342               }
343             }
344     
345             if (!isDeclaration && !isUndeclaredReference && v.isConst() && reference.isLvalue()) {
346               compiler.report(JSError.make(referenceNode, REASSIGNED_CONSTANT, v.name));
347             }
348     
349             if (isDeclaration
350                 && !reference.isVarDeclaration()
351                 && reference.getGrandparent().isAddedBlock()) {
352               compiler.report(JSError.make(referenceNode, DECLARATION_NOT_DIRECTLY_IN_BLOCK, v.name));
353             }
354     
355             if (isDeclaration) {
356               blocksWithDeclarations.add(basicBlock);
357               isDeclaredInScope = true;
358             }
359           }
360     
361           // Only check for unused local if there are no other errors, and if not in a goog.scope
362           // function.
363           // TODO(tbreisacher): Consider moving UNUSED_LOCAL_ASSIGNMENT into its own check pass, so
364           // that we can run it after goog.scope processing, and get rid of the inGoogScope check.
365           if (unusedAssignment != null && !isRead && !hasErrors) {
366             boolean inGoogScope = false;
367             Scope s = v.getScope();
368             Node function = null;
369             if (s.isFunctionBlockScope()) {
370               function = s.getRootNode().getParent();
371             } else if (s.isFunctionScope()) {
372               // TODO(tbreisacher): Remove this branch when everything is switched to
373               // Es6SyntacticScopeCreator.
374               function = s.getRootNode();
375             }
376             if (function != null) {
377               Node callee = function.getParent().getChildBefore(function);
378               inGoogScope = callee != null && callee.matchesQualifiedName("goog.scope");
379             }
380     
381             if (!inGoogScope) {
382               compiler.report(
383                   JSError.make(unusedAssignment.getNode(), UNUSED_LOCAL_ASSIGNMENT, v.name));
384             }
385           }
386         }
387       }
