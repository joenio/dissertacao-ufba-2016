121       /**
122        * @param fnName The name of this function. This either the name of the
123        *  variable to which the function is assigned or the name from the FUNCTION
124        *  node.
125        * @param fnNode The FUNCTION node of the function to inspect.
126        * @return Whether the function node meets the minimum requirements for
127        * inlining.
128        */
129       boolean doesFunctionMeetMinimumRequirements(
130           final String fnName, Node fnNode) {
131         Node block = NodeUtil.getFunctionBody(fnNode);
132     
133         // Basic restrictions on functions that can be inlined:
134         // 0) The function is inlinable by convention
135         // 1) It contains a reference to itself.
136         // 2) It uses its parameters indirectly using "arguments" (it isn't
137         //    handled yet.
138         // 3) It references "eval". Inline a function containing eval can have
139         //    large performance implications.
140     
141         if (!compiler.getCodingConvention().isInlinableFunction(fnNode)) {
142           return false;
143         }
144     
145         final String fnRecursionName = fnNode.getFirstChild().getString();
146         Preconditions.checkState(fnRecursionName != null);
147     
148         // If the function references "arguments" directly in the function
149         boolean referencesArguments = NodeUtil.isNameReferenced(
150             block, "arguments", NodeUtil.MATCH_NOT_FUNCTION);
151     
152         // or it references "eval" or one of its names anywhere.
153         Predicate<Node> p = new Predicate<Node>(){
154           @Override
155           public boolean apply(Node n) {
156             if (n.isName()) {
157               return n.getString().equals("eval")
158                 || (!fnName.isEmpty()
159                     && n.getString().equals(fnName))
160                 || (!fnRecursionName.isEmpty()
161                     && n.getString().equals(fnRecursionName));
162             }
163             return false;
164           }
165         };
166     
167         return !referencesArguments
168             && !NodeUtil.has(block, p, Predicates.<Node>alwaysTrue());
169       }

171       /**
172        * @param fnNode The function to evaluate for inlining.
173        * @param needAliases A set of function parameter names that can not be
174        *     used without aliasing. Returned by getUnsafeParameterNames().
175        * @param referencesThis Whether fnNode contains references to its this
176        *     object.
177        * @param containsFunctions Whether fnNode contains inner functions.
178        * @return Whether the inlining can occur.
179        */
180       CanInlineResult canInlineReferenceToFunction(
181           Reference ref, Node fnNode, Set<String> needAliases,
182           boolean referencesThis, boolean containsFunctions) {
183         // TODO(johnlenz): This function takes too many parameter, without
184         // context.  Modify the API to take a structure describing the function.
185     
186         // Allow direct function calls or "fn.call" style calls.
187         Node callNode = ref.callNode;
188         if (!isSupportedCallType(callNode)) {
189           return CanInlineResult.NO;
190         }
191     
192         // Limit where functions that contain functions can be inline.  Introducing
193         // an inner function into another function can capture a variable and cause
194         // a memory leak.  This isn't a problem in the global scope as those values
195         // last until explicitly cleared.
196         if (containsFunctions) {
197           if (!assumeMinimumCapture && !ref.scope.isGlobal()) {
198             // TODO(johnlenz): Allow inlining into any scope without local names or
199             // inner functions.
200             return CanInlineResult.NO;
201           } else if (NodeUtil.isWithinLoop(callNode)) {
202             // An inner closure maybe relying on a local value holding a value for a
203             // single iteration through a loop.
204             return CanInlineResult.NO;
205           }
206         }
207     
208         // TODO(johnlenz): Add support for 'apply'
209         if (referencesThis && !NodeUtil.isFunctionObjectCall(callNode)) {
210           // TODO(johnlenz): Allow 'this' references to be replaced with a
211           // global 'this' object.
212           return CanInlineResult.NO;
213         }
214     
215         if (ref.mode == InliningMode.DIRECT) {
216           return canInlineReferenceDirectly(ref, fnNode, needAliases);
217         } else {
218           return canInlineReferenceAsStatementBlock(
219               ref, fnNode, needAliases);
220         }
221       }

223       /**
224        * Only ".call" calls and direct calls to functions are supported.
225        * @param callNode The call evaluate.
226        * @return Whether the call is of a type that is supported.
227        */
228       private boolean isSupportedCallType(Node callNode) {
229         if (!callNode.getFirstChild().isName()) {
230           if (NodeUtil.isFunctionObjectCall(callNode)) {
231             if (!assumeStrictThis) {
232               Node thisValue = callNode.getSecondChild();
233               if (thisValue == null || !thisValue.isThis()) {
234                 return false;
235               }
236             }
237           } else if (NodeUtil.isFunctionObjectApply(callNode)) {
238             return false;
239           }
240         }
241     
242         return true;
243       }

246        * Inline a function into the call site.
247        */
248       Node inline(Reference ref, String fnName, Node fnNode) {
249         Preconditions.checkState(compiler.getLifeCycleStage().isNormalized());
250     
251         if (ref.mode == InliningMode.DIRECT) {
252           return inlineReturnValue(ref, fnNode);
253         } else {
254           return inlineFunction(ref, fnNode, fnName);
255         }

258       /**
259        * Inline a function that fulfills the requirements of
260        * canInlineReferenceDirectly into the call site, replacing only the CALL
261        * node.
262        */
263       private Node inlineReturnValue(Reference ref, Node fnNode) {
264         Node callNode = ref.callNode;
265         Node block = fnNode.getLastChild();
266         Node callParentNode = callNode.getParent();
267     
268         // NOTE: As the normalize pass guarantees globals aren't being
269         // shadowed and an expression can't introduce new names, there is
270         // no need to check for conflicts.
271     
272         // Create an argName -> expression map, checking for side effects.
273         Map<String, Node> argMap =
274             FunctionArgumentInjector.getFunctionCallParameterMap(
275                 fnNode, callNode, this.safeNameIdSupplier);
276     
277         Node newExpression;
278         if (!block.hasChildren()) {
279           Node srcLocation = block;
280           newExpression = NodeUtil.newUndefinedNode(srcLocation);
281         } else {
282           Node returnNode = block.getFirstChild();
283           Preconditions.checkArgument(returnNode.isReturn());
284     
285           // Clone the return node first.
286           Node safeReturnNode = returnNode.cloneTree();
287           Node inlineResult = FunctionArgumentInjector.inject(
288               null, safeReturnNode, null, argMap);
289           Preconditions.checkArgument(safeReturnNode == inlineResult);
290           newExpression = safeReturnNode.removeFirstChild();
291         }
292     
293         callParentNode.replaceChild(callNode, newExpression);
294         return newExpression;
295       }

297       /**
298        * Supported call site types.
299        */
300       private static enum CallSiteType {
301     
302         /**
303          * Used for a call site for which there does not exist a method
304          * to inline it.
305          */
306         UNSUPPORTED() {
307           @Override
308           public void prepare(FunctionInjector injector, Reference ref) {
309             throw new IllegalStateException("unexpected");
310           }
311         },
312     
313         /**
314          * A call as a statement. For example: "foo();".
315          *   EXPR_RESULT
316          *     CALL
317          */
318         SIMPLE_CALL() {
319           @Override
320           public void prepare(FunctionInjector injector, Reference ref) {
321             // Nothing to do.
322           }
323         },
324     
325         /**
326          * An assignment, where the result of the call is assigned to a simple
327          * name. For example: "a = foo();".
328          *   EXPR_RESULT
329          *     NAME A
330          *     CALL
331          *       FOO
332          */
333         SIMPLE_ASSIGNMENT() {
334           @Override
335           public void prepare(FunctionInjector injector, Reference ref) {
336             // Nothing to do.
337           }
338         },
339         /**
340          * An var declaration and initialization, where the result of the call is
341          * assigned to the declared name
342          * name. For example: "a = foo();".
343          *   VAR
344          *     NAME A
345          *       CALL
346          *         FOO
347          */
348         VAR_DECL_SIMPLE_ASSIGNMENT() {
349           @Override
350           public void prepare(FunctionInjector injector, Reference ref) {
351             // Nothing to do.
352           }
353         },
354         /**
355          * An arbitrary expression, the root of which is a EXPR_RESULT, IF,
356          * RETURN, SWITCH or VAR.  The call must be the first side-effect in
357          * the expression.
358          *
359          * Examples include:
360          *   "if (foo()) {..."
361          *   "return foo();"
362          *   "var a = 1 + foo();"
363          *   "a = 1 + foo()"
364          *   "foo() ? 1:0"
365          *   "foo() && x"
366          */
367         EXPRESSION() {
368           @Override
369           public void prepare(FunctionInjector injector, Reference ref) {
370             Node callNode = ref.callNode;
371             injector.getDecomposer(ref.scope).moveExpression(callNode);
372     
373             // Reclassify after move
374             CallSiteType callSiteType = injector.classifyCallSite(ref);
375             Preconditions.checkState(this != callSiteType);
376             callSiteType.prepare(injector, ref);
377           }
378         },
379     
380         /**
381          * An arbitrary expression, the root of which is a EXPR_RESULT, IF,
382          * RETURN, SWITCH or VAR.  Where the call is not the first side-effect in
383          * the expression.
384          */
385         DECOMPOSABLE_EXPRESSION() {
386           @Override
387           public void prepare(FunctionInjector injector, Reference ref) {
388             Node callNode = ref.callNode;
389             injector.getDecomposer(ref.scope).maybeExposeExpression(callNode);
390     
391             // Reclassify after decomposition
392             CallSiteType callSiteType = injector.classifyCallSite(ref);
393             Preconditions.checkState(this != callSiteType);
394             callSiteType.prepare(injector, ref);
395           }
396         };
397     
398         public abstract void prepare(FunctionInjector injector, Reference ref);
399       }

401       /**
402        * Determine which, if any, of the supported types the call site is.
403        *
404        * Constant vars are treated differently so that we don't break their
405        * const-ness when we decompose the expression. Once the CONSTANT_VAR
406        * annotation is used everywhere instead of coding conventions, we should just
407        * teach this pass how to remove the annotation.
408        */
409       private CallSiteType classifyCallSite(Reference ref) {
410         Node callNode = ref.callNode;
411         Node parent = callNode.getParent();
412         Node grandParent = parent.getParent();
413     
414         // Verify the call site:
415         if (NodeUtil.isExprCall(parent)) {
416           // This is a simple call?  Example: "foo();".
417           return CallSiteType.SIMPLE_CALL;
418         } else if (NodeUtil.isExprAssign(grandParent)
419             && !NodeUtil.isVarOrSimpleAssignLhs(callNode, parent)
420             && parent.getFirstChild().isName()
421             // TODO(nicksantos): Remove this once everyone is using
422             // the CONSTANT_VAR annotation. We know how to remove that.
423             && !NodeUtil.isConstantName(parent.getFirstChild())) {
424           // This is a simple assignment.  Example: "x = foo();"
425           return CallSiteType.SIMPLE_ASSIGNMENT;
426         } else if (parent.isName()
427             // TODO(nicksantos): Remove this once everyone is using
428             // the CONSTANT_VAR annotation.
429             && !NodeUtil.isConstantName(parent)
430             && grandParent.isVar()
431             && grandParent.hasOneChild()) {
432           // This is a var declaration.  Example: "var x = foo();"
433           // TODO(johnlenz): Should we be checking for constants on the
434           // left-hand-side of the assignments and handling them as EXPRESSION?
435           return CallSiteType.VAR_DECL_SIMPLE_ASSIGNMENT;
436         } else {
437           Node expressionRoot = ExpressionDecomposer.findExpressionRoot(callNode);
438           if (expressionRoot != null) {
439             ExpressionDecomposer decomposer = new ExpressionDecomposer(
440                 compiler, safeNameIdSupplier, knownConstants, ref.scope);
441             DecompositionType type = decomposer.canExposeExpression(
442                 callNode);
443             if (type == DecompositionType.MOVABLE) {
444               return CallSiteType.EXPRESSION;
445             } else if (type == DecompositionType.DECOMPOSABLE) {
446               return CallSiteType.DECOMPOSABLE_EXPRESSION;
447             } else {
448               Preconditions.checkState(type == DecompositionType.UNDECOMPOSABLE);
449             }
450           }
451         }
452     
453         return CallSiteType.UNSUPPORTED;
454       }

470       /**
471        * Inline a function which fulfills the requirements of
472        * canInlineReferenceAsStatementBlock into the call site, replacing the
473        * parent expression.
474        */
475       private Node inlineFunction(
476           Reference ref, Node fnNode, String fnName) {
477         Node callNode = ref.callNode;
478         Node parent = callNode.getParent();
479         Node grandParent = parent.getParent();
480     
481         // TODO(johnlenz): Consider storing the callSite classification in the
482         // reference object and passing it in here.
483         CallSiteType callSiteType = classifyCallSite(ref);
484         Preconditions.checkArgument(callSiteType != CallSiteType.UNSUPPORTED);
485     
486         // Store the name for the result. This will be used to
487         // replace "return expr" statements with "resultName = expr"
488         // to replace
489         String resultName = null;
490         boolean needsDefaultReturnResult = true;
491         switch (callSiteType) {
492           case SIMPLE_ASSIGNMENT:
493             resultName = parent.getFirstChild().getString();
494             removeConstantVarAnnotation(ref.scope, resultName);
495             break;
496     
497           case VAR_DECL_SIMPLE_ASSIGNMENT:
498             resultName = parent.getString();
499             removeConstantVarAnnotation(ref.scope, resultName);
500             break;
501     
502           case SIMPLE_CALL:
503             resultName = null;  // "foo()" doesn't need a result.
504             needsDefaultReturnResult = false;
505             break;
506     
507           case EXPRESSION:
508             throw new IllegalStateException(
509                 "Movable expressions must be moved before inlining.");
510     
511           case DECOMPOSABLE_EXPRESSION:
512             throw new IllegalStateException(
513                 "Decomposable expressions must be decomposed before inlining.");
514     
515           default:
516             throw new IllegalStateException("Unexpected call site type.");
517         }
518     
519         FunctionToBlockMutator mutator = new FunctionToBlockMutator(
520             compiler, this.safeNameIdSupplier);
521     
522         boolean isCallInLoop = NodeUtil.isWithinLoop(callNode);
523         Node newBlock = mutator.mutate(
524             fnName, fnNode, callNode, resultName,
525             needsDefaultReturnResult, isCallInLoop);
526     
527         // TODO(nicksantos): Create a common mutation function that
528         // can replace either a VAR name assignment, assignment expression or
529         // a EXPR_RESULT.
530         Node greatGrandParent = grandParent.getParent();
531         switch (callSiteType) {
532           case VAR_DECL_SIMPLE_ASSIGNMENT:
533             // Remove the call from the name node.
534             parent.removeChild(parent.getFirstChild());
535             Preconditions.checkState(parent.getFirstChild() == null);
536             // Add the call, after the VAR.
537             greatGrandParent.addChildAfter(newBlock, grandParent);
538             break;
539     
540           case SIMPLE_ASSIGNMENT:
541             // The assignment is now part of the inline function so
542             // replace it completely.
543             Preconditions.checkState(grandParent.isExprResult());
544             greatGrandParent.replaceChild(grandParent, newBlock);
545             break;
546     
547           case SIMPLE_CALL:
548             // If nothing is looking at the result just replace the call.
549             Preconditions.checkState(parent.isExprResult());
550             grandParent.replaceChild(parent, newBlock);
551             break;
552     
553           default:
554             throw new IllegalStateException("Unexpected call site type.");
555         }
556     
557         return newBlock;
558       }

560       private static void removeConstantVarAnnotation(Scope scope, String name) {
561         Var var = scope.getVar(name);
562         Node nameNode = var == null ? null : var.getNameNode();
563         if (nameNode == null) return;
564     
565         if (nameNode.getBooleanProp(Node.IS_CONSTANT_VAR)) {
566           nameNode.removeProp(Node.IS_CONSTANT_VAR);
567         }
568       }

570       /**
571        * Checks if the given function matches the criteria for an inlinable
572        * function, and if so, adds it to our set of inlinable functions.
573        */
574       static boolean isDirectCallNodeReplacementPossible(Node fnNode) {
575         // Only inline single-statement functions
576         Node block = NodeUtil.getFunctionBody(fnNode);
577     
578         // Check if this function is suitable for direct replacement of a CALL node:
579         // a function that consists of single return that returns an expression.
580         if (!block.hasChildren()) {
581           // special case empty functions.
582           return true;
583         } else if (block.hasOneChild()) {
584           // Only inline functions that return something.
585           if (block.getFirstChild().isReturn()
586               && block.getFirstFirstChild() != null) {
587             return true;
588           }
589         }
590     
591         return false;
592       }

594       enum CanInlineResult {
595         YES,
596         AFTER_PREPARATION,
597         NO

600       /**
601        * Determines whether a function can be inlined at a particular call site.
602        * There are several criteria that the function and reference must hold in
603        * order for the functions to be inlined:
604        * - It must be a simple call, or assignment, or var initialization.
605        * <pre>
606        *    f();
607        *    a = foo();
608        *    var a = foo();
609        * </pre>
610        */
611       private CanInlineResult canInlineReferenceAsStatementBlock(
612           Reference ref, Node fnNode, Set<String> namesToAlias) {
613         CallSiteType callSiteType = classifyCallSite(ref);
614         if (callSiteType == CallSiteType.UNSUPPORTED) {
615           return CanInlineResult.NO;
616         }
617     
618         if (!allowDecomposition
619             && (callSiteType == CallSiteType.DECOMPOSABLE_EXPRESSION
620                 || callSiteType == CallSiteType.EXPRESSION)) {
621           return CanInlineResult.NO;
622         }
623     
624         if (!callMeetsBlockInliningRequirements(
625                 ref, fnNode, namesToAlias)) {
626           return CanInlineResult.NO;
627         }
628     
629         if (callSiteType == CallSiteType.DECOMPOSABLE_EXPRESSION
630             || callSiteType == CallSiteType.EXPRESSION) {
631           return CanInlineResult.AFTER_PREPARATION;
632         } else {
633           return CanInlineResult.YES;
634         }
635       }

637       /**
638        * Determines whether a function can be inlined at a particular call site.
639        * - Don't inline if the calling function contains an inner function and
640        * inlining would introduce new globals.
641        */
642       private boolean callMeetsBlockInliningRequirements(
643           Reference ref, final Node fnNode,
644           Set<String> namesToAlias) {
645         final boolean assumeMinimumCapture = this.assumeMinimumCapture;
646         // Note: functions that contain function definitions are filtered out
647         // in isCandidateFunction.
648     
649         // TODO(johnlenz): Determining if the called function contains VARs
650         // or if the caller contains inner functions accounts for 20% of the
651         // run-time cost of this pass.
652     
653         // Don't inline functions with var declarations into a scope with inner
654         // functions as the new vars would leak into the inner function and
655         // cause memory leaks.
656         boolean fnContainsVars = NodeUtil.has(
657             NodeUtil.getFunctionBody(fnNode),
658             new NodeUtil.MatchDeclaration(),
659             new NodeUtil.MatchShallowStatement());
660         boolean forbidTemps = false;
661         if (!ref.scope.getClosestHoistScope().isGlobal()) {
662           Node fnCallerBody = ref.scope.getClosestHoistScope().getRootNode();
663     
664           // Don't allow any new vars into a scope that contains eval or one
665           // that contains functions (excluding the function being inlined).
666           Predicate<Node> match = new Predicate<Node>(){
667             @Override
668             public boolean apply(Node n) {
669               if (n.isName()) {
670                 return n.getString().equals("eval");
671               }
672               if (!assumeMinimumCapture && n.isFunction()) {
673                 return n != fnNode;
674               }
675               return false;
676             }
677           };
678           forbidTemps = NodeUtil.has(fnCallerBody, match, NodeUtil.MATCH_NOT_FUNCTION);
679         }
680     
681         if (fnContainsVars && forbidTemps) {
682           return false;
683         }
684     
685         // If the caller contains functions or evals, verify we aren't adding any
686         // additional VAR declarations because aliasing is needed.
687         if (forbidTemps) {
688           Map<String, Node> args =
689               FunctionArgumentInjector.getFunctionCallParameterMap(
690                   fnNode, ref.callNode, this.safeNameIdSupplier);
691           boolean hasArgs = !args.isEmpty();
692           if (hasArgs) {
693             // Limit the inlining
694             Set<String> allNamesToAlias = new HashSet<>(namesToAlias);
695             FunctionArgumentInjector.maybeAddTempsForCallArguments(
696                 fnNode, args, allNamesToAlias, compiler.getCodingConvention());
697             if (!allNamesToAlias.isEmpty()) {
698               return false;
699             }
700           }
701         }
702     
703         return true;
704       }

706       /**
707        * Determines whether a function can be inlined at a particular call site.
708        * There are several criteria that the function and reference must hold in
709        * order for the functions to be inlined:
710        * 1) If a call's arguments have side effects,
711        * the corresponding argument in the function must only be referenced once.
712        * For instance, this will not be inlined:
713        * <pre>
714        *     function foo(a) { return a + a }
715        *     x = foo(i++);
716        * </pre>
717        */
718       private CanInlineResult canInlineReferenceDirectly(
719           Reference ref, Node fnNode, Set<String> namesToAlias) {
720         if (!isDirectCallNodeReplacementPossible(fnNode)) {
721           return CanInlineResult.NO;
722         }
723     
724         // CALL NODE: [ NAME, ARG1, ARG2, ... ]
725         Node callNode = ref.callNode;
726         Node cArg = callNode.getSecondChild();
727     
728         // Functions called via 'call' and 'apply' have a this-object as
729         // the first parameter, but this is not part of the called function's
730         // parameter list.
731         if (!callNode.getFirstChild().isName()) {
732           if (NodeUtil.isFunctionObjectCall(callNode)) {
733             // TODO(johnlenz): Support replace this with a value.
734             if (cArg == null || !cArg.isThis()) {
735               return CanInlineResult.NO;
736             }
737             cArg = cArg.getNext();
738           } else {
739             // ".apply" call should be filtered before this.
740             Preconditions.checkState(!NodeUtil.isFunctionObjectApply(callNode));
741           }
742         }
743     
744         Map<String, Node> args =
745             FunctionArgumentInjector.getFunctionCallParameterMap(
746                 fnNode, callNode, this.throwawayNameSupplier);
747         boolean hasArgs = !args.isEmpty();
748         if (hasArgs) {
749           // Limit the inlining
750           Set<String> allNamesToAlias = new HashSet<>(namesToAlias);
751           FunctionArgumentInjector.maybeAddTempsForCallArguments(
752               fnNode, args, allNamesToAlias, compiler.getCodingConvention());
753           if (!allNamesToAlias.isEmpty()) {
754             return CanInlineResult.NO;
755           }
756         }
757     
758         return CanInlineResult.YES;
759       }

761       /**
762        * Determine if inlining the function is likely to reduce the code size.
763        * @param namesToAlias
764        */
765       boolean inliningLowersCost(
766           JSModule fnModule, Node fnNode, Collection<? extends Reference> refs,
767           Set<String> namesToAlias, boolean isRemovable, boolean referencesThis) {
768         int referenceCount = refs.size();
769         if (referenceCount == 0) {
770           return true;
771         }
772     
773         int referencesUsingBlockInlining = 0;
774     
775         boolean checkModules = isRemovable && fnModule != null;
776         JSModuleGraph moduleGraph = compiler.getModuleGraph();
777     
778         for (Reference ref : refs) {
779           if (ref.mode == InliningMode.BLOCK) {
780             referencesUsingBlockInlining++;
781           }
782     
783           // Check if any of the references cross the module boundaries.
784           if (checkModules && ref.module != null) {
785             if (ref.module != fnModule &&
786                 !moduleGraph.dependsOn(ref.module, fnModule)) {
787               // Calculate the cost as if the function were non-removable,
788               // if it still lowers the cost inline it.
789               isRemovable = false;
790               checkModules = false;  // no need to check additional modules.
791             }
792           }
793         }
794     
795         int referencesUsingDirectInlining = referenceCount -
796             referencesUsingBlockInlining;
797     
798         // Don't bother calculating the cost of function for simple functions where
799         // possible.
800         // However, when inlining a complex function, even a single reference may be
801         // larger than the original function if there are many returns (resulting
802         // in additional assignments) or many parameters that need to be aliased
803         // so use the cost estimating.
804         if (referenceCount == 1 && isRemovable &&
805             referencesUsingDirectInlining == 1) {
806           return true;
807         }
808     
809         int callCost = estimateCallCost(fnNode, referencesThis);
810         int overallCallCost = callCost * referenceCount;
811     
812         int costDeltaDirect = inlineCostDelta(
813             fnNode, namesToAlias, InliningMode.DIRECT);
814         int costDeltaBlock = inlineCostDelta(
815             fnNode, namesToAlias, InliningMode.BLOCK);
816     
817         return doesLowerCost(fnNode, overallCallCost,
818             referencesUsingDirectInlining, costDeltaDirect,
819             referencesUsingBlockInlining, costDeltaBlock,
820             isRemovable);
821       }
822     
823       /**
824        * @return Whether inlining will lower cost.
825        */
826       private static boolean doesLowerCost(
827           Node fnNode, int callCost,
828           int directInlines, int costDeltaDirect,
829           int blockInlines, int costDeltaBlock,
830           boolean removable) {
831     
832         // Determine the threshold value for this inequality:
833         //     inline_cost < call_cost
834         // But solve it for the function declaration size so the size of it
835         // is only calculated once and terminated early if possible.
836     
837         int fnInstanceCount = directInlines + blockInlines - (removable ? 1 : 0);
838         // Prevent division by zero.
839         if (fnInstanceCount == 0) {
840           // Special case single reference function that are being block inlined:
841           // If the cost of the inline is greater than the function definition size,
842           // don't inline.
843           return blockInlines <= 0 || costDeltaBlock <= 0;
844         }
845     
846         int costDelta = (directInlines * -costDeltaDirect) + (blockInlines * -costDeltaBlock);
847         int threshold = (callCost + costDelta) / fnInstanceCount;
848     
849         return InlineCostEstimator.getCost(fnNode, threshold + 1) <= threshold;
850       }

852       /**
853        * Gets an estimate of the cost in characters of making the function call:
854        * the sum of the identifiers and the separators.
855        * @param referencesThis
856        */
857       private static int estimateCallCost(Node fnNode, boolean referencesThis) {
858         Node argsNode = NodeUtil.getFunctionParameters(fnNode);
859         int numArgs = argsNode.getChildCount();
860     
861         int callCost = NAME_COST_ESTIMATE + PAREN_COST;
862         if (numArgs > 0) {
863           callCost += (numArgs * NAME_COST_ESTIMATE) + ((numArgs - 1) * COMMA_COST);
864         }
865     
866         if (referencesThis) {
867           // TODO(johnlenz): Update this if we start supporting inlining
868           // other functions that reference this.
869           // The only functions that reference this that are currently inlined
870           // are those that are called via ".call" with an explicit "this".
871           callCost += 5 + 5;  // ".call" + "this,"
872         }
873     
874         return callCost;
875       }

877       /**
878        * @return The difference between the function definition cost and
879        *     inline cost.
880        */
881       private static int inlineCostDelta(
882           Node fnNode, Set<String> namesToAlias, InliningMode mode) {
883         // The part of the function that is never inlined:
884         //    "function xx(xx,xx){}" (15 + (param count * 3) -1;
885         int paramCount = NodeUtil.getFunctionParameters(fnNode).getChildCount();
886         int commaCount = (paramCount > 1) ? paramCount - 1 : 0;
887         int costDeltaFunctionOverhead = 15 + commaCount +
888             (paramCount * InlineCostEstimator.ESTIMATED_IDENTIFIER_COST);
889     
890         Node block = fnNode.getLastChild();
891         if (!block.hasChildren()) {
892           // Assume the inline cost is zero for empty functions.
893           return -costDeltaFunctionOverhead;
894         }
895     
896         if (mode == InliningMode.DIRECT) {
897           // The part of the function that is inlined using direct inlining:
898           //    "return " (7)
899           return -(costDeltaFunctionOverhead + 7);
900         } else {
901           int aliasCount = namesToAlias.size();
902     
903           // Originally, we estimated purely base on the function code size, relying
904           // on later optimizations. But that did not produce good results, so here
905           // we try to estimate the something closer to the actual inlined coded.
906     
907           // NOTE 1: Result overhead is only if there is an assignment, but
908           // getting that information would require some refactoring.
909           // NOTE 2: The aliasing overhead is currently an under-estimate,
910           // as some parameters are aliased because of the parameters used.
911           // Perhaps we should just assume all parameters will be aliased?
912           final int inlineBlockOverhead = 4; // "X:{}"
913           final int perReturnOverhead = 2;   // "return" --> "break X"
914           final int perReturnResultOverhead = 3; // "XX="
915           final int perAliasOverhead = 3; // "XX="
916     
917           // TODO(johnlenz): Counting the number of returns is relatively expensive
918           //   this information should be determined during the traversal and
919           //   cached.
920           int returnCount = NodeUtil.getNodeTypeReferenceCount(
921               block, Token.RETURN, new NodeUtil.MatchShallowStatement());
922           int resultCount = (returnCount > 0) ? returnCount - 1 : 0;
923           int baseOverhead = (returnCount > 0) ? inlineBlockOverhead : 0;
924     
925           int overhead = baseOverhead
926               + returnCount * perReturnOverhead
927               + resultCount * perReturnResultOverhead
928               + aliasCount * perAliasOverhead;
929     
930           return (overhead - costDeltaFunctionOverhead);
931         }
932       }
