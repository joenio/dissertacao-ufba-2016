18     import java.io.IOException;

20     /**
21      * @author johnlenz@google.com (John Lenz)
22      */
23     public class Util {
24     
25       private static final char[] HEX_CHARS
26           = { '0', '1', '2', '3', '4', '5', '6', '7',
27               '8', '9', 'a', 'b', 'c', 'd', 'e', 'f' };
28     
29       /**
30        * Escapes the given string to a double quoted (") JavaScript/JSON string
31        */
32       static String escapeString(String s) {
33         return escapeString(s, '"',  "\\\"", "\'", "\\\\");
34       }
35     
36       /** Helper to escape JavaScript string as well as regular expression */
37       private static String escapeString(String s, char quote,
38                               String doublequoteEscape,
39                               String singlequoteEscape,
40                               String backslashEscape) {
41         StringBuilder sb = new StringBuilder(s.length() + 2);
42         sb.append(quote);
43         for (int i = 0; i < s.length(); i++) {
44           char c = s.charAt(i);
45           switch (c) {
46             case '\n': sb.append("\\n"); break;
47             case '\r': sb.append("\\r"); break;
48             case '\t': sb.append("\\t"); break;
49             case '\\': sb.append(backslashEscape); break;
50             case '\"': sb.append(doublequoteEscape); break;
51             case '\'': sb.append(singlequoteEscape); break;
52             case '>':                       // Break --> into --\> or ]]> into ]]\>
53               if (i >= 2 &&
54                   ((s.charAt(i - 1) == '-' && s.charAt(i - 2) == '-') ||
55                    (s.charAt(i - 1) == ']' && s.charAt(i - 2) == ']'))) {
56                 sb.append("\\>");
57               } else {
58                 sb.append(c);
59               }
60               break;
61             case '<':
62               // Break </script into <\/script
63               final String END_SCRIPT = "/script";
64     
65               // Break <!-- into <\!--
66               final String START_COMMENT = "!--";
67     
68               if (s.regionMatches(true, i + 1, END_SCRIPT, 0,
69                                   END_SCRIPT.length())) {
70                 sb.append("<\\");
71               } else if (s.regionMatches(false, i + 1, START_COMMENT, 0,
72                                          START_COMMENT.length())) {
73                 sb.append("<\\");
74               } else {
75                 sb.append(c);
76               }
77               break;
78             default:
79               // No charsetEncoder provided - pass straight Latin characters
80               // through, and escape the rest.  Doing the explicit character
81               // check is measurably faster than using the CharsetEncoder.
82               if (c > 0x1f && c <= 0x7f) {
83                 sb.append(c);
84               } else {
85                 // Other characters can be misinterpreted by some JS parsers,
86                 // or perhaps mangled by proxies along the way,
87                 // so we play it safe and Unicode escape them.
88                 appendHexJavaScriptRepresentation(sb, c);
89               }
90           }
91         }
92         sb.append(quote);
93         return sb.toString();
94       }
95     
96       /**
97        * @see #appendHexJavaScriptRepresentation(int, Appendable)
98        */
99       public static void appendHexJavaScriptRepresentation(
100           StringBuilder sb, char c) {
101         try {
102           appendHexJavaScriptRepresentation(c, sb);
103         } catch (IOException ex) {
104           // StringBuilder does not throw IOException.
105           throw new RuntimeException(ex);
106         }
107       }
108     
109       /**
110        * Returns a JavaScript representation of the character in a hex escaped
111        * format.
112        * @param codePoint The code point to append.
113        * @param out The buffer to which the hex representation should be appended.
114        */
115       private static void appendHexJavaScriptRepresentation(
116           int codePoint, Appendable out)
117           throws IOException {
118         if (Character.isSupplementaryCodePoint(codePoint)) {
119           // Handle supplementary Unicode values which are not representable in
120           // JavaScript.  We deal with these by escaping them as two 4B sequences
121           // so that they will round-trip properly when sent from Java to JavaScript
122           // and back.
123           char[] surrogates = Character.toChars(codePoint);
124           appendHexJavaScriptRepresentation(surrogates[0], out);
125           appendHexJavaScriptRepresentation(surrogates[1], out);
126           return;
127         }
128         out.append("\\u")
129             .append(HEX_CHARS[(codePoint >>> 12) & 0xf])
130             .append(HEX_CHARS[(codePoint >>> 8) & 0xf])
131             .append(HEX_CHARS[(codePoint >>> 4) & 0xf])
132             .append(HEX_CHARS[codePoint & 0xf]);
133       }
