88       void checkValid() {
89         if (isTopFunction() || isQmarkFunction()) {
90           return;
91         }
92         Preconditions.checkNotNull(requiredFormals,
93             "null required formals for function: %s", this);
94         for (JSType formal : requiredFormals) {
95           Preconditions.checkNotNull(formal);
96           // A loose function has bottom formals in the bwd direction of NTI.
97           // See NTI#analyzeLooseCallNodeBwd.
98           Preconditions.checkState(isLoose || !formal.isBottom());
99         }
100         Preconditions.checkNotNull(optionalFormals,
101             "null optional formals for function: %s", this);
102         for (JSType formal : optionalFormals) {
103           Preconditions.checkNotNull(formal);
104           Preconditions.checkState(!formal.isBottom());
105         }
106         Preconditions.checkState(restFormals == null || !restFormals.isBottom());
107         Preconditions.checkNotNull(returnType);
108       }

126       static FunctionType normalized(
127           List<JSType> requiredFormals,
128           List<JSType> optionalFormals,
129           JSType restFormals,
130           JSType retType,
131           JSType nominalType,
132           JSType receiverType,
133           Map<String, JSType> outerVars,
134           ImmutableList<String> typeParameters,
135           boolean isLoose) {
136         if (requiredFormals == null) {
137           requiredFormals = ImmutableList.of();
138         }
139         if (optionalFormals == null) {
140           optionalFormals = ImmutableList.of();
141         }
142         if (outerVars == null) {
143           outerVars = ImmutableMap.of();
144         }
145         if (typeParameters == null) {
146           typeParameters = ImmutableList.of();
147         }
148         if (restFormals != null) {
149           // Remove trailing optional params w/ type equal to restFormals
150           for (int i = optionalFormals.size() - 1; i >= 0; i--) {
151             if (restFormals.equals(optionalFormals.get(i))) {
152               optionalFormals.remove(i);
153             } else {
154               break;
155             }
156           }
157         }
158         return new FunctionType(
159             ImmutableList.copyOf(requiredFormals),
160             ImmutableList.copyOf(optionalFormals),
161             restFormals, retType, nominalType, receiverType,
162             ImmutableMap.copyOf(outerVars),
163             typeParameters,
164             isLoose);
165       }

167       // We want to warn about argument mismatch, so we don't consider a function
168       // with N required arguments to have restFormals of type TOP.
169       // But we allow joins (eg after an IF) to change arity, eg,
170       // number->number \/ number,number->number = number,number->number
171     
172       // Theoretically, the top function takes an infinite number of required
173       // arguments of type BOTTOM and returns TOP. If this function is ever called,
174       // it's a type error. Despite that, we want to represent it and not go
175       // directly to JSType.TOP, to avoid spurious warnings.
176       // Eg, after an IF, we may see a type (number | top_function); this type could
177       // get specialized to number and used legitimately.
178     
179       // We can't represent the theoretical top function, so we special-case
180       // TOP_FUNCTION below. However, the outcome is the same; if our top function
181       // is ever called, a warning is inevitable.
182       static final FunctionType TOP_FUNCTION = new FunctionType(false);
183       private static final FunctionType LOOSE_TOP_FUNCTION = new FunctionType(true);

188       private static final FunctionType BOTTOM_FUNCTION = normalized(
189           null, null, null, JSType.BOTTOM, null, null, null, null, false);

240       // Returns null if argpos indexes past the arguments
241       public JSType getFormalType(int argpos) {
242         Preconditions.checkArgument(!isTopFunction());
243         int numReqFormals = requiredFormals.size();
244         if (argpos < numReqFormals) {
245           Preconditions.checkState(null != requiredFormals.get(argpos));
246           return requiredFormals.get(argpos);
247         } else if (argpos < numReqFormals + optionalFormals.size()) {
248           Preconditions.checkState(
249               null != optionalFormals.get(argpos - numReqFormals));
250           return optionalFormals.get(argpos - numReqFormals);
251         } else {
252           return restFormals;
253         }
254       }

305       public FunctionType transformByCallProperty() {
306         if (isTopFunction() || isQmarkFunction() || isLoose) {
307           return QMARK_FUNCTION;
308         }
309         FunctionTypeBuilder builder = new FunctionTypeBuilder();
310         builder.addReqFormal(fromReceiverToFirstFormal());
311         for (JSType type : this.requiredFormals) {
312           builder.addReqFormal(type);
313         }
314         for (JSType type : this.optionalFormals) {
315           builder.addOptFormal(type);
316         }
317         builder.addRestFormals(this.restFormals);
318         builder.addRetType(this.returnType);
319         builder.addTypeParameters(this.typeParameters);
320         return builder.buildFunction();
321       }

323       // We only typecheck the receiver type for a .apply function. To typecheck all
324       // arguments we either need tuple types or special handling in NTI to gather
325       // the types inside the array.
326       public FunctionType transformByApplyProperty(JSTypes commonTypes) {
327         if (isTopFunction() || isQmarkFunction() || isLoose) {
328           return QMARK_FUNCTION;
329         }
330         if (isGeneric()) {
331           return instantiateGenericsWithUnknown(this).transformByApplyProperty(commonTypes);
332         }
333         FunctionTypeBuilder builder = new FunctionTypeBuilder();
334         builder.addReqFormal(fromReceiverToFirstFormal());
335         builder.addOptFormal(JSType.join(
336             commonTypes.getArrayInstance(), commonTypes.getArgumentsArrayType()));
337         builder.addRetType(this.returnType);
338         return builder.buildFunction();
339       }

341       private JSType fromReceiverToFirstFormal() {
342         if (this.receiverType == null) {
343           return JSType.UNKNOWN;
344         }
345         NominalType nt = this.receiverType.getNominalTypeIfSingletonObj();
346         if (nt == null || nt.isBuiltinObject()) {
347           return this.receiverType;
348         }
349         if (nt.isGeneric()) {
350           return nt.instantiateGenerics(JSType.MAP_TO_UNKNOWN).getInstanceAsJSType();
351         }
352         return nt.getInstanceAsJSType();
353       }

356       public DeclaredFunctionType toDeclaredFunctionType() {
357         if (isQmarkFunction()) {
358           return FunctionTypeBuilder.qmarkFunctionBuilder().buildDeclaration();
359         }
360         Preconditions.checkState(!isLoose(), "Loose function: %s", this);
361         Preconditions.checkState(!isGeneric(), "Generic function: %s", this);
362         FunctionTypeBuilder builder = new FunctionTypeBuilder();
363         for (JSType type : this.requiredFormals) {
364           builder.addReqFormal(type);
365         }
366         for (JSType type : this.optionalFormals) {
367           builder.addOptFormal(type);
368         }
369         builder.addRestFormals(this.restFormals);
370         builder.addRetType(this.returnType);
371         builder.addNominalType(this.nominalType);
372         builder.addReceiverType(this.receiverType);
373         return builder.buildDeclaration();
374       }

376       private static JSType nullAcceptingMeet(JSType t1, JSType t2) {
377         if (t1 == null) {
378           return t2;
379         }
380         if (t2 == null) {
381           return t1;
382         }
383         JSType tmp = JSType.meet(t1, t2);
384         return tmp.isBottom() ? null : tmp;
385       }

387       // TODO(dimvar): we need to clean up the combination of loose functions with
388       // new: and/or this: types. Eg, this.nominalType doesn't appear at all.
389       private static FunctionType looseJoin(FunctionType f1, FunctionType f2) {
390         Preconditions.checkArgument(f1.isLoose() || f2.isLoose());
391     
392         FunctionTypeBuilder builder = new FunctionTypeBuilder();
393         int minRequiredArity = Math.min(f1.getMinArity(), f2.getMinArity());
394         for (int i = 0; i < minRequiredArity; i++) {
395           builder.addReqFormal(JSType.nullAcceptingJoin(
396               f1.getFormalType(i), f2.getFormalType(i)));
397         }
398         int maxTotalArity = Math.max(
399             f1.requiredFormals.size() + f1.optionalFormals.size(),
400             f2.requiredFormals.size() + f2.optionalFormals.size());
401         for (int i = minRequiredArity; i < maxTotalArity; i++) {
402           JSType t = JSType.nullAcceptingJoin(f1.getFormalType(i), f2.getFormalType(i));
403           if (t != null && t.isBottom()) {
404             // We will add the optional formal of the loose function in the fwd
405             // direction, when we have better type information.
406             break;
407           }
408           builder.addOptFormal(t);
409         }
410         // Loose types never have varargs, because there is no way for that
411         // information to make it to a function summary
412         return builder.addRetType(
413             JSType.nullAcceptingJoin(f1.returnType, f2.returnType))
414             .addLoose().buildFunction();
415       }

440       private boolean isSubtypeOfHelper(FunctionType other, boolean checkThisType,
441           SubtypeCache subSuperMap, MismatchInfo[] boxedInfo) {
442         if (other.isTopFunction() ||
443             other.isQmarkFunction() || this.isQmarkFunction()) {
444           return true;
445         }
446         if (isTopFunction()) {
447           return false;
448         }
449         // NOTE(dimvar): We never happen to call isSubtypeOf for loose functions.
450         // If some analyzed program changes this, the preconditions check will tell
451         // us so we can handle looseness correctly.
452         Preconditions.checkState(!isLoose() && !other.isLoose());
453         if (this.isGeneric()) {
454           if (this.equals(other)) {
455             return true;
456           }
457           // NOTE(dimvar): This is a bug. The code that triggers this should be rare
458           // and the fix is not trivial, so for now we decided to not fix.
459           // See unit tests in NewTypeInferenceES5OrLowerTest#testGenericsSubtyping
460           return instantiateGenericsWithUnknown(this)
461               .isSubtypeOfHelper(other, checkThisType, subSuperMap, boxedInfo);
462         }
463     
464         if (!other.acceptsAnyArguments()) {
465           // The subtype must have an equal or smaller number of required formals
466           if (requiredFormals.size() > other.requiredFormals.size()) {
467             return false;
468           }
469           int otherMaxTotalArity =
470               other.requiredFormals.size() + other.optionalFormals.size();
471           for (int i = 0; i < otherMaxTotalArity; i++) {
472             // contravariance in the arguments
473             JSType thisFormal = getFormalType(i);
474             JSType otherFormal = other.getFormalType(i);
475             if (thisFormal != null
476                 && !thisFormal.isUnknown() && !otherFormal.isUnknown()
477                 && !otherFormal.isSubtypeOf(thisFormal, subSuperMap)) {
478               if (boxedInfo != null) {
479                 boxedInfo[0] =
480                     MismatchInfo.makeArgTypeMismatch(i, otherFormal, thisFormal);
481               }
482               return false;
483             }
484           }
485     
486           if (other.restFormals != null) {
487             int thisMaxTotalArity =
488                 this.requiredFormals.size() + this.optionalFormals.size();
489             if (this.restFormals != null) {
490               thisMaxTotalArity++;
491             }
492             for (int i = otherMaxTotalArity; i < thisMaxTotalArity; i++) {
493               JSType thisFormal = getFormalType(i);
494               JSType otherFormal = other.getFormalType(i);
495               if (thisFormal != null
496                   && !thisFormal.isUnknown() && !otherFormal.isUnknown()
497                   && !otherFormal.isSubtypeOf(thisFormal, subSuperMap)) {
498                 return false;
499               }
500             }
501           }
502         }
503     
504         // covariance for the new: type
505         if (this.nominalType == null && other.nominalType != null
506             || this.nominalType != null && other.nominalType == null
507             || this.nominalType != null && other.nominalType != null
508                && !this.nominalType.isSubtypeOf(other.nominalType)) {
509           return false;
510         }
511     
512         if (checkThisType) {
513           // A function without @this can be a subtype of a function with @this.
514           if (this.receiverType != null && other.receiverType == null
515               || this.receiverType != null && other.receiverType != null
516               // Contravariance for the receiver type
517               && !other.receiverType.isSubtypeOf(this.receiverType, subSuperMap)
518               // NOTE(dimvar): Covariance for the receiver type.
519               // Not correct, but allowed to make migration easier.
520               // After bounded generics, we could probably drop support for this.
521               && !this.receiverType.isSubtypeOf(other.receiverType, subSuperMap)) {
522             return false;
523           }
524         }
525     
526         // covariance in the return type
527         boolean areRetTypesSubtypes = this.returnType.isUnknown()
528             || other.returnType.isUnknown()
529             || this.returnType.isSubtypeOf(other.returnType, subSuperMap);
530         if (boxedInfo != null) {
531           boxedInfo[0] =
532               MismatchInfo.makeRetTypeMismatch(other.returnType, this.returnType);
533         }
534         return areRetTypesSubtypes;
535       }

537       // Avoid using JSType#join if possible, to avoid creating new types
538       private static JSType joinNominalTypes(JSType nt1, JSType nt2) {
539         if (nt1 == null || nt2 == null) {
540           return null;
541         }
542         NominalType n1 = getNominalTypeIfSingletonObj(nt1);
543         NominalType n2 = getNominalTypeIfSingletonObj(nt2);
544         if (n1 != null && n2 != null) {
545           NominalType tmp = NominalType.pickSuperclass(n1, n2);
546           return tmp == null ? null : tmp.getInstanceAsJSType();
547         }
548         // One of the nominal types is non-standard; can't avoid the join
549         return JSType.join(nt1, nt2);
550       }

552       // Avoid using JSType#meet if possible, to avoid creating new types
553       private static JSType meetNominalTypes(JSType nt1, JSType nt2) {
554         if (nt1 == null) {
555           return nt2;
556         }
557         if (nt2 == null) {
558           return nt1;
559         }
560         NominalType n1 = getNominalTypeIfSingletonObj(nt1);
561         NominalType n2 = getNominalTypeIfSingletonObj(nt2);
562         if (n1 != null && n2 != null) {
563           NominalType tmp = NominalType.pickSubclass(n1, n2);
564           return tmp == null ? null : tmp.getInstanceAsJSType();
565         }
566         // One of the nominal types is non-standard; can't avoid the meet
567         return JSType.meet(nt1, nt2);
568       }

570       static FunctionType join(FunctionType f1, FunctionType f2) {
571         if (f1 == null) {
572           return f2;
573         } else if (f2 == null || f1.equals(f2)) {
574           return f1;
575         } else if (f1.isQmarkFunction()) {
576           return f2 == QMARK_FUNCTION ? QMARK_FUNCTION : f1;
577         } else if (f2.isQmarkFunction()) {
578           return f2;
579         } else if (f1.isTopFunction() || f2.isTopFunction()) {
580           return TOP_FUNCTION;
581         }
582     
583         if (f1.isLoose() || f2.isLoose()) {
584           return looseJoin(f1, f2);
585         }
586     
587         if (f1.isGeneric() && f2.isSubtypeOf(f1, SubtypeCache.create())) {
588           return f1;
589         } else if (f2.isGeneric() && f1.isSubtypeOf(f2, SubtypeCache.create())) {
590           return f2;
591         }
592     
593         // We lose precision for generic funs that are not in a subtype relation.
594         if (f1.isGeneric()) {
595           f1 = instantiateGenericsWithUnknown(f1);
596         }
597         if (f2.isGeneric()) {
598           f2 = instantiateGenericsWithUnknown(f2);
599         }
600     
601         FunctionTypeBuilder builder = new FunctionTypeBuilder();
602         int maxRequiredArity = Math.max(
603             f1.requiredFormals.size(), f2.requiredFormals.size());
604         for (int i = 0; i < maxRequiredArity; i++) {
605           JSType reqFormal = nullAcceptingMeet(f1.getFormalType(i), f2.getFormalType(i));
606           if (reqFormal == null) {
607             return BOTTOM_FUNCTION;
608           }
609           builder.addReqFormal(reqFormal);
610         }
611         int maxTotalArity = Math.max(
612             f1.requiredFormals.size() + f1.optionalFormals.size(),
613             f2.requiredFormals.size() + f2.optionalFormals.size());
614         for (int i = maxRequiredArity; i < maxTotalArity; i++) {
615           JSType optFormal = nullAcceptingMeet(f1.getFormalType(i), f2.getFormalType(i));
616           if (optFormal == null) {
617             return BOTTOM_FUNCTION;
618           }
619           builder.addOptFormal(optFormal);
620         }
621         if (f1.restFormals != null && f2.restFormals != null) {
622           JSType newRestFormals = nullAcceptingMeet(f1.restFormals, f2.restFormals);
623           if (newRestFormals == null) {
624             return BOTTOM_FUNCTION;
625           }
626           builder.addRestFormals(newRestFormals);
627         }
628         builder.addRetType(JSType.join(f1.returnType, f2.returnType));
629         builder.addNominalType(joinNominalTypes(f1.nominalType, f2.nominalType));
630         builder.addReceiverType(meetNominalTypes(f1.receiverType, f2.receiverType));
631         return builder.buildFunction();
632       }

634       FunctionType specialize(FunctionType other) {
635         if (other == null
636             || other.isQmarkFunction() || other.isTopFunction() || equals(other)
637             || !isLoose()) {
638           return this;
639         }
640         return isTopFunction() || isQmarkFunction()
641             ? other.withLoose() : looseJoin(this, other);
642       }

644       static FunctionType meet(FunctionType f1, FunctionType f2) {
645         if (f1 == null || f2 == null) {
646           return null;
647         } else if (f2.isTopFunction() || f1.equals(f2)) {
648           return f1;
649         } else if (f1.isTopFunction()) {
650           return f2;
651         }
652     
653         // War is peace, freedom is slavery, meet is join
654         if (f1.isLoose() || f2.isLoose()) {
655           return looseJoin(f1, f2);
656         }
657     
658         if (f1.isGeneric() && f1.isSubtypeOf(f2, SubtypeCache.create())) {
659           return f1;
660         } else if (f2.isGeneric() && f2.isSubtypeOf(f1, SubtypeCache.create())) {
661           return f2;
662         }
663     
664         // We lose precision for generic funs that are not in a subtype relation.
665         if (f1.isGeneric()) {
666           f1 = instantiateGenericsWithUnknown(f1);
667         }
668         if (f2.isGeneric()) {
669           f2 = instantiateGenericsWithUnknown(f2);
670         }
671     
672         FunctionTypeBuilder builder = new FunctionTypeBuilder();
673         int minRequiredArity = Math.min(
674             f1.requiredFormals.size(), f2.requiredFormals.size());
675         for (int i = 0; i < minRequiredArity; i++) {
676           builder.addReqFormal(
677               JSType.nullAcceptingJoin(f1.getFormalType(i), f2.getFormalType(i)));
678         }
679         int maxTotalArity = Math.max(
680             f1.requiredFormals.size() + f1.optionalFormals.size(),
681             f2.requiredFormals.size() + f2.optionalFormals.size());
682         for (int i = minRequiredArity; i < maxTotalArity; i++) {
683           JSType optFormalType =
684               JSType.nullAcceptingJoin(f1.getFormalType(i), f2.getFormalType(i));
685           if (optFormalType.isBottom()) {
686             return BOTTOM_FUNCTION;
687           }
688           builder.addOptFormal(optFormalType);
689         }
690         if (f1.restFormals != null || f2.restFormals != null) {
691           JSType restFormalsType =
692               JSType.nullAcceptingJoin(f1.restFormals, f2.restFormals);
693           if (restFormalsType.isBottom()) {
694             return BOTTOM_FUNCTION;
695           }
696           builder.addRestFormals(restFormalsType);
697         }
698         JSType retType = JSType.meet(f1.returnType, f2.returnType);
699         if (retType.isBottom()) {
700           return BOTTOM_FUNCTION;
701         }
702         builder.addRetType(retType);
703         // NOTE(dimvar): these two are not correct. We should be picking the
704         // greatest lower bound of the types if they are incomparable.
705         // Eg, this case arises when an interface extends multiple interfaces.
706         // OTOH, it may be enough to detect that during GTI, and not implement the
707         // more expensive methods (in NominalType or ObjectType).
708         builder.addNominalType(meetNominalTypes(f1.nominalType, f2.nominalType));
709         builder.addReceiverType(joinNominalTypes(f1.receiverType, f2.receiverType));
710         return builder.buildFunction();
711       }

713       // We may consider true subtyping for deferred checks when the formal
714       // parameter has a loose function type.
715       boolean isLooseSubtypeOf(FunctionType f2, SubtypeCache subSuperMap) {
716         Preconditions.checkState(this.isLoose() || f2.isLoose());
717         if (this.isTopFunction() || f2.isTopFunction()) {
718           return true;
719         }
720         int minRequiredArity =
721             Math.min(this.requiredFormals.size(), f2.requiredFormals.size());
722         for (int i = 0; i < minRequiredArity; i++) {
723           if (!JSType.haveCommonSubtype(this.getFormalType(i), f2.getFormalType(i))) {
724             return false;
725           }
726         }
727         return JSType.haveCommonSubtype(this.getReturnType(), f2.getReturnType());
728       }

738       boolean unifyWithSubtype(FunctionType other, List<String> typeParameters,
739           Multimap<String, JSType> typeMultimap, SubtypeCache subSuperMap) {
740         Preconditions.checkState(this.typeParameters.isEmpty());
741         Preconditions.checkState(this.outerVarPreconditions.isEmpty());
742         Preconditions.checkState(this != TOP_FUNCTION);
743     
744         if (this == LOOSE_TOP_FUNCTION || other.isTopFunction() || other.isLoose()) {
745           return true;
746         }
747         if (!acceptsAnyArguments()) {
748           if (other.requiredFormals.size() > this.requiredFormals.size()) {
749             return false;
750           }
751           int maxNonInfiniteArity = getMaxArityWithoutRestFormals();
752           for (int i = 0; i < maxNonInfiniteArity; i++) {
753             JSType thisFormal = getFormalType(i);
754             JSType otherFormal = other.getFormalType(i);
755             // NOTE(dimvar): The correct handling here would be to implement
756             // unifyWithSupertype for JSType, ObjectType, etc, to handle the
757             // contravariance here.
758             // But it's probably an overkill to do, so instead we just do a subtype
759             // check if unification fails. Same for restFormals and receiverType.
760             // Altenatively, maybe the unifyWith function could handle both subtype
761             // and supertype, and we'd catch type errors as invalid-argument-type
762             // after unification. (Not sure this is correct, I'd have to try it.)
763             if (otherFormal != null
764                 && !thisFormal.unifyWithSubtype(
765                     otherFormal, typeParameters, typeMultimap, subSuperMap)
766                 && !thisFormal.isSubtypeOf(otherFormal, SubtypeCache.create())) {
767               return false;
768             }
769           }
770           if (this.restFormals != null) {
771             JSType otherRestFormals = other.getFormalType(maxNonInfiniteArity);
772             if (otherRestFormals != null
773                 && !this.restFormals.unifyWithSubtype(
774                     otherRestFormals, typeParameters, typeMultimap, subSuperMap)
775                 && !this.restFormals.isSubtypeOf(
776                     otherRestFormals, SubtypeCache.create())) {
777               return false;
778             }
779           }
780         }
781     
782         if (nominalType == null && other.nominalType != null
783             || nominalType != null && other.nominalType == null) {
784           return false;
785         }
786         if (nominalType != null && !nominalType.unifyWithSubtype(
787             other.nominalType, typeParameters, typeMultimap, subSuperMap)) {
788           return false;
789         }
790     
791         // If one of the two functions doesn't use THIS in the body, we can still
792         // unify.
793         if (this.receiverType != null && other.receiverType != null
794             && !this.receiverType.unifyWithSubtype(
795                 other.receiverType, typeParameters, typeMultimap, subSuperMap)
796             && !this.receiverType.isSubtypeOf(
797                 other.receiverType, SubtypeCache.create())) {
798           return false;
799         }
800     
801         return this.returnType.unifyWithSubtype(
802             other.returnType, typeParameters, typeMultimap, subSuperMap);
803       }

805       private static FunctionType instantiateGenericsWithUnknown(FunctionType f) {
806         if (!f.isGeneric()) {
807           return f;
808         }
809         return f.instantiateGenerics(JSType.MAP_TO_UNKNOWN);
810       }

812       /**
813        * Unify the two types symmetrically, given that we have already instantiated
814        * the type variables of interest in {@code f1} and {@code f2}, treating
815        * JSType.UNKNOWN as a "hole" to be filled.
816        * @return The unified type, or null if unification fails
817        */
818       static FunctionType unifyUnknowns(FunctionType f1, FunctionType f2) {
819         Preconditions.checkState(f1 != null || f2 != null);
820         if (f1 == null || f2 == null) {
821           return null;
822         }
823         if (!f1.typeParameters.isEmpty()) {
824           f1 = instantiateGenericsWithUnknown(f1);
825         }
826         if (!f2.typeParameters.isEmpty()) {
827           f2 = instantiateGenericsWithUnknown(f2);
828         }
829         Preconditions.checkState(!f1.isLoose() && !f2.isLoose());
830         if (f1.equals(f2)) {
831           return f1;
832         }
833     
834         ImmutableList<JSType> formals1 = f1.requiredFormals;
835         ImmutableList<JSType> formals2 = f2.requiredFormals;
836         if (formals1.size() != formals2.size()) {
837           return null;
838         }
839         FunctionTypeBuilder builder = new FunctionTypeBuilder();
840         int numReqFormals = formals1.size();
841         for (int i = 0; i < numReqFormals; i++) {
842           JSType t = JSType.unifyUnknowns(formals1.get(i), formals2.get(i));
843           if (t == null) {
844             return null;
845           }
846           builder.addReqFormal(t);
847         }
848     
849         formals1 = f1.optionalFormals;
850         formals2 = f2.optionalFormals;
851         if (formals1.size() != formals2.size()) {
852           return null;
853         }
854         int numOptFormals = formals1.size();
855         for (int i = 0; i < numOptFormals; i++) {
856           JSType t = JSType.unifyUnknowns(formals1.get(i), formals2.get(i));
857           if (t == null) {
858             return null;
859           }
860           builder.addOptFormal(t);
861         }
862     
863         if (f1.restFormals == null && f2.restFormals != null
864             || f1.restFormals != null && f2.restFormals == null) {
865           return null;
866         }
867         if (f1.restFormals != null) {
868           JSType t = JSType.unifyUnknowns(f1.restFormals, f2.restFormals);
869           if (t == null) {
870             return null;
871           }
872           builder.addRestFormals(t);
873         }
874     
875         JSType t = JSType.unifyUnknowns(f1.returnType, f2.returnType);
876         if (t == null) {
877           return null;
878         }
879         builder.addRetType(t);
880     
881         // Don't unify unknowns in nominal types; it's going to be rare.
882         if (!Objects.equals(f1.nominalType, f2.nominalType)) {
883           return null;
884         }
885         builder.addNominalType(f1.nominalType);
886     
887         if (!Objects.equals(f1.receiverType, f2.receiverType)) {
888           return null;
889         }
890         builder.addReceiverType(f1.receiverType);
891     
892         return builder.buildFunction();
893       }

895       // Avoid JSType#substituteGenerics if possible, to avoid creating new types.
896       private static JSType substGenericsInNomType(JSType nt, Map<String, JSType> typeMap) {
897         if (nt == null) {
898           return null;
899         }
900         NominalType tmp = nt.getNominalTypeIfSingletonObj();
901         if (tmp == null) {
902           return nt.substituteGenerics(typeMap);
903         }
904         if (!tmp.isGeneric()) {
905           return tmp.getInstanceAsJSType();
906         }
907         if (typeMap.isEmpty()) {
908           return nt;
909         }
910         return JSType.fromObjectType(ObjectType.fromNominalType(
911             tmp.instantiateGenerics(typeMap)));
912       }

914       private FunctionType substituteNominalGenerics(Map<String, JSType> typeMap) {
915         if (typeMap.isEmpty()) {
916           return this;
917         }
918         Map<String, JSType> reducedMap = typeMap;
919         if (!JSType.MAP_TO_UNKNOWN.equals(typeMap)) {
920           boolean foundShadowedTypeParam = false;
921           for (String typeParam : this.typeParameters) {
922             if (typeMap.containsKey(typeParam)) {
923               foundShadowedTypeParam = true;
924               break;
925             }
926           }
927           if (foundShadowedTypeParam) {
928             ImmutableMap.Builder<String, JSType> builder = ImmutableMap.builder();
929             for (Map.Entry<String, JSType> entry : typeMap.entrySet()) {
930               if (!typeParameters.contains(entry.getKey())) {
931                 builder.put(entry);
932               }
933             }
934             reducedMap = builder.build();
935           }
936         }
937         FunctionTypeBuilder builder = new FunctionTypeBuilder();
938         for (JSType reqFormal : this.requiredFormals) {
939           builder.addReqFormal(reqFormal.substituteGenerics(reducedMap));
940         }
941         for (JSType optFormal : this.optionalFormals) {
942           builder.addOptFormal(optFormal.substituteGenerics(reducedMap));
943         }
944         if (this.restFormals != null) {
945           builder.addRestFormals(restFormals.substituteGenerics(reducedMap));
946         }
947         builder.addRetType(this.returnType.substituteGenerics(reducedMap));
948         if (isLoose()) {
949           builder.addLoose();
950         }
951         builder.addNominalType(substGenericsInNomType(this.nominalType, typeMap));
952         builder.addReceiverType(substGenericsInNomType(this.receiverType, typeMap));
953         // TODO(blickly): Do we need instatiation here?
954         for (String var : this.outerVarPreconditions.keySet()) {
955           builder.addOuterVarPrecondition(var, this.outerVarPreconditions.get(var));
956         }
957         builder.addTypeParameters(this.typeParameters);
958         return builder.buildFunction();
959       }

961       private FunctionType substituteParametricGenerics(Map<String, JSType> typeMap) {
962         if (typeMap.isEmpty()) {
963           return this;
964         }
965         FunctionTypeBuilder builder = new FunctionTypeBuilder();
966         for (JSType reqFormal : this.requiredFormals) {
967           builder.addReqFormal(reqFormal.substituteGenerics(typeMap));
968         }
969         for (JSType optFormal : this.optionalFormals) {
970           builder.addOptFormal(optFormal.substituteGenerics(typeMap));
971         }
972         if (this.restFormals != null) {
973           builder.addRestFormals(restFormals.substituteGenerics(typeMap));
974         }
975         builder.addRetType(this.returnType.substituteGenerics(typeMap));
976         if (isLoose()) {
977           builder.addLoose();
978         }
979         builder.addNominalType(substGenericsInNomType(this.nominalType, typeMap));
980         if (this.receiverType != null) {
981           // NOTE(dimvar):
982           // We have no way of knowing if receiverType comes from an @this
983           // annotation, or because this type represents a method.
984           // In case 1, we would want to substitute in receiverType, in case 2 we
985           // don't, it's a different scope for the type variables.
986           // To properly track this we would need two separate fields instead of
987           // just receiverType.
988           // Instead, the IF test is a heuristic that works in most cases.
989           // In the else branch, we are substituting incorrectly when receiverType
990           // comes from a method declaration, but I have not been able to find a
991           // test that exposes the bug.
992           NominalType recvType = getNominalTypeIfSingletonObj(this.receiverType);
993           if (recvType != null && recvType.isUninstantiatedGenericType()) {
994             builder.addReceiverType(this.receiverType);
995           } else {
996             builder.addReceiverType(substGenericsInNomType(this.receiverType, typeMap));
997           }
998         }
999         // TODO(blickly): Do we need instatiation here?
1000         for (String var : outerVarPreconditions.keySet()) {
1001           builder.addOuterVarPrecondition(var, outerVarPreconditions.get(var));
1002         }
1003         return builder.buildFunction();
1004       }

1007        * FunctionType#substituteGenerics is called while instantiating prototype
1008        * methods of generic nominal types.
1009        */
1010       FunctionType substituteGenerics(Map<String, JSType> concreteTypes) {
1011         if (!isGeneric() || JSType.MAP_TO_UNKNOWN.equals(concreteTypes)) {
1012           return substituteNominalGenerics(concreteTypes);
1013         }
1014         ImmutableMap.Builder<String, JSType> builder = ImmutableMap.builder();
1015         for (Map.Entry<String, JSType> concreteTypeEntry
1016                  : concreteTypes.entrySet()) {
1017           if (!typeParameters.contains(concreteTypeEntry.getKey())) {
1018             builder.put(concreteTypeEntry);
1019           }
1020         }
1021         return substituteNominalGenerics(builder.build());
1022       }

1029       public FunctionType instantiateGenericsFromArgumentTypes(
1030           List<JSType> argTypes) {
1031         Preconditions.checkState(isGeneric());
1032         if (argTypes.size() < getMinArity() || argTypes.size() > getMaxArity()) {
1033           return null;
1034         }
1035         Multimap<String, JSType> typeMultimap = LinkedHashMultimap.create();
1036         for (int i = 0, size = argTypes.size(); i < size; i++) {
1037           if (!this.getFormalType(i)
1038               .unifyWithSubtype(argTypes.get(i), typeParameters, typeMultimap,
1039                   SubtypeCache.create())) {
1040             return null;
1041           }
1042         }
1043         ImmutableMap.Builder<String, JSType> builder = ImmutableMap.builder();
1044         for (String typeParam : typeParameters) {
1045           Collection<JSType> types = typeMultimap.get(typeParam);
1046           if (types.size() != 1) {
1047             return null;
1048           }
1049           builder.put(typeParam, Iterables.getOnlyElement(types));
1050         }
1051         return substituteParametricGenerics(builder.build());
1052       }

1083       public StringBuilder appendTo(StringBuilder builder) {
1084         if (this == LOOSE_TOP_FUNCTION) {
1085           return builder.append("LOOSE_TOP_FUNCTION");
1086         } else if (this == TOP_FUNCTION) {
1087           return builder.append("TOP_FUNCTION");
1088         } else if (this == QMARK_FUNCTION) {
1089           return builder.append("Function");
1090         }
1091         builder.append("function(");
1092         if (nominalType != null) {
1093           builder.append("new:");
1094           builder.append(nominalType);
1095           builder.append(',');
1096         } else if (receiverType != null) {
1097           builder.append("this:");
1098           builder.append(receiverType);
1099           builder.append(',');
1100         }
1101         for (int i = 0; i < requiredFormals.size(); ++i) {
1102           requiredFormals.get(i).appendTo(builder);
1103           builder.append(',');
1104         }
1105         for (int i = 0; i < optionalFormals.size(); ++i) {
1106           optionalFormals.get(i).appendTo(builder);
1107           builder.append("=,");
1108         }
1109         if (restFormals != null) {
1110           builder.append("...");
1111           restFormals.appendTo(builder);
1112         }
1113         // Delete the trailing comma, if present
1114         if (builder.charAt(builder.length() - 1) == ',') {
1115           builder.deleteCharAt(builder.length() - 1);
1116         }
1117         builder.append(')');
1118         if (returnType != null) {
1119           builder.append(':');
1120           returnType.appendTo(builder);
1121         }
1122         if (isLoose()) {
1123           builder.append(" (loose)");
1124         }
1125         if (DEBUGGING && !outerVarPreconditions.isEmpty()) {
1126           builder.append("\tFV: {");
1127           boolean firstIteration = true;
1128           for (Map.Entry<String, JSType> entry : outerVarPreconditions.entrySet()) {
1129             if (!firstIteration) {
1130               builder.append(',');
1131             }
1132             builder.append(entry.getKey());
1133             builder.append('=');
1134             entry.getValue().appendTo(builder);
1135           }
1136           builder.append('}');
1137         }
1138         return builder;
1139       }
