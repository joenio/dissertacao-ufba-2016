789       static CompilationParam[] getSortedValues() {
790         ArrayList<CompilationParam> values = new ArrayList<>(
791             Arrays.asList(CompilationParam.values()));
792     
793         Collections.sort(values, new java.util.Comparator<CompilationParam>() {
794           @Override
795           public int compare(CompilationParam o1, CompilationParam o2) {
796             return o1.toString().compareTo(o2.toString());
797           }
798         });
799     
800         return values.toArray(new CompilationParam[0]);
801       }
