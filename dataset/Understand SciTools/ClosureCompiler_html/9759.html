17     package com.google.javascript.jscomp;
18     
19     import java.util.ArrayList;
20     import java.util.Comparator;
21     import java.util.List;
22     import java.util.SortedSet;
23     import java.util.TreeSet;
24     
25     /**
26      * <p>A basic error manager that sorts all errors and warnings reported to it to
27      * generate a sorted report when the {@link #generateReport()} method
28      * is called.</p>
29      *
30      * <p>This error manager does not produce any output, but subclasses can
31      * override the {@link #println(CheckLevel, JSError)} method to generate custom
32      * output.</p>
33      *
34      */
35     public abstract class BasicErrorManager implements ErrorManager {
36       private final SortedSet<ErrorWithLevel> messages =
37           new TreeSet<>(new LeveledJSErrorComparator());
38       private int errorCount = 0;
39       private int warningCount = 0;
40       private double typedPercent = 0.0;
41     
42       @Override
43       public void report(CheckLevel level, JSError error) {
44         if (messages.add(new ErrorWithLevel(error, level))) {
45           if (level == CheckLevel.ERROR) {
46             errorCount++;
47           } else if (level == CheckLevel.WARNING) {
48             warningCount++;
49           }
50         }
51       }
52     
53       @Override
54       public void generateReport() {
55         for (ErrorWithLevel message : messages) {
56           println(message.level, message.error);
57         }
58         printSummary();
59       }
60     
61       /**
62        * Print a message with a trailing new line. This method is called by the
63        * {@link #generateReport()} method when generating messages.
64        */
65       public abstract void println(CheckLevel level, JSError error);
66     
67       /**
68        * Print the summary of the compilation - number of errors and warnings.
69        */
70       protected abstract void printSummary();
71     
72       @Override
73       public int getErrorCount() {
74         return errorCount;
75       }
76     
77       @Override
78       public int getWarningCount() {
79         return warningCount;
80       }
81     
82       @Override
83       public JSError[] getErrors() {
84         return toArray(CheckLevel.ERROR);
85       }
86     
87       @Override
88       public JSError[] getWarnings() {
89         return toArray(CheckLevel.WARNING);
90       }
91     
92       @Override
93       public void setTypedPercent(double typedPercent) {
94         this.typedPercent = typedPercent;
95       }
96     
97       @Override
98       public double getTypedPercent() {
99         return typedPercent;
100       }
101     
102       private JSError[] toArray(CheckLevel level) {
103         List<JSError> errors = new ArrayList<>(messages.size());
104         for (ErrorWithLevel p : messages) {
105           if (p.level == level) {
106             errors.add(p.error);
107           }
108         }
109         return errors.toArray(new JSError[0]);
110       }
111     
112       /**
113        * <p>Comparator of {@link JSError} with an associated {@link CheckLevel}.
114        * The ordering is the standard lexical ordering on the quintuple
115        * (file name, line number, {@link CheckLevel},
116        * character number, description).</p>
117        *
118        * <p>Note: this comparator imposes orderings that are inconsistent with
119        * {@link JSError#equals(Object)}.</p>
120        */
121       static final class LeveledJSErrorComparator
122           implements Comparator<ErrorWithLevel> {
123         private static final int P1_LT_P2 = -1;
124         private static final int P1_GT_P2 = 1;
125     
126         @Override
127         public int compare(ErrorWithLevel p1, ErrorWithLevel p2) {
128           // null is the smallest value
129           if (p2 == null) {
130             if (p1 == null) {
131               return 0;
132             } else {
133               return P1_GT_P2;
134             }
135           }
136     
137           // check level
138           if (p1.level != p2.level) {
139             return p2.level.compareTo(p1.level);
140           }
141     
142           // sourceName comparison
143           String source1 = p1.error.sourceName;
144           String source2 = p2.error.sourceName;
145           if (source1 != null && source2 != null) {
146             int sourceCompare = source1.compareTo(source2);
147             if (sourceCompare != 0) {
148               return sourceCompare;
149             }
150           } else if (source1 == null && source2 != null) {
151             return P1_LT_P2;
152           } else if (source1 != null && source2 == null) {
153             return P1_GT_P2;
154           }
155           // lineno comparison
156           int lineno1 = p1.error.lineNumber;
157           int lineno2 = p2.error.lineNumber;
158           if (lineno1 != lineno2) {
159             return lineno1 - lineno2;
160           } else if (lineno1 < 0 && 0 <= lineno2) {
161             return P1_LT_P2;
162           } else if (0 <= lineno1 && lineno2 < 0) {
163             return P1_GT_P2;
164           }
165           // charno comparison
166           int charno1 = p1.error.getCharno();
167           int charno2 = p2.error.getCharno();
168           if (charno1 != charno2) {
169             return charno1 - charno2;
170           } else if (charno1 < 0 && 0 <= charno2) {
171             return P1_LT_P2;
172           } else if (0 <= charno1 && charno2 < 0) {
173             return P1_GT_P2;
174           }
175           // description
176           return p1.error.description.compareTo(p2.error.description);
177         }
178       }
179     
180       static class ErrorWithLevel {
181         final JSError error;
182         final CheckLevel level;
183     
184         ErrorWithLevel(JSError error, CheckLevel level) {
185           this.error = error;
186           this.level = level;
187         }
188       }
189     }
