52       @Test
53       public void testAllOf() {
54         String input = "goog.require('goog.dom');";
55         Node root = compileToScriptRoot(getCompiler(input));
56         Node fnCall = root.getFirstFirstChild();
57         assertTrue(fnCall.isCall());

59         Matcher notMatcher = Matchers.not(Matchers.anything());

360       /**
361        * Returns the root script node produced from the compiled JS input.
362        */
363       private Node compileToScriptRoot(Compiler compiler) {
364         Node root = compiler.getRoot();
365         // The last child of the compiler root is a Block node, and the first child
366         // of that is the Script node.
367         return root.getLastChild().getFirstChild();
368       }
369     
370       private Compiler getCompiler(String jsInput) {
371         return getCompiler("", jsInput);
372       }
373     
374       private Compiler getCompiler(String externs, String jsInput) {
375         Compiler compiler = new Compiler();
376         BlackHoleErrorManager.silence(compiler);
377         compiler.disableThreads();
378         CompilerOptions options = RefactoringDriver.getCompilerOptions();
379         compiler.compile(
380             ImmutableList.of(SourceFile.fromCode("externs", externs)),
381             ImmutableList.of(SourceFile.fromCode("test", jsInput)),
382             options);
383         return compiler;
384       }
