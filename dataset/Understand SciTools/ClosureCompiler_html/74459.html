40     package com.google.javascript.rhino.jstype;
41     
42     /**
43      * A "can cast to" relationship visitor.
44      */
45     class CanCastToVisitor implements RelationshipVisitor<Boolean> {
46     
47       @Override
48       public Boolean caseUnknownType(JSType thisType, JSType thatType) {
49         return true;
50       }
51     
52       @Override
53       public Boolean caseNoType(JSType thatType) {
54         return true;
55       }
56     
57       @Override
58       public Boolean caseNoObjectType(JSType thatType) {
59         return true; // TODO(johnlenz): restrict to objects
60       }
61     
62       @Override
63       public Boolean caseAllType(JSType thatType) {
64         return true;
65       }
66     
67       boolean canCastToUnion(JSType thisType, UnionType unionType) {
68         for (JSType type : unionType.getAlternates()) {
69           if (thisType.visit(this, type)) {
70             return true;
71           }
72         }
73         return false;
74       }
75     
76       boolean canCastToFunction(JSType thisType, FunctionType functionType) {
77         if (thisType.isFunctionType()) {
78           // TODO(johnlenz): visit function parts
79           return true;
80         } else {
81           return thisType.isSubtype(functionType)
82               || functionType.isSubtype(thisType);
83         }
84       }
85     
86       private boolean isInterface(JSType type) {
87         ObjectType objType = type.toObjectType();
88         if (objType != null) {
89           JSType constructor = objType.getConstructor();
90           return constructor != null && constructor.isInterface();
91         }
92         return false;
93       }
94     
95       Boolean castCastToHelper(JSType thisType, JSType thatType) {
96         if (thatType.isUnknownType()
97             || thatType.isAllType()
98             || thatType.isNoObjectType() // TODO(johnlenz): restrict to objects
99             || thatType.isNoType()) {
100           return true;
101         } else if (thisType.isRecordType() || thatType.isRecordType()) {
102           return true;  // TODO(johnlenz): are there any misuses we can catch?
103         } else if (isInterface(thisType) || isInterface(thatType)) {
104           return true;  // TODO(johnlenz): are there any misuses we can catch?
105         } else if (thatType.isEnumElementType()) {
106           return thisType.visit(this,
107               thatType.toMaybeEnumElementType().getPrimitiveType());
108         } else if (thatType.isUnionType()) {
109           return canCastToUnion(thisType, thatType.toMaybeUnionType());
110         } else if (thatType.isFunctionType()) {
111           return canCastToFunction(thisType, thatType.toMaybeFunctionType());
112         } else if (thatType.isTemplatizedType()) {
113           // TODO(johnlenz): once the templated type work is finished,
114           // restrict the type parameters.
115           return thisType.visit(this,
116               thatType.toMaybeTemplatizedType().getReferencedTypeInternal());
117         }
118     
119         return thisType.isSubtype(thatType) || thatType.isSubtype(thisType);
120       }
121     
122       @Override
123       public Boolean caseValueType(ValueType thisType, JSType thatType) {
124         return castCastToHelper(thisType, thatType);
125       }
126     
127       @Override
128       public Boolean caseObjectType(ObjectType thisType, JSType thatType) {
129         return castCastToHelper(thisType, thatType);
130       }
131     
132       @Override
133       public Boolean caseFunctionType(FunctionType thisType, JSType thatType) {
134         return castCastToHelper(thisType, thatType);
135       }
136     
137       @Override
138       public Boolean caseUnionType(UnionType thisType, JSType thatType) {
139         boolean visited = false;
140         for (JSType type : thisType.getAlternates()) {
141           if (type.isVoidType() || type.isNullType()) {
142             // Don't allow if the only match between the types is null or void,
143             // otherwise any nullable type would be castable to any other nullable
144             // type and we don't want that.
145           } else {
146             visited = true;
147             if (type.visit(this, thatType)) {
148               return true;
149             }
150           }
151         }
152     
153         // Special case the "null|undefined" union and allow it to be cast
154         // to any cast to any type containing allowing either null|undefined.
155         if (!visited) {
156           JSType NULL_TYPE = thisType.getNativeType(JSTypeNative.NULL_TYPE);
157           JSType VOID_TYPE = thisType.getNativeType(JSTypeNative.VOID_TYPE);
158           return NULL_TYPE.visit(this, thatType) || VOID_TYPE.visit(this, thatType);
159         }
160     
161         return false;
162       }
163     
164       @Override
165       public Boolean caseTemplatizedType(
166           TemplatizedType thisType, JSType thatType) {
167         // TODO(johnlenz): once the templated type work is finished,
168         // restrict the type parameters.
169         return thisType.getReferencedTypeInternal().visit(this, thatType);
170       }
171     
172       @Override
173       public Boolean caseTemplateType(TemplateType thisType, JSType thatType) {
174         return true;
175       }
176     
177       @Override
178       public Boolean caseEnumElementType(
179           EnumElementType typeType, JSType thatType) {
180         return typeType.getPrimitiveType().visit(this, thatType);
181       }
182     }
