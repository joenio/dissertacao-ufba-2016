28     /**
29      * Methods necessary for partially or full decomposing an expression.  Initially
30      * this is intended to expanded the locations were inlining can occur, but has
31      * other uses as well.
32      *
33      * For example:
34      *    var x = y() + z();
35      *
36      * Becomes:
37      *    var a = y();
38      *    var b = z();
39      *    x = a + b;
40      *
41      * @author johnlenz@google.com (John Lenz)
42      */
43     class ExpressionDecomposer {
44     
45       /**
46        * @see #canExposeExpression
47        */
48       enum DecompositionType {
49         UNDECOMPOSABLE,
50         MOVABLE,
51         DECOMPOSABLE
52       }

74       private static final int MAX_INTERATIONS = 100;

76       /**
77        * If required, rewrite the statement containing the expression.
78        * @param expression The expression to be exposed.
79        * @see #canExposeExpression
80        */
81       void maybeExposeExpression(Node expression) {
82         // If the expression needs to exposed.
83         int i = 0;
84         while (DecompositionType.DECOMPOSABLE == canExposeExpression(expression)) {
85           exposeExpression(expression);
86           i++;
87           if (i > MAX_INTERATIONS) {
88             throw new IllegalStateException(
89                 "DecomposeExpression depth exceeded on :\n" +
90                 expression.toStringTree());
91           }
92         }
93       }

95       /**
96        * Perform any rewriting necessary so that the specified expression
97        * is movable. This is a partial expression decomposition.
98        * @see #canExposeExpression
99        */
100       void exposeExpression(Node expression) {
101         Node expressionRoot = findExpressionRoot(expression);
102         Preconditions.checkState(expressionRoot != null);
103         exposeExpression(expressionRoot, expression);
104         compiler.reportCodeChange();
105       }

111       /**
112        * Extract the specified expression from its parent expression.
113        * @see #canExposeExpression
114        */
115       void moveExpression(Node expression) {
116         String resultName = getResultValueName();
117         Node injectionPoint = findInjectionPoint(expression);
118         Preconditions.checkNotNull(injectionPoint);
119         Node injectionPointParent = injectionPoint.getParent();
120         Preconditions.checkNotNull(injectionPointParent);
121         Preconditions.checkState(NodeUtil.isStatementBlock(injectionPointParent));
122     
123         // Replace the expression with a reference to the new name.
124         Node expressionParent = expression.getParent();
125         expressionParent.replaceChild(
126             expression, IR.name(resultName));
127     
128         // Re-add the expression at the appropriate place.
129         Node newExpressionRoot = NodeUtil.newVarNode(resultName, expression);
130         injectionPointParent.addChildBefore(newExpressionRoot, injectionPoint);
131         compiler.reportCodeChange();
132       }

134       /**
135        * Rewrite the expression such that the sub-expression is in a movable
136        * expression statement while maintaining evaluation order.
137        *
138        * Two types of subexpressions are extracted from the source expression:
139        * 1) subexpressions with side-effects.
140        * 2) conditional expressions, that contain the call, which are transformed
141        * into IF statements.
142        *
143        * The following terms are used:
144        *    expressionRoot: The top-level node before which the any extracted
145        *                    expressions should be placed before.
146        *    nonconditionalExpr: The node that will be extracted either express.
147        *
148        */
149       private void exposeExpression(Node expressionRoot, Node subExpression) {
150         Node nonconditionalExpr = findNonconditionalParent(
151             subExpression, expressionRoot);
152         // Before extraction, record whether there are side-effect
153         boolean hasFollowingSideEffects = NodeUtil.mayHaveSideEffects(
154             nonconditionalExpr, compiler);
155     
156         Node exprInjectionPoint = findInjectionPoint(nonconditionalExpr);
157         DecompositionState state = new DecompositionState();
158         state.sideEffects = hasFollowingSideEffects;
159         state.extractBeforeStatement = exprInjectionPoint;
160     
161         // Extract expressions in the reverse order of their evaluation.
162         for (Node grandchild = null,
163                 child = nonconditionalExpr,
164                 parent = child.getParent();
165              parent != expressionRoot;
166              grandchild = child,
167                  child = parent,
168                  parent = child.getParent()) {
169           Token parentType = parent.getType();
170           Preconditions.checkState(
171               !isConditionalOp(parent) || child == parent.getFirstChild());
172           if (parentType == Token.ASSIGN) {
173               if (isSafeAssign(parent, state.sideEffects)) {
174                 // It is always safe to inline "foo()" for expressions such as
175                 // "a = b = c = foo();"
176                 // As the assignment is unaffected by side effect of "foo()"
177                 // and the names assigned-to can not influence the state before
178                 // the call to foo.
179                 //
180                 // This is not true of more complex LHS values, such as
181                 // a.x = foo();
182                 // next().x = foo();
183                 // in these cases the checks below are necessary.
184               } else {
185                 // Alias "next()" in "next().foo"
186                 Node left = parent.getFirstChild();
187                 Token type = left.getType();
188                 if (left != child) {
189                   Preconditions.checkState(NodeUtil.isGet(left));
190                   if (type == Token.GETELEM) {
191                     decomposeSubExpressions(left.getLastChild(), null, state);
192                   }
193                   decomposeSubExpressions(left.getFirstChild(), null, state);
194                 }
195               }
196           } else if (parentType == Token.CALL
197               && NodeUtil.isGet(parent.getFirstChild())) {
198             Node functionExpression = parent.getFirstChild();
199             decomposeSubExpressions(functionExpression.getNext(), child, state);
200             // Now handle the call expression
201             if (isExpressionTreeUnsafe(functionExpression, state.sideEffects)
202                 && functionExpression.getFirstChild() != grandchild) {
203               // TODO(johnlenz): In Internet Explorer, non-JavaScript objects such
204               // as DOM objects can not be decomposed.
205               Preconditions.checkState(allowObjectCallDecomposing(),
206                   "Object method calls can not be decomposed.");
207               // Either there were preexisting side-effects, or this node has
208               // side-effects.
209               state.sideEffects = true;
210     
211               // Rewrite the call so "this" is preserved.
212               Node replacement = rewriteCallExpression(parent, state);
213               // Continue from here.
214               parent = replacement;
215             }
216           } else if (parentType == Token.OBJECTLIT) {
217             decomposeObjectLiteralKeys(parent.getFirstChild(), child, state);
218           } else {
219             decomposeSubExpressions(parent.getFirstChild(), child, state);
220           }
221         }
222     
223         // Now extract the expression that the decomposition is being performed to
224         // to allow to be moved.  All expressions that need to be evaluated before
225         // this have been extracted, so add the expression statement after the
226         // other extracted expressions and the original statement (or replace
227         // the original statement.
228         if (nonconditionalExpr == subExpression) {
229           // Don't extract the call, as that introduces an extra constant VAR
230           // that will simply need to be inlined back.  It will be handled as
231           // an EXPRESSION call site type.
232           // Node extractedCall = extractExpression(decomposition, expressionRoot);
233         } else {
234           Node parent = nonconditionalExpr.getParent();
235           boolean needResult = !parent.isExprResult();
236           extractConditional(nonconditionalExpr, exprInjectionPoint, needResult);
237         }
238       }

252       /**
253        * @return "expression" or the node closest to "expression", that does not
254        * have a conditional ancestor.
255        */
256       private static Node findNonconditionalParent(
257           Node subExpression, Node expressionRoot) {
258          Node result = subExpression;
259     
260          for (Node child = subExpression, parent = child.getParent();
261               parent != expressionRoot;
262               child = parent, parent = child.getParent()) {
263            if (isConditionalOp(parent)) {
264              // Only the first child is always executed, if the function may never
265              // be called, don't inline it.
266              if (child != parent.getFirstChild()) {
267                result = parent;
268              }
269            }
270          }
271     
272          return result;
273       }

285       /**
286        * Decompose an object literal.
287        * @param key The object literal key.
288        * @param stopNode A node after which to stop iterating.
289        */
290       private void decomposeObjectLiteralKeys(
291           Node key, Node stopNode, DecompositionState state) {
292         if (key == null || key == stopNode) {
293           return;
294         }
295         decomposeObjectLiteralKeys(key.getNext(), stopNode, state);
296         decomposeSubExpressions(key.getFirstChild(), stopNode, state);
297       }

299       /**
300        * @param n The node with which to start iterating.
301        * @param stopNode A node after which to stop iterating.
302        */
303       private void decomposeSubExpressions(
304           Node n, Node stopNode, DecompositionState state) {
305         if (n == null || n == stopNode) {
306           return;
307         }
308     
309         // Never try to decompose an object literal key.
310         Preconditions.checkState(!NodeUtil.isObjectLitKey(n));
311     
312         // Decompose the children in reverse evaluation order.  This simplifies
313         // determining if the any of the children following have side-effects.
314         // If they do we need to be more aggressive about removing values
315         // from the expression.
316         decomposeSubExpressions(
317             n.getNext(), stopNode, state);
318     
319         // Now this node.
320         // TODO(johnlenz): Move "safety" code to a shared class.
321         if (isExpressionTreeUnsafe(n, state.sideEffects)) {
322           // Either there were preexisting side-effects, or this node has
323           // side-effects.
324           state.sideEffects = true;
325           state.extractBeforeStatement = extractExpression(
326               n, state.extractBeforeStatement);
327         }
328       }

330       /**
331        *
332        * @param expr The conditional expression to extract.
333        * @param injectionPoint The before which extracted expression, would be
334        *     injected.
335        * @param needResult  Whether the result of the expression is required.
336        * @return The node that contains the logic of the expression after
337        *     extraction.
338        */
339       private Node extractConditional(
340           Node expr, Node injectionPoint, boolean needResult) {
341         Node parent = expr.getParent();
342         String tempName = getTempValueName();
343     
344         // Break down the conditional.
345         Node first = expr.getFirstChild();
346         Node second = first.getNext();
347         Node last = expr.getLastChild();
348     
349         // Isolate the children nodes.
350         expr.detachChildren();
351     
352         // Transform the conditional to an IF statement.
353         Node cond = null;
354         Node trueExpr = IR.block().srcref(expr);
355         Node falseExpr = IR.block().srcref(expr);
356         switch (expr.getType()) {
357           case HOOK:
358             // a = x?y:z --> if (x) {a=y} else {a=z}
359             cond = first;
360             trueExpr.addChildToFront(NodeUtil.newExpr(
361                 buildResultExpression(second, needResult, tempName)));
362             falseExpr.addChildToFront(NodeUtil.newExpr(
363                 buildResultExpression(last, needResult, tempName)));
364             break;
365           case AND:
366             // a = x&&y --> if (a=x) {a=y} else {}
367             cond = buildResultExpression(first, needResult, tempName);
368             trueExpr.addChildToFront(NodeUtil.newExpr(
369                 buildResultExpression(last, needResult, tempName)));
370             break;
371           case OR:
372             // a = x||y --> if (a=x) {} else {a=y}
373             cond = buildResultExpression(first, needResult, tempName);
374             falseExpr.addChildToFront(NodeUtil.newExpr(
375                 buildResultExpression(last, needResult, tempName)));
376             break;
377           default:
378             // With a valid tree we should never get here.
379             throw new IllegalStateException("Unexpected.");
380         }
381     
382         Node ifNode;
383         if (falseExpr.hasChildren()) {
384           ifNode = IR.ifNode(cond, trueExpr, falseExpr);
385         } else {
386           ifNode = IR.ifNode(cond, trueExpr);
387         }
388         ifNode.useSourceInfoIfMissingFrom(expr);
389     
390         if (needResult) {
391           Node tempVarNode = NodeUtil.newVarNode(tempName, null)
392               .useSourceInfoIfMissingFromForTree(expr);
393           Node injectionPointParent = injectionPoint.getParent();
394           injectionPointParent.addChildBefore(tempVarNode, injectionPoint);
395           injectionPointParent.addChildAfter(ifNode, tempVarNode);
396     
397           // Replace the expression with the temporary name.
398           Node replacementValueNode = IR.name(tempName);
399           parent.replaceChild(expr, replacementValueNode);
400         } else {
401           // Only conditionals that are the direct child of an expression statement
402           // don't need results, for those simply replace the expression statement.
403           Preconditions.checkArgument(parent.isExprResult());
404           Node grandparent = parent.getParent();
405           grandparent.replaceChild(parent, ifNode);
406         }
407     
408         return ifNode;
409       }

411       /**
412        * Create an expression tree for an expression.
413        * If the result of the expression is needed, then:
414        *    ASSIGN
415        *       tempName
416        *       expr
417        * otherwise, simply:
418        *       expr
419        */
420       private static Node buildResultExpression(
421           Node expr, boolean needResult, String tempName) {
422         if (needResult) {
423           return IR.assign(
424               IR.name(tempName),
425               expr).srcrefTree(expr);
426         } else {
427           return expr;
428         }
429       }

431       private boolean isConstantNameNode(Node n, Set<String> knownConstants) {
432         // Non-constant names values may have been changed.
433         return n.isName() && (NodeUtil.isConstantVar(n, scope)
434             || knownConstants.contains(n.getString()));
435       }

437       /**
438        * @param expr The expression to extract.
439        * @param injectionPoint The node before which to added the extracted
440        *     expression.
441        * @return The extract statement node.
442        */
443       private Node extractExpression(Node expr, Node injectionPoint) {
444         Node parent = expr.getParent();
445     
446         boolean isLhsOfAssignOp = NodeUtil.isAssignmentOp(parent)
447             && !parent.isAssign()
448             && parent.getFirstChild() == expr;
449     
450         Node firstExtractedNode = null;
451     
452         // Expressions on the LHS of an assignment-op must have any possible
453         // side-effects extracted as the value must be duplicated:
454         //    next().foo += 2;
455         // becomes:
456         //    var t1 = next();
457         //    t1.foo = t1.foo + 2;
458         if (isLhsOfAssignOp && NodeUtil.isGet(expr)) {
459           for (Node n : expr.children()) {
460             if (!n.isString() && !isConstantNameNode(n, knownConstants)) {
461               Node extractedNode = extractExpression(n, injectionPoint);
462               if (firstExtractedNode == null) {
463                 firstExtractedNode = extractedNode;
464               }
465             }
466           }
467         }
468     
469         // The temp is known to be constant.
470         String tempName = getTempConstantValueName();
471         Node replacementValueNode = IR.name(tempName).srcref(expr);
472     
473         Node tempNameValue;
474     
475         // If it is ASSIGN_XXX, keep the assignment in place and extract the
476         // original value of the LHS operand.
477         if (isLhsOfAssignOp) {
478           Preconditions.checkState(expr.isName() || NodeUtil.isGet(expr));
479           // Transform "x += 2" into "x = temp + 2"
480           Node opNode = new Node(NodeUtil.getOpFromAssignmentOp(parent))
481               .useSourceInfoIfMissingFrom(parent);
482     
483           Node rightOperand = parent.getLastChild();
484     
485           parent.setType(Token.ASSIGN);
486           parent.replaceChild(rightOperand, opNode);
487           opNode.addChildToFront(replacementValueNode);
488           opNode.addChildToBack(rightOperand);
489     
490           // The original expression is still being used, so make a clone.
491           tempNameValue = expr.cloneTree();
492         } else {
493           // Replace the expression with the temporary name.
494           parent.replaceChild(expr, replacementValueNode);
495     
496           // Keep the original node so that CALL expressions can still be found
497           // and inlined properly.
498           tempNameValue = expr;
499         }
500     
501         // Re-add the expression in the declaration of the temporary name.
502         Node tempVarNode = NodeUtil.newVarNode(tempName, tempNameValue);
503     
504         Node injectionPointParent = injectionPoint.getParent();
505         injectionPointParent.addChildBefore(tempVarNode, injectionPoint);
506     
507         if (firstExtractedNode == null) {
508           firstExtractedNode = tempVarNode;
509         }
510         return firstExtractedNode;
511       }

513       /**
514        * Rewrite the call so "this" is preserved.
515        *   a.b(c);
516        * becomes:
517        *   var temp1 = a;
518        *   var temp0 = temp1.b;
519        *   temp0.call(temp1,c);
520        *
521        * @return The replacement node.
522        */
523       private Node rewriteCallExpression(Node call, DecompositionState state) {
524         Preconditions.checkArgument(call.isCall());
525         Node first = call.getFirstChild();
526         Preconditions.checkArgument(NodeUtil.isGet(first));
527     
528         // Extracts the expression representing the function to call. For example:
529         //   "a['b'].c" from "a['b'].c()"
530         Node getVarNode = extractExpression(
531             first, state.extractBeforeStatement);
532         state.extractBeforeStatement = getVarNode;
533     
534         // Extracts the object reference to be used as "this". For example:
535         //   "a['b']" from "a['b'].c"
536         Node getExprNode = getVarNode.getFirstFirstChild();
537         Preconditions.checkArgument(NodeUtil.isGet(getExprNode));
538         Node thisVarNode = extractExpression(
539             getExprNode.getFirstChild(), state.extractBeforeStatement);
540         state.extractBeforeStatement = thisVarNode;
541     
542         // Rewrite the CALL expression.
543         Node thisNameNode = thisVarNode.getFirstChild();
544         Node functionNameNode = getVarNode.getFirstChild();
545     
546         // CALL
547         //   GETPROP
548         //     functionName
549         //     "call"
550         //   thisName
551         //   original-parameter1
552         //   original-parameter2
553         //   ...
554         Node newCall = IR.call(
555             IR.getprop(
556                 functionNameNode.cloneNode(),
557                 IR.string("call")),
558             thisNameNode.cloneNode()).srcref(call);
559     
560         // Throw away the call name
561         call.removeFirstChild();
562         if (call.hasChildren()) {
563           // Add the call parameters to the new call.
564           newCall.addChildrenToBack(call.removeChildren());
565         }
566     
567         // Replace the call.
568         Node callParent = call.getParent();
569         callParent.replaceChild(call, newCall);
570     
571         return newCall;
572       }

620       /**
621        * @return For the subExpression, find the nearest statement Node before which
622        * it can be inlined.  Null if no such location can be found.
623        */
624       static Node findInjectionPoint(Node subExpression) {
625         Node expressionRoot = findExpressionRoot(subExpression);
626         Preconditions.checkNotNull(expressionRoot);
627     
628         Node injectionPoint = expressionRoot;
629     
630         Node parent = injectionPoint.getParent();
631         while (parent.isLabel()) {
632           injectionPoint = parent;
633           parent = injectionPoint.getParent();
634         }
635     
636         Preconditions.checkState(
637             NodeUtil.isStatementBlock(injectionPoint.getParent()));
638         return injectionPoint;
639       }

641       /**
642        * @return Whether the node is a conditional op.
643        */
644       private static boolean isConditionalOp(Node n) {
645         switch(n.getType()) {
646           case HOOK:
647           case AND:
648           case OR:
649             return true;
650           default:
651             return false;
652         }
653       }

655       /**
656        * @return The statement containing the expression or null if the subExpression
657        *     is not contain in a Node where inlining is known to be possible.
658        *     For example, a WHILE node condition expression.
659        */
660       static Node findExpressionRoot(Node subExpression) {
661         Node child = subExpression;
662         for (Node parent : child.getAncestors()) {
663           Token parentType = parent.getType();
664           switch (parentType) {
665             // Supported expression roots:
666             // SWITCH and IF can have multiple children, but the CASE, DEFAULT,
667             // or BLOCK will be encountered first for any of the children other
668             // than the condition.
669             case EXPR_RESULT:
670             case IF:
671             case SWITCH:
672             case RETURN:
673             case THROW:
674             case VAR:
675             case CONST:
676             case LET:
677               Preconditions.checkState(child == parent.getFirstChild());
678               return parent;
679             // Any of these indicate an unsupported expression:
680             case FOR:
681               if (!NodeUtil.isForIn(parent) && child == parent.getFirstChild()) {
682                 return parent;
683               }
684             case SCRIPT:
685             case BLOCK:
686             case LABEL:
687             case CASE:
688             case DEFAULT_CASE:
689               return null;
690           }
691           child = parent;
692         }
693     
694         throw new IllegalStateException("Unexpected AST structure.");
695       }

697       /**
698        * Determine whether a expression is movable, or can be be made movable after
699        * decomposing the containing expression.
700        *
701        * A subexpression is MOVABLE if it can be replaced with a temporary holding
702        * its results and moved to immediately before the root of the expression.
703        * There are three conditions that must be met for this to occur:
704        * 1) There must be a location to inject a statement for the expression.  For
705        * example, this condition can not be met if the expression is a loop
706        * condition or CASE condition.
707        * 2) If the expression can be affect by side-effects, there can not be a
708        * side-effect between original location and the expression root.
709        * 3) If the expression has side-effects, there can not be any other
710        * expression that can be effected between the original location and the
711        * expression root.
712        *
713        * An expression is DECOMPOSABLE if it can be rewritten so that an
714        * subExpression is MOVABLE.
715        *
716        * An expression is decomposed by moving any other sub-expressions that
717        * preventing an subExpression from being MOVABLE.
718        *
719        * @return Whether This is a call that can be moved to an new point in the
720        * AST to allow it to be inlined.
721        */
722       DecompositionType canExposeExpression(Node subExpression) {
723         Node expressionRoot = findExpressionRoot(subExpression);
724         if (expressionRoot != null) {
725           return isSubexpressionMovable(expressionRoot, subExpression);
726         }
727         return DecompositionType.UNDECOMPOSABLE;
728       }
729     
730       /**
731        * Walk the AST from the call site to the expression root and verify that
732        * the portions of the expression that are evaluated before the call are:
733        * 1) Unaffected by the the side-effects, if any, of the call.
734        * 2) That there are no side-effects, that may influence the call.
735        *
736        * For example, if x has side-effects:
737        *   a = 1 + x();
738        * the call to x can be moved because "a" final value of a can not be
739        * influenced by x(), but in:
740        *   a = b + x();
741        * the call to x can not be moved because the value of b may be modified
742        * by the call to x.
743        *
744        * If x is without side-effects in:
745        *   a = b + x();
746        * the call to x can be moved, but in:
747        *   a = (b.foo = c) + x();
748        * the call to x can not be moved because the value of b.foo may be referenced
749        * by x().  Note: this is true even if b is a local variable; the object that
750        * b refers to may have a global alias.
751        *
752        * @return UNDECOMPOSABLE if the expression can not be moved, DECOMPOSABLE if
753        * decomposition is required before the expression can be moved, otherwise
754        * MOVABLE.
755        */
756       private DecompositionType isSubexpressionMovable(
757           Node expressionRoot, Node subExpression) {
758         boolean requiresDecomposition = false;
759         boolean seenSideEffects = NodeUtil.mayHaveSideEffects(
760             subExpression, compiler);
761     
762         Node child = subExpression;
763         for (Node parent : child.getAncestors()) {
764           if (parent == expressionRoot) {
765             // Done. The walk back to the root of the expression is complete, and
766             // nothing was encountered that blocks the call from being moved.
767             return requiresDecomposition
768                 ? DecompositionType.DECOMPOSABLE
769                 : DecompositionType.MOVABLE;
770           }
771     
772           if (isConditionalOp(parent)) {
773             // Only the first child is always executed, otherwise it must be
774             // decomposed.
775             if (child != parent.getFirstChild()) {
776               requiresDecomposition = true;
777             }
778           } else {
779             // Only inline the call if none of the preceding siblings in the
780             // expression have side-effects, and are unaffected by the side-effects,
781             // if any, of the call in question.
782             // NOTE: This depends on the siblings being in the same order as they
783             // are evaluated.
784     
785             // SPECIAL CASE: Assignment to a simple name
786             if (isSafeAssign(parent, seenSideEffects)) {
787               // It is always safe to inline "foo()" for expressions such as
788               //   "a = b = c = foo();"
789               // As the assignment is unaffected by side effect of "foo()"
790               // and the names assigned-to can not influence the state before
791               // the call to foo.
792               //
793               // This is not true of more complex LHS values, such as
794               //    a.x = foo();
795               //    next().x = foo();
796               // in these cases the checks below are necessary.
797             } else {
798               // Everything else.
799               for (Node n : parent.children()) {
800                 if (n == child) {
801                   // None of the preceding siblings have side-effects.
802                   // This is OK.
803                   break;
804                 }
805     
806                 if (isExpressionTreeUnsafe(
807                     n, seenSideEffects)) {
808                   seenSideEffects = true;
809                   requiresDecomposition = true;
810                 }
811               }
812     
813               // In Internet Explorer, DOM objects and other external objects
814               // methods can not be called indirectly, as is required when the
815               // object or its property can be side-effected.  For example,
816               // when exposing expression f() (with side-effects) in: x.m(f())
817               // either the value of x or its property m might have changed, so
818               // both the 'this' value ('x') and the function to be called ('x.m')
819               // need to be preserved. Like so:
820               //   var t1 = x, t2 = x.m, t3 = f();
821               //   t2.call(t1, t3);
822               // As IE doesn't support the call to these non-JavaScript objects
823               // methods in this way. We can't do this.
824               // We don't currently distinguish between these types of objects
825               // in the extern definitions and if we did we would need accurate
826               // type information.
827               //
828               Node first = parent.getFirstChild();
829               if (requiresDecomposition
830                   && parent.isCall()
831                   && NodeUtil.isGet(first)) {
832                 if (maybeExternMethod(first)) {
833                   return DecompositionType.UNDECOMPOSABLE;
834                 } else {
835                   return DecompositionType.DECOMPOSABLE;
836                 }
837               }
838             }
839           }
840           // Continue looking up the expression tree.
841           child = parent;
842         }
843     
844         // With a valid tree we should never get here.
845         throw new IllegalStateException("Unexpected.");
846       }
847     
848       /**
849        * It is always safe to inline "foo()" for expressions such as
850        *    "a = b = c = foo();"
851        * As the assignment is unaffected by side effect of "foo()"
852        * and the names assigned-to can not influence the state before
853        * the call to foo.
854        *
855        * It is also safe in cases like where the object is constant:
856        *    CONST_NAME.a = foo()
857        *    CONST_NAME[CONST_VALUE] = foo();
858        *
859        * This is not true of more complex LHS values, such as
860        *     a.x = foo();
861        *     next().x = foo();
862        * in these cases the checks below are necessary.
863        *
864        * @param seenSideEffects If true, check to see if node-tree maybe affected by
865        * side-effects, otherwise if the tree has side-effects. @see
866        * isExpressionTreeUnsafe
867        * @return Whether the assignment is safe from side-effects.
868        */
869       private boolean isSafeAssign(Node n, boolean seenSideEffects) {
870         if (n.isAssign()) {
871           Node lhs = n.getFirstChild();
872           switch (lhs.getType()) {
873             case NAME:
874               return true;
875             case GETPROP:
876               return !isExpressionTreeUnsafe(lhs.getFirstChild(), seenSideEffects);
877             case GETELEM:
878               return !isExpressionTreeUnsafe(lhs.getFirstChild(), seenSideEffects)
879                   && !isExpressionTreeUnsafe(lhs.getLastChild(), seenSideEffects);
880           }
881         }
882         return false;
883       }
884     
885       /**
886        * @return Whether anything in the expression tree prevents a call from
887        * being moved.
888        */
889       private boolean isExpressionTreeUnsafe(
890           Node n, boolean followingSideEffectsExist) {
891         if (followingSideEffectsExist) {
892           // If the call to be inlined has side-effects, check to see if this
893           // expression tree can be affected by any side-effects.
894     
895           // This is a superset of "NodeUtil.mayHaveSideEffects".
896           return NodeUtil.canBeSideEffected(n, this.knownConstants, scope);
897         } else {
898           // The function called doesn't have side-effects but check to see if there
899           // are side-effects that that may affect it.
900           return NodeUtil.mayHaveSideEffects(n, compiler);
901         }
902       }
