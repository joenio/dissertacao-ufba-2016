35     /**
36      * Converts ES6 generator functions to valid ES3 code. This pass runs after all ES6 features
37      * except for yield and generators have been transpiled.
38      *
39      * @author mattloring@google.com (Matthew Loring)
40      */
41     public final class Es6RewriteGenerators
42         extends NodeTraversal.AbstractPostOrderCallback implements HotSwapCompilerPass {

44       // Name of the variable that holds the state at which the generator
45       // should resume execution after a call to yield or return.
46       // The beginning state is 0 and the end state is -1.
47       private static final String GENERATOR_STATE = "$jscomp$generator$state";
48       private static final String GENERATOR_DO_WHILE_INITIAL = "$jscomp$generator$first$do";
49       private static final String GENERATOR_YIELD_ALL_NAME = "$jscomp$generator$yield$all";
50       private static final String GENERATOR_YIELD_ALL_ENTRY = "$jscomp$generator$yield$entry";
51       private static final String GENERATOR_ARGUMENTS = "$jscomp$generator$arguments";
52       private static final String GENERATOR_THIS = "$jscomp$generator$this";
53       private static final String GENERATOR_NEXT_ARG = "$jscomp$generator$next$arg";
54       private static final String GENERATOR_THROW_ARG = "$jscomp$generator$throw$arg";

111       @Override
112       public void visit(NodeTraversal t, Node n, Node parent) {
113         switch (n.getType()) {
114           case FUNCTION:
115             if (n.isGeneratorFunction()) {
116               generatorCaseCount = 0;
117               visitGenerator(n, parent);
118             }
119             break;
120           case NAME:
121             Node enclosing = NodeUtil.getEnclosingFunction(n);
122             if (enclosing != null
123                 && enclosing.isGeneratorFunction()
124                 && n.matchesQualifiedName("arguments")) {
125               n.setString(GENERATOR_ARGUMENTS);
126             }
127             break;
128           case THIS:
129             enclosing = NodeUtil.getEnclosingFunction(n);
130             if (enclosing != null && enclosing.isGeneratorFunction()) {
131               n.getParent().replaceChild(n, IR.name(GENERATOR_THIS));
132             }
133             break;
134           case YIELD:
135             if (n.isYieldFor()) {
136               visitYieldFor(t, n, parent);
137             } else if (!parent.isExprResult()) {
138               visitYieldExpr(n, parent);
139             } else {
140               visitYieldThrows(parent, parent.getParent());
141             }
142             break;
143           default:
144             break;
145         }
146       }

148       private void visitYieldThrows(Node n, Node parent) {
149         Node ifThrows =
150             IR.ifNode(
151                 IR.shne(IR.name(GENERATOR_THROW_ARG), IR.name("undefined")),
152                 IR.block(IR.throwNode(IR.name(GENERATOR_THROW_ARG))));
153         parent.addChildAfter(ifThrows, n);
154         compiler.reportCodeChange();
155       }

157       /**
158        * Translates expressions using the new yield-for syntax.
159        *
160        * <p>Sample translation:
161        *
162        * <pre>
163        * var i = yield * gen();
164        * </pre>
165        *
166        * <p>Is rewritten to:
167        *
168        * <pre>
169        * var $jscomp$generator$yield$all = gen();
170        * var $jscomp$generator$yield$entry;
171        * while (!($jscomp$generator$yield$entry =
172        *     $jscomp$generator$yield$all.next($jscomp$generator$next$arg)).done) {
173        *   yield $jscomp$generator$yield$entry.value;
174        * }
175        * var i = $jscomp$generator$yield$entry.value;
176        * </pre>
177        */
178       private void visitYieldFor(NodeTraversal t, Node n, Node parent) {
179         Node enclosingStatement = NodeUtil.getEnclosingStatement(n);
180         Node generator = IR.var(
181             IR.name(GENERATOR_YIELD_ALL_NAME),
182             makeIterator(compiler, n.removeFirstChild()));
183         Node entryDecl = IR.var(IR.name(GENERATOR_YIELD_ALL_ENTRY));
184         Node assignIterResult =
185             IR.assign(
186                 IR.name(GENERATOR_YIELD_ALL_ENTRY),
187                 IR.call(
188                     IR.getprop(IR.name(GENERATOR_YIELD_ALL_NAME), IR.string("next")),
189                     IR.name(GENERATOR_NEXT_ARG)));
190         Node loopCondition = IR.not(IR.getprop(assignIterResult, IR.string("done")));
191         Node elemValue = IR.getprop(IR.name(GENERATOR_YIELD_ALL_ENTRY), IR.string("value"));
192         Node yieldStatement = IR.exprResult(IR.yield(elemValue.cloneTree()));
193         Node loop = IR.whileNode(loopCondition, IR.block(yieldStatement));
194     
195         enclosingStatement.getParent().addChildBefore(generator, enclosingStatement);
196         enclosingStatement.getParent().addChildBefore(entryDecl, enclosingStatement);
197         enclosingStatement.getParent().addChildBefore(loop, enclosingStatement);
198         if (parent.isExprResult()) {
199           parent.detachFromParent();
200         } else {
201           parent.replaceChild(n, elemValue);
202         }
203     
204         visitYieldThrows(yieldStatement, yieldStatement.getParent());
205         compiler.reportCodeChange();
206       }

208       private void visitYieldExpr(Node n, Node parent) {
209         Node enclosingStatement = NodeUtil.getEnclosingStatement(n);
210         Node yieldStatement =
211             IR.exprResult(n.hasChildren() ? IR.yield(n.removeFirstChild()) : IR.yield());
212         Node yieldResult = IR.name(GENERATOR_NEXT_ARG + generatorCounter.get());
213         Node yieldResultDecl = IR.var(yieldResult.cloneTree(), IR.name(GENERATOR_NEXT_ARG));
214     
215         parent.replaceChild(n, yieldResult);
216         enclosingStatement.getParent().addChildBefore(yieldStatement, enclosingStatement);
217         enclosingStatement.getParent().addChildBefore(yieldResultDecl, enclosingStatement);
218     
219         visitYieldThrows(yieldStatement, yieldStatement.getParent());
220         compiler.reportCodeChange();
221       }

223       private void visitGenerator(Node n, Node parent) {
224         compiler.ensureLibraryInjected("es6_runtime", false);
225         hasTranslatedTry = false;
226         Node genBlock = compiler.parseSyntheticCode(Joiner.on('\n').join(
227                 "function generatorBody() {",
228                 "  var " + GENERATOR_STATE + " = " + generatorCaseCount + ";",
229                 "  function $jscomp$generator$impl(" + GENERATOR_NEXT_ARG + ", ",
230                 "      " + GENERATOR_THROW_ARG + ") {",
231                 "    while (1) switch (" + GENERATOR_STATE + ") {",
232                 "      case " + generatorCaseCount + ":",
233                 "      default:",
234                 "        return {value: undefined, done: true};",
235                 "    }",
236                 "  }",
237                 // TODO(tbreisacher): Remove this cast if we start returning an actual Generator object.
238                 "  var iterator = /** @type {!Generator<?>} */ ({",
239                 "    next: function(arg) { return $jscomp$generator$impl(arg, undefined); },",
240                 "    throw: function(arg) { return $jscomp$generator$impl(undefined, arg); },",
241                 // TODO(tbreisacher): Implement Generator.return:
242                 // http://www.ecma-international.org/ecma-262/6.0/#sec-generator.prototype.return
243                 "    return: function(arg) { throw Error('Not yet implemented'); },",
244                 "  });",
245                 "  $jscomp.initSymbolIterator();",
246                 "  /** @this {!Generator<?>} */",
247                 "  iterator[Symbol.iterator] = function() { return this; };",
248                 "  return iterator;",
249                 "}"))
250             .getFirstChild() // function
251             .getLastChild() // function body
252             .detachFromParent();
253         generatorCaseCount++;
254     
255         originalGeneratorBody = n.getLastChild();
256         n.replaceChild(originalGeneratorBody, genBlock);
257         n.setIsGeneratorFunction(false);
258     
259         // TODO(mattloring): remove this suppression once we can optimize the switch statement to
260         // remove unused cases.
261         JSDocInfoBuilder builder = JSDocInfoBuilder.maybeCopyFrom(n.getJSDocInfo());
262         // TODO(mattloring): copy existing suppressions.
263         builder.recordSuppressions(ImmutableSet.of("uselessCode"));
264         JSDocInfo info = builder.build();
265         n.setJSDocInfo(info);
266     
267         // Set state to the default after the body of the function has completed.
268         originalGeneratorBody.addChildToBack(
269             IR.exprResult(IR.assign(IR.name(GENERATOR_STATE), IR.number(-1))));
270     
271         enclosingBlock = getUnique(genBlock, Token.CASE).getLastChild();
272         hoistRoot = genBlock.getFirstChild();
273     
274         if (NodeUtil.isNameReferenced(originalGeneratorBody, GENERATOR_ARGUMENTS)) {
275           hoistRoot
276               .getParent()
277               .addChildAfter(IR.var(IR.name(GENERATOR_ARGUMENTS), IR.name("arguments")), hoistRoot);
278         }
279         if (NodeUtil.isNameReferenced(originalGeneratorBody, GENERATOR_THIS)) {
280           hoistRoot
281               .getParent()
282               .addChildAfter(IR.var(IR.name(GENERATOR_THIS), IR.thisNode()), hoistRoot);
283         }
284     
285         while (originalGeneratorBody.hasChildren()) {
286           currentStatement = originalGeneratorBody.removeFirstChild();
287           boolean advanceCase = translateStatementInOriginalBody();
288     
289           if (advanceCase) {
290             int caseNumber;
291             if (currentStatement.isGeneratorMarker()) {
292               caseNumber = (int) currentStatement.getFirstChild().getDouble();
293             } else {
294               caseNumber = generatorCaseCount;
295               generatorCaseCount++;
296             }
297             Node oldCase = enclosingBlock.getParent();
298             Node newCase = IR.caseNode(IR.number(caseNumber), IR.block());
299             enclosingBlock = newCase.getLastChild();
300             if (oldCase.isTry()) {
301               oldCase = oldCase.getGrandparent();
302               if (!currentExceptionContext.isEmpty()) {
303                 Node newTry =
304                     IR.tryCatch(IR.block(), currentExceptionContext.get(0).catchBlock.cloneTree());
305                 newCase.getLastChild().addChildToBack(newTry);
306                 enclosingBlock = newCase.getLastChild().getLastChild().getFirstChild();
307               }
308             }
309             oldCase.getParent().addChildAfter(newCase, oldCase);
310           }
311         }
312     
313         parent.useSourceInfoIfMissingFromForTree(parent);
314         compiler.reportCodeChange();
315       }

317       /** Returns {@code true} if a new case node should be added */
318       private boolean translateStatementInOriginalBody() {
319         if (currentStatement.isVar()) {
320           visitVar();
321           return false;
322         } else if (currentStatement.isGeneratorMarker()) {
323           visitGeneratorMarker();
324           return true;
325         } else if (currentStatement.isFunction()) {
326           visitFunctionStatement();
327           return false;
328         } else if (currentStatement.isBlock()) {
329           visitBlock();
330           return false;
331         } else if (controlCanExit(currentStatement)) {
332           switch (currentStatement.getType()) {
333             case WHILE:
334             case DO:
335             case FOR:
336               if (NodeUtil.isForIn(currentStatement)) {
337                 visitForIn();
338                 return false;
339               }
340               visitLoop(null);
341               return false;
342             case LABEL:
343               visitLabel();
344               return false;
345             case SWITCH:
346               visitSwitch();
347               return false;
348             case IF:
349               if (!currentStatement.isGeneratorSafe()) {
350                 visitIf();
351                 return false;
352               }
353               break;
354             case TRY:
355               visitTry();
356               return false;
357             case EXPR_RESULT:
358               if (currentStatement.getFirstChild().isYield()) {
359                 visitYieldExprResult();
360                 return true;
361               }
362               break;
363             case RETURN:
364               visitReturn();
365               return false;
366             case CONTINUE:
367               visitContinue();
368               return false;
369             case BREAK:
370               if (!currentStatement.isGeneratorSafe()) {
371                 visitBreak();
372                 return false;
373               }
374               break;
375             case THROW:
376               visitThrow();
377               return false;
378             default:
379               // We never want to copy over an untranslated statement for which control exits.
380               throw new RuntimeException(
381                   "Untranslatable control-exiting statement in generator function: "
382                       + currentStatement.getType());
383           }
384         }
385     
386         // In the default case, add the statement to the current case block unchanged.
387         enclosingBlock.addChildToBack(currentStatement);
388         return false;
389       }

391       private void visitFunctionStatement() {
392         hoistRoot.getParent().addChildAfter(currentStatement, hoistRoot);
393       }

395       private void visitTry() {
396         Node tryBody = currentStatement.getFirstChild();
397         Node caughtError;
398         Node catchBody;
399         Node catchBlock = tryBody.getNext();
400         if (catchBlock.hasChildren()) {
401           // There is a catch block
402           caughtError = catchBlock.getFirstChild().removeFirstChild();
403           catchBody = catchBlock.getFirstChild().removeFirstChild();
404         } else {
405           caughtError = IR.name(GENERATOR_ERROR + "temp");
406           catchBody = IR.block(IR.throwNode(caughtError.cloneTree()));
407           catchBody.getFirstChild().setGeneratorSafe(true);
408         }
409         Node finallyBody = catchBlock.getNext();
410         int catchStartState = generatorCaseCount++;
411         Node catchStart = makeGeneratorMarker(catchStartState);
412     
413         Node errorNameGenerated = IR.name("$jscomp$generator$" + caughtError.getString());
414     
415         originalGeneratorBody.addChildToFront(catchStart);
416         originalGeneratorBody.addChildAfter(catchBody, catchStart);
417     
418         Node assignError = IR.assign(IR.name(GENERATOR_ERROR), errorNameGenerated.cloneTree());
419         Node newCatchBody =
420             IR.block(IR.exprResult(assignError), createStateUpdate(catchStartState), createSafeBreak());
421         Node newCatch = IR.catchNode(errorNameGenerated, newCatchBody);
422     
423         currentExceptionContext.add(0, new ExceptionContext(catchStartState, newCatch));
424     
425         if (finallyBody != null) {
426           Node finallyName = IR.name(GENERATOR_FINALLY_JUMP + generatorCounter.get());
427           int finallyStartState = generatorCaseCount++;
428           Node finallyStart = makeGeneratorMarker(finallyStartState);
429           int finallyEndState = generatorCaseCount++;
430           Node finallyEnd = makeGeneratorMarker(finallyEndState);
431     
432           NodeTraversal.traverseEs6(
433               compiler, tryBody, new ControlExitsCheck(finallyName, finallyStartState));
434           NodeTraversal.traverseEs6(
435               compiler, catchBody, new ControlExitsCheck(finallyName, finallyStartState));
436           originalGeneratorBody.addChildToFront(tryBody.detachFromParent());
437     
438           originalGeneratorBody.addChildAfter(finallyStart, catchBody);
439           originalGeneratorBody.addChildAfter(finallyBody.detachFromParent(), finallyStart);
440           originalGeneratorBody.addChildAfter(finallyEnd, finallyBody);
441           originalGeneratorBody.addChildToFront(IR.var(finallyName.cloneTree()));
442     
443           finallyBody.addChildToBack(
444               IR.exprResult(IR.assign(IR.name(GENERATOR_STATE), finallyName.cloneTree())));
445           finallyBody.addChildToBack(createSafeBreak());
446           tryBody.addChildToBack(
447               IR.exprResult(IR.assign(finallyName.cloneTree(), IR.number(finallyEndState))));
448           tryBody.addChildToBack(createStateUpdate(finallyStartState));
449           tryBody.addChildToBack(createSafeBreak());
450           catchBody.addChildToBack(
451               IR.exprResult(IR.assign(finallyName.cloneTree(), IR.number(finallyEndState))));
452         } else {
453           int catchEndState = generatorCaseCount++;
454           Node catchEnd = makeGeneratorMarker(catchEndState);
455           originalGeneratorBody.addChildAfter(catchEnd, catchBody);
456           tryBody.addChildToBack(createStateUpdate(catchEndState));
457           tryBody.addChildToBack(createSafeBreak());
458           originalGeneratorBody.addChildToFront(tryBody.detachFromParent());
459         }
460     
461         catchBody.addChildToFront(IR.var(caughtError, IR.name(GENERATOR_ERROR)));
462     
463         if (enclosingBlock.getParent().isTry()) {
464           enclosingBlock = enclosingBlock.getGrandparent();
465         }
466     
467         enclosingBlock.addChildToBack(IR.tryCatch(IR.block(), newCatch));
468         enclosingBlock = enclosingBlock.getLastChild().getFirstChild();
469         if (!hasTranslatedTry) {
470           hasTranslatedTry = true;
471           hoistRoot.getParent().addChildAfter(IR.var(IR.name(GENERATOR_ERROR)), hoistRoot);
472         }
473       }

475       private void visitContinue() {
476         Preconditions.checkState(currentLoopContext.get(0).continueCase != -1);
477         int continueCase;
478         if (currentStatement.hasChildren()) {
479           continueCase = getLoopContext(currentStatement.removeFirstChild().getString()).continueCase;
480         } else {
481           continueCase = currentLoopContext.get(0).continueCase;
482         }
483         enclosingBlock.addChildToBack(createStateUpdate(continueCase));
484         enclosingBlock.addChildToBack(createSafeBreak());
485       }
486     
487       private void visitThrow() {
488         enclosingBlock.addChildToBack(createStateUpdate(-1));
489         enclosingBlock.addChildToBack(currentStatement);
490       }
491     
492       private void visitBreak() {
493         int breakCase;
494         if (currentStatement.hasChildren()) {
495           LoopContext loop = getLoopContext(currentStatement.removeFirstChild().getString());
496           if (loop == null) {
497             compiler.report(
498                 JSError.make(
499                     currentStatement,
500                     Es6ToEs3Converter.CANNOT_CONVERT_YET,
501                     "Breaking to a label that is not a loop"));
502             return;
503           }
504           breakCase = loop.breakCase;
505         } else {
506           breakCase = currentLoopContext.get(0).breakCase;
507         }
508         enclosingBlock.addChildToBack(createStateUpdate(breakCase));
509         enclosingBlock.addChildToBack(createSafeBreak());
510       }

512       private void visitLabel() {
513         Node labelName = currentStatement.removeFirstChild();
514         Node child = currentStatement.removeFirstChild();
515         if (NodeUtil.isLoopStructure(child)) {
516           currentStatement = child;
517           visitLoop(labelName.getString());
518         } else {
519           originalGeneratorBody.addChildToFront(child);
520         }
521       }

523       /**
524        * Pops the loop information off of our stack if we reach the marker cooresponding
525        * to the end of the current loop.
526        */
527       private void visitGeneratorMarker() {
528         if (!currentLoopContext.isEmpty()
529             && currentLoopContext.get(0).breakCase == currentStatement.getFirstChild().getDouble()) {
530           currentLoopContext.remove(0);
531         }
532         if (!currentExceptionContext.isEmpty()
533             && currentExceptionContext.get(0).catchStartCase
534                 == currentStatement.getFirstChild().getDouble()) {
535           currentExceptionContext.remove(0);
536         }
537       }

539       /**
540        * Uses a case statement to jump over the body if the condition of the
541        * if statement is false. Additionally, lift the body of the {@code if}
542        * statement to the top level.
543        */
544       private void visitIf() {
545         Node condition = currentStatement.removeFirstChild();
546         Node ifBody = currentStatement.removeFirstChild();
547         boolean hasElse = currentStatement.hasChildren();
548     
549         int ifEndState = generatorCaseCount++;
550     
551         Node invertedConditional =
552             IR.ifNode(IR.not(condition), IR.block(createStateUpdate(ifEndState), createSafeBreak()));
553         invertedConditional.setGeneratorSafe(true);
554         Node endIf = makeGeneratorMarker(ifEndState);
555     
556         originalGeneratorBody.addChildToFront(invertedConditional);
557         originalGeneratorBody.addChildAfter(ifBody, invertedConditional);
558         originalGeneratorBody.addChildAfter(endIf, ifBody);
559     
560         if (hasElse) {
561           Node elseBlock = currentStatement.removeFirstChild();
562     
563           int elseEndState = generatorCaseCount++;
564     
565           Node endElse = makeGeneratorMarker(elseEndState);
566     
567           ifBody.addChildToBack(createStateUpdate(elseEndState));
568           ifBody.addChildToBack(createSafeBreak());
569           originalGeneratorBody.addChildAfter(elseBlock, endIf);
570           originalGeneratorBody.addChildAfter(endElse, elseBlock);
571         }
572       }

574       /**
575        * Translates switch statements into a series of if statements.
576        *
577        * <p>Sample translation:
578        * <pre>
579        * switch (i) {
580        *   case 1:
581        *     s;
582        *   case 2:
583        *     t;
584        *   ...
585        * }
586        * </pre>
587        *
588        * <p>Is eventually rewritten to:
589        *
590        * <pre>
591        * $jscomp$generator$switch$entered0 = false;
592        * if ($jscomp$generator$switch$entered0 || i == 1) {
593        *   $jscomp$generator$switch$entered0 = true;
594        *   s;
595        * }
596        * if ($jscomp$generator$switch$entered0 || i == 2) {
597        *   $jscomp$generator$switch$entered0 = true;
598        *   t;
599        * }
600        * ...
601        *
602        * </pre>
603        */
604       private void visitSwitch() {
605         Node didEnter = IR.name(GENERATOR_SWITCH_ENTERED + generatorCounter.get());
606         Node didEnterDecl = IR.var(didEnter.cloneTree(), IR.falseNode());
607         Node switchVal = IR.name(GENERATOR_SWITCH_VAL + generatorCounter.get());
608         Node switchValDecl = IR.var(switchVal.cloneTree(), currentStatement.removeFirstChild());
609         originalGeneratorBody.addChildToFront(didEnterDecl);
610         originalGeneratorBody.addChildAfter(switchValDecl, didEnterDecl);
611         Node insertionPoint = switchValDecl;
612     
613         while (currentStatement.hasChildren()) {
614           Node currCase = currentStatement.removeFirstChild();
615           Node equivBlock;
616           currCase
617               .getLastChild()
618               .addChildToFront(IR.exprResult(IR.assign(didEnter.cloneTree(), IR.trueNode())));
619           if (currCase.isDefaultCase()) {
620             if (currentStatement.hasChildren()) {
621               compiler.report(
622                   JSError.make(
623                       currentStatement,
624                       Es6ToEs3Converter.CANNOT_CONVERT_YET,
625                       "Default case as intermediate case"));
626             }
627             equivBlock = IR.block(currCase.removeFirstChild());
628           } else {
629             equivBlock =
630                 IR.ifNode(
631                     IR.or(
632                         didEnter.cloneTree(),
633                         IR.sheq(switchVal.cloneTree(), currCase.removeFirstChild())),
634                     currCase.removeFirstChild());
635           }
636           originalGeneratorBody.addChildAfter(equivBlock, insertionPoint);
637           insertionPoint = equivBlock;
638         }
639     
640         int breakTarget = generatorCaseCount++;
641         int cont = currentLoopContext.isEmpty() ? -1 : currentLoopContext.get(0).continueCase;
642         currentLoopContext.add(0, new LoopContext(breakTarget, cont, null));
643         Node breakCase = makeGeneratorMarker(breakTarget);
644         originalGeneratorBody.addChildAfter(breakCase, insertionPoint);
645       }
646     
647       /**
648        * Lifts all children to the body of the original generator to flatten the block.
649        */
650       private void visitBlock() {
651         if (currentStatement.getChildCount() == 0) {
652           return;
653         }
654         Node insertionPoint = currentStatement.removeFirstChild();
655         originalGeneratorBody.addChildToFront(insertionPoint);
656         for (Node child = currentStatement.removeFirstChild();
657             child != null;
658             child = currentStatement.removeFirstChild()) {
659           originalGeneratorBody.addChildAfter(child, insertionPoint);
660           insertionPoint = child;
661         }
662       }

664       /**
665        * Translates for in loops to a for in loop which produces an array of
666        * values iterated over followed by a plain for loop which performs the logic
667        * contained in the body of the original for in.
668        *
669        * <p>Sample translation:
670        * <pre>
671        * for (i in j) {
672        *   s;
673        * }
674        * </pre>
675        *
676        * <p>Is eventually rewritten to:
677        *
678        * <pre>
679        * $jscomp$arr = [];
680        * $jscomp$iter = j;
681        * for (i in $jscomp$iter) {
682        *   $jscomp$arr.push(i);
683        * }
684        * for ($jscomp$var = 0; $jscomp$var < $jscomp$arr.length; $jscomp$var++) {
685        *   i = $jscomp$arr[$jscomp$var];
686        *   if (!(i in $jscomp$iter)) {
687        *     continue;
688        *   }
689        *   s;
690        * }
691        * </pre>
692        */
693       private void visitForIn() {
694         Node variable = currentStatement.removeFirstChild();
695         Node iterable = currentStatement.removeFirstChild();
696         Node body = currentStatement.removeFirstChild();
697     
698         String loopId = generatorCounter.get();
699         Node arrayName = IR.name(GENERATOR_FOR_IN_ARRAY + loopId);
700         Node varName = IR.name(GENERATOR_FOR_IN_VAR + loopId);
701         Node iterableName = IR.name(GENERATOR_FOR_IN_ITER + loopId);
702     
703         if (variable.isVar()) {
704           variable = variable.removeFirstChild();
705         }
706         body.addChildToFront(
707             IR.ifNode(
708                 IR.not(IR.in(variable.cloneTree(), iterableName.cloneTree())),
709                 IR.block(IR.continueNode())));
710         body.addChildToFront(
711             IR.var(variable.cloneTree(), IR.getelem(arrayName.cloneTree(), varName.cloneTree())));
712         hoistRoot.getParent().addChildAfter(IR.var(arrayName.cloneTree()), hoistRoot);
713         hoistRoot.getParent().addChildAfter(IR.var(varName.cloneTree()), hoistRoot);
714         hoistRoot.getParent().addChildAfter(IR.var(iterableName.cloneTree()), hoistRoot);
715     
716         Node arrayDef = IR.exprResult(IR.assign(arrayName.cloneTree(), IR.arraylit()));
717         Node iterDef = IR.exprResult(IR.assign(iterableName.cloneTree(), iterable));
718         Node newForIn =
719             IR.forIn(
720                 variable.cloneTree(),
721                 iterableName,
722                 IR.block(
723                     IR.exprResult(
724                         IR.call(IR.getprop(arrayName.cloneTree(), IR.string("push")), variable))));
725         Node newFor =
726             IR.forNode(
727                 IR.assign(varName.cloneTree(), IR.number(0)),
728                 IR.lt(varName.cloneTree(), IR.getprop(arrayName, IR.string("length"))),
729                 IR.inc(varName, true),
730                 body);
731     
732         enclosingBlock.addChildToBack(arrayDef);
733         enclosingBlock.addChildToBack(iterDef);
734         enclosingBlock.addChildToBack(newForIn);
735         originalGeneratorBody.addChildToFront(newFor);
736       }

738       /**
739        * Translates loops to a case statement followed by an if statement
740        * containing the loop body. The if statement finishes by
741        * jumping back to the initial case statement to enter the loop again.
742        * In the case of for and do loops, initialization and post loop statements are inserted
743        * before and after the if statement. Below is a sample translation for a while loop:
744        *
745        * <p>Sample translation:
746        * <pre>
747        * while (b) {
748        *   s;
749        * }
750        * </pre>
751        *
752        * <p>Is eventually rewritten to:
753        * <pre>
754        * case n:
755        *   if (b) {
756        *     s;
757        *     state = n;
758        *     break;
759        *   }
760        * </pre>
761        */
762       private void visitLoop(String label) {
763         Node initializer;
764         Node guard;
765         Node incr;
766         Node body;
767     
768         if (currentStatement.isWhile()) {
769           guard = currentStatement.removeFirstChild();
770           body = currentStatement.removeFirstChild();
771           initializer = IR.empty();
772           incr = IR.empty();
773         } else if (currentStatement.isFor()) {
774           initializer = currentStatement.removeFirstChild();
775           if (initializer.isAssign()) {
776             initializer = IR.exprResult(initializer);
777           }
778           guard = currentStatement.removeFirstChild();
779           incr = currentStatement.removeFirstChild();
780           body = currentStatement.removeFirstChild();
781         } else {
782           Preconditions.checkState(currentStatement.isDo());
783           initializer = IR.empty();
784           incr = IR.assign(IR.name(GENERATOR_DO_WHILE_INITIAL), IR.falseNode());
785     
786           body = currentStatement.removeFirstChild();
787           guard = currentStatement.removeFirstChild();
788         }
789     
790         Node condition, prestatement;
791     
792         if (guard.isBlock()) {
793           prestatement = guard.removeFirstChild();
794           condition = guard.removeFirstChild();
795         } else {
796           prestatement = IR.block();
797           condition = guard;
798         }
799     
800         int loopBeginState = generatorCaseCount++;
801         int continueState = loopBeginState;
802     
803         if (!incr.isEmpty()) {
804           continueState = generatorCaseCount++;
805           Node continueCase = makeGeneratorMarker(continueState);
806           body.addChildToBack(continueCase);
807           body.addChildToBack(incr.isBlock() ? incr : IR.exprResult(incr));
808         }
809     
810         currentLoopContext.add(0, new LoopContext(generatorCaseCount, continueState, label));
811     
812         Node beginCase = makeGeneratorMarker(loopBeginState);
813         Node conditionalBranch = IR.ifNode(condition.isEmpty() ? IR.trueNode() : condition, body);
814         Node setStateLoopStart = createStateUpdate(loopBeginState);
815         Node breakToStart = createSafeBreak();
816     
817         originalGeneratorBody.addChildToFront(conditionalBranch);
818         if (!prestatement.isEmpty()) {
819           originalGeneratorBody.addChildToFront(prestatement);
820         }
821         originalGeneratorBody.addChildToFront(beginCase);
822         if (!initializer.isEmpty()) {
823           originalGeneratorBody.addChildToFront(initializer);
824         }
825         body.addChildToBack(setStateLoopStart);
826         body.addChildToBack(breakToStart);
827       }

829       /**
830        * Hoists {@code var} statements into the closure containing the iterator
831        * to preserve their state across
832        * multiple calls to next().
833        */
834       private void visitVar() {
835         Node name = currentStatement.removeFirstChild();
836         while (name != null) {
837           if (name.hasChildren()) {
838             enclosingBlock.addChildToBack(IR.exprResult(IR.assign(name, name.removeFirstChild())));
839           }
840           hoistRoot.getParent().addChildAfter(IR.var(name.cloneTree()), hoistRoot);
841           name = currentStatement.removeFirstChild();
842         }
843       }

845       /**
846        * Translates {@code yield} to set the state so that execution resume at the next statement
847        * when the function is next called and then returns an iterator result with
848        * the desired value.
849        */
850       private void visitYieldExprResult() {
851         enclosingBlock.addChildToBack(createStateUpdate());
852         Node yield = currentStatement.getFirstChild();
853         Node value = yield.hasChildren() ? yield.removeFirstChild() : IR.name("undefined");
854         enclosingBlock.addChildToBack(IR.returnNode(createIteratorResult(value, false)));
855       }

857       /**
858        * Translates {@code return} statements to set the state to done before returning the
859        * desired value.
860        */
861       private void visitReturn() {
862         enclosingBlock.addChildToBack(createStateUpdate(-1));
863         enclosingBlock.addChildToBack(
864             IR.returnNode(
865                 createIteratorResult(
866                     currentStatement.hasChildren()
867                         ? currentStatement.removeFirstChild()
868                         : IR.name("undefined"),
869                     true)));
870       }

873         return IR.exprResult(IR.assign(IR.name(GENERATOR_STATE), IR.number(generatorCaseCount)));
874       }

877         return IR.exprResult(IR.assign(IR.name(GENERATOR_STATE), IR.number(state)));
878       }

881         return IR.objectlit(
882             IR.propdef(IR.stringKey("value"), value),
883             IR.propdef(IR.stringKey("done"), done ? IR.trueNode() : IR.falseNode()));
884       }

887         Node breakNode = IR.breakNode();
888         breakNode.setGeneratorSafe(true);

892       private static Node createFinallyJumpBlock(Node finallyName, int finallyStartState) {
893         int jumpPoint = generatorCaseCount++;
894         Node setReturnState = IR.exprResult(IR.assign(finallyName.cloneTree(), IR.number(jumpPoint)));
895         Node toFinally = createStateUpdate(finallyStartState);
896         Node returnPoint = makeGeneratorMarker(jumpPoint);
897         Node returnBlock = IR.block(setReturnState, toFinally, createSafeBreak());
898         returnBlock.addChildToBack(returnPoint);
899         return returnBlock;
900       }

939       /**
940        * Decomposes expressions with yields inside of them to equivalent
941        * sequence of expressions in which all non-statement yields are
942        * of the form:
943        *
944        * <pre>
945        *   var name = yield expr;
946        * </pre>
947        *
948        * <p>For example, change the following code:
949        * <pre>
950        *   return x || yield y;
951        * </pre>
952        * <p>Into:
953        * <pre>
954        *  var temp$$0;
955        *  if (temp$$0 = x); else temp$$0 = yield y;
956        *  return temp$$0;
957        * </pre>
958        *
959        * This uses the {@link ExpressionDecomposer} class
960        */
961       private final class DecomposeYields extends NodeTraversal.AbstractPreOrderCallback {
962     
963         private final AbstractCompiler compiler;
964         private final ExpressionDecomposer decomposer;
965     
966         DecomposeYields(AbstractCompiler compiler) {
967           this.compiler = compiler;
968           Set<String> consts = new HashSet<>();
969           decomposer =
970               new ExpressionDecomposer(
971                   compiler,
972                   compiler.getUniqueNameIdSupplier(),
973                   consts,
974                   Scope.createGlobalScope(new Node(Token.SCRIPT)));
975         }
976     
977         @Override
978         public boolean shouldTraverse(NodeTraversal t, Node n, Node parent) {
979           switch (n.getType()) {
980             case YIELD:
981               visitYieldExpression(n);
982               break;
983             case DO:
984             case FOR:
985             case WHILE:
986               visitLoop(n);
987               break;
988             case CASE:
989               if (controlCanExit(n.getFirstChild())) {
990                 compiler.report(
991                     JSError.make(
992                         n,
993                         Es6ToEs3Converter.CANNOT_CONVERT_YET,
994                         "Case statements that contain yields"));
995                 return false;
996               }
997               break;
998             default:
999               break;
1000           }
1001           return true;
1002         }
1003     
1004         private void visitYieldExpression(Node n) {
1005           if (n.getParent().isExprResult()) {
1006             return;
1007           }
1008           if (decomposer.canExposeExpression(n)
1009               != ExpressionDecomposer.DecompositionType.UNDECOMPOSABLE) {
1010             decomposer.exposeExpression(n);
1011             compiler.reportCodeChange();
1012           } else {
1013             compiler.report(
1014                 JSError.make(n, Es6ToEs3Converter.CANNOT_CONVERT, "Undecomposable expression"));
1015           }
1016         }
1017     
1018         private void visitLoop(Node n) {
1019           Node enclosingFunc = NodeUtil.getEnclosingFunction(n);
1020           if (enclosingFunc == null || !enclosingFunc.isGeneratorFunction() || NodeUtil.isForIn(n)) {
1021             return;
1022           }
1023           Node enclosingBlock = NodeUtil.getEnclosingBlock(n);
1024           Node guard = null;
1025           Node incr = null;
1026           switch (n.getType()) {
1027             case FOR:
1028               guard = n.getSecondChild();
1029               incr = guard.getNext();
1030               break;
1031             case WHILE:
1032               guard = n.getFirstChild();
1033               incr = IR.empty();
1034               break;
1035             case DO:
1036               guard = n.getLastChild();
1037               if (!guard.isEmpty()) {
1038                 Node firstEntry = IR.name(GENERATOR_DO_WHILE_INITIAL);
1039                 enclosingBlock.addChildToFront(IR.var(firstEntry.cloneTree(), IR.trueNode()));
1040                 guard = IR.or(firstEntry, n.getLastChild().detachFromParent());
1041                 n.addChildToBack(guard);
1042               }
1043               incr = IR.empty();
1044               break;
1045             default:
1046               break;
1047           }
1048           if (!controlCanExit(guard) && !controlCanExit(incr)) {
1049             return;
1050           }
1051           Node guardName = IR.name(GENERATOR_LOOP_GUARD + generatorCounter.get());
1052           if (!guard.isEmpty()) {
1053             Node container = new Node(Token.BLOCK);
1054             n.replaceChild(guard, container);
1055             container.addChildToFront(
1056                 IR.block(IR.exprResult(IR.assign(guardName.cloneTree(), guard.cloneTree()))));
1057             container.addChildToBack(guardName.cloneTree());
1058           }
1059           if (!incr.isEmpty()) {
1060             n.addChildBefore(IR.block(IR.exprResult(incr.detachFromParent())), n.getLastChild());
1061           }
1062           enclosingBlock.addChildToFront(IR.var(guardName));
1063           compiler.reportCodeChange();
1064         }
1065       }

1067       private static Node makeGeneratorMarker(int i) {
1068         Node n = IR.exprResult(IR.number(i));
1069         n.setGeneratorMarker(true);
1070         return n;
1071       }

1073       private static final class ControlExitsCheck implements NodeTraversal.Callback {
1074     
1075         int continueCatchers;
1076         int breakCatchers;
1077         int throwCatchers;
1078         List<String> labels = new ArrayList<>();
1079         boolean exited;
1080         boolean addJumps;
1081         private Node finallyName;
1082         private int finallyStartState;
1083     
1084         ControlExitsCheck(Node finallyName, int finallyStartState) {
1085           this.finallyName = finallyName;
1086           this.finallyStartState = finallyStartState;
1087           addJumps = true;
1088         }
1089     
1090         ControlExitsCheck() {
1091           addJumps = false;
1092         }
1093     
1094         @Override
1095         public boolean shouldTraverse(NodeTraversal nodeTraversal, Node n, Node parent) {
1096           switch (n.getType()) {
1097             case FUNCTION:
1098               return false;
1099             case LABEL:
1100               labels.add(0, n.getFirstChild().getString());
1101               break;
1102             case DO:
1103             case WHILE:
1104             case FOR:
1105               continueCatchers++;
1106               breakCatchers++;
1107               break;
1108             case SWITCH:
1109               breakCatchers++;
1110               break;
1111             case BLOCK:
1112               parent = n.getParent();
1113               if (parent != null
1114                   && parent.isTry()
1115                   && parent.getFirstChild() == n
1116                   && n.getNext().hasChildren()) {
1117                 throwCatchers++;
1118               }
1119               break;
1120             case BREAK:
1121               if (!n.isGeneratorSafe()
1122                   && ((breakCatchers == 0 && !n.hasChildren())
1123                       || (n.hasChildren() && !labels.contains(n.getFirstChild().getString())))) {
1124                 exited = true;
1125                 if (addJumps) {
1126                   parent.addChildBefore(createFinallyJumpBlock(finallyName, finallyStartState), n);
1127                 }
1128               }
1129               break;
1130             case CONTINUE:
1131               if (continueCatchers == 0
1132                   || (n.hasChildren() && !labels.contains(n.getFirstChild().getString()))) {
1133                 exited = true;
1134                 if (addJumps) {
1135                   parent.addChildBefore(createFinallyJumpBlock(finallyName, finallyStartState), n);
1136                 }
1137               }
1138               break;
1139             case THROW:
1140               if (throwCatchers == 0) {
1141                 exited = true;
1142                 if (addJumps && !n.isGeneratorSafe()) {
1143                   parent.addChildBefore(createFinallyJumpBlock(finallyName, finallyStartState), n);
1144                 }
1145               }
1146               break;
1147             case RETURN:
1148               exited = true;
1149               if (addJumps) {
1150                 parent.addChildBefore(createFinallyJumpBlock(finallyName, finallyStartState), n);
1151               }
1152               break;
1153             case YIELD:
1154               exited = true;
1155               break;
1156             default:
1157               break;
1158           }
1159           return true;
1160         }
1161     
1162         @Override
1163         public void visit(NodeTraversal t, Node n, Node parent) {
1164           switch (n.getType()) {
1165             case LABEL:
1166               labels.remove(0);
1167               break;
1168             case DO:
1169             case WHILE:
1170             case FOR:
1171               continueCatchers--;
1172               breakCatchers--;
1173               break;
1174             case SWITCH:
1175               breakCatchers--;
1176               break;
1177             case BLOCK:
1178               parent = n.getParent();
1179               if (parent != null
1180                   && parent.isTry()
1181                   && parent.getFirstChild() == n
1182                   && n.getNext().hasChildren()) {
1183                 throwCatchers--;
1184               }
1185               break;
1186             default:
1187               break;
1188           }
1189         }
1190     
1191         public boolean didExit() {
1192           return exited;
1193         }
1194       }
