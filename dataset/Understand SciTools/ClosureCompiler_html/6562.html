17     package com.google.javascript.jscomp;
18     
19     import com.google.common.base.Preconditions;
20     import com.google.javascript.jscomp.NodeTraversal.Callback;
21     import com.google.javascript.rhino.Node;
22     import com.google.javascript.rhino.Token;
23     
24     /**
25      * The goal with this pass is to reverse the simplifications done in the
26      * normalization pass that are not handled by other passes (such as
27      * CollapseVariableDeclarations) to avoid making the resulting code larger.
28      *
29      * Currently this pass only does two things:
30      *
31      * 1. Push statements into for-loop initializer. This:
32      *   var a = 0; for(;a<0;a++) {}
33      * becomes:
34      *   for(var a = 0;a<0;a++) {}
35      *
36      * 2. Fold assignments like x = x + 1 into x += 1
37      *
38      * @author johnlenz@google.com (johnlenz)
39      */
40     class Denormalize implements CompilerPass, Callback {
41     
42       private final AbstractCompiler compiler;
43     
44       Denormalize(AbstractCompiler compiler) {
45         this.compiler = compiler;
46       }
47     
48       @Override
49       public void process(Node externs, Node root) {
50         NodeTraversal.traverseEs6(compiler, root, this);
51       }
52     
53       @Override
54       public boolean shouldTraverse(NodeTraversal t, Node n, Node parent) {
55         return true;
56       }
57     
58       @Override
59       public void visit(NodeTraversal t, Node n, Node parent) {
60         maybeCollapseIntoForStatements(n, parent);
61         maybeCollapseAssignShorthand(n, parent);
62       }
63     
64       /**
65        * Collapse VARs and EXPR_RESULT node into FOR loop initializers where
66        * possible.
67        */
68       private void maybeCollapseIntoForStatements(Node n, Node parent) {
69         // Only SCRIPT, BLOCK, and LABELs can have FORs that can be collapsed into.
70         // LABELs are not supported here.
71         if (parent == null || !NodeUtil.isStatementBlock(parent)) {
72           return;
73         }
74     
75         // Is the current node something that can be in a for loop initializer?
76         if (!n.isExprResult() && !n.isVar()) {
77           return;
78         }
79     
80         // Is the next statement a valid FOR?
81         Node nextSibling = n.getNext();
82         if (nextSibling == null) {
83           return;
84         } else if (NodeUtil.isForIn(nextSibling)) {
85           Node forNode = nextSibling;
86           Node forVar = forNode.getFirstChild();
87           if (forVar.isName()
88               && n.isVar() && n.hasOneChild()) {
89             Node name = n.getFirstChild();
90             if (!name.hasChildren()
91                 && forVar.getString().equals(name.getString())) {
92               // OK, the names match, and the var declaration does not have an
93               // initializer. Move it into the loop.
94               parent.removeChild(n);
95               forNode.replaceChild(forVar, n);
96               compiler.reportCodeChange();
97             }
98           }
99         } else if (nextSibling.isFor()
100             && nextSibling.getFirstChild().isEmpty()) {
101     
102           // Does the current node contain an in operator?  If so, embedding
103           // the expression in a for loop can cause some JavaScript parsers (such
104           // as the PlayStation 3's browser based on Access's NetFront
105           // browser) to fail to parse the code.
106           // See bug 1778863 for details.
107           if (NodeUtil.containsType(n, Token.IN)) {
108             return;
109           }
110     
111           // Move the current node into the FOR loop initializer.
112           Node forNode = nextSibling;
113           Node oldInitializer = forNode.getFirstChild();
114           parent.removeChild(n);
115     
116           Node newInitializer;
117           if (n.isVar()) {
118             newInitializer = n;
119           } else {
120             // Extract the expression from EXPR_RESULT node.
121             Preconditions.checkState(n.hasOneChild(), n);
122             newInitializer = n.getFirstChild();
123             n.removeChild(newInitializer);
124           }
125     
126           forNode.replaceChild(oldInitializer, newInitializer);
127     
128           compiler.reportCodeChange();
129         }
130       }
131     
132       private void maybeCollapseAssignShorthand(Node n, Node parent) {
133         if (n.isAssign() && n.getFirstChild().isName()
134             && NodeUtil.hasCorrespondingAssignmentOp(n.getLastChild())
135             && n.getLastChild().getFirstChild().isName()) {
136           Node op = n.getLastChild();
137           Token assignOp = NodeUtil.getAssignOpFromOp(op);
138           if (n.getFirstChild().getString().equals(op.getFirstChild().getString())) {
139             op.setType(assignOp);
140             Node opDetached = op.detachFromParent();
141             opDetached.setJSDocInfo(n.getJSDocInfo());
142             parent.replaceChild(n, opDetached);
143             compiler.reportCodeChange();
144           }
145         }
146       }
147     }
