38     /**
39      * A graph represents all the referencing of global names in the program. In
40      * other words, it is a call and variable-name graph.
41      *
42      * <p>The NameReferenceGraph G for a program P is a directed graph G = (V, E)
43      * where:
44      *
45      * <P>V ({@link Name}) represents all global names in P and E = (v, v'), v and
46      * v' in V ({@link Reference} represents a reference use or definition from the
47      * name v to v' in P.
48      *
49      * <p>There are two core results we are trying to compute. The first being able
50      * to precisely identify the function body at any given call site with
51      * {@link #getReferencesAt(Node)}.
52      *
53      * <p>The second result come directly from the previous one. The directed edge
54      * provides us with dependency information. If A->B, B might be needed (in this
55      * module) if A is needed (in this module). The converse of the this result is
56      * more useful. B is not needed if A is not needed.
57      *
58      */
59     class NameReferenceGraph extends
60         LinkedDirectedGraph<NameReferenceGraph.Name, NameReferenceGraph.Reference>
61         implements DefinitionProvider {
62     
63       // This is the key result of the name graph. Given a node in the AST, this map
64       // will give us the Reference edges. For example a CALL node will map to a
65       // list of possible call edge destinations.
66       private final Multimap<Node, Name>
67           referenceMap = HashMultimap.create();
68     
69       // Given a qualified name, provides the Name object.
70       private Map<String, Name> nameMap = new HashMap<>();
71     
72       // The following are some implicit nodes of the graph.
73     
74       // If we have a call site that we absolutely have no idea what variable it
75       // it calls or reference, we'd point it to UNKNOWN.
76       final Name unknown;
77     
78       // Represents the "main" global block as well as externs.
79       final Name main;
80     
81       // The implicit "window" object.
82       final Name window;
83     
84       final AbstractCompiler compiler;
85     
86       public NameReferenceGraph(AbstractCompiler compiler) {
87         super(true, true);
88         this.compiler = compiler;
89     
90         // Initialize builtins.
91         unknown = new Name("{UNKNOWN}", true);
92         unknown.isAliased = true;
93         unknown.type = compiler.getTypeIRegistry().getNativeType(JSTypeNative.NO_TYPE);
94         this.createNode(unknown);
95     
96         main = new Name("{Global Main}", true);
97         this.createNode(main);
98     
99         window = new Name("window", true);
100         this.createNode(window);
101       }
102     
103       public Name defineNameIfNotExists(String name, boolean isExtern) {
104         Name symbol = null;
105         if (nameMap.containsKey(name)) {
106           // This is a re-declaration.
107           symbol = nameMap.get(name);
108         } else {
109           symbol = new Name(name, isExtern);
110           nameMap.put(name, symbol);
111           createNode(symbol);
112         }
113         return symbol;
114       }
115     
116       /**
117        * Retrieves a list of all possible Names that this site is referring to.
118        */
119       public List<Name> getReferencesAt(Node site) {
120         Preconditions.checkArgument(
121             site.isGetProp() || site.isName());
122         List<Name> result = new ArrayList<>();
123         result.addAll(referenceMap.get(site));
124         return result;
125       }
126     
127       @Override
128       public Collection<Definition> getDefinitionsReferencedAt(Node useSite) {
129         List<Name> nameRefs = getReferencesAt(useSite);
130         if (nameRefs.isEmpty()) {
131           return null;
132         }
133     
134         List<Definition> result = new ArrayList<>();
135         for (Name nameRef : nameRefs) {
136           List<Definition> decls = nameRef.getDeclarations();
137           if (!decls.isEmpty()) {
138             result.addAll(decls);
139           }
140         }
141     
142         if (!result.isEmpty()) {
143           return result;
144         } else {
145           return null;
146         }
147       }
148     
149       public Name getSymbol(String name) {
150         return nameMap.get(name);
151       }
152     
153       @Override
154       public GraphNode<Name, Reference> createNode(Name value) {
155         nameMap.put(value.qName, value);
156         return super.createNode(value);
157       }
158     
159       @Override
160       public void connect(Name src, Reference ref, Name dest) {
161         super.connect(src, ref, dest);
162         referenceMap.put(ref.site, dest);
163       }

165       /**
166        * Represents function or variable names that can be referenced globally.
167        */
168       class Name {
169         // Full name
170         private final String qName;
171     
172         private JSType type;
173     
174         // A list (re)declarations
175         private List<Definition> declarations = new LinkedList<>();
176     
177         final boolean isExtern;
178     
179         private boolean isExported = false;
180     
181         private boolean isAliased = false;
182     
183         // Function invocations that use ".call" and ".apply" syntax may prevent
184         // several of the possible optimizations.  We keep track of all functions
185         // invoked in this way so those passes can exclude them.
186         // Ex:
187         // some_func.call(some_obj, 1, 2 , 3);
188         // The name graph does not currently recognize this as a call to some_func.
189         // This Set is meant to keep track of such occurrence until the name graph
190         // becomes aware of those cases.
191         private boolean exposedToCallOrApply = false;
192     
193         public Name(String qName, boolean isExtern) {
194           this.qName = qName;
195           this.isExtern = isExtern;
196           int lastDot = qName.lastIndexOf('.');
197           String name = (lastDot == -1) ? qName : qName.substring(lastDot + 1);
198           this.isExported = compiler.getCodingConvention().isExported(name);
199           this.type = compiler.getTypeIRegistry().getNativeType(JSTypeNative.UNKNOWN_TYPE);
200         }
201     
202         public JSType getType() {
203           return type;
204         }
205     
206         public void setType(JSType type) {
207           this.type = type;
208         }
209     
210         public List<Definition> getDeclarations() {
211           return declarations;
212         }
213     
214         public void addAssignmentDeclaration(Node node) {
215           declarations.add(new AssignmentDefinition(node, isExtern));
216         }
217     
218         public void addFunctionDeclaration(Node node) {
219           declarations.add(new NamedFunctionDefinition(node, isExtern));
220         }
221     
222         public void markExported() {
223           this.isExported = true;
224         }
225     
226         public boolean isExported() {
227           return isExported;
228         }
229     
230         /**
231          * @return {@code} True if this name has been dereferenced. Removing from
232          *     the program or the module is no longer safe unless further analysis
233          *     can prove otherwise.
234          */
235         public boolean isAliased() {
236           return isAliased;
237         }
238     
239         public void setAliased(boolean isAliased) {
240           this.isAliased = isAliased;
241         }
242     
243         public String getQualifiedName() {
244           return qName;
245         }
246     
247         /**
248          * @return The short property name of this object if it is a property, else
249          *     {@code null}.
250          */
251         public String getPropertyName() {
252           int lastIndexOfDot = qName.lastIndexOf('.');
253           if (lastIndexOfDot == -1) {
254             return null;
255           } else {
256             return qName.substring(lastIndexOfDot + 1);
257           }
258         }
259     
260         public boolean exposedToCallOrApply() {
261           return exposedToCallOrApply;
262         }
263     
264         public void markExposedToCallOrApply() {
265           exposedToCallOrApply = true;
266         }
267     
268         @Override
269         public String toString() {
270           return qName + " : " + type;
271         }
272     
273         @Override
274         public int hashCode() {
275           return qName.hashCode();
276         }
277       }

279       /**
280        * A reference site for a function or a variable reference. It can be a
281        * reference use or an assignment to that name.
282        */
283       static class Reference {
284         // The node that references the name.
285         public final Node site;
286     
287         public Reference(Node site) {
288           this.site = site;
289         }
290     
291         /**
292          * Get accessor for retrieving the actual node corresponding to the
293          * reference.
294          *
295          * @return node representing the access/reference/call site
296          */
297         public Node getSite() {
298           return site;
299         }
300       }
