18

20     import com.google.javascript.rhino.JSDocInfo;
21     import com.google.javascript.rhino.JSDocInfo.Visibility;
22     import com.google.javascript.rhino.Node;
23     import com.google.javascript.rhino.TypeI;

25     import java.util.ArrayList;
26     import java.util.HashSet;
27     import java.util.List;

30     /**
31      * This pass looks for properties that are never read.
32      * These can be properties created using "this", or static properties of
33      * constructors or interfaces. Explicitly ignored is the possibility that
34      * these properties may be indirectly referenced using "for-in" or
35      * "Object.keys".
36      *
37      * This class is based on RemoveUnusedClassProperties, some effort should
38      * be made to extract the common pieces.
39      *
40      * @author johnlenz@google.com (John Lenz)
41      */
42     class CheckUnusedPrivateProperties
43         implements HotSwapCompilerPass, NodeTraversal.Callback {
44     
45       static final DiagnosticType UNUSED_PRIVATE_PROPERTY =
46           DiagnosticType.disabled(
47               "JSC_UNUSED_PRIVATE_PROPERTY", "Private property {0} is never read");
48     
49       private final AbstractCompiler compiler;
50       private Set<String> used = new HashSet<>();
51       private List<Node> candidates = new ArrayList<>();
52     
53       CheckUnusedPrivateProperties(AbstractCompiler compiler) {
54         this.compiler = compiler;
55       }
56     
57       @Override
58       public void process(Node externs, Node root) {
59         NodeTraversal.traverseEs6(compiler, root, this);
60       }
61     
62       @Override
63       public void hotSwapScript(Node scriptRoot, Node originalRoot) {
64         NodeTraversal.traverseEs6(compiler, scriptRoot, this);
65       }
66     
67       private void reportUnused(NodeTraversal t) {
68         for (Node n : candidates) {
69           String propName = getPropName(n);
70           if (!used.contains(propName)) {
71             t.report(n, UNUSED_PRIVATE_PROPERTY, propName);
72           }
73         }
74       }
75     
76       private String getPropName(Node n) {
77         switch (n.getType()) {
78           case GETPROP:
79             return n.getLastChild().getString();
80           case MEMBER_FUNCTION_DEF:
81             return n.getString();
82         }
83         throw new RuntimeException("Unexpected node type: " + n);
84       }
85     
86       @Override
87       public boolean shouldTraverse(NodeTraversal t, Node n, Node parent) {
88         if (n.isScript()) {
89           used.clear();
90           candidates.clear();
91         }
92         return true;
93       }
94     
95       @Override
96       public void visit(NodeTraversal t, Node n, Node parent) {
97          switch (n.getType()) {
98            case SCRIPT: {
99              // exiting the script, report any privates not used in the file.
100              reportUnused(t);
101              break;
102            }
103     
104            case GETPROP: {
105              String propName = n.getLastChild().getString();
106              if (compiler.getCodingConvention().isExported(propName)
107                  || isPinningPropertyUse(n)
108                  || !isCandidatePropertyDefinition(n)) {
109                used.add(propName);
110              } else {
111                // Only consider "private" properties.
112                if (isCheckablePrivatePropDecl(n)) {
113                  candidates.add(n);
114                }
115              }
116              break;
117            }
118     
119            case MEMBER_FUNCTION_DEF: {
120              // Only consider "private" methods.
121              if (isCheckablePrivatePropDecl(n)) {
122                candidates.add(n);
123              }
124              break;
125            }
126     
127            case OBJECTLIT: {
128              // Assume any object literal definition might be a reflection on the
129              // class property.
130              for (Node c : n.children()) {
131                used.add(c.getString());
132              }
133              break;
134            }
135     
136           case CALL:
137             // Look for properties referenced through a property rename function.
138             Node target = n.getFirstChild();
139             if (n.hasMoreThanOneChild()
140                 && compiler
141                     .getCodingConvention()
142                     .isPropertyRenameFunction(target.getOriginalQualifiedName())) {
143                Node propName = target.getNext();
144                if (propName.isString()) {
145                  used.add(propName.getString());
146                }
147              }
148              break;
149          }
150       }
151     
152       private boolean isPrivatePropDecl(Node n) {
153         // TODO(johnlenz): add support private by convention property definitions without JSDoc.
154         // TODO(johnlenz): add support for checking protected properties in final classes
155         // TODO(johnlenz): add support for checking "package" properties when checking an entire
156         // library.
157         JSDocInfo info = NodeUtil.getBestJSDocInfo(n);
158         return (info != null && info.getVisibility() == Visibility.PRIVATE);
159       }
160     
161       private boolean isCheckablePrivatePropDecl(Node n) {
162         // TODO(tbreisacher): Look for uses of the typedef/interface in type expressions; warn if there
163         // are no uses.
164         JSDocInfo info = NodeUtil.getBestJSDocInfo(n);
165         return isPrivatePropDecl(n) && !info.hasTypedefType() && !info.isInterface();
166       }
167     
168       private boolean isCandidatePropertyDefinition(Node n) {
169         Preconditions.checkState(n.isGetProp(), n);
170         Node target = n.getFirstChild();
171         return target.isThis()
172             || (isConstructor(target))
173             || (target.isGetProp()
174                 && target.getLastChild().getString().equals("prototype"));
175       }
176     
177       private boolean isConstructor(Node n) {
178         // If type checking is enabled (not just a per-file lint check),
179         // we can check constructor properties too. But it isn't required.
180         TypeI type = n.getTypeI();
181         return type != null && (type.isConstructor() || type.isInterface());
182       }
183     
184       /**
185        * @return Whether the property is used in a way that prevents its removal.
186        */
187       private static boolean isPinningPropertyUse(Node n) {
188         // Rather than looking for cases that are uses, we assume all references are
189         // pinning uses unless they are:
190         //  - a simple assignment (x.a = 1)
191         //  - a compound assignment or increment (x++, x += 1) whose result is
192         //    otherwise unused
193     
194         Node parent = n.getParent();
195         if (n == parent.getFirstChild()) {
196           if (parent.isExprResult()) {
197             // A stub declaration "this.x;" isn't a pinning use.
198             return false;
199           } else if (parent.isAssign()) {
200             // A simple assignment doesn't pin the property.
201             return false;
202           } else if (NodeUtil.isAssignmentOp(parent)
203                 || parent.isInc() || parent.isDec()) {
204             // In general, compound assignments are both reads and writes, but
205             // if the property is never otherwise read we can consider it simply
206             // a write.
207             // However if the assign expression is used as part of a larger
208             // expression, we must consider it a read. For example:
209             //    x = (y.a += 1);
210             return NodeUtil.isExpressionResultUsed(parent);
211           }
212         }
213         return true;
214       }
215     }
