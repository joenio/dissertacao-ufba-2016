26     public final class CheckRequiresAndProvidesSortedTest extends Es6CompilerTestCase {

32       public void testNoWarning_require() {
33         testSame("goog.require('a.b');\ngoog.require('a.c')");
34         testSame(
35             LINE_JOINER.join(
36                 "goog.require('namespace');",
37                 "goog.require('namespace.ExampleClass');",
38                 "goog.require('namespace.ExampleClass.ExampleInnerClass');"));
39         testSame(
40             LINE_JOINER.join(
41                 "goog.require('namespace.Example');",
42                 "goog.require('namespace.example');"));
43       }
44     
45       public void testNoWarning_provide() {
46         testSame("goog.provide('a.b');\ngoog.provide('a.c')");
47         testSame(
48             LINE_JOINER.join(
49                 "goog.provide('namespace');",
50                 "goog.provide('namespace.ExampleClass');",
51                 "goog.provide('namespace.ExampleClass.ExampleInnerClass');"));
52         testSame(
53             LINE_JOINER.join(
54                 "goog.provide('namespace.Example');",
55                 "goog.provide('namespace.example');"));
56       }
57     
58       public void testWarning_require() {
59         testWarning("goog.require('a.c');\ngoog.require('a.b')", REQUIRES_NOT_SORTED);
60         testWarning("goog.require('a.c');\ngoog.require('a')", REQUIRES_NOT_SORTED);
61       }
62     
63       public void testWarning_provide() {
64         testWarning("goog.provide('a.c');\ngoog.provide('a.b')", PROVIDES_NOT_SORTED);
65         testWarning("goog.provide('a.c');\ngoog.provide('a')", PROVIDES_NOT_SORTED);
66       }
67     
68       public void testWarning_requiresFirst() {
69         testWarning("goog.require('a');\ngoog.provide('b')", PROVIDES_AFTER_REQUIRES);
70       }

81       /**
82        * If a goog.module uses the "var x = goog.require('x')" form, don't warn.
83        */
84       public void testGoogModuleWithShorthand() {
85         testSame(
86             LINE_JOINER.join(
87                 "goog.module('m');",
88                 "",
89                 "goog.require('a.c');",
90                 "var d = goog.require('a.b.d');",
91                 "goog.require('a.b.c');",
92                 "",
93                 "alert(1);"));
94     
95         testSame(
96             LINE_JOINER.join(
97                 "goog.module('m');",
98                 "",
99                 "var d = goog.require('a.b.d');",
100                 "var c = goog.require('a.c');",
101                 "",
102                 "alert(1);"));
103       }
104     
105       public void testGoogModuleNoShorthand() {
106         testWarning(
107             LINE_JOINER.join(
108                 "goog.module('m');",
109                 "",
110                 "goog.require('a.c');",
111                 "goog.require('a.b.d');",
112                 "goog.require('a.b.c');",

118       public void testGoogModuleWithDestructuring() {
119         testSameEs6(
120             LINE_JOINER.join(
121                 "goog.module('m');",
122                 "",
123                 "goog.require('z');",
124                 "goog.require('a');",
125                 "var {someFunction, anotherFunction} = goog.require('example.utils');",
126                 "var {A_CONST, ANOTHER_CONST} = goog.require('example.constants');",
127                 "",
128                 "alert(1);"));
129       }
