28     /**
29      * A peephole optimization that minimizes conditional expressions
30      * according to De Morgan's laws.
31      * Also rewrites conditional statements as expressions by replacing them
32      * with HOOKs and short-circuit binary operators.
33      *
34      * Based on PeepholeSubstituteAlternateSyntax
35      */
36     class PeepholeMinimizeConditions
37       extends AbstractPeepholeOptimization {
38     
39       private static final int AND_PRECEDENCE = NodeUtil.precedence(Token.AND);
40     
41       private final boolean late;
42       private final boolean useTypes;
43     
44       /**
45        * @param late When late is false, this mean we are currently running before
46        * most of the other optimizations. In this case we would avoid optimizations
47        * that would make the code harder to analyze (such as using string splitting,
48        * merging statements with commas, etc). When this is true, we would
49        * do anything to minimize for size.
50        */
51       PeepholeMinimizeConditions(boolean late, boolean useTypes) {
52         this.late = late;
53         this.useTypes = useTypes;
54       }
55     
56       /**
57        * Tries to apply our various peephole minimizations on the passed in node.
58        */
59       @Override
60       @SuppressWarnings("fallthrough")
61       public Node optimizeSubtree(Node node) {
62         switch(node.getType()) {
63           case THROW:
64           case RETURN: {
65             Node result = tryRemoveRedundantExit(node);
66             if (result != node) {
67               return result;
68             }
69             return tryReplaceExitWithBreak(node);
70           }
71     
72           // TODO(johnlenz): Maybe remove redundant BREAK and CONTINUE. Overlaps
73           // with MinimizeExitPoints.
74     
75           case NOT:
76             tryMinimizeCondition(node.getFirstChild());
77             return tryMinimizeNot(node);
78     
79           case IF:
80             performCoercionSubstitutions(node.getFirstChild());
81             performConditionSubstitutions(node.getFirstChild());
82             return tryMinimizeIf(node);
83     
84           case EXPR_RESULT:
85             performConditionSubstitutions(node.getFirstChild());
86             return tryMinimizeExprResult(node);
87     
88           case HOOK:
89             performCoercionSubstitutions(node.getFirstChild());
90             performConditionSubstitutions(node.getFirstChild());
91             return tryMinimizeHook(node);
92     
93           case WHILE:
94           case DO:
95             tryMinimizeCondition(NodeUtil.getConditionExpression(node));
96             return node;
97     
98           case FOR:
99             if (!NodeUtil.isForIn(node)) {
100               tryJoinForCondition(node);
101               tryMinimizeCondition(NodeUtil.getConditionExpression(node));
102             }
103             return node;
104     
105           case BLOCK:
106             return tryReplaceIf(node);
107     
108           default:
109             return node; //Nothing changed
110         }
111       }

113       private void tryJoinForCondition(Node n) {
114         if (!late) {
115           return;
116         }
117     
118         Node block = n.getLastChild();
119         Node maybeIf = block.getFirstChild();
120         if (maybeIf != null && maybeIf.isIf()) {
121           Node thenBlock = maybeIf.getSecondChild();
122           Node maybeBreak = thenBlock.getFirstChild();
123           if (maybeBreak != null && maybeBreak.isBreak()
124               && !maybeBreak.hasChildren()) {
125     
126             // Preserve the IF ELSE expression is there is one.
127             if (maybeIf.getChildCount() == 3) {
128               block.replaceChild(maybeIf,
129                   maybeIf.getLastChild().detachFromParent());
130             } else {
131               NodeUtil.redeclareVarsInsideBranch(thenBlock);
132               block.removeFirstChild();
133             }
134     
135             Node ifCondition = maybeIf.removeFirstChild();
136             Node fixedIfCondition = IR.not(ifCondition)
137                 .srcref(ifCondition);
138     
139             // OK, join the IF expression with the FOR expression
140             Node forCondition = NodeUtil.getConditionExpression(n);
141             if (forCondition.isEmpty()) {
142               n.replaceChild(forCondition, fixedIfCondition);
143             } else {
144               Node replacement = new Node(Token.AND);
145               n.replaceChild(forCondition, replacement);
146               replacement.addChildToBack(forCondition);
147               replacement.addChildToBack(fixedIfCondition);
148             }
149     
150             reportCodeChange();
151           }
152         }
153       }

155       /**
156        * Use "return x?1:2;" in place of "if(x)return 1;return 2;"
157        */
158       private Node tryReplaceIf(Node n) {
159     
160         for (Node child = n.getFirstChild();
161              child != null; child = child.getNext()){
162           if (child.isIf()){
163             Node cond = child.getFirstChild();
164             Node thenBranch = cond.getNext();
165             Node elseBranch = thenBranch.getNext();
166             Node nextNode = child.getNext();
167     
168             if (nextNode != null && elseBranch == null
169                 && isReturnBlock(thenBranch)
170                 && nextNode.isIf()) {
171               Node nextCond = nextNode.getFirstChild();
172               Node nextThen = nextCond.getNext();
173               Node nextElse = nextThen.getNext();
174               if (thenBranch.isEquivalentToTyped(nextThen)) {
175                 // Transform
176                 //   if (x) return 1; if (y) return 1;
177                 // to
178                 //   if (x||y) return 1;
179                 child.detachFromParent();
180                 child.detachChildren();
181                 Node newCond = new Node(Token.OR, cond);
182                 nextNode.replaceChild(nextCond, newCond);
183                 newCond.addChildToBack(nextCond);
184                 reportCodeChange();
185               } else if (nextElse != null
186                   && thenBranch.isEquivalentToTyped(nextElse)) {
187                 // Transform
188                 //   if (x) return 1; if (y) foo() else return 1;
189                 // to
190                 //   if (!x&&y) foo() else return 1;
191                 child.detachFromParent();
192                 child.detachChildren();
193                 Node newCond = new Node(Token.AND,
194                     IR.not(cond).srcref(cond));
195                 nextNode.replaceChild(nextCond, newCond);
196                 newCond.addChildToBack(nextCond);
197                 reportCodeChange();
198               }
199             } else if (nextNode != null && elseBranch == null &&
200                 isReturnBlock(thenBranch) && isReturnExpression(nextNode)) {
201               Node thenExpr = null;
202               // if(x)return; return 1 -> return x?void 0:1
203               if (isReturnExpressBlock(thenBranch)) {
204                 thenExpr = getBlockReturnExpression(thenBranch);
205                 thenExpr.detachFromParent();
206               } else {
207                 thenExpr = NodeUtil.newUndefinedNode(child);
208               }
209     
210               Node elseExpr = nextNode.getFirstChild();
211     
212               cond.detachFromParent();
213               elseExpr.detachFromParent();
214     
215               Node returnNode = IR.returnNode(
216                                     IR.hook(cond, thenExpr, elseExpr)
217                                         .srcref(child));
218               n.replaceChild(child, returnNode);
219               n.removeChild(nextNode);
220               reportCodeChange();
221             } else if (elseBranch != null && statementMustExitParent(thenBranch)) {
222               child.removeChild(elseBranch);
223               n.addChildAfter(elseBranch, child);
224               reportCodeChange();
225             }
226           }
227         }
228         return n;
229       }

231       private static boolean statementMustExitParent(Node n) {
232         switch (n.getType()) {
233           case THROW:
234           case RETURN:
235             return true;
236           case BLOCK:
237             if (n.hasChildren()) {
238               Node child = n.getLastChild();
239               return statementMustExitParent(child);
240             }
241             return false;
242           // TODO(johnlenz): handle TRY/FINALLY
243           case FUNCTION:
244           default:
245             return false;
246         }
247       }

249       /**
250        * Replace duplicate exits in control structures.  If the node following
251        * the exit node expression has the same effect as exit node, the node can
252        * be replaced or removed.
253        * For example:
254        *   "while (a) {return f()} return f();" ==> "while (a) {break} return f();"
255        *   "while (a) {throw 'ow'} throw 'ow';" ==> "while (a) {break} throw 'ow';"
256        *
257        * @param n An follow control exit expression (a THROW or RETURN node)
258        * @return The replacement for n, or the original if no change was made.
259        */
260       private Node tryReplaceExitWithBreak(Node n) {
261         Node result = n.getFirstChild();
262     
263         // Find the enclosing control structure, if any, that a "break" would exit
264         // from.
265         Node breakTarget = n;
266         for (; !ControlFlowAnalysis.isBreakTarget(breakTarget, null /* no label */);
267             breakTarget = breakTarget.getParent()) {
268           if (breakTarget.isFunction() || breakTarget.isScript()) {
269             // No break target.
270             return n;
271           }
272         }
273     
274         Node follow = ControlFlowAnalysis.computeFollowNode(breakTarget);
275     
276         // Skip pass all the finally blocks because both the break and return will
277         // also trigger all the finally blocks. However, the order of execution is
278         // slightly changed. Consider:
279         //
280         // return a() -> finally { b() } -> return a()
281         //
282         // which would call a() first. However, changing the first return to a
283         // break will result in calling b().
284     
285         Node prefinallyFollows = follow;
286         follow = skipFinallyNodes(follow);
287     
288         if (prefinallyFollows != follow) {
289           // There were finally clauses
290           if (!isPure(result)) {
291             // Can't defer the exit
292             return n;
293           }
294         }
295     
296         if (follow == null && (n.isThrow() || result != null)) {
297           // Can't complete remove a throw here or a return with a result.
298           return n;
299         }
300     
301         // When follow is null, this mean the follow of a break target is the
302         // end of a function. This means a break is same as return.
303         if (follow == null || areMatchingExits(n, follow)) {
304           Node replacement = IR.breakNode();
305           n.getParent().replaceChild(n, replacement);
306           this.reportCodeChange();
307           return replacement;
308         }
309     
310         return n;
311       }

313       /**
314        * Remove duplicate exits.  If the node following the exit node expression
315        * has the same effect as exit node, the node can be removed.
316        * For example:
317        *   "if (a) {return f()} return f();" ==> "if (a) {} return f();"
318        *   "if (a) {throw 'ow'} throw 'ow';" ==> "if (a) {} throw 'ow';"
319        *
320        * @param n An follow control exit expression (a THROW or RETURN node)
321        * @return The replacement for n, or the original if no change was made.
322        */
323       private Node tryRemoveRedundantExit(Node n) {
324         Node exitExpr = n.getFirstChild();
325     
326         Node follow = ControlFlowAnalysis.computeFollowNode(n);
327     
328         // Skip pass all the finally blocks because both the fall through and return
329         // will also trigger all the finally blocks.
330         Node prefinallyFollows = follow;
331         follow = skipFinallyNodes(follow);
332         if (prefinallyFollows != follow) {
333           // There were finally clauses
334           if (!isPure(exitExpr)) {
335             // Can't replace the return
336             return n;
337           }
338         }
339     
340         if (follow == null && (n.isThrow() || exitExpr != null)) {
341           // Can't complete remove a throw here or a return with a result.
342           return n;
343         }
344     
345         // When follow is null, this mean the follow of a break target is the
346         // end of a function. This means a break is same as return.
347         if (follow == null || areMatchingExits(n, follow)) {
348           n.detachFromParent();
349           reportCodeChange();
350           return null;
351         }
352     
353         return n;
354       }

356       /**
357        * @return Whether the expression does not produces and can not be affected
358        * by side-effects.
359        */
360       boolean isPure(Node n) {
361         return n == null
362             || (!NodeUtil.canBeSideEffected(n)
363                 && !mayHaveSideEffects(n));
364       }

366       /**
367        * @return n or the node following any following finally nodes.
368        */
369       static Node skipFinallyNodes(Node n) {
370         while (n != null && NodeUtil.isTryFinallyNode(n.getParent(), n)) {
371           n = ControlFlowAnalysis.computeFollowNode(n);
372         }
373         return n;
374       }

376       /**
377        * Check whether one exit can be replaced with another. Verify:
378        * 1) They are identical expressions
379        * 2) If an exception is possible that the statements, the original
380        * and the potential replacement are in the same exception handler.
381        */
382       boolean areMatchingExits(Node nodeThis, Node nodeThat) {
383         return nodeThis.isEquivalentTo(nodeThat)
384             && (!isExceptionPossible(nodeThis)
385                 || getExceptionHandler(nodeThis) == getExceptionHandler(nodeThat));
386       }
387     
388       static boolean isExceptionPossible(Node n) {
389         // TODO(johnlenz): maybe use ControlFlowAnalysis.mayThrowException?
390         Preconditions.checkState(n.isReturn() || n.isThrow(), n);
391         return n.isThrow()
392             || (n.hasChildren()
393                 && !NodeUtil.isLiteralValue(n.getLastChild(), true));
394       }
395     
396       static Node getExceptionHandler(Node n) {
397         return ControlFlowAnalysis.getExceptionHandler(n);
398       }

400       /**
401        * Try to minimize NOT nodes such as !(x==y).
402        *
403        * Returns the replacement for n or the original if no change was made
404        */
405       private Node tryMinimizeNot(Node n) {
406         Preconditions.checkArgument(n.isNot());
407         Node parent = n.getParent();
408     
409         Node notChild = n.getFirstChild();
410         // negative operator of the current one : == -> != for instance.
411         Token complementOperator;
412         switch (notChild.getType()) {
413           case EQ:
414             complementOperator = Token.NE;
415             break;
416           case NE:
417             complementOperator = Token.EQ;
418             break;
419           case SHEQ:
420             complementOperator = Token.SHNE;
421             break;
422           case SHNE:
423             complementOperator = Token.SHEQ;
424             break;
425           // GT, GE, LT, LE are not handled in this because !(x<NaN) != x>=NaN.
426           default:
427             return n;
428         }
429         Node newOperator = n.removeFirstChild();
430         newOperator.setType(complementOperator);
431         parent.replaceChild(n, newOperator);
432         reportCodeChange();
433         return newOperator;
434       }

437       /**
438        * Try to remove leading NOTs from EXPR_RESULTS.
439        *
440        * Returns the replacement for n or the original if no replacement was
441        * necessary.
442        */
443       private Node tryMinimizeExprResult(Node n) {
444         MinimizedCondition minCond =
445             MinimizedCondition.fromConditionNode(n.getFirstChild());
446         MinimizedCondition.MeasuredNode mNode =
447             minCond.getMinimized(MinimizationStyle.ALLOW_LEADING_NOT);
448         Node placeholder = minCond.getPlaceholder();
449         if (mNode.getNode().isNot()) {
450           // Remove the leading NOT in the EXPR_RESULT.
451           n.replaceChild(placeholder, mNode.getNode().removeFirstChild());
452           reportCodeChange();
453         } else {
454           replaceNode(placeholder, mNode);
455         }
456         return n;
457       }

459       /**
460        * Try flipping HOOKs that have negated conditions.
461        *
462        * Returns the replacement for n or the original if no replacement was
463        * necessary.
464        */
465       private Node tryMinimizeHook(Node n) {
466         MinimizedCondition minCond =
467           MinimizedCondition.fromConditionNode(n.getFirstChild());
468         MinimizedCondition.MeasuredNode mNode =
469             minCond.getMinimized(MinimizationStyle.ALLOW_LEADING_NOT);
470         Node placeholder = minCond.getPlaceholder();
471         if (mNode.getNode().isNot()) {
472           // Swap the HOOK
473           Node thenBranch = n.getSecondChild();
474           n.replaceChild(placeholder, mNode.getNode().removeFirstChild());
475           n.removeChild(thenBranch);
476           n.addChildToBack(thenBranch);
477           reportCodeChange();
478         } else {
479           replaceNode(placeholder, mNode);
480         }
481         return n;
482       }

484       /**
485        * Try turning IF nodes into smaller HOOKs
486        *
487        * Returns the replacement for n or the original if no replacement was
488        * necessary.
489        */
490       private Node tryMinimizeIf(Node n) {
491     
492         Node parent = n.getParent();
493     
494         Node originalCond = n.getFirstChild();
495     
496         /* If the condition is a literal, we'll let other
497          * optimizations try to remove useless code.
498          */
499         if (NodeUtil.isLiteralValue(originalCond, true)) {
500           return n;
501         }
502     
503         Node thenBranch = originalCond.getNext();
504         Node elseBranch = thenBranch.getNext();
505     
506         MinimizedCondition minCond =
507             MinimizedCondition.fromConditionNode(originalCond);
508         originalCond = null;  // originalCond was mutated and should not be used.
509     
510         Node placeholder = minCond.getPlaceholder();
511     
512         MinimizedCondition.MeasuredNode unnegatedCond;
513         MinimizedCondition.MeasuredNode shortCond;
514         // Compute two minimized representations. The first representation counts
515         // a leading NOT node, and the second ignores a leading NOT node.
516         // If we can fold the if statement into a HOOK or boolean operation,
517         // then the NOT node does not matter, and we prefer the second condition.
518         // If we cannot fold the if statement, then we prefer the first condition.
519         unnegatedCond = minCond.getMinimized(MinimizationStyle.PREFER_UNNEGATED);
520         shortCond = minCond.getMinimized(MinimizationStyle.ALLOW_LEADING_NOT);
521     
522         if (elseBranch == null) {
523           if (isFoldableExpressBlock(thenBranch)) {
524             Node expr = getBlockExpression(thenBranch);
525             if (!late && isPropertyAssignmentInExpression(expr)) {
526               // Keep opportunities for CollapseProperties such as
527               // a.longIdentifier || a.longIdentifier = ... -> var a = ...;
528               // until CollapseProperties has been run.
529               replaceNode(placeholder, unnegatedCond);
530               return n;
531             }
532     
533             if (shortCond.getNode().isNot()) {
534               // if(!x)bar(); -> x||bar();
535               Node or = IR.or(
536                   shortCond.getNode().removeFirstChild(),
537                   expr.removeFirstChild()).srcref(n);
538               Node newExpr = NodeUtil.newExpr(or);
539               parent.replaceChild(n, newExpr);
540               reportCodeChange();
541     
542               return newExpr;
543             }
544             // True, but removed for performance reasons.
545             // Preconditions.checkState(shortCond.isEquivalentTo(unnegatedCond));
546     
547             // if(x)foo(); -> x&&foo();
548             if (isLowerPrecedence(shortCond.getNode(), AND_PRECEDENCE) &&
549                 isLowerPrecedence(expr.getFirstChild(),
550                     AND_PRECEDENCE)) {
551               // One additional set of parentheses is worth the change even if
552               // there is no immediate code size win. However, two extra pair of
553               // {}, we would have to think twice. (unless we know for sure the
554               // we can further optimize its parent.
555               replaceNode(placeholder, shortCond);
556               return n;
557             }
558     
559             n.removeChild(placeholder);
560             Node and = IR.and(shortCond.getNode(), expr.removeFirstChild()).srcref(n);
561             Node newExpr = NodeUtil.newExpr(and);
562             parent.replaceChild(n, newExpr);
563             reportCodeChange();
564     
565             return newExpr;
566           } else {
567     
568             // Try to combine two IF-ELSE
569             if (NodeUtil.isStatementBlock(thenBranch) &&
570                 thenBranch.hasOneChild()) {
571               Node innerIf = thenBranch.getFirstChild();
572     
573               if (innerIf.isIf()) {
574                 Node innerCond = innerIf.getFirstChild();
575                 Node innerThenBranch = innerCond.getNext();
576                 Node innerElseBranch = innerThenBranch.getNext();
577     
578                 if (innerElseBranch == null &&
579                      !(isLowerPrecedence(unnegatedCond.getNode(), AND_PRECEDENCE) &&
580                        isLowerPrecedence(innerCond, AND_PRECEDENCE))) {
581                   n.detachChildren();
582                   n.addChildToBack(
583                       IR.and(
584                           unnegatedCond.getNode(),
585                           innerCond.detachFromParent())
586                           .srcref(placeholder));
587                   n.addChildrenToBack(innerThenBranch.detachFromParent());
588                   reportCodeChange();
589                   // Not worth trying to fold the current IF-ELSE into && because
590                   // the inner IF-ELSE wasn't able to be folded into && anyways.
591                   return n;
592                 }
593               }
594             }
595           }
596           replaceNode(placeholder, unnegatedCond);
597           return n;
598         }
599     
600         /* TODO(dcc) This modifies the siblings of n, which is undesirable for a
601          * peephole optimization. This should probably get moved to another pass.
602          */
603         tryRemoveRepeatedStatements(n);
604     
605         // if(!x)foo();else bar(); -> if(x)bar();else foo();
606         // An additional set of curly braces isn't worth it.
607         if (shortCond.getNode().isNot() && !consumesDanglingElse(elseBranch)) {
608           n.replaceChild(placeholder, shortCond.getNode().removeFirstChild());
609           n.removeChild(thenBranch);
610           n.addChildToBack(thenBranch);
611           reportCodeChange();
612           return n;
613         }
614     
615         // if(x)return 1;else return 2; -> return x?1:2;
616         if (isReturnExpressBlock(thenBranch) && isReturnExpressBlock(elseBranch)) {
617           Node thenExpr = getBlockReturnExpression(thenBranch);
618           Node elseExpr = getBlockReturnExpression(elseBranch);
619           n.removeChild(placeholder);
620           thenExpr.detachFromParent();
621           elseExpr.detachFromParent();
622     
623           // note - we ignore any cases with "return;", technically this
624           // can be converted to "return undefined;" or some variant, but
625           // that does not help code size.
626           Node returnNode = IR.returnNode(
627                                 IR.hook(shortCond.getNode(), thenExpr, elseExpr)
628                                     .srcref(n));
629           parent.replaceChild(n, returnNode);
630           reportCodeChange();
631           return returnNode;
632         }
633     
634         boolean thenBranchIsExpressionBlock = isFoldableExpressBlock(thenBranch);
635         boolean elseBranchIsExpressionBlock = isFoldableExpressBlock(elseBranch);
636     
637         if (thenBranchIsExpressionBlock && elseBranchIsExpressionBlock) {
638           Node thenOp = getBlockExpression(thenBranch).getFirstChild();
639           Node elseOp = getBlockExpression(elseBranch).getFirstChild();
640           if (thenOp.getType() == elseOp.getType()) {
641             // if(x)a=1;else a=2; -> a=x?1:2;
642             if (NodeUtil.isAssignmentOp(thenOp)) {
643               Node lhs = thenOp.getFirstChild();
644               if (areNodesEqualForInlining(lhs, elseOp.getFirstChild()) &&
645                   // if LHS has side effects, don't proceed [since the optimization
646                   // evaluates LHS before cond]
647                   // NOTE - there are some circumstances where we can
648                   // proceed even if there are side effects...
649                   !mayEffectMutableState(lhs) &&
650                   (!mayHaveSideEffects(unnegatedCond.getNode()) ||
651                       (thenOp.isAssign() && thenOp.getFirstChild().isName()))) {
652     
653                 n.removeChild(placeholder);
654                 Node assignName = thenOp.removeFirstChild();
655                 Node thenExpr = thenOp.removeFirstChild();
656                 Node elseExpr = elseOp.getLastChild();
657                 elseOp.removeChild(elseExpr);
658     
659                 Node hookNode = IR.hook(shortCond.getNode(), thenExpr, elseExpr)
660                     .srcref(n);
661                 Node assign = new Node(thenOp.getType(), assignName, hookNode)
662                                   .srcref(thenOp);
663                 Node expr = NodeUtil.newExpr(assign);
664                 parent.replaceChild(n, expr);
665                 reportCodeChange();
666     
667                 return expr;
668               }
669             }
670           }
671           // if(x)foo();else bar(); -> x?foo():bar()
672           n.removeChild(placeholder);
673           thenOp.detachFromParent();
674           elseOp.detachFromParent();
675           Node expr = IR.exprResult(
676               IR.hook(shortCond.getNode(), thenOp, elseOp).srcref(n));
677           parent.replaceChild(n, expr);
678           reportCodeChange();
679           return expr;
680         }
681     
682         boolean thenBranchIsVar = isVarBlock(thenBranch);
683         boolean elseBranchIsVar = isVarBlock(elseBranch);
684     
685         // if(x)var y=1;else y=2  ->  var y=x?1:2
686         if (thenBranchIsVar && elseBranchIsExpressionBlock &&
687             getBlockExpression(elseBranch).getFirstChild().isAssign()) {
688     
689           Node var = getBlockVar(thenBranch);
690           Node elseAssign = getBlockExpression(elseBranch).getFirstChild();
691     
692           Node name1 = var.getFirstChild();
693           Node maybeName2 = elseAssign.getFirstChild();
694     
695           if (name1.hasChildren()
696               && maybeName2.isName()
697               && name1.getString().equals(maybeName2.getString())) {
698             Node thenExpr = name1.removeChildren();
699             Node elseExpr = elseAssign.getLastChild().detachFromParent();
700             placeholder.detachFromParent();
701             Node hookNode = IR.hook(shortCond.getNode(), thenExpr, elseExpr)
702                                 .srcref(n);
703             var.detachFromParent();
704             name1.addChildrenToBack(hookNode);
705             parent.replaceChild(n, var);
706             reportCodeChange();
707             return var;
708           }
709     
710         // if(x)y=1;else var y=2  ->  var y=x?1:2
711         } else if (elseBranchIsVar && thenBranchIsExpressionBlock &&
712             getBlockExpression(thenBranch).getFirstChild().isAssign()) {
713     
714           Node var = getBlockVar(elseBranch);
715           Node thenAssign = getBlockExpression(thenBranch).getFirstChild();
716     
717           Node maybeName1 = thenAssign.getFirstChild();
718           Node name2 = var.getFirstChild();
719     
720           if (name2.hasChildren()
721               && maybeName1.isName()
722               && maybeName1.getString().equals(name2.getString())) {
723             Node thenExpr = thenAssign.getLastChild().detachFromParent();
724             Node elseExpr = name2.removeChildren();
725             placeholder.detachFromParent();
726             Node hookNode = IR.hook(shortCond.getNode(), thenExpr, elseExpr)
727                                 .srcref(n);
728             var.detachFromParent();
729             name2.addChildrenToBack(hookNode);
730             parent.replaceChild(n, var);
731             reportCodeChange();
732     
733             return var;
734           }
735         }
736     
737         replaceNode(placeholder, unnegatedCond);
738         return n;
739       }

741       /**
742        * Try to remove duplicate statements from IF blocks. For example:
743        *
744        * if (a) {
745        *   x = 1;
746        *   return true;
747        * } else {
748        *   x = 2;
749        *   return true;
750        * }
751        *
752        * becomes:
753        *
754        * if (a) {
755        *   x = 1;
756        * } else {
757        *   x = 2;
758        * }
759        * return true;
760        *
761        * @param n The IF node to examine.
762        */
763       private void tryRemoveRepeatedStatements(Node n) {
764         Preconditions.checkState(n.isIf(), n);
765     
766         Node parent = n.getParent();
767         if (!NodeUtil.isStatementBlock(parent)) {
768           // If the immediate parent is something like a label, we
769           // can't move the statement, so bail.
770           return;
771         }
772     
773         Node cond = n.getFirstChild();
774         Node trueBranch = cond.getNext();
775         Node falseBranch = trueBranch.getNext();
776         Preconditions.checkNotNull(trueBranch);
777         Preconditions.checkNotNull(falseBranch);
778     
779         while (true) {
780           Node lastTrue = trueBranch.getLastChild();
781           Node lastFalse = falseBranch.getLastChild();
782           if (lastTrue == null || lastFalse == null
783               || !areNodesEqualForInlining(lastTrue, lastFalse)) {
784             break;
785           }
786           lastTrue.detachFromParent();
787           lastFalse.detachFromParent();
788           parent.addChildAfter(lastTrue, n);
789           reportCodeChange();
790         }
791       }

793       /**
794        * @return Whether the node is a block with a single statement that is
795        *     an expression.
796        */
797       private static boolean isFoldableExpressBlock(Node n) {
798         if (n.isBlock()) {
799           if (n.hasOneChild()) {
800             Node maybeExpr = n.getFirstChild();
801             if (maybeExpr.isExprResult()) {
802               // IE has a bug where event handlers behave differently when
803               // their return value is used vs. when their return value is in
804               // an EXPR_RESULT. It's pretty freaking weird. See:
805               // http://code.google.com/p/closure-compiler/issues/detail?id=291
806               // We try to detect this case, and not fold EXPR_RESULTs
807               // into other expressions.
808               if (maybeExpr.getFirstChild().isCall()) {
809                 Node calledFn = maybeExpr.getFirstFirstChild();
810     
811                 // We only have to worry about methods with an implicit 'this'
812                 // param, or this doesn't happen.
813                 if (calledFn.isGetElem()) {
814                   return false;
815                 } else if (calledFn.isGetProp() &&
816                            calledFn.getLastChild().getString().startsWith("on")) {
817                   return false;
818                 }
819               }
820     
821               return true;
822             }
823             return false;
824           }
825         }
826     
827         return false;
828       }

913       /**
914        * Does a statement consume a 'dangling else'? A statement consumes
915        * a 'dangling else' if an 'else' token following the statement
916        * would be considered by the parser to be part of the statement.
917        */
918       private static boolean consumesDanglingElse(Node n) {
919         while (true) {
920           switch (n.getType()) {
921             case IF:
922               if (n.getChildCount() < 3) {
923                 return true;
924               }
925               // This IF node has no else clause.
926               n = n.getLastChild();
927               continue;
928             case BLOCK:
929               if (n.getChildCount() != 1) {
930                 return false;
931               }
932               // This BLOCK has no curly braces.
933               n = n.getLastChild();
934               continue;
935             case WITH:
936             case WHILE:
937             case FOR:
938               n = n.getLastChild();
939               continue;
940             default:
941               return false;
942           }
943         }
944       }

970       /**
971        * Try to minimize condition expression, as there are additional
972        * assumptions that can be made when it is known that the final result
973        * is a boolean.
974        *
975        * @return The replacement for n, or the original if no change was made.
976        */
977       private Node tryMinimizeCondition(Node n) {
978         n = performCoercionSubstitutions(n);
979         n = performConditionSubstitutions(n);
980         MinimizedCondition minCond = MinimizedCondition.fromConditionNode(n);
981         return replaceNode(
982             minCond.getPlaceholder(),
983             minCond.getMinimized(MinimizationStyle.PREFER_UNNEGATED));
984       }

986       /**
987        * Replaces 'foo ==/!=/===/!== null' with 'foo' or '!foo'. Should only be used for expressions
988        * used in conditions where the final result is coerced to a boolean.
989        *
990        * @return The replacement for n, or the original if no change was made.
991        */
992       private Node performCoercionSubstitutions(Node n) {
993         if (!useTypes) {
994           return n;
995         }
996     
997         Node parent = n.getParent();
998     
999         switch (n.getType()) {
1000           case OR:
1001           case AND:
1002             performCoercionSubstitutions(n.getFirstChild());
1003             performCoercionSubstitutions(n.getLastChild());
1004             break;
1005     
1006           case EQ:
1007           case NE:
1008           case SHEQ:
1009           case SHNE:
1010             Node left = n.getFirstChild();
1011             Node right = n.getLastChild();
1012             boolean leftIsNull = NodeUtil.isNullOrUndefined(left);
1013             boolean rightIsNull = NodeUtil.isNullOrUndefined(right);
1014             boolean leftIsObject = isObjectType(left);
1015             boolean rightIsObject = isObjectType(right);
1016             if (leftIsObject && rightIsNull || rightIsObject && leftIsNull) {
1017               n.detachChildren();
1018               Node objExpression = leftIsObject ? left : right;
1019               Node replacement = n.getType() == Token.EQ || n.getType() == Token.SHEQ
1020                   ? IR.not(objExpression)
1021                   : objExpression;
1022               parent.replaceChild(n, replacement);
1023               reportCodeChange();
1024               return replacement;
1025             }
1026             break;
1027         }
1028         return n;
1029       }

1031       private static boolean isObjectType(Node n) {
1032         JSType jsType = n.getJSType();
1033         if (jsType == null) {
1034           return false;
1035         }
1036         return !jsType.isUnknownType() && jsType.isObject();
1037       }

1040         Node parent = lhs.getParent();

1042         if (rhs.isChanged()) {
1043           reportCodeChange();
1044         }
1045         return rhs.getNode();

1048       /**
1049        * Try to minimize the given condition by applying local substitutions.
1050        *
1051        * The following types of transformations are performed:
1052        *   x || true        --> true
1053        *   x && true        --> x
1054        *   x ? false : true --> !x
1055        *   x ? true : y     --> x || y
1056        *
1057        *   Returns the replacement for n, or the original if no change was made
1058        */
1059       private Node performConditionSubstitutions(Node n) {
1060         Node parent = n.getParent();
1061     
1062         switch (n.getType()) {
1063           case OR:
1064           case AND: {
1065             Node left = n.getFirstChild();
1066             Node right = n.getLastChild();
1067     
1068             // Because the expression is in a boolean context minimize
1069             // the children, this can't be done in the general case.
1070             left = performConditionSubstitutions(left);
1071             right = performConditionSubstitutions(right);
1072     
1073             // Remove useless conditionals
1074             // Handle the following cases:
1075             //   x || false --> x
1076             //   x && true --> x
1077             // This works regardless of whether x has side effects.
1078             //
1079             // If x does not have side effects:
1080             //   x || true  --> true
1081             //   x && false  --> false
1082             //
1083             // If x may have side effects:
1084             //   x || true  --> x,true
1085             //   x && false  --> x,false
1086             //
1087             // In the last two cases, code size may increase slightly (adding
1088             // some parens because the comma operator has a low precedence) but
1089             // the new AST is easier for other passes to handle.
1090             TernaryValue rightVal = NodeUtil.getPureBooleanValue(right);
1091             if (NodeUtil.getPureBooleanValue(right) != TernaryValue.UNKNOWN) {
1092               Token type = n.getType();
1093               Node replacement = null;
1094               boolean rval = rightVal.toBoolean(true);
1095     
1096               // (x || FALSE) => x
1097               // (x && TRUE) => x
1098               if (type == Token.OR && !rval ||
1099                   type == Token.AND && rval) {
1100                 replacement = left;
1101               } else if (!mayHaveSideEffects(left)) {
1102                 replacement = right;
1103               } else {
1104                 // expr_with_sideeffects || true  =>  expr_with_sideeffects, true
1105                 // expr_with_sideeffects && false  =>  expr_with_sideeffects, false
1106                 n.detachChildren();
1107                 replacement = IR.comma(left, right);
1108               }
1109     
1110               if (replacement != null) {
1111                 n.detachChildren();
1112                 parent.replaceChild(n, replacement);
1113                 reportCodeChange();
1114                 return replacement;
1115               }
1116             }
1117             return n;
1118           }
1119     
1120           case HOOK: {
1121             Node condition = n.getFirstChild();
1122             Node trueNode = n.getSecondChild();
1123             Node falseNode = n.getLastChild();
1124     
1125             // Because the expression is in a boolean context minimize
1126             // the result children, this can't be done in the general case.
1127             // The condition is handled in the general case in #optimizeSubtree
1128             trueNode = performConditionSubstitutions(trueNode);
1129             falseNode = performConditionSubstitutions(falseNode);
1130     
1131             // Handle four cases:
1132             //   x ? true : false --> x
1133             //   x ? false : true --> !x
1134             //   x ? true : y     --> x || y
1135             //   x ? y : false    --> x && y
1136             Node replacement = null;
1137             TernaryValue trueNodeVal = NodeUtil.getPureBooleanValue(trueNode);
1138             TernaryValue falseNodeVal = NodeUtil.getPureBooleanValue(falseNode);
1139             if (trueNodeVal == TernaryValue.TRUE
1140                 && falseNodeVal == TernaryValue.FALSE) {
1141               // Remove useless conditionals, keep the condition
1142               condition.detachFromParent();
1143               replacement = condition;
1144             } else if (trueNodeVal == TernaryValue.FALSE
1145                 && falseNodeVal == TernaryValue.TRUE) {
1146               // Remove useless conditionals, keep the condition
1147               condition.detachFromParent();
1148               replacement = IR.not(condition);
1149             } else if (trueNodeVal == TernaryValue.TRUE) {
1150               // Remove useless true case.
1151               n.detachChildren();
1152               replacement = IR.or(condition, falseNode);
1153             } else if (falseNodeVal == TernaryValue.FALSE) {
1154               // Remove useless false case
1155               n.detachChildren();
1156               replacement = IR.and(condition, trueNode);
1157             }
1158     
1159             if (replacement != null) {
1160               parent.replaceChild(n, replacement);
1161               n = replacement;
1162               reportCodeChange();
1163             }
1164     
1165             return n;
1166           }
1167     
1168           default:
1169             // while(true) --> while(1)
1170             TernaryValue nVal = NodeUtil.getPureBooleanValue(n);
1171             if (nVal != TernaryValue.UNKNOWN) {
1172               boolean result = nVal.toBoolean(true);
1173               int equivalentResult = result ? 1 : 0;
1174               return maybeReplaceChildWithNumber(n, parent, equivalentResult);
1175             }
1176             // We can't do anything else currently.
1177             return n;
1178         }
1179       }

1181       /**
1182        * Replaces a node with a number node if the new number node is not equivalent
1183        * to the current node.
1184        *
1185        * Returns the replacement for n if it was replaced, otherwise returns n.
1186        */
1187       private Node maybeReplaceChildWithNumber(Node n, Node parent, int num) {
1188         Node newNode = IR.number(num);
1189         if (!newNode.isEquivalentTo(n)) {
1190           parent.replaceChild(n, newNode);
1191           reportCodeChange();
1192     
1193           return newNode;
1194         }
1195     
1196         return n;
1197       }
