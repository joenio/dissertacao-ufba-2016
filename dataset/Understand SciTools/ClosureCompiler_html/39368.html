56     /**
57      * Represents JavaScript value types.<p>
58      *
59      * Types are split into two separate families: value types and object types.
60      *
61      * A special {@link UnknownType} exists to represent a wildcard type on which
62      * no information can be gathered. In particular, it can assign to everyone,
63      * is a subtype of everyone (and everyone is a subtype of it).<p>
64      *
65      * If you remove the {@link UnknownType}, the set of types in the type system
66      * forms a lattice with the {@link #isSubtype} relation defining the partial
67      * order of types. All types are united at the top of the lattice by the
68      * {@link AllType} and at the bottom by the {@link NoType}.<p>
69      *
70      */

261       /**
262        * Returns true iff {@code this} can be a {@code struct}.
263        * UnionType overrides the method, assume {@code this} is not a union here.
264        */
265       public boolean isStruct() {
266         if (isObject()) {
267           ObjectType objType = toObjectType();
268           ObjectType iproto = objType.getImplicitPrototype();
269           // For the case when a @struct constructor is assigned to a function's
270           // prototype property
271           if (iproto != null && iproto.isStruct()) {
272             return true;
273           }
274           FunctionType ctor = objType.getConstructor();
275           // This test is true for object literals
276           if (ctor == null) {
277             JSDocInfo info = objType.getJSDocInfo();
278             return info != null && info.makesStructs();
279           } else {
280             return ctor.makesStructs();
281           }
282         }
283         return false;
284       }

550        * Whether this type is meaningfully different from {@code that} type for
551        * the purposes of data flow analysis.
552        *
553        * This is a trickier check than pure equality, because it has to properly
554        * handle unknown types. See {@code EquivalenceMethod} for more info.
555        *
556        * @see <a href="http://www.youtube.com/watch?v=_RpSv3HjpEw">Unknown unknowns</a>

559         return !checkEquivalenceHelper(that, EquivalenceMethod.DATA_FLOW);

562       /**
563        * An equivalence visitor.
564        */
565       boolean checkEquivalenceHelper(
566           final JSType that, EquivalenceMethod eqMethod) {
567         return checkEquivalenceHelper(that, eqMethod, EqCache.create());
568       }
569     
570       boolean checkEquivalenceHelper(final JSType that, EquivalenceMethod eqMethod,
571           EqCache eqCache) {
572         if (this == that) {
573           return true;
574         }
575     
576         boolean thisUnknown = isUnknownType();
577         boolean thatUnknown = that.isUnknownType();
578         if (thisUnknown || thatUnknown) {
579           if (eqMethod == EquivalenceMethod.INVARIANT) {
580             // If we're checking for invariance, the unknown type is invariant
581             // with everyone.
582             return true;
583           } else if (eqMethod == EquivalenceMethod.DATA_FLOW) {
584             // If we're checking data flow, then two types are the same if they're
585             // both unknown.
586             return thisUnknown && thatUnknown;
587           } else if (thisUnknown && thatUnknown &&
588               (isNominalType() ^ that.isNominalType())) {
589             // If they're both unknown, but one is a nominal type and the other
590             // is not, then we should fail out immediately. This ensures that
591             // we won't unbox the unknowns further down.
592             return false;
593           }
594         }
595     
596         if (isUnionType() && that.isUnionType()) {
597           return toMaybeUnionType().checkUnionEquivalenceHelper(
598               that.toMaybeUnionType(), eqMethod, eqCache);
599         }
600     
601         if (isFunctionType() && that.isFunctionType()) {
602           return toMaybeFunctionType().checkFunctionEquivalenceHelper(
603               that.toMaybeFunctionType(), eqMethod, eqCache);
604         }
605     
606         if (!getTemplateTypeMap().checkEquivalenceHelper(
607             that.getTemplateTypeMap(), eqMethod, eqCache, SubtypingMode.NORMAL)) {
608           return false;
609         }
610     
611         if (eqCache.isStructuralTyping() && this.isStructuralType() && that.isStructuralType()) {
612           return toMaybeObjectType()
613               .checkStructuralEquivalenceHelper(that.toMaybeObjectType(), eqMethod, eqCache);
614         }
615     
616         if (isNominalType() && that.isNominalType()) {
617           // TODO(johnlenz): is this valid across scopes?
618           return getConcreteNominalTypeName(this.toObjectType())
619               .equals(getConcreteNominalTypeName(that.toObjectType()));
620         }
621     
622         if (isTemplateType() && that.isTemplateType()) {
623           // TemplateType are they same only if they are object identical,
624           // which we check at the start of this function.
625           return false;
626         }
627     
628         // Unbox other proxies.
629         if (this instanceof ProxyObjectType) {
630           return ((ProxyObjectType) this)
631               .getReferencedTypeInternal().checkEquivalenceHelper(
632                   that, eqMethod, eqCache);
633         }
634     
635         if (that instanceof ProxyObjectType) {
636           return checkEquivalenceHelper(
637               ((ProxyObjectType) that).getReferencedTypeInternal(),
638               eqMethod, eqCache);
639         }
640     
641         // Relies on the fact that for the base {@link JSType}, only one
642         // instance of each sub-type will ever be created in a given registry, so
643         // there is no need to verify members. If the object pointers are not
644         // identical, then the type member must be different.
645         return false;
646       }

648       // Named types may be proxies of concrete types.
649       private String getConcreteNominalTypeName(ObjectType objType) {
650         if (objType instanceof ProxyObjectType) {
651           ObjectType internal = ((ProxyObjectType) objType)
652               .getReferencedObjTypeInternal();
653           if (internal != null && internal.isNominalType()) {
654             return getConcreteNominalTypeName(internal);
655           }
656         }
657         return objType.getReferenceName();
658       }

827        * Compares {@code this} and {@code that}.
828        * @return <ul>
829        * <li>{@link TernaryValue#TRUE} if the comparison of values of
830        *   {@code this} type and {@code that} always succeed (such as
831        *   {@code undefined} compared to {@code null})</li>
832        * <li>{@link TernaryValue#FALSE} if the comparison of values of
833        *   {@code this} type and {@code that} always fails (such as
834        *   {@code undefined} compared to {@code number})</li>
835        * <li>{@link TernaryValue#UNKNOWN} if the comparison can succeed or
836        *   fail depending on the concrete values</li>
837        * </ul>
838        */

843       TernaryValue testForEqualityHelper(JSType aType, JSType bType) {
844         if (bType.isAllType() || bType.isUnknownType() ||
845             bType.isNoResolvedType() ||
846             aType.isAllType() || aType.isUnknownType() ||
847             aType.isNoResolvedType()) {
848           return UNKNOWN;
849         }
850     
851         boolean aIsEmpty = aType.isEmptyType();
852         boolean bIsEmpty = bType.isEmptyType();
853         if (aIsEmpty || bIsEmpty) {
854           if (aIsEmpty && bIsEmpty) {
855             return TernaryValue.TRUE;
856           } else {
857             return UNKNOWN;
858           }
859         }
860     
861         if (aType.isFunctionType() || bType.isFunctionType()) {
862           JSType otherType = aType.isFunctionType() ? bType : aType;
863           // In theory, functions are comparable to anything except
864           // null/undefined. For example, on FF3:
865           // function() {} == 'function () {\n}'
866           // In practice, how a function serializes to a string is
867           // implementation-dependent, so it does not really make sense to test
868           // for equality with a string.
869           JSType meet = otherType.getGreatestSubtype(
870               getNativeType(JSTypeNative.OBJECT_TYPE));
871           if (meet.isNoType() || meet.isNoObjectType()) {
872             return TernaryValue.FALSE;
873           } else {
874             return TernaryValue.UNKNOWN;
875           }
876         }
877         if (bType.isEnumElementType() || bType.isUnionType()) {
878           return bType.testForEquality(aType);
879         }
880         return null;
881       }

981       /**
982        * A generic implementation meant to be used as a helper for common
983        * getGreatestSubtype implementations.
984        */
985       static JSType getGreatestSubtype(JSType thisType, JSType thatType) {
986         if (thisType.isFunctionType() && thatType.isFunctionType()) {
987           // The FunctionType sub-lattice is not well-defined. i.e., the
988           // proposition
989           // A < B => sup(A, B) == B
990           // does not hold because of unknown parameters and return types.
991           // See the comment in supAndInfHelper for more info on this.
992           return thisType.toMaybeFunctionType().supAndInfHelper(
993               thatType.toMaybeFunctionType(), false);
994         } else if (thisType.isEquivalentTo(thatType)) {
995           return thisType;
996         } else if (thisType.isUnknownType() || thatType.isUnknownType()) {
997           // The greatest subtype with any unknown type is the universal
998           // unknown type, unless the two types are equal.
999           return thisType.isEquivalentTo(thatType) ? thisType :
1000               thisType.getNativeType(JSTypeNative.UNKNOWN_TYPE);
1001         } else if (thisType.isUnionType()) {
1002           return thisType.toMaybeUnionType().meet(thatType);
1003         } else if (thatType.isUnionType()) {
1004           return thatType.toMaybeUnionType().meet(thisType);
1005         } else if (thisType.isTemplatizedType()) {
1006           return thisType.toMaybeTemplatizedType().getGreatestSubtypeHelper(
1007               thatType);
1008         }  else if (thatType.isTemplatizedType()) {
1009           return thatType.toMaybeTemplatizedType().getGreatestSubtypeHelper(
1010               thisType);
1011         } else if (thisType.isSubtype(thatType)) {
1012           return filterNoResolvedType(thisType);
1013         } else if (thatType.isSubtype(thisType)) {
1014           return filterNoResolvedType(thatType);
1015         } else if (thisType.isRecordType()) {
1016           return thisType.toMaybeRecordType().getGreatestSubtypeHelper(thatType);
1017         } else if (thatType.isRecordType()) {
1018           return thatType.toMaybeRecordType().getGreatestSubtypeHelper(thisType);
1019         }
1020     
1021         if (thisType.isEnumElementType()) {
1022           JSType inf = thisType.toMaybeEnumElementType().meet(thatType);
1023           if (inf != null) {
1024             return inf;
1025           }
1026         } else if (thatType.isEnumElementType()) {
1027           JSType inf = thatType.toMaybeEnumElementType().meet(thisType);
1028           if (inf != null) {
1029             return inf;
1030           }
1031         }
1032     
1033         if (thisType.isObject() && thatType.isObject()) {
1034           return thisType.getNativeType(JSTypeNative.NO_OBJECT_TYPE);
1035         }
1036         return thisType.getNativeType(JSTypeNative.NO_TYPE);
1037       }

1039       /**
1040        * When computing infima, we may get a situation like
1041        * inf(Type1, Type2)
1042        * where both types are unresolved, so they're technically
1043        * subtypes of one another.
1044        *
1045        * If this happens, filter them down to NoResolvedType.
1046        */
1047       static JSType filterNoResolvedType(JSType type) {
1048         if (type.isNoResolvedType()) {
1049           // inf(UnresolvedType1, UnresolvedType2) needs to resolve
1050           // to the base unresolved type, so that the relation is symmetric.
1051           return type.getNativeType(JSTypeNative.NO_RESOLVED_TYPE);
1052         } else if (type.isUnionType()) {
1053           UnionType unionType = type.toMaybeUnionType();
1054           boolean needsFiltering = false;
1055           for (JSType alt : unionType.getAlternates()) {
1056             if (alt.isNoResolvedType()) {
1057               needsFiltering = true;
1058               break;
1059             }
1060           }
1061     
1062           if (needsFiltering) {
1063             UnionTypeBuilder builder = new UnionTypeBuilder(type.registry);
1064             builder.addAlternate(type.getNativeType(JSTypeNative.NO_RESOLVED_TYPE));
1065             for (JSType alt : unionType.getAlternates()) {
1066               if (!alt.isNoResolvedType()) {
1067                 builder.addAlternate(alt);
1068               }
1069             }
1070             return builder.build();
1071           }
1072         }
1073         return type;
1074       }

1251        * Note this function also returns true if this type structurally
1252        * matches the protocol define by that type (if that type is an
1253        * interface function type)
1254        *
1255        * Subtyping rules:
1256        * <ul>
1257        * <li>(unknown) &mdash; every type is a subtype of the Unknown type.</li>
1258        * <li>(no) &mdash; the No type is a subtype of every type.</li>
1259        * <li>(no-object) &mdash; the NoObject type is a subtype of every object
1260        * type (i.e. subtypes of the Object type).</li>

1263        * union type's constituents are a subtype of U. Formally<br>
1264        * <code>(T<sub>1</sub>, &hellip;, T<sub>n</sub>) &lt;: U</code> if and only
1265        * <code>T<sub>k</sub> &lt;: U</code> for all <code>k &isin; 1..n</code>.</li>
1266        * <li>(union-r) &mdash; A type U is a subtype of a union type if it is a
1267        * subtype of one of the union type's constituents. Formally<br>
1268        * <code>U &lt;: (T<sub>1</sub>, &hellip;, T<sub>n</sub>)</code> if and only
1269        * if <code>U &lt;: T<sub>k</sub></code> for some index {@code k}.</li>
1270        * <li>(objects) &mdash; an Object <code>O<sub>1</sub></code> is a subtype
1271        * of an object <code>O<sub>2</sub></code> if it has more properties
1272        * than <code>O<sub>2</sub></code> and all common properties are
1273        * pairwise subtypes.</li>
1274        * </ul>
1275        *
1276        * @return <code>this &lt;: that</code>
1277        */

1304       /**
1305        * if implicitImplCache is null, there is no structural interface matching
1306        */
1307       static boolean isSubtypeHelper(JSType thisType, JSType thatType,
1308           ImplCache implicitImplCache, SubtypingMode subtypingMode) {
1309         Preconditions.checkNotNull(thisType);
1310         // unknown
1311         if (thatType.isUnknownType()) {
1312           return true;
1313         }
1314         // all type
1315         if (thatType.isAllType()) {
1316           return true;
1317         }
1318         // equality
1319         if (thisType.isEquivalentTo(thatType, implicitImplCache.isStructuralTyping())) {
1320           return true;
1321         }
1322         // unions
1323         if (thatType.isUnionType()) {
1324           UnionType union = thatType.toMaybeUnionType();
1325           for (JSType element : union.alternatesWithoutStucturalTyping) {
1326             if (thisType.isSubtype(element, implicitImplCache, subtypingMode)) {
1327               return true;
1328             }
1329           }
1330           return false;
1331         }
1332     
1333         // TemplateTypeMaps. This check only returns false if the TemplateTypeMaps
1334         // are not equivalent.
1335         TemplateTypeMap thisTypeParams = thisType.getTemplateTypeMap();
1336         TemplateTypeMap thatTypeParams = thatType.getTemplateTypeMap();
1337         boolean templateMatch = true;
1338         if (isExemptFromTemplateTypeInvariance(thatType)) {
1339           // Array and Object are exempt from template type invariance; their
1340           // template types maps are considered a match only if the ObjectElementKey
1341           // values are subtypes/supertypes of one another.
1342           TemplateType key = thisType.registry.getObjectElementKey();
1343           JSType thisElement = thisTypeParams.getResolvedTemplateType(key);
1344           JSType thatElement = thatTypeParams.getResolvedTemplateType(key);
1345     
1346           templateMatch = thisElement.isSubtype(thatElement, implicitImplCache, subtypingMode)
1347               || thatElement.isSubtype(thisElement, implicitImplCache, subtypingMode);
1348         } else {
1349           templateMatch = thisTypeParams.checkEquivalenceHelper(
1350               thatTypeParams, EquivalenceMethod.INVARIANT, subtypingMode);
1351         }
1352         if (!templateMatch) {
1353           return false;
1354         }
1355     
1356         // If the super type is a structural type, then we can't safely remove a templatized type
1357         // (since it might affect the types of the properties)
1358         if (implicitImplCache.isStructuralTyping()
1359             && thisType.isObject() && thatType.isStructuralType()) {
1360           return thisType.toMaybeObjectType().isStructuralSubtype(
1361               thatType.toMaybeObjectType(), implicitImplCache, subtypingMode);
1362         }
1363     
1364         // Templatized types. For nominal types, the above check guarantees TemplateTypeMap
1365         // equivalence; check if the base type is a subtype.
1366         if (thisType.isTemplatizedType()) {
1367           return thisType.toMaybeTemplatizedType().getReferencedType().isSubtype(
1368               thatType, implicitImplCache, subtypingMode);
1369         }
1370     
1371         // proxy types
1372         if (thatType instanceof ProxyObjectType) {
1373           return thisType.isSubtype(
1374               ((ProxyObjectType) thatType).getReferencedTypeInternal(),
1375               implicitImplCache, subtypingMode);
1376         }
1377         return false;
1378       }

1403       /**
1404        * Resolve this type in the given scope.
1405        *
1406        * The returned value must be equal to {@code this}, as defined by
1407        * {@link #isEquivalentTo}. It may or may not be the same object. This method
1408        * may modify the internal state of {@code this}, as long as it does
1409        * so in a way that preserves Object equality.
1410        *
1411        * For efficiency, we should only resolve a type once per compilation job.
1412        * For incremental compilations, one compilation job may need the
1413        * artifacts from a previous generation, so we will eventually need
1414        * a generational flag instead of a boolean one.
1415        */
1416       public final JSType resolve(ErrorReporter t, StaticTypedScope<JSType> scope) {
1417         if (resolved) {
1418           // TODO(nicksantos): Check to see if resolve() looped back on itself.
1419           // Preconditions.checkNotNull(resolveResult);
1420           if (resolveResult == null) {
1421             return registry.getNativeType(JSTypeNative.UNKNOWN_TYPE);
1422           }
1423           return resolveResult;
1424         }
1425         resolved = true;
1426         resolveResult = resolveInternal(t, scope);
1427         resolveResult.setResolvedTypeInternal(resolveResult);
1428         return resolveResult;
1429       }

1434       abstract JSType resolveInternal(ErrorReporter t, StaticTypedScope<JSType> scope);

1437         resolveResult = type;
1438         resolved = true;
1439       }

1540       /**
1541        * describe the status of checking that a function
1542        * implicitly implements an interface.
1543        *
1544        * it also be used to describe the status of checking
1545        * that a record type structurally matches another
1546        * record type
1547        *
1548        * A function implicitly implements an interface if
1549        * the function does not use @implements to declare
1550        * that it implements the interface, but its class
1551        * structure complies with the protocol defined
1552        * by the interface
1553        */
1554       static enum MatchStatus {
1555         /**
1556          * indicate that a function implicitly
1557          * implements an interface (i.e., the function
1558          * structurally complies with the protocol
1559          * defined in interface)
1560          *
1561          * or a record type matches another record type
1562          */
1563         MATCH(true),
1564         /**
1565          * indicate that a function does not implicitly
1566          * implements an interface (i.e., the function
1567          * does not structurally comply with the protocol
1568          * defined in interface)
1569          *
1570          * or a record type does not match another
1571          * record type
1572          */
1573         NOT_MATCH(false),
1574         /**
1575          * indicate that the interface and function
1576          * relationship is under processing
1577          */
1578         PROCESSING(true);
1579     
1580         MatchStatus(boolean isSubtype) {
1581           this.isSubtype = isSubtype;
1582         }
1583     
1584         private boolean isSubtype;
1585         boolean subtypeValue() {
1586           return this.isSubtype;
1587         }
1588       }

1605       /**
1606        * cache used by equivalence check logic
1607        */
1608       static class EqCache extends MatchCache {
1609         private IdentityHashMap<JSType, IdentityHashMap<JSType, MatchStatus>> matchCache;
1610     
1611         static EqCache create() {
1612           return new EqCache(true);
1613         }
1614     
1615         static EqCache createWithoutStructuralTyping() {
1616           return new EqCache(false);
1617         }
1618     
1619         private EqCache(boolean isStructuralTyping) {
1620           super(isStructuralTyping);
1621           this.matchCache = null;
1622         }
1623     
1624         void updateCache(JSType t1, JSType t2, MatchStatus isMatch) {
1625           IdentityHashMap<JSType, MatchStatus> map = this.matchCache.get(t1);
1626           if (map == null) {
1627             map = new IdentityHashMap<>();
1628           }
1629           map.put(t2, isMatch);
1630           this.matchCache.put(t1, map);
1631         }
1632     
1633         MatchStatus checkCache(JSType t1, JSType t2) {
1634           if (this.matchCache == null) {
1635             this.matchCache = new IdentityHashMap<>();
1636           }
1637           // check the cache
1638           if (this.matchCache.containsKey(t1) && this.matchCache.get(t1).containsKey(t2)) {
1639             return this.matchCache.get(t1).get(t2);
1640           } else if (this.matchCache.containsKey(t2) && this.matchCache.get(t2).containsKey(t1)) {
1641             return this.matchCache.get(t2).get(t1);
1642           } else {
1643             this.updateCache(t1, t2, MatchStatus.PROCESSING);
1644             return null;
1645           }
1646         }
1647       }

1649       /**
1650        * cache used by check sub-type logic
1651        */
1652       static class ImplCache extends MatchCache {
1653         private HashBasedTable<JSType, JSType, MatchStatus> matchCache;
1654     
1655         static ImplCache create() {
1656           return new ImplCache(true);
1657         }
1658     
1659         static ImplCache createWithoutStructuralTyping() {
1660           return new ImplCache(false);
1661         }
1662     
1663         private ImplCache(boolean isStructuralTyping) {
1664           super(isStructuralTyping);
1665           this.matchCache = null;
1666         }
1667     
1668         void updateCache(JSType subType, JSType superType, MatchStatus isMatch) {
1669           this.matchCache.put(subType, superType, isMatch);
1670         }
1671     
1672         MatchStatus checkCache(JSType subType, JSType superType) {
1673           if (this.matchCache == null) {
1674             this.matchCache = HashBasedTable.create();
1675           }
1676           // check the cache
1677           if (this.matchCache.contains(subType, superType)) {
1678             return this.matchCache.get(subType, superType);
1679           } else {
1680             this.updateCache(subType, superType, MatchStatus.PROCESSING);
1681             return null;
1682           }
1683         }
1684       }
