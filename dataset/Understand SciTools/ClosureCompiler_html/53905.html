17     package com.google.javascript.jscomp;
18     
19     /**
20      * Tests for {@link ClosureCodeRemoval}
21      *
22      * @author robbyw@google.com (Robby Walker)
23      */
24     public final class ClosureCodeRemovalTest extends CompilerTestCase {
25     
26       private static String EXTERNS = "var window;";
27     
28       public ClosureCodeRemovalTest() {
29         super(EXTERNS);
30       }
31     
32       public void testRemoveAbstract() {
33         test("function Foo() {}; Foo.prototype.doSomething = goog.abstractMethod;",
34             "function Foo() {};");
35       }
36     
37       public void testRemoveMultiplySetAbstract() {
38         test("function Foo() {}; Foo.prototype.doSomething = " +
39             "Foo.prototype.doSomethingElse = Foo.prototype.oneMore = " +
40             "goog.abstractMethod;",
41             "function Foo() {};");
42       }
43     
44       public void testDoNotRemoveNormal() {
45         testSame("function Foo() {}; Foo.prototype.doSomething = function() {};");
46       }
47     
48       public void testDoNotRemoveOverride() {
49         test("function Foo() {}; Foo.prototype.doSomething = goog.abstractMethod;" +
50              "function Bar() {}; goog.inherits(Bar, Foo);" +
51              "Bar.prototype.doSomething = function() {}",
52              "function Foo() {}; function Bar() {}; goog.inherits(Bar, Foo);" +
53              "Bar.prototype.doSomething = function() {}");
54       }
55     
56       public void testDoNotRemoveNonQualifiedName() {
57         testSame("document.getElementById('x').y = goog.abstractMethod;");
58       }
59     
60       public void testStopRemovalAtNonQualifiedName() {
61         test("function Foo() {}; function Bar() {};" +
62              "Foo.prototype.x = document.getElementById('x').y = Bar.prototype.x" +
63              " = goog.abstractMethod;",
64              "function Foo() {}; function Bar() {};" +
65              "Foo.prototype.x = document.getElementById('x').y = " +
66              "goog.abstractMethod;");
67       }
68     
69       public void testRemoveAbstract_annotation() {
70         test(
71             LINE_JOINER.join(
72                 "function Foo() {};",
73                 "/** @abstract */",
74                 "Foo.prototype.doSomething = function() {};"),
75             "function Foo() {};");
76       }
77     
78       public void testAssertionRemoval1() {
79         test("var x = goog.asserts.assert(y(), 'message');", "var x = y();");
80       }
81     
82       public void testAssertionRemoval2() {
83         test("goog.asserts.assert(y(), 'message');", "");
84       }
85     
86       public void testAssertionRemoval3() {
87         test("goog.asserts.assert();", "");
88       }
89     
90       public void testAssertionRemoval4() {
91         test("var x = goog.asserts.assert();", "var x = void 0;");
92       }
93     
94       @Override
95       protected ClosureCodeRemoval getProcessor(Compiler compiler) {
96         return new ClosureCodeRemoval(compiler, true, true);
97       }
98     }
