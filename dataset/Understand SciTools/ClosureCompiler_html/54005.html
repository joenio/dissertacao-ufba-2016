50     /**
51      * A {@code NamedType} is a named reference to some other type.  This provides
52      * a convenient mechanism for implementing forward references to types; a
53      * {@code NamedType} can be used as a placeholder until its reference is
54      * resolved.  It is also useful for representing type names in JsDoc type
55      * annotations, some of which may never be resolved (as they may refer to
56      * types in host systems not yet supported by JSCompiler, such as the JVM.)<p>
57      *
58      * An important distinction: {@code NamedType} is a type name reference,
59      * whereas {@link ObjectType} is a named type object, such as an Enum name.
60      * The Enum itself is typically used only in a dot operator to name one of its
61      * constants, or in a declaration, where its name will appear in a
62      * NamedType.<p>
63      *
64      * A {@code NamedType} is not currently a full-fledged typedef, because it
65      * cannot resolve to any JavaScript type.  It can only resolve to a named
66      * {@link JSTypeRegistry} type, or to {@link FunctionType} or
67      * {@link EnumType}.<p>
68      *
69      * If full typedefs are to be supported, then each method on each type class
70      * needs to be reviewed to make sure that everything works correctly through
71      * typedefs.  Alternatively, we would need to walk through the parse tree and
72      * unroll each reference to a {@code NamedType} to its resolved type before
73      * applying the rest of the analysis.<p>
74      *
75      * TODO(user): Revisit all of this logic.<p>
76      *
77      * The existing typing logic is hacky.  Unresolved types should get processed
78      * in a more consistent way, but with the Rhino merge coming, there will be
79      * much that has to be changed.<p>
80      *
81      */
82     public class NamedType extends ProxyObjectType {

100       /**
101        * Create a named type based on the reference.
102        */
103       NamedType(JSTypeRegistry registry, String reference,
104           String sourceName, int lineno, int charno) {
105         super(registry, registry.getNativeObjectType(JSTypeNative.UNKNOWN_TYPE));
106     
107         Preconditions.checkNotNull(reference);
108         this.reference = reference;
109         this.sourceName = sourceName;
110         this.lineno = lineno;
111         this.charno = charno;
112       }

114       @Override
115       boolean defineProperty(String propertyName, JSType type,
116           boolean inferred, Node propertyNode) {
117         if (!isResolved()) {
118           // If this is an unresolved object type, we need to save all its
119           // properties and define them when it is resolved.
120           if (propertyContinuations == null) {
121             propertyContinuations = new ArrayList<>();
122           }
123           propertyContinuations.add(
124               new PropertyContinuation(
125                   propertyName, type, inferred, propertyNode));
126           return true;
127         } else {
128           return super.defineProperty(
129               propertyName, type, inferred, propertyNode);
130         }
131       }
132     
133       private void finishPropertyContinuations() {
134         ObjectType referencedObjType = getReferencedObjTypeInternal();
135         if (referencedObjType != null
136             && !referencedObjType.isUnknownType()
137             && propertyContinuations != null) {
138           for (PropertyContinuation c : propertyContinuations) {
139             c.commit(this);
140           }
141         }
142         propertyContinuations = null;
143       }

180       /**
181        * Resolve the referenced type within the enclosing scope.
182        */
183       @Override
184       JSType resolveInternal(ErrorReporter t, StaticTypedScope<JSType> enclosing) {
185         // TODO(user): Investigate whether it is really necessary to keep two
186         // different mechanisms for resolving named types, and if so, which order
187         // makes more sense. Now, resolution via registry is first in order to
188         // avoid triggering the warnings built into the resolution via properties.
189         boolean resolved = resolveViaRegistry(t);
190         if (detectInheritanceCycle()) {
191           handleTypeCycle(t);
192         }
193     
194         if (resolved) {
195           super.resolveInternal(t, enclosing);
196           finishPropertyContinuations();
197           return registry.isLastGeneration() ?
198               getReferencedType() : this;
199         }
200     
201         resolveViaProperties(t, enclosing);
202         if (detectInheritanceCycle()) {
203           handleTypeCycle(t);
204         }
205     
206         super.resolveInternal(t, enclosing);
207         if (isResolved()) {
208           finishPropertyContinuations();
209         }
210         return registry.isLastGeneration() ?
211             getReferencedType() : this;
212       }
213     
214       /**
215        * Resolves a named type by looking it up in the registry.
216        * @return True if we resolved successfully.
217        */
218       private boolean resolveViaRegistry(ErrorReporter reporter) {
219         JSType type = registry.getType(reference);
220         if (type != null) {
221           setReferencedAndResolvedType(type, reporter);
222           return true;
223         }
224         return false;
225       }

227       /**
228        * Resolves a named type by looking up its first component in the scope, and
229        * subsequent components as properties. The scope must have been fully
230        * parsed and a symbol table constructed.
231        */
232       private void resolveViaProperties(ErrorReporter reporter,
233                                         StaticTypedScope<JSType> enclosing) {
234         JSType value = lookupViaProperties(reporter, enclosing);
235         // last component of the chain
236         if (value != null && value.isFunctionType() &&
237             (value.isConstructor() || value.isInterface())) {
238           FunctionType functionType = value.toMaybeFunctionType();
239           setReferencedAndResolvedType(functionType.getInstanceType(), reporter);
240         } else if (value != null && value.isNoObjectType()) {
241           setReferencedAndResolvedType(
242               registry.getNativeFunctionType(
243                   JSTypeNative.NO_OBJECT_TYPE).getInstanceType(), reporter);
244         } else if (value instanceof EnumType) {
245           setReferencedAndResolvedType(
246               ((EnumType) value).getElementsType(), reporter);
247         } else {
248           // We've been running into issues where people forward-declare
249           // non-named types. (This is legitimate...our dependency management
250           // code doubles as our forward-declaration code.)
251           //
252           // So if the type does resolve to an actual value, but it's not named,
253           // then don't respect the forward declaration.
254           handleUnresolvedType(reporter, value == null || value.isUnknownType());
255         }
256       }
257     
258       /**
259        * Resolves a type by looking up its first component in the scope, and
260        * subsequent components as properties. The scope must have been fully
261        * parsed and a symbol table constructed.
262        * @return The type of the symbol, or null if the type could not be found.
263        */
264       private JSType lookupViaProperties(ErrorReporter reporter,
265           StaticTypedScope<JSType> enclosing) {
266         String[] componentNames = reference.split("\\.", -1);
267         if (componentNames[0].length() == 0) {
268           return null;
269         }
270         StaticTypedSlot<JSType> slot = enclosing.getSlot(componentNames[0]);
271         if (slot == null) {
272           return null;
273         }
274         // If the first component has a type of 'Unknown', then any type
275         // names using it should be regarded as silently 'Unknown' rather than be
276         // noisy about it.
277         JSType slotType = slot.getType();
278         if (slotType == null || slotType.isAllType() || slotType.isNoType()) {
279           return null;
280         }
281         JSType value = getTypedefType(reporter, slot);
282         if (value == null) {
283           return null;
284         }
285     
286         // resolving component by component
287         for (int i = 1; i < componentNames.length; i++) {
288           ObjectType parentClass = ObjectType.cast(value);
289           if (parentClass == null) {
290             return null;
291           }
292           if (componentNames[i].length() == 0) {
293             return null;
294           }
295           value = parentClass.getPropertyType(componentNames[i]);
296         }
297         return value;
298       }
299     
300       private void setReferencedAndResolvedType(
301           JSType type, ErrorReporter reporter) {
302         if (validator != null) {
303           validator.apply(type);
304         }
305         setReferencedType(type);
306         checkEnumElementCycle(reporter);
307         checkProtoCycle(reporter);
308         setResolvedTypeInternal(getReferencedType());
309       }
310     
311       private void handleTypeCycle(ErrorReporter t) {
312         setReferencedType(
313             registry.getNativeObjectType(JSTypeNative.UNKNOWN_TYPE));
314         warning(t, "Cycle detected in inheritance chain of type " + reference);
315         setResolvedTypeInternal(getReferencedType());
316       }
317     
318       private void checkEnumElementCycle(ErrorReporter t) {
319         JSType referencedType = getReferencedType();
320         if (referencedType instanceof EnumElementType &&
321             ((EnumElementType) referencedType).getPrimitiveType() == this) {
322           handleTypeCycle(t);
323         }
324       }
325     
326       private void checkProtoCycle(ErrorReporter t) {
327         JSType referencedType = getReferencedType();
328         if (referencedType == this) {
329           handleTypeCycle(t);
330         }
331       }
332     
333       // Warns about this type being unresolved iff it's not a forward-declared
334       // type name.
335       private void handleUnresolvedType(
336           ErrorReporter t, boolean ignoreForwardReferencedTypes) {
337         if (registry.isLastGeneration()) {
338           boolean isForwardDeclared =
339               ignoreForwardReferencedTypes &&
340               registry.isForwardDeclaredType(reference);
341           if (!isForwardDeclared && registry.isLastGeneration()) {
342             warning(t, "Bad type annotation. Unknown type " + reference);
343           } else {
344             setReferencedType(
345                 registry.getNativeObjectType(
346                     JSTypeNative.NO_RESOLVED_TYPE));
347     
348             if (registry.isLastGeneration() && validator != null) {
349               validator.apply(getReferencedType());
350             }
351           }
352     
353           setResolvedTypeInternal(getReferencedType());
354         } else {
355           setResolvedTypeInternal(this);
356         }
357       }
358     
359       private JSType getTypedefType(ErrorReporter t, StaticTypedSlot<JSType> slot) {
360         JSType type = slot.getType();
361         if (type != null) {
362           return type;
363         }
364         handleUnresolvedType(t, true);
365         return null;
366       }
367     
368       @Override
369       public boolean setValidator(Predicate<JSType> validator) {
370         // If the type is already resolved, we can validate it now. If
371         // the type has not been resolved yet, we need to wait till its
372         // resolved before we can validate it.
373         if (this.isResolved()) {
374           return super.setValidator(validator);
375         } else {
376           this.validator = validator;
377           return true;
378         }
379       }
380     
381       void warning(ErrorReporter reporter, String message) {
382         reporter.warning(message, sourceName, lineno, charno);
383       }

385       /** Store enough information to define a property at a later time. */
386       private static final class PropertyContinuation {
387         private final String propertyName;
388         private final JSType type;
389         private final boolean inferred;
390         private final Node propertyNode;
391     
392         private PropertyContinuation(
393             String propertyName,
394             JSType type,
395             boolean inferred,
396             Node propertyNode) {
397           this.propertyName = propertyName;
398           this.type = type;
399           this.inferred = inferred;
400           this.propertyNode = propertyNode;
401         }
402     
403         void commit(ObjectType target) {
404           target.defineProperty(
405               propertyName, type, inferred, propertyNode);
406         }
407       }

409       @Override
410       public <T> T visit(Visitor<T> visitor) {
411         return visitor.caseNamedType(this);
