23     import java.util.List;

27     /**
28      *
29      * @author blickly@google.com (Ben Lickly)
30      * @author dimvar@google.com (Dimitris Vardoulakis)
31      */
32     class Property {
33       // AST node where the property is defined; most commonly null.
34       // TODO(dimvar): if having this field on every property wastes memory,
35       // consider alternatives:
36       // 1) Subclass Property for the few ones that have jsdoc.
37       // 2) Store the jsdocs scattered in NominalType and Namespace.
38       //    This is hard b/c for objlit namespaces, we don't keep around the
39       //    Namespace class in NTI; they're plain-old object literals.
40       private final Node defSite;
41       private final JSType inferredType;
42       private final JSType declaredType;
43       // Attributes are ordered: constant <= required <= optional
44       private enum Attribute {
45         CONSTANT, // For required props only
46         OPTIONAL,
47         REQUIRED;
48       }
49       private Attribute attribute;
50     
51       private Property(Node defSite, JSType inferredType, JSType declaredType, Attribute attribute) {
52         Preconditions.checkArgument(inferredType != null);
53         this.defSite = defSite;
54         this.inferredType = inferredType;
55         this.declaredType = declaredType;
56         this.attribute = attribute;
57       }
58     
59       static Property make(JSType inferredType, JSType declaredType) {
60         return new Property(null, inferredType, declaredType, Attribute.REQUIRED);
61       }
62     
63       static Property makeWithDefsite(Node defSite, JSType inferredType, JSType declaredType) {
64         return new Property(defSite, inferredType, declaredType, Attribute.REQUIRED);
65       }
66     
67       static Property makeConstant(Node defSite, JSType inferredType, JSType declaredType) {
68         return new Property(defSite, inferredType, declaredType, Attribute.CONSTANT);
69       }
70     
71       static Property makeOptional(Node defSite, JSType inferredType, JSType declaredType) {
72         return new Property(defSite, inferredType, declaredType, Attribute.OPTIONAL);
73       }
74     
75       boolean isRequired() {
76         return attribute == Attribute.REQUIRED;
77       }
78     
79       boolean isOptional() {
80         return attribute == Attribute.OPTIONAL;
81       }
82     
83       boolean isConstant() {
84         return attribute == Attribute.CONSTANT;
85       }
86     
87       boolean isDeclared() {
88         return declaredType != null;
89       }
90     
91       JSType getType() {
92         return inferredType;
93       }
94     
95       Node getDefSite() {
96         return this.defSite;
97       }
98     
99       JSType getDeclaredType() {
100         return declaredType;
101       }
102     
103       Property withOptional() {
104         return isOptional() ? this
105             : new Property(defSite, inferredType, declaredType, Attribute.OPTIONAL);
106       }
107     
108       Property withRequired() {
109         return isRequired() ? this
110             : new Property(defSite, inferredType, declaredType, Attribute.REQUIRED);
111       }
112     
113       private static Attribute meetAttributes(Attribute a1, Attribute a2) {
114         if (a1 == Attribute.CONSTANT || a2 == Attribute.CONSTANT) {
115           return Attribute.CONSTANT;
116         }
117         if (a1 == Attribute.REQUIRED || a2 == Attribute.REQUIRED) {
118           return Attribute.REQUIRED;
119         }
120         return Attribute.OPTIONAL;
121       }
122     
123       private static Attribute joinAttributes(Attribute a1, Attribute a2) {
124         if (a1 == Attribute.OPTIONAL || a2 == Attribute.OPTIONAL) {
125           return Attribute.OPTIONAL;
126         }
127         if (a1 == Attribute.REQUIRED || a2 == Attribute.REQUIRED) {
128           return Attribute.REQUIRED;
129         }
130         return Attribute.CONSTANT;
131       }
132     
133       Property specialize(Property other) {
134         // TODO(dimvar): consider not always creating a new property here; only
135         // when you can actually specialize.
136         return new Property(
137             this.defSite,
138             this.inferredType.specialize(other.inferredType),
139             this.declaredType,
140             meetAttributes(this.attribute, other.attribute));
141       }
142     
143       static Property meet(Property p1, Property p2) {
144         return new Property(
145             p1.defSite == p2.defSite ? p1.defSite : null,
146             JSType.meet(p1.inferredType, p2.inferredType),
147             null,
148             meetAttributes(p1.attribute, p2.attribute));
149       }
150     
151       static Property join(Property p1, Property p2) {
152         JSType declType, p1decl = p1.declaredType, p2decl = p2.declaredType;
153         if (p1decl == null || p2decl == null) {
154           declType = null;
155         } else if (p1decl.equals(p2decl)) {
156           declType = p1decl;
157         } else {
158           declType = null;
159         }
160         return new Property(
161             p1.defSite == p2.defSite ? p1.defSite : null,
162             JSType.join(p1.inferredType, p2.inferredType),
163             declType,
164             joinAttributes(p1.attribute, p2.attribute));
165       }
166     
167       /**
168        * Unify the two types bidirectionally, ignoring type variables, but
169        * treating JSType.UNKNOWN as a "hole" to be filled.
170        * @return The unified type, or null if unification fails
171        */
172       static Property unifyUnknowns(Property p1, Property p2) {
173         JSType unifiedDeclaredType = null;
174         if (p1.declaredType != null && p2.declaredType != null) {
175           unifiedDeclaredType =
176             JSType.unifyUnknowns(p1.declaredType, p2.declaredType);
177           if (unifiedDeclaredType == null) {
178             return null;
179           }
180         }
181         JSType unifiedInferredType =
182             JSType.unifyUnknowns(p1.inferredType, p2.inferredType);
183         if (unifiedInferredType == null) {
184           return null;
185         }
186         return new Property(
187             p1.defSite == p2.defSite ? p1.defSite : null,
188             unifiedInferredType, unifiedDeclaredType,
189             meetAttributes(p1.attribute, p2.attribute));
190       }
191     
192       /** Returns whether unification succeeded */
193       boolean unifyWithSubtype(Property other, List<String> typeParameters,
194           Multimap<String, JSType> typeMultimap, SubtypeCache subSuperMap) {
195         if (!inferredType.unifyWithSubtype(
196             other.inferredType, typeParameters, typeMultimap, subSuperMap)) {
197           return false;
198         } else if (declaredType != null && other.declaredType != null &&
199             !declaredType.unifyWithSubtype(
200                 other.declaredType, typeParameters, typeMultimap, subSuperMap)) {
201           return false;
202         }
203         return true;
204       }
205     
206       Property substituteGenerics(Map<String, JSType> concreteTypes) {
207         if (concreteTypes.isEmpty()) {
208           return this;
209         }
210         return new Property(
211             this.defSite,
212             inferredType.substituteGenerics(concreteTypes),
213             declaredType == null ?
214             null : declaredType.substituteGenerics(concreteTypes),
215             attribute);
216       }
217     
218       @Override
219       public String toString() {
220         return appendTo(new StringBuilder()).toString();
221       }
222     
223       public StringBuilder appendTo(StringBuilder builder) {
224         switch (attribute) {
225           case CONSTANT:
226             return inferredType.appendTo(builder).append('^');
227           case REQUIRED:
228             return inferredType.appendTo(builder);
229           case OPTIONAL:
230             return inferredType.appendTo(builder).append('=');
231           default:
232             throw new RuntimeException("Unknown Attribute value " + attribute);
233         }
234       }
235     
236       @Override
237       public boolean equals(Object o) {
238         if (o == null) {
239           return false;
240         }
241         if (this == o) {
242           return true;
243         }
244         Preconditions.checkArgument(o instanceof Property);
245         Property p2 = (Property) o;
246         return inferredType.equals(p2.inferredType) &&
247             attribute == p2.attribute;
248       }
249     
250       @Override
251       public int hashCode() {
252         return Objects.hash(inferredType, attribute);
253       }
254     }
