33     /**
34      * Generate a nice HTML file describing the name reference graph.
35      * For each declaration, list the sites where the declaration's name
36      * is referenced, and list all the names that the declaration references.
37      * For each, name exactly where use occurs in the source code.
38      *
39      * <p>This report should be useful both for internal compiler
40      * developers and for engineers trying to understand running behavior
41      * of their code or who want to understand why the compiler won't
42      * move their code into a new module.
43      *
44      * @author bowdidge@google.com (Robert Bowdidge)
45      */
46     
47     final class NameReferenceGraphReport {
48       private NameReferenceGraph graph = null;
49     
50       /**
51        * Create a NameReferenceGraphReport object.
52        *
53        * @param g  name reference graph to describe in report.
54        */
55       NameReferenceGraphReport(NameReferenceGraph g) {
56         this.graph = g;
57       }
58     
59       /**
60        * Generate a nice HTML file describing the name reference graph.
61        * For each declaration, list the sites where the declaration's name
62        * is referenced, and list all the names that the declaration references.
63        * For each, name exactly where use occurs in the source code.
64        *
65        * <p>This report should be useful both for internal compiler
66        * developers and for engineers trying to understand running
67        * behavior of their code or who want to understand why
68        * AbstractCompiler won't move their code into a new module.
69        *
70        * @return String containing the entire HTML for the report.
71        */
72       public String getHtmlReport() {
73         StringBuilder builder = new StringBuilder();
74         List<DiGraphNode<Name, Reference>> nodes = new ArrayList<>();
75         Iterables.addAll(nodes, graph.getDirectedGraphNodes());
76     
77         generateHtmlReportHeader(builder);
78     
79         builder.append("<h1>Name Reference Graph Dump</h1>\n");
80         builder.append("OVERALL STATS\n");
81         builder.append("<ul>\n");
82         builder.append("<li>Total names: ").append(nodes.size());
83         builder.append("</ul>\n");
84     
85         builder.append("ALL NAMES\n");
86         builder.append("<UL>\n");
87     
88         // Sort declarations in alphabetical order.
89         Collections.sort(nodes, new DiGraphNodeComparator());
90     
91         for (DiGraphNode<Name, Reference> n : nodes) {
92           // Generate the HTML describing the declaration itself.
93           generateDeclarationReport(builder, n);
94     
95           // Next, list the places where this name is used (REFERS TO), and the
96           // names that this declaration refers to (REFERENCED BY).
97           List<DiGraphEdge<Name, Reference>> outEdges =
98               graph.getOutEdges(n.getValue());
99           List<DiGraphEdge<Name, Reference>> inEdges =
100               graph.getInEdges(n.getValue());
101     
102           // Don't bother to create the dotted list if we don't have anything to
103           // put in it.
104           if (!outEdges.isEmpty() || !inEdges.isEmpty()) {
105             builder.append("<ul>");
106     
107             if (!outEdges.isEmpty()) {
108               builder.append("<li>REFERS TO:<br>\n");
109               builder.append("<ul>");
110               for (DiGraphEdge<Name, Reference> edge : outEdges) {
111                 generateEdgeReport(builder, edge.getDestination().getValue(),
112                     edge);
113               }
114               builder.append("</ul>\n");
115             }
116     
117             if (!inEdges.isEmpty()) {
118               builder.append("<li>REFERENCED BY:<br>\n");
119               builder.append("<ul>");
120               for (DiGraphEdge<Name, Reference> edge : inEdges) {
121                 generateEdgeReport(builder, edge.getSource().getValue(), edge);
122               }
123               builder.append("</ul>");
124             }
125             builder.append("</ul>\n");
126           }
127         }
128         builder.append("</ul>\n");
129         generateHtmlReportFooter(builder);
130         return builder.toString();
131       }
132     
133       /**
134        * Given a node, find the name of the containing source file.
135        *
136        * @param node Parse tree node whose filename is requested
137        * @return String containing name of source file, or empty string if name
138        *     cannot be identified.
139        */
140       private static String getSourceFile(Node node) {
141         String filename = node.getSourceFileName();
142         if (filename == null) {
143           return "";
144         }
145         return filename;
146       }
147     
148       /**
149        * Generate the HTML for describing a specific declaration.
150        * @param builder  contents of report to be generated
151        * @param declarationNode declaration to describe
152        */
153       private void generateDeclarationReport(StringBuilder builder,
154           DiGraphNode<Name, Reference> declarationNode) {
155         // Provide the name and location of declaration,
156         // with an anchor to allow navigation to this declaration.
157         String declName = declarationNode.getValue().getQualifiedName();
158         JSType declType = declarationNode.getValue().getType();
159     
160         builder.append("<LI> ");
161         builder.append("<A NAME=\"").append(declName).append("\">");
162         builder.append(declName);
163         builder.append("\n");
164     
165         // Provide the type of the declaration.
166         // This is helpful for debugging.
167         generateType(builder, declType);
168     
169         // List all the definitions of this name that were found in the code.
170         // For each, list
171         List<DefinitionsRemover.Definition> defs =
172             declarationNode.getValue().getDeclarations();
173     
174         if (defs.isEmpty()) {
175            builder.append("<br>No definitions found<br>");
176         } else {
177           // Otherwise, provide a list of definitions in a dotted list.
178           // For each definition, print the location where that definition is
179           // found.
180           builder.append("<ul>");
181           for (DefinitionsRemover.Definition def : defs) {
182             Node fnDef = def.getRValue();
183             String sourceFileName = getSourceFile(fnDef);
184             builder.append("<li> Defined: ");
185             generateSourceReferenceLink(builder,
186                 sourceFileName, fnDef.getLineno(), fnDef.getCharno());
187           }
188           builder.append("</ul>");
189         }
190       }

192       /**
193        *  Generate the HTML header for the report style.
194        * Borrowed straight from NameAnalyzer's report style.
195        *
196        * @param builder contents of the report to be generated
197        */
198       private static void generateHtmlReportHeader(StringBuilder builder) {
199         builder.append("<!DOCTYPE html>\n" +
200             "<html>" +
201             "<head>" +
202             "<meta http-equiv=\"Content-Type\" " +
203             "content=\"text/html;charset=utf-8\" >" +
204             "<title>Name Reference Graph Dump</title>" +
205             "<style type=\"text/css\">body, td, ");
206         builder.append("p {font-family: Arial; font-size: 83%} ");
207         builder.append("ul {margin-top:2px; margin-left:0px; padding-left:1em;}");
208         builder.append("li {margin-top:3px; margin-left:24px;" +
209             "padding-left:0px;padding-bottom: 4px}");
210         builder.append("</style></head><body>\n");
211       }
212     
213       /**
214        * Generate the HTML footer for the report style.
215        */
216       private static void generateHtmlReportFooter(StringBuilder builder) {
217         builder.append("</body></html>");
218       }

220       /**
221        * Generate a description of a specific edge between two nodes.
222        * For each edge, name the element being linked, the location of the
223        * reference in the source file, and the type of the reference.
224        *
225        * @param builder contents of the report to be generated
226        * @param referencedDecl name of the declaration being referenced
227        * @param edge the graph edge being described
228        */
229       private void generateEdgeReport(StringBuilder builder,
230           Name referencedDecl, DiGraphEdge<Name, Reference> edge) {
231         String srcDeclName = referencedDecl.getQualifiedName();
232         builder.append("<li><A HREF=\"#").append(srcDeclName).append("\">");
233         builder.append(srcDeclName);
234         builder.append("</a> ");
235     
236         Node def = edge.getValue().getSite();
237         int lineNumber = def.getLineno();
238         int columnNumber = def.getCharno();
239         String sourceFile = getSourceFile(def);
240     
241         generateSourceReferenceLink(builder, sourceFile, lineNumber, columnNumber);
242     
243         JSType defType = edge.getValue().getSite().getJSType();
244         generateType(builder, defType);
245       }
246     
247       /**
248        * Generate a link and text for a reference to a particular location
249        * in a source file.  Selecting the link should take the programmer
250        * to a browsable version of the file.
251        *
252        * @param builder  contents of the report to be generated
253        * @param sourceFile  Path to the file
254        * @param lineNumber  line where the object to view is located
255        * @param columnNumber  column where the object to highlight is located.
256        */
257       private static void generateSourceReferenceLink(StringBuilder builder,
258           String sourceFile, int lineNumber, int columnNumber) {
259         assert(sourceFile != null);
260     
261         builder.append("(");
262     
263     
264         // Print out the text path so the user knows where things come from.
265         builder.append(sourceFile).append(":").append(lineNumber).append(",").append(columnNumber);
266     
267     
268         builder.append(")");
269       }
270     
271       /**
272        * Dump a type in a nice, readable way.
273        *
274        * @param builder contents of the report to be generated.
275        * @param defType type to describe
276        */
277       private static void generateType(StringBuilder builder, JSType defType) {
278         if (defType == null) {
279           builder.append(" (type: null) ");
280         } else if (defType.isUnknownType()) {
281           builder.append(" (type: unknown) ");
282         } else {
283           builder.append(" (type: ").append(defType).append(") ");
284         }
285       }

287       /**
288        * DiGraphNodeComparator gives us a way to generate sorted lists
289        * of DiGraphNodes.  It provides a compare function used by the
290        * String class's sort method.
291        */
292       static class DiGraphNodeComparator implements Comparator<DiGraphNode<Name, Reference>> {
293         @Override
294         public int compare(DiGraphNode<Name, Reference> node1,
295             DiGraphNode<Name, Reference> node2) {
296           Preconditions.checkNotNull(node1.getValue());
297           Preconditions.checkNotNull(node2.getValue());
298     
299           if ((node1.getValue().getQualifiedName() == null) &&
300               (node2.getValue().getQualifiedName() == null)) {
301             return 0;
302           }
303     
304           // Node 1, if null, comes before node 2.
305           if (node1.getValue().getQualifiedName() == null) {
306             return -1;
307           }
308     
309           // Node 2, if null, comes before node 1.
310           if (node2.getValue().getQualifiedName() == null) {
311             return 1;
312           }
313     
314           return node1.getValue().getQualifiedName().compareTo(
315               node2.getValue().getQualifiedName());
316         }
317       }
