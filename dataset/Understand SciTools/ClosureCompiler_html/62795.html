32       @Override
33       public CompilerPass getProcessor(final Compiler compiler) {
34         PeepholeOptimizationsPass peepholePass =
35           new PeepholeOptimizationsPass(compiler,
36             new PeepholeMinimizeConditions(late, false /* useTypes */),
37             new PeepholeSubstituteAlternateSyntax(late),
38             new PeepholeRemoveDeadCode(),
39             new PeepholeFoldConstants(late, false)
40           );
41     
42         return peepholePass;
43       }

72       /** Check that removing blocks with 1 child works */
73       public void testFoldOneChildBlocksIntegration() {
74          fold("function f(){switch(foo()){default:{break}}}",
75               "function f(){foo()}");
76     
77          fold("function f(){switch(x){default:{break}}}",
78               "function f(){}");
79     
80          fold("function f(){switch(x){default:x;case 1:return 2}}",
81               "function f(){switch(x){default:case 1:return 2}}");
82     
83          // ensure that block folding does not break hook ifs
84          fold("if(x){if(true){foo();foo()}else{bar();bar()}}",
85               "if(x){foo();foo()}");
86     
87          fold("if(x){if(false){foo();foo()}else{bar();bar()}}",
88               "if(x){bar();bar()}");
89     
90          // Cases where the then clause has no side effects.
91          fold("if(x()){}", "x()");
92     
93          fold("if(x()){} else {x()}", "x()||x()");
94          fold("if(x){}", ""); // Even the condition has no side effect.
95          fold("if(a()){A()} else if (b()) {} else {C()}", "a()?A():b()||C()");
96     
97          fold("if(a()){} else if (b()) {} else {C()}",
98               "a() || (b() || C())");
99          fold("if(a()){A()} else if (b()) {} else if (c()) {} else{D()}",
100               "a() ? A() : b() || (c() || D())");
101          fold("if(a()){} else if (b()) {} else if (c()) {} else{D()}",
102               "a() || (b() || (c() || D()))");
103          fold("if(a()){A()} else if (b()) {} else if (c()) {} else{}",
104               "a()?A():b()||c()");
105     
106          // Verify that non-global scope works.
107          fold("function foo(){if(x()){}}", "function foo(){x()}");
108     
109       }

129       public void testBug1059649() {
130         // ensure that folding blocks with a single var node doesn't explode
131         fold("if(x){var y=3;}var z=5", "if(x)var y=3;var z=5");
132     
133         // With normalization, we no longer have this case.
134         foldSame("if(x){var y=3;}else{var y=4;}var z=5");
135         fold("while(x){var y=3;}var z=5", "while(x)var y=3;var z=5");
136         fold("for(var i=0;i<10;i++){var y=3;}var z=5",
137              "for(var i=0;i<10;i++)var y=3;var z=5");
138         fold("for(var i in x){var y=3;}var z=5",
139              "for(var i in x)var y=3;var z=5");
140         fold("do{var y=3;}while(x);var z=5", "do var y=3;while(x);var z=5");
141       }
