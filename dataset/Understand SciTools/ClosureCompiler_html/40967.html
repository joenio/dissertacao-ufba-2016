57       /** Check that removing blocks with 1 child works */
58       public void testFoldOneChildBlocks() {
59         late = false;
60         fold("function f(){if(x)a();x=3}",
61             "function f(){x&&a();x=3}");
62         fold("function f(){if(x){a()}x=3}",
63             "function f(){x&&a();x=3}");
64         fold("function f(){if(x){return 3}}",
65             "function f(){if(x)return 3}");
66         fold("function f(){if(x){a()}}",
67             "function f(){x&&a()}");
68         fold("function f(){if(x){throw 1}}", "function f(){if(x)throw 1;}");
69     
70         // Try it out with functions
71         fold("function f(){if(x){foo()}}", "function f(){x&&foo()}");
72         fold("function f(){if(x){foo()}else{bar()}}",
73              "function f(){x?foo():bar()}");
74     
75         // Try it out with properties and methods
76         fold("function f(){if(x){a.b=1}}", "function f(){if(x)a.b=1}");
77         fold("function f(){if(x){a.b*=1}}", "function f(){x&&(a.b*=1)}");
78         fold("function f(){if(x){a.b+=1}}", "function f(){x&&(a.b+=1)}");
79         fold("function f(){if(x){++a.b}}", "function f(){x&&++a.b}");
80         fold("function f(){if(x){a.foo()}}", "function f(){x&&a.foo()}");
81     
82         // Try it out with throw/catch/finally [which should not change]
83         fold("function f(){try{foo()}catch(e){bar(e)}finally{baz()}}",
84              "function f(){try{foo()}catch(e){bar(e)}finally{baz()}}");
85     
86         // Try it out with switch statements
87         fold("function f(){switch(x){case 1:break}}",
88              "function f(){switch(x){case 1:break}}");
89     
90         // Do while loops stay in a block if that's where they started
91         fold("function f(){if(e1){do foo();while(e2)}else foo2()}",
92              "function f(){if(e1){do foo();while(e2)}else foo2()}");
93         // Test an obscure case with do and while
94         fold("if(x){do{foo()}while(y)}else bar()",
95              "if(x){do foo();while(y)}else bar()");
96     
97         // Play with nested IFs
98         fold("function f(){if(x){if(y)foo()}}",
99              "function f(){x && (y && foo())}");
100         fold("function f(){if(x){if(y)foo();else bar()}}",
101              "function f(){x&&(y?foo():bar())}");
102         fold("function f(){if(x){if(y)foo()}else bar()}",
103              "function f(){x?y&&foo():bar()}");
104         fold("function f(){if(x){if(y)foo();else bar()}else{baz()}}",
105              "function f(){x?y?foo():bar():baz()}");
106     
107         fold("if(e1){while(e2){if(e3){foo()}}}else{bar()}",
108              "if(e1)while(e2)e3&&foo();else bar()");
109     
110         fold("if(e1){with(e2){if(e3){foo()}}}else{bar()}",
111              "if(e1)with(e2)e3&&foo();else bar()");
112     
113         fold("if(a||b){if(c||d){var x;}}", "if(a||b)if(c||d)var x");
114         fold("if(x){ if(y){var x;}else{var z;} }",
115              "if(x)if(y)var x;else var z");
116     
117         // NOTE - technically we can remove the blocks since both the parent
118         // and child have elses. But we don't since it causes ambiguities in
119         // some cases where not all descendent ifs having elses
120         fold("if(x){ if(y){var x;}else{var z;} }else{var w}",
121              "if(x)if(y)var x;else var z;else var w");
122         fold("if (x) {var x;}else { if (y) { var y;} }",
123              "if(x)var x;else if(y)var y");
124     
125         // Here's some of the ambiguous cases
126         fold("if(a){if(b){f1();f2();}else if(c){f3();}}else {if(d){f4();}}",
127              "if(a)if(b){f1();f2()}else c&&f3();else d&&f4()");
128     
129         fold("function f(){foo()}", "function f(){foo()}");
130         fold("switch(x){case y: foo()}", "switch(x){case y:foo()}");
131         fold("try{foo()}catch(ex){bar()}finally{baz()}",
132              "try{foo()}catch(ex){bar()}finally{baz()}");
133       }

463              "function f() { while(x) { if (y) { break  }}}");

481         foldSame("function f() { while(x) { while (y) { return } } }");

487                  "  try { while(x) {return f()}} catch (e) { } return f()}");

492
