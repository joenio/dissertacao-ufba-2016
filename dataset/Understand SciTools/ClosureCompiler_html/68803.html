35     import java.io.ByteArrayOutputStream;
36     import java.io.IOException;
37     import java.io.PrintStream;
38     import java.io.UnsupportedEncodingException;
39     import java.nio.charset.StandardCharsets;
40     import java.util.concurrent.Callable;
41     import java.util.concurrent.ExecutionException;
42     
43     import javax.annotation.concurrent.NotThreadSafe;
44     
45     /**
46      * {@link ClosureBundler} that transpiles its sources.
47      */
48     @NotThreadSafe
49     public final class TranspilingClosureBundler extends ClosureBundler {
50     
51       private static final HashFunction HASH_FUNCTION = Hashing.goodFastHash(64);
52       private static final int CACHE_SIZE = 100;
53     
54       // TODO(sdh): Not all transpilation requires the runtime, only inject if actually needed.
55       private final String es6Runtime;
56       private boolean needToBundleEs6Runtime = true;
57     
58       public TranspilingClosureBundler() {
59         this(getEs6Runtime());
60       }
61     
62       @VisibleForTesting
63       TranspilingClosureBundler(String es6Runtime) {
64         this.es6Runtime = es6Runtime;
65       }
66     
67       /**
68        * Cache recent transpilations, keyed by the hash code of the input
69        * to avoid storing the whole input.
70        */
71       @VisibleForTesting final Cache<Long, String> cachedTranspilations =
72           CacheBuilder.newBuilder().maximumSize(CACHE_SIZE).build();
73     
74       @Override
75       public void appendTo(Appendable out, DependencyInfo info, CharSource content) throws IOException {
76         if (needToBundleEs6Runtime) {
77           // Piggyback on the first call to transformInput to include the ES6 runtime as well.
78           super.appendTo(out, SimpleDependencyInfo.EMPTY, CharSource.wrap(es6Runtime));
79           needToBundleEs6Runtime = false;
80         }
81         super.appendTo(out, info, content);
82       }
83     
84       private static CompilerOptions getOptions() {
85         CompilerOptions options = new CompilerOptions();
86         options.setLanguageIn(LanguageMode.ECMASCRIPT6_STRICT);
87         options.setLanguageOut(LanguageMode.ECMASCRIPT5);
88         // Quoting keyword properties is only needed in ES3, so basically only in IE8.
89         // But we set it explicitly here because the way the test bundler works, it invokes
90         // the compiler without giving information about the browser, so we have to quote
91         // every time to be safe :-/
92         options.setQuoteKeywordProperties(true);
93         options.setSkipNonTranspilationPasses(true);
94         options.setVariableRenaming(VariableRenamingPolicy.OFF);
95         options.setPropertyRenaming(PropertyRenamingPolicy.OFF);
96         options.setWrapGoogModulesForWhitespaceOnly(false);
97         options.setPrettyPrint(true);
98         return options;
99       }
100     
101       @Override
102       protected String transformInput(final String js, final String path) {
103         try {
104           // Don't use built-in hashCode to decrease the likelihood of a collision.
105           long hashCode = HASH_FUNCTION.hashString(js, StandardCharsets.UTF_8).asLong();
106           return cachedTranspilations.get(
107               hashCode,
108               new Callable<String>() {
109                 @Override
110                 public String call() {
111                   // Neither the compiler nor the options is thread safe, so they can't be
112                   // saved as instance state.
113                   ByteArrayOutputStream baos = new ByteArrayOutputStream();
114                   Compiler compiler = new Compiler(new PrintStream(baos));
115                   // Threads can't be used in small unit tests.
116                   compiler.disableThreads();
117                   SourceFile externs = SourceFile.fromCode("externs", "function Symbol() {}");
118                   SourceFile sourceFile = SourceFile.fromCode(path, js);
119                   compiler.<SourceFile, SourceFile>compile(
120                       ImmutableList.<SourceFile>of(externs),
121                       ImmutableList.<SourceFile>of(sourceFile),
122                       getOptions());
123                   if (compiler.getErrorManager().getErrorCount() > 0) {
124                     String message;
125                     try {
126                       message = baos.toString(StandardCharsets.UTF_8.name());
127                     } catch (UnsupportedEncodingException e) {
128                       throw new RuntimeException(e);
129                     }
130                     throw new IllegalStateException(message);
131                   }
132                   return compiler.toSource();
133                 }
134               });
135         } catch (ExecutionException | UncheckedExecutionException e) {
136           // IllegalStateExceptions thrown from the callable above will end up here as
137           // UncheckedExecutionExceptions, per the contract of Cache#get. Throw the underlying
138           // IllegalStateException so that the compiler error message is at the top of the stack trace.
139           if (e.getCause() instanceof IllegalStateException) {
140             throw (IllegalStateException) e.getCause();
141           } else {
142             throw Throwables.propagate(e);
143           }
144         }
145       }
146     
147       /** Generates the runtime by requesting the "es6_runtime" library from the compiler. */
148       private static String getEs6Runtime() {
149         CompilerOptions options = getOptions();
150         options.setLanguageOut(LanguageMode.ECMASCRIPT3); // change .delete to ['delete']
151         options.setForceLibraryInjection(ImmutableList.of("es6_runtime"));
152         Compiler compiler = new Compiler();
153         // Threads can't be used in small unit tests.
154         compiler.disableThreads();
155         SourceFile externs = SourceFile.fromCode("externs", "function Symbol() {}");
156         SourceFile sourceFile = SourceFile.fromCode("source", "");
157         compiler.compile(ImmutableList.of(externs), ImmutableList.of(sourceFile), options);
158         return compiler.toSource();
159       }
160     }
