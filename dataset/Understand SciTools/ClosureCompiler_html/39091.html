31     /**
32      * Rewrites/inlines some J2CL constructs to be more optimizable.
33      */
34     public class J2clPass implements CompilerPass {
35       private static final String ALL_CLASS_FILE_NAMES = "*";
36       private final AbstractCompiler compiler;
37       private final Supplier<String> safeNameIdSupplier;
38     
39       private class GetDefineRewriter extends AbstractPostOrderCallback {
40         private Set<String> defines;
41     
42         GetDefineRewriter(Set<String> defines) {
43           this.defines = defines;
44         }
45     
46         @Override
47         public void visit(NodeTraversal t, Node n, Node parent) {
48           if (isUtilGetDefineCall(n)) {
49             substituteUtilGetDefine(n);
50           }
51         }
52     
53         private void substituteUtilGetDefine(Node callNode) {
54           Node firstExpr = callNode.getSecondChild();
55           Node secondExpr = callNode.getLastChild();
56     
57           if (secondExpr != firstExpr) {
58             secondExpr.detachFromParent();
59           } else {
60             // There is no secondExpr; default to null.
61             secondExpr = IR.nullNode();
62           }
63     
64           Node replacement = getDefineReplacement(firstExpr, secondExpr);
65           replacement.useSourceInfoIfMissingFromForTree(callNode);
66           callNode.getParent().replaceChild(callNode, replacement);
67           compiler.reportCodeChange();
68         }
69     
70         private Node getDefineReplacement(Node firstExpr, Node secondExpr) {
71           if (defines.contains(firstExpr.getString())) {
72             Node define = NodeUtil.newQName(compiler, firstExpr.getString());
73             Node defineStringValue = NodeUtil.newCallNode(IR.name("String"), define);
74             return IR.comma(secondExpr, defineStringValue);
75           } else {
76             return secondExpr;
77           }
78         }
79     
80         private boolean isUtilGetDefineCall(Node n) {
81           return n.isCall() && isUtilGetDefineMethodName(n.getFirstChild().getQualifiedName());
82         }
83     
84         private boolean isUtilGetDefineMethodName(String fnName) {
85           return fnName != null && fnName.endsWith("Util.$getDefine");
86         }
87       }
88     
89       /**
90        * Forcefully inlines aliases to native types to allow further optimizations.
91        *
92        * <p>e.g. "let $RegExp = window.RegExp; var foo = new $RegExp;" becomes "var foo = new RegExp;"
93        */
94       private class NativeAliasInliner extends AbstractPostOrderCallback {
95     
96         @Override
97         public void visit(NodeTraversal t, Node n, Node parent) {
98           if (n.isName()
99               && !NodeUtil.isNameDeclaration(parent)
100               && !parent.isAssign()) {
101             Node declaringNode = getDeclaringNode(t, n);
102             if (declaringNode != null
103                 && declaringNode.getFirstChild() != n
104                 && isNativeAlias(declaringNode)) {
105               parent.replaceChild(n, declaringNode.getFirstFirstChild().cloneTree());
106               compiler.reportCodeChange();
107             }
108           }
109         }
110     
111         private Node getDeclaringNode(NodeTraversal t, Node nameNode) {
112           Var var = t.getScope().getVar(nameNode.getString());
113           return var == null ? null : var.getParentNode();
114         }
115     
116         private boolean isNativeAlias(Node n) {
117           return NodeUtil.isNameDeclaration(n)
118               && n.getParent().isScript()
119               && n.getJSDocInfo() != null
120               && n.getJSDocInfo().isConstructor()
121               && n.getFirstFirstChild() != null
122               && n.getFirstFirstChild().isGetProp()
123               && n.getFirstFirstChild().getQualifiedName().matches("window.[A-Z][A-Za-z]+");
124         }
125       }
126     
127       /**
128        * Collects references to certain function definitions in a certain class and then inlines fully
129        * qualified static method calls to those functions anywhere in the program.
130        *
131        * <p>Assumes that the set of provided short function names will not collide with any of the
132        * collected fully qualified function names once the module prefix has been added.
133        */
134       private class ClassStaticFunctionsInliner {
135         private final String classFileName;
136         private final Set<String> fnNamesToInline;
137         private final InliningMode inliningMode;
138         private final Map<String, Node> fnsToInlineByQualifiedName = new HashMap<>();
139         private final FunctionInjector injector;
140         private final Node root;
141     
142         private ClassStaticFunctionsInliner(
143             Node root, String classFileName, Set<String> fnNamesToInline, InliningMode inliningMode) {
144           this.root = root;
145           this.classFileName = classFileName;
146           this.fnNamesToInline = fnNamesToInline;
147           this.inliningMode = inliningMode;
148     
149           this.injector = new FunctionInjector(compiler, safeNameIdSupplier, true, true, true);
150           this.injector.setKnownConstants(fnNamesToInline);
151         }
152     
153         private void run() {
154           NodeTraversal.traverseEs6(compiler, root, new FunctionDefsCollector());
155           NodeTraversal.traverseEs6(compiler, root, new StaticCallInliner());
156         }
157     
158         private class FunctionDefsCollector implements Callback {
159           @Override
160           public boolean shouldTraverse(NodeTraversal nodeTraversal, Node n, Node parent) {
161             // Only look inside the referenced class file.
162             return !n.isScript()
163                 || n.getSourceFileName().endsWith(classFileName)
164                 || classFileName.equals(ALL_CLASS_FILE_NAMES);
165           }
166     
167           @Override
168           public void visit(NodeTraversal t, Node n, Node parent) {
169             // If we arrive here then we're already inside the desired script.
170     
171             // Only look at named function declarations
172             if (!n.isAssign() || !n.getLastChild().isFunction()) {
173               return;
174             }
175     
176             // ... that are fully qualified
177             Node qualifiedNameNode = n.getFirstChild();
178             if (!qualifiedNameNode.isGetProp() || !qualifiedNameNode.isQualifiedName()) {
179               return;
180             }
181     
182             Node fnNode = n.getLastChild();
183             String qualifiedFnName = qualifiedNameNode.getQualifiedName();
184             String fnName = qualifiedNameNode.getLastChild().getString();
185             if (fnNamesToInline.contains(fnName)) {
186               // Then store a reference to it.
187               fnsToInlineByQualifiedName.put(qualifiedFnName, fnNode);
188             }
189           }
190         }
191     
192         private class StaticCallInliner extends AbstractPostOrderCallback {
193           @Override
194           public void visit(NodeTraversal t, Node n, Node parent) {
195             // Only look at method calls
196             if (!n.isCall()) {
197               return;
198             }
199     
200             // ... that are fully qualified
201             Node qualifiedNameNode = n.getFirstChild();
202             if (!qualifiedNameNode.isGetProp() || !qualifiedNameNode.isQualifiedName()) {
203               return;
204             }
205     
206             // ... and that reference a function definition we want to inline
207             String qualifiedFnName = qualifiedNameNode.getQualifiedName();
208             String fnName = qualifiedNameNode.getLastChild().getString();
209             Node fnImpl = fnsToInlineByQualifiedName.get(qualifiedFnName);
210             if (fnImpl == null) {
211               return;
212             }
213     
214             // Otherwise inline the call.
215             Node inlinedCall =
216                 injector.inline(
217                     new Reference(n, t.getScope(), t.getModule(), inliningMode), fnName, fnImpl);
218             t.getCompiler().reportChangeToEnclosingScope(inlinedCall);
219           }
220         }
221       }
222     
223       public J2clPass(AbstractCompiler compiler) {
224         this.compiler = compiler;
225         this.safeNameIdSupplier = compiler.getUniqueNameIdSupplier();
226       }
227     
228       @Override
229       public void process(Node externs, Node root) {
230         /*
231          * Re-writes Util.getDefine to make it work for compiled mode.
232          */
233         Set<String> defines = new ProcessDefines(compiler, null).collectDefines(root).keySet();
234         NodeTraversal.traverseEs6(compiler, root, new GetDefineRewriter(defines));
235     
236         /*
237          * Inlines Arrays.$create(), Arrays.$init(), Arrays.$instanceIsOfType(), Arrays.$castTo() and
238          * Casts.to() so that all references to Object.$isInstance() functions will be fully qualified
239          * and easy to strip.
240          */
241         inlineFunctionsInFile(
242             root,
243             "vmbootstrap/Arrays.impl.js",
244             ImmutableSet.of("$create", "$init", "$instanceIsOfType", "$castTo"),
245             InliningMode.DIRECT);
246         inlineFunctionsInFile(
247             root,
248             "vmbootstrap/Casts.impl.js",
249             ImmutableSet.of("to"),
250             InliningMode.DIRECT);
251     
252         /*
253          * Inlines all Interface.$markImplementor(FooClass) metaclass calls so that FooClass and others
254          * like it are not unnecessarily retained and so that static analysis of interface instanceof
255          * calls becomes possible.
256          */
257         inlineFunctionsInFile(
258             root, ALL_CLASS_FILE_NAMES, ImmutableSet.of("$markImplementor"), InliningMode.BLOCK);
259     
260         /*
261          * Inlines class metadata calls so they become optimizable and avoids escaping of constructor.
262          */
263         inlineFunctionsInFile(
264             root,
265             "nativebootstrap/Util.impl.js",
266             ImmutableSet.of(
267                 "$setClassMetadata",
268                 "$setClassMetadataForInterface",
269                 "$setClassMetadataForEnum",
270                 "$setClassMetadataForPrimitive"),
271             InliningMode.BLOCK);
272     
273         NodeTraversal.traverseEs6(compiler, root, new NativeAliasInliner());
274       }
275     
276       private void inlineFunctionsInFile(
277           Node root, String classFileName, Set<String> fnNamesToInline, InliningMode inliningMode) {
278         new ClassStaticFunctionsInliner(root, classFileName, fnNamesToInline, inliningMode).run();
279       }
