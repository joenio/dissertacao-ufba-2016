17     package com.google.javascript.jscomp;
18     
19     import static com.google.javascript.jscomp.JsMessage.Style.RELAX;
20     
21     import com.google.javascript.jscomp.JsMessage.Style;
22     
23     /**
24      * Test which checks that replacer works correctly.
25      *
26      * @author tbreisacher@google.com (Tyler Breisacher)
27      */
28     public final class ReplaceMessagesForChromeTest extends CompilerTestCase {
29     
30       private Style style = RELAX;
31     
32       @Override
33       protected CompilerPass getProcessor(Compiler compiler) {
34         return new ReplaceMessagesForChrome(compiler,
35             new GoogleJsMessageIdGenerator(null), false, style);
36       }
37     
38       @Override
39       protected int getNumRepetitions() {
40         // No longer valid on the second run.
41         return 1;
42       }
43     
44       @Override
45       protected void setUp()  {
46         style = RELAX;
47         compareJsDoc = false;
48       }
49     
50       public void testReplaceSimpleMessage() {
51         test("/** @desc A simple message. */\n" +
52              "var MSG_A = goog.getMsg('Hello world');",
53              "var MSG_A=chrome.i18n.getMessage('8660696502365331902');");
54     
55         test("/** @desc A message attached to an object. */\n" +
56             "foo.bar.MSG_B = goog.getMsg('Goodbye world');",
57             "foo.bar.MSG_B=chrome.i18n.getMessage('2356086230621084760');");
58       }
59     
60       public void testReplaceSinglePlaceholder() {
61         test("/** @desc A message with one placeholder. */\n" +
62              "var MSG_C = goog.getMsg('Hello, {$name}', {name: 'Tyler'});",
63              "var MSG_C=chrome.i18n.getMessage('4985325380591528435', ['Tyler']);");
64       }
65     
66       public void testReplaceTwoPlaceholders() {
67         test("/** @desc A message with two placeholders. */\n" +
68              "var MSG_D = goog.getMsg('{$greeting}, {$name}', " +
69              "{greeting: 'Hi', name: 'Tyler'});",
70              "var MSG_D=chrome.i18n.getMessage('3605047247574980322', " +
71              "['Hi', 'Tyler']);");
72     
73         test("/** @desc A message with two placeholders, but their order is\n" +
74              " * reversed in the object literal. (Shouldn't make a difference.)\n" +
75              " */\n" +
76              "var MSG_E = goog.getMsg('{$greeting}, {$name}!', " +
77              "{name: 'Tyler', greeting: 'Hi'});",
78              "var MSG_E=chrome.i18n.getMessage('691522386483664339', " +
79              "['Hi', 'Tyler']);");
80       }
81     
82       public void testReplacePlaceholderMissingValue() {
83         testError("/** @desc A message with two placeholders, but one is missing. */\n" +
84              "var MSG_F = goog.getMsg('{$greeting}, {$name}!', {name: 'Tyler'});",
85              JsMessageVisitor.MESSAGE_TREE_MALFORMED);
86       }
87     
88       public void testReplaceTwoPlaceholdersNonAlphaOrder() {
89         test("/** @desc A message with two placeholders not in order .*/\n" +
90              "var MSG_G = goog.getMsg('{$name}: {$greeting}', " +
91              "{greeting: 'Salutations', name: 'Tyler'});",
92              "var MSG_G=chrome.i18n.getMessage('7437383242562773138', " +
93              "['Salutations', 'Tyler']);");
94       }
95     
96       public void testReplaceExternalMessage() {
97         test("/** @desc A message that was extracted with SoyMsgExtractor. */\n" +
98              "var MSG_EXTERNAL_1357902468 = goog.getMsg('Hello world');",
99              "var MSG_EXTERNAL_1357902468 = chrome.i18n.getMessage('1357902468');");
100       }
101     
102       /**
103        * Test that messages are handled correctly if they contain the same
104        * placeholder twice.
105        */
106       public void testReplaceMessageWithDuplicatePlaceholders() {
107         String original = "" +
108             "/** @desc A message that contains two instances of the same placeholder. */\n" +
109             "var MSG_EXTERNAL_987654321 = goog.getMsg(" +
110             "'{$startDiv_1}You are signed in as{$endDiv}{$img}{$startDiv_2}{$name}{$endDiv}'," +
111             "{'startDiv_1': '<div>'," +
112             "'endDiv': '</div>'," +
113             "'img': '<img src=\"http://example.com/photo.png\">'," +
114             "'startDiv_2': '<div class=\"name\">'," +
115             "'name': name});";
116     
117         String compiled = "" +
118             "var MSG_EXTERNAL_987654321 = chrome.i18n.getMessage('987654321', " +
119             "[" +
120             "'</div>', " +  // endDiv, only included once, even though it appears twice in the message.
121             "'<img src=\"http://example.com/photo.png\">', " +  // img
122             "name, " +  // name
123             "'<div>', " +  // startDiv_1
124             "'<div class=\"name\">'" +  // startDiv_2
125             "]);";
126     
127         test(original, compiled);
128       }
129     }
