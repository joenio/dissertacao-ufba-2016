41     package com.google.javascript.rhino;
42     
43     /**
44      * This is interface defines a protocol for the reporting of
45      * errors during JavaScript translation or execution.
46      *
47      */
48     
49     public interface ErrorReporter {
50     
51         /**
52          * Report a warning.
53          *
54          * The implementing class may choose to ignore the warning
55          * if it desires.
56          *
57          * @param message a String describing the warning
58          * @param sourceName a String describing the JavaScript source
59          * where the warning occurred; typically a filename or URL
60          * @param line the line number associated with the warning
61          * @param lineOffset the offset into lineSource where problem was detected
62          */
63         void warning(String message, String sourceName, int line, int lineOffset);
64     
65         /**
66          * Report an error.
67          *
68          * The implementing class is free to throw an exception if
69          * it desires.
70          *
71          * If execution has not yet begun, the JavaScript engine is
72          * free to find additional errors rather than terminating
73          * the translation. It will not execute a script that had
74          * errors, however.
75          *
76          * @param message a String describing the error
77          * @param sourceName a String describing the JavaScript source
78          * where the error occurred; typically a filename or URL
79          * @param line the line number associated with the error
80          * @param lineOffset the offset into lineSource where problem was detected
81          */
82         void error(String message, String sourceName, int line, int lineOffset);
83     
84     }
