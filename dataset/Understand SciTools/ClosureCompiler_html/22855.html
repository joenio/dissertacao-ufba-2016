49     /**
50      * Renames unrelated properties to the same name, using type information.
51      * This allows better compression as more properties can be given short names.
52      *
53      * <p>Properties are considered unrelated if they are never referenced from the
54      * same type or from a subtype of each others' types, thus this pass is only
55      * effective if type checking is enabled.
56      *
57      * Example:
58      * <code>
59      *   Foo.fooprop = 0;
60      *   Foo.fooprop2 = 0;
61      *   Bar.barprop = 0;
62      * </code>
63      *
64      * becomes:
65      *
66      * <code>
67      *   Foo.a = 0;
68      *   Foo.b = 0;
69      *   Bar.a = 0;
70      * </code>
71      *
72      */
73     class AmbiguateProperties implements CompilerPass {
74       private static final Logger logger = Logger.getLogger(
75           AmbiguateProperties.class.getName());
76

78

95       /**
96        * Sorts Property objects by their count, breaking ties alphabetically to
97        * ensure a deterministic total ordering.
98        */
99       private static final Comparator<Property> FREQUENCY_COMPARATOR =
100           new Comparator<Property>() {
101             @Override
102             public int compare(Property p1, Property p2) {
103               if (p1.numOccurrences != p2.numOccurrences) {
104                 return p2.numOccurrences - p1.numOccurrences;
105               }
106               return p1.oldName.compareTo(p2.oldName);
107             }
108           };

128       AmbiguateProperties(AbstractCompiler compiler,
129           char[] reservedCharacters) {
130         Preconditions.checkState(compiler.getLifeCycleStage().isNormalized());
131         this.compiler = compiler;
132         this.reservedCharacters = reservedCharacters;
133     
134         JSTypeRegistry r = compiler.getTypeRegistry();
135         invalidatingTypes = new HashSet<>(ImmutableSet.of(
136             r.getNativeType(JSTypeNative.ALL_TYPE),
137             r.getNativeType(JSTypeNative.FUNCTION_FUNCTION_TYPE),
138             r.getNativeType(JSTypeNative.FUNCTION_INSTANCE_TYPE),
139             r.getNativeType(JSTypeNative.FUNCTION_PROTOTYPE),
140             r.getNativeType(JSTypeNative.GLOBAL_THIS),
141             r.getNativeType(JSTypeNative.OBJECT_TYPE),
142             r.getNativeType(JSTypeNative.OBJECT_PROTOTYPE),
143             r.getNativeType(JSTypeNative.OBJECT_FUNCTION_TYPE),
144             r.getNativeType(JSTypeNative.TOP_LEVEL_PROTOTYPE)));
145     
146         for (TypeMismatch mis : compiler.getTypeMismatches()) {
147           addInvalidatingType(mis.typeA);
148           addInvalidatingType(mis.typeB);
149         }
150     
151         for (TypeMismatch mis : compiler.getImplicitInterfaceUses()) {
152           addInvalidatingType(mis.typeA);
153           addInvalidatingType(mis.typeB);
154         }
155     
156         externedNames = compiler.getExternProperties();
157       }

167       /**
168        * Invalidates the given type, so that no properties on it will be renamed.
169        */
170       private void addInvalidatingType(JSType type) {
171         type = type.restrictByNotNullOrUndefined();
172         if (type.isUnionType()) {
173           for (JSType alt : type.toMaybeUnionType().getAlternatesWithoutStructuralTyping()) {
174             addInvalidatingType(alt);
175           }
176         }
177     
178         invalidatingTypes.add(type);
179         ObjectType objType = ObjectType.cast(type);
180         if (objType != null && objType.isInstanceType()) {
181           invalidatingTypes.add(objType.getImplicitPrototype());
182         }
183       }

190       /** Returns an integer that uniquely identifies a JSType. */
191       private int getIntForType(JSType type) {
192         // Templatized types don't exist at runtime, so collapse to raw type
193         if (type != null && type.isTemplatizedType()) {
194           type = type.toMaybeTemplatizedType().getReferencedType();
195         }
196         if (intForType.containsKey(type)) {
197           return intForType.get(type).intValue();
198         }
199         int newInt = intForType.size() + 1;
200         intForType.put(type, newInt);
201         return newInt;
202       }

204       @Override
205       public void process(Node externs, Node root) {
206         // Find all property references and record the types on which they occur.
207         // Populate stringNodesToRename, propertyMap, quotedNames.
208         NodeTraversal.traverseEs6(compiler, root, new ProcessProperties());
209     
210         ImmutableSet.Builder<String> reservedNames = ImmutableSet.<String>builder()
211             .addAll(externedNames)
212             .addAll(quotedNames);
213     
214         int numRenamedPropertyNames = 0;
215         int numSkippedPropertyNames = 0;
216         ArrayList<PropertyGraphNode> nodes = new ArrayList<>(propertyMap.size());
217         for (Property prop : propertyMap.values()) {
218           if (prop.skipAmbiguating) {
219             ++numSkippedPropertyNames;
220             reservedNames.add(prop.oldName);
221           } else {
222             ++numRenamedPropertyNames;
223             nodes.add(new PropertyGraphNode(prop));
224           }
225         }
226     
227         PropertyGraph graph = new PropertyGraph(nodes);
228         GraphColoring<Property, Void> coloring =
229             new GreedyGraphColoring<>(graph, FREQUENCY_COMPARATOR);
230         int numNewPropertyNames = coloring.color();
231     
232         // Generate new names for the properties that will be renamed.
233         NameGenerator nameGen = new DefaultNameGenerator(
234             reservedNames.build(), "", reservedCharacters);
235         String[] colorMap = new String[numNewPropertyNames];
236         for (int i = 0; i < numNewPropertyNames; ++i) {
237           colorMap[i] = nameGen.generateNextName();
238         }
239     
240         // Translate the color of each Property instance to a name.
241         for (PropertyGraphNode node : graph.getNodes()) {
242           node.getValue().newName = colorMap[node.getAnnotation().hashCode()];
243           if (renamingMap != null) {
244             renamingMap.put(node.getValue().oldName, node.getValue().newName);
245           }
246         }
247     
248         // Actually assign the new names to the relevant STRING nodes in the AST.
249         for (Node n : stringNodesToRename) {
250           String oldName = n.getString();
251           Property p = propertyMap.get(oldName);
252           if (p != null && p.newName != null) {
253             Preconditions.checkState(oldName.equals(p.oldName));
254             if (!p.newName.equals(oldName)) {
255               n.setString(p.newName);
256               compiler.reportCodeChange();
257             }
258           }
259         }
260     
261         logger.fine("Collapsed " + numRenamedPropertyNames + " properties into "
262                     + numNewPropertyNames + " and skipped renaming "
263                     + numSkippedPropertyNames + " properties.");
264       }

266       private BitSet getRelatedTypesOnNonUnion(JSType type) {
267         // All of the types we encounter should have been added to the
268         // relatedBitsets via computeRelatedTypes.
269         if (relatedBitsets.containsKey(type)) {
270           return relatedBitsets.get(type);
271         } else {
272           throw new RuntimeException("Related types should have been computed for"
273                                      + " type: " + type + " but have not been.");
274         }
275       }

277       /**
278        * Adds subtypes - and implementors, in the case of interfaces - of the type
279        * to its JSTypeBitSet of related types. Union types are decomposed into their
280        * alternative types.
281        *
282        * <p>The 'is related to' relationship is best understood graphically. Draw an
283        * arrow from each instance type to the prototype of each of its
284        * subclass. Draw an arrow from each prototype to its instance type. Draw an
285        * arrow from each interface to its implementors. A type is related to another
286        * if there is a directed path in the graph from the type to other. Thus, the
287        * 'is related to' relationship is reflexive and transitive.
288        *
289        * <p>Example with Foo extends Bar which extends Baz and Bar implements I:
290        * <pre>
291        * Foo -> Bar.prototype -> Bar -> Baz.prototype -> Baz
292        *                          ^
293        *                          |
294        *                          I
295        * </pre>
296        *
297        * <p>Note that we don't need to correctly handle the relationships between
298        * functions, because the function type is invalidating (i.e. its properties
299        * won't be ambiguated).
300        */
301       private void computeRelatedTypes(JSType type) {
302         if (type.isUnionType()) {
303           type = type.restrictByNotNullOrUndefined();
304           if (type.isUnionType()) {
305             for (JSType alt : type.toMaybeUnionType().getAlternates()) {
306               computeRelatedTypes(alt);
307             }
308             return;
309           }
310         }
311     
312         if (relatedBitsets.containsKey(type)) {
313           // We only need to generate the bit set once.
314           return;
315         }
316     
317         JSTypeBitSet related = new JSTypeBitSet(intForType.size());
318         relatedBitsets.put(type, related);
319         related.set(getIntForType(type));
320     
321         // A prototype is related to its instance.
322         if (type.isFunctionPrototypeType()) {
323           addRelatedInstance(((ObjectType) type).getOwnerFunction(), related);
324           return;
325         }
326     
327         // An instance is related to its subclasses.
328         FunctionType constructor = type.toObjectType().getConstructor();
329         if (constructor != null && constructor.getSubTypes() != null) {
330           for (FunctionType subType : constructor.getSubTypes()) {
331             addRelatedInstance(subType, related);
332           }
333         }
334     
335         // An interface is related to its implementors.
336         for (FunctionType implementor : compiler.getTypeRegistry()
337             .getDirectImplementors(type.toObjectType())) {
338           addRelatedInstance(implementor, related);
339         }
340       }
341     
342       /**
343        * Adds the instance of the given constructor, its implicit prototype and all
344        * its related types to the given bit set.
345        */
346       private void addRelatedInstance(
347           FunctionType constructor, JSTypeBitSet related) {
348         // TODO(user): A constructor which doesn't have an instance type
349         // (e.g. it's missing the @constructor annotation) should be an invalidating
350         // type which doesn't reach this code path.
351         if (constructor.hasInstanceType()) {
352           ObjectType instanceType = constructor.getInstanceType();
353           related.set(getIntForType(instanceType.getImplicitPrototype()));
354           computeRelatedTypes(instanceType);
355           related.or(relatedBitsets.get(instanceType));
356         }
357       }

395        * A {@link SubGraph} that represents properties. The related types of
396        * the properties are used to efficiently calculate adjacency information.
397        */
398       class PropertySubGraph implements SubGraph<Property, Void> {
399         /** Types related to properties referenced in this subgraph. */
400         JSTypeBitSet relatedTypes = new JSTypeBitSet(intForType.size());
401     
402         /**
403          * Returns true if prop is in an independent set from all properties in this
404          * sub graph.  That is, if none of its related types intersects with the
405          * related types for this sub graph.
406          */
407         @Override
408         public boolean isIndependentOf(Property prop) {
409           return !relatedTypes.intersects(prop.relatedTypes);
410         }
411     
412         /**
413          * Adds the node to the sub graph, adding all its related types to the
414          * related types for the sub graph.
415          */
416         @Override
417         public void addNode(Property prop) {
418           relatedTypes.or(prop.relatedTypes);
419         }
420       }

446       /** Finds all property references, recording the types on which they occur. */
447       private class ProcessProperties extends AbstractPostOrderCallback {
448         @Override
449         public void visit(NodeTraversal t, Node n, Node parent) {
450           switch (n.getType()) {
451             case GETPROP: {
452               Node propNode = n.getSecondChild();
453               JSType jstype = getJSType(n.getFirstChild());
454               maybeMarkCandidate(propNode, jstype);
455               break;
456             }
457             case CALL: {
458               Node target = n.getFirstChild();
459                 if (!target.isQualifiedName()) {
460                 break;
461               }
462     
463                 String renameFunctionName = target.getOriginalQualifiedName();
464                 if (renameFunctionName != null
465                     && compiler.getCodingConvention().isPropertyRenameFunction(renameFunctionName)) {
466     
467                   if (n.getChildCount() != 2 && n.getChildCount() != 3) {
468                     compiler.report(
469                         JSError.make(
470                             n,
471                             DisambiguateProperties.Warnings.INVALID_RENAME_FUNCTION,
472                             renameFunctionName,
473                             " Must be called with 1 or 2 arguments."));
474                     break;
475                   }
476     
477                   Node propName = n.getSecondChild();
478                   if (!propName.isString()) {
479                     compiler.report(
480                         JSError.make(
481                             n,
482                             DisambiguateProperties.Warnings.INVALID_RENAME_FUNCTION,
483                             renameFunctionName,
484                             " The first argument must be a string literal."));
485                     break;
486                   }
487     
488                   if (propName.getString().contains(".")) {
489                     compiler.report(
490                         JSError.make(
491                             n,
492                             DisambiguateProperties.Warnings.INVALID_RENAME_FUNCTION,
493                             renameFunctionName,
494                             " The first argument must not be a property path."));
495                     break;
496                   }
497     
498                   JSType jstype = getJSType(n.getSecondChild());
499     
500                   maybeMarkCandidate(propName, jstype);
501                 } else if (NodeUtil.isObjectDefinePropertiesDefinition(n)) {
502                   Node typeObj = n.getSecondChild();
503                   JSType jstype = getJSType(typeObj);
504                   Node objectLiteral = typeObj.getNext();
505     
506                   if (!objectLiteral.isObjectLit()) {
507                     break;
508                   }
509     
510                   for (Node key : objectLiteral.children()) {
511                     if (key.isQuotedString()) {
512                       quotedNames.add(key.getString());
513                     } else {
514                       maybeMarkCandidate(key, jstype);
515                     }
516                   }
517                 }
518               break;
519             }
520             case OBJECTLIT:
521               // Object.defineProperties literals are handled at the CALL node.
522               if (n.getParent().isCall()
523                   && NodeUtil.isObjectDefinePropertiesDefinition(n.getParent())) {
524                 break;
525               }
526     
527               // The children of an OBJECTLIT node are keys, where the values
528               // are the children of the keys.
529               for (Node key = n.getFirstChild(); key != null; key = key.getNext()) {
530                 // We only want keys that were unquoted.
531                 // Keys are STRING, GET, SET
532                 if (!key.isQuotedString()) {
533                   JSType jstype = getJSType(n.getFirstChild());
534                   maybeMarkCandidate(key, jstype);
535                 } else {
536                   // Ensure that we never rename some other property in a way
537                   // that could conflict with this quoted key.
538                   quotedNames.add(key.getString());
539                 }
540               }
541               break;
542             case GETELEM:
543               // If this is a quoted property access (e.g. x['myprop']), we need to
544               // ensure that we never rename some other property in a way that
545               // could conflict with this quoted name.
546               Node child = n.getLastChild();
547               if (child.isString()) {
548                 quotedNames.add(child.getString());
549               }
550               break;
551           }
552         }

554         /**
555          * If a property node is eligible for renaming, stashes a reference to it
556          * and increments the property name's access count.
557          *
558          * @param n The STRING node for a property
559          */
560         private void maybeMarkCandidate(Node n, JSType type) {
561           String name = n.getString();
562           if (!externedNames.contains(name)) {
563             stringNodesToRename.add(n);
564             recordProperty(name, type);
565           }
566         }
567     
568         private Property recordProperty(String name, JSType type) {
569           Property prop = getProperty(name);
570           prop.addType(type);
571           return prop;
572         }

575       /** Returns true if properties on this type should not be renamed. */
576       private boolean isInvalidatingType(JSType type) {
577         if (type.isUnionType()) {
578           type = type.restrictByNotNullOrUndefined();
579           if (type.isUnionType()) {
580             for (JSType alt : type.toMaybeUnionType().getAlternates()) {
581               if (isInvalidatingType(alt)) {
582                 return true;
583               }
584             }
585             return false;
586           }
587         }
588         ObjectType objType = ObjectType.cast(type);
589         return objType == null
590             || invalidatingTypes.contains(objType)
591             || !objType.hasReferenceName()
592             || objType.isUnknownType()
593             || objType.isEmptyType() /* unresolved types */
594             || objType.isEnumType()
595             || objType.autoboxesTo() != null;
596       }

598       private Property getProperty(String name) {
599         Property prop = propertyMap.get(name);
600         if (prop == null) {
601           prop = new Property(name);
602           propertyMap.put(name, prop);
603         }
604         return prop;
605       }

607       /**
608        * This method gets the JSType from the Node argument and verifies that it is
609        * present.
610        */
611       private JSType getJSType(Node n) {
612         if (n == null) {
613           return compiler.getTypeRegistry().getNativeType(JSTypeNative.UNKNOWN_TYPE);
614         }
615     
616         JSType jsType = n.getJSType();
617         if (jsType == null) {
618           // TODO(user): This branch indicates a compiler bug, not worthy of
619           // halting the compilation but we should log this and analyze to track
620           // down why it happens. This is not critical and will be resolved over
621           // time as the type checker is extended.
622           return compiler.getTypeRegistry().getNativeType(JSTypeNative.UNKNOWN_TYPE);
623         } else {
624           return jsType;
625         }
626       }

628       /** Encapsulates the information needed for renaming a property. */
629       private class Property {
630         final String oldName;
631         String newName;
632         int numOccurrences;
633         boolean skipAmbiguating;
634         JSTypeBitSet relatedTypes = new JSTypeBitSet(intForType.size());
635     
636         Property(String name) {
637           this.oldName = name;
638     
639           // Properties with this suffix are handled in RenameProperties.
640           if (name.startsWith(SKIP_PREFIX)) {
641             skipAmbiguating = true;
642           }
643         }
644     
645         /** Add this type to this property, calculating */
646         void addType(JSType newType) {
647           if (skipAmbiguating) {
648             return;
649           }
650     
651           ++numOccurrences;
652     
653           if (newType.isUnionType()) {
654             newType = newType.restrictByNotNullOrUndefined();
655             if (newType.isUnionType()) {
656               for (JSType alt : newType.toMaybeUnionType().getAlternatesWithoutStructuralTyping()) {
657                 addNonUnionType(alt);
658               }
659               return;
660             }
661           }
662           addNonUnionType(newType);
663         }
664     
665         private void addNonUnionType(JSType newType) {
666           if (skipAmbiguating || isInvalidatingType(newType)) {
667             skipAmbiguating = true;
668             return;
669           }
670     
671           if (!relatedTypes.get(getIntForType(newType))) {
672             computeRelatedTypes(newType);
673             relatedTypes.or(getRelatedTypesOnNonUnion(newType));
674           }
675         }
676       }

678       // A BitSet that stores type info. Adds pretty-print routines.
679       private class JSTypeBitSet extends BitSet {
680         private static final long serialVersionUID = 1L;
681     
682         private JSTypeBitSet(int size) {
683           super(size);
684         }
685     
686         private JSTypeBitSet() {
687           super();
688         }
689     
690         /**
691          * Pretty-printing, for diagnostic purposes.
692          */
693         @Override
694         public String toString() {
695           int from = 0;
696           int current = 0;
697           List<String> types = new ArrayList<>();
698           while (-1 != (current = nextSetBit(from))) {
699             types.add(intForType.inverse().get(current).toString());
700             from = current + 1;
701           }
702           return Joiner.on(" && ").join(types);
703         }
704       }
