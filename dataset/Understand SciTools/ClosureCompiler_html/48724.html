33     /**
34      * Computes "may be" reaching use for all definitions of each variables.
35      *
36      * A use of {@code A} in {@code alert(A)} is a "may be" reaching use of
37      * the definition of {@code A} at {@code A = foo()} if at least one path from
38      * the use node reaches that definition and it is the last definition before
39      * the use on that path.
40      *
41      */
42     class MaybeReachingVariableUse extends
43         DataFlowAnalysis<Node, MaybeReachingVariableUse.ReachingUses> {
44     
45       // The scope of the function that we are analyzing.
46       private final Scope jsScope;
47       private final Set<Var> escaped;
48     
49       MaybeReachingVariableUse(
50           ControlFlowGraph<Node> cfg, Scope jsScope, AbstractCompiler compiler) {
51         super(cfg, new ReachingUsesJoinOp());
52         this.jsScope = jsScope;
53         this.escaped = new HashSet<>();
54     
55         // TODO(user): Maybe compute it somewhere else and re-use the escape
56         // local set here.
57         computeEscaped(jsScope, escaped, compiler);
58       }

60       /**
61        * May use definition lattice representation. It captures a product
62        * lattice for each local (non-escaped) variable. The sub-lattice is
63        * a n + 2 power set element lattice with all the Nodes in the program,
64        * TOP and BOTTOM. This is better explained with an example:
65        *
66        * Consider: A sub-lattice element representing the variable A represented
67        * by { N_4, N_5} where N_x is a Node in the program. This implies at
68        * that particular point in the program the content of A is "upward exposed"
69        * at point N_4 and N_5.
70        *
71        * Example:
72        *
73        * A = 1;
74        * ...
75        * N_3:
76        * N_4: print(A);
77        * N_5: y = A;
78        * N_6: A = 1;
79        * N_7: print(A);
80        *
81        * At N_3, reads of A in {N_4, N_5} are said to be upward exposed.
82        */
83       static final class ReachingUses implements LatticeElement {
84         final Multimap<Var, Node> mayUseMap;
85     
86         public ReachingUses() {
87           mayUseMap = HashMultimap.create();
88         }
89     
90         /**
91          * Copy constructor.
92          *
93          * @param other The constructed object is a replicated copy of this element.
94          */
95         public ReachingUses(ReachingUses other) {
96           mayUseMap = HashMultimap.create(other.mayUseMap);
97         }
98     
99         @Override
100         public boolean equals(Object other) {
101           return (other instanceof ReachingUses) &&
102               ((ReachingUses) other).mayUseMap.equals(this.mayUseMap);
103         }
104     
105         @Override
106         public int hashCode() {
107           return mayUseMap.hashCode();
108         }
109       }
110     
111       /**
112        * The join is a simple union because of the "may be" nature of the analysis.
113        *
114        * Consider: A = 1; if (x) { A = 2 }; alert(A);
115        *
116        * The read of A "may be" exposed to A = 1 in the beginning.
117        */
118       private static class ReachingUsesJoinOp implements JoinOp<ReachingUses> {
119         @Override
120         public ReachingUses apply(List<ReachingUses> from) {
121           ReachingUses result = new ReachingUses();
122           for (ReachingUses uses : from) {
123             result.mayUseMap.putAll(uses.mayUseMap);
124           }
125           return result;
126         }
127       }
128     
129       @Override
130       boolean isForward() {
131         return false;
132       }
133     
134       @Override
135       ReachingUses createEntryLattice() {
136         return new ReachingUses();
137       }
138     
139       @Override
140       ReachingUses createInitialEstimateLattice() {
141         return new ReachingUses();
142       }
143     
144       @Override
145       ReachingUses flowThrough(Node n, ReachingUses input) {
146         ReachingUses output = new ReachingUses(input);
147     
148         // If there's an ON_EX edge, this cfgNode may or may not get executed.
149         // We can express this concisely by just pretending this happens in
150         // a conditional.
151         boolean conditional = hasExceptionHandler(n);
152         computeMayUse(n, n, output, conditional);
153     
154         return output;
155       }
156     
157       private boolean hasExceptionHandler(Node cfgNode) {
158         List<DiGraphEdge<Node, Branch>> branchEdges = getCfg().getOutEdges(cfgNode);
159         for (DiGraphEdge<Node, Branch> edge : branchEdges) {
160           if (edge.getValue() == Branch.ON_EX) {
161             return true;
162           }
163         }
164         return false;
165       }
166     
167       private void computeMayUse(
168           Node n, Node cfgNode, ReachingUses output, boolean conditional) {
169         switch (n.getType()) {
170     
171           case BLOCK:
172           case FUNCTION:
173             return;
174     
175           case NAME:
176             addToUseIfLocal(n.getString(), cfgNode, output);
177             return;
178     
179           case WHILE:
180           case DO:
181           case IF:
182             computeMayUse(
183                 NodeUtil.getConditionExpression(n), cfgNode, output, conditional);
184             return;
185     
186           case FOR:
187             if (!NodeUtil.isForIn(n)) {
188               computeMayUse(
189                   NodeUtil.getConditionExpression(n), cfgNode, output, conditional);
190             } else {
191               // for(x in y) {...}
192               Node lhs = n.getFirstChild();
193               Node rhs = lhs.getNext();
194               if (lhs.isVar()) {
195                 lhs = lhs.getLastChild(); // for(var x in y) {...}
196               }
197               if (lhs.isName() && !conditional) {
198                 removeFromUseIfLocal(lhs.getString(), output);
199               }
200               computeMayUse(rhs, cfgNode, output, conditional);
201             }
202             return;
203     
204           case AND:
205           case OR:
206             computeMayUse(n.getLastChild(), cfgNode, output, true);
207             computeMayUse(n.getFirstChild(), cfgNode, output, conditional);
208             return;
209     
210           case HOOK:
211             computeMayUse(n.getLastChild(), cfgNode, output, true);
212             computeMayUse(n.getSecondChild(), cfgNode, output, true);
213             computeMayUse(n.getFirstChild(), cfgNode, output, conditional);
214             return;
215     
216           case VAR:
217             Node varName = n.getFirstChild();
218             Preconditions.checkState(n.hasChildren(), "AST should be normalized", n);
219     
220             if (varName.hasChildren()) {
221               computeMayUse(varName.getFirstChild(), cfgNode, output, conditional);
222               if (!conditional) {
223                 removeFromUseIfLocal(varName.getString(), output);
224               }
225             }
226             return;
227     
228           default:
229             if (NodeUtil.isAssignmentOp(n) && n.getFirstChild().isName()) {
230               Node name = n.getFirstChild();
231               if (!conditional) {
232                 removeFromUseIfLocal(name.getString(), output);
233               }
234     
235               // In case of a += "Hello". There is a read of a.
236               if (!n.isAssign()) {
237                 addToUseIfLocal(name.getString(), cfgNode, output);
238               }
239     
240               computeMayUse(name.getNext(), cfgNode, output, conditional);
241             } else {
242               /*
243                * We want to traverse in reverse order because we want the LAST
244                * definition in the sub-tree....
245                * But we have no better way to traverse in reverse other :'(
246                */
247               for (Node c = n.getLastChild(); c != null; c = n.getChildBefore(c)) {
248                 computeMayUse(c, cfgNode, output, conditional);
249               }
250             }
251         }
252       }

254       /**
255        * Sets the variable for the given name to the node value in the upward
256        * exposed lattice. Do nothing if the variable name is one of the escaped
257        * variable.
258        */
259       private void addToUseIfLocal(String name, Node node, ReachingUses use) {
260         Var var = jsScope.getVar(name);
261         if (var == null || var.scope != jsScope) {
262           return;
263         }
264         if (!escaped.contains(var)) {
265           use.mayUseMap.put(var, node);
266         }
267       }
268     
269       /**
270        * Removes the variable for the given name from the node value in the upward
271        * exposed lattice. Do nothing if the variable name is one of the escaped
272        * variable.
273        */
274       private void removeFromUseIfLocal(String name, ReachingUses use) {
275         Var var = jsScope.getVar(name);
276         if (var == null || var.scope != jsScope) {
277           return;
278         }
279         if (!escaped.contains(var)) {
280           use.mayUseMap.removeAll(var);
281         }
282       }

284       /**
285        * Gets a list of nodes that may be using the value assigned to {@code name}
286        * in {@code defNode}. {@code defNode} must be one of the control flow graph
287        * nodes.
288        *
289        * @param name name of the variable. It can only be names of local variable
290        *     that are not function parameters, escaped variables or variables
291        *     declared in catch.
292        * @param defNode The list of upward exposed use for the variable.
293        */
294       Collection<Node> getUses(String name, Node defNode) {
295         GraphNode<Node, Branch> n = getCfg().getNode(defNode);
296         Preconditions.checkNotNull(n);
297         FlowState<ReachingUses> state = n.getAnnotation();
298         return state.getOut().mayUseMap.get(jsScope.getVar(name));
299       }
