32     /**
33      * Rewrites a given call to Polymer({}) to a set of declarations and assignments which can be
34      * understood by the compiler.
35      */
36     final class PolymerClassRewriter {
37     
38       private final AbstractCompiler compiler;
39     
40       private Node polymerElementExterns;
41     
42       PolymerClassRewriter(AbstractCompiler compiler, Node polymerElementExterns) {
43         this.compiler = compiler;
44         this.polymerElementExterns = polymerElementExterns;
45       }

47       /**
48        * Rewrites a given call to Polymer({}) to a set of declarations and assignments which can be
49        * understood by the compiler.
50        *
51        * @param exprRoot The root expression of the call to Polymer({}).
52        * @param cls The extracted {@link PolymerClassDefinition} for the Polymer element created by this
53        *     call.
54        */
55       void rewritePolymerClass(
56           Node exprRoot, final PolymerClassDefinition cls, boolean isInGlobalScope) {
57         Node call = exprRoot.getFirstChild();
58         if (call.isAssign()) {
59           call = call.getSecondChild();
60         } else if (call.isName()) {
61           call = call.getFirstChild();
62         }
63     
64         Node objLit = cls.descriptor;
65         if (hasShorthandAssignment(objLit)) {
66           compiler.report(JSError.make(objLit, PolymerPassErrors.POLYMER_SHORTHAND_NOT_SUPPORTED));
67           return;
68         }
69     
70         // Add {@code @lends} to the object literal.
71         JSDocInfoBuilder objLitDoc = new JSDocInfoBuilder(true);
72         objLitDoc.recordLends(cls.target.getQualifiedName() + ".prototype");
73         objLit.setJSDocInfo(objLitDoc.build());
74     
75         addTypesToFunctions(objLit, cls.target.getQualifiedName());
76         PolymerPassStaticUtils.switchDollarSignPropsToBrackets(objLit, compiler);
77         PolymerPassStaticUtils.quoteListenerAndHostAttributeKeys(objLit);
78     
79         for (MemberDefinition prop : cls.props) {
80           if (prop.value.isObjectLit()) {
81             PolymerPassStaticUtils.switchDollarSignPropsToBrackets(prop.value, compiler);
82           }
83         }
84     
85         // For simplicity add everything into a block, before adding it to the AST.
86         Node block = IR.block();
87     
88         JSDocInfoBuilder constructorDoc = this.getConstructorDoc(cls);
89     
90         // Remove the original constructor JS docs from the objlit.
91         Node ctorKey = cls.constructor.value.getParent();
92         if (ctorKey != null) {
93           ctorKey.removeProp(Node.JSDOC_INFO_PROP);
94         }
95     
96         if (cls.target.isGetProp()) {
97           // foo.bar = Polymer({...});
98           Node assign = IR.assign(
99               cls.target.cloneTree(),
100               cls.constructor.value.cloneTree());
101           assign.setJSDocInfo(constructorDoc.build());
102           Node exprResult = IR.exprResult(assign);
103           block.addChildToBack(exprResult);
104         } else {
105           // var foo = Polymer({...}); OR Polymer({...});
106           Node var = IR.var(cls.target.cloneTree(), cls.constructor.value.cloneTree());
107           var.setJSDocInfo(constructorDoc.build());
108           block.addChildToBack(var);
109         }
110     
111         appendPropertiesToBlock(cls, block, cls.target.getQualifiedName() + ".prototype.");
112         appendBehaviorMembersToBlock(cls, block);
113         ImmutableList<MemberDefinition> readOnlyProps = parseReadOnlyProperties(cls, block);
114         addInterfaceExterns(cls, readOnlyProps);
115         removePropertyDocs(objLit);
116     
117         block.useSourceInfoIfMissingFromForTree(exprRoot);
118         Node statements = block.removeChildren();
119         Node parent = exprRoot.getParent();
120     
121         // If the call to Polymer() is not in the global scope and the assignment target is not
122         // namespaced (which likely means it's exported to the global scope), put the type declaration
123         // into the global scope at the start of the current script.
124         //
125         // This avoids unknown type warnings which are a result of the compiler's poor understanding of
126         // types declared inside IIFEs or any non-global scope. We should revisit this decision after
127         // moving to the new type inference system which should be able to infer these types better.
128         if (!isInGlobalScope && !cls.target.isGetProp()) {
129           Node scriptNode = NodeUtil.getEnclosingScript(exprRoot);
130           scriptNode.addChildrenToFront(statements);
131         } else {
132           Node beforeRoot = parent.getChildBefore(exprRoot);
133           if (beforeRoot == null) {
134             parent.addChildrenToFront(statements);
135           } else {
136             parent.addChildrenAfter(statements, beforeRoot);
137           }
138         }
139     
140         if (NodeUtil.isNameDeclaration(exprRoot)) {
141           Node assignExpr = varToAssign(exprRoot);
142           parent.replaceChild(exprRoot, assignExpr);
143         }
144     
145         compiler.reportCodeChange();
146       }

148       /**
149        * Adds an @this annotation to all functions in the objLit.
150        */
151       private void addTypesToFunctions(Node objLit, String thisType) {
152         Preconditions.checkState(objLit.isObjectLit());
153         for (Node keyNode : objLit.children()) {
154           Node value = keyNode.getLastChild();
155           if (value != null && value.isFunction()) {
156             JSDocInfoBuilder fnDoc = JSDocInfoBuilder.maybeCopyFrom(keyNode.getJSDocInfo());
157             fnDoc.recordThisType(new JSTypeExpression(
158                 new Node(Token.BANG, IR.string(thisType)), PolymerPass.VIRTUAL_FILE));
159             keyNode.setJSDocInfo(fnDoc.build());
160           }
161         }
162     
163         // Add @this and @return to default property values.
164         for (MemberDefinition property : PolymerPassStaticUtils.extractProperties(objLit)) {
165           if (!property.value.isObjectLit()) {
166             continue;
167           }
168           if (hasShorthandAssignment(property.value)){
169             compiler.report(
170                 JSError.make(property.value, PolymerPassErrors.POLYMER_SHORTHAND_NOT_SUPPORTED));
171             return;
172           }
173     
174           Node defaultValue = NodeUtil.getFirstPropMatchingKey(property.value, "value");
175           if (defaultValue == null || !defaultValue.isFunction()) {
176             continue;
177           }
178           Node defaultValueKey = defaultValue.getParent();
179           JSDocInfoBuilder fnDoc = JSDocInfoBuilder.maybeCopyFrom(defaultValueKey.getJSDocInfo());
180           fnDoc.recordThisType(new JSTypeExpression(
181               new Node(Token.BANG, IR.string(thisType)), PolymerPass.VIRTUAL_FILE));
182           fnDoc.recordReturnType(PolymerPassStaticUtils.getTypeFromProperty(property, compiler));
183           defaultValueKey.setJSDocInfo(fnDoc.build());
184         }
185       }

187       /**
188        * Generates the _set* setters for readonly properties and appends them to the given block.
189        * @return A List of all readonly properties.
190        */
191       private ImmutableList<MemberDefinition> parseReadOnlyProperties(
192           final PolymerClassDefinition cls, Node block) {
193         String qualifiedPath = cls.target.getQualifiedName() + ".prototype.";
194         ImmutableList.Builder<MemberDefinition> readOnlyProps = ImmutableList.builder();
195     
196         for (MemberDefinition prop : cls.props) {
197           // Generate the setter for readOnly properties.
198           if (prop.value.isObjectLit()) {
199             Node readOnlyValue = NodeUtil.getFirstPropMatchingKey(prop.value, "readOnly");
200             if (readOnlyValue != null && readOnlyValue.isTrue()) {
201               block.addChildToBack(makeReadOnlySetter(prop.name.getString(), qualifiedPath));
202               readOnlyProps.add(prop);
203             }
204           }
205         }
206     
207         return readOnlyProps.build();
208       }

210       /**
211        * @return The proper constructor doc for the Polymer call.
212        */
213       private JSDocInfoBuilder getConstructorDoc(final PolymerClassDefinition cls) {
214         JSDocInfoBuilder constructorDoc = JSDocInfoBuilder.maybeCopyFrom(cls.constructor.info);
215         constructorDoc.recordConstructor();
216     
217         JSTypeExpression baseType = new JSTypeExpression(
218             new Node(Token.BANG, IR.string(PolymerPassStaticUtils.getPolymerElementType(cls))),
219             PolymerPass.VIRTUAL_FILE);
220         constructorDoc.recordBaseType(baseType);
221     
222         String interfaceName = getInterfaceName(cls);
223         JSTypeExpression interfaceType = new JSTypeExpression(
224             new Node(Token.BANG, IR.string(interfaceName)), PolymerPass.VIRTUAL_FILE);
225         constructorDoc.recordImplementedInterface(interfaceType);
226     
227         return constructorDoc;
228       }

230       /**
231        * Appends all properties in the ClassDefinition to the prototype of the custom element.
232        */
233       private void appendPropertiesToBlock(
234           final PolymerClassDefinition cls, Node block, String basePath) {
235         for (MemberDefinition prop : cls.props) {
236           Node propertyNode = IR.exprResult(
237               NodeUtil.newQName(compiler, basePath + prop.name.getString()));
238           JSDocInfoBuilder info = JSDocInfoBuilder.maybeCopyFrom(prop.info);
239     
240           JSTypeExpression propType = PolymerPassStaticUtils.getTypeFromProperty(prop, compiler);
241           if (propType == null) {
242             return;
243           }
244           info.recordType(propType);
245           propertyNode.getFirstChild().setJSDocInfo(info.build());
246     
247           block.addChildToBack(propertyNode);
248         }
249       }
250     
251       /**
252        * Remove all JSDocs from properties of a class definition
253        */
254       private void removePropertyDocs(final Node objLit) {
255         for (MemberDefinition prop : PolymerPassStaticUtils.extractProperties(objLit)) {
256           prop.name.removeProp(Node.JSDOC_INFO_PROP);
257         }
258       }

260       /**
261        * Appends all required behavior functions and non-property members to the given block.
262        */
263       private void appendBehaviorMembersToBlock(final PolymerClassDefinition cls, Node block) {
264         String qualifiedPath = cls.target.getQualifiedName() + ".prototype.";
265         Map<String, Node> nameToExprResult = new HashMap<>();
266         for (BehaviorDefinition behavior : cls.behaviors) {
267           for (MemberDefinition behaviorFunction : behavior.functionsToCopy) {
268             String fnName = behaviorFunction.name.getString();
269             // Don't copy functions already defined by the element itself.
270             if (NodeUtil.getFirstPropMatchingKey(cls.descriptor, fnName) != null) {
271               continue;
272             }
273     
274             // Avoid copying over the same function twice. The last definition always wins.
275             if (nameToExprResult.containsKey(fnName)) {
276               block.removeChild(nameToExprResult.get(fnName));
277             }
278     
279             Node fnValue = behaviorFunction.value.cloneTree();
280             Node exprResult = IR.exprResult(
281                 IR.assign(NodeUtil.newQName(compiler, qualifiedPath + fnName), fnValue));
282             JSDocInfoBuilder info = JSDocInfoBuilder.maybeCopyFrom(behaviorFunction.info);
283     
284             // Behaviors whose declarations are not in the global scope may contain references to
285             // symbols which do not exist in the element's scope. Only copy a function stub.
286             if (!behavior.isGlobalDeclaration) {
287               NodeUtil.getFunctionBody(fnValue).removeChildren();
288             }
289     
290             exprResult.getFirstChild().setJSDocInfo(info.build());
291             block.addChildToBack(exprResult);
292             nameToExprResult.put(fnName, exprResult);
293           }
294     
295           // Copy other members.
296           for (MemberDefinition behaviorProp : behavior.nonPropertyMembersToCopy) {
297             String propName = behaviorProp.name.getString();
298             if (nameToExprResult.containsKey(propName)) {
299               block.removeChild(nameToExprResult.get(propName));
300             }
301     
302             Node exprResult = IR.exprResult(NodeUtil.newQName(compiler, qualifiedPath + propName));
303             JSDocInfoBuilder info = JSDocInfoBuilder.maybeCopyFrom(behaviorProp.info);
304     
305             if (behaviorProp.name.isGetterDef()) {
306               info = new JSDocInfoBuilder(true);
307               if (behaviorProp.info != null && behaviorProp.info.getReturnType() != null) {
308                 info.recordType(behaviorProp.info.getReturnType());
309               }
310             }
311     
312             exprResult.getFirstChild().setJSDocInfo(info.build());
313             block.addChildToBack(exprResult);
314             nameToExprResult.put(propName, exprResult);
315           }
316         }
317       }

319       /**
320        * Adds the generated setter for a readonly property.
321        * @see https://www.polymer-project.org/0.8/docs/devguide/properties.html#read-only
322        */
323       private Node makeReadOnlySetter(String propName, String qualifiedPath) {
324         String setterName = "_set" + propName.substring(0, 1).toUpperCase() + propName.substring(1);
325         Node fnNode = IR.function(IR.name(""), IR.paramList(IR.name(propName)), IR.block());
326         Node exprResNode = IR.exprResult(
327             IR.assign(NodeUtil.newQName(compiler, qualifiedPath + setterName), fnNode));
328     
329         JSDocInfoBuilder info = new JSDocInfoBuilder(true);
330         // This is overriding a generated function which was added to the interface in
331         // {@code addInterfaceExterns}.
332         info.recordOverride();
333         exprResNode.getFirstChild().setJSDocInfo(info.build());
334     
335         return exprResNode;
336       }

338       /**
339        * Adds an interface for the given ClassDefinition to externs. This allows generated setter
340        * functions for read-only properties to avoid renaming altogether.
341        * @see https://www.polymer-project.org/0.8/docs/devguide/properties.html#read-only
342        */
343       private void addInterfaceExterns(
344           final PolymerClassDefinition cls, List<MemberDefinition> readOnlyProps) {
345         Node block = IR.block();
346     
347         String interfaceName = getInterfaceName(cls);
348         Node fnNode = IR.function(IR.name(""), IR.paramList(), IR.block());
349         Node varNode = IR.var(NodeUtil.newQName(compiler, interfaceName), fnNode);
350     
351         JSDocInfoBuilder info = new JSDocInfoBuilder(true);
352         info.recordInterface();
353         varNode.setJSDocInfo(info.build());
354         block.addChildToBack(varNode);
355     
356         appendPropertiesToBlock(cls, block, interfaceName + ".prototype.");
357         for (MemberDefinition prop : readOnlyProps) {
358           // Add all _set* functions to avoid renaming.
359           String propName = prop.name.getString();
360           String setterName = "_set" + propName.substring(0, 1).toUpperCase() + propName.substring(1);
361           Node setterExprNode = IR.exprResult(
362               NodeUtil.newQName(compiler, interfaceName + ".prototype." + setterName));
363     
364           JSDocInfoBuilder setterInfo = new JSDocInfoBuilder(true);
365           JSTypeExpression propType = PolymerPassStaticUtils.getTypeFromProperty(prop, compiler);
366           setterInfo.recordParameter(propName, propType);
367           setterExprNode.getFirstChild().setJSDocInfo(setterInfo.build());
368     
369           block.addChildToBack(setterExprNode);
370         }
371     
372         block.useSourceInfoIfMissingFromForTree(polymerElementExterns);
373     
374         Node parent = polymerElementExterns.getParent();
375         Node stmts = block.removeChildren();
376         parent.addChildrenToBack(stmts);
377     
378         compiler.reportCodeChange();
379       }

381       private static boolean hasShorthandAssignment(Node objLit) {
382         Preconditions.checkState(objLit.isObjectLit());
383         for (Node property : objLit.children()){
384           if (property.isStringKey() && !property.hasChildren()){
385             return true;
386           }
387         }
388         return false;
389       }

391       /**
392        * @return The name of the generated extern interface which the element implements.
393        */
394       private static String getInterfaceName(final PolymerClassDefinition cls) {
395         return "Polymer" + cls.target.getQualifiedName().replaceAll("\\.", "_") + "Interface";
396       }

398       /**
399        * @return An assign replacing the equivalent var or let declaration.
400        */
401       private static Node varToAssign(Node var) {
402         Node assign = IR.assign(
403             var.getFirstChild().cloneNode(),
404             var.getFirstChild().removeFirstChild());
405         return IR.exprResult(assign).useSourceInfoFromForTree(var);
406       }
