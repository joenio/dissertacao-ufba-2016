47     /**
48      * The arrow type models a "bare" function type: from some parameter types to
49      * a return type. JavaScript functions include more things like properties, type
50      * of THIS, etc, and are modeled by {@link FunctionType}.
51      */
52     final class ArrowType extends JSType {
53       private static final long serialVersionUID = 1L;
54     
55       final Node parameters;
56       JSType returnType;
57     
58       // Whether the return type is inferred.
59       final boolean returnTypeInferred;
60     
61       ArrowType(JSTypeRegistry registry, Node parameters, JSType returnType) {
62         this(registry, parameters, returnType, false);
63       }
64     
65       ArrowType(JSTypeRegistry registry, Node parameters,
66           JSType returnType, boolean returnTypeInferred) {
67         super(registry);
68     
69         this.parameters = parameters == null ?
70             registry.createParametersWithVarArgs(getNativeType(UNKNOWN_TYPE)) :
71             parameters;
72         this.returnType = returnType == null ?
73             getNativeType(UNKNOWN_TYPE) : returnType;
74         this.returnTypeInferred = returnTypeInferred;
75       }

77       @Override
78       public boolean isSubtype(JSType that) {
79         return isSubtype(that, ImplCache.create(), SubtypingMode.NORMAL);
80       }
81     
82       @Override
83       protected boolean isSubtype(JSType other,
84           ImplCache implicitImplCache, SubtypingMode subtypingMode) {
85         if (!(other instanceof ArrowType)) {
86           return false;
87         }
88     
89         ArrowType that = (ArrowType) other;
90     
91         // This is described in Draft 2 of the ES4 spec,
92         // Section 3.4.7: Subtyping Function Types.
93     
94         // this.returnType <: that.returnType (covariant)
95         if (!this.returnType.isSubtype(that.returnType, implicitImplCache, subtypingMode)) {
96           return false;
97         }
98     
99         // that.paramType[i] <: this.paramType[i] (contravariant)
100         //
101         // If this.paramType[i] is required,
102         // then that.paramType[i] is required.
103         //
104         // In theory, the "required-ness" should work in the other direction as
105         // well. In other words, if we have
106         //
107         // function f(number, number) {}
108         // function g(number) {}
109         //
110         // Then f *should* not be a subtype of g, and g *should* not be
111         // a subtype of f. But in practice, we do not implement it this way.
112         // We want to support the use case where you can pass g where f is
113         // expected, and pretend that g ignores the second argument.
114         // That way, you can have a single "no-op" function, and you don't have
115         // to create a new no-op function for every possible type signature.
116         //
117         // So, in this case, g < f, but f !< g
118         Node thisParam = parameters.getFirstChild();
119         Node thatParam = that.parameters.getFirstChild();
120         while (thisParam != null && thatParam != null) {
121           JSType thisParamType = thisParam.getJSType();
122           JSType thatParamType = thatParam.getJSType();
123           if (thisParamType != null) {
124             if (thatParamType == null ||
125                 !thatParamType.isSubtype(thisParamType, implicitImplCache, subtypingMode)) {
126               return false;
127             }
128           }
129     
130           boolean thisIsVarArgs = thisParam.isVarArgs();
131           boolean thatIsVarArgs = thatParam.isVarArgs();
132           boolean thisIsOptional = thisIsVarArgs || thisParam.isOptionalArg();
133           boolean thatIsOptional = thatIsVarArgs || thatParam.isOptionalArg();
134     
135           // "that" can't be a supertype, because it's missing a required argument.
136           if (!thisIsOptional && thatIsOptional) {
137             // NOTE(nicksantos): In our type system, we use {function(...?)} and
138             // {function(...NoType)} to to indicate that arity should not be
139             // checked. Strictly speaking, this is not a correct formulation,
140             // because now a sub-function can required arguments that are var_args
141             // in the super-function. So we special-case this.
142             boolean isTopFunction =
143                 thatIsVarArgs &&
144                 (thatParamType == null ||
145                  thatParamType.isUnknownType() ||
146                  thatParamType.isNoType());
147             if (!isTopFunction) {
148               return false;
149             }
150           }
151     
152           // don't advance if we have variable arguments
153           if (!thisIsVarArgs) {
154             thisParam = thisParam.getNext();
155           }
156           if (!thatIsVarArgs) {
157             thatParam = thatParam.getNext();
158           }
159     
160           // both var_args indicates the end
161           if (thisIsVarArgs && thatIsVarArgs) {
162             thisParam = null;
163             thatParam = null;
164           }
165         }
166     
167         // "that" can't be a supertype, because it's missing a required argument.
168         return thisParam == null || thisParam.isOptionalArg() || thisParam.isVarArgs()
169             || thatParam != null;
170       }

172       /**
173        * @return True if our parameter spec is equal to {@code that}'s parameter
174        *     spec.
175        */
176       boolean hasEqualParameters(ArrowType that, EquivalenceMethod eqMethod, EqCache eqCache) {
177         Node thisParam = parameters.getFirstChild();
178         Node otherParam = that.parameters.getFirstChild();
179         while (thisParam != null && otherParam != null) {
180           JSType thisParamType = thisParam.getJSType();
181           JSType otherParamType = otherParam.getJSType();
182           if (thisParamType != null) {
183             // Both parameter lists give a type for this param, it should be equal
184             if (otherParamType != null &&
185                 !thisParamType.checkEquivalenceHelper(otherParamType, eqMethod, eqCache)) {
186               return false;
187             }
188           } else {
189             if (otherParamType != null) {
190               return false;
191             }
192           }
193     
194           // Check var_args/optionality
195           if (thisParam.isOptionalArg() != otherParam.isOptionalArg()) {
196             return false;
197           }
198     
199           if (thisParam.isVarArgs() != otherParam.isVarArgs()) {
200             return false;
201           }
202     
203           thisParam = thisParam.getNext();
204           otherParam = otherParam.getNext();
205         }
206         // One of the parameters is null, so the types are only equal if both
207         // parameter lists are null (they are equal).
208         return thisParam == otherParam;
209       }

211       boolean checkArrowEquivalenceHelper(
212           ArrowType that, EquivalenceMethod eqMethod, EqCache eqCache) {
213         // Please keep this method in sync with the hashCode() method below.
214         if (!returnType.checkEquivalenceHelper(
215             that.returnType, eqMethod, eqCache)) {
216           return false;
217         }
218         return hasEqualParameters(that, eqMethod, eqCache);
219       }

221       @Override
222       public int hashCode() {
223         int hashCode = 0;
224         if (returnType != null) {
225           hashCode += returnType.hashCode();
226         }
227         if (returnTypeInferred) {
228           hashCode += 1;
229         }
230         if (parameters != null) {
231           Node param = parameters.getFirstChild();
232           while (param != null) {
233             JSType paramType = param.getJSType();
234             if (paramType != null) {
235               hashCode += paramType.hashCode();
236             }
237             param = param.getNext();
238           }
239         }
240         return hashCode;
241       }

272       @Override
273       JSType resolveInternal(ErrorReporter t, StaticTypedScope<JSType> scope) {
274         returnType = safeResolve(returnType, t, scope);
275         if (parameters != null) {
276           for (Node paramNode = parameters.getFirstChild();
277                paramNode != null; paramNode = paramNode.getNext()) {
278             paramNode.setJSType(paramNode.getJSType().resolve(t, scope));
279           }
280         }
281         return this;
282       }
283     
284       boolean hasUnknownParamsOrReturn() {
285         if (parameters != null) {
286           for (Node paramNode = parameters.getFirstChild();
287                paramNode != null; paramNode = paramNode.getNext()) {
288             JSType type = paramNode.getJSType();
289             if (type == null || type.isUnknownType()) {
290               return true;
291             }
292           }
293         }
294         return returnType == null || returnType.isUnknownType();
295       }

297       @Override
298       String toStringHelper(boolean forAnnotations) {
299         return "[ArrowType]";
300       }
301     
302       @Override
303       public boolean hasAnyTemplateTypesInternal() {
304         return returnType.hasAnyTemplateTypes()
305             || hasTemplatedParameterType();
306       }
307     
308       private boolean hasTemplatedParameterType() {
309         if (parameters != null) {
310           for (Node paramNode = parameters.getFirstChild();
311                paramNode != null; paramNode = paramNode.getNext()) {
312             JSType type = paramNode.getJSType();
313             if (type != null && type.hasAnyTemplateTypes()) {
314               return true;
315             }
316           }
317         }
318         return false;
319       }
