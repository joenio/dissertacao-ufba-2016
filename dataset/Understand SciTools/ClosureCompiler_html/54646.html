34       public void testVarAndOptionalParams() {
35         Node args = new Node(Token.PARAM_LIST,
36             Node.newString(Token.NAME, "a"),
37             Node.newString(Token.NAME, "b"));

39             Node.newString(Token.NAME, "opt_a"),
40             Node.newString(Token.NAME, "opt_b"));
41     
42         assertFalse(conv.isVarArgsParameter(args.getFirstChild()));
43         assertFalse(conv.isVarArgsParameter(args.getLastChild()));
44         assertFalse(conv.isVarArgsParameter(optArgs.getFirstChild()));
45         assertFalse(conv.isVarArgsParameter(optArgs.getLastChild()));
46     
47         assertFalse(conv.isOptionalParameter(args.getFirstChild()));
48         assertFalse(conv.isOptionalParameter(args.getLastChild()));
49         assertFalse(conv.isOptionalParameter(optArgs.getFirstChild()));
50         assertFalse(conv.isOptionalParameter(optArgs.getLastChild()));
51       }

212       public void testApplySubclassRelationship() {
213         JSTypeRegistry registry = new JSTypeRegistry(null);
214     
215         Node nodeA = new Node(Token.FUNCTION);
216         FunctionType ctorA = registry.createConstructorType("A", nodeA,
217             new Node(Token.PARAM_LIST), null, null);
218     
219         Node nodeB = new Node(Token.FUNCTION);
220         FunctionType ctorB = registry.createConstructorType("B", nodeB,
221             new Node(Token.PARAM_LIST), null, null);
222     
223         conv.applySubclassRelationship(ctorA, ctorB, SubclassType.INHERITS);
224     
225         assertTrue(ctorB.getPrototype().hasOwnProperty("constructor"));
226         assertEquals(nodeB, ctorB.getPrototype().getPropertyNode("constructor"));
227     
228         assertTrue(ctorB.hasOwnProperty("superClass_"));
229         assertEquals(nodeB, ctorB.getPropertyNode("superClass_"));
230       }

232       public void testDescribeCachingCall() {
233         assertCachingCall("goog.reflect.cache(obj, 10, function() {})");
234         assertCachingCall("goog.reflect.cache(obj, 10, function() {}, function() {})");
235         assertCachingCall("goog$reflect$cache(obj, 10, function() {})");
236         assertCachingCall("goog$reflect$cache(obj, 10, function() {}, function() {})");
237         assertNotCachingCall("goog.reflect.cache()");
238         assertNotCachingCall("goog.reflect.cache(obj)");
239         assertNotCachingCall("goog.reflect.cache(obj, 10)");
240         assertNotCachingCall("foo.cache(obj, 10, function() {}, function() {})");
241       }

278       private void assertDefinesClasses(String code, String subclassName,
279           String superclassName) {
280         Node n = parseTestCode(code);
