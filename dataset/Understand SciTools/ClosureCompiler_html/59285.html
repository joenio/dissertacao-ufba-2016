40     /**
41      * Rewrite "let"s and "const"s as "var"s.
42      * Rename block-scoped declarations and their references when necessary.
43      *
44      * TODO(moz): Try to use MakeDeclaredNamesUnique
45      *
46      * @author moz@google.com (Michael Zhou)
47      */
48     public final class Es6RewriteBlockScopedDeclaration extends AbstractPostOrderCallback
49         implements HotSwapCompilerPass {
50     
51       private final AbstractCompiler compiler;
52       private final Map<Node, Map<String, String>> renameMap = new LinkedHashMap<>();
53       private final Set<Node> letConsts = new HashSet<>();
54       private final Set<String> undeclaredNames = new HashSet<>();
55     
56       public Es6RewriteBlockScopedDeclaration(AbstractCompiler compiler) {
57         this.compiler = compiler;
58       }
59     
60       @Override
61       public void visit(NodeTraversal t, Node n, Node parent) {
62         if (!n.hasChildren() || !NodeUtil.isBlockScopedDeclaration(n.getFirstChild())) {
63           return;
64         }
65     
66         Scope scope = t.getScope();
67         Node nameNode = n.getFirstChild();
68         if (!n.isClass() && !n.isFunction() && !nameNode.hasChildren()
69             && (parent == null || !NodeUtil.isEnhancedFor(parent))
70             && !n.isCatch()
71             && inLoop(n)) {
72           Node undefined = IR.name("undefined");
73           if (nameNode.getJSDocInfo() != null || n.getJSDocInfo() != null) {
74             JSDocInfoBuilder jsDoc = new JSDocInfoBuilder(false);
75             jsDoc.recordType(new JSTypeExpression(new Node(Token.QMARK), n.getSourceFileName()));
76             undefined = IR.cast(undefined, jsDoc.build());
77           }
78           undefined.useSourceInfoFromForTree(nameNode);
79           nameNode.addChildToFront(undefined);
80         }
81     
82         String oldName = nameNode.getString();
83         if (n.isLet() || n.isConst()) {
84           letConsts.add(n);
85         }
86         Scope hoistScope = scope.getClosestHoistScope();
87         boolean doRename = false;
88         if (scope != hoistScope) {
89           doRename = hoistScope.isDeclared(oldName, true)
90               || undeclaredNames.contains(oldName);
91           String newName = doRename
92               ? oldName + "$" + compiler.getUniqueNameIdSupplier().get()
93               : oldName;
94           Var oldVar = scope.getVar(oldName);
95           scope.undeclare(oldVar);
96           hoistScope.declare(newName, nameNode, oldVar.input);
97           if (doRename) {
98             nameNode.setString(newName);
99             Node scopeRoot = scope.getRootNode();
100             if (!renameMap.containsKey(scopeRoot)) {
101               renameMap.put(scopeRoot, new HashMap<String, String>());
102             }
103     
104             renameMap.get(scopeRoot).put(oldName, newName);
105           }
106         }
107         if (doRename) {
108           compiler.reportCodeChange();
109         }
110       }

112       @Override
113       public void process(Node externs, Node root) {
114         NodeTraversal.traverseRootsEs6(compiler, new CollectUndeclaredNames(), externs, root);
115         NodeTraversal.traverseRootsEs6(compiler, this, externs, root);
116         NodeTraversal.traverseRootsEs6(compiler, new Es6RenameReferences(renameMap), externs, root);
117     
118         LoopClosureTransformer transformer = new LoopClosureTransformer();
119         NodeTraversal.traverseRootsEs6(compiler, transformer, externs, root);
120         transformer.transformLoopClosure();
121         varify();
122         NodeTraversal.traverseRootsEs6(
123             compiler, new RewriteBlockScopedFunctionDeclaration(), externs, root);
124       }
125     
126       @Override
127       public void hotSwapScript(Node scriptRoot, Node originalRoot) {
128         NodeTraversal.traverseEs6(compiler, scriptRoot, new CollectUndeclaredNames());
129         NodeTraversal.traverseEs6(compiler, scriptRoot, this);
130         NodeTraversal.traverseEs6(compiler, scriptRoot, new Es6RenameReferences(renameMap));
131     
132         LoopClosureTransformer transformer = new LoopClosureTransformer();
133         NodeTraversal.traverseEs6(compiler, scriptRoot, transformer);
134         transformer.transformLoopClosure();
135         varify();
136         NodeTraversal.traverseEs6(compiler, scriptRoot, new RewriteBlockScopedFunctionDeclaration());
137       }

139       /**
140        * Whether n is inside a loop. If n is inside a function which is inside a loop, we do not
141        * consider it to be inside a loop.
142        */
143       private boolean inLoop(Node n) {
144         Node enclosingNode = NodeUtil.getEnclosingNode(n, loopPredicate);
145         return enclosingNode != null && enclosingNode.getType() != Token.FUNCTION;
146       }
147     
148       private static final Predicate<Node> loopPredicate = new Predicate<Node>() {
149         @Override
150         public boolean apply(Node n) {
151           return n.getType() == Token.WHILE
152               || n.getType() == Token.FOR
153               || n.getType() == Token.FOR_OF
154               || n.getType() == Token.DO
155               || n.getType() == Token.FUNCTION;
156         }
157       };
158     
159       private static void extractInlineJSDoc(Node srcDeclaration, Node srcName, Node destDeclaration) {
160         JSDocInfo existingInfo = srcDeclaration.getJSDocInfo();
161         if (existingInfo == null) {
162           // Extract inline JSDoc from "src" and add it to the "dest" node.
163           existingInfo = srcName.getJSDocInfo();
164           srcName.setJSDocInfo(null);
165         }
166         JSDocInfoBuilder builder = JSDocInfoBuilder.maybeCopyFrom(existingInfo);
167         destDeclaration.setJSDocInfo(builder.build());
168       }
169     
170       private static void maybeAddConstJSDoc(Node srcDeclaration, Node srcParent, Node srcName,
171           Node destDeclaration) {
172         if (srcDeclaration.isConst()
173             // Don't add @const for the left side of a for/in. If we do we get warnings from the NTI.
174             && !(NodeUtil.isForIn(srcParent) && srcDeclaration == srcParent.getFirstChild())) {
175           extractInlineJSDoc(srcDeclaration, srcName, destDeclaration);
176           JSDocInfoBuilder builder = JSDocInfoBuilder.maybeCopyFrom(destDeclaration.getJSDocInfo());
177           builder.recordConstancy();
178           destDeclaration.setJSDocInfo(builder.build());
179         }
180       }
181     
182       private static void handleDeclarationList(Node declarationList, Node parent) {
183         // Normalize: "const i = 0, j = 0;" becomes "/** @const */ var i = 0; /** @const */ var j = 0;"
184         while (declarationList.hasMoreThanOneChild()) {
185           Node name = declarationList.getLastChild();
186           Node newDeclaration = IR.var(name.detachFromParent()).useSourceInfoFrom(declarationList);
187           maybeAddConstJSDoc(declarationList, parent, name, newDeclaration);
188           parent.addChildAfter(newDeclaration, declarationList);
189         }
190         maybeAddConstJSDoc(declarationList, parent, declarationList.getFirstChild(), declarationList);
191         declarationList.setType(Token.VAR);
192       }
193     
194       private void varify() {
195         if (!letConsts.isEmpty()) {
196           for (Node n : letConsts) {
197             if (n.isConst()) {
198               handleDeclarationList(n, n.getParent());
199             }
200             n.setType(Token.VAR);
201           }
202           compiler.reportCodeChange();
203         }
204       }

206       private class RewriteBlockScopedFunctionDeclaration extends
207           AbstractPostOrderCallback {
208     
209         @Override
210         public void visit(NodeTraversal t, Node n, Node parent) {
211           if (n.isFunction() && NormalizeStatements.maybeNormalizeFunctionDeclaration(n)) {
212             compiler.reportCodeChange();
213           }
214         }

217       /**
218        * Record undeclared names and aggressively rename possible references to them.
219        * Eg: In "{ let inner; } use(inner);", we rename the let declared variable.
220        */
221       private class CollectUndeclaredNames extends AbstractPostOrderCallback {
222         @Override
223         public void visit(NodeTraversal t, Node n, Node parent) {
224           if (n.isName() && !t.getScope().isDeclared(n.getString(), true)) {
225             undeclaredNames.add(n.getString());
226           }
227         }
228       }

230       /**
231        * Transforms let/const declarations captured by loop closures.
232        */
233       private class LoopClosureTransformer extends AbstractPostOrderCallback {
234         private static final String LOOP_OBJECT_NAME = "$jscomp$loop";
235     
236         private final Map<Node, LoopObject> loopObjectMap = new LinkedHashMap<>();
237         private final Multimap<Node, LoopObject> functionLoopObjectsMap =
238             LinkedHashMultimap.create();
239         private final Multimap<Node, String> functionHandledMap = HashMultimap.create();
240         private final Multimap<Var, Node> referenceMap = LinkedHashMultimap.create();
241     
242         @Override
243         public void visit(NodeTraversal t, Node n, Node parent) {
244           if (!NodeUtil.isReferenceName(n)) {
245             return;
246           }
247     
248           String name = n.getString();
249           Scope referencedIn = t.getScope();
250           Var var = referencedIn.getVar(name);
251           if (var == null) {
252             return;
253           }
254     
255           if (!var.isLet() && !var.isConst()) {
256             return;
257           }
258     
259           if (n.getParent().isLet() || n.getParent().isConst()) {
260             letConsts.add(n.getParent());
261           }
262     
263           // Traverse nodes up from let/const declaration:
264           // If we hit a function or the root before a loop - Not a loop closure.
265           // if we hit a loop first - maybe loop closure.
266           Scope declaredIn = var.getScope();
267           Node loopNode = null;
268           for (Scope s = declaredIn;; s = s.getParent()) {
269             Node scopeRoot = s.getRootNode();
270             if (NodeUtil.isLoopStructure(s.getRootNode())) {
271               loopNode = scopeRoot;
272               break;
273             } else if (scopeRoot.getParent() != null
274                 && NodeUtil.isLoopStructure(scopeRoot.getParent())) {
275               loopNode = scopeRoot.getParent();
276               break;
277             } else if (s.isFunctionBlockScope() || s.isGlobal()) {
278               return;
279             }
280           }
281     
282           referenceMap.put(var, n);
283     
284           // Traverse scopes from reference scope to declaration scope.
285           // If we hit a function - loop closure detected.
286           for (Scope s = referencedIn; s != declaredIn; s = s.getParent()) {
287             if (s.isFunctionBlockScope()) {
288               Node function = s.getRootNode().getParent();
289               if (functionHandledMap.containsEntry(function, name)) {
290                 return;
291               }
292               functionHandledMap.put(function, name);
293     
294               if (!loopObjectMap.containsKey(loopNode)) {
295                 loopObjectMap.put(loopNode,
296                     new LoopObject(
297                         LOOP_OBJECT_NAME + "$" + compiler.getUniqueNameIdSupplier().get()));
298               }
299               LoopObject object = loopObjectMap.get(loopNode);
300               object.vars.add(var);
301     
302               functionLoopObjectsMap.put(function,  object);
303               return;
304             }
305           }
306         }
307     
308         private void transformLoopClosure() {
309           if (loopObjectMap.isEmpty()) {
310             return;
311           }
312     
313           for (Node loopNode : loopObjectMap.keySet()) {
314             // Introduce objects to reflect the captured scope variables.
315             // Fields are initially left as undefined to avoid cases like:
316             //   var $jscomp$loop$0 = {i: 0, j: $jscomp$loop$0.i}
317             // They are initialized lazily by changing declarations into assignments
318             // later.
319             LoopObject object = loopObjectMap.get(loopNode);
320             Node objectLitNextIteration = IR.objectlit();
321             for (Var var : object.vars) {
322               objectLitNextIteration.addChildToBack(
323                   IR.stringKey(var.name, IR.getprop(IR.name(object.name), IR.string(var.name))));
324             }
325     
326             Node updateLoopObject = IR.assign(IR.name(object.name), objectLitNextIteration);
327             Node objectLit =
328                 IR.var(IR.name(object.name), IR.objectlit()).useSourceInfoFromForTree(loopNode);
329             loopNode.getParent().addChildBefore(objectLit, loopNode);
330             if (NodeUtil.isVanillaFor(loopNode)) { // For
331               // The initializer is pulled out and placed prior to the loop.
332               Node initializer = loopNode.getFirstChild();
333               loopNode.replaceChild(initializer, IR.empty());
334               if (!initializer.isEmpty()) {
335                 if (!NodeUtil.isNameDeclaration(initializer)) {
336                   initializer = IR.exprResult(initializer).useSourceInfoFrom(initializer);
337                 }
338                 loopNode.getParent().addChildBefore(initializer, loopNode);
339               }
340     
341               Node increment = loopNode.getChildAtIndex(2);
342               if (increment.isEmpty()) {
343                 loopNode.replaceChild(
344                     increment,
345                     updateLoopObject.useSourceInfoIfMissingFromForTree(loopNode));
346               } else {
347                 Node placeHolder = IR.empty();
348                 loopNode.replaceChild(increment, placeHolder);
349                 loopNode.replaceChild(
350                     placeHolder,
351                     IR.comma(updateLoopObject, increment)
352                         .useSourceInfoIfMissingFromForTree(loopNode));
353               }
354             } else if (loopNode.isDo()) { // do-while, put at the end of the block
355               loopNode.getFirstChild().addChildToBack(IR.exprResult(updateLoopObject)
356                   .useSourceInfoIfMissingFromForTree(loopNode));
357             } else { // For-in, for-of or while, put at the end of the block
358               loopNode.getLastChild().addChildToBack(IR.exprResult(updateLoopObject)
359                   .useSourceInfoIfMissingFromForTree(loopNode));
360             }
361     
362             // For captured variables, change declarations to assignments on the
363             // corresponding field of the introduced object. Rename all references
364             // accordingly.
365             for (Var var : object.vars) {
366               for (Node reference : referenceMap.get(var)) {
367                 // For-of and for-in declarations are not altered, since they are
368                 // used as temporary variables for assignment.
369                 if (NodeUtil.isEnhancedFor(loopNode)
370                     && loopNode.getFirstChild() == reference.getParent()) {
371                   loopNode.getLastChild().addChildToFront(
372                       IR.exprResult(IR.assign(
373                           IR.getprop(IR.name(object.name), IR.string(var.name)),
374                           var.getNameNode().cloneNode()))
375                           .useSourceInfoIfMissingFromForTree(reference));
376                 } else {
377                   if (NodeUtil.isNameDeclaration(reference.getParent())) {
378                     Node declaration = reference.getParent();
379                     Node grandParent = declaration.getParent();
380                     handleDeclarationList(declaration, grandParent);
381                     declaration = reference.getParent(); // Might have changed after normalization.
382                     // Change declaration to assignment, or just drop it if there's
383                     // no initial value.
384                     if (reference.hasChildren()) {
385                       Node newReference = reference.cloneNode();
386                       Node assign = IR.assign(newReference, reference.removeFirstChild());
387                       extractInlineJSDoc(declaration, reference, declaration);
388                       maybeAddConstJSDoc(declaration, grandParent, reference, declaration);
389                       assign.setJSDocInfo(declaration.getJSDocInfo());
390     
391                       Node replacement = IR.exprResult(assign)
392                           .useSourceInfoIfMissingFromForTree(declaration);
393                       grandParent.replaceChild(declaration, replacement);
394                       reference = newReference;
395                     } else {
396                       grandParent.removeChild(declaration);
397                     }
398                     letConsts.remove(declaration);
399                     compiler.reportCodeChange();
400                   }
401     
402                   if (reference.getParent().isCall()
403                       && reference.getParent().getFirstChild() == reference) {
404                     reference.getParent().putBooleanProp(Node.FREE_CALL, false);
405                   }
406                   // Change reference to GETPROP.
407                   reference.getParent().replaceChild(
408                       reference,
409                       IR.getprop(IR.name(object.name), IR.string(var.name))
410                           .useSourceInfoIfMissingFromForTree(reference));
411                 }
412               }
413             }
414           }
415     
416           // Create wrapper functions and call them.
417           for (Node function : functionLoopObjectsMap.keySet()) {
418             Node returnNode = IR.returnNode();
419             Collection<LoopObject> objects = functionLoopObjectsMap.get(function);
420             Node[] objectNames = new Node[objects.size()];
421             Node[] objectNamesForCall = new Node[objects.size()];
422             int i = 0;
423             for (LoopObject object : objects) {
424               objectNames[i] = IR.name(object.name);
425               objectNamesForCall[i] = IR.name(object.name);
426               i++;
427             }
428     
429             Node iife = IR.function(
430                 IR.name(""),
431                 IR.paramList(objectNames),
432                 IR.block(returnNode));
433             Node call = IR.call(iife, objectNamesForCall);
434             call.putBooleanProp(Node.FREE_CALL, true);
435             Node replacement;
436             if (NodeUtil.isFunctionDeclaration(function)) {
437               replacement = IR.var(IR.name(function.getFirstChild().getString()), call)
438                   .useSourceInfoIfMissingFromForTree(function);
439             } else {
440               replacement = call.useSourceInfoIfMissingFromForTree(function);
441             }
442             function.getParent().replaceChild(function, replacement);
443             returnNode.addChildToFront(function);
444           }
445         }
