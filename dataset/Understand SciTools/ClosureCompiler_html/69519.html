17     package com.google.javascript.jscomp;
18     
19     import com.google.javascript.rhino.Node;
20     import com.google.javascript.rhino.StaticSourceFile;
21     
22     import java.util.regex.Matcher;
23     import java.util.regex.Pattern;
24     
25     /**
26      * This describes the Google-specific JavaScript coding conventions.
27      * Within Google, variable names are semantically significant.
28      *
29      */
30     public class GoogleCodingConvention extends CodingConventions.Proxy {
31     
32       private static final long serialVersionUID = 1L;
33     
34       private static final String OPTIONAL_ARG_PREFIX = "opt_";
35     
36       private static final String VAR_ARGS_NAME = "var_args";
37     
38       private static final Pattern ENUM_KEY_PATTERN =
39         Pattern.compile("[A-Z0-9][A-Z0-9_]*");
40     
41       private static final Pattern PACKAGE_WITH_TEST_DIR =
42         Pattern.compile("^(.*)/(?:test|tests|testing)/(?:[^/]+)$");
43     
44       private static final Pattern GENFILES_DIR = Pattern.compile("^.*/genfiles/(.*)$");
45     
46       /** By default, decorate the ClosureCodingConvention. */
47       public GoogleCodingConvention() {
48         this(new ClosureCodingConvention());
49       }
50     
51       /** Decorates a wrapped CodingConvention. */
52       public GoogleCodingConvention(CodingConvention convention) {
53         super(convention);
54       }
55     
56       /**
57        * {@inheritDoc}
58        *
59        * <p>This enforces the Google const name convention, that the first character
60        * after the last $ must be an upper-case letter and all subsequent letters
61        * must be upper case. The name must be at least 2 characters long.
62        *
63        * <p>Examples:
64        * <pre>
65        *      aaa          Not constant - lower-case letters in the name
66        *      A            Not constant - too short
67        *      goog$A       Constant - letters after the $ are upper-case.
68        *      AA17         Constant - digits can appear after the first letter
69        *      goog$7A      Not constant - first character after the $ must be
70        *                   upper case.
71        *      $A           Constant - doesn't have to be anything in front of the $
72        * </pre>
73        */
74       @Override
75       public boolean isConstant(String name) {
76         if (name.length() <= 1) {
77           return false;
78         }
79     
80         // In compiled code, '$' is often a namespace delimiter. To allow inlining
81         // of namespaced constants, we strip off any namespaces here.
82         int pos = name.lastIndexOf('$');
83         if (pos >= 0) {
84           name = name.substring(pos + 1);
85           if (name.isEmpty()) {
86             return false;
87           }
88         }
89     
90         return isConstantKey(name);
91       }
92     
93       @Override
94       public boolean isConstantKey(String name) {
95         if (name.isEmpty() || !Character.isUpperCase(name.charAt(0))) {
96           return false;
97         }
98     
99         // hack way of checking that there aren't any lower-case letters
100         return name.toUpperCase().equals(name);
101       }
102     
103       /**
104        * {@inheritDoc}
105        *
106        * <p>This enforces Google's convention about enum key names. They must match
107        * the regular expression {@code [A-Z0-9][A-Z0-9_]*}.
108        *
109        * <p>Examples:
110        * <ul>
111        * <li>A</li>
112        * <li>213</li>
113        * <li>FOO_BAR</li>
114        * </ul>
115        */
116       @Override
117       public boolean isValidEnumKey(String key) {
118         return ENUM_KEY_PATTERN.matcher(key).matches();
119       }
120     
121       /**
122        * {@inheritDoc}
123        *
124        * <p>In Google code, parameter names beginning with {@code opt_} are
125        * treated as optional arguments.
126        */
127       @Override
128       public boolean isOptionalParameter(Node parameter) {
129         return super.isOptionalParameter(parameter)
130             || parameter.getString().startsWith(OPTIONAL_ARG_PREFIX);
131       }
132     
133       @Override
134       public boolean isVarArgsParameter(Node parameter) {
135         return super.isVarArgsParameter(parameter)
136             || VAR_ARGS_NAME.equals(parameter.getString());
137       }
138     
139       /**
140        * {@inheritDoc}
141        *
142        * <p>In Google code, any global name starting with an underscore is
143        * considered exported.
144        */
145       @Override
146       public boolean isExported(String name, boolean local) {
147         return super.isExported(name, local) || (!local && name.startsWith("_"));
148       }
149     
150       @Override
151       public boolean isClassFactoryCall(Node callNode) {
152         return super.isClassFactoryCall(callNode)
153             || callNode.getFirstChild().matchesQualifiedName("Polymer");
154       }
155     
156       /**
157        * {@inheritDoc}
158        *
159        * <p>In Google code, the package name of a source file is its file path.
160        * Exceptions: if a source file's parent directory is "test", "tests", or
161        * "testing", that directory is stripped from the package name.
162        * If a file is generated, strip the "genfiles" prefix to try
163        * to match the package of the generating file.
164        */
165       @Override
166       public String getPackageName(StaticSourceFile source) {
167         String name = source.getName();
168         Matcher genfilesMatcher = GENFILES_DIR.matcher(name);
169         if (genfilesMatcher.find()) {
170           name = genfilesMatcher.group(1);
171         }
172         Matcher m = PACKAGE_WITH_TEST_DIR.matcher(name);
173         if (m.find()) {
174           return m.group(1);
175         } else {
176           int lastSlash = name.lastIndexOf('/');
177           return lastSlash == -1 ? "" : name.substring(0, lastSlash);
178         }
179       }
180     
181       /**
182        * {@inheritDoc}
183        *
184        * <p>In Google code, private names end with an underscore, and exported
185        * names are never considered private (see {@link #isExported}).
186        */
187       @Override
188       public boolean isPrivate(String name) {
189         return name.endsWith("_") && !name.endsWith("__") && !isExported(name);
190       }
191     
192       @Override
193       public boolean hasPrivacyConvention() {
194         return true;
195       }
196     }
