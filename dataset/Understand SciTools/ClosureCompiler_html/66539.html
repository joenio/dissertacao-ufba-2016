50     /**
51      * A builder for {@link JSDocInfo} objects. This builder abstracts the
52      * construction process of {@link JSDocInfo} objects whilst minimizing the
53      * number of instances of {@link JSDocInfo} objects. It provides early
54      * incompatibility detection among properties stored on the {@code JSDocInfo}
55      * object being created.
56      *
57      */
58     public final class JSDocInfoBuilder {
59       // the current JSDoc which is being populated
60       private JSDocInfo currentInfo;
61     
62       // whether the current JSDocInfo has valuable information
63       private boolean populated;
64     
65       // whether to include the documentation itself when parsing the JsDoc
66       private boolean parseDocumentation;
67     
68       // the current marker, if any.
69       private JSDocInfo.Marker currentMarker;
70     
71       public JSDocInfoBuilder(boolean parseDocumentation) {
72         this(new JSDocInfo(parseDocumentation), parseDocumentation, false);
73       }
74     
75       private JSDocInfoBuilder(
76           JSDocInfo info, boolean parseDocumentation, boolean populated) {
77         this.currentInfo = info;
78         this.parseDocumentation = parseDocumentation;
79         this.populated = populated;
80       }

82       public static JSDocInfoBuilder copyFrom(JSDocInfo info) {
83         JSDocInfo clone = info.clone();
84         if (clone.getVisibility() == Visibility.INHERITED) {
85           clone.setVisibility(null);
86         }
87         return new JSDocInfoBuilder(clone, info.isDocumentationIncluded(), true);
88       }
89     
90       public static JSDocInfoBuilder maybeCopyFrom(@Nullable JSDocInfo info) {
91         if (info == null) {
92           return new JSDocInfoBuilder(true);
93         }
94         return copyFrom(info);
95       }

159       /**
160        * Builds a {@link JSDocInfo} object based on the populated information and
161        * returns it. Once this method is called, the builder can be reused to build
162        * another {@link JSDocInfo} object.
163        *
164        * @return a {@link JSDocInfo} object populated with the values given to this
165        *     builder. If no value was populated, this method simply returns
166        *     {@code null}
167        */
168       public JSDocInfo buildAndReset() {
169         JSDocInfo info = build(false);
170         if (currentInfo == null) {
171           currentInfo = new JSDocInfo(parseDocumentation);
172           populated = false;
173         }
174         return info;
175       }

177       /**
178        * Builds a {@link JSDocInfo} object based on the populated information and
179        * returns it.
180        *
181        * @param always Return an default JSDoc object.
182        * @return a {@link JSDocInfo} object populated with the values given to this
183        *     builder. If no value was populated and {@code always} is false, returns
184        *     {@code null}. If {@code always} is true, returns a default JSDocInfo.
185        */
186       public JSDocInfo build(boolean always) {
187         if (populated || always) {
188           Preconditions.checkState(currentInfo != null);
189           JSDocInfo built = currentInfo;
190           currentInfo = null;
191           populateDefaults(built);
192           populated = false;
193           return built;
194         } else {
195           return null;
196         }
197       }

199       /** Generate defaults when certain parameters are not specified. */
200       private static void populateDefaults(JSDocInfo info) {
201         if (info.getVisibility() == null) {
202           info.setVisibility(Visibility.INHERITED);
203         }
204       }

206       /**
207        * Adds a marker to the current JSDocInfo and populates the marker with the
208        * annotation information.
209        */
210       public void markAnnotation(String annotation, int lineno, int charno) {
211         JSDocInfo.Marker marker = currentInfo.addMarker();
212     
213         if (marker != null) {
214           JSDocInfo.TrimmedStringPosition position =
215               new JSDocInfo.TrimmedStringPosition();
216           position.setItem(annotation);
217           position.setPositionInformation(lineno, charno, lineno,
218               charno + annotation.length());
219           marker.setAnnotation(position);
220           populated = true;
221         }
222     
223         currentMarker = marker;
224       }

240       /**
241        * Adds a type declaration to the current marker.
242        */
243       public void markTypeNode(Node typeNode, int lineno, int startCharno,
244           int endLineno, int endCharno, boolean hasLC) {
245         if (currentMarker != null) {
246           JSDocInfo.TypePosition position = new JSDocInfo.TypePosition();
247           position.setItem(typeNode);
248           position.setHasBrackets(hasLC);
249           position.setPositionInformation(lineno, startCharno,
250               endLineno, endCharno);
251           currentMarker.setType(position);
252         }
253       }

255       /**
256        * Adds a name declaration to the current marker.
257        */
258       public void markName(String name, StaticSourceFile file,
259           int lineno, int charno) {
260         if (currentMarker != null) {
261           // Record the name as both a SourcePosition<String> and a
262           // SourcePosition<Node>. The <String> form is deprecated,
263           // because <Node> is more consistent with how other name
264           // references are handled (see #markTypeNode)
265           //
266           // TODO(nicksantos): Remove all uses of the Name position
267           // and replace them with the NameNode position.
268           JSDocInfo.TrimmedStringPosition position =
269               new JSDocInfo.TrimmedStringPosition();
270           position.setItem(name);
271           position.setPositionInformation(lineno, charno,
272               lineno, charno + name.length());
273           currentMarker.setName(position);
274     
275           JSDocInfo.NamePosition nodePos = new JSDocInfo.NamePosition();
276           Node node = Node.newString(Token.NAME, name, lineno, charno);
277           node.setLength(name.length());
278           node.setStaticSourceFile(file);
279           nodePos.setItem(node);
280           nodePos.setPositionInformation(lineno, charno,
281               lineno, charno + name.length());
282           currentMarker.setNameNode(nodePos);
283         }
284       }

1344       private boolean hasAnyTypeRelatedTags() {
1345         return currentInfo.isConstructor()
1346             || currentInfo.isInterface()
1347             || currentInfo.isAbstract()
1348             || currentInfo.getParameterCount() > 0
1349             || currentInfo.hasReturnType()
1350             || currentInfo.hasBaseType()
1351             || currentInfo.getExtendedInterfacesCount() > 0
1352             || currentInfo.getLendsName() != null
1353             || currentInfo.hasThisType()
1354             || hasAnySingletonTypeTags();
