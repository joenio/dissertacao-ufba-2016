19     import com.google.common.annotations.VisibleForTesting;
20     import com.google.common.base.Preconditions;
21     import com.google.javascript.jscomp.DefinitionsRemover.Definition;
22     import com.google.javascript.rhino.IR;
23     import com.google.javascript.rhino.Node;
24     
25     import java.util.ArrayList;
26     import java.util.Collection;
27     import java.util.List;

29     /**
30      * A compiler pass for optimize function return results.  Currently this
31      * pass looks for results that are complete unused and rewrite then to be:
32      *   "return x()" -->"x(); return"
33      * , but it can easily be
34      * expanded to look for use context to avoid unneeded type coercion:
35      *   - "return x.toString()" --> "return x"
36      *   - "return !!x" --> "return x"
37      * @author johnlenz@google.com (John Lenz)
38      */
39     class OptimizeReturns
40         implements OptimizeCalls.CallGraphCompilerPass, CompilerPass {
41     
42       private AbstractCompiler compiler;
43     
44       OptimizeReturns(AbstractCompiler compiler) {
45         this.compiler = compiler;
46       }
47     
48       @Override
49       @VisibleForTesting
50       public void process(Node externs, Node root) {
51         SimpleDefinitionFinder defFinder = new SimpleDefinitionFinder(compiler);
52         defFinder.process(externs, root);
53         process(externs, root, defFinder);
54       }
55     
56       @Override
57       public void process(
58           Node externs, Node root, SimpleDefinitionFinder definitions) {
59         // Find all function nodes whose callers ignore the return values.
60         List<Node> toOptimize = new ArrayList<>();
61         for (DefinitionSite defSite : definitions.getDefinitionSites()) {
62           if (!defSite.inExterns && !callResultsMaybeUsed(definitions, defSite)) {
63             toOptimize.add(defSite.definition.getRValue());
64           }
65         }
66         // Optimize the return statements.
67         for (Node node : toOptimize) {
68           rewriteReturns(definitions, node);
69         }
70       }
71     
72       /**
73        * Determines if a function result might be used.  A result might be use if:
74        * - Function must is exported.
75        * - The definition is never accessed outside a function call context.
76        */
77       private static boolean callResultsMaybeUsed(
78           SimpleDefinitionFinder defFinder, DefinitionSite definitionSite) {
79     
80         Definition definition = definitionSite.definition;
81     
82         // Assume non-function definitions results are used.
83         Node rValue = definition.getRValue();
84         if (rValue == null || !rValue.isFunction()) {
85           return true;
86         }
87     
88         // Be conservative, don't try to optimize any declaration that isn't as
89         // simple function declaration or assignment.
90         if (!SimpleDefinitionFinder.isSimpleFunctionDeclaration(rValue)) {
91           return true;
92         }
93     
94         if (!defFinder.canModifyDefinition(definition)) {
95           return true;
96         }
97     
98         Collection<UseSite> useSites = defFinder.getUseSites(definition);
99         for (UseSite site : useSites) {
100           // Assume indirect definitions references use the result
101           Node useNodeParent = site.node.getParent();
102           if (isCall(site)) {
103             Node callNode = useNodeParent;
104             Preconditions.checkState(callNode.isCall());
105             if (NodeUtil.isExpressionResultUsed(callNode)) {
106               return true;
107             }
108           } else {
109             // Allow a standalone name reference.
110             //     var a;
111             if (!useNodeParent.isVar()) {
112               return true;
113             }
114           }
115         }
116     
117         // No possible use of the definition result
118         return false;
119       }
120     
121       /**
122        * For the supplied function node, rewrite all the return expressions so that:
123        *    return foo();
124        * becomes:
125        *    foo(); return;
126        * Useless return will be removed later by the peephole optimization passes.
127        */
128       private void rewriteReturns(
129           final SimpleDefinitionFinder defFinder, Node fnNode) {
130         Preconditions.checkState(fnNode.isFunction());
131         NodeUtil.visitPostOrder(
132           fnNode.getLastChild(),
133           new NodeUtil.Visitor() {
134             @Override
135             public void visit(Node node) {
136               if (node.isReturn() && node.hasOneChild()) {
137                 boolean keepValue = NodeUtil.mayHaveSideEffects(
138                     node.getFirstChild(), compiler);
139                 if (!keepValue) {
140                   defFinder.removeReferences(node.getFirstChild());
141                 }
142                 Node result = node.removeFirstChild();
143                 if (keepValue) {
144                   node.getParent().addChildBefore(
145                     IR.exprResult(result).srcref(result), node);
146                 }
147                 compiler.reportCodeChange();
148               }
149             }
150           },
151           new NodeUtil.MatchShallowStatement());
152       }
153     
154       /**
155        * Determines if the name node acts as the function name in a call expression.
156        */
157       private static boolean isCall(UseSite site) {
158         Node node = site.node;
159         Node parent = node.getParent();
160         return (parent.getFirstChild() == node) && parent.isCall();
161       }
162     }
