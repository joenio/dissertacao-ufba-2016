62       /**
63        * All the information parsed out of a single file.
64        * Exported as a JSON object:
65        * <pre> {@code {
66        *   "custom_annotations": {?Array<[string, string]>},  @.*
67        *   "goog": {?bool},  whether 'goog' is implicitly required
68        *   "has_soy_delcalls": {?Array<string>},  @fileoverview @hassoydelcall {.*}
69        *   "has_soy_deltemplates": {?Array<string>},  @fileoverview @hassoydeltemplate {.*}
70        *   "imported_modules": {?Array<string>},  import ... from .*
71        *   "is_config": {?bool},  @fileoverview @config
72        *   "is_externs": {?bool},  @fileoverview @externs
73        *   "load_flags": {?Array<[string, string]>},
74        *   "mod_name": {?Array<string>},  @fileoverview @modName .*, @modName {.*}
75        *   "mods": {?Array<string>},  @fileoverview @mods {.*}
76        *   "provide_goog": {?bool},  @fileoverview @provideGoog
77        *   "provides": {?Array<string>},
78        *   "requires": {?Array<string>},  note: look for goog.* for 'goog'
79        *   "requires_css": {?Array<string>},  @fileoverview @requirecss {.*}
80        *   "testonly": {?bool},  goog.setTestOnly
81        *   "visibility: {?Array<string>},  @fileoverview @visibility {.*}
82        * }}</pre>
83        * Any trivial values are omitted.
84        */
85       static final class FileInfo {
86         final ErrorReporter reporter;
87     
88         boolean goog = false;
89         boolean isConfig = false;
90         boolean isExterns = false;
91         boolean provideGoog = false;
92         boolean testonly = false;
93     
94         final Set<String> hasSoyDelcalls = new TreeSet<>();
95         final Set<String> hasSoyDeltemplates = new TreeSet<>();
96         final Set<String> importedModules = new TreeSet<>();
97         final List<String> modName = new ArrayList<>();
98         final List<String> mods = new ArrayList<>();
99     
100         // Note: multiple copies doesn't make much sense, but we report
101         // each copy so that calling code can choose how to handle it
102         final Multiset<String> provides = TreeMultiset.create();
103         final Multiset<String> requires = TreeMultiset.create();
104         final Multiset<String> requiresCss = TreeMultiset.create();
105         final Multiset<String> visibility = TreeMultiset.create();
106     
107         final Set<JsArray<String>> customAnnotations = assoc();
108         final Set<JsArray<String>> loadFlags = assoc();
109     
110         FileInfo(ErrorReporter reporter) {
111           this.reporter = reporter;
112         }
113     
114         private void handleGoog() {
115           if (provideGoog) {
116             provides.add("goog");
117           } else if (goog) {
118             requires.add("goog");
119           }
120         }
121     
122         /** Exports the file info as a JSON object. */
123         JsObject<Object> full() {
124           handleGoog();
125           return new SparseObject()
126               .set("custom_annotations", customAnnotations)
127               .set("goog", goog)
128               .set("has_soy_delcalls", hasSoyDelcalls)
129               .set("has_soy_deltemplates", hasSoyDeltemplates)
130               .set("imported_modules", importedModules)
131               .set("is_config", isConfig)
132               .set("is_externs", isExterns)
133               .set("load_flags", loadFlags)
134               .set("modName", modName)
135               .set("mods", mods)
136               .set("provide_goog", provideGoog)
137               .set("provides", provides)
138               .set("requires", requires)
139               .set("requiresCss", requiresCss)
140               .set("testonly", testonly)
141               .set("visibility", visibility)
142               .object;
143         }
144       }

146       /** Represents a single JSDoc annotation, with an optional argument. */
147       private static class CommentAnnotation {
148     
149         /** Annotation name, e.g. "@fileoverview" or "@externs". */
150         final String name;
151         /**
152          * Annotation value: either the bare identifier immediately after the
153          * annotation, or else string in braces.
154          */
155         final String value;
156     
157         CommentAnnotation(String name, String value) {
158           this.name = name;
159           this.value = value;
160         }
161     
162         /** Returns all the annotations in a given comment string. */
163         static List<CommentAnnotation> parse(String comment) {
164           // TODO(sdh): This is reinventing a large part of JSDocInfoParser.  We should
165           // try to consolidate as much as possible.  This requires several steps:
166           //   1. Make all the annotations we look for first-class in JSDocInfo
167           //   2. Support custom annotations (may already be done?)
168           //   3. Fix up existing code so that all these annotations are in @fileoverview
169           //   4. Change this code to simply inspect the script's JSDocInfo instead of re-parsing
170           JsRegExp re = new JsRegExp(
171               ANNOTATION_RE,
172               "g");
173           JsRegExp.Match match;
174           List<CommentAnnotation> out = new ArrayList<>();
175           while ((match = re.exec(comment)) != null) {
176             boolean modName = match.get(OTHER_ANNOTATION_GROUP) == null;
177             String name = modName ? "@modName" : match.get(OTHER_ANNOTATION_GROUP);
178             String value =
179                 Strings.nullToEmpty(match.get(modName ? MODNAME_VALUE_GROUP : OTHER_VALUE_GROUP));
180             out.add(new CommentAnnotation(name, value));
181           }
182           return out;
183         }
184     
185         private static final String ANNOTATION_RE =
186             Joiner.on("").join(
187                 // Don't match "@" in the middle of a word
188                 "(?:[^a-zA-Z0-9_$]|^)",
189                 "(?:",
190                 // Case 1: @modName with a single identifier and no braces
191                 "@modName[\\t\\v\\f ]*([^{\\t\\n\\v\\f\\r ][^\\t\\n\\v\\f\\r ]*)",
192                 "|",
193                 // Case 2: Everything else, with an optional brace-delimited argument
194                 "(@[a-zA-Z]+)(?:\\s*\\{\\s*([^}\\t\\n\\v\\f\\r ]+)\\s*\\})?",
195                 ")");
196         private static final int MODNAME_VALUE_GROUP = 1;
197         private static final int OTHER_ANNOTATION_GROUP = 2;
198         private static final int OTHER_VALUE_GROUP = 3;
199       }

201       /** Method exported to JS to parse a file for dependencies and annotations. */
202       @JsMethod(name = "gjd", namespace = "jscomp")
203       public static JsObject<Object> gjd(String code, String filename, @Nullable Reporter reporter) {
204         return parse(code, filename, reporter).full();

207       /** Internal implementation to produce the {@link FileInfo} object. */
208       private static FileInfo parse(String code, String filename, @Nullable Reporter reporter) {
209         ErrorReporter errorReporter = new DelegatingReporter(reporter);
210         Compiler compiler = new Compiler();
211         compiler.init(
212             ImmutableList.<SourceFile>of(), ImmutableList.<SourceFile>of(), new CompilerOptions());
213     
214         Config config =
215             ParserRunner.createConfig(
216                 // TODO(sdh): ES6_STRICT, with a non-strict fallback - then give warnings.
217                 Config.LanguageMode.ECMASCRIPT6,
218                 Config.JsDocParsing.INCLUDE_DESCRIPTIONS_NO_WHITESPACE,
219                 Config.SourceLocationInformation.PRESERVE,
220                 Config.RunMode.KEEP_GOING,
221                 /* extraAnnotationNames */ ImmutableSet.<String>of());
222     
223         SourceFile source = SourceFile.fromCode(filename, code);
224         FileInfo info = new FileInfo(errorReporter);
225         ParserRunner.ParseResult parsed = ParserRunner.parse(source, code, config, errorReporter);
226         parsed.ast.setInputId(new InputId(filename));
227         String version = parsed.features.version();
228         if (!version.equals("es3")) {
229           info.loadFlags.add(JsArray.of("lang", version));
230         }
231     
232         for (Comment comment : parsed.comments) {
233           if (comment.type == Comment.Type.JSDOC) {
234             parseComment(comment, info);
235           }
236         }
237         NodeTraversal.traverseEs6(compiler, parsed.ast, new Traverser(info));
238         return info;
239       }

241       /** Mutates {@code info} with information from the given {@code comment}. */
242       private static void parseComment(Comment comment, FileInfo info) {
243         boolean fileOverview = comment.value.contains("@fileoverview");
244         for (CommentAnnotation annotation : CommentAnnotation.parse(comment.value)) {
245           switch (annotation.name) {
246             case "@fileoverview":
247             case "@author":
248             case "@see":
249             case "@link":
250               break;
251             case "@mods":
252               if (!annotation.value.isEmpty()) {
253                 info.mods.add(annotation.value);
254               }
255               break;
256             case "@visibility":
257               if (!annotation.value.isEmpty()) {
258                 info.visibility.add(annotation.value);
259               }
260               break;
261             case "@modName":
262               if (!annotation.value.isEmpty()) {
263                 info.modName.add(annotation.value);
264               }
265               break;
266             case "@config":
267               info.isConfig = true;
268               break;
269             case "@provideGoog":
270               info.provideGoog = true;
271               break;
272             case "@requirecss":
273               if (!annotation.value.isEmpty()) {
274                 info.requiresCss.add(annotation.value);
275               }
276               break;
277             case "@hassoydeltemplate":
278               if (!annotation.value.isEmpty()) {
279                 info.hasSoyDeltemplates.add(annotation.value);
280               }
281               break;
282             case "@hassoydelcall":
283               if (!annotation.value.isEmpty()) {
284                 info.hasSoyDelcalls.add(annotation.value);
285               }
286               break;
287             case "@externs":
288               info.isExterns = true;
289               break;
290             case "@enhanceable":
291             case "@pintomodule":
292               info.customAnnotations.add(
293                   JsArray.of(annotation.name.substring(1), annotation.value));
294               break;
295             case "@enhance":
296               if (!annotation.value.isEmpty()) {
297                 info.customAnnotations.add(
298                     JsArray.of(annotation.name.substring(1), annotation.value));
299               }
300               break;
301             default:
302               if (fileOverview) {
303                 info.customAnnotations.add(
304                     JsArray.of(annotation.name.substring(1), annotation.value));
305               }
306           }
307         }
308       }

310       /** Traverser that mutates {@code #info} with information from the AST. */
311       private static class Traverser extends AbstractPostOrderCallback {
312         final FileInfo info;
313     
314         Traverser(FileInfo info) {
315           this.info = info;
316         }
317     
318         @Override
319         public void visit(NodeTraversal traversal, Node node, Node parent) {
320           // Look for goog.* calls
321           if (node.isGetProp() && node.getFirstChild().isName()
322               && node.getFirstChild().getString().equals("goog")) {
323             Var root = traversal.getScope().getVar("goog");
324             if (root == null) {
325               info.goog = true;
326               if (parent.isCall() && parent.getChildCount() < 3) {
327                 Node arg;
328                 switch (node.getLastChild().getString()) {
329                   case "module":
330                     info.loadFlags.add(JsArray.of("module", "goog"));
331                     // fall through
332                   case "provide":
333                     arg = parent.getSecondChild();
334                     if (arg.isString()) {
335                       info.provides.add(arg.getString());
336                     } // TODO(sdh): else warning?
337                     break;
338                   case "require":
339                     arg = parent.getSecondChild();
340                     if (arg.isString()) {
341                       info.requires.add(arg.getString());
342                     } // TODO(sdh): else warning?
343                     break;
344                   case "setTestOnly":
345                     info.testonly = true;
346                     break;
347                   default:
348                     // Do nothing
349                 }
350               }
351             }
352           }
353     
354           // Look for ES6 import statements
355           if (node.isImport()) {
356             Node moduleSpecifier = node.getChildAtIndex(2);
357             // NOTE: previous tool was more forgiving here.
358             Preconditions.checkState(moduleSpecifier.isString());
359             info.loadFlags.add(JsArray.of("module", "es6"));
360             info.importedModules.add(moduleSpecifier.getString());
361           } else if (node.isExport()) {
362             info.loadFlags.add(JsArray.of("module", "es6"));
363           }
364         }
365       }

367       /** JS function interface for reporting errors. */
368       @JsFunction
369       public interface Reporter {
370         void report(boolean fatal, String message, String sourceName, int line, int lineOffset);
371       }
372     
373       private static final class DelegatingReporter implements ErrorReporter {
374         final Reporter delegate;
375     
376         DelegatingReporter(Reporter delegate) {
377           this.delegate = delegate != null ? delegate : NULL_REPORTER;
378         }
379     
380         @Override
381         public void warning(String message, String sourceName, int line, int lineOffset) {
382           delegate.report(false, message, sourceName, line, lineOffset);
383         }
384     
385         @Override
386         public void error(String message, String sourceName, int line, int lineOffset) {
387           delegate.report(true, message, sourceName, line, lineOffset);
388         }
389       }
390     
391       private static final Reporter NULL_REPORTER = new Reporter() {
392         @Override
393         public void report(
394             boolean fatal, String message, String sourceName, int line, int lineOffset) {}
395       };

398       public void onModuleLoad() {}

400       /** Returns an associative multimap. */
401       private static Set<JsArray<String>> assoc() {
402         return new TreeSet<>(
403             Ordering.<String>natural()
404                 .lexicographical()
405                 .onResultOf(
406                     new Function<JsArray<String>, List<String>>() {
407                       @Override public List<String> apply(JsArray<String> arg) {
408                         return arg.asList();
409                       }
410                     }));
411       }

413       /** Sparse object helper class: only adds non-trivial values. */
414       private static class SparseObject {
415         final JsObject<Object> object = new JsObject<>();
416     
417         SparseObject set(String key, Iterable<?> iterable) {
418           JsArray<?> array = JsArray.copyOf(iterable);
419           if (array.getLength() > 0) {
420             object.set(key, array);
421           }
422           return this;
423         }
424     
425         SparseObject set(String key, String value) {
426           if (value != null && !value.isEmpty()) {
427             object.set(key, value);
428           }
429           return this;
430         }
431     
432         SparseObject set(String key, boolean value) {
433           if (value) {
434             object.set(key, value);
435           }
436           return this;
437         }
438       }
