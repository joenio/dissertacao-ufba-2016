789       private Node parse(String source, String... warnings) {
790         TestErrorReporter testErrorReporter = new TestErrorReporter(null, warnings);
791         Config config =
792             ParserRunner.createConfig(
793                 mode,
794                 Config.JsDocParsing.INCLUDE_DESCRIPTIONS_NO_WHITESPACE,
795                 Config.SourceLocationInformation.PRESERVE,
796                 Config.RunMode.KEEP_GOING,
797                 null);
798         Node script = ParserRunner.parse(
799             new SimpleSourceFile("input", false),
800             source,
801             config,
802             testErrorReporter).ast;
803     
804         // verifying that all warnings were seen
805         testErrorReporter.assertHasEncounteredAllErrors();
806         testErrorReporter.assertHasEncounteredAllWarnings();
807     
808         return script;
809       }
