17     package com.google.javascript.jscomp;
18     
19     import com.google.javascript.jscomp.CompilerOptions.LanguageMode;
20     import com.google.javascript.jscomp.NodeTraversal.AbstractPostOrderCallback;
21     import com.google.javascript.rhino.IR;
22     import com.google.javascript.rhino.Node;
23     
24     /**
25      * Converts property accesses from quoted string syntax to dot syntax, where
26      * possible. Dot syntax is more compact and avoids an object allocation in
27      * IE 6.
28      *
29      */
30     class ConvertToDottedProperties extends AbstractPostOrderCallback
31         implements CompilerPass {
32     
33       private final AbstractCompiler compiler;
34     
35       ConvertToDottedProperties(AbstractCompiler compiler) {
36         this.compiler = compiler;
37       }
38     
39       @Override
40       public void process(Node externs, Node root) {
41         NodeTraversal.traverseEs6(compiler, root, this);
42       }
43     
44       @Override
45       public void visit(NodeTraversal t, Node n, Node parent) {
46         switch (n.getType()) {
47           case GETTER_DEF:
48           case SETTER_DEF:
49           case STRING_KEY:
50             if (NodeUtil.isValidPropertyName(LanguageMode.ECMASCRIPT3, n.getString())) {
51               n.putBooleanProp(Node.QUOTED_PROP, false);
52               compiler.reportCodeChange();
53             }
54             break;
55     
56           case GETELEM:
57             Node left = n.getFirstChild();
58             Node right = left.getNext();
59             if (right.isString() &&
60                 NodeUtil.isValidPropertyName(LanguageMode.ECMASCRIPT3, right.getString())) {
61               n.removeChild(left);
62               n.removeChild(right);
63               parent.replaceChild(n, IR.getprop(left, right));
64               compiler.reportCodeChange();
65             }
66             break;
67         }
68       }
69     }
