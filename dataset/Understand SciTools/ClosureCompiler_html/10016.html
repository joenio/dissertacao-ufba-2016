19     import com.google.common.collect.ArrayListMultimap;
20     import com.google.common.collect.ListMultimap;
21     import com.google.common.collect.Lists;
22     import com.google.common.collect.Multimaps;
23     import com.google.javascript.jscomp.NodeTraversal.Callback;
24     import com.google.javascript.rhino.IR;
25     import com.google.javascript.rhino.Node;
26     
27     import java.util.List;
28     import java.util.Map.Entry;

30     /**
31      * Moves top-level function declarations to the top.
32      *
33      * Enable this pass if a try catch block wraps the output after compilation,
34      * and the output runs on Firefox because function declarations are only
35      * defined when reached inside a try catch block on Firefox.
36      *
37      * On Firefox, this code works:
38      *
39      * var g = f;
40      * function f() {}
41      *
42      * but this code does not work:
43      *
44      * try {
45      *   var g = f;
46      *   function f() {}
47      * } catch(e) {}
48      *
49      * NOTE(dimvar):
50      * This pass is safe to turn on by default and delete the associated compiler
51      * option. However, we don't do that because the pass is only useful for code
52      * wrapped in a try/catch, and otherwise it makes debugging harder because it
53      * moves code around.
54      *
55      */
56     class MoveFunctionDeclarations implements Callback, CompilerPass {
57       private final AbstractCompiler compiler;
58       private final ListMultimap<JSModule, Node> functions;
59     
60       MoveFunctionDeclarations(AbstractCompiler compiler) {
61         this.compiler = compiler;
62         functions = ArrayListMultimap.create();
63       }
64     
65       @Override
66       public void process(Node externs, Node root) {
67         NodeTraversal.traverseEs6(compiler, root, this);
68         for (Entry<JSModule, List<Node>> entry : Multimaps.asMap(functions).entrySet()) {
69           Node addingRoot = compiler.getNodeForCodeInsertion(entry.getKey());
70           for (Node n : Lists.reverse(entry.getValue())) {
71             Node nameNode = n.getFirstChild();
72             String name = nameNode.getString();
73             nameNode.setString("");
74             addingRoot.addChildToFront(
75                 IR.var(IR.name(name), n).useSourceInfoIfMissingFromForTree(n));
76           }
77         }
78       }
79     
80       @Override
81       public boolean shouldTraverse(NodeTraversal t, Node n, Node parent) {
82         Node grandparent = n.getAncestor(2);
83         return grandparent == null || !grandparent.isScript();
84       }
85     
86       @Override
87       public void visit(NodeTraversal t, Node n, Node parent) {
88         if (parent == null || !parent.isScript()) {
89           return;
90         }
91     
92         if (NodeUtil.isFunctionDeclaration(n)) {
93           parent.removeChild(n);
94           compiler.reportCodeChange();
95     
96           functions.put(t.getModule(), n);
97         }
98       }
99     }
