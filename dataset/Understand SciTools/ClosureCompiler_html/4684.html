17     package com.google.javascript.jscomp;
18     
19     /**
20      * Tests for {@link ClosureOptimizePrimitives}.
21      *
22      * @author agrieve@google.com (Andrew Grieve)
23      */
24     public final class ClosureOptimizePrimitivesTest extends CompilerTestCase {
25     
26       @Override public CompilerPass getProcessor(final Compiler compiler) {
27         return new ClosureOptimizePrimitives(compiler);
28       }
29     
30       public void testObjectCreateNonConstKey() {
31         testSame("goog.object.create('a',1,2,3,foo,bar);");
32       }
33     
34       public void testObjectCreateOddParams() {
35         testSame("goog.object.create('a',1,2);");
36       }
37     
38       public void testObjectCreate1() {
39         test("var a = goog.object.create()", "var a = {}");
40       }
41     
42       public void testObjectCreate2() {
43         test("var a = goog$object$create('b',goog$object$create('c','d'))",
44              "var a = {'b':{'c':'d'}};");
45       }
46     
47       public void testObjectCreate3() {
48         test("var a = goog.object.create(1,2)", "var a = {1:2}");
49       }
50     
51       public void testObjectCreate4() {
52         test("alert(goog.object.create(1,2).toString())",
53              "alert({1:2}.toString())");
54       }
55     
56       public void testObjectCreate5() {
57         test("goog.object.create('a',2).toString()", "({'a':2}).toString()");
58       }
59     
60       public void testObjectCreateSetNonConstKey() {
61         testSame("goog.object.createSet('a',1,2,3,foo,bar);");
62       }
63     
64       public void testObjectCreateSet1() {
65         test("var a = goog.object.createSet()", "var a = {}");
66       }
67     
68       public void testObjectCreateSet2() {
69         test("var a = goog.object.createSet(1,2)", "var a = {1:true,2:true}");
70       }
71     
72       public void testObjectCreateSet3() {
73         test("alert(goog.object.createSet(1).toString())",
74              "alert({1:true}.toString())");
75       }
76     
77       public void testObjectCreateSet4() {
78         test("goog.object.createSet('a').toString()", "({'a':true}).toString()");
79       }
80     }
