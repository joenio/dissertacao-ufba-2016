33       // Possible enhancements:
34       // * verify NAME, LABEL_NAME, GETPROP property name and unquoted

40       /** Violation handler */
41       public interface ViolationHandler {
42         void handleViolation(String message, Node n);
43       }
44     
45       private final AbstractCompiler compiler;
46       private final ViolationHandler violationHandler;
47     
48       public AstValidator(AbstractCompiler compiler, ViolationHandler handler) {
49         this.compiler = compiler;
50         this.violationHandler = handler;
51       }
52     
53       public AstValidator(AbstractCompiler compiler) {
54         this(compiler, new ViolationHandler() {
55           @Override
56           public void handleViolation(String message, Node n) {
57             throw new IllegalStateException(
58                 message + ". Reference node:\n" + n.toStringTree()
59                 + "\n Parent node:\n"
60                 + ((n.getParent() != null) ? n.getParent().toStringTree() : " no parent "));
61           }
62         });
63       }
64     
65       @Override
66       public void process(Node externs, Node root) {
67         if (externs != null) {
68           validateCodeRoot(externs);
69         }
70         if (root != null) {
71           validateCodeRoot(root);
72         }
73       }

75       public void validateRoot(Node n) {
76         validateNodeType(Token.BLOCK, n);
77         validateIsSynthetic(n);
78         validateChildCount(n, 2);
79         validateCodeRoot(n.getFirstChild());
80         validateCodeRoot(n.getLastChild());
81       }

83       public void validateCodeRoot(Node n) {
84         validateNodeType(Token.BLOCK, n);
85         validateIsSynthetic(n);
86         for (Node c = n.getFirstChild(); c != null; c = c.getNext()) {
87           validateScript(c);
88         }
89       }

91       public void validateScript(Node n) {
92         validateNodeType(Token.SCRIPT, n);
93         validateHasSourceName(n);
94         validateHasInputId(n);
95         for (Node c = n.getFirstChild(); c != null; c = c.getNext()) {
96           validateStatement(c);
97         }
98       }

101         validateStatement(n, false);

104       public void validateStatement(Node n, boolean isAmbient) {
105         switch (n.getType()) {
106           case LABEL:
107             validateLabel(n);
108             return;
109           case BLOCK:
110             validateBlock(n);
111             return;
112           case FUNCTION:
113             if (isAmbient) {
114               validateFunctionSignature(n);
115             } else {
116               validateFunctionStatement(n);
117             }
118             return;
119           case WITH:
120             validateWith(n);
121             return;
122           case FOR:
123             validateFor(n);
124             return;
125           case FOR_OF:
126             validateForOf(n);
127             return;
128           case WHILE:
129             validateWhile(n);
130             return;
131           case DO:
132             validateDo(n);
133             return;
134           case SWITCH:
135             validateSwitch(n);
136             return;
137           case IF:
138             validateIf(n);
139             return;
140           case VAR:
141           case LET:
142           case CONST:
143             validateNameDeclarationHelper(n.getType(), n);
144             return;
145           case EXPR_RESULT:
146             validateExprStmt(n);
147             return;
148           case RETURN:
149             validateReturn(n);
150             return;
151           case THROW:
152             validateThrow(n);
153             return;
154           case TRY:
155             validateTry(n);
156             return;
157           case BREAK:
158             validateBreak(n);
159             return;
160           case CONTINUE:
161             validateContinue(n);
162             return;
163           case EMPTY:
164           case DEBUGGER:
165             validateChildless(n);
166             return;
167           case CLASS:
168             validateClassDeclaration(n, isAmbient);
169             return;
170           case IMPORT:
171             validateImport(n);
172             return;
173           case EXPORT:
174             validateExport(n, isAmbient);
175             return;
176           case INTERFACE:
177             validateInterface(n);
178             return;
179           case ENUM:
180             validateEnum(n);
181             return;
182           case TYPE_ALIAS:
183             validateTypeAlias(n);
184             return;
185           case DECLARE:
186             validateAmbientDeclaration(n);
187             return;
188           case NAMESPACE:
189             validateNamespace(n, isAmbient);
190             return;
191           default:
192             violation("Expected statement but was " + n.getType() + ".", n);
193         }
194       }

201           case NULL:
202           case SUPER:

204           case TRUE:
205             validateChildless(n);
206             return;
207

209           case DELPROP:
210           case POS:
211           case NEG:
212           case NOT:
213           case INC:
214           case DEC:
215           case TYPEOF:
216           case VOID:
217           case BITNOT:

219             validateUnaryOp(n);

240             return;

249             return;
250

253             return;
254

262           case BITAND:

274           case RSH:

290             return;
291

293             validateObjectLit(n);
294             return;
295     
296           case REGEXP:

298             return;
299

302             return;
303

306             return;
307     
308           case NEW:

311

313             validateFunctionExpression(n);
314             return;
315

337       private void validateYield(Node n) {
338         validateEs6Feature("yield", n);
339         validateNodeType(Token.YIELD, n);
340         validateChildCountIn(n, 0, 1);
341         if (n.hasChildren()) {
342           validateExpression(n.getFirstChild());
343         }

346       private void validateImport(Node n) {
347         validateEs6Feature("import statement", n);
348         validateNodeType(Token.IMPORT, n);
349         validateChildCount(n);
350     
351         if (n.getFirstChild().isName()) {
352           validateName(n.getFirstChild());
353         } else {
354           validateNodeType(Token.EMPTY, n.getFirstChild());
355         }
356     
357         Node secondChild = n.getSecondChild();
358         switch (secondChild.getType()) {
359           case IMPORT_SPECS:
360             validateImportSpecifiers(secondChild);
361             break;
362           case IMPORT_STAR:
363             validateNonEmptyString(secondChild);
364             break;
365           default:
366             validateNodeType(Token.EMPTY, secondChild);
367         }
368     
369         validateString(n.getChildAtIndex(2));
370       }
371     
372       private void validateImportSpecifiers(Node n) {
373         validateNodeType(Token.IMPORT_SPECS, n);
374         for (Node child : n.children()) {
375           validateImportSpecifier(child);
376         }
377       }
378     
379       private void validateImportSpecifier(Node n) {
380         validateNodeType(Token.IMPORT_SPEC, n);
381         validateChildCountIn(n, 1, 2);
382         for (Node child : n.children()) {
383           validateName(child);
384         }
385       }

387       private void validateExport(Node n, boolean isAmbient) {
388         validateNodeType(Token.EXPORT, n);
389         if (n.getBooleanProp(Node.EXPORT_ALL_FROM)) { // export * from "mod"
390           validateChildCount(n, 2);
391           validateNodeType(Token.EMPTY, n.getFirstChild());
392           validateString(n.getSecondChild());
393         } else if (n.getBooleanProp(Node.EXPORT_DEFAULT)) { // export default foo = 2
394           validateChildCount(n, 1);
395           validateExpression(n.getFirstChild());
396         } else {
397           validateChildCountIn(n, 1, 2);
398           if (n.getFirstChild().getType() == Token.EXPORT_SPECS) {
399             validateExportSpecifiers(n.getFirstChild());
400           } else {
401             validateStatement(n.getFirstChild(), isAmbient);
402           }
403           if (n.getChildCount() == 2) {
404             validateString(n.getSecondChild());
405           }
406         }
407       }

410         validateNodeType(Token.EXPORT_SPECS, n);
411         for (Node child : n.children()) {
412           validateExportSpecifier(child);
413         }
414       }

418         validateChildCountIn(n, 1, 2);

420           validateName(child);
421         }

424       private void validateTaggedTemplateLit(Node n) {
425         validateEs6Feature("template literal", n);
426         validateNodeType(Token.TAGGED_TEMPLATELIT, n);
427         validateChildCount(n);
428         validateExpression(n.getFirstChild());
429         validateTemplateLit(n.getLastChild());
430       }
431     
432       private void validateTemplateLit(Node n) {
433         validateEs6Feature("template literal", n);
434         validateNodeType(Token.TEMPLATELIT, n);
435         if (!n.hasChildren()) {
436           return;
437         }
438         for (int i = 0; i < n.getChildCount(); i++) {
439           Node child = n.getChildAtIndex(i);
440           // If the first child is not a STRING, this is a tagged template.
441           if (i == 0 && !child.isString()) {
442             validateExpression(child);
443           } else if (child.isString()) {
444             validateString(child);
445           } else {
446             validateTemplateLitSub(child);
447           }
448         }
449       }
450     
451       private void validateTemplateLitSub(Node n) {
452         validateNodeType(Token.TEMPLATELIT_SUB, n);
453         validateChildCount(n);
454         validateExpression(n.getFirstChild());
455       }

457       private void validateInterface(Node n) {
458         validateEs6TypedFeature("interface", n);
459         validateNodeType(Token.INTERFACE, n);
460         validateChildCount(n);
461         Node name = n.getFirstChild();
462         validateName(name);
463         Node superTypes = name.getNext();
464         if (superTypes.isEmpty()) {
465           validateChildless(superTypes);
466         } else {
467           validateInterfaceExtends(superTypes);
468         }
469         validateInterfaceMembers(n.getLastChild());
470       }
471     
472       private void validateInterfaceExtends(Node n) {
473         validateNodeType(Token.INTERFACE_EXTENDS, n);
474         for (Node child : n.children()) {
475           validateNamedType(child);
476         }
477       }
478     
479       private void validateInterfaceMembers(Node n) {
480         validateNodeType(Token.INTERFACE_MEMBERS, n);
481         for (Node child : n.children()) {
482           validateInterfaceMember(child);
483         }
484       }
485     
486       private void validateInterfaceMember(Node n) {
487         switch (n.getType()) {
488           case MEMBER_FUNCTION_DEF:
489             validateChildCount(n);
490             validateFunctionSignature(n.getFirstChild());
491             break;
492           case MEMBER_VARIABLE_DEF:
493             validateChildless(n);
494             break;
495           case INDEX_SIGNATURE:
496             validateChildCount(n);
497             validateChildless(n.getFirstChild());
498             break;
499           case CALL_SIGNATURE:
500             validateChildCount(n);
501             break;
502           default:
503             violation("Interface contained member of invalid type " + n.getType(), n);
504         }
505       }

508         validateNodeType(Token.ENUM, n);
509         validateName(n.getFirstChild());
510         validateEnumMembers(n.getLastChild());
511       }

517         }

533       private void validateClassHelper(Node n, boolean isAmbient) {
534         validateEs6Feature("classes", n);
535         validateNodeType(Token.CLASS, n);
536         validateChildCount(n);
537     
538         Node name = n.getFirstChild();
539         if (name.isEmpty()) {
540           validateChildless(name);
541         } else {
542           validateName(name);
543         }
544     
545         Node superClass = name.getNext();
546         if (superClass.isEmpty()) {
547           validateChildless(superClass);
548         } else {
549           validateExpression(superClass);
550         }
551     
552         validateClassMembers(n.getLastChild(), isAmbient);
553       }
554     
555       private void validateClassMembers(Node n, boolean isAmbient) {
556         validateNodeType(Token.CLASS_MEMBERS, n);
557         for (Node c : n.children()) {
558           validateClassMember(c, isAmbient);
559         }
560       }
561     
562       private void validateClassMember(Node n, boolean isAmbient) {
563         switch (n.getType()) {
564           case MEMBER_FUNCTION_DEF:
565           case GETTER_DEF:
566           case SETTER_DEF:
567             validateChildCount(n);
568             Node function = n.getFirstChild();
569             if (isAmbient) {
570               validateFunctionSignature(function);
571             } else {
572               validateFunctionExpression(function);
573             }
574             break;
575           case MEMBER_VARIABLE_DEF:
576             validateChildless(n);
577             break;
578           case COMPUTED_PROP:
579             validateComputedPropClassMethod(n);
580             break;
581           case INDEX_SIGNATURE:
582             validateChildCount(n);
583             validateChildless(n.getFirstChild());
584             break;
585           case CALL_SIGNATURE:
586             validateChildCount(n);
587             break;
588           case EMPTY: // Empty is allowed too.
589             break;
590           default:
591             violation("Class contained member of invalid type " + n.getType(), n);
592         }
593       }

697       private void validateFunctionExpressionHelper(Node n, boolean isAmbient) {
698         validateNodeType(Token.FUNCTION, n);
699         validateChildCount(n);
700     
701         validateParameters(n.getSecondChild());
702     
703         Node name = n.getFirstChild();
704         Node body = n.getLastChild();
705         if (n.isArrowFunction()) {
706           validateEs6Feature("arrow functions", n);
707           validateEmptyName(name);
708           if (body.getType() == Token.BLOCK) {
709             validateBlock(body);
710           } else {
711             validateExpression(body);
712           }
713         } else {
714           validateOptionalName(name);
715           validateFunctionBody(body, isAmbient);
716         }
717       }
718     
719       private void validateFunctionBody(Node n, boolean noBlock) {
720         if (noBlock) {
721           validateNodeType(Token.EMPTY, n);
722         } else {
723           validateBlock(n);
724         }
725       }

727       private void validateParameters(Node n) {
728         validateNodeType(Token.PARAM_LIST, n);
729     
730         if (isEs6OrHigher()) {
731           validateParametersEs6(n);
732         } else {
733           validateParametersEs5(n);
734         }
735       }

737       private void validateParametersEs5(Node n) {
738         for (Node c = n.getFirstChild(); c != null; c = c.getNext()) {
739           validateName(c);
740         }
741       }
742     
743       private void validateParametersEs6(Node n) {
744         boolean defaultParams = false;
745         for (Node c = n.getFirstChild(); c != null; c = c.getNext()) {
746           if (c.isRest()) {
747             if (c.getNext() != null) {
748               violation("Rest parameters must come after all other parameters.", c);
749             }
750             validateRest(Token.PARAM_LIST, c);
751           } else if (c.isDefaultValue()) {
752             defaultParams = true;
753             validateDefaultValue(Token.PARAM_LIST, c);
754           } else {
755             if (defaultParams) {
756               violation("Cannot have a parameter without a default value,"
757                   + " after one with a default value.", c);
758             }
759     
760             if (c.isName()) {
761               validateName(c);
762             } else if (c.isArrayPattern()) {
763               validateArrayPattern(Token.PARAM_LIST, c);
764             } else {
765               validateObjectPattern(Token.PARAM_LIST, c);
766             }
767           }
768         }
769       }
770     
771       private void validateDefaultValue(Token type, Node n) {
772         validateAssignmentExpression(n);
773         Node lhs = n.getFirstChild();
774     
775         // LHS can only be a name or destructuring pattern.
776         if (lhs.isName()) {
777           validateName(lhs);
778         } else if (lhs.isArrayPattern()) {
779           validateArrayPattern(type, lhs);
780         } else {
781           validateObjectPattern(type, lhs);
782         }
783       }

785       private void validateCall(Node n) {
786         validateNodeType(Token.CALL, n);
787         validateMinimumChildCount(n, 1);
788         for (Node c = n.getFirstChild(); c != null; c = c.getNext()) {
789           validateExpression(c);
790         }
791       }

794        * @param contextType A {@link Token} constant value indicating that {@code n} should be validated
795        *     appropriately for a descendant of a {@link Node} of this type.
796        * @param n
797        */
798       private void validateRest(Token contextType, Node n) {
799         validateNodeType(Token.REST, n);
800         validateChildCount(n);
801         validateLHS(contextType, n.getFirstChild());
802       }

804       private void validateSpread(Node n) {
805         validateNodeType(Token.SPREAD, n);
806         validateChildCount(n);
807         Node parent = n.getParent();
808         switch (parent.getType()) {
809           case CALL:
810           case NEW:
811             if (n == parent.getFirstChild()) {
812               violation("SPREAD node is not callable.", n);
813             }
814             break;
815           case ARRAYLIT:
816             break;
817           default:
818             violation("SPREAD node should not be the child of a " + parent.getType() + " node.", n);
819         }
820       }

822       private void validateNew(Node n) {
823         validateNodeType(Token.NEW, n);

830       private void validateNameDeclarationHelper(Token type, Node n) {
831         validateMinimumChildCount(n, 1);
832         for (Node c = n.getFirstChild(); c != null; c = c.getNext()) {
833           validateNameDeclarationChild(type, c);
834         }
835       }

837       private void validateNameDeclarationChild(Token type, Node n) {
838         if (n.isName()) {
839           validateLHS(type, n);
840         } else if (n.isDestructuringLhs()) {
841           validateLHS(type, n.getFirstChild());
842         } else {
843           violation("Invalid child for " + type + " node", n);
844         }
845       }

847       /**
848        * @param contextType A {@link Token} constant value indicating that {@code n} should be validated
849        *     appropriately for a descendant of a {@link Node} of this type.
850        * @param n
851        */
852       private void validateLHS(Token contextType, Node n) {
853         if (n.isName()) {
854           // Don't use validateName here since this NAME node may have
855           // a child.
856           validateNonEmptyString(n);
857           validateMaximumChildCount(n, 1);
858           if (n.hasChildren()) {
859             validateExpression(n.getFirstChild());
860           }
861         } else if (n.isArrayPattern()) {
862           validateArrayPattern(contextType, n);
863         } else if (n.isObjectPattern()) {
864           validateObjectPattern(contextType, n);
865         } else if (n.isDefaultValue()) {
866           validateDefaultValue(contextType, n);
867         } else if (n.isComputedProp()) {
868           validateObjectPatternComputedPropKey(contextType, n);
869         } else {
870           violation("Invalid child for " + contextType + " node", n);
871         }
872       }
873     
874       private void validateArrayPattern(Token type, Node n) {
875         validateNodeType(Token.ARRAY_PATTERN, n);
876         for (Node c = n.getFirstChild(); c != null; c = c.getNext()) {
877           // When the array pattern is a direct child of a var/let/const node,
878           // the last element is the RHS of the assignment.
879           if (c == n.getLastChild() && NodeUtil.isNameDeclaration(n.getParent())) {
880             validateExpression(c);
881           } else if (c.isRest()) {
882             validateRest(type, c);
883           } else if (c.isEmpty()) {
884             validateChildless(c);
885           } else {
886             // The members of the array pattern can be simple names,
887             // or nested array/object patterns, e.g. "var [a,[b,c]]=[1,[2,3]];"
888             validateLHS(type, c);
889           }
890         }
891       }
892     
893       private void validateObjectPattern(Token type, Node n) {
894         validateNodeType(Token.OBJECT_PATTERN, n);
895         for (Node c = n.getFirstChild(); c != null; c = c.getNext()) {
896           // When the object pattern is a direct child of a var/let/const node,
897           // the last element is the RHS of the assignment.
898           if (c == n.getLastChild() && NodeUtil.isNameDeclaration(n.getParent())) {
899             validateExpression(c);
900           } else if (c.isStringKey()) {
901             validateObjectPatternStringKey(type, c);
902           } else {
903             // Nested destructuring pattern.
904             validateLHS(type, c);
905           }
906         }
907       }

909       private void validateFor(Node n) {
910         validateNodeType(Token.FOR, n);
911         if (NodeUtil.isForIn(n)) {
912           // FOR-IN
913           validateChildCount(n, 3);
914           validateVarOrAssignmentTarget(n.getFirstChild());
915           validateExpression(n.getSecondChild());
916         } else {
917           // FOR
918           validateChildCount(n, 4);
919           validateVarOrOptionalExpression(n.getFirstChild());
920           validateOptionalExpression(n.getSecondChild());
921           validateOptionalExpression(n.getChildAtIndex(2));
922         }
923         validateBlock(n.getLastChild());
924       }

1019       private void validateTry(Node n) {
1020         validateNodeType(Token.TRY, n);
1021         validateChildCountIn(n, 2, 3);
1022         validateBlock(n.getFirstChild());
1023     
1024         boolean seenCatchOrFinally = false;
1025     
1026         // Validate catch
1027         Node catches = n.getSecondChild();
1028         validateNodeType(Token.BLOCK, catches);
1029         validateMaximumChildCount(catches, 1);
1030         if (catches.hasChildren()) {
1031           validateCatch(catches.getFirstChild());
1032           seenCatchOrFinally = true;
1033         }
1034     
1035         // Validate finally
1036         if (n.getChildCount() == 3) {
1037           validateBlock(n.getLastChild());
1038           seenCatchOrFinally = true;
1039         }
1040     
1041         if (!seenCatchOrFinally) {
1042           violation("Missing catch or finally for try statement.", n);
1043         }
1044       }

1050         validateBlock(n.getLastChild());
1051       }

1053       private void validateSwitch(Node n) {
1054         validateNodeType(Token.SWITCH, n);
1055         validateMinimumChildCount(n, 1);
1056         validateExpression(n.getFirstChild());
1057         int defaults = 0;
1058         for (Node c = n.getSecondChild(); c != null; c = c.getNext()) {
1059           validateSwitchMember(n.getLastChild());
1060           if (c.isDefaultCase()) {
1061             defaults++;
1062           }
1063         }
1064         if (defaults > 1) {
1065           violation("Expected at most 1 'default' in switch but was "
1066               + defaults, n);
1067         }
1068       }

1077             return;

1080         }
1081       }

1174       private void validateObjectLitKey(Node n) {
1175         switch (n.getType()) {
1176           case GETTER_DEF:
1177             validateObjectLitGetKey(n);
1178             return;
1179           case SETTER_DEF:
1180             validateObjectLitSetKey(n);
1181             return;
1182           case STRING_KEY:
1183             validateObjectLitStringKey(n);
1184             return;
1185           case MEMBER_FUNCTION_DEF:
1186             validateClassMember(n, false);
1187             if (n.isStaticMember()) {
1188               violation("Keys in an object literal should not be static.", n);
1189             }
1190             return;
1191           case COMPUTED_PROP:
1192             validateObjectLitComputedPropKey(n);
1193             return;
1194           default:
1195             violation("Expected object literal key expression but was " + n.getType(), n);
1196         }
1197       }

1199       private void validateObjectLitGetKey(Node n) {
1200         validateNodeType(Token.GETTER_DEF, n);
1201         validateChildCount(n);
1202         validateObjectLiteralKeyName(n);
1203         Node function = n.getFirstChild();
1204         validateFunctionExpression(function);
1205         // objlit get functions must be nameless, and must have zero parameters.
1206         if (!function.getFirstChild().getString().isEmpty()) {
1207           violation("Expected unnamed function expression.", n);
1208         }
1209         Node functionParams = function.getSecondChild();
1210         if (functionParams.hasChildren()) {
1211           violation("get methods must not have parameters.", n);
1212         }
1213       }

1216         validateNodeType(Token.SETTER_DEF, n);
1217         validateChildCount(n);

1219         Node function = n.getFirstChild();
1220         validateFunctionExpression(function);
1221         // objlit set functions must be nameless, and must have 1 parameter.
1222         if (!function.getFirstChild().getString().isEmpty()) {
1223           violation("Expected unnamed function expression.", n);
1224         }
1225         Node functionParams = function.getSecondChild();
1226         if (!functionParams.hasOneChild()) {
1227           violation("set methods must have exactly one parameter.", n);
1228         }
1229       }

1411         int count = n.getChildCount();
1412         if (expected != count) {
1413           violation(
1414               "Expected " + expected + " children, but was "
1415                   + n.getChildCount(), n);
1416         }

1419       private void validateChildCountIn(Node n, int min, int max) {
1420         int count = n.getChildCount();
1421         if (count < min || count > max) {
1422           violation("Expected child count in [" + min + ", " + max
1423               + "], but was " + count, n);
1424         }
1425       }

1427       private void validateMinimumChildCount(Node n, int i) {
1428         boolean valid = false;
1429         if (i == 1) {
1430           valid = n.hasChildren();
1431         } else if (i == 2) {
1432           valid = n.hasMoreThanOneChild();
1433         } else {
1434           valid = n.getChildCount() >= i;
1435         }
1436     
1437         if (!valid) {
1438           violation(
1439               "Expected at least " + i + " children, but was "
1440                   + n.getChildCount(), n);
1441         }
1442       }
1443     
1444       private void validateMaximumChildCount(Node n, int i) {
1445         boolean valid = false;
1446         if (i == 1) {
1447           valid = !n.hasMoreThanOneChild();
1448         } else if (i == -1) {
1449           valid = true;  // Varying number of children.
1450         } else {
1451           valid = n.getChildCount() <= i;
1452         }
1453         if (!valid) {
1454           violation(
1455               "Expected no more than " + i + " children, but was "
1456                   + n.getChildCount(), n);
1457         }
1458       }
