179        * Parse a description as a {@code @type}.

181       public JSDocInfo parseInlineTypeDoc() {
182         skipEOLs();
183     
184         JsDocToken token = next();
185         int lineno = stream.getLineno();
186         int startCharno = stream.getCharno();
187         Node typeAst = parseParamTypeExpression(token);
188         recordTypeNode(lineno, startCharno, typeAst, token == JsDocToken.LEFT_CURLY);
189     
190         JSTypeExpression expr = createJSTypeExpression(typeAst);
191         if (expr != null) {
192           jsdocBuilder.recordType(expr);
193           jsdocBuilder.recordInlineType();
194           return retrieveAndResetParsedJSDocInfo();
195         }
196         return null;
197       }

244       /**
245        * Parses a {@link JSDocInfo} object. This parsing method reads all tokens
246        * returned by the {@link JsDocTokenStream#getJsDocToken()} method until the
247        * {@link JsDocToken#EOC} is returned.
248        *
249        * @return {@code true} if JSDoc information was correctly parsed,
250        *     {@code false} otherwise
251        */
252       boolean parse() {
253         state = State.SEARCHING_ANNOTATION;
254         skipEOLs();
255     
256         JsDocToken token = next();
257     
258         // Always record that we have a comment.
259         if (jsdocBuilder.shouldParseDocumentation()) {
260           ExtractionInfo blockInfo = extractBlockComment(token);
261           token = blockInfo.token;
262           if (!blockInfo.string.isEmpty()) {
263             jsdocBuilder.recordBlockDescription(blockInfo.string);
264           }
265         } else {
266           if (token != JsDocToken.ANNOTATION &&
267               token != JsDocToken.EOC) {
268             // Mark that there was a description, but don't bother marking
269             // what it was.
270             jsdocBuilder.recordBlockDescription("");
271           }
272         }
273     
274         return parseHelperLoop(token, new ArrayList<ExtendedTypeInfo>());
275       }

277       private boolean parseHelperLoop(JsDocToken token,
278                                       List<ExtendedTypeInfo> extendedTypes) {
279         while (true) {
280           switch (token) {
281             case ANNOTATION:
282               if (state == State.SEARCHING_ANNOTATION) {
283                 state = State.SEARCHING_NEWLINE;
284                 token = parseAnnotation(token, extendedTypes);
285               } else {
286                 token = next();
287               }
288               break;
289     
290             case EOC:
291               boolean success = true;
292               // TODO(johnlenz): It should be a parse error to have an @extends
293               // or similiar annotations in a file overview block.
294               checkExtendedTypes(extendedTypes);
295               if (hasParsedFileOverviewDocInfo()) {
296                 fileOverviewJSDocInfo = retrieveAndResetParsedJSDocInfo();
297                 Visibility visibility = fileOverviewJSDocInfo.getVisibility();
298                 switch (visibility) {
299                   case PRIVATE: // fallthrough
300                   case PROTECTED:
301                     // PRIVATE and PROTECTED are not allowed in @fileoverview JsDoc.
302                     addParserWarning(
303                         "msg.bad.fileoverview.visibility.annotation",
304                         visibility.toString().toLowerCase());
305                     success = false;
306                     break;
307                   default:
308                     // PACKAGE, PUBLIC, and (implicitly) INHERITED are allowed
309                     // in @fileoverview JsDoc.
310                     break;
311                 }
312               }
313               return success;
314     
315             case EOF:
316               // discard any accumulated information
317               jsdocBuilder.build();
318               addParserWarning("msg.unexpected.eof");
319               checkExtendedTypes(extendedTypes);
320               return false;
321     
322             case EOL:
323               if (state == State.SEARCHING_NEWLINE) {
324                 state = State.SEARCHING_ANNOTATION;
325               }
326               token = next();
327               break;
328     
329             default:
330               if (token == JsDocToken.STAR && state == State.SEARCHING_ANNOTATION) {
331                 token = next();
332               } else {
333                 state = State.SEARCHING_NEWLINE;
334                 token = eatTokensUntilEOL();
335               }
336               break;
337           }
338         }
339       }

341       private JsDocToken parseAnnotation(JsDocToken token,
342           List<ExtendedTypeInfo> extendedTypes) {
343         // JSTypes are represented as Rhino AST nodes, and then resolved later.
344         JSTypeExpression type;
345         int lineno = stream.getLineno();
346         int charno = stream.getCharno();
347     
348         String annotationName = stream.getString();
349         Annotation annotation = annotationNames.get(annotationName);
350         if (annotation == null || annotationName.isEmpty()) {
351           addParserWarning("msg.bad.jsdoc.tag", annotationName);
352         } else {
353           // Mark the beginning of the annotation.
354           jsdocBuilder.markAnnotation(annotationName, lineno, charno);
355     
356           switch (annotation) {
357             case NG_INJECT:
358               if (jsdocBuilder.isNgInjectRecorded()) {
359                 addParserWarning("msg.jsdoc.nginject.extra");
360               } else {
361                 jsdocBuilder.recordNgInject(true);
362               }
363               return eatUntilEOLIfNotAnnotation();
364     
365             case JAGGER_INJECT:
366               if (jsdocBuilder.isJaggerInjectRecorded()) {
367                 addParserWarning("msg.jsdoc.jaggerInject.extra");
368               } else {
369                 jsdocBuilder.recordJaggerInject(true);
370               }
371               return eatUntilEOLIfNotAnnotation();
372     
373             case JAGGER_MODULE:
374               if (jsdocBuilder.isJaggerModuleRecorded()) {
375                 addParserWarning("msg.jsdoc.jaggerModule.extra");
376               } else {
377                 jsdocBuilder.recordJaggerModule(true);
378               }
379               return eatUntilEOLIfNotAnnotation();
380     
381             case JAGGER_PROVIDE:
382               if (jsdocBuilder.isJaggerProvideRecorded()) {
383                 addParserWarning("msg.jsdoc.jaggerProvide.extra");
384               } else {
385                 jsdocBuilder.recordJaggerProvide(true);
386               }
387               return eatUntilEOLIfNotAnnotation();
388     
389             case JAGGER_PROVIDE_PROMISE:
390               if (jsdocBuilder.isJaggerProvidePromiseRecorded()) {
391                 addParserWarning("msg.jsdoc.jaggerProvidePromise.extra");
392               } else {
393                 jsdocBuilder.recordJaggerProvidePromise(true);
394               }
395               return eatUntilEOLIfNotAnnotation();
396     
397             case ABSTRACT:
398               if (!jsdocBuilder.recordAbstract()) {
399                 addTypeWarning("msg.jsdoc.incompat.type");
400               }
401               return eatUntilEOLIfNotAnnotation();
402     
403             case AUTHOR:
404               if (jsdocBuilder.shouldParseDocumentation()) {
405                 ExtractionInfo authorInfo = extractSingleLineBlock();
406                 String author = authorInfo.string;
407     
408                 if (author.isEmpty()) {
409                   addParserWarning("msg.jsdoc.authormissing");
410                 } else {
411                   jsdocBuilder.addAuthor(author);
412                 }
413                 token = authorInfo.token;
414               } else {
415                 token = eatUntilEOLIfNotAnnotation();
416               }
417               return token;
418     
419             case CONSISTENTIDGENERATOR:
420               if (!jsdocBuilder.recordConsistentIdGenerator()) {
421                 addParserWarning("msg.jsdoc.consistidgen");
422               }
423               return eatUntilEOLIfNotAnnotation();
424     
425             case UNRESTRICTED:
426               if (!jsdocBuilder.recordUnrestricted()) {
427                 addTypeWarning("msg.jsdoc.incompat.type");
428               }
429               return eatUntilEOLIfNotAnnotation();
430     
431             case STRUCT:
432               if (!jsdocBuilder.recordStruct()) {
433                 addTypeWarning("msg.jsdoc.incompat.type");
434               }
435               return eatUntilEOLIfNotAnnotation();
436     
437             case DICT:
438               if (!jsdocBuilder.recordDict()) {
439                 addTypeWarning("msg.jsdoc.incompat.type");
440               }
441               return eatUntilEOLIfNotAnnotation();
442     
443             case CONSTRUCTOR:
444               if (!jsdocBuilder.recordConstructor()) {
445                 if (jsdocBuilder.isInterfaceRecorded()) {
446                   addTypeWarning("msg.jsdoc.interface.constructor");
447                 } else {
448                   addTypeWarning("msg.jsdoc.incompat.type");
449                 }
450               }
451               return eatUntilEOLIfNotAnnotation();
452     
453             case RECORD:
454               if (!jsdocBuilder.recordImplicitMatch()) {
455                 addTypeWarning("msg.jsdoc.record");
456               }
457               return eatUntilEOLIfNotAnnotation();
458     
459             case DEPRECATED:
460               if (!jsdocBuilder.recordDeprecated()) {
461                 addParserWarning("msg.jsdoc.deprecated");
462               }
463     
464               // Find the reason/description, if any.
465               ExtractionInfo reasonInfo = extractMultilineTextualBlock(token);
466     
467               String reason = reasonInfo.string;
468     
469               if (reason.length() > 0) {
470                 jsdocBuilder.recordDeprecationReason(reason);
471               }
472     
473               token = reasonInfo.token;
474               return token;
475     
476             case INTERFACE:
477               if (!jsdocBuilder.recordInterface()) {
478                 if (jsdocBuilder.isConstructorRecorded()) {
479                   addTypeWarning("msg.jsdoc.interface.constructor");
480                 } else {
481                   addTypeWarning("msg.jsdoc.incompat.type");
482                 }
483               }
484               return eatUntilEOLIfNotAnnotation();
485     
486             case DESC:
487               if (jsdocBuilder.isDescriptionRecorded()) {
488                 addParserWarning("msg.jsdoc.desc.extra");
489                 return eatUntilEOLIfNotAnnotation();
490               } else {
491                 ExtractionInfo descriptionInfo =
492                     extractMultilineTextualBlock(token);
493     
494                 String description = descriptionInfo.string;
495     
496                 jsdocBuilder.recordDescription(description);
497                 token = descriptionInfo.token;
498                 return token;
499               }
500     
501             case FILE_OVERVIEW:
502               String fileOverview = "";
503               if (jsdocBuilder.shouldParseDocumentation() && !lookAheadForAnnotation()) {
504                 ExtractionInfo fileOverviewInfo = extractMultilineTextualBlock(
505                     token, getWhitespaceOption(WhitespaceOption.TRIM), false);
506     
507                 fileOverview = fileOverviewInfo.string;
508     
509                 token = fileOverviewInfo.token;
510               } else {
511                 token = eatUntilEOLIfNotAnnotation();
512               }
513     
514               if (!jsdocBuilder.recordFileOverview(fileOverview)) {
515                 addParserWarning("msg.jsdoc.fileoverview.extra");
516               }
517               return token;
518     
519             case LICENSE:
520             case PRESERVE:
521               // Always use PRESERVE for @license and @preserve blocks.
522               ExtractionInfo preserveInfo = extractMultilineTextualBlock(
523                   token, WhitespaceOption.PRESERVE, true);
524     
525               String preserve = preserveInfo.string;
526     
527               if (preserve.length() > 0) {
528                 if (fileLevelJsDocBuilder != null) {
529                   fileLevelJsDocBuilder.addLicense(preserve);
530                 }
531               }
532     
533               token = preserveInfo.token;
534               return token;
535     
536             case ENUM:
537               token = next();
538               lineno = stream.getLineno();
539               charno = stream.getCharno();
540     
541               type = null;
542               if (token != JsDocToken.EOL && token != JsDocToken.EOC) {
543                 Node typeNode = parseAndRecordTypeNode(token);
544                 if (typeNode != null && typeNode.getType() == Token.STRING) {
545                   String typeName = typeNode.getString();
546                   if (!typeName.equals("number") && !typeName.equals("string")
547                       && !typeName.equals("boolean")) {
548                     typeNode = wrapNode(Token.BANG, typeNode);
549                   }
550                 }
551                 type = createJSTypeExpression(typeNode);
552               } else {
553                 restoreLookAhead(token);
554               }
555     
556               if (type == null) {
557                 type = createJSTypeExpression(newStringNode("number"));
558               }
559               if (!jsdocBuilder.recordEnumParameterType(type)) {
560                 addTypeWarning("msg.jsdoc.incompat.type", lineno, charno);
561               }
562               return eatUntilEOLIfNotAnnotation();
563     
564             case EXPOSE:
565               if (!jsdocBuilder.recordExpose()) {
566                 addParserWarning("msg.jsdoc.expose");
567               }
568               return eatUntilEOLIfNotAnnotation();
569     
570             case EXTERNS:
571               if (!jsdocBuilder.recordExterns()) {
572                 addParserWarning("msg.jsdoc.externs");
573               }
574               return eatUntilEOLIfNotAnnotation();
575     
576             case EXTENDS:
577             case IMPLEMENTS:
578               skipEOLs();
579               token = next();
580               lineno = stream.getLineno();
581               charno = stream.getCharno();
582               boolean matchingRc = false;
583     
584               if (token == JsDocToken.LEFT_CURLY) {
585                 token = next();
586                 matchingRc = true;
587               }
588     
589               if (token == JsDocToken.STRING) {
590                 Node typeNode = parseAndRecordTypeNameNode(
591                     token, lineno, charno, matchingRc);
592     
593                 lineno = stream.getLineno();
594                 charno = stream.getCharno();
595     
596                 typeNode = wrapNode(Token.BANG, typeNode);
597                 type = createJSTypeExpression(typeNode);
598     
599                 if (annotation == Annotation.EXTENDS) {
600                   // record the extended type, check later
601                   extendedTypes.add(new ExtendedTypeInfo(
602                       type, stream.getLineno(), stream.getCharno()));
603                 } else {
604                   Preconditions.checkState(
605                       annotation == Annotation.IMPLEMENTS);
606                   if (!jsdocBuilder.recordImplementedInterface(type)) {
607                     addTypeWarning("msg.jsdoc.implements.duplicate", lineno, charno);
608                   }
609                 }
610                 token = next();
611                 if (matchingRc) {
612                   if (token != JsDocToken.RIGHT_CURLY) {
613                     addTypeWarning("msg.jsdoc.missing.rc");
614                   } else {
615                     token = next();
616                   }
617                 } else if (token != JsDocToken.EOL &&
618                     token != JsDocToken.EOF && token != JsDocToken.EOC) {
619                   addTypeWarning("msg.end.annotation.expected");
620                 }
621               } else {
622                 addTypeWarning("msg.no.type.name", lineno, charno);
623               }
624               token = eatUntilEOLIfNotAnnotation(token);
625               return token;
626     
627             case HIDDEN:
628               if (!jsdocBuilder.recordHiddenness()) {
629                 addParserWarning("msg.jsdoc.hidden");
630               }
631               return eatUntilEOLIfNotAnnotation();
632     
633             case LENDS:
634               skipEOLs();
635     
636               matchingRc = false;
637               if (match(JsDocToken.LEFT_CURLY)) {
638                 token = next();
639                 matchingRc = true;
640               }
641     
642               if (match(JsDocToken.STRING)) {
643                 token = next();
644                 if (!jsdocBuilder.recordLends(stream.getString())) {
645                   addTypeWarning("msg.jsdoc.lends.incompatible");
646                 }
647               } else {
648                 addTypeWarning("msg.jsdoc.lends.missing");
649               }
650     
651               if (matchingRc && !match(JsDocToken.RIGHT_CURLY)) {
652                 addTypeWarning("msg.jsdoc.missing.rc");
653               }
654               return eatUntilEOLIfNotAnnotation();
655     
656             case MEANING:
657               ExtractionInfo meaningInfo = extractMultilineTextualBlock(token);
658               String meaning = meaningInfo.string;
659               token = meaningInfo.token;
660               if (!jsdocBuilder.recordMeaning(meaning)) {
661                 addParserWarning("msg.jsdoc.meaning.extra");
662               }
663               return token;
664     
665             case NO_ALIAS:
666               if (!jsdocBuilder.recordNoAlias()) {
667                 addParserWarning("msg.jsdoc.noalias");
668               }
669               return eatUntilEOLIfNotAnnotation();
670     
671             case NO_COMPILE:
672               if (!jsdocBuilder.recordNoCompile()) {
673                 addParserWarning("msg.jsdoc.nocompile");
674               }
675               return eatUntilEOLIfNotAnnotation();
676     
677             case NO_COLLAPSE:
678               if (!jsdocBuilder.recordNoCollapse()) {
679                 addParserWarning("msg.jsdoc.nocollapse");
680               }
681               return eatUntilEOLIfNotAnnotation();
682     
683             case NOT_IMPLEMENTED:
684               return eatUntilEOLIfNotAnnotation();
685     
686             case INHERIT_DOC:
687             case OVERRIDE:
688               if (!jsdocBuilder.recordOverride()) {
689                 addTypeWarning("msg.jsdoc.override");
690               }
691               return eatUntilEOLIfNotAnnotation();
692     
693             case POLYMER_BEHAVIOR:
694               if (jsdocBuilder.isPolymerBehaviorRecorded()) {
695                 addParserWarning("msg.jsdoc.polymerBehavior.extra");
696               } else {
697                 jsdocBuilder.recordPolymerBehavior();
698               }
699               return eatUntilEOLIfNotAnnotation();
700     
701             case THROWS: {
702               skipEOLs();
703               token = next();
704               lineno = stream.getLineno();
705               charno = stream.getCharno();
706               type = null;
707     
708               if (token == JsDocToken.LEFT_CURLY) {
709                 type = createJSTypeExpression(
710                     parseAndRecordTypeNode(token));
711     
712                 if (type == null) {
713                   // parsing error reported during recursive descent
714                   // recovering parsing
715                   return eatUntilEOLIfNotAnnotation();
716                 }
717               }
718     
719               // *Update* the token to that after the type annotation.
720               token = current();
721     
722               // Save the throw type.
723               jsdocBuilder.recordThrowType(type);
724     
725               boolean isAnnotationNext = lookAheadForAnnotation();
726     
727               // Find the throw's description (if applicable).
728               if (jsdocBuilder.shouldParseDocumentation() && !isAnnotationNext) {
729                 ExtractionInfo descriptionInfo =
730                     extractMultilineTextualBlock(token);
731     
732                 String description = descriptionInfo.string;
733     
734                 if (description.length() > 0) {
735                   jsdocBuilder.recordThrowDescription(type, description);
736                 }
737     
738                 token = descriptionInfo.token;
739               } else {
740                 token = eatUntilEOLIfNotAnnotation();
741               }
742               return token;
743             }
744     
745             case PARAM:
746               skipEOLs();
747               token = next();
748               lineno = stream.getLineno();
749               charno = stream.getCharno();
750               type = null;
751     
752               boolean hasParamType = false;
753     
754               if (token == JsDocToken.LEFT_CURLY) {
755                 type = createJSTypeExpression(
756                     parseAndRecordParamTypeNode(token));
757     
758                 if (type == null) {
759                   // parsing error reported during recursive descent
760                   // recovering parsing
761                   return eatUntilEOLIfNotAnnotation();
762                 }
763                 skipEOLs();
764                 token = next();
765                 lineno = stream.getLineno();
766                 charno = stream.getCharno();
767                 hasParamType = true;
768               }
769     
770     
771               String name = null;
772               boolean isBracketedParam = JsDocToken.LEFT_SQUARE == token;
773               if (isBracketedParam) {
774                 token = next();
775               }
776     
777               if (JsDocToken.STRING != token) {
778                 addTypeWarning("msg.missing.variable.name", lineno, charno);
779               } else {
780                 if (!hasParamType) {
781                   addMissingTypeWarning(stream.getLineno(), stream.getCharno());
782                 }
783     
784                 name = stream.getString();
785     
786                 if (isBracketedParam) {
787                   token = next();
788     
789                   // Throw out JsDocToolkit's "default" parameter
790                   // annotation.  It makes no sense under our type
791                   // system.
792                   if (JsDocToken.EQUALS == token) {
793                     token = next();
794                     if (JsDocToken.STRING == token) {
795                       token = next();
796                     }
797                   }
798     
799                   if (JsDocToken.RIGHT_SQUARE != token) {
800                     reportTypeSyntaxWarning("msg.jsdoc.missing.rb");
801                   } else if (type != null) {
802                     // Make the type expression optional, if it isn't
803                     // already.
804                     type = JSTypeExpression.makeOptionalArg(type);
805                   }
806                 }
807     
808                 // We do not handle the JsDocToolkit method
809                 // for handling properties of params, so if the param name has a DOT
810                 // in it, report a warning and throw it out.
811                 // See https://github.com/google/closure-compiler/issues/499
812                 if (!TokenStream.isJSIdentifier(name)) {
813                   addParserWarning("msg.invalid.variable.name", name, lineno, charno);
814                   name = null;
815                 } else if (!jsdocBuilder.recordParameter(name, type)) {
816                   if (jsdocBuilder.hasParameter(name)) {
817                     addTypeWarning("msg.dup.variable.name", name, lineno, charno);
818                   } else {
819                     addTypeWarning("msg.jsdoc.incompat.type", name, lineno, charno);
820                   }
821                 }
822               }
823     
824               if (name == null) {
825                 token = eatUntilEOLIfNotAnnotation(token);
826                 return token;
827               }
828     
829               jsdocBuilder.markName(name, sourceFile, lineno, charno);
830     
831               // Find the parameter's description (if applicable).
832               if (jsdocBuilder.shouldParseDocumentation()
833                   && token != JsDocToken.ANNOTATION) {
834                 ExtractionInfo paramDescriptionInfo =
835                     extractMultilineTextualBlock(token);
836     
837                 String paramDescription = paramDescriptionInfo.string;
838     
839                 if (paramDescription.length() > 0) {
840                   jsdocBuilder.recordParameterDescription(name,
841                       paramDescription);
842                 }
843     
844                 token = paramDescriptionInfo.token;
845               } else if (token != JsDocToken.EOC && token != JsDocToken.EOF) {
846                 token = eatUntilEOLIfNotAnnotation();
847               }
848               return token;
849     
850             case PRESERVE_TRY:
851               if (!jsdocBuilder.recordPreserveTry()) {
852                 addParserWarning("msg.jsdoc.preservertry");
853               }
854               return eatUntilEOLIfNotAnnotation();
855     
856             case NO_SIDE_EFFECTS:
857               if (!jsdocBuilder.recordNoSideEffects()) {
858                 addParserWarning("msg.jsdoc.nosideeffects");
859               }
860               return eatUntilEOLIfNotAnnotation();
861     
862             case MODIFIES:
863               token = parseModifiesTag(next());
864               return token;
865     
866             case IMPLICIT_CAST:
867               if (!jsdocBuilder.recordImplicitCast()) {
868                 addTypeWarning("msg.jsdoc.implicitcast");
869               }
870               return eatUntilEOLIfNotAnnotation();
871     
872             case SEE:
873               if (jsdocBuilder.shouldParseDocumentation()) {
874                 ExtractionInfo referenceInfo = extractSingleLineBlock();
875                 String reference = referenceInfo.string;
876     
877                 if (reference.isEmpty()) {
878                   addParserWarning("msg.jsdoc.seemissing");
879                 } else {
880                   jsdocBuilder.addReference(reference);
881                 }
882     
883                 token = referenceInfo.token;
884               } else {
885                 token = eatUntilEOLIfNotAnnotation();
886               }
887               return token;
888     
889             case STABLEIDGENERATOR:
890               if (!jsdocBuilder.recordStableIdGenerator()) {
891                 addParserWarning("msg.jsdoc.stableidgen");
892               }
893               return eatUntilEOLIfNotAnnotation();
894     
895             case SUPPRESS:
896               token = parseSuppressTag(next());
897               return token;
898     
899             case TEMPLATE: {
900               int templateLineno = stream.getLineno();
901               int templateCharno = stream.getCharno();
902               // Always use TRIM for template TTL expressions.
903               ExtractionInfo templateInfo =
904                   extractMultilineTextualBlock(token, WhitespaceOption.TRIM, false);
905               String templateString = templateInfo.string;
906               // TTL stands for type transformation language
907               // TODO(lpino): This delimiter needs to be further discussed
908               String ttlStartDelimiter = ":=";
909               String ttlEndDelimiter = "=:";
910               String templateNames;
911               String typeTransformationExpr = "";
912               boolean isTypeTransformation = false;
913               boolean validTypeTransformation = true;
914               // Detect if there is a type transformation
915               if (!templateString.contains(ttlStartDelimiter)) {
916                 // If there is no type transformation take the first line
917                 if (templateString.contains("\n")) {
918                   templateNames =
919                       templateString.substring(0, templateString.indexOf('\n'));
920                 } else {
921                   templateNames = templateString;
922                 }
923               } else {
924                 // Split the part with the template type names
925                 int ttlStartIndex = templateString.indexOf(ttlStartDelimiter);
926                 templateNames = templateString.substring(0, ttlStartIndex);
927                 // Check if the type transformation expression ends correctly
928                 if (!templateString.contains(ttlEndDelimiter)) {
929                   validTypeTransformation = false;
930                   addTypeWarning(
931                         "msg.jsdoc.typetransformation.missing.delimiter",
932                         templateLineno,
933                         templateCharno);
934                   } else {
935                   isTypeTransformation = true;
936                   // Split the part of the type transformation
937                   int ttlEndIndex = templateString.indexOf(ttlEndDelimiter);
938                   typeTransformationExpr = templateString.substring(
939                       ttlStartIndex + ttlStartDelimiter.length(),
940                       ttlEndIndex).trim();
941                 }
942               }
943     
944               // Obtain the template type names
945               List<String> names = Splitter.on(',')
946                   .trimResults()
947                   .splitToList(templateNames);
948     
949               if (names.size() == 1 && names.get(0).isEmpty()) {
950                 addTypeWarning("msg.jsdoc.templatemissing", templateLineno, templateCharno);
951               } else {
952                 for (String typeName : names) {
953                   if (!validTemplateTypeName(typeName)) {
954                     addTypeWarning(
955                         "msg.jsdoc.template.invalid.type.name", templateLineno, templateCharno);
956                   } else if (!isTypeTransformation) {
957                     if (!jsdocBuilder.recordTemplateTypeName(typeName)) {
958                       addTypeWarning(
959                           "msg.jsdoc.template.name.declared.twice", templateLineno, templateCharno);
960                     }
961                   }
962                 }
963               }
964     
965               if (isTypeTransformation) {
966                 // A type transformation must be associated to a single type name
967                 if (names.size() > 1) {
968                     addTypeWarning(
969                         "msg.jsdoc.typetransformation.with.multiple.names",
970                         templateLineno, templateCharno);
971                 }
972                 if (typeTransformationExpr.isEmpty()) {
973                   validTypeTransformation = false;
974                   addTypeWarning(
975                       "msg.jsdoc.typetransformation.expression.missing",
976                       templateLineno,
977                       templateCharno);
978                 }
979                 // Build the AST for the type transformation
980                 if (validTypeTransformation) {
981                   TypeTransformationParser ttlParser =
982                       new TypeTransformationParser(typeTransformationExpr,
983                           sourceFile, errorReporter, templateLineno, templateCharno);
984                   // If the parsing was successful store the type transformation
985                   if (ttlParser.parseTypeTransformation()
986                       && !jsdocBuilder.recordTypeTransformation(
987                           names.get(0), ttlParser.getTypeTransformationAst())) {
988                     addTypeWarning(
989                         "msg.jsdoc.template.name.declared.twice", templateLineno, templateCharno);
990                   }
991                 }
992               }
993               token = templateInfo.token;
994               return token;
995             }
996     
997             case IDGENERATOR:
998               token = parseIdGeneratorTag(next());
999               return token;
1000     
1001             case WIZACTION:
1002               if (!jsdocBuilder.recordWizaction()) {
1003                 addParserWarning("msg.jsdoc.wizaction");
1004               }
1005               return eatUntilEOLIfNotAnnotation();
1006     
1007             case DISPOSES:
1008               {
1009               ExtractionInfo templateInfo = extractSingleLineBlock();
1010               List<String> names = Splitter.on(',')
1011                   .trimResults()
1012                   .splitToList(templateInfo.string);
1013     
1014               if (names.isEmpty() || names.get(0).isEmpty()) {
1015                 addTypeWarning("msg.jsdoc.disposeparameter.missing");
1016               } else if (!jsdocBuilder.recordDisposesParameter(names)) {
1017                 addTypeWarning("msg.jsdoc.disposeparameter.error");
1018               }
1019     
1020               token = templateInfo.token;
1021               return token;
1022             }
1023     
1024             case VERSION:
1025               ExtractionInfo versionInfo = extractSingleLineBlock();
1026               String version = versionInfo.string;
1027     
1028               if (version.isEmpty()) {
1029                 addParserWarning("msg.jsdoc.versionmissing");
1030               } else {
1031                 if (!jsdocBuilder.recordVersion(version)) {
1032                   addParserWarning("msg.jsdoc.extraversion");
1033                 }
1034               }
1035     
1036               token = versionInfo.token;
1037               return token;
1038     
1039             case CONSTANT:
1040             case DEFINE:
1041             case EXPORT:
1042             case RETURN:
1043             case PACKAGE:
1044             case PRIVATE:
1045             case PROTECTED:
1046             case PUBLIC:
1047             case THIS:
1048             case TYPE:
1049             case TYPEDEF:
1050               lineno = stream.getLineno();
1051               charno = stream.getCharno();
1052     
1053               Node typeNode = null;
1054               boolean hasType = lookAheadForType();
1055               boolean isAlternateTypeAnnotation =
1056                   annotation == Annotation.PACKAGE
1057                       || annotation == Annotation.PRIVATE
1058                       || annotation == Annotation.PROTECTED
1059                       || annotation == Annotation.PUBLIC
1060                       || annotation == Annotation.CONSTANT
1061                       || annotation == Annotation.EXPORT;
1062               boolean canSkipTypeAnnotation =
1063                   isAlternateTypeAnnotation || annotation == Annotation.RETURN;
1064               type = null;
1065     
1066               if (annotation == Annotation.RETURN && !hasType) {
1067                 addMissingTypeWarning(stream.getLineno(), stream.getCharno());
1068               }
1069     
1070               if (hasType || !canSkipTypeAnnotation) {
1071                 skipEOLs();
1072                 token = next();
1073                 typeNode = parseAndRecordTypeNode(token);
1074     
1075                 if (annotation == Annotation.THIS) {
1076                   typeNode = wrapNode(Token.BANG, typeNode);
1077                 }
1078                 type = createJSTypeExpression(typeNode);
1079               }
1080     
1081               // The error was reported during recursive descent
1082               // recovering parsing
1083               boolean hasError = type == null && !canSkipTypeAnnotation;
1084               if (!hasError) {
1085                 // Record types for @type.
1086                 // If the @package, @private, @protected, or @public annotations
1087                 // have a type attached, pretend that they actually wrote:
1088                 // @type {type}\n@private
1089                 // This will have some weird behavior in some cases
1090                 // (for example, @private can now be used as a type-cast),
1091                 // but should be mostly OK.
1092                 if (((type != null && isAlternateTypeAnnotation) || annotation == Annotation.TYPE)
1093                     && !jsdocBuilder.recordType(type)) {
1094                   addTypeWarning("msg.jsdoc.incompat.type", lineno, charno);
1095                 }
1096     
1097                 boolean isAnnotationNext = lookAheadForAnnotation();
1098     
1099                 switch (annotation) {
1100                   case CONSTANT:
1101                     if (!jsdocBuilder.recordConstancy()) {
1102                       addParserWarning("msg.jsdoc.const");
1103                     }
1104                     break;
1105     
1106                   case DEFINE:
1107                     if (!jsdocBuilder.recordDefineType(type)) {
1108                       addParserWarning("msg.jsdoc.define", lineno, charno);
1109                     }
1110                     if (!isAnnotationNext) {
1111                       return recordDescription(token);
1112                     }
1113                     break;
1114     
1115                   case EXPORT:
1116                     if (!jsdocBuilder.recordExport()) {
1117                       addParserWarning("msg.jsdoc.export", lineno, charno);
1118                     } else if (!jsdocBuilder.recordVisibility(Visibility.PUBLIC)) {
1119                       addParserWarning("msg.jsdoc.extra.visibility", lineno, charno);
1120                     }
1121                     if (!isAnnotationNext) {
1122                       return recordDescription(token);
1123                     }
1124                     break;
1125     
1126                   case PRIVATE:
1127                     if (!jsdocBuilder.recordVisibility(Visibility.PRIVATE)) {
1128                       addParserWarning("msg.jsdoc.extra.visibility", lineno, charno);
1129                     }
1130                     if (!isAnnotationNext) {
1131                       return recordDescription(token);
1132                     }
1133                     break;
1134     
1135                   case PACKAGE:
1136                     if (!jsdocBuilder.recordVisibility(Visibility.PACKAGE)) {
1137                       addParserWarning("msg.jsdoc.extra.visibility", lineno, charno);
1138                     }
1139                     if (!isAnnotationNext) {
1140                       return recordDescription(token);
1141                     }
1142                     break;
1143     
1144                   case PROTECTED:
1145                     if (!jsdocBuilder.recordVisibility(Visibility.PROTECTED)) {
1146                       addParserWarning("msg.jsdoc.extra.visibility", lineno, charno);
1147                     }
1148                     if (!isAnnotationNext) {
1149                       return recordDescription(token);
1150                     }
1151                     break;
1152     
1153                   case PUBLIC:
1154                     if (!jsdocBuilder.recordVisibility(Visibility.PUBLIC)) {
1155                       addParserWarning("msg.jsdoc.extra.visibility", lineno, charno);
1156                     }
1157                     if (!isAnnotationNext) {
1158                       return recordDescription(token);
1159                     }
1160                     break;
1161     
1162                   case RETURN:
1163                     if (type == null) {
1164                       type = createJSTypeExpression(newNode(Token.QMARK));
1165                     }
1166     
1167                     if (!jsdocBuilder.recordReturnType(type)) {
1168                       addTypeWarning("msg.jsdoc.incompat.type", lineno, charno);
1169                       break;
1170                     }
1171     
1172                     // TODO(johnlenz): The extractMultilineTextualBlock method
1173                     // and friends look directly at the stream, regardless of
1174                     // last token read, so we don't want to read the first
1175                     // "STRING" out of the stream.
1176     
1177                     // Find the return's description (if applicable).
1178                     if (jsdocBuilder.shouldParseDocumentation()
1179                         && !isAnnotationNext) {
1180                       ExtractionInfo returnDescriptionInfo =
1181                           extractMultilineTextualBlock(token);
1182     
1183                       String returnDescription =
1184                           returnDescriptionInfo.string;
1185     
1186                       if (returnDescription.length() > 0) {
1187                         jsdocBuilder.recordReturnDescription(
1188                             returnDescription);
1189                       }
1190     
1191                       token = returnDescriptionInfo.token;
1192                     } else {
1193                       token = eatUntilEOLIfNotAnnotation();
1194                     }
1195                     return token;
1196     
1197                   case THIS:
1198                     if (!jsdocBuilder.recordThisType(type)) {
1199                       addTypeWarning("msg.jsdoc.incompat.type", lineno, charno);
1200                     }
1201                     break;
1202     
1203                   case TYPEDEF:
1204                     if (!jsdocBuilder.recordTypedef(type)) {
1205                       addTypeWarning("msg.jsdoc.incompat.type", lineno, charno);
1206                     }
1207                     break;
1208                 }
1209               }
1210     
1211               return eatUntilEOLIfNotAnnotation();
1212           }
1213         }
1214     
1215         return next();
1216       }

1257       /**
1258        * Parse a {@code @suppress} tag of the form
1259        * {@code @suppress&#123;warning1|warning2&#125;}.
1260        *
1261        * @param token The current token.
1262        */
1263       private JsDocToken parseSuppressTag(JsDocToken token) {
1264         if (token != JsDocToken.LEFT_CURLY) {
1265           addParserWarning("msg.jsdoc.suppress");
1266           return token;
1267         } else {
1268           Set<String> suppressions = new HashSet<>();
1269           while (true) {
1270             if (match(JsDocToken.STRING)) {
1271               String name = stream.getString();
1272               if (!suppressionNames.contains(name)) {
1273                 addParserWarning("msg.jsdoc.suppress.unknown", name);
1274               }
1275     
1276               suppressions.add(stream.getString());
1277               token = next();
1278             } else {
1279               addParserWarning("msg.jsdoc.suppress");
1280               return token;
1281             }
1282     
1283             if (match(JsDocToken.PIPE, JsDocToken.COMMA)) {
1284               token = next();
1285             } else {
1286               break;
1287             }
1288           }
1289     
1290           if (!match(JsDocToken.RIGHT_CURLY)) {
1291             addParserWarning("msg.jsdoc.suppress");
1292           } else {
1293             token = next();
1294             if (!jsdocBuilder.recordSuppressions(suppressions)) {
1295               addParserWarning("msg.jsdoc.suppress.duplicate");
1296             }
1297           }
1298           return eatUntilEOLIfNotAnnotation();
1299         }
1300       }

1303        * Parse a {@code @modifies} tag of the form
1304        * {@code @modifies&#123;this|arguments|param&#125;}.
1305        *
1306        * @param token The current token.
1307        */
1308       private JsDocToken parseModifiesTag(JsDocToken token) {
1309         if (token == JsDocToken.LEFT_CURLY) {
1310           Set<String> modifies = new HashSet<>();
1311           while (true) {
1312             if (match(JsDocToken.STRING)) {
1313               String name = stream.getString();
1314               if (!modifiesAnnotationKeywords.contains(name)
1315                   && !jsdocBuilder.hasParameter(name)) {
1316                 addParserWarning("msg.jsdoc.modifies.unknown", name);
1317               }
1318     
1319               modifies.add(stream.getString());
1320               token = next();
1321             } else {
1322               addParserWarning("msg.jsdoc.modifies");
1323               return token;
1324             }
1325     
1326             if (match(JsDocToken.PIPE)) {
1327               token = next();
1328             } else {
1329               break;
1330             }
1331           }
1332     
1333           if (!match(JsDocToken.RIGHT_CURLY)) {
1334             addParserWarning("msg.jsdoc.modifies");
1335           } else {
1336             token = next();
1337             if (!jsdocBuilder.recordModifies(modifies)) {
1338               addParserWarning("msg.jsdoc.modifies.duplicate");
1339             }
1340           }
1341         }
1342         return token;
1343       }

1345       /**
1346        * Parse a {@code @idgenerator} tag of the form
1347        * {@code @idgenerator} or
1348        * {@code @idgenerator&#123;consistent&#125;}.
1349        *
1350        * @param token The current token.
1351        */
1352       private JsDocToken parseIdGeneratorTag(JsDocToken token) {
1353         String idgenKind = "unique";
1354         if (token == JsDocToken.LEFT_CURLY) {
1355           if (match(JsDocToken.STRING)) {
1356             String name = stream.getString();
1357             if (!idGeneratorAnnotationKeywords.contains(name)
1358                 && !jsdocBuilder.hasParameter(name)) {
1359               addParserWarning("msg.jsdoc.idgen.unknown", name);
1360             }
1361     
1362             idgenKind = name;
1363             token = next();
1364           } else {
1365             addParserWarning("msg.jsdoc.idgen.bad");
1366             return token;
1367           }
1368     
1369           if (!match(JsDocToken.RIGHT_CURLY)) {
1370             addParserWarning("msg.jsdoc.idgen.bad");
1371           } else {
1372             token = next();
1373           }
1374         }
1375     
1376         switch (idgenKind) {
1377           case "unique":
1378             if (!jsdocBuilder.recordIdGenerator()) {
1379               addParserWarning("msg.jsdoc.idgen.duplicate");
1380             }
1381             break;
1382           case "consistent":
1383             if (!jsdocBuilder.recordConsistentIdGenerator()) {
1384               addParserWarning("msg.jsdoc.idgen.duplicate");
1385             }
1386             break;
1387           case "stable":
1388             if (!jsdocBuilder.recordStableIdGenerator()) {
1389               addParserWarning("msg.jsdoc.idgen.duplicate");
1390             }
1391             break;
1392           case "mapped":
1393             if (!jsdocBuilder.recordMappedIdGenerator()) {
1394               addParserWarning("msg.jsdoc.idgen.duplicate");
1395             }
1396             break;
1397         }
1398     
1399         return token;
1400       }

1416        * returns it. Note that this method consumes input.
1417        *

1674       /**
1675        * Extracts text from the stream until the end of the comment, end of the
1676        * file, or an annotation token is encountered. If the text is being
1677        * extracted for a JSDoc marker, the first line in the stream will always be
1678        * included in the extract text.
1679        *
1680        * @param token The starting token.
1681        * @param option How to handle whitespace.
1682        * @param isMarker Whether the extracted text is for a JSDoc marker or a
1683        *     block comment.
1684        * @param includeAnnotations Whether the extracted text may include
1685        *     annotations. If set to false, text extraction will stop on the first
1686        *     encountered annotation token.
1687        *
1688        * @return The extraction information.
1689        */
1690       private ExtractionInfo extractMultilineComment(
1691           JsDocToken token, WhitespaceOption option, boolean isMarker, boolean includeAnnotations) {
1692     
1693         StringBuilder builder = new StringBuilder();
1694         int startLineno = -1;
1695         int startCharno = -1;
1696     
1697         if (isMarker) {
1698           stream.update();
1699           startLineno = stream.getLineno();
1700           startCharno = stream.getCharno() + 1;
1701     
1702           String line = getRemainingJSDocLine();
1703           if (option != WhitespaceOption.PRESERVE) {
1704             line = line.trim();
1705           }
1706           builder.append(line);
1707     
1708           state = State.SEARCHING_ANNOTATION;
1709           token = next();
1710         }
1711     
1712         boolean ignoreStar = false;
1713     
1714         // Track the start of the line to count whitespace that
1715         // the tokenizer skipped. Because this case is rare, it's easier
1716         // to do this here than in the tokenizer.
1717         int lineStartChar = -1;
1718     
1719         do {
1720           switch (token) {
1721             case STAR:
1722               if (ignoreStar) {
1723                 // Mark the position after the star as the new start of the line.
1724                 lineStartChar = stream.getCharno() + 1;
1725                 ignoreStar = false;
1726               } else {
1727                 // The star is part of the comment.
1728                 padLine(builder, lineStartChar, option);
1729                 lineStartChar = -1;
1730                 builder.append('*');
1731               }
1732     
1733               token = next();
1734               while (token == JsDocToken.STAR) {
1735                 if (lineStartChar != -1) {
1736                   padLine(builder, lineStartChar, option);
1737                   lineStartChar = -1;
1738                 }
1739                 builder.append('*');
1740                 token = next();
1741               }
1742               continue;
1743     
1744             case EOL:
1745               if (option != WhitespaceOption.SINGLE_LINE) {
1746                 builder.append('\n');
1747               }
1748     
1749               ignoreStar = true;
1750               lineStartChar = 0;
1751               token = next();
1752               continue;
1753     
1754             default:
1755               ignoreStar = false;
1756               state = State.SEARCHING_ANNOTATION;
1757     
1758               boolean isEOC = token == JsDocToken.EOC;
1759               if (!isEOC) {
1760                 padLine(builder, lineStartChar, option);
1761                 lineStartChar = -1;
1762               }
1763     
1764               if (token == JsDocToken.EOC ||
1765                   token == JsDocToken.EOF ||
1766                   // When we're capturing a license block, annotations
1767                   // in the block are OK.
1768                   (token == JsDocToken.ANNOTATION && !includeAnnotations)) {
1769                 String multilineText = builder.toString();
1770     
1771                 if (option != WhitespaceOption.PRESERVE) {
1772                   multilineText = multilineText.trim();
1773                 }
1774     
1775                 if (isMarker && !multilineText.isEmpty()) {
1776                   int endLineno = stream.getLineno();
1777                   int endCharno = stream.getCharno();
1778                   jsdocBuilder.markText(multilineText, startLineno, startCharno,
1779                       endLineno, endCharno);
1780                 }
1781     
1782                 return new ExtractionInfo(multilineText, token);
1783               }
1784     
1785               builder.append(toString(token));
1786     
1787               String line = getRemainingJSDocLine();
1788     
1789               if (option != WhitespaceOption.PRESERVE) {
1790                 line = trimEnd(line);
1791               }
1792     
1793               builder.append(line);
1794               token = next();
1795           }
1796         } while (true);
1797       }

1799       private void padLine(StringBuilder builder, int lineStartChar, WhitespaceOption option) {
1800         if (lineStartChar != -1 && option == WhitespaceOption.PRESERVE) {
1801           int numSpaces = stream.getCharno() - lineStartChar;
1802           for (int i = 0; i < numSpaces; i++) {
1803             builder.append(' ');
1804           }
1805         } else if (builder.length() > 0) {
1806           if (builder.charAt(builder.length() - 1) != '\n' || option == WhitespaceOption.PRESERVE) {
1807             builder.append(' ');
1808           }
1809         }
1810       }

1812       /**
1813        * Trim characters from only the end of a string.
1814        * This method will remove all whitespace characters
1815        * (defined by TokenUtil.isWhitespace(char), in addition to the characters
1816        * provided, from the end of the provided string.
1817        *
1818        * @param s String to be trimmed
1819        * @return String with whitespace and characters in extraChars removed
1820        *                   from the end.
1821        */
1822       private static String trimEnd(String s) {
1823         int trimCount = 0;
1824         while (trimCount < s.length()) {
1825           char ch = s.charAt(s.length() - trimCount - 1);
1826           if (TokenUtil.isWhitespace(ch)) {
1827             trimCount++;
1828           } else {
1829             break;
1830           }
1831         }
1832     
1833         if (trimCount == 0) {
1834           return s;
1835         }
1836         return s.substring(0, s.length() - trimCount);
1837       }

1849        * TypeExpressionAnnotation := TypeExpression |
1850        *     '{' TopLevelTypeExpression '}'

1852       private Node parseTypeExpressionAnnotation(JsDocToken token) {

1856           if (typeNode != null) {

1860             } else {

1866         } else {

1868           reportTypeSyntaxWarning("msg.jsdoc.missing.braces");
1869           return parseTypeExpression(token);
1870         }

1873       /**
1874        * ParamTypeExpression :=
1875        *     OptionalParameterType |
1876        *     TopLevelTypeExpression |
1877        *     '...' TopLevelTypeExpression
1878        *
1879        * OptionalParameterType :=
1880        *     TopLevelTypeExpression '='
1881        *
1882        */
1883       private Node parseParamTypeExpression(JsDocToken token) {
1884         boolean restArg = false;
1885         if (token == JsDocToken.ELLIPSIS) {
1886           token = next();
1887           if (token == JsDocToken.RIGHT_CURLY) {
1888             restoreLookAhead(token);
1889             // EMPTY represents the UNKNOWN type in the Type AST.
1890             return wrapNode(Token.ELLIPSIS, IR.empty());
1891           }
1892           restArg = true;
1893         }
1894     
1895         Node typeNode = parseTopLevelTypeExpression(token);
1896         if (typeNode != null) {
1897           skipEOLs();
1898           if (restArg) {
1899             typeNode = wrapNode(Token.ELLIPSIS, typeNode);
1900           } else if (match(JsDocToken.EQUALS)) {
1901             next();
1902             skipEOLs();
1903             typeNode = wrapNode(Token.EQUALS, typeNode);
1904           }
1905         }
1906     
1907         return typeNode;
1908       }

1973       /**
1974        * TypeExpressionList := TopLevelTypeExpression
1975        *     | TopLevelTypeExpression ',' TypeExpressionList
1976        */
1977       private Node parseTypeExpressionList(JsDocToken token) {
1978         Node typeExpr = parseTopLevelTypeExpression(token);
1979         if (typeExpr == null) {
1980           return null;
1981         }
1982         Node typeList = IR.block();
1983         typeList.addChildToBack(typeExpr);
1984         while (match(JsDocToken.COMMA)) {
1985           next();
1986           skipEOLs();
1987           typeExpr = parseTopLevelTypeExpression(next());
1988           if (typeExpr == null) {
1989             return null;
1990           }
1991           typeList.addChildToBack(typeExpr);
1992         }
1993         return typeList;
1994       }

1996       /**
1997        * TypeExpression := BasicTypeExpression
1998        *     | '?' BasicTypeExpression
1999        *     | '!' BasicTypeExpression
2000        *     | BasicTypeExpression '?'
2001        *     | BasicTypeExpression '!'
2002        *     | '?'
2003        */
2004       private Node parseTypeExpression(JsDocToken token) {
2005         if (token == JsDocToken.QMARK) {
2006           // A QMARK could mean that a type is nullable, or that it's unknown.
2007           // We use look-ahead 1 to determine whether it's unknown. Otherwise,
2008           // we assume it means nullable. There are 8 cases:
2009           // {?} - right curly
2010           // ? - EOF (possible when the parseTypeString method is given a bare type expression)
2011           // {?=} - equals
2012           // {function(?, number)} - comma
2013           // {function(number, ?)} - right paren
2014           // {function(): ?|number} - pipe
2015           // {Array.<?>} - greater than
2016           // /** ? */ - EOC (inline types)
2017           // I'm not a big fan of using look-ahead for this, but it makes
2018           // the type language a lot nicer.
2019           token = next();
2020           if (token == JsDocToken.COMMA
2021               || token == JsDocToken.EQUALS
2022               || token == JsDocToken.RIGHT_SQUARE
2023               || token == JsDocToken.RIGHT_CURLY
2024               || token == JsDocToken.RIGHT_PAREN
2025               || token == JsDocToken.PIPE
2026               || token == JsDocToken.RIGHT_ANGLE
2027               || token == JsDocToken.EOC
2028               || token == JsDocToken.EOF) {
2029             restoreLookAhead(token);
2030             return newNode(Token.QMARK);
2031           }
2032     
2033           return wrapNode(Token.QMARK, parseBasicTypeExpression(token));
2034         } else if (token == JsDocToken.BANG) {
2035           return wrapNode(Token.BANG, parseBasicTypeExpression(next()));
2036         } else {
2037           Node basicTypeExpr = parseBasicTypeExpression(token);
2038           if (basicTypeExpr != null) {
2039             if (match(JsDocToken.QMARK)) {
2040               next();
2041               return wrapNode(Token.QMARK, basicTypeExpr);
2042             } else if (match(JsDocToken.BANG)) {
2043               next();
2044               return wrapNode(Token.BANG, basicTypeExpr);
2045             }
2046           }
2047     
2048           return basicTypeExpr;
2049         }
2050       }

2064       /**
2065        * BasicTypeExpression := '*' | 'null' | 'undefined' | TypeName
2066        *     | FunctionType | UnionType | RecordType
2067        */
2068       private Node parseBasicTypeExpression(JsDocToken token) {
2069         if (token == JsDocToken.STAR) {
2070           return newNode(Token.STAR);
2071         } else if (token == JsDocToken.LEFT_CURLY) {
2072           skipEOLs();
2073           return parseRecordType(next());
2074         } else if (token == JsDocToken.LEFT_PAREN) {
2075           skipEOLs();
2076           return parseUnionType(next());
2077         } else if (token == JsDocToken.STRING) {
2078           String string = stream.getString();
2079           switch (string) {
2080             case "function":
2081               skipEOLs();
2082               return parseFunctionType(next());
2083             case "null":
2084             case "undefined":
2085               return newStringNode(string);
2086             default:
2087               return parseTypeName(token);
2088           }
2089         }
2090     
2091         restoreLookAhead(token);
2092         return reportGenericTypeSyntaxWarning();
2093       }

2095       /**
2096        * TypeName := NameExpression | NameExpression TypeApplication
2097        * TypeApplication := '.<' TypeExpressionList '>'
2098        */
2099       private Node parseTypeName(JsDocToken token) {
2100         if (token != JsDocToken.STRING) {
2101           return reportGenericTypeSyntaxWarning();
2102         }
2103     
2104         String typeName = stream.getString();
2105         int lineno = stream.getLineno();
2106         int charno = stream.getCharno();
2107         while (match(JsDocToken.EOL) &&
2108             typeName.charAt(typeName.length() - 1) == '.') {
2109           skipEOLs();
2110           if (match(JsDocToken.STRING)) {
2111             next();
2112             typeName += stream.getString();
2113           }
2114         }
2115     
2116         Node typeNameNode = newStringNode(typeName, lineno, charno);
2117     
2118         if (match(JsDocToken.LEFT_ANGLE)) {
2119           next();
2120           skipEOLs();
2121           Node memberType = parseTypeExpressionList(next());
2122           if (memberType != null) {

2136       /**
2137        * FunctionType := 'function' FunctionSignatureType
2138        * FunctionSignatureType :=
2139        *    TypeParameters '(' 'this' ':' TypeName, ParametersType ')' ResultType
2140        *
2141        * <p>The Node that is produced has type Token.FUNCTION but does not look like a typical
2142        * function node. If there is a 'this:' or 'new:' type, that type is added as a child.
2143        * Then, if there are parameters, a PARAM_LIST node is added as a child. Finally, if
2144        * there is a return type, it is added as a child. This means that the parameters
2145        * could be the first or second child, and the return type could be
2146        * the first, second, or third child.
2147        */
2148       private Node parseFunctionType(JsDocToken token) {
2149         // NOTE(nicksantos): We're not implementing generics at the moment, so
2150         // just throw out TypeParameters.
2151         if (token != JsDocToken.LEFT_PAREN) {
2152           restoreLookAhead(token);
2153           return reportTypeSyntaxWarning("msg.jsdoc.missing.lp");
2154         }
2155     
2156         Node functionType = newNode(Token.FUNCTION);
2157         Node parameters = null;
2158         skipEOLs();
2159         if (!match(JsDocToken.RIGHT_PAREN)) {
2160           token = next();
2161     
2162           boolean hasParams = true;
2163           if (token == JsDocToken.STRING) {
2164             String tokenStr = stream.getString();
2165             boolean isThis = "this".equals(tokenStr);
2166             boolean isNew = "new".equals(tokenStr);
2167             if (isThis || isNew) {
2168               if (match(JsDocToken.COLON)) {
2169                 next();
2170                 skipEOLs();
2171                 Node contextType = wrapNode(
2172                     isThis ? Token.THIS : Token.NEW,
2173                     parseContextTypeExpression(next()));
2174                 if (contextType == null) {
2175                   return null;
2176                 }
2177     
2178                 functionType.addChildToFront(contextType);
2179               } else {
2180                 return reportTypeSyntaxWarning("msg.jsdoc.missing.colon");
2181               }
2182     
2183               if (match(JsDocToken.COMMA)) {
2184                 next();
2185                 skipEOLs();
2186                 token = next();
2187               } else {
2188                 hasParams = false;
2189               }
2190             }
2191           }
2192     
2193           if (hasParams) {
2194             parameters = parseParametersType(token);
2195             if (parameters == null) {
2196               return null;
2197             }
2198           }
2199         }
2200     
2201         if (parameters != null) {
2202           functionType.addChildToBack(parameters);
2203         }
2204     
2205         skipEOLs();
2206         if (!match(JsDocToken.RIGHT_PAREN)) {
2207           return reportTypeSyntaxWarning("msg.jsdoc.missing.rp");
2208         }
2209     
2210         skipEOLs();
2211         next();

2213         if (resultType == null) {
2214           return null;
2215         } else {
2216           functionType.addChildToBack(resultType);
2217         }
2218         return functionType;
2219       }

2221       /**
2222        * ParametersType := RestParameterType | NonRestParametersType
2223        *     | NonRestParametersType ',' RestParameterType
2224        * RestParameterType := '...' Identifier
2225        * NonRestParametersType := ParameterType ',' NonRestParametersType
2226        *     | ParameterType
2227        *     | OptionalParametersType
2228        * OptionalParametersType := OptionalParameterType
2229        *     | OptionalParameterType, OptionalParametersType
2230        * OptionalParameterType := ParameterType=
2231        * ParameterType := TypeExpression | Identifier ':' TypeExpression
2232        */
2233       // NOTE(nicksantos): The official ES4 grammar forces optional and rest
2234       // arguments to come after the required arguments. Our parser does not
2235       // enforce this. Instead we allow them anywhere in the function at parse-time,
2236       // and then warn about them during type resolution.
2237       //
2238       // In theory, it might be mathematically nicer to do the order-checking here.
2239       // But in practice, the order-checking for structural functions is exactly
2240       // the same as the order-checking for @param annotations. And the latter
2241       // has to happen during type resolution. Rather than duplicate the
2242       // order-checking in two places, we just do all of it in type resolution.
2243       private Node parseParametersType(JsDocToken token) {
2244         Node paramsType = newNode(Token.PARAM_LIST);
2245         boolean isVarArgs = false;
2246         Node paramType = null;
2247         if (token != JsDocToken.RIGHT_PAREN) {
2248           do {
2249             if (paramType != null) {
2250               // skip past the comma
2251               next();
2252               skipEOLs();
2253               token = next();
2254             }
2255     
2256             if (token == JsDocToken.ELLIPSIS) {
2257               // In the latest ES4 proposal, there are no type constraints allowed
2258               // on variable arguments. We support the old syntax for backwards
2259               // compatibility, but we should gradually tear it out.
2260               skipEOLs();
2261               if (match(JsDocToken.RIGHT_PAREN)) {
2262                 paramType = newNode(Token.ELLIPSIS);
2263               } else {
2264                 skipEOLs();
2265                 paramType = wrapNode(Token.ELLIPSIS, parseTypeExpression(next()));
2266                 skipEOLs();
2267               }
2268     
2269               isVarArgs = true;
2270             } else {
2271               paramType = parseTypeExpression(token);
2272               if (match(JsDocToken.EQUALS)) {
2273                 skipEOLs();
2274                 next();
2275                 paramType = wrapNode(Token.EQUALS, paramType);
2276               }
2277             }
2278     
2279             if (paramType == null) {
2280               return null;
2281             }
2282             paramsType.addChildToBack(paramType);
2283             if (isVarArgs) {
2284               break;
2285             }
2286           } while (match(JsDocToken.COMMA));
2287         }
2288     
2289         if (isVarArgs && match(JsDocToken.COMMA)) {
2290           return reportTypeSyntaxWarning("msg.jsdoc.function.varargs");
2291         }
2292     
2293         // The right paren will be checked by parseFunctionType
2294     
2295         return paramsType;
2296       }

2394        */

2396         Node fieldTypeList = newNode(Token.LB);
2397     
2398         Set<String> names = new HashSet<>();

2401           Node fieldType = parseFieldType(token);
2402

2408               : fieldType.getFirstChild().getString();
