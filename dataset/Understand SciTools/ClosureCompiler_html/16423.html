17     package com.google.javascript.jscomp.gwt.linker;
18     
19     import com.google.common.base.Preconditions;
20     import com.google.gwt.core.ext.LinkerContext;
21     import com.google.gwt.core.ext.TreeLogger;
22     import com.google.gwt.core.ext.UnableToCompleteException;
23     import com.google.gwt.core.ext.linker.AbstractLinker;
24     import com.google.gwt.core.ext.linker.ArtifactSet;
25     import com.google.gwt.core.ext.linker.CompilationResult;
26     import com.google.gwt.core.ext.linker.LinkerOrder;
27     import com.google.gwt.core.ext.linker.LinkerOrder.Order;
28     import com.google.gwt.core.linker.SymbolMapsLinker;
29     import com.google.gwt.dev.util.Util;
30     
31     /**
32      * Simple single-script linker that doesn't add any dependence on the browser.
33      * This is intended to generate server-side runnable JS.  It doesn't support
34      * permutations, nor does it allow late-loading code.
35      */
36     @LinkerOrder(Order.PRIMARY)
37     public class MinimalLinker extends AbstractLinker {
38     
39       private static final String PREFIX = "(function(){var $wnd=this;var $doc={};";
40       private static final String SUFFIX = "})();";
41     
42       @Override
43       public String getDescription() {
44         return "Minimal";
45       }
46     
47       @Override
48       public ArtifactSet link(TreeLogger logger, LinkerContext context, ArtifactSet artifacts)
49           throws UnableToCompleteException {
50         ArtifactSet toReturn = link(logger, context, artifacts, true);
51         toReturn = link(logger, context, toReturn, false);
52         return toReturn;
53       }
54     
55       @Override
56       public ArtifactSet link(
57           TreeLogger logger, LinkerContext context, ArtifactSet artifacts, boolean onePermutation)
58           throws UnableToCompleteException {
59         ArtifactSet toReturn = new ArtifactSet(artifacts);
60         ArtifactSet writableArtifacts = new ArtifactSet(artifacts);
61     
62         for (CompilationResult result : toReturn.find(CompilationResult.class)) {
63           String[] js = result.getJavaScript();
64           Preconditions.checkArgument(js.length == 1, "MinimalLinker doesn't support GWT.runAsync");
65     
66           byte[] primary = Util.getBytes(PREFIX + js[0] + SUFFIX);
67           toReturn.add(emitBytes(logger, primary, context.getModuleName() + ".js"));
68         }
69     
70         for (SymbolMapsLinker.ScriptFragmentEditsArtifact ea :
71             writableArtifacts.find(SymbolMapsLinker.ScriptFragmentEditsArtifact.class)) {
72           toReturn.add(ea);
73         }
74         return toReturn;
75       }
76     
77       @Override
78       public boolean supportsDevModeInJunit(LinkerContext context) {
79         return false;
80       }
81     }
