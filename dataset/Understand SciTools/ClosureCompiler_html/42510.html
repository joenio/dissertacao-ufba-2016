33     /**
34      * Tests for {@link PhaseOptimizer}.
35      * @author nicksantos@google.com (Nick Santos)
36      */
37     public final class PhaseOptimizerTest extends TestCase {
38       private final List<String> passesRun = new ArrayList<>();
39       private Node dummyRoot;
40       private PhaseOptimizer optimizer;
41       private Compiler compiler;
42       private PerformanceTracker tracker;
43     
44       @Override
45       public void setUp() {
46         passesRun.clear();
47         dummyRoot = new Node(Token.BLOCK);
48         // Needed if we are validating the AST using AstValidator.
49         dummyRoot.setIsSyntheticBlock(true);
50         compiler = new Compiler();
51         compiler.initCompilerOptionsIfTesting();
52         tracker = new PerformanceTracker(dummyRoot, TracerMode.TIMING_ONLY);
53         optimizer = new PhaseOptimizer(compiler, tracker, null);
54       }
55     
56       public void testOneRun() {
57         addOneTimePass("x");
58         assertPasses("x");
59       }
60     
61       public void testLoop1() {
62         Loop loop = optimizer.addFixedPointLoop();
63         addLoopedPass(loop, "x", 0);
64         assertPasses("x");
65       }
66     
67       public void testLoop2() {
68         Loop loop = optimizer.addFixedPointLoop();
69         addLoopedPass(loop, "x", 3);
70         assertPasses("x", "x", "x", "x");
71       }
72     
73       public void testSchedulingOfLoopablePasses() {
74         Loop loop = optimizer.addFixedPointLoop();
75         addLoopedPass(loop, "x", 3);
76         addLoopedPass(loop, "y", 1);
77         // The pass iterations can be grouped as: [x y] [x y] [x] [x] [y]
78         assertPasses("x", "y", "x", "y", "x", "x", "y");
79       }
80     
81       public void testNotInfiniteLoop() {
82         Loop loop = optimizer.addFixedPointLoop();
83         addLoopedPass(loop, "x", PhaseOptimizer.MAX_LOOPS - 1);
84         optimizer.process(null, dummyRoot);
85         assertEquals("There should be no errors.", 0, compiler.getErrorCount());
86       }
87     
88       public void testInfiniteLoop() {
89         Loop loop = optimizer.addFixedPointLoop();
90         addLoopedPass(loop, "x", PhaseOptimizer.MAX_LOOPS + 1);
91         try {
92           optimizer.process(null, dummyRoot);
93           fail("Expected RuntimeException");
94         } catch (RuntimeException e) {
95           assertTrue(e.getMessage(),
96               e.getMessage().contains(PhaseOptimizer.OPTIMIZE_LOOP_ERROR));
97         }
98       }

100       public void testSchedulingOfAnyKindOfPasses1() {
101         addOneTimePass("a");
102         Loop loop = optimizer.addFixedPointLoop();
103         addLoopedPass(loop, "x", 3);
104         addLoopedPass(loop, "y", 1);
105         addOneTimePass("z");
106         assertPasses("a", "x", "y", "x", "y", "x", "x", "y", "z");
107       }

109       public void testSchedulingOfAnyKindOfPasses2() {
110         optimizer.consume(
111             ImmutableList.of(
112                 createPassFactory("a", 0, true),
113                 createPassFactory("b", 1, false),
114                 createPassFactory("c", 2, false),
115                 createPassFactory("d", 1, false),
116                 createPassFactory("e", 1, true),
117                 createPassFactory("f", 0, true)));
118         // The pass iterations can be grouped as:
119         // [a] [b c d] [b c d] [c] [b d] [e] [f]
120         assertPasses("a", "b", "c", "d", "b", "c", "d", "c", "b", "d", "e", "f");
121       }

141       public void testDuplicateLoop() {
142         Loop loop = optimizer.addFixedPointLoop();
143         addLoopedPass(loop, "x", 1);
144         try {
145           addLoopedPass(loop, "x", 1);
146           fail("Expected exception");
147         } catch (IllegalArgumentException e) {
148           return;
149         }
150       }
151     
152       public void testPassOrdering() {
153         Loop loop = optimizer.addFixedPointLoop();
154         List<String> optimalOrder = new ArrayList<>(
155             PhaseOptimizer.OPTIMAL_ORDER);
156         Random random = new Random();
157         while (!optimalOrder.isEmpty()) {
158           addLoopedPass(
159               loop, optimalOrder.remove(random.nextInt(optimalOrder.size())), 0);
160         }
161         optimizer.process(null, dummyRoot);
162         assertEquals(PhaseOptimizer.OPTIMAL_ORDER, passesRun);
163       }
164     
165       public void testProgress() {
166         final List<Double> progressList = new ArrayList<>();
167         compiler = new Compiler() {
168           @Override void setProgress(double p, String name) {
169             progressList.add(p);
170           }
171         };
172         compiler.initCompilerOptionsIfTesting();
173         optimizer = new PhaseOptimizer(compiler, null,
174             new PhaseOptimizer.ProgressRange(0, 100));
175         addOneTimePass("x1");
176         addOneTimePass("x2");
177         addOneTimePass("x3");
178         addOneTimePass("x4");
179         optimizer.process(null, dummyRoot);
180         assertThat(progressList).hasSize(4);
181         assertEquals(25, Math.round(progressList.get(0)));
182         assertEquals(50, Math.round(progressList.get(1)));
183         assertEquals(75, Math.round(progressList.get(2)));
184         assertEquals(100, Math.round(progressList.get(3)));
185       }
186     
187       public void assertPasses(String ... names) {
188         optimizer.process(null, dummyRoot);
189         assertEquals(ImmutableList.copyOf(names), passesRun);
190       }
191     
192       private void addOneTimePass(String name) {
193         optimizer.addOneTimePass(
194             createPassFactory(name, 0, true));
195       }
196     
197       private void addLoopedPass(Loop loop, String name, int numChanges) {
198         loop.addLoopedPass(
199             createPassFactory(name, numChanges, false));
200       }
201     
202       private PassFactory createPassFactory(
203           String name, int numChanges, boolean isOneTime) {
204         return createPassFactory(name, createPass(name, numChanges), isOneTime);
205       }
206     
207       private PassFactory createPassFactory(
208           String name, final CompilerPass pass, boolean isOneTime) {
209         return new PassFactory(name, isOneTime) {
210           @Override
211           protected CompilerPass create(AbstractCompiler compiler) {
212             return pass;
213           }
214         };
215       }
216     
217       private CompilerPass createPass(final String name, int numChanges) {
218         final int[] numChangesClosure = new int[] {numChanges};
219         return new CompilerPass() {
220           @Override public void process(Node externs, Node root) {
221             passesRun.add(name);
222             if (numChangesClosure[0] > 0) {
223               compiler.reportCodeChange();
224               numChangesClosure[0] = numChangesClosure[0] - 1;
225             }
226           }
227         };
228       }
