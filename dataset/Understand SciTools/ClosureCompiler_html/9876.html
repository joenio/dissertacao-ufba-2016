37     @RunWith(JUnit4.class)

39       protected static final Joiner LINE_JOINER = Joiner.on('\n');

41       private String srcCode = "";

43       @Test
44       public void testSimple() {
45         setSourceCode("var a;");
46         SourceAst realAst = createRealAst();
47     
48         // Note that here we do not test the caching of the tree between calls,
49         // since that is the responsibility of the JsAst class.  Here we just test
50         // that proxying is successful and a copy is happening
51     
52         // Initial compile.
53         RecoverableJsAst ast1 = new RecoverableJsAst(realAst, true);
54         checkCompile(realAst, makeDefensiveCopy(ast1), "var a;\n");
55     
56         // Change in the file-system.
57         setSourceCode("var b;");
58         realAst.clearAst();
59     
60         // The first RecoverableJsAst should continue to have the same value.
61         checkCompile(realAst, makeDefensiveCopy(ast1), "var a;\n");
62     
63         // A newly created one from the source should have a different value.
64         RecoverableJsAst ast2 = new RecoverableJsAst(realAst, true);
65         checkCompile(realAst, makeDefensiveCopy(ast2), "var b;\n");
66     
67         // Clearing the first AST should also pick up the new changes.
68         ast1.clearAst();
69         checkCompile(realAst, makeDefensiveCopy(ast1), "var b;\n");
70       }

72       @Test
73       public void testWarningReplay() {
74         setSourceCode("var f() = a;");
75         SourceAst realAst = createRealAst();
76     
77         // Note that here we do not test the caching of the tree between calls,
78         // since that is the responsibility of the JsAst class.  Here we just test
79         // that proxying is successful and a copy is happening
80     
81         // Initial compile.
82         RecoverableJsAst ast1 = new RecoverableJsAst(realAst, true);
83         checkParseErrors(realAst, makeDefensiveCopy(ast1), "Parse error. Semi-colon expected");
84     
85         // The first RecoverableJsAst should continue to have the same value.
86         checkParseErrors(realAst, makeDefensiveCopy(ast1), "Parse error. Semi-colon expected");
87       }

90       private RecoverableJsAst makeDefensiveCopy(SourceAst ast) {
91         // NOTE: We reuse RecoverableJsAst as a way of making tree clones, because
92         // compilation mutates the tree.  This is unrelated to testing
93         // RecoverableJsAst.
94         return new RecoverableJsAst(ast, true);
95       }

116         CompilerOptions options = new CompilerOptions();

121       private void checkParseErrors(SourceAst realAst, RecoverableJsAst ast, String... expectedErrors) {
122         checkCompile(realAst, ast, null, ImmutableList.copyOf(expectedErrors));
123       }

125       private void checkCompile(SourceAst realAst, RecoverableJsAst ast, String expected) {
126         checkCompile(realAst, ast, expected, ImmutableList.<String>of());
127       }

129       private void checkCompile(SourceAst realAst, RecoverableJsAst ast, String expected,
130           ImmutableList<String> expectedErrors) {
131         Compiler compiler = new Compiler();
132     
133         // Keep this a "small" test. Don't use threads.
134         compiler.disableThreads();
135     
136         JSModule module = new JSModule("m0");
137         module.add(new CompilerInput(ast));
138         compiler.compileModules(ImmutableList.<SourceFile>of(),
139             ImmutableList.of(module),
140             createCompilerOptions());
141     
142         Node mainRoot = compiler.getRoot().getLastChild();
143     
144         Node expectedRoot = null;
145         if (expected != null) {
146           expectedRoot = parseExpectedJs(ImmutableList.of(
147               SourceFile.fromCode("expected.js", expected)));
148           expectedRoot.detachFromParent();
149         }
150     
151         if (expectedRoot == null) {
152           // We use null to signal a parse failure, which results in an empty sources root.
153           assertTrue(mainRoot.isBlock() && !mainRoot.hasChildren());
154         } else {
155           String explanation = expectedRoot.checkTreeEqualsIncludingJsDoc(mainRoot);
156           if (explanation != null) {
157             String expectedAsSource = compiler.toSource(expectedRoot);
158             String mainAsSource = compiler.toSource(mainRoot);
159             if (expectedAsSource.equals(mainAsSource)) {
160               fail("In: " + expectedAsSource + "\n" + explanation);
161             } else {
162               fail("\nExpected: "
163                   + expectedAsSource
164                   + "\nResult:   "
165                   + mainAsSource
166                   + "\n" + explanation);
167             }
168           }
169         }
170     
171         JSError[] errors = compiler.getResult().errors;
172         if (!expectedErrors.isEmpty()) {
173           for (int i = 0; i < expectedErrors.size(); i++) {
174             if (i < errors.length) {
175               assertThat(errors[i].toString()).contains(expectedErrors.get(i));
176             } else {
177               fail("missing error: " + expectedErrors.get(i));
178             }
179           }
180         } else {
181           assertThat(errors).isEmpty();
182         }
183     
184         assertThat(ast.getAstRoot(compiler)).isNotSameAs(realAst.getAstRoot(compiler));
185       }
186     
187       /**
188        * Parses expected JS inputs and returns the root of the parse tree.
189        */
190       protected Node parseExpectedJs(List<SourceFile> inputs) {
191         Compiler compiler = new Compiler();
192     
193         compiler.init(ImmutableList.<SourceFile>of(), inputs, createCompilerOptions());
194         compiler.parse();
195         Node root = compiler.getRoot();
196         assertThat(root).isNotNull();
197         Node externsRoot = root.getFirstChild();
198         Node mainRoot = externsRoot.getNext();
199         return mainRoot;
200       }
