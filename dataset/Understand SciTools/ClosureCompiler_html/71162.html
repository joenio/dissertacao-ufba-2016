17     package com.google.javascript.jscomp.type;
18     
19     import com.google.javascript.jscomp.graph.LatticeElement;
20     import com.google.javascript.rhino.Node;
21     import com.google.javascript.rhino.jstype.JSType;
22     import com.google.javascript.rhino.jstype.StaticTypedScope;
23     import com.google.javascript.rhino.jstype.StaticTypedSlot;
24     
25     /**
26      * A symbol table for inferring types during data flow analysis.
27      *
28      * Each flow scope represents the types of all variables in the scope at
29      * a particular point in the flow analysis.
30      *
31      * @author nicksantos@google.com (Nick Santos)
32      */
33     public interface FlowScope extends StaticTypedScope<JSType>, LatticeElement {
34     
35       /**
36        * Creates a child of this flow scope, to represent an instruction
37        * directly following this one.
38        */
39       FlowScope createChildFlowScope();
40     
41       /**
42        * Defines the type of a symbol at this point in the flow.
43        * @throws IllegalArgumentException If no slot for this symbol exists.
44        */
45       void inferSlotType(String symbol, JSType type);
46     
47       /**
48        * Infer the type of a qualified name.
49        *
50        * When traversing the control flow of a function, simple names are
51        * declared at the bottom of the flow lattice. But there are far too many
52        * qualified names to be able to do this and be performant. So the bottoms
53        * of qualified names are declared lazily.
54        *
55        * Therefore, when inferring a qualified slot, we need both the "bottom"
56        * type of the slot when we enter the scope, and the current type being
57        * inferred.
58        */
59       void inferQualifiedSlot(Node node, String symbol, JSType bottomType,
60           JSType inferredType, boolean declare);
61     
62       /**
63        * Optimize this scope and return a new FlowScope with faster lookup.
64        */
65       FlowScope optimize();
66     
67       /**
68        * Tries to find a unique refined variable in the refined scope, up to the
69        * the blind scope.
70        * @param blindScope The scope before the refinement, i.e. some parent of the
71        *     this scope or itself.
72        * @return The unique refined variable if found or null.
73        */
74       StaticTypedSlot<JSType> findUniqueRefinedSlot(FlowScope blindScope);
75     
76       /**
77        * Look through the given scope, and try to find slots where it doesn't
78        * have enough type information. Then fill in that type information
79        * with stuff that we've inferred in the local flow.
80        */
81       void completeScope(StaticTypedScope<JSType> scope);
82     }
