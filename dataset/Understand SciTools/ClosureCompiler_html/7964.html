17     package com.google.javascript.refactoring;
18     
19     import com.google.common.base.Preconditions;
20     import com.google.common.base.Splitter;
21     import com.google.common.base.Strings;
22     import com.google.javascript.jscomp.CommandLineRunner;
23     import com.google.javascript.jscomp.CompilerOptions;
24     import com.google.javascript.jscomp.ErrorManager;
25     import com.google.javascript.jscomp.TypeMatchingStrategy;
26     
27     import org.kohsuke.args4j.Argument;
28     import org.kohsuke.args4j.CmdLineParser;
29     import org.kohsuke.args4j.Option;
30     import org.kohsuke.args4j.spi.BooleanOptionHandler;
31     
32     import java.io.File;
33     import java.io.IOException;
34     import java.util.ArrayList;
35     import java.util.HashSet;
36     import java.util.List;
37     import java.util.Set;
38     import java.util.TreeSet;
39     import java.util.logging.Level;
40     import java.util.logging.Logger;
41     
42     /**
43      * Main binary that drives a RefasterJS refactoring.
44      *
45      * @author mknichel@google.com (Mark Knichel)
46      */
47     final class RefasterJs {
48     
49       @Option(name = "--help",
50             hidden = true,
51             handler = BooleanOptionHandler.class,
52             usage = "Show instructions for how to use RefasterJS")
53       private boolean displayHelp = false;
54     
55       @Option(
56           name = "--inputs",
57           usage = "List of input files for the refactoring. You may also use glob patterns to "
58               + "match files. For example, use --js='**.js' --js='!**_test.js' "
59               + "to recursively include all js files that do not end in _test.js")
60       private List<String> inputs = new ArrayList<>();
61     
62       @Option(name = "--externs", usage = "List of externs files to use in the compilation.")
63       private List<String> externs = new ArrayList<>();
64     
65       @Option(
66           name = "--refasterjs_template",
67           usage = "Location of the JS file to use as the RefasterJS template.")
68       private String refasterJsTemplate = null;
69     
70       @Option(name = "--env",
71           usage = "Which set of externs to include. Defaults to BROWSER.")
72       private CompilerOptions.Environment environment =
73           CompilerOptions.Environment.BROWSER;
74     
75       @Option(name = "--type_matching",
76         usage = "Which type matching strategy to use. Defaults to SUBTYPES.")
77       private TypeMatchingStrategy typeMatchingStrategy = TypeMatchingStrategy.SUBTYPES;
78     
79       @Option(name = "--dry_run",
80           usage = "Use this to display what changes would be made without applying the changes.")
81       private boolean dryRun = false;
82     
83       @Option(name = "--verbose", usage = "Use this to print verbose statements from RefasterJS.")
84       private boolean verbose = false;
85     
86       @Argument
87       private List<String> arguments = new ArrayList<>();
88     
89       private void doMain(String[] args) throws Exception {
90         CmdLineParser parser = new CmdLineParser(this);
91         parser.parseArgument(args);
92         if (args.length < 1 || displayHelp) {
93           CmdLineParser p = new CmdLineParser(this);
94           p.printUsage(System.out);
95           return;
96         }
97         Preconditions.checkArgument(
98             !Strings.isNullOrEmpty(refasterJsTemplate), "--refasterjs_template must be provided");
99         List<String> fileInputs = getInputs();
100         Preconditions.checkArgument(
101             !fileInputs.isEmpty(), "At least one input must be provided in the --inputs flag.");
102         for (String input : fileInputs) {
103           Preconditions.checkArgument(
104               new File(input).exists(), "Input file %s does not exist.", input);
105         }
106     
107         if (!verbose) {
108           // This is done here instead of using the Compiler#setLoggingLevel function since the
109           // Compiler is created and then run inside of RefactoringDriver.
110           Logger errorManagerLogger = Logger.getLogger("com.google.javascript.jscomp");
111           errorManagerLogger.setLevel(Level.OFF);
112         }
113     
114         RefasterJsScanner scanner = new RefasterJsScanner();
115         scanner.setTypeMatchingStrategy(typeMatchingStrategy);
116         scanner.loadRefasterJsTemplate(refasterJsTemplate);
117         CompilerOptions options = new CompilerOptions();
118         options.setEnvironment(environment);
119         RefactoringDriver driver = new RefactoringDriver.Builder(scanner)
120             .addExterns(CommandLineRunner.getBuiltinExterns(environment))
121             .addExternsFromFile(getExterns())
122             .addInputsFromFile(fileInputs)
123             .build();
124         System.out.println("Compiling JavaScript code and searching for suggested fixes.");
125         List<SuggestedFix> fixes = driver.drive();
126     
127         if (!verbose) {
128           // When running in quiet mode, the Compiler's error manager will not have printed
129           // this information itself.
130           ErrorManager errorManager = driver.getCompiler().getErrorManager();
131           System.out.println("Compiler results: " + errorManager.getErrorCount()
132               + " errors and " + errorManager.getWarningCount() + " warnings.");
133         }
134         System.out.println("Found " + fixes.size() + " suggested fixes.");
135         if (dryRun) {
136           if (!fixes.isEmpty()) {
137             System.out.println("SuggestedFixes: " + fixes);
138           }
139         } else {
140           Set<String> affectedFiles = new TreeSet<>();
141           for (SuggestedFix fix : fixes) {
142             affectedFiles.addAll(fix.getReplacements().keySet());
143           }
144           System.out.println("Modifying affected files: " + affectedFiles);
145           ApplySuggestedFixes.applySuggestedFixesToFiles(fixes);
146         }
147       }
148     
149       private List<String> getInputs() throws IOException {
150         Set<String> patterns = new HashSet<>();
151         // The args4j library can't handle multiple files provided within the same flag option,
152         // like --inputs=file1.js,file2.js so handle that here.
153         Splitter commaSplitter = Splitter.on(',');
154         for (String input : inputs) {
155           patterns.addAll(commaSplitter.splitToList(input));
156         }
157         patterns.addAll(arguments);
158         return CommandLineRunner.findJsFiles(patterns);
159       }
160     
161       private List<String> getExterns() throws IOException {
162         Set<String> patterns = new HashSet<>();
163         // The args4j library can't handle multiple files provided within the same flag option,
164         // like --externs=file1.js,file2.js so handle that here.
165         Splitter commaSplitter = Splitter.on(',');
166         for (String extern : externs) {
167           patterns.addAll(commaSplitter.splitToList(extern));
168         }
169         return CommandLineRunner.findJsFiles(patterns);
170       }
171     
172       public static void main(String[] args) throws Exception {
173         new RefasterJs().doMain(args);
174       }
175     }
