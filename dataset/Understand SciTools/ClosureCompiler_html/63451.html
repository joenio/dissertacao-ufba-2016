17     package com.google.javascript.jscomp;
18     
19     import com.google.common.collect.ImmutableSet;
20     import com.google.javascript.jscomp.CodingConvention.AssertionFunctionSpec;
21     import com.google.javascript.jscomp.NodeTraversal.AbstractPostOrderCallback;
22     import com.google.javascript.jscomp.NodeTraversal.Callback;
23     import com.google.javascript.rhino.Node;
24     
25     import java.util.ArrayList;
26     import java.util.List;
27     import java.util.Set;
28     
29     /**
30      * <p>Compiler pass that removes Closure-specific code patterns.</p>
31      *
32      * <p>Currently does the following:</p>
33      *
34      * <ul>
35      *   <li> Instead of setting abstract methods to a function that throws an
36      *        informative error, this pass allows some binary size reduction by
37      *        removing these methods altogether for production builds.</li>
38      *   <li> Remove calls to assertion functions (like goog.asserts.assert).
39      *        If the return value of the assertion function is used, then
40      *        the first argument (the asserted value) will be directly inlined.
41      *        Otherwise, the entire call will be removed. It is well-known that
42      *        this is not provably safe, much like the equivalent assert
43      *        statement in Java.</li>
44      * </ul>
45      *
46      * @author robbyw@google.com (Robby Walker)
47      */
48     final class ClosureCodeRemoval implements CompilerPass {
49     
50       /** Reference to the JS compiler */
51       private final AbstractCompiler compiler;
52     
53       /** Name used to denote an abstract function */
54       static final String ABSTRACT_METHOD_NAME = "goog.abstractMethod";
55     
56       private final boolean removeAbstractMethods;
57       private final boolean removeAssertionCalls;
58     
59       /**
60        * List of names referenced in successive generations of finding referenced
61        * nodes.
62        */
63       private final List<RemovableAssignment> abstractMethodAssignmentNodes =
64            new ArrayList<>();
65     
66       /**
67        * List of assertion functions.
68        */
69       private final List<Node> assertionCalls = new ArrayList<>();
70     
71     
72       /**
73        * Utility class to track a node and its parent.
74        */
75       private class RemovableAssignment {
76         /**
77          * The node
78          */
79         final Node node;
80     
81         /**
82          * Its parent
83          */
84         final Node parent;
85     
86         /**
87          * Full chain of ASSIGN ancestors
88          */
89         final List<Node> assignAncestors = new ArrayList<>();
90     
91         /**
92          * The last ancestor
93          */
94         final Node lastAncestor;
95     
96         /**
97          * Data structure for information about a removable assignment.
98          *
99          * @param nameNode The LHS
100          * @param assignNode The parent ASSIGN node
101          * @param traversal Access to further levels, assumed to start at 1
102          */
103         public RemovableAssignment(Node nameNode, Node assignNode,
104             NodeTraversal traversal) {
105           this.node = nameNode;
106           this.parent = assignNode;
107     
108           Node ancestor = assignNode;
109           do {
110             ancestor = ancestor.getParent();
111             assignAncestors.add(ancestor);
112           } while (ancestor.isAssign() &&
113                    ancestor.getFirstChild().isQualifiedName());
114           lastAncestor = ancestor.getParent();
115         }
116     
117         /**
118          * Remove this node.
119          */
120         public void remove() {
121           Node rhs = node.getNext();
122           Node last = parent;
123           for (Node ancestor : assignAncestors) {
124             if (ancestor.isExprResult()) {
125               lastAncestor.removeChild(ancestor);
126             } else {
127               rhs.detachFromParent();
128               ancestor.replaceChild(last, rhs);
129             }
130             last = ancestor;
131           }
132           compiler.reportCodeChange();
133         }
134       }
135     
136       /**
137        * Identifies all assignments of the abstract method to a variable.
138        */
139       private class FindAbstractMethods extends AbstractPostOrderCallback {
140         @Override
141         public void visit(NodeTraversal t, Node n, Node parent) {
142           if (n.isAssign()) {
143             Node nameNode = n.getFirstChild();
144             Node valueNode = n.getLastChild();
145     
146             if (nameNode.isQualifiedName() &&
147                 valueNode.isQualifiedName() &&
148                 valueNode.matchesQualifiedName(ABSTRACT_METHOD_NAME)) {
149               // Foo.prototype.bar = goog.abstractMethod
150               abstractMethodAssignmentNodes.add(
151                   new RemovableAssignment(n.getFirstChild(), n, t));
152             } else if (n.getJSDocInfo() != null
153                 && n.getJSDocInfo().isAbstract()
154                 && !n.getJSDocInfo().isConstructor()) {
155               // @abstract
156               abstractMethodAssignmentNodes.add(
157                   new RemovableAssignment(n.getFirstChild(), n, t));
158             }
159           }
160         }
161       }
162     
163     
164       /**
165        * Identifies all assertion calls.
166        */
167       private class FindAssertionCalls extends AbstractPostOrderCallback {
168         final Set<String> assertionNames;
169     
170         FindAssertionCalls() {
171           ImmutableSet.Builder<String> assertionNamesBuilder = ImmutableSet.builder();
172           for (AssertionFunctionSpec spec :
173                    compiler.getCodingConvention().getAssertionFunctions()) {
174             assertionNamesBuilder.add(spec.getFunctionName());
175           }
176           assertionNames = assertionNamesBuilder.build();
177         }
178     
179     
180         @Override
181         public void visit(NodeTraversal t, Node n, Node parent) {
182           if (n.isCall()) {
183             String fnName = n.getFirstChild().getQualifiedName();
184             if (assertionNames.contains(fnName)) {
185               assertionCalls.add(n);
186             }
187           }
188         }
189       }
190     
191     
192       /**
193        * Creates a Closure code remover.
194        *
195        * @param compiler The AbstractCompiler
196        * @param removeAbstractMethods Remove declarations of abstract methods.
197        * @param removeAssertionCalls Remove calls to goog.assert functions.
198        */
199       ClosureCodeRemoval(AbstractCompiler compiler, boolean removeAbstractMethods,
200                          boolean removeAssertionCalls) {
201         this.compiler = compiler;
202         this.removeAbstractMethods = removeAbstractMethods;
203         this.removeAssertionCalls = removeAssertionCalls;
204       }
205     
206       @Override
207       public void process(Node externs, Node root) {
208         List<Callback> passes = new ArrayList<>();
209         if (removeAbstractMethods) {
210           passes.add(new FindAbstractMethods());
211         }
212         if (removeAssertionCalls) {
213           passes.add(new FindAssertionCalls());
214         }
215         CombinedCompilerPass.traverse(compiler, root, passes);
216     
217         for (RemovableAssignment assignment : abstractMethodAssignmentNodes) {
218           assignment.remove();
219         }
220     
221         for (Node call : assertionCalls) {
222           // If the assertion is an expression, just strip the whole thing.
223           Node parent = call.getParent();
224           if (parent.isExprResult()) {
225             parent.getParent().removeChild(parent);
226           } else {
227             // Otherwise, replace the assertion with its first argument,
228             // which is the return value of the assertion.
229             Node firstArg = call.getSecondChild();
230             if (firstArg == null) {
231               parent.replaceChild(call, NodeUtil.newUndefinedNode(call));
232             } else {
233               parent.replaceChild(call, firstArg.detachFromParent());
234             }
235           }
236           compiler.reportCodeChange();
237         }
238       }
239     }
