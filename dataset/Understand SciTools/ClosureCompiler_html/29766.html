17     package com.google.javascript.jscomp;
18     
19     import com.google.common.base.Preconditions;
20     import com.google.javascript.jscomp.NodeTraversal.AbstractPostOrderCallback;
21     import com.google.javascript.rhino.IR;
22     import com.google.javascript.rhino.Node;
23     
24     /**
25      * An optimization that does peephole optimizations of ES6 code.
26      *
27      */
28     class SubstituteEs6Syntax extends AbstractPostOrderCallback implements HotSwapCompilerPass {
29     
30       private final AbstractCompiler compiler;
31     
32       public SubstituteEs6Syntax(AbstractCompiler compiler) {
33         this.compiler = compiler;
34       }
35     
36       @Override
37       public void process(Node externs, Node root) {
38         hotSwapScript(root, null);
39       }
40     
41       @Override
42       public void hotSwapScript(Node scriptRoot, Node originalRoot) {
43         NodeTraversal.traverseEs6(compiler, scriptRoot, this);
44       }
45     
46       @Override
47       public void visit(NodeTraversal t, Node n, Node parent) {
48         switch(n.getType()) {
49           case FUNCTION:
50             if (n.isArrowFunction()) {
51               maybeSimplifyArrowFunctionBody(n, n.getLastChild());
52             }
53             break;
54         }
55       }
56     
57       /**
58        * If possible, replace functions of the form ()=>{ return x; } with ()=>x
59        */
60       private void maybeSimplifyArrowFunctionBody(Node arrowFunction, Node body) {
61         Preconditions.checkArgument(arrowFunction.isArrowFunction());
62         if (!body.isBlock() || body.getChildCount() != 1 || !body.getFirstChild().isReturn()) {
63           return;
64         }
65         Node returnValue = body.getFirstChild().removeFirstChild();
66         arrowFunction.replaceChild(body, returnValue != null ? returnValue : IR.name("undefined"));
67         compiler.reportCodeChange();
68       }
69     }
