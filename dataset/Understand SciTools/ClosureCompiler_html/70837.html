144       /** Creates an instance for a function that might be a constructor. */
145       FunctionType(JSTypeRegistry registry, String name, Node source,
146                    ArrowType arrowType, JSType typeOfThis,
147                    TemplateTypeMap templateTypeMap,
148                    boolean isConstructor, boolean nativeType) {
149         super(registry, name,
150             registry.getNativeObjectType(JSTypeNative.FUNCTION_INSTANCE_TYPE),
151             nativeType, templateTypeMap);
152         setPrettyPrint(true);
153     
154         Preconditions.checkArgument(source == null ||
155             Token.FUNCTION == source.getType());
156         Preconditions.checkNotNull(arrowType);
157         this.source = source;
158         if (isConstructor) {
159           this.kind = Kind.CONSTRUCTOR;
160           this.propAccess = PropAccess.ANY;
161           this.typeOfThis = typeOfThis != null ?
162               typeOfThis : new InstanceObjectType(registry, this, nativeType);
163         } else {
164           this.kind = Kind.ORDINARY;
165           this.typeOfThis = typeOfThis != null ?
166               typeOfThis :
167               registry.getNativeObjectType(JSTypeNative.UNKNOWN_TYPE);
168         }
169         this.call = arrowType;
170         this.isStructuralInterface = false;
171       }

175           TemplateTypeMap typeParameters) {
176         super(registry, name,

221        * When a class B inherits from A and A is annotated as a struct, then B
222        * automatically gets the annotation, even if B's constructor is not
223        * explicitly annotated.
224        */
225       public boolean makesStructs() {
226         if (!hasInstanceType()) {
227           return false;
228         }
229         if (propAccess == PropAccess.STRUCT) {
230           return true;
231         }
232         FunctionType superc = getSuperClassConstructor();
233         if (superc != null && superc.makesStructs()) {
234           setStruct();
235           return true;
236         }
237         return false;
238       }

304       /** Gets the minimum number of arguments that this function requires. */
305       public int getMinArguments() {
306         // NOTE(nicksantos): There are some native functions that have optional
307         // parameters before required parameters. This algorithm finds the position
308         // of the last required parameter.
309         int i = 0;
310         int min = 0;
311         for (Node n : getParameters()) {
312           i++;
313           if (!n.isOptionalArg() && !n.isVarArgs()) {
314             min = i;
315           }
316         }
317         return min;
318       }

320       /**
321        * Gets the maximum number of arguments that this function requires,
322        * or Integer.MAX_VALUE if this is a variable argument function.
323        */
324       public int getMaxArguments() {
325         Node params = getParametersNode();
326         if (params != null) {
327           Node lastParam = params.getLastChild();
328           if (lastParam == null || !lastParam.isVarArgs()) {
329             return params.getChildCount();
330           }
331         }
332     
333         return Integer.MAX_VALUE;
334       }

377       /**
378        * Gets the {@code prototype} property of this function type. This is
379        * equivalent to {@code (ObjectType) getPropertyType("prototype")}.
380        */
381       public ObjectType getPrototype() {
382         // lazy initialization of the prototype field
383         if (prototypeSlot == null) {
384           String refName = getReferenceName();
385           if (refName == null) {
386             // Someone is trying to access the prototype of a structural function.
387             // We don't want to give real properties to this prototype, because
388             // then it would propagate to all structural functions.
389             setPrototypeNoCheck(
390                registry.getNativeObjectType(JSTypeNative.UNKNOWN_TYPE),
391                null);
392           } else {
393             setPrototype(
394                 new PrototypeObjectType(
395                     registry,
396                     getReferenceName() + ".prototype",
397                     registry.getNativeObjectType(OBJECT_TYPE),
398                     isNativeObjectType(), null),
399                 null);
400           }
401         }
402         return (ObjectType) prototypeSlot.getType();
403       }

414       void setPrototypeBasedOn(ObjectType baseType, Node propertyNode) {
415         // This is a bit weird. We need to successfully handle these
416         // two cases:
417         // Foo.prototype = new Bar();
418         // and
419         // Foo.prototype = {baz: 3};
420         // In the first case, we do not want new properties to get
421         // added to Bar. In the second case, we do want new properties
422         // to get added to the type of the anonymous object.
423         //
424         // We handle this by breaking it into two cases:
425         //
426         // In the first case, we create a new PrototypeObjectType and set
427         // its implicit prototype to the type being assigned. This ensures
428         // that Bar will not get any properties of Foo.prototype, but properties
429         // later assigned to Bar will get inherited properly.
430         //
431         // In the second case, we just use the anonymous object as the prototype.
432         if (baseType.hasReferenceName() ||
433             isNativeObjectType() ||
434             baseType.isFunctionPrototypeType()) {
435           baseType = new PrototypeObjectType(
436               registry, getReferenceName() + ".prototype", baseType);
437         }
438         setPrototype(baseType, propertyNode);
439       }

450       /**
451        * Sets the prototype.
452        * @param prototype the prototype. If this value is {@code null} it will
453        *        silently be discarded.
454        */
455       boolean setPrototype(ObjectType prototype, Node propertyNode) {
456         if (prototype == null) {
457           return false;
458         }
459         // getInstanceType fails if the function is not a constructor
460         if (isConstructor() && prototype == getInstanceType()) {
461           return false;
462         }
463         return setPrototypeNoCheck(prototype, propertyNode);
464       }

466       /** Set the prototype without doing any sanity checks. */
467       private boolean setPrototypeNoCheck(ObjectType prototype, Node propertyNode) {
468         ObjectType oldPrototype = prototypeSlot == null
469             ? null : (ObjectType) prototypeSlot.getType();
470         boolean replacedPrototype = oldPrototype != null;
471     
472         this.prototypeSlot = new Property("prototype", prototype, true,
473             propertyNode == null ? source : propertyNode);
474         prototype.setOwnerFunction(this);
475     
476         if (oldPrototype != null) {
477           // Disassociating the old prototype makes this easier to debug--
478           // we don't have to worry about two prototypes running around.
479           oldPrototype.setOwnerFunction(null);
480         }
481     
482         if (isConstructor() || isInterface()) {
483           FunctionType superClass = getSuperClassConstructor();
484           if (superClass != null) {
485             superClass.addSubType(this);
486           }
487     
488           if (isInterface()) {
489             for (ObjectType interfaceType : getExtendedInterfaces()) {
490               if (interfaceType.getConstructor() != null) {
491                 interfaceType.getConstructor().addSubType(this);
492               }
493             }
494           }
495         }
496     
497         if (replacedPrototype) {
498           clearCachedValues();
499         }
500     
501         return true;
502       }

504       /**
505        * check whether or not this function type has implemented
506        * the given interface
507        * if this function is an interface, check whether or not
508        * this interface has extended the given interface
509        * @param interfaceType the interface type
510        * @return true if implemented
511        */
512       public boolean explicitlyImplOrExtInterface(FunctionType interfaceType) {
513         Preconditions.checkArgument(interfaceType.isInterface());
514         for (ObjectType implementedInterface : getAllImplementedInterfaces()) {
515           FunctionType ctor = implementedInterface.getConstructor();
516           if (ctor != null && ctor.checkEquivalenceHelper(
517               interfaceType, EquivalenceMethod.IDENTITY)) {
518             return true;
519           }
520         }
521         for (ObjectType implementedInterface : getExtendedInterfaces()) {
522           FunctionType ctor = implementedInterface.getConstructor();
523           if (ctor != null && ctor.checkEquivalenceHelper(
524               interfaceType, EquivalenceMethod.IDENTITY)) {
525             return true;
526           } else if (ctor != null) {
527             return ctor.explicitlyImplOrExtInterface(interfaceType);
528           }
529         }
530         return false;
531       }

535        * superclasses for any of those interfaces. If this is called before all
536        * types are resolved, it may return an incomplete set.

539         // Store them in a linked hash set, so that the compile job is
540         // deterministic.

545         }

549       private void addRelatedInterfaces(ObjectType instance, Set<ObjectType> set) {
550         FunctionType constructor = instance.getConstructor();
551         if (constructor != null) {
552           if (!constructor.isInterface()) {
553             return;
554           }
555     
556           if (!set.add(instance)) {
557             return;
558           }
559     
560           for (ObjectType interfaceType : instance.getCtorExtendedInterfaces()) {
561             addRelatedInterfaces(interfaceType, set);
562           }
563         }
564       }

566       /** Returns interfaces implemented directly by a class or its superclass. */
567       public Iterable<ObjectType> getImplementedInterfaces() {
568         FunctionType superCtor =
569             isConstructor() ? getSuperClassConstructor() : null;
570         if (superCtor == null) {
571           return implementedInterfaces;
572         }
573         ImmutableList.Builder<ObjectType> builder = ImmutableList.builder();
574         builder.addAll(implementedInterfaces);
575         while (superCtor != null) {
576           builder.addAll(superCtor.implementedInterfaces);
577           superCtor = superCtor.getSuperClassConstructor();
578         }
579         return builder.build();
580       }

587       public void setImplementedInterfaces(List<ObjectType> implementedInterfaces) {
588         if (isConstructor()) {
589           // Records this type for each implemented interface.
590           for (ObjectType type : implementedInterfaces) {
591             registry.registerTypeImplementingInterface(this, type);
592             typeOfThis.extendTemplateTypeMap(type.getTemplateTypeMap());
593           }
594           this.implementedInterfaces = ImmutableList.copyOf(implementedInterfaces);
595         } else {
596           throw new UnsupportedOperationException(
597               "An interface cannot implement other inferfaces");
598         }
599       }

638       @Override
639       public JSType getPropertyType(String name) {
640         if (!hasOwnProperty(name)) {
641           // Define the "call", "apply", and "bind" functions lazily.
642           boolean isCall = "call".equals(name);
643           boolean isBind = "bind".equals(name);
644           if (isCall || isBind) {
645             defineDeclaredProperty(name, getCallOrBindSignature(isCall), source);
646           } else if ("apply".equals(name)) {
647             // Define the "apply" function lazily.
648             FunctionParamBuilder builder = new FunctionParamBuilder(registry);
649     
650             // ECMA-262 says that apply's second argument must be an Array
651             // or an arguments object. We don't model the arguments object,
652             // so let's just be forgiving for now.
653             // TODO(nicksantos): Model the Arguments object.
654             builder.addOptionalParams(
655                 registry.createNullableType(getTypeOfThis()),
656                 registry.createNullableType(
657                     registry.getNativeType(JSTypeNative.OBJECT_TYPE)));
658     
659             defineDeclaredProperty(name,
660                 new FunctionBuilder(registry)
661                 .withParamsNode(builder.build())
662                 .withReturnType(getReturnType())
663                 .withTemplateKeys(getTemplateTypeMap().getTemplateKeys())
664                 .build(),
665                 source);
666           }
667         }
668     
669         return super.getPropertyType(name);
670       }

672       /**
673        * Get the return value of calling "bind" on this function
674        * with the specified number of arguments.
675        *
676        * If -1 is passed, then we will return a result that accepts
677        * any parameters.
678        */
679       public FunctionType getBindReturnType(int argsToBind) {
680         FunctionBuilder builder = new FunctionBuilder(registry)
681             .withReturnType(getReturnType())
682             .withTemplateKeys(getTemplateTypeMap().getTemplateKeys());
683         if (argsToBind >= 0) {
684           Node origParams = getParametersNode();
685           if (origParams != null) {
686             Node params = origParams.cloneTree();
687             for (int i = 1; i < argsToBind && params.getFirstChild() != null; i++) {
688               if (params.getFirstChild().isVarArgs()) {
689                 break;
690               }
691               params.removeFirstChild();
692             }
693             builder.withParamsNode(params);
694           }
695         }
696         return builder.build();
697       }

699       /**
700        * Notice that "call" and "bind" have the same argument signature,
701        * except that all the arguments of "bind" (except the first)
702        * are optional.
703        */
704       private FunctionType getCallOrBindSignature(boolean isCall) {
705         boolean isBind = !isCall;
706         FunctionBuilder builder = new FunctionBuilder(registry)
707             .withReturnType(isCall ? getReturnType() : getBindReturnType(-1))
708             .withTemplateKeys(getTemplateTypeMap().getTemplateKeys());
709     
710         Node origParams = getParametersNode();
711         if (origParams != null) {
712           Node params = origParams.cloneTree();
713     
714           Node thisTypeNode = Node.newString(Token.NAME, "thisType");
715           thisTypeNode.setJSType(
716               registry.createOptionalNullableType(getTypeOfThis()));
717           params.addChildToFront(thisTypeNode);
718     
719           if (isBind) {
720             // The arguments of bind() are unique in that they are all
721             // optional but not undefinable.
722             for (Node current = thisTypeNode.getNext();
723                  current != null; current = current.getNext()) {
724               current.setOptionalArg(true);
725             }
726           } else if (isCall) {
727             // The first argument of call() is optional iff all the arguments
728             // are optional. It's sufficient to check the first argument.
729             Node firstArg = thisTypeNode.getNext();
730             if (firstArg == null
731                 || firstArg.isOptionalArg()
732                 || firstArg.isVarArgs()) {
733               thisTypeNode.setOptionalArg(true);
734             }
735           }
736     
737           builder.withParamsNode(params);
738         }
739     
740         return builder.build();
741       }

743       @Override
744       boolean defineProperty(String name, JSType type,
745           boolean inferred, Node propertyNode) {
746         if ("prototype".equals(name)) {
747           ObjectType objType = type.toObjectType();
748           if (objType != null) {
749             if (prototypeSlot != null &&
750                 objType.isEquivalentTo(prototypeSlot.getType())) {
751               return true;
752             }
753             setPrototypeBasedOn(objType, propertyNode);
754             return true;
755           } else {
756             return false;
757           }
758         }
759         return super.defineProperty(name, type, inferred, propertyNode);
760       }

762       /**
763        * Computes the supremum or infimum of two functions.
764        * Because sup() and inf() share a lot of logic for functions, we use
765        * a single helper.
766        * @param leastSuper If true, compute the supremum of {@code this} with
767        *     {@code that}. Otherwise, compute the infimum.
768        * @return The least supertype or greatest subtype.
769        */
770       FunctionType supAndInfHelper(FunctionType that, boolean leastSuper) {
771         // NOTE(nicksantos): When we remove the unknown type, the function types
772         // form a lattice with the universal constructor at the top of the lattice,
773         // and the LEAST_FUNCTION_TYPE type at the bottom of the lattice.
774         //
775         // When we introduce the unknown type, it's much more difficult to make
776         // heads or tails of the partial ordering of types, because there's no
777         // clear hierarchy between the different components (parameter types and
778         // return types) in the ArrowType.
779         //
780         // Rather than make the situation more complicated by introducing new
781         // types (like unions of functions), we just fallback on the simpler
782         // approach of getting things right at the top and the bottom of the
783         // lattice.
784         //
785         // If there are unknown parameters or return types making things
786         // ambiguous, then sup(A, B) is always the top function type, and
787         // inf(A, B) is always the bottom function type.
788         Preconditions.checkNotNull(that);
789     
790         if (isEquivalentTo(that)) {
791           return this;
792         }
793     
794         // If these are ordinary functions, then merge them.
795         // Don't do this if any of the params/return
796         // values are unknown, because then there will be cycles in
797         // their local lattice and they will merge in weird ways.
798         if (isOrdinaryFunction() && that.isOrdinaryFunction() &&
799             !this.call.hasUnknownParamsOrReturn() &&
800             !that.call.hasUnknownParamsOrReturn()) {
801     
802           // Check for the degenerate case, but double check
803           // that there's not a cycle.
804           boolean isSubtypeOfThat = isSubtype(that);
805           boolean isSubtypeOfThis = that.isSubtype(this);
806           if (isSubtypeOfThat && !isSubtypeOfThis) {
807             return leastSuper ? that : this;
808           } else if (isSubtypeOfThis && !isSubtypeOfThat) {
809             return leastSuper ? this : that;
810           }
811     
812           // Merge the two functions component-wise.
813           FunctionType merged = tryMergeFunctionPiecewise(that, leastSuper);
814           if (merged != null) {
815             return merged;
816           }
817         }
818     
819         // The function instance type is a special case
820         // that lives above the rest of the lattice.
821         JSType functionInstance = registry.getNativeType(
822             JSTypeNative.FUNCTION_INSTANCE_TYPE);
823         if (functionInstance.isEquivalentTo(that)) {
824           return leastSuper ? that : this;
825         } else if (functionInstance.isEquivalentTo(this)) {
826           return leastSuper ? this : that;
827         }
828     
829         // In theory, we should be using the GREATEST_FUNCTION_TYPE as the
830         // greatest function. In practice, we don't because it's way too
831         // broad. The greatest function takes var_args None parameters, which
832         // means that all parameters register a type warning.
833         //
834         // Instead, we use the U2U ctor type, which has unknown type args.
835         FunctionType greatestFn =
836             registry.getNativeFunctionType(JSTypeNative.U2U_CONSTRUCTOR_TYPE);
837         FunctionType leastFn =
838             registry.getNativeFunctionType(JSTypeNative.LEAST_FUNCTION_TYPE);
839         return leastSuper ? greatestFn : leastFn;
840       }

842       /**
843        * Try to get the sup/inf of two functions by looking at the
844        * piecewise components.
845        */
846       private FunctionType tryMergeFunctionPiecewise(
847           FunctionType other, boolean leastSuper) {
848         Node newParamsNode = null;
849         if (call.hasEqualParameters(other.call, EquivalenceMethod.IDENTITY, EqCache.create())) {
850           newParamsNode = call.parameters;
851         } else {
852           // If the parameters are not equal, don't try to merge them.
853           // Someday, we should try to merge the individual params.
854           return null;
855         }
856     
857         JSType newReturnType = leastSuper ?
858             call.returnType.getLeastSupertype(other.call.returnType) :
859             call.returnType.getGreatestSubtype(other.call.returnType);
860     
861         JSType newTypeOfThis = null;
862         if (isEquivalent(typeOfThis, other.typeOfThis)) {
863           newTypeOfThis = typeOfThis;
864         } else {
865           JSType maybeNewTypeOfThis = leastSuper ?
866               typeOfThis.getLeastSupertype(other.typeOfThis) :
867               typeOfThis.getGreatestSubtype(other.typeOfThis);
868           newTypeOfThis = maybeNewTypeOfThis;
869         }
870     
871         boolean newReturnTypeInferred =
872             call.returnTypeInferred || other.call.returnTypeInferred;
873     
874         return new FunctionType(
875             registry, null, null,
876             new ArrowType(
877                 registry, newParamsNode, newReturnType, newReturnTypeInferred),
878             newTypeOfThis, null, false, false);
879       }

896        * that has the property defined (including this interface).
897        */
898       public static ObjectType getTopDefiningInterface(ObjectType type,
899           String propertyName) {
900         ObjectType foundType = null;
901         if (type.hasProperty(propertyName)) {
902           foundType = type;
903         }
904         for (ObjectType interfaceType : type.getCtorExtendedInterfaces()) {
905           if (interfaceType.hasProperty(propertyName)) {
906             foundType = getTopDefiningInterface(interfaceType, propertyName);
907           }
908         }
909         return foundType;

912       /**
913        * Given a constructor or an interface type and a property, finds the
914        * top-most superclass that has the property defined (including this
915        * constructor).
916        */
917       public ObjectType getTopMostDefiningType(String propertyName) {
918         Preconditions.checkState(isConstructor() || isInterface());
919         Preconditions.checkArgument(getInstanceType().hasProperty(propertyName));
920         FunctionType ctor = this;
921     
922         if (isInterface()) {
923           return getTopDefiningInterface(getInstanceType(), propertyName);
924         }
925     
926         ObjectType topInstanceType = null;
927         do {
928           topInstanceType = ctor.getInstanceType();
929           ctor = ctor.getSuperClassConstructor();
930         } while (ctor != null
931             && ctor.getPrototype().hasProperty(propertyName));
932     
933         return topInstanceType;
934       }

936       /**
937        * Two function types are equal if their signatures match. Since they don't
938        * have signatures, two interfaces are equal if their names match.
939        */
940       boolean checkFunctionEquivalenceHelper(
941           FunctionType that, EquivalenceMethod eqMethod, EqCache eqCache) {
942         if (isConstructor()) {
943           if (that.isConstructor()) {
944             return this == that;
945           }
946           return false;
947         }
948         if (isInterface()) {
949           if (that.isInterface()) {
950             return getReferenceName().equals(that.getReferenceName());
951           }
952           return false;
953         }
954         if (that.isInterface()) {
955           return false;
956         }
957     
958         return typeOfThis.checkEquivalenceHelper(that.typeOfThis, eqMethod, eqCache) &&
959             call.checkArrowEquivalenceHelper(that.call, eqMethod, eqCache);
960       }

972       /**
973        * Informally, a function is represented by
974        * {@code function (params): returnType} where the {@code params} is a comma
975        * separated list of types, the first one being a special
976        * {@code this:T} if the function expects a known type for {@code this}.
977        */
978       @Override
979       String toStringHelper(boolean forAnnotations) {
980         if (!isPrettyPrint() ||
981             this == registry.getNativeType(JSTypeNative.FUNCTION_INSTANCE_TYPE)) {
982           return "Function";
983         }
984     
985         setPrettyPrint(false);
986     
987         StringBuilder b = new StringBuilder(32);
988         b.append("function (");
989         int paramNum = call.parameters.getChildCount();
990         boolean hasKnownTypeOfThis = !(typeOfThis instanceof UnknownType);
991         if (hasKnownTypeOfThis) {
992           if (isConstructor()) {
993             b.append("new:");
994           } else {
995             b.append("this:");
996           }
997           b.append(typeOfThis.toStringHelper(forAnnotations));
998         }
999         if (paramNum > 0) {
1000           if (hasKnownTypeOfThis) {
1001             b.append(", ");
1002           }
1003           Node p = call.parameters.getFirstChild();
1004           appendArgString(b, p, forAnnotations);
1005     
1006           p = p.getNext();
1007           while (p != null) {
1008             b.append(", ");
1009             appendArgString(b, p, forAnnotations);
1010             p = p.getNext();
1011           }
1012         }
1013         b.append("): ");
1014         b.append(call.returnType.toStringHelper(forAnnotations));
1015     
1016         setPrettyPrint(true);
1017         return b.toString();
1018       }

1027           b.append(p.getJSType().toStringHelper(forAnnotations));

1057        * the prototype chain.

1064       @Override
1065       protected boolean isSubtype(JSType that,
1066           ImplCache implicitImplCache, SubtypingMode subtypingMode) {
1067         if (JSType.isSubtypeHelper(this, that, implicitImplCache, subtypingMode)) {
1068           return true;
1069         }
1070     
1071         if (that.isFunctionType()) {
1072           FunctionType other = that.toMaybeFunctionType();
1073           if (other.isInterface()) {
1074             // Any function can be assigned to an interface function.
1075             return true;
1076           }
1077           if (isInterface()) {
1078             // An interface function cannot be assigned to anything.
1079             return false;
1080           }
1081     
1082           return treatThisTypesAsCovariant(other, implicitImplCache)
1083               && this.call.isSubtype(other.call, implicitImplCache, subtypingMode);
1084         }
1085     
1086         return getNativeType(JSTypeNative.FUNCTION_PROTOTYPE)
1087             .isSubtype(that, implicitImplCache, subtypingMode);
1088       }

1090       private boolean treatThisTypesAsCovariant(FunctionType other,
1091           ImplCache implicitImplCache) {
1092         // If functionA is a subtype of functionB, then their "this" types
1093         // should be contravariant. However, this causes problems because
1094         // of the way we enforce overrides. Because function(this:SubFoo)
1095         // is not a subtype of function(this:Foo), our override check treats
1096         // this as an error. Let's punt on all this for now.
1097         // TODO(nicksantos): fix this.
1098         boolean treatThisTypesAsCovariant =
1099           // An interface 'this'-type is non-restrictive.
1100           // In practical terms, if C implements I, and I has a method m,
1101           // then any m doesn't necessarily have to C#m's 'this'
1102           // type doesn't need to match I.
1103           (other.typeOfThis.toObjectType() != null &&
1104                other.typeOfThis.toObjectType().getConstructor() != null &&
1105                other.typeOfThis.toObjectType().getConstructor().isInterface()) ||
1106     
1107           // If one of the 'this' types is covariant of the other,
1108           // then we'll treat them as covariant (see comment above).
1109           other.typeOfThis.isSubtype(this.typeOfThis, implicitImplCache, SubtypingMode.NORMAL)
1110           || this.typeOfThis.isSubtype(other.typeOfThis, implicitImplCache, SubtypingMode.NORMAL);
1111         return treatThisTypesAsCovariant;
1112       }

1235       @Override
1236       JSType resolveInternal(ErrorReporter t, StaticTypedScope<JSType> scope) {
1237         setResolvedTypeInternal(this);
1238     
1239         call = (ArrowType) safeResolve(call, t, scope);
1240         if (prototypeSlot != null) {
1241           prototypeSlot.setType(
1242               safeResolve(prototypeSlot.getType(), t, scope));
1243         }
1244     
1245         // Warning about typeOfThis if it doesn't resolve to an ObjectType
1246         // is handled further upstream.
1247         //
1248         // TODO(nicksantos): Handle this correctly if we have a UnionType.
1249         JSType maybeTypeOfThis = safeResolve(typeOfThis, t, scope);
1250         if (maybeTypeOfThis != null) {
1251           if (maybeTypeOfThis.isNullType() || maybeTypeOfThis.isVoidType()) {
1252             typeOfThis = maybeTypeOfThis;
1253           } else {
1254             maybeTypeOfThis = ObjectType.cast(
1255                 maybeTypeOfThis.restrictByNotNullOrUndefined());
1256             if (maybeTypeOfThis != null) {
1257               typeOfThis = maybeTypeOfThis;
1258             }
1259           }
1260         }
1261     
1262         ImmutableList<ObjectType> resolvedImplemented =
1263             resolveTypeListHelper(implementedInterfaces, t, scope);
1264         if (resolvedImplemented != null) {
1265           implementedInterfaces = resolvedImplemented;
1266         }
1267     
1268         ImmutableList<ObjectType> resolvedExtended =
1269             resolveTypeListHelper(extendedInterfaces, t, scope);
1270         if (resolvedExtended != null) {
1271           extendedInterfaces = resolvedExtended;
1272         }
1273     
1274         if (subTypes != null) {
1275           for (int i = 0; i < subTypes.size(); i++) {
1276             subTypes.set(
1277                 i, JSType.toMaybeFunctionType(subTypes.get(i).resolve(t, scope)));
1278           }
1279         }
1280     
1281         return super.resolveInternal(t, scope);
1282       }

1284       /**
1285        * Resolve each item in the list, and return a new list if any
1286        * references changed. Otherwise, return null.
1287        */
1288       private ImmutableList<ObjectType> resolveTypeListHelper(
1289           ImmutableList<ObjectType> list,
1290           ErrorReporter t,
1291           StaticTypedScope<JSType> scope) {
1292         boolean changed = false;
1293         ImmutableList.Builder<ObjectType> resolvedList =
1294             ImmutableList.builder();
1295         for (ObjectType type : list) {
1296           ObjectType resolved = (ObjectType) type.resolve(t, scope);
1297           resolvedList.add(resolved);
1298           changed |= (resolved != type);
1299         }
1300         return changed ? resolvedList.build() : null;
1301       }

1409       // cache is added to prevent infinite recursion when retrieving
1410       // the super type: see testInterfaceExtendsLoop in TypeCheckTest.java
1411       private static void updatePropertyTypeMap(
1412           FunctionType type, Map<String, JSType> propTypeMap,
1413           HashSet<FunctionType> cache) {
1414         if (type == null) { return; }
1415         // retrieve all property types on the prototype of this class
1416         ObjectType prototype = type.getPrototype();
1417         if (prototype != null) {
1418           Set<String> propNames = prototype.getOwnPropertyNames();
1419           for (String name : propNames) {
1420             if (!propTypeMap.containsKey(name)) {
1421               JSType propType = prototype.getPropertyType(name);
1422               propTypeMap.put(name, propType);
1423             }
1424           }
1425         }
1426         // retrieve all property types from its super class
1427         Iterable<ObjectType> iterable = type.getExtendedInterfaces();
1428         if (iterable != null) {
1429           for (ObjectType interfaceType : iterable) {
1430             FunctionType superConstructor = interfaceType.getConstructor();
1431             if (superConstructor == null
1432                 || cache.contains(superConstructor)) { continue; }
1433             cache.add(superConstructor);
1434             updatePropertyTypeMap(superConstructor, propTypeMap, cache);
1435             cache.remove(superConstructor);
1436           }
1437         }
1438       }

1440       /**
1441        * check if there is a loop in the type extends chain
1442        * @return an array of all functions in the loop chain if
1443        * a loop exists, otherwise returns null
1444        */
1445       public List<FunctionType> checkExtendsLoop() {
1446         return checkExtendsLoop(new HashSet<FunctionType>(),
1447             new ArrayList<FunctionType>());
1448       }

1450       public List<FunctionType> checkExtendsLoop(HashSet<FunctionType> cache,
1451           List<FunctionType> path) {
1452         Iterable<ObjectType> iterable = this.getExtendedInterfaces();
1453         if (iterable != null) {
1454           for (ObjectType interfaceType : iterable) {
1455             FunctionType superConstructor = interfaceType.getConstructor();
1456             if (superConstructor == null) { continue; }
1457             if (cache.contains(superConstructor)) {
1458               // after detecting a loop, prune and return the path, e.g.,:
1459               // A -> B -> C -> D -> C, will be pruned into:
1460               // c -> D -> C
1461               path.add(superConstructor);
1462               while (path.get(0) != superConstructor) {
1463                 path.remove(0);
1464               }
1465               return path;
1466             }
1467             cache.add(superConstructor);
1468             path.add(superConstructor);
1469             List<FunctionType> result =
1470                 superConstructor.checkExtendsLoop(cache, path);
1471             if (result != null) {
1472               return result;
1473             }
1474             cache.remove(superConstructor);
1475             path.remove(path.size() - 1);
1476           }
1477         }
1478         return null;
1479       }
