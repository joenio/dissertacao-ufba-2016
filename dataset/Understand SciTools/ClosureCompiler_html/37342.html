40     package com.google.javascript.rhino.jstype;
41     
42     
43     import com.google.common.base.Preconditions;
44     import com.google.javascript.rhino.Node;
45     
46     /**
47      * An object type that is an instance of some function constructor.
48      */
49     class InstanceObjectType extends PrototypeObjectType {
50       private static final long serialVersionUID = 1L;
51     
52       private final FunctionType constructor;
53     
54       InstanceObjectType(JSTypeRegistry registry, FunctionType constructor) {
55         this(registry, constructor, false);
56       }
57     
58       InstanceObjectType(JSTypeRegistry registry, FunctionType constructor,
59                          boolean isNativeType) {
60         super(registry, null, null, isNativeType, constructor.getTemplateTypeMap());
61         Preconditions.checkNotNull(constructor);
62         this.constructor = constructor;
63       }
64     
65       @Override
66       public String getReferenceName() {
67         return getConstructor().getReferenceName();
68       }
69     
70       @Override
71       public boolean hasReferenceName() {
72         return getConstructor().hasReferenceName();
73       }
74     
75       @Override
76       public ObjectType getImplicitPrototype() {
77         return getConstructor().getPrototype();
78       }
79     
80       @Override
81       public FunctionType getConstructor() {
82         return constructor;
83       }
84     
85       @Override
86       boolean defineProperty(String name, JSType type, boolean inferred,
87           Node propertyNode) {
88         ObjectType proto = getImplicitPrototype();
89         if (proto != null && proto.hasOwnDeclaredProperty(name)) {
90           return false;
91         }
92         return super.defineProperty(name, type, inferred, propertyNode);
93       }
94     
95       @Override
96       String toStringHelper(boolean forAnnotations) {
97         if (constructor.hasReferenceName()) {
98           return constructor.getReferenceName();
99         } else {
100           return super.toStringHelper(forAnnotations);
101         }
102       }
103     
104       @Override
105       boolean isTheObjectType() {
106         return getConstructor().isNativeObjectType()
107             && "Object".equals(getReferenceName());
108       }
109     
110       @Override
111       public boolean isInstanceType() {
112         return true;
113       }
114     
115       @Override
116       public boolean isArrayType() {
117         return getConstructor().isNativeObjectType()
118             && "Array".equals(getReferenceName());
119       }
120     
121       @Override
122       public boolean isStringObjectType() {
123         return getConstructor().isNativeObjectType()
124             && "String".equals(getReferenceName());
125       }
126     
127       @Override
128       public boolean isBooleanObjectType() {
129         return getConstructor().isNativeObjectType()
130             && "Boolean".equals(getReferenceName());
131       }
132     
133       @Override
134       public boolean isNumberObjectType() {
135         return getConstructor().isNativeObjectType()
136             && "Number".equals(getReferenceName());
137       }
138     
139       @Override
140       public boolean isDateType() {
141         return getConstructor().isNativeObjectType()
142             && "Date".equals(getReferenceName());
143       }
144     
145       @Override
146       public boolean isRegexpType() {
147         return getConstructor().isNativeObjectType()
148             && "RegExp".equals(getReferenceName());
149       }
150     
151       @Override
152       public boolean isNominalType() {
153         return hasReferenceName();
154       }
155     
156       /**
157        * If this is equal to a NamedType object, its hashCode must be equal
158        * to the hashCode of the NamedType object.
159        */
160       @Override
161       public int hashCode() {
162         if (hasReferenceName()) {
163           return getReferenceName().hashCode();
164         } else {
165           return super.hashCode();
166         }
167       }
168     
169       @Override
170       public Iterable<ObjectType> getCtorImplementedInterfaces() {
171         return getConstructor().getImplementedInterfaces();
172       }
173     
174       @Override
175       public Iterable<ObjectType> getCtorExtendedInterfaces() {
176         return getConstructor().getExtendedInterfaces();
177       }
178     
179       // The owner will always be a resolved type, so there's no need to set
180       // the constructor in resolveInternal.
181       // (it would lead to infinite loops if we did).
182       // JSType resolveInternal(ErrorReporter t, StaticTypedScope<JSType> scope);
183     }
