35     /**
36      * A pass that analyzes side effects to determine when it is safe to move
37      * code from one program point to another.
38      *
39      * In its current form, SideEffectsAnalysis is very incomplete; this is
40      * mostly a sketch to prototype the interface and the broad strokes of
41      * a possible implementation based on flow-insensitive MOD and REF sets.
42      *
43      * See:
44      *
45      * Banning, John. "An efficient way to find the side effects of procedure
46      *      calls and the aliases of variables." POPL '79.
47      *
48      * For an introduction to MOD and REF sets.
49      *
50      * @author dcc@google.com (Devin Coughlin)

136       /**
137        * Determines whether it is safe to move code ({@code source}) across
138        * an environment to another program point (immediately preceding
139        * {@code destination}).
140        *
141        * <p>The notion of "environment" is optimization-specific, but it should
142        * include any code that could be executed between the source program point
143        * and the destination program point.
144        *
145        * {@code destination} must not be a descendant of {@code source}.
146        *
147        * @param source The node that would be moved
148        * @param environment An environment representing the code across which
149        *    the source will be moved.
150        * @param destination The node before which the source would be moved
151        * @return Whether it is safe to move the source to the destination
152        */
153       public boolean safeToMoveBefore(Node source,
154           AbstractMotionEnvironment environment,
155           Node destination) {
156         Preconditions.checkNotNull(locationAbstraction);
157         Preconditions.checkArgument(!nodeHasAncestor(destination, source));
158     
159         // It is always safe to move pure code.
160         if (isPure(source)) {
161           return true;
162         }
163     
164         // Don't currently support interprocedural analysis
165         if (nodeHasCall(source)) {
166           return false;
167         }
168     
169         LocationSummary sourceLocationSummary =
170             locationAbstraction.calculateLocationSummary(source);
171     
172         EffectLocation sourceModSet = sourceLocationSummary.getModSet();
173     
174         // If the source has side effects, then we require that the source
175         // is executed exactly as many times as the destination.
176         if (!sourceModSet.isEmpty() &&
177             !nodesHaveSameControlFlow(source, destination)) {
178             return false;
179         }
180     
181         EffectLocation sourceRefSet = sourceLocationSummary.getRefSet();
182     
183         Set<Node> environmentNodes = environment.calculateEnvironment();
184     
185         for (Node environmentNode : environmentNodes) {
186           if (nodeHasCall(environmentNode)) {
187             return false;
188           }
189         }
190     
191         LocationSummary environmentLocationSummary =
192             locationAbstraction.calculateLocationSummary(environmentNodes);
193     
194         EffectLocation environmentModSet = environmentLocationSummary.getModSet();
195     
196         EffectLocation environmentRefSet = environmentLocationSummary.getRefSet();
197     
198         // If MOD(environment) intersects REF(source) then moving the
199         // source across the environment could cause the source
200         // to read an incorrect value.
201         // If REF(environment) intersects MOD(source) then moving the
202         // source across the environment could cause the environment
203         // to read an incorrect value.
204         // If MOD(environment) intersects MOD(source) then moving the
205         // source across the environment could cause some later code that reads
206         // a modified location to get an incorrect value.
207     
208         return !environmentModSet.intersectsLocation(sourceRefSet)
209             && !environmentRefSet.intersectsLocation(sourceModSet)
210             && !environmentModSet.intersectsLocation(sourceModSet);
211       }

224       /**
225        * Returns true if the two nodes have the same control flow properties,
226        * that is, is node1 be executed every time node2 is executed and vice versa?
227        */
228       private static boolean nodesHaveSameControlFlow(Node node1, Node node2) {
229         /*
230          * We conservatively approximate this with the following criteria:
231          *
232          * Define the "deepest control dependent block" for a node to be the
233          * closest ancestor whose *parent* is a control structure and where that
234          * ancestor may or may be executed depending on the parent.
235          *
236          * So, for example, in:
237          * if (a) {
238          *  b;
239          * } else {
240          *  c;
241          * }
242          *
243          * a has not deepest control dependent block.
244          * b's deepest control dependent block is the "then" block of the IF.
245          * c's deepest control dependent block is the "else" block of the IF.
246          *
247          * We'll say two nodes have the same control flow if
248          *
249          * 1) they have the same deepest control dependent block
250          * 2) that block is either a CASE (which can't have early exits) or it
251          * doesn't have any early exits (e.g. breaks, continues, returns.)
252          *
253          */
254     
255         Node node1DeepestControlDependentBlock =
256             closestControlDependentAncestor(node1);
257     
258         Node node2DeepestControlDependentBlock =
259           closestControlDependentAncestor(node2);
260     
261         if (node1DeepestControlDependentBlock ==
262             node2DeepestControlDependentBlock) {
263     
264           if (node2DeepestControlDependentBlock != null) {
265             // CASE is complicated because we have to deal with fall through and
266             // because some BREAKs are early exits and some are not.
267             // For now, we don't allow movement within a CASE.
268             //
269             // TODO(dcc): be less conservative about movement within CASE
270             if (node2DeepestControlDependentBlock.isCase()) {
271               return false;
272             }
273     
274             // Don't allow breaks, continues, returns in control dependent
275             // block because we don't actually create a control-flow graph
276             // and so don't know if early exits site between the source
277             // and the destination.
278             //
279             // This is overly conservative as it doesn't allow, for example,
280             // moving in the following case:
281             // while (a) {
282             //   source();
283             //
284             //   while(b) {
285             //     break;
286             //   }
287             //
288             //   destination();
289             // }
290             //
291             // To fully support this kind of movement, we'll probably have to use
292             // a CFG-based analysis rather than just looking at the AST.
293             //
294             // TODO(dcc): have nodesHaveSameControlFlow() use a CFG
295             Predicate<Node> isEarlyExitPredicate = new Predicate<Node>() {
296               @Override
297               public boolean apply(Node input) {
298                 Token nodeType = input.getType();
299     
300                 return nodeType == Token.RETURN
301                     || nodeType == Token.BREAK
302                     || nodeType == Token.CONTINUE;
303               }
304             };
305     
306             return !NodeUtil.has(node2DeepestControlDependentBlock,
307                 isEarlyExitPredicate, NOT_FUNCTION_PREDICATE);
308           } else {
309             return true;
310           }
311         } else {
312           return false;
313         }
314       }

316       /**
317        * Returns true if the number of times the child executes depends on the
318        * parent.
319        *
320        * For example, the guard of an IF is not control dependent on the
321        * IF, but its two THEN/ELSE blocks are.
322        *
323        * Also, the guard of WHILE and DO are control dependent on the parent
324        * since the number of times it executes depends on the parent.
325        */
326       private static boolean isControlDependentChild(Node child) {
327         Node parent = child.getParent();
328     
329         if (parent == null) {
330           return false;
331         }
332     
333         ArrayList<Node> siblings = new ArrayList<>();
334         Iterables.addAll(siblings, parent.children());
335     
336         int indexOfChildInParent = siblings.indexOf(child);
337     
338         switch (parent.getType()) {
339           case IF:
340           case HOOK:
341             return (indexOfChildInParent == 1 || indexOfChildInParent == 2);
342           case FOR:
343             // Only initializer is not control dependent
344             return indexOfChildInParent != 0;
345           case SWITCH:
346             return indexOfChildInParent > 0;
347           case WHILE:
348           case DO:
349           case AND:
350           case OR:
351           case FUNCTION:
352             return true;
353     
354           default:
355             return false;
356         }
357       }
358     
359       private static Node closestControlDependentAncestor(Node node) {
360         if (isControlDependentChild(node)) {
361           return node;
362         }
363     
364         // Note: node is not considered one of its ancestors
365         for (Node ancestor : node.getAncestors()) {
366           if (isControlDependentChild(ancestor)) {
367             return ancestor;
368           }
369         }
370     
371         return null;
372       }

374       /**
375        * Returns true if {@code possibleAncestor} is an ancestor of{@code node}.
376        * A node is not considered to be an ancestor of itself.
377        */
378       private static boolean nodeHasAncestor(Node node, Node possibleAncestor) {
379         // Note node is not in node.getAncestors()
380     
381         for (Node ancestor : node.getAncestors()) {
382           if (ancestor == possibleAncestor) {
383             return true;
384           }
385         }
386     
387         return false;
388       }

530       /**
531        * Interface representing the notion of an effect location -- an abstract
532        * location that can be modified or referenced.
533        *
534        * <p>Since there are an infinite number of possible concrete locations
535        * in a running program, this abstraction must be imprecise (i.e. there
536        * will be some distinct concrete locations that are indistinguishable
537        * under the abstraction).
538        *
539        * <p>Different location abstractions will provide their
540        * own implementations of this interface, based on the level and kind
541        * of precision they provide.
542        */
543       private static interface EffectLocation {
544     
545         /**
546          * Does the receiver's effect location intersect a given effect location?
547          * That is, could any of the concrete storage locations (fields, variables,
548          * etc.) represented by the receiver be contained in the set of concrete
549          * storage locations represented by the given abstract effect location.
550          */
551         public boolean intersectsLocation(EffectLocation otherLocation);
552     
553         /**
554          * Returns the result of merging the given effect location with
555          * the receiver. The concrete locations represented by the result must
556          * include all the concrete locations represented by each of the merged
557          * locations and may also possibly include more (i.e., a join may
558          * introduce a loss of precision).
559          */
560         public EffectLocation join(EffectLocation otherLocation);
561     
562         /**
563          * Does the effect location represent any possible concrete locations?
564          */
565         public boolean isEmpty();
566       }

568       /**
569        * An abstract class representing a location abstraction. (Here "abstraction"
570        * means an imprecise representation of concrete side effects.)
571        *
572        * <p>Implementations of this class will each provide own their
573        * implementation(s) of SideEffectLocation and methods to determine the side
574        * effect locations of a given piece of code.
575        */
576       private abstract static class LocationAbstraction  {
577     
578         /** Calculates the abstraction-specific side effects
579          * for the node.
580          */
581         abstract LocationSummary calculateLocationSummary(Node node);
582     
583         /**
584          * Returns an abstraction-specific EffectLocation representing
585          * no location.
586          *
587          * <p>The bottom location joined with any location should return
588          * that location.
589          */
590         abstract EffectLocation getBottomLocation();
591     
592         /**
593          * Calculates the abstraction-specific side effects
594          * for the node.
595          */
596         public LocationSummary calculateLocationSummary(Set<Node> nodes) {
597           EffectLocation modAccumulator = getBottomLocation();
598           EffectLocation refAccumulator = getBottomLocation();
599     
600           for (Node node : nodes) {
601             LocationSummary nodeLocationSummary = calculateLocationSummary(node);
602     
603             modAccumulator = modAccumulator.join(nodeLocationSummary.getModSet());
604             refAccumulator = refAccumulator.join(nodeLocationSummary.getRefSet());
605           }
606     
607           return new LocationSummary(modAccumulator, refAccumulator);
608         }
609       }
610       /**
611        * A very imprecise location abstraction in which there are only two abstract
612        * locations: one representing all concrete locations and one for bottom
613        * (no concrete locations).
614        *
615        * This implementation is a thin wrapper on NodeUtil.mayHaveSideEffects()
616        * and NodeUtil.canBeSideEffected() -- it doesn't add any real value other
617        * than to prototype the LocationAbstraction interface.
618        */
619       private static class DegenerateLocationAbstraction
620           extends LocationAbstraction {
621     
622         private static final EffectLocation EVERY_LOCATION =
623             new DegenerateEffectLocation();
624     
625         private static final EffectLocation NO_LOCATION =
626             new DegenerateEffectLocation();
627     
628         @Override
629         EffectLocation getBottomLocation() {
630           return NO_LOCATION;
631         }
632     
633         @Override
634         public LocationSummary calculateLocationSummary(Node node) {
635           return new LocationSummary(calculateModSet(node), calculateRefSet(node));
636         }
637     
638         static EffectLocation calculateRefSet(Node node) {
639           if (NodeUtil.canBeSideEffected(node)) {
640             return EVERY_LOCATION;
641           } else {
642             return NO_LOCATION;
643           }
644         }
645     
646         static EffectLocation calculateModSet(Node node) {
647           if (NodeUtil.mayHaveSideEffects(node)) {
648             return EVERY_LOCATION;
649           } else {
650             return NO_LOCATION;
651           }
652         }

654         private static class DegenerateEffectLocation implements EffectLocation {
655            @Override
656           public EffectLocation join(EffectLocation otherLocation) {
657             if (otherLocation == EVERY_LOCATION) {
658               return otherLocation;
659             } else {
660               return this;
661             }
662           }
663     
664           @Override
665           public boolean intersectsLocation(EffectLocation otherLocation) {
666             return this == EVERY_LOCATION && otherLocation == EVERY_LOCATION;
667           }
668     
669           @Override
670           public boolean isEmpty() {
671             return this == NO_LOCATION;
672           }
673         }

676       /**
677        * A location abstraction based on the visibility of concrete locations.
678        *
679        * A global variables are treated as one common location, as are all heap
680        * storage locations.
681        *
682        * Local variables are broken up into two classes, one for truly local
683        * variables and one for local variables captured by an inner scope. Each
684        * of these classes has their own separate location representing the
685        * variables in the class.
686        *
687        * Parameter variables are considered to be heap locations since they
688        * can be accessed via the arguments object which itself can be aliased.
689        *
690        * A more precise analysis could:
691        *    1) put parameters on the heap only when "arguments" is actually used
692        *        in a method
693        *    2) recognize that GETPROPs cannot access or modify parameters, only
694        *        GETELEMs
695        *
696        * TODO(dcc): Don't merge parameters with the heap unless necessary.
697        *
698        * Internally, abstract locations are represented as integers
699        * with bits set (masks) representing the storage classes in the location, so
700        * that joining is bit-wise ORing and intersection is bitwise AND.
701        */
702       private static class VisibilityLocationAbstraction
703           extends LocationAbstraction {
704     
705         /** The "bottom" location. Used to signify an empty location set */
706         private static final int VISIBILITY_LOCATION_NONE = 0;
707     
708         /** The "top" location. Used to signify the set containing all locations */
709         private static final int UNKNOWN_LOCATION_MASK = 0xFFFFFFFF;
710     
711         private static final int LOCAL_VARIABLE_LOCATION_MASK = 1 << 1;
712     
713         private static final int CAPTURED_LOCAL_VARIABLE_LOCATION_MASK = 1 << 2;
714     
715         private static final int GLOBAL_VARIABLE_LOCATION_MASK = 1 << 3;
716     
717         private static final int HEAP_LOCATION_MASK = 1 << 4;
718     
719         AbstractCompiler compiler;
720     
721         VariableVisibilityAnalysis variableVisibilityAnalysis;
722         VariableUseDeclarationMap variableUseMap;
723     
724         private VisibilityLocationAbstraction(AbstractCompiler compiler,
725             VariableVisibilityAnalysis variableVisibilityAnalysis,
726             VariableUseDeclarationMap variableUseMap) {
727           this.compiler = compiler;
728           this.variableVisibilityAnalysis = variableVisibilityAnalysis;
729           this.variableUseMap = variableUseMap;
730         }
731     
732         /**
733          * Calculates the MOD/REF summary for the given node.
734          */
735         @Override
736         LocationSummary calculateLocationSummary(Node node) {
737           int visibilityRefLocations = VISIBILITY_LOCATION_NONE;
738           int visibilityModLocations = VISIBILITY_LOCATION_NONE;
739     
740           for (Node reference : findStorageLocationReferences(node)) {
741             int effectMask;
742     
743             if (reference.isName()) {
744               // Variable access
745               effectMask = effectMaskForVariableReference(reference);
746              } else {
747               // Heap access
748               effectMask = HEAP_LOCATION_MASK;
749             }
750     
751             if (storageNodeIsLValue(reference)) {
752               visibilityModLocations |= effectMask;
753             }
754     
755             if (storageNodeIsRValue(reference)) {
756               visibilityRefLocations |= effectMask;
757             }
758           }
759     
760           VisibilityBasedEffectLocation modSet =
761               new VisibilityBasedEffectLocation(visibilityModLocations);
762     
763           VisibilityBasedEffectLocation refSet =
764             new VisibilityBasedEffectLocation(visibilityRefLocations);
765     
766           return new LocationSummary(modSet, refSet);
767         }

769         /**
770          * Returns the set of references to storage locations (both variables
771          * and the heap) under {@code root}.
772          */
773         private Set<Node> findStorageLocationReferences(Node root) {
774           final Set<Node> references = new HashSet<>();
775     
776           NodeTraversal.traverseEs6(compiler, root, new AbstractShallowCallback() {
777             @Override
778             public void visit(NodeTraversal t, Node n, Node parent) {
779               if (NodeUtil.isGet(n)
780                   || (n.isName() && !parent.isFunction())) {
781                 references.add(n);
782               }
783             }
784           });
785     
786           return references;
787         }

789         /**
790          * Calculates the effect mask for a variable reference.
791          */
792         private int effectMaskForVariableReference(Node variableReference) {
793           Preconditions.checkArgument(variableReference.isName());
794     
795           int effectMask = VISIBILITY_LOCATION_NONE;
796     
797           Node declaringNameNode =
798             variableUseMap.findDeclaringNameNodeForUse(variableReference);
799     
800           if (declaringNameNode != null) {
801             VariableVisibility visibility =
802               variableVisibilityAnalysis.getVariableVisibility(declaringNameNode);
803     
804             switch (visibility) {
805               case LOCAL:
806                 effectMask = LOCAL_VARIABLE_LOCATION_MASK;
807                 break;
808               case CAPTURED_LOCAL:
809                 effectMask = CAPTURED_LOCAL_VARIABLE_LOCATION_MASK;
810                 break;
811               case PARAMETER:
812                 // Parameters are considered to be on the heap since they
813                 // can be accessed via the arguments object.
814                 effectMask = HEAP_LOCATION_MASK;
815                 break;
816               case GLOBAL:
817                 effectMask = GLOBAL_VARIABLE_LOCATION_MASK;
818                 break;
819               default:
820                 throw new IllegalStateException("Unrecognized variable" +
821                     " visibility: " + visibility);
822             }
823           } else {
824             // Couldn't find a variable for the reference
825             effectMask = UNKNOWN_LOCATION_MASK;
826           }
827     
828           return effectMask;
829         }

832         EffectLocation getBottomLocation() {
833           return new VisibilityBasedEffectLocation(VISIBILITY_LOCATION_NONE);
834         }

836         /**
837          * Returns true if the node is a storage node.
838          *
839          * Only NAMEs, GETPROPs, and GETELEMs are storage nodes.
840          */
841         private static boolean isStorageNode(Node node) {
842           return node.isName() || NodeUtil.isGet(node);
843         }
844     
845         /**
846          * Return true if the storage node is an r-value.
847          */
848         private static boolean storageNodeIsRValue(Node node) {
849           Preconditions.checkArgument(isStorageNode(node));
850     
851           // We consider all names to be r-values unless
852           // LHS of Token.ASSIGN
853           // LHS of of for in expression
854           // Child of VAR
855     
856           Node parent = node.getParent();
857     
858           if (storageNodeIsLValue(node)) {
859             // Assume l-value is NOT an r-value
860             // unless it is a non-simple assign
861             // or an increment/decrement
862     
863             boolean nonSimpleAssign =
864               NodeUtil.isAssignmentOp(parent) && !parent.isAssign();
865     
866             return (nonSimpleAssign
867                 || parent.isDec()
868                 || parent.isInc());
869           }
870     
871           return true;
872         }
873     
874         /**
875          * Return true if the storage node is an l-value.
876          */
877         private static boolean storageNodeIsLValue(Node node) {
878           Preconditions.checkArgument(isStorageNode(node));
879           return NodeUtil.isLValue(node);
880         }

882         /**
883          * An abstract effect location based the visibility of the
884          * concrete storage location.
885          *
886          * See {@link VisibilityLocationAbstraction} for deeper description
887          * of this abstraction.
888          *
889          * The effect locations are stored as bits set on an integer, so
890          * intersect, join, etc. are the standard bitwise operations.
891          */
892         private static class VisibilityBasedEffectLocation
893             implements EffectLocation {
894           int visibilityMask = VISIBILITY_LOCATION_NONE;
895     
896           public VisibilityBasedEffectLocation(int visibilityMask) {
897             this.visibilityMask = visibilityMask;
898           }
899     
900           @Override
901           public boolean intersectsLocation(EffectLocation otherLocation) {
902             Preconditions.checkArgument(otherLocation instanceof
903                 VisibilityBasedEffectLocation);
904     
905             int otherMask =
906                 ((VisibilityBasedEffectLocation) otherLocation).visibilityMask;
907     
908             return (visibilityMask & otherMask) > 0;
909           }
910     
911           @Override
912           public boolean isEmpty() {
913             return visibilityMask == VISIBILITY_LOCATION_NONE;
914           }
915     
916           @Override
917           public EffectLocation join(EffectLocation otherLocation) {
918             Preconditions.checkArgument(otherLocation instanceof
919                 VisibilityBasedEffectLocation);
920     
921             int otherMask =
922                 ((VisibilityBasedEffectLocation) otherLocation).visibilityMask;
923     
924             int joinedMask = visibilityMask | otherMask;
925     
926             return new VisibilityBasedEffectLocation(joinedMask);
927           }
928         }
929       }

931       /**
932        * Maps NAME nodes that refer to variables to the NAME
933        * nodes that declared them.
934        */
935       private static class VariableUseDeclarationMap {
936     
937         private AbstractCompiler compiler;
938     
939         // Maps a using name to its declaring name
940         private Map<Node, Node> referencesByNameNode;
941     
942         public VariableUseDeclarationMap(AbstractCompiler compiler) {
943           this.compiler = compiler;
944         }
945     
946         /**
947          * Adds a map from each use NAME in {@code root} to its corresponding
948          * declaring name, *provided the declaration is also under root*.
949          *
950          * If the declaration is not under root, then the reference will
951          * not be added to the map.
952          */
953         public void mapUses(Node root) {
954           referencesByNameNode = new HashMap<>();
955     
956           ReferenceCollectingCallback callback =
957             new ReferenceCollectingCallback(compiler,
958                 ReferenceCollectingCallback.DO_NOTHING_BEHAVIOR);
959     
960           NodeTraversal.traverseEs6(compiler, root, callback);
961     
962           for (Var variable : callback.getAllSymbols()) {
963             ReferenceCollection referenceCollection =
964                 callback.getReferences(variable);
965     
966             for (Reference reference : referenceCollection.references) {
967               Node referenceNameNode = reference.getNode();
968     
969               // Note that this counts a declaration as a reference to itself
970               referencesByNameNode.put(referenceNameNode, variable.getNameNode());
971             }
972           }
973         }
974     
975         /**
976          * Returns the NAME node for the declaration of the variable
977          * that {@code usingNameNode} refers to, if it is in the map,
978          * or {@code null} otherwise.
979          */
980         public Node findDeclaringNameNodeForUse(Node usingNameNode) {
981           Preconditions.checkArgument(usingNameNode.isName());
982     
983           return referencesByNameNode.get(usingNameNode);
984         }
985       }
