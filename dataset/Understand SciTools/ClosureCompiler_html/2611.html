34     /**
35      * A PassConfig for the standalone linter, which runs on a single file at a time. This runs a
36      * similar set of checks to what you would get when running the compiler with the lintChecks
37      * DiagnosticGroup enabled, but some of the lint checks depend on type information, which is not
38      * available when looking at a single file, so those are omitted here.
39      */
40     class LintPassConfig extends PassConfig.PassConfigDelegate {
41       LintPassConfig(CompilerOptions options) {
42         super(new DefaultPassConfig(options));
43       }
44     
45       @Override protected List<PassFactory> getChecks() {
46         return ImmutableList.of(
47             earlyLintChecks,
48             closureGoogScopeAliases,
49             closureRewriteClass,
50             lateLintChecks,
51             checkRequires);
52       }
53     
54       @Override protected List<PassFactory> getOptimizations() {
55         return ImmutableList.of();
56       }
57     
58       private final PassFactory earlyLintChecks =
59           new PassFactory("earlyLintChecks", true) {
60             @Override
61             protected CompilerPass create(AbstractCompiler compiler) {
62               return new CombinedCompilerPass(
63                   compiler,
64                   ImmutableList.<Callback>of(
65                       new CheckDuplicateCase(compiler),
66                       new CheckEmptyStatements(compiler),
67                       new CheckEnums(compiler),
68                       new CheckJSDocStyle(compiler),
69                       new CheckJSDoc(compiler),
70                       new CheckMissingSemicolon(compiler),
71                       new CheckMissingSuper(compiler),
72                       new CheckPrimitiveAsObject(compiler),
73                       new CheckRequiresAndProvidesSorted(compiler),
74                       new CheckUnusedLabels(compiler),
75                       new CheckUselessBlocks(compiler),
76                       new ClosureCheckModule(compiler),
77                       new Es6SuperCheck(compiler)));
78             }
79           };
80     
81       private final PassFactory closureGoogScopeAliases =
82           new PassFactory("closureGoogScopeAliases", true) {
83             @Override
84             protected HotSwapCompilerPass create(AbstractCompiler compiler) {
85               return new ScopedAliases(compiler, null, options.getAliasTransformationHandler());
86             }
87           };
88     
89       private final PassFactory closureRewriteClass =
90           new PassFactory("closureRewriteClass", true) {
91             @Override
92             protected HotSwapCompilerPass create(AbstractCompiler compiler) {
93               return new ClosureRewriteClass(compiler);
94             }
95           };
96     
97       private final PassFactory lateLintChecks =
98           new PassFactory("lateLintChecks", true) {
99             @Override
100             protected CompilerPass create(AbstractCompiler compiler) {
101               return new CombinedCompilerPass(
102                   compiler,
103                   ImmutableList.<Callback>of(
104                       new CheckInterfaces(compiler),
105                       new CheckPrototypeProperties(compiler)));
106             }
107           };
108     
109       // This cannot be part of lintChecks because the callbacks in the CombinedCompilerPass don't
110       // get access to the externs.
111       private final PassFactory checkRequires =
112           new PassFactory("checkRequires", true) {
113             @Override
114             protected CompilerPass create(AbstractCompiler compiler) {
115               return new CheckRequiresForConstructors(
116                   compiler, CheckRequiresForConstructors.Mode.SINGLE_FILE);
117             }
118           };
119     }
