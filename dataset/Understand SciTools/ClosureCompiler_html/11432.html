38     /**
39      * A compiler pass that checks that the programmer has obeyed all the access
40      * control restrictions indicated by JSDoc annotations, like
41      * {@code @private} and {@code @deprecated}.
42      *
43      * Because access control restrictions are attached to type information,
44      * it's important that TypedScopeCreator, TypeInference, and InferJSDocInfo
45      * all run before this pass. TypedScopeCreator creates and resolves types,
46      * TypeInference propagates those types across the AST, and InferJSDocInfo
47      * propagates JSDoc across the types.
48      *
49      * @author nicksantos@google.com (Nick Santos)
50      */
51     class CheckAccessControls implements ScopedCallback, HotSwapCompilerPass {

175       @Override
176       public void enterScope(NodeTraversal t) {
177         if (!t.inGlobalScope()) {
178           Node n = t.getScopeRoot();
179           Node parent = n.getParent();
180           if (isDeprecatedFunction(n)) {
181             deprecatedDepth++;
182           }
183           JSType prevClass = getCurrentClass();
184           JSType currentClass = prevClass == null
185               ? getClassOfMethod(n, parent)
186               : prevClass;
187           // ArrayDeques can't handle nulls, so we reuse the bottom type
188           // as a null sentinel.
189           currentClassStack.addFirst(currentClass == null
190               ? noTypeSentinel
191               : currentClass);
192         }
193       }
194     
195       @Override
196       public void exitScope(NodeTraversal t) {
197         if (!t.inGlobalScope()) {
198           Node n = t.getScopeRoot();
199           if (isDeprecatedFunction(n)) {
200             deprecatedDepth--;
201           }
202           currentClassStack.pop();
203         }
204       }

206       /**
207        * Gets the type of the class that "owns" a method, or null if
208        * we know that its un-owned.
209        */
210       private JSType getClassOfMethod(Node n, Node parent) {
211         if (parent.isAssign()) {
212           Node lValue = parent.getFirstChild();
213           if (NodeUtil.isGet(lValue)) {
214             // We have an assignment of the form "a.b = ...".
215             JSType lValueType = lValue.getJSType();
216             if (lValueType != null && lValueType.isNominalConstructor()) {
217               // If a.b is a constructor, then everything in this function
218               // belongs to the "a.b" type.
219               return (lValueType.toMaybeFunctionType()).getInstanceType();
220             } else {
221               // If a.b is not a constructor, then treat this as a method
222               // of whatever type is on "a".
223               return normalizeClassType(lValue.getFirstChild().getJSType());
224             }
225           } else {
226             // We have an assignment of the form "a = ...", so pull the
227             // type off the "a".
228             return normalizeClassType(lValue.getJSType());
229           }
230         } else if (NodeUtil.isFunctionDeclaration(n) ||
231                    parent.isName()) {
232           return normalizeClassType(n.getJSType());
233         } else if (parent.isStringKey()
234             || parent.isGetterDef() || parent.isSetterDef()) {
235           Node objectLitParent = parent.getGrandparent();
236           if (!objectLitParent.isAssign()) {
237             return null;
238           }
239           Node className = NodeUtil.getPrototypeClassName(objectLitParent.getFirstChild());
240           if (className != null) {
241             return normalizeClassType(className.getJSType());
242           }
243         }
244     
245         return null;
246       }

248       /**
249        * Normalize the type of a constructor, its instance, and its prototype
250        * all down to the same type (the instance type).
251        */
252       private static JSType normalizeClassType(JSType type) {
253         if (type == null || type.isUnknownType()) {
254           return type;
255         } else if (type.isNominalConstructor()) {
256           return (type.toMaybeFunctionType()).getInstanceType();
257         } else if (type.isFunctionPrototypeType()) {
258           FunctionType owner = ((ObjectType) type).getOwnerFunction();
259           if (owner.isConstructor()) {
260             return owner.getInstanceType();
261           }
262         }
263         return type;
264       }

271       @Override
272       public void visit(NodeTraversal t, Node n, Node parent) {
273         switch (n.getType()) {
274           case NAME:
275             checkNameDeprecation(t, n, parent);
276             checkNameVisibility(t, n, parent);
277             break;
278           case GETPROP:
279             checkPropertyDeprecation(t, n, parent);
280             checkPropertyVisibility(t, n, parent);
281             checkConstantProperty(t, n);
282             break;
283           case STRING_KEY:
284           case GETTER_DEF:
285           case SETTER_DEF:
286             checkKeyVisibilityConvention(t, n, parent);
287             break;
288           case NEW:
289             checkConstructorDeprecation(t, n, parent);
290             break;
291           case FUNCTION:
292             checkFinalClassOverrides(t, n, parent);
293             break;
294         }
295       }

297       /**
298        * Checks the given NEW node to ensure that access restrictions are obeyed.
299        */
300       private void checkConstructorDeprecation(NodeTraversal t, Node n,
301           Node parent) {
302         JSType type = n.getJSType();
303     
304         if (type != null) {
305           String deprecationInfo = getTypeDeprecationInfo(type);
306     
307           if (deprecationInfo != null &&
308               shouldEmitDeprecationWarning(t, n, parent)) {
309     
310             if (!deprecationInfo.isEmpty()) {
311                 compiler.report(
312                     t.makeError(n, DEPRECATED_CLASS_REASON,
313                         type.toString(), deprecationInfo));
314             } else {
315               compiler.report(
316                   t.makeError(n, DEPRECATED_CLASS, type.toString()));
317             }
318           }
319         }
320       }

322       /**
323        * Checks the given NAME node to ensure that access restrictions are obeyed.
324        */
325       private void checkNameDeprecation(NodeTraversal t, Node n, Node parent) {
326         // Don't bother checking definitions or constructors.
327         if (parent.isFunction() || parent.isVar() ||
328             parent.isNew()) {
329           return;
330         }
331     
332         TypedVar var = t.getTypedScope().getVar(n.getString());
333         JSDocInfo docInfo = var == null ? null : var.getJSDocInfo();
334     
335         if (docInfo != null && docInfo.isDeprecated() &&
336             shouldEmitDeprecationWarning(t, n, parent)) {
337           if (docInfo.getDeprecationReason() != null) {
338             compiler.report(
339                 t.makeError(n, DEPRECATED_NAME_REASON, n.getString(),
340                     docInfo.getDeprecationReason()));
341           } else {
342             compiler.report(
343                 t.makeError(n, DEPRECATED_NAME, n.getString()));
344           }
345         }
346       }

348       /**
349        * Checks the given GETPROP node to ensure that access restrictions are
350        * obeyed.
351        */
352       private void checkPropertyDeprecation(NodeTraversal t, Node n, Node parent) {
353         // Don't bother checking constructors.
354         if (parent.isNew()) {
355           return;
356         }
357     
358         ObjectType objectType =
359             ObjectType.cast(dereference(n.getFirstChild().getJSType()));
360         String propertyName = n.getLastChild().getString();
361     
362         if (objectType != null) {
363           String deprecationInfo
364               = getPropertyDeprecationInfo(objectType, propertyName);
365     
366           if (deprecationInfo != null &&
367               shouldEmitDeprecationWarning(t, n, parent)) {
368     
369             if (!deprecationInfo.isEmpty()) {
370               compiler.report(
371                   t.makeError(n, DEPRECATED_PROP_REASON, propertyName,
372                       typeRegistry.getReadableTypeName(n.getFirstChild()),
373                       deprecationInfo));
374             } else {
375               compiler.report(
376                   t.makeError(n, DEPRECATED_PROP, propertyName,
377                       typeRegistry.getReadableTypeName(n.getFirstChild())));
378             }
379           }
380         }
381       }

384         return enforceCodingConventions
385             && compiler.getCodingConvention().isPrivate(name);

388       /**
389        * Determines whether the given OBJECTLIT property visibility
390        * violates the coding convention.
391        * @param t The current traversal.
392        * @param key The objectlit key node (STRING_KEY, GETTER_DEF, SETTER_DEF).
393        */
394       private void checkKeyVisibilityConvention(NodeTraversal t,
395           Node key, Node parent) {
396         JSDocInfo info = key.getJSDocInfo();
397         if (info == null) {
398           return;
399         }
400         if (!isPrivateByConvention(key.getString())) {
401           return;
402         }
403         Node assign = parent.getParent();
404         if (assign == null || !assign.isAssign()) {
405           return;
406         }
407         Node left = assign.getFirstChild();
408         if (!left.isGetProp()
409             || !left.getLastChild().getString().equals("prototype")) {
410           return;
411         }
412         Visibility declaredVisibility = info.getVisibility();
413         // Visibility is declared to be something other than private.
414         if (declaredVisibility != Visibility.INHERITED
415             && declaredVisibility != Visibility.PRIVATE) {
416           compiler.report(t.makeError(key, CONVENTION_MISMATCH));
417         }
418       }

420       /**
421        * Reports an error if the given name is not visible in the current context.
422        * @param t The current traversal.
423        * @param name The name node.
424        */
425       private void checkNameVisibility(NodeTraversal t, Node name, Node parent) {
426         TypedVar var = t.getTypedScope().getVar(name.getString());
427         if (var == null) {
428           return;
429         }
430     
431         Visibility v = checkPrivateNameConvention(
432             AccessControlUtils.getEffectiveNameVisibility(
433                 name, var, defaultVisibilityForFiles), name);
434     
435         switch (v) {
436           case PACKAGE:
437             if (!isPackageAccessAllowed(var, name)) {
438               compiler.report(
439                   t.makeError(name, BAD_PACKAGE_PROPERTY_ACCESS,
440                       name.getString(), var.getSourceFile().getName()));
441             }
442             break;
443           case PRIVATE:
444             if (!isPrivateAccessAllowed(var, name, parent)) {
445               compiler.report(
446                   t.makeError(name, BAD_PRIVATE_GLOBAL_ACCESS,
447                       name.getString(), var.getSourceFile().getName()));
448             }
449             break;
450           default:
451             // Nothing to do for PUBLIC and PROTECTED
452             // (which is irrelevant for names).
453             break;
454         }
455       }

458       /**
459        * Returns the effective visibility of the given name, reporting an error
460        * if there is a contradiction in the various sources of visibility
461        * (example: a variable with a trailing underscore that is declared
462        * {@code @public}).
463        */
464       private Visibility checkPrivateNameConvention(Visibility v, Node name) {
465         if (isPrivateByConvention(name.getString())) {
466           if (v != Visibility.PRIVATE && v != Visibility.INHERITED) {
467             compiler.report(JSError.make(name, CONVENTION_MISMATCH));
468           }
469           return Visibility.PRIVATE;
470         }
471         return v;
472       }

474       private static boolean isPrivateAccessAllowed(TypedVar var, Node name, Node parent) {
475         StaticSourceFile varSrc = var.getSourceFile();
476         StaticSourceFile refSrc = name.getStaticSourceFile();
477         JSDocInfo docInfo = var.getJSDocInfo();
478         if (varSrc != null
479             && refSrc != null
480             && !varSrc.getName().equals(refSrc.getName())) {
481           return docInfo != null && docInfo.isConstructor()
482               && isValidPrivateConstructorAccess(parent);
483         } else {
484           return true;
485         }
486       }

488       private boolean isPackageAccessAllowed(TypedVar var, Node name) {
489         StaticSourceFile varSrc = var.getSourceFile();
490         StaticSourceFile refSrc = name.getStaticSourceFile();
491         CodingConvention codingConvention = compiler.getCodingConvention();
492         if (varSrc != null && refSrc != null) {
493           String srcPackage = codingConvention.getPackageName(varSrc);
494           String refPackage = codingConvention.getPackageName(refSrc);
495           return srcPackage != null
496               && refPackage != null
497               && srcPackage.equals(refPackage);
498         } else {
499           // If the source file of either var or name is unavailable, conservatively
500           // assume they belong to different packages.
501           // TODO(brndn): by contrast, isPrivateAccessAllowed does allow
502           // private access when a source file is unknown. I didn't change it
503           // in order not to break existing code.
504           return false;
505         }
506       }

508       private void checkOverriddenPropertyVisibilityMismatch(
509           Visibility overriding,
510           Visibility overridden,
511           @Nullable Visibility fileOverview,
512           NodeTraversal t,
513           Node getprop) {
514         if (overriding == Visibility.INHERITED
515             && overriding != overridden
516             && fileOverview != null
517             && fileOverview != Visibility.INHERITED) {
518           String propertyName = getprop.getLastChild().getString();
519           compiler.report(
520               t.makeError(getprop,
521                   BAD_PROPERTY_OVERRIDE_IN_FILE_WITH_FILEOVERVIEW_VISIBILITY,
522                   propertyName,
523                   fileOverview.name()));
524         }
525       }

527       @Nullable private static Visibility getOverridingPropertyVisibility(Node parent) {
528         JSDocInfo overridingInfo = parent.getJSDocInfo();
529         return overridingInfo == null || !overridingInfo.isOverride()
530             ? null
531             : overridingInfo.getVisibility();

537        * Checks if a constructor is trying to override a final class.
538        */
539       private void checkFinalClassOverrides(NodeTraversal t, Node fn, Node parent) {
540         JSType type = fn.getJSType().toMaybeFunctionType();
541         if (type != null && type.isConstructor()) {
542           JSType finalParentClass = getFinalParentClass(getClassOfMethod(fn, parent));
543           if (finalParentClass != null) {
544             compiler.report(
545                 t.makeError(fn, EXTEND_FINAL_CLASS,
546                     type.getDisplayName(), finalParentClass.getDisplayName()));
547           }
548         }

551       /**
552        * Determines whether the given constant property got reassigned
553        * @param t The current traversal.
554        * @param getprop The getprop node.
555        */
556       private void checkConstantProperty(NodeTraversal t,
557           Node getprop) {
558         // Check whether the property is modified
559         Node parent = getprop.getParent();
560         boolean isDelete = parent.isDelProp();
561         if (!(NodeUtil.isAssignmentOp(parent) && parent.getFirstChild() == getprop)
562             && !parent.isInc() && !parent.isDec()
563             && !isDelete) {
564           return;
565         }
566     
567         ObjectType objectType =
568           ObjectType.cast(dereference(getprop.getFirstChild().getJSType()));
569     
570         String propertyName = getprop.getLastChild().getString();
571     
572         boolean isConstant = isPropertyDeclaredConstant(objectType, propertyName);
573     
574         // Check whether constant properties are reassigned
575         if (isConstant) {
576           JSDocInfo info = parent.getJSDocInfo();
577           if (info != null && info.getSuppressions().contains("const")) {
578             return;
579           }
580     
581           if (isDelete) {
582             compiler.report(
583                 t.makeError(getprop, CONST_PROPERTY_DELETED, propertyName));
584             return;
585           }
586     
587           // Can't check for constant properties on generic function types.
588           // TODO(johnlenz): I'm not 100% certain this is necessary, or if
589           // the type is being inspected incorrectly.
590           if (objectType == null
591               || (objectType.isFunctionType()
592                   && !objectType.toMaybeFunctionType().isConstructor())) {
593             return;
594           }
595     
596           ObjectType oType = objectType;
597           while (oType != null) {
598             if (initializedConstantProperties.containsEntry(
599                 oType, propertyName)) {
600               compiler.report(
601                   t.makeError(getprop, CONST_PROPERTY_REASSIGNED_VALUE,
602                       propertyName));
603                 break;
604               }
605             oType = oType.getImplicitPrototype();
606           }
607     
608           initializedConstantProperties.put(objectType,
609               propertyName);
610     
611           // Add the prototype when we're looking at an instance object
612           if (objectType.isInstanceType()) {
613             ObjectType prototype = objectType.getImplicitPrototype();
614             if (prototype != null && prototype.hasProperty(propertyName)) {
615               initializedConstantProperties.put(prototype, propertyName);
616             }
617           }
618         }
619       }

621       /**
622        * Reports an error if the given property is not visible in the current
623        * context.
624        * @param t The current traversal.
625        * @param getprop The getprop node.
626        */
627       private void checkPropertyVisibility(NodeTraversal t,
628           Node getprop, Node parent) {
629         JSDocInfo jsdoc = parent.getJSDocInfo();
630         if (jsdoc != null && jsdoc.getSuppressions().contains("visibility")) {
631           return;
632         }
633     
634         ObjectType referenceType =
635             ObjectType.cast(dereference(getprop.getFirstChild().getJSType()));
636     
637         String propertyName = getprop.getLastChild().getString();
638         boolean isPrivateByConvention = isPrivateByConvention(propertyName);
639     
640         if (isPrivateByConvention
641             && propertyIsDeclaredButNotPrivate(getprop, parent)) {
642           compiler.report(t.makeError(getprop, CONVENTION_MISMATCH));
643           return;
644         }
645     
646         StaticSourceFile definingSource = getDefiningSource(
647             getprop, referenceType, propertyName);
648     
649         boolean isClassType = false;
650     
651         // Is this a normal property access, or are we trying to override
652         // an existing property?
653         boolean isOverride = jsdoc != null
654             && parent.isAssign()
655             && parent.getFirstChild() == getprop;
656     
657         ObjectType objectType = getObjectType(
658             referenceType, isOverride, propertyName);
659     
660         Visibility fileOverviewVisibility =
661             defaultVisibilityForFiles.get(definingSource);
662     
663         Visibility visibility = AccessControlUtils.getEffectivePropertyVisibility(
664             getprop,
665             referenceType,
666             defaultVisibilityForFiles,
667             enforceCodingConventions ? compiler.getCodingConvention() : null);
668     
669         if (isOverride) {
670           Visibility overriding = getOverridingPropertyVisibility(parent);
671           if (overriding != null) {
672             checkOverriddenPropertyVisibilityMismatch(
673                 overriding, visibility, fileOverviewVisibility, t, getprop);
674           }
675         }
676     
677         if (objectType != null) {
678           Property p = objectType.getOwnSlot(propertyName);
679           Node node = p.getNode();
680           if (node == null) {
681             // Assume the property is public.
682             return;
683           }
684           definingSource = node.getStaticSourceFile();
685           isClassType = p.getJSDocInfo().isConstructor();
686         } else if (isPrivateByConvention) {
687           // We can only check visibility references if we know what file
688           // it was defined in.
689           objectType = referenceType;
690         } else if (fileOverviewVisibility == null) {
691           // Otherwise just assume the property is public.
692           return;
693         }
694     
695         StaticSourceFile referenceSource = getprop.getStaticSourceFile();
696     
697         if (isOverride) {
698           boolean sameInput = referenceSource != null
699               && referenceSource.getName().equals(definingSource.getName());
700           checkOverriddenPropertyVisibility(
701               t,
702               getprop,
703               parent,
704               visibility,
705               fileOverviewVisibility,
706               objectType,
707               sameInput);
708         } else {
709           checkNonOverriddenPropertyVisibility(
710               t,
711               getprop,
712               parent,
713               visibility,
714               isClassType,
715               objectType,
716               referenceSource,
717               definingSource);
718         }
719       }

721       private static boolean propertyIsDeclaredButNotPrivate(Node getprop, Node parent) {
722         // This is a declaration with JSDoc
723         JSDocInfo info = NodeUtil.getBestJSDocInfo(getprop);
724         if ((parent.isAssign() || parent.isExprResult())
725             && parent.getFirstChild() == getprop
726             && info != null) {
727           Visibility declaredVisibility = info.getVisibility();
728           if (declaredVisibility != Visibility.PRIVATE
729           && declaredVisibility != Visibility.INHERITED) {
730               return true;
731           }
732         }
733         return false;
734       }

736       @Nullable private static StaticSourceFile getDefiningSource(
737           Node getprop,
738           @Nullable ObjectType referenceType,
739           String propertyName) {
740           if (referenceType != null) {
741             Node propDefNode = referenceType.getPropertyNode(propertyName);
742             if (propDefNode != null) {
743               return propDefNode.getStaticSourceFile();
744             }
745           }
746         return getprop.getStaticSourceFile();
747       }

749       @Nullable private static ObjectType getObjectType(
750           @Nullable ObjectType referenceType,
751           boolean isOverride,
752           String propertyName) {
753         if (referenceType == null) {
754           return null;
755         }
756     
757         // Find the lowest property defined on a class with visibility
758         // information.
759         ObjectType objectType = isOverride
760             ? referenceType.getImplicitPrototype()
761             : referenceType;
762         for (; objectType != null;
763             objectType = objectType.getImplicitPrototype()) {
764           JSDocInfo docInfo = objectType.getOwnPropertyJSDocInfo(propertyName);
765           if (docInfo != null
766               && docInfo.getVisibility() != Visibility.INHERITED) {
767             return objectType;
768           }
769         }
770         return null;
771       }

773       private void checkOverriddenPropertyVisibility(
774           NodeTraversal t,
775           Node getprop,
776           Node parent,
777           Visibility visibility,
778           Visibility fileOverviewVisibility,
779           ObjectType objectType,
780           boolean sameInput) {
781         // Check an ASSIGN statement that's trying to override a property
782         // on a superclass.
783         JSDocInfo overridingInfo = parent.getJSDocInfo();
784         Visibility overridingVisibility = overridingInfo == null
785             ? Visibility.INHERITED
786             : overridingInfo.getVisibility();
787     
788         // Check that:
789         // (a) the property *can* be overridden,
790         // (b) the visibility of the override is the same as the
791         //     visibility of the original property,
792         // (c) the visibility is explicitly redeclared if the override is in
793         //     a file with default visibility in the @fileoverview block.
794         if (visibility == Visibility.PRIVATE && !sameInput) {
795           compiler.report(
796               t.makeError(getprop, PRIVATE_OVERRIDE,
797                   objectType.toString()));
798         } else if (overridingVisibility != Visibility.INHERITED
799             && overridingVisibility != visibility
800             && fileOverviewVisibility == null) {
801           compiler.report(
802               t.makeError(getprop, VISIBILITY_MISMATCH,
803                   visibility.name(), objectType.toString(),
804                   overridingVisibility.name()));
805         }
806       }
807     
808       private void checkNonOverriddenPropertyVisibility(
809           NodeTraversal t,
810           Node getprop,
811           Node parent,
812           Visibility visibility,
813           boolean isClassType,
814           JSType objectType,
815           StaticSourceFile referenceSource,
816           StaticSourceFile definingSource) {
817         // private access is always allowed in the same file.
818         if (referenceSource != null
819             && definingSource != null
820             && referenceSource.getName().equals(definingSource.getName())) {
821           return;
822         }
823     
824         JSType ownerType = normalizeClassType(objectType);
825     
826         switch (visibility) {
827           case PACKAGE:
828             checkPackagePropertyVisibility(t, getprop, referenceSource, definingSource);
829             break;
830           case PRIVATE:
831             checkPrivatePropertyVisibility(t, getprop, parent, isClassType, ownerType);
832             break;
833           case PROTECTED:
834             checkProtectedPropertyVisibility(t, getprop, ownerType);
835             break;
836           default:
837             break;
838         }
839       }

841       private void checkPackagePropertyVisibility(
842           NodeTraversal t,
843           Node getprop,
844           StaticSourceFile referenceSource,
845           StaticSourceFile definingSource) {
846         CodingConvention codingConvention = compiler.getCodingConvention();
847         String refPackage = codingConvention.getPackageName(referenceSource);
848         String defPackage = codingConvention.getPackageName(definingSource);
849         if (refPackage == null
850             || defPackage == null
851             || !refPackage.equals(defPackage)) {
852           String propertyName = getprop.getLastChild().getString();
853           compiler.report(
854               t.makeError(getprop, BAD_PACKAGE_PROPERTY_ACCESS,
855                   propertyName,
856                   typeRegistry.getReadableTypeName(getprop.getFirstChild())));
857           }
858       }

860       @Nullable private JSType getCurrentClass() {
861         JSType cur = currentClassStack.peekFirst();
862         return cur == noTypeSentinel
863             ? null
864             : cur;
865       }

867       private void checkPrivatePropertyVisibility(
868           NodeTraversal t,
869           Node getprop,
870           Node parent,
871           boolean isClassType,
872           JSType ownerType) {
873         JSType currentClass = getCurrentClass();
874         if (currentClass != null && ownerType.isEquivalentTo(currentClass)) {
875           return;
876         }
877     
878         if (isClassType && isValidPrivateConstructorAccess(parent)) {
879           return;
880         }
881     
882         // private access is not allowed outside the file from a different
883         // enclosing class.
884         JSType accessedType = getprop.getFirstChild().getJSType();
885         String propertyName = getprop.getLastChild().getString();
886         String readableTypeName = ownerType.equals(accessedType)
887             ? typeRegistry.getReadableTypeName(getprop.getFirstChild())
888             : ownerType.toString();
889         compiler.report(
890             t.makeError(getprop,
891                 BAD_PRIVATE_PROPERTY_ACCESS,
892                 propertyName,
893                 readableTypeName));
894       }

896       private void checkProtectedPropertyVisibility(
897           NodeTraversal t,
898           Node getprop,
899           JSType ownerType) {
900         // There are 3 types of legal accesses of a protected property:
901         // 1) Accesses in the same file
902         // 2) Overriding the property in a subclass
903         // 3) Accessing the property from inside a subclass
904         // The first two have already been checked for.
905         JSType currentClass = getCurrentClass();
906         if (currentClass == null || !currentClass.isSubtype(ownerType)) {
907           String propertyName = getprop.getLastChild().getString();
908           compiler.report(
909               t.makeError(getprop,  BAD_PROTECTED_PROPERTY_ACCESS,
910                   propertyName,
911                   typeRegistry.getReadableTypeName(getprop.getFirstChild())));
912         }
913       }

915       /**
916        * Whether the given access of a private constructor is legal.
917        *
918        * For example,
919        * new PrivateCtor_(); // not legal
920        * PrivateCtor_.newInstance(); // legal
921        * x instanceof PrivateCtor_ // legal
922        *
923        * This is a weird special case, because our visibility system is inherited
924        * from Java, and JavaScript has no distinction between classes and
925        * constructors like Java does.
926        *
927        * We may want to revisit this if we decide to make the restrictions tighter.
928        */
929       private static boolean isValidPrivateConstructorAccess(Node parent) {
930         return !parent.isNew();
931       }

933       /**
934        * Determines whether a deprecation warning should be emitted.
935        * @param t The current traversal.
936        * @param n The node which we are checking.
937        * @param parent The parent of the node which we are checking.
938        */
939       private boolean shouldEmitDeprecationWarning(
940           NodeTraversal t, Node n, Node parent) {
941         // In the global scope, there are only two kinds of accesses that should
942         // be flagged for warnings:
943         // 1) Calls of deprecated functions and methods.
944         // 2) Instantiations of deprecated classes.
945         // For now, we just let everything else by.
946         if (t.inGlobalScope()) {
947           if (!((parent.isCall() && parent.getFirstChild() == n) ||
948                   n.isNew())) {
949             return false;
950           }
951         }
952     
953         // We can always assign to a deprecated property, to keep it up to date.
954         if (n.isGetProp() && n == parent.getFirstChild() &&
955             NodeUtil.isAssignmentOp(parent)) {
956           return false;
957         }
958     
959         // Don't warn if the node is just declaring the property, not reading it.
960         if (n.isGetProp() && parent.isExprResult() &&
961             n.getJSDocInfo().isDeprecated()) {
962           return false;
963         }
964     
965         return !canAccessDeprecatedTypes(t);
966       }
967     
968       /**
969        * Returns whether it's currently OK to access deprecated names and
970        * properties.
971        *
972        * There are 3 exceptions when we're allowed to use a deprecated
973        * type or property:
974        * 1) When we're in a deprecated function.
975        * 2) When we're in a deprecated class.
976        * 3) When we're in a static method of a deprecated class.
977        */
978       private boolean canAccessDeprecatedTypes(NodeTraversal t) {
979         Node scopeRoot = t.getScopeRoot();
980         Node scopeRootParent = scopeRoot.getParent();
981         return
982           // Case #1
983           (deprecatedDepth > 0) ||
984           // Case #2
985           (getTypeDeprecationInfo(t.getTypedScope().getTypeOfThis()) != null) ||
986             // Case #3
987           (scopeRootParent != null && scopeRootParent.isAssign() &&
988            getTypeDeprecationInfo(
989                getClassOfMethod(scopeRoot, scopeRootParent)) != null);
990       }

992       /**
993        * Returns whether this is a function node annotated as deprecated.
994        */
995       private static boolean isDeprecatedFunction(Node n) {
996         if (n.isFunction()) {
997           JSType type = n.getJSType();
998           if (type != null) {
999             return getTypeDeprecationInfo(type) != null;
1000           }
1001         }
1002     
1003         return false;
1004       }

1006       /**
1007        * Returns the deprecation reason for the type if it is marked
1008        * as being deprecated. Returns empty string if the type is deprecated
1009        * but no reason was given. Returns null if the type is not deprecated.
1010        */
1011       private static String getTypeDeprecationInfo(JSType type) {
1012         if (type == null) {
1013           return null;
1014         }
1015     
1016         JSDocInfo info = type.getJSDocInfo();
1017         if (info != null && info.isDeprecated()) {
1018           if (info.getDeprecationReason() != null) {
1019             return info.getDeprecationReason();
1020           }
1021           return "";
1022         }
1023         ObjectType objType = ObjectType.cast(type);
1024         if (objType != null) {
1025           ObjectType implicitProto = objType.getImplicitPrototype();
1026           if (implicitProto != null) {
1027             return getTypeDeprecationInfo(implicitProto);
1028           }
1029         }
1030         return null;
1031       }

1033       /**
1034        * Returns if a property is declared constant.
1035        */
1036       private boolean isPropertyDeclaredConstant(
1037           ObjectType objectType, String prop) {
1038         if (enforceCodingConventions
1039             && compiler.getCodingConvention().isConstant(prop)) {
1040           return true;
1041         }
1042         for (;
1043              objectType != null;
1044              objectType = objectType.getImplicitPrototype()) {
1045           JSDocInfo docInfo = objectType.getOwnPropertyJSDocInfo(prop);
1046           if (docInfo != null && docInfo.isConstant()) {
1047             return true;
1048           }
1049         }
1050         return false;
1051       }

1053       /**
1054        * Returns the deprecation reason for the property if it is marked
1055        * as being deprecated. Returns empty string if the property is deprecated
1056        * but no reason was given. Returns null if the property is not deprecated.
1057        */
1058       private static String getPropertyDeprecationInfo(ObjectType type,
1059                                                        String prop) {
1060         JSDocInfo info = type.getOwnPropertyJSDocInfo(prop);
1061         if (info != null && info.isDeprecated()) {
1062           if (info.getDeprecationReason() != null) {
1063             return info.getDeprecationReason();
1064           }
1065     
1066           return "";
1067         }
1068         ObjectType implicitProto = type.getImplicitPrototype();
1069         if (implicitProto != null) {
1070           return getPropertyDeprecationInfo(implicitProto, prop);
1071         }
1072         return null;
1073       }

1082       /**
1083        * Returns the super class of the given type that has a constructor.
1084        */
1085       private static JSType getFinalParentClass(JSType type) {
1086         if (type != null) {
1087           ObjectType iproto = ObjectType.cast(type).getImplicitPrototype();
1088           while (iproto != null && iproto.getConstructor() == null) {
1089             iproto = iproto.getImplicitPrototype();
1090           }
1091           if (iproto != null) {
1092             Node source = iproto.getConstructor().getSource();
1093             JSDocInfo jsDoc = source != null ? NodeUtil.getBestJSDocInfo(source) : null;
1094             if (jsDoc != null && jsDoc.isConstant()) {
1095               return iproto;
1096             }
1097           }
1098         }
1099         return null;
1100       }
