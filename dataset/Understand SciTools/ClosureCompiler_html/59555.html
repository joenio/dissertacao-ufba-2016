36     /**
37      * A sorted list of inputs following the ES6 module ordering spec.
38      * <p>
39      * Orders such that each input always comes after its dependencies.
40      * Circular references are allowed by emitting the current input in the moment
41      * before a loop would complete.
42      * <p>
43      * The resulting order is not the same as the user-provided order but is
44      * influenced by it since it may take more than one graph traversal to account
45      * for all provided inputs and the graph traversals start with the first user
46      * provided input and continue from there.
47      * <p>
48      * Also exposes other information about the inputs, like which inputs
49      * do not provide symbols.
50      *
51      */
52     public final class Es6SortedDependencies<INPUT extends DependencyInfo>
53         implements SortedDependencies<INPUT> {
54     
55       private final List<INPUT> userOrderedInputs = new ArrayList<>();
56       private final List<INPUT> importOrderedInputs = new ArrayList<>();
57       private final Set<INPUT> completedInputs = new HashSet<>();
58       private final Map<String, INPUT> nonExportingInputs = new LinkedHashMap<>();
59       private final Map<String, INPUT> exportingInputBySymbolName = new HashMap<>();
60       // Maps an input A to the inputs it depends on, ie, inputs that provide stuff that A requires.
61       private final Multimap<INPUT, INPUT> importedInputByImportingInput = LinkedHashMultimap.create();
62     
63       public Es6SortedDependencies(List<INPUT> userOrderedInputs) {
64         this.userOrderedInputs.addAll(userOrderedInputs);
65         processInputs();
66       }
67     
68       @Override
69       public List<INPUT> getDependenciesOf(List<INPUT> rootInputs, boolean sorted) {
70         Preconditions.checkArgument(userOrderedInputs.containsAll(rootInputs));
71     
72         Set<INPUT> includedInputs = new HashSet<>();
73         Deque<INPUT> worklist = new ArrayDeque<>(rootInputs);
74         while (!worklist.isEmpty()) {
75           INPUT input = worklist.pop();
76           if (includedInputs.add(input)) {
77             for (String symbolName : input.getRequires()) {
78               INPUT importedSymbolName = exportingInputBySymbolName.get(symbolName);
79               if (importedSymbolName != null) {
80                 worklist.add(importedSymbolName);
81               }
82             }
83           }
84         }
85     
86         ImmutableList.Builder<INPUT> builder = ImmutableList.builder();
87         for (INPUT input : (sorted ? importOrderedInputs : userOrderedInputs)) {
88           if (includedInputs.contains(input)) {
89             builder.add(input);
90           }
91         }
92         return builder.build();
93       }
94     
95       @Override
96       public INPUT getInputProviding(String symbolName) throws MissingProvideException {
97         INPUT input = maybeGetInputProviding(symbolName);
98         if (input != null) {
99           return input;
100         }
101     
102         throw new MissingProvideException(symbolName);
103       }
104     
105       @Override
106       public List<INPUT> getInputsWithoutProvides() {
107         return ImmutableList.copyOf(nonExportingInputs.values());
108       }
109     
110       @Override
111       public List<INPUT> getSortedDependenciesOf(List<INPUT> roots) {
112         return getDependenciesOf(roots, true);
113       }
114     
115       @Override
116       public List<INPUT> getSortedList() {
117         return Collections.unmodifiableList(importOrderedInputs);
118       }
119     
120       @Override
121       public INPUT maybeGetInputProviding(String symbol) {
122         if (exportingInputBySymbolName.containsKey(symbol)) {
123           return exportingInputBySymbolName.get(symbol);
124         }
125     
126         return nonExportingInputs.get(toModuleName(createUri(symbol)));
127       }
128     
129       /**
130        * Turns a filename into a JS identifier that is used for moduleNames in
131        * rewritten code. Removes leading ./, replaces / with $, removes trailing .js
132        * and replaces - with _. All moduleNames get a "module$" prefix.
133        *
134        * @see com.google.javascript.jscomp.ES6ModuleLoader
135        *
136        * TODO(tbreisacher): Switch to using the ModuleIdentifier class once
137        * it no longer causes circular dependencies in Google builds.
138        */
139       private static String toModuleName(URI filename) {
140         String moduleName = filename.toString();
141         if (moduleName.endsWith(".js")) {
142           moduleName = moduleName.substring(0, moduleName.length() - 3);
143         }
144     
145         moduleName =
146             moduleName
147                 .replaceAll("^\\./", "")
148                 .replace('/', '$')
149                 .replace('\\', '$')
150                 .replace('-', '_')
151                 .replace(':', '_')
152                 .replace('.', '_')
153                 .replace("%20", "_");
154         return "module$" + moduleName;
155       }
156     
157       /**
158        * Copied from ES6ModuleLoader because our BUILD rules are written in such a way that we can't
159        * depend on ES6ModuleLoader from here.
160        * TODO(tbreisacher): Switch to using the ES6ModuleLoader once the BUILD graph allows it.
161        */
162       private static URI createUri(String input) {
163         // Handle special characters
164         String encodedInput = input.replace(':', '-')
165             .replace('\\', '/')
166             .replace(" ", "%20")
167             .replace("[", "%5B")
168             .replace("]", "%5D")
169             .replace("<", "%3C")
170             .replace(">", "%3E");
171     
172         return URI.create(encodedInput).normalize();
173       }
174     
175       private void orderInput(INPUT input) {
176         if (completedInputs.contains(input)) {
177           return;
178         }
179     
180         completedInputs.add(input);
181         for (INPUT importedInput : importedInputByImportingInput.get(input)) {
182           orderInput(importedInput);
183         }
184     
185         // Emit an input after its imports have been emitted.
186         importOrderedInputs.add(input);
187       }
188     
189       private void processInputs() {
190         // Index.
191         for (INPUT userOrderedInput : userOrderedInputs) {
192           if (userOrderedInput.getProvides().isEmpty()) {
193             nonExportingInputs.put(
194                 toModuleName(createUri(userOrderedInput.getName())), userOrderedInput);
195           }
196           for (String providedSymbolName : userOrderedInput.getProvides()) {
197             exportingInputBySymbolName.put(providedSymbolName, userOrderedInput);
198           }
199         }
200         for (INPUT userOrderedInput : userOrderedInputs) {
201           for (String symbolName : userOrderedInput.getRequires()) {
202             INPUT importedInput = exportingInputBySymbolName.get(symbolName);
203             if (importedInput != null) {
204               importedInputByImportingInput.put(userOrderedInput, importedInput);
205             }
206           }
207         }
208     
209         // Order.
210         // For each input, traverse in user-provided order.
211         for (INPUT userOrderedInput : userOrderedInputs) {
212           // Traverse the graph starting from this input and record any
213           // newly-reached inputs.
214           orderInput(userOrderedInput);
215         }
216     
217         // Free temporary indexes.
218         completedInputs.clear();
219         importedInputByImportingInput.clear();
220       }
221     }
