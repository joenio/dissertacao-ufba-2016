17     package com.google.javascript.jscomp;
18     
19     import com.google.javascript.jscomp.NodeTraversal.Callback;
20     import com.google.javascript.jscomp.graph.LinkedDirectedGraph;
21     import com.google.javascript.rhino.Node;
22     
23     import java.util.Comparator;
24     
25     /**
26      * Control flow graph.
27      *
28      *
29      * @param <N> The instruction type of the control flow graph.
30      */
31     public class ControlFlowGraph<N> extends
32         LinkedDirectedGraph<N, ControlFlowGraph.Branch> {
33     
34       /**
35        * A special node marked by the node value key null to a singleton
36        * "return" when control is transferred outside of the current control flow
37        * graph.
38        */
39       private final DiGraphNode<N, ControlFlowGraph.Branch> implicitReturn;
40     
41       private final DiGraphNode<N, ControlFlowGraph.Branch> entry;
42     
43       /**
44        * Constructor.
45        */
46       ControlFlowGraph(
47           N entry, boolean nodeAnnotations, boolean edgeAnnotations) {
48         super(nodeAnnotations, edgeAnnotations);
49         implicitReturn = createDirectedGraphNode(null);
50         this.entry = createDirectedGraphNode(entry);
51       }
52     
53       /**
54        * Gets the implicit return node.
55        *
56        * @return Return node.
57        */
58       public DiGraphNode<N, ControlFlowGraph.Branch> getImplicitReturn() {
59         return implicitReturn;
60       }
61     
62       /**
63        * Gets the entry point of the control flow graph. In general, this should be
64        * the beginning of the global script or beginning of a function.
65        *
66        * @return The entry point.
67        */
68       public DiGraphNode<N, ControlFlowGraph.Branch> getEntry() {
69         return entry;
70       }
71     
72       /**
73        * Checks whether node is the implicit return.
74        *
75        * @param node Node.
76        * @return True if the node is the implicit return.
77        */
78       public boolean isImplicitReturn(
79           DiGraphNode<N, ControlFlowGraph.Branch> node) {
80         return node == implicitReturn;
81       }
82     
83       /**
84        * Gets a comparator for the nodes. The default implementation returns
85        * {@code null}. See {@link ControlFlowGraph#getOptionalNodeComparator}.
86        * @param isForward Whether the comparator sorts the nodes in the direction of
87        *    the flow.
88        * @return a comparator or null (in particular, if not overridden)
89        */
90       public Comparator<DiGraphNode<N, Branch>> getOptionalNodeComparator(
91           boolean isForward) {
92         return null;
93       }
94     
95       /**
96        * The edge object for the control flow graph.
97        */
98       public static enum Branch {
99         /** Edge is taken if the condition is true. */
100         ON_TRUE,
101         /** Edge is taken if the condition is false. */
102         ON_FALSE,
103         /** Unconditional branch. */
104         UNCOND,
105         /**
106          * Exception-handling code paths.
107          * Conflates two kind of control flow passing:
108          * - An exception is thrown, and falls into a catch or finally block
109          * - During exception handling, a finally block finishes and control
110          *   passes to the next finally block.
111          * In theory, we need 2 different edge types. In practice, we
112          * can just treat them as "the edges we can't really optimize".
113          */
114         ON_EX,
115         /** Possible folded-away template */
116         SYN_BLOCK;
117     
118         public boolean isConditional() {
119           return this == ON_TRUE || this == ON_FALSE;
120         }
121       }
122     
123       /**
124        * Abstract callback to visit a control flow graph node without going into
125        * subtrees of the node that are also represented by other
126        * control flow graph nodes.
127        *
128        * <p>For example, traversing an IF node as root will visit the two subtrees
129        * pointed by the {@link ControlFlowGraph.Branch#ON_TRUE} and
130        * {@link ControlFlowGraph.Branch#ON_FALSE} edges.
131        */
132       public abstract static class AbstractCfgNodeTraversalCallback implements
133           Callback {
134         @Override
135         public final boolean shouldTraverse(NodeTraversal nodeTraversal, Node n,
136             Node parent) {
137           if (parent == null) {
138             return true;
139           }
140           return !isEnteringNewCfgNode(n);
141         }
142       }
143     
144       /**
145        * @return True if n should be represented by a new CFG node in the control
146        * flow graph.
147        */
148       public static boolean isEnteringNewCfgNode(Node n) {
149         Node parent = n.getParent();
150         switch (parent.getType()) {
151           case BLOCK:
152           case SCRIPT:
153           case TRY:
154             return true;
155           case FUNCTION:
156             // A function node represents the start of a function where the name
157             // bleeds into the local scope and parameters are assigned
158             // to the formal argument names. The node includes the name of the
159             // function and the LP list since we assume the whole set up process
160             // is atomic without change in control flow. The next change of
161             // control is going into the function's body, represented by the second
162             // child.
163             return n != parent.getSecondChild();
164           case WHILE:
165           case DO:
166           case IF:
167             // These control structures are represented by a node that holds the
168             // condition. Each of them is a branch node based on its condition.
169             return NodeUtil.getConditionExpression(parent) != n;
170     
171           case FOR:
172             // The FOR(;;) node differs from other control structures in that
173             // it has an initialization and an increment statement. Those
174             // two statements have corresponding CFG nodes to represent them.
175             // The FOR node only represents the condition check for each iteration.
176             // That way the following:
177             // for(var x = 0; x < 10; x++) { } has a graph that is isomorphic to
178             // var x = 0; while(x<10) {  x++; }
179             if (NodeUtil.isForIn(parent)) {
180               // TODO(user): Investigate how we should handle the case where
181               // we have a very complex expression inside the FOR-IN header.
182               return n != parent.getFirstChild();
183             } else {
184               return NodeUtil.getConditionExpression(parent) != n;
185             }
186           case SWITCH:
187           case CASE:
188           case CATCH:
189           case WITH:
190             return n != parent.getFirstChild();
191           default:
192             return false;
193         }
194       }
195     
196       @Override
197       public String toString() {
198         String s = "CFG:\n";
199         for (GraphvizEdge e : getGraphvizEdges()) {
200           s += e.toString() + '\n';
201         }
202         return s;
203       }
204     }
