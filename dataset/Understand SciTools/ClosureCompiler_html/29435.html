17     package com.google.javascript.jscomp.newtypes;
18     
19     import com.google.common.base.MoreObjects;
20     import com.google.common.base.Preconditions;
21     import com.google.common.collect.ImmutableSet;
22     
23     import java.util.Collection;
24     import java.util.Iterator;
25     import java.util.Map;
26     
27     /**
28      * A persistent map from variables to abstract values (types)
29      *
30      * @author blickly@google.com (Ben Lickly)
31      * @author dimvar@google.com (Dimitris Vardoulakis)
32      */
33     public final class TypeEnv {
34       private final PersistentMap<String, JSType> typeMap;
35     
36       // Above this threshold, the type env keeps track of which variables have
37       // changed, in order to improve the speed of joins.
38       private static final int SIZE_THRESHOLD = 100;
39       private PersistentSet<String> changedVars = null;
40     
41       public TypeEnv() {
42         this.typeMap = PersistentMap.create();
43       }
44     
45       private TypeEnv(PersistentMap<String, JSType> typeMap) {
46         this.typeMap = typeMap;
47         if (typeMap.size() >= SIZE_THRESHOLD) {
48           this.changedVars = PersistentSet.create();
49         }
50       }
51     
52       private TypeEnv(PersistentMap<String, JSType> typeMap,
53           PersistentSet<String> changedVars) {
54         Preconditions.checkState(typeMap.size() >= SIZE_THRESHOLD);
55         this.typeMap = typeMap;
56         this.changedVars = changedVars;
57       }
58     
59       public JSType getType(String n) {
60         Preconditions.checkArgument(!n.contains("."));
61         return typeMap.get(n);
62       }
63     
64       public TypeEnv putType(String n, JSType t) {
65         Preconditions.checkArgument(!n.contains("."));
66         Preconditions.checkArgument(t != null);
67         if (changedVars == null) {
68           return new TypeEnv(typeMap.with(n, t));
69         }
70         JSType oldType = typeMap.get(n);
71         if (oldType == null) {
72           // The environment is being initialized; don't keep a log yet.
73           return new TypeEnv(typeMap.with(n, t));
74         } else if (t.equals(oldType)) {
75           return this;
76         } else {
77           return new TypeEnv(typeMap.with(n, t), changedVars.with(n));
78         }
79       }
80     
81       // This method clears the change log, so it can change the correctness
82       // because it influences what is joined.
83       // In NewTypeInference, we call it only from statements at the top level of a
84       // scope, where we know that we are not in a branch, so we can forget the log.
85       public TypeEnv clearChangeLog() {
86         if (changedVars == null || changedVars.isEmpty()) {
87           return this;
88         }
89         return new TypeEnv(typeMap);
90       }
91     
92       public static TypeEnv join(TypeEnv e1, TypeEnv e2) {
93         return join(ImmutableSet.of(e1, e2));
94       }
95     
96       // In NewTypeInference, we call join with a set of type envs.
97       // This is OK because we use reference equality for type envs.
98       // Don't override equals to do logical equality!
99       public static TypeEnv join(Collection<TypeEnv> envs) {
100         Preconditions.checkArgument(!envs.isEmpty());
101         Iterator<TypeEnv> envsIter = envs.iterator();
102         TypeEnv firstEnv = envsIter.next();
103     
104         if (!envsIter.hasNext()) {
105           return firstEnv;
106         }
107         PersistentMap<String, JSType> newMap = firstEnv.typeMap;
108     
109         if (firstEnv.changedVars == null) {
110           while (envsIter.hasNext()) {
111             TypeEnv env = envsIter.next();
112             for (Map.Entry<String, JSType> entry : env.typeMap.entrySet()) {
113               String name = entry.getKey();
114               // TODO(dimvar):
115               // If the iteration order in the type envs is guaranteed to get the
116               // keys in the same order for any env, then we can iterate through the
117               // two type envs at the same time, to avoid the map lookup here.
118               JSType currentType = newMap.get(name);
119               JSType otherType = entry.getValue();
120               Preconditions.checkNotNull(
121                   currentType, "%s is missing from an env", name);
122               if (!currentType.equals(otherType)) {
123                 newMap = newMap.with(name, JSType.join(currentType, otherType));
124               }
125             }
126           }
127           return new TypeEnv(newMap);
128         }
129     
130         PersistentSet<String> newLog = PersistentSet.create();
131         for (TypeEnv env : envs) {
132           for (String varName : env.changedVars) {
133             newLog = newLog.with(varName);
134           }
135         }
136         while (envsIter.hasNext()) {
137           TypeEnv env = envsIter.next();
138           for (String changedVar : newLog) {
139             JSType currentType = newMap.get(changedVar);
140             JSType otherType = env.typeMap.get(changedVar);
141             if (!currentType.equals(otherType)) {
142               newMap = newMap.with(changedVar, JSType.join(currentType, otherType));
143             }
144           }
145         }
146         return new TypeEnv(newMap, newLog);
147       }
148     
149       @Override
150       public String toString() {
151         MoreObjects.ToStringHelper helper = MoreObjects.toStringHelper(this);
152         for (String key : typeMap.keySet()) {
153           helper.add(key, getType(key));
154         }
155         return helper.toString();
156       }
157     }
