27     /**
28      * This class is mostly used by passes testing the old type checker.
29      * Passes that run after type checking and need type information use
30      * the class TypeICompilerTestCase.
31      */
32     abstract class CompilerTypeTestCase extends BaseJSTypeTestCase {
33     
34       static final String CLOSURE_DEFS =
35           "var goog = {};" +
36           "goog.inherits = function(x, y) {};" +
37           "/** @type {!Function} */ goog.abstractMethod = function() {};" +
38           "goog.isArray = function(x) {};" +
39           "goog.isDef = function(x) {};" +
40           "goog.isFunction = function(x) {};" +
41           "goog.isNull = function(x) {};" +
42           "goog.isString = function(x) {};" +
43           "goog.isObject = function(x) {};" +
44           "goog.isDefAndNotNull = function(x) {};" +
45           "goog.array = {};" +
46           // simplified ArrayLike definition
47           "/**\n" +
48           " * @typedef {Array|{length: number}}\n" +
49           " */\n" +
50           "goog.array.ArrayLike;" +
51           "/**\n" +
52           " * @param {Array.<T>|{length:number}} arr\n" +
53           " * @param {function(this:S, T, number, goog.array.ArrayLike):boolean} f\n" +
54           " * @param {S=} opt_obj\n" +
55           " * @return {!Array.<T>}\n" +
56           " * @template T,S\n" +
57           " */" +
58           // return empty array to satisfy return type
59           "goog.array.filter = function(arr, f, opt_obj){ return []; };" +
60           "goog.asserts = {};" +
61           "/** @return {*} */ goog.asserts.assert = function(x) { return x; };";
62     
63       /** A default set of externs for testing. */
64       static final String DEFAULT_EXTERNS =
65           CompilerTestCase.DEFAULT_EXTERNS;
66     
67       protected Compiler compiler;
68     
69       protected CompilerOptions getOptions() {
70         CompilerOptions options = new CompilerOptions();
71         options.setLanguageIn(LanguageMode.ECMASCRIPT5);
72         options.setWarningLevel(
73             DiagnosticGroups.MISSING_PROPERTIES, CheckLevel.WARNING);
74         options.setWarningLevel(
75             DiagnosticGroups.MISPLACED_TYPE_ANNOTATION, CheckLevel.WARNING);
76         options.setWarningLevel(
77             DiagnosticGroups.INVALID_CASTS, CheckLevel.WARNING);
78         options.setWarningLevel(DiagnosticGroups.LINT_CHECKS, CheckLevel.WARNING);
79         options.setCodingConvention(getCodingConvention());
80         return options;
81       }
82     
83       protected CodingConvention getCodingConvention() {
84         return new GoogleCodingConvention();
85       }
86     
87       protected void checkReportedWarningsHelper(String[] expected) {
88         JSError[] warnings = compiler.getWarnings();
89         for (String element : expected) {
90           if (element != null) {
91             assertThat(warnings.length).named("Number of warnings").isGreaterThan(0);
92             assertThat(warnings[0].description).isEqualTo(element);
93             warnings =
94                 Arrays.asList(warnings)
95                     .subList(1, warnings.length)
96                     .toArray(new JSError[warnings.length - 1]);
97           }
98         }
99         if (warnings.length > 0) {
100           fail("unexpected warnings(s):\n" + LINE_JOINER.join(warnings));
101         }
102       }
103     
104       @Override
105       protected void setUp() {
106         compiler = new Compiler();
107         compiler.initOptions(getOptions());
108         registry = compiler.getTypeRegistry();
109         initTypes();
110       }
111     }
