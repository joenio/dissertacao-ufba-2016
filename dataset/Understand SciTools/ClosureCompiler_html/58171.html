16     package com.google.javascript.jscomp;

18     import static com.google.common.truth.Truth.assertThat;
19     
20     import com.google.common.base.Predicates;
21     import com.google.javascript.rhino.Node;
22     
23     public final class PolymerClassDefinitionTest extends CompilerTypeTestCase {
24     
25       private Node polymerCall;
26     
27       @Override
28       protected void setUp() {
29         super.setUp();
30         polymerCall = null;
31       }
32     
33       // TODO(jlklein): Add more complex test cases and verify behaviors and descriptors.
34     
35       public void testSimpleBehavior() {
36         PolymerClassDefinition def = parseAndExtractClassDef(
37             LINE_JOINER.join(
38                 "/** @polymerBehavior */",
39                 "var FunBehavior = {",
40                 "  properties: {",
41                 "    /** @type {boolean} */",
42                 "    isFun: {",
43                 "      type: Boolean,",
44                 "      value: true,",
45                 "    }",
46                 "  },",
47                 "  listeners: {",
48                 "    click: 'doSomethingFun',",
49                 "  },",
50                 "  /** @type {string} */",
51                 "  foo: 'hooray',",
52                 "",
53                 "  /** @param {string} funAmount */",
54                 "  doSomethingFun: function(funAmount) { alert('Something ' + funAmount + ' fun!'); },",
55                 "  /** @override */",
56                 "  created: function() {}",
57                 "};",
58                 "var A = Polymer({",
59                 "  is: 'x-element',",
60                 "  properties: {",
61                 "    pets: {",
62                 "      type: Array,",
63                 "      notify: true,",
64                 "    },",
65                 "    name: String,",
66                 "  },",
67                 "  behaviors: [ FunBehavior ],",
68                 "});"));
69     
70         assertNotNull(def);
71         assertTrue(def.target.isName());
72         assertEquals("A", def.target.getString());
73         assertNull(def.nativeBaseElement);
74         assertThat(def.behaviors).hasSize(1);
75         assertThat(def.props).hasSize(3);
76       }
77     
78       private PolymerClassDefinition parseAndExtractClassDef(String code) {
79         Node rootNode = compiler.parseTestCode(code);
80         GlobalNamespace globalNamespace =  new GlobalNamespace(compiler, rootNode);
81     
82         NodeUtil.visitPostOrder(rootNode, new NodeUtil.Visitor() {
83           @Override
84           public void visit(Node node) {
85             if (PolymerPass.isPolymerCall(node)) {
86               polymerCall = node;
87             }
88           }
89         }, Predicates.<Node>alwaysTrue());
90     
91         assertNotNull(polymerCall);
92         return PolymerClassDefinition.extractFromCallNode(polymerCall, compiler, globalNamespace);
93       }
94     }
