38     /**
39      * A reverse abstract interpreter (RAI) for specific closure patterns such as
40      * {@code goog.isDef}.
41      *
42      */
43     public final class ClosureReverseAbstractInterpreter
44         extends ChainableReverseAbstractInterpreter {
45     
46       /**
47        * For when {@code goog.isObject} returns true. This includes functions, but
48        * not {@code null}.
49        */
50       private final Visitor<JSType> restrictToObjectVisitor =
51           new RestrictByTrueTypeOfResultVisitor() {
52             @Override
53             protected JSType caseTopType(JSType topType) {
54               return getNativeType(NO_OBJECT_TYPE);
55             }
56     
57             @Override
58             public JSType caseObjectType(ObjectType type) {
59               return type;
60             }
61     
62             @Override
63             public JSType caseFunctionType(FunctionType type) {
64               return type;
65             }
66           };
67     
68       /**
69        * For when {@code goog.isObject} returns false.
70        */
71       private final Visitor<JSType> restrictToNotObjectVisitor =
72           new RestrictByFalseTypeOfResultVisitor() {
73     
74             @Override
75             public JSType caseAllType() {
76               return typeRegistry.createUnionType(
77                   getNativeType(NUMBER_STRING_BOOLEAN), getNativeType(NULL_VOID));
78             }
79     
80             @Override
81             public JSType caseObjectType(ObjectType type) {
82               return null;
83             }
84     
85             @Override
86             public JSType caseFunctionType(FunctionType type) {
87               return null;
88             }
89           };

91       /** Functions used to restrict types. */
92       private Map<String, Function<TypeRestriction, JSType>> restricters;
93     
94       public ClosureReverseAbstractInterpreter(final JSTypeRegistry typeRegistry) {
95         super(typeRegistry);
96         this.restricters =
97           new ImmutableMap.Builder<String, Function<TypeRestriction, JSType>>()
98           .put("isDef", new Function<TypeRestriction, JSType>() {
99             @Override
100             public JSType apply(TypeRestriction p) {
101               if (p.outcome) {
102                 return getRestrictedWithoutUndefined(p.type);
103               } else {
104                 return  p.type != null ?
105                     getNativeType(VOID_TYPE).getGreatestSubtype(p.type) : null;
106               }
107              }
108           })
109           .put("isNull", new Function<TypeRestriction, JSType>() {
110             @Override
111             public JSType apply(TypeRestriction p) {
112               if (p.outcome) {
113                 return p.type != null ?
114                     getNativeType(NULL_TYPE).getGreatestSubtype(p.type) : null;
115               } else {
116                 return getRestrictedWithoutNull(p.type);
117               }
118             }
119           })
120           .put("isDefAndNotNull", new Function<TypeRestriction, JSType>() {
121             @Override
122             public JSType apply(TypeRestriction p) {
123               if (p.outcome) {
124                 return getRestrictedWithoutUndefined(
125                     getRestrictedWithoutNull(p.type));
126               } else {
127                 return p.type != null ?
128                     getNativeType(NULL_VOID).getGreatestSubtype(p.type) : null;
129               }
130             }
131           })
132           .put("isString", new Function<TypeRestriction, JSType>() {
133             @Override
134             public JSType apply(TypeRestriction p) {
135               return getRestrictedByTypeOfResult(p.type, "string", p.outcome);
136             }
137           })
138           .put("isBoolean", new Function<TypeRestriction, JSType>() {
139             @Override
140             public JSType apply(TypeRestriction p) {
141               return getRestrictedByTypeOfResult(p.type, "boolean", p.outcome);
142             }
143           })
144           .put("isNumber", new Function<TypeRestriction, JSType>() {
145             @Override
146             public JSType apply(TypeRestriction p) {
147               return getRestrictedByTypeOfResult(p.type, "number", p.outcome);
148             }
149           })
150           .put("isFunction", new Function<TypeRestriction, JSType>() {
151             @Override
152             public JSType apply(TypeRestriction p) {
153               return getRestrictedByTypeOfResult(p.type, "function", p.outcome);
154             }
155           })
156           .put("isArray", new Function<TypeRestriction, JSType>() {
157             @Override
158             public JSType apply(TypeRestriction p) {
159               if (p.type == null) {
160                 return p.outcome ? getNativeType(ARRAY_TYPE) : null;
161               }
162     
163               Visitor<JSType> visitor = p.outcome ? restrictToArrayVisitor :
164                   restrictToNotArrayVisitor;
165               return p.type.visit(visitor);
166             }
167           })
168           .put("isObject", new Function<TypeRestriction, JSType>() {
169             @Override
170             public JSType apply(TypeRestriction p) {
171               if (p.type == null) {
172                 return p.outcome ? getNativeType(OBJECT_TYPE) : null;
173               }
174     
175               Visitor<JSType> visitor = p.outcome ? restrictToObjectVisitor :
176                   restrictToNotObjectVisitor;
177               return p.type.visit(visitor);
178             }
179           })
180           .build();
181       }

183       @Override
184       public FlowScope getPreciserScopeKnowingConditionOutcome(Node condition,
185           FlowScope blindScope, boolean outcome) {
186         if (condition.isCall() && condition.getChildCount() == 2) {
187           Node callee = condition.getFirstChild();
188           Node param = condition.getLastChild();
189           if (callee.isGetProp() && param.isQualifiedName()) {
190             JSType paramType =  getTypeIfRefinable(param, blindScope);
191             Node left = callee.getFirstChild();
192             Node right = callee.getLastChild();
193             if (left.isName() && "goog".equals(left.getString()) &&
194                 right.isString()) {
195               Function<TypeRestriction, JSType> restricter =
196                   restricters.get(right.getString());
197               if (restricter != null) {
198                 return restrictParameter(param, paramType, blindScope, restricter,
199                     outcome);
200               }
201             }
202           }
203         }
204         return nextPreciserScopeKnowingConditionOutcome(
205             condition, blindScope, outcome);
206       }
207     
208       private FlowScope restrictParameter(Node parameter, JSType type,
209           FlowScope blindScope, Function<TypeRestriction, JSType> restriction,
210           boolean outcome) {
211         // restricting
212         type = restriction.apply(new TypeRestriction(type, outcome));
213     
214         // changing the scope
215         if (type != null) {
216           FlowScope informed = blindScope.createChildFlowScope();
217           declareNameInScope(informed, parameter, type);
218           return informed;
219         } else {
220           return blindScope;
221         }
222       }
223     
224       private static class TypeRestriction {
225         private final JSType type;
226         private final boolean outcome;
227     
228         private TypeRestriction(JSType type, boolean outcome) {
229           this.type = type;
230           this.outcome = outcome;
231         }
232       }
