17     package com.google.javascript.jscomp;
18     
19     import com.google.javascript.jscomp.ExpressionDecomposer.DecompositionType;
20     import com.google.javascript.rhino.IR;
21     import com.google.javascript.rhino.Node;
22     import com.google.javascript.rhino.Token;
23     
24     import java.util.HashSet;
25     import java.util.Set;
26     
27     /**
28      * Extracts ES6 classes defined in function calls to local constants.
29      * <p>
30      * Example:
31      * Before: <code>foo(class { constructor() {} });</code>
32      * After:
33      * <code>
34      *   const $jscomp$classdecl$var0 = class { constructor() {} };
35      *   foo($jscomp$classdecl$var0);
36      * </code>
37      * <p>
38      * This must be done before {@link Es6ToEs3Converter}, because that pass only handles classes
39      * that are declarations or simple assignments.
40      * @see Es6ToEs3Converter#visitClass(Node, Node)
41      */
42     public final class Es6ExtractClasses
43         extends NodeTraversal.AbstractPostOrderCallback implements HotSwapCompilerPass {
44     
45       static final String CLASS_DECL_VAR = "$classdecl$var";
46     
47       private final AbstractCompiler compiler;
48       private final ExpressionDecomposer expressionDecomposer;
49       private int classDeclVarCounter = 0;
50     
51       Es6ExtractClasses(AbstractCompiler compiler) {
52         this.compiler = compiler;
53         Set<String> consts = new HashSet<>();
54         this.expressionDecomposer = new ExpressionDecomposer(
55             compiler,
56             compiler.getUniqueNameIdSupplier(),
57             consts,
58             Scope.createGlobalScope(new Node(Token.SCRIPT)));
59       }
60     
61       @Override
62       public void process(Node externs, Node root) {
63         NodeTraversal.traverseRootsEs6(compiler, this, externs, root);
64       }
65     
66       @Override
67       public void hotSwapScript(Node scriptRoot, Node originalRoot) {
68         NodeTraversal.traverseEs6(compiler, scriptRoot, this);
69       }
70     
71       @Override
72       public void visit(NodeTraversal t, Node n, Node parent) {
73         if (n.isClass() && shouldExtractClass(n, parent)) {
74           extractClass(n, parent);
75         }
76       }
77     
78       private boolean shouldExtractClass(Node classNode, Node parent) {
79         if (NodeUtil.isClassDeclaration(classNode)
80             || parent.isName()
81             || (parent.isAssign() && parent.getParent().isExprResult())) {
82           // No need to extract. Handled directly by Es6ToEs3Converter.ClassDeclarationMetadata#create.
83           return false;
84         }
85         // Don't extract the class if it's not safe to do so. For example,
86         // var c = maybeTrue() && class extends someSideEffect() {};
87         // TODO(brndn): it is possible to be less conservative. If the classNode is DECOMPOSABLE,
88         // we could use the expression decomposer to move it out of the way.
89         return expressionDecomposer.canExposeExpression(classNode) == DecompositionType.MOVABLE;
90       }
91     
92       private void extractClass(Node classNode, Node parent) {
93         String name = ES6ModuleLoader.toJSIdentifier(
94             ES6ModuleLoader.createUri(classNode.getStaticSourceFile().getName()))
95             + CLASS_DECL_VAR
96             + (classDeclVarCounter++);
97         Node statement = NodeUtil.getEnclosingStatement(parent);
98         parent.replaceChild(classNode, IR.name(name));
99         Node classDeclaration = IR.constNode(IR.name(name), classNode)
100             .useSourceInfoIfMissingFromForTree(classNode);
101         statement.getParent().addChildBefore(classDeclaration, statement);
102         compiler.reportCodeChange();
103       }
104     }
