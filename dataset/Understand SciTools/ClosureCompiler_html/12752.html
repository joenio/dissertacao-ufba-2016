17     package com.google.javascript.jscomp.newtypes;
18     
19     import java.util.AbstractMap;
20     
21     /** A wrapper around a naive persistent map. */
22     abstract class PersistentMap<K, V> extends AbstractMap<K, V> {
23       private static final PersistentMap EMPTY = NaivePersistentMap.create();
24     
25       public abstract PersistentMap<K, V> with(K key, V value);
26     
27       public abstract PersistentMap<K, V> without(K key);
28     
29       @SuppressWarnings("unchecked")
30       public static <K, V> PersistentMap<K, V> create() {
31         return EMPTY;
32       }
33     
34       public static <K, V> PersistentMap<K, V> of(K key, V value) {
35         return PersistentMap.<K, V>create().with(key, value);
36       }
37     }
