17     package com.google.javascript.refactoring;
18     
19     import com.google.common.collect.ArrayListMultimap;
20     import com.google.common.collect.ListMultimap;
21     import com.google.javascript.jscomp.AbstractCompiler;
22     import com.google.javascript.jscomp.BasicErrorManager;
23     import com.google.javascript.jscomp.CheckLevel;
24     import com.google.javascript.jscomp.JSError;
25     
26     import java.util.Collection;
27     import java.util.List;
28     
29     /**
30      * <p>An error manager that finds a SuggestedFix for all errors if possible.
31      */
32     public class FixingErrorManager extends BasicErrorManager {
33       private AbstractCompiler compiler;
34       private final ListMultimap<JSError, SuggestedFix> fixes = ArrayListMultimap.create();
35     
36       public FixingErrorManager() {}
37     
38       public void setCompiler(AbstractCompiler compiler) {
39         this.compiler = compiler;
40       }
41     
42       @Override
43       public void report(CheckLevel level, JSError error) {
44         super.report(level, error);
45         fixes.putAll(error, ErrorToFixMapper.getFixesForJsError(error, compiler));
46       }
47     
48       public List<SuggestedFix> getFixesForJsError(JSError error) {
49         return fixes.get(error);
50       }
51     
52       public Collection<SuggestedFix> getAllFixes() {
53         return fixes.values();
54       }
55     
56       @Override
57       public void printSummary() {
58         // No-op
59       }
60     
61       @Override
62       public void println(CheckLevel level, JSError error) {
63         // No-op
64       }
65     }
