19     import com.google.common.collect.ImmutableList;
20     import com.google.common.collect.Iterables;

22     import junit.framework.TestCase;

24     import java.util.Arrays;
25     import java.util.List;

27     /**
28      * Tests for error message filtering.
29      * @author nicksantos@google.com (Nick Santos)
30      */
31     public final class RhinoErrorReporterTest extends TestCase {
32     
33       private boolean reportEs3Props;
34       private boolean reportLintWarnings;
35     
36       @Override
37       protected void setUp() throws Exception {
38         reportEs3Props = true;
39         reportLintWarnings = true;
40         super.setUp();
41       }
42     
43       public void testTrailingComma() throws Exception {
44         String message =
45             "Parse error. IE8 (and below) will parse trailing commas in " +
46             "array and object literals incorrectly. " +
47             "If you are targeting newer versions of JS, " +
48             "set the appropriate language_in option.";
49         assertError(
50             "var x = [1,];",
51             RhinoErrorReporter.TRAILING_COMMA,
52             message);
53         JSError error = assertError(
54             "var x = {\n" +
55             "    1: 2,\n" +
56             "};",
57             RhinoErrorReporter.TRAILING_COMMA,
58             message);
59     
60         assertEquals(2, error.getLineNumber());
61         assertEquals(8, error.getCharno());
62       }
63     
64       public void testInvalidEs3Prop() throws Exception {
65         reportEs3Props = false;
66     
67         assertNoWarningOrError("var x = y.function;");
68     
69         reportEs3Props = true;
70     
71         String message =
72             "Keywords and reserved words are not allowed as unquoted property " +
73             "names in older versions of JavaScript. " +
74             "If you are targeting newer versions of JavaScript, " +
75             "set the appropriate language_in option.";
76         JSError error = assertWarning(
77             "var x = y.function;",
78             RhinoErrorReporter.INVALID_ES3_PROP_NAME,
79             message);
80     
81         assertEquals(1, error.getLineNumber());
82         assertEquals(10, error.getCharno());
83       }
84     
85     
86       public void testMissingTypeWarnings() throws Exception {
87         reportLintWarnings = false;
88     
89         assertNoWarningOrError("/** @return */ function f() {}");
90     
91         reportLintWarnings = true;
92     
93         String message =
94             "Missing type declaration.";
95         JSError error = assertWarning(
96             "/** @return */ function f() {}",
97             RhinoErrorReporter.JSDOC_MISSING_TYPE_WARNING,
98             message);
99     
100         assertEquals(1, error.getLineNumber());
101         assertEquals(4, error.getCharno());
102       }
103     
104       /**
105        * Verifies that the compiler emits an error for the given code.
106        */
107       private void assertNoWarningOrError(String code) {
108         Compiler compiler = parseCode(code);
109         assertEquals("Expected error", 0, compiler.getErrorCount());
110         assertEquals("Expected warning", 0, compiler.getErrorCount());
111       }
112     
113       /**
114        * Verifies that the compiler emits an error for the given code.
115        */
116       private JSError assertError(
117           String code, DiagnosticType type, String description) {
118         Compiler compiler = parseCode(code);
119         assertEquals("Expected error", 1, compiler.getErrorCount());
120     
121         JSError error =
122             Iterables.getOnlyElement(Arrays.asList(compiler.getErrors()));
123         assertEquals(type, error.getType());
124         assertEquals(description, error.description);
125         return error;
126       }
127     
128       /**
129        * Verifies that the compiler emits an error for the given code.
130        */
131       private JSError assertWarning(
132           String code, DiagnosticType type, String description) {
133         Compiler compiler = parseCode(code);
134         assertEquals("Expected warning", 1, compiler.getWarningCount());
135     
136         JSError error =
137             Iterables.getOnlyElement(Arrays.asList(compiler.getWarnings()));
138         assertEquals(type, error.getType());
139         assertEquals(description, error.description);
140         return error;
141       }
142     
143       private Compiler parseCode(String code) {
144         Compiler compiler = new Compiler();
145         CompilerOptions options = new CompilerOptions();
146     
147         if (!reportEs3Props) {
148           options.setWarningLevel(
149               DiagnosticGroups.ES3,
150               CheckLevel.OFF);
151         }
152     
153         if (!reportLintWarnings) {
154           options.setWarningLevel(
155               DiagnosticGroups.LINT_CHECKS,
156               CheckLevel.OFF);
157         } else {
158           options.setWarningLevel(
159               DiagnosticGroups.LINT_CHECKS,
160               CheckLevel.WARNING);
161         }
162     
163         List<SourceFile> externs = ImmutableList.of();
164         List<SourceFile> inputs = ImmutableList.of(
165             SourceFile.fromCode("input", code));
166         compiler.init(externs, inputs, options);
167         compiler.parseInputs();
168         return compiler;
169       }
170     }
