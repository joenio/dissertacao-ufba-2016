77     /**
78      * CommandLineRunner translates flags into Java API calls on the Compiler.
79      *
80      * This class may be extended and used to create other Java classes
81      * that behave the same as running the Compiler from the command line. If you
82      * want to run the compiler in-process in Java, you should look at this class
83      * for hints on what API calls to make, but you should not use this class
84      * directly.
85      *
86      * Example:
87      * <pre>
88      * class MyCommandLineRunner extends CommandLineRunner {
89      *   MyCommandLineRunner(String[] args) {
90      *     super(args);
91      *   }
92      *
93      *   {@code @Override} protected CompilerOptions createOptions() {
94      *     CompilerOptions options = super.createOptions();
95      *     addMyCrazyCompilerPassThatOutputsAnExtraFile(options);
96      *     return options;
97      *   }
98      *
99      *   public static void main(String[] args) {
100      *     MyCommandLineRunner runner = new MyCommandLineRunner(args);
101      *     if (runner.shouldRunCompiler()) {
102      *       runner.run();
103      *     }
104      *     if (runner.hasErrors()) {
105      *       System.exit(-1);
106      *     }
107      *   }
108      * }
109      * </pre>
110      *
111      * This class is totally not thread-safe.
112      *
113      * @author bolinfest@google.com (Michael Bolin)
114      */

835         private void printStringLineWrapped(String input, OutputStreamWriter outputStream)
836             throws IOException {
837           if (input.length() < maxLineLength) {
838             outputStream.write(input);
839             return;
840           }
841     
842           int endIndex = maxLineLength;
843           String subString = input.substring(0, maxLineLength);
844           Matcher whitespaceMatcher = whitespacePattern.matcher(subString);
845           boolean foundMatch = false;
846           while (whitespaceMatcher.find()) {
847             endIndex = whitespaceMatcher.start();
848             foundMatch = true;
849           }
850           outputStream.write(input.substring(0, endIndex) + "\n");
851           printStringLineWrapped(
852               "    " + input.substring(foundMatch ? endIndex + 1 : endIndex), outputStream);
853         }

864         /**
865          * Users may specify JS inputs via the {@code --js} flag, as well
866          * as via additional arguments to the Closure Compiler. For example, it is
867          * convenient to leverage the additional arguments feature when using the
868          * Closure Compiler in combination with {@code find} and {@code xargs}:
869          * <pre>
870          * find MY_JS_SRC_DIR -name '*.js' \
871          *     | xargs java -jar compiler.jar --dependency_mode=LOOSE
872          * </pre>
873          * The {@code find} command will produce a list of '*.js' source files in
874          * the {@code MY_JS_SRC_DIR} directory while {@code xargs} will convert them
875          * to a single, space-delimited set of arguments that are appended to the
876          * {@code java} command to run the Compiler.
877          * <p>
878          * Note that it is important to use the
879          * {@code --dependency_mode=LOOSE or STRICT} option in this case because the
880          * order produced by {@code find} is unlikely to be sorted correctly with
881          * respect to {@code goog.provide()} and {@code goog.requires()}.
882          */
883         protected List<String> getJsFiles() throws CmdLineException, IOException {
884           List<String> patterns = new ArrayList<>();
885           patterns.addAll(js);
886           patterns.addAll(arguments);
887           List<String> allJsInputs = findJsFiles(patterns);
888           if (!patterns.isEmpty() && allJsInputs.isEmpty()) {
889             throw new CmdLineException(parser, "No inputs matched");
890           }
891           return allJsInputs;
892         }

894         protected List<FlagEntry<JsSourceType>> getMixedJsSources()
895             throws CmdLineException, IOException {
896           List<FlagEntry<JsSourceType>> mixedSources = new ArrayList<>();
897           Set<String> excludes = new HashSet<>();
898           for (FlagEntry<JsSourceType> source : Flags.mixedJsSources) {
899             if (source.getValue().endsWith(".zip")) {
900               mixedSources.add(source);
901             } else if (source.getValue().startsWith("!")) {
902               for (String filename : findJsFiles(
903                   Collections.singletonList(source.getValue().substring(1)))) {
904                 excludes.add(filename);
905                 mixedSources.remove(new FlagEntry<>(JsSourceType.JS, filename));
906               }
907             } else {
908               for (String filename : findJsFiles(Collections.singletonList(source.getValue()), true)) {
909                 if (!excludes.contains(filename)) {
910                   mixedSources.add(new FlagEntry<>(JsSourceType.JS, filename));
911                 }
912               }
913             }
914           }
915           List<String> fromArguments = findJsFiles(arguments);
916           for (String filename : fromArguments) {
917             mixedSources.add(new FlagEntry<>(JsSourceType.JS, filename));
918           }
919           if (!Flags.mixedJsSources.isEmpty() && !arguments.isEmpty() && mixedSources.isEmpty()) {
920             throw new CmdLineException(parser, "No inputs matched");
921           }
922           return mixedSources;
923         }

943             String flagName) throws CmdLineException {
944           ImmutableMap.Builder<String, String> result = new ImmutableMap.Builder<>();
945

947           for (String inputSourceMap : input) {

952             }
953             result.put(parts.get(0), parts.get(1));
954           }

1148       private static List<String> processArgs(String[] args) {
1149         // Args4j has a different format that the old command-line parser.
1150         // So we use some voodoo to get the args into the format that args4j
1151         // expects.
1152         Pattern argPattern = Pattern.compile("(--?[a-zA-Z_]+)=(.*)");
1153         Pattern quotesPattern = Pattern.compile("^['\"](.*)['\"]$");
1154         List<String> processedArgs = new ArrayList<>();
1155     
1156         for (String arg : args) {
1157           Matcher matcher = argPattern.matcher(arg);
1158           if (matcher.matches()) {
1159             processedArgs.add(matcher.group(1));
1160     
1161             String value = matcher.group(2);
1162             Matcher quotesMatcher = quotesPattern.matcher(value);
1163             if (quotesMatcher.matches()) {
1164               processedArgs.add(quotesMatcher.group(1));
1165             } else {
1166               processedArgs.add(value);
1167             }
1168           } else {
1169             processedArgs.add(arg);
1170           }
1171         }
1172     
1173         return processedArgs;
1174       }

1182       private void processFlagFile()
1183                 throws CmdLineException, IOException {
1184         Path flagFile = Paths.get(flags.flagFile);
1185     
1186         BufferedReader buffer =
1187           java.nio.file.Files.newBufferedReader(flagFile, UTF_8);
1188         // Builds the tokens.
1189         StringBuilder builder = new StringBuilder();
1190         // Stores the built tokens.
1191         List<String> tokens = new ArrayList<>();
1192         // Indicates if we are in a "quoted" token.
1193         boolean quoted = false;
1194         // Indicates if the char being processed has been escaped.
1195         boolean escaped = false;
1196         // Indicates whether this is the beginning of the file.
1197         boolean isFirstCharacter = true;
1198     
1199         int c;
1200     
1201         while ((c = buffer.read()) != -1) {
1202     
1203           // Ignoring the BOM.
1204           if (isFirstCharacter) {
1205             isFirstCharacter = false;
1206             if (c == UTF8_BOM_CODE) {
1207               continue;
1208             }
1209           }
1210     
1211           if (c == 32 || c == 9 || c == 10 || c == 13) {
1212             if (quoted) {
1213               builder.append((char) c);
1214             } else if (builder.length() != 0) {
1215               tokens.add(builder.toString());
1216               builder.setLength(0);
1217             }
1218           } else if (c == 34) {
1219             if (escaped) {
1220               if (quoted) {
1221                 builder.setCharAt(builder.length() - 1, (char) c);
1222               } else {
1223                 builder.append((char) c);
1224               }
1225             } else {
1226               quoted = !quoted;
1227             }
1228           } else {
1229             builder.append((char) c);
1230           }
1231     
1232           escaped = c == 92;
1233         }
1234     
1235         buffer.close();
1236     
1237         if (builder.length() != 0) {
1238           tokens.add(builder.toString());
1239         }
1240     
1241         flags.flagFile = "";
1242     
1243         tokens = processArgs(tokens.toArray(new String[0]));
1244     
1245         // Command-line warning levels should override flag file settings,
1246         // which means they should go last.
1247         List<FlagEntry<CheckLevel>> previousGuardLevels = new ArrayList<>(Flags.guardLevels);
1248         List<FlagEntry<JsSourceType>> previousMixedJsSources = new ArrayList<>(Flags.mixedJsSources);
1249         Flags.guardLevels.clear();
1250         Flags.mixedJsSources.clear();
1251         flags.parse(tokens);
1252         Flags.guardLevels.addAll(previousGuardLevels);
1253         Flags.mixedJsSources.addAll(previousMixedJsSources);
1254     
1255         // Currently we are not supporting this (prevent direct/indirect loops)
1256         if (!flags.flagFile.isEmpty()) {
1257           reportError("ERROR - Arguments in the file cannot contain "
1258               + "--flagfile option.");
1259         }
1260       }

1262       private void initConfigFromFlags(String[] args, PrintStream out, PrintStream err) {
1263     
1264         errorStream = err;
1265         List<String> processedArgs = processArgs(args);
1266     
1267         Flags.guardLevels.clear();
1268         Flags.mixedJsSources.clear();
1269     
1270         List<String> jsFiles = null;
1271         List<FlagEntry<JsSourceType>> mixedSources = null;
1272         List<LocationMapping> mappings = null;
1273         ImmutableMap<String, String> sourceMapInputs = null;
1274         try {
1275           flags.parse(processedArgs);
1276     
1277           // For contains --flagfile flag
1278           if (!flags.flagFile.isEmpty()) {
1279             processFlagFile();
1280           }
1281     
1282           jsFiles = flags.getJsFiles();
1283           mixedSources = flags.getMixedJsSources();
1284           mappings = flags.getSourceMapLocationMappings();
1285           sourceMapInputs = flags.getSourceMapInputs();
1286         } catch (CmdLineException e) {
1287           reportError(e.getMessage());
1288         } catch (IOException ioErr) {
1289           reportError("ERROR - " + flags.flagFile + " read error.");
1290         }
1291     
1292         List<ModuleIdentifier> entryPoints = new ArrayList<>();
1293     
1294         if (flags.processCommonJsModules) {
1295           flags.processClosurePrimitives = true;
1296           if (flags.commonJsEntryModule != null) {
1297             if (flags.entryPoints.isEmpty()) {
1298               entryPoints.add(ModuleIdentifier.forFile(flags.commonJsEntryModule));
1299             } else {
1300               reportError("--common_js_entry_module cannot be used with --entry_point.");
1301             }
1302           }
1303         }
1304     
1305         if (flags.outputWrapperFile != null && !flags.outputWrapperFile.isEmpty()) {
1306           flags.outputWrapper = "";

1664       /**
1665        * Returns all the JavaScript files from the set of patterns.
1666        *
1667        * @param patterns A collection of filename patterns.
1668        * @param sortAlphabetically Whether the output filenames should be in alphabetical order.
1669        * @return The list of JS filenames found by expanding the patterns.
1670        */
1671       private static List<String> findJsFiles(Collection<String> patterns, boolean sortAlphabetically)
1672           throws IOException {
1673         // A map from normalized absolute paths to original paths. We need to return original paths to
1674         // support whitelist files that depend on them.
1675         Map<String, String> allJsInputs = sortAlphabetically
1676             ? new TreeMap<String, String>() : new LinkedHashMap<String, String>();
1677         Set<String> excludes = new HashSet<>();
1678         for (String pattern : patterns) {
1679           if (!pattern.contains("*") && !pattern.startsWith("!")) {
1680             File matchedFile = new File(pattern);
1681             if (matchedFile.isDirectory()) {
1682               matchPaths(new File(matchedFile, "**.js").toString(), allJsInputs, excludes);
1683             } else {
1684               Path original = Paths.get(pattern);
1685               String pathStringAbsolute = original.normalize().toAbsolutePath().toString();
1686               if (!excludes.contains(pathStringAbsolute)) {
1687                 allJsInputs.put(pathStringAbsolute, original.toString());
1688               }
1689             }
1690           } else {
1691             matchPaths(pattern, allJsInputs, excludes);
1692           }
1693         }
1694     
1695         return new ArrayList<>(allJsInputs.values());
1696       }
1697     
1698       private static void matchPaths(String pattern, final Map<String, String> allJsInputs,
1699           final Set<String> excludes) throws IOException {
1700         FileSystem fs = FileSystems.getDefault();
1701         final boolean remove = pattern.indexOf('!') == 0;
1702         if (remove) {
1703           pattern = pattern.substring(1);
1704         }
1705     
1706         String separator = File.separator.equals("\\") ? "\\\\" : File.separator;
1707     
1708         // Split the pattern into two pieces: the globbing part
1709         // and the non-globbing prefix.
1710         List<String> patternParts = Splitter.on(File.separator).splitToList(pattern);
1711         String prefix = ".";
1712         for (int i = 0; i < patternParts.size(); i++) {
1713           if (patternParts.get(i).contains("*")) {
1714             if (i > 0) {
1715               prefix = Joiner.on(separator).join(patternParts.subList(0, i));
1716               pattern = Joiner.on(separator).join(patternParts.subList(i, patternParts.size()));
1717             }
1718             break;
1719           }
1720         }
1721     
1722         final PathMatcher matcher = fs.getPathMatcher("glob:" + prefix + separator + pattern);
1723         java.nio.file.Files.walkFileTree(
1724             fs.getPath(prefix), new SimpleFileVisitor<Path>() {
1725               @Override
1726               public FileVisitResult visitFile(Path p, BasicFileAttributes attrs) {
1727                 if (matcher.matches(p) || matcher.matches(p.normalize())) {
1728                   String pathStringAbsolute = p.normalize().toAbsolutePath().toString();
1729                   if (remove) {
1730                     excludes.add(pathStringAbsolute);
1731                     allJsInputs.remove(pathStringAbsolute);
1732                   } else if (!excludes.contains(pathStringAbsolute)) {
1733                     allJsInputs.put(pathStringAbsolute, p.toString());
1734                   }
1735                 }
1736                 return FileVisitResult.CONTINUE;
1737               }
1738     
1739               @Override
1740               public FileVisitResult visitFileFailed(Path file, IOException e) {
1741                 return FileVisitResult.SKIP_SUBTREE;
1742               }
1743             });
1744       }
