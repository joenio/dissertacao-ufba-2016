17     package com.google.javascript.jscomp;
18     
19     import com.google.common.collect.ImmutableMap;
20     import com.google.javascript.rhino.Node;
21     
22     import java.util.Collections;
23     import java.util.HashMap;
24     import java.util.Map;
25     import java.util.Set;
26     import java.util.logging.Logger;
27     
28     /**
29      * Gives anonymous function names that are optimized to be small and provides a
30      * mapping back to the original names. This makes it way easier to debug because
31      * debuggers and stack traces use the function names. So if you have
32      *
33      * goog.string.htmlEscape = function(str) {
34      * }
35      *
36      * It will become
37      *
38      * goog.string.htmlEscape = function $qv(str) {
39      * }
40      *
41      * And there will be mapping from $qv to goog.string.htmlEscape
42      *
43      */
44     class NameAnonymousFunctionsMapped implements CompilerPass {
45     
46       private static Logger logger = Logger.getLogger(
47           NameAnonymousFunctionsMapped.class.getName());
48     
49       static final char PREFIX = '$';
50       static final String PREFIX_STRING = "$";
51     
52       private final AbstractCompiler compiler;
53       private final NameGenerator nameGenerator;
54       private final VariableMap previousMap;
55       private final Map<String, String> renameMap;
56     
57       private int namedCount = 0;
58       private int bytesUsed = 0;
59     
60       NameAnonymousFunctionsMapped(
61           AbstractCompiler compiler, VariableMap previousMap) {
62         this.compiler = compiler;
63         Set<String> reserved =
64             previousMap != null ?
65                 previousMap.getNewNameToOriginalNameMap().keySet() :
66                 Collections.<String>emptySet();
67         this.nameGenerator = new DefaultNameGenerator(
68             reserved, PREFIX_STRING, null);
69         this.previousMap = previousMap;
70         this.renameMap = new HashMap<>();
71       }
72     
73       @Override
74       public void process(Node externs, Node root) {
75         AnonymousFunctionNamingCallback namingCallback =
76             new AnonymousFunctionNamingCallback(new MappedFunctionNamer());
77         NodeTraversal.traverseEs6(compiler, root, namingCallback);
78         logger.fine("Named " + namedCount + " anon functions using " +
79             bytesUsed + " bytes");
80         if (namedCount > 0) {
81           compiler.reportCodeChange();
82         }
83       }
84     
85       /**
86        * Names anonymous functions. The function names don't have to be globally
87        * unique or even locally unique. We make them somewhat unique because of a
88        * bug in IE (and there may be other bugs we haven't found). See unit test for
89        * more info.
90        */
91       private class MappedFunctionNamer
92           implements AnonymousFunctionNamingCallback.FunctionNamer {
93         static final char DELIMITER = '.';
94     
95         @Override
96         public final String getName(Node node) {
97           switch (node.getType()) {
98             case NAME:
99             case STRING:
100             case STRING_KEY:
101               return node.getString();
102             default:
103               return compiler.toSource(node);
104           }
105         }
106     
107         @Override
108         public final void setFunctionName(String name, Node fnNode) {
109           Node fnNameNode = fnNode.getFirstChild();
110           String newName = getAlternateName(name);
111           fnNameNode.setString(newName);
112           namedCount++;
113           bytesUsed += newName.length();
114         }
115     
116         String getAlternateName(String name) {
117           String newName = renameMap.get(name);
118           if (newName == null) {
119             // Use the previously used name, if possible.
120             if (previousMap != null) {
121               newName = previousMap.lookupNewName(name);
122             }
123             if (newName == null) {
124               // otherwise generate a new name.
125               newName = nameGenerator.generateNextName();
126             }
127             renameMap.put(name, newName);
128           }
129           return newName;
130         }
131     
132         @Override
133         public final String getCombinedName(String lhs, String rhs) {
134           return lhs + DELIMITER + rhs;
135         }
136       }
137     
138       /**
139        * Gets the function renaming map (the "answer key").
140        *
141        * @return A mapping from original names to new names
142        */
143       VariableMap getFunctionMap() {
144         return new VariableMap(ImmutableMap.copyOf(renameMap));
145       }
146     }
