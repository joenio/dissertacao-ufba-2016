17     package com.google.javascript.jscomp;
18     
19     /**
20      * Convert the warnings level to an Options object.
21      *
22      */
23     public enum WarningLevel {
24       QUIET,
25     
26       DEFAULT,
27     
28       VERBOSE;
29     
30       public void setOptionsForWarningLevel(CompilerOptions options) {
31         switch (this) {
32           case QUIET:
33             silenceAllWarnings(options);
34             break;
35           case DEFAULT:
36             addDefaultWarnings(options);
37             break;
38           case VERBOSE:
39             addVerboseWarnings(options);
40             break;
41           default:
42             throw new RuntimeException("Unknown warning level.");
43         }
44       }
45     
46       /**
47        * Silence all non-essential warnings.
48        */
49       private static void silenceAllWarnings(CompilerOptions options) {
50         // Just use a ShowByPath warnings guard, so that we don't have
51         // to maintain a separate class of warnings guards for silencing warnings.
52         options.addWarningsGuard(
53             new ShowByPathWarningsGuard(
54                 "the_longest_path_that_cannot_be_expressed_as_a_string"));
55     
56         // Allow passes that aren't going to report anything to be skipped.
57     
58         options.setWarningLevel(DiagnosticGroups.MISSING_PROVIDE, CheckLevel.OFF);
59         options.setCheckMissingGetCssNameLevel(CheckLevel.OFF);
60         options.setCheckTypes(false);
61         options.setWarningLevel(DiagnosticGroups.CHECK_TYPES, CheckLevel.OFF);
62         options.setWarningLevel(DiagnosticGroups.CHECK_USELESS_CODE, CheckLevel.OFF);
63         options.setWarningLevel(DiagnosticGroups.MISSING_RETURN, CheckLevel.OFF);
64         options.setWarningLevel(DiagnosticGroups.ACCESS_CONTROLS, CheckLevel.OFF);
65         options.setWarningLevel(DiagnosticGroups.CONST, CheckLevel.OFF);
66         options.setWarningLevel(DiagnosticGroups.CONSTANT_PROPERTY, CheckLevel.OFF);
67         options.setCheckGlobalNamesLevel(CheckLevel.OFF);
68         options.setCheckSuspiciousCode(false);
69         options.setCheckGlobalThisLevel(CheckLevel.OFF);
70         options.setWarningLevel(DiagnosticGroups.GLOBAL_THIS, CheckLevel.OFF);
71         options.setWarningLevel(DiagnosticGroups.ES5_STRICT, CheckLevel.OFF);
72     
73         // Allows annotations that are not standard.
74         options.setWarningLevel(DiagnosticGroups.NON_STANDARD_JSDOC,
75             CheckLevel.OFF);
76       }
77     
78       /**
79        * Add the default checking pass to the compilation options.
80        * @param options The CompilerOptions object to set the options on.
81        */
82       private static void addDefaultWarnings(CompilerOptions options) {
83         options.setCheckSuspiciousCode(true);
84     
85         // Allows annotations that are not standard.
86         options.setWarningLevel(DiagnosticGroups.NON_STANDARD_JSDOC,
87             CheckLevel.OFF);
88       }
89     
90       /**
91        * Add all the check pass that are possibly relevant to a non-googler.
92        * @param options The CompilerOptions object to set the options on.
93        */
94       private static void addVerboseWarnings(CompilerOptions options) {
95         addDefaultWarnings(options);
96     
97         // checkSuspiciousCode needs to be enabled for CheckGlobalThis to get run.
98         options.setCheckSuspiciousCode(true);
99         options.setCheckGlobalThisLevel(CheckLevel.WARNING);
100         options.setCheckSymbols(true);
101     
102         // checkTypes has the side-effect of asserting that the
103         // correct number of arguments are passed to a function.
104         // Because the CodingConvention used with the web service does not provide a
105         // way for optional arguments to be specified, these warnings may result in
106         // false positives.
107         options.setCheckTypes(true);
108         options.setCheckGlobalNamesLevel(CheckLevel.WARNING);
109         options.setWarningLevel(DiagnosticGroups.MISSING_PROPERTIES, CheckLevel.WARNING);
110         options.setWarningLevel(
111             DiagnosticGroups.DEPRECATED, CheckLevel.WARNING);
112         options.setWarningLevel(
113             DiagnosticGroups.ES5_STRICT, CheckLevel.WARNING);
114         options.setWarningLevel(
115             DiagnosticGroups.VISIBILITY, CheckLevel.WARNING);
116         options.setWarningLevel(
117             DiagnosticGroups.CONST, CheckLevel.WARNING);
118         options.setWarningLevel(
119             DiagnosticGroups.CHECK_REGEXP, CheckLevel.WARNING);
120         options.setWarningLevel(
121             DiagnosticGroups.STRICT_MODULE_DEP_CHECK, CheckLevel.WARNING);
122         options.setWarningLevel(
123             DiagnosticGroups.MISSING_RETURN, CheckLevel.WARNING);
124     
125         // Kindly tell the user that they have JsDocs that we don't understand.
126         options.setWarningLevel(DiagnosticGroups.NON_STANDARD_JSDOC,
127             CheckLevel.WARNING);
128     
129         // Transitional.
130         options.enforceAccessControlCodingConventions = true;
131       }
132     }
