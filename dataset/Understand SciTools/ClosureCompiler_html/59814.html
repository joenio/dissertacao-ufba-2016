17     package com.google.javascript.jscomp;
18     
19     /**
20      * An interface for providing alternative values for user-visible messages in
21      * JavaScript code.
22      *
23      */
24     public interface MessageBundle {
25     
26       /**
27        * Gets the message ID generator to use to compute message IDs for this
28        * type of bundle.
29        * @return idGenerator instance or null if we do not want to use any custom
30        *     id generation. In case if idGenerator is null caller should decide how
31        *     to create id by itself. In the most cases using the message key is
32        *     enough.
33        */
34       public JsMessage.IdGenerator idGenerator();
35     
36       /**
37        * Gets a message replacement.
38        *
39        * @param id the id of the message being replaced; the key is message ID
40        *     generated by {@link JsMessage.IdGenerator}
41        * @return the message replacement, which may be null.
42        */
43       public JsMessage getMessage(String id);
44     
45       /**
46        * Returns an iterable over the keys that this object has replacements for.
47        * @return all messages from this bundle.
48        */
49       public Iterable<JsMessage> getAllMessages();
50     }
