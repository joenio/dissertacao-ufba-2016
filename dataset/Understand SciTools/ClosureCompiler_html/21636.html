17     package com.google.javascript.jscomp;
18     
19     /** Unit tests for {@link ImplicitNullabilityCheck}. */
20     public final class ImplicitNullabilityCheckTest extends TypeICompilerTestCase {
21     
22       @Override
23       public CompilerPass getProcessor(Compiler compiler) {
24         return new ImplicitNullabilityCheck(compiler);
25       }
26     
27       @Override
28       protected int getNumRepetitions() {
29         return 1;
30       }
31     
32       public void testExplicitJsdocDoesntWarn() {
33         noWarning("/** @type {boolean} */ var x;");
34         noWarning("/** @type {null} */ var x;");
35         noWarning("/** @type {!Object} */ var x;");
36         noWarning("/** @type {?Object} */ var x;");
37         noWarning("/** @type {function(new:Object)} */ function f(){}");
38         noWarning("/** @type {function(this:Object)} */ function f(){}");
39         noWarning("/** @typedef {!Object} */ var Obj; var /** Obj */ x;");
40       }
41     
42       public void testExplicitlyNullableUnion() {
43         noWarning("/** @type {(Object|null)} */ var x;");
44         noWarning("/** @type {(Object|number)?} */ var x;");
45         noWarning("/** @type {?(Object|number)} */ var x;");
46         noWarning("/** @type {(Object|?number)} */ var x;");
47         warnImplicitlyNullable("/** @type {(Object|number)} */ var x;");
48       }
49     
50       public void testJsdocPositions() {
51         warnImplicitlyNullable("/** @type {Object} */ var x;");
52         warnImplicitlyNullable("var /** Object */ x;");
53         warnImplicitlyNullable("/** @typedef {Object} */ var x;");
54         warnImplicitlyNullable("/** @param {Object} x */ function f(x){}");
55         warnImplicitlyNullable(
56             "/** @return {Object} */ function f(x){ return {}; }");
57       }
58     
59       public void testNullableTypedef() {
60         // Arguable whether or not this deserves a warning, so leaving
61         // out of NTI for now.
62         testSameOtiOnly(
63             DEFAULT_EXTERNS,
64             "/** @typedef {?number} */ var Num; var /** Num */ x;",
65             ImplicitNullabilityCheck.IMPLICITLY_NULLABLE_JSDOC);
66       }
67     
68       public void testUnkownTypenameDoesntWarn() {
69         // Different warnings in OTI and NTI
70         testSameOtiOnly(
71             DEFAULT_EXTERNS, "/** @type {gibberish} */ var x;", RhinoErrorReporter.TYPE_PARSE_ERROR);
72     
73         testSameNtiOnly(
74             DEFAULT_EXTERNS, "/** @type {gibberish} */ var x;", GlobalTypeInfo.UNRECOGNIZED_TYPE_NAME);
75       }
76     
77       public void testThrowsDoesntWarn() {
78         noWarning("/** @throws {Error} */ function f() {}");
79         noWarning("/** @throws {TypeError}\n * @throws {SyntaxError} */ function f() {}");
80       }
81     
82       public void testUserDefinedClass() {
83         warnImplicitlyNullable(LINE_JOINER.join(
84             "/** @constructor */",
85             "function Foo() {}",
86             "/** @type {Foo} */ var x;"));
87     
88         // TODO(aravindpg): this ought to pass under both, or at any rate NTI.
89         noWarning(LINE_JOINER.join(
90             "function f() {",
91             "  /** @constructor */",
92             "  function Foo() {}",
93             "  /** @type {Foo} */ var x;",
94             "}"));
95       }
96     
97       private void warnImplicitlyNullable(String js) {
98         testSame(DEFAULT_EXTERNS, js, ImplicitNullabilityCheck.IMPLICITLY_NULLABLE_JSDOC);
99       }
100     
101       private void noWarning(String js) {
102         testSame(DEFAULT_EXTERNS, js, null);
103       }
104     }
