19     import com.google.javascript.rhino.ErrorReporter;
20     
21     /**
22      * An error reporter which consumes all calls and performs no actions.
23      *
24      */
25     public abstract class NullErrorReporter  {
26       private NullErrorReporter() {
27       }
28     
29       public void error(String message, String sourceName, int line,
30           int lineOffset) {
31       }
32     
33       public void warning(String message, String sourceName, int line,
34           int lineOffset) {
35       }
36     
37       public static ErrorReporter forOldRhino() {
38         return new OldRhinoNullReporter();
39       }
40     
41       private static class OldRhinoNullReporter extends NullErrorReporter
42           implements ErrorReporter {
43       }
44     }
