17     package com.google.javascript.jscomp;
18     
19     import com.google.javascript.jscomp.ReferenceCollectingCallback.ReferenceCollection;
20     import com.google.javascript.rhino.JSDocInfo;
21     import com.google.javascript.rhino.Node;
22     
23     /**
24      * Attaches the CONST_VAR annotation to any variable that's
25      * 1) Provably well-defined and assigned once in its lifetime.
26      * 2) Annotated 'const'
27      * 3) Declared with the 'const' keyword.
28      * 4) Is constant by naming convention.
29      *
30      * These 3 are considered semantically equivalent. Notice that a variable
31      * in a loop is never considered const.
32      *
33      * Note that criteria (1) is only used for normal code, not externs.
34      *
35      * @author nicholas.j.santos@gmail.com (Nick Santos)
36      */
37     class InferConsts implements CompilerPass {
38       private final AbstractCompiler compiler;
39     
40       InferConsts(AbstractCompiler compiler) {
41         this.compiler = compiler;
42       }
43     
44       @Override
45       public void process(Node externs, Node js) {
46         ReferenceCollectingCallback collector = new ReferenceCollectingCallback(
47             compiler, ReferenceCollectingCallback.DO_NOTHING_BEHAVIOR);
48         NodeTraversal.traverseEs6(compiler, js, collector);
49     
50         for (Var v : collector.getAllSymbols()) {
51           considerVar(v, collector.getReferences(v));
52         }
53     
54         Scope globalExternsScope =
55             SyntacticScopeCreator.makeUntyped(compiler).createScope(externs, null);
56         for (Var v : globalExternsScope.getAllSymbols()) {
57           considerVar(v, null);
58         }
59       }
60     
61       private void considerVar(Var v, ReferenceCollection refCollection) {
62         Node nameNode = v.getNameNode();
63         JSDocInfo docInfo = v.getJSDocInfo();
64         if (docInfo != null && docInfo.isConstant()) {
65           nameNode.putBooleanProp(Node.IS_CONSTANT_VAR, true);
66         } else if (nameNode != null && nameNode.getParent().isConst()) {
67           nameNode.putBooleanProp(Node.IS_CONSTANT_VAR, true);
68         } else if (nameNode != null &&
69             compiler.getCodingConvention().isConstant(nameNode.getString())) {
70           nameNode.putBooleanProp(Node.IS_CONSTANT_VAR, true);
71         } else if (nameNode != null &&
72                    refCollection != null &&
73                    refCollection.isWellDefined() &&
74                    refCollection.isAssignedOnceInLifetime() &&
75                    refCollection.firstReferenceIsAssigningDeclaration()) {
76           nameNode.putBooleanProp(Node.IS_CONSTANT_VAR, true);
77         }
78       }
79     }
