16     package com.google.javascript.jscomp;
17     
18     import com.google.common.collect.ImmutableSet;
19     import com.google.javascript.jscomp.NodeTraversal.AbstractPostOrderCallback;
20     import com.google.javascript.jscomp.regex.RegExpTree;
21     import com.google.javascript.rhino.Node;
22     import com.google.javascript.rhino.Token;
23     
24     import java.util.Set;
25     
26     /**
27      * Look for references to the global RegExp object that would cause
28      * regular expressions to be unoptimizable, and checks that regular expressions
29      * are syntactically valid.
30      *
31      * @author johnlenz@google.com (John Lenz)
32      */
33     class CheckRegExp extends AbstractPostOrderCallback implements CompilerPass {
34     
35       static final DiagnosticType REGEXP_REFERENCE =
36         DiagnosticType.warning("JSC_REGEXP_REFERENCE",
37             "References to the global RegExp object prevents " +
38             "optimization of regular expressions.");
39       static final DiagnosticType MALFORMED_REGEXP = DiagnosticType.warning(
40             "JSC_MALFORMED_REGEXP",
41             "Malformed Regular Expression: {0}");
42     
43       private static final Set<String> REGEXP_PROPERTY_BLACKLIST = ImmutableSet.of(
44           "$1", "$2", "$3", "$4", "$5", "$6", "$7", "$8", "$9",
45           "$_", "$input",
46           // The following would also be blacklisted, but they aren't valid
47           // identifiers, so can't be accessed with the '.' operator anyway.
48           // "$*", "$&", "$+", "$`", "$'",
49           "input", "lastMatch", "lastParen", "leftContext", "rightContext",
50           "global", "ignoreCase", "lastIndex", "multiline", "source");
51     
52       private final AbstractCompiler compiler;
53       private boolean globalRegExpPropertiesUsed = false;
54     
55       public boolean isGlobalRegExpPropertiesUsed() {
56         return globalRegExpPropertiesUsed;
57       }
58     
59       public CheckRegExp(AbstractCompiler compiler) {
60         this.compiler = compiler;
61       }
62     
63       @Override
64       public void process(Node externs, Node root) {
65         NodeTraversal.traverseEs6(compiler, root, this);
66       }
67     
68       @Override
69       public void visit(NodeTraversal t, Node n, Node parent) {
70         if (NodeUtil.isReferenceName(n)) {
71           String name = n.getString();
72           if (name.equals("RegExp") && t.getScope().getVar(name) == null) {
73             Token parentType = parent.getType();
74             boolean first = (n == parent.getFirstChild());
75             if (!((parentType == Token.NEW && first)
76                 || (parentType == Token.CALL && first)
77                 || (parentType == Token.INSTANCEOF && !first)
78                 || parentType == Token.EQ || parentType == Token.NE
79                 || parentType == Token.SHEQ || parentType == Token.SHNE
80                 || parentType == Token.CASE
81                 || (parentType == Token.GETPROP && first
82                 && !REGEXP_PROPERTY_BLACKLIST.contains(
83                 parent.getLastChild().getString())))) {
84               t.report(n, REGEXP_REFERENCE);
85               globalRegExpPropertiesUsed = true;
86             }
87           }
88     
89         // Check the syntax of regular expression patterns.
90         } else if (n.isRegExp()) {
91           String pattern = n.getFirstChild().getString();
92           String flags = n.getChildCount() == 2
93               ? n.getLastChild().getString() : "";
94           try {
95             RegExpTree.parseRegExp(pattern, flags);
96           } catch (IllegalArgumentException | IndexOutOfBoundsException ex) {
97             t.report(n, MALFORMED_REGEXP, ex.getMessage());
98           }
99         }
100       }
101     }
