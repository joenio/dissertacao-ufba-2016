32     /**
33      * Collects information mapping the generated (compiled) source back to
34      * its original source for debugging purposes.
35      *
36      * @see CodeConsumer
37      * @see CodeGenerator
38      * @see CodePrinter
39      *
40      * @author johnlenz@google.com (John Lenz)
41      */
42     public final class SourceMap {
43     
44       /**
45        * An enumeration of available source map formats
46        */
47       public static enum Format {
48          DEFAULT {
49            @Override SourceMap getInstance() {
50              return new SourceMap(
51                SourceMapGeneratorFactory.getInstance(SourceMapFormat.DEFAULT));
52            }
53          },
54          V3 {
55            @Override SourceMap getInstance() {
56              return new SourceMap(
57                SourceMapGeneratorFactory.getInstance(SourceMapFormat.V3));
58             }
59          };
60          abstract SourceMap getInstance();
61       }
62     
63       /**
64        * Source maps can be very large different levels of detail can be specified.
65        */
66       public static enum DetailLevel implements Predicate<Node> {
67         // ALL is best when the fullest details are needed for debugging or for
68         // code-origin analysis.
69         ALL {
70           @Override public boolean apply(Node node) {
71             return true;
72           }
73         },
74         // SYMBOLS is intended to be used for stack trace deobfuscation when full
75         // detail is not needed.
76         SYMBOLS {
77           @Override public boolean apply(Node node) {
78             return node.isCall()
79                 || node.isNew()
80                 || node.isFunction()
81                 || node.isName()
82                 || NodeUtil.isGet(node)
83                 || NodeUtil.isObjectLitKey(node)
84                 || (node.isString() && NodeUtil.isGet(node.getParent()))
85                 || node.isTaggedTemplateLit();
86           }
87         }
88       }
89     
90       /**
91        * A simple pair of path prefixes to the desired "destination" location to use within the
92        * source map.
93        */
94       public static class LocationMapping {
95         final String prefix;
96         final String replacement;
97         public LocationMapping(String prefix, String replacement) {
98           this.prefix = prefix;
99           this.replacement = replacement;
100         }
101         @Override
102         public String toString() {
103           return "(" + prefix + "|" + replacement + ")";
104         }
105       }
106     
107       private final SourceMapGenerator generator;
108       private List<LocationMapping> prefixMappings = Collections.emptyList();
109       private final Map<String, String> sourceLocationFixupCache =
110            new HashMap<>();
111     
112       private SourceMap(SourceMapGenerator generator) {
113         this.generator = generator;
114       }
115     
116       public void addMapping(
117           Node node,
118           FilePosition outputStartPosition,
119           FilePosition outputEndPosition) {
120         String sourceFile = node.getSourceFileName();
121     
122         // If the node does not have an associated source file or
123         // its line number is -1, then the node does not have sufficient
124         // information for a mapping to be useful.
125         if (sourceFile == null || node.getLineno() < 0) {
126           return;
127         }
128     
129         sourceFile = fixupSourceLocation(sourceFile);
130     
131         String originalName = node.getOriginalName();
132     
133         // Rhino source lines are one based but for v3 source maps, we make
134         // them zero based.
135         int lineBaseOffset = 1;
136     
137         generator.addMapping(
138             sourceFile, originalName,
139             new FilePosition(node.getLineno() - lineBaseOffset, node.getCharno()),
140             outputStartPosition, outputEndPosition);
141       }
142     
143       /**
144        * @param sourceFile The source file location to fixup.
145        * @return a remapped source file.
146        */
147       private String fixupSourceLocation(String sourceFile) {
148         if (prefixMappings.isEmpty()) {
149           return sourceFile;
150         }
151     
152         String fixed = sourceLocationFixupCache.get(sourceFile);
153         if (fixed != null) {
154           return fixed;
155         }
156     
157         // Replace the first prefix found with its replacement
158         for (LocationMapping mapping : prefixMappings) {
159           if (sourceFile.startsWith(mapping.prefix)) {
160             fixed = mapping.replacement + sourceFile.substring(
161               mapping.prefix.length());
162             break;
163           }
164         }
165     
166         // If none of the mappings match then use the original file path.
167         if (fixed == null) {
168           fixed = sourceFile;
169         }
170     
171         sourceLocationFixupCache.put(sourceFile, fixed);
172         return fixed;
173       }
174     
175       public void appendTo(Appendable out, String name) throws IOException {
176         generator.appendTo(out, fixupSourceLocation(name));
177       }
178     
179       public void reset() {
180         generator.reset();
181         sourceLocationFixupCache.clear();
182       }
183     
184       public void setStartingPosition(int offsetLine, int offsetIndex) {
185         generator.setStartingPosition(offsetLine, offsetIndex);
186       }
187     
188       public void setWrapperPrefix(String prefix) {
189         generator.setWrapperPrefix(prefix);
190       }
191     
192       public void validate(boolean validate) {
193         generator.validate(validate);
194       }
195     
196       /**
197        * @param sourceMapLocationMappings
198        */
199       public void setPrefixMappings(List<LocationMapping> sourceMapLocationMappings) {
200          this.prefixMappings = sourceMapLocationMappings;
201       }
202     }
