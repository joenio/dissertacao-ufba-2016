16     package com.google.javascript.jscomp;
17     
18     import com.google.common.collect.ImmutableMap;
19     
20     /**
21      * A decoder for strings encoded by the ReplaceStrings JS compiler pass. This class is immutable.
22      *
23      */
24     public final class ReplacedStringsDecoder {
25     
26       // The default place holder for replaced string arguments.
27       // Note: We're not allowing this to be changed here because it's present in
28       // the replacement strings in the js_storage protocol buffer.
29       public static final String ARGUMENT_PLACE_HOLDER = "`";
30     
31       private final ImmutableMap<String, String> originalToNewNameMap;
32     
33       /** A null decoder that does no mapping. */
34       public static final ReplacedStringsDecoder NULL_DECODER =
35           new ReplacedStringsDecoder(VariableMap.fromMap(
36               ImmutableMap.<String, String>of())
37           );
38     
39       public ReplacedStringsDecoder(VariableMap variableMap) {
40         // VariableMap is not an immutable type, so we extract the map instead of directly using it.
41         this.originalToNewNameMap = ImmutableMap.copyOf(variableMap.getOriginalNameToNewNameMap());
42       }
43     
44       /**
45        * Decodes an encoded string from the JS Compiler ReplaceStrings pass.
46        *
47        * <p>An original string with args might look like this:
48        * <pre>  Error('Some ' + arg1 + ' error ' + arg2 + ' message.');</pre>
49        * Which gets replaced with:
50        * <pre>  Error('key' + '`' + arg1 + '`' + arg2);</pre>
51        * Where ` is the argument place holder. The replacement mapping would be:
52        * <pre>  key â†’ 'Some ` error ` message.'</pre>
53        * Where key is some arbitrary replacement string. An encoded string,
54        * with args, from the client will look like:
55        * <pre>  'key`arg1`arg2'</pre>
56        *
57        * @param encodedStr An encoded string.
58        * @return The decoded string, or the encoded string if it fails to decode.
59        * @see com.google.javascript.jscomp.ReplaceStrings
60        */
61       public String decode(String encodedStr) {
62         String[] suppliedBits = encodedStr.split(ARGUMENT_PLACE_HOLDER, -1);
63         String originalStr = originalToNewNameMap.get(suppliedBits[0]);
64         if (originalStr == null) {
65           return encodedStr; // Failed to decode.
66         }
67         String[] originalBits = originalStr.split(ARGUMENT_PLACE_HOLDER, -1);
68         StringBuilder sb = new StringBuilder(originalBits[0]);
69         for (int i = 1; i < Math.max(originalBits.length, suppliedBits.length); i++) {
70           // Replace missing bits with "-". Shouldn't happen except that we aren't
71           // escaping the replacement token at the moment.
72           sb.append(i < suppliedBits.length ? suppliedBits[i] : "-");
73           sb.append(i < originalBits.length ? originalBits[i] : "-");
74         }
75         return sb.toString();
76       }
77     
78     }
