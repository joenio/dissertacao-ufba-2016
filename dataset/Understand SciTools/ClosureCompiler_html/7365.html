16     package com.google.javascript.jscomp;
17     
18     import com.google.debugging.sourcemap.proto.Mapping.OriginalMapping;
19     
20     /**
21      * A source excerpt provider is responsible for building source code excerpt
22      * of specific locations, such as a specific line or a region around a
23      * given line number.
24      *
25      */
26     public interface SourceExcerptProvider {
27       /**
28        * Source excerpt variety.
29        */
30       enum SourceExcerpt {
31         /**
32          * Line excerpt.
33          */
34         LINE {
35           @Override
36           public String get(SourceExcerptProvider source, String sourceName,
37               int lineNumber, ExcerptFormatter formatter) {
38             return formatter.formatLine(
39                 source.getSourceLine(sourceName, lineNumber), lineNumber);
40           }
41         },
42         /**
43          * Region excerpt.
44          */
45         REGION {
46           @Override
47           public String get(SourceExcerptProvider source, String sourceName,
48               int lineNumber, ExcerptFormatter formatter) {
49             return formatter.formatRegion(
50                 source.getSourceRegion(sourceName, lineNumber));
51           }
52         };
53     
54         /**
55          * Get a source excerpt string based on the type of the source excerpt.
56          */
57         public abstract String get(SourceExcerptProvider source, String sourceName,
58             int lineNumber, ExcerptFormatter formatter);
59       }
60     
61       /**
62        * Get the line indicated by the line number. This call will return only the
63        * specific line.
64        *
65        * @param lineNumber the line number, 1 being the first line of the file
66        * @return the line indicated, or {@code null} if it does not exist
67        */
68       String getSourceLine(String sourceName, int lineNumber);
69     
70       /**
71        * Get a region around the indicated line number. The exact definition of a
72        * region is implementation specific, but it must contain the line indicated
73        * by the line number. A region must not start or end by a carriage return.
74        *
75        * @param lineNumber the line number, 1 being the first line of the file
76        * @return the region around the line number indicated, or {@code null}
77        * if it does not exist
78        */
79       Region getSourceRegion(String sourceName, int lineNumber);
80     
81       /**
82        * If the source file is associated with an inputSourceMap, does a lookup
83        * for the original file position based on this source map.
84        */
85       OriginalMapping getSourceMapping(String sourceName, int lineNumber, int columnNumber);
86     
87       /**
88        * A excerpt formatter is responsible of formatting source excerpts.
89        */
90       interface ExcerptFormatter {
91         /**
92          * Format a line excerpt.
93          */
94         String formatLine(String line, int lineNumber);
95     
96         /**
97          * Format a region excerpt.
98          */
99         String formatRegion(Region region);
100       }
101     }
