39     /**
40      * Gathers property names defined in externs.
41      */
42     class GatherExternProperties extends AbstractPostOrderCallback
43         implements CompilerPass {
44       private final Set<String> externProperties = new LinkedHashSet<>();
45       private final AbstractCompiler compiler;
46       private final ExtractRecordTypePropertyNames typeVisitor =
47           new ExtractRecordTypePropertyNames();
48     
49       public GatherExternProperties(AbstractCompiler compiler) {
50         this.compiler = compiler;
51       }
52     
53       @Override
54       public void process(Node externs, Node root) {
55         NodeTraversal.traverseEs6(compiler, externs, this);
56         compiler.setExternProperties(ImmutableSet.copyOf(externProperties));
57       }
58     
59       @Override
60       public void visit(NodeTraversal t, Node n, Node parent) {
61         switch (n.getType()) {
62           case GETPROP:
63             // Gathers "name" from (someObject.name).
64             Node dest = n.getSecondChild();
65             if (dest.isString()) {
66               externProperties.add(dest.getString());
67             }
68             break;
69           case OBJECTLIT:
70             // Gathers "name" and "address" from ({name: null, address: null}).
71             for (Node child = n.getFirstChild();
72                  child != null;
73                  child = child.getNext()) {
74               externProperties.add(child.getString());
75             }
76             break;
77         }
78     
79         // Gather field names from the type of the node (if any).
80         JSType type = n.getJSType();
81         if (type != null) {
82           typeVisitor.visitOnce(type);
83         }
84     
85         // Gather field names from the @typedef declaration.
86         // Typedefs are declared on qualified name nodes.
87         if (n.isQualifiedName()) {
88           // Get the JSDoc for the current node and check if it contains a
89           // typedef.
90           JSDocInfo jsDoc = NodeUtil.getBestJSDocInfo(n);
91           if (jsDoc != null && jsDoc.hasTypedefType()) {
92             // Get the corresponding type by looking at the type registry.
93             JSType typedefType = compiler.getTypeIRegistry().getType(n.getQualifiedName());
94             if (typedefType != null) {
95               typeVisitor.visitOnce(typedefType);
96             }
97           }
98         }
99       }
100     
101       private class ExtractRecordTypePropertyNames
102           implements Visitor<Set<String>> {
103         private final Set<JSType> seenTypes = Sets.newIdentityHashSet();
104     
105         public void visitOnce(JSType type) {
106           // Handle recursive types by only ever visiting the same type once.
107           if (seenTypes.add(type)) {
108             type.visit(this);
109           }
110         }
111     
112         // Interesting cases first, no-ops later.
113     
114         @Override
115         public Set<String> caseEnumElementType(EnumElementType type) {
116           // Descend into the enum's element type.
117           // @enum {T}
118           visitOnce(type.getPrimitiveType());
119     
120           return externProperties;
121         }
122     
123         @Override
124         public Set<String> caseFunctionType(FunctionType type) {
125           // Visit parameter types.
126           // function(T1, T2), as well as @param {T}
127           for (Node param : type.getParameters()) {
128             visitOnce(param.getJSType());
129           }
130     
131           // Visit the return type.
132           // function(): T, as well as @return {T}
133           visitOnce(type.getReturnType());
134     
135           // @interface
136           if (type.isInterface()) {
137             // Visit the extended interfaces.
138             // @extends {T}
139             for (JSType extendedType : type.getExtendedInterfaces()) {
140               visitOnce(extendedType);
141             }
142           }
143     
144           // @constructor
145           if (type.isConstructor()) {
146             // Visit the implemented interfaces.
147             // @implements {T}
148             for (JSType implementedType : type.getOwnImplementedInterfaces()) {
149               visitOnce(implementedType);
150             }
151     
152             // Visit the parent class (if any).
153             // @extends {T}
154             JSType superClass = type.getPrototype().getImplicitPrototype();
155             if (superClass != null) {
156               visitOnce(superClass);
157             }
158           }
159           return externProperties;
160         }
161     
162         @Override
163         public Set<String> caseObjectType(ObjectType type) {
164           // Record types.
165           // {a: T1, b: T2}.
166           if (type.isRecordType()) {
167             for (String propertyName : type.getOwnPropertyNames()) {
168               // After type inference it is possible that some nodes in externs
169               // can have types which are defined in non-extern code. To avoid
170               // bleeding property names of such types into externs we check that
171               // the node for each property was defined in externs.
172               if (type.getPropertyNode(propertyName).isFromExterns()) {
173                 externProperties.add(propertyName);
174                 visitOnce(type.getPropertyType(propertyName));
175               }
176             }
177           }
178     
179           return externProperties;
180         }
181     
182         @Override
183         public Set<String> caseNamedType(NamedType type) {
184           // Treat as all other proxy objects.
185           return caseProxyObjectType(type);
186         }
187     
188         @Override
189         public Set<String> caseProxyObjectType(ProxyObjectType type) {
190           // Visit the proxied type.
191           // @typedef {T}
192           type.visitReferenceType(this);
193     
194           return externProperties;
195         }
196     
197         @Override
198         public Set<String> caseUnionType(UnionType type) {
199           // Visit the alternatives.
200           // T1|T2|T3
201           for (JSType alternateType : type.getAlternates()) {
202             visitOnce(alternateType);
203           }
204           return externProperties;
205         }
206     
207         @Override
208         public Set<String> caseTemplatizedType(TemplatizedType type) {
209           // Visit the type arguments.
210           // SomeType.<T1, T2>
211           for (JSType templateType : type.getTemplateTypes()) {
212             visitOnce(templateType);
213           }
214           return externProperties;
215         }
216     
217         @Override
218         public Set<String> caseNoType(NoType type) {
219           return externProperties;
220         }
221     
222         @Override
223         public Set<String> caseAllType() {
224           return externProperties;
225         }
226     
227         @Override
228         public Set<String> caseBooleanType() {
229           return externProperties;
230         }
231     
232         @Override
233         public Set<String> caseNoObjectType() {
234           return externProperties;
235         }
236     
237         @Override
238         public Set<String> caseUnknownType() {
239           return externProperties;
240         }
241     
242         @Override
243         public Set<String> caseNullType() {
244           return externProperties;
245         }
246     
247         @Override
248         public Set<String> caseNumberType() {
249           return externProperties;
250         }
251     
252         @Override
253         public Set<String> caseStringType() {
254           return externProperties;
255         }
256     
257         @Override
258         public Set<String> caseVoidType() {
259           return externProperties;
260         }
261     
262         @Override
263         public Set<String> caseTemplateType(TemplateType templateType) {
264           return externProperties;
265         }
266       }
