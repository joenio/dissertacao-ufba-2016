40     /**
41      * Rewrites a ES6 module into a form that can be safely concatenated.
42      * Note that we treat a file as an ES6 module if it has at least one import or
43      * export statement.
44      *
45      * @author moz@google.com (Michael Zhou)
46      */
47     public final class ProcessEs6Modules extends AbstractPostOrderCallback {

53               "The left side of a goog.require() must use ''const'' (not ''let'' or ''var'')");

92       /**
93        * Creates a new ProcessEs6Modules instance which can be used to rewrite
94        * ES6 modules to a concatenable form.
95        *
96        * @param compiler The compiler
97        * @param loader The module loader which is used to locate ES6 modules
98        * @param reportDependencies Whether the rewriter should report dependency
99        *     information to the Closure dependency manager. This needs to be true
100        *     if we want to sort ES6 module inputs correctly. Note that goog.provide
101        *     and goog.require calls will still be generated if this argument is
102        *     false.
103        */
104       public ProcessEs6Modules(Compiler compiler, ES6ModuleLoader loader,
105           boolean reportDependencies) {
106         this.compiler = compiler;
107         this.loader = loader;
108         this.reportDependencies = reportDependencies;
109       }

111       /**
112        * If a file contains an ES6 "import" or "export" statement, or the forceRewrite
113        * option is true, rewrite the source as a module.
114        */
115       public void processFile(Node root, boolean forceRewrite) {
116         FindGoogProvideOrGoogModule finder = new FindGoogProvideOrGoogModule();
117         NodeTraversal.traverseEs6(compiler, root, finder);
118         if (finder.isFound()) {
119           return;
120         }
121         this.forceRewrite = forceRewrite;
122         isEs6Module = forceRewrite;
123         NodeTraversal.traverseEs6(compiler, root, this);
124       }

126       @Override
127       public void visit(NodeTraversal t, Node n, Node parent) {
128         if (n.isImport()) {
129           isEs6Module = true;
130           visitImport(t, n, parent);
131         } else if (n.isExport()) {
132           isEs6Module = true;
133           visitExport(t, n, parent);
134         } else if (n.isScript()) {
135           scriptNodeCount++;
136           visitScript(t, n);
137         }
138       }

140       private void visitImport(NodeTraversal t, Node importDecl, Node parent) {
141         String moduleName;
142         String importName = importDecl.getLastChild().getString();
143         boolean isNamespaceImport = importName.startsWith("goog:");
144         if (isNamespaceImport) {
145           // Allow importing Closure namespace objects (e.g. from goog.provide or goog.module) as
146           //   import ... from 'goog:my.ns.Object'.
147           // These are rewritten to plain namespace object accesses.
148           moduleName = importName.substring("goog:".length());
149         } else {
150           URI loadAddress = checkNotNull(loader.locateEs6Module(importName, t.getInput()));
151           moduleName = ES6ModuleLoader.toModuleName(loadAddress);
152         }
153     
154         for (Node child : importDecl.children()) {
155           if (child.isEmpty() || child.isString()) {
156             continue;
157           } else if (child.isName()) { // import a from "mod"
158             // Namespace imports' default export is the namespace itself.
159             String name = isNamespaceImport ? "" : "default";
160             importMap.put(child.getString(), new ModuleOriginalNamePair(moduleName, name));
161           } else if (child.getType() == Token.IMPORT_SPECS) {
162             for (Node grandChild : child.children()) {
163               String origName = grandChild.getFirstChild().getString();
164               if (grandChild.getChildCount() == 2) { // import {a as foo} from "mod"
165                 importMap.put(
166                     grandChild.getLastChild().getString(),
167                     new ModuleOriginalNamePair(moduleName, origName));
168               } else { // import {a} from "mod"
169                 importMap.put(
170                     origName,
171                     new ModuleOriginalNamePair(moduleName, origName));
172               }
173             }
174           } else {
175             // import * as ns from "mod"
176             Preconditions.checkState(child.getType() == Token.IMPORT_STAR,
177                 "Expected an IMPORT_STAR node, but was: %s", child);
178             // Namespace imports cannot be imported "as *".
179             if (isNamespaceImport) {
180               compiler.report(t.makeError(importDecl, NAMESPACE_IMPORT_CANNOT_USE_STAR,
181                   child.getString(), moduleName));
182             }
183             importMap.put(
184                 child.getString(),
185                 new ModuleOriginalNamePair(moduleName, ""));
186           }
187         }
188     
189         Node script = NodeUtil.getEnclosingScript(parent);
190         // Emit goog.require call for the module.
191         if (alreadyRequired.add(moduleName)) {
192           Node require = IR.exprResult(
193               IR.call(NodeUtil.newQName(compiler, "goog.require"), IR.string(moduleName)));
194           require.copyInformationFromForTree(importDecl);
195           script.addChildAfter(require, googRequireInsertSpot);
196           googRequireInsertSpot = require;
197     
198           if (reportDependencies) {
199             t.getInput().addRequire(moduleName);
200           }
201         }
202     
203         parent.removeChild(importDecl);
204         compiler.reportCodeChange();
205       }

207       private void visitExport(NodeTraversal t, Node export, Node parent) {
208         if (export.getBooleanProp(Node.EXPORT_DEFAULT)) {
209           // export default
210     
211           // If the thing being exported is a class or function that has a name,
212           // extract it from the export statement, so that it can be referenced
213           // from within the module.
214           //
215           //   export default class X {} -> class X {}; ... moduleName.default = X;
216           //   export default function X() {} -> function X() {}; ... moduleName.default = X;
217           //
218           // Otherwise, create a local variable for it and export that.
219           //
220           //   export default 'someExpression'
221           //     ->
222           //   var $jscompDefaultExport = 'someExpression';
223           //   ...
224           //   moduleName.default = $jscompDefaultExport;
225           Node child = export.getFirstChild();
226           String name = null;
227     
228           if (child.isFunction() || child.isClass()) {
229             name = NodeUtil.getName(child);
230           }
231     
232           if (name != null) {
233             Node decl = child.cloneTree();
234             decl.setJSDocInfo(child.getJSDocInfo());
235             parent.replaceChild(export, decl);
236             exportMap.put("default", new NameNodePair(name, child));
237           } else {
238             Node var = IR.var(IR.name(DEFAULT_EXPORT_NAME), export.removeFirstChild());
239             var.setJSDocInfo(child.getJSDocInfo());
240             child.setJSDocInfo(null);
241             var.useSourceInfoIfMissingFromForTree(export);
242             parent.replaceChild(export, var);
243             exportMap.put("default", new NameNodePair(DEFAULT_EXPORT_NAME, child));
244           }
245         } else if (export.getBooleanProp(Node.EXPORT_ALL_FROM)) {
246           //   export * from 'moduleIdentifier';
247           compiler.report(JSError.make(export, Es6ToEs3Converter.CANNOT_CONVERT_YET,
248               "Wildcard export"));
249         } else if (export.getChildCount() == 2) {
250           //   export {x, y as z} from 'moduleIdentifier';
251           Node moduleIdentifier = export.getLastChild();
252           Node importNode = new Node(Token.IMPORT, moduleIdentifier.cloneNode());
253           importNode.copyInformationFrom(export);
254           parent.addChildBefore(importNode, export);
255           visit(t, importNode, parent);
256     
257           URI loadAddress = checkNotNull(
258               loader.locateEs6Module(moduleIdentifier.getString(), t.getInput()));
259           String moduleName = ES6ModuleLoader.toModuleName(loadAddress);
260     
261           for (Node exportSpec : export.getFirstChild().children()) {
262             String nameFromOtherModule = exportSpec.getFirstChild().getString();
263             String exportedName = exportSpec.getLastChild().getString();
264             exportMap.put(exportedName,
265                 new NameNodePair(moduleName + "." + nameFromOtherModule, exportSpec));
266           }
267           parent.removeChild(export);
268         } else {
269           if (export.getFirstChild().getType() == Token.EXPORT_SPECS) {
270             //     export {Foo};
271             for (Node exportSpec : export.getFirstChild().children()) {
272               Node origName = exportSpec.getFirstChild();
273               exportMap.put(
274                   exportSpec.getChildCount() == 2
275                       ? exportSpec.getLastChild().getString()
276                       : origName.getString(),
277                   new NameNodePair(origName.getString(), exportSpec));
278             }
279             parent.removeChild(export);
280           } else {
281             //    export var Foo;
282             //    export function Foo() {}
283             // etc.
284             Node declaration = export.getFirstChild();
285             for (int i = 0; i < declaration.getChildCount(); i++) {
286               Node maybeName = declaration.getChildAtIndex(i);
287               if (!maybeName.isName()) {
288                 break;
289               }
290               // Break out on "B" in "class A extends B"
291               if (declaration.isClass() && i > 0) {
292                 break;
293               }
294               String name = maybeName.getString();
295               Var v = t.getScope().getVar(name);
296               if (v == null || v.isGlobal()) {
297                 exportMap.put(name, new NameNodePair(name, maybeName));
298               }
299     
300               // If the declaration declares a new type, create annotations for
301               // the type checker.
302               // TODO(moz): Currently we only record ES6 classes and typedefs,
303               // need to handle other kinds of type declarations too.
304               if (declaration.isClass()) {
305                 classes.add(name);
306               }
307               if (declaration.getJSDocInfo() != null && declaration.getJSDocInfo().hasTypedefType()) {
308                 typedefs.add(name);
309               }
310             }
311             parent.replaceChild(export, declaration.detachFromParent());
312           }
313           compiler.reportCodeChange();
314         }
315       }

317       private void visitScript(NodeTraversal t, Node script) {
318         if (!isEs6Module) {
319           return;
320         }
321     
322         ClosureRewriteModule.checkAndSetStrictModeDirective(t, script);
323     
324         Preconditions.checkArgument(scriptNodeCount == 1,
325             "ProcessEs6Modules supports only one invocation per "
326             + "CompilerInput / script node");
327     
328         // rewriteRequires is here (rather than being part of the main visit()
329         // method, because we only want to rewrite the requires if this is an
330         // ES6 module.
331         rewriteRequires(script);
332     
333         URI normalizedAddress = loader.normalizeInputAddress(t.getInput());
334         String moduleName = ES6ModuleLoader.toModuleName(normalizedAddress);
335     
336         for (Map.Entry<String, NameNodePair> entry : exportMap.entrySet()) {
337           String exportedName = entry.getKey();
338           String withSuffix = entry.getValue().name;
339           Node nodeForSourceInfo = entry.getValue().nodeForSourceInfo;
340           Node getProp = IR.getprop(IR.name(moduleName), IR.string(exportedName));
341     
342           if (typedefs.contains(exportedName)) {
343             // /** @typedef {foo} */
344             // moduleName.foo;
345             JSDocInfoBuilder builder = new JSDocInfoBuilder(true);
346             JSTypeExpression typeExpr = new JSTypeExpression(
347                 IR.string(exportedName), script.getSourceFileName());
348             builder.recordTypedef(typeExpr);
349             JSDocInfo info = builder.build();
350             getProp.setJSDocInfo(info);
351             Node exprResult = IR.exprResult(getProp)
352                 .useSourceInfoIfMissingFromForTree(nodeForSourceInfo);
353             script.addChildToBack(exprResult);
354           } else {
355             //   moduleName.foo = foo;
356             // with a @const annotation if needed.
357             Node assign = IR.assign(
358                 getProp,
359                 NodeUtil.newQName(compiler, withSuffix));
360             Node exprResult = IR.exprResult(assign)
361                 .useSourceInfoIfMissingFromForTree(nodeForSourceInfo);
362             if (classes.contains(exportedName)) {
363               JSDocInfoBuilder builder = new JSDocInfoBuilder(true);
364               builder.recordConstancy();
365               JSDocInfo info = builder.build();
366               assign.setJSDocInfo(info);
367             }
368             script.addChildToBack(exprResult);
369           }
370         }
371     
372         // Rename vars to not conflict in global scope.
373         NodeTraversal.traverseEs6(compiler, script, new RenameGlobalVars(moduleName));
374     
375         if (!exportMap.isEmpty() || forceRewrite) {
376           // Add goog.provide call.
377           Node googProvide = IR.exprResult(
378               IR.call(NodeUtil.newQName(compiler, "goog.provide"),
379                   IR.string(moduleName)));
380           script.addChildToFront(googProvide.copyInformationFromForTree(script));
381           if (reportDependencies) {
382             t.getInput().addProvide(moduleName);
383           }
384         }
385     
386         JSDocInfoBuilder jsDocInfo = script.getJSDocInfo() == null
387             ? new JSDocInfoBuilder(false)
388             : JSDocInfoBuilder.copyFrom(script.getJSDocInfo());
389         if (!jsDocInfo.isPopulatedWithFileOverview()) {
390           jsDocInfo.recordFileOverview("");
391         }
392         // Don't check provides and requires, since most of them are auto-generated.
393         jsDocInfo.recordSuppressions(ImmutableSet.of("missingProvide", "missingRequire"));
394         script.setJSDocInfo(jsDocInfo.build());
395     
396         exportMap.clear();
397         compiler.reportCodeChange();
398       }

400       private void rewriteRequires(Node script) {
401         NodeTraversal.traverseEs6(
402             compiler,
403             script,
404             new NodeTraversal.AbstractShallowCallback() {
405               @Override
406               public void visit(NodeTraversal t, Node n, Node parent) {
407                 if (n.isCall()
408                     && n.getFirstChild().matchesQualifiedName("goog.require")
409                     && NodeUtil.isNameDeclaration(parent.getParent())) {
410                   visitRequire(n, parent);
411                 }
412               }
413     
414               /**
415                * Rewrites
416                *   const foo = goog.require('bar.foo');
417                * to
418                *   goog.require('bar.foo');
419                *   const foo = bar.foo;
420                */
421               private void visitRequire(Node requireCall, Node parent) {
422                 String namespace = requireCall.getLastChild().getString();
423                 if (!parent.getParent().isConst()) {
424                   compiler.report(JSError.make(parent.getParent(), LHS_OF_GOOG_REQUIRE_MUST_BE_CONST));
425                 }
426     
427                 // If the LHS is a destructuring pattern with the "shorthand" syntax,
428                 // desugar it because otherwise the renaming will not be done correctly.
429                 //   const {x} = goog.require('y')
430                 // becomes
431                 //   const {x: x} = goog.require('y');
432                 if (parent.isObjectPattern()) {
433                   for (Node key = parent.getFirstChild(); key != null; key = key.getNext()) {
434                     if (!key.hasChildren()) {
435                       key.addChildToBack(IR.name(key.getString()).useSourceInfoFrom(key));
436                     }
437                   }
438                 }
439     
440                 Node replacement = NodeUtil.newQName(compiler, namespace).srcrefTree(requireCall);
441                 parent.replaceChild(requireCall, replacement);
442                 Node varNode = parent.getParent();
443                 varNode.getParent().addChildBefore(
444                     IR.exprResult(requireCall).srcrefTree(requireCall),
445                     varNode);
446               }
447             });
448       }

450       /**
451        * Traverses a node tree and
452        * <ol>
453        *   <li>Appends a suffix to all global variable names defined in this module.
454        *   <li>Changes references to imported values to be property accesses on the
455        *    imported module object.
456        * </ol>
457        */
458       private class RenameGlobalVars extends AbstractPostOrderCallback {
459         private final String suffix;
460     
461         RenameGlobalVars(String suffix) {
462           this.suffix = suffix;
463         }
464     
465         @Override
466         public void visit(NodeTraversal t, Node n, Node parent) {
467           JSDocInfo info = n.getJSDocInfo();
468           if (info != null) {
469             for (Node typeNode : info.getTypeNodes()) {
470               fixTypeNode(t, typeNode);
471             }
472           }
473     
474           boolean isShorthandObjLitKey = n.isStringKey() && !n.hasChildren();
475           if (n.isName() || isShorthandObjLitKey) {
476             String name = n.getString();
477             if (suffix.equals(name)) {
478               // TODO(moz): Investigate whether we need to return early in this unlikely situation.
479               return;
480             }
481     
482             Var var = t.getScope().getVar(name);
483             if (var != null && var.isGlobal()) {
484               // Avoid polluting the global namespace.
485               String newName = name + "$$" + suffix;
486               if (isShorthandObjLitKey) {
487                 // Change {a} to {a: a$$module$foo}
488                 n.addChildToBack(IR.name(newName).useSourceInfoIfMissingFrom(n));
489               } else {
490                 n.setString(newName);
491                 n.setOriginalName(name);
492               }
493             } else if (var == null && importMap.containsKey(name)) {
494               // Change to property access on the imported module object.
495               if (parent.isCall() && parent.getFirstChild() == n) {
496                 parent.putBooleanProp(Node.FREE_CALL, false);
497               }
498     
499               ModuleOriginalNamePair pair = importMap.get(name);
500               Node moduleAccess = NodeUtil.newQName(compiler, pair.module);
501               if (pair.originalName.isEmpty()) {
502                 n.getParent().replaceChild(
503                     n, moduleAccess.useSourceInfoIfMissingFromForTree(n));
504               } else {
505                 n.getParent().replaceChild(n,
506                     IR.getprop(moduleAccess, IR.string(pair.originalName))
507                         .useSourceInfoIfMissingFromForTree(n));
508               }
509             }
510           }
511         }

513         /**
514          * Replace type name references. Change short names to fully qualified names
515          * with namespace prefixes. Eg: {Foo} becomes {module$test.Foo}.
516          */
517         private void fixTypeNode(NodeTraversal t, Node typeNode) {
518           if (typeNode.isString()) {
519             String name = typeNode.getString();
520             if (ES6ModuleLoader.isRelativeIdentifier(name)
521                 || ES6ModuleLoader.isAbsoluteIdentifier(name)) {
522               int lastSlash = name.lastIndexOf('/');
523               int endIndex = name.indexOf('.', lastSlash);
524               String localTypeName = null;
525               if (endIndex == -1) {
526                 endIndex = name.length();
527               } else {
528                 localTypeName = name.substring(endIndex);
529               }
530     
531               String moduleName = name.substring(0, endIndex);
532               URI loadAddress = checkNotNull(loader.locateEs6Module(moduleName, t.getInput()));
533               String globalModuleName = ES6ModuleLoader.toModuleName(loadAddress);
534               typeNode.setString(
535                   localTypeName == null ? globalModuleName : globalModuleName + localTypeName);
536             } else {
537               List<String> splitted = Splitter.on('.').limit(2).splitToList(name);
538               String baseName = splitted.get(0);
539               String rest = "";
540               if (splitted.size() == 2) {
541                 rest = "." + splitted.get(1);
542               }
543               Var var = t.getScope().getVar(baseName);
544               if (var != null && var.isGlobal()) {
545                 typeNode.setString(baseName + "$$" + suffix + rest);
546               } else if (var == null && importMap.containsKey(baseName)) {
547                 ModuleOriginalNamePair pair = importMap.get(baseName);
548                 if (pair.originalName.isEmpty()) {
549                   typeNode.setString(pair.module + rest);
550                 } else {
551                   typeNode.setString(baseName + "$$" + pair.module + rest);
552                 }
553               }
554               typeNode.setOriginalName(name);
555             }
556           }
557     
558           for (Node child = typeNode.getFirstChild(); child != null;
559                child = child.getNext()) {
560             fixTypeNode(t, child);
561           }
562           compiler.reportCodeChange();
563         }
564       }

566       private static class ModuleOriginalNamePair {
567         private String module;
568         private String originalName;
569     
570         private ModuleOriginalNamePair(String module, String originalName) {
571           this.module = module;
572           this.originalName = originalName;

574     
575         @Override
576         public String toString() {
577           return "(" + module + ", " + originalName + ")";
