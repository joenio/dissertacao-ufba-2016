16     package com.google.javascript.jscomp;
17     
18     import com.google.javascript.jscomp.NodeTraversal.AbstractPostOrderCallback;
19     import com.google.javascript.rhino.Node;
20     
21     /**
22      * {@link CheckDebuggerStatement} checks for the presence of the "debugger"
23      * statement in JavaScript code. It is appropriate to use this statement while
24      * developing JavaScript; however, it is generally undesirable to include it in
25      * production code.
26      *
27      * @author bolinfest@google.com (Michael Bolin)
28      */
29     class CheckDebuggerStatement extends AbstractPostOrderCallback
30         implements CompilerPass {
31     
32       static final DiagnosticType DEBUGGER_STATEMENT_PRESENT =
33         DiagnosticType.disabled("JSC_DEBUGGER_STATEMENT_PRESENT",
34             "Using the debugger statement can halt your application if the user " +
35             "has a JavaScript debugger running.");
36     
37       private final AbstractCompiler compiler;
38     
39       public CheckDebuggerStatement(AbstractCompiler compiler) {
40         this.compiler = compiler;
41       }
42     
43       @Override
44       public void process(Node externs, Node root) {
45         NodeTraversal.traverseEs6(compiler, root, this);
46       }
47     
48       @Override
49       public void visit(NodeTraversal t, Node n, Node parent) {
50         if (n.isDebugger()) {
51           t.report(n, DEBUGGER_STATEMENT_PRESENT);
52         }
53       }
54     }
