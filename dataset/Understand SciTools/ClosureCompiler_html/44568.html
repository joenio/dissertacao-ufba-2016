59       public void testUndeclaredConst() {
60         assertEarlyReferenceError("if (a) { x = 3; const x = 3;}");
61     
62         // For the following, IE 11 gives "Assignment to const", but technically
63         // they are also undeclared references, which get caught in the first place.
64         assertEarlyReferenceError(LINE_JOINER.join(
65             "var x = 1;",
66             "if (true) {",
67             "  x++;",
68             "  const x = 3;",
69             "}"));
70     
71         assertEarlyReferenceError("a = 1; const a = 0;");
72         assertEarlyReferenceError("a++; const a = 0;");
73       }

75       public void testIllegalLetShadowing() {
76         assertRedeclareError("if (a) { let x; var x;}");

78         assertRedeclareError("if (a) { let x; let x;}");
79     
80         assertRedeclareError(LINE_JOINER.join(
81             "function f() {",
82             "  let x;",
83             "  if (a) {",
84             "    var x;",
85             "  }",
86             "}"));

89             "function f() {",
90             "  if (a) {",
91             "    let x;",
92             "  }",
93             "  var x;",
94             "}"));
95     
96         assertNoWarning(
97             LINE_JOINER.join(
98                 "function f() {",
99                 "  if (a) { let x; }",

101                 "}"));
102

104         assertRedeclareError("var x; let x;");

330       /**
331        * We can't catch all possible runtime errors but it's useful to have some
332        * basic checks.
333        */
334       public void testDefaultParam() {
335         assertEarlyReferenceError("function f(x=a) { let a; }");
336         assertEarlyReferenceError(LINE_JOINER.join(
337             "function f(x=a) { let a; }",
338             "function g(x=1) { var a; }"));
339         assertEarlyReferenceError("function f(x=a) { var a; }");
340         assertEarlyReferenceError("function f(x=a()) { function a() {} }");
341         assertEarlyReferenceError("function f(x=[a]) { var a; }");
342         assertEarlyReferenceError("function f(x=y, y=2) {}");
343         assertNoWarning("function f(x=a) {}");
344         assertNoWarning("function f(x=a) {} var a;");
345         assertNoWarning("let b; function f(x=b) { var b; }");
346         assertNoWarning("function f(y = () => x, x = 5) { return y(); }");
347         assertNoWarning("function f(x = new foo.bar()) {}");
348         assertNoWarning("var foo = {}; foo.bar = class {}; function f(x = new foo.bar()) {}");
349       }

351       public void testDestructuring() {
352         testSame(LINE_JOINER.join(
353             "function f() { ",
354             "  var obj = {a:1, b:2}; ",
355             "  var {a:c, b:d} = obj; ",
356             "}"));
357         testSame(LINE_JOINER.join(
358             "function f() { ",
359             "  var obj = {a:1, b:2}; ",
360             "  var {a, b} = obj; ",
361             "}"));
362     
363         assertRedeclare(LINE_JOINER.join(
364             "function f() { ",
365             "  var obj = {a:1, b:2}; ",
366             "  var {a:c, b:d} = obj; ",
367             "  var c = b;",
368             "}"));
369     
370         assertEarlyReference(LINE_JOINER.join(
371             "function f() { ",
372             "  var {a:c, b:d} = obj;",
373             "  var obj = {a:1, b:2};",
374             "}"));
375         assertEarlyReference(LINE_JOINER.join(
376             "function f() { ",
377             "  var {a, b} = obj;",
378             "  var obj = {a:1, b:2};",
379             "}"));
380         assertEarlyReference(LINE_JOINER.join(
381             "function f() { ",
382             "  var e = c;",
383             "  var {a:c, b:d} = {a:1, b:2};",
384             "}"));
385       }
