275       private IRFactory(String sourceString,
276                         StaticSourceFile sourceFile,
277                         Config config,
278                         ErrorReporter errorReporter,
279                         ImmutableList<Comment> comments) {
280         this.sourceString = sourceString;
281         this.nextCommentIter = comments.iterator();
282         this.currentComment = skipNonJsDoc(nextCommentIter);
283         this.newlines = new ArrayList<>();
284         this.sourceFile = sourceFile;
285         this.fileLevelJsDocBuilder =
286             new JSDocInfoBuilder(config.parseJsDocDocumentation.shouldParseDescriptions());
287     
288         // Pre-generate all the newlines in the file.
289         for (int charNo = 0; true; charNo++) {
290           charNo = sourceString.indexOf('\n', charNo);
291           if (charNo == -1) {
292             break;
293           }
294           newlines.add(charNo);
295         }
296     
297         // Sometimes this will be null in tests.
298         this.sourceName = sourceFile == null ? null : sourceFile.getName();
299     
300         this.config = config;
301         this.errorReporter = errorReporter;
302         this.transformDispatcher = new TransformDispatcher();
303         // The template node properties are applied to all nodes in this transform.
304         this.templateNode = createTemplateNode();
305     
306         switch (config.languageMode) {
307           case ECMASCRIPT3:
308             reservedKeywords = null; // use TokenStream.isKeyword instead
309             break;
310           case ECMASCRIPT5:
311           case ECMASCRIPT6:
312             reservedKeywords = ES5_RESERVED_KEYWORDS;
313             break;
314           case ECMASCRIPT5_STRICT:
315           case ECMASCRIPT6_STRICT:
316           case ECMASCRIPT6_TYPED:
317           case ECMASCRIPT7:
318           case ECMASCRIPT8:
319             reservedKeywords = ES5_STRICT_RESERVED_KEYWORDS;
320             break;
321           default:
322             throw new IllegalStateException("unknown language mode: " + config.languageMode);
323         }
324       }

326       private static Comment skipNonJsDoc(UnmodifiableIterator<Comment> comments) {
327         while (comments.hasNext()) {
328           Comment comment = comments.next();
329           if (comment.type == Comment.Type.JSDOC) {
330             return comment;
331           }
332         }
333         return null;
334       }

346       public static IRFactory transformTree(ProgramTree tree,
347                                             StaticSourceFile sourceFile,
348                                             String sourceString,
349                                             Config config,
350                                             ErrorReporter errorReporter) {
351         IRFactory irFactory = new IRFactory(sourceString, sourceFile,
352             config, errorReporter, tree.sourceComments);
353     
354         // don't call transform as we don't want standard jsdoc handling.
355         Node n = irFactory.transformDispatcher.process(tree);
356         irFactory.setSourceInfo(n, tree);
357     
358         if (tree.sourceComments != null) {
359           for (Comment comment : tree.sourceComments) {
360             if (comment.type == Comment.Type.JSDOC
361                 && !irFactory.parsedComments.contains(comment)) {
362               irFactory.handlePossibleFileOverviewJsDoc(comment);
363             } else if (comment.type == Comment.Type.BLOCK) {
364               irFactory.handleBlockComment(comment);
365             }
366           }
367         }
368     
369         irFactory.setFileOverviewJsDoc(n);
370     
371         irFactory.validateAll(n);
372         irFactory.resultNode = n;
373     
374         return irFactory;
375       }

407       private void validateAll(Node n) {
408         ArrayDeque<Node> work = new ArrayDeque<>();
409         while (n != null) {
410           validate(n);
411           Node nextSibling = n.getNext();
412           Node firstChild = n.getFirstChild();
413           if (firstChild != null) {
414             if (nextSibling != null) {
415               // handle the siblings later
416               work.push(nextSibling);
417             }
418             n = firstChild;
419           } else if (nextSibling != null) {
420             // no children, handle the next sibling
421             n = nextSibling;
422           } else {
423             // no siblings, continue with work we have saved on the work queue
424             n = work.poll();
425           }
426         }
427         Preconditions.checkState(work.isEmpty());
428       }

437       private void validateReturn(Node n) {
438         if (n.isReturn()) {
439           Node parent = n;
440           while ((parent = parent.getParent()) != null) {
441             if (parent.isFunction()) {

450       private void validateBreakContinue(Node n) {
451         if (n.isBreak() || n.isContinue()) {
452           Node labelName = n.getFirstChild();
453           if (labelName != null) {
454             Node parent = n.getParent();
455             while (!parent.isLabel() || !labelsMatch(parent, labelName)) {
456               if (parent.isFunction() || parent.isScript()) {
457                 // report missing label
458                 errorReporter.error(
459                     SimpleFormat.format(UNDEFINED_LABEL, labelName.getString()),
460                     sourceName,
461                     n.getLineno(), n.getCharno());
462                 break;
463               }
464               parent = parent.getParent();
465             }
466             if (parent.isLabel() && labelsMatch(parent, labelName)) {
467               if (n.isContinue() && !isContinueTarget(parent.getLastChild())) {
468                 // report invalid continue target
469                 errorReporter.error(
470                     UNEXPECTED_LABLED_CONTINUE,
471                     sourceName,
472                     n.getLineno(), n.getCharno());
473               }
474             }
475           } else {
476             if (n.isContinue()) {
477               Node parent = n.getParent();
478               while (!isContinueTarget(parent)) {
479                 if (parent.isFunction() || parent.isScript()) {
480                   // report invalid continue
481                   errorReporter.error(
482                       UNEXPECTED_CONTINUE,
483                       sourceName,
484                       n.getLineno(), n.getCharno());
485                   break;
486                 }
487                 parent = parent.getParent();
488               }
489             } else {
490               Node parent = n.getParent();
491               while (!isBreakTarget(parent)) {
492                 if (parent.isFunction() || parent.isScript()) {
493                   // report invalid break
494                   errorReporter.error(
495                       UNLABELED_BREAK,
496                       sourceName,
497                       n.getLineno(), n.getCharno());
498                   break;
499                 }
500                 parent = parent.getParent();
501               }
502             }
503           }
504         }
505       }

537       private void validateLabel(Node n) {
538         if (n.isLabel()) {
539           Node labelName = n.getFirstChild();
540           for (Node parent = n.getParent();
541                parent != null && !parent.isFunction(); parent = parent.getParent()) {
542             if (parent.isLabel() && labelsMatch(parent, labelName)) {
543               errorReporter.error(
544                   SimpleFormat.format(DUPLICATE_LABEL, labelName.getString()),
545                   sourceName,
546                   n.getLineno(), n.getCharno());
547               break;
548             }
549           }
550         }

553       private void validateParameters(Node n) {
554         if (n.isParamList()) {
555           Node c = n.getFirstChild();
556           for (; c != null; c = c.getNext()) {
557             if (!c.isName()) {
558               continue;
559             }
560             Node sibling = c.getNext();
561             for (; sibling != null; sibling = sibling.getNext()) {
562               if (sibling.isName() && c.getString().equals(sibling.getString())) {
563                 errorReporter.warning(
564                     SimpleFormat.format(DUPLICATE_PARAMETER, c.getString()),
565                     sourceName,

708       private boolean shouldAttachJSDocHere(ParseTree tree) {
709         switch (tree.type) {
710           case EXPRESSION_STATEMENT:
711           case LABELLED_STATEMENT:
712           case EXPORT_DECLARATION:
713             return false;
714           case CALL_EXPRESSION:
715           case CONDITIONAL_EXPRESSION:
716           case BINARY_OPERATOR:
717           case MEMBER_EXPRESSION:
718           case MEMBER_LOOKUP_EXPRESSION:
719           case POSTFIX_EXPRESSION:
720             ParseTree nearest = findNearestNode(tree);
721             if (nearest.type == ParseTreeType.PAREN_EXPRESSION) {
722               return false;
723             }
724             return true;
725           default:
726             return true;
727         }
728       }

730       private static ParseTree findNearestNode(ParseTree tree) {
731         while (true) {
732           switch (tree.type) {
733             case EXPRESSION_STATEMENT:
734               tree = tree.asExpressionStatement().expression;
735               continue;
736             case CALL_EXPRESSION:
737               tree = tree.asCallExpression().operand;
738               continue;
739             case BINARY_OPERATOR:
740               tree = tree.asBinaryOperator().left;
741               continue;
742             case CONDITIONAL_EXPRESSION:
743               tree = tree.asConditionalExpression().condition;
744               continue;
745             case MEMBER_EXPRESSION:
746               tree = tree.asMemberExpression().operand;
747               continue;
748             case MEMBER_LOOKUP_EXPRESSION:
749               tree = tree.asMemberLookupExpression().operand;
750               continue;
751             case POSTFIX_EXPRESSION:
752               tree = tree.asPostfixExpression().operand;
753               continue;
754             default:
755               return tree;
756           }
757         }
758       }

953          * Transforms the given node and then sets its type to Token.STRING if it
954          * was Token.NAME. If its type was already Token.STRING, then quotes it.
955          * Used for properties, as the old AST uses String tokens, while the new one
956          * uses Name tokens for unquoted strings. For example, in
957          * var o = {'a' : 1, b: 2};
958          * the string 'a' is quoted, while the name b is turned into a string, but
959          * unquoted.

962             com.google.javascript.jscomp.parsing.parser.Token token) {
963           Node ret;

973             ret.putBooleanProp(Node.QUOTED_PROP, true);
974           }

1034          * Parse the directives, encode them in the AST, and remove their nodes.
1035          *
1036          * For information on ES5 directives, see section 14.1 of
1037          * ECMA-262, Edition 5.
1038          *
1039          * It would be nice if Rhino would eventually take care of this for
1040          * us, but right now their directive-processing is a one-off.
1041          */
1042         private void parseDirectives(Node node) {
1043           // Remove all the directives, and encode them in the AST.
1044           ImmutableSet.Builder<String> directives = null;
1045           while (isDirective(node.getFirstChild())) {
1046             String directive = node.removeFirstChild().getFirstChild().getString();
1047             if (directives == null) {
1048               directives = new ImmutableSet.Builder<>();
1049             }
1050             directives.add(directive);
1051           }
1052     
1053           if (directives != null) {
1054             ImmutableSet<String> result = directives.build();
1055             if (result.size() == 1 && result.contains("use strict")) {
1056               // Use a shared set.
1057               result = USE_STRICT_ONLY;
1058             }
1059             node.setDirectives(result);
1060           }
1061         }

1356         // Deep binary ops (typical string concatentations) can cause stack overflows,
1357         // avoid recursing in this case and loop instead.
1358         private Node processBinaryExpressionHelper(BinaryOperatorTree exprTree) {
1359           Node root = null;
1360           Node current = null;
1361           Node previous = null;
1362           while (exprTree != null) {
1363             previous = current;
1364             // Skip the first child but recurse normally into the right operand as typically this isn't
1365             // deep and because we have already checked that there isn't any JSDoc we can traverse
1366             // out of order.
1367             current = newNode(
1368                 transformBinaryTokenType(exprTree.operator.type),
1369                 transform(exprTree.right));
1370             // We have inlined "transform" here. Normally, we would need to handle the JSDoc here but we
1371             // know there is no JSDoc to attach, which simplifies things.
1372             setSourceInfo(current, exprTree);
1373     
1374             // As the iteration continues add the left operand.
1375             if (previous != null) {
1376               previous.addChildToFront(current);
1377             }
1378     
1379             if (exprTree.left instanceof BinaryOperatorTree) {
1380               // continue with the left hand child
1381               exprTree = (BinaryOperatorTree) exprTree.left;
1382             } else {
1383               // Finish things off, add the left operand to the current node.
1384               Node leftNode = transform(exprTree.left);
1385               current.addChildToFront(leftNode);
1386               // Nothing left to do.
1387               exprTree = null;
1388             }
1389     
1390             // Save the top binary op, this is the result to return.
1391             if (root == null) {
1392               root = current;
1393             }
1394           }
1395           return root;
1396         }

1749         Node processStringLiteral(LiteralExpressionTree literalTree) {
1750           LiteralToken token = literalTree.literalToken.asLiteral();
1751     
1752           Node n = processString(token);
1753           String value = n.getString();
1754           if (value.indexOf('\u000B') != -1) {
1755             // NOTE(nicksantos): In JavaScript, there are 3 ways to
1756             // represent a vertical tab: \v, \x0B, \u000B.
1757             // The \v notation was added later, and is not understood
1758             // on IE. So we need to preserve it as-is. This is really
1759             // obnoxious, because we do not have a good way to represent
1760             // how the original string was encoded without making the
1761             // representation of strings much more complicated.
1762             //
1763             // To handle this, we look at the original source test, and
1764             // mark the string as \v-encoded or not. If a string is
1765             // \v encoded, then all the vertical tabs in that string
1766             // will be encoded with a \v.
1767             int start = token.location.start.offset;
1768             int end = token.location.end.offset;
1769             if (start < sourceString.length() &&
1770                 (sourceString.substring(
1771                     start, Math.min(sourceString.length(), end)).contains("\\v"))) {
1772               n.putBooleanProp(Node.SLASH_V, true);
1773             }
1774           }
1775           return n;
1776         }

1843           Node node = newNode(Token.TRY,
1844               transformBlock(statementNode.body));
1845           Node block = newNode(Token.BLOCK);
1846           boolean lineSet = false;

1849           if (cc != null) {

1851             // clause.

1860           if (finallyBlock != null) {

2401           boolean seenOptional = false;
2402           boolean good = true;
2403           for (int i = 0; i < params.size(); i++) {
2404             ParseTree param = params.get(i);

2406             if (param.type == ParseTreeType.TYPED_PARAMETER) {
2407               TypedParameterTree typedParam = param.asTypedParameter();

2825       String normalizeString(LiteralToken token, boolean templateLiteral) {
2826         String value = token.value;
2827         if (templateLiteral) {
2828           // <CR><LF> and <CR> are normalized as <LF> for raw string value
2829           value = value.replaceAll("\r\n?", "\n");
2830         }
2831         int start = templateLiteral ? 0 : 1; // skip the leading quote
2832         int cur = value.indexOf('\\');
2833         if (cur == -1) {
2834           // short circuit no escapes.
2835           return templateLiteral ? value : value.substring(1, value.length() - 1);
2836         }
2837         StringBuilder result = new StringBuilder();
2838         while (cur != -1) {
2839           if (cur - start > 0) {
2840             result.append(value, start, cur);
2841           }
2842           cur += 1; // skip the escape char.
2843           char c = value.charAt(cur);
2844           switch (c) {
2845             case '\'':
2846             case '"':
2847             case '\\':
2848               result.append(c);
2849               break;
2850             case 'b':
2851               result.append('\b');
2852               break;
2853             case 'f':
2854               result.append('\f');
2855               break;
2856             case 'n':
2857               result.append('\n');
2858               break;
2859             case 'r':
2860               result.append('\r');
2861               break;
2862             case 't':
2863               result.append('\t');
2864               break;
2865             case 'v':
2866               result.append('\u000B');
2867               break;
2868             case '\n':
2869               features = features.require(Feature.STRING_CONTINUATION);
2870               if (isEs5OrBetterMode()) {
2871                 errorReporter.warning(STRING_CONTINUATION_WARNING,
2872                     sourceName,
2873                     lineno(token.location.start), charno(token.location.start));
2874               } else {
2875                 errorReporter.error(STRING_CONTINUATION_ERROR,
2876                     sourceName,
2877                     lineno(token.location.start), charno(token.location.start));
2878               }
2879               // line continuation, skip the line break
2880               break;
2881             case '0': case '1': case '2': case '3': case '4': case '5': case '6': case '7':
2882               char next1 = value.charAt(cur + 1);
2883     
2884               if (inStrictContext()) {
2885                 if (c == '0' && !isOctalDigit(next1)) {
2886                   // No warning: "\0" followed by a character which is not an octal digit
2887                   // is allowed in strict mode.
2888                 } else {
2889                   errorReporter.warning(OCTAL_STRING_LITERAL_WARNING,
2890                       sourceName,
2891                       lineno(token.location.start), charno(token.location.start));
2892                 }
2893               }
2894     
2895               if (!isOctalDigit(next1)) {
2896                 result.append((char) octaldigit(c));
2897               } else {
2898                 char next2 = value.charAt(cur + 2);
2899                 if (!isOctalDigit(next2)) {
2900                   result.append((char) (8 * octaldigit(c) + octaldigit(next1)));
2901                   cur += 1;
2902                 } else {
2903                   result.append((char)
2904                       (8 * 8 * octaldigit(c) + 8 * octaldigit(next1) + octaldigit(next2)));
2905                   cur += 2;
2906                 }
2907               }
2908     
2909               break;
2910             case 'x':
2911               result.append((char) (
2912                   hexdigit(value.charAt(cur + 1)) * 0x10
2913                   + hexdigit(value.charAt(cur + 2))));
2914               cur += 2;
2915               break;
2916             case 'u':
2917               int escapeEnd;
2918               String hexDigits;
2919               if (value.charAt(cur + 1) != '{') {
2920                 // Simple escape with exactly four hex digits: \\uXXXX
2921                 escapeEnd = cur + 5;
2922                 hexDigits = value.substring(cur + 1, escapeEnd);
2923               } else {
2924                 // Escape with braces can have any number of hex digits: \\u{XXXXXXX}
2925                 escapeEnd = cur + 2;
2926                 while (Character.digit(value.charAt(escapeEnd), 0x10) >= 0) {
2927                   escapeEnd++;
2928                 }
2929                 hexDigits = value.substring(cur + 2, escapeEnd);
2930                 escapeEnd++;
2931               }
2932               result.append(Character.toChars(Integer.parseInt(hexDigits, 0x10)));
2933               cur = escapeEnd - 1;
2934               break;
2935             default:
2936               // TODO(tbreisacher): Add a warning because the user probably
2937               // intended to type an escape sequence.
2938               result.append(c);
2939               break;
2940           }
2941           start = cur + 1;
2942           cur = value.indexOf('\\', start);
2943         }
2944         // skip the trailing quote.
2945         result.append(value, start, templateLiteral ? value.length() : value.length() - 1);
2946     
2947         return result.toString();
2948       }
