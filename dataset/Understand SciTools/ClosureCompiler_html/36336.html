17     package com.google.javascript.jscomp.newtypes;
18     
19     import com.google.common.annotations.GwtIncompatible;
20     
21     import java.lang.reflect.Method;
22     import java.util.Map;
23     import java.util.Set;
24     
25     /** A persistent map that simply wraps Clojure's implementation */
26     @GwtIncompatible("java.lang.reflect")
27     final class ClojurePersistentHashMap<K, V> extends PersistentMap<K, V> {
28       private static Method assoc;
29       private static Method without;
30       private final Map map;
31     
32       private ClojurePersistentHashMap(Map m) {
33         this.map = m;
34       }
35     
36       public static <K, V> PersistentMap<K, V> create(Class<? extends Map> cls)  {
37         try {
38           assoc = cls.getDeclaredMethod("assoc", Object.class, Object.class);
39           without = cls.getDeclaredMethod("without", Object.class);
40           Map m = (Map) cls.getDeclaredField("EMPTY").get(null);
41           return new ClojurePersistentHashMap<>(m);
42         } catch (ReflectiveOperationException e) {
43           throw new RuntimeException(e);
44         }
45       }
46     
47       public PersistentMap<K, V> with(K key, V value) {
48         try {
49           Map m = (Map) assoc.invoke(this.map, key, value);
50           return new ClojurePersistentHashMap<>(m);
51         } catch (ReflectiveOperationException e) {
52           throw new RuntimeException(e);
53         }
54       }
55     
56       public PersistentMap<K, V> without(K key) {
57         try {
58           Map m = (Map) without.invoke(this.map, key);
59           return new ClojurePersistentHashMap<>(m);
60         } catch (ReflectiveOperationException e) {
61           throw new RuntimeException(e);
62         }
63       }
64     
65       @Override
66       @SuppressWarnings("unchecked")
67       public V get(Object key) {
68         return (V) this.map.get(key);
69       }
70     
71       @Override
72       @SuppressWarnings("unchecked")
73       public Set<K> keySet() {
74         return this.map.keySet();
75       }
76     
77       @Override
78       public boolean containsKey(Object key) {
79         return this.map.containsKey(key);
80       }
81     
82       @Override
83       @SuppressWarnings("unchecked")
84       public Set<Map.Entry<K, V>> entrySet() {
85         return this.map.entrySet();
86       }
87     
88       @Override
89       public int size() {
90         return this.map.size();
91       }
92     
93       @Override
94       public boolean isEmpty() {
95         return this.map.isEmpty();
96       }
97     
98       @Override
99       public boolean equals(Object o) {
100         if (o instanceof ClojurePersistentHashMap) {
101           ClojurePersistentHashMap pm = (ClojurePersistentHashMap) o;
102           return this.map.equals(pm.map);
103         }
104         return false;
105       }
106     
107       @Override
108       public int hashCode() {
109         return this.map.hashCode();
110       }
111     }
