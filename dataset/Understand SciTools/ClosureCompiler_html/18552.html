129     /**
130      * Parses a javascript file.
131      *
132      * <p>The various parseX() methods never return null - even when parse errors are encountered.
133      * Typically parseX() will return a XTree ParseTree. Each ParseTree that is created includes its
134      * source location. The typical pattern for a parseX() method is:
135      *
136      * <pre>
137      * XTree parseX() {
138      *   SourcePosition start = getTreeStartLocation();
139      *   parse X grammar element and its children
140      *   return new XTree(getTreeLocation(start), children);
141      * }
142      * </pre>
143      *
144      * <p>parseX() methods must consume at least 1 token - even in error cases. This prevents infinite
145      * loops in the parser.
146      *
147      * <p>Many parseX() methods are matched by a 'boolean peekX()' method which will return true if
148      * the beginning of an X appears at the current location. There are also peek() methods which
149      * examine the next token. peek() methods must not consume any tokens.
150      *
151      * <p>The eat() method consumes a token and reports an error if the consumed token is not of the
152      * expected type. The eatOpt() methods consume the next token iff the next token is of the expected
153      * type and return the consumed token or null if no token was consumed.
154      *
155      * <p>When parse errors are encountered, an error should be reported and the parse should return a
156      * best guess at the current parse tree.
157      *
158      * <p>When parsing lists, the preferred pattern is:
159      * <pre>
160      *   eat(LIST_START);
161      *   ImmutableList.Builder&lt;ParseTree&gt; elements = ImmutableList.builder();
162      *   while (peekListElement()) {
163      *     elements.add(parseListElement());
164      *   }
165      *   eat(LIST_END);
166      * </pre>
167      */

260       // 14 Program
261       public ProgramTree parseProgram() {
262         Timer t = new Timer("Parse Program");
263         try {
264           SourcePosition start = getTreeStartLocation();
265           ImmutableList<ParseTree> sourceElements = parseGlobalSourceElements();
266           eat(TokenType.END_OF_FILE);
267           t.end();
268           return new ProgramTree(
269               getTreeLocation(start), sourceElements, commentRecorder.getComments());
270         } catch (StackOverflowError e) {
271           reportError("Too deep recursion while parsing");
272           return null;
273         }
274       }

362       private ParseTree parseImportDeclaration() {
363         SourcePosition start = getTreeStartLocation();
364         eat(TokenType.IMPORT);
365     
366         // import ModuleSpecifier ;
367         if (peek(TokenType.STRING)) {
368           LiteralToken moduleSpecifier = eat(TokenType.STRING).asLiteral();
369           eatPossibleImplicitSemiColon();
370     
371           return new ImportDeclarationTree(
372               getTreeLocation(start), null, null, null, moduleSpecifier);
373         }
374     
375         // import ImportedDefaultBinding from ModuleSpecifier
376         // import NameSpaceImport from ModuleSpecifier
377         // import NamedImports from ModuleSpecifier ;
378         // import ImportedDefaultBinding , NameSpaceImport from ModuleSpecifier ;
379         // import ImportedDefaultBinding , NamedImports from ModuleSpecifier ;
380         IdentifierToken defaultBindingIdentifier = null;
381         IdentifierToken nameSpaceImportIdentifier = null;
382         ImmutableList<ParseTree> identifierSet = null;
383     
384         boolean parseExplicitNames = true;
385         if (peekId()) {
386           defaultBindingIdentifier = eatId();
387           if (peek(TokenType.COMMA)) {
388             eat(TokenType.COMMA);
389           } else {
390             parseExplicitNames = false;
391           }
392         } else if (Keywords.isKeyword(peekType())) {
393             Token keyword = nextToken();
394             reportError(keyword, "cannot use keyword '%s' here.", keyword);
395         }
396     
397         if (parseExplicitNames) {
398           if (peek(TokenType.STAR)) {
399             eat(TokenType.STAR);
400             eatPredefinedString(PredefinedName.AS);
401             nameSpaceImportIdentifier = eatId();
402           } else {
403             identifierSet = parseImportSpecifierSet();
404           }
405         }
406     
407         eatPredefinedString(PredefinedName.FROM);
408         Token moduleStr = eat(TokenType.STRING);
409         LiteralToken moduleSpecifier = (moduleStr == null)
410             ? null : moduleStr.asLiteral();
411         eatPossibleImplicitSemiColon();
412     
413         return new ImportDeclarationTree(
414             getTreeLocation(start),
415             defaultBindingIdentifier, identifierSet, nameSpaceImportIdentifier, moduleSpecifier);
416       }

418       //  ImportSpecifierSet ::= '{' (ImportSpecifier (',' ImportSpecifier)* (,)? )?  '}'
419       private ImmutableList<ParseTree> parseImportSpecifierSet() {
420         ImmutableList.Builder<ParseTree> elements;
421         elements = ImmutableList.builder();
422         eat(TokenType.OPEN_CURLY);
423         while (peekIdOrKeyword()) {
424           elements.add(parseImportSpecifier());
425           if (!peek(TokenType.CLOSE_CURLY)) {

427           }
428         }
429         eat(TokenType.CLOSE_CURLY);
430         return elements.build();

458       /*
459       ExportDeclaration :
460         export * FromClause ;
461         export ExportClause [NoReference] FromClause ;
462         export ExportClause ;
463         export VariableStatement
464         export Declaration[Default]
465         export default AssignmentExpression ;
466       ExportClause [NoReference] :
467         { }
468         { ExportsList [?NoReference] }
469         { ExportsList [?NoReference] , }
470       ExportsList [NoReference] :
471         ExportSpecifier [?NoReference]
472         ExportsList [?NoReference] , ExportSpecifier [?NoReference]
473       ExportSpecifier [NoReference] :
474         [~NoReference] IdentifierReference
475         [~NoReference] IdentifierReference as IdentifierName
476         [+NoReference] IdentifierName
477         [+NoReference] IdentifierName as IdentifierName
478        */
479       private ParseTree parseExportDeclaration(boolean isAmbient) {
480         SourcePosition start = getTreeStartLocation();
481         boolean isDefault = false;
482         boolean isExportAll = false;
483         boolean isExportSpecifier = false;
484         boolean needsSemiColon = true;
485         eat(TokenType.EXPORT);
486         ParseTree export = null;
487         ImmutableList<ParseTree> exportSpecifierList = null;
488         switch (peekType()) {
489           case STAR:
490             isExportAll = true;
491             nextToken();
492             break;
493           case FUNCTION:
494             // TODO(bradfordcsmith): handle async functions here
495             export = isAmbient ? parseAmbientFunctionDeclaration() : parseFunctionDeclaration();
496             needsSemiColon = isAmbient;
497             break;
498           case CLASS:
499             export = parseClassDeclaration(isAmbient);
500             needsSemiColon = false;
501             break;
502           case INTERFACE:
503             export = parseInterfaceDeclaration();
504             needsSemiColon = false;
505             break;
506           case ENUM:
507             export = parseEnumDeclaration();
508             needsSemiColon = false;
509             break;
510           case MODULE:
511           case NAMESPACE:
512             export = parseNamespaceDeclaration(isAmbient);
513             needsSemiColon = false;
514             break;
515           case DECLARE:
516             export = parseAmbientDeclaration();
517             needsSemiColon = false;
518             break;
519           case DEFAULT:
520             isDefault = true;
521             nextToken();
522             export = parseExpression();
523             needsSemiColon = false;
524             break;
525           case OPEN_CURLY:
526             isExportSpecifier = true;
527             exportSpecifierList = parseExportSpecifierSet();
528             break;
529           case TYPE:
530             export = parseTypeAlias();
531             break;
532           default: // unreachable, parse as a var decl to get a parse error.
533           case VAR:
534           case LET:
535           case CONST:
536             export = isAmbient ? parseAmbientVariableDeclarationList() : parseVariableDeclarationList();
537             break;
538         }
539     
540         LiteralToken moduleSpecifier = null;
541         if (isExportAll ||
542             (isExportSpecifier && peekPredefinedString(PredefinedName.FROM))) {
543           eatPredefinedString(PredefinedName.FROM);
544           moduleSpecifier = eat(TokenType.STRING).asLiteral();
545         } else if (isExportSpecifier) {
546           for (ParseTree tree : exportSpecifierList) {
547             IdentifierToken importedName = tree.asExportSpecifier().importedName;
548             if (Keywords.isKeyword(importedName.value)) {
549               reportError(importedName, "cannot use keyword '%s' here.", importedName.value);
550             }
551           }
552         }
553     
554         if (needsSemiColon || peekImplicitSemiColon()) {
555           eatPossibleImplicitSemiColon();
556         }
557     
558         return new ExportDeclarationTree(
559             getTreeLocation(start), isDefault, isExportAll,
560             export, exportSpecifierList, moduleSpecifier);
561       }

616       private ParseTree parseInterfaceDeclaration() {
617         SourcePosition start = getTreeStartLocation();
618         eat(TokenType.INTERFACE);
619         IdentifierToken name = eatId();
620         GenericTypeListTree generics = maybeParseGenericTypes();
621         ImmutableList.Builder<ParseTree> superTypes = ImmutableList.builder();
622         if (peek(TokenType.EXTENDS)) {
623           eat(TokenType.EXTENDS);
624           ParseTree type = parseType();
625           superTypes.add(type);
626     
627           while (peek(TokenType.COMMA)) {
628             eat(TokenType.COMMA);
629             type = parseType();
630             if (type != null) {
631               superTypes.add(type);
632             }
633           }
634         }
635         eat(TokenType.OPEN_CURLY);
636         ImmutableList<ParseTree> elements = parseInterfaceElements();
637         eat(TokenType.CLOSE_CURLY);
638         return new InterfaceDeclarationTree(getTreeLocation(start), name, generics,
639             superTypes.build(), elements);
640       }

642       private ImmutableList<ParseTree> parseInterfaceElements() {
643         ImmutableList.Builder<ParseTree> result = ImmutableList.builder();
644     
645         while (peekInterfaceElement()) {
646           result.add(parseInterfaceElement());
647           if (!peek(TokenType.CLOSE_CURLY)) {
648             // The standard delimiter is semicolon, but we also accept comma
649             if (peekImplicitSemiColon()) {
650               eatPossibleImplicitSemiColon();
651             } else {
652               eat(TokenType.COMMA);
653             }
654           }
655         }
656     
657         return result.build();
658       }

664           case IDENTIFIER:
665           case OPEN_SQUARE:
666           case STAR:

675       private ParseTree parseInterfaceElement() {
676         SourcePosition start = getTreeStartLocation();
677     
678         boolean isGenerator = eatOpt(TokenType.STAR) != null;
679     
680         IdentifierToken name = null;
681         TokenType type = peekType();
682     
683         if (type == TokenType.NEW) {
684           return parseCallSignature(true); // ConstructSignature
685         } else if (type == TokenType.IDENTIFIER || Keywords.isKeyword(type)) {
686           name = eatIdOrKeywordAsId();
687         } else if (type == TokenType.OPEN_SQUARE) { // IndexSignature
688           return parseIndexSignature();
689         } else if (type == TokenType.OPEN_ANGLE || type == TokenType.OPEN_PAREN) { // CallSignature
690           return parseCallSignature(false);
691         }
692     
693         boolean isOptional = false;
694         if (peek(TokenType.QUESTION)) {
695           eat(TokenType.QUESTION);
696           isOptional = true;
697         }
698     
699         if (peek(TokenType.OPEN_PAREN) || peek(TokenType.OPEN_ANGLE)) {
700           // Method signature.
701           ParseTree function = parseMethodSignature(
702               start, name, false, isGenerator, isOptional, null);
703           return function;
704         } else {
705           // Property signature.
706           ParseTree declaredType = maybeParseColonType();
707           return new MemberVariableTree(
708               getTreeLocation(start), name, false, isOptional, null, declaredType);
709         }
710       }

740         SourcePosition start = getTreeStartLocation();
741         eat(TokenType.CLASS);
742         IdentifierToken name = null;
743         if (!isExpression || peekId()) {
744           name = eatId();
745         }
746     
747         GenericTypeListTree generics = maybeParseGenericTypes();
748         ParseTree superClass = null;
749         if (peek(TokenType.EXTENDS)) {
750           eat(TokenType.EXTENDS);
751           superClass = parseExpression();
752         }
753     
754         ImmutableList.Builder<ParseTree> interfaces = ImmutableList.builder();
755         if (config.parseTypeSyntax && peek(TokenType.IMPLEMENTS)) {
756           eat(TokenType.IMPLEMENTS);
757           ParseTree type = parseType();
758           interfaces.add(type);
759     
760           while (peek(TokenType.COMMA)) {
761             eat(TokenType.COMMA);
762             type = parseType();
763             if (type != null) {
764               interfaces.add(type);
765             }
766           }
767         }
768     
769         eat(TokenType.OPEN_CURLY);
770         ImmutableList<ParseTree> elements = parseClassElements(isAmbient);
771         eat(TokenType.CLOSE_CURLY);
772         return new ClassDeclarationTree(getTreeLocation(start), name, generics,
773             superClass, interfaces.build(), elements);
774       }

815       private ParseTree parseClassMemberDeclaration(
816           boolean allowStatic, boolean isAmbient, TokenType access) {
817         SourcePosition start = getTreeStartLocation();
818         boolean isStatic = false;
819         if (allowStatic && peek(TokenType.STATIC) && peekType(1) != TokenType.OPEN_PAREN) {
820           eat(TokenType.STATIC);
821           isStatic = true;
822         }
823         boolean isGenerator = eatOpt(TokenType.STAR) != null;
824     
825         ParseTree nameExpr;
826         IdentifierToken name;
827         if (peekIdOrKeyword()) {
828           nameExpr = null;
829           name = eatIdOrKeywordAsId();
830         } else {
831           if (config.parseTypeSyntax && peekIndexSignature()) {
832             ParseTree indexSignature = parseIndexSignature();
833             eatPossibleImplicitSemiColon();
834             return indexSignature;
835           }
836           nameExpr = parseComputedPropertyName();
837           name = null;
838         }
839     
840         if (peek(TokenType.OPEN_PAREN) || peek(TokenType.OPEN_ANGLE)) {
841           // Member function.
842           FunctionDeclarationTree.Kind kind;
843           TokenType accessOnFunction;
844           if (nameExpr == null) {
845             kind = FunctionDeclarationTree.Kind.MEMBER;
846             accessOnFunction = access;
847           } else {
848             kind = FunctionDeclarationTree.Kind.EXPRESSION;
849             accessOnFunction = null; // Accessibility modifier goes on the ComputedPropertyMethodTree
850           }
851     
852           ParseTree function;
853           if (isAmbient) {
854             function = parseMethodSignature(
855                 start, name, isStatic, isGenerator, false, accessOnFunction);
856             eatPossibleImplicitSemiColon();
857           } else {
858             FunctionDeclarationTree.Builder builder =
859                 FunctionDeclarationTree.builder(kind)
860                     .setName(name)
861                     .setStatic(isStatic)
862                     .setAccess(accessOnFunction);
863             parseFunctionTail(builder, isGenerator);
864     
865             function = builder.build(getTreeLocation(start));
866           }
867           if (kind == FunctionDeclarationTree.Kind.MEMBER) {
868             return function;
869           } else {
870             return new ComputedPropertyMethodTree(
871                 getTreeLocation(start), access, nameExpr, function);
872           }
873         } else {
874           // Member variable.
875           if (isGenerator) {
876             reportError("Member variable cannot be prefixed by '*' (generator function)");
877           }
878           ParseTree declaredType = maybeParseColonType();
879           if (peek(TokenType.EQUAL)) {
880             reportError("Member variable initializers ('=') are not supported");
881           }
882           eatPossibleImplicitSemiColon();
883           if (nameExpr == null) {
884             return new MemberVariableTree(
885                 getTreeLocation(start), name, isStatic, false, access, declaredType);
886           } else {
887             return new ComputedPropertyMemberVariableTree(getTreeLocation(start),
888                 nameExpr, isStatic, access, declaredType);
889           }
890         }
891       }

966         if (peekAsyncFunctionStart()) {
967           return parseAsyncFunctionDeclaration();

1176       private boolean peekParameter(ParamContext context) {
1177         if (peekId() || peek(TokenType.SPREAD)) {
1178           return true;
1179         }
1180         if (context != ParamContext.TYPE_EXPRESSION) {
1181           return peek(TokenType.OPEN_SQUARE) || peek(TokenType.OPEN_CURLY);
1182         }
1183         return false;
1184       }

1186       private ParseTree parseParameter(ParamContext context) {
1187         SourcePosition start = getTreeStartLocation();
1188         ParseTree parameter = null;
1189     
1190         if (peek(TokenType.SPREAD)) {
1191           parameter = parseRestParameter();
1192         } else if (peekId()) {
1193           parameter = parseIdentifierExpression();
1194           if (peek(TokenType.QUESTION)) {
1195             eat(TokenType.QUESTION);
1196             parameter = new OptionalParameterTree(getTreeLocation(start), parameter);
1197           }
1198         } else if (context != ParamContext.TYPE_EXPRESSION && peekPatternStart()) {
1199           parameter = parsePattern(PatternKind.INITIALIZER);
1200         } else {
1201           throw new IllegalStateException(
1202               "parseParameterCalled() without confirming a parameter exists.");
1203         }
1204     
1205         ParseTree typeAnnotation = null;
1206         SourceRange typeLocation = null;
1207         if (peek(TokenType.COLON)) {
1208           if (peek(1, TokenType.STRING)) {
1209             eat(TokenType.COLON);
1210             typeAnnotation = parseLiteralExpression(); // Specialized Signature
1211           } else {
1212             typeAnnotation = parseTypeAnnotation();
1213           }
1214           typeLocation = getTreeLocation(getTreeStartLocation());
1215         }
1216     
1217         if (context == ParamContext.IMPLEMENTATION
1218             && !parameter.isRestParameter()
1219             && peek(TokenType.EQUAL)) {
1220           features = features.require(Feature.DEFAULT_PARAMETERS);
1221           eat(TokenType.EQUAL);
1222           ParseTree defaultValue = parseAssignmentExpression();
1223           parameter = new DefaultParameterTree(getTreeLocation(start), parameter, defaultValue);
1224         }
1225     
1226         if (typeAnnotation != null) {
1227           // Must be a direct child of the parameter list.
1228           parameter = new TypedParameterTree(typeLocation, parameter, typeAnnotation);
1229         }
1230     
1231         return parameter;
1232       }

1242       private FormalParameterListTree parseFormalParameterList(ParamContext context) {
1243         SourcePosition listStart = getTreeStartLocation();
1244         eat(TokenType.OPEN_PAREN);
1245     
1246         ImmutableList.Builder<ParseTree> result = ImmutableList.builder();
1247     
1248         while (peekParameter(context)) {
1249           result.add(parseParameter(context));
1250     
1251           if (!peek(TokenType.CLOSE_PAREN)) {
1252             Token comma = eat(TokenType.COMMA);
1253             if (peek(TokenType.CLOSE_PAREN)) {
1254               reportError(comma, "Invalid trailing comma in formal parameter list");
1255             }
1256           }
1257         }
1258     
1259         eat(TokenType.CLOSE_PAREN);
1260     
1261         return new FormalParameterListTree(
1262             getTreeLocation(listStart), result.build());
1263       }

1270       private ParseTree parseType() {
1271         SourcePosition start = getTreeStartLocation();
1272         if (!peekId() && !EnumSet.of(TokenType.VOID, TokenType.OPEN_PAREN, TokenType.OPEN_CURLY,
1273               TokenType.TYPEOF).contains(peekType())) {
1274           reportError("Unexpected token '%s' in type expression", peekType());
1275           return new TypeNameTree(getTreeLocation(start), ImmutableList.of("error"));
1276         }
1277     
1278         ParseTree typeExpression = parseFunctionTypeExpression();
1279         if (!peek(TokenType.BAR)) {
1280           return typeExpression;
1281         }
1282         ImmutableList.Builder<ParseTree> unionType = ImmutableList.builder();

1343       private ParseTree parseTypeQuery() {
1344         SourcePosition start = getTreeStartLocation();
1345         if (peek(TokenType.TYPEOF)) {
1346           eat(TokenType.TYPEOF);
1347     
1348           IdentifierToken token = eatId();
1349           ImmutableList.Builder<String> identifiers = ImmutableList.builder();
1350           if (token != null) {
1351             identifiers.add(token.value);
1352           }
1353           while (peek(TokenType.PERIOD)) {
1354             // TypeQueryExpression . IdentifierName
1355             eat(TokenType.PERIOD);
1356             token = eatId();
1357             if (token == null) {
1358               break;
1359             }
1360             identifiers.add(token.value);
1361           }
1362           return new TypeQueryTree(getTreeLocation(start), identifiers.build());
1363         } else {
1364           return parseTypeReference();
1365         }

1380         // < TypeArgumentList >
1381         // TypeArgumentList , TypeArgument
1382         eat(TokenType.OPEN_ANGLE);
1383         scanner.incTypeParameterLevel();

1385         ParseTree type = parseType();

1609       private VariableDeclarationListTree parseVariableDeclarationList(
1610           Expression expressionIn) {
1611         SourcePosition start = getTreeStartLocation();
1612         TokenType token = peekType();
1613     
1614         switch (token) {
1615         case CONST:
1616         case LET:
1617         case VAR:
1618           eat(token);
1619           break;
1620         default:
1621           reportError(peekToken(), "expected declaration");
1622           return null;
1623         }

1625         ImmutableList.Builder<VariableDeclarationTree> declarations =
1626             ImmutableList.builder();
1627     
1628         declarations.add(parseVariableDeclaration(token, expressionIn));
1629         while (peek(TokenType.COMMA)) {
1630           eat(TokenType.COMMA);
1631           declarations.add(parseVariableDeclaration(token, expressionIn));
1632         }
1633         return new VariableDeclarationListTree(
1634             getTreeLocation(start), token, declarations.build());

1637       private VariableDeclarationTree parseVariableDeclaration(
1638           final TokenType binding, Expression expressionIn) {
1639     
1640         SourcePosition start = getTreeStartLocation();
1641         ParseTree lvalue;
1642         ParseTree typeAnnotation = null;
1643         if (peekPatternStart()) {
1644           lvalue = parsePattern(PatternKind.INITIALIZER);
1645         } else {
1646           lvalue = parseIdentifierExpression();
1647           if (peek(TokenType.COLON)) {
1648             typeAnnotation = parseTypeAnnotation();
1649           }
1650         }
1651     
1652         ParseTree initializer = null;
1653         if (peek(TokenType.EQUAL)) {
1654           initializer = parseInitializer(expressionIn);
1655         } else if (expressionIn != Expression.NO_IN) {
1656           // NOTE(blickly): this is a bit of a hack, declarations outside of for statements allow "in",
1657           // and by chance, also must have initializers for const/destructuring. Vanilla for loops
1658           // also require intializers, but are handled separately in checkVanillaForInitializers
1659           maybeReportNoInitializer(binding, lvalue);
1660         }
1661         return new VariableDeclarationTree(getTreeLocation(start), lvalue, typeAnnotation, initializer);
1662       }

1729       // 12.6.3 The for Statement
1730       // 12.6.4 The for-in Statement
1731       // The for-of Statement
1732       private ParseTree parseForStatement() {
1733         SourcePosition start = getTreeStartLocation();
1734         eat(TokenType.FOR);
1735         eat(TokenType.OPEN_PAREN);
1736         if (peekVariableDeclarationList()) {
1737           VariableDeclarationListTree variables = parseVariableDeclarationListNoIn();
1738           if (peek(TokenType.IN)) {
1739             // for-in: only one declaration allowed
1740             if (variables.declarations.size() > 1) {
1741               reportError("for-in statement may not have more than one variable declaration");
1742             }
1743             VariableDeclarationTree declaration = variables.declarations.get(0);
1744             if (declaration.initializer != null) {
1745               // An initializer is allowed here in ES5 and below, but not in ES6.
1746               // Warn about it, to encourage people to eliminate it from their code.
1747               // http://esdiscuss.org/topic/initializer-expression-on-for-in-syntax-subject
1748               if (config.atLeast6) {
1749                 reportError("for-in statement may not have initializer");
1750               } else {
1751                 errorReporter.reportWarning(declaration.location.start,
1752                     "for-in statement should not have initializer");
1753               }
1754             }
1755     
1756             return parseForInStatement(start, variables);
1757           } else if (peekPredefinedString(PredefinedName.OF)) {
1758             // for-of: only one declaration allowed
1759             if (variables.declarations.size() > 1) {
1760               reportError("for-of statement may not have more than one variable declaration");
1761             }
1762             // for-of: initializer is illegal
1763             VariableDeclarationTree declaration = variables.declarations.get(0);
1764             if (declaration.initializer != null) {
1765               reportError("for-of statement may not have initializer");
1766             }
1767     
1768             return parseForOfStatement(start, variables);
1769           } else {
1770             // "Vanilla" for statement: const/destructuring must have initializer
1771             checkVanillaForInitializers(variables);
1772             return parseForStatement(start, variables);
1773           }
1774         }
1775     
1776         if (peek(TokenType.SEMI_COLON)) {
1777           return parseForStatement(start, null);
1778         }
1779     
1780         ParseTree initializer = parseExpressionNoIn();
1781         if (peek(TokenType.IN) || peek(TokenType.EQUAL) || peekPredefinedString(PredefinedName.OF)) {
1782           initializer = transformLeftHandSideExpression(initializer);
1783         }
1784     
1785         if (peek(TokenType.IN) || peekPredefinedString(PredefinedName.OF)) {
1786           if (initializer.type != ParseTreeType.BINARY_OPERATOR
1787               && initializer.type != ParseTreeType.COMMA_EXPRESSION) {
1788             if (peek(TokenType.IN)) {
1789               return parseForInStatement(start, initializer);
1790             } else {
1791               return parseForOfStatement(start, initializer);
1792             }
1793           }
1794         }
1795     
1796         return parseForStatement(start, initializer);
1797       }

1937       private ImmutableList<ParseTree> parseCaseClauses() {
1938         boolean foundDefaultClause = false;
1939         ImmutableList.Builder<ParseTree> result = ImmutableList.builder();
1940     
1941         while (true) {
1942           SourcePosition start = getTreeStartLocation();
1943           switch (peekType()) {
1944           case CASE:
1945             eat(TokenType.CASE);
1946             ParseTree expression = parseExpression();
1947             eat(TokenType.COLON);
1948             ImmutableList<ParseTree> statements = parseCaseStatementsOpt();
1949             result.add(new CaseClauseTree(getTreeLocation(start), expression, statements));
1950             break;
1951           case DEFAULT:
1952             if (foundDefaultClause) {
1953               reportError("Switch statements may have at most one default clause");
1954             } else {
1955               foundDefaultClause = true;
1956             }
1957             eat(TokenType.DEFAULT);
1958             eat(TokenType.COLON);
1959             result.add(new DefaultClauseTree(getTreeLocation(start), parseCaseStatementsOpt()));
1960             break;
1961           default:
1962             return result.build();
1963           }
1964         }

2113        * Constructs a template literal expression tree. "operand" is used to handle
2114        * the case like "foo`bar`", which is a CallExpression or MemberExpression that
2115        * calls the function foo() with the template literal as the argument (with extra
2116        * handling). In this case, operand would be "foo", which is the callsite.
2117        *
2118        * <p>We store this operand in the TemplateLiteralExpressionTree and
2119        * generate a TAGGED_TEMPLATELIT node if it's not null later when transpiling.
2120        *
2121        * @param operand A non-null value would represent the callsite
2122        * @return The template literal expression
2123        */

2125         SourcePosition start = operand == null
2126             ? getTreeStartLocation()
2127             : operand.location.start;
2128         Token token = nextToken();
2129         ImmutableList.Builder<ParseTree> elements = ImmutableList.builder();
2130         elements.add(new TemplateLiteralPortionTree(token.location, token));
2131         if (token.type == TokenType.NO_SUBSTITUTION_TEMPLATE) {
2132           return new TemplateLiteralExpressionTree(
2133               getTreeLocation(start), operand, elements.build());

2136         // `abc${
2137         ParseTree expression = parseExpression();
2138         elements.add(new TemplateSubstitutionTree(expression.location, expression));

2147             break;
2148           }
2149     
2150           expression = parseExpression();
2151           elements.add(new TemplateSubstitutionTree(expression.location, expression));

2337       private ParseTree parsePropertyAssignment() {
2338         TokenType type = peekType();
2339         if (type == TokenType.STAR) {

2510       //   CoverParenthesizedExpressionAndArrowParameterList ::=
2511       //     ( Expression )
2512       //     ( )
2513       //     ( ... BindingIdentifier )
2514       //     ( Expression , ... BindingIdentifier )
2515       private ParseTree parseCoverParenthesizedExpressionAndArrowParameterList() {
2516         if (peekType(1) == TokenType.FOR) {
2517           return parseGeneratorComprehension();
2518         }
2519     
2520         SourcePosition start = getTreeStartLocation();
2521         eat(TokenType.OPEN_PAREN);
2522         // Case ( )
2523         if (peek(TokenType.CLOSE_PAREN)) {
2524           eat(TokenType.CLOSE_PAREN);
2525           return new FormalParameterListTree(getTreeLocation(start), ImmutableList.<ParseTree>of());
2526         }

2528         if (peek(TokenType.SPREAD)) {
2529           ParseTree result = new FormalParameterListTree(
2530               getTreeLocation(start), ImmutableList.of(parseParameter(ParamContext.IMPLEMENTATION)));
2531           eat(TokenType.CLOSE_PAREN);
2532           return result;
2533         }
2534         // For either of the two remaining cases:
2535         //     ( Expression )
2536         //     ( Expression , ... BindingIdentifier )
2537         // we can parse as an expression.
2538         ParseTree result = parseExpression();
2539         // If it follows witha comma, we must be in the
2540         //     ( Expression , ... BindingIdentifier )
2541         // case.
2542         if (peek(TokenType.COMMA)) {
2543           eat(TokenType.COMMA);
2544           // Since we already parsed as an expression, we will guaranteed reparse this expression
2545           // as an arrow function parameter list, but just leave it as a comma expression for now.
2546           result = new CommaExpressionTree(
2547               getTreeLocation(start),
2548               ImmutableList.of(result, parseParameter(ParamContext.IMPLEMENTATION)));
2549         }

2677       private ParseTree parseAssignment(Expression expressionIn) {
2678         if (peek(TokenType.YIELD) && inGeneratorContext()) {
2679           return parseYield(expressionIn);
2680         }

2682         SourcePosition start = getTreeStartLocation();

2684         ParseTree left = parseConditional(expressionIn);
2685         if (peek(TokenType.ARROW)) {
2686           return completeAssignmentExpressionParseAtArrow(left, expressionIn);
2687         }

2699           return new BinaryOperatorTree(getTreeLocation(start), left, operator, right);

2735       private FormalParameterListTree transformToArrowFormalParameters(ParseTree leftOfArrow) {
2736         FormalParameterListTree arrowParameterList;

2752             // parseFormalParameterList will take care of reporting errors.

2763           CallExpressionTree callExpression, Expression expressionIn) {
2764         ParseTree operand = callExpression.operand;
2765         ParseTree arguments = callExpression.arguments;
2766         ParseTree result;

3379       // Pattern ::= ... | "[" Element? ("," Element?)* "]"
3380       private ParseTree parseArrayPattern(PatternKind kind) {
3381         features = features.require(Feature.DESTRUCTURING);
3382         SourcePosition start = getTreeStartLocation();
3383         ImmutableList.Builder<ParseTree> elements = ImmutableList.builder();
3384         eat(TokenType.OPEN_SQUARE);
3385         while (peek(TokenType.COMMA) || peekArrayPatternElement()) {
3386           if (peek(TokenType.COMMA)) {
3387             eat(TokenType.COMMA);
3388             elements.add(new NullTree(getTreeLocation(getTreeStartLocation())));
3389           } else {
3390             ParseTree element = parseArrayPatternElement(kind);
3391             elements.add(element);
3392     
3393             if (element.isAssignmentRestElement()) {
3394               // Rest can only appear in the posterior, so we must be done
3395               break;
3396             } else if (peek(TokenType.COMMA)) {
3397               // Consume the comma separator
3398               eat(TokenType.COMMA);
3399               if (peek(TokenType.CLOSE_SQUARE)) {
3400                 reportError("Array pattern may not end with a comma");
3401                 break;
3402               }
3403             } else {
3404               // Otherwise we must be done
3405               break;
3406             }
3407           }
3408         }
3409         eat(TokenType.CLOSE_SQUARE);
3410         return new ArrayPatternTree(getTreeLocation(start), elements.build());
3411       }

3413       // Pattern ::= "{" (Field ("," Field)* ","?)? "}" | ...
3414       private ParseTree parseObjectPattern(PatternKind kind) {
3415         features = features.require(Feature.DESTRUCTURING);
3416         SourcePosition start = getTreeStartLocation();
3417         ImmutableList.Builder<ParseTree> fields = ImmutableList.builder();
3418         eat(TokenType.OPEN_CURLY);
3419         while (peekObjectPatternField()) {
3420           fields.add(parseObjectPatternField(kind));
3421           if (peek(TokenType.COMMA)) {
3422             // Consume the comma separator
3423             eat(TokenType.COMMA);
3424           } else {
3425             // Otherwise we must be done
3426             break;
3427           }
3428         }
3429         eat(TokenType.CLOSE_CURLY);
3430         return new ObjectPatternTree(getTreeLocation(start), fields.build());
3431       }

3437       private ParseTree parseObjectPatternField(PatternKind kind) {
3438         SourcePosition start = getTreeStartLocation();
3439         if (peekType() == TokenType.OPEN_SQUARE) {
3440           ParseTree key = parseComputedPropertyName();
3441           eat(TokenType.COLON);
3442           ParseTree value = parsePatternAssignmentTarget(kind);
3443           return new ComputedPropertyDefinitionTree(getTreeLocation(start), key, value);
3444         }

3446         Token name;
3447         if (peekIdOrKeyword()) {
3448           name = eatIdOrKeywordAsId();

3450             IdentifierToken idToken = (IdentifierToken) name;
3451             if (Keywords.isKeyword(idToken.value)
3452                 && !Keywords.isTypeScriptSpecificKeyword(idToken.value)) {
3453               reportError("cannot use keyword '" + name + "' here.");
3454             }
3455             if (peek(TokenType.EQUAL)) {
3456               IdentifierExpressionTree idTree = new IdentifierExpressionTree(
3457                   getTreeLocation(start), idToken);
3458               eat(TokenType.EQUAL);
3459               ParseTree defaultValue = parseAssignmentExpression();
3460               return new DefaultParameterTree(getTreeLocation(start), idTree, defaultValue);
3461             }
3462             return new PropertyNameAssignmentTree(getTreeLocation(start), name, null);
3463           }
3464         } else {

3466         }
3467

3469         ParseTree value = parsePatternAssignmentTarget(kind);
3470         return new PropertyNameAssignmentTree(getTreeLocation(start), name, value);
3471       }

3473       /**
3474        * A PatternAssignmentTarget is the location where the assigned value gets stored, including an
3475        * optional default value.
3476        *
3477        * <dl>
3478        *   <dt> Spec AssignmentElement === PatternAssignmentTarget(PatternKind.ANY)
3479        *   <dd> Valid in an assignment that is not a formal parameter list or variable declaration.
3480        *        Sub-patterns and arbitrary left hand side expressions are allowed.
3481        *
3482        *   <dt> Spec BindingElement === PatternAssignmentElement(PatternKind.INITIALIZER)
3483        *   <dd> Valid in a formal parameter list or variable declaration statement.
3484        *        Only sub-patterns and identifiers are allowed.
3485        * </dl>
3486        * Examples:
3487        * <pre>
3488        *   <code>
3489        *     [a, {foo: b = 'default'}] = someArray;          // valid
3490        *     [x.a, {foo: x.b = 'default'}] = someArray;      // valid
3491        *
3492        *     let [a, {foo: b = 'default'}] = someArray;      // valid
3493        *     let [x.a, {foo: x.b = 'default'}] = someArray;  // invalid
3494        *
3495        *     function f([a, {foo: b = 'default'}]) {...}     // valid
3496        *     function f([x.a, {foo: x.b = 'default'}]) {...} // invalid
3497        *   </code>
3498        * </pre>
3499        */
3500       private ParseTree parsePatternAssignmentTarget(PatternKind patternKind) {
3501         SourcePosition start = getTreeStartLocation();
3502         ParseTree assignmentTarget;
3503         assignmentTarget = parsePatternAssignmentTargetNoDefault(patternKind);
3504     
3505         if (peek(TokenType.EQUAL)) {
3506           eat(TokenType.EQUAL);
3507           ParseTree defaultValue = parseAssignmentExpression();
3508           assignmentTarget =
3509               new DefaultParameterTree(getTreeLocation(start), assignmentTarget, defaultValue);
3510         }
3511         return assignmentTarget;
3512       }

3514       private ParseTree parsePatternAssignmentTargetNoDefault(PatternKind kind) {
3515         ParseTree assignmentTarget;
3516         if (peekPatternStart()) {
3517           assignmentTarget = parsePattern(kind);
3518         } else {

3520           if (!assignmentTarget.isValidAssignmentTarget()) {

3525             // We're in the context of a formal parameter list or a variable declaration statement
3526             reportError("Only an identifier or destructuring pattern is allowed here.");
3527           }
3528         }
3529         return assignmentTarget;
3530       }
