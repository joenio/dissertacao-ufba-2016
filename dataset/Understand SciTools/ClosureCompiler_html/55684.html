232     
233         // Verify JsDoc annotations
234         checks.add(checkJsDoc);
235     
236         if (options.enables(DiagnosticGroups.LINT_CHECKS)) {
237           checks.add(lintChecks);
238         }
239     
240         if (!options.skipNonTranspilationPasses && options.closurePass
241             && options.enables(DiagnosticGroups.LINT_CHECKS)) {
242           checks.add(checkRequiresAndProvidesSorted);
243         }
244     
245         if (!options.skipNonTranspilationPasses && options.closurePass) {
246           checks.add(closureCheckModule);
247           checks.add(closureRewriteModule);
248         }
249     
250         if (!options.skipNonTranspilationPasses && options.declaredGlobalExternsOnWindow) {
251           checks.add(declaredGlobalExternsOnWindow);
252         }
253     
254         if (options.getLanguageIn() == LanguageMode.ECMASCRIPT6_TYPED
255                 && options.getLanguageOut() != LanguageMode.ECMASCRIPT6_TYPED) {
256           checks.add(convertEs6TypedToEs6);
257         }
258     
259         checks.add(checkMissingSuper);
260         checks.add(checkVariableReferences);
261     
262         if (!options.skipNonTranspilationPasses && options.closurePass) {
263           checks.add(closureGoogScopeAliases);
264           checks.add(closureRewriteClass);
265         }
266     
267         if (options.enables(DiagnosticGroups.MISSING_REQUIRE)
268             || options.enables(DiagnosticGroups.EXTRA_REQUIRE)) {
269           checks.add(checkRequires);
270         }
271     
272         if (!options.skipNonTranspilationPasses) {
273           checks.add(checkSideEffects);
274         }
275     
276         if (options.enables(DiagnosticGroups.MISSING_PROVIDE)) {
277           checks.add(checkProvides);
278         }
279     
280         if (options.jqueryPass && !options.skipNonTranspilationPasses) {
281           checks.add(jqueryAliases);
282         }
283     
284         if (options.angularPass && !options.skipNonTranspilationPasses) {
285           checks.add(angularPass);
286         }
287     
288         if (!options.generateExportsAfterTypeChecking
289             && options.generateExports && !options.skipNonTranspilationPasses) {
290           checks.add(generateExports);
291         }
292     
293         if (options.exportTestFunctions && !options.skipNonTranspilationPasses) {
294           checks.add(exportTestFunctions);
295         }
296     
297         if (options.closurePass && !options.skipNonTranspilationPasses) {
298           checks.add(closurePrimitives);
299         }
300     
301         // It's important that the PolymerPass run *after* the ClosurePrimitives
302         // rewrite and *before* the suspicious code checks.
303         // This is enforced in the assertValidOrder method.
304         if (options.polymerPass && !options.skipNonTranspilationPasses) {
305           checks.add(polymerPass);
306         }
307     
308         if ((options.checkSuspiciousCode
309                 || options.enables(DiagnosticGroups.GLOBAL_THIS)
310                 || options.enables(DiagnosticGroups.DEBUGGER_STATEMENT_PRESENT))
311             && !options.skipNonTranspilationPasses) {
312           checks.add(suspiciousCode);
313         }
314     
315         if (options.closurePass && options.checkMissingGetCssNameLevel.isOn()
316             && !options.skipNonTranspilationPasses) {
317           checks.add(closureCheckGetCssName);
318         }
319     
320         if (options.syntheticBlockStartMarker != null && !options.skipNonTranspilationPasses) {
321           // This pass must run before the first fold constants pass.
322           checks.add(createSyntheticBlocks);
323         }
324     
325         if (!options.skipNonTranspilationPasses) {
326           checks.add(checkVars);
327         }
328     
329         if (options.inferConsts && !options.skipNonTranspilationPasses) {
330           checks.add(inferConsts);
331         }
332     
333         if (options.computeFunctionSideEffects && !options.skipNonTranspilationPasses) {
334           checks.add(checkRegExp);
335         }
336     
337         // This pass should run before types are assigned.
338         if (options.processObjectPropertyString && !options.skipNonTranspilationPasses) {
339           checks.add(objectPropertyStringPreprocess);
340         }
341     
342         if (options.getLanguageIn().isEs6OrHigher() && !options.skipTranspilationAndCrash) {
343           checks.add(es6ExternsCheck);
344           checks.add(es6SuperCheck);
345           TranspilationPasses.addEs6EarlyPasses(checks);
346         }
347     
348         // It's important that the Dart super accessors pass run *before* es6ConvertSuper,

351           checks.add(dartSuperAccessorsPass);
352         }
353     
354         if (options.getLanguageIn().isEs6OrHigher() && !options.skipTranspilationAndCrash) {
355           TranspilationPasses.addEs6LatePasses(checks);
356           checks.add(markTranspilationDone);
357         }
358     
359         if (options.raiseToEs6Typed()) {
360           checks.add(convertToTypedES6);
361         }
362     
363         if (!options.forceLibraryInjection.isEmpty()) {
364           checks.add(injectRuntimeLibraries);
365         }
366     
367         if (options.skipNonTranspilationPasses) {
368           return checks;
369         }
370     
371         checks.add(convertStaticInheritance);
372     
373         // End of ES6 transpilation passes.
374     
375         checks.add(createEmptyPass("beforeTypeChecking"));
376     
377         if (options.getNewTypeInference()) {
378           checks.add(symbolTableForNewTypeInference);
379           checks.add(newTypeInference);
380         }
381     
382         if (!options.allowsHotswapReplaceScript()) {
383           checks.add(inlineTypeAliases);
384         }
385     
386         if (!options.getNewTypeInference()) {
387           addOldTypeCheckerPasses(checks, options);
388         }
389     
390         // NOTE(dimvar): This will move later into the checks as we convert checks
391         // to handle types from the new type inference
392         if (options.getNewTypeInference()) {
393           addOldTypeCheckerPasses(checks, options);
394         }
395     
396         if (options.generateExportsAfterTypeChecking && options.generateExports) {
397           checks.add(generateExports);
398         }
399     
400         if (!options.disables(DiagnosticGroups.CHECK_USELESS_CODE) ||
401             (!options.getNewTypeInference() && !options.disables(DiagnosticGroups.MISSING_RETURN))) {
402           checks.add(checkControlFlow);
403         }
404     
405         // CheckAccessControls only works if check types is on.
406         if (options.checkTypes &&
407             (!options.disables(DiagnosticGroups.ACCESS_CONTROLS)
408              || options.enables(DiagnosticGroups.CONSTANT_PROPERTY))) {
409           checks.add(checkAccessControls);
410         }
411     
412         // Analyzer checks must be run after typechecking.
413         if (options.enables(DiagnosticGroups.ANALYZER_CHECKS)) {
414           checks.add(analyzerChecks);
415         }
416     
417         if (options.checkEventfulObjectDisposalPolicy != CompilerOptions.DisposalCheckingPolicy.OFF) {
418           checks.add(checkEventfulObjectDisposal);
419         }
420     
421         if (options.checkGlobalNamesLevel.isOn()) {
422           checks.add(checkGlobalNames);
423         }
424     
425         if (options.enables(DiagnosticGroups.ES5_STRICT)) {
426           checks.add(checkStrictMode);
427         }
428     
429         if (!options.getConformanceConfigs().isEmpty()) {
430           checks.add(checkConformance);
431         }
432     
433         // Replace 'goog.getCssName' before processing defines but after the
434         // other checks have been done.
435         if (options.closurePass) {
436           checks.add(closureReplaceGetCssName);
437         }
438     
439         // i18n
440         // If you want to customize the compiler to use a different i18n pass,
441         // you can create a PassConfig that calls replacePassFactory
442         // to replace this.
443         if (!options.shouldGenerateTypedExterns()) {
444           if (options.replaceMessagesWithChromeI18n) {
445             checks.add(replaceMessagesForChrome);
446           } else if (options.messageBundle != null) {
447             checks.add(replaceMessages);
448           }
449         }
450     
451         if (options.getTweakProcessing().isOn()) {
452           checks.add(processTweaks);
453         }
454     
455         // Defines in code always need to be processed.
456         checks.add(processDefines);
457     
458         if (options.shouldGenerateTypedExterns()) {
459           checks.add(removeBodies);
460         }
461     
462         if (options.instrumentationTemplate != null ||
463             options.recordFunctionInformation) {
464           checks.add(computeFunctionNames);
465         }
466     
467         if (options.nameReferenceGraphPath != null &&
468             !options.nameReferenceGraphPath.isEmpty()) {
469           checks.add(printNameReferenceGraph);
470         }
471     
472         if (options.nameReferenceReportPath != null &&
473             !options.nameReferenceReportPath.isEmpty()) {
474           checks.add(printNameReferenceReport);
475         }
476     
477         checks.add(createEmptyPass("afterStandardChecks"));
478     
479         assertAllOneTimePasses(checks);
480         assertValidOrder(checks);
481         return checks;
482       }

484       @Override
485       protected List<PassFactory> getOptimizations() {
486         List<PassFactory> passes = new ArrayList<>();
487     
488         if (options.skipNonTranspilationPasses) {
489           return passes;
490         }
491     
492         // Gather property names in externs so they can be queried by the
493         // optimising passes.
494         passes.add(gatherExternProperties);
495     
496         passes.add(garbageCollectChecks);
497     
498         // TODO(nicksantos): The order of these passes makes no sense, and needs
499         // to be re-arranged.
500     
501         if (options.instrumentForCoverage) {
502           passes.add(instrumentForCodeCoverage);
503         }
504     
505         if (options.runtimeTypeCheck) {
506           passes.add(runtimeTypeCheck);
507         }
508     
509         // Inlines functions that perform dynamic accesses to static properties of parameters that are
510         // typed as {Function}.
511         //
512         // Inlining these functions turns a dynamic access to a static property of a class definition
513         // into a fully qualified access and in so doing enables better dead code stripping.
514         if (options.j2clPass) {
515           passes.add(j2clPass);
516           passes.add(j2clPropertyInlinerPass);
517         }
518     
519         passes.add(createEmptyPass("beforeStandardOptimizations"));
520     
521         if (options.replaceIdGenerators) {
522           passes.add(replaceIdGenerators);
523         }
524     
525         // Optimizes references to the arguments variable.
526         if (options.optimizeArgumentsArray) {
527           passes.add(optimizeArgumentsArray);
528         }
529     
530         // Abstract method removal works best on minimally modified code, and also
531         // only needs to run once.
532         if (options.closurePass &&
533             (options.removeAbstractMethods || options.removeClosureAsserts)) {
534           passes.add(closureCodeRemoval);
535         }
536     
537         // Property disambiguation should only run once and needs to be done
538         // soon after type checking, both so that it can make use of type
539         // information and so that other passes can take advantage of the renamed
540         // properties.
541         if (options.disambiguatePrivateProperties) {
542           passes.add(disambiguatePrivateProperties);
543         }
544     
545         // Collapsing properties can undo constant inlining, so we do this before
546         // the main optimization loop.
547         if (options.collapseProperties) {
548           passes.add(collapseProperties);
549         }
550     
551         if (options.inferConsts) {
552           passes.add(inferConsts);
553         }
554     
555         if (options.reportPath != null && (options.extraSmartNameRemoval || options.smartNameRemoval)) {
556           passes.add(initNameAnalyzeReport);
557         }
558     
559         // Running smart name removal before disambiguate properties allows disambiguate properties
560         // to be more effective if code that would prevent disambiguation can be removed.
561         if (options.extraSmartNameRemoval && options.smartNameRemoval) {
562     
563           // These passes remove code that is dead because of define flags.
564           // If the dead code is weakly typed, running these passes before property
565           // disambiguation results in more code removal.
566           // The passes are one-time on purpose. (The later runs are loopable.)
567           if (options.foldConstants && (options.inlineVariables || options.inlineLocalVariables)) {
568             passes.add(earlyInlineVariables);
569             passes.add(earlyPeepholeOptimizations);
570           }
571     
572           passes.add(extraSmartNamePass);
573         }
574     
575         // Property disambiguation should only run once and needs to be done
576         // soon after type checking, both so that it can make use of type
577         // information and so that other passes can take advantage of the renamed
578         // properties.
579         if (options.disambiguateProperties) {
580           passes.add(disambiguateProperties);
581         }
582     
583         if (options.computeFunctionSideEffects) {
584           passes.add(markPureFunctions);
585         } else if (options.markNoSideEffectCalls) {
586           // TODO(user) The properties that this pass adds to CALL and NEW
587           // AST nodes increase the AST's in-memory size.  Given that we are
588           // already running close to our memory limits, we could run into
589           // trouble if we end up using the @nosideeffects annotation a lot
590           // or compute @nosideeffects annotations by looking at function
591           // bodies.  It should be easy to propagate @nosideeffects
592           // annotations as part of passes that depend on this property and
593           // store the result outside the AST (which would allow garbage
594           // collection once the pass is done).
595           passes.add(markNoSideEffectCalls);
596         }
597     
598         if (options.chainCalls) {
599           passes.add(chainCalls);
600         }
601     
602         // Constant checking must be done after property collapsing because
603         // property collapsing can introduce new constants (e.g. enum values).
604         // TODO(johnlenz): make checkConsts namespace aware so it can be run
605         // as during the checks phase.
606         passes.add(checkConsts);
607     
608         // Detects whether invocations of the method goog.string.Const.from are done
609         // with an argument which is a string literal.
610         passes.add(checkConstParams);
611     
612         assertAllOneTimePasses(passes);
613     
614         if (options.smartNameRemoval || options.reportPath != null) {
615           passes.addAll(getCodeRemovingPasses());
616           passes.add(smartNamePass);
617         }
618     
619         // This needs to come after the inline constants pass, which is run within
620         // the code removing passes.
621         if (options.closurePass) {
622           passes.add(closureOptimizePrimitives);
623         }
624     
625         // ReplaceStrings runs after CollapseProperties in order to simplify
626         // pulling in values of constants defined in enums structures. It also runs
627         // after disambiguate properties and smart name removal so that it can
628         // correctly identify logging types and can replace references to string
629         // expressions.
630         if (!options.replaceStringsFunctionDescriptions.isEmpty()) {
631           passes.add(replaceStrings);
632         }
633     
634         // TODO(user): This forces a first crack at crossModuleCodeMotion
635         // before devirtualization. Once certain functions are devirtualized,
636         // it confuses crossModuleCodeMotion ability to recognized that
637         // it is recursive.
638     
639         // TODO(user): This is meant for a temporary quick win.
640         // In the future, we might want to improve our analysis in
641         // CrossModuleCodeMotion so we don't need to do this.
642         if (options.crossModuleCodeMotion) {
643           passes.add(crossModuleCodeMotion);
644         }
645     
646         // Method devirtualization benefits from property disambiguation so
647         // it should run after that pass but before passes that do
648         // optimizations based on global names (like cross module code motion
649         // and inline functions).  Smart Name Removal does better if run before
650         // this pass.
651         if (options.devirtualizePrototypeMethods) {
652           passes.add(devirtualizePrototypeMethods);
653         }
654     
655         if (options.customPasses != null) {
656           passes.add(getCustomPasses(
657               CustomPassExecutionTime.BEFORE_OPTIMIZATION_LOOP));
658         }
659     
660         passes.add(createEmptyPass("beforeMainOptimizations"));
661     
662         // Because FlowSensitiveInlineVariables does not operate on the global scope due to compilation
663         // time, we need to run it once before InlineFunctions so that we don't miss inlining
664         // opportunities when a function will be inlined into the global scope.
665         if (options.flowSensitiveInlineVariables) {
666           passes.add(flowSensitiveInlineVariables);
667         }
668     
669         passes.addAll(getMainOptimizationLoop());
670         passes.add(createEmptyPass("afterMainOptimizations"));
671     
672         passes.add(createEmptyPass("beforeModuleMotion"));
673     
674         if (options.crossModuleCodeMotion) {
675           passes.add(crossModuleCodeMotion);
676         }
677     
678         if (options.crossModuleMethodMotion) {
679           passes.add(crossModuleMethodMotion);
680         }
681     
682         passes.add(createEmptyPass("afterModuleMotion"));
683     
684         // Some optimizations belong outside the loop because running them more
685         // than once would either have no benefit or be incorrect.
686         if (options.customPasses != null) {
687           passes.add(getCustomPasses(
688               CustomPassExecutionTime.AFTER_OPTIMIZATION_LOOP));
689         }
690     
691         if (options.flowSensitiveInlineVariables) {
692           passes.add(flowSensitiveInlineVariables);
693     
694           // After inlining some of the variable uses, some variables are unused.
695           // Re-run remove unused vars to clean it up.
696           if (options.removeUnusedVars || options.removeUnusedLocalVars) {
697             passes.add(getRemoveUnusedVars("removeUnusedVars", false));
698           }
699         }
700     
701         // Running this pass again is required to have goog.events compile down to
702         // nothing when compiled on its own.
703         if (options.smartNameRemoval) {
704           passes.add(smartNamePass2);
705         }
706     
707         if (options.collapseAnonymousFunctions) {
708           passes.add(collapseAnonymousFunctions);
709         }
710     
711         // Move functions before extracting prototype member declarations.
712         if (options.moveFunctionDeclarations ||
713             // renamePrefixNamescape relies on moveFunctionDeclarations
714             // to preserve semantics.
715             options.renamePrefixNamespace != null) {
716           passes.add(moveFunctionDeclarations);
717         }
718     
719         if (options.anonymousFunctionNaming ==
720             AnonymousFunctionNamingPolicy.MAPPED) {
721           passes.add(nameMappedAnonymousFunctions);
722         }
723     
724         // The mapped name anonymous function pass makes use of information that
725         // the extract prototype member declarations pass removes so the former
726         // happens before the latter.
727         if (options.extractPrototypeMemberDeclarations != ExtractPrototypeMemberDeclarationsMode.OFF) {
728           passes.add(extractPrototypeMemberDeclarations);
729         }
730     
731         if (options.ambiguateProperties &&
732             (options.propertyRenaming == PropertyRenamingPolicy.ALL_UNQUOTED)) {
733           passes.add(ambiguateProperties);
734         }
735     
736         if (options.propertyRenaming == PropertyRenamingPolicy.ALL_UNQUOTED) {
737           passes.add(renameProperties);
738         }
739     
740         // Reserve global names added to the "windows" object.
741         if (options.reserveRawExports) {
742           passes.add(gatherRawExports);
743         }
744     
745         // This comes after property renaming because quoted property names must
746         // not be renamed.
747         if (options.convertToDottedProperties) {
748           passes.add(convertToDottedProperties);
749         }
750     
751         // Property renaming must happen before this pass runs since this
752         // pass may convert dotted properties into quoted properties.  It
753         // is beneficial to run before alias strings, alias keywords and
754         // variable renaming.
755         if (options.rewriteFunctionExpressions) {
756           passes.add(rewriteFunctionExpressions);
757         }
758     
759         // This comes after converting quoted property accesses to dotted property
760         // accesses in order to avoid aliasing property names.
761         if (!options.aliasableStrings.isEmpty() || options.aliasAllStrings) {
762           passes.add(aliasStrings);
763         }
764     
765         // Passes after this point can no longer depend on normalized AST
766         // assumptions.
767         passes.add(markUnnormalized);
768     
769         if (options.coalesceVariableNames) {
770           passes.add(coalesceVariableNames);
771     
772           // coalesceVariables creates identity assignments and more redundant code
773           // that can be removed, rerun the peephole optimizations to clean them
774           // up.
775           if (options.foldConstants) {
776             passes.add(peepholeOptimizations);
777           }
778         }
779     
780         if (options.collapseVariableDeclarations) {
781           passes.add(exploitAssign);
782           passes.add(collapseVariableDeclarations);
783         }
784     
785         // This pass works best after collapseVariableDeclarations.
786         passes.add(denormalize);
787     
788         if (options.instrumentationTemplate != null) {
789           passes.add(instrumentFunctions);
790         }
791     
792         if (options.variableRenaming != VariableRenamingPolicy.ALL) {
793           // If we're leaving some (or all) variables with their old names,
794           // then we need to undo any of the markers we added for distinguishing
795           // local variables ("$$1").
796           passes.add(invertContextualRenaming);
797         }
798     
799         if (options.variableRenaming != VariableRenamingPolicy.OFF) {
800           passes.add(renameVars);
801         }
802     
803         // This pass should run after names stop changing.
804         if (options.processObjectPropertyString) {
805           passes.add(objectPropertyStringPostprocess);
806         }
807     
808         if (options.labelRenaming) {
809           passes.add(renameLabels);
810         }
811     
812         if (options.foldConstants) {
813           passes.add(latePeepholeOptimizations);
814         }
815     
816         if (options.anonymousFunctionNaming ==
817             AnonymousFunctionNamingPolicy.UNMAPPED) {
818           passes.add(nameUnmappedAnonymousFunctions);
819         }
820     
821         if (protectHiddenSideEffects) {
822           passes.add(stripSideEffectProtection);
823         }
824     
825         if (options.renamePrefixNamespace != null) {
826           if (!GLOBAL_SYMBOL_NAMESPACE_PATTERN.matcher(
827               options.renamePrefixNamespace).matches()) {
828             throw new IllegalArgumentException(
829                 "Illegal character in renamePrefixNamespace name: "
830                 + options.renamePrefixNamespace);
831           }
832           passes.add(rescopeGlobalSymbols);
833         }
834     
835         // Safety checks
836         passes.add(sanityCheckAst);
837         passes.add(sanityCheckVars);
838     
839         // Raise to ES6, if allowed
840         if (options.getLanguageOut().isEs6OrHigher()) {
841           passes.add(optimizeToEs6);
842           passes.add(objectLitAssignmentShortening);
843           passes.add(rewriteBindThis);
844         }
845     
846         return passes;
847       }

849       /** Creates the passes for the main optimization loop. */
850       private List<PassFactory> getMainOptimizationLoop() {
851         List<PassFactory> passes = new ArrayList<>();
852         if (options.inlineGetters) {
853           passes.add(inlineSimpleMethods);
854         }
855     
856         passes.addAll(getCodeRemovingPasses());
857     
858         if (options.inlineFunctions || options.inlineLocalFunctions) {
859           passes.add(inlineFunctions);
860         }
861     
862         if (options.inlineProperties) {
863           passes.add(inlineProperties);
864         }
865     
866         boolean runOptimizeCalls = options.optimizeCalls
867             || options.optimizeParameters
868             || options.optimizeReturns;
869     
870         if (options.removeUnusedVars || options.removeUnusedLocalVars) {
871           if (options.deadAssignmentElimination) {
872             passes.add(deadAssignmentsElimination);
873     
874             // The Polymer source is usually not included in the compilation, but it creates
875             // getters/setters for many properties in compiled code. Dead property assignment
876             // elimination is only safe when it knows about getters/setters. Therefore, we skip
877             // it if the polymer pass is enabled.
878             if (!options.polymerPass) {
879               passes.add(deadPropertyAssignmentElimination);
880             }
881           }
882           if (!runOptimizeCalls) {
883             passes.add(getRemoveUnusedVars("removeUnusedVars", false));
884           }
885         }
886     
887         if (runOptimizeCalls) {
888           passes.add(optimizeCalls);
889           // RemoveUnusedVars cleans up after optimizeCalls, so we run it here.
890           // It has a special name because otherwise PhaseOptimizer would change its
891           // position in the optimization loop.
892           if (options.optimizeCalls) {
893             passes.add(
894                 getRemoveUnusedVars("removeUnusedVars_afterOptimizeCalls", true));
895           }
896         }
897     
898         assertAllLoopablePasses(passes);
899         return passes;
900       }

902       /** Creates several passes aimed at removing code. */
903       private List<PassFactory> getCodeRemovingPasses() {
904         List<PassFactory> passes = new ArrayList<>();
905         if (options.collapseObjectLiterals) {
906           passes.add(collapseObjectLiterals);
907         }
908     
909         if (options.inlineVariables || options.inlineLocalVariables) {
910           passes.add(inlineVariables);
911         } else if (options.inlineConstantVars) {
912           passes.add(inlineConstants);
913         }
914     
915         if (options.foldConstants) {
916           // These used to be one pass.
917           passes.add(minimizeExitPoints);
918           passes.add(peepholeOptimizations);
919         }
920     
921         if (options.removeDeadCode) {
922           passes.add(removeUnreachableCode);
923         }
924     
925         if (options.removeUnusedPrototypeProperties) {
926           passes.add(removeUnusedPrototypeProperties);
927         }
928     
929         if (options.removeUnusedClassProperties) {
930           passes.add(removeUnusedClassProperties);
931         }
932     
933         if (options.j2clPass) {
934           passes.add(j2clClinitPrunerPass);
935           passes.add(j2clConstantHoisterPass);
936           passes.add(j2clEqualitySameRewriterPass);
937         }
938     
939         assertAllLoopablePasses(passes);
940         return passes;
941       }
