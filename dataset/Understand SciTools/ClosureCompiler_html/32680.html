114       private static int lo0bits(int y) {
115         int k;
116         int x = y;
117     
118         if ((x & 7) != 0) {
119           if ((x & 1) != 0) return 0;
120           if ((x & 2) != 0) {
121             return 1;
122           }
123           return 2;
124         }
125         k = 0;
126         if ((x & 0xffff) == 0) {
127           k = 16;
128           x >>>= 16;
129         }
130         if ((x & 0xff) == 0) {
131           k += 8;
132           x >>>= 8;
133         }
134         if ((x & 0xf) == 0) {
135           k += 4;
136           x >>>= 4;
137         }
138         if ((x & 0x3) == 0) {
139           k += 2;
140           x >>>= 2;
141         }
142         if ((x & 1) == 0) {
143           k++;
144           x >>>= 1;
145           if ((x & 1) == 0) return 32;
146         }
147         return k;
148       }

150       /* Return the number (0 through 32) of most significant zero bits in x. */
151       private static int hi0bits(int x) {
152         int k = 0;
153     
154         if ((x & 0xffff0000) == 0) {
155           k = 16;
156           x <<= 16;
157         }
158         if ((x & 0xff000000) == 0) {
159           k += 8;
160           x <<= 8;
161         }
162         if ((x & 0xf0000000) == 0) {
163           k += 4;
164           x <<= 4;
165         }
166         if ((x & 0xc0000000) == 0) {
167           k += 2;
168           x <<= 2;
169         }
170         if ((x & 0x80000000) == 0) {
171           k++;
172           if ((x & 0x40000000) == 0) return 32;
173         }
174         return k;
175       }

177       private static void stuffBits(byte bits[], int offset, int val) {
178         bits[offset] = (byte) (val >> 24);
179         bits[offset + 1] = (byte) (val >> 16);
180         bits[offset + 2] = (byte) (val >> 8);
181         bits[offset + 3] = (byte) (val);
182       }

184       /* Convert d into the form b*2^e, where b is an odd integer.  b is the returned
185        * Bigint and e is the returned binary exponent.  Return the number of significant
186        * bits in b in bits.  d must be finite and nonzero. */
187       private static BigInteger d2b(double d, int[] e, int[] bits) {
188         byte dbl_bits[];
189         int i, k, y, z, de;
190         long dBits = Double.doubleToLongBits(d);
191         int d0 = (int) (dBits >>> 32);
192         int d1 = (int) (dBits);
193     
194         z = d0 & Frac_mask;
195         d0 &= 0x7fffffff; /* clear sign bit, which we ignore */
196     
197         if ((de = (d0 >>> Exp_shift)) != 0) z |= Exp_msk1;
198     
199         if ((y = d1) != 0) {
200           dbl_bits = new byte[8];
201           k = lo0bits(y);
202           y >>>= k;
203           if (k != 0) {
204             stuffBits(dbl_bits, 4, y | z << (32 - k));
205             z >>= k;
206           } else stuffBits(dbl_bits, 4, y);
207           stuffBits(dbl_bits, 0, z);
208           i = (z != 0) ? 2 : 1;
209         } else {
210           //        JS_ASSERT(z);
211           dbl_bits = new byte[4];
212           k = lo0bits(z);
213           z >>>= k;
214           stuffBits(dbl_bits, 0, z);
215           k += 32;
216           i = 1;
217         }
218         if (de != 0) {
219           e[0] = de - Bias - (P - 1) + k;
220           bits[0] = P - k;
221         } else {
222           e[0] = de - Bias - (P - 1) + 1 + k;
223           bits[0] = 32 * i - hi0bits(z);
224         }
225         return new BigInteger(dbl_bits);
226       }

228       private static String JS_dtobasestr(int base, double d) {
229         if (!(2 <= base && base <= 36)) throw new IllegalArgumentException("Bad base: " + base);
230     
231         /* Check for Infinity and NaN */
232         if (Double.isNaN(d)) {
233           return "NaN";
234         } else if (Double.isInfinite(d)) {
235           return (d > 0.0) ? "Infinity" : "-Infinity";
236         } else if (d == 0) {
237           // ALERT: should it distinguish -0.0 from +0.0 ?
238           return "0";
239         }
240     
241         boolean negative;
242         if (d >= 0.0) {
243           negative = false;
244         } else {
245           negative = true;
246           d = -d;
247         }
248     
249         /* Get the integer part of d including '-' sign. */
250         String intDigits;
251     
252         double dfloor = Math.floor(d);
253         long lfloor = (long) dfloor;
254         if (lfloor == dfloor) {
255           // int part fits long
256           intDigits = Long.toString((negative) ? -lfloor : lfloor, base);
257         } else {
258           // BigInteger should be used
259           long floorBits = Double.doubleToLongBits(dfloor);
260           int exp = (int) (floorBits >> Exp_shiftL) & Exp_mask_shifted;
261           long mantissa;
262           if (exp == 0) {
263             mantissa = (floorBits & Frac_maskL) << 1;
264           } else {
265             mantissa = (floorBits & Frac_maskL) | Exp_msk1L;
266           }
267           if (negative) {
268             mantissa = -mantissa;
269           }
270           exp -= 1075;
271           BigInteger x = BigInteger.valueOf(mantissa);
272           if (exp > 0) {
273             x = x.shiftLeft(exp);
274           } else if (exp < 0) {
275             x = x.shiftRight(-exp);
276           }
277           intDigits = x.toString(base);
278         }
279     
280         if (d == dfloor) {
281           // No fraction part
282           return intDigits;
283         } else {
284           /* We have a fraction. */
285     
286           StringBuilder buffer; /* The output string */
287           int digit;
288           double df; /* The fractional part of d */
289           BigInteger b;
290     
291           buffer = new StringBuilder();
292           buffer.append(intDigits).append('.');
293           df = d - dfloor;
294     
295           long dBits = Double.doubleToLongBits(d);
296           int word0 = (int) (dBits >> 32);
297           int word1 = (int) (dBits);
298     
299           int[] e = new int[1];
300           int[] bbits = new int[1];
301     
302           b = d2b(df, e, bbits);
303           //            JS_ASSERT(e < 0);
304           /* At this point df = b * 2^e.  e must be less than zero because 0 < df < 1. */
305     
306           int s2 = -(word0 >>> Exp_shift1 & Exp_mask >> Exp_shift1);
307           if (s2 == 0) s2 = -1;
308           s2 += Bias + P;
309           /* 1/2^s2 = (nextDouble(d) - d)/2 */
310           //            JS_ASSERT(-s2 < e);
311           BigInteger mlo = BigInteger.valueOf(1);
312           BigInteger mhi = mlo;
313           if ((word1 == 0)
314               && ((word0 & Bndry_mask) == 0)
315               && ((word0 & (Exp_mask & Exp_mask << 1)) != 0)) {
316             /* The special case.  Here we want to be within a quarter of the last input
317             significant digit instead of one half of it when the output string's value is less than d.  */
318             s2 += Log2P;
319             mhi = BigInteger.valueOf(1 << Log2P);
320           }
321     
322           b = b.shiftLeft(e[0] + s2);
323           BigInteger s = BigInteger.valueOf(1);
324           s = s.shiftLeft(s2);
325           /* At this point we have the following:
326            *   s = 2^s2;
327            *   1 > df = b/2^s2 > 0;
328            *   (d - prevDouble(d))/2 = mlo/2^s2;
329            *   (nextDouble(d) - d)/2 = mhi/2^s2. */
330           BigInteger bigBase = BigInteger.valueOf(base);
331     
332           boolean done = false;
333           do {
334             b = b.multiply(bigBase);
335             BigInteger[] divResult = b.divideAndRemainder(s);
336             b = divResult[1];
337             digit = (char) (divResult[0].intValue());
338             if (mlo == mhi) mlo = mhi = mlo.multiply(bigBase);
339             else {
340               mlo = mlo.multiply(bigBase);
341               mhi = mhi.multiply(bigBase);
342             }
343     
344             /* Do we yet have the shortest string that will round to d? */
345             int j = b.compareTo(mlo);
346             /* j is b/2^s2 compared with mlo/2^s2. */
347             BigInteger delta = s.subtract(mhi);
348             int j1 = (delta.signum() <= 0) ? 1 : b.compareTo(delta);
349             /* j1 is b/2^s2 compared with 1 - mhi/2^s2. */
350             if (j1 == 0 && ((word1 & 1) == 0)) {
351               if (j > 0) digit++;
352               done = true;
353             } else if (j < 0 || (j == 0 && ((word1 & 1) == 0))) {
354               if (j1 > 0) {
355                 /* Either dig or dig+1 would work here as the least significant digit.
356                 Use whichever would produce an output value closer to d. */
357                 b = b.shiftLeft(1);
358                 j1 = b.compareTo(s);
359                 if (j1
360                     > 0) /* The even test (|| (j1 == 0 && (digit & 1))) is not here because it messes up odd base output
361                           * such as 3.5 in base 3.  */ digit++;
362               }
363               done = true;
364             } else if (j1 > 0) {
365               digit++;
366               done = true;
367             }
368             //                JS_ASSERT(digit < (uint32)base);
369             buffer.append(BASEDIGIT(digit));
370           } while (!done);
371     
372           return buffer.toString();
373         }

376       /* dtoa for IEEE arithmetic (dmg): convert double to ASCII string.
377        *
378        * Inspired by "How to Print Floating-Point Numbers Accurately" by
379        * Guy L. Steele, Jr. and Jon L. White [Proc. ACM SIGPLAN '90, pp. 92-101].
380        *
381        * Modifications:
382        *  1. Rather than iterating, we use a simple numeric overestimate
383        *     to determine k = floor(log10(d)).  We scale relevant
384        *     quantities using O(log2(k)) rather than O(k) multiplications.
385        *  2. For some modes > 2 (corresponding to ecvt and fcvt), we don't
386        *     try to generate digits strictly left to right.  Instead, we
387        *     compute with fewer bits and propagate the carry if necessary
388        *     when rounding the final digit up.  This is often faster.
389        *  3. Under the assumption that input will be rounded nearest,
390        *     mode 0 renders 1e23 as 1e23 rather than 9.999999999999999e22.
391        *     That is, we allow equality in stopping tests when the
392        *     round-nearest rule will give the same floating-point value
393        *     as would satisfaction of the stopping test with strict
394        *     inequality.
395        *  4. We remove common factors of powers of 2 from relevant
396        *     quantities.
397        *  5. When converting floating-point integers less than 1e16,
398        *     we use floating-point arithmetic rather than resorting
399        *     to multiple-precision integers.
400        *  6. When asked to produce fewer than 15 digits, we first try
401        *     to get by with floating-point arithmetic; we resort to
402        *     multiple-precision integer arithmetic only if we cannot
403        *     guarantee that the floating-point calculation has given
404        *     the correctly rounded result.  For k requested digits and
405        *     "uniformly" distributed input, the probability is
406        *     something like 10^(k-15) that we must resort to the Long
407        *     calculation.
408        */
409     
410       private static int word0(double d) {
411         long dBits = Double.doubleToLongBits(d);
412         return (int) (dBits >> 32);
413       }

415       private static double setWord0(double d, int i) {
416         long dBits = Double.doubleToLongBits(d);
417         dBits = ((long) i << 32) | (dBits & 0x0FFFFFFFFL);
418         return Double.longBitsToDouble(dBits);
419       }

421       private static int word1(double d) {
422         long dBits = Double.doubleToLongBits(d);
423         return (int) (dBits);
424       }

432       private static boolean roundOff(StringBuilder buf) {
433         int i = buf.length();
434         while (i != 0) {
435           --i;
436           char c = buf.charAt(i);
437           if (c != '9') {
438             buf.setCharAt(i, (char) (c + 1));
439             buf.setLength(i + 1);
440             return false;
441           }
442         }
443         buf.setLength(0);
444         return true;
445       }

447       /* Always emits at least one digit. */
448       /* If biasUp is set, then rounding in modes 2 and 3 will round away from zero
449        * when the number is exactly halfway between two representable values.  For example,
450        * rounding 2.5 to zero digits after the decimal point will return 3 and not 2.
451        * 2.49 will still round to 2, and 2.51 will still round to 3. */
452       /* bufsize should be at least 20 for modes 0 and 1.  For the other modes,
453        * bufsize should be two greater than the maximum number of output characters expected. */
454       private static int JS_dtoa(
455           double d, int mode, boolean biasUp, int ndigits, boolean[] sign, StringBuilder buf) {
456         /*  Arguments ndigits, decpt, sign are similar to those
457             of ecvt and fcvt; trailing zeros are suppressed from
458             the returned string.  If not null, *rve is set to point
459             to the end of the return value.  If d is +-Infinity or NaN,
460             then *decpt is set to 9999.
461     
462             mode:
463             0 ==> shortest string that yields d when read in
464             and rounded to nearest.
465             1 ==> like 0, but with Steele & White stopping rule;
466             e.g. with IEEE P754 arithmetic , mode 0 gives
467             1e23 whereas mode 1 gives 9.999999999999999e22.
468             2 ==> max(1,ndigits) significant digits.  This gives a
469             return value similar to that of ecvt, except
470             that trailing zeros are suppressed.
471             3 ==> through ndigits past the decimal point.  This
472             gives a return value similar to that from fcvt,
473             except that trailing zeros are suppressed, and
474             ndigits can be negative.
475             4-9 should give the same return values as 2-3, i.e.,
476             4 <= mode <= 9 ==> same return as mode
477             2 + (mode & 1).  These modes are mainly for
478             debugging; often they run slower but sometimes
479             faster than modes 2-3.
480             4,5,8,9 ==> left-to-right digit generation.
481             6-9 ==> don't try fast floating-point estimate
482             (if applicable).
483     
484             Values of mode other than 0-9 are treated as mode 0.
485     
486             Sufficient space is allocated to the return value
487             to hold the suppressed trailing zeros.
488         */
489     
490         int b2, b5, i, ieps, ilim, ilim0, ilim1, j, j1, k, k0, m2, m5, s2, s5;
491         char dig;
492         long L;
493         long x;
494         BigInteger b, b1, delta, mlo, mhi, S;
495         int[] be = new int[1];
496         int[] bbits = new int[1];
497         double d2, ds, eps;
498         boolean spec_case, denorm, k_check, try_quick, leftright;
499     
500         if ((word0(d) & Sign_bit) != 0) {
501           /* set sign for everything, including 0's and NaNs */
502           sign[0] = true;
503           // word0(d) &= ~Sign_bit;  /* clear sign bit */
504           d = setWord0(d, word0(d) & ~Sign_bit);
505         } else sign[0] = false;
506     
507         if ((word0(d) & Exp_mask) == Exp_mask) {
508           /* Infinity or NaN */
509           buf.append(((word1(d) == 0) && ((word0(d) & Frac_mask) == 0)) ? "Infinity" : "NaN");
510           return 9999;
511         }
512         if (d == 0) {
513           //          no_digits:
514           buf.setLength(0);
515           buf.append('0'); /* copy "0" to buffer */
516           return 1;
517         }
518     
519         b = d2b(d, be, bbits);
520         if ((i = (word0(d) >>> Exp_shift1 & (Exp_mask >> Exp_shift1))) != 0) {
521           d2 = setWord0(d, (word0(d) & Frac_mask1) | Exp_11);
522           /* log(x)   ~=~ log(1.5) + (x-1.5)/1.5
523            * log10(x)  =  log(x) / log(10)
524            *      ~=~ log(1.5)/log(10) + (x-1.5)/(1.5*log(10))
525            * log10(d) = (i-Bias)*log(2)/log(10) + log10(d2)
526            *
527            * This suggests computing an approximation k to log10(d) by
528            *
529            * k = (i - Bias)*0.301029995663981
530            *  + ( (d2-1.5)*0.289529654602168 + 0.176091259055681 );
531            *
532            * We want k to be too large rather than too small.
533            * The error in the first-order Taylor series approximation
534            * is in our favor, so we just round up the constant enough
535            * to compensate for any error in the multiplication of
536            * (i - Bias) by 0.301029995663981; since |i - Bias| <= 1077,
537            * and 1077 * 0.30103 * 2^-52 ~=~ 7.2e-14,
538            * adding 1e-13 to the constant term more than suffices.
539            * Hence we adjust the constant term to 0.1760912590558.
540            * (We could get a more accurate k by invoking log10,
541            *  but this is probably not worthwhile.)
542            */
543           i -= Bias;
544           denorm = false;
545         } else {
546           /* d is denormalized */
547           i = bbits[0] + be[0] + (Bias + (P - 1) - 1);
548           x =
549               (i > 32)
550                   ? ((long) word0(d)) << (64 - i) | word1(d) >>> (i - 32)
551                   : ((long) word1(d)) << (32 - i);
552           //            d2 = x;
553           //            word0(d2) -= 31*Exp_msk1; /* adjust exponent */
554           d2 = setWord0(x, word0(x) - 31 * Exp_msk1);
555           i -= (Bias + (P - 1) - 1) + 1;
556           denorm = true;
557         }
558         /* At this point d = f*2^i, where 1 <= f < 2.  d2 is an approximation of f. */
559         ds = (d2 - 1.5) * 0.289529654602168 + 0.1760912590558 + i * 0.301029995663981;
560         k = (int) ds;
561         if (ds < 0.0 && ds != k) k--; /* want k = floor(ds) */
562         k_check = true;
563         if (k >= 0 && k <= Ten_pmax) {
564           if (d < tens[k]) k--;
565           k_check = false;
566         }
567         /* At this point floor(log10(d)) <= k <= floor(log10(d))+1.
568         If k_check is zero, we're guaranteed that k = floor(log10(d)). */
569         j = bbits[0] - i - 1;
570         /* At this point d = b/2^j, where b is an odd integer. */
571         if (j >= 0) {
572           b2 = 0;
573           s2 = j;
574         } else {
575           b2 = -j;
576           s2 = 0;
577         }
578         if (k >= 0) {
579           b5 = 0;
580           s5 = k;
581           s2 += k;
582         } else {
583           b2 -= k;
584           b5 = -k;
585           s5 = 0;
586         }
587         /* At this point d/10^k = (b * 2^b2 * 5^b5) / (2^s2 * 5^s5), where b is an odd integer,
588         b2 >= 0, b5 >= 0, s2 >= 0, and s5 >= 0. */
589         if (mode < 0 || mode > 9) mode = 0;
590         try_quick = true;
591         if (mode > 5) {
592           mode -= 4;
593           try_quick = false;
594         }
595         leftright = true;
596         ilim = ilim1 = 0;
597         switch (mode) {
598           case 0:
599           case 1:
600             ilim = ilim1 = -1;
601             i = 18;
602             ndigits = 0;
603             break;
604           case 2:
605             leftright = false;
606             /* no break */
607           case 4:
608             if (ndigits <= 0) ndigits = 1;
609             ilim = ilim1 = i = ndigits;
610             break;
611           case 3:
612             leftright = false;
613             /* no break */
614           case 5:
615             i = ndigits + k + 1;
616             ilim = i;
617             ilim1 = i - 1;
618             if (i <= 0) i = 1;
619         }
620         /* ilim is the maximum number of significant digits we want, based on k and ndigits. */
621         /* ilim1 is the maximum number of significant digits we want, based on k and ndigits,
622         when it turns out that k was computed too high by one. */
623     
624         boolean fast_failed = false;
625         if (ilim >= 0 && ilim <= Quick_max && try_quick) {
626     
627           /* Try to get by with floating-point arithmetic. */
628     
629           i = 0;
630           d2 = d;
631           k0 = k;
632           ilim0 = ilim;
633           ieps = 2; /* conservative */
634           /* Divide d by 10^k, keeping track of the roundoff error and avoiding overflows. */
635           if (k > 0) {
636             ds = tens[k & 0xf];
637             j = k >> 4;
638             if ((j & Bletch) != 0) {
639               /* prevent overflows */
640               j &= Bletch - 1;
641               d /= bigtens[n_bigtens - 1];
642               ieps++;
643             }
644             for (; (j != 0); j >>= 1, i++)
645               if ((j & 1) != 0) {
646                 ieps++;
647                 ds *= bigtens[i];
648               }
649             d /= ds;
650           } else if ((j1 = -k) != 0) {
651             d *= tens[j1 & 0xf];
652             for (j = j1 >> 4; (j != 0); j >>= 1, i++)
653               if ((j & 1) != 0) {
654                 ieps++;
655                 d *= bigtens[i];
656               }
657           }
658           /* Check that k was computed correctly. */
659           if (k_check && d < 1.0 && ilim > 0) {
660             if (ilim1 <= 0) fast_failed = true;
661             else {
662               ilim = ilim1;
663               k--;
664               d *= 10.;
665               ieps++;
666             }
667           }
668           /* eps bounds the cumulative error. */
669           //            eps = ieps*d + 7.0;
670           //            word0(eps) -= (P-1)*Exp_msk1;
671           eps = ieps * d + 7.0;
672           eps = setWord0(eps, word0(eps) - (P - 1) * Exp_msk1);
673           if (ilim == 0) {
674             S = mhi = null;
675             d -= 5.0;
676             if (d > eps) {
677               buf.append('1');
678               k++;
679               return k + 1;
680             }
681             if (d < -eps) {
682               buf.setLength(0);
683               buf.append('0'); /* copy "0" to buffer */
684               return 1;
685             }
686             fast_failed = true;
687           }
688           if (!fast_failed) {
689             fast_failed = true;
690             if (leftright) {
691               /* Use Steele & White method of only
692                * generating digits needed.
693                */
694               eps = 0.5 / tens[ilim - 1] - eps;
695               for (i = 0; ; ) {
696                 L = (long) d;
697                 d -= L;
698                 buf.append((char) ('0' + L));
699                 if (d < eps) {
700                   return k + 1;
701                 }
702                 if (1.0 - d < eps) {
703                   //                            goto bump_up;
704                   char lastCh;
705                   while (true) {
706                     lastCh = buf.charAt(buf.length() - 1);
707                     buf.setLength(buf.length() - 1);
708                     if (lastCh != '9') break;
709                     if (buf.length() == 0) {
710                       k++;
711                       lastCh = '0';
712                       break;
713                     }
714                   }
715                   buf.append((char) (lastCh + 1));
716                   return k + 1;
717                 }
718                 if (++i >= ilim) break;
719                 eps *= 10.0;
720                 d *= 10.0;
721               }
722             } else {
723               /* Generate ilim digits, then fix them up. */
724               eps *= tens[ilim - 1];
725               for (i = 1; ; i++, d *= 10.0) {
726                 L = (long) d;
727                 d -= L;
728                 buf.append((char) ('0' + L));
729                 if (i == ilim) {
730                   if (d > 0.5 + eps) {
731                     //                                goto bump_up;
732                     char lastCh;
733                     while (true) {
734                       lastCh = buf.charAt(buf.length() - 1);
735                       buf.setLength(buf.length() - 1);
736                       if (lastCh != '9') break;
737                       if (buf.length() == 0) {
738                         k++;
739                         lastCh = '0';
740                         break;
741                       }
742                     }
743                     buf.append((char) (lastCh + 1));
744                     return k + 1;
745                   } else if (d < 0.5 - eps) {
746                     stripTrailingZeroes(buf);
747                     //                                    while(*--s == '0') ;
748                     //                                    s++;
749                     return k + 1;
750                   }
751                   break;
752                 }
753               }
754             }
755           }
756           if (fast_failed) {
757             buf.setLength(0);
758             d = d2;
759             k = k0;
760             ilim = ilim0;
761           }
762         }
763     
764         /* Do we have a "small" integer? */
765     
766         if (be[0] >= 0 && k <= Int_max) {
767           /* Yes. */
768           ds = tens[k];
769           if (ndigits < 0 && ilim <= 0) {
770             S = mhi = null;
771             if (ilim < 0 || d < 5 * ds || (!biasUp && d == 5 * ds)) {
772               buf.setLength(0);
773               buf.append('0'); /* copy "0" to buffer */
774               return 1;
775             }
776             buf.append('1');
777             k++;
778             return k + 1;
779           }
780           for (i = 1; ; i++) {
781             L = (long) (d / ds);
782             d -= L * ds;
783             buf.append((char) ('0' + L));
784             if (i == ilim) {
785               d += d;
786               if ((d > ds) || (d == ds && (((L & 1) != 0) || biasUp))) {
787                 //                    bump_up:
788                 //                        while(*--s == '9')
789                 //                            if (s == buf) {
790                 //                                k++;
791                 //                                *s = '0';
792                 //                                break;
793                 //                            }
794                 //                        ++*s++;
795                 char lastCh;
796                 while (true) {
797                   lastCh = buf.charAt(buf.length() - 1);
798                   buf.setLength(buf.length() - 1);
799                   if (lastCh != '9') break;
800                   if (buf.length() == 0) {
801                     k++;
802                     lastCh = '0';
803                     break;
804                   }
805                 }
806                 buf.append((char) (lastCh + 1));
807               }
808               break;
809             }
810             d *= 10.0;
811             if (d == 0) break;
812           }
813           return k + 1;
814         }
815     
816         m2 = b2;
817         m5 = b5;
818         mhi = mlo = null;
819         if (leftright) {
820           if (mode < 2) {
821             i = (denorm) ? be[0] + (Bias + (P - 1) - 1 + 1) : 1 + P - bbits[0];
822             /* i is 1 plus the number of trailing zero bits in d's significand. Thus,
823             (2^m2 * 5^m5) / (2^(s2+i) * 5^s5) = (1/2 lsb of d)/10^k. */
824           } else {
825             j = ilim - 1;
826             if (m5 >= j) m5 -= j;
827             else {
828               s5 += j -= m5;
829               b5 += j;
830               m5 = 0;
831             }
832             if ((i = ilim) < 0) {
833               m2 -= i;
834               i = 0;
835             }
836             /* (2^m2 * 5^m5) / (2^(s2+i) * 5^s5) = (1/2 * 10^(1-ilim))/10^k. */
837           }
838           b2 += i;
839           s2 += i;
840           mhi = BigInteger.valueOf(1);
841           /* (mhi * 2^m2 * 5^m5) / (2^s2 * 5^s5) = one-half of last printed (when mode >= 2) or
842           input (when mode < 2) significant digit, divided by 10^k. */
843         }
844         /* We still have d/10^k = (b * 2^b2 * 5^b5) / (2^s2 * 5^s5).  Reduce common factors in
845         b2, m2, and s2 without changing the equalities. */
846         if (m2 > 0 && s2 > 0) {
847           i = (m2 < s2) ? m2 : s2;
848           b2 -= i;
849           m2 -= i;
850           s2 -= i;
851         }
852     
853         /* Fold b5 into b and m5 into mhi. */
854         if (b5 > 0) {
855           if (leftright) {
856             if (m5 > 0) {
857               mhi = pow5mult(mhi, m5);
858               b1 = mhi.multiply(b);
859               b = b1;
860             }
861             if ((j = b5 - m5) != 0) b = pow5mult(b, j);
862           } else b = pow5mult(b, b5);
863         }
864         /* Now we have d/10^k = (b * 2^b2) / (2^s2 * 5^s5) and
865         (mhi * 2^m2) / (2^s2 * 5^s5) = one-half of last printed or input significant digit, divided by 10^k. */
866     
867         S = BigInteger.valueOf(1);
868         if (s5 > 0) S = pow5mult(S, s5);
869         /* Now we have d/10^k = (b * 2^b2) / (S * 2^s2) and
870         (mhi * 2^m2) / (S * 2^s2) = one-half of last printed or input significant digit, divided by 10^k. */
871     
872         /* Check for special case that d is a normalized power of 2. */
873         spec_case = false;
874         if (mode < 2) {
875           if ((word1(d) == 0)
876               && ((word0(d) & Bndry_mask) == 0)
877               && ((word0(d) & (Exp_mask & Exp_mask << 1)) != 0)) {
878             /* The special case.  Here we want to be within a quarter of the last input
879             significant digit instead of one half of it when the decimal output string's value is less than d.  */
880             b2 += Log2P;
881             s2 += Log2P;
882             spec_case = true;
883           }
884         }
885     
886         /* Arrange for convenient computation of quotients:
887          * shift left if necessary so divisor has 4 leading 0 bits.
888          *
889          * Perhaps we should just compute leading 28 bits of S once
890          * and for all and pass them and a shift to quorem, so it
891          * can do shifts and ors to compute the numerator for q.
892          */
893         byte[] S_bytes = S.toByteArray();
894         int S_hiWord = 0;
895         for (int idx = 0; idx < 4; idx++) {
896           S_hiWord = (S_hiWord << 8);
897           if (idx < S_bytes.length) S_hiWord |= (S_bytes[idx] & 0xFF);
898         }
899         if ((i = (((s5 != 0) ? 32 - hi0bits(S_hiWord) : 1) + s2) & 0x1f) != 0) i = 32 - i;
900         /* i is the number of leading zero bits in the most significant word of S*2^s2. */
901         if (i > 4) {
902           i -= 4;
903           b2 += i;
904           m2 += i;
905           s2 += i;
906         } else if (i < 4) {
907           i += 28;
908           b2 += i;
909           m2 += i;
910           s2 += i;
911         }
912         /* Now S*2^s2 has exactly four leading zero bits in its most significant word. */
913         if (b2 > 0) b = b.shiftLeft(b2);
914         if (s2 > 0) S = S.shiftLeft(s2);
915         /* Now we have d/10^k = b/S and
916         (mhi * 2^m2) / S = maximum acceptable error, divided by 10^k. */
917         if (k_check) {
918           if (b.compareTo(S) < 0) {
919             k--;
920             b = b.multiply(BigInteger.valueOf(10)); /* we botched the k estimate */
921             if (leftright) mhi = mhi.multiply(BigInteger.valueOf(10));
922             ilim = ilim1;
923           }
924         }
925         /* At this point 1 <= d/10^k = b/S < 10. */
926     
927         if (ilim <= 0 && mode > 2) {
928           /* We're doing fixed-mode output and d is less than the minimum nonzero output in this mode.
929           Output either zero or the minimum nonzero output depending on which is closer to d. */
930           if ((ilim < 0)
931               || ((i = b.compareTo(S = S.multiply(BigInteger.valueOf(5)))) < 0)
932               || ((i == 0 && !biasUp))) {
933             /* Always emit at least one digit.  If the number appears to be zero
934             using the current mode, then emit one '0' digit and set decpt to 1. */
935             /*no_digits:
936             k = -1 - ndigits;
937             goto ret; */
938             buf.setLength(0);
939             buf.append('0'); /* copy "0" to buffer */
940             return 1;
941             //                goto no_digits;
942           }
943           //        one_digit:
944           buf.append('1');
945           k++;
946           return k + 1;
947         }
948         if (leftright) {
949           if (m2 > 0) mhi = mhi.shiftLeft(m2);
950     
951           /* Compute mlo -- check for special case
952            * that d is a normalized power of 2.
953            */
954     
955           mlo = mhi;
956           if (spec_case) {
957             mhi = mlo;
958             mhi = mhi.shiftLeft(Log2P);
959           }
960           /* mlo/S = maximum acceptable error, divided by 10^k, if the output is less than d. */
961           /* mhi/S = maximum acceptable error, divided by 10^k, if the output is greater than d. */
962     
963           for (i = 1; ; i++) {
964             BigInteger[] divResult = b.divideAndRemainder(S);
965             b = divResult[1];
966             dig = (char) (divResult[0].intValue() + '0');
967             /* Do we yet have the shortest decimal string
968              * that will round to d?
969              */
970             j = b.compareTo(mlo);
971             /* j is b/S compared with mlo/S. */
972             delta = S.subtract(mhi);
973             j1 = (delta.signum() <= 0) ? 1 : b.compareTo(delta);
974             /* j1 is b/S compared with 1 - mhi/S. */
975             if ((j1 == 0) && (mode == 0) && ((word1(d) & 1) == 0)) {
976               if (dig == '9') {
977                 buf.append('9');
978                 if (roundOff(buf)) {
979                   k++;
980                   buf.append('1');
981                 }
982                 return k + 1;
983                 //                        goto round_9_up;
984               }
985               if (j > 0) dig++;
986               buf.append(dig);
987               return k + 1;
988             }
989             if ((j < 0) || ((j == 0) && (mode == 0) && ((word1(d) & 1) == 0))) {
990               if (j1 > 0) {
991                 /* Either dig or dig+1 would work here as the least significant decimal digit.
992                 Use whichever would produce a decimal value closer to d. */
993                 b = b.shiftLeft(1);
994                 j1 = b.compareTo(S);
995                 if (((j1 > 0) || (j1 == 0 && (((dig & 1) == 1) || biasUp))) && (dig++ == '9')) {
996                   buf.append('9');
997                   if (roundOff(buf)) {
998                     k++;
999                     buf.append('1');
1000                   }
1001                   return k + 1;
1002                   //                                goto round_9_up;
1003                 }
1004               }
1005               buf.append(dig);
1006               return k + 1;
1007             }
1008             if (j1 > 0) {
1009               if (dig == '9') {
1010                 /* possible if i == 1 */
1011                 //                    round_9_up:
1012                 //                        *s++ = '9';
1013                 //                        goto roundoff;
1014                 buf.append('9');
1015                 if (roundOff(buf)) {
1016                   k++;
1017                   buf.append('1');
1018                 }
1019                 return k + 1;
1020               }
1021               buf.append((char) (dig + 1));
1022               return k + 1;
1023             }
1024             buf.append(dig);
1025             if (i == ilim) break;
1026             b = b.multiply(BigInteger.valueOf(10));
1027             if (mlo == mhi) mlo = mhi = mhi.multiply(BigInteger.valueOf(10));
1028             else {
1029               mlo = mlo.multiply(BigInteger.valueOf(10));
1030               mhi = mhi.multiply(BigInteger.valueOf(10));
1031             }
1032           }
1033         } else
1034           for (i = 1; ; i++) {
1035             //                (char)(dig = quorem(b,S) + '0');
1036             BigInteger[] divResult = b.divideAndRemainder(S);
1037             b = divResult[1];
1038             dig = (char) (divResult[0].intValue() + '0');
1039             buf.append(dig);
1040             if (i >= ilim) break;
1041             b = b.multiply(BigInteger.valueOf(10));
1042           }
1043     
1044         /* Round off last digit */
1045     
1046         b = b.shiftLeft(1);
1047         j = b.compareTo(S);
1048         if ((j > 0) || (j == 0 && (((dig & 1) == 1) || biasUp))) {
1049           //        roundoff:
1050           //            while(*--s == '9')
1051           //                if (s == buf) {
1052           //                    k++;
1053           //                    *s++ = '1';
1054           //                    goto ret;
1055           //                }
1056           //            ++*s++;
1057           if (roundOff(buf)) {
1058             k++;
1059             buf.append('1');
1060             return k + 1;
1061           }
1062         } else {
1063           stripTrailingZeroes(buf);
1064           //            while(*--s == '0') ;
1065           //            s++;
1066         }
1067         //      ret:
1068         //        Bfree(S);
1069         //        if (mhi) {
1070         //            if (mlo && mlo != mhi)
1071         //                Bfree(mlo);
1072         //            Bfree(mhi);
1073         //        }
1074         //      ret1:
1075         //        Bfree(b);
1076         //        JS_ASSERT(s < buf + bufsize);
1077         return k + 1;
1078       }

1080       private static void stripTrailingZeroes(StringBuilder buf) {
1081         //      while(*--s == '0') ;
1082         //      s++;
1083         int bl = buf.length();
1084         while (bl-- > 0 && buf.charAt(bl) == '0') {
1085           // empty
1086         }
1087         buf.setLength(bl + 1);
1088       }

1091       private static final int dtoaModes[] = {
1092         0, /* DTOSTR_STANDARD */ 0, /* DTOSTR_STANDARD_EXPONENTIAL, */ 3, /* DTOSTR_FIXED, */
1093         2, /* DTOSTR_EXPONENTIAL, */ 2
1094       }; /* DTOSTR_PRECISION */

1096       private static void JS_dtostr(StringBuilder buffer, int mode, int precision, double d) {
1097         int decPt; /* Position of decimal point relative to first digit returned by JS_dtoa */
1098         boolean[] sign = new boolean[1]; /* true if the sign bit was set in d */
1099         int nDigits; /* Number of significand digits returned by JS_dtoa */
1100     
1101         //        JS_ASSERT(bufferSize >= (size_t)(mode <= DTOSTR_STANDARD_EXPONENTIAL ? DTOSTR_STANDARD_BUFFER_SIZE :
1102         //                DTOSTR_VARIABLE_BUFFER_SIZE(precision)));
1103     
1104         if (mode == DTOSTR_FIXED && (d >= 1e21 || d <= -1e21))
1105           mode =
1106               DTOSTR_STANDARD; /* Change mode here rather than below because the buffer may not be large enough to hold a large integer. */
1107     
1108         decPt = JS_dtoa(d, dtoaModes[mode], mode >= DTOSTR_FIXED, precision, sign, buffer);
1109         nDigits = buffer.length();
1110     
1111         /* If Infinity, -Infinity, or NaN, return the string regardless of the mode. */
1112         if (decPt != 9999) {
1113           boolean exponentialNotation = false;
1114           int minNDigits = 0; /* Minimum number of significand digits required by mode and precision */
1115           int p;
1116     
1117           switch (mode) {
1118             case DTOSTR_STANDARD:
1119               if (decPt < -5 || decPt > 21) exponentialNotation = true;
1120               else minNDigits = decPt;
1121               break;
1122     
1123             case DTOSTR_FIXED:
1124               if (precision >= 0) minNDigits = decPt + precision;
1125               else minNDigits = decPt;
1126               break;
1127     
1128             case DTOSTR_EXPONENTIAL:
1129               //                    JS_ASSERT(precision > 0);
1130               minNDigits = precision;
1131               /* Fall through */
1132             case DTOSTR_STANDARD_EXPONENTIAL:
1133               exponentialNotation = true;
1134               break;
1135     
1136             case DTOSTR_PRECISION:
1137               //                    JS_ASSERT(precision > 0);
1138               minNDigits = precision;
1139               if (decPt < -5 || decPt > precision) exponentialNotation = true;
1140               break;
1141           }
1142     
1143           /* If the number has fewer than minNDigits, pad it with zeros at the end */
1144           if (nDigits < minNDigits) {
1145             p = minNDigits;
1146             nDigits = minNDigits;
1147             do {
1148               buffer.append('0');
1149             } while (buffer.length() != p);
1150           }
1151     
1152           if (exponentialNotation) {
1153             /* Insert a decimal point if more than one significand digit */
1154             if (nDigits != 1) {
1155               buffer.insert(1, '.');
1156             }
1157             buffer.append('e');
1158             if ((decPt - 1) >= 0) buffer.append('+');
1159             buffer.append(decPt - 1);
1160             //                JS_snprintf(numEnd, bufferSize - (numEnd - buffer), "e%+d", decPt-1);
1161           } else if (decPt != nDigits) {
1162             /* Some kind of a fraction in fixed notation */
1163             //                JS_ASSERT(decPt <= nDigits);
1164             if (decPt > 0) {
1165               /* dd...dd . dd...dd */
1166               buffer.insert(decPt, '.');
1167             } else {
1168               /* 0 . 00...00dd...dd */
1169               for (int i = 0; i < 1 - decPt; i++) buffer.insert(0, '0');
1170               buffer.insert(1, '.');
1171             }
1172           }
1173         }
1174     
1175         /* If negative and neither -0.0 nor NaN, output a leading '-'. */
1176         if (sign[0]
1177             && !(word0(d) == Sign_bit && word1(d) == 0)
1178             && !((word0(d) & Exp_mask) == Exp_mask
1179                 && ((word1(d) != 0) || ((word0(d) & Frac_mask) != 0)))) {
1180           buffer.insert(0, '-');
1181         }
1182       }

1184       public static String numberToString(double value) {
1185         if (Double.isNaN(value)) return "NaN";
1186         if (value == 0.0) return "0";
1187         if (value < 0) return "-" + numberToString(-value);
1188         if (Double.isInfinite(value)) return "Infinity";
1189     
1190         StringBuilder buffer = new StringBuilder();
1191         DToA.JS_dtostr(buffer, DToA.DTOSTR_STANDARD, 0, value);
1192         return buffer.toString();
1193       }
