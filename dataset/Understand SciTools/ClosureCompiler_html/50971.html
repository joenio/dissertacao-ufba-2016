34     /**
35      * Provides a framework for checking code against a set of user configured
36      * conformance rules.  The rules are specified by the ConformanceConfig
37      * proto, which allows for both standard checks (forbidden properties,
38      * variables, or dependencies) and allow for more complex checks using
39      * custom rules than specify
40      *
41      */
42     @GwtIncompatible("com.google.protobuf")
43     public final class CheckConformance implements Callback, CompilerPass {
44     
45       static final DiagnosticType CONFORMANCE_VIOLATION =
46           DiagnosticType.warning(
47               "JSC_CONFORMANCE_VIOLATION",
48               "Violation: {0}{1}{2}");
49     
50       static final DiagnosticType CONFORMANCE_POSSIBLE_VIOLATION =
51           DiagnosticType.warning(
52               "JSC_CONFORMANCE_POSSIBLE_VIOLATION",
53               "Possible violation: {0}{1}{2}");
54     
55       static final DiagnosticType INVALID_REQUIREMENT_SPEC =
56           DiagnosticType.error(
57               "JSC_INVALID_REQUIREMENT_SPEC",
58               "Invalid requirement. Reason: {0}\nRequirement spec:\n{1}");
59     
60       private final AbstractCompiler compiler;
61       private final ImmutableList<Rule> rules;
62     
63       public static interface Rule {
64         /** Perform conformance check */
65         void check(NodeTraversal t, Node n);
66       }
67     
68       /**
69        * @param configs The rules to check.
70        */
71       CheckConformance(
72           AbstractCompiler compiler,
73           ImmutableList<ConformanceConfig> configs) {
74         this.compiler = compiler;
75         // Initialize the map of functions to inspect for renaming candidates.
76         this.rules = initRules(compiler, configs);
77       }
78     
79       @Override
80       public void process(Node externs, Node root) {
81         if (!rules.isEmpty()) {
82           NodeTraversal.traverseRootsEs6(compiler, this, externs, root);
83         }
84       }
85     
86       @Override
87       public final boolean shouldTraverse(NodeTraversal t, Node n, Node parent) {
88         // Don't inspect extern files
89         return !n.isScript() || !t.getInput().getSourceFile().isExtern();
90       }
91     
92       @Override
93       public void visit(NodeTraversal t, Node n, Node parent) {
94         for (int i = 0, len = rules.size(); i < len; i++) {
95           Rule rule = rules.get(i);
96           rule.check(t, n);
97         }
98       }
99     
100       /**
101        * Build the data structures need by this pass from the provided
102        * configurations.
103        */
104       private static ImmutableList<Rule> initRules(
105           AbstractCompiler compiler, ImmutableList<ConformanceConfig> configs) {
106         ImmutableList.Builder<Rule> builder = ImmutableList.builder();
107         List<Requirement> requirements = mergeRequirements(compiler, configs);
108         for (Requirement requirement : requirements) {
109           Rule rule = initRule(compiler, requirement);
110           if (rule != null) {
111             builder.add(rule);
112           }
113         }
114         return builder.build();
115       }
116     
117       private static final Set<String> EXTENDABLE_FIELDS = ImmutableSet.of(
118           "extends", "whitelist", "whitelist_regexp", "only_apply_to", "only_apply_to_regexp");
119     
120       /**
121        * Gets requirements from all configs. Merges whitelists of requirements with 'extends' equal to
122        * 'rule_id' of other rule.
123        */
124       static List<Requirement> mergeRequirements(AbstractCompiler compiler,
125           List<ConformanceConfig> configs) {
126         List<Requirement.Builder> builders = new ArrayList<>();
127         Map<String, Requirement.Builder> extendable = new HashMap<>();
128         for (ConformanceConfig config : configs) {
129           for (Requirement requirement : config.getRequirementList()) {
130             Requirement.Builder builder = requirement.toBuilder();
131             if (requirement.hasRuleId()) {
132               if (requirement.getRuleId().isEmpty()) {
133                 reportInvalidRequirement(compiler, requirement, "empty rule_id");
134                 continue;
135               }
136               if (extendable.containsKey(requirement.getRuleId())) {
137                 reportInvalidRequirement(compiler, requirement,
138                     "two requirements with the same rule_id: " + requirement.getRuleId());
139                 continue;
140               }
141               extendable.put(requirement.getRuleId(), builder);
142             }
143             if (!requirement.hasExtends()) {
144               builders.add(builder);
145             }
146           }
147         }
148     
149         for (ConformanceConfig config : configs) {
150           for (Requirement requirement : config.getRequirementList()) {
151             if (requirement.hasExtends()) {
152               Requirement.Builder existing = extendable.get(requirement.getExtends());
153               if (existing == null) {
154                 reportInvalidRequirement(compiler, requirement,
155                     "no requirement with rule_id: " + requirement.getExtends());
156                 continue;
157               }
158               for (Descriptors.FieldDescriptor field : requirement.getAllFields().keySet()) {
159                 if (!EXTENDABLE_FIELDS.contains(field.getName())) {
160                   reportInvalidRequirement(compiler, requirement,
161                       "extending rules allow only " + EXTENDABLE_FIELDS);
162                 }
163               }
164               existing.addAllWhitelist(requirement.getWhitelistList());
165               existing.addAllWhitelistRegexp(requirement.getWhitelistRegexpList());
166               existing.addAllOnlyApplyTo(requirement.getOnlyApplyToList());
167               existing.addAllOnlyApplyToRegexp(requirement.getOnlyApplyToRegexpList());
168             }
169           }
170         }
171     
172         List<Requirement> requirements = new ArrayList<>(builders.size());
173         for (Requirement.Builder builder : builders) {
174           Requirement requirement = builder.build();
175           checkRequirementList(compiler, requirement, "whitelist");
176           checkRequirementList(compiler, requirement, "whitelist_regexp");
177           checkRequirementList(compiler, requirement, "only_apply_to");
178           checkRequirementList(compiler, requirement, "only_apply_to_regexp");
179           requirements.add(requirement);
180         }
181         return requirements;
182       }
183     
184       private static void checkRequirementList(AbstractCompiler compiler, Requirement requirement,
185           String field) {
186         Set<String> existing = new HashSet<>();
187         for (String value : getRequirementList(requirement, field)) {
188           if (!existing.add(value)) {
189             reportInvalidRequirement(compiler, requirement, "duplicate " + field + " value: " + value);
190           }
191         }
192       }
193     
194       private static List<String> getRequirementList(Requirement requirement, String field) {
195         switch (field) {
196           case "whitelist":
197             return requirement.getWhitelistList();
198           case "whitelist_regexp":
199             return requirement.getWhitelistRegexpList();
200           case "only_apply_to":
201             return requirement.getOnlyApplyToList();
202           case "only_apply_to_regexp":
203             return requirement.getOnlyApplyToRegexpList();
204           default:
205             throw new AssertionError("Unrecognized field: " + field);
206         }
207       }
208     
209       private static Rule initRule(
210           AbstractCompiler compiler, Requirement requirement) {
211         try {
212           switch (requirement.getType()) {
213             case CUSTOM:
214               return new ConformanceRules.CustomRuleProxy(compiler, requirement);
215             case BANNED_CODE_PATTERN:
216               return new ConformanceRules.BannedCodePattern(compiler, requirement);
217             case BANNED_DEPENDENCY:
218               return new ConformanceRules.BannedDependency(compiler, requirement);
219             case BANNED_NAME:
220               return new ConformanceRules.BannedName(compiler, requirement);
221             case BANNED_PROPERTY:
222             case BANNED_PROPERTY_READ:
223             case BANNED_PROPERTY_WRITE:
224             case BANNED_PROPERTY_CALL:
225               return new ConformanceRules.BannedProperty(compiler, requirement);
226             case RESTRICTED_NAME_CALL:
227               return new ConformanceRules.RestrictedNameCall(
228                   compiler, requirement);
229             case RESTRICTED_METHOD_CALL:
230               return new ConformanceRules.RestrictedMethodCall(
231                   compiler, requirement);
232             default:
233               reportInvalidRequirement(
234                   compiler, requirement, "unknown requirement type");
235               return null;
236           }
237         } catch (InvalidRequirementSpec e){
238           reportInvalidRequirement(compiler, requirement, e.getMessage());
239           return null;
240         }
241       }
242     
243       public static class InvalidRequirementSpec extends Exception {
244         InvalidRequirementSpec(String message) {
245           super(message);
246         }
247       }
248     
249       /**
250        * @param requirement
251        */
252       private static void reportInvalidRequirement(
253           AbstractCompiler compiler, Requirement requirement, String reason) {
254         compiler.report(JSError.make(INVALID_REQUIREMENT_SPEC,
255             reason,
256             TextFormat.printToString(requirement)));
257       }
