17     package com.google.javascript.jscomp;
18     
19     import com.google.javascript.rhino.Node;
20     
21     import java.util.Set;
22     
23     /**
24      * A compiler pass that verifies the structure of the AST conforms
25      * to a number of invariants. Because this can add a lot of overhead,
26      * we only run this in development mode.
27      *
28      * @author nicksantos@google.com (Nick Santos)
29      */
30     class SanityCheck implements CompilerPass {
31     
32       static final DiagnosticType CANNOT_PARSE_GENERATED_CODE =
33           DiagnosticType.error("JSC_CANNOT_PARSE_GENERATED_CODE",
34               "Internal compiler error. Cannot parse generated code: {0}");
35     
36       static final DiagnosticType GENERATED_BAD_CODE = DiagnosticType.error(
37           "JSC_GENERATED_BAD_CODE",
38           "Internal compiler error. Generated bad code." +
39           "----------------------------------------\n" +
40           "Expected:\n{0}\n" +
41           "----------------------------------------\n" +
42           "Actual:\n{1}");
43     
44       static final DiagnosticType EXTERN_PROPERTIES_CHANGED =
45           DiagnosticType.error("JSC_EXTERN_PROPERTIES_CHANGED",
46               "Internal compiler error. Extern properties modified.");
47     
48       private final AbstractCompiler compiler;
49       private final AstValidator astValidator;
50     
51       SanityCheck(AbstractCompiler compiler) {
52         this.compiler = compiler;
53         this.astValidator = new AstValidator(compiler);
54       }
55     
56       @Override
57       public void process(Node externs, Node root) {
58         sanityCheckAst(externs, root);
59         sanityCheckNormalization(externs, root);
60         sanityCheckCodeGeneration(root);
61         sanityCheckVars(externs, root);
62         sanityCheckExternProperties(externs);
63       }
64     
65       /**
66        * Sanity check the AST is structurally accurate.
67        */
68       private void sanityCheckAst(Node externs, Node root) {
69         astValidator.validateCodeRoot(externs);
70         astValidator.validateCodeRoot(root);
71       }
72     
73       private void sanityCheckVars(Node externs, Node root) {
74         if (compiler.getLifeCycleStage().isNormalized()) {
75           (new VarCheck(compiler, true)).process(externs, root);
76         }
77       }
78     
79       /**
80        * Sanity checks code generation by performing it once, parsing the result,
81        * then generating code from the second parse tree to verify that it matches
82        * the code generated from the first parse tree.
83        *
84        * @return The regenerated parse tree. Null on error.
85        */
86       private Node sanityCheckCodeGeneration(Node root) {
87         if (compiler.hasHaltingErrors()) {
88           // Don't even bother checking code generation if we already know the
89           // the code is bad.
90           return null;
91         }
92     
93         String source = compiler.toSource(root);
94         Node root2 = compiler.parseSyntheticCode(source);
95         if (compiler.hasHaltingErrors()) {
96           compiler.report(JSError.make(CANNOT_PARSE_GENERATED_CODE,
97                   Strings.truncateAtMaxLength(source, 100, true)));
98     
99           // Throw an exception, so that the infrastructure will tell us
100           // which pass violated the sanity check.
101           throw new IllegalStateException("Sanity Check failed");
102         }
103     
104         String source2 = compiler.toSource(root2);
105         if (!source.equals(source2)) {
106           compiler.report(JSError.make(GENERATED_BAD_CODE, source, source2));
107     
108           // Throw an exception, so that the infrastructure will tell us
109           // which pass violated the sanity check.
110           throw new IllegalStateException("Sanity Check failed");
111         }
112     
113         return root2;
114       }
115     
116       /**
117        * Sanity checks the AST. This is by verifying the normalization passes do
118        * nothing.
119        */
120       private void sanityCheckNormalization(Node externs, Node root) {
121         // Verify nothing has inappropriately denormalize the AST.
122         CodeChangeHandler handler = new ForbiddenChange();
123         compiler.addChangeHandler(handler);
124     
125         // TODO(johnlenz): Change these normalization checks Preconditions and
126         // Exceptions into Errors so that it is easier to find the root cause
127         // when there are cascading issues.
128         new PrepareAst(compiler, true).process(null, root);
129         if (compiler.getLifeCycleStage().isNormalized()) {
130           (new Normalize(compiler, true)).process(externs, root);
131     
132           if (compiler.getLifeCycleStage().isNormalizedUnobfuscated()) {
133             boolean checkUserDeclarations = true;
134             CompilerPass pass = new Normalize.VerifyConstants(
135                 compiler, checkUserDeclarations);
136             pass.process(externs, root);
137           }
138         }
139     
140         compiler.removeChangeHandler(handler);
141       }
142     
143       private void sanityCheckExternProperties(Node externs) {
144         Set<String> externProperties = compiler.getExternProperties();
145         if (externProperties == null) {
146           // GatherExternProperties hasn't run yet. Don't report a violation.
147           return;
148         }
149         (new GatherExternProperties(compiler)).process(externs, null);
150         if (!compiler.getExternProperties().equals(externProperties)) {
151           compiler.report(JSError.make(EXTERN_PROPERTIES_CHANGED));
152           // Throw an exception, so that the infrastructure will tell us
153           // which pass violated the sanity check.
154           throw new IllegalStateException("Sanity Check failed");
155         }
156       }
157     }
