40     package com.google.javascript.rhino.testing;

42     import static com.google.common.truth.Truth.assertThat;
43     import static org.junit.Assert.assertEquals;
44     
45     import com.google.common.collect.Iterables;
46     import com.google.javascript.rhino.ErrorReporter;
47     import com.google.javascript.rhino.jstype.JSType;

54     /**
55      * Helper methods for making assertions about the validity of types.
56      * @author nicksantos@google.com (Nick Santos)
57      */
58     public class Asserts {
59       private Asserts() {} // all static
60     
61       public static JSType assertResolvesToSame(JSType type) {
62         Assert.assertSame(type, assertValidResolve(type));
63         return type;
64       }
65     
66       /** @return The resolved type */
67       public static JSType assertValidResolve(JSType type) {
68         return assertValidResolve(type, MapBasedScope.emptyScope());
69       }
70     
71       /** @return The resolved type */
72       public static JSType assertValidResolve(
73           JSType type, StaticTypedScope<JSType> scope) {
74         ErrorReporter t = TestErrorReporter.forNoExpectedReports();
75         JSType resolvedType = type.resolve(t, scope);
76         assertTypeEquals("JSType#resolve should not affect object equality",
77             type, resolvedType);
78         return resolvedType;
79       }
80     
81       public static void assertTypeNotEquals(JSType a, JSType b) {
82         assertTypeNotEquals("", a, b);
83       }
84     
85       public static void assertTypeNotEquals(String message, JSType a, JSType b) {
86         Assert.assertFalse(
87             message +
88             (message.isEmpty() ? "" : "\n") +
89             "Type: " + b + "\n",
90             a.isEquivalentTo(b));
91         Assert.assertFalse(
92             message +
93             " Equals is not symmetric.\n" +
94             "Type: " + b + "\n",
95             b.isEquivalentTo(a));
96       }
97     
98       public static void assertTypeEquals(JSType a, JSType b) {
99         assertTypeEquals("", a, b);
100       }
101     
102       public static void assertTypeEquals(String message, JSType a, JSType b) {
103         assertEquals(
104             "Both types must be null, or both must be non-null " + a + "," + b,
105             (b == null), (a == null));
106         if (a == null) {
107           return;
108         }
109         Assert.assertTrue(
110             message +
111             (message.isEmpty() ? "" : "\n") +
112             "Expected: " + a + "\n" +
113             "Actual  : " + b,
114             a.isEquivalentTo(b));
115         Assert.assertTrue(
116             message +
117             " Equals is not symmetric.\n" +
118             "Expected: " + b + "\n" +
119             "Actual  : " + a,
120             b.isEquivalentTo(a));
121       }
122     
123       public static <T extends JSType, S extends JSType> void
124           assertTypeCollectionEquals(Iterable<T> a, Iterable<S> b) {
125         assertThat(b).hasSize(Iterables.size(a));
126         Iterator<T> aIterator = a.iterator();
127         Iterator<S> bIterator = b.iterator();
128         while (aIterator.hasNext()) {
129           assertTypeEquals(aIterator.next(), bIterator.next());
130         }
131       }
132     
133       /**
134        * For the given equivalent types, run all type operations that
135        * should have trivial solutions (getGreatestSubtype, isEquivalentTo, etc)
136        */
137       public static void assertEquivalenceOperations(JSType a, JSType b) {
138         Assert.assertTrue(a.isEquivalentTo(b));
139         Assert.assertTrue(a.isEquivalentTo(a));
140         Assert.assertTrue(b.isEquivalentTo(b));
141         Assert.assertTrue(b.isEquivalentTo(a));
142     
143         Assert.assertTrue(a.isSubtype(b));
144         Assert.assertTrue(a.isSubtype(a));
145         Assert.assertTrue(b.isSubtype(b));
146         Assert.assertTrue(b.isSubtype(a));
147     
148         assertTypeEquals(a, a.getGreatestSubtype(b));
149         assertTypeEquals(a, a.getGreatestSubtype(a));
150         assertTypeEquals(a, b.getGreatestSubtype(b));
151         assertTypeEquals(a, b.getGreatestSubtype(a));
152     
153         assertTypeEquals(a, a.getLeastSupertype(b));
154         assertTypeEquals(a, a.getLeastSupertype(a));
155         assertTypeEquals(a, b.getLeastSupertype(b));
156         assertTypeEquals(a, b.getLeastSupertype(a));
157     
158         Assert.assertTrue(a.canCastTo(b));
159         Assert.assertTrue(a.canCastTo(a));
160         Assert.assertTrue(b.canCastTo(b));
161         Assert.assertTrue(b.canCastTo(a));
162       }
163     }
