332       @Override
333       public void process(Node externs, Node root) {
334         Preconditions.checkNotNull(warnings, "Cannot rerun GlobalTypeInfo.process");
335         Preconditions.checkArgument(externs == null || externs.isSyntheticBlock());
336         Preconditions.checkArgument(root.isSyntheticBlock());
337         this.globalScope = new NTIScope(root, null, ImmutableList.<String>of(), commonTypes);
338         this.globalScope.addUnknownTypeNames(this.unknownTypeNames);
339         this.unknownTypeNames = null; // Don't retain the LinkedHashSet
340         scopes.add(this.globalScope);
341     
342         // Processing of a scope is split into many separate phases, and it's not
343         // straightforward to remember which phase does what.
344     
345         // (1) Find names of classes, interfaces, typedefs, enums, and namespaces
346         //   defined in the global scope.
347         CollectNamedTypes rootCnt = new CollectNamedTypes(this.globalScope);
348         if (externs != null) {
349           NodeTraversal.traverseEs6(compiler, externs, rootCnt);
350         }
351         NodeTraversal.traverseEs6(compiler, root, rootCnt);
352         // (2) Determine the type represented by each typedef and each enum
353         this.globalScope.resolveTypedefs(typeParser);
354         this.globalScope.resolveEnums(typeParser);
355         // (3) Repeat steps 1-2 for all the other scopes (outer-to-inner)
356         for (int i = 1; i < scopes.size(); i++) {
357           NTIScope s = scopes.get(i);
358           CollectNamedTypes cnt = new CollectNamedTypes(s);
359           NodeTraversal.traverseEs6(compiler, s.getBody(), cnt);
360           s.resolveTypedefs(typeParser);
361           s.resolveEnums(typeParser);
362           if (NewTypeInference.measureMem) {
363             NewTypeInference.updatePeakMem();
364           }
365         }
366     
367         // If the Function constructor isn't defined, we cannot create function
368         // types. Exit early.
369         if (this.commonTypes.getFunctionType() == null) {
370           warnings.add(JSError.make(root, FUNCTION_CONSTRUCTOR_NOT_DEFINED));
371           return;
372         }
373     
374         // (4) The bulk of the global-scope processing happens here:
375         //     - Create scopes for functions
376         //     - Declare properties on types
377         ProcessScope rootPs = new ProcessScope(this.globalScope);
378         if (externs != null) {
379           NodeTraversal.traverseEs6(compiler, externs, rootPs);
380         }
381         NodeTraversal.traverseEs6(compiler, root, rootPs);
382         // (5) Things that must happen after the traversal of the scope
383         rootPs.finishProcessingScope();
384     
385         // (6) Repeat steps 4-5 for all the other scopes (outer-to-inner)
386         for (int i = 1; i < scopes.size(); i++) {
387           NTIScope s = scopes.get(i);
388           ProcessScope ps = new ProcessScope(s);
389           NodeTraversal.traverseEs6(compiler, s.getBody(), ps);
390           ps.finishProcessingScope();
391           if (NewTypeInference.measureMem) {
392             NewTypeInference.updatePeakMem();
393           }
394         }
395     
396         // (7) Adjust types of properties based on inheritance information.
397         //     Report errors in the inheritance chain. Do Window last.
398         RawNominalType win = null;
399         for (Map.Entry<Node, RawNominalType> entry : nominaltypesByNode.entrySet()) {
400           RawNominalType rawType = entry.getValue();
401           if (rawType.getName().equals(WINDOW_CLASS) && entry.getKey().isFromExterns()) {
402             win = rawType;
403             continue;
404           }
405           checkAndFinalizeNominalType(rawType);
406         }
407         if (win != null) {
408           // Copy properties from window to Window.prototype, because in rare cases
409           // people pass window around rather than using it directly.
410           Namespace winNs = this.globalScope.getNamespace(WINDOW_INSTANCE);
411           if (winNs != null) {
412             winNs.copyWindowProperties(this.commonTypes, win);
413           }
414           checkAndFinalizeNominalType(win);
415         }
416     
417         nominaltypesByNode = null;
418         propertyDefs = null;
419         for (NTIScope s : scopes) {
420           s.finalizeScope();
421         }
422         Map<Node, String> unknownTypes = typeParser.getUnknownTypesMap();
423         for (Map.Entry<Node, String> unknownTypeEntry : unknownTypes.entrySet()) {
424           this.warnings.add(JSError.make(unknownTypeEntry.getKey(),
425                   UNRECOGNIZED_TYPE_NAME, unknownTypeEntry.getValue()));
426         }
427         // The jsdoc parser doesn't have access to the error functions in the jscomp
428         // package, so we collect its warnings here.
429         for (JSError warning : typeParser.getWarnings()) {
430           this.warnings.add(warning);
431         }
432         this.warnings = null;
433         this.varNameGen = null;
434         this.funNameGen = null;
435     
436         // If a scope s1 contains a scope s2, then s2 must be before s1 in scopes.
437         // The type inference relies on this fact to process deeper scopes
438         // before shallower scopes.
439         Collections.reverse(scopes);
440       }

442       private Collection<PropertyDef> getPropDefsFromInterface(
443           NominalType nominalType, String pname) {
444         Preconditions.checkArgument(nominalType.isFinalized());
445         Preconditions.checkArgument(
446             nominalType.isInterface() || nominalType.isBuiltinObject());
447         if (nominalType.getPropDeclaredType(pname) == null) {
448           return ImmutableSet.of();
449         } else if (propertyDefs.get(nominalType.getId(), pname) != null) {
450           return ImmutableSet.of(propertyDefs.get(nominalType.getId(), pname));
451         }
452         ImmutableSet.Builder<PropertyDef> result = ImmutableSet.builder();
453         for (NominalType interf : nominalType.getInstantiatedInterfaces()) {
454           result.addAll(getPropDefsFromInterface(interf, pname));
455         }
456         return result.build();
457       }

459       private PropertyDef getPropDefFromClass(
460           NominalType nominalType, String pname) {
461         while (nominalType.getPropDeclaredType(pname) != null) {
462           Preconditions.checkArgument(nominalType.isFinalized());
463           Preconditions.checkArgument(nominalType.isClass());
464     
465           if (propertyDefs.get(nominalType.getId(), pname) != null) {
466             return propertyDefs.get(nominalType.getId(), pname);
467           }
468           nominalType = nominalType.getInstantiatedSuperclass();
469         }
470         return null;
471       }

473       private void checkAndFinalizeNominalType(RawNominalType rawType) {
474         if (rawType.isFinalized()) {
475           return;
476         }
477         NominalType superClass = rawType.getSuperClass();
478         Set<String> nonInheritedPropNames = rawType.getAllOwnProps();
479         if (superClass != null && !superClass.isFinalized()) {
480           checkAndFinalizeNominalType(superClass.getRawNominalType());
481         }
482         for (NominalType superInterf : rawType.getInterfaces()) {
483           if (!superInterf.isFinalized()) {
484             checkAndFinalizeNominalType(superInterf.getRawNominalType());
485           }
486         }
487     
488         Multimap<String, DeclaredFunctionType> propMethodTypesToProcess =
489             LinkedHashMultimap.create();
490         Multimap<String, JSType> propTypesToProcess = LinkedHashMultimap.create();
491         // Collect inherited types for extended classes
492         if (superClass != null) {
493           Preconditions.checkState(superClass.isFinalized());
494           // TODO(blickly): Can we optimize this to skip unnecessary iterations?
495           for (String pname : superClass.getAllPropsOfClass()) {
496             nonInheritedPropNames.remove(pname);
497             checkSuperProperty(rawType, superClass, pname,
498                 propMethodTypesToProcess, propTypesToProcess);
499           }
500         }
501     
502         // Collect inherited types for extended/implemented interfaces
503         for (NominalType superInterf : rawType.getInterfaces()) {
504           Preconditions.checkState(superInterf.isFinalized());
505           for (String pname : superInterf.getAllPropsOfInterface()) {
506             nonInheritedPropNames.remove(pname);
507             checkSuperProperty(rawType, superInterf, pname,
508                 propMethodTypesToProcess, propTypesToProcess);
509           }
510         }
511     
512         // Munge inherited types of methods
513         for (String pname : propMethodTypesToProcess.keySet()) {
514           Collection<DeclaredFunctionType> methodTypes =
515               propMethodTypesToProcess.get(pname);
516           Preconditions.checkState(!methodTypes.isEmpty());
517           PropertyDef localPropDef =
518               propertyDefs.get(rawType, pname);
519           // To find the declared type of a method, we must meet declared types
520           // from all inherited methods.
521           DeclaredFunctionType superMethodType = DeclaredFunctionType.meet(methodTypes);
522           DeclaredFunctionType localMethodType = localPropDef.methodType;
523           boolean getsTypeFromParent = getsTypeInfoFromParentMethod(localPropDef);
524           if (superMethodType == null) {
525             // If the inherited types are not compatible, pick one.
526             superMethodType = methodTypes.iterator().next();
527             warnings.add(JSError.make(localPropDef.defSite,
528                     SUPER_INTERFACES_HAVE_INCOMPATIBLE_PROPERTIES,
529                     rawType.getName(), pname, methodTypes.toString()));
530           } else if (getsTypeFromParent
531               && localMethodType.getMaxArity() > superMethodType.getMaxArity()) {
532             // When getsTypeFromParent is true, we will miss the invalid override
533             // earlier, so we check here.
534             warnings.add(JSError.make(
535                 localPropDef.defSite, INVALID_PROP_OVERRIDE, pname,
536                 superMethodType.toFunctionType().toString(),
537                 localMethodType.toFunctionType().toString()));
538           }
539           DeclaredFunctionType updatedMethodType = localMethodType
540               .withTypeInfoFromSuper(superMethodType, getsTypeFromParent);
541           localPropDef.updateMethodType(updatedMethodType);
542           propTypesToProcess.put(pname,
543               commonTypes.fromFunctionType(updatedMethodType.toFunctionType()));
544         }
545     
546         // Check inherited types of all props
547      add_interface_props:
548         for (String pname : propTypesToProcess.keySet()) {
549           Collection<JSType> defs = propTypesToProcess.get(pname);
550           Preconditions.checkState(!defs.isEmpty());
551           JSType resultType = JSType.TOP;
552           for (JSType inheritedType : defs) {
553             resultType = JSType.meet(resultType, inheritedType);
554             if (!resultType.isBottom()) {
555               resultType = inheritedType;
556             } else {
557               warnings.add(JSError.make(
558                   rawType.getDefSite(),
559                   SUPER_INTERFACES_HAVE_INCOMPATIBLE_PROPERTIES,
560                   rawType.getName(), pname,
561                   defs.toString()));
562               continue add_interface_props;
563             }
564           }
565           // TODO(dimvar): check if we can have @const props here
566           rawType.addProtoProperty(pname, null, resultType, false);
567         }
568     
569         // Warn when inheriting from incompatible IObject types
570         if (rawType.inheritsFromIObject()) {
571           JSType wrapped = rawType.getInstanceAsJSType();
572           if (wrapped.getIndexType() == null) {
573             warnings.add(JSError.make(
574                 rawType.getDefSite(),
575                 SUPER_INTERFACES_HAVE_INCOMPATIBLE_PROPERTIES,
576                 rawType.getName(),
577                 "IObject<K,V>#index",
578                 "the keys K have types that can't be joined."));
579           } else if (wrapped.getIndexedType() == null) {
580             warnings.add(JSError.make(
581                 rawType.getDefSite(),
582                 SUPER_INTERFACES_HAVE_INCOMPATIBLE_PROPERTIES,
583                 rawType.getName(),
584                 "IObject<K,V>#index",
585                 "the values V should have a common subtype."));
586           }
587         }
588     
589         // Warn for a prop declared with @override that isn't overriding anything.
590         for (String pname : nonInheritedPropNames) {
591           PropertyDef propDef = propertyDefs.get(rawType, pname);
592           Preconditions.checkState(propDef != null || rawType.getName().equals(WINDOW_CLASS));
593           if (propDef != null) {
594             Node propDefsite = propDef.defSite;
595             JSDocInfo jsdoc = NodeUtil.getBestJSDocInfo(propDefsite);
596             if (jsdoc != null && jsdoc.isOverride()) {
597               warnings.add(JSError.make(propDefsite, UNKNOWN_OVERRIDE,
598                       pname, rawType.getName()));
599             }
600           }
601         }
602     
603         // Finalize nominal type once all properties are added.
604         rawType.finalize();
605       }

607       // TODO(dimvar): the finalization method and this one should be cleaned up;
608       // they are very hard to understand.
609       private void checkSuperProperty(
610           RawNominalType current, NominalType superType, String pname,
611           Multimap<String, DeclaredFunctionType> propMethodTypesToProcess,
612           Multimap<String, JSType> propTypesToProcess) {
613         JSType inheritedPropType = superType.getPropDeclaredType(pname);
614         if (inheritedPropType == null) {
615           // No need to go further for undeclared props.
616           return;
617         }
618         Collection<PropertyDef> inheritedPropDefs;
619         if (superType.isInterface()) {
620           inheritedPropDefs = getPropDefsFromInterface(superType, pname);
621         } else {
622           inheritedPropDefs =
623               ImmutableSet.of(getPropDefFromClass(superType, pname));
624         }
625         if (superType.isInterface() && current.isClass()
626             && !current.mayHaveProp(pname)) {
627           warnings.add(JSError.make(
628               inheritedPropDefs.iterator().next().defSite,
629               INTERFACE_METHOD_NOT_IMPLEMENTED,
630               pname, superType.toString(), current.toString()));
631           return;
632         }
633         PropertyDef localPropDef = propertyDefs.get(current, pname);
634         JSType localPropType = localPropDef == null
635             ? null : current.getInstancePropDeclaredType(pname);
636         if (localPropDef != null && superType.isClass()
637             && localPropType != null
638             && localPropType.getFunType() != null
639             && superType.hasConstantProp(pname)) {
640           // TODO(dimvar): This doesn't work for multiple levels in the hierarchy.
641           // Clean up how we process inherited properties and then fix this.
642           warnings.add(JSError.make(
643               localPropDef.defSite, CANNOT_OVERRIDE_FINAL_METHOD, pname));
644           return;
645         }
646         if (localPropType == null && superType.isInterface()) {
647           // Add property from interface to class
648           propTypesToProcess.put(pname, inheritedPropType);
649         } else if (localPropType != null
650             && !getsTypeInfoFromParentMethod(localPropDef)
651             && !isValidOverride(localPropType, inheritedPropType)) {
652           warnings.add(JSError.make(
653               localPropDef.defSite, INVALID_PROP_OVERRIDE, pname,
654               inheritedPropType.toString(), localPropType.toString()));
655         } else if (localPropType != null && localPropDef.methodType != null) {
656           // If we are looking at a method definition, munging may be needed
657           for (PropertyDef inheritedPropDef : inheritedPropDefs) {
658             if (inheritedPropDef.methodType != null) {
659               propMethodTypesToProcess.put(pname,
660                   inheritedPropDef.methodType.substituteNominalGenerics(superType));
661             }
662           }
663         }
664       }

666       private boolean isValidOverride(JSType localPropType, JSType inheritedPropType) {
667         FunctionType localFunType = localPropType.getFunTypeIfSingletonObj();
668         FunctionType inheritedFunType = inheritedPropType.getFunTypeIfSingletonObj();
669         if (localFunType == null) {
670           return localPropType.isSubtypeOf(inheritedPropType);
671         } else if (inheritedFunType == null) {
672           return false;
673         } else {
674           return localFunType.isValidOverride(inheritedFunType);
675         }

688        * Collects names of classes, interfaces, namespaces, typedefs and enums.
689        * This way, if a type name appears before its declaration, we know what
690        * it refers to.

696           this.currentScope = s;

699         @Override
700         public void visit(NodeTraversal t, Node n, Node parent) {
701           switch (n.getType()) {
702             case FUNCTION: {
703               visitFunctionEarly(n);
704               break;
705             }
706             case VAR: {
707               Node nameNode = n.getFirstChild();
708               String varName = nameNode.getString();
709               if (NodeUtil.isNamespaceDecl(nameNode)) {
710                 visitObjlitNamespace(nameNode);
711               } else if (NodeUtil.isTypedefDecl(nameNode)) {
712                 visitTypedef(nameNode);
713               } else if (NodeUtil.isEnumDecl(nameNode)) {
714                 visitEnum(nameNode);
715               } else if (isAliasedNamespaceDefinition(nameNode)) {
716                 visitAliasedNamespace(nameNode);
717               } else if (varName.equals(WINDOW_INSTANCE)
718                   && nameNode.isFromExterns()) {
719                 visitWindowVar(nameNode);
720               } else if (isCtorDefinedByCall(nameNode)) {
721                 visitNewCtorDefinedByCall(nameNode);
722               }
723               if (!n.isFromExterns()
724                   && !this.currentScope.isDefinedLocally(varName, false)) {
725                 // Add a dummy local to avoid shadowing errors, and to calculate
726                 // escaped variables.
727                 this.currentScope.addLocal(varName, JSType.UNKNOWN, false, false);
728               }
729               break;
730             }
731             case NAME: {
732               if (this.currentScope.isFunction()) {
733                 NTIScope.mayRecordEscapedVar(this.currentScope, n.getString());
734               }
735               break;
736             }
737             case EXPR_RESULT: {
738               Node expr = n.getFirstChild();
739               switch (expr.getType()) {
740                 case ASSIGN:
741                   Node lhs = expr.getFirstChild();
742                   if (isCtorDefinedByCall(lhs)) {
743                     visitNewCtorDefinedByCall(lhs);
744                     return;
745                   }
746                   if (!lhs.isGetProp()) {
747                     return;
748                   }
749                   expr = lhs;
750                   // fall through
751                 case GETPROP:
752                   if (isPrototypeProperty(expr)
753                       || NodeUtil.referencesThis(expr)
754                       || !expr.isQualifiedName()) {
755                     // Class & prototype properties are handled in ProcessScope
756                     return;
757                   }
758                   processQualifiedDefinition(expr);
759                   break;
760               }
761               break;
762             }
763           }
764         }

766         private void visitWindowVar(Node nameNode) {
767           JSType typeInJsdoc = getVarTypeFromAnnotation(nameNode, this.currentScope);
768           if (!this.currentScope.isDefinedLocally(WINDOW_INSTANCE, false)) {
769             this.currentScope.addLocal(WINDOW_INSTANCE, typeInJsdoc, false, true);
770             return;
771           }
772           // The externs may contain multiple definitions of window, or they may add
773           // properties to the window instance before "var window" is defined.
774           // In those cases, we want to pick the definition that has the Window
775           // nominal type.
776           NominalType maybeWin = typeInJsdoc == null
777               ? null : typeInJsdoc.getNominalTypeIfSingletonObj();
778           if (maybeWin != null && maybeWin.getName().equals(WINDOW_CLASS)) {
779             this.currentScope.addLocal(WINDOW_INSTANCE, typeInJsdoc, false, true);
780           }
781         }

783         private void processQualifiedDefinition(Node qnameNode) {
784           Preconditions.checkArgument(qnameNode.isGetProp());
785           Preconditions.checkArgument(qnameNode.isQualifiedName());
786           Node recv = qnameNode.getFirstChild();
787           if (!currentScope.isNamespace(recv)
788               && !mayCreateFunctionNamespace(recv)
789               && !mayCreateWindowNamespace(recv)) {
790             return;
791           }
792           if (NodeUtil.isNamespaceDecl(qnameNode)) {
793             visitObjlitNamespace(qnameNode);
794           } else if (NodeUtil.isTypedefDecl(qnameNode)) {
795             visitTypedef(qnameNode);
796           } else if (NodeUtil.isEnumDecl(qnameNode)) {
797             visitEnum(qnameNode);
798           } else if (isAliasedNamespaceDefinition(qnameNode)) {
799             visitAliasedNamespace(qnameNode);
800           } else if (isQualifiedFunctionDefinition(qnameNode)) {
801             maybeAddFunctionToNamespace(qnameNode);
802           }
803         }

805         private boolean isAliasedNamespaceDefinition(Node qnameNode) {
806           Node rhs = NodeUtil.getRValueOfLValue(qnameNode);
807           if (rhs == null || !rhs.isQualifiedName()) {
808             return false;
809           }
810           JSDocInfo jsdoc = NodeUtil.getBestJSDocInfo(qnameNode);
811           if (jsdoc == null) {
812             return qnameNode.isFromExterns()
813                 // If the aliased namespace is a variable, do the aliasing at
814                 // declaration, not at a random assignment later.
815                 && (qnameNode.getParent().isVar() || !qnameNode.isName())
816                 && this.currentScope.getNamespace(QualifiedName.fromNode(rhs)) != null;
817           } else {
818             return jsdoc.isConstructorOrInterface() || jsdoc.hasConstAnnotation();
819           }
820         }

822         private boolean isQualifiedFunctionDefinition(Node qnameNode) {
823           Preconditions.checkArgument(qnameNode.isGetProp());
824           Preconditions.checkArgument(qnameNode.isQualifiedName());
825           Node parent = qnameNode.getParent();
826           JSDocInfo jsdoc = NodeUtil.getBestJSDocInfo(qnameNode);
827           return parent.isAssign()
828               && parent.getParent().isExprResult()
829               && parent.getLastChild().isFunction()
830               && (jsdoc == null || jsdoc.containsFunctionDeclaration());
831         }

833         // Returns true iff it creates a new function namespace
834         private boolean mayCreateFunctionNamespace(Node qnameNode) {
835           if (!qnameNode.isQualifiedName()) {
836             return false;
837           }
838           QualifiedName qname = QualifiedName.fromNode(qnameNode);
839           Preconditions.checkState(!currentScope.isNamespace(qname));
840           if (!currentScope.isKnownFunction(qname)) {
841             return false;
842           }
843           if (qnameNode.isGetProp()) {
844             markAssignNodeAsAnalyzed(qnameNode.getGrandparent());
845           }
846           NTIScope s;
847           if (qnameNode.isName()) {
848             // s is the scope that contains the function
849             s = currentScope.getScope(qnameNode.getString()).getParent();
850           } else {
851             s = currentScope;
852           }
853           s.addFunNamespace(qnameNode);
854           return true;
855         }

858           if (qnameNode.isName() && qnameNode.getString().equals(WINDOW_INSTANCE)
859               && this.currentScope.isGlobalVar(WINDOW_INSTANCE)) {
860             globalScope.addNamespaceLit(qnameNode);
861             return true;
862           }
863           return false;

866         private void visitObjlitNamespace(Node qnameNode) {
867           if (qnameNode.isGetProp()) {
868             markAssignNodeAsAnalyzed(qnameNode.getParent());
869           }
870           if (currentScope.isDefined(qnameNode)) {
871             if (qnameNode.isGetProp()
872                 && !NodeUtil.getRValueOfLValue(qnameNode).isOr()) {
873               warnings.add(JSError.make(qnameNode, REDECLARED_PROPERTY,
874                       qnameNode.getLastChild().getString(),
875                       qnameNode.getFirstChild().getQualifiedName()));
876             }
877             return;
878           }
879           currentScope.addNamespaceLit(qnameNode);
880         }

884             maybeAssign.putBooleanProp(Node.ANALYZED_DURING_GTI, true);

887             Preconditions.checkState(maybeAssign.isExprResult());

891         private void visitTypedef(Node qnameNode) {
892           Preconditions.checkState(qnameNode.isQualifiedName());
893           qnameNode.putBooleanProp(Node.ANALYZED_DURING_GTI, true);
894           if (NodeUtil.getRValueOfLValue(qnameNode) != null) {
895             warnings.add(JSError.make(qnameNode, CANNOT_INIT_TYPEDEF));
896           }
897           if (currentScope.isDefined(qnameNode)) {
898             return;
899           }
900           JSDocInfo jsdoc = NodeUtil.getBestJSDocInfo(qnameNode);
901           Typedef td = Typedef.make(jsdoc.getTypedefType());
902           currentScope.addTypedef(qnameNode, td);
903         }

905         private void visitEnum(Node qnameNode) {
906           Preconditions.checkState(qnameNode.isQualifiedName());
907           qnameNode.putBooleanProp(Node.ANALYZED_DURING_GTI, true);
908           if (currentScope.isDefined(qnameNode)) {
909             return;
910           }
911           Node init = NodeUtil.getRValueOfLValue(qnameNode);
912           // First check if the definition is an alias of a previous enum.
913           if (init != null && init.isQualifiedName()) {
914             EnumType et = currentScope.getEnum(QualifiedName.fromNode(init));
915             if (et != null) {
916               currentScope.addNamespace(qnameNode, et);
917               return;
918             }
919           }
920           // Then check if the enum initializer is an object literal.
921           if (init == null || !init.isObjectLit() || init.getFirstChild() == null) {
922             warnings.add(JSError.make(qnameNode, MALFORMED_ENUM));
923             return;
924           }
925           // Last, read the object-literal properties and create the EnumType.
926           JSDocInfo jsdoc = NodeUtil.getBestJSDocInfo(qnameNode);
927           Set<String> propNames = new LinkedHashSet<>();
928           for (Node prop : init.children()) {
929             String pname = NodeUtil.getObjectLitKeyName(prop);
930             if (propNames.contains(pname)) {
931               warnings.add(JSError.make(qnameNode, DUPLICATE_PROP_IN_ENUM, pname));
932             }
933             propNames.add(pname);
934           }
935           currentScope.addNamespace(qnameNode,
936               EnumType.make(
937                   qnameNode.getQualifiedName(),
938                   jsdoc.getEnumParameterType(),
939                   ImmutableSet.copyOf(propNames)));
940         }

942         private void visitFunctionEarly(Node fn) {
943           JSDocInfo fnDoc = NodeUtil.getBestJSDocInfo(fn);
944           Node nameNode = NodeUtil.getNameNode(fn);
945           String internalName = createFunctionInternalName(fn, nameNode);
946           boolean isRedeclaration;
947           if (nameNode == null || !nameNode.isQualifiedName()) {
948             isRedeclaration = false;
949           } else if (nameNode.isName()) {
950             isRedeclaration = currentScope.isDefinedLocally(nameNode.getString(), false);
951           } else {
952             Preconditions.checkState(
953                 nameNode.isGetProp(), "Expected getprop, found %s", nameNode.getType());
954             isRedeclaration = currentScope.isDefined(nameNode);
955             if (isRedeclaration && fnDoc != null) {
956               nameNode.getParent().putBooleanProp(Node.ANALYZED_DURING_GTI, true);
957               warnings.add(JSError.make(nameNode, REDECLARED_PROPERTY,
958                       nameNode.getLastChild().getString(),
959                       nameNode.getFirstChild().getQualifiedName()));
960             }
961           }
962           NTIScope fnScope = new NTIScope(fn, this.currentScope, collectFormals(fn, fnDoc), null);
963           if (!fn.isFromExterns()) {
964             scopes.add(fnScope);
965           }
966           this.currentScope.addLocalFunDef(internalName, fnScope);
967           maybeRecordNominalType(fn, nameNode, fnDoc, isRedeclaration);
968         }

970         private String createFunctionInternalName(Node fn, Node nameNode) {
971           String internalName = null;
972           if (nameNode == null || !nameNode.isName()
973               || nameNode.getParent().isAssign()) {
974             // Anonymous functions, qualified names, and stray assignments
975             // (eg, f = function(x) { ... }; ) get gensymed names.
976             internalName = ANON_FUN_PREFIX + funNameGen.generateNextName();
977             anonFunNames.put(fn, internalName);
978           } else if (currentScope.isDefinedLocally(nameNode.getString(), false)) {
979             String fnName = nameNode.getString();
980             Preconditions.checkState(!fnName.contains("."));
981             internalName = ANON_FUN_PREFIX + funNameGen.generateNextName();
982             anonFunNames.put(fn, internalName);
983           } else {
984             // fnNameNode is undefined simple name
985             internalName = nameNode.getString();
986           }
987           return internalName;
988         }

990         private ArrayList<String> collectFormals(Node fn, JSDocInfo fnDoc) {
991           Preconditions.checkArgument(fn.isFunction());
992           // Collect the names of the formals.
993           // If a formal is a placeholder for variable arity, eg,
994           // /** @param {...?} var_args */ function f(var_args) { ... }
995           // then we don't collect it.
996           // But to decide that we can't just use the jsdoc b/c the type parser
997           // may ignore the jsdoc; the only reliable way is to collect the names of
998           // formals after building the declared function type.
999           ArrayList<String> formals = new ArrayList<>();
1000           // tmpRestFormals is used only for error checking
1001           ArrayList<String> tmpRestFormals = new ArrayList<>();
1002           Node param = NodeUtil.getFunctionParameters(fn).getFirstChild();
1003           while (param != null) {
1004             if (JSTypeCreatorFromJSDoc.isRestArg(fnDoc, param.getString())
1005                 && param.getNext() == null) {
1006               tmpRestFormals.add(param.getString());
1007             } else {
1008               formals.add(param.getString());
1009             }
1010             param = param.getNext();
1011           }
1012           if (fnDoc != null) {
1013             for (String formalInJsdoc : fnDoc.getParameterNames()) {
1014               if (!formals.contains(formalInJsdoc)
1015                   && !tmpRestFormals.contains(formalInJsdoc)) {
1016                 String functionName = NodeUtil.getNearestFunctionName(fn);
1017                 warnings.add(JSError.make(fn, INEXISTENT_PARAM, formalInJsdoc, functionName));
1018               }
1019             }
1020           }
1021           return formals;
1022         }

1024         private void maybeRecordNominalType(
1025             Node defSite, Node nameNode, JSDocInfo fnDoc, boolean isRedeclaration) {
1026           Preconditions.checkState(nameNode == null || nameNode.isQualifiedName());
1027           if (fnDoc == null) {
1028             return;
1029           }
1030           if (fnDoc.isConstructorOrInterface()) {
1031             if (nameNode == null) {
1032               warnings.add(JSError.make(defSite, ANONYMOUS_NOMINAL_TYPE));
1033               return;
1034             }
1035             String qname = nameNode.getQualifiedName();
1036             ImmutableList.Builder<String> builder = ImmutableList.builder();
1037             for (String typeParam : fnDoc.getTemplateTypeNames()) {
1038               builder.add(varNameGen.getNextName(typeParam));
1039             }
1040             ImmutableList<String> typeParameters = builder.build();
1041             RawNominalType rawType;
1042             if (fnDoc.usesImplicitMatch()) {
1043               rawType = RawNominalType.makeStructuralInterface(defSite, qname, typeParameters);
1044             } else if (fnDoc.isInterface()) {
1045               rawType = RawNominalType.makeNominalInterface(defSite, qname, typeParameters);
1046             } else if (fnDoc.makesStructs()) {
1047               rawType = RawNominalType.makeStructClass(defSite, qname, typeParameters);
1048             } else if (fnDoc.makesDicts()) {
1049               rawType = RawNominalType.makeDictClass(defSite, qname, typeParameters);
1050             } else {
1051               rawType = RawNominalType.makeUnrestrictedClass(defSite, qname, typeParameters);
1052             }
1053             nominaltypesByNode.put(defSite, rawType);
1054             if (isRedeclaration) {
1055               return;
1056             }
1057             Node firstChild = nameNode.getFirstChild();
1058             if (nameNode.isName()
1059                 || currentScope.isNamespace(firstChild)
1060                 || mayCreateFunctionNamespace(firstChild)
1061                 || mayCreateWindowNamespace(firstChild)) {
1062               if (nameNode.isGetProp()) {
1063                 defSite.getParent().getFirstChild().putBooleanProp(Node.ANALYZED_DURING_GTI, true);
1064               } else if (currentScope.isTopLevel()) {
1065                 maybeRecordBuiltinType(qname, rawType);
1066               }
1067               currentScope.addNamespace(nameNode, rawType);
1068             }
1069           } else if (fnDoc.makesStructs()) {
1070             warnings.add(JSError.make(defSite, STRUCTDICT_WITHOUT_CTOR, "@struct"));
1071           } else if (fnDoc.makesDicts()) {
1072             warnings.add(JSError.make(defSite, STRUCTDICT_WITHOUT_CTOR, "@dict"));
1073           }
1074         }

1080               commonTypes.setArgumentsType(rawType);

1083               commonTypes.setFunctionType(rawType);
1084               break;
1085             case "Object":

1090               break;

1109         private void visitAliasedNamespace(Node lhs) {
1110           if (this.currentScope.isDefined(lhs)) {
1111             return;
1112           }
1113           Node rhs = NodeUtil.getRValueOfLValue(lhs);
1114           QualifiedName rhsQname = QualifiedName.fromNode(rhs);
1115           JSDocInfo jsdoc = NodeUtil.getBestJSDocInfo(lhs);
1116     
1117           if (jsdoc != null && jsdoc.isConstructorOrInterface()) {
1118             RawNominalType rawType = this.currentScope.getNominalType(rhsQname);
1119             if (jsdoc.isConstructor()
1120                 && (rawType == null || rawType.isInterface())) {
1121               warnings.add(JSError.make(rhs, EXPECTED_CONSTRUCTOR, rhsQname.toString()));
1122               return;
1123             }
1124             if (jsdoc.isInterface()
1125                 && (rawType == null || rawType.isClass())) {
1126               warnings.add(JSError.make(rhs, EXPECTED_INTERFACE, rhsQname.toString()));
1127               return;
1128             }
1129           }
1130           Namespace ns = this.currentScope.getNamespace(rhsQname);
1131           if (ns != null) {
1132             lhs.getParent().putBooleanProp(Node.ANALYZED_DURING_GTI, true);
1133             this.currentScope.addNamespace(lhs, ns);
1134           }
1135         }

1139           String internalName = getFunInternalName(funQname.getParent().getLastChild());

1141           QualifiedName pname = new QualifiedName(funQname.getLastChild().getString());
1142           if (!ns.isDefined(pname)) {
1143             ns.addNamespace(pname,
1144                 new FunctionNamespace(funQname.getQualifiedName(), s));

1171         /**
1172          * @lends can lend properties to an object X being defined in the same
1173          * statement as the @lends. To make sure that we've seen the definition of
1174          * X, we process @lends annotations after we've traversed the scope.
1175          *
1176          * @lends can only add properties to namespaces, constructors and prototypes
1177          */
1178         void processLendsNode(Node objlit) {
1179           JSDocInfo jsdoc = objlit.getJSDocInfo();
1180           String lendsName = jsdoc.getLendsName();
1181           Preconditions.checkNotNull(lendsName);
1182           QualifiedName lendsQname = QualifiedName.fromQualifiedString(lendsName);
1183           if (currentScope.isNamespace(lendsQname)) {
1184             processLendsToNamespace(lendsQname, lendsName, objlit);
1185           } else {
1186             RawNominalType rawType = checkValidLendsToPrototypeAndGetClass(
1187                 lendsQname, lendsName, objlit);
1188             if (rawType != null) {
1189               for (Node prop : objlit.children()) {
1190                 String pname =  NodeUtil.getObjectLitKeyName(prop);
1191                 mayAddPropToPrototype(rawType, pname, prop, prop.getFirstChild());
1192               }
1193             } else {
1194               // When rawType is null, some invalid declaration prevented the
1195               // raw type from being registered. We must still compute the
1196               // declared type of any function literals.
1197               for (Node prop : objlit.children()) {
1198                 Node propInit = prop.getFirstChild();
1199                 if (propInit.isFunction()) {
1200                   visitFunctionLate(propInit, null);
1201                 }
1202               }
1203             }
1204           }
1205         }

1207         void processLendsToNamespace(
1208             QualifiedName lendsQname, String lendsName, Node objlit) {
1209           RawNominalType rawType = currentScope.getNominalType(lendsQname);
1210           if (rawType != null && rawType.isInterface()) {
1211             warnings.add(JSError.make(objlit, LENDS_ON_BAD_TYPE, lendsName));
1212             return;
1213           }
1214           Namespace borrowerNamespace = currentScope.getNamespace(lendsQname);
1215           for (Node prop : objlit.children()) {
1216             String pname = NodeUtil.getObjectLitKeyName(prop);
1217             JSType propDeclType = declaredObjLitProps.get(prop);
1218             if (propDeclType != null) {
1219               borrowerNamespace.addProperty(pname, prop, propDeclType, false);
1220             } else {
1221               JSType t = simpleInferExprType(prop.getFirstChild());
1222               if (t == null) {
1223                 t = JSType.UNKNOWN;
1224               }
1225               borrowerNamespace.addProperty(pname, prop, t, false);
1226             }
1227           }
1228         }

1230         RawNominalType checkValidLendsToPrototypeAndGetClass(
1231             QualifiedName lendsQname, String lendsName, Node objlit) {
1232           if (!lendsQname.getRightmostName().equals("prototype")) {
1233             warnings.add(JSError.make(objlit, LENDS_ON_BAD_TYPE, lendsName));
1234             return null;
1235           }
1236           QualifiedName recv = lendsQname.getAllButRightmost();
1237           RawNominalType rawType = currentScope.getNominalType(recv);
1238           if (rawType == null || rawType.isInterface()) {
1239             warnings.add(JSError.make(objlit, LENDS_ON_BAD_TYPE, lendsName));
1240           }
1241           return rawType;
1242         }

1244         @Override
1245         public void visit(NodeTraversal t, Node n, Node parent) {
1246           switch (n.getType()) {
1247             case FUNCTION:
1248               Node grandparent = parent.getParent();
1249               // We skip property and prototype declarations because if we compute their
1250               // declared type here it will be wrong, and it will block the correct
1251               // computation later.
1252               if (grandparent == null
1253                   || (!isPrototypePropertyDeclaration(grandparent)
1254                       && !isClassPropertyDeclaration(
1255                           parent.getFirstChild(), currentScope))) {
1256                 RawNominalType ownerType = maybeGetOwnerType(n, parent);
1257                 visitFunctionLate(n, ownerType);
1258               }
1259               break;
1260             case NAME: {
1261               String name = n.getString();
1262               if (name == null || parent.isFunction()) {
1263                 return;
1264               }
1265               // TODO(dimvar): Handle local scopes introduced by catch properly,
1266               // after we decide what to do with variables in general, eg, will we
1267               // use unique numeric ids?
1268               if (parent.isVar() || parent.isCatch()) {
1269                 visitVar(n, parent);
1270               } else if (currentScope.isOuterVarEarly(name)) {
1271                 currentScope.addOuterVar(name);
1272               } else if (// Typedef variables can't be referenced in the source.
1273                   currentScope.getTypedef(name) != null
1274                   || !name.equals(currentScope.getName())
1275                   && !currentScope.isDefinedLocally(name, false)) {
1276               }
1277               break;
1278             }
1279             case GETPROP:
1280               if (parent.isExprResult() && n.isQualifiedName()) {
1281                 visitPropertyDeclaration(n);
1282               }
1283               break;
1284             case ASSIGN: {
1285               Node lvalue = n.getFirstChild();
1286               if (lvalue.isGetProp() && lvalue.isQualifiedName() && parent.isExprResult()) {
1287                 visitPropertyDeclaration(lvalue);
1288               }
1289               break;
1290             }
1291             case CAST:
1292               castTypes.put(n,
1293                   getDeclaredTypeOfNode(n.getJSDocInfo(), currentScope));
1294               break;
1295             case OBJECTLIT:
1296               visitObjectLit(n, parent);
1297               break;
1298             case CALL:
1299               visitCall(n);
1300               break;
1301           }
1302         }

1304         private void visitVar(Node nameNode, Node parent) {
1305           String name = nameNode.getString();
1306           boolean isDefinedLocally = this.currentScope.isDefinedLocally(name, false);
1307           if (isCtorDefinedByCall(nameNode)) {
1308             computeFnDeclaredType(NodeUtil.getBestJSDocInfo(nameNode), name,
1309                 nameNode.getFirstChild(), null, this.currentScope);
1310             return;
1311           }
1312           if (isDefinedLocally && this.currentScope.isNamespace(name)) {
1313             return;
1314           }
1315           if (NodeUtil.isTypedefDecl(nameNode) || NodeUtil.isEnumDecl(nameNode)) {
1316             if (!isDefinedLocally) {
1317               // Malformed enum or typedef
1318               this.currentScope.addLocal(
1319                   name, JSType.UNKNOWN, false, nameNode.isFromExterns());
1320             }
1321             return;
1322           }
1323           Node initializer = nameNode.getFirstChild();
1324           if (initializer != null && initializer.isFunction()) {
1325             return;
1326           }
1327           if (parent.isCatch()) {
1328             this.currentScope.addLocal(name, JSType.UNKNOWN, false, false);
1329           } else {
1330             boolean isConst = isConst(nameNode);
1331             JSType declType = getVarTypeFromAnnotation(nameNode, this.currentScope);
1332             if (declType == null) {
1333               declType = mayInferFromRhsIfConst(nameNode);
1334             }
1335             this.currentScope.addLocal(name, declType, isConst, nameNode.isFromExterns());
1336           }
1337         }

1339         private void visitObjectLit(Node objLitNode, Node parent) {
1340           JSDocInfo jsdoc = objLitNode.getJSDocInfo();
1341           if (jsdoc != null && jsdoc.getLendsName() != null) {
1342             lendsObjlits.add(objLitNode);
1343           }
1344           Node maybeLvalue = parent.isAssign() ? parent.getFirstChild() : parent;
1345           if (NodeUtil.isNamespaceDecl(maybeLvalue)
1346               && currentScope.isNamespace(maybeLvalue)) {
1347             for (Node prop : objLitNode.children()) {
1348               visitNamespacePropertyDeclaration(
1349                   prop, maybeLvalue, prop.getString());
1350             }
1351           } else if (!NodeUtil.isEnumDecl(maybeLvalue)
1352               && !NodeUtil.isPrototypeAssignment(maybeLvalue)) {
1353             for (Node prop : objLitNode.children()) {
1354               if (prop.getJSDocInfo() != null) {
1355                 declaredObjLitProps.put(prop,
1356                     getDeclaredTypeOfNode(
1357                         prop.getJSDocInfo(), currentScope));
1358               }
1359               if (isAnnotatedAsConst(prop)) {
1360                 warnings.add(JSError.make(prop, MISPLACED_CONST_ANNOTATION));
1361               }
1362             }
1363           }
1364         }

1366         private void visitCall(Node call) {
1367           String className = convention.getSingletonGetterClassName(call);
1368           if (className == null) {
1369             return;
1370           }
1371           QualifiedName qname = QualifiedName.fromQualifiedString(className);
1372           RawNominalType rawType = this.currentScope.getNominalType(qname);
1373           if (rawType != null) {
1374             JSType instanceType = rawType.getInstanceAsJSType();
1375             FunctionType getInstanceFunType =
1376                 (new FunctionTypeBuilder()).addRetType(instanceType).buildFunction();
1377             JSType getInstanceType = commonTypes.fromFunctionType(getInstanceFunType);
1378             convention.applySingletonGetterNew(rawType, getInstanceType, instanceType);
1379           }
1380         }

1382         private void visitPropertyDeclaration(Node getProp) {
1383           // Class property
1384           if (isClassPropertyDeclaration(getProp, currentScope)) {
1385             visitClassPropertyDeclaration(getProp);
1386           }
1387           // Prototype property
1388           else if (isPrototypeProperty(getProp)) {
1389             visitPrototypePropertyDeclaration(getProp);
1390           }
1391           // Direct assignment to the prototype
1392           else if (NodeUtil.isPrototypeAssignment(getProp)) {
1393             visitPrototypeAssignment(getProp);
1394           }
1395           // "Static" property on constructor
1396           else if (isStaticCtorProp(getProp, currentScope)) {
1397             visitConstructorPropertyDeclaration(getProp);
1398           }
1399           // Namespace property
1400           else if (currentScope.isNamespace(getProp.getFirstChild())) {
1401             visitNamespacePropertyDeclaration(getProp);
1402           }
1403           // Other property
1404           else {
1405             visitOtherPropertyDeclaration(getProp);
1406           }
1407         }

1409         private boolean isStaticCtorProp(Node getProp, NTIScope s) {
1410           Preconditions.checkArgument(getProp.isGetProp());
1411           if (!getProp.isQualifiedName()) {
1412             return false;
1413           }
1414           Node receiverObj = getProp.getFirstChild();
1415           if (!s.isLocalFunDef(receiverObj.getQualifiedName())) {
1416             return false;
1417           }
1418           return null != currentScope.getNominalType(
1419               QualifiedName.fromNode(receiverObj));
1420         }

1422         /** Compute the declared type for a given scope. */
1423         private NTIScope visitFunctionLate(Node fn, RawNominalType ownerType) {
1424           Preconditions.checkArgument(fn.isFunction());
1425           String internalName = getFunInternalName(fn);
1426           NTIScope fnScope = currentScope.getScope(internalName);
1427           JSDocInfo jsdoc = NodeUtil.getBestJSDocInfo(fn);
1428           DeclaredFunctionType declFunType = fnScope.getDeclaredFunctionType();
1429           if (declFunType == null) {
1430             declFunType = computeFnDeclaredType(
1431                 jsdoc, internalName, fn, ownerType, currentScope);
1432             fnScope.setDeclaredType(declFunType);
1433           }
1434           return fnScope;
1435         }

1437         // Used when declaring constructor and namespace properties. Prototype property
1438         // declarations are similar, but different enough that they didn't neatly fit
1439         // in this method (eg, redeclaration warnings are stricter).
1440         PropertyType getPropTypeHelper(
1441             JSDocInfo jsdoc, Node initializer, RawNominalType thisType) {
1442           PropertyType result = new PropertyType();
1443           DeclaredFunctionType dft = null;
1444           if (initializer != null && initializer.isFunction()) {
1445             dft = visitFunctionLate(initializer, thisType).getDeclaredFunctionType();
1446           }
1447           if (jsdoc != null && jsdoc.hasType()) {
1448             result.declType = getDeclaredTypeOfNode(jsdoc, currentScope);
1449           } else if (initializer != null && initializer.isFunction()) {
1450             JSType funType = commonTypes.fromFunctionType(dft.toFunctionType());
1451             if ((jsdoc != null && jsdoc.containsFunctionDeclaration())
1452                 || NodeUtil.functionHasInlineJsdocs(initializer)) {
1453               result.declType = funType;
1454             } else {
1455               result.inferredFunType = funType;
1456             }
1457           }
1458           return result;
1459         }

1461         private void visitPrototypePropertyDeclaration(Node getProp) {
1462           Preconditions.checkArgument(getProp.isGetProp());
1463           Node parent = getProp.getParent();
1464           Node initializer = parent.isAssign() ? parent.getLastChild() : null;
1465           Node ctorNameNode = NodeUtil.getPrototypeClassName(getProp);
1466           QualifiedName ctorQname = QualifiedName.fromNode(ctorNameNode);
1467           RawNominalType rawType = currentScope.getNominalType(ctorQname);
1468     
1469           if (rawType == null) {
1470             if (initializer != null && initializer.isFunction()) {
1471               visitFunctionLate(initializer, null);
1472             }
1473             // We don't look at assignments to prototypes of non-constructors.
1474             return;
1475           }
1476           if (initializer != null && initializer.isFunction()) {
1477             parent.putBooleanProp(Node.ANALYZED_DURING_GTI, true);
1478           }
1479           // We only add properties to the prototype of a class if the
1480           // property creations are in the same scope as the constructor
1481           // TODO(blickly): Rethink this
1482           if (!currentScope.isDefined(ctorNameNode)) {
1483             warnings.add(JSError.make(getProp, CTOR_IN_DIFFERENT_SCOPE));
1484             if (initializer != null && initializer.isFunction()) {
1485               visitFunctionLate(initializer, rawType);
1486             }
1487             return;
1488           }
1489           mayWarnAboutInterfacePropInit(rawType, initializer);
1490           mayAddPropToPrototype(
1491               rawType, getProp.getLastChild().getString(), getProp, initializer);
1492         }

1494         private void mayWarnAboutInterfacePropInit(RawNominalType rawType, Node initializer) {
1495           if (rawType.isInterface() && initializer != null) {
1496             String abstractMethodName = convention.getAbstractMethodName();
1497             if (initializer.isFunction()
1498                 && !NodeUtil.isEmptyFunctionExpression(initializer)) {
1499               warnings.add(JSError.make(initializer, INTERFACE_METHOD_NOT_EMPTY));
1500             } else if (!initializer.isFunction()
1501                 && !initializer.matchesQualifiedName(abstractMethodName)) {
1502               warnings.add(JSError.make(initializer, INVALID_INTERFACE_PROP_INITIALIZER));
1503             }
1504           }
1505         }

1508           Preconditions.checkArgument(getProp.isGetProp());
1509           Node ctorNameNode = NodeUtil.getPrototypeClassName(getProp);
1510           QualifiedName ctorQname = QualifiedName.fromNode(ctorNameNode);
1511           RawNominalType rawType = currentScope.getNominalType(ctorQname);
1512           if (rawType == null) {
1513             return;
1514           }
1515           getProp.putBooleanProp(Node.ANALYZED_DURING_GTI, true);
1516           for (Node objLitChild : getProp.getParent().getLastChild().children()) {
1517             mayAddPropToPrototype(rawType, objLitChild.getString(), objLitChild,
1518                 objLitChild.getLastChild());
1519           }

1522         private void visitConstructorPropertyDeclaration(Node getProp) {
1523           Preconditions.checkArgument(getProp.isGetProp());
1524           // Named types have already been crawled in CollectNamedTypes
1525           if (isNamedType(getProp)) {
1526             return;
1527           }
1528           String ctorName = getProp.getFirstChild().getQualifiedName();
1529           QualifiedName ctorQname = QualifiedName.fromNode(getProp.getFirstChild());
1530           Preconditions.checkState(currentScope.isLocalFunDef(ctorName));
1531           RawNominalType classType = currentScope.getNominalType(ctorQname);
1532           String pname = getProp.getLastChild().getString();
1533           JSDocInfo jsdoc = NodeUtil.getBestJSDocInfo(getProp);
1534           JSType propDeclType = getDeclaredTypeOfNode(jsdoc, currentScope);
1535           boolean isConst = isConst(getProp);
1536           if (propDeclType != null || isConst) {
1537             JSType previousPropType = classType.getCtorPropDeclaredType(pname);
1538             if (classType.hasCtorProp(pname)
1539                 && previousPropType != null
1540                 && !suppressDupPropWarning(jsdoc, propDeclType, previousPropType)) {
1541               warnings.add(JSError.make(
1542                   getProp, REDECLARED_PROPERTY, pname, "type " + classType));
1543               return;
1544             }
1545             if (propDeclType == null) {
1546               propDeclType = mayInferFromRhsIfConst(getProp);
1547             }
1548             classType.addCtorProperty(pname, getProp, propDeclType, isConst);
1549             getProp.putBooleanProp(Node.ANALYZED_DURING_GTI, true);
1550             if (isConst) {
1551               getProp.putBooleanProp(Node.CONSTANT_PROPERTY_DEF, true);
1552             }
1553           } else {
1554             classType.addUndeclaredCtorProperty(pname, getProp);
1555           }
1556         }

1558         private void visitNamespacePropertyDeclaration(Node getProp) {
1559           Preconditions.checkArgument(getProp.isGetProp());
1560           if (isCtorDefinedByCall(getProp)) {
1561             computeFnDeclaredType(
1562                 NodeUtil.getBestJSDocInfo(getProp),
1563                 getProp.getQualifiedName(),
1564                 getProp.getNext(), null, this.currentScope);
1565             return;
1566           }
1567           // Named types have already been crawled in CollectNamedTypes
1568           if (isNamedType(getProp)) {
1569             return;
1570           }
1571           Node recv = getProp.getFirstChild();
1572           String pname = getProp.getLastChild().getString();
1573           visitNamespacePropertyDeclaration(getProp, recv, pname);
1574         }

1576         private void visitNamespacePropertyDeclaration(
1577             Node declNode, Node recv, String pname) {
1578           Preconditions.checkArgument(declNode.isGetProp() || declNode.isStringKey());
1579           Preconditions.checkArgument(currentScope.isNamespace(recv));
1580           EnumType et = currentScope.getEnum(QualifiedName.fromNode(recv));
1581           // If there is a reassignment to one of the enum's members, don't consider
1582           // that a definition of a new property.
1583           if (et != null && et.enumLiteralHasKey(pname)) {
1584             return;
1585           }
1586     
1587           Namespace ns = currentScope.getNamespace(QualifiedName.fromNode(recv));
1588           JSDocInfo jsdoc = NodeUtil.getBestJSDocInfo(declNode);
1589           Node initializer = NodeUtil.getRValueOfLValue(declNode);
1590           PropertyType pt = getPropTypeHelper(jsdoc, initializer, null);
1591           JSType propDeclType = pt.declType;
1592           JSType propInferredFunType = pt.inferredFunType;
1593           boolean isConst = isConst(declNode);
1594           if (propDeclType != null || isConst) {
1595             JSType previousPropType = ns.getPropDeclaredType(pname);
1596             if (ns.hasSubnamespace(new QualifiedName(pname))
1597                 || ns.hasProp(pname)
1598                 && previousPropType != null
1599                 && !suppressDupPropWarning(jsdoc, propDeclType, previousPropType)) {
1600               warnings.add(JSError.make(
1601                   declNode, REDECLARED_PROPERTY, pname, "namespace " + ns));
1602               declNode.getParent().putBooleanProp(Node.ANALYZED_DURING_GTI, true);
1603               return;
1604             }
1605             if (propDeclType == null) {
1606               propDeclType = mayInferFromRhsIfConst(declNode);
1607             }
1608             ns.addProperty(pname, declNode, propDeclType, isConst);
1609             declNode.putBooleanProp(Node.ANALYZED_DURING_GTI, true);
1610             if (declNode.isGetProp() && isConst) {
1611               declNode.putBooleanProp(Node.CONSTANT_PROPERTY_DEF, true);
1612             }
1613           } else if (propInferredFunType != null) {
1614             ns.addUndeclaredProperty(pname, declNode, propInferredFunType, false);
1615           } else {
1616             // Try to infer the prop type, but don't say that the prop is declared.
1617             JSType t = initializer == null
1618                 ? null : simpleInferExprType(initializer);
1619             if (t == null) {
1620               t = JSType.UNKNOWN;
1621             }
1622             ns.addUndeclaredProperty(pname, declNode, t, false);
1623           }
1624         }

1626         private void visitClassPropertyDeclaration(Node getProp) {
1627           Preconditions.checkArgument(getProp.isGetProp());
1628           JSType t = currentScope.getDeclaredFunctionType().getThisType();
1629           NominalType thisType = t == null ? null : t.getNominalTypeIfSingletonObj();
1630           Node parent = getProp.getParent();
1631           Node initializer = parent.isAssign() ? parent.getLastChild() : null;
1632           if (thisType == null) {
1633             if (initializer != null && initializer.isFunction()) {
1634               visitFunctionLate(initializer, null);
1635             }
1636             // This will get caught in NewTypeInference
1637             return;
1638           }
1639           RawNominalType rawType = thisType.getRawNominalType();
1640           String pname = getProp.getLastChild().getString();
1641           JSDocInfo jsdoc = NodeUtil.getBestJSDocInfo(getProp);
1642           PropertyType pt = getPropTypeHelper(jsdoc, initializer, rawType);
1643           JSType propDeclType = pt.declType;
1644           JSType propInferredFunType = pt.inferredFunType;
1645           boolean isConst = isConst(getProp);
1646           if (initializer != null && initializer.isFunction()) {
1647             parent.putBooleanProp(Node.ANALYZED_DURING_GTI, true);
1648           }
1649           if (propDeclType != null || isConst) {
1650             mayWarnAboutExistingProp(rawType, pname, getProp, propDeclType);
1651             // Intentionally, we keep going even if we warned for redeclared prop.
1652             // The reason is that if a prop is defined on a class and on its proto
1653             // with conflicting types, we prefer the type of the class.
1654             if (propDeclType == null) {
1655               propDeclType = mayInferFromRhsIfConst(getProp);
1656             }
1657             if (mayAddPropToType(getProp, rawType)) {
1658               rawType.addClassProperty(pname, getProp, propDeclType, isConst);
1659             }
1660             if (isConst) {
1661               getProp.putBooleanProp(Node.CONSTANT_PROPERTY_DEF, true);
1662             }
1663           } else if (mayAddPropToType(getProp, rawType)) {
1664             if (propInferredFunType != null) {
1665               rawType.addUndeclaredClassProperty(pname, propInferredFunType, getProp);
1666             } else {
1667               rawType.addUndeclaredClassProperty(pname, JSType.UNKNOWN, getProp);
1668             }
1669     
1670           }
1671           // Only add the definition node if the property is not already defined.
1672           if (!propertyDefs.contains(rawType, pname)) {
1673             propertyDefs.put(rawType, pname,
1674                 new PropertyDef(getProp, null, null));
1675           }
1676         }

1678         private void visitOtherPropertyDeclaration(Node getProp) {
1679           Preconditions.checkArgument(getProp.isGetProp());
1680           Preconditions.checkArgument(getProp.isQualifiedName());
1681           if (isAnnotatedAsConst(getProp)) {
1682             warnings.add(JSError.make(getProp, MISPLACED_CONST_ANNOTATION));
1683           }
1684           Node recv = getProp.getFirstChild();
1685           QualifiedName recvQname = QualifiedName.fromNode(recv);
1686           Declaration d = this.currentScope.getDeclaration(recvQname, false);
1687           if (d != null && d.getTypedef() != null) {
1688             warnings.add(JSError.make(getProp, CANNOT_ADD_PROPERTIES_TO_TYPEDEF));
1689             getProp.getParent().putBooleanProp(Node.ANALYZED_DURING_GTI, true);
1690             return;
1691           }
1692           JSType recvType = simpleInferExprType(recv);
1693           if (recvType == null) {
1694             return;
1695           }
1696           recvType = recvType.removeType(JSType.NULL);

1785         private JSType simpleInferCallNewType(Node n) {
1786           Node callee = n.getFirstChild();
1787           // We special-case the function goog.getMsg, which is used by the
1788           // compiler for i18n.
1789           if (callee.matchesQualifiedName("goog.getMsg")) {
1790             return JSType.STRING;
1791           }
1792           FunctionType funType = simpleInferFunctionType(callee);
1793           if (funType == null) {
1794             return null;
1795           }
1796           if (funType.isGeneric()) {
1797             ImmutableList.Builder<JSType> argTypes = ImmutableList.builder();
1798             for (Node argNode = n.getSecondChild();
1799                  argNode != null;
1800                  argNode = argNode.getNext()) {
1801               JSType t = simpleInferExprType(argNode);
1802               if (t == null) {
1803                 return null;
1804               }
1805               argTypes.add(t);
1806             }
1807             funType = funType.instantiateGenericsFromArgumentTypes(argTypes.build());
1808             if (funType == null) {
1809               return null;
1810             }
1811           }
1812           JSType retType = n.isNew() ? funType.getThisType() : funType.getReturnType();
1813           return retType;
1814         }

1816         private JSType simpleInferExprType(Node n) {
1817           switch (n.getType()) {
1818             case REGEXP:
1819               return commonTypes.getRegexpType();
1820             case CAST:
1821               return castTypes.get(n);
1822             case ARRAYLIT: {
1823               if (!n.hasChildren()) {
1824                 return commonTypes.getArrayInstance();
1825               }
1826               Node child = n.getFirstChild();
1827               JSType arrayType = simpleInferExprType(child);
1828               if (arrayType == null) {
1829                 return null;
1830               }
1831               while (null != (child = child.getNext())) {
1832                 if (!arrayType.equals(simpleInferExprType(child))) {
1833                   return null;
1834                 }
1835               }
1836               return commonTypes.getArrayInstance(arrayType);
1837             }
1838             case TRUE:
1839               return JSType.TRUE_TYPE;
1840             case FALSE:
1841               return JSType.FALSE_TYPE;
1842             case THIS:
1843               return this.currentScope.getDeclaredTypeOf("this");
1844             case NAME:
1845               return simpleInferDeclaration(
1846                   this.currentScope.getDeclaration(n.getString(), false));
1847             case OBJECTLIT: {
1848               JSType objLitType = JSType.TOP_OBJECT;
1849               for (Node prop : n.children()) {
1850                 JSType propType = simpleInferExprType(prop.getFirstChild());
1851                 if (propType == null) {
1852                   return null;
1853                 }
1854                 objLitType = objLitType.withProperty(
1855                     new QualifiedName(NodeUtil.getObjectLitKeyName(prop)),
1856                     propType);
1857               }
1858               return objLitType;
1859             }
1860             case GETPROP:
1861               return simpleInferPropAccessType(
1862                   n.getFirstChild(), n.getLastChild().getString());
1863             case GETELEM:
1864               return simpleInferGetelemType(n);
1865             case COMMA:
1866             case ASSIGN:
1867               return simpleInferExprType(n.getLastChild());
1868             case CALL:
1869             case NEW:
1870               return simpleInferCallNewType(n);
1871             case AND:
1872             case OR:
1873               return simpleInferAndOrType(n);
1874             case HOOK: {
1875               JSType lhs = simpleInferExprType(n.getSecondChild());
1876               JSType rhs = simpleInferExprType(n.getLastChild());
1877               return lhs == null || rhs == null ? null : JSType.join(lhs, rhs);
1878             }
1879             default:
1880               switch (NodeUtil.getKnownValueType(n)) {
1881                 case NULL:
1882                   return JSType.NULL;
1883                 case VOID:
1884                   return JSType.UNDEFINED;
1885                 case NUMBER:
1886                   return JSType.NUMBER;
1887                 case STRING:
1888                   return JSType.STRING;
1889                 case BOOLEAN:
1890                   return JSType.BOOLEAN;
1891                 default:
1892                   return null;
1893               }
1894           }
1895         }

2049             JSDocInfo fnDoc, String functionName, Node declNode,
2050             RawNominalType ownerType, NTIScope parentScope) {
2051           Preconditions.checkArgument(
2052               declNode.isFunction() || declNode.isGetProp() || declNode.isCall());
2053     
2054           // For an unannotated function, check if we can grab a type signature for
2055           // it from the surrounding code where it appears.
2056           if (fnDoc == null && !NodeUtil.functionHasInlineJsdocs(declNode)) {
2057             DeclaredFunctionType t = getDeclaredFunctionTypeFromContext(
2058                 functionName, declNode, parentScope);
2059             if (t != null) {
2060               return t;
2061             }
2062           }
2063           // TODO(dimvar): warn if multiple jsdocs for a fun
2064           RawNominalType ctorType =
2065               !declNode.isGetProp() ? nominaltypesByNode.get(declNode) : null;
2066           FunctionAndSlotType result = typeParser.getFunctionType(
2067               fnDoc, functionName, declNode, ctorType, ownerType, parentScope);
2068           Node qnameNode;

2070             qnameNode = declNode;
2071           } else if (declNode.isFunction()) {
2072             qnameNode = NodeUtil.getNameNode(declNode);

2104         // We only return a non-null result if the arity of declNode matches the
2105         // arity we get from declaredTypeAsJSType.
2106         private DeclaredFunctionType computeFnDeclaredTypeFromCallee(
2107             Node declNode, JSType declaredTypeAsJSType) {
2108           Preconditions.checkArgument(declNode.isFunction());
2109           Preconditions.checkArgument(declNode.getParent().isCall());
2110           Preconditions.checkNotNull(declaredTypeAsJSType);
2111     
2112           FunctionType funType = declaredTypeAsJSType.getFunType();
2113           if (funType == null
2114               || funType.isUniqueConstructor()
2115               || funType.isInterfaceDefinition()) {
2116             return null;
2117           }
2118           DeclaredFunctionType declType = funType.toDeclaredFunctionType();
2119           if (declType == null) {
2120             return null;
2121           }
2122           int numFormals = declNode.getSecondChild().getChildCount();
2123           int reqArity = declType.getRequiredArity();
2124           int optArity = declType.getOptionalArity();
2125           boolean hasRestFormals = declType.hasRestFormals();
2126           if (reqArity == optArity && !hasRestFormals) {
2127             return numFormals == reqArity ? declType : null;
2128           }
2129           if (numFormals == optArity && !hasRestFormals
2130               || numFormals == (optArity + 1) && hasRestFormals) {

2134         }

2136         // Returns null if it can't find a suitable type in the context
2137         private DeclaredFunctionType getDeclaredFunctionTypeFromContext(
2138             String functionName, Node declNode, NTIScope parentScope) {
2139           Node parent = declNode.getParent();
2140           Node maybeBind = parent.isCall() ? parent.getFirstChild() : parent;
2141     
2142           // The function literal is used with .bind or goog.bind
2143           if (NodeUtil.isFunctionBind(maybeBind) && !NodeUtil.isGoogPartial(maybeBind)) {
2144             Node call = maybeBind.getParent();

2146             JSType recvType = simpleInferExprType(bindComponents.thisValue);
2147             if (recvType == null) {

2151             DeclaredFunctionType allButRecvType = typeParser.getFunctionType(
2152                 null, functionName, declNode, null, null, parentScope).functionType;
2153             return allButRecvType.withReceiverType(recvType);
2154           }

2156           // The function literal is an argument at a call
2157           if (parent.isCall() && declNode != parent.getFirstChild()) {
2158             DeclaredFunctionType calleeDeclType = getDeclaredFunctionTypeOfCalleeIfAny(
2159                 parent.getFirstChild(), parentScope);
2160             if (calleeDeclType != null && !calleeDeclType.isGeneric()) {
2161               int index = parent.getIndexOfChild(declNode) - 1;

2163               if (declTypeFromCallee != null) {
2164                 DeclaredFunctionType t =
2165                     computeFnDeclaredTypeFromCallee(declNode, declTypeFromCallee);
2166                 if (t != null) {

2200           }

2278           Node fn, NTIScope currentScope) {
2279         Preconditions.checkArgument(fn.getParent().isCall());
2280         if (fn.isThis() || !fn.isFunction() && !fn.isQualifiedName()) {
2281           return null;
2282         }
2283         if (fn.isFunction()) {
2284           return currentScope.getScope(getFunInternalName(fn)).getDeclaredFunctionType();

2287         Declaration decl = currentScope.getDeclaration(QualifiedName.fromNode(fn), false);

2289           return null;

2291         if (decl.getFunctionScope() != null) {
2292           return decl.getFunctionScope().getDeclaredFunctionType();
2293         }
2294         if (decl.getTypeOfSimpleDecl() != null) {
2295           FunctionType funType = decl.getTypeOfSimpleDecl().getFunType();
