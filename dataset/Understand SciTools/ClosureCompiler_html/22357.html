16     package com.google.javascript.jscomp;
17     
18     import com.google.common.base.Strings;
19     import com.google.javascript.jscomp.parsing.parser.util.format.SimpleFormat;
20     
21     /**
22      * Verbose message formatter. This formatter generates very loud and long
23      * messages with multi-line source excerpts.
24      *
25      */
26     class VerboseMessageFormatter extends AbstractMessageFormatter {
27       VerboseMessageFormatter(SourceExcerptProvider source) {
28         super(source);
29       }
30     
31       @Override
32       public String formatError(JSError error) {
33         return getLevelName(CheckLevel.ERROR) + ": " + format(error);
34       }
35     
36       @Override
37       public String formatWarning(JSError warning) {
38         return getLevelName(CheckLevel.WARNING) + ": " + format(warning);
39       }
40     
41       private String format(JSError message) {
42         String description = message.description;
43         String sourceName = message.sourceName;
44         int lineNumber = message.lineNumber;
45         Region sourceRegion = getSource().getSourceRegion(sourceName, lineNumber);
46         String lineSource = null;
47         if (sourceRegion != null) {
48           lineSource = sourceRegion.getSourceExcerpt();
49         }
50         return SimpleFormat.format("%s at %s line %s %s", description,
51             (Strings.isNullOrEmpty(sourceName) ? "(unknown source)" : sourceName),
52             ((lineNumber < 0) ? String.valueOf(lineNumber) : "(unknown line)"),
53             ((lineSource != null) ? ":\n\n" + lineSource : "."));
54       }
55     }
