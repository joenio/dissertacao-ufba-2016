1374       public void testInvertibleOperators() {
1375         Map<String, String> inverses = ImmutableMap.<String, String>builder()
1376             .put("==", "!=")
1377             .put("===", "!==")
1378             .put("<=", ">")
1379             .put("<", ">=")
1380             .put(">=", "<")
1381             .put(">", "<=")
1382             .put("!=", "==")
1383             .put("!==", "===")
1384             .build();
1385         Set<String> comparators = ImmutableSet.of("<=", "<", ">=", ">");
1386         Set<String> equalitors = ImmutableSet.of("==", "===");
1387         Set<String> uncomparables = ImmutableSet.of("undefined", "void 0");
1388         List<String> operators = ImmutableList.copyOf(inverses.values());
1389         for (int iOperandA = 0; iOperandA < LITERAL_OPERANDS.size(); iOperandA++) {
1390           for (int iOperandB = 0;
1391                iOperandB < LITERAL_OPERANDS.size();
1392                iOperandB++) {
1393             for (int iOp = 0; iOp < operators.size(); iOp++) {
1394               String a = LITERAL_OPERANDS.get(iOperandA);
1395               String b = LITERAL_OPERANDS.get(iOperandB);
1396               String op = operators.get(iOp);
1397               String inverse = inverses.get(op);
1398     
1399               // Test invertability.
1400               if (comparators.contains(op)) {
1401                   if (uncomparables.contains(a) || uncomparables.contains(b)
1402                       || (a.equals("null") && NodeUtil.getStringNumberValue(b) == null)) {
1403                     assertSameResults(join(a, op, b), "false");
1404                     assertSameResults(join(a, inverse, b), "false");
1405                   }
1406               } else if (a.equals(b) && equalitors.contains(op)) {
1407                 if (a.equals("NaN") ||
1408                     a.equals("Infinity") ||
1409                     a.equals("-Infinity")) {
1410                   fold(join(a, op, b), a.equals("NaN") ? "false" : "true");
1411                 } else {
1412                   assertSameResults(join(a, op, b), "true");
1413                   assertSameResults(join(a, inverse, b), "false");
1414                 }
1415               } else {
1416                 assertNotSameResults(join(a, op, b), join(a, inverse, b));
1417               }
1418             }
1419           }
1420         }
1421       }
