48       /**
49        * @param n The last statement of a block to check for an always throws
50        *     function call. Used by CheckMissingReturn.
51        * @param alwaysThrowsFunctionName The name of a function that always throws.
52        * @return {@code true} if n is call to alwaysThrowsFunctionName, otherwise
53        *     {@code false}.
54        */
55       public static boolean defaultIsFunctionCallThatAlwaysThrows(
56           Node n, String alwaysThrowsFunctionName) {
57         if (n.isExprResult()) {
58           if (!n.getFirstChild().isCall()) {
59             return false;
60           }
61         } else if (!n.isCall()) {
62           return false;
63         }
64         if (n.isExprResult()) {
65           n = n.getFirstChild();
66         }
67         // n is a call
68         return n.getFirstChild().matchesQualifiedName(alwaysThrowsFunctionName);
69       }

554         @Override
555         public Bind describeFunctionBind(
556             Node n, boolean callerChecksTypes, boolean iCheckTypes) {
557           if (!n.isCall()) {
558             return null;
559           }
560     
561           Node callTarget = n.getFirstChild();
562           if (callTarget.isQualifiedName()) {
563             if (callTarget.matchesQualifiedName("Function.prototype.bind.call")) {
564               // goog.bind(fn, self, args...);
565               Node fn = callTarget.getNext();
566               if (fn == null) {
567                 return null;
568               }
569               Node thisValue = safeNext(fn);
570               Node parameters = safeNext(thisValue);
571               return new Bind(fn, thisValue, parameters);
572             }
573           }
574     
575           if (callTarget.isGetProp()
576               && callTarget.getLastChild().getString().equals("bind")) {
577             Node maybeFn = callTarget.getFirstChild();
578             com.google.javascript.rhino.jstype.JSType maybeFnType =
579                 maybeFn.getJSType();
580             FunctionType fnType = null;
581             if (iCheckTypes && maybeFnType != null) {
582               fnType = maybeFnType.restrictByNotNullOrUndefined()
583                   .toMaybeFunctionType();
584             }
585     
586             if (fnType != null || callerChecksTypes || maybeFn.isFunction()) {
587               // (function(){}).bind(self, args...);
588               Node thisValue = callTarget.getNext();
589               Node parameters = safeNext(thisValue);
590               return new Bind(maybeFn, thisValue, parameters);
591             }
592           }
593     
594           return null;
595         }
