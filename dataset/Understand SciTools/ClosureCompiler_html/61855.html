32     /**
33      * A code generator that outputs type annotations for functions and
34      * constructors.
35      */
36     class TypedCodeGenerator extends CodeGenerator {
37       private final TypeIRegistry registry;
38     
39       TypedCodeGenerator(
40           CodeConsumer consumer, CompilerOptions options, TypeIRegistry registry) {
41         super(consumer, options);
42         Preconditions.checkNotNull(registry);
43         this.registry = registry;
44       }
45     
46       @Override
47       protected void add(Node n, Context context) {
48         Node parent = n.getParent();
49         if (parent != null
50             && (parent.isBlock()
51                 || parent.isScript())) {
52           if (n.isFunction()) {
53             add(getFunctionAnnotation(n));
54           } else if (n.isExprResult()
55               && n.getFirstChild().isAssign()) {
56             Node assign = n.getFirstChild();
57             if (NodeUtil.isNamespaceDecl(assign.getFirstChild())) {
58               add(JSDocInfoPrinter.print(assign.getJSDocInfo()));
59             } else {
60               Node rhs = assign.getLastChild();
61               add(getTypeAnnotation(rhs));
62             }
63           } else if (n.isVar()
64               && n.getFirstFirstChild() != null) {
65             if (NodeUtil.isNamespaceDecl(n.getFirstChild())) {
66               add(JSDocInfoPrinter.print(n.getJSDocInfo()));
67             } else {
68               add(getTypeAnnotation(n.getFirstFirstChild()));
69             }
70           }
71         }
72     
73         super.add(n, context);
74       }
75     
76       private String getTypeAnnotation(Node node) {
77         // Only add annotations for things with JSDoc, or function literals.
78         JSDocInfo jsdoc = NodeUtil.getBestJSDocInfo(node);
79         if (jsdoc == null && !node.isFunction()) {
80           return "";
81         }
82     
83         JSType type = node.getJSType();
84         if (type == null) {
85           return "";
86         } else if (type.isFunctionType()) {
87           return getFunctionAnnotation(node);
88         } else if (type.isEnumType()) {
89           return "/** @enum {" +
90               type.toMaybeEnumType().getElementsType().toAnnotationString() +
91               "} */\n";
92         } else if (!type.isUnknownType()
93             && !type.isEmptyType()
94             && !type.isVoidType()
95             && !type.isFunctionPrototypeType()) {
96           return "/** @type {" + node.getJSType().toAnnotationString() + "} */\n";
97         } else {
98           return "";
99         }
100       }

102       /**
103        * @param fnNode A node for a function for which to generate a type annotation
104        */
105       private String getFunctionAnnotation(Node fnNode) {
106         JSType type = fnNode.getJSType();
107         Preconditions.checkState(fnNode.isFunction() || type.isFunctionType());
108     
109         if (type == null || type.isUnknownType()) {
110           return "";
111         }
112     
113         FunctionType funType = type.toMaybeFunctionType();
114     
115         if (JSType.isEquivalent(
116             type, (JSType) registry.getNativeType(JSTypeNative.FUNCTION_INSTANCE_TYPE))) {
117           return "/** @type {!Function} */\n";
118         }
119     
120         StringBuilder sb = new StringBuilder("/**\n");
121     
122     
123         Node paramNode = null;
124         // We need to use the child nodes of the function as the nodes for the
125         // parameters of the function type do not have the real parameter names.
126         // FUNCTION
127         //   NAME
128         //   LP
129         //     NAME param1
130         //     NAME param2
131         if (fnNode != null && fnNode.isFunction()) {
132           paramNode = NodeUtil.getFunctionParameters(fnNode).getFirstChild();
133         }
134     
135         // Param types
136         int i = 0;
137         for (Node n : funType.getParameters()) {
138           sb.append(" * ");
139           appendAnnotation(sb, "param", getParameterNodeJSDocType(n));
140           sb.append(" ")
141               .append(paramNode == null ? "p" + i : paramNode.getString())
142               .append("\n");
143           if (paramNode != null) {
144             paramNode = paramNode.getNext();
145           } else {
146             i++;
147           }
148         }
149     
150         // Return type
151         JSType retType = funType.getReturnType();
152         if (retType != null &&
153             !retType.isEmptyType() && // There is no annotation for the empty type.
154             !funType.isInterface() && // Interfaces never return a value.
155             !(funType.isConstructor() && retType.isVoidType())) {
156           sb.append(" * ");
157           appendAnnotation(sb, "return", retType.toNonNullAnnotationString());
158           sb.append("\n");
159         }
160     
161         // Constructor/interface
162         if (funType.isConstructor() || funType.isInterface()) {
163     
164           FunctionType superConstructor = funType.getSuperClassConstructor();
165     
166           if (superConstructor != null) {
167             ObjectType superInstance =
168               funType.getSuperClassConstructor().getInstanceType();
169             if (!superInstance.toString().equals("Object")) {
170               sb.append(" * ");
171               appendAnnotation(sb, "extends", superInstance.toAnnotationString());
172               sb.append("\n");
173             }
174           }
175     
176           if (funType.isInterface()) {
177             for (ObjectType interfaceType : funType.getExtendedInterfaces()) {
178               sb.append(" * ");
179               appendAnnotation(sb, "extends", interfaceType.toAnnotationString());
180               sb.append("\n");
181             }
182           }
183     
184           // Avoid duplicates, add implemented type to a set first
185           Set<String> interfaces = new TreeSet<>();
186           for (ObjectType interfaze : funType.getImplementedInterfaces()) {
187             interfaces.add(interfaze.toAnnotationString());
188           }
189           for (String interfaze : interfaces) {
190             sb.append(" * ");
191             appendAnnotation(sb, "implements", interfaze);
192             sb.append("\n");
193           }
194     
195           if (funType.isConstructor()) {
196             sb.append(" * @constructor\n");
197           } else if (funType.isStructuralInterface()) {
198             sb.append(" * @record\n");
199           } else if (funType.isInterface()) {
200             sb.append(" * @interface\n");
201           }
202         }
203     
204         if (!funType.getTemplateTypeMap().getTemplateKeys().isEmpty()) {
205           sb.append(" * @template ");
206           Joiner.on(",").appendTo(sb, funType.getTemplateTypeMap().getTemplateKeys());
207           sb.append("\n");
208         }
209     
210         sb.append(" */\n");
211         return sb.toString();
212       }
213     
214       private static void appendAnnotation(StringBuilder sb, String name, String type) {
215         sb.append("@").append(name).append(" {").append(type).append("}");
216       }
217     
218       /**
219        * Creates a JSDoc-suitable String representation the type of a parameter.
220        *
221        * @param parameterNode The parameter node.
222        */
223       private String getParameterNodeJSDocType(Node parameterNode) {
224         JSType parameterType = parameterNode.getJSType();
225         String typeString;
226     
227         if (parameterNode.isOptionalArg()) {
228           typeString = restrictByUndefined(parameterType).toNonNullAnnotationString() +
229               "=";
230         } else if (parameterNode.isVarArgs()) {
231           typeString = "..." +
232               restrictByUndefined(parameterType).toNonNullAnnotationString();
233         } else {
234           typeString = parameterType.toNonNullAnnotationString();
235         }
236     
237         return typeString;
238       }
239     
240       private JSType restrictByUndefined(JSType type) {
241         if (type.isUnionType()) {
242           return type.toMaybeUnionType().getRestrictedUnion(
243               (JSType) registry.getNativeType(JSTypeNative.VOID_TYPE));
244         }
245         return type;
246       }
