29     /**
30      * A pass for stripping a list of provided JavaScript object types.
31      *
32      * The stripping strategy is as follows:
33      *   - Provide: 1) a list of types that should be stripped, and 2) a list of
34      *     suffixes of field/variable names that should be stripped.
35      *   - Remove declarations of variables that are initialized using static
36      *     methods of strip types (e.g. var x = goog.debug.Logger.getLogger(...);).
37      *   - Remove all references to variables that are stripped.
38      *   - Remove all object literal keys with strip names.
39      *   - Remove all assignments to 1) field names that are strip names and
40      *     2) qualified names that begin with strip types.
41      *   - Remove all statements containing calls to static methods of strip types.
42      *
43      */
44     class StripCode implements CompilerPass {
45     
46       // TODO(user): Try eliminating the need for a list of strip names by instead
47       // recording which field names are assigned to debug types in each JS input.
48       private final AbstractCompiler compiler;
49       private final Set<String> stripTypes;
50       private final Set<String> stripNameSuffixes;
51       private final Set<String> stripTypePrefixes;
52       private final Set<String> stripNamePrefixes;
53       private final Set<Var> varsToRemove;
54     
55       static final DiagnosticType STRIP_TYPE_INHERIT_ERROR = DiagnosticType.error(
56           "JSC_STRIP_TYPE_INHERIT_ERROR",
57           "Non-strip type {0} cannot inherit from strip type {1}");
58     
59       static final DiagnosticType STRIP_ASSIGNMENT_ERROR = DiagnosticType.error(
60           "JSC_STRIP_ASSIGNMENT_ERROR",
61           "Unable to strip assignment to {0}");

63       /**
64        * Creates an instance.
65        *
66        * @param compiler The compiler
67        */
68       StripCode(AbstractCompiler compiler,
69                 Set<String> stripTypes,
70                 Set<String> stripNameSuffixes,
71                 Set<String> stripTypePrefixes,
72                 Set<String> stripNamePrefixes) {
73     
74         this.compiler = compiler;
75         this.stripTypes = new HashSet<>(stripTypes);
76         this.stripNameSuffixes = new HashSet<>(stripNameSuffixes);
77         this.stripTypePrefixes = new HashSet<>(stripTypePrefixes);
78         this.stripNamePrefixes = new HashSet<>(stripNamePrefixes);
79         this.varsToRemove = new HashSet<>();
80       }

82       /**
83        * Enables stripping of goog.tweak functions.
84        */
85       public void enableTweakStripping() {
86         stripTypes.add("goog.tweak");
87       }

89       @Override
90       public void process(Node externs, Node root) {
91         // Always strip types that defined on a type that is being stripped, otherwise the
92         // resulting code will be invalid, so add "prefix" stripping that isn't a partial name.
93         // TODO(johnlenz): I'm not sure what the original intent of "type prefix" stripping was.
94         // Verify that we can always assume a complete namespace and simplify this logic.
95         for (String type : stripTypes) {
96           stripTypePrefixes.add(type + ".");
97         }
98     
99         NodeTraversal.traverseEs6(compiler, root, new Strip());
100       }

104       /**
105        * A callback that strips debug code from a JavaScript parse tree.
106        */
107       private class Strip extends AbstractPostOrderCallback {
108     
109         @Override
110         public void visit(NodeTraversal t, Node n, Node parent) {
111           switch (n.getType()) {
112             case VAR:
113               removeVarDeclarationsByNameOrRvalue(t, n, parent);
114               break;
115     
116             case NAME:
117               maybeRemoveReferenceToRemovedVariable(t, n, parent);
118               break;
119     
120             case ASSIGN:
121             case ASSIGN_BITOR:
122             case ASSIGN_BITXOR:
123             case ASSIGN_BITAND:
124             case ASSIGN_LSH:
125             case ASSIGN_RSH:
126             case ASSIGN_URSH:
127             case ASSIGN_ADD:
128             case ASSIGN_SUB:
129             case ASSIGN_MUL:
130             case ASSIGN_DIV:
131             case ASSIGN_MOD:
132               maybeEliminateAssignmentByLvalueName(t, n, parent);
133               break;
134     
135             case CALL:
136             case NEW:
137               maybeRemoveCall(t, n, parent);
138               break;
139     
140             case OBJECTLIT:
141               eliminateKeysWithStripNamesFromObjLit(t, n);
142               break;
143     
144             case EXPR_RESULT:
145               maybeEliminateExpressionByName(t, n, parent);
146               break;
147           }
148         }
149     
150         /**
151          * Removes declarations of any variables whose names are strip names or
152          * whose whose r-values are static method calls on strip types. Builds a set
153          * of removed variables so that all references to them can be removed.
154          *
155          * @param t The traversal
156          * @param n A VAR node
157          * @param parent {@code n}'s parent
158          */
159         void removeVarDeclarationsByNameOrRvalue(NodeTraversal t, Node n,
160             Node parent) {
161           for (Node nameNode = n.getFirstChild(); nameNode != null;
162               nameNode = nameNode.getNext()) {
163             String name = nameNode.getString();
164             if (isStripName(name) ||
165                 isCallWhoseReturnValueShouldBeStripped(nameNode.getFirstChild())) {
166               // Remove the NAME.
167               Scope scope = t.getScope();
168               varsToRemove.add(scope.getVar(name));
169               n.removeChild(nameNode);
170               compiler.reportCodeChange();
171             }
172           }
173           if (!n.hasChildren()) {
174             // Must also remove the VAR.
175             replaceWithEmpty(n, parent);
176             compiler.reportCodeChange();
177           }
178         }
179     
180         /**
181          * Removes a reference if it is a reference to a removed variable.
182          *
183          * @param t The traversal
184          * @param n A NAME node
185          * @param parent {@code n}'s parent
186          */
187         void maybeRemoveReferenceToRemovedVariable(NodeTraversal t, Node n,
188                                                    Node parent) {
189           switch (parent.getType()) {
190             case VAR:
191               // This is a variable declaration, not a reference.
192               break;
193     
194             case GETPROP:
195               // GETPROP
196               //   NAME
197               //   STRING (property name)
198             case GETELEM:
199               // GETELEM
200               //   NAME
201               //   NUMBER|STRING|NAME|...
202               if (parent.getFirstChild() == n && isReferenceToRemovedVar(t, n)) {
203                 replaceHighestNestedCallWithNull(parent, parent.getParent());
204               }
205               break;
206     
207             case ASSIGN:
208             case ASSIGN_BITOR:
209             case ASSIGN_BITXOR:
210             case ASSIGN_BITAND:
211             case ASSIGN_LSH:
212             case ASSIGN_RSH:
213             case ASSIGN_URSH:
214             case ASSIGN_ADD:
215             case ASSIGN_SUB:
216             case ASSIGN_MUL:
217             case ASSIGN_DIV:
218             case ASSIGN_MOD:
219               if (isReferenceToRemovedVar(t, n)) {
220                 if (parent.getFirstChild() == n) {
221                   Node grandparent = parent.getParent();
222                   if (grandparent.isExprResult()) {
223                     // Remove the assignment.
224                     Node greatGrandparent = grandparent.getParent();
225                     replaceWithEmpty(grandparent, greatGrandparent);
226                     compiler.reportCodeChange();
227                   } else {
228                     // Substitute the r-value for the assignment.
229                     Node rvalue = n.getNext();
230                     parent.removeChild(rvalue);
231                     grandparent.replaceChild(parent, rvalue);
232                     compiler.reportCodeChange();
233                   }
234                 } else {
235                   // The var reference is the r-value. Replace it with null.
236                   replaceWithNull(n, parent);
237                   compiler.reportCodeChange();
238                 }
239               }
240               break;
241     
242             default:
243               if (isReferenceToRemovedVar(t, n)) {
244                 replaceWithNull(n, parent);
245                 compiler.reportCodeChange();
246               }
247               break;
248           }
249         }
250     
251         /**
252          * Use a while loop to get up out of any nested calls. For example,
253          * if we have just detected that we need to remove the a.b() call
254          * in a.b().c().d(), we'll have to remove all of the calls, and it
255          * will take a few iterations through this loop to get up to d().
256          */
257         void replaceHighestNestedCallWithNull(Node node, Node parent) {
258           Node ancestor = parent;
259           Node ancestorChild = node;
260           while (true) {
261             if (ancestor.getFirstChild() != ancestorChild) {
262               replaceWithNull(ancestorChild, ancestor);
263               break;
264             }
265             if (ancestor.isExprResult()) {
266               // Remove the entire expression statement.
267               Node ancParent = ancestor.getParent();
268               replaceWithEmpty(ancestor, ancParent);
269               break;
270             }
271             if (ancestor.isAssign()) {
272               Node ancParent = ancestor.getParent();
273               ancParent.replaceChild(
274                   ancestor, ancestor.getLastChild().detachFromParent());
275               break;
276             }
277             if (!NodeUtil.isGet(ancestor)
278                 && !ancestor.isCall()) {
279               replaceWithNull(ancestorChild, ancestor);
280               break;
281             }
282             ancestorChild = ancestor;
283             ancestor = ancestor.getParent();
284           }
285           compiler.reportCodeChange();
286         }

288         /**
289          * Eliminates an assignment if the l-value is:
290          *  - A field name that's a strip name
291          *  - A qualified name that begins with a strip type
292          *
293          * @param t The traversal
294          * @param n An ASSIGN node
295          * @param parent {@code n}'s parent
296          */
297         void maybeEliminateAssignmentByLvalueName(NodeTraversal t, Node n,
298                                                   Node parent) {
299           // ASSIGN
300           //   l-value
301           //   r-value
302           Node lvalue = n.getFirstChild();
303           if (nameIncludesFieldNameToStrip(lvalue) ||
304               qualifiedNameBeginsWithStripType(lvalue)) {
305     
306             // Limit to EXPR_RESULT because it is not
307             // safe to eliminate assignment in complex expressions,
308             // e.g. in ((x = 7) + 8)
309             if (parent.isExprResult()) {
310               Node grandparent = parent.getParent();
311               replaceWithEmpty(parent, grandparent);
312               compiler.reportCodeChange();
313             } else {
314               t.report(n, STRIP_ASSIGNMENT_ERROR, lvalue.getQualifiedName());
315             }
316           }
317         }
318     
319         /**
320          * Eliminates an expression if it refers to:
321          *  - A field name that's a strip name
322          *  - A qualified name that begins with a strip type
323          * This gets rid of construct like:
324          *  a.prototype.logger; (used instead of a.prototype.logger = null;)
325          * This expression is not an assignment and so will not be caught by
326          * maybeEliminateAssignmentByLvalueName.
327          * @param t The traversal
328          * @param n An EXPR_RESULT node
329          * @param parent {@code n}'s parent
330          */
331         void maybeEliminateExpressionByName(NodeTraversal t, Node n,
332                                             Node parent) {
333           // EXPR_RESULT
334           //   expression
335           Node expression = n.getFirstChild();
336           if (nameIncludesFieldNameToStrip(expression) ||
337               qualifiedNameBeginsWithStripType(expression)) {
338             if (parent.isExprResult()) {
339               Node grandparent = parent.getParent();
340               replaceWithEmpty(parent, grandparent);
341             } else {
342               replaceWithEmpty(n, parent);
343             }
344             compiler.reportCodeChange();
345           }
346         }

348         /**
349          * Removes a method call if {@link #isMethodOrCtorCallThatTriggersRemoval}
350          * indicates that it should be removed.
351          *
352          * @param t The traversal
353          * @param n A CALL node
354          * @param parent {@code n}'s parent
355          */
356         void maybeRemoveCall(NodeTraversal t, Node n, Node parent) {
357           // CALL/NEW
358           //   function
359           //   arguments
360           if (isMethodOrCtorCallThatTriggersRemoval(t, n, parent)) {
361             replaceHighestNestedCallWithNull(n, parent);
362           }
363         }

365         /**
366          * Eliminates any object literal keys in an object literal declaration that
367          * have strip names.
368          *
369          * @param t The traversal
370          * @param n An OBJLIT node
371          */
372         void eliminateKeysWithStripNamesFromObjLit(NodeTraversal t, Node n) {
373           // OBJLIT
374           //   key1
375           //     value1
376           //   key2
377           //   ...
378           Node key = n.getFirstChild();
379           while (key != null) {
380             if (isStripName(key.getString())) {
381               Node next = key.getNext();
382               n.removeChild(key);
383               key = next;
384               compiler.reportCodeChange();
385             } else {
386               key = key.getNext();
387             }
388           }
389         }

391         /**
392          * Gets whether a node is a CALL node whose return value should be
393          * stripped. A call's return value should be stripped if the function
394          * getting called is a static method in a class that gets stripped. For
395          * example, if "goog.debug.Logger" is a strip name, then this function
396          * returns true for a call such as "goog.debug.Logger.getLogger(...)".  It
397          * may also simply be a function that is getting stripped.  For example,
398          * if "getLogger" is a strip name, but not "goog.debug.Logger", this will
399          * still return true.
400          *
401          * @param n A node (typically a CALL node)
402          * @return Whether the call's return value should be stripped
403          */
404         boolean isCallWhoseReturnValueShouldBeStripped(@Nullable Node n) {
405           return n != null &&
406               (n.isCall() ||
407                n.isNew()) &&
408               n.hasChildren() &&
409               (qualifiedNameBeginsWithStripType(n.getFirstChild()) ||
410                   nameIncludesFieldNameToStrip(n.getFirstChild()));
411         }

413         /**
414          * Gets whether a qualified name begins with a strip name. The names
415          * "goog.debug", "goog.debug.Logger", and "goog.debug.Logger.Level" are
416          * examples of strip names that would result in this function returning
417          * true for a node representing the name "goog.debug.Logger.Level".
418          *
419          * @param n A node (typically a NAME or GETPROP node)
420          * @return Whether the name begins with a strip name
421          */
422         boolean qualifiedNameBeginsWithStripType(Node n) {
423           String name = n.getQualifiedName();
424           return qualifiedNameBeginsWithStripType(name);
425         }
426     
427         /**
428          * Gets whether a qualified name begins with a strip name. The names
429          * "goog.debug", "goog.debug.Logger", and "goog.debug.Logger.Level" are
430          * examples of strip names that would result in this function returning
431          * true for a node representing the name "goog.debug.Logger.Level".
432          *
433          * @param name A qualified class name
434          * @return Whether the name begins with a strip name
435          */
436         boolean qualifiedNameBeginsWithStripType(String name) {
437           if (name != null) {
438             for (String type : stripTypes) {
439               if (name.equals(type)) {
440                 return true;
441               }
442             }
443             for (String type : stripTypePrefixes) {
444               if (name.startsWith(type)) {
445                 return true;
446               }
447             }
448           }
449           return false;
450         }

452         /**
453          * Determines whether a NAME node represents a reference to a variable that
454          * has been removed.
455          *
456          * @param t The traversal
457          * @param n A NAME node
458          * @return Whether the variable was removed
459          */
460         boolean isReferenceToRemovedVar(NodeTraversal t, Node n) {
461           String name = n.getString();
462           Scope scope = t.getScope();
463           Var var = scope.getVar(name);
464           return varsToRemove.contains(var);
465         }

467         /**
468          * Gets whether a CALL node triggers statement removal, based on the name
469          * of the object whose method is being called, or the name of the method.
470          * Checks whether the name begins with a strip type, includes a field name
471          * that's a strip name, or belongs to the set of global class-defining
472          * functions (e.g. goog.inherits).
473          *
474          * @param t The traversal
475          * @param n A CALL node
476          * @return Whether the node triggers statement removal
477          */
478         boolean isMethodOrCtorCallThatTriggersRemoval(
479             NodeTraversal t, Node n, Node parent) {
480           // CALL/NEW
481           //   GETPROP (function)         <-- we're interested in this, the function
482           //     GETPROP (callee object)  <-- or the object on which it is called
483           //       ...
484           //       STRING (field name)
485           //     STRING (method name)
486           //   ... (arguments)
487     
488           Node function = n.getFirstChild();
489           if (function == null || !function.isGetProp()) {
490             // We are only interested in calls on object references that are
491             // properties. We don't need to eliminate method calls on variables
492             // that are getting removed, since that's already done by the code
493             // that removes all references to those variables.
494             return false;
495           }
496     
497           if (parent != null && parent.isName()) {
498             Node grandparent = parent.getParent();
499             if (grandparent != null && grandparent.isVar()) {
500               // The call's return value is being used to initialize a newly
501               // declared variable. We should leave the call intact for now.
502               // That way, when the traversal reaches the variable declaration,
503               // we'll recognize that the variable and all references to it need
504               // to be eliminated.
505               return false;
506             }
507           }
508     
509           Node callee = function.getFirstChild();
510           return nameIncludesFieldNameToStrip(callee) ||
511               nameIncludesFieldNameToStrip(function) ||
512               qualifiedNameBeginsWithStripType(function) ||
513               actsOnStripType(t, n);
514         }
515     
516         /**
517          * @return Whether a name includes a field name that should be stripped.
518          * E.g., "foo.stripMe.bar", "(foo.bar).stripMe", etc.
519          */
520         boolean nameIncludesFieldNameToStrip(@Nullable Node n) {
521           if (n != null && n.isGetProp()) {
522             Node propNode = n.getLastChild();
523             return isStripName(propNode.getString())
524                 || nameIncludesFieldNameToStrip(n.getFirstChild());
525           }
526           return false;
527         }

529         /**
530          * Determines whether the given node helps to define a
531          * strip type. For example, goog.inherits(stripType, Object)
532          * would be such a call.
533          *
534          * Also reports an error if a non-strip type inherits from a strip type.
535          *
536          * @param t The current traversal
537          * @param callNode The CALL node
538          */
539         private boolean actsOnStripType(NodeTraversal t, Node callNode) {
540           SubclassRelationship classes =
541               compiler.getCodingConvention().getClassesDefinedByCall(callNode);
542           if (classes != null) {
543             // It's okay to strip a type that inherits from a non-stripped type
544             // e.g. goog.inherits(goog.debug.Logger, Object)
545             if (qualifiedNameBeginsWithStripType(classes.subclassName)) {
546               return true;
547             }
548     
549             // report an error if a non-strip type inherits from a
550             // strip type.
551             if (qualifiedNameBeginsWithStripType(classes.superclassName)) {
552               t.report(callNode, STRIP_TYPE_INHERIT_ERROR,
553                        classes.subclassName, classes.superclassName);
554             }
555           }
556     
557           return false;
558         }
559     
560         /**
561          * Gets whether a JavaScript identifier is the name of a variable or
562          * property that should be stripped.
563          *
564          * @param name A JavaScript identifier
565          * @return Whether {@code name} is a name that triggers removal
566          */
567         boolean isStripName(String name) {
568           if (stripNameSuffixes.contains(name) ||
569               stripNamePrefixes.contains(name)) {
570             return true;
571           }
572     
573           if (name.isEmpty() || Character.isUpperCase(name.charAt(0))) {
574             return false;
575           }
576     
577           String lcName = name.toLowerCase();
578           for (String stripName : stripNamePrefixes) {
579             if (lcName.startsWith(stripName.toLowerCase())) {
580               return true;
581             }
582           }
583     
584           for (String stripName : stripNameSuffixes) {
585             if (lcName.endsWith(stripName.toLowerCase())) {
586               return true;
587             }
588           }
589     
590           return false;
591         }

593         /**
594          * Replaces a node with a NULL node. This is useful where a value is
595          * expected.
596          *
597          * @param n A node
598          * @param parent {@code n}'s parent
599          */
600         void replaceWithNull(Node n, Node parent) {
601           parent.replaceChild(n, IR.nullNode());
602         }
603     
604         /**
605          * Replaces a node with an EMPTY node. This is useful where a statement is
606          * expected.
607          *
608          * @param n A node
609          * @param parent {@code n}'s parent
610          */
611         void replaceWithEmpty(Node n, Node parent) {
612           NodeUtil.removeChild(parent, n);
613         }
