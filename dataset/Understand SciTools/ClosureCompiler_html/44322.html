16     package com.google.javascript.jscomp.testing;
17     
18     import static com.google.common.truth.Truth.THROW_ASSERTION_ERROR;
19     import static org.junit.Assert.assertEquals;
20     
21     import com.google.common.truth.FailureStrategy;
22     import com.google.common.truth.Subject;
23     import com.google.javascript.rhino.Node;
24     import com.google.javascript.rhino.Token;
25     
26     /**
27      * A Truth Subject for the Node class. Usage:
28      * <pre>
29      *   import static com.google.javascript.jscomp.testing.NodeSubject.assertNode;
30      *   ...
31      *   assertNode(node1).isEqualTo(node2);
32      *   assertNode(node1).hasType(Token.FUNCTION);
33      * </pre>
34      */
35     public final class NodeSubject extends Subject<NodeSubject, Node> {
36       public static NodeSubject assertNode(Node node) {
37         return new NodeSubject(THROW_ASSERTION_ERROR, node);
38       }
39     
40       public NodeSubject(FailureStrategy fs, Node node) {
41         super(fs, node);
42       }
43     
44       public void isEqualTo(Node node) {
45         String treeDiff = node.checkTreeEquals(getSubject());
46         if (treeDiff != null) {
47           failWithRawMessage("%s", treeDiff);
48         }
49       }
50     
51       public void hasType(Token type) {
52         String message = "Node is of type " + getSubject().getType() + " not of type " + type;
53         assertEquals(message, type, getSubject().getType());
54       }
55     
56       public void hasCharno(int charno) {
57         assertEquals(charno, getSubject().getCharno());
58       }
59     
60       public void hasLineno(int lineno) {
61         assertEquals(lineno, getSubject().getLineno());
62       }
63     
64       public void hasLength(int length) {
65         assertEquals(length, getSubject().getLength());
66       }
67     }
