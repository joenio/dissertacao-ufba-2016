27     /**
28      * Checks for misplaced, misused or deprecated JSDoc annotations.
29      *
30      * @author chadkillingsworth@gmail.com (Chad Killingsworth)
31      */
32     final class CheckJSDoc extends AbstractPostOrderCallback implements HotSwapCompilerPass {
33     
34       public static final DiagnosticType MISPLACED_MSG_ANNOTATION =
35           DiagnosticType.disabled("JSC_MISPLACED_MSG_ANNOTATION",
36               "Misplaced message annotation. @desc, @hidden, and @meaning annotations should only"
37                       + "be on message nodes.");
38     
39       public static final DiagnosticType MISPLACED_ANNOTATION =
40           DiagnosticType.warning("JSC_MISPLACED_ANNOTATION",
41               "Misplaced {0} annotation. {1}");
42     
43       public static final DiagnosticType ANNOTATION_DEPRECATED =
44           DiagnosticType.warning("JSC_ANNOTATION_DEPRECATED",
45               "The {0} annotation is deprecated. {1}");
46     
47       public static final DiagnosticType DISALLOWED_MEMBER_JSDOC =
48           DiagnosticType.warning("JSC_DISALLOWED_MEMBER_JSDOC",
49               "Class level JSDocs (@interface, @extends, etc.) are not allowed on class members");
50     
51       static final DiagnosticType ARROW_FUNCTION_AS_CONSTRUCTOR = DiagnosticType.error(
52           "JSC_ARROW_FUNCTION_AS_CONSTRUCTOR",
53           "Arrow functions cannot be used as constructors");
54     
55       static final DiagnosticType DEFAULT_PARAM_MUST_BE_MARKED_OPTIONAL = DiagnosticType.error(
56           "JSC_DEFAULT_PARAM_MUST_BE_MARKED_OPTIONAL",
57           "Inline JSDoc on default parameters must be marked as optional");

59       private final AbstractCompiler compiler;

61       CheckJSDoc(AbstractCompiler compiler) {
62         this.compiler = compiler;
63       }

65       @Override
66       public void process(Node externs, Node root) {
67         NodeTraversal.traverseEs6(compiler, externs, this);
68         NodeTraversal.traverseEs6(compiler, root, this);
69       }

71       @Override
72       public void hotSwapScript(Node scriptRoot, Node originalRoot) {
73         NodeTraversal.traverseEs6(compiler, scriptRoot, this);
74       }

76       @Override
77       public void visit(NodeTraversal t, Node n, Node parent) {
78         JSDocInfo info = n.getJSDocInfo();
79         validateTypeAnnotations(n, info);
80         validateFunctionJsDoc(n, info);
81         validateMsgJsDoc(n, info);
82         validateDeprecatedJsDoc(n, info);
83         validateNoCollapse(n, info);
84         validateClassLevelJsDoc(n, info);
85         validateArrowFunction(n);
86         validateDefaultValue(n, info);
87         validateTemplates(n, info);
88         validateTypedefs(n, info);
89         validateNoSideEffects(n, info);
90         validateAbstractJsDoc(n, info);
91       }
92     
93       private void validateTypedefs(Node n, JSDocInfo info) {
94         if (info != null && info.getTypedefType() != null && isClassDecl(n)) {
95           reportMisplaced(n, "typedef", "@typedef does not make sense on a class declaration.");
96         }
97       }
98     
99       private void validateTemplates(Node n, JSDocInfo info) {
100         if (info != null
101             && !info.getTemplateTypeNames().isEmpty()
102             && !info.isConstructorOrInterface()
103             && !isClassDecl(n)
104             && !info.containsFunctionDeclaration()) {
105           if (getFunctionDecl(n) != null) {
106             reportMisplaced(n, "template",
107                 "The template variable is unused."
108                 + " Please remove the @template annotation.");
109           } else {
110             reportMisplaced(n, "template",
111                 "@template is only allowed in class, constructor, interface, function "
112                 + "or method declarations");
113           }
114         }
115       }

117       /**
118        * @return The function node associated with the function declaration associated with the
119        *     specified node, no null if no such function exists.
120        */
121       @Nullable
122       private Node getFunctionDecl(Node n) {
123         if (n.isFunction()) {
124           return n;
125         }
126         if (n.isMemberFunctionDef()) {
127           return n.getFirstChild();
128         }
129         if (n.isVar()
130             && n.getFirstFirstChild() != null
131             && n.getFirstFirstChild().isFunction()) {
132           return n.getFirstFirstChild();
133         }
134     
135         if (n.isAssign() && n.getFirstChild().isQualifiedName() && n.getLastChild().isFunction()) {
136           return n.getLastChild();
137         }
138     
139         return null;
140       }
141     
142       private boolean isClassDecl(Node n) {
143         return isClass(n)
144             || (n.isAssign() && isClass(n.getLastChild()))
145             || (NodeUtil.isNameDeclaration(n) && isNameIntializeWithClass(n.getFirstChild()))
146             || isNameIntializeWithClass(n);
147       }
148     
149       private boolean isNameIntializeWithClass(Node n) {
150         return n != null && n.isName() && n.hasChildren() && isClass(n.getFirstChild());
151       }
152     
153       private boolean isClass(Node n) {
154         return n.isClass()
155             || (n.isCall() && compiler.getCodingConvention().isClassFactoryCall(n));
156       }

158       /**
159        * Checks that class-level annotations like @interface/@extends are not used on member functions.
160        */
161       private void validateClassLevelJsDoc(Node n, JSDocInfo info) {
162         if (info != null && n.isMemberFunctionDef()
163             && hasClassLevelJsDoc(info)) {
164           report(n, DISALLOWED_MEMBER_JSDOC);
165         }
166       }
167     
168       private void validateAbstractJsDoc(Node n, JSDocInfo info) {
169         if (info == null || !info.isAbstract()) {
170           return;
171         }
172         Node functionNode = getFunctionDecl(n);
173     
174         if (functionNode == null) {
175           // @abstract annotation on a non-function
176           report(n, MISPLACED_ANNOTATION);
177           return;
178         }
179     
180         if (NodeUtil.getFunctionBody(functionNode).hasChildren()) {
181           // @abstract annotation on a function with a non-empty body
182           report(n, MISPLACED_ANNOTATION);
183           return;
184         }
185     
186         if (n.isMemberFunctionDef() && "constructor".equals(n.getString())) {
187           // @abstract annotation on an ES6 constructor
188           report(n, MISPLACED_ANNOTATION);
189           return;
190         }
191     
192         if (!n.isMemberFunctionDef() && !NodeUtil.isPrototypeMethod(functionNode)) {
193           // @abstract annotation on a non-method (or static method) in ES5
194           report(n, MISPLACED_ANNOTATION);
195           return;
196         }
197     
198         if (n.isStaticMember()) {
199           // @abstract annotation on a static method in ES6
200           report(n, MISPLACED_ANNOTATION);
201           return;
202         }
203       }
204     
205       private boolean hasClassLevelJsDoc(JSDocInfo info) {
206         return info.isConstructorOrInterface()
207             || info.hasBaseType()
208             || info.getImplementedInterfaceCount() != 0
209             || info.getExtendedInterfacesCount() != 0;
210       }

212       /**
213        * Checks that deprecated annotations such as @expose are not present
214        */
215       private void validateDeprecatedJsDoc(Node n, JSDocInfo info) {
216         if (info != null && info.isExpose()) {
217           report(n, ANNOTATION_DEPRECATED, "@expose",
218                   "Use @nocollapse or @export instead.");
219         }
220       }
221     
222       /**
223        * Warns when nocollapse annotations are present on nodes
224        * which are not eligible for property collapsing.
225        */
226       private void validateNoCollapse(Node n, JSDocInfo info) {
227         if (n.isFromExterns()) {
228           if (info != null && info.isNoCollapse()) {
229             // @nocollapse has no effect in externs
230             reportMisplaced(n, "nocollapse", "This JSDoc has no effect in externs.");
231           }
232           return;
233         }
234         if (!NodeUtil.isPrototypePropertyDeclaration(n.getParent())) {
235           return;
236         }
237         JSDocInfo jsdoc = n.getJSDocInfo();
238         if (jsdoc != null && jsdoc.isNoCollapse()) {
239           reportMisplaced(n, "nocollapse", "This JSDoc has no effect on prototype properties.");
240         }
241       }

243       /**
244        * Checks that JSDoc intended for a function is actually attached to a
245        * function.
246        */
247       private void validateFunctionJsDoc(Node n, JSDocInfo info) {
248         if (info == null) {
249           return;
250         }
251     
252         if (info.containsFunctionDeclaration() && !info.hasType()) {
253           // This JSDoc should be attached to a FUNCTION node, or an assignment
254           // with a function as the RHS, etc.
255           switch (n.getType()) {
256             case FUNCTION:
257             case VAR:
258             case LET:
259             case CONST:
260             case GETTER_DEF:
261             case SETTER_DEF:
262             case MEMBER_FUNCTION_DEF:
263             case STRING_KEY:
264             case COMPUTED_PROP:
265             case EXPORT:
266               return;
267             case GETELEM:
268             case GETPROP:
269               if (n.getFirstChild().isQualifiedName()) {
270                 return;
271               }
272               break;
273             case ASSIGN: {
274               // TODO(tbreisacher): Check that the RHS of the assignment is a
275               // function. Note that it can be a FUNCTION node, but it can also be
276               // a call to goog.abstractMethod, goog.functions.constant, etc.
277               return;
278             }
279           }
280           reportMisplaced(n,
281               "function", "This JSDoc is not attached to a function node. "
282                   + "Are you missing parentheses?");
283         }
284       }

286       /**
287        * Checks that annotations for messages ({@code @desc}, {@code @hidden},
288        * and {@code @meaning})
289        * are in the proper place, namely on names starting with MSG_ which
290        * indicates they should be
291        * extracted for translation. A later pass checks that the right side is
292        * a call to goog.getMsg.
293        */
294       private void validateMsgJsDoc(Node n,
295           JSDocInfo info) {
296         if (info == null) {
297           return;
298         }
299     
300         if (info.getDescription() != null || info.isHidden() || info.getMeaning() != null) {
301           boolean descOkay = false;
302           switch (n.getType()) {
303             case ASSIGN: {
304               Node lhs = n.getFirstChild();
305               if (lhs.isName()) {
306                 descOkay = lhs.getString().startsWith("MSG_");
307               } else if (lhs.isQualifiedName()) {
308                 descOkay = lhs.getLastChild().getString().startsWith("MSG_");
309               }
310               break;
311             }
312             case VAR:
313             case LET:
314             case CONST:
315               descOkay = n.getFirstChild().getString().startsWith("MSG_");
316               break;
317             case STRING_KEY:
318               descOkay = n.getString().startsWith("MSG_");
319               break;
320           }
321           if (!descOkay) {
322             report(n, MISPLACED_MSG_ANNOTATION);
323           }
324         }
325       }

327       /**
328        * Check that JSDoc with a {@code @type} annotation is in a valid place.
329        */
330       private void validateTypeAnnotations(Node n, JSDocInfo info) {
331         if (info != null && info.hasType()) {
332           boolean valid = false;
333           switch (n.getType()) {
334             // Function declarations are valid
335             case FUNCTION:
336               valid = NodeUtil.isFunctionDeclaration(n);
337               break;
338             // Object literal properties, catch declarations and variable
339             // initializers are valid.
340             case NAME:
341             case DEFAULT_VALUE:
342             case ARRAY_PATTERN:
343             case OBJECT_PATTERN:
344               Node parent = n.getParent();
345               switch (parent.getType()) {
346                 case GETTER_DEF:
347                 case SETTER_DEF:
348                 case CATCH:
349                 case FUNCTION:
350                 case VAR:
351                 case LET:
352                 case CONST:
353                 case PARAM_LIST:
354                   valid = true;
355                   break;
356               }
357               break;
358             // Casts, variable declarations, exports, and Object literal properties are valid.
359             case CAST:
360             case VAR:
361             case LET:
362             case CONST:
363             case EXPORT:
364             case STRING_KEY:
365             case GETTER_DEF:
366             case SETTER_DEF:
367               valid = true;
368               break;
369             // Property assignments are valid, if at the root of an expression.
370             case ASSIGN: {
371               Node lvalue = n.getFirstChild();
372               valid = n.getParent().isExprResult()
373                   && (lvalue.isGetProp()
374                       || lvalue.isGetElem()
375                       || lvalue.matchesQualifiedName("exports"));
376               break;
377             }
378             case GETPROP:
379               valid = n.getParent().isExprResult() && n.isQualifiedName();
380               break;
381             case CALL:
382               valid = info.isDefine();
383               break;
384             default:
385               break;
386           }
387     
388           if (!valid) {
389             reportMisplaced(n, "type", "Type annotations are not allowed here. "
390                 + "Are you missing parentheses?");
391           }
392         }
393       }
394     
395       private void reportMisplaced(Node n, String annotationName, String note) {
396         compiler.report(JSError.make(n, MISPLACED_ANNOTATION,
397             annotationName, note));
398       }
399     
400       private void report(Node n, DiagnosticType type, String... arguments) {
401         compiler.report(JSError.make(n, type, arguments));
402       }

404       /**
405        * Check that an arrow function is not annotated with {@constructor}.
406        */
407       private void validateArrowFunction(Node n) {
408         if (n.isArrowFunction()) {
409           JSDocInfo info = NodeUtil.getBestJSDocInfo(n);
410           if (info != null && info.isConstructorOrInterface()) {
411             report(n, ARROW_FUNCTION_AS_CONSTRUCTOR);
412           }
413         }
414       }
415     
416       /**
417        * Check that an arrow function is not annotated with {@constructor}.
418        */
419       private void validateDefaultValue(Node n, JSDocInfo info) {
420         if (n.isDefaultValue() && n.getParent().isParamList() && info != null) {
421           JSTypeExpression typeExpr = info.getType();
422           if (typeExpr == null) {
423             return;
424           }
425           Node typeNode = typeExpr.getRoot();
426           if (typeNode.getType() != Token.EQUALS) {
427             report(typeNode, DEFAULT_PARAM_MUST_BE_MARKED_OPTIONAL);
428           }
429         }
430       }
431     
432       /**
433        * Check that @nosideeeffects annotations are only present in externs.
434        */
435       private void validateNoSideEffects(Node n, JSDocInfo info) {
436         if (info != null && info.isNoSideEffects() && !n.isFromExterns()) {
437           reportMisplaced(n, "nosideeffects", "@nosideeffects is only supported in externs.");
438         }
439       }
