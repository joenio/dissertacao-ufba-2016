17     package com.google.javascript.jscomp.deps;
18     
19     import static com.google.common.truth.Truth.assertThat;
20     
21     import com.google.common.base.Joiner;
22     import com.google.common.collect.ImmutableList;
23     
24     import junit.framework.TestCase;
25     
26     import java.util.Collection;
27     import java.util.HashSet;
28     import java.util.Set;
29     
30     /**
31      * Tests for DependencyResolver.
32      */
33     public final class DependencyResolverTest extends TestCase {
34     
35       DependencyFile fakeDeps1 = new DependencyFile(new VirtualFile("deps1",
36           "goog.addDependency('a.js', ['a'], []);\n"
37           + "goog.addDependency('b.js', ['b'], []);\n"
38           + "goog.addDependency('c.js', ['c', 'c2'], ['a']);\n"
39           + "goog.addDependency('d.js', ['d'], ['b', 'c']);\n"));
40     
41       DependencyFile fakeDeps2 = new DependencyFile(new VirtualFile("deps2",
42           "goog.addDependency('e.js', ['e'], ['c2']);\n"
43           + "goog.addDependency('f.js', ['f'], ['b', 'c']);\n"
44           + "goog.addDependency('g.js', ['g'], ['a', 'b', 'c']);\n"
45           + "goog.addDependency('h.js', ['h', 'i'], ['g', 'd', 'c']);\n"));
46     
47       DefaultDependencyResolver resolver;
48     
49       @Override protected void setUp() throws Exception {
50         super.setUp();
51         resolver = new DefaultDependencyResolver(ImmutableList.of(fakeDeps1, fakeDeps2), false);
52       }
53     
54       public void testBasicCase() throws Exception {
55          Collection<String> deps = resolver.getDependencies("goog.require('a');");
56          assertThat(Joiner.on(",").useForNull("null").join(deps)).isEqualTo("base.js,a.js");
57       }
58     
59       public void testSimpleDependencies() throws Exception {
60          Collection<String> deps = resolver.getDependencies("goog.require('c');");
61          assertThat(Joiner.on(",").useForNull("null").join(deps)).isEqualTo("base.js,a.js,c.js");
62       }
63     
64       public void testTransitiveDependencies() throws Exception {
65          Collection<String> deps = resolver.getDependencies("goog.require('e');");
66          assertThat(Joiner.on(",").useForNull("null").join(deps)).isEqualTo("base.js,a.js,c.js,e.js");
67       }
68     
69       public void testMultipleRequires() throws Exception {
70          Collection<String> deps = resolver.getDependencies(
71              "goog.require('e');goog.require('a');goog.require('b');");
72          assertThat(Joiner.on(",").useForNull("null").join(deps))
73              .isEqualTo("base.js,a.js,c.js,e.js,b.js");
74       }
75     
76       public void testOneMoreForGoodMeasure() throws Exception {
77         Collection<String> deps = resolver.getDependencies(
78             "goog.require('g');goog.require('f');goog.require('c');");
79         assertThat(Joiner.on(",").useForNull("null").join(deps))
80             .isEqualTo("base.js,a.js,b.js,c.js,g.js,f.js");
81       }
82     
83       public void testSharedSeenSetNoBaseFile() throws Exception {
84         Set<String> seen = new HashSet<>();
85     
86         Collection<String> deps = resolver.getDependencies(
87         "goog.require('g');goog.require('f');goog.require('c');", seen, false);
88     
89         Collection<String> depsLater = resolver.getDependencies(
90         "goog.require('f');goog.require('c');", seen, false);
91     
92         assertThat(Joiner.on(",").useForNull("null").join(deps)).isEqualTo("a.js,b.js,c.js,g.js,f.js");
93         assertThat(depsLater).isEmpty();
94       }
95     
96       public void testSharedSeenSetNoBaseFileNewRequires() throws Exception {
97         Set<String> seen = new HashSet<>();
98     
99         Collection<String> deps = resolver.getDependencies(
100             "goog.require('f');goog.require('c');", seen, false);
101     
102         Collection<String> depsLater = resolver.getDependencies(
103             "goog.require('g');goog.require('c');", seen, false);
104     
105         assertThat(Joiner.on(",").useForNull("null").join(deps)).isEqualTo("b.js,a.js,c.js,f.js");
106         assertThat(Joiner.on(",").useForNull("null").join(depsLater)).isEqualTo("g.js");
107       }
108     
109       public void testSharedSeenSetNoBaseFileMultipleProvides() throws Exception {
110         Set<String> seen = new HashSet<>();
111     
112         Collection<String> deps = resolver.getDependencies(
113             "goog.require('h');goog.require('i');", seen, false);
114     
115         assertThat(Joiner.on(",").useForNull("null").join(deps))
116             .isEqualTo("a.js,b.js,c.js,g.js,d.js,h.js");
117       }
118     
119       public void testNonExistentProvideLoose() throws Exception {
120         Set<String> seen = new HashSet<>();
121         resolver = new DefaultDependencyResolver(ImmutableList.of(fakeDeps1), false);
122         Collection<String> deps = resolver.getDependencies(
123             "goog.require('foo');goog.require('d');", seen, false);
124     
125         assertThat(Joiner.on(",").useForNull("null").join(deps)).isEqualTo("b.js,a.js,c.js,d.js");
126       }
127     
128       public void testNonExistentProvideStrict() throws Exception {
129         Set<String> seen = new HashSet<>();
130         resolver = new DefaultDependencyResolver(ImmutableList.of(fakeDeps1), true);
131         try {
132           Collection<String> deps = resolver.getDependencies(
133               "goog.require('foo');goog.require('a');", seen, false);
134           fail("Service exception should be thrown");
135         } catch (ServiceException expected) {}
136       }

138     }
