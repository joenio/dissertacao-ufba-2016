17     package com.google.javascript.jscomp.webservice.common;
18     
19     /**
20      * All the exceptions that can be returned as error to the client of the API.
21      *
22      */
23     public abstract class AbstractWebServiceException extends Exception {
24       public abstract ErrorCode getErrorCode();
25       public abstract String getFormattedError();
26     
27       @Override
28       public String toString() {
29         return String.format("Error(%d): %s", getErrorCode().getCode(),
30             getFormattedError());
31       }
32     }
