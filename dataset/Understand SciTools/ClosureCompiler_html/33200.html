41     /**
42      * This pass walks the AST to create a Collection of 'new' nodes and
43      * 'goog.require' nodes. It reconciles these Collections, creating a
44      * warning for each discrepancy.
45      *
46      * <p>The rules on when a warning is reported are: <ul>
47      * <li>Type is referenced in code -> goog.require is required
48      *     (missingRequires check fails if it's not there)
49      * <li>Type is referenced in an @extends or @implements -> goog.require is required
50      *     (missingRequires check fails if it's not there)
51      * <li>Type is referenced in other JsDoc (@type etc) -> goog.require is optional
52      *     (don't warn, regardless of if it is there)
53      * <li>Type is not referenced at all -> goog.require is forbidden
54      *     (extraRequires check fails if it is there)
55      * </ul>
56      *
57      */
58     class CheckRequiresForConstructors implements HotSwapCompilerPass, NodeTraversal.Callback {
59       private final AbstractCompiler compiler;
60       private final CodingConvention codingConvention;

133       // Return true if the name is a class name (starts with an uppercase
134       // character, but is not in all-caps).
135       private static boolean isClassName(String name) {
136         return isClassOrConstantName(name)
137             && !name.equals(name.toUpperCase());
138       }
139     
140       // Return true if the name looks like a class name or a constant name.
141       private static boolean isClassOrConstantName(String name) {
142         return name != null && name.length() > 1
143             && Character.isUpperCase(name.charAt(0));
144       }
145     
146       // Return the shortest prefix of the className that refers to a class,
147       // or null if no part refers to a class.
148       private static List<String> getClassNames(String qualifiedName) {
149         ImmutableList.Builder<String> classNames = ImmutableList.builder();
150         List<String> parts = DOT_SPLITTER.splitToList(qualifiedName);
151         for (int i = 0; i < parts.size(); i++) {
152           String part = parts.get(i);
153           if (isClassOrConstantName(part)) {
154             classNames.add(DOT_JOINER.join(parts.subList(0, i + 1)));
155           }
156         }
157         return classNames.build();
158       }

160       // TODO(tbreisacher): Update CodingConvention.extractClassNameIf{Require,Provide} to match this.
161       private String extractNamespace(Node call, String functionName) {
162         Node callee = call.getFirstChild();
163         if (callee.isGetProp() && callee.matchesQualifiedName(functionName)) {
164           Node target = callee.getNext();
165           if (target != null && target.isString()) {
166             return target.getString();
167           }
168         }
169         return null;
170       }

172       private String extractNamespaceIfRequire(Node call) {
173         return extractNamespace(call, "goog.require");
174       }

176       private String extractNamespaceIfProvide(Node call) {
177         return extractNamespace(call, "goog.provide");

180       @Override
181       public boolean shouldTraverse(NodeTraversal t, Node n, Node parent) {
182         if (n.isCall() && n.getFirstChild().matchesQualifiedName("goog.scope")) {
183           Node function = n.getSecondChild();
184           if (function.isFunction()) {
185             googScopeBlock = NodeUtil.getFunctionBody(function);
186           }
187         }
188     
189         return parent == null || !parent.isScript() || !t.getInput().isExtern();
190       }

192       @Override
193       public void visit(NodeTraversal t, Node n, Node parent) {
194         maybeAddJsDocUsages(t, n);
195         switch (n.getType()) {
196           case ASSIGN:
197             maybeAddProvidedName(n);
198             break;
199           case VAR:
200           case LET:
201           case CONST:
202             maybeAddProvidedName(n);
203             maybeAddGoogScopeUsage(t, n, parent);
204             break;
205           case FUNCTION:
206             // Exclude function expressions.
207             if (NodeUtil.isStatement(n)) {
208               maybeAddProvidedName(n);
209             }
210             break;
211           case NAME:
212             if (!NodeUtil.isLValue(n)) {
213               visitQualifiedName(n);
214             }
215             break;
216           case GETPROP:
217             // If parent is a GETPROP, they will handle the weak usages.
218             if (!parent.isGetProp() && n.isQualifiedName()) {
219               visitQualifiedName(n);
220             }
221             break;
222           case CALL:
223             visitCallNode(t, n, parent);
224             break;
225           case SCRIPT:
226             visitScriptNode(t);
227             reset();
228             break;
229           case NEW:
230             visitNewNode(t, n);
231             break;
232           case CLASS:
233             visitClassNode(t, n);
234             break;
235           case IMPORT:
236             visitImportNode(n);
237             break;
238         }
239       }

241       private void reset() {
242         this.usages.clear();
243         this.weakUsages.clear();
244         this.requires.clear();
245         this.closurizedNamespaces.clear();
246         this.providedNames.clear();
247         this.googScopeBlock = null;
248       }

250       private void visitScriptNode(NodeTraversal t) {
251         if (mode == Mode.SINGLE_FILE && requires.isEmpty() && closurizedNamespaces.isEmpty()) {
252           // Likely a file that isn't using Closure at all.
253           return;
254         }
255     
256         Set<String> namespaces = new HashSet<>();
257     
258         // For every usage, check that there is a goog.require, and warn if not.
259         for (Map.Entry<String, Node> entry : usages.entrySet()) {
260           String namespace = entry.getKey();
261           if (namespace.endsWith(".call") || namespace.endsWith(".apply")) {
262             namespace = namespace.substring(0, namespace.lastIndexOf('.'));
263           }
264           if (namespace.startsWith("goog.global.")
265               // Most functions in base.js are goog.someName, but
266               // goog.module.{get,declareLegacyNamespace} are the exceptions, so just check for them
267               // explicitly.
268               || namespace.equals("goog.module.get")
269               || namespace.equals("goog.module.declareLegacyNamespace")) {
270             continue;
271           }
272     
273           Node node = entry.getValue();
274           JSDocInfo info = NodeUtil.getBestJSDocInfo(NodeUtil.getEnclosingStatement(node));
275           if (info != null && info.getSuppressions().contains("missingRequire")) {
276             continue;
277           }
278     
279           List<String> classNames = getClassNames(namespace);
280           // The parent namespace of the outermost class is also checked, so that classes
281           // used by goog.module are still checked properly. This may cause missing requires
282           // to be missed but in practice that should happen rarely.
283           String nonNullClassName = Iterables.getFirst(classNames, namespace);
284           String parentNamespace = null;
285           int separatorIndex = nonNullClassName.lastIndexOf('.');
286           if (separatorIndex > 0) {
287             parentNamespace = nonNullClassName.substring(0, separatorIndex);
288           }
289           if ("goog".equals(parentNamespace)
290               && !isClassName(nonNullClassName.substring(separatorIndex + 1))) {
291             // This is probably something provided in Closure's base.js so it doesn't need
292             // to be required.
293             continue;
294           }
295     
296           boolean providedByConstructors =
297               providedNames.contains(namespace) || providedNames.contains(parentNamespace);
298           boolean providedByRequires =
299               requires.containsKey(namespace) || requires.containsKey(parentNamespace);
300     
301           for (String className : classNames) {
302             if (providedNames.contains(className)) {
303               providedByConstructors = true;
304             }
305             if (requires.containsKey(className)) {
306               providedByRequires = true;
307             }
308           }
309     
310           if (!providedByConstructors && !providedByRequires && !namespaces.contains(namespace)) {
311             // TODO(mknichel): If the symbol is not explicitly provided, find the next best
312             // symbol from the provides in the same file.
313             String rootName = Splitter.on('.').split(namespace).iterator().next();
314             if (mode != Mode.SINGLE_FILE || closurizedNamespaces.contains(rootName)) {
315               if (node.isCall()) {
316                 String defaultName = parentNamespace != null ? parentNamespace : namespace;
317                 String nameToReport = Iterables.getFirst(classNames, defaultName);
318                 compiler.report(t.makeError(node, MISSING_REQUIRE_CALL_WARNING, nameToReport));
319               } else {
320                 compiler.report(t.makeError(node, MISSING_REQUIRE_WARNING, namespace));
321               }
322               namespaces.add(namespace);
323             }
324           }
325         }
326     
327         // For every goog.require, check that there is a usage (in either usages or weakUsages)
328         // and warn if there is not.
329         for (Map.Entry<String, Node> entry : requires.entrySet()) {
330           String require = entry.getKey();
331           Node call = entry.getValue();
332           if (!usages.containsKey(require) && !weakUsages.contains(require)) {
333             reportExtraRequireWarning(call, require);
334           }
335         }
336       }

338       private void reportExtraRequireWarning(Node call, String require) {
339         if (DEFAULT_EXTRA_NAMESPACES.contains(require)) {
340           return;
341         }
342         JSDocInfo jsDoc = NodeUtil.getBestJSDocInfo(call);
343         if (jsDoc != null && jsDoc.getSuppressions().contains("extraRequire")) {
344           // There is a @suppress {extraRequire} on the call node or its enclosing statement.
345           // This is one of the acceptable places for a @suppress, per
346           // https://github.com/google/closure-compiler/wiki/@suppress-annotations
347           return;
348         }
349         compiler.report(JSError.make(call, EXTRA_REQUIRE_WARNING, require));
350       }

352       private void reportDuplicateRequireWarning(Node call, String require) {
353         compiler.report(JSError.make(call, DUPLICATE_REQUIRE_WARNING, require));
354       }

356       /**
357        * @param localName The name that should be used in this file.
358        *
359        * <pre>
360        * Require style                        | localName
361        * -------------------------------------|----------
362        * goog.require('foo.bar');             | foo.bar
363        * var bar = goog.require('foo.bar');   | bar
364        * var {qux} = goog.require('foo.bar'); | qux
365        * import {qux} from 'foo.bar';         | qux
366        * </pre>
367        */
368       private void visitRequire(String localName, Node node) {
369         if (requires.containsKey(localName)) {
370           reportDuplicateRequireWarning(node, localName);
371         } else {
372           requires.put(localName, node);
373         }
374       }

376       private void visitImportNode(Node importNode) {
377         Node defaultImport = importNode.getFirstChild();
378         if (defaultImport.isName()) {
379           visitRequire(defaultImport.getString(), importNode);
380         }
381         Node namedImports = defaultImport.getNext();
382         if (namedImports.getType() == Token.IMPORT_SPECS) {
383           for (Node importSpec : namedImports.children()) {
384             visitRequire(importSpec.getLastChild().getString(), importNode);
385           }
386         }
387       }

391           String rootName = Splitter.on('.').split(requiredName).iterator().next();
392           closurizedNamespaces.add(rootName);
393         }

396       private void visitCallNode(NodeTraversal t, Node call, Node parent) {
397         String required = extractNamespaceIfRequire(call);
398         if (required != null) {
399           maybeAddClosurizedNamespace(required);
400           if (parent.isName()) {
401             visitRequire(parent.getString(), call);
402           } else if (parent.isDestructuringLhs() && parent.getFirstChild().isObjectPattern()) {
403             for (Node stringKey : parent.getFirstChild().children()) {
404               if (stringKey.hasChildren()) {
405                 visitRequire(stringKey.getFirstChild().getString(), call);
406               } else {
407                 visitRequire(stringKey.getString(), call);
408               }
409             }
410           } else {
411             visitRequire(required, call);
412           }
413           return;
414         }
415         String provided = extractNamespaceIfProvide(call);
416         if (provided != null) {
417           providedNames.add(provided);
418           return;
419         }
420     
421         Node callee = call.getFirstChild();
422         if (callee.matchesQualifiedName("goog.module.get") && call.getSecondChild().isString()) {
423           weakUsages.add(call.getSecondChild().getString());
424         }
425     
426         if (codingConvention.isClassFactoryCall(call)) {
427           if (parent.isName()) {
428             providedNames.add(parent.getString());
429           } else if (parent.isAssign()) {
430             providedNames.add(parent.getFirstChild().getQualifiedName());
431           }
432         }
433     
434         if (callee.isName()) {
435           weakUsages.add(callee.getString());
436         } else if (callee.isQualifiedName()) {
437           Node root = NodeUtil.getRootOfQualifiedName(callee);
438           if (root.isName()) {
439             Var var = t.getScope().getVar(root.getString());
440             if (var == null || (!var.isExtern() && !var.isLocal())) {
441               usages.put(callee.getQualifiedName(), call);
442             }
443           }
444         }
445       }

447       private void addWeakUsagesOfAllPrefixes(String qualifiedName) {
448         // For "foo.bar.baz.qux" add weak usages for "foo.bar.baz.qux", "foo.bar.baz",
449         // "foo.bar", and "foo" because those might all be goog.provide'd in different files,
450         // so it doesn't make sense to require the user to goog.require all of them.
451         for (int i = qualifiedName.indexOf('.'); i != -1; i = qualifiedName.indexOf('.', i + 1)) {
452           String prefix = qualifiedName.substring(0, i);
453           weakUsages.add(prefix);
454         }
455         weakUsages.add(qualifiedName);
456       }

458       private void visitQualifiedName(Node getpropOrName) {
459         if (getpropOrName.isName() && getpropOrName.getString() != null) {
460           // If the referenced thing is a goog.require as desugared from goog.module().
461           if (getpropOrName.getBooleanProp(Node.GOOG_MODULE_REQUIRE)) {
462             Node declStatement = NodeUtil.getEnclosingStatement(getpropOrName);
463             if (NodeUtil.isNameDeclaration(declStatement)) {
464               for (Node varChild : declStatement.children()) {
465                 // Normal declaration.
466                 if (varChild.isName()) {
467                   requires.put(varChild.getString(), getpropOrName);
468                 }
469                 // Object destructuring declaration.
470                 if (varChild.isObjectPattern()) {
471                   for (Node objectChild : varChild.children()) {
472                     if (objectChild.isStringKey()) {
473                       requires.put(objectChild.getString(), getpropOrName);
474                     }
475                   }
476                 }
477               }
478             }
479           }
480         }
481     
482         addWeakUsagesOfAllPrefixes(getpropOrName.getQualifiedName());
483       }

485       private void visitNewNode(NodeTraversal t, Node newNode) {
486         Node qNameNode = newNode.getFirstChild();
487     
488         // Single names are likely external, but if this is running in single-file mode, they
489         // will not be in the externs, so add a weak usage.
490         if (mode == Mode.SINGLE_FILE && qNameNode.isName()) {
491           weakUsages.add(qNameNode.getString());
492           return;
493         }
494     
495         // If the ctor is something other than a qualified name, ignore it.
496         if (!qNameNode.isQualifiedName()) {
497           return;
498         }
499     
500         // Grab the root ctor namespace.
501         Node root = NodeUtil.getRootOfQualifiedName(qNameNode);
502     
503         // We only consider programmer-defined constructors that are
504         // global variables, or are defined on global variables.
505         if (!root.isName()) {
506           return;
507         }
508     
509         String name = root.getString();
510         Var var = t.getScope().getVar(name);
511         if (var != null
512             && (var.isExtern()
513                 || var.getSourceFile() == newNode.getStaticSourceFile()
514                 || ClosureRewriteModule.isModuleExport(name))) {
515           return;
516         }
517         usages.put(qNameNode.getQualifiedName(), newNode);
518     
519         // for "new foo.bar.Baz.Qux" add weak usages for "foo.bar.Baz", "foo.bar", and "foo"
520         // because those might be goog.provide'd from a different file than foo.bar.Baz.Qux,
521         // so it doesn't make sense to require the user to goog.require all of them.
522         for (; qNameNode != null; qNameNode = qNameNode.getFirstChild()) {
523           weakUsages.add(qNameNode.getQualifiedName());
524         }
525       }

527       private void visitClassNode(NodeTraversal t, Node classNode) {
528         String name = NodeUtil.getName(classNode);
529         if (name != null) {
530           providedNames.add(name);
531         }
532     
533         Node extendClass = classNode.getSecondChild();
534     
535         // If the superclass is something other than a qualified name, ignore it.
536         if (!extendClass.isQualifiedName()) {
537           return;
538         }
539     
540         // Single names are likely external, but if this is running in single-file mode, they
541         // will not be in the externs, so add a weak usage.
542         if (mode == Mode.SINGLE_FILE && extendClass.isName()) {
543           weakUsages.add(extendClass.getString());
544           return;
545         }
546     
547         Node root = NodeUtil.getRootOfQualifiedName(extendClass);
548     
549         // It should always be a name. Extending this.something or
550         // super.something is unlikely.
551         // We only consider programmer-defined superclasses that are
552         // global variables, or are defined on global variables.
553         if (root.isName()) {
554           String rootName = root.getString();
555           Var var = t.getScope().getVar(rootName);
556           if (var != null
557               && (var.isLocal() || var.isExtern() || ClosureRewriteModule.isModuleExport(rootName))) {
558             // "require" not needed for these
559           } else {
560             usages.put(extendClass.getQualifiedName(), extendClass);
561           }
562         }
563       }

565       private void maybeAddProvidedName(Node n) {
566         Node name = n.getFirstChild();
567         if (name.isQualifiedName()) {
568           providedNames.add(name.getQualifiedName());
569         }
570       }

572       /**
573        * "var Dog = some.cute.Dog;" counts as a usage of some.cute.Dog, if it's immediately
574        * inside a goog.scope function.
575        */
576       private void maybeAddGoogScopeUsage(NodeTraversal t, Node n, Node parent) {
577         Preconditions.checkState(NodeUtil.isNameDeclaration(n));
578         if (n.getChildCount() == 1 && parent == googScopeBlock) {
579           Node rhs = n.getFirstFirstChild();
580           if (rhs != null && rhs.isQualifiedName()) {
581             Node root = NodeUtil.getRootOfQualifiedName(rhs);
582             if (root.isName()) {
583               Var var = t.getScope().getVar(root.getString());
584               if (var == null || (var.isGlobal() && !var.isExtern())) {
585                 usages.put(rhs.getQualifiedName(), rhs);
586               }
587             }
588           }
589         }
590       }

592       /**
593        * If this returns true, check for @extends and @implements annotations on this node.
594        * Otherwise, it's probably an alias for an existing class, so skip those annotations.
595        *
596        * @return Whether the given node declares a function. True for the following forms:
597        *      <li><pre>function foo() {}</pre>
598        *      <li><pre>var foo = function() {};</pre>
599        *      <li><pre>foo.bar = function() {};</pre>
600        */
601       private boolean declaresFunction(Node n) {
602         if (n.isFunction()) {
603           return true;
604         }
605     
606         if (n.isAssign() && n.getLastChild().isFunction()) {
607           return true;
608         }
609     
610         if (NodeUtil.isNameDeclaration(n)
611             && n.getFirstChild().hasChildren()
612             && n.getFirstFirstChild().isFunction()) {
613           return true;
614         }
615     
616         return false;
617       }

619       private void maybeAddJsDocUsages(NodeTraversal t, Node n) {
620         JSDocInfo info = n.getJSDocInfo();
621         if (info == null) {
622           return;
623         }
624     
625         if (declaresFunction(n)) {
626           for (JSTypeExpression expr : info.getImplementedInterfaces()) {
627             maybeAddUsage(t, n, expr);
628           }
629           if (info.getBaseType() != null) {
630             maybeAddUsage(t, n, info.getBaseType());
631           }
632           for (JSTypeExpression extendedInterface : info.getExtendedInterfaces()) {
633             maybeAddUsage(t, n, extendedInterface);
634           }
635         }
636     
637         for (Node typeNode : info.getTypeNodes()) {
638           maybeAddWeakUsage(t, n, typeNode);
639         }
640       }

642       /**
643        * Adds a weak usage for the given type expression (unless it references a variable that is
644        * defined in the externs, in which case no goog.require() is needed). When a "weak usage"
645        * is added, it means that a goog.require for that type is optional: No
646        * warning is given whether the require is there or not.
647        */
648       private void maybeAddWeakUsage(NodeTraversal t, Node n, Node typeNode) {
649         maybeAddUsage(t, n, typeNode, false, Predicates.<Node>alwaysTrue());
650       }
651     
652       /**
653        * Adds a usage for the given type expression (unless it references a variable that is
654        * defined in the externs, in which case no goog.require() is needed). When a usage is
655        * added, it means that there should be a goog.require for that type.
656        */
657       private void maybeAddUsage(NodeTraversal t, Node n, final JSTypeExpression expr) {
658         // Just look at the root node, don't traverse.
659         Predicate<Node> pred =
660             new Predicate<Node>() {
661               @Override
662               public boolean apply(Node n) {
663                 return n == expr.getRoot();
664               }
665             };
666         maybeAddUsage(t, n, expr.getRoot(), true, pred);
667       }
668     
669       private void maybeAddUsage(
670           final NodeTraversal t,
671           final Node n,
672           Node rootTypeNode,
673           final boolean markStrongUsages,
674           Predicate<Node> pred) {
675         Visitor visitor =
676             new Visitor() {
677               @Override
678               public void visit(Node typeNode) {
679                 if (typeNode.isString()) {
680                   String typeString = typeNode.getString();
681                   if (mode == Mode.SINGLE_FILE && !typeString.contains(".")) {
682                     // If using a single-name type, it's probably something like Error, which we
683                     // don't have externs for.
684                     weakUsages.add(typeString);
685                     return;
686                   }
687                   String rootName = Splitter.on('.').split(typeString).iterator().next();
688                   Var var = t.getScope().getVar(rootName);
689                   if ((var == null || !var.isExtern())
690                       && !ClosureRewriteModule.isModuleExport(rootName)) {
691                     if (markStrongUsages) {
692                       usages.put(typeString, n);
693                     } else {
694                       // If we're not adding strong usages here, add weak usages for the prefixes of the
695                       // namespace, like we do for GETPROP nodes. Otherwise we get an extra require
696                       // warning for cases like:
697                       //
698                       //     goog.require('foo.bar.SomeService');
699                       //
700                       //     /** @constructor @extends {foo.bar.SomeService.Handler} */
701                       //     var MyHandler = function() {};
702                       addWeakUsagesOfAllPrefixes(typeString);
703                     }
704                   } else {
705                     // Even if the root namespace is in externs, add a weak usage because the full
706                     // namespace may still be goog.provided.
707                     weakUsages.add(typeString);
708                   }
709                 }
710               }
711             };
712     
713         NodeUtil.visitPreOrder(rootTypeNode, visitor, pred);
714       }
715     }
