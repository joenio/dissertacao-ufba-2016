88         if (parent == null) {

95         }
96         this.root = root;

99         this.commonTypes = commonTypes;

193       // In other languages, type names and variable names are in distinct
194       // namespaces and don't clash.
195       // But because our typedefs and enums are var declarations, they are in the
196       // same namespace as other variables.
197       boolean isDefinedLocally(String name, boolean includeTypes) {
198         Preconditions.checkNotNull(name);
199         Preconditions.checkState(!name.contains("."));
200         if (locals.containsKey(name)
201             || formals.contains(name)
202             || localNamespaces.containsKey(name)
203             || localFunDefs.containsKey(name)
204             || "this".equals(name)
205             || externs.containsKey(name)
206             || localTypedefs.containsKey(name)) {
207           return true;
208         }
209         if (includeTypes) {
210           return unknownTypeNames.contains(name)
211               || declaredType != null && declaredType.isTypeVariableDefinedLocally(name);
212         }
213         return false;
214       }

216       // For variables it is the same as isDefinedLocally; for properties it looks
217       // for a definition in any scope.
218       boolean isDefined(Node qnameNode) {
219         Preconditions.checkArgument(qnameNode.isQualifiedName());
220         if (qnameNode.isName()) {
221           return isDefinedLocally(qnameNode.getString(), false);
222         } else if (qnameNode.isThis()) {
223           return true;
224         }
225         QualifiedName qname = QualifiedName.fromNode(qnameNode);
226         String leftmost = qname.getLeftmostName();
227         if (isNamespace(leftmost)) {
228           return getNamespace(leftmost).isDefined(qname.getAllButLeftmost());
229         }
230         return parent == null ? false : parent.isDefined(qnameNode);
231       }

340       @Override
341       public JSType getDeclaredTypeOf(String name) {
342         Preconditions.checkArgument(!name.contains("."));
343         if ("this".equals(name)) {
344           if (!hasThis()) {
345             return null;
346           }
347           return getDeclaredFunctionType().getThisType();
348         }
349         Declaration decl = getLocalDeclaration(name, false);
350         if (decl != null) {
351           if (decl.getTypeOfSimpleDecl() != null) {
352             Preconditions.checkState(!decl.getTypeOfSimpleDecl().isBottom(),
353                 "%s was bottom", name);
354             return decl.getTypeOfSimpleDecl();
355           }
356           NTIScope funScope = (NTIScope) decl.getFunctionScope();
357           if (funScope != null) {
358             return getCommonTypes().fromFunctionType(
359                 funScope.getDeclaredFunctionType().toFunctionType());
360           }
361           Preconditions.checkState(decl.getNamespace() == null);
362           return null;
363         }
364         // When a function is a namespace, the parent scope has a better type.
365         if (name.equals(this.name) && !parent.isFunctionNamespace(name)) {
366           return getCommonTypes()
367               .fromFunctionType(getDeclaredFunctionType().toFunctionType());
368         }
369         if (parent != null) {
370           return parent.getDeclaredTypeOf(name);
371         }
372         return null;
373       }

375       boolean hasUndeclaredFormalsOrOuters() {
376         for (String formal : formals) {
377           if (getDeclaredTypeOf(formal) == null) {
378             return true;
379           }
380         }
381         for (String outer : outerVars) {
382           JSType declType = getDeclaredTypeOf(outer);
383           if (declType == null
384               // Undeclared functions have a non-null declared type,
385               //  but they always have a return type of unknown
386               || (declType.getFunType() != null
387                   && declType.getFunType().getReturnType().isUnknown())) {
388             return true;
389           }
390         }
391         return false;
392       }

427       // We don't check for duplicates here, mainly because we add some
428       // intentionally during the two phases of GlobalTypeInfo.
429       // If a variable is declared many times in a scope, the last definition
430       // overwrites the previous ones. For correctness, we rely on the fact that
431       // the var-check passes run before type checking.
432       void addLocal(String name, JSType declType,
433           boolean isConstant, boolean isFromExterns) {
434         Preconditions.checkArgument(!name.contains("."));
435         if (isConstant) {
436           constVars.add(name);
437         }
438         if (isFromExterns) {
439           externs.put(name, declType);
440         } else {
441           locals.put(name, declType);
442         }
443       }

445       static void mayRecordEscapedVar(NTIScope s, String name) {
446         if (s.isDefinedLocally(name, false)) {
447           return;
448         }
449         while (s != null) {
450           if (s.isDefinedLocally(name, false)) {
451             s.escapedVars.add(name);
452             return;
453           }
454           s = s.parent;
455         }
456       }

480       void addFunNamespace(Node qnameNode) {
481         if (qnameNode.isName()) {
482           String varName = qnameNode.getString();
483           Preconditions.checkArgument(isDefinedLocally(varName, false));
484           Preconditions.checkState(!this.localNamespaces.containsKey(varName));
485           NTIScope s = Preconditions.checkNotNull(this.localFunDefs.get(varName));
486           this.localNamespaces.put(varName, new FunctionNamespace(varName, s));
487         } else {
488           Preconditions.checkArgument(!isNamespace(qnameNode));
489           QualifiedName qname = QualifiedName.fromNode(qnameNode);
490           Namespace ns = getNamespace(qname.getLeftmostName());
491           NTIScope s = (NTIScope) ns.getDeclaration(qname).getFunctionScope();
492           ns.addNamespace(qname.getAllButLeftmost(),
493               new FunctionNamespace(qname.toString(), s));
494         }
495       }

529       void addNamespace(Node qnameNode, Namespace ns) {
530         if (ns instanceof EnumType) {
531           this.localEnums.add((EnumType) ns);
532         }
533         if (qnameNode.isName()) {
534           String varName = qnameNode.getString();
535           Preconditions.checkState(!this.localNamespaces.containsKey(varName));
536           this.localNamespaces.put(varName, ns);
537           if (qnameNode.isFromExterns() && !this.externs.containsKey(varName)) {
538             // We don't know the full type of a namespace until after we see all
539             // its properties. But we want to add it to the externs, otherwise it
540             // is treated as a local and initialized to the wrong thing in NTI.
541             this.externs.put(qnameNode.getString(), null);
542           }
543         } else {
544           Preconditions.checkState(!isDefined(qnameNode));
545           QualifiedName qname = QualifiedName.fromNode(qnameNode);
546           Namespace rootns = getNamespace(qname.getLeftmostName());
547           rootns.addNamespace(qname.getAllButLeftmost(), ns);
548         }
549       }

557       private Declaration getLocalDeclaration(String name, boolean includeTypes) {
558         Preconditions.checkArgument(!name.contains("."));
559         if (!isDefinedLocally(name, includeTypes)) {
560           return null;
561         }
562         JSType type = null;
563         boolean isTypeVar = false;
564         if ("this".equals(name)) {
565           type = getDeclaredTypeOf("this");
566         } else if (locals.containsKey(name)) {
567           type = locals.get(name);
568         } else if (formals.contains(name)) {
569           int formalIndex = formals.indexOf(name);
570           if (declaredType != null && formalIndex != -1) {
571             JSType formalType = declaredType.getFormalType(formalIndex);
572             if (formalType != null && !formalType.isBottom()) {
573               type = formalType;
574             }
575           }
576         } else if (localFunDefs.containsKey(name)) {
577           // After finalization, the externs contain the correct type for
578           // external function namespaces, don't rely on localFunDefs
579           if (isFinalized && externs.containsKey(name)) {
580             type = externs.get(name);
581           }
582         } else if (localTypedefs.containsKey(name) || localNamespaces.containsKey(name)) {
583           // Any further declarations are shadowed
584         } else if (declaredType != null && declaredType.isTypeVariableDefinedLocally(name)) {
585           isTypeVar = true;
586           type = JSType.fromTypeVar(declaredType.getTypeVariableDefinedLocally(name));
587         } else if (externs.containsKey(name)) {
588           type = externs.get(name);
589         }
590         return new Declaration(type, localTypedefs.get(name),
591             localNamespaces.get(name), localFunDefs.get(name), isTypeVar,
592             constVars.contains(name));
593       }

595       @Override
596       public Declaration getDeclaration(QualifiedName qname, boolean includeTypes) {
597         if (qname.isIdentifier()) {
598           return getDeclaration(qname.getLeftmostName(), includeTypes);
599         }
600         Preconditions.checkState(!this.isFinalized,
601             "Namespaces are removed from scopes after finalization");
602         Namespace ns = getNamespace(qname.getLeftmostName());
603         if (ns == null) {
604           return maybeGetForwardDeclaration(qname.toString());
605         }
606         Declaration decl = ns.getDeclaration(qname.getAllButLeftmost());
607         return decl != null ? decl : maybeGetForwardDeclaration(qname.toString());
608       }

610       private Declaration maybeGetForwardDeclaration(String qname) {
611         NTIScope globalScope = this;
612         while (globalScope.parent != null) {
613           globalScope = globalScope.parent;
614         }
615         if (globalScope.unknownTypeNames.contains(qname)) {
616           return new Declaration(JSType.UNKNOWN, null, null, null, false, false);
617         }
618         return null;
619       }

621       public Declaration getDeclaration(String name, boolean includeTypes) {
622         Preconditions.checkArgument(!name.contains("."));
623         Declaration decl = getLocalDeclaration(name, includeTypes);
624         if (decl != null) {
625           return decl;
626         }
627         return parent == null ? null : parent.getDeclaration(name, includeTypes);

654       void finalizeScope() {
655         Preconditions.checkState(isTopLevel() || this.declaredType != null,
656             "No declared type for function-scope: %s", this.root);
657         unknownTypeNames = ImmutableSet.of();
658         JSTypes commonTypes = getCommonTypes();
659         // For now, we put types of namespaces directly into the locals.
660         // Alternatively, we could move this into NewTypeInference.initEdgeEnvs
661         for (Map.Entry<String, Namespace> entry : localNamespaces.entrySet()) {
662           String name = entry.getKey();
663           Namespace ns = entry.getValue();
664           JSType t;
665           if (ns instanceof NamespaceLit) {
666             constVars.add(name);
667             NamespaceLit nslit = (NamespaceLit) ns;
668             // The argument to maybeSetWindowInstance should only be non-null for
669             // window, but we don't check here to avoid hard-coding the name.
670             // Enforced in GlobalTypeInfo.
671             nslit.maybeSetWindowInstance(externs.get(name));
672             t = nslit.toJSType(commonTypes);
673           } else {
674             t = ns.toJSType(commonTypes);
675           }
676           if (externs.containsKey(name)) {
677             externs.put(name, t);
678           } else {
679             locals.put(name, t);
680           }
681         }
682         for (String typedefName : localTypedefs.keySet()) {
683           locals.put(typedefName, JSType.UNDEFINED);
684         }
685         copyOuterVarsTransitively(this);
686         preservedNamespaces = localNamespaces;
687         localNamespaces = ImmutableMap.of();
688         localTypedefs = ImmutableMap.of();
689         escapedVars = ImmutableSet.of();
690         isFinalized = true;
691       }

693       // A scope must know about the free variables used in outer scopes,
694       // otherwise we end up with invalid type envs.
695       private static void copyOuterVarsTransitively(NTIScope s) {
696         if (s.isTopLevel()) {
697           return;
698         }
699         NTIScope parent = s.parent;
700         Set<String> outerVars = s.outerVars;
701         while (parent.isFunction()) {
702           boolean copiedOneVar = false;
703           for (String v : outerVars) {
704             if (!parent.isDefinedLocally(v, false)) {
705               copiedOneVar = true;
706               parent.addOuterVar(v);
707             }
708           }
709           if (!copiedOneVar) {
710             break;
711           }
712           outerVars = parent.outerVars;
713           parent = parent.parent;
714         }
715       }

717       @Override
718       public String toString() {
719         StringBuilder sb = new StringBuilder();
720         if (isTopLevel()) {
721           sb.append("<TOP SCOPE>");
722         } else {
723           sb.append(getReadableName());
724           sb.append('(');
725           Joiner.on(',').appendTo(sb, formals);
726           sb.append(')');
727         }
728         sb.append(" with root: ");
729         sb.append(root);
730         return sb.toString();
731       }
