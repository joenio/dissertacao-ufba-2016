244       /** If the arity isn't always the same, this function returns -1 */
245       public static int arity(Token token) {
246         switch (token) {
247           case RETURN:
248           case NEW:
249           case CALL:
250           case STRING_KEY:
251           case REGEXP:
252           case ARRAYLIT:
253           case OBJECTLIT:
254           case TEMPLATELIT:
255           case TRY:
256           case PARAM_LIST:
257           case IF:
258           case SWITCH:
259           case FOR:
260           case BREAK:
261           case CONTINUE:
262           case VAR:
263           case BLOCK:
264           case SCRIPT:
265           case CONST:
266           case DEBUGGER:
267           case ANNOTATION:
268           case PIPE:
269           case STAR:
270           case EOC:
271           case QMARK:
272           case ELLIPSIS:
273           case BANG:
274           case EQUALS:
275           case LB:
276           case LC:
277           case COLON:
278           case YIELD:
279             return -1;
280           case NAME:
281           case LABEL_NAME:
282           case NUMBER:
283           case STRING:
284           case NULL:
285           case THIS:
286           case FALSE:
287           case TRUE:
288           case EMPTY:
289           case IMPORT_STAR:
290           case MEMBER_VARIABLE_DEF:
291             return 0;
292           case NOT:
293           case BITNOT:
294           case POS:
295           case NEG:
296           case DELPROP:
297           case TYPEOF:
298           case THROW:
299           case TEMPLATELIT_SUB:
300           case MEMBER_FUNCTION_DEF:
301           case INC:
302           case DEC:
303           case DEFAULT_CASE:
304           case EXPR_RESULT:
305           case GETTER_DEF:
306           case SETTER_DEF:
307           case CAST:
308           case SPREAD:
309           case VOID:
310           case NAMED_TYPE:
311           case TYPE_ALIAS:
312           case INDEX_SIGNATURE:
313           case CALL_SIGNATURE:
314           case REST:
315             return 1;
316           case BITOR:
317           case BITXOR:
318           case BITAND:
319           case EQ:
320           case NE:
321           case LT:
322           case LE:
323           case GT:
324           case GE:
325           case LSH:
326           case RSH:
327           case URSH:
328           case ADD:
329           case SUB:
330           case MUL:
331           case DIV:
332           case MOD:
333           case GETPROP:
334           case GETELEM:
335           case SHEQ:
336           case SHNE:
337           case IN:
338           case INSTANCEOF:
339           case TAGGED_TEMPLATELIT:
340           case DEFAULT_VALUE:
341           case COMMA:
342           case ASSIGN:
343           case ASSIGN_BITOR:
344           case ASSIGN_BITXOR:
345           case ASSIGN_BITAND:
346           case ASSIGN_LSH:
347           case ASSIGN_RSH:
348           case ASSIGN_URSH:
349           case ASSIGN_ADD:
350           case ASSIGN_SUB:
351           case ASSIGN_MUL:
352           case ASSIGN_DIV:
353           case ASSIGN_MOD:
354           case OR:
355           case AND:
356           case CASE:
357           case WHILE:
358           case DO:
359           case WITH:
360           case CATCH:
361           case LABEL:
362           case COMPUTED_PROP:
363           case ENUM:
364           case NAMESPACE:
365             return 2;
366           case CLASS:
367           case HOOK:
368           case FUNCTION:
369           case FOR_OF:
370           case IMPORT:
371           case INTERFACE:
372             return 3;
373           default:
374             throw new IllegalStateException("No arity defined for " + token);
375         }
376       }
