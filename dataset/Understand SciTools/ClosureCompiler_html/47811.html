30     /**
31      * Compiler pass for AngularJS-specific needs. Generates {@code $inject} \
32      * properties for functions (class constructors, wrappers, etc) annotated with
33      * @ngInject. Without this pass, AngularJS will not work properly if variable
34      * renaming is enabled, because the function arguments will be renamed.
35      * @see http://docs.angularjs.org/tutorial/step_05#a-note-on-minification
36      *
37      * <p>For example, the following code:</p>
38      * <pre><code>
39      *
40      * /** @ngInject * /
41      * function Controller(dependency1, dependency2) {
42      *   // do something
43      * }
44      *
45      * </code></pre>
46      *
47      * <p>will be transformed into:
48      * <pre><code>
49      *
50      * function Controller(dependency1, dependency2) {
51      *   // do something
52      * }
53      * Controller.$inject = ['dependency1', 'dependency2'];
54      *
55      * </code></pre>
56      *
57      * <p> This pass also supports assignments of function expressions to variables
58      * like:
59      * <pre><code>
60      *
61      * /** @ngInject * /
62      * var filter = function(a, b) {};
63      *
64      * var ns = {};
65      * /** @ngInject * /
66      * ns.method = function(a,b,c) {};
67      *
68      * /** @ngInject * /
69      * var shorthand = ns.method2 = function(a,b,c,) {}
70      *
71      * </code></pre>
72      */
73     class AngularPass extends AbstractPostOrderCallback
74         implements HotSwapCompilerPass {
75       final AbstractCompiler compiler;
76     
77       /** Nodes annotated with @ngInject */
78       private final List<NodeContext> injectables = new ArrayList<>();
79     
80       public AngularPass(AbstractCompiler compiler) {
81         this.compiler = compiler;
82       }
83     
84       public static final String INJECT_PROPERTY_NAME = "$inject";
85     
86       static final DiagnosticType INJECT_IN_NON_GLOBAL_OR_BLOCK_ERROR =
87           DiagnosticType.error("JSC_INJECT_IN_NON_GLOBAL_OR_BLOCK_ERROR",
88               "@ngInject only applies to functions defined in blocks or " +
89               "global scope.");
90     
91       static final DiagnosticType INJECT_NON_FUNCTION_ERROR =
92           DiagnosticType.error("JSC_INJECT_NON_FUNCTION_ERROR",
93               "@ngInject can only be used when defining a function or " +
94               "assigning a function expression.");
95     
96       static final DiagnosticType INJECTED_FUNCTION_HAS_DESTRUCTURED_PARAM =
97           DiagnosticType.error("JSC_INJECTED_FUNCTION_HAS_DESTRUCTURED_PARAM",
98               "@ngInject cannot be used on functions containing "
99               + "destructured parameter.");
100     
101       static final DiagnosticType INJECTED_FUNCTION_HAS_DEFAULT_VALUE =
102           DiagnosticType.error("JSC_INJECTED_FUNCTION_HAS_DEFAULT_VALUE",
103               "@ngInject cannot be used on functions containing default value.");
104     
105       static final DiagnosticType INJECTED_FUNCTION_ON_NON_QNAME =
106           DiagnosticType.error("JSC_INJECTED_FUNCTION_ON_NON_QNAME",
107               "@ngInject can only be used on qualified names.");
108     
109       @Override
110       public void process(Node externs, Node root) {
111         hotSwapScript(root, null);
112       }

114       @Override
115       public void hotSwapScript(Node scriptRoot, Node originalRoot) {
116         // Traverses AST looking for nodes annotated with @ngInject.
117         NodeTraversal.traverseEs6(compiler, scriptRoot, this);
118         boolean codeChanged = false;
119         // iterates through annotated nodes adding $inject property to elements.
120         for (NodeContext entry : injectables) {
121           String name = entry.getName();
122           Node fn = entry.getFunctionNode();
123           List<Node> dependencies = createDependenciesList(fn);
124           // skips entry if it does not have any dependencies.
125           if (dependencies.isEmpty()) {
126             continue;
127           }
128           Node dependenciesArray = IR.arraylit(dependencies.toArray(new Node[0]));
129           // creates `something.$inject = ['param1', 'param2']` node.
130           Node statement = IR.exprResult(
131               IR.assign(
132                   IR.getelem(
133                       NodeUtil.newQName(compiler, name),
134                       IR.string(INJECT_PROPERTY_NAME)),
135                   dependenciesArray
136               )
137           );
138           NodeUtil.setDebugInformation(statement, entry.getNode(), name);
139           // Set the visibility of the newly created property.
140           JSDocInfoBuilder newPropertyDoc = new JSDocInfoBuilder(false);
141           newPropertyDoc.recordVisibility(Visibility.PUBLIC);
142           statement.getFirstChild().setJSDocInfo(newPropertyDoc.build());
143     
144           // adds `something.$inject = [...]` node after the annotated node or the following
145           // goog.inherits call.
146           Node insertionPoint = entry.getTarget();
147           Node next = insertionPoint.getNext();
148           while (next != null
149                  && NodeUtil.isExprCall(next)
150                  && compiler.getCodingConvention().getClassesDefinedByCall(
151                      next.getFirstChild()) != null) {
152             insertionPoint = next;
153             next = insertionPoint.getNext();
154           }
155     
156           insertionPoint.getParent().addChildAfter(statement, insertionPoint);
157           codeChanged = true;
158         }
159         if (codeChanged) {
160           compiler.reportCodeChange();
161         }
162       }

164       /**
165        * Given a FUNCTION node returns array of STRING nodes representing function
166        * parameters.
167        * @param n the FUNCTION node.
168        * @return STRING nodes.
169        */
170       private List<Node> createDependenciesList(Node n) {
171         Preconditions.checkArgument(n.isFunction());
172         Node params = NodeUtil.getFunctionParameters(n);
173         if (params != null) {
174           return createStringsFromParamList(params);
175         }
176         return new ArrayList<>();
177       }
178     
179       /**
180        * Given a PARAM_LIST node creates an array of corresponding STRING nodes.
181        * @param params PARAM_LIST node.
182        * @return array of STRING nodes.
183        */
184       private List<Node> createStringsFromParamList(Node params) {
185         Node param = params.getFirstChild();
186         ArrayList<Node> names = new ArrayList<>();
187         while (param != null) {
188           if (param.isName()) {
189             names.add(IR.string(param.getString()).srcref(param));
190           } else if (param.isDestructuringPattern()) {
191             compiler.report(JSError.make(param,
192                 INJECTED_FUNCTION_HAS_DESTRUCTURED_PARAM));
193             return new ArrayList<>();
194           } else if (param.isDefaultValue()) {
195             compiler.report(JSError.make(param,
196                 INJECTED_FUNCTION_HAS_DEFAULT_VALUE));
197             return new ArrayList<>();
198           }
199           param = param.getNext();
200         }
201         return names;
202       }

204       @Override
205       public void visit(NodeTraversal t, Node n, Node parent) {
206         JSDocInfo docInfo = n.getJSDocInfo();
207         if (docInfo != null && docInfo.isNgInject()) {
208           addNode(n, t);
209         }
210       }
211     
212       /**
213        * Add node to the list of injectables.
214        * @param n node to add.
215        * @param t node traversal instance.
216        */
217       private void addNode(Node n, NodeTraversal t) {
218         Node target = null;
219         Node fn = null;
220         String name = null;
221     
222         switch (n.getType()) {
223           // handles assignment cases like:
224           // a = function() {}
225           // a = b = c = function() {}
226           case ASSIGN:
227             name = n.getFirstChild().getQualifiedName();
228             if (name == null) {
229               compiler.report(t.makeError(n, INJECTED_FUNCTION_ON_NON_QNAME));
230               return;
231             }
232             // last node of chained assignment.
233             fn = n;
234             while (fn.isAssign()) {
235               fn = fn.getLastChild();
236             }
237             target = n.getParent();
238             break;
239     
240           // handles function case:
241           // function fnName() {}
242           case FUNCTION:
243             name = NodeUtil.getName(n);
244             fn = n;
245             target = n;
246             if (n.getParent().isAssign()
247                 && n.getParent().getJSDocInfo().isNgInject()) {
248               // This is a function assigned into a symbol, e.g. a regular function
249               // declaration in a goog.module or goog.scope.
250               // Skip in this traversal, it is handled when visiting the assign.
251               return;
252             }
253             break;
254     
255           // handles var declaration cases like:
256           // var a = function() {}
257           // var a = b = function() {}
258           case VAR:
259           case LET:
260           case CONST:
261             name = n.getFirstChild().getString();
262             // looks for a function node.
263             fn = getDeclarationRValue(n);
264             target = n;
265             break;
266     
267           // handles class method case:
268           // class clName(){
269           //   constructor(){}
270           //   someMethod(){} <===
271           // }
272           case MEMBER_FUNCTION_DEF:
273             Node parent = n.getParent();
274             if (parent.isClassMembers()){
275               Node classNode = parent.getParent();
276               String midPart = n.isStaticMember() ? "." : ".prototype.";
277               name = NodeUtil.getName(classNode) + midPart + n.getString();
278               if (n.getString().equals("constructor")) {
279                 name = NodeUtil.getName(classNode);
280               }
281               fn = n.getFirstChild();
282               if (classNode.getParent().isAssign() || classNode.getParent().isName()) {
283                 target = classNode.getGrandparent();
284               } else {
285                 target = classNode;
286               }
287             }
288             break;
289         }
290     
291         if (fn == null || !fn.isFunction()) {
292           compiler.report(t.makeError(n, INJECT_NON_FUNCTION_ERROR));
293           return;
294         }
295         // report an error if the function declaration did not take place in a block or global scope
296         if (!target.getParent().isScript() && !target.getParent().isBlock()) {
297           compiler.report(t.makeError(n, INJECT_IN_NON_GLOBAL_OR_BLOCK_ERROR));
298           return;
299         }
300         // checks that name is present, which must always be the case unless the
301         // compiler allowed a syntax error or a dangling anonymous function
302         // expression.
303         Preconditions.checkNotNull(name);
304         // registers the node.
305         injectables.add(new NodeContext(name, n, fn, target));
306       }

308       /**
309        * Given a VAR node (variable declaration) returns the node of initial value.
310        *
311        * <pre><code>
312        * var x;  // null
313        * var y = "value"; // STRING "value" node
314        * var z = x = y = function() {}; // FUNCTION node
315        * <code></pre>
316        * @param n VAR node.
317        * @return the assigned initial value, or the rightmost rvalue of an assignment
318        * chain, or null.
319        */
320       private static Node getDeclarationRValue(Node n) {
321         Preconditions.checkNotNull(n);
322         Preconditions.checkArgument(NodeUtil.isNameDeclaration(n));
323         n = n.getFirstFirstChild();
324         if (n == null) {
325           return null;
326         }
327         while (n.isAssign()) {
328           n = n.getLastChild();
329         }
330         return n;
331       }

333       static class NodeContext {
334         /** Name of the function/object. */
335         private final String name;
336         /** Node jsDoc is attached to. */
337         private final Node node;
338         /** Function node */
339         private final Node functionNode;
340         /** Node after which to inject the new code */
341         private final Node target;
342     
343         public NodeContext(String name, Node node, Node functionNode, Node target) {
344           this.name = name;
345           this.node = node;
346           this.functionNode = functionNode;
347           this.target = target;
348         }
349     
350         /**
351          * @return the name.
352          */
353         public String getName() {
354           return name;
355         }
356     
357         /**
358          * @return the node.
359          */
360         public Node getNode() {
361           return node;
362         }
363     
364         /**
365          * @return the context.
366          */
367         public Node getFunctionNode() {
368           return functionNode;
369         }
370     
371         /**
372          * @return the context.
373          */
374         public Node getTarget() {
375           return target;
376         }
377       }
