16     package com.google.javascript.jscomp;
17     
18     import com.google.javascript.jscomp.NodeTraversal.AbstractPostOrderCallback;
19     import com.google.javascript.rhino.IR;
20     import com.google.javascript.rhino.Node;
21     
22     /**
23      * An optimization pass to re-write J2CL Equality.$same.
24      */
25     public class J2clEqualitySameRewriterPass extends AbstractPostOrderCallback
26         implements CompilerPass {
27     
28       private final AbstractCompiler compiler;
29     
30       J2clEqualitySameRewriterPass(AbstractCompiler compiler) {
31         this.compiler = compiler;
32       }
33     
34       @Override
35       public void process(Node externs, Node root) {
36         NodeTraversal.traverseEs6(compiler, root, this);
37       }
38     
39       @Override
40       public void visit(NodeTraversal t, Node node, Node parent) {
41         if (isEqualitySameCall(node)) {
42           trySubstituteEqualitySame(node);
43         }
44       }
45     
46       private void trySubstituteEqualitySame(Node callNode) {
47         Node firstExpr = callNode.getSecondChild();
48         Node secondExpr = callNode.getLastChild();
49         if (!NodeUtil.isLiteralValue(firstExpr, true) && !NodeUtil.isLiteralValue(secondExpr, true)) {
50           return;
51         }
52     
53         // At least one is literal value. So we can replace w/ a simpler form.
54         firstExpr.detachFromParent();
55         secondExpr.detachFromParent();
56         Node replacement = asEqOperation(firstExpr, secondExpr);
57         callNode.getParent().replaceChild(callNode, replacement.useSourceInfoIfMissingFrom(callNode));
58         compiler.reportCodeChange();
59       }
60     
61       private Node asEqOperation(Node firstExpr, Node secondExpr) {
62         return (NodeUtil.isNullOrUndefined(firstExpr) || NodeUtil.isNullOrUndefined(secondExpr))
63             ? IR.eq(firstExpr, secondExpr)
64             : IR.sheq(firstExpr, secondExpr);
65       }
66     
67       private static boolean isEqualitySameCall(Node node) {
68         return node.isCall() && isEqualitySameMethodName(node.getFirstChild().getQualifiedName());
69       }
70     
71       private static boolean isEqualitySameMethodName(String fnName) {
72         // The '.$same' case only happens when collapseProperties is off.
73         return fnName != null
74             && (fnName.endsWith("Equality$$0same") || fnName.endsWith("Equality.$same"));
75       }
76     }
