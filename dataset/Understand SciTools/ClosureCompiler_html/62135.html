17     package com.google.javascript.jscomp;
18     
19     import com.google.javascript.jscomp.CompilerOptions.LanguageMode;
20     
21     /**
22      * @author johnlenz@google.com (John Lenz)
23      */
24     public final class CheckRegExpTest extends CompilerTestCase {
25       CheckRegExp last = null;
26     
27       public CheckRegExpTest() {
28         super("var RegExp;", true);
29       }
30     
31       @Override
32       protected CompilerPass getProcessor(Compiler compiler) {
33         compiler.options.setWarningLevel(
34             DiagnosticGroups.CHECK_REGEXP, CheckLevel.WARNING);
35         last = new CheckRegExp(compiler);
36         return last;
37       }
38     
39       private void testReference(String code, boolean expected) {
40         testSame(code, (expected) ? CheckRegExp.REGEXP_REFERENCE : null);
41         assertEquals(expected, last.isGlobalRegExpPropertiesUsed());
42       }
43     
44       public void testRegExp() {
45         // Creating RegExp's is OK.
46         testReference("RegExp();", false);
47         testReference("var x = RegExp();", false);
48         testReference("new RegExp();", false);
49         testReference("var x = new RegExp();", false);
50     
51         // Checking for RegExp instances is OK, as well.
52         testReference("x instanceof RegExp;", false);
53     
54         // Comparing for equality with RegExp is OK.
55         testReference("x === RegExp;", false);
56         testReference("x !== RegExp;", false);
57         testReference("switch (x) { case RegExp: }", false);
58         testReference("x == RegExp;", false);
59         testReference("x != RegExp;", false);
60     
61         // Access to non-magical properties is OK.
62         testReference("RegExp.test();", false);
63         testReference("var x = RegExp.test();", false);
64         testReference("RegExp.exec();", false);
65         testReference("RegExp.foobar;", false);
66     
67         // Magical properties aren't allowed.
68         testReference("RegExp.$1;", true);
69         testReference("RegExp.$_;", true);
70         testReference("RegExp.$input;", true);
71         testReference("RegExp.rightContext;", true);
72         testReference("RegExp.multiline;", true);
73     
74         // Any other reference isn't allowed.
75         testReference("delete RegExp;", true);
76         testReference("RegExp;", true);
77         testReference("if (RegExp);", true);
78         testReference("if (!RegExp);", true);
79     
80         // Aliases aren't allowed.
81         testReference("var x = RegExp;", true);
82         testReference("f(RegExp);", true);
83         testReference("new f(RegExp);", true);
84         testReference("var x = RegExp; x.test()", true);
85         setAcceptedLanguage(LanguageMode.ECMASCRIPT6);
86         testReference("let x = RegExp;", true);
87         testReference("const x = RegExp;", true);
88     
89         // No RegExp reference is OK.
90         testReference("var x;", false);
91     
92         // Local RegExp is OK.
93         testReference("function f() {var RegExp; RegExp.test();}", false);
94         testReference("function f() {let RegExp; RegExp.test();}", false);
95         testReference("function *gen() {var RegExp; yield RegExp.test();}", false);
96     
97         // Property named 'RegExp' is OK.
98         testReference("var x = {RegExp: {}}; x.RegExp.$1;", false);
99     
100         // Class property is also OK.
101         setAcceptedLanguage(LanguageMode.ECMASCRIPT6);
102         testReference(LINE_JOINER.join(
103             "class x {",
104             "  constructor() {this.RegExp = {};}",
105             "  method() {",
106             "    this.RegExp.$1;",
107             "    this.RegExp.test();",
108             "  }",
109             "}"), false);
110       }
111     
112       public void testInvalidRange() {
113         testSame("\"asdf\".match(/[z-a]/)", CheckRegExp.MALFORMED_REGEXP);
114       }
115     }
