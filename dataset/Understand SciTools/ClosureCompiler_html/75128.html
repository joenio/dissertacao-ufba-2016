40     /**
41      * A parser that can extract dependency information from existing deps.js files.
42      *
43      * <p>See //javascript/closure/deps.js for an example file.</p>
44      *
45      * @author agrieve@google.com (Andrew Grieve)
46      */
47     @GwtIncompatible("java.util.regex")
48     public final class DepsFileParser extends JsFileLineParser {
49     
50       private static Logger logger = Logger.getLogger(DepsFileParser.class.getName());
51     
52       /**
53        * Pattern for matching JavaScript string literals. The group is:
54        * goog.addDependency({1});
55        */
56       private final Matcher depMatcher =
57           Pattern.compile("\\s*goog.addDependency\\((.*)\\);?\\s*").matcher("");
58     
59       /**
60        * Pattern for matching the args of a goog.addDependency(). The group is:
61        * goog.addDependency({1}, {2}, {3}, {4?});
62        */
63       private final Matcher depArgsMatch =
64           Pattern.compile(
65                   "\\s*([^,]*), (\\[[^\\]]*\\]), (\\[[^\\]]*\\])"
66                       + "(?:, (true|false|\\{[^{}]*\\}))?\\s*")
67               .matcher("");
68     
69       /**
70        * The dependency information extracted from the current file.
71        */
72       private List<DependencyInfo> depInfos;
73     
74       /** Translates paths in different build systems. */
75       private final Function<String, String> pathTranslator;
76     
77       /**
78        * Constructor
79        *
80        * @param errorManager Handles parse errors.
81        */
82       public DepsFileParser(ErrorManager errorManager) {
83         this(Functions.<String>identity(), errorManager);
84       }
85     
86       /**
87        * @param pathTranslator Translates paths in different build systems.
88        * @param errorManager Handles parse errors.
89        */
90       public DepsFileParser(Function<String, String> pathTranslator,
91           ErrorManager errorManager) {
92         super(errorManager);
93         this.pathTranslator = pathTranslator;
94       }
95     
96       /**
97        * Parses the given file and returns a list of dependency information that it
98        * contained.
99        *
100        * @param filePath Path to the file to parse.
101        * @return A list of DependencyInfo objects.
102        * @throws IOException Thrown if the file could not be read.
103        */
104       public List<DependencyInfo> parseFile(String filePath) throws IOException {
105         return parseFileReader(filePath, Files.newReader(new File(filePath), StandardCharsets.UTF_8));
106       }
107     
108       /**
109        * Parses the given file and returns a list of dependency information that it
110        * contained.
111        * It uses the passed in fileContents instead of reading the file.
112        *
113        * @param filePath Path to the file to parse.
114        * @param fileContents The contents to parse.
115        * @return A list of DependencyInfo objects.
116        */
117       public List<DependencyInfo> parseFile(String filePath, String fileContents) {
118         return parseFileReader(filePath, new StringReader(fileContents));
119       }
120     
121     
122       /**
123        * Parses the file from the given reader and returns a list of
124        * dependency information that it contained.
125        *
126        * @param filePath Path to the file to parse.
127        * @param reader A reader for the file.
128        * @return A list of DependencyInfo objects.
129        */
130       public List<DependencyInfo> parseFileReader(String filePath, Reader reader) {
131         depInfos = new ArrayList<>();
132         logger.fine("Parsing Dep: " + filePath);
133         doParse(filePath, reader);
134         return depInfos;
135       }
136     
137       /**
138        * Extracts dependency information from lines that look like
139        *   goog.addDependency('pathRelativeToClosure', ['provides'], ['requires']);
140        * Adds the dependencies to depInfos.
141        *
142        * @throws ParseException Thrown if the given line has a malformed
143        *     goog.addDependency().
144        */
145       @Override
146       protected boolean parseLine(String line) throws ParseException {
147         boolean hasDependencies = false;
148     
149         // Quick sanity check that will catch most cases. This is a performance
150         // win for people with a lot of JS.
151         if (line.contains("addDependency")) {
152           depMatcher.reset(line);
153           // See if the line looks like: goog.addDependency(...)
154           if (depMatcher.matches()) {
155             hasDependencies = true;
156             String addDependencyParams = depMatcher.group(1);
157             depArgsMatch.reset(addDependencyParams);
158             // Extract the three parameters.
159             if (!depArgsMatch.matches()) {
160               // Although we could recover, we mark this as fatal since there should
161               // not be problems with generated deps.js files.
162               throw new ParseException("Invalid arguments to goog.addDependency(). Found: "
163                   + addDependencyParams, true);
164             }
165             // Parse the file path.
166             String path = pathTranslator.apply(parseJsString(depArgsMatch.group(1)));
167     
168             DependencyInfo depInfo = new SimpleDependencyInfo(path, filePath,
169                 // Parse the provides.
170                 parseJsStringArray(depArgsMatch.group(2)),
171                 // Parse the requires.
172                 parseJsStringArray(depArgsMatch.group(3)),
173                 // Parse the loadFlags map.
174                 parseLoadFlags(depArgsMatch.group(4))
175                 );
176     
177             if (logger.isLoggable(Level.FINE)) {
178               logger.fine("Found dep: " + depInfo);
179             }
180             depInfos.add(depInfo);
181           }
182         }
183     
184         return !shortcutMode || hasDependencies ||
185             CharMatcher.whitespace().matchesAllOf(line);
186       }
187     
188       private Map<String, String> parseLoadFlags(String loadFlags) throws ParseException {
189         if (loadFlags == null || loadFlags.equals("false")) {
190           return ImmutableMap.of();
191         } else if (loadFlags.equals("true")) {
192           return ImmutableMap.of("module", "goog");
193         } else {
194           return parseJsStringMap(loadFlags);
195         }
196       }
197     }
