16     package com.google.javascript.jscomp;

18     import com.google.common.base.Preconditions;
19     import com.google.javascript.rhino.Node;

21     /**
22      * Tries to chain assignments together.
23      *
24      * @author nicksantos@google.com (Nick Santos)
25      *
26      */
27     class ExploitAssigns extends AbstractPeepholeOptimization {
28     
29       @Override
30       Node optimizeSubtree(Node subtree) {
31         for (Node child = subtree.getFirstChild(); child != null;) {
32           Node next = child.getNext();
33           if (NodeUtil.isExprAssign(child)) {
34             collapseAssign(child.getFirstChild(), child, subtree);
35           }
36           child = next;
37         }
38         return subtree;
39       }
40     
41       /**
42        * Try to collapse the given assign into subsequent expressions.
43        */
44       private void collapseAssign(Node assign, Node expr,
45           Node exprParent) {
46         Node leftValue = assign.getFirstChild();
47         Node rightValue = leftValue.getNext();
48         if (isCollapsibleValue(leftValue, true) &&
49             collapseAssignEqualTo(expr, exprParent, leftValue)) {
50           reportCodeChange();
51         } else if (isCollapsibleValue(rightValue, false) &&
52             collapseAssignEqualTo(expr, exprParent, rightValue)) {
53           reportCodeChange();
54         } else if (rightValue.isAssign()) {
55           // Recursively deal with nested assigns.
56           collapseAssign(rightValue, expr, exprParent);
57         }
58       }
59     
60       /**
61        * Determines whether we know enough about the given value to be able
62        * to collapse it into subsequent expressions.
63        *
64        * For example, we can collapse booleans and variable names:
65        * <code>
66        * x = 3; y = x; // y = x = 3;
67        * a = true; b = true; // b = a = true;
68        * <code>
69        * But we won't try to collapse complex expressions.
70        *
71        * @param value The value node.
72        * @param isLValue Whether it's on the left-hand side of an expr.
73        */
74       private static boolean isCollapsibleValue(Node value, boolean isLValue) {
75         switch (value.getType()) {
76           case GETPROP:
77             // Do not collapse GETPROPs on arbitrary objects, because
78             // they may be implemented setter functions, and oftentimes
79             // setter functions fail on native objects. This is OK for "THIS"
80             // objects, because we assume that they are non-native.
81             return !isLValue || value.getFirstChild().isThis();
82     
83           case NAME:
84             return true;
85     
86           default:
87             return NodeUtil.isImmutableValue(value);
88         }
89       }
90     
91       /**
92        * Collapse the given assign expression into the expression directly
93        * following it, if possible.
94        *
95        * @param expr The expression that may be moved.
96        * @param exprParent The parent of {@code expr}.
97        * @param value The value of this expression, expressed as a node. Each
98        *     expression may have multiple values, so this function may be called
99        *     multiple times for the same expression. For example,
100        *     <code>
101        *     a = true;
102        *     </code>
103        *     is equal to the name "a" and the boolean "true".
104        * @return Whether the expression was collapsed successfully.
105        */
106       private boolean collapseAssignEqualTo(Node expr, Node exprParent,
107           Node value) {
108         Node assign = expr.getFirstChild();
109         Node parent = exprParent;
110         Node next = expr.getNext();
111         while (next != null) {
112           switch (next.getType()) {
113             case AND:
114             case OR:
115             case HOOK:
116             case IF:
117             case RETURN:
118             case EXPR_RESULT:
119               // Dive down the left side
120               parent = next;
121               next = next.getFirstChild();
122               break;
123     
124             case VAR:
125               if (next.getFirstChild().hasChildren()) {
126                 parent = next.getFirstChild();
127                 next = parent.getFirstChild();
128                 break;
129               }
130               return false;
131     
132             case GETPROP:
133             case NAME:
134               if (next.isQualifiedName()) {
135                 if (value.isQualifiedName() &&
136                     next.matchesQualifiedName(value)) {
137                   // If the previous expression evaluates to value of a
138                   // qualified name, and that qualified name is used again
139                   // shortly, then we can exploit the assign here.
140     
141                   // Verify the assignment doesn't change its own value.
142                   if (!isSafeReplacement(next, assign)) {
143                     return false;
144                   }
145     
146                   exprParent.removeChild(expr);
147                   expr.removeChild(assign);
148                   parent.replaceChild(next, assign);
149                   return true;
150                 }
151               }
152               return false;
153     
154             case ASSIGN:
155               // Assigns are really tricky. In lots of cases, we want to inline
156               // into the right side of the assign. But the left side of the
157               // assign is evaluated first, and it may have convoluted logic:
158               //   a = null;
159               //   (a = b).c = null;
160               // We don't want to exploit the first assign. Similarly:
161               //   a.b = null;
162               //   a.b.c = null;
163               // We don't want to exploit the first assign either.
164               //
165               // To protect against this, we simply only inline when the left side
166               // is guaranteed to evaluate to the same L-value no matter what.
167               Node leftSide = next.getFirstChild();
168               if (leftSide.isName() ||
169                   leftSide.isGetProp() &&
170                   leftSide.getFirstChild().isThis()) {
171                 // Dive down the right side of the assign.
172                 parent = next;
173                 next = leftSide.getNext();
174                 break;
175               } else {
176                 return false;
177               }
178     
179             default:
180               if (NodeUtil.isImmutableValue(next)
181                   && next.isEquivalentTo(value)) {
182                 // If the r-value of the expr assign is an immutable value,
183                 // and the value is used again shortly, then we can exploit
184                 // the assign here.
185                 exprParent.removeChild(expr);
186                 expr.removeChild(assign);
187                 parent.replaceChild(next, assign);
188                 return true;
189               }
190               // Return without inlining a thing
191               return false;
192           }
193         }
194     
195         return false;
196       }

198       /**
199        * Checks name referenced in node to determine if it might have
200        * changed.
201        * @return Whether the replacement can be made.
202        */
203       private boolean isSafeReplacement(Node node, Node replacement) {
204         // No checks are needed for simple names.
205         if (node.isName()) {
206           return true;
207         }
208         Preconditions.checkArgument(node.isGetProp());
209     
210         while (node.isGetProp()) {
211           node = node.getFirstChild();
212         }
213         return !(node.isName() && isNameAssignedTo(node.getString(), replacement));
214     
215       }
216     
217       /**
218        * @return Whether name is assigned in the expression rooted at node.
219        */
220     
221       private static boolean isNameAssignedTo(String name, Node node) {
222         for (Node c = node.getFirstChild(); c != null; c = c.getNext()) {
223           if (isNameAssignedTo(name, c)) {
224             return true;
225           }
226         }
227     
228         if (node.isName()) {
229           Node parent = node.getParent();
230           if (parent.isAssign() && parent.getFirstChild() == node) {
231             if (name.equals(node.getString())) {
232               return true;
233             }
234           }
235         }
236     
237         return false;
238       }
