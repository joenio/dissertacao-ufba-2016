29     /**
30      * A peephole optimization that minimizes code by simplifying conditional
31      * expressions, replacing IFs with HOOKs, replacing object constructors
32      * with literals, and simplifying returns.
33      *
34      */

51       /**
52        * @param late When late is false, this mean we are currently running before
53        * most of the other optimizations. In this case we would avoid optimizations
54        * that would make the code harder to analyze (such as using string splitting,
55        * merging statements with commas, etc). When this is true, we would
56        * do anything to minimize for size.
57        */
58       PeepholeSubstituteAlternateSyntax(boolean late) {
59         this.late = late;

62       /**
63        * Tries apply our various peephole minimizations on the passed in node.
64        */
65       @Override
66       @SuppressWarnings("fallthrough")
67       public Node optimizeSubtree(Node node) {
68         switch(node.getType()) {
69           case ASSIGN_SUB:
70             return reduceSubstractionAssignment(node);
71     
72           case TRUE:
73           case FALSE:
74             return reduceTrueFalse(node);
75     
76           case NEW:
77             node = tryFoldStandardConstructors(node);
78             if (!node.isCall()) {
79               return node;
80             }
81             // Fall through on purpose because tryFoldStandardConstructors() may
82             // convert a NEW node into a CALL node
83           case CALL:
84             Node result =  tryFoldLiteralConstructor(node);
85             if (result == node) {
86               result = tryFoldSimpleFunctionCall(node);
87               if (result == node) {
88                 result = tryFoldImmediateCallToBoundFunction(node);
89               }
90             }
91             return result;
92     
93           case RETURN:
94             return tryReduceReturn(node);
95     
96           case COMMA:
97             return trySplitComma(node);
98     
99           case NAME:
100             return tryReplaceUndefined(node);
101     
102           case ARRAYLIT:
103             return tryMinimizeArrayLiteral(node);
104     
105           case GETPROP:
106             return tryMinimizeWindowRefs(node);
107     
108           case MUL:
109           case AND:
110           case OR:
111           case BITOR:
112           case BITXOR:
113           case BITAND:
114             return tryRotateAssociativeOperator(node);
115     
116           default:
117             return node; //Nothing changed
118         }
119       }

122           "Object",
123           "Array",
124           "Error",
125           "RegExp",

128       private Node tryMinimizeWindowRefs(Node node) {
129         // Normalization needs to be done to ensure there's no shadowing. The window prefix is also
130         // required if the global externs are not on the window.
131         if (!isASTNormalized() || !areDeclaredGlobalExternsOnWindow()) {
132           return node;
133         }
134     
135         Preconditions.checkArgument(node.isGetProp());
136     
137         if (node.getFirstChild().isName()) {
138           Node nameNode = node.getFirstChild();
139           Node stringNode = node.getLastChild();
140     
141           // Since normalization has run we know we're referring to the global window.
142           if ("window".equals(nameNode.getString())
143               && BUILTIN_EXTERNS.contains(stringNode.getString())) {
144             Node newNameNode = IR.name(stringNode.getString());
145             Node parentNode = node.getParent();
146     
147             newNameNode.useSourceInfoFrom(stringNode);
148             parentNode.replaceChild(node, newNameNode);
149     
150             if (parentNode.isCall()) {
151               parentNode.putBooleanProp(Node.FREE_CALL, true);
152             }
153             reportCodeChange();
154             return newNameNode;
155           }
156         }
157     
158         return node;
159       }

161       private Node tryRotateAssociativeOperator(Node n) {
162         if (!late) {
163           return n;
164         }
165         // All commutative operators are also associative
166         Preconditions.checkArgument(NodeUtil.isAssociative(n.getType()));
167         Node rhs = n.getLastChild();
168         if (n.getType() == rhs.getType()) {
169           // Transform a * (b * c) to a * b * c
170           Node first = n.getFirstChild().detachFromParent();
171           Node second = rhs.getFirstChild().detachFromParent();
172           Node third = rhs.getLastChild().detachFromParent();
173           Node newLhs = new Node(n.getType(), first, second)
174               .useSourceInfoIfMissingFrom(n);
175           Node newRoot = new Node(rhs.getType(), newLhs, third)
176               .useSourceInfoIfMissingFrom(rhs);
177           n.getParent().replaceChild(n, newRoot);
178           reportCodeChange();
179           return newRoot;
180         } else if (NodeUtil.isCommutative(n.getType()) &&
181                    !NodeUtil.mayHaveSideEffects(n)) {
182           // Transform a * (b / c) to b / c * a
183           Node lhs = n.getFirstChild();
184           while (lhs.getType() == n.getType()) {
185             lhs = lhs.getFirstChild();
186           }
187           int precedence = NodeUtil.precedence(n.getType());
188           int lhsPrecedence = NodeUtil.precedence(lhs.getType());
189           int rhsPrecedence = NodeUtil.precedence(rhs.getType());
190           if (rhsPrecedence == precedence && lhsPrecedence != precedence) {
191             n.removeChild(rhs);
192             lhs.getParent().replaceChild(lhs, rhs);
193             n.addChildToBack(lhs);
194             reportCodeChange();
195             return n;
196           }
197         }
198         return n;
199       }

201       private Node tryFoldSimpleFunctionCall(Node n) {
202         Preconditions.checkState(n.isCall(), n);
203         Node callTarget = n.getFirstChild();
204         if (callTarget == null || !callTarget.isName()) {
205           return n;
206         }
207         String targetName = callTarget.getString();
208         switch (targetName) {
209           case "Boolean": {
210             // Fold Boolean(a) to !!a
211             // http://www.ecma-international.org/ecma-262/6.0/index.html#sec-boolean-constructor-boolean-value
212             // and
213             // http://www.ecma-international.org/ecma-262/6.0/index.html#sec-logical-not-operator-runtime-semantics-evaluation
214             int paramCount = n.getChildCount() - 1;
215             // only handle the single known parameter case
216             if (paramCount == 1) {
217               Node value = n.getLastChild().detachFromParent();
218               Node replacement;
219               if (NodeUtil.isBooleanResult(value)) {
220                 // If it is already a boolean do nothing.
221                 replacement = value;
222               } else {
223                 // Replace it with a "!!value"
224                 replacement = IR.not(IR.not(value).srcref(n));
225               }
226               n.getParent().replaceChild(n, replacement);
227               reportCodeChange();
228             }
229             break;
230           }
231     
232           case "String": {
233             // Fold String(a) to '' + (a) on immutable literals,
234             // which allows further optimizations
235             //
236             // We can't do this in the general case, because String(a) has
237             // slightly different semantics than '' + (a). See
238             // http://code.google.com/p/closure-compiler/issues/detail?id=759
239             Node value = callTarget.getNext();
240             if (value != null && value.getNext() == null &&
241                 NodeUtil.isImmutableValue(value)) {
242               Node addition = IR.add(
243                   IR.string("").srcref(callTarget),
244                   value.detachFromParent());
245               n.getParent().replaceChild(n, addition);
246               reportCodeChange();
247               return addition;
248             }
249             break;
250           }
251     
252           default:
253             // nothing.
254             break;
255         }
256         return n;
257       }

259       private Node tryFoldImmediateCallToBoundFunction(Node n) {
260         // Rewriting "(fn.bind(a,b))()" to "fn.call(a,b)" makes it inlinable
261         Preconditions.checkState(n.isCall());
262         Node callTarget = n.getFirstChild();
263         Bind bind = getCodingConvention()
264             .describeFunctionBind(callTarget, false, false);
265         if (bind != null) {
266           // replace the call target
267           bind.target.detachFromParent();
268           n.replaceChild(callTarget, bind.target);
269           callTarget = bind.target;
270     
271           // push the parameters
272           addParameterAfter(bind.parameters, callTarget);
273     
274           // add the this value before the parameters if necessary
275           if (bind.thisValue != null && !NodeUtil.isUndefined(bind.thisValue)) {
276             // rewrite from "fn(a, b)" to "fn.call(thisValue, a, b)"
277             Node newCallTarget = IR.getprop(
278                 callTarget.cloneTree(),
279                 IR.string("call").srcref(callTarget));
280             n.replaceChild(callTarget, newCallTarget);
281             n.addChildAfter(bind.thisValue.cloneTree(), newCallTarget);
282             n.putBooleanProp(Node.FREE_CALL, false);
283           } else {
284             n.putBooleanProp(Node.FREE_CALL, true);
285           }
286           reportCodeChange();
287         }
288         return n;
289       }

291       private static void addParameterAfter(Node parameterList, Node after) {
292         if (parameterList != null) {
293           // push the last parameter to the head of the list first.
294           addParameterAfter(parameterList.getNext(), after);
295           after.getParent().addChildAfter(parameterList.cloneTree(), after);
296         }
297       }

299       private Node trySplitComma(Node n) {
300         if (late) {
301           return n;
302         }
303         Node parent = n.getParent();
304         Node left = n.getFirstChild();
305         Node right = n.getLastChild();
306     
307         if (parent.isExprResult()
308             && !parent.getParent().isLabel()) {
309           // split comma
310           n.detachChildren();
311           // Replace the original expression with the left operand.
312           parent.replaceChild(n, left);
313           // Add the right expression afterward.
314           Node newStatement = IR.exprResult(right);
315           newStatement.useSourceInfoIfMissingFrom(n);
316     
317           //This modifies outside the subtree, which is not
318           //desirable in a peephole optimization.
319           parent.getParent().addChildAfter(newStatement, parent);
320           reportCodeChange();
321           return left;
322         } else {
323           return n;
324         }
325       }

328        * Use "void 0" in place of "undefined"
329        */
330       private Node tryReplaceUndefined(Node n) {
331         // TODO(johnlenz): consider doing this as a normalization.
332         if (isASTNormalized()
333             && NodeUtil.isUndefined(n)
334             && !NodeUtil.isLValue(n)) {
335           Node replacement = NodeUtil.newUndefinedNode(n);
336           n.getParent().replaceChild(n, replacement);
337           reportCodeChange();
338           return replacement;
339         }
340         return n;
341       }

343       /**
344        * Reduce "return undefined" or "return void 0" to simply "return".
345        *
346        * @return The original node, maybe simplified.
347        */
348       private Node tryReduceReturn(Node n) {
349         Node result = n.getFirstChild();
350     
351         if (result != null) {
352           switch (result.getType()) {
353             case VOID:
354               Node operand = result.getFirstChild();
355               if (!mayHaveSideEffects(operand)) {
356                 n.removeFirstChild();
357                 reportCodeChange();
358               }
359               break;
360             case NAME:
361               String name = result.getString();
362               if (name.equals("undefined")) {
363                 n.removeFirstChild();
364                 reportCodeChange();
365               }
366               break;
367           }
368         }
369     
370         return n;
371       }

373       private static final ImmutableSet<String> STANDARD_OBJECT_CONSTRUCTORS =
374         // String, Number, and Boolean functions return non-object types, whereas
375         // new String, new Number, and new Boolean return object types, so don't
376         // include them here.
377         ImmutableSet.of(
378           "Object",
379           "Array",
380           "Error"
381           );

383       /**
384        * Fold "new Object()" to "Object()".
385        */
386       private Node tryFoldStandardConstructors(Node n) {
387         Preconditions.checkState(n.isNew());
388     
389         if (canFoldStandardConstructors(n)) {
390           n.setType(Token.CALL);
391           n.putBooleanProp(Node.FREE_CALL, true);
392           reportCodeChange();
393         }
394     
395         return n;
396       }

398       /**
399        * @return Whether "new Object()" can be folded to "Object()" on {@code n}.
400        */
401       private boolean canFoldStandardConstructors(Node n) {
402         // If name normalization has been run then we know that
403         // new Object() does in fact refer to what we think it is
404         // and not some custom-defined Object().
405         if (isASTNormalized() && n.getFirstChild().isName()) {
406           String className = n.getFirstChild().getString();
407           if (STANDARD_OBJECT_CONSTRUCTORS.contains(className)) {
408             return true;
409           }
410           if ("RegExp".equals(className)) {
411             // Fold "new RegExp()" to "RegExp()", but only if the argument is a string.
412             // See issue 1260.
413             if (n.getSecondChild() == null || n.getSecondChild().isString()) {
414               return true;
415             }
416           }
417         }
418     
419         return false;
420       }

422       /**
423        * Replaces a new Array, Object, or RegExp node with a literal, unless the
424        * call is to a local constructor function with the same name.
425        */
426       private Node tryFoldLiteralConstructor(Node n) {
427         Preconditions.checkArgument(n.isCall()
428             || n.isNew());
429     
430         Node constructorNameNode = n.getFirstChild();
431     
432         Node newLiteralNode = null;
433     
434         // We require the AST to be normalized to ensure that, say,
435         // Object() really refers to the built-in Object constructor
436         // and not a user-defined constructor with the same name.
437     
438         if (isASTNormalized() && Token.NAME == constructorNameNode.getType()) {
439     
440           String className = constructorNameNode.getString();
441     
442           if ("RegExp".equals(className)) {
443             // "RegExp("boo", "g")" --> /boo/g
444             return tryFoldRegularExpressionConstructor(n);
445           } else {
446             boolean constructorHasArgs = constructorNameNode.getNext() != null;
447     
448             if ("Object".equals(className) && !constructorHasArgs) {
449               // "Object()" --> "{}"
450               newLiteralNode = IR.objectlit();
451             } else if ("Array".equals(className)) {
452               // "Array(arg0, arg1, ...)" --> "[arg0, arg1, ...]"
453               Node arg0 = constructorNameNode.getNext();
454               FoldArrayAction action = isSafeToFoldArrayConstructor(arg0);
455     
456               if (action == FoldArrayAction.SAFE_TO_FOLD_WITH_ARGS ||
457                   action == FoldArrayAction.SAFE_TO_FOLD_WITHOUT_ARGS) {
458                 newLiteralNode = IR.arraylit();
459                 n.removeChildren();
460                 if (action == FoldArrayAction.SAFE_TO_FOLD_WITH_ARGS) {
461                   newLiteralNode.addChildrenToFront(arg0);
462                 }
463               }
464             }
465     
466             if (newLiteralNode != null) {
467               n.getParent().replaceChild(n, newLiteralNode);
468               reportCodeChange();
469               return newLiteralNode;
470             }
471           }
472         }
473         return n;
474       }

476       private static enum FoldArrayAction {
477         NOT_SAFE_TO_FOLD, SAFE_TO_FOLD_WITH_ARGS, SAFE_TO_FOLD_WITHOUT_ARGS}
478     
479       /**
480        * Checks if it is safe to fold Array() constructor into []. It can be
481        * obviously done, if the initial constructor has either no arguments or
482        * at least two. The remaining case may be unsafe since Array(number)
483        * actually reserves memory for an empty array which contains number elements.
484        */
485       private static FoldArrayAction isSafeToFoldArrayConstructor(Node arg) {
486         FoldArrayAction action = FoldArrayAction.NOT_SAFE_TO_FOLD;
487     
488         if (arg == null) {
489           action = FoldArrayAction.SAFE_TO_FOLD_WITHOUT_ARGS;
490         } else if (arg.getNext() != null) {
491           action = FoldArrayAction.SAFE_TO_FOLD_WITH_ARGS;
492         } else {
493           switch (arg.getType()) {
494             case STRING:
495               // "Array('a')" --> "['a']"
496               action = FoldArrayAction.SAFE_TO_FOLD_WITH_ARGS;
497               break;
498             case NUMBER:
499               // "Array(0)" --> "[]"
500               if (arg.getDouble() == 0) {
501                 action = FoldArrayAction.SAFE_TO_FOLD_WITHOUT_ARGS;
502               }
503               break;
504             case ARRAYLIT:
505               // "Array([args])" --> "[[args]]"
506               action = FoldArrayAction.SAFE_TO_FOLD_WITH_ARGS;
507               break;
508             default:
509           }
510         }
511         return action;
512       }

514       private Node tryFoldRegularExpressionConstructor(Node n) {
515         Node parent = n.getParent();
516         Node constructor = n.getFirstChild();
517         Node pattern = constructor.getNext();  // e.g.  ^foobar$
518         Node flags = null != pattern ? pattern.getNext() : null;  // e.g. gi
519     
520         if (null == pattern || (null != flags && null != flags.getNext())) {
521           // too few or too many arguments
522           return n;
523         }
524     
525         if (// is pattern folded
526             pattern.isString()
527             // make sure empty pattern doesn't fold to //
528             && !"".equals(pattern.getString())
529     
530             && (null == flags || flags.isString())
531             // don't escape patterns with Unicode escapes since Safari behaves badly
532             // (read can't parse or crashes) on regex literals with Unicode escapes
533             && (isEcmaScript5OrGreater()
534                 || !containsUnicodeEscape(pattern.getString()))) {
535     
536           // Make sure that / is escaped, so that it will fit safely in /brackets/
537           // and make sure that no LineTerminatorCharacters appear literally inside
538           // the pattern.
539           // pattern is a string value with \\ and similar already escaped
540           pattern = makeForwardSlashBracketSafe(pattern);
541     
542           Node regexLiteral;
543           if (null == flags || "".equals(flags.getString())) {
544             // fold to /foobar/
545             regexLiteral = IR.regexp(pattern);
546           } else {
547             // fold to /foobar/gi
548             if (!areValidRegexpFlags(flags.getString())) {
549               report(INVALID_REGULAR_EXPRESSION_FLAGS, flags);
550               return n;
551             }
552             if (!areSafeFlagsToFold(flags.getString())) {
553               return n;
554             }
555             n.removeChild(flags);
556             regexLiteral = IR.regexp(pattern, flags);
557           }
558     
559           parent.replaceChild(n, regexLiteral);
560           reportCodeChange();
561           return regexLiteral;
562         }
563     
564         return n;
565       }

567       private Node reduceSubstractionAssignment(Node n) {
568         Node right = n.getLastChild();
569         if (right.isNumber()) {
570           if (right.getDouble() == 1) {
571             Node newNode = IR.dec(n.removeFirstChild(), false);
572             n.getParent().replaceChild(n, newNode);
573             reportCodeChange();
574             return newNode;
575           } else if (right.getDouble() == -1) {
576             Node newNode = IR.inc(n.removeFirstChild(), false);
577             n.getParent().replaceChild(n, newNode);
578             reportCodeChange();
579             return newNode;
580           }
581         }
582         return n;
583       }

585       private Node reduceTrueFalse(Node n) {
586         if (late) {
587           switch (n.getParent().getType()) {
588             case EQ:
589             case GT:
590             case GE:
591             case LE:
592             case LT:
593             case NE:
594               Node number = IR.number(n.isTrue() ? 1 : 0);
595               n.getParent().replaceChild(n, number);
596               reportCodeChange();
597               return number;
598           }
599     
600           Node not = IR.not(IR.number(n.isTrue() ? 0 : 1));
601           not.useSourceInfoIfMissingFromForTree(n);
602           n.getParent().replaceChild(n, not);
603           reportCodeChange();
604           return not;
605         }
606         return n;
607       }

609       private Node tryMinimizeArrayLiteral(Node n) {
610         boolean allStrings = true;
611         for (Node cur = n.getFirstChild(); cur != null; cur = cur.getNext()) {
612           if (!cur.isString()) {
613             allStrings = false;
614           }
615         }
616     
617         if (allStrings) {
618           return tryMinimizeStringArrayLiteral(n);
619         } else {
620           return n;
621         }
622       }
623     
624       private Node tryMinimizeStringArrayLiteral(Node n) {
625         if (!late) {
626           return n;
627         }
628     
629         int numElements = n.getChildCount();
630         // We save two bytes per element.
631         int saving = numElements * 2 - STRING_SPLIT_OVERHEAD;
632         if (saving <= 0) {
633           return n;
634         }
635     
636         String[] strings = new String[n.getChildCount()];
637         int idx = 0;
638         for (Node cur = n.getFirstChild(); cur != null; cur = cur.getNext()) {
639           strings[idx++] = cur.getString();
640         }
641     
642         // These delimiters are chars that appears a lot in the program therefore
643         // probably have a small Huffman encoding.
644         String delimiter = pickDelimiter(strings);
645         if (delimiter != null) {
646           String template = Joiner.on(delimiter).join(strings);
647           Node call = IR.call(
648               IR.getprop(
649                   IR.string(template),
650                   IR.string("split")),
651               IR.string("" + delimiter));
652           call.useSourceInfoIfMissingFromForTree(n);
653           n.getParent().replaceChild(n, call);
654           reportCodeChange();
655           return call;
656         }
657         return n;
658       }

660       /**
661        * Find a delimiter that does not occur in the given strings
662        * @param strings The strings that must be separated.
663        * @return a delimiter string or null
664        */
665       private static String pickDelimiter(String[] strings) {
666         boolean allLength1 = true;
667         for (String s : strings) {
668           if (s.length() != 1) {
669             allLength1 = false;
670             break;
671           }
672         }
673     
674         if (allLength1) {
675           return "";
676         }
677     
678         String[] delimiters = new String[]{" ", ";", ",", "{", "}", null};
679         int i = 0;
680         NEXT_DELIMITER: for (; delimiters[i] != null; i++) {
681           for (String cur : strings) {
682             if (cur.contains(delimiters[i])) {
683               continue NEXT_DELIMITER;
684             }
685           }
686           break;
687         }
688         return delimiters[i];
689       }

691       private static final Pattern REGEXP_FLAGS_RE = Pattern.compile("^[gmi]*$");
692     
693       /**
694        * are the given flags valid regular expression flags?
695        * JavaScript recognizes several suffix flags for regular expressions,
696        * 'g' - global replace, 'i' - case insensitive, 'm' - multi-line.
697        * They are case insensitive, and JavaScript does not recognize the extended
698        * syntax mode, single-line mode, or expression replacement mode from Perl 5.
699        */
700       private static boolean areValidRegexpFlags(String flags) {
701         return REGEXP_FLAGS_RE.matcher(flags).matches();
702       }
703     
704       /**
705        * are the given flags safe to fold?
706        * We don't fold the regular expression if global ('g') flag is on,
707        * because in this case it isn't really a constant: its 'lastIndex'
708        * property contains the state of last execution, so replacing
709        * 'new RegExp('foobar','g')' with '/foobar/g' may change the behavior of
710        * the program if the RegExp is used inside a loop, for example.
711        * <p>
712        * ECMAScript 5 explicitly disallows pooling of regular expression literals so
713        * in ECMAScript 5, {@code /foo/g} and {@code new RegExp('foo', 'g')} are
714        * equivalent.
715        * From section 7.8.5:
716        * "Then each time the literal is evaluated, a new object is created as if by
717        * the expression new RegExp(Pattern, Flags) where RegExp is the standard
718        * built-in constructor with that name."
719        */
720       private boolean areSafeFlagsToFold(String flags) {
721         return isEcmaScript5OrGreater() || flags.indexOf('g') < 0;
722       }

724       /**
725        * returns a string node that can safely be rendered inside /brackets/.
726        */
727       private static Node makeForwardSlashBracketSafe(Node n) {
728         String s = n.getString();
729         // sb contains everything in s[0:pos]
730         StringBuilder sb = null;
731         int pos = 0;
732         boolean isEscaped = false, inCharset = false;
733         for (int i = 0; i < s.length(); ++i) {
734           char ch = s.charAt(i);
735           switch (ch) {
736             case '\\':
737               isEscaped = !isEscaped;
738               continue;
739             case '/':
740               // Escape a literal forward slash if it is not already escaped and is
741               // not inside a character set.
742               //     new RegExp('/') -> /\//
743               // but the following do not need extra escaping
744               //     new RegExp('\\/') -> /\//
745               //     new RegExp('[/]') -> /[/]/
746               if (!isEscaped && !inCharset) {
747                 if (null == sb) { sb = new StringBuilder(s.length() + 16); }
748                 sb.append(s, pos, i).append('\\');
749                 pos = i;
750               }
751               break;
752             case '[':
753               if (!isEscaped) {
754                 inCharset = true;
755               }
756               break;
757             case ']':
758               if (!isEscaped) {
759                 inCharset = false;
760               }
761               break;
762             case '\r': case '\n': case '\u2028': case '\u2029':
763               // LineTerminators cannot appear raw inside a regular
764               // expression literal.
765               // They can't appear legally in a quoted string, but when
766               // the quoted string from
767               //     new RegExp('\n')
768               // reaches here, the quoting has been removed.
769               // Requote just these code-points.
770               if (null == sb) { sb = new StringBuilder(s.length() + 16); }
771               if (isEscaped) {
772                 sb.append(s, pos, i - 1);
773               } else {
774                 sb.append(s, pos, i);
775               }
776               switch (ch) {
777                 case '\r': sb.append("\\r"); break;
778                 case '\n': sb.append("\\n"); break;
779                 case '\u2028': sb.append("\\u2028"); break;
780                 case '\u2029': sb.append("\\u2029"); break;
781               }
782               pos = i + 1;
783               break;
784           }
785           isEscaped = false;
786         }
787     
788         if (null == sb) { return n.cloneTree(); }
789     
790         sb.append(s, pos, s.length());
791         return IR.string(sb.toString()).srcref(n);
792       }

794       /**
795        * true if the JavaScript string would contain a Unicode escape when written
796        * out as the body of a regular expression literal.
797        */
798       static boolean containsUnicodeEscape(String s) {
799         String esc = REGEXP_ESCAPER.regexpEscape(s);
800         for (int i = -1; (i = esc.indexOf("\\u", i + 1)) >= 0;) {
801           int nSlashes = 0;
802           while (i - nSlashes > 0 && '\\' == esc.charAt(i - nSlashes - 1)) {
803             ++nSlashes;
804           }
805           // if there are an even number of slashes before the \ u then it is a
806           // Unicode literal.
807           if (0 == (nSlashes & 1)) { return true; }
808         }
809         return false;
810       }
