17     package com.google.javascript.jscomp.deps;
18     
19     import java.util.List;
20     
21     /**
22      * A sorted list of inputs with dependency information.
23      * <p>
24      * Performs a sort to make sure that an input always comes after its
25      * dependencies.
26      * <p>
27      * Also exposes other information about the inputs, like which inputs
28      * do not provide symbols.
29      *
30      * TODO(tbreisacher): Consider removing this interface, since it now
31      * has only one class implementing it.
32      */
33     public interface SortedDependencies<INPUT extends DependencyInfo> {
34     
35       /**
36        * Return the input that gives us the given symbol.
37        * @throws MissingProvideException An exception if there is no
38        *     input for this symbol.
39        */
40       public INPUT getInputProviding(String symbol) throws MissingProvideException;
41     
42       /**
43        * Return the input that gives us the given symbol, or null.
44        */
45       public INPUT maybeGetInputProviding(String symbol);
46     
47       public List<INPUT> getSortedList();
48     
49       /**
50        * Gets all the dependencies of the given roots. The inputs must be returned
51        * in a stable order. In other words, if A comes before B, and A does not
52        * transitively depend on B, then A must also come before B in the returned
53        * list.
54        */
55       public List<INPUT> getSortedDependenciesOf(List<INPUT> roots);
56     
57       /**
58        * Gets all the dependencies of the given roots. The inputs must be returned
59        * in a stable order. In other words, if A comes before B, and A does not
60        * transitively depend on B, then A must also come before B in the returned
61        * list.
62        *
63        * @param sorted If true, get them in topologically sorted order. If false,
64        *     get them in the original order they were passed to the compiler.
65        */
66       public List<INPUT> getDependenciesOf(List<INPUT> roots, boolean sorted);
67     
68       public List<INPUT> getInputsWithoutProvides();
69     
70       public static class MissingProvideException extends Exception {
71         public MissingProvideException(String provide) {
72           super(provide);
73         }
74     
75         public MissingProvideException(String provide, Exception e) {
76           super(provide, e);
77         }
78       }
79     }
