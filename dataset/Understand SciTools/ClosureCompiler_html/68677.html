16     package com.google.javascript.jscomp;
17     
18     import static com.google.common.truth.Truth.assert_;
19     import static java.nio.charset.StandardCharsets.UTF_8;
20     
21     import com.google.common.collect.ImmutableList;
22     import com.google.common.io.Files;
23     
24     import java.io.File;
25     import java.io.IOException;
26     import java.util.ArrayList;
27     import java.util.List;
28     
29     /**
30      * Test compiler by comparing output to data files aka golden files.
31      * Always start with the options object from this class.
32      */
33     public class GoldenFileComparer {
34     
35       private static final String DATA_DIR =
36           "test/"
37               + "com/google/javascript/jscomp/testdata/";
38     
39       private static List<SourceFile> coverageExterns() {
40           SourceFile externs = SourceFile.fromCode("externs", "function Symbol() {}; var window;");
41           return ImmutableList.of(externs);
42       }
43     
44       private static String compile(
45           List<SourceFile> externs, List<SourceFile> inputs, CompilerOptions options) {
46         Compiler compiler = new Compiler();
47     
48         List<SourceFile> allExterns = new ArrayList<>();
49         allExterns.addAll(externs);
50         allExterns.addAll(coverageExterns());
51     
52         compiler.compile(allExterns, inputs, options);
53         return compiler.toSource();
54       }
55     
56       private static String readFile(String filePath) throws IOException {
57         return Files.toString(new File(filePath), UTF_8);
58       }
59     
60       private static SourceFile readSource(String fileName) throws IOException {
61         String source = readFile(toFullPath(fileName));
62        return SourceFile.fromCode(fileName, source);
63       }
64     
65       private static boolean compare(String compiledSource, String goldenSource) {
66         String[] compiledLines = compiledSource.split("\n");
67         String[] goldenLines = goldenSource.split("\n");
68     
69         assert_()
70             .withFailureMessage("Num lines of compiled code do not match that of expected")
71             .that(compiledLines.length)
72             .isEqualTo(goldenLines.length);
73     
74         // Loop through each line to make it convenient to pin-point the faulty one
75         for (int i = 0; i < compiledLines.length; i++) {
76           assert_()
77               .withFailureMessage(
78                   "Instrumented code does not match expected at line no: "
79                       + (i + 1)
80                       + "\nExpected:\n"
81                       + goldenLines[i]
82                       + "\nFound:\n"
83                       + compiledLines[i]
84                       + "\n")
85               .that(compiledLines[i])
86               .isEqualTo(goldenLines[i]);
87         }
88     
89         return true;
90       }
91     
92       private static String toFullPath(String fileName) {
93         return DATA_DIR + fileName;
94       }
95     
96       private static void compileAndCompare(
97           String goldenFileName,
98           CompilerOptions options,
99           List<SourceFile> sourceFiles,
100           List<SourceFile> externsFiles)
101           throws Exception {
102         String compiledSource = compile(externsFiles, sourceFiles, options);
103         String referenceSource = readFile(toFullPath(goldenFileName));
104         compare(compiledSource, referenceSource);
105       }
106     
107       /**
108        * Always use these options, they set --pretty_print option for easy verification.
109        */
110       public static CompilerOptions options() {
111         CompilerOptions options = new CompilerOptions();
112         // Instrumentation done
113         options.setPrettyPrint(true);
114         return options;
115       }
116     
117       /**
118        * Compile one input file and throw if the result does not match golden.
119        * Pass options from this class, mutated with desired options
120        */
121       public static void compileAndCompare(
122           String goldenFileName, CompilerOptions options, String sourceFileName) throws Exception {
123         List<SourceFile> sourceFiles =
124             ImmutableList.of(readSource(sourceFileName));
125         List<SourceFile> externsFiles = ImmutableList.of();
126         compileAndCompare(goldenFileName, options, sourceFiles, externsFiles);
127       }
128     
129       /**
130        * Compile two input files and throw if the result does not match golden.
131        * Pass options from this class, mutated with desired options
132        */
133       public static void compileAndCompare(
134           String goldenFileName,
135           CompilerOptions options,
136           String sourceFileName1,
137           String sourceFileName2,
138           String externsFileName)
139           throws Exception {
140         // Prepare sources
141         List<SourceFile> sourceFiles = ImmutableList.of(
142             readSource(sourceFileName1),
143             readSource(sourceFileName2));
144     
145         List<SourceFile> externsFiles =
146             ImmutableList.of(SourceFile.fromFile(toFullPath(externsFileName)));
147     
148         compileAndCompare(goldenFileName, options, sourceFiles, externsFiles);
149       }
150     }
