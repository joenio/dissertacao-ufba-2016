16     package com.google.javascript.jscomp;
17     
18     import com.google.javascript.rhino.IR;
19     import com.google.javascript.rhino.Node;
20     
21     /**
22      * Splits variable declarations that declare multiple variables into
23      * separate declarations, if at least one of the declarations is a
24      * destructuring declaration. For example
25      * <pre>
26      *   var [a, b] = foo(), c = bar();}
27      * </pre>
28      * becomes
29      * <pre>
30      *   var [a, b] = foo();
31      *   var c = bar();
32      * </pre>
33      *
34      * <p>This runs before the main ES6 transpilation step, to simplify
35      * the transpilation of destructuring syntax.
36      *
37      * @author tbreisacher@google.com (Tyler Breisacher)
38      */
39     public final class Es6SplitVariableDeclarations extends
40         NodeTraversal.AbstractPostOrderCallback implements HotSwapCompilerPass {
41       private final AbstractCompiler compiler;
42       public Es6SplitVariableDeclarations(AbstractCompiler compiler) {
43         this.compiler = compiler;
44       }
45     
46       @Override
47       public void process(Node externs, Node root) {
48         NodeTraversal.traverseEs6(compiler, externs, this);
49         NodeTraversal.traverseEs6(compiler, root, this);
50       }
51     
52       @Override
53       public void hotSwapScript(Node scriptRoot, Node originalRoot) {
54         NodeTraversal.traverseEs6(compiler, scriptRoot, this);
55       }
56     
57       @Override
58       public void visit(NodeTraversal t, Node n, Node parent) {
59         if (NodeUtil.isDestructuringDeclaration(n)) {
60           splitDeclaration(n, parent);
61         }
62       }
63     
64       public void splitDeclaration(Node n, Node parent) {
65         while (n.getFirstChild() != n.getLastChild()) {
66           Node child = n.getLastChild().detachFromParent();
67           Node newVar = IR.declaration(child, n.getType()).srcref(n);
68           parent.addChildAfter(newVar, n);
69           compiler.reportCodeChange();
70         }
71       }
72     }
