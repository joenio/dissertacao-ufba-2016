17     package com.google.javascript.jscomp;
18     
19     import com.google.common.base.Preconditions;
20     import com.google.common.collect.ImmutableMap;
21     import com.google.javascript.rhino.JSDocInfo;
22     import com.google.javascript.rhino.JSDocInfo.Visibility;
23     import com.google.javascript.rhino.Node;
24     import com.google.javascript.rhino.StaticSourceFile;
25     
26     /**
27      * Compiler pass that collects visibility annotations in {@code @fileoverview}
28      * blocks. Used by {@link CheckAccessControls}.
29      *
30      * @author brndn@google.com (Brendan Linn)
31      */
32     class CollectFileOverviewVisibility implements HotSwapCompilerPass {
33     
34       private final AbstractCompiler compiler;
35       private final ImmutableMap.Builder<StaticSourceFile, Visibility> builder =
36           ImmutableMap.builder();
37     
38       CollectFileOverviewVisibility(AbstractCompiler compiler) {
39         this.compiler = compiler;
40       }
41     
42       @Override
43       public void process(Node externs, Node root) {
44         for (Node script = root.getFirstChild();
45             script != null;
46             script = script.getNext()) {
47           Preconditions.checkState(script.isScript());
48           visit(script);
49         }
50       }
51     
52       private void visit(Node scriptNode) {
53         JSDocInfo jsDocInfo = scriptNode.getJSDocInfo();
54         if (jsDocInfo == null) {
55           return;
56         }
57         Visibility v = jsDocInfo.getVisibility();
58         if (v == null) {
59           return;
60         }
61         builder.put(scriptNode.getStaticSourceFile(), v);
62       }
63     
64       ImmutableMap<StaticSourceFile, Visibility> getFileOverviewVisibilityMap() {
65         return builder.build();
66       }
67     
68       @Override
69       public void hotSwapScript(Node scriptRoot, Node originalRoot) {
70         compiler.process(this);
71       }
72     
73     }
