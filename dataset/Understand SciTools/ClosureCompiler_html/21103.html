30     /**
31      * Finds all references to global symbols and rewrites them to be property
32      * accesses to a special object with the same name as the global symbol.
33      *
34      * Given the name of the global object is NS
35      * <pre> var a = 1; function b() { return a }</pre>
36      * becomes
37      * <pre> NS.a = 1; NS.b = function b() { return NS.a }</pre>
38      *
39      * This allows splitting code into modules that depend on each other's
40      * global symbols, without using polluting JavaScript's global scope with those
41      * symbols. You typically define just a single global symbol, wrap each module
42      * in a function wrapper, and pass the global symbol around, eg,
43      * <pre> var uniqueNs = uniqueNs || {}; </pre>
44      * <pre> (function (NS) { ...your module code here... })(uniqueNs); </pre>
45      *
46      *
47      * <p>This compile step requires moveFunctionDeclarations to be turned on
48      * to guarantee semantics.
49      *
50      * <p>For lots of examples, see the unit test.
51      *
52      *
53      */
54     final class RescopeGlobalSymbols implements CompilerPass {
55     
56       // Appended to variables names that conflict with globalSymbolNamespace.
57       private static final String DISAMBIGUATION_SUFFIX = "$";
58       private static final String WINDOW = "window";
59       private static final Set<String> SPECIAL_EXTERNS =
60           ImmutableSet.of(WINDOW, "eval", "arguments", "undefined",
61               // The javascript built-in objects (listed in Ecma 262 section 4.2)
62               "Object", "Function", "Array", "String", "Boolean", "Number", "Math",
63               "Date", "RegExp", "JSON", "Error", "EvalError", "ReferenceError",
64               "SyntaxError", "TypeError", "URIError");
65     
66       private final AbstractCompiler compiler;
67       private final String globalSymbolNamespace;
68       private final boolean addExtern;
69       private final boolean assumeCrossModuleNames;
70       private final Set<String> crossModuleNames = new HashSet<>();
71       private final Set<String> maybeReferencesThis = new HashSet<>();
72

74        * Constructor for the RescopeGlobalSymbols compiler pass.
75        *

77        * @param globalSymbolNamespace Name of namespace into which all global

80        *     cross module boundaries and thus require renaming.

90        * Constructor for the RescopeGlobalSymbols compiler pass for use in testing.
91        *

93        * @param globalSymbolNamespace Name of namespace into which all global
94        *     symbols are transferred.
95        * @param addExtern If true, the compiler will consider the
96        *    globalSymbolNamespace an extern name.
97        * @param assumeCrossModuleNames If true, all global symbols will be assumed
98        *     cross module boundaries and thus require renaming.
99        * VisibleForTesting
100        */
101       RescopeGlobalSymbols(
102           AbstractCompiler compiler,
103           String globalSymbolNamespace,
104           boolean addExtern,
105           boolean assumeCrossModuleNames) {
106         this.compiler = compiler;
107         this.globalSymbolNamespace = globalSymbolNamespace;
108         this.addExtern = addExtern;
109         this.assumeCrossModuleNames = assumeCrossModuleNames;
110       }

112       private boolean isCrossModuleName(String name) {
113         return assumeCrossModuleNames || crossModuleNames.contains(name)
114             || compiler.getCodingConvention().isExported(name, false);
115       }
116     
117       private void addExternForGlobalSymbolNamespace() {
118         Node varNode = IR.var(IR.name(globalSymbolNamespace));
119         CompilerInput input = compiler.getSynthesizedExternsInput();
120         input.getAstRoot(compiler).addChildrenToBack(varNode);
121         compiler.reportCodeChange();
122       }
123     
124       @Override
125       public void process(Node externs, Node root) {
126         // Make the name of the globalSymbolNamespace an extern.
127         if (addExtern) {
128           addExternForGlobalSymbolNamespace();
129         }
130         // Rewrite all references to global symbols to properties of a
131         // single symbol by:
132         // (If necessary the 4 traversals could be combined. They are left
133         // separate for readability reasons.)
134         // 1. turning global named function statements into var assignments.
135         NodeTraversal.traverseEs6(
136             compiler,
137             root,
138             new RewriteGlobalFunctionStatementsToVarAssignmentsCallback());
139         // 2. find global names that are used in more than one module. Those that
140         //    are have to be rewritten.
141         List<Callback> nonMutatingPasses = new ArrayList<>();
142         nonMutatingPasses.add(new FindCrossModuleNamesCallback());
143         //    and find names that may refering functions that reference this.
144         nonMutatingPasses.add(new FindNamesReferencingThis());
145         CombinedCompilerPass.traverse(compiler, root, nonMutatingPasses);
146         // 3. rewriting all references to be property accesses of the single symbol.
147         RewriteScopeCallback rewriteScope = new RewriteScopeCallback();
148         NodeTraversal.traverseEs6(compiler, root, rewriteScope);
149         // 4. removing the var from statements in global scope if the declared names
150         //    have been rewritten in the previous pass.
151         NodeTraversal.traverseEs6(compiler, root, new RemoveGlobalVarCallback());
152         rewriteScope.declareModuleGlobals();
153     
154         // Extra pass which makes all extern global symbols reference window
155         // explicitly.
156         NodeTraversal.traverseEs6(
157             compiler,
158             root,
159             new MakeExternsReferenceWindowExplicitly());
160       }

162       /**
163        * Rewrites function statements to var statements + assignment.
164        *
165        * <pre>function test(){}</pre>
166        * becomes
167        * <pre>var test = function (){}</pre>
168        *
169        * After this traversal, the special case of global function statements
170        * can be ignored.
171        */
172       private class RewriteGlobalFunctionStatementsToVarAssignmentsCallback
173           extends AbstractShallowStatementCallback {
174         @Override
175         public void visit(NodeTraversal t, Node n, Node parent) {
176           if (NodeUtil.isFunctionDeclaration(n)) {
177             String name = NodeUtil.getName(n);
178             n.getFirstChild().setString("");
179             Node prev = parent.getChildBefore(n);
180             n.detachFromParent();
181             Node var = NodeUtil.newVarNode(name, n);
182             if (prev == null) {
183               parent.addChildToFront(var);
184             } else {
185               parent.addChildAfter(var, prev);
186             }
187             compiler.reportCodeChange();
188           }
189         }
190       }

192       /**
193        * Find all global names that are used in more than one module. The following
194        * compiler transformations can ignore the globals that are not.
195        */
196       private class FindCrossModuleNamesCallback extends
197           AbstractPostOrderCallback {
198         @Override
199         public void visit(NodeTraversal t, Node n, Node parent) {
200           if (n.isName()) {
201             String name = n.getString();
202             if ("".equals(name) || crossModuleNames.contains(name)) {
203               return;
204             }
205             Scope s = t.getScope();
206             Var v = s.getVar(name);
207             if (v == null || !v.isGlobal()) {
208               return;
209             }
210             CompilerInput input = v.getInput();
211             if (input == null) {
212               // We know nothing. Assume name is used across modules.
213               crossModuleNames.add(name);
214               return;
215             }
216             // Compare the module where the variable is declared to the current
217             // module. If they are different, the variable is used across modules.
218             JSModule module = input.getModule();
219             if (module != t.getModule()) {
220               crossModuleNames.add(name);
221             }
222           }
223         }
224       }

226       /**
227        * Builds the maybeReferencesThis set of names that may reference a function
228        * that references this. If the function a name references does not reference
229        * this it can be called as a method call where the this value is not the
230        * same as in a normal function call.
231        */
232       private class FindNamesReferencingThis extends
233           AbstractPostOrderCallback {
234         @Override
235         public void visit(NodeTraversal t, Node n, Node parent) {
236           if (n.isName()) {
237             String name = n.getString();
238             if (name.isEmpty()) {
239               return;
240             }
241             Node value = null;
242             if (parent.isAssign() && n == parent.getFirstChild()) {
243               value = parent.getLastChild();
244             } else if (parent.isVar()) {
245               value = n.getFirstChild();
246             } else if (parent.isFunction()) {
247               value = parent;
248             }
249             if (value == null) {
250               return;
251             }
252             // We already added this symbol. Done after checks above because those
253             // are comparatively cheap.
254             if (maybeReferencesThis.contains(name)) {
255               return;
256             }
257             Scope s = t.getScope();
258             Var v = s.getVar(name);
259             if (v == null || !v.isGlobal()) {
260               return;
261             }
262             // If anything but a function is assignment we assume that possibly
263             // a function referencing this is being assignment. Otherwise we
264             // check whether the function that is being assigned references this.
265             if (!value.isFunction() || NodeUtil.referencesThis(value)) {
266               maybeReferencesThis.add(name);
267             }
268           }
269         }
270       }

272       /**
273        * Visits each NAME token and checks whether it refers to a global variable.
274        * If yes, rewrites the name to be a property access on the
275        * "globalSymbolNamespace".
276        *
277        * <pre>var a = 1, b = 2, c = 3;</pre>
278        * becomes
279        * <pre>var NS.a = 1, NS.b = 2, NS.c = 4</pre>
280        * (The var token is removed in a later traversal.)
281        *
282        * <pre>a + b</pre>
283        * becomes
284        * <pre>NS.a + NS.b</pre>
285        *
286        * <pre>a()</pre>
287        * becomes
288        * <pre>(0,NS.a)()</pre>
289        * Notice the special syntax here to preserve the *this* semantics in the
290        * function call.
291        */
292       private class RewriteScopeCallback extends AbstractPostOrderCallback {
293     
294         List<ModuleGlobal> preDeclarations = new ArrayList<>();
295     
296         @Override
297         public void visit(NodeTraversal t, Node n, Node parent) {
298           if (!n.isName()) {
299             return;
300           }
301           String name = n.getString();
302           // Ignore anonymous functions
303           if (parent.isFunction() && name.isEmpty()) {
304             return;
305           }
306           Var var = t.getScope().getVar(name);
307           if (var == null) {
308             return;
309           }
310           // Don't touch externs.
311           if (var.isExtern()) {
312             return;
313           }
314           // When the globalSymbolNamespace is used as a local variable name
315           // add suffix to avoid shadowing the namespace. Also add a suffix
316           // if a name starts with the name of the globalSymbolNamespace and
317           // the suffix.
318           if (!var.isExtern() && !var.isGlobal()
319               && (name.equals(globalSymbolNamespace)
320                   || name.startsWith(
321                       globalSymbolNamespace + DISAMBIGUATION_SUFFIX))) {
322             n.setString(name + DISAMBIGUATION_SUFFIX);
323             compiler.reportCodeChange();
324           }
325           // We only care about global vars.
326           if (!var.isGlobal()) {
327             return;
328           }
329           Node nameNode = var.getNameNode();
330           // The exception variable (e in try{}catch(e){}) should not be rewritten.
331           if (nameNode != null && nameNode.getParent() != null
332               && nameNode.getParent().isCatch()) {
333             return;
334           }
335           replaceSymbol(n, name, t.getInput());
336         }

338         private void replaceSymbol(Node node, String name, CompilerInput input) {
339           Node parent = node.getParent();
340           boolean isCrossModule = isCrossModuleName(name);
341           if (!isCrossModule) {
342             // When a non cross module name appears outside a var declaration we
343             // never have to do anything.
344             if (!parent.isVar()) {
345               return;
346             }
347             // If it is a var declaration, but no cross module names are declared
348             // we also don't have to do anything.
349             boolean hasCrossModuleChildren = false;
350             for (Node c : parent.children()) {
351               // Var child is no longer a name means it was transformed already
352               // which means there was a cross module name.
353               if (!c.isName() || isCrossModuleName(c.getString())) {
354                 hasCrossModuleChildren = true;
355                 break;
356               }
357             }
358             if (!hasCrossModuleChildren) {
359               return;
360             }
361           }
362           Node replacement = isCrossModule
363               ? IR.getprop(
364                   IR.name(globalSymbolNamespace).srcref(node),
365                   IR.string(name).srcref(node))
366               : IR.name(name).srcref(node);
367           replacement.srcref(node);
368           if (node.hasChildren()) {
369             // var declaration list: var a = 1, b = 2;
370             Node assign = IR.assign(
371                 replacement,
372                 node.removeFirstChild());
373             parent.replaceChild(node, assign);
374           } else if (isCrossModule) {
375             parent.replaceChild(node, replacement);
376             if (parent.isCall() && !maybeReferencesThis.contains(name)) {
377               // Do not write calls like this: (0, _a)() but rather as _.a(). The
378               // this inside the function will be wrong, but it doesn't matter
379               // because the this is never read.
380               parent.putBooleanProp(Node.FREE_CALL, false);
381             }
382           }
383           // If we changed a non cross module name that was in a var declaration
384           // we need to preserve that var declaration. Because it is global
385           // anyway, we just put it at the beginning of the current input.
386           // Example:
387           // var crossModule = i++, notCrossModule = i++
388           // becomes
389           // var notCrossModule;_.crossModule = i++, notCrossModule = i++
390           if (!isCrossModule && parent.isVar()) {
391             preDeclarations.add(new ModuleGlobal(
392                 input.getAstRoot(compiler),
393                 IR.name(name).srcref(node)));
394           }
395           compiler.reportCodeChange();
396         }
397     
398         /**
399          * Adds back declarations for variables that do not cross module boundaries.
400          * Must be called after RemoveGlobalVarCallback.
401          */
402         void declareModuleGlobals() {
403           for (ModuleGlobal global : preDeclarations) {
404             if (global.root.getFirstChild() != null
405                 && global.root.getFirstChild().isVar()) {
406               global.root.getFirstChild().addChildToBack(global.name);
407             } else {
408               global.root.addChildToFront(
409                   IR.var(global.name).srcref(global.name));
410             }
411             compiler.reportCodeChange();
412           }
413         }

416          * Variable that doesn't cross module boundaries.

418         private class ModuleGlobal {
419           final Node root;
420           final Node name;
421     
422           ModuleGlobal(Node root, Node name) {
423             this.root = root;
424             this.name = name;
425           }
426         }

429       /**
430        * Removes every occurrence of var that declares a global variable.
431        *
432        * <pre>var NS.a = 1, NS.b = 2;</pre>
433        * becomes
434        * <pre>NS.a = 1; NS.b = 2;</pre>
435        *
436        * <pre>for (var a = 0, b = 0;;)</pre>
437        * becomes
438        * <pre>for (NS.a = 0, NS.b = 0;;)</pre>
439        *
440        * Declarations without assignments are optimized away:
441        * <pre>var a = 1, b;</pre>
442        * becomes
443        * <pre>NS.a = 1</pre>
444        */
445       private class RemoveGlobalVarCallback extends
446           AbstractShallowStatementCallback {
447         @Override
448         public void visit(NodeTraversal t, Node n, Node parent) {
449           if (!n.isVar()) {
450             return;
451           }
452     
453           List<Node> commas = new ArrayList<>();
454           List<Node> interestingChildren = new ArrayList<>();
455           // Filter out declarations without assignments.
456           // As opposed to regular var nodes, there are always assignments
457           // because the previous traversal in RewriteScopeCallback creates
458           // them.
459           boolean allName = true;
460           for (Node c : n.children()) {
461             if (!c.isName()) {
462               allName = false;
463             }
464             if (c.isAssign() || parent.isFor()) {
465               interestingChildren.add(c);
466             }
467           }
468           // If every child of a var declares a name, it must stay in place.
469           // This is the case if none of the declared variables cross module
470           // boundaries.
471           if (allName) {
472             return;
473           }
474           for (Node c : interestingChildren) {
475             if (parent.isFor() && parent.getFirstChild() == n) {
476               commas.add(c.cloneTree());
477             } else {
478               // Var statement outside of for-loop.
479               Node expr = IR.exprResult(c.cloneTree()).srcref(c);
480               parent.addChildBefore(expr, n);
481             }
482           }
483           if (!commas.isEmpty()) {
484             Node comma = joinOnComma(commas, n);
485             parent.addChildBefore(comma, n);
486           }
487           // Remove the var node.
488           parent.removeChild(n);
489           compiler.reportCodeChange();
490         }
491     
492         private Node joinOnComma(List<Node> commas, Node source) {
493           Node comma = commas.get(0);
494           for (int i = 1; i < commas.size(); i++) {
495             Node nextComma = IR.comma(comma, commas.get(i));
496             nextComma.useSourceInfoIfMissingFrom(source);
497             comma = nextComma;
498           }
499           return comma;
500         }
501       }

503       /**
504        * Rewrites extern names to be explicit children of window instead of only
505        * implicitly referencing it.
506        * This enables injecting window into a scope and make all global symbol
507        * depend on the injected object.
508        */
509       private class MakeExternsReferenceWindowExplicitly extends
510           AbstractPostOrderCallback {
511     
512         @Override
513         public void visit(NodeTraversal t, Node n, Node parent) {
514           if (!n.isName()) {
515             return;
516           }
517           String name = n.getString();
518           if (globalSymbolNamespace.equals(name)
519               || SPECIAL_EXTERNS.contains(name)) {
520             return;
521           }
522           Var var = t.getScope().getVar(name);
523           if (name.length() > 0 && (var == null || var.isExtern())) {
524             parent.replaceChild(n, IR.getprop(IR.name(WINDOW), IR.string(name))
525                 .srcrefTree(n));
526             compiler.reportCodeChange();
527           }
528         }
529       }
