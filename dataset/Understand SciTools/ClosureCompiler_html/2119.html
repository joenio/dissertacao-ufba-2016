67       /**
68        * Closure's goog.inherits adds a {@code superClass_} property to the
69        * subclass, and a {@code constructor} property.
70        */
71       @Override
72       public void applySubclassRelationship(FunctionType parentCtor,
73           FunctionType childCtor, SubclassType type) {
74         super.applySubclassRelationship(parentCtor, childCtor, type);
75         if (type == SubclassType.INHERITS) {
76           childCtor.defineDeclaredProperty("superClass_",
77               parentCtor.getPrototype(), childCtor.getSource());
78           childCtor.getPrototype().defineDeclaredProperty("constructor",
79               // Notice that constructor functions do not need to be covariant
80               // on the superclass.
81               // So if G extends F, new G() and new F() can accept completely
82               // different argument types, but G.prototype.constructor needs
83               // to be covariant on F.prototype.constructor.
84               // To get around this, we just turn off type-checking on arguments
85               // and return types of G.prototype.constructor.
86               childCtor.forgetParameterAndReturnTypes(),
87               childCtor.getSource());
88         }
89       }

91       /**
92        * {@inheritDoc}
93        *
94        * <p>Understands several different inheritance patterns that occur in
95        * Google code (various uses of {@code inherits} and {@code mixin}).
96        */
97       @Override
98       public SubclassRelationship getClassesDefinedByCall(Node callNode) {
99         SubclassRelationship relationship =
100             super.getClassesDefinedByCall(callNode);
101         if (relationship != null) {
102           return relationship;
103         }
104     
105         Node callName = callNode.getFirstChild();
106         SubclassType type = typeofClassDefiningName(callName);
107         if (type != null) {
108           Node subclass = null;
109           Node superclass = callNode.getLastChild();
110     
111           // There are six possible syntaxes for a class-defining method:
112           // SubClass.inherits(SuperClass)
113           // goog.inherits(SubClass, SuperClass)
114           // goog$inherits(SubClass, SuperClass)
115           // SubClass.mixin(SuperClass.prototype)
116           // goog.mixin(SubClass.prototype, SuperClass.prototype)
117           // goog$mixin(SubClass.prototype, SuperClass.prototype)
118           boolean isDeprecatedCall = callNode.getChildCount() == 2 &&
119               callName.isGetProp();
120           if (isDeprecatedCall) {
121             // SubClass.inherits(SuperClass)
122             subclass = callName.getFirstChild();
123           } else if (callNode.getChildCount() == 3) {
124             // goog.inherits(SubClass, SuperClass)
125             subclass = callName.getNext();
126           } else {
127             return null;
128           }
129     
130           if (type == SubclassType.MIXIN) {
131             // Only consider mixins that mix two prototypes as related to
132             // inheritance.
133             if (!endsWithPrototype(superclass)) {
134               return null;
135             }
136             if (!isDeprecatedCall) {
137               if (!endsWithPrototype(subclass)) {
138                 return null;
139               }
140               // Strip off the prototype from the name.
141               subclass = subclass.getFirstChild();
142             }
143             superclass = superclass.getFirstChild();
144           }
145     
146           // bail out if either of the side of the "inherits"
147           // isn't a real class name. This prevents us from
148           // doing something weird in cases like:
149           // goog.inherits(MySubClass, cond ? SuperClass1 : BaseClass2)
150           if (subclass != null &&
151               subclass.isUnscopedQualifiedName() &&
152               superclass.isUnscopedQualifiedName()) {
153             return new SubclassRelationship(type, subclass, superclass);
154           }
155         }
156     
157         return null;
158       }

165       /**
166        * Determines whether the given node is a class-defining name, like
167        * "inherits" or "mixin."
168        * @return The type of class-defining name, or null.
169        */
170       private static SubclassType typeofClassDefiningName(Node callName) {
171         // Check if the method name matches one of the class-defining methods.
172         String methodName = null;
173         if (callName.isGetProp()) {
174           methodName = callName.getLastChild().getString();
175         } else if (callName.isName()) {
176           String name = callName.getString();
177           int dollarIndex = name.lastIndexOf('$');
178           if (dollarIndex != -1) {
179             methodName = name.substring(dollarIndex + 1);
180           }
181         }
182     
183         if (methodName != null) {
184           if (methodName.equals("inherits")) {
185             return SubclassType.INHERITS;
186           } else if (methodName.equals("mixin")) {
187             return SubclassType.MIXIN;
188           }
189         }
190         return null;
191       }

199       /**
200        * Given a qualified name node, returns whether "prototype" is at the end.
201        * For example:
202        * a.b.c => false
203        * a.b.c.prototype => true
204        */
205       private static boolean endsWithPrototype(Node qualifiedName) {
206         return qualifiedName.isGetProp() &&
207             qualifiedName.getLastChild().getString().equals("prototype");
208       }

244       private static String extractClassNameIfGoog(Node node, Node parent,
245           String functionName){
246         String className = null;
247         if (NodeUtil.isExprCall(parent)) {
248           Node callee = node.getFirstChild();
249           if (callee != null && callee.isGetProp() && callee.matchesQualifiedName(functionName)) {
250             Node target = callee.getNext();
251             if (target != null && target.isString()) {
252               className = target.getString();
253             }
254           }
255         }
256         return className;
257       }

277       @Override
278       public List<String> identifyTypeDeclarationCall(Node n) {
279         Node callName = n.getFirstChild();
280         if (callName.matchesQualifiedName("goog.addDependency") &&
281             n.getChildCount() >= 3) {
282           Node typeArray = callName.getNext().getNext();
283           if (typeArray.isArrayLit()) {
284             List<String> typeNames = new ArrayList<>();
285             for (Node name = typeArray.getFirstChild(); name != null;
286                  name = name.getNext()) {
287               if (name.isString()) {
288                 typeNames.add(name.getString());
289               }
290             }
291             return typeNames;
292           }
293         }
294     
295         // Identify forward declaration of form goog.forwardDeclare('foo.bar')
296         if (callName.matchesQualifiedName("goog.forwardDeclare") &&
297             n.getChildCount() == 2) {
298           Node typeDeclaration = n.getSecondChild();
299           if (typeDeclaration.isString()) {
300             return ImmutableList.of(typeDeclaration.getString());
301           }
302         }
303     
304         return super.identifyTypeDeclarationCall(n);
305       }

314         Node callArg = callNode.getFirstChild();
315         // Use both the original name and the post-CollapseProperties name.
316         if (callNode.getChildCount() == 2
317             && (callArg.matchesQualifiedName("goog.addSingletonGetter")
318                 || callArg.matchesQualifiedName("goog$addSingletonGetter"))) {
319           return callArg.getNext().getQualifiedName();
320         }
321         return super.getSingletonGetterClassName(callNode);

369       @Override
370       public ObjectLiteralCast getObjectLiteralCast(Node callNode) {
371         Preconditions.checkArgument(callNode.isCall());
372         ObjectLiteralCast proxyCast = super.getObjectLiteralCast(callNode);
373         if (proxyCast != null) {
374           return proxyCast;
375         }
376     
377         Node callName = callNode.getFirstChild();
378         if (!callName.matchesQualifiedName("goog.reflect.object") ||
379             callNode.getChildCount() != 3) {
380           return null;
381         }
382     
383         Node typeNode = callName.getNext();
384         if (!typeNode.isQualifiedName()) {
385           return null;
386         }
387     
388         Node objectNode = typeNode.getNext();
389         if (!objectNode.isObjectLit()) {
390           return new ObjectLiteralCast(null, null, OBJECTLIT_EXPECTED);
391         }
392     
393         return new ObjectLiteralCast(
394             typeNode.getQualifiedName(), typeNode.getNext(), null);
395       }

419       @Override
420       public Bind describeFunctionBind(
421           Node n, boolean callerChecksTypes, boolean iCheckTypes) {
422         if (!n.isCall()) {
423           return null;
424         }
425         Node callTarget = n.getFirstChild();
426         if (callTarget.isQualifiedName()) {
427           if (callTarget.matchesQualifiedName("goog.bind")
428               || callTarget.matchesQualifiedName("goog$bind")) {
429             // goog.bind(fn, self, args...);
430             Node fn = callTarget.getNext();
431             if (fn == null) {
432               return null;
433             }
434             Node thisValue = safeNext(fn);
435             Node parameters = safeNext(thisValue);
436             return new Bind(fn, thisValue, parameters);
437           }
438     
439           if (callTarget.matchesQualifiedName("goog.partial") ||
440               callTarget.matchesQualifiedName("goog$partial")) {
441             // goog.partial(fn, args...);
442             Node fn = callTarget.getNext();
443             if (fn == null) {
444               return null;
445             }
446             Node thisValue = null;
447             Node parameters = safeNext(fn);
448             return new Bind(fn, thisValue, parameters);
449           }
450         }
451         return super.describeFunctionBind(n, callerChecksTypes, iCheckTypes);
452       }

454       @Override
455       public Cache describeCachingCall(Node node) {
456         if (!node.isCall()) {
457           return null;
458         }
459     
460         Node callTarget = node.getFirstChild();
461         if (callTarget.isQualifiedName()
462             && (callTarget.matchesQualifiedName("goog.reflect.cache")
463                 || callTarget.matchesQualifiedName("goog$reflect$cache"))) {
464           int paramCount = node.getChildCount() - 1;
465           if (3 <= paramCount && paramCount <= 4) {
466             Node cacheObj = callTarget.getNext();
467             Node keyNode = cacheObj.getNext();
468             Node valueFn = keyNode.getNext();
469             Node keyFn = valueFn.getNext();
470     
471             return new Cache(cacheObj, keyNode, valueFn, keyFn);
472           }
473         }
474     
475         return super.describeCachingCall(node);
476       }

499         /**
500          * Returns the type for a type assertion, or null if the function asserts
501          * that the node must not be null or undefined.
502          */
503         @Override
504         public com.google.javascript.rhino.jstype.JSType
505             getAssertedOldType(Node call, JSTypeRegistry registry) {
506           if (call.getChildCount() > 2) {
507             Node constructor = call.getSecondChild().getNext();
508             if (constructor != null) {
509               com.google.javascript.rhino.jstype.JSType ownerType =
510                   constructor.getJSType();
511               if (ownerType != null
512                   && ownerType.isFunctionType()
513                   && ownerType.isConstructor()) {
514                 FunctionType functionType = ((FunctionType) ownerType);
515                 return functionType.getInstanceType();
516               }
517             }
518           }
519           return registry.getNativeType(JSTypeNative.UNKNOWN_TYPE);
520         }
521     
522         @Override
523         public JSType getAssertedNewType(Node call, DeclaredTypeRegistry scope) {
524           if (call.getChildCount() > 2) {
525             Node constructor = call.getSecondChild().getNext();
526             if (constructor != null && constructor.isQualifiedName()) {
527               QualifiedName qname = QualifiedName.fromNode(constructor);
528               JSType functionType = scope.getDeclaredTypeOf(qname.getLeftmostName());
529               if (functionType != null) {
530                 if (!qname.isIdentifier()) {
531                   functionType = functionType.getProp(qname.getAllButLeftmost());
532                 }
533                 com.google.javascript.jscomp.newtypes.FunctionType ctorType =
534                     functionType.getFunTypeIfSingletonObj();
535                 if (ctorType != null && ctorType.isUniqueConstructor()) {
536                   return ctorType.getInstanceTypeOfCtor();
537                 }
538               }
539             }
540           }
541           return JSType.UNKNOWN;
542         }
543       }
