17     package com.google.javascript.jscomp;
18     
19     import com.google.common.base.Preconditions;
20     import com.google.javascript.rhino.IR;
21     import com.google.javascript.rhino.Node;
22     
23     /**
24      * A pass that looks for assignments to properties of an object or array
25      * immediately following its creation using the abbreviated syntax.
26      * <p>
27      * E.g. {@code var a = [];a[0] = 0} is optimized to {@code var a = [0]} and
28      * similarly for the object constructor.
29      *
30      */
31     final class PeepholeCollectPropertyAssignments extends AbstractPeepholeOptimization {
32     
33       @Override
34       Node optimizeSubtree(Node subtree) {
35         if (!subtree.isScript() && !subtree.isBlock()) {
36           return subtree;
37         }
38     
39         boolean codeChanged = false;
40     
41         // Look for variable declarations or simple assignments
42         // and start processing there.
43         for (Node child = subtree.getFirstChild();
44              child != null; child = child.getNext()) {
45           if (!child.isVar() && !NodeUtil.isExprAssign(child)) {
46             continue;
47           }
48           if (!isPropertyAssignmentToName(child.getNext())) {
49             // Quick check to see if there's anything to collapse.
50             continue;
51           }
52     
53           Preconditions.checkState(child.hasOneChild());
54           Node name = getName(child);
55           if (!name.isName()) {
56             // The assignment target is not a simple name.
57             continue;
58           }
59           Node value = getValue(child);
60           if (value == null || !isInterestingValue(value)) {
61             // No initializer or not an Object or Array literal.
62             continue;
63           }
64     
65           Node propertyCandidate;
66           while ((propertyCandidate = child.getNext()) != null) {
67             // This does not infinitely loop because collectProperty always
68             // removes propertyCandidate from its parent when it returns true.
69             if (!collectProperty(propertyCandidate, name.getString(), value)) {
70               break;
71             }
72             codeChanged = true;
73           }
74         }
75     
76         if (codeChanged) {
77           reportCodeChange();
78         }
79         return subtree;
80       }
81     
82       private static Node getName(Node n) {
83         if (n.isVar()) {
84           return n.getFirstChild();
85         } else if (NodeUtil.isExprAssign(n)) {
86           return n.getFirstFirstChild();
87         }
88         throw new IllegalStateException();
89       }
90     
91       private static Node getValue(Node n) {
92         if (n.isVar()) {
93           return n.getFirstFirstChild();
94         } else if (NodeUtil.isExprAssign(n)) {
95           return n.getFirstChild().getLastChild();
96         }
97         throw new IllegalStateException();
98       }
99     
100       static boolean isInterestingValue(Node n) {
101         return n.isObjectLit() || n.isArrayLit();
102       }
103     
104       private static boolean isPropertyAssignmentToName(Node propertyCandidate) {
105         if (propertyCandidate == null) { return false; }
106         // Must be an assignment...
107         if (!NodeUtil.isExprAssign(propertyCandidate)) {
108           return false;
109         }
110     
111         Node expr = propertyCandidate.getFirstChild();
112     
113         // to a property...
114         Node lhs = expr.getFirstChild();
115         if (!NodeUtil.isGet(lhs)) {
116           return false;
117         }
118     
119         // of a variable.
120         Node obj = lhs.getFirstChild();
121         return obj.isName();
122       }
123     
124       private boolean collectProperty(
125           Node propertyCandidate, String name, Node value) {
126         if (!isPropertyAssignmentToName(propertyCandidate)) {
127           return false;
128         }
129     
130         Node lhs = propertyCandidate.getFirstFirstChild();
131         // Must be an assignment to the recent variable...
132         if (!name.equals(lhs.getFirstChild().getString())) {
133           return false;
134         }
135     
136         Node rhs = lhs.getNext();
137         // with a value that cannot change the values of the variables,
138         if (mayHaveSideEffects(rhs)
139             || NodeUtil.canBeSideEffected(rhs)) {
140           return false;
141         }
142         // and does not have a reference to a variable initialized after it.
143         if (!NodeUtil.isLiteralValue(rhs, true)
144             && mightContainForwardReference(rhs, name)) {
145           return false;
146         }
147     
148         switch (value.getType()) {
149           case ARRAYLIT:
150             if (!collectArrayProperty(value, propertyCandidate)) {
151               return false;
152             }
153             break;
154           case OBJECTLIT:
155             if (!collectObjectProperty(value, propertyCandidate)) {
156               return false;
157             }
158             break;
159           default:
160             throw new IllegalStateException();
161         }
162         return true;
163       }

166       private static boolean collectArrayProperty(
167           Node arrayLiteral, Node propertyCandidate) {
168         Node assignment = propertyCandidate.getFirstChild();
169         final int sizeOfArrayAtStart = arrayLiteral.getChildCount();
170         int maxIndexAssigned = sizeOfArrayAtStart - 1;
171     
172         Node lhs = assignment.getFirstChild();
173         Node rhs = lhs.getNext();
174         if (!lhs.isGetElem()) {
175           return false;
176         }
177         Node obj = lhs.getFirstChild();
178         Node property = obj.getNext();
179         // The left hand side must have a numeric index
180         if (!property.isNumber()) {
181           return false;
182         }
183         // that is a valid array index
184         double dindex = property.getDouble();
185         if (!(dindex >= 0)  // Handles NaN and negatives.
186             || Double.isInfinite(dindex) || dindex > 0x7fffffffL) {
187           return false;
188         }
189         int index = (int) dindex;
190         if (dindex != index) {
191           return false;
192         }
193         // that would not make the array so sparse that they take more space
194         // when rendered than x[9]=1.
195         if (maxIndexAssigned + 4 < index) {
196           return false;
197         }
198         if (index > maxIndexAssigned) {
199           while (maxIndexAssigned < index - 1) {
200             // Pad the array if it is sparse.
201             // So if array is [0] and integer 3 is assigned at index is 2, then
202             // we want to produce [0,,2].
203             Node emptyNode = IR.empty().srcref(arrayLiteral);
204             arrayLiteral.addChildToBack(emptyNode);
205             ++maxIndexAssigned;
206           }
207           arrayLiteral.addChildToBack(rhs.detachFromParent());
208         } else {
209           // An out of order assignment.  Allow it if it's a hole.
210           Node currentValue = arrayLiteral.getChildAtIndex(index);
211           if (!currentValue.isEmpty()) {
212             // We've already collected a value for this index.
213             return false;
214           }
215           arrayLiteral.replaceChild(currentValue, rhs.detachFromParent());
216         }
217     
218         propertyCandidate.detachFromParent();
219         return true;
220       }

222       private static boolean collectObjectProperty(
223           Node objectLiteral, Node propertyCandidate) {
224         Node assignment = propertyCandidate.getFirstChild();
225         Node lhs = assignment.getFirstChild(), rhs = lhs.getNext();
226         Node obj = lhs.getFirstChild();
227         Node property = obj.getNext();
228     
229         // The property must be statically known.
230         if (lhs.isGetElem()
231             && (!property.isString()
232                 && !property.isNumber())) {
233           return false;
234         }
235     
236         String propertyName;
237         if (property.isNumber()) {
238           propertyName = NodeUtil.getStringValue(property);
239         } else {
240           propertyName = property.getString();
241         }
242     
243         Node newProperty = IR.stringKey(propertyName)
244             .useSourceInfoIfMissingFrom(property);
245         // Preserve the quotedness of a property reference
246         if (lhs.isGetElem()) {
247           newProperty.setQuotedString();
248         }
249         Node newValue = rhs.detachFromParent();
250         newProperty.addChildToBack(newValue);
251         // Check if the new property already exists in the object literal
252         // Note: Duplicate keys are invalid in strict mode
253         boolean propertyExists = false;
254         for (Node currentProperty : objectLiteral.children()) {
255           // Get the name of the current property
256           String currentPropertyName = currentProperty.getString();
257           // Get the value of the property
258           Node currentValue = currentProperty.getFirstChild();
259           // Compare the current property name with the new property name
260           if (currentPropertyName.equals(propertyName)) {
261             propertyExists = true;
262             // Check if the current value and the new value are side-effect
263             boolean isCurrentValueSideEffect = NodeUtil.canBeSideEffected(currentValue);
264             boolean isNewValueSideEffect = NodeUtil.canBeSideEffected(newValue);
265             // If they are side-effect free then replace the current value with the new one
266             if (!isCurrentValueSideEffect && !isNewValueSideEffect) {
267               objectLiteral.removeChild(currentProperty);
268               objectLiteral.addChildToBack(newProperty);
269               propertyCandidate.detachFromParent();
270               return true;
271             }
272             // Break the loop if the property exists
273             break;
274           }
275         }
276         // If the property does not already exist we can safely add it
277         if (!propertyExists) {
278           objectLiteral.addChildToBack(newProperty);
279         }
280         propertyCandidate.detachFromParent();
281         return true;
282       }

285       private static boolean mightContainForwardReference(
286           Node node, String varName) {
287         if (node.isName()) {
288           return varName.equals(node.getString());
289         }
290         for (Node child = node.getFirstChild(); child != null;
291              child = child.getNext()) {
292           if (mightContainForwardReference(child, varName)) {
293             return true;
294           }
295         }
296         return false;
297       }
