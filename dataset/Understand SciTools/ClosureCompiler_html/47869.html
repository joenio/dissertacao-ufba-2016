36     /**
37      * A {@link Compiler} pass for moving code to a deeper module if possible.
38      * - currently it only moves functions + variables
39      *
40      */
41     class CrossModuleCodeMotion implements CompilerPass {
42     
43       private static final Logger logger =
44           Logger.getLogger(CrossModuleCodeMotion.class.getName());
45     
46       private final AbstractCompiler compiler;
47       private final JSModuleGraph graph;

49       /**
50        * Map from module to the node in that module that should parent any string
51        * variable declarations that have to be moved into that module
52        */
53       private final Map<JSModule, Node> moduleVarParentMap =
54           new HashMap<>();
55     
56       /*
57        * NOTE - I made this a LinkedHashMap to make testing easier. With a regular
58        * HashMap, the variables may not output in a consistent order
59        */
60       private final Map<Var, NamedInfo> namedInfo =
61           new LinkedHashMap<>();

63       private final Map<Node, InstanceofInfo> instanceofNodes =
64           new LinkedHashMap<>();

68       /**
69        * Creates an instance.
70        *
71        * @param compiler The compiler
72        */
73       CrossModuleCodeMotion(
74           AbstractCompiler compiler,
75           JSModuleGraph graph,
76           boolean parentModuleCanSeeSymbolsDeclaredInChildren) {
77         this.compiler = compiler;
78         this.graph = graph;
79         this.parentModuleCanSeeSymbolsDeclaredInChildren =
80             parentModuleCanSeeSymbolsDeclaredInChildren;
81       }

83       @Override
84       public void process(Node externs, Node root) {
85         logger.fine("Moving functions + variable into deeper modules");
86     
87         // If there are <2 modules, then we will never move anything, so we're done
88         if (graph != null && graph.getModuleCount() > 1) {
89     
90           // Traverse the tree and find the modules where a var is declared + used
91           collectReferences(root);
92     
93           // Make is so we can ignore constructor references in instanceof.
94           if (parentModuleCanSeeSymbolsDeclaredInChildren) {
95             makeInstanceOfCodeOrderIndependent();
96           }
97     
98           // Move the functions + variables to a deeper module [if possible]
99           moveCode();
100         }
101       }
102     
103       /** move the code accordingly */
104       private void moveCode() {
105         for (NamedInfo info : namedInfo.values()) {
106           JSModule deepestDependency = info.deepestModule;
107     
108           // Only move if all are true:
109           // a) allowMove is true
110           // b) it was used + declared somewhere [if not, then it will be removed
111           // as dead or invalid code elsewhere]
112           // c) the new dependency depends on the declModule
113           if (info.allowMove && deepestDependency != null) {
114             Iterator<Declaration> it = info.declarationIterator();
115             JSModuleGraph moduleGraph = compiler.getModuleGraph();
116             while (it.hasNext()) {
117               Declaration decl = it.next();
118               if (decl.module != null &&
119                   moduleGraph.dependsOn(deepestDependency,
120                       decl.module)) {
121     
122                 // Find the appropriate spot to move it to
123                 Node destParent = moduleVarParentMap.get(deepestDependency);
124                 if (destParent == null) {
125                   destParent = compiler.getNodeForCodeInsertion(deepestDependency);
126                   moduleVarParentMap.put(deepestDependency, destParent);
127                 }
128     
129                 // VAR Nodes are normalized to have only one child.
130                 Node declParent = decl.node.getParent();
131                 Preconditions.checkState(
132                     !declParent.isVar() || declParent.hasOneChild(),
133                     "AST not normalized.");
134     
135                 // Remove it
136                 declParent.detachFromParent();
137     
138                 // Add it to the new spot
139                 destParent.addChildToFront(declParent);
140     
141                 compiler.reportCodeChange();
142               }
143             }
144           }
145         }
146       }

148       /** useful information for each variable candidate */
149       private class NamedInfo {
150         boolean allowMove = true;
151     
152         // The deepest module where the variable is used. Starts at null
153         private JSModule deepestModule = null;
154     
155         // The module where declarations appear
156         private JSModule declModule = null;
157     
158         // information on the spot where the item was declared
159         private final Deque<Declaration> declarations =
160             new ArrayDeque<>();
161     
162         // Add a Module where it is used
163         void addUsedModule(JSModule m) {
164           // If we are not allowed to move it, all bets are off
165           if (!allowMove) {
166             return;
167           }
168     
169           // If we have no deepest module yet, set this one
170           if (deepestModule == null) {
171             deepestModule = m;
172           } else {
173             // Find the deepest common dependency
174             deepestModule =
175                 graph.getDeepestCommonDependencyInclusive(m, deepestModule);
176           }
177         }
178     
179         boolean isUsedInOrDependencyOfModule(JSModule m) {
180           if (deepestModule == null || m == null) {
181             return false;
182           }
183           return m == deepestModule || graph.dependsOn(m, deepestModule);
184         }
185     
186         /**
187          * Add a declaration for this name.
188          * @return Whether this is a valid declaration. If this returns false,
189          *    this should be added as a reference.
190          */
191         boolean addDeclaration(Declaration d) {
192           // all declarations must appear in the same module.
193           if (declModule != null && d.module != declModule) {
194             return false;
195           }
196           declarations.push(d);
197           declModule = d.module;
198           return true;
199         }
200     
201         /**
202          * Returns an iterator over the declarations, in the order that they were
203          * declared.
204          */
205         Iterator<Declaration> declarationIterator() {
206           return declarations.iterator();
207         }
208       }

210       private static class Declaration {
211         final JSModule module;
212         final Node node;
213     
214         Declaration(JSModule module, Node node) {
215           this.module = module;
216           this.node = node;
217         }

220       /**
221        * return true if the node has any form of conditional in its ancestry
222        * TODO(nicksantos) keep track of the conditionals in the ancestry, so
223        * that we don't have to recrawl it.
224        */
225       private static boolean hasConditionalAncestor(Node n) {
226         for (Node ancestor : n.getAncestors()) {
227           switch (ancestor.getType()) {
228             case DO:
229             case FOR:
230             case HOOK:
231             case IF:
232             case SWITCH:
233             case WHILE:
234             case FUNCTION:
235               return true;
236           }
237         }
238         return false;
239       }

241       /**
242        * get the information on a variable
243        */
244       private NamedInfo getNamedInfo(Var v) {
245         NamedInfo info = namedInfo.get(v);
246         if (info == null) {
247           info = new NamedInfo();
248           namedInfo.put(v, info);
249         }
250         return info;
251       }

253       /**
254        * Process the reads to named variables
255        */
256       private void processRead(Reference ref, NamedInfo info) {
257         // A name is recursively defined if:
258         //   1: It is calling itself.
259         //   2: One of its property calls itself.
260         // Recursive definition should not block movement.
261         String name = ref.getNode().getString();
262         boolean recursive = false;
263         Scope hoistTarget = ref.getScope().getClosestHoistScope();
264         if (hoistTarget.isFunctionBlockScope()) {
265           Node rootNode = hoistTarget.getRootNode().getParent();
266           // CASE #1:
267           String scopeFuncName = rootNode.getFirstChild().getString();
268           Node scopeFuncParent = rootNode.getParent();
269           if (scopeFuncName.equals(name)) {
270             recursive = true;
271           } else if (scopeFuncParent.isName() &&
272               scopeFuncParent.getString().equals(name)) {
273             recursive = true;
274           } else {
275             // CASE #2:
276             // Suppose name is Foo, we keep look up the scope stack to look for
277             // a scope with "Foo.prototype.bar = function() { ..... "
278             for (Scope s = ref.getScope();
279                  s.getParent() != null; s = s.getParent()) {
280               Node curRoot = s.getRootNode();
281               if (curRoot.getParent().isAssign()) {
282                 Node owner = curRoot.getParent().getFirstChild();
283                 while (owner.isGetProp()) {
284                   owner = owner.getFirstChild();
285                 }
286                 if (owner.isName() &&
287                     owner.getString().equals(name)) {
288                   recursive = true;
289                   break;
290                 }
291               }
292             }
293           }
294         }
295     
296         if (!recursive) {
297           info.addUsedModule(getModule(ref));
298         }
299       }

301       private void collectReferences(Node root) {
302         ReferenceCollectingCallback collector = new ReferenceCollectingCallback(
303             compiler, ReferenceCollectingCallback.DO_NOTHING_BEHAVIOR,
304             new Predicate<Var>() {
305               @Override public boolean apply(Var var) {
306                 // Only collect global and non-exported names.
307                 return var.isGlobal() &&
308                     !compiler.getCodingConvention().isExported(var.getName());
309               }
310             });
311         NodeTraversal.traverseEs6(compiler, root, collector);
312     
313         for (Var v : collector.getAllSymbols()) {
314           ReferenceCollection refCollection = collector.getReferences(v);
315           NamedInfo info = getNamedInfo(v);
316           for (Reference ref : refCollection) {
317             processReference(collector, ref, info);
318           }
319         }
320       }
321     
322       private void processReference(
323           ReferenceCollectingCallback collector, Reference ref, NamedInfo info) {
324         Node n = ref.getNode();
325         Node parent = n.getParent();
326         if (info.allowMove) {
327           if (maybeProcessDeclaration(collector, ref, info)) {
328             // Check to see if the declaration is conditional starting at the
329             // grandparent of the name node. Since a function declaration
330             // is considered conditional (the function might not be called)
331             // we would need to skip the parent in this check as the name could
332             // just be a function itself.
333             if (hasConditionalAncestor(parent.getParent())) {
334               info.allowMove = false;
335             }
336           } else {
337             if (parentModuleCanSeeSymbolsDeclaredInChildren &&
338                 parent.isInstanceOf() && parent.getLastChild() == n) {
339               instanceofNodes.put(parent, new InstanceofInfo(getModule(ref), info));
340             } else {
341               // Otherwise, it's a read
342               processRead(ref, info);
343             }
344           }
345         }
346       }

348       private JSModule getModule(Reference ref) {
349         return compiler.getInput(ref.getInputId()).getModule();
350       }

352       /**
353        * Determines whether the given NAME node belongs to a declaration that
354        * can be moved across modules. If it is, registers it properly.
355        *
356        * There are four types of movable declarations:
357        * 1) var NAME = [movable object];
358        * 2) function NAME() {}
359        * 3) NAME = [movable object];
360        *    NAME.prop = [movable object];
361        *    NAME.prop.prop2 = [movable object];
362        *    etc.
363        * 4) Class-defining function calls, like "inherits" and "mixin".
364        *    NAME.inherits([some other name]);
365        * where "movable object" is a literal or a function.
366        */
367       private boolean maybeProcessDeclaration(
368           ReferenceCollectingCallback collector, Reference ref, NamedInfo info) {
369         Node name = ref.getNode();
370         Node parent = name.getParent();
371         Node grandparent = parent.getParent();
372         switch (parent.getType()) {
373           case VAR:
374             if (canMoveValue(collector, ref.getScope(), name.getFirstChild())) {
375               return info.addDeclaration(
376                   new Declaration(getModule(ref), name));
377             }
378             return false;
379     
380           case FUNCTION:
381             if (NodeUtil.isFunctionDeclaration(parent)) {
382               return info.addDeclaration(
383                   new Declaration(getModule(ref), name));
384             }
385             return false;
386     
387           case ASSIGN:
388           case GETPROP:
389             Node child = name;
390     
391             // Look for assignment expressions where the name is the root
392             // of a qualified name on the left hand side of the assignment.
393             for (Node current : name.getAncestors()) {
394               if (current.isGetProp()) {
395                 // fallthrough
396               } else if (current.isAssign() &&
397                          current.getFirstChild() == child) {
398                 Node currentParent = current.getParent();
399                 if (currentParent.isExprResult() &&
400                     canMoveValue(
401                         collector, ref.getScope(), current.getLastChild())) {
402                   return info.addDeclaration(
403                       new Declaration(getModule(ref), current));
404                 }
405               } else {
406                 return false;
407               }
408     
409               child = current;
410             }
411             return false;
412     
413           case CALL:
414             if (NodeUtil.isExprCall(grandparent)) {
415               SubclassRelationship relationship =
416                   compiler.getCodingConvention().getClassesDefinedByCall(parent);
417               if (relationship != null &&
418                   name.getString().equals(relationship.subclassName)) {
419                 return info.addDeclaration(
420                     new Declaration(getModule(ref), parent));
421               }
422             }
423             return false;
424     
425           default:
426             return false;
427         }
428       }

430       /**
431        * Determines whether the given value is eligible to be moved across modules.
432        */
433       private static boolean canMoveValue(
434           ReferenceCollectingCallback collector, Scope scope, Node n) {
435         // the value is only movable if it's
436         // a) nothing,
437         // b) a constant literal,
438         // c) a function, or
439         // d) an array/object literal of movable values.
440         // e) a function stub generated by CrossModuleMethodMotion.
441         if (n == null || NodeUtil.isLiteralValue(n, true) ||
442             n.isFunction()) {
443           return true;
444         } else if (n.isCall()) {
445           Node functionName = n.getFirstChild();
446           return functionName.isName() &&
447               (functionName.getString().equals(
448                   CrossModuleMethodMotion.STUB_METHOD_NAME) ||
449                functionName.getString().equals(
450                   CrossModuleMethodMotion.UNSTUB_METHOD_NAME));
451         } else if (n.isArrayLit() || n.isObjectLit()) {
452           boolean isObjectLit = n.isObjectLit();
453           for (Node child = n.getFirstChild(); child != null;
454                child = child.getNext()) {
455             if (!canMoveValue(collector, scope,
456                               isObjectLit ? child.getFirstChild() : child)) {
457               return false;
458             }
459           }
460     
461           return true;
462         } else if (n.isName()) {
463           // If the value is guaranteed to never be changed after
464           // this reference, then we can move it.
465           Var v = scope.getVar(n.getString());
466           if (v != null && v.isGlobal()) {
467             ReferenceCollection refCollection = collector.getReferences(v);
468             if (refCollection != null &&
469                 refCollection.isWellDefined() &&
470                 refCollection.isAssignedOnceInLifetime()) {
471               return true;
472             }
473           }
474         }
475     
476         return false;
477       }

479       /**
480        * Transforms instanceof usages into an expression that short circuits to
481        * false if tested with a constructor that is undefined. This allows ignoring
482        * instanceof with respect to cross module code motion.
483        */
484       private void makeInstanceOfCodeOrderIndependent() {
485         Node tmp = IR.block();
486         for (Map.Entry<Node, InstanceofInfo> entry : instanceofNodes.entrySet()) {
487           Node n = entry.getKey();
488           InstanceofInfo info = entry.getValue();
489           if (!info.namedInfo.allowMove || !info.mustBeGuardedByTypeof()) {
490             continue;
491           }
492           // In order for the compiler pass to be idempotent, this checks whether
493           // the instanceof is already wrapped in the code that is generated below.
494           Node parent = n.getParent();
495           if (parent.isAnd() && parent.getLastChild() == n
496               && parent.getFirstChild().isNE()) {
497             Node ne = parent.getFirstChild();
498             if (ne.getFirstChild().isString()
499                 && "undefined".equals(ne.getFirstChild().getString())
500                 && ne.getLastChild().isTypeOf()) {
501               Node ref = ne.getLastChild().getFirstChild();
502               if (ref.isEquivalentTo(n.getLastChild())) {
503                 continue;
504               }
505             }
506           }
507           // Wrap "foo instanceof Bar" in
508           // "('undefined' != typeof Bar && foo instanceof Bar)"
509           Node reference = n.getLastChild().cloneNode();
510           Preconditions.checkState(reference.isName());
511           n.getParent().replaceChild(n, tmp);
512           Node and = IR.and(
513               new Node(Token.NE,
514                   IR.string("undefined"),
515                   new Node(Token.TYPEOF, reference)
516               ),
517               n
518           );
519           and.useSourceInfoIfMissingFromForTree(n);
520           tmp.getParent().replaceChild(tmp, and);
521           compiler.reportCodeChange();
522         }
523       }

525       private static class InstanceofInfo {
526         private final JSModule module;
527         private final NamedInfo namedInfo;
528     
529         InstanceofInfo(JSModule module, NamedInfo namedInfo) {
530           this.module = module;
531           this.namedInfo = namedInfo;
532         }
533     
534         /**
535          * Returns true if this instance of instanceof is in a deeper module than
536          * the deepest module (by reference) of the related name.
537          * In that case the name may be undefined when the instanceof runs and we
538          * have to guard it with typeof.
539          */
540         boolean mustBeGuardedByTypeof() {
541           return !this.namedInfo.isUsedInOrDependencyOfModule(this.module);
542         }
