1130           implements AliasTransformationHandler {
1131     
1132         private final Map<String, List<SourcePosition<AliasTransformation>>> observedPositions =
1133             new HashMap<>();
1134     
1135         public final List<AliasTransformation> constructedAliases =
1136              new ArrayList<>();
1137     
1138         @Override
1139         public AliasTransformation logAliasTransformation(
1140             String sourceFile, SourcePosition<AliasTransformation> position) {
1141           if (!observedPositions.containsKey(sourceFile)) {
1142             observedPositions.put(sourceFile,
1143                  new ArrayList<SourcePosition<AliasTransformation>>());
1144           }
1145           observedPositions.get(sourceFile).add(position);
1146           AliasTransformation spy = new AliasSpy();
1147           constructedAliases.add(spy);
1148           return spy;
1149         }
1150       }

1161       private static class TypeVerifyingPass
1162           implements CompilerPass, NodeTraversal.Callback {
1163         private final Compiler compiler;
1164         private List<Node> actualTypes = null;
1165     
1166         public TypeVerifyingPass(Compiler compiler) {
1167           this.compiler = compiler;
1168         }
1169     
1170         @Override
1171         public void process(Node externs, Node root) {
1172           NodeTraversal.traverseEs6(compiler, root, this);
1173         }
1174     
1175         @Override
1176         public boolean shouldTraverse(NodeTraversal nodeTraversal, Node n,
1177             Node parent) {
1178           return true;
1179         }
1180     
1181         @Override
1182         public void visit(NodeTraversal t, Node n, Node parent) {
1183           JSDocInfo info = n.getJSDocInfo();
1184           if (info != null) {
1185             Collection<Node> typeNodes = info.getTypeNodes();
