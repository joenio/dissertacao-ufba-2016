17     package com.google.javascript.jscomp;
18     
19     import com.google.common.annotations.GwtIncompatible;
20     
21     import java.io.PrintStream;
22     
23     /**
24      * <p>An error manager that prints errors and warnings to the print stream
25      * provided in addition to the functionality of the
26      * {@link BasicErrorManager}.</p>
27      *
28      * <p>It collaborates with a {@link SourceExcerptProvider} via a
29      * {@link MessageFormatter} to display error messages with source context.</p>
30      *
31      */
32     @GwtIncompatible("java.io.PrintStream")
33     public class PrintStreamErrorManager extends BasicErrorManager {
34       private final MessageFormatter formatter;
35       private final PrintStream stream;
36       private int summaryDetailLevel = 1;
37     
38       /**
39        * Creates an error manager.
40        * @param formatter the message formatter used to format the messages
41        * @param stream the stream on which the errors and warnings should be
42        *     printed. This class does not close the stream
43        */
44       public PrintStreamErrorManager(MessageFormatter formatter,
45                                      PrintStream stream) {
46         this.formatter = formatter;
47         this.stream = stream;
48       }
49     
50       /**
51        * Creates an instance with a source-less error formatter.
52        */
53       public PrintStreamErrorManager(PrintStream stream) {
54         this(ErrorFormat.SOURCELESS.toFormatter(null, false), stream);
55       }
56     
57       @Override
58       public void println(CheckLevel level, JSError error) {
59         stream.println(error.format(level, formatter));
60       }
61     
62       public void setSummaryDetailLevel(int summaryDetailLevel) {
63         this.summaryDetailLevel = summaryDetailLevel;
64       }
65     
66       @Override
67       public void printSummary() {
68         if (summaryDetailLevel >= 3 ||
69             (summaryDetailLevel >= 1 && getErrorCount() + getWarningCount() > 0) ||
70             (summaryDetailLevel >= 2 && getTypedPercent() > 0.0)) {
71           if (getTypedPercent() > 0.0) {
72             stream.format("%d error(s), %d warning(s), %.1f%% typed%n",
73                 getErrorCount(), getWarningCount(), getTypedPercent());
74           } else {
75             stream.format("%d error(s), %d warning(s)%n", getErrorCount(),
76                 getWarningCount());
77           }
78         }
79       }
80     }
