81       @Override
82       Node optimizeSubtree(Node subtree) {
83         switch(subtree.getType()) {
84           case CALL:
85             return tryFoldCall(subtree);
86           case NEW:
87             return tryFoldCtorCall(subtree);
88     
89           case TYPEOF:
90             return tryFoldTypeof(subtree);
91     
92           case NOT:
93           case POS:
94           case NEG:
95           case BITNOT:
96             tryReduceOperandsForOp(subtree);
97             return tryFoldUnaryOperator(subtree);
98     
99           case VOID:
100             return tryReduceVoid(subtree);
101     
102           default:
103             tryReduceOperandsForOp(subtree);
104             return tryFoldBinaryOperator(subtree);
105         }
106       }

108       private Node tryFoldBinaryOperator(Node subtree) {
109         Node left = subtree.getFirstChild();
110     
111         if (left == null) {
112           return subtree;
113         }
114     
115         Node right = left.getNext();
116     
117         if (right == null) {
118           return subtree;
119         }
120     
121         // If we've reached here, node is truly a binary operator.
122         switch(subtree.getType()) {
123           case GETPROP:
124             return tryFoldGetProp(subtree, left, right);
125     
126           case GETELEM:
127             return tryFoldGetElem(subtree, left, right);
128     
129           case INSTANCEOF:
130             return tryFoldInstanceof(subtree, left, right);
131     
132           case AND:
133           case OR:
134             return tryFoldAndOr(subtree, left, right);
135     
136           case LSH:
137           case RSH:
138           case URSH:
139             return tryFoldShift(subtree, left, right);
140     
141           case ASSIGN:
142             return tryFoldAssign(subtree, left, right);
143     
144           case ASSIGN_BITOR:
145           case ASSIGN_BITXOR:
146           case ASSIGN_BITAND:
147           case ASSIGN_LSH:
148           case ASSIGN_RSH:
149           case ASSIGN_URSH:
150           case ASSIGN_ADD:
151           case ASSIGN_SUB:
152           case ASSIGN_MUL:
153           case ASSIGN_DIV:
154           case ASSIGN_MOD:
155             return tryUnfoldAssignOp(subtree, left, right);
156     
157           case ADD:
158             return tryFoldAdd(subtree, left, right);
159     
160           case SUB:
161           case DIV:
162           case MOD:
163             return tryFoldArithmeticOp(subtree, left, right);
164     
165           case MUL:
166           case BITAND:
167           case BITOR:
168           case BITXOR:
169             Node result = tryFoldArithmeticOp(subtree, left, right);
170             if (result != subtree) {
171               return result;
172             }
173             return tryFoldLeftChildOp(subtree, left, right);
174     
175           case LT:
176           case GT:
177           case LE:
178           case GE:
179           case EQ:
180           case NE:
181           case SHEQ:
182           case SHNE:
183             return tryFoldComparison(subtree, left, right);
184     
185           default:
186             return subtree;
187         }
188       }

190       private Node tryReduceVoid(Node n) {
191         Node child = n.getFirstChild();
192         if ((!child.isNumber() || child.getDouble() != 0.0) && !mayHaveSideEffects(n)) {
193           n.replaceChild(child, IR.number(0));
194           reportCodeChange();
195         }
196         return n;
197       }

199       private void tryReduceOperandsForOp(Node n) {
200         switch (n.getType()) {
201           case ADD:
202             Node left = n.getFirstChild();
203             Node right = n.getLastChild();
204             if (!NodeUtil.mayBeString(left, shouldUseTypes)
205                 && !NodeUtil.mayBeString(right, shouldUseTypes)) {
206               tryConvertOperandsToNumber(n);
207             }
208             break;
209           case ASSIGN_BITOR:
210           case ASSIGN_BITXOR:
211           case ASSIGN_BITAND:
212             // TODO(johnlenz): convert these to integers.
213           case ASSIGN_LSH:
214           case ASSIGN_RSH:
215           case ASSIGN_URSH:
216           case ASSIGN_SUB:
217           case ASSIGN_MUL:
218           case ASSIGN_MOD:
219           case ASSIGN_DIV:
220             tryConvertToNumber(n.getLastChild());
221             break;
222           case BITNOT:
223           case BITOR:
224           case BITXOR:
225           case BITAND:
226           case LSH:
227           case RSH:
228           case URSH:
229           case SUB:
230           case MUL:
231           case MOD:
232           case DIV:
233           case POS:
234           case NEG:
235             tryConvertOperandsToNumber(n);
236             break;
237         }
238       }

240       private void tryConvertOperandsToNumber(Node n) {
241         Node next;
242         for (Node c = n.getFirstChild(); c != null; c = next) {
243           next = c.getNext();
244           tryConvertToNumber(c);
245         }
246       }

248       private void tryConvertToNumber(Node n) {
249         switch (n.getType()) {
250           case NUMBER:
251             // Nothing to do
252             return;
253           case AND:
254           case OR:
255           case COMMA:
256             tryConvertToNumber(n.getLastChild());
257             return;
258           case HOOK:
259             tryConvertToNumber(n.getSecondChild());
260             tryConvertToNumber(n.getLastChild());
261             return;
262           case NAME:
263             if (!NodeUtil.isUndefined(n)) {
264               return;
265             }
266             break;
267         }
268     
269         Double result = NodeUtil.getNumberValue(n, shouldUseTypes);
270         if (result == null) {
271           return;
272         }
273     
274         double value = result;
275     
276         Node replacement = NodeUtil.numberNode(value, n);
277         if (replacement.isEquivalentTo(n)) {
278           return;
279         }
280     
281         n.getParent().replaceChild(n, replacement);
282         reportCodeChange();
283       }

285       /**
286        * Folds 'typeof(foo)' if foo is a literal, e.g.
287        * typeof("bar") --> "string"
288        * typeof(6) --> "number"
289        */
290       private Node tryFoldTypeof(Node originalTypeofNode) {
291         Preconditions.checkArgument(originalTypeofNode.isTypeOf());
292     
293         Node argumentNode = originalTypeofNode.getFirstChild();
294         if (argumentNode == null || !NodeUtil.isLiteralValue(argumentNode, true)) {
295           return originalTypeofNode;
296         }
297     
298         String typeNameString = null;
299     
300         switch (argumentNode.getType()) {
301           case FUNCTION:
302             typeNameString = "function";
303             break;
304           case STRING:
305             typeNameString = "string";
306             break;
307           case NUMBER:
308             typeNameString = "number";
309             break;
310           case TRUE:
311           case FALSE:
312             typeNameString = "boolean";
313             break;
314           case NULL:
315           case OBJECTLIT:
316           case ARRAYLIT:
317             typeNameString = "object";
318             break;
319           case VOID:
320             typeNameString = "undefined";
321             break;
322           case NAME:
323             // We assume here that programs don't change the value of the
324             // keyword undefined to something other than the value undefined.
325             if ("undefined".equals(argumentNode.getString())) {
326               typeNameString = "undefined";
327             }
328             break;
329         }
330     
331         if (typeNameString != null) {
332           Node newNode = IR.string(typeNameString);
333           originalTypeofNode.getParent().replaceChild(originalTypeofNode, newNode);
334           reportCodeChange();
335     
336           return newNode;
337         }
338     
339         return originalTypeofNode;
340       }

342       private Node tryFoldUnaryOperator(Node n) {
343         Preconditions.checkState(n.hasOneChild(), n);
344     
345         Node left = n.getFirstChild();
346         Node parent = n.getParent();
347     
348         if (left == null) {
349           return n;
350         }
351     
352         TernaryValue leftVal = NodeUtil.getPureBooleanValue(left);
353         if (leftVal == TernaryValue.UNKNOWN) {
354           return n;
355         }
356     
357         switch (n.getType()) {
358           case NOT:
359             // Don't fold !0 and !1 back to false.
360             if (late && left.isNumber()) {
361               double numValue = left.getDouble();
362               if (numValue == 0 || numValue == 1) {
363                 return n;
364               }
365             }
366             Node replacementNode = NodeUtil.booleanNode(!leftVal.toBoolean(true));
367             parent.replaceChild(n, replacementNode);
368             reportCodeChange();
369             return replacementNode;
370           case POS:
371             if (NodeUtil.isNumericResult(left)) {
372               // POS does nothing to numeric values.
373               parent.replaceChild(n, left.detachFromParent());
374               reportCodeChange();
375               return left;
376             }
377             return n;
378           case NEG:
379             if (left.isName()) {
380               if (left.getString().equals("Infinity")) {
381                 // "-Infinity" is valid and a literal, don't modify it.
382                 return n;
383               } else if (left.getString().equals("NaN")) {
384                 // "-NaN" is "NaN".
385                 n.removeChild(left);
386                 parent.replaceChild(n, left);
387                 reportCodeChange();
388                 return left;
389               }
390             }
391     
392             if (left.isNumber()) {
393               double negNum = -left.getDouble();
394     
395               Node negNumNode = IR.number(negNum);
396               parent.replaceChild(n, negNumNode);
397               reportCodeChange();
398               return negNumNode;
399             } else {
400               // left is not a number node, so do not replace, but warn the
401               // user because they can't be doing anything good
402               report(NEGATING_A_NON_NUMBER_ERROR, left);
403               return n;
404             }
405           case BITNOT:
406             try {
407               double val = left.getDouble();
408               if (val >= Integer.MIN_VALUE && val <= Integer.MAX_VALUE) {
409                 int intVal = (int) val;
410                 if (intVal == val) {
411                   Node notIntValNode = IR.number(~intVal);
412                   parent.replaceChild(n, notIntValNode);
413                   reportCodeChange();
414                   return notIntValNode;
415                 } else {
416                   report(FRACTIONAL_BITWISE_OPERAND, left);
417                   return n;
418                 }
419               } else {
420                 report(BITWISE_OPERAND_OUT_OF_RANGE, left);
421                 return n;
422               }
423             } catch (UnsupportedOperationException ex) {
424               // left is not a number node, so do not replace, but warn the
425               // user because they can't be doing anything good
426               report(NEGATING_A_NON_NUMBER_ERROR, left);
427               return n;
428             }
429             default:
430               return n;
431         }
432       }

434       /**
435        * Try to fold {@code left instanceof right} into {@code true}
436        * or {@code false}.
437        */
438       private Node tryFoldInstanceof(Node n, Node left, Node right) {
439         Preconditions.checkArgument(n.isInstanceOf());
440     
441         // TODO(johnlenz) Use type information if available to fold
442         // instanceof.
443         if (NodeUtil.isLiteralValue(left, true)
444             && !mayHaveSideEffects(right)) {
445     
446           Node replacementNode = null;
447     
448           if (NodeUtil.isImmutableValue(left)) {
449             // Non-object types are never instances.
450             replacementNode = IR.falseNode();
451           } else if (right.isName()
452               && "Object".equals(right.getString())) {
453             replacementNode = IR.trueNode();
454           }
455     
456           if (replacementNode != null) {
457             n.getParent().replaceChild(n, replacementNode);
458             reportCodeChange();
459             return replacementNode;
460           }
461         }
462     
463         return n;
464       }

466       private Node tryFoldAssign(Node n, Node left, Node right) {
467         Preconditions.checkArgument(n.isAssign());
468     
469         if (!late) {
470           return n;
471         }
472     
473         // Tries to convert x = x + y -> x += y;
474         if (!right.hasChildren() ||
475             right.getSecondChild() != right.getLastChild()) {
476           // RHS must have two children.
477           return n;
478         }
479     
480         if (mayHaveSideEffects(left)) {
481           return n;
482         }
483     
484         Node newRight;
485         if (areNodesEqualForInlining(left, right.getFirstChild())) {
486           newRight = right.getLastChild();
487         } else if (NodeUtil.isCommutative(right.getType()) &&
488               areNodesEqualForInlining(left, right.getLastChild())) {
489           newRight = right.getFirstChild();
490         } else {
491           return n;
492         }
493     
494         Token newType = null;
495         switch (right.getType()) {
496           case ADD:
497             newType = Token.ASSIGN_ADD;
498             break;
499           case BITAND:
500             newType = Token.ASSIGN_BITAND;
501             break;
502           case BITOR:
503             newType = Token.ASSIGN_BITOR;
504             break;
505           case BITXOR:
506             newType = Token.ASSIGN_BITXOR;
507             break;
508           case DIV:
509             newType = Token.ASSIGN_DIV;
510             break;
511           case LSH:
512             newType = Token.ASSIGN_LSH;
513             break;
514           case MOD:
515             newType = Token.ASSIGN_MOD;
516             break;
517           case MUL:
518             newType = Token.ASSIGN_MUL;
519             break;
520           case RSH:
521             newType = Token.ASSIGN_RSH;
522             break;
523           case SUB:
524             newType = Token.ASSIGN_SUB;
525             break;
526           case URSH:
527             newType = Token.ASSIGN_URSH;
528             break;
529           default:
530             return n;
531         }
532     
533         Node newNode = new Node(newType,
534             left.detachFromParent(), newRight.detachFromParent());
535         n.getParent().replaceChild(n, newNode);
536     
537         reportCodeChange();
538     
539         return newNode;
540       }

542       private Node tryUnfoldAssignOp(Node n, Node left, Node right) {
543         if (late) {
544           return n;
545         }
546     
547         if (!n.hasChildren() ||
548             n.getSecondChild() != n.getLastChild()) {
549           return n;
550         }
551     
552         if (mayHaveSideEffects(left)) {
553           return n;
554         }
555     
556         // Tries to convert x += y -> x = x + y;
557         Token op = NodeUtil.getOpFromAssignmentOp(n);
558         Node replacement = IR.assign(left.detachFromParent(),
559             new Node(op, left.cloneTree(), right.detachFromParent())
560                 .srcref(n));
561         n.getParent().replaceChild(n, replacement);
562         reportCodeChange();
563     
564         return replacement;
565       }

567       /**
568        * Try to fold a AND/OR node.
569        */
570       private Node tryFoldAndOr(Node n, Node left, Node right) {
571         Node parent = n.getParent();
572     
573         Node result = null;
574     
575         Token type = n.getType();
576     
577         TernaryValue leftVal = NodeUtil.getImpureBooleanValue(left);
578     
579         if (leftVal != TernaryValue.UNKNOWN) {
580           boolean lval = leftVal.toBoolean(true);
581     
582           // (TRUE || x) => TRUE (also, (3 || x) => 3)
583           // (FALSE && x) => FALSE
584           if (lval && type == Token.OR ||
585               !lval && type == Token.AND) {
586             result = left;
587     
588           } else if (!mayHaveSideEffects(left)) {
589             // (FALSE || x) => x
590             // (TRUE && x) => x
591             result = right;
592           } else {
593             // Left side may have side effects, but we know its boolean value.
594             // e.g. true_with_sideeffects || foo() => true_with_sideeffects, foo()
595             // or: false_with_sideeffects && foo() => false_with_sideeffects, foo()
596             // This, combined with PeepholeRemoveDeadCode, helps reduce expressions
597             // like "x() || false || z()".
598             n.detachChildren();
599             result = IR.comma(left, right);
600           }
601         }
602     
603         // Note: Right hand side folding is handled by
604         // PeepholeMinimizeConditions#tryMinimizeCondition
605     
606         if (result != null) {
607           // Fold it!
608           n.detachChildren();
609           parent.replaceChild(n, result);
610           reportCodeChange();
611     
612           return result;
613         } else {
614           return n;
615         }
616       }

618       /**
619        * Expressions such as [foo() + 'a' + 'b'] generate parse trees
620        * where no node has two const children ((foo() + 'a') + 'b'), so
621        * tryFoldAdd() won't fold it -- tryFoldLeftChildAdd() will (for Strings).
622        * Specifically, it folds Add expressions where:
623        *  - The left child is also and add expression
624        *  - The right child is a constant value
625        *  - The left child's right child is a STRING constant.
626        */
627       private Node tryFoldChildAddString(Node n, Node left, Node right) {
628     
629         if (NodeUtil.isLiteralValue(right, false) &&
630             left.isAdd()) {
631     
632           Node ll = left.getFirstChild();
633           Node lr = ll.getNext();
634     
635           // Left's right child MUST be a string. We would not want to fold
636           // foo() + 2 + 'a' because we don't know what foo() will return, and
637           // therefore we don't know if left is a string concat, or a numeric add.
638           if (lr.isString()) {
639             String leftString = NodeUtil.getStringValue(lr);
640             String rightString = NodeUtil.getStringValue(right);
641             if (leftString != null && rightString != null) {
642               left.removeChild(ll);
643               String result = leftString + rightString;
644               n.replaceChild(left, ll);
645               n.replaceChild(right, IR.string(result));
646               reportCodeChange();
647               return n;
648             }
649           }
650         }
651     
652         if (NodeUtil.isLiteralValue(left, false) &&
653             right.isAdd()) {
654     
655           Node rl = right.getFirstChild();
656           Node rr = right.getLastChild();
657     
658           // Left's right child MUST be a string. We would not want to fold
659           // foo() + 2 + 'a' because we don't know what foo() will return, and
660           // therefore we don't know if left is a string concat, or a numeric add.
661           if (rl.isString()) {
662             String leftString = NodeUtil.getStringValue(left);
663             String rightString = NodeUtil.getStringValue(rl);
664             if (leftString != null && rightString != null) {
665               right.removeChild(rr);
666               String result = leftString + rightString;
667               n.replaceChild(right, rr);
668               n.replaceChild(left, IR.string(result));
669               reportCodeChange();
670               return n;
671             }
672           }
673         }
674     
675         return n;
676       }

681       private Node tryFoldAddConstantString(Node n, Node left, Node right) {

699        * Try to fold arithmetic binary operators
700        */
701       private Node tryFoldArithmeticOp(Node n, Node left, Node right) {
702         Node result = performArithmeticOp(n.getType(), left, right);
703         if (result != null) {
704           result.useSourceInfoIfMissingFromForTree(n);
705           n.getParent().replaceChild(n, result);
706           reportCodeChange();
707           return result;
708         }
709         return n;
710       }

712       /**
713        * Try to fold arithmetic binary operators
714        */
715       private Node performArithmeticOp(Token opType, Node left, Node right) {
716         // Unlike other operations, ADD operands are not always converted
717         // to Number.
718         if (opType == Token.ADD
719             && (NodeUtil.mayBeString(left, shouldUseTypes)
720                 || NodeUtil.mayBeString(right, shouldUseTypes))) {
721           return null;
722         }
723     
724         double result;
725     
726         // TODO(johnlenz): Handle NaN with unknown value. BIT ops convert NaN
727         // to zero so this is a little awkward here.
728     
729         Double lValObj = NodeUtil.getNumberValue(left, shouldUseTypes);
730         if (lValObj == null) {
731           return null;
732         }
733         Double rValObj = NodeUtil.getNumberValue(right, shouldUseTypes);
734         if (rValObj == null) {
735           return null;
736         }
737     
738         double lval = lValObj;
739         double rval = rValObj;
740     
741         switch (opType) {
742           case BITAND:
743             result = NodeUtil.toInt32(lval) & NodeUtil.toInt32(rval);
744             break;
745           case BITOR:
746             result = NodeUtil.toInt32(lval) | NodeUtil.toInt32(rval);
747             break;
748           case BITXOR:
749             result = NodeUtil.toInt32(lval) ^ NodeUtil.toInt32(rval);
750             break;
751           case ADD:
752             result = lval + rval;
753             break;
754           case SUB:
755             result = lval - rval;
756             break;
757           case MUL:
758             result = lval * rval;
759             break;
760           case MOD:
761             if (rval == 0) {
762               return null;
763             }
764             result = lval % rval;
765             break;
766           case DIV:
767             if (rval == 0) {
768               return null;
769             }
770             result = lval / rval;
771             break;
772           default:
773             throw new Error("Unexpected arithmetic operator");
774         }
775     
776         // TODO(johnlenz): consider removing the result length check.
777         // length of the left and right value plus 1 byte for the operator.
778         if ((String.valueOf(result).length() <=
779             String.valueOf(lval).length() + String.valueOf(rval).length() + 1
780     
781             // Do not try to fold arithmetic for numbers > 2^53. After that
782             // point, fixed-point math starts to break down and become inaccurate.
783             && Math.abs(result) <= MAX_FOLD_NUMBER)
784             || Double.isNaN(result)
785             || result == Double.POSITIVE_INFINITY
786             || result == Double.NEGATIVE_INFINITY) {
787           return NodeUtil.numberNode(result, null);
788         }
789         return null;
790       }

792       /**
793        * Expressions such as [foo() * 10 * 20] generate parse trees
794        * where no node has two const children ((foo() * 10) * 20), so
795        * performArithmeticOp() won't fold it -- tryFoldLeftChildOp() will.
796        * Specifically, it folds associative expressions where:
797        *  - The left child is also an associative expression of the same time.
798        *  - The right child is a constant NUMBER constant.
799        *  - The left child's right child is a NUMBER constant.
800        */
801       private Node tryFoldLeftChildOp(Node n, Node left, Node right) {
802         Token opType = n.getType();
803         Preconditions.checkState(
804             (NodeUtil.isAssociative(opType) && NodeUtil.isCommutative(opType))
805             || n.isAdd());
806     
807         Preconditions.checkState(!n.isAdd() || !NodeUtil.mayBeString(n, shouldUseTypes));
808     
809         // Use getNumberValue to handle constants like "NaN" and "Infinity"
810         // other values are converted to numbers elsewhere.
811         Double rightValObj = NodeUtil.getNumberValue(right, shouldUseTypes);
812         if (rightValObj != null && left.getType() == opType) {
813           Preconditions.checkState(left.getChildCount() == 2);
814     
815           Node ll = left.getFirstChild();
816           Node lr = ll.getNext();
817     
818           Node valueToCombine = ll;
819           Node replacement = performArithmeticOp(opType, valueToCombine, right);
820           if (replacement == null) {
821             valueToCombine = lr;
822             replacement = performArithmeticOp(opType, valueToCombine, right);
823           }
824           if (replacement != null) {
825             // Remove the child that has been combined
826             left.removeChild(valueToCombine);
827             // Replace the left op with the remaining child.
828             n.replaceChild(left, left.removeFirstChild());
829             // New "-Infinity" node need location info explicitly
830             // added.
831             replacement.useSourceInfoIfMissingFromForTree(right);
832             n.replaceChild(right, replacement);
833             reportCodeChange();
834           }
835         }
836     
837         return n;
838       }

840       private Node tryFoldAdd(Node node, Node left, Node right) {
841         Preconditions.checkArgument(node.isAdd());
842     
843         if (NodeUtil.mayBeString(node, shouldUseTypes)) {
844           if (NodeUtil.isLiteralValue(left, false) &&
845               NodeUtil.isLiteralValue(right, false)) {
846             // '6' + 7
847             return tryFoldAddConstantString(node, left, right);
848           } else {
849             // a + 7 or 6 + a
850             return tryFoldChildAddString(node, left, right);
851           }
852         } else {
853           // Try arithmetic add
854           Node result = tryFoldArithmeticOp(node, left, right);
855           if (result != node) {
856             return result;
857           }
858           return tryFoldLeftChildOp(node, left, right);
859         }
860       }

862       /**
863        * Try to fold shift operations
864        */
865       private Node tryFoldShift(Node n, Node left, Node right) {
866         if (left.isNumber() &&
867             right.isNumber()) {
868     
869           double result;
870           double lval = left.getDouble();
871           double rval = right.getDouble();
872     
873           // check ranges.  We do not do anything that would clip the double to
874           // a 32-bit range, since the user likely does not intend that.
875           if (lval < Integer.MIN_VALUE) {
876             report(BITWISE_OPERAND_OUT_OF_RANGE, left);
877             return n;
878           }
879           // only the lower 5 bits are used when shifting, so don't do anything
880           // if the shift amount is outside [0,32)
881           if (!(rval >= 0 && rval < 32)) {
882             report(SHIFT_AMOUNT_OUT_OF_BOUNDS, n);
883             return n;
884           }
885     
886           int rvalInt = (int) rval;
887           if (rvalInt != rval) {
888             report(FRACTIONAL_BITWISE_OPERAND, right);
889             return n;
890           }
891     
892           switch (n.getType()) {
893             case LSH:
894             case RSH:
895               // Convert the numbers to ints
896               if (lval > Integer.MAX_VALUE) {
897                 report(BITWISE_OPERAND_OUT_OF_RANGE, left);
898                 return n;
899               }
900               int lvalInt = (int) lval;
901               if (lvalInt != lval) {
902                 report(FRACTIONAL_BITWISE_OPERAND, left);
903                 return n;
904               }
905               if (n.getType() == Token.LSH) {
906                 result = lvalInt << rvalInt;
907               } else {
908                 result = lvalInt >> rvalInt;
909               }
910               break;
911             case URSH:
912               // JavaScript handles zero shifts on signed numbers differently than
913               // Java as an Java int can not represent the unsigned 32-bit number
914               // where JavaScript can so use a long here.
915               long maxUint32 = 0xffffffffL;
916               if (lval > maxUint32) {
917                 report(BITWISE_OPERAND_OUT_OF_RANGE, left);
918                 return n;
919               }
920               long lvalLong = (long) lval;
921               if (lvalLong != lval) {
922                 report(FRACTIONAL_BITWISE_OPERAND, left);
923                 return n;
924               }
925               result = (lvalLong & maxUint32) >>> rvalInt;
926               break;
927             default:
928               throw new AssertionError("Unknown shift operator: " + n.getType());
929           }
930     
931           Node newNumber = IR.number(result);
932           n.getParent().replaceChild(n, newNumber);
933           reportCodeChange();
934     
935           return newNumber;
936         }
937     
938         return n;
939       }

942        * Try to fold comparison nodes, e.g ==
943        */
944       private Node tryFoldComparison(Node n, Node left, Node right) {
945         TernaryValue result = evaluateComparison(n.getType(), left, right, shouldUseTypes);
946         if (result == TernaryValue.UNKNOWN) {
947           return n;
948         }
949     
950         Node newNode = NodeUtil.booleanNode(result.toBoolean(true));
951         n.getParent().replaceChild(n, newNode);
952         reportCodeChange();
953     
954         return newNode;
955       }

957       /** http://www.ecma-international.org/ecma-262/6.0/#sec-abstract-relational-comparison */
958       private static TernaryValue tryAbstractRelationalComparison(Node left, Node right,
959           boolean useTypes, boolean willNegate) {
960         // First, try to evaluate based on the general type.
961         ValueType leftValueType = NodeUtil.getKnownValueType(left);
962         ValueType rightValueType = NodeUtil.getKnownValueType(right);
963         if (leftValueType != ValueType.UNDETERMINED && rightValueType != ValueType.UNDETERMINED) {
964           if (leftValueType == ValueType.STRING && rightValueType == ValueType.STRING) {
965             String lv = NodeUtil.getStringValue(left);
966             String rv = NodeUtil.getStringValue(right);
967             if (lv != null && rv != null) {
968               // In JS, browsers parse \v differently. So do not compare strings if one contains \v.
969               if (lv.indexOf('\u000B') != -1 || rv.indexOf('\u000B') != -1) {
970                 return TernaryValue.UNKNOWN;
971               } else {
972                 return TernaryValue.forBoolean(lv.compareTo(rv) < 0);
973               }
974             } else if (left.isTypeOf() && right.isTypeOf()
975                 && left.getFirstChild().isName() && right.getFirstChild().isName()
976                 && left.getFirstChild().getString().equals(right.getFirstChild().getString())) {
977               // Special case: `typeof a < typeof a` is always false.
978               return TernaryValue.FALSE;
979             }
980           }
981         }
982         // Then, try to evaluate based on the value of the node. Try comparing as numbers.
983         Double lv = NodeUtil.getNumberValue(left, useTypes);
984         Double rv = NodeUtil.getNumberValue(right, useTypes);
985         if (lv == null || rv == null) {
986           // Special case: `x < x` is always false.
987           //
988           // TODO(moz): If we knew the named value wouldn't be NaN, it would be nice to handle
989           // LE and GE. We should use type information if available here.
990           if (!willNegate && left.isName() && right.isName()) {
991             if (left.getString().equals(right.getString())) {
992               return TernaryValue.FALSE;
993             }
994           }
995           return TernaryValue.UNKNOWN;
996         }
997         if (Double.isNaN(lv) || Double.isNaN(rv)) {
998           return TernaryValue.forBoolean(willNegate);
999         } else {
1000           return TernaryValue.forBoolean(lv.doubleValue() < rv.doubleValue());
1001         }
1002       }

1004       /** http://www.ecma-international.org/ecma-262/6.0/#sec-abstract-equality-comparison */
1005       private static TernaryValue tryAbstractEqualityComparison(Node left, Node right,
1006           boolean useTypes) {
1007         // Evaluate based on the general type.
1008         ValueType leftValueType = NodeUtil.getKnownValueType(left);
1009         ValueType rightValueType = NodeUtil.getKnownValueType(right);
1010         if (leftValueType != ValueType.UNDETERMINED && rightValueType != ValueType.UNDETERMINED) {
1011           // Delegate to strict equality comparison for values of the same type.
1012           if (leftValueType == rightValueType) {
1013             return tryStrictEqualityComparison(left, right, useTypes);
1014           }
1015           if ((leftValueType == ValueType.NULL && rightValueType == ValueType.VOID)
1016               || (leftValueType == ValueType.VOID && rightValueType == ValueType.NULL)) {
1017             return TernaryValue.TRUE;
1018           }
1019           if ((leftValueType == ValueType.NUMBER && rightValueType == ValueType.STRING)
1020               || rightValueType == ValueType.BOOLEAN) {
1021             Double rv = NodeUtil.getNumberValue(right, useTypes);
1022             return rv == null
1023                 ? TernaryValue.UNKNOWN
1024                 : tryAbstractEqualityComparison(left, IR.number(rv), useTypes);
1025           }
1026           if ((leftValueType == ValueType.STRING && rightValueType == ValueType.NUMBER)
1027               || leftValueType == ValueType.BOOLEAN) {
1028             Double lv = NodeUtil.getNumberValue(left, useTypes);
1029             return lv == null
1030                 ? TernaryValue.UNKNOWN
1031                 : tryAbstractEqualityComparison(IR.number(lv), right, useTypes);
1032           }
1033           if ((leftValueType == ValueType.STRING || leftValueType == ValueType.NUMBER)
1034               && rightValueType == ValueType.OBJECT) {
1035             return TernaryValue.UNKNOWN;
1036           }
1037           if (leftValueType == ValueType.OBJECT
1038               && (rightValueType == ValueType.STRING || rightValueType == ValueType.NUMBER)) {
1039             return TernaryValue.UNKNOWN;
1040           }
1041           return TernaryValue.FALSE;
1042         }
1043         // In general, the rest of the cases cannot be folded.
1044         return TernaryValue.UNKNOWN;
1045       }

1047       /** http://www.ecma-international.org/ecma-262/6.0/#sec-strict-equality-comparison */
1048       private static TernaryValue tryStrictEqualityComparison(Node left, Node right, boolean useTypes) {
1049         // First, try to evaluate based on the general type.
1050         ValueType leftValueType = NodeUtil.getKnownValueType(left);
1051         ValueType rightValueType = NodeUtil.getKnownValueType(right);
1052         if (leftValueType != ValueType.UNDETERMINED && rightValueType != ValueType.UNDETERMINED) {
1053           // Strict equality can only be true for values of the same type.
1054           if (leftValueType != rightValueType) {
1055             return TernaryValue.FALSE;
1056           }
1057           switch (leftValueType) {
1058             case VOID:
1059             case NULL:
1060               return TernaryValue.TRUE;
1061             case NUMBER: {
1062               if (NodeUtil.isNaN(left)) {
1063                 return TernaryValue.FALSE;
1064               }
1065               if (NodeUtil.isNaN(right)) {
1066                 return TernaryValue.FALSE;
1067               }
1068               Double lv = NodeUtil.getNumberValue(left, useTypes);
1069               Double rv = NodeUtil.getNumberValue(right, useTypes);
1070               if (lv != null && rv != null) {
1071                 return TernaryValue.forBoolean(lv.doubleValue() == rv.doubleValue());
1072               }
1073               break;
1074             }
1075             case STRING: {
1076               String lv = NodeUtil.getStringValue(left);
1077               String rv = NodeUtil.getStringValue(right);
1078               if (lv != null && rv != null) {
1079                 // In JS, browsers parse \v differently. So do not consider strings
1080                 // equal if one contains \v.
1081                 if (lv.indexOf('\u000B') != -1 || rv.indexOf('\u000B') != -1) {
1082                   return TernaryValue.UNKNOWN;
1083                 } else {
1084                   return lv.equals(rv) ? TernaryValue.TRUE : TernaryValue.FALSE;
1085                 }
1086               } else if (left.isTypeOf() && right.isTypeOf()
1087                   && left.getFirstChild().isName() && right.getFirstChild().isName()
1088                   && left.getFirstChild().getString().equals(right.getFirstChild().getString())) {
1089                 // Special case, typeof a == typeof a is always true.
1090                 return TernaryValue.TRUE;
1091               }
1092               break;
1093             }
1094             case BOOLEAN: {
1095               TernaryValue lv = NodeUtil.getPureBooleanValue(left);
1096               TernaryValue rv = NodeUtil.getPureBooleanValue(right);
1097               return lv.and(rv).or(lv.not().and(rv.not()));
1098             }
1099             default: // Symbol and Object cannot be folded in the general case.
1100               return TernaryValue.UNKNOWN;
1101           }
1102         }
1103     
1104         // Then, try to evaluate based on the value of the node. There's only one special case:
1105         // Any strict equality comparison against NaN returns false.
1106         if (NodeUtil.isNaN(left) || NodeUtil.isNaN(right)) {
1107           return TernaryValue.FALSE;
1108         }
1109         return TernaryValue.UNKNOWN;
1110       }

1112       static TernaryValue evaluateComparison(Token op, Node left, Node right, boolean useTypes) {
1113         // Don't try to minimize side-effects here.
1114         if (NodeUtil.mayHaveSideEffects(left) || NodeUtil.mayHaveSideEffects(right)) {
1115           return TernaryValue.UNKNOWN;
1116         }
1117     
1118         switch (op) {
1119           case EQ:
1120             return tryAbstractEqualityComparison(left, right, useTypes);
1121           case NE:
1122             return tryAbstractEqualityComparison(left, right, useTypes).not();
1123           case SHEQ:
1124             return tryStrictEqualityComparison(left, right, useTypes);
1125           case SHNE:
1126             return tryStrictEqualityComparison(left, right, useTypes).not();
1127           case LT:
1128             return tryAbstractRelationalComparison(left, right, useTypes, false);
1129           case GT:
1130             return tryAbstractRelationalComparison(right, left, useTypes, false);
1131           case LE:
1132             return tryAbstractRelationalComparison(right, left, useTypes, true).not();
1133           case GE:
1134             return tryAbstractRelationalComparison(left, right, useTypes, true).not();
1135         }
1136         throw new IllegalStateException("Unexpected operator for comparison");
1137       }

1241       /**
1242        * Try to fold array-length. e.g [1, 2, 3].length ==> 3, [x, y].length ==> 2
1243        */
1244       private Node tryFoldGetProp(Node n, Node left, Node right) {
1245         Preconditions.checkArgument(n.isGetProp());
1246     
1247         if (left.isObjectLit()) {
1248           return tryFoldObjectPropAccess(n, left, right);
1249         }
1250     
1251         if (right.isString() &&
1252             right.getString().equals("length")) {
1253           int knownLength = -1;
1254           switch (left.getType()) {
1255             case ARRAYLIT:
1256               if (mayHaveSideEffects(left)) {
1257                 // Nope, can't fold this, without handling the side-effects.
1258                 return n;
1259               }
1260               knownLength = left.getChildCount();
1261               break;
1262             case STRING:
1263               knownLength = left.getString().length();
1264               break;
1265             default:
1266               // Not a foldable case, forget it.
1267               return n;
1268           }
1269     
1270           Preconditions.checkState(knownLength != -1);
1271           Node lengthNode = IR.number(knownLength);
1272           n.getParent().replaceChild(n, lengthNode);
1273           reportCodeChange();
1274     
1275           return lengthNode;
1276         }
1277     
1278         return n;
1279       }

1281       private Node tryFoldArrayAccess(Node n, Node left, Node right) {
1282         // If GETPROP/GETELEM is used as assignment target the array literal is
1283         // acting as a temporary we can't fold it here:
1284         //    "[][0] += 1"
1285         if (NodeUtil.isAssignmentTarget(n)) {
1286           return n;
1287         }
1288     
1289         if (!right.isNumber()) {
1290           // Sometimes people like to use complex expressions to index into
1291           // arrays, or strings to index into array methods.
1292           return n;
1293         }
1294     
1295         double index = right.getDouble();
1296         int intIndex = (int) index;
1297         if (intIndex != index) {
1298           report(INVALID_GETELEM_INDEX_ERROR, right);
1299           return n;
1300         }
1301     
1302         if (intIndex < 0) {
1303           report(INDEX_OUT_OF_BOUNDS_ERROR, right);
1304           return n;
1305         }
1306     
1307         Node current = left.getFirstChild();
1308         Node elem = null;
1309         for (int i = 0; current != null; i++) {
1310           if (i != intIndex) {
1311             if (mayHaveSideEffects(current)) {
1312               return n;
1313             }
1314           } else {
1315             elem = current;
1316           }
1317     
1318           current = current.getNext();
1319         }
1320     
1321         if (elem == null) {
1322           report(INDEX_OUT_OF_BOUNDS_ERROR, right);
1323           return n;
1324         }
1325     
1326         if (elem.isEmpty()) {
1327           elem = NodeUtil.newUndefinedNode(elem);
1328         } else {
1329           left.removeChild(elem);
1330         }
1331     
1332         // Replace the entire GETELEM with the value
1333         n.getParent().replaceChild(n, elem);
1334         reportCodeChange();
1335         return elem;
1336       }

1338       private Node tryFoldStringArrayAccess(Node n, Node left, Node right) {
1339         // If GETPROP/GETELEM is used as assignment target the array literal is
1340         // acting as a temporary we can't fold it here:
1341         //    "[][0] += 1"
1342         if (NodeUtil.isAssignmentTarget(n)) {
1343           return n;
1344         }
1345     
1346         if (!right.isNumber()) {
1347           // Sometimes people like to use complex expressions to index into
1348           // arrays, or strings to index into array methods.
1349           return n;
1350         }
1351     
1352         double index = right.getDouble();
1353         int intIndex = (int) index;
1354         if (intIndex != index) {
1355           report(INVALID_GETELEM_INDEX_ERROR, right);
1356           return n;
1357         }
1358     
1359         if (intIndex < 0) {
1360           report(INDEX_OUT_OF_BOUNDS_ERROR, right);
1361           return n;
1362         }
1363     
1364         Preconditions.checkState(left.isString());
1365         String value = left.getString();
1366         if (intIndex >= value.length()) {
1367           report(INDEX_OUT_OF_BOUNDS_ERROR, right);
1368           return n;
1369         }
1370     
1371         char c = 0;
1372         // Note: For now skip the strings with unicode
1373         // characters as I don't understand the differences
1374         // between Java and JavaScript.
1375         for (int i = 0; i <= intIndex; i++) {
1376           c = value.charAt(i);
1377           if (c < 32 || c > 127) {
1378             return n;
1379           }
1380         }
1381         Node elem = IR.string(Character.toString(c));
1382     
1383         // Replace the entire GETELEM with the value
1384         n.getParent().replaceChild(n, elem);
1385         reportCodeChange();
1386         return elem;
1387       }

1389       private Node tryFoldObjectPropAccess(Node n, Node left, Node right) {
1390         Preconditions.checkArgument(NodeUtil.isGet(n));
1391     
1392         if (!left.isObjectLit() || !right.isString()) {
1393           return n;
1394         }
1395     
1396         if (NodeUtil.isAssignmentTarget(n)) {
1397           // If GETPROP/GETELEM is used as assignment target the object literal is
1398           // acting as a temporary we can't fold it here:
1399           //    "{a:x}.a += 1" is not "x += 1"
1400           return n;
1401         }
1402     
1403         // find the last definition in the object literal
1404         Node key = null;
1405         Node value = null;
1406         for (Node c = left.getFirstChild(); c != null; c = c.getNext()) {
1407           if (c.getString().equals(right.getString())) {
1408             switch (c.getType()) {
1409               case SETTER_DEF:
1410                 continue;
1411               case GETTER_DEF:
1412               case STRING_KEY:
1413                 if (value != null && mayHaveSideEffects(value)) {
1414                   // The previously found value had side-effects
1415                   return n;
1416                 }
1417                 key = c;
1418                 value = key.getFirstChild();
1419                 break;
1420               default:
1421                 throw new IllegalStateException();
1422             }
1423           } else if (mayHaveSideEffects(c.getFirstChild())) {
1424             // We don't handle the side-effects here as they might need a temporary
1425             // or need to be reordered.
1426             return n;
1427           }
1428         }
1429     
1430         // Didn't find a definition of the name in the object literal, it might
1431         // be coming from the Object prototype
1432         if (value == null) {
1433           return n;
1434         }
1435     
1436         if (value.isFunction() && NodeUtil.referencesThis(value)) {
1437           // 'this' may refer to the object we are trying to remove
1438           return n;
1439         }
1440     
1441         Node replacement = value.detachFromParent();
1442         if (key.isGetterDef()){
1443           replacement = IR.call(replacement);
1444           replacement.putBooleanProp(Node.FREE_CALL, true);
1445         }
1446     
1447         n.getParent().replaceChild(n, replacement);
1448         reportCodeChange();
1449         return n;
1450       }
1451     }
