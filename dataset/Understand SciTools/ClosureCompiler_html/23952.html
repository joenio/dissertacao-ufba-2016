21     /**
22      * An immutable sparse bitset that deals well where the data is chunky:
23      * where P(bit[x+1] == bit[x]).  E.g. [101,102,103,104,105,1001,1002,1003,1004]
24      * is chunky.
25      *
26      * @author mikesamuel@gmail.com (Mike Samuel)
27      */
28     final class CharRanges {
29       /**
30        * A strictly increasing set of bit indices where even members are the
31        * inclusive starts of ranges, and odd members are the exclusive ends.
32        * <p>
33        * E.g., { 1, 5, 6, 10 } represents the set ( 1, 2, 3, 4, 6, 7, 8, 9 ).
34        */
35       private final int[] ranges;

40           = new CharRanges(new int[] { 0, 0x10000 });

42       public static CharRanges inclusive(int start, int end) {
43         if (start > end) {
44           throw new IndexOutOfBoundsException(start + " > " + end);
45         }
46         return new CharRanges(new int[] { start, end + 1 });
47       }

52       public static CharRanges withMembers(int... members) {
53         return new CharRanges(intArrayToRanges(members));

56       /**
57        * Returns an instance containing the given ranges.
58        * @param ranges An even-length ordered sequence of non-overlapping,
59        *     non-contiguous, [inclusive start, exclusive end) ranges.
60        */
61       public static CharRanges withRanges(int... ranges) {
62         if ((ranges.length & 1) != 0) { throw new IllegalArgumentException(); }
63         for (int i = 1; i < ranges.length; ++i) {
64           if (ranges[i] <= ranges[i - 1]) {
65             throw new IllegalArgumentException(ranges[i] + " > " + ranges[i - 1]);
66           }
67         }
68         return new CharRanges(ranges);
69       }

75       private static int[] intArrayToRanges(int[] members) {
76         int nMembers = members.length;
77         if (nMembers == 0) {
78           return new int[0];
79         }
80     
81         Arrays.sort(members);
82     
83         // Count the number of runs.
84         int nRuns = 1;
85         for (int i = 1; i < nMembers; ++i) {
86           int current = members[i], last = members[i - 1];
87           if (current == last) { continue; }
88           if (current != last + 1) { ++nRuns; }
89         }
90     
91         int[] ranges = new int[nRuns * 2];
92         ranges[0] = members[0];
93         int k = 0;
94         for (int i = 1; k + 2 < ranges.length; ++i) {
95           int current = members[i], last = members[i - 1];
96           if (current == last) { continue; }
97           if (current != last + 1) {
98             ranges[++k] = last + 1;  // add 1 to make end exclusive
99             ranges[++k] = current;
100           }
101         }
102         ranges[++k] = members[nMembers - 1] + 1;  // add 1 to make end exclusive
103         return ranges;
104       }

106       public boolean contains(int bit) {
107         return (Arrays.binarySearch(ranges, bit) & 1) == 0;
108         // By the contract of Arrays.binarySearch, its result is either the position
109         // of bit in ranges or it is the bitwise inverse of the position of the
110         // least element greater than bit.
111     
112         // Two cases
113         // case (idx >= 0)
114         //     We ended up exactly on a range boundary.
115         //     Starts are inclusive and ends are both exclusive, so this contains
116         //     bit iff idx is even.
117         //
118         // case (idx < 0)
119         //     If the least element greater than bit is an odd element,
120         //     then bit must be greater than a start and less than an end, so
121         //     contained.
122         //
123         //     If bit is greater than all elements, then idx will be past the end of
124         //     the array, and will be even since ranges.length is even.
125         //
126         //     Otherwise, bit must be in the space between two runs, so not
127         //     contained.
128         //
129         //     In all cases, oddness is equivalent to containedness.
130     
131         // Those two cases lead to
132         //     idx >= 0 ? ((idx & 1) == 0) : ((~idx & 1) == 1)
133     
134         // But ~n & bit == bit   <=>   n & bit == 0, so
135         //     idx >= 0 ? ((idx & 1) == 0) : ((~idx & 1) == 1)
136         // =>  idx >= 0 ? ((idx & 1) == 0) : ((idx & 1) == 0)
137         // =>  (idx & 1) == 0
138       }

150       public CharRanges union(CharRanges other) {
151         // Index of the input ranges
152         int[] q = this.ranges, r = other.ranges;
153         // Lengths of the inputs
154         int m = q.length, n = r.length;
155     
156         if (m == 0) { return other; }
157         if (n == 0) { return this; }
158     
159         // The output array.  The length is m+n in the worst case when all the
160         // ranges in a are disjoint from the ranges in b.
161         int[] out = new int[m + n];
162     
163         // Indexes into the various arrays
164         int i = 0, j = 0, k = 0;
165         // Since there are three arrays, and indices into them the following
166         // should never occur in this function:
167         // (1) q[j] or q[k]                         -- q is indexed by i
168         // (2) r[i] or r[k]                         -- r is indexed by j
169         // (3) out[i] or out[j]                     -- out is indexed by k
170         // (4) i < n or j < m                       -- index compared to wrong limit
171     
172         // This loop exits because we always increment at least one of i,j.
173         while (i < m && j < n) {
174           // Range starts and ends.
175           int a0 = q[i], a1 = q[i + 1],
176               b0 = r[j], b1 = r[j + 1];
177           if (a1 < b0) {  // [a0, a1) ends before [b0, b1) starts
178             out[k++] = a0;
179             out[k++] = a1;
180             i += 2;
181           } else if (b1 < a0) {  // [b0, b1) ends before [a0, a1) starts
182             out[k++] = b0;
183             out[k++] = b1;
184             j += 2;
185           } else {  // ranges overlap
186             // We need to compute a new range based on the set of ranges that
187             // transitively overlap.
188             //       AAAAAAAAA AAA
189             //     BBB  BBB* BBB
190             // In the range above, the start comes from one set, and the end from
191             // another.  The range with the asterisk next to it is subsumed entirely
192             // by a range from the other, and so not all ranges on the input
193             // contribute a value to the output.
194             // The last BBB run serves only as a bridge -- it overlaps two
195             // disjoint ranges in the other one so establishes that they
196             // transitively overlap.
197             int start = Math.min(a0, b0);
198             // Guess at the end, and lookahead to come up with a more complete
199             // estimate.
200             int end = Math.max(a1, b1);
201             i += 2;
202             j += 2;
203             while (i < m || j < n) {
204               if (i < m && q[i] <= end) {
205                 end = Math.max(end, q[i + 1]);
206                 i += 2;
207               } else if (j < n && r[j] <= end) {
208                 end = Math.max(end, r[j + 1]);
209                 j += 2;
210               } else {
211                 break;
212               }
213             }
214             out[k++] = start;
215             out[k++] = end;
216           }
217         }
218         // There may be unprocessed ranges at the end of one of the inputs.
219         if (i < m) {
220           System.arraycopy(q, i, out, k, m - i);
221           k += m - i;
222         } else if (j < n) {
223           System.arraycopy(r, j, out, k, n - j);
224           k += n - j;
225         }
226         // We guessed at the output length above.  Cut off the tail.
227         if (k != out.length) {
228           int[] clipped = Arrays.copyOf(out, k);
229           out = clipped;
230         }
231         return new CharRanges(out);
232       }

234       public CharRanges intersection(CharRanges other) {
235         int[] aRanges = ranges, bRanges = other.ranges;
236         int aLen = aRanges.length, bLen = bRanges.length;
237         if (aLen == 0) { return this; }
238         if (bLen == 0) { return other; }
239         int aIdx = 0, bIdx = 0;
240         int[] intersection = new int[Math.min(aLen, bLen)];
241         int intersectionIdx = 0;
242         int pos = Math.min(aRanges[0], bRanges[0]);
243         while (aIdx < aLen && bIdx < bLen) {
244           if (aRanges[aIdx + 1] <= pos) {
245             aIdx += 2;
246           } else if (bRanges[bIdx + 1] <= pos) {
247             bIdx += 2;
248           } else {
249             int start = Math.max(aRanges[aIdx], bRanges[bIdx]);
250             if (pos < start) {  // Advance to start of common block.
251               pos = start;
252             } else {
253               // Now we know that pos is less than the ends of the two ranges and
254               // greater or equal to the starts of the two ranges.
255               int end = Math.min(aRanges[aIdx + 1], bRanges[bIdx + 1]);
256               if (intersectionIdx != 0
257                   && pos == intersection[intersectionIdx - 1]) {
258                 intersection[intersectionIdx - 1] = end;
259               } else {
260                 if (intersectionIdx == intersection.length) {
261                   int[] newArr = new int[intersectionIdx * 2];
262                   System.arraycopy(intersection, 0, newArr, 0, intersectionIdx);
263                   intersection = newArr;
264                 }
265                 intersection[intersectionIdx++] = pos;
266                 intersection[intersectionIdx++] = end;
267               }
268               pos = end;
269             }
270           }
271         }
272         if (intersectionIdx != intersection.length) {
273           int[] newArr = Arrays.copyOf(intersection, intersectionIdx);
274           intersection = newArr;
275         }
276         return new CharRanges(intersection);
277       }

279       public CharRanges difference(CharRanges subtrahendRanges) {
280         // difference = minuend - subtrahend
281         int[] minuend = this.ranges;
282         int[] subtrahend = subtrahendRanges.ranges;
283     
284         int mn = minuend.length, sn = subtrahend.length;
285         if (mn == 0 || sn == 0) { return this; }
286     
287         int[] difference = new int[minuend.length];
288     
289         // Indices into minuend.ranges, subtrahend.ranges, and difference.
290         int mIdx = 0, sIdx = 0, dIdx = 0;
291     
292         int pos = minuend[0];
293         while (mIdx < mn) {
294           if (pos >= minuend[mIdx + 1]) {
295             mIdx += 2;
296           } else if (pos < minuend[mIdx]) {
297             // Skip gaps in the minuend.
298             pos = minuend[mIdx];
299           } else if (sIdx < sn && pos >= subtrahend[sIdx]) {
300             // Skip over a removed part.
301             pos = subtrahend[sIdx + 1];
302             sIdx += 2;
303           } else {
304             // Now we know that pos is between [minuend[i], minuend[i + 1])
305             // and outside [subtrahend[j], subtrahend[j + 1]).
306             int end = sIdx < sn
307                 ? Math.min(minuend[mIdx + 1], subtrahend[sIdx]) : minuend[mIdx + 1];
308             if (dIdx != 0 && difference[dIdx - 1] == pos) {
309               difference[dIdx - 1] = pos;
310             } else {
311               if (dIdx == difference.length) {
312                 int[] newArr = new int[dIdx * 2];
313                 System.arraycopy(difference, 0, newArr, 0, dIdx);
314                 difference = newArr;
315               }
316               difference[dIdx++] = pos;
317               difference[dIdx++] = end;
318             }
319             pos = end;
320           }
321         }
322     
323         if (dIdx != difference.length) {
324           int[] newArr = Arrays.copyOf(difference, dIdx);
325           difference = newArr;
326         }
327     
328         return new CharRanges(difference);
329       }

331       public boolean containsAll(CharRanges sub) {
332         int[] superRanges = this.ranges;
333         int[] subRanges = sub.ranges;
334     
335         int superIdx = 0, subIdx = 0;
336         int superLen = superRanges.length, subLen = subRanges.length;
337         while (subIdx < subLen) {
338           if (superIdx == superLen) {
339             return false;
340           }
341           if (superRanges[superIdx + 1] <= subRanges[subIdx]) {
342             // Super range ends before subRange starts.
343             superIdx += 2;
344           } else if (superRanges[superIdx] > subRanges[subIdx]) {
345             // Uncontained portion at start of sub range.
346             return false;
347           } else if (superRanges[superIdx + 1] >= subRanges[subIdx + 1]) {
348             // A sub range is completely contained in the super range.
349             // We know this because of the above condition and we have already
350             // ruled out that subRanges[subIdx] < superRanges[superIdx].
351             subIdx += 2;
352           } else {
353             // Uncontained portion at end of sub range.
354             return false;
355           }
356         }
357         return subIdx == subLen;
358       }

360       /**
361        * Shifts the bits matched by the given delta.
362        * So if this has the bits (a, b, c, ..., z) set then the result has the bits
363        * ((a - delta), (b - delta), (c - delta), ...., (z - delta)) set.
364        *
365        * @throws IndexOutOfBoundsException if shifting by delta would cause an
366        *     overflow or underflow in a 32 bit {@code signed int} range boundary.
367        *     Since the end boundaries of ranges are exclusive, even if there is no
368        *     range containing {@link Integer#MAX_VALUE}, shifting by a delta of 1
369        *     can cause an overflow.
370        */
371       public CharRanges shift(int delta) {
372         int n = ranges.length;
373         if (delta == 0 || n == 0) { return this; }
374         // Test overflow/underflow
375         if (delta < 0) {
376           long lmin = ranges[0] + delta;
377           if (lmin < Integer.MIN_VALUE) { throw new IndexOutOfBoundsException(); }
378         } else {
379           long lmax = ranges[n - 1] + delta;
380           if (lmax > Integer.MAX_VALUE) { throw new IndexOutOfBoundsException(); }
381         }
382         // Create a shifted range.
383         int[] shiftedRanges = new int[n];
384         for (int i = n; --i >= 0;) {
385           shiftedRanges[i] = ranges[i] + delta;
386         }
387         return new CharRanges(shiftedRanges);
388       }

390       @Override
391       public String toString() {
392         StringBuilder sb = new StringBuilder();
393         sb.append('[');
394         for (int i = 0; i < ranges.length; ++i) {
395           if ((i & 1) != 0 && ranges[i] == ranges[i - 1] + 1) { continue; }
396           if (i != 0) { sb.append((i & 1) == 0 ? ' ' : '-'); }
397           sb.append("0x").append(Integer.toString(ranges[i] - (i & 1), 16));
398         }
399         sb.append(']');
400         return sb.toString();
401       }

411         int hc = 0;
