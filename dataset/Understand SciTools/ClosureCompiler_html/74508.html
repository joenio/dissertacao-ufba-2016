105       public final void testArrayOutOfOrder() {
106         test("var a = []; a[1] = 1; a[0] = 0;",
107              "var a = [0, 1];");
108         test("var a; a = []; a[1] = 1; a[0] = 0;",
109              "var a; a = [0, 1];");
110         // We cannot change the order of side-effects.
111         // The below should not be
112         //   var x = 0; var a = [x++, x++]
113         // since that would produce
114         //   var a = [0, 1], x = 2;
115         // instead of
116         //   var a = [1, 0], x = 2;
117         testSame("var x = 0; var a = []; a[1] = x++; a[0] = x++;");
118         testSame("var x; x = 0; var a = []; a[1] = x++; a[0] = x++;");
119       }
