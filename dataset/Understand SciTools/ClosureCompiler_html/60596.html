16     package com.google.javascript.jscomp;
17     
18     /**
19      * Basic information on an entry point module.
20      * While closure entry points are namespaces,
21      * ES6 and CommonJS entry points are file paths
22      * which are normalized to a closure namespace.
23      *
24      * This class allows error messages to the user to
25      * be based on the input name rather than the
26      * normalized version.
27      */
28     public class ModuleIdentifier {
29       private final String name;
30       private final String closureNamespace;
31       private final String moduleName;
32     
33       /**
34        * @param name as provided by the user
35        * @param closureNamespace entry point normalized to a closure namespace
36        * @param moduleName For closure namespaces, the module name may be different than
37        *     the namespace
38        */
39       ModuleIdentifier(String name, String closureNamespace, String moduleName) {
40         this.name = name;
41         this.closureNamespace = closureNamespace;
42         this.moduleName = moduleName;
43       }
44     
45       public String getName() {
46         return name;
47       }
48     
49       public String getClosureNamespace() {
50         return closureNamespace;
51       }
52     
53       public String getModuleName() {
54         return moduleName;
55       }
56     
57       @Override
58       public String toString() {
59         if (closureNamespace.equals(moduleName)) {
60           return closureNamespace;
61         }
62         return moduleName + ":" + closureNamespace;
63       }
64     
65       /**
66        * @param name Closure namespace used as an entry point. May start
67        *     "goog:" when provided as a flag from the command line.
68        *
69        * Closure entry points may also be formatted as:
70        *     'goog:moduleName:name.space'
71        * which specifies that the module name and provided namespace
72        * are different
73        */
74       public static ModuleIdentifier forClosure(String name) {
75         String normalizedName = name;
76         if (normalizedName.startsWith("goog:")) {
77           normalizedName = normalizedName.substring("goog:".length());
78         }
79     
80         String namespace = normalizedName;
81         String moduleName = normalizedName;
82         int splitPoint = normalizedName.indexOf(':');
83         if (splitPoint != -1) {
84           moduleName = normalizedName.substring(0, splitPoint);
85           namespace = normalizedName.substring(Math.min(splitPoint + 1, normalizedName.length() - 1));
86         }
87     
88         return new ModuleIdentifier(normalizedName, namespace, moduleName);
89       }
90     
91       /**
92        * @param filepath ES6 or CommonJS module used as an entry point.
93        */
94       public static ModuleIdentifier forFile(String filepath) {
95         String normalizedName = ES6ModuleLoader.toModuleName(ES6ModuleLoader.createUri(filepath));
96         return new ModuleIdentifier(filepath, normalizedName, normalizedName);
97       }
98     }
