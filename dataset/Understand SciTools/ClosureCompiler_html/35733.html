93       /**
94        * Callback for tree-based traversals
95        */
96       public interface Callback {
97         /**
98          * <p>Visits a node in pre order (before visiting its children) and decides
99          * whether this node's children should be traversed. If children are
100          * traversed, they will be visited by
101          * {@link #visit(NodeTraversal, Node, Node)} in postorder.</p>
102          * <p>Implementations can have side effects (e.g. modifying the parse
103          * tree).</p>
104          * @return whether the children of this node should be visited
105          */
106         boolean shouldTraverse(NodeTraversal nodeTraversal, Node n, Node parent);
107     
108         /**
109          * <p>Visits a node in postorder (after its children have been visited).
110          * A node is visited only if all its parents should be traversed
111          * ({@link #shouldTraverse(NodeTraversal, Node, Node)}).</p>
112          * <p>Implementations can have side effects (e.g. modifying the parse
113          * tree).</p>
114          */
115         void visit(NodeTraversal t, Node n, Node parent);
116       }

260       private void throwUnexpectedException(Exception unexpectedException) {
261         // If there's an unexpected exception, try to get the
262         // line number of the code that caused it.
263         String message = unexpectedException.getMessage();
264     
265         // TODO(user): It is possible to get more information if curNode or
266         // its parent is missing. We still have the scope stack in which it is still
267         // very useful to find out at least which function caused the exception.
268         if (inputId != null) {
269           message =
270               unexpectedException.getMessage() + "\n" +
271               formatNodeContext("Node", curNode) +
272               (curNode == null ?
273                   "" :
274                   formatNodeContext("Parent", curNode.getParent()));
275         }
276         compiler.throwInternalError(message, unexpectedException);
277       }

279       private String formatNodeContext(String label, Node n) {
280         if (n == null) {
281           return "  " + label + ": NULL";
282         }
283         return "  " + label + "(" + n.toString(false, false, false) + "): "
284             + formatNodePosition(n);

287       /**
288        * Traverses a parse tree recursively.
289        */
290       public void traverse(Node root) {
291         try {
292           inputId = NodeUtil.getInputId(root);
293           sourceName = "";
294           curNode = root;
295           pushScope(root);
296           // null parent ensures that the shallow callbacks will traverse root
297           traverseBranch(root, null);
298           popScope();
299         } catch (Exception unexpectedException) {
300           throwUnexpectedException(unexpectedException);
301         }
302       }

304       void traverseRoots(Node externs, Node root) {
305         try {
306           Node scopeRoot = externs.getParent();
307           Preconditions.checkNotNull(scopeRoot);
308     
309           inputId = NodeUtil.getInputId(scopeRoot);
310           sourceName = "";
311           curNode = scopeRoot;
312           pushScope(scopeRoot);
313     
314           traverseBranch(externs, scopeRoot);
315           Preconditions.checkState(root.getParent() == scopeRoot);
316           traverseBranch(root, scopeRoot);
317     
318           popScope();
319         } catch (Exception unexpectedException) {
320           throwUnexpectedException(unexpectedException);
321         }
322       }

326       private String formatNodePosition(Node n) {
327         String sourceFileName = getBestSourceFileName(n);
328         if (sourceFileName == null) {
329           return MISSING_SOURCE + "\n";
330         }
331     
332         int lineNumber = n.getLineno();
333         int columnNumber = n.getCharno();
334         String src = compiler.getSourceLine(sourceFileName, lineNumber);
335         if (src == null) {
336           src = MISSING_SOURCE;
337         }
338         return sourceFileName + ":" + lineNumber + ":" + columnNumber + "\n"
339             + src + "\n";
340       }

361       /**
362        * Traverses a parse tree recursively with a scope, starting at that scope's
363        * root.
364        */
365       void traverseAtScope(Scope s) {
366         Node n = s.getRootNode();
367         if (n.isFunction()) {
368           // We need to do some extra magic to make sure that the scope doesn't
369           // get re-created when we dive into the function.
370           if (inputId == null) {
371             inputId = NodeUtil.getInputId(n);
372           }
373           sourceName = getSourceName(n);
374           curNode = n;
375           pushScope(s);
376     
377           Node args = n.getSecondChild();
378           Node body = args.getNext();
379           traverseBranch(args, n);
380           traverseBranch(body, n);
381     
382           popScope();
383         } else if (n.isBlock()) {
384           if (inputId == null) {
385             inputId = NodeUtil.getInputId(n);
386           }
387           sourceName = getSourceName(n);
388           curNode = n;
389           pushScope(s);
390     
391           // traverseBranch is not called here to avoid re-creating the block scope.
392           for (Node child = n.getFirstChild(); child != null; ) {
393             Node next = child.getNext();
394             traverseBranch(child, n);
395             child = next;
396           }
397     
398           popScope();
399         } else {
400           Preconditions.checkState(s.isGlobal(), "Expected global scope. Got:", s);
401           traverseWithScope(n, s);
402         }
403       }

405       /**
406        * Traverse a function out-of-band of normal traversal.
407        *
408        * @param node The function node.
409        * @param scope The scope the function is contained in. Does not fire enter/exit
410        *     callback events for this scope.
411        */
412       public void traverseFunctionOutOfBand(Node node, Scope scope) {
413         Preconditions.checkNotNull(scope);
414         Preconditions.checkState(node.isFunction());
415         Preconditions.checkState(scope.getRootNode() != null);
416         if (inputId == null) {
417           inputId = NodeUtil.getInputId(node);
418         }
419         curNode = node.getParent();
420         pushScope(scope, true /* quietly */);
421         traverseBranch(node, curNode);
422         popScope(true /* quietly */);
423       }

425       /**
426        * Traverses an inner node recursively with a refined scope. An inner node may
427        * be any node with a non {@code null} parent (i.e. all nodes except the
428        * root).
429        *
430        * @param node the node to traverse
431        * @param parent the node's parent, it may not be {@code null}
432        * @param refinedScope the refined scope of the scope currently at the top of
433        *     the scope stack or in trivial cases that very scope or {@code null}
434        */
435       void traverseInnerNode(Node node, Node parent, Scope refinedScope) {
436         Preconditions.checkNotNull(parent);
437         if (inputId == null) {
438           inputId = NodeUtil.getInputId(node);
439         }
440         if (refinedScope != null && getScope() != refinedScope) {
441           curNode = node;
442           pushScope(refinedScope);
443           traverseBranch(node, parent);
444           popScope();
445         } else {
446           traverseBranch(node, parent);
447         }
448       }

458       public int getLineNumber() {
459         Node cur = curNode;
460         while (cur != null) {
461           int line = cur.getLineno();

515       /**
516        * Traversal for passes that work only on changed functions.
517        * Suppose a loopable pass P1 uses this traversal.
518        * Then, if a function doesn't change between two runs of P1, it won't look at
519        * the function the second time.
520        * (We're assuming that P1 runs to a fixpoint, o/w we may miss optimizations.)
521        *
522        * <p>Most changes are reported with calls to Compiler.reportCodeChange(), which
523        * doesn't know which scope changed. We keep track of the current scope by
524        * calling Compiler.setScope inside pushScope and popScope.
525        * The automatic tracking can be wrong in rare cases when a pass changes scope
526        * w/out causing a call to pushScope or popScope. It's very hard to find the
527        * places where this happens unless a bug is triggered.
528        * Passes that do cross-scope modifications call
529        * Compiler.reportChangeToEnclosingScope(Node n).
530        */
531       public static void traverseChangedFunctions(
532           AbstractCompiler compiler, FunctionCallback callback) {
533         final AbstractCompiler comp = compiler;
534         final FunctionCallback cb = callback;
535         final Node jsRoot = comp.getJsRoot();
536         NodeTraversal.traverseEs6(comp, jsRoot,
537             new AbstractPreOrderCallback() {
538               @Override
539               public final boolean shouldTraverse(NodeTraversal t, Node n, Node p) {
540                 if ((n == jsRoot || n.isFunction()) && comp.hasScopeChanged(n)) {
541                   cb.enterFunction(comp, n);
542                 }
543                 return true;
544               }
545             });
546       }

591       /**
592        * Traverses a branch.
593        */
594       private void traverseBranch(Node n, Node parent) {
595         Token type = n.getType();
596         if (type == Token.SCRIPT) {
597           inputId = n.getInputId();
598           sourceName = getSourceName(n);
599         }
600     
601         curNode = n;
602         if (!callback.shouldTraverse(this, n, parent)) {
603           return;
604         }
605     
606         if (type == Token.FUNCTION) {
607           traverseFunction(n, parent);
608         } else if (type == Token.CLASS) {
609           traverseClass(n, parent);
610         } else if (useBlockScope && NodeUtil.createsBlockScope(n)) {
611           traverseBlockScope(n);
612         } else {
613           for (Node child = n.getFirstChild(); child != null; ) {
614             // child could be replaced, in which case our child node
615             // would no longer point to the true next
616             Node next = child.getNext();
617             traverseBranch(child, n);
618             child = next;
619           }
620         }
621     
622         curNode = n;
623         callback.visit(this, n, parent);
624       }

626       /** Traverses a function. */
627       private void traverseFunction(Node n, Node parent) {
628         final Node fnName = n.getFirstChild();
629         boolean isFunctionExpression = (parent != null)
630             && NodeUtil.isFunctionExpression(n);
631     
632         if (!isFunctionExpression) {
633           // Function declarations are in the scope containing the declaration.
634           traverseBranch(fnName, n);
635         }
636     
637         curNode = n;
638         pushScope(n);
639     
640         if (isFunctionExpression) {
641           // Function expression names are only accessible within the function
642           // scope.
643           traverseBranch(fnName, n);
644         }
645     
646         final Node args = fnName.getNext();
647         final Node body = args.getNext();
648     
649         // Args
650         traverseBranch(args, n);
651     
652         // Body
653         // ES6 "arrow" function may not have a block as a body.
654         traverseBranch(body, n);
655     
656         popScope();
657       }
658     
659       /** Traverses a class. */
660       private void traverseClass(Node n, Node parent) {
661         final Node className = n.getFirstChild();
662         boolean isClassExpression = NodeUtil.isClassExpression(n);
663     
664         if (!isClassExpression) {
665           // Class declarations are in the scope containing the declaration.
666           traverseBranch(className, n);
667         }
668     
669         curNode = n;
670         pushScope(n);
671     
672         if (isClassExpression) {
673           // Class expression names are only accessible within the function
674           // scope.
675           traverseBranch(className, n);
676         }
677     
678         final Node extendsClause = className.getNext();
679         final Node body = extendsClause.getNext();
680     
681         // Extends
682         traverseBranch(extendsClause, n);
683     
684         // Body
685         traverseBranch(body, n);
686     
687         popScope();
688       }

690       /** Traverses a non-function block. */
691       private void traverseBlockScope(Node n) {
692         pushScope(n);
693         for (Node child : n.children()) {
694           traverseBranch(child, n);
695         }
696         popScope();
697       }

707       /** Creates a new scope (e.g. when entering a function). */
708       private void pushScope(Node node) {
709         Preconditions.checkState(curNode != null);
710         Preconditions.checkState(node != null);
711         compiler.setScope(node);
712         scopeRoots.push(node);
713         if (NodeUtil.isValidCfgRoot(node)) {
714           cfgRoots.push(node);
715           cfgs.push(null);
716         }
717         if (scopeCallback != null) {
718           scopeCallback.enterScope(this);
719         }
720       }

722       /** Creates a new scope (e.g. when entering a function). */
723       private void pushScope(Scope s) {
724         pushScope(s, false);
725       }

727       /**
728        * Creates a new scope (e.g. when entering a function).
729        * @param quietly Don't fire an enterScope callback.
730        */
731       private void pushScope(Scope s, boolean quietly) {
732         Preconditions.checkState(curNode != null);
733         compiler.setScope(s.getRootNode());
734         scopes.push(s);
735         if (NodeUtil.isValidCfgRoot(s.getRootNode())) {
736           cfgRoots.push(s.getRootNode());
737           cfgs.push(null);
738         }
739         if (!quietly && scopeCallback != null) {
740           scopeCallback.enterScope(this);
741         }
742       }

748       /**
749        * Pops back to the previous scope (e.g. when leaving a function).
750        * @param quietly Don't fire the exitScope callback.
751        */
752       private void popScope(boolean quietly) {
753         if (!quietly && scopeCallback != null) {
754           scopeCallback.exitScope(this);
755         }
756         Node scopeRoot;
757         if (scopeRoots.isEmpty()) {
758           scopeRoot = scopes.pop().getRootNode();
759         } else {
760           scopeRoot = scopeRoots.pop();
761         }
762         if (NodeUtil.isValidCfgRoot(scopeRoot)) {
763           Preconditions.checkState(!cfgRoots.isEmpty());
764           Preconditions.checkState(cfgRoots.pop() == scopeRoot);
765           cfgs.pop();
766         }
767         if (hasScope()) {
768           compiler.setScope(getScopeRoot());
769         }
770       }
771     
772       /** Gets the current scope. */
773       public Scope getScope() {
774         Scope scope = scopes.isEmpty() ? null : scopes.peek();
775         if (scopeRoots.isEmpty()) {
776           return scope;
777         }
778     
779         Iterator<Node> it = scopeRoots.descendingIterator();
780         while (it.hasNext()) {
781           scope = scopeCreator.createScope(it.next(), scope);
782           scopes.push(scope);
783         }
784         scopeRoots.clear();
785         // No need to call compiler.setScope; the top scopeRoot is now the top scope
786         return scope;
787       }

789       public Scope getClosestHoistScope() {
790         // TODO(moz): This should not call getScope(). We should find the root of the closest hoist
791         // scope and effectively getScope() from there, which avoids scanning inner scopes that might
792         // not be needed.
793         return getScope().getClosestHoistScope();
794       }

796       public TypedScope getTypedScope() {
797         Scope s = getScope();
798         Preconditions.checkState(s instanceof TypedScope,
799             "getTypedScope called for untyped traversal");
800         return (TypedScope) s;
801       }

803       /** Gets the control flow graph for the current JS scope. */
804       public ControlFlowGraph<Node> getControlFlowGraph() {
805         if (cfgs.peek() == null) {
806           ControlFlowAnalysis cfa = new ControlFlowAnalysis(compiler, false, true);
807           cfa.process(null, getCfgRoot());
808           cfgs.pop();
809           cfgs.push(cfa.getCfg());
810         }
811         return cfgs.peek();
812       }

814       /** Returns the current scope's root. */
815       public Node getScopeRoot() {
816         if (scopeRoots.isEmpty()) {
817           return scopes.peek().getRootNode();
818         } else {
819           return scopeRoots.peek();
820         }
821       }

823       private Node getCfgRoot() {
824         return cfgRoots.peek();
825       }

827       /**
828        * Determines whether the traversal is currently in the global scope. Note that this returns false
829        * in a global block scope.
830        */
831       public boolean inGlobalScope() {
832         return getScopeDepth() == 0;
833       }

835       /**
836        * Determines whether the hoist scope of the current traversal is global.
837        */
838       public boolean inGlobalHoistScope() {
839         return !getCfgRoot().isFunction();
840       }

842       int getScopeDepth() {
843         int sum = scopes.size() + scopeRoots.size();
844         Preconditions.checkState(sum > 0);
845         return sum - 1; // Use 0-based scope depth to be consistent within the compiler
846       }

848       public boolean hasScope() {
849         return !(scopes.isEmpty() && scopeRoots.isEmpty());
850       }
