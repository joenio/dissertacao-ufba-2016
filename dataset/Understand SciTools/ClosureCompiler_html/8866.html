39      * Constructs a name reference graph.
40      *

44     class NameReferenceGraphConstruction implements CompilerPass {

46       private final AbstractCompiler compiler;
47       private final NameReferenceGraph graph;

49       // Maps "foo" -> (curFuncName, unknownObject.foo) if we have no idea what
50       // the unknown object is. After we finish one pass, we must go through all
51       // the nodes that might have a name foo and connect that to the curFuncName.
52       // The accuracy of the analysis will depend heavily on eliminating the need
53       // to resort to this map.
54       private final Multimap<String, NameUse> unknownNameUse =
55           HashMultimap.create();
56     
57       // Should we continue even if we found a type checker bug.
58       private static final boolean CONSERVATIVE = false;
59     
60       // The symbol for the current function so we can quickly create a reference
61       // edge when we see a call: Example when this symbol is foo() and we see
62       // bar(), we connect foo -> bar.
63       private final ArrayList<Name> currentFunctionStack = new ArrayList<>();

65       NameReferenceGraphConstruction(AbstractCompiler compiler) {
66         this.compiler = compiler;
67         this.graph = new NameReferenceGraph(compiler);
68       }

70       NameReferenceGraph getNameReferenceGraph() {

74       @Override
75       public void process(Node externs, Node root) {
76         ScopeCreator scopeCreator = new MemoizedScopeCreator(new TypedScopeCreator(compiler));
77         NodeTraversal externsTraversal = new NodeTraversal(compiler,
78             new Traversal(true), scopeCreator);
79         NodeTraversal codeTraversal = new NodeTraversal(compiler,
80             new Traversal(false), scopeCreator);
81         Scope topScope = compiler.getTopScope();
82         if (topScope != null) {
83           externsTraversal.traverseWithScope(externs, topScope);
84           codeTraversal.traverseWithScope(root, topScope);
85         } else {
86           externsTraversal.traverse(externs);
87           codeTraversal.traverse(root);
88         }
89         connectUnknowns();
90       }

92       private class Traversal implements ScopedCallback {

94         final boolean isExtern;
95     
96         private Traversal(boolean isExtern) {
97           this.isExtern = isExtern;
98           pushContainingFunction(graph.main);
99         }

101         @Override
102         public void enterScope(NodeTraversal t) {
103           Node root = t.getScopeRoot();
104           Node parent = root.getParent();
105     
106           // When we are not in a {{GLOBAL MAIN}}, we need to determine what the
107           // current function is.
108           if (!t.inGlobalScope()) {
109     
110             // TODO(user): A global function foo() is treated as the same
111             // function as a inner function named foo(). We should use some clever
112             // naming scheme to avoid this lost of precision.
113             String name = NodeUtil.getName(root);
114     
115             if (name == null) {
116               // When the name is null, we have a function that is presumably not
117               // reference-able again and should not be modeled in the name graph.
118               // A common example would be (function() { ... })();
119               pushContainingFunction(graph.unknown);
120               return;
121             }
122     
123             // If we've done type analysis, then we should be able to get the
124             // correct JSFunctionType for the containing function.  If not,
125             // we're probably going to get an unknown type here.
126             JSType type = getType(root);
127     
128             if (parent.isAssign() &&
129                 NodeUtil.isPrototypeProperty(parent.getFirstChild())) {
130               pushContainingFunction(
131                   recordPrototypePropDefinition(parent.getFirstChild(), type, parent));
132             } else {
133               pushContainingFunction(
134                   recordStaticNameDefinition(
135                     name, type, root, root.getLastChild()));
136             }
137           }
138         }
139     
140         @Override
141         public void exitScope(NodeTraversal t) {
142           if (!t.inGlobalScope()) {
143             popContainingFunction();
144           }
145         }

147         @Override
148         public boolean shouldTraverse(NodeTraversal t, Node n, Node parent) {
149           return true;
150         }

152         @SuppressWarnings("fallthrough")
153         @Override
154         public void visit(NodeTraversal t, Node n, Node parent) {
155           switch (n.getType()) {
156             case NAME:
157             case GETPROP:
158               if (parent.isGetProp()) {
159                 // We will resolve this when we visit parent later in the traversal.
160                 return;
161               } else if (parent.isFunction()) {
162                 // Function declarations have been taken care of in enterScope();
163                 return;
164               } else if (parent.isAssign()) {
165                 // Handled below.
166                 return;
167               }
168     
169               if (isLocalNameReference(t, n)) {
170                 // Ignore all local variable references unless is creates a closure.
171                 return;
172               }
173     
174               if (isPrototypeNameReference(n)) {
175                 recordPrototypePropUse(n);
176               } else if (isStaticNameReference(n, t.getScope())) {
177                 recordStaticNameUse(n);
178               } else {
179                 recordUnknownUse(n);
180               }
181               break;
182     
183             case ASSIGN:
184               Node lhs = n.getFirstChild();
185               Node rhs = n.getLastChild();
186               if (rhs.isFunction()) {
187                 // These are recorded when entering the scope.
188                 return;
189               }
190               if (lhs.isName() ||
191                   lhs.isGetProp() ||
192                   rhs.isGetProp()) {
193                 if (NodeUtil.isPrototypeProperty(lhs)) {
194                   Name name = recordPrototypePropDefinition(
195                       lhs, getType(rhs), n);
196                   name.setAliased(true);
197                 }
198               }
199               maybeAliasNamesOnAssign(lhs, rhs);
200               break;
201     
202             case VAR:
203               // var foo = bar;
204               Node varName = n.getFirstChild();
205               Node assignedValue = varName.getFirstChild();
206               if (assignedValue == null) {
207                 return;
208               }
209               maybeAliasNamesOnAssign(varName, assignedValue);
210               break;
211     
212             case CALL:
213               Node param = n.getFirstChild();
214               // We need to alias every name that is passed as a parameter because
215               // they have different names inside the function's scope.
216               while ((param = param.getNext()) != null) {
217                 if (param.isName() || param.isGetProp()) {
218                   safeAlias(param);
219                 }
220               }
221     
222               maybeRecordExport(n);
223               break;
224           }
225         }
226     
227         private boolean containsName(Node n) {
228           return NodeUtil.containsType(n, Token.NAME) ||
229               NodeUtil.containsType(n, Token.GETELEM) ||
230               NodeUtil.containsType(n, Token.GETPROP);
231         }
232     
233         /**
234          * Given a node, this alias all the names in the node that need aliasing.
235          * This is safer than just calling getQualifiedName() because it can return
236          * null it several situations.
237          * @param n node to alias
238          */
239         private void safeAlias(Node n) {
240           if (n.isName() || n.isGetProp()) {
241             String name = n.getQualifiedName();
242             // getQualifiedName can return null in cases like bar[0].baz
243             if (name != null) {
244               defineAndAlias(name);
245               return;
246             }
247           }
248     
249           if (n.isGetProp()) {
250             // var foo = bar[0].baz;
251             defineAndAlias(n.getLastChild().getString());
252           } else if (n.isAssign()) {
253             // In case of nested assignment, we only consider the name of the
254             // immediate neighbor.
255             safeAlias(n.getFirstChild());
256           } else if (n.hasChildren()) {
257             Node cur = n.getFirstChild();
258             do {
259               safeAlias(cur);
260             } while ((cur = cur.getNext()) != null);
261           } else {
262             // No name to alias
263           }
264         }
265     
266         private void maybeAliasNamesOnAssign(Node lhs, Node rhs) {
267           if ((lhs.isName() || lhs.isGetProp()) &&
268               containsName(rhs) &&
269               !rhs.isFunction() &&
270               !rhs.isNew()) {
271             safeAlias(lhs);
272             safeAlias(rhs);
273           }
274         }
275     
276         private void defineAndAlias(String name) {
277           graph.defineNameIfNotExists(name, isExtern).setAliased(true);
278         }
279     
280         private void maybeRecordExport(Node call) {
281           Preconditions.checkArgument(call.isCall());
282           Node getProp = call.getFirstChild();
283           if (!getProp.isGetProp()) {
284             return;
285           }
286     
287           String propQName = getProp.getQualifiedName();
288     
289           if (propQName == null) {
290             return;
291           }
292     
293           // Keep track of calls to "call" and "apply" because they mess up the name
294           // graph.
295           if (propQName.endsWith(".call") || propQName.endsWith(".apply")) {
296             graph.defineNameIfNotExists(getProp.getFirstChild().getQualifiedName(),
297                 isExtern).markExposedToCallOrApply();
298           }
299     
300           if (!"goog.exportSymbol".equals(propQName)) {
301             return;
302           }
303     
304           Node symbol = getProp.getNext();
305           if (!symbol.isString()) {
306             return;
307           }
308     
309           Node obj = symbol.getNext();
310           String qName = obj.getQualifiedName();
311     
312           if (qName == null || obj.getNext() != null) {
313             return;
314           }
315     
316           graph.defineNameIfNotExists(qName, false).markExported();
317         }

319         /**
320          * @return true if n MUST be a local name reference.
321          */
322         private boolean isLocalNameReference(NodeTraversal t, Node n) {
323           // TODO(user): What happen if it is a reference to an outer local
324           // variable (closures)?
325           if (n.isName()) {
326             Var v = t.getScope().getVar(n.getString());
327             return v != null && v.isLocal();
328           }
329           return false;
330         }

332         /**
333          * @return true if n MUST be a static name reference.
334          */
335         private boolean isStaticNameReference(Node n, Scope scope) {
336           Preconditions.checkArgument(n.isName() || n.isGetProp());
337           if (n.isName()) {
338             return true;
339           }
340           String qName = n.getQualifiedName();
341           if (qName == null) {
342             return false;
343           }
344           // TODO(user): This does not always work due to type system bugs.
345           return scope.isDeclared(qName, true);
346         }
347     
348         /**
349          * @return true if n MUST be a prototype name reference.
350          */
351         private boolean isPrototypeNameReference(Node n) {
352           if (!n.isGetProp()) {
353             return false;
354           }
355           JSType type = getType(n.getFirstChild());
356           if (type.isUnknownType() || type.isUnionType()) {
357             return false;
358           }
359           return (type.isInstanceType() || type.autoboxesTo() != null);
360         }
361     
362         private Name recordStaticNameDefinition(String name, JSType type,
363             Node n, Node rValue) {
364           if (getNamedContainingFunction() != graph.main) {
365             // TODO(user): if A.B() defines A.C(), there is a dependence from
366             // A.C() -> A.B(). However, this is not important in module code motion
367             // and will be ignored (for now).
368           }
369           if (type.isConstructor()) {
370             return recordClassConstructorOrInterface(
371                 name, type.toMaybeFunctionType(),
372                 n, rValue);
373           } else {
374             Name symbol = graph.defineNameIfNotExists(name, isExtern);
375             symbol.setType(type);
376             if (n.isAssign()) {
377               symbol.addAssignmentDeclaration(n);
378             } else {
379               symbol.addFunctionDeclaration(n);
380             }
381             return symbol;
382           }
383         }

385         /**
386          * @param assign The assignment node, null if it is just a "forward"
387          *     declaration for recording the rValue's type.
388          */
389         private Name recordPrototypePropDefinition(
390             Node qName, JSType type, @Nullable Node assign) {
391           JSType constructor = getType(NodeUtil.getPrototypeClassName(qName));
392           FunctionType classType = null;
393           String className = null;
394     
395           if (constructor != null && constructor.isConstructor()) {
396             // Case where the class has been properly declared with @constructor
397             classType = constructor.toMaybeFunctionType();
398             className = classType.getReferenceName();
399           } else {
400             // We'll guess it is a constructor even if it didn't have @constructor
401             classType = compiler.getTypeIRegistry()
402                 .getNativeFunctionType(JSTypeNative.U2U_CONSTRUCTOR_TYPE);
403             className = NodeUtil.getPrototypeClassName(qName).getQualifiedName();
404           }
405           // In case we haven't seen the function yet.
406           recordClassConstructorOrInterface(
407               className, classType, null, null);
408     
409           String qNameStr = className + ".prototype." +
410               NodeUtil.getPrototypePropertyName(qName);
411           Name prototypeProp = graph.defineNameIfNotExists(qNameStr, isExtern);
412           Preconditions.checkNotNull(prototypeProp,
413               "%s should be in the name graph as a node.", qNameStr);
414           if (assign != null) {
415             prototypeProp.addAssignmentDeclaration(assign);
416           }
417           prototypeProp.setType(type);
418           return prototypeProp;
419         }

421         private Reference recordStaticNameUse(Node n) {
422           if (isExtern) {
423             // Don't count reference in extern as a use.
424             return null;
425           } else {
426             Reference reference = new Reference(n);
427             Name name = graph.defineNameIfNotExists(n.getQualifiedName(), isExtern);
428             name.setType(getType(n));
429             graph.connect(getNamedContainingFunction(), reference, name);
430             return reference;
431           }
432         }
433     
434         private void recordPrototypePropUse(Node n) {
435           Preconditions.checkArgument(n.isGetProp());
436           Node instance = n.getFirstChild();
437           JSType instanceType = getType(instance);
438           JSType boxedType = instanceType.autoboxesTo();
439           instanceType = boxedType != null ? boxedType : instanceType;
440     
441           // Retrieves the property.
442           ObjectType objType = instanceType.toObjectType();
443           Preconditions.checkState(objType != null);
444     
445           if (!isExtern) {
446             // Don't count reference in extern as a use.
447             Reference ref = new Reference(n);
448     
449             FunctionType constructor = objType.getConstructor();
450             if (constructor != null) {
451               String propName = n.getLastChild().getString();
452               if (!constructor.getPrototype().hasOwnProperty(propName)) {
453                 recordSuperClassPrototypePropUse(constructor, propName, ref);
454               }
455     
456               // TODO(user): TightenType can help a whole lot here.
457               recordSubclassPrototypePropUse(constructor, propName, ref);
458             } else {
459               recordUnknownUse(n);
460             }
461           }
462         }

464         /**
465          * Look for the super class implementation up the tree.
466          */
467         private void recordSuperClassPrototypePropUse(
468             FunctionType classType, String prop, Reference ref) {
469           FunctionType superClass = classType.getSuperClassConstructor();
470           while (superClass != null) {
471             if (superClass.getPrototype().hasOwnProperty(prop)) {
472               graph.connect(getNamedContainingFunction(), ref,
473                   graph.defineNameIfNotExists(
474                      superClass.getReferenceName() + ".prototype." + prop, false));
475               return;
476             } else {
477               superClass = superClass.getSuperClassConstructor();
478             }
479           }
480         }
481     
482         /**
483          * Conservatively assumes that all subclass implementation of this property
484          * might be called.
485          */
486         private void recordSubclassPrototypePropUse(
487             FunctionType classType, String prop, Reference ref) {
488           if (classType.getPrototype().hasOwnProperty(prop)) {
489             graph.connect(getNamedContainingFunction(), ref,
490                graph.defineNameIfNotExists(
491                    classType.getReferenceName() + ".prototype." + prop, false));
492           }
493           if (classType.getSubTypes() != null) {
494             for (FunctionType subclass : classType.getSubTypes()) {
495                 recordSubclassPrototypePropUse(subclass, prop, ref);
496             }
497           }
498         }

500         private void recordUnknownUse(Node n) {
501           if (isExtern) {
502             // Don't count reference in extern as a use.
503             return;
504           } else {
505             Preconditions.checkArgument(n.isGetProp());
506             Reference ref = new Reference(n);
507             unknownNameUse.put(n.getLastChild().getString(),
508                 new NameUse(getNamedContainingFunction(), ref));
509           }
510         }

512         /**
513          * Creates the name in the graph if it does not already exist. Also puts all
514          * the properties and prototype properties of this name in the graph.
515          */
516         private Name recordClassConstructorOrInterface(
517             String name, FunctionType type, @Nullable Node n, @Nullable Node rhs) {
518           Preconditions.checkArgument(type.isConstructor() || type.isInterface());
519           Name symbol = graph.defineNameIfNotExists(name, isExtern);
520           if (rhs != null) {
521             // TODO(user): record the definition.
522             symbol.setType(getType(rhs));
523             if (n.isAssign()) {
524               symbol.addAssignmentDeclaration(n);
525             } else {
526               symbol.addFunctionDeclaration(n);
527             }
528           }
529           ObjectType prototype = type.getPrototype();
530           for (String prop : prototype.getOwnPropertyNames()) {
531             graph.defineNameIfNotExists(
532                 name + ".prototype." + prop, isExtern);
533           }
534           return symbol;
535         }
536       }
537     
538       private void connectUnknowns() {
539         for (GraphNode<Name, Reference> node : graph.getNodes()) {
540           Name name = node.getValue();
541           String propName = name.getPropertyName();
542           if (propName == null) {
543             continue;
544           }
545           Collection<NameUse> uses = unknownNameUse.get(propName);
546           if (uses != null) {
547             for (NameUse use : uses) {
548               graph.connect(use.name, use.reference, name);
549             }
550           }
551         }
552       }

554       /**
555        * A helper to retrieve the type of a node.
556        */
557       private JSType getType(Node n) {
558         JSType type = n.getJSType();
559         if (type == null) {
560           if (CONSERVATIVE) {
561             throw new RuntimeException("Type system failed us :(");
562           } else {
563             return compiler.getTypeIRegistry().getNativeType(JSTypeNative.UNKNOWN_TYPE);
564           }
565         }
566         // Null-ability does not affect the name graph's result.
567         return type.restrictByNotNullOrUndefined();
568       }

570       /**
571        * Mark the provided node as the current function that we are analyzing.
572        * and add it to the stack of scopes we are inside.
573        *
574        * @param functionNode node representing current function.
575        */
576       private void pushContainingFunction(Name functionNode) {
577         currentFunctionStack.add(functionNode);
578       }
579     
580       /**
581        * Remove the top item off the containing function stack, and restore the
582        * previous containing scope to the be the current containing function.
583        */
584       private void popContainingFunction() {
585         currentFunctionStack.remove(currentFunctionStack.size() - 1);
586       }
587     
588       /**
589        * Find the first containing function that's not an function expression
590        * closure.
591        */
592       private Name getNamedContainingFunction() {
593         Name containingFn = null;
594         int pos;
595         for (pos = currentFunctionStack.size() - 1; pos >= 0; pos = pos - 1) {
596           Name cf = currentFunctionStack.get(pos);
597           if (cf != graph.unknown) {
598             containingFn = cf;
599             break;
600           }
601         }
602         Preconditions.checkNotNull(containingFn);
603         return containingFn;
604       }

606       private static class NameUse {
607         private final Name name;
608         private final Reference reference;
609     
610         private NameUse(Name name, Reference reference) {
611           this.name = name;
612           this.reference = reference;
613         }
614       }
