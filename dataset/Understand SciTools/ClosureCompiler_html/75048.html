184       protected void test(CompilerOptions options,
185           String[] original, String[] compiled) {
186         Compiler compiler = compile(options, original);

190             0, compiler.getErrors().length + compiler.getWarnings().length);

192         Node root = compiler.getRoot().getLastChild();

195         assertNull("\n"
196             + "Expected: " + compiler.toSource(expectedRoot) + "\n"
197             + "Result:   " + compiler.toSource(root) + "\n"

222       /**
223        * Asserts that when compiling with the given compiler options,
224        * there is an error or warning.
225        */
226       protected void test(CompilerOptions options,
227           String[] original, String[] compiled, DiagnosticType warning) {
228         Compiler compiler = compile(options, original);
229         checkUnexpectedErrorsOrWarnings(compiler, 1);
230         assertEquals("Expected exactly one warning or error",
231             1, compiler.getErrors().length + compiler.getWarnings().length);
232         if (compiler.getErrors().length > 0) {
233           assertError(compiler.getErrors()[0]).hasType(warning);
234         } else {
235           assertError(compiler.getWarnings()[0]).hasType(warning);
236         }
237     
238         if (compiled != null) {
239           Node root = compiler.getRoot().getLastChild();
240           Node expectedRoot = parseExpectedCode(compiled, options, normalizeResults);
241           String explanation = expectedRoot.checkTreeEquals(root);
242           assertNull("\nExpected: " + compiler.toSource(expectedRoot) +
243               "\nResult: " + compiler.toSource(root) +
244               "\n" + explanation, explanation);
245         }
246       }

255       /**
256        * Asserts that there is at least one parse error.
257        */
258       protected void testParseError(CompilerOptions options,
259           String original, String compiled) {
260         Compiler compiler = compile(options, original);
261         for (JSError error : compiler.getErrors()) {
262           if (!error.getType().equals(RhinoErrorReporter.PARSE_ERROR)) {
263             fail("Found unexpected error type " + error.getType() + ":\n" + error);
264           }
265         }
266         assertEquals("Unexpected warnings: " +
267             Joiner.on("\n").join(compiler.getWarnings()),
268             0, compiler.getWarnings().length);
269     
270         if (compiled != null) {
271           Node root = compiler.getRoot().getLastChild();
272           Node expectedRoot = parseExpectedCode(
273               new String[] {compiled}, options, normalizeResults);
274           String explanation = expectedRoot.checkTreeEquals(root);
275           assertNull("\nExpected: " + compiler.toSource(expectedRoot) +
276               "\nResult: " + compiler.toSource(root) +
277               "\n" + explanation, explanation);
278         }
279       }

299     
300       protected void checkUnexpectedErrorsOrWarnings(
301           Compiler compiler, int expected) {
302         int actual = compiler.getErrors().length + compiler.getWarnings().length;
303         if (actual != expected) {
304           String msg = "";
305           for (JSError err : compiler.getErrors()) {
306             msg += "Error:" + err + "\n";
307           }
308           for (JSError err : compiler.getWarnings()) {
309             msg += "Warning:" + err + "\n";
310           }
311           assertEquals("Unexpected warnings or errors.\n " + msg,
312             expected, actual);
313         }
314       }

320       protected Compiler compile(CompilerOptions options, String[] original) {
321         Compiler compiler = lastCompiler = new Compiler();
322         BlackHoleErrorManager.silence(compiler);
323         compiler.compileModules(
324             externs, ImmutableList.copyOf(CompilerTestCase.createModuleChain(original)),
325             options);
326         return compiler;
327       }
328     
329       /**
330        * Parse the expected code to compare against.
331        * We want to run this with similar parsing options, but don't
332        * want to run the commonjs preprocessing passes (so that we can use this
333        * to test the commonjs code).
334        */
335       protected Node parseExpectedCode(
336           String[] original, CompilerOptions options, boolean normalize) {
337         boolean oldProcessCommonJsModules = options.processCommonJSModules;
338         options.processCommonJSModules = false;
339         Node expectedRoot = parse(original, options, normalize);
340         options.processCommonJSModules = oldProcessCommonJsModules;
341         return expectedRoot;
342       }
343     
344       protected Node parse(
345           String[] original, CompilerOptions options, boolean normalize) {
346         Compiler compiler = new Compiler();
347         List<SourceFile> inputs = new ArrayList<>();
348         for (int i = 0; i < original.length; i++) {
349           inputs.add(SourceFile.fromCode("input" + i, original[i]));
350         }
351         compiler.init(externs, inputs, options);
352         checkUnexpectedErrorsOrWarnings(compiler, 0);
353         Node all = compiler.parseInputs();
354         checkUnexpectedErrorsOrWarnings(compiler, 0);
355         Node n = all.getLastChild();
356         Node externs = all.getFirstChild();
357     
358         (new CreateSyntheticBlocks(
359             compiler, "synStart", "synEnd")).process(externs, n);
360     
361         if (normalize) {
362           compiler.normalize();
363         }
364     
365         return n;
366       }
