1176       /**
1177        * Verifies that the compiler pass's JS output matches the expected output
1178        * and (optionally) that an expected warning is issued. Or, if an error is
1179        * expected, this method just verifies that the error is encountered.
1180        *
1181        * @param compiler A compiler that has been initialized via
1182        *     {@link Compiler#init}
1183        * @param expected Expected output, or null if an error is expected
1184        * @param error Expected error, or null if no error is expected
1185        * @param warning Expected warning, or null if no warning is expected
1186        * @param description The description of the expected warning,
1187        *      or null if no warning is expected or if the warning's description
1188        *      should not be examined
1189        */
1190       private void test(
1191           Compiler compiler,
1192           List<SourceFile> expected,
1193           DiagnosticType error,
1194           DiagnosticType warning,
1195           String description) {
1196         RecentChange recentChange = new RecentChange();
1197         compiler.addChangeHandler(recentChange);
1198     
1199         compiler.getOptions().setNewTypeInference(newTypeInferenceEnabled);
1200     
1201         Node root = compiler.parseInputs();
1202     
1203         String errorMsg = LINE_JOINER.join(compiler.getErrors());
1204         if (root == null && expected == null && error != null) {
1205           // Might be an expected parse error.
1206           assert_().withFailureMessage("Expected one parse error, but got " + errorMsg)
1207               .that(compiler.getErrorCount())
1208               .isEqualTo(1);
1209           JSError actualError = compiler.getErrors()[0];
1210           assert_().withFailureMessage("Unexpected parse error(s): " + errorMsg)
1211               .that(actualError.getType())
1212               .isEqualTo(error);
1213           if (description != null) {
1214             assertThat(actualError.description).isEqualTo(description);
1215           }
1216           return;
1217         }
1218         assert_().withFailureMessage("Unexpected parse error(s): " + errorMsg).that(root).isNotNull();
1219         if (!expectParseWarningsThisTest) {
1220           assertEquals(
1221               "Unexpected parse warning(s): " + LINE_JOINER.join(compiler.getWarnings()),
1222               0,
1223               compiler.getWarnings().length);
1224         } else {
1225           assertThat(compiler.getWarningCount()).isGreaterThan(0);
1226         }
1227     
1228         if (astValidationEnabled) {
1229           (new AstValidator(compiler)).validateRoot(root);
1230         }
1231         Node externsRoot = root.getFirstChild();
1232         Node mainRoot = root.getLastChild();
1233     
1234         // Save the tree for later comparison.
1235         Node rootClone = root.cloneTree();
1236         Node externsRootClone = rootClone.getFirstChild();
1237         Node mainRootClone = rootClone.getLastChild();
1238         Map<Node, Node> mtoc = NodeUtil.mapMainToClone(mainRoot, mainRootClone);
1239     
1240         int numRepetitions = getNumRepetitions();
1241         ErrorManager[] errorManagers = new ErrorManager[numRepetitions];
1242         int aggregateWarningCount = 0;
1243         List<JSError> aggregateWarnings = new ArrayList<>();
1244         boolean hasCodeChanged = false;
1245     
1246         for (int i = 0; i < numRepetitions; ++i) {
1247           if (compiler.getErrorCount() == 0) {
1248             errorManagers[i] = new BlackHoleErrorManager();
1249             compiler.setErrorManager(errorManagers[i]);
1250     
1251             if (rewriteClosureCode && i == 0) {
1252               new ClosureRewriteClass(compiler).process(null, mainRoot);
1253               new ClosureRewriteModule(compiler, null, null).process(null, mainRoot);
1254               new ScopedAliases(compiler, null, CompilerOptions.NULL_ALIAS_TRANSFORMATION_HANDLER)
1255                   .process(null, mainRoot);
1256               hasCodeChanged = hasCodeChanged || recentChange.hasCodeChanged();
1257             }
1258     
1259             // Only run process closure primitives once, if asked.
1260             if (closurePassEnabled && i == 0) {
1261               recentChange.reset();
1262               new ProcessClosurePrimitives(compiler, null, CheckLevel.ERROR, false)
1263                   .process(null, mainRoot);
1264               hasCodeChanged = hasCodeChanged || recentChange.hasCodeChanged();
1265             }
1266     
1267             if (transpileEnabled && i == 0) {
1268               recentChange.reset();
1269               transpileToEs5(compiler, externsRoot, mainRoot);
1270               hasCodeChanged = hasCodeChanged || recentChange.hasCodeChanged();
1271             }
1272     
1273             // Only run the type checking pass once, if asked.
1274             // Running it twice can cause unpredictable behavior because duplicate
1275             // objects for the same type are created, and the type system
1276             // uses reference equality to compare many types.
1277             if (!runTypeCheckAfterProcessing && typeCheckEnabled && i == 0) {
1278               TypeCheck check = createTypeCheck(compiler, reportMissingOverrideCheckLevel);
1279               check.processForTesting(externsRoot, mainRoot);
1280             } else if (!this.runNTIAfterProcessing
1281                 && this.newTypeInferenceEnabled
1282                 && i == 0) {
1283               runNewTypeInference(compiler, externsRoot, mainRoot);
1284             }
1285     
1286             // Only run the normalize pass once, if asked.
1287             if (normalizeEnabled && i == 0) {
1288               normalizeActualCode(compiler, externsRoot, mainRoot);
1289             }
1290     
1291             if (enableInferConsts && i == 0) {
1292               new InferConsts(compiler).process(externsRoot, mainRoot);
1293             }
1294     
1295             if (computeSideEffects && i == 0) {
1296               PureFunctionIdentifier.Driver mark =
1297                   new PureFunctionIdentifier.Driver(compiler, null, false);
1298               mark.process(externsRoot, mainRoot);
1299             }
1300     
1301             if (markNoSideEffects && i == 0) {
1302               MarkNoSideEffectCalls mark = new MarkNoSideEffectCalls(compiler);
1303               mark.process(externsRoot, mainRoot);
1304             }
1305     
1306             if (gatherExternPropertiesEnabled && i == 0) {
1307               (new GatherExternProperties(compiler)).process(externsRoot, mainRoot);
1308             }
1309     
1310             recentChange.reset();
1311     
1312             getProcessor(compiler).process(externsRoot, mainRoot);
1313             if (astValidationEnabled) {
1314               (new AstValidator(compiler)).validateRoot(root);
1315             }
1316             if (checkLineNumbers) {
1317               (new LineNumberCheck(compiler)).process(externsRoot, mainRoot);
1318             }
1319     
1320             if (runTypeCheckAfterProcessing && typeCheckEnabled && i == 0) {
1321               TypeCheck check = createTypeCheck(compiler, reportMissingOverrideCheckLevel);
1322               check.processForTesting(externsRoot, mainRoot);
1323             } else if (this.runNTIAfterProcessing
1324                 && this.newTypeInferenceEnabled
1325                 && i == 0) {
1326               runNewTypeInference(compiler, externsRoot, mainRoot);
1327             }
1328     
1329             hasCodeChanged = hasCodeChanged || recentChange.hasCodeChanged();
1330             aggregateWarningCount += errorManagers[i].getWarningCount();
1331             Collections.addAll(aggregateWarnings, compiler.getWarnings());
1332     
1333             if (normalizeEnabled) {
1334               boolean verifyDeclaredConstants = true;
1335               new Normalize.VerifyConstants(compiler, verifyDeclaredConstants)
1336                   .process(externsRoot, mainRoot);
1337             }
1338           }
1339         }
1340     
1341         if (error == null) {
1342           assertEquals(
1343               "Unexpected error(s):\n" + LINE_JOINER.join(compiler.getErrors()),
1344               0,
1345               compiler.getErrorCount());
1346     
1347           // Verify the symbol table.
1348           ErrorManager symbolTableErrorManager = new BlackHoleErrorManager();
1349           compiler.setErrorManager(symbolTableErrorManager);
1350           Node expectedRoot = null;
1351           if (expected != null) {
1352             expectedRoot = parseExpectedJs(expected);
1353             expectedRoot.detachFromParent();
1354           }
1355     
1356           JSError[] stErrors = symbolTableErrorManager.getErrors();
1357           if (expectedSymbolTableError != null) {
1358             assertEquals("There should be one error.", 1, stErrors.length);
1359             assertThat(stErrors[0].getType()).isEqualTo(expectedSymbolTableError);
1360           } else {
1361             assertEquals(
1362                 "Unexpected symbol table error(s): " + LINE_JOINER.join(stErrors), 0, stErrors.length);
1363           }
1364     
1365           if (warning == null) {
1366             assertEquals(
1367                 "Unexpected warning(s): " + LINE_JOINER.join(aggregateWarnings),
1368                 0,
1369                 aggregateWarningCount);
1370           } else {
1371             assertEquals(
1372                 "There should be one warning, repeated "
1373                     + numRepetitions
1374                     + " time(s). Warnings: \n"
1375                     + LINE_JOINER.join(aggregateWarnings),
1376                 numRepetitions,
1377                 aggregateWarningCount);
1378             for (int i = 0; i < numRepetitions; ++i) {
1379               JSError[] warnings = errorManagers[i].getWarnings();
1380               JSError actual = warnings[0];
1381               assertThat(actual.getType()).isEqualTo(warning);
1382               validateSourceLocation(actual);
1383     
1384               if (description != null) {
1385                 assertThat(actual.description).isEqualTo(description);
1386               }
1387             }
1388           }
1389     
1390           // If we ran normalize on the AST, we must also run normalize on the
1391           // clone before checking for changes.
1392           if (normalizeEnabled) {
1393             normalizeActualCode(compiler, externsRootClone, mainRootClone);
1394           }
1395     
1396           boolean codeChange = !mainRootClone.isEquivalentTo(mainRoot);
1397           boolean externsChange = !externsRootClone.isEquivalentTo(externsRoot);
1398     
1399           // Generally, externs should not be changed by the compiler passes.
1400           if (externsChange && !allowExternsChanges) {
1401             String explanation = externsRootClone.checkTreeEquals(externsRoot);
1402             fail(
1403                 "Unexpected changes to externs"
1404                     + "\nExpected: "
1405                     + compiler.toSource(externsRootClone)
1406                     + "\nResult:   "
1407                     + compiler.toSource(externsRoot)
1408                     + "\n"
1409                     + explanation);
1410           }
1411     
1412           if (!codeChange && !externsChange) {
1413             assertFalse(
1414                 "compiler.reportCodeChange() was called " + "even though nothing changed",
1415                 hasCodeChanged);
1416           } else {
1417             assertTrue(
1418                 "compiler.reportCodeChange() should have been called."
1419                     + "\nOriginal: "
1420                     + mainRootClone.toStringTree()
1421                     + "\nNew: "
1422                     + mainRoot.toStringTree(),
1423                 hasCodeChanged);
1424           }
1425     
1426           // Check correctness of the changed-scopes-only traversal
1427           NodeUtil.verifyScopeChanges(mtoc, mainRoot, false);
1428     
1429           if (expected != null) {
1430             if (compareAsTree) {
1431               String explanation;
1432               if (compareJsDoc) {
1433                 explanation = expectedRoot.checkTreeEqualsIncludingJsDoc(mainRoot);
1434               } else {
1435                 explanation = expectedRoot.checkTreeEquals(mainRoot);
1436               }
1437               if (explanation != null) {
1438                 String expectedAsSource = compiler.toSource(expectedRoot);
1439                 String mainAsSource = compiler.toSource(mainRoot);
1440                 if (expectedAsSource.equals(mainAsSource)) {
1441                   fail("In: " + expectedAsSource + "\n" + explanation);
1442                 } else {
1443                   fail("\nExpected: "
1444                       + expectedAsSource
1445                       + "\nResult:   "
1446                       + mainAsSource
1447                       + "\n" + explanation);
1448                 }
1449               }
1450             } else if (expected != null) {
1451               String[] expectedSources = new String[expected.size()];
1452               for (int i = 0; i < expected.size(); ++i) {
1453                 try {
1454                   expectedSources[i] = expected.get(i).getCode();
1455                 } catch (IOException e) {
1456                   throw new RuntimeException("failed to get source code", e);
1457                 }
1458               }
1459               assertThat(compiler.toSource(mainRoot)).isEqualTo(Joiner.on("").join(expectedSources));
1460             }
1461           }
1462     
1463           // Verify normalization is not invalidated.
1464           Node normalizeCheckRootClone = root.cloneTree();
1465           Node normalizeCheckExternsRootClone = normalizeCheckRootClone.getFirstChild();
1466           Node normalizeCheckMainRootClone = normalizeCheckRootClone.getLastChild();
1467           new PrepareAst(compiler).process(normalizeCheckExternsRootClone, normalizeCheckMainRootClone);
1468           String explanation = normalizeCheckMainRootClone.checkTreeEquals(mainRoot);
1469           assertNull(
1470               "Node structure normalization invalidated."
1471                   + "\nExpected: "
1472                   + compiler.toSource(normalizeCheckMainRootClone)
1473                   + "\nResult:   "
1474                   + compiler.toSource(mainRoot)
1475                   + "\n"
1476                   + explanation,
1477               explanation);
1478     
1479           // TODO(johnlenz): enable this for most test cases.
1480           // Currently, this invalidates test for while-loops, for-loop
1481           // initializers, and other naming.  However, a set of code
1482           // (Closure primitive rewrites, etc) runs before the Normalize pass,
1483           // so this can't be force on everywhere.
1484           if (normalizeEnabled) {
1485             new Normalize(compiler, true)
1486                 .process(normalizeCheckExternsRootClone, normalizeCheckMainRootClone);
1487             explanation = normalizeCheckMainRootClone.checkTreeEquals(mainRoot);
1488             assertNull(
1489                 "Normalization invalidated."
1490                     + "\nExpected: "
1491                     + compiler.toSource(normalizeCheckMainRootClone)
1492                     + "\nResult:   "
1493                     + compiler.toSource(mainRoot)
1494                     + "\n"
1495                     + explanation,
1496                 explanation);
1497           }
1498         } else {
1499           assertNull("expected must be null if error != null", expected);
1500           assertEquals(
1501               "There should be one error of type '" + error.key + "' but there were: "
1502               + Arrays.toString(compiler.getErrors()),
1503               1, compiler.getErrorCount());
1504           JSError actualError = compiler.getErrors()[0];
1505           assertEquals(errorMsg, error, actualError.getType());
1506           validateSourceLocation(actualError);
1507           if (description != null) {
1508             assertThat(actualError.description).isEqualTo(description);
1509           }
1510           assert_()
1511               .withFailureMessage("Some placeholders in the error message were not replaced")
1512               .that(actualError.description)
1513               .doesNotContainMatch("\\{\\d\\}");
1514     
1515           if (warning != null) {
1516             String warnings = "";
1517             for (JSError actualWarning : compiler.getWarnings()) {
1518               warnings += actualWarning.description + "\n";
1519               assert_()
1520                   .withFailureMessage("Some placeholders in the warning message were not replaced")
1521                   .that(actualWarning.description)
1522                   .doesNotContainMatch("\\{\\d\\}");
1523             }
1524             assertEquals("There should be one warning. " + warnings, 1, compiler.getWarningCount());
1525             assertEquals(warnings, warning, compiler.getWarnings()[0].getType());
1526           }
1527         }
1528       }

1567        * Parses expected JS inputs and returns the root of the parse tree.
1568        */
1569       protected Node parseExpectedJs(List<SourceFile> inputs) {
1570         Compiler compiler = createCompiler();
1571     
1572         compiler.init(externsInputs, inputs, getOptions());
1573         Node root = compiler.parseInputs();
1574         assertNotNull("Unexpected parse error(s): " + LINE_JOINER.join(compiler.getErrors()), root);
1575         Node externsRoot = root.getFirstChild();
1576         Node mainRoot = externsRoot.getNext();
1577         // Only run the normalize pass, if asked.
1578         if (normalizeEnabled && !compiler.hasErrors()) {
1579           Normalize normalize = new Normalize(compiler, false);
1580           normalize.process(externsRoot, mainRoot);
1581         }
1582     
1583         if (closurePassEnabled && closurePassEnabledForExpected && !compiler.hasErrors()) {
1584           new ProcessClosurePrimitives(compiler, null, CheckLevel.ERROR, false).process(null, mainRoot);
1585         }
1586     
1587         if (rewriteClosureCode) {
1588           new ClosureRewriteClass(compiler).process(externsRoot, mainRoot);
1589           new ClosureRewriteModule(compiler, null, null).process(externsRoot, mainRoot);
1590           new ScopedAliases(compiler, null, CompilerOptions.NULL_ALIAS_TRANSFORMATION_HANDLER)
1591               .process(externsRoot, mainRoot);
1592         }
1593     
1594         if (transpileEnabled && transpileExpected && !compiler.hasErrors()) {
1595           transpileToEs5(compiler, externsRoot, mainRoot);
1596         }
1597         return mainRoot;
1598       }

1683        * depends on the module before it.

1705       /**
1706        * Generates a list of modules from a list of inputs, such that modules
1707        * form a bush formation. In a bush formation, module 2 depends
1708        * on module 1, and all other modules depend on module 2.
1709        */
1710       static JSModule[] createModuleBush(String... inputs) {
1711         Preconditions.checkState(inputs.length > 2);
1712         JSModule[] modules = createModules(inputs);
1713         for (int i = 1; i < modules.length; i++) {
1714           modules[i].addDependency(modules[i == 1 ? 0 : 1]);
1715         }
1716         return modules;
1717       }
1718     
1719       /**
1720        * Generates a list of modules from a list of inputs, such that modules
1721        * form a tree formation. In a tree formation, module N depends on
1722        * module `floor(N/2)`, So the modules form a balanced binary tree.
1723        */
1724       static JSModule[] createModuleTree(String... inputs) {
1725         JSModule[] modules = createModules(inputs);
1726         for (int i = 1; i < modules.length; i++) {
1727           modules[i].addDependency(modules[(i - 1) / 2]);
1728         }
1729         return modules;
1730       }
1731     
1732       /**
1733        * Generates a list of modules from a list of inputs. Does not generate any
1734        * dependencies between the modules.
1735        */
1736       static JSModule[] createModules(String... inputs) {
1737         JSModule[] modules = new JSModule[inputs.length];
1738         for (int i = 0; i < inputs.length; i++) {
1739           JSModule module = modules[i] = new JSModule("m" + i);
1740           module.add(SourceFile.fromCode("i" + i + ".js", inputs[i]));
1741         }
1742         return modules;
1743       }

1760       public final List<Node> findQualifiedNameNodes(final String name, Node root) {

1762         NodeUtil.visitPostOrder(
1763             root,
1764             new NodeUtil.Visitor() {
1765               @Override

1767                 if (name.equals(n.getQualifiedName())) {

1771             },
1772             Predicates.<Node>alwaysTrue());
1773         return matches;
