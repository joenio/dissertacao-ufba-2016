61     /**
62      * A symbol table for people that want to use Closure Compiler as an indexer.
63      *
64      * Contains an index of all the symbols in the code within a compilation
65      * job. The API is designed for people who want to visit all the symbols, rather
66      * than people who want to lookup a specific symbol by a certain key.
67      *
68      * We can use this to combine different types of symbol tables. For example,
69      * one class might have a {@code StaticSymbolTable} of all variable references,
70      * and another class might have a {@code StaticSymbolTable} of all type names
71      * in JSDoc comments. This class allows you to combine them into a unified
72      * index.
73      *
74      * Most passes build their own "partial" symbol table that implements the same
75      * interface (StaticSymbolTable, StaticSlot, and friends). Individual compiler
76      * passes usually need more or less metadata about the certainty of symbol
77      * information. Building a complete symbol table with all the necessary metadata
78      * for all passes would be too slow. However, as long as these "partial" symbol
79      * tables implement the proper interfaces, we should be able to add them to this
80      * symbol table to make it more complete.
81      *
82      * If clients want fast lookup, they should build their own wrapper around
83      * this symbol table that indexes symbols or references by the desired lookup
84      * key.
85      *
86      * By design, when this symbol table creates symbols for types, it tries
87      * to mimic the symbol table you would get in an OO language. For example,
88      * the "type Foo" and "the constructor that creates objects of type Foo"
89      * are the same symbol. The types of "Foo.prototype" and "new Foo()" also
90      * have the same symbol. Although JSCompiler internally treats these as
91      * distinct symbols, we assume that most clients will not care about
92      * the distinction.
93      *
94      * @see #addSymbolsFrom For more information on how to write plugins for this
95      *    symbol table.
96      *
97      * @author nicksantos@google.com (Nick Santos)
98      */
99     public final class SymbolTable {

188       /**
189        * Gets the scope that contains the given node.
190        * If {@code n} is a function name, we return the scope that contains the
191        * function, not the function itself.
192        */
193       public SymbolScope getEnclosingScope(Node n) {
194         Node current = n.getParent();
195         if (n.isName() &&
196             n.getParent().isFunction()) {
197           current = current.getParent();
198         }
199     
200         for (; current != null; current = current.getParent()) {
201           if (scopes.containsKey(current)) {
202             return scopes.get(current);
203           }
204         }
205         return null;
206       }

208       /**
209        * If {@code sym} is a function, try to find a Symbol for
210        * a parameter with the given name.
211        *
212        * Returns null if we couldn't find one.
213        *
214        * Notice that this just makes a best effort, and may not be able
215        * to find parameters for non-conventional function definitions.
216        * For example, we would not be able to find "y" in this code:
217        * <code>
218        * var x = x() ? function(y) {} : function(y) {};
219        * </code>
220        */
221       public Symbol getParameterInFunction(Symbol sym, String paramName) {
222         SymbolScope scope = getScopeInFunction(sym);
223         if (scope != null) {
224           Symbol param = scope.getSlot(paramName);
225           if (param != null && param.scope == scope) {
226             return param;
227           }
228         }
229         return null;
230       }

232       private SymbolScope getScopeInFunction(Symbol sym) {
233         FunctionType type = sym.getFunctionType();
234         if (type == null) {
235           return null;
236         }
237     
238         Node functionNode = type.getSource();
239         if (functionNode == null) {
240           return null;
241         }
242     
243         return scopes.get(functionNode);
244       }

258       /**
259        * Find the symbol associated with the given scope.
260        * Notice that we won't always be able to figure out this association
261        * dynamically, so sometimes we'll just create the association when we
262        * create the scope.
263        */
264       private Symbol findSymbolForScope(SymbolScope scope) {
265         Node rootNode = scope.getRootNode();
266         if (rootNode.getParent() == null) {
267           return globalScope.getSlot(GLOBAL_THIS);
268         }
269     
270         if (!rootNode.isFunction()) {
271           return null;
272         }
273     
274         String name = NodeUtil.getBestLValueName(
275             NodeUtil.getBestLValue(rootNode));
276         return name == null ? null : scope.getParentScope().getQualifiedSlot(name);
277       }

336       private Symbol getSymbolForName(Node source, String name) {
337         if (name == null || globalScope == null) {
338           return null;
339         }
340     
341         SymbolScope scope = source == null ?
342             globalScope : getEnclosingScope(source);
343     
344         // scope will sometimes be null if one of the type-stripping passes
345         // was run, and the symbol isn't in the AST anymore.
346         return scope == null ? null : scope.getQualifiedSlot(name);
347       }

349       /**
350        * Gets all symbols associated with the given type.
351        * For union types, this may be multiple symbols.
352        * For instance types, this will return the constructor of
353        * that instance.
354        */
355       public List<Symbol> getAllSymbolsForType(JSType type) {
356         if (type == null) {
357           return ImmutableList.of();
358         }
359     
360         UnionType unionType = type.toMaybeUnionType();
361         if (unionType != null) {
362           List<Symbol> result = new ArrayList<>(2);
363           for (JSType alt : unionType.getAlternates()) {
364             // Our type system never has nested unions.
365             Symbol altSym = getSymbolForTypeHelper(alt, true);
366             if (altSym != null) {
367               result.add(altSym);
368             }
369           }
370           return result;
371         }
372         Symbol result = getSymbolForTypeHelper(type, true);
373         return result == null
374             ? ImmutableList.<Symbol>of() : ImmutableList.of(result);
375       }

377       /**
378        * Gets all symbols associated with the given type.
379        * If there is more that one symbol associated with the given type,
380        * return null.
381        * @param type The type.
382        * @param linkToCtor If true, we should link instance types back
383        *     to their constructor function. If false, we should link
384        *     instance types back to their prototype. See the comments
385        *     at the top of this file for more information on how
386        *     our internal type system is more granular than Symbols.
387        */
388       private Symbol getSymbolForTypeHelper(JSType type, boolean linkToCtor) {
389         if (type == null) {
390           return null;
391         }
392     
393         if (type.isGlobalThisType()) {
394           return globalScope.getSlot(GLOBAL_THIS);
395         } else if (type.isNominalConstructor()) {
396           return linkToCtor ?
397               globalScope.getSlot("Function") :
398               getSymbolDeclaredBy(type.toMaybeFunctionType());
399         } else if (type.isFunctionPrototypeType()) {
400           FunctionType ownerFn = ((ObjectType) type).getOwnerFunction();
401           if (!ownerFn.isConstructor() && !ownerFn.isInterface()) {
402             return null;
403           }
404           return linkToCtor ?
405               getSymbolDeclaredBy(ownerFn) :
406               getSymbolForInstancesOf(ownerFn);
407         } else if (type.isInstanceType()) {
408           FunctionType ownerFn = ((ObjectType) type).getConstructor();
409           return linkToCtor ?
410               getSymbolDeclaredBy(ownerFn) :
411               getSymbolForInstancesOf(ownerFn);
412         } else if (type.isFunctionType()) {
413           return linkToCtor ?
414               globalScope.getSlot("Function") :
415               globalScope.getQualifiedSlot("Function.prototype");
416         } else if (type.autoboxesTo() != null) {
417           return getSymbolForTypeHelper(type.autoboxesTo(), linkToCtor);
418         } else {
419           return null;
420         }
421       }

432       private void toDebugString(StringBuilder builder, Symbol symbol) {
433         SymbolScope scope = symbol.scope;
434         if (scope.isGlobalScope()) {
435           builder.append(
436               SimpleFormat.format("'%s' : in global scope:\n", symbol.getName()));
437         } else if (scope.getRootNode() != null) {
438           builder.append(
439               SimpleFormat.format("'%s' : in scope %s:%d\n",
440                   symbol.getName(),
441                   scope.getRootNode().getSourceFileName(),
442                   scope.getRootNode().getLineno()));
443         } else if (scope.getSymbolForScope() != null) {
444           builder.append(
445               SimpleFormat.format("'%s' : in scope %s\n", symbol.getName(),
446                   scope.getSymbolForScope().getName()));
447         } else {
448           builder.append(
449               SimpleFormat.format("'%s' : in unknown scope\n", symbol.getName()));
450         }
451     
452         int refCount = 0;
453         for (Reference ref : getReferences(symbol)) {
454           builder.append(
455               SimpleFormat.format("  Ref %d: %s:%d\n",
456                   refCount,
457                   ref.getNode().getSourceFileName(),
458                   ref.getNode().getLineno()));
459           refCount++;
460         }
461       }

495        * Finds anonymous functions in local scopes, and gives them names
496        * and symbols. They will show up as local variables with names
497        * "function%0", "function%1", etc.
498        */
499       public void addAnonymousFunctions() {
500         TreeSet<SymbolScope> scopes = new TreeSet<>(lexicalScopeOrdering);
501         for (SymbolScope scope : getAllScopes()) {
502           if (scope.isLexicalScope()) {
503             scopes.add(scope);
504           }
505         }
506     
507         for (SymbolScope scope : scopes) {
508           addAnonymousFunctionsInScope(scope);
509         }
510       }

512       private void addAnonymousFunctionsInScope(SymbolScope scope) {
513         Symbol sym = getSymbolForScope(scope);
514         if (sym == null) {
515           // JSCompiler has no symbol for this scope. Check to see if it's a
516           // local function. If it is, give it a name.
517           if (scope.isLexicalScope() &&
518               !scope.isGlobalScope() &&
519               scope.getRootNode() != null &&
520               !scope.getRootNode().isFromExterns() &&
521               scope.getParentScope() != null) {
522             SymbolScope parent = scope.getParentScope();
523     
524             int count = parent.innerAnonFunctionsWithNames++;
525             String innerName = "function%" + count;
526             scope.setSymbolForScope(
527                 declareInferredSymbol(
528                     parent, innerName, scope.getRootNode()));
529           }
530         }
531       }

533       /**
534        * Make sure all the symbols and references in {@code otherSymbolTable}
535        * are in this symbol table.
536        *
537        * Uniqueness of symbols and references is determined by the associated
538        * node.
539        *
540        * If multiple symbol tables are mixed in, we do not check for consistency
541        * between symbol tables. The first symbol we see dictates the type
542        * information for that symbol.
543        */
544       <S extends StaticSlot, R extends StaticRef>
545       void addSymbolsFrom(StaticSymbolTable<S, R> otherSymbolTable) {
546         for (S otherSymbol : otherSymbolTable.getAllSymbols()) {
547           String name = otherSymbol.getName();
548           SymbolScope myScope = createScopeFrom(
549               otherSymbolTable.getScope(otherSymbol));
550     
551           StaticRef decl = findBestDeclToAdd(otherSymbolTable, otherSymbol);
552           Symbol mySymbol = null;
553           if (decl != null) {
554             Node declNode = decl.getNode();
555     
556             // If we have a declaration node, we can ensure the symbol is declared.
557             mySymbol = isAnySymbolDeclared(name, declNode, myScope);
558             if (mySymbol == null) {
559               mySymbol = copySymbolTo(otherSymbol, declNode, myScope);
560             }
561           } else {
562             // If we don't have a declaration node, we won't be able to declare
563             // a symbol in this symbol table. But we may be able to salvage the
564             // references if we already have a symbol.
565             mySymbol = myScope.getOwnSlot(name);
566           }
567     
568           if (mySymbol != null) {
569             for (R otherRef : otherSymbolTable.getReferences(otherSymbol)) {
570               if (isGoodRefToAdd(otherRef)) {
571                 mySymbol.defineReferenceAt(otherRef.getNode());
572               }
573             }
574           }
575         }
576       }

578       /**
579        * Checks if any symbol is already declared at the given node and scope
580        * for the given name. If so, returns it.
581        */
582       private Symbol isAnySymbolDeclared(
583           String name, Node declNode, SymbolScope scope) {
584         Symbol sym = symbols.get(declNode, name);
585         if (sym == null) {
586           // Sometimes, our symbol tables will disagree on where the
587           // declaration node should be. In the rare case where this happens,
588           // trust the existing symbol.
589           // See SymbolTableTest#testDeclarationDisagreement.
590           return scope.ownSymbols.get(name);
591         }
592         return sym;
593       }
594     
595       /** Helper for addSymbolsFrom, to determine the best declaration spot. */
596       private <S extends StaticSlot, R extends StaticRef>
597       StaticRef findBestDeclToAdd(StaticSymbolTable<S, R> otherSymbolTable, S slot) {
598         StaticRef decl = slot.getDeclaration();
599         if (isGoodRefToAdd(decl)) {
600           return decl;
601         }
602     
603         for (R ref : otherSymbolTable.getReferences(slot)) {
604           if (isGoodRefToAdd(ref)) {
605             return ref;
606           }
607         }
608     
609         return null;
610       }

612       /**
613        * Helper for addSymbolsFrom, to determine whether a reference is
614        * acceptable. A reference must be in the normal source tree.
615        */
616       private boolean isGoodRefToAdd(@Nullable StaticRef ref) {
617         return ref != null && ref.getNode() != null
618             && ref.getNode().getStaticSourceFile() != null
619             && !Compiler.SYNTHETIC_EXTERNS.equals(
620                 ref.getNode().getStaticSourceFile().getName());
621       }

627       private Symbol copySymbolTo(
628           StaticSlot sym, Node declNode, SymbolScope scope) {
629         // All symbols must have declaration nodes.

636       private Symbol addSymbol(
637           String name, JSType type, boolean inferred, SymbolScope scope,
638           Node declNode) {
639         Symbol symbol = new Symbol(name, type, inferred, scope);
640         Symbol replacedSymbol = symbols.put(declNode, name, symbol);
641         Preconditions.checkState(
642             replacedSymbol == null,
643             "Found duplicate symbol %s in global index. Type %s", name, type);
644     
645         replacedSymbol = scope.ownSymbols.put(name, symbol);
646         Preconditions.checkState(
647             replacedSymbol == null,
648             "Found duplicate symbol %s in its scope. Type %s", name, type);
649         return symbol;
650       }

652       private Symbol declareSymbol(
653           String name, JSType type, boolean inferred,
654           SymbolScope scope, Node declNode, JSDocInfo info) {
655         Symbol symbol = addSymbol(name, type, inferred, scope, declNode);
656         symbol.setJSDocInfo(info);
657         symbol.setDeclaration(symbol.defineReferenceAt(declNode));
658         return symbol;
659       }

661       private void removeSymbol(Symbol s) {
662         SymbolScope scope = getScope(s);
663         if (scope.ownSymbols.remove(s.getName()) != s) {
664           throw new IllegalStateException("Symbol not found in scope " + s);
665         }
666         if (symbols.remove(s.getDeclaration().getNode(), s.getName()) != s) {
667           throw new IllegalStateException("Symbol not found in table " + s);
668         }
669       }

671       /**
672        * Not all symbol tables record references to "namespace" objects.
673        * For example, if you have:
674        * goog.dom.DomHelper = function() {};
675        * The symbol table may not record that as a reference to "goog.dom",
676        * because that would be redundant.
677        */
678       void fillNamespaceReferences() {
679         for (Symbol symbol : getAllSymbolsSorted()) {
680           String qName = symbol.getName();
681           int rootIndex = qName.indexOf('.');
682           if (rootIndex == -1) {
683             continue;
684           }
685     
686           Symbol root = symbol.scope.getQualifiedSlot(
687               qName.substring(0, rootIndex));
688           if (root == null) {
689             // In theory, this should never happen, but we fail quietly anyway
690             // just to be safe.
691             continue;
692           }
693     
694           for (Reference ref : getReferences(symbol)) {
695             Node currentNode = ref.getNode();
696             if (!currentNode.isQualifiedName()) {
697               continue;
698             }
699     
700             while (currentNode.isGetProp()) {
701               currentNode = currentNode.getFirstChild();
702     
703               String name = currentNode.getQualifiedName();
704               if (name != null) {
705                 Symbol namespace =
706                     isAnySymbolDeclared(name, currentNode, root.scope);
707                 if (namespace == null) {
708                   namespace = root.scope.getQualifiedSlot(name);
709                 }
710     
711                 if (namespace == null && root.scope.isGlobalScope()) {
712                   namespace = declareSymbol(name,
713                       registry.getNativeType(JSTypeNative.UNKNOWN_TYPE),
714                       true,
715                       root.scope,
716                       currentNode,
717                       null /* JsDoc info */);
718                 }
719     
720                 if (namespace != null) {
721                   namespace.defineReferenceAt(currentNode);
722                 }
723               }
724             }
725           }
726         }
727       }

729       void fillPropertyScopes() {
730         // Collect all object symbols.
731         // All symbols that came from goog.module are collected separately because they will have to
732         // be processed first. See explanation below.
733         List<Symbol> types = new ArrayList<>();
734         List<Symbol> googModuleExportTypes = new ArrayList<>();
735     
736         // Create a property scope for each named type and each anonymous object,
737         // and populate it with that object's properties.
738         //
739         // We notably don't want to create a property scope for 'x' in
740         // var x = new Foo();
741         // where x is just an instance of another type.
742         for (Symbol sym : getAllSymbols()) {
743           if (needsPropertyScope(sym)) {
744             if (sym.getName().startsWith("module$exports")) {
745               googModuleExportTypes.add(sym);
746             } else {
747               types.add(sym);
748             }
749           }
750         }
751     
752         // The order of operations here is significant.
753         //
754         // When we add properties to Foo, we'll remove Foo.prototype from
755         // the symbol table and replace it with a fresh symbol in Foo's
756         // property scope. So the symbol for Foo.prototype in
757         // {@code instances} will be stale.
758         //
759         // To prevent this, we sort the list by the reverse of the
760         // default symbol order, which will do the right thing. Essentially going from leaf symbols
761         // to roots.
762         //
763         // Also sorting all symbols is not enough. There is a tricky case with symbols declared in
764         // goog.module that also has declareLegacyNamespace. Example:
765         //
766         // goog.module('x.y');
767         // goog.module.declareLegacyNamespace();
768         // exports.foo = function() {};
769         //
770         // Symbols are following:
771         // x.y
772         // x
773         // module$exports$x$y.foo
774         // module$exports$x$y
775         //
776         // If we order them in reverse lexicographical order symbols x.y and x will be processed before
777         // foo. This is wrong as foo is in fact property of x.y namespace. So we must process all
778         // module$exports$ symbols first. That's why we collected them in separate list.
779         //
780         Collections.sort(types, getNaturalSymbolOrdering().reverse());
781         Collections.sort(googModuleExportTypes, getNaturalSymbolOrdering().reverse());
782         for (Symbol s : Iterables.concat(googModuleExportTypes, types)) {
783           createPropertyScopeFor(s);
784         }
785     
786         pruneOrphanedNames();
787       }

789       private boolean needsPropertyScope(Symbol sym) {
790         ObjectType type = ObjectType.cast(getType(sym));
791         if (type == null) {
792           return false;
793         }
794     
795         // Anonymous objects
796         if (type.getReferenceName() == null) {
797           return true;
798         }
799     
800         // Constructors/prototypes
801         // Should this check for
802         // (type.isNominalConstructor() || type.isFunctionPrototypeType())
803         // ?
804         if (sym.getName().equals(type.getReferenceName())) {
805           return true;
806         }
807     
808         // Enums
809         return type.isEnumType()
810             && sym.getName().equals(type.toMaybeEnumType().getElementsType().getReferenceName());
811       }

813       /**
814        * Removes symbols where the namespace they're on has been removed.
815        *
816        * After filling property scopes, we may have two symbols represented
817        * in different ways. For example, "A.superClass_.foo" and B.prototype.foo".
818        *
819        * This resolves that ambiguity by pruning the duplicates.
820        * If we have a lexical symbol with a constructor in its property
821        * chain, then we assume there's also a property path to this symbol.
822        * In other words, we can remove "A.superClass_.foo" because it's rooted
823        * at "A", and we built a property scope for "A" above.
824        */
825       void pruneOrphanedNames() {
826         nextSymbol: for (Symbol s : getAllSymbolsSorted()) {
827           if (s.isProperty()) {
828             continue;
829           }
830     
831           String currentName = s.getName();
832           int dot = -1;
833           while (-1 != (dot = currentName.lastIndexOf('.'))) {
834             currentName = currentName.substring(0, dot);
835     
836             Symbol owner = s.scope.getQualifiedSlot(currentName);
837             if (owner != null
838                 && getType(owner) != null
839                 && (getType(owner).isNominalConstructor() ||
840                     getType(owner).isFunctionPrototypeType() ||
841                     getType(owner).isEnumType())) {
842               removeSymbol(s);
843               continue nextSymbol;
844             }
845           }
846         }
847       }

849       /**
850        * Create symbols and references for all properties of types in
851        * this symbol table.
852        *
853        * This gets a little bit tricky, because of the way this symbol table
854        * conflates "type Foo" and "the constructor of type Foo". So if you
855        * have:
856        *
857        * <code>
858        * SymbolTable symbolTable = for("var x = new Foo();");
859        * Symbol x = symbolTable.getGlobalScope().getSlot("x");
860        * Symbol type = symbolTable.getAllSymbolsForType(getType(x)).get(0);
861        * </code>
862        *
863        * Then type.getPropertyScope() will have the properties of the
864        * constructor "Foo". To get the properties of instances of "Foo",
865        * you will need to call:
866        *
867        * <code>
868        * Symbol instance = symbolTable.getSymbolForInstancesOf(type);
869        * </code>
870        *
871        * As described at the top of this file, notice that "new Foo()" and
872        * "Foo.prototype" are represented by the same symbol.
873        */
874       void fillPropertySymbols(
875           AbstractCompiler compiler, Node externs, Node root) {
876         (new PropertyRefCollector(compiler)).process(externs, root);

879       /** Index JSDocInfo. */
880       void fillJSDocInfo(
881           AbstractCompiler compiler, Node externs, Node root) {
882         NodeTraversal.traverseRoots(
883             compiler, new JSDocInfoCollector(compiler.getTypeRegistry()), externs, root);
884     
885         // Create references to parameters in the JSDoc.
886         for (Symbol sym : getAllSymbolsSorted()) {
887           JSDocInfo info = sym.getJSDocInfo();
888           if (info == null) {
889             continue;
890           }
891     
892           for (Marker marker : info.getMarkers()) {
893             SourcePosition<Node> pos = marker.getNameNode();
894             if (pos == null) {
895               continue;
896             }
897     
898             Node paramNode = pos.getItem();
899             String name = paramNode.getString();
900             Symbol param = getParameterInFunction(sym, name);
901             if (param == null) {
902               // There is no reference to this parameter in the actual JavaScript
903               // code, so we'll try to create a special JsDoc-only symbol in
904               // a JsDoc-only scope.
905               SourcePosition<Node> typePos = marker.getType();
906               JSType type = null;
907               if (typePos != null) {
908                 type = typePos.getItem().getJSType();
909               }
910     
911               if (sym.docScope == null) {
912                 sym.docScope = new SymbolScope(null /* root */,
913                     null /* parent scope */, null /* type of this */, sym);
914               }
915     
916               // Check to make sure there's no existing symbol. In theory, this
917               // should never happen, but we check anyway and fail silently
918               // if our assumptions are wrong. (We do not want to put the symbol
919               // table into an invalid state).
920               Symbol existingSymbol =
921                   isAnySymbolDeclared(name, paramNode, sym.docScope);
922               if (existingSymbol == null) {
923                 declareSymbol(name, type, type == null, sym.docScope, paramNode,
924                     null /* info */);
925               }
926             } else {
927               param.defineReferenceAt(paramNode);
928             }
929           }
930         }
931       }

947       /**
948        * Build a property scope for the given symbol. Any properties of the symbol
949        * will be added to the property scope.
950        *
951        * It is important that property scopes are created in order from the leaves
952        * up to the root, so this should only be called from #fillPropertyScopes.
953        * If you try to create a property scope for a parent before its leaf,
954        * then the leaf will get cut and re-added to the parent property scope,
955        * and weird things will happen.
956        */
957       private void createPropertyScopeFor(Symbol s) {
958         // In order to build a property scope for s, we will need to build
959         // a property scope for all its implicit prototypes first. This means
960         // that sometimes we will already have built its property scope
961         // for a previous symbol.
962         if (s.propertyScope != null) {
963           return;
964         }
965     
966         SymbolScope parentPropertyScope = null;
967         ObjectType type = getType(s) == null ? null : getType(s).toObjectType();
968         if (type == null) {
969           return;
970         }
971     
972         ObjectType proto = type.getParentScope();
973         if (proto != null && proto != type && proto.getConstructor() != null) {
974           Symbol parentSymbol = getSymbolForInstancesOf(proto.getConstructor());
975           if (parentSymbol != null) {
976             createPropertyScopeFor(parentSymbol);
977             parentPropertyScope = parentSymbol.getPropertyScope();
978           }
979         }
980     
981         ObjectType instanceType = type;
982         Iterable<String> propNames = type.getOwnPropertyNames();
983         if (instanceType.isFunctionPrototypeType()) {
984           // Guard against modifying foo.prototype when foo is a regular (non-constructor) function.
985           if (instanceType.getOwnerFunction().hasInstanceType()) {
986             // Merge the properties of "Foo.prototype" and "new Foo()" together.
987             instanceType = instanceType.getOwnerFunction().getInstanceType();
988             propNames = Iterables.concat(propNames, instanceType.getOwnPropertyNames());
989           }
990         }
991     
992         s.setPropertyScope(new SymbolScope(null, parentPropertyScope, type, s));
993         for (String propName : propNames) {
994           StaticSlot newProp = instanceType.getSlot(propName);
995           if (newProp.getDeclaration() == null) {
996             // Skip properties without declarations. We won't know how to index
997             // them, because we index things by node.
998             continue;
999           }
1000     
1001           // We have symbol tables that do not do type analysis. They just try
1002           // to build a complete index of all objects in the program. So we might
1003           // already have symbols for things like "Foo.bar". If this happens,
1004           // throw out the old symbol and use the type-based symbol.
1005           Symbol oldProp = symbols.get(newProp.getDeclaration().getNode(),
1006               s.getName() + "." + propName);
1007           if (oldProp != null) {
1008             removeSymbol(oldProp);
1009           }
1010     
1011           // If we've already have an entry in the table for this symbol,
1012           // then skip it. This should only happen if we screwed up,
1013           // and declared multiple distinct properties with the same name
1014           // at the same node. We bail out here to be safe.
1015           if (symbols.get(newProp.getDeclaration().getNode(),
1016                   newProp.getName()) != null) {
1017             logger.fine("Found duplicate symbol " + newProp);
1018             continue;
1019           }
1020     
1021           Symbol newSym = copySymbolTo(newProp, s.propertyScope);
1022           if (oldProp != null) {
1023             if (newSym.getJSDocInfo() == null) {
1024               newSym.setJSDocInfo(oldProp.getJSDocInfo());
1025             }
1026             newSym.setPropertyScope(oldProp.propertyScope);
1027             for (Reference ref : oldProp.references.values()) {
1028               newSym.defineReferenceAt(ref.getNode());
1029             }
1030           }
1031         }
1032       }

1042       /**
1043        * Given a scope from another symbol table, returns the {@code SymbolScope}
1044        * rooted at the same node. Creates one if it doesn't exist yet.
1045        */
1046       private SymbolScope createScopeFrom(StaticScope otherScope) {
1047         Node otherScopeRoot = otherScope.getRootNode();
1048         SymbolScope myScope = scopes.get(otherScopeRoot);
1049         if (myScope == null) {
1050           StaticScope otherScopeParent = otherScope.getParentScope();
1051     
1052           // If otherScope is a global scope, and we already have a global scope,
1053           // then something has gone seriously wrong.
1054           //
1055           // Not all symbol tables are rooted at the same global node, and
1056           // we do not want to mix and match symbol tables that are rooted
1057           // differently.
1058     
1059           if (otherScopeParent == null) {
1060             // The global scope must be created before any local scopes.
1061             Preconditions.checkState(
1062                 globalScope == null, "Global scopes found at different roots");
1063           }
1064     
1065           myScope = new SymbolScope(
1066               otherScopeRoot,
1067               otherScopeParent == null ? null : createScopeFrom(otherScopeParent),
1068               getTypeOfThis(otherScope),
1069               null);
1070           scopes.put(otherScopeRoot, myScope);
1071           if (myScope.isGlobalScope()) {
1072             globalScope = myScope;
1073           }
1074         }
1075         return myScope;
1076       }

1376         private boolean maybeDefineTypedReference(
1377             Node n, String propName, JSType owner) {
1378           if (owner.isGlobalThisType()) {
1379             Symbol sym = globalScope.getSlot(propName);
1380             if (sym != null) {
1381               sym.defineReferenceAt(n);
1382               return true;
1383             }
1384           } else if (owner.isNominalConstructor()) {
1385             return maybeDefineReference(
1386                 n, propName, getSymbolDeclaredBy(owner.toMaybeFunctionType()));
1387           } else if (owner.isEnumType()) {
1388             return maybeDefineReference(
1389                 n, propName, getSymbolDeclaredBy(owner.toMaybeEnumType()));
1390           } else {
1391             boolean defined = false;
1392             for (Symbol ctor : getAllSymbolsForType(owner)) {
1393               if (maybeDefineReference(
1394                       n, propName, getSymbolForInstancesOf(ctor))) {
1395                 defined = true;
1396               }
1397             }
1398             return defined;
1399           }
1400           return false;
1401         }
1402     
1403         @Override
1404         public void visit(NodeTraversal t, Node n, Node parent) {
1405           // There are two ways to define a property reference:
1406           // 1) As a fully qualified lexical symbol (e.g., x.y)
1407           // 2) As a property of another object (e.g., x's y)
1408           // Property definitions should take precedence over lexical
1409           // definitions. e.g., for "a.b", it's more useful to record
1410           // this as "property b of the type of a", than as "symbol a.b".
1411     
1412           if (n.isGetProp()) {
1413             JSType owner = n.getFirstChild().getJSType();
1414             if (owner != null) {
1415               boolean defined = maybeDefineTypedReference(
1416                   n, n.getLastChild().getString(), owner);
1417     
1418               if (defined) {
1419                 tryRemoveLexicalQualifiedNameRef(n.getQualifiedName(), n);
1420                 return;
1421               }
1422             }
1423     
1424             tryDefineLexicalQualifiedNameRef(n.getQualifiedName(), n);
1425           } else if (n.isStringKey()) {
1426             JSType owner = parent.getJSType();
1427             if (owner != null) {
1428               boolean defined =
1429                   maybeDefineTypedReference(n, n.getString(), owner);
1430     
1431               if (defined) {
1432                 tryRemoveLexicalQualifiedNameRef(
1433                     NodeUtil.getBestLValueName(n), n);
1434                 return;
1435               }
1436             }
1437     
1438             tryDefineLexicalQualifiedNameRef(
1439                 NodeUtil.getBestLValueName(n), n);
1440           }
1441         }
1442       }

1444       private class ThisRefCollector
1445           extends NodeTraversal.AbstractScopedCallback
1446           implements CompilerPass {
1447         private final AbstractCompiler compiler;
1448     
1449         // The 'this' symbols in the current scope chain.
1450         //
1451         // If we don't know how to declare 'this' in a scope chain,
1452         // then null should be on the stack. But this should be a rare
1453         // occurrence. We should strive to always be able to come up
1454         // with some symbol for 'this'.
1455         private final List<Symbol> thisStack = new ArrayList<>();
1456     
1457         ThisRefCollector(AbstractCompiler compiler) {
1458           this.compiler = compiler;
1459         }
1460     
1461         @Override
1462         public void process(Node externs, Node root) {
1463           NodeTraversal.traverseRoots(compiler, this, externs, root);
1464         }
1465     
1466         @Override
1467         public void enterScope(NodeTraversal t) {
1468           Symbol symbol = null;
1469           if (t.inGlobalScope()) {
1470             // Declare the global this at the first input root.
1471             // This is a bizarre place to put it, but we need some
1472             // location with a real file path (because all symbols
1473             // must have a path).
1474             // Note that root.lastChild.firstChild is the first non-extern input.
1475             Node firstInputRoot = t.getScopeRoot().getLastChild().getFirstChild();
1476             if (firstInputRoot != null) {
1477               symbol = addSymbol(
1478                   GLOBAL_THIS,
1479                   registry.getNativeType(JSTypeNative.GLOBAL_THIS),
1480                   false /* declared */,
1481                   globalScope,
1482                   firstInputRoot);
1483               symbol.setDeclaration(new Reference(symbol, firstInputRoot));
1484             }
1485           } else {
1486             // Otherwise, declare a "this" property when possible.
1487             SymbolScope scope = scopes.get(t.getScopeRoot());
1488             Preconditions.checkNotNull(scope);
1489             Symbol scopeSymbol = getSymbolForScope(scope);
1490             if (scopeSymbol != null) {
1491               SymbolScope propScope = scopeSymbol.getPropertyScope();
1492               if (propScope != null) {
1493                 // If a function is assigned multiple times, we only want
1494                 // one addressable "this" symbol.
1495                 symbol = propScope.getOwnSlot("this");
1496                 if (symbol == null) {
1497                   JSType rootType = t.getScopeRoot().getJSType();
1498                   FunctionType fnType = rootType == null
1499                       ? null : rootType.toMaybeFunctionType();
1500                   JSType type = fnType == null
1501                       ? null : fnType.getTypeOfThis();
1502                   symbol = addSymbol(
1503                       "this",
1504                       type,
1505                       false /* declared */,
1506                       scope,
1507                       t.getScopeRoot());
1508                 }
1509     
1510                 // TODO(nicksantos): It's non-obvious where the declaration of
1511                 // the 'this' symbol should be. Figure this out later.
1512               }
1513             }
1514           }
1515     
1516           thisStack.add(symbol);
1517         }
1518     
1519         @Override
1520         public void exitScope(NodeTraversal t) {
1521           thisStack.remove(thisStack.size() - 1);
1522         }
1523     
1524         @Override
1525         public void visit(NodeTraversal t, Node n, Node parent) {
1526           if (!n.isThis()) {
1527             return;
1528           }
1529     
1530           Symbol symbol = Iterables.getLast(thisStack);
1531           if (symbol != null) {
1532             Reference ref = symbol.defineReferenceAt(n);
1533             if (symbol.getDeclaration() == null) {
1534               symbol.setDeclaration(ref);
1535             }
1536           }
1537         }
1538       }

1540       /** Collects references to types in JSDocInfo. */
1541       private class JSDocInfoCollector
1542           extends NodeTraversal.AbstractPostOrderCallback {
1543         private final JSTypeRegistry typeRegistry;
1544     
1545         private JSDocInfoCollector(JSTypeRegistry registry) {
1546           this.typeRegistry = registry;
1547         }
1548     
1549         @Override public void visit(NodeTraversal t, Node n, Node parent) {
1550           if (n.getJSDocInfo() != null) {
1551     
1552             // Find references in the JSDocInfo.
1553             JSDocInfo info = n.getJSDocInfo();
1554             docInfos.add(n);
1555     
1556             for (Node typeAst : info.getTypeNodes()) {
1557               SymbolScope scope = scopes.get(t.getScopeRoot());
1558               visitTypeNode(
1559                   n,
1560                   info.getTemplateTypeNames(),
1561                   scope == null ? globalScope : scope,
1562                   typeAst);
1563             }
1564           }
1565         }

1567         private boolean isNativeSourcelessType(String name) {
1568           switch (name) {
1569             case "null":
1570             case "undefined":
1571             case "void":
1572               return true;
1573     
1574             default:
1575               return false;
1576           }
1577         }
1578     
1579         public void visitTypeNode(Node refNode, ImmutableList<String> templateTypeNames,
1580             SymbolScope scope, Node n) {
1581           if (n.isString()
1582               && !isNativeSourcelessType(n.getString())
1583               && !templateTypeNames.contains(n.getString())) {
1584             Symbol symbol = lookupPossiblyDottedName(scope, n.getString());
1585             if (symbol != null) {
1586               symbol.defineReferenceAt(n);
1587             }
1588           }
1589     
1590           for (Node child = n.getFirstChild();
1591                child != null; child = child.getNext()) {
1592             visitTypeNode(refNode, templateTypeNames, scope, child);
1593           }
1594         }
1595     
1596         // TODO(peterhal): @template types.
1597         private Symbol lookupPossiblyDottedName(SymbolScope scope, String dottedName) {
1598           // Try the dotted name to start.
1599           String[] names = dottedName.split("\\.");
1600           Symbol result = null;
1601           SymbolScope currentScope = scope;
1602           for (int i = 0; i < names.length; i++) {
1603             String name = names[i];
1604             result = currentScope.getSlot(name);
1605             if (result == null) {
1606               break;
1607             }
1608             if (i < (names.length - 1)) {
1609               currentScope = result.getPropertyScope();
1610               if (currentScope == null) {
1611                 result = null;
1612                 break;
1613               }
1614             }
1615           }
1616     
1617           if (result == null) {
1618             // If we can't find this type, it might be a reference to a
1619             // primitive type (like {string}). Autobox it to check.
1620             JSType type = typeRegistry.getType(dottedName);
1621             JSType autobox = type == null ? null : type.autoboxesTo();
1622             result = autobox == null
1623                 ? null : getSymbolForTypeHelper(autobox, true);
1624           }
1625           return result;
1626         }
1627       }

1629       /** Collects the visibility information for each name/property. */
1630       private class VisibilityCollector
1631           extends NodeTraversal.AbstractPostOrderCallback {
1632         private final ImmutableMap<StaticSourceFile, Visibility> fileVisibilityMap;
1633         private final CodingConvention codingConvention;
1634     
1635         private VisibilityCollector(
1636             ImmutableMap<StaticSourceFile, Visibility> fileVisibilityMap,
1637             CodingConvention codingConvention) {
1638           this.fileVisibilityMap = fileVisibilityMap;
1639           this.codingConvention = codingConvention;
1640         }
1641     
1642         @Override public void visit(NodeTraversal t, Node n, Node parent) {
1643           if (n.isName()) {
1644             visitName(t, n);
1645           } else if (n.isGetProp()) {
1646             visitProperty(n, parent);
1647           }
1648         }
1649     
1650         private void visitName(NodeTraversal t, Node n) {
1651           Symbol symbol = symbols.get(n, n.getString());
1652           if (symbol == null) {
1653             return;
1654           }
1655           // Visibility already set.
1656           if (symbol.getVisibility() != null) {
1657             return;
1658           }
1659           Var var = t.getScope().getVar(n.getString());
1660           if (var == null) {
1661             return;
1662           }
1663           Visibility v = AccessControlUtils.getEffectiveNameVisibility(
1664               n, var, fileVisibilityMap);
1665           if (v == null) {
1666             return;
1667           }
1668           symbol.setVisibility(v);
1669         }
1670     
1671         private void visitProperty(Node getprop, Node parent) {
1672           String propertyName = getprop.getLastChild().getString();
1673           Symbol symbol = symbols.get(getprop, propertyName);
1674           if (symbol == null) {
1675             return;
1676           }
1677           // Visibility already set.
1678           if (symbol.getVisibility() != null) {
1679             return;
1680           }
1681           JSType jsType = getprop.getFirstChild().getJSType();
1682           if (jsType == null) {
1683             return;
1684           }
1685           boolean isOverride = parent.getJSDocInfo() != null
1686               && parent.isAssign()
1687               && parent.getFirstChild() == getprop;
1688           if (isOverride) {
1689             // Don't bother with AccessControlUtils for overridden properties.
1690             // AccessControlUtils currently has complicated logic for detecting
1691             // visibility mismatches for overridden properties that is still
1692             // too tightly coupled to CheckAccessControls. TODO(brndn): simplify.
1693             symbol.setVisibility(Visibility.INHERITED);
1694           } else {
1695             ObjectType referenceType = ObjectType.cast(jsType.dereference());
1696             Visibility v = AccessControlUtils.getEffectivePropertyVisibility(
1697                 getprop,
1698                 referenceType,
1699                 fileVisibilityMap,
1700                 codingConvention);
1701             if (v == null) {
1702               return;
1703             }
1704             symbol.setVisibility(v);
1705           }
1706         }
1707       }

1709       // Comparators
1710       private final Ordering<String> sourceNameOrdering =
1711           Ordering.natural().nullsFirst();
1712     
1713       private final Ordering<Node> nodeOrdering = new Ordering<Node>() {
1714         @Override
1715         public int compare(Node a, Node b) {
1716           int result = sourceNameOrdering.compare(
1717               a.getSourceFileName(), b.getSourceFileName());
1718           if (result != 0) {
1719             return result;
1720           }
1721     
1722           // Source position is a bit mask of line in the top 4 bits, so this
1723           // is a quick way to compare order without computing absolute position.
1724           return a.getSourcePosition() - b.getSourcePosition();
1725         }
1726       };
1727     
1728       private final Ordering<SymbolScope> lexicalScopeOrdering =
1729           new Ordering<SymbolScope>() {
1730         @Override
1731         public int compare(SymbolScope a, SymbolScope b) {
1732           Preconditions.checkState(a.isLexicalScope() && b.isLexicalScope(),
1733                                    "We can only sort lexical scopes");
1734           return nodeOrdering.compare(a.getRootNode(), b.getRootNode());
1735         }
1736       };
1737     
1738       private final Ordering<Symbol> symbolOrdering = new Ordering<Symbol>() {
1739         @Override
1740         public int compare(Symbol a, Symbol b) {
1741           SymbolScope scopeA = getScope(a);
1742           SymbolScope scopeB = getScope(b);
1743     
1744           // More deeply nested symbols should go later.
1745           int result = getLexicalScopeDepth(scopeA) - getLexicalScopeDepth(scopeB);
1746           if (result != 0) {
1747             return result;
1748           }
1749     
1750           // After than, just use lexicographic ordering.
1751           // This ensures "a.b" comes before "a.b.c".
1752           return a.getName().compareTo(b.getName());
1753         }
1754       };
1755     
1756       /**
1757        * For a lexical scope, just returns the normal scope depth.
1758        *
1759        * For a property scope, returns the number of scopes we have to search
1760        *     to find the nearest lexical scope, plus that lexical scope's depth.
1761        *
1762        * For a doc info scope, returns 0.
1763        */
1764       private int getLexicalScopeDepth(SymbolScope scope) {
1765         if (scope.isLexicalScope() || scope.isDocScope()) {
1766           return scope.getScopeDepth();
1767         } else {
1768           Preconditions.checkState(scope.isPropertyScope());
1769           Symbol sym = scope.getSymbolForScope();
1770           Preconditions.checkNotNull(sym);
1771           return getLexicalScopeDepth(getScope(sym)) + 1;
1772         }
1773       }
