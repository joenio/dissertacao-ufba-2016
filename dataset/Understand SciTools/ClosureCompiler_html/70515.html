301       /**
302        * A Bind instance or null.
303        *
304        * When seeing an expression exp1.bind(recv, arg1, ...);
305        * we only know that it's a function bind if exp1 has type function.
306        * W/out type info, exp1 has certainly a function type only if it's a
307        * function literal.
308        *
309        * If (the old) type checking has already happened, exp1's type is attached to
310        * the AST node.
311        * When iCheckTypes is true, describeFunctionBind looks for that type.
312        *
313        * The new type inference does not yet attach types to nodes, but we can still
314        * use type information in describeFunctionBind by passing true for
315        * callerChecksTypes.
316        *
317        * @param callerChecksTypes Trust that the caller of this method has verified
318        *        that the bound node has a function type.
319        * @param iCheckTypes Check that the bound node has a function type.
320        */
321       public Bind describeFunctionBind(
322           Node n, boolean callerChecksTypes, boolean iCheckTypes);

324       /** Bind class */
325       public static class Bind {
326         // The target of the bind action
327         final Node target;
328         // The node representing the "this" value, maybe null
329         final Node thisValue;
330         // The head of a Node list representing the parameters
331         final Node parameters;
332     
333         public Bind(Node target, Node thisValue, Node parameters) {
334           this.target = target;
335           this.thisValue = thisValue;
336           this.parameters = parameters;
337         }
338     
339         /**
340          * The number of parameters bound (not including the 'this' value).
341          */
342         int getBoundParameterCount() {
343           if (parameters == null) {
344             return 0;
345           }
346           Node paramParent = parameters.getParent();
347           return paramParent.getChildCount() - paramParent.getIndexOfChild(parameters);
348         }
349       }

419       /** Record subclass relations */
420       static class SubclassRelationship {
421         final SubclassType type;
422         final String subclassName;
423         final String superclassName;
424     
425         public SubclassRelationship(SubclassType type,
426             Node subclassNode, Node superclassNode) {
427           this.type = type;
428           this.subclassName = subclassNode.getQualifiedName();
429           this.superclassName = superclassNode.getQualifiedName();

434        * Delegates provides a mechanism and structure for identifying where classes

474       /**
475        * A function that will throw an exception when either:
476        *   -One or more of its parameters evaluate to false.
477        *   -One or more of its parameters are not of a certain type.
478        */
479       public class AssertionFunctionSpec {
480         protected final String functionName;
481         // Old type system type
482         protected final JSTypeNative assertedType;
483         // New type system type
484         protected final JSType assertedNewType;
485     
486         @Deprecated
487         public AssertionFunctionSpec(String functionName) {
488           this(functionName, JSType.UNKNOWN, null);
489         }
490     
491         public AssertionFunctionSpec(String functionName, JSType assertedNewType) {
492           this(functionName, assertedNewType, null);
493         }
494     
495         public AssertionFunctionSpec(String functionName,
496             JSType assertedNewType, JSTypeNative assertedType) {
497           this.functionName = functionName;
498           this.assertedNewType = assertedNewType;
499           this.assertedType = assertedType;
500         }
501     
502         /** Returns the name of the function. */
503         public String getFunctionName() {
504           return functionName;
505         }
506     
507         /**
508          * Returns the parameter of the assertion function that is being checked.
509          * @param firstParam The first parameter of the function call.
510          */
511         public Node getAssertedParam(Node firstParam) {
512           return firstParam;
513         }
514     
515         /**
516          * Returns the old type system type for a type assertion, or null if
517          * the function asserts that the node must not be null or undefined.
518          * @param call The asserting call
519          */
520         public com.google.javascript.rhino.jstype.JSType
521             getAssertedOldType(Node call, JSTypeRegistry registry) {
522           return assertedType != null ? registry.getNativeType(assertedType) : null;
523         }
524     
525         /**
526          * Returns the new type system type for a type assertion.
527          * @param call The asserting call
528          */
529         public JSType getAssertedNewType(Node call, DeclaredTypeRegistry scope) {
530           return assertedNewType;
531         }
