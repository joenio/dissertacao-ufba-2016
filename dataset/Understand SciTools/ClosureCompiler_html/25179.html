30      * Peephole optimization to remove useless code such as IF's with false
31      * guard conditions, comma operator left hand sides with no side effects, etc.
32      *
33      */

41       @Override
42       Node optimizeSubtree(Node subtree) {
43         switch(subtree.getType()) {
44           case ASSIGN:
45             return tryFoldAssignment(subtree);
46           case COMMA:
47             return tryFoldComma(subtree);
48           case SCRIPT:
49           case BLOCK:
50             return tryOptimizeBlock(subtree);
51           case EXPR_RESULT:
52             subtree = tryFoldExpr(subtree);
53             return subtree;
54           case HOOK:
55             return tryFoldHook(subtree);
56           case SWITCH:
57             return tryOptimizeSwitch(subtree);
58           case IF:
59             return tryFoldIf(subtree);
60           case WHILE:
61             return tryFoldWhile(subtree);
62           case FOR: {
63             Node condition = NodeUtil.getConditionExpression(subtree);
64             if (condition != null) {
65               tryFoldForCondition(condition);
66             }
67             return tryFoldFor(subtree);
68           }
69           case DO:
70             Node foldedDo = tryFoldDoAway(subtree);
71             if (foldedDo.isDo()) {
72               return tryFoldEmptyDo(foldedDo);
73             }
74             return foldedDo;
75     
76           case TRY:
77             return tryFoldTry(subtree);
78           default:
79               return subtree;
80         }
81       }

83       /**
84        * Remove try blocks without catch blocks and with empty or not
85        * existent finally blocks.
86        * Or, only leave the finally blocks if try body blocks are empty
87        * @return the replacement node, if changed, or the original if not
88        */
89       private Node tryFoldTry(Node n) {
90         Preconditions.checkState(n.isTry(), n);
91         Node body = n.getFirstChild();
92         Node catchBlock = body.getNext();
93         Node finallyBlock = catchBlock.getNext();
94     
95         // Removes TRYs that had its CATCH removed and/or empty FINALLY.
96         if (!catchBlock.hasChildren() &&
97             (finallyBlock == null || !finallyBlock.hasChildren())) {
98           n.removeChild(body);
99           n.getParent().replaceChild(n, body);
100           reportCodeChange();
101           return body;
102         }
103     
104         // Only leave FINALLYs if TRYs are empty
105         if (!body.hasChildren()) {
106           NodeUtil.redeclareVarsInsideBranch(catchBlock);
107           if (finallyBlock != null) {
108             n.removeChild(finallyBlock);
109             n.getParent().replaceChild(n, finallyBlock);
110           } else {
111             n.getParent().removeChild(n);
112           }
113           reportCodeChange();
114           return finallyBlock;
115         }
116     
117         return n;
118       }

120       /**
121        * Try removing identity assignments
122        * @return the replacement node, if changed, or the original if not
123        */
124       private Node tryFoldAssignment(Node subtree) {
125         Preconditions.checkState(subtree.isAssign());
126         Node left = subtree.getFirstChild();
127         Node right = subtree.getLastChild();
128         // Only names
129         if (left.isName()
130             && right.isName()
131             && left.getString().equals(right.getString())) {
132           subtree.getParent().replaceChild(subtree, right.detachFromParent());
133           reportCodeChange();
134           return right;
135         }
136         return subtree;
137       }

139       /**
140        * Try folding EXPR_RESULT nodes by removing useless Ops and expressions.
141        * @return the replacement node, if changed, or the original if not
142        */
143       private Node tryFoldExpr(Node subtree) {
144         Node result = trySimplifyUnusedResult(subtree.getFirstChild());
145         if (result == null) {
146           Node parent = subtree.getParent();
147           // If the EXPR_RESULT no longer has any children, remove it as well.
148           if (parent.isLabel()) {
149             Node replacement = IR.block().srcref(subtree);
150             parent.replaceChild(subtree, replacement);
151             subtree = replacement;
152           } else {
153             subtree.detachFromParent();
154             subtree = null;
155           }
156         }
157         return subtree;
158       }

162        * @param n The root of the expression to simplify.
163        * @return The replacement node, or null if the node was is not useful.

166         return trySimplifyUnusedResult(n, true);

169       /**
170        * General cascading unused operation node removal.
171        * @param n The root of the expression to simplify.
172        * @param removeUnused If true, the node is removed from the AST if
173        *     it is not useful, otherwise it replaced with an EMPTY node.
174        * @return The replacement node, or null if the node was is not useful.
175        */
176       private Node trySimplifyUnusedResult(Node n, boolean removeUnused) {
177         Node result = n;
178     
179         // Simplify the results of conditional expressions
180         switch (n.getType()) {
181           case HOOK:
182             Node trueNode = trySimplifyUnusedResult(n.getSecondChild());
183             Node falseNode = trySimplifyUnusedResult(n.getLastChild());
184             // If one or more of the conditional children were removed,
185             // transform the HOOK to an equivalent operation:
186             //    x() ? foo() : 1 --> x() && foo()
187             //    x() ? 1 : foo() --> x() || foo()
188             //    x() ? 1 : 1 --> x()
189             //    x ? 1 : 1 --> null
190             if (trueNode == null && falseNode != null) {
191               n.setType(Token.OR);
192               Preconditions.checkState(n.getChildCount() == 2, n);
193             } else if (trueNode != null && falseNode == null) {
194               n.setType(Token.AND);
195               Preconditions.checkState(n.getChildCount() == 2, n);
196             } else if (trueNode == null && falseNode == null) {
197               result = trySimplifyUnusedResult(n.getFirstChild());
198             } else {
199               // The structure didn't change.
200               result = n;
201             }
202             break;
203           case AND:
204           case OR:
205             // Try to remove the second operand from a AND or OR operations:
206             //    x() || f --> x()
207             //    x() && f --> x()
208             Node conditionalResultNode = trySimplifyUnusedResult(
209                 n.getLastChild());
210             if (conditionalResultNode == null) {
211               Preconditions.checkState(n.hasOneChild(), n);
212               // The conditionally executed code was removed, so
213               // replace the AND/OR with its LHS or remove it if it isn't useful.
214               result = trySimplifyUnusedResult(n.getFirstChild());
215             }
216             break;
217           case FUNCTION:
218             // A function expression isn't useful if it isn't used, remove it and
219             // don't bother to look at its children.
220             result = null;
221             break;
222           case COMMA:
223             // We rewrite other operations as COMMA expressions (which will later
224             // get split into individual EXPR_RESULT statement, if possible), so
225             // we special case COMMA (we don't want to rewrite COMMAs as new COMMAs
226             // nodes.
227             Node left = trySimplifyUnusedResult(n.getFirstChild());
228             Node right = trySimplifyUnusedResult(n.getLastChild());
229             if (left == null && right == null) {
230               result = null;
231             } else if (left == null) {
232               result = right;
233             } else if (right == null){
234               result = left;
235             } else {
236               // The structure didn't change.
237               result = n;
238             }
239             break;
240           default:
241             if (!nodeTypeMayHaveSideEffects(n)) {
242               // This is the meat of this function. The node itself doesn't generate
243               // any side-effects but preserve any side-effects in the children.
244               Node resultList = null;
245               for (Node next, c = n.getFirstChild(); c != null; c = next) {
246                 next = c.getNext();
247                 c = trySimplifyUnusedResult(c);
248                 if (c != null) {
249                   c.detachFromParent();
250                   if (resultList == null)  {
251                     // The first side-effect can be used stand-alone.
252                     resultList = c;
253                   } else {
254                     // Leave the side-effects in-place, simplifying it to a COMMA
255                     // expression.
256                     resultList = IR.comma(resultList, c).srcref(c);
257                   }
258                 }
259               }
260               result = resultList;
261             }
262         }
263     
264         // Fix up the AST, replace or remove the an unused node (if requested).
265         if (n != result) {
266           Node parent = n.getParent();
267           if (result == null) {
268             if (removeUnused) {
269               parent.removeChild(n);
270             } else {
271               result = IR.empty().srcref(n);
272               parent.replaceChild(n, result);
273             }
274           } else {
275             // A new COMMA expression may not have an existing parent.
276             if (result.getParent() != null) {
277               result.detachFromParent();
278             }
279             n.getParent().replaceChild(n, result);
280           }
281           reportCodeChange();
282         }
283     
284         return result;
285       }

291         @Override

299       private void removeIfUnnamedBreak(Node maybeBreak) {
300         if (maybeBreak != null && maybeBreak.isBreak() && !maybeBreak.hasChildren()) {
301           maybeBreak.detachFromParent();
302           reportCodeChange();

306       private Node tryRemoveSwitchWithSingleCase(Node n, boolean shouldIncludeCondition) {
307         Node caseBlock = n.getLastChild().getLastChild();
308         removeIfUnnamedBreak(caseBlock.getLastChild());
309         // Back off if the switch contains statements like "if (a) { break; }"
310         if (NodeUtil.has(caseBlock, MATCH_UNNAMED_BREAK, NodeUtil.MATCH_NOT_FUNCTION)) {
311           return n;
312         }
313         // TODO(moz): This needs to change when we optimize for ES6. The block might need to be
314         // preserved in the presence of block-scoped declarations.
315         if (shouldIncludeCondition) {
316           caseBlock.addChildToFront(IR.exprResult(n.removeFirstChild()).srcref(n));
317         }
318         caseBlock.setIsSyntheticBlock(false);
319         n.getParent().replaceChild(n, caseBlock.detachFromParent());
320         reportCodeChange();
321         return caseBlock;
322       }

324       private Node tryRemoveSwitch(Node n) {
325         if (n.hasOneChild()) {
326           // Remove the switch if there are no remaining cases
327           Node condition = n.removeFirstChild();
328           Node replacement = IR.exprResult(condition).srcref(n);
329           n.getParent().replaceChild(n, replacement);
330           reportCodeChange();
331           return replacement;
332         } else if (n.getChildCount() == 2 && n.getLastChild().isDefaultCase()) {
333           return tryRemoveSwitchWithSingleCase(n, true);
334         } else {
335           return n;
336         }
337       }

339       /**
340        * Remove useless switches and cases.
341        */
342       private Node tryOptimizeSwitch(Node n) {
343         Preconditions.checkState(n.isSwitch(), n);
344     
345         Node defaultCase = tryOptimizeDefaultCase(n);
346     
347         // Generally, it is unsafe to remove other cases when the default case is not the last one.
348         if (defaultCase == null || n.getLastChild().isDefaultCase()) {
349           Node cond = n.getFirstChild(), prev = null, next = null, cur;
350     
351           for (cur = cond.getNext(); cur != null; cur = next) {
352             next = cur.getNext();
353             if (!mayHaveSideEffects(cur.getFirstChild()) && isUselessCase(cur, prev, defaultCase)) {
354               removeCase(n, cur);
355             } else {
356               prev = cur;
357             }
358           }
359     
360           // Optimize switches with constant condition
361           if (NodeUtil.isLiteralValue(cond, false)) {
362             Node caseLabel;
363             TernaryValue caseMatches = TernaryValue.TRUE;
364             // Remove cases until you find one that may match
365             for (cur = cond.getNext(); cur != null; cur = next) {
366               next = cur.getNext();
367               caseLabel = cur.getFirstChild();
368               // TODO(moz): Use type information if allowed.
369               caseMatches = PeepholeFoldConstants.evaluateComparison(
370                   Token.SHEQ, cond, caseLabel, false);
371               if (caseMatches == TernaryValue.TRUE) {
372                 break;
373               } else if (caseMatches == TernaryValue.UNKNOWN) {
374                 break;
375               } else {
376                 removeCase(n, cur);
377               }
378             }
379             if (caseMatches != TernaryValue.UNKNOWN) {
380               Node block, lastStm;
381               // Skip cases until you find one whose last stm is a removable break
382               while (cur != null) {
383                 block = cur.getLastChild();
384                 lastStm = block.getLastChild();
385                 cur = cur.getNext();
386                 if (lastStm != null && isExit(lastStm)) {
387                   removeIfUnnamedBreak(lastStm);
388                   break;
389                 }
390               }
391               // Remove any remaining cases
392               for (; cur != null; cur = next) {
393                 next = cur.getNext();
394                 removeCase(n, cur);
395               }
396               // If there is one case left, we may be able to fold it
397               cur = cond.getNext();
398               if (cur != null && cur.getNext() == null) {
399                 return tryRemoveSwitchWithSingleCase(n, false);
400               }
401             }
402           }
403         }
404     
405         return tryRemoveSwitch(n);
406       }

408       /**
409        * @return the default case node or null if there is no default case or
410        *     if the default case is removed.
411        */
412       private Node tryOptimizeDefaultCase(Node n) {
413         Preconditions.checkState(n.isSwitch(), n);
414     
415         Node lastNonRemovable = n.getFirstChild();  // The switch condition
416     
417         // The first child is the switch conditions skip it when looking for cases.
418         for (Node c = n.getSecondChild(); c != null; c = c.getNext()) {
419           if (c.isDefaultCase()) {
420             // Remove cases that fall-through to the default case
421             Node caseToRemove = lastNonRemovable.getNext();
422             for (Node next; caseToRemove != c; caseToRemove = next) {
423               next = caseToRemove.getNext();
424               removeCase(n, caseToRemove);
425             }
426     
427             // Don't use the switch condition as the previous case.
428             Node prevCase = (lastNonRemovable == n.getFirstChild())
429                 ? null : lastNonRemovable;
430     
431             // Remove the default case if we can
432             if (isUselessCase(c, prevCase, c)) {
433               removeCase(n, c);
434               return null;
435             }
436             return c;
437           } else {
438             Preconditions.checkState(c.isCase());
439             if (c.getLastChild().hasChildren()
440                 || mayHaveSideEffects(c.getFirstChild())) {
441               lastNonRemovable = c;
442             }
443           }
444         }
445         return null;
446       }

458       /**
459        * The function assumes that when checking a CASE node there is no DEFAULT_CASE node in the
460        * SWITCH, or the DEFAULT_CASE is the last case in the SWITCH.
461        *
462        * @return Whether the CASE or DEFAULT_CASE block does anything useful.
463        */
464       private boolean isUselessCase(
465           Node caseNode, @Nullable Node previousCase, @Nullable Node defaultCase) {
466         Preconditions.checkState(
467             previousCase == null || previousCase.getNext() == caseNode);
468         // A case isn't useless if a previous case falls through to it unless it happens to be the last
469         // case in the switch.
470         Node switchNode = caseNode.getParent();
471         if (switchNode.getLastChild() != caseNode
472             && previousCase != null) {
473           Node previousBlock = previousCase.getLastChild();
474           if (!previousBlock.hasChildren()
475               || !isExit(previousBlock.getLastChild())) {
476             return false;
477           }
478         }
479     
480         Node executingCase = caseNode;
481         while (executingCase != null) {
482           Preconditions.checkState(executingCase.isDefaultCase()
483               || executingCase.isCase());
484           // We only expect a DEFAULT case if the case we are checking is the
485           // DEFAULT case.  Otherwise, we assume the DEFAULT case has already
486           // been removed.
487           Preconditions.checkState(caseNode == executingCase
488               || !executingCase.isDefaultCase());
489           Node block = executingCase.getLastChild();
490           Preconditions.checkState(block.isBlock());
491           if (block.hasChildren()) {
492             for (Node blockChild : block.children()) {
493               // If this is a block with a labelless break, it is useless.
494               switch (blockChild.getType()) {
495                 case BREAK:
496                   // A case with a single labelless break is useless if it is the default case or if
497                   // there is no default case. A break to a different control structure isn't useless.
498                   return !blockChild.hasChildren()
499                       && (defaultCase == null || defaultCase == executingCase);
500                 case VAR:
501                   if (blockChild.hasOneChild()
502                       && blockChild.getFirstFirstChild() == null) {
503                     // Variable declarations without initializations are OK.
504                     continue;
505                   }
506                   return false;
507                 default:
508                   return false;
509               }
510             }
511           }
512           // Look at the fallthrough case
513           executingCase = executingCase.getNext();
514         }
515         return true;
516       }

519        * @return Whether the node is an obvious control flow exit.

522         switch (n.getType()) {
523           case BREAK:
524           case CONTINUE:
525           case RETURN:
526           case THROW:
527             return true;
528           default:
529             return false;

533       private Node tryFoldComma(Node n) {
534         // If the left side does nothing replace the comma with the result.
535         Node parent = n.getParent();
536         Node left = n.getFirstChild();
537         Node right = left.getNext();
538     
539         left = trySimplifyUnusedResult(left);
540         if (left == null || !mayHaveSideEffects(left)) {
541           // Fold it!
542           n.removeChild(right);
543           parent.replaceChild(n, right);
544           reportCodeChange();
545           return right;
546         }
547         return n;
548       }

550       /**
551        * Try removing unneeded block nodes and their useless children
552        */
553       Node tryOptimizeBlock(Node n) {
554         // Remove any useless children
555         for (Node c = n.getFirstChild(), prev = null; c != null; ) {
556           Node next = c.getNext();  // save c.next, since 'c' may be removed
557           if (!isUnremovableNode(c) && !mayHaveSideEffects(c)) {
558             // TODO(johnlenz): determine what this is actually removing. Candidates
559             //    include: EMPTY nodes, control structures without children
560             //    (removing infinite loops), empty try blocks.  What else?
561             n.removeFirstOrChildAfter(prev);
562             reportCodeChange();
563           } else {
564             prev = c;
565             tryOptimizeConditionalAfterAssign(c);
566           }
567           c = next;
568         }
569     
570         if (n.isSyntheticBlock() || n.isScript() || n.getParent() == null) {
571           return n;
572         }
573     
574         // Try to remove the block.
575         if (NodeUtil.tryMergeBlock(n)) {
576           reportCodeChange();
577           return null;
578         }
579     
580         return n;
581       }

584        * Some nodes unremovable node don't have side-effects.
585        */
586       private static boolean isUnremovableNode(Node n) {
587         return (n.isBlock() && n.isSyntheticBlock()) || n.isScript();
588       }

590       // TODO(johnlenz): Consider moving this to a separate peephole pass.
591       /**
592        * Attempt to replace the condition of if or hook immediately that is a
593        * reference to a name that is assigned immediately before.
594        */
595       private void tryOptimizeConditionalAfterAssign(Node n) {
596         Node next = n.getNext();
597     
598         // Look for patterns like the following and replace the if-condition with
599         // a constant value so it can later be folded:
600         //   var a = /a/;
601         //   if (a) {foo(a)}
602         // or
603         //   a = 0;
604         //   a ? foo(a) : c;
605         // or
606         //   a = 0;
607         //   a || foo(a);
608         // or
609         //   a = 0;
610         //   a && foo(a)
611         //
612         // TODO(johnlenz): This would be better handled by control-flow sensitive
613         // constant propagation. As the other case that I want to handle is:
614         //   i=0; for(;i<0;i++){}
615         // as right now nothing facilitates removing a loop like that.
616         // This is here simply to remove the cruft left behind goog.userAgent and
617         // similar cases.
618     
619         if (isSimpleAssignment(n) && isConditionalStatement(next)) {
620           Node lhsAssign = getSimpleAssignmentName(n);
621     
622           Node condition = getConditionalStatementCondition(next);
623           if (lhsAssign.isName() && condition.isName()
624               && lhsAssign.getString().equals(condition.getString())) {
625             Node rhsAssign = getSimpleAssignmentValue(n);
626             TernaryValue value = NodeUtil.getImpureBooleanValue(rhsAssign);
627             if (value != TernaryValue.UNKNOWN) {
628               Node replacementConditionNode =
629                   NodeUtil.booleanNode(value.toBoolean(true));
630               condition.getParent().replaceChild(condition,
631                   replacementConditionNode);
632               reportCodeChange();
633             }
634           }
635         }
636       }

638       /**
639        * @return whether the node is a assignment to a simple name, or simple var
640        * declaration with initialization.
641        */
642       private static boolean isSimpleAssignment(Node n) {
643         // For our purposes we define a simple assignment to be a assignment
644         // to a NAME node, or a VAR declaration with one child and a initializer.
645         if (NodeUtil.isExprAssign(n)
646             && n.getFirstFirstChild().isName()) {
647           return true;
648         } else if (n.isVar() && n.hasOneChild() &&
649             n.getFirstFirstChild() != null) {
650           return true;
651         }
652     
653         return false;
654       }

661         if (NodeUtil.isExprAssign(n)) {
662           return n.getFirstFirstChild();
663         } else {

678        * @return Whether the node is a conditional statement.
679        */
680       private boolean isConditionalStatement(Node n) {
681         // We defined a conditional statement to be a IF or EXPR_RESULT rooted with
682         // a HOOK, AND, or OR node.
683         return n != null && (n.isIf() || isExprConditional(n));

686       /**
687        * @return Whether the node is a rooted with a HOOK, AND, or OR node.
688        */
689       private static boolean isExprConditional(Node n) {
690         if (n.isExprResult()) {
691           switch (n.getFirstChild().getType()) {
692             case HOOK:
693             case AND:
694             case OR:
695               return true;
696           }
697         }
698         return false;
699       }

701       /**
702        * @return The condition of a conditional statement.
703        */
704       private Node getConditionalStatementCondition(Node n) {
705         if (n.isIf()) {
706           return NodeUtil.getConditionExpression(n);
707         } else {
708           Preconditions.checkState(isExprConditional(n));
709           return n.getFirstFirstChild();
710         }
711       }

713       /**
714        * Try folding IF nodes by removing dead branches.
715        * @return the replacement node, if changed, or the original if not
716        */
717       private Node tryFoldIf(Node n) {
718         Preconditions.checkState(n.isIf(), n);
719         Node parent = n.getParent();
720         Preconditions.checkNotNull(parent);
721         Token type = n.getType();
722         Node cond = n.getFirstChild();
723         Node thenBody = cond.getNext();
724         Node elseBody = thenBody.getNext();
725     
726         // if (x) { .. } else { } --> if (x) { ... }
727         if (elseBody != null && !mayHaveSideEffects(elseBody)) {
728           n.removeChild(elseBody);
729           elseBody = null;
730           reportCodeChange();
731         }
732     
733         // if (x) { } else { ... } --> if (!x) { ... }
734         if (!mayHaveSideEffects(thenBody) && elseBody != null) {
735           n.removeChild(elseBody);
736           n.replaceChild(thenBody, elseBody);
737           Node notCond = new Node(Token.NOT);
738           n.replaceChild(cond, notCond);
739           notCond.addChildToFront(cond);
740           cond = notCond;
741           thenBody = cond.getNext();
742           elseBody = null;
743           reportCodeChange();
744         }
745     
746         // if (x()) { }
747         if (!mayHaveSideEffects(thenBody) && elseBody == null) {
748           if (mayHaveSideEffects(cond)) {
749             // x() has side effects, just leave the condition on its own.
750             n.removeChild(cond);
751             Node replacement = NodeUtil.newExpr(cond);
752             parent.replaceChild(n, replacement);
753             reportCodeChange();
754             return replacement;
755           } else {
756             // x() has no side effects, the whole tree is useless now.
757             NodeUtil.removeChild(parent, n);
758             reportCodeChange();
759             return null;
760           }
761         }
762     
763         // Try transforms that apply to both IF and HOOK.
764         TernaryValue condValue = NodeUtil.getImpureBooleanValue(cond);
765         if (condValue == TernaryValue.UNKNOWN) {
766           return n;  // We can't remove branches otherwise!
767         }
768     
769         if (mayHaveSideEffects(cond)) {
770           // Transform "if (a = 2) {x =2}" into "if (true) {a=2;x=2}"
771           boolean newConditionValue = condValue == TernaryValue.TRUE;
772           // Add an elseBody if it is needed.
773           if (!newConditionValue && elseBody == null) {
774             elseBody = IR.block().srcref(n);
775             n.addChildToBack(elseBody);
776           }
777           Node newCond = NodeUtil.booleanNode(newConditionValue);
778           n.replaceChild(cond, newCond);
779           Node branchToKeep = newConditionValue ? thenBody : elseBody;
780           branchToKeep.addChildToFront(IR.exprResult(cond).srcref(cond));
781           reportCodeChange();
782           cond = newCond;
783         }
784     
785         boolean condTrue = condValue.toBoolean(true);
786         if (n.getChildCount() == 2) {
787           Preconditions.checkState(type == Token.IF);
788     
789           if (condTrue) {
790             // Replace "if (true) { X }" with "X".
791             Node thenStmt = n.getSecondChild();
792             n.removeChild(thenStmt);
793             parent.replaceChild(n, thenStmt);
794             reportCodeChange();
795             return thenStmt;
796           } else {
797             // Remove "if (false) { X }" completely.
798             NodeUtil.redeclareVarsInsideBranch(n);
799             NodeUtil.removeChild(parent, n);
800             reportCodeChange();
801             return null;
802           }
803         } else {
804           // Replace "if (true) { X } else { Y }" with X, or
805           // replace "if (false) { X } else { Y }" with Y.
806           Node trueBranch = n.getSecondChild();
807           Node falseBranch = trueBranch.getNext();
808           Node branchToKeep = condTrue ? trueBranch : falseBranch;
809           Node branchToRemove = condTrue ? falseBranch : trueBranch;
810           NodeUtil.redeclareVarsInsideBranch(branchToRemove);
811           n.removeChild(branchToKeep);
812           parent.replaceChild(n, branchToKeep);
813           reportCodeChange();
814           return branchToKeep;
815         }
816       }

818       /**
819        * Try folding HOOK (?:) if the condition results of the condition is known.
820        * @return the replacement node, if changed, or the original if not
821        */
822       private Node tryFoldHook(Node n) {
823         Preconditions.checkState(n.isHook(), n);
824         Node parent = n.getParent();
825         Preconditions.checkNotNull(parent);
826         Node cond = n.getFirstChild();
827         Node thenBody = cond.getNext();
828         Node elseBody = thenBody.getNext();
829     
830         TernaryValue condValue = NodeUtil.getImpureBooleanValue(cond);
831         if (condValue == TernaryValue.UNKNOWN) {
832           // If the result nodes are equivalent, then one of the nodes can be
833           // removed and it doesn't matter which.
834           if (!areNodesEqualForInlining(thenBody, elseBody)) {
835             return n;  // We can't remove branches otherwise!
836           }
837         }
838     
839         // Transform "(a = 2) ? x =2 : y" into "a=2,x=2"
840         Node branchToKeep = condValue.toBoolean(true) ? thenBody : elseBody;
841         Node replacement;
842         boolean condHasSideEffects = mayHaveSideEffects(cond);
843         // Must detach after checking for side effects, to ensure that the parents
844         // of nodes are set correctly.
845         n.detachChildren();
846     
847         if (condHasSideEffects) {
848           replacement = IR.comma(cond, branchToKeep).srcref(n);
849         } else {
850           replacement = branchToKeep;
851         }
852     
853         parent.replaceChild(n, replacement);
854         reportCodeChange();
855         return replacement;
856       }

858       /**
859        * Removes WHILEs that always evaluate to false.
860        */
861       Node tryFoldWhile(Node n) {
862         Preconditions.checkArgument(n.isWhile());
863         Node cond = NodeUtil.getConditionExpression(n);
864         if (NodeUtil.getPureBooleanValue(cond) != TernaryValue.FALSE) {
865           return n;
866         }
867         NodeUtil.redeclareVarsInsideBranch(n);
868         NodeUtil.removeChild(n.getParent(), n);
869         reportCodeChange();
870     
871         return null;
872       }

874       /**
875        * Removes FORs that always evaluate to false.
876        */
877       Node tryFoldFor(Node n) {
878         Preconditions.checkArgument(n.isFor());
879         // If this is a FOR-IN loop skip it.
880         if (NodeUtil.isForIn(n)) {
881           return n;
882         }
883     
884         Node init = n.getFirstChild();
885         Node cond = init.getNext();
886         Node increment = cond.getNext();
887     
888         if (!init.isEmpty() && !init.isVar()) {
889           init = trySimplifyUnusedResult(init, false);
890         }
891     
892         if (!increment.isEmpty()) {
893           increment = trySimplifyUnusedResult(increment, false);
894         }
895     
896         // There is an initializer skip it
897         if (!n.getFirstChild().isEmpty()) {
898           return n;
899         }
900     
901         if (NodeUtil.getImpureBooleanValue(cond) != TernaryValue.FALSE) {
902           return n;
903         }
904     
905         Node parent = n.getParent();
906         NodeUtil.redeclareVarsInsideBranch(n);
907         if (!mayHaveSideEffects(cond)) {
908           NodeUtil.removeChild(parent, n);
909         } else {
910           Node statement = IR.exprResult(cond.detachFromParent())
911               .useSourceInfoIfMissingFrom(cond);
912           if (parent.isLabel()) {
913             Node block = IR.block();
914             block.useSourceInfoIfMissingFrom(statement);
915             block.addChildToFront(statement);
916             statement = block;
917           }
918           parent.replaceChild(n, statement);
919         }
920         reportCodeChange();
921         return null;
922       }

924       /**
925        * Removes DOs that always evaluate to false. This leaves the
926        * statements that were in the loop in a BLOCK node.
927        * The block will be removed in a later pass, if possible.
928        */
929       Node tryFoldDoAway(Node n) {
930         Preconditions.checkArgument(n.isDo());
931     
932         Node cond = NodeUtil.getConditionExpression(n);
933         if (NodeUtil.getImpureBooleanValue(cond) != TernaryValue.FALSE) {
934           return n;
935         }
936     
937         // TODO(johnlenz): The do-while can be turned into a label with
938         // named breaks and the label optimized away (maybe).
939         if (hasBreakOrContinue(n)) {
940           return n;
941         }
942     
943         Preconditions.checkState(
944             NodeUtil.isControlStructureCodeBlock(n, n.getFirstChild()));
945         Node block = n.removeFirstChild();
946     
947         Node parent =  n.getParent();
948         parent.replaceChild(n, block);
949         if (mayHaveSideEffects(cond)) {
950           Node condStatement = IR.exprResult(cond.detachFromParent())
951               .srcref(cond);
952           parent.addChildAfter(condStatement, block);
953         }
954         reportCodeChange();
955     
956         return block;
957       }

959       /**
960        * Removes DOs that have empty bodies into FORs, which are
961        * much easier for the CFA to analyze.
962        */
963       Node tryFoldEmptyDo(Node n) {
964         Preconditions.checkArgument(n.isDo());
965     
966         Node body = NodeUtil.getLoopCodeBlock(n);
967         if (body.isBlock() && !body.hasChildren()) {
968           Node cond = NodeUtil.getConditionExpression(n);
969           Node whileNode =
970               IR.forNode(IR.empty().srcref(n),
971                          cond.detachFromParent(),
972                          IR.empty().srcref(n),
973                          body.detachFromParent())
974               .srcref(n);
975           n.getParent().replaceChild(n, whileNode);
976           reportCodeChange();
977           return whileNode;
978         }
979         return n;
980       }

982       /**
983        *
984        */
985       static boolean hasBreakOrContinue(Node n) {
986         // TODO(johnlenz): This is overkill as named breaks may refer to outer
987         // loops or labels, and any break my refer to an inner loop.
988         // More generally, this check may be more expensive than we like.
989         return NodeUtil.has(
990             n,
991             Predicates.or(
992                 new NodeUtil.MatchNodeType(Token.BREAK),
993                 new NodeUtil.MatchNodeType(Token.CONTINUE)),
994             NodeUtil.MATCH_NOT_FUNCTION);
995       }

998        * Remove always true loop conditions.
999        */
1000       private void tryFoldForCondition(Node forCondition) {
1001         if (NodeUtil.getPureBooleanValue(forCondition) == TernaryValue.TRUE) {
1002           forCondition.getParent().replaceChild(forCondition,
1003               IR.empty());
1004           reportCodeChange();
1005         }
