17     package com.google.javascript.jscomp.newtypes;
18     
19     import com.google.common.annotations.GwtIncompatible;
20     
21     import java.lang.reflect.Method;
22     import java.util.Iterator;
23     import java.util.Set;
24     
25     @GwtIncompatible("java.lang.reflect")
26     /** A persistent set that simply wraps Clojure's implementation */
27     final class ClojurePersistentHashSet<K> extends PersistentSet<K> {
28       private static Method cons;
29       private static Method disjoin;
30       private final Set set;
31     
32       private ClojurePersistentHashSet(Set s) {
33         this.set = s;
34       }
35     
36       public static <K> PersistentSet<K> create(Class<? extends Set> cls)  {
37         try {
38           cons = cls.getDeclaredMethod("cons", Object.class);
39           disjoin = cls.getDeclaredMethod("disjoin", Object.class);
40           Set m = (Set) cls.getDeclaredField("EMPTY").get(null);
41           return new ClojurePersistentHashSet<>(m);
42         } catch (ReflectiveOperationException e) {
43           throw new RuntimeException(e);
44         }
45       }
46     
47       public PersistentSet<K> with(K key) {
48         try {
49           Set s = (Set) cons.invoke(this.set, key);
50           return new ClojurePersistentHashSet<>(s);
51         } catch (ReflectiveOperationException e) {
52           throw new RuntimeException(e);
53         }
54       }
55     
56       public PersistentSet<K> without(K key) {
57         try {
58           Set s = (Set) disjoin.invoke(this.set, key);
59           return new ClojurePersistentHashSet<>(s);
60         } catch (ReflectiveOperationException e) {
61           throw new RuntimeException(e);
62         }
63       }
64     
65       @Override
66       public boolean contains(Object key) {
67         return this.set.contains(key);
68       }
69     
70       @Override
71       public int size() {
72         return this.set.size();
73       }
74     
75       @Override
76       public boolean isEmpty() {
77         return this.set.isEmpty();
78       }
79     
80       @Override
81       @SuppressWarnings("unchecked")
82       public Iterator<K> iterator() {
83         return this.set.iterator();
84       }
85     
86       @Override
87       public boolean equals(Object o) {
88         if (o instanceof ClojurePersistentHashSet) {
89           ClojurePersistentHashSet ps = (ClojurePersistentHashSet) o;
90           return this.set.equals(ps.set);
91         }
92         return false;
93       }
94     
95       @Override
96       public int hashCode() {
97         return this.set.hashCode();
98       }
99     }
