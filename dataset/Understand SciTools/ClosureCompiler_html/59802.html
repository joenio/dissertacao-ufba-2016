17     package com.google.javascript.jscomp;
18     
19     import com.google.common.base.Function;
20     import com.google.common.base.Preconditions;
21     import com.google.javascript.jscomp.graph.LatticeElement;
22     
23     import java.util.Iterator;
24     import java.util.List;
25     
26     /**
27      * Defines a way join a list of LatticeElements.
28      */
29     interface JoinOp<L extends LatticeElement> extends Function<List<L>, L> {
30     
31       /**
32        * An implementation of {@code JoinOp} that makes it easy to join to
33        * lattice elements at a time.
34        */
35       abstract static class BinaryJoinOp<L extends LatticeElement>
36           implements JoinOp<L> {
37         @Override
38         public L apply(List<L> values) {
39           Preconditions.checkArgument(!values.isEmpty());
40           int size = values.size();
41           if (size == 1) {
42             return values.get(0);
43           } else if (size == 2) {
44             return apply(values.get(0), values.get(1));
45           } else if (size <= 16) {
46             // Do a Tree merge which request the least number of merges
47             int mid = computeMidPoint(size);
48             return apply(
49                 apply(values.subList(0, mid)),
50                 apply(values.subList(mid, size)));
51           } else {
52             // If the merge set is getting out of hand, avoid tree merges as it is
53             // more memory intense. Instead, stick with linear merges that requires
54             // more merges but less memory. This avoids GC trashing.
55             Iterator<L> iter = values.iterator();
56             L result = iter.next(); // Never null, we have 16+ values.
57             while (iter.hasNext()) {
58               result = apply(result, iter.next());
59             }
60             return result;
61           }
62         }
63     
64         /**
65          * Creates a new lattice that will be the join of two input lattices.
66          *
67          * @return The join of {@code latticeA} and {@code latticeB}.
68          */
69         abstract L apply(L latticeA, L latticeB);
70     
71         /**
72          * Finds the midpoint of a list. The function will favor two lists of
73          * even length instead of two lists of the same odd length. The list
74          * must be at least length two.
75          *
76          * @param size Size of the list.
77          */
78         static int computeMidPoint(int size) {
79           int midpoint = size >>> 1;
80           if (size > 4) {
81             /* Any list longer than 4 should prefer an even split point
82              * over the true midpoint, so that [0,6] splits at 2, not 3. */
83             midpoint &= -2; // (0xfffffffe) clears low bit so midpoint is even
84           }
85           return midpoint;
86         }
87       }
88     }
