17     package com.google.javascript.refactoring;
18     
19     import java.util.Objects;
20     
21     /**
22      * Class that represents a set of changes to make to the code.
23      *
24      * @author mknichel@google.com (Mark Knichel)
25      */
26     public final class CodeReplacement {
27     
28       private final int startPosition;
29       private final int length;
30       private final String newContent;
31     
32       // Used for sorting a list of CodeReplacements, in case of a "tie": two replacements with the
33       // same startPosition.
34       private final String sortKey;
35     
36       // TODO(tbreisacher): Make this package-private by refactoring tests so they
37       // don't need to call it directly.
38       public CodeReplacement(int startPosition, int length, String newContent) {
39         this(startPosition, length, newContent, "");
40       }
41     
42       CodeReplacement(int startPosition, int length, String newContent, String sortKey) {
43         this.startPosition = startPosition;
44         this.length = length;
45         this.newContent = newContent;
46         this.sortKey = sortKey;
47       }
48     
49       /**
50        * Returns the start position within the file that the modification
51        * should be applied starting at.
52        */
53       public int getStartPosition() {
54         return startPosition;
55       }
56     
57       /**
58        * Returns how many bytes the new content should replace in the
59        * original content.
60        */
61       public int getLength() {
62         return length;
63       }
64     
65       /**
66        * Returns the new content that should be inserted into the file.
67        */
68       public String getNewContent() {
69         return newContent;
70       }
71     
72       /**
73        * Returns an additional String key that can be used to sort replacements using lexical ordering.
74        */
75       public String getSortKey() {
76         return sortKey;
77       }
78     
79       @Override public String toString() {
80         return String.format(
81             "Start position: %d\nLength: %d\nNew Content: \"%s\"", startPosition, length, newContent);
82       }
83     
84       @Override public boolean equals(Object o) {
85         if (o == this) {
86           return true;
87         }
88         if (!(o instanceof CodeReplacement)) {
89           return false;
90         }
91         CodeReplacement other = (CodeReplacement) o;
92         return startPosition == other.startPosition
93             && length == other.length
94             && newContent.equals(other.newContent);
95       }
96     
97       @Override public int hashCode() {
98         return Objects.hash(startPosition, length, newContent);
99       }
100     }
