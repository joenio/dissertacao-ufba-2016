17     package com.google.javascript.jscomp.parsing.parser.trees;
18     
19     import static com.google.common.base.Preconditions.checkNotNull;
20     
21     import com.google.javascript.jscomp.parsing.parser.IdentifierToken;
22     import com.google.javascript.jscomp.parsing.parser.TokenType;
23     import com.google.javascript.jscomp.parsing.parser.util.SourceRange;
24     
25     import javax.annotation.Nullable;
26     
27     public class FunctionDeclarationTree extends ParseTree {
28     
29       public static enum Kind {
30         DECLARATION,
31         EXPRESSION,
32         MEMBER,
33         ARROW
34       }
35     
36       @Nullable public final IdentifierToken name;
37       @Nullable public final GenericTypeListTree generics;
38       public final FormalParameterListTree formalParameterList;
39       @Nullable public final ParseTree returnType;
40       public final ParseTree functionBody;
41       public final boolean isStatic;
42       public final boolean isGenerator;
43       public final boolean isOptional;
44       public final boolean isAsync;
45       @Nullable public final TokenType access;
46       public final Kind kind;
47     
48       public static Builder builder(Kind kind) {
49         return new Builder(kind);
50       }
51     
52       private FunctionDeclarationTree(Builder builder) {
53         super(ParseTreeType.FUNCTION_DECLARATION, builder.location);
54     
55         this.name = builder.name;
56         this.generics = builder.generics;
57         this.isStatic = builder.isStatic;
58         this.isGenerator = builder.isGenerator;
59         this.isOptional = builder.isOptional;
60         this.access = builder.access;
61         this.kind = checkNotNull(builder.kind);
62         this.formalParameterList = checkNotNull(builder.formalParameterList);
63         this.returnType = builder.returnType;
64         this.functionBody = checkNotNull(builder.functionBody);
65         this.isAsync = builder.isAsync;
66       }
67     
68       /**
69        * Builds a {@link FunctionDeclarationTree}.
70        */
71       public static class Builder {
72         private final Kind kind;
73     
74         @Nullable private IdentifierToken name = null;
75         @Nullable private GenericTypeListTree generics = null;
76         @Nullable private FormalParameterListTree formalParameterList = null;
77         @Nullable private ParseTree returnType = null;
78         @Nullable private ParseTree functionBody = null;
79         @Nullable private TokenType access = null;
80         private boolean isStatic = false;
81         private boolean isGenerator = false;
82         private boolean isOptional = false;
83         private boolean isAsync = false;
84         private SourceRange location;
85     
86         Builder(Kind kind) {
87           this.kind = kind;
88         }
89     
90         /**
91          * Optional function name.
92          *
93          * <p> Default is {@code null}.
94          */
95         public Builder setName(IdentifierToken name) {
96           this.name = name;
97           return this;
98         }
99     
100         /**
101          * Optional generics information.
102          *
103          * <p> Default is {@code null}.
104          */
105         public Builder setGenerics(GenericTypeListTree generics) {
106           this.generics = generics;
107           return this;
108         }
109     
110         /**
111          * Required parameter list.
112          */
113         public Builder setFormalParameterList(FormalParameterListTree formalParameterList) {
114           this.formalParameterList = formalParameterList;
115           return this;
116         }
117     
118         /**
119          * Optional return type.
120          *
121          * <p> Default is {@code null}.
122          */
123         public Builder setReturnType(ParseTree returnType) {
124           this.returnType = returnType;
125           return this;
126         }
127     
128         /**
129          * Required function body.
130          */
131         public Builder setFunctionBody(ParseTree functionBody) {
132           this.functionBody = functionBody;
133           return this;
134         }
135     
136         /**
137          * Optional TypeScript accessibility modifier (PUBLIC, PROTECTED, PRIVATE).
138          *
139          * <p> Default is {@code null}.
140          * Only relevant for method member declaration.
141          */
142         public Builder setAccess(TokenType access) {
143           this.access = access;
144           return this;
145         }
146     
147         /**
148          * Is the method static?
149          *
150          * <p> Default is {@code false}.
151          * Only relevant for method member declarations.
152          */
153         public Builder setStatic(boolean isStatic) {
154           this.isStatic = isStatic;
155           return this;
156         }
157     
158         /**
159          * Is this a generator function?
160          *
161          * <p> Default is {@code false}.
162          */
163         public Builder setGenerator(boolean isGenerator) {
164           this.isGenerator = isGenerator;
165           return this;
166         }
167     
168         /**
169          * Is this the declaration of an optional function parameter? Default is {@code false}.
170          *
171          * <p> Only relevant for function declaration as a parameter to another function.
172          */
173         public Builder setOptional(boolean isOptional) {
174           this.isOptional = isOptional;
175           return this;
176         }
177     
178         /**
179          * Is this an asynchronous function?
180          *
181          * <p> Default is {@code false}.
182          */
183         public Builder setAsync(boolean isAsync) {
184           this.isAsync = isAsync;
185           return this;
186         }
187     
188         /**
189          * Return a new {@link FunctionDeclarationTree}.
190          *
191          * <p> The location is provided at this point because it cannot be correctly calculated
192          * until the whole function has been parsed.
193          */
194         public FunctionDeclarationTree build(SourceRange location) {
195           this.location = location;
196           return new FunctionDeclarationTree(this);
197         }
198       }
199     }
