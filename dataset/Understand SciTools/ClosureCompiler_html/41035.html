95       /**
96        * Parses the given contents containing a source map.
97        */
98       public void parse(JsonObject sourceMapRoot, SourceMapSupplier sectionSupplier)
99           throws SourceMapParseException {
100         try {
101           // Check basic assertions about the format.
102           int version = sourceMapRoot.get("version").getAsInt();
103           if (version != 3) {
104             throw new SourceMapParseException("Unknown version: " + version);
105           }
106     
107           if (sourceMapRoot.has("file")
108               && sourceMapRoot.get("file").getAsString().isEmpty()) {
109             throw new SourceMapParseException("File entry is empty");
110           }
111     
112           if (sourceMapRoot.has("sections")) {
113             // Looks like a index map, try to parse it that way.
114             parseMetaMap(sourceMapRoot, sectionSupplier);
115             return;
116           }
117     
118           lineCount = sourceMapRoot.has("lineCount")
119               ? sourceMapRoot.get("lineCount").getAsInt() : -1;
120           String lineMap = sourceMapRoot.get("mappings").getAsString();
121     
122           sources = getJavaStringArray(sourceMapRoot.get("sources").getAsJsonArray());
123           names = getJavaStringArray(sourceMapRoot.get("names").getAsJsonArray());
124     
125           if (lineCount >= 0) {
126             lines = new ArrayList<>(lineCount);
127           } else {
128             lines = new ArrayList<>();
129           }
130     
131           if (sourceMapRoot.has("sourceRoot")) {
132             sourceRoot = sourceMapRoot.get("sourceRoot").getAsString();
133           }
134     
135           for (Map.Entry<String, JsonElement> entry : sourceMapRoot.entrySet()) {
136             if (entry.getKey().startsWith("x_")) {
137               extensions.put(entry.getKey(), entry.getValue());
138             }
139           }
140     
141           new MappingBuilder(lineMap).build();
142         } catch (JsonParseException ex) {
143           throw new SourceMapParseException("JSON parse exception: " + ex);
144         }
145       }

147       /**
148        * @param sourceMapRoot
149        * @throws SourceMapParseException
150        */
151       private void parseMetaMap(
152           JsonObject sourceMapRoot, SourceMapSupplier sectionSupplier)
153           throws SourceMapParseException {
154         if (sectionSupplier == null) {
155           sectionSupplier = new DefaultSourceMapSupplier();
156         }
157     
158         try {
159           // Check basic assertions about the format.
160           int version = sourceMapRoot.get("version").getAsInt();
161           if (version != 3) {
162             throw new SourceMapParseException("Unknown version: " + version);
163           }
164     
165           String file = sourceMapRoot.get("file").getAsString();
166           if (file.isEmpty()) {
167             throw new SourceMapParseException("File entry is missing or empty");
168           }
169     
170           if (sourceMapRoot.has("lineCount")
171               || sourceMapRoot.has("mappings")
172               || sourceMapRoot.has("sources")
173               || sourceMapRoot.has("names")) {
174             throw new SourceMapParseException("Invalid map format");
175           }
176     
177           SourceMapGeneratorV3 generator = new SourceMapGeneratorV3();
178           JsonArray sections = sourceMapRoot.get("sections").getAsJsonArray();
179           for (int i = 0, count = sections.size(); i < count; i++) {
180             JsonObject section = sections.get(i).getAsJsonObject();
181             if (section.has("map") && section.has("url")) {
182               throw new SourceMapParseException(
183                   "Invalid map format: section may not have both 'map' and 'url'");
184             }
185             JsonObject offset = section.get("offset").getAsJsonObject();
186             int line = offset.get("line").getAsInt();
187             int column =  offset.get("column").getAsInt();
188             String mapSectionContents;
189             if (section.has("url")) {
190               String url = section.get("url").getAsString();
191               mapSectionContents = sectionSupplier.getSourceMap(url);
192               if (mapSectionContents == null) {
193                 throw new SourceMapParseException("Unable to retrieve: " + url);
194               }
195             } else if (section.has("map")) {
196               mapSectionContents = section.get("map").toString();
197             } else {
198               throw new SourceMapParseException(
199                   "Invalid map format: section must have either 'map' or 'url'");
200             }
201             generator.mergeMapSection(line, column, mapSectionContents);
202           }
203     
204           StringBuilder sb = new StringBuilder();
205           try {
206             generator.appendTo(sb, file);
207           } catch (IOException e) {
208             // Can't happen.
209             throw new RuntimeException(e);
210           }
211     
212           parse(sb.toString());
213         } catch (IOException ex) {
214           throw new SourceMapParseException("IO exception: " + ex);
215         } catch (JsonParseException ex) {
216           throw new SourceMapParseException("JSON parse exception: " + ex);
217         }
218       }

220       @Override
221       public OriginalMapping getMappingForLine(int lineNumber, int column) {
222         // Normalize the line and column numbers to 0.
223         lineNumber--;
224         column--;
225     
226         if (lineNumber < 0 || lineNumber >= lines.size()) {
227           return null;
228         }
229     
230         Preconditions.checkState(lineNumber >= 0);
231         Preconditions.checkState(column >= 0);
232     
233     
234         // If the line is empty return the previous mapping.
235         if (lines.get(lineNumber) == null) {
236           return getPreviousMapping(lineNumber);
237         }
238     
239         ArrayList<Entry> entries = lines.get(lineNumber);
240         // No empty lists.
241         Preconditions.checkState(!entries.isEmpty());
242         if (entries.get(0).getGeneratedColumn() > column) {
243           return getPreviousMapping(lineNumber);
244         }
245     
246         int index = search(entries, column, 0, entries.size() - 1);
247         Preconditions.checkState(index >= 0, "unexpected:%s", index);
248         return getOriginalMappingForEntry(entries.get(index));
249       }

256       @Override
257       public Collection<OriginalMapping> getReverseMapping(String originalFile,
258           int line, int column) {
259         // TODO(user): This implementation currently does not make use of the column
260         // parameter.
261     
262         // Synchronization needs to be handled by callers.
263         if (reverseSourceMapping == null) {
264           createReverseMapping();
265         }
266     
267         Map<Integer, Collection<OriginalMapping>> sourceLineToCollectionMap =
268             reverseSourceMapping.get(originalFile);
269     
270         if (sourceLineToCollectionMap == null) {
271           return Collections.emptyList();
272         } else {
273           Collection<OriginalMapping> mappings =
274               sourceLineToCollectionMap.get(line);
275     
276           if (mappings == null) {
277             return Collections.emptyList();
278           } else {
279             return mappings;
280           }
281         }

299       private String[] getJavaStringArray(JsonArray array) throws JsonParseException {
300         int len = array.size();
301         String[] result = new String[len];
302         for (int i = 0; i < len; i++) {
303           result[i] = array.get(i).getAsString();
304         }
305         return result;

308       private class MappingBuilder {
309         private static final int MAX_ENTRY_VALUES = 5;
310         private final StringCharIterator content;
311         private int line = 0;
312         private int previousCol = 0;
313         private int previousSrcId = 0;
314         private int previousSrcLine = 0;
315         private int previousSrcColumn = 0;
316         private int previousNameId = 0;
317     
318         MappingBuilder(String lineMap) {
319           this.content = new StringCharIterator(lineMap);
320         }
321     
322         void build() {
323           int [] temp = new int[MAX_ENTRY_VALUES];
324           ArrayList<Entry> entries = new ArrayList<>();
325           while (content.hasNext()) {
326             // ';' denotes a new line.
327             if (tryConsumeToken(';')) {
328               // The line is complete, store the result
329               completeLine(entries);
330               if (!entries.isEmpty()) {
331                 // A new array list for the next line.
332                 entries = new ArrayList<>();
333               }
334             } else {
335               // grab the next entry for the current line.
336               int entryValues = 0;
337               while (!entryComplete()) {
338                 temp[entryValues] = nextValue();
339                 entryValues++;
340               }
341               Entry entry = decodeEntry(temp, entryValues);
342     
343               validateEntry(entry);
344               entries.add(entry);
345     
346               // Consume the separating token, if there is one.
347               tryConsumeToken(',');
348             }
349           }
350     
351           // Some source map generator (e.g.UglifyJS) generates lines without
352           // a trailing line separator. So add the rest of the content.
353           if (!entries.isEmpty()) {
354             completeLine(entries);
355           }
356         }
357     
358         private void completeLine(ArrayList<Entry> entries) {
359           // The line is complete, store the result for the line,
360           // null if the line is empty.
361           if (!entries.isEmpty()) {
362             lines.add(entries);
363           } else {
364             lines.add(null);
365           }
366           line++;
367           previousCol = 0;
368         }
369     
370         /**
371          * Sanity check the entry.
372          */
373         private void validateEntry(Entry entry) {
374           Preconditions.checkState((lineCount < 0) || (line < lineCount),
375               "line=%s, lineCount=%s", line, lineCount);
376           Preconditions.checkState(entry.getSourceFileId() == UNMAPPED
377               || entry.getSourceFileId() < sources.length);
378           Preconditions.checkState(entry.getNameId() == UNMAPPED
379               || entry.getNameId() < names.length);
380         }
381     
382         /**
383          * Decodes the next entry, using the previous encountered values to
384          * decode the relative values.
385          *
386          * @param vals An array of integers that represent values in the entry.
387          * @param entryValues The number of entries in the array.
388          * @return The entry object.
389          */
390         private Entry decodeEntry(int[] vals, int entryValues) {
391           Entry entry;
392           switch (entryValues) {
393             // The first values, if present are in the following order:
394             //   0: the starting column in the current line of the generated file
395             //   1: the id of the original source file
396             //   2: the starting line in the original source
397             //   3: the starting column in the original source
398             //   4: the id of the original symbol name
399             // The values are relative to the last encountered value for that field.
400             // Note: the previously column value for the generated file is reset
401             // to '0' when a new line is encountered.  This is done in the 'build'
402             // method.
403     
404             case 1:
405               // An unmapped section of the generated file.
406               entry = new UnmappedEntry(
407                   vals[0] + previousCol);
408               // Set the values see for the next entry.
409               previousCol = entry.getGeneratedColumn();
410               return entry;
411     
412             case 4:
413               // A mapped section of the generated file.
414               entry = new UnnamedEntry(
415                   vals[0] + previousCol,
416                   vals[1] + previousSrcId,
417                   vals[2] + previousSrcLine,
418                   vals[3] + previousSrcColumn);
419               // Set the values see for the next entry.
420               previousCol = entry.getGeneratedColumn();
421               previousSrcId = entry.getSourceFileId();
422               previousSrcLine = entry.getSourceLine();
423               previousSrcColumn = entry.getSourceColumn();
424               return entry;
425     
426             case 5:
427               // A mapped section of the generated file, that has an associated
428               // name.
429               entry = new NamedEntry(
430                   vals[0] + previousCol,
431                   vals[1] + previousSrcId,
432                   vals[2] + previousSrcLine,
433                   vals[3] + previousSrcColumn,
434                   vals[4] + previousNameId);
435               // Set the values see for the next entry.
436               previousCol = entry.getGeneratedColumn();
437               previousSrcId = entry.getSourceFileId();
438               previousSrcLine = entry.getSourceLine();
439               previousSrcColumn = entry.getSourceColumn();
440               previousNameId = entry.getNameId();
441               return entry;
442     
443             default:
444               throw new IllegalStateException(
445                   "Unexpected number of values for entry:" + entryValues);
446           }
447         }
448     
449         private boolean tryConsumeToken(char token) {
450           if (content.hasNext() && content.peek() == token) {
451             // consume the comma
452             content.next();
453             return true;
454           }
455           return false;
456         }
457     
458         private boolean entryComplete() {
459           if (!content.hasNext()) {
460             return true;
461           }
462     
463           char c = content.peek();
464           return (c == ';' || c == ',');
465         }
466     
467         private int nextValue() {
468           return Base64VLQ.decode(content);
469         }
470       }

472       /**
473        * Perform a binary search on the array to find a section that covers
474        * the target column.
475        */
476       private int search(ArrayList<Entry> entries, int target, int start, int end) {
477         while (true) {
478           int mid = ((end - start) / 2) + start;
479           int compare = compareEntry(entries, mid, target);
480           if (compare == 0) {
481             return mid;
482           } else if (compare < 0) {
483             // it is in the upper half
484             start = mid + 1;
485             if (start > end) {
486               return end;
487             }
488           } else {
489             // it is in the lower half
490             end = mid - 1;
491             if (end < start) {
492               return end;
493             }
494           }
495         }
496       }

498       /**
499        * Compare an array entry's column value to the target column value.
500        */
501       private int compareEntry(ArrayList<Entry> entries, int entry, int target) {
502         return entries.get(entry).getGeneratedColumn() - target;
503       }

505       /**
506        * Returns the mapping entry that proceeds the supplied line or null if no
507        * such entry exists.
508        */
509       private OriginalMapping getPreviousMapping(int lineNumber) {
510         do {
511           if (lineNumber == 0) {
512             return null;
513           }
514           lineNumber--;
515         } while (lines.get(lineNumber) == null);
516         ArrayList<Entry> entries = lines.get(lineNumber);
517         return getOriginalMappingForEntry(Iterables.getLast(entries));
518       }

520       /**
521        * Creates an "OriginalMapping" object for the given entry object.
522        */
523       private OriginalMapping getOriginalMappingForEntry(Entry entry) {
524         if (entry.getSourceFileId() == UNMAPPED) {
525           return null;
526         } else {
527           // Adjust the line/column here to be start at 1.
528           Builder x = OriginalMapping.newBuilder()
529             .setOriginalFile(sources[entry.getSourceFileId()])
530             .setLineNumber(entry.getSourceLine() + 1)
531             .setColumnPosition(entry.getSourceColumn() + 1);
532           if (entry.getNameId() != UNMAPPED) {
533             x.setIdentifier(names[entry.getNameId()]);
534           }
535           return x.build();
536         }
537       }

539       /**
540        * Reverse the source map; the created mapping will allow us to quickly go
541        * from a source file and line number to a collection of target
542        * OriginalMappings.
543        */
544       private void createReverseMapping() {
545         reverseSourceMapping = new HashMap<>();
546     
547         for (int targetLine = 0; targetLine < lines.size(); targetLine++) {
548           ArrayList<Entry> entries = lines.get(targetLine);
549     
550           if (entries != null) {
551             for (Entry entry : entries) {
552               if (entry.getSourceFileId() != UNMAPPED
553                   && entry.getSourceLine() != UNMAPPED) {
554                 String originalFile = sources[entry.getSourceFileId()];
555     
556                 if (!reverseSourceMapping.containsKey(originalFile)) {
557                   reverseSourceMapping.put(originalFile,
558                       new HashMap<Integer, Collection<OriginalMapping>>());
559                 }
560     
561                 Map<Integer, Collection<OriginalMapping>> lineToCollectionMap =
562                     reverseSourceMapping.get(originalFile);
563     
564                 int sourceLine = entry.getSourceLine();
565     
566                 if (!lineToCollectionMap.containsKey(sourceLine)) {
567                   lineToCollectionMap.put(sourceLine,
568                       new ArrayList<OriginalMapping>(1));
569                 }
570     
571                 Collection<OriginalMapping> mappings =
572                     lineToCollectionMap.get(sourceLine);
573     
574                 Builder builder = OriginalMapping.newBuilder().setLineNumber(
575                     targetLine).setColumnPosition(entry.getGeneratedColumn());
576     
577                 mappings.add(builder.build());
578               }
579             }
580           }
581         }
582       }

584       /**
585        * A implementation of the Base64VLQ CharIterator used for decoding the
586        * mappings encoded in the JSON string.
587        */
588       private static class StringCharIterator implements CharIterator {
589         final String content;
590         final int length;
591         int current = 0;
592     
593         StringCharIterator(String content) {
594           this.content = content;
595           this.length = content.length();
596         }
597     
598         @Override
599         public char next() {
600           return content.charAt(current++);
601         }
602     
603         char peek() {
604           return content.charAt(current);
605         }
606     
607         @Override
608         public boolean hasNext() {
609           return current < length;
610         }
611       }

716       public static interface EntryVisitor {
717         void visit(String sourceName,
718                    String symbolName,
719                    FilePosition sourceStartPosition,
720                    FilePosition startPosition,
721                    FilePosition endPosition);
722       }
723     
724       public void visitMappings(EntryVisitor visitor) {
725         boolean pending = false;
726         String sourceName = null;
727         String symbolName = null;
728         FilePosition sourceStartPosition = null;
729         FilePosition startPosition = null;
730     
731         final int lineCount = lines.size();
732         for (int i = 0; i < lineCount; i++) {
733           ArrayList<Entry> line = lines.get(i);
734           if (line != null) {
735             final int entryCount = line.size();
736             for (int j = 0; j < entryCount; j++) {
737               Entry entry = line.get(j);
738               if (pending) {
739                 FilePosition endPosition = new FilePosition(
740                     i, entry.getGeneratedColumn());
741                 visitor.visit(
742                     sourceName,
743                     symbolName,
744                     sourceStartPosition,
745                     startPosition,
746                     endPosition);
747                 pending = false;
748               }
749     
750               if (entry.getSourceFileId() != UNMAPPED) {
751                 pending = true;
752                 sourceName = sources[entry.getSourceFileId()];
753                 symbolName = (entry.getNameId() != UNMAPPED)
754                     ? names[entry.getNameId()] : null;
755                 sourceStartPosition = new FilePosition(
756                     entry.getSourceLine(), entry.getSourceColumn());
757                 startPosition = new FilePosition(
758                     i, entry.getGeneratedColumn());
759               }
760             }
761           }
762         }
763       }
