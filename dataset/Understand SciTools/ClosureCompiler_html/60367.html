17     package com.google.javascript.jscomp.newtypes;
18     
19     import java.util.HashMap;
20     import java.util.Map;
21     import java.util.Set;
22     
23     /** A naive persistent map that does too many copies */
24     final class NaivePersistentMap<K, V> extends PersistentMap<K, V> {
25       private Map<K, V> map;
26     
27       private NaivePersistentMap(Map<K, V> m) {
28         this.map = m;
29       }
30     
31       public static <K, V> PersistentMap<K, V> create()  {
32         return new NaivePersistentMap<>(new HashMap<K, V>());
33       }
34     
35       public PersistentMap<K, V> with(K key, V value) {
36         Map<K, V> newMap = new HashMap<>(this);
37         newMap.put(key, value);
38         return new NaivePersistentMap<>(newMap);
39       }
40     
41       public PersistentMap<K, V> without(K key) {
42         Map<K, V> newMap = new HashMap<>(this);
43         newMap.remove(key);
44         return new NaivePersistentMap<>(newMap);
45       }
46     
47       @Override
48       public Set<Map.Entry<K, V>> entrySet() {
49         return this.map.entrySet();
50       }
51     }
