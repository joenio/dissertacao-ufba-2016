16     package com.google.javascript.jscomp;
17     
18     import static com.google.common.truth.Truth.assertThat;
19     
20     import com.google.common.collect.ImmutableList;
21     import com.google.javascript.rhino.Node;
22     
23     public final class PolymerPassFindExternsTest extends Es6CompilerTestCase {
24     
25       private static final String EXTERNS =
26           LINE_JOINER.join(
27               "/** @constructor */",
28               "var HTMLElement = function() {};",
29               "/** @constructor @extends {HTMLElement} */",
30               "var HTMLInputElement = function() {};",
31               "/** @constructor @extends {HTMLElement} */",
32               "var PolymerElement = function() {};",
33               "/** @type {!Object} */",
34               "PolymerElement.prototype.$;",
35               "PolymerElement.prototype.created = function() {};",
36               "PolymerElement.prototype.ready = function() {};",
37               "PolymerElement.prototype.attached = function() {};",
38               "PolymerElement.prototype.domReady = function() {};",
39               "PolymerElement.prototype.detached = function() {};",
40               "/**",
41               " * Call the callback after a timeout. Calling job again with the same name",
42               " * resets the timer but will not result in additional calls to callback.",
43               " *",
44               " * @param {string} name",
45               " * @param {Function} callback",
46               " * @param {number} timeoutMillis The minimum delay in milliseconds before",
47               " *     calling the callback.",
48               " */",
49               "PolymerElement.prototype.job = function(name, callback, timeoutMillis) {};",
50               "/**",
51               " * @param a {!Object}",
52               " * @return {!function()}",
53               " */",
54               "var Polymer = function(a) {};",
55               "var alert = function(msg) {};");
56     
57       private PolymerPassFindExterns findExternsCallback;
58     
59       public PolymerPassFindExternsTest() {
60         super(EXTERNS);
61       }
62     
63       @Override
64       protected CompilerPass getProcessor(final Compiler compiler) {
65         findExternsCallback = new PolymerPassFindExterns();
66         return new CompilerPass() {
67           @Override
68           public void process(Node externs, Node root) {
69             NodeTraversal.traverseEs6(compiler, externs, findExternsCallback);
70           }
71         };
72       }
73     
74       @Override
75       protected void setUp() throws Exception {
76         super.setUp();
77         allowExternsChanges(true);
78         enableTypeCheck();
79         runTypeCheckAfterProcessing = true;
80         parseTypeInfo = true;
81       }
82     
83       @Override
84       protected int getNumRepetitions() {
85         return 1;
86       }
87     
88       public void testFindsPolymerElementRoot() {
89         testSame("");
90         Node polymerElementNode = findExternsCallback.getPolymerElementExterns();
91     
92         assertNotNull(polymerElementNode);
93         assertTrue(polymerElementNode.isVar());
94         assertTrue(polymerElementNode.getFirstChild().matchesQualifiedName("PolymerElement"));
95       }
96     
97       public void testFindsPolymerElementProps() {
98         testSame("");
99         final ImmutableList<String> expectedProps = ImmutableList.of(
100             "$", "created", "ready", "attached", "domReady", "detached", "job");
101         ImmutableList<Node> polymerElementProps = findExternsCallback.getPolymerElementProps();
102     
103         assertNotNull(polymerElementProps);
104         assertThat(polymerElementProps).hasSize(expectedProps.size());
105         for (int i = 0; i < polymerElementProps.size(); ++i) {
106           assertThat(getPropertyName(polymerElementProps.get(i))).isEqualTo(expectedProps.get(i));
107         }
108       }
109     
110       private String getPropertyName(Node node) {
111         Node rightName = node.getFirstChild().getChildAtIndex(1);
112         return rightName.isFunction()
113             ? node.getFirstFirstChild().getChildAtIndex(1).getString() : rightName.getString();
114       }
115     }
