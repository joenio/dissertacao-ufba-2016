54     /**
55      * The {@code UnionType} implements a common JavaScript idiom in which the
56      * code is specifically designed to work with multiple input types.  Because
57      * JavaScript always knows the run-time type of an object value, this is safer
58      * than a C union.<p>
59      *
60      * For instance, values of the union type {@code (String,boolean)} can be of
61      * type {@code String} or of type {@code boolean}. The commutativity of the
62      * statement is captured by making {@code (String,boolean)} and
63      * {@code (boolean,String)} equal.<p>
64      *
65      * The implementation of this class prevents the creation of nested
66      * unions.<p>
67      */
68     public class UnionType extends JSType {
69       private static final long serialVersionUID = 1L;
70     
71       // alternates without merging structural interfaces and their subtypes
72       Collection<JSType> alternatesWithoutStucturalTyping;
73       // alternates under structural typing
74       Collection<JSType> alternates;
75       private int hashcode;

77       /**
78        * Creates a union type.
79        *
80        * @param alternatesWithoutStructuralTyping the alternates of the union without
81        * structural typing subtype
82        */
83       UnionType(JSTypeRegistry registry,
84           Collection<JSType> alternatesWithoutStructuralTyping) {
85         super(registry);
86         this.alternatesWithoutStucturalTyping = alternatesWithoutStructuralTyping;
87     
88         UnionTypeBuilder builder = new UnionTypeBuilder(registry);
89         for (JSType alternate : alternatesWithoutStructuralTyping) {
90           builder.addAlternate(alternate, true);
91         }
92         this.alternates = builder.getAlternates();
93         this.hashcode = this.alternatesWithoutStucturalTyping.hashCode();
94       }

96       /**
97        * Gets the alternate types of this union type.
98        * @return The alternate types of this union type. The returned set is
99        *     immutable.
100        */
101       public Collection<JSType> getAlternates() {
102         for (JSType t : alternatesWithoutStucturalTyping) {
103           if (t.isUnionType()) {
104             rebuildAlternates();
105             break;
106           }
107         }
108         return alternates;
109       }

127       /**
128        * Use UnionTypeBuilder to rebuild the list of alternates and hashcode
129        * of the current UnionType.
130        */
131       private void rebuildAlternates() {
132         UnionTypeBuilder builder = new UnionTypeBuilder(registry);
133         for (JSType alternate : alternatesWithoutStucturalTyping) {
134           builder.addAlternate(alternate);
135         }
136         alternatesWithoutStucturalTyping = builder.getAlternates();
137         builder = new UnionTypeBuilder(registry);
138         for (JSType alternate : alternatesWithoutStucturalTyping) {
139           builder.addAlternate(alternate, true);
140         }
141         alternates = builder.getAlternates();
142         hashcode = alternatesWithoutStucturalTyping.hashCode();
143       }

211       @Override
212       public JSType findPropertyType(String propertyName) {
213         JSType propertyType = null;
214     
215         for (JSType alternate : getAlternates()) {
216           // Filter out the null/undefined type.
217           if (alternate.isNullType() || alternate.isVoidType()) {
218             continue;
219           }
220     
221           JSType altPropertyType = alternate.findPropertyType(propertyName);
222           if (altPropertyType == null) {
223             continue;
224           }
225     
226           if (propertyType == null) {
227             propertyType = altPropertyType;
228           } else {
229             propertyType = propertyType.getLeastSupertype(altPropertyType);
230           }
231         }
232     
233         return propertyType;
234       }

352       @Override
353       public JSType getLeastSupertype(JSType that) {
354         if (!that.isUnknownType() && !that.isUnionType()) {
355           for (JSType alternate : alternatesWithoutStucturalTyping) {
356             if (!alternate.isUnknownType() && that.isSubtype(alternate)) {
357               return this;
358             }
359           }
360         }
361     
362         return getLeastSupertype(this, that);
363       }

365       JSType meet(JSType that) {
366         UnionTypeBuilder builder = new UnionTypeBuilder(registry);
367         for (JSType alternate : alternatesWithoutStucturalTyping) {
368           if (alternate.isSubtype(that)) {
369             builder.addAlternate(alternate);
370           }
371         }
372     
373         if (that.isUnionType()) {
374           for (JSType otherAlternate : that.toMaybeUnionType().alternatesWithoutStucturalTyping) {
375             if (otherAlternate.isSubtype(this)) {
376               builder.addAlternate(otherAlternate);
377             }
378           }
379         } else if (that.isSubtype(this)) {
380           builder.addAlternate(that);
381         }
382         JSType result = builder.build();
383         if (!result.isNoType()) {
384           return result;
385         } else if (this.isObject() && (that.isObject() && !that.isNoType())) {
386           return getNativeType(JSTypeNative.NO_OBJECT_TYPE);
387         } else {
388           return getNativeType(JSTypeNative.NO_TYPE);
389         }
390       }

392       /**
393        * Two union types are equal if, after flattening nested union types,
394        * they have the same number of alternates and all alternates are equal.
395        */
396       boolean checkUnionEquivalenceHelper(
397           UnionType that, EquivalenceMethod eqMethod, EqCache eqCache) {
398         Collection<JSType> thatAlternates = that.getAlternatesWithoutStructuralTyping();
399         if (eqMethod == EquivalenceMethod.IDENTITY
400             && getAlternatesWithoutStructuralTyping().size() != thatAlternates.size()) {
401           return false;
402         }
403         for (JSType alternate : thatAlternates) {
404           if (!hasAlternate(alternate, eqMethod, eqCache)) {
405             return false;
406           }
407         }
408         return true;
409       }

411       private boolean hasAlternate(JSType type, EquivalenceMethod eqMethod,
412           EqCache eqCache) {
413         for (JSType alternate : getAlternatesWithoutStructuralTyping()) {
414           if (alternate.checkEquivalenceHelper(type, eqMethod, eqCache)) {
415             return true;
416           }
417         }
418         return false;
419       }

468       /**
469        * Returns a more restricted union type than {@code this} one, in which all
470        * subtypes of {@code type} have been removed.<p>
471        *
472        * Examples:
473        * <ul>
474        * <li>{@code (number,string)} restricted by {@code number} is
475        *     {@code string}</li>
476        * <li>{@code (null, EvalError, URIError)} restricted by
477        *     {@code Error} is {@code null}</li>
478        * </ul>
479        *
480        * @param type the supertype of the types to remove from this union type
481        */
482       public JSType getRestrictedUnion(JSType type) {
483         UnionTypeBuilder restricted = new UnionTypeBuilder(registry);
484         for (JSType t : alternatesWithoutStucturalTyping) {
485           // Keep all unknown/unresolved types.
486           if (t.isUnknownType() || t.isNoResolvedType() || !t.isSubtype(type)) {
487             restricted.addAlternate(t);
488           }
489         }
490         return restricted.build();
491       }

493       @Override String toStringHelper(boolean forAnnotations) {
494         StringBuilder result = new StringBuilder();
495         boolean firstAlternate = true;
496     
497         result.append("(");
498         SortedSet<JSType> sorted = new TreeSet<>(ALPHA);
499         sorted.addAll(alternatesWithoutStucturalTyping);
500         for (JSType t : sorted) {
501           if (!firstAlternate) {
502             result.append("|");
503           }
504           result.append(t.toStringHelper(forAnnotations));
505           firstAlternate = false;
506         }
507         result.append(")");
508         return result.toString();
509       }

516       @Override
517       protected boolean isSubtype(JSType that,
518           ImplCache implicitImplCache, SubtypingMode subtypingMode) {
519         // unknown
520         if (that.isUnknownType() || this.isUnknownType()) {
521           return true;
522         }
523         // all type
524         if (that.isAllType()) {
525           return true;
526         }
527         for (JSType element : alternatesWithoutStucturalTyping) {
528           if (subtypingMode == SubtypingMode.IGNORE_NULL_UNDEFINED
529               && (element.isNullType() || element.isVoidType())) {
530             continue;
531           }
532           if (!element.isSubtype(that, implicitImplCache, subtypingMode)) {
533             return false;
534           }
535         }
536         return true;
537       }

625       @Override
626       JSType resolveInternal(ErrorReporter t, StaticTypedScope<JSType> scope) {
627         setResolvedTypeInternal(this); // for circularly defined types.
628     
629         // Just resolve the alternates, but do not update as that breaks some error
630         // reporting cases.
631         for (JSType alternate : alternatesWithoutStucturalTyping) {
632           alternate.resolve(t, scope);
633         }
634         // Ensure the union is in a normalized state.
635         rebuildAlternates();
636         return this;
637       }

656       @Override
657       public JSType collapseUnion() {
658         JSType currentValue = null;
659         ObjectType currentCommonSuper = null;
660         for (JSType a : alternatesWithoutStucturalTyping) {
661           if (a.isUnknownType()) {
662             return getNativeType(JSTypeNative.UNKNOWN_TYPE);
663           }
664     
665           ObjectType obj = a.toObjectType();
666           if (obj == null) {
667             if (currentValue == null && currentCommonSuper == null) {
668               // If obj is not an object, then it must be a value.
669               currentValue = a;
670             } else {
671               // Multiple values and objects will always collapse to the ALL_TYPE.
672               return getNativeType(JSTypeNative.ALL_TYPE);
673             }
674           } else if (currentValue != null) {
675             // Values and objects will always collapse to the ALL_TYPE.
676             return getNativeType(JSTypeNative.ALL_TYPE);
677           } else if (currentCommonSuper == null) {
678             currentCommonSuper = obj;
679           } else {
680             currentCommonSuper =
681                 registry.findCommonSuperObject(currentCommonSuper, obj);
682           }
683         }
684         return currentCommonSuper;
685       }
