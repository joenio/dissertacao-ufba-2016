28     /**
29      * Rewrites ES6 destructuring patterns and default parameters to valid ES3 code.
30      */
31     public final class Es6RewriteDestructuring implements NodeTraversal.Callback, HotSwapCompilerPass {
32       private final AbstractCompiler compiler;
33     
34       private static final String DESTRUCTURING_TEMP_VAR = "$jscomp$destructuring$var";
35     
36       private int destructuringVarCounter = 0;
37     
38       public Es6RewriteDestructuring(AbstractCompiler compiler) {
39         this.compiler = compiler;
40       }

42       @Override
43       public void process(Node externs, Node root) {
44         NodeTraversal.traverseEs6(compiler, externs, this);
45         NodeTraversal.traverseEs6(compiler, root, this);
46       }

48       @Override
49       public void hotSwapScript(Node scriptRoot, Node originalRoot) {
50         NodeTraversal.traverseEs6(compiler, scriptRoot, this);
51       }

53       @Override
54       public boolean shouldTraverse(NodeTraversal nodeTraversal, Node n, Node parent) {
55         switch (n.getType()) {
56           case PARAM_LIST:
57             visitParamList(n, parent);
58             break;
59           case FOR_OF:
60             visitForOf(n);
61             break;
62         }
63         return true;
64       }
65     
66       @Override
67       public void visit(NodeTraversal t, Node n, Node parent) {
68         if (parent != null && parent.isDestructuringLhs()) {
69           parent = parent.getParent();
70         }
71         switch (n.getType()) {
72           case ARRAY_PATTERN:
73             visitArrayPattern(t, n, parent);
74             break;
75           case OBJECT_PATTERN:
76             visitObjectPattern(t, n, parent);
77             break;
78         }
79       }

81       /**
82        * Processes trailing default and rest parameters.
83        */
84       private void visitParamList(Node paramList, Node function) {
85         Node insertSpot = null;
86         Node body = function.getLastChild();
87         for (int i = 0; i < paramList.getChildCount(); i++) {
88           Node param = paramList.getChildAtIndex(i);
89           if (param.isDefaultValue()) {
90             JSDocInfo jsDoc = param.getJSDocInfo();
91             Node nameOrPattern = param.removeFirstChild();
92             Node defaultValue = param.removeFirstChild();
93             Node newParam;
94     
95             // Treat name=undefined (and equivalent) as if it was just name.  There
96             // is no need to generate a (name===void 0?void 0:name) statement for
97             // such arguments.
98             boolean isNoop = false;
99             if (!nameOrPattern.isName()) {
100               // Do not try to optimize unless nameOrPattern is a simple name.
101             } else if (defaultValue.isName()) {
102               isNoop = "undefined".equals(defaultValue.getString());
103             } else if (defaultValue.isVoid()) {
104               // Any kind of 'void literal' is fine, but 'void fun()' or anything
105               // else with side effects isn't.  We're not trying to be particularly
106               // smart here and treat 'void {}' for example as if it could cause
107               // side effects.  Any sane person will type 'name=undefined' or
108               // 'name=void 0' so this should not be an issue.
109               isNoop = NodeUtil.isImmutableValue(defaultValue.getFirstChild());
110             }
111     
112             if (isNoop) {
113               newParam = nameOrPattern.cloneTree();
114             } else {
115               newParam =
116                   nameOrPattern.isName()
117                       ? nameOrPattern
118                       : IR.name(DESTRUCTURING_TEMP_VAR + (destructuringVarCounter++));
119               Node lhs = nameOrPattern.cloneTree();
120               Node rhs = defaultValueHook(newParam.cloneTree(), defaultValue);
121               Node newStatement =
122                   nameOrPattern.isName() ? IR.exprResult(IR.assign(lhs, rhs)) : IR.var(lhs, rhs);
123               newStatement.useSourceInfoIfMissingFromForTree(param);
124               body.addChildAfter(newStatement, insertSpot);
125               insertSpot = newStatement;
126             }
127     
128             paramList.replaceChild(param, newParam);
129             newParam.setOptionalArg(true);
130             newParam.setJSDocInfo(jsDoc);
131     
132             compiler.reportCodeChange();
133           } else if (param.isDestructuringPattern()) {
134             insertSpot =
135                 replacePatternParamWithTempVar(
136                     function, insertSpot, param, getTempParameterName(function, i));
137             compiler.reportCodeChange();
138           } else if (param.isRest() && param.getFirstChild().isDestructuringPattern()) {
139             insertSpot =
140                 replacePatternParamWithTempVar(
141                     function, insertSpot, param.getFirstChild(), getTempParameterName(function, i));
142             compiler.reportCodeChange();
143           }
144         }
145       }

147       /**
148        * Replace a destructuring pattern parameter with a a temporary parameter name and add a new
149        * local variable declaration to the function assigning the temporary parameter to the pattern.
150        *
151        * <p> Note: Rewrites of variable declaration destructuring will happen later to rewrite
152        * this declaration as non-destructured code.
153        * @param function
154        * @param insertSpot The local variable declaration will be inserted after this statement.
155        * @param patternParam
156        * @param tempVarName the name to use for the temporary variable
157        * @return the declaration statement that was generated for the local variable
158        */
159       private Node replacePatternParamWithTempVar(
160           Node function, Node insertSpot, Node patternParam, String tempVarName) {
161         Node newParam = IR.name(tempVarName);
162         newParam.setJSDocInfo(patternParam.getJSDocInfo());
163         patternParam.getParent().replaceChild(patternParam, newParam);
164         Node newDecl = IR.var(patternParam, IR.name(tempVarName));
165         function.getLastChild().addChildAfter(newDecl, insertSpot);
166         return newDecl;
167       }

169       /**
170        * Find or create the best name to use for a parameter we need to rewrite.
171        *
172        * <ol>
173        * <li> Use the JS Doc function parameter name at the given index, if possible.
174        * <li> Otherwise, build one of our own.
175        * </ol>
176        * @param function
177        * @param parameterIndex
178        * @return name to use for the given parameter
179        */
180       private String getTempParameterName(Node function, int parameterIndex) {
181         String tempVarName;
182         JSDocInfo fnJSDoc = NodeUtil.getBestJSDocInfo(function);
183         if (fnJSDoc != null && fnJSDoc.getParameterNameAt(parameterIndex) != null) {
184           tempVarName = fnJSDoc.getParameterNameAt(parameterIndex);
185         } else {
186           tempVarName = DESTRUCTURING_TEMP_VAR + (destructuringVarCounter++);
187         }
188         Preconditions.checkState(TokenStream.isJSIdentifier(tempVarName));
189         return tempVarName;
190       }

192       private void visitForOf(Node node) {
193         Node lhs = node.getFirstChild();
194         if (lhs.isDestructuringLhs()) {
195           visitDestructuringPatternInEnhancedFor(lhs.getFirstChild());
196         }
197       }

199       private void visitObjectPattern(NodeTraversal t, Node objectPattern, Node parent) {
200         Node rhs, nodeToDetach;
201         if (NodeUtil.isNameDeclaration(parent) && !NodeUtil.isEnhancedFor(parent.getParent())) {
202           rhs = objectPattern.getNext();
203           nodeToDetach = parent;
204         } else if (parent.isAssign() && parent.getParent().isExprResult()) {
205           rhs = parent.getLastChild();
206           nodeToDetach = parent.getParent();
207         } else if (parent.isRest()
208             || parent.isStringKey()
209             || parent.isArrayPattern()
210             || parent.isDefaultValue()) {
211           // Nested object pattern; do nothing. We will visit it after rewriting the parent.
212           return;
213         } else if (NodeUtil.isEnhancedFor(parent) || NodeUtil.isEnhancedFor(parent.getParent())) {
214           visitDestructuringPatternInEnhancedFor(objectPattern);
215           return;
216         } else if (parent.isCatch()) {
217           visitDestructuringPatternInCatch(objectPattern);
218           return;
219         } else {
220           throw new IllegalStateException("Unexpected OBJECT_PATTERN parent: " + parent);
221         }
222     
223         // Convert 'var {a: b, c: d} = rhs' to:
224         // /** @const */ var temp = rhs;
225         // var b = temp.a;
226         // var d = temp.c;
227         String tempVarName = DESTRUCTURING_TEMP_VAR + (destructuringVarCounter++);
228         Node tempDecl = IR.var(IR.name(tempVarName), rhs.detachFromParent())
229                 .useSourceInfoIfMissingFromForTree(objectPattern);
230         // TODO(tbreisacher): Remove the "if" and add this JSDoc unconditionally.
231         if (parent.isConst()) {
232           JSDocInfoBuilder jsDoc = new JSDocInfoBuilder(false);
233           jsDoc.recordConstancy();
234           tempDecl.setJSDocInfo(jsDoc.build());
235         }
236         nodeToDetach.getParent().addChildBefore(tempDecl, nodeToDetach);
237     
238         for (Node child = objectPattern.getFirstChild(), next; child != null; child = next) {
239           next = child.getNext();
240     
241           Node newLHS, newRHS;
242           if (child.isStringKey()) {
243             if (!child.hasChildren()) { // converting shorthand
244               Node name = IR.name(child.getString());
245               name.useSourceInfoIfMissingFrom(child);
246               child.addChildToBack(name);
247             }
248             Node getprop =
249                 new Node(
250                     child.isQuotedString() ? Token.GETELEM : Token.GETPROP,
251                     IR.name(tempVarName),
252                     IR.string(child.getString()));
253     
254             Node value = child.removeFirstChild();
255             if (!value.isDefaultValue()) {
256               newLHS = value;
257               newRHS = getprop;
258             } else {
259               newLHS = value.removeFirstChild();
260               Node defaultValue = value.removeFirstChild();
261               newRHS = defaultValueHook(getprop, defaultValue);
262             }
263           } else if (child.isComputedProp()) {
264             if (child.getLastChild().isDefaultValue()) {
265               newLHS = child.getLastChild().removeFirstChild();
266               Node getelem = IR.getelem(IR.name(tempVarName), child.removeFirstChild());
267     
268               String intermediateTempVarName = DESTRUCTURING_TEMP_VAR + (destructuringVarCounter++);
269               Node intermediateDecl = IR.var(IR.name(intermediateTempVarName), getelem);
270               intermediateDecl.useSourceInfoIfMissingFromForTree(child);
271               nodeToDetach.getParent().addChildBefore(intermediateDecl, nodeToDetach);
272     
273               newRHS =
274                   defaultValueHook(
275                       IR.name(intermediateTempVarName), child.getLastChild().removeFirstChild());
276             } else {
277               newRHS = IR.getelem(IR.name(tempVarName), child.removeFirstChild());
278               newLHS = child.removeFirstChild();
279             }
280           } else if (child.isDefaultValue()) {
281             newLHS = child.removeFirstChild();
282             Node defaultValue = child.removeFirstChild();
283             Node getprop = IR.getprop(IR.name(tempVarName), IR.string(newLHS.getString()));
284             newRHS = defaultValueHook(getprop, defaultValue);
285           } else {
286             throw new IllegalStateException("Unexpected OBJECT_PATTERN child: " + child);
287           }
288     
289           Node newNode;
290           if (NodeUtil.isNameDeclaration(parent)) {
291             newNode = IR.declaration(newLHS, newRHS, parent.getType());
292           } else if (parent.isAssign()) {
293             newNode = IR.exprResult(IR.assign(newLHS, newRHS));
294           } else {
295             throw new IllegalStateException("not reached");
296           }
297           newNode.useSourceInfoIfMissingFromForTree(child);
298     
299           nodeToDetach.getParent().addChildBefore(newNode, nodeToDetach);
300     
301           // Explicitly visit the LHS of the new node since it may be a nested
302           // destructuring pattern.
303           visit(t, newLHS, newLHS.getParent());
304         }
305     
306         nodeToDetach.detachFromParent();
307         compiler.reportCodeChange();
308       }

310       private void visitArrayPattern(NodeTraversal t, Node arrayPattern, Node parent) {
311         Node rhs, nodeToDetach;
312         if (NodeUtil.isNameDeclaration(parent) && !NodeUtil.isEnhancedFor(parent.getParent())) {
313           rhs = arrayPattern.getNext();
314           nodeToDetach = parent;
315         } else if (parent.isAssign()) {
316           rhs = arrayPattern.getNext();
317           nodeToDetach = parent.getParent();
318           Preconditions.checkState(nodeToDetach.isExprResult());
319         } else if (parent.isArrayPattern()
320             || parent.isRest()
321             || parent.isDefaultValue()
322             || parent.isStringKey()) {
323           // This is a nested array pattern. Don't do anything now; we'll visit it
324           // after visiting the parent.
325           return;
326         } else if (NodeUtil.isEnhancedFor(parent) || NodeUtil.isEnhancedFor(parent.getParent())) {
327           visitDestructuringPatternInEnhancedFor(arrayPattern);
328           return;
329         } else if (parent.isCatch()) {
330           visitDestructuringPatternInCatch(arrayPattern);
331           return;
332         } else {
333           throw new IllegalStateException("Unexpected ARRAY_PATTERN parent: " + parent);
334         }
335     
336         // Convert 'var [x, y] = rhs' to:
337         // var temp = $jscomp.makeIterator(rhs);
338         // var x = temp.next().value;
339         // var y = temp.next().value;
340         String tempVarName = DESTRUCTURING_TEMP_VAR + (destructuringVarCounter++);
341         Node tempDecl = IR.var(
342             IR.name(tempVarName),
343             makeIterator(compiler, rhs.detachFromParent()));
344         tempDecl.useSourceInfoIfMissingFromForTree(arrayPattern);
345         nodeToDetach.getParent().addChildBefore(tempDecl, nodeToDetach);
346         boolean needsRuntime = false;
347     
348         for (Node child = arrayPattern.getFirstChild(), next; child != null; child = next) {
349           next = child.getNext();
350           if (child.isEmpty()) {
351             // Just call the next() method to advance the iterator, but throw away the value.
352             Node nextCall = IR.exprResult(IR.call(IR.getprop(IR.name(tempVarName), IR.string("next"))));
353             nextCall.useSourceInfoIfMissingFromForTree(child);
354             nodeToDetach.getParent().addChildBefore(nextCall, nodeToDetach);
355             continue;
356           }
357     
358           Node newLHS, newRHS;
359           if (child.isDefaultValue()) {
360             //   [x = defaultValue] = rhs;
361             // becomes
362             //   var temp0 = $jscomp.makeIterator(rhs);
363             //   var temp1 = temp.next().value
364             //   x = (temp1 === undefined) ? defaultValue : temp1;
365             String nextVarName = DESTRUCTURING_TEMP_VAR + (destructuringVarCounter++);
366             Node var = IR.var(
367                 IR.name(nextVarName),
368                 IR.getprop(
369                     IR.call(IR.getprop(IR.name(tempVarName), IR.string("next"))),
370                     IR.string("value")));
371             var.useSourceInfoIfMissingFromForTree(child);
372             nodeToDetach.getParent().addChildBefore(var, nodeToDetach);
373     
374             newLHS = child.getFirstChild().detachFromParent();
375             newRHS = defaultValueHook(IR.name(nextVarName), child.getLastChild().detachFromParent());
376           } else if (child.isRest()) {
377             //   [...x] = rhs;
378             // becomes
379             //   var temp = $jscomp.makeIterator(rhs);
380             //   x = $jscomp.arrayFromIterator(temp);
381             newLHS = child.getFirstChild().detachFromParent();
382             newRHS =
383                 IR.call(
384                     NodeUtil.newQName(compiler, "$jscomp.arrayFromIterator"),
385                     IR.name(tempVarName));
386             needsRuntime = true;
387           } else {
388             // LHS is just a name (or a nested pattern).
389             //   var [x] = rhs;
390             // becomes
391             //   var temp = $jscomp.makeIterator(rhs);
392             //   var x = temp.next().value;
393             newLHS = child.detachFromParent();
394             newRHS = IR.getprop(
395                 IR.call(IR.getprop(IR.name(tempVarName), IR.string("next"))),
396                 IR.string("value"));
397           }
398           Node newNode;
399           if (parent.isAssign()) {
400             Node assignment = IR.assign(newLHS, newRHS);
401             newNode = IR.exprResult(assignment);
402           } else {
403             newNode = IR.declaration(newLHS, newRHS, parent.getType());
404           }
405           newNode.useSourceInfoIfMissingFromForTree(arrayPattern);
406     
407           nodeToDetach.getParent().addChildBefore(newNode, nodeToDetach);
408           // Explicitly visit the LHS of the new node since it may be a nested
409           // destructuring pattern.
410           visit(t, newLHS, newLHS.getParent());
411         }
412         nodeToDetach.detachFromParent();
413     
414         if (needsRuntime) {
415           compiler.ensureLibraryInjected("es6_runtime", false);
416         }
417         compiler.reportCodeChange();
418       }

420       private void visitDestructuringPatternInEnhancedFor(Node pattern) {
421         Preconditions.checkArgument(pattern.isDestructuringPattern());
422         String tempVarName = DESTRUCTURING_TEMP_VAR + (destructuringVarCounter++);
423         if (NodeUtil.isEnhancedFor(pattern.getParent())) {
424           Node forNode = pattern.getParent();
425           Node block = forNode.getLastChild();
426           Node decl = IR.var(IR.name(tempVarName));
427           decl.useSourceInfoIfMissingFromForTree(pattern);
428           forNode.replaceChild(pattern, decl);
429           Node exprResult = IR.exprResult(IR.assign(pattern, IR.name(tempVarName)));
430           exprResult.useSourceInfoIfMissingFromForTree(pattern);
431           block.addChildToFront(exprResult);
432         } else {
433           Node destructuringLhs = pattern.getParent();
434           Preconditions.checkState(destructuringLhs.isDestructuringLhs());
435           Node declarationNode = destructuringLhs.getParent();
436           Node forNode = declarationNode.getParent();
437           Preconditions.checkState(NodeUtil.isEnhancedFor(forNode));
438           Node block = forNode.getLastChild();
439           declarationNode.replaceChild(
440               destructuringLhs, IR.name(tempVarName).useSourceInfoFrom(pattern));
441           Token declarationType = declarationNode.getType();
442           Node decl = IR.declaration(pattern.detachFromParent(), IR.name(tempVarName), declarationType);
443           decl.useSourceInfoIfMissingFromForTree(pattern);
444           block.addChildToFront(decl);
445         }
446       }
447     
448       private void visitDestructuringPatternInCatch(Node pattern) {
449         String tempVarName = DESTRUCTURING_TEMP_VAR + (destructuringVarCounter++);
450         Node catchBlock = pattern.getNext();
451     
452         pattern.getParent().replaceChild(pattern, IR.name(tempVarName));
453         catchBlock.addChildToFront(IR.declaration(pattern, IR.name(tempVarName), Token.LET));
454       }

456       /**
457        * Helper for transpiling DEFAULT_VALUE trees.
458        */
459       private static Node defaultValueHook(Node getprop, Node defaultValue) {
460         return IR.hook(IR.sheq(getprop, IR.name("undefined")), defaultValue, getprop.cloneTree());
461       }
