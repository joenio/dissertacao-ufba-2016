40     package com.google.javascript.rhino.jstype;
41     
42     
43     
44     import com.google.javascript.rhino.ErrorReporter;
45     import com.google.javascript.rhino.Node;
46     
47     /**
48      * The type of individual elements of an enum type
49      * (see {@link EnumType}).
50      */
51     public class EnumElementType extends ObjectType {
52       private static final long serialVersionUID = 1L;
53     
54       /**
55        * The primitive type this enum element type wraps. For instance, in
56        * the following code defining the {@code LOCAL_CODES} enum
57        * <pre>var LOCAL_CODES = {A: 3, B: 9, C: 8}</pre>
58        * the primitive type of the the constants is {@code number}.
59        */
60       private JSType primitiveType;
61     
62       // The primitive type, if it is an object.
63       private ObjectType primitiveObjectType;
64     
65       private final String name;
66     
67       private final EnumType enumType;
68     
69       EnumElementType(JSTypeRegistry registry, JSType elementType,
70           String name, EnumType enumType) {
71         super(registry);
72         this.primitiveType = elementType;
73         this.primitiveObjectType = elementType.toObjectType();
74         this.name = name;
75         this.enumType = enumType;
76       }
77     
78       public EnumType getEnumType() {
79         return enumType;
80       }
81     
82       @Override public PropertyMap getPropertyMap() {
83         return primitiveObjectType == null
84             ? PropertyMap.immutableEmptyMap()
85             : primitiveObjectType.getPropertyMap();
86       }
87     
88       @Override
89       public EnumElementType toMaybeEnumElementType() {
90         return this;
91       }
92     
93       @Override
94       public boolean matchesNumberContext() {
95         return primitiveType.matchesNumberContext();
96       }
97     
98       @Override
99       public boolean matchesStringContext() {
100         return primitiveType.matchesStringContext();
101       }
102     
103       @Override
104       public boolean matchesObjectContext() {
105         return primitiveType.matchesObjectContext();
106       }
107     
108       @Override
109       public boolean canBeCalled() {
110         return primitiveType.canBeCalled();
111       }
112     
113       @Override
114       public boolean isObject() {
115         return primitiveType.isObject();
116       }
117     
118       @Override
119       public TernaryValue testForEquality(JSType that) {
120         return primitiveType.testForEquality(that);
121       }
122     
123       /**
124        * This predicate determines whether objects of this type can have the null
125        * value, and therefore can appear in contexts where null is expected.
126        *
127        * @return true for everything but Number and Boolean types.
128        */
129       @Override
130       public boolean isNullable() {
131         return primitiveType.isNullable();
132       }
133     
134       @Override
135       public boolean isNominalType() {
136         return hasReferenceName();
137       }
138     
139       /**
140        * If this is equal to a NamedType object, its hashCode must be equal
141        * to the hashCode of the NamedType object.
142        */
143       @Override
144       public int hashCode() {
145         if (hasReferenceName()) {
146           return getReferenceName().hashCode();
147         } else {
148           return super.hashCode();
149         }
150       }
151     
152       @Override
153       String toStringHelper(boolean forAnnotations) {
154         return forAnnotations ?
155             primitiveType.toString() :
156             (getReferenceName() + "<" + primitiveType + ">");
157       }
158     
159       @Override
160       public String getReferenceName() {
161         return name;
162       }
163     
164       @Override
165       public boolean hasReferenceName() {
166         return true;
167       }
168     
169       @Override
170       public boolean isSubtype(JSType that) {
171         return isSubtype(that, ImplCache.create(), SubtypingMode.NORMAL);
172       }
173     
174       @Override
175       protected boolean isSubtype(JSType that,
176           ImplCache implicitImplCache, SubtypingMode subtypingMode) {
177         if (JSType.isSubtypeHelper(this, that, implicitImplCache, subtypingMode)) {
178           return true;
179         } else {
180           return primitiveType.isSubtype(that, implicitImplCache, subtypingMode);
181         }
182       }
183     
184       @Override
185       public <T> T visit(Visitor<T> visitor) {
186         return visitor.caseEnumElementType(this);
187       }
188     
189       @Override <T> T visit(RelationshipVisitor<T> visitor, JSType that) {
190         return visitor.caseEnumElementType(this, that);
191       }
192     
193       @Override
194       boolean defineProperty(String propertyName, JSType type,
195           boolean inferred, Node propertyNode) {
196         // nothing
197         return true;
198       }
199     
200       @Override
201       public ObjectType getImplicitPrototype() {
202         return null;
203       }
204     
205       @Override
206       public JSType findPropertyType(String propertyName) {
207         return primitiveType.findPropertyType(propertyName);
208       }
209     
210       @Override
211       public FunctionType getConstructor() {
212         return primitiveObjectType == null ?
213             null : primitiveObjectType.getConstructor();
214       }
215     
216       @Override
217       public JSType autoboxesTo() {
218         return primitiveType.autoboxesTo();
219       }
220     
221       /**
222        * Gets the primitive type of this enum element.
223        */
224       public JSType getPrimitiveType() {
225         return primitiveType;
226       }
227     
228       /**
229        * Returns the infimum of a enum element type and another type, or null
230        * if the infimum is empty.
231        *
232        * This can be a little bit weird. For example, suppose you have an enum
233        * of {(string|number)}, and you want the greatest subtype of the enum
234        * and a {number}.
235        *
236        * The infimum is non-empty. But at the same time, we don't really have
237        * a name for this infimum. It's equivalent to "elements of this enum that
238        * are numbers".
239        *
240        * The best we can do is make up a new type. This is similar to what
241        * we do in UnionType#meet, which kind-of-sort-of makes sense, because
242        * an EnumElementType is a union of instances of a type.
243        */
244       JSType meet(JSType that) {
245         JSType meetPrimitive = primitiveType.getGreatestSubtype(that);
246         if (meetPrimitive.isEmptyType()) {
247           return null;
248         } else {
249           return new EnumElementType(registry, meetPrimitive, name, getEnumType());
250         }
251       }
252     
253       @Override
254       JSType resolveInternal(ErrorReporter t, StaticTypedScope<JSType> scope) {
255         primitiveType = primitiveType.resolve(t, scope);
256         primitiveObjectType = ObjectType.cast(primitiveType);
257         return this;
258       }
259     }
