17     package com.google.javascript.jscomp;
18     
19     import com.google.common.base.Preconditions;
20     import com.google.javascript.rhino.Node;
21     
22     /**
23      * An abstract class whose implementations run peephole optimizations:
24      * optimizations that look at a small section of code and either remove
25      * that code (if it is not needed) or replaces it with smaller code.
26      *
27      */
28     abstract class AbstractPeepholeOptimization {
29     
30       private AbstractCompiler compiler;
31     
32       /**
33        * Given a node to optimize and a traversal, optimize the node. Subclasses
34        * should override to provide their own peephole optimization.
35        *
36        * @param subtree The subtree that will be optimized.
37        * @return The new version of the subtree (or null if the subtree or one of
38        * its parents was removed from the AST). If the subtree has not changed,
39        * this method must return {@code subtree}.
40        */
41       abstract Node optimizeSubtree(Node subtree);
42     
43       /**
44        * Helper method for reporting an error to the compiler when applying a
45        * peephole optimization.
46        *
47        * @param diagnostic The error type
48        * @param n The node for which the error should be reported
49        */
50       protected void report(DiagnosticType diagnostic, Node n) {
51         JSError error = JSError.make(n, diagnostic, n.toString());
52         compiler.report(error);
53       }
54     
55       /**
56        * Helper method for telling the compiler that something has changed.
57        * Subclasses must call these if they have changed the AST.
58        */
59       protected void reportCodeChange() {
60         Preconditions.checkNotNull(compiler);
61         compiler.reportCodeChange();
62       }
63     
64       /**
65        * Are the nodes equal for the purpose of inlining?
66        * If type aware optimizations are on, type equality is checked.
67        */
68       protected boolean areNodesEqualForInlining(Node n1, Node n2) {
69         /* Our implementation delegates to the compiler. We provide this
70          * method because we don't want to expose Compiler to PeepholeOptimizations.
71          */
72         Preconditions.checkNotNull(compiler);
73         return compiler.areNodesEqualForInlining(n1, n2);
74       }
75     
76       /**
77        *  Is the current AST normalized? (e.g. has the Normalize pass been run
78        *  and has the Denormalize pass not yet been run?)
79        */
80       protected boolean isASTNormalized() {
81         Preconditions.checkNotNull(compiler);
82     
83         return compiler.getLifeCycleStage().isNormalized();
84       }
85     
86       /**
87        * Informs the optimization that a traversal will begin.
88        */
89       void beginTraversal(AbstractCompiler compiler) {
90         this.compiler = compiler;
91       }
92     
93       /**
94        * Informs the optimization that a traversal has completed.
95        * @param compiler The current compiler.
96        */
97       void endTraversal(AbstractCompiler compiler) {
98         this.compiler = null;
99       }
100     
101       // NodeUtil's mayEffectMutableState and mayHaveSideEffects need access to the
102       // compiler object, route them through here to give them access.
103     
104       /**
105        * @return Whether the node may create new mutable state, or change existing
106        * state.
107        */
108       boolean mayEffectMutableState(Node n) {
109         return NodeUtil.mayEffectMutableState(n, compiler);
110       }
111     
112       /**
113        * @return Whether the node may have side effects when executed.
114        */
115       boolean mayHaveSideEffects(Node n) {
116         return NodeUtil.mayHaveSideEffects(n, compiler);
117       }
118     
119       /**
120        * Returns true if the current node's type implies side effects.
121        *
122        * This is a non-recursive version of the may have side effects
123        * check; used to check wherever the current node's type is one of
124        * the reason's why a subtree has side effects.
125        */
126       boolean nodeTypeMayHaveSideEffects(Node n) {
127         return NodeUtil.nodeTypeMayHaveSideEffects(n, compiler);
128       }
129     
130       /**
131        * @return Whether the output language is ECMAScript 5 or later.
132        *     Workarounds for quirks in browsers that do not support ES5 can be
133        *     ignored when this is true.
134        */
135       boolean isEcmaScript5OrGreater() {
136         return compiler != null && compiler.getOptions().getLanguageOut().isEs5OrHigher();
137       }
138     
139       /**
140        * @return the current coding convention.
141        */
142       CodingConvention getCodingConvention() {
143         // Note: this assumes a thread safe coding convention object.
144         return compiler.getCodingConvention();
145       }
146     
147       final boolean areDeclaredGlobalExternsOnWindow() {
148         return compiler != null && compiler.getOptions().declaredGlobalExternsOnWindow;
149       }
150     
151     }
