137       private StaticTypedScope<JSType> getScope(StaticTypedScope<JSType> scope, String name) {
138         StaticTypedSlot<JSType> slot = scope.getOwnSlot(name);
139         if (slot != null) {
140           return scope;
141         }
142         return getScope(scope.getParentScope(), name);
143       }

145       private JSType getType(String name) {
146         // Case template type names inside a class
147         // (borrowed from JSTypeRegistry#getType
148         JSType type = null;
149         JSType thisType = null;
150         if (scope != null && scope.getTypeOfThis() != null) {
151           thisType = scope.getTypeOfThis().toObjectType();
152         }
153         if (thisType != null) {
154           type = thisType.getTemplateTypeMap().getTemplateTypeKeyByName(name);
155           if (type != null) {
156             Preconditions.checkState(type.isTemplateType(),
157                 "Expected a template type, but found: %s", type);
158             return type;
159           }
160         }
161     
162         // Resolve the name and get the corresponding type
163         StaticTypedSlot<JSType> slot = scope.getSlot(name);
164         if (slot != null) {
165           JSType rawType = slot.getType();
166           if (rawType != null) {
167             // Case constructor, get the instance type
168             if ((rawType.isConstructor() || rawType.isInterface())
169                 && rawType.isFunctionType() && rawType.isNominalConstructor()) {
170               return rawType.toMaybeFunctionType().getInstanceType();
171             }
172             // Case enum
173             if (rawType.isEnumType()) {
174               return rawType.toMaybeEnumType().getElementsType();
175             }
176           }
177           // Case typedef
178           JSDocInfo info = slot.getJSDocInfo();
179           if (info != null && info.hasTypedefType()) {
180             JSTypeExpression expr = info.getTypedefType();
181             StaticTypedScope<JSType> typedefScope = getScope(scope, name);
182             return expr.evaluate(typedefScope, typeRegistry);
183           }
184         }
185         // Otherwise handle native types
186         return typeRegistry.getType(name);
187       }

296       /** Evaluates the type transformation expression and returns the resulting
297        * type.
298        *
299        * @param ttlAst The node representing the type transformation
300        * expression
301        * @param typeVars The environment containing the information about
302        * the type variables
303        * @param nameVars The environment containing the information about
304        * the name variables
305        * @return JSType The resulting type after the transformation
306        */
307       JSType eval(Node ttlAst, ImmutableMap<String, JSType> typeVars,
308           ImmutableMap<String, String> nameVars) {
309         return evalInternal(ttlAst, new NameResolver(typeVars, nameVars));
310       }

312       private JSType evalInternal(Node ttlAst, NameResolver nameResolver) {
313         if (isTypeName(ttlAst)) {
314           return evalTypeName(ttlAst);
315         }
316         if (isTypeVar(ttlAst)) {
317           return evalTypeVar(ttlAst, nameResolver);
318         }
319         String name = getCallName(ttlAst);
320         Keywords keyword = nameToKeyword(name);
321         switch (keyword.kind) {
322           case TYPE_CONSTRUCTOR:
323             return evalTypeExpression(ttlAst, nameResolver);
324           case OPERATION:
325             return evalOperationExpression(ttlAst, nameResolver);
326           default:
327             throw new IllegalStateException(
328                 "Could not evaluate the type transformation expression");
329         }
330       }
331     
332       private JSType evalOperationExpression(Node ttlAst, NameResolver nameResolver) {
333         String name = getCallName(ttlAst);
334         Keywords keyword = nameToKeyword(name);
335         switch (keyword) {
336           case COND:
337             return evalConditional(ttlAst, nameResolver);
338           case MAPUNION:
339             return evalMapunion(ttlAst, nameResolver);
340           case MAPRECORD:
341             return evalMaprecord(ttlAst, nameResolver);
342           case TYPEOFVAR:
343             return evalTypeOfVar(ttlAst);
344           case INSTANCEOF:
345             return evalInstanceOf(ttlAst, nameResolver);
346           case PRINTTYPE:
347             return evalPrintType(ttlAst, nameResolver);
348           case PROPTYPE:
349             return evalPropType(ttlAst, nameResolver);
350           default:
351             throw new IllegalStateException("Invalid type transformation operation");
352         }
353       }

355       private JSType evalTypeExpression(Node ttlAst, NameResolver nameResolver) {
356         String name = getCallName(ttlAst);
357         Keywords keyword = nameToKeyword(name);
358         switch (keyword) {
359           case TYPE:
360             return evalTemplatizedType(ttlAst, nameResolver);
361           case UNION:
362             return evalUnionType(ttlAst, nameResolver);
363           case NONE:
364             return getNoType();
365           case ALL:
366             return getAllType();
367           case UNKNOWN:
368              return getUnknownType();
369           case RAWTYPEOF:
370             return evalRawTypeOf(ttlAst, nameResolver);
371           case TEMPLATETYPEOF:
372             return evalTemplateTypeOf(ttlAst, nameResolver);
373           case RECORD:
374             return evalRecordType(ttlAst, nameResolver);
375           case TYPEEXPR:
376             return evalNativeTypeExpr(ttlAst);
377           default:
378             throw new IllegalStateException("Invalid type expression");
379         }
380       }

382       private JSType evalTypeName(Node ttlAst) {
383         String typeName = ttlAst.getString();
384         JSType resultingType = getType(typeName);
385         // If the type name is not defined then return UNKNOWN and report a warning
386         if (resultingType == null) {
387           reportWarning(ttlAst, UNKNOWN_TYPENAME, typeName);
388           return getUnknownType();
389         }
390         return resultingType;
391       }

393       private JSType evalTemplatizedType(Node ttlAst, NameResolver nameResolver) {
394         ImmutableList<Node> params = getCallParams(ttlAst);
395         JSType firstParam = evalInternal(params.get(0), nameResolver);
396         if (!isTemplatizable(firstParam)) {
397           reportWarning(ttlAst, BASETYPE_INVALID, firstParam.toString());
398           return getUnknownType();
399         }
400         ObjectType baseType = firstParam.toObjectType();
401         // TODO(lpino): Check that the number of parameters correspond with the
402         // number of template types that the base type can take when creating
403         // a templatized type. For instance, if the base type is Array then there
404         // must be just one parameter.
405         JSType[] templatizedTypes = new JSType[params.size() - 1];
406         for (int i = 0; i < templatizedTypes.length; i++) {
407           templatizedTypes[i] = evalInternal(params.get(i + 1), nameResolver);
408         }
409         return createTemplatizedType(baseType, templatizedTypes);
410       }

413         String typeVar = ttlAst.getString();
414         JSType resultingType = nameResolver.typeVars.get(typeVar);
415         // If the type variable is not defined then return UNKNOWN and report a warning
416         if (resultingType == null) {
417           reportWarning(ttlAst, UNKNOWN_TYPEVAR, typeVar);
418           return getUnknownType();
419         }
420         return resultingType;

423       private JSType evalUnionType(Node ttlAst, NameResolver nameResolver) {
424         // Get the parameters of the union
425         ImmutableList<Node> params = getCallParams(ttlAst);
426         int paramCount = params.size();
427         // Create an array of types after evaluating each parameter
428         JSType[] basicTypes = new JSType[paramCount];
429         for (int i = 0; i < paramCount; i++) {
430           basicTypes[i] = evalInternal(params.get(i), nameResolver);
431         }
432         return createUnionType(basicTypes);
433       }

445       private String evalString(Node ttlAst, NameResolver nameResolver) {
446         if (ttlAst.isName()) {
447           // Return the empty string if the name variable cannot be resolved
448           if (!nameResolver.nameVars.containsKey(ttlAst.getString())) {
449             reportWarning(ttlAst, UNKNOWN_STRVAR, ttlAst.getString());
450             return "";
451           }
452           return nameResolver.nameVars.get(ttlAst.getString());
453         }
454         return ttlAst.getString();
455       }

458         ImmutableList<Node> params = getCallParams(ttlAst);

467       private boolean evalTypePredicate(Node ttlAst,
468           NameResolver nameResolver) {
469         JSType[] params = evalTypeParams(ttlAst, nameResolver);
470         String name = getCallName(ttlAst);
471         Keywords keyword = nameToKeyword(name);
472         switch (keyword) {
473           case EQ:
474             return params[0].isEquivalentTo(params[1]);
475           case SUB:
476             return params[0].isSubtype(params[1]);
477           case ISCTOR:
478             return params[0].isConstructor();
479           case ISTEMPLATIZED:
480             return params[0].isTemplatizedType();
481           case ISRECORD:
482             return params[0].isRecordType();
483           case ISUNKNOWN:
484             return params[0].isUnknownType() || params[0].isCheckedUnknownType();
485           default:
486             throw new IllegalStateException(
487                 "Invalid type predicate in the type transformation");
488         }
489       }
490     
491       private boolean evalStringPredicate(Node ttlAst,
492           NameResolver nameResolver) {
493         String[] params = evalStringParams(ttlAst, nameResolver);
494         // If any of the parameters evaluates to the empty string then they were
495         // not resolved by the name resolver. In this case we always return false.
496         for (int i = 0; i < params.length; i++) {
497           if (params[i].isEmpty()) {
498             return false;
499           }
500         }
501         String name = getCallName(ttlAst);
502         Keywords keyword = nameToKeyword(name);
503         switch (keyword) {
504           case STREQ:
505             return params[0].equals(params[1]);
506           default:
507             throw new IllegalStateException(
508                 "Invalid string predicate in the type transformation");
509         }
510       }

512       private boolean evalTypevarPredicate(Node ttlAst, NameResolver nameResolver) {
513         String name = getCallName(ttlAst);
514         Keywords keyword = nameToKeyword(name);
515         switch (keyword) {
516           case ISDEFINED:
517             return nameResolver.typeVars.containsKey(getCallArgument(ttlAst, 0).getString());
518           default:
519             throw new IllegalStateException(
520                 "Invalid typevar predicate in the type transformation");
521         }
522       }

524        private boolean evalBooleanOperation(Node ttlAst, NameResolver nameResolver) {
525          boolean param0 = evalBoolean(ttlAst.getFirstChild(), nameResolver);
526          if (ttlAst.isNot()) {
527            return !param0;
528          }
529          if (ttlAst.isAnd()) {
530            return param0 && evalBoolean(ttlAst.getLastChild(), nameResolver);
531          }
532          if (ttlAst.isOr()) {
533            return param0 || evalBoolean(ttlAst.getLastChild(), nameResolver);
534          }
535          throw new IllegalStateException(
536              "Invalid boolean predicate in the type transformation");
537       }
538     
539       private boolean evalBoolean(Node ttlAst, NameResolver nameResolver) {
540         if (isBooleanOperation(ttlAst)) {
541           return evalBooleanOperation(ttlAst, nameResolver);
542         }
543         String name = getCallName(ttlAst);
544         Keywords keyword = nameToKeyword(name);
545         switch (keyword.kind) {
546           case STRING_PREDICATE:
547             return evalStringPredicate(ttlAst, nameResolver);
548           case TYPE_PREDICATE:
549             return evalTypePredicate(ttlAst, nameResolver);
550           case TYPEVAR_PREDICATE:
551             return evalTypevarPredicate(ttlAst, nameResolver);
552           default:
553             throw new IllegalStateException(
554                 "Invalid boolean predicate in the type transformation");
555         }
556       }

558       private JSType evalConditional(Node ttlAst, NameResolver nameResolver) {
559         ImmutableList<Node> params = getCallParams(ttlAst);
560         if (evalBoolean(params.get(0), nameResolver)) {
561           return evalInternal(params.get(1), nameResolver);
562         } else {
563           return evalInternal(params.get(2), nameResolver);
564         }
565       }

567       private JSType evalMapunion(Node ttlAst, NameResolver nameResolver) {
568         ImmutableList<Node> params = getCallParams(ttlAst);
569         Node unionParam = params.get(0);
570         Node mapFunction = params.get(1);
571         String paramName = getFunctionParameter(mapFunction, 0);
572     
573         // The mapunion variable must not be defined in the environment
574         if (nameResolver.typeVars.containsKey(paramName)) {
575           reportWarning(ttlAst, DUPLICATE_VARIABLE, paramName);
576           return getUnknownType();
577         }
578     
579         Node mapFunctionBody = getFunctionBody(mapFunction);
580         JSType unionType = evalInternal(unionParam, nameResolver);
581         // If the first parameter does not correspond to a union type then
582         // consider it as a union with a single type and evaluate
583         if (!unionType.isUnionType()) {
584           NameResolver newNameResolver = new NameResolver(
585               addNewEntry(nameResolver.typeVars, paramName, unionType),
586               nameResolver.nameVars);
587           return evalInternal(mapFunctionBody, newNameResolver);
588         }
589     
590         // Otherwise obtain the elements in the union type. Note that the block
591         // above guarantees the casting to be safe
592         Collection<JSType> unionElms = ((UnionType) unionType).getAlternates();
593         // Evaluate the map function body using each element in the union type
594         int unionSize = unionElms.size();
595         JSType[] newUnionElms = new JSType[unionSize];
596         int i = 0;
597         for (JSType elm : unionElms) {
598           NameResolver newNameResolver = new NameResolver(
599               addNewEntry(nameResolver.typeVars, paramName, elm),
600               nameResolver.nameVars);
601           newUnionElms[i] = evalInternal(mapFunctionBody, newNameResolver);
602           i++;
603         }
604     
605         return createUnionType(newUnionElms);
606       }

608       private JSType evalRawTypeOf(Node ttlAst, NameResolver nameResolver) {
609         ImmutableList<Node> params = getCallParams(ttlAst);
610         JSType type = evalInternal(params.get(0), nameResolver);
611         if (!type.isTemplatizedType()) {
612           reportWarning(ttlAst, TEMPTYPE_INVALID, "rawTypeOf", type.toString());
613           return getUnknownType();
614         }
615         return ((TemplatizedType) type).getReferencedType();
616       }

618       private JSType evalTemplateTypeOf(Node ttlAst, NameResolver nameResolver) {
619         ImmutableList<Node> params = getCallParams(ttlAst);
620         JSType type = evalInternal(params.get(0), nameResolver);
621         if (!type.isTemplatizedType()) {
622           reportWarning(ttlAst, TEMPTYPE_INVALID, "templateTypeOf", type.toString());
623           return getUnknownType();
624         }
625         int index = (int) params.get(1).getDouble();
626         ImmutableList<JSType> templateTypes =
627             ((TemplatizedType) type).getTemplateTypes();
628         if (index > templateTypes.size()) {
629           reportWarning(ttlAst, INDEX_OUTOFBOUNDS,
630               Integer.toString(index), Integer.toString(templateTypes.size()));
631           return getUnknownType();
632         }
633         return templateTypes.get(index);
634       }

636       private JSType evalRecord(Node record, NameResolver nameResolver) {
637         RecordTypeBuilder builder = new RecordTypeBuilder(typeRegistry);
638         for (Node propNode : record.children()) {
639           // If it is a computed property then find the property name using the resolver
640           if (propNode.isComputedProp()) {
641             String compPropName = getComputedPropName(propNode);
642             // If the name does not exist then report a warning
643             if (!nameResolver.nameVars.containsKey(compPropName)) {
644               reportWarning(record, UNKNOWN_NAMEVAR, compPropName);
645               return getUnknownType();
646             }
647             // Otherwise add the property
648             Node propValue = getComputedPropValue(propNode);
649             String resolvedName = nameResolver.nameVars.get(compPropName);
650             JSType resultingType = evalInternal(propValue, nameResolver);
651             builder.addProperty(resolvedName, resultingType, null);
652           } else {
653             String propName = propNode.getString();
654             JSType resultingType = evalInternal(propNode.getFirstChild(),
655                 nameResolver);
656             builder.addProperty(propName, resultingType, null);
657           }
658         }
659         return builder.build();
660       }

663         if (ttlAst.isObjectLit()) {
664           return evalRecord(ttlAst, nameResolver);
665         }
666         // The parameter of record can be a type transformation expression
667         return evalInternal(ttlAst, nameResolver);
668       }

670       private JSType buildRecordTypeFromObject(ObjectType objType) {
671         RecordType recType = objType.toMaybeRecordType();
672         // If it can be casted to a record type then return
673         if (recType != null) {
674           return recType;
675         }
676         // TODO(lpino): Handle inherited properties
677         Set<String> propNames = objType.getOwnPropertyNames();
678         // If the type has no properties then return Object
679         if (propNames.isEmpty()) {
680           return getObjectType();
681         }
682         ImmutableMap.Builder<String, JSType> props = new ImmutableMap.Builder<>();
683         // Otherwise collect the properties and build a record type
684         for (String propName : propNames) {
685           props.put(propName, objType.getPropertyType(propName));
686         }
687         return createRecordType(props.build());
688       }

690       private JSType evalRecordType(Node ttlAst, NameResolver nameResolver) {
691         int paramCount = getCallParamCount(ttlAst);
692         ImmutableList.Builder<RecordType> recTypesBuilder = new ImmutableList.Builder<>();
693         for (int i = 0; i < paramCount; i++) {
694           JSType type = evalRecordParam(getCallArgument(ttlAst, i), nameResolver);
695           // Check that each parameter evaluates to an object
696           ObjectType objType = type.toObjectType();
697           if (objType == null || objType.isUnknownType()) {
698             reportWarning(ttlAst, RECPARAM_INVALID, type.toString());
699             return getUnknownType();
700           }
701           JSType recType = buildRecordTypeFromObject(objType);
702           if (!recType.isEquivalentTo(getObjectType())) {
703             recTypesBuilder.add(recType.toMaybeRecordType());
704           }
705         }
706         return joinRecordTypes(recTypesBuilder.build());
707       }

709       private void putNewPropInPropertyMap(Map<String, JSType> props,
710           String newPropName, JSType newPropValue) {
711         // TODO(lpino): Decide if the best strategy is to collapse the properties
712         // to a union type or not. So far, new values replace the old ones except
713         // if they are two record types in which case the properties are joined
714         // together
715     
716         // Three cases:
717         // (i) If the key does not exist then add it to the map with the new value
718         // (ii) If the key to be added already exists in the map and the new value
719         // is not a record type then the current value is replaced with the new one
720         // (iii) If the new value is a record type and the current is not then
721         // the current value is replaced with the new one
722         if (!props.containsKey(newPropName)
723             || !newPropValue.isRecordType()
724             || !props.get(newPropName).isRecordType()) {
725           props.put(newPropName, newPropValue);
726           return;
727         }
728         // Otherwise join the current value with the new one since both are records
729         props.put(newPropName,
730             joinRecordTypes(ImmutableList.<RecordType>of(
731                 (RecordType) props.get(newPropName),
732                 (RecordType) newPropValue)));
733       }

735       private void addNewPropsFromRecordType(Map<String, JSType> props,
736           RecordType recType) {
737         for (String newPropName : recType.getOwnPropertyNames()) {
738           JSType newPropValue = recType.getSlot(newPropName).getType();
739           // Put the new property depending if it already exists in the map
740           putNewPropInPropertyMap(props, newPropName, newPropValue);
741         }
742       }

744       /**
745        * Merges a list of record types.
746        * Example
747        * {r:{s:string, n:number}} and {a:boolean}
748        * is transformed into {r:{s:string, n:number}, a:boolean}
749        */
750       private JSType joinRecordTypes(ImmutableList<RecordType> recTypes) {
751         Map<String, JSType> props = new LinkedHashMap<>();
752         for (int i = 0; i < recTypes.size(); i++) {
753           addNewPropsFromRecordType(props, recTypes.get(i));
754         }
755         return createRecordType(ImmutableMap.copyOf(props));
756       }

758       private JSType evalMaprecord(Node ttlAst, NameResolver nameResolver) {
759         ImmutableList<Node> params = getCallParams(ttlAst);
760         JSType type = evalInternal(params.get(0), nameResolver);
761     
762         // If it is an empty record type (Object) then return
763         if (type.isEquivalentTo(getObjectType())) {
764           return getObjectType();
765         }
766     
767         // The parameter must be a valid record type
768         if (!type.isRecordType()) {
769           // TODO(lpino): Decide how to handle non-record types
770           reportWarning(params.get(0), RECTYPE_INVALID, type.toString());
771           return getUnknownType();
772         }
773     
774         // Obtain the elements in the record type. Note that the block
775         // above guarantees the casting to be safe
776         RecordType recType = ((RecordType) type);
777         Set<String> ownPropsNames = recType.getOwnPropertyNames();
778     
779         // Fetch the information of the map function
780         Node mapFunction = params.get(1);
781         String paramKey = getFunctionParameter(mapFunction, 0);
782         String paramValue = getFunctionParameter(mapFunction, 1);
783     
784         // The maprecord variables must not be defined in the environment
785         if (nameResolver.nameVars.containsKey(paramKey)) {
786           reportWarning(ttlAst, DUPLICATE_VARIABLE, paramKey);
787           return getUnknownType();
788         }
789         if (nameResolver.typeVars.containsKey(paramValue)) {
790           reportWarning(ttlAst, DUPLICATE_VARIABLE, paramValue);
791           return getUnknownType();
792         }
793     
794         // Compute the new properties using the map function
795         Node mapFnBody = getFunctionBody(mapFunction);
796         Map<String, JSType> newProps = new LinkedHashMap<>();
797         for (String propName : ownPropsNames) {
798           // The value of the current property
799           JSType propValue = recType.getSlot(propName).getType();
800     
801           // Evaluate the map function body with paramValue and paramKey replaced
802           // by the values of the current property
803           NameResolver newNameResolver = new NameResolver(
804               addNewEntry(nameResolver.typeVars, paramValue, propValue),
805               addNewEntry(nameResolver.nameVars, paramKey, propName));
806           JSType body = evalInternal(mapFnBody, newNameResolver);
807     
808           // If the body returns unknown then the whole expression returns unknown
809           if (body.isUnknownType()) {
810             return getUnknownType();
811           }
812     
813           // Skip the property when the body evaluates to NO_TYPE
814           // or the empty record (Object)
815           if (body.isNoType() || body.isEquivalentTo(getObjectType())) {
816             continue;
817           }
818     
819           // Otherwise the body must evaluate to a record type
820           if (!body.isRecordType()) {
821             reportWarning(ttlAst, MAPRECORD_BODY_INVALID, body.toString());
822             return getUnknownType();
823           }
824     
825           // Add the properties of the resulting record type to the original one
826           RecordType bodyAsRecType = ((RecordType) body);
827           for (String newPropName : bodyAsRecType.getOwnPropertyNames()) {
828             JSType newPropValue = bodyAsRecType.getSlot(newPropName).getType();
829             // If the key already exists then we have to mix it with the current
830             // property value
831             putNewPropInPropertyMap(newProps, newPropName, newPropValue);
832           }
833         }
834         return createRecordType(ImmutableMap.copyOf(newProps));
835       }
