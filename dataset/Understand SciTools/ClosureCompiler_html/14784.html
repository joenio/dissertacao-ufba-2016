35       public CheckGlobalNamesTest() {
36         super("function alert() {}" +
37               "/** @constructor */ function Object(){}" +
38               "Object.prototype.hasOwnProperty = function() {};" +
39               "/** @constructor */ function Function(){}" +
40               "Function.prototype.call = function() {};");

43       @Override
44       protected CompilerPass getProcessor(final Compiler compiler) {
45         final CheckGlobalNames checkGlobalNames = new CheckGlobalNames(
46             compiler, CheckLevel.WARNING);
47         if (injectNamespace) {
48           return new CompilerPass() {
49             @Override
50             public void process(Node externs, Node js) {
51               checkGlobalNames.injectNamespace(
52                   new GlobalNamespace(compiler, externs, js))
53                   .process(externs, js);
54             }
55           };
56         } else {
57           return checkGlobalNames;
58         }
59       }
