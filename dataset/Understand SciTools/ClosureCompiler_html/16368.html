16     package com.google.javascript.refactoring.testing;

18     import static com.google.common.truth.Truth.assertThat;
19     import static org.junit.Assert.assertEquals;
20     
21     import com.google.common.collect.ImmutableList;
22     import com.google.common.collect.ImmutableMap;
23     import com.google.common.collect.ImmutableSet;
24     import com.google.common.collect.SetMultimap;
25     import com.google.javascript.refactoring.ApplySuggestedFixes;
26     import com.google.javascript.refactoring.CodeReplacement;
27     import com.google.javascript.refactoring.SuggestedFix;
28     
29     import java.util.List;
30     import java.util.Set;
31     
32     
33     /**
34      * Utilities for testing SuggestedFix-related code.
35      */
36     public final class SuggestedFixes {
37       private SuggestedFixes() {}
38     
39       public static void assertReplacement(SuggestedFix fix, CodeReplacement expectedReplacement) {
40         assertReplacements(fix, ImmutableSet.of(expectedReplacement));
41       }
42     
43       private static void assertReplacements(
44           SuggestedFix fix, Set<CodeReplacement> expectedReplacements) {
45         SetMultimap<String, CodeReplacement> replacementMap = fix.getReplacements();
46         assertEquals(1, replacementMap.size());
47         Set<CodeReplacement> replacements = replacementMap.get("test");
48         assertThat(replacements).hasSize(expectedReplacements.size());
49         assertEquals(expectedReplacements, replacements);
50       }
51     
52       public static void assertChanges(
53           SuggestedFix fix, String externs, String originalCode, String expectedCode) {
54         assertChanges(ImmutableList.of(fix), externs, originalCode, expectedCode);
55       }
56     
57       public static void assertChanges(
58           List<SuggestedFix> fixes, String externs, String originalCode, String expectedCode) {
59         assertThat(fixes).isNotEmpty();
60         String newCode = ApplySuggestedFixes.applySuggestedFixesToCode(
61             fixes, ImmutableMap.of("test", originalCode)).get("test");
62         assertEquals(expectedCode, newCode);
63       }
64     }
