31     public final class CheckNullableReturnTest extends TypeICompilerTestCase {
32       private static String externs = DEFAULT_EXTERNS + "/** @constructor */ function SomeType() {}";
33     
34       @Override
35       protected CompilerPass getProcessor(Compiler compiler) {
36         return new CheckNullableReturn(compiler);
37       }

39       @Override
40       protected CompilerOptions getOptions(CompilerOptions options) {
41         super.getOptions(options);
42         options.setWarningLevel(DiagnosticGroups.LINT_CHECKS, CheckLevel.WARNING);
43         return options;
44       }

46       @Override
47       protected int getNumRepetitions() {
48         return 1;
49       }

51       public void testSimpleWarning() {
52         testError(LINE_JOINER.join(
53             "/** @return {SomeType} */",
54             "function f() {",
55             "  return new SomeType();",
56             "}"));
57       }

59       public void testNullableReturn() {
60         testBodyOk("return null;");
61         testBodyOk("if (a) { return null; } return {};");
62         testBodyOk("switch(1) { case 12: return null; } return {};");
63         testBodyOk(
64             "/** @return {number} */ function f() { return 42; }; return null;");
65       }
66     
67       public void testNotNullableReturn()  {
68         // Empty function body. Ignore this case. The remainder of the functions in
69         // this test have non-empty bodies.
70         testBodyOkOti("");
71     
72         // Simple case.
73         testBodyError("return {};");
74     
75         // This implementation of an abstract method should not be reported.
76         testBodyOk("throw new Error('Not implemented');");
77     
78         // Nested function.
79         testBodyError(
80             "/** @return {number} */ function f() { return 1; }; return {};");
81     
82         testBodyError("switch(1) { default: return {}; } return null;");
83         testBodyError("switch(g) { case 1: return {}; default: return {}; } return null;");
84       }
85     
86       public void testFinallyStatements() {
87         testBodyOk("try { return null; } finally { return {}; }");
88         testBodyOk("try { } finally { return null; }");
89         testBodyOk("try { return {}; } finally { return null; }");
90         testBodyOk("try { return null; } finally { return {}; }");
91         testBodyErrorOti("try { } catch (e) { return null; } finally { return {}; }");
92       }
93     
94       public void testKnownConditions() {
95         testBodyOk("if (true) return {}; return null;");
96         testBodyOk("if (true) return null; else return {};");
97     
98         testBodyOk("if (false) return {}; return null;");
99         testBodyOk("if (false) return null; else return {};");
100     
101         testBodyError("if (1) return {}; return {x: 42};");
102         testBodyOk("if (1) { return null; } else { return {}; }");
103     
104         testBodyOk("if (0) return {}; return null;");
105         testBodyOk("if (0) { return null; } else { return {}; }");
106     
107         testBodyOk("if (3) return null; else return {};");
108       }
109     
110       public void testKnownWhileLoop() {
111         testBodyError("while (1) return {}");
112         testBodyError("while (1) { if (x) { return {}; } else { return {}; }}");
113         testBodyError("while (0) {} return {}");
114     
115         // Not known.
116         testBodyErrorOti("while(x) { return {}; }");
117       }

246       private static String createFunction(String body) {
247         return "/** @return {?Object} */ function foo() {" + body + "}";
248       }
249     
250       private static String createShorthandFunctionInObjLit(String body) {
251         return "var obj = {/** @return {?Object} */ foo() {" + body + "}}";
252       }
253     
254       private void testOk(String js) {
255         testSame(externs, js, null);
256       }
257     
258       private void testOkOti(String js) {
259         testSameOtiOnly(externs, js, null);
260       }
261     
262       private void testError(String js) {
263         testSame(externs, js, CheckNullableReturn.NULLABLE_RETURN_WITH_NAME);
264       }
265     
266       private void testErrorOti(String js) {
267         testSameOtiOnly(externs, js, CheckNullableReturn.NULLABLE_RETURN_WITH_NAME);
268       }

270       private void testBodyOk(String body) {
271         testOk(createFunction(body));
272         setAcceptedLanguage(LanguageMode.ECMASCRIPT6);
273         testOk(createShorthandFunctionInObjLit(body));
274       }

276       private void testBodyOkOti(String body) {
277         testOkOti(createFunction(body));
278         setAcceptedLanguage(LanguageMode.ECMASCRIPT6);
279         testOkOti(createShorthandFunctionInObjLit(body));
280       }

285         testError(createShorthandFunctionInObjLit(body));
286       }

289         testErrorOti(createFunction(body));
