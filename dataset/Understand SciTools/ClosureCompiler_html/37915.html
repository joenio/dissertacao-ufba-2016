30     public abstract class BaseReplaceScriptTestCase extends TestCase {
31       protected static final Joiner LINE_JOINER = Joiner.on('\n');
32     
33       protected static final String CLOSURE_BASE =
34           LINE_JOINER.join(
35               "/** @const */ var goog = goog || {};",
36               "goog.require = function(x) {};",
37               "goog.provide = function(x) {};");
38     
39       protected static final List<SourceFile> EXTERNS =
40           ImmutableList.of(SourceFile.fromCode("externs", "var extVar = 3;"));
41     
42       /**
43        * In addition to the passed parameter adds a few options necessary options for
44        * {@code replaceScript} and creates a {@code CompilerOptions}.
45        */
46       protected CompilerOptions getOptions(DiagnosticGroup... typesOfGuard) {
47         CompilerOptions options = new CompilerOptions();
48         options.declaredGlobalExternsOnWindow = false;
49         options.setClosurePass(true);
50         // These are the options that are always on in JsDev which is the only
51         // use-case for replaceScript currently.
52         options.setInferTypes(true);
53         options.setIdeMode(true);
54         for (DiagnosticGroup group : typesOfGuard) {
55           options.setWarningLevel(group, CheckLevel.ERROR);
56         }
57         return options;
58       }
59     
60       protected void flushResults(Compiler compiler) {
61         // TODO(bashir) Maybe implement error-flush functionality in Compiler?
62         compiler.setErrorManager(new PrintStreamErrorManager(System.err));
63       }
64     
65       protected void runReplaceScriptNoWarnings(
66           List<String> sources, String newSource, int newSourceInd) {
67         Result result =
68             runReplaceScript(getOptions(), sources, 0, 0, newSource, newSourceInd, true).getResult();
69         assertNoWarningsOrErrors(result);
70         assertTrue(result.success);
71       }
72     
73       protected void runReplaceScriptWithError(
74           List<String> sources, String newSource, int newSourceInd, DiagnosticType errorType) {
75         Result result =
76             runReplaceScript(getOptions(), sources, 0, 0, newSource, newSourceInd, true).getResult();
77         assertNumWarningsAndErrors(result, 1, 0);
78         assertEquals(result.errors[0].getType(), errorType);
79       }

81       /**
82        * For a given set of sources, first runs a full compile, then replaces one source with a given
83        * new version and calls {@code replaceScript}.
84        *
85        * @param options Compiler options.
86        * @param sources The list of sources.
87        * @param expectedCompileErrors Expected number of errors after full compile.
88        * @param expectedCompileWarnings Expected number of warnings of full compile.
89        * @param newSource The source version.
90        * @param newSourceInd Index of the source in {@code sources} to be replaced.
91        * @param flushResults Whether to flush results after full-build or not.
92        * @return The compiler which can be reused for further inc-compiles.
93        */
94       protected Compiler runReplaceScript(
95           CompilerOptions options,
96           List<String> sources,
97           int expectedCompileErrors,
98           int expectedCompileWarnings,
99           String newSource,
100           int newSourceInd,
101           boolean flushResults) {
102         Preconditions.checkArgument(newSourceInd < sources.size());
103     
104         // First do a full compile.
105         Compiler compiler =
106             runFullCompile(
107                 options, sources, expectedCompileErrors, expectedCompileWarnings, flushResults);
108     
109         // Now replace one of the source files and run replaceScript.
110         doReplaceScript(compiler, newSource, newSourceInd);
111         return compiler;
112       }
113     
114       protected Compiler runAddScript(
115           CompilerOptions options,
116           List<String> sources,
117           int expectedCompileErrors,
118           int expectedCompileWarnings,
119           String newSource,
120           boolean flushResults) {
121         // First do a full compile.
122         Compiler compiler =
123             runFullCompile(
124                 options, sources, expectedCompileErrors, expectedCompileWarnings, flushResults);
125     
126         // Now replace one of the source files and run replaceScript.
127         doAddScript(compiler, newSource, sources.size());
128         return compiler;
129       }
130     
131       protected Compiler runFullCompile(
132           CompilerOptions options,
133           List<String> sources,
134           int expectedCompileErrors,
135           int expectedCompileWarnings,
136           boolean flushResults) {
137         List<SourceFile> inputs = new ArrayList<>();
138         int i = 0;
139         for (String source : sources) {
140           inputs.add(SourceFile.fromCode("in" + i, source));
141           i++;
142         }
143         Compiler compiler = new Compiler();
144         Compiler.setLoggingLevel(Level.INFO);
145         Result result = compiler.compile(EXTERNS, inputs, options);
146         if (expectedCompileErrors == 0) {
147           assertTrue("Expected no errors, found: " + Arrays.toString(result.errors), result.success);
148         } else {
149           assertFalse(result.success);
150           assertEquals(expectedCompileErrors, compiler.getErrorCount());
151         }
152         assertEquals(expectedCompileWarnings, compiler.getWarningCount());
153         if (flushResults) {
154           flushResults(compiler);
155         }
156     
157         return compiler;
158       }
159     
160       protected void doReplaceScript(Compiler compiler, String newSource, int newSourceInd) {
161         SourceFile replacedSource = SourceFile.fromCode("in" + newSourceInd, newSource);
162         JsAst ast = new JsAst(replacedSource);
163         compiler.replaceScript(ast);
164       }
165     
166       protected void doAddScript(Compiler compiler, String newSource, int newSourceInd) {
167         SourceFile replacedSource = SourceFile.fromCode("in" + newSourceInd, newSource);
168         JsAst ast = new JsAst(replacedSource);
169         compiler.addNewScript(ast);
170       }
171     
172       protected void assertNoWarningsOrErrors(Result result) {
173         assertNumWarningsAndErrors(result, 0, 0);
174       }
175     
176       protected void assertNumWarningsAndErrors(Result result, int e, int w) {
177         assertEquals(
178             "Unexpected warnings:\n" + Joiner.on("\n").join(result.warnings),
179             w,
180             result.warnings.length);
181         assertEquals(
182             "Unexpected errors:\n" + Joiner.on("\n").join(result.errors), e, result.errors.length);
183         assertEquals(e == 0, result.success);
184       }
185     
186       protected void assertErrorType(JSError e, DiagnosticType type, int lineNumber) {
187         assertEquals(e.getType(), type);
188         assertEquals(e.lineNumber, lineNumber);
189       }
