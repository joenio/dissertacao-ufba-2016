33     /**
34      * Rewrites a CommonJS module http://wiki.commonjs.org/wiki/Modules/1.1.1
35      * into a form that can be safely concatenated.
36      * Does not add a function around the module body but instead adds suffixes
37      * to global variables to avoid conflicts.
38      * Calls to require are changed to reference the required module directly.
39      * goog.provide and goog.require are emitted for closure compiler automatic
40      * ordering.
41      */
42     public final class ProcessCommonJSModules implements CompilerPass {
43       private static final String EXPORTS = "exports";
44       private static final String MODULE = "module";
45     
46       private static final DiagnosticType LOAD_ERROR = DiagnosticType.error(
47           "JSC_COMMONJS_MODULE_LOAD_ERROR",
48           "Failed to load module \"{0}\"");
49     
50       private final Compiler compiler;
51       private final ES6ModuleLoader loader;
52       private final boolean reportDependencies;
53     
54       /**
55        * Creates a new ProcessCommonJSModules instance which can be used to

57        *
58        * @param compiler The compiler
59        * @param loader The module loader which is used to locate CommonJS modules
60        */
61       public ProcessCommonJSModules(Compiler compiler, ES6ModuleLoader loader) {
62         this(compiler, loader, true);

66        * Creates a new ProcessCommonJSModules instance which can be used to
67        * rewrite CommonJS modules to a concatenable form.

97        * Avoid processing if we find the appearance of goog.provide or goog.module.
98        *
99        * TODO(moz): Let ES6, CommonJS and goog.provide live happily together.
100        */
101       static class FindGoogProvideOrGoogModule extends AbstractPreOrderCallback {
102     
103         private boolean found;
104     
105         boolean isFound() {
106           return found;
107         }
108     
109         @Override
110         public boolean shouldTraverse(NodeTraversal nodeTraversal, Node n, Node parent) {
111           if (found) {
112             return false;
113           }
114           // Shallow traversal, since we don't need to inspect within functions or expressions.
115           if (parent == null
116               || NodeUtil.isControlStructure(parent)
117               || NodeUtil.isStatementBlock(parent)) {
118             if (n.isExprResult()) {
119               Node maybeGetProp = n.getFirstFirstChild();
120               if (maybeGetProp != null
121                   && (maybeGetProp.matchesQualifiedName("goog.provide")
122                       || maybeGetProp.matchesQualifiedName("goog.module"))) {
123                 found = true;
124                 return false;
125               }
126             }
127             return true;
128           }
129           return false;
130         }

181       /**
182        * Visits require, every "script" and special module.exports assignments.
183        */
184       private class ProcessCommonJsModulesCallback extends
185           AbstractPostOrderCallback {
186     
187         private int scriptNodeCount = 0;
188         private List<Node> moduleExportRefs = new ArrayList<>();
189         private List<Node> exportRefs = new ArrayList<>();
190         Multiset<String> propertyExportRefCount = HashMultiset.create();
191         private final boolean allowFullRewrite;
192     
193         public ProcessCommonJsModulesCallback(boolean allowFullRewrite) {
194           this.allowFullRewrite = allowFullRewrite;
195         }
196     
197         @Override
198         public void visit(NodeTraversal t, Node n, Node parent) {
199           if (n.isCall() && n.getChildCount() == 2 &&
200               n.getFirstChild().matchesQualifiedName("require") &&
201               n.getSecondChild().isString()) {
202             visitRequireCall(t, n, parent);
203           }
204     
205           // Finds calls to require.ensure which is a method to allow async loading of
206           // CommonJS modules:
207           //
208           // require.ensure(['/path/to/module1', '/path/to/module2'], function(require) {
209           //    var module1 = require('/path/to/module1');
210           //    var module2 = require('/path/to/module2');
211           // });
212           //
213           // will be rewritten as an IIFE
214           //
215           // (function() {
216           //   var module1 = require('/path/to/module1');
217           //   var module2 = require('/path/to/module2');
218           // })()
219           //
220           // See http://wiki.commonjs.org/wiki/Modules/Async/A
221           //     http://www.injectjs.com/docs/0.7.x/cjs/require.ensure.html
222           if (n.isCall()
223               && n.getChildCount() == 3
224               && n.getFirstChild().matchesQualifiedName("require.ensure")) {
225             visitRequireEnsureCall(n);
226           }
227     
228           // Detects UMD pattern, by checking for CommonJS exports and AMD define
229           // statements in if-conditions and rewrites the if-then-else block as
230           // follows to make sure the CommonJS exports can be reached:
231           // 1. When detecting a CommonJS exports statement, it removes the
232           // if-condition and the else-branch and adds the then-branch directly
233           // to the current parent node:
234           //
235           // if (typeof module == "object" && module.exports) {
236           //   module.exports = foobar;
237           // } else if (typeof define === "function" && define.amd) {...}
238           //
239           // will be rewritten to:
240           //
241           // module.exports = foobar;
242           //
243           // 2. When detecting an AMD define statement, it removes the if-condition and
244           // the then-branch and adds the else-branch directly to the current parent node:
245           //
246           // if (typeof define === "function" && define.amd) {
247           // ...} else if (typeof module == "object" && module.exports) {...}
248           //
249           // will be rewritten to:
250           //
251           // if (typeof module == "object" && module.exports) {...}
252           if (allowFullRewrite && n.isIf()) {
253             FindModuleExportStatements commonjsFinder = new FindModuleExportStatements();
254             Node condition = n.getFirstChild();
255             NodeTraversal.traverseEs6(compiler, condition, commonjsFinder);
256     
257             if (commonjsFinder.isFound()) {
258               visitCommonJSIfStatement(n);
259             } else {
260               FindDefineAmdStatements amdFinder = new FindDefineAmdStatements();
261               NodeTraversal.traverseEs6(compiler, condition, amdFinder);
262     
263               if (amdFinder.isFound()) {
264                 visitAMDIfStatement(n);
265               }
266             }
267           }
268     
269           if (n.isScript()) {
270             scriptNodeCount++;
271             visitScript(t, n);
272           }
273     
274           if (allowFullRewrite &&  n.isGetProp() &&
275               "module.exports".equals(n.getQualifiedName())) {
276             Var v = t.getScope().getVar(MODULE);
277             // only rewrite "module.exports" if "module" is a free variable,
278             // meaning it is not defined in the current scope as a local
279             // variable or function parameter
280             if (v == null) {
281               moduleExportRefs.add(n);
282               maybeAddReferenceCount(n);
283             }
284           }
285     
286           if (allowFullRewrite && n.isName() && EXPORTS.equals(n.getString())) {
287             Var v = t.getScope().getVar(n.getString());
288             if (v == null || v.isGlobal()) {
289               exportRefs.add(n);
290               maybeAddReferenceCount(n);
291             }
292           }
293         }

295         private Node getBaseQualifiedNameNode(Node n) {
296           Node refParent = n;
297           while (refParent.getParent() != null && refParent.getParent().isQualifiedName()) {
298             refParent = refParent.getParent();
299           }
300     
301           if (refParent == null || !refParent.getParent().isAssign()) {
302             return null;
303           }
304     
305           return refParent;
306         }
307     
308         private void maybeAddReferenceCount(Node n) {
309           Node refParent = getBaseQualifiedNameNode(n);
310     
311           if (refParent == null) {
312             return;
313           }
314     
315           String qName = refParent.getQualifiedName();
316           if (qName.startsWith("module.exports.")) {
317             qName = qName.substring("module.exports.".length());
318           } else if (qName.startsWith("exports.")) {
319             qName = qName.substring("exports.".length());
320           } else {
321             return;
322           }
323     
324           propertyExportRefCount.add(qName);
325         }

327         /**
328          * Visit require calls. Emit corresponding goog.require and rewrite require
329          * to be a direct reference to name of require module.
330          */
331         private void visitRequireCall(NodeTraversal t, Node require, Node parent) {
332           String requireName = require.getSecondChild().getString();
333           URI loadAddress = loader.locateCommonJsModule(requireName, t.getInput());
334           if (loadAddress == null) {
335             compiler.report(t.makeError(require, LOAD_ERROR, requireName));
336             return;
337           }
338     
339           String moduleName = ES6ModuleLoader.toModuleName(loadAddress);
340           Node script = getCurrentScriptNode(parent);
341     
342           // When require("name") is used as a standalone statement (the result isn't used)
343           // it indicates that a module is being loaded for the side effects it produces.
344           // In this case the require statement should just be removed as the goog.require
345           // call inserted will import the module.
346           if (!NodeUtil.isExpressionResultUsed(require)
347               && parent.isExprResult()
348               && NodeUtil.isStatementBlock(parent.getParent())) {
349             parent.getParent().removeChild(parent);
350           } else {
351             Node moduleRef = IR.name(moduleName).srcref(require);
352             parent.replaceChild(require, moduleRef);
353           }
354           if (reportDependencies) {
355             t.getInput().addRequire(moduleName);
356           }
357           // Rewrite require("name").
358           script.addChildToFront(IR.exprResult(
359               IR.call(IR.getprop(IR.name("goog"), IR.string("require")),
360                   IR.string(moduleName))).useSourceInfoIfMissingFromForTree(require));
361           compiler.reportCodeChange();
362         }

364         /**
365          * Visit require.ensure calls. Replace the call with an IIFE.
366          */
367         private void visitRequireEnsureCall(Node n) {
368           Preconditions.checkState(n.getChildCount() == 3);
369           Node callbackFunction = n.getChildAtIndex(2);
370     
371           // We only support the form where the first argument is an array literal and
372           // the the second a callback function which has a single argument
373           // with the name "require".
374           if (!(n.getSecondChild().isArrayLit()
375               && callbackFunction.isFunction()
376               && callbackFunction.getChildCount() == 3
377               && callbackFunction.getSecondChild().getChildCount() == 1
378               && callbackFunction.getSecondChild().getFirstChild().matchesQualifiedName("require"))) {
379             return;
380           }
381     
382           callbackFunction.detachFromParent();
383     
384           // Remove the "require" argument from the parameter list.
385           callbackFunction.getSecondChild().removeChildren();
386           n.removeChildren();
387           n.putBooleanProp(Node.FREE_CALL, true);
388           n.addChildrenToFront(callbackFunction);
389     
390           compiler.reportCodeChange();
391         }

393         /**
394          * Emit goog.provide and add suffix to all global vars to avoid conflicts
395          * with other modules.
396          */
397         private void visitScript(NodeTraversal t, Node script) {
398           Preconditions.checkArgument(scriptNodeCount == 1,
399               "ProcessCommonJSModules supports only one invocation per " +
400               "CompilerInput / script node");
401     
402           String moduleName = inputToModuleName(t.getInput());
403     
404           boolean hasExports = !(moduleExportRefs.isEmpty() && exportRefs.isEmpty());
405     
406           // Rename vars to not conflict in global scope - but only if the script exports something.
407           // If there are no exports, we still need to rewrite type annotations which
408           // are module paths
409           NodeTraversal.traverseEs6(compiler, script,
410               new SuffixVarsCallback(moduleName, hasExports));
411     
412           // If the script has no exports, we don't want to output a goog.provide statement
413           if (!hasExports) {
414             return;
415           }
416     
417           // Replace all refs to module.exports and exports
418           processExports(script, moduleName);
419           moduleExportRefs.clear();
420           exportRefs.clear();
421     
422           // Add goog.provide calls.
423           if (reportDependencies) {
424             CompilerInput ci = t.getInput();
425             ci.addProvide(moduleName);
426           }
427           script.addChildToFront(IR.exprResult(
428               IR.call(IR.getprop(IR.name("goog"), IR.string("provide")),
429                   IR.string(moduleName))).useSourceInfoIfMissingFromForTree(script));
430     
431           compiler.reportCodeChange();
432         }

434         /**
435          * Rewrites CommonJS part of UMD pattern by removing the if-condition and the
436          * else-branch and adds the then-branch directly to the current parent node.
437          */
438         private void visitCommonJSIfStatement(Node n) {
439           Node p = n.getParent();
440           if (p != null) {
441             // pull out then-branch
442             replaceIfStatementWithBranch(n, n.getSecondChild());
443           }
444         }
445     
446         /**
447          * Rewrites AMD part of UMD pattern by removing the if-condition and the
448          * then-branch and adds the else-branch directly to the current parent node.
449          */
450         private void visitAMDIfStatement(Node n) {
451           Node p = n.getParent();
452           if (p != null) {
453             if (n.getChildCount() == 3) {
454               // pull out else-branch
455               replaceIfStatementWithBranch(n, n.getChildAtIndex(2));
456             } else {
457               // remove entire if-statement if it doesn't have an else-branch
458               p.removeChild(n);
459             }
460           }
461         }
462     
463         private void replaceIfStatementWithBranch(Node ifStatement, Node branch) {
464           Node p = ifStatement.getParent();
465           Node newNode = branch;
466           // Remove redundant block node. Not strictly necessary, but makes tests more legible.
467           if (branch.isBlock() && branch.getChildCount() == 1) {
468             newNode = branch.getFirstChild();
469             branch.detachChildren();
470           } else {
471             ifStatement.detachChildren();
472           }
473           p.replaceChild(ifStatement, newNode);
474         }

476         /**
477          * Process all references to module.exports and exports.
478          *
479          * In CommonJS systems, module.exports and exports point to
480          * the same object, unless one of them is re-assigned.
481          *
482          * We handle 2 special forms:
483          * 1) Exactly 1 top-level assign to module.exports.
484          *    module.exports = ...;
485          * 2) Direct reads of exports and module.exports.
486          *    This includes assignments to properties of exports,
487          *    because these only read the slot itself.
488          *    module.exports.prop = ...; // 1 or more times.
489          *
490          * We do this so that these forms type-check better.
491          *
492          * All other forms are handled by a more general algorithm.
493          */
494         private void processExports(Node script, String moduleName) {
495           if (hasOneTopLevelModuleExportAssign()) {
496             // One top-level assign: transform to
497             // moduleName = rhs
498             Node ref = moduleExportRefs.get(0);
499             Node newName = IR.name(moduleName).srcref(ref);
500             newName.setOriginalName(ref.getQualifiedName());
501             Node rhsValue = ref.getNext();
502     
503             if (rhsValue.isObjectLit()) {
504               addConstToObjLitKeys(rhsValue);
505             }
506     
507             Node assign = ref.getParent();
508             assign.replaceChild(ref, newName);
509             JSDocInfoBuilder builder = new JSDocInfoBuilder(true);
510             builder.recordConstancy();
511             JSDocInfo info = builder.build();
512             assign.setJSDocInfo(info);
513             return;
514           }
515     
516           Iterable<Node> exports;
517           boolean hasLValues = hasExportLValues();
518           if (hasLValues) {
519             exports = moduleExportRefs;
520           } else {
521             exports = Iterables.concat(moduleExportRefs, exportRefs);
522           }
523     
524           // Transform to:
525           //
526           // moduleName.prop0 = 0; // etc.
527           boolean declaredModuleExports = false;
528           for (Node ref : exports) {
529             // If there is a module exports assignment at this point, we need to
530             // add a variable declaration for the module name, because otherwise
531             // the default declaration for the goog.provide is a constant and the
532             // assignment would violate that constant.
533             // Note that the hasOneTopLevelModuleExportAssign() case handles the
534             // more common case of assigning to module.exports on the top level,
535             // but CommonJS code also sometimes assigns to module.exports inside
536             // of more complex expressions.
537             if (ref.getParent().isAssign()
538                 && !ref.getGrandparent().isExprResult()
539                 && !declaredModuleExports) {
540               // Adds "var moduleName" to front of the current file.
541               script.addChildToFront(
542                   IR.var(IR.name(moduleName))
543                       .useSourceInfoIfMissingFromForTree(ref));
544               declaredModuleExports = true;
545             }
546     
547             String qName = null;
548             if (ref.isQualifiedName()) {
549               Node baseName = getBaseQualifiedNameNode(ref);
550               if (baseName != null) {
551                 qName = baseName.getQualifiedName();
552                 if (qName.startsWith("module.exports.")) {
553                   qName = qName.substring("module.exports.".length());
554                 } else {
555                   qName = qName.substring("exports.".length());
556                 }
557               }
558             }
559     
560             Node rhsValue = ref.getNext();
561             Node newName = IR.name(moduleName).srcref(ref);
562             newName.setOriginalName(qName);
563     
564             Node parent = ref.getParent();
565             parent.replaceChild(ref, newName);
566     
567             // If the property was assigned to exactly once, add an @const annotation
568             if (parent.isAssign() && qName != null &&  propertyExportRefCount.count(qName) == 1) {
569               if (rhsValue != null && rhsValue.isObjectLit()) {
570                 addConstToObjLitKeys(rhsValue);
571               }
572     
573               JSDocInfoBuilder builder = new JSDocInfoBuilder(true);
574               builder.recordConstancy();
575               JSDocInfo info = builder.build();
576               parent.setJSDocInfo(info);
577             }
578           }
579     
580           if(!hasLValues) {
581             return;
582           }
583     
584           // Transform exports to exports$$moduleName and set to point
585           // to module namespace: exports$$moduleName = moduleName;
586           if (!exportRefs.isEmpty()) {
587             String aliasName = "exports$$" + moduleName;
588             Node aliasNode = IR.var(IR.name(aliasName), IR.name(moduleName))
589                 .useSourceInfoIfMissingFromForTree(script);
590             script.addChildToFront(aliasNode);
591     
592             for (Node ref : exportRefs) {
593               ref.setOriginalName(ref.getString());
594               ref.setString(aliasName);
595             }
596           }
597         }
598     
599         /**
600          * Add an @const annotation to each key of an object literal
601          */
602         private void addConstToObjLitKeys(Node n) {
603           Preconditions.checkState(n.isObjectLit());
604           for (Node key = n.getFirstChild();
605                key != null; key = key.getNext()) {
606             if (key.getJSDocInfo() == null) {
607               JSDocInfoBuilder builder = new JSDocInfoBuilder(true);
608               builder.recordConstancy();
609               JSDocInfo info = builder.build();
610               key.setJSDocInfo(info);
611             }
612           }
613         }

615         /**
616          * Recognize export pattern [1] (see above).
617          */
618         private boolean hasOneTopLevelModuleExportAssign() {
619           return moduleExportRefs.size() == 1 &&
620               exportRefs.isEmpty() &&
621               isTopLevelAssignLhs(moduleExportRefs.get(0));
622         }
623     
624         private boolean isTopLevelAssignLhs(Node n) {
625           Node parent = n.getParent();
626           return parent.isAssign() && n == parent.getFirstChild() &&
627               parent.getParent().isExprResult() &&
628               parent.getGrandparent().isScript();
629         }
630     
631         /**
632          * Recognize the opposite of export pattern [2] (see above).
633          */
634         private boolean hasExportLValues() {
635           for (Node ref : Iterables.concat(moduleExportRefs, exportRefs)) {
636             if (NodeUtil.isLValue(ref)) {
637               return true;
638             }
639           }
640           return false;
641         }

643         /**
644          * Returns next script node in parents.
645          */
646         private Node getCurrentScriptNode(Node n) {
647           while (true) {
648             if (n.isScript()) {
649               return n;
650             }
651             n = n.getParent();
652           }
653         }

656       /**
657        * Traverses a node tree and appends a suffix to all global variable names.
658        */
659       private class SuffixVarsCallback extends AbstractPostOrderCallback {
660         private final String suffix;
661         private final boolean fullRewrite;
662     
663         SuffixVarsCallback(String suffix, boolean fullRewrite) {
664           this.suffix = suffix;
665           this.fullRewrite = fullRewrite;
666         }
667     
668         @Override
669         public void visit(NodeTraversal t, Node n, Node parent) {
670           JSDocInfo info = n.getJSDocInfo();
671           if (info != null) {
672             for (Node typeNode : info.getTypeNodes()) {
673               fixTypeNode(t, typeNode);
674             }
675           }
676     
677           boolean isShorthandObjLitKey = n.isStringKey() && !n.hasChildren();
678           if (fullRewrite && n.isName() || isShorthandObjLitKey) {
679             String name = n.getString();
680             if (suffix.equals(name)) {
681               // TODO(moz): Investigate whether we need to return early in this unlikely situation.
682               return;
683             }
684     
685             // refs to 'exports' are handled separately.
686             if (EXPORTS.equals(name)) {
687               return;
688             }
689     
690             // closure_test_suite looks for test*() functions
691             if (compiler.getOptions().exportTestFunctions && name.startsWith("test")) {
692               return;
693             }
694     
695             Var var = t.getScope().getVar(name);
696             if (var != null && var.isGlobal()) {
697               String newName = name + "$$" + suffix;
698               if (isShorthandObjLitKey) {
699                 // Change {a} to {a: a$$module$foo}
700                 n.addChildToBack(IR.name(newName).useSourceInfoIfMissingFrom(n));
701               } else {
702                 n.setString(newName);
703                 n.setOriginalName(name);
704               }
705             }
706           }
707         }
708     
709         /**
710          * Replace type name references.
711          */
712         private void fixTypeNode(NodeTraversal t, Node typeNode) {
713           if (typeNode.isString()) {
714             String name = typeNode.getString();
715             if (ES6ModuleLoader.isRelativeIdentifier(name)
716                 || ES6ModuleLoader.isAbsoluteIdentifier(name)) {
717               int lastSlash = name.lastIndexOf('/');
718               int endIndex = name.indexOf('.', lastSlash);
719               String localTypeName = null;
720               if (endIndex == -1) {
721                 endIndex = name.length();
722               } else {
723                 localTypeName = name.substring(endIndex);
724               }
725     
726               String moduleName = name.substring(0, endIndex);
727               URI loadAddress = loader.locateCommonJsModule(moduleName, t.getInput());
728               if (loadAddress == null) {
729                 t.makeError(typeNode, LOAD_ERROR, moduleName);
730                 return;
731               }
732     
733               String globalModuleName = ES6ModuleLoader.toModuleName(loadAddress);
734               typeNode.setString(
735                   localTypeName == null ? globalModuleName : globalModuleName + localTypeName);
736             } else if (fullRewrite) {
737               int endIndex = name.indexOf('.');
738               if (endIndex == -1) {
739                 endIndex = name.length();
740               }
741               String baseName = name.substring(0, endIndex);
742               Var var = t.getScope().getVar(baseName);
743               if (var != null && var.isGlobal()) {
744                 typeNode.setString(baseName + "$$" + suffix + name.substring(endIndex));
745                 typeNode.setOriginalName(name);
746               }
747             }
748           }
749     
750           for (Node child = typeNode.getFirstChild(); child != null;
751                child = child.getNext()) {
752             fixTypeNode(t, child);
753           }
754         }
755       }
