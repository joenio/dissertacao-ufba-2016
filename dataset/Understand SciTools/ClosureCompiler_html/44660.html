52     /**
53      * Tests the conversion of closure-style type declarations in JSDoc
54      * to inline type declarations, by running both syntaxes through the parser
55      * and verifying the resulting AST is the same.
56      */
57     public final class JsdocToEs6TypedConverterTest extends CompilerTestCase {
58     
59       @Override
60       public void setUp() {
61         setAcceptedLanguage(LanguageMode.ECMASCRIPT6_TYPED);
62         compareJsDoc = false;
63       }
64     
65       @Override
66       protected CompilerOptions getOptions() {
67         CompilerOptions options = super.getOptions();
68         options.setLanguageOut(LanguageMode.ECMASCRIPT6_TYPED);
69         return options;
70       }

73       public CompilerPass getProcessor(Compiler compiler) {
74         return new JsdocToEs6TypedConverter(compiler);

99       public void testFunctionInsideAssignment() throws Exception {
100         test("/** @param {boolean} b @return {boolean} */ "
101                 + "var f = function(b){return !b};",
102             "var f = function(b: boolean): boolean { return !b; };");

109                 + "  var t = function(l) {"
110                 + "  };"

112                 + "};");
113       }

125       public void testConvertSimpleTypes() {
126         assertParseTypeAndConvert("?").hasType(ANY_TYPE);
127         assertParseTypeAndConvert("*").hasType(ANY_TYPE);
128         assertParseTypeAndConvert("boolean").hasType(BOOLEAN_TYPE);
129         assertParseTypeAndConvert("number").hasType(NUMBER_TYPE);
130         assertParseTypeAndConvert("string").hasType(STRING_TYPE);
131       }
132     
133       public void testConvertNamedTypes() throws Exception {
134         assertParseTypeAndConvert("Window")
135             .isEqualTo(namedType("Window"));
136         assertParseTypeAndConvert("goog.ui.Menu")
137             .isEqualTo(namedType("goog.ui.Menu"));
138     
139         assertNode(namedType("goog.ui.Menu"))
140             .isEqualTo(new TypeDeclarationNode(NAMED_TYPE,
141                 IR.getprop(IR.getprop(IR.name("goog"), IR.string("ui")), IR.string("Menu"))));
142       }
143     
144       public void testConvertTypeApplication() throws Exception {
145         assertParseTypeAndConvert("Array.<string>")
146             .isEqualTo(arrayType(stringType()));
147         assertParseTypeAndConvert("Object.<string, number>")
148             .isEqualTo(parameterizedType(namedType("Object"), asList(stringType(), numberType())));
149     
150         assertNode(parameterizedType(namedType("Array"), asList(stringType())))
151             .isEqualTo(new TypeDeclarationNode(PARAMETERIZED_TYPE,
152                 new TypeDeclarationNode(NAMED_TYPE, IR.name("Array")),
153                 new TypeDeclarationNode(STRING_TYPE)));
154       }
155     
156       public void testConvertTypeUnion() throws Exception {
157         assertParseTypeAndConvert("(number|boolean)")
158             .isEqualTo(unionType(numberType(), booleanType()));
159       }

161       public void testConvertRecordType() throws Exception {
162         LinkedHashMap<String, TypeDeclarationNode> properties = new LinkedHashMap<>();
163         properties.put("myNum", numberType());
164         properties.put("myObject", null);
165     
166         assertParseTypeAndConvert("{myNum: number, myObject}")
167             .isEqualTo(recordType(properties));
168       }
169     
170       public void testCreateRecordType() throws Exception {
171         LinkedHashMap<String, TypeDeclarationNode> properties = new LinkedHashMap<>();
172         properties.put("myNum", numberType());
173         properties.put("myObject", null);
174         TypeDeclarationNode node = recordType(properties);
175     
176         Node prop1 = IR.stringKey("myNum");
177         prop1.addChildToFront(new TypeDeclarationNode(NUMBER_TYPE));
178         Node prop2 = IR.stringKey("myObject");
179     
180         assertNode(node)
181             .isEqualTo(new TypeDeclarationNode(RECORD_TYPE, prop1, prop2));
182       }

184       public void testConvertRecordTypeWithTypeApplication() throws Exception {
185         Node prop1 = IR.stringKey("length");
186         assertParseTypeAndConvert("Array.<{length}>")
187             .isEqualTo(new TypeDeclarationNode(ARRAY_TYPE,
188                 new TypeDeclarationNode(RECORD_TYPE, prop1)));
189       }

202       public void testConvertFunctionType() throws Exception {
203         Node p1 = IR.name("p1");
204         p1.setDeclaredTypeExpression(stringType());
205         Node p2 = IR.name("p2");
206         p2.setDeclaredTypeExpression(booleanType());
207         assertParseTypeAndConvert("function(string, boolean)")
208             .isEqualTo(new TypeDeclarationNode(FUNCTION_TYPE, anyType(), p1, p2));
209       }

230       public void testConvertVariableParameters() throws Exception {
231         Node p1 = IR.name("p1");
232         p1.setDeclaredTypeExpression(stringType());
233         Node p2 = IR.rest("p2");
234         p2.setDeclaredTypeExpression(arrayType(numberType()));
235         assertParseTypeAndConvert("function(string, ...number): number")
236             .isEqualTo(new TypeDeclarationNode(FUNCTION_TYPE, numberType(), p1, p2));
237       }
238     
239       public void testConvertOptionalFunctionParameters() throws Exception {
240         LinkedHashMap<String, TypeDeclarationNode> requiredParams = new LinkedHashMap<>();
241         LinkedHashMap<String, TypeDeclarationNode> optionalParams = new LinkedHashMap<>();
242         optionalParams.put("p1", stringType());
243         optionalParams.put("p2", numberType());
244         assertParseTypeAndConvert("function(?string=, number=)")
245             .isEqualTo(functionType(anyType(), requiredParams, optionalParams, null, null));
246       }
247     
248       private NodeSubject assertParseTypeAndConvert(final String typeExpr) {
249         Node oldAST = JsDocInfoParser.parseTypeString(typeExpr);
250         assertNotNull(typeExpr + " did not produce a parsed AST", oldAST);
251         return new NodeSubject(THROW_ASSERTION_ERROR,
252             TypeDeclarationsIRFactory.convertTypeNodeAST(oldAST));
253       }
