40     /**
41      * For functions with function(this: T, ...) and T as arguments, type inference
42      * will set the type of this on a function literal argument to the actual type
43      * of T.
44      *
45      */
46     package com.google.javascript.rhino.jstype;
47     
48     import com.google.javascript.rhino.Node;
49     
50     public class TemplateType extends ProxyObjectType {
51       private static final long serialVersionUID = 1L;
52     
53       private final String name;
54       private final Node typeTransformation;
55     
56       TemplateType(JSTypeRegistry registry, String name) {
57         this(registry, name, null);
58       }
59     
60       TemplateType(JSTypeRegistry registry, String name, Node typeTransformation) {
61         super(registry, registry.getNativeObjectType(JSTypeNative.UNKNOWN_TYPE));
62         this.name = name;
63         this.typeTransformation = typeTransformation;
64       }
65     
66       @Override
67       public String getReferenceName() {
68         return name;
69       }
70     
71       @Override
72       String toStringHelper(boolean forAnnotations) {
73         return name;
74       }
75     
76       @Override
77       public TemplateType toMaybeTemplateType() {
78         return this;
79       }
80     
81       @Override
82       public boolean hasAnyTemplateTypesInternal() {
83         return true;
84       }
85     
86       @Override
87       public <T> T visit(Visitor<T> visitor) {
88         return visitor.caseTemplateType(this);
89       }
90     
91       @Override <T> T visit(RelationshipVisitor<T> visitor, JSType that) {
92         return visitor.caseTemplateType(this, that);
93       }
94     
95       public boolean isTypeTransformation() {
96         return typeTransformation != null;
97       }
98     
99       public Node getTypeTransformation() {
100         return typeTransformation;
101       }
102     }
