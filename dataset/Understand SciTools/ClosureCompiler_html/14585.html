28       @Override
29       protected CompilerOptions getOptions() {
30         CompilerOptions options = super.getOptions();
31         options.setLanguageIn(LanguageMode.ECMASCRIPT6_TYPED);
32         options.setLanguageOut(LanguageMode.ECMASCRIPT6);
33         return options;

36       @Override
37       protected CompilerPass getProcessor(Compiler compiler) {
38         PhaseOptimizer optimizer = new PhaseOptimizer(compiler, null, null);
39         optimizer.addOneTimePass(new PassFactory("convertDeclaredTypesToJSDoc", true) {
40           // To make sure types copied.
41           @Override protected CompilerPass create(AbstractCompiler compiler) {
42             return new Es6TypedToEs6Converter(compiler);
43           }
44         });
45         return optimizer;
46       }
