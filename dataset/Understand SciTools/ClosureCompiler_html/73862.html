40     package com.google.javascript.rhino.jstype;
41     
42     import com.google.javascript.rhino.ErrorReporter;
43     import com.google.javascript.rhino.JSDocInfo;
44     import com.google.javascript.rhino.Node;
45     
46     /**
47      * The bottom Object type, representing the subclass of all objects.
48      *
49      * Although JavaScript programmers can't explicitly denote the bottom
50      * Object type, it comes up in static analysis. For example, if we have:
51      * <code>
52      * var x = function() {};
53      * if (x instanceof Array) {
54      *   f(x);
55      * }
56      * </code>
57      * We need to be able to assign {@code x} a type within the {@code f(x)}
58      * call. It has no possible type, but {@code x} would not be legal if f
59      * expected a string. So we assign it the {@code NoObjectType}.
60      *
61      * @see <a href="http://en.wikipedia.org/wiki/Bottom_type">Bottom types</a>
62      */
63     public class NoObjectType extends FunctionType {
64       private static final long serialVersionUID = 1L;
65     
66       NoObjectType(JSTypeRegistry registry) {
67         super(registry, null, null,
68               registry.createArrowType(null, null),
69               null, null, true, true);
70         getInternalArrowType().returnType = this;
71         this.setInstanceType(this);
72       }
73     
74       @Override
75       public boolean isSubtype(JSType that) {
76         return isSubtype(that, ImplCache.create(), SubtypingMode.NORMAL);
77       }
78     
79       @Override
80       protected boolean isSubtype(JSType that,
81           ImplCache implicitImplCache, SubtypingMode subtypingMode) {
82         if (JSType.isSubtypeHelper(this, that, implicitImplCache, subtypingMode)) {
83           return true;
84         } else {
85           return that.isObject() && !that.isNoType() && !that.isNoResolvedType();
86         }
87       }
88     
89       @Override
90       public FunctionType toMaybeFunctionType() {
91         return null;
92       }
93     
94       @Override
95       public boolean isNoObjectType() {
96         return true;
97       }
98     
99       @Override
100       public ObjectType getImplicitPrototype() {
101         return null;
102       }
103     
104       @Override
105       public String getReferenceName() {
106         return null;
107       }
108     
109       @Override
110       public boolean hasReferenceName() {
111         return false;
112       }
113     
114       @Override
115       public boolean matchesNumberContext() {
116         return true;
117       }
118     
119       @Override
120       public boolean matchesObjectContext() {
121         return true;
122       }
123     
124       @Override
125       public boolean matchesStringContext() {
126         return true;
127       }
128     
129       @Override
130       public int hashCode() {
131         return System.identityHashCode(this);
132       }
133     
134       @Override
135       boolean defineProperty(String propertyName, JSType type,
136           boolean inferred, Node propertyNode) {
137         // nothing, all properties are defined
138         return true;
139       }
140     
141       @Override
142       public boolean removeProperty(String name) {
143         return false;
144       }
145     
146       @Override
147       public void setPropertyJSDocInfo(String propertyName, JSDocInfo info) {
148         // Do nothing, specific properties do not have JSDocInfo.
149       }
150     
151       @Override
152       public <T> T visit(Visitor<T> visitor) {
153         return visitor.caseNoObjectType();
154       }
155     
156       @Override <T> T visit(RelationshipVisitor<T> visitor, JSType that) {
157         return visitor.caseNoObjectType(that);
158       }
159     
160       @Override
161       String toStringHelper(boolean forAnnotations) {
162         return forAnnotations ? "?" : "NoObject";
163       }
164     
165       @Override
166       public FunctionType getConstructor() {
167         return null;
168       }
169     
170       @Override
171       JSType resolveInternal(ErrorReporter t, StaticTypedScope<JSType> scope) {
172         return this;
173       }
174     }
