50     /**
51      * DisambiguateProperties renames properties to disambiguate between unrelated
52      * fields with the same name. Two properties are considered related if they
53      * share a definition on their prototype chains, or if they are potentially
54      * referenced together via union types.
55      *
56      * <p> Renamimg only occurs if there are two or more distinct properties with
57      * the same name.
58      *
59      * <p> This pass allows other passes, such as inlining and code removal to take
60      * advantage of type information implicitly.
61      *
62      * <pre>
63      *   Foo.a;
64      *   Bar.a;
65      * </pre>
66      *
67      * <p> will become
68      *
69      * <pre>
70      *   Foo.a$Foo;
71      *   Bar.a$Bar;
72      * </pre>
73      *
74      * NOTE(dimvar): For every property, this pass groups together the types that
75      * can't be disambiguated. If a type inherits from another type, their common
76      * properties can never be disambiguated, yet we have to compute this info once
77      * per property rather than just once in the pass. This is where the bulk of the
78      * time is spent.
79      * We have added many caches that help a lot, but it is probably worth it to
80      * revisit this pass and rewrite it in a way that does not compute the same
81      * thing over and over.
82      *
83      */
84     class DisambiguateProperties implements CompilerPass {

86       // that we tell the user to fix per-property.
87       private static final int MAX_INVALIDATION_WARNINGS_PER_PROPERTY = 10;

141       private ObjectType gtwpCacheGet(String field, JSType type) {
142         IdentityHashMap<JSType, ObjectType> m = gtwpCache.get(field);
143         return m == null ? null : m.get(type);
144       }

146       private void gtwpCachePut(String field, JSType type, ObjectType top) {
147         IdentityHashMap<JSType, ObjectType> m = gtwpCache.get(field);
148         if (m == null) {
149           m = new IdentityHashMap<>();
150           gtwpCache.put(field, m);
151         }
152         Preconditions.checkState(null == m.put(type, top));
153       }

155       private class Property {
156         /** The name of the property. */
157         final String name;
158     
159         /**
160          * All top types on which the field exists, grouped together if related.
161          * See getTypeWithProperty. If a property exists on a parent class and a
162          * subclass, only the parent class is recorded here.
163          */
164         private UnionFind<JSType> types;
165     
166         /**
167          * A set of types for which renaming this field should be skipped. This
168          * list is first filled by fields defined in the externs file.
169          */
170         Set<JSType> typesToSkip = new HashSet<>();
171     
172         /**
173          * If true, do not rename any instance of this field, as it has been
174          * referenced from an unknown type.
175          */
176         boolean skipRenaming;
177     
178         /**
179          * A map from nodes that need renaming to the highest type in the prototype
180          * chain containing the field for each node. In the case of a union, the
181          * type is the highest type of one of the types in the union.
182          */
183         Map<Node, JSType> rootTypesByNode = new HashMap<>();
184     
185         /**
186          * For every property p and type t, we only need to run recordInterfaces
187          * once. Use this cache to avoid needless calls.
188          */
189         private final Set<JSType> recordInterfacesCache = new HashSet<>();
190     
191         Property(String name) {
192           this.name = name;
193         }
194     
195         /** Returns the types on which this field is referenced. */
196         UnionFind<JSType> getTypes() {
197           if (types == null) {
198             types = new StandardUnionFind<>();
199           }
200           return types;
201         }
202     
203         /**
204          * Record that this property is referenced from this type.
205          */
206         void addType(JSType type, JSType relatedType) {
207           checkState(!skipRenaming, "Attempt to record skipped property: %s", name);
208           JSType top = getTypeWithProperty(this.name, type);
209           if (isInvalidatingType(top)) {
210             invalidate();
211             return;
212           }
213           if (isTypeToSkip(top)) {
214             addTypeToSkip(top);
215           }
216           if (relatedType == null) {
217             getTypes().add(top);
218           } else {
219             getTypes().union(top, relatedType);
220           }
221           FunctionType constructor = getConstructor(type);
222           if (constructor != null && recordInterfacesCache.add(type)) {
223             recordInterfaces(constructor, top, this);
224           }
225         }
226     
227         /** Records the given type as one to skip for this property. */
228         void addTypeToSkip(JSType type) {
229           for (JSType skipType : getTypesToSkipForType(type)) {
230             typesToSkip.add(skipType);
231             getTypes().union(skipType, type);
232           }
233         }
234     
235         /** Invalidates any types related to invalid types. */
236         void expandTypesToSkip() {
237           // If we are not going to rename any properties, then we do not need to
238           // update the list of invalid types, as they are all invalid.
239           if (shouldRename()) {
240             int count = 0;
241             while (true) {
242               // It should usually only take one time through this do-while.
243               checkState(++count < 10, "Stuck in loop expanding types to skip.");
244     
245               // Make sure that the representative type for each type to skip is
246               // marked as being skipped.
247               Set<JSType> rootTypesToSkip = new HashSet<>();
248               for (JSType subType : typesToSkip) {
249                 rootTypesToSkip.add(types.find(subType));
250               }
251               typesToSkip.addAll(rootTypesToSkip);
252     
253               Set<JSType> newTypesToSkip = new HashSet<>();
254               Set<JSType> allTypes = types.elements();
255               int originalTypesSize = allTypes.size();
256               for (JSType subType : allTypes) {
257                 if (!typesToSkip.contains(subType)
258                     && typesToSkip.contains(types.find(subType))) {
259                   newTypesToSkip.add(subType);
260                 }
261               }
262     
263               for (JSType newType : newTypesToSkip) {
264                 addTypeToSkip(newType);
265               }
266     
267               // If there were not any new types added, we are done here.
268               if (types.elements().size() == originalTypesSize) {
269                 break;
270               }
271             }
272           }
273         }
274     
275         /** Returns true if any instance of this property should be renamed. */
276         boolean shouldRename() {
277           return !skipRenaming && types != null
278               && types.allEquivalenceClasses().size() > 1;
279         }
280     
281         /**
282          * Returns true if this property should be renamed on this type.
283          * expandTypesToSkip() should be called before this, if anything has been
284          * added to the typesToSkip list.
285          */
286         boolean shouldRename(JSType type) {
287           return !skipRenaming && !typesToSkip.contains(type);
288         }
289     
290         /**
291          * Invalidates a field from renaming.  Used for field references on an
292          * object with unknown type.
293          */
294         boolean invalidate() {
295           boolean changed = !skipRenaming;
296           skipRenaming = true;
297           types = null;
298           typesToSkip = null;
299           rootTypesByNode = null;
300           return changed;
301         }
302     
303         /**
304          * Schedule the node to potentially be renamed.
305          * @param node the node to rename
306          * @param type the highest type in the prototype chain for which the
307          *     property is defined
308          * @return True if type was accepted without invalidation or if the property
309          *     was already invalidated.  False if this property was invalidated this
310          *     time.
311          */
312         boolean scheduleRenaming(Node node, JSType type) {
313           if (!skipRenaming) {
314             if (isInvalidatingType(type)) {
315               invalidate();
316               return false;
317             }
318             rootTypesByNode.put(node, type);
319           }
320           return true;
321         }
322       }

326       DisambiguateProperties(
327           AbstractCompiler compiler, Map<String, CheckLevel> propertiesToErrorFor) {
328         this.compiler = compiler;
329         this.registry = compiler.getTypeRegistry();
330         this.BOTTOM_OBJECT =
331             this.registry.getNativeType(JSTypeNative.NO_OBJECT_TYPE).toObjectType();
332         this.invalidatingTypes = new HashSet<>(ImmutableSet.of(
333             registry.getNativeType(JSTypeNative.ALL_TYPE),
334             registry.getNativeType(JSTypeNative.NO_OBJECT_TYPE),
335             registry.getNativeType(JSTypeNative.NO_TYPE),
336             registry.getNativeType(JSTypeNative.FUNCTION_PROTOTYPE),
337             registry.getNativeType(JSTypeNative.FUNCTION_INSTANCE_TYPE),
338             registry.getNativeType(JSTypeNative.OBJECT_PROTOTYPE),
339             registry.getNativeType(JSTypeNative.TOP_LEVEL_PROTOTYPE),
340             registry.getNativeType(JSTypeNative.UNKNOWN_TYPE)));
341         this.propertiesToErrorFor = propertiesToErrorFor;
342         if (!this.propertiesToErrorFor.isEmpty()) {
343           this.invalidationMap = LinkedHashMultimap.create();
344         } else {
345           this.invalidationMap = null;
346         }
347       }

349       @Override
350       public void process(Node externs, Node root) {
351         Preconditions.checkState(
352             compiler.getLifeCycleStage() == LifeCycleStage.NORMALIZED);
353         this.implementedInterfaces = new HashMap<>();
354         this.gtwpCache = new HashMap<>();
355         // TypeValidator records places where a type A is used in a context that
356         // expects a type B.
357         // For each pair (A, B), here we mark both A and B as types whose properties
358         // cannot be renamed.
359         for (TypeMismatch mis : compiler.getTypeMismatches()) {
360           recordInvalidatingType(mis.typeA, mis.src);
361           recordInvalidatingType(mis.typeB, mis.src);
362         }
363         for (TypeMismatch mis : compiler.getImplicitInterfaceUses()) {
364           recordInvalidatingType(mis.typeA, mis.src);
365           recordInvalidatingType(mis.typeB, mis.src);
366         }
367         // Gather names of properties in externs; these properties can't be renamed.
368         NodeTraversal.traverseEs6(compiler, externs, new FindExternProperties());
369         // Look at each unquoted property access and decide if that property will
370         // be renamed.
371         NodeTraversal.traverseEs6(compiler, root, new FindRenameableProperties());
372         // Do the actual renaming.
373         renameProperties();
374       }

376       private void recordInvalidationError(JSType t, JSError error) {
377         if (!t.isObject()) {
378           return;
379         }
380         if (invalidationMap != null) {
381           Collection<JSError> errors = this.invalidationMap.get(t);
382           if (errors.size() < MAX_INVALIDATION_WARNINGS_PER_PROPERTY) {
383             errors.add(error);
384           }
385         }
386       }

388       /**
389        * Invalidates the given type, so that no properties on it will be renamed.
390        */
391       private void recordInvalidatingType(JSType type, JSError error) {
392         type = type.restrictByNotNullOrUndefined();
393         if (type.isUnionType()) {
394           for (JSType alt : type.toMaybeUnionType().getAlternatesWithoutStructuralTyping()) {
395             recordInvalidatingType(alt, error);
396           }
397         } else if (type.isEnumElementType()) {
398           recordInvalidatingType(
399               type.toMaybeEnumElementType().getPrimitiveType(), error);
400         } else {
401           addInvalidatingType(type);
402           recordInvalidationError(type, error);
403           ObjectType objType = ObjectType.cast(type);
404           if (objType != null && objType.getImplicitPrototype() != null) {
405             addInvalidatingType(objType.getImplicitPrototype());
406             recordInvalidationError(objType.getImplicitPrototype(), error);
407           }
408           if (objType != null
409               && objType.isConstructor() && objType.isFunctionType()) {
410             addInvalidatingType(objType.toMaybeFunctionType().getInstanceType());
411           }
412         }
413       }

424       /**
425        * Finds all properties defined in the externs file and sets them as
426        * ineligible for renaming from the type on which they are defined.
427        */
428       private class FindExternProperties extends AbstractScopedCallback {
429         @Override
430         public void visit(NodeTraversal t, Node n, Node parent) {
431           // TODO(johnlenz): Support object-literal property definitions.
432           if (n.isGetProp()) {
433             String field = n.getLastChild().getString();
434             JSType type = getType(n.getFirstChild());
435             Property prop = getProperty(field);
436             if (isInvalidatingType(type) || isStructuralInterfacePrototype(type)) {
437               prop.invalidate();
438             } else if (!prop.skipRenaming) {
439               prop.addTypeToSkip(type);
440               // If this is a prototype property, then we want to skip assignments
441               // to the instance type as well.  These assignments are not usually
442               // seen in the extern code itself, so we must handle them here.
443               if ((type = getInstanceFromPrototype(type)) != null) {
444                 prop.getTypes().add(type);
445                 prop.typesToSkip.add(type);
446               }
447             }
448           }
449         }
450     
451         private boolean isStructuralInterfacePrototype(JSType type) {
452           return type.isFunctionPrototypeType()
453               && type.toObjectType().getOwnerFunction().isStructuralInterface();
454         }
455       }

457       /**
458        * Traverses the tree, building a map from field names to Nodes for all
459        * fields that can be renamed.
460        */
461       private class FindRenameableProperties extends AbstractScopedCallback {
462         @Override
463         public void visit(NodeTraversal t, Node n, Node parent) {
464           if (n.isGetProp()) {
465             handleGetProp(t, n);
466           } else if (n.isObjectLit()) {
467             handleObjectLit(t, n);
468           } else if (n.isCall()) {
469             handleCall(t, n);
470           }
471         }
472     
473         private void handleGetProp(NodeTraversal t, Node n) {
474           String name = n.getLastChild().getString();
475           JSType type = getType(n.getFirstChild());
476     
477           Property prop = getProperty(name);
478           if (!prop.scheduleRenaming(
479                  n.getLastChild(),
480                  processProperty(t, prop, type, null))
481               && propertiesToErrorFor.containsKey(name)) {
482             String suggestion = "";
483             if (type.isAllType() || type.isUnknownType()) {
484               if (n.getFirstChild().isThis()) {
485                 suggestion = "The \"this\" object is unknown in the function, consider using @this";
486               } else {
487                 String qName = n.getFirstChild().getQualifiedName();
488                 suggestion = "Consider casting " + qName + " if you know its type.";
489               }
490             } else {
491               List<String> errors = new ArrayList<>();
492               printErrorLocations(errors, type);
493               if (!errors.isEmpty()) {
494                 suggestion = "Consider fixing errors for the following types:\n";
495                 suggestion += Joiner.on("\n").join(errors);
496               }
497             }
498             compiler.report(JSError.make(n, propertiesToErrorFor.get(name),
499                     Warnings.INVALIDATION, name, String.valueOf(type), n.toString(),
500                     suggestion));
501           }
502         }
503     
504         private void handleObjectLit(NodeTraversal t, Node n) {
505           // Object.defineProperties literals are handled at the CALL node.
506           if (n.getParent().isCall() && NodeUtil.isObjectDefinePropertiesDefinition(n.getParent())) {
507             return;
508           }
509     
510           for (Node child = n.getFirstChild();
511               child != null;
512               child = child.getNext()) {
513             // Maybe STRING, GET, SET
514             if (child.isQuotedString()) {
515               continue;
516             }
517     
518             // We should never see a mix of numbers and strings.
519             String name = child.getString();
520             JSType type = getType(n);
521     
522             Property prop = getProperty(name);
523             if (!prop.scheduleRenaming(child,
524                                        processProperty(t, prop, type, null))) {
525               // TODO(user): It doesn't look like the user can do much in this
526               // case right now.
527               if (propertiesToErrorFor.containsKey(name)) {
528                 compiler.report(JSError.make(child, propertiesToErrorFor.get(name),
529                     Warnings.INVALIDATION, name, String.valueOf(type), n.toString(), ""));
530               }
531             }
532           }
533         }
534     
535         private void handleCall(NodeTraversal t, Node call) {
536           Node target = call.getFirstChild();
537           if (!target.isQualifiedName()) {
538             return;
539           }
540     
541           String functionName = target.getOriginalQualifiedName();
542           if (functionName != null
543               && compiler.getCodingConvention().isPropertyRenameFunction(functionName)) {
544             handlePropertyRenameFunctionCall(t, call, functionName);
545           } else if (NodeUtil.isObjectDefinePropertiesDefinition(call)) {
546             handleObjectDefineProperties(t, call);
547           }
548         }
549     
550         private void handlePropertyRenameFunctionCall(
551             NodeTraversal t, Node call, String renameFunctionName) {
552           if (call.getChildCount() != 2 && call.getChildCount() != 3) {
553             compiler.report(
554                 JSError.make(
555                     call,
556                     Warnings.INVALID_RENAME_FUNCTION,
557                     renameFunctionName,
558                     " Must be called with 1 or 2 arguments"));
559             return;
560           }
561     
562           if (!call.getSecondChild().isString()) {
563             compiler.report(
564                 JSError.make(
565                     call,
566                     Warnings.INVALID_RENAME_FUNCTION,
567                     renameFunctionName,
568                     " The first argument must be a string literal."));
569             return;
570           }
571     
572           String propName = call.getSecondChild().getString();
573     
574           if (propName.contains(".")) {
575             compiler.report(
576                 JSError.make(
577                     call,
578                     Warnings.INVALID_RENAME_FUNCTION,
579                     renameFunctionName,
580                     " The first argument must not be a property path."));
581             return;
582           }
583     
584           Node obj = call.getChildAtIndex(2);
585           JSType type = getType(obj);
586           Property prop = getProperty(propName);
587           if (!prop.scheduleRenaming(call.getSecondChild(), processProperty(t, prop, type, null))
588               && propertiesToErrorFor.containsKey(propName)) {
589             String suggestion = "";
590             if (type.isAllType() || type.isUnknownType()) {
591               if (obj.isThis()) {
592                 suggestion = "The \"this\" object is unknown in the function, consider using @this";
593               } else {
594                 String qName = obj.getQualifiedName();
595                 suggestion = "Consider casting " + qName + " if you know its type.";
596               }
597             } else {
598               List<String> errors = new ArrayList<>();
599               printErrorLocations(errors, type);
600               if (!errors.isEmpty()) {
601                 suggestion = "Consider fixing errors for the following types:\n";
602                 suggestion += Joiner.on("\n").join(errors);
603               }
604             }
605     
606             compiler.report(
607                 JSError.make(
608                     call,
609                     propertiesToErrorFor.get(propName),
610                     Warnings.INVALIDATION,
611                     propName,
612                     String.valueOf(type),
613                     renameFunctionName,
614                     suggestion));
615           }
616         }

618         private void handleObjectDefineProperties(NodeTraversal t, Node call) {
619           Node typeObj = call.getSecondChild();
620           JSType type = getType(typeObj);
621           Node objectLiteral = typeObj.getNext();
622           if (!objectLiteral.isObjectLit()) {
623             return;
624           }
625     
626           for (Node key : objectLiteral.children()) {
627             if (key.isQuotedString()) {
628               continue;
629             }
630     
631             String propName = key.getString();
632             Property prop = getProperty(propName);
633             prop.scheduleRenaming(key, processProperty(t, prop, type, null));
634           }
635         }

637         private void printErrorLocations(List<String> errors, JSType t) {
638           if (!t.isObject() || t.isAllType()) {
639             return;
640           }
641     
642           if (t.isUnionType()) {
643             for (JSType alt : t.toMaybeUnionType().getAlternates()) {
644               printErrorLocations(errors, alt);
645             }
646             return;
647           }
648     
649           for (JSError error : invalidationMap.get(t)) {
650             errors.add(t + " at " + error.sourceName + ":" + error.lineNumber);
651           }
652         }

654         /**
655          * Processes a property, adding it to the list of properties to rename.
656          * @return a representative type for the property reference, which will be
657          *   the highest type on the prototype chain of the provided type.  In the
658          *   case of a union type, it will be the highest type on the prototype
659          *   chain of one of the members of the union.
660          */
661         private JSType processProperty(
662             NodeTraversal t, Property prop, JSType type, JSType relatedType) {
663           type = type.restrictByNotNullOrUndefined();
664           if (prop.skipRenaming || isInvalidatingType(type)) {
665             return null;
666           }
667     
668           Iterable<JSType> alternatives = getTypeAlternatives(type);
669           if (alternatives != null) {
670             JSType firstType = relatedType;
671             for (JSType subType : alternatives) {
672               JSType lastType = processProperty(t, prop, subType, firstType);
673               if (lastType != null) {
674                 firstType = firstType == null ? lastType : firstType;
675               }
676             }
677             return firstType;
678           } else {
679             JSType topType = getTypeWithProperty(prop.name, type);
680             if (isInvalidatingType(topType)) {
681               return null;
682             }
683             prop.addType(type, relatedType);
684             return topType;
685           }
686         }
687       }

689       /** Renames all properties with references on more than one type. */
690       void renameProperties() {
691         int propsRenamed = 0, propsSkipped = 0, instancesRenamed = 0,
692             instancesSkipped = 0, singleTypeProps = 0;
693     
694         Set<String> reported = new HashSet<>();
695         for (Property prop : properties.values()) {
696           if (prop.shouldRename()) {
697             UnionFind<JSType> pTypes = prop.getTypes();
698             Map<JSType, String> propNames = buildPropNames(prop);
699     
700             ++propsRenamed;
701             prop.expandTypesToSkip();
702             // This loop has poor locality, because instead of walking the AST,
703             // we iterate over all accesses of a property, which can be in very
704             // different places in the code.
705             for (Map.Entry<Node, JSType> entry : prop.rootTypesByNode.entrySet()) {
706               Node node = entry.getKey();
707               JSType rootType = entry.getValue();
708               if (prop.shouldRename(rootType)) {
709                 String newName = propNames.get(pTypes.find(rootType));
710                 node.setString(newName);
711                 compiler.reportCodeChange();
712                 ++instancesRenamed;
713               } else {
714                 ++instancesSkipped;
715     
716                 CheckLevel checkLevelForProp = propertiesToErrorFor.get(prop.name);
717                 if (checkLevelForProp != null &&
718                     checkLevelForProp != CheckLevel.OFF &&
719                     !reported.contains(prop.name)) {
720                   reported.add(prop.name);
721                   compiler.report(JSError.make(
722                       node,
723                       checkLevelForProp,
724                       Warnings.INVALIDATION_ON_TYPE, prop.name,
725                       rootType.toString(), ""));
726                 }
727               }
728             }
729           } else {
730             if (prop.skipRenaming) {
731               ++propsSkipped;
732             } else {
733               ++singleTypeProps;
734             }
735           }
736         }
737         logger.fine("Renamed " + instancesRenamed + " instances of "
738                     + propsRenamed + " properties.");
739         logger.fine("Skipped renaming " + instancesSkipped + " invalidated "
740                     + "properties, " + propsSkipped + " instances of properties "
741                     + "that were skipped for specific types and " + singleTypeProps
742                     + " properties that were referenced from only one type.");
743       }

745       /**
746        * Chooses a name to use for renaming in each equivalence class and maps
747        * the representative type of that class to that name.
748        */
749       private Map<JSType, String> buildPropNames(Property prop) {
750         UnionFind<JSType> pTypes = prop.getTypes();
751         String pname = prop.name;
752         Map<JSType, String> names = new HashMap<>();
753         for (Set<JSType> set : pTypes.allEquivalenceClasses()) {
754           checkState(!set.isEmpty());
755           JSType representative = pTypes.find(set.iterator().next());
756           String typeName = null;
757           for (JSType type : set) {
758             String typeString = type.toString();
759             if (typeName == null || typeString.compareTo(typeName) < 0) {
760               typeName = typeString;
761             }
762           }
763           String newName;
764           if ("{...}".equals(typeName)) {
765             newName = pname;
766           } else {
767             newName = NONWORD_PATTERN.matcher(typeName).replaceAll("_") + '$' + pname;
768           }
769           names.put(representative, newName);
770         }
771         return names;
772       }

774       /** Returns a map from field name to types for which it will be renamed. */
775       Multimap<String, Collection<JSType>> getRenamedTypesForTesting() {
776         Multimap<String, Collection<JSType>> ret = HashMultimap.create();
777         for (Map.Entry<String, Property> entry : properties.entrySet()) {
778           Property prop = entry.getValue();
779           if (!prop.skipRenaming) {
780             for (Collection<JSType> c : prop.getTypes().allEquivalenceClasses()) {
781               if (!c.isEmpty() && !prop.typesToSkip.contains(c.iterator().next())) {
782                 ret.put(entry.getKey(), c);
783               }
784             }
785           }
786         }
787         return ret;
788       }

802       /**
803        * Returns true if a field reference on this type will invalidate all
804        * references to that field as candidates for renaming. This is true if the
805        * type is unknown or all-inclusive, as variables with such a type could be
806        * references to any object.
807        */
808       private boolean isInvalidatingType(JSType type) {
809         if (type == null || invalidatingTypes.contains(type) ||
810             type.isUnknownType() /* unresolved types */) {
811           return true;
812         }
813         ObjectType objType = ObjectType.cast(type);
814         return objType != null && !objType.hasReferenceName();
815       }

817       /**
818        * Returns a set of types that should be skipped given the given type. This is
819        * necessary for interfaces, as all super interfaces must also be skipped.
820        */
821       private ImmutableSet<JSType> getTypesToSkipForType(JSType type) {
822         type = type.restrictByNotNullOrUndefined();
823         if (type.isUnionType()) {
824           ImmutableSet.Builder<JSType> types = ImmutableSet.builder();
825           types.add(type);
826           for (JSType alt : type.toMaybeUnionType().getAlternates()) {
827             types.addAll(getTypesToSkipForTypeNonUnion(alt));
828           }
829           return types.build();
830         } else if (type.isEnumElementType()) {
831           return getTypesToSkipForType(
832               type.toMaybeEnumElementType().getPrimitiveType());
833         }
834         return ImmutableSet.copyOf(getTypesToSkipForTypeNonUnion(type));
835       }
836     
837       private Set<JSType> getTypesToSkipForTypeNonUnion(JSType type) {
838         Set<JSType> types = new HashSet<>();
839         JSType skipType = type;
840         while (skipType != null) {
841           types.add(skipType);
842           ObjectType objSkipType = skipType.toObjectType();
843           if (objSkipType != null) {
844             skipType = objSkipType.getImplicitPrototype();
845           } else {
846             break;
847           }
848         }
849         return types;
850       }

852       /**
853        * Determines whether the given type is one whose properties should not be
854        * considered for renaming.
855        */
856       private boolean isTypeToSkip(JSType type) {
857         return type.isEnumType() || (type.autoboxesTo() != null);
858       }

860       /**
861        * Returns the alternatives if this is a type that represents multiple
862        * types, and null if not. Union and interface types can correspond to
863        * multiple other types.
864        */
865       private Iterable<JSType> getTypeAlternatives(JSType type) {
866         if (type.isUnionType()) {
867           return type.toMaybeUnionType().getAlternatesWithoutStructuralTyping();
868         } else {
869           ObjectType objType = type.toObjectType();
870           if (objType != null &&
871               objType.getConstructor() != null &&
872               objType.getConstructor().isInterface()) {
873             List<JSType> list = new ArrayList<>();
874             for (FunctionType impl
875                      : registry.getDirectImplementors(objType)) {
876               list.add(impl.getInstanceType());
877             }
878             return list;
879           } else {
880             return null;
881           }
882         }
883       }

885       /**
886        * Returns the type in the chain from the given type that contains the given
887        * field or null if it is not found anywhere.
888        * Can return a subtype of the input type.
889        */
890       @VisibleForTesting
891       ObjectType getTypeWithProperty(String field, JSType type) {
892         if (type == null) {
893           return null;
894         }
895     
896         ObjectType foundType = gtwpCacheGet(field, type);
897         if (foundType != null) {
898           return foundType.equals(BOTTOM_OBJECT) ? null : foundType;
899         }
900     
901         if (type.isEnumElementType()) {
902           foundType = getTypeWithProperty(
903               field, type.toMaybeEnumElementType().getPrimitiveType());
904           gtwpCachePut(field, type, foundType == null ? BOTTOM_OBJECT : foundType);
905           return foundType;
906         }
907     
908         if (!(type instanceof ObjectType)) {
909           if (type.autoboxesTo() != null) {
910             foundType = getTypeWithProperty(field, type.autoboxesTo());
911             gtwpCachePut(field, type, foundType == null ? BOTTOM_OBJECT : foundType);
912             return foundType;
913           } else {
914             gtwpCachePut(field, type, BOTTOM_OBJECT);
915             return null;
916           }
917         }
918     
919         // Ignore the prototype itself at all times.
920         if ("prototype".equals(field)) {
921           gtwpCachePut(field, type, BOTTOM_OBJECT);
922           return null;
923         }
924     
925         // We look up the prototype chain to find the highest place (if any) that
926         // this appears.  This will make references to overridden properties look
927         // like references to the initial property, so they are renamed alike.
928         ObjectType objType = ObjectType.cast(type);
929         if (objType != null && objType.getConstructor() != null
930             && objType.getConstructor().isInterface()) {
931           ObjectType topInterface = FunctionType.getTopDefiningInterface(
932               objType, field);
933           if (topInterface != null && topInterface.getConstructor() != null) {
934             foundType = topInterface.getConstructor().getPrototype();
935           }
936         } else {
937           while (objType != null && objType.getImplicitPrototype() != objType) {
938             if (objType.hasOwnProperty(field)) {
939               foundType = objType;
940             }
941             objType = objType.getImplicitPrototype();
942           }
943         }
944     
945         // If the property does not exist on the referenced type but the original
946         // type is an object type, see if any subtype has the property.
947         if (foundType == null) {
948           ObjectType maybeType = ObjectType.cast(
949               registry.getGreatestSubtypeWithProperty(type, field));
950           // getGreatestSubtypeWithProperty does not guarantee that the property
951           // is defined on the returned type, it just indicates that it might be,
952           // so we have to double check.
953           if (maybeType != null && maybeType.hasOwnProperty(field)) {
954             foundType = maybeType;
955           }
956         }
957     
958         // Unwrap templatized types, they are not unique at runtime.
959         if (foundType != null && foundType.isTemplatizedType()) {
960           foundType = foundType.toMaybeTemplatizedType().getReferencedType();
961         }
962     
963         // Since disambiguation just looks at names, we must return a uniquely named type rather
964         // than an "equivalent" type. In particular, we must manually unwrap named types
965         // so that the returned type has the correct name.
966         if (foundType != null && foundType.isNamedType()) {
967           foundType = foundType.toMaybeNamedType().getReferencedType().toMaybeObjectType();
968         }
969     
970         gtwpCachePut(field, type, foundType == null ? BOTTOM_OBJECT : foundType);
971         return foundType;
972       }

974       private JSType getInstanceFromPrototype(JSType type) {
975         if (type.isFunctionPrototypeType()) {
976           ObjectType prototype = (ObjectType) type;
977           FunctionType owner = prototype.getOwnerFunction();
978           if (owner.isConstructor() || owner.isInterface()) {
979             return prototype.getOwnerFunction().getInstanceType();
980           }
981         }
982         return null;
983       }

985       /**
986        * Records that this property could be referenced from any interface that
987        * this type, or any type in its superclass chain, implements.
988        *
989        * If the property p is defined only on a subtype of constructor, then this
990        * method has no effect. But we tried modifying getTypeWithProperty to tell us
991        * when the returned type is a subtype, and then skip those calls to
992        * recordInterface, and there was no speed-up.
993        * And it made the code harder to understand, so we don't do it.
994        */
995       private void recordInterfaces(FunctionType constructor, JSType relatedType,
996           DisambiguateProperties.Property p) {
997         Preconditions.checkArgument(constructor.isConstructor());
998         Iterable<ObjectType> interfaces = implementedInterfaces.get(constructor);
999         if (interfaces == null) {
1000           interfaces = constructor.getImplementedInterfaces();
1001           implementedInterfaces.put(constructor, interfaces);
1002         }
1003         for (ObjectType itype : interfaces) {
1004           JSType top = getTypeWithProperty(p.name, itype);
1005           if (top != null) {
1006             p.addType(itype, relatedType);
1007           }
1008           // If this interface invalidated this property, return now.
1009           if (p.skipRenaming) {
1010             return;
1011           }
1012         }
1013       }

1015       private FunctionType getConstructor(JSType type) {
1016         ObjectType objType = ObjectType.cast(type);
1017         if (objType == null) {
1018           return null;
1019         }
1020         FunctionType constructor = null;
1021         if (objType.isFunctionType()) {
1022           constructor = objType.toMaybeFunctionType();
1023         } else if (objType.isFunctionPrototypeType()) {
1024           constructor = objType.getOwnerFunction();
1025         } else {
1026           constructor = objType.getConstructor();
1027         }
1028         return constructor != null && constructor.isConstructor()
1029             ? constructor : null;
1030       }
