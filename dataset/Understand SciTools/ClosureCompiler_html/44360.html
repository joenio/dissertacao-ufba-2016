34     /**
35      * Rewrites calls to ES6 library functions to use compiler-provided polyfills,
36      * e.g., <code>var m = new Map();</code> becomes
37      * <code>$jscomp.Map$install(); var m = new $jscomp.Map();</code>
38      */
39     public class RewritePolyfills implements HotSwapCompilerPass {
40     
41       static final DiagnosticType INSUFFICIENT_OUTPUT_VERSION_ERROR = DiagnosticType.warning(
42           "JSC_INSUFFICIENT_OUTPUT_VERSION",
43           "Built-in ''{0}'' not supported in output version {1}: set --language_out to at least {2}");
44     
45       // Also polyfill references to e.g. goog.global.Map or window.Map.
46       private static final String GLOBAL = "goog.global.";
47       private static final String WINDOW = "window.";
48     
49       /**
50        * Represents a single polyfill: specifically, a native symbol
51        * (either a qualified name or a property name) that can be
52        * rewritten and/or installed to provide the functionality to
53        * a lower version.  This is a simple value type.
54        */
55       private static class Polyfill {
56         /**
57          * The language version at (or above) which the native symbol is
58          * available and sufficient.  If the language out flag is at least
59          * as high as {@code nativeVersion} then no rewriting will happen.
60          */
61         final FeatureSet nativeVersion;
62     
63         /**
64          * The required language version for the polyfill to work.  This
65          * should not be higher than {@code nativeVersion}, but may be the same
66          * in cases where there is no polyfill provided.  This is used to
67          * emit a warning if the language out flag is too low.
68          */
69         final FeatureSet polyfillVersion;
70     
71         /**
72          * Optional qualified name to drop-in replace for the native symbol.
73          * May be empty if no direct rewriting is to take place.
74          */
75         final String rewrite;
76     
77         /**
78          * Optional "installer" to insert (once) at the top of a source file.
79          * If present, this should be a JavaScript statement, or empty if no
80          * installer should be inserted.
81          */
82         final String installer;
83     
84         Polyfill(
85             FeatureSet nativeVersion, FeatureSet polyfillVersion, String rewrite, String installer) {
86           this.nativeVersion = nativeVersion;
87           this.polyfillVersion = polyfillVersion;
88           this.rewrite = rewrite;
89           this.installer = installer;
90         }
91       }
92     
93       /**
94        * Describes all the available polyfills, including native and
95        * required versions, and how to use them.
96        */
97       static class Polyfills {
98         // Map of method polyfills, keyed by native method name.
99         private final ImmutableMultimap<String, Polyfill> methods;
100         // Map of static polyfills, keyed by fully-qualified native name.
101         private final ImmutableMap<String, Polyfill> statics;
102     
103         private Polyfills(Builder builder) {
104           this.methods = builder.methodsBuilder.build();
105           this.statics = builder.staticsBuilder.build();
106         }
107     
108         /**
109          * Provides a DSL for building a {@link Polyfills} object by calling
110          * {@link #addStatics}, {@link #addMethods}, and {@link #addClasses}
111          * to register the various polyfills and provide information about
112          * the native and polyfilled versions, and how to use the polyfills.
113          */
114         static class Builder {
115           private final ImmutableMultimap.Builder<String, Polyfill> methodsBuilder =
116               ImmutableMultimap.builder();
117           private final ImmutableMap.Builder<String, Polyfill> staticsBuilder = ImmutableMap.builder();
118     
119           /**
120            * Registers one or more prototype method in a single namespace.
121            * The pass is agnostic with regard to the class whose prototype
122            * is being augmented.  The {@code base} parameter specifies the
123            * qualified namespace where all the {@code methods} reside.  Each
124            * method is expected to have a sibling named with the {@code $install}
125            * suffix.  The method calls themselves are not rewritten, but
126            * whenever one is detected, its installer(s) will be added to the
127            * top of the source file whenever the output version is less than
128            * {@code nativeVersion}.  For example, defining {@code
129            *    addMethods(ES6, ES5, "$jscomp.string", "startsWith", "endsWith")}
130            * will cause {@code $jscomp.string.startsWith$install();} to be
131            * added to any source file that calls, e.g. {@code foo.startsWith}.
132            *
133            * <p>If {@code base} is blank, then no polyfills will be installed.
134            * This is useful for documenting unimplemented polyfills.
135            */
136           Builder addMethods(
137               FeatureSet nativeVersion, FeatureSet polyfillVersion, String base, String... methods) {
138             if (!base.isEmpty()) {
139               for (String method : methods) {
140                 methodsBuilder.put(
141                     method,
142                     new Polyfill(
143                         nativeVersion, polyfillVersion, "", base + "." + method + "$install();"));
144               }
145             }
146             // TODO(sdh): If base.isEmpty() then it means no polyfill is implemented.  Is there
147             // any way we can warn if the output language is too low?  It's not likely, since
148             // there's no good way to determine if it's actually intended as an ES6 method or
149             // else is defined elsewhere.
150             return this;
151           }
152     
153           /**
154            * Registers one or more static rewrite polyfill, which is a
155            * simple rewrite of one qualified name to another.  For each
156            * {@code name} in {@code statics}, {@code nativeBase + '.' + name}
157            * will be replaced with {@code polyfillBase + '.' + name}
158            * whenever the output version is less than {@code nativeVersion}.
159            * For eaxmple, defining {@code
160            *    addStatics(ES6, ES5, "$jscomp.math", "Math", "clz32", "imul")}
161            * will cause {@code Math.clz32} to be rewritten as
162            * {@code $jscomp.math.clz32}.
163            *
164            * <p>If {@code polyfillBase} is blank, then no polyfills will be
165            * installed.  This is useful for documenting unimplemented polyfills,
166            * and will trigger a warning if the language output mode is less than
167            * the native version.
168            */
169           Builder addStatics(
170               FeatureSet nativeVersion,
171               FeatureSet polyfillVersion,
172               String polyfillBase,
173               String nativeBase,
174               String... statics) {
175             for (String item : statics) {
176               String nativeName = nativeBase + "." + item;
177               String polyfillName = !polyfillBase.isEmpty() ? polyfillBase + "." + item : "";
178               Polyfill polyfill = new Polyfill(nativeVersion, polyfillVersion, polyfillName, "");
179               staticsBuilder.put(nativeName, polyfill);
180               staticsBuilder.put(GLOBAL + nativeName, polyfill);
181               staticsBuilder.put(WINDOW + nativeName, polyfill);
182             }
183             return this;
184           }
185     
186           /**
187            * Registers one or more class polyfill.  Class polyfills
188            * are both rewritten in place and also installed (so that
189            * faster native versions may be preferred if available).
190            * The {@code base} parameter is a qualified name prefix
191            * added to the class name to get the polyfill's name.
192            * A sibling method with the {@code $install} suffix should
193            * also be present.
194            * For example, defining {@code
195            *    addClasses(ES6, ES5, "$jscomp", "Map", "Set")}
196            * will cause {@code new Map()} to be rewritten as
197            * {@code new $jscomp.Map()} and will insert {@code
198            * $jscomp.Map$install();} at the top of the source
199            * file whenever the output version is less than
200            * {@code nativeVersion}.
201            *
202            * <p>If {@code base} is blank, then no polyfills will be
203            * installed.  This is useful for documenting unimplemented
204            * polyfills, and will trigger a warning if the language
205            * output mode is less than the native version.
206            */
207           Builder addClasses(
208               FeatureSet nativeVersion, FeatureSet polyfillVersion, String base, String... classes) {
209             for (String className : classes) {
210               String polyfillName = base + "." + className;
211               Polyfill polyfill =
212                   !base.isEmpty()
213                       ? new Polyfill(
214                           nativeVersion, polyfillVersion, polyfillName, polyfillName + "$install();")
215                       : new Polyfill(nativeVersion, polyfillVersion, "", "");
216               staticsBuilder.put(className, polyfill);
217               staticsBuilder.put(GLOBAL + className, polyfill);
218               staticsBuilder.put(WINDOW + className, polyfill);
219             }
220             return this;
221           }
222     
223           /** Builds the {@link Polyfills}. */
224           Polyfills build() {
225             return new Polyfills(this);
226           }
227         }
228       }

289       @Override
290       public void hotSwapScript(Node scriptRoot, Node originalRoot) {
291         Traverser traverser = new Traverser();
292         NodeTraversal.traverseEs6(compiler, scriptRoot, traverser);
293         if (traverser.changed) {
294           compiler.reportCodeChange();
295         }
296       }
297     
298       @Override
299       public void process(Node externs, Node root) {
300         if (languageOutIsAtLeast(ES6) || !compiler.getOptions().rewritePolyfills) {
301           return; // no rewriting in this case.
302         }
303         this.globals = new GlobalNamespace(compiler, externs, root);
304         hotSwapScript(root, null);
305       }

307       private class Traverser extends AbstractPostOrderCallback {
308     
309         Node injectedLibraryNode = null;
310         Set<String> installers = new HashSet<>();
311         boolean changed = false;
312     
313         @Override
314         public void visit(NodeTraversal traversal, Node node, Node parent) {
315     
316           // Fix types in JSDoc.
317           JSDocInfo doc = node.getJSDocInfo();
318           if (doc != null) {
319             fixJsdoc(traversal.getScope(), doc);
320           }
321     
322           // Find qualified names that match static calls
323           if (node.isQualifiedName()) {
324             String name = node.getQualifiedName();
325             Polyfill polyfill = null;
326     
327             if (polyfills.statics.containsKey(name)) {
328               polyfill = polyfills.statics.get(name);
329             }
330     
331             if (polyfill != null) {
332               // Check the scope to make sure it's a global name.
333               if (isRootInScope(node, traversal) || NodeUtil.isVarOrSimpleAssignLhs(node, parent)) {
334                 return;
335               }
336     
337               if (!languageOutIsAtLeast(polyfill.polyfillVersion)) {
338                 traversal.report(
339                     node,
340                     INSUFFICIENT_OUTPUT_VERSION_ERROR,
341                     name,
342                     compiler.getOptions().getLanguageOut().toString(),
343                     polyfill.polyfillVersion.toLanguageModeString());
344               }
345               if (!languageOutIsAtLeast(polyfill.nativeVersion)) {
346     
347                 if (!polyfill.installer.isEmpty()) {
348                   // Note: add the installer *before* replacing the node!
349                   addInstaller(polyfill.installer);
350                 }
351     
352                 if (!polyfill.rewrite.isEmpty()) {
353                   changed = true;
354                   injectRuntime();
355                   Node replacement = NodeUtil.newQName(compiler, polyfill.rewrite);
356                   replacement.useSourceInfoIfMissingFromForTree(node);
357                   parent.replaceChild(node, replacement);
358                 }
359               }
360     
361               // TODO(sdh): consider warning if language_in is too low?  it's not really any
362               // harm, and we can't do it consistently for the prototype methods, so maybe
363               // it's not worth doing here, either.
364     
365               return; // isGetProp (below) overlaps, so just bail out now
366             }
367           }
368     
369           // Add any requires that *might* match method calls (but don't rewrite anything)
370           if (node.isGetProp() && node.getLastChild().isString()) {
371             for (Polyfill polyfill : polyfills.methods.get(node.getLastChild().getString())) {
372               if (!languageOutIsAtLeast(polyfill.nativeVersion) && !polyfill.installer.isEmpty()) {
373                 // Check if this is a global function.
374                 if (!isStaticFunction(node, traversal)) {
375                   addInstaller(polyfill.installer);
376                 }
377               }
378             }
379           }
380         }
381     
382         private boolean isStaticFunction(Node node, NodeTraversal traversal) {
383           if (!node.isQualifiedName()) {
384             return false;
385           }
386           String root = NodeUtil.getRootOfQualifiedName(node).getQualifiedName();
387           if (globals == null) {
388             return false;
389           }
390           GlobalNamespace.Name fullName = globals.getOwnSlot(node.getQualifiedName());
391           GlobalNamespace.Name rootName = globals.getOwnSlot(root);
392           if (fullName == null || rootName == null) {
393             return false;
394           }
395           GlobalNamespace.Ref rootDecl = rootName.getDeclaration();
396           if (rootDecl == null) {
397             return false;
398           }
399           Node globalDeclNode = rootDecl.getNode();
400           if (globalDeclNode == null) {
401             return false; // don't know where the root came from so assume it could be anything
402           }
403           Var rootScope = traversal.getScope().getVar(root);
404           if (rootScope == null) {
405             return true; // root is not in the current scope, so it's a static function
406           }
407           Node scopeDeclNode = rootScope.getNode();
408           return scopeDeclNode == globalDeclNode; // is the global name currently in scope?
409         }

411         // Fix all polyfill type references in any JSDoc.
412         private void fixJsdoc(Scope scope, JSDocInfo doc) {
413           for (Node node : doc.getTypeNodes()) {
414             fixJsdocType(scope, node);
415           }
416         }
417     
418         private void fixJsdocType(Scope scope, Node node) {
419           if (node.isString()) {
420             Polyfill polyfill = polyfills.statics.get(node.getString());
421             // Note: all classes are unqualified names, so we don't need to deal with dots
422             if (polyfill != null
423                 && scope.getVar(node.getString()) == null
424                 && !languageOutIsAtLeast(polyfill.nativeVersion)) {
425               node.setString(polyfill.rewrite);
426             }
427           }
428           for (Node child = node.getFirstChild(); child != null; child = child.getNext()) {
429             fixJsdocType(scope, child);
430           }
431         }
432     
433         private void addInstaller(String function) {
434           if (installers.add(function)) {
435             changed = true;
436             Node installer = compiler.parseSyntheticCode(function).removeChildren();
437     
438             injectRuntime();
439     
440             if (injectedLibraryNode != null) {
441               injectedLibraryNode.getParent().addChildrenAfter(installer, injectedLibraryNode);
442             } else {
443               compiler.getNodeForCodeInsertion(null).addChildrenToFront(installer);
444             }
445           }
446         }
447     
448         private void injectRuntime() {
449           if (injectedLibraryNode == null) {
450             injectedLibraryNode = compiler.ensureLibraryInjected("es6_runtime", false);
451           }
452         }
453       }

455       private boolean languageOutIsAtLeast(LanguageMode mode) {
456         return compiler.getOptions().getLanguageOut().compareTo(mode) >= 0;
457       }
458     
459       private boolean languageOutIsAtLeast(FeatureSet features) {
460         switch (features.version()) {
461           case "ts":
462             return languageOutIsAtLeast(LanguageMode.ECMASCRIPT6_TYPED);
463           case "es6":
464           case "es6-impl": // TODO(sdh): support a separate language mode for es6-impl?
465             return languageOutIsAtLeast(LanguageMode.ECMASCRIPT6);
466           case "es5":
467             return languageOutIsAtLeast(LanguageMode.ECMASCRIPT5);
468           case "es3":
469             return languageOutIsAtLeast(LanguageMode.ECMASCRIPT3);
470           default:
471             return false;
472         }
473       }
474     
475       private static boolean isRootInScope(Node node, NodeTraversal traversal) {
476         String rootName = NodeUtil.getRootOfQualifiedName(node).getQualifiedName();
477         return traversal.getScope().getVar(rootName) != null;
478       }
