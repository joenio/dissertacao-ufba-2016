16     package com.google.javascript.jscomp;
17     
18     import com.google.javascript.jscomp.CompilerOptions.LanguageMode;
19     
20     /**
21      * Round-trip test for TypeScript-style inline type syntax.  Each expression is
22      * parsed and then printed, and we assert that the pretty-printed result is identical
23      * to the input.
24      *
25      * <p>See {@link JsdocToEs6TypedConverterTest} for tests which start from
26      * closure-style JSDoc type declaration syntax.
27      */
28     public final class CodePrinterEs6TypedTest extends CodePrinterTestBase {
29     
30       @Override
31       public void setUp() {
32         super.setUp();
33         languageMode = LanguageMode.ECMASCRIPT6_TYPED;
34       }
35     
36       void assertPrettyPrintSame(String js) {
37         String parsed = parsePrint(js, newCompilerOptions(new CompilerOptionBuilder() {
38           @Override void setOptions(CompilerOptions options) {
39             options.setPrettyPrint(true);
40             options.setPreferLineBreakAtEndOfFile(false);
41             options.setPreferSingleQuotes(true);
42           }
43         }));
44         parsed = parsed.trim(); // strip trailing line break.
45         assertEquals(js, parsed);
46       }
47     
48       public void testVariableDeclaration() {
49         assertPrettyPrintSame("var foo: any = 'hello';");
50         assertPrettyPrintSame("var foo: number = 'hello';");
51         assertPrettyPrintSame("var foo: boolean = 'hello';");
52         assertPrettyPrintSame("var foo: string = 'hello';");
53         assertPrettyPrintSame("var foo: void = 'hello';");
54         assertPrettyPrintSame("var foo: hello = 'hello';");
55       }
56     
57       public void testFunctionParamDeclaration() {
58         assertPrettyPrintSame("function foo(x: string) {\n}");
59       }
60     
61       public void testFunctionParamDeclaration_defaultValue() {
62         assertPrettyPrintSame("function foo(x: string = 'hello') {\n}");
63       }
64     
65       public void disabled_testFunctionParamDeclaration_arrow() {
66         assertPrettyPrintSame("(x: string) => 'hello' + x;");
67       }
68     
69       public void testFunctionReturn() {
70         assertPrettyPrintSame("function foo(): string {\n  return 'hello';\n}");
71       }
72     
73       public void disabled_testFunctionReturn_arrow() {
74         assertPrettyPrintSame("(): string => 'hello';");
75       }
76     
77       public void testCompositeType() {
78         assertPrettyPrintSame("var foo: mymod.ns.Type;");
79       }
80     
81       public void testArrayType() {
82         assertPrettyPrintSame("var foo: string[];");
83       }
84     
85       public void testArrayType_qualifiedType() {
86         assertPrettyPrintSame("var foo: mymod.ns.Type[];");
87       }
88     
89       public void testParameterizedType() {
90         assertPrettyPrintSame("var x: my.parameterized.Type<ns.A, ns.B>;");
91       }
92     }
