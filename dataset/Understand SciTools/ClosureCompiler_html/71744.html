40     package com.google.javascript.rhino.jstype;
41     
42     /**
43      * Constants corresponding to types that are built into a JavaScript engine
44      * and other types that occur very often in the type system. See
45      * {@link com.google.javascript.rhino.jstype.JSTypeRegistry#getNativeType(JSTypeNative)}.
46      */
47     public enum JSTypeNative {
48       // Built-in types (please keep alphabetized)
49     
50       ARRAY_TYPE,
51       ARRAY_FUNCTION_TYPE,
52     
53       BOOLEAN_TYPE,
54       BOOLEAN_OBJECT_TYPE,
55       BOOLEAN_OBJECT_FUNCTION_TYPE,
56     
57       /**
58        * A checked unknown type is a type that we know something about,
59        * but we're not really sure what we know about it.
60        *
61        * Examples of checked unknown types include:
62        * <code>
63        * if (x) { // x is unknown
64        *   alert(x); // x is checked unknown
65        * }
66        * </code>
67        *
68        * <code>
69        * /* @param {SomeForwardDeclaredType} x /
70        * function f(x) {
71        *   // x is checked unknown. We know it's some type, but the type
72        *   // has not been included in this binary.
73        * }
74        * </code>
75        *
76        * This is useful for missing property warnings, where we don't
77        * want to emit warnings on things that have been checked.
78        */
79       CHECKED_UNKNOWN_TYPE,
80     
81       DATE_TYPE,
82       DATE_FUNCTION_TYPE,
83     
84       ERROR_FUNCTION_TYPE,
85       ERROR_TYPE,
86     
87       EVAL_ERROR_FUNCTION_TYPE,
88       EVAL_ERROR_TYPE,
89     
90       FUNCTION_FUNCTION_TYPE,
91       FUNCTION_INSTANCE_TYPE, // equivalent to U2U_CONSTRUCTOR_TYPE
92       FUNCTION_PROTOTYPE,
93     
94       NULL_TYPE,
95     
96       NUMBER_TYPE,
97       NUMBER_OBJECT_TYPE,
98       NUMBER_OBJECT_FUNCTION_TYPE,
99     
100       OBJECT_TYPE,
101       OBJECT_FUNCTION_TYPE,
102       OBJECT_PROTOTYPE,
103     
104       RANGE_ERROR_FUNCTION_TYPE,
105       RANGE_ERROR_TYPE,
106     
107       REFERENCE_ERROR_FUNCTION_TYPE,
108       REFERENCE_ERROR_TYPE,
109     
110       REGEXP_TYPE,
111       REGEXP_FUNCTION_TYPE,
112     
113       STRING_OBJECT_TYPE,
114       STRING_OBJECT_FUNCTION_TYPE,
115       STRING_TYPE,
116     
117       SYNTAX_ERROR_FUNCTION_TYPE,
118       SYNTAX_ERROR_TYPE,
119     
120       TYPE_ERROR_FUNCTION_TYPE,
121       TYPE_ERROR_TYPE,
122     
123       UNKNOWN_TYPE,
124     
125       URI_ERROR_FUNCTION_TYPE,
126       URI_ERROR_TYPE,
127     
128       VOID_TYPE,
129     
130       // Commonly used types
131     
132       TOP_LEVEL_PROTOTYPE,
133       STRING_VALUE_OR_OBJECT_TYPE,
134       NUMBER_VALUE_OR_OBJECT_TYPE,
135       ALL_TYPE,
136       NO_TYPE,
137       NO_OBJECT_TYPE,
138       NO_RESOLVED_TYPE,
139       GLOBAL_THIS,
140       // TODO(dimvar): merge these two if possible.
141       U2U_CONSTRUCTOR_TYPE,
142       // unknown function type, i.e. (?...) -> ?
143       U2U_FUNCTION_TYPE,
144     
145       LEAST_FUNCTION_TYPE,
146       GREATEST_FUNCTION_TYPE,
147     
148       /**
149        * (null, void)
150        */
151       NULL_VOID,
152     
153       /**
154        * (Object,number,string)
155        */
156       OBJECT_NUMBER_STRING,
157     
158       /**
159        * (Object,number,string,boolean)
160        */
161       OBJECT_NUMBER_STRING_BOOLEAN,
162     
163       /**
164        * (number,string,boolean)
165        */
166       NUMBER_STRING_BOOLEAN,
167     
168       /**
169        * (number,string)
170        */
171       NUMBER_STRING,
172     }
