35       public void testDelegation() {
36         Compiler compiler = new Compiler();
37         JsAst ast = new JsAst(SourceFile.fromCode("file.js", "// nothing here"));
38         SimpleDependencyInfo delegate =
39             new SimpleDependencyInfo(
40                 "path/to/1.js",
41                 "path/2.js",
42                 ImmutableList.of("foo", "bar"),
43                 ImmutableList.of("baz", "qux"),
44                 ImmutableMap.<String, String>of());
45         DependencyInfo info = new LazyParsedDependencyInfo(delegate, ast, compiler);
46     
47         assertThat(info.getName()).isEqualTo("path/2.js");
48         assertThat(info.getPathRelativeToClosureBase()).isEqualTo("path/to/1.js");
49         assertThat(info.getProvides()).containsExactly("foo", "bar");
50         assertThat(info.getRequires()).containsExactly("baz", "qux");
51       }

53       public void testLoadFlagsParsesEs3() {
54         Compiler compiler = new Compiler();
55         compiler.initOptions(new CompilerOptions());
56         JsAst ast = new JsAst(SourceFile.fromCode("file.js", "// nothing here"));
57         SimpleDependencyInfo delegate =
58             new SimpleDependencyInfo("", "", EMPTY, EMPTY, ImmutableMap.of("foo", "bar"));
59         DependencyInfo info = new LazyParsedDependencyInfo(delegate, ast, compiler);
60     
61         // TODO(sdh): We're currently stuck on an earlier version of Truth that doesn't
62         // provide MapSubject#containsExactly(Object, Object, Object...), and it is very
63         // hard to upgrade due to parallel ant and maven builds.  Once this restriction
64         // is lifted and we can depend on a newer Truth, these assertions should be
65         // changed to assertThat(info.getLoadFlags()).containsExactly(...)
66         assertThat(info.getLoadFlags()).isEqualTo(ImmutableMap.of("foo", "bar"));
67         assertFalse(info.isModule());
68       }

106       public void testParseIsLazy() {
107         Compiler compiler = new Compiler();
108         compiler.initOptions(new CompilerOptions());
109         JsAst ast = new JsAst(SourceFile.fromCode("file.js", "parse error"));
110         SimpleDependencyInfo delegate =
111             new SimpleDependencyInfo("", "", EMPTY, EMPTY, ImmutableMap.<String, String>of());
112         DependencyInfo info = new LazyParsedDependencyInfo(delegate, ast, compiler);
113     
114         info.getName();
115         info.getPathRelativeToClosureBase();
116         info.getProvides();
117         info.getRequires();
118     
119         assertThat(compiler.getErrorManager().getErrorCount()).isEqualTo(0);
120         info.getLoadFlags();
121         assertThat(compiler.getErrorManager().getErrorCount()).isAtLeast(1);
122       }

124       public void testModuleConflict() {
125         Compiler compiler = new Compiler();
126         compiler.initOptions(new CompilerOptions());
127         JsAst ast = new JsAst(SourceFile.fromCode("file.js", "export let foo = 42;"));
128         SimpleDependencyInfo delegate =
129             new SimpleDependencyInfo("", "my/js.js", EMPTY, EMPTY, ImmutableMap.of("module", "goog"));
130         DependencyInfo info = new LazyParsedDependencyInfo(delegate, ast, compiler);
131     
132         assertThat(Arrays.asList(compiler.getErrorManager().getWarnings())).isEmpty();
133         assertThat(info.getLoadFlags()).isEqualTo(ImmutableMap.of("module", "es6", "lang", "es6"));
134         assertThat(Arrays.asList(compiler.getErrorManager().getWarnings()))
135             .containsExactly(JSError.make(LazyParsedDependencyInfo.MODULE_CONFLICT, "my/js.js"));
136       }
