79       /**
80        * Gets the boolean value of a node that represents a expression. This method
81        * effectively emulates the <code>Boolean()</code> JavaScript cast function.
82        * Note: unlike getPureBooleanValue this function does not return UNKNOWN
83        * for expressions with side-effects.
84        */
85       static TernaryValue getImpureBooleanValue(Node n) {
86         switch (n.getType()) {
87           case ASSIGN:
88           case COMMA:
89             // For ASSIGN and COMMA the value is the value of the RHS.
90             return getImpureBooleanValue(n.getLastChild());
91           case NOT:
92             TernaryValue value = getImpureBooleanValue(n.getLastChild());
93             return value.not();
94           case AND: {
95             TernaryValue lhs = getImpureBooleanValue(n.getFirstChild());
96             TernaryValue rhs = getImpureBooleanValue(n.getLastChild());
97             return lhs.and(rhs);
98           }
99           case OR:  {
100             TernaryValue lhs = getImpureBooleanValue(n.getFirstChild());
101             TernaryValue rhs = getImpureBooleanValue(n.getLastChild());
102             return lhs.or(rhs);
103           }
104           case HOOK:  {
105             TernaryValue trueValue = getImpureBooleanValue(
106                 n.getSecondChild());
107             TernaryValue falseValue = getImpureBooleanValue(n.getLastChild());
108             if (trueValue.equals(falseValue)) {
109               return trueValue;
110             } else {
111               return TernaryValue.UNKNOWN;
112             }
113           }
114           case NEW:
115           case ARRAYLIT:
116           case OBJECTLIT:
117             // ignoring side-effects
118             return TernaryValue.TRUE;
119     
120           case VOID:
121             return TernaryValue.FALSE;
122     
123           default:
124             return getPureBooleanValue(n);
125         }
126       }

128       /**
129        * Gets the boolean value of a node that represents a literal. This method
130        * effectively emulates the <code>Boolean()</code> JavaScript cast function
131        * except it return UNKNOWN for known values with side-effects, use
132        * getImpureBooleanValue if you don't care about side-effects.
133        */
134       static TernaryValue getPureBooleanValue(Node n) {
135         switch (n.getType()) {
136           case TEMPLATELIT:
137             if (n.hasOneChild()) {
138               return TernaryValue.forBoolean(!n.getFirstChild().getString().isEmpty());
139             }
140             break;
141     
142           case STRING:
143             return TernaryValue.forBoolean(n.getString().length() > 0);
144     
145           case NUMBER:
146             return TernaryValue.forBoolean(n.getDouble() != 0);
147     
148           case NOT:
149             return getPureBooleanValue(n.getLastChild()).not();
150     
151           case NULL:
152           case FALSE:
153             return TernaryValue.FALSE;
154     
155           case VOID:
156             if (!mayHaveSideEffects(n.getFirstChild())) {
157               return TernaryValue.FALSE;
158             }
159             break;
160     
161           case NAME:
162             String name = n.getString();
163             if ("undefined".equals(name)
164                 || "NaN".equals(name)) {
165               // We assume here that programs don't change the value of the keyword
166               // undefined to something other than the value undefined.
167               return TernaryValue.FALSE;
168             } else if ("Infinity".equals(name)) {
169               return TernaryValue.TRUE;
170             }
171             break;
172     
173           case TRUE:
174           case REGEXP:
175             return TernaryValue.TRUE;
176     
177           case FUNCTION:
178           case CLASS:
179           case NEW:
180           case ARRAYLIT:
181           case OBJECTLIT:
182             if (!mayHaveSideEffects(n)) {
183               return TernaryValue.TRUE;
184             }
185             break;
186         }
187     
188         return TernaryValue.UNKNOWN;
189       }

192        * Gets the value of a node as a String, or null if it cannot be converted.
193        * When it returns a non-null String, this method effectively emulates the
194        * <code>String()</code> JavaScript cast function.
195        */
196       public static String getStringValue(Node n) {
197         // TODO(user): regex literals as well.
198         switch (n.getType()) {
199           case STRING:
200           case STRING_KEY:
201             return n.getString();
202     
203           case NAME:
204             String name = n.getString();
205             if ("undefined".equals(name)
206                 || "Infinity".equals(name)
207                 || "NaN".equals(name)) {
208               return name;
209             }
210             break;
211     
212           case NUMBER:
213             return DToA.numberToString(n.getDouble());
214     
215           case FALSE:
216             return "false";
217     
218           case TRUE:
219             return "true";
220     
221           case NULL:
222             return "null";
223     
224           case VOID:
225             return "undefined";
226     
227           case NOT:
228             TernaryValue child = getPureBooleanValue(n.getFirstChild());
229             if (child != TernaryValue.UNKNOWN) {
230               return child.toBoolean(true) ? "false" : "true"; // reversed.
231             }
232             break;
233     
234           case ARRAYLIT:
235             return arrayToString(n);
236     
237           case OBJECTLIT:
238             return "[object Object]";
239         }
240         return null;
241       }

256       static String arrayToString(Node literal) {
257         Node first = literal.getFirstChild();
258         StringBuilder result = new StringBuilder();
259         for (Node n = first; n != null; n = n.getNext()) {
260           String childValue = getArrayElementStringValue(n);
261           if (childValue == null) {
262             return null;
263           }
264           if (n != first) {
265             result.append(',');
266           }
267           result.append(childValue);
268         }
269         return result.toString();
270       }

276       /**
277        * Gets the value of a node as a Number, or null if it cannot be converted.
278        * When it returns a non-null Double, this method effectively emulates the
279        * <code>Number()</code> JavaScript cast function.
280        *
281        * @param n The node.
282        * @param useType If true, return 0.0 if the type is null, and NaN if the type is undefined.
283        * @return The value of a node as a Number, or null if it cannot be converted.
284        */
285       static Double getNumberValue(Node n, boolean useType) {
286         switch (n.getType()) {
287           case TRUE:
288             return 1.0;
289     
290           case FALSE:
291           case NULL:
292             return 0.0;
293     
294           case NUMBER:
295             return n.getDouble();
296     
297           case VOID:
298             if (mayHaveSideEffects(n.getFirstChild())) {
299               return null;
300             } else {
301               return Double.NaN;
302             }
303     
304           case NAME:
305             // Check for known constants
306             String name = n.getString();
307             if (name.equals("undefined")) {
308               return Double.NaN;
309             }
310             if (name.equals("NaN")) {
311               return Double.NaN;
312             }
313             if (name.equals("Infinity")) {
314               return Double.POSITIVE_INFINITY;
315             }
316             if (useType) {
317               JSType type = n.getJSType();
318               if (type != null) {
319                 if (type.isVoidType()) {
320                   return Double.NaN;
321                 } else if (type.isNullType()) {
322                   return 0.0;
323                 }
324               }
325             }
326             return null;
327     
328           case NEG:
329             if (n.getChildCount() == 1 && n.getFirstChild().isName()
330                 && n.getFirstChild().getString().equals("Infinity")) {
331               return Double.NEGATIVE_INFINITY;
332             }
333             return null;
334     
335           case NOT:
336             TernaryValue child = getPureBooleanValue(n.getFirstChild());
337             if (child != TernaryValue.UNKNOWN) {
338               return child.toBoolean(true) ? 0.0 : 1.0; // reversed.
339             }
340             break;
341     
342           case STRING:
343             return getStringNumberValue(n.getString());
344     
345           case ARRAYLIT:
346           case OBJECTLIT:
347             String value = getStringValue(n);
348             return value != null ? getStringNumberValue(value) : null;
349         }
350     
351         return null;
352       }

354       static Double getStringNumberValue(String rawJsString) {
355         if (rawJsString.contains("\u000b")) {
356           // vertical tab is not always whitespace
357           return null;
358         }
359     
360         String s = trimJsWhiteSpace(rawJsString);
361         // return ScriptRuntime.toNumber(s);
362         if (s.isEmpty()) {
363           return 0.0;
364         }
365     
366         if (s.length() > 2
367             && s.charAt(0) == '0'
368             && (s.charAt(1) == 'x' || s.charAt(1) == 'X')) {
369           // Attempt to convert hex numbers.
370           try {
371             return Double.valueOf(Integer.parseInt(s.substring(2), 16));
372           } catch (NumberFormatException e) {
373             return Double.NaN;
374           }
375         }
376     
377         if (s.length() > 3
378             && (s.charAt(0) == '-' || s.charAt(0) == '+')
379             && s.charAt(1) == '0'
380             && (s.charAt(2) == 'x' || s.charAt(2) == 'X')) {
381           // hex numbers with explicit signs vary between browsers.
382           return null;
383         }
384     
385         // Firefox and IE treat the "Infinity" differently. Firefox is case
386         // insensitive, but IE treats "infinity" as NaN.  So leave it alone.
387         if (s.equals("infinity")
388             || s.equals("-infinity")
389             || s.equals("+infinity")) {
390           return null;
391         }
392     
393         try {
394           return Double.parseDouble(s);
395         } catch (NumberFormatException e) {
396           return Double.NaN;
397         }
398       }

400       static String trimJsWhiteSpace(String s) {
401         int start = 0;
402         int end = s.length();
403         while (end > 0
404             && TokenUtil.isStrWhiteSpaceChar(s.charAt(end - 1)) == TernaryValue.TRUE) {
405           end--;
406         }
407         while (start < end
408             && TokenUtil.isStrWhiteSpaceChar(s.charAt(start)) == TernaryValue.TRUE) {
409           start++;
410         }
411         return s.substring(start, end);
412       }

423       /**
424        * Gets the node of a function or class's name. This method recognizes five forms:
425        * <ul>
426        * <li>{@code class name {...}}</li>
427        * <li>{@code var name = class {...}}</li>
428        * <li>{@code qualified.name = class {...}}</li>
429        * <li>{@code var name2 = class name1 {...}}</li>
430        * <li>{@code qualified.name2 = class name1 {...}}</li>
431        * </ul>
432        * In two last cases with named function expressions, the second name is
433        * returned (the variable or qualified name).
434        *
435        * @param n A function or class node
436        * @return the node best representing the class's name
437        */
438       public static Node getNameNode(Node n) {
439         Preconditions.checkState(n.isFunction() || n.isClass());
440         Node parent = n.getParent();
441         switch (parent.getType()) {
442           case NAME:
443             // var name = function() ...
444             // var name2 = function name1() ...
445             return parent;
446     
447           case ASSIGN: {
448             // qualified.name = function() ...
449             // qualified.name2 = function name1() ...
450             Node firstChild = parent.getFirstChild();
451             return firstChild.isQualifiedName() ? firstChild : null;
452           }
453     
454           default:
455             // function name() ...
456             Node funNameNode = n.getFirstChild();
457             // Don't return the name node for anonymous functions/classes.
458             // TODO(tbreisacher): Currently we do two kinds of "empty" checks because
459             // anonymous classes have an EMPTY name node while anonymous functions
460             // have a STRING node with an empty string. Consider making these the same.
461             return (funNameNode.isEmpty() || funNameNode.getString().isEmpty())
462                 ? null : funNameNode;
463         }
464       }

467        * Gets the function's name. This method recognizes the forms:
468        * <ul>
469        * <li>{@code &#123;'name': function() ...&#125;}</li>
470        * <li>{@code &#123;name: function() ...&#125;}</li>
471        * <li>{@code function name() ...}</li>
472        * <li>{@code var name = function() ...}</li>
473        * <li>{@code var obj = {name() {} ...}}</li>
474        * <li>{@code qualified.name = function() ...}</li>
475        * <li>{@code var name2 = function name1() ...}</li>
476        * <li>{@code qualified.name2 = function name1() ...}</li>
477        * </ul>
478        *
479        * @param n a node whose type is {@link Token#FUNCTION}
480        * @return the function's name, or {@code null} if it has no name
481        */
482       public static String getNearestFunctionName(Node n) {
483         if (!n.isFunction()) {
484           return null;
485         }
486     
487         String name = getName(n);
488         if (name != null) {
489           return name;
490         }
491     
492         // Check for the form { 'x' : function() { }} and {x() {}}
493         Node parent = n.getParent();
494         switch (parent.getType()) {
495           case MEMBER_FUNCTION_DEF:
496           case SETTER_DEF:
497           case GETTER_DEF:
498           case STRING_KEY:
499             // Return the name of the literal's key.
500             return parent.getString();
501           case NUMBER:
502             return getStringValue(parent);
503         }
504     
505         return null;
506       }

516       static boolean isImmutableValue(Node n) {
517         // TODO(johnlenz): rename this function.  It is currently being used
518         // in two disjoint cases:
519         // 1) We only care about the result of the expression
520         //    (in which case NOT here should return true)
521         // 2) We care that expression is a side-effect free and can't
522         //    be side-effected by other expressions.
523         // This should only be used to say the value is immuable and
524         // hasSideEffects and canBeSideEffected should be used for the other case.
525     
526         switch (n.getType()) {
527           case STRING:

530           case TRUE:
531           case FALSE:
532             return true;
533           case CAST:
534           case NOT:
535           case VOID:
536           case NEG:
537             return isImmutableValue(n.getFirstChild());
538           case NAME:
539             String name = n.getString();

541             // undefined to something other than the value undefined.
542             return "undefined".equals(name) || "Infinity".equals(name) || "NaN".equals(name);
543         }
544     
545         return false;
546       }

612       /**
613        * Returns true if this is a literal value. We define a literal value
614        * as any node that evaluates to the same thing regardless of when or
615        * where it is evaluated. So /xyz/ and [3, 5] are literals, but
616        * the name a is not.
617        *
618        * <p>Function literals do not meet this definition, because they
619        * lexically capture variables. For example, if you have
620        * <code>
621        * function() { return a; }
622        * </code>
623        * If it is evaluated in a different scope, then it
624        * captures a different variable. Even if the function did not read
625        * any captured variables directly, it would still fail this definition,
626        * because it affects the lifecycle of variables in the enclosing scope.
627        *
628        * <p>However, a function literal with respect to a particular scope is
629        * a literal.
630        *
631        * @param includeFunctions If true, all function expressions will be
632        *     treated as literals.
633        */
634       public static boolean isLiteralValue(Node n, boolean includeFunctions) {
635         switch (n.getType()) {
636           case CAST:
637             return isLiteralValue(n.getFirstChild(), includeFunctions);
638     
639           case ARRAYLIT:
640             for (Node child = n.getFirstChild(); child != null;
641                  child = child.getNext()) {
642               if ((!child.isEmpty()) && !isLiteralValue(child, includeFunctions)) {
643                 return false;
644               }
645             }
646             return true;
647     
648           case REGEXP:
649             // Return true only if all descendants are const.
650             for (Node child = n.getFirstChild(); child != null;
651                  child = child.getNext()) {
652               if (!isLiteralValue(child, includeFunctions)) {
653                 return false;
654               }
655             }
656             return true;
657     
658           case OBJECTLIT:
659             // Return true only if all values are const.
660             for (Node child = n.getFirstChild(); child != null;
661                  child = child.getNext()) {
662               if (!isLiteralValue(child.getFirstChild(), includeFunctions)) {
663                 return false;
664               }
665             }
666             return true;
667     
668           case FUNCTION:
669             return includeFunctions && !NodeUtil.isFunctionDeclaration(n);
670     
671           default:
672             return isImmutableValue(n);
673         }
674       }

676       /**
677        * Determines whether the given value may be assigned to a define.
678        *
679        * @param val The value being assigned.
680        * @param defines The list of names of existing defines.
681        */
682       static boolean isValidDefineValue(Node val, Set<String> defines) {
683         switch (val.getType()) {
684           case STRING:
685           case NUMBER:
686           case TRUE:
687           case FALSE:
688             return true;
689     
690           // Binary operators are only valid if both children are valid.
691           case AND:
692           case OR:
693           case ADD:
694           case BITAND:
695           case BITNOT:
696           case BITOR:
697           case BITXOR:
698           case DIV:
699           case EQ:
700           case GE:
701           case GT:
702           case LE:
703           case LSH:
704           case LT:
705           case MOD:
706           case MUL:
707           case NE:
708           case RSH:
709           case SHEQ:
710           case SHNE:
711           case SUB:
712           case URSH:
713             return isValidDefineValue(val.getFirstChild(), defines)
714                 && isValidDefineValue(val.getLastChild(), defines);
715     
716           case HOOK:
717             return isValidDefineValue(val.getFirstChild(), defines)
718                 && isValidDefineValue(val.getSecondChild(), defines)
719                 && isValidDefineValue(val.getLastChild(), defines);
720     
721           // Unary operators are valid if the child is valid.
722           case NOT:
723           case NEG:
724           case POS:
725             return isValidDefineValue(val.getFirstChild(), defines);
726     
727           // Names are valid if and only if they are defines themselves.
728           case NAME:
729           case GETPROP:
730             if (val.isQualifiedName()) {
731               return defines.contains(val.getQualifiedName());
732             }
733         }
734         return false;
735       }

893       /**
894        * Returns true iff this node defines a namespace, e.g.,
895        *
896        * /** @const * / var goog = {};
897        * /** @const * / var goog = goog || {};
898        * /** @const * / goog.math = goog.math || {};
899        */
900       static boolean isNamespaceDecl(Node n) {
901         JSDocInfo jsdoc = getBestJSDocInfo(n);
902         if (jsdoc != null && !jsdoc.getTypeNodes().isEmpty()) {
903           return false;
904         }
905         // In externs, we allow namespace definitions without @const.
906         // This is a worse design than always requiring @const, but it helps with
907         // namespaces that are defined in many places, such as gapi.
908         // Also, omitting @const in externs is not as confusing as in source code,
909         // because assigning an object literal in externs only makes sense when
910         // defining a namespace or enum.
911         if (!n.isFromExterns()
912             && (jsdoc == null || !jsdoc.isConstant())) {
913           return false;
914         }
915         Node qnameNode;
916         Node initializer;
917         if (n.getParent().isVar()) {
918           qnameNode = n;
919           initializer = n.getFirstChild();
920         } else if (n.isExprResult()) {
921           Node expr = n.getFirstChild();
922           if (!expr.isAssign() || !expr.getFirstChild().isGetProp()) {
923             return false;
924           }
925           qnameNode = expr.getFirstChild();
926           initializer = expr.getLastChild();
927         } else if (n.isGetProp()) {
928           Node parent = n.getParent();
929           if (!parent.isAssign() || !parent.getParent().isExprResult()) {
930             return false;
931           }
932           qnameNode = n;
933           initializer = parent.getLastChild();
934         } else {
935           return false;
936         }
937         if (initializer == null || qnameNode == null) {
938           return false;
939         }
940         if (initializer.isObjectLit()) {
941           return true;
942         }
943         return initializer.isOr()
944             && qnameNode.matchesQualifiedName(initializer.getFirstChild())
945             && initializer.getLastChild().isObjectLit();
946       }

985       /**
986        * Returns true if some node in n's subtree changes application state.
987        * If {@code checkForNewObjects} is true, we assume that newly created
988        * mutable objects (like object literals) change state. Otherwise, we assume
989        * that they have no side effects.
990        */
991       private static boolean checkForStateChangeHelper(
992           Node n, boolean checkForNewObjects, AbstractCompiler compiler) {
993         // Rather than id which ops may have side effects, id the ones
994         // that we know to be safe
995         switch (n.getType()) {
996           // other side-effect free statements and expressions
997           // Throws are by definition side effects, and yields are similar.
998           case THROW:
999           case YIELD:
1000             return true;
1001     
1002           case OBJECTLIT:
1003             if (checkForNewObjects) {
1004               return true;
1005             }
1006             for (Node key = n.getFirstChild(); key != null; key = key.getNext()) {
1007               for (Node c = key.getFirstChild(); c != null; c = c.getNext()) {
1008                 if (checkForStateChangeHelper(c, checkForNewObjects, compiler)) {
1009                   return true;
1010                 }
1011               }
1012             }
1013             return false;
1014     
1015           case ARRAYLIT:
1016           case REGEXP:
1017             if (checkForNewObjects) {
1018               return true;
1019             }
1020             break;
1021     
1022           case VAR:    // empty var statement (no declaration)
1023           case NAME:   // variable by itself
1024             if (n.getFirstChild() != null) {
1025               return true;
1026             }
1027             break;
1028     
1029           case FUNCTION:
1030             // Function expressions don't have side-effects, but function
1031             // declarations change the namespace. Either way, we don't need to
1032             // check the children, since they aren't executed at declaration time.
1033             return checkForNewObjects || !isFunctionExpression(n);
1034     
1035           case CLASS:
1036             return checkForNewObjects || isClassDeclaration(n)
1037                 // Check the extends clause for side effects.
1038                 || checkForStateChangeHelper(n.getSecondChild(), checkForNewObjects, compiler)
1039                 // Check for class members that are computed properties with side effects.
1040                 || checkForStateChangeHelper(n.getLastChild(), checkForNewObjects, compiler);
1041     
1042           case CLASS_MEMBERS:
1043             for (Node member = n.getFirstChild(); member != null; member = member.getNext()) {
1044               if (member.isComputedProp()
1045                   && checkForStateChangeHelper(member.getFirstChild(), checkForNewObjects, compiler)) {
1046                 return true;
1047               }
1048             }
1049             return false;
1050     
1051           case NEW:
1052             if (checkForNewObjects) {
1053               return true;
1054             }
1055     
1056             if (!constructorCallHasSideEffects(n)) {
1057               // loop below will see if the constructor parameters have
1058               // side-effects
1059               break;
1060             }
1061             return true;
1062     
1063           case CALL:
1064             // calls to functions that have no side effects have the no
1065             // side effect property set.
1066             if (!functionCallHasSideEffects(n, compiler)) {
1067               // loop below will see if the function parameters have
1068               // side-effects
1069               break;
1070             }
1071             return true;
1072     
1073           case TAGGED_TEMPLATELIT:
1074             return functionCallHasSideEffects(n, compiler);
1075     
1076           case CAST:
1077           case AND:
1078           case BLOCK:
1079           case EXPR_RESULT:
1080           case HOOK:
1081           case IF:
1082           case IN:
1083           case PARAM_LIST:
1084           case NUMBER:
1085           case OR:
1086           case THIS:
1087           case TRUE:
1088           case FALSE:
1089           case NULL:
1090           case STRING:
1091           case STRING_KEY:
1092           case SWITCH:
1093           case TEMPLATELIT_SUB:
1094           case TRY:
1095           case EMPTY:
1096           case TEMPLATELIT:
1097             break;
1098     
1099           default:
1100             if (isSimpleOperator(n)) {
1101               break;
1102             }
1103     
1104             if (isAssignmentOp(n)) {
1105               Node assignTarget = n.getFirstChild();
1106               if (assignTarget.isName()) {
1107                 return true;
1108               }
1109     
1110               // Assignments will have side effects if
1111               // a) The RHS has side effects, or
1112               // b) The LHS has side effects, or
1113               // c) A name on the LHS will exist beyond the life of this statement.
1114               if (checkForStateChangeHelper(
1115                       n.getFirstChild(), checkForNewObjects, compiler) ||
1116                   checkForStateChangeHelper(
1117                       n.getLastChild(), checkForNewObjects, compiler)) {
1118                 return true;
1119               }
1120     
1121               if (isGet(assignTarget)) {
1122                 // If the object being assigned to is a local object, don't
1123                 // consider this a side-effect as it can't be referenced
1124                 // elsewhere.  Don't do this recursively as the property might
1125                 // be an alias of another object, unlike a literal below.
1126                 Node current = assignTarget.getFirstChild();
1127                 if (evaluatesToLocalValue(current)) {
1128                   return false;
1129                 }
1130     
1131                 // A literal value as defined by "isLiteralValue" is guaranteed
1132                 // not to be an alias, or any components which are aliases of
1133                 // other objects.
1134                 // If the root object is a literal don't consider this a
1135                 // side-effect.
1136                 while (isGet(current)) {
1137                   current = current.getFirstChild();
1138                 }
1139     
1140                 return !isLiteralValue(current, true);
1141               } else {
1142                 // TODO(johnlenz): remove this code and make this an exception. This
1143                 // is here only for legacy reasons, the AST is not valid but
1144                 // preserve existing behavior.
1145                 return !isLiteralValue(assignTarget, true);
1146               }
1147             }
1148     
1149             return true;
1150         }
1151     
1152         for (Node c = n.getFirstChild(); c != null; c = c.getNext()) {
1153           if (checkForStateChangeHelper(c, checkForNewObjects, compiler)) {
1154             return true;
1155           }
1156         }
1157     
1158         return false;
1159       }

1209        * @param callNode The call node to inspected.
1210        * @param compiler A compiler object to provide program state changing
1211        *     context information. Can be null.
1212        */
1213       static boolean functionCallHasSideEffects(
1214           Node callNode, @Nullable AbstractCompiler compiler) {
1215         Preconditions.checkState(callNode.isCall() || callNode.isTaggedTemplateLit(), callNode);
1216     
1217         if (callNode.isNoSideEffectsCall()) {
1218           return false;
1219         }
1220     
1221         if (callNode.isOnlyModifiesArgumentsCall()
1222             && allArgsUnescapedLocal(callNode)) {
1223           return false;
1224         }
1225     
1226         Node nameNode = callNode.getFirstChild();
1227     
1228         // Built-in functions with no side effects.
1229         if (nameNode.isName()) {
1230           String name = nameNode.getString();
1231           if (BUILTIN_FUNCTIONS_WITHOUT_SIDEEFFECTS.contains(name)) {
1232             return false;
1233           }
1234         } else if (nameNode.isGetProp()) {
1235           if (callNode.hasOneChild()
1236               && OBJECT_METHODS_WITHOUT_SIDEEFFECTS.contains(
1237                     nameNode.getLastChild().getString())) {
1238             return false;
1239           }
1240     
1241           if (callNode.isOnlyModifiesThisCall()
1242               && evaluatesToLocalValue(nameNode.getFirstChild())) {
1243             return false;
1244           }
1245     
1246           // Many common Math functions have no side-effects.
1247           // TODO(nicksantos): This is a terrible terrible hack, until
1248           // I create a definitionProvider that understands namespacing.
1249           if (nameNode.getFirstChild().isName() && nameNode.isQualifiedName()
1250               && nameNode.getFirstChild().getString().equals("Math")) {
1251             switch(nameNode.getLastChild().getString()) {
1252               case "abs":
1253               case "acos":
1254               case "acosh":
1255               case "asin":
1256               case "asinh":
1257               case "atan":
1258               case "atanh":
1259               case "atan2":
1260               case "cbrt":
1261               case "ceil":
1262               case "cos":
1263               case "cosh":
1264               case "exp":
1265               case "expm1":
1266               case "floor":
1267               case "hypot":
1268               case "log":
1269               case "log10":
1270               case "log1p":
1271               case "log2":
1272               case "max":
1273               case "min":
1274               case "pow":
1275               case "round":
1276               case "sign":
1277               case "sin":
1278               case "sinh":
1279               case "sqrt":
1280               case "tan":
1281               case "tanh":
1282               case "trunc":
1283                 return false;
1284               case "random":
1285                 return !callNode.hasOneChild(); // no parameters
1286             }
1287           }
1288     
1289           if (compiler != null && !compiler.hasRegExpGlobalReferences()) {
1290             if (nameNode.getFirstChild().isRegExp()
1291                 && REGEXP_METHODS.contains(nameNode.getLastChild().getString())) {
1292               return false;
1293             } else if (nameNode.getFirstChild().isString()) {
1294               String method = nameNode.getLastChild().getString();

1299                     return false;
1300                   }

1302                   if ("replace".equals(method)) {
1303                     // Assume anything but a string constant has side-effects
1304                     return !param.getNext().isString();
1305                   } else if (STRING_REGEXP_METHODS.contains(method)) {
1306                     return false;
1307                   }
1308                 }
1309               }
1310             }
1311           }
1312         }
1313     
1314         return true;
1315       }

1431       /*
1432        *  0 comma ,
1433        *  1 assignment = += -= *= /= %= <<= >>= >>>= &= ^= |=
1434        *  2 conditional ?:
1435        *  3 logical-or ||
1436        *  4 logical-and &&
1437        *  5 bitwise-or |
1438        *  6 bitwise-xor ^
1439        *  7 bitwise-and &
1440        *  8 equality == !=
1441        *  9 relational < <= > >=
1442        * 10 bitwise shift << >> >>>
1443        * 11 addition/subtraction + -
1444        * 12 multiply/divide * / %
1445        * 13 negation/increment ! ~ - ++ --
1446        * 14 call, member () [] .
1447        */
1448       static int precedence(Token type) {
1449         switch (type) {
1450           case COMMA:  return 0;
1451           case ASSIGN_BITOR:
1452           case ASSIGN_BITXOR:
1453           case ASSIGN_BITAND:
1454           case ASSIGN_LSH:
1455           case ASSIGN_RSH:
1456           case ASSIGN_URSH:
1457           case ASSIGN_ADD:
1458           case ASSIGN_SUB:
1459           case ASSIGN_MUL:
1460           case ASSIGN_DIV:
1461           case ASSIGN_MOD:
1462           case ASSIGN: return 1;
1463           case YIELD:  return 2;
1464           case HOOK:   return 3;  // ?: operator
1465           case OR:     return 4;
1466           case AND:    return 5;
1467           case BITOR:  return 6;
1468           case BITXOR: return 7;
1469           case BITAND: return 8;
1470           case EQ:

1622           case COMMA:
1623             return getKnownValueType(n.getLastChild());
1624           case AND:

1627                 getKnownValueType(n.getFirstChild()),
1628                 getKnownValueType(n.getLastChild()));
1629           case HOOK:
1630             return and(
1631                 getKnownValueType(n.getSecondChild()),
1632                 getKnownValueType(n.getLastChild()));
1633     
1634           case ADD: {
1635             ValueType last = getKnownValueType(n.getLastChild());
1636             if (last == ValueType.STRING) {
1637               return ValueType.STRING;
1638             }
1639             ValueType first = getKnownValueType(n.getFirstChild());
1640             if (first == ValueType.STRING) {
1641               return ValueType.STRING;
1642             }
1643     
1644             // There are some pretty weird cases for object types:
1645             //   {} + [] === "0"
1646             //   [] + {} ==== "[object Object]"
1647             if (first == ValueType.OBJECT || last == ValueType.OBJECT) {
1648               return ValueType.UNDETERMINED;
1649             }
1650     
1651             if (!mayBeString(first) && !mayBeString(last)) {
1652               // ADD used with compilations of null, undefined, boolean and number always result
1653               // in numbers.
1654               return ValueType.NUMBER;

1657             // There are some pretty weird cases for object types:
1658             //   {} + [] === "0"
1659             //   [] + {} ==== "[object Object]"

1661             return ValueType.UNDETERMINED;
1662           }
1663     
1664           case ASSIGN_ADD: {

1672           case NAME:

1675               return ValueType.VOID;
1676             }
1677             if (name.equals("NaN")) {
1678               return ValueType.NUMBER;

1682             }

2489       /**
2490        * Safely remove children while maintaining a valid node structure.
2491        * In some cases, this is done by removing the parent from the AST as well.
2492        */
2493       public static void removeChild(Node parent, Node node) {
2494         if (isTryFinallyNode(parent, node)) {
2495           if (NodeUtil.hasCatchHandler(getCatchBlock(parent))) {
2496             // A finally can only be removed if there is a catch.
2497             parent.removeChild(node);
2498           } else {
2499             // Otherwise, only its children can be removed.
2500             node.detachChildren();
2501           }
2502         } else if (node.isCatch()) {
2503           // The CATCH can can only be removed if there is a finally clause.
2504           Node tryNode = node.getGrandparent();
2505           Preconditions.checkState(NodeUtil.hasFinally(tryNode));
2506           node.detachFromParent();
2507         } else if (isTryCatchNodeContainer(node)) {
2508           // The container node itself can't be removed, but the contained CATCH
2509           // can if there is a 'finally' clause
2510           Node tryNode = node.getParent();
2511           Preconditions.checkState(NodeUtil.hasFinally(tryNode));
2512           node.detachChildren();
2513         } else if (node.isBlock()) {
2514           // Simply empty the block.  This maintains source location and
2515           // "synthetic"-ness.
2516           node.detachChildren();
2517         } else if (isStatementBlock(parent)
2518             || isSwitchCase(node)) {
2519           // A statement in a block can simply be removed.
2520           parent.removeChild(node);
2521         } else if (parent.isVar() || parent.isExprResult()) {
2522           if (parent.hasMoreThanOneChild()) {
2523             parent.removeChild(node);
2524           } else {
2525             // Remove the node from the parent, so it can be reused.
2526             parent.removeChild(node);
2527             // This would leave an empty VAR, remove the VAR itself.
2528             removeChild(parent.getParent(), parent);
2529           }
2530         } else if (parent.isLabel()
2531             && node == parent.getLastChild()) {
2532           // Remove the node from the parent, so it can be reused.
2533           parent.removeChild(node);
2534           // A LABEL without children can not be referred to, remove it.
2535           removeChild(parent.getParent(), parent);
2536         } else if (parent.isFor()
2537             && parent.getChildCount() == 4) {
2538           // Only Token.FOR can have an Token.EMPTY other control structure
2539           // need something for the condition. Others need to be replaced
2540           // or the structure removed.
2541           parent.replaceChild(node, IR.empty());
2542         } else {
2543           throw new IllegalStateException("Invalid attempt to remove node: " + node + " of " + parent);
2544         }
2545       }

3958        * @return Whether the node is known to be a value that is not a reference
3959        *     outside the expression scope.
3960        */
3961       static boolean evaluatesToLocalValue(Node value, Predicate<Node> locals) {
3962         switch (value.getType()) {
3963           case CAST:
3964             return evaluatesToLocalValue(value.getFirstChild(), locals);
3965           case ASSIGN:
3966             // A result that is aliased by a non-local name, is the effectively the
3967             // same as returning a non-local name, but this doesn't matter if the
3968             // value is immutable.
3969             return NodeUtil.isImmutableValue(value.getLastChild())
3970                 || (locals.apply(value)
3971                     && evaluatesToLocalValue(value.getLastChild(), locals));
3972           case COMMA:
3973             return evaluatesToLocalValue(value.getLastChild(), locals);
3974           case AND:
3975           case OR:
3976             return evaluatesToLocalValue(value.getFirstChild(), locals)
3977                && evaluatesToLocalValue(value.getLastChild(), locals);
3978           case HOOK:
3979             return evaluatesToLocalValue(value.getSecondChild(), locals)
3980                && evaluatesToLocalValue(value.getLastChild(), locals);
3981           case INC:
3982           case DEC:
3983             return true;
3984           case THIS:
3985             return locals.apply(value);
3986           case NAME:
3987             return isImmutableValue(value) || locals.apply(value);
3988           case GETELEM:
3989           case GETPROP:
3990             // There is no information about the locality of object properties.
3991             return locals.apply(value);
3992           case CALL:
3993             return callHasLocalResult(value)
3994                 || isToStringMethodCall(value)
3995                 || locals.apply(value);
3996           case NEW:
3997             return newHasLocalResult(value)
3998                    || locals.apply(value);
3999           case FUNCTION:
4000           case REGEXP:
4001           case ARRAYLIT:
4002           case OBJECTLIT:
4003             // Literals objects with non-literal children are allowed.
4004             return true;
4005           case DELPROP:
4006           case IN:
4007             // TODO(johnlenz): should IN operator be included in #isSimpleOperator?
4008             return true;
4009           default:
4010             // Other op force a local value:
4011             //  x = '' + g (x is now an local string)
4012             //  x -= g (x is now an local number)
4013             if (isAssignmentOp(value)
4014                 || isSimpleOperator(value)
4015                 || isImmutableValue(value)) {
4016               return true;
4017             }

4019             throw new IllegalStateException(
4020                 "Unexpected expression node" + value +
4021                 "\n parent:" + value.getParent());

4359       /** Checks that the scope roots marked as changed have indeed changed */
4360       public static void verifyScopeChanges(Map<Node, Node> map, Node main,
4361           boolean verifyUnchangedNodes) {
4362         // compiler is passed only to call compiler.toSource during debugging to see
4363         // mismatches in scopes
4364     
4365         // If verifyUnchangedNodes is false, we are comparing the initial AST to the
4366         // final AST. Don't check unmarked nodes b/c they may have been changed by
4367         // non-loopable passes.
4368         // If verifyUnchangedNodes is true, we are comparing the ASTs before & after
4369         // a pass. Check all scope roots.
4370         final Map<Node, Node> mtoc = map;
4371         final boolean checkUnchanged = verifyUnchangedNodes;
4372         Node clone = mtoc.get(main);
4373         if (main.getChangeTime() > clone.getChangeTime()) {
4374           Preconditions.checkState(!isEquivalentToExcludingFunctions(main, clone));
4375         } else if (checkUnchanged) {
4376           Preconditions.checkState(isEquivalentToExcludingFunctions(main, clone));
4377         }
4378         visitPreOrder(main,
4379             new Visitor() {
4380               @Override
4381               public void visit(Node n) {
4382                 if (n.isFunction() && mtoc.containsKey(n)) {
4383                   Node clone = mtoc.get(n);
4384                   if (n.getChangeTime() > clone.getChangeTime()) {
4385                     Preconditions.checkState(
4386                         !isEquivalentToExcludingFunctions(n, clone));
4387                   } else if (checkUnchanged) {
4388                     Preconditions.checkState(
4389                         isEquivalentToExcludingFunctions(n, clone));
4390                   }
4391                 }
4392               }
4393             },
4394             Predicates.<Node>alwaysTrue());
4395       }

4417       /**
4418        * @return Whether the two node are equivalent while ignoring
4419        * differences any descendant functions differences.
4420        */
4421       private static boolean isEquivalentToExcludingFunctions(
4422           Node thisNode, Node thatNode) {
4423         if (thisNode == null || thatNode == null) {
4424           return thisNode == null && thatNode == null;
4425         }
4426         if (!thisNode.isEquivalentToShallow(thatNode)) {
4427           return false;
4428         }
4429         if (thisNode.getChildCount() != thatNode.getChildCount()) {
4430           return false;
4431         }
4432         Node thisChild = thisNode.getFirstChild();
4433         Node thatChild = thatNode.getFirstChild();
4434         while (thisChild != null && thatChild != null) {
4435           if (thisChild.isFunction()) {
4436             //  don't compare function name, parameters or bodies.
4437             return thatChild.isFunction();
4438           }
4439           if (!isEquivalentToExcludingFunctions(thisChild, thatChild)) {
4440             return false;
4441           }
4442           thisChild = thisChild.getNext();
4443           thatChild = thatChild.getNext();
4444         }
4445         return true;
4446       }

4454       static int toInt32(double d) {
4455         int id = (int) d;
4456         if (id == d) {
4457             // This covers -0.0 as well
4458             return id;
4459         }
4460     
4461         if (Double.isNaN(d)
4462             || d == Double.POSITIVE_INFINITY
4463             || d == Double.NEGATIVE_INFINITY) {
4464             return 0;
4465         }
4466     
4467         d = (d >= 0) ? Math.floor(d) : Math.ceil(d);
4468     
4469         double two32 = 4294967296.0;
4470         d = d % two32;
4471         // (double)(long)d == d should hold here
4472     
4473         long l = (long) d;
4474         // returning (int)d does not work as d can be outside int range
4475         // but the result must always be 32 lower bits of l
4476         return (int) l;
4477       }
