38     /**
39      * An abstract compiler, to help remove the circular dependency of
40      * passes on JSCompiler.
41      *
42      * This is an abstract class, so that we can make the methods package-private.
43      *
44      * @author nicksantos@google.com (Nick Santos)
45      */
46     public abstract class AbstractCompiler implements SourceExcerptProvider {
47       static final DiagnosticType READ_ERROR = DiagnosticType.error(
48           "JSC_READ_ERROR", "Cannot read: {0}");

50       /**
51        * Will be called before each pass runs.

53       abstract void beforePass(String passName);

55       /**
56        * Will be called after each pass finishes.

58       abstract void afterPass(String passName);

331       static enum LifeCycleStage {
332         RAW,
333     
334         // See constraints put on the tree by Normalize.java
335         NORMALIZED,
336     
337         // The normalize pass has put constraints on the tree,
338         // but variables and properties have been renamed so
339         // coding conventions no longer apply.
340         NORMALIZED_OBFUSCATED;
341     
342         boolean isNormalized() {
343           return this == NORMALIZED || this == NORMALIZED_OBFUSCATED;
344         }
345     
346         boolean isNormalizedUnobfuscated() {
347           return this == NORMALIZED;
348         }
349     
350         boolean isNormalizedObfuscated() {
351           return this == NORMALIZED_OBFUSCATED;
352         }
353       }

435        * The subdir js/ contains libraries of code that we inject
