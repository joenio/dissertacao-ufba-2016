16     package com.google.javascript.jscomp;
17     
18     import com.google.common.annotations.GwtIncompatible;
19     import com.google.javascript.jscomp.NodeTraversal.AbstractPostOrderCallback;
20     import com.google.javascript.rhino.Node;
21     
22     import java.util.regex.Matcher;
23     import java.util.regex.Pattern;
24     
25     /**
26      * Ensures string literals matching certain patterns are only used as
27      * goog.getCssName parameters.
28      *
29      * @author mkretzschmar@google.com (Martin Kretzschmar)
30      */
31     @GwtIncompatible("java.util.regex")
32     class CheckMissingGetCssName
33         extends AbstractPostOrderCallback implements CompilerPass {
34       private final AbstractCompiler compiler;
35       private final CheckLevel level;
36       private final Matcher blacklist;
37     
38       static final String GET_CSS_NAME_FUNCTION = "goog.getCssName";
39       static final String GET_UNIQUE_ID_FUNCTION = ".getUniqueId";
40     
41       static final DiagnosticType MISSING_GETCSSNAME =
42           DiagnosticType.disabled(
43               "JSC_MISSING_GETCSSNAME",
44               "missing goog.getCssName around literal ''{0}''");
45     
46       CheckMissingGetCssName(AbstractCompiler compiler, CheckLevel level,
47           String blacklistRegex) {
48         this.compiler = compiler;
49         this.level = level;
50         this.blacklist =
51             Pattern.compile("\\b(?:" + blacklistRegex + ")").matcher("");
52       }
53     
54       @Override
55       public void process(Node externs, Node root) {
56         NodeTraversal.traverseEs6(compiler, root, this);
57       }
58     
59       @Override
60       public void visit(NodeTraversal t, Node n, Node parent) {
61         if (n.isString() &&
62             !parent.isGetProp() &&
63             !parent.isRegExp()) {
64           String s = n.getString();
65     
66           for (blacklist.reset(s); blacklist.find();) {
67             if (parent.isTemplateLit()) {
68               if (parent.getChildCount() > 1) {
69                 // Ignore template string with substitutions
70                 continue;
71               } else {
72                 n = parent;
73               }
74             }
75             if (insideGetCssNameCall(n)) {
76               continue;
77             }
78             if (insideGetUniqueIdCall(n)) {
79               continue;
80             }
81             if (insideAssignmentToIdConstant(n)) {
82               continue;
83             }
84             compiler.report(t.makeError(n, level, MISSING_GETCSSNAME,
85                     blacklist.group()));
86           }
87         }
88       }
89     
90       /** Returns whether the node is an argument of a goog.getCssName call. */
91       private static boolean insideGetCssNameCall(Node n) {
92         Node parent = n.getParent();
93         return parent.isCall() &&
94             parent.getFirstChild().matchesQualifiedName(GET_CSS_NAME_FUNCTION);
95       }
96     
97       /**
98        * Returns whether the node is an argument of a function that returns
99        * a unique id (the last part of the qualified name matches
100        * GET_UNIQUE_ID_FUNCTION).
101        */
102       private static boolean insideGetUniqueIdCall(Node n) {
103         Node parent = n.getParent();
104         String name = parent.isCall() ?
105             parent.getFirstChild().getQualifiedName() : null;
106     
107         return name != null && name.endsWith(GET_UNIQUE_ID_FUNCTION);
108       }
109     
110       /**
111        * Returns whether the node is the right hand side of an assignment or
112        * initialization of a variable named *_ID of *_ID_.
113        */
114       private boolean insideAssignmentToIdConstant(Node n) {
115         Node parent = n.getParent();
116         if (parent.isAssign()) {
117           String qname = parent.getFirstChild().getQualifiedName();
118           return qname != null && isIdName(qname);
119         } else if (parent.isName()) {
120           Node grandParent = parent.getParent();
121           if (grandParent != null && NodeUtil.isNameDeclaration(grandParent)) {
122             String name = parent.getString();
123             return isIdName(name);
124           } else {
125             return false;
126           }
127         } else {
128           return false;
129         }
130       }
131     
132       private static boolean isIdName(String name) {
133         return name.endsWith("ID") || name.endsWith("ID_");
134       }
135     }
