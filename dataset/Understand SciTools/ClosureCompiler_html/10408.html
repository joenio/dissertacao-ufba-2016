35     /**
36      * Primary driver of a refactoring. This class collects the inputs, runs the refactoring over
37      * the compiled input, and then collects the suggested fixes based on the refactoring.
38      *
39      * @author mknichel@google.com (Mark Knichel)
40      */
41     public final class RefactoringDriver {
42     
43       private final Scanner scanner;
44       private final Compiler compiler;
45       private final Node rootNode;
46     
47       private RefactoringDriver(
48           Scanner scanner,
49           List<SourceFile> inputs,
50           List<SourceFile> externs,
51           CompilerOptions compilerOptions) {
52         this.scanner = scanner;
53         this.compiler = createCompiler(inputs, externs, compilerOptions);
54         this.rootNode = this.compiler.getRoot();
55       }
56     
57       /**
58        * Run the refactoring and return any suggested fixes as a result.
59        */
60       public List<SuggestedFix> drive() {
61         JsFlumeCallback callback = new JsFlumeCallback(scanner, null);
62         NodeTraversal.traverseEs6(compiler, rootNode, callback);
63         List<SuggestedFix> fixes = callback.getFixes();
64         fixes.addAll(scanner.processAllMatches(callback.getMatches()));
65         return fixes;
66       }
67     
68       public Compiler getCompiler() {
69         return compiler;
70       }
71     
72       private Compiler createCompiler(
73           List<SourceFile> inputs, List<SourceFile> externs, CompilerOptions compilerOptions) {
74         Compiler compiler = new Compiler();
75         compiler.disableThreads();
76         compiler.compile(externs, inputs, compilerOptions);
77         return compiler;
78       }
79     
80       // TODO(tbreisacher): Make this package-private by refactoring tests so they
81       // don't need to call it directly.
82       @VisibleForTesting
83       public static CompilerOptions getCompilerOptions() {
84         CompilerOptions options = new CompilerOptions();
85     
86         DependencyOptions deps = new DependencyOptions();
87         deps.setDependencySorting(true);
88         options.setDependencyOptions(deps);
89     
90         options.setIdeMode(true);
91         options.setCheckSuspiciousCode(true);
92         options.setCheckSymbols(true);
93         options.setCheckTypes(true);
94         options.setClosurePass(true);
95         options.setGenerateExports(true);
96         options.setPreserveGoogRequires(true);
97     
98         options.setWarningLevel(DiagnosticGroups.MISSING_REQUIRE, CheckLevel.ERROR);
99     
100         return options;
101       }
102     
103       public static class Builder {
104         private static final Function<String, SourceFile> TO_SOURCE_FILE_FN =
105             new Function<String, SourceFile>() {
106               @Override public SourceFile apply(String file) {
107                 return new SourceFile.Builder().buildFromFile(file);
108               }
109             };
110     
111         private final Scanner scanner;
112         private final ImmutableList.Builder<SourceFile> inputs = ImmutableList.builder();
113         private final ImmutableList.Builder<SourceFile> externs = ImmutableList.builder();
114         private CompilerOptions compilerOptions = getCompilerOptions();
115     
116         public Builder(Scanner scanner) {
117           this.scanner = scanner;
118         }
119     
120         public Builder addExternsFromFile(String filename) {
121           externs.add(SourceFile.fromFile(filename));
122           return this;
123         }
124     
125         public Builder addExternsFromCode(String code) {
126           externs.add(SourceFile.fromCode("externs", code));
127           return this;
128         }
129     
130         public Builder addExterns(Iterable<SourceFile> externs) {
131           this.externs.addAll(externs);
132           return this;
133         }
134     
135         public Builder addExternsFromFile(Iterable<String> externs) {
136           this.externs.addAll(Lists.transform(ImmutableList.copyOf(externs), TO_SOURCE_FILE_FN));
137           return this;
138         }
139     
140         public Builder addInputsFromFile(String filename) {
141           inputs.add(SourceFile.fromFile(filename));
142           return this;
143         }
144     
145         public Builder addInputsFromCode(String code) {
146           return addInputsFromCode(code, "input");
147         }
148     
149         public Builder addInputsFromCode(String code, String filename) {
150           inputs.add(SourceFile.fromCode(filename, code));
151           return this;
152         }
153     
154         public Builder addInputs(Iterable<SourceFile> inputs) {
155           this.inputs.addAll(inputs);
156           return this;
157         }
158     
159         public Builder addInputsFromFile(Iterable<String> inputs) {
160           this.inputs.addAll(Lists.transform(ImmutableList.copyOf(inputs), TO_SOURCE_FILE_FN));
161           return this;
162         }
163     
164         public Builder withCompilerOptions(CompilerOptions compilerOptions) {
165           this.compilerOptions = Preconditions.checkNotNull(compilerOptions);
166           return this;
167         }
168     
169         public RefactoringDriver build() {
170           return new RefactoringDriver(scanner, inputs.build(), externs.build(), compilerOptions);
171         }
172       }
173     }
