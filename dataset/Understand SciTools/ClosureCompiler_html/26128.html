30     /**
31      * Optimize function calls and function signatures.
32      *
33      * <ul>
34      * <li>Removes optional parameters if no caller specifies it as argument.</li>
35      * <li>Removes arguments at call site to function that
36      *     ignores the parameter.</li>
37      * <li>Inline a parameter if the function is always called with that constant.
38      *     </li>
39      * </ul>
40      *
41      * @author johnlenz@google.com (John Lenz)
42      */
43     class OptimizeParameters
44         implements CompilerPass, OptimizeCalls.CallGraphCompilerPass {
45     
46       private final AbstractCompiler compiler;
47       private final List<Node> removedNodes = new ArrayList<>();
48     
49       OptimizeParameters(AbstractCompiler compiler) {
50         this.compiler = compiler;
51       }

53       @Override
54       @VisibleForTesting
55       public void process(Node externs, Node root) {
56         Preconditions.checkState(
57             compiler.getLifeCycleStage() == LifeCycleStage.NORMALIZED);
58         SimpleDefinitionFinder defFinder = new SimpleDefinitionFinder(compiler);
59         defFinder.process(externs, root);
60         process(externs, root, defFinder);
61       }
62     
63       @Override
64       public void process(
65           Node externs, Node root, SimpleDefinitionFinder definitions) {
66         for (DefinitionSite defSite : definitions.getDefinitionSites()) {
67           if (canChangeSignature(defSite, definitions)) {
68             tryEliminateConstantArgs(defSite, definitions);
69             tryEliminateOptionalArgs(defSite, definitions);
70           }
71         }
72     
73         // Remove any references or definitions that have been removed to keep it
74         // in a consistent state for the next pass.
75         for (Node n : removedNodes) {
76           definitions.removeReferences(n);
77         }
78       }

80       /**
81        * @return Whether the definitionSite represents a function whose call
82        *      signature can be modified.
83        */
84       private static boolean canChangeSignature(
85           DefinitionSite definitionSite, SimpleDefinitionFinder defFinder) {
86         Definition definition = definitionSite.definition;
87     
88         if (definitionSite.inExterns) {
89           return false;
90         }
91     
92         // Only functions may be rewritten.
93         // Functions that access "arguments" are not eligible since
94         // rewrite changes the structure of this object.
95         Node rValue = definition.getRValue();
96         if (rValue == null ||
97             !rValue.isFunction() ||
98             NodeUtil.isVarArgsFunction(rValue)) {
99           return false;
100         }
101     
102         // We don't want to re-write $jscomp.inherits to not stop recognizing
103         // 'inherits' calls. (b/27244988)
104         Node lValue = definition.getLValue();
105         if (lValue.matchesQualifiedName("$jscomp.inherits")
106             || lValue.matchesQualifiedName("$jscomp$inherits")) {
107           return false;
108         }
109     
110         // TODO(johnlenz): support rewriting methods defined as part of
111         // object literals (they are generally problematic because they may be
112         // maps of functions use in for-in expressions, etc).
113         // Be conservative, don't try to optimize any declaration that isn't as
114         // simple function declaration or assignment.
115         if (!SimpleDefinitionFinder.isSimpleFunctionDeclaration(rValue)) {
116           return false;
117         }
118     
119         // Assume an exported method result is used.
120         if (!defFinder.canModifyDefinition(definition)) {
121           return false;
122         }
123     
124         Collection<UseSite> useSites = defFinder.getUseSites(definition);
125     
126         if (useSites.isEmpty()) {
127           return false;
128         }
129     
130         for (UseSite site : useSites) {
131           // Any non-call reference maybe introducing an alias. Don't try to
132           // change the function signature, if all the aliases can't also be
133           // changed.
134           // TODO(johnlenz): Support .call signature changes.
135           if (!SimpleDefinitionFinder.isCallOrNewSite(site)) {
136             return false;
137           }
138     
139           // TODO(johnlenz): support specialization
140     
141           // Multiple definitions prevent rewrite.
142           // TODO(johnlenz): Allow rewrite all definitions are valid.
143           Node nameNode = site.node;
144           Collection<Definition> singleSiteDefinitions =
145               defFinder.getDefinitionsReferencedAt(nameNode);
146           if (singleSiteDefinitions.size() > 1) {
147             return false;
148           }
149           Preconditions.checkState(!singleSiteDefinitions.isEmpty());
150           Preconditions.checkState(singleSiteDefinitions.contains(definition));
151         }
152     
153         return true;
154       }

156       /**
157        * Removes any optional parameters if no callers specifies it as an argument.
158        */
159       private void tryEliminateOptionalArgs(
160           DefinitionSite defSite, SimpleDefinitionFinder defFinder) {
161         // Count the maximum number of arguments passed into this function all
162         // all points of the program.
163         int maxArgs = -1;
164     
165         Definition definition = defSite.definition;
166         Collection<UseSite> useSites = defFinder.getUseSites(definition);
167         for (UseSite site : useSites) {
168           Preconditions.checkState(SimpleDefinitionFinder.isCallOrNewSite(site));
169           Node call = site.node.getParent();
170     
171           int numArgs = call.getChildCount() - 1;
172           if (numArgs > maxArgs) {
173             maxArgs = numArgs;
174           }
175         }
176     
177         eliminateParamsAfter(definition.getRValue(), maxArgs);
178       }

180       /**
181        * Eliminate parameters if they are always constant.
182        *
183        * function foo(a, b) {...}
184        * foo(1,2);
185        * foo(1,3)
186        * becomes
187        * function foo(b) { var a = 1 ... }
188        * foo(2);
189        * foo(3);
190        */
191       private void tryEliminateConstantArgs(
192           DefinitionSite defSite, SimpleDefinitionFinder defFinder) {
193     
194         List<Parameter> parameters = new ArrayList<>();
195         boolean firstCall = true;
196     
197         // Build a list of parameters to remove
198         Definition definition = defSite.definition;
199         Collection<UseSite> useSites = defFinder.getUseSites(definition);
200         boolean continueLooking = false;
201         for (UseSite site : useSites) {
202           Preconditions.checkState(SimpleDefinitionFinder.isCallOrNewSite(site));
203           Node call = site.node.getParent();
204     
205           Node cur = call.getFirstChild();
206           if (firstCall) {
207             // Use the first call to construct a list of parameters of the
208             // function.
209             continueLooking = buildParameterList(parameters, cur, site.scope);
210             firstCall = false;
211           } else {
212             continueLooking = findFixedParameters(parameters, cur);
213           }
214           if (!continueLooking) {
215             return;
216           }
217         }
218     
219         continueLooking = adjustForSideEffects(parameters);
220         if (!continueLooking) {
221           return;
222         }
223     
224         // Remove the constant parameters in all the calls
225         for (UseSite site : useSites) {
226           Preconditions.checkState(SimpleDefinitionFinder.isCallOrNewSite(site));
227           Node call = site.node.getParent();
228     
229           optimizeCallSite(defFinder, parameters, call);
230         }
231     
232         // Remove the constant parameters in the definitions and add it as a local
233         // variable.
234         Node function = definition.getRValue();
235         if (function.isFunction()) {
236           optimizeFunctionDefinition(parameters, function);
237         }
238       }

240       /**
241        * Adjust the parameters to move based on the side-effects seen.
242        * @return Whether there are any movable parameters.
243        */
244       private static boolean adjustForSideEffects(List<Parameter> parameters) {
245         // If a parameter is moved, that has side-effect no parameters that
246         // can be effected by side-effects can be left.
247     
248         // A parameter can be moved if it can't be side-effected (immutable),
249         // or there are no following side-effects, that aren't moved.
250     
251         boolean anyMovable = false;
252         boolean seenUnmovableSideEffects = false;
253         boolean seenUnmoveableSideEfffected = false;
254         for (int i = parameters.size() - 1; i >= 0; i--) {
255           Parameter current = parameters.get(i);
256     
257           // Preserve side-effect ordering, don't move this parameter if:
258           // * the current parameter has side-effects and a following
259           // parameters that will not be move can be effected.
260           // * the current parameter can be effected and a following
261           // parameter that will not be moved has side-effects
262     
263           if (current.shouldRemove
264               && ((seenUnmovableSideEffects && current.canBeSideEffected())
265               || (seenUnmoveableSideEfffected && current.hasSideEffects()))) {
266             current.shouldRemove = false;
267           }
268     
269           if (current.shouldRemove) {
270             anyMovable = true;
271           } else {
272             if (current.canBeSideEffected) {
273               seenUnmoveableSideEfffected = true;
274             }
275     
276             if (current.hasSideEffects) {
277               seenUnmovableSideEffects = true;
278             }
279           }
280         }
281         return anyMovable;
282       }

284       /**
285        * Determine which parameters use the same expression.
286        * @return Whether any parameter was found that can be updated.
287        */
288       private boolean findFixedParameters(List<Parameter> parameters, Node cur) {
289         boolean anyMovable = false;
290         int index = 0;
291         while ((cur = cur.getNext()) != null) {
292           Parameter p;
293           if (index >= parameters.size()) {
294             p = new Parameter(cur, false);
295             parameters.add(p);
296           } else {
297             p = parameters.get(index);
298             if (p.shouldRemove()) {
299               Node value = p.getArg();
300               if (!cur.isEquivalentTo(value)) {
301                 p.setShouldRemove(false);
302               } else {
303                 anyMovable = true;
304               }
305             }
306           }
307     
308           setParameterSideEffectInfo(p, cur);
309           index++;
310         }
311     
312         for (; index < parameters.size(); index++) {
313           parameters.get(index).setShouldRemove(false);
314         }
315     
316         return anyMovable;
317       }

319       /**
320        * @return Whether any parameter was movable.
321        */
322       private boolean buildParameterList(
323           List<Parameter> parameters, Node cur, Scope s) {
324         boolean anyMovable = false;
325         while ((cur = cur.getNext()) != null) {
326           boolean movable = isMovableValue(cur, s);
327           Parameter p = new Parameter(cur, movable);
328           setParameterSideEffectInfo(p, cur);
329           parameters.add(p);
330           if (movable) {
331             anyMovable = true;
332           }
333         }
334         return anyMovable;
335       }
336     
337       private void setParameterSideEffectInfo(Parameter p, Node value) {
338         if (!p.hasSideEffects()) {
339           p.setHasSideEffects(NodeUtil.mayHaveSideEffects(value, compiler));
340         }
341     
342         if (!p.canBeSideEffected()) {
343           p.setCanBeSideEffected(NodeUtil.canBeSideEffected(value));
344         }
345       }

348       /**
349        * @return Whether the expression can be safely moved to another function
350        *   in another scope.
351        */
352       private static boolean isMovableValue(Node n, Scope s) {
353         // Things that can change value or are inaccessible can't be moved, these
354         // are "this", "arguments", local names, and functions that capture local
355         // values.
356         switch (n.getType()) {
357           case THIS:
358             return false;
359           case FUNCTION:
360             // Don't move function closures.
361             // TODO(johnlenz): Closure that only contain global reference can be
362             // moved.
363             return false;
364           case NAME:
365             if (n.getString().equals("arguments")) {
366               return false;
367             } else {
368               Var v = s.getVar(n.getString());
369               // Make sure that the variable is global. A caught exception, while
370               // it is in the global scope object in the compiler, it is not a
371               // global variable.
372               if (v != null &&
373                   (v.isLocal() ||
374                    v.nameNode.getParent().isCatch())) {
375                 return false;
376               }
377             }
378             break;
379         }
380     
381         for (Node c = n.getFirstChild(); c != null; c = c.getNext()) {
382           if (!isMovableValue(c, s)) {
383             return false;
384           }
385         }
386         return true;
387       }

389       private void optimizeFunctionDefinition(List<Parameter> parameters,
390           Node function) {
391         for (int index = parameters.size() - 1; index >= 0; index--) {
392           if (parameters.get(index).shouldRemove()) {
393             Node paramName = eliminateFunctionParamAt(function, index);
394             addVariableToFunction(function, paramName,
395                 parameters.get(index).getArg());
396           }
397         }
398       }

400       private void optimizeCallSite(
401           SimpleDefinitionFinder defFinder, List<Parameter> parameters, Node call) {
402         boolean mayMutateArgs = call.mayMutateArguments();
403         boolean mayMutateGlobalsOrThrow = call.mayMutateGlobalStateOrThrow();
404         for (int index = parameters.size() - 1; index >= 0; index--) {
405           Parameter p = parameters.get(index);
406           if (p.shouldRemove()) {
407             eliminateCallParamAt(defFinder, p, call, index);
408     
409             if (mayMutateArgs && !mayMutateGlobalsOrThrow &&
410                 // We want to cover both global-state arguments, and
411                 // expressions that might throw exceptions.
412                 // We're deliberately conservative here b/c it's
413                 // difficult to test all the edge cases.
414                 !NodeUtil.isImmutableValue(p.getArg())) {
415               mayMutateGlobalsOrThrow = true;
416               call.setSideEffectFlags(
417                   new Node.SideEffectFlags(call.getSideEffectFlags())
418                   .setMutatesGlobalState());
419             }
420           }
421         }
422       }

424       /**
425        * Simple container class that keeps tracks of a parameter and whether it
426        * should be removed.
427        */
428       private static class Parameter {
429         private final Node arg;
430         private boolean shouldRemove;
431         private boolean hasSideEffects;
432         private boolean canBeSideEffected;
433     
434         public Parameter(Node arg, boolean shouldRemove) {
435           this.shouldRemove = shouldRemove;
436           this.arg = arg;
437         }
438     
439         public Node getArg() {
440           return arg;
441         }
442     
443         public boolean shouldRemove() {
444           return shouldRemove;
445         }
446     
447         public void setShouldRemove(boolean value) {
448           shouldRemove = value;
449         }
450     
451         public void setHasSideEffects(boolean hasSideEffects) {
452           this.hasSideEffects = hasSideEffects;
453         }
454     
455         public boolean hasSideEffects() {
456           return hasSideEffects;
457         }
458     
459         public void setCanBeSideEffected(boolean canBeSideEffected) {
460           this.canBeSideEffected = canBeSideEffected;
461         }
462     
463         public boolean canBeSideEffected() {
464           return canBeSideEffected;
465         }
466       }

468       /**
469        * Adds a variable to the top of a function block.
470        * @param function A function node.
471        * @param varName The name of the variable.
472        * @param value The initial value of the variable.
473        */
474       private void addVariableToFunction(Node function, Node varName, Node value) {
475         Preconditions.checkArgument(function.isFunction(),
476             "Node must be a function.");
477     
478         Node block = function.getLastChild();
479         Preconditions.checkArgument(block.isBlock(),
480             "Node must be a block.");
481     
482         Preconditions.checkState(value.getParent() == null);
483         Node stmt;
484         if (varName != null) {
485           stmt = NodeUtil.newVarNode(varName.getString(), value);
486         } else {
487           stmt = IR.exprResult(value);
488         }
489         block.addChildToFront(stmt);
490         compiler.reportCodeChange();
491       }

493       /**
494        * Removes all formal parameters starting at argIndex.
495        * @return true if a parameter has been removed.
496        */
497       private boolean eliminateParamsAfter(Node function, int argIndex) {
498         Node formalArgPtr = function.getSecondChild().getFirstChild();
499         while (argIndex != 0 && formalArgPtr != null) {
500           formalArgPtr = formalArgPtr.getNext();
501           argIndex--;
502         }
503     
504         return eliminateParamsAfter(function, formalArgPtr);
505       }
506     
507       private boolean eliminateParamsAfter(Node fnNode, Node argNode) {
508         if (argNode != null) {
509           // Keep the args in the same order, do the last first.
510           eliminateParamsAfter(fnNode, argNode.getNext());
511           argNode.detachFromParent();
512           Node var = IR.var(argNode).useSourceInfoIfMissingFrom(argNode);
513           fnNode.getLastChild().addChildrenToFront(var);
514           compiler.reportCodeChange();
515           return true;
516         }
517         return false;
518       }

520       /**
521        * Eliminates the parameter from a function definition.
522        * @param function The function node
523        * @param argIndex The index of the the argument to remove.
524        * @return The Node of the argument removed.
525        */
526       private static Node eliminateFunctionParamAt(Node function, int argIndex) {
527         Preconditions.checkArgument(function.isFunction(),
528             "Node must be a function.");
529     
530         Node formalArgPtr = NodeUtil.getArgumentForFunction(
531             function, argIndex);
532     
533         if (formalArgPtr != null) {
534           function.getSecondChild().removeChild(formalArgPtr);
535         }
536         return formalArgPtr;
537       }

539       /**
540        * Eliminates the parameter from a function call.
541        * @param defFinder
542        * @param p
543        * @param call The function call node
544        * @param argIndex The index of the the argument to remove.
545        * @return The Node of the argument removed.
546        */
547       private Node eliminateCallParamAt(
548           SimpleDefinitionFinder defFinder, Parameter p, Node call, int argIndex) {
549         Preconditions.checkArgument(
550             NodeUtil.isCallOrNew(call), "Node must be a call or new.");
551     
552         Node formalArgPtr = NodeUtil.getArgumentForCallOrNew(
553             call, argIndex);
554     
555         if (formalArgPtr != null) {
556           call.removeChild(formalArgPtr);
557           // The value in the parameter object is the one that is being moved into
558           // function definition leave that one's references.  For everything else,
559           // remove any references.
560           if (p.getArg() != formalArgPtr) {
561             removedNodes.add(formalArgPtr);
562           }
563           compiler.reportCodeChange();
564         }
565         return formalArgPtr;
566       }
