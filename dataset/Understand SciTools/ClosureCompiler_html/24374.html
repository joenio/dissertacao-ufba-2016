36       public void testSimpleClasses() {
37         typeCheck("class Foo {}");
38     
39         typeCheck(LINE_JOINER.join(
40             "class Foo {}",
41             "class Bar {}",
42             "/** @type {!Foo} */ var x = new Bar;"),
43             NewTypeInference.MISTYPED_ASSIGN_RHS);
44     
45         typeCheck(LINE_JOINER.join(
46             "class Foo {",
47             "  constructor(x) {",
48             "    /** @type {string} */",
49             "    this.x = x;",
50             "  }",
51             "}",
52             "(new Foo('')).x - 5;"),
53             NewTypeInference.INVALID_OPERAND_TYPE);
54       }

74       public void testFunctionSubtypingForReceiverType() {
75         typeCheck(LINE_JOINER.join(
76             "class Foo {",
77             "  method() {}",
78             "}",
79             "class Bar extends Foo {}",
80             "function f(/** function(this:Bar) */ x) {}",
81             "f(Foo.prototype.method);"));
82     
83         typeCheck(LINE_JOINER.join(
84             "class Foo {",
85             "  method() { return 123; }",
86             "}",
87             "class Bar extends Foo {}",
88             "/**",
89             " * @template T",
90              " * @param {function(this:Bar):T} x",
91             " */",
92             "function f(x) {}",
93             "f(Foo.prototype.method);"));
94     
95         typeCheck(LINE_JOINER.join(
96             "class Controller {}",
97             "class SubController extends Controller {",
98             "  method() {}",
99             "}",
100             "/** @param {{always: function(this:Controller)}} spec */",
101             "function vsyncMethod(spec) {}",
102             "vsyncMethod({always: (new SubController).method});"));
103       }

105       public void testDetectPropertyDefinitionOnNullableObject() {
106         typeCheck(LINE_JOINER.join(
107             "/** @unrestricted */",
108             "class Foo {}",
109             "function f(/** ?Foo */ x) {",
110             "  /** @type {number} */",
111             "  x.prop = 123;",
112             "}",
113             "function g(/** !Foo */ x) {",
114             "  return x.prop - 5;",
115             "}"),
116             NewTypeInference.NULLABLE_DEREFERENCE);
117       }

149       public void testFunctionsWithUntypecheckedArguments() {
150         typeCheck(LINE_JOINER.join(
151             "class Foo {}",
152             "/**",
153             " * @param {function(function(new:Foo, ...?))} f1",
154             " * @param {function(new:Foo, ?)} f2",
155             " */",
156             "function f(f1, f2) {",
157             "  f1(f2);",
158             "}"));
159     
160         typeCheck(LINE_JOINER.join(
161             "class Foo {}",
162             "/**",
163             " * @template T",
164             " * @param {function(...?):T} x",
165             " * @return {T}",
166             " */",
167             "function f(x) {",
168             "  return x();",
169             "}",
170             "/** @type {function(?):!Foo} */",
171             "function g(x) { return new Foo; }",
172             "f(g) - 5;"),
173             NewTypeInference.INVALID_OPERAND_TYPE);

176       public void testMethodOverridesWithoutJsdoc() {
177         typeCheck(LINE_JOINER.join(
178             "class A {  someMethod(x) {}  }",
179             "class B extends A {  someMethod() {}  }"));
180     
181         typeCheck(LINE_JOINER.join(
182             "class A {  someMethod(x) {}  }",
183             "class B extends A {  someMethod(x, y) { return y + 1; }  }"),
184             GlobalTypeInfo.INVALID_PROP_OVERRIDE);
185     
186         typeCheck(LINE_JOINER.join(
187             "class Foo {",
188             "  /** @param {...number} var_args */",
189             "  method(var_args) {}",
190             "}",
191             "class Bar extends Foo {",
192             "  method(x) {}",
193             "}",
194             "(new Bar).method('str');"),
195             NewTypeInference.INVALID_ARGUMENT_TYPE);
196     
197         typeCheck(LINE_JOINER.join(
198             "class Foo {",
199             "  /** @param {...number} var_args */",
200             "  method(var_args) {}",
201             "}",
202             "class Bar extends Foo {",
203             "  method(x,y,z) {}",
204             "}",
205             "(new Bar).method('str');"),
206             NewTypeInference.WRONG_ARGUMENT_COUNT);
207       }
