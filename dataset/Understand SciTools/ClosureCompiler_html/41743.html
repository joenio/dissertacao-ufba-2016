164       // If there are two 'var' statements for the same variable, and both are in the JS
165       // (not in externs), the second will be normalized to an assignment, and the
166       // JSDoc with the suppression will be on the new node.
167       public void testConstSuppressionOnVar() {
168         String before = "/** @const */ var xyz = 1;\n/** @suppress {const} */ var xyz = 3;";
169         String after = "/** @const */ var xyz = 1;\n/** @suppress {const} */ xyz = 3;";
170         test(before, after);
171       }
172     
173       // If there are two 'var' statements for the same variable, one in externs and
174       // one in the JS, there is no normalization, and the suppression remains on the
175       // statement in the JS.
176       public void testConstSuppressionOnVarFromExterns() {
177         String externs = "/** @const */ var xyz;";
178         String js = "/** @suppress {const} */ var xyz = 3;";
179         test(externs, js, js, null, null);
180       }
181     
182       public void testConstSuppressionOnInc() {
183         testSame("/** @const */ var xyz = 1; /** @suppress {const} */ xyz++;");
184       }
185     
186       public void testConstNameInExterns() {
187         String externs = "/** @const */ var FOO;";
188         String js = "FOO = 1;";
189         test(externs, js, (String) null, null, ConstCheck.CONST_REASSIGNED_VALUE_ERROR);
190       }
