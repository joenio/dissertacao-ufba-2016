97       /**
98        * Pattern for unnamed messages.
99        * <p>
100        * All JS messages in JS code should have unique name but messages in
101        * generated code (i.e. from soy template) could have duplicated message names.
102        * Later we replace the message names with ids constructed as a hash of the
103        * message content.
104        * <p>
105        * <link href="http://code.google.com/p/closure-templates/">
106        * Soy</link> generates messages with names MSG_UNNAMED_<NUMBER> . This
107        * pattern recognizes such messages.
108        */
109       private static final Pattern MSG_UNNAMED_PATTERN =
110           Pattern.compile("MSG_UNNAMED_\\d+");

159           JsMessage.Style style, JsMessage.IdGenerator idGenerator) {
160     
161         this.compiler = compiler;
162         this.needToCheckDuplications = needToCheckDuplications;
163         this.style = style;
164         this.idGenerator = idGenerator;
165     
166         checkLevel = (style == JsMessage.Style.CLOSURE)
167             ? CheckLevel.ERROR : CheckLevel.WARNING;

174       @Override
175       public void process(Node externs, Node root) {
176         NodeTraversal.traverseEs6(compiler, root, this);
177     
178         for (Node msgNode : googMsgNodes) {
179           compiler.report(JSError.make(msgNode,
180               checkLevel, MESSAGE_NODE_IS_ORPHANED));
181         }
182       }

184       @Override
185       public void visit(NodeTraversal traversal, Node node, Node parent) {
186         String messageKey;
187         String originalMessageKey;
188         boolean isVar;
189         Node msgNode;
190     
191         switch (node.getType()) {
192           case NAME:
193             // var MSG_HELLO = 'Message'
194             if ((parent != null) && (NodeUtil.isNameDeclaration(parent))) {
195               messageKey = node.getString();
196               originalMessageKey = node.getOriginalName();
197               isVar = true;
198             } else {
199               return;
200             }
201     
202             msgNode = node.getFirstChild();
203             break;
204           case ASSIGN:
205             // somenamespace.someclass.MSG_HELLO = 'Message'
206             isVar = false;
207     
208             Node getProp = node.getFirstChild();
209             if (!getProp.isGetProp()) {
210               return;
211             }
212     
213             Node propNode = getProp.getLastChild();
214     
215             messageKey = propNode.getString();
216             originalMessageKey = getProp.getOriginalName();
217             msgNode = node.getLastChild();
218             break;
219     
220           case STRING_KEY:
221             if (node.isQuotedString() || node.getFirstChild() == null) {
222               return;
223             }
224             isVar = false;
225             messageKey = node.getString();
226             originalMessageKey = node.getOriginalName();
227             msgNode = node.getFirstChild();
228             break;
229     
230           case CALL:
231             // goog.getMsg()
232             if (node.getFirstChild().matchesQualifiedName(MSG_FUNCTION_NAME)) {
233               googMsgNodes.add(node);
234             } else if (node.getFirstChild().matchesQualifiedName(
235                 MSG_FALLBACK_FUNCTION_NAME)) {
236               visitFallbackFunctionCall(traversal, node);
237             }
238             return;
239           default:
240             return;
241         }
242     
243         if (originalMessageKey != null) {
244           messageKey = originalMessageKey;
245         }
246     
247         // Is this a message name?
248         boolean isNewStyleMessage =
249             msgNode != null && msgNode.isCall();
250         if (!isMessageName(messageKey, isNewStyleMessage)) {
251           return;
252         }
253     
254         if (msgNode == null) {
255           compiler.report(
256               traversal.makeError(node, MESSAGE_HAS_NO_VALUE, messageKey));
257           return;
258         }
259     
260         if (msgNode.isQualifiedName()
261             && msgNode.getLastChild().getString().equals(messageKey)) {
262           // foo.Thing.MSG_EXAMPLE = bar.OtherThing.MSG_EXAMPLE;
263           // This kind of construct is created by Es6ToEs3ClassSideInheritance. Just ignore it; the
264           // message will have already been extracted from the base class.
265           return;
266         }
267     
268         // Report a warning if a qualified messageKey that looks like a message
269         // (e.g. "a.b.MSG_X") doesn't use goog.getMsg().
270         if (isNewStyleMessage) {
271           googMsgNodes.remove(msgNode);
272         } else if (style != JsMessage.Style.LEGACY) {
273           // TODO(johnlenz): promote this to an error once existing conflicts have been
274           // cleaned up.
275           compiler.report(traversal.makeError(node,
276               MESSAGE_NOT_INITIALIZED_USING_NEW_SYNTAX));
277           if (style == JsMessage.Style.CLOSURE) {
278             // Don't extract the message if we aren't accepting LEGACY messages
279             return;
280           }
281         }
282     
283         boolean isUnnamedMsg = isUnnamedMessageName(messageKey);
284     
285         Builder builder = new Builder(
286             isUnnamedMsg ? null : messageKey);
287         OriginalMapping mapping = compiler.getSourceMapping(
288             traversal.getSourceName(), traversal.getLineNumber(),
289             traversal.getCharno());
290         if (mapping != null) {
291           builder.setSourceName(mapping.getOriginalFile());
292         } else {
293           builder.setSourceName(traversal.getSourceName());
294         }
295     
296         try {
297           if (isVar) {
298             extractMessageFromVariable(builder, node, parent, parent.getParent());
299           } else {
300             extractMessageFrom(builder, msgNode, node);
301           }
302         } catch (MalformedException ex) {
303           compiler.report(traversal.makeError(ex.getNode(),
304               MESSAGE_TREE_MALFORMED, ex.getMessage()));
305           return;
306         }
307     
308         JsMessage extractedMessage = builder.build(idGenerator);
309     
310         // If asked to check named internal messages.
311         if (needToCheckDuplications
312             && !isUnnamedMsg
313             && !extractedMessage.isExternal()) {
314           checkIfMessageDuplicated(messageKey, msgNode);
315         }
316         trackMessage(traversal, extractedMessage,
317             messageKey, msgNode, isUnnamedMsg);
318     
319         if (extractedMessage.isEmpty()) {
320           // value of the message is an empty string. Translators do not like it.
321           compiler.report(traversal.makeError(node, MESSAGE_HAS_NO_TEXT,
322               messageKey));
323         }
324     
325         // New-style messages must have descriptions. We don't emit a warning
326         // for legacy-style messages, because there are thousands of
327         // them in legacy code that are not worth the effort to fix, since they've
328         // already been translated anyway.
329         String desc = extractedMessage.getDesc();
330         if (isNewStyleMessage
331             && (desc == null || desc.trim().isEmpty())
332             && !extractedMessage.isExternal()) {
333           compiler.report(traversal.makeError(node, MESSAGE_HAS_NO_DESCRIPTION,
334               messageKey));
335         }
336     
337         JsMessageDefinition msgDefinition = new JsMessageDefinition(msgNode);
338         processJsMessage(extractedMessage, msgDefinition);
339       }

341       /**
342        * Track a message for later retrieval.
343        *
344        * This is used for tracking duplicates, and for figuring out message
345        * fallback. Not all message types are trackable, because that would
346        * require a more sophisticated analysis. e.g.,
347        * function f(s) { s.MSG_UNNAMED_X = 'Some untrackable message'; }
348        */
349       private void trackMessage(
350           NodeTraversal t, JsMessage message, String msgName,
351           Node msgNode, boolean isUnnamedMessage) {
352         if (!isUnnamedMessage) {
353           MessageLocation location = new MessageLocation(message, msgNode);
354           messageNames.put(msgName, location);
355         } else {
356           Var var = t.getScope().getVar(msgName);
357           if (var != null) {
358             unnamedMessages.put(var, message);
359           }
360         }
361       }
362     
363       /** Get a previously tracked message. */
364       private JsMessage getTrackedMessage(NodeTraversal t, String msgName) {
365         boolean isUnnamedMessage = isUnnamedMessageName(msgName);
366         if (!isUnnamedMessage) {
367           MessageLocation location = messageNames.get(msgName);
368           return location == null ? null : location.message;
369         } else {
370           Var var = t.getScope().getVar(msgName);
371           if (var != null) {
372             return unnamedMessages.get(var);
373           }
374         }
375         return null;
376       }

378       /**
379        * Checks if message already processed. If so - it generates 'message
380        * duplicated' compiler error.
381        *
382        * @param msgName the name of the message
383        * @param msgNode the node that represents JS message
384        */
385       private void checkIfMessageDuplicated(String msgName, Node msgNode) {
386         if (messageNames.containsKey(msgName)) {
387           MessageLocation location = messageNames.get(msgName);
388           compiler.report(JSError.make(msgNode, MESSAGE_DUPLICATE_KEY,
389               msgName, location.messageNode.getSourceFileName(),
390               Integer.toString(location.messageNode.getLineno())));
391         }
392       }

394       /**
395        * Creates a {@link JsMessage} for a JS message defined using a JS variable
396        * declaration (e.g <code>var MSG_X = ...;</code>).
397        *
398        * @param builder the message builder
399        * @param nameNode a NAME node for a JS message variable
400        * @param parentNode a VAR node, parent of {@code nameNode}
401        * @param grandParentNode the grandparent of {@code nameNode}. This node is
402        *        only used to get meta data about the message that might be
403        *        surrounding it (e.g. a message description). This argument may be
404        *        null if the meta data is not needed.
405        * @throws MalformedException if {@code varNode} does not
406        *         correspond to a valid JS message VAR node
407        */
408       private void extractMessageFromVariable(
409           Builder builder, Node nameNode, Node parentNode,
410           @Nullable Node grandParentNode) throws MalformedException {
411     
412         // Determine the message's value
413         Node valueNode = nameNode.getFirstChild();
414         switch (valueNode.getType()) {
415           case STRING:
416           case ADD:
417             maybeInitMetaDataFromJsDocOrHelpVar(builder, parentNode,
418                 grandParentNode);
419             builder.appendStringPart(extractStringFromStringExprNode(valueNode));
420             break;
421           case FUNCTION:
422             maybeInitMetaDataFromJsDocOrHelpVar(builder, parentNode,
423                 grandParentNode);
424             extractFromFunctionNode(builder, valueNode);
425             break;
426           case CALL:
427             maybeInitMetaDataFromJsDoc(builder, parentNode);
428             extractFromCallNode(builder, valueNode);
429             break;
430           default:
431             throw new MalformedException("Cannot parse value of message "
432                 + builder.getKey(), valueNode);
433         }
434       }

437        * Creates a {@link JsMessage} for a JS message defined using an assignment to

439        *
440        * @param builder the message builder
441        * @param valueNode a node in a JS message value
442        * @param docNode the node containing the jsdoc.

444        *         correspond to a valid JS message node

453       /**
454        * Initializes the meta data in a JsMessage by examining the nodes just before
455        * and after a message VAR node.
456        *
457        * @param builder the message builder whose meta data will be initialized
458        * @param varNode the message VAR node
459        * @param parentOfVarNode {@code varNode}'s parent node
460        */
461       private void maybeInitMetaDataFromJsDocOrHelpVar(
462           Builder builder, Node varNode, @Nullable Node parentOfVarNode)
463           throws MalformedException {
464     
465         // First check description in @desc
466         if (maybeInitMetaDataFromJsDoc(builder, varNode)) {
467           return;
468         }
469     
470         // Check the preceding node for meta data
471         if ((parentOfVarNode != null) &&
472             maybeInitMetaDataFromHelpVar(builder,
473                 parentOfVarNode.getChildBefore(varNode))) {
474           return;
475         }
476     
477         // Check the subsequent node for meta data
478         maybeInitMetaDataFromHelpVar(builder, varNode.getNext());
479       }

481       /**
482        * Initializes the meta data in a JsMessage by examining a node just before or
483        * after a message VAR node.
484        *
485        * @param builder the message builder whose meta data will be initialized
486        * @param sibling a node adjacent to the message VAR node
487        * @return true iff message has corresponding description variable
488        */
489       private static boolean maybeInitMetaDataFromHelpVar(Builder builder,
490           @Nullable Node sibling) throws MalformedException {
491         if ((sibling != null) && (sibling.isVar())) {
492           Node nameNode = sibling.getFirstChild();
493           String name = nameNode.getString();
494           if (name.equals(builder.getKey() + DESC_SUFFIX)) {
495             Node valueNode = nameNode.getFirstChild();
496             String desc = extractStringFromStringExprNode(valueNode);
497             if (desc.startsWith(HIDDEN_DESC_PREFIX)) {
498               builder.setDesc(desc.substring(HIDDEN_DESC_PREFIX.length()).trim());
499               builder.setIsHidden(true);
500             } else {
501               builder.setDesc(desc);
502             }
503             return true;
504           }
505         }
506         return false;
507       }

509       /**
510        * Initializes the meta data in a message builder given a node that may
511        * contain JsDoc properties.
512        *
513        * @param builder the message builder whose meta data will be initialized
514        * @param node the node with the message's JSDoc properties
515        * @return true if message has JsDoc with valid description in @desc
516        *         annotation
517        */
518       private static boolean maybeInitMetaDataFromJsDoc(Builder builder, Node node) {
519         boolean messageHasDesc = false;
520         JSDocInfo info = node.getJSDocInfo();
521         if (info != null) {
522           String desc = info.getDescription();
523           if (desc != null) {
524             builder.setDesc(desc);
525             messageHasDesc = true;
526           }
527           if (info.isHidden()) {
528             builder.setIsHidden(true);
529           }
530           if (info.getMeaning() != null) {
531             builder.setMeaning(info.getMeaning());
532           }
533         }
534     
535         return messageHasDesc;
536       }

538       /**
539        * Returns the string value associated with a node representing a JS string or
540        * several JS strings added together (e.g. {@code 'str'} or {@code 's' + 't' +
541        * 'r'}).
542        *
543        * @param node the node from where we extract the string
544        * @return String representation of the node
545        * @throws MalformedException if the parsed message is invalid
546        */
547       private static String extractStringFromStringExprNode(Node node)
548           throws MalformedException {
549         switch (node.getType()) {
550           case STRING:
551             return node.getString();
552           case ADD:
553             StringBuilder sb = new StringBuilder();
554             for (Node child : node.children()) {
555               sb.append(extractStringFromStringExprNode(child));
556             }
557             return sb.toString();
558           default:
559             throw new MalformedException("STRING or ADD node expected; found: " + node.getType(), node);
560         }
561       }

563       /**
564        * Initializes a message builder from a FUNCTION node.
565        * <p>
566        * <pre>
567        * The tree should look something like:
568        *
569        * function
570        *  |-- name
571        *  |-- lp
572        *  |   |-- name <arg1>
573        *  |    -- name <arg2>
574        *   -- block
575        *      |
576        *       --return
577        *           |
578        *            --add
579        *               |-- string foo
580        *                -- name <arg1>
581        * </pre>
582        *
583        * @param builder the message builder
584        * @param node the function node that contains a message
585        * @throws MalformedException if the parsed message is invalid
586        */
587       private void extractFromFunctionNode(Builder builder, Node node)
588           throws MalformedException {
589         Set<String> phNames = new HashSet<>();
590     
591         for (Node fnChild : node.children()) {
592           switch (fnChild.getType()) {
593             case NAME:
594               // This is okay. The function has a name, but it is empty.
595               break;
596             case PARAM_LIST:
597               // Parse the placeholder names from the function argument list.
598               for (Node argumentNode : fnChild.children()) {
599                 if (argumentNode.isName()) {
600                   String phName = argumentNode.getString();
601                   if (phNames.contains(phName)) {
602                     throw new MalformedException("Duplicate placeholder name: "
603                         + phName, argumentNode);
604                   } else {
605                     phNames.add(phName);
606                   }
607                 }
608               }
609               break;
610             case BLOCK:
611               // Build the message's value by examining the return statement
612               Node returnNode = fnChild.getFirstChild();
613               if (!returnNode.isReturn()) {
614                 throw new MalformedException(
615                     "RETURN node expected; found: " + returnNode.getType(), returnNode);
616               }
617               for (Node child : returnNode.children()) {
618                 extractFromReturnDescendant(builder, child);
619               }
620     
621               // Check that all placeholders from the message text have appropriate
622               // object literal keys
623               for (String phName : builder.getPlaceholders()) {
624                 if (!phNames.contains(phName)) {
625                   throw new MalformedException(
626                       "Unrecognized message placeholder referenced: " + phName,
627                       returnNode);
628                 }
629               }
630               break;
631             default:
632               throw new MalformedException(
633                   "NAME, LP, or BLOCK node expected; found: " + node.getType(), fnChild);
634           }
635         }
636       }

638       /**
639        * Appends value parts to the message builder by traversing the descendants
640        * of the given RETURN node.
641        *
642        * @param builder the message builder
643        * @param node the node from where we extract a message
644        * @throws MalformedException if the parsed message is invalid
645        */
646       private static void extractFromReturnDescendant(Builder builder, Node node)
647           throws MalformedException {
648     
649         switch (node.getType()) {
650           case STRING:
651             builder.appendStringPart(node.getString());
652             break;
653           case NAME:
654             builder.appendPlaceholderReference(node.getString());
655             break;
656           case ADD:
657             for (Node child : node.children()) {
658               extractFromReturnDescendant(builder, child);
659             }
660             break;
661           default:
662             throw new MalformedException(
663                 "STRING, NAME, or ADD node expected; found: " + node.getType(), node);
664         }
665       }

667       /**
668        * Initializes a message builder from a CALL node.
669        * <p>
670        * The tree should look something like:
671        *
672        * <pre>
673        * call
674        *  |-- getprop
675        *  |   |-- name 'goog'
676        *  |   +-- string 'getMsg'
677        *  |
678        *  |-- string 'Hi {$userName}! Welcome to {$product}.'
679        *  +-- objlit
680        *      |-- string 'userName'
681        *      |-- name 'someUserName'
682        *      |-- string 'product'
683        *      +-- call
684        *          +-- name 'getProductName'
685        * </pre>
686        *
687        * @param builder the message builder
688        * @param node the call node from where we extract the message
689        * @throws MalformedException if the parsed message is invalid
690        */
691       private void extractFromCallNode(Builder builder,
692           Node node) throws MalformedException {
693         // Check the function being called
694         if (!node.isCall()) {
695           throw new MalformedException(
696               "Message must be initialized using " + MSG_FUNCTION_NAME +
697               " function.", node);
698         }
699     
700         Node fnNameNode = node.getFirstChild();
701         if (!fnNameNode.matchesQualifiedName(MSG_FUNCTION_NAME)) {
702           throw new MalformedException(
703               "Message initialized using unrecognized function. " +
704               "Please use " + MSG_FUNCTION_NAME + "() instead.", fnNameNode);
705         }
706     
707         // Get the message string
708         Node stringLiteralNode = fnNameNode.getNext();
709         if (stringLiteralNode == null) {
710           throw new MalformedException("Message string literal expected",
711               stringLiteralNode);
712         }
713         // Parse the message string and append parts to the builder
714         parseMessageTextNode(builder, stringLiteralNode);
715     
716         Node objLitNode = stringLiteralNode.getNext();
717         Set<String> phNames = new HashSet<>();
718         if (objLitNode != null) {
719           // Register the placeholder names
720           if (!objLitNode.isObjectLit()) {
721             throw new MalformedException("OBJLIT node expected", objLitNode);
722           }
723           for (Node aNode = objLitNode.getFirstChild(); aNode != null;
724                aNode = aNode.getNext()) {
725             if (!aNode.isStringKey()) {
726               throw new MalformedException("STRING_KEY node expected as OBJLIT key",
727                   aNode);
728             }
729             String phName = aNode.getString();
730             if (!isLowerCamelCaseWithNumericSuffixes(phName)) {
731               throw new MalformedException(
732                   "Placeholder name not in lowerCamelCase: " + phName, aNode);
733             }
734     
735             if (phNames.contains(phName)) {
736               throw new MalformedException("Duplicate placeholder name: "
737                   + phName, aNode);
738             }
739     
740             phNames.add(phName);
741           }
742         }
743     
744         // Check that all placeholders from the message text have appropriate objlit
745         // values
746         Set<String> usedPlaceholders = builder.getPlaceholders();
747         for (String phName : usedPlaceholders) {
748           if (!phNames.contains(phName)) {
749             throw new MalformedException(
750                 "Unrecognized message placeholder referenced: " + phName,
751                 node);
752           }
753         }
754     
755         // Check that objLiteral have only names that are present in the
756         // message text
757         for (String phName : phNames) {
758           if (!usedPlaceholders.contains(phName)) {
759             throw new MalformedException(
760                 "Unused message placeholder: " + phName,
761                 node);
762           }
763         }
764       }

766       /**
767        * Appends the message parts in a JS message value extracted from the given
768        * text node.
769        *
770        * @param builder the JS message builder to append parts to
771        * @param node the node with string literal that contains the message text
772        * @throws MalformedException if {@code value} contains a reference to
773        *         an unregistered placeholder
774        */
775       private static void parseMessageTextNode(Builder builder, Node node)
776           throws MalformedException {
777         String value = extractStringFromStringExprNode(node);
778     
779         while (true) {
780           int phBegin = value.indexOf(PH_JS_PREFIX);
781           if (phBegin < 0) {
782             // Just a string literal
783             builder.appendStringPart(value);
784             return;
785           } else {
786             if (phBegin > 0) {
787               // A string literal followed by a placeholder
788               builder.appendStringPart(value.substring(0, phBegin));
789             }
790     
791             // A placeholder. Find where it ends
792             int phEnd = value.indexOf(PH_JS_SUFFIX, phBegin);
793             if (phEnd < 0) {
794               throw new MalformedException(
795                   "Placeholder incorrectly formatted in: " + builder.getKey(),
796                   node);
797             }
798     
799             String phName = value.substring(phBegin + PH_JS_PREFIX.length(),
800                 phEnd);
801             builder.appendPlaceholderReference(phName);
802             int nextPos = phEnd + PH_JS_SUFFIX.length();
803             if (nextPos < value.length()) {
804               // Iterate on the rest of the message value
805               value = value.substring(nextPos);
806             } else {
807               // The message is parsed
808               return;
809             }
810           }
811         }
812       }

814       /** Visit a call to goog.getMsgWithFallback. */
815       private void visitFallbackFunctionCall(NodeTraversal t, Node call) {
816         // Check to make sure the function call looks like:
817         // goog.getMsgWithFallback(MSG_1, MSG_2);
818         if (call.getChildCount() != 3 ||
819             !call.getChildAtIndex(1).isName() ||
820             !call.getChildAtIndex(2).isName()) {
821           compiler.report(t.makeError(call, BAD_FALLBACK_SYNTAX));
822           return;
823         }
824     
825         Node firstArg = call.getChildAtIndex(1);
826         JsMessage firstMessage = getTrackedMessage(t, firstArg.getString());
827         if (firstMessage == null) {
828           compiler.report(
829               t.makeError(firstArg, FALLBACK_ARG_ERROR, firstArg.getString()));
830           return;
831         }
832     
833         Node secondArg = firstArg.getNext();
834         JsMessage secondMessage = getTrackedMessage(
835             t, call.getChildAtIndex(2).getString());
836         if (secondMessage == null) {
837           compiler.report(
838               t.makeError(secondArg, FALLBACK_ARG_ERROR, secondArg.getString()));
839           return;
840         }
841     
842         processMessageFallback(call, firstMessage, secondMessage);
843       }

888       /**
889        * Returns whether a string is nonempty, begins with a lowercase letter, and
890        * contains only digits and underscores after the first underscore.
891        */
892       static boolean isLowerCamelCaseWithNumericSuffixes(String input) {
893         return CAMELCASE_PATTERN.matcher(input).matches();
894       }

896       /**
897        * Converts the given string from upper-underscore case to lower-camel case,
898        * preserving numeric suffixes. For example: "NAME" -> "name" "A4_LETTER" ->
899        * "a4Letter" "START_SPAN_1_23" -> "startSpan_1_23".
900        */
901       static String toLowerCamelCaseWithNumericSuffixes(String input) {
902         // Determine where the numeric suffixes begin
903         int suffixStart = input.length();
904         while (suffixStart > 0) {
905           char ch = '\0';
906           int numberStart = suffixStart;
907           while (numberStart > 0) {
908             ch = input.charAt(numberStart - 1);
909             if (Character.isDigit(ch)) {
910               numberStart--;
911             } else {
912               break;
913             }
914           }
915           if ((numberStart > 0) && (numberStart < suffixStart) && (ch == '_')) {
916             suffixStart = numberStart - 1;
917           } else {
918             break;
919           }
920         }
921     
922         if (suffixStart == input.length()) {
923           return CaseFormat.UPPER_UNDERSCORE.to(CaseFormat.LOWER_CAMEL, input);
924         } else {
925           return CaseFormat.UPPER_UNDERSCORE.to(CaseFormat.LOWER_CAMEL,
926               input.substring(0, suffixStart)) +
927               input.substring(suffixStart);
928         }
929       }
