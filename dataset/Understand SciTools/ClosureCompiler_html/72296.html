16     package com.google.javascript.jscomp;
17     
18     import static com.google.javascript.jscomp.CheckRequiresForConstructors.EXTRA_REQUIRE_WARNING;
19     import static com.google.javascript.jscomp.CheckRequiresForConstructors.MISSING_REQUIRE_WARNING;
20     
21     /**
22      * Tests for {@link CheckRequiresForConstructors} in single-file mode.
23      */
24     public final class SingleFileCheckRequiresTest extends Es6CompilerTestCase {
25     
26       @Override
27       protected CompilerOptions getOptions(CompilerOptions options) {
28         options.setWarningLevel(DiagnosticGroups.MISSING_REQUIRE, CheckLevel.ERROR);
29         options.setWarningLevel(DiagnosticGroups.EXTRA_REQUIRE, CheckLevel.ERROR);
30         return super.getOptions(options);
31       }
32     
33       @Override
34       protected CompilerPass getProcessor(final Compiler compiler) {
35         return new CheckRequiresForConstructors(compiler,
36             CheckRequiresForConstructors.Mode.SINGLE_FILE);
37       }
38     
39       public void testReferenceToSingleName() {
40         testSame("new Foo();");
41         testSame("new Array();");
42         testSame("new Error();");
43       }
44     
45       public void testCtorExtendsSingleName() {
46         testSame("/** @constructor @extends {Foo} */ function MyFoo() {}");
47         testSame("/** @constructor @extends {Error} */ function MyError() {}");
48         testSame("/** @constructor @extends {Array} */ function MyArray() {}");
49       }
50     
51       public void testClassExtendsSingleName() {
52         testSameEs6("class MyFoo extends Foo {}");
53         testSameEs6("class MyError extends Error {}");
54         testSameEs6("class MyArray extends Array {}");
55       }
56     
57       public void testReferenceToQualifiedName() {
58         testError(
59             LINE_JOINER.join(
60                 "goog.require('x.y.z');",
61                 "goog.require('bar.Abc');",
62                 "new x.y.z();",
63                 "new bar.Abc();",
64                 "new bar.Foo();"),
65             MISSING_REQUIRE_WARNING);
66       }
67     
68       // Since there are no goog.require()s for any bar.* names, assume that bar
69       // is a "non-Closurized" namespace, i.e. that all bar.* names come from the externs.
70       public void testReferenceToQualifiedName_nonClosurizedNamespace() {
71         testSame(
72             LINE_JOINER.join(
73                 "goog.require('x.y.z');",
74                 "new x.y.z();",
75                 "new bar.Foo();"));
76       }
77     
78       public void testReferenceToUnqualifiedName() {
79         testSameEs6(
80             LINE_JOINER.join(
81                 "goog.module('a.b.c');",
82                 "var z = goog.require('x.y.z');",
83                 "",
84                 "exports = { foobar : z };"));
85     
86         testSameEs6(
87             LINE_JOINER.join(
88                 "goog.module('a.b.c');",
89                 "var {z} = goog.require('x.y');",
90                 "",
91                 "exports = { foobar : z };"));
92     
93         testSameEs6(
94             LINE_JOINER.join(
95                 "import {z} from 'x.y'",
96                 "",
97                 "export var foobar = z;"));
98     
99         testSameEs6(
100             LINE_JOINER.join(
101                 "import z from 'x.y.z'",
102                 "",
103                 "export var foobar = z;"));
104       }
105     
106       public void testExtraRequire() {
107         testErrorEs6("goog.require('foo.Bar');", EXTRA_REQUIRE_WARNING);
108       }
109     
110       public void testUnqualifiedRequireUsedInJSDoc() {
111         testSameEs6("goog.require('Bar'); /** @type {Bar} */ var x;");
112       }
113     
114       public void testUnqualifiedImportUsedInJSDoc() {
115         testSameEs6("import { Something } from 'somewhere'; /** @type {Something} */ var x;");
116       }
117     
118       public void testReferenceToSingleNameWithRequire() {
119         testSameEs6("goog.require('Foo'); new Foo();");
120       }
121     
122       public void testReferenceInDefaultParam() {
123         testSameEs6("function func( a = new Bar() ){}; func();");
124       }
125     
126       public void testReferenceInDestructuringParam() {
127         testSameEs6("var {a = new Bar()} = b;");
128       }
129     }
