37     /**
38      * Analyzes properties on prototypes.
39      *
40      * Uses a reference graph to analyze prototype properties. Each unique property
41      * name is represented by a node in this graph. An edge from property A to
42      * property B means that there's a GETPROP access of a property B on some
43      * object inside of a method named A.
44      *
45      * Global functions are also represented by nodes in this graph, with
46      * similar semantics.
47      *
48      * @author nicksantos@google.com (Nick Santos)
49      */
50     class AnalyzePrototypeProperties implements CompilerPass {
51     
52       // Constants for symbol types, for easier readability.
53       private static final SymbolType PROPERTY = SymbolType.PROPERTY;
54       private static final SymbolType VAR = SymbolType.VAR;
55     
56       private final AbstractCompiler compiler;
57       private final boolean canModifyExterns;
58       private final boolean anchorUnusedVars;
59       private final JSModuleGraph moduleGraph;
60       private final JSModule firstModule;
61     
62       // Properties that are implicitly used as part of the JS language.
63       private static final Set<String> IMPLICITLY_USED_PROPERTIES =
64           ImmutableSet.of("length", "toString", "valueOf");
65     
66       // A graph where the nodes are property names or variable names,
67       // and the edges signify the modules where the property is referenced.
68       // For example, if we had the code:
69       //
70       // Foo.prototype.bar = function(x) { x.baz(); }; // in module 2.;
71       //
72       // then this would be represented in the graph by a node representing
73       // "bar", a node representing "baz", and an edge between them representing
74       // module #2.
75       //
76       // Similarly, if we had:
77       //
78       // var scotch = function(f) { return f.age(); };
79       //
80       // then there would be a node for "scotch", a node for "age", and an edge
81       // from scotch to age.
82       private final LinkedDirectedGraph<NameInfo, JSModule> symbolGraph =
83           LinkedDirectedGraph.createWithoutAnnotations();
84     
85       // A dummy node for representing global references.
86       private final NameInfo globalNode = new NameInfo("[global]");
87     
88       // A dummy node for representing extern references.
89       private final NameInfo externNode = new NameInfo("[extern]");
90     
91       // A dummy node for representing all anonymous functions with no names.
92       private final NameInfo anonymousNode = new NameInfo("[anonymous]");
93     
94       // All the real NameInfo for prototype properties, hashed by the name
95       // of the property that they represent.
96       private final Map<String, NameInfo> propertyNameInfo = new LinkedHashMap<>();
97     
98       // All the NameInfo for global functions, hashed by the name of the
99       // global variable that it's assigned to.
100       private final Map<String, NameInfo> varNameInfo = new LinkedHashMap<>();

102       /**
103        * Creates a new pass for analyzing prototype properties.
104        * @param compiler The compiler.
105        * @param moduleGraph The graph for resolving module dependencies. May be
106        *     null if we don't care about module dependencies.
107        * @param canModifyExterns If true, then we can move prototype
108        *     properties that are declared in the externs file.
109        * @param anchorUnusedVars If true, then we must mark all vars as referenced,
110        *     even if they are never used.
111        */
112       AnalyzePrototypeProperties(AbstractCompiler compiler,
113           JSModuleGraph moduleGraph, boolean canModifyExterns,
114           boolean anchorUnusedVars) {
115         this.compiler = compiler;
116         this.moduleGraph = moduleGraph;
117         this.canModifyExterns = canModifyExterns;
118         this.anchorUnusedVars = anchorUnusedVars;
119     
120         if (moduleGraph != null) {
121           firstModule = moduleGraph.getRootModule();
122         } else {
123           firstModule = null;
124         }
125     
126         globalNode.markReference(null);
127         externNode.markReference(null);
128         symbolGraph.createNode(globalNode);
129         symbolGraph.createNode(externNode);
130     
131         for (String property : IMPLICITLY_USED_PROPERTIES) {
132           NameInfo nameInfo = getNameInfoForName(property, PROPERTY);
133           if (moduleGraph == null) {
134             symbolGraph.connect(externNode, null, nameInfo);
135           } else {
136             for (JSModule module : moduleGraph.getAllModules()) {
137               symbolGraph.connect(externNode, module, nameInfo);
138             }
139           }
140         }
141       }
142     
143       @Override
144       public void process(Node externRoot, Node root) {
145         if (!canModifyExterns) {
146           NodeTraversal.traverseEs6(compiler, externRoot,
147               new ProcessExternProperties());
148         }
149     
150         NodeTraversal.traverseEs6(compiler, root, new ProcessProperties());
151     
152         FixedPointGraphTraversal<NameInfo, JSModule> t =
153             FixedPointGraphTraversal.newTraversal(new PropagateReferences());
154         t.computeFixedPoint(symbolGraph,
155             ImmutableSet.of(externNode, globalNode));
156       }

159        * Returns information on all prototype properties.
160        */
161       public Collection<NameInfo> getAllNameInfo() {
162         List<NameInfo> result = new ArrayList<>(propertyNameInfo.values());
163         result.addAll(varNameInfo.values());
164         return result;

167       /**
168        * Gets the name info for the property or variable of a given name,
169        * and creates a new one if necessary.
170        *
171        * @param name The name of the symbol.
172        * @param type The type of symbol.
173        */
174       private NameInfo getNameInfoForName(String name, SymbolType type) {
175         Map<String, NameInfo> map = type == PROPERTY ?
176             propertyNameInfo : varNameInfo;
177         if (map.containsKey(name)) {
178           return map.get(name);
179         } else {
180           NameInfo nameInfo = new NameInfo(name);
181           map.put(name, nameInfo);
182           symbolGraph.createNode(nameInfo);
183           return nameInfo;
184         }
185       }

187       private class ProcessProperties implements NodeTraversal.ScopedCallback {
188         // There are two types of context information on this stack:
189         // 1) Every scope has a NameContext corresponding to its scope.
190         //    Variables are given VAR contexts.
191         //    Prototype properties are given PROPERTY contexts.
192         //    The global scope is given the special [global] context.
193         //    And function expressions that we aren't able to give a reasonable
194         //    name are given a special [anonymous] context.
195         // 2) Every assignment of a prototype property of a non-function is
196         //    given a name context. These contexts do not have scopes.
197         private final Stack<NameContext> symbolStack = new Stack<>();
198     
199         @Override
200         public void enterScope(NodeTraversal t) {
201           Node n = t.getCurrentNode();
202           Scope scope = t.getScope();
203           Node root = scope.getRootNode();
204           if (root.isFunction()) {
205             String propName = getPrototypePropertyNameFromRValue(n);
206             if (propName != null) {
207               symbolStack.push(
208                   new NameContext(
209                       getNameInfoForName(propName, PROPERTY),
210                       scope));
211             } else if (isGlobalFunctionDeclaration(t, n)) {
212               Node parent = n.getParent();
213               String name = parent.isName() ?
214                   parent.getString() /* VAR */ :
215                   n.getFirstChild().getString() /* named function */;
216               symbolStack.push(
217                   new NameContext(getNameInfoForName(name, VAR), scope.getClosestHoistScope()));
218             } else {
219               // NOTE(nicksantos): We use the same anonymous node for all
220               // functions that do not have reasonable names. I can't remember
221               // at the moment why we do this. I think it's because anonymous
222               // nodes can never have in-edges. They're just there as a placeholder
223               // for scope information, and do not matter in the edge propagation.
224               symbolStack.push(new NameContext(anonymousNode, scope));
225             }
226           } else if (t.inGlobalScope()) {
227             symbolStack.push(new NameContext(globalNode, scope));
228           } else {
229             // TODO(moz): It's not yet clear if we need another kind of NameContext for block scopes
230             // in ES6, use anonymous node for now and investigate later.
231             Preconditions.checkState(NodeUtil.createsBlockScope(root), scope);
232             symbolStack.push(new NameContext(anonymousNode, scope));
233           }
234         }
235     
236         @Override
237         public void exitScope(NodeTraversal t) {
238           symbolStack.pop();
239         }
240     
241         @Override
242         public boolean shouldTraverse(NodeTraversal t, Node n, Node parent) {
243           // Process prototype assignments to non-functions.
244           String propName = processNonFunctionPrototypeAssign(n, parent);
245           if (propName != null) {
246             symbolStack.push(
247                 new NameContext(
248                     getNameInfoForName(propName, PROPERTY), null));
249           }
250           return true;
251         }
252     
253         @Override
254         public void visit(NodeTraversal t, Node n, Node parent) {
255           if (n.isGetProp()) {
256             String propName = n.getSecondChild().getString();
257     
258             if (n.isQualifiedName()) {
259               if (propName.equals("prototype")) {
260                 if (processPrototypeRef(t, n)) {
261                   return;
262                 }
263               } else if (compiler.getCodingConvention().isExported(propName)) {
264                 addGlobalUseOfSymbol(propName, t.getModule(), PROPERTY);
265                 return;
266               } else {
267                 // Do not mark prototype prop assigns as a 'use' in the global scope.
268                 if (n.getParent().isAssign() && n.getNext() != null) {
269                   String rValueName = getPrototypePropertyNameFromRValue(n);
270                   if (rValueName != null) {
271                     return;
272                   }
273                 }
274               }
275             }
276     
277             addSymbolUse(propName, t.getModule(), PROPERTY);
278           } else if (n.isObjectLit()) {
279             // Make sure that we're not handling object literals being
280             // assigned to a prototype, as in:
281             // Foo.prototype = {bar: 3, baz: 5};
282             String lValueName = NodeUtil.getBestLValueName(
283                 NodeUtil.getBestLValue(n));
284             if (lValueName != null && lValueName.endsWith(".prototype")) {
285               return;
286             }
287     
288             // var x = {a: 1, b: 2}
289             // should count as a use of property a and b.
290             for (Node propNameNode = n.getFirstChild(); propNameNode != null;
291                  propNameNode = propNameNode.getNext()) {
292               // May be STRING, GET, or SET, but NUMBER isn't interesting.
293               if (!propNameNode.isQuotedString()) {
294                 addSymbolUse(propNameNode.getString(), t.getModule(), PROPERTY);
295               }
296             }
297           } else if (n.isName()) {
298             String name = n.getString();
299     
300             Var var = t.getScope().getVar(name);
301             if (var != null) {
302               // Only process global functions.
303               if (var.isGlobal()) {
304                 if (var.getInitialValue() != null &&
305                     var.getInitialValue().isFunction()) {
306                   if (t.inGlobalHoistScope()) {
307                     if (!processGlobalFunctionDeclaration(t, n, var)) {
308                       addGlobalUseOfSymbol(name, t.getModule(), VAR);
309                     }
310                   } else {
311                     addSymbolUse(name, t.getModule(), VAR);
312                   }
313                 }
314     
315               // If it is not a global, it might be accessing a local of the outer
316               // scope. If that's the case the functions between the variable's
317               // declaring scope and the variable reference scope cannot be moved.
318               } else if (var.getScope() != t.getScope()) {
319                 for (int i = symbolStack.size() - 1; i >= 0; i--) {
320                   NameContext context = symbolStack.get(i);
321                   if (context.scope == var.getScope()) {
322                     break;
323                   }
324     
325                   context.name.readClosureVariables = true;
326                 }
327               }
328             }
329           }
330     
331           // Process prototype assignments to non-functions.
332           if (processNonFunctionPrototypeAssign(n, parent) != null) {
333             symbolStack.pop();
334           }
335         }

337         private void addSymbolUse(String name, JSModule module, SymbolType type) {
338           NameInfo info = getNameInfoForName(name, type);
339           NameInfo def = null;
340           // Skip all anonymous nodes. We care only about symbols with names.
341           for (int i = symbolStack.size() - 1; i >= 0; i--) {
342             def = symbolStack.get(i).name;
343             if (def != anonymousNode) {
344               break;
345             }
346           }
347           if (!def.equals(info)) {
348             symbolGraph.connect(def, module, info);
349           }
350         }

352         /**
353          * If this is a non-function prototype assign, return the prop name.
354          * Otherwise, return null.
355          */
356         private String processNonFunctionPrototypeAssign(Node n, Node parent) {
357           if (isAssignRValue(n, parent) && !n.isFunction()) {
358             return getPrototypePropertyNameFromRValue(n);
359           }
360           return null;
361         }

363         /**
364          * Determines whether {@code n} is the FUNCTION node in a global function
365          * declaration.
366          */
367         private boolean isGlobalFunctionDeclaration(NodeTraversal t, Node n) {
368           // Make sure we're not in a function scope, or if we are then the function we're looking at
369           // is defined in the global scope.
370           if (!(t.inGlobalHoistScope()
371                 || n.isFunction() && t.getScopeRoot() == n
372                    && t.getScope().getParent().getClosestHoistScope().isGlobal())) {
373             return false;
374           }
375     
376           return NodeUtil.isFunctionDeclaration(n)
377               || n.isFunction() && n.getParent().isName();
378         }
379     
380         /**
381          * Returns true if this is the r-value of an assignment.
382          */
383         private boolean isAssignRValue(Node n, Node parent) {
384           return parent != null && parent.isAssign() && parent.getFirstChild() != n;
385         }

387         /**
388          * Returns the name of a prototype property being assigned to this r-value.
389          *
390          * Returns null if this is not the R-value of a prototype property, or if
391          * the R-value is used in multiple expressions (i.e., if there's
392          * a prototype property assignment in a more complex expression).
393          */
394         private String getPrototypePropertyNameFromRValue(Node rValue) {
395           Node lValue = NodeUtil.getBestLValue(rValue);
396           if (lValue == null ||
397               lValue.getParent() == null ||
398               lValue.getGrandparent() == null ||
399               !((NodeUtil.isObjectLitKey(lValue) && !lValue.isQuotedString()) ||
400                 NodeUtil.isExprAssign(lValue.getGrandparent()))) {
401             return null;
402           }
403     
404           String lValueName =
405               NodeUtil.getBestLValueName(NodeUtil.getBestLValue(rValue));
406           if (lValueName == null) {
407             return null;
408           }
409           int lastDot = lValueName.lastIndexOf('.');
410           if (lastDot == -1) {
411             return null;
412           }
413     
414           String firstPart = lValueName.substring(0, lastDot);
415           if (!firstPart.endsWith(".prototype")) {
416             return null;
417           }
418     
419           return lValueName.substring(lastDot + 1);
420         }

422         /**
423          * Processes a NAME node to see if it's a global function declaration.
424          * If it is, record it and return true. Otherwise, return false.
425          */
426         private boolean processGlobalFunctionDeclaration(NodeTraversal t,
427             Node nameNode, Var v) {
428           Node firstChild = nameNode.getFirstChild();
429           Node parent = nameNode.getParent();
430     
431           if (// Check for a named FUNCTION.
432               isGlobalFunctionDeclaration(t, parent) ||
433               // Check for a VAR declaration.
434               firstChild != null &&
435               isGlobalFunctionDeclaration(t, firstChild)) {
436             String name = nameNode.getString();
437             getNameInfoForName(name, VAR).getDeclarations().add(
438                 new GlobalFunction(nameNode, v, t.getModule()));
439     
440             // If the function name is exported, we should create an edge here
441             // so that it's never removed.
442             if (compiler.getCodingConvention().isExported(name) ||
443                 anchorUnusedVars) {
444               addGlobalUseOfSymbol(name, t.getModule(), VAR);
445             }
446     
447             return true;
448           }
449           return false;
450         }

452         /**
453          * Processes the GETPROP of prototype, which can either be under
454          * another GETPROP (in the case of Foo.prototype.bar), or can be
455          * under an assignment (in the case of Foo.prototype = ...).
456          * @return True if a declaration was added.
457          */
458         private boolean processPrototypeRef(NodeTraversal t, Node ref) {
459           Node root = NodeUtil.getRootOfQualifiedName(ref);
460     
461           Node n = ref.getParent();
462           switch (n.getType()) {
463             // Foo.prototype.getBar = function() { ... }
464             case GETPROP:
465               Node dest = n.getSecondChild();
466               Node parent = n.getParent();
467               Node grandParent = parent.getParent();
468     
469               if (dest.isString() &&
470                   NodeUtil.isExprAssign(grandParent) &&
471                   NodeUtil.isVarOrSimpleAssignLhs(n, parent)) {
472                 String name = dest.getString();
473                 Property prop = new AssignmentProperty(
474                     grandParent,
475                     maybeGetVar(t, root),
476                     t.getModule());
477                 getNameInfoForName(name, PROPERTY).getDeclarations().add(prop);
478                 return true;
479               }
480               break;
481     
482             // Foo.prototype = { "getBar" : function() { ... } }
483             case ASSIGN:
484               Node map = n.getSecondChild();
485               if (map.isObjectLit()) {
486                 for (Node key = map.getFirstChild();
487                      key != null; key = key.getNext()) {
488                   if (!key.isQuotedString()) {
489                     // May be STRING, GETTER_DEF, or SETTER_DEF,
490                     String name = key.getString();
491                     Property prop = new LiteralProperty(
492                         key, key.getFirstChild(), map, n,
493                         maybeGetVar(t, root),
494                         t.getModule());
495                     getNameInfoForName(name, PROPERTY).getDeclarations().add(prop);
496                   }
497                 }
498                 return true;
499               }
500               break;
501           }
502           return false;
503         }

505         private Var maybeGetVar(NodeTraversal t, Node maybeName) {
506           return maybeName.isName()
507               ? t.getScope().getVar(maybeName.getString()) : null;
508         }

526       private class PropagateReferences
527           implements EdgeCallback<NameInfo, JSModule> {
528         @Override
529         public boolean traverseEdge(NameInfo start, JSModule edge, NameInfo dest) {
530           if (start.isReferenced()) {
531             JSModule startModule = start.getDeepestCommonModuleRef();
532             if (startModule != null &&
533                 moduleGraph.dependsOn(startModule, edge)) {
534               return dest.markReference(startModule);
535             } else {
536               return dest.markReference(edge);
537             }
538           }
539           return false;
540         }

572       /**
573        * A function initialized as a VAR statement or a function declaration.
574        */
575       static class GlobalFunction implements Symbol {
576         private final Node nameNode;
577         private final Var var;
578         private final JSModule module;
579     
580         GlobalFunction(Node nameNode, Var var, JSModule module) {
581           Node parent = nameNode.getParent();
582           Preconditions.checkState(
583               parent.isVar() ||
584               NodeUtil.isFunctionDeclaration(parent));
585           this.nameNode = nameNode;
586           this.var = var;
587           this.module = module;
588         }
589     
590         @Override
591         public Var getRootVar() {
592           return var;
593         }
594     
595         @Override
596         public void remove(AbstractCompiler compiler) {
597           Node parent = nameNode.getParent();
598           compiler.reportChangeToEnclosingScope(parent);
599           if (parent.isFunction() || parent.hasOneChild()) {
600             NodeUtil.removeChild(parent.getParent(), parent);
601           } else {
602             Preconditions.checkState(parent.isVar());
603             parent.removeChild(nameNode);
604           }
605         }
606     
607         @Override
608         public JSModule getModule() {
609           return module;
610         }
611       }

639          */

641           this.exprNode = node;
642           this.rootVar = rootVar;
643           this.module = module;
644         }

649         }

679        *

683       static class LiteralProperty implements Property {
684         private final Node key;
685         private final Node value;
686         private final Node map;
687         private final Node assign;
688         private final Var rootVar;
689         private final JSModule module;
690     
691         LiteralProperty(Node key, Node value, Node map, Node assign,
692             Var rootVar, JSModule module) {
693           this.key = key;
694           this.value = value;
695           this.map = map;
696           this.assign = assign;
697           this.rootVar = rootVar;
698           this.module = module;
699         }

702         public Var getRootVar() {
703           return rootVar;
704         }
705     
706         @Override
707         public void remove(AbstractCompiler compiler) {
708           compiler.reportChangeToEnclosingScope(key);
709           map.removeChild(key);
710         }
711     
712         @Override
713         public Node getPrototype() {
714           return assign.getFirstChild();
715         }
716     
717         @Override
718         public Node getValue() {
719           return value;

721     
722         @Override
723         public JSModule getModule() {
724           return module;
725         }

728       /**
729        * The context of the current name. This includes the NameInfo and the scope
730        * if it is a scope defining name (function).
731        */
732       private static class NameContext {
733         final NameInfo name;
734     
735         // If this is a function context, then scope will be the scope of the
736         // corresponding function. Otherwise, it will be null.
737         final Scope scope;
738     
739         NameContext(NameInfo name, Scope scope) {
740           this.name = name;
741           this.scope = scope;
742         }
743       }

745       /**
746        * Information on all properties or global variables of a given name.
747        */
748       class NameInfo {
749     
750         final String name;
751     
752         private boolean referenced = false;
753         private final Deque<Symbol> declarations = new ArrayDeque<>();
754         private JSModule deepestCommonModuleRef = null;
755     
756         // True if this property is a function that reads a variable from an
757         // outer scope which isn't the global scope.
758         private boolean readClosureVariables = false;
759     
760         /**
761          * Constructs a new NameInfo.
762          * @param name The name of the property that this represents. May be null
763          *     to signify dummy nodes in the property graph.
764          */
765         NameInfo(String name) {
766           this.name = name;
767         }
768     
769         @Override public String toString() { return name; }
770     
771         /** Determines whether we've marked a reference to this property name. */
772         boolean isReferenced() {
773           return referenced;
774         }
775     
776         /** Determines whether it reads a closure variable. */
777         boolean readsClosureVariables() {
778           return readClosureVariables;
779         }
780     
781         /**
782          * Mark a reference in a given module to this property name, and record
783          * the deepest common module reference.
784          * @param module The module where it was referenced.
785          * @return Whether the name info has changed.
786          */
787         boolean markReference(JSModule module) {
788           boolean hasChanged = false;
789           if (!referenced) {
790             referenced = true;
791             hasChanged = true;
792           }
793     
794           if (moduleGraph != null) {
795             JSModule originalDeepestCommon = deepestCommonModuleRef;
796     
797             if (deepestCommonModuleRef == null) {
798               deepestCommonModuleRef = module;
799             } else {
800               deepestCommonModuleRef =
801                   moduleGraph.getDeepestCommonDependencyInclusive(
802                       deepestCommonModuleRef, module);
803             }
804     
805             if (originalDeepestCommon != deepestCommonModuleRef) {
806               hasChanged = true;
807             }
808           }
809     
810           return hasChanged;
811         }
812     
813         /**
814          * Returns the deepest common module of all the references to this
815          * property.
816          */
817         JSModule getDeepestCommonModuleRef() {
818           return deepestCommonModuleRef;
819         }
820     
821         /**
822          * Returns a mutable collection of all the prototype property declarations
823          * of this property name.
824          */
825         Deque<Symbol> getDeclarations() {
826           return declarations;
827         }
828       }
