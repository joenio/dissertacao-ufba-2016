372       /**
373        * Main entry point for this phase of processing. This follows the pattern for
374        * JSCompiler phases.
375        *
376        * @param externsRoot The root of the externs parse tree.
377        * @param jsRoot The root of the input parse tree to be checked.
378        */
379       @Override
380       public void process(Node externsRoot, Node jsRoot) {
381         Preconditions.checkNotNull(scopeCreator);
382         Preconditions.checkNotNull(topScope);
383     
384         Node externsAndJs = jsRoot.getParent();
385         Preconditions.checkState(externsAndJs != null);
386         Preconditions.checkState(
387             externsRoot == null || externsAndJs.hasChild(externsRoot));
388     
389         if (externsRoot != null) {
390           check(externsRoot, true);
391         }
392         check(jsRoot, false);
393       }

395       /** Main entry point of this phase for testing code. */
396       public TypedScope processForTesting(Node externsRoot, Node jsRoot) {
397         Preconditions.checkState(scopeCreator == null);
398         Preconditions.checkState(topScope == null);
399     
400         Preconditions.checkState(jsRoot.getParent() != null);
401         Node externsAndJsRoot = jsRoot.getParent();
402     
403         scopeCreator = new MemoizedScopeCreator(new TypedScopeCreator(compiler));
404         topScope = scopeCreator.createScope(externsAndJsRoot, null);
405     
406         TypeInferencePass inference = new TypeInferencePass(compiler,
407             reverseInterpreter, topScope, scopeCreator);
408     
409         inference.process(externsRoot, jsRoot);
410         process(externsRoot, jsRoot);
411     
412         return topScope;
413       }

416       void check(Node node, boolean externs) {
417         Preconditions.checkNotNull(node);
418     
419         NodeTraversal t = new NodeTraversal(compiler, this, scopeCreator);
420         inExterns = externs;
421         t.traverseWithScope(node, topScope);
422         if (externs) {
423           inferJSDocInfo.process(node, null);
424         } else {
425           inferJSDocInfo.process(null, node);
426         }
427       }

434       @Override
435       public boolean shouldTraverse(
436           NodeTraversal t, Node n, Node parent) {
437         switch (n.getType()) {
438           case FUNCTION:
439             // normal type checking
440             final TypedScope outerScope = t.getTypedScope();
441             final String functionPrivateName = n.getFirstChild().getString();
442             if (functionPrivateName != null && functionPrivateName.length() > 0 &&
443                 outerScope.isDeclared(functionPrivateName, false) &&
444                 // Ideally, we would want to check whether the type in the scope
445                 // differs from the type being defined, but then the extern
446                 // redeclarations of built-in types generates spurious warnings.
447                 !(outerScope.getVar(
448                     functionPrivateName).getType() instanceof FunctionType)) {
449               report(t, n, FUNCTION_MASKS_VARIABLE, functionPrivateName);
450             }
451     
452             // TODO(user): Only traverse the function's body. The function's
453             // name and arguments are traversed by the scope creator, and ideally
454             // should not be traversed by the type checker.
455             break;
456         }
457         return true;
458       }

460       /**
461        * This is the meat of the type checking.  It is basically one big switch,
462        * with each case representing one type of parse tree node.  The individual
463        * cases are usually pretty straightforward.
464        *
465        * @param t The node traversal object that supplies context, such as the
466        * scope chain to use in name lookups as well as error reporting.
467        * @param n The node being visited.
468        * @param parent The parent of the node n.
469        */
470       @Override
471       public void visit(NodeTraversal t, Node n, Node parent) {
472         JSType childType;
473         JSType leftType, rightType;
474         Node left, right;
475         // To be explicitly set to false if the node is not typeable.
476         boolean typeable = true;
477     
478         switch (n.getType()) {
479           case CAST:
480             Node expr = n.getFirstChild();
481             JSType exprType = getJSType(expr);
482             JSType castType = getJSType(n);
483     
484             // TODO(johnlenz): determine if we can limit object literals in some
485             // way.
486             if (!expr.isObjectLit()) {
487               validator.expectCanCast(t, n, castType, exprType);
488             }
489             ensureTyped(t, n, castType);
490     
491             expr.putProp(Node.TYPE_BEFORE_CAST, exprType);
492             if (castType.isSubtype(exprType) || expr.isObjectLit()) {
493               expr.setJSType(castType);
494             }
495             break;
496     
497           case NAME:
498             typeable = visitName(t, n, parent);
499             break;
500     
501           case COMMA:
502             ensureTyped(t, n, getJSType(n.getLastChild()));
503             break;
504     
505           case THIS:
506             ensureTyped(t, n, t.getTypedScope().getTypeOfThis());
507             break;
508     
509           case NULL:
510             ensureTyped(t, n, NULL_TYPE);
511             break;
512     
513           case NUMBER:
514             ensureTyped(t, n, NUMBER_TYPE);
515             break;
516     
517           case GETTER_DEF:
518           case SETTER_DEF:
519             // Object literal keys are handled with OBJECTLIT
520             break;
521     
522           case ARRAYLIT:
523             ensureTyped(t, n, ARRAY_TYPE);
524             break;
525     
526           case REGEXP:
527             ensureTyped(t, n, REGEXP_TYPE);
528             break;
529     
530           case GETPROP:
531             visitGetProp(t, n, parent);
532             typeable = !(parent.isAssign() && parent.getFirstChild() == n);
533             break;
534     
535           case GETELEM:
536             visitGetElem(t, n);
537             // The type of GETELEM is always unknown, so no point counting that.
538             // If that unknown leaks elsewhere (say by an assignment to another
539             // variable), then it will be counted.
540             typeable = false;
541             break;
542     
543           case VAR:
544             visitVar(t, n);
545             typeable = false;
546             break;
547     
548           case NEW:
549             visitNew(t, n);
550             break;
551     
552           case CALL:
553             visitCall(t, n);
554             typeable = !parent.isExprResult();
555             break;
556     
557           case RETURN:
558             visitReturn(t, n);
559             typeable = false;
560             break;
561     
562           case DEC:
563           case INC:
564             left = n.getFirstChild();
565             checkPropCreation(t, left);
566             validator.expectNumber(t, left, getJSType(left), "increment/decrement");
567             ensureTyped(t, n, NUMBER_TYPE);
568             break;
569     
570           case VOID:
571             ensureTyped(t, n, VOID_TYPE);
572             break;
573     
574           case STRING:
575           case TYPEOF:
576             ensureTyped(t, n, STRING_TYPE);
577             break;
578     
579           case BITNOT:
580             childType = getJSType(n.getFirstChild());
581             if (!childType.matchesInt32Context()) {
582               report(t, n, BIT_OPERATION, NodeUtil.opToStr(n.getType()), childType.toString());
583             }
584             ensureTyped(t, n, NUMBER_TYPE);
585             break;
586     
587           case POS:
588           case NEG:
589             left = n.getFirstChild();
590             validator.expectNumber(t, left, getJSType(left), "sign operator");
591             ensureTyped(t, n, NUMBER_TYPE);
592             break;
593     
594           case EQ:
595           case NE:
596           case SHEQ:
597           case SHNE: {
598             left = n.getFirstChild();
599             right = n.getLastChild();
600     
601             if (left.isTypeOf()) {
602               if (right.isString()) {
603                 checkTypeofString(t, right, right.getString());
604               }
605             } else if (right.isTypeOf() && left.isString()) {
606               checkTypeofString(t, left, left.getString());
607             }
608     
609             leftType = getJSType(left);
610             rightType = getJSType(right);
611     
612             // We do not want to warn about explicit comparisons to VOID. People
613             // often do this if they think their type annotations screwed up.
614             //
615             // We do want to warn about cases where people compare things like
616             // (Array|null) == (Function|null)
617             // because it probably means they screwed up.
618             //
619             // This heuristic here is not perfect, but should catch cases we
620             // care about without too many false negatives.
621             JSType leftTypeRestricted = leftType.restrictByNotNullOrUndefined();
622             JSType rightTypeRestricted = rightType.restrictByNotNullOrUndefined();
623     
624             TernaryValue result = TernaryValue.UNKNOWN;
625             if (n.getType() == Token.EQ || n.getType() == Token.NE) {
626               result = leftTypeRestricted.testForEquality(rightTypeRestricted);
627               if (n.isNE()) {
628                 result = result.not();
629               }
630             } else {
631               // SHEQ or SHNE
632               if (!leftTypeRestricted.canTestForShallowEqualityWith(rightTypeRestricted)) {
633                 result = n.getType() == Token.SHEQ ? TernaryValue.FALSE : TernaryValue.TRUE;
634               }
635             }
636     
637             if (result != TernaryValue.UNKNOWN) {
638               report(
639                   t,
640                   n,
641                   DETERMINISTIC_TEST,
642                   leftType.toString(),
643                   rightType.toString(),
644                   result.toString());
645             }
646             ensureTyped(t, n, BOOLEAN_TYPE);
647             break;
648           }
649     
650           case LT:
651           case LE:
652           case GT:
653           case GE:
654             leftType = getJSType(n.getFirstChild());
655             rightType = getJSType(n.getLastChild());
656             if (rightType.isNumber()) {
657               validator.expectNumber(t, n, leftType, "left side of numeric comparison");
658             } else if (leftType.isNumber()) {
659               validator.expectNumber(t, n, rightType, "right side of numeric comparison");
660             } else if (leftType.matchesNumberContext() && rightType.matchesNumberContext()) {
661               // OK.
662             } else {
663               // Whether the comparison is numeric will be determined at runtime
664               // each time the expression is evaluated. Regardless, both operands
665               // should match a string context.
666               String message = "left side of comparison";
667               validator.expectString(t, n, leftType, message);
668               validator.expectNotNullOrUndefined(t, n, leftType, message, getNativeType(STRING_TYPE));
669               message = "right side of comparison";
670               validator.expectString(t, n, rightType, message);
671               validator.expectNotNullOrUndefined(t, n, rightType, message, getNativeType(STRING_TYPE));
672             }
673             ensureTyped(t, n, BOOLEAN_TYPE);
674             break;
675     
676           case IN:
677             left = n.getFirstChild();
678             right = n.getLastChild();
679             rightType = getJSType(right);
680             validator.expectString(t, left, getJSType(left), "left side of 'in'");
681             validator.expectObject(t, n, rightType, "'in' requires an object");
682             if (rightType.isStruct()) {
683               report(t, right, IN_USED_WITH_STRUCT);
684             }
685             ensureTyped(t, n, BOOLEAN_TYPE);
686             break;
687     
688           case INSTANCEOF:
689             left = n.getFirstChild();
690             right = n.getLastChild();
691             rightType = getJSType(right).restrictByNotNullOrUndefined();
692             validator.expectAnyObject(
693                 t, left, getJSType(left), "deterministic instanceof yields false");
694             validator.expectActualObject(t, right, rightType, "instanceof requires an object");
695             ensureTyped(t, n, BOOLEAN_TYPE);
696             break;
697     
698           case ASSIGN:
699             visitAssign(t, n);
700             typeable = false;
701             break;
702     
703           case ASSIGN_LSH:
704           case ASSIGN_RSH:
705           case ASSIGN_URSH:
706           case ASSIGN_DIV:
707           case ASSIGN_MOD:
708           case ASSIGN_BITOR:
709           case ASSIGN_BITXOR:
710           case ASSIGN_BITAND:
711           case ASSIGN_SUB:
712           case ASSIGN_ADD:
713           case ASSIGN_MUL:
714             checkPropCreation(t, n.getFirstChild());
715             // fall through
716     
717           case LSH:
718           case RSH:
719           case URSH:
720           case DIV:
721           case MOD:
722           case BITOR:
723           case BITXOR:
724           case BITAND:
725           case SUB:
726           case ADD:
727           case MUL:
728             visitBinaryOperator(n.getType(), t, n);
729             break;
730     
731           case TRUE:
732           case FALSE:
733           case NOT:
734           case DELPROP:
735             ensureTyped(t, n, BOOLEAN_TYPE);
736             break;
737     
738           case CASE:
739             JSType switchType = getJSType(parent.getFirstChild());
740             JSType caseType = getJSType(n.getFirstChild());
741             validator.expectSwitchMatchesCase(t, n, switchType, caseType);
742             typeable = false;
743             break;
744     
745           case WITH: {
746             Node child = n.getFirstChild();
747             childType = getJSType(child);
748             validator.expectObject(t, child, childType, "with requires an object");
749             typeable = false;
750             break;
751           }
752     
753           case MEMBER_FUNCTION_DEF:
754             ensureTyped(t, n, getJSType(n.getFirstChild()));
755             break;
756     
757           case FUNCTION:
758             visitFunction(t, n);
759             break;
760     
761             // These nodes have no interesting type behavior.
762             // These nodes require data flow analysis.
763           case PARAM_LIST:
764           case STRING_KEY:
765           case LABEL:
766           case LABEL_NAME:
767           case SWITCH:
768           case BREAK:
769           case CATCH:
770           case TRY:
771           case SCRIPT:
772           case EXPR_RESULT:
773           case BLOCK:
774           case EMPTY:
775           case DEFAULT_CASE:
776           case CONTINUE:
777           case DEBUGGER:
778           case THROW:
779           case DO:
780           case IF:
781           case WHILE:
782             typeable = false;
783             break;
784     
785           case FOR:
786             if (NodeUtil.isForIn(n)) {
787               Node obj = n.getSecondChild();
788               if (getJSType(obj).isStruct()) {
789                 report(t, obj, IN_USED_WITH_STRUCT);
790               }
791             }
792             typeable = false;
793             break;
794     
795           // These nodes are typed during the type inference.
796           case AND:
797           case HOOK:
798           case OBJECTLIT:
799           case OR:
800             if (n.getJSType() != null) { // If we didn't run type inference.
801               ensureTyped(t, n);
802             } else {
803               // If this is an enum, then give that type to the objectlit as well.
804               if ((n.isObjectLit()) && (parent.getJSType() instanceof EnumType)) {
805                 ensureTyped(t, n, parent.getJSType());
806               } else {
807                 ensureTyped(t, n);
808               }
809             }
810             if (n.isObjectLit()) {
811               JSType typ = getJSType(n);
812               for (Node key : n.children()) {
813                 visitObjLitKey(t, key, n, typ);
814               }
815             }
816             break;
817     
818           default:
819             report(t, n, UNEXPECTED_TOKEN, n.getType().toString());
820             ensureTyped(t, n);
821             break;
822         }
823     
824         // Don't count externs since the user's code may not even use that part.
825         typeable = typeable && !inExterns;
826     
827         if (typeable) {
828           doPercentTypedAccounting(t, n);
829         }
830     
831         checkJsdocInfoContainsObjectWithBadKey(t, n);
832       }

860       /**
861        * Visits an assignment <code>lvalue = rvalue</code>. If the
862        * <code>lvalue</code> is a prototype modification, we change the schema
863        * of the object type it is referring to.
864        * @param t the traversal
865        * @param assign the assign node
866        * (<code>assign.isAssign()</code> is an implicit invariant)
867        */
868       private void visitAssign(NodeTraversal t, Node assign) {
869         JSDocInfo info = assign.getJSDocInfo();
870         Node lvalue = assign.getFirstChild();
871         Node rvalue = assign.getLastChild();
872     
873         // Check property sets to 'object.property' when 'object' is known.
874         if (lvalue.isGetProp()) {
875           Node object = lvalue.getFirstChild();
876           JSType objectJsType = getJSType(object);
877           Node property = lvalue.getLastChild();
878           String pname = property.getString();
879     
880           // the first name in this getprop refers to an interface
881           // we perform checks in addition to the ones below
882           if (object.isGetProp()) {
883             JSType jsType = getJSType(object.getFirstChild());
884             if (jsType.isInterface() &&
885                 object.getLastChild().getString().equals("prototype")) {
886               visitInterfaceGetprop(t, assign, object, pname, lvalue, rvalue);
887             }
888           }
889     
890           checkEnumAlias(t, info, rvalue);
891           checkPropCreation(t, lvalue);
892     
893           // Prototype assignments are special, because they actually affect
894           // the definition of a class. These are mostly validated
895           // during TypedScopeCreator, and we only look for the "dumb" cases here.
896           // object.prototype = ...;
897           if (pname.equals("prototype")) {
898             if (objectJsType != null && objectJsType.isFunctionType()) {
899               FunctionType functionType = objectJsType.toMaybeFunctionType();
900               if (functionType.isConstructor()) {
901                 JSType rvalueType = rvalue.getJSType();
902                 validator.expectObject(t, rvalue, rvalueType,
903                     OVERRIDING_PROTOTYPE_WITH_NON_OBJECT);
904                 return;
905               }
906             }
907           }
908     
909           // The generic checks for 'object.property' when 'object' is known,
910           // and 'property' is declared on it.
911           // object.property = ...;
912           ObjectType type = ObjectType.cast(
913               objectJsType.restrictByNotNullOrUndefined());
914           if (type != null) {
915             if (type.hasProperty(pname) &&
916                 !type.isPropertyTypeInferred(pname)) {
917               JSType expectedType = type.getPropertyType(pname);
918               if (!expectedType.isUnknownType()) {
919                 if (!propertyIsImplicitCast(type, pname)) {
920                   validator.expectCanAssignToPropertyOf(
921                       t, assign, getJSType(rvalue),
922                       expectedType, object, pname);
923                   checkPropertyInheritanceOnGetpropAssign(
924                       t, assign, object, pname, info, expectedType);
925                 }
926                 return;
927               }
928             }
929           }
930     
931           // If we couldn't get the property type with normal object property
932           // lookups, then check inheritance anyway with the unknown type.
933           checkPropertyInheritanceOnGetpropAssign(
934               t, assign, object, pname, info, getNativeType(UNKNOWN_TYPE));
935         }
936     
937         // Check qualified name sets to 'object' and 'object.property'.
938         // This can sometimes handle cases when the type of 'object' is not known.
939         // e.g.,
940         // var obj = createUnknownType();
941         // /** @type {number} */ obj.foo = true;
942         JSType leftType = getJSType(lvalue);
943         if (lvalue.isQualifiedName()) {
944           // variable with inferred type case
945           TypedVar var = t.getTypedScope().getVar(lvalue.getQualifiedName());
946           if (var != null) {
947             if (var.isTypeInferred()) {
948               return;
949             }
950     
951             if (NodeUtil.getRootOfQualifiedName(lvalue).isThis() &&
952                 t.getTypedScope() != var.getScope()) {
953               // Don't look at "this.foo" variables from other scopes.
954               return;
955             }
956     
957             if (var.getType() != null) {
958               leftType = var.getType();
959             }
960           }
961         }
962     
963         // Fall through case for arbitrary LHS and arbitrary RHS.
964         Node rightChild = assign.getLastChild();
965         JSType rightType = getJSType(rightChild);
966         if (validator.expectCanAssignTo(
967                 t, assign, rightType, leftType, "assignment")) {
968           ensureTyped(t, assign, rightType);
969         } else {
970           ensureTyped(t, assign);
971         }
972       }

984       private void checkPropertyInheritanceOnGetpropAssign(
985           NodeTraversal t, Node assign, Node object, String property,
986           JSDocInfo info, JSType propertyType) {
987         // Inheritance checks for prototype properties.
988         //
989         // TODO(nicksantos): This isn't the right place to do this check. We
990         // really want to do this when we're looking at the constructor.
991         // We'd find all its properties and make sure they followed inheritance
992         // rules, like we currently do for @implements to make sure
993         // all the methods are implemented.
994         //
995         // As-is, this misses many other ways to override a property.
996         //
997         // object.prototype.property = ...;
998         if (object.isGetProp()) {
999           Node object2 = object.getFirstChild();
1000           String property2 = NodeUtil.getStringValue(object.getLastChild());
1001     
1002           if ("prototype".equals(property2)) {
1003             JSType jsType = getJSType(object2);
1004             if (jsType.isFunctionType()) {
1005               FunctionType functionType = jsType.toMaybeFunctionType();
1006               if (functionType.isConstructor() || functionType.isInterface()) {
1007                 checkDeclaredPropertyInheritance(
1008                     t, assign, functionType, property, info, propertyType);
1009               }
1010             }
1011           }
1012         }
1013       }

1016           NodeTraversal t, Node key, String propertyName, ObjectType type) {
1017         // Inheritance checks for prototype objlit properties.
1018         //
1019         // TODO(nicksantos): This isn't the right place to do this check. We
1020         // really want to do this when we're looking at the constructor.
1021         // We'd find all its properties and make sure they followed inheritance
1022         // rules, like we currently do for @implements to make sure
1023         // all the methods are implemented.
1024         //
1025         // As-is, this misses many other ways to override a property.
1026         //
1027         // object.prototype = { key: function() {} };
1028         FunctionType ctorType = type.getOwnerFunction();
1029         if (ctorType == null || (!ctorType.isConstructor() && !ctorType.isInterface())) {
1030           return;

1033         JSType propertyType = type.getPropertyType(propertyName);

1035             t, key.getFirstChild(), ctorType, propertyName,

1039       /**
1040        * Visits an object literal field definition <code>key : value</code>.
1041        *
1042        * If the <code>lvalue</code> is a prototype modification, we change the
1043        * schema of the object type it is referring to.
1044        *
1045        * @param t the traversal
1046        * @param key the assign node
1047        */
1048       private void visitObjLitKey(
1049           NodeTraversal t, Node key, Node objlit, JSType litType) {
1050         // Do not validate object lit value types in externs. We don't really care,
1051         // and it makes it easier to generate externs.
1052         if (objlit.isFromExterns()) {
1053           ensureTyped(t, key);
1054           return;
1055         }
1056     
1057         // Structs must have unquoted keys and dicts must have quoted keys
1058         if (litType.isStruct() && key.isQuotedString()) {
1059           report(t, key, ILLEGAL_OBJLIT_KEY, "struct");
1060         } else if (litType.isDict() && !key.isQuotedString()) {
1061           report(t, key, ILLEGAL_OBJLIT_KEY, "dict");
1062         }
1063     
1064         // TODO(johnlenz): Validate get and set function declarations are valid
1065         // as is the functions can have "extraneous" bits.
1066     
1067         // For getter and setter property definitions the
1068         // r-value type != the property type.
1069         Node rvalue = key.getFirstChild();
1070         JSType rightType = getObjectLitKeyTypeFromValueType(key, getJSType(rvalue));
1071         if (rightType == null) {
1072           rightType = getNativeType(UNKNOWN_TYPE);
1073         }
1074     
1075         Node owner = objlit;
1076     
1077         // Validate value is assignable to the key type.
1078     
1079         JSType keyType = getJSType(key);
1080     
1081         JSType allowedValueType = keyType;
1082         if (allowedValueType.isEnumElementType()) {
1083           allowedValueType =
1084               allowedValueType.toMaybeEnumElementType().getPrimitiveType();
1085         }
1086     
1087         boolean valid = validator.expectCanAssignToPropertyOf(t, key,
1088             rightType, allowedValueType,
1089             owner, NodeUtil.getObjectLitKeyName(key));
1090         if (valid) {
1091           ensureTyped(t, key, rightType);
1092         } else {
1093           ensureTyped(t, key);
1094         }
1095     
1096         // Validate that the key type is assignable to the object property type.
1097         // This is necessary as the objlit may have been cast to a non-literal
1098         // object type.
1099         // TODO(johnlenz): consider introducing a CAST node to the AST (or
1100         // perhaps a parentheses node).
1101     
1102         JSType objlitType = getJSType(objlit);
1103         ObjectType type = ObjectType.cast(
1104             objlitType.restrictByNotNullOrUndefined());
1105         if (type != null) {
1106           String property = NodeUtil.getObjectLitKeyName(key);
1107           checkPropertyInheritanceOnPrototypeLitKey(t, key, property, type);
1108           if (type.hasProperty(property) &&
1109               !type.isPropertyTypeInferred(property) &&
1110               !propertyIsImplicitCast(type, property)) {
1111             validator.expectCanAssignToPropertyOf(
1112                 t, key, keyType,
1113                 type.getPropertyType(property), owner, property);
1114           }
1115           return;
1116         }
1117       }

1133       /**
1134        * Given a constructor type and a property name, check that the property has
1135        * the JSDoc annotation @override iff the property is declared on a
1136        * superclass. Several checks regarding inheritance correctness are also
1137        * performed.
1138        */
1139       private void checkDeclaredPropertyInheritance(
1140           NodeTraversal t, Node n, FunctionType ctorType, String propertyName,
1141           JSDocInfo info, JSType propertyType) {
1142         // If the supertype doesn't resolve correctly, we've warned about this
1143         // already.
1144         if (hasUnknownOrEmptySupertype(ctorType)) {
1145           return;
1146         }
1147     
1148         FunctionType superClass = ctorType.getSuperClassConstructor();
1149         boolean superClassHasProperty = superClass != null &&
1150             superClass.getInstanceType().hasProperty(propertyName);
1151         boolean superClassHasDeclaredProperty = superClass != null &&
1152             superClass.getInstanceType().isPropertyTypeDeclared(propertyName);
1153     
1154         // For interface
1155         boolean superInterfaceHasProperty = false;
1156         boolean superInterfaceHasDeclaredProperty = false;
1157         if (ctorType.isInterface()) {
1158           for (ObjectType interfaceType : ctorType.getExtendedInterfaces()) {
1159             superInterfaceHasProperty =
1160                 superInterfaceHasProperty ||
1161                 interfaceType.hasProperty(propertyName);
1162             superInterfaceHasDeclaredProperty =
1163                 superInterfaceHasDeclaredProperty ||
1164                 interfaceType.isPropertyTypeDeclared(propertyName);
1165           }
1166         }
1167         boolean declaredOverride = info != null && info.isOverride();
1168     
1169         boolean foundInterfaceProperty = false;
1170         if (ctorType.isConstructor()) {
1171           for (JSType implementedInterface :
1172               ctorType.getAllImplementedInterfaces()) {
1173             if (implementedInterface.isUnknownType() ||
1174                 implementedInterface.isEmptyType()) {
1175               continue;
1176             }
1177             FunctionType interfaceType =
1178                 implementedInterface.toObjectType().getConstructor();
1179             Preconditions.checkNotNull(interfaceType);
1180     
1181             boolean interfaceHasProperty =
1182                 interfaceType.getPrototype().hasProperty(propertyName);
1183             foundInterfaceProperty = foundInterfaceProperty ||
1184                 interfaceHasProperty;
1185             if (reportMissingOverride.isOn()
1186                 && !declaredOverride
1187                 && interfaceHasProperty
1188                 && !"__proto__".equals(propertyName)) {
1189               // @override not present, but the property does override an interface
1190               // property
1191               compiler.report(t.makeError(n, reportMissingOverride,
1192                   HIDDEN_INTERFACE_PROPERTY, propertyName,
1193                   interfaceType.getTopMostDefiningType(propertyName).toString()));
1194             }
1195           }
1196         }
1197     
1198         if (!declaredOverride
1199             && !superClassHasProperty
1200             && !superInterfaceHasProperty) {
1201           // nothing to do here, it's just a plain new property
1202           return;
1203         }
1204     
1205         ObjectType topInstanceType = superClassHasDeclaredProperty ?
1206             superClass.getTopMostDefiningType(propertyName) : null;
1207         boolean declaredLocally =
1208             ctorType.isConstructor() &&
1209             (ctorType.getPrototype().hasOwnProperty(propertyName) ||
1210              ctorType.getInstanceType().hasOwnProperty(propertyName));
1211         if (reportMissingOverride.isOn()
1212             && !declaredOverride
1213             && superClassHasDeclaredProperty
1214             && declaredLocally
1215             && !"__proto__".equals(propertyName)) {
1216           // @override not present, but the property does override a superclass
1217           // property
1218           compiler.report(t.makeError(n, reportMissingOverride,
1219               HIDDEN_SUPERCLASS_PROPERTY, propertyName,
1220               topInstanceType.toString()));
1221         }
1222     
1223         // @override is present and we have to check that it is ok
1224         if (superClassHasDeclaredProperty) {
1225           // there is a superclass implementation
1226           JSType superClassPropType =
1227               superClass.getInstanceType().getPropertyType(propertyName);
1228           TemplateTypeMap ctorTypeMap =
1229               ctorType.getTypeOfThis().getTemplateTypeMap();
1230           if (!ctorTypeMap.isEmpty()) {
1231             superClassPropType = superClassPropType.visit(
1232                 new TemplateTypeMapReplacer(typeRegistry, ctorTypeMap));
1233           }
1234     
1235           if (!propertyType.isSubtype(superClassPropType)) {
1236             compiler.report(
1237                 t.makeError(n, HIDDEN_SUPERCLASS_PROPERTY_MISMATCH,
1238                     propertyName, topInstanceType.toString(),
1239                     superClassPropType.toString(), propertyType.toString()));
1240           }
1241         } else if (superInterfaceHasDeclaredProperty) {
1242           // there is an super interface property
1243           for (ObjectType interfaceType : ctorType.getExtendedInterfaces()) {
1244             if (interfaceType.hasProperty(propertyName)) {
1245               JSType superPropertyType =
1246                   interfaceType.getPropertyType(propertyName);
1247               if (!propertyType.isSubtype(superPropertyType)) {
1248                 topInstanceType = interfaceType.getConstructor().
1249                     getTopMostDefiningType(propertyName);
1250                 compiler.report(
1251                     t.makeError(n, HIDDEN_SUPERCLASS_PROPERTY_MISMATCH,
1252                         propertyName, topInstanceType.toString(),
1253                         superPropertyType.toString(),
1254                         propertyType.toString()));
1255               }
1256             }
1257           }
1258         } else if (!foundInterfaceProperty
1259             && !superClassHasProperty
1260             && !superInterfaceHasProperty) {
1261           // there is no superclass nor interface implementation
1262           compiler.report(
1263               t.makeError(n, UNKNOWN_OVERRIDE,
1264                   propertyName, ctorType.getInstanceType().toString()));
1265         }
1266       }

1268       /**
1269        * Given a constructor or an interface type, find out whether the unknown
1270        * type is a supertype of the current type.
1271        */
1272       private static boolean hasUnknownOrEmptySupertype(FunctionType ctor) {
1273         Preconditions.checkArgument(ctor.isConstructor() || ctor.isInterface());
1274         Preconditions.checkArgument(!ctor.isUnknownType());
1275     
1276         // The type system should notice inheritance cycles on its own
1277         // and break the cycle.
1278         while (true) {
1279           ObjectType maybeSuperInstanceType =
1280               ctor.getPrototype().getImplicitPrototype();
1281           if (maybeSuperInstanceType == null) {
1282             return false;
1283           }
1284           if (maybeSuperInstanceType.isUnknownType() ||
1285               maybeSuperInstanceType.isEmptyType()) {
1286             return true;
1287           }
1288           ctor = maybeSuperInstanceType.getConstructor();
1289           if (ctor == null) {
1290             return false;
1291           }
1292           Preconditions.checkState(ctor.isConstructor() || ctor.isInterface());
1293         }
1294       }

1365       /**
1366        * Visits a NAME node.
1367        *
1368        * @param t The node traversal object that supplies context, such as the
1369        * scope chain to use in name lookups as well as error reporting.
1370        * @param n The node being visited.
1371        * @param parent The parent of the node n.
1372        * @return whether the node is typeable or not
1373        */
1374       boolean visitName(NodeTraversal t, Node n, Node parent) {
1375         // At this stage, we need to determine whether this is a leaf
1376         // node in an expression (which therefore needs to have a type
1377         // assigned for it) versus some other decorative node that we
1378         // can safely ignore.  Function names, arguments (children of LP nodes) and
1379         // variable declarations are ignored.
1380         // TODO(user): remove this short-circuiting in favor of a
1381         // pre order traversal of the FUNCTION, CATCH, LP and VAR nodes.
1382         Token parentNodeType = parent.getType();
1383         if (parentNodeType == Token.FUNCTION ||
1384             parentNodeType == Token.CATCH ||
1385             parentNodeType == Token.PARAM_LIST ||
1386             parentNodeType == Token.VAR) {
1387           return false;
1388         }
1389     
1390         // Not need to type first key in for in.
1391         if (NodeUtil.isForIn(parent) && parent.getFirstChild() == n) {
1392           return false;
1393         }
1394     
1395         JSType type = n.getJSType();
1396         if (type == null) {
1397           type = getNativeType(UNKNOWN_TYPE);
1398           TypedVar var = t.getTypedScope().getVar(n.getString());
1399           if (var != null) {
1400             JSType varType = var.getType();
1401             if (varType != null) {
1402               type = varType;
1403             }
1404           }
1405         }
1406         ensureTyped(t, n, type);
1407         return true;
1408       }

1436       /**
1437        * Emit a warning if we can prove that a property cannot possibly be
1438        * defined on an object. Note the difference between JS and a strictly
1439        * statically typed language: we're checking if the property
1440        * *cannot be defined*, whereas a java compiler would check if the
1441        * property *can be undefined*.
1442        */
1443       private void checkPropertyAccess(JSType childType, String propName,
1444           NodeTraversal t, Node n) {
1445         // If the property type is unknown, check the object type to see if it
1446         // can ever be defined. We explicitly exclude CHECKED_UNKNOWN (for
1447         // properties where we've checked that it exists, or for properties on
1448         // objects that aren't in this binary).
1449         JSType propType = getJSType(n);
1450         if (propType.isEquivalentTo(typeRegistry.getNativeType(UNKNOWN_TYPE))) {
1451           childType = childType.autobox();
1452           ObjectType objectType = ObjectType.cast(childType);
1453           if (objectType != null) {
1454             // We special-case object types so that checks on enums can be
1455             // much stricter, and so that we can use hasProperty (which is much
1456             // faster in most cases).
1457             if (!objectType.hasProperty(propName) ||
1458                 objectType.isEquivalentTo(
1459                     typeRegistry.getNativeType(UNKNOWN_TYPE))) {
1460               if (objectType instanceof EnumType) {
1461                 report(t, n, INEXISTENT_ENUM_ELEMENT, propName);
1462               } else {
1463                 checkPropertyAccessHelper(objectType, propName, t, n);
1464               }
1465             }
1466     
1467           } else {
1468             checkPropertyAccessHelper(childType, propName, t, n);
1469           }
1470         }
1471       }

1475         if (!objectType.isEmptyType() && reportMissingProperties

1571       /**
1572        * Check whether there's any property conflict for for a particular super
1573        * interface
1574        * @param t The node traversal object that supplies context
1575        * @param n The node being visited
1576        * @param functionName The function name being checked
1577        * @param properties The property names in the super interfaces that have
1578        * been visited
1579        * @param currentProperties The property names in the super interface
1580        * that have been visited
1581        * @param interfaceType The super interface that is being visited
1582        */
1583       private void checkInterfaceConflictProperties(NodeTraversal t, Node n,
1584           String functionName, Map<String, ObjectType> properties,
1585           Map<String, ObjectType> currentProperties,
1586           ObjectType interfaceType) {
1587         ObjectType implicitProto = interfaceType.getImplicitPrototype();
1588         Set<String> currentPropertyNames;
1589         if (implicitProto == null) {
1590           // This can be the case if interfaceType is proxy to a non-existent
1591           // object (which is a bad type annotation, but shouldn't crash).
1592           currentPropertyNames = ImmutableSet.of();
1593         } else {
1594           currentPropertyNames = implicitProto.getOwnPropertyNames();
1595         }
1596         for (String name : currentPropertyNames) {
1597           ObjectType oType = properties.get(name);
1598           currentProperties.put(name, interfaceType);
1599           if (oType != null) {
1600             JSType thisPropType = interfaceType.getPropertyType(name);
1601             JSType oPropType = oType.getPropertyType(name);
1602             if (thisPropType.isEquivalentTo(oPropType)
1603                 || thisPropType.isFunctionType() && oPropType.isFunctionType()
1604                    && thisPropType.toMaybeFunctionType().hasEqualCallType(
1605                       oPropType.toMaybeFunctionType())) {
1606               continue;
1607             }
1608             compiler.report(
1609                 t.makeError(n, INCOMPATIBLE_EXTENDED_PROPERTY_TYPE,
1610                     functionName, name, oType.toString(),
1611                     interfaceType.toString()));
1612           }
1613         }
1614         for (ObjectType iType : interfaceType.getCtorExtendedInterfaces()) {
1615           checkInterfaceConflictProperties(t, n, functionName, properties,
1616               currentProperties, iType);
1617         }
1618       }

1620       /**
1621        * Visits a {@link Token#FUNCTION} node.
1622        *
1623        * @param t The node traversal object that supplies context, such as the
1624        * scope chain to use in name lookups as well as error reporting.
1625        * @param n The node being visited.
1626        */
1627       private void visitFunction(NodeTraversal t, Node n) {
1628         FunctionType functionType = JSType.toMaybeFunctionType(n.getJSType());
1629         String functionPrivateName = n.getFirstChild().getString();
1630         if (functionType.isConstructor()) {
1631           FunctionType baseConstructor = functionType.getSuperClassConstructor();
1632           if (baseConstructor != getNativeType(OBJECT_FUNCTION_TYPE) &&
1633               baseConstructor != null &&
1634               baseConstructor.isInterface()) {
1635             compiler.report(
1636                 t.makeError(n, CONFLICTING_EXTENDED_TYPE,
1637                             "constructor", functionPrivateName));
1638           } else {
1639             // All interfaces are properly implemented by a class
1640             for (JSType baseInterface : functionType.getImplementedInterfaces()) {
1641               boolean badImplementedType = false;
1642               ObjectType baseInterfaceObj = ObjectType.cast(baseInterface);
1643               if (baseInterfaceObj != null) {
1644                 FunctionType interfaceConstructor =
1645                   baseInterfaceObj.getConstructor();
1646                 if (interfaceConstructor != null &&
1647                     !interfaceConstructor.isInterface()) {
1648                   badImplementedType = true;
1649                 }
1650               } else {
1651                 badImplementedType = true;
1652               }
1653               if (badImplementedType) {
1654                 report(t, n, BAD_IMPLEMENTED_TYPE, functionPrivateName);
1655               }
1656             }
1657             // check properties
1658             validator.expectAllInterfaceProperties(t, n, functionType);
1659           }
1660         } else if (functionType.isInterface()) {
1661           // Interface must extend only interfaces
1662           for (ObjectType extInterface : functionType.getExtendedInterfaces()) {
1663             if (extInterface.getConstructor() != null
1664                 && !extInterface.getConstructor().isInterface()) {
1665               compiler.report(
1666                   t.makeError(n, CONFLICTING_EXTENDED_TYPE,
1667                               "interface", functionPrivateName));
1668             }
1669           }
1670     
1671           // Check whether the extended interfaces have any conflicts
1672           if (functionType.getExtendedInterfacesCount() > 1) {
1673             // Only check when extending more than one interfaces
1674             HashMap<String, ObjectType> properties
1675                 = new HashMap<>();
1676             LinkedHashMap<String, ObjectType> currentProperties
1677                 = new LinkedHashMap<>();
1678             for (ObjectType interfaceType : functionType.getExtendedInterfaces()) {
1679               currentProperties.clear();
1680               checkInterfaceConflictProperties(t, n, functionPrivateName,
1681                   properties, currentProperties, interfaceType);
1682               properties.putAll(currentProperties);
1683             }
1684           }
1685     
1686           List<FunctionType> loopPath = functionType.checkExtendsLoop();
1687           if (loopPath != null) {
1688             String strPath = "";
1689             for (int i = 0; i < loopPath.size() - 1; i++) {
1690               strPath += loopPath.get(i).getDisplayName() + " -> ";
1691             }
1692             strPath += Iterables.getLast(loopPath).getDisplayName();
1693             compiler.report(t.makeError(n, INTERFACE_EXTENDS_LOOP,
1694                 loopPath.get(0).getDisplayName(), strPath));
1695           }
1696         }
1697       }

1722        * Visits a CALL node.
1723        *
1724        * @param t The node traversal object that supplies context, such as the
1725        * scope chain to use in name lookups as well as error reporting.
1726        * @param n The node being visited.
1727        */
1728       private void visitCall(NodeTraversal t, Node n) {
1729         checkCallConventions(t, n);
1730     
1731         Node child = n.getFirstChild();
1732         JSType childType = getJSType(child).restrictByNotNullOrUndefined();
1733     
1734         if (!childType.canBeCalled()) {
1735           report(t, n, NOT_CALLABLE, childType.toString());
1736           ensureTyped(t, n);
1737           return;
1738         }
1739     
1740         // A couple of types can be called as if they were functions.
1741         // If it is a function type, then validate parameters.
1742         if (childType.isFunctionType()) {
1743           FunctionType functionType = childType.toMaybeFunctionType();
1744     
1745           // Non-native constructors should not be called directly
1746           // unless they specify a return type
1747           if (functionType.isConstructor() &&
1748               !functionType.isNativeObjectType() &&
1749               (functionType.getReturnType().isUnknownType() ||
1750                functionType.getReturnType().isVoidType())) {
1751             report(t, n, CONSTRUCTOR_NOT_CALLABLE, childType.toString());
1752           }
1753     
1754           // Functions with explicit 'this' types must be called in a GETPROP
1755           // or GETELEM.
1756           if (functionType.isOrdinaryFunction() &&
1757               !functionType.getTypeOfThis().isUnknownType() &&
1758               !(functionType.getTypeOfThis().toObjectType() != null &&
1759               functionType.getTypeOfThis().toObjectType().isNativeObjectType()) &&
1760               !(child.isGetElem() ||
1761                 child.isGetProp())) {
1762             report(t, n, EXPECTED_THIS_TYPE, functionType.toString());
1763           }
1764     
1765           visitParameterList(t, n, functionType);
1766           ensureTyped(t, n, functionType.getReturnType());
1767         } else {
1768           ensureTyped(t, n);
1769         }
1770     
1771         // TODO(nicksantos): Add something to check for calls of RegExp objects,
1772         // which is not supported by IE. Either say something about the return type
1773         // or warn about the non-portability of the call or both.
1774       }

1776       /**
1777        * Visits the parameters of a CALL or a NEW node.
1778        */
1779       private void visitParameterList(NodeTraversal t, Node call,
1780           FunctionType functionType) {
1781         Iterator<Node> arguments = call.children().iterator();
1782         arguments.next(); // skip the function name
1783     
1784         Iterator<Node> parameters = functionType.getParameters().iterator();
1785         int ordinal = 0;
1786         Node parameter = null;
1787         Node argument = null;
1788         while (arguments.hasNext() &&
1789                (parameters.hasNext() ||
1790                 parameter != null && parameter.isVarArgs())) {
1791           // If there are no parameters left in the list, then the while loop
1792           // above implies that this must be a var_args function.
1793           if (parameters.hasNext()) {
1794             parameter = parameters.next();
1795           }
1796           argument = arguments.next();
1797           ordinal++;
1798     
1799           validator.expectArgumentMatchesParameter(t, argument,
1800               getJSType(argument), getJSType(parameter), call, ordinal);
1801         }
1802     
1803         int numArgs = call.getChildCount() - 1;
1804         int minArgs = functionType.getMinArguments();
1805         int maxArgs = functionType.getMaxArguments();
1806         if (minArgs > numArgs || maxArgs < numArgs) {
1807           report(t, call, WRONG_ARGUMENT_COUNT,
1808                   typeRegistry.getReadableTypeNameNoDeref(call.getFirstChild()),
1809                   String.valueOf(numArgs), String.valueOf(minArgs),
1810                   maxArgs != Integer.MAX_VALUE ?
1811                   " and no more than " + maxArgs + " argument(s)" : "");
1812         }
1813       }

1852       /**
1853        * This function unifies the type checking involved in the core binary
1854        * operators and the corresponding assignment operators.  The representation
1855        * used internally is such that common code can handle both kinds of
1856        * operators easily.
1857        *
1858        * @param op The operator.
1859        * @param t The traversal object, needed to report errors.
1860        * @param n The node being checked.
1861        */
1862       private void visitBinaryOperator(Token op, NodeTraversal t, Node n) {
1863         Node left = n.getFirstChild();
1864         JSType leftType = getJSType(left);
1865         Node right = n.getLastChild();
1866         JSType rightType = getJSType(right);
1867         switch (op) {
1868           case ASSIGN_LSH:
1869           case ASSIGN_RSH:
1870           case LSH:
1871           case RSH:
1872           case ASSIGN_URSH:
1873           case URSH:
1874             if (!leftType.matchesInt32Context()) {
1875               report(t, left, BIT_OPERATION,
1876                        NodeUtil.opToStr(n.getType()), leftType.toString());
1877             }
1878             if (!rightType.matchesUint32Context()) {
1879               report(t, right, BIT_OPERATION,
1880                        NodeUtil.opToStr(n.getType()), rightType.toString());
1881             }
1882             break;
1883     
1884           case ASSIGN_DIV:
1885           case ASSIGN_MOD:
1886           case ASSIGN_MUL:
1887           case ASSIGN_SUB:
1888           case DIV:
1889           case MOD:
1890           case MUL:
1891           case SUB:
1892             validator.expectNumber(t, left, leftType, "left operand");
1893             validator.expectNumber(t, right, rightType, "right operand");
1894             break;
1895     
1896           case ASSIGN_BITAND:
1897           case ASSIGN_BITXOR:
1898           case ASSIGN_BITOR:
1899           case BITAND:
1900           case BITXOR:
1901           case BITOR:
1902             validator.expectBitwiseable(t, left, leftType,
1903                 "bad left operand to bitwise operator");
1904             validator.expectBitwiseable(t, right, rightType,
1905                 "bad right operand to bitwise operator");
1906             break;
1907     
1908           case ASSIGN_ADD:
1909           case ADD:
1910             break;

1913             report(t, n, UNEXPECTED_TOKEN, op.toString());
1914         }
1915         ensureTyped(t, n);

1921        *
1922        * <p>We verify that the enum element type of the enum used
1923        * for initialization is a subtype of the enum element type of
1924        * the enum the value is being copied in.</p>
1925        *
1926        * <p>Example:</p>
1927        * <pre>var myEnum = myOtherEnum;</pre>
1928        *
1929        * <p>Enum aliases are irregular, so we need special code for this :(</p>
1930        *
1931        * @param value the value used for initialization of the enum
1932        */
1933       private void checkEnumAlias(
1934           NodeTraversal t, JSDocInfo declInfo, Node value) {
1935         if (declInfo == null || !declInfo.hasEnumParameterType()) {
1936           return;
1937         }
1938     
1939         JSType valueType = getJSType(value);
1940         if (!valueType.isEnumType()) {
1941           return;
1942         }

1944         EnumType valueEnumType = valueType.toMaybeEnumType();

1946             valueEnumType.getElementsType().getPrimitiveType();
1947         validator.expectCanAssignTo(t, value, valueEnumPrimitiveType,
1948             declInfo.getEnumParameterType().evaluate(t.getTypedScope(), typeRegistry),
1949             "incompatible enum element types");
1950       }

1987       /**
1988        * Enforces type casts, and ensures the node is typed.
1989        *
1990        * A cast in the way that we use it in JSDoc annotations never
1991        * alters the generated code and therefore never can induce any runtime
1992        * operation. What this means is that a 'cast' is really just a compile
1993        * time constraint on the underlying value. In the future, we may add
1994        * support for run-time casts for compiled tests.
1995        *
1996        * To ensure some shred of sanity, we enforce the notion that the
1997        * type you are casting to may only meaningfully be a narrower type
1998        * than the underlying declared type. We also invalidate optimizations
1999        * on bad type casts.
2000        *
2001        * @param t The traversal object needed to report errors.
2002        * @param n The node getting a type assigned to it.
2003        * @param type The type to be assigned.
2004        */
2005       private void ensureTyped(NodeTraversal t, Node n, JSType type) {
2006         // Make sure FUNCTION nodes always get function type.
2007         Preconditions.checkState(!n.isFunction() ||
2008                 type.isFunctionType() ||
2009                 type.isUnknownType());
2010         // TODO(johnlenz): this seems like a strange place to check "@implicitCast"
2011         JSDocInfo info = n.getJSDocInfo();
2012         if (info != null && (info.isImplicitCast() && !inExterns)) {
2013           String propName = n.isGetProp() ? n.getLastChild().getString() : "(missing)";
2014           compiler.report(t.makeError(n, ILLEGAL_IMPLICIT_CAST, propName));
2015         }
2016     
2017         if (n.getJSType() == null) {
2018           n.setJSType(type);
2019         }
2020       }

2062       /**
2063        * Checks whether type is stringifiable. Stringifiable is a type that can be converted to string
2064        * and give unique results for different objects. For example objects have native toString()
2065        * method that on chrome returns "[object Object]" for all objects making it useless when used
2066        * as keys. At the same time native types like numbers can be safely converted to strings and
2067        * used as keys. Also user might have provided custom toString() methods for a class making it
2068        * suitable for using as key.
2069        */
2070       private boolean isStringifiable(JSType type) {
2071         // Check built-in types
2072         if (type.isUnknownType() || type.isNumber() || type.isString() || type.isBooleanObjectType()
2073             || type.isBooleanValueType() || type.isDateType() || type.isRegexpType()
2074             || type.isInterface() || type.isRecordType() || type.isNullType() || type.isVoidType()) {
2075           return true;
2076         }
2077     
2078         // For enums check that underlying type is stringifiable.
2079         if (type.toMaybeEnumElementType() != null) {
2080           return isStringifiable(type.toMaybeEnumElementType().getPrimitiveType());
2081         }
2082     
2083         // Array is stringifiable if it doesn't have template type or if it does have it, the template
2084         // type must be also stringifiable.
2085         // Good: Array, Array.<number>
2086         // Bad: Array.<!Object>
2087         if (type.isArrayType()) {
2088           return true;
2089         }
2090         if (type.isTemplatizedType()) {
2091           TemplatizedType templatizedType = type.toMaybeTemplatizedType();
2092           if (templatizedType.getReferencedType().isArrayType()) {
2093             return isStringifiable(templatizedType.getTemplateTypes().get(0));
2094           }
2095         }
2096     
2097         // Named types are usually @typedefs. For such types we need to check underlying type specified
2098         // in @typedef annotation.
2099         if (type instanceof NamedType) {
2100           return isStringifiable(((NamedType) type).getReferencedType());
2101         }
2102     
2103         // For union type every alternate must be stringifiable.
2104         if (type.isUnionType()) {
2105           for (JSType alternateType : type.toMaybeUnionType().getAlternates()) {
2106             if (!isStringifiable(alternateType)) {
2107               return false;
2108             }
2109           }
2110           return true;
2111         }
2112     
2113         // Handle interfaces and classes.
2114         if (type.isObject()) {
2115           ObjectType objectType = type.toMaybeObjectType();
2116           JSType constructor = objectType.getConstructor();
2117           // Interfaces considered stringifiable as user might implement toString() method in
2118           // classes-implementations.
2119           if (constructor != null && constructor.isInterface()) {
2120             return true;
2121           }
2122           // This is user-defined class so check if it has custom toString() method.
2123           return classHasToString(objectType);
2124         }
2125         return false;
2126       }

2128       /**
2129        * Checks whether current type is Object type with non-stringifable key.
2130        */
2131       private boolean isObjectTypeWithNonStringifiableKey(JSType type) {
2132         if (!type.isTemplatizedType()) {
2133           return false;
2134         }
2135         TemplatizedType templatizedType = type.toMaybeTemplatizedType();
2136         if (templatizedType.getReferencedType().isNativeObjectType()
2137             && templatizedType.getTemplateTypes().size() > 1) {
2138           return !isStringifiable(templatizedType.getTemplateTypes().get(0));
2139         } else {
2140           return false;
2141         }
2142       }

2144       /**
2145        * Checks whether type (or one of its component if is composed type like union or templatized
2146        * type) has Object with non-stringifiable key. For example {@code Object.<!Object, number>}.
2147        *
2148        * @return non-stringifiable type which is used as key or null if all there are no such types.
2149        */
2150       private JSType findObjectWithNonStringifiableKey(JSType type) {
2151         if (isObjectTypeWithNonStringifiableKey(type)) {
2152           return type;
2153         }
2154         if (type.isUnionType()) {
2155           for (JSType alternateType : type.toMaybeUnionType().getAlternates()) {
2156             JSType result = findObjectWithNonStringifiableKey(alternateType);
2157             if (result != null) {
2158               return result;
2159             }
2160           }
2161         }
2162         if (type.isTemplatizedType()) {
2163           for (JSType templateType : type.toMaybeTemplatizedType().getTemplateTypes()) {
2164             JSType result = findObjectWithNonStringifiableKey(templateType);
2165             if (result != null) {
2166               return result;
2167             }
2168           }
2169         }
2170         if (type.isOrdinaryFunction()) {
2171           FunctionType function = type.toMaybeFunctionType();
2172           for (Node parameter : function.getParameters()) {
2173             JSType result = findObjectWithNonStringifiableKey(parameter.getJSType());
2174             if (result != null) {
2175               return result;
2176             }
2177           }
2178           return findObjectWithNonStringifiableKey(function.getReturnType());
2179         }
2180         return null;
2181       }

2185        * method but we ignore it as it is not useful so we need user-provided toString() method.
2186        */
2187       private boolean classHasToString(ObjectType type) {
2188         Property toStringProperty = type.getOwnSlot("toString");
2189         if (toStringProperty != null) {
2190           return toStringProperty.getType().isFunctionType();
2191         }
2192         ObjectType parent = type.getParentScope();
2193         if (parent != null && !parent.isNativeObjectType()) {
2194           return classHasToString(parent);
2195         }
2196         return false;
