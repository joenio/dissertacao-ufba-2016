238       public void testModules() {
239         String module1Js = "function Bar(){} Bar.prototype.getA=function(x){};" +
240                            "var foo;foo.getA(foo);foo.doo=foo;foo.bloo=foo;";
241     
242         String module2Js = "function Far(){} Far.prototype.getB=function(x){};" +
243                            "var too;too.getB(too);too.woo=too;too.bloo=too;";
244     
245         String module3Js = "function Car(){} Car.prototype.getC=function(x){};" +
246                            "var noo;noo.getC(noo);noo.zoo=noo;noo.cloo=noo;";
247     
248         JSModule module1 = new JSModule("m1");
249         module1.add(SourceFile.fromCode("input1", module1Js));
250     
251         JSModule module2 = new JSModule("m2");
252         module2.add(SourceFile.fromCode("input2", module2Js));
253     
254         JSModule module3 = new JSModule("m3");
255         module3.add(SourceFile.fromCode("input3", module3Js));
256     
257         JSModule[] modules = new JSModule[] { module1, module2, module3 };
258         Compiler compiler = compileModules("", modules);
259     
260         Result result = compiler.getResult();
261         assertTrue(result.success);
262     
263         assertEquals("function Bar(){}Bar.prototype.b=function(x){};" +
264                      "var foo;foo.b(foo);foo.f=foo;foo.a=foo;",
265                      compiler.toSource(module1));
266     
267         assertEquals("function Far(){}Far.prototype.c=function(x){};" +
268                      "var too;too.c(too);too.g=too;too.a=too;",
269                      compiler.toSource(module2));
270     
271         // Note that properties that occur most often globally get the earliest
272         // names. The "getC" property, which doesn't occur until module 3, is
273         // renamed to an earlier name in the alphabet than "woo", which appears
274         // in module 2, because "getC" occurs more total times across all modules.
275         // Might be better to give early modules the shortest names, but this is
276         // how the pass currently works.
277         assertEquals("function Car(){}Car.prototype.d=function(x){};" +
278                      "var noo;noo.d(noo);noo.h=noo;noo.e=noo;",
279                      compiler.toSource(module3));
280       }

394       private void testStableRenaming(String input1, String expected1,
395                                       String input2, String expected2) {

398         test(input2, expected2);

401       private Compiler compileModules(String externs, JSModule[] modules) {
402         SourceFile externsInput = SourceFile.fromCode("externs", externs);
403     
404         CompilerOptions options = new CompilerOptions();
405         options.setPropertyRenaming(PropertyRenamingPolicy.ALL_UNQUOTED);
406     
407         Compiler compiler = new Compiler();
408         compiler.compileModules(
409             ImmutableList.of(externsInput), ImmutableList.copyOf(modules), options);
410         return compiler;
411       }
