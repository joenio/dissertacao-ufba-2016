16     package com.google.javascript.jscomp;
17     
18     import static com.google.common.collect.ObjectArrays.concat;
19     
20     import com.google.common.collect.ImmutableList;
21     import com.google.javascript.jscomp.CompilerOptions.LanguageMode;
22     import com.google.javascript.refactoring.ApplySuggestedFixes;
23     import com.google.javascript.refactoring.ErrorToFixMapper;
24     import com.google.javascript.refactoring.SuggestedFix;
25     
26     import java.io.IOException;
27     import java.nio.file.Path;
28     import java.nio.file.Paths;
29     import java.util.ArrayList;
30     import java.util.List;
31     
32     /**
33      * Minimal binary that just runs the "lint" checks which can be run on a single file at a time.
34      * This means some checks in the lintChecks DiagnosticGroup are skipped, since they depend on
35      * type information.
36      */
37     public class Linter {
38       public static void main(String[] args) throws IOException {
39         for (String filename : args) {
40           // TODO(tbreisacher): Add a command line flag that causes this
41           // to call fix() instead of lint().
42           lint(filename);
43         }
44       }
45     
46     
47       static void lint(String filename) throws IOException {
48         lint(Paths.get(filename), false);
49       }
50     
51       static void fix(String filename) throws IOException {
52         lint(Paths.get(filename), true);
53       }
54     
55       private static void lint(Path path, boolean fix) throws IOException {
56         SourceFile file = SourceFile.fromFile(path.toString());
57         Compiler compiler = new Compiler(System.out);
58         CompilerOptions options = new CompilerOptions();
59         options.setLanguage(LanguageMode.ECMASCRIPT6_STRICT);
60     
61         // For a full compile, this would cause a crash, as the method name implies. But the passes
62         // in LintPassConfig can all handle untranspiled ES6.
63         options.setSkipTranspilationAndCrash(true);
64     
65         options.setPreserveDetailedSourceInfo(true);
66         options.setCodingConvention(new GoogleCodingConvention());
67     
68         // Even though we're not running the typechecker, enable the checkTypes DiagnosticGroup, since
69         // it contains some warnings we do want to report, such as JSDoc parse warnings.
70         options.setWarningLevel(DiagnosticGroups.CHECK_TYPES, CheckLevel.WARNING);
71     
72         options.setWarningLevel(DiagnosticGroups.LINT_CHECKS, CheckLevel.WARNING);
73         options.setWarningLevel(DiagnosticGroups.MISSING_REQUIRE, CheckLevel.WARNING);
74         options.setWarningLevel(DiagnosticGroups.EXTRA_REQUIRE, CheckLevel.WARNING);
75         options.setWarningLevel(DiagnosticGroups.USE_OF_GOOG_BASE, CheckLevel.WARNING);
76         compiler.setPassConfig(new LintPassConfig(options));
77         compiler.disableThreads();
78         SourceFile externs = SourceFile.fromCode("<Linter externs>", "");
79         compiler.compile(ImmutableList.<SourceFile>of(externs), ImmutableList.of(file), options);
80         if (fix) {
81           List<SuggestedFix> fixes = new ArrayList<>();
82           for (JSError warning : concat(compiler.getErrors(), compiler.getWarnings(), JSError.class)) {
83             SuggestedFix suggestedFix = ErrorToFixMapper.getFixForJsError(warning, compiler);
84             if (suggestedFix != null) {
85               fixes.add(suggestedFix);
86             }
87           }
88           ApplySuggestedFixes.applySuggestedFixesToFiles(fixes);
89         }
90       }
91     }
