35     /**
36      * Maps a JSError to a SuggestedFix.
37      * TODO(tbreisacher): Move this into the compiler itself (i.e. into the jscomp package). This will
38      *     make it easier for people adding new warnings to also add fixes for them.
39      */
40     public final class ErrorToFixMapper {
41       private ErrorToFixMapper() {} // All static
42     
43       private static final Pattern DID_YOU_MEAN = Pattern.compile(".*Did you mean (.*)\\?");
44       private static final Pattern MISSING_REQUIRE =
45           Pattern.compile("missing require: '([^']+)'");
46       private static final Pattern EXTRA_REQUIRE =
47           Pattern.compile("extra require: '([^']+)'");
48     
49       public static List<SuggestedFix> getFixesForJsError(JSError error, AbstractCompiler compiler) {
50         SuggestedFix fix = getFixForJsError(error, compiler);
51         if (fix != null) {
52           return ImmutableList.of(fix);
53         }
54         switch (error.getType().key) {
55           case "JSC_IMPLICITLY_NULLABLE_JSDOC":
56             return getFixesForImplicitlyNullableJsDoc(error);
57           default:
58             return ImmutableList.of();
59         }
60       }
61     
62       /**
63        * Creates a SuggestedFix for the given error. Note that some errors have multiple fixes
64        * so getFixesForJsError should often be used instead of this.
65        */
66       public static SuggestedFix getFixForJsError(JSError error, AbstractCompiler compiler) {
67         switch (error.getType().key) {
68           case "JSC_MISSING_SEMICOLON":
69             return getFixForMissingSemicolon(error);
70           case "JSC_REQUIRES_NOT_SORTED":
71             return getFixForUnsortedRequiresOrProvides("goog.require", error, compiler);
72           case "JSC_PROVIDES_NOT_SORTED":
73             return getFixForUnsortedRequiresOrProvides("goog.provide", error, compiler);
74           case "JSC_DEBUGGER_STATEMENT_PRESENT":
75           case "JSC_USELESS_EMPTY_STATEMENT":
76             return removeNode(error);
77           case "JSC_INEXISTENT_PROPERTY":
78             return getFixForInexistentProperty(error);
79           case "JSC_MISSING_CALL_TO_SUPER":
80             return getFixForMissingSuper(error);
81           case "JSC_INVALID_SUPER_CALL_WITH_SUGGESTION":
82             return getFixForInvalidSuper(error, compiler);
83           case "JSC_MISSING_REQUIRE_WARNING":
84           case "JSC_MISSING_REQUIRE_CALL_WARNING":
85             return getFixForMissingRequire(error, compiler);
86           case "JSC_DUPLICATE_REQUIRE_WARNING":
87           case "JSC_EXTRA_REQUIRE_WARNING":
88             return getFixForExtraRequire(error, compiler);
89           default:
90             return null;
91         }
92       }
93     
94       private static List<SuggestedFix> getFixesForImplicitlyNullableJsDoc(JSError error) {
95         SuggestedFix qmark = new SuggestedFix.Builder()
96             .setOriginalMatchedNode(error.node)
97             .insertBefore(error.node, "?")
98             .setDescription("Make nullability explicit")
99             .build();
100         SuggestedFix bang = new SuggestedFix.Builder()
101             .setOriginalMatchedNode(error.node)
102             .insertBefore(error.node, "!")
103             .setDescription("Make type non-nullable")
104             .build();
105         return ImmutableList.of(bang, qmark);
106       }
107     
108       private static SuggestedFix removeNode(JSError error) {
109         return new SuggestedFix.Builder()
110             .setOriginalMatchedNode(error.node)
111             .delete(error.node).build();
112       }
113     
114       private static SuggestedFix getFixForMissingSemicolon(JSError error) {
115         return new SuggestedFix.Builder()
116             .setOriginalMatchedNode(error.node)
117             .insertAfter(error.node, ";")
118             .build();
119       }
120     
121       private static SuggestedFix getFixForMissingSuper(JSError error) {
122         Node body = NodeUtil.getFunctionBody(error.node);
123         return new SuggestedFix.Builder()
124             .setOriginalMatchedNode(error.node)
125             .addChildToFront(body, "super();")
126             .build();
127       }
128     
129       private static SuggestedFix getFixForInvalidSuper(JSError error, AbstractCompiler compiler) {
130         Matcher m = DID_YOU_MEAN.matcher(error.description);
131         if (m.matches()) {
132           return new SuggestedFix.Builder()
133               .setOriginalMatchedNode(error.node)
134               .replace(error.node, NodeUtil.newQName(compiler, m.group(1)), compiler)
135               .build();
136         }
137         return null;
138       }
139     
140       private static SuggestedFix getFixForInexistentProperty(JSError error) {
141         Matcher m = DID_YOU_MEAN.matcher(error.description);
142         if (m.matches()) {
143           String suggestedPropName = m.group(1);
144           return new SuggestedFix.Builder()
145               .setOriginalMatchedNode(error.node)
146               .rename(error.node, suggestedPropName).build();
147         }
148         return null;
149       }
150     
151       private static SuggestedFix getFixForMissingRequire(JSError error, AbstractCompiler compiler) {
152         Matcher regexMatcher = MISSING_REQUIRE.matcher(error.description);
153         Preconditions.checkState(regexMatcher.matches(),
154             "Unexpected error description: %s", error.description);
155         String namespaceToRequire = regexMatcher.group(1);
156         NodeMetadata metadata = new NodeMetadata(compiler);
157         Match match = new Match(error.node, metadata);
158         return new SuggestedFix.Builder()
159             .setOriginalMatchedNode(error.node)
160             .addGoogRequire(match, namespaceToRequire)
161             .build();
162       }
163     
164       private static SuggestedFix getFixForExtraRequire(JSError error, AbstractCompiler compiler) {
165         Matcher regexMatcher = EXTRA_REQUIRE.matcher(error.description);
166         Preconditions.checkState(regexMatcher.matches(),
167             "Unexpected error description: %s", error.description);
168         String namespace = regexMatcher.group(1);
169         NodeMetadata metadata = new NodeMetadata(compiler);
170         Match match = new Match(error.node, metadata);
171         return new SuggestedFix.Builder()
172             .setOriginalMatchedNode(error.node)
173             .removeGoogRequire(match, namespace)
174             .build();
175       }
176     
177       private static SuggestedFix getFixForUnsortedRequiresOrProvides(
178           String closureFunction, JSError error, AbstractCompiler compiler) {
179         SuggestedFix.Builder fix = new SuggestedFix.Builder();
180         fix.setOriginalMatchedNode(error.node);
181         Node script = NodeUtil.getEnclosingScript(error.node);
182         RequireProvideSorter cb = new RequireProvideSorter(closureFunction);
183         NodeTraversal.traverseEs6(compiler, script, cb);
184         Node first = cb.calls.get(0);
185         Node last = Iterables.getLast(cb.calls);
186     
187         cb.sortCallsAlphabetically();
188         StringBuilder sb = new StringBuilder();
189         for (Node n : cb.calls) {
190           String statement = fix.generateCode(compiler, n);
191           JSDocInfo jsDoc = NodeUtil.getBestJSDocInfo(n);
192           if (jsDoc != null) {
193             statement = jsDoc.getOriginalCommentString() + "\n" + statement;
194           }
195           sb.append(statement);
196         }
197         // Trim to remove the newline after the last goog.require/provide.
198         String newContent = sb.toString().trim();
199         return fix.replaceRange(first, last, newContent).build();
200       }

202       private static String getNamespaceFromClosureNode(Node exprResult) {
203         Preconditions.checkState(exprResult.isExprResult());
204         return exprResult.getFirstChild().getLastChild().getString();
205       }
206     
207       private static class RequireProvideSorter extends NodeTraversal.AbstractShallowCallback
208           implements Comparator<Node> {
209         private final String closureFunction;
210         private final List<Node> calls = new ArrayList<>();
211     
212         RequireProvideSorter(String closureFunction) {
213           this.closureFunction = closureFunction;
214         }
215     
216         @Override
217         public final void visit(NodeTraversal nodeTraversal, Node n, Node parent) {
218           if (n.isCall()
219               && parent.isExprResult()
220               && n.getFirstChild().matchesQualifiedName(closureFunction)) {
221             calls.add(parent);
222           }
223         }
224     
225         public void sortCallsAlphabetically() {
226           Collections.sort(calls, this);
227         }
228     
229         @Override
230         public int compare(Node n1, Node n2) {
231           String namespace1 = getNamespaceFromClosureNode(n1);
232           String namespace2 = getNamespaceFromClosureNode(n2);
233           return namespace1.compareTo(namespace2);
234         }
235       }
