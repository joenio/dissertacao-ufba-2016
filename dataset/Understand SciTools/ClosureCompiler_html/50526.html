55     /**
56      * This pass identifies all global names, simple (e.g. <code>a</code>) or
57      * qualified (e.g. <code>a.b.c</code>), and the dependencies between them, then
58      * removes code associated with unreferenced names. It starts by assuming that
59      * only externally accessible names (e.g. <code>window</code>) are referenced,
60      * then iteratively marks additional names as referenced (e.g. <code>Foo</code>
61      * in <code>window['foo'] = new Foo();</code>). This makes it possible to
62      * eliminate code containing circular references.
63      *
64      * <p>Qualified names can be defined using dotted or object literal syntax
65      * (<code>a.b.c = x;</code> or <code>a.b = {c: x};</code>, respectively).
66      *
67      * <p>Removal of prototype classes is currently all or nothing. In other words,
68      * prototype properties and methods are never individually removed.
69      *
70      * <p>Optionally generates pretty HTML output of data so that it is easy to
71      * analyze dependencies.
72      *
73      * <p>Only operates on names defined in the global scope, but it would be easy
74      * to extend the pass to names defined in local scopes.
75      *
76      * TODO(nicksantos): In the initial implementation of this pass, it was
77      * important to understand namespaced names (e.g., that a.b is distinct from
78      * a.b.c). Now that this pass comes after CollapseProperties, this is no longer
79      * necessary. For now, I've changed so that {@code referenceParentNames}
80      * creates a two-way reference between a.b and a.b.c, so that they're
81      * effectively the same name. When someone has the time, we should completely
82      * rip out all the logic that understands namespaces.
83      *
84      */

276       /**
277        * Class for nodes that reference a fully-qualified JS name. Fully qualified
278        * names are of form A or A.B (A.B.C, etc.). References can get the value or
279        * set the value of the JS name.
280        */
281       private class JsNameRefNode implements RefNode {
282         /** JsName node for this reference */
283         JsName name;
284     
285         /**
286          * Parent node of the name access
287          * (ASSIGN, VAR, FUNCTION, OBJECTLIT, or CALL)
288          */
289         Node parent;
290     
291     
292         /**
293          * Create a node that refers to a name
294          *
295          * @param name The name
296          * @param node The top node representing the name (GETPROP, NAME, STRING)
297          */
298         JsNameRefNode(JsName name, Node node) {
299           this.name = name;
300           this.parent = node.getParent();
301         }
302     
303         @Override
304         public JsName name() {
305           return name;
306         }
307     
308         @Override
309         public void remove() {
310           // Setters have VAR, FUNCTION, or ASSIGN parent nodes. CALL parent
311           // nodes are global refs, and are handled later in this function.
312           Node containingNode = parent.getParent();
313           switch (parent.getType()) {
314             case VAR:
315               Preconditions.checkState(parent.hasOneChild());
316               replaceWithRhs(containingNode, parent);
317               break;
318             case FUNCTION:
319               replaceWithRhs(containingNode, parent);
320               break;
321             case ASSIGN:
322               if (containingNode.isExprResult()) {
323                 replaceWithRhs(containingNode.getParent(), containingNode);
324               } else {
325                 replaceWithRhs(containingNode, parent);
326               }
327               break;
328             case OBJECTLIT:
329               // TODO(nicksantos): Come up with a way to remove this.
330               // If we remove object lit keys, then we will need to also
331               // create dependency scopes for them.
332               break;
333             case EXPR_RESULT:
334               Preconditions.checkState(isAnalyzableObjectDefinePropertiesDefinition(parent.getFirstChild()));
335               replaceWithRhs(containingNode, parent);
336               break;
337             default:
338               throw new IllegalArgumentException(
339                   "Unsupported parent node type in JsNameRefNode.remove: " + parent.getType());
340           }
341         }
342       }

481       /**
482        * <p>Identifies all dependency scopes.
483        *
484        * <p>A dependency scope is a relationship between a node tree and a name that
485        * implies that the node tree will not execute (and thus can be eliminated) if
486        * the name is never referenced.
487        *
488        * <p>The entire parse tree is ultimately in a dependency scope relationship
489        * with <code>window</code> (or an equivalent name for the global scope), but
490        * the goal here is to find finer-grained relationships. This callback creates
491        * dependency scopes for every assignment statement, variable declaration, and
492        * function call in the global scope.
493        *
494        * <p>Note that dependency scope node trees aren't necessarily disjoint.
495        * In the following code snippet, for example, the function definition
496        * forms a dependency scope with the name <code>f</code> and the assignment
497        * inside the function forms a dependency scope with the name <code>x</code>.
498        * <pre>
499        * var x; function f() { x = 1; }
500        * </pre>
501        */
502       private class FindDependencyScopes extends AbstractPostOrderCallback {
503         @Override
504         public void visit(NodeTraversal t, Node n, Node parent) {
505           if (!t.inGlobalHoistScope()) {
506             return;
507           }
508     
509           if (n.isAssign()) {
510             recordAssignment(t, n, n);
511             if (!NodeUtil.isImmutableResult(n.getLastChild())) {
512               recordConsumers(t, n, n);
513             }
514           } else if (NodeUtil.isVarDeclaration(n)) {
515             NameInformation ns = createNameInformation(t, n);
516             recordDepScope(n, ns);
517           } else if (NodeUtil.isFunctionDeclaration(n) && t.inGlobalScope()) {
518             NameInformation ns = createNameInformation(t, n.getFirstChild());
519             recordDepScope(n, ns);
520           } else if (NodeUtil.isExprCall(n)) {
521             Node callNode = n.getFirstChild();
522             Node nameNode = callNode.getFirstChild();
523             NameInformation ns = createNameInformation(t, nameNode);
524             if (ns != null && ns.onlyAffectsClassDef) {
525               recordDepScope(n, ns);
526             }
527           } else if (isAnalyzableObjectDefinePropertiesDefinition(n)) {
528             Node targetObject = n.getSecondChild();
529             NameInformation ns = createNameInformation(t, targetObject);
530             recordDepScope(n, ns);
531           }
532         }
533     
534         private void recordConsumers(NodeTraversal t, Node n, Node recordNode) {
535           Node parent = n.getParent();
536           switch (parent.getType()) {
537             case ASSIGN:
538               if (n == parent.getLastChild()) {
539                 recordAssignment(t, parent, recordNode);
540               }
541               recordConsumers(t, parent, recordNode);
542               break;
543             case NAME:
544               NameInformation ns = createNameInformation(t, parent);
545               recordDepScope(recordNode, ns);
546               break;
547             case OR:
548               recordConsumers(t, parent, recordNode);
549               break;
550             case AND:
551               // In "a && b" only "b" can be meaningfully aliased.
552               // "a" must be falsy, which it must be an immutable, non-Object
553             case COMMA:
554             case HOOK:
555               if (n != parent.getFirstChild()) {
556                 recordConsumers(t, parent, recordNode);
557               }
558               break;
559           }
560         }
561     
562         private void recordAssignment(NodeTraversal t, Node n, Node recordNode) {
563           Node nameNode = n.getFirstChild();
564           Node parent = n.getParent();
565           NameInformation ns = createNameInformation(t, nameNode);
566           if (ns != null) {
567             if (parent.isFor() && !NodeUtil.isForIn(parent)) {
568               // Patch for assignments that appear in the init,
569               // condition or iteration part of a FOR loop.  Without
570               // this change, all 3 of those parts try to claim the for
571               // loop as their dependency scope.  The last assignment in
572               // those three fields wins, which can result in incorrect
573               // reference edges between referenced and assigned variables.
574               //
575               // TODO(user) revisit the dependency scope calculation
576               // logic.
577               if (parent.getSecondChild() != n) {
578                 recordDepScope(recordNode, ns);
579               } else {
580                 recordDepScope(nameNode, ns);
581               }
582             } else if (!parent.isCall() || n != parent.getFirstChild()) {
583               // The rhs of the assignment is the caller, so it's used by the
584               // context. Don't associate it w/ the lhs.
585               // FYI: this fixes only the specific case where the assignment is the
586               // caller expression, but it could be nested deeper in the caller and
587               // we would still get a bug.
588               // See testAssignWithCall2 for an example of this.
589               recordDepScope(recordNode, ns);
590             }
591           }
592         }

627       /**
628        * Identifies all declarations of global names and setter statements
629        * affecting global symbols (assignments to global names).
630        *
631        * All declarations and setters must be gathered in a single
632        * traversal and stored in traversal order so "removeUnreferenced"
633        * can perform modifications in traversal order.
634        */
635       private class FindDeclarationsAndSetters extends AbstractPostOrderCallback {
636     
637         @Override
638         public void visit(NodeTraversal t, Node n, Node parent) {
639     
640           // Record global variable and function declarations
641           if (t.inGlobalHoistScope()) {
642             if (NodeUtil.isVarDeclaration(n)) {
643               NameInformation ns = createNameInformation(t, n);
644               Preconditions.checkNotNull(ns);
645               recordSet(ns.name, n);
646             } else if (NodeUtil.isFunctionDeclaration(n) && t.inGlobalScope()) {
647               Node nameNode = n.getFirstChild();
648               NameInformation ns = createNameInformation(t, nameNode);
649               if (ns != null) {
650                 JsName nameInfo = getName(nameNode.getString(), true);
651                 recordSet(nameInfo.name, nameNode);
652               }
653             } else if (NodeUtil.isObjectLitKey(n)) {
654               NameInformation ns = createNameInformation(t, n);
655               if (ns != null) {
656                 recordSet(ns.name, n);
657               }
658             }
659           }
660     
661           // Record assignments and call sites
662           if (n.isAssign() || isAnalyzableObjectDefinePropertiesDefinition(n)) {
663             Node nameNode = n.isAssign() ? n.getFirstChild() : n;
664     
665             NameInformation ns = createNameInformation(t, nameNode);
666             if (ns != null) {
667               if (ns.isPrototype) {
668                 recordPrototypeSet(ns.prototypeClass, ns.prototypeProperty, n);
669               } else {
670                 recordSet(ns.name, nameNode);
671               }
672             }
673           } else if (n.isCall()) {
674             Node nameNode = n.getFirstChild();
675             NameInformation ns = createNameInformation(t, nameNode);
676             if (ns != null && ns.onlyAffectsClassDef) {
677               JsName name = getName(ns.name, true);
678               refNodes.add(new ClassDefiningFunctionNode(name, n));
679             }
680           }
681         }

722          * Record that the properties of this name have been written to.
723          */
724         private void recordWriteOnProperties(String parentName) {
725           do {
726             JsName parent = getName(parentName, true);
727             if (parent.hasWrittenDescendants) {
728               // If we already recorded this name, then all its parents must
729               // also be recorded. short-circuit this loop.
730               return;
731             } else {
732               parent.hasWrittenDescendants = true;
733             }
734     
735             if (parentName.indexOf('.') == -1) {
736               return;
737             }
738             parentName = parentName.substring(0, parentName.lastIndexOf('.'));
739           } while(true);
740         }

785         private void addSimplifiedExpression(Node n, Node parent) {
786           if (parent.isVar()) {
787             Node value = n.getFirstChild();
788             if (value != null) {
789               addSimplifiedChildren(value);
790             }
791           } else if (n.isAssign() &&
792               (parent.isExprResult() ||
793                parent.isFor() ||
794                parent.isReturn())) {
795             for (Node child : n.children()) {
796               addSimplifiedChildren(child);
797             }
798           } else if (isAnalyzableObjectDefinePropertiesDefinition(n)) {
799             addSimplifiedChildren(n.getLastChild());
800           } else if (n.isCall() && parent.isExprResult()) {
801             addSimplifiedChildren(n);
802           } else {
803             addAllChildren(n);
804           }
805         }

807         @Override
808         public boolean shouldTraverse(NodeTraversal t, Node n, Node parent) {
809           if (parent == null) {
810             return true;
811           }
812     
813           // Gather the list of nodes that either have side effects, are
814           // arguments to function calls with side effects or are used in
815           // control structure predicates.  These names are always
816           // referenced when the enclosing function is called.
817           if (n.isFor()) {
818             if (!NodeUtil.isForIn(n)) {
819               Node decl = n.getFirstChild();
820               Node pred = decl.getNext();
821               Node step = pred.getNext();
822               addSimplifiedExpression(decl, n);
823               addSimplifiedExpression(pred, n);
824               addSimplifiedExpression(step, n);
825             } else { // n.getChildCount() == 3
826               Node decl = n.getFirstChild();
827               Node iter = decl.getNext();
828               addAllChildren(decl);
829               addAllChildren(iter);
830             }
831           }
832     
833           if (parent.isVar() ||
834               parent.isExprResult() ||
835               parent.isReturn() ||
836               parent.isThrow()) {
837             addSimplifiedExpression(n, parent);
838           }
839     
840           if ((parent.isIf() ||
841                parent.isWhile() ||
842                parent.isWith() ||
843                parent.isSwitch() ||
844                parent.isCase()) &&
845               parent.getFirstChild() == n) {
846             addAllChildren(n);
847           }
848     
849           if (parent.isDo() && parent.getLastChild() == n) {
850             addAllChildren(n);
851           }
852     
853           return true;
854         }

856         @Override
857         public void visit(NodeTraversal t, Node n, Node parent) {
858           if (!n.isName() && (!NodeUtil.isGet(n) || parent.isGetProp())) {
859             // This is not a simple or qualified name.
860             return;
861           }
862     
863           NameInformation nameInfo = createNameInformation(t, n);
864           if (nameInfo == null) {
865             // The name is not a global name
866             return;
867           }
868     
869           if (nameInfo.onlyAffectsClassDef) {
870             if (nameInfo.superclass != null) {
871               recordReference(
872                   nameInfo.name, nameInfo.superclass, RefType.INHERITANCE);
873             }
874     
875             // Make sure that we record a reference to the function that does
876             // the inheritance, so that the inherits() function itself does
877             // not get stripped.
878             String nodeName = n.getQualifiedName();
879             if (nodeName != null) {
880               recordReference(
881                   nameInfo.name, nodeName, RefType.REGULAR);
882             }
883     
884             return;
885           }
886     
887           // instanceof checks are not handled like regular read references.
888           boolean isInstanceOfCheck = parent.isInstanceOf() &&
889               parent.getLastChild() == n;
890           if (isInstanceOfCheck) {
891             JsName checkedClass = getName(nameInfo.name, true);
892     
893             // If we know where this constructor is created, and we
894             // know we can find all 'new' calls on it, then treat
895             // this as a special reference. It will be replaced with
896             // false if there are no other references, because we
897             // know the class can't be instantiated.
898             if (checkedClass.hasSetterReference &&
899                 !nameInfo.isExternallyReferenceable &&
900                 // Exclude GETELEMs.
901                 n.isQualifiedName()) {
902               refNodes.add(new InstanceOfCheckNode(checkedClass, n));
903               checkedClass.hasInstanceOfReference = true;
904               return;
905             }
906           }
907     
908           // Determine which name might be potentially referring to this one by
909           // looking up the nearest enclosing dependency scope. It's unnecessary to
910           // determine all enclosing dependency scopes because this callback should
911           // create a chain of references between them.
912           List<NameInformation> referers = getDependencyScope(n);
913           if (referers.isEmpty()) {
914             maybeRecordReferenceOrAlias(t, n, parent, nameInfo, null);
915           } else {
916             for (NameInformation referring : referers) {
917               maybeRecordReferenceOrAlias(t, n, parent, nameInfo, referring);
918             }
919             recordAliases(referers);
920           }
921         }

923         private void maybeRecordReferenceOrAlias(
924             NodeTraversal t, Node n, Node parent,
925             NameInformation nameInfo, NameInformation referring) {
926           String referringName = "";
927           if (referring != null) {
928             referringName = referring.isPrototype
929                           ? referring.prototypeClass
930                           : referring.name;
931           }
932     
933           String name = nameInfo.name;
934     
935           // A value whose result is the return value of a function call
936           // can be an alias to global object.
937           // Here we add an alias to the general "global" object
938           // to act as a placeholder for the actual (unnamed) value.
939           if (maybeHiddenAlias(n)) {
940             recordAlias(name, WINDOW);
941           }
942     
943           // An externally referenceable name must always be defined, so we add a
944           // reference to it from the global scope (a.k.a. window).
945           if (nameInfo.isExternallyReferenceable) {
946             recordReference(WINDOW, name, RefType.REGULAR);
947             maybeRecordAlias(name, parent, referring, referringName);
948             return;
949           }
950     
951           // An assignment implies a reference from the enclosing dependency scope.
952           // For example, foo references bar in: function foo() {bar=5}.
953           if (NodeUtil.isVarOrSimpleAssignLhs(n, parent)) {
954             if (referring != null) {
955               recordReference(referringName, name, RefType.REGULAR);
956             }
957             return;
958           }
959     
960           if (nodesToKeep.contains(n)) {
961             List<NameInformation> functionScopes =
962                 getEnclosingFunctionDependencyScope(t);
963             if (!functionScopes.isEmpty()) {
964               for (NameInformation functionScope : functionScopes) {
965                 recordReference(functionScope.name, name, RefType.REGULAR);
966               }
967             } else {
968               recordReference(WINDOW, name, RefType.REGULAR);
969               if (referring != null) {
970                 maybeRecordAlias(name, parent, referring, referringName);
971               }
972             }
973           } else if (referring != null) {
974             if (!maybeRecordAlias(name, parent, referring, referringName)) {
975               RefType depType = referring.onlyAffectsClassDef ?
976                   RefType.INHERITANCE : RefType.REGULAR;
977               recordReference(referringName, name, depType);
978             }
979           } else {
980             // No named dependency scope found.  Unfortunately that might
981             // mean that the expression is a child of an function expression
982             // or assignment with a complex lhs.  In those cases,
983             // protect this node by creating a reference to WINDOW.
984             for (Node ancestor : n.getAncestors()) {
985               if (NodeUtil.isAssignmentOp(ancestor) ||
986                   ancestor.isFunction()) {
987                 recordReference(WINDOW, name, RefType.REGULAR);
988                 break;
989               }
990             }
991           }
992         }

994         private void recordAliases(List<NameInformation> referers) {
995           int size = referers.size();
996           for (int i = 0; i < size; i++) {
997             for (int j = i + 1; j < size; j++) {
998               recordAlias(referers.get(i).name, referers.get(j).name);
999               recordAlias(referers.get(j).name, referers.get(i).name);
1000             }
1001           }
1002         }

1004         /**
1005          * A value whose result is the return value of a function call
1006          * can be an alias to global object. The dependency on the call target will
1007          * prevent the removal of the function and its dependent values, but won't
1008          * prevent the alias' removal.
1009          */
1010         private boolean maybeHiddenAlias(Node n) {
1011           Node parent = n.getParent();
1012           if (NodeUtil.isVarOrSimpleAssignLhs(n, parent)) {
1013             Node rhs = (parent.isVar())
1014                 ? n.getFirstChild() : parent.getLastChild();
1015             return (rhs != null && !NodeUtil.evaluatesToLocalValue(
1016                 rhs, NON_LOCAL_RESULT_PREDICATE));
1017           }
1018           return false;
1019         }

1021         /**
1022          * @return Whether the alias was recorded.
1023          */
1024         private boolean maybeRecordAlias(
1025             String name, Node parent,
1026             NameInformation referring, String referringName) {
1027           // A common type of reference is
1028           // function F() {}
1029           // F.prototype.bar = goog.nullFunction;
1030           //
1031           // In this specific case, we do not want a reference to goog.nullFunction
1032           // to preserve F.
1033           //
1034           // In the general case, the user could do something like
1035           // function F() {}
1036           // F.prototype.bar = goog.nullFunction;
1037           // F.prototype.bar.baz = 3;
1038           // where it would not be safe to remove F.
1039           //
1040           // So we do not treat this alias as a backdoor for people to mutate the
1041           // original object. We think that this heuristic will always be
1042           // OK in real code.
1043           boolean isPrototypePropAssignment =
1044               parent.isAssign()
1045               && NodeUtil.isPrototypeProperty(parent.getFirstChild());
1046     
1047           if ((parent.isName() || parent.isAssign()) && !isPrototypePropAssignment && referring != null
1048               && scopes.containsEntry(parent, referring)) {
1049             recordAlias(referringName, name);
1050             return true;
1051           }
1052           return false;
1053         }

1168       /**
1169        * Records an alias of one name to another name.
1170        */
1171       private void recordAlias(String fromName, String toName) {
1172         recordReference(fromName, toName, RefType.REGULAR);
1173     
1174         // We need to add an edge to the alias graph. The alias graph is expressed
1175         // implicitly as a set of connected components, called AliasSets.
1176         //
1177         // There are three possibilities:
1178         // 1) Neither name is part of a connected component. Create a new one.
1179         // 2) Exactly one name is part of a connected component. Merge the new
1180         //    name into the component.
1181         // 3) The two names are already part of connected components. Merge
1182         //    those components together.
1183         AliasSet toNameAliasSet = aliases.get(toName);
1184         AliasSet fromNameAliasSet = aliases.get(fromName);
1185         AliasSet resultSet = null;
1186         if (toNameAliasSet == null && fromNameAliasSet == null) {
1187           resultSet = new AliasSet(toName, fromName);
1188         } else if (toNameAliasSet != null && fromNameAliasSet != null) {
1189           resultSet = toNameAliasSet;
1190           resultSet.names.addAll(fromNameAliasSet.names);
1191           for (String name : fromNameAliasSet.names) {
1192             aliases.put(name, resultSet);
1193           }
1194         } else if (toNameAliasSet != null) {
1195           resultSet = toNameAliasSet;
1196           resultSet.names.add(fromName);
1197         } else {
1198           resultSet = fromNameAliasSet;
1199           resultSet.names.add(toName);
1200         }
1201         aliases.put(fromName, resultSet);
1202         aliases.put(toName, resultSet);
1203       }

1374       /**

1376        * name in the global scope. When multiple names in the global scope
1377        * point to the same object, things start to break down.
1378        *
1379        * For example, if we have
1380        * <code>
1381        * var a = {};
1382        * var b = a;
1383        * a.foo = 3;
1384        * alert(b.foo);
1385        * </code>
1386        * then a.foo and b.foo are the same name, even though NameAnalyzer doesn't
1387        * represent them as such.
1388        *
1389        * To handle this case, we look at all the aliases in the program.
1390        * If descendant properties of that alias are assigned, then we create a
1391        * directional reference from the original name to the alias. For example,
1392        * in this case, the assign to {@code a.foo} triggers a reference from
1393        * {@code b} to {@code a}, but NOT from a to b.
1394        *
1395        * Similarly, "instanceof" checks do not prevent the removal
1396        * of a unaliased name but an instanceof check on an alias can only be removed
1397        * if the other aliases are also removed, so we add a connection here.
1398        */
1399       private void referenceAliases() {
1400     
1401         // Minimize the number of connections in the graph by creating a connected
1402         // cluster for names that are used to modify the object and then ensure
1403         // there is at least one link to the cluster from the other names (which are
1404         // removalable on there own) in the AliasSet.
1405     
1406         Set<AliasSet> sets = new HashSet<>(aliases.values());
1407         for (AliasSet set : sets) {
1408           DiGraphNode<JsName, RefType> first = null;
1409           Set<DiGraphNode<JsName, RefType>> required = new HashSet<>();
1410           for (String key : set.names) {
1411             JsName name = getName(key, false);
1412             if (name.hasWrittenDescendants || name.hasInstanceOfReference) {
1413               DiGraphNode<JsName, RefType> node = getGraphNode(name);
1414               required.add(node);
1415               if (first == null) {
1416                 first = node;
1417               }
1418             }
1419           }
1420     
1421           if (!required.isEmpty()) {
1422             // link the required nodes together to form a cluster so that if one
1423             // is needed, all are kept.
1424             for (DiGraphNode<JsName, RefType> node : required) {
1425               recordReference(node, first, RefType.REGULAR);
1426               recordReference(first, node, RefType.REGULAR);
1427             }
1428     
1429             // link all the other aliases to the one of the required nodes, so
1430             // that if they are kept only if referenced directly, but all the
1431             // required nodes are kept if any are referenced.
1432             for (String key : set.names) {
1433               DiGraphNode<JsName, RefType> alias = getGraphNode(getName(key, false));
1434               recordReference(alias, first, RefType.REGULAR);
1435             }
1436           }
1437         }
1438       }

1472       /**
1473        * Creates name information for the current node during a traversal.
1474        *
1475        * @param t The node traversal
1476        * @param n The current node
1477        * @return The name information, or null if the name is irrelevant to this
1478        *     pass
1479        */
1480       private NameInformation createNameInformation(NodeTraversal t, Node n) {
1481         Node parent = n.getParent();
1482         // Build the full name and find its root node by iterating down through all
1483         // GETPROP/GETELEM nodes.
1484         String name = "";
1485         Node rootNameNode = n;
1486         boolean bNameWasShortened = false;
1487         while (true) {
1488           if (NodeUtil.isGet(rootNameNode)) {
1489             Node prop = rootNameNode.getLastChild();
1490             if (rootNameNode.isGetProp()) {
1491               name = "." + prop.getString() + name;
1492             } else {
1493               // We consider the name to be "a.b" in a.b['c'] or a.b[x].d.
1494               bNameWasShortened = true;
1495               name = "";
1496             }
1497             rootNameNode = rootNameNode.getFirstChild();
1498           } else if (NodeUtil.isObjectLitKey(rootNameNode)) {
1499             name = "." + rootNameNode.getString() + name;
1500     
1501             // Check if this is an object literal assigned to something.
1502             Node objLit = rootNameNode.getParent();
1503             Node objLitParent = objLit.getParent();
1504             if (objLitParent.isAssign()) {
1505               // This must be the right side of the assign.
1506               rootNameNode = objLitParent.getFirstChild();
1507             } else if (objLitParent.isName()) {
1508               // This must be a VAR initialization.
1509               rootNameNode = objLitParent;
1510             } else if (objLitParent.isStringKey()) {
1511               // This must be a object literal key initialization.
1512               rootNameNode = objLitParent;
1513             } else {
1514               return null;
1515             }
1516           } else if (isAnalyzableObjectDefinePropertiesDefinition(rootNameNode)) {
1517             Node target = rootNameNode.getSecondChild();
1518             if (!target.isQualifiedName()) {
1519               return null;
1520             }
1521             rootNameNode = target;
1522           } else {
1523             break;
1524           }
1525         }
1526     
1527         // Check whether this is a class-defining call. Classes may only be defined
1528         // in the global scope.
1529         if (parent.isCall() && t.inGlobalHoistScope()) {
1530           CodingConvention convention = compiler.getCodingConvention();
1531           SubclassRelationship classes = convention.getClassesDefinedByCall(parent);
1532           if (classes != null) {
1533             NameInformation nameInfo = new NameInformation();
1534             nameInfo.name = classes.subclassName;
1535             nameInfo.onlyAffectsClassDef = true;
1536             nameInfo.superclass = classes.superclassName;
1537             return nameInfo;
1538           }
1539     
1540           String singletonGetterClass =
1541               convention.getSingletonGetterClassName(parent);
1542           if (singletonGetterClass != null) {
1543             NameInformation nameInfo = new NameInformation();
1544             nameInfo.name = singletonGetterClass;
1545             nameInfo.onlyAffectsClassDef = true;
1546             return nameInfo;
1547           }
1548         }
1549     
1550         switch (rootNameNode.getType()) {
1551           case NAME:
1552             // Check whether this is an assignment to a prototype property
1553             // of an object defined in the global scope.
1554             if (!bNameWasShortened &&
1555                 n.isGetProp() &&
1556                 parent.isAssign() &&
1557                 "prototype".equals(n.getLastChild().getString())) {
1558               if (createNameInformation(t, n.getFirstChild()) != null) {
1559                 name = rootNameNode.getString() + name;
1560                 name = name.substring(0, name.length() - PROTOTYPE_SUFFIX_LEN);
1561                 NameInformation nameInfo = new NameInformation();
1562                 nameInfo.name = name;
1563                 return nameInfo;
1564               } else {
1565                 return null;
1566               }
1567             }
1568             return createNameInformation(
1569                 rootNameNode.getString() + name, t.getScope(), rootNameNode);
1570           case THIS:
1571             if (t.inGlobalHoistScope()) {
1572               NameInformation nameInfo = new NameInformation();
1573               if (name.indexOf('.') == 0) {
1574                 nameInfo.name = name.substring(1);  // strip leading "."
1575               } else {
1576                 nameInfo.name = name;
1577               }
1578               nameInfo.isExternallyReferenceable = true;
1579               return nameInfo;
1580             }
1581             return null;
1582           default:
1583             return null;
1584         }
1585       }

1587       /**
1588        * Creates name information for a particular qualified name that occurs in a
1589        * particular scope.
1590        *
1591        * @param name A qualified name (e.g. "x" or "a.b.c")
1592        * @param scope The scope in which {@code name} occurs
1593        * @param rootNameNode The NAME node for the first token of {@code name}
1594        * @return The name information, or null if the name is irrelevant to this
1595        *     pass
1596        */
1597       private NameInformation createNameInformation(
1598           String name, Scope scope, Node rootNameNode) {
1599         // Check the scope. Currently we're only looking at globally scoped vars.
1600         String rootName = rootNameNode.getString();
1601         Var v = scope.getVar(rootName);
1602         boolean isExtern = (v == null && externalNames.contains(rootName));
1603         boolean isGlobalRef = (v != null && v.isGlobal()) || isExtern ||
1604             rootName.equals(WINDOW);
1605         if (!isGlobalRef) {
1606           return null;
1607         }
1608     
1609         NameInformation nameInfo = new NameInformation();
1610     
1611         // If a prototype property or method, fill in prototype information.
1612         int idx = name.indexOf(PROTOTYPE_SUBSTRING);
1613         if (idx != -1) {
1614           nameInfo.isPrototype = true;
1615           nameInfo.prototypeClass = name.substring(0, idx);
1616           nameInfo.prototypeProperty = name.substring(
1617               idx + PROTOTYPE_SUBSTRING_LEN);
1618         }
1619     
1620         nameInfo.name = name;
1621         nameInfo.isExternallyReferenceable =
1622             isExtern || isExternallyReferenceable(scope, name);
1623         return nameInfo;
1624       }

1627        * Checks whether a name can be referenced outside of the compiled code.
1628        * These names will be the root of dependency trees.
1629        *
1630        * @param scope The current variable scope
1631        * @param name The name
1632        * @return True if can be referenced outside
1633        */
1634       private boolean isExternallyReferenceable(Scope scope, String name) {
1635         if (compiler.getCodingConvention().isExported(name)) {
1636           return true;
1637         }
1638         if (scope.isLocal()) {
1639           return false;
1640         }
1641         for (String s : globalNames) {
1642           if (name.startsWith(s)) {
1643             return true;
1644           }
1645         }
1646         return false;

1663       /**
1664        * Get dependency scope defined by the enclosing function, or null.
1665        * If enclosing function is a function expression, determine scope based on
1666        * its parent if the parent node is a variable declaration or
1667        * assignment.
1668        */
1669       private List<NameInformation> getEnclosingFunctionDependencyScope(
1670           NodeTraversal t) {
1671         Node function = t.getEnclosingFunction();
1672         if (function == null) {
1673           return Collections.emptyList();
1674         }
1675     
1676         List<NameInformation> refs = scopes.get(function);
1677         if (!refs.isEmpty()) {
1678           return refs;
1679         }
1680     
1681         // Function expression.  try to get a name from the parent var
1682         // declaration or assignment.
1683         Node parent = function.getParent();
1684         if (parent != null) {
1685           // Account for functions defined in the form:
1686           //   var a = cond ? function a() {} : function b() {};
1687           while (parent.isHook()) {
1688             parent = parent.getParent();
1689           }
1690     
1691           if (parent.isName()) {
1692             return scopes.get(parent);
1693           }
1694     
1695           if (parent.isAssign()) {
1696             return scopes.get(parent);
1697           }
1698         }
1699     
1700         return Collections.emptyList();
1701       }

1715       private void propagateReference(JsName ... names) {
1716         Deque<DiGraphNode<JsName, RefType>> work = new ArrayDeque<>();
1717         for (JsName name : names) {
1718           work.push(referenceGraph.createDirectedGraphNode(name));
1719         }
1720         while (!work.isEmpty()) {
1721           DiGraphNode<JsName, RefType> source = work.pop();
1722           List<DiGraphEdge<JsName, RefType>> outEdges = source.getOutEdges();
1723           int len = outEdges.size();
1724           for (int i = 0; i < len; i++) {
1725             DiGraphNode<JsName, RefType> item = outEdges.get(i).getDestination();
1726             JsName destNode = item.getValue();
1727             if (!destNode.referenced) {
1728               destNode.referenced = true;
1729               work.push(item);
1730             }
1731           }
1732         }
1733       }

1795       /**
1796        * Replace n with a simpler expression, while preserving program
1797        * behavior.
1798        *
1799        * If the n's value is used, replace it with its RHS; otherwise
1800        * replace it with the subexpressions that have side effects.
1801        */
1802       private void replaceWithRhs(Node parent, Node n) {
1803         if (valueConsumedByParent(n, parent)) {
1804           // parent reads from n directly; replace it with n's rhs + lhs
1805           // subexpressions with side effects.
1806           List<Node> replacements = getRhsSubexpressions(n);
1807           List<Node> newReplacements = new ArrayList<>();
1808           for (int i = 0; i < replacements.size() - 1; i++) {
1809             newReplacements.addAll(getSideEffectNodes(replacements.get(i)));
1810           }
1811           Node valueExpr = Iterables.getLast(replacements);
1812           valueExpr.detachFromParent();
1813           newReplacements.add(valueExpr);
1814           changeProxy.replaceWith(
1815               parent, n, collapseReplacements(newReplacements));
1816         } else if (n.isAssign() && !parent.isFor()) {
1817           // assignment appears in a RHS expression.  we have already
1818           // considered names in the assignment's RHS as being referenced;
1819           // replace the assignment with its RHS.
1820           // TODO(user) make the pass smarter about these cases and/or run
1821           // this pass and RemoveConstantExpressions together in a loop.
1822           Node replacement = n.getLastChild();
1823           replacement.detachFromParent();
1824           changeProxy.replaceWith(parent, n, replacement);
1825         } else {
1826           replaceTopLevelExpressionWithRhs(parent, n);
1827         }
1828       }

1830       /**
1831        * Simplify a toplevel expression, while preserving program
1832        * behavior.
1833        */
1834       private void replaceTopLevelExpressionWithRhs(Node parent, Node n) {
1835         // validate inputs
1836         switch (parent.getType()) {
1837           case BLOCK:
1838           case SCRIPT:
1839           case FOR:
1840           case LABEL:
1841             break;
1842           default:
1843             throw new IllegalArgumentException(
1844                 "Unsupported parent node type in replaceWithRhs " + parent.getType());
1845         }
1846     
1847         switch (n.getType()) {
1848           case EXPR_RESULT:
1849           case FUNCTION:
1850           case VAR:
1851             break;
1852           case ASSIGN:
1853             Preconditions.checkArgument(
1854                 parent.isFor(),
1855                 "Unsupported assignment in replaceWithRhs. parent: %s",
1856                 parent.getType());
1857             break;
1858           default:
1859             throw new IllegalArgumentException(
1860                 "Unsupported node type in replaceWithRhs " + n.getType());
1861         }
1862     
1863         // gather replacements
1864         List<Node> replacements = new ArrayList<>();
1865         for (Node rhs : getRhsSubexpressions(n)) {
1866           replacements.addAll(getSideEffectNodes(rhs));
1867         }
1868     
1869         if (parent.isFor()) {
1870           // tweak replacements array s.t. it is a single expression node.
1871           if (replacements.isEmpty()) {
1872             replacements.add(IR.empty());
1873           } else {
1874             Node expr = collapseReplacements(replacements);
1875             replacements.clear();
1876             replacements.add(expr);
1877           }
1878         }
1879     
1880         changeProxy.replaceWith(parent, n, replacements);
1881       }

1883       /**
1884        * Determine if the parent reads the value of a child expression
1885        * directly.  This is true children used in predicates, RETURN
1886        * statements and, RHS of variable declarations and assignments.
1887        *
1888        * In the case of:
1889        * if (a) b else c
1890        *
1891        * This method returns true for "a", and false for "b" and "c": the
1892        * IF expression does something special based on "a"'s value.  "b"
1893        * and "c" are effectively outputs.  Same logic applies to FOR,
1894        * WHILE and DO loop predicates.  AND/OR/HOOK expressions are
1895        * syntactic sugar for IF statements; therefore this method returns
1896        * true for the predicate and false otherwise.
1897        */
1898       private static boolean valueConsumedByParent(Node n, Node parent) {
1899         if (NodeUtil.isAssignmentOp(parent)) {
1900           return parent.getLastChild() == n;
1901         }
1902     
1903         switch (parent.getType()) {
1904           case NAME:
1905           case RETURN:
1906             return true;
1907     
1908           case AND:
1909           case OR:
1910           case HOOK:
1911           case IF:
1912           case WHILE:
1913             return parent.getFirstChild() == n;
1914     
1915           case FOR:
1916             return parent.getSecondChild() == n;
1917     
1918           case DO:
1919             return parent.getLastChild() == n;
1920     
1921           default:
1922             return false;
1923         }
1924       }

1926       /**
1927        * Merge a list of nodes into a single expression.  The value of the
1928        * new expression is determined by the last expression in the list.
1929        */
1930       private static Node collapseReplacements(List<Node> replacements) {
1931         Node expr = null;
1932         for (Node rep : replacements) {
1933           if (rep.isExprResult()) {
1934             rep = rep.getFirstChild();
1935             rep.detachFromParent();
1936           }
1937     
1938           if (expr == null) {
1939             expr = rep;
1940           } else {
1941             expr = IR.comma(expr, rep);
1942           }
1943         }
1944     
1945         return expr;
1946       }

1948       /**
1949        * Extract a list of subexpressions that act as right hand sides.
1950        */
1951       private static List<Node> getRhsSubexpressions(Node n) {
1952         switch (n.getType()) {
1953           case EXPR_RESULT:
1954             // process body
1955             return getRhsSubexpressions(n.getFirstChild());
1956           case FUNCTION:
1957             // function nodes have no RHS
1958             return ImmutableList.of();
1959           case CALL:
1960             {
1961               // In our analyzable case, only the last argument to Object.defineProperties
1962               // (the object literal) can have side-effects
1963               Preconditions.checkState(isAnalyzableObjectDefinePropertiesDefinition(n));
1964               return ImmutableList.of(n.getLastChild());
1965             }
1966           case NAME:
1967             {
1968               // parent is a var node.  RHS is the first child
1969               Node rhs = n.getFirstChild();
1970               if (rhs != null) {
1971                 return ImmutableList.of(rhs);
1972               } else {
1973                 return ImmutableList.of();
1974               }
1975             }
1976           case ASSIGN:
1977             {
1978               // add LHS and RHS expressions - LHS may be a complex expression
1979               Node lhs = n.getFirstChild();
1980               Node rhs = lhs.getNext();
1981               return ImmutableList.of(lhs, rhs);
1982             }
1983           case VAR:
1984             {
1985               // recurse on all children
1986               ImmutableList.Builder<Node> nodes = ImmutableList.builder();
1987               for (Node child : n.children()) {
1988                 nodes.addAll(getRhsSubexpressions(child));
1989               }
1990               return nodes.build();
1991             }
1992           default:
1993             throw new IllegalArgumentException("AstChangeProxy::getRhs " + n);
1994         }
1995       }
