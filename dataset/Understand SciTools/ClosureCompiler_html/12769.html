16     package com.google.javascript.jscomp.lint;
17     
18     import com.google.javascript.jscomp.CheckLevel;
19     import com.google.javascript.jscomp.Compiler;
20     import com.google.javascript.jscomp.CompilerOptions;
21     import com.google.javascript.jscomp.CompilerPass;
22     import com.google.javascript.jscomp.DiagnosticGroups;
23     import com.google.javascript.jscomp.Es6CompilerTestCase;
24     
25     public class CheckMissingSemicolonTest extends Es6CompilerTestCase {
26       @Override
27       public CompilerPass getProcessor(Compiler compiler) {
28         return new CheckMissingSemicolon(compiler);
29       }
30     
31       @Override
32       protected CompilerOptions getOptions(CompilerOptions options) {
33         super.getOptions(options);
34         options.setWarningLevel(DiagnosticGroups.LINT_CHECKS, CheckLevel.WARNING);
35         options.setPreserveDetailedSourceInfo(true);
36         return options;
37       }
38     
39       private void testWarning(String js) {
40         testWarning(js, CheckMissingSemicolon.MISSING_SEMICOLON);
41       }
42     
43       private void testWarningEs6(String js) {
44         testWarningEs6(js, CheckMissingSemicolon.MISSING_SEMICOLON);
45       }
46     
47       public void testWarning() {
48         testWarning("var x");
49         testWarning("alert(1)");
50         testWarning("do { things; } while (true)");
51       }
52     
53       public void testNoWarning_withSemicolon() {
54         testSame("var x;");
55         testSame("alert(1);");
56       }
57     
58       public void testNoWarning_controlStructure() {
59         testSame("if (true) {}");
60         testSame("while (true) {}");
61         testSame("do { things; } while (true);");
62         testSame("switch (n) { case 3: alert(4); }");
63         testSame("switch (n) { case 5: { alert(6); } }");
64         testSame("alert(1); LABEL: while (true) { alert(2); }");
65         testSame("for (;;) {}");
66         testSameEs6("for (x of y) {}");
67         testSame("for (x in y) {}");
68       }
69     
70       public void testNoWarning_functionOrClass() {
71         testSame("function f() {}");
72         testSameEs6("function* f() {}");
73         testSameEs6("class Example {}");
74       }
75     
76       public void testWarning_export() {
77         testWarningEs6("export var x = 3");
78         testWarningEs6("export * from 'other'");
79       }
80     
81       public void testNoWarning_export() {
82         testSameEs6("export function f() {}");
83         testSameEs6("export class C {}");
84       }
85     }
