26     import java.util.ArrayList;

30     /**
31      * Checks for non side effecting statements such as
32      * <pre>
33      * var s = "this string is "
34      *         "continued on the next line but you forgot the +";
35      * x == foo();  // should that be '='?
36      * foo();;  // probably just a stray-semicolon. Doesn't hurt to check though
37      * </pre>
38      * and generates warnings.
39      *
40      */
41     final class CheckSideEffects extends AbstractPostOrderCallback
42         implements HotSwapCompilerPass {
43     
44       static final DiagnosticType USELESS_CODE_ERROR = DiagnosticType.warning(
45           "JSC_USELESS_CODE",
46           "Suspicious code. {0}");
47     
48       static final String PROTECTOR_FN = "JSCOMPILER_PRESERVE";
49     
50       private final boolean report;
51     
52       private final List<Node> problemNodes = new ArrayList<>();
53     
54       private final LinkedHashMap<String, String> noSideEffectExterns =
55          new LinkedHashMap<>();
56     
57       private final AbstractCompiler compiler;
58     
59       private final boolean protectSideEffectFreeCode;
60     
61       CheckSideEffects(AbstractCompiler compiler, boolean report,
62           boolean protectSideEffectFreeCode) {
63         this.compiler = compiler;
64         this.report = report;
65         this.protectSideEffectFreeCode = protectSideEffectFreeCode;
66       }
67     
68       @Override
69       public void process(Node externs, Node root) {
70         NodeTraversal.traverseEs6(compiler, externs, new GetNoSideEffectExterns());
71     
72         NodeTraversal.traverseEs6(compiler, root, this);
73     
74         // Code with hidden side-effect code is common, for example
75         // accessing "el.offsetWidth" forces a reflow in browsers, to allow this
76         // will still allowing local dead code removal in general,
77         // protect the "side-effect free" code in the source.
78         //
79         // This also includes function calls such as with document.createElement
80         if (protectSideEffectFreeCode) {
81           protectSideEffects();
82         }
83       }
84     
85       @Override
86       public void hotSwapScript(Node scriptRoot, Node originalRoot) {
87         NodeTraversal.traverseEs6(compiler, scriptRoot, this);
88       }
89     
90       @Override
91       public void visit(NodeTraversal t, Node n, Node parent) {
92         // VOID nodes appear when there are extra semicolons at the BLOCK level.
93         // I've been unable to think of any cases where this indicates a bug,
94         // and apparently some people like keeping these semicolons around,
95         // so we'll allow it.
96         if (n.isEmpty() ||
97             n.isComma()) {
98           return;
99         }
100     
101         if (parent == null) {
102           return;
103         }
104     
105         // Do not try to remove a block or an expr result. We already handle
106         // these cases when we visit the child, and the peephole passes will
107         // fix up the tree in more clever ways when these are removed.
108         if (n.isExprResult() || n.isBlock()) {
109           return;
110         }
111     
112         // This no-op statement was there so that JSDoc information could
113         // be attached to the name. This check should not complain about it.
114         if (n.isQualifiedName() && n.getJSDocInfo() != null) {
115           return;
116         }
117     
118         boolean isResultUsed = NodeUtil.isExpressionResultUsed(n);
119         boolean isSimpleOp = NodeUtil.isSimpleOperator(n);
120         if (!isResultUsed) {
121           if (isSimpleOp || !NodeUtil.mayHaveSideEffects(n, t.getCompiler())) {
122             if (report) {
123               String msg = "This code lacks side-effects. Is there a bug?";
124               if (n.isString() || n.isTemplateLit()) {
125                 msg = "Is there a missing '+' on the previous line?";
126               } else if (isSimpleOp) {
127                 msg = "The result of the '" + Token.name(n.getType()).toLowerCase() +
128                     "' operator is not being used.";
129               }
130     
131               t.report(n, USELESS_CODE_ERROR, msg);
132             }
133             // TODO(johnlenz): determine if it is necessary to
134             // try to protect side-effect free statements as well.
135             if (!NodeUtil.isStatement(n)) {
136               problemNodes.add(n);
137             }
138           } else if (n.isCall() && (n.getFirstChild().isGetProp() ||
139               n.getFirstChild().isName() || n.getFirstChild().isString())) {
140             String qname = n.getFirstChild().getQualifiedName();
141     
142             // The name should not be defined in src scopes - only externs
143             boolean isDefinedInSrc = false;
144             if (qname != null) {
145               if (n.getFirstChild().isGetProp()) {
146                 Node rootNameNode =
147                     NodeUtil.getRootOfQualifiedName(n.getFirstChild());
148                 isDefinedInSrc = rootNameNode != null && rootNameNode.isName() &&
149                     t.getScope().getVar(rootNameNode.getString()) != null;
150               } else {
151                 isDefinedInSrc = t.getScope().getVar(qname) != null;
152               }
153             }
154     
155             if (qname != null && noSideEffectExterns.containsKey(qname) &&
156                 !isDefinedInSrc) {
157               problemNodes.add(n);
158               if (report) {
159                 String msg = "The result of the extern function call '" + qname +
160                     "' is not being used.";
161                 t.report(n, USELESS_CODE_ERROR, msg);
162               }
163             }
164           }
165         }
166       }
167     
168       /**
169        * Protect side-effect free nodes by making them parameters
170        * to a extern function call.  This call will be removed
171        * after all the optimizations passes have run.
172        */
173       private void protectSideEffects() {
174         if (!problemNodes.isEmpty()) {
175           addExtern();
176           for (Node n : problemNodes) {
177             Node name = IR.name(PROTECTOR_FN).srcref(n);
178             name.putBooleanProp(Node.IS_CONSTANT_NAME, true);
179             Node replacement = IR.call(name).srcref(n);
180             replacement.putBooleanProp(Node.FREE_CALL, true);
181             n.getParent().replaceChild(n, replacement);
182             replacement.addChildToBack(n);
183           }
184           compiler.reportCodeChange();
185         }
186       }
187     
188       private void addExtern() {
189         Node name = IR.name(PROTECTOR_FN);
190         name.putBooleanProp(Node.IS_CONSTANT_NAME, true);
191         Node var = IR.var(name);
192         // Add "@noalias" so we can strip the method when AliasExternals is enabled.
193         JSDocInfoBuilder builder = new JSDocInfoBuilder(false);
194         builder.recordNoAlias();
195         var.setJSDocInfo(builder.build());
196         CompilerInput input = compiler.getSynthesizedExternsInput();
197         name.setStaticSourceFile(input.getSourceFile());
198         var.setStaticSourceFile(input.getSourceFile());
199         input.getAstRoot(compiler).addChildrenToBack(var);
200         compiler.reportCodeChange();
201       }

203       /**
204        * Remove side-effect sync functions.
205        */
206       static class StripProtection extends AbstractPostOrderCallback implements CompilerPass {
207     
208         private final AbstractCompiler compiler;
209     
210         StripProtection(AbstractCompiler compiler) {
211           this.compiler = compiler;
212         }
213     
214         @Override
215         public void process(Node externs, Node root) {
216           NodeTraversal.traverseEs6(compiler, root, this);
217         }
218     
219         @Override
220         public void visit(NodeTraversal t, Node n, Node parent) {
221           if (n.isCall()) {
222             Node target = n.getFirstChild();
223             // TODO(johnlenz): add this to the coding convention
224             // so we can remove goog.reflect.sinkValue as well.
225             if (target.isName() && target.getString().equals(PROTECTOR_FN)) {
226               Node expr = n.getLastChild();
227               n.detachChildren();
228               parent.replaceChild(n, expr);
229             }
230           }
231         }
232       }
233     
234       /**
235        * Get fully qualified function names which are marked
236        * with @nosideeffects
237        *
238        * TODO(ChadKillingsworth) Add support for object literals
239        */
240       private class GetNoSideEffectExterns extends AbstractPostOrderCallback {
241         @Override
242         public void visit(NodeTraversal t, Node n, Node parent) {
243           if (n.isFunction()) {
244             String name = NodeUtil.getName(n);
245             JSDocInfo jsDoc = NodeUtil.getBestJSDocInfo(n);
246             if (jsDoc != null && jsDoc.isNoSideEffects()) {
247               noSideEffectExterns.put(name, null);
248             }
249           }
250         }
251       }
