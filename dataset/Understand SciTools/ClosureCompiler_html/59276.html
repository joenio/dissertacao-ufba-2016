16     package com.google.javascript.jscomp.newtypes;
17     
18     import com.google.common.base.Joiner;
19     import com.google.common.base.Preconditions;
20     import com.google.common.base.Splitter;
21     import com.google.common.collect.ImmutableList;
22     
23     import com.google.javascript.rhino.Node;
24     
25     /**
26      * Represents a qualified name.
27      * (e.g. namespace.inner.Foo)
28      *
29      * @author blickly@google.com (Ben Lickly)
30      * @author dimvar@google.com (Dimitris Vardoulakis)
31      */
32     public final class QualifiedName {
33       private ImmutableList<String> parts;
34     
35       private QualifiedName(ImmutableList<String> parts) {
36         this.parts = parts;
37       }
38     
39       public QualifiedName(String s) {
40         this.parts = ImmutableList.of(s);
41       }
42     
43       public static QualifiedName join(QualifiedName lhs, QualifiedName rhs) {
44         return new QualifiedName(ImmutableList.<String>builder()
45             .addAll(lhs.parts).addAll(rhs.parts).build());
46       }
47     
48       public static QualifiedName fromNode(Node qnameNode) {
49         if (qnameNode == null || !qnameNode.isQualifiedName()) {
50           return null;
51         }
52         return qnameNode.isName()
53             ? new QualifiedName(qnameNode.getString())
54             : new QualifiedName(ImmutableList.copyOf(
55                   Splitter.on('.').split(qnameNode.getQualifiedName())));
56       }
57     
58       public static QualifiedName fromQualifiedString(String qname) {
59         return qname.contains(".") ?
60             new QualifiedName(ImmutableList.copyOf(Splitter.on('.').split(qname))) :
61             new QualifiedName(qname);
62       }
63     
64       public boolean isIdentifier() {
65         return parts.size() == 1;
66       }
67     
68       public QualifiedName getAllButLeftmost() {
69         Preconditions.checkArgument(!isIdentifier());
70         return new QualifiedName(parts.subList(1, parts.size()));
71       }
72     
73       public String getLeftmostName() {
74         return parts.get(0);
75       }
76     
77       public QualifiedName getAllButRightmost() {
78         Preconditions.checkArgument(!isIdentifier());
79         return new QualifiedName(parts.subList(0, parts.size() - 1));
80       }
81     
82       public String getRightmostName() {
83         return parts.get(parts.size() - 1);
84       }
85     
86       @Override
87       public String toString() {
88         return Joiner.on(".").join(parts);
89       }
90     }
