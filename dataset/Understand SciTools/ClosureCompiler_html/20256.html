94       class Assignment {
95         final String oldName;
96         final int orderOfOccurrence;
97         String newName;
98         int count; // Number of times this is referenced
99     
100         Assignment(String name) {
101           this.oldName = name;
102           this.newName = null;
103           this.count = 0;
104     
105           // Represents the order at which a symbol appears in the source.
106           this.orderOfOccurrence = assignmentCount++;
107         }
108     
109         /**
110          * Assigns the new name.
111          */
112         void setNewName(String newName) {
113           Preconditions.checkState(this.newName == null);
114           this.newName = newName;
115         }
116       }

149        * the instance may reset or reconfigure it, so the caller should

152       RenameVars(AbstractCompiler compiler, String prefix,
153           boolean localRenamingOnly, boolean preserveFunctionExpressionNames,
154           boolean generatePseudoNames, boolean shouldShadow,
155           boolean preferStableNames, VariableMap prevUsedRenameMap,
156           @Nullable char[] reservedCharacters,
157           @Nullable Set<String> reservedNames,
158           NameGenerator nameGenerator) {
159         this.compiler = compiler;
160         this.prefix = nullToEmpty(prefix);
161         this.localRenamingOnly = localRenamingOnly;
162         this.preserveFunctionExpressionNames = preserveFunctionExpressionNames;
163         if (generatePseudoNames) {
164           this.pseudoNameMap = new HashMap<>();
165         } else {
166           this.pseudoNameMap = null;
167         }
168         this.prevUsedRenameMap = prevUsedRenameMap;
169         this.reservedCharacters = reservedCharacters;
170         this.shouldShadow = shouldShadow;
171         this.preferStableNames = preferStableNames;
172         if (reservedNames == null) {
173           this.reservedNames = new HashSet<>();
174         } else {
175           this.reservedNames = new HashSet<>(reservedNames);
176         }
177         this.nameGenerator = nameGenerator;
178       }

180       /**
181        * Iterate through the nodes, collect all the NAME nodes that need to be
182        * renamed, and count how many times each variable name is referenced.
183        *
184        * There are 2 passes:
185        * - externs: keep track of the global vars in the externNames_ map.
186        * - source: keep track of all name references in globalNameNodes_, and
187        *   localNameNodes_.
188        *
189        * To get shorter local variable renaming, we rename local variables to a
190        * temporary name "LOCAL_VAR_PREFIX + index" where index is the index of the
191        * variable declared in the local scope stack.
192        * e.g.
193        * Foo(fa, fb) {
194        *   var c = function(d, e) { return fa; }
195        * }
196        * The indexes are: fa:0, fb:1, c:2, d:3, e:4
197        *
198        * In that way, local variable names are reused in each global function.
199        * e.g. the final code might look like
200        * function x(a,b) { ... }
201        * function y(a,b,c) { ... }
202        */
203       class ProcessVars extends AbstractPostOrderCallback
204           implements ScopedCallback {
205         private final boolean isExternsPass_;
206     
207         ProcessVars(boolean isExterns) {
208           isExternsPass_ = isExterns;
209         }
210     
211         @Override
212         public void enterScope(NodeTraversal t) {
213           if (t.inGlobalHoistScope() ||
214               !shouldTemporarilyRenameLocalsInScope(t.getScope())) {
215             return;
216           }
217           Scope scope = t.getScope();
218           for (Var current : scope.getVarIterable()) {
219             if (current.isBleedingFunction()) {
220               localBleedingFunctions.add(current);
221               localBleedingFunctionsPerScope.put(
222                   scope.getParent(), current);
223             }
224           }
225         }
226     
227         @Override
228         public void exitScope(NodeTraversal t) {}
229     
230         @Override
231         public void visit(NodeTraversal t, Node n, Node parent) {
232           if (!n.isName()) {
233             return;
234           }
235     
236           String name = n.getString();
237     
238           // Ignore anonymous functions
239           if (name.isEmpty()) {
240             return;
241           }
242     
243           // Is this local or Global?
244           // Bleeding functions should be treated as part of their outer
245           // scope, because IE has bugs in how it handles bleeding
246           // functions.
247           Var var = t.getScope().getVar(name);
248           boolean local = (var != null) && var.isLocal() &&
249               (!var.scope.getParent().isGlobal() ||
250                !var.isBleedingFunction());
251     
252           // Never rename references to the arguments array
253           if (var != null && var.isArguments()) {
254             reservedNames.add(name);
255             return;
256           }
257     
258           // Are we renaming global variables?
259           if (!local && localRenamingOnly) {
260             reservedNames.add(name);
261             return;
262           }
263     
264           // Are we renaming function expression names?
265           if (preserveFunctionExpressionNames && var != null
266               && NodeUtil.isFunctionExpression(var.getParentNode())) {
267             reservedNames.add(name);
268             return;
269           }
270     
271           // Check if we can rename this.
272           if (!okToRenameVar(name, local)) {
273             if (local) {
274               // Blindly de-uniquify for the Prototype library for issue 103.
275               String newName = MakeDeclaredNamesUnique.ContextualRenameInverter
276                   .getOriginalName(name);
277               if (!newName.equals(name)) {
278                 n.setString(newName);
279               }
280             }
281             return;
282           }
283     
284           if (isExternsPass_) {
285             // Keep track of extern globals.
286             if (!local) {
287               externNames.add(name);
288             }
289             return;
290           }
291     
292           if (pseudoNameMap != null) {
293             recordPseudoName(n);
294           }
295     
296           if (local && shouldTemporarilyRenameLocalsInScope(var.getScope())) {
297             // Give local variables a temporary name based on the
298             // variable's index in the scope to enable name reuse across
299             // locals in independent scopes.
300             String tempName = LOCAL_VAR_PREFIX + getLocalVarIndex(var);
301             incCount(tempName);
302             localNameNodes.add(n);
303             n.setString(tempName);
304           } else if (var != null) { // Not an extern
305             // If it's global, increment global count
306             incCount(name);
307             globalNameNodes.add(n);
308           }
309         }
310     
311         // Increment count of an assignment
312         void incCount(String name) {
313           Assignment s = assignments.get(name);
314           if (s == null) {
315             s = new Assignment(name);
316             assignments.put(name, s);
317           }
318           s.count++;
319         }
320       }

322       /**
323        * Sorts Assignment objects by their count, breaking ties by their order of
324        * occurrence in the source to ensure a deterministic total ordering.
325        */
326       private static final Comparator<Assignment> FREQUENCY_COMPARATOR =
327           new Comparator<Assignment>() {
328         @Override
329         public int compare(Assignment a1, Assignment a2) {
330           if (a1.count != a2.count) {
331             return a2.count - a1.count;
332           }
333           // Break a tie using the order in which the variable first appears in
334           // the source.
335           return ORDER_OF_OCCURRENCE_COMPARATOR.compare(a1, a2);
336         }
337       };
338     
339       /**
340        * Sorts Assignment objects by the order the variable name first appears in
341        * the source.
342        */
343       private static final Comparator<Assignment> ORDER_OF_OCCURRENCE_COMPARATOR =
344           new Comparator<Assignment>() {
345             @Override
346             public int compare(Assignment a1, Assignment a2) {
347               return a1.orderOfOccurrence - a2.orderOfOccurrence;
348             }
349           };

351       @Override
352       public void process(Node externs, Node root) {
353         assignmentLog = new StringBuilder();
354     
355         // Do variable reference counting.
356         NodeTraversal.traverseEs6(compiler, externs, new ProcessVars(true));
357         NodeTraversal.traverseEs6(compiler, root, new ProcessVars(false));
358     
359         // Make sure that new names don't overlap with extern names.
360         reservedNames.addAll(externNames);
361     
362         // Rename vars, sorted by frequency of occurrence to minimize code size.
363         SortedSet<Assignment> varsByFrequency =
364             new TreeSet<>(FREQUENCY_COMPARATOR);
365         varsByFrequency.addAll(assignments.values());
366     
367         if (shouldShadow) {
368           new ShadowVariables(
369               compiler, assignments, varsByFrequency, pseudoNameMap).process(
370                   externs, root);
371         }
372     
373         // First try to reuse names from an earlier compilation.
374         if (prevUsedRenameMap != null) {
375           reusePreviouslyUsedVariableMap();
376         }
377     
378         // Assign names, sorted by descending frequency to minimize code size.
379         assignNames(varsByFrequency);
380     
381         boolean changed = false;
382     
383         // Rename the globals!
384         for (Node n : globalNameNodes) {
385           String newName = getNewGlobalName(n);
386           // Note: if newName is null, then oldName is an extern.
387           if (newName != null) {
388             n.setString(newName);
389             changed = true;
390           }
391         }
392     
393         // Rename the locals!
394         for (Node n : localNameNodes) {
395           String newName = getNewLocalName(n);
396           if (newName != null) {
397             n.setString(newName);
398             changed = true;
399           }
400         }
401     
402         if (changed) {
403           compiler.reportCodeChange();
404         }
405     
406         // Lastly, write the name assignments to the debug log.
407         compiler.addToDebugLog("JS var assignments:\n" + assignmentLog);
408         assignmentLog = null;
409       }

411       private String getNewGlobalName(Node n) {
412         String oldName = n.getString();
413         Assignment a = assignments.get(oldName);
414         if (a.newName != null && !a.newName.equals(oldName)) {
415           if (pseudoNameMap != null) {
416             return pseudoNameMap.get(n);
417           }
418           return a.newName;
419         } else {
420           return null;
421         }
422       }
423     
424       private String getNewLocalName(Node n) {
425         String oldTempName = n.getString();
426         Assignment a = assignments.get(oldTempName);
427         if (!a.newName.equals(oldTempName)) {
428           if (pseudoNameMap != null) {
429             return pseudoNameMap.get(n);
430           }
431           return a.newName;
432         }
433         return null;
434       }

436       private void recordPseudoName(Node n) {
437         // Variable names should be in a different name space than
438         // property pseudo names.
439         pseudoNameMap.put(n, '$' + n.getString() + "$$");
440       }

442       /**
443        * Runs through the assignments and reuses as many names as possible from the
444        * previously used variable map. Updates reservedNames with the set of names
445        * that were reused.
446        */
447       private void reusePreviouslyUsedVariableMap() {
448         // If prevUsedRenameMap had duplicate values then this pass would be
449         // non-deterministic.
450         // In such a case, the following will throw an IllegalArgumentException.
451         Preconditions.checkNotNull(prevUsedRenameMap.getNewNameToOriginalNameMap());
452         for (Assignment a : assignments.values()) {
453           String prevNewName = prevUsedRenameMap.lookupNewName(a.oldName);
454           if (prevNewName == null || reservedNames.contains(prevNewName)) {
455             continue;
456           }
457     
458           if (a.oldName.startsWith(LOCAL_VAR_PREFIX)
459               || (!externNames.contains(a.oldName)
460                   && prevNewName.startsWith(prefix))) {
461             reservedNames.add(prevNewName);
462             finalizeNameAssignment(a, prevNewName);
463           }
464         }
465       }

467       /**
468        * Determines which new names to substitute for the original names.
469        */
470       private void assignNames(SortedSet<Assignment> varsToRename) {
471         NameGenerator globalNameGenerator = null;
472         NameGenerator localNameGenerator = null;
473     
474         globalNameGenerator = nameGenerator;
475         nameGenerator.reset(reservedNames, prefix, reservedCharacters);
476     
477         // Local variables never need a prefix.
478         // Also, we need to avoid conflicts between global and local variable
479         // names; we do this by having using the same generator (not two
480         // instances). The case where global variables have a prefix (and
481         // therefore we use two different generators) but a local variable name
482         // might nevertheless conflict with a global one is not handled.
483         localNameGenerator =
484             prefix.isEmpty()
485             ? globalNameGenerator
486             : nameGenerator.clone(reservedNames, "", reservedCharacters);
487     
488         // Generated names and the assignments for non-local vars.
489         List<Assignment> pendingAssignments = new ArrayList<>();
490         List<String> generatedNamesForAssignments = new ArrayList<>();
491     
492         for (Assignment a : varsToRename) {
493           if (a.newName != null) {
494             continue;
495           }
496     
497           if (externNames.contains(a.oldName)) {
498             continue;
499           }
500     
501           String newName;
502           if (a.oldName.startsWith(LOCAL_VAR_PREFIX)) {
503             // For local variable, we make the assignment right away.
504             newName = localNameGenerator.generateNextName();
505             finalizeNameAssignment(a, newName);
506           } else {
507             // For non-local variable, delay finalizing the name assignment
508             // until we know how many new names we'll have of length 2, 3, etc.
509             newName = globalNameGenerator.generateNextName();
510             pendingAssignments.add(a);
511             generatedNamesForAssignments.add(newName);
512           }
513           reservedNames.add(newName);
514         }
515     
516         // Now that we have a list of generated names, and a list of variable
517         // Assignment objects, we assign the generated names to the vars as
518         // follows:
519         // 1) The most frequent vars get the shorter names.
520         // 2) If N number of vars are going to be assigned names of the same
521         //    length, we assign the N names based on the order at which the vars
522         //    first appear in the source. This makes the output somewhat less
523         //    random, because symbols declared close together are assigned names
524         //    that are quite similar. With this heuristic, the output is more
525         //    compressible.
526         //    For instance, the output may look like:
527         //    var da = "..", ea = "..";
528         //    function fa() { .. } function ga() { .. }
529     
530         int numPendingAssignments = generatedNamesForAssignments.size();
531         for (int i = 0; i < numPendingAssignments;) {
532           SortedSet<Assignment> varsByOrderOfOccurrence =
533               new TreeSet<>(ORDER_OF_OCCURRENCE_COMPARATOR);
534     
535           // Add k number of Assignment to the set, where k is the number of
536           // generated names of the same length.
537           int len = generatedNamesForAssignments.get(i).length();
538           for (int j = i; j < numPendingAssignments
539               && generatedNamesForAssignments.get(j).length() == len; j++) {
540             varsByOrderOfOccurrence.add(pendingAssignments.get(j));
541           }
542     
543           // Now, make the assignments
544           for (Assignment a : varsByOrderOfOccurrence) {
545             finalizeNameAssignment(a, generatedNamesForAssignments.get(i));
546             ++i;
547           }
548         }
549       }

551       /**
552        * Makes a final name assignment.
553        */
554       private void finalizeNameAssignment(Assignment a, String newName) {
555         a.setNewName(newName);
556     
557         // Keep track of the mapping
558         renameMap.put(a.oldName, newName);
559     
560         // Log the mapping
561         assignmentLog.append(a.oldName).append(" => ").append(newName).append('\n');
562       }

571       /**
572        * Determines whether a variable name is okay to rename.
573        */
574       private boolean okToRenameVar(String name, boolean isLocal) {
575         return !compiler.getCodingConvention().isExported(name, isLocal);
576       }

578       /**
579        * Returns the index within the scope stack.
580        * e.g. function Foo(a) { var b; function c(d) { } }
581        * a = 0, b = 1, c = 2, d = 3
582        */
583       private int getLocalVarIndex(Var v) {
584         int num = v.index;
585         Scope s = v.scope.getParent();
586         if (s == null) {
587           throw new IllegalArgumentException("Var is not local");
588         }
589     
590         boolean isBleedingIntoScope = s.getParent() != null &&
591             localBleedingFunctions.contains(v);
592     
593         while (s.getParent() != null) {
594           if (isBleedingIntoScope) {
595             num += localBleedingFunctionsPerScope.get(s).indexOf(v) + 1;
596             isBleedingIntoScope = false;
597           } else {
598             num += localBleedingFunctionsPerScope.get(s).size();
599           }
600           if (shouldTemporarilyRenameLocalsInScope(s)) {
601             num += s.getVarCount();
602           }
603           s = s.getParent();
604         }
605         return num;
606       }

608       /**
609        * Returns true if the local variables in a scope should be given
610        * temporary names (eg, 'L 123') prior to renaming to allow reuse of
611        * names across scopes.  With {@code preferStableNames}, temporary
612        * renaming is disabled if the number of locals in the scope is
613        * above a heuristic threshold to allow effective reuse of rename
614        * maps (see {@code prevUsedRenameMap}).  In scopes with many
615        * variables the temporary name given to a variable is unlikely to
616        * be the same temporary name used when the rename map was created.
617        */
618       private boolean shouldTemporarilyRenameLocalsInScope(Scope s) {
619         return (!preferStableNames ||
620             s.getVarCount() <= MAX_LOCALS_IN_SCOPE_TO_TEMP_RENAME);
621       }
