25       // VariableStatement, Declaration (let, const, function or class),
26       // or default with assignmentExpression
27       public final boolean isDefault;
28       public final boolean isExportAll;
29       public final ParseTree declaration;
30       public final ImmutableList<ParseTree> exportSpecifierList;
31       public final LiteralToken from;
32     
33       public ExportDeclarationTree(SourceRange location,
34           boolean isDefault,
35           boolean isExportAll,
36           ParseTree declaration,
37           ImmutableList<ParseTree> exportSpecifierList,
38           LiteralToken from) {
39         super(ParseTreeType.EXPORT_DECLARATION, location);
40         this.isDefault = isDefault;
41         this.isExportAll = isExportAll;
42         this.declaration = declaration;
43         this.exportSpecifierList = exportSpecifierList;
44         this.from = from;
45       }
