34     /**
35      * A flow scope that tries to store as little symbol information as possible,
36      * instead delegating to its parents. Optimized for low memory use.
37      *
38      * @author nicksantos@google.com (Nick Santos)
39      */
40     class LinkedFlowScope implements FlowScope {
41       // The closest flow scope cache.
42       private final FlatFlowScopeCache cache;
43     
44       // The parent flow scope.
45       private final LinkedFlowScope parent;
46     
47       // The distance between this flow scope and the closest flat flow scope.
48       private int depth;
49     
50       static final int MAX_DEPTH = 250;
51     
52       // A FlatFlowScopeCache equivalent to this scope.
53       private FlatFlowScopeCache flattened;
54     
55       // Flow scopes assume that all their ancestors are immutable.
56       // So once a child scope is created, this flow scope may not be modified.
57       private boolean frozen = false;
58     
59       // The last slot defined in this flow instruction, and the head of the
60       // linked list of slots.
61       private LinkedFlowSlot lastSlot;
62     
63       private LinkedFlowScope(FlatFlowScopeCache cache,
64           LinkedFlowScope directParent) {
65         this.cache = cache;
66         if (directParent == null) {
67           this.lastSlot = null;
68           this.depth = 0;
69           this.parent = cache.linkedEquivalent;
70         } else {
71           this.lastSlot = directParent.lastSlot;
72           this.depth = directParent.depth + 1;
73           this.parent = directParent;
74         }
75       }
76     
77       LinkedFlowScope(FlatFlowScopeCache cache) {
78         this(cache, null);
79       }
80     
81       LinkedFlowScope(LinkedFlowScope directParent) {
82         this(directParent.cache, directParent);
83       }

85       /** Gets the function scope for this flow scope. */
86       private TypedScope getFunctionScope() {
87         return cache.functionScope;
88       }

90       /** Whether this flows from a bottom scope. */
91       private boolean flowsFromBottom() {
92         return getFunctionScope().isBottom();
93       }

95       /**
96        * Creates an entry lattice for the flow.
97        */
98       public static LinkedFlowScope createEntryLattice(TypedScope scope) {
99         return new LinkedFlowScope(new FlatFlowScopeCache(scope));

102       @Override
103       public void inferSlotType(String symbol, JSType type) {
104         Preconditions.checkState(!frozen);
105         lastSlot = new LinkedFlowSlot(symbol, type, lastSlot);
106         depth++;
107         cache.dirtySymbols.add(symbol);
108       }

110       @Override
111       public void inferQualifiedSlot(Node node, String symbol, JSType bottomType,
112           JSType inferredType, boolean declared) {
113         TypedScope functionScope = getFunctionScope();
114         if (functionScope.isLocal()) {
115           TypedVar v  = functionScope.getVar(symbol);
116           if (v == null && !functionScope.isBottom()) {
117             v = functionScope.declare(symbol, node, bottomType, null, !declared);
118           }
119     
120           if (v != null && !v.isTypeInferred()) {
121             JSType declaredType = v.getType();
122             // Use the inferred type over the declared type only if the
123             // inferred type is a strict subtype of the declared type.
124             if (declaredType != null && inferredType.isSubtype(declaredType)
125                 && !declaredType.isSubtype(inferredType)
126                 && !inferredType.isEquivalentTo(declaredType)) {
127               inferSlotType(symbol, inferredType);
128             }
129           } else {
130             inferSlotType(symbol, inferredType);
131           }
132         }
133       }

150       /**
151        * Get the slot for the given symbol.
152        */
153       @Override
154       public StaticTypedSlot<JSType> getSlot(String name) {
155         if (cache.dirtySymbols.contains(name)) {
156           for (LinkedFlowSlot slot = lastSlot;
157                slot != null; slot = slot.parent) {
158             if (slot.getName().equals(name)) {
159               return slot;
160             }
161           }
162         }
163         return cache.getSlot(name);
164       }

166       @Override
167       public StaticTypedSlot<JSType> getOwnSlot(String name) {
168         throw new UnsupportedOperationException();
169       }

171       @Override
172       public FlowScope createChildFlowScope() {
173         frozen = true;
174     
175         if (depth > MAX_DEPTH) {
176           if (flattened == null) {
177             flattened = new FlatFlowScopeCache(this);
178           }
179           return new LinkedFlowScope(flattened);
180         }
181     
182         return new LinkedFlowScope(this);
183       }

185       /**
186        * Iterate through all the linked flow scopes before this one.
187        * If there's one and only one slot defined between this scope
188        * and the blind scope, return it.
189        */
190       @Override
191       public StaticTypedSlot<JSType> findUniqueRefinedSlot(FlowScope blindScope) {
192         StaticTypedSlot<JSType> result = null;
193     
194         for (LinkedFlowScope currentScope = this;
195              currentScope != blindScope;
196              currentScope = currentScope.parent) {
197           for (LinkedFlowSlot currentSlot = currentScope.lastSlot;
198                currentSlot != null &&
199                (currentScope.parent == null ||
200                 currentScope.parent.lastSlot != currentSlot);
201                currentSlot = currentSlot.parent) {
202             if (result == null) {
203               result = currentSlot;
204             } else if (!currentSlot.getName().equals(result.getName())) {
205               return null;
206             }
207           }
208         }
209     
210         return result;
211       }

213       /**
214        * Look through the given scope, and try to find slots where it doesn't
215        * have enough type information. Then fill in that type information
216        * with stuff that we've inferred in the local flow.
217        */
218       @Override
219       public void completeScope(StaticTypedScope<JSType> staticScope) {
220         TypedScope scope = (TypedScope) staticScope;
221         for (TypedVar var : scope.getVarIterable()) {
222           if (var.isTypeInferred()) {
223             JSType type = var.getType();
224             if (type == null || type.isUnknownType()) {
225               JSType flowType = getSlot(var.getName()).getType();
226               var.setType(flowType);
227             }
228           }
229         }
230       }

232       /**
233        * Remove flow scopes that add nothing to the flow.
234        */
235       // NOTE(nicksantos): This function breaks findUniqueRefinedSlot, because
236       // findUniqueRefinedSlot assumes that this scope is a direct descendant
237       // of blindScope. This is not necessarily true if this scope has been
238       // optimize()d and blindScope has not. This should be fixed. For now,
239       // we only use optimize() where we know that we won't have to do
240       // a findUniqueRefinedSlot on it.
241       @Override
242       public LinkedFlowScope optimize() {
243         LinkedFlowScope current;
244         for (current = this;
245              current.parent != null &&
246                  current.lastSlot == current.parent.lastSlot;
247              current = current.parent) {}
248         return current;
249       }

251       /** Join the two FlowScopes. */
252       static class FlowScopeJoinOp extends JoinOp.BinaryJoinOp<FlowScope> {
253         @SuppressWarnings("unchecked")
254         @Override
255         public FlowScope apply(FlowScope a, FlowScope b) {
256           // To join the two scopes, we have to
257           LinkedFlowScope linkedA = (LinkedFlowScope) a;
258           LinkedFlowScope linkedB = (LinkedFlowScope) b;
259           linkedA.frozen = true;
260           linkedB.frozen = true;
261           if (linkedA.optimize() == linkedB.optimize()) {
262             return linkedA.createChildFlowScope();
263           }
264           return new LinkedFlowScope(new FlatFlowScopeCache(linkedA, linkedB));
265         }
266       }

268       @Override
269       public boolean equals(Object other) {
270         if (other instanceof LinkedFlowScope) {
271           LinkedFlowScope that = (LinkedFlowScope) other;
272           if (this.optimize() == that.optimize()) {
273             return true;
274           }
275     
276           // If two flow scopes are in the same function, then they could have
277           // two possible function scopes: the real one and the BOTTOM scope.
278           // If they have different function scopes, we *should* iterate through all
279           // the variables in each scope and compare. However, 99.9% of the time,
280           // they're not equal. And the other .1% of the time, we can pretend
281           // they're equal--this just means that data flow analysis will have
282           // to propagate the entry lattice a little bit further than it
283           // really needs to. Everything will still come out ok.
284           if (this.getFunctionScope() != that.getFunctionScope()) {
285             return false;
286           }
287     
288           if (cache == that.cache) {
289             // If the two flow scopes have the same cache, then we can check
290             // equality a lot faster: by just looking at the "dirty" elements
291             // in the cache, and comparing them in both scopes.
292             for (String name : cache.dirtySymbols) {
293               if (diffSlots(getSlot(name), that.getSlot(name))) {
294                 return false;
295               }
296             }
297     
298             return true;
299           }
300     
301           Map<String, StaticTypedSlot<JSType>> myFlowSlots = allFlowSlots();
302           Map<String, StaticTypedSlot<JSType>> otherFlowSlots = that.allFlowSlots();
303     
304           for (StaticTypedSlot<JSType> slot : myFlowSlots.values()) {
305             if (diffSlots(slot, otherFlowSlots.get(slot.getName()))) {
306               return false;
307             }
308             otherFlowSlots.remove(slot.getName());
309           }
310           for (StaticTypedSlot<JSType> slot : otherFlowSlots.values()) {
311             if (diffSlots(slot, myFlowSlots.get(slot.getName()))) {
312               return false;
313             }
314           }
315           return true;
316         }
317         return false;
318       }

320       /**
321        * Determines whether two slots are meaningfully different for the
322        * purposes of data flow analysis.
323        */
324       private static boolean diffSlots(StaticTypedSlot<JSType> slotA,
325                                        StaticTypedSlot<JSType> slotB) {
326         boolean aIsNull = slotA == null || slotA.getType() == null;
327         boolean bIsNull = slotB == null || slotB.getType() == null;
328         if (aIsNull && bIsNull) {
329           return false;
330         } else if (aIsNull ^ bIsNull) {
331           return true;
332         }
333     
334         // Both slots and types must be non-null.
335         return slotA.getType().differsFrom(slotB.getType());
336       }

338       /**
339        * Gets all the symbols that have been defined before this point
340        * in the current flow. Does not return slots that have not changed during
341        * the flow.
342        *
343        * For example, consider the code:
344        * <code>
345        * var x = 3;
346        * function f() {
347        *   var y = 5;
348        *   y = 6; // FLOW POINT
349        *   var z = y;
350        *   return z;
351        * }
352        * </code>
353        * A FlowScope at FLOW POINT will return a slot for y, but not
354        * a slot for x or z.
355        */
356       private Map<String, StaticTypedSlot<JSType>> allFlowSlots() {
357         Map<String, StaticTypedSlot<JSType>> slots = new LinkedHashMap<>();
358         for (LinkedFlowSlot slot = lastSlot;
359              slot != null; slot = slot.parent) {
360           if (!slots.containsKey(slot.getName())) {
361             slots.put(slot.getName(), slot);
362           }
363         }
364     
365         for (Map.Entry<String, StaticTypedSlot<JSType>> symbolEntry : cache.symbols.entrySet()) {
366           if (!slots.containsKey(symbolEntry.getKey())) {
367             slots.put(symbolEntry.getKey(), symbolEntry.getValue());
368           }
369         }
370     
371         return slots;
372       }

391       /**
392        * A map that tries to cache as much symbol table information
393        * as possible in a map. Optimized for fast lookup.
394        */
395       private static class FlatFlowScopeCache {
396         // The TypedScope for the entire function or for the global scope.
397         private final TypedScope functionScope;
398     
399         // The linked flow scope that this cache represents.
400         private final LinkedFlowScope linkedEquivalent;
401     
402         // All the symbols defined before this point in the local flow.
403         // May not include lazily declared qualified names.
404         private Map<String, StaticTypedSlot<JSType>> symbols = new LinkedHashMap<>();
405     
406         // Used to help make lookup faster for LinkedFlowScopes by recording
407         // symbols that may be redefined "soon", for an arbitrary definition
408         // of "soon". ;)
409         //
410         // More rigorously, if a symbol is redefined in a LinkedFlowScope,
411         // and this is the closest FlatFlowScopeCache, then that symbol is marked
412         // "dirty". In this way, we don't waste time looking in the LinkedFlowScope
413         // list for symbols that aren't defined anywhere nearby.
414         final Set<String> dirtySymbols = new LinkedHashSet<>();
415     
416         // The cache at the bottom of the lattice.
417         FlatFlowScopeCache(TypedScope functionScope) {
418           this.functionScope = functionScope;
419           symbols = ImmutableMap.of();
420           linkedEquivalent = null;
421         }
422     
423         // A cache in the middle of a long scope chain.
424         FlatFlowScopeCache(LinkedFlowScope directParent) {
425           FlatFlowScopeCache cache = directParent.cache;
426     
427           functionScope = cache.functionScope;
428           symbols = directParent.allFlowSlots();
429           linkedEquivalent = directParent;
430         }
431     
432         // A cache at the join of two scope chains.
433         FlatFlowScopeCache(LinkedFlowScope joinedScopeA,
434             LinkedFlowScope joinedScopeB) {
435           linkedEquivalent = null;
436     
437           // Always prefer the "real" function scope to the faked-out
438           // bottom scope.
439           functionScope = joinedScopeA.flowsFromBottom() ?
440               joinedScopeB.getFunctionScope() : joinedScopeA.getFunctionScope();
441     
442           Map<String, StaticTypedSlot<JSType>> slotsA = joinedScopeA.allFlowSlots();
443           Map<String, StaticTypedSlot<JSType>> slotsB = joinedScopeB.allFlowSlots();
444     
445           symbols = slotsA;
446     
447           // There are 5 different join cases:
448           // 1) The type is declared in joinedScopeA, not in joinedScopeB,
449           //    and not in functionScope. Just use the one in A.
450           // 2) The type is declared in joinedScopeB, not in joinedScopeA,
451           //    and not in functionScope. Just use the one in B.
452           // 3) The type is declared in functionScope and joinedScopeA, but
453           //    not in joinedScopeB. Join the two types.
454           // 4) The type is declared in functionScope and joinedScopeB, but
455           //    not in joinedScopeA. Join the two types.
456           // 5) The type is declared in joinedScopeA and joinedScopeB. Join
457           //    the two types.
458     
459           for (String name : Iterables.concat(symbols.keySet(), slotsB.keySet())) {
460             StaticTypedSlot<JSType> slotA = slotsA.get(name);
461             StaticTypedSlot<JSType> slotB = slotsB.get(name);
462     
463             JSType joinedType = null;
464             if (slotB == null || slotB.getType() == null) {
465               StaticTypedSlot<JSType> fnSlot
466                   = joinedScopeB.getFunctionScope().getSlot(name);
467               JSType fnSlotType = fnSlot == null ? null : fnSlot.getType();
468               if (fnSlotType == null) {
469                 // Case #1 -- already inserted.
470               } else {
471                 // Case #3
472                 joinedType = slotA.getType().getLeastSupertype(fnSlotType);
473               }
474             } else if (slotA == null || slotA.getType() == null) {
475               StaticTypedSlot<JSType> fnSlot
476                   = joinedScopeA.getFunctionScope().getSlot(name);
477               JSType fnSlotType = fnSlot == null ? null : fnSlot.getType();
478               if (fnSlotType == null) {
479                 // Case #2
480                 symbols.put(name, slotB);
481               } else {
482                 // Case #4
483                 joinedType = slotB.getType().getLeastSupertype(fnSlotType);
484               }
485             } else {
486               // Case #5
487               joinedType =
488                   slotA.getType().getLeastSupertype(slotB.getType());
489             }
490     
491             if (joinedType != null) {
492               symbols.put(name, new SimpleSlot(name, joinedType, true));
493             }
494           }
495         }
496     
497         /**
498          * Get the slot for the given symbol.
499          */
500         public StaticTypedSlot<JSType> getSlot(String name) {
501           if (symbols.containsKey(name)) {
502             return symbols.get(name);
503           } else {
504             return functionScope.getSlot(name);
505           }
506         }
507       }
