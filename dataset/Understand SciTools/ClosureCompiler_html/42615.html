54     /**
55      * New type inference algorithm.
56      *
57      * Main differences from the old type checker:
58      * - Infers types for unannotated functions
59      * - Stricter with missing-property warnings
60      * - Stricter when checking the operands of primitive operators
61      * - Requires constants to be typed
62      * - Tries to warn about misplaced annotations, rather than silently ignore them
63      *
64      * @author blickly@google.com (Ben Lickly)
65      * @author dimvar@google.com (Dimitris Vardoulakis)
66      *
67      */

390       @Override
391       public void process(Node externs, Node root) {
392         try {
393           this.symbolTable = (GlobalTypeInfo) compiler.getSymbolTable();
394           this.commonTypes = symbolTable.getTypesUtilObject();
395           for (NTIScope scope : symbolTable.getScopes()) {
396             analyzeFunction(scope);
397             envs.clear();
398           }
399           for (DeferredCheck check : deferredChecks.values()) {
400             check.runCheck(summaries, warnings);
401           }
402           if (measureMem) {
403             System.out.println("Peak mem: " + peakMem + "MB");
404           }
405         } catch (Exception unexpectedException) {
406           String message = unexpectedException.getMessage();
407           if (currentScope != null) {
408             message += "\nIn scope: " + currentScope;
409           }
410           this.compiler.throwInternalError(message, unexpectedException);
411         }
412       }

418           peakMem = currentUsedMem;

432       private TypeEnv getInEnv(DiGraphNode<Node, ControlFlowGraph.Branch> dn) {
433         List<DiGraphEdge<Node, ControlFlowGraph.Branch>> inEdges = dn.getInEdges();
434         // True for code considered dead in the CFG
435         if (inEdges.isEmpty()) {
436           return getEntryTypeEnv();
437         }
438         if (inEdges.size() == 1) {
439           return envs.get(inEdges.get(0));
440         }
441     
442         Set<TypeEnv> envSet = new LinkedHashSet<>();
443         for (DiGraphEdge<Node, ControlFlowGraph.Branch> de : inEdges) {
444           TypeEnv env = envs.get(de);
445           if (env != null) {
446             envSet.add(env);
447           }
448         }
449         if (envSet.isEmpty()) {
450           return null;
451         }
452         return TypeEnv.join(envSet);
453       }

455       private TypeEnv getOutEnv(DiGraphNode<Node, ControlFlowGraph.Branch> dn) {
456         List<DiGraphEdge<Node, ControlFlowGraph.Branch>> outEdges = dn.getOutEdges();
457         if (outEdges.isEmpty()) {
458           // This occurs when visiting a throw in the backward direction.
459           Preconditions.checkArgument(dn.getValue().isThrow());
460           return this.typeEnvFromDeclaredTypes;
461         }
462         if (outEdges.size() == 1) {
463           return envs.get(outEdges.get(0));
464         }
465         Set<TypeEnv> envSet = new LinkedHashSet<>();
466         for (DiGraphEdge<Node, ControlFlowGraph.Branch> de : outEdges) {
467           TypeEnv env = envs.get(de);
468           if (env != null) {
469             envSet.add(env);
470           }
471         }
472         Preconditions.checkState(!envSet.isEmpty());
473         return TypeEnv.join(envSet);
474       }

484       // Initialize the type environments on the CFG edges before the FWD analysis.
485       private void initEdgeEnvsFwd(TypeEnv entryEnv) {
486         envs.clear();
487     
488         // For function scopes, add the formal parameters and the free variables
489         // from outer scopes to the environment.
490         Set<String> nonLocals = new LinkedHashSet<>();
491         if (currentScope.isFunction()) {
492           if (currentScope.getName() != null) {
493             nonLocals.add(currentScope.getName());
494           }
495           nonLocals.addAll(currentScope.getOuterVars());
496           nonLocals.addAll(currentScope.getFormals());
497           if (currentScope.hasThis()) {
498             nonLocals.add(THIS_ID);
499           }
500           entryEnv = envPutType(entryEnv, RETVAL_ID, JSType.UNDEFINED);
501         } else {
502           nonLocals.addAll(currentScope.getExterns());
503         }
504         for (String name : nonLocals) {
505           JSType declType = currentScope.getDeclaredTypeOf(name);
506           JSType initType =
507               declType == null ? envGetType(entryEnv, name) : declType;
508           println("Adding non-local ", name,
509               " with decltype: ", declType,
510               " and inittype: ", initType);
511           entryEnv = envPutType(entryEnv, name, initType);
512         }
513     
514         // For all scopes, add local variables and (local) function definitions
515         // to the environment.
516         for (String local : currentScope.getLocals()) {
517           if (!currentScope.isFunctionNamespace(local)) {
518             entryEnv = envPutType(entryEnv, local, JSType.UNDEFINED);
519           }
520         }
521         for (String fnName : currentScope.getLocalFunDefs()) {
522           entryEnv = envPutType(entryEnv, fnName, getSummaryOfLocalFunDef(fnName));
523         }
524         println("Keeping env: ", entryEnv);
525         setOutEnv(this.cfg.getEntry(), entryEnv);
526       }

528       private TypeEnv getTypeEnvFromDeclaredTypes() {
529         TypeEnv env = new TypeEnv();
530         Set<String> varNames = currentScope.getOuterVars();
531         Set<String> locals = currentScope.getLocals();
532         varNames.addAll(locals);
533         varNames.addAll(currentScope.getExterns());
534         if (currentScope.isFunction()) {
535           if (currentScope.getName() != null) {
536             varNames.add(currentScope.getName());
537           }
538           varNames.addAll(currentScope.getFormals());
539           if (currentScope.hasThis()) {
540             varNames.add(THIS_ID);
541           }
542           // In the rare case when there is a local variable named "arguments",
543           // this entry will be overwritten in the foreach loop below.
544           JSType argumentsType;
545           DeclaredFunctionType dft = currentScope.getDeclaredFunctionType();
546           if (dft.getOptionalArity() == 0 && dft.hasRestFormals()) {
547             argumentsType = dft.getRestFormalsType();
548           } else {
549             argumentsType = JSType.UNKNOWN;
550           }
551           env = envPutType(env, "arguments",
552               commonTypes.getArgumentsArrayType(argumentsType));
553         }
554         for (String varName : varNames) {
555           if (!locals.contains(varName) || !currentScope.isFunctionNamespace(varName)) {
556             JSType declType = currentScope.getDeclaredTypeOf(varName);
557             env = envPutType(env, varName, declType == null ? JSType.UNKNOWN : declType);
558           }
559         }
560         for (String fnName : currentScope.getLocalFunDefs()) {
561           env = envPutType(env, fnName, getSummaryOfLocalFunDef(fnName));
562         }
563         return env;
564       }

566       private JSType getSummaryOfLocalFunDef(String name) {
567         NTIScope fnScope = currentScope.getScope(name);
568         JSType fnType = summaries.get(fnScope);
569         if (fnType != null) {
570           return fnType;
571         }
572         // Functions defined in externs have no summary, so use the declared type
573         fnType = currentScope.getDeclaredTypeOf(name);
574         Preconditions.checkState(fnType.getFunType() != null,
575             "Needed function but found %s", fnType);
576         return changeTypeIfFunctionNamespace(fnScope, fnType);
577       }

586       private void buildWorksetHelper(
587           DiGraphNode<Node, ControlFlowGraph.Branch> dn,
588           List<DiGraphNode<Node, ControlFlowGraph.Branch>> workset,
589           Set<DiGraphNode<Node, ControlFlowGraph.Branch>> seen) {
590         if (seen.contains(dn) || dn == this.cfg.getImplicitReturn()) {
591           return;
592         }
593         switch (dn.getValue().getType()) {
594           case DO:
595           case WHILE:
596           case FOR:
597             // Do the loop body first, then the loop follow.
598             // For DO loops, we do BODY-CONDT-CONDF-FOLLOW
599             // Since CONDT is currently unused, this could be optimized.
600             List<DiGraphEdge<Node, ControlFlowGraph.Branch>> outEdges =
601                 dn.getOutEdges();
602             seen.add(dn);
603             workset.add(dn);
604             for (DiGraphEdge<Node, ControlFlowGraph.Branch> outEdge : outEdges) {
605               if (outEdge.getValue() == ControlFlowGraph.Branch.ON_TRUE) {
606                 buildWorksetHelper(outEdge.getDestination(), workset, seen);
607               }
608             }
609             workset.add(dn);
610             for (DiGraphEdge<Node, ControlFlowGraph.Branch> outEdge : outEdges) {
611               if (outEdge.getValue() == ControlFlowGraph.Branch.ON_FALSE) {
612                 buildWorksetHelper(outEdge.getDestination(), workset, seen);
613               }
614             }
615             break;
616           default: {
617             // Wait for all other incoming edges at join nodes.
618             for (DiGraphEdge<Node, ControlFlowGraph.Branch> inEdge :
619                 dn.getInEdges()) {
620               if (!seen.contains(inEdge.getSource())
621                   && !inEdge.getSource().getValue().isDo()) {
622                 return;
623               }
624             }
625             seen.add(dn);
626             if (this.cfg.getEntry() != dn) {
627               workset.add(dn);
628             }
629             // Don't recur for straight-line code
630             while (true) {
631               Node n = dn.getValue();
632               if (n.isTry()) {
633                 maybeAddDeadCode(workset, seen, n.getSecondChild());
634               } else if (n.isBreak() || n.isContinue() || n.isThrow()) {
635                 maybeAddDeadCode(workset, seen, n.getNext());
636               }
637               List<DiGraphNode<Node, ControlFlowGraph.Branch>> succs =
638                   this.cfg.getDirectedSuccNodes(dn);
639               if (succs.size() != 1) {
640                 break;
641               }
642               DiGraphNode<Node, ControlFlowGraph.Branch> succ = succs.get(0);
643               if (succ == this.cfg.getImplicitReturn()) {
644                 if (n.getNext() != null) {
645                   maybeAddDeadCode(workset, seen, n.getNext());
646                 }
647                 return;
648               }
649               // Make sure that succ isn't a join node
650               if (this.cfg.getDirectedPredNodes(succ).size() > 1) {
651                 break;
652               }
653               workset.add(succ);
654               seen.add(succ);
655               dn = succ;
656             }
657             for (DiGraphNode<Node, ControlFlowGraph.Branch> succ :
658               this.cfg.getDirectedSuccNodes(dn)) {
659               buildWorksetHelper(succ, workset, seen);
660             }
661             break;
662           }
663         }
664       }
665     
666       // Analyze dead code, such as a catch that is never executed or a statement
667       // following a return/break/continue. This code can be a predecessor of live
668       // code in the cfg. We wait on incoming edges before adding nodes to the
669       // workset, and don't want dead code to block live code from being analyzed.
670       private void maybeAddDeadCode(
671           List<DiGraphNode<Node, ControlFlowGraph.Branch>> workset,
672           Set<DiGraphNode<Node, ControlFlowGraph.Branch>> seen,
673           Node maybeDeadNode) {
674         if (maybeDeadNode == null) {
675           return;
676         }
677         DiGraphNode<Node, ControlFlowGraph.Branch> cfgNode =
678             this.cfg.getDirectedGraphNode(maybeDeadNode);
679         if (cfgNode == null) {
680           return;
681         }
682         if (this.cfg.getDirectedPredNodes(cfgNode).isEmpty()) {
683           buildWorksetHelper(cfgNode, workset, seen);
684         }
685       }

692         this.cfg = cfa.getCfg();
693         println(this.cfg);

696         List<DiGraphNode<Node, ControlFlowGraph.Branch>> workset =

698         buildWorkset(this.cfg.getEntry(), workset);

736       private void analyzeFunctionBwd(
737           List<DiGraphNode<Node, ControlFlowGraph.Branch>> workset) {
738         for (DiGraphNode<Node, ControlFlowGraph.Branch> dn : workset) {
739           Node n = dn.getValue();
740           TypeEnv outEnv = Preconditions.checkNotNull(getOutEnv(dn));
741           TypeEnv inEnv;
742           println("\tBWD Statment: ", n);
743           println("\t\toutEnv: ", outEnv);
744           switch (n.getType()) {
745             case EXPR_RESULT:
746               inEnv = analyzeExprBwd(n.getFirstChild(), outEnv, JSType.UNKNOWN).env;
747               break;
748             case RETURN: {
749               Node retExp = n.getFirstChild();
750               if (retExp == null) {
751                 inEnv = outEnv;
752               } else {
753                 JSType declRetType = currentScope.getDeclaredFunctionType().getReturnType();
754                 declRetType = declRetType == null ? JSType.UNKNOWN : declRetType;
755                 inEnv = analyzeExprBwd(retExp, outEnv, declRetType).env;
756               }
757               break;
758             }
759             case VAR: {
760               if (NodeUtil.isTypedefDecl(n)) {
761                 inEnv = outEnv;
762                 break;
763               }
764               inEnv = outEnv;
765               for (Node nameNode = n.getFirstChild(); nameNode != null;
766                    nameNode = nameNode.getNext()) {
767                 String varName = nameNode.getString();
768                 Node rhs = nameNode.getFirstChild();
769                 JSType declType = currentScope.getDeclaredTypeOf(varName);
770                 inEnv = envPutType(inEnv, varName, JSType.UNKNOWN);
771                 if (rhs == null || currentScope.isLocalFunDef(varName)) {
772                   continue;
773                 }
774                 JSType inferredType = envGetType(outEnv, varName);
775                 JSType requiredType;
776                 if (declType == null) {
777                   requiredType = inferredType;
778                 } else {
779                   // TODO(dimvar): look if the meet is needed
780                   requiredType = JSType.meet(declType, inferredType);
781                   if (requiredType.isBottom()) {
782                     requiredType = JSType.UNKNOWN;
783                   }
784                 }
785                 inEnv = analyzeExprBwd(rhs, inEnv, requiredType).env;
786               }
787               break;
788             }
789             case BLOCK:
790             case BREAK:
791             case CATCH:
792             case CONTINUE:
793             case DEFAULT_CASE:
794             case DEBUGGER:
795             case EMPTY:
796             case SCRIPT:
797             case TRY:
798               inEnv = outEnv;
799               break;
800             case DO:
801             case FOR:
802             case IF:
803             case WHILE:
804               Node expr = NodeUtil.isForIn(n) ?
805                   n.getFirstChild() : NodeUtil.getConditionExpression(n);
806               inEnv = analyzeExprBwd(expr, outEnv).env;
807               break;
808             case THROW:
809             case CASE:
810             case SWITCH:
811               inEnv = analyzeExprBwd(n.getFirstChild(), outEnv).env;
812               break;
813             default:
814               if (NodeUtil.isStatement(n)) {
815                 throw new RuntimeException("Unhandled statement type: " + n.getType());
816               } else {
817                 inEnv = analyzeExprBwd(n, outEnv).env;
818                 break;
819               }
820           }
821           println("\t\tinEnv: ", inEnv);
822           for (DiGraphEdge<Node, ControlFlowGraph.Branch> de : dn.getInEdges()) {
823             envs.put(de, inEnv);
824           }
825         }
826       }

828       private void analyzeFunctionFwd(
829           List<DiGraphNode<Node, ControlFlowGraph.Branch>> workset) {
830         for (DiGraphNode<Node, ControlFlowGraph.Branch> dn : workset) {
831           Node n = dn.getValue();
832           Node parent = n.getParent();
833           Preconditions.checkState(n != null,
834               "Implicit return should not be in workset.");
835           TypeEnv inEnv = getInEnv(dn);
836           TypeEnv outEnv = null;
837           if (parent.isScript()
838               || (parent.isBlock() && parent.getParent().isFunction())) {
839             // All joins have merged; forget changes
840             inEnv = inEnv.clearChangeLog();
841           }
842     
843           println("\tFWD Statment: ", n);
844           println("\t\tinEnv: ", inEnv);
845           boolean conditional = false;
846           switch (n.getType()) {
847             case BLOCK:
848             case BREAK:
849             case CONTINUE:
850             case DEFAULT_CASE:
851             case DEBUGGER:
852             case EMPTY:
853             case FUNCTION:
854             case SCRIPT:
855             case TRY:
856               outEnv = inEnv;
857               break;
858             case CATCH:
859               String catchVarname = n.getFirstChild().getString();
860               outEnv = envPutType(inEnv, catchVarname, JSType.UNKNOWN);
861               break;
862             case EXPR_RESULT:
863               println("\tsemi ", n.getFirstChild().getType());
864               if (n.getBooleanProp(Node.ANALYZED_DURING_GTI)) {
865                 n.removeProp(Node.ANALYZED_DURING_GTI);
866                 outEnv = inEnv;
867               } else {
868                 outEnv = analyzeExprFwd(n.getFirstChild(), inEnv, JSType.UNKNOWN).env;
869               }
870               break;
871             case RETURN: {
872               Node retExp = n.getFirstChild();
873               JSType declRetType = currentScope.getDeclaredFunctionType().getReturnType();
874               declRetType = declRetType == null ? JSType.UNKNOWN : declRetType;
875               JSType actualRetType;
876               if (retExp == null) {
877                 actualRetType = JSType.UNDEFINED;
878                 outEnv = envPutType(inEnv, RETVAL_ID, actualRetType);
879               } else {
880                 EnvTypePair retPair = analyzeExprFwd(retExp, inEnv, declRetType);
881                 actualRetType = retPair.type;
882                 outEnv = envPutType(retPair.env, RETVAL_ID, actualRetType);
883               }
884               if (!actualRetType.isSubtypeOf(declRetType)) {
885                 warnings.add(JSError.make(n, RETURN_NONDECLARED_TYPE,
886                         errorMsgWithTypeDiff(declRetType, actualRetType)));
887               }
888               break;
889             }
890             case DO:
891             case IF:
892             case FOR:
893             case WHILE:
894               if (NodeUtil.isForIn(n)) {
895                 Node obj = n.getSecondChild();
896                 EnvTypePair pair = analyzeExprFwd(obj, inEnv, pickReqObjType(n));
897                 pair = mayWarnAboutNullableReferenceAndTighten(n, pair.type, null, inEnv);
898                 JSType objType = pair.type;
899                 if (!objType.isSubtypeOf(JSType.TOP_OBJECT)) {
900                   warnings.add(JSError.make(
901                       obj, FORIN_EXPECTS_OBJECT, objType.toString()));
902                 } else if (objType.isStruct()) {
903                   warnings.add(JSError.make(obj, IN_USED_WITH_STRUCT));
904                 }
905                 Node lhs = n.getFirstChild();
906                 LValueResultFwd lval = analyzeLValueFwd(lhs, inEnv, JSType.STRING);
907                 if (lval.declType != null &&
908                     !commonTypes.isStringScalarOrObj(lval.declType)) {
909                   warnings.add(JSError.make(lhs, FORIN_EXPECTS_STRING_KEY,
910                       lval.declType.toString()));
911                   outEnv = lval.env;
912                 } else {
913                   outEnv = updateLvalueTypeInEnv(lval.env, lhs, lval.ptr, JSType.STRING);
914                 }
915                 break;
916               }
917               conditional = true;
918               analyzeConditionalStmFwd(dn, NodeUtil.getConditionExpression(n), inEnv);
919               break;
920             case CASE: {
921               conditional = true;
922               // See analyzeExprFwd#Token.CASE for how to handle this precisely
923               analyzeConditionalStmFwd(dn, n, inEnv);
924               break;
925             }
926             case VAR:
927               outEnv = inEnv;
928               if (NodeUtil.isTypedefDecl(n)) {
929                 break;
930               }
931               for (Node nameNode = n.getFirstChild(); nameNode != null;
932                    nameNode = nameNode.getNext()) {
933                 outEnv = processVarDeclaration(nameNode, outEnv);
934               }
935               break;
936             case SWITCH:
937             case THROW:
938               outEnv = analyzeExprFwd(n.getFirstChild(), inEnv).env;
939               break;
940             default:
941               if (NodeUtil.isStatement(n)) {
942                 throw new RuntimeException("Unhandled statement type: " + n.getType());
943               } else {
944                 outEnv = analyzeExprFwd(n, inEnv, JSType.UNKNOWN).env;
945                 break;
946               }
947           }
948     
949           if (!conditional) {
950             println("\t\toutEnv: ", outEnv);
951             setOutEnv(dn, outEnv);
952           }
953         }
954       }

956       private void analyzeConditionalStmFwd(
957           DiGraphNode<Node, ControlFlowGraph.Branch> stm, Node cond, TypeEnv inEnv) {
958         for (DiGraphEdge<Node, ControlFlowGraph.Branch> outEdge :
959             stm.getOutEdges()) {
960           JSType specializedType;
961           switch (outEdge.getValue()) {
962             case ON_TRUE:
963               specializedType = JSType.TRUTHY;
964               break;
965             case ON_FALSE:
966               specializedType = JSType.FALSY;
967               break;
968             case ON_EX:
969               specializedType = JSType.UNKNOWN;
970               break;
971             default:
972               throw new RuntimeException(
973                   "Condition with an unexpected edge type: " + outEdge.getValue());
974           }
975           envs.put(outEdge,
976               analyzeExprFwd(cond, inEnv, JSType.UNKNOWN, specializedType).env);
977         }
978       }

980       private void createSummary(NTIScope fn) {
981         Node fnRoot = fn.getRoot();
982         Preconditions.checkArgument(!fnRoot.isFromExterns());
983         FunctionTypeBuilder builder = new FunctionTypeBuilder();
984         TypeEnv entryEnv = getEntryTypeEnv();
985         TypeEnv exitEnv = getInEnv(this.cfg.getImplicitReturn());
986         if (exitEnv == null) {
987           // This function only exits with THROWs
988           exitEnv = envPutType(new TypeEnv(), RETVAL_ID, JSType.BOTTOM);
989         }
990     
991         DeclaredFunctionType declType = fn.getDeclaredFunctionType();
992         int reqArity = declType.getRequiredArity();
993         int optArity = declType.getOptionalArity();
994         if (declType.isGeneric()) {
995           builder.addTypeParameters(declType.getTypeParameters());
996         }
997     
998         // Every trailing undeclared formal whose inferred type is ?
999         // or contains undefined can be marked as optional.
1000         List<String> formals = fn.getFormals();
1001         for (int i = reqArity - 1; i >= 0; i--) {
1002           JSType formalType = fn.getDeclaredFunctionType().getFormalType(i);
1003           if (formalType != null) {
1004             break;
1005           }
1006           String formalName = formals.get(i);
1007           formalType = getTypeAfterFwd(formalName, entryEnv, exitEnv);
1008           if (formalType.isUnknown() || JSType.UNDEFINED.isSubtypeOf(formalType)) {
1009             reqArity--;
1010           } else {
1011             break;
1012           }
1013         }
1014     
1015         // Collect types of formals in the builder
1016         int formalIndex = 0;
1017         for (String formal : formals) {
1018           JSType formalType = fn.getDeclaredTypeOf(formal);
1019           if (formalType == null) {
1020             formalType = getTypeAfterFwd(formal, entryEnv, exitEnv);
1021           }
1022           if (formalIndex < reqArity) {
1023             builder.addReqFormal(formalType);
1024           } else if (formalIndex < optArity) {
1025             builder.addOptFormal(formalType);
1026           }
1027           formalIndex++;
1028         }
1029         if (declType.hasRestFormals()) {
1030           builder.addRestFormals(declType.getFormalType(formalIndex));
1031         }
1032     
1033         for (String outer : fn.getOuterVars()) {
1034           println("Free var ", outer, " going in summary");
1035           builder.addOuterVarPrecondition(outer, envGetType(entryEnv, outer));
1036         }
1037     
1038         builder.addNominalType(declType.getNominalType());
1039         builder.addReceiverType(declType.getReceiverType());
1040         JSType declRetType = declType.getReturnType();
1041         JSType actualRetType = Preconditions.checkNotNull(envGetType(exitEnv, RETVAL_ID));
1042     
1043         if (declRetType != null) {
1044           builder.addRetType(declRetType);
1045           if (!isAllowedToNotReturn(fn) &&
1046               !JSType.UNDEFINED.isSubtypeOf(declRetType) &&
1047               hasPathWithNoReturn(this.cfg)) {
1048             warnings.add(JSError.make(
1049                 fnRoot, MISSING_RETURN_STATEMENT, declRetType.toString()));
1050           }
1051         } else if (declType.getNominalType() == null) {
1052           // If someone uses the result of a function that doesn't return, they get a warning.
1053           builder.addRetType(actualRetType.isBottom() ? JSType.TOP : actualRetType);
1054         } else {
1055           // Don't infer a return type for constructors. We want to warn for
1056           // constructors called without new who don't explicitly declare @return.
1057           builder.addRetType(JSType.UNDEFINED);
1058         }
1059         JSType summary = commonTypes.fromFunctionType(builder.buildFunction());
1060         println("Function summary for ", fn.getReadableName());
1061         println("\t", summary);
1062         summary = changeTypeIfFunctionNamespace(fn, summary);
1063         summaries.put(fn, summary);
1064         maybeSetTypeI(fnRoot, summary);
1065         Node fnNameNode = NodeUtil.getNameNode(fnRoot);
1066         if (fnNameNode != null) {
1067           maybeSetTypeI(fnNameNode, summary);
1068         }
1069       }

1071       private JSType changeTypeIfFunctionNamespace(NTIScope fnScope, JSType fnType) {
1072         NTIScope enclosingScope = fnScope.getParent();
1073         Node fnNameNode = NodeUtil.getNameNode(fnScope.getRoot());
1074         JSType namespaceType = null;
1075         if (fnNameNode == null) {
1076           return fnType;
1077         }
1078         if (fnNameNode.isName()) {
1079           String fnName = fnNameNode.getString();
1080           if (enclosingScope.isFunctionNamespace(fnName)) {
1081             namespaceType = enclosingScope.getDeclaredTypeOf(fnName);
1082           }
1083         } else if (fnNameNode.isQualifiedName()) {
1084           QualifiedName qname = QualifiedName.fromNode(fnNameNode);
1085           JSType rootNs = enclosingScope.getDeclaredTypeOf(qname.getLeftmostName());
1086           namespaceType = rootNs == null ? null : rootNs.getProp(qname.getAllButLeftmost());
1087         }
1088         if (namespaceType != null && namespaceType.isNamespace()) {
1089           // Replace the less-precise declared function type
1090           // with the new function summary.
1091           return namespaceType.withFunction(
1092               fnType.getFunTypeIfSingletonObj(), commonTypes.getFunctionType());
1093         }
1094         return fnType;
1095       }

1098       // able to know whether a property was added during the evaluation of the

1100       private JSType getTypeAfterFwd(
1101           String varName, TypeEnv entryEnv, TypeEnv exitEnv) {
1102         JSType typeAfterBwd = envGetType(entryEnv, varName);
1103         if (!typeAfterBwd.hasNonScalar() || typeAfterBwd.getFunType() != null) {
1104           // The type of a formal after fwd is more precise than the type after bwd,
1105           // so we use typeAfterFwd in the summary.
1106           // Trade-off: If the formal is assigned in the body of a function, and the
1107           // new value has a different type, we compute a wrong summary.
1108           // Since this is rare, we prefer typeAfterFwd to typeAfterBwd.
1109           JSType typeAfterFwd = envGetType(exitEnv, varName);
1110           if (typeAfterFwd != null) {
1111             return typeAfterFwd;
1112           }
1113         }
1114         return typeAfterBwd;
1115       }

1117       private static boolean isAllowedToNotReturn(NTIScope methodScope) {
1118         Node fn = methodScope.getRoot();
1119         if (fn.isFromExterns()) {
1120           return true;
1121         }
1122         if (!NodeUtil.isPrototypeMethod(fn)) {
1123           return false;
1124         }
1125         JSType maybeInterface;
1126         Node ntQnameNode = NodeUtil.getPrototypeClassName(fn.getParent().getFirstChild());
1127         if (ntQnameNode.isName()) {
1128           maybeInterface = methodScope.getDeclaredTypeOf(ntQnameNode.getString());
1129         } else {
1130           QualifiedName ntQname = QualifiedName.fromNode(ntQnameNode);
1131           JSType rootNamespace = methodScope.getDeclaredTypeOf(ntQname.getLeftmostName());
1132           maybeInterface = rootNamespace == null
1133               ? null : rootNamespace.getProp(ntQname.getAllButLeftmost());
1134         }
1135         return maybeInterface != null && maybeInterface.isInterfaceDefinition();
1136       }

1138       private static boolean hasPathWithNoReturn(ControlFlowGraph<Node> cfg) {
1139         for (DiGraphNode<Node, ControlFlowGraph.Branch> dn :
1140                  cfg.getDirectedPredNodes(cfg.getImplicitReturn())) {
1141           Node stm = dn.getValue();
1142           if (NodeUtil.isLoopStructure(stm)) {
1143             Node cond = NodeUtil.getConditionExpression(stm);
1144             if (!(cond != null && NodeUtil.isImpureTrue(cond))) {
1145               return true;
1146             }
1147           } else if (!stm.isReturn()) {
1148             return true;
1149           }
1150         }
1151         return false;
1152       }

1154       /** Processes a single variable declaration in a VAR statement. */
1155       private TypeEnv processVarDeclaration(Node nameNode, TypeEnv inEnv) {
1156         String varName = nameNode.getString();
1157         JSType declType = currentScope.getDeclaredTypeOf(varName);
1158     
1159         if (currentScope.isLocalFunDef(varName)) {
1160           return inEnv;
1161         }
1162         if (NodeUtil.isNamespaceDecl(nameNode)
1163             || nameNode.getParent().getBooleanProp(Node.ANALYZED_DURING_GTI)) {
1164           Preconditions.checkNotNull(declType,
1165               "Can't skip var declaration with undeclared type at: %s", nameNode);
1166           maybeSetTypeI(nameNode, declType);
1167           return envPutType(inEnv, varName, declType);
1168         }
1169     
1170         Node rhs = nameNode.getFirstChild();
1171         TypeEnv outEnv = inEnv;
1172         JSType rhsType;
1173         if (rhs == null) {
1174           rhsType = JSType.UNDEFINED;
1175         } else {
1176           EnvTypePair pair = analyzeExprFwd(rhs, inEnv,
1177               declType != null ? declType : JSType.UNKNOWN);
1178           outEnv = pair.env;
1179           rhsType = pair.type;
1180         }
1181         if (declType != null && rhs != null) {
1182           if (!rhsType.isSubtypeOf(declType)) {
1183             warnings.add(JSError.make(
1184                 rhs, MISTYPED_ASSIGN_RHS,
1185                 errorMsgWithTypeDiff(declType, rhsType)));
1186             // Don't flow the wrong initialization
1187             rhsType = declType;
1188           } else {
1189             // We do this to preserve type attributes like declaredness of
1190             // a property
1191             rhsType = declType.specialize(rhsType);
1192           }
1193         }
1194         maybeSetTypeI(nameNode, rhsType);
1195         return envPutType(outEnv, varName, rhsType);
1196       }

1207       /**
1208        * @param requiredType The context requires this type; warn if the expression
1209        *                     doesn't have this type.
1210        * @param specializedType Used in boolean contexts to infer types of names.
1211        *
1212        * Invariant: specializedType is a subtype of requiredType.
1213        */
1214       private EnvTypePair analyzeExprFwd(
1215           Node expr, TypeEnv inEnv, JSType requiredType, JSType specializedType) {
1216         Preconditions.checkArgument(requiredType != null && !requiredType.isBottom());
1217         EnvTypePair resultPair = null;
1218         switch (expr.getType()) {
1219           case EMPTY: // can be created by a FOR with empty condition
1220             resultPair = new EnvTypePair(inEnv, JSType.UNKNOWN);
1221             break;
1222           case FUNCTION: {
1223             String fnName = symbolTable.getFunInternalName(expr);
1224             JSType fnType = envGetType(inEnv, fnName);
1225             Preconditions.checkState(fnType != null, "Could not find type for %s", fnName);
1226             TypeEnv outEnv = collectTypesForFreeVarsFwd(expr, inEnv);
1227             resultPair = new EnvTypePair(outEnv, fnType);
1228             break;
1229           }
1230           case FALSE:
1231           case NULL:
1232           case NUMBER:
1233           case STRING:
1234           case TRUE:
1235             resultPair = new EnvTypePair(inEnv, scalarValueToType(expr.getType()));
1236             break;
1237           case OBJECTLIT:
1238             resultPair = analyzeObjLitFwd(expr, inEnv, requiredType, specializedType);
1239             break;
1240           case THIS: {
1241             if (!currentScope.hasThis()) {
1242               mayWarnAboutGlobalThis(expr, currentScope);
1243               resultPair = new EnvTypePair(inEnv, JSType.UNKNOWN);
1244             } else {
1245               // A trimmed-down version of analyzeNameFwd.
1246               JSType inferredType = envGetType(inEnv, THIS_ID);
1247               if (!inferredType.isSubtypeOf(requiredType)) {
1248                 return new EnvTypePair(inEnv, inferredType);
1249               }
1250               JSType preciseType = inferredType.specialize(specializedType);
1251               resultPair = EnvTypePair.addBinding(inEnv, THIS_ID, preciseType);
1252             }
1253             break;
1254           }
1255           case NAME:
1256             resultPair = analyzeNameFwd(expr, inEnv, requiredType, specializedType);
1257             break;
1258           case AND:
1259           case OR:
1260             resultPair = analyzeLogicalOpFwd(expr, inEnv, requiredType, specializedType);
1261             break;
1262           case INC:
1263           case DEC:
1264             resultPair = analyzeIncDecFwd(expr, inEnv, requiredType);
1265             break;
1266           case BITNOT:
1267           case NEG:
1268             resultPair = analyzeUnaryNumFwd(expr, inEnv);
1269             break;
1270           case POS: {
1271             // We are more permissive with +, because it is used to coerce to number
1272             resultPair = analyzeExprFwd(expr.getFirstChild(), inEnv);
1273             resultPair.type = JSType.NUMBER;
1274             break;
1275           }
1276           case TYPEOF: {
1277             resultPair = analyzeExprFwd(expr.getFirstChild(), inEnv);
1278             resultPair.type = JSType.STRING;
1279             break;
1280           }
1281           case INSTANCEOF:
1282             resultPair = analyzeInstanceofFwd(expr, inEnv, specializedType);
1283             break;
1284           case ADD:
1285             resultPair = analyzeAddFwd(expr, inEnv, requiredType);
1286             break;
1287           case BITOR:
1288           case BITAND:
1289           case BITXOR:
1290           case DIV:
1291           case LSH:
1292           case MOD:
1293           case MUL:
1294           case RSH:
1295           case SUB:
1296           case URSH:
1297             resultPair = analyzeBinaryNumericOpFwd(expr, inEnv);
1298             break;
1299           case ASSIGN:
1300             resultPair = analyzeAssignFwd(expr, inEnv, requiredType, specializedType);
1301             break;
1302           case ASSIGN_ADD:
1303             resultPair = analyzeAssignAddFwd(expr, inEnv, requiredType);
1304             break;
1305           case ASSIGN_BITOR:
1306           case ASSIGN_BITXOR:
1307           case ASSIGN_BITAND:
1308           case ASSIGN_LSH:
1309           case ASSIGN_RSH:
1310           case ASSIGN_URSH:
1311           case ASSIGN_SUB:
1312           case ASSIGN_MUL:
1313           case ASSIGN_DIV:
1314           case ASSIGN_MOD:
1315             resultPair = analyzeAssignNumericOpFwd(expr, inEnv);
1316             break;
1317           case SHEQ:
1318           case SHNE:
1319             resultPair = analyzeStrictComparisonFwd(expr.getType(),
1320                 expr.getFirstChild(), expr.getLastChild(), inEnv, specializedType);
1321             break;
1322           case EQ:
1323           case NE:
1324             resultPair = analyzeNonStrictComparisonFwd(expr, inEnv, specializedType);
1325             break;
1326           case LT:
1327           case GT:
1328           case LE:
1329           case GE:
1330             resultPair = analyzeLtGtFwd(expr, inEnv);
1331             break;
1332           case GETPROP:
1333             Preconditions.checkState(
1334                 !NodeUtil.isAssignmentOp(expr.getParent()) ||
1335                 !NodeUtil.isLValue(expr));
1336             if (expr.getBooleanProp(Node.ANALYZED_DURING_GTI)) {
1337               expr.removeProp(Node.ANALYZED_DURING_GTI);
1338               resultPair = new EnvTypePair(inEnv, requiredType);
1339             } else {
1340               resultPair = analyzePropAccessFwd(
1341                   expr.getFirstChild(), expr.getLastChild().getString(),
1342                   inEnv, requiredType, specializedType);
1343             }
1344             break;
1345           case HOOK:
1346             resultPair = analyzeHookFwd(expr, inEnv, requiredType, specializedType);
1347             break;
1348           case CALL:
1349           case NEW:
1350             resultPair = analyzeCallNewFwd(expr, inEnv, requiredType, specializedType);
1351             break;
1352           case COMMA:
1353             resultPair = analyzeExprFwd(
1354                 expr.getLastChild(),
1355                 analyzeExprFwd(expr.getFirstChild(), inEnv).env,
1356                 requiredType,
1357                 specializedType);
1358             break;
1359           case NOT: {
1360             resultPair = analyzeExprFwd(expr.getFirstChild(),
1361                 inEnv, JSType.UNKNOWN, specializedType.negate());
1362             resultPair.type = resultPair.type.negate().toBoolean();
1363             break;
1364           }
1365           case GETELEM:
1366             resultPair = analyzeGetElemFwd(expr, inEnv, requiredType, specializedType);
1367             break;
1368           case VOID: {
1369             resultPair = analyzeExprFwd(expr.getFirstChild(), inEnv);
1370             resultPair.type = JSType.UNDEFINED;
1371             break;
1372           }
1373           case IN:
1374             resultPair = analyzeInFwd(expr, inEnv, specializedType);
1375             break;
1376           case DELPROP: {
1377             // IRFactory checks that the operand is a name, getprop or getelem.
1378             // No further warnings here.
1379             resultPair = analyzeExprFwd(expr.getFirstChild(), inEnv);
1380             resultPair.type = JSType.BOOLEAN;
1381             break;
1382           }
1383           case REGEXP:
1384             resultPair = new EnvTypePair(inEnv, commonTypes.getRegexpType());
1385             break;
1386           case ARRAYLIT:
1387             resultPair = analyzeArrayLitFwd(expr, inEnv);
1388             break;
1389           case CAST:
1390             resultPair = analyzeCastFwd(expr, inEnv);
1391             break;
1392           case CASE:
1393             // For a statement of the form: switch (exp1) { ... case exp2: ... }
1394             // we analyze the case as if it were (exp1 === exp2).
1395             // We analyze the body of the case when the test is true and the stm
1396             // following the body when the test is false.
1397             resultPair = analyzeStrictComparisonFwd(Token.SHEQ,
1398                 expr.getParent().getFirstChild(), expr.getFirstChild(),
1399                 inEnv, specializedType);
1400             break;
1401           default:
1402             throw new RuntimeException("Unhandled expression type: " + expr.getType());
1403         }
1404         mayWarnAboutUnknownType(expr, resultPair.type);
1405         // TODO(dimvar): We attach the type to every expression because that's also
1406         // what the old type inference does.
1407         // But maybe most of these types are never looked at.
1408         // See if the passes that use types only need types on a small subset of the
1409         // nodes, and only store these types to save mem.
1410         maybeSetTypeI(expr, resultPair.type);
1411         if (this.currentScope.isFunction()) {
1412           // In global scope, the env is too big and produces too much output
1413           println("AnalyzeExprFWD: ", expr,
1414               " ::reqtype: ", requiredType, " ::spectype: ", specializedType,
1415               " ::resulttype: ", resultPair.type);
1416         }
1417         return resultPair;
1418       }

1420       private void mayWarnAboutUnknownType(Node expr, JSType t) {
1421         boolean isKnownGetElem = expr.isGetElem() && expr.getLastChild().isString();
1422         if (t.isUnknown()
1423             && this.reportUnknownTypes
1424             // Don't warn for expressions whose value isn't used
1425             && !expr.getParent().isExprResult()
1426             // The old type checker doesn't warn about unknown getelems.
1427             // Maybe because we can't do anything about them, so why warn?
1428             // We mimic that behavior here.
1429             && (!expr.isGetElem() || isKnownGetElem)) {
1430           warnings.add(JSError.make(expr, UNKNOWN_EXPR_TYPE, expr.getType().toString()));
1431         }
1432       }

1434       private EnvTypePair analyzeNameFwd(
1435           Node expr, TypeEnv inEnv, JSType requiredType, JSType specializedType) {
1436         String varName = expr.getString();
1437         if (varName.equals("undefined")) {
1438           return new EnvTypePair(inEnv, JSType.UNDEFINED);
1439         }
1440         JSType inferredType = envGetType(inEnv, varName);
1441         if (inferredType == null) {
1442           println("Found global variable ", varName);
1443           // For now, we don't warn for global variables
1444           return new EnvTypePair(inEnv, JSType.UNKNOWN);
1445         }
1446         println(varName, "'s inferredType: ", inferredType,
1447             " requiredType:  ", requiredType,
1448             " specializedType:  ", specializedType);
1449         if (!inferredType.isSubtypeOf(requiredType)) {
1450           // The inferred type of a variable is always an upper bound, but
1451           // sometimes it's also a lower bound, eg, if x was the lhs of an =
1452           // where we know the type of the rhs.
1453           // We don't track whether the inferred type is a lower bound, so we
1454           // conservatively assume that it always is.
1455           // This is why we warn when !inferredType.isSubtypeOf(requiredType).
1456           // In some rare cases, the inferred type is only an upper bound,
1457           // and we would falsely warn.
1458           // (These usually include the polymorphic operators += and <.)
1459           // We have a heuristic check to avoid the spurious warnings,
1460           // but we also miss some true warnings.
1461           JSType declType = currentScope.getDeclaredTypeOf(varName);
1462           if (tightenTypeAndDontWarn(varName, expr, declType, inferredType, requiredType)) {
1463             inferredType = inferredType.specialize(requiredType);
1464           } else {
1465             // Propagate incorrect type so that the context catches
1466             // the mismatch
1467             return new EnvTypePair(inEnv, inferredType);
1468           }
1469         }
1470         // If preciseType is bottom, there is a condition that can't be true,
1471         // but that's not necessarily a type error.
1472         JSType preciseType = inferredType.specialize(specializedType);
1473         println(varName, "'s preciseType: ", preciseType);
1474         if (!preciseType.isBottom()
1475             && (currentScope.isUndeclaredFormal(varName)
1476                 || currentScope.isUndeclaredOuterVar(varName))
1477             && preciseType.hasNonScalar()) {
1478           // In the bwd direction, we may infer a loose type and then join w/
1479           // top and forget it. That's why we also loosen types going fwd.
1480           preciseType = preciseType.withLoose();
1481         }
1482         return EnvTypePair.addBinding(inEnv, varName, preciseType);
1483       }

1485       private EnvTypePair analyzeLogicalOpFwd(
1486           Node expr, TypeEnv inEnv, JSType requiredType, JSType specializedType) {
1487         Token exprKind = expr.getType();
1488         Node lhs = expr.getFirstChild();
1489         Node rhs = expr.getLastChild();
1490         if ((specializedType.isTrueOrTruthy() && exprKind == Token.AND)
1491             || (specializedType.isFalseOrFalsy() && exprKind == Token.OR)) {
1492           EnvTypePair lhsPair =
1493               analyzeExprFwd(lhs, inEnv, JSType.UNKNOWN, specializedType);
1494           EnvTypePair rhsPair =
1495               analyzeExprFwd(rhs, lhsPair.env, JSType.UNKNOWN, specializedType);
1496           return rhsPair;
1497         } else if ((specializedType.isFalseOrFalsy() && exprKind == Token.AND)
1498             || (specializedType.isTrueOrTruthy() && exprKind == Token.OR)) {
1499           EnvTypePair shortCircuitPair =
1500               analyzeExprFwd(lhs, inEnv, JSType.UNKNOWN, specializedType);
1501           EnvTypePair lhsPair = analyzeExprFwd(
1502               lhs, inEnv, JSType.UNKNOWN, specializedType.negate());
1503           EnvTypePair rhsPair =
1504               analyzeExprFwd(rhs, lhsPair.env, JSType.UNKNOWN, specializedType);
1505           JSType lhsUnspecializedType = JSType.join(shortCircuitPair.type, lhsPair.type);
1506           return combineLhsAndRhsForLogicalOps(
1507               exprKind, lhsUnspecializedType, shortCircuitPair, rhsPair);
1508         } else {
1509           // Independently of the specializedType, && rhs is only analyzed when
1510           // lhs is truthy, and || rhs is only analyzed when lhs is falsy.
1511           JSType stopAfterLhsType = exprKind == Token.AND ? JSType.FALSY : JSType.TRUTHY;
1512           EnvTypePair shortCircuitPair =
1513               analyzeExprFwd(lhs, inEnv, JSType.UNKNOWN, stopAfterLhsType);
1514           EnvTypePair lhsPair = analyzeExprFwd(
1515               lhs, inEnv, JSType.UNKNOWN, stopAfterLhsType.negate());
1516           EnvTypePair rhsPair =
1517               analyzeExprFwd(rhs, lhsPair.env, requiredType, specializedType);
1518           JSType lhsUnspecializedType = JSType.join(shortCircuitPair.type, lhsPair.type);
1519           return combineLhsAndRhsForLogicalOps(
1520               exprKind, lhsUnspecializedType, shortCircuitPair, rhsPair);
1521         }
1522       }
1523     
1524       private EnvTypePair combineLhsAndRhsForLogicalOps(Token logicalOp,
1525           JSType lhsUnspecializedType, EnvTypePair lhsPair, EnvTypePair rhsPair) {
1526         if (logicalOp == Token.OR) {
1527           if (lhsUnspecializedType.isAnyTruthyType()) {
1528             return lhsPair;
1529           }
1530           if (lhsUnspecializedType.isAnyFalsyType()) {
1531             return rhsPair;
1532           }
1533           lhsPair.type = lhsPair.type.specialize(JSType.TRUTHY);
1534           return EnvTypePair.join(lhsPair, rhsPair);
1535         }
1536         Preconditions.checkState(logicalOp == Token.AND);
1537         if (lhsUnspecializedType.isAnyFalsyType()) {
1538           return lhsPair;
1539         }
1540         if (lhsUnspecializedType.isAnyTruthyType()) {
1541           return rhsPair;
1542         }
1543         lhsPair.type = lhsPair.type.specialize(JSType.FALSY);
1544         return EnvTypePair.join(lhsPair, rhsPair);
1545       }

1547       private EnvTypePair analyzeIncDecFwd(
1548           Node expr, TypeEnv inEnv, JSType requiredType) {
1549         mayWarnAboutConst(expr);
1550         Node ch = expr.getFirstChild();
1551         if (ch.isGetProp() || ch.isGetElem() && ch.getLastChild().isString()) {
1552           // We prefer to analyze the child of INC/DEC one extra time here,
1553           // to putting the @const prop check in analyzePropAccessFwd.
1554           Node recv = ch.getFirstChild();
1555           String pname = ch.getLastChild().getString();
1556           EnvTypePair pair = analyzeExprFwd(recv, inEnv);
1557           JSType recvType = pair.type;
1558           if (mayWarnAboutConstProp(ch, recvType, new QualifiedName(pname))) {
1559             pair.type = requiredType;
1560             return pair;
1561           }
1562         }
1563         return analyzeUnaryNumFwd(expr, inEnv);
1564       }

1566       private EnvTypePair analyzeUnaryNumFwd(Node expr, TypeEnv inEnv) {
1567         // For inc and dec on a getprop, we don't want to create a property on
1568         // a struct by accident.
1569         // But we will get an inexistent-property warning, so we don't check
1570         // for structness separately here.
1571         Node child = expr.getFirstChild();
1572         EnvTypePair pair = analyzeExprFwd(child, inEnv, JSType.NUMBER);
1573         if (!commonTypes.isNumberScalarOrObj(pair.type)) {
1574           warnInvalidOperand(child, expr.getType(), JSType.NUMBER, pair.type);
1575         }
1576         pair.type = JSType.NUMBER;
1577         return pair;
1578       }

1580       private EnvTypePair analyzeInstanceofFwd(
1581           Node expr, TypeEnv inEnv, JSType specializedType) {
1582         Node obj = expr.getFirstChild();
1583         Node ctor = expr.getLastChild();
1584         EnvTypePair objPair, ctorPair;
1585     
1586         // First, evaluate ignoring the specialized context
1587         objPair = analyzeExprFwd(obj, inEnv);
1588         JSType objType = objPair.type;
1589         if (!objType.isTop()
1590             && !objType.isUnknown()
1591             && !objType.hasNonScalar()
1592             && !objType.hasTypeVariable()) {
1593           warnInvalidOperand(
1594               obj, Token.INSTANCEOF,
1595               "an object or a union type that includes an object",
1596               objPair.type);
1597         }
1598         ctorPair = analyzeExprFwd(ctor, objPair.env, commonTypes.topFunction());
1599         JSType ctorType = ctorPair.type;
1600         FunctionType ctorFunType = ctorType.getFunType();
1601         boolean mayBeConstructorFunction = ctorFunType != null
1602             && (ctorFunType.isLoose()
1603                 || ctorFunType.isQmarkFunction()
1604                 || ctorFunType.isSomeConstructorOrInterface());
1605         if (!(ctorType.isUnknown() || mayBeConstructorFunction)) {
1606           warnInvalidOperand(
1607               ctor, Token.INSTANCEOF, "a constructor function", ctorType);
1608         }
1609         if (ctorFunType == null
1610             || !ctorFunType.isUniqueConstructor()
1611             || (!specializedType.isTrueOrTruthy()
1612                 && !specializedType.isFalseOrFalsy())) {
1613           ctorPair.type = JSType.BOOLEAN;
1614           return ctorPair;
1615         }
1616     
1617         // We are in a specialized context *and* we know the constructor type
1618         JSType instanceType = ctorFunType.getInstanceTypeOfCtor();
1619         JSType instanceSpecType = specializedType.isTrueOrTruthy()
1620             ? objType.specialize(instanceType)
1621             : objType.removeType(instanceType);
1622         if (!instanceSpecType.isBottom()) {
1623           objPair = analyzeExprFwd(obj, inEnv, JSType.UNKNOWN, instanceSpecType);
1624           ctorPair = analyzeExprFwd(ctor, objPair.env, commonTypes.topFunction());
1625         }
1626         ctorPair.type = JSType.BOOLEAN;
1627         return ctorPair;
1628       }

1630       private EnvTypePair analyzeAddFwd(Node expr, TypeEnv inEnv, JSType requiredType) {
1631         Node lhs = expr.getFirstChild();
1632         Node rhs = expr.getLastChild();
1633         JSType operandType = requiredType.isNumber() ? JSType.NUMBER : JSType.UNKNOWN;
1634         EnvTypePair lhsPair = analyzeExprFwd(lhs, inEnv, operandType);
1635         EnvTypePair rhsPair = analyzeExprFwd(rhs, lhsPair.env, operandType);
1636         JSType lhsType = lhsPair.type;
1637         JSType rhsType = rhsPair.type;
1638         if (lhsType.isString() || rhsType.isString()) {
1639           // Return early and don't warn, since '' + expr is used for type coercions
1640           rhsPair.type = JSType.STRING;
1641           return rhsPair;
1642         }
1643         if (!commonTypes.isNumStrScalarOrObj(lhsType)) {
1644           warnInvalidOperand(lhs, expr.getType(), JSType.NUM_OR_STR, lhsType);
1645         }
1646         if (!commonTypes.isNumStrScalarOrObj(rhsType)) {
1647           warnInvalidOperand(rhs, expr.getType(), JSType.NUM_OR_STR, rhsType);
1648         }
1649         return new EnvTypePair(rhsPair.env, JSType.plus(lhsType, rhsType));
1650       }

1653         Node lhs = expr.getFirstChild();
1654         Node rhs = expr.getLastChild();
1655         EnvTypePair lhsPair = analyzeExprFwd(lhs, inEnv, JSType.NUMBER);

1660         if (!commonTypes.isNumberScalarOrObj(rhsPair.type)) {
1661           warnInvalidOperand(rhs, expr.getType(), JSType.NUMBER, rhsPair.type);
1662         }
1663         rhsPair.type = JSType.NUMBER;
1664         return rhsPair;

1667       private EnvTypePair analyzeAssignFwd(
1668           Node expr, TypeEnv inEnv, JSType requiredType, JSType specializedType) {
1669         if (expr.getBooleanProp(Node.ANALYZED_DURING_GTI)) {
1670           expr.removeProp(Node.ANALYZED_DURING_GTI);
1671           // No need to set a type on the assignment expression
1672           return new EnvTypePair(inEnv, requiredType);
1673         }
1674         mayWarnAboutConst(expr);
1675         Node lhs = expr.getFirstChild();
1676         Node rhs = expr.getLastChild();
1677         if (lhs.getBooleanProp(Node.ANALYZED_DURING_GTI)) {
1678           lhs.removeProp(Node.ANALYZED_DURING_GTI);
1679           JSType declType = markAndGetTypeOfPreanalyzedNode(lhs, inEnv, true);
1680           if (rhs.matchesQualifiedName(ABSTRACT_METHOD_NAME)) {
1681             return new EnvTypePair(inEnv, requiredType);
1682           }
1683           EnvTypePair rhsPair = analyzeExprFwd(rhs, inEnv, declType);
1684           if (!rhsPair.type.isSubtypeOf(declType)
1685               && !NodeUtil.isPrototypeAssignment(lhs)) {
1686             warnings.add(JSError.make(expr, MISTYPED_ASSIGN_RHS,
1687                     errorMsgWithTypeDiff(declType, rhsPair.type)));
1688           }
1689           return rhsPair;
1690         }
1691         LValueResultFwd lvalue = analyzeLValueFwd(lhs, inEnv, requiredType);
1692         JSType declType = lvalue.declType;
1693         EnvTypePair rhsPair =
1694             analyzeExprFwd(rhs, lvalue.env, requiredType, specializedType);
1695         if (declType != null && !rhsPair.type.isSubtypeOf(declType)) {
1696           warnings.add(JSError.make(expr, MISTYPED_ASSIGN_RHS,
1697                   errorMsgWithTypeDiff(declType, rhsPair.type)));
1698         } else {
1699           rhsPair.env = updateLvalueTypeInEnv(rhsPair.env, lhs, lvalue.ptr, rhsPair.type);
1700         }
1701         return rhsPair;
1702       }

1704       private EnvTypePair analyzeAssignAddFwd(
1705           Node expr, TypeEnv inEnv, JSType requiredType) {
1706         mayWarnAboutConst(expr);
1707         Node lhs = expr.getFirstChild();
1708         Node rhs = expr.getLastChild();
1709         JSType lhsReqType =
1710             specializeKeep2ndWhenBottom(requiredType, JSType.NUM_OR_STR);
1711         LValueResultFwd lvalue = analyzeLValueFwd(lhs, inEnv, lhsReqType);
1712         JSType lhsType = lvalue.type;
1713         if (!lhsType.isSubtypeOf(JSType.NUM_OR_STR)) {
1714           warnInvalidOperand(lhs, Token.ASSIGN_ADD, JSType.NUM_OR_STR, lhsType);
1715         }
1716         // if lhs is a string, rhs can still be a number
1717         JSType rhsReqType = lhsType.isNumber() ? JSType.NUMBER : JSType.NUM_OR_STR;
1718         EnvTypePair pair = analyzeExprFwd(rhs, lvalue.env, rhsReqType);
1719         if (!pair.type.isSubtypeOf(rhsReqType)) {
1720           warnInvalidOperand(rhs, Token.ASSIGN_ADD, rhsReqType, pair.type);
1721         }
1722         return pair;
1723       }

1786       private EnvTypePair analyzeCallNewFwd(
1787           Node expr, TypeEnv inEnv, JSType requiredType, JSType specializedType) {
1788         if (isPropertyTestCall(expr)) {
1789           return analyzePropertyTestCallFwd(expr, inEnv, specializedType);
1790         }
1791         Node callee = expr.getFirstChild();
1792         if (isFunctionBind(callee, inEnv, true)) {
1793           return analyzeFunctionBindFwd(expr, inEnv);
1794         }
1795         AssertionFunctionSpec assertionFunctionSpec =
1796             assertionFunctionsMap.get(callee.getQualifiedName());
1797         if (assertionFunctionSpec != null) {
1798           return analyzeAssertionCall(expr, inEnv, assertionFunctionSpec);
1799         }
1800         EnvTypePair calleePair = analyzeExprFwd(callee, inEnv, commonTypes.topFunction());
1801         TypeEnv envAfterCallee = calleePair.env;
1802         calleePair = mayWarnAboutNullableReferenceAndTighten(
1803             callee, calleePair.type, null, envAfterCallee);
1804         JSType calleeType = calleePair.type;
1805         if (calleeType.isBottom() || !calleeType.isSubtypeOf(commonTypes.topFunction())) {
1806           warnings.add(JSError.make(
1807               expr, NOT_CALLABLE, calleeType.toString()));
1808         }
1809         FunctionType funType = calleeType.getFunTypeIfSingletonObj();
1810         if (funType == null
1811             || funType.isTopFunction() || funType.isQmarkFunction()) {
1812           return analyzeCallNodeArgsFwdWhenError(expr, envAfterCallee);
1813         } else if (funType.isLoose()) {
1814           return analyzeLooseCallNodeFwd(expr, envAfterCallee, requiredType);
1815         } else if (expr.isCall()
1816             && funType.isSomeConstructorOrInterface()
1817             && (funType.getReturnType().isUnknown()
1818                 || funType.getReturnType().isUndefined())) {
1819           warnings.add(JSError.make(expr, CONSTRUCTOR_NOT_CALLABLE, funType.toString()));
1820           return analyzeCallNodeArgsFwdWhenError(expr, envAfterCallee);
1821         } else if (expr.isNew()
1822             && (!funType.isSomeConstructorOrInterface()
1823                 || funType.isInterfaceDefinition())) {
1824           warnings.add(JSError.make(expr, NOT_A_CONSTRUCTOR, funType.toString()));
1825           return analyzeCallNodeArgsFwdWhenError(expr, envAfterCallee);
1826         }
1827         int maxArity = funType.getMaxArity();
1828         int minArity = funType.getMinArity();
1829         int numArgs = expr.getChildCount() - 1;
1830         if (numArgs < minArity || numArgs > maxArity) {
1831           warnings.add(JSError.make(
1832               expr, WRONG_ARGUMENT_COUNT,
1833               getReadableCalleeName(callee),
1834               Integer.toString(numArgs), Integer.toString(minArity),
1835               " and at most " + maxArity));
1836           return analyzeCallNodeArgsFwdWhenError(expr, envAfterCallee);
1837         }
1838         FunctionType origFunType = funType; // save for later
1839         if (funType.isGeneric()) {
1840           Map<String, JSType> typeMap = calcTypeInstantiationFwd(
1841               expr,
1842               callee.isGetProp() ? callee.getFirstChild() : null,
1843               expr.getSecondChild(), funType, envAfterCallee);
1844           funType = funType.instantiateGenerics(typeMap);
1845           println("Instantiated function type: " + funType);
1846         }
1847         // argTypes collects types of actuals for deferred checks.
1848         List<JSType> argTypes = new ArrayList<>();
1849         TypeEnv tmpEnv = analyzeCallNodeArgumentsFwd(
1850             expr, expr.getSecondChild(), funType, argTypes, envAfterCallee);
1851         if (callee.isName()) {
1852           String calleeName = callee.getString();
1853           if (currentScope.isKnownFunction(calleeName)
1854               && !currentScope.isExternalFunction(calleeName)) {
1855             // Local function definitions will be type-checked more
1856             // exactly using their summaries, and don't need deferred checks
1857             if (currentScope.isLocalFunDef(calleeName)) {
1858               tmpEnv = collectTypesForFreeVarsFwd(callee, tmpEnv);
1859             } else if (!origFunType.isGeneric()) {
1860               JSType expectedRetType = requiredType;
1861               println("Updating deferred check with ret: ", expectedRetType,
1862                   " and args: ", argTypes);
1863               DeferredCheck dc;
1864               if (expr.isCall()) {
1865                 dc = deferredChecks.get(expr);
1866                 if (dc != null) {
1867                   dc.updateReturn(expectedRetType);
1868                 } else {
1869                   // The backward analysis of a function is skipped when all
1870                   // variables, including outer vars, are declared.
1871                   // So, we check that dc is null iff bwd was skipped.
1872                   Preconditions.checkState(
1873                       !currentScope.hasUndeclaredFormalsOrOuters(),
1874                       "No deferred check created in backward direction for %s",
1875                       expr);
1876                 }
1877               } else { // call to constructor
1878                 dc = new DeferredCheck(expr, null,
1879                     currentScope, currentScope.getScope(calleeName));
1880                 deferredChecks.put(expr, dc);
1881               }
1882               if (dc != null) {
1883                 dc.updateArgTypes(argTypes);
1884               }
1885             }
1886           }
1887         }
1888         JSType retType = expr.isNew() ? funType.getThisType() : funType.getReturnType();
1889         if (retType.isSubtypeOf(requiredType)) {
1890           retType = retType.specialize(specializedType);
1891         }
1892         return new EnvTypePair(tmpEnv, retType);
1893       }

1895       private EnvTypePair analyzeFunctionBindFwd(Node call, TypeEnv inEnv) {
1896         Preconditions.checkArgument(call.isCall());
1897         Bind bindComponents = this.convention.describeFunctionBind(call, true, false);
1898         Node boundFunNode = bindComponents.target;
1899         EnvTypePair pair = analyzeExprFwd(boundFunNode, inEnv);
1900         TypeEnv env = pair.env;
1901         FunctionType boundFunType = pair.type.getFunTypeIfSingletonObj();
1902         if (!pair.type.isSubtypeOf(commonTypes.topFunction())) {
1903           warnings.add(JSError.make(boundFunNode, GOOG_BIND_EXPECTS_FUNCTION, pair.type.toString()));
1904         }
1905         // For some function types, we don't know enough to handle .bind specially.
1906         if (boundFunType == null
1907             || boundFunType.isTopFunction()
1908             || boundFunType.isQmarkFunction()
1909             || boundFunType.isLoose()) {
1910           return analyzeCallNodeArgsFwdWhenError(call, env);
1911         }
1912         if (boundFunType.isSomeConstructorOrInterface()) {
1913           warnings.add(JSError.make(call, CANNOT_BIND_CTOR));
1914           return new EnvTypePair(env, JSType.UNKNOWN);
1915         }
1916         // Check if the receiver argument is there
1917         if (NodeUtil.isGoogBind(call.getFirstChild()) && call.getChildCount() <= 2
1918             || !NodeUtil.isGoogPartial(call.getFirstChild()) && call.getChildCount() == 1) {
1919           warnings.add(JSError.make(
1920               call, WRONG_ARGUMENT_COUNT,
1921               getReadableCalleeName(call.getFirstChild()),
1922               "0", "1", ""));
1923         }
1924         // Check that there are not too many of the other arguments
1925         int maxArity = boundFunType.hasRestFormals()
1926             ? Integer.MAX_VALUE : boundFunType.getMaxArity();
1927         int numArgs = bindComponents.getBoundParameterCount();
1928         if (numArgs > maxArity) {
1929           warnings.add(JSError.make(
1930               call, WRONG_ARGUMENT_COUNT,
1931               getReadableCalleeName(call.getFirstChild()),
1932               Integer.toString(numArgs), "0",
1933               " and at most " + maxArity));
1934           return analyzeCallNodeArgsFwdWhenError(call, inEnv);
1935         }
1936     
1937         // If the bound function is polymorphic, we only support the case where we
1938         // can completely calculate the type instantiation at the .bind call site.
1939         // We don't support splitting the instantiation between call sites.
1940         //
1941         Node receiver = bindComponents.thisValue;
1942         if (boundFunType.isGeneric()) {
1943           Map<String, JSType> typeMap = calcTypeInstantiationFwd(
1944               call, receiver, bindComponents.parameters, boundFunType, env);
1945           boundFunType = boundFunType.instantiateGenerics(typeMap);
1946         }
1947         FunctionTypeBuilder builder = new FunctionTypeBuilder();
1948         if (receiver != null) {// receiver is null for goog.partial
1949           JSType reqThisType = boundFunType.getThisType();
1950           if (reqThisType == null || boundFunType.isSomeConstructorOrInterface()) {
1951             reqThisType = JSType.join(JSType.NULL, JSType.TOP_OBJECT);
1952           }
1953           pair = analyzeExprFwd(receiver, env, reqThisType);
1954           env = pair.env;
1955           if (!pair.type.isSubtypeOf(reqThisType)) {
1956             warnings.add(JSError.make(call, INVALID_THIS_TYPE_IN_BIND,
1957                     errorMsgWithTypeDiff(reqThisType, pair.type)));
1958           }
1959         }
1960     
1961         // We're passing an arraylist but don't do deferred checks for bind.
1962         env = analyzeCallNodeArgumentsFwd(call, bindComponents.parameters,
1963             boundFunType, new ArrayList<JSType>(), env);
1964         // For any formal not bound here, add it to the resulting function type.
1965         for (int j = numArgs; j < boundFunType.getMaxArityWithoutRestFormals(); j++) {
1966           JSType formalType = boundFunType.getFormalType(j);
1967           if (boundFunType.isRequiredArg(j)) {
1968             builder.addReqFormal(formalType);
1969           } else {
1970             builder.addOptFormal(formalType);
1971           }
1972         }
1973         if (boundFunType.hasRestFormals()) {
1974           builder.addRestFormals(boundFunType.getRestFormalsType());
1975         }
1976         return new EnvTypePair(env, commonTypes.fromFunctionType(
1977             builder.addRetType(boundFunType.getReturnType()).buildFunction()));
1978       }

1980       private TypeEnv analyzeCallNodeArgumentsFwd(Node call, Node firstArg,
1981           FunctionType funType, List<JSType> argTypesForDeferredCheck,
1982           TypeEnv inEnv) {
1983         TypeEnv env = inEnv;
1984         Node arg = firstArg;
1985         int i = 0;
1986         while (arg != null) {
1987           JSType formalType = funType.getFormalType(i);
1988           Preconditions.checkState(!formalType.isBottom());
1989           EnvTypePair pair = analyzeExprFwd(arg, env, formalType);
1990           JSType argTypeForDeferredCheck = pair.type;
1991           // Allow passing undefined for an optional argument.
1992           if (funType.isOptionalArg(i) && pair.type.equals(JSType.UNDEFINED)) {
1993             argTypeForDeferredCheck = null; // No deferred check needed.
1994           } else if (!pair.type.isSubtypeOf(formalType)) {
1995             String fnName = getReadableCalleeName(call.getFirstChild());
1996             warnings.add(JSError.make(arg, INVALID_ARGUMENT_TYPE,
1997                     Integer.toString(i + 1), fnName,
1998                     errorMsgWithTypeDiff(formalType, pair.type)));
1999             argTypeForDeferredCheck = null; // No deferred check needed.
2000           }
2001           argTypesForDeferredCheck.add(argTypeForDeferredCheck);
2002           env = pair.env;
2003           arg = arg.getNext();
2004           i++;
2005         }
2006         return env;
2007       }

2009       private EnvTypePair analyzeAssertionCall(
2010           Node callNode, TypeEnv env, AssertionFunctionSpec assertionFunctionSpec) {
2011         Node firstParam = callNode.getSecondChild();
2012         if (firstParam == null) {
2013           return new EnvTypePair(env, JSType.UNKNOWN);
2014         }
2015         Node assertedNode = assertionFunctionSpec.getAssertedParam(firstParam);
2016         if (assertedNode == null) {
2017           return new EnvTypePair(env, JSType.UNKNOWN);
2018         }
2019         JSType assertedType = assertionFunctionSpec.getAssertedNewType(callNode, currentScope);
2020         if (assertedType.isUnknown()) {
2021           warnings.add(JSError.make(callNode, NewTypeInference.UNKNOWN_ASSERTION_TYPE));
2022         }
2023         EnvTypePair pair = analyzeExprFwd(assertedNode, env, JSType.UNKNOWN, assertedType);
2024         if (!pair.type.isSubtypeOf(assertedType)
2025             && JSType.haveCommonSubtype(assertedType, pair.type)) {
2026           // We do this because the assertion needs to return a subtype of the
2027           // asserted type to its context, but sometimes the asserted expression
2028           // can't be specialized.
2029           pair.type = assertedType;
2030         }
2031         if (pair.type.isBottom()) {
2032           JSType t = analyzeExprFwd(assertedNode, env).type.substituteGenericsWithUnknown();
2033           if (t.isSubtypeOf(assertedType)) {
2034             pair.type = t;
2035           } else {
2036             warnings.add(JSError.make(assertedNode, NewTypeInference.ASSERT_FALSE));
2037             pair.type = JSType.UNKNOWN;
2038             pair.env = env;
2039           }
2040         }
2041         return pair;
2042       }

2071       private EnvTypePair analyzeInFwd(
2072           Node expr, TypeEnv inEnv, JSType specializedType) {
2073         Node lhs = expr.getFirstChild();
2074         Node rhs = expr.getLastChild();
2075         JSType reqObjType = pickReqObjType(expr);
2076         EnvTypePair pair;
2077     
2078         pair = analyzeExprFwd(lhs, inEnv, JSType.NUM_OR_STR);
2079         if (!pair.type.isSubtypeOf(JSType.NUM_OR_STR)) {
2080           warnInvalidOperand(lhs, Token.IN, JSType.NUM_OR_STR, pair.type);
2081         }
2082         pair = analyzeExprFwd(rhs, pair.env, reqObjType);
2083         if (!pair.type.isSubtypeOf(JSType.TOP_OBJECT)) {
2084           warnInvalidOperand(rhs, Token.IN, "Object", pair.type);
2085           pair.type = JSType.BOOLEAN;
2086           return pair;
2087         }
2088         if (pair.type.isStruct()) {
2089           warnings.add(JSError.make(rhs, IN_USED_WITH_STRUCT));
2090           pair.type = JSType.BOOLEAN;
2091           return pair;
2092         }
2093     
2094         JSType resultType = JSType.BOOLEAN;
2095         if (lhs.isString()) {
2096           QualifiedName pname = new QualifiedName(lhs.getString());
2097           if (specializedType.isTrueOrTruthy()) {
2098             pair = analyzeExprFwd(rhs, inEnv, reqObjType,
2099                 reqObjType.withPropertyRequired(pname.getLeftmostName()));
2100             resultType = JSType.TRUE_TYPE;
2101           } else if (specializedType.isFalseOrFalsy()) {
2102             pair = analyzeExprFwd(rhs, inEnv, reqObjType);
2103             // If the rhs is a loose object, we won't warn about missing
2104             // properties, despite removing the type here.
2105             // The only way to have that warning would be to keep track of props
2106             // that a loose object *cannot* have; but the implementation cost
2107             // is probably not worth it.
2108             pair = analyzeExprFwd(
2109                 rhs, inEnv, reqObjType, pair.type.withoutProperty(pname));
2110             resultType = JSType.FALSE_TYPE;
2111           }
2112         }
2113         pair.type = resultType;
2114         return pair;
2115       }

2156       private EnvTypePair analyzeStrictComparisonFwd(Token comparisonOp,
2157           Node lhs, Node rhs, TypeEnv inEnv, JSType specializedType) {
2158         if (specializedType.isTrueOrTruthy() || specializedType.isFalseOrFalsy()) {
2159           if (lhs.isTypeOf()) {
2160             return analyzeSpecializedTypeof(
2161                 lhs, rhs, comparisonOp, inEnv, specializedType);
2162           } else if (rhs.isTypeOf()) {
2163             return analyzeSpecializedTypeof(
2164                 rhs, lhs, comparisonOp, inEnv, specializedType);
2165           } else if (isGoogTypeof(lhs)) {
2166             return analyzeGoogTypeof(lhs, rhs, inEnv, specializedType);
2167           } else if (isGoogTypeof(rhs)) {
2168             return analyzeGoogTypeof(rhs, lhs, inEnv, specializedType);
2169           }
2170         }
2171     
2172         EnvTypePair lhsPair = analyzeExprFwd(lhs, inEnv);
2173         EnvTypePair rhsPair = analyzeExprFwd(rhs, lhsPair.env);
2174         // This env may contain types that have been tightened after nullable deref.
2175         TypeEnv preciseEnv = rhsPair.env;
2176     
2177         if ((comparisonOp == Token.SHEQ && specializedType.isTrueOrTruthy())
2178             || (comparisonOp == Token.SHNE && specializedType.isFalseOrFalsy())) {
2179           lhsPair = analyzeExprFwd(lhs, preciseEnv,
2180               JSType.UNKNOWN, lhsPair.type.specialize(rhsPair.type));
2181           rhsPair = analyzeExprFwd(rhs, lhsPair.env,
2182               JSType.UNKNOWN, rhsPair.type.specialize(lhsPair.type));
2183         } else if ((comparisonOp == Token.SHEQ && specializedType.isFalseOrFalsy()) ||
2184             (comparisonOp == Token.SHNE && specializedType.isTrueOrTruthy())) {
2185           JSType lhsType = lhsPair.type;
2186           JSType rhsType = rhsPair.type;
2187           if (lhsType.isNullOrUndef()) {
2188             rhsType = rhsType.removeType(lhsType);
2189           } else if (rhsType.isNullOrUndef()) {
2190             lhsType = lhsType.removeType(rhsType);
2191           }
2192           lhsPair = analyzeExprFwd(lhs, preciseEnv, JSType.UNKNOWN, lhsType);
2193           rhsPair = analyzeExprFwd(rhs, lhsPair.env, JSType.UNKNOWN, rhsType);
2194         }
2195         rhsPair.type = JSType.BOOLEAN;
2196         return rhsPair;
2197       }

2199       private EnvTypePair analyzeSpecializedTypeof(Node typeof, Node typeString,
2200           Token comparisonOp, TypeEnv inEnv, JSType specializedType) {
2201         EnvTypePair pair;
2202         Node typeofRand = typeof.getFirstChild();
2203         JSType comparedType = getTypeFromString(typeString);
2204         checkInvalidTypename(typeString);
2205         if (comparedType.isUnknown()) {
2206           pair = analyzeExprFwd(typeofRand, inEnv);
2207           pair = analyzeExprFwd(typeString, pair.env);
2208         } else if ((specializedType.isTrueOrTruthy()
2209                 && (comparisonOp == Token.SHEQ || comparisonOp == Token.EQ))
2210             || (specializedType.isFalseOrFalsy()
2211                 && (comparisonOp == Token.SHNE || comparisonOp == Token.NE))) {
2212           pair = analyzeExprFwd(typeofRand, inEnv, JSType.UNKNOWN, comparedType);
2213         } else {
2214           pair = analyzeExprFwd(typeofRand, inEnv);
2215           JSType rmType = pair.type.removeType(comparedType);
2216           if (!rmType.isBottom()) {
2217             pair = analyzeExprFwd(typeofRand, inEnv, JSType.UNKNOWN, rmType);
2218           }
2219         }
2220         pair.type = specializedType.toBoolean();
2221         return pair;
2222       }

2224       private JSType getTypeFromString(Node typeString) {
2225         if (!typeString.isString()) {
2226           return JSType.UNKNOWN;
2227         }
2228         switch (typeString.getString()) {
2229           case "number":
2230             return JSType.NUMBER;

2276       /**
2277        * We don't use the requiredType of the context to unify with the return
2278        * type. There are several difficulties:
2279        * 1) A polymorphic function is allowed to return ANY subtype of the
2280        *    requiredType, so we would need to use a heuristic to determine the type
2281        *    to unify with.
2282        * 2) It's hard to give good error messages in cases like: id('str') - 5
2283        *    We want an invalid-operand-type, not a not-unique-instantiation.
2284        *
2285        *
2286        * We don't take the arg evaluation order into account during instantiation.
2287        *
2288        *
2289        * When calculating the instantiation, when do we use the receiver type?
2290        * See the following snippet:
2291        * /**
2292        *  * @constructor
2293        *  * @template T
2294        *  * @param {T} x
2295        *  * /
2296        * function Foo(x) {}
2297        * /**
2298        *  * @template T
2299        *  * @param {T} x
2300        *  * /
2301        * Foo.prototype.f = function(x) {};
2302        * Foo.prototype.f.bind(new Foo(123), 'asdf');
2303        *
2304        * Here, the receiver type of f is Foo<T>, but the T is the class's T,
2305        * not the T of f's template declaration.
2306        * OTOH, if f had a @this annotation that contained T, T would refer to
2307        * f's T. There is no way of knowing what's the scope of the type variables
2308        * in the receiver of the function type.
2309        * But when THIS comes from the class, it is always a singleton object. So,
2310        * we use a heuristic: if THIS is not a singleton obj, we know it comes from
2311        * @this, and we use it for the instantiation.
2312        */
2313       private ImmutableMap<String, JSType> calcTypeInstantiation(
2314           Node callNode, Node receiver, Node firstArg,
2315           FunctionType funType, TypeEnv typeEnv, boolean isFwd) {
2316         Preconditions.checkState(receiver == null || isFwd);
2317         List<String> typeParameters = funType.getTypeParameters();
2318         Multimap<String, JSType> typeMultimap = LinkedHashMultimap.create();
2319         JSType funRecvType = funType.getThisType();
2320         if (receiver != null && funRecvType != null && !funRecvType.isSingletonObj()) {
2321           EnvTypePair pair = analyzeExprFwd(receiver, typeEnv);
2322           unifyWithSubtypeWarnIfFail(funRecvType, pair.type, typeParameters,
2323               typeMultimap, receiver, isFwd);
2324           typeEnv = pair.env;
2325         }
2326         Node arg = firstArg;
2327         int i = 0;
2328         while (arg != null) {
2329           EnvTypePair pair =
2330               isFwd ? analyzeExprFwd(arg, typeEnv) : analyzeExprBwd(arg, typeEnv);
2331           unifyWithSubtypeWarnIfFail(funType.getFormalType(i), pair.type,
2332               typeParameters, typeMultimap, arg, isFwd);
2333           arg = arg.getNext();
2334           typeEnv = pair.env;
2335           i++;
2336         }
2337         ImmutableMap.Builder<String, JSType> builder = ImmutableMap.builder();
2338         for (String typeParam : typeParameters) {
2339           Collection<JSType> types = typeMultimap.get(typeParam);
2340           if (types.size() > 1) {
2341             if (isFwd) {
2342               warnings.add(JSError.make(
2343                   callNode, NOT_UNIQUE_INSTANTIATION,
2344                   Integer.toString(types.size()),
2345                   UniqueNameGenerator.getOriginalName(typeParam),
2346                   types.toString(),
2347                   funType.toString()));
2348             }
2349             builder.put(typeParam, JSType.UNKNOWN);
2350           } else if (types.size() == 1) {
2351             JSType t = Iterables.getOnlyElement(types);
2352             builder.put(typeParam, t.isBottom() ? JSType.UNKNOWN : t);
2353           } else {
2354             // Put ? for any uninstantiated type variables
2355             builder.put(typeParam, JSType.UNKNOWN);
2356           }
2357         }
2358         return builder.build();
2359       }

2362           List<String> typeParameters, Multimap<String, JSType> typeMultimap,

2377       private EnvTypePair analyzeNonStrictComparisonFwd(
2378           Node expr, TypeEnv inEnv, JSType specializedType) {
2379         Token tokenType = expr.getType();
2380         Node lhs = expr.getFirstChild();
2381         Node rhs = expr.getLastChild();
2382     
2383         if (specializedType.isTrueOrTruthy() || specializedType.isFalseOrFalsy()) {
2384           if (lhs.isTypeOf()) {
2385             return analyzeSpecializedTypeof(
2386                 lhs, rhs, tokenType, inEnv, specializedType);
2387           } else if (rhs.isTypeOf()) {
2388             return analyzeSpecializedTypeof(
2389                 rhs, lhs, tokenType, inEnv, specializedType);
2390           } else if (isGoogTypeof(lhs)) {
2391             return analyzeGoogTypeof(lhs, rhs, inEnv, specializedType);
2392           } else if (isGoogTypeof(rhs)) {
2393             return analyzeGoogTypeof(rhs, lhs, inEnv, specializedType);
2394           }
2395         }
2396     
2397         EnvTypePair lhsPair = analyzeExprFwd(lhs, inEnv);
2398         EnvTypePair rhsPair = analyzeExprFwd(rhs, lhsPair.env);
2399         // This env may contain types that have been tightened after nullable deref.
2400         TypeEnv preciseEnv = rhsPair.env;
2401         JSType lhsType = lhsPair.type;
2402         JSType rhsType = rhsPair.type;
2403     
2404         if (tokenType == Token.EQ && specializedType.isTrueOrTruthy()
2405             || tokenType == Token.NE && specializedType.isFalseOrFalsy()) {
2406           if (lhsType.isNullOrUndef()) {
2407             rhsPair = analyzeExprFwd(
2408                 rhs, preciseEnv, JSType.UNKNOWN, JSType.NULL_OR_UNDEF);
2409           } else if (rhsType.isNullOrUndef()) {
2410             lhsPair = analyzeExprFwd(
2411                 lhs, preciseEnv, JSType.UNKNOWN, JSType.NULL_OR_UNDEF);
2412             rhsPair = analyzeExprFwd(rhs, lhsPair.env);
2413           } else if (!JSType.NULL.isSubtypeOf(lhsType) && !JSType.UNDEFINED.isSubtypeOf(lhsType)) {
2414             rhsType = rhsType.removeType(JSType.NULL_OR_UNDEF);
2415             rhsPair = analyzeExprFwd(rhs, preciseEnv, JSType.UNKNOWN, rhsType);
2416           } else if (!JSType.NULL.isSubtypeOf(rhsType) && !JSType.UNDEFINED.isSubtypeOf(rhsType)) {
2417             lhsType = lhsType.removeType(JSType.NULL_OR_UNDEF);
2418             lhsPair = analyzeExprFwd(lhs, preciseEnv, JSType.UNKNOWN, lhsType);
2419             rhsPair = analyzeExprFwd(rhs, lhsPair.env);
2420           }
2421         } else if (tokenType == Token.EQ && specializedType.isFalseOrFalsy()
2422             || tokenType == Token.NE && specializedType.isTrueOrTruthy()) {
2423           if (lhsType.isNullOrUndef()) {
2424             rhsType = rhsType.removeType(JSType.NULL_OR_UNDEF);
2425             rhsPair = analyzeExprFwd(rhs, preciseEnv, JSType.UNKNOWN, rhsType);
2426           } else if (rhsType.isNullOrUndef()) {
2427             lhsType = lhsType.removeType(JSType.NULL_OR_UNDEF);
2428             lhsPair = analyzeExprFwd(lhs, preciseEnv, JSType.UNKNOWN, lhsType);
2429             rhsPair = analyzeExprFwd(rhs, lhsPair.env);
2430           }
2431         }
2432         rhsPair.type = JSType.BOOLEAN;
2433         return rhsPair;
2434       }

2436       private EnvTypePair analyzeObjLitFwd(
2437           Node objLit, TypeEnv inEnv, JSType requiredType, JSType specializedType) {
2438         if (NodeUtil.isEnumDecl(objLit.getParent())) {
2439           return analyzeEnumObjLitFwd(objLit, inEnv, requiredType);
2440         }
2441         JSDocInfo jsdoc = objLit.getJSDocInfo();
2442         boolean isStruct = jsdoc != null && jsdoc.makesStructs();
2443         boolean isDict = jsdoc != null && jsdoc.makesDicts();
2444         TypeEnv env = inEnv;
2445         JSType result = pickReqObjType(objLit);
2446         for (Node prop : objLit.children()) {
2447           if (isStruct && prop.isQuotedString()) {
2448             warnings.add(JSError.make(prop, ILLEGAL_OBJLIT_KEY, "struct"));
2449           } else if (isDict && !prop.isQuotedString()) {
2450             warnings.add(JSError.make(prop, ILLEGAL_OBJLIT_KEY, "dict"));
2451           }
2452           String pname = NodeUtil.getObjectLitKeyName(prop);
2453           // We can't assign to a getter to change its value.
2454           // We can't do a prop access on a setter.
2455           // So, we don't associate pname with a getter/setter.
2456           // We add a property with a name that's weird enough to hopefully avoid
2457           // an accidental clash.
2458           if (prop.isGetterDef() || prop.isSetterDef()) {
2459             EnvTypePair pair = analyzeExprFwd(prop.getFirstChild(), env);
2460             FunctionType funType = pair.type.getFunType();
2461             Preconditions.checkNotNull(funType);
2462             String specialPropName;
2463             JSType propType;
2464             if (prop.isGetterDef()) {
2465               specialPropName = GETTER_PREFIX + pname;
2466               propType = funType.getReturnType();
2467             } else {
2468               specialPropName = SETTER_PREFIX + pname;
2469               propType = pair.type;
2470             }
2471             result = result.withProperty(new QualifiedName(specialPropName), propType);
2472             env = pair.env;
2473           } else {
2474             QualifiedName qname = new QualifiedName(pname);
2475             JSType jsdocType = symbolTable.getPropDeclaredType(prop);
2476             JSType reqPtype, specPtype;
2477             if (jsdocType != null) {
2478               reqPtype = specPtype = jsdocType;
2479             } else if (requiredType.mayHaveProp(qname)) {
2480               reqPtype = specPtype = requiredType.getProp(qname);
2481               if (specializedType.mayHaveProp(qname)) {
2482                 specPtype = specializedType.getProp(qname);
2483               }
2484             } else {
2485               reqPtype = specPtype = JSType.UNKNOWN;
2486             }
2487             EnvTypePair pair = analyzeExprFwd(prop.getFirstChild(), env, reqPtype, specPtype);
2488             if (jsdocType != null) {
2489               // First declare it; then set the maybe more precise inferred type
2490               result = result.withDeclaredProperty(qname, jsdocType, false);
2491               if (!pair.type.isSubtypeOf(jsdocType)) {
2492                 warnings.add(JSError.make(
2493                     prop, INVALID_OBJLIT_PROPERTY_TYPE,
2494                     errorMsgWithTypeDiff(jsdocType, pair.type)));
2495                 pair.type = jsdocType;
2496               }
2497             }
2498             result = result.withProperty(qname, pair.type);
2499             env = pair.env;
2500           }
2501         }
2502         return new EnvTypePair(env, result);
2503       }

2626       private boolean tightenTypeAndDontWarn(
2627           String varName, Node n, JSType declared, JSType inferred, JSType required) {
2628         boolean isSpecializableTop = declared != null && declared.isTop()
2629             && (!inferred.isTop() || NodeUtil.isPropertyTest(compiler, n.getParent()));
2630         boolean fuzzyDeclaration = declared == null || declared.isUnknown() || isSpecializableTop;
2631     
2632         return fuzzyDeclaration
2633             // The intent is to be looser about warnings in the case when a value
2634             // is passed to a function (as opposed to being created locally),
2635             // because then we have less information about the value.
2636             // The accurate way to do this is to taint types so that we can track
2637             // where a type comes from (local or not).
2638             // Without tainting (eg, we used to have locations), we approximate this
2639             // by checking the variable name.
2640             && (varName == null || currentScope.isFormalParam(varName)
2641                 || currentScope.isOuterVar(varName))
2642             // If required is loose, it's easier for it to be a subtype of inferred.
2643             // We only tighten the type if the non-loose required is also a subtype.
2644             // Otherwise, we would be skipping warnings too often.
2645             // This is important b/c analyzePropAccess & analyzePropLvalue introduce
2646             // loose objects, even if there are no undeclared formals.
2647             && required.isNonLooseSubtypeOf(inferred);
2648       }

2873       private EnvTypePair analyzePropAccessFwd(Node receiver, String pname,
2874           TypeEnv inEnv, JSType requiredType, JSType specializedType) {
2875         QualifiedName propQname = new QualifiedName(pname);
2876         Node propAccessNode = receiver.getParent();
2877         EnvTypePair pair;
2878         JSType reqObjType = pickReqObjType(propAccessNode);
2879         JSType recvReqType, recvSpecType;
2880     
2881         // First, analyze the receiver object.
2882         if (NodeUtil.isPropertyTest(compiler, propAccessNode)
2883             && !specializedType.isFalseOrFalsy()
2884             // The NodeUtil method doesn't use types, so it can't see that the
2885             // else branch of "if (!x.prop)" is a property test.
2886             || specializedType.isTrueOrTruthy()) {
2887           recvReqType = reqObjType;
2888           pair = analyzeExprFwd(receiver, inEnv, recvReqType);
2889           JSType subtypeWithProp = pair.type.findSubtypeWithProp(propQname);
2890           if (subtypeWithProp.isBottom()) {
2891             recvSpecType = reqObjType;
2892           } else {
2893             recvSpecType = subtypeWithProp;
2894           }
2895           if (specializedType.isTrueOrTruthy()) {
2896             // This handles cases like: if (x.prop1 && x.prop1.prop2) { ... }
2897             // In the THEN branch, the only thing we know about x.prop1 is that it
2898             // has a truthy property, so x.prop1 should be a loose object to avoid
2899             // spurious warnings.
2900             recvSpecType = recvSpecType.withLoose().withProperty(propQname, specializedType);
2901           } else {
2902             recvSpecType = recvSpecType.withProperty(propQname, specializedType);
2903           }
2904         } else if (specializedType.isFalseOrFalsy()) {
2905           recvReqType = recvSpecType = reqObjType;
2906         } else {
2907           recvReqType = reqObjType.withProperty(propQname, requiredType);
2908           recvSpecType = reqObjType.withProperty(propQname, specializedType);
2909         }
2910         pair = analyzeExprFwd(receiver, inEnv, recvReqType, recvSpecType);
2911         pair = mayWarnAboutNullableReferenceAndTighten(
2912             receiver, pair.type, recvSpecType, pair.env);
2913         JSType recvType = pair.type.autobox();
2914         if (recvType.isUnknown()
2915             || mayWarnAboutNonObject(receiver, recvType, specializedType)) {
2916           return new EnvTypePair(pair.env, requiredType);
2917         }
2918         FunctionType ft = recvType.getFunTypeIfSingletonObj();
2919         if (ft != null && pname.equals("call")) {
2920           return new EnvTypePair(pair.env,
2921               commonTypes.fromFunctionType(ft.transformByCallProperty()));
2922         }
2923         if (ft != null && pname.equals("apply")) {
2924           return new EnvTypePair(pair.env,
2925               commonTypes.fromFunctionType(ft.transformByApplyProperty(commonTypes)));
2926         }
2927         if (this.convention.isSuperClassReference(pname)) {
2928           if (ft != null && ft.isUniqueConstructor()) {
2929             JSType result = ft.getSuperPrototype();
2930             pair.type = result != null ? result : JSType.UNDEFINED;
2931             return pair;
2932           }
2933         }
2934         if (propAccessNode.isGetProp() &&
2935             mayWarnAboutDictPropAccess(receiver, recvType)) {
2936           return new EnvTypePair(pair.env, requiredType);
2937         }
2938         if (recvType.isTop()) {
2939           recvType = JSType.TOP_OBJECT;
2940         }
2941         // Then, analyze the property access.
2942         QualifiedName getterPname = new QualifiedName(GETTER_PREFIX + pname);
2943         if (recvType.hasProp(getterPname)) {
2944           return new EnvTypePair(pair.env, recvType.getProp(getterPname));
2945         }
2946         JSType resultType = recvType.getProp(propQname);
2947         if (resultType != null && resultType.isBottom()) {
2948           warnings.add(JSError.make(propAccessNode, BOTTOM_PROP,
2949                   pname, recvType.toString()));
2950           return new EnvTypePair(pair.env, JSType.UNKNOWN);
2951         }
2952         if (!propAccessNode.getParent().isExprResult()
2953             && !specializedType.isTrueOrTruthy()
2954             && !specializedType.isFalseOrFalsy()
2955             && !recvType.mayBeDict()
2956             && !mayWarnAboutInexistentProp(propAccessNode, recvType, propQname, pair.env)
2957             && recvType.hasProp(propQname)
2958             && !resultType.isSubtypeOf(requiredType)
2959             && tightenTypeAndDontWarn(
2960                 receiver.isName() ? receiver.getString() : null,
2961                 receiver,
2962                 recvType.getDeclaredProp(propQname),
2963                 resultType, requiredType)) {
2964           // Tighten the inferred type and don't warn.
2965           // See analyzeNameFwd for explanation about types as lower/upper bounds.
2966           resultType = resultType.specialize(requiredType);
2967           LValueResultFwd lvr =
2968               analyzeLValueFwd(propAccessNode, inEnv, resultType);
2969           TypeEnv updatedEnv =
2970               updateLvalueTypeInEnv(lvr.env, propAccessNode, lvr.ptr, resultType);
2971           return new EnvTypePair(updatedEnv, resultType);
2972         }
2973         // We've already warned about missing props, and never want to return null.
2974         if (resultType == null) {
2975           resultType = JSType.UNKNOWN;
2976         }
2977         // Any potential type mismatch will be caught by the context
2978         return new EnvTypePair(pair.env, resultType);
2979       }

3018       private TypeEnv collectTypesForFreeVarsFwd(Node n, TypeEnv env) {
3019         Preconditions.checkArgument(n.isFunction()
3020             || n.isName() && NodeUtil.isCallOrNewTarget(n));
3021         String fnName = n.isFunction() ? symbolTable.getFunInternalName(n) : n.getString();
3022         NTIScope innerScope = currentScope.getScope(fnName);
3023         for (String freeVar : innerScope.getOuterVars()) {
3024           if (innerScope.getDeclaredTypeOf(freeVar) == null) {
3025             FunctionType summary = summaries.get(innerScope).getFunType();
3026             JSType outerType = envGetType(env, freeVar);
3027             if (outerType == null) {
3028               outerType = JSType.UNKNOWN;
3029             }
3030             JSType innerType = summary.getOuterVarPrecondition(freeVar);
3031             if (// We don't warn if the innerType is a loose object, because we
3032                 // haven't found an easy way to avoid false positives.
3033                 !innerType.isLoose()
3034                 // If n is a function expression, we don't know where it's called,
3035                 // so we don't warn for uninitialized variables.
3036                 && (n.isName() || n.isFunction() && !outerType.isUndefined())
3037                 && !JSType.haveCommonSubtype(outerType, innerType)) {
3038               warnings.add(JSError.make(n, CROSS_SCOPE_GOTCHA,
3039                       freeVar, outerType.toString(), innerType.toString()));
3040             }
3041             // If n is a callee node, we only want to keep the type in the callee.
3042             // If n is a function expression, we're not sure if it gets called,
3043             // so we join the types.
3044             env = envPutType(env, freeVar,
3045                 n.isFunction() ? JSType.join(innerType, outerType) : innerType);
3046           }
3047         }
3048         return env;
3049       }

3266       private EnvTypePair analyzeNameBwd(
3267           Node expr, TypeEnv outEnv, JSType requiredType) {
3268         String varName = expr.getString();
3269         if (varName.equals("undefined")) {
3270           return new EnvTypePair(outEnv, JSType.UNDEFINED);
3271         }
3272         JSType inferredType = envGetType(outEnv, varName);
3273         println(varName, "'s inferredType: ", inferredType,
3274             " requiredType:  ", requiredType);
3275         if (inferredType == null) {
3276           return new EnvTypePair(outEnv, JSType.UNKNOWN);
3277         }
3278         JSType preciseType = inferredType.specialize(requiredType);
3279         if ((currentScope.isUndeclaredFormal(varName)
3280             || currentScope.isUndeclaredOuterVar(varName))
3281             && preciseType.hasNonScalar()) {
3282           preciseType = preciseType.withLoose();
3283         }
3284         println(varName, "'s preciseType: ", preciseType);
3285         if (preciseType.isBottom()) {
3286           // If there is a type mismatch, we can propagate the previously
3287           // inferred type or the required type.
3288           // Propagating the already inferred type means that the type of the
3289           // variable is stable throughout the function body.
3290           // Propagating the required type means that the type chosen for a
3291           // formal is the one closest to the function header, which helps
3292           // generate more intuitive warnings in the fwd direction.
3293           // But there is a small chance that the different types of the same

3409       private EnvTypePair analyzeCallNewBwd(
3410           Node expr, TypeEnv outEnv, JSType requiredType) {
3411         Preconditions.checkArgument(expr.isNew() || expr.isCall());
3412         Node callee = expr.getFirstChild();
3413         EnvTypePair pair = analyzeExprBwd(callee, outEnv, commonTypes.topFunction());
3414         TypeEnv envAfterCallee = pair.env;
3415         FunctionType funType = pair.type.getFunType();
3416         if (funType == null) {
3417           return analyzeCallNodeArgumentsBwd(expr, envAfterCallee);
3418         } else if (funType.isLoose()) {
3419           return analyzeLooseCallNodeBwd(expr, envAfterCallee, requiredType);
3420         } else if (expr.isCall() && funType.isSomeConstructorOrInterface()
3421             || expr.isNew() && !funType.isSomeConstructorOrInterface()) {
3422           return analyzeCallNodeArgumentsBwd(expr, envAfterCallee);
3423         } else if (funType.isTopFunction()) {
3424           return analyzeCallNodeArgumentsBwd(expr, envAfterCallee);
3425         }
3426         if (callee.isName() && !funType.isGeneric() && expr.isCall()) {
3427           createDeferredCheckBwd(expr, requiredType);
3428         }
3429         int numArgs = expr.getChildCount() - 1;
3430         if (numArgs < funType.getMinArity() || numArgs > funType.getMaxArity()) {
3431           return analyzeCallNodeArgumentsBwd(expr, envAfterCallee);
3432         }
3433         if (funType.isGeneric()) {
3434           Map<String, JSType> typeMap =
3435               calcTypeInstantiationBwd(expr, funType, envAfterCallee);
3436           funType = funType.instantiateGenerics(typeMap);
3437         }
3438         TypeEnv tmpEnv = envAfterCallee;
3439         // In bwd direction, analyze arguments in reverse
3440         for (int i = expr.getChildCount() - 2; i >= 0; i--) {
3441           JSType formalType = funType.getFormalType(i);
3442           // The type of a formal can be BOTTOM as the result of a join.
3443           // Don't use this as a requiredType.
3444           if (formalType.isBottom()) {
3445             formalType = JSType.UNKNOWN;
3446           }
3447           Node arg = expr.getChildAtIndex(i + 1);
3448           tmpEnv = analyzeExprBwd(arg, tmpEnv, formalType).env;
3449           // We don't need deferred checks for args in BWD
3450         }
3451         JSType retType =
3452             expr.isNew() ? funType.getThisType() : funType.getReturnType();
3453         return new EnvTypePair(tmpEnv, retType);
3454       }

3796           Node expr, TypeEnv inEnv, JSType type, boolean insideQualifiedName) {

3799           case THIS: {

3802                   currentScope.getDeclaredTypeOf(THIS_ID),

3806               lvalResult = new LValueResultFwd(inEnv, JSType.UNKNOWN, null, null);
3807             }
3808             break;

3810           case NAME: {

3814                 currentScope.getDeclaredTypeOf(varName),

3819           case GETELEM: {

3823                 ? new QualifiedName(prop.getString()) : null;
3824             LValueResultFwd recvLvalue = analyzeReceiverLvalFwd(obj, pname, inEnv, type);
3825             if (!recvLvalue.type.isSubtypeOf(JSType.TOP_OBJECT)) {
3826               EnvTypePair pair = analyzeExprFwd(prop, recvLvalue.env, type);
3827               lvalResult = new LValueResultFwd(pair.env, type, null, null);
3828               break;
3829             }
3830             JSType indexType = recvLvalue.type.getIndexType();
3831             // (1) A getelem where the receiver is an IObject
3832             if (expr.isGetElem() && indexType != null) {
3833               lvalResult = analyzeIObjectElmLvalFwd(prop, recvLvalue, indexType);
3834               break;
3835             }
3836             // (2) A getelem where the prop is a string literal is like a getprop
3837             if (expr.isGetProp() || prop.isString()) {
3838               lvalResult = analyzePropLValFwd(obj, pname, recvLvalue, type, insideQualifiedName);
3839               break;
3840             }
3841             // (3) All other getelems
3842             // TODO(dimvar): there is some recomputation here; the receiver will be
3843             // analyzed again. Some more refactoring can fix this.
3844             EnvTypePair pair = analyzeExprFwd(expr, recvLvalue.env, type);
3845             lvalResult = new LValueResultFwd(pair.env, pair.type, null, null);
3846             break;
3847           }
3848           case VAR: { // Can happen iff its parent is a for/in.
3849             Preconditions.checkState(NodeUtil.isForIn(expr.getParent()));
3850             Node vdecl = expr.getFirstChild();
3851             String name = vdecl.getString();
3852             // For/in can never have rhs of its VAR
3853             Preconditions.checkState(!vdecl.hasChildren());
3854             return new LValueResultFwd(inEnv, JSType.STRING, null, new QualifiedName(name));
3855           }
3856           default: {
3857             // Expressions that aren't lvalues should be handled because they may
3858             // be, e.g., the left child of a getprop.
3859             // We must check that they are not the direct lvalues.
3860             Preconditions.checkState(insideQualifiedName);
3861             EnvTypePair pair = analyzeExprFwd(expr, inEnv, type);
3862             return new LValueResultFwd(pair.env, pair.type, null, null);
3863           }
3864         }
3865         maybeSetTypeI(expr, lvalResult.type);
3866         mayWarnAboutUnknownType(expr, lvalResult.type);
3867         return lvalResult;
3868       }

3887       private EnvTypePair mayWarnAboutNullableReferenceAndTighten(
3888           Node obj, JSType recvType, JSType maybeSpecType, TypeEnv inEnv) {
3889         if (!recvType.isUnknown()
3890             && !recvType.isTop()
3891             && (JSType.NULL.isSubtypeOf(recvType)
3892                 || JSType.UNDEFINED.isSubtypeOf(recvType))) {
3893           JSType minusNull = recvType.removeType(JSType.NULL_OR_UNDEF);
3894           if (!minusNull.isBottom()) {
3895             if (this.reportNullDeref) {
3896               warnings.add(JSError.make(
3897                   obj, NULLABLE_DEREFERENCE, recvType.toString()));
3898             }
3899             TypeEnv outEnv = inEnv;
3900             if (obj.isQualifiedName()) {
3901               QualifiedName qname = QualifiedName.fromNode(obj);
3902               if (maybeSpecType != null && maybeSpecType.isSubtypeOf(minusNull)) {
3903                 minusNull = maybeSpecType;
3904               }
3905               outEnv = updateLvalueTypeInEnv(inEnv, obj, qname, minusNull);
3906             }
3907             return new EnvTypePair(outEnv, minusNull);
3908           }
3909         }
3910         return new EnvTypePair(inEnv, recvType);
3911       }

3913       private LValueResultFwd analyzePropLValFwd(Node obj, QualifiedName pname,
3914           LValueResultFwd recvLvalue, JSType requiredType, boolean insideQualifiedName) {
3915         Preconditions.checkArgument(pname.isIdentifier());
3916         TypeEnv inEnv = recvLvalue.env;
3917         JSType recvType = recvLvalue.type;
3918         if (!recvType.isUnion() && !recvType.isSingletonObj()) {
3919           // The lvalue is a subtype of TOP_OBJECT, but does not contain an object
3920           // yet, eg, it is ?, truthy, or bottom.
3921           recvType = JSType.TOP_OBJECT.withLoose();
3922         }
3923         Node propAccessNode = obj.getParent();
3924         if (propAccessNode.isGetProp() &&
3925             propAccessNode.getParent().isAssign() &&
3926             mayWarnAboutPropCreation(pname, propAccessNode, recvType)) {
3927           return new LValueResultFwd(inEnv, requiredType, null, null);
3928         }
3929         if (!insideQualifiedName
3930             && mayWarnAboutConstProp(propAccessNode, recvType, pname)) {
3931           return new LValueResultFwd(inEnv, requiredType, null, null);
3932         }
3933         if (!recvType.hasProp(pname)) {
3934           if (recvType.isLoose()) {
3935             // For loose objects, create the inner property if it doesn't exist.
3936             recvType = recvType.withProperty(pname, JSType.TOP_OBJECT.withLoose());
3937             inEnv = updateLvalueTypeInEnv(inEnv, obj, recvLvalue.ptr, recvType);
3938           } else {
3939             // Warn for inexistent prop either on the non-top-level of a qualified
3940             // name, or for assignment ops that won't create a new property.
3941             boolean warnForInexistentProp = insideQualifiedName ||
3942                 propAccessNode.getParent().getType() != Token.ASSIGN;
3943             if (warnForInexistentProp
3944                 && !recvType.isUnknown()
3945                 && !recvType.mayBeDict()) {
3946               mayWarnAboutInexistentProp(propAccessNode, recvType, pname, inEnv);
3947               return new LValueResultFwd(inEnv, requiredType, null, null);
3948             }
3949           }
3950         }
3951         if (propAccessNode.isGetElem()) {
3952           mayWarnAboutStructPropAccess(obj, recvType);
3953         } else if (propAccessNode.isGetProp()) {
3954           mayWarnAboutDictPropAccess(obj, recvType);
3955         }
3956         QualifiedName setterPname =
3957             new QualifiedName(SETTER_PREFIX + pname.getLeftmostName());
3958         if (recvType.hasProp(setterPname)) {
3959           FunctionType funType = recvType.getProp(setterPname).getFunType();
3960           Preconditions.checkNotNull(funType);
3961           JSType formalType = funType.getFormalType(0);
3962           Preconditions.checkState(!formalType.isBottom());
3963           return new LValueResultFwd(inEnv, formalType, formalType, null);
3964         }
3965         QualifiedName ptr = recvLvalue.ptr == null
3966             ? null : QualifiedName.join(recvLvalue.ptr, pname);
3967         return recvType.mayHaveProp(pname)
3968             ? new LValueResultFwd(
3969                 inEnv, recvType.getProp(pname), recvType.getDeclaredProp(pname), ptr)
3970             : new LValueResultFwd(inEnv, JSType.UNKNOWN, null, ptr);
3971       }

3973       private LValueResultFwd analyzeReceiverLvalFwd(
3974           Node obj, QualifiedName pname, TypeEnv inEnv, JSType propType) {
3975         // pname is null when the property name is not known.
3976         Preconditions.checkArgument(pname == null || pname.isIdentifier());
3977         JSType reqObjType = pickReqObjType(obj.getParent());
3978         if (pname != null) {
3979           reqObjType = reqObjType.withProperty(pname, propType);
3980         }
3981         LValueResultFwd lvalue = analyzeLValueFwd(obj, inEnv, reqObjType, true);
3982         EnvTypePair pair = mayWarnAboutNullableReferenceAndTighten(
3983             obj, lvalue.type, null, lvalue.env);
3984         JSType lvalueType = pair.type.autobox();
3985         if (!lvalueType.isSubtypeOf(JSType.TOP_OBJECT)) {
3986           warnings.add(JSError.make(obj, PROPERTY_ACCESS_ON_NONOBJECT,
3987                   getPropNameForErrorMsg(obj.getParent()), lvalueType.toString()));
3988         }
3989         lvalue.type = lvalueType;
3990         lvalue.env = pair.env;
3991         return lvalue;
3992       }

4012       /** When {@code doSlicing} is set, remove the lvalue from the returned env */
4013       private LValueResultBwd analyzeLValueBwd(Node expr, TypeEnv outEnv,
4014           JSType type, boolean doSlicing, boolean insideQualifiedName) {
4015         switch (expr.getType()) {
4016           case THIS:
4017           case NAME: {
4018             EnvTypePair pair = analyzeExprBwd(expr, outEnv, type);
4019             String name = expr.getQualifiedName();
4020             JSType declType = currentScope.getDeclaredTypeOf(name);
4021             if (doSlicing) {
4022               pair.env = envPutType(pair.env, name,
4023                   declType != null ? declType : JSType.UNKNOWN);
4024             }
4025             return new LValueResultBwd(pair.env, pair.type,
4026                 pair.type.hasNonScalar() ? new QualifiedName(name) : null);
4027           }
4028           case GETPROP: {
4029             Node obj = expr.getFirstChild();
4030             QualifiedName pname =
4031                 new QualifiedName(expr.getLastChild().getString());
4032             return analyzePropLValBwd(obj, pname, outEnv, type, doSlicing);
4033           }
4034           case GETELEM: {
4035             if (expr.getLastChild().isString()) {
4036               Node obj = expr.getFirstChild();
4037               QualifiedName pname =
4038                   new QualifiedName(expr.getLastChild().getString());
4039               return analyzePropLValBwd(obj, pname, outEnv, type, doSlicing);
4040             }
4041             EnvTypePair pair = analyzeExprBwd(expr, outEnv, type);
4042             return new LValueResultBwd(pair.env, pair.type, null);
4043           }
4044           default: {
4045             // Expressions that aren't lvalues should be handled because they may
4046             // be, e.g., the left child of a getprop.
4047             // We must check that they are not the direct lvalues.
4048             Preconditions.checkState(insideQualifiedName);
4049             EnvTypePair pair = analyzeExprBwd(expr, outEnv, type);
4050             return new LValueResultBwd(pair.env, pair.type, null);
4051           }
4052         }
4053       }

4055       private LValueResultBwd analyzePropLValBwd(Node obj, QualifiedName pname,
4056           TypeEnv outEnv, JSType type, boolean doSlicing) {
4057         Preconditions.checkArgument(pname.isIdentifier());
4058         JSType reqObjType =
4059             pickReqObjType(obj.getParent()).withProperty(pname, type);
4060         LValueResultBwd lvalue =
4061             analyzeLValueBwd(obj, outEnv, reqObjType, false, true);
4062         if (lvalue.ptr != null) {
4063           lvalue.ptr = QualifiedName.join(lvalue.ptr, pname);
4064           if (doSlicing) {
4065             String objName = lvalue.ptr.getLeftmostName();
4066             QualifiedName props = lvalue.ptr.getAllButLeftmost();
4067             JSType objType = envGetType(lvalue.env, objName);
4068             // withoutProperty only removes inferred properties
4069             JSType slicedObjType = objType.withoutProperty(props);
4070             lvalue.env = envPutType(lvalue.env, objName, slicedObjType);
4071           }
4072         }
4073         lvalue.type = lvalue.type.mayHaveProp(pname) ?
4074             lvalue.type.getProp(pname) : JSType.UNKNOWN;
4075         return lvalue;
4076       }

4078       private static JSType pickReqObjType(Node expr) {
4079         Token exprKind = expr.getType();
4080         switch (exprKind) {
4081           case OBJECTLIT: {
4082             JSDocInfo jsdoc = expr.getJSDocInfo();
4083             if (jsdoc != null && jsdoc.makesStructs()) {
4084               return JSType.TOP_STRUCT;
4085             }
4086             if (jsdoc != null && jsdoc.makesDicts()) {
4087               return JSType.TOP_DICT;
4088             }
4089             return JSType.TOP_OBJECT;
4090           }
4091           case FOR:
4092             Preconditions.checkState(NodeUtil.isForIn(expr));
4093             return JSType.TOP_OBJECT;
4094           case GETPROP:
4095           case GETELEM:
4096           case IN:
4097             return JSType.TOP_OBJECT;
4098           default:
4099             throw new RuntimeException("Unhandled node for pickReqObjType: " + exprKind);
4100         }
4101       }

4117       private static class DeferredCheck {
4118         final Node callSite;
4119         final NTIScope callerScope;
4120         final NTIScope calleeScope;
4121         // Null types means that they were declared
4122         // (and should have been checked during inference)
4123         JSType expectedRetType;
4124         List<JSType> argTypes;
4125     
4126         DeferredCheck(
4127             Node callSite,
4128             JSType expectedRetType,
4129             NTIScope callerScope,
4130             NTIScope calleeScope) {
4131           this.callSite = callSite;
4132           this.expectedRetType = expectedRetType;
4133           this.callerScope = callerScope;
4134           this.calleeScope = calleeScope;
4135         }
4136     
4137         void updateReturn(JSType expectedRetType) {
4138           if (this.expectedRetType != null) {
4139             this.expectedRetType = JSType.meet(this.expectedRetType, expectedRetType);
4140           }
4141         }
4142     
4143         void updateArgTypes(List<JSType> argTypes) {
4144           this.argTypes = argTypes;
4145         }
4146     
4147         private void runCheck(
4148             Map<NTIScope, JSType> summaries, WarningReporter warnings) {
4149           FunctionType fnSummary = summaries.get(this.calleeScope).getFunType();
4150           println(
4151               "Running deferred check of function: ", calleeScope.getReadableName(),
4152               " with FunctionSummary of: ", fnSummary, " and callsite ret: ",
4153               expectedRetType, " args: ", argTypes);
4154           if (this.expectedRetType != null &&
4155               !fnSummary.getReturnType().isSubtypeOf(this.expectedRetType)) {
4156             warnings.add(JSError.make(
4157                 this.callSite, INVALID_INFERRED_RETURN_TYPE,
4158                 errorMsgWithTypeDiff(
4159                     this.expectedRetType, fnSummary.getReturnType())));
4160           }
4161           int i = 0;
4162           Node argNode = callSite.getSecondChild();
4163           // this.argTypes can be null if in the fwd direction the analysis of the
4164           // call return prematurely, eg, because of a WRONG_ARGUMENT_COUNT.
4165           if (this.argTypes == null) {
4166             return;
4167           }
4168           for (JSType argType : this.argTypes) {
4169             JSType formalType = fnSummary.getFormalType(i);
4170             if (argNode.isName() && callerScope.isKnownFunction(argNode.getString())) {
4171               argType = summaries.get(callerScope.getScope(argNode.getString()));
4172             }
4173             if (argType != null && !argType.isSubtypeOf(formalType)) {
4174               warnings.add(JSError.make(
4175                   argNode, INVALID_ARGUMENT_TYPE,
4176                   Integer.toString(i + 1), calleeScope.getReadableName(),
4177                   formalType.toString(),
4178                   argType.toString()));
4179             }
4180             i++;
4181             argNode = argNode.getNext();
4182           }
4183         }
