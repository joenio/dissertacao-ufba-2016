17     package com.google.javascript.jscomp.newtypes;
18     
19     import com.google.common.base.Preconditions;
20     
21     /**
22      *
23      * @author dimvar@google.com (Dimitris Vardoulakis)
24      */
25     public final class FunctionNamespace extends Namespace {
26       private DeclaredTypeRegistry scope;
27     
28       public FunctionNamespace(String name, DeclaredTypeRegistry scope) {
29         Preconditions.checkNotNull(name);
30         Preconditions.checkNotNull(scope);
31         this.name = name;
32         this.scope = scope;
33       }
34     
35       @Override
36       protected JSType computeJSType(JSTypes commonTypes) {
37         Preconditions.checkState(this.namespaceType == null);
38         return JSType.fromObjectType(ObjectType.makeObjectType(
39             commonTypes.getFunctionType(),
40             null,
41             this.scope.getDeclaredFunctionType().toFunctionType(),
42             this,
43             false,
44             ObjectKind.UNRESTRICTED));
45       }
46     
47       DeclaredTypeRegistry getScope() {
48         return this.scope;
49       }
50     }
