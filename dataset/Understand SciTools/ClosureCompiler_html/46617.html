17     package com.google.javascript.jscomp;
18     
19     import com.google.javascript.rhino.InputId;
20     import com.google.javascript.rhino.Node;
21     
22     import java.io.Serializable;
23     
24     /**
25      * An interface for accessing the AST root of an input.
26      *
27      */
28     public interface SourceAst extends Serializable {
29       /**
30        * Gets the root node of the AST for the source file this represents. The AST
31        * is lazily instantiated and cached.
32        */
33       public Node getAstRoot(AbstractCompiler compiler);
34     
35       /**
36        * Removes any references to root node of the AST.  If it is requested again,
37        * another parse will be performed.  This method is needed to allow the ASTs
38        * to be garbage collected if the inputs are still around after compilation.
39        */
40       public void clearAst();
41     
42       /** @return The input id associated with this AST */
43       public InputId getInputId();
44     
45       /** Returns the source file the generated AST represents. */
46       public SourceFile getSourceFile();
47     
48       /**
49        * Sets the source file the generated AST represents. This can be called after
50        * deserializing if access to the source file is needed. If a different file
51        * is provided than that with which this was created, an IllegalStateException
52        * will be thrown.
53        */
54       public void setSourceFile(SourceFile file);
55     }
