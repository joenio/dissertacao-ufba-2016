42     /**
43      * Process aliases in goog.scope blocks.
44      *
45      * <pre>
46      * goog.scope(function() {
47      *   var dom = goog.dom;
48      *   var DIV = dom.TagName.DIV;
49      *
50      *   dom.createElement(DIV);
51      * });
52      * </pre>
53      *
54      * should become
55      *
56      * <pre>
57      * goog.dom.createElement(goog.dom.TagName.DIV);
58      * </pre>
59      *
60      * The advantage of using goog.scope is that the compiler will *guarantee*
61      * the anonymous function will be inlined, even if it can't prove
62      * that it's semantically correct to do so. For example, consider this case:
63      *
64      * <pre>
65      * goog.scope(function() {
66      *   goog.getBar = function () { return alias; };
67      *   ...
68      *   var alias = foo.bar;
69      * })
70      * </pre>
71      *
72      * <p>In theory, the compiler can't inline 'alias' unless it can prove that
73      * goog.getBar is called only after 'alias' is defined. In practice, the
74      * compiler will inline 'alias' anyway, at the risk of 'fixing' bad code.
75      *
76      * @author robbyw@google.com (Robby Walker)
77      */

144       @Override
145       public void hotSwapScript(Node root, Node originalRoot) {
146         Traversal traversal = new Traversal();
147         NodeTraversal.traverseEs6(compiler, root, traversal);
148     
149         if (!traversal.hasErrors()) {
150           // Apply the aliases.
151           List<AliasUsage> aliasWorkQueue =
152                new ArrayList<>(traversal.getAliasUsages());
153           while (!aliasWorkQueue.isEmpty()) {
154             List<AliasUsage> newQueue = new ArrayList<>();
155             for (AliasUsage aliasUsage : aliasWorkQueue) {
156               if (aliasUsage.referencesOtherAlias()) {
157                 newQueue.add(aliasUsage);
158               } else {
159                 aliasUsage.applyAlias();
160               }
161             }
162     
163             // Prevent an infinite loop.
164             if (newQueue.size() == aliasWorkQueue.size()) {
165               Var cycleVar = newQueue.get(0).aliasVar;
166               compiler.report(JSError.make(
167                   cycleVar.getNode(), GOOG_SCOPE_ALIAS_CYCLE, cycleVar.getName()));
168               break;
169             } else {
170               aliasWorkQueue = newQueue;
171             }
172           }
173     
174           // Remove the alias definitions.
175           for (Node aliasDefinition : traversal.getAliasDefinitionsInOrder()) {
176             if (NodeUtil.isNameDeclaration(aliasDefinition.getParent())
177                 && aliasDefinition.getParent().hasOneChild()) {
178               aliasDefinition.getParent().detachFromParent();
179             } else {
180               aliasDefinition.detachFromParent();
181             }
182           }
183     
184           // Collapse the scopes.
185           for (Node scopeCall : traversal.getScopeCalls()) {
186             Node expressionWithScopeCall = scopeCall.getParent();
187             Node scopeClosureBlock = scopeCall.getLastChild().getLastChild();
188             scopeClosureBlock.detachFromParent();
189             expressionWithScopeCall.getParent().replaceChild(
190                 expressionWithScopeCall,
191                 scopeClosureBlock);
192             NodeUtil.tryMergeBlock(scopeClosureBlock);
193           }
194     
195           if (!traversal.getAliasUsages().isEmpty() || !traversal.getAliasDefinitionsInOrder().isEmpty()
196               || !traversal.getScopeCalls().isEmpty()) {
197             compiler.reportCodeChange();
198           }
199         }
200       }

202       private abstract static class AliasUsage {
203         final Var aliasVar;
204         final Node aliasReference;
205     
206         AliasUsage(Var aliasVar, Node aliasReference) {
207           this.aliasVar = aliasVar;
208           this.aliasReference = aliasReference;
209         }
210     
211         /** Checks to see if this references another alias. */
212         public boolean referencesOtherAlias() {
213           Node aliasDefinition = aliasVar.getInitialValue();
214           Node root = NodeUtil.getRootOfQualifiedName(aliasDefinition);
215           Var otherAliasVar = aliasVar.getScope().getOwnSlot(root.getString());
216           return otherAliasVar != null;
217         }
218     
219         public abstract void applyAlias();
220       }
221     
222       private static class AliasedNode extends AliasUsage {
223         AliasedNode(Var aliasVar, Node aliasReference) {
224           super(aliasVar, aliasReference);
225         }
226     
227         @Override
228         public void applyAlias() {
229           Node aliasDefinition = aliasVar.getInitialValue();
230           Node replacement = aliasDefinition.cloneTree();
231           replacement.useSourceInfoFromForTree(aliasReference);
232           if (aliasReference.getType() == Token.STRING_KEY) {
233             Preconditions.checkState(!aliasReference.hasChildren());
234             aliasReference.addChildToFront(replacement);
235           } else {
236             aliasReference.getParent().replaceChild(aliasReference, replacement);
237           }
238         }
239       }
240     
241       private static class AliasedTypeNode extends AliasUsage {
242         AliasedTypeNode(Var aliasVar, Node aliasReference) {
243           super(aliasVar, aliasReference);
244         }
245     
246         @Override
247         public void applyAlias() {
248           Node aliasDefinition = aliasVar.getInitialValue();
249           String aliasName = aliasVar.getName();
250           String typeName = aliasReference.getString();
251           if (typeName.startsWith("$jscomp.scope.")) {
252             // Already visited.
253             return;
254           }
255           String aliasExpanded =
256               Preconditions.checkNotNull(aliasDefinition.getQualifiedName());
257           Preconditions.checkState(typeName.startsWith(aliasName),
258               "%s must start with %s", typeName, aliasName);
259           String replacement =
260               aliasExpanded + typeName.substring(aliasName.length());
261           aliasReference.setString(replacement);
262         }
263       }

283         // Suppose you create an alias.
284         // var x = goog.x;
285         // As a side-effect, this means you can shadow the namespace 'goog'
286         // in inner scopes. When we inline the namespaces, we have to rename

290         // normalization (before optimizations). We run it here on a limited
291         // set of variables, but only as a last resort (because this will screw
292         // up warning messages downstream).

341          * @param scopeRoot the Node which is the root of the current scope
342          * @return the goog.scope() CALL node containing the scopeRoot, or null if scopeRoot is not
343          *     in a goog.scope() call.
344          */
345         private Node findScopeMethodCall(Node scopeRoot) {
346           Node n = scopeRoot.getGrandparent();
347           if (isCallToScopeMethod(n)) {
348             return n;
349           }
350           return null;

353         @Override
354         public void enterScope(NodeTraversal t) {
355           if (t.inGlobalHoistScope()) {
356             return;
357           }
358           if (inGoogScopeBody()) {
359             Scope hoistedScope = t.getClosestHoistScope();
360             if (isGoogScopeFunctionBody(hoistedScope.getRootNode())) {
361               findAliases(t, hoistedScope);
362             }
363           }
364           Node scopeMethodCall = findScopeMethodCall(t.getScope().getRootNode());
365           if (scopeMethodCall != null) {
366             transformation = transformationHandler.logAliasTransformation(
367                 scopeMethodCall.getSourceFileName(), getSourceRegion(scopeMethodCall));
368             findAliases(t, t.getScope());
369             scopeFunctionBody = scopeMethodCall.getLastChild().getLastChild();
370           }
371         }
372     
373         @Override
374         public void exitScope(NodeTraversal t) {
375           if (isGoogScopeFunctionBody(t.getScopeRoot())) {
376             scopeFunctionBody = null;
377             renameNamespaceShadows(t);
378             injectedDecls.clear();
379             aliases.clear();
380             forbiddenLocals.clear();
381             transformation = null;
382             hasNamespaceShadows = false;
383           } else if (inGoogScopeBody()) {
384             findNamespaceShadows(t);
385             reportInvalidVariables(t);
386           }
387         }

389         private void reportInvalidVariables(NodeTraversal t) {
390           Node scopeRoot = t.getScopeRoot();
391           Node enclosingFunctionBody = t.getEnclosingFunction().getLastChild();
392           if (isGoogScopeFunctionBody(enclosingFunctionBody) && scopeRoot.isBlock()
393               && !scopeRoot.getParent().isFunction()) {
394             for (Var v : t.getScope().getVarIterable()) {
395               Node parent = v.getNameNode().getParent();
396               if (NodeUtil.isFunctionDeclaration(parent)) {
397                 // Disallow block-scoped function declarations that leak into the goog.scope
398                 // function body. Technically they shouldn't leak in ES6 but the browsers don't agree
399                 // on that yet.
400                 report(t, v.getNode(), GOOG_SCOPE_INVALID_VARIABLE, v.getName());
401               }
402             }
403           }
404         }

406         private SourcePosition<AliasTransformation> getSourceRegion(Node n) {
407           Node testNode = n;
408           Node next = null;
409           for (; next != null || testNode.isScript();) {
410             next = testNode.getNext();
411             testNode = testNode.getParent();
412           }
413     
414           int endLine = next == null ? Integer.MAX_VALUE : next.getLineno();
415           int endChar = next == null ? Integer.MAX_VALUE : next.getCharno();
416           SourcePosition<AliasTransformation> pos =
417               new SourcePosition<AliasTransformation>() {};
418           pos.setPositionInformation(
419               n.getLineno(), n.getCharno(), endLine, endChar);
420           return pos;
421         }

429         private void findAliases(NodeTraversal t, Scope scope) {
430           for (Var v : scope.getVarIterable()) {
431             Node n = v.getNode();
432             Node parent = n.getParent();
433             // We use isBlock to avoid variables declared in loop headers.
434             boolean isVar = NodeUtil.isNameDeclaration(parent) && parent.getParent().isBlock();
435             boolean isFunctionDecl = NodeUtil.isFunctionDeclaration(parent);
436             if (isVar && n.getFirstChild() != null && n.getFirstChild().isQualifiedName()) {
437               recordAlias(v);
438             } else if (v.isBleedingFunction()) {
439               // Bleeding functions already get a BAD_PARAMETERS error, so just
440               // do nothing.
441             } else if (parent.getType() == Token.PARAM_LIST) {
442               // Parameters of the scope function also get a BAD_PARAMETERS
443               // error.
444             } else if (isVar || isFunctionDecl || NodeUtil.isClassDeclaration(parent)) {
445               boolean isHoisted = NodeUtil.isHoistedFunctionDeclaration(parent);
446               Node grandparent = parent.getParent();
447               Node value = v.getInitialValue();
448               Node varNode = null;
449     
450               // Pull out inline type declaration if present.
451               if (n.getJSDocInfo() != null) {
452                 JSDocInfoBuilder builder = JSDocInfoBuilder.maybeCopyFrom(parent.getJSDocInfo());
453                 if (isFunctionDecl) { // Fix inline return type.
454                   builder.recordReturnType(n.getJSDocInfo().getType());
455                 } else {
456                   builder.recordType(n.getJSDocInfo().getType());
457                 }
458                 parent.setJSDocInfo(builder.build());
459                 n.setJSDocInfo(null);
460               }
461               // Grab the docinfo before we do any AST manipulation.
462               JSDocInfo varDocInfo = v.getJSDocInfo();
463     
464               String name = n.getString();
465               int nameCount = scopedAliasNames.count(name);
466               scopedAliasNames.add(name);
467               String globalName =
468                   "$jscomp.scope." + name + (nameCount == 0 ? "" : ("$jscomp$" + nameCount));
469     
470               compiler.ensureLibraryInjected("base", false);
471     
472               // First, we need to free up the function expression (EXPR)
473               // to be used in another expression.
474               if (isFunctionDecl || NodeUtil.isClassDeclaration(parent)) {
475                 // Replace "function NAME() { ... }" with "var NAME;".
476                 // Replace "class NAME { ... }" with "var NAME;".
477     
478                 // We can't keep the local name on the function expression,
479                 // because IE is buggy and will leak the name into the global
480                 // scope. This is covered in more detail here:
481                 // http://wiki.ecmascript.org/lib/exe/fetch.php?id=resources:resources&cache=cache&media=resources:jscriptdeviationsfromes3.pdf
482                 //
483                 // This will only cause problems if this is a hoisted, recursive
484                 // function, and the programmer is using the hoisting.
485                 Node newName;
486                 if (isFunctionDecl) {
487                   newName = IR.name("");
488                 } else {
489                   newName = IR.empty();
490                 }
491                 newName.useSourceInfoFrom(n);
492                 value.replaceChild(n, newName);
493     
494                 varNode = IR.var(n).useSourceInfoFrom(n);
495                 grandparent.replaceChild(parent, varNode);
496               } else {
497                 if (value != null) {
498                   // If this is a VAR, we can just detach the expression and
499                   // the tree will still be valid.
500                   value.detachFromParent();
501                 }
502                 varNode = parent;
503               }
504     
505               // Add $jscomp.scope.name = EXPR;
506               // Make sure we copy over all the jsdoc and debug info.
507               if (value != null || varDocInfo != null) {
508                 Node newDecl = NodeUtil.newQNameDeclaration(
509                     compiler,
510                     globalName,
511                     value,
512                     varDocInfo)
513                     .useSourceInfoIfMissingFromForTree(n);
514                 NodeUtil.setDebugInformation(
515                     newDecl.getFirstFirstChild(), n, name);
516     
517                 if (isHoisted) {
518                   grandparent.addChildToFront(newDecl);
519                 } else {
520                   grandparent.addChildBefore(newDecl, varNode);
521                 }
522                 injectedDecls.add(newDecl.getFirstChild());
523               }
524     
525               // Rewrite "var name = EXPR;" to "var name = $jscomp.scope.name;"
526               v.getNameNode().addChildToFront(
527                   NodeUtil.newQName(
528                       compiler, globalName, n, name));
529     
530               recordAlias(v);
531             } else {
532               // Do not other kinds of local symbols, like catch params.
533               report(t, n, GOOG_SCOPE_NON_ALIAS_LOCAL, n.getString());
534             }
535           }
536         }

538         private void recordAlias(Var aliasVar) {
539           String name = aliasVar.getName();
540           aliases.put(name, aliasVar);
541     
542           String qualifiedName =
543             aliasVar.getInitialValue().getQualifiedName();
544           transformation.addAlias(name, qualifiedName);
545     
546           int rootIndex = qualifiedName.indexOf('.');
547           if (rootIndex != -1) {
548             String qNameRoot = qualifiedName.substring(0, rootIndex);
549             if (!aliases.containsKey(qNameRoot)) {
550               forbiddenLocals.add(qNameRoot);
551             }
552           }
553         }

555         /** Find out if there are any local shadows of namespaces. */
556         private void findNamespaceShadows(NodeTraversal t) {
557           if (hasNamespaceShadows) {
558             return;
559           }
560     
561           Scope scope = t.getScope();
562           for (Var v : scope.getVarIterable()) {
563             if (forbiddenLocals.contains(v.getName())) {
564               hasNamespaceShadows = true;
565               return;
566             }
567           }
568         }

570         /**
571          * Rename any local shadows of namespaces.
572          * This should be a very rare occurrence, so only do this traversal
573          * if we know that we need it.
574          */
575         private void renameNamespaceShadows(NodeTraversal t) {
576           if (hasNamespaceShadows) {
577             MakeDeclaredNamesUnique.Renamer renamer =
578                 new MakeDeclaredNamesUnique.WhitelistedRenamer(
579                     new MakeDeclaredNamesUnique.ContextualRenamer(),
580                     forbiddenLocals);
581             for (String s : forbiddenLocals) {
582               renamer.addDeclaredName(s, false);
583             }
584             MakeDeclaredNamesUnique uniquifier =
585                 new MakeDeclaredNamesUnique(renamer);
586             Node parent = t.getScopeRoot().getParent();
587             NodeTraversal.traverseEs6(compiler, parent, uniquifier);
588           }
589         }

591         private void validateScopeCall(NodeTraversal t, Node n, Node parent) {
592           if (preprocessorSymbolTable != null) {
593             preprocessorSymbolTable.addReference(n.getFirstChild());
594           }
595           if (!parent.isExprResult()) {
596             report(t, n, GOOG_SCOPE_MUST_BE_ALONE);
597           }
598           if (t.getEnclosingFunction() != null) {
599             report(t, n, GOOG_SCOPE_MUST_BE_IN_GLOBAL_SCOPE);
600           }
601           if (n.getChildCount() != 2) {
602             // The goog.scope call should have exactly 1 parameter.  The first
603             // child is the "goog.scope" and the second should be the parameter.
604             report(t, n, GOOG_SCOPE_HAS_BAD_PARAMETERS);
605           } else {
606             Node anonymousFnNode = n.getSecondChild();
607             if (!anonymousFnNode.isFunction()
608                 || NodeUtil.getName(anonymousFnNode) != null
609                 || NodeUtil.getFunctionParameters(anonymousFnNode).hasChildren()) {
610               report(t, anonymousFnNode, GOOG_SCOPE_HAS_BAD_PARAMETERS);
611             } else {
612               scopeCalls.add(n);
613             }
614           }
615         }

617         @Override
618         public void visit(NodeTraversal t, Node n, Node parent) {
619           if (isCallToScopeMethod(n)) {
620             validateScopeCall(t, n, n.getParent());
621           }
622     
623           if (!inGoogScopeBody()) {
624             return;
625           }
626     
627           Token type = n.getType();
628           boolean isObjLitShorthand = type == Token.STRING_KEY && !n.hasChildren();
629           Var aliasVar = null;
630           if (type == Token.NAME || isObjLitShorthand) {
631             String name = n.getString();
632             Var lexicalVar = t.getScope().getVar(name);
633             if (lexicalVar != null && lexicalVar == aliases.get(name)) {
634               aliasVar = lexicalVar;
635             }
636           }
637     
638           if (isGoogScopeFunctionBody(t.getEnclosingFunction().getLastChild())) {
639             if (aliasVar != null && !isObjLitShorthand && NodeUtil.isLValue(n)) {
640               if (aliasVar.getNode() == n) {
641                 aliasDefinitionsInOrder.add(n);
642     
643                 // Return early, to ensure that we don't record a definition
644                 // twice.
645                 return;
646               } else {
647                 report(t, n, GOOG_SCOPE_ALIAS_REDEFINED, n.getString());
648               }
649             }
650     
651             if (type == Token.RETURN) {
652               report(t, n, GOOG_SCOPE_USES_RETURN);
653             } else if (type == Token.THIS) {
654               report(t, n, GOOG_SCOPE_REFERENCES_THIS);
655             } else if (type == Token.THROW) {
656               report(t, n, GOOG_SCOPE_USES_THROW);
657             }
658           }
659     
660           // Validate all descendent scopes of the goog.scope block.
661           if (inGoogScopeBody()) {
662             // Check if this name points to an alias.
663             if (aliasVar != null) {
664               // Note, to support the transitive case, it's important we don't
665               // clone aliasedNode here.  For example,
666               // var g = goog; var d = g.dom; d.createElement('DIV');
667               // The node in aliasedNode (which is "g") will be replaced in the
668               // changes pass above with "goog".  If we cloned here, we'd end up
669               // with <code>g.dom.createElement('DIV')</code>.
670               aliasUsages.add(new AliasedNode(aliasVar, n));
671             }
672     
673             // When we inject declarations, we duplicate jsdoc. Make sure
674             // we only process that jsdoc once.
675             JSDocInfo info = n.getJSDocInfo();
676             if (info != null && !injectedDecls.contains(n)) {
677               for (Node node : info.getTypeNodes()) {
678                 fixTypeNode(node);
679               }
680             }
681           }
682         }
683     
684         private void fixTypeNode(Node typeNode) {
685           if (typeNode.isString()) {
686             String name = typeNode.getString();
687             int endIndex = name.indexOf('.');
688             if (endIndex == -1) {
689               endIndex = name.length();
690             }
691             String baseName = name.substring(0, endIndex);
692             Var aliasVar = aliases.get(baseName);
693             if (aliasVar != null) {
694               aliasUsages.add(new AliasedTypeNode(aliasVar, typeNode));
695             }
696           }
697     
698           for (Node child = typeNode.getFirstChild(); child != null;
699                child = child.getNext()) {
700             fixTypeNode(child);
701           }
702         }
703       }
