28     import java.util.List;
29     import java.util.logging.Level;

32     /**
33      * Removes dead code from a parse tree. The kinds of dead code that this pass
34      * removes are:
35      *  - Any code following a return statement, such as the <code>alert</code>
36      *    call in: <code>if (x) { return; alert('unreachable'); }</code>.
37      *  - Statements that have no side effects, such as:
38      *    <code>a.b.MyClass.prototype.propertyName;</code> or <code>true;</code>.
39      *    That first kind of statement sometimes appears intentionally, so that
40      *    prototype properties can be annotated using JSDoc without actually
41      *    being initialized.
42      *
43      */
44     
45     // TODO(dimvar): Besides dead code after returns, this pass removes useless live
46     // code such as breaks/continues/returns and stms w/out side effects.
47     // These things don't require reachability info, consider making them their own
48     // pass or putting them in some other, more related pass.
49     
50     class UnreachableCodeElimination implements CompilerPass {
51       private static final Logger logger =
52         Logger.getLogger(UnreachableCodeElimination.class.getName());
53       private final AbstractCompiler compiler;
54       private final boolean removeNoOpStatements;
55       private boolean codeChanged;
56     
57       UnreachableCodeElimination(AbstractCompiler compiler,
58           boolean removeNoOpStatements) {
59         this.compiler = compiler;
60         this.removeNoOpStatements = removeNoOpStatements;
61       }
62     
63       @Override
64       public void process(Node externs, Node toplevel) {
65         NodeTraversal.traverseChangedFunctions(compiler, new FunctionCallback() {
66             @Override
67             public void enterFunction(AbstractCompiler compiler, Node root) {
68               // Computes the control flow graph.
69               ControlFlowAnalysis cfa =
70                   new ControlFlowAnalysis(compiler, false, false);
71               cfa.process(null, root);
72               ControlFlowGraph<Node> cfg = cfa.getCfg();
73               new GraphReachability<>(cfg)
74                   .compute(cfg.getEntry().getValue());
75               if (root.isFunction()) {
76                 root = root.getLastChild();
77               }
78               do {
79                 codeChanged = false;
80                 NodeTraversal.traverseEs6(compiler, root, new EliminationPass(cfg));
81               } while (codeChanged);
82             }
83           });
84       }
85     
86       private class EliminationPass extends AbstractShallowCallback {
87         private final ControlFlowGraph<Node> cfg;
88         private EliminationPass(ControlFlowGraph<Node> cfg) {
89           this.cfg = cfg;
90         }
91     
92         @Override
93         public void visit(NodeTraversal t, Node n, Node parent) {
94           if (parent == null || n.isFunction() || n.isScript()) {
95             return;
96           }
97           DiGraphNode<Node, Branch> gNode = cfg.getDirectedGraphNode(n);
98           if (gNode == null) { // Not in CFG.
99             return;
100           }
101           if (gNode.getAnnotation() != GraphReachability.REACHABLE ||
102               (removeNoOpStatements && !NodeUtil.mayHaveSideEffects(n, compiler))) {
103             removeDeadExprStatementSafely(n);
104             return;
105           }
106           tryRemoveUnconditionalBranching(n);
107         }
108     
109         /**
110          * Tries to remove n if it is an unconditional branch node (break, continue,
111          * or return) and the target of n is the same as the the follow of n.
112          * That is, if removing n preserves the control flow. Also if n targets
113          * another unconditional branch, this function will recursively try to
114          * remove the target branch as well. The reason why we want to cascade this
115          * removal is because we only run this pass once. If we have code such as
116          *
117          * break -> break -> break
118          *
119          * where all 3 breaks are useless, then the order of removal matters. When
120          * we first look at the first break, we see that it branches to the 2nd
121          * break. However, if we remove the last break, the 2nd break becomes
122          * useless and finally the first break becomes useless as well.
123          */
124         @SuppressWarnings("fallthrough")
125         private void tryRemoveUnconditionalBranching(Node n) {
126           /*
127            * For each unconditional branching control flow node, check to see
128            * if the ControlFlowAnalysis.computeFollowNode of that node is same as
129            * the branching target. If it is, the branch node is safe to be removed.
130            *
131            * This is not as clever as MinimizeExitPoints because it doesn't do any
132            * if-else conversion but it handles more complicated switch statements
133            * much more nicely.
134            */
135     
136           // If n is null the target is the end of the function, nothing to do.
137           if (n == null) {
138              return;
139           }
140     
141           DiGraphNode<Node, Branch> gNode = cfg.getDirectedGraphNode(n);
142     
143           if (gNode == null) {
144             return;
145           }
146     
147           switch (n.getType()) {
148             case RETURN:
149               if (n.hasChildren()) {
150                 break;
151               }
152             case BREAK:
153             case CONTINUE:
154               // We are looking for a control flow changing statement that always
155               // branches to the same node. If after removing it control still
156               // branches to the same node, it is safe to remove.
157               List<DiGraphEdge<Node, Branch>> outEdges = gNode.getOutEdges();
158               if (outEdges.size() == 1 &&
159                   // If there is a next node, this jump is not useless.
160                   (n.getNext() == null || n.getNext().isFunction())) {
161     
162                 Preconditions.checkState(
163                     outEdges.get(0).getValue() == Branch.UNCOND);
164                 Node fallThrough = computeFollowing(n);
165                 Node nextCfgNode = outEdges.get(0).getDestination().getValue();
166                 if (nextCfgNode == fallThrough && !inFinally(n.getParent(), n)) {
167                   removeNode(n);
168                 }
169               }
170           }
171         }
172     
173         private boolean inFinally(Node parent, Node child) {
174           if (parent == null || parent.isFunction()) {
175             return false;
176           } else if (NodeUtil.isTryFinallyNode(parent, child)) {
177             return true;
178           } else {
179             return inFinally(parent.getParent(), parent);
180           }
181         }
182     
183         private Node computeFollowing(Node n) {
184           Node next = ControlFlowAnalysis.computeFollowNode(n);
185           while (next != null && next.isBlock()) {
186             if (next.hasChildren()) {
187               next = next.getFirstChild();
188             } else {
189               next = computeFollowing(next);
190             }
191           }
192           return next;
193         }
194     
195         private void removeDeadExprStatementSafely(Node n) {
196           Node parent = n.getParent();
197           if (n.isEmpty() || (n.isBlock() && !n.hasChildren())) {
198             // Not always trivial to remove, let FoldConstants work its magic later.
199             return;
200           }
201     
202           // TODO(user): This is a problem with removeNoOpStatements.
203           // Every expression in a FOR-IN header looks side effect free on its own.
204           if (NodeUtil.isForIn(parent)) {
205             return;
206           }
207     
208           switch (n.getType()) {
209             // In the CFG, the only incoming edges the the DO node are from
210             // breaks/continues and the condition. The edge from the previous
211             // statement connects directly to the body of the DO.
212             //
213             // Removing an unreachable DO node is messy b/c it means we still have
214             // to execute one iteration of the body. If the DO's body has breaks in
215             // the middle, it can get even more tricky and code size might actually
216             // increase.
217             case DO:
218               return;
219     
220             case BLOCK:
221               // BLOCKs are used in several ways including wrapping CATCH
222               // blocks in TRYs
223               if (parent.isTry() && NodeUtil.isTryCatchNodeContainer(n)) {
224                 return;
225               }
226               break;
227     
228             case CATCH:
229               Node tryNode = parent.getParent();
230               NodeUtil.maybeAddFinally(tryNode);
231               break;
232           }
233     
234           if (n.isVar() && !n.getFirstChild().hasChildren()) {
235             // Very unlikely case, Consider this:
236             // File 1: {throw 1}
237             // File 2: {var x}
238             // The node var x is unreachable in the global scope.
239             // Before we remove the node, redeclareVarsInsideBranch
240             // would basically move var x to the beginning of File 2,
241             // which resulted in zero changes to the AST but triggered
242             // reportCodeChange().
243             // Instead, we should just ignore dead variable declarations.
244             return;
245           }
246     
247           removeNode(n);
248         }
249     
250         private void removeNode(Node n) {
251           codeChanged = true;
252           NodeUtil.redeclareVarsInsideBranch(n);
253           compiler.reportChangeToEnclosingScope(n);
254           if (logger.isLoggable(Level.FINE)) {
255             logger.fine("Removing " + n);
256           }
257           NodeUtil.removeChild(n.getParent(), n);
258         }
259       }
260     }
