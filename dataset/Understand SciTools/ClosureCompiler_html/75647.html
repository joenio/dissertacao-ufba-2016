40     package com.google.javascript.rhino.jstype;
41     
42     import static com.google.javascript.rhino.jstype.TernaryValue.FALSE;
43     import static com.google.javascript.rhino.jstype.TernaryValue.UNKNOWN;
44     
45     
46     /**
47      * Number type.
48      */
49     public class NumberType extends ValueType {
50       private static final long serialVersionUID = 1L;
51     
52       NumberType(JSTypeRegistry registry) {
53         super(registry);
54       }
55     
56       @Override
57       public boolean isNullable() {
58         return false;
59       }
60     
61       @Override
62       public TernaryValue testForEquality(JSType that) {
63         TernaryValue result = super.testForEquality(that);
64         if (result != null) {
65           return result;
66         }
67         if (that.isUnknownType() || that.isSubtype(
68                 getNativeType(JSTypeNative.OBJECT_NUMBER_STRING_BOOLEAN))) {
69           return UNKNOWN;
70         }
71         return FALSE;
72       }
73     
74       @Override
75       public boolean isNumberValueType() {
76         return true;
77       }
78     
79       @Override
80       public boolean matchesNumberContext() {
81         return true;
82       }
83     
84       @Override
85       public boolean matchesStringContext() {
86         return true;
87       }
88     
89       @Override
90       public boolean matchesObjectContext() {
91         // TODO(user): Revisit this for ES4, which is stricter.
92         return true;
93       }
94     
95       @Override
96       String toStringHelper(boolean forAnnotations) {
97         return getDisplayName();
98       }
99     
100       @Override
101       public String getDisplayName() {
102         return "number";
103       }
104     
105       @Override
106       public BooleanLiteralSet getPossibleToBooleanOutcomes() {
107         return BooleanLiteralSet.BOTH;
108       }
109     
110       @Override
111       public <T> T visit(Visitor<T> visitor) {
112         return visitor.caseNumberType();
113       }
114     
115       @Override
116       public JSType autoboxesTo() {
117         return getNativeType(JSTypeNative.NUMBER_OBJECT_TYPE);
118       }
119     }
