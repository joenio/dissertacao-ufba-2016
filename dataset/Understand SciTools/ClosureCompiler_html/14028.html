196       private final void parseAndTypeCheck(String externs, String js) {
197         // NOTE(dimvar): it's unusual that we run setUp for each test in a test
198         // method rather than once per test method. But the parent class creates a
199         // new compiler object at every setUp call, and we need that.
200         setUp();
201         final CompilerOptions options = compiler.getOptions();
202         options.setClosurePass(true);
203         options.setNewTypeInference(true);
204         options.setWarningLevel(
205             DiagnosticGroups.NEW_CHECK_TYPES_ALL_CHECKS, CheckLevel.WARNING);
206         if (this.reportUnknownTypes) {
207           options.setWarningLevel(
208               DiagnosticGroups.REPORT_UNKNOWN_TYPES, CheckLevel.WARNING);
209         }
210         compiler.init(
211             ImmutableList.of(SourceFile.fromCode("[externs]", externs)),
212             ImmutableList.of(SourceFile.fromCode("[testcode]", js)),
213             options);
214     
215         Node externsRoot = IR.block();
216         externsRoot.setIsSyntheticBlock(true);
217         externsRoot.addChildToFront(
218             compiler.getInput(new InputId("[externs]")).getAstRoot(compiler));
219         Node astRoot = IR.block();
220         astRoot.setIsSyntheticBlock(true);
221         astRoot.addChildToFront(
222             compiler.getInput(new InputId("[testcode]")).getAstRoot(compiler));
223     
224         assertEquals("parsing error: " + Joiner.on(", ").join(compiler.getErrors()),
225             0, compiler.getErrorCount());
226         assertEquals(
227             "parsing warning: " + Joiner.on(", ").join(compiler.getWarnings()), 0,
228             compiler.getWarningCount());
229     
230         // Create common parent of externs and ast; needed by Es6RewriteBlockScopedDeclaration.
231         Node block = IR.block(externsRoot, astRoot);
232         block.setIsSyntheticBlock(true);
233     
234         // Run ASTValidator
235         (new AstValidator(compiler)).validateRoot(block);
236     
237         DeclaredGlobalExternsOnWindow rewriteExterns =
238             new DeclaredGlobalExternsOnWindow(compiler);
239         passes.add(makePassFactory("globalExternsOnWindow", rewriteExterns));
240         ProcessClosurePrimitives closurePass =
241             new ProcessClosurePrimitives(compiler, null, CheckLevel.ERROR, false);
242         passes.add(makePassFactory("ProcessClosurePrimitives", closurePass));
243         if (options.getLanguageIn() == CompilerOptions.LanguageMode.ECMASCRIPT6_TYPED) {
244           passes.add(makePassFactory("convertEs6TypedToEs6",
245                   new Es6TypedToEs6Converter(compiler)));
246         }
247         if (options.getLanguageIn().isEs6OrHigher()) {
248           TranspilationPasses.addEs6EarlyPasses(passes);
249           TranspilationPasses.addEs6LatePasses(passes);
250         }
251         passes.add(makePassFactory("GlobalTypeInfo", compiler.getSymbolTable()));
252         passes.add(makePassFactory("NewTypeInference", new NewTypeInference(compiler)));
253     
254         PhaseOptimizer phaseopt = new PhaseOptimizer(compiler, null, null);
255         phaseopt.consume(passes);
256         phaseopt.process(externsRoot, astRoot);
257       }

268       private final void typeCheck(
269           String externs, String js, DiagnosticType... warningKinds) {
270         parseAndTypeCheck(externs, js);
271         JSError[] warnings = compiler.getWarnings();
272         JSError[] errors = compiler.getErrors();
273         String errorMessage = LINE_JOINER.join(
274             "Expected warning of type:",
275             "================================================================",
276             LINE_JOINER.join(warningKinds),
277             "================================================================",
278             "but found:",
279             "----------------------------------------------------------------",
280             LINE_JOINER.join(errors) + "\n" + LINE_JOINER.join(warnings),
281             "----------------------------------------------------------------\n");
282         assertEquals(
283             errorMessage + "Warning count",
284             warningKinds.length,
285             warnings.length + errors.length);
286         for (JSError warning : warnings) {
287           assertTrue(
288               "Wrong warning type\n" + errorMessage,
289               Arrays.asList(warningKinds).contains(warning.getType()));
290         }
291         for (JSError error : errors) {
292           assertTrue(
293               "Wrong warning type\n" + errorMessage,
294               Arrays.asList(warningKinds).contains(error.getType()));
295         }
296       }

298       // Used only in the cases where we provide extra details in the error message.
299       // Don't use in other cases.
300       // It is deliberately less general; no custom externs and only a single
301       // warning per test.
302       protected final void typeCheckMessageContents(
303           String js, DiagnosticType warningKind, String warningMsg) {
304         parseAndTypeCheck(DEFAULT_EXTERNS, js);
305         JSError[] warnings = compiler.getWarnings();
306         JSError[] errors = compiler.getErrors();
307         assertEquals(
308             "Expected no errors, but found:\n" + Arrays.toString(errors),
309             0, errors.length);
310         assertEquals(
311             "Expected one warning, but found:\n" + Arrays.toString(warnings),
312             1, warnings.length);
313         JSError warning = warnings[0];
314         assertEquals(LINE_JOINER.join(
315             "Wrong warning type",
316             "Expected warning of type:",
317             "================================================================",
318             warningKind.toString(),
319             "================================================================",
320             "but found:",
321             "----------------------------------------------------------------",
322             warning.toString(),
323             "----------------------------------------------------------------\n"),
324             warningKind, warning.getType());
325         assertEquals(LINE_JOINER.join(
326             "Wrong warning message",
327             "Expected:",
328             "================================================================",
329             warningMsg,
330             "================================================================",
331             "but found:",
332             "----------------------------------------------------------------",
333             warning.description,
334             "----------------------------------------------------------------\n"),
335             warningMsg, warning.description);
336       }
