17     package com.google.javascript.jscomp;
18     
19     import com.google.javascript.jscomp.PassFactory.HotSwapPassFactory;
20     
21     import java.util.List;
22     
23     /**
24      * Provides a single place to manage transpilation passes.
25      */
26     public class TranspilationPasses {
27       private TranspilationPasses() {}
28     
29       /**
30        * Adds all the early ES6 transpilation passes, which go before the Dart pass.
31        *
32        * <p>Includes ES6 features that are straightforward to transpile.
33        * We won't handle them natively in the rest of the compiler, so we always
34        * transpile them, even if the output language is also ES6.
35        */
36       public static void addEs6EarlyPasses(List<PassFactory> passes) {
37         passes.add(es6RewriteArrowFunction);
38         passes.add(es6RenameVariablesInParamLists);
39         passes.add(es6SplitVariableDeclarations);
40         passes.add(es6RewriteDestructuring);
41       }
42     
43       /**
44        * Adds all the late ES6 transpilation passes, which go after the Dart pass.
45        *
46        * <p>Includes ES6 features that are best handled natively by the compiler.
47        * As we convert more passes to handle these features, we will be moving the
48        * transpilation later in the compilation, and eventually only transpiling
49        * when the output is lower than ES6.
50        */
51       public static void addEs6LatePasses(List<PassFactory> passes) {
52         passes.add(es6ConvertSuper);
53         passes.add(es6ExtractClasses);
54         passes.add(convertEs6ToEs3);
55         passes.add(rewriteBlockScopedDeclaration);
56         passes.add(rewriteGenerators);
57         passes.add(rewritePolyfills);
58       }
59     
60     
61       static final HotSwapPassFactory es6ExtractClasses =
62           new HotSwapPassFactory("Es6ExtractClasses", true) {
63             @Override
64             protected HotSwapCompilerPass create(AbstractCompiler compiler) {
65               return new Es6ExtractClasses(compiler);
66             }
67           };
68     
69       static final HotSwapPassFactory es6RewriteDestructuring =
70           new HotSwapPassFactory("Es6RewriteDestructuring", true) {
71             @Override
72             protected HotSwapCompilerPass create(final AbstractCompiler compiler) {
73               return new Es6RewriteDestructuring(compiler);
74             }
75           };
76     
77       static final HotSwapPassFactory es6RenameVariablesInParamLists =
78           new HotSwapPassFactory("Es6RenameVariablesInParamLists", true) {
79             @Override
80             protected HotSwapCompilerPass create(final AbstractCompiler compiler) {
81               return new Es6RenameVariablesInParamLists(compiler);
82             }
83           };
84     
85       static final HotSwapPassFactory es6RewriteArrowFunction =
86           new HotSwapPassFactory("Es6RewriteArrowFunction", true) {
87             @Override
88             protected HotSwapCompilerPass create(final AbstractCompiler compiler) {
89               return new Es6RewriteArrowFunction(compiler);
90             }
91           };
92     
93       static final HotSwapPassFactory rewritePolyfills =
94           new HotSwapPassFactory("RewritePolyfills", true) {
95             @Override
96             protected HotSwapCompilerPass create(final AbstractCompiler compiler) {
97               return new RewritePolyfills(compiler);
98             }
99           };
100     
101       static final HotSwapPassFactory es6SplitVariableDeclarations =
102           new HotSwapPassFactory("Es6SplitVariableDeclarations", true) {
103             @Override
104             protected HotSwapCompilerPass create(final AbstractCompiler compiler) {
105               return new Es6SplitVariableDeclarations(compiler);
106             }
107           };
108     
109       static final HotSwapPassFactory es6ConvertSuper =
110           new HotSwapPassFactory("es6ConvertSuper", true) {
111         @Override
112         protected HotSwapCompilerPass create(final AbstractCompiler compiler) {
113           return new Es6ConvertSuper(compiler);
114         }
115       };
116     
117       /**
118        * Does the main ES6 to ES3 conversion.
119        * There are a few other passes which run before or after this one,
120        * to convert constructs which are not converted by this pass.
121        */
122       static final HotSwapPassFactory convertEs6ToEs3 =
123           new HotSwapPassFactory("convertEs6", true) {
124         @Override
125         protected HotSwapCompilerPass create(final AbstractCompiler compiler) {
126           return new Es6ToEs3Converter(compiler);
127         }
128       };
129     
130       static final HotSwapPassFactory rewriteBlockScopedDeclaration =
131           new HotSwapPassFactory("Es6RewriteBlockScopedDeclaration", true) {
132         @Override
133         protected HotSwapCompilerPass create(final AbstractCompiler compiler) {
134           return new Es6RewriteBlockScopedDeclaration(compiler);
135         }
136       };
137     
138       static final HotSwapPassFactory rewriteGenerators =
139           new HotSwapPassFactory("rewriteGenerators", true) {
140         @Override
141         protected HotSwapCompilerPass create(final AbstractCompiler compiler) {
142           return new Es6RewriteGenerators(compiler);
143         }
144       };
145     }
