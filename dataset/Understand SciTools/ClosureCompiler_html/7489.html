18     
19     import com.google.common.base.Predicate;
20     import com.google.javascript.jscomp.ControlFlowGraph.Branch;
21     import com.google.javascript.jscomp.NodeTraversal.ScopedCallback;
22     import com.google.javascript.jscomp.graph.DiGraph.DiGraphEdge;
23     import com.google.javascript.rhino.Node;
24     import com.google.javascript.rhino.jstype.FunctionType;
25     import com.google.javascript.rhino.jstype.JSType;
26     import com.google.javascript.rhino.jstype.JSTypeNative;
27     import com.google.javascript.rhino.jstype.TernaryValue;

29     /**
30      * Checks functions for missing return statements. Return statements are only
31      * expected for functions with return type information. Functions with empty
32      * bodies are ignored.
33      *
34      *
35      * NOTE(dimvar):
36      * Do not convert this pass to use TypeI. The pass is only used with the old type checker.
37      * The new type inference checks missing returns on its own.
38      */
39     class CheckMissingReturn implements ScopedCallback {
40     
41       static final DiagnosticType MISSING_RETURN_STATEMENT =
42           DiagnosticType.warning(
43               "JSC_MISSING_RETURN_STATEMENT",
44               "Missing return statement. Function expected to return {0}.");
45     
46       private final AbstractCompiler compiler;
47       private final CodingConvention convention;
48     
49       private static final Predicate<Node> IS_RETURN = new Predicate<Node>() {
50         @Override
51         public boolean apply(Node input) {
52           // Check for null because the control flow graph's implicit return node is
53           // represented by null, so this value might be input.
54           return input != null && input.isReturn();
55         }
56       };
57     
58       /* Skips all exception edges and impossible edges. */
59       private static final Predicate<DiGraphEdge<Node, ControlFlowGraph.Branch>>
60           GOES_THROUGH_TRUE_CONDITION_PREDICATE =
61             new Predicate<DiGraphEdge<Node, ControlFlowGraph.Branch>>() {
62         @Override
63         public boolean apply(DiGraphEdge<Node, ControlFlowGraph.Branch> input) {
64           // First skill all exceptions.
65           Branch branch = input.getValue();
66           if (branch == Branch.ON_EX) {
67             return false;
68           } else if (branch.isConditional()) {
69             Node condition = NodeUtil.getConditionExpression(
70                 input.getSource().getValue());
71             // TODO(user): We CAN make this bit smarter just looking at
72             // constants. We DO have a full blown ReverseAbstractInterupter and
73             // type system that can evaluate some impressions' boolean value but
74             // for now we will keep this pass lightweight.
75             if (condition != null) {
76               TernaryValue val = NodeUtil.getImpureBooleanValue(condition);
77               if (val != TernaryValue.UNKNOWN) {
78                 return val.toBoolean(true) == (Branch.ON_TRUE == branch);
79               }
80             }
81           }
82           return true;
83         }
84       };
85     
86       CheckMissingReturn(AbstractCompiler compiler) {
87         this.compiler = compiler;
88         this.convention = compiler.getCodingConvention();
89       }
90     
91       @Override
92       public void enterScope(NodeTraversal t) {
93         JSType returnType = explicitReturnExpected(t.getScopeRoot());
94         if (returnType == null) {
95           return;
96         }
97     
98         if (fastAllPathsReturnCheck(t.getControlFlowGraph())) {
99           return;
100         }
101     
102         CheckPathsBetweenNodes<Node, ControlFlowGraph.Branch> test =
103             new CheckPathsBetweenNodes<>(
104                 t.getControlFlowGraph(),
105                 t.getControlFlowGraph().getEntry(),
106                 t.getControlFlowGraph().getImplicitReturn(),
107                 IS_RETURN, GOES_THROUGH_TRUE_CONDITION_PREDICATE);
108     
109         if (!test.allPathsSatisfyPredicate()) {
110           compiler.report(
111               t.makeError(t.getScopeRoot(), MISSING_RETURN_STATEMENT, returnType.toString()));
112         }
113       }
114     
115       /**
116        * Fast check to see if all execution paths contain a return statement.
117        * May spuriously report that a return statement is missing.
118        *
119        * @return true if all paths return, converse not necessarily true
120        */
121       private boolean fastAllPathsReturnCheck(ControlFlowGraph<Node> cfg) {
122         for (DiGraphEdge<Node, Branch> s : cfg.getImplicitReturn().getInEdges()) {
123           Node n = s.getSource().getValue();
124           // NOTE(dimvar): it is possible to change ControlFlowAnalysis.java, so
125           // that the calls that always throw are treated in the same way as THROW
126           // in the CFG. Then, we would not need to use the coding convention here.
127           if (!n.isReturn() && !convention.isFunctionCallThatAlwaysThrows(n)) {
128             return false;
129           }
130         }
131         return true;
132       }
133     
134       @Override
135       public void exitScope(NodeTraversal t) {
136       }
137     
138       @Override
139       public boolean shouldTraverse(
140           NodeTraversal nodeTraversal, Node n, Node parent) {
141         return true;
142       }
143     
144       @Override
145       public void visit(NodeTraversal t, Node n, Node parent) {
146       }
147     
148       /**
149        * Determines if the given scope should explicitly return. All functions
150        * with non-void or non-unknown return types must have explicit returns.
151        *
152        * Exception: Constructors which specifically specify a return type are
153        * used to allow invocation without requiring the "new" keyword. They
154        * have an implicit return type. See unit tests.
155        *
156        * @return If a return type is expected, returns it. Otherwise, returns null.
157        */
158       private JSType explicitReturnExpected(Node scope) {
159         FunctionType scopeType = JSType.toMaybeFunctionType(scope.getJSType());
160     
161         if (scopeType == null) {
162           return null;
163         }
164     
165         if (isEmptyFunction(scope)) {
166           return null;
167         }
168     
169         if (scopeType.isConstructor()) {
170           return null;
171         }
172     
173         JSType returnType = scopeType.getReturnType();
174     
175         if (returnType == null) {
176           return null;
177         }
178     
179         if (!isVoidOrUnknown(returnType)) {
180           return returnType;
181         }
182     
183         return null;
184       }
185     
186       /**
187        * @return {@code true} if function represents a JavaScript function
188        *     with an empty body
189        */
190       private static boolean isEmptyFunction(Node function) {
191         return function.getChildCount() == 3 &&
192                !function.getSecondChild().getNext().hasChildren();
193       }
194     
195       /**
196        * @return {@code true} if returnType is void, unknown, or a union
197        *     containing void or unknown
198        */
199       private boolean isVoidOrUnknown(JSType returnType) {
200         final JSType voidType = compiler.getTypeIRegistry().getNativeType(JSTypeNative.VOID_TYPE);
201         return voidType.isSubtype(returnType);
202       }
203     }
