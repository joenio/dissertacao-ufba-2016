17     package com.google.javascript.jscomp.parsing;

19     import com.google.common.collect.ImmutableMap;

25     /**
26      * Configuration for the AST factory. Should be shared across AST creation
27      * for all files of a compilation process.
28      *
29      * @author nicksantos@google.com (Nick Santos)
30      */
31     public final class Config {
32     
33       /**
34        * Level of language strictness required for the input source code.
35        */
36       public enum StrictMode {
37         STRICT, SLOPPY;
38       }
39     
40       /** JavaScript mode */
41       public enum LanguageMode {
42     
43         // Note that minimumRequiredFor() relies on these being defined in order from fewest features to
44         // most features, and _STRICT versions should be supplied after unspecified strictness.
45         ECMASCRIPT3(FeatureSet.ES3, StrictMode.SLOPPY),
46         ECMASCRIPT5(FeatureSet.ES5, StrictMode.SLOPPY),
47         ECMASCRIPT5_STRICT(FeatureSet.ES5, StrictMode.STRICT),
48         ECMASCRIPT6(FeatureSet.ES6_MODULES, StrictMode.SLOPPY),
49         ECMASCRIPT6_STRICT(FeatureSet.ES6_MODULES, StrictMode.STRICT),
50         ECMASCRIPT7(FeatureSet.ES7_MODULES, StrictMode.STRICT),
51         ECMASCRIPT8(FeatureSet.ES8_MODULES, StrictMode.STRICT),
52         // TODO(bradfordcsmith): This should be renamed so it doesn't seem tied to es6
53         ECMASCRIPT6_TYPED(FeatureSet.TYPESCRIPT, StrictMode.STRICT),
54         ;
55     
56         public final FeatureSet featureSet;
57         public final StrictMode strictMode;
58     
59         LanguageMode(FeatureSet featureSet, StrictMode strictMode) {
60           this.featureSet = featureSet;
61           this.strictMode = strictMode;
62         }
63     
64         /**
65          * Returns the lowest {@link LanguageMode} that supports the specified feature.
66          */
67         public static LanguageMode minimumRequiredFor(FeatureSet.Feature feature) {
68           // relies on the LanguageMode enums being in the right order
69           for (LanguageMode mode : LanguageMode.values()) {
70             if (mode.featureSet.contains(feature)) {
71               return mode;
72             }
73           }
74           throw new IllegalStateException("No input language mode supports feature: " + feature);
75         }
76       }
77     
78       /**
79        * Whether to parse the descriptions of JsDoc comments.
80        */
81       public enum JsDocParsing {
82         TYPES_ONLY,
83         INCLUDE_DESCRIPTIONS_NO_WHITESPACE,
84         INCLUDE_DESCRIPTIONS_WITH_WHITESPACE;
85     
86         boolean shouldParseDescriptions() {
87           return this != TYPES_ONLY;
88         }
89       }
90       final JsDocParsing parseJsDocDocumentation;
91     
92       /**
93        * Whether to keep detailed source location information such as the exact length of every node.
94        */
95       public enum SourceLocationInformation {
96         DISCARD,
97         PRESERVE,
98       }
99       final SourceLocationInformation preserveDetailedSourceInfo;
100     
101       /**
102        * Whether to keep going after encountering a parse error.
103        */
104       public enum RunMode {
105         STOP_AFTER_ERROR,
106         KEEP_GOING,
107       }
108       final RunMode keepGoing;
109     
110       /**
111        * Recognized JSDoc annotations, mapped from their name to their internal
112        * representation.
113        */
114       final ImmutableMap<String, Annotation> annotationNames;
115     
116       /**
117        * Recognized names in a {@code @suppress} tag.
118        */
119       final ImmutableSet<String> suppressionNames;
120     
121       /**
122        * Accept ECMAScript5 syntax, such as getter/setter.
123        */
124       final LanguageMode languageMode;
125     
126       Config(Set<String> annotationWhitelist, Set<String> suppressionNames, LanguageMode languageMode) {
127         this(
128             annotationWhitelist,
129             JsDocParsing.TYPES_ONLY,
130             SourceLocationInformation.DISCARD,
131             RunMode.STOP_AFTER_ERROR,
132             suppressionNames,
133             languageMode);
134       }
135     
136       Config(
137           Set<String> annotationWhitelist,
138           JsDocParsing parseJsDocDocumentation,
139           SourceLocationInformation preserveDetailedSourceInfo,
140           RunMode keepGoing,
141           Set<String> suppressionNames,
142           LanguageMode languageMode) {
143         this.annotationNames = buildAnnotationNames(annotationWhitelist);
144         this.parseJsDocDocumentation = parseJsDocDocumentation;
145         this.preserveDetailedSourceInfo = preserveDetailedSourceInfo;
146         this.keepGoing = keepGoing;
147         this.suppressionNames = ImmutableSet.copyOf(suppressionNames);
148         this.languageMode = languageMode;
149       }
150     
151       /**
152        * Create the annotation names from the user-specified
153        * annotation whitelist.
154        */
155       private static ImmutableMap<String, Annotation> buildAnnotationNames(
156           Set<String> annotationWhitelist) {
157         ImmutableMap.Builder<String, Annotation> annotationBuilder =
158             ImmutableMap.builder();
159         annotationBuilder.putAll(Annotation.recognizedAnnotations);
160         for (String unrecognizedAnnotation : annotationWhitelist) {
161           if (!unrecognizedAnnotation.isEmpty()
162               && !Annotation.recognizedAnnotations.containsKey(
163                   unrecognizedAnnotation)) {
164             annotationBuilder.put(
165                 unrecognizedAnnotation, Annotation.NOT_IMPLEMENTED);
166           }
167         }
168         return annotationBuilder.build();
169       }
170     }
