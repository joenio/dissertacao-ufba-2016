31     /**
32      * Removes local variable assignments that are useless based on information from
33      * {@link LiveVariablesAnalysis}. If there is an assignment to variable
34      * {@code x} and {@code x} is dead after this assignment, we know that the
35      * current content of {@code x} will not be read and this assignment is useless.
36      *
37      */
38     class DeadAssignmentsElimination extends AbstractPostOrderCallback implements
39         CompilerPass, ScopedCallback {
40     
41       private final AbstractCompiler compiler;
42       private LiveVariablesAnalysis liveness;
43     
44       // Matches all assignment operators and increment/decrement operators.
45       // Does *not* match VAR initialization, since RemoveUnusedVariables
46       // will already remove variables that are initialized but unused.
47       private static final Predicate<Node> matchRemovableAssigns =
48           new Predicate<Node>() {
49         @Override
50         public boolean apply(Node n) {
51           return (NodeUtil.isAssignmentOp(n) &&
52                   n.getFirstChild().isName()) ||
53               n.isInc() || n.isDec();
54         }
55       };
56     
57       public DeadAssignmentsElimination(AbstractCompiler compiler) {
58         this.compiler = compiler;
59       }

62       public void process(Node externs, Node root) {
63         Preconditions.checkNotNull(externs);
64         Preconditions.checkNotNull(root);
65         NodeTraversal.traverseEs6(compiler, root, this);
66       }

72       @Override
73       public void exitScope(NodeTraversal t) {
74         // Do nothing on global scope / global blocks
75         if (t.inGlobalHoistScope()) {
76           return;
77         }
78     
79         Scope scope = t.getScope();
80         // Elevate all variable declarations up till the enclosing function scope
81         // so the liveness analysis has all variables for the process.
82         if (!scope.isFunctionScope()) {
83           for (Var var : scope.getVarIterable()) {
84             Preconditions.checkArgument(
85                 !var.isClass() && !var.isLet() && !var.isConst());
86             scope.getParent().declare(
87                 var.getName(), var.getNameNode(), var.getInput());
88           }
89         } else {
90           if (LiveVariablesAnalysis.MAX_VARIABLES_TO_ANALYZE <
91               scope.getVarCount()) {
92             return;
93           }
94     
95           // We are not going to do any dead assignment elimination in when there is
96           // at least one inner function because in most browsers, when there is a
97           // closure, ALL the variables are saved (escaped).
98           Node fnBlock = t.getScopeRoot().getLastChild();
99           if (NodeUtil.containsFunction(fnBlock)) {
100             return;
101           }
102     
103           // We don't do any dead assignment elimination if there are no assigns
104           // to eliminate. :)
105           if (!NodeUtil.has(fnBlock, matchRemovableAssigns,
106                   Predicates.<Node>alwaysTrue())) {
107             return;
108           }
109     
110           // Computes liveness information first.
111           ControlFlowGraph<Node> cfg = t.getControlFlowGraph();
112           liveness = new LiveVariablesAnalysis(cfg, scope, compiler);
113           liveness.analyze();
114           tryRemoveDeadAssignments(t, cfg);
115         }
116       }

122       /**
123        * Try to remove useless assignments from a control flow graph that has been
124        * annotated with liveness information.
125        *
126        * @param t The node traversal.
127        * @param cfg The control flow graph of the program annotated with liveness
128        *        information.
129        */
130       private void tryRemoveDeadAssignments(NodeTraversal t,
131           ControlFlowGraph<Node> cfg) {
132         Iterable<DiGraphNode<Node, Branch>> nodes = cfg.getDirectedGraphNodes();
133     
134         for (DiGraphNode<Node, Branch> cfgNode : nodes) {
135           FlowState<LiveVariableLattice> state =
136               cfgNode.getAnnotation();
137           Node n = cfgNode.getValue();
138           if (n == null) {
139             continue;
140           }
141           switch (n.getType()) {
142             case IF:
143             case WHILE:
144             case DO:
145               tryRemoveAssignment(t, NodeUtil.getConditionExpression(n), state);
146               continue;
147             case FOR:
148               if (!NodeUtil.isForIn(n)) {
149                 tryRemoveAssignment(
150                     t, NodeUtil.getConditionExpression(n), state);
151               }
152               continue;
153             case SWITCH:
154             case CASE:
155             case RETURN:
156               if (n.hasChildren()) {
157                 tryRemoveAssignment(t, n.getFirstChild(), state);
158               }
159               continue;
160             // TODO(user): case VAR: Remove var a=1;a=2;.....
161           }
162     
163           tryRemoveAssignment(t, n, state);
164         }
165       }
166     
167       private void tryRemoveAssignment(NodeTraversal t, Node n,
168           FlowState<LiveVariableLattice> state) {
169         tryRemoveAssignment(t, n, n, state);
170       }

172       /**
173        * Determines if any local variables are dead after the instruction {@code n}
174        * and are assigned within the subtree of {@code n}. Removes those assignments
175        * if there are any.
176        *
177        * @param n Target instruction.
178        * @param exprRoot The CFG node where the liveness information in state is
179        *     still correct.
180        * @param state The liveness information at {@code n}.
181        */
182       private void tryRemoveAssignment(NodeTraversal t, Node n, Node exprRoot,
183           FlowState<LiveVariableLattice> state) {
184     
185         Node parent = n.getParent();
186         boolean isDeclarationNode = NodeUtil.isNameDeclaration(parent);
187     
188         if (NodeUtil.isAssignmentOp(n) || n.isInc() || n.isDec() || isDeclarationNode) {
189     
190           Node lhs = isDeclarationNode ? n : n.getFirstChild();
191           Node rhs = NodeUtil.getRValueOfLValue(lhs);
192     
193           // Recurse first. Example: dead_x = dead_y = 1; We try to clean up dead_y
194           // first.
195           if (rhs != null) {
196             tryRemoveAssignment(t, rhs, exprRoot, state);
197             rhs = NodeUtil.getRValueOfLValue(lhs);
198           }
199     
200           // Multiple declarations should be processed from right-to-left to ensure side-effects
201           // are run in the correct order.
202           if (isDeclarationNode && lhs.getNext() != null) {
203             tryRemoveAssignment(t, lhs.getNext(), exprRoot, state);
204           }
205     
206           // Ignore declarations that don't initialize a value. Dead code removal will kill those nodes.
207           // Also ignore the var declaration if it's in a for-loop instantiation since there's not a
208           // safe place to move the side-effects.
209           if (isDeclarationNode && (rhs == null || parent.getParent().isFor())) {
210             return;
211           }
212     
213           Scope scope = t.getScope();
214           if (!lhs.isName()) {
215             return; // Not a local variable assignment.
216           }
217           String name = lhs.getString();
218           if (!scope.isDeclared(name, false)) {
219             return;
220           }
221           Var var = scope.getVar(name);
222     
223           if (liveness.getEscapedLocals().contains(var)) {
224             return; // Local variable that might be escaped due to closures.
225           }
226     
227           // If we have an identity assignment such as a=a, always remove it
228           // regardless of what the liveness results because it
229           // does not change the result afterward.
230           if (rhs != null &&
231               rhs.isName() &&
232               rhs.getString().equals(var.name) &&
233               n.isAssign()) {
234             n.removeChild(rhs);
235             n.getParent().replaceChild(n, rhs);
236             compiler.reportCodeChange();
237             return;
238           }
239     
240           if (state.getOut().isLive(var)) {
241             return; // Variable not dead.
242           }
243     
244           if (state.getIn().isLive(var) &&
245               isVariableStillLiveWithinExpression(n, exprRoot, var.name)) {
246             // The variable is killed here but it is also live before it.
247             // This is possible if we have say:
248             //    if (X = a && a = C) {..} ; .......; a = S;
249             // In this case we are safe to remove "a = C" because it is dead.
250             // However if we have:
251             //    if (a = C && X = a) {..} ; .......; a = S;
252             // removing "a = C" is NOT correct, although the live set at the node
253             // is exactly the same.
254             // TODO(user): We need more fine grain CFA or we need to keep track
255             // of GEN sets when we recurse here.
256             return;
257           }
258     
259           if (n.isAssign()) {
260             n.removeChild(rhs);
261             n.getParent().replaceChild(n, rhs);
262           } else if (NodeUtil.isAssignmentOp(n)) {
263             n.removeChild(rhs);
264             n.removeChild(lhs);
265             Node op = new Node(NodeUtil.getOpFromAssignmentOp(n), lhs, rhs);
266             parent.replaceChild(n, op);
267           } else if (n.isInc() || n.isDec()) {
268             if (parent.isExprResult()) {
269               parent.replaceChild(n,
270                   IR.voidNode(IR.number(0).srcref(n)));
271             } else if (n.isComma() && n != parent.getLastChild()) {
272               parent.removeChild(n);
273             } else if (parent.isFor() && !NodeUtil.isForIn(parent) &&
274                 NodeUtil.getConditionExpression(parent) != n) {
275               parent.replaceChild(n, IR.empty());
276             } else {
277               // Cannot replace x = a++ with x = a because that's not valid
278               // when a is not a number.
279               return;
280             }
281           } else if (isDeclarationNode) {
282             lhs.removeChild(rhs);
283             parent.getParent().addChildAfter(IR.exprResult(rhs), parent);
284             rhs.getParent().useSourceInfoFrom(rhs);
285           } else {
286             // Not reachable.
287             throw new IllegalStateException("Unknown statement");
288           }
289     
290           compiler.reportCodeChange();
291           return;
292         } else {
293           for (Node c = n.getFirstChild(); c != null;) {
294             Node next = c.getNext();
295             if (!ControlFlowGraph.isEnteringNewCfgNode(c)) {
296               tryRemoveAssignment(t, c, exprRoot, state);
297             }
298             c = next;
299           }
300           return;
301         }
302       }

304       /**
305        * Given a variable, node n in the tree and a sub-tree denoted by exprRoot as
306        * the root, this function returns true if there exists a read of that
307        * variable before a write to that variable that is on the right side of n.
308        *
309        * For example, suppose the node is x = 1:
310        *
311        * y = 1, x = 1; // false, there is no reads at all.
312        * y = 1, x = 1, print(x) // true, there is a read right of n.
313        * y = 1, x = 1, x = 2, print(x) // false, there is a read right of n but
314        *                               // it is after a write.
315        *
316        * @param n The current node we should look at.
317        * @param exprRoot The node
318        */
319       private boolean isVariableStillLiveWithinExpression(
320           Node n, Node exprRoot, String variable) {
321         while (n != exprRoot) {
322           VariableLiveness state = VariableLiveness.MAYBE_LIVE;
323           switch (n.getParent().getType()) {
324             case OR:
325             case AND:
326               // If the currently node is the first child of
327               // AND/OR, be conservative only consider the READs
328               // of the second operand.
329               if (n.getNext() != null) {
330                 state = isVariableReadBeforeKill(
331                     n.getNext(), variable);
332                 if (state == VariableLiveness.KILL) {
333                   state = VariableLiveness.MAYBE_LIVE;
334                 }
335               }
336               break;
337     
338             case HOOK:
339               // If current node is the condition, check each following
340               // branch, otherwise it is a conditional branch and the
341               // other branch can be ignored.
342               if (n.getNext() != null && n.getNext().getNext() != null) {
343                 state = checkHookBranchReadBeforeKill(
344                     n.getNext(), n.getNext().getNext(), variable);
345               }
346               break;
347     
348             default:
349               for (Node sibling = n.getNext(); sibling != null;
350                    sibling = sibling.getNext()) {
351                 state = isVariableReadBeforeKill(sibling, variable);
352                 if (state != VariableLiveness.MAYBE_LIVE) {
353                   break;
354                 }
355               }
356           }
357     
358           // If we see a READ or KILL there is no need to continue.
359           if (state == VariableLiveness.READ) {
360             return true;
361           } else if (state == VariableLiveness.KILL) {
362             return false;
363           }
364           n = n.getParent();
365         }
366         return false;
367       }

369       // The current liveness of the variable
370       private enum VariableLiveness {
371         MAYBE_LIVE, // May be still live in the current expression tree.
372         READ, // Known there is a read left of it.
373         KILL, // Known there is a write before any read.
374       }
375     
376       /**
377        * Give an expression and a variable. It returns READ, if the first
378        * reference of that variable is a read. It returns KILL, if the first
379        * reference of that variable is an assignment. It returns MAY_LIVE otherwise.
380        */
381       private VariableLiveness isVariableReadBeforeKill(
382           Node n, String variable) {
383         if (ControlFlowGraph.isEnteringNewCfgNode(n)) { // Not a FUNCTION
384           return VariableLiveness.MAYBE_LIVE;
385         }
386     
387         if (n.isName() && variable.equals(n.getString())) {
388           if (NodeUtil.isVarOrSimpleAssignLhs(n, n.getParent())) {
389             Preconditions.checkState(n.getParent().isAssign(), n.getParent());
390             // The expression to which the assignment is made is evaluated before
391             // the RHS is evaluated (normal left to right evaluation) but the KILL
392             // occurs after the RHS is evaluated.
393             Node rhs = n.getNext();
394             VariableLiveness state = isVariableReadBeforeKill(rhs, variable);
395             if (state == VariableLiveness.READ) {
396               return state;
397             }
398             return VariableLiveness.KILL;
399           } else {
400             return VariableLiveness.READ;
401           }
402         }
403     
404         switch (n.getType()) {
405           // Conditionals
406           case OR:
407           case AND:
408             VariableLiveness v1 = isVariableReadBeforeKill(
409               n.getFirstChild(), variable);
410             VariableLiveness v2 = isVariableReadBeforeKill(
411               n.getLastChild(), variable);
412             // With a AND/OR the first branch always runs, but the second is
413             // may not.
414             if (v1 != VariableLiveness.MAYBE_LIVE) {
415               return v1;
416             } else if (v2 == VariableLiveness.READ) {
417               return VariableLiveness.READ;
418             } else {
419               return VariableLiveness.MAYBE_LIVE;
420             }
421           case HOOK:
422             VariableLiveness first = isVariableReadBeforeKill(
423                 n.getFirstChild(), variable);
424             if (first != VariableLiveness.MAYBE_LIVE) {
425               return first;
426             }
427             return checkHookBranchReadBeforeKill(
428                 n.getSecondChild(), n.getLastChild(), variable);
429     
430           default:
431             // Expressions are evaluated left-right, depth first.
432             for (Node child = n.getFirstChild();
433                 child != null; child = child.getNext()) {
434               VariableLiveness state = isVariableReadBeforeKill(child, variable);
435               if (state != VariableLiveness.MAYBE_LIVE) {
436                 return state;
437               }
438             }
439         }
440     
441         return VariableLiveness.MAYBE_LIVE;
442       }
443     
444       private VariableLiveness checkHookBranchReadBeforeKill(
445           Node trueCase, Node falseCase, String variable) {
446         VariableLiveness v1 = isVariableReadBeforeKill(
447           trueCase, variable);
448         VariableLiveness v2 = isVariableReadBeforeKill(
449           falseCase, variable);
450         // With a hook it is unknown which branch will run, so
451         // we must be conservative.  A read by either is a READ, and
452         // a KILL is only considered if both KILL.
453         if (v1 == VariableLiveness.READ || v2 == VariableLiveness.READ) {
454           return VariableLiveness.READ;
455         } else if (v1 == VariableLiveness.KILL && v2 == VariableLiveness.KILL) {
456           return VariableLiveness.KILL;
457         } else {
458           return VariableLiveness.MAYBE_LIVE;
459         }
460       }
