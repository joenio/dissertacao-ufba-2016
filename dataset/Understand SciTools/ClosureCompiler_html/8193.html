31     /**
32      * Pass factories and meta-data for native Compiler passes.
33      *
34      * @author nicksantos@google.com (Nick Santos)
35      */
36     public abstract class PassConfig {
37     
38       // Used by subclasses in this package.
39       final CompilerOptions options;
40     
41       /**
42        * A memoized version of scopeCreator. It must be memoized so that
43        * we can make two separate passes over the AST, one for inferring types
44        * and one for checking types.
45        */
46       private MemoizedScopeCreator typedScopeCreator;
47     
48       /**
49        * This is the scope creator that {@code TypedScopeCreator} delegates to.
50        */
51       private TypedScopeCreator internalScopeCreator;
52     
53       /** The global typed scope. */
54       TypedScope topScope = null;
55     
56       public PassConfig(CompilerOptions options) {
57         this.options = options;
58       }
59     
60       /**
61        * Regenerates the top scope from scratch.
62        *
63        * @param compiler The compiler for which the global scope is regenerated.
64        * @param root The root of the AST.
65        */
66       void regenerateGlobalTypedScope(AbstractCompiler compiler, Node root) {
67         internalScopeCreator = new TypedScopeCreator(compiler);
68         typedScopeCreator = new MemoizedScopeCreator(internalScopeCreator);
69         topScope = typedScopeCreator.createScope(root, null);
70       }
71     
72       void clearTypedScope() {
73         internalScopeCreator = null;
74         typedScopeCreator = null;
75         topScope = null;
76       }
77     
78       /**
79        * Regenerates the top scope potentially only for a sub-tree of AST and then
80        * copies information for the old global scope.
81        *
82        * @param compiler The compiler for which the global scope is generated.
83        * @param scriptRoot The root of the AST used to generate global scope.
84        */
85       void patchGlobalTypedScope(AbstractCompiler compiler, Node scriptRoot) {
86         Preconditions.checkNotNull(internalScopeCreator);
87         internalScopeCreator.patchGlobalScope(topScope, scriptRoot);
88       }
89     
90       /**
91        * Gets the scope creator for typed scopes.
92        */
93       MemoizedScopeCreator getTypedScopeCreator() {
94         return typedScopeCreator;
95       }
96     
97       /**
98        * Gets the global scope, with type information.
99        */
100       TypedScope getTopScope() {
101         return topScope;
102       }
103     
104       /**
105        * Gets additional checking passes that are run always, even in "whitespace only" mode.
106        * For very specific cases where processing is required even in a mode which is intended
107        * not to have any processing - specifically introduced to support goog.module() usage.
108        */
109       protected List<PassFactory> getWhitespaceOnlyPasses() {
110         return Collections.emptyList();
111       }
112     
113       /**
114        * Gets the checking passes to run.
115        *
116        * Checking passes revolve around emitting warnings and errors.
117        * They also may include pre-processor passes needed to do
118        * error analysis more effectively.
119        *
120        * Clients that only want to analyze code (like IDEs) and not emit
121        * code will only run checks and not optimizations.
122        */
123       protected abstract List<PassFactory> getChecks();
124     
125       /**
126        * Gets the optimization passes to run.
127        *
128        * Optimization passes revolve around producing smaller and faster code.
129        * They should always run after checking passes.
130        */
131       protected abstract List<PassFactory> getOptimizations();
132     
133       /**
134        * Gets a graph of the passes run. For debugging.
135        */
136       GraphvizGraph getPassGraph() {
137         LinkedDirectedGraph<String, String> graph =
138             LinkedDirectedGraph.createWithoutAnnotations();
139         Iterable<PassFactory> allPasses =
140             Iterables.concat(getChecks(), getOptimizations());
141         String lastPass = null;
142         String loopStart = null;
143         for (PassFactory pass : allPasses) {
144           String passName = pass.getName();
145           int i = 1;
146           while (graph.hasNode(passName)) {
147             passName = pass.getName() + (i++);
148           }
149           graph.createNode(passName);
150     
151           if (loopStart == null && !pass.isOneTimePass()) {
152             loopStart = passName;
153           } else if (loopStart != null && pass.isOneTimePass()) {
154             graph.connect(lastPass, "loop", loopStart);
155             loopStart = null;
156           }
157     
158           if (lastPass != null) {
159             graph.connect(lastPass, "", passName);
160           }
161           lastPass = passName;
162         }
163         return graph;
164       }
165     
166       /**
167        * Create a type inference pass.
168        */
169       final TypeInferencePass makeTypeInference(AbstractCompiler compiler) {
170         return new TypeInferencePass(
171             compiler, compiler.getReverseAbstractInterpreter(),
172             topScope, typedScopeCreator);
173       }
174     
175       static final InferJSDocInfo makeInferJsDocInfo(AbstractCompiler compiler) {
176         return new InferJSDocInfo(compiler);
177       }
178     
179       /**
180        * Create a type-checking pass.
181        */
182       final TypeCheck makeTypeCheck(AbstractCompiler compiler) {
183         return new TypeCheck(
184             compiler,
185             compiler.getReverseAbstractInterpreter(),
186             compiler.getTypeRegistry(),
187             topScope,
188             typedScopeCreator,
189             options.reportMissingOverride)
190             .reportMissingProperties(options.enables(
191                 DiagnosticGroup.forType(TypeCheck.INEXISTENT_PROPERTY)));
192       }
193     
194       /**
195        * Insert the given pass factory before the factory of the given name.
196        */
197       static final void addPassFactoryBefore(
198           List<PassFactory> factoryList, PassFactory factory, String passName) {
199         factoryList.add(
200             findPassIndexByName(factoryList, passName), factory);
201       }
202     
203       /**
204        * Find a pass factory with the same name as the given one, and replace it.
205        */
206       static final void replacePassFactory(
207           List<PassFactory> factoryList, PassFactory factory) {
208         factoryList.set(
209             findPassIndexByName(factoryList, factory.getName()), factory);
210       }
211     
212       /**
213        * Throws an exception if no pass with the given name exists.
214        */
215       private static int findPassIndexByName(
216           List<PassFactory> factoryList, String name) {
217         for (int i = 0; i < factoryList.size(); i++) {
218           if (factoryList.get(i).getName().equals(name)) {
219             return i;
220           }
221         }
222     
223         throw new IllegalArgumentException(
224             "No factory named '" + name + "' in the factory list");
225       }
226     
227       /**
228        * Find the first pass provider that does not have a delegate.
229        */
230       final PassConfig getBasePassConfig() {
231         PassConfig current = this;
232         while (current instanceof PassConfigDelegate) {
233           current = ((PassConfigDelegate) current).delegate;
234         }
235         return current;
236       }

239        * Get intermediate state for a running pass config, so it can
240        * be paused and started again later.
241        */
242       protected abstract State getIntermediateState();

244       /**
245        * An implementation of PassConfig that just proxies all its method calls
246        * into an inner class.
247        */
248       static class PassConfigDelegate extends PassConfig {
249     
250         private final PassConfig delegate;
251     
252         PassConfigDelegate(PassConfig delegate) {
253           super(delegate.options);
254           this.delegate = delegate;
255         }
256     
257         @Override
258         protected List<PassFactory> getWhitespaceOnlyPasses() {
259           return delegate.getWhitespaceOnlyPasses();
260         }
261     
262         @Override protected List<PassFactory> getChecks() {
263           return delegate.getChecks();
264         }
265     
266         @Override protected List<PassFactory> getOptimizations() {
267           return delegate.getOptimizations();
268         }
269     
270         @Override MemoizedScopeCreator getTypedScopeCreator() {
271           return delegate.getTypedScopeCreator();
272         }
273     
274         @Override TypedScope getTopScope() {
275           return delegate.getTopScope();
276         }
277     
278         @Override protected State getIntermediateState() {
279           return delegate.getIntermediateState();
280         }
281       }

283       /**
284        * Intermediate state for a running pass configuration.
285        */
286       public static class State implements Serializable {
287         private static final long serialVersionUID = 1L;
288     
289         final Map<String, Integer> cssNames;
290         final VariableMap variableMap;
291         final VariableMap propertyMap;
292         final VariableMap anonymousFunctionNameMap;
293         final VariableMap stringMap;
294         final FunctionNames functionNames;
295         final String idGeneratorMap;
296     
297         public State(Map<String, Integer> cssNames, Set<String> exportedNames,
298             CrossModuleMethodMotion.IdGenerator crossModuleIdGenerator,
299             VariableMap variableMap, VariableMap propertyMap,
300             VariableMap anonymousFunctionNameMap,
301             VariableMap stringMap, FunctionNames functionNames,
302             String idGeneratorMap) {
303           this.cssNames = cssNames;
304           this.variableMap = variableMap;
305           this.propertyMap = propertyMap;
306           this.anonymousFunctionNameMap = anonymousFunctionNameMap;
307           this.stringMap = stringMap;
308           this.idGeneratorMap = idGeneratorMap;
309           this.functionNames = functionNames;
310         }
311       }
