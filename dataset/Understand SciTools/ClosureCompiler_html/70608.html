44     /**
45      * Check to ensure there exists a path to dispose of each eventful object
46      * created.
47      *
48      * <p>An eventful class is any class that derives from goog.events.EventHandler
49      * or (in aggressive mode) is disposable and disposes of an eventful class when
50      * it is disposed (see http://research.google.com/pubs/pub40738.html).
51      *
52      * <p>This pass is heuristic based and should not be used for any check
53      * of pass/fail testing. The pass traverses the AST and marks as errors
54      * cases where an eventful object is allocated but a dispose call is not found.
55      * It only tracks eventful objects that has a easily identifiable static name,
56      * i.e., objects assigned to arrays, returned from functions or captured in
57      * closures are not considered. It simply tries to see if there exists a call to
58      * a dispose method in the AST for every object seen as eventful.
59      *
60      * <p>This compiler pass uses the inferred types and hence either type checking or
61      * type inference needs to be enabled.
62      *
63      *
64      */
65      // TODO(tbreisacher): Find out if anyone is still using this pass. Delete if not.
66      // TODO(user): Pass needs to be updated for listenable interfaces.
67     public final class CheckEventfulObjectDisposal implements CompilerPass {

163       /**
164        * Add a new call that is used to dispose an JS object.
165        * @param functionOrMethodName The name or suffix of a function or method
166        *  that disposes of/registers an object as disposable
167        * @param argumentsThatAreDisposed An array of integers (ideally sorted) that
168        *   specifies the arguments of the function being disposed
169        */
170       private void addDisposeCall(String functionOrMethodName,
171           List<Integer> argumentsThatAreDisposed) {
172         String potentiallyTypeName, propertyName;
173         JSType objectType = null;
174     
175         int lastPeriod = functionOrMethodName.lastIndexOf('.');
176         // If function call has a period it is potentially a method function.
177         if (lastPeriod >= 0) {
178           potentiallyTypeName = functionOrMethodName.substring(0, lastPeriod).
179             replaceFirst(".prototype$", "");
180           propertyName = functionOrMethodName.substring(lastPeriod);
181           objectType = this.typeRegistry.getType(potentiallyTypeName);
182         } else {
183           propertyName = functionOrMethodName;
184         }
185     
186         // Find or create property map for object type
187         Map<String, List<Integer>> map = this.disposeCalls.get(objectType);
188         if (map == null) {
189           map = new HashMap<>();
190           this.disposeCalls.put(objectType, map);
191         }
192     
193         /*
194          * If this is a static function call store the full function name,
195          * else only the method of the object.
196          */
197         if (objectType == null) {
198           map.put(functionOrMethodName, argumentsThatAreDisposed);
199         } else {
200           map.put(propertyName, argumentsThatAreDisposed);
201         }
202       }

205       /*
206        * Initialize disposeMethods map with calls to dispose calls.
207        */
208       private void initializeDisposeMethodsMap() {
209         this.disposeCalls = new HashMap<>();
210     
211         /*
212          * Initialize dispose calls map. Checks for:
213          *    - Y.registerDisposable(X)
214          *      (Y has to be of type goog.Disposable)
215          *    - X.dispose()
216          *    - goog.dispose(X)
217          *    - goog.disposeAll(X...)
218          *    - X.removeAll() (X is of type goog.events.EventHandler)
219          *    - goog.array.extend(_, X...)
220          *    - Y.add(X...) or Y.push(X)
221          */
222         this.addDisposeCall("goog.array.extend", ImmutableList.of(DISPOSE_ALL));
223         this.addDisposeCall("goog.dispose", ImmutableList.of(0));
224         this.addDisposeCall("goog.Disposable.registerDisposable", ImmutableList.of(0));
225         this.addDisposeCall("goog.disposeAll", ImmutableList.of(DISPOSE_ALL));
226         this.addDisposeCall("goog.events.EventHandler.removeAll", ImmutableList.of(DISPOSE_SELF));
227         this.addDisposeCall(".dispose", ImmutableList.of(DISPOSE_SELF));
228         this.addDisposeCall(".push", ImmutableList.of(0));
229         this.addDisposeCall(".add", ImmutableList.of(DISPOSE_SELF));
230       }

257       /**
258        * Determines if thisType is possibly a subtype of thatType.
259        *
260        *  <p>It differs from isSubtype only in that thisType gets expanded
261        *  if it is a union.
262        *
263        *  <p>Common case targeted is a function returning an eventful object
264        *  that may also return a null.
265        *
266        *  @param thisType the JSType being tested
267        *  @param thatType the JSType that is possibly a base of thisType
268        *  @return whether thisType is possibly subtype of thatType
269        */
270       private static boolean isPossiblySubtype(JSType thisType, JSType thatType) {
271         if (thisType == null) {
272           return false;
273         }
274     
275         JSType type = thisType;
276     
277         if (type.isUnionType()) {
278           for (JSType alternate : type.toMaybeUnionType().getAlternates()) {
279             if (alternate.isSubtype(thatType)) {
280               return true;
281             }
282           }
283         } else {
284           if (type.isSubtype(thatType)) {
285             return true;
286           }
287         }
288     
289         return false;
290       }

296       /*
297        * Create a unique identification string for Node n, or null if function
298        * called with invalid argument.
299        *
300        * This function is basically used to distinguish between:
301        *   A.B = function() {
302        *     this.eh = new ...
303        *   }
304        * and
305        *   C.D = function() {
306        *     this.eh = new ...
307        *   }
308        *
309        * As well as
310        *   A.B = function() {
311        *     var eh = new ...
312        *   }
313        * and
314        *   C.D = function() {
315        *     var eh = new ...
316        *   }
317        *
318        * Warning: Inheritance is not currently handled.
319        */
320       private static String generateKey(NodeTraversal t, Node n,
321           boolean noLocalVariables) {
322         if (n == null) {
323           return null;
324         }
325         String key;
326     
327         Node scopeNode = t.getScopeRoot();
328     
329         if (n.isName()) {
330           if (noLocalVariables) {
331             return null;
332           }
333           key = n.getQualifiedName();
334     
335           if (scopeNode.isFunction()) {
336             JSType parentScopeType = t.getTypedScope().getParentScope().getTypeOfThis();
337             /*
338              * If the locally defined variable is defined within a function, use
339              * the function name to create ID.
340              */
341             if (!parentScopeType.isGlobalThisType()) {
342               key = parentScopeType + "~" + key;
343             }
344             key = NodeUtil.getName(scopeNode) + "=" + key;
345           }
346         } else {
347           /*
348            * Only handle cases such as a.b.c.X and not cases where the
349            * eventful object is stored in an array or uses a function to
350            * determine the index.
351            *
352            * Note: Inheritance changes the name that should be returned here
353            */
354           if (!n.isQualifiedName()) {
355             return null;
356           }
357           key = n.getQualifiedName();
358     
359           /*
360            * If it is not a simple variable and doesn't use this, then we assume
361            * global variable.
362            */
363           Node base = getBase(n);
364           if (base != null && base.isThis()) {
365             if (base.getJSType().isUnknownType()) {
366               // Handle anonymous function created in constructor:
367               //
368               // /**
369               // * @extends {goog.SubDisposable}
370               // * @constructor */
371               // speel.Person = function() {
372               //  this.run = function() {
373               //    this.eh = new goog.events.EventHandler();
374               //  }
375               //};
376               key = t.getTypedScope().getParentScope().getTypeOfThis() + "~" + key;
377             } else {
378               if (n.getFirstChild() == null) {
379                 key = base.getJSType() + "=" + key;
380               } else {
381                 ObjectType objectType =
382                     ObjectType.cast(dereference(n.getFirstChild().getJSType()));
383                 if (objectType == null) {
384                   return null;
385                 }
386     
387                 ObjectType hObjT = objectType;
388                 String propertyName = n.getLastChild().getString();
389     
390                 while (objectType != null) {
391                   hObjT = objectType;
392                   objectType = objectType.getImplicitPrototype();
393                   if (objectType == null) {
394                     break;
395                   }
396                   if (objectType.getDisplayName().endsWith("prototype")) {
397                     continue;
398                   }
399                   if (!objectType.getPropertyNames().contains(propertyName)) {
400                     break;
401                   }
402                 }
403                 key = hObjT + "=" + key;
404               }
405             }
406           }
407         }
408     
409         return key;
410       }

412       @Override
413       public void process(Node externs, Node root) {
414         // This pass should not have gotten added in this case
415         Preconditions.checkArgument(checkingPolicy != DisposalCheckingPolicy.OFF);
416     
417         // Initialize types
418         googDisposableInterfaceType = this.typeRegistry.getType(DISPOSABLE_INTERFACE_TYPE_NAME);
419         googEventsEventHandlerType = this.typeRegistry.getType(EVENT_HANDLER_TYPE_NAME);
420     
421         /*
422          * Required types not found therefore the kind of pattern considered
423          * will not be found.
424          */
425         if (googEventsEventHandlerType == null || googDisposableInterfaceType == null) {
426           return;
427         }
428     
429         // Seed list of disposable stype
430         eventfulTypes = new HashSet<>();
431         eventfulTypes.add(googEventsEventHandlerType);
432     
433         // Construct eventizer graph
434         if (checkingPolicy == DisposalCheckingPolicy.AGGRESSIVE) {
435           NodeTraversal.traverseTyped(compiler, root, new ComputeEventizeTraversal());
436           computeEventful();
437         }
438     
439         /*
440          * eventfulObjectMap maps a eventful object's "name" to its corresponding
441          * EventfulObjectState which tracks the state (allocated, disposed of)
442          * as well as allocation site.
443          */
444         eventfulObjectMap = new HashMap<>();
445     
446         // Traverse tree
447         NodeTraversal.traverseTyped(compiler, root, new Traversal());
448     
449         /*
450          * Scan eventfulObjectMap for allocated eventful objects that
451          * had no dispose calls.
452          */
453         for (EventfulObjectState e : eventfulObjectMap.values()) {
454           Node n = e.allocationSite;
455           if (e.seen == SeenType.ALLOCATED) {
456             compiler.report(JSError.make(n, EVENTFUL_OBJECT_NOT_DISPOSED));
457           } else if (e.seen == SeenType.ALLOCATED_LOCALLY
458               && checkingPolicy == DisposalCheckingPolicy.AGGRESSIVE) {
459             compiler.report(JSError.make(n, EVENTFUL_OBJECT_PURELY_LOCAL));
460           }
461         }
462       }

464       private void computeEventful() {
465         /*
466          * Topological order of Eventize DAG
467          */
468         String[] order = new String[eventizes.keySet().size()];
469     
470         /*
471          * Perform topological sort
472          */
473         int white = 0, gray = 1, black = 2;
474         int last = eventizes.keySet().size() - 1;
475         Map<String, Integer> color = new HashMap<>();
476         Stack<String> dfsStack = new Stack<>();
477     
478         /*
479          * Initialize color.
480          * Some types are only on one or the other side of the
481          * inference.
482          */
483         for (Map.Entry<String, Set<String>> eventizesEntry : Multimaps.asMap(eventizes).entrySet()) {
484           color.put(eventizesEntry.getKey(), white);
485           for (String s : eventizesEntry.getValue()) {
486             color.put(s, white);
487           }
488         }
489     
490         int indx = 0;
491         for (String s : eventizes.keySet()) {
492           dfsStack.push(s);
493           while (!dfsStack.isEmpty()) {
494             String top = dfsStack.pop();
495             if (!color.containsKey(top)) {
496               continue;
497             }
498             if (color.get(top) == white) {
499               color.put(top, gray);
500               dfsStack.push(top);
501               // for v in Adj[s]
502               if (eventizes.containsKey(top)) {
503                 for (String v : eventizes.get(top)) {
504                   if (color.get(v) == white) {
505                     dfsStack.push(v);
506                   }
507                 }
508               }
509             } else if (color.get(top) == gray && eventizes.containsKey(top)) {
510               order[last - indx] = top;
511               ++indx;
512               color.put(top, black);
513             }
514           }
515         }
516     
517         /*
518          * Propagate eventfulness by iterating in topological order
519          */
520         for (String s : order) {
521           if (eventfulTypes.contains(typeRegistry.getType(s))) {
522             for (String v : eventizes.get(s)) {
523               eventfulTypes.add((JSType) typeRegistry.getType(v));
524             }
525           }
526         }
527       }

529       private JSType maybeReturnDisposedType(Node n, boolean checkDispose) {
530         /*
531          * Checks for:
532          *    - Y.registerDisposable(X)
533          *      (Y has to be of type goog.Disposable)
534          *    - X.dispose()
535          *    - goog.dispose(X)
536          *    - X.removeAll() (X is of type goog.events.EventHandler)
537          *    - <array>.property(X) or Y.push(X)
538          */
539         Node first = n.getFirstChild();
540     
541         if (first == null || !first.isQualifiedName()) {
542           return null;
543         }
544         String property = first.getQualifiedName();
545     
546         if (property.endsWith(".registerDisposable"))  {
547           /*
548            *  Ensure object is of type disposable
549            */
550           Node base = first.getFirstChild();
551           JSType baseType = base.getJSType();
552     
553           if (baseType == null
554               || !isPossiblySubtype(baseType, googDisposableInterfaceType)) {
555             return null;
556           }
557     
558           return n.getLastChild().getJSType();
559         }
560     
561         if (checkDispose) {
562           if (property.equals("goog.dispose")) {
563             return n.getLastChild().getJSType();
564           }
565           if (property.endsWith(".dispose")) {
566             /*
567              * n -> call
568              *   n.firstChild -> "dispose"
569              *   n.firstChild.firstChild -> object
570              */
571             return n.getFirstFirstChild().getJSType();
572           }
573         }
574     
575         return null;
576       }

578       /*
579        * Compute eventize relationship graph.
580        */
581       private class ComputeEventizeTraversal extends AbstractPostOrderCallback
582           implements ScopedCallback {
583     
584         /*
585          * Keep track of whether in the constructor or disposal scope.
586          */
587         Stack<Boolean> isConstructorStack;
588         Stack<Boolean> isDisposalStack;
589     
590     
591         public ComputeEventizeTraversal() {
592           isConstructorStack = new Stack<>();
593           isDisposalStack = new Stack<>();
594           eventizes = HashMultimap.create();
595         }
596     
597         private Boolean inConstructorScope() {
598           Preconditions.checkNotNull(isConstructorStack);
599           if (!isDisposalStack.isEmpty()) {
600             return isConstructorStack.peek();
601           }
602           return null;
603         }
604     
605         private Boolean inDisposalScope() {
606           Preconditions.checkNotNull(isDisposalStack);
607           if (!isDisposalStack.isEmpty()) {
608             return isDisposalStack.peek();
609           }
610           return null;
611         }
612     
613         /*
614          * Filter types not interested in for eventize graph
615          */
616         private boolean collectorFilterType(JSType type) {
617           if (type == null) {
618             return true;
619           }
620     
621           return type.isEmptyType() || type.isUnknownType()
622               || !isPossiblySubtype(type, googDisposableInterfaceType);
623         }
624     
625         /*
626          * Log that thisType eventizes thatType.
627          */
628         private void addEventize(JSType thisType, JSType thatType) {
629           if (collectorFilterType(thisType)
630               || collectorFilterType(thatType)
631               || thisType.isEquivalentTo(thatType)) {
632             return;
633           }
634     
635           String className = thisType.getDisplayName();
636           if (thatType.isUnionType()) {
637             UnionType ut = thatType.toMaybeUnionType();
638             for (JSType type : ut.getAlternates()) {
639               if (type.isObject()) {
640                 addEventizeClass(className, type);
641               }
642             }
643           } else {
644             addEventizeClass(className, thatType);
645           }
646         }
647     
648         private void addEventizeClass(String className, JSType thatType) {
649           String propertyJsTypeName = thatType.getDisplayName();
650           eventizes.put(propertyJsTypeName, className);
651         }

653         @Override
654         public void enterScope(NodeTraversal t) {
655           Node n = t.getScopeRoot();
656           boolean isConstructor = false;
657           boolean isInDisposal = false;
658           String functionName = null;
659     
660           /*
661            * TypedScope entered is a function definition
662            */
663           if (n.isFunction()) {
664             functionName = NodeUtil.getName(n);
665     
666             /*
667              *  Skip anonymous functions
668              */
669             if (functionName != null) {
670     
671               JSDocInfo jsDocInfo = NodeUtil.getBestJSDocInfo(n);
672               if (jsDocInfo != null) {
673                 /*
674                  *  Record constructor of a type
675                  */
676                 if (jsDocInfo.isConstructor()) {
677                   isConstructor = true;
678     
679                   /*
680                    * Initialize eventizes relationship
681                    */
682                   if (t.getTypedScope() != null
683                       && t.getTypedScope().getTypeOfThis() != null) {
684                     ObjectType objectType = ObjectType.cast(t.getTypedScope()
685                         .getTypeOfThis().dereference());
686     
687                     /*
688                      * Eventize due to inheritance
689                      */
690     
691                     while (objectType != null) {
692                       objectType = objectType.getImplicitPrototype();
693                       if (objectType == null) {
694                         break;
695                       }
696     
697                       if (objectType.getDisplayName().endsWith("prototype")) {
698                         continue;
699                       }
700     
701                       addEventize((JSType) compiler.getTypeIRegistry().getType(functionName),
702                           objectType);
703     
704                       /*
705                        * Don't add transitive eventize edges here, it will be
706                        * taken care of in computeEventful
707                        */
708                       break;
709                     }
710                   }
711                 }
712               }
713     
714               /*
715                *  Indicate within a disposeInternal member
716                */
717               if (functionName.endsWith(".disposeInternal")) {
718                 isInDisposal = true;
719               }
720             }
721     
722             isConstructorStack.push(isConstructor);
723             isDisposalStack.push(isInDisposal);
724           } else {
725             isConstructorStack.push(inConstructorScope());
726             isDisposalStack.push(inDisposalScope());
727           }
728         }

731         public void exitScope(NodeTraversal t) {
732           isConstructorStack.pop();
733           isDisposalStack.pop();
734         }

736         /*
737          * Is the current node a call to goog.events.unlisten
738          */
739         private void isGoogEventsUnlisten(Node n) {
740           Preconditions.checkArgument(n.getChildCount() > 3);
741     
742           Node listener = n.getChildAtIndex(3);
743     
744           Node objectWithListener = n.getSecondChild();
745           if (!objectWithListener.isQualifiedName()) {
746             return;
747           }
748     
749           if (listener.isFunction()) {
750             /*
751              * Anonymous function
752              */
753             compiler.report(JSError.make(n, UNLISTEN_WITH_ANONBOUND));
754           } else if (listener.isCall()) {
755             if (!listener.getFirstChild().isQualifiedName()) {
756               /*
757                * Anonymous function
758                */
759               compiler.report(JSError.make(n, UNLISTEN_WITH_ANONBOUND));
760             } else if (listener.getFirstChild().matchesQualifiedName("goog.bind")) {
761               /*
762                * Using goog.bind to unlisten
763                */
764               compiler.report(JSError.make(n, UNLISTEN_WITH_ANONBOUND));
765             }
766           }
767         }

770         private void visitCall(NodeTraversal t, Node n) {
771           Node functionCalled = n.getFirstChild();
772           if (functionCalled == null || !functionCalled.isQualifiedName()) {
773               return;
774           }
775           JSType typeOfThis = getTypeOfThisForScope(t);
776           if (typeOfThis == null) {
777             return;
778           }
779     
780           /*
781            * Class considered eventful if there is an unlisten call in the
782            * disposal.
783            */
784           if (functionCalled.matchesQualifiedName("goog.events.unlisten")) {
785     
786             if (inDisposalScope()) {
787               eventfulTypes.add(typeOfThis);
788             }
789             isGoogEventsUnlisten(n);
790           }
791           if (inDisposalScope()
792               && functionCalled.matchesQualifiedName("goog.events.removeAll")) {
793             eventfulTypes.add(typeOfThis);
794           }
795     
796           /*
797            * If member with qualified name gets disposed of when this class
798            * gets disposed, consider the member type as an eventizer of this
799            * class.
800            */
801           JSType disposedType = maybeReturnDisposedType(n, inDisposalScope());
802           if (!collectorFilterType(disposedType)) {
803             addEventize(getTypeOfThisForScope(t), disposedType);
804           }
805         }

809           switch (n.getType()) {
810             case CALL:
811               visitCall(t, n);
812               break;
813             default:
814               break;
815           }
816         }

821         /*
822          * Checks if the input node correspond to the creation of an eventful object
823          */
824         private boolean createsEventfulObject(Node n) {
825           Node first = n.getFirstChild();
826           JSType type = n.getJSType();
827           if (first == null
828               || !first.isQualifiedName()
829               || type.isEmptyType()
830               || type.isUnknownType()) {
831             return false;
832           }
833     
834           boolean isOfTypeNeedingDisposal = false;
835           for (JSType disposableType : eventfulTypes) {
836             if (type.isSubtype(disposableType)) {
837               isOfTypeNeedingDisposal = true;
838               break;
839             }
840           }
841           return isOfTypeNeedingDisposal;
842         }
843     
844         /*
845          * This function traverses the current scope to see if a locally
846          * defined eventful object is assigned to a live-out variable.
847          *
848          * Note: This function could be called multiple times to traverse
849          * the same scope if multiple local eventful objects are created in the
850          * scope.
851          */
852         private Node localEventfulObjectAssign(
853             NodeTraversal t, Node propertyNode) {
854           Node parent;
855           if (!t.getTypedScope().isGlobal()) {
856             /*
857              * In function
858              */
859             parent = NodeUtil.getFunctionBody(t.getScopeRoot());
860           } else {
861             /*
862              * In global scope
863              */
864             parent = t.getScopeRoot().getFirstChild();
865           }
866     
867           /*
868            * Check to see if locally created EventHandler is assigned to field
869            */
870           for (Node sibling : parent.children()) {
871             if (sibling.isExprResult()) {
872               Node assign = sibling.getFirstChild();
873               if (assign.isAssign()) {
874                 // assign.getLastChild().isEquivalentTo(propertyNode) did not work
875                 if (propertyNode.matchesQualifiedName(assign.getLastChild())) {
876                   if (!assign.getFirstChild().isName()) {
877                     return assign.getFirstChild();
878                   }
879                 }
880               }
881             }
882           }
883     
884           /*
885            * Eventful object created and assigned to a local variable which is not
886            * assigned to another variable in a way to allow disposal.
887            */
888           String key = generateKey(t, propertyNode, false);
889           if (key == null) {
890             return null;
891           }
892     
893           EventfulObjectState e;
894           if (eventfulObjectMap.containsKey(key)) {
895             e = eventfulObjectMap.get(key);
896     
897             if (e.seen == SeenType.ALLOCATED) {
898               e.seen = SeenType.ALLOCATED_LOCALLY;
899             }
900           } else {
901             e = new EventfulObjectState();
902             e.seen = SeenType.ALLOCATED_LOCALLY;
903     
904             eventfulObjectMap.put(key, e);
905           }
906           e.allocationSite = propertyNode;
907     
908     
909           return null;
910         }
911     
912         /*
913          * Record the creation of a new eventful object.
914          */
915         private void visitNew(NodeTraversal t, Node n, Node parent) {
916           if (!createsEventfulObject(n)) {
917             return;
918           }
919     
920           /*
921            * Insert allocation site and construct into eventfulObjectMap
922            */
923           String key;
924           Node propertyNode;
925     
926           /*
927            * Handles (E is an eventful class):
928            *  - object.something = new E();
929            *  - local = new E();
930            *  - var local = new E();
931            */
932           if (parent.isAssign()) {
933             propertyNode = parent.getFirstChild();
934           } else {
935             propertyNode = parent;
936           }
937     
938           key = generateKey(t, propertyNode, false);
939           if (key == null) {
940             return;
941           }
942     
943           EventfulObjectState e;
944           if (eventfulObjectMap.containsKey(key)) {
945             e = eventfulObjectMap.get(key);
946           } else {
947             e = new EventfulObjectState();
948             e.seen = SeenType.ALLOCATED;
949     
950             eventfulObjectMap.put(key, e);
951           }
952           e.allocationSite = propertyNode;
953     
954           /*
955            * Check if locally defined eventful object is assigned to global variable
956            * and create an entry mapping to the previous site.
957            */
958           if (propertyNode.isName()) {
959             Node globalVarNode = localEventfulObjectAssign(t, propertyNode);
960             if (globalVarNode != null) {
961               key = generateKey(t, globalVarNode, false);
962               if (key == null) {
963                 /*
964                  * Local variable is assigned to an array or in a manner requiring
965                  * a function call.
966                  */
967                 e.seen = SeenType.POSSIBLY_DISPOSED;
968                 return;
969               }
970               eventfulObjectMap.put(key, e);
971             }
972           }
973         }

975         private void addDisposeArgumentsMatched(Map<String, List<Integer>> map,
976             Node n, String property, List<Node> foundDisposeCalls) {
977           for (Map.Entry<String, List<Integer>> disposeCallsEntry : map.entrySet()) {
978             if (property.endsWith(disposeCallsEntry.getKey())) {
979               List<Integer> disposeArguments = disposeCallsEntry.getValue();
980     
981               // Dispose specific arguments only
982               Node t = n.getNext();
983               int tsArgument = 0;
984               for (Integer disposeArgument : disposeArguments) {
985                 switch (disposeArgument) {
986                   // Dispose all arguments
987                   case DISPOSE_ALL:
988                     for (Node tt = n.getNext(); tt != null; tt = tt.getNext()) {
989                       foundDisposeCalls.add(tt);
990                     }
991                     break;
992                   // Dispose objects called on
993                   case DISPOSE_SELF:
994                     Node calledOn = n.getFirstChild();
995     
996                     foundDisposeCalls.add(calledOn);
997                     break;
998                   default:
999                     // The current item pointed to by t is beyond that requested in
1000                     // current array element.
1001                     if (tsArgument > disposeArgument) {
1002                       t = n.getNext();
1003                       tsArgument = 0;
1004                     }
1005                     for (; tsArgument < disposeArgument && t != null;
1006                             ++tsArgument) {
1007                       t = t.getNext();
1008                     }
1009                     if (tsArgument == disposeArgument && t != null) {
1010                       foundDisposeCalls.add(t);
1011                     }
1012                     break;
1013                 }
1014               }
1015             }
1016           }
1017         }

1019         private List<Node> maybeGetValueNodesFromCall(Node n) {
1020           List<Node> ret = new ArrayList<>();
1021           Node first = n.getFirstChild();
1022     
1023           if (first == null || !first.isQualifiedName()) {
1024             return ret;
1025           }
1026           String property = first.getQualifiedName();
1027     
1028           Node base = first.getFirstChild();
1029           JSType baseType = null;
1030           if (base != null) {
1031             baseType = base.getJSType();
1032           }
1033     
1034           for (Map.Entry<JSType, Map<String, List<Integer>>> disposeCallEntry :
1035               disposeCalls.entrySet()) {
1036             JSType key = disposeCallEntry.getKey();
1037             if (key == null
1038                 || (baseType != null && isPossiblySubtype(baseType, key))) {
1039               addDisposeArgumentsMatched(disposeCallEntry.getValue(), first,
1040                   property, ret);
1041             }
1042           }
1043     
1044           return ret;
1045         }

1047         /*
1048          * Look for calls to an eventful object's disposal functions.
1049          * (dispose or removeAll will remove all event listeners from
1050          * an EventHandler).
1051          */
1052         private void visitCall(NodeTraversal t, Node n) {
1053           // Filter the calls to find a "dispose" call
1054           List<Node> variableNodes = maybeGetValueNodesFromCall(n);
1055     
1056           for (Node variableNode : variableNodes) {
1057             Preconditions.checkState(variableNode != null);
1058     
1059             // Only consider removals on eventful object
1060             boolean isTrackedRemoval = false;
1061             JSType vnType = variableNode.getJSType();
1062             for (JSType type : eventfulTypes) {
1063               if (isPossiblySubtype(vnType, type)) {
1064                 isTrackedRemoval = true;
1065               }
1066             }
1067             if (!isTrackedRemoval) {
1068               continue;
1069             }
1070     
1071             String key = generateKey(t, variableNode, false);
1072             if (key == null) {
1073               continue;
1074             }
1075     
1076             eventfulObjectDisposed(t, variableNode);
1077           }
1078         }

1080         /**
1081          * Dereference a type, autoboxing it and filtering out null.
1082          * From {@link CheckAccessControls}
1083          */
1084         private JSType dereference(JSType type) {
1085           return type == null ? null : type.dereference();
1086         }

1088         /*
1089          * Check function definitions to add custom dispose methods.
1090          */
1091         public void visitFunction(NodeTraversal t, Node n) {
1092           Preconditions.checkArgument(n.isFunction());
1093           JSDocInfo jsDocInfo = NodeUtil.getBestJSDocInfo(n);
1094     
1095           // Function annotated to dispose of
1096           if (jsDocInfo != null && jsDocInfo.isDisposes()) {
1097             JSType type = n.getJSType();
1098             if (type == null || type.isUnknownType()) {
1099               return;
1100             }
1101     
1102             FunctionType funType = type.toMaybeFunctionType();
1103             Node paramNode = NodeUtil.getFunctionParameters(n).getFirstChild();
1104             List<Integer> positionalDisposedParameters = new ArrayList<>();
1105     
1106             if (jsDocInfo.disposesOf("*")) {
1107               positionalDisposedParameters.add(DISPOSE_ALL);
1108             } else {
1109               // Record index of parameters that are disposed.
1110               for (int index = 0; index < funType.getMaxArguments(); ++index) {
1111                 // Bail out if the paramNode is not there.
1112                 if (paramNode == null) {
1113                   break;
1114                 }
1115                 if (jsDocInfo.disposesOf(paramNode.getString())) {
1116                   positionalDisposedParameters.add(index);
1117                 }
1118                 paramNode = paramNode.getNext();
1119               }
1120             }
1121             addDisposeCall(NodeUtil.getName(n), positionalDisposedParameters);
1122           }
1123         }

1125         /*
1126          * Track assignments to see if a private field is being
1127          * overwritten.
1128          *
1129          * Assigning to an array element is taken care of by the generateKey
1130          * returning null on array ("complex") variable names.
1131          */
1132         public void visitAssign(NodeTraversal t, Node n) {
1133           Node assignedTo = n.getFirstChild();
1134           JSType assignedToType = assignedTo.getJSType();
1135           if (assignedToType == null || assignedToType.isEmptyType()) {
1136             return;
1137           }
1138     
1139           if (n.getFirstChild().isGetProp()) {
1140             boolean isTrackedAssign = false;
1141             for (JSType disposalType : eventfulTypes) {
1142               if (assignedToType.isSubtype(disposalType)) {
1143                 isTrackedAssign = true;
1144                 break;
1145               }
1146             }
1147             if (!isTrackedAssign) {
1148               return;
1149             }
1150     
1151             JSDocInfo di = n.getJSDocInfo();
1152             ObjectType objectType =
1153                 ObjectType.cast(dereference(n.getFirstFirstChild()
1154                     .getJSType()));
1155             String propertyName = n.getFirstChild().getLastChild().getString();
1156     
1157             boolean fieldIsPrivate =
1158                 (di != null) && (di.getVisibility() == Visibility.PRIVATE);
1159     
1160             /*
1161              * See if field is defined as private in superclass
1162              */
1163             while (objectType != null) {
1164               di = null;
1165               objectType = objectType.getImplicitPrototype();
1166               if (objectType == null) {
1167                 break;
1168               }
1169     
1170               /*
1171                * Skip prototype definitions:
1172                *   Don't flag a field declared private in assignment as well
1173                *   as in prototype declaration
1174                * Assumption: The inheritance hierarchy is similar to
1175                *   class
1176                *   class.prototype
1177                *   superclass
1178                *   superclass.prototype
1179                */
1180               if (objectType.getDisplayName().endsWith("prototype")) {
1181                 continue;
1182               }
1183     
1184               di = objectType.getOwnPropertyJSDocInfo(propertyName);
1185               if (di != null) {
1186                 if (fieldIsPrivate || di.getVisibility() == Visibility.PRIVATE) {
1187                   compiler.report(
1188                       t.makeError(n, OVERWRITE_PRIVATE_EVENTFUL_OBJECT));
1189                   break;
1190                 }
1191               }
1192             }
1193           }
1194         }

1196         /*
1197          * Filter out any eventful objects returned.
1198          */
1199         private void visitReturn(NodeTraversal t, Node n) {
1200           Node variableNode = n.getFirstChild();
1201           if (variableNode == null) {
1202             return;
1203           }
1204     
1205           if (!variableNode.isArrayLit()) {
1206             eventfulObjectDisposed(t, variableNode);
1207           } else {
1208             for (Node child : variableNode.children()) {
1209               eventfulObjectDisposed(t, child);
1210             }
1211           }
1212         }

1214         /*
1215          * Mark an eventful object as being disposed.
1216          */
1217         private void eventfulObjectDisposed(NodeTraversal t, Node variableNode) {
1218           String key = generateKey(t, variableNode, false);
1219           if (key == null) {
1220             return;
1221           }
1222     
1223           EventfulObjectState e = eventfulObjectMap.get(key);
1224           if (e == null) {
1225             e = new EventfulObjectState();
1226             eventfulObjectMap.put(key, e);
1227           }
1228           e.seen = SeenType.POSSIBLY_DISPOSED;
1229         }

1231         @Override
1232         public void enterScope(NodeTraversal t) {
1233           TypedScope scope = t.getTypedScope();
1234           if (scope.getVarCount() > LiveVariablesAnalysis.MAX_VARIABLES_TO_ANALYZE) {
1235             // Too many variables to analyze, so just assume that all eventful objects are
1236             // disposed. This will miss some errors but only in very large scopes.
1237             for (TypedVar v : scope.getVarIterable()) {
1238               eventfulObjectDisposed(t, v.getNode());
1239             }
1240             return;
1241           }
1242     
1243           /*
1244            * Local variables captured in scope are filtered at present.
1245            * LiveVariableAnalysis used to filter such variables.
1246            */
1247           ControlFlowGraph<Node> cfg = t.getControlFlowGraph();
1248           LiveVariablesAnalysis liveness =
1249               new LiveVariablesAnalysis(cfg, t.getTypedScope(), compiler);
1250           liveness.analyze();
1251     
1252           for (TypedVar v : ((Set<TypedVar>) liveness.getEscapedLocals())) {
1253             eventfulObjectDisposed(t, v.getNode());
1254           }
1255         }

1259         }

1261         @Override
1262         public void visit(NodeTraversal t, Node n, Node parent) {
1263           switch (n.getType()) {
1264             case ASSIGN:
1265               visitAssign(t, n);
1266               break;
1267             case CALL:
1268               visitCall(t, n);
1269               break;
1270             case FUNCTION:
1271               visitFunction(t, n);
1272               break;
1273             case NEW:
1274               visitNew(t, n, parent);
1275               break;
1276             case RETURN:
1277               visitReturn(t, n);
1278               break;
1279             default:
1280               break;
1281           }
1282         }
