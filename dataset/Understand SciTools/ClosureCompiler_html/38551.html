157       private static JSType makeType(int mask,
158           ImmutableSet<ObjectType> objs, String typeVar,
159           ImmutableSet<EnumType> enums) {
160         // Fix up the mask for objects and enums
161         if (Preconditions.checkNotNull(enums).isEmpty()) {
162           mask &= ~ENUM_MASK;
163         } else {
164           mask |= ENUM_MASK;
165         }
166     
167         if (Preconditions.checkNotNull(objs).isEmpty()) {
168           mask &= ~NON_SCALAR_MASK;
169         } else {
170           mask |= NON_SCALAR_MASK;
171         }
172     
173         if (objs.isEmpty() && enums.isEmpty()
174             && typeVar == null && (mask & TYPEVAR_MASK) == 0) {
175           return MaskType.make(mask);
176         }
177         if (!JSType.isInhabitable(objs)) {
178           return BOTTOM;
179         }
180         if (mask == NON_SCALAR_MASK) {
181           return new ObjsType(objs);
182         }
183         if (mask == (NON_SCALAR_MASK | NULL_MASK)) {
184           return new NullableObjsType(objs);
185         }
186         return new UnionType(mask, objs, typeVar, enums);
187       }

222       boolean isValidType() {
223         if (isUnknown() || isTop()) {
224           return true;
225         }
226         if ((getMask() & NON_SCALAR_MASK) != 0
227             && getObjs().isEmpty()) {
228           return false;
229         }
230         if ((getMask() & NON_SCALAR_MASK) == 0 && !getObjs().isEmpty()) {
231           return false;
232         }
233         if ((getMask() & ENUM_MASK) != 0
234             && getEnums().isEmpty()) {
235           return false;
236         }
237         if ((getMask() & ENUM_MASK) == 0 && !getEnums().isEmpty()) {
238           return false;
239         }
240         return ((getMask() & TYPEVAR_MASK) != 0) == (getTypeVar() != null);
241       }

421       // Returns null if this type doesn't inherit from IObject
422       public JSType getIndexType() {
423         if (getMask() != NON_SCALAR_MASK) {
424           return null;
425         }
426         // This (union) type is a supertype of all indexed types in the union.
427         // Different from NominalType#getIndexType, which uses join.
428         JSType result = TOP;
429         // We need this because the index type may explicitly be TOP.
430         boolean foundIObject = false;
431         for (ObjectType objType : getObjs()) {
432           JSType tmp = objType.getNominalType().getIndexType();
433           if (tmp == null) {
434             return null;
435           }
436           foundIObject = true;
437           result = meet(result, tmp);
438         }
439         return foundIObject ? result : null;
440       }

497       public JSType autobox() {
498         if (isTop() || isUnknown()) {
499           return this;
500         }
501         int mask = getMask();
502         if ((mask & (NUMBER_MASK | STRING_MASK | BOOLEAN_MASK | ENUM_MASK)) == BOTTOM_MASK) {
503           return this;
504         }
505         switch (mask) {
506           case NUMBER_MASK:
507             return commonTypes.getNumberInstance();
508           case BOOLEAN_MASK:
509           case TRUE_MASK:
510           case FALSE_MASK:
511             return commonTypes.getBooleanInstance();
512           case STRING_MASK:
513             return commonTypes.getStringInstance();
514         }
515         // For each set bit, add the corresponding obj to the new objs
516         // construct and return the new type.
517         // Don't bother autoboxing enums.
518         ImmutableSet.Builder<ObjectType> builder = ImmutableSet.builder();
519         builder.addAll(getObjs());
520         if ((mask & NUMBER_MASK) != 0) {
521           builder.add(commonTypes.getNumberInstanceObjType());
522         }
523         if ((mask & STRING_MASK) != 0) {
524           builder.add(commonTypes.getStringInstanceObjType());
525         }
526         if ((mask & BOOLEAN_MASK) != 0) { // may have truthy or falsy
527           builder.add(commonTypes.getBooleanInstanceObjType());
528         }
529         JSType result = makeType(
530             mask & ~(NUMBER_MASK | STRING_MASK | BOOLEAN_MASK),
531             builder.build(), getTypeVar(), NO_ENUMS);
532         for (EnumType e : getEnums()) {
533           result = join(result, e.getEnumeratedType().autobox());
534         }
535         return result;
536       }

547       // When joining w/ TOP or UNKNOWN, avoid setting more fields on them, eg, obj.
548       public static JSType join(JSType lhs, JSType rhs) {
549         Preconditions.checkNotNull(lhs);
550         Preconditions.checkNotNull(rhs);
551         if (lhs.isTop() || rhs.isTop()) {
552           return TOP;
553         }
554         if (lhs.isUnknown() || rhs.isUnknown()) {
555           return UNKNOWN;
556         }
557         if (lhs.isBottom()) {
558           return rhs;
559         }
560         if (rhs.isBottom()) {
561           return lhs;
562         }
563         if (lhs.hasTruthyMask() || lhs.hasFalsyMask()
564             || rhs.hasTruthyMask() || rhs.hasFalsyMask()) {
565           return UNKNOWN;
566         }
567         if (lhs.getTypeVar() != null && rhs.getTypeVar() != null
568             && !lhs.getTypeVar().equals(rhs.getTypeVar())) {
569           // For now return ? when joining two type vars. This is probably uncommon.
570           return UNKNOWN;
571         }
572     
573         int newMask = lhs.getMask() | rhs.getMask();
574         ImmutableSet<ObjectType> newObjs =
575             ObjectType.joinSets(lhs.getObjs(), rhs.getObjs());
576         String newTypevar =
577             lhs.getTypeVar() != null ? lhs.getTypeVar() : rhs.getTypeVar();
578         ImmutableSet<EnumType> newEnums =
579             EnumType.union(lhs.getEnums(), rhs.getEnums());
580         if (newEnums.isEmpty()) {
581           return makeType(newMask, newObjs, newTypevar, NO_ENUMS);
582         }
583         JSType tmpJoin =
584             makeType(newMask & ~ENUM_MASK, newObjs, newTypevar, NO_ENUMS);
585         return makeType(newMask, newObjs, newTypevar,
586             EnumType.normalizeForJoin(newEnums, tmpJoin));
587       }

589       public JSType substituteGenerics(Map<String, JSType> concreteTypes) {
590         if (isTop()
591             || isUnknown()
592             || getObjs().isEmpty() && getTypeVar() == null
593             || concreteTypes.isEmpty()) {
594           return this;
595         }
596         ImmutableSet.Builder<ObjectType> builder = ImmutableSet.builder();
597         for (ObjectType obj : getObjs()) {
598           builder.add(obj.substituteGenerics(concreteTypes));
599         }
600         JSType current = makeType(
601             getMask() & ~TYPEVAR_MASK, builder.build(), null, getEnums());
602         if (hasTypeVariable()) {
603           current = JSType.join(current, concreteTypes.containsKey(getTypeVar()) ?
604               concreteTypes.get(getTypeVar()) : fromTypeVar(getTypeVar()));
605         }
606         return current;
607       }

613       private static void updateTypemap(
614           Multimap<String, JSType> typeMultimap,
615           String typeParam, JSType type) {
616         Preconditions.checkNotNull(type);
617         Set<JSType> typesToRemove = new LinkedHashSet<>();
618         for (JSType other : typeMultimap.get(typeParam)) {
619           if (type.isUnknown()) {
620             typesToRemove.add(other);
621             continue;
622           }
623           if (other.isUnknown()) {
624             type = null;
625             break;
626           }
627           // The only way to instantiate with a loose type is if there are no
628           // concrete types available. We may miss some warnings this way but we
629           // also avoid false positives.
630           if (type.isLoose()) {
631             type = null;
632             break;
633           } else if (other.isLoose()) {
634             typesToRemove.add(other);
635             continue;
636           }
637           JSType unified = unifyUnknowns(type, other);
638           if (unified != null) {
639             // Can't remove elms while iterating over the collection, so do it later
640             typesToRemove.add(other);
641             type = unified;
642           } else if (other.isSubtypeOf(type, SubtypeCache.create())) {
643             typesToRemove.add(other);
644           } else if (type.isSubtypeOf(other, SubtypeCache.create())) {
645             type = null;
646             break;
647           }
648         }
649         for (JSType typeToRemove : typesToRemove) {
650           typeMultimap.remove(typeParam, typeToRemove);
651         }
652         if (type != null) {
653           typeMultimap.put(typeParam, type);
654         }
655       }

664       /**
665        * Unify the two types symmetrically, given that we have already instantiated
666        * the type variables of interest in {@code t1} and {@code t2}, treating
667        * JSType.UNKNOWN as a "hole" to be filled.
668        * @return The unified type, or null if unification fails */
669       static JSType unifyUnknowns(JSType t1, JSType t2) {
670         Preconditions.checkNotNull(t1);
671         Preconditions.checkNotNull(t2);
672         if (t1.isUnknown() || t1.isLoose()) {
673           return t2;
674         } else if (t2.isUnknown() || t2.isLoose()) {
675           return t1;
676         } else if (t1.isTop() && t2.isTop()) {
677           return TOP;
678         } else if (t1.isTop() || t2.isTop()) {
679           return null;
680         }
681     
682         if (!t1.getEnums().equals(t2.getEnums())) {
683           return null;
684         }
685         ImmutableSet<EnumType> newEnums = t1.getEnums();
686     
687         int t1Mask = promoteBoolean(t1.getMask());
688         int t2Mask = promoteBoolean(t2.getMask());
689         if (t1Mask != t2Mask || !Objects.equals(t1.getTypeVar(), t2.getTypeVar())) {
690           return null;
691         }
692         // All scalar types are equal
693         if ((t1Mask & NON_SCALAR_MASK) == 0) {
694           return t1;
695         }
696         if (t1.getObjs().size() != t2.getObjs().size()) {
697           return null;
698         }
699     
700         Set<ObjectType> ununified = new LinkedHashSet<>(t2.getObjs());
701         Set<ObjectType> unifiedObjs = new LinkedHashSet<>();
702         for (ObjectType objType1 : t1.getObjs()) {
703           ObjectType unified = objType1;
704           boolean hasUnified = false;
705           for (ObjectType objType2 : t2.getObjs()) {
706             ObjectType tmp = ObjectType.unifyUnknowns(unified, objType2);
707             if (tmp != null) {
708               hasUnified = true;
709               ununified.remove(objType2);
710               unified = tmp;
711             }
712           }
713           if (!hasUnified) {
714             return null;
715           }
716           unifiedObjs.add(unified);
717         }
718         if (!ununified.isEmpty()) {
719           return null;
720         }
721         return makeType(t1Mask, ImmutableSet.copyOf(unifiedObjs),
722             t1.getTypeVar(), newEnums);
723       }

725       /**
726        * Unify {@code this}, which may contain free type variables,
727        * with {@code other}, a concrete subtype, modifying the supplied
728        * {@code typeMultimap} to add any new template variable type bindings.
729        * Note that if {@code this} is a union type, some of the union members may
730        * be ignored if they are not present in {@code other}.
731        * @return Whether unification succeeded
732        *
733        * This method should only be called outside the newtypes package;
734        * classes inside the package should use unifyWithSubtype.
735        */
736       public boolean unifyWith(JSType other, List<String> typeParameters,
737           Multimap<String, JSType> typeMultimap) {
738         return unifyWithSubtype(
739             other, typeParameters, typeMultimap, SubtypeCache.create());
740       }

742       boolean unifyWithSubtype(JSType other, List<String> typeParameters,
743           Multimap<String, JSType> typeMultimap, SubtypeCache subSuperMap) {
744         Preconditions.checkNotNull(other);
745         if (this.isUnknown() || this.isTop()) {
746           return true;
747         } else if (getMask() == TYPEVAR_MASK
748             && typeParameters.contains(getTypeVar())) {
749           updateTypemap(typeMultimap, getTypeVar(), other);
750           return true;
751         } else if (other.isUnknown()) {
752           return true;
753         } else if (other.isTop()) {
754           // T|number doesn't unify with TOP
755           return false;
756         }
757     
758         Set<EnumType> ununifiedEnums = ImmutableSet.of();
759         if (!other.getEnums().isEmpty()) {
760           ununifiedEnums = new LinkedHashSet<>();
761           for (EnumType e : other.getEnums()) {
762             if (!fromEnum(e).isSubtypeOf(this, SubtypeCache.create())) {
763               ununifiedEnums.add(e);
764             }
765           }
766         }
767     
768         Set<ObjectType> ununifiedObjs = new LinkedHashSet<>(other.getObjs());
769         // We don't check that two different objects of this don't unify
770         // with the same other type.
771         // Fancy cases are unfortunately iteration-order dependent, eg,
772         // Foo<number>|Foo<string> may or may not unify with Foo<T>|Foo<string>
773         for (ObjectType targetObj : getObjs()) {
774           for (ObjectType sourceObj : other.getObjs()) {
775             if (targetObj.unifyWithSubtype(
776                   sourceObj, typeParameters, typeMultimap, subSuperMap)) {
777               ununifiedObjs.remove(sourceObj);
778             }
779           }
780         }
781     
782         String thisTypevar = getTypeVar();
783         String otherTypevar = other.getTypeVar();
784         if (thisTypevar == null || !typeParameters.contains(thisTypevar)) {
785           return ununifiedObjs.isEmpty() && ununifiedEnums.isEmpty()
786               && (otherTypevar == null || otherTypevar.equals(thisTypevar))
787               && getMask() == (getMask() | (other.getMask() & ~ENUM_MASK));
788         } else {
789           // this is (T | ...)
790           int thisScalarBits = getMask() & ~NON_SCALAR_MASK & ~TYPEVAR_MASK;
791           int templateMask = other.getMask() & ~thisScalarBits;
792           if (ununifiedObjs.isEmpty()) {
793             templateMask &= ~NON_SCALAR_MASK;
794           }
795     
796           if (templateMask == BOTTOM_MASK) {
797             // nothing left in other to assign to thisTypevar, so don't update typemap
798             return ununifiedObjs.isEmpty() && ununifiedEnums.isEmpty();
799           }
800           JSType templateType = makeType(
801               promoteBoolean(templateMask),
802               ImmutableSet.copyOf(ununifiedObjs),
803               otherTypevar,
804               ImmutableSet.copyOf(ununifiedEnums));
805           updateTypemap(typeMultimap, getTypeVar(), templateType);
806           return true;
807         }
808       }

810       public JSType specialize(JSType other) {
811         JSType t = specializeHelper(other);
812         if (t.isBottom() && (isLoose() || other.isLoose())) {
813           t = autobox().specializeHelper(other.autobox());
814           // If the autoboxed specialization is not null, this means that one of
815           // the two types contains scalars that when autoboxed are compatible with
816           // the loose object in the other type. In this case, don't return bottom,
817           // just leave the type unspecialized.
818           if (!t.isBottom()) {
819             return this;
820           }
821         }
822         if (t.isLoose()) {
823           JSType maybeScalar = ObjectType.mayTurnLooseObjectToScalar(t, JSType.commonTypes);
824           if (t != maybeScalar) { // ref equality on purpose
825             return maybeScalar;
826           }
827         }
828         return t;
829       }

831       private JSType specializeHelper(JSType other) {
832         if (other.isTop() || other.isUnknown() || this == other) {
833           return this;
834         }
835         if (other.hasTruthyMask()) {
836           return makeTruthy();
837         }
838         if (hasTruthyMask()) {
839           // If the only thing we know about this type is that it's truthy, that's very
840           // little information, so we loosen the other type to avoid spurious warnings.
841           JSType otherTruthy = other.makeTruthy();
842           return otherTruthy.hasNonScalar()
843               ? otherTruthy.withLoose()
844               : otherTruthy;
845         }
846         if (other.hasFalsyMask()) {
847           return makeFalsy();
848         }
849         if (hasFalsyMask()) {
850           return other.makeFalsy();
851         }
852         if (this.isTop()) {
853           return other;
854         }
855         if (this.isUnknown()) {
856           NominalType otherNt = other.getNominalTypeIfSingletonObj();
857           return otherNt != null && otherNt.isBuiltinObject() ? other.withLoose() : other;
858         }
859         int newMask = getMask() & other.getMask();
860         String newTypevar;
861         if (Objects.equals(getTypeVar(), other.getTypeVar())) {
862           newTypevar = getTypeVar();
863         } else if (getTypeVar() != null && other.getTypeVar() == null) {
864           // Consider, e.g., function f(/** T|number */ x) { if (typeof x === 'string') { ... } }
865           // We want to specialize the T to string, rather than going to bottom.
866           return other;
867         } else {
868           newTypevar = null;
869           newMask &= ~TYPEVAR_MASK;
870         }
871         return meetEnums(
872             newMask, getMask() | other.getMask(),
873             ObjectType.specializeSet(getObjs(), other.getObjs()),
874             newTypevar, getObjs(), other.getObjs(), getEnums(), other.getEnums());
875       }

877       public static JSType meet(JSType lhs, JSType rhs) {
878         JSType t = meetHelper(lhs, rhs);
879         if (t.isBottom() && (lhs.isLoose() || rhs.isLoose())) {
880           t = meetHelper(lhs.autobox(), rhs.autobox());
881           // If the autoboxed meet is not null, this means that one of
882           // the two types contains scalars that when autoboxed are compatible with
883           // the loose object in the other type. In this case, don't return bottom,
884           // use some heuristic to return a result.
885           if (!t.isBottom()) {
886             if (!lhs.isLoose()) {
887               return lhs;
888             } else {
889               // If this fails, find repro case, add a test, and return ? here.
890               Preconditions.checkState(!rhs.isLoose(), "Two loose types %s and %s"
891                   + " that meet to bottom, meet to non-bottom after autoboxing.",
892                   lhs.toString(), rhs.toString());
893               return rhs;
894             }
895           }
896         }
897         return t;
898       }

900       private static JSType meetHelper(JSType lhs, JSType rhs) {
901         if (lhs.isTop()) {
902           return rhs;
903         } else if (rhs.isTop()) {
904           return lhs;
905         } else if (lhs.isUnknown()) {
906           return rhs;
907         } else if (rhs.isUnknown()) {
908           return lhs;
909         } else if (lhs.isBottom() || rhs.isBottom()) {
910           return BOTTOM;
911         }
912         int newMask = lhs.getMask() & rhs.getMask();
913         String newTypevar;
914         if (Objects.equals(lhs.getTypeVar(), rhs.getTypeVar())) {
915           newTypevar = lhs.getTypeVar();
916         } else {
917           newTypevar = null;
918           newMask = newMask & ~TYPEVAR_MASK;
919         }
920         return meetEnums(
921             newMask, lhs.getMask() | rhs.getMask(),
922             ObjectType.meetSets(lhs.getObjs(), rhs.getObjs()),
923             newTypevar, lhs.getObjs(), rhs.getObjs(),
924             lhs.getEnums(), rhs.getEnums());
925       }

927       /**
928        * Both {@code meet} and {@code specialize} do the same computation for enums.
929        * They don't just compute the set of enums; they may modify mask and objs.
930        * So, both methods finish off by calling this one.
931        */
932       private static JSType meetEnums(int newMask, int unionMask,
933           ImmutableSet<ObjectType> newObjs, String newTypevar,
934           ImmutableSet<ObjectType> objs1, ImmutableSet<ObjectType> objs2,
935           ImmutableSet<EnumType> enums1, ImmutableSet<EnumType> enums2) {
936         if (Objects.equals(enums1, enums2)) {
937           return makeType(newMask, newObjs, newTypevar, enums1);
938         }
939         ImmutableSet.Builder<EnumType> enumBuilder = ImmutableSet.builder();
940         ImmutableSet<EnumType> allEnums = EnumType.union(enums1, enums2);
941         for (EnumType e : allEnums) {
942           // An enum in the intersection will always be in the result
943           if (enums1 != null && enums1.contains(e)
944               && enums2 != null && enums2.contains(e)) {
945             enumBuilder.add(e);
946             continue;
947           }
948           // An enum {?} in the union will always be in the result
949           JSType enumeratedType = e.getEnumeratedType();
950           if (enumeratedType.isUnknown()) {
951             enumBuilder.add(e);
952             continue;
953           }
954           // An enum {TypeA} meets with any supertype of TypeA. When this happens,
955           // we put the enum in the result and remove the supertype.
956           // The following would be much more complex if we allowed the type of
957           // an enum to be a union.
958           if (enumeratedType.getMask() != NON_SCALAR_MASK) {
959             if ((enumeratedType.getMask() & unionMask) != 0) {
960               enumBuilder.add(e);
961               newMask &= ~enumeratedType.getMask();
962             }
963           } else if (!objs1.isEmpty() || !objs2.isEmpty()) {
964             Set<ObjectType> objsToRemove = new LinkedHashSet<>();
965             ObjectType enumObj = Iterables.getOnlyElement(enumeratedType.getObjs());
966             for (ObjectType obj1 : objs1) {
967               if (enumObj.isSubtypeOf(obj1, SubtypeCache.create())) {
968                 enumBuilder.add(e);
969                 objsToRemove.add(obj1);
970               }
971             }
972             for (ObjectType obj2 : objs2) {
973               if (enumObj.isSubtypeOf(obj2, SubtypeCache.create())) {
974                 enumBuilder.add(e);
975                 objsToRemove.add(obj2);
976               }
977             }
978             if (!objsToRemove.isEmpty()) {
979               newObjs = Sets.difference(newObjs, objsToRemove).immutableCopy();
980             }
981           }
982         }
983         return makeType(newMask, newObjs, newTypevar, enumBuilder.build());
984       }

1080       boolean isSubtypeOf(TypeI other, SubtypeCache subSuperMap) {

1085         if (isLoose() || type2.isLoose()) {

1092       private boolean isSubtypeOfHelper(
1093           boolean keepLoosenessOfThis, JSType other, SubtypeCache subSuperMap,
1094           MismatchInfo[] boxedInfo) {
1095         if (isUnknown() || other.isUnknown() || other.isTop()) {
1096           return true;
1097         }
1098         if (hasTruthyMask()) {
1099           return !other.makeTruthy().isBottom();
1100         }
1101         if (hasFalsyMask()) {
1102           return !other.makeFalsy().isBottom();
1103         }
1104         if (!EnumType.areSubtypes(this, other, subSuperMap)) {
1105           return false;
1106         }
1107         int mask = getMask() & ~ENUM_MASK;
1108         if ((mask | other.getMask()) != other.getMask()) {
1109           if (boxedInfo != null && isUnion()) {
1110             whyNotUnionSubtypes(this, other, boxedInfo);
1111           }
1112           return false;
1113         }
1114         if (getTypeVar() != null && !getTypeVar().equals(other.getTypeVar())) {
1115           return false;
1116         }
1117         if (getObjs().isEmpty()) {
1118           return true;
1119         }
1120         // Because of optional properties,
1121         //   x \le y \iff x \join y = y does not hold.
1122         boolean result = ObjectType.isUnionSubtype(
1123             keepLoosenessOfThis, getObjs(), other.getObjs(), subSuperMap);
1124         if (boxedInfo != null) {
1125           ObjectType.whyNotUnionSubtypes(
1126               keepLoosenessOfThis, getObjs(), other.getObjs(),
1127               subSuperMap, boxedInfo);
1128         }
1129         return result;
1130       }

1132       private static void whyNotUnionSubtypes(
1133           JSType found, JSType expected, MismatchInfo[] boxedInfo) {
1134         if (NUMBER.isSubtypeOf(found) && !NUMBER.isSubtypeOf(expected)) {
1135           boxedInfo[0] = MismatchInfo.makeUnionTypeMismatch(NUMBER);
1136         } else if (STRING.isSubtypeOf(found) && !STRING.isSubtypeOf(expected)) {
1137           boxedInfo[0] = MismatchInfo.makeUnionTypeMismatch(STRING);
1138         } else if (BOOLEAN.isSubtypeOf(found) && !BOOLEAN.isSubtypeOf(expected)) {
1139           boxedInfo[0] = MismatchInfo.makeUnionTypeMismatch(BOOLEAN);
1140         } else if (NULL.isSubtypeOf(found) && !NULL.isSubtypeOf(expected)) {
1141           boxedInfo[0] = MismatchInfo.makeUnionTypeMismatch(NULL);
1142         } else if (UNDEFINED.isSubtypeOf(found) && !UNDEFINED.isSubtypeOf(expected)) {
1143           boxedInfo[0] = MismatchInfo.makeUnionTypeMismatch(UNDEFINED);
1144         } else if (found.hasTypeVariable() && !expected.hasTypeVariable()) {
1145           boxedInfo[0] = MismatchInfo.makeUnionTypeMismatch(
1146               fromTypeVar(found.getTypeVar()));
1147         } else if ((found.getMask() & NON_SCALAR_MASK) != 0
1148             && (expected.getMask() & NON_SCALAR_MASK) == 0) {
1149           boxedInfo[0] = MismatchInfo.makeUnionTypeMismatch(makeType(
1150               NON_SCALAR_MASK, found.getObjs(), null, NO_ENUMS));
1151         }
1152       }

1154       public JSType removeType(JSType other) {
1155         int otherMask = other.getMask();
1156         Preconditions.checkState(
1157             !other.isTop() && !other.isUnknown()
1158             && (otherMask & TYPEVAR_MASK) == 0 && (otherMask & ENUM_MASK) == 0,
1159             "Requested invalid type to remove: %s", other);
1160         if (isUnknown()) {
1161           return this;
1162         }
1163         if (isTop()) {
1164           return ALMOST_TOP.removeType(other);
1165         }
1166         int newMask = getMask() & ~otherMask;
1167         if ((otherMask & NON_SCALAR_MASK) == 0) {
1168           return newMask == getMask()
1169               ? this : makeType(newMask, getObjs(), getTypeVar(), getEnums());
1170         }
1171         // TODO(dimvar): If objs and enums stay unchanged, reuse, don't recreate.
1172         Preconditions.checkState(other.getObjs().size() == 1,
1173             "Invalid type to remove: %s", other);
1174         ObjectType otherObj = Iterables.getOnlyElement(other.getObjs());
1175         ImmutableSet.Builder<ObjectType> objsBuilder = ImmutableSet.builder();
1176         for (ObjectType obj : getObjs()) {
1177           if (!obj.isSubtypeOf(otherObj, SubtypeCache.create())) {
1178             objsBuilder.add(obj);
1179           }
1180         }
1181         ImmutableSet.Builder<EnumType> enumBuilder = ImmutableSet.builder();
1182         for (EnumType e : getEnums()) {
1183           if (!e.getEnumeratedType().isSubtypeOf(other, SubtypeCache.create())) {
1184             enumBuilder.add(e);
1185           }
1186         }
1187         return makeType(newMask, objsBuilder.build(), getTypeVar(), enumBuilder.build());
1188       }

1335       // For a type A, this method tries to return the greatest subtype of A that
1336       // has a property called pname. If it can't safely find a subtype, it
1337       // returns bottom.
1338       public JSType findSubtypeWithProp(QualifiedName pname) {
1339         Preconditions.checkArgument(pname.isIdentifier());
1340         // common cases first
1341         if (isTop() || isUnknown() || (getMask() & NON_SCALAR_MASK) == 0) {
1342           return BOTTOM;
1343         }
1344         // For simplicity, if this type has scalars with pname, return bottom.
1345         // If it has enums, return bottom.
1346         if (NUMBER.isSubtypeOf(this)
1347             && commonTypes.getNumberInstance().mayHaveProp(pname)
1348             || STRING.isSubtypeOf(this)
1349             && commonTypes.getNumberInstance().mayHaveProp(pname)
1350             || BOOLEAN.isSubtypeOf(this)
1351             && commonTypes.getBooleanInstance().mayHaveProp(pname)) {
1352           return BOTTOM;
1353         }
1354         if ((getMask() & ENUM_MASK) != 0) {
1355           return BOTTOM;
1356         }
1357         // NOTE(dimvar): Nothing for type variables for now, but we will have to
1358         // handle them here when we implement bounded generics.
1359         if (getObjs().size() == 1) {
1360           ObjectType obj = Iterables.getOnlyElement(getObjs());
1361           return obj.mayHaveProp(pname) ? this : BOTTOM;
1362         }
1363         ImmutableSet.Builder<ObjectType> builder = ImmutableSet.builder();
1364         boolean foundObjWithProp = false;
1365         for (ObjectType o : getObjs()) {
1366           if (o.mayHaveProp(pname)) {
1367             foundObjWithProp = true;
1368             builder.add(o);
1369           }
1370         }
1371         return foundObjWithProp
1372             ? makeType(NON_SCALAR_MASK, builder.build(), null, NO_ENUMS) : BOTTOM;
1373       }

1387       /** For use in {@link #typeToString} */
1388       private static final Joiner PIPE_JOINER = Joiner.on("|");
1389     
1390       private StringBuilder typeToString(StringBuilder builder) {
1391         switch (getMask()) {
1392           case BOTTOM_MASK:
1393             return builder.append("bottom");
1394           case TOP_MASK:
1395             return builder.append("*");
1396           case UNKNOWN_MASK:
1397             return builder.append("?");
1398           default:
1399             int tags = getMask();
1400             boolean firstIteration = true;
1401             for (int tag = 1; tag != END_MASK; tag <<= 1) {
1402               if ((tags & tag) != 0) {
1403                 if (!firstIteration) {
1404                   builder.append('|');
1405                 }
1406                 firstIteration = false;
1407                 switch (tag) {
1408                   case TRUE_MASK:
1409                   case FALSE_MASK:
1410                     builder.append("boolean");
1411                     tags &= ~BOOLEAN_MASK;
1412                     continue;
1413                   case NULL_MASK:
1414                     builder.append("null");
1415                     tags &= ~NULL_MASK;
1416                     continue;
1417                   case NUMBER_MASK:
1418                     builder.append("number");
1419                     tags &= ~NUMBER_MASK;
1420                     continue;
1421                   case STRING_MASK:
1422                     builder.append("string");
1423                     tags &= ~STRING_MASK;
1424                     continue;
1425                   case UNDEFINED_MASK:
1426                     builder.append("undefined");
1427                     tags &= ~UNDEFINED_MASK;
1428                     continue;
1429                   case TYPEVAR_MASK:
1430                     builder.append(UniqueNameGenerator.getOriginalName(getTypeVar()));
1431                     tags &= ~TYPEVAR_MASK;
1432                     continue;
1433                   case NON_SCALAR_MASK: {
1434                     if (getObjs().size() == 1) {
1435                       Iterables.getOnlyElement(getObjs()).appendTo(builder);
1436                     } else {
1437                       Set<String> strReps = new TreeSet<>();
1438                       for (ObjectType obj : getObjs()) {
1439                         strReps.add(obj.toString());
1440                       }
1441                       PIPE_JOINER.appendTo(builder, strReps);
1442                     }
1443                     tags &= ~NON_SCALAR_MASK;
1444                     continue;
1445                   }
1446                   case ENUM_MASK: {
1447                     if (getEnums().size() == 1) {
1448                       builder.append(Iterables.getOnlyElement(getEnums()));
1449                     } else {
1450                       Set<String> strReps = new TreeSet<>();
1451                       for (EnumType e : getEnums()) {
1452                         strReps.add(e.toString());
1453                       }
1454                       PIPE_JOINER.appendTo(builder, strReps);
1455                     }
1456                     tags &= ~ENUM_MASK;
1457                     continue;
1458                   }
1459                 }
1460               }
1461             }
1462             if (tags == 0) { // Found all types in the union
1463               return builder;
1464             } else if (tags == TRUTHY_MASK) {
1465               return builder.append("truthy");
1466             } else if (tags == FALSY_MASK) {
1467               return builder.append("falsy");
1468             } else {
1469               return builder.append("Unrecognized type: " + tags);
1470             }
1471         }
1472       }
