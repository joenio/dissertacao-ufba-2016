40     package com.google.javascript.rhino;
41     
42     /**
43      * Represents a position in some piece of source code, with an associated
44      * item of type T found at that position.
45      *
46      */
47     public abstract class SourcePosition<T> {
48       /**
49        * The (well typed) item found at the source position.
50        */
51       private T item = null;
52     
53       /**
54        * The starting line number.
55        */
56       private int startLineno = 0;
57     
58       /**
59        * The character position on the starting line.
60        */
61       private int startCharno = 0;
62     
63       /**
64        * The ending line number.
65        */
66       private int endLineno = 0;
67     
68       /**
69        * The character position on the ending line.
70        */
71       private int endCharno = 0;
72     
73       /**
74        * Sets the item that this source position references.
75        */
76       public void setItem(T item) {
77         this.item = item;
78       }
79     
80       /**
81        * Sets the position information contained in this source position.
82        */
83       public void setPositionInformation(int startLineno, int startCharno,
84                                          int endLineno, int endCharno) {
85         if (startLineno > endLineno) {
86           throw new IllegalStateException(
87               "Recorded bad position information\n"
88               + "start-line: " + startLineno + "\n"
89               + "end-line: " + endLineno);
90         } else if (startLineno == endLineno && startCharno >= endCharno) {
91           throw new IllegalStateException(
92               "Recorded bad position information\n"
93               + "line: " + startLineno + "\n"
94               + "start-char: " + startCharno + "\n"
95               + "end-char: " + endCharno);
96         }
97     
98         this.startLineno = startLineno;
99         this.startCharno = startCharno;
100         this.endLineno = endLineno;
101         this.endCharno = endCharno;
102       }
103     
104       /**
105        * Returns the item found at this source position.
106        */
107       public T getItem() {
108         return item;
109       }
110     
111       /**
112        * Returns the starting line number of this position.
113        */
114       public int getStartLine() {
115         return startLineno;
116       }
117     
118       /**
119        * Returns the character position on the starting line.
120        */
121       public int getPositionOnStartLine() {
122         return startCharno;
123       }
124     
125       /**
126        * Returns the ending line number of this position.
127        */
128       public int getEndLine() {
129         return endLineno;
130       }
131     
132       /**
133        * Returns the character position on the ending line.
134        */
135       public int getPositionOnEndLine() {
136         return endCharno;
137       }
138     }
