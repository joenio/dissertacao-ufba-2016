41     /**
42      * Checks for various JSDoc-related style issues, such as function definitions without JsDoc, params
43      * with no corresponding {@code @param} annotation, coding conventions not being respected, etc.
44      */
45     public final class CheckJSDocStyle extends AbstractPostOrderCallback implements CompilerPass {

47           DiagnosticType.disabled(
48               "JSC_INVALID_SUPPRESS",
49               "@suppress annotation not allowed here. See"
50                   + " https://github.com/google/closure-compiler/wiki/@suppress-annotations");

53           DiagnosticType.disabled("JSC_CONSTRUCTOR_DISALLOWED_JSDOC",
54               "Visibility annotations on constructors are not supported.\n"
55               + "Please mark the visibility on the class instead.");

58           DiagnosticType.disabled("JSC_MISSING_JSDOC", "Function must have JSDoc.");

61           DiagnosticType.disabled("JSC_MISSING_PARAMETER_JSDOC", "Parameter must have JSDoc.");

64           DiagnosticType.disabled("JSC_MIXED_PARAM_JSDOC_STYLES",

126       @Override
127       public void visit(NodeTraversal t, Node n, Node parent) {
128         switch (n.getType()) {
129           case FUNCTION:
130             visitFunction(t, n, parent);
131             break;
132           case ASSIGN:
133             // If the right side is a function it will be handled when the function is visited.
134             if (!n.getLastChild().isFunction()) {
135               visitNonFunction(t, n);
136             }
137             checkStyleForPrivateProperties(t, n);
138             break;
139           case VAR:
140           case LET:
141           case CONST:
142             for (Node decl : n.children()) {
143               // If the right side is a function it will be handled when the function is visited.
144               if (decl.getFirstChild() == null || !decl.getFirstChild().isFunction()) {
145                 visitNonFunction(t, n);
146               }
147             }
148             break;
149           case STRING_KEY:
150             // If the value is a function it will be handled when the function is visited.
151             if (n.getFirstChild() == null || !n.getFirstChild().isFunction()) {
152               visitNonFunction(t, n);
153             }
154             break;
155           case MEMBER_FUNCTION_DEF:
156           case GETTER_DEF:
157           case SETTER_DEF:
158             // Don't need to call visitFunction because this JSDoc will be visited when the function is
159             // visited.
160             if (NodeUtil.getEnclosingClass(n) != null) {
161               checkStyleForPrivateProperties(t, n);
162             }
163             break;
164           default:
165             visitNonFunction(t, n);
166         }
167       }
168     
169       private void visitNonFunction(NodeTraversal t, Node n) {
170         JSDocInfo jsDoc = n.getJSDocInfo();
171         if (jsDoc == null) {
172           return;
173         }
174     
175         if (!n.isScript()) {
176           checkSuppressionsOnNonFunction(t, n, jsDoc);
177         }
178       }
179     
180       private void checkStyleForPrivateProperties(NodeTraversal t, Node n) {
181         JSDocInfo jsDoc = NodeUtil.getBestJSDocInfo(n);
182         String name;
183         if (n.isMemberFunctionDef() || n.isGetterDef() || n.isSetterDef()) {
184           name = n.getString();
185         } else {
186           Preconditions.checkState(n.isAssign());
187           Node lhs = n.getFirstChild();
188           if (!lhs.isGetProp()) {
189             return;
190           }
191           name = lhs.getLastChild().getString();
192         }
193         if (name.equals("constructor")) {
194           return;
195         }
196     
197         if (jsDoc != null && name != null) {
198           if (compiler.getCodingConvention().isPrivate(name)
199               && !jsDoc.getVisibility().equals(Visibility.PRIVATE)) {
200             t.report(n, MUST_BE_PRIVATE, name);
201           } else if (compiler.getCodingConvention().hasPrivacyConvention()
202               && !compiler.getCodingConvention().isPrivate(name)
203               && jsDoc.getVisibility().equals(Visibility.PRIVATE)) {
204             t.report(n, MUST_HAVE_TRAILING_UNDERSCORE, name);
205           }
206         }
207       }
208       private void checkSuppressionsOnNonFunction(NodeTraversal t, Node n, JSDocInfo jsDoc) {
209         // Suppressions that are allowed to be in places other than functions and @fileoverview blocks.
210         Set<String> specialSuppressions =
211             ImmutableSet.of("const", "duplicate", "extraRequire", "missingRequire");
212     
213         Set<String> suppressions = Sets.difference(jsDoc.getSuppressions(), specialSuppressions);
214         if (!suppressions.isEmpty()) {
215           t.report(n, INVALID_SUPPRESS);
216         }
217       }

219       private void visitFunction(NodeTraversal t, Node function, Node parent) {
220         JSDocInfo jsDoc = NodeUtil.getBestJSDocInfo(function);
221     
222         if (jsDoc == null && !hasAnyInlineJsDoc(function)) {
223           checkMissingJsDoc(t, function);
224         } else {
225           if (t.inGlobalScope()
226               || hasAnyInlineJsDoc(function)
227               || !jsDoc.getParameterNames().isEmpty()
228               || jsDoc.hasReturnType()) {
229             checkParams(t, function, jsDoc);
230           }
231           checkReturn(t, function, jsDoc);
232         }
233     
234         if (parent.isMemberFunctionDef()
235             && "constructor".equals(parent.getString())
236             && jsDoc != null
237             && !jsDoc.getVisibility().equals(Visibility.INHERITED)) {
238           t.report(function, CONSTRUCTOR_DISALLOWED_JSDOC);
239         }
240       }
241     
242       private void checkMissingJsDoc(NodeTraversal t, Node function) {
243         if (isFunctionThatShouldHaveJsDoc(t, function)) {
244           String name = NodeUtil.getName(function);
245           // Don't warn for test functions, setUp, tearDown, etc.
246           if (name == null || !ExportTestFunctions.isTestFunction(name)) {
247             t.report(function, MISSING_JSDOC);
248           }
249         }
250       }

252       /**
253        * Whether the given function should have JSDoc. True if it's a function declared
254        * in the global scope, or a method on a class which is declared in the global scope.
255        */
256       private boolean isFunctionThatShouldHaveJsDoc(NodeTraversal t, Node function) {
257         if (!t.inGlobalHoistScope()) {
258           return false;
259         }
260         if (NodeUtil.isFunctionDeclaration(function)) {
261           return true;
262         }
263         if (NodeUtil.isNameDeclaration(function.getGrandparent()) || function.getParent().isAssign()) {
264           return true;
265         }
266     
267         if (function.getGrandparent().isClassMembers()) {
268           Node memberNode = function.getParent();
269           if (memberNode.isMemberFunctionDef()) {
270             // A constructor with no parameters doesn't need JSDoc,
271             // but all other member functions do.
272             return !isConstructorWithoutParameters(function);
273           } else if (memberNode.isGetterDef() || memberNode.isSetterDef()) {
274             return true;
275           }
276         }
277     
278         return false;
279       }
280     
281       private boolean isConstructorWithoutParameters(Node function) {
282         return function.getParent().matchesQualifiedName("constructor")
283             && !NodeUtil.getFunctionParameters(function).hasChildren();
284       }

286       private void checkParams(NodeTraversal t, Node function, JSDocInfo jsDoc) {
287         if (jsDoc != null && jsDoc.isOverride()) {
288           return;
289         }
290     
291         if (jsDoc != null && jsDoc.getType() != null) {
292           // Sometimes functions are declared with @type {function(Foo, Bar)} instead of
293           //   @param {Foo} foo
294           //   @param {Bar} bar
295           // which is fine.
296           return;
297         }
298     
299         List<String> paramsFromJsDoc =
300             jsDoc == null
301                 ? ImmutableList.<String>of()
302                 : ImmutableList.<String>copyOf(jsDoc.getParameterNames());
303         if (paramsFromJsDoc.isEmpty()) {
304           checkInlineParams(t, function);
305         } else {
306           Node paramList = NodeUtil.getFunctionParameters(function);
307           if (paramsFromJsDoc.size() != paramList.getChildCount()) {
308             t.report(paramList, WRONG_NUMBER_OF_PARAMS);
309             return;
310           }
311     
312           Node param = paramList.getFirstChild();
313           for (int i = 0; i < paramsFromJsDoc.size(); i++) {
314             if (param.getJSDocInfo() != null) {
315               t.report(param, MIXED_PARAM_JSDOC_STYLES);
316             }
317             String name = paramsFromJsDoc.get(i);
318             JSTypeExpression paramType = jsDoc.getParameterType(name);
319             if (checkParam(t, param, name, paramType)) {
320               return;
321             }
322             param = param.getNext();
323           }
324         }
325       }

327       /**
328        * Checks that the inline type annotations are correct.
329        */
330       private void checkInlineParams(NodeTraversal t, Node function) {
331         Node paramList = NodeUtil.getFunctionParameters(function);
332     
333         for (Node param : paramList.children()) {
334           JSDocInfo jsDoc = param.getJSDocInfo();
335           if (jsDoc == null) {
336             t.report(param, MISSING_PARAMETER_JSDOC);
337             return;
338           } else {
339             JSTypeExpression paramType = jsDoc.getType();
340             Preconditions.checkNotNull(paramType, "Inline JSDoc info should always have a type");
341             checkParam(t, param, null, paramType);
342           }
343         }
344       }

346       /**
347        * Checks that the given parameter node has the given name, and that the given type is
348        * compatible.
349        * @param param If this is a non-NAME node, such as a destructuring pattern, skip the name check.
350        * @param name If null, skip the name check
351        * @return Whether a warning was reported
352        */
353       private boolean checkParam(
354           NodeTraversal t, Node param, @Nullable String name, JSTypeExpression paramType) {
355         boolean nameOptional;
356         Node nodeToCheck = param;
357         if (param.isDefaultValue()) {
358           nodeToCheck = param.getFirstChild();
359           nameOptional = true;
360         } else if (param.isName()) {
361           nameOptional = param.getString().startsWith("opt_");
362         } else {
363           Preconditions.checkState(param.isDestructuringPattern() || param.isRest(), param);
364           nameOptional = false;
365         }
366     
367         if (name == null || !nodeToCheck.isName()) {
368           // Skip the name check, but use "<unknown name>" for other errors that might be reported.
369           name = "<unknown name>";
370         } else if (!nodeToCheck.matchesQualifiedName(name)) {
371           t.report(nodeToCheck, INCORRECT_PARAM_NAME);
372           return true;
373         }
374     
375         boolean jsDocOptional = paramType != null && paramType.isOptionalArg();
376         if (nameOptional && !jsDocOptional) {
377           t.report(nodeToCheck, OPTIONAL_PARAM_NOT_MARKED_OPTIONAL, name);
378           return true;
379         } else if (!nameOptional && jsDocOptional) {
380           t.report(nodeToCheck, OPTIONAL_TYPE_NOT_USING_OPTIONAL_NAME, name);
381           return true;
382         }
383         return false;
384       }

386       private boolean hasAnyInlineJsDoc(Node function) {
387         if (function.getFirstChild().getJSDocInfo() != null) {
388           // Inline return annotation.
389           return true;
390         }
391         for (Node param : NodeUtil.getFunctionParameters(function).children()) {
392           if (param.getJSDocInfo() != null) {
393             return true;
394           }
395         }
396         return false;
397       }

399       private void checkReturn(NodeTraversal t, Node function, JSDocInfo jsDoc) {
400         if (jsDoc != null
401             && (jsDoc.hasType()
402                 || jsDoc.hasReturnType()
403                 || jsDoc.isOverride())) {
404           return;
405         }
406         if (function.getFirstChild().getJSDocInfo() != null) {
407           return;
408         }
409     
410         FindNonTrivialReturn finder = new FindNonTrivialReturn();
411         NodeTraversal.traverseEs6(compiler, function.getLastChild(), finder);
412         if (finder.found) {
413           t.report(function, MISSING_RETURN_JSDOC);
414         }
415       }
416     
417       private static class FindNonTrivialReturn extends AbstractPreOrderCallback {
418         private boolean found;
419     
420         @Override
421         public boolean shouldTraverse(NodeTraversal t, Node n, Node parent) {
422           if (found) {
423             return false;
424           }
425     
426           // Shallow traversal, since we don't need to inspect within functions or expressions.
427           if (parent == null
428               || NodeUtil.isControlStructure(parent)
429               || NodeUtil.isStatementBlock(parent)) {
430             if (n.isReturn() && n.hasChildren()) {
431               found = true;
432               return false;
433             }
434             return true;
435           }
436           return false;
437         }
438       }

440       private static class ExternsCallback implements NodeTraversal.Callback {
441         @Override
442         public boolean shouldTraverse(NodeTraversal t, Node n, Node parent) {
443           return parent == null || n.isScript();
444         }
445     
446         @Override
447         public void visit(NodeTraversal t, Node n, Node parent) {
448           if (n.isScript()) {
449             JSDocInfo info = n.getJSDocInfo();
450             if (info == null || !info.isExterns()) {
451               t.report(n, EXTERNS_FILES_SHOULD_BE_ANNOTATED);
452             }
453           }
454         }
455       }
