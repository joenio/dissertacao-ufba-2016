634       /**
635        * Wrapper that gathers function, call, and self variable type strings after
636        * the pass under test runs.  For use to test passes that modify JSType
637        * annotations.
638        */
639       private static class TypeInformationGatherer
640           implements CompilerPass {
641         private final Compiler compiler;
642         private final CompilerPass passUnderTest;
643         private final List<String> typeInformation;
644     
645         TypeInformationGatherer(Compiler compiler,
646                                         CompilerPass passUnderTest,
647                                         List<String> typeInformation) {
648           this.compiler = compiler;
649           this.passUnderTest = passUnderTest;
650           this.typeInformation = typeInformation;
651         }
652     
653         @Override
654         public void process(Node externs, Node root) {
655           passUnderTest.process(externs, root);
656           NodeTraversal.traverseEs6(compiler, externs, new GatherCallback());
657           NodeTraversal.traverseEs6(compiler, root, new GatherCallback());
658         }
659     
660         public String getNameString(Node n) {
661           Token type = n.getType();
662           if (type == Token.NAME) {
663             return n.getString();
664           } else if (type == Token.GETPROP) {
665             String left = getNameString(n.getFirstChild());
666             if (left == null) {
667               return null;
668             }
669             return left + "." + n.getLastChild().getString();
670           } else if (type == Token.GETELEM) {
671             String left = getNameString(n.getFirstChild());
672             if (left == null) {
673               return null;
674             }
675             return left + "[" + n.getLastChild().getString() + "]";
676           } else if (type == Token.THIS) {
677             return "this";
678           } else if (type == Token.FUNCTION){
679             return "{ANON FUNCTION}";
680           } else {
681             // I wonder if we should just die on this.
682             return null;
683           }
684         }

686         private class GatherCallback extends AbstractPostOrderCallback {
687           @Override
688           public void visit(NodeTraversal traversal, Node node, Node parent) {
689             Node nameNode = null;
690             if (node.isFunction()) {
691               if (parent.isName()) {
692                 nameNode = parent;
693               } else if (parent.isAssign()) {
694                 nameNode = parent.getFirstChild();
695               } else {
696                 nameNode = node.getFirstChild();
697               }
698             } else if (node.isCall() || node.isNew()) {
699               nameNode = node.getFirstChild();
700             }
701     
702             if (nameNode != null) {
703               JSType type = node.getJSType();
704               typeInformation.add(
705                   Joiner.on("")
706                       .join(node.getType(), " ", getNameString(nameNode), " = ", String.valueOf(type)));
707             }
708     
709             if (node.isGetProp()) {
710               Node child = node.getFirstChild();
711               if (child.isName() && child.getString().endsWith("$self")) {
712                 JSType type = child.getJSType();
713                 typeInformation.add(
714                     Joiner.on("")
715                         .join(child.getType(), " ", child.getString(), " = ", String.valueOf(type)));
716               }
717             }
718           }
719         }
720       }
721     }
