43     /**
44      * Flattens global objects/namespaces by replacing each '.' with '$' in
45      * their names. This reduces the number of property lookups the browser has
46      * to do and allows the {@link RenameVars} pass to shorten namespaced names.
47      * For example, goog.events.handleEvent() -> goog$events$handleEvent() -> Za().
48      *
49      * <p>If a global object's name is assigned to more than once, or if a property
50      * is added to the global object in a complex expression, then none of its
51      * properties will be collapsed (for safety/correctness).
52      *
53      * <p>If, after a global object is declared, it is never referenced except when
54      * its properties are read or set, then the object will be removed after its
55      * properties have been collapsed.
56      *
57      * <p>Uninitialized variable stubs are created at a global object's declaration
58      * site for any of its properties that are added late in a local scope.
59      *
60      * <p> Static properties of constructors are always collapsed, unsafely!
61      * For other objects: if, after an object is declared, it is referenced directly
62      * in a way that might create an alias for it, then none of its properties will
63      * be collapsed.
64      * This behavior is a safeguard to prevent the values associated with the
65      * flattened names from getting out of sync with the object's actual property
66      * values. For example, in the following case, an alias a$b, if created, could
67      * easily keep the value 0 even after a.b became 5:
68      * <code> a = {b: 0}; c = a; c.b = 5; </code>.
69      *
70      * <p>This pass doesn't flatten property accesses of the form: a[b].
71      *
72      * <p>For lots of examples, see the unit test.
73      *
74      */
75     class CollapseProperties implements CompilerPass {

109       @Override
110       public void process(Node externs, Node root) {
111         GlobalNamespace namespace;
112         namespace = new GlobalNamespace(compiler, root);
113         inlineAliases(namespace);
114         nameMap = namespace.getNameIndex();
115         globalNames = namespace.getNameForest();
116         checkNamespaces();
117     
118         for (Name name : globalNames) {
119           flattenReferencesToCollapsibleDescendantNames(name, name.getBaseName());
120         }
121     
122         // We collapse property definitions after collapsing property references
123         // because this step can alter the parse tree above property references,
124         // invalidating the node ancestry stored with each reference.
125         for (Name name : globalNames) {
126           collapseDeclarationOfNameAndDescendants(name, name.getBaseName());
127         }
128       }

130       /**
131        * For each qualified name N in the global scope, we check if:
132        * (a) No ancestor of N is ever aliased or assigned an unknown value type.
133        *     (If N = "a.b.c", "a" and "a.b" are never aliased).
134        * (b) N has exactly one write, and it lives in the global scope.
135        * (c) N is aliased in a local scope.
136        * (d) N is aliased in global scope
137        *
138        * If (a) is true, then GlobalNamespace must know all the writes to N.
139        * If (a) and (b) are true, then N cannot change during the execution of
140        *    a local scope.
141        * If (a) and (b) and (c) are true, then the alias can be inlined if the
142        *    alias obeys the usual rules for how we decide whether a variable is
143        *    inlineable.
144        * If (a) and (b) and (d) are true, then inline the alias if possible (if
145        * it is assigned exactly once unconditionally).
146        * @see InlineVariables
147        */
148       private void inlineAliases(GlobalNamespace namespace) {
149         // Invariant: All the names in the worklist meet condition (a).
150         Deque<Name> workList = new ArrayDeque<>(namespace.getNameForest());
151     
152         while (!workList.isEmpty()) {
153           Name name = workList.pop();
154     
155           // Don't attempt to inline a getter or setter property as a variable.
156           if (name.type == Name.Type.GET || name.type == Name.Type.SET) {
157             continue;
158           }
159     
160           if (!name.inExterns && name.globalSets == 1 && name.localSets == 0 &&
161               name.aliasingGets > 0) {
162             // {@code name} meets condition (b). Find all of its local aliases
163             // and try to inline them.
164             List<Ref> refs = new ArrayList<>(name.getRefs());
165             for (Ref ref : refs) {
166               if (ref.type == Type.ALIASING_GET && ref.scope.isLocal()) {
167                 // {@code name} meets condition (c). Try to inline it.
168                 // TODO(johnlenz): consider picking up new aliases at the end
169                 // of the pass instead of immediately like we do for global
170                 // inlines.
171                 if (inlineAliasIfPossible(name, ref, namespace)) {
172                   name.removeRef(ref);
173                 }
174               } else if (ref.type == Type.ALIASING_GET
175                   && ref.scope.isGlobal()
176                   && ref.getTwin() == null) { // ignore aliases in chained assignments
177                 if (inlineGlobalAliasIfPossible(name, ref, namespace)) {
178                   name.removeRef(ref);
179                 }
180               }
181             }
182           }
183     
184           // Check if {@code name} has any aliases left after the
185           // local-alias-inlining above.
186           if ((name.type == Name.Type.OBJECTLIT ||
187                name.type == Name.Type.FUNCTION) &&
188               name.aliasingGets == 0 && name.props != null) {
189             // All of {@code name}'s children meet condition (a), so they can be
190             // added to the worklist.
191             workList.addAll(name.props);
192           }
193         }
194       }

196       /**
197        * Attempt to inline an global alias of a global name. This requires that
198        * the name is well defined: assigned unconditionally, assigned exactly once.
199        * It is assumed that, the name for which it is an alias must already
200        * meet these same requirements.
201        *
202        * @param alias The alias to inline
203        * @return Whether the alias was inlined.
204        */
205       private boolean inlineGlobalAliasIfPossible(
206           Name name, Ref alias, GlobalNamespace namespace) {
207         // Ensure that the alias is assigned to global name at that the
208         // declaration.
209         Node aliasParent = alias.node.getParent();
210         if ((aliasParent.isAssign() || aliasParent.isName())
211             && NodeUtil.isExecutedExactlyOnce(aliasParent)
212             // We special-case for constructors here, to inline constructor aliases
213             // more aggressively in global scope.
214             // We do this because constructor properties are always collapsed,
215             // so we want to inline the aliases also to avoid breakages.
216             // TODO(tbreisacher): Do we still need this special case?
217             || aliasParent.isName() && name.isConstructor()) {
218           Node lvalue = aliasParent.isName() ? aliasParent : aliasParent.getFirstChild();
219           if (!lvalue.isQualifiedName()) {
220             return false;
221           }
222           if (lvalue.isName()
223               && compiler.getCodingConvention().isExported(
224                   lvalue.getString(),
225                   /* local */ false)) {
226             return false;
227           }
228           name = namespace.getSlot(lvalue.getQualifiedName());
229           if (name != null && name.isInlinableGlobalAlias()) {
230             Set<AstChange> newNodes = new LinkedHashSet<>();
231     
232             List<Ref> refs = new ArrayList<>(name.getRefs());
233             for (Ref ref : refs) {
234               switch (ref.type) {
235                 case SET_FROM_GLOBAL:
236                   continue;
237                 case DIRECT_GET:
238                 case ALIASING_GET:
239                 case PROTOTYPE_GET:
240                 case CALL_GET:
241                   Node newNode = alias.node.cloneTree();
242                   Node node = ref.node;
243                   node.getParent().replaceChild(node, newNode);
244                   newNodes.add(new AstChange(ref.module, ref.scope, newNode));
245                   name.removeRef(ref);
246                   break;
247                 default:
248                   throw new IllegalStateException();
249               }
250             }
251     
252             rewriteAliasProps(name, alias.node, 0, newNodes);
253     
254             // just set the original alias to null.
255             aliasParent.replaceChild(alias.node, IR.nullNode());
256             compiler.reportCodeChange();
257     
258             // Inlining the variable may have introduced new references
259             // to descendants of {@code name}. So those need to be collected now.
260             namespace.scanNewNodes(newNodes);
261     
262             return true;
263           }
264         }
265         return false;
266       }

268       /**
269        * @param name The Name whose properties references should be updated.
270        * @param value The value to use when rewriting.
271        * @param depth The chain depth.
272        * @param newNodes Expression nodes that have been updated.
273        */
274       private static void rewriteAliasProps(
275           Name name, Node value, int depth, Set<AstChange> newNodes) {
276         if (name.props == null) {
277           return;
278         }
279         Preconditions.checkState(
280             !value.matchesQualifiedName(name.getFullName()),
281             "%s should not match name %s", value, name.getFullName());
282         for (Name prop : name.props) {
283           rewriteAliasProps(prop, value, depth + 1, newNodes);
284           List<Ref> refs = new ArrayList<>(prop.getRefs());
285           for (Ref ref : refs) {
286             Node target = ref.node;
287             for (int i = 0; i <= depth; i++) {
288               if (target.isGetProp()) {
289                 target = target.getFirstChild();
290               } else if (NodeUtil.isObjectLitKey(target)) {
291                 // Object literal key definitions are a little trickier, as we
292                 // need to find the assignment target
293                 Node gparent = target.getGrandparent();
294                 if (gparent.isAssign()) {
295                   target = gparent.getFirstChild();
296                 } else {
297                   Preconditions.checkState(NodeUtil.isObjectLitKey(gparent));
298                   target = gparent;
299                 }
300               } else {
301                 throw new IllegalStateException("unexpected: " + target);
302               }
303             }
304             Preconditions.checkState(target.isGetProp() || target.isName());
305             target.getParent().replaceChild(target, value.cloneTree());
306             prop.removeRef(ref);
307             // Rescan the expression root.
308             newNodes.add(new AstChange(ref.module, ref.scope, ref.node));
309           }
310         }
311       }

313       private boolean inlineAliasIfPossible(
314           Name name, Ref alias, GlobalNamespace namespace) {
315         // Ensure that the alias is assigned to a local variable at that
316         // variable's declaration. If the alias's parent is a NAME,
317         // then the NAME must be the child of a VAR node, and we must
318         // be in a VAR assignment.
319         Node aliasParent = alias.node.getParent();
320         if (aliasParent.isName()) {
321           // Ensure that the local variable is well defined and never reassigned.
322           Scope scope = alias.scope;
323           String aliasVarName = aliasParent.getString();
324           Var aliasVar = scope.getVar(aliasVarName);
325     
326           ReferenceCollectingCallback collector =
327               new ReferenceCollectingCallback(compiler,
328                   ReferenceCollectingCallback.DO_NOTHING_BEHAVIOR,
329                   Predicates.equalTo(aliasVar));
330           collector.processScope(scope);
331     
332           ReferenceCollection aliasRefs = collector.getReferences(aliasVar);
333           Set<AstChange> newNodes = new LinkedHashSet<>();
334     
335           if (aliasRefs.isWellDefined()
336               && aliasRefs.firstReferenceIsAssigningDeclaration()) {
337             if (!aliasRefs.isAssignedOnceInLifetime()) {
338               // Static properties of constructors are always collapsed.
339               // So, if a constructor is aliased and its properties are accessed from
340               // the alias, we would like to inline the alias here to access the
341               // properties correctly.
342               // But if the aliased variable is assigned more than once, we can't
343               // inline, so we warn.
344               if (name.isConstructor()) {
345                 boolean accessPropsAfterAliasing = false;
346                 for (Reference ref : aliasRefs.references) {
347                   if (ref.getNode().getParent().isGetProp()) {
348                     accessPropsAfterAliasing = true;
349                     break;
350                   }
351                 }
352                 if (accessPropsAfterAliasing) {
353                   compiler.report(
354                       JSError.make(aliasParent, UNSAFE_CTOR_ALIASING, aliasVarName));
355                 }
356               }
357               return false;
358             }
359     
360             // The alias is well-formed, so do the inlining now.
361             int size = aliasRefs.references.size();
362             for (int i = 1; i < size; i++) {
363               ReferenceCollectingCallback.Reference aliasRef =
364                   aliasRefs.references.get(i);
365     
366               Node newNode = alias.node.cloneTree();
367               aliasRef.getParent().replaceChild(aliasRef.getNode(), newNode);
368               newNodes.add(new AstChange(
369                   getRefModule(aliasRef), aliasRef.getScope(), newNode));
370             }
371     
372             // just set the original alias to null.
373             aliasParent.replaceChild(alias.node, IR.nullNode());
374             compiler.reportCodeChange();
375     
376             // Inlining the variable may have introduced new references
377             // to descendants of {@code name}. So those need to be collected now.
378             namespace.scanNewNodes(newNodes);
379             return true;
380           }
381         }
382     
383         return false;
384       }

388         return input == null ? null : input.getModule();

391       /**
392        * Runs through all namespaces (prefixes of classes and enums), and checks if
393        * any of them have been used in an unsafe way.
394        */
395       private void checkNamespaces() {
396         for (Name name : nameMap.values()) {
397           if (name.isNamespaceObjectLit() &&
398               (name.aliasingGets > 0 || name.localSets + name.globalSets > 1 ||
399                name.deleteProps > 0)) {
400             boolean initialized = name.getDeclaration() != null;
401             for (Ref ref : name.getRefs()) {
402               if (ref == name.getDeclaration()) {
403                 continue;
404               }
405     
406               if (ref.type == Ref.Type.DELETE_PROP) {
407                 if (initialized) {
408                   warnAboutNamespaceRedefinition(name, ref);
409                 }
410               } else if (
411                   ref.type == Ref.Type.SET_FROM_GLOBAL
412                   || ref.type == Ref.Type.SET_FROM_LOCAL) {
413                 if (initialized && !isSafeNamespaceReinit(ref)) {
414                   warnAboutNamespaceRedefinition(name, ref);
415                 }
416     
417                 initialized = true;
418               } else if (ref.type == Ref.Type.ALIASING_GET) {
419                 warnAboutNamespaceAliasing(name, ref);
420               }
421             }
422           }
423         }
424       }

426       private boolean isSafeNamespaceReinit(Ref ref) {
427         // allow "a = a || {}" or "var a = a || {}"
428         Node valParent = getValueParent(ref);
429         Node val = valParent.getLastChild();
430         if (val.getType() == Token.OR) {
431           Node maybeName = val.getFirstChild();
432           if (ref.node.matchesQualifiedName(maybeName)) {
433             return true;
434           }
435         }
436     
437         return false;
438       }

477       /**
478        * Flattens all references to collapsible properties of a global name except
479        * their initial definitions. Recurs on subnames.
480        *
481        * @param n An object representing a global name
482        * @param alias The flattened name for {@code n}
483        */
484       private void flattenReferencesToCollapsibleDescendantNames(
485           Name n, String alias) {
486         if (n.props == null || n.isCollapsingExplicitlyDenied()) {
487           return;
488         }
489     
490         for (Name p : n.props) {
491           String propAlias = appendPropForAlias(alias, p.getBaseName());
492     
493           if (p.canCollapse()) {
494             flattenReferencesTo(p, propAlias);
495           } else if (p.isSimpleStubDeclaration() && !p.isCollapsingExplicitlyDenied()) {
496             flattenSimpleStubDeclaration(p, propAlias);
497           }
498     
499           flattenReferencesToCollapsibleDescendantNames(p, propAlias);
500         }
501       }

503       /**
504        * Flattens a stub declaration.
505        * This is mostly a hack to support legacy users.
506        */
507       private void flattenSimpleStubDeclaration(Name name, String alias) {
508         Ref ref = Iterables.getOnlyElement(name.getRefs());
509         Node nameNode = NodeUtil.newName(
510             compiler, alias, ref.node,
511             name.getFullName());
512         Node varNode = IR.var(nameNode).useSourceInfoIfMissingFrom(nameNode);
513     
514         Preconditions.checkState(ref.node.getParent().isExprResult());
515         Node parent = ref.node.getParent();
516         Node grandparent = parent.getParent();
517         grandparent.replaceChild(parent, varNode);
518         compiler.reportCodeChange();
519       }

521       /**
522        * Flattens all references to a collapsible property of a global name except
523        * its initial definition.
524        *
525        * @param n A global property name (e.g. "a.b" or "a.b.c.d")
526        * @param alias The flattened name (e.g. "a$b" or "a$b$c$d")
527        */
528       private void flattenReferencesTo(Name n, String alias) {
529         String originalName = n.getFullName();
530         for (Ref r : n.getRefs()) {
531           if (r == n.getDeclaration()) {
532             // Declarations are handled separately.
533             continue;
534           }
535           Node rParent = r.node.getParent();
536           // There are two cases when we shouldn't flatten a reference:
537           // 1) Object literal keys, because duplicate keys show up as refs.
538           // 2) References inside a complex assign. (a = x.y = 0). These are
539           //    called TWIN references, because they show up twice in the
540           //    reference list. Only collapse the set, not the alias.
541           if (!NodeUtil.isObjectLitKey(r.node) &&
542               (r.getTwin() == null || r.isSet())) {
543             flattenNameRef(alias, r.node, rParent, originalName);
544           }
545         }
546     
547         // Flatten all occurrences of a name as a prefix of its subnames. For
548         // example, if {@code n} corresponds to the name "a.b", then "a.b" will be
549         // replaced with "a$b" in all occurrences of "a.b.c", "a.b.c.d", etc.
550         if (n.props != null) {
551           for (Name p : n.props) {
552             flattenPrefixes(alias, p, 1);
553           }
554         }
555       }

557       /**
558        * Flattens all occurrences of a name as a prefix of subnames beginning
559        * with a particular subname.
560        *
561        * @param n A global property name (e.g. "a.b.c.d")
562        * @param alias A flattened prefix name (e.g. "a$b")
563        * @param depth The difference in depth between the property name and
564        *    the prefix name (e.g. 2)
565        */
566       private void flattenPrefixes(String alias, Name n, int depth) {
567         // Only flatten the prefix of a name declaration if the name being
568         // initialized is fully qualified (i.e. not an object literal key).
569         String originalName = n.getFullName();
570         Ref decl = n.getDeclaration();
571         if (decl != null && decl.node != null &&
572             decl.node.isGetProp()) {
573           flattenNameRefAtDepth(alias, decl.node, depth, originalName);
574         }
575     
576         for (Ref r : n.getRefs()) {
577           if (r == decl) {
578             // Declarations are handled separately.
579             continue;
580           }
581     
582           // References inside a complex assign (a = x.y = 0)
583           // have twins. We should only flatten one of the twins.
584           if (r.getTwin() == null || r.isSet()) {
585             flattenNameRefAtDepth(alias, r.node, depth, originalName);
586           }
587         }
588     
589         if (n.props != null) {
590           for (Name p : n.props) {
591             flattenPrefixes(alias, p, depth + 1);
592           }
593         }
594       }

596       /**
597        * Flattens a particular prefix of a single name reference.
598        *
599        * @param alias A flattened prefix name (e.g. "a$b")
600        * @param n The node corresponding to a subproperty name (e.g. "a.b.c.d")
601        * @param depth The difference in depth between the property name and
602        *    the prefix name (e.g. 2)
603        * @param originalName String version of the property name.
604        */
605       private void flattenNameRefAtDepth(String alias, Node n, int depth,
606           String originalName) {
607         // This method has to work for both GETPROP chains and, in rare cases,
608         // OBJLIT keys, possibly nested. That's why we check for children before
609         // proceeding. In the OBJLIT case, we don't need to do anything.
610         Token nType = n.getType();
611         boolean isQName = nType == Token.NAME || nType == Token.GETPROP;
612         boolean isObjKey = NodeUtil.isObjectLitKey(n);
613         Preconditions.checkState(isObjKey || isQName);
614         if (isQName) {
615           for (int i = 1; i < depth && n.hasChildren(); i++) {
616             n = n.getFirstChild();
617           }
618           if (n.isGetProp() && n.getFirstChild().isGetProp()) {
619             flattenNameRef(alias, n.getFirstChild(), n, originalName);
620           }
621         }
622       }

624       /**
625        * Replaces a GETPROP a.b.c with a NAME a$b$c.
626        *
627        * @param alias A flattened prefix name (e.g. "a$b")
628        * @param n The GETPROP node corresponding to the original name (e.g. "a.b")
629        * @param parent {@code n}'s parent
630        * @param originalName String version of the property name.
631        */
632       private void flattenNameRef(String alias, Node n, Node parent,
633           String originalName) {
634         Preconditions.checkArgument(
635             n.isGetProp(), "Expected GETPROP, found %s. Node: %s", n.getType(), n);
636     
637         // BEFORE:
638         //   getprop
639         //     getprop
640         //       name a
641         //       string b
642         //     string c
643         // AFTER:
644         //   name a$b$c
645         Node ref = NodeUtil.newName(compiler, alias, n, originalName);
646         NodeUtil.copyNameAnnotations(n.getLastChild(), ref);
647         if (parent.isCall() && n == parent.getFirstChild()) {
648           // The node was a call target, we are deliberately flatten these as
649           // we node the "this" isn't provided by the namespace. Mark it as such:
650           parent.putBooleanProp(Node.FREE_CALL, true);
651         }
652     
653         TypeI type = n.getTypeI();
654         if (type != null) {
655           ref.setTypeI(type);
656         }
657     
658         parent.replaceChild(n, ref);
659         compiler.reportCodeChange();
660       }

662       /**
663        * Collapses definitions of the collapsible properties of a global name.
664        * Recurs on subnames that also represent JavaScript objects with
665        * collapsible properties.
666        *
667        * @param n A node representing a global name
668        * @param alias The flattened name for {@code n}
669        */
670       private void collapseDeclarationOfNameAndDescendants(Name n, String alias) {
671         boolean canCollapseChildNames = n.canCollapseUnannotatedChildNames();
672     
673         // Handle this name first so that nested object literals get unrolled.
674         if (n.canCollapse()) {
675           updateObjLitOrFunctionDeclaration(n, alias, canCollapseChildNames);
676         }
677     
678         if (n.props == null) {
679           return;
680         }
681         for (Name p : n.props) {
682           // Recur first so that saved node ancestries are intact when needed.
683           collapseDeclarationOfNameAndDescendants(
684               p, appendPropForAlias(alias, p.getBaseName()));
685           if (!p.inExterns && canCollapseChildNames &&
686               p.getDeclaration() != null &&
687               p.canCollapse() &&
688               p.getDeclaration().node != null &&
689               p.getDeclaration().node.getParent() != null &&
690               p.getDeclaration().node.getParent().isAssign()) {
691             updateSimpleDeclaration(
692                 appendPropForAlias(alias, p.getBaseName()), p, p.getDeclaration());
693           }
694         }
695       }

697       /**
698        * Updates the initial assignment to a collapsible property at global scope
699        * by changing it to a variable declaration (e.g. a.b = 1 -> var a$b = 1).
700        * The property's value may either be a primitive or an object literal or
701        * function whose properties aren't collapsible.
702        *
703        * @param alias The flattened property name (e.g. "a$b")
704        * @param refName The name for the reference being updated.
705        * @param ref An object containing information about the assignment getting
706        *     updated
707        */
708       private void updateSimpleDeclaration(String alias, Name refName, Ref ref) {
709         Node rvalue = ref.node.getNext();
710         Node parent = ref.node.getParent();
711         Node grandparent = parent.getParent();
712         Node greatGrandparent = grandparent.getParent();
713     
714         if (rvalue != null && rvalue.isFunction()) {
715           checkForHosedThisReferences(rvalue, refName.docInfo, refName);
716         }
717     
718         // Create the new alias node.
719         Node nameNode = NodeUtil.newName(compiler, alias, grandparent.getFirstChild(),
720             refName.getFullName());
721         NodeUtil.copyNameAnnotations(ref.node.getLastChild(), nameNode);
722     
723         if (grandparent.isExprResult()) {
724           // BEFORE: a.b.c = ...;
725           //   exprstmt
726           //     assign
727           //       getprop
728           //         getprop
729           //           name a
730           //           string b
731           //         string c
732           //       NODE
733           // AFTER: var a$b$c = ...;
734           //   var
735           //     name a$b$c
736           //       NODE
737     
738           // Remove the r-value (NODE).
739           parent.removeChild(rvalue);
740           nameNode.addChildToFront(rvalue);
741     
742           Node varNode = IR.var(nameNode);
743           greatGrandparent.replaceChild(grandparent, varNode);
744         } else {
745           // This must be a complex assignment.
746           Preconditions.checkNotNull(ref.getTwin());
747     
748           // BEFORE:
749           // ... (x.y = 3);
750           //
751           // AFTER:
752           // var x$y;
753           // ... (x$y = 3);
754     
755           Node current = grandparent;
756           Node currentParent = grandparent.getParent();
757           for (; !currentParent.isScript() &&
758                  !currentParent.isBlock();
759                current = currentParent,
760                currentParent = currentParent.getParent()) {}
761     
762           // Create a stub variable declaration right
763           // before the current statement.
764           Node stubVar = IR.var(nameNode.cloneTree())
765               .useSourceInfoIfMissingFrom(nameNode);
766           currentParent.addChildBefore(stubVar, current);
767     
768           parent.replaceChild(ref.node, nameNode);
769         }
770     
771         compiler.reportCodeChange();
772       }

774       /**
775        * Updates the first initialization (a.k.a "declaration") of a global name.
776        * This involves flattening the global name (if it's not just a global
777        * variable name already), collapsing object literal keys into global
778        * variables, declaring stub global variables for properties added later
779        * in a local scope.
780        *
781        * It may seem odd that this function also takes care of declaring stubs
782        * for direct children. The ultimate goal of this function is to eliminate
783        * the global name entirely (when possible), so that "middlemen" namespaces
784        * disappear, and to do that we need to make sure that all the direct children
785        * will be collapsed as well.
786        *
787        * @param n An object representing a global name (e.g. "a", "a.b.c")
788        * @param alias The flattened name for {@code n} (e.g. "a", "a$b$c")
789        * @param canCollapseChildNames Whether it's possible to collapse children of
790        *     this name. (This is mostly passed for convenience; it's equivalent to
791        *     n.canCollapseChildNames()).
792        */
793       private void updateObjLitOrFunctionDeclaration(
794           Name n, String alias, boolean canCollapseChildNames) {
795         Ref decl = n.getDeclaration();
796         if (decl == null) {
797           // Some names do not have declarations, because they
798           // are only defined in local scopes.
799           return;
800         }
801     
802         if (decl.getTwin() != null) {
803           // Twin declarations will get handled when normal references
804           // are handled.
805           return;
806         }
807     
808         switch (decl.node.getParent().getType()) {
809           case ASSIGN:
810             updateObjLitOrFunctionDeclarationAtAssignNode(
811                 n, alias, canCollapseChildNames);
812             break;
813           case VAR:
814             updateObjLitOrFunctionDeclarationAtVarNode(n, canCollapseChildNames);
815             break;
816           case FUNCTION:
817             updateFunctionDeclarationAtFunctionNode(n, canCollapseChildNames);
818             break;
819         }
820       }

822       /**
823        * Updates the first initialization (a.k.a "declaration") of a global name
824        * that occurs at an ASSIGN node. See comment for
825        * {@link #updateObjLitOrFunctionDeclaration}.
826        *
827        * @param n An object representing a global name (e.g. "a", "a.b.c")
828        * @param alias The flattened name for {@code n} (e.g. "a", "a$b$c")
829        */
830       private void updateObjLitOrFunctionDeclarationAtAssignNode(
831           Name n, String alias, boolean canCollapseChildNames) {
832         // NOTE: It's important that we don't add additional nodes
833         // (e.g. a var node before the exprstmt) because the exprstmt might be
834         // the child of an if statement that's not inside a block).
835     
836         Ref ref = n.getDeclaration();
837         Node rvalue = ref.node.getNext();
838         Node varNode = new Node(Token.VAR);
839         Node varParent = ref.node.getAncestor(3);
840         Node grandparent = ref.node.getAncestor(2);
841         boolean isObjLit = rvalue.isObjectLit();
842         boolean insertedVarNode = false;
843     
844         if (isObjLit && n.canEliminate()) {
845           // Eliminate the object literal altogether.
846           varParent.replaceChild(grandparent, varNode);
847           ref.node = null;
848           insertedVarNode = true;
849     
850         } else if (!n.isSimpleName()) {
851           // Create a VAR node to declare the name.
852           if (rvalue.isFunction()) {
853             checkForHosedThisReferences(rvalue, n.docInfo, n);
854           }
855     
856           ref.node.getParent().removeChild(rvalue);
857     
858           Node nameNode = NodeUtil.newName(compiler,
859               alias, ref.node.getAncestor(2), n.getFullName());
860     
861           JSDocInfo info = NodeUtil.getBestJSDocInfo(ref.node.getParent());
862           if (ref.node.getLastChild().getBooleanProp(Node.IS_CONSTANT_NAME) ||
863               (info != null && info.isConstant())) {
864             nameNode.putBooleanProp(Node.IS_CONSTANT_NAME, true);
865           }
866     
867           if (info != null) {
868             varNode.setJSDocInfo(info);
869           }
870           varNode.addChildToBack(nameNode);
871           nameNode.addChildToFront(rvalue);
872           varParent.replaceChild(grandparent, varNode);
873     
874           // Update the node ancestry stored in the reference.
875           ref.node = nameNode;
876           insertedVarNode = true;
877         }
878     
879         if (canCollapseChildNames) {
880           if (isObjLit) {
881             declareVarsForObjLitValues(
882                 n, alias, rvalue,
883                 varNode, varParent.getChildBefore(varNode), varParent);
884           }
885     
886           addStubsForUndeclaredProperties(n, alias, varParent, varNode);
887         }
888     
889         if (insertedVarNode) {
890           if (!varNode.hasChildren()) {
891             varParent.removeChild(varNode);
892           }
893           compiler.reportCodeChange();
894         }
895       }

897       /**
898        * Warns about any references to "this" in the given FUNCTION. The function
899        * is getting collapsed, so the references will change.
900        */
901       private void checkForHosedThisReferences(Node function, JSDocInfo docInfo,
902           final Name name) {
903         // A function is getting collapsed. Make sure that if it refers to
904         // "this", it must be a constructor or documented with @this.
905         if (docInfo == null ||
906             (!docInfo.isConstructor() && !docInfo.hasThisType())) {
907           NodeTraversal.traverseEs6(compiler, function.getLastChild(),
908               new NodeTraversal.AbstractShallowCallback() {
909                 @Override
910                 public void visit(NodeTraversal t, Node n, Node parent) {
911                   if (n.isThis()) {
912                     compiler.report(
913                         JSError.make(n, UNSAFE_THIS, name.getFullName()));
914                   }
915                 }
916               });
917         }
918       }

920       /**
921        * Updates the first initialization (a.k.a "declaration") of a global name
922        * that occurs at a VAR node. See comment for
923        * {@link #updateObjLitOrFunctionDeclaration}.
924        *
925        * @param n An object representing a global name (e.g. "a")
926        */
927       private void updateObjLitOrFunctionDeclarationAtVarNode(
928           Name n, boolean canCollapseChildNames) {
929         if (!canCollapseChildNames) {
930           return;
931         }
932     
933         Ref ref = n.getDeclaration();
934         String name = ref.node.getString();
935         Node rvalue = ref.node.getFirstChild();
936         Node varNode = ref.node.getParent();
937         Node grandparent = varNode.getParent();
938     
939         boolean isObjLit = rvalue.isObjectLit();
940         int numChanges = 0;
941     
942         if (isObjLit) {
943           numChanges += declareVarsForObjLitValues(
944               n, name, rvalue, varNode, grandparent.getChildBefore(varNode),
945               grandparent);
946         }
947     
948         numChanges += addStubsForUndeclaredProperties(n, name, grandparent, varNode);
949     
950         if (isObjLit && n.canEliminate()) {
951           varNode.removeChild(ref.node);
952           if (!varNode.hasChildren()) {
953             grandparent.removeChild(varNode);
954           }
955           numChanges++;
956     
957           // Clear out the object reference, since we've eliminated it from the
958           // parse tree.
959           ref.node = null;
960         }
961     
962         if (numChanges > 0) {
963           compiler.reportCodeChange();
964         }
965       }

967       /**
968        * Updates the first initialization (a.k.a "declaration") of a global name
969        * that occurs at a FUNCTION node. See comment for
970        * {@link #updateObjLitOrFunctionDeclaration}.
971        *
972        * @param n An object representing a global name (e.g. "a")
973        */
974       private void updateFunctionDeclarationAtFunctionNode(
975           Name n, boolean canCollapseChildNames) {
976         if (!canCollapseChildNames || !n.canCollapse()) {
977           return;
978         }
979     
980         Ref ref = n.getDeclaration();
981         String fnName = ref.node.getString();
982         addStubsForUndeclaredProperties(
983             n, fnName, ref.node.getAncestor(2), ref.node.getParent());
984       }

986       /**
987        * Declares global variables to serve as aliases for the values in an object
988        * literal, optionally removing all of the object literal's keys and values.
989        *
990        * @param alias The object literal's flattened name (e.g. "a$b$c")
991        * @param objlit The OBJLIT node
992        * @param varNode The VAR node to which new global variables should be added
993        *     as children
994        * @param nameToAddAfter The child of {@code varNode} after which new
995        *     variables should be added (may be null)
996        * @param varParent {@code varNode}'s parent
997        * @return The number of variables added
998        */
999       private int declareVarsForObjLitValues(
1000           Name objlitName, String alias, Node objlit, Node varNode,
1001           Node nameToAddAfter, Node varParent) {
1002         int numVars = 0;
1003         int arbitraryNameCounter = 0;
1004         boolean discardKeys = !objlitName.shouldKeepKeys();
1005     
1006         for (Node key = objlit.getFirstChild(), nextKey; key != null;
1007              key = nextKey) {
1008           Node value = key.getFirstChild();
1009           nextKey = key.getNext();
1010     
1011           // A get or a set can not be rewritten as a VAR.
1012           if (key.isGetterDef() || key.isSetterDef()) {
1013             continue;
1014           }
1015     
1016           // We generate arbitrary names for keys that aren't valid JavaScript
1017           // identifiers, since those keys are never referenced. (If they were,
1018           // this object literal's child names wouldn't be collapsible.) The only
1019           // reason that we don't eliminate them entirely is the off chance that
1020           // their values are expressions that have side effects.
1021           boolean isJsIdentifier = !key.isNumber() &&
1022                                    TokenStream.isJSIdentifier(key.getString());
1023           String propName = isJsIdentifier ?
1024               key.getString() : String.valueOf(++arbitraryNameCounter);
1025     
1026           // If the name cannot be collapsed, skip it.
1027           String qName = objlitName.getFullName() + '.' + propName;
1028           Name p = nameMap.get(qName);
1029           if (p != null && !p.canCollapse()) {
1030             continue;
1031           }
1032     
1033           String propAlias = appendPropForAlias(alias, propName);
1034           Node refNode = null;
1035           if (discardKeys) {
1036             objlit.removeChild(key);
1037             value.detachFromParent();
1038           } else {
1039             // Substitute a reference for the value.
1040             refNode = IR.name(propAlias);
1041             if (key.getBooleanProp(Node.IS_CONSTANT_NAME)) {
1042               refNode.putBooleanProp(Node.IS_CONSTANT_NAME, true);
1043             }
1044     
1045             key.replaceChild(value, refNode);
1046           }
1047     
1048           // Declare the collapsed name as a variable with the original value.
1049           Node nameNode = IR.name(propAlias);
1050           nameNode.addChildToFront(value);
1051           if (key.getBooleanProp(Node.IS_CONSTANT_NAME)) {
1052             nameNode.putBooleanProp(Node.IS_CONSTANT_NAME, true);
1053           }
1054           Node newVar = IR.var(nameNode).useSourceInfoIfMissingFromForTree(key);
1055           if (nameToAddAfter != null) {
1056             varParent.addChildAfter(newVar, nameToAddAfter);
1057           } else {
1058             varParent.addChildBefore(newVar, varNode);
1059           }
1060           compiler.reportCodeChange();
1061           nameToAddAfter = newVar;
1062     
1063           // Update the global name's node ancestry if it hasn't already been
1064           // done. (Duplicate keys in an object literal can bring us here twice
1065           // for the same global name.)
1066           if (isJsIdentifier && p != null) {
1067             if (!discardKeys) {
1068               Ref newAlias =
1069                   p.getDeclaration().cloneAndReclassify(Ref.Type.ALIASING_GET);
1070               newAlias.node = refNode;
1071               p.addRef(newAlias);
1072             }
1073     
1074             p.getDeclaration().node = nameNode;
1075     
1076             if (value.isFunction()) {
1077               checkForHosedThisReferences(value, key.getJSDocInfo(), p);
1078             }
1079           }
1080     
1081           numVars++;
1082         }
1083         return numVars;
1084       }

1086       /**
1087        * Adds global variable "stubs" for any properties of a global name that are
1088        * only set in a local scope or read but never set.
1089        *
1090        * @param n An object representing a global name (e.g. "a", "a.b.c")
1091        * @param alias The flattened name of the object whose properties we are
1092        *     adding stubs for (e.g. "a$b$c")
1093        * @param parent The node to which new global variables should be added
1094        *     as children
1095        * @param addAfter The child of after which new
1096        *     variables should be added
1097        * @return The number of variables added
1098        */
1099       private int addStubsForUndeclaredProperties(
1100           Name n, String alias, Node parent, Node addAfter) {
1101         Preconditions.checkState(n.canCollapseUnannotatedChildNames(), n);
1102         Preconditions.checkArgument(NodeUtil.isStatementBlock(parent), parent);
1103         Preconditions.checkNotNull(addAfter);
1104         if (n.props == null) {
1105           return 0;
1106         }
1107         int numStubs = 0;
1108         for (Name p : n.props) {
1109           if (p.needsToBeStubbed()) {
1110             String propAlias = appendPropForAlias(alias, p.getBaseName());
1111             Node nameNode = IR.name(propAlias);
1112             Node newVar = IR.var(nameNode).useSourceInfoIfMissingFromForTree(addAfter);
1113             parent.addChildAfter(newVar, addAfter);
1114             addAfter = newVar;
1115             numStubs++;
1116             compiler.reportCodeChange();
1117             // Determine if this is a constant var by checking the first
1118             // reference to it. Don't check the declaration, as it might be null.
1119             if (p.getRefs().get(0).node.getLastChild().getBooleanProp(
1120                 Node.IS_CONSTANT_NAME)) {
1121               nameNode.putBooleanProp(Node.IS_CONSTANT_NAME, true);
1122             }
1123           }
1124         }
1125         return numStubs;
1126       }

1128       private String appendPropForAlias(String root, String prop) {
1129         if (prop.indexOf('$') != -1) {
1130           // Encode '$' in a property as '$0'. Because '0' cannot be the
1131           // start of an identifier, this will never conflict with our
1132           // encoding from '.' -> '$'.
1133           prop = prop.replace("$", "$0");
1134         }
1135         String result = root + '$' + prop;
1136         int id = 1;
1137         while (nameMap.containsKey(result)) {
1138           result = root + '$' + prop + '$' + id;
1139           id++;
1140         }
1141         return result;
1142       }
