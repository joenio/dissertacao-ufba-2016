75       private void testNotEs6Typed(String source, String... features) {
76         for (int i = 0; i < features.length; i++) {
77           features[i] =
78               "type syntax is only supported in ES6 typed mode: "
79                   + features[i]
80                   + ". Use --language_in=ECMASCRIPT6_TYPED to enable ES6 typed features.";

82         expectErrors(features);
83         parse(source, LanguageMode.ECMASCRIPT6);
84         expectErrors(features);
85         parse(source, LanguageMode.ECMASCRIPT6_STRICT);

310       public void testFunctionType() {
311         parse("var n: (p1) => boolean;");
312         parse("var n: (p1, p2) => boolean;");
313         parse("var n: (p1: string) => boolean;");
314         parse("var n: (p1: string, p2: number) => boolean;");
315         parse("var n: () => () => number;");
316         parse("var n: (p1: string) => {};");
317         // parse("(number): () => number => number;");
318     
319         Node ast = parse("var n: (p1: string, p2: number) => boolean[];");
320         TypeDeclarationNode function = (TypeDeclarationNode)
321             (ast.getFirstFirstChild().getProp(Node.DECLARED_TYPE_EXPR));
322         assertNode(function).hasType(Token.FUNCTION_TYPE);
323     
324         Node ast2 = parse("var n: (p1: string, p2: number) => boolean | number;");
325         TypeDeclarationNode function2 = (TypeDeclarationNode)
326             (ast2.getFirstFirstChild().getProp(Node.DECLARED_TYPE_EXPR));
327         assertNode(function2).hasType(Token.FUNCTION_TYPE);
328     
329         Node ast3 = parse("var n: (p1: string, p2: number) => Array<Foo>;");
330         TypeDeclarationNode function3 = (TypeDeclarationNode)
331             (ast3.getFirstFirstChild().getProp(Node.DECLARED_TYPE_EXPR));

786       private Node parse(String source, String expected, LanguageMode languageIn) {
787         CompilerOptions options = new CompilerOptions();
788         options.setLanguageIn(languageIn);
789         options.setLanguageOut(LanguageMode.ECMASCRIPT6_TYPED);
790         options.setPreserveTypeAnnotations(true);
791         options.setPrettyPrint(true);
792         options.setLineLengthThreshold(80);
793         options.setPreferSingleQuotes(true);
794     
795         Compiler compiler = new Compiler();
796         compiler.setErrorManager(testErrorManager);
797         compiler.initOptions(options);
798     
799         Node script = compiler.parse(SourceFile.fromCode("[test]", source));
800     
801         // Verifying that all warnings were seen
802         assertTrue("Missing an error", testErrorManager.hasEncounteredAllErrors());
803         assertTrue("Missing a warning", testErrorManager.hasEncounteredAllWarnings());
804     
805         if (script != null && testErrorManager.getErrorCount() == 0) {
806           // if it can be parsed, it should round trip.
807           String actual = new CodePrinter.Builder(script)
808               .setCompilerOptions(options)
809               .setTypeRegistry(compiler.getTypeIRegistry())
810               .build() // does the actual printing.
811               .trim();
812           assertThat(actual).isEqualTo(expected);
813         }
814     
815         return script;
816       }
