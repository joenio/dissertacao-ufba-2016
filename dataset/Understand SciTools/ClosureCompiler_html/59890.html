17     package com.google.javascript.jscomp;
18     
19     import static com.google.common.base.MoreObjects.toStringHelper;
20     
21     import java.io.Serializable;
22     import java.util.Collection;
23     import java.util.HashSet;
24     import java.util.Set;
25     
26     /**
27      * Options for how to manage dependencies between input files.
28      *
29      * Dependency information is usually pulled out from the JS code by
30      * looking for primitive dependency functions (like Closure Library's
31      * goog.provide/goog.require). Analysis of this dependency information is
32      * controlled by {@code CodingConvention}, which lets you define those
33      * dependency primitives.
34      *
35      * This options class determines how we use that dependency information
36      * to change how code is built.
37      *
38      * @author nicksantos@google.com (Nick Santos)
39      */
40     public final class DependencyOptions implements Serializable {
41       private static final long serialVersionUID = 1L;
42     
43       private boolean sortDependencies = false;
44       private boolean pruneDependencies = false;
45       private boolean dropMoochers = false;
46       private final Set<ModuleIdentifier> entryPoints = new HashSet<>();
47     
48       /**
49        * Enables or disables dependency sorting mode.
50        *
51        * If true, we will sort the input files based on dependency information
52        * in them. Otherwise, we will use the order of files specified
53        * on the command-line.
54        * @return this for easy building.
55        */
56       public DependencyOptions setDependencySorting(boolean enabled) {
57         this.sortDependencies = enabled;
58         return this;
59       }
60     
61       /**
62        * Enables or disables dependency pruning mode.
63        *
64        * In dependency pruning mode, we will look for all files that provide a
65        * symbol. Unless that file is a transitive dependency of a file that
66        * we're using, we will remove it from the compilation job.
67        *
68        * This does not affect how we handle files that do not provide symbols.
69        * See setMoocherDropping for information on how these are handled.
70        *
71        * @return this for easy chaining.
72        */
73       public DependencyOptions setDependencyPruning(boolean enabled) {
74         this.pruneDependencies = enabled;
75         return this;
76       }
77     
78       /**
79        * Enables or disables moocher dropping mode.
80        *
81        * A 'moocher' is a file that does not provide any symbols (though they
82        * may require symbols). This is usually because they don't want to
83        * tie themselves to a particular dependency system (e.g., Closure's
84        * goog.provide, CommonJS modules). So they rely on other people to
85        * manage dependencies on them.
86        *
87        * If true, we drop these files when we prune dependencies.
88        * If false, we always keep these files an anything they depend on.
89        * The default is false.
90        *
91        * Notice that this option only makes sense if dependency pruning is on,
92        * and a set of entry points is specified.
93        *
94        * @return this for easy chaining.
95        */
96       public DependencyOptions setMoocherDropping(boolean enabled) {
97         this.dropMoochers = enabled;
98         return this;
99       }
100     
101       /**
102        * Adds a collection of symbols to always keep.
103        *
104        * In dependency pruning mode, we will automatically keep all the
105        * transitive dependencies of these symbols.
106        *
107        * The syntactic form of a symbol depends on the type of dependency
108        * primitives we're using. For example, goog.provide('foo.bar')
109        * provides the symbol 'foo.bar'.
110        *
111        * Entry points can be scoped to a module by specifying 'mod2:foo.bar'.
112        *
113        * @return this for easy chaining.
114        */
115       public DependencyOptions setEntryPoints(Collection<ModuleIdentifier> symbols) {
116         entryPoints.clear();
117         entryPoints.addAll(symbols);
118         return this;
119       }
120     
121       /** Returns whether re-ordering of files is needed. */
122       boolean needsManagement() {
123         return sortDependencies || pruneDependencies;
124       }
125     
126       boolean shouldSortDependencies() {
127         return sortDependencies;
128       }
129     
130       boolean shouldPruneDependencies() {
131         return pruneDependencies;
132       }
133     
134       boolean shouldDropMoochers() {
135         return pruneDependencies && dropMoochers;
136       }
137     
138       Collection<ModuleIdentifier> getEntryPoints() {
139         return entryPoints;
140       }
141     
142       @Override
143       public String toString() {
144         return toStringHelper(this)
145             .add("sortDependencies", sortDependencies)
146             .add("pruneDependencies", pruneDependencies)
147             .add("dropMoochers", dropMoochers)
148             .add("entryPoints", entryPoints)
149             .toString();
150       }
151     }
