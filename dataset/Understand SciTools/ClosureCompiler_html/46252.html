17     package com.google.javascript.jscomp;
18     
19     import com.google.common.annotations.GwtIncompatible;
20     import com.google.common.base.Preconditions;
21     import com.google.javascript.jscomp.CoverageInstrumentationPass.CoverageReach;
22     import com.google.javascript.rhino.IR;
23     import com.google.javascript.rhino.Node;
24     
25     import java.util.Map;
26     
27     /**
28      * This class implements a traversal to instrument an AST for code coverage.
29      * @author praveenk@google.com (Praveen Kumashi)
30      *
31      */
32     @GwtIncompatible("FileInstrumentationData")
33     class CoverageInstrumentationCallback extends
34         NodeTraversal.AbstractPostOrderCallback {
35     
36       private final AbstractCompiler compiler;
37       private final Map<String, FileInstrumentationData> instrumentationData;
38       private final CoverageReach reach;
39     
40       static final String ARRAY_NAME_PREFIX = "JSCompiler_lcov_data_";
41     
42       public CoverageInstrumentationCallback(
43           AbstractCompiler compiler,
44           Map<String, FileInstrumentationData> instrumentationData,
45           CoverageReach reach) {
46         this.compiler = compiler;
47         this.instrumentationData = instrumentationData;
48         this.reach = reach;
49       }
50     
51       /**
52        * Returns the name of the source file from which the given node originates.
53        * @param traversal the traversal
54        * @return the name of the file it originates from
55        */
56       private static String getFileName(NodeTraversal traversal) {
57         return traversal.getSourceName();
58       }
59     
60       /**
61        * Returns a string that can be used as array name. The name is based on the
62        * source filename of the AST node.
63        */
64       private String createArrayName(NodeTraversal traversal) {
65         return ARRAY_NAME_PREFIX
66             + CoverageUtil.createIdentifierFromText(getFileName(traversal));
67       }
68     
69       /**
70        * Creates and return a new instrumentation node. The instrumentation Node is
71        * of the form: "arrayName[lineNumber] = true;"
72        * Note 1: Node returns a 1-based line number.
73        * Note 2: Line numbers in instrumentation are made 0-based. This is done to
74        * map to their bit representation in BitField. Thus, there's a one-to-one
75        * correspondence of the line number seen on instrumented files and their bit
76        * encodings.
77        *
78        * @return an instrumentation node corresponding to the line number
79        */
80       private Node newInstrumentationNode(NodeTraversal traversal, Node node) {
81         int lineNumber = node.getLineno();
82         String arrayName = createArrayName(traversal);
83     
84         // Create instrumentation Node
85         Node getElemNode = IR.getelem(
86             IR.name(arrayName),
87             IR.number(lineNumber - 1));  // Make line number 0-based
88         Node exprNode = IR.exprResult(IR.assign(getElemNode, IR.trueNode()));
89     
90         // Note line as instrumented
91         String fileName = getFileName(traversal);
92         if (!instrumentationData.containsKey(fileName)) {
93           instrumentationData.put(fileName,
94                                   new FileInstrumentationData(fileName, arrayName));
95         }
96         instrumentationData.get(fileName).setLineAsInstrumented(lineNumber);
97     
98         return exprNode.useSourceInfoIfMissingFromForTree(node);
99       }
100     
101       /**
102        * Create and return a new array declaration node. The array name is
103        * generated based on the source filename, and declaration is of the form:
104        * "var arrayNameUsedInFile = [];"
105        */
106       private Node newArrayDeclarationNode(NodeTraversal traversal) {
107         return IR.var(
108             IR.name(createArrayName(traversal)),
109             IR.arraylit());
110       }
111     
112       /**
113        * @return a Node containing file specific setup logic.
114        */
115       private Node newHeaderNode(NodeTraversal traversal, Node srcref) {
116         String fileName = getFileName(traversal);
117         String arrayName = createArrayName(traversal);
118         FileInstrumentationData data = instrumentationData.get(fileName);
119         Preconditions.checkNotNull(data);
120     
121         String objName = CoverageInstrumentationPass.JS_INSTRUMENTATION_OBJECT_NAME;
122     
123         // var JSCompiler_lcov_data_xx = [];
124         // __jscov.executedLines.push(JSCompiler_lcov_data_xx);
125         // __jscov.instrumentedLines.push(hex-data);
126         // __jscov.fileNames.push(filename);
127         return IR.block(
128                 newArrayDeclarationNode(traversal),
129                 IR.exprResult(
130                     IR.call(
131                         NodeUtil.newQName(compiler, objName + ".executedLines.push"),
132                         IR.name(arrayName))),
133                 IR.exprResult(
134                     IR.call(
135                         NodeUtil.newQName(compiler, objName + ".instrumentedLines.push"),
136                         IR.string(data.getInstrumentedLinesAsHexString()))),
137                 IR.exprResult(
138                     IR.call(
139                         NodeUtil.newQName(compiler, objName + ".fileNames.push"), IR.string(fileName))))
140             .useSourceInfoIfMissingFromForTree(srcref);
141       }
142     
143       /**
144        * Instruments the JS code by inserting appropriate nodes into the AST. The
145        * instrumentation logic is tuned to collect "line coverage" data only.
146        */
147       @Override
148       public void visit(NodeTraversal traversal, Node node, Node parent) {
149         // SCRIPT node is special - it is the root of the AST for code from a file.
150         // Append code to declare and initialize structures used in instrumentation.
151         if (node.isScript()) {
152           String fileName = getFileName(traversal);
153           if (instrumentationData.get(fileName) != null) {
154             node.addChildToFront(newHeaderNode(traversal, node));
155           }
156           compiler.reportCodeChange();
157           return;
158         }
159     
160         // Don't instrument global statements
161         if (reach == CoverageReach.CONDITIONAL
162             && parent != null && parent.isScript()) {
163           return;
164         }
165     
166         // For arrow functions whose body is an expression instead of a block,
167         // convert it to a block so that it can be instrumented.
168         if (node.isFunction() && !NodeUtil.getFunctionBody(node).isBlock()) {
169           Node returnValue = NodeUtil.getFunctionBody(node);
170           Node body = IR.block(IR.returnNode(returnValue.detachFromParent()));
171           body.useSourceInfoIfMissingFromForTree(returnValue);
172           node.addChildToBack(body);
173         }
174     
175         // Add instrumentation code just before a function block.
176         // Similarly before other constructs: 'with', 'case', 'default', 'catch'
177         if (node.isFunction()
178             || node.isWith()
179             || node.isCase()
180             || node.isDefaultCase()
181             || node.isCatch()) {
182           Node codeBlock = node.getLastChild();
183           codeBlock.addChildToFront(
184               newInstrumentationNode(traversal, node));
185           compiler.reportCodeChange();
186           return;
187         }
188     
189         // Add instrumentation code as the first child of a 'try' block.
190         if (node.isTry()) {
191           Node firstChild = node.getFirstChild();
192           firstChild.addChildToFront(
193               newInstrumentationNode(traversal, node));
194           compiler.reportCodeChange();
195           return;
196         }
197     
198         // For any other statement, add instrumentation code just before it.
199         if (parent != null && NodeUtil.isStatementBlock(parent)) {
200           parent.addChildBefore(
201               newInstrumentationNode(traversal, node),
202               node);
203           compiler.reportCodeChange();
204           return;
205         }
206       }
207     }
