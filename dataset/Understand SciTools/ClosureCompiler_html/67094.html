1072       private void assertSymmetricOrdering(
1073           Ordering<Symbol> ordering, Symbol first, Symbol second) {
1074         assertEquals(0, ordering.compare(first, first));
1075         assertEquals(0, ordering.compare(second, second));
1076         assertTrue(ordering.compare(first, second) < 0);
1077         assertTrue(ordering.compare(second, first) > 0);
1078       }

1087             return sym;

1094         for (SymbolScope scope : table.getAllScopes()) {
1095           if (!scope.isGlobalScope() && scope.isLexicalScope() &&
1096               scope.getQualifiedSlot(name) != null) {
1097             return scope.getQualifiedSlot(name);
1098           }
1099         }
1100         return null;
1101       }

1106         for (Symbol symbol : table.getAllSymbols()) {
1107           if (symbol.getDeclaration() != null &&
1108               !symbol.getDeclaration().getNode().isFromExterns()) {

1112         return result;

1115       private SymbolTable createSymbolTable(String input) {
1116         List<SourceFile> inputs = ImmutableList.of(
1117             SourceFile.fromCode("in1", input));
1118         List<SourceFile> externs = ImmutableList.of(
1119             SourceFile.fromCode("externs1", EXTERNS));
1120     
1121         Compiler compiler = new Compiler();
1122         BlackHoleErrorManager.silence(compiler);
1123         compiler.compile(externs, inputs, options);
1124         return assertSymbolTableValid(compiler.buildKnownSymbolTable());
1125       }

1127       /**
1128        * Asserts that the symbol table meets some invariants.
1129        * Returns the same table for easy chaining.
1130        */
1131       private SymbolTable assertSymbolTableValid(SymbolTable table) {
1132         Set<Symbol> allSymbols = new HashSet<>();
1133         Iterables.addAll(allSymbols, table.getAllSymbols());
1134         for (Symbol sym : table.getAllSymbols()) {
1135           // Make sure that grabbing the symbol's scope and looking it up
1136           // again produces the same symbol.
1137           assertEquals(sym, table.getScope(sym).getQualifiedSlot(sym.getName()));
1138     
1139           for (Reference ref : table.getReferences(sym)) {
1140             // Make sure that the symbol and reference are mutually linked.
1141             assertEquals(sym, ref.getSymbol());
1142           }
1143     
1144           Symbol scope = table.getSymbolForScope(table.getScope(sym));
1145           assertTrue(
1146               "The symbol's scope is a zombie scope that shouldn't exist.\n" +
1147               "Symbol: " + sym + "\n" +
1148               "Scope: " + table.getScope(sym),
1149               scope == null || allSymbols.contains(scope));
1150         }
1151     
1152         // Make sure that the global "this" is declared at the first input root.
1153         Symbol global = getGlobalVar(table, SymbolTable.GLOBAL_THIS);
1154         assertNotNull(global);
1155         assertNotNull(global.getDeclaration());
1156         assertEquals(Token.SCRIPT, global.getDeclaration().getNode().getType());
1157     
1158         List<Reference> globalRefs = table.getReferenceList(global);
1159     
1160         // The main reference list should never contain the synthetic declaration
1161         // for the global root.
1162         assertThat(globalRefs).doesNotContain(global.getDeclaration());
1163     
1164         return table;
1165       }
