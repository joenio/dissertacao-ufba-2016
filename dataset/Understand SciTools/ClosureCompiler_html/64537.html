32      * Unit tests for PerformanceTracker.

34      * @author dimvar@google.com (Dimitris Vardoulakis)

36     public final class PerformanceTrackerTest extends TestCase {
37       private Node emptyScript = new Node(Token.SCRIPT);
38     
39       public void testStatsCalculation() {
40         PerformanceTracker tracker =
41             new PerformanceTracker(emptyScript, TracerMode.ALL);
42         CodeChangeHandler handler = tracker.getCodeChangeHandler();
43     
44         // It's sufficient for this test to assume that a single run of any pass
45         // takes some fixed amount of time, say 5ms.
46         int passRuntime = 5;
47     
48         tracker.recordPassStart("noloopA", true);
49         handler.reportChange();
50         tracker.recordPassStop("noloopA", passRuntime);
51     
52         tracker.recordPassStart("noloopB", true);
53         handler.reportChange();
54         tracker.recordPassStop("noloopB", passRuntime);
55     
56         tracker.recordPassStart("loopA", false);
57         handler.reportChange();
58         tracker.recordPassStop("loopA", passRuntime);
59     
60         tracker.recordPassStart("loopA", false);
61         tracker.recordPassStop("loopA", passRuntime);
62     
63         tracker.recordPassStart("noloopB", true);
64         handler.reportChange();
65         tracker.recordPassStop("noloopB", passRuntime);
66     
67         tracker.recordPassStart("loopB", false);
68         tracker.recordPassStop("loopB", passRuntime);
69     
70         tracker.recordPassStart("noloopB", true);
71         tracker.recordPassStop("noloopB", passRuntime);
72     
73         int numRuns = tracker.getRuns();
74     
75         assertEquals(numRuns, 7);
76         assertEquals(tracker.getRuntime(), numRuns * passRuntime);
77         assertEquals(tracker.getLoopRuns(), 3);
78         assertEquals(tracker.getChanges(), 4); /* reportChange was called 4 times */
79         assertEquals(tracker.getLoopChanges(), 1);
80     
81         ImmutableMap<String, Stats> stats = tracker.getStats();
82         Stats st = stats.get("noloopA");
83         assertEquals(st.runs, 1);
84         assertEquals(st.runtime, passRuntime);
85         assertEquals(st.changes, 1);
86     
87         st = stats.get("noloopB");
88         assertEquals(st.runs, 3);
89         assertEquals(st.runtime, 3 * passRuntime);
90         assertEquals(st.changes, 2);
91     
92         st = stats.get("loopA");
93         assertEquals(st.runs, 2);
94         assertEquals(st.runtime, 2 * passRuntime);
95         assertEquals(st.changes, 1);
96     
97         st = stats.get("loopB");
98         assertEquals(st.runs, 1);
99         assertEquals(st.runtime, passRuntime);
100         assertEquals(st.changes, 0);
101       }
102     
103       public void testOutputFormat() {
104         PerformanceTracker tracker =
105             new PerformanceTracker(emptyScript, TracerMode.ALL);
106         ByteArrayOutputStream output = new ByteArrayOutputStream();
107         PrintStream outstream = new PrintStream(output);
108         tracker.outputTracerReport(outstream);
109         outstream.close();
110         Pattern p = Pattern.compile(
111             ".*Summary:\npass,runtime,runs,changingRuns,reduction,gzReduction" +
112             ".*TOTAL:" +
113             "\nRuntime\\(ms\\): [0-9]+" +
114             "\n#Runs: [0-9]+" +
115             "\n#Changing runs: [0-9]+" +
116             "\n#Loopable runs: [0-9]+" +
117             "\n#Changing loopable runs: [0-9]+" +
118             "\nEstimated Reduction\\(bytes\\): [0-9]+" +
119             "\nEstimated GzReduction\\(bytes\\): [0-9]+" +
120             "\nEstimated Size\\(bytes\\): -?[0-9]+" +
121             "\nEstimated GzSize\\(bytes\\): -?[0-9]+" +
122             "\n\nLog:\n" +
123             "pass,runtime,runs,changingRuns,reduction,gzReduction,size,gzSize.*",
124             Pattern.DOTALL);
125         String outputString = output.toString();
126         assertTrue("Unexpected output from PerformanceTracker:\n" + outputString,
127             p.matcher(outputString).matches());
128       }
