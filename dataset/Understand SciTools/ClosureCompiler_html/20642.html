40     package com.google.javascript.rhino.jstype;
41     
42     import com.google.javascript.rhino.Node;
43     import com.google.javascript.rhino.jstype.RecordTypeBuilder.RecordProperty;
44     
45     import java.util.Map;
46     
47     /**
48      * A record (structural) type.
49      *
50      * Subtyping: The subtyping of a record type is defined via structural
51      * comparison of a record type's properties. For example, a record
52      * type of the form { a : TYPE_1 } is a supertype of a record type
53      * of the form { b : TYPE_2, a : TYPE_1 } because B can be assigned to
54      * A and matches all constraints. Similarly, a defined type can be assigned
55      * to a record type so long as that defined type matches all property
56      * constraints of the record type. A record type of the form { a : A, b : B }
57      * can be assigned to a record of type { a : A }.
58      *
59      */
60     public class RecordType extends PrototypeObjectType {
61       private static final long serialVersionUID = 1L;
62     
63       private final boolean declared;
64       private boolean isFrozen = false;
65     
66       RecordType(JSTypeRegistry registry, Map<String, RecordProperty> properties) {
67         this(registry, properties, true);
68       }
69     
70       /**
71        * Creates a record type.
72        *
73        * @param registry The type registry under which this type lives.
74        * @param properties A map of all the properties of this record type.
75        * @param declared Whether this is a declared or synthesized type.
76        *     A synthesized record type is just used for bookkeeping
77        *     in the type system. A declared record type was actually used in the
78        *     user's program.
79        * @throws IllegalStateException if the {@code RecordProperty} associated
80        *         with a property is null.
81        */
82       RecordType(JSTypeRegistry registry, Map<String, RecordProperty> properties,
83           boolean declared) {
84         super(registry, null, null);
85         setPrettyPrint(true);
86         this.declared = declared;
87     
88         for (String property : properties.keySet()) {
89           RecordProperty prop = properties.get(property);
90           if (prop == null) {
91             throw new IllegalStateException(
92                 "RecordProperty associated with a property should not be null!");
93           }
94           if (declared) {
95             defineDeclaredProperty(
96                 property, prop.getType(), prop.getPropertyNode());
97           } else {
98             defineSynthesizedProperty(
99                 property, prop.getType(), prop.getPropertyNode());
100           }
101         }
102         // Freeze the record type.
103         isFrozen = true;
104       }
105     
106       /** @return Is this synthesized for internal bookkeeping? */
107       boolean isSynthetic() {
108         return !declared;
109       }
110     
111       @Override
112       public ObjectType getImplicitPrototype() {
113         return registry.getNativeObjectType(JSTypeNative.OBJECT_TYPE);
114       }
115     
116       @Override
117       boolean defineProperty(String propertyName, JSType type,
118           boolean inferred, Node propertyNode) {
119         if (isFrozen) {
120           return false;
121         }
122     
123         return super.defineProperty(propertyName, type, inferred,
124             propertyNode);
125       }
126     
127       JSType getGreatestSubtypeHelper(JSType that) {
128         if (that.isRecordType()) {
129           RecordType thatRecord = that.toMaybeRecordType();
130           RecordTypeBuilder builder = new RecordTypeBuilder(registry);
131           builder.setSynthesized(true);
132     
133           JSType noType = registry.getNativeObjectType(JSTypeNative.NO_TYPE);
134     
135           // The greatest subtype consists of those *unique* properties of both
136           // record types. If any property conflicts, then the NO_TYPE type
137           // is returned.
138           for (String property : getOwnPropertyNames()) {
139             JSType thisPropertyType = getPropertyType(property);
140             JSType propType = null;
141             if (thatRecord.hasProperty(property)) {
142               JSType thatPropertyType = thatRecord.getPropertyType(property);
143               propType = thisPropertyType.getGreatestSubtype(thatPropertyType);
144               if (propType.isEquivalentTo(noType)) { return noType; }
145             } else {
146               propType = thisPropertyType;
147             }
148             builder.addProperty(property, propType, getPropertyNode(property));
149           }
150     
151           for (String property : thatRecord.getOwnPropertyNames()) {
152             if (!hasProperty(property)) {
153               builder.addProperty(property, thatRecord.getPropertyType(property),
154                   thatRecord.getPropertyNode(property));
155             }
156           }
157           return builder.build();
158         }
159     
160         JSType greatestSubtype = registry.getNativeType(
161             JSTypeNative.NO_OBJECT_TYPE);
162         JSType thatRestrictedToObj =
163             registry.getNativeType(JSTypeNative.OBJECT_TYPE)
164             .getGreatestSubtype(that);
165         if (!thatRestrictedToObj.isEmptyType()) {
166           // In this branch, the other type is some object type. We find
167           // the greatest subtype with the following algorithm:
168           // 1) For each property "x" of this record type, take the union
169           //    of all classes with a property "x" with a compatible property type.
170           //    and which are a subtype of {@code that}.
171           // 2) Take the intersection of all of these unions.
172           for (String propName : getOwnPropertyNames()) {
173             JSType propType = getPropertyType(propName);
174             UnionTypeBuilder builder = new UnionTypeBuilder(registry);
175             for (ObjectType alt :
176               registry.getEachReferenceTypeWithProperty(propName)) {
177               JSType altPropType = alt.getPropertyType(propName);
178               if (altPropType != null && !alt.isEquivalentTo(this)
179                   && alt.isSubtype(that) && altPropType.isSubtype(propType)) {
180                 builder.addAlternate(alt);
181               }
182             }
183             greatestSubtype = greatestSubtype.getLeastSupertype(builder.build());
184           }
185         }
186         return greatestSubtype;
187       }
188     
189       @Override
190       public RecordType toMaybeRecordType() {
191         return this;
192       }
193     
194       @Override
195       public boolean isStructuralType() {
196         return true;
197       }
198     
199       @Override
200       public boolean isSubtype(JSType that) {
201         return isSubtype(that, ImplCache.create(), SubtypingMode.NORMAL);
202       }
203     
204       @Override
205       protected boolean isSubtype(JSType that,
206           ImplCache implicitImplCache, SubtypingMode subtypingMode) {
207         if (JSType.isSubtypeHelper(this, that, implicitImplCache, subtypingMode)) {
208           return true;
209         }
210     
211         // Top of the record types is the empty record, or OBJECT_TYPE.
212         if (registry.getNativeObjectType(
213                 JSTypeNative.OBJECT_TYPE).isSubtype(that, implicitImplCache, subtypingMode)) {
214           return true;
215         }
216     
217         // A type is a subtype of a record type if it itself is a record
218         // type and it has at least the same members as the parent record type
219         // with the same types.
220         if (!that.isRecordType()) {
221           return false;
222         }
223     
224         return this.isStructuralSubtype(that.toMaybeRecordType(), implicitImplCache, subtypingMode);
225       }
226     }
