46       /**
47        * With the map provided, replace the names with expression trees.
48        * @param node The root of the node tree within which to perform the
49        *     substitutions.
50        * @param parent The parent root node.
51        * @param replacements The map of names to template node trees with which
52        *     to replace the name Nodes.
53        * @return The root node or its replacement.
54        */
55       static Node inject(AbstractCompiler compiler, Node node, Node parent,
56           Map<String, Node> replacements) {
57         return inject(compiler, node, parent, replacements, true);
58       }

60       static Node inject(AbstractCompiler compiler, Node node, Node parent,
61           Map<String, Node> replacements, boolean replaceThis) {
62         if (node.isName()) {
63           Node replacementTemplate = replacements.get(node.getString());
64           if (replacementTemplate != null) {
65             // This should not be replacing declared names.
66             Preconditions.checkState(!parent.isFunction()
67                 || !parent.isVar()
68                 || !parent.isCatch());
69             // The name may need to be replaced more than once,
70             // so we need to clone the node.
71             Node replacement = replacementTemplate.cloneTree();
72             parent.replaceChild(node, replacement);
73             return replacement;
74           }
75         } else if (replaceThis && node.isThis()) {
76           Node replacementTemplate = replacements.get(THIS_MARKER);
77           Preconditions.checkNotNull(replacementTemplate);
78           if (!replacementTemplate.isThis()) {
79             // The name may need to be replaced more than once,
80             // so we need to clone the node.
81             Node replacement = replacementTemplate.cloneTree();
82             parent.replaceChild(node, replacement);
83     
84             // Remove the value.  This isn't required but it ensures that we won't
85             // inject side-effects multiple times as it will trigger the null
86             // check above if we do.
87             if (NodeUtil.mayHaveSideEffects(replacementTemplate, compiler)) {
88               replacements.remove(THIS_MARKER);
89             }
90     
91             return replacement;
92           }
93         } else if (node.isFunction()) {
94           // Once we enter another scope the "this" value changes, don't try
95           // to replace it within an inner scope.
96           replaceThis = false;
97         }
98     
99         for (Node c = node.getFirstChild(); c != null; c = c.getNext()) {
100           // We have to reassign c in case it was replaced, because the removed c's
101           // getNext() would no longer be correct.
102           c = inject(compiler, c, node, replacements, replaceThis);
103         }
104     
105         return node;
106       }

108       /**
109        * Get a mapping for function parameter names to call arguments.
110        */
111       static LinkedHashMap<String, Node> getFunctionCallParameterMap(
112           Node fnNode, Node callNode, Supplier<String> safeNameIdSupplier) {
113         // Create an argName -> expression map
114         // NOTE: A linked map is created here to provide ordering.
115         LinkedHashMap<String, Node> argMap = new LinkedHashMap<>();
116     
117         // CALL NODE: [ NAME, ARG1, ARG2, ... ]
118         Node cArg = callNode.getSecondChild();
119         if (cArg != null && NodeUtil.isFunctionObjectCall(callNode)) {
120           argMap.put(THIS_MARKER, cArg);
121           cArg = cArg.getNext();
122         } else {
123           // 'apply' isn't supported yet.
124           Preconditions.checkState(!NodeUtil.isFunctionObjectApply(callNode));
125           argMap.put(THIS_MARKER, NodeUtil.newUndefinedNode(callNode));
126         }
127     
128         for (Node fnArg : NodeUtil.getFunctionParameters(fnNode).children()) {
129           if (cArg != null) {
130             argMap.put(fnArg.getString(), cArg);
131             cArg = cArg.getNext();
132           } else {
133             Node srcLocation = callNode;
134             argMap.put(fnArg.getString(), NodeUtil.newUndefinedNode(srcLocation));
135           }
136         }
137     
138         // Add temp names for arguments that don't have named parameters in the
139         // called function.
140         while (cArg != null) {
141           String uniquePlaceholder =
142             getUniqueAnonymousParameterName(safeNameIdSupplier);
143           argMap.put(uniquePlaceholder, cArg);
144           cArg = cArg.getNext();
145         }
146     
147         return argMap;
148       }

158       /**
159        * Retrieve a set of names that can not be safely substituted in place.
160        * Example:
161        *   function(a) {
162        *     a = 0;
163        *   }
164        * Inlining this without taking precautions would cause the call site value
165        * to be modified (bad).
166        */
167       static Set<String> findModifiedParameters(Node fnNode) {
168         Set<String> names = getFunctionParameterSet(fnNode);
169         Set<String> unsafeNames = new HashSet<>();
170         return findModifiedParameters(
171             fnNode.getLastChild(), null, names, unsafeNames, false);
172       }

174       /**
175        * Check for uses of the named value that imply a pass-by-value
176        * parameter is expected.  This is used to prevent cases like:
177        *
178        *   function (x) {
179        *     x=2;
180        *     return x;
181        *   }
182        *
183        * We don't want "undefined" to be substituted for "x", and get
184        *   undefined=2
185        *
186        * @param n The node in question.
187        * @param parent The parent of the node.
188        * @param names The set of names to check.
189        * @param unsafe The set of names that require aliases.
190        * @param inInnerFunction Whether the inspection is occurring on a inner
191        *     function.
192        */
193       private static Set<String> findModifiedParameters(
194           Node n, Node parent, Set<String> names, Set<String> unsafe,
195           boolean inInnerFunction) {
196         Preconditions.checkArgument(unsafe != null);
197         if (n.isName()) {
198           if (names.contains(n.getString()) && (inInnerFunction || canNameValueChange(n, parent))) {
199             unsafe.add(n.getString());
200           }
201         } else if (n.isFunction()) {
202           // A function parameter can not be replaced with a direct inlined value
203           // if it is referred to by an inner function. The inner function
204           // can out live the call we are replacing, so inner function must
205           // capture a unique name.  This approach does not work within loop
206           // bodies so those are forbidden elsewhere.
207           inInnerFunction = true;
208         }
209     
210         for (Node c : n.children()) {
211           findModifiedParameters(c, n, names, unsafe, inInnerFunction);
212         }
213     
214         return unsafe;
215       }

217       /**
218        * This is similar to NodeUtil.isLValue except that object properties and
219        * array member modification aren't important ("o" in "o.a = 2" is still "o"
220        * after assignment, where in as "o = x", "o" is now "x").
221        *
222        * This also looks for the redefinition of a name.
223        *   function (x){var x;}
224        *
225        * @param n The NAME node in question.
226        * @param parent The parent of the node.
227        */
228       private static boolean canNameValueChange(Node n, Node parent) {
229         Token type = parent.getType();
230         return (type == Token.VAR || type == Token.INC || type == Token.DEC ||
231             (NodeUtil.isAssignmentOp(parent) && parent.getFirstChild() == n) ||
232             (NodeUtil.isForIn(parent)));
233       }

235       /**
236        * Updates the set of parameter names in set unsafe to include any
237        * arguments from the call site that require aliases.
238        * @param fnNode The FUNCTION node to be inlined.
239        * @param argMap The argument list for the call to fnNode.
240        * @param namesNeedingTemps The set of names to update.
241        */
242       static void maybeAddTempsForCallArguments(
243           Node fnNode, Map<String, Node> argMap, Set<String> namesNeedingTemps,
244           CodingConvention convention) {
245         if (argMap.isEmpty()) {
246           // No arguments to check, we are done.
247           return;
248         }
249     
250         Preconditions.checkArgument(fnNode.isFunction());
251         Node block = fnNode.getLastChild();
252     
253         int argCount = argMap.size();
254         // We limit the "trivial" bodies to those where there is a single expression or
255         // return, the expression is
256         boolean isTrivialBody = (!block.hasChildren()
257             || (block.hasOneChild() && !bodyMayHaveConditionalCode(block.getLastChild())));
258         boolean hasMinimalParameters = NodeUtil.isUndefined(argMap.get(THIS_MARKER))
259             && argCount <= 2; // this + one parameter
260     
261         Set<String> parameters = argMap.keySet();
262     
263         // Get the list of parameters that may need temporaries due to
264         // side-effects.
265         Set<String> namesAfterSideEffects = findParametersReferencedAfterSideEffect(
266             parameters, block);
267     
268         // Check for arguments that are evaluated more than once.
269         for (Map.Entry<String, Node> entry : argMap.entrySet()) {
270           String argName = entry.getKey();
271           if (namesNeedingTemps.contains(argName)) {
272             continue;
273           }
274           Node cArg = entry.getValue();
275           boolean safe = true;
276           int references = NodeUtil.getNameReferenceCount(block, argName);
277     
278           boolean argSideEffects = NodeUtil.mayHaveSideEffects(cArg);
279           if (!argSideEffects && references == 0) {
280             safe = true;
281           } else if (isTrivialBody && hasMinimalParameters
282               && references == 1
283               && !(NodeUtil.canBeSideEffected(cArg) && namesAfterSideEffects.contains(argName))) {
284             // For functions with a trivial body, and where the parameter evaluation order
285             // can't change, and there aren't any side-effect before the parameter, we can
286             // avoid creating a temporary.
287             //
288             // This is done to help inline common trivial functions
289             safe = true;
290           } else  if (NodeUtil.mayEffectMutableState(cArg) && references > 0) {
291             // Note: Mutable arguments should be assigned to temps, as the
292             // may be within in a loop:
293             //   function x(a) {
294             //     for(var i=0; i<0; i++) {
295             //       foo(a);
296             //     }
297             //   x( [] );
298             //
299             //   The parameter in the call to foo should not become "[]".
300             safe = false;
301           } else if (argSideEffects) {
302             // Even if there are no references, we still need to evaluate the
303             // expression if it has side-effects.
304             safe = false;
305           } else if (NodeUtil.canBeSideEffected(cArg)
306               && namesAfterSideEffects.contains(argName)) {
307             safe = false;
308           } else if (references > 1) {
309             // Safe is a misnomer, this is a check for "large".
310             switch (cArg.getType()) {
311               case NAME:
312                 String name = cArg.getString();
313                 safe = !(convention.isExported(name));
314                 break;
315               case THIS:
316                 safe = true;
317                 break;
318               case STRING:
319                 safe = (cArg.getString().length() < 2);
320                 break;
321               default:
322                 safe = NodeUtil.isImmutableValue(cArg);
323                 break;
324             }
325           }
326     
327           if (!safe) {
328             namesNeedingTemps.add(argName);
329           }
330         }
331       }

333       /**
334        * We consider a return or expression trivial if it doesn't contain a conditional expression or
335        * a function.
336        */
337       static boolean bodyMayHaveConditionalCode(Node n) {
338         if (!n.isReturn() && !n.isExprResult()) {
339           return true;
340         }
341         return mayHaveConditionalCode(n);
342       }
343     
344       /**
345        * We consider an expression trivial if it doesn't contain a conditional expression or
346        * a function.
347        */
348       static boolean mayHaveConditionalCode(Node n) {
349         for (Node c = n.getFirstChild(); c != null; c = c.getNext()) {
350           switch (c.getType()) {
351             case FUNCTION:
352             case AND:
353             case OR:
354             case HOOK:
355               return true;
356           }
357           if (mayHaveConditionalCode(c)) {
358             return true;
359           }
360         }
361         return false;
362       }

364       /**
365        * Boot strap a traversal to look for parameters referenced
366        * after a non-local side-effect.
367        * NOTE: This assumes no-inner functions.
368        * @param parameters The set of parameter names.
369        * @param root The function code block.
370        * @return The subset of parameters referenced after the first
371        *     seen non-local side-effect.
372        */
373       private static Set<String> findParametersReferencedAfterSideEffect(
374           Set<String> parameters, Node root) {
375     
376         // TODO(johnlenz): Consider using scope for this.
377         Set<String> locals = new HashSet<>(parameters);
378         gatherLocalNames(root, locals);
379     
380         ReferencedAfterSideEffect collector = new ReferencedAfterSideEffect(
381             parameters, locals);
382         NodeUtil.visitPostOrder(
383             root,
384             collector,
385             collector);
386         return collector.getResults();
387       }

389       /**
390        * Collect parameter names referenced after a non-local side-effect.
391        *
392        * Assumptions:
393        * - We assume parameters are not modified in the function body
394        * (that is checked separately).
395        * - There are no inner functions (also checked separately).
396        *
397        * As we are trying to replace parameters with there passed in values
398        * we are interested in anything that may affect those value.  So, ignoring
399        * changes to local variables, we look for things that may affect anything
400        * outside the local-state.  Once such a side-effect is seen any following
401        * reference to the function parameters are collected.  These will need
402        * to be assigned to temporaries to prevent changes to their value as would
403        * have happened during the function call.
404        *
405        * To properly handle loop structures all references to the function
406        * parameters are recorded and the decision to keep or throw away those
407        * references is deferred until exiting the loop structure.
408        */
409       private static class ReferencedAfterSideEffect
410           implements Visitor, Predicate<Node> {
411         private final Set<String> parameters;
412         private final Set<String> locals;
413         private boolean sideEffectSeen = false;
414         private Set<String> parametersReferenced = new HashSet<>();
415         private int loopsEntered = 0;
416     
417         ReferencedAfterSideEffect(Set<String> parameters, Set<String> locals) {
418           this.parameters = parameters;
419           this.locals = locals;
420         }
421     
422         Set<String> getResults() {
423           return parametersReferenced;
424         }
425     
426         @Override
427         public boolean apply(Node node) {
428           // Keep track of any loop structures entered.
429           if (NodeUtil.isLoopStructure(node)) {
430             loopsEntered++;
431           }
432     
433           // If we have found all the parameters, don't bother looking
434           // at the children.
435           return !(sideEffectSeen
436               && parameters.size() == parametersReferenced.size());
437         }
438     
439         boolean inLoop() {
440           return loopsEntered != 0;
441         }
442     
443         @Override
444         public void visit(Node n) {
445           // If we are exiting a loop.
446           if (NodeUtil.isLoopStructure(n)) {
447             loopsEntered--;
448             if (!inLoop() && !sideEffectSeen) {
449               // Now that the loops has been fully traversed and
450               // no side-effects have been seen, throw away
451               // the references seen in them.
452               parametersReferenced.clear();
453             }
454           }
455     
456           if (!sideEffectSeen) {
457             // Look for side-effects.
458             if (hasNonLocalSideEffect(n)) {
459               sideEffectSeen = true;
460             }
461           }
462     
463           // If traversing the nodes of a loop save any references
464           // that are seen.
465           if (inLoop() || sideEffectSeen) {
466             // Record references to parameters.
467             if (n.isName()) {
468               String name = n.getString();
469               if (parameters.contains(name)) {
470                 parametersReferenced.add(name);
471               }
472             } else if (n.isThis()) {
473               parametersReferenced.add(THIS_MARKER);
474             }
475           }
476         }
477     
478         /**
479          * @return Whether the node may have non-local side-effects.
480          */
481         private boolean hasNonLocalSideEffect(Node n) {
482           boolean sideEffect = false;
483           Token type = n.getType();
484           // Note: Only care about changes to non-local names, specifically
485           // ignore VAR declaration assignments.
486           if (NodeUtil.isAssignmentOp(n)
487               || type == Token.INC
488               || type == Token.DEC) {
489             Node lhs = n.getFirstChild();
490             // Ignore changes to local names.
491             if (!isLocalName(lhs)) {
492               sideEffect = true;
493             }
494           } else if (type == Token.CALL) {
495             sideEffect = NodeUtil.functionCallHasSideEffects(n);
496           } else if (type == Token.NEW) {
497             sideEffect = NodeUtil.constructorCallHasSideEffects(n);
498           } else if (type == Token.DELPROP) {
499             sideEffect = true;
500           }
501     
502           return sideEffect;
503         }
504     
505         /**
506          * @return Whether node is a reference to locally declared name.
507          */
508         private boolean isLocalName(Node node) {
509           if (node.isName()) {
510             String name = node.getString();
511             return locals.contains(name);
512           }
513           return false;
514         }
515       }

517       /**
518        * Gather any names declared in the local scope.
519        */
520       private static void gatherLocalNames(Node n, Set<String> names) {
521         if (n.isFunction()) {
522           if (NodeUtil.isFunctionDeclaration(n)) {
523             names.add(n.getFirstChild().getString());
524           }
525           // Don't traverse into inner function scopes;
526           return;
527         } else if (n.isName()) {
528           switch (n.getParent().getType()) {
529             case VAR:
530             case CATCH:
531               names.add(n.getString());
532           }
533         }
534     
535         for (Node c = n.getFirstChild(); c != null; c = c.getNext()) {
536           gatherLocalNames(c, names);
537         }
538       }

543       private static Set<String> getFunctionParameterSet(Node fnNode) {
