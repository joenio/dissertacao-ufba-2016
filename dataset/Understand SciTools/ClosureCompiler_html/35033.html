32     /**
33      * An object that optimizes the order of compiler passes.
34      *
35      * @author nicksantos@google.com (Nick Santos)
36      * @author dimvar@google.com (Dimitris Vardoulakis)
37      */
38     class PhaseOptimizer implements CompilerPass {

40       private static final Logger logger =
41           Logger.getLogger(PhaseOptimizer.class.getName());
42       private final AbstractCompiler compiler;
43       private final PerformanceTracker tracker;
44       private final List<CompilerPass> passes;
45       private boolean inLoop;
46       private PassFactory sanityCheck;
47       private boolean printAstHashcodes = false;

49       private double progress = 0.0;
50       private double progressStep = 0.0;
51       private final ProgressRange progressRange;

53       // These fields are used during optimization loops.
54       // They are declared here for two reasons:
55       // 1) Loop and ScopedChangeHandler can communicate via shared state
56       // 2) Compiler talks to PhaseOptimizer, not Loop or ScopedChangeHandler
57       private NamedPass currentPass;
58       // For each pass, remember the time at the end of the pass's last run.
59       private Map<NamedPass, Integer> lastRuns;
60       private Node currentScope;
61       // Starts at 0, increases as "interesting" things happen.
62       // Nothing happens at time START_TIME, the first pass starts at time 1.
63       // The correctness of scope-change tracking relies on Node/getIntProp
64       // returning 0 if the custom attribute on a node hasn't been set.
65       private int timestamp;
66       // The time of the last change made to the program by any pass.
67       private int lastChange;
68       private static final int START_TIME = 0;
69       private final Node jsRoot;
70       // Compiler/reportChangeToScope must call reportCodeChange to update all
71       // change handlers. This flag prevents double update in ScopedChangeHandler.
72       private boolean crossScopeReporting;
73     
74       // Used for sanity checks between loopable passes
75       private Node lastAst;
76       private Map<Node, Node> mtoc; // Stands for "main to clone"

78       /**
79        * When processing loopable passes in order, the PhaseOptimizer can be in one
80        * of these two states.
81        * <p>
82        * This enum is used by Loop/process only, but enum types can't be local.
83        */
84       enum State {
85         RUN_PASSES_NOT_RUN_IN_PREV_ITER,
86         RUN_PASSES_THAT_CHANGED_STH_IN_PREV_ITER
87       }
88     
89       // NOTE(dimvar): There used to be some code that tried various orderings of
90       // loopable passes and picked the fastest one. This code became stale
91       // gradually and I decided to remove it. It was also never tried after the
92       // new pass scheduler was written. If we need to revisit this order in the
93       // future, we should write new code to do it.
94       @VisibleForTesting
95       static final List<String> OPTIMAL_ORDER = ImmutableList.of(
96          "deadAssignmentsElimination",
97          "inlineFunctions",
98          "removeUnusedPrototypeProperties",
99          "removeUnreachableCode",
100          "removeUnusedVars",
101          "minimizeExitPoints",
102          "inlineVariables",
103          "collapseObjectLiterals",
104          "peepholeOptimizations");

106       static final int MAX_LOOPS = 100;
107       static final String OPTIMIZE_LOOP_ERROR =
108           "Fixed point loop exceeded the maximum number of iterations.";

110       /**
111        * @param comp the compiler that owns/creates this.
112        * @param tracker an optional performance tracker
113        * @param range the progress range for the process function or null
114        *        if progress should not be reported.
115        */
116       PhaseOptimizer(
117           AbstractCompiler comp, PerformanceTracker tracker, ProgressRange range) {
118         this.compiler = comp;
119         this.jsRoot = comp.getJsRoot();
120         this.tracker = tracker;
121         this.passes = new ArrayList<>();
122         this.progressRange = range;
123         this.inLoop = false;
124         this.crossScopeReporting = false;
125         this.timestamp = this.lastChange = START_TIME;
126       }
127     
128       /**
129        * Add the passes generated by the given factories to the compile sequence.
130        * <p>
131        * Automatically pulls multi-run passes into fixed point loops. If there
132        * are 1 or more multi-run passes in a row, they will run together in
133        * the same fixed point loop. The passes will run until they are finished
134        * making changes.
135        * <p>
136        * The PhaseOptimizer is free to tweak the order and frequency of multi-run
137        * passes in a fixed-point loop.
138        */
139       void consume(List<PassFactory> factories) {
140         Loop currentLoop = new Loop();
141         boolean isCurrentLoopPopulated = false;
142         for (PassFactory factory : factories) {
143           if (factory.isOneTimePass()) {
144             if (isCurrentLoopPopulated) {
145               passes.add(currentLoop);
146               currentLoop = new Loop();
147               isCurrentLoopPopulated = false;
148             }
149             addOneTimePass(factory);
150           } else {
151             currentLoop.addLoopedPass(factory);
152             isCurrentLoopPopulated = true;
153           }
154         }
155     
156         if (isCurrentLoopPopulated) {
157           passes.add(currentLoop);
158         }
159       }
160     
161       /**
162        * Add the pass generated by the given factory to the compile sequence.
163        * This pass will be run once.
164        */
165       @VisibleForTesting
166       void addOneTimePass(PassFactory factory) {
167         passes.add(new NamedPass(factory));
168       }
169     
170       /**
171        * Add a loop to the compile sequence. This loop will continue running
172        * until the AST stops changing.
173        * @return The loop structure. Pass suppliers should be added to the loop.
174        */
175       Loop addFixedPointLoop() {
176         Loop loop = new Loop();
177         passes.add(loop);
178         return loop;
179       }

181       /**
182        * Adds a sanity checker to be run after every pass. Intended for development.
183        */
184       void setSanityCheck(PassFactory sanityCheck) {
185         this.sanityCheck = sanityCheck;
186         setSanityCheckState();
187       }

189       private void setSanityCheckState() {
190         if (inLoop) {
191           lastAst = jsRoot.cloneTree();
192           mtoc = NodeUtil.mapMainToClone(jsRoot, lastAst);
193         }
194       }
195     
196       /**
197        * Sets the hashcode of the AST to be logged every pass.
198        * Intended for development.
199        */
200       void setPrintAstHashcodes(boolean printAstHashcodes) {
201         this.printAstHashcodes = printAstHashcodes;
202       }

204       /**
205        * Run all the passes in the optimizer.
206        */
207       @Override
208       public void process(Node externs, Node root) {
209         progress = 0.0;
210         progressStep = 0.0;
211         if (progressRange != null) {
212           progressStep = (progressRange.maxValue - progressRange.initialValue)
213               / passes.size();
214           progress = progressRange.initialValue;
215         }
216         for (CompilerPass pass : passes) {
217           pass.process(externs, root);
218           if (hasHaltingErrors()) {
219             return;
220           }
221         }
222       }
223     
224       private void maybePrintAstHashcodes(String passName, Node root) {
225         if (printAstHashcodes) {
226           String hashCodeMsg = "AST hashCode after " + passName + ": " +
227               compiler.toSource(root).hashCode();
228           System.err.println(hashCodeMsg);
229           compiler.addToDebugLog(hashCodeMsg);
230         }
231       }
232     
233       /**
234        * Runs the sanity check if it is available.
235        */
236       private void maybeSanityCheck(Node externs, Node root) {
237         if (sanityCheck != null) {
238           sanityCheck.create(compiler).process(externs, root);
239           // The cross-module passes are loopable and ran together, but do not
240           // participate in the other optimization loops, and are not relevant to
241           // tracking changed scopes.
242           if (inLoop &&
243               !currentPass.name.equals(Compiler.CROSS_MODULE_CODE_MOTION_NAME) &&
244               !currentPass.name.equals(Compiler.CROSS_MODULE_METHOD_MOTION_NAME)) {
245             NodeUtil.verifyScopeChanges(mtoc, jsRoot, true);
246           }
247         }
248       }

250       private boolean hasHaltingErrors() {
251         return compiler.hasHaltingErrors();
252       }

254       /**
255        * A single compiler pass.
256        */
257       class NamedPass implements CompilerPass {
258         final String name;
259         private final PassFactory factory;
260         private Tracer tracer;
261     
262         NamedPass(PassFactory factory) {
263           this.name = factory.getName();
264           this.factory = factory;
265         }
266     
267         @Override
268         public void process(Node externs, Node root) {
269           logger.fine(name);
270           if (sanityCheck != null) {
271             // Before running the pass, clone the AST so you can sanity-check the
272             // changed AST against the clone after the pass finishes.
273             setSanityCheckState();
274           }
275           if (tracker != null) {
276             tracker.recordPassStart(name, factory.isOneTimePass());
277           }
278           tracer = new Tracer("JSCompiler");
279     
280           compiler.beforePass(name);
281     
282           // Delay the creation of the actual pass until *after* all previous passes
283           // have been processed.
284           // Some precondition checks rely on this, eg, in CoalesceVariableNames.
285           factory.create(compiler).process(externs, root);
286     
287           compiler.afterPass(name);
288     
289           try {
290             if (progressRange == null) {
291               compiler.setProgress(-1, name);
292             } else {
293               progress += progressStep;
294               compiler.setProgress(progress, name);
295             }
296             // Don't move this line in the IF. We create a Tracer even when the tracker
297             // is null; so we must also stop the tracer when the tracker is null.
298             // Otherwise, Tracer.ThreadTrace#events can become too big.
299             long traceRuntime = tracer.stop();
300             if (tracker != null) {
301               tracker.recordPassStop(name, traceRuntime);
302             }
303             maybePrintAstHashcodes(name, root);
304             maybeSanityCheck(externs, root);
305           } catch (IllegalStateException e) {
306             // TODO(johnlenz): Remove this once the normalization checks report
307             // errors instead of exceptions.
308             throw new RuntimeException("Sanity check failed for " + name, e);
309           }
310         }
311     
312         @Override
313         public String toString() {
314           return "pass: " + name;
315         }
316       }

318       void setScope(Node n) {
319         // NodeTraversal causes setScope calls outside loops; ignore them.
320         if (inLoop) {
321           // Find the top-level node in the scope.
322           currentScope = n.isFunction() ? n : getEnclosingScope(n);
323         }
324       }
325     
326       boolean hasScopeChanged(Node n) {
327         // Outside loops we don't track changed scopes, so we visit them all.
328         if (!inLoop) {
329           return true;
330         }
331         int timeOfLastRun = lastRuns.get(currentPass);
332         // A pass looks at all functions when it first runs
333         return timeOfLastRun == START_TIME
334             || n.getChangeTime() > timeOfLastRun;
335       }
336     
337       private Node getEnclosingScope(Node n) {
338         while (n != jsRoot && n.getParent() != null) {
339           n = n.getParent();
340           if (n.isFunction()) {
341             return n;
342           }
343         }
344         return n;
345       }
346     
347       void reportChangeToEnclosingScope(Node n) {
348         lastChange = timestamp;
349         getEnclosingScope(n).setChangeTime(timestamp);
350         // Every code change happens at a different time
351         timestamp++;
352       }

354       /**
355        * Records that the currently-running pass may report cross-scope changes.
356        * When this happens, we don't want to falsely report the current scope as
357        * changed when reportChangeToScope is called from Compiler.
358        */
359       void startCrossScopeReporting() {
360         crossScopeReporting = true;
361       }
362     
363       /** The currently-running pass won't report cross-scope changes. */
364       void endCrossScopeReporting() {
365         crossScopeReporting = false;
366       }

368       /**
369        * A change handler that marks scopes as changed when reportChange is called.
370        */
371       private class ScopedChangeHandler implements CodeChangeHandler {
372         private int lastCodeChangeQuery;
373     
374         ScopedChangeHandler() {
375           this.lastCodeChangeQuery = timestamp;
376         }
377     
378         @Override
379         public void reportChange() {
380           if (crossScopeReporting) {
381             // This call was caused by Compiler/reportChangeToEnclosingScope,
382             // do nothing.
383             return;
384           }
385           lastChange = timestamp;
386           currentScope.setChangeTime(timestamp);
387           // Every code change happens at a different time
388           timestamp++;
389         }
390     
391         private boolean hasCodeChangedSinceLastCall() {
392           boolean result = lastChange > lastCodeChangeQuery;
393           lastCodeChangeQuery = timestamp;
394           // The next call to the method will happen at a different time
395           timestamp++;
396           return result;
397         }
398       }

400       /**
401        * A compound pass that contains atomic passes and runs them until they reach
402        * a fixed point.
403        * <p>
404        * Notice that this is a non-static class, because it includes the closure
405        * of PhaseOptimizer.
406        */
407       @VisibleForTesting
408       class Loop implements CompilerPass {
409         private final List<NamedPass> myPasses = new ArrayList<>();
410         private final Set<String> myNames = new HashSet<>();
411         private ScopedChangeHandler scopeHandler;
412     
413         void addLoopedPass(PassFactory factory) {
414           String name = factory.getName();
415           Preconditions.checkArgument(!myNames.contains(name),
416               "Already a pass with name '%s' in this loop", name);
417           myNames.add(name);
418           myPasses.add(new NamedPass(factory));
419         }
420     
421         @Override
422         public void process(Node externs, Node root) {
423           Preconditions.checkState(!inLoop, "Nested loops are forbidden");
424           inLoop = true;
425           optimizePasses();
426     
427           // Set up function-change tracking
428           scopeHandler = new ScopedChangeHandler();
429           compiler.addChangeHandler(scopeHandler);
430           setScope(root);
431           // lastRuns is initialized before each loop. This way, when a pass is run
432           // in the 2nd loop for the 1st time, it looks at all scopes.
433           lastRuns = new HashMap<>();
434           for (NamedPass pass : myPasses) {
435             lastRuns.put(pass, START_TIME);
436           }
437           // Contains a pass iff it made changes the last time it was run.
438           Set<NamedPass> madeChanges = new HashSet<>();
439           // Contains a pass iff it was run during the last inner loop.
440           Set<NamedPass> runInPrevIter = new HashSet<>();
441           State state = State.RUN_PASSES_NOT_RUN_IN_PREV_ITER;
442           boolean lastIterMadeChanges;
443           int count = 0;
444     
445           try {
446             while (true) {
447               if (count++ > MAX_LOOPS) {
448                 compiler.throwInternalError(OPTIMIZE_LOOP_ERROR, null);
449               }
450               lastIterMadeChanges = false;
451               for (NamedPass pass : myPasses) {
452                 if ((state == State.RUN_PASSES_NOT_RUN_IN_PREV_ITER
453                         && !runInPrevIter.contains(pass))
454                     || (state == State.RUN_PASSES_THAT_CHANGED_STH_IN_PREV_ITER
455                             && madeChanges.contains(pass))) {
456                   timestamp++;
457                   currentPass = pass;
458                   pass.process(externs, root);
459                   runInPrevIter.add(pass);
460                   lastRuns.put(pass, timestamp);
461                   if (hasHaltingErrors()) {
462                     return;
463                   } else if (scopeHandler.hasCodeChangedSinceLastCall()) {
464                     madeChanges.add(pass);
465                     lastIterMadeChanges = true;
466                   } else {
467                     madeChanges.remove(pass);
468                   }
469                 } else {
470                   runInPrevIter.remove(pass);
471                 }
472               }
473     
474               if (state == State.RUN_PASSES_NOT_RUN_IN_PREV_ITER) {
475                 if (lastIterMadeChanges) {
476                   state = State.RUN_PASSES_THAT_CHANGED_STH_IN_PREV_ITER;
477                 } else {
478                   return;
479                 }
480               } else { // state == State.RUN_PASSES_THAT_CHANGED_STH_IN_PREV_ITER
481                 if (!lastIterMadeChanges) {
482                   state = State.RUN_PASSES_NOT_RUN_IN_PREV_ITER;
483                 }
484               }
485             }
486           } finally {
487             inLoop = false;
488             compiler.removeChangeHandler(scopeHandler);
489           }
490         }
491     
492         /** Re-arrange the passes in an optimal order. */
493         private void optimizePasses() {
494           // It's important that this ordering is deterministic, so that
495           // multiple compiles with the same input produce exactly the same
496           // results.
497           //
498           // To do this, grab any passes we recognize, and move them to the end
499           // in an "optimal" order.
500           List<NamedPass> optimalPasses = new ArrayList<>();
501           for (String passInOptimalOrder : OPTIMAL_ORDER) {
502             for (NamedPass loopablePass : myPasses) {
503               if (loopablePass.name.equals(passInOptimalOrder)) {
504                 optimalPasses.add(loopablePass);
505                 break;
506               }
507             }
508           }
509     
510           myPasses.removeAll(optimalPasses);
511           myPasses.addAll(optimalPasses);
512         }
513       }
