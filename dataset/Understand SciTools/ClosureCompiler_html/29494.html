40     package com.google.javascript.rhino.jstype;
41     
42     import static com.google.javascript.rhino.jstype.TernaryValue.FALSE;
43     import static com.google.javascript.rhino.jstype.TernaryValue.TRUE;
44     
45     import com.google.javascript.rhino.ErrorReporter;
46     import com.google.javascript.rhino.Node;
47     
48     import java.util.Collections;
49     import java.util.HashSet;
50     import java.util.Set;
51     
52     /**
53      * An enum type representing a branded collection of elements. Each element
54      * is referenced by its name, and has an {@link EnumElementType} type.
55      */
56     public class EnumType extends PrototypeObjectType {
57       private static final long serialVersionUID = 1L;
58     
59       // the type of the individual elements
60       private EnumElementType elementsType;
61       // the elements' names (they all have the same type)
62       private final Set<String> elements = new HashSet<>();
63     
64       /**
65        * Creates an enum type.
66        *
67        * @param name the enum's name
68        * @param elementsType the base type of the individual elements
69        */
70       EnumType(JSTypeRegistry registry, String name, Node source,
71           JSType elementsType) {
72         super(registry, "enum{" + name + "}", null);
73         this.elementsType = new EnumElementType(registry, elementsType, name, this);
74       }
75     
76       @Override
77       public EnumType toMaybeEnumType() {
78         return this;
79       }
80     
81       @Override
82       public ObjectType getImplicitPrototype() {
83         return registry.getNativeObjectType(JSTypeNative.OBJECT_TYPE);
84       }
85     
86       /**
87        * Gets the elements defined on this enum.
88        * @return the elements' names defined on this enum. The returned set is
89        *         immutable.
90        */
91       public Set<String> getElements() {
92         return Collections.unmodifiableSet(elements);
93       }
94     
95       /**
96        * Defines a new element on this enum.
97        * @param name the name of the new element
98        * @param definingNode the {@code Node} that defines this new element
99        * @return true iff the new element is added successfully
100        */
101       public boolean defineElement(String name, Node definingNode) {
102         elements.add(name);
103         return defineDeclaredProperty(name, elementsType, definingNode);
104       }
105     
106       /**
107        * Gets the elements' type.
108        */
109       public EnumElementType getElementsType() {
110         return elementsType;
111       }
112     
113       @Override
114       public TernaryValue testForEquality(JSType that) {
115         TernaryValue result = super.testForEquality(that);
116         if (result != null) {
117           return result;
118         }
119         return this.isEquivalentTo(that) ? TRUE : FALSE;
120       }
121     
122       @Override
123       public boolean isSubtype(JSType that) {
124         return isSubtype(that, ImplCache.create(), SubtypingMode.NORMAL);
125       }
126     
127       @Override
128       protected boolean isSubtype(JSType that,
129           ImplCache implicitImplCache, SubtypingMode subtypingMode) {
130         return that.isEquivalentTo(getNativeType(JSTypeNative.OBJECT_TYPE)) ||
131             that.isEquivalentTo(getNativeType(JSTypeNative.OBJECT_PROTOTYPE)) ||
132             JSType.isSubtypeHelper(this, that, implicitImplCache, subtypingMode);
133       }
134     
135       @Override
136       String toStringHelper(boolean forAnnotations) {
137         return forAnnotations ? "Object" : getReferenceName();
138       }
139     
140       @Override
141       public String getDisplayName() {
142         return elementsType.getDisplayName();
143       }
144     
145       @Override
146       public <T> T visit(Visitor<T> visitor) {
147         return visitor.caseObjectType(this);
148       }
149     
150       @Override <T> T visit(RelationshipVisitor<T> visitor, JSType that) {
151         return visitor.caseObjectType(this, that);
152       }
153     
154       @Override
155       public FunctionType getConstructor() {
156         return null;
157       }
158     
159       @Override
160       public boolean matchesNumberContext() {
161         return false;
162       }
163     
164       @Override
165       public boolean matchesStringContext() {
166         return true;
167       }
168     
169       @Override
170       public boolean matchesObjectContext() {
171         return true;
172       }
173     
174       @Override
175       JSType resolveInternal(ErrorReporter t, StaticTypedScope<JSType> scope) {
176         elementsType = (EnumElementType) elementsType.resolve(t, scope);
177         return super.resolveInternal(t, scope);
178       }
179     }
