17     package com.google.javascript.jscomp.parsing.parser;
18     
19     import java.io.Serializable;
20     import java.util.Objects;
21     
22     /**
23      * Represents various aspects of language version and support.
24      *
25      * <p>This is somewhat redundant with LanguageMode, but is separate
26      * for two reasons: (1) it's used for parsing, which cannot
27      * depend on LanguageMode, and (2) it's concerned with slightly
28      * different nuances: implemented features and modules rather
29      * than strictness.
30      *
31      * <p>In the long term, it would be good to disentangle all these
32      * concerns and pull out a single LanguageSyntax enum with a
33      * separate strict mode flag, and then these could possibly be
34      * unified.
35      *
36      * <p>Instances of this class are immutable.
37      */
38     public final class FeatureSet implements Serializable {
39     
40       /** The number of the language version: 3, 5, or 6. */
41       private final int number;
42       /** Whether this includes features not supported in current stable browsers. */
43       private final boolean unsupported;
44       /** Whether ES6 modules are included. */
45       private final boolean es6Modules;
46       /** Whether TypeScript syntax is included (for .d.ts support). */
47       private final boolean typeScript;
48     
49       /** The bare minimum set of features in ES3. */
50       public static final FeatureSet ES3 = new FeatureSet(3, false, false, false);
51       /** Features from ES5 only. */
52       public static final FeatureSet ES5 = new FeatureSet(5, false, false, false);
53       /** The subset of ES6 features that are implemented in stable Chrome, Firefox, and Edge. */
54       public static final FeatureSet ES6_IMPL = new FeatureSet(6, false, false, false);
55       /** The full set of ES6 features, not including modules. */
56       public static final FeatureSet ES6 = new FeatureSet(6, true, false, false);
57       /** All ES6 features, including modules. */
58       public static final FeatureSet ES6_MODULES = new FeatureSet(6, true, true, false);
59       public static final FeatureSet ES7 = new FeatureSet(7, true, false, false);
60       public static final FeatureSet ES7_MODULES = new FeatureSet(7, true, true, false);
61       public static final FeatureSet ES8 = new FeatureSet(8, true, false, false);
62       public static final FeatureSet ES8_MODULES = new FeatureSet(8, true, true, false);
63       /** TypeScript syntax. */
64       public static final FeatureSet TYPESCRIPT = new FeatureSet(8, true, true, true);
65     
66       /**
67        * Specific features that can be included (indirectly) in a FeatureSet.
68        * This primarily adds a name so that helper functions can simultaneously
69        * update the detected features and also warn about unsupported features
70        * by name.  Additionally, collecting these all in one place provides a
71        * single file that needs to be edited to update the current browser support.
72        */
73       public enum Feature {
74         // ES5 features
75         ES3_KEYWORDS_AS_IDENTIFIERS("ES3 keywords as identifiers", ES5),
76         GETTER("getters", ES5),
77         KEYWORDS_AS_PROPERTIES("reserved words as properties", ES5),
78         SETTER("setters", ES5),
79         STRING_CONTINUATION("string continuation", ES5),
80         TRAILING_COMMA("trailing comma", ES5),
81     
82         // ES6 features that are already implemented in modern stable browsers
83         // (Chrome 50, Firefox 46, and Edge 13)
84         ARROW_FUNCTIONS("arrow function", ES6_IMPL),
85         BINARY_LITERALS("binary literal", ES6_IMPL),
86         OCTAL_LITERALS("octal literal", ES6_IMPL),
87         CLASSES("class", ES6_IMPL),
88         COMPUTED_PROPERTIES("computed property", ES6_IMPL),
89         EXTENDED_OBJECT_LITERALS("extended object literal", ES6_IMPL),
90         FOR_OF("for-of loop", ES6_IMPL),
91         GENERATORS("generator", ES6_IMPL),
92         LET_DECLARATIONS("let declaration", ES6_IMPL),
93         MEMBER_DECLARATIONS("member declaration", ES6_IMPL),
94         REGEXP_FLAG_Y("RegExp flag 'y'", ES6_IMPL),
95         REST_PARAMETERS("rest parameter", ES6_IMPL),
96         SPREAD_EXPRESSIONS("spread expression", ES6_IMPL),
97         SUPER("super", ES6_IMPL),
98         TEMPLATE_LITERALS("template literal", ES6_IMPL),
99     
100         // ES6 features that are not yet implemented in all "modern" browsers
101         // The following features will become stable once Edge 14 is released:
102         CONST_DECLARATIONS("const declaration", ES6),
103         DESTRUCTURING("destructuring", ES6),
104         NEW_TARGET("new.target", ES6), // Not fully supported until Edge 14
105         REGEXP_FLAG_U("RegExp flag 'u'", ES6), // (note: case folding still broken even in Edge 14)
106         // The following features are still broken even in Firefox 49:
107         // See https://bugzilla.mozilla.org/show_bug.cgi?id=1187502 for details
108         DEFAULT_PARAMETERS("default parameter", ES6),
109     
110         // ES6 features that include modules
111         MODULES("modules", ES6_MODULES),
112     
113         // '**' operator
114         EXPONENT_OP("exponent operator (**)", ES7),
115     
116         // http://tc39.github.io/ecmascript-asyncawait/
117         ASYNC_FUNCTIONS("async function", ES8),
118     
119         // ES6 typed features that are not at all implemented in browsers
120         AMBIENT_DECLARATION("ambient declaration", TYPESCRIPT),
121         CALL_SIGNATURE("call signature", TYPESCRIPT),
122         CONSTRUCTOR_SIGNATURE("constructor signature", TYPESCRIPT),
123         ENUM("enum", TYPESCRIPT),
124         GENERICS("generics", TYPESCRIPT),
125         IMPLEMENTS("implements", TYPESCRIPT),
126         INDEX_SIGNATURE("index signature", TYPESCRIPT),
127         INTERFACE("interface", TYPESCRIPT),
128         MEMBER_VARIABLE_IN_CLASS("member variable in class", TYPESCRIPT),
129         NAMESPACE_DECLARATION("namespace declaration", TYPESCRIPT),
130         OPTIONAL_PARAMETER("optional parameter", TYPESCRIPT),
131         TYPE_ALIAS("type alias", TYPESCRIPT),
132         TYPE_ANNOTATION("type annotation", TYPESCRIPT);
133     
134         private final String name;
135         private final FeatureSet features;
136     
137         private Feature(String name, FeatureSet features) {
138           this.name = name;
139           this.features = features;
140         }
141     
142         public FeatureSet features() {
143           return features;
144         }
145     
146         @Override
147         public String toString() {
148           return name;
149         }
150       }
151     
152       private FeatureSet(int number, boolean unsupported, boolean es6Modules, boolean typeScript) {
153         this.number = number;
154         this.unsupported = unsupported;
155         this.es6Modules = es6Modules;
156         this.typeScript = typeScript;
157       }
158     
159       /** Returns a string representation suitable for encoding in depgraph and deps.js files. */
160       public String version() {
161         if (typeScript) {
162           return "ts";
163         } else if (number > 6) {
164           return "es" + number;
165         } else if (unsupported || es6Modules) {
166           return "es6";
167         } else if (number > 5) {
168           return "es6-impl";
169         } else if (number > 3) {
170           return "es5";
171         }
172         return "es3";
173       }
174     
175       /** Returns whether this feature set includes ES6 modules. */
176       public boolean hasEs6Modules() {
177         return es6Modules;
178       }
179     
180       /** Returns whether this feature set includes typescript features. */
181       public boolean isTypeScript() {
182         return typeScript;
183       }
184     
185       /** Returns a feature set combining all the features from {@code this} and {@code other}. */
186       public FeatureSet require(FeatureSet other) {
187         return this.contains(other) ? this : this.union(other);
188       }
189     
190       /**
191        * Returns a new {@link FeatureSet} including all features of both {@code this} and {@code other}.
192        */
193       public FeatureSet union(FeatureSet other) {
194         return new FeatureSet(
195             Math.max(number, other.number),
196             unsupported || other.unsupported,
197             es6Modules || other.es6Modules,
198             typeScript || other.typeScript);
199       }
200     
201       /**
202        * Does this {@link FeatureSet} contain all of the features of {@code other}?
203        */
204       public boolean contains(FeatureSet other) {
205         return this.number >= other.number
206             && (this.unsupported || !other.unsupported)
207             && (this.es6Modules || !other.es6Modules)
208             && (this.typeScript || !other.typeScript);
209       }
210     
211       /** Returns a feature set combining all the features from {@code this} and {@code feature}. */
212       public FeatureSet require(Feature feature) {
213         return require(feature.features);
214       }
215     
216       /**
217        * Does this {@link FeatureSet} include {@code feature}?
218        */
219       public boolean contains(Feature feature) {
220         return contains(feature.features());
221       }
222     
223       @Override
224       public boolean equals(Object other) {
225         return other instanceof FeatureSet
226             && ((FeatureSet) other).number == number
227             && ((FeatureSet) other).unsupported == unsupported
228             && ((FeatureSet) other).es6Modules == es6Modules
229             && ((FeatureSet) other).typeScript == typeScript;
230       }
231     
232       @Override
233       public int hashCode() {
234         return Objects.hash(number, unsupported, es6Modules, typeScript);
235       }
236     
237       @Override
238       public String toString() {
239         return "FeatureSet{number=" + number
240             + (unsupported ? ", unsupported" : "")
241             + (es6Modules ? ", es6Modules" : "")
242             + (typeScript ? ", typeScript" : "")
243             + "}";
244       }
245     
246       /** Returns a the name of a corresponding LanguageMode enum element. */
247       public String toLanguageModeString() {
248         return "ECMASCRIPT" + number;
249       }
250     }
