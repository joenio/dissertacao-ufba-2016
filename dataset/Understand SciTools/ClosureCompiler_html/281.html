19     import com.google.common.collect.ImmutableMap;
20     import com.google.javascript.rhino.JSDocInfo;
21     import com.google.javascript.rhino.JSDocInfo.Visibility;
22     import com.google.javascript.rhino.Node;

24     import com.google.javascript.rhino.jstype.JSType;

30     /**
31      * Helper functions for computing the visibility of names and properties
32      * in JavaScript source code.
33      *
34      * @author brndn@google.com (Brendan Linn)
35      * @see CheckAccessControls
36      */
37     public final class AccessControlUtils {
38     
39       /** Non-instantiable. */
40       private AccessControlUtils() {}
41     
42       /**
43        * Returns the effective visibility of the given name. This can differ
44        * from the name's declared visibility if the file's {@code @fileoverview}
45        * JsDoc specifies a default visibility.
46        *
47        * @param name The name node to compute effective visibility for.
48        * @param var The name to compute effective visibility for.
49        * @param fileVisibilityMap A map of {@code @fileoverview} visibility
50        *     annotations, used to compute the name's default visibility.
51        */
52       static Visibility getEffectiveNameVisibility(Node name, Var var,
53           ImmutableMap<StaticSourceFile, Visibility> fileVisibilityMap) {
54         JSDocInfo jsDocInfo = var.getJSDocInfo();
55         Visibility raw = (jsDocInfo == null || jsDocInfo.getVisibility() == null)
56             ? Visibility.INHERITED
57             : jsDocInfo.getVisibility();
58         if (raw != Visibility.INHERITED) {
59           return raw;
60         }
61         Visibility defaultVisibilityForFile =
62             fileVisibilityMap.get(var.getSourceFile());
63         JSType type = name.getJSType();
64         boolean createdFromGoogProvide = (type instanceof PrototypeObjectType
65             && ((PrototypeObjectType) type).isAnonymous());
66         // Ignore @fileoverview visibility when computing the effective visibility
67         // for names created by goog.provide.
68         //
69         // ProcessClosurePrimitives rewrites goog.provide()s as object literal
70         // declarations, but the exact form depends on the ordering of the
71         // input files. If goog.provide('a.b') occurs in the inputs before
72         // goog.provide('a'), it is rewritten like
73         //
74         // var a={};a.b={};
75         //
76         // If the file containing goog.provide('a.b') also declares a @fileoverview
77         // visibility, it must not apply to a, as this would make every a.* namespace
78         // effectively package-private.
79         return (createdFromGoogProvide || defaultVisibilityForFile == null)
80             ? raw
81             : defaultVisibilityForFile;
82       }
83     
84       /**
85        * Returns the effective visibility of the given property. This can differ
86        * from the property's declared visibility if the property is inherited from
87        * a superclass, or if the file's {@code @fileoverview} JsDoc specifies
88        * a default visibility.
89        *
90        * @param property The property to compute effective visibility for.
91        * @param referenceType The JavaScript type of the property.
92        * @param fileVisibilityMap A map of {@code @fileoverview} visibility
93        *     annotations, used to compute the property's default visibility.
94        * @param codingConvention The coding convention in effect (if any),
95        *     used to determine whether the property is private by lexical convention
96        *     (example: trailing underscore).
97        */
98       static Visibility getEffectivePropertyVisibility(
99           Node property,
100           ObjectType referenceType,
101           ImmutableMap<StaticSourceFile, Visibility> fileVisibilityMap,
102           @Nullable CodingConvention codingConvention) {
103         String propertyName = property.getLastChild().getString();
104         StaticSourceFile definingSource = getDefiningSource(
105             property, referenceType, propertyName);
106         Visibility fileOverviewVisibility = fileVisibilityMap.get(definingSource);
107         Node parent = property.getParent();
108         boolean isOverride = parent.getJSDocInfo() != null
109             && parent.isAssign()
110             && parent.getFirstChild() == property;
111         ObjectType objectType = getObjectType(
112             referenceType, isOverride, propertyName);
113         if (isOverride) {
114           Visibility overridden = getOverriddenPropertyVisibility(
115               objectType, propertyName);
116           return getEffectiveVisibilityForOverriddenProperty(
117               overridden, fileOverviewVisibility, propertyName, codingConvention);
118         } else {
119           return getEffectiveVisibilityForNonOverriddenProperty(
120               property, objectType, fileOverviewVisibility, codingConvention);
121         }
122       }
123     
124       /**
125        * Returns the source file in which the given property is defined,
126        * or null if it is not known.
127        */
128       @Nullable private static StaticSourceFile getDefiningSource(
129           Node getprop, @Nullable ObjectType referenceType, String propertyName) {
130         if (referenceType != null) {
131           Node propDefNode = referenceType.getPropertyNode(propertyName);
132           if (propDefNode != null) {
133             return propDefNode.getStaticSourceFile();
134           }
135         }
136         return getprop.getStaticSourceFile();
137       }
138     
139       /**
140        * Returns the lowest property defined on a class with visibility information.
141        */
142       @Nullable private static ObjectType getObjectType(
143           @Nullable ObjectType referenceType,
144           boolean isOverride,
145           String propertyName) {
146         if (referenceType == null) {
147           return null;
148         }
149     
150         ObjectType objectType = isOverride
151             ? referenceType.getImplicitPrototype()
152             : referenceType;
153         for (; objectType != null;
154             objectType = objectType.getImplicitPrototype()) {
155           JSDocInfo docInfo = objectType.getOwnPropertyJSDocInfo(propertyName);
156           if (docInfo != null
157               && docInfo.getVisibility() != Visibility.INHERITED) {
158             return objectType;
159           }
160         }
161         return null;
162       }
163     
164       /**
165        * Returns the original visibility of an overridden property.
166        */
167       private static Visibility getOverriddenPropertyVisibility(
168           ObjectType objectType, String propertyName) {
169         return objectType != null
170             ? objectType.getOwnPropertyJSDocInfo(propertyName).getVisibility()
171             : Visibility.INHERITED;
172       }
173     
174       /**
175        * Returns the effective visibility of the given overridden property.
176        * An overridden propertiy inherits the visibility of the property it
177        * overrides.
178        */
179       private static Visibility getEffectiveVisibilityForOverriddenProperty(
180           Visibility visibility,
181           @Nullable Visibility fileOverviewVisibility,
182           String propertyName,
183           @Nullable CodingConvention codingConvention) {
184         if (codingConvention != null && codingConvention.isPrivate(propertyName)) {
185           return Visibility.PRIVATE;
186         }
187         return (fileOverviewVisibility != null
188             && visibility == Visibility.INHERITED)
189             ? fileOverviewVisibility
190             : visibility;
191       }
192     
193       /**
194        * Returns the effective visibility of the given non-overridden property.
195        * Non-overridden properties without an explicit visibility annotation
196        * receive the default visibility declared in the file's {@code @fileoverview}
197        * block, if one exists.
198        */
199       private static Visibility getEffectiveVisibilityForNonOverriddenProperty(
200           Node getprop,
201           ObjectType objectType,
202           @Nullable Visibility fileOverviewVisibility,
203           @Nullable CodingConvention codingConvention) {
204         String propertyName = getprop.getLastChild().getString();
205         if (codingConvention != null && codingConvention.isPrivate(propertyName)) {
206           return Visibility.PRIVATE;
207         }
208         Visibility raw = Visibility.INHERITED;
209         if (objectType != null) {
210           raw = objectType.getOwnPropertyJSDocInfo(propertyName).getVisibility();
211         }
212         JSType type = getprop.getJSType();
213         boolean createdFromGoogProvide = (type instanceof PrototypeObjectType
214             && ((PrototypeObjectType) type).isAnonymous());
215         // Ignore @fileoverview visibility when computing the effective visibility
216         // for properties created by goog.provide.
217         //
218         // ProcessClosurePrimitives rewrites goog.provide()s as object literal
219         // declarations, but the exact form depends on the ordering of the
220         // input files. If goog.provide('a.b.c') occurs in the inputs before
221         // goog.provide('a'), it is rewritten like
222         //
223         // var a={};a.b={}a.b.c={};
224         //
225         // If the file containing goog.provide('a.b.c') also declares
226         // a @fileoverview visibility, it must not apply to b, as this would make
227         // every a.b.* namespace effectively package-private.
228         return (raw != Visibility.INHERITED
229             || fileOverviewVisibility == null
230             || createdFromGoogProvide)
231             ? raw
232             : fileOverviewVisibility;
233       }
