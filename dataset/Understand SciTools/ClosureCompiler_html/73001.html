200       /**
201        * If the client adds new nodes to the AST, scan these new nodes
202        * to see if they've added any references to the global namespace.
203        * @param newNodes New nodes to check.
204        */
205       void scanNewNodes(Set<AstChange> newNodes) {
206         BuildGlobalNamespace builder = new BuildGlobalNamespace();
207     
208         for (AstChange info : newNodes) {
209           if (!info.node.isQualifiedName() && !NodeUtil.isObjectLitKey(info.node)) {
210             continue;
211           }
212           scanFromNode(builder, info.module, info.scope, info.node);
213         }
214       }
215     
216       private void scanFromNode(
217         BuildGlobalNamespace builder, JSModule module, Scope scope, Node n) {
218         // Check affected parent nodes first.
219         if (n.isName() || n.isGetProp()) {
220           scanFromNode(builder, module, scope, n.getParent());
221         }
222         builder.collect(module, scope, n);
223       }

225       /**
226        * Builds the namespace lazily.
227        */
228       private void process() {
229         if (externsRoot != null) {
230           inExterns = true;
231           NodeTraversal.traverseEs6(compiler, externsRoot, new BuildGlobalNamespace());
232         }
233         inExterns = false;
234     
235         NodeTraversal.traverseEs6(compiler, root, new BuildGlobalNamespace());
236         generated = true;
237       }

239       /**
240        * Determines whether a name reference in a particular scope is a global name
241        * reference.
242        *
243        * @param name A variable or property name (e.g. "a" or "a.b.c.d")
244        * @param s The scope in which the name is referenced
245        * @return Whether the name reference is a global name reference
246        */
247       private boolean isGlobalNameReference(String name, Scope s) {
248         String topVarName = getTopVarName(name);
249         return isGlobalVarReference(topVarName, s);
250       }

252       /**
253        * Gets the top variable name from a possibly namespaced name.
254        *
255        * @param name A variable or qualified property name (e.g. "a" or "a.b.c.d")
256        * @return The top variable name (e.g. "a")
257        */
258       private static String getTopVarName(String name) {
259         int firstDotIndex = name.indexOf('.');
260         return firstDotIndex == -1 ? name : name.substring(0, firstDotIndex);
261       }
262     
263       /**
264        * Determines whether a variable name reference in a particular scope is a
265        * global variable reference.
266        *
267        * @param name A variable name (e.g. "a")
268        * @param s The scope in which the name is referenced
269        * @return Whether the name reference is a global variable reference
270        */
271       private boolean isGlobalVarReference(String name, Scope s) {
272         Var v = s.getVar(name);
273         if (v == null && externsScope != null) {
274           v = externsScope.getVar(name);
275         }
276         return v != null && !v.isLocal();
277       }

281       /**
282        * Builds a tree representation of the global namespace. Omits prototypes.
283        */
284       private class BuildGlobalNamespace implements NodeTraversal.Callback {
285     
286         BuildGlobalNamespace() {
287         }
288     
289         @Override
290         public void visit(NodeTraversal t, Node n, Node parent) {}
291     
292         /** Collect the references in pre-order. */
293         @Override
294         public boolean shouldTraverse(NodeTraversal t, Node n, Node parent) {
295           // If we are traversing the externs, then we save a pointer to the scope
296           // generated by them, so that we can do lookups in it later.
297           if (externsRoot != null && n == externsRoot) {
298             externsScope = t.getScope();
299           }
300     
301           collect(t.getModule(), t.getScope(), n);
302     
303           return true;
304         }
305     
306         public void collect(JSModule module, Scope scope, Node n) {
307           Node parent = n.getParent();
308     
309           String name;
310           boolean isSet = false;
311           Name.Type type = Name.Type.OTHER;
312           boolean isPropAssign = false;
313     
314           switch (n.getType()) {
315             case GETTER_DEF:
316             case SETTER_DEF:
317             case STRING_KEY:
318             case MEMBER_FUNCTION_DEF:
319               // This may be a key in an object literal declaration.
320               name = null;
321               if (parent != null && parent.isObjectLit()) {
322                 name = getNameForObjLitKey(n);
323               } else if (parent != null && parent.isClassMembers()) {
324                 name = getNameForClassMembers(n);
325               }
326               if (name == null) {
327                 return;
328               }
329               isSet = true;
330               switch (n.getType()) {
331                 case MEMBER_FUNCTION_DEF:
332                 case STRING_KEY:
333                   type = getValueType(n.getFirstChild());
334                   break;
335                 case GETTER_DEF:
336                   type = Name.Type.GET;
337                   break;
338                 case SETTER_DEF:
339                   type = Name.Type.SET;
340                   break;
341                 default:
342                   throw new IllegalStateException("unexpected:" + n);
343               }
344               break;
345             case NAME:
346               // This may be a variable get or set.
347               if (parent != null) {
348                 switch (parent.getType()) {
349                   case VAR:
350                   case LET:
351                   case CONST:
352                     isSet = true;
353                     Node rvalue = n.getFirstChild();
354                     type = rvalue == null ? Name.Type.OTHER : getValueType(rvalue);
355                     break;
356                   case ASSIGN:
357                     if (parent.getFirstChild() == n) {
358                       isSet = true;
359                       type = getValueType(n.getNext());
360                     }
361                     break;
362                   case GETPROP:
363                     return;
364                   case FUNCTION:
365                     Node grandparent = parent.getParent();
366                     if (grandparent == null || NodeUtil.isFunctionExpression(parent)) {
367                       return;
368                     }
369                     isSet = true;
370                     type = Name.Type.FUNCTION;
371                     break;
372                   case CATCH:
373                   case INC:
374                   case DEC:
375                     isSet = true;
376                     type = Name.Type.OTHER;
377                     break;
378                   case CLASS:
379                     isSet = true;
380                     type = Name.Type.CLASS;
381                     break;
382                   default:
383                     if (NodeUtil.isAssignmentOp(parent) &&
384                         parent.getFirstChild() == n) {
385                       isSet = true;
386                       type = Name.Type.OTHER;
387                     }
388                 }
389               }
390               name = n.getString();
391               break;
392             case GETPROP:
393               // This may be a namespaced name get or set.
394               if (parent != null) {
395                 switch (parent.getType()) {
396                   case ASSIGN:
397                     if (parent.getFirstChild() == n) {
398                       isSet = true;
399                       type = getValueType(n.getNext());
400                       isPropAssign = true;
401                     }
402                     break;
403                   case INC:
404                   case DEC:
405                     isSet = true;
406                     type = Name.Type.OTHER;
407                     break;
408                   case GETPROP:
409                     return;
410                   default:
411                     if (NodeUtil.isAssignmentOp(parent) &&
412                         parent.getFirstChild() == n) {
413                       isSet = true;
414                       type = Name.Type.OTHER;
415                     }
416                 }
417               }
418               name = n.getQualifiedName();
419               if (name == null) {
420                 return;
421               }
422               break;
423             default:
424               return;
425           }
426     
427           // We are only interested in global names.
428           if (!isGlobalNameReference(name, scope)) {
429             return;
430           }
431     
432           scope = scope.getClosestHoistScope();
433           if (isSet) {
434             if (scope.isGlobal()) {
435               handleSetFromGlobal(module, scope, n, parent, name, isPropAssign, type);
436             } else {
437               handleSetFromLocal(module, scope, n, parent, name);
438             }
439           } else {
440             handleGet(module, scope, n, parent, name);
441           }
442         }

444         /**
445          * Gets the fully qualified name corresponding to an object literal key,
446          * as long as it and its prefix property names are valid JavaScript
447          * identifiers. The object literal may be nested inside of other object
448          * literals.
449          *
450          * For example, if called with node {@code n} representing "z" in any of
451          * the following expressions, the result would be "w.x.y.z":
452          * <code> var w = {x: {y: {z: 0}}}; </code>
453          * <code> w.x = {y: {z: 0}}; </code>
454          * <code> w.x.y = {'a': 0, 'z': 0}; </code>
455          *
456          * @param n A child of an OBJLIT node
457          * @return The global name, or null if {@code n} doesn't correspond to the
458          *   key of an object literal that can be named
459          */
460         String getNameForObjLitKey(Node n) {
461           Node parent = n.getParent();
462           Preconditions.checkState(parent.isObjectLit());
463     
464           Node grandparent = parent.getParent();
465           if (grandparent == null) {
466             return null;
467           }
468     
469           Node greatGrandparent = grandparent.getParent();
470           String name;
471           switch (grandparent.getType()) {
472             case NAME:
473               // VAR
474               //   NAME (grandparent)
475               //     OBJLIT (parent)
476               //       STRING (n)
477               if (greatGrandparent == null || !NodeUtil.isNameDeclaration(greatGrandparent)) {
478                 return null;
479               }
480               name = grandparent.getString();
481               break;
482             case ASSIGN:
483               // ASSIGN (grandparent)
484               //   NAME|GETPROP
485               //   OBJLIT (parent)
486               //     STRING (n)
487               Node lvalue = grandparent.getFirstChild();
488               name = lvalue.getQualifiedName();
489               break;
490             case STRING_KEY:
491               // OBJLIT
492               //   STRING (grandparent)
493               //     OBJLIT (parent)
494               //       STRING (n)
495               if (greatGrandparent != null &&
496                   greatGrandparent.isObjectLit()) {
497                 name = getNameForObjLitKey(grandparent);
498               } else {
499                 return null;
500               }
501               break;
502             default:
503               return null;
504           }
505           if (name != null) {
506             String key = n.getString();
507             if (TokenStream.isJSIdentifier(key)) {
508               return name + '.' + key;
509             }
510           }
511           return null;
512         }

514         /**
515          * Gets the fully qualified name corresponding to an class member function,
516          * as long as it and its prefix property names are valid JavaScript
517          * identifiers.
518          *
519          * For example, if called with node {@code n} representing "y" in any of
520          * the following expressions, the result would be "x.y":
521          * <code> class x{y(){}}; </code>
522          * <code> var x = class{y(){}}; </code>
523          * <code> var x; x = class{y(){}}; </code>
524          *
525          * @param n A child of an CLASS_MEMBERS node
526          * @return The global name, or null if {@code n} doesn't correspond to
527          *   a class member function that can be named
528          */
529         String getNameForClassMembers(Node n) {
530           Node parent = n.getParent();
531           Preconditions.checkState(parent.isClassMembers());
532           String className = NodeUtil.getName(parent.getParent());
533           return className == null ? null : className + '.' + n.getString();
534         }

536         /**
537          * Gets the type of a value or simple expression.
538          *
539          * @param n An r-value in an assignment or variable declaration (not null)
540          * @return A {@link Name.Type}
541          */
542         Name.Type getValueType(Node n) {
543           // Shorthand assignment of extended object literal
544           if (n == null) {
545             return Name.Type.OTHER;
546           }
547           switch (n.getType()) {
548             case CLASS:
549               return Name.Type.CLASS;
550             case OBJECTLIT:
551               return Name.Type.OBJECTLIT;
552             case FUNCTION:
553               return Name.Type.FUNCTION;
554             case OR:
555               // Recurse on the second value. If the first value were an object
556               // literal or function, then the OR would be meaningless and the
557               // second value would be dead code. Assume that if the second value
558               // is an object literal or function, then the first value will also
559               // evaluate to one when it doesn't evaluate to false.
560               return getValueType(n.getLastChild());
561             case HOOK:
562               // The same line of reasoning used for the OR case applies here.
563               Node second = n.getSecondChild();
564               Name.Type t = getValueType(second);
565               if (t != Name.Type.OTHER) {
566                 return t;
567               }
568               Node third = second.getNext();
569               return getValueType(third);
570           }
571           return Name.Type.OTHER;
572         }

574         /**
575          * Updates our representation of the global namespace to reflect an
576          * assignment to a global name in global scope.
577          *
578          * @param module the current module
579          * @param scope the current scope
580          * @param n The node currently being visited
581          * @param parent {@code n}'s parent
582          * @param name The global name (e.g. "a" or "a.b.c.d")
583          * @param isPropAssign Whether this set corresponds to a property
584          *     assignment of the form <code>a.b.c = ...;</code>
585          * @param type The type of the value that the name is being assigned
586          */
587         void handleSetFromGlobal(JSModule module, Scope scope,
588             Node n, Node parent, String name,
589             boolean isPropAssign, Name.Type type) {
590           if (maybeHandlePrototypePrefix(module, scope, n, parent, name)) {
591             return;
592           }
593     
594           Name nameObj = getOrCreateName(name);
595           nameObj.type = type;
596     
597           Ref set = new Ref(module, scope, n, nameObj, Ref.Type.SET_FROM_GLOBAL,
598               currentPreOrderIndex++);
599           nameObj.addRef(set);
600     
601           if (isNestedAssign(parent)) {
602             // This assignment is both a set and a get that creates an alias.
603             Ref get = new Ref(module, scope, n, nameObj, Ref.Type.ALIASING_GET,
604                 currentPreOrderIndex++);
605             nameObj.addRef(get);
606             Ref.markTwins(set, get);
607           } else if (isTypeDeclaration(n)) {
608             // Names with a @constructor or @enum annotation are always collapsed
609             nameObj.setDeclaredType();
610           }
611         }

613         /**
614          * Determines whether a set operation is a constructor or enumeration
615          * or interface declaration. The set operation may either be an assignment
616          * to a name, a variable declaration, or an object literal key mapping.
617          *
618          * @param n The node that represents the name being set
619          * @return Whether the set operation is either a constructor or enum
620          *     declaration
621          */
622         private boolean isTypeDeclaration(Node n) {
623           Node valueNode = NodeUtil.getRValueOfLValue(n);
624           JSDocInfo info = NodeUtil.getBestJSDocInfo(n);
625           // Heed the annotations only if they're sensibly used.
626           return info != null && valueNode != null &&
627                  (info.isConstructor() && valueNode.isFunction() ||
628                   info.isInterface() && valueNode.isFunction() ||
629                   info.hasEnumParameterType() && valueNode.isObjectLit());
630         }

632         /**
633          * Updates our representation of the global namespace to reflect an
634          * assignment to a global name in a local scope.
635          *
636          * @param module The current module
637          * @param scope The current scope
638          * @param n The node currently being visited
639          * @param parent {@code n}'s parent
640          * @param name The global name (e.g. "a" or "a.b.c.d")
641          */
642         void handleSetFromLocal(JSModule module, Scope scope, Node n, Node parent,
643                                 String name) {
644           if (maybeHandlePrototypePrefix(module, scope, n, parent, name)) {
645             return;
646           }
647     
648           Name nameObj = getOrCreateName(name);
649           Ref set = new Ref(module, scope, n, nameObj,
650               Ref.Type.SET_FROM_LOCAL, currentPreOrderIndex++);
651           nameObj.addRef(set);
652     
653           if (isNestedAssign(parent)) {
654             // This assignment is both a set and a get that creates an alias.
655             Ref get = new Ref(module, scope, n, nameObj,
656                 Ref.Type.ALIASING_GET, currentPreOrderIndex++);
657             nameObj.addRef(get);
658             Ref.markTwins(set, get);
659           }
660         }

662         /**
663          * Updates our representation of the global namespace to reflect a read
664          * of a global name.
665          *
666          * @param module The current module
667          * @param scope The current scope
668          * @param n The node currently being visited
669          * @param parent {@code n}'s parent
670          * @param name The global name (e.g. "a" or "a.b.c.d")
671          */
672         void handleGet(JSModule module, Scope scope,
673             Node n, Node parent, String name) {
674           if (maybeHandlePrototypePrefix(module, scope, n, parent, name)) {
675             return;
676           }
677     
678           Ref.Type type = Ref.Type.DIRECT_GET;
679           if (parent != null) {
680             switch (parent.getType()) {
681               case EXPR_RESULT:
682               case IF:
683               case INSTANCEOF:
684               case TYPEOF:
685               case VOID:
686               case NOT:
687               case BITNOT:
688               case POS:
689               case NEG:
690                 break;
691               case CALL:
692                 if (n == parent.getFirstChild()) {
693                   // It is a call target
694                   type = Ref.Type.CALL_GET;
695                 } else if (isClassDefiningCall(parent)) {
696                   type = Ref.Type.DIRECT_GET;
697                 } else {
698                   type = Ref.Type.ALIASING_GET;
699                 }
700                 break;
701               case NEW:
702                 type = n == parent.getFirstChild() ? Ref.Type.DIRECT_GET : Ref.Type.ALIASING_GET;
703                 break;
704               case OR:
705               case AND:
706                 // This node is x or y in (x||y) or (x&&y). We only know that an
707                 // alias is not getting created for this name if the result is used
708                 // in a boolean context or assigned to the same name
709                 // (e.g. var a = a || {}).
710                 type = determineGetTypeForHookOrBooleanExpr(module, scope, parent, name);
711                 break;
712               case HOOK:
713                 if (n != parent.getFirstChild()) {
714                   // This node is y or z in (x?y:z). We only know that an alias is
715                   // not getting created for this name if the result is assigned to
716                   // the same name (e.g. var a = a ? a : {}).
717                   type = determineGetTypeForHookOrBooleanExpr(module, scope, parent, name);
718                 }
719                 break;
720               case DELPROP:
721                 type = Ref.Type.DELETE_PROP;
722                 break;
723               default:
724                 type = Ref.Type.ALIASING_GET;
725                 break;
726             }
727           }
728     
729           handleGet(module, scope, n, parent, name, type);
730         }

732         private boolean isClassDefiningCall(Node callNode) {
733           CodingConvention convention = compiler.getCodingConvention();
734           // Look for goog.inherits, goog.mixin
735           SubclassRelationship classes =
736               convention.getClassesDefinedByCall(callNode);
737           if (classes != null) {
738             return true;
739           }
740     
741           // Look for calls to goog.addSingletonGetter calls.
742           String className = convention.getSingletonGetterClassName(callNode);
743           return className != null;
744         }

746         /**
747          * Determines whether the result of a hook (x?y:z) or boolean expression
748          * (x||y) or (x&&y) is assigned to a specific global name.
749          *
750          * @param module The current module
751          * @param scope The current scope
752          * @param parent The parent of the current node in the traversal. This node
753          *     should already be known to be a HOOK, AND, or OR node.
754          * @param name A name that is already known to be global in the current
755          *     scope (e.g. "a" or "a.b.c.d")
756          * @return The expression's get type, either {@link Ref.Type#DIRECT_GET} or
757          *     {@link Ref.Type#ALIASING_GET}
758          */
759         Ref.Type determineGetTypeForHookOrBooleanExpr(
760             JSModule module, Scope scope, Node parent, String name) {
761           Node prev = parent;
762           for (Node anc : parent.getAncestors()) {
763             switch (anc.getType()) {
764               case INSTANCEOF:
765               case EXPR_RESULT:
766               case VAR:
767               case LET:
768               case CONST:
769               case IF:
770               case WHILE:
771               case FOR:
772               case TYPEOF:
773               case VOID:
774               case NOT:
775               case BITNOT:
776               case POS:
777               case NEG:
778                 return Ref.Type.DIRECT_GET;
779               case HOOK:
780                 if (anc.getFirstChild() == prev) {
781                   return Ref.Type.DIRECT_GET;
782                 }
783                 break;
784               case ASSIGN:
785                 if (!name.equals(anc.getFirstChild().getQualifiedName())) {
786                   return Ref.Type.ALIASING_GET;
787                 }
788                 break;
789               case NAME:  // a variable declaration
790                 if (!name.equals(anc.getString())) {
791                   return Ref.Type.ALIASING_GET;
792                 }
793                 break;
794               case CALL:
795                 if (anc.getFirstChild() != prev) {
796                   return Ref.Type.ALIASING_GET;
797                 }
798                 break;
799               case DELPROP:
800                 return Ref.Type.DELETE_PROP;
801             }
802             prev = anc;
803           }
804           return Ref.Type.ALIASING_GET;
805         }

808          * Updates our representation of the global namespace to reflect a read
809          * of a global name.
810          *
811          * @param module The current module
812          * @param scope The current scope
813          * @param n The node currently being visited
814          * @param parent {@code n}'s parent
815          * @param name The global name (e.g. "a" or "a.b.c.d")
816          * @param type The reference type
817          */
818         void handleGet(JSModule module, Scope scope, Node n, Node parent,
819             String name, Ref.Type type) {
820           Name nameObj = getOrCreateName(name);
821

823           nameObj.addRef(
824               new Ref(module, scope, n, nameObj, type, currentPreOrderIndex++));

827         /**
828          * Updates our representation of the global namespace to reflect a read
829          * of a global name's longest prefix before the "prototype" property if the
830          * name includes the "prototype" property. Does nothing otherwise.
831          *
832          * @param module The current module
833          * @param scope The current scope
834          * @param n The node currently being visited
835          * @param parent {@code n}'s parent
836          * @param name The global name (e.g. "a" or "a.b.c.d")
837          * @return Whether the name was handled
838          */
839         boolean maybeHandlePrototypePrefix(JSModule module, Scope scope,
840             Node n, Node parent, String name) {
841           // We use a string-based approach instead of inspecting the parse tree
842           // to avoid complexities with object literals, possibly nested, beneath
843           // assignments.
844     
845           int numLevelsToRemove;
846           String prefix;
847           if (name.endsWith(".prototype")) {
848             numLevelsToRemove = 1;
849             prefix = name.substring(0, name.length() - 10);
850           } else {
851             int i = name.indexOf(".prototype.");
852             if (i == -1) {
853               return false;
854             }
855             prefix = name.substring(0, i);
856             numLevelsToRemove = 2;
857             i = name.indexOf('.', i + 11);
858             while (i >= 0) {
859               numLevelsToRemove++;
860               i = name.indexOf('.', i + 1);
861             }
862           }
863     
864           if (parent != null && NodeUtil.isObjectLitKey(n)) {
865             // Object literal keys have no prefix that's referenced directly per
866             // key, so we're done.
867             return true;
868           }
869     
870           for (int i = 0; i < numLevelsToRemove; i++) {
871             parent = n;
872             n = n.getFirstChild();
873           }
874     
875           handleGet(module, scope, n, parent, prefix, Ref.Type.PROTOTYPE_GET);
876           return true;
877         }

880          * Determines whether an assignment is nested (i.e. whether its return
881          * value is used).
882          *
883          * @param parent The parent of the current traversal node (not null)
884          * @return Whether it appears that the return value of the assignment is
885          *     used
886          */
887         boolean isNestedAssign(Node parent) {
888           return parent.isAssign() &&
889                  !parent.getParent().isExprResult();

892         /**
893          * Gets a {@link Name} instance for a global name. Creates it if necessary,
894          * as well as instances for any of its prefixes that are not yet defined.
895          *
896          * @param name A global name (e.g. "a", "a.b.c.d")
897          * @return The {@link Name} instance for {@code name}
898          */
899         Name getOrCreateName(String name) {
900           Name node = nameMap.get(name);
901           if (node == null) {
902             int i = name.lastIndexOf('.');
903             if (i >= 0) {
904               String parentName = name.substring(0, i);
905               Name parent = getOrCreateName(parentName);
906               node = parent.addProperty(name.substring(i + 1), inExterns);
907             } else {
908               node = new Name(name, null, inExterns);
909               globalNames.add(node);
910             }
911             nameMap.put(name, node);
912           }
913           return node;
914         }

929           FUNCTION,

937         List<Name> props;

1003         void addRef(Ref ref) {
1004           addRefInternal(ref);
1005           switch (ref.type) {
1006             case SET_FROM_GLOBAL:
1007               if (declaration == null) {
1008                 declaration = ref;
1009                 docInfo = getDocInfoForDeclaration(ref);
1010               }
1011               globalSets++;
1012               break;
1013             case SET_FROM_LOCAL:
1014               localSets++;
1015               JSDocInfo info = ref.getNode() == null ? null :
1016                   NodeUtil.getBestJSDocInfo(ref.getNode());
1017               if (info != null && info.isNoCollapse()) {
1018                 localSetsWithNoCollapse++;
1019               }
1020               break;
1021             case PROTOTYPE_GET:
1022             case DIRECT_GET:
1023               Node node = ref.getNode();
1024               if (node != null && node.isGetProp() && node.getParent().isExprResult()) {
1025                 docInfo = node.getJSDocInfo();
1026               }
1027               totalGets++;
1028               break;
1029             case ALIASING_GET:
1030               aliasingGets++;
1031               totalGets++;
1032               break;
1033             case CALL_GET:
1034               callGets++;
1035               totalGets++;
1036               break;
1037             case DELETE_PROP:
1038               deleteProps++;
1039               break;
1040             default:
1041               throw new IllegalStateException();
1042           }
1043         }
1044     
1045         void removeRef(Ref ref) {
1046           if (refs != null && refs.remove(ref)) {
1047             if (ref == declaration) {
1048               declaration = null;
1049               if (refs != null) {
1050                 for (Ref maybeNewDecl : refs) {
1051                   if (maybeNewDecl.type == Ref.Type.SET_FROM_GLOBAL) {
1052                     declaration = maybeNewDecl;
1053                     break;
1054                   }
1055                 }
1056               }
1057             }
1058     
1059             JSDocInfo info;
1060             switch (ref.type) {
1061               case SET_FROM_GLOBAL:
1062                 globalSets--;
1063                 break;
1064               case SET_FROM_LOCAL:
1065                 localSets--;
1066                 info = ref.getNode() == null ? null :
1067                     NodeUtil.getBestJSDocInfo(ref.getNode());
1068                 if (info != null && info.isNoCollapse()) {
1069                   localSetsWithNoCollapse--;
1070                 }
1071                 break;
1072               case PROTOTYPE_GET:
1073               case DIRECT_GET:
1074                 totalGets--;
1075                 break;
1076               case ALIASING_GET:
1077                 aliasingGets--;
1078                 totalGets--;
1079                 break;
1080               case CALL_GET:
1081                 callGets--;
1082                 totalGets--;
1083                 break;
1084               case DELETE_PROP:
1085                 deleteProps--;
1086                 break;
1087               default:
1088                 throw new IllegalStateException();
1089             }
1090           }
1091         }

1104         boolean canEliminate() {
1105           if (!canCollapseUnannotatedChildNames() || totalGets > 0) {
1106             return false;
1107           }
1108     
1109           if (props != null) {
1110             for (Name n : props) {
1111               if (!n.canCollapse()) {
1112                 return false;
1113               }
1114             }

1119         boolean isSimpleStubDeclaration() {
1120           if (getRefs().size() == 1) {
1121             Ref ref = refs.get(0);
1122             if (ref.node.getParent() != null &&
1123                 ref.node.getParent().isExprResult()) {
1124               return true;
1125             }
1126           }
1127           return false;

1130         boolean isCollapsingExplicitlyDenied() {
1131           if (docInfo == null) {
1132             Ref ref = getDeclaration();
1133             if (ref != null) {
1134               docInfo = getDocInfoForDeclaration(ref);
1135             }
1136           }
1137     
1138           return docInfo != null && docInfo.isNoCollapse();
1139         }

1141         boolean isInlinableGlobalAlias() {
1142           // Only simple aliases with direct usage are inlinable.
1143           if (inExterns || globalSets != 1 || localSets != 0 || !canCollapse()) {
1144             return false;
1145           }
1146     
1147           // Only allow inlining of simple references.
1148           for (Ref ref : getRefs()) {
1149             switch (ref.type) {
1150               case SET_FROM_GLOBAL:
1151                 // Expect one global set
1152                 continue;
1153               case SET_FROM_LOCAL:
1154                 throw new IllegalStateException();
1155               case ALIASING_GET:
1156               case DIRECT_GET:
1157               case PROTOTYPE_GET:
1158               case CALL_GET:
1159                 continue;
1160               case DELETE_PROP:
1161                 return false;
1162               default:
1163                 throw new IllegalStateException();
1164             }
1165           }
1166           return true;
1167         }

1169         boolean canCollapse() {
1170           return !inExterns && !isGetOrSetDefinition() &&
1171               !isCollapsingExplicitlyDenied() &&
1172               (declaredType ||
1173               (parent == null || parent.canCollapseUnannotatedChildNames()) &&
1174               (globalSets > 0 || localSets > 0) && localSetsWithNoCollapse == 0 &&
1175               deleteProps == 0);
1176         }
1177     
1178         boolean isGetOrSetDefinition() {
1179           return this.type == Type.GET || this.type == Type.SET;
1180         }
1181     
1182         boolean canCollapseUnannotatedChildNames() {
1183           if (type == Type.OTHER || isGetOrSetDefinition()
1184               || globalSets != 1 || localSets != 0 || deleteProps != 0) {
1185             return false;
1186           }
1187     
1188           // Don't try to collapse if the one global set is a twin reference.
1189           // We could theoretically handle this case in CollapseProperties, but
1190           // it's probably not worth the effort.
1191           Preconditions.checkNotNull(declaration);
1192           if (declaration.getTwin() != null) {
1193             return false;
1194           }
1195     
1196           if (isCollapsingExplicitlyDenied()) {
1197             return false;
1198           }
1199     
1200           if (declaredType) {
1201             return true;
1202           }
1203     
1204           // If this is a key of an aliased object literal, then it will be aliased
1205           // later. So we won't be able to collapse its properties.
1206           if (parent != null && parent.shouldKeepKeys()) {
1207             return false;
1208           }
1209     
1210           // If this is aliased, then its properties can't be collapsed either.
1211           if (aliasingGets > 0) {
1212             return false;
1213           }
1214     
1215           return (parent == null || parent.canCollapseUnannotatedChildNames());
1216         }
1217     
1218         /** Whether this is an object literal that needs to keep its keys. */
1219         boolean shouldKeepKeys() {
1220           return type == Type.OBJECTLIT &&
1221               (aliasingGets > 0 || isCollapsingExplicitlyDenied());
1222         }
1223     
1224         boolean needsToBeStubbed() {
1225           return globalSets == 0 && localSets > 0 && localSetsWithNoCollapse == 0 &&
1226               !isCollapsingExplicitlyDenied();
1227         }

1229         void setDeclaredType() {
1230           declaredType = true;
1231           for (Name ancestor = parent; ancestor != null;
1232                ancestor = ancestor.parent) {
1233             ancestor.isDeclared = true;
1234           }
1235         }

1238           return declaredType;
1239         }

1241         boolean isConstructor() {
1242           Node declNode = declaration.node;
1243           Node rvalueNode = NodeUtil.getRValueOfLValue(declNode);
1244           JSDocInfo jsdoc = NodeUtil.getBestJSDocInfo(declNode);
1245           return rvalueNode != null && rvalueNode.isFunction()
1246               && jsdoc != null && jsdoc.isConstructor();
1247         }

1283         private static JSDocInfo getDocInfoForDeclaration(Ref ref) {
1284           if (ref.node != null) {
1285             Node refParent = ref.node.getParent();
1286             if (refParent == null) {
1287               // May happen when inlineAliases removes refs from the AST.
1288               return null;

1291               case FUNCTION:
1292               case ASSIGN:
1293               case CLASS:
1294                 return refParent.getJSDocInfo();
1295               case VAR:

1298                 return ref.node == refParent.getFirstChild() ?
1299                     refParent.getJSDocInfo() : ref.node.getJSDocInfo();
1300               case OBJECTLIT:
1301                 return ref.node.getJSDocInfo();

1303           }
