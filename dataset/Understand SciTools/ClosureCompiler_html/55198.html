44       public void testExprExploitation() {
45         test("a = null; b = null; var c = b",
46              "var c = b = a = null");
47         test("a = null; b = null",
48              "b = a = null");
49         test("a = undefined; b = undefined",
50              "b = a = undefined");
51         test("a = 0; b = 0", "b=a=0");
52         test("a = 'foo'; b = 'foo'",
53              "b = a = \"foo\"");
54         test("a = c; b = c", "b=a=c");
55     
56         testSame("a = 0; b = 1");
57         testSame("a = \"foo\"; b = \"foox\"");
58     
59         test("a = null; a && b;", "(a = null)&&b");
60         test("a = null; a || b;", "(a = null)||b");
61     
62         test("a = null; a ? b : c;", "(a = null) ? b : c");
63     
64         test("a = null; this.foo = null;",
65              "this.foo = a = null");
66         test("function f(){ a = null; return null; }",
67              "function f(){return a = null}");
68     
69         test("a = true; if (a) { foo(); }",
70              "if (a = true) { foo() }");
71         test("a = true; if (a && a) { foo(); }",
72              "if ((a = true) && a) { foo() }");
73         test("a = false; if (a) { foo(); }",
74              "if (a = false) { foo() }");
75     
76         test("a = !0; if (a) { foo(); }",
77             "if (a = !0) { foo() }");
78         test("a = !0; if (a && a) { foo(); }",
79             "if ((a = !0) && a) { foo() }");
80         test("a = !1; if (a) { foo(); }",
81             "if (a = !1) { foo() }");
82     
83         testSame("a = this.foo; a();");
84         test("a = b; b = a;",
85              "b = a = b");
86         testSame("a = b; a.c = a");
87         test("this.foo = null; this.bar = null;",
88              "this.bar = this.foo = null");
89         test("this.foo = null; this.bar = null; this.baz = this.bar",
90              "this.baz = this.bar = this.foo = null");
91         test("this.foo = null; a = null;",
92              "a = this.foo = null");
93         test("this.foo = null; a = this.foo;",
94              "a = this.foo = null");
95         test("a.b.c=null; a=null;",
96              "a = a.b.c = null");
97         testSame("a = null; a.b.c = null");
98         test("(a=b).c = null; this.b = null;",
99              "this.b = (a=b).c = null");
100         testSame("if(x) a = null; else b = a");
101       }

103       public void testNestedExprExploitation() {
104         test("this.foo = null; this.bar = null; this.baz = null;",
105              "this.baz = this.bar = this.foo = null");
106     
107         test("a = 3; this.foo = a; this.bar = a; this.baz = 3;",
108              "this.baz = this.bar = this.foo = a = 3");
109         test("a = 3; this.foo = a; this.bar = this.foo; this.baz = a;",
110              "this.baz = this.bar = this.foo = a = 3");
111         test("a = 3; this.foo = a; this.bar = 3; this.baz = this.foo;",
112              "this.baz = this.bar = this.foo = a = 3");
113         test("a = 3; this.foo = a; a = 3; this.bar = 3; " +
114              "a = 3; this.baz = this.foo;",
115              "this.baz = a = this.bar = a = this.foo = a = 3");
116     
117         test("a = 4; this.foo = a; a = 3; this.bar = 3; " +
118              "a = 3; this.baz = this.foo;",
119              "this.foo = a = 4; a = this.bar = a = 3; this.baz = this.foo");
120         test("a = 3; this.foo = a; a = 4; this.bar = 3; " +
121              "a = 3; this.baz = this.foo;",
122              "this.foo = a = 3; a = 4; a = this.bar = 3; this.baz = this.foo");
123         test("a = 3; this.foo = a; a = 3; this.bar = 3; " +
124              "a = 4; this.baz = this.foo;",
125              "this.bar = a = this.foo = a = 3; a = 4; this.baz = this.foo");
126       }

128       public void testBug1840071() {
129         // Some external properties are implemented as setters. Let's
130         // make sure that we don't collapse them inappropriately.
131         test("a.b = a.x; if (a.x) {}", "if (a.b = a.x) {}");
132         testSame("a.b = a.x; if (a.b) {}");
133         test("a.b = a.c = a.x; if (a.x) {}", "if (a.b = a.c = a.x) {}");
134         testSame("a.b = a.c = a.x; if (a.c) {}");
135         testSame("a.b = a.c = a.x; if (a.b) {}");
136       }
