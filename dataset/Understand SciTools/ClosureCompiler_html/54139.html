21     /**
22      * Implements the ECMAScript 5
23      * <a href="http://es5.github.com/#Canonicalize">Canonicalize</a> operation
24      * used to specify how case-insensitive regular expressions match.
25      *
26      * <p>
27      * From section <a href="http://es5.github.com/#x15.10.2.9">15.10.2.9</a>,
28      * <blockquote>
29      * The abstract operation Canonicalize takes a character parameter ch and
30      * performs the following steps:
31      * <ul>
32      *   <li>If IgnoreCase is false, return ch.
33      *   <li>Let u be ch converted to upper case as if by calling the standard
34      *   built-in method {@code String.prototype.toUpperCase} on the one-character
35      *   String ch.
36      *   <li>If u does not consist of a single character, return ch.
37      *   <li>Let cu be u's character.
38      *   <li>If ch's code unit value is greater than or equal to decimal 128 and
39      *   cu's code unit value is less than decimal 128, then return ch.
40      *   <li>Return cu.
41      * </ul>
42      * </blockquote>
43      *
44      * @author Mike Samuel &lt;mikesamuel@gmail.com&gt;
45      */
46     public final class CaseCanonicalize {

188       /**
189        * Returns the case canonical version of the given string.
190        */
191       public static String caseCanonicalize(String s) {
192         for (int i = 0, n = s.length(); i < n; ++i) {
193           char ch = s.charAt(i);
194           char cu = caseCanonicalize(ch);
195           if (cu != ch) {
196             StringBuilder sb = new StringBuilder(s);
197             sb.setCharAt(i, cu);
198             while (++i < n) {
199               sb.setCharAt(i, caseCanonicalize(s.charAt(i)));
200             }
201             return sb.toString();
202           }
203         }
204         return s;
205       }

207       /**
208        * Returns the case canonical version of the given code-unit.  ECMAScript 5
209        * explicitly says that code-units are to be treated as their code-point
210        * equivalent, even surrogates.
211        */
212       public static char caseCanonicalize(char ch) {
213         if (ch < 0x80) {  // Normal case.
214           return ('A' <= ch && ch <= 'Z') ? (char) (ch | 32) : ch;
215         }
216         // Non-ASCII case.
217         if (CASE_SENSITIVE.contains(ch)) {
218           for (DeltaSet ds : CANON_DELTA_SETS) {
219             if (ds.codeUnits.contains(ch)) {
220               return (char) (ch - ds.delta);
221             }
222           }
223         }
224         return ch;
225       }

227       /**
228        * Given a character range that may include case sensitive code-units,
229        * such as {@code [0-9B-M]}, returns the character range that includes all
230        * the code-units in the input and those that are case-insensitively
231        * equivalent to a code-unit in the input.
232        */
233       public static CharRanges expandToAllMatched(CharRanges ranges) {
234         CharRanges caseSensitive = ranges.intersection(CASE_SENSITIVE);
235         if (caseSensitive.isEmpty()) { return ranges; }
236         CharRanges expanded = CharRanges.EMPTY;
237         for (DeltaSet ds : DELTA_SETS) {
238           expanded = expanded.union(
239               caseSensitive.intersection(ds.codeUnits).shift(-ds.delta));
240         }
241         return ranges.union(expanded);
242       }

245       private static final CharRanges UCASE_ASCII_LETTERS
246           = CharRanges.inclusive('A', 'Z');
247     
248       /**
249        * Given a character range that may include case sensitive code-units,
250        * such as {@code [0-9B-M]}, returns the character range that includes
251        * the minimal set of code units such that for every code unit in the
252        * input there is a case-sensitively equivalent canonical code unit in the
253        * output.
254        */
255       public static CharRanges reduceToMinimum(CharRanges ranges) {
256         CharRanges caseSensitive = ranges.intersection(CASE_SENSITIVE);
257         if (caseSensitive.isEmpty()) { return ranges; }
258         CharRanges expanded = CharRanges.EMPTY;
259         for (DeltaSet ds : CANON_DELTA_SETS) {
260           expanded = expanded.union(
261               caseSensitive.intersection(ds.codeUnits).shift(-ds.delta));
262         }
263         // Letters a-z gzip better than uppercase A-Z since JavaScript keywords
264         // are lower-case, so, even though the definition of Canonicalize is
265         // based on String.prototype.toUpperCase, we use lowercase ASCII characters
266         // in the minimal form.
267         expanded = expanded.difference(UCASE_ASCII_LETTERS).union(
268             expanded.intersection(UCASE_ASCII_LETTERS).shift(32));
269         return ranges.difference(caseSensitive).union(expanded);
270       }
