40     package com.google.javascript.rhino.jstype;
41     
42     import java.util.ArrayDeque;
43     
44     /**
45      * Uses a TemplateTypeMap to replace TemplateTypes with their associated JSType
46      * values.
47      *
48      * @author izaakr@google.com (Izaak Rubin)
49      */
50     public class TemplateTypeMapReplacer extends ModificationVisitor {
51       private final TemplateTypeMap replacements;
52       private ArrayDeque<TemplateType> visitedTypes;
53     
54       public TemplateTypeMapReplacer(
55           JSTypeRegistry registry, TemplateTypeMap replacements) {
56         super(registry, false);
57         this.replacements = replacements;
58         this.visitedTypes = new ArrayDeque<>();
59       }
60     
61       @Override
62       public JSType caseTemplateType(TemplateType type) {
63         if (replacements.hasTemplateKey(type)) {
64           if (hasVisitedType(type) || !replacements.hasTemplateType(type)) {
65             // If we have already encountered this TemplateType during replacement
66             // (i.e. there is a reference loop), or there is no JSType substitution
67             // for the TemplateType, return the TemplateType type itself.
68             return type;
69           } else {
70             JSType replacement = replacements.getUnresolvedOriginalTemplateType(type);
71             if (isRecursive(type, replacement)) {
72               // Recursive templated type definition (e.g. T resolved to Foo<T>).
73               return type;
74             }
75     
76             visitedTypes.push(type);
77             JSType visitedReplacement = replacement.visit(this);
78             visitedTypes.pop();
79     
80             return visitedReplacement;
81           }
82         } else {
83           return type;
84         }
85       }
86     
87       /**
88        * Returns whether the replacement type is a templatized type which contains the current type.
89        * e.g. current type T is being replaced with Foo<T>
90        */
91       private boolean isRecursive(TemplateType currentType, JSType replacementType) {
92         TemplatizedType replacementTemplatizedType =
93             replacementType.restrictByNotNullOrUndefined().toMaybeTemplatizedType();
94         if (replacementTemplatizedType == null) {
95           return false;
96         }
97     
98         Iterable<JSType> replacementTemplateTypes = replacementTemplatizedType.getTemplateTypes();
99         for (JSType replacementTemplateType : replacementTemplateTypes) {
100           if (replacementTemplateType.isTemplateType()
101               && isSameType(currentType, replacementTemplateType.toMaybeTemplateType())) {
102             return true;
103           }
104         }
105     
106         return false;
107       }
108     
109       private boolean isSameType(TemplateType currentType, TemplateType replacementType) {
110         return currentType == replacementType
111             || currentType == replacements.getUnresolvedOriginalTemplateType(replacementType);
112       }
113     
114       /**
115        * Checks if the specified type has already been visited during the Visitor's
116        * traversal of a JSType.
117        */
118       private boolean hasVisitedType(TemplateType type) {
119         for (TemplateType visitedType : visitedTypes) {
120           if (visitedType == type) {
121             return true;
122           }
123         }
124         return false;
125       }
126     }
