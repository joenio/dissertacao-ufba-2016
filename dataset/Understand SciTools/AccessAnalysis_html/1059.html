30     			page = window.getActivePage();
31     		}
32     		
33         	IConfigurationElement[] config = Platform.getExtensionRegistry()
34     				.getConfigurationElementsFor(AccessAnalysisResultsConsumer.EXTENSION_POINT_ID);
35         	
36         	Collection<AccessAnalysisResultsConsumer> consumers = new ArrayList<AccessAnalysisResultsConsumer>(config.length);
37     		for (IConfigurationElement e : config) {
38     			final Object extension = e.createExecutableExtension("class");
39     			if (extension instanceof AccessAnalysisResultsConsumer) {
40     				AccessAnalysisResultsConsumer consumer = (AccessAnalysisResultsConsumer) extension;
41     				if (page != null) {
42     					consumer.setWorkbenchPage(page);
43     				}
44     				consumers.add(consumer);
45     			}
46     		}
47     		
48     		return consumers;
49         }

51     	public static void showErrorMessage(String message, Exception exception) {
52     		final StringBuilder errorMessage = new StringBuilder();

54     			errorMessage.append(message);
55     		}
56     		if (message != null && exception != null) {

59     		if (exception != null) {

61     		}
62     		if (message != null && exception != null) {
63     			errorMessage.append(")");
64     		}
65     		
66     		Display.getDefault().asyncExec(new Runnable() {
67     			
68     			@Override
