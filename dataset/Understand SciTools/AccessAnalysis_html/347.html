26     /**
27      * Runs the analysis.
28      */
29     class AccessAnalysisJob extends WorkspaceJob
30     {
31         private List<IJavaProject> _projects;
32         private Analysis _analysis;
33         private Collection<AccessAnalysisResultsConsumer> _resultsConsumer;
34         
35         /**
36          * Initializes an object of AnalysisJob. 
37          * 
38          * @param IJavaProject projects The projects to be analyzed
39          * @param AccessAnalysisView view The view that shows the results
40          */
41         AccessAnalysisJob(List<IJavaProject> projects, Collection<AccessAnalysisResultsConsumer> resultsConsumer)
42         {
43             super("AccessAnalysis");
44             
45             _projects = projects;
46             _analysis = AnalysisFactory.analyzer(_projects);
47             _resultsConsumer = resultsConsumer;
48             
49             setUser(true);
50             try {
51                 setRule(createModifyRule(_analysis.filesToBeLocked()));
52             }
53             catch (AnalysisException exc) {
54             	RunAnalysisActionActivator.showErrorMessage("Error while listing files!", exc);
55             }
56         }
57         
58         @Override
59         public IStatus runInWorkspace(IProgressMonitor monitor)
60         {
61             try
62             {
63             	clearResultsConsumer();
64                 monitor.beginTask("", 1000);
65                 _analysis.run(new SubProgressMonitor(monitor, 990, SubProgressMonitor.PREPEND_MAIN_LABEL_TO_SUBTASK));
66                 if (! monitor.isCanceled())
67                 {
68                     monitor.subTask("Show results");
69                     fillResultsConsumer(_analysis.getResults());
70                     monitor.worked(10);
71                 }
72             }
73             catch (AnalysisException exc)
74             {
75             	RunAnalysisActionActivator.showErrorMessage(null, exc);
76                 return Status.CANCEL_STATUS;
77             }
78             finally 
79             {
80                 monitor.done();
81             }
82             return Status.OK_STATUS;
83         }
84         
85         private void clearResultsConsumer() {
86     		for (AccessAnalysisResultsConsumer resultsConsumer : _resultsConsumer) {
87     			resultsConsumer.clear();
88     		}
89     	}
90         
91         private void fillResultsConsumer(List<Result> results) {
92     		for (AccessAnalysisResultsConsumer resultsConsumer : _resultsConsumer) {
93     			resultsConsumer.takeResults(results);
94     		}
95     	}
96         
97     	private ISchedulingRule createModifyRule(IFile[] files) 
98         {
99             ISchedulingRule combinedRule = null;
100             IResourceRuleFactory ruleFactory = ResourcesPlugin.getWorkspace().getRuleFactory();
101             for (int i = 0; i < files.length; i++) 
102             {
103                 ISchedulingRule rule = ruleFactory.modifyRule(files[i]);
104                 combinedRule = MultiRule.combine(rule, combinedRule);
105             }
106             return combinedRule;
107         }
