26     /**
27      * Determines the types of expressions.
28      */
29     public class ExpressionTypeDeterminator
30     {
31         private final Stack<ITypeBinding> _currentTypes;
32     
33         private CompilationUnit _currentCompilationUnit;
34         
35         /**
36          * Initializes an object of ExpressionTypeDeterminator. 
37          */
38         public ExpressionTypeDeterminator()
39         {
40             _currentTypes = new Stack<ITypeBinding>();
41         }
42         
43         /**
44          * Handles a CompilationUnit
45          * 
46          * @param CompilationUnit node
47          */
48         public void handle(CompilationUnit node)
49         {
50             _currentCompilationUnit = node;
51         }
52         
53         /**
54          * Handles a TypeDeclaration
55          * 
56          * @param TypeDeclaration node
57          */
58         public void handle(AbstractTypeDeclaration node)
59         {
60             ITypeBinding typeBinding = node.resolveBinding();
61             setCurrentType(typeBinding);
62         }
63         
64         /**
65          * Handles the end of a TypeDeclaration
66          * 
67          * @param TypeDeclaration node
68          */
69         public void handleEnd(AbstractTypeDeclaration node)
70         {
71             unsetCurrentType();
72         }
73         
74         /**
75          * Determines the type of method invocation.
76          * 
77          * Returns not the return type of the invoked method but the expression where the method is called at. 
78          * 
79          * @param MethodInvocation node The method invocation to be examined.
80          * @return ITypeBinding The type of the expression
81          * @throws JavaModelException
82          */
83         public ITypeBinding getExpressionType(MethodInvocation node) throws JavaModelException
84         {
85             Expression expression = node.getExpression();
86             if (expression != null)
87             {
88                 ITypeBinding binding = expression.resolveTypeBinding();
89                 if (binding == null)
90                 {
91                     binding = node.resolveMethodBinding().getDeclaringClass();
92                 }
93                 while (binding.isTypeVariable())
94                 {
95                     ITypeBinding[] bounds = binding.getTypeBounds();
96                     if (bounds.length == 0)
97                     {
98                         return null;
99                     }
100                     binding = bounds[0];
101                 }
102                 return binding.getTypeDeclaration();
103             }
104             
105             IMethodBinding method = node.resolveMethodBinding();
106             ITypeBinding enclosingType = getCurrentType();
107             
108             while (enclosingType != null)
109             {
110                 if (Hierarchy.findOverriddenMethodInHierarchy(enclosingType, method) != null)
111                 {
112                     return enclosingType.getTypeDeclaration();
113                 }
114                 enclosingType = enclosingType.getDeclaringClass();
115             }
116             
117             if (method.getJavaElement() != null && Flags.isStatic(((IMethod) method.getJavaElement()).getFlags()))
118             {
119                 return getTypeOfStaticImportedMethods(method);
120             }
121     
122             return null;
123         }
124         
125         /**
126          * Determines the type of field access.
127          * 
128          * Returns not the type of the field but the expression where the field is accessed at. 
129          * 
130          * @param FieldAccess node The field access to be examined.
131          * @return ITypeBinding The type of the expression
132          * @throws JavaModelException
133          */
134         public ITypeBinding getExpressionType(FieldAccess node) throws JavaModelException
135         {
136             Expression expression = node.getExpression();
137             if (expression != null)
138             {
139                 return expression.resolveTypeBinding().getTypeDeclaration();
140             }
141             
142             IVariableBinding field = node.resolveFieldBinding();
143             ITypeBinding enclosingType = getCurrentType();
144             
145             while (enclosingType != null)
146             {
147                 if (Hierarchy.findFieldInHierarchy(enclosingType, field) != null)
148                 {
149                     return enclosingType.getTypeDeclaration();
150                 }
151                 enclosingType = enclosingType.getDeclaringClass();
152             }
153     
154             return null;
155         }
156         
157         private ITypeBinding getTypeOfStaticImportedMethods(IMethodBinding method)
158         {
159             ITypeBinding type = method.getDeclaringClass();
160             
161             for (ImportDeclaration importDeclaration : getStaticImports())
162             {
163                 IBinding importedBinding = importDeclaration.resolveBinding();
164                 
165                 if (importedBinding.getKind() == IBinding.TYPE)
166                 {
167                     ITypeBinding importedType = (ITypeBinding) importedBinding;
168                     if (importedType.isSubTypeCompatible(type))
169                     {
170                         return importedType.getTypeDeclaration();
171                     }
172                 }
173                 else if (importedBinding.getKind() == IBinding.METHOD)
174                 {
175                     IMethodBinding importedMethodBinding = (IMethodBinding) importedBinding;
176                     if (method.equals(importedMethodBinding))
177                     {
178                         return type.getTypeDeclaration();
179                     }
180                 }
181             }
182             
183             return null;
184         }
185         
186         private void setCurrentType(ITypeBinding typeBinding)
187         {
188             _currentTypes.push(typeBinding);
189         }
190         
191         private void unsetCurrentType()
192         {
193             if (! _currentTypes.empty())
194             {
195                 _currentTypes.pop();
196             }
197         }
198         
199         private ITypeBinding getCurrentType()
200         {
201             if (_currentTypes.empty())
202             {
203                 return null;
204             }
205             
206             return _currentTypes.peek();
207         }
208         
209         private Iterable<ImportDeclaration> getStaticImports()
210         {
211             Collection<ImportDeclaration> staticImports = new LinkedList<ImportDeclaration>();
212             
213             for (Object o : _currentCompilationUnit.imports())
214             {
215                 ImportDeclaration importDeclaration = (ImportDeclaration) o;
216                 if (importDeclaration.isStatic())
217                 {
218                     staticImports.add(importDeclaration);
219                 }
220             }
221             
222             return staticImports;
223         }
