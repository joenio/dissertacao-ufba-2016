18     /**
19      * Determines the lowest access mode needed for the invocation of a method.
20      * 
21      * @author christian
22      * 
23      */
24     public class InvokedMethodAccessDeterminator
25     {
26         private final IMethodBinding _invokingMethodBinding;
27     
28         /**
29          * Initializes an object of InvokedMethodAccessDeterminator.
30          * 
31          * @param IMethodBinding
32          *            invokingMethod The method that invokes other methods
33          */
34         public InvokedMethodAccessDeterminator(IMethodBinding invokingMethod)
35         {
36             _invokingMethodBinding = invokingMethod;
37         }
38     
39         /**
40          * Determines the minimal access of an invoked method needed for the
41          * invocation Assumes that the invoking method is the current one.
42          * 
43          * @param IMethodBinding
44          *            binding The binding of the invoked method
45          * @param ITypeBinding
46          *            expressionType The binding of the type of the expression the
47          *            method is called on
48          * @return The minimal access needed for the invocation
49          */
50         public Access determineAccess(IMethodBinding binding, ITypeBinding expressionType) throws JavaModelException
51         {
52             if (binding.getDeclaringClass().isInterface())
53             {
54                 return Access.PUBLIC;
55             }
56     
57             if (canBeNoUse(binding, expressionType))
58             {
59                 return Access.NO_USE;
60             }
61     
62             if (canBePrivate(binding, expressionType))
63             {
64                 return Access.PRIVATE;
65             }
66     
67             if (canBeDefault(binding, expressionType))
68             {
69                 return Access.DEFAULT;
70             }
71     
72             if (canBeProtected(binding, expressionType))
73             {
74                 return Access.PROTECTED;
75             }
76     
77             return Access.PUBLIC;
78         }
79     
80         boolean canBeNoUse(IMethodBinding binding, ITypeBinding expressionType)
81         {
82             return binding.equals(getCurrentMethodBinding()) && (expressionType == null || expressionType.equals(getCurrentTypeBinding()));
83         }
84     
85         private boolean canBePrivate(IMethodBinding binding, ITypeBinding expressionType) throws JavaModelException
86         {
87             if (Flags.isAbstract(((IMethod) binding.getJavaElement()).getFlags()))
88             {
89                 return false;
90             }
91     
92             ITypeBinding currentTopLevelTyp = Nesting.getTopLevelType(getCurrentTypeBinding());
93             
94             return Nesting.getTopLevelType(binding.getDeclaringClass()).equals(currentTopLevelTyp) && (expressionType == null || Nesting.getTopLevelType(expressionType).equals(currentTopLevelTyp));
95         }
96     
97         private boolean canBeDefault(IMethodBinding binding, ITypeBinding expressionType) throws JavaModelException
98         {
99             ITypeBinding declaringType = binding.getDeclaringClass();
100     
101             if (!declaringType.getPackage().equals(getCurrentPackageBinding()))
102             {
103                 return false;
104             }
105     
106             ITypeBinding intermediateType = null;
107             if (expressionType != null)
108             {
109                 intermediateType = expressionType.getTypeDeclaration();
110             }
111             if (intermediateType == null)
112             {
113                 intermediateType = getCurrentTypeBinding();
114             }
115     
116             while (intermediateType != null && !intermediateType.getName().equals("void") && !intermediateType.equals(declaringType))
117             {
118                 ITypeBinding typeBinding = intermediateType;
119                 
120                 while (typeBinding.getPackage() == null)
121                 {
122                     typeBinding = typeBinding.getDeclaringClass();
123                 }
124                 
125                 if (! typeBinding.getPackage().equals(getCurrentPackageBinding()))
126                 {
127                     return false;
128                 }
129     
130                 intermediateType = intermediateType.getSuperclass().getTypeDeclaration();
131             }
132     
133             return true;
134         }
135     
136         private boolean canBeProtected(IMethodBinding binding, ITypeBinding expressionType)
137         {
138             ITypeBinding declaringType = binding.getDeclaringClass();
139     
140             ITypeBinding currentType = getCurrentTypeBinding();
141             while (currentType != null)
142             {
143                 if (Hierarchy.isSuperType(declaringType, currentType) && 
144                     (expressionType == null || Flags.isStatic(binding.getModifiers()) || Hierarchy.isSuperType(currentType, expressionType)))
145                 {
146                     return true;
147                 }
148                 currentType = currentType.getDeclaringClass();
149             }
150     
151             if (declaringType.getPackage().equals(getCurrentPackageBinding()))
152             {
153                 return true;
154             }
155             
156             return false;
157         }
158     
159         private IMethodBinding getCurrentMethodBinding()
160         {
161             return _invokingMethodBinding;
162         }
163     
164         ITypeBinding getCurrentTypeBinding()
165         {
166             return getCurrentMethodBinding().getDeclaringClass().getTypeDeclaration();
167         }
168     
169         private IPackageBinding getCurrentPackageBinding()
170         {
171             return getCurrentTypeBinding().getPackage();
172         }
173     }
