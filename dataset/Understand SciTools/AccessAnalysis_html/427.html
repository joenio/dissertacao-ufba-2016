15     /**
16      * Creates a console and put out text to it.
17      * 
18      * Code based on the example at
19      * http://www.eclipsezone.com/eclipse/forums/t52910.html.
20      */
21     class ConsoleDisplayMgr
22     {
23         private String _Title = null;
24         private MessageConsole _MessageConsole = null;
25     
26         /**
27          * Initializes an object of ConsoleDisplayMgr.
28          * 
29          * @param String
30          *            messageTitle The name of the console
31          */
32         ConsoleDisplayMgr(String messageTitle)
33         {
34             _Title = messageTitle;
35         }
36     
37         /**
38          * Prints one line of the default message type.
39          * 
40          * @param String
41          *            msg The text to be printed
42          * @param LoggerMessageType
43          *            type The type of message to be printed
44          */
45         void println(String msg, LoggerMessageType msgType)
46         {
47             if (msg == null)
48                 return;
49     
50             /*
51              * if console-view in Java-perspective is not active, then show it and
52              * then display the message in the console attached to it
53              */
54             if (!displayConsoleView())
55             {
56                 /*
57                  * If an exception occurs while displaying in the console, then just
58                  * diplay atleast the same in a message-box
59                  */
60                 MessageDialog.openError(PlatformUI.getWorkbench().getActiveWorkbenchWindow().getShell(), "Error", msg);
61                 return;
62             }
63     
64             /* display message on console */
65             getNewMessageConsoleStream(msgType).println(msg);
66         }
67     
68         private boolean displayConsoleView()
69         {
70             try
71             {
72                 IWorkbenchWindow activeWorkbenchWindow = PlatformUI.getWorkbench().getActiveWorkbenchWindow();
73                 if (activeWorkbenchWindow != null)
74                 {
75                     IWorkbenchPage activePage = activeWorkbenchWindow.getActivePage();
76                     if (activePage != null)
77                         activePage.showView(IConsoleConstants.ID_CONSOLE_VIEW, null, IWorkbenchPage.VIEW_VISIBLE);
78                 }
79     
80             }
81             catch (PartInitException partEx)
82             {
83                 return false;
84             }
85     
86             return true;
87         }
88     
89         /**
90          * Returns a MessageConsoleStream
91          * 
92          * @param msgType
93          *            The type of message
94          * @return MessageConsoleStream
95          */
96         MessageConsoleStream getNewMessageConsoleStream(LoggerMessageType msgType)
97         {
98             MessageConsoleStream msgConsoleStream = getMessageConsole().newMessageStream();
99             setColor(msgConsoleStream, msgType);
100             return msgConsoleStream;
101         }
102     
103         private void setColor(final MessageConsoleStream msgConsoleStream, final LoggerMessageType msgType)
104         {
105             Display.getDefault().syncExec(new Runnable()
106             {
107                 @Override
108                 public void run()
109                 {
110                     msgConsoleStream.setColor(Display.getCurrent().getSystemColor(msgType.getColor()));
111                 }
112             });
113         }
114     
115         private MessageConsole getMessageConsole()
116         {
117             if (_MessageConsole == null)
118             {
119                 makeMessageConsoleStream(_Title);
120             }
121     
122             return _MessageConsole;
123         }
124     
125         private void makeMessageConsoleStream(String title)
126         {
127             _MessageConsole = new MessageConsole(title, null);
128             ConsolePlugin.getDefault().getConsoleManager().addConsoles(new IConsole[] { _MessageConsole });
129         }
130     }
