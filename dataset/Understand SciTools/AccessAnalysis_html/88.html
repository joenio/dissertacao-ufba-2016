31      * Extracts all packages, types and methods from given projects,
32      * determines the minimal access for all types and methods and 
33      * generates a result tree.
34      */
35     public class AccessAnalysis implements Analysis
36     {

44          * Initializes an object of AccessAnalysis.

49         public AccessAnalysis(List<IJavaProject> projects, boolean verbose)

51             _projects = projects;

53             if (verbose)

63         @Override
64         public IFile[] filesToBeLocked() throws AnalysisException
65         {
66             Set<IFile> files = new HashSet<IFile>();
67             
68             try
69             {
70                 for (ICompilationUnit compilationUnit : getCompilationUnits())
71                 {
72                     files.add((IFile) compilationUnit.getCorrespondingResource());
73                 }
74             }
75             catch (JavaModelException e)
76             {
77                 throw new AnalysisException(e);
78             }
79             
80             return files.toArray(new IFile[files.size()]);
81         }

83         @Override
84         public void run(IProgressMonitor monitor) throws AnalysisException
85         {
86             try
87             {
88                 createResults(monitor);
89             }
90             catch (ProgressMonitorCanceledException cancelExc)
91             {
92                 _printer.message("Canceled by user.");
93             }
94             catch (CoreException coreExc)
95             {
96                 _printer.error(coreExc);
97                 throw new AnalysisException(coreExc);
98             }
99             finally
100             {
101                 monitor.done();
102             }
103         }

105         /**
106          * Returns all compilation units of all projects
107          * 
108          * @return Set<ICompilationUnit> All compilation units
109          * 
110          * @throws JavaModelException if something goes wrong
111          */
112         private Set<ICompilationUnit> getCompilationUnits() throws JavaModelException
113         {
114             if (_compilationUnits == null)
115             {
116                 _compilationUnits = new HashSet<ICompilationUnit>();
117                 
118                 for (IJavaProject project : _projects)
119                 {
120                     for (IPackageFragment packageFragment : project.getPackageFragments())
121                     {
122                         if (packageFragment.getKind() == IPackageFragmentRoot.K_SOURCE)
123                         {
124                             _compilationUnits.addAll(Arrays.asList(packageFragment.getCompilationUnits()));
125                         }
126                     }
127                 }
128             }
129             
130             return _compilationUnits;
131         }
132         
133         /**
134          * Runs the actual analysis for all projects, 
135          * prints some basic information, 
136          * forwards some information to the progress monitor and
137          * puts the results to the fields when finished. 
138          * 
139          * Starts the progress monitor and increments it to the end when successful, but doesn't stop it.
140          * 
141          * @param IProgressMonitor monitor The current progress monitor
142          * 
143          * @throws AnalysisException if a project can't be analyzed 
144          * @throws CoreException if something goes wrong
145          * @throws ProgressMonitorCanceledException if the progress monitor is in canceled state
146          */
147         private void createResults(IProgressMonitor monitor) throws AnalysisException, CoreException, ProgressMonitorCanceledException
148         {
149             monitor.beginTask("", getCompilationUnits().size() + 2);
150             
151             checkAllReadable(_projects, monitor);
152             
153             _printer.analysisBefore();
154             Set<TypeAccessConstraintChecker> typeConstraints = ConstraintsCollector.getTypeConstraints();
155             Set<MethodAccessConstraintChecker> methodConstraints = ConstraintsCollector.getMethodConstraints();
156             AccessAnalysisResultCollector resultCollector = new AccessAnalysisResultCollector(_projects, typeConstraints, methodConstraints, monitor, _printer);
157             resultCollector.collect();
158             _projectResults = resultCollector.getResults(monitor);
159             _printer.analysisAfter();
160         }
161         
162     	/**
163          * Checks whether some given projects are readable for the analysis. 
164          * If this is not the case an exception is thrown.
165          * 
166          * Increments the progress monitor by 1 when successful.
167          * 
168          * @param Collection<IJavaProject> projects The projects to be checked
169          * @param IProgressMonitor monitor The current progress monitor
170          * 
171          * @throws CoreException if something goes wrong
172          * @throws AnalysisException AnalysisException if a project can't be analyzed
173          */
174         private void checkAllReadable(Collection<IJavaProject> projects, IProgressMonitor monitor) throws CoreException, AnalysisException, ProgressMonitorCanceledException
175         {
176             monitor.subTask("Check projects");
177             for (IJavaProject project : projects)
178             {
179                 ProgressMonitorCanceledException.checkMonitor(monitor);
180                 checkReadable(project);
181             }
182             monitor.worked(1);
183         }
184         
185         /**
186          * Checks whether a single given project is readable for the analysis. 
187          * If this is not the case an exception is thrown.
188          * 
189          * @param IJavaProject project The project to be checked
190          * 
191          * @throws CoreException if something goes wrong
192          * @throws AnalysisException AnalysisException if the project can't be analyzed
193          */
194         private void checkReadable(IJavaProject project) throws CoreException, AnalysisException
195         {
196             if (hasError(project, IJavaModelMarker.BUILDPATH_PROBLEM_MARKER) || 
197                 hasError(project, IJavaModelMarker.JAVA_MODEL_PROBLEM_MARKER))
198             {
199                 throw new AnalysisException("There are some errors in project \"" + project.getElementName() + "\". Analysis impossible!");
200             }           
201             if (project.hasUnsavedChanges())
202             {
203                 throw new AnalysisException("There are unsaved changes in project \"" + project.getElementName() + "\". Analysis impossible!");
204             }
205         }
206         
207         /**
208          * Checks whether a project contains an error of a specific type
209          * 
210          * @param IJavaProject project The project to be checked
211          * @param String errorType The type of error
212          * 
213          * @throws CoreException if something goes wrong
214          */
215         private boolean hasError(IJavaProject project, String errorType) throws CoreException
216         {
217             return IMarker.SEVERITY_ERROR == project.getUnderlyingResource().findMaxProblemSeverity(errorType, true, IResource.DEPTH_INFINITE);
218         }
