31         private final List<IJavaProject> _projects;
32         private final Set<TypeAccessConstraintChecker> _typeConstraints;
33         private final Set<MethodAccessConstraintChecker> _methodConstraints;
34         private final IProgressMonitor _monitor;

36         
37         private List<ProjectResult> _projectResults;
38         private List<Result> _mergedResults;
39         private ExpressionTypeDeterminator _expressionTypeDeterminator;

46         AccessAnalysisResultCollector(List<IJavaProject> projects, Set<TypeAccessConstraintChecker> typeConstraints,
47                 Set<MethodAccessConstraintChecker> methodConstraints, IProgressMonitor monitor, ElementPrinter printer)
48         {
49             _projects = projects;
50             _typeConstraints = typeConstraints;
51             _methodConstraints = methodConstraints;
52             _monitor = monitor;
53             _printer = printer;
54         }
55         
56         void collect() throws JavaModelException, ProgressMonitorCanceledException
57         {
58             _projectResults = new ArrayList<ProjectResult>();
59             _mergedResults = null;
60             _expressionTypeDeterminator = new ExpressionTypeDeterminator();
61             _typeHandler = new TypeHandler(_expressionTypeDeterminator, _typeConstraints);
62             _methodHandler = new MethodHandler(_expressionTypeDeterminator, _methodConstraints);
63             _currentProjectResult = null;
64             _currentSourceFolderResult = null;
65             _currentPackageResult = null;
66             
67             for (IJavaProject project : _projects)
68             {
69                 analyzeProject(project);
70             }
71         }
72         
73         private void analyzeProject(IJavaProject project) throws JavaModelException, ProgressMonitorCanceledException
74         {
75             _printer.projectBefore(project);
76             _currentProjectResult = new ProjectResult(project);
77             _projectResults.add(_currentProjectResult);
78             for (IPackageFragmentRoot packageFragmentRoot : project.getPackageFragmentRoots())
79             {
80                 analyzeSourceFolder(packageFragmentRoot);
81             }
82             _printer.projectAfter();
83             _currentProjectResult = null;
84         }
85         
86         private void analyzeSourceFolder(IPackageFragmentRoot packageFragmentRoot) throws JavaModelException, ProgressMonitorCanceledException
87         {
88             if (packageFragmentRoot.getKind() == IPackageFragmentRoot.K_SOURCE)
89             {
90                 _printer.sourceFolderBefore(packageFragmentRoot);
91                 _currentSourceFolderResult = new SourceFolderResult(packageFragmentRoot, _currentProjectResult);
92                 _currentProjectResult.addSourceFolder(_currentSourceFolderResult);
93                 for (IJavaElement child : packageFragmentRoot.getChildren())
94                 {
95                     if (child instanceof IPackageFragment)
96                     {
97                         analyzePackage((IPackageFragment) child);
98                     }
99                 }
100                 _printer.sourceFolderAfter();
101                 _currentSourceFolderResult = null;
102             }
103         }
104     
105         private void analyzePackage(IPackageFragment packageFragment) throws JavaModelException, ProgressMonitorCanceledException
106         {
107             if (packageFragment.getCompilationUnits().length > 0)
108             {
109                 _printer.packageBefore(packageFragment);
110                 _currentPackageResult = new PackageResult(packageFragment, _currentSourceFolderResult);
111                 _currentSourceFolderResult.addPackage(_currentPackageResult);
112                 for (ICompilationUnit unit : packageFragment.getCompilationUnits())
113                 {
114                     analyzeCompilationUnit(unit);
115                 }
116                 _printer.packageAfter();
117                 _currentPackageResult = null;
118             }
119         }
120     
121         private void analyzeCompilationUnit(ICompilationUnit unit) throws ProgressMonitorCanceledException
122         {
123             ProgressMonitorCanceledException.checkMonitor(_monitor);
124             _monitor.subTask("Read " + unit.getPath().toString());
125             CompilationUnit ast = parse(unit);
126             AccessAnalysisASTVisitor visitor = new AccessAnalysisASTVisitor(_currentPackageResult, _typeHandler, _methodHandler, _expressionTypeDeterminator, _printer);
127             ast.accept(visitor);
128             _monitor.worked(1);
129         }
130         
131         /**
132          * Creates the AST for a compilation unit.
133          * 
134          * @param ICompilationUnit
135          *            unit The compilation unit
136          * @return CompilationUnit The root element of the AST
137          */
138         private CompilationUnit parse(ICompilationUnit unit)
139         {
140             ASTParser parser = ASTParser.newParser(AST.JLS3);
141             parser.setKind(ASTParser.K_COMPILATION_UNIT);
142             parser.setSource(unit);
143             parser.setResolveBindings(true);
144             return (CompilationUnit) parser.createAST(null);
145         }
146         
147         /**
148          * Returns prepares and returns the results of last collect() call.
149          * 
150          * @return List<Result> All top level result elements (ProjectResults)
151          *      or null if collect() haven't been called yet
152          * @throws ProgressMonitorCanceledException if ProgressMonitor has canceled state.
153          */
154         List<Result> getResults(IProgressMonitor monitor) throws ProgressMonitorCanceledException
155         {
156             if (_projectResults != null && _mergedResults == null)
157             {
158                 monitor.subTask("Merge results");
159                 mergeResults(monitor);
160                 monitor.worked(1);
161             }
162             
163             return _mergedResults;
164         }
165         
166         /**
167          * Extracts the minimal access of types and methods from the maps 
168          * and sets the results.
169          * 
170          * Starts a new progress monitor subtask and increments it by 1 when successful.
171          * 
172          * @param Collection<Result> projectResults The results to be set
173          * @param MethodMap methodMap Map of minimal access of all methods
174          * @param TypeMap typeMap Map of minimal access of all types  
175          * 
176          * @throws ProgressMonitorCanceledException if progress monitor get in canceled state
177          */
178         private void mergeResults(IProgressMonitor monitor) throws ProgressMonitorCanceledException
179         {
180             MethodMap methodMap = _methodHandler.getMethodMap();
181             TypeMap typeMap = _typeHandler.getTypeMap();
182             
183             for (Result projectResult : _projectResults)
184             {
185                 for (Result sourceFolderResult : projectResult.getChildren()) //Source Folder
186                 {
187                     for (Result packageResult : sourceFolderResult.getChildren()) //Packages
188                     {
189                         for (Result typeResult : packageResult.getChildren()) //Types
190                         {
191                             ProgressMonitorCanceledException.checkMonitor(monitor);
192                             
193                             for (Result methodResult : typeResult.getChildren()) //Methods
194                             {
195                                 ((MethodResult) methodResult).setMinimalAccess(methodMap.getAccess(((MethodResult) methodResult).getJavaElement()));
196                             }
197                             ((TypeResult) typeResult).setMinimalAccess(typeMap.getAccess(((TypeResult) typeResult).getJavaElement()));
198                         }
199                     }
200                 }
201             }
202             
203             _mergedResults = new ArrayList<Result>(_projectResults);
204         }
