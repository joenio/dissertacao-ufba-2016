90         private void makeTreeViewer(Composite parent)
91         {
92             _viewer = new TreeViewer(parent, SWT.MULTI | SWT.H_SCROLL | SWT.V_SCROLL | SWT.FULL_SELECTION);

98         private void makeTreeLayout(TreeViewer viewer)
99         {
100             Tree tree = viewer.getTree();
101             tree.setHeaderVisible(true);
102             tree.setLinesVisible(true);
103             TableLayout layout = new TableLayout();
104     
105             TreeColumn treeColumn = new TreeColumn(tree, SWT.LEFT);
106             treeColumn.setText("Element");
107             TreeViewerColumn treeViewerColumn = new TreeViewerColumn(viewer, treeColumn);
108             treeViewerColumn.setLabelProvider(new ElementLabelProvider());
109             layout.addColumnData(new ColumnWeightData(100, 110));
110     
111             treeColumn = new TreeColumn(tree, SWT.LEFT);
112             treeColumn.setText("IGAT");
113             treeViewerColumn = new TreeViewerColumn(viewer, treeColumn);
114             treeViewerColumn.setLabelProvider(new IGATLabelProvider());
115             layout.addColumnData(new ColumnWeightData(10, 30));
116             
117             treeColumn = new TreeColumn(tree, SWT.LEFT);
118             treeColumn.setText("IGAM");
119             treeViewerColumn = new TreeViewerColumn(viewer, treeColumn);
120             treeViewerColumn.setLabelProvider(new IGAMLabelProvider());
121             layout.addColumnData(new ColumnWeightData(10, 30));
122             
123             treeColumn = new TreeColumn(tree, SWT.LEFT);
124             treeColumn.setText("Minimal Access Modifier");
125             treeViewerColumn = new TreeViewerColumn(viewer, treeColumn);
126             treeViewerColumn.setLabelProvider(new MinimalAccessLabelProvider());
127             layout.addColumnData(new ColumnWeightData(26, 80));
128             
129             treeColumn = new TreeColumn(tree, SWT.LEFT);
130             treeColumn.setText("Actual Access Modifier");
131             treeViewerColumn = new TreeViewerColumn(viewer, treeColumn);
132             treeViewerColumn.setLabelProvider(new ActualAccessLabelProvider());

212         private Result getSelectedResult()
213         {
214             ISelection candidate = _viewer.getSelection();
215             
216             if (!(candidate instanceof IStructuredSelection))
217             {
218                 return null;
219             }
220     
221             Object firstElement = ((IStructuredSelection) candidate).getFirstElement();
222             if (!(firstElement instanceof Result))
223             {
224                 return null;
225             }
226             
227             return (Result) firstElement;
228         }

367         private class DoubleClickAction extends Action
368         {
369             public void run()
370             {
371                 Result result = getSelectedResult();
372                 
373                 if (result.isElementEditable())
374                 {
375                     openElement(result.getJavaElement());
376                 }
377                 
378                 _viewer.expandToLevel(result, 1);
379             }
380             
381             private void openElement(IJavaElement element)
382             {
383                 try
384                 {
385                     IEditorPart editor = JavaUI.openInEditor(element);
386                     JavaUI.revealInEditor(editor, element);
387                 }
388                 catch (PartInitException e)
389                 {
390                     showMessage(e.getMessage());
391                 }
392                 catch (JavaModelException e)
393                 {
394                     showMessage(e.getMessage());
395                 }
396             }
397         }
