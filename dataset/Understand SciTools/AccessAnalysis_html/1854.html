103         private void createAnnotationTypeField(Composite parent)
104         {
105             Label label = new Label(parent, SWT.NONE);
106             label.setText("Annotation:");
107     
108             _annotationTypeTextField = new Text(parent, SWT.BORDER);
109             _annotationTypeTextField.setLayoutData(GridDataFactory.swtDefaults().minSize(120, SWT.DEFAULT).span(3, 1).grab(true, false)
110                     .align(SWT.FILL, SWT.CENTER).create());
111             _annotationTypeTextField.setEditable(false);
112     
113             Button browseButton = new Button(parent, SWT.PUSH);
114             browseButton.setText("Browse");
115             browseButton.addSelectionListener(new SelectionAdapter() {
116                 @Override
117                 public void widgetSelected(SelectionEvent event)
118                 {
119                     try
120                     {
121                         IJavaSearchScope scope = SearchEngine.createJavaSearchScope(new IJavaElement[] { _project });
122     
123                         SelectionDialog dialog = JavaUI.createTypeDialog(getShell(), new ProgressMonitorDialog(getShell()),
124                                 scope, IJavaElementSearchConstants.CONSIDER_ANNOTATION_TYPES, false);
125                         dialog.open();
126     
127                         if (dialog.getReturnCode() == SelectionDialog.OK)
128                         {
129                             for (Object result : dialog.getResult())
130                             {
131                                 if (result instanceof IType && ((IType) result).isAnnotation())
132                                 {
133                                     _annotationType = (IType) result;
134                                     updateTextFieldAndOkButton();
135                                 }
136                             }
137                         }
138                     } catch (JavaModelException e)
139                     {
140                         showErrorMessage("Error while looking for Annotation Types.", e);
141                     }
142                 }
143             });
144         }

162         private void initFields()
163         {
164             assert _inputConstraint != null : "Precondition failed: _inputConstraint != null";
165             
166             _annotationType = _inputConstraint.getAnnotationType();
167             
168             ApplyTo applyTo = _inputConstraint.applyTo();
169             _typesCheckbox.setSelection(applyTo == ApplyTo.TYPES);
170             _methodsCheckbox.setSelection(applyTo == ApplyTo.METHODS);
171             _bothCheckbox.setSelection(applyTo == ApplyTo.BOTH);
172             
173             _accessComboViewer.setSelection(new StructuredSelection(_inputConstraint.getMinimalAccess()));
174         }
175     
176         private void createLeastMinimalAccessModifierField(Composite parent)
177         {
178             Composite main = new Composite(parent, SWT.NONE);
179             main.setLayout(GridLayoutFactory.swtDefaults().numColumns(2).margins(0, 0).create());
180             main.setLayoutData(GridDataFactory.swtDefaults().span(5, 1).create());
181         
182             Label accessLabel = new Label(main, SWT.NONE);
183             accessLabel.setText("Minimal Access Modifier at least:");
184         
185             _accessComboViewer = new ComboViewer(main, SWT.DROP_DOWN | SWT.READ_ONLY);
186             _accessComboViewer.setContentProvider(new ArrayContentProvider());
187             _accessComboViewer.setInput(new Access[] { Access.PUBLIC, Access.PROTECTED, Access.DEFAULT, Access.PRIVATE });
188             _accessComboViewer.setSelection(new StructuredSelection(Access.PUBLIC));
189         }

191         private void updateTextFieldAndOkButton()
192         {
193             String text = "";
194             if (_annotationType != null)
195             {
196                 getButton(IDialogConstants.OK_ID).setEnabled(true);
197                 text = _annotationType.getFullyQualifiedName();
198             }
199             _annotationTypeTextField.setText(text);
200         }

202         @Override
203         protected void okPressed()
204         {
205             assert _annotationType != null : "Precondition failed: OK button not clickable as long as no Annotation is selected";
206             assert !_accessComboViewer.getSelection().isEmpty() : "Precondition failed: OK button not clickable as long as no Minimal Access Modifier is selected";
207             
208             Access access = (Access) ((IStructuredSelection) _accessComboViewer.getSelection()).getFirstElement();
209             ApplyTo applyTo = ApplyTo.BOTH;
210             if (_typesCheckbox.getSelection())
211             {
212                 applyTo = ApplyTo.TYPES;
213             }
214             else if (_methodsCheckbox.getSelection())
215             {
216                 applyTo = ApplyTo.METHODS;
217             }
218             
219             _resultConstraint = new AnnotationConstraint(_annotationType, access, applyTo);
220             
221             super.okPressed();
222         }

229         private void showErrorMessage(String message, Exception exception) {
230             final StringBuilder errorMessage = new StringBuilder();
231             if (message != null) {
232                 errorMessage.append(message);
233             }
234             if (message != null && exception != null) {
235                 errorMessage.append(" (");
236             }
237             if (exception != null) {
238                 errorMessage.append(exception.getMessage());
239             }
240             if (message != null && exception != null) {
241                 errorMessage.append(")");
242             }
243             
244             getShell().getDisplay().asyncExec(new Runnable() {
245                 
246                 @Override
247                 public void run() {
248                     MessageDialog.openError(getShell(), "Error", errorMessage.toString());
249                 }
250             });
251         }
