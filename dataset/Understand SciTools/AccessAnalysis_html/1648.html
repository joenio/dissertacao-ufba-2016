4     package de.uni_hamburg.informatik.swt.accessanalysis;

6     import java.util.LinkedList;
7     import java.util.List;
8     
9     import org.eclipse.jdt.core.IJavaProject;
10     
11     import de.uni_hamburg.informatik.swt.accessanalysis.analysis.AccessAnalysis;
12     
13     
14     /**
15      * Generates the right analysis object
16      */
17     public class AnalysisFactory
18     {   
19         /**
20          * All possible analysis modes
21          */
22         public static enum AnalysisMode 
23         {
24             ACCESS_QUIET,
25             ACCESS_VERBOSE,
26         }
27      
28         /**
29          * The default analysis mode
30          */
31         public static final AnalysisMode DEFAULT = AnalysisMode.ACCESS_QUIET;
32         
33         /**
34          * Returns the analysis for a list of projects at a given mode.
35          * 
36          * @param List<IJavaProject> projects The projects to be analyzed
37          * @param AnalysisMode mode The analysis mode of choice
38          * @return Analysis The analysis object
39          */
40         public static Analysis analyzer(List<IJavaProject> projects, AnalysisMode mode)
41         {
42             switch (mode)
43             {
44                 case ACCESS_QUIET : 
45                     return new AccessAnalysis(projects, false);
46                 case ACCESS_VERBOSE : 
47                     return new AccessAnalysis(projects, true);
48             }
49             
50             return null;
51         }
52         
53         /**
54          * Returns the analysis for a single project at a given mode.
55          * 
56          * @param IJavaProject project The project to be analyzed
57          * @param AnalysisMode mode The analysis mode of choice
58          * @return Analysis The analysis object
59          */
60         public static Analysis analyzer(IJavaProject project, AnalysisMode type)
61         {
62             List<IJavaProject> projects = new LinkedList<IJavaProject>();
63             projects.add(project);
64             return analyzer(projects, type);
65         }
66         
67         /**
68          * Returns the analysis for a list of projects at the default mode.
69          * 
70          * @param List<IJavaProject> projects The projects to be analyzed
71          * @return Analysis The analysis object
72          */
73         public static Analysis analyzer(List<IJavaProject> projects)
74         {
75             return analyzer(projects, DEFAULT);
76         }
77     }
