15     /**
16      * Collects minimal access and overridden methods of methods
17      * and maps the methods to their actual minimal access.
18      *
19      */
20     class MethodMap
21     {
22         private final Map<String, Access> _accessMap = new HashMap<String, Access>();
23         private final Map<String, String> _overrideMap = new HashMap<String, String>();;
24         
25         
26         void putAccess(IMethodBinding binding, Access access)
27         {
28             putAccess(binding.getKey(), access);
29         }
30         
31         private void putAccess(String key, Access access)
32         {
33             Access old = _accessMap.get(key);
34             if (old == null || access.compareTo(old) > 0)
35             {
36                 _accessMap.put(key, access);
37             }
38         }
39     
40         void putOverride(IMethodBinding overriding, IMethodBinding overridden)
41         {
42             putOverride(overriding.getKey(), overridden.getKey());
43         }
44         
45         private void putOverride(String overridingKey, String overriddenKey)
46         {
47             _overrideMap.put(overridingKey, overriddenKey);
48         }
49         
50         Access getAccess(IMethod method)
51         {
52             return getAccess(method.getKey());
53         }
54     
55         private Access getAccess(String key)
56         {
57             Access access = Access.NO_USE;
58             
59             Access ownAccess = _accessMap.get(key);
60             if (ownAccess != null)
61             {
62                 access = ownAccess;
63             }
64             
65             String overrideKey = _overrideMap.get(key);
66             if (overrideKey != null)
67             {
68                 Access overrideAccess = getAccess(overrideKey);
69                 if (overrideAccess.compareTo(access) > 0)
70                 {
71                     access = overrideAccess;
72                 }
73             }
74             
75             return access;
76         }
77         
78         @Override
79         public String toString()
80         {
81             return _accessMap.toString();
82         }
83     }
