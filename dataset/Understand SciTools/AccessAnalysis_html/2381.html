4     package de.uni_hamburg.informatik.swt.accessanalysis.access;

6     import org.eclipse.jdt.core.Flags;
7     import org.eclipse.jdt.core.IMember;
8     import org.eclipse.jdt.core.JavaModelException;
9     import org.eclipse.swt.graphics.Image;

11     import de.uni_hamburg.informatik.swt.accessanalysis.internal.AccessAnalysisActivator;
12     
13     /**
14      * All possible access modes of Java elements
15      */
16     public enum Access
17     {   
18         NO_USE("not used", "icons/access_not_used.png"),
19         PRIVATE("private", "icons/access_private.png"),
20         DEFAULT("default", "icons/access_default.png"),
21         PROTECTED("protected", "icons/access_protected.png"),
22         PUBLIC("public", "icons/access_public.png");
23         
24         private final String _name;
25         private final Image _image;
26         
27         /**
28          * Initializes an object of Access. 
29          *
30          * @param String name The name of the access mode
31          */
32         private Access(String name, String imagePath)
33         {
34             _name = name;
35             _image = AccessAnalysisActivator.imageDescriptorFromPlugin(imagePath).createImage();
36         }
37         
38         /**
39          * Returns an icon representing this access level
40          * 
41          * @return Image The Icon image
42          */
43         public Image getImage()
44         {
45             return _image;
46         }
47         
48         /**
49          * Returns a string representation
50          * 
51          * @return String The name of this access mode
52          */
53         @Override
54         public String toString()
55         {
56             return _name;
57         }
58         
59         /**
60          * Returns the access mode of an IMember by examining the modifier flags
61          * 
62          * @param IMember member The IMember to be examined
63          * @return Access The access mode of the given IMember  
64          * 
65          * @throws JavaModelException if the <code>member</code> does not exist or if an exception 
66          *          occurs while accessing its corresponding resource.
67          */
68         public static Access fromFlags(IMember member) throws JavaModelException
69         {
70             int flags = member.getFlags();
71             
72             if (Flags.isPrivate(flags))
73             {
74                 return Access.PRIVATE;
75             }
76             else if (Flags.isPackageDefault(flags))
77             {
78                 return Access.DEFAULT;
79             }
80             else if (Flags.isProtected(flags))
81             {
82                 return Access.PROTECTED;
83             }
84             else if (Flags.isPublic(flags))
85             {
86                 return Access.PUBLIC;
87             }
88         
89             return Access.NO_USE; 
90         }
91         
92         /**
93          * Returns the more generous Access of two.
94          * 
95          * @param access1 The first Access to compare
96          * @param access2 The second Access
97          * 
98          * @return Access <code>access2<code> if it is less restrictive than <code>access1</code>, 
99          * else <code>access1</code>. 
100          */
101         public static Access max(Access access1, Access access2)
102         {
103         	if (access1.compareTo(access2) < 0)
104         	{
105         		return access2;
106         	}
107         	
108         	return access1;
109         }
110         
111         /**
112          * Returns the Access modes that can really be the actual access level modes
113          * of Java elements such as types or methods.
114          * Namely these are PUBLIC, PROTECTED, DEFAULT and PRIVATE. The pseudo access
115          * modifier NO_USE which can only be the minimal access modifier of an
116          * element, is omited.
117          * 
118          * @return Access[] The named Access modes
119          */
120         public static Access[] real()
121         {
122             return new Access[] { Access.PRIVATE, Access.DEFAULT, Access.PROTECTED, Access.PUBLIC };
123         }
124     }
