6     import org.eclipse.jface.resource.ColorRegistry;
7     import org.eclipse.jface.viewers.StyledCellLabelProvider;
8     import org.eclipse.jface.viewers.ViewerCell;
9     import org.eclipse.swt.custom.StyleRange;
10     import org.eclipse.swt.graphics.RGB;
11     import org.eclipse.swt.widgets.Event;

17     /**
18      * Provides the labels for the "Actual Access"-column 
19      * in the AccessAnalysisView-TreeTable.
20      */
21     class ActualAccessLabelProvider extends StyledCellLabelProvider
22     {
23         private final ColorRegistry _colorRegistry;
24         
25         /**
26          * Initializes an object of MinimalAccessLabelProvider. 
27          */
28         ActualAccessLabelProvider()
29         {
30             super(StyledCellLabelProvider.COLORS_ON_SELECTION);
31             _colorRegistry = new ColorRegistry();
32             _colorRegistry.put("red", new RGB(255, 0, 0));
33         }
34         
35         
36         
37         @Override
38         public void update(ViewerCell cell)
39         {
40             ResultFormatter formatter = ((Result) cell.getElement()).getFormatter();
41             cell.setText(formatter.actualAccess());
42             cell.setStyleRanges(createStyle(formatter));
43             super.update(cell);
44         }
45         
46         @Override
47         protected void paint(Event event, Object element)
48         {
49             super.paint(event, element);
50         }
51         
52         /**
53          * Returns the style for the label. 
54          *  
55          * @param ResultFormatter formatter The result formatter
56          * @return StyleRange[] The style
57          */
58         private StyleRange[] createStyle(ResultFormatter formatter)
59         {
60             StyleRange[] styleRanges = new StyleRange[1];
61             styleRanges[0] = new StyleRange();
62     
63             if (formatter.accessAlert())
64             {
65                 styleRanges[0].foreground = _colorRegistry.get("red");
66             }
67             else
68             {
69                 styleRanges[0].foreground = _colorRegistry.get("black");
70             }
71             
72             styleRanges[0].start = 0;
73             styleRanges[0].length = formatter.actualAccess().length();
74             
75             return styleRanges;
76         }
77     }
