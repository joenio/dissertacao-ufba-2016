30     @RcpExtension
31     public class AccessAnalysisActionDelegate extends ActionDelegate implements IViewActionDelegate
32     {
33         private ISelection _selection;
34         
35         @Override
36         public void init(IViewPart view)
37         {
38         }
39     
40         @Override
41         public void selectionChanged(IAction ignored, ISelection selection)
42         {
43             _selection = selection;
44         }
45         
46         @Override
47         public void run(IAction ignored)
48         {
49             if (_selection != null)
50             {
51                 scheduleJob(getProjects());
52             }
53         }
54         
55         private List<IJavaProject> getProjects()
56         {
57             LinkedList<IJavaProject> projectList = new LinkedList<IJavaProject>();
58             
59             if (_selection instanceof IStructuredSelection)
60             {
61                 Set<IJavaProject> projectSet = new HashSet<IJavaProject>();
62                 for (Object element : ((IStructuredSelection) _selection).toArray())
63                 {
64                     if (element instanceof IJavaElement)
65                     {
66                         projectSet.add(((IJavaElement) element).getJavaProject());
67                     }
68                 }
69                 projectList.addAll(projectSet);
70             }
71             
72             return projectList;
73         }
74         
75         private void scheduleJob(List<IJavaProject> projects)
76         {
77             if (! projects.isEmpty())
78             {
79             	try {
80         			Collection<AccessAnalysisResultsConsumer> resultsConsumer = RunAnalysisActionActivator.getResultsConsumer();
81         			Job job = new AccessAnalysisJob(projects, resultsConsumer);
82         			job.schedule();
83         		} 
84                 catch (CoreException e) {
85         			RunAnalysisActionActivator.showErrorMessage(null, e);
86         			e.printStackTrace();
87         		}
88             }
89             else
90             {
91                 MessageDialog.openWarning(Display.getCurrent().getActiveShell(), "AccessAnalysis", "No Project selected!");
92             }
93         }
