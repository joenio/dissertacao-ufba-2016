15     /**
16      * Maps types to their minimal access.
17      */
18     class TypeMap
19     {
20         private final Map<String, Access> _accessMap = new HashMap<String, Access>();
21         
22         void putAccess(ITypeBinding binding, Access access)
23         {
24             putAccess(binding.getKey(), access);
25         }
26         
27         private void putAccess(String key, Access access)
28         {
29             Access old = _accessMap.get(key);
30             if (old == null || access.compareTo(old) > 0)
31             {
32                 _accessMap.put(key, access);
33             }
34         }
35         
36         Access getAccess(IType type)
37         {
38             return getAccess(type.getKey());
39         }
40     
41         private Access getAccess(String key)
42         {
43             if (_accessMap.containsKey(key))
44             {
45                 return _accessMap.get(key);
46             }
47             
48             return Access.NO_USE;
49         }
50         
51         @Override
52         public String toString()
53         {
54             return _accessMap.toString();
55         }
56     }
