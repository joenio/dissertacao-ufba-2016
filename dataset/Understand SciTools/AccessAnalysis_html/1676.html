24     /**
25      * Copies selected results from the AccessAnalysisView to the clipboard.
26      */
27     @RcpExtension
28     public class ClipboardHandler extends AbstractHandler
29     {
30         @Override
31         public Object execute(ExecutionEvent event) throws ExecutionException
32         {
33             Clipboard clipboard = new Clipboard(HandlerUtil.getActiveShell(event).getDisplay());
34             try
35             {
36                 copyResults(event, clipboard);
37                 return null;
38             }
39             finally
40             {
41                 clipboard.dispose();
42             }
43         }
44     
45         private void copyResults(ExecutionEvent event, Clipboard clipboard) throws ExecutionException
46         {
47             ISelection selection = HandlerUtil.getCurrentSelection(event);
48             if (selection instanceof IStructuredSelection)
49             {
50                 Object[] objects = ((IStructuredSelection) selection).toArray();
51                 if (objects.length > 0)
52                 {
53                     try
54                     {
55                         Object[] contentObjects = new Object[] { asResource(objects), asText(objects) };
56                         Transfer[] transfers = new Transfer[] { EditorInputTransfer.getInstance(), TextTransfer.getInstance() };
57                         clipboard.setContents(contentObjects, transfers);
58                     }
59                     catch (SWTError error) { }
60                 }
61             }
62             
63         }
64     
65         private IResource[] asResource(Object[] objects)
66         {
67             Collection<IResource> resources = new HashSet<IResource>(objects.length);
68             
69             for (Object object : objects)
70             {
71                 if (object instanceof Result)
72                 {
73                     Result result = (Result) object;
74                     if (result.isElementEditable())
75                     {
76                         try
77                         {
78                             resources.add(result.getJavaElement().getUnderlyingResource());
79                         }
80                         catch (JavaModelException e)
81                         {
82                         }
83                     }
84                 }
85             }
86             return resources.toArray(new IResource[resources.size()]);
87         }
88     
89         private String asText(Object[] objects)
90         {
91             StringBuffer buf = new StringBuffer();
92             for (Object object : objects)
93             {
94                 buf.append(object.toString());
95                 buf.append(System.getProperty("line.separator"));
96             }
97             return buf.toString();
98         }
