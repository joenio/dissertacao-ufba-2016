13     /**
14      * Determines the lowest access mode needed for the usage of a type.
15      */
16     public abstract class TypeAccessDeterminator
17     {
18         /**
19          * Determines the minimal access of an used type needed for the usage.
20          * Assumes that the using type is the current one.
21          * 
22          * @param ITypeBinding expressionType The binding of the type that is to be used.
23          * @return The minimal access needed for the usage
24          */
25         public Access determineAccess(ITypeBinding binding) throws JavaModelException
26         {
27             if (binding.isArray())
28             {
29                 return determineAccess(binding.getElementType());
30             }
31     
32             if (canBeNoUse(binding))
33             {
34                 return Access.NO_USE;
35             }
36             
37             if (canBePrivate(binding))
38             {
39                 return Access.PRIVATE;
40             }
41             
42             if (canBeDefault(binding))
43             {
44                 return Access.DEFAULT;
45             }
46     
47             if (canBeProtected(binding))
48             {
49                 return Access.PROTECTED;
50             }
51     
52             return Access.PUBLIC;
53         }
54         
55         abstract boolean canBeNoUse(ITypeBinding binding) throws JavaModelException;
56         
57         abstract boolean canBePrivate(ITypeBinding binding) throws JavaModelException;
58         
59         private boolean canBeDefault(ITypeBinding binding) throws JavaModelException
60         {
61             return binding.getPackage().equals(getCurrentPackageBinding());
62         }
63         
64         abstract boolean canBeProtected(ITypeBinding binding);
65         
66         abstract IPackageBinding getCurrentPackageBinding();
67     }
