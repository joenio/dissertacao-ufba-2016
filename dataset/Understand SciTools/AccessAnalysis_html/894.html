15     /**
16      * Determines the lowest access mode needed for the usage of a type.
17      */
18     public class UsedTypeAccessDeterminator extends TypeAccessDeterminator
19     {
20         private final ITypeBinding _usingTypeBinding;
21         
22         /**
23          * Initializes an object of UsedTypeAccessDeterminator. 
24          * 
25          * @param ITypeBinding usingType The Type that uses other types
26          */
27         public UsedTypeAccessDeterminator(ITypeBinding usingType)
28         {
29             _usingTypeBinding = usingType;
30         }
31         
32         boolean canBeNoUse(ITypeBinding binding)
33         {
34             return binding.equals(getCurrentTypeBinding()) ||
35                    Nesting.isNestedIn(getCurrentTypeBinding(), binding);
36         }
37         
38         boolean canBePrivate(ITypeBinding binding) throws JavaModelException
39         {
40             if (! binding.isNested())
41             {
42                 return false;
43             }
44             
45             ICompilationUnit compilationUnit = ((IType) binding.getJavaElement()).getCompilationUnit();
46             return getCurrentCompilationUnit().equals(compilationUnit);
47         }
48     
49         boolean canBeProtected(ITypeBinding binding)
50         {
51             if (! binding.isNested())
52             {
53                 return false;
54             }
55     
56             ITypeBinding declaringType = binding.getDeclaringClass();
57             
58             ITypeBinding currentType = getCurrentTypeBinding();
59             while (currentType != null)
60             {
61                 if (Hierarchy.isSuperType(declaringType, currentType))
62                 {
63                     return true;
64                 }
65                 currentType = currentType.getDeclaringClass();
66             }
67             
68             if (declaringType.getPackage().equals(getCurrentPackageBinding()))
69             {
70                 return true;
71             }
72             
73             return false;
74         }
75         
76         private ITypeBinding getCurrentTypeBinding()
77         {
78             return _usingTypeBinding;
79         }
80         
81         private ICompilationUnit getCurrentCompilationUnit()
82         {
83             return ((IType) getCurrentTypeBinding().getJavaElement()).getCompilationUnit();
84         }
85         
86         IPackageBinding getCurrentPackageBinding()
87         {
88             return getCurrentTypeBinding().getPackage();
89         }
90     }
