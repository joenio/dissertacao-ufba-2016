13     /**
14      * Determines the lowest access mode needed to override a method.
15      * 
16      * @author christian
17      */
18     public class OverriddenMethodAccessDeterminator
19     {
20     	public static Access determineAccess(IMethodBinding overridingBinding, IMethodBinding overriddenBinding)
21         {
22             if (overriddenBinding.getDeclaringClass().isInterface())
23             {
24                 return Access.PUBLIC;
25             }
26             
27             if (samePackage(overridingBinding, overriddenBinding))
28             {
29                 return Access.DEFAULT;
30             }
31             
32             return Access.PROTECTED;
33         }
34            
35         private static boolean samePackage(IMethodBinding overridingBinding, IMethodBinding overriddenBinding)
36         {
37             return overridingBinding.getDeclaringClass().getPackage().equals(overriddenBinding.getDeclaringClass().getPackage());
38         }
39         
40         public static Access getActualAccess(IMethodBinding methodBinding) throws JavaModelException
41         {
42             int modifiers = methodBinding.getModifiers();
43             
44             if (Modifier.isPublic(modifiers))
45             {
46                 return Access.PUBLIC;
47             }
48             else if (Modifier.isProtected(modifiers))
49             {
50                 return Access.PROTECTED;
51             }
52             else if (Modifier.isPrivate(modifiers))
53             {
54                 return Access.PRIVATE;
55             }
56             
57             return Access.DEFAULT;
58         }
