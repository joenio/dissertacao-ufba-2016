31     extern mtValuesNode mtValuesNode_create (cstringList values) 
32     {
33       mtValuesNode res = (mtValuesNode) dmalloc (sizeof (*res));
34     
35       DPRINTF (("Values node: %s", cstringList_unparse (values)));
36       res->values = values;
37       return res;
38     }
39     
40     void mtValuesNode_free (mtValuesNode node)
41     {
42       DPRINTF (("Free values: %s", mtValuesNode_unparse (node)));
43       cstringList_free (node->values); /*! why shouldn't we free this? */
44       sfree (node);
45     }
46     
47     extern cstring mtValuesNode_unparse (mtValuesNode node)
48     {
49       return message ("values %q", cstringList_unparse (node->values));
50     }
