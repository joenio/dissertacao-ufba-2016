28     # include "splintMacros.nf"
29     # include "basic.h"
30     
31     extern mtTransferClause mtTransferClause_create (mttok fromname, mttok toname, mtTransferAction action)
32     {
33       mtTransferClause res = (mtTransferClause) dmalloc (sizeof (*res));
34     
35       llassert (mttok_isIdentifier (fromname));
36       res->fromname = mttok_getText (fromname);
37     
38       llassert (mttok_isIdentifier (toname));
39       res->toname = mttok_getText (toname);
40     
41       res->action = action;
42       res->loc = mttok_stealLoc (fromname);
43       if (fileloc_isUndefined (res->loc))
44         {
45           llbug (message ("Not fileloc: %q", mttok_unparse (fromname)));
46         }
47     
48       mttok_free (fromname);
49       mttok_free (toname);
50       return res;
51     }
52     
53     extern void mtTransferClause_free (mtTransferClause transfer)
54     {
55       cstring_free (transfer->fromname);
56       cstring_free (transfer->toname);
57       fileloc_free (transfer->loc);
58       mtTransferAction_free (transfer->action);
59       sfree (transfer);
60     }
61     
62     extern cstring mtTransferClause_unparse (mtTransferClause node)
63     {
64       return message ("%s as %s ==> %q",
65     		  node->fromname, node->toname, 
66     		  mtTransferAction_unparse (node->action));
67     }
