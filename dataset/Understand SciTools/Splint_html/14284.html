33     valueTable valueTable_copy (valueTable s)
34     {
35       if (valueTable_size (s) > 0)
36         {
37           valueTable t;
38           t = valueTable_create (valueTable_size (s));
39           
40           valueTable_elements (s, key, val) 
41     	{
42     	  valueTable_insert (t, cstring_copy (key), stateValue_copy (val));
43     	} end_valueTable_elements ;
44     	
45     	llassert (valueTable_size (s) == valueTable_size (t));
46     	return t;
47         }
48       else
49         {
50           return valueTable_undefined;
51         }
52     }
53     
54     cstring valueTable_unparse (valueTable h)
55     {
56       cstring res = cstring_newEmpty ();
57     
58       valueTable_elements (h, key, val) {
59         DPRINTF (("Using key: %s", key));
60         res = message ("%q%s: %q [%q]; ", res, key, 
61     		   stateValue_unparseValue (val, context_lookupMetaStateInfo (key)),
62     		   stateValue_unparse (val));
63       } end_valueTable_elements ;
64       
65       return res;
66     }
67     
68     void valueTable_insert (valueTable h, cstring key, stateValue value)
69     {
70       llassert (stateValue_isDefined (value));
71       genericTable_insert ((genericTable) (h), key, (void *) (value));
72     }
73     
74     void valueTable_update (valueTable h, cstring key, stateValue newval) 
75     {
76       DPRINTF (("Update: %s -> %s", key, stateValue_unparse (newval)));
77       llassert (stateValue_isDefined (newval));
78       genericTable_update ((genericTable) (h), key, (void *) (newval));
79     }
