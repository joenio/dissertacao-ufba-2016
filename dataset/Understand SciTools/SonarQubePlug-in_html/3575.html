32     import java.util.Arrays;

34     
35     import org.sonar.api.measures.Metric;
36     import org.sonar.plugins.SourceMeterCore.api.SourceMeterMetricFinder;
37     
38     /**
39      * Metric finder class for get SourceMeter Metric objects
40      */
41     public class SourceMeterJavaMetricFinder extends SourceMeterMetricFinder {
42     
43         /**
44          * Default Constructor
45          */
46         public SourceMeterJavaMetricFinder() {
47             super();
48             SourceMeterJavaMetrics javaMetrics = new SourceMeterJavaMetrics();
49             for (Metric metric : javaMetrics.getMetrics()) {
50                 this.metricMap.put(metric.getKey(), metric);
51             }
52         }
53     
54         @Override
55         public List<Metric> findLanguageSpecificRulesetMetrics() {
56             return Arrays.asList(
57                     SourceMeterJavaMetrics.ISSUEGROUP_ANDROID,
58                     SourceMeterJavaMetrics.ISSUEGROUP_BAD_PRACTICE,
59                     SourceMeterJavaMetrics.ISSUEGROUP_BRACE,
60                     SourceMeterJavaMetrics.ISSUEGROUP_CLONE_IMPLEMENTATION,
61                     SourceMeterJavaMetrics.ISSUEGROUP_CODE_SIZE,
62                     SourceMeterJavaMetrics.ISSUEGROUP_COMMENT,
63                     SourceMeterJavaMetrics.ISSUEGROUP_CONTROVERSIAL,
64                     SourceMeterJavaMetrics.ISSUEGROUP_CORRECTNESS,
65                     SourceMeterJavaMetrics.ISSUEGROUP_COUPLING,
66                     SourceMeterJavaMetrics.ISSUEGROUP_DODGY_CODE,
67                     SourceMeterJavaMetrics.ISSUEGROUP_EMPTY_CODE,
68                     SourceMeterJavaMetrics.ISSUEGROUP_EXPERIMENTAL,
69                     SourceMeterJavaMetrics.ISSUEGROUP_FINALIZER,
70                     SourceMeterJavaMetrics.ISSUEGROUP_IMPORT_STATEMENT,
71                     SourceMeterJavaMetrics.ISSUEGROUP_INTERNATIONALIZATION,
72                     SourceMeterJavaMetrics.ISSUEGROUP_J2EE,
73                     SourceMeterJavaMetrics.ISSUEGROUP_JAKARTA_COMMONS_LOGGING,
74                     SourceMeterJavaMetrics.ISSUEGROUP_JAVA_LOGGING,
75                     SourceMeterJavaMetrics.ISSUEGROUP_JAVABEAN,
76                     SourceMeterJavaMetrics.ISSUEGROUP_JUNIT,
77                     SourceMeterJavaMetrics.ISSUEGROUP_MIGRATION,
78                     SourceMeterJavaMetrics.ISSUEGROUP_MULTITHREADED_CORRECTNESS,
79                     SourceMeterJavaMetrics.ISSUEGROUP_OPTIMIZATION,
80                     SourceMeterJavaMetrics.ISSUEGROUP_SECURITY_CODE_GUIDELINE,
81                     SourceMeterJavaMetrics.ISSUEGROUP_STRICT_EXCEPTION,
82                     SourceMeterJavaMetrics.ISSUEGROUP_STRING_AND_STRINGBUFFER,
83                     SourceMeterJavaMetrics.ISSUEGROUP_TYPE_RESOLUTION,
84                     SourceMeterJavaMetrics.ISSUEGROUP_VULNERABILITY);
85         }
