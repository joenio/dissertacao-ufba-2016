222       public static String[] keywordValues() {
223         RPGKeyword[] keywordsEnum = RPGKeyword.values();
224         String[] keywords = new String[keywordsEnum.length];
225         for (int i = 0; i < keywords.length; i++) {
226           keywords[i] = keywordsEnum[i].getValue();
227         }
228         return keywords;
229       }

231         public static Set<String> keywordValuesSet() {

234             for (int i = 0; i < keywordsEnum.length; i++) {
235               keywords.add(keywordsEnum[i].getValue());
236             }
237             return keywords;
