30     package com.sourcemeter.analyzer.rpg.colorizer;
31     
32     import java.util.regex.Matcher;
33     import java.util.regex.Pattern;
34     
35     import org.sonar.channel.CodeReader;
36     import org.sonar.colorizer.HtmlCodeBuilder;
37     import org.sonar.colorizer.NotThreadSafeTokenizer;
38     
39     /**
40      * Detect case-sensitive keywords
41      */
42     public class RPGRecordTypeColorizer extends NotThreadSafeTokenizer {
43     
44         private final String tagBefore;
45         private final String tagAfter;
46         private final Matcher matcher;
47         private final StringBuilder tmpBuilder = new StringBuilder();
48         private static final String DEFAULT_REGEX = "[DFICOEHdficoeh]{1}";
49         public static final int COLUMN_INDEX = 5;
50     
51         public RPGRecordTypeColorizer(String tagBefore, String tagAfter) {
52             this.tagBefore = tagBefore;
53             this.tagAfter = tagAfter;
54             this.matcher = Pattern.compile(DEFAULT_REGEX).matcher("");
55         }
56     
57         @Override
58         public boolean consume(CodeReader code, HtmlCodeBuilder codeBuilder) {
59             if (code.getColumnPosition() == COLUMN_INDEX
60                     && code.popTo(matcher, tmpBuilder) > 0) {
61                 codeBuilder.appendWithoutTransforming(tagBefore);
62                 codeBuilder.append(tmpBuilder);
63                 codeBuilder.appendWithoutTransforming(tagAfter);
64                 tmpBuilder.delete(0, tmpBuilder.length());
65                 return true;
66             }
67             return false;
68         }
69     
70         @SuppressWarnings("PMD")
71         @edu.umd.cs.findbugs.annotations.SuppressWarnings("CN_IMPLEMENTS_CLONE_BUT_NOT_CLONEABLE")
72         @Override
73         public RPGRecordTypeColorizer clone() {
74             // There's no super.clone() method in this case. Parent class is
75             // abstract and has abstract clone() method.
76     
77             RPGRecordTypeColorizer clone = new RPGRecordTypeColorizer(tagBefore, tagAfter);
78             return clone;
79         }
80     }
