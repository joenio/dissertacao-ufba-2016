30     package com.sourcemeter.analyzer.base.core;
31     
32     import java.util.Set;
33     import java.util.TreeSet;
34     
35     /**
36      * Class for storing SourceMeter license information for every SM tool on
37      * projects. License can be: full, limited, inactive.
38      */
39     public class LicenseInformation {
40     
41         public static final String FULL = "full";
42     
43         public static final String LIMITED = "limited";
44     
45         public static final String INACTIVE = "inactive";
46     
47         private final Set<String> full;
48     
49         private final Set<String> limited;
50     
51         private final Set<String> inactive;
52     
53         /**
54          * Creates an empty LicenseInformation.
55          */
56         public LicenseInformation() {
57             full = new TreeSet<String>();
58             limited = new TreeSet<String>();
59             inactive = new TreeSet<String>();
60         }
61     
62         /**
63          * A tool is added to the current license informations with the given
64          * license
65          *
66          * @param tool
67          * @param license
68          */
69         public void addTool(String tool, String license) {
70             if (FULL.equals(license)) {
71                 full.add(tool);
72             } else if (LIMITED.equals(license)) {
73                 limited.add(tool);
74             } else {
75                 inactive.add(tool);
76             }
77         }
78     
79         /**
80          * A tool is updated with the given license. This method can only grant
81          * higher value for the given tool, to merge more license informations into
82          * one.
83          *
84          * @param tool
85          * @param license
86          */
87         public void updateTool(String tool, String license) {
88             if (FULL.equals(license)) {
89                 full.add(tool);
90                 limited.remove(tool);
91                 inactive.remove(tool);
92             } else if (LIMITED.equals(tool) && !full.contains(tool)) {
93                 limited.add(tool);
94                 inactive.remove(tool);
95             } else if (!full.contains(tool) && !limited.contains(tool)) {
96                 inactive.add(tool);
97             }
98         }
99     
100         /**
101          * All tools in the given licenseInformation are updated with the given
102          * license. This method can only grant higher value for the given tools, to
103          * merge more license informations into one.
104          *
105          * @param tool
106          * @param license
107          */
108         public void updateLicenseInformations(LicenseInformation licenseInformation) {
109             for (String full : licenseInformation.getFull()) {
110                 updateTool(full, LicenseInformation.FULL);
111             }
112             for (String limited : licenseInformation.getLimited()) {
113                 updateTool(limited, LicenseInformation.LIMITED);
114             }
115             for (String inactive : licenseInformation.getInactive()) {
116                 updateTool(inactive, LicenseInformation.INACTIVE);
117             }
118         }
119     
120         /**
121          * Returns the Set of tools with full license.
122          *
123          * @return full license tools
124          */
125         public Set<String> getFull() {
126             return full;
127         }
128     
129         /**
130          * Returns a Set of tools with limited license.
131          *
132          * @return limited license tools
133          */
134         public Set<String> getLimited() {
135             return limited;
136         }
137     
138         /**
139          * Returns the Set of tools which were inactive during the analyzis.
140          *
141          * @return inactive tools
142          */
143         public Set<String> getInactive() {
144             return inactive;
145         }
146     }
