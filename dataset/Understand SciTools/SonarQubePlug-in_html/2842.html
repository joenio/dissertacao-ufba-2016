112     
113         private void addSourceMeterWidget(String name, int column, String metricKey) {
114             w = dashboard.addWidget(name, column);
115             w.setProperty(this.METRIC, metricKey);
116             w.setProperty(this.LIMIT, this.LIMIT_VALUE);
117         }
118     
119         private void addWidgetProperty(String... metrics) {
120             for (int i = 0; i < metrics.length; i++) {
121                 w.setProperty("metric" + (i + 1), metrics[i]);
122             }
123         }
124     
125         @Override
126         public String getName() {
127             return "SourceMeter";
128         }

130     }
