30     package com.sourcemeter.analyzer.rpg.helper;
31     
32     import com.sourcemeter.analyzer.base.helper.FileHelper;
33     
34     import java.io.File;
35     
36     import org.sonar.api.batch.fs.FilePredicate;
37     import org.sonar.api.batch.fs.FileSystem;
38     import org.sonar.api.batch.fs.InputFile;
39     
40     public class FileHelperRPG extends FileHelper {
41     
42         /**
43          * Returns the correct file path from the given path. If the path was a
44          * compile listing path, the method searches in the source folder for the
45          * original file.
46          * 
47          * @param path the path of the examined file
48          * @param fileSystem FileSystem object for the path to be searched on
49          * @return the generated RPG file's path if it's a spool path, otherwise the original path
50          */
51         public static String getCorrectedFilePath(String path, FileSystem fileSystem) {
52             FilePredicate predicate = fileSystem.predicates().hasType(
53                     InputFile.Type.MAIN);
54             Iterable<File> files = fileSystem.files(predicate);
55             for (File file : files) {
56                 if (file.getAbsolutePath().endsWith(path)) {
57                     return file.getAbsolutePath();
58                 }
59             }
60     
61             return path;
62         }
63     }
