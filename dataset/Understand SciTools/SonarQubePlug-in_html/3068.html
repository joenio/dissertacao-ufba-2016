58     public abstract class SourceMeterSensor implements Sensor {
59     
60         protected static final Logger LOG = LoggerFactory.getLogger(SourceMeterSensor.class);
61         protected static final String MS = " ms";
62     
63         protected final Settings settings;
64         protected String resultGraph = "";
65         protected String projectName = "";
66         protected boolean isIncrementalMode = false;
67         protected final ResourcePerspectives perspectives;
68         protected final ModuleFileSystem moduleFileSystem;
69         protected final FileSystem fileSystem;
70         private static final Map<String, String> activeInfoRules = new TreeMap<String, String>();
71     
72         public static Map<String, String> getActiveInfoRules() {
73             return activeInfoRules;
74         }
75     
76         /**
77          * Constructor: Use of IoC to get Settings
78          */
79         public SourceMeterSensor(ModuleFileSystem moduleFileSystem, FileSystem fileSystem, Settings settings,
80                 ResourcePerspectives perspectives, RulesProfile rulesProfile) {
81             this.moduleFileSystem = moduleFileSystem;
82             this.fileSystem = fileSystem;
83             this.perspectives = perspectives;
84             this.settings = settings;
85     
86             List<ActiveRule> activeRulesList = rulesProfile
87                     .getActiveRulesByRepository(SourceMeterRuleRepository.getRepositoryKey());
88             if (activeRulesList != null) {
89                 for (ActiveRule activeRule : activeRulesList) {
90                     if (Severity.INFO.equals(activeRule.getSeverity().name())) {
91                         activeInfoRules.put(activeRule.getRuleKey(), "");
92                     }
93                 }
94             }
95         }
96     
97         /**
98          * Load result graph binary
99          *
100          * @param filename
101          * @param project
102          * @param sensorContext
103          */
104         protected abstract void loadDataFromGraphBin(String filename,
105                 Project project, SensorContext sensorContext) throws GraphlibException;
106     
107         /**
108          * {@inheritDoc}
109          */
110         @Override
111         public abstract void analyse(Project project, SensorContext sensorContext);
112     
113         /**
114          * {@inheritDoc}
115          */
116         @Override
117         public boolean shouldExecuteOnProject(Project project) {
118             return this.fileSystem.hasFiles(fileSystem.predicates().hasLanguage(
119                     SourceMeterInitializer.getPluginLanguage().getKey()));
120         }
121     
122         @Override
123         public String toString() {
124             return getClass().getSimpleName();
125         }
126     
127         protected void saveLicense(Graph graph, SensorContext sensorContext,
128                 Map<String, String> headerLicenseInformations, Metric targetMetric) {
129             Gson gson = new Gson();
130     
131             LicenseInformation licenseInformation = new LicenseInformation();
132             for (Map.Entry<String, String> entry : headerLicenseInformations.entrySet()) {
133                 String modeInfo = entry.getKey() + "-mode";
134                 String value = graph.getHeaderInfo(modeInfo);
135     
136                 if (value == null) {
137                     continue;
138                 }
139     
140                 licenseInformation.addTool(entry.getValue(), value);
141             }
142     
143             sensorContext.saveMeasure(new Measure(targetMetric, gson.toJson(
144                     licenseInformation).toString()));
145         }
146     }
