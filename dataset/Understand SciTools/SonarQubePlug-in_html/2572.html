30     package com.sourcemeter.analyzer.cpp.profile;
31     
32     import java.io.InputStream;
33     import java.io.InputStreamReader;
34     import java.nio.charset.Charset;
35     
36     import org.apache.commons.io.IOUtils;
37     import org.sonar.api.profiles.ProfileDefinition;
38     import org.sonar.api.profiles.RulesProfile;
39     import org.sonar.api.profiles.XMLProfileParser;
40     import org.sonar.api.utils.ValidationMessages;
41     
42     public class SourceMeterCppProfile extends ProfileDefinition {
43     
44         private final XMLProfileParser parser;
45     
46         public SourceMeterCppProfile(XMLProfileParser parser) {
47             this.parser = parser;
48         }
49     
50         @Override
51         public RulesProfile createProfile(ValidationMessages validationMessages) {
52             InputStream input = getClass().getResourceAsStream(
53                     "/SourceMeter_way_default_profile.xml");
54             InputStreamReader reader = new InputStreamReader(input,
55                     Charset.defaultCharset());
56             try {
57                 RulesProfile profile = this.parser
58                         .parse(reader, validationMessages);
59                 profile.setDefaultProfile(true);
60                 return profile;
61             } finally {
62                 IOUtils.closeQuietly(reader);
63             }
64         }
65     }
