30     package com.sourcemeter.analyzer.java.visitor;

36     import java.io.File;

38     
39     import org.sonar.api.batch.SensorContext;

41     import org.sonar.api.component.ResourcePerspectives;
42     import org.sonar.api.config.Settings;
43     import org.sonar.api.resources.Project;
44     import org.sonar.api.resources.Resource;
45     import org.sonar.api.utils.SonarException;
46     
47     import com.sourcemeter.analyzer.base.helper.GraphHelper;
48     import com.sourcemeter.analyzer.base.visitor.LogicalTreeLoaderVisitor;
49     import com.sourcemeter.analyzer.java.core.resources.JavaClass;
50     import com.sourcemeter.analyzer.java.core.resources.JavaMethod;
51     import com.sourcemeter.analyzer.java.helper.VisitorHelperJava;
52     
53     public class LogicalTreeLoaderVisitorJava extends LogicalTreeLoaderVisitor {
54     
55         private final boolean uploadMethods;
56     
57         public LogicalTreeLoaderVisitorJava(FileSystem fileSystem, Settings settings,
58                 ResourcePerspectives perspectives, Project project, SensorContext sensorContext, long numOfNodes) {
59     
60             super(fileSystem, settings, perspectives, project, sensorContext, numOfNodes,
61                     new VisitorHelperJava(project, sensorContext, perspectives, fileSystem));
62     
63             this.uploadMethods = settings.getBoolean("sm.java.uploadMethods");
64         }
65     
66         /**
67          * {@inheritDoc}
68          */
69         @Override
70         public void edgeVisitorFunc(Edge e) {
71         }
72     
73         /**
74          * {@inheritDoc}
75          */
76         @Override
77         public void preNodeVisitorFunc(Node node) {
78             if (this.emptyProject) {
79                 return;
80             }
81     
82             String nodeName = GraphHelper.getNodeNameAttribute(node);
83             if (nodeName == null || "__LogicalRoot__".equals(nodeName)) {
84                 return;
85             }
86     
87             long startLogicalTime = System.currentTimeMillis();
88             String nodeLongName = GraphHelper.getNodeLongNameAttribute(node);
89             Position nodePosition = graphsupportlib.Metric.getFirstPositionAttribute(node);
90     
91             Resource resource = null;
92             Resource parentResource = null;
93             String nodeType = node.getType().getType();
94             String nodeTUID = GraphHelper.getNodeTUID(node);
95     
96             if (nodeTUID == null) {
97                 String warningMessage = "A node has no TUID attribute: "
98                         + nodeLongName + ", UID: " + node.getUID();
99     
100                 if (skipTUID) {
101                     LOG.warn(warningMessage);
102                     return;
103                 } else {
104                     throw new SonarException(warningMessage);
105                 }
106             }
107     
108             if (GraphHelper.isClass(node)) {
109                 resource = new JavaClass(nodeTUID, nodeName, nodeLongName,
110                         nodeType.toLowerCase(Locale.ENGLISH));
111                 parentResource = org.sonar.api.resources.File.fromIOFile(new File(nodePosition.path), this.project);
112     
113                 indexResource(node, resource, parentResource, nodePosition);
114             } else if (this.uploadMethods && "Method".equals(nodeType)) {
115                 Node parentNode = GraphHelper.getParentNode(node);
116                 if (GraphHelper.isClass(parentNode)) {
117                     String parentNodeType = parentNode.getType().getType().toLowerCase(Locale.ENGLISH);
118                     parentResource = new JavaClass(GraphHelper.getNodeTUID(parentNode),
119                             GraphHelper.getNodeNameAttribute(parentNode),
120                             GraphHelper.getNodeLongNameAttribute(parentNode),
121                             parentNodeType);
122                     resource = new JavaMethod(nodeTUID, nodeName, nodeLongName)
123                             .setParent(parentResource);
124     
125                     indexResource(node, resource, parentResource, nodePosition);
126                 }
127             }
128     
129             uploadMetricsAndWarnings(node, resource, nodePosition, true);
130     
131             this.logicalTime += (System.currentTimeMillis() - startLogicalTime);
132         }
133     }
