30     package com.sourcemeter.analyzer.base.batch.decorators;
31     
32     import com.sourcemeter.analyzer.base.batch.SourceMeterInitializer;
33     
34     import java.util.HashMap;
35     import java.util.HashSet;
36     import java.util.List;
37     import java.util.Map;
38     import java.util.Set;
39     
40     import org.sonar.api.batch.Decorator;
41     import org.sonar.api.batch.DecoratorContext;
42     import org.sonar.api.batch.Phase;
43     import org.sonar.api.measures.Measure;
44     import org.sonar.api.measures.Metric;
45     import org.sonar.api.measures.Metric.ValueType;
46     import org.sonar.api.resources.Project;
47     import org.sonar.api.resources.Qualifiers;
48     import org.sonar.api.resources.Resource;
49     import org.sonar.plugins.SourceMeterCore.api.SourceMeterCoreInitializer;
50     import org.sonar.plugins.SourceMeterCore.api.SourceMeterCoreMetrics;
51     
52     @Phase(name = Phase.Name.POST)
53     public class ProjectDecorator implements Decorator {
54     
55         private static final Map<Metric, Metric> weightedMetrics;
56         static {
57             weightedMetrics = new HashMap<>();
58             weightedMetrics.put(SourceMeterCoreMetrics.TAD, SourceMeterCoreMetrics.TLOC);
59             weightedMetrics.put(SourceMeterCoreMetrics.TCD, SourceMeterCoreMetrics.TCLOC);
60             weightedMetrics.put(SourceMeterCoreMetrics.TPDA, SourceMeterCoreMetrics.TLOC);
61             weightedMetrics.put(SourceMeterCoreMetrics.TPUA, SourceMeterCoreMetrics.TLOC);
62             weightedMetrics.put(SourceMeterCoreMetrics.CLC, SourceMeterCoreMetrics.TLOC);
63             weightedMetrics.put(SourceMeterCoreMetrics.CLLC, SourceMeterCoreMetrics.TLLOC);
64             weightedMetrics.put(SourceMeterCoreMetrics.CC, SourceMeterCoreMetrics.TLOC);
65             weightedMetrics.put(SourceMeterCoreMetrics.CCO, SourceMeterCoreMetrics.CI);
66             weightedMetrics.put(SourceMeterCoreMetrics.NCR, SourceMeterCoreMetrics.CCL);
67         }
68     
69         @Override
70         public boolean shouldExecuteOnProject(Project project) {
71             return  SourceMeterCoreInitializer.allDetectedLanguages
72                     .contains(SourceMeterInitializer.getPluginLanguage().getKey());
73         }
74     
75         @Override
76         public void decorate(Resource resource, DecoratorContext context) {
77             //execute only on projects, not modules
78             if (Qualifiers.isProject(resource, false)) {
79                 Set<Metric> allMetrics = new HashSet<>();
80                 List<Set<Measure>> measuresForLanguages = SourceMeterCoreInitializer.MEASURES_FOR_LANGUAGES;
81     
82                 // get all the metrics from each plugin's stored measures
83                 for (Set<Measure> measures : measuresForLanguages) {
84                     for (Measure measure : measures) {
85                         allMetrics.add(measure.getMetric());
86                     }
87                 }
88     
89                 for (Metric metric : allMetrics) {
90                     if (context.getMeasure(metric) != null) {
91                         continue;
92                     }
93                     if (weightedMetrics.containsKey(metric)
94                             || metric.getType() == ValueType.PERCENT) {
95                         avgTargetMetrics(context, metric, measuresForLanguages);
96                     } else {
97                         sumTargetMetrics(context, metric, measuresForLanguages);
98                     }
99                 }
100             }
101         }
102     
103         /**
104          * Sums the target metric from the given Sets and saves it to the context.
105          *
106          * @param context
107          * @param metric
108          * @param measuresForLanguages
109          */
110         private void sumTargetMetrics(DecoratorContext context, Metric metric,
111                 List<Set<Measure>> measuresForLanguages) {
112             int sum = 0;
113     
114             for (Set<Measure> measures : measuresForLanguages) {
115                 sum += (int) getMetricValueFromMeasures(measures, metric);
116             }
117     
118             Measure measure = new Measure(metric);
119             measure.setIntValue(sum);
120             context.saveMeasure(measure);
121         }
122     
123         /**
124          * Averages the target metric from the given Sets and saves it to the
125          * context. If a metric's average should be weighted, it's value is weighted
126          * by another metric.
127          *
128          * @param context
129          * @param metric
130          * @param measuresForLanguages
131          */
132         private void avgTargetMetrics(DecoratorContext context, Metric metric,
133                 List<Set<Measure>> measuresForLanguages) {
134             double avg = 0;
135             double sumWeightedByMetrics = 0.0;
136             boolean shouldBeWeighted = weightedMetrics.containsKey(metric);
137     
138             for (Set<Measure> measures : measuresForLanguages) {
139                 double currentValue = getMetricValueFromMeasures(measures, metric);
140                 if (shouldBeWeighted) {
141                     double weightBy = getMetricValueFromMeasures(measures,
142                             weightedMetrics.get(metric));
143                     if (weightBy > 0.0) {
144                         sumWeightedByMetrics += weightBy;
145                         currentValue *= weightBy;
146                     } else {
147                         sumWeightedByMetrics++;
148                     }
149                 } else {
150                     sumWeightedByMetrics++;
151                 }
152                 avg += currentValue;
153             }
154     
155             if (sumWeightedByMetrics > 0.0) {
156                 avg /= sumWeightedByMetrics;
157             }
158     
159             Measure measure = new Measure(metric, avg);
160             context.saveMeasure(measure);
161         }
162     
163         /**
164          * Searches for a metric in the given Set and returns it's value.
165          *
166          * @param measures
167          * @param metric
168          * @return the metric's value.
169          */
170         private double getMetricValueFromMeasures(Set<Measure> measures,
171                 Metric metric) {
172             double val = 0.0;
173             for (Measure measure : measures) {
174                 if (measure.getMetric().equals(metric)) {
175                     if (measure.getValue() != null) {
176                         val = measure.getValue();
177                     } else if (measure.getIntValue() != null) {
178                         val = measure.getIntValue();
179                     }
180                     break;
181                 }
182             }
183     
184             return val;
185         }
186     }
