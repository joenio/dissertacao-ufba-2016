53     /**
54      * Class for initializing MetricHunter\"s input file
55      */
56     public class ProfileInitializer {
57     
58         public static final String CLASS_TRESHOLD_VIOLATION_SUFFIX = "_warning_Class";
59         public static final String METHOD_TRESHOLD_VIOLATION_SUFFIX = "_warning_Method";
60         public static final String FUNCTION_TRESHOLD_VIOLATION_SUFFIX = "_warning_Function";
61         public static final String ClONE_CLASS_TRESHOLD_VIOLATION_SUFFIX = "_warning_CloneClass";
62         public static final String CLONE_INSTANCE_TRESHOLD_VIOLATION_SUFFIX = "_warning_CloneInstance";
63     
64         private final Settings settings;
65         private final List<MetricHunterCategory> categories;
66         private final Set<String> activeRuleKeys;
67         private final List<Rule> allRules;
68     
69         /**
70          * Sets the needed properties for generating a profile file for SourceMeter
71          * toolchain.
72          *
73          * @param settings for getting the metric threshold properties from SonarQube
74          * @param categories for setting which categories are passed to MetricHunter
75          * @param activeRules needed for the list of active rules passed to profile file
76          */
77         public ProfileInitializer(Settings settings,
78                 List<MetricHunterCategory> categories, RulesProfile profile,
79                 RuleRepository ruleRepository) {
80             this.settings = settings;
81             this.categories = categories;
82             this.allRules = ruleRepository.createRules();
83             String repositoryKey = ruleRepository.getKey();
84             activeRuleKeys = new HashSet<String>();
85             for (ActiveRule rule : profile.getActiveRulesByRepository(repositoryKey)) {
86                 activeRuleKeys.add(rule.getRuleKey());
87             }
88         }
89     
90         /**
91          * Generates a profile file for sourceMeter toolchain.
92          *
93          * @param path the generated profile file\"s path
94          * @throws IOException
95          */
96         public void generatePofileFile(String path) throws IOException {
97             File file = new File(path);
98     
99             if (!file.exists() && !file.createNewFile()) {
100                 throw new SonarException("Profile file could not be created: "
101                         + path);
102             }
103     
104             BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(
105                     new FileOutputStream(path), Charset.defaultCharset()));
106     
107             bw.write("<sourcemeter-profile>\n");
108             bw.write("  <tool-options>\n");
109             bw.write(getMetricHunterThresholds());
110             bw.write("  </tool-options>\n");
111             bw.write(getRuleOptions());
112             bw.write("</sourcemeter-profile>\n");
113             bw.close();
114         }
115     
116         private String getRuleOptions() {
117             StringBuffer buffer = new StringBuffer();
118             buffer.append("  <rule-options>\n");
119     
120             for (Rule rule : allRules) {
121                 if (rule.getSeverity() == RulePriority.INFO) {
122                     continue;
123                 }
124     
125                 String[] splittedKey = rule.getKey().split("_");
126                 String key = splittedKey[splittedKey.length - 1];
127     
128                 String priority = "Blocker";
129     
130                 // have to check the value for setting priority, RulePriority enum's
131                 // ordinal is not in order
132                 if (rule.getSeverity() == RulePriority.CRITICAL) {
133                     priority = "Critical";
134                 } else if (rule.getSeverity() == RulePriority.MAJOR) {
135                     priority = "Major";
136                 } else if (rule.getSeverity() == RulePriority.MINOR) {
137                     priority = "Minor";
138                 }
139     
140                 buffer.append("    <rule id=\"").append(key).append("\"  enabled=\"");
141                 if (activeRuleKeys.contains(rule.getKey())) {
142                     buffer.append("true");
143                 } else {
144                     buffer.append("false");
145                 }
146                 buffer.append("\" priority=\"").append(priority).append("\"");
147                 buffer.append("/>\n");
148             }
149     
150             buffer.append("  </rule-options>\n");
151     
152             return buffer.toString();
153         }
154     
155         private String getMetricHunterThresholds() throws IOException {
156             StringBuffer buffer = new StringBuffer();
157     
158             buffer.append("    <tool name = \"MetricHunter\" enabled = \"true\">\n");
159             buffer.append("      <metric-thresholds>\n");
160     
161             for (MetricHunterCategory category : categories) {
162                 List<Metric> metrics = category.getMetrics();
163                 for (Metric metric : metrics) {
164                     String line = getTresholdLine(metric,
165                             category.getCategoryName(), category.getpropertyName());
166                     if (!line.isEmpty()) {
167                         buffer.append(line);
168                     }
169                 }
170                 buffer.append("\n");
171             }
172     
173             buffer.append("      </metric-thresholds>\n");
174             buffer.append("    </tool>\n");
175             return buffer.toString();
176         }
177     
178         private String getTresholdLine(Metric metric, String entity, String property) {
179             StringBuffer baseline = new StringBuffer("sm.");
180             baseline.append(SourceMeterInitializer.getPluginLanguage().getKey().toLowerCase(Locale.ENGLISH));
181             baseline.append(".").append(property.toLowerCase(Locale.ENGLISH))
182                     .append(".baseline.").append(metric.getKey());
183             Double threshold = this.settings.getDouble(baseline.toString());
184             if (threshold == null) {
185                 return "";
186             }
187     
188             if (metric.getType() == ValueType.PERCENT) {
189                 threshold /= 100;
190             }
191     
192             String thresholdString = threshold.toString();
193     
194             if (threshold % 1 == 0) {
195                 thresholdString = Long.toString(Math.round(threshold));
196             }
197     
198             StringBuffer line = new StringBuffer();
199             String relation = "gt";
200     
201             if (metric.getDirection() > 0) {
202                 relation = "lt";
203             }
204     
205             line.append("        <threshold metric-id=\"").append(metric.getKey()).append("\" relation=\"").append(relation);
206             line.append("\" value=\"").append(thresholdString).append("\" entity=\"").append(entity).append("\" />\n");
207     
208             return line.toString();
209         }
