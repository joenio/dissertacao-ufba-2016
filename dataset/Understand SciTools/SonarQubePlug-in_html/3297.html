30     package com.sourcemeter.analyzer.base.visitor;

32     import graphlib.Attribute;
33     import graphlib.Node;
34     
35     import java.util.HashSet;
36     import java.util.List;
37     import java.util.Set;
38     
39     import org.sonar.api.batch.SensorContext;
40     import org.sonar.api.component.ResourcePerspectives;
41     import org.sonar.api.measures.Measure;
42     import org.sonar.api.resources.Project;
43     import org.sonar.api.resources.Resource;
44     import org.sonar.plugins.SourceMeterCore.api.SourceMeterCoreInitializer;
45     
46     import com.sourcemeter.analyzer.base.helper.GraphHelper;
47     import com.sourcemeter.analyzer.base.helper.VisitorHelper;
48     
49     /**
50      * Class for visiting and storing component nodes from the result graph.
51      */
52     public abstract class ComponentTreeLoaderVisitor extends BaseVisitor {
53     
54         protected long componentTime;
55         protected long numOfVisitedNodes;
56         protected final long numOfNodes;
57     
58         public ComponentTreeLoaderVisitor(ResourcePerspectives perspectives,
59                 Project project, SensorContext sensorContext, long numOfNodes,
60                 VisitorHelper visitorHelper) {
61             super(perspectives, visitorHelper);
62             this.project = project;
63             this.sensorContext = sensorContext;
64     
65             this.componentTime = 0;
66             this.numOfVisitedNodes = 0;
67             this.numOfNodes = numOfNodes;
68         }
69     
70         /**
71          * @return execution time.
72          */
73         public long getComponentTime() {
74             return this.componentTime;
75         }
76     
77         /**
78          * {@inheritDoc}
79          */
80         @Override
81         public void postNodeVisitorFunc(Node node) {
82             if (this.isDebugMode) {
83                 this.numOfVisitedNodes++;
84                 printProgress(this.numOfVisitedNodes, this.numOfNodes,
85                         this.componentTime);
86             }
87         }
88     
89         /**
90          * {@inheritDoc}
91          */
92         @Override
93         public void preNodeVisitorFunc(Node node) {
94             long startTime = System.currentTimeMillis();
95     
96             String nodeName = GraphHelper.getNodeNameAttribute(node);
97             String nodeType = node.getType().getType();
98     
99             if ("Component".equals(nodeType) && "<System>".equals(nodeName)) {
100                 Resource resource = this.project;
101                 createMetricsForProject(node, resource);
102                 this.componentTime += (System.currentTimeMillis() - startTime);
103             }
104         }
105     
106         /**
107          * Creates and stores metrics measures for the current language to be
108          * aggregated on project level.
109          *
110          * @param node
111          * @param resource
112          */
113         protected void createMetricsForProject(Node node, Resource resource) {
114             List<Attribute> attributes = node.getAttributes();
115             Set<Measure> measures = new HashSet<>();
116             for (Attribute attribute : attributes) {
117                 String context = attribute.getContext();
118                 if (resource != null
119                     && ("metric".equals(context) || "metricgroup".equals(context))) {
120     
121                     Measure measure = visitorHelper.createMeasureFromAttribute(attribute, resource);
122                     if (measure != null) {
123                         measures.add(measure);
124                     }
125                 }
126             }
127     
128             // put the Set for the current language's metrics
129             SourceMeterCoreInitializer.MEASURES_FOR_LANGUAGES.add(measures);
130         }
131     }
