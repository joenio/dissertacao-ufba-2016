59     /**
60      * Visitor class for reading clone informations from the result graph.
61      */
62     public abstract class CloneTreeLoaderVisitor extends BaseVisitor {
63     
64         private final Map<Resource, Set<String>> duplicationsMap = new HashMap<Resource, Set<String>>();
65     
66         private long cloneTime;
67     
68         private long numOfVisitedNodes;
69         private final long numOfNodes;
70         private boolean emptyProject = false;
71         private final VisitorHelper visitorHelper;
72     
73         /**
74          * Initializes a CloneTreeLoaderVisitor by all necessary attributes
75          *
76          * @param fileSystem
77          * @param perspectives
78          * @param project
79          * @param sensorContext
80          * @param numOfNodes
81          * @param visitorHelper
82          * @param countThresholdViolations
83          */
84         public CloneTreeLoaderVisitor(FileSystem fileSystem,
85                 ResourcePerspectives perspectives, Project project,
86                 SensorContext sensorContext, long numOfNodes,
87                 VisitorHelper visitorHelper) {
88             super(perspectives, visitorHelper);
89             this.project = project;
90             this.sensorContext = sensorContext;
91             this.visitorHelper = visitorHelper;
92     
93             this.cloneTime = 0;
94             this.numOfVisitedNodes = 0;
95             this.numOfNodes = numOfNodes;
96     
97             FilePredicate mainFilePredicate = fileSystem.predicates().hasType(InputFile.Type.MAIN);
98     
99             if (!fileSystem.hasFiles(mainFilePredicate)) {
100                 this.emptyProject = true;
101             }
102         }
103     
104         /**
105          * Different toolchains can have different keys. This method should return
106          * the used key in every analyzer plugin.
107          *
108          * @param node
109          * @return
110          */
111         protected abstract String getKeyFromNode(Node node);
112     
113         /**
114          * @return Duplications container0
115          */
116         public Map<Resource, Set<String>> getDuplicationsMap() {
117             return this.duplicationsMap;
118         }
119     
120         /**
121          * @return execution time.
122          */
123         public long getCloneTime() {
124             return this.cloneTime;
125         }
126     
127         /**
128          * Clone instance to string
129          *
130          * @param node
131          * @param sensorContext
132          * @param projectKey
133          * @return Pair of the containing class key and clone instance in xml format
134          */
135         private SimpleEntry<org.sonar.api.resources.File, String> cloneInstanceToString(
136                 Node node, SensorContext sensorContext, Project project) {
137             SimpleEntry<org.sonar.api.resources.File, String> result = null;
138             String projectKey = project.getKey();
139             org.sonar.api.resources.File fileResource = org.sonar.api.resources.File
140                     .fromIOFile(new File(visitorHelper.getPathFromNode(node)), this.project);
141     
142             if (fileResource != null) {
143                 Position pos = graphsupportlib.Metric.getFirstPositionAttribute(node);
144                 String value = "<b c=\"" + projectKey + ":" + getKeyFromNode(node)
145                         + "\" s=\"" + pos.line + "\" l=\""
146                         + (pos.endline - pos.line)
147                         + "\" r=\"" + projectKey + ":" + fileResource.getKey() + "\"/>";
148                 result = new SimpleEntry<org.sonar.api.resources.File, String>(fileResource, value);
149             }
150             return result;
151         }
152     
153         /**
154          * Check Clone class or instance is Disappearing clone smell or not.
155          *
156          * @param node (CloneClass or CloneInstance)
157          * @return true if disappearing
158          */
159         private boolean isDisappearing(Node node) {
160             AttributeString cloneSmellType = (AttributeString) graphsupportlib.Metric.getNodeAttribute(node, "CloneSmellType");
161             if (cloneSmellType != null && "cstDisappearing".equals(cloneSmellType.getValue())) {
162                 return true;
163             }
164     
165             return false;
166         }
167     
168         /**
169          * Collect clone instances of a CloneClass node
170          *
171          * @param node
172          * @param project
173          * @param sensorContext
174          *
175          * @return first CI node
176          */
177         private Node collectCloneClass(Node node, Project project, SensorContext sensorContext) {
178             List<Resource> resources = new ArrayList<Resource>();
179             List<Node> nodes = graphsupportlib.Metric.getNodesByEdgeType(node, "CloneTree", eDirectionType.edtDirectional);
180             StringBuffer duplicationsData = new StringBuffer("<g c=\"" + project.getKey() + ":" + getKeyFromNode(node) + "\">");
181     
182             for (Node n : nodes) {
183                 if (n != null && !isDisappearing(n)) {
184                     SimpleEntry<org.sonar.api.resources.File, String> pair = cloneInstanceToString(n, sensorContext, project);
185                     if (pair != null) {
186                         duplicationsData.append(pair.getValue());
187                         resources.add(pair.getKey());
188                     }
189                 }
190             }
191             duplicationsData.append("</g>");
192             for (Resource resource : resources) {
193                 Set<String> list = this.duplicationsMap.get(resource);
194                 if (list == null) {
195                     list = new HashSet<String>();
196                 }
197                 list.add(duplicationsData.toString());
198                 this.duplicationsMap.put(resource, list);
199             }
200     
201             if (!nodes.isEmpty()) {
202                 return nodes.get(0);
203             } else {
204                 return null;
205             }
206         }

208         /**
209          * {@inheritDoc}
210          */
211         @Override
212         public void postNodeVisitorFunc(Node node) {
213             if (this.isDebugMode) {
214                 this.numOfVisitedNodes++;
215                 printProgress(this.numOfVisitedNodes, this.numOfNodes, this.cloneTime);
216             }
217         }
218     
219         /**
220          * {@inheritDoc}
221          */
222         @Override
223         public void preNodeVisitorFunc(Node node) {
224             if (this.emptyProject) {
225                 return;
226             }
227     
228             long startTime = System.currentTimeMillis();
229             String nodeName = GraphHelper.getNodeNameAttribute(node);
230             Position nodePosition = graphsupportlib.Metric.getFirstPositionAttribute(node);
231             if (null != nodePosition) {
232                 nodePosition.path = visitorHelper.getPathFromNode(node);
233             }
234     
235             Resource resource = null;
236             org.sonar.api.resources.File parentResource = null;
237             String nodeType = node.getType().getType();
238     
239             boolean disappearing = isDisappearing(node);
240             if ("CloneClass".equals(nodeType) && !disappearing) {
241                 Node ci = collectCloneClass(node, this.project, this.sensorContext);
242                 parentResource = org.sonar.api.resources.File.fromIOFile(
243                         new File(visitorHelper.getPathFromNode(ci)), this.project);
244                 if (parentResource != null) {
245                     resource = new CloneClass(getKeyFromNode(node), nodeName, parentResource);
246                     this.sensorContext.index(resource);
247                 } else {
248                     return;
249                 }
250             } else if ("CloneInstance".equals(nodeType) && !disappearing && nodePosition != null) {
251                 parentResource = org.sonar.api.resources.File.fromIOFile(new File(nodePosition.path), this.project);
252                 if (parentResource != null) {
253                     resource = new CloneInstance(getKeyFromNode(node), nodeName, parentResource);
254                     this.sensorContext.index(resource);
255                 } else {
256                     return;
257                 }
258             } else {
259                 return;
260             }
261     
262             uploadMetricsAndWarnings(node, resource, nodePosition, false);
263             this.cloneTime += (System.currentTimeMillis() - startTime);
264         }
265     }
