30     package com.sourcemeter.analyzer.base.core.resources;
31     
32     import org.sonar.api.resources.File;
33     
34     public abstract class BaseMethod extends BaseResource {
35     
36         private static final long serialVersionUID = 6435646875613564768L;
37         public static final String DEFAULT_METHOD_DESCRIPTION = "method";
38         public static final String DEFAULT_FUNCTION_DESCRIPTION = "function";
39     
40         /**
41          * Initialize method resource with only a key for the resource.
42          *
43          * @param key
44          */
45         public BaseMethod(String key) {
46             super(key);
47             this.setDescription(DEFAULT_METHOD_DESCRIPTION);
48         }
49     
50         /**
51          * Initialize a method resource with key, name and parent attributes
52          *
53          * @param key
54          * @param name
55          * @param parent
56          */
57         public BaseMethod(String key, String fileName, String longName,
58                 String description) {
59             super(key, fileName, longName);
60             this.setDescription(description);
61         }
62     
63         /**
64          * Initialize a method resource with key, name and parent attributes
65          *
66          * @param key
67          * @param name
68          * @param parent
69          */
70         public BaseMethod(String key, String name, File parent) {
71             super(key, name, parent);
72             this.setDescription(DEFAULT_METHOD_DESCRIPTION);
73         }
74     
75     }
