20     package com.sourcemeter.analyzer.cpp.colorizer;
21     
22     import com.sourcemeter.analyzer.cpp.core.Cpp;
23     
24     import java.util.ArrayList;
25     import java.util.List;
26     
27     import org.sonar.api.web.CodeColorizerFormat;
28     import org.sonar.colorizer.CDocTokenizer;
29     import org.sonar.colorizer.CppDocTokenizer;
30     import org.sonar.colorizer.KeywordsTokenizer;
31     import org.sonar.colorizer.LiteralTokenizer;
32     import org.sonar.colorizer.RegexpTokenizer;
33     import org.sonar.colorizer.Tokenizer;
34     
35     /**
36      * {@inheritDoc}
37      */
38     public final class CppColorizer extends CodeColorizerFormat {
39     
40       private static final String SPAN = "</span>";
41     
42       /**
43        * {@inheritDoc}
44        */
45       public CppColorizer() {
46             super(Cpp.KEY);
47       }
48     
49       @Override
50       public List<Tokenizer> getTokenizers() {
51         List<Tokenizer> tokenizers = new ArrayList<Tokenizer>();
52         tokenizers.add(new CDocTokenizer("<span class=\"p\">", SPAN)); // C style comments
53         tokenizers.add(new CppDocTokenizer("<span class=\"p\">", SPAN)); // C++ style comments
54         tokenizers.add(new KeywordsTokenizer("<span class=\"k\">", SPAN, CppKeywords.keywordValues())); // keywords
55         tokenizers.add(new LiteralTokenizer("<span class=\"s\">", SPAN)); // strings, characters
56         tokenizers.add(new RegexpTokenizer("<span class=\"h\">", SPAN, "#[^\\n\\r]*+")); // preprocessor directives
57         tokenizers.add(new RegexpTokenizer("<span class=\"c\">", SPAN, "[+-]?[0-9]+[xX]?+(\\.[0-9]*+)?")); // numbers
58         return tokenizers;
59       }
60     }
