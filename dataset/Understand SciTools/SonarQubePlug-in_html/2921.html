78                 FileSystem fileSystem, ResourcePerspectives perspectives,

88         @Override
89         public void analyse(Project module, SensorContext context) {
90             String analyseMode = this.settings.getString("sonar.analysis.mode");
91             this.projectName = settings.getString("sonar.projectKey");
92             this.projectName = StringUtils.replace(projectName, ":", "_");
93     
94             if ("incremental".equals(analyseMode)) {
95                 LOG.warn("Incremental mode is on. There are no metric based (INFO level) issues in this mode.");
96                 this.isIncrementalMode = true;
97             }
98     
99             this.resultGraph = FileHelper.getSMSourcePath(settings, fileSystem, '-')
100                     + File.separator + projectName + ".graph";
101     
102             long startTime = System.currentTimeMillis();
103             LOG.info("      Graph: " + resultGraph);
104     
105             try {
106                 loadDataFromGraphBin(this.resultGraph, project, sensorContext);
107             } catch (GraphlibException e) {
108                 throw new SonarException("Error during graph loading!", e);
109             }
110     
111             LOG.info("    Load data from graph bin and save resources and metrics done: " + (System.currentTimeMillis() - startTime) + MS);
112         }

114         /**
115          * Load result graph binary
116          *
117          * @param filename
118          * @param project
119          * @param sensorContext
120          * @throws GraphlibException
121          */
122         @Override
123         protected void loadDataFromGraphBin(String filename, Project project, SensorContext sensorContext) throws GraphlibException {
124             Graph graph = new Graph();
125             graph.loadBinary(filename);
126     
127             saveLicense(graph, sensorContext);
128     
129             Node componentRoot = null;
130             ComponentTreeLoaderVisitorCpp componentVisitor = null;
131             NodeCounterVisitor nodeCounter = null;
132     
133             List<Node> components = graph.findNodes(new NodeType("Component"));
134             for (Node component : components) {
135                 String name = GraphHelper.getNodeNameAttribute(component);
136                 if (name != null && name.equals("<System>")) {
137                     componentRoot = component;
138                     break;
139                 }
140             }
141     
142             try {
143                 LOG.info("      * Initialization...");
144                 long startTime = System.currentTimeMillis();
145     
146                 if (componentRoot != null) {
147                     nodeCounter = new NodeCounterVisitor();
148                     GraphHelper.processGraph(graph, componentRoot, "ComponentTree", nodeCounter);
149                     componentVisitor = new ComponentTreeLoaderVisitorCpp(
150                             this.perspectives, this.project,
151                             this.sensorContext, this.fileSystem,
152                             nodeCounter.getNumberOfNodes());
153                 }
154     
155                 nodeCounter = new NodeCounterVisitor();
156                 GraphHelper.processGraph(graph, "__LogicalRoot__", "LogicalTree", nodeCounter);
157                 LogicalTreeLoaderVisitorCpp logicalVisitor = new LogicalTreeLoaderVisitorCpp(
158                         this.fileSystem, this.settings, this.perspectives, project,
159                         sensorContext, nodeCounter.getNumberOfNodes());
160     
161                 nodeCounter = new NodeCounterVisitor();
162                 GraphHelper.processGraph(graph, "__PhysicalRoot__", "PhysicalTree", nodeCounter);
163                 PhysicalTreeLoaderVisitorCpp physicalVisitor = new PhysicalTreeLoaderVisitorCpp(
164                         fileSystem, this.perspectives, this.project,
165                         this.sensorContext, nodeCounter.getNumberOfNodes());
166     
167                 LOG.info("      * Initialization done: " + (System.currentTimeMillis() - startTime) + MS);
168     
169                 if (componentVisitor != null) {
170                     LOG.info("      * Start processing ComponentTree...");
171                     GraphHelper.processGraph(graph, componentRoot, "ComponentTree", componentVisitor);
172                     LOG.info("      * ComponentTree processing time: " + componentVisitor.getComponentTime() + MS);
173                     componentVisitor = null;
174                 }
175     
176                 LOG.info("      * Processing LogicalTree...");
177                 GraphHelper.processGraph(graph, "__LogicalRoot__", "LogicalTree", logicalVisitor);
178                 saveClassesInFilesMetric(logicalVisitor.getClassesInFiles(), SourceMeterCppMetrics.CLASSES_IN_FILE);
179                 saveClassesInFilesMetric(logicalVisitor.getMethodsInFiles(), SourceMeterCppMetrics.METHODS_IN_FILE);
180                 saveClassesInFilesMetric(logicalVisitor.getFilePathsForClasses(), SourceMeterCppMetrics.FILE_PATHS);
181                 saveClassesInFilesMetric(logicalVisitor.getFilePathsForMethods(), SourceMeterCppMetrics.FILE_PATHS);
182                 LOG.info("      * Processing LogicalTree done: " + logicalVisitor.getLogicalTime() + MS);
183                 logicalVisitor = null;
184     
185                 LOG.info("      * Processing PhysicalTree...");
186                 GraphHelper.processGraph(graph, "__PhysicalRoot__", "PhysicalTree", physicalVisitor);
187                 LOG.info("      * Processing PhysicalTree done: " + physicalVisitor.getFileTime() + MS);
188                 physicalVisitor = null;
189     
190                 CloneTreeLoaderVisitorCpp cloneVisitor = null;
191                 if (!this.isIncrementalMode) {
192                     nodeCounter = new NodeCounterVisitor();
193                     GraphHelper.processGraph(graph, "__CloneRoot__", "CloneTree", nodeCounter);
194                     cloneVisitor = new CloneTreeLoaderVisitorCpp(this.fileSystem,
195                             this.perspectives, project,
196                             sensorContext, nodeCounter.getNumberOfNodes());
197                 }
198     
199                 if (!this.isIncrementalMode) {
200                     LOG.info("      * Start processing CloneTree...");
201                     GraphHelper.processGraph(graph, "__CloneRoot__", "CloneTree", cloneVisitor);
202                     LOG.info("      * CloneTree processing time: " + cloneVisitor.getCloneTime() + MS);
203     
204                     // Save duplications
205                     startTime = System.currentTimeMillis();
206                     LOG.info("    Save duplications...");
207     
208                     Iterator<Entry<Resource, Set<String>>> dupIt = cloneVisitor.getDuplicationsMap().entrySet().iterator();
209                     while (dupIt.hasNext()) {
210                         Map.Entry<Resource, Set<String>> pairs = dupIt.next();
211                         Set<String> set = pairs.getValue();
212                         StringBuffer tmp = new StringBuffer("<duplications>");
213                         for (String cloneClass : set) {
214                             tmp.append(cloneClass);
215                         }
216                         tmp.append("</duplications>");
217                         sensorContext.saveMeasure(pairs.getKey(), new Measure(CoreMetrics.DUPLICATIONS_DATA, tmp.toString()));
218                         dupIt.remove(); // avoids a ConcurrentModificationException
219                     }
220                     LOG.info("    Save duplications done: " + (System.currentTimeMillis() - startTime) + MS);
221                     cloneVisitor = null;
222                 }
223             } catch (VisitorException e) {
224                 throw (SonarException) new SonarException(e.getMessage()).initCause(e);
225             } finally {
226                 graph = null;
227             }
228         }

230         private void saveLicense(Graph graph, SensorContext sensorContext) {
231             Map<String, String> headerLicenseInformations = new HashMap<String, String>();
232             headerLicenseInformations.put("FaultHunterCPP", "FaultHunter");
233             headerLicenseInformations.put("MetricHunter", "MetricHunter");
234             headerLicenseInformations.put("DuplicatedCodeFinder", "Duplicated Code");
235             headerLicenseInformations.put("LIM2Metrics", "Metrics");
236             headerLicenseInformations.put("Cppcheck2Graph", "CPPCheck");
237     
238             super.saveLicense(graph, sensorContext, headerLicenseInformations, SourceMeterCppMetrics.CPP_LICENSE);
239         }

241         /**
242          * Saves ClassData list for resources to the given metric.
243          *
244          * @param classData
245          * @param metric
246          */
247         private void saveClassesInFilesMetric(Map<Resource, List<ClassData>> classData, Metric metric) {
248             Gson gson = new Gson();
249             Iterator<Entry<Resource, List<ClassData>>> iterator = classData.entrySet().iterator();
250             while (iterator.hasNext()) {
251                 Entry<Resource, List<ClassData>> entry = iterator.next();
252                 String jsonClassesList = gson.toJson(entry.getValue()).toString();
253                 Measure measure = new Measure(metric);
254                 measure.setData(jsonClassesList);
255                 this.sensorContext.saveMeasure(entry.getKey(), measure);
256             }
257         }
