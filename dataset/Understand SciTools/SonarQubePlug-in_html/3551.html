30     package com.sourcemeter.analyzer.base.batch.decorators;
31     
32     import com.sourcemeter.analyzer.base.batch.SourceMeterInitializer;
33     import com.sourcemeter.analyzer.base.core.resources.BaseMethod;
34     
35     import org.sonar.api.batch.Decorator;
36     import org.sonar.api.batch.DecoratorContext;
37     import org.sonar.api.batch.fs.FileSystem;
38     import org.sonar.api.measures.CoreMetrics;
39     import org.sonar.api.measures.Measure;
40     import org.sonar.api.measures.PersistenceMode;
41     import org.sonar.api.measures.RangeDistributionBuilder;
42     import org.sonar.api.resources.Project;
43     
44     public abstract class BaseDecorator implements Decorator {
45     
46         private static final Number[] FUNCTIONS_DISTRIB_BOTTOM_LIMITS = {1, 2, 4, 6, 8, 10, 12, 20, 30};
47         private static final Number[] FILES_DISTRIB_BOTTOM_LIMITS = {0, 5, 10, 20, 30, 60, 90};
48     
49         private final FileSystem fileSystem;
50     
51         public BaseDecorator(FileSystem fileSystem) {
52             this.fileSystem = fileSystem;
53         }
54     
55         @Override
56         public boolean shouldExecuteOnProject(Project project) {
57             return fileSystem.hasFiles(fileSystem.predicates().hasLanguage(
58                     SourceMeterInitializer.getPluginLanguage().getKey()));
59         }
60     
61         protected void saveFunctionsComplexityDistribution(DecoratorContext context) {
62             RangeDistributionBuilder complexityDistribution = new RangeDistributionBuilder(
63                     CoreMetrics.FUNCTION_COMPLEXITY_DISTRIBUTION,
64                     FUNCTIONS_DISTRIB_BOTTOM_LIMITS);
65     
66             // FUNCTION_COMPLEXITY_DISTRIBUTION metric is aggregated automatically
67             // for parent resources therefore uploading is only needed for functions
68             if (!(context.getResource() instanceof BaseMethod)) {
69                 return;
70             }
71             Measure complexity = context.getMeasure(CoreMetrics.COMPLEXITY);
72             if (complexity != null) {
73                 complexityDistribution.add(complexity.getValue());
74                 context.saveMeasure(complexityDistribution.build().setPersistenceMode(PersistenceMode.MEMORY));
75             }
76         }
77     
78         protected void saveFilesComplexityDistribution(DecoratorContext context) {
79             RangeDistributionBuilder complexityDistribution = new RangeDistributionBuilder(
80                     CoreMetrics.FILE_COMPLEXITY_DISTRIBUTION,
81                     FILES_DISTRIB_BOTTOM_LIMITS);
82             Measure measure = context.getMeasure(CoreMetrics.COMPLEXITY);
83             if (measure != null) {
84                 complexityDistribution.add(measure.getValue());
85                 context.saveMeasure(complexityDistribution.build().setPersistenceMode(PersistenceMode.MEMORY));
86             }
87         }
88     }
