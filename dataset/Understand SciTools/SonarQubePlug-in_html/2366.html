30     package com.sourcemeter.analyzer.cpp.batch.decorators;
31     
32     import org.sonar.api.batch.DecoratorContext;
33     import org.sonar.api.batch.Phase;
34     import org.sonar.api.batch.fs.FileSystem;
35     import org.sonar.api.measures.CoreMetrics;
36     import org.sonar.api.measures.Measure;
37     import org.sonar.api.resources.Resource;
38     import org.sonar.plugins.SourceMeterCore.api.SourceMeterCoreMetrics;
39     
40     import com.sourcemeter.analyzer.base.batch.decorators.FileMetricsDecorator;
41     import com.sourcemeter.analyzer.base.core.resources.BaseResource;
42     import com.sourcemeter.analyzer.base.helper.DecoratorHelper;
43     import com.sourcemeter.analyzer.cpp.SourceMeterCppMetrics;
44     import com.sourcemeter.analyzer.cpp.visitor.LogicalTreeLoaderVisitorCpp;
45     
46     @Phase(name = Phase.Name.POST)
47     public class FileMetricsDecoratorCpp extends FileMetricsDecorator {
48     
49         public FileMetricsDecoratorCpp(FileSystem fileSystem) {
50             super(fileSystem);
51         }
52     
53         @Override
54         protected void decorateMetricsWithHelper(Resource resource,
55                                                  DecoratorContext context,
56                                                  DecoratorHelper helper) {
57             Measure functionsMeasure = context.getMeasure(CoreMetrics.FUNCTIONS);
58             if (functionsMeasure == null) {
59                 functionsMeasure = new Measure(CoreMetrics.FUNCTIONS);
60             }
61             double numOfFunctions = 0.0;
62             if (LogicalTreeLoaderVisitorCpp.FUNCTIONS_FOR_FILES.get(resource) != null) {
63                 numOfFunctions = LogicalTreeLoaderVisitorCpp.FUNCTIONS_FOR_FILES.get(resource);
64             }
65             functionsMeasure.setValue(numOfFunctions);
66             context.saveMeasure(functionsMeasure);
67     
68             helper.copyMetricResult(SourceMeterCppMetrics.LLOC, CoreMetrics.NCLOC);
69             helper.copyMetricResult(SourceMeterCoreMetrics.LOC, CoreMetrics.LINES);
70             helper.copyMetricResult(SourceMeterCppMetrics.CLOC, CoreMetrics.COMMENT_LINES);
71             helper.copyMetricResult(SourceMeterCoreMetrics.NOS, CoreMetrics.STATEMENTS);
72             helper.sumAndCopyMeasures(CoreMetrics.PUBLIC_API, SourceMeterCoreMetrics.PDA, SourceMeterCoreMetrics.PUA);
73             helper.copyMetricResult(SourceMeterCoreMetrics.PUA, CoreMetrics.PUBLIC_UNDOCUMENTED_API);
74     
75             Measure publicDocAPIDensityMeasure = context.getMeasure(CoreMetrics.PUBLIC_DOCUMENTED_API_DENSITY);
76             if (publicDocAPIDensityMeasure == null) {
77                 publicDocAPIDensityMeasure = new Measure(CoreMetrics.PUBLIC_DOCUMENTED_API_DENSITY);
78             }
79             Measure publicAPIMeasure = context.getMeasure(CoreMetrics.PUBLIC_API);
80             Measure publicDocAPIMeasure = context.getMeasure(SourceMeterCoreMetrics.PDA);
81             if (publicAPIMeasure != null
82                 && publicDocAPIMeasure != null
83                 && publicAPIMeasure.getIntValue() > 0) {
84     
85                 publicDocAPIDensityMeasure.setValue((publicDocAPIMeasure.getValue() / publicAPIMeasure.getValue()) * 100);
86             }
87     
88             Measure commentLinesDensityMeasure = context.getMeasure(CoreMetrics.COMMENT_LINES_DENSITY);
89             if (commentLinesDensityMeasure == null) {
90                 commentLinesDensityMeasure = new Measure(CoreMetrics.COMMENT_LINES_DENSITY);
91             }
92             Measure clocMeasure = context.getMeasure(SourceMeterCppMetrics.CLOC);
93             Measure llocMeasure = context.getMeasure(SourceMeterCppMetrics.LLOC);
94             if (clocMeasure != null && llocMeasure != null) {
95                 int cloc = clocMeasure.getIntValue();
96                 int lloc = llocMeasure.getIntValue();
97                 if ((cloc + lloc > 0)) {
98                     commentLinesDensityMeasure.setValue((clocMeasure.getValue() / (llocMeasure.getValue() + clocMeasure.getValue())) * 100);
99                 } else {
100                     LOG.warn("Cannot calculate comment lines density [CLOC: " + cloc + ", LLOC: " + lloc + "]");
101                 }
102             }
103             helper.sumChildClassMeasures(CoreMetrics.ACCESSORS);
104     
105             helper.sumChildCloneClassMeasures(SourceMeterCoreMetrics.CLLOC);
106             helper.sumChildCloneClassMeasures(SourceMeterCoreMetrics.CI);
107             helper.copyMetricResult(SourceMeterCoreMetrics.CLLOC, CoreMetrics.DUPLICATED_LINES);
108             helper.copyMetricResult(SourceMeterCoreMetrics.CI, CoreMetrics.DUPLICATED_BLOCKS);
109     
110             helper.avgChildMeasures(CoreMetrics.FUNCTION_COMPLEXITY, BaseResource.class);
111             helper.avgChildClassMeasuresAndCreateMeasure(CoreMetrics.CLASS_COMPLEXITY);
112         }
113     
114         @Override
115         protected DecoratorHelper getDecoratorHelper(DecoratorContext context) {
116             return new DecoratorHelper(context);
117         }
118     
119         @Override
120         protected void decorateComplexity(DecoratorHelper helper) {
121             helper.copyMetricResult(SourceMeterCoreMetrics.McCC, CoreMetrics.COMPLEXITY);
122         }
123     }
