20     package com.sourcemeter.analyzer.python.colorizer;
21     
22     import com.sonar.sslr.api.AstNode;
23     import com.sonar.sslr.api.TokenType;
24     
25     /**
26      * http://docs.python.org/reference/lexical_analysis.html#keywords
27      */
28     public enum PythonKeyword implements TokenType {
29     
30       FALSE("False"),
31       NONE("None"),
32       TRUE("True"),
33     
34       /**
35        * Logical AND.
36        */
37       AND("and"),
38     
39       AS("as"),
40       ASSERT("assert"),
41       BREAK("break"),
42       CLASS("class"),
43       CONTINUE("continue"),
44       DEF("def"),
45       DEL("del"),
46       ELIF("elif"),
47       ELSE("else"),
48       EXCEPT("except"),
49       FINALLY("finally"),
50       FOR("for"),
51       FROM("from"),
52       GLOBAL("global"),
53       IF("if"),
54       IMPORT("import"),
55       IN("in"),
56       IS("is"),
57       LAMBDA("lambda"),
58       NONLOCAL("nonlocal"),
59       NOT("not"),
60     
61       /**
62        * Logical OR.
63        */
64       OR("or"),
65     
66       PASS("pass"),
67       RAISE("raise"),
68       RETURN("return"),
69       TRY("try"),
70       WHILE("while"),
71       WITH("with"),
72       YIELD("yield");
73     
74       private final String value;
75     
76       private PythonKeyword(String value) {
77         this.value = value;
78       }
79     
80       public String getName() {
81         return name();
82       }
83     
84       public String getValue() {
85         return value;
86       }
87     
88       public boolean hasToBeSkippedFromAst(AstNode node) {
89         return false;
90       }
91     
92       public static String[] keywordValues() {
93         PythonKeyword[] keywordsEnum = PythonKeyword.values();
94         String[] keywords = new String[keywordsEnum.length];
95         for (int i = 0; i < keywords.length; i++) {
96           keywords[i] = keywordsEnum[i].getValue();
97         }
98         return keywords;
99       }
100     
101     }
