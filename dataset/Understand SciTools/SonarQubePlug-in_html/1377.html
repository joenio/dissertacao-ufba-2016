30     package com.sourcemeter.analyzer.rpg.colorizer;
31     
32     import com.sourcemeter.analyzer.rpg.core.RPG;
33     
34     import java.util.List;
35     
36     import org.sonar.api.web.CodeColorizerFormat;
37     import org.sonar.colorizer.KeywordsTokenizer;
38     import org.sonar.colorizer.RegexpTokenizer;
39     import org.sonar.colorizer.Tokenizer;
40     
41     import com.google.common.collect.Lists;
42     
43     public final class RPGColorizer extends CodeColorizerFormat {
44     
45         private List<Tokenizer> tokenizers;
46         private static final String END_TAG = "</span>";
47         private static final String K_CLASS_START_TAG = "<span class=\"k\">";
48         private static final String S_CLASS_START_TAG = "<span class=\"s\">";
49     
50         public RPGColorizer() {
51             super(RPG.KEY);
52         }
53     
54         @Override
55         public List<Tokenizer> getTokenizers() {
56             if (tokenizers == null) {
57                 tokenizers = Lists.newArrayList();
58     
59                 tokenizers.add(new RPGRecordTypeColorizer(K_CLASS_START_TAG, END_TAG));
60                 tokenizers.add(new RPGCommentTokenizer("<span class=\"cd\">", END_TAG));
61                 tokenizers.add(new RPGKeywordsTokenizer(K_CLASS_START_TAG, END_TAG));
62                 tokenizers.add(new RPGDocStringTokenizer(S_CLASS_START_TAG,END_TAG));
63     
64                 KeywordsTokenizer defaultKeywordsTokenizer = new KeywordsTokenizer(
65                         K_CLASS_START_TAG, END_TAG, RPGKeyword.keywordValuesSet());
66                 defaultKeywordsTokenizer.setCaseInsensitive(true);
67     
68                 tokenizers.add(defaultKeywordsTokenizer);
69                 tokenizers.add(new RegexpTokenizer(K_CLASS_START_TAG, END_TAG,
70                         "[*]{1}[a-zA-Z0-9_\\-]*+"));
71                 tokenizers.add(new RegexpTokenizer(S_CLASS_START_TAG, END_TAG,
72                         " \\d+[a-zA-Z_\\-]*"));
73             }
74             return tokenizers;
75         }
76     
77     }
