30     package com.sourcemeter.analyzer.rpg.batch.decorators;
31     
32     import com.sourcemeter.analyzer.base.batch.decorators.FileMetricsDecorator;
33     import com.sourcemeter.analyzer.base.helper.DecoratorHelper;
34     import com.sourcemeter.analyzer.rpg.core.RPG;
35     
36     import org.slf4j.Logger;
37     import org.slf4j.LoggerFactory;
38     import org.sonar.api.batch.DecoratorContext;
39     import org.sonar.api.batch.Phase;
40     import org.sonar.api.batch.fs.FileSystem;
41     import org.sonar.api.measures.CoreMetrics;
42     import org.sonar.api.measures.Measure;
43     import org.sonar.api.resources.Qualifiers;
44     import org.sonar.api.resources.Resource;
45     import org.sonar.api.utils.SonarException;
46     import org.sonar.plugins.SourceMeterCore.api.SourceMeterCoreMetrics;
47     
48     @Phase(name = Phase.Name.POST)
49     public class FileMetricsDecoratorRPG extends FileMetricsDecorator {
50     
51         private static final Logger LOG = LoggerFactory.getLogger(FileMetricsDecoratorRPG.class);
52     
53     
54         public FileMetricsDecoratorRPG(FileSystem fileSystem) {
55             super(fileSystem);
56         }
57     
58         @Override
59         public void decorate(Resource resource, DecoratorContext context) {
60             DecoratorHelper helper = getDecoratorHelper(context);
61             try {
62                 if (Qualifiers.isFile(resource) && RPG.KEY.equals(resource.getLanguage().getKey())) {
63                     Measure smResourceMeasure = new Measure(SourceMeterCoreMetrics.SM_RESOURCE);
64                     smResourceMeasure.setValue(1.0);
65                     context.saveMeasure(smResourceMeasure);
66     
67                     decorateMetricsWithHelper(resource, context, helper);
68                     saveFunctionsComplexityDistribution(context);
69                     saveFilesComplexityDistribution(context);
70                 }
71             } catch (SonarException e) {
72                 LOG.error("Cannot decorate file metrics! Cause: " + e.getMessage());
73             }
74         }
75     
76         @Override
77         protected void decorateMetricsWithHelper(Resource resource,
78                 DecoratorContext context, DecoratorHelper helper) {
79             helper.sumChildClassMeasures(SourceMeterCoreMetrics.TLLOC);
80             helper.copyMetricResult(SourceMeterCoreMetrics.TLLOC, CoreMetrics.NCLOC);
81             helper.sumChildClassMeasures(SourceMeterCoreMetrics.TLOC);
82             helper.copyMetricResult(SourceMeterCoreMetrics.TLOC, CoreMetrics.LINES);
83     
84             helper.sumChildClassMeasures(CoreMetrics.STATEMENTS);
85             helper.sumChildClassMeasures(CoreMetrics.ACCESSORS);
86     
87             helper.sumChildClassMeasures(CoreMetrics.COMMENT_LINES);
88             helper.avgChildClassMeasures(CoreMetrics.COMMENT_LINES_DENSITY);
89             helper.sumChildClassMeasures(CoreMetrics.PUBLIC_API);
90             helper.sumChildClassMeasures(CoreMetrics.PUBLIC_UNDOCUMENTED_API);
91             helper.avgChildClassMeasures(CoreMetrics.PUBLIC_DOCUMENTED_API_DENSITY);
92     
93             helper.sumChildClassMeasures(CoreMetrics.DUPLICATED_BLOCKS);
94             helper.sumChildClassMeasures(CoreMetrics.DUPLICATED_LINES);
95             helper.avgChildClassMeasures(CoreMetrics.DUPLICATED_LINES_DENSITY);
96     
97             helper.avgChildClassMeasuresAndCreateMeasure(CoreMetrics.FUNCTION_COMPLEXITY);
98         }
99     
100         @Override
101         protected DecoratorHelper getDecoratorHelper(DecoratorContext context) {
102             return new DecoratorHelper(context);
103         }
104     }
