103         /**
104          * {@inheritDoc}
105          */
106         @Override
107         public boolean shouldExecuteOnProject(Project project) {
108             // This needs to be set, whether or not the current initializer is
109             // needed to run on the current language.
110             SourceMeterCoreInitializer.updateDetectedLanguages(fileSystem);
111     
112             boolean shouldExecute = false;
113     
114             // On top level of a module analyzation, SM toolchain
115             // does not need to run.
116             if (!getSourcesDirectoriesForProject().isEmpty()) {
117                 if (pluginLanguage.getKey().equals(this.settings.getString("sonar.language"))) {
118                     shouldExecute = true;
119                 } else {
120                     shouldExecute = SourceMeterCoreInitializer.allDetectedLanguages
121                             .contains(pluginLanguage.getKey());
122                 }
123     
124                 // if the actual plugin is needed to be executed,
125                 // the actual language is an SM specific language.
126                 if (shouldExecute) {
127                     SourceMeterCoreInitializer.SM_SPECIFIC_LANGUAGES.add(pluginLanguage.getKey());
128                 }
129             }
130     
131             return shouldExecute;
132         }

143         /**
144          * Creates file exclusions for not analyzed files (e.g.
145          *
146          * @throws IOException
147          * @throws GraphlibException
148          * @throws VisitorException
149          */
150         protected void createExcludesFromNotAnalyzedFiles(char dateSeparator)
151                 throws IOException, GraphlibException, VisitorException {
152             String resultsDir = FileHelper.getSMSourcePath(settings, fileSystem, dateSeparator);
153             String projectName = settings.getString("sonar.projectKey");
154             projectName = StringUtils.replace(projectName, ":", "_");
155             String resultGraph = resultsDir + File.separator + projectName + ".graph";
156     
157             Graph graph = new Graph();
158             graph.loadBinary(resultGraph);
159     
160             IncludedFilesVisitor fileVisitor = new IncludedFilesVisitor(fileSystem);
161             GraphHelper.processGraph(graph, "__PhysicalRoot__", "PhysicalTree", fileVisitor);
162     
163             Set<String> includedFiles = fileVisitor.getIncludedFiles();
164             Set<String> excludedFiles = new HashSet<String>();
165     
166             FilePredicate predicate = this.fileSystem.predicates().hasType(InputFile.Type.MAIN);
167             Iterable<File> files = this.fileSystem.files(predicate);
168             String baseDir = this.fileSystem.baseDir().getCanonicalPath()
169                     .replace("\\", "/").toLowerCase(Locale.getDefault());
170             for (File file : files) {
171                 String path = file.getAbsolutePath().replace("\\", "/");
172                 String lowerCasePath = path.toLowerCase(Locale.getDefault());
173                 if (pluginLanguage.isFileForCurrentLanguage(file) && !includedFiles.contains(lowerCasePath)) {
174                     String excludedPath = "**" + path.substring(baseDir.length(), path.length());
175                     excludedFiles.add(excludedPath);
176                 }
177             }
178     
179             if (!excludedFiles.isEmpty()) {
180                 String[] exclusions = (String[]) ArrayUtils.addAll(
181                         this.settings.getStringArray("sonar.exclusions"),
182                         excludedFiles.toArray());
183                 this.settings.setProperty("sonar.exclusions", exclusions);
184             }
185         }

187         /**
188          * Run SourceMeter toolchain
189          *
190          * @throws IOException
191          * @throws InterruptedException
192          * @throws SonarException
193          */
194         protected void runSourceMeter(List<String> commands) throws SonarException {
195             LOG.debug("    Calling SourceMeter toolchain:");
196             StringBuffer commandLineOptions = new StringBuffer();
197             for (String command : commands) {
198                 commandLineOptions.append(command + " ");
199             }
200             if (LOG.isDebugEnabled()) {
201                 LOG.debug("        " + commandLineOptions.toString());
202             }
203             ProcessBuilder sourceMeterProcessBuilder = new ProcessBuilder(commands);
204             sourceMeterProcessBuilder.directory(this.fileSystem.baseDir());
205     
206             Process sourceMeterProcess = null;
207             BufferedReader br = null;
208             try {
209                 sourceMeterProcess = sourceMeterProcessBuilder.start();
210     
211                 br = new BufferedReader(new InputStreamReader(
212                         sourceMeterProcess.getInputStream(),
213                         Charset.defaultCharset()));
214                 StringBuilder smLogContent = new StringBuilder();
215                 String line = null;
216                 while ((line = br.readLine()) != null) {
217                     smLogContent.append(line);
218                     smLogContent.append(System.getProperty("line.separator"));
219                 }
220                 sourceMeterProcess.waitFor();
221     
222                 String logFilePath = this.fileSystem.workDir().getAbsolutePath()
223                         + File.separator + "sourcemeter.log";
224                 File smLogFile = new File(logFilePath);
225                 if (!smLogFile.exists() && !smLogFile.createNewFile()) {
226                     throw new SonarException(
227                         "SourceMeter log file could not be created: "
228                         + logFilePath);
229                 }
230     
231                 BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(
232                         new FileOutputStream(smLogFile.getAbsolutePath()),
233                         Charset.defaultCharset()));
234                 bw.write(smLogContent.toString());
235                 bw.close();
236     
237                 if (sourceMeterProcess.exitValue() != 0) {
238                     LOG.error("SourceMeter toolchain could not be executed properly. Please check the log file for more information: "
239                             + smLogFile.getAbsolutePath());
240                     throw new SonarException();
241                 }
242             } catch (IOException | InterruptedException e) {
243                 LOG.error("An exception occured while trying to run SourceMeter toolchain. Perhaps the permissions were not set correctly. Please check the trace for more information.", e);
244                 throw new SonarException(e);
245             } finally {
246                 if (br != null) {
247                     try {
248                         br.close();
249                     } catch (IOException e) {
250                         LOG.error("Could not close opened resource!", e);
251                     }
252                 }
253             }
254         }

256         /**
257          * List of source directories for project.
258          *
259          * @return source directories list
260          */
261         protected List<File> getSourcesDirectoriesForProject() {
262             List<File> sourceDirectories = new ArrayList<File>();
263     
264             for (String sourcePath : this.projectDefinition.getSourceDirs()) {
265                 File file = pathResolver.relativeFile(this.projectDefinition.getBaseDir(), sourcePath);
266                 if (file.exists() && file.isDirectory()) {
267                     sourceDirectories.add(file);
268                 }
269             }
270     
271             return sourceDirectories;
272         }

274         private String writeFilterToFile(String filter, String fileName) throws IOException {
275             String workDir = this.fileSystem.workDir().getCanonicalPath();
276             if (workDir.endsWith("\\.") || workDir.endsWith("/.")) {
277                 workDir = workDir.substring(0, workDir.length() - 2);
278             }
279             String filterFilePath = workDir + File.separator + fileName;
280             File filterFile = new File(filterFilePath);
281             if (!filterFile.exists() && !filterFile.createNewFile()) {
282                 throw new SonarException("Filter file could not be created: "
283                         + filterFilePath);
284             }
285     
286             BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(
287                     new FileOutputStream(filterFile.getAbsolutePath()),
288                     Charset.defaultCharset()));
289             bw.write(filter);
290             bw.close();
291     
292             return filterFilePath;
293         }

319         /**
320          * Assemble filter file's content by exclusions.
321          *
322          * @return filter file's content
323          * @throws IOException
324          */
325         protected String getFilterContent() throws IOException {
326             StringBuffer filter = new StringBuffer("-.*\n");
327             FileExclusions fe = new FileExclusions(this.settings);
328             if (ArrayUtils.isNotEmpty(fe.sourceExclusions()) || ArrayUtils.isNotEmpty(fe.sourceInclusions()) || isIncrementalMode) {
329                 FilePredicate predicate = this.fileSystem.predicates().hasType(InputFile.Type.MAIN);
330                 Iterable<File> files = this.fileSystem.files(predicate);
331                 for (File file : files) {
332                     String filePath = file.getCanonicalPath();
333                     filter.append("+");
334                     filter.append(Pattern.quote(filePath));
335                     filter.append("\n");
336                 }
337             } else {
338                 List<File> sourceDirectories = getSourcesDirectoriesForProject();
339                 for (File file : sourceDirectories) {
340                     String dirPath = file.getCanonicalPath();
341                     filter.append("+" + Pattern.quote(dirPath + File.separator) + "\n");
342                 }
343             }
344     
345             return filter.toString();
346         }

348         /**
349          * Assemble a filter file by only using the source exclusions.
350          *
351          * @return filter file's content
352          */
353         protected String getHardFilterForSourceExclusions() {
354             StringBuffer filter = new StringBuffer();
355             FileExclusions fe = new FileExclusions(this.settings);
356     
357             for (String exclusion : fe.sourceExclusions()) {
358                 filter.append("-").append(exclusion).append("\n");
359             }
360     
361             return filter.toString();
362         }

364         /**
365          * Removes "\." or "/." suffixes from file paths.
366          *
367          * @param path
368          * @return corrected path
369          */
370         protected String getCorrectedFilePath(String path) {
371             if (OSValidator.isWindows()) {
372                 path = path.substring(0, 2).toLowerCase(Locale.ENGLISH) + path.substring(2, path.length());
373             }
374     
375             if (path.endsWith("\\.") || path.endsWith("/.")) {
376                 path = path.substring(0, path.length() - 2);
377             }
378     
379             return path;
380         }
