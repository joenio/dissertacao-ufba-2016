30     package com.sourcemeter.analyzer.rpg.core;
31     
32     import java.util.Arrays;
33     
34     import org.apache.commons.lang.ArrayUtils;
35     import org.apache.commons.lang.StringUtils;
36     
37     import com.sourcemeter.analyzer.base.core.AbstractSMLanguage;
38     
39     public class RPG extends AbstractSMLanguage {
40     
41         public static final RPG INSTANCE = new RPG();
42     
43         /**
44          * RPG key
45          */
46         public static final String KEY = "rpg";
47     
48         /**
49          * RPG name
50          */
51         public static final String NAME = "RPG";
52     
53         /**
54          * RPG files knows suffixes
55          */
56         private static String[] suffixes = { ".rpg", ".rpgle" };
57     
58         /**
59          * Default constructor
60          */
61         public RPG() {
62             super(KEY, NAME);
63         }
64     
65         /**
66          * {@inheritDoc}
67          *
68          * @see org.sonar.api.resources.AbstractLanguage#getFileSuffixes()
69          */
70         @Override
71         public String[] getFileSuffixes() {
72             return Arrays.copyOf(suffixes, suffixes.length);
73         }
74     
75         @Override
76         public boolean isFileForCurrentLanguage(java.io.File file) {
77             String suffix = "." + StringUtils.substringAfterLast(file.getName(), ".");
78             return ArrayUtils.contains(suffixes, suffix);
79         }
80     
81         public static void removeSuffixesForCurrentAnalyze() {
82             suffixes = new String[1];
83         }
84     
85         public static void updateSuffixes(String[] suffixes) {
86             System.arraycopy(suffixes, 0, RPG.suffixes, 0, suffixes.length);
87         }
88     
89     }
