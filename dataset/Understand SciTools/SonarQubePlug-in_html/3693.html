30     package org.sonar.plugins.SourceMeterCore.api;
31     
32     import java.util.ArrayList;
33     import java.util.Collection;
34     import java.util.List;
35     import java.util.Map;
36     import java.util.TreeMap;
37     
38     import org.sonar.api.measures.Metric;
39     import org.sonar.api.measures.MetricFinder;
40     
41     /**
42      * Metric finder class for get SourceMeter Metric objects
43      */
44     public abstract class SourceMeterMetricFinder implements MetricFinder {
45     
46         protected final Map<String, Metric> metricMap = new TreeMap<String, Metric>();
47     
48         /**
49          * Default Constructor
50          */
51         public SourceMeterMetricFinder() {
52             SourceMeterCoreMetrics cm = new SourceMeterCoreMetrics();
53             List<Metric> metricList = cm.getMetrics();
54             for (Metric metric : metricList) {
55                 metricMap.put(metric.getKey(), metric);
56             }
57         }
58     
59         /**
60          * Returns the language specific Ruleset metrics.
61          * 
62          * @return
63          */
64         public abstract List<Metric> findLanguageSpecificRulesetMetrics();
65     
66         /**
67          * Find a SourceMeter Metric by ID
68          * 
69          * @param id
70          * @return Metric
71          */
72         @Override
73         public Metric findById(int id) {
74             Collection<Metric> metrics = metricMap.values();
75             for (Metric metric : metrics) {
76                 if (metric.getId() == id) {
77                     return metric;
78                 }
79             }
80             return null;
81         }
82     
83         /**
84          * Find a SourceMeter Metric by key (example: "SM:LOC")
85          * 
86          * @param key (short name)
87          * @return Metric
88          */
89         @Override
90         public Metric findByKey(String key) {
91             return metricMap.get(key);
92         }
93     
94         /**
95          * Find a list of SourceMeter metrics by keys
96          * 
97          * @param List of metric keys
98          * @return Collection of Metrics
99          */
100         @Override
101         public Collection<Metric> findAll(List<String> metricKeys) {
102             List<Metric> retValue = new ArrayList<Metric>();
103             for (String key : metricKeys) {
104                 Metric tmp = findByKey(key);
105                 if (tmp != null) {
106                     retValue.add(tmp);
107                 }
108             }
109             return retValue;
110         }
111     
112         /**
113          * Gives back all of the SourceMeter metrics
114          * 
115          * @return Collection of Metrics
116          */
117         @Override
118         public Collection<Metric> findAll() {
119             return metricMap.values();
120         }
121     }
