30     package com.sourcemeter.analyzer.base.profile;
31     
32     import com.sourcemeter.analyzer.base.batch.SourceMeterInitializer;
33     
34     import java.io.InputStream;
35     import java.util.List;
36     import java.util.Locale;
37     
38     import org.apache.commons.io.IOUtils;
39     import org.sonar.api.rules.Rule;
40     import org.sonar.api.rules.RuleRepository;
41     import org.sonar.api.rules.XMLRuleParser;
42     
43     public abstract class SourceMeterRuleRepository extends RuleRepository {
44     
45         public static final String BASE_REPOSITORY_KEY = "SourceMeter_";
46     
47         private final XMLRuleParser ruleParser;
48     
49         /**
50          * Initializes a SourceMeterRuleRepository. Rules are parsed by an
51          * XMLRuleParser. Language is needed to determine which SourceMeter plugin's
52          * rules are parsed.
53          * 
54          * @param ruleParser
55          * @param language
56          */
57         public SourceMeterRuleRepository(XMLRuleParser ruleParser, String language) {
58             super(BASE_REPOSITORY_KEY + language, language
59                     .toLowerCase(Locale.ENGLISH));
60             this.ruleParser = ruleParser;
61         }
62     
63         @Override
64         public List<Rule> createRules() {
65             InputStream input = getClass().getResourceAsStream("/rules.xml");
66             try {
67                 return this.ruleParser.parse(input);
68             } finally {
69                 IOUtils.closeQuietly(input);
70             }
71         }
72     
73         /**
74          * Returns the repositoryKey of the current RuleRepository
75          * 
76          * @return repositoryKey
77          */
78         public static String getRepositoryKey() {
79             return BASE_REPOSITORY_KEY
80                     + SourceMeterInitializer.getPluginLanguage().getKey();
81         }
82     }
