48          * "resource" : [resource_key], "trace" : [{ "line" : [line], "sourcelink" :
49          * [{ "resource" : [resource_key], "line" : [line] }, ... { "resource" :
50          * [resource_key], "line" : [line] }], "issue-key" : [issue-key] }, ... {
51          * ... }] }
52          */
53         public static final String TRACE_KEY = "SM:stacktrace";
54         public static final Metric TRACE = new Builder(TRACE_KEY, "Stacktrace",
55                 ValueType.DATA)
56                 .setHidden(true)
57                 .create();
