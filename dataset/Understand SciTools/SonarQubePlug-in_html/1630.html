61     public class SourceMeterCppInitializer extends SourceMeterInitializer {
62     
63         private static final Logger LOG = LoggerFactory.getLogger(SourceMeterCppInitializer.class);
64     
65         /**
66          * Command and parameters for running SourceMeter C++ analyzer
67          */
68         private final List<String> commands;
69     
70         public SourceMeterCppInitializer(FileSystem fileSystem, Settings settings,
71                 ProjectDefinition projectDefinition, PathResolver pathResolver,
72                 RulesProfile profile) {
73             super(fileSystem, settings, projectDefinition, pathResolver, profile);
74             this.commands = new ArrayList<String>();
75             SourceMeterInitializer.updatePluginLanguage(Cpp.INSTANCE);
76         }
77     
78         @Override
79         public boolean shouldExecuteOnProject(Project project) {
80             boolean skipCpp = this.settings.getBoolean("sm.cpp.skip");
81             if (skipCpp) {
82                 Cpp.removeSuffixesForCurrentAnalyze();
83                 return false;
84             }
85     
86             Cpp.updateSuffixes(settings);
87             return super.shouldExecuteOnProject(project);
88         }
89     
90         @Override
91         public void execute(Project project) {
92             this.settings.setProperty(CoreProperties.CPD_SKIP_PROPERTY, true); // Disable built in CPD plugin
93     
94             if (this.settings.getBoolean("sm.cpp.skipToolchain")) {
95                 LOG.info("    SourceMeter toolchain is skipped for C++. Results will be uploaded from former results directory, if it exists.");
96             } else {
97                 if (!checkProperties()) {
98                     throw new SonarException("Cannot run SourceMeterPlugin!");
99                 }
100     
101                 LOG.info("    Running SourceMeter toolchain...");
102                 long startTime = System.currentTimeMillis();
103     
104                 runSourceMeter(this.commands);
105     
106                 LOG.info("    Running SourceMeter toolchain done: "
107                         + (System.currentTimeMillis() - startTime) + " ms");
108             }
109     
110             try {
111                 long startTime = System.currentTimeMillis();
112                 LOG.info("    Creating exclusions for analyzed files...");
113                 createExcludesFromNotAnalyzedFiles('-');
114                 LOG.info("    Creating exclusions done: "
115                         + (System.currentTimeMillis() - startTime) + " ms");
116             } catch (IOException | GraphlibException | VisitorException e) {
117                 LOG.warn("Could not create exlcusions from result graph! All cpp files are indexed by SonarQube automatically.", e);
118             }
119         }
120     
121         /**
122          * Check for property settings
123          *
124          * @return true if every property set
125          */
126         private boolean checkProperties() {
127             String pathToCA = this.settings.getString("sm.toolchaindir");
128             if (pathToCA == null) {
129                 LOG.error("C++ SourceMeter path must be set! Check it on the settings page of your SonarQube!");
130                 return false;
131             }
132     
133             String pathToBuild = this.settings.getString("sm.cpp.buildfile");
134             if (pathToBuild == null) {
135                 LOG.error("Build script path must be set! (sm.cpp.buildfile)");
136                 return false;
137             }
138     
139             String resultsDir = this.settings.getString("sm.resultsdir");
140             if (resultsDir == null) {
141                 LOG.error("Results directory must be set! Check it on the settings page of your SonarQube!");
142                 return false;
143             }
144     
145             String cleanResults = this.settings.getString("sm.cleanresults");
146             String projectName = this.settings.getString("sonar.projectKey");
147             projectName = StringUtils.replace(projectName, ":", "_");
148     
149             String softFilter = "";
150             String softFilterFilePath = null;
151             String hardFilter = this.settings.getString("sm.cpp.hardFilter");
152             String additionalParameters = this.settings.getString("sm.cpp.toolchainOptions");
153     
154             try {
155                 softFilter = getFilterContent();
156                 softFilterFilePath = writeSoftFilterToFile(softFilter);
157             } catch (IOException e) {
158                 LOG.warn("Cannot create softFilter file for toolchain! No softFilter is used during analyzis.", e);
159             }
160     
161             ProfileInitializer profileInitializer = new ProfileInitializer(
162                     this.settings, getMetricHunterCategories(), profile,
163                     new SourceMeterCppRuleRepository(new XMLRuleParser()));
164     
165             String profilePath = this.fileSystem.workDir() + File.separator
166                     + "SM-Profile.xml";
167     
168             this.commands.add(pathToCA + File.separator
169                     + Cpp.KEY.toUpperCase(Locale.ENGLISH)
170                     + File.separator + "SourceMeterCPP");
171     
172             try {
173                 profileInitializer.generatePofileFile(profilePath);
174                 this.commands.add("-profileXML=" + profilePath);
175             } catch (IOException e) {
176                 LOG.warn("An error occured while creating SourceMeter profile file. Default profile is used!!", e);
177             }
178     
179             String baseDir = "";
180             try {
181                 baseDir = this.fileSystem.baseDir().getCanonicalPath();
182             } catch (IOException e) {
183                 LOG.warn("Could not get base directory's canonical path. Absolute path is used.");
184                 baseDir = this.fileSystem.baseDir().getAbsolutePath();
185             }
186     
187             // Setting command and parameters for SourceMeter C++ analyzer
188             this.commands.add("-resultsDir=" + resultsDir);
189             this.commands.add("-projectName=" + projectName);
190             this.commands.add("-projectBaseDir=" + baseDir);
191             this.commands.add("-buildScript=" + pathToBuild);
192             this.commands.add("-cleanResults=" + cleanResults);
193             this.commands.add("-runChangeTracker=true");
194     
195             String cloneGenealogy = this.settings.getString("sm.cloneGenealogy");
196             String cloneMinLines = this.settings.getString("sm.cloneMinLines");
197             this.commands.add("-cloneGenealogy=" + cloneGenealogy);
198             this.commands.add("-cloneMinLines=" + cloneMinLines);
199     
200             if (null != softFilterFilePath) {
201                 this.commands.add("-externalSoftFilter=" + softFilterFilePath);
202             }
203     
204             if (null != hardFilter) {
205                 this.commands.add("-externalHardFilter=" + hardFilter);
206             }
207     
208             if (null != additionalParameters) {
209                 this.commands.add(additionalParameters);
210             }
211     
212             return true;
213         }
214     
215         @Override
216         protected List<MetricHunterCategory> getMetricHunterCategories() {
217             List<MetricHunterCategory> categories = new ArrayList<MetricHunterCategory>();
218     
219             categories.add(new MetricHunterCategory("Class", SourceMeterCppMetrics
220                     .getClassThresholdMetrics()));
221     
222             categories.add(new MetricHunterCategory("Method",
223                     SourceMeterCppMetrics.getMethodThresholdMetrics()));
224             categories.add(new MetricHunterCategory("Function", "method",
225                     SourceMeterCppMetrics.getFunctionThresholdMetrics()));
226     
227             categories.addAll(super.getMetricHunterCategories());
228     
229             return categories;
230         }
