6     package org.homeunix.wap.table.tainted;
7     
8     import org.homeunix.wap.table.symbol.Symbol;
9     import java.util.*;
10     
11     /**
12      *
13      * @author Iberia Medeiros
14      */
15     public class VariableTaintedSymbol extends Symbol{
16         // criacao do mapa de variaveis/funcoes tainted a inserir nos mapas das relacoes
17         List<Integer> linesVars; // Lista que armazenara as linhas de vars, fun tainteds
18         List<String> filesVars; // Lista que armazenara as ficheiros a que pertencem as vars, fun tainteds
19     
20         /* CONSTRUCTOR */
21         public VariableTaintedSymbol(String name, int line, int tainted, String file){
22             super(name, tainted, line, file);
23             linesVars = new ArrayList<Integer>();
24             filesVars = new ArrayList<String>();
25             linesVars.add(line);
26             filesVars.add(file);
27         }
28     
29     
30         /* METHODS of the class*/
31     
32         /*
33          * Insert a code line in List of the tainted variable included in RelatedTaintedSymbol Map.
34          * This appened when dependent tainted variable is identified, and your code line
35          * is included in List
36          */
37         public void InsertLine (int line){
38             linesVars.add(line);
39         }
40     
41         /*
42          * Give the List of line codes from the variable included in RelatedTainted Map
43          */
44         public List getVariableLines (){
45             return linesVars;
46         }    
47     
48         /*
49          * Insert name of file in List of the tainted variable included in RelatedTaintedSymbol Map.
50          * This appened when dependent tainted variable is identified, and your file name
51          * is included in List
52          */
53         public void InsertFile (String file){
54             filesVars.add(file);
55         }
56     
57         /*
58          * Give the List of files from the variable included in RelatedTainted Map
59          */
60         public List getVariableFiles (){
61             return filesVars;
62         }
63     }
