86                 // Wrap Weka classifier in bridge 
87                 Classifier javaml_lr = new WekaClassifier(lr);

103                 // Wrap Weka classifier in bridge 
104                 Classifier javaml_rt = new WekaClassifier(rt);
105                 
106                 // Build the classifier         
107                 javaml_rt.buildClassifier(data);

109             }
110

112             if (nameClassifier.equals("jrip") == true){
113                 jrip = new weka.classifiers.rules.JRip();
114                 jrip.setCheckErrorRate(true);
115                 jrip.setDebug(false);
116                 jrip.setFolds(3);
117                 jrip.setMinNo(2.0);
118                 jrip.setOptimizations(2);
119                 jrip.setSeed(1);
