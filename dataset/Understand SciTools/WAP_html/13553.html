124         private void haveSQL(String codeline, List inp, collect_atts c_i){
125             int found = 0;
126             boolean is_num = false;
127             String upper = codeline.toUpperCase();
128             String sql_key;
129             int i, index, j = 0, ind_where = -1, ind_from = -1, ii;
130             Iterator <String> it = GlobalDataSqli.SQLKeywords.listIterator();
131             for (;it.hasNext();){
132                 sql_key = it.next();
133                 if (upper.contains(sql_key) == true){
134                     if (sql_key.equals("WHERE") == true)
135                         ind_where = upper.indexOf(sql_key);
136                     if (sql_key.equals("FROM") == true)
137                         ind_from = upper.indexOf(sql_key);
138                     j++;
139                 }
140             }
141             
142             if (j > 1){ // Para index 0, 1 e 2
143                 // Para index 0 e 1
144                 for (Map.Entry<String, Integer> sql_att : GlobalDataDatamining.SQLAttributes.entrySet()){
145                     if (upper.contains(sql_att.getKey()) == true){
146                         j = upper.indexOf(sql_att.getKey()) + sql_att.getKey().length();
147                         i = -1;
148                         for (ii = 0; ii < inp.size(); ii++){
149                             i = codeline.indexOf((String)inp.get(ii));
150                             if (i > j){
151                                 if (i < ind_from && sql_att.getKey().equals("FROM") == false){ // aggregated functions
152                                     index = sql_att.getValue();
153                                     this.setValueOfAttribute(index);
154                                     if (c_i.who.isEmpty())
155                                         c_i.who += sql_att.getKey();
156                                     else
157                                         c_i.who += ", " + sql_att.getKey();
158                                     found = 1;
159                                 }
160                                 
161                                 // FROM
162                                 if (sql_att.getKey().equals("FROM") == true && (i < ind_where || ind_where == -1)){
163                                     index = sql_att.getValue();
164                                     this.setValueOfAttribute(index);
165                                     if (c_i.who.isEmpty())
166                                         c_i.who += sql_att.getKey();
167                                     else
168                                         c_i.who += ", " + sql_att.getKey();                                
169                                     found = 1;                            
170                                 }
171                                     
172                             }
173                             
174                         }
175                     }
176                 }
177                 
178                 // Para index 2            
179                 if (ind_where > -1){
180                     i = -1;
181                     int iii = -1;
182                     String input;
183                     for (ii = 0; ii < inp.size(); ii++){
184                         input = (String)inp.get(ii);

186

191                                 String substr_before = codeline.substring(i-6, i-1);
192                                 String substr_after = codeline.substring(iii+1, iii+6);
193                                 it = GlobalDataDatamining.SQLOperators_aritm.listIterator();

195                                     sql_key = it.next();
196                                     if (substr_before.contains(sql_key) == true || substr_after.contains(sql_key) == true){
197                                         this.setValueOfAttribute(2);
198                                         found = 1;
199                                         is_num = true;
200                                         if (c_i.who.isEmpty())
201                                             c_i.who += "is_numeric_field";
202                                         else
203                                             c_i.who += ", is_numeric_field";                                    
204                                         break;
205                                     }
206                                 }
207                                 
208                                 it = GlobalDataDatamining.SQLOperators_comp.listIterator();
209                                 for (;it.hasNext();){
210                                     sql_key = it.next();
211                                     if (substr_before.contains(sql_key) == true || substr_after.contains(sql_key) == true){
212                                         this.setValueOfAttribute(2);

248             int found = 0;        
249             int i, j, k, open_square_brace, index, ii;
250             char chars[];
251             String aux;
252             
253             // Para index 4 a 9 sem concatenation
254             for (Map.Entry<String, Integer> str_att : GlobalDataDatamining.StringAttributes.entrySet()){          
255                 if (codeline.contains(str_att.getKey()) == true){
256                     j = codeline.indexOf(str_att.getKey());
257                     aux = codeline.substring(j);
258                     j = str_att.getKey().length()+1;
259                     k = 0;
260                     chars = aux.substring(j).toCharArray();
261                     open_square_brace = 1;
262                     while (open_square_brace != 0){
263                         if (chars[k] == '(')
264                             open_square_brace++;
265                         if (chars[k] == ')')
266                             open_square_brace--;
267                         k++;
268                     }                                
269                     i = -1;
270                     String a;
271                     for (ii = 0; ii < inp.size(); ii++){
272                         i = aux.indexOf((String) inp.get(ii));                     
273                         if (i >= j && i < j+k){
274                             index = str_att.getValue();                        
275                             this.setValueOfAttribute(index);
276                             if (c_j.who.isEmpty())
277                                 c_j.who += str_att.getKey();
278                             else
279                                 c_j.who += ", " + str_att.getKey();                        
280                             found = 1;
281                         }
282                     }
283                 }
284             }

432         private int applyClassifier(String nameClassifier, Instance inst) throws IOException{
433             int classe;
434             
435             /* Create the instance to classify */
436             //Instance instance = new DenseInstance(this.attributes);
437             
438             // Classify the instance 
439             Object predictedClassValue = "No";
440             if (nameClassifier.equals("lr") == true)
441                 predictedClassValue = GlobalDataApp.cf_lr.classify(inst);
442             
443             if (nameClassifier.equals("rt") == true)
444                 predictedClassValue = GlobalDataApp.cf_rt.classify(inst);
445             
446             if (nameClassifier.equals("svm") == true)
447                 predictedClassValue = GlobalDataApp.cf_svm.classify(inst);
448             
449             if (nameClassifier.equals("jrip") == true)
450                 predictedClassValue = GlobalDataApp.cf_jrip.classify(inst);        
451             
452             if (predictedClassValue.equals("Yes"))
453                 classe = 1;
454             else
455                 classe = 0;
456             
457             return classe;
