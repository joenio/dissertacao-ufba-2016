63         public Instance getInstanceOfListInstances(String instName){
64             Instance inst_final = null;
65             Iterator <Instance> it = this.getMembers().iterator();
66             Instance ins;
67             for (;it.hasNext();){
68                     ins = it.next();
69                     if (ins.getInstanceName().equals(instName) == true){
70                         inst_final = ins;
71                         break;
72                     }
73             }
74             return inst_final;
75         }

81         public Boolean containsInstance(String instName){
82             Iterator <Instance> it = this.getMembers().iterator();
83             Instance ins;
84             Boolean haveInst = false;
85             for (;it.hasNext();){
86                     ins = it.next();
87                     if (ins.getInstanceName().equals(instName) == true){
88                         haveInst = true;
89                         break;
90                     }
91             }
92             return haveInst;
93         }
