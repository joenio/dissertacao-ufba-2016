131         public String getInstanceNameFromAliases(String s, InstanceTable instTab){
132             String instName = "";
133             boolean found = false;
134             while (found == false){
135                 if (instTab.containsInstance(s) == true){
136                     instName = s;
137                     found = true;
138                 }
139                 else{
140                     if (this.getAliases().containsKey(s)){
141                         Symbol sym = (Symbol) this.getAliases().get(s);
142                         s = sym.getName();
143                     }
144                 }
145             }
146             return instName;
147         }

150          * Return a specific symbol
151          */
152         public Symbol getSymbolOfListSymbols(String symName){
153             Symbol sym_final = null;
154             Iterator <Symbol> it = this.getMembers().iterator();
155             Symbol sym;
156             for (;it.hasNext();){
157                     sym = it.next();
158                     if (sym.getName().equals(symName) == true){
159                         sym_final = sym;
160                         break;
161                     }
162             }
163             return sym_final;
164         }

430         /*
431          * Print the elements of the main Map scope.
432          */
433         public void print(Scope scp){
434             Symbol aux1;
435             Scope s;
436             if (!scp.getMembers().isEmpty()){
437                 System.out.println("\nScope: "+scp.getScopeName()+"\tsize="+scp.getMembers().size());
438                 Iterator <Symbol> it = scp.getMembers().iterator();
439                 for(; it.hasNext();){
440                     aux1 = it.next();
441                     if ( aux1.getRootScope() != null ) { // is a root of a new scope?
442                             s = aux1.getRootScope();
443                             s.print(s);
444                     }
445                 }
446             }
447              else{
448                 aux1 = scp.getScopeSymbol();
449                  System.out.println("Name: "+aux1.getName()+"\t\tline: "+aux1.getCodeLine()+"\t\ttainted: "
450                                     +aux1.getTainted()+"\t\tScope: "+aux1.getScope().getScopeName()+"\n");
451              }
452         }
