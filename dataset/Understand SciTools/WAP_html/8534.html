3      * entre horas de inicio e fim

5     package org.homeunix.wap.utils;
6     
7     /**
8      *
9      * @author iberiam
10      */
11     public class Dates {
12         public static String DiffDates(Long seconds){
13             String diff_date = "", m, s, h;
14             int horas, min;
15             
16             if (seconds >= 1000){
17                 seconds = seconds/1000;
18                 horas = (int)Math.floor(seconds / 3600);
19                 seconds -= horas * 3600;
20                 min = (int)Math.floor(seconds / 60);
21                 seconds -= min * 60;
22                 
23                 if (horas < 10)
24                     h = "0"+ Integer.toString(horas);
25                 else
26                     h = Integer.toString(horas);
27                 
28                 if (min < 10)
29                     m = "0"+ Integer.toString(min);
30                 else
31                     m = Integer.toString(min);
32     
33                 if (seconds < 10)
34                     s = "0"+ Long.toString(seconds);
35                 else
36                     s = Long.toString(seconds);
37                 
38                 diff_date = h + ":" + m + ":" + s + " H";
39             }
40             else
41                 //System.out.println("Tempo decorrido " + (end_date - begin_date) + " ms");
42                 diff_date = seconds + " ms";
43             
44             return diff_date;
45         }
46     }
