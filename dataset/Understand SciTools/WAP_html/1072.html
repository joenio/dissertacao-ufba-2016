100         public void copyToMTS(TaintedTable m, UntaintedTable mu) { // m = mts
101             Iterator <RootTaintedSymbol> it = this.getTaintedMembers().values().iterator(); // mts_aux
102             RootTaintedSymbol rts, rts_m;
103             RelatedTaintedSymbol rtt=null, rtt_m=null;
104             VariableTaintedSymbol vts = null, vts_m;
105             for(;it.hasNext();){//mts_aux
106                 rts = it.next();
107                 if (m.getTaintedMembers().containsKey(rts.getScopeTaintedName()) == false)
108                     m.getTaintedMembers().put(rts.getScopeTaintedName(), rts);
109                 else{
110                     rts_m = (RootTaintedSymbol) m.getTaintedMembers().get(rts.getScopeTaintedName());
111                     Iterator <Integer> it1;
112                     Iterator <String> it2 = rts.getFilesList().iterator();
113                     int i;
114                     String s;
115                     for (it1 = rts.getLinesList().iterator(); it1.hasNext();){
116                         i = it1.next();
117                         s = it2.next();
118                         if (i != 0){ // as var globais em mts_aux tem linha = 0
119                             if (rts_m.getLinesList().contains(i) == false ||
120                                 rts_m.getLinesList().contains(i) == true && rts_m.getFilesList().contains(s) == false){
121                                 rts_m.getLinesList().add(i);
122                                 rts_m.getFilesList().add(s);
123                                 for (Iterator <RelatedTaintedSymbol> it3 = rts.getListTaintedMembers().iterator(); it3.hasNext();){
124                                     rtt = it3.next();
125                                     if (rtt.getLineRts() == i && rtt.getFileRts().equals(s)){
126                                         rts_m.define(rtt);
127                                         break;
128                                     }
129                                 }
130                             }
131                             else{
132                                 // Obter RelatedTaintedSymbol de mts_princ de i e s
133                                 for (Iterator <RelatedTaintedSymbol> it3 = rts_m.getListTaintedMembers().iterator(); it3.hasNext();){
134                                     rtt_m = it3.next();
135                                     if (rtt_m.getLineRts() == i && rtt_m.getFileRts().equals(s))
136                                         break;
137                                 }
138                                 // Obter RelatedTaintedSymbol de mts_aux de i e s para copiar os vts para mts_princ
139                                 for (Iterator <RelatedTaintedSymbol> it3 = rts.getListTaintedMembers().iterator(); it3.hasNext();){
140                                     rtt = it3.next();
141                                     if (rtt.getLineRts() == i && rtt.getFileRts().equals(s))
142                                         break;
143                                 }
144                                 
145                                 Iterator <VariableTaintedSymbol> it3 = rtt.getTaintedMembers().values().iterator();
146                                 for (;it3.hasNext();){
147                                     vts = it3.next();
148                                     if (rtt_m.getTaintedMembers().containsKey(vts.getName()) == false){
149                                         rtt_m.define(vts);
150                                     }
151                                     else{
152                                         vts_m = (VariableTaintedSymbol) rtt_m.getTaintedMembers().get(vts.getName());
153                                         Iterator <Integer> it4 = vts.getVariableLines().iterator();
154                                         Iterator <String> it5 = vts.getVariableFiles().iterator();
155                                         for (; it4.hasNext();){
156                                             int ii = it4.next();
157                                             String ss = it5.next();
158                                             if (vts_m.getVariableLines().contains(ii) == false ||
159                                                 vts_m.getVariableLines().contains(ii) == true && vts_m.getVariableFiles().contains(ss) == false){
160                                                 vts_m.getVariableLines().add(ii);
161                                                 vts_m.getVariableFiles().add(ss);
162                                             }
163                                         }
164                                     }
165                                 }
166                             }
167                         }
168                     }
169                     
170                     // Se rts passou de untainted (vindo de mus) para tainted (em mts_aux), retirar rts de mus
171                     Symbol sym = rts.getSymbol();
172                     if (mu.existSymbol(sym.getName()) == true) // verify if symbol pass to untaint to taint
173                         mu.removeUntaintSymbol(sym);
174                 }
175             }
176         }

178         public void copyGlobalsToMTS(TaintedTable mts_princ , UntaintedTable mus_princ , UntaintedTable mus_aux, MethodSymbol meth_sym_clone){
179             for(Iterator <Symbol> it = meth_sym_clone.getGlobalVarsOfFunction().values().iterator(); it.hasNext();){
180                 Symbol sym = it.next();
181                 if (this.getTaintedMembers().containsKey(sym.getName()) == true && mus_aux.existSymbol(sym.getName()) == false){
182                     if (mts_princ.getTaintedMembers().containsKey(sym.getName()) == true){
183                         // Se passou de untainted para tainted, retirar de untainted
184                         if(mus_princ.existSymbol(sym.getName()) == true)
185                             mus_princ.removeUntaintSymbol(sym);                   
186                     else{
187                         RootTaintedSymbol rts_aux = (RootTaintedSymbol) this.getTaintedMembers().get(sym.getName());
188                         RootTaintedSymbol rts_princ = (RootTaintedSymbol) mts_princ.getTaintedMembers().get(sym.getName());
189                         int index = rts_aux.getLinesList().size()-1;
190                         int i = (Integer) rts_aux.getLinesList().get(index);
191                         String s = (String) rts_aux.getFilesList().get(index);
192                         RelatedTaintedSymbol rtt_aux = (RelatedTaintedSymbol) rts_aux.getListTaintedMembers().get(index);
193                         if (rts_princ.getLinesList().contains(i) == false){
194                             rts_princ.getLinesList().add(i);
195                             rts_princ.getFilesList().add(s);
196                             rts_princ.getListTaintedMembers().add(rtt_aux);
197                         }
198                         else{
199                             int index_princ = rts_princ.getLinesList().indexOf(i);
200                             if (rts_princ.getFilesList().get(index_princ).equals(s) == false){
201                                 rts_princ.getLinesList().add(i);
202                                 rts_princ.getFilesList().add(s);
203                                 rts_princ.getListTaintedMembers().add(rtt_aux);
204                             }
205                             else{
206                                 rts_princ.getListTaintedMembers().set(index_princ, rtt_aux);
207                             }
208                         }
209                     }
210                     }
211                 }
212             }

215     public void removeDuplicates(Symbol sym){
216         if (this.getTaintedMembers().containsKey(sym.getScope().getScopeName()) == true){
217             RootTaintedSymbol Rrt = (RootTaintedSymbol) this.getTaintedMembers().get(sym.getScope().getScopeName());
218             if (Rrt.getLinesList().size() > 1){
219                 int last_i = (Integer) Rrt.getLinesList().get(Rrt.getLinesList().size()-1);
220                 String last_s = (String) Rrt.getFilesList().get(Rrt.getFilesList().size()-1);
221                 
222                 int penultimate_i = (Integer) Rrt.getLinesList().get(Rrt.getLinesList().size()-2);
223                 String penultimate_s = (String) Rrt.getFilesList().get(Rrt.getFilesList().size()-2);
224                 if (last_i == penultimate_i && last_s.equals(penultimate_s) == true){
225                     Rrt.getLinesList().remove(Rrt.getLinesList().size()-1);
226                     Rrt.getFilesList().remove(Rrt.getFilesList().size()-1);
227                 }
228             }
229          }
