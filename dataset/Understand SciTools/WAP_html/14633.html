14      *
15      * @author Iberia Medeiros

17     public class UntaintedTable {
18         Map<String, Symbol> symbolsUntainted;
19     
20         /* CONSTRUCTOR */
21         public UntaintedTable() {
22          symbolsUntainted = new LinkedHashMap<String, Symbol>();
23         }
24     
25     
26         /* METHODS of the class*/
27     
28         /*
29          * Give the Map that contain the untainted variables
30          */
31         public Map getUntaintedMembers() {
32             return symbolsUntainted;
33         }
34     
35         /*
36          * Verify if a untainted variable exist in Map
37          */
38         public Boolean existSymbol(String s) {
39             return symbolsUntainted.containsKey(s);
40         }
41     
42         /*
43          * Insert a untainted variable in Map
44          */
45         public void insertUntaintSymbol(Symbol sym) {
46         	symbolsUntainted.put(sym.getName(), sym);
47         }
48     
49         /*
50          * Remove a untainted variable from Map
51          */
52         public void removeUntaintSymbol(Symbol sym) {
53         	symbolsUntainted.remove(sym.getName());
54         }
55     
56         public void copyToMUS(UntaintedTable m) { // m = mus
57             Iterator <Symbol> it = this.getUntaintedMembers().values().iterator(); // mus_aux
58             Symbol sym;
59             for(;it.hasNext();){
60                 sym = it.next();
61                 if (!sym.getName().equals("return"))
62                     m.getUntaintedMembers().put(sym.getName(), sym);
63             }
64         }
65     
66         public void copyGlobalsToMUS(TaintedTable mts_princ , UntaintedTable mus_princ, MethodSymbol meth_sym_clone){
67             for(Iterator <Symbol> it = meth_sym_clone.getGlobalVarsOfFunction().values().iterator(); it.hasNext();){
68                 Symbol sym = it.next();
69                 if (this.getUntaintedMembers().containsKey(sym.getName()) == true && mts_princ.getTaintedMembers().containsKey(sym.getName()) == true) {
70                     if (mus_princ.existSymbol(sym.getName()) == false){
71                         // Se passou de tainted para untainted, colocar em untainted
72                         mus_princ.insertUntaintSymbol(sym);
73                     }
74                 }
75             }
76         }    
77         
78         
79         public UntaintedTable copyUntaintedTable() {
80             UntaintedTable tab = new UntaintedTable();
81             for(Iterator <Symbol> it = this.getUntaintedMembers().values().iterator(); it.hasNext();){
82     		Symbol cs = it.next();
83                 try {
84                     tab.getUntaintedMembers().put(cs.getName(), cs.clone());
85                 } catch (CloneNotSupportedException ex) {
86                     Logger.getLogger(UntaintedTable.class.getName()).log(Level.SEVERE, null, ex);
87                 }
88     	 }
89         	return tab;
90         }
91         
92          public void cleanAll(){
93             Collection <Symbol> c = this.symbolsUntainted.values();
94         	c.removeAll(c);
95         	this.symbolsUntainted.clear();
96         }
