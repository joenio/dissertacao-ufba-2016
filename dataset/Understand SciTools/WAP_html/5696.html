5     package org.homeunix.wap.table.tainted;
6     
7     import org.homeunix.wap.table.symbol.Symbol;
8     import java.util.*;
9     
10     /**
11      *
12      * @author iberiam
13      */
14     public class RootTaintedSymbol extends Symbol{
15         String nameRootSymbol;
16         List<RelatedTaintedSymbol> symbolsTainted;
17         List<Integer> lines; // Lista que armazenara as linhas de vars, fun tainteds
18         List<String> files; // Lista que armazenara os files names de vars, fun tainteds
19         
20         /* CONSTRUCTOR */
21         public RootTaintedSymbol(String name, int line, int tainted, Boolean alfanumeric, String file){
22             super(name, tainted, line, alfanumeric, file);
23             symbolsTainted = new ArrayList<RelatedTaintedSymbol>();
24             this.nameRootSymbol = name;
25             lines = new ArrayList<Integer>();
26             files = new ArrayList<String>();
27             RelatedTaintedSymbol rrt = new RelatedTaintedSymbol(line, file);
28             this.define(rrt);
29         }
30         
31         /* METHODS that satisfy TaintedScope interface*/
32     
33         /*
34          * Give the the name of taintedScope. How this scope don't have parent, then
35          * your name is "globaltainted"
36          */
37         public String getScopeTaintedName() {
38             return nameRootSymbol;
39         }
40     
41         /*
42          * If the tainted scope is enclosing in another (if have parent), give the
43          * name of the parent scope.
44          * How the Tainted Table is the main scope, then don't have parent (null)
45          */
46         public ScopeTainted getEnclosingScopeTainted() {
47         	return null;
48         }
49     
50         /*
51          * Give the main Map that contain the tainted variables
52          */
53         public List getListTaintedMembers() {
54             return symbolsTainted;
55         }
56     
57         /*
58          * Give the List of line codes from the variable included in main Map
59          */
60         public List getLinesList (){
61             return lines;
62         }
63     
64         /*
65          * Give the List of the files names from the variable included in main Map
66          */
67         public List getFilesList (){
68             return files;
69         }    
70         
71         /*
72          * Insert a tainted variable in main Map.
73          * Each member of main Map is a RelatedTaintedSymbol, composed by:
74          * Map VariableTaintedSymbol and List to store code lines of tainted variables
75          */
76         public final void define(RelatedTaintedSymbol rtt) {
77         	symbolsTainted.add(rtt);
78             lines.add(rtt.getLineRts());
79             files.add(rtt.getFileRts());
80         }
81         
82         /*
83          * Insert a code line in List of the tainted variable included in main Map.
84          * This appened when new tainted variable is identified, and your code line
85          * is included in List
86          */
87         public void setLineCode (int line){
88             lines.add(line);
89         }
90     
91         /*
92          * Insert a file name in List of the tainted variable included in main Map.
93          * This appened when new tainted variable is identified, and your code line
94          * is included in List
95          */
96         public void setFile (String file){
97            files.add(file);
98         }
99         
100     }
