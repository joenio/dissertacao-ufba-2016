59         public MethodClass getMethodClass(String methName, String methType, Map mct) {
60             MethodSymbol mss_final = null;
61             ClassSymbol cs = null;
62             MethodClass mc = null;
63             int i;
64             String f = "";
65             Iterator <MethodSymbol> it = this.getMembersMethods().getMembers().iterator();
66             MethodSymbol ms = null;
67             Boolean contConstruct = false;
68             Boolean cont_methName = false;
69             for (;it.hasNext();){
70                 ms = it.next();
71                 if (ms.getFunctionName().equals(methName) == true){
72                     f = methName;
73                     cont_methName = true;
74                     break;
75                 }
76                 if (methType.equals("construct") == true && ms.getFunctionName().equals("__construct") == true){
77                     f = "__construct";
78                     contConstruct = true;
79                     break;
80                 }
81             }
82     
83             if (cont_methName == true || contConstruct == true){
84                 mss_final = ms;
85                 cs = this;
86             }
87             else{
88                 String ff = this.fileExtendsClass;
89                 String fff = this.extendsNameClass;
90                 try{
91                     ClassTable ct = (ClassTable) mct.get(ff);
92                     cs = ct.getClassOfListClasses(fff);
93                     mc = cs.getMethodClass(methName, methType, mct);
94                 } catch (Exception e){}
95             }
96     
97             if (mss_final != null)
98                 mc = new MethodClass(mss_final, cs);
99             return mc;
100         }

192         private void copyScope(Scope cs, Scope csClone){
193             Symbol aux1, aux2;
194             Scope s, s1 = null;
195             CallSymbol curScope1;
196             if (!cs.getMembers().isEmpty()){
197                 for(Iterator <Symbol> it = cs.getMembers().iterator(); it.hasNext();){
198                     aux1 = it.next();
199                     if ( aux1.getRootScope() != null ) { // is a root of a new scope?
200                             s = aux1.getRootScope();
201                             try {
202                                 s1 = (Scope) cs.getScopeSymbol().clone();
203                             } catch (CloneNotSupportedException ex) {
204                                 Logger.getLogger(MethodSymbol.class.getName()).log(Level.SEVERE, null, ex);
205                             }                        
206                             curScope1 = new CallSymbol (aux1.getName(), aux1.getCodeLine(), s1, aux1.getTainted(), aux1.getFileSymbol());
207                             curScope1.getScopeSymbol().setIsFunction(aux1.getIsFunction());
208                             curScope1.getScopeSymbol().setIsSecureFunction(aux1.getIsSecureFunction());
209                             curScope1.getScopeSymbol().setIsTaintedFunction(aux1.getIsTaintedFunction());
210                             csClone.define(curScope1, curScope1, false);
211                             copyScope(s, curScope1);
212                     }
213                     else{
214                         VariableSymbol vs = new VariableSymbol(aux1.getName(), aux1.getTainted(), aux1.getCodeLine(), aux1.getFileSymbol());
215                         csClone.define(vs, null, true);
216                     }
217                         
218                 }
219             }       
220         }
