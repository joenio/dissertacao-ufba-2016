386         private static void analysisOfFile(ListVulners lv, FileWriter outFile, String filename) throws IOException{
387             VulnerXSS v;
388             String cl, correctedCode = "", aux = "";
389             Iterator <VulnerXSS> it;
390             int i=0;
391             LinesToCorrect ltc;
392             Object key;
393             
394             String setPlainText = "";
395             String setBoldText = "";
396             if (GlobalDataApp.isWindows.booleanValue() == false){
397                 setPlainText = "\033[0;0m";
398                 setBoldText = "\033[0;1m";
399             }
400             
401             //ltc = GlobalDataXSS.MainLinesToCorrect.get(filename);
402             //Map sortedMap = new TreeMap(ltc.getMapLinesToCorrect());
403     
404     
405             for (it = lv.getListOfVulners().iterator(); it.hasNext();){
406                 v = it.next();
407                 if (v.IsFP() == 0)
408                     System.out.println(setBoldText + "\n\t= = = =  Vulnerability n.: " + (i+1) + "  = = = =");
409                 else
410                     System.out.println(setBoldText + "\n\t= = = =  Vulnerability n.: " + (i+1) + "   >>> is a possible False Positive <<<  = = = =");
411                 System.out.println("\tVulnerable code:" + setPlainText);
412                 if(GlobalDataApp.args_flags[4] == 1){
413                     if (v.IsFP() == 0)
414                         outFile.write("\n\t= = = =  Vulnerability n.: " + (i+1) + "  = = = =\n");
415                     else
416                         outFile.write("\n\t= = = =  Vulnerability n.: " + (i+1) + "   >>> is a possible False Positive <<<  = = = =\n");
417                     outFile.write("\tVulnerable code:\n");
418                 }
419                 i++;
420     
421                 try{
422                 // escrever vulnerabilidade no ecra
423                 String f;
424                 correctedCode = "";
425                 for (int l = v.getLinesOfVulner().size() -1; l >= 0; l--){
426                     f = "";
427                     if (v.getFilesOfVulner().get(l).equals(lv.getFilename()) == false)
428                         f = (String) v.getFilesOfVulner().get(l);
429     
430                     ManageFiles fff = new ManageFiles ((String) v.getFilesOfVulner().get(l));
431                     cl = fff.getLineOfCode((Integer) v.getLinesOfVulner().get(l));
432                     System.out.println("\t" + v.getLinesOfVulner().get(l) + ": "+ cl);
433                     if(GlobalDataApp.args_flags[4] == 1){
434                         outFile.write("\t" + v.getLinesOfVulner().get(l) + ": "+ cl+"\n");
435                     }
436                     if (f.equals("") == false){
437                         System.out.println("\t    (" + f + ")");
438                         if(GlobalDataApp.args_flags[4] == 1){
439                             outFile.write("\t    (" + f + ")\n");
440                         }
441                     }
442                     
443     
444                     if (v.IsFP() == 0){
445                         // escrita do corrected code no ecra
446                         ltc = GlobalDataXSS.MainLinesToCorrect.get(v.getFileOfLineToSanitize());
447                         Map sortedMap = new TreeMap(ltc.getMapLinesToCorrect());                
448                         if (ltc.getMapLinesToCorrect().containsKey(v.getLinesOfVulner().get(l)) == true){
449                             key = v.getLinesOfVulner().get(l);
450                             aux =  "\t" + (Integer)key + ": " + sortedMap.get(key).toString()+"\n";
451                         }
452                         else
453                             aux =  "\t" + v.getLinesOfVulner().get(l) + ": "+ cl + "\n";
454                         correctedCode = correctedCode + aux;
455                     }  
456                 }
457             }catch (Exception e){
458                 aux = "\tSome unexpected error in analysis happened. Sorry :-(";
459                 correctedCode = correctedCode + aux;
460             }            
461     
462             if (v.IsFP() == 0){
463                 System.out.println(setBoldText + "\n\tCorrected code:" + setPlainText);
464                 System.out.println(correctedCode);
465                 if(GlobalDataApp.args_flags[4] == 1)
466                     outFile.write("\n\tCorrected code:\n" + correctedCode);
467             }
468             else{
469                 String justifyFP = v.getJustificationOfFP();
470                 System.out.println(setBoldText + "\n\tFalse Positive justification:" + setPlainText);
471                 System.out.println(justifyFP);
472                 if(GlobalDataApp.args_flags[4] == 1)
473                     outFile.write("\n\tFalse Positive justification:\n" + justifyFP); 
474             }            
475             }
476         }
