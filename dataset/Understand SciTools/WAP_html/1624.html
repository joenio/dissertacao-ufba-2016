15     public class AutoCorrectSQL {
16         VulnerSQLI vv;

18         // *** CONSTRUCTOR
19         public AutoCorrectSQL(VulnerSQLI vul) {
20             this.vv = vul;
21         }

24         
25         // Cria a string para a correcao automatica
26         public void buildCorrectSQL(Map MainLinesToCorrect) {    
27             String sql_final = "";
28             LinesToCorrect ltc = getLinesToCorrect(vv.getFileOfLineToSanitize(), MainLinesToCorrect);
29             ManageFiles fff = new ManageFiles (vv.getFileOfLineToSanitize());
30             
31             if (vv.getMysqlFunc().equals("mysqli_execute") == true){
32                     if (vv.getLineMysqliBindParam() > 0){
33                         // Linha de mysqli_bind_param passar para mysqli_stmt_bind_param
34                         // buscar linha do ficheiro
35                         String cLine = fff.getLineOfCode(vv.getLineMysqliBindParam());
36                         cLine = cLine.replace("mysqli_bind_param", "mysqli_stmt_bind_param");
37                         ltc.getMapLinesToCorrect().put(vv.getLineMysqliBindParam(), cLine);
38                     }
39                    vv.setCorrectedCode(vv.getVulCode());
40             }
41           
42             String codeLine = fff.getLineOfCode(vv.getLineToSanitize());
43             String cod_antes = codeLine.toString();
44             codeLine = codeLine.replace(vv.getVulCode(), vv.getCorrectedCode());
45             if (vv.getDecreptedMysqlFuncToSubst().equals("") == false)
46                 codeLine = codeLine.replace(vv.getMysqlFunc(), vv.getDecreptedMysqlFuncToSubst());
47            
48             if (cod_antes.equals(vv.getVulCode()) == false)
49                 ltc.getMapLinesToCorrect().put(vv.getLineToSanitize(), codeLine);
50             else
51                 ltc.getMapLinesToCorrect().put(vv.getLineToSanitize(), vv.getCorrectedCode()); 
52         }
53        
54         private LinesToCorrect getLinesToCorrect(String filename, Map MainLinesToCorrect){
55             if (MainLinesToCorrect.containsKey(filename))
56                 return (LinesToCorrect) MainLinesToCorrect.get(filename);
57             else{
58                 LinesToCorrect l = new LinesToCorrect(filename);
59                 MainLinesToCorrect.put(filename, l);
60                 return l;
61             }
62         }
