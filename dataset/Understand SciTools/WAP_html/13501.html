136         public Instance getInstanceOfReturn(String methName, int line, MethodSymbolCall mscAnaliseMeth, Scope sc, Boolean inAnaliseMeth) {
137             Instance inst_final = null;
138             MethodSymbolCall msc = mscAnaliseMeth;
139     
140             if (inAnaliseMeth == false){
141                 Iterator <MethodSymbolCall> it;
142                 for (it = this.methodCallInstance.getListMethodsMembers().iterator(); it.hasNext();){
143                     msc = it.next();
144                     if (msc.getFunctionNameCall().equals(methName) == true && msc.getLineNumberCall() == line)
145                         break;
146                 }
147             }
148     
149             try{
150                 Scope scp = sc;
151                 Iterator <Symbol> it1;
152                 if (inAnaliseMeth == false){
153                     for (it1 = msc.getCallFunctionSymbols().iterator(); it1.hasNext();){
154                         scp = (Scope) it1.next();
155                         if (scp.getScopeName().equals("return")){
156                             break;
157                         }
158                     }
159                 }
160                 
161                 Symbol sy = (Symbol) scp.getMembers().get(0);
162     
163                 if (sy.getRootScope() != null){
164                     scp = (Scope) sy;
165                     String s = scp.getScopeSymbol().getName();
166                     Instance inst = getInstanceFromAliases(s, msc);
167                     if (inst != null){
168                         scp = (Scope) scp.getMembers().get(0);
169                         String methName_aux = scp.getScopeSymbol().getName();
170                         inst.getInstanceOfReturn(methName_aux, scp.getScopeSymbol().getCodeLine(), null, null, false);
171                     }
172                 }
173                 else{
174                     Instance inst = getInstanceFromAliases(sy.getName(), msc);
175                     inst_final = inst;
176                 }
177             } catch (Exception e){}
178     
179             return inst_final;
180         }

182         private static Instance getInstanceFromAliases(String s, MethodSymbolCall methSymCall){
183             Instance inst = null;
184             boolean found = false;
185             while (found == false){
186                 if (methSymCall.getObjectsOfMethod().containsInstance(s) == true){
187                     inst = (Instance) methSymCall.getObjectsOfMethod().getInstanceOfListInstances(s);
188                     found = true;
189                 }
190                 else{
191                     if (methSymCall.getAliases().containsKey(s)){
192                         Symbol sym = (Symbol) methSymCall.getAliases().get(s);
193                         s = sym.getName();
194                     }
195                     else
196                         break;
197                 }
198             }
199     
200             return inst;
201         }

209         public void updateTainted(Scope cs_att, Symbol sym) {
210             cs_att.getScopeSymbol().setTainted(sym.getTainted());
211             cs_att.resolveSymbol(cs_att, sym, this.getMtsInstanceAttributes(), this.getMusInstanceAttributes(), sym.getFileSymbol());
212             if (cs_att.getScopeSymbol().getTainted() == 1)
213                 this.setTainted(1);
214             else{
215                 if (this.getTainted() == 1){
216                     Symbol sy;
217                     int numTaints = 0;
218                     for (int j = 0; j < this.getInstanceAttributes().getMembers().size(); j++){
219                         sy = (Symbol) this.getInstanceAttributes().getMembers().get(j);
220                         if (sy.getTainted() == 1)
221                             numTaints++;
222                     }
223                     if (numTaints == 0)
224                         this.setTainted(0);
225                 }
226             }
