39     // The return value of fgetc(), getc(), ungetc(), getchar() etc. is an integer value.
40     // If this return value is stored in a character variable and then compared
41     // to compared to EOF, which is an integer, the comparison maybe be false.
42     //
43     // Reference:
44     // - Ticket #160
45     // - http://www.cplusplus.com/reference/cstdio/fgetc/
46     // - http://www.cplusplus.com/reference/cstdio/getc/
47     // - http://www.cplusplus.com/reference/cstdio/getchar/
48     // - http://www.cplusplus.com/reference/cstdio/ungetc/ ...
49     //----------------------------------------------------------------------------------
50     void CheckOther::checkCastIntToCharAndBack()
51     {
52         if (!_settings->isEnabled("warning"))
53             return;
54     
55         const SymbolDatabase *symbolDatabase = _tokenizer->getSymbolDatabase();
56         const std::size_t functions = symbolDatabase->functionScopes.size();
57         for (std::size_t i = 0; i < functions; ++i) {
58             const Scope * scope = symbolDatabase->functionScopes[i];
59             std::map<unsigned int, std::string> vars;
60             for (const Token* tok = scope->classStart->next(); tok != scope->classEnd; tok = tok->next()) {
61                 // Quick check to see if any of the matches below have any chances
62                 if (!tok->isName() || !Token::Match(tok, "%var%|EOF %comp%|="))
63                     continue;
64                 if (Token::Match(tok, "%var% = fclose|fflush|fputc|fputs|fscanf|getchar|getc|fgetc|putchar|putc|puts|scanf|sscanf|ungetc (")) {
65                     const Variable *var = tok->variable();
66                     if (var && var->typeEndToken()->str() == "char" && !var->typeEndToken()->isSigned()) {
67                         vars[tok->varId()] = tok->strAt(2);
68                     }
69                 } else if (Token::Match(tok, "EOF %comp% ( %var% = fclose|fflush|fputc|fputs|fscanf|getchar|getc|fgetc|putchar|putc|puts|scanf|sscanf|ungetc (")) {
70                     tok = tok->tokAt(3);
71                     const Variable *var = tok->variable();
72                     if (var && var->typeEndToken()->str() == "char" && !var->typeEndToken()->isSigned()) {
73                         checkCastIntToCharAndBackError(tok, tok->strAt(2));
74                     }
75                 } else if (_tokenizer->isCPP() && (Token::Match(tok, "EOF %comp% ( %var% = std :: cin . get (") || Token::Match(tok, "EOF %comp% ( %var% = cin . get ("))) {
76                     tok = tok->tokAt(3);
77                     const Variable *var = tok->variable();
78                     if (var && var->typeEndToken()->str() == "char" && !var->typeEndToken()->isSigned()) {
79                         checkCastIntToCharAndBackError(tok, "cin.get");
80                     }
81                 } else if (_tokenizer->isCPP() && (Token::Match(tok, "%var% = std :: cin . get (") || Token::Match(tok, "%var% = cin . get ("))) {
82                     const Variable *var = tok->variable();
83                     if (var && var->typeEndToken()->str() == "char" && !var->typeEndToken()->isSigned()) {
84                         vars[tok->varId()] = "cin.get";
85                     }
86                 } else if (Token::Match(tok, "%var% %comp% EOF")) {
87                     if (vars.find(tok->varId()) != vars.end()) {
88                         checkCastIntToCharAndBackError(tok, vars[tok->varId()]);
89                     }
90                 } else if (Token::Match(tok, "EOF %comp% %var%")) {
91                     tok = tok->tokAt(2);
92                     if (vars.find(tok->varId()) != vars.end()) {
93                         checkCastIntToCharAndBackError(tok, vars[tok->varId()]);
94                     }
95                 }
96             }
97         }
98     }

117     // Clarify calculation precedence for ternary operators.
118     //---------------------------------------------------------------------------
119     void CheckOther::clarifyCalculation()
120     {
121         if (!_settings->isEnabled("style"))
122             return;
123     
124         const SymbolDatabase *symbolDatabase = _tokenizer->getSymbolDatabase();
125         const std::size_t functions = symbolDatabase->functionScopes.size();
126         for (std::size_t i = 0; i < functions; ++i) {
127             const Scope * scope = symbolDatabase->functionScopes[i];
128             for (const Token* tok = scope->classStart->next(); tok != scope->classEnd; tok = tok->next()) {
129                 // ? operator where lhs is arithmetical expression
130                 if (tok->str() != "?" || !tok->astOperand1() || !tok->astOperand1()->isCalculation())
131                     continue;
132                 if (!tok->astOperand1()->isArithmeticalOp() && tok->astOperand1()->tokType() != Token::eBitOp)
133                     continue;
134     
135                 // Is code clarified by parentheses already?
136                 const Token *tok2 = tok->astOperand1();
137                 for (; tok2; tok2 = tok2->next()) {
138                     if (tok2->str() == "(")
139                         tok2 = tok2->link();
140                     else if (tok2->str() == ")")
141                         break;
142                     else if (tok2->str() == "?") {
143                         clarifyCalculationError(tok, tok->astOperand1()->str());
144                         break;
145                     }
146                 }
147             }
148         }
149     }

209     void CheckOther::checkSuspiciousSemicolon()
210     {
211         if (!_settings->inconclusive || !_settings->isEnabled("warning"))
212             return;
213     
214         const SymbolDatabase* const symbolDatabase = _tokenizer->getSymbolDatabase();
215     
216         // Look for "if(); {}", "for(); {}" or "while(); {}"
217         for (std::list<Scope>::const_iterator i = symbolDatabase->scopeList.begin(); i != symbolDatabase->scopeList.end(); ++i) {
218             if (i->type == Scope::eIf || i->type == Scope::eElse || i->type == Scope::eWhile || i->type == Scope::eFor) {
219                 // Ensure the semicolon is at the same line number as the if/for/while statement
220                 // and the {..} block follows it without an extra empty line.
221                 if (Token::simpleMatch(i->classStart, "{ ; } {") &&
222                     i->classStart->previous()->linenr() == i->classStart->tokAt(2)->linenr()
223                     && i->classStart->linenr()+1 >= i->classStart->tokAt(3)->linenr()) {
224                     SuspiciousSemicolonError(i->classDef);
225                 }
226             }
227         }
228     }

238     // For C++ code, warn if C-style casts are used on pointer types
239     //---------------------------------------------------------------------------
240     void CheckOther::warningOldStylePointerCast()
241     {
242         // Only valid on C++ code
243         if (!_settings->isEnabled("style") || !_tokenizer->isCPP())
244             return;
245     
246         const SymbolDatabase *symbolDatabase = _tokenizer->getSymbolDatabase();
247         for (std::size_t i = 0; i < symbolDatabase->functionScopes.size(); ++i) {
248             const Scope * scope = symbolDatabase->functionScopes[i];
249             const Token* tok;
250             if (scope->function && scope->function->isConstructor())
251                 tok = scope->classDef;
252             else
253                 tok = scope->classStart;
254             for (; tok && tok != scope->classEnd; tok = tok->next()) {
255                 // Old style pointer casting..
256                 if (!Token::Match(tok, "( const| %type% * const| ) (| %name%|%num%|%bool%|%char%|%str%"))
257                     continue;
258     
259                 // skip first "const" in "const Type* const"
260                 if (tok->strAt(1) == "const")
261                     tok = tok->next();
262                 const Token* typeTok = tok->next();
263                 // skip second "const" in "const Type* const"
264                 if (tok->strAt(3) == "const")
265                     tok = tok->next();
266     
267                 if (tok->strAt(4) == "0") // Casting nullpointers is safe
268                     continue;
269     
270                 // Is "type" a class?
271                 if (typeTok->type())
272                     cstyleCastError(tok);
273             }
274         }
275     }

291     static std::string analyzeType(const Token* tok)
292     {
293         if (tok->str() == "double") {
294             if (tok->isLong())
295                 return "long double";
296             else
297                 return "double";
298         }
299         if (tok->str() == "float")
300             return "float";
301         if (Token::Match(tok, "int|long|short|char|size_t"))
302             return "integer";
303         return "";

306     void CheckOther::invalidPointerCast()
307     {
308         if (!_settings->isEnabled("portability"))
309             return;
310     
311         const bool printInconclusive = _settings->inconclusive;
312         const SymbolDatabase* const symbolDatabase = _tokenizer->getSymbolDatabase();
313         const std::size_t functions = symbolDatabase->functionScopes.size();
314         for (std::size_t i = 0; i < functions; ++i) {
315             const Scope * scope = symbolDatabase->functionScopes[i];
316             for (const Token* tok = scope->classStart->next(); tok != scope->classEnd; tok = tok->next()) {
317                 const Token* toTok = nullptr;
318                 const Token* nextTok = nullptr;
319                 // Find cast
320                 if (Token::Match(tok, "( const| %type% %type%| const| * )")) {
321                     toTok = tok->next();
322                     nextTok = tok->link()->next();
323                     if (nextTok && nextTok->str() == "(")
324                         nextTok = nextTok->next();
325                 } else if (Token::Match(tok, "reinterpret_cast < const| %type% %type%| const| * > (")) {
326                     nextTok = tok->tokAt(5);
327                     while (nextTok->str() != "(")
328                         nextTok = nextTok->next();
329                     nextTok = nextTok->next();
330                     toTok = tok->tokAt(2);
331                 }
332                 if (!nextTok)
333                     continue;
334                 if (toTok && toTok->str() == "const")
335                     toTok = toTok->next();
336     
337                 if (!toTok || !toTok->isStandardType())
338                     continue;
339     
340                 // Find casted variable
341                 const Variable *var = nullptr;
342                 bool allocation = false;
343                 bool ref = false;
344                 if (_tokenizer->isCPP() && Token::Match(nextTok, "new %type%"))
345                     allocation = true;
346                 else if (Token::Match(nextTok, "%var% !!["))
347                     var = nextTok->variable();
348                 else if (Token::Match(nextTok, "& %var%") && !Token::Match(nextTok->tokAt(2), "(|[")) {
349                     var = nextTok->next()->variable();
350                     ref = true;
351                 }
352     
353                 const Token* fromTok = nullptr;
354     
355                 if (allocation) {
356                     fromTok = nextTok->next();
357                 } else {
358                     if (!var || (!ref && !var->isPointer() && !var->isArray()) || (ref && (var->isPointer() || var->isArray())))
359                         continue;
360                     fromTok = var->typeStartToken();
361                 }
362     
363                 while (Token::Match(fromTok, "static|const"))
364                     fromTok = fromTok->next();
365                 if (!fromTok->isStandardType())
366                     continue;
367     
368                 std::string fromType = analyzeType(fromTok);
369                 std::string toType = analyzeType(toTok);
370                 if (fromType != toType && !fromType.empty() && !toType.empty() && (toTok->str() != "char" || printInconclusive))
371                     invalidPointerCastError(tok, fromType, toType, toTok->str() == "char");
372             }
373         }
374     }

462     static bool checkExceptionHandling(const Token* tok)
463     {
464         const Variable* var = tok->variable();
465         const Scope* upperScope = tok->scope();
466         if (var && upperScope == var->scope())
467             return true;
468         while (upperScope && upperScope->type != Scope::eTry && upperScope->type != Scope::eLambda && (!var || upperScope->nestedIn != var->scope()) && upperScope->isExecutable()) {
469             upperScope = upperScope->nestedIn;
470         }
471         if (var && upperScope && upperScope->type == Scope::eTry) {
472             // Check all exception han
473             const Token* tok2 = upperScope->classEnd;
474             while (Token::simpleMatch(tok2, "} catch (")) {
475                 tok2 = tok2->linkAt(2)->next();
476                 if (Token::findmatch(tok2, "%varid%", tok2->link(), var->declarationId()))
477                     return false;
478                 tok2 = tok2->link();
479             }
480         }
481         return true;
482     }

484     void CheckOther::checkRedundantAssignment()
485     {
486         const bool printPerformance = _settings->isEnabled("performance");
487         const bool printWarning = _settings->isEnabled("warning");
488         if (!printWarning && !printPerformance)
489             return;
490     
491         const bool printInconclusive = _settings->inconclusive;
492         const SymbolDatabase* symbolDatabase = _tokenizer->getSymbolDatabase();
493     
494         for (std::list<Scope>::const_iterator scope = symbolDatabase->scopeList.begin(); scope != symbolDatabase->scopeList.end(); ++scope) {
495             if (!scope->isExecutable())
496                 continue;
497     
498             std::map<unsigned int, const Token*> varAssignments;
499             std::map<unsigned int, const Token*> memAssignments;
500             std::map<unsigned int, std::set<unsigned int> > membervars;
501             std::set<unsigned int> initialized;
502             const Token* writtenArgumentsEnd = nullptr;
503     
504             for (const Token* tok = scope->classStart->next(); tok && tok != scope->classEnd; tok = tok->next()) {
505                 if (tok == writtenArgumentsEnd)
506                     writtenArgumentsEnd = nullptr;
507     
508                 if (tok->str() == "?" && tok->astOperand2()) {
509                     tok = Token::findmatch(tok->astOperand2(), ";|}");
510                     if (!tok)
511                         break;
512                     varAssignments.clear();
513                     memAssignments.clear();
514                 } else if (tok->str() == "{" && tok->strAt(-1) != "{" && tok->strAt(-1) != "=" && tok->strAt(-4) != "case" && tok->strAt(-3) != "default") { // conditional or non-executable inner scope: Skip it and reset status
515                     tok = tok->link();
516                     varAssignments.clear();
517                     memAssignments.clear();
518                 } else if (Token::Match(tok, "for|if|while (")) {
519                     tok = tok->linkAt(1);
520                 } else if (Token::Match(tok, "break|return|continue|throw|goto|asm")) {
521                     varAssignments.clear();
522                     memAssignments.clear();
523                 } else if (tok->tokType() == Token::eVariable && !Token::Match(tok, "%name% (")) {
524                     // Set initialization flag
525                     if (!Token::Match(tok, "%var% ["))
526                         initialized.insert(tok->varId());
527                     else {
528                         const Token *tok2 = tok->next();
529                         while (tok2 && tok2->str() == "[")
530                             tok2 = tok2->link()->next();
531                         if (tok2 && tok2->str() != ";")
532                             initialized.insert(tok->varId());
533                     }
534     
535                     const Token *startToken = tok;
536                     while (Token::Match(startToken, "%name%|::|.")) {
537                         startToken = startToken->previous();
538                         if (Token::Match(startToken, "%name% . %var%"))
539                             membervars[startToken->varId()].insert(startToken->tokAt(2)->varId());
540                     }
541     
542                     std::map<unsigned int, const Token*>::iterator it = varAssignments.find(tok->varId());
543                     if (Token::simpleMatch(tok->next(), "=") && Token::Match(startToken, "[;{}]")) { // Assignment
544                         if (it != varAssignments.end()) {
545                             bool error = true; // Ensure that variable is not used on right side
546                             for (const Token* tok2 = tok->tokAt(2); tok2; tok2 = tok2->next()) {
547                                 if (tok2->str() == ";")
548                                     break;
549                                 else if (tok2->varId() == tok->varId()) {
550                                     error = false;
551                                     break;
552                                 } else if (Token::Match(tok2, "%name% (") && nonLocal(tok->variable())) { // Called function might use the variable
553                                     const Function* const func = tok2->function();
554                                     const Variable* const var = tok->variable();
555                                     if (!var || var->isGlobal() || var->isReference() || ((!func || func->nestedIn) && tok2->strAt(-1) != ".")) {// Global variable, or member function
556                                         error = false;
557                                         break;
558                                     }
559                                 }
560                             }
561                             if (error) {
562                                 if (printWarning && scope->type == Scope::eSwitch && Token::findmatch(it->second, "default|case", tok))
563                                     redundantAssignmentInSwitchError(it->second, tok, tok->str());
564                                 else if (printPerformance) {
565                                     // See #7133
566                                     const bool nonlocal = it->second->variable() && nonLocal(it->second->variable());
567                                     if (printInconclusive || !nonlocal) // see #5089 - report inconclusive only when requested
568                                         if (_tokenizer->isC() || checkExceptionHandling(tok)) // see #6555 to see how exception handling might have an impact
569                                             redundantAssignmentError(it->second, tok, tok->str(), nonlocal); // Inconclusive for non-local variables
570                                 }
571                             }
572                             it->second = tok;
573                         }
574                         if (!Token::simpleMatch(tok->tokAt(2), "0 ;") || (tok->variable() && tok->variable()->nameToken() != tok->tokAt(-2)))
575                             varAssignments[tok->varId()] = tok;
576                         memAssignments.erase(tok->varId());
577                         eraseMemberAssignments(tok->varId(), membervars, varAssignments);
578                     } else if ((tok->next() && tok->next()->tokType() == Token::eIncDecOp) || (tok->previous()->tokType() == Token::eIncDecOp && tok->strAt(1) == ";")) { // Variable incremented/decremented; Prefix-Increment is only suspicious, if its return value is unused
579                         varAssignments[tok->varId()] = tok;
580                         memAssignments.erase(tok->varId());
581                         eraseMemberAssignments(tok->varId(), membervars, varAssignments);
582                     } else if (!Token::simpleMatch(tok->tokAt(-2), "sizeof (")) { // Other usage of variable
583                         if (it != varAssignments.end())
584                             varAssignments.erase(it);
585                         if (!writtenArgumentsEnd) // Indicates that we are in the first argument of strcpy/memcpy/... function
586                             memAssignments.erase(tok->varId());
587                     }
588                 } else if (Token::Match(tok, "%name% (") && _settings->library.functionpure.find(tok->str()) == _settings->library.functionpure.end()) { // Function call. Global variables might be used. Reset their status
589                     const bool memfunc = Token::Match(tok, "memcpy|memmove|memset|strcpy|strncpy|sprintf|snprintf|strcat|strncat|wcscpy|wcsncpy|swprintf|wcscat|wcsncat");
590                     if (tok->varId()) // operator() or function pointer
591                         varAssignments.erase(tok->varId());
592     
593                     if (memfunc && tok->strAt(-1) != "(" && tok->strAt(-1) != "=") {
594                         const Token* param1 = tok->tokAt(2);
595                         writtenArgumentsEnd = param1->next();
596                         if (param1->varId() && param1->strAt(1) == "," && !Token::Match(tok, "strcat|strncat|wcscat|wcsncat")) {
597                             if (tok->str() == "memset" && initialized.find(param1->varId()) == initialized.end() && param1->variable() && param1->variable()->isLocal() && param1->variable()->isArray())
598                                 initialized.insert(param1->varId());
599                             else {
600                                 const std::map<unsigned int, const Token*>::const_iterator it = memAssignments.find(param1->varId());
601                                 if (it == memAssignments.end())
602                                     memAssignments[param1->varId()] = tok;
603                                 else {
604                                     if (printWarning && scope->type == Scope::eSwitch && Token::findmatch(it->second, "default|case", tok))
605                                         redundantCopyInSwitchError(it->second, tok, param1->str());
606                                     else if (printPerformance)
607                                         redundantCopyError(it->second, tok, param1->str());
608                                 }
609                             }
610                         }
611                     } else if (scope->type == Scope::eSwitch) { // Avoid false positives if noreturn function is called in switch
612                         const Function* const func = tok->function();
613                         if (!func || !func->hasBody()) {
614                             varAssignments.clear();
615                             memAssignments.clear();
616                             continue;
617                         }
618                         const Token* funcEnd = func->functionScope->classEnd;
619                         bool noreturn;
620                         if (!_tokenizer->IsScopeNoReturn(funcEnd, &noreturn) && !noreturn) {
621                             eraseNotLocalArg(varAssignments, symbolDatabase);
622                             eraseNotLocalArg(memAssignments, symbolDatabase);
623                         } else {
624                             varAssignments.clear();
625                             memAssignments.clear();
626                         }
627                     } else { // Noreturn functions outside switch don't cause problems
628                         eraseNotLocalArg(varAssignments, symbolDatabase);
629                         eraseNotLocalArg(memAssignments, symbolDatabase);
630                     }
631                 }
632             }
633         }
634     }

671     //    switch (x)
672     //    {
673     //        case 2:
674     //            y = a;        // <- this assignment is redundant
675     //        case 3:
676     //            y = b;        // <- case 2 falls through and sets y twice
677     //    }

679     static inline bool isFunctionOrBreakPattern(const Token *tok)
680     {
681         if (Token::Match(tok, "%name% (") || Token::Match(tok, "break|continue|return|exit|goto|throw"))
682             return true;
683     
684         return false;

687     void CheckOther::checkRedundantAssignmentInSwitch()
688     {
689         if (!_settings->isEnabled("warning"))
690             return;
691     
692         const SymbolDatabase *symbolDatabase = _tokenizer->getSymbolDatabase();
693     
694         // Find the beginning of a switch. E.g.:
695         //   switch (var) { ...
696         for (std::list<Scope>::const_iterator i = symbolDatabase->scopeList.begin(); i != symbolDatabase->scopeList.end(); ++i) {
697             if (i->type != Scope::eSwitch || !i->classStart)
698                 continue;
699     
700             // Check the contents of the switch statement
701             std::map<unsigned int, const Token*> varsWithBitsSet;
702             std::map<unsigned int, std::string> bitOperations;
703     
704             for (const Token *tok2 = i->classStart->next(); tok2 != i->classEnd; tok2 = tok2->next()) {
705                 if (tok2->str() == "{") {
706                     // Inside a conditional or loop. Don't mark variable accesses as being redundant. E.g.:
707                     //   case 3: b = 1;
708                     //   case 4: if (a) { b = 2; }    // Doesn't make the b=1 redundant because it's conditional
709                     if (Token::Match(tok2->previous(), ")|else {") && tok2->link()) {
710                         const Token* endOfConditional = tok2->link();
711                         for (const Token* tok3 = tok2; tok3 != endOfConditional; tok3 = tok3->next()) {
712                             if (tok3->varId() != 0) {
713                                 varsWithBitsSet.erase(tok3->varId());
714                                 bitOperations.erase(tok3->varId());
715                             } else if (isFunctionOrBreakPattern(tok3)) {
716                                 varsWithBitsSet.clear();
717                                 bitOperations.clear();
718                             }
719                         }
720                         tok2 = endOfConditional;
721                     }
722                 }
723     
724                 // Variable assignment. Report an error if it's assigned to twice before a break. E.g.:
725                 //    case 3: b = 1;    // <== redundant
726                 //    case 4: b = 2;
727     
728                 if (Token::Match(tok2->previous(), ";|{|}|: %var% = %any% ;")) {
729                     varsWithBitsSet.erase(tok2->varId());
730                     bitOperations.erase(tok2->varId());
731                 }
732     
733                 // Bitwise operation. Report an error if it's performed twice before a break. E.g.:
734                 //    case 3: b |= 1;    // <== redundant
735                 //    case 4: b |= 1;
736                 else if (Token::Match(tok2->previous(), ";|{|}|: %var% %assign% %num% ;") &&
737                          (tok2->strAt(1) == "|=" || tok2->strAt(1) == "&=") &&
738                          Token::Match(tok2->next()->astOperand2(), "%num%")) {
739                     std::string bitOp = tok2->strAt(1)[0] + tok2->strAt(2);
740                     std::map<unsigned int, const Token*>::const_iterator i2 = varsWithBitsSet.find(tok2->varId());
741     
742                     // This variable has not had a bit operation performed on it yet, so just make a note of it
743                     if (i2 == varsWithBitsSet.end()) {
744                         varsWithBitsSet[tok2->varId()] = tok2;
745                         bitOperations[tok2->varId()] = bitOp;
746                     }
747     
748                     // The same bit operation has been performed on the same variable twice, so report an error
749                     else if (bitOperations[tok2->varId()] == bitOp)
750                         redundantBitwiseOperationInSwitchError(i2->second, i2->second->str());
751     
752                     // A different bit operation was performed on the variable, so clear it
753                     else {
754                         varsWithBitsSet.erase(tok2->varId());
755                         bitOperations.erase(tok2->varId());
756                     }
757                 }
758     
759                 // Bitwise operation. Report an error if it's performed twice before a break. E.g.:
760                 //    case 3: b = b | 1;    // <== redundant
761                 //    case 4: b = b | 1;
762                 else if (Token::Match(tok2->previous(), ";|{|}|: %var% = %name% %or%|& %num% ;") &&
763                          tok2->varId() == tok2->tokAt(2)->varId()) {
764                     std::string bitOp = tok2->strAt(3) + tok2->strAt(4);
765                     std::map<unsigned int, const Token*>::const_iterator i2 = varsWithBitsSet.find(tok2->varId());
766     
767                     // This variable has not had a bit operation performed on it yet, so just make a note of it
768                     if (i2 == varsWithBitsSet.end()) {
769                         varsWithBitsSet[tok2->varId()] = tok2;
770                         bitOperations[tok2->varId()] = bitOp;
771                     }
772     
773                     // The same bit operation has been performed on the same variable twice, so report an error
774                     else if (bitOperations[tok2->varId()] == bitOp)
775                         redundantBitwiseOperationInSwitchError(i2->second, i2->second->str());
776     
777                     // A different bit operation was performed on the variable, so clear it
778                     else {
779                         varsWithBitsSet.erase(tok2->varId());
780                         bitOperations.erase(tok2->varId());
781                     }
782                 }
783     
784                 // Not a simple assignment so there may be good reason if this variable is assigned to twice. E.g.:
785                 //    case 3: b = 1;
786                 //    case 4: b++;
787                 else if (tok2->varId() != 0 && tok2->strAt(1) != "|" && tok2->strAt(1) != "&") {
788                     varsWithBitsSet.erase(tok2->varId());
789                     bitOperations.erase(tok2->varId());
790                 }
791     
792                 // Reset our record of assignments if there is a break or function call. E.g.:
793                 //    case 3: b = 1; break;
794                 if (isFunctionOrBreakPattern(tok2)) {
795                     varsWithBitsSet.clear();
796                     bitOperations.clear();
797                 }
798             }
799         }
800     }

809     //---------------------------------------------------------------------------
810     // Detect fall through cases (experimental).
811     //---------------------------------------------------------------------------
812     void CheckOther::checkSwitchCaseFallThrough()
813     {
814         if (!(_settings->isEnabled("style") && _settings->experimental))
815             return;
816     
817         const SymbolDatabase* const symbolDatabase = _tokenizer->getSymbolDatabase();
818     
819         for (std::list<Scope>::const_iterator i = symbolDatabase->scopeList.begin(); i != symbolDatabase->scopeList.end(); ++i) {
820             if (i->type != Scope::eSwitch || !i->classStart) // Find the beginning of a switch
821                 continue;
822     
823             // Check the contents of the switch statement
824             std::stack<std::pair<Token *, bool> > ifnest;
825             bool justbreak = true;
826             bool firstcase = true;
827             for (const Token *tok2 = i->classStart; tok2 != i->classEnd; tok2 = tok2->next()) {
828                 if (Token::simpleMatch(tok2, "if (")) {
829                     tok2 = tok2->next()->link()->next();
830                     ifnest.push(std::make_pair(tok2->link(), false));
831                     justbreak = false;
832                 } else if (Token::simpleMatch(tok2, "while (")) {
833                     tok2 = tok2->next()->link()->next();
834                     if (tok2->link()) // skip over "do { } while ( ) ;" case
835                         tok2 = tok2->link();
836                     justbreak = false;
837                 } else if (Token::simpleMatch(tok2, "do {")) {
838                     tok2 = tok2->next()->link();
839                     justbreak = false;
840                 } else if (Token::simpleMatch(tok2, "for (")) {
841                     tok2 = tok2->next()->link()->next()->link();
842                     justbreak = false;
843                 } else if (Token::simpleMatch(tok2, "switch (")) {
844                     // skip over nested switch, we'll come to that soon
845                     tok2 = tok2->next()->link()->next()->link();
846                 } else if (Token::Match(tok2, "break|continue|return|exit|goto|throw")) {
847                     justbreak = true;
848                     tok2 = Token::findsimplematch(tok2, ";");
849                 } else if (Token::Match(tok2, "case|default")) {
850                     if (!justbreak && !firstcase) {
851                         switchCaseFallThrough(tok2);
852                     }
853                     tok2 = Token::findsimplematch(tok2, ":");
854                     justbreak = true;
855                     firstcase = false;
856                 } else if (tok2->str() == "}") {
857                     if (!ifnest.empty() && tok2 == ifnest.top().first) {
858                         if (tok2->next()->str() == "else") {
859                             tok2 = tok2->tokAt(2);
860                             ifnest.pop();
861                             ifnest.push(std::make_pair(tok2->link(), justbreak));
862                             justbreak = false;
863                         } else {
864                             justbreak &= ifnest.top().second;
865                             ifnest.pop();
866                         }
867                     }
868                 } else if (tok2->str() != ";") {
869                     justbreak = false;
870                 }
871             }
872         }
873     }

924     //---------------------------------------------------------------------------
925     //    if (x == 1)
926     //        x == 0;       // <- suspicious equality comparison.
927     //---------------------------------------------------------------------------
928     void CheckOther::checkSuspiciousEqualityComparison()
929     {
930         if (!_settings->isEnabled("warning") || !_settings->inconclusive)
931             return;
932     
933         const SymbolDatabase* symbolDatabase = _tokenizer->getSymbolDatabase();
934         const std::size_t functions = symbolDatabase->functionScopes.size();
935         for (std::size_t i = 0; i < functions; ++i) {
936             const Scope * scope = symbolDatabase->functionScopes[i];
937             for (const Token* tok = scope->classStart; tok != scope->classEnd; tok = tok->next()) {
938                 if (Token::simpleMatch(tok, "for (")) {
939                     const Token* const openParen = tok->next();
940                     const Token* const closeParen = tok->linkAt(1);
941     
942                     // Search for any suspicious equality comparison in the initialization
943                     // or increment-decrement parts of the for() loop.
944                     // For example:
945                     //    for (i == 2; i < 10; i++)
946                     // or
947                     //    for (i = 0; i < 10; i == a)
948                     if (Token::Match(openParen->next(), "%name% =="))
949                         suspiciousEqualityComparisonError(openParen->tokAt(2));
950                     if (closeParen->strAt(-2) == "==")
951                         suspiciousEqualityComparisonError(closeParen->tokAt(-2));
952     
953                     // Skip over for() loop conditions because "for (;running==1;)"
954                     // is a bit strange, but not necessarily incorrect.
955                     tok = closeParen;
956                 } else if (Token::Match(tok, "[;{}] *| %name% == %any% ;")) {
957     
958                     // Exclude compound statements surrounded by parentheses, such as
959                     //    printf("%i\n", ({x==0;}));
960                     // because they may appear as an expression in GNU C/C++.
961                     // See http://gcc.gnu.org/onlinedocs/gcc/Statement-Exprs.html
962                     const Token* afterStatement = tok->strAt(1) == "*" ? tok->tokAt(6) : tok->tokAt(5);
963                     if (!Token::simpleMatch(afterStatement, "} )"))
964                         suspiciousEqualityComparisonError(tok->next());
965                 }
966             }
967         }
968     }

977     //---------------------------------------------------------------------------
978     //    Find consecutive return, break, continue, goto or throw statements. e.g.:
979     //        break; break;
980     //    Detect dead code, that follows such a statement. e.g.:
981     //        return(0); foo();
982     //---------------------------------------------------------------------------
983     void CheckOther::checkUnreachableCode()
984     {
985         if (!_settings->isEnabled("style"))
986             return;
987         const bool printInconclusive = _settings->inconclusive;
988         const SymbolDatabase* symbolDatabase = _tokenizer->getSymbolDatabase();
989         const std::size_t functions = symbolDatabase->functionScopes.size();
990         for (std::size_t i = 0; i < functions; ++i) {
991             const Scope * scope = symbolDatabase->functionScopes[i];
992     
993             for (const Token* tok = scope->classStart; tok && tok != scope->classEnd; tok = tok->next()) {
994                 const Token* secondBreak = 0;
995                 const Token* labelName = 0;
996                 if (tok->link() && Token::Match(tok, "(|[|<"))
997                     tok = tok->link();
998                 else if (Token::Match(tok, "break|continue ;"))
999                     secondBreak = tok->tokAt(2);
1000                 else if (Token::Match(tok, "[;{}:] return|throw")) {
1001                     tok = tok->next(); // tok should point to return or throw
1002                     for (const Token *tok2 = tok->next(); tok2; tok2 = tok2->next()) {
1003                         if (tok2->str() == "(" || tok2->str() == "{")
1004                             tok2 = tok2->link();
1005                         if (tok2->str() == ";") {
1006                             secondBreak = tok2->next();
1007                             break;
1008                         }
1009                     }
1010                 } else if (Token::Match(tok, "goto %any% ;")) {
1011                     secondBreak = tok->tokAt(3);
1012                     labelName = tok->next();
1013                 } else if (Token::Match(tok, "%name% (") && _settings->library.isnoreturn(tok) && !Token::Match(tok->next()->astParent(), "?|:")) {
1014                     if ((!tok->function() || (tok->function()->token != tok && tok->function()->tokenDef != tok)) && tok->linkAt(1)->strAt(1) != "{")
1015                         secondBreak = tok->linkAt(1)->tokAt(2);
1016                 }
1017     
1018                 // Statements follow directly, no line between them. (#3383)
1019                 // TODO: Try to find a better way to avoid false positives due to preprocessor configurations.
1020                 const bool inconclusive = secondBreak && (secondBreak->linenr() - 1 > secondBreak->previous()->linenr());
1021     
1022                 if (secondBreak && (printInconclusive || !inconclusive)) {
1023                     if (Token::Match(secondBreak, "continue|goto|throw") ||
1024                         (secondBreak->str() == "return" && (tok->str() == "return" || secondBreak->strAt(1) == ";"))) { // return with value after statements like throw can be necessary to make a function compile
1025                         duplicateBreakError(secondBreak, inconclusive);
1026                         tok = Token::findmatch(secondBreak, "[}:]");
1027                     } else if (secondBreak->str() == "break") { // break inside switch as second break statement should not issue a warning
1028                         if (tok->str() == "break") // If the previous was a break, too: Issue warning
1029                             duplicateBreakError(secondBreak, inconclusive);
1030                         else {
1031                             if (tok->scope()->type != Scope::eSwitch) // Check, if the enclosing scope is a switch
1032                                 duplicateBreakError(secondBreak, inconclusive);
1033                         }
1034                         tok = Token::findmatch(secondBreak, "[}:]");
1035                     } else if (!Token::Match(secondBreak, "return|}|case|default") && secondBreak->strAt(1) != ":") { // TODO: No bailout for unconditional scopes
1036                         // If the goto label is followed by a loop construct in which the label is defined it's quite likely
1037                         // that the goto jump was intended to skip some code on the first loop iteration.
1038                         bool labelInFollowingLoop = false;
1039                         if (labelName && Token::Match(secondBreak, "while|do|for")) {
1040                             const Token *scope2 = Token::findsimplematch(secondBreak, "{");
1041                             if (scope2) {
1042                                 for (const Token *tokIter = scope2; tokIter != scope2->link() && tokIter; tokIter = tokIter->next()) {
1043                                     if (Token::Match(tokIter, "[;{}] %any% :") && labelName->str() == tokIter->strAt(1)) {
1044                                         labelInFollowingLoop = true;
1045                                         break;
1046                                     }
1047                                 }
1048                             }
1049                         }
1050     
1051                         // hide FP for statements that just hide compiler warnings about unused function arguments
1052                         bool silencedCompilerWarningOnly = false;
1053                         const Token *silencedWarning = secondBreak;
1054                         for (;;) {
1055                             if (Token::Match(silencedWarning, "( void ) %name% ;")) {
1056                                 silencedWarning = silencedWarning->tokAt(5);
1057                                 continue;
1058                             } else if (silencedWarning && silencedWarning == scope->classEnd)
1059                                 silencedCompilerWarningOnly = true;
1060     
1061                             break;
1062                         }
1063                         if (silencedWarning)
1064                             secondBreak = silencedWarning;
1065     
1066                         if (!labelInFollowingLoop && !silencedCompilerWarningOnly)
1067                             unreachableCodeError(secondBreak, inconclusive);

1182     //---------------------------------------------------------------------------
1183     // Check scope of variables..
1184     //---------------------------------------------------------------------------
1185     void CheckOther::checkVariableScope()
1186     {
1187         if (!_settings->isEnabled("style"))
1188             return;
1189     
1190         const SymbolDatabase *symbolDatabase = _tokenizer->getSymbolDatabase();
1191     
1192         for (unsigned int i = 1; i < symbolDatabase->getVariableListSize(); i++) {
1193             const Variable* var = symbolDatabase->getVariableFromVarId(i);
1194             if (!var || !var->isLocal() || (!var->isPointer() && !var->typeStartToken()->isStandardType()))
1195                 continue;
1196     
1197             if (var->isConst())
1198                 continue;
1199     
1200             bool forHead = false; // Don't check variables declared in header of a for loop
1201             for (const Token* tok = var->typeStartToken(); tok; tok = tok->previous()) {
1202                 if (tok->str() == "(") {
1203                     forHead = true;
1204                     break;
1205                 } else if (tok->str() == "{" || tok->str() == ";" || tok->str() == "}")
1206                     break;
1207             }
1208             if (forHead)
1209                 continue;
1210     
1211             const Token* tok = var->nameToken()->next();
1212             if (Token::Match(tok, "; %varid% = %any% ;", var->declarationId())) {
1213                 tok = tok->tokAt(3);
1214                 if (!tok->isNumber() && tok->tokType() != Token::eString && tok->tokType() != Token::eChar && !tok->isBoolean())
1215                     continue;
1216             }
1217             bool reduce = true;
1218             bool used = false; // Don't warn about unused variables
1219             for (; tok && tok != var->scope()->classEnd; tok = tok->next()) {
1220                 if (tok->str() == "{" && tok->scope() != tok->previous()->scope() && !tok->isExpandedMacro() && tok->scope()->type != Scope::eLambda) {
1221                     if (used) {
1222                         bool used2 = false;
1223                         if (!checkInnerScope(tok, var, used2) || used2) {
1224                             reduce = false;
1225                             break;
1226                         }
1227                     } else if (!checkInnerScope(tok, var, used)) {
1228                         reduce = false;
1229                         break;
1230                     }
1231     
1232                     tok = tok->link();
1233     
1234                     // parse else if blocks..
1235                 } else if (Token::simpleMatch(tok, "else { if (") && Token::simpleMatch(tok->linkAt(3), ") {")) {
1236                     const Token *endif = tok->linkAt(3)->linkAt(1);
1237                     bool elseif = false;
1238                     if (Token::simpleMatch(endif, "} }"))
1239                         elseif = true;
1240                     else if (Token::simpleMatch(endif, "} else {") && Token::simpleMatch(endif->linkAt(2),"} }"))
1241                         elseif = true;
1242                     if (elseif && Token::findmatch(tok->next(), "%varid%", tok->linkAt(1), var->declarationId())) {
1243                         reduce = false;
1244                         break;
1245                     }
1246                 } else if (tok->varId() == var->declarationId() || tok->str() == "goto") {
1247                     reduce = false;
1248                     break;
1249                 }
1250             }
1251     
1252             if (reduce && used)
1253                 variableScopeError(var->nameToken(), var->name());
1254         }
1255     }

1257     bool CheckOther::checkInnerScope(const Token *tok, const Variable* var, bool& used)
1258     {
1259         const Scope* scope = tok->next()->scope();
1260         bool loopVariable = scope->type == Scope::eFor || scope->type == Scope::eWhile || scope->type == Scope::eDo;
1261         bool noContinue = true;
1262         const Token* forHeadEnd = 0;
1263         const Token* end = tok->link();
1264         if (scope->type == Scope::eUnconditional && (tok->strAt(-1) == ")" || tok->previous()->isName())) // Might be an unknown macro like BOOST_FOREACH
1265             loopVariable = true;
1266     
1267         if (scope->type == Scope::eDo) {
1268             end = end->linkAt(2);
1269         } else if (loopVariable && tok->strAt(-1) == ")") {
1270             tok = tok->linkAt(-1); // Jump to opening ( of for/while statement
1271         } else if (scope->type == Scope::eSwitch) {
1272             for (std::list<Scope*>::const_iterator i = scope->nestedList.begin(); i != scope->nestedList.end(); ++i) {
1273                 if (used) {
1274                     bool used2 = false;
1275                     if (!checkInnerScope((*i)->classStart, var, used2) || used2) {
1276                         return false;
1277                     }
1278                 } else if (!checkInnerScope((*i)->classStart, var, used)) {
1279                     return false;
1280                 }
1281             }
1282         }
1283     
1284         bool bFirstAssignment=false;
1285         for (; tok && tok != end; tok = tok->next()) {
1286             if (tok->str() == "goto")
1287                 return false;
1288             if (tok->str() == "continue")
1289                 noContinue = false;
1290     
1291             if (Token::simpleMatch(tok, "for ("))
1292                 forHeadEnd = tok->linkAt(1);
1293             if (tok == forHeadEnd)
1294                 forHeadEnd = 0;
1295     
1296             if (loopVariable && noContinue && tok->scope() == scope && !forHeadEnd && scope->type != Scope::eSwitch && Token::Match(tok, "%varid% =", var->declarationId())) { // Assigned in outer scope.
1297                 loopVariable = false;
1298                 unsigned int indent = 0;
1299                 for (const Token* tok2 = tok->tokAt(2); tok2; tok2 = tok2->next()) { // Ensure that variable isn't used on right side of =, too
1300                     if (tok2->str() == "(")
1301                         indent++;
1302                     else if (tok2->str() == ")") {
1303                         if (indent == 0)
1304                             break;
1305                         indent--;
1306                     } else if (tok2->str() == ";")
1307                         break;
1308                     else if (tok2->varId() == var->declarationId()) {
1309                         loopVariable = true;
1310                         break;
1311                     }
1312                 }
1313             }
1314     
1315             if (loopVariable && Token::Match(tok, "%varid% !!=", var->declarationId())) // Variable used in loop
1316                 return false;
1317     
1318             if (Token::Match(tok, "& %varid%", var->declarationId())) // Taking address of variable
1319                 return false;
1320     
1321             if (Token::Match(tok, "%varid% =", var->declarationId()))
1322                 bFirstAssignment = true;
1323     
1324             if (!bFirstAssignment && Token::Match(tok, "* %varid%", var->declarationId())) // dereferencing means access to previous content
1325                 return false;
1326     
1327             if (Token::Match(tok, "= %varid%", var->declarationId()) && (var->isArray() || var->isPointer())) // Create a copy of array/pointer. Bailout, because the memory it points to might be necessary in outer scope
1328                 return false;
1329     
1330             if (tok->varId() == var->declarationId()) {
1331                 used = true;
1332                 if (scope->type == Scope::eSwitch && scope == tok->scope())
1333                     return false; // Used in outer switch scope - unsafe or impossible to reduce scope
1334             }
1335         }
1336     
1337         return true;
1338     }

1530     void CheckOther::checkIncompleteStatement()
1531     {
1532         if (!_settings->isEnabled("warning"))
1533             return;
1534     
1535         for (const Token *tok = _tokenizer->tokens(); tok; tok = tok->next()) {
1536             if (tok->str() == "(") {
1537                 tok = tok->link();
1538                 if (Token::simpleMatch(tok, ") {") && Token::simpleMatch(tok->next()->link(), "} ;"))
1539                     tok = tok->next()->link();
1540             }
1541     
1542             else if (Token::simpleMatch(tok, "= {"))
1543                 tok = tok->next()->link();
1544     
1545             // C++11 struct/array/etc initialization in initializer list
1546             else if (Token::Match(tok->previous(), "%name%|] {") && !Token::findsimplematch(tok,";",tok->link()))
1547                 tok = tok->link();
1548     
1549             // C++11 vector initialization / return { .. }
1550             else if (Token::Match(tok,"> %name% {") || Token::Match(tok, "[;{}] return {"))
1551                 tok = tok->linkAt(2);
1552     
1553             // C++11 initialize set in initalizer list : [,:] std::set<int>{1} [{,]
1554             else if (Token::simpleMatch(tok,"> {") && tok->link())
1555                 tok = tok->next()->link();
1556     
1557             else if (Token::Match(tok, "[;{}] %str%|%num%")) {
1558                 // No warning if numeric constant is followed by a "." or ","
1559                 if (Token::Match(tok->next(), "%num% [,.]"))
1560                     continue;
1561     
1562                 // No warning for [;{}] (void *) 0 ;
1563                 if (Token::Match(tok, "[;{}] 0 ;") && (tok->next()->isCast() || tok->next()->isExpandedMacro()))
1564                     continue;
1565     
1566                 // bailout if there is a "? :" in this statement
1567                 bool bailout = false;
1568                 for (const Token *tok2 = tok->tokAt(2); tok2; tok2 = tok2->next()) {
1569                     if (tok2->str() == "?") {
1570                         bailout = true;
1571                         break;
1572                     } else if (tok2->str() == ";")
1573                         break;
1574                 }
1575                 if (bailout)
1576                     continue;
1577     
1578                 // no warning if this is the last statement in a ({})
1579                 for (const Token *tok2 = tok->next(); tok2; tok2 = tok2->next()) {
1580                     if (tok2->str() == "(")
1581                         tok2 = tok2->link();
1582                     else if (Token::Match(tok2, "[;{}]")) {
1583                         bailout = Token::simpleMatch(tok2, "; } )");
1584                         break;
1585                     }
1586                 }
1587                 if (bailout)
1588                     continue;
1589     
1590                 constStatementError(tok->next(), tok->next()->isNumber() ? "numeric" : "string");
1591             }
1592         }
1593     }

1711     //-----------------------------------------------------------------------------
1712     // check for duplicate code in if and else branches
1713     // if (a) { b = true; } else { b = true; }
1714     //-----------------------------------------------------------------------------
1715     void CheckOther::checkDuplicateBranch()
1716     {
1717         // This is inconclusive since in practice most warnings are noise:
1718         // * There can be unfixed low-priority todos. The code is fine as it
1719         //   is but it could be possible to enhance it. Writing a warning
1720         //   here is noise since the code is fine (see cppcheck, abiword, ..)
1721         // * There can be overspecified code so some conditions can't be true
1722         //   and their conditional code is a duplicate of the condition that
1723         //   is always true just in case it would be false. See for instance
1724         //   abiword.
1725         if (!_settings->isEnabled("style") || !_settings->inconclusive)
1726             return;
1727     
1728         const SymbolDatabase *symbolDatabase = _tokenizer->getSymbolDatabase();
1729     
1730         std::list<Scope>::const_iterator scope;
1731     
1732         for (scope = symbolDatabase->scopeList.begin(); scope != symbolDatabase->scopeList.end(); ++scope) {
1733             if (scope->type != Scope::eIf)
1734                 continue;
1735     
1736             // check all the code in the function for if (..) else
1737             if (Token::simpleMatch(scope->classEnd, "} else {")) {
1738                 // Make sure there are no macros (different macros might be expanded
1739                 // to the same code)
1740                 bool macro = false;
1741                 for (const Token *tok = scope->classStart; tok != scope->classEnd->linkAt(2); tok = tok->next()) {
1742                     if (tok->isExpandedMacro()) {
1743                         macro = true;
1744                         break;
1745                     }
1746                 }
1747                 if (macro)
1748                     continue;
1749     
1750                 // save if branch code
1751                 std::string branch1 = scope->classStart->next()->stringifyList(scope->classEnd);
1752     
1753                 if (branch1.empty())
1754                     continue;
1755     
1756                 // save else branch code
1757                 std::string branch2 = scope->classEnd->tokAt(3)->stringifyList(scope->classEnd->linkAt(2));
1758     
1759                 // check for duplicates
1760                 if (branch1 == branch2)
1761                     duplicateBranchError(scope->classDef, scope->classEnd->next());
1762             }
1763         }
1764     }

1777     //-----------------------------------------------------------------------------
1778     // Check for a free() of an invalid address
1779     // char* p = malloc(100);
1780     // free(p + 10);
1781     //-----------------------------------------------------------------------------
1782     void CheckOther::checkInvalidFree()
1783     {
1784         std::map<unsigned int, bool> allocatedVariables;
1785     
1786         const bool printInconclusive = _settings->inconclusive;
1787         const SymbolDatabase* symbolDatabase = _tokenizer->getSymbolDatabase();
1788         const std::size_t functions = symbolDatabase->functionScopes.size();
1789         for (std::size_t i = 0; i < functions; ++i) {
1790             const Scope * scope = symbolDatabase->functionScopes[i];
1791             for (const Token* tok = scope->classStart->next(); tok != scope->classEnd; tok = tok->next()) {
1792     
1793                 // Keep track of which variables were assigned addresses to newly-allocated memory
1794                 if (Token::Match(tok, "%var% = malloc|g_malloc|new")) {
1795                     allocatedVariables.insert(std::make_pair(tok->varId(), false));
1796                 }
1797     
1798                 // If a previously-allocated pointer is incremented or decremented, any subsequent
1799                 // free involving pointer arithmetic may or may not be invalid, so we should only
1800                 // report an inconclusive result.
1801                 else if (Token::Match(tok, "%var% = %name% +|-") &&
1802                          tok->varId() == tok->tokAt(2)->varId() &&
1803                          allocatedVariables.find(tok->varId()) != allocatedVariables.end()) {
1804                     if (printInconclusive)
1805                         allocatedVariables[tok->varId()] = true;
1806                     else
1807                         allocatedVariables.erase(tok->varId());
1808                 }
1809     
1810                 // If a previously-allocated pointer is assigned a completely new value,
1811                 // we can't know if any subsequent free() on that pointer is valid or not.
1812                 else if (Token::Match(tok, "%var% =")) {
1813                     allocatedVariables.erase(tok->varId());
1814                 }
1815     
1816                 // If a variable that was previously assigned a newly-allocated memory location is
1817                 // added or subtracted from when used to free the memory, report an error.
1818                 else if (Token::Match(tok, "free|g_free|delete ( %any% +|-") ||
1819                          Token::Match(tok, "delete [ ] ( %any% +|-") ||
1820                          Token::Match(tok, "delete %any% +|- %any%")) {
1821     
1822                     const int varIndex = tok->strAt(1) == "(" ? 2 :
1823                                          tok->strAt(3) == "(" ? 4 : 1;
1824                     const unsigned int var1 = tok->tokAt(varIndex)->varId();
1825                     const unsigned int var2 = tok->tokAt(varIndex + 2)->varId();
1826                     const std::map<unsigned int, bool>::const_iterator alloc1 = allocatedVariables.find(var1);
1827                     const std::map<unsigned int, bool>::const_iterator alloc2 = allocatedVariables.find(var2);
1828                     if (alloc1 != allocatedVariables.end()) {
1829                         invalidFreeError(tok, alloc1->second);
1830                     } else if (alloc2 != allocatedVariables.end()) {
1831                         invalidFreeError(tok, alloc2->second);
1832                     }
1833                 }
1834     
1835                 // If the previously-allocated variable is passed in to another function
1836                 // as a parameter, it might be modified, so we shouldn't report an error
1837                 // if it is later used to free memory
1838                 else if (Token::Match(tok, "%name% (") && _settings->library.functionpure.find(tok->str()) == _settings->library.functionpure.end()) {
1839                     const Token* tok2 = Token::findmatch(tok->next(), "%var%", tok->linkAt(1));
1840                     while (tok2 != nullptr) {
1841                         allocatedVariables.erase(tok2->varId());
1842                         tok2 = Token::findmatch(tok2->next(), "%var%", tok->linkAt(1));
1843                     }
1844                 }
1845             }
1846         }
1847     }

1896             return;

1898         // Parse all executing scopes..

1903         const std::set<std::string> temp; // Can be used as dummy for isSameExpression()
1904         getConstFunctions(symbolDatabase, constFunctions);
1905     
1906         for (scope = symbolDatabase->scopeList.begin(); scope != symbolDatabase->scopeList.end(); ++scope) {
1907             // only check functions

1916                         if (isWithoutSideEffects(_tokenizer->isCPP(), tok->astOperand1())) {

1922                                     const Token* parent = tok->astParent();
1923                                     while (parent && parent->astParent()) {
1924                                         parent = parent->astParent();
1925                                     }
1926                                     if (parent && parent->previous() && parent->previous()->str() == "static_assert") {
1927                                         continue;
1928                                     }
1929                                 }
1930                                 if (styleEnabled)
1931                                     duplicateExpressionError(tok, tok, tok->str());
1932                             }
1933                         }
1934                     } else if (!Token::Match(tok, "[-/%]")) { // These operators are not associative
1935                         if (styleEnabled && tok->astOperand2() && tok->str() == tok->astOperand1()->str() && isSameExpression(_tokenizer->isCPP(), true, tok->astOperand2(), tok->astOperand1()->astOperand2(), _settings->library.functionpure) && isWithoutSideEffects(_tokenizer->isCPP(), tok->astOperand2()))
1936                             duplicateExpressionError(tok->astOperand2(), tok->astOperand2(), tok->str());

1940                                 if (isSameExpression(_tokenizer->isCPP(), true, ast1->astOperand1(), tok->astOperand2(), _settings->library.functionpure) && isWithoutSideEffects(_tokenizer->isCPP(), ast1->astOperand1()))
1941                                     // TODO: warn if variables are unchanged. See #5683
1942                                     // Probably the message should be changed to 'duplicate expressions X in condition or something like that'.
1943                                     ;//duplicateExpressionError(ast1->astOperand1(), tok->astOperand2(), tok->str());
1944                                 else if (styleEnabled && isSameExpression(_tokenizer->isCPP(), true, ast1->astOperand2(), tok->astOperand2(), _settings->library.functionpure) && isWithoutSideEffects(_tokenizer->isCPP(), ast1->astOperand2()))
1945                                     duplicateExpressionError(ast1->astOperand2(), tok->astOperand2(), tok->str());
1946                                 if (!isConstExpression(ast1->astOperand2(), _settings->library.functionpure))
1947                                     break;
1948                                 ast1 = ast1->astOperand1();
1949                             }
1950                         }
1951                     }
1952                 } else if (styleEnabled && tok->astOperand1() && tok->astOperand2() && tok->str() == ":" && tok->astParent() && tok->astParent()->str() == "?") {
1953                     if (isSameExpression(_tokenizer->isCPP(), true, tok->astOperand1(), tok->astOperand2(), temp))

2508     void CheckOther::checkEvaluationOrder()
2509     {
2510         // This checker is not written according to C++11 sequencing rules
2511         if (_tokenizer->isCPP() && _settings->standards.cpp >= Standards::CPP11)
2512             return;
2513     
2514         const SymbolDatabase *symbolDatabase = _tokenizer->getSymbolDatabase();
2515         const std::size_t functions = symbolDatabase->functionScopes.size();
2516         for (std::size_t i = 0; i < functions; ++i) {
2517             const Scope * functionScope = symbolDatabase->functionScopes[i];
2518             for (const Token* tok = functionScope->classStart; tok != functionScope->classEnd; tok = tok->next()) {
2519                 if (!Token::Match(tok, "++|--") && !tok->isAssignmentOp())
2520                     continue;
2521                 if (!tok->astOperand1())
2522                     continue;
2523                 for (const Token *tok2 = tok;; tok2 = tok2->astParent()) {
2524                     // If ast parent is a sequence point then break
2525                     const Token * const parent = tok2->astParent();
2526                     if (!parent)
2527                         break;
2528                     if (Token::Match(parent, "%oror%|&&|?|:|;"))
2529                         break;
2530                     if (parent->str() == ",") {
2531                         const Token *par = parent;
2532                         while (Token::simpleMatch(par,","))
2533                             par = par->astParent();
2534                         // not function => break
2535                         if (!(par && par->str() == "(" && par->astOperand2()))
2536                             break;
2537                         // sequence point in function argument: dostuff((1,2),3) => break
2538                         par = par->next();
2539                         while (par && (par->previous() != parent))
2540                             par = par->nextArgument();
2541                         if (!par)
2542                             break;
2543                     }
2544                     if (parent->str() == "(" && parent->astOperand2())
2545                         break;
2546     
2547                     // self assignment..
2548                     if (tok2 == tok &&
2549                         tok->str() == "=" &&
2550                         parent->str() == "=" &&
2551                         isSameExpression(_tokenizer->isCPP(), false, tok->astOperand1(), parent->astOperand1(), _settings->library.functionpure)) {
2552                         if (_settings->isEnabled("warning") &&
2553                             isSameExpression(_tokenizer->isCPP(), true, tok->astOperand1(), parent->astOperand1(), _settings->library.functionpure))
2554                             selfAssignmentError(parent, tok->astOperand1()->expressionString());
2555                         break;
2556                     }
2557     
2558                     // Is expression used?
2559                     bool foundError = false;
2560                     std::stack<const Token *> tokens;
2561                     tokens.push((parent->astOperand1() != tok2) ? parent->astOperand1() : parent->astOperand2());
2562                     while (!tokens.empty() && !foundError) {
2563                         const Token * const tok3 = tokens.top();
2564                         tokens.pop();
2565                         if (!tok3)
2566                             continue;
2567                         if (tok3->str() == "&" && !tok3->astOperand2())
2568                             continue; // don't handle adress-of for now
2569                         if (tok3->str() == "(" && Token::simpleMatch(tok3->previous(), "sizeof"))
2570                             continue; // don't care about sizeof usage
2571                         tokens.push(tok3->astOperand1());
2572                         tokens.push(tok3->astOperand2());
2573                         if (isSameExpression(_tokenizer->isCPP(), false, tok->astOperand1(), tok3, _settings->library.functionpure)) {
2574                             foundError = true;
2575                         }
2576                     }
2577     
2578                     if (foundError) {
2579                         unknownEvaluationOrder(parent);
2580                         break;
2581                     }
2582                 }
2583             }
2584         }
2585     }
