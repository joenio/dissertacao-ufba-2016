33     void Check64BitPortability::pointerassignment()
34     {
35         if (!_settings->isEnabled("portability"))
36             return;
37     
38         const SymbolDatabase *symbolDatabase = _tokenizer->getSymbolDatabase();
39     
40         // Check return values
41         const std::size_t functions = symbolDatabase->functionScopes.size();
42         for (std::size_t i = 0; i < functions; ++i) {
43             const Scope * scope = symbolDatabase->functionScopes[i];
44             if (scope->function == 0 || !scope->function->hasBody()) // We only look for functions with a body
45                 continue;
46     
47             bool retPointer = false;
48             if (scope->function->token->strAt(-1) == "*") // Function returns a pointer
49                 retPointer = true;
50             else if (Token::Match(scope->function->token->previous(), "int|long|DWORD")) // Function returns an integer
51                 ;
52             else
53                 continue;
54     
55             for (const Token* tok = scope->classStart->next(); tok != scope->classEnd; tok = tok->next()) {
56                 // skip nested functions
57                 if (tok->str() == "{") {
58                     if (tok->scope()->type == Scope::ScopeType::eFunction)
59                         tok = tok->link();
60                 }
61     
62                 if (tok->str() != "return")
63                     continue;
64     
65                 if (!tok->astOperand1() || tok->astOperand1()->isNumber())
66                     continue;
67     
68                 const ValueType * const returnType = tok->astOperand1()->valueType();
69                 if (!returnType)
70                     continue;
71     
72                 if (retPointer && returnType->pointer == 0U)
73                     returnIntegerError(tok);
74     
75                 if (!retPointer && returnType->pointer >= 1U)
76                     returnPointerError(tok);
77             }
78         }
79     
80         // Check assignments
81         for (std::size_t i = 0; i < functions; ++i) {
82             const Scope * scope = symbolDatabase->functionScopes[i];
83             for (const Token *tok = scope->classStart; tok && tok != scope->classEnd; tok = tok->next()) {
84                 if (tok->str() != "=")
85                     continue;
86     
87                 const ValueType *lhstype = tok->astOperand1() ? tok->astOperand1()->valueType() : nullptr;
88                 const ValueType *rhstype = tok->astOperand2() ? tok->astOperand2()->valueType() : nullptr;
89                 if (!lhstype || !rhstype)
90                     continue;
91     
92                 // Assign integer to pointer..
93                 if (lhstype->pointer >= 1U &&
94                     !tok->astOperand2()->isNumber() &&
95                     rhstype->pointer == 0U &&
96                     rhstype->originalTypeName.empty() &&
97                     rhstype->type == ValueType::Type::INT)
98                     assignmentIntegerToAddressError(tok);
99     
100                 // Assign pointer to integer..
101                 if (rhstype->pointer >= 1U &&
102                     lhstype->pointer == 0U &&
103                     lhstype->originalTypeName.empty() &&
104                     lhstype->type == ValueType::Type::INT)
105                     assignmentAddressToIntegerError(tok);
106             }
107         }
108     }

110     void Check64BitPortability::assignmentAddressToIntegerError(const Token *tok)
111     {
112         reportError(tok, Severity::portability,
113                     "AssignmentAddressToInteger",
114                     "Assigning a pointer to an integer is not portable.\n"
115                     "Assigning a pointer to an integer (int/long/etc) is not portable across different platforms and "
116                     "compilers. For example in 32-bit Windows and linux they are same width, but in 64-bit Windows and linux "
117                     "they are of different width. In worst case you end up assigning 64-bit address to 32-bit integer. The safe "
118                     "way is to store addresses only in pointer types (or typedefs like uintptr_t).");
119     }

125                     "Assigning an integer to a pointer is not portable.\n"
126                     "Assigning an integer (int/long/etc) to a pointer is not portable across different platforms and "

132     void Check64BitPortability::returnPointerError(const Token *tok)
133     {
134         reportError(tok, Severity::portability,
135                     "CastAddressToIntegerAtReturn",
136                     "Returning an address value in a function with integer return type is not portable.\n"
137                     "Returning an address value in a function with integer (int/long/etc) return type is not portable across "
138                     "different platforms and compilers. For example in 32-bit Windows and Linux they are same width, but in "
139                     "64-bit Windows and Linux they are of different width. In worst case you end up casting 64-bit address down "
140                     "to 32-bit integer. The safe way is to always return an integer.");
141     }
