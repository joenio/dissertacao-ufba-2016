32     static bool astIsBool(const Token *expr)
33     {
34         return Token::Match(expr, "%comp%|%bool%|%oror%|&&|!") && !expr->link();
35     }

39     void CheckBool::checkIncrementBoolean()
40     {
41         if (!_settings->isEnabled("style"))
42             return;
43     
44         const SymbolDatabase *symbolDatabase = _tokenizer->getSymbolDatabase();
45         const std::size_t functions = symbolDatabase->functionScopes.size();
46         for (std::size_t i = 0; i < functions; ++i) {
47             const Scope * scope = symbolDatabase->functionScopes[i];
48             for (const Token* tok = scope->classStart->next(); tok != scope->classEnd; tok = tok->next()) {
49                 if (Token::Match(tok, "%var% ++")) {
50                     const Variable *var = tok->variable();
51                     if (var && var->typeEndToken()->str() == "bool")
52                         incrementBooleanError(tok);
53                 }
54             }
55         }
56     }

70     // if (bool & bool) -> if (bool && bool)
71     // if (bool | bool) -> if (bool || bool)

73     void CheckBool::checkBitwiseOnBoolean()
74     {
75         if (!_settings->isEnabled("style"))
76             return;
77     
78         // danmar: this is inconclusive because I don't like that there are
79         //         warnings for calculations. Example: set_flag(a & b);
80         if (!_settings->inconclusive)
81             return;
82     
83         const SymbolDatabase *symbolDatabase = _tokenizer->getSymbolDatabase();
84         const std::size_t functions = symbolDatabase->functionScopes.size();
85         for (std::size_t i = 0; i < functions; ++i) {
86             const Scope * scope = symbolDatabase->functionScopes[i];
87             for (const Token* tok = scope->classStart->next(); tok != scope->classEnd; tok = tok->next()) {
88                 if (Token::Match(tok, "(|.|return|&&|%oror%|throw|, %var% [&|]")) {
89                     const Variable *var = tok->next()->variable();
90                     if (var && var->typeEndToken()->str() == "bool") {
91                         bitwiseOnBooleanError(tok->next(), var->name(), tok->strAt(2) == "&" ? "&&" : "||");
92                         tok = tok->tokAt(2);
93                     }
94                 } else if (Token::Match(tok, "[&|] %var% )|.|return|&&|%oror%|throw|,") && (!tok->previous() || !tok->previous()->isExtendedOp() || tok->strAt(-1) == ")" || tok->strAt(-1) == "]")) {
95                     const Variable *var = tok->next()->variable();
96                     if (var && var->typeEndToken()->str() == "bool") {
97                         bitwiseOnBooleanError(tok->next(), var->name(), tok->str() == "&" ? "&&" : "||");
98                         tok = tok->tokAt(2);
99                     }
100                 }
101             }
102         }
103     }

125     void CheckBool::checkComparisonOfBoolWithInt()
126     {
127         if (!_settings->isEnabled("warning") || !_tokenizer->isCPP())
128             return;
129     
130         const SymbolDatabase* const symbolDatabase = _tokenizer->getSymbolDatabase();
131         const std::size_t functions = symbolDatabase->functionScopes.size();
132         for (std::size_t i = 0; i < functions; ++i) {
133             const Scope * scope = symbolDatabase->functionScopes[i];
134             for (const Token* tok = scope->classStart->next(); tok != scope->classEnd; tok = tok->next()) {
135                 const Token* const left = tok->astOperand1();
136                 const Token* const right = tok->astOperand2();
137                 if (left && right && tok->isComparisonOp()) {
138                     if ((left->varId() && right->isNumber()) || (left->isNumber() && right->varId())) { // Comparing variable with number
139                         const Token* varTok = left;
140                         const Token* numTok = right;
141                         if (left->isNumber() && right->varId()) // num with var
142                             std::swap(varTok, numTok);
143                         if (isBool(varTok->variable()) && // Variable has to be a boolean
144                             ((tok->str() != "==" && tok->str() != "!=") ||
145                              (MathLib::toLongNumber(numTok->str()) != 0 && MathLib::toLongNumber(numTok->str()) != 1))) { // == 0 and != 0 are allowed, for C also == 1 and != 1
146                             comparisonOfBoolWithIntError(varTok, numTok->str(), tok->str() == "==" || tok->str() == "!=");
147                         }
148                     } else if (left->isBoolean() && right->varId()) { // Comparing boolean constant with variable
149                         if (isNonBoolStdType(right->variable())) { // Variable has to be of non-boolean standard type
150                             comparisonOfBoolWithIntError(right, left->str(), false);
151                         } else if (tok->str() != "==" && tok->str() != "!=") {
152                             comparisonOfBoolWithInvalidComparator(right, left->str());
153                         }
154                     } else if (left->varId() && right->isBoolean()) { // Comparing variable with boolean constant
155                         if (isNonBoolStdType(left->variable())) { // Variable has to be of non-boolean standard type
156                             comparisonOfBoolWithIntError(left, right->str(), false);
157                         } else if (tok->str() != "==" && tok->str() != "!=") {
158                             comparisonOfBoolWithInvalidComparator(right, left->str());
159                         }
160                     } else if (left->isNumber() && right->isBoolean()) { // number constant with boolean constant
161                         comparisonOfBoolWithIntError(left, right->str(), false);
162                     } else if (left->isBoolean() && right->isNumber()) { // number constant with boolean constant
163                         comparisonOfBoolWithIntError(left, left->str(), false);
164                     } else if (left->varId() && right->varId()) { // Comparing two variables, one of them boolean, one of them integer
165                         const Variable* var1 = right->variable();
166                         const Variable* var2 = left->variable();
167                         if (isBool(var1) && isNonBoolStdType(var2)) // Comparing boolean with non-bool standard type
168                             comparisonOfBoolWithIntError(left, var1->name(), false);
169                         else if (isNonBoolStdType(var1) && isBool(var2)) // Comparing non-bool standard type with boolean
170                             comparisonOfBoolWithIntError(left, var2->name(), false);
171                     }
172                 }
173             }
174         }
175     }

205     static bool tokenIsFunctionReturningBool(const Token* tok)
206     {
207         const Function* func = tok->function();
208         if (func && Token::Match(tok, "%name% (")) {
209             if (func->tokenDef && func->tokenDef->strAt(-1) == "bool") {
210                 return true;
211             }
212         }
213         return false;
214     }

216     void CheckBool::checkComparisonOfFuncReturningBool()
217     {
218         if (!_settings->isEnabled("style"))
219             return;
220     
221         if (!_tokenizer->isCPP())
222             return;
223     
224         const SymbolDatabase * const symbolDatabase = _tokenizer->getSymbolDatabase();
225     
226         const std::size_t functionsCount = symbolDatabase->functionScopes.size();
227         for (std::size_t i = 0; i < functionsCount; ++i) {
228             const Scope * scope = symbolDatabase->functionScopes[i];
229             for (const Token* tok = scope->classStart->next(); tok != scope->classEnd; tok = tok->next()) {
230                 if (tok->tokType() != Token::eComparisonOp || tok->str() == "==" || tok->str() == "!=")
231                     continue;
232                 const Token *firstToken = tok->previous();
233                 if (tok->strAt(-1) == ")") {
234                     firstToken = firstToken->link()->previous();
235                 }
236                 const Token *secondToken = tok->next();
237                 while (secondToken->str() == "!") {
238                     secondToken = secondToken->next();
239                 }
240                 const bool firstIsFunctionReturningBool = tokenIsFunctionReturningBool(firstToken);
241                 const bool secondIsFunctionReturningBool = tokenIsFunctionReturningBool(secondToken);
242                 if (firstIsFunctionReturningBool && secondIsFunctionReturningBool) {
243                     comparisonOfTwoFuncsReturningBoolError(firstToken->next(), firstToken->str(), secondToken->str());
244                 } else if (firstIsFunctionReturningBool) {
245                     comparisonOfFuncReturningBoolError(firstToken->next(), firstToken->str());
246                 } else if (secondIsFunctionReturningBool) {
247                     comparisonOfFuncReturningBoolError(secondToken->previous(), secondToken->str());
248                 }
249             }
250         }
251     }

275     void CheckBool::checkComparisonOfBoolWithBool()
276     {
277         // FIXME: This checking is "experimental" because of the false positives
278         //        when self checking lib/tokenize.cpp (#2617)
279         if (!_settings->experimental)
280             return;
281     
282         if (!_settings->isEnabled("style"))
283             return;
284     
285         if (!_tokenizer->isCPP())
286             return;
287     
288         const SymbolDatabase* const symbolDatabase = _tokenizer->getSymbolDatabase();
289     
290         const std::size_t functions = symbolDatabase->functionScopes.size();
291         for (std::size_t i = 0; i < functions; ++i) {
292             const Scope * scope = symbolDatabase->functionScopes[i];
293             for (const Token* tok = scope->classStart->next(); tok != scope->classEnd; tok = tok->next()) {
294                 if (tok->tokType() != Token::eComparisonOp || tok->str() == "==" || tok->str() == "!=")
295                     continue;
296                 bool firstTokenBool = false;
297     
298                 const Token *firstToken = tok->previous();
299                 if (firstToken->varId()) {
300                     if (isBool(firstToken->variable())) {
301                         firstTokenBool = true;
302                     }
303                 }
304                 if (!firstTokenBool)
305                     continue;
306     
307                 bool secondTokenBool = false;
308                 const Token *secondToken = tok->next();
309                 if (secondToken->varId()) {
310                     if (isBool(secondToken->variable())) {
311                         secondTokenBool = true;
312                     }
313                 }
314                 if (secondTokenBool) {
315                     comparisonOfBoolWithBoolError(firstToken->next(), secondToken->str());
316                 }
317             }
318         }
319     }

332     void CheckBool::checkAssignBoolToPointer()
333     {
334         const SymbolDatabase *symbolDatabase = _tokenizer->getSymbolDatabase();
335         const std::size_t functions = symbolDatabase->functionScopes.size();
336         for (std::size_t i = 0; i < functions; ++i) {
337             const Scope * scope = symbolDatabase->functionScopes[i];
338             for (const Token* tok = scope->classStart; tok != scope->classEnd; tok = tok->next()) {
339                 if (tok->str() == "=" && astIsBool(tok->astOperand2())) {
340                     const Token *lhs = tok->astOperand1();
341                     while (lhs && (lhs->str() == "." || lhs->str() == "::"))
342                         lhs = lhs->astOperand2();
343                     if (!lhs || !lhs->variable() || !lhs->variable()->isPointer())
344                         continue;
345     
346                     assignBoolToPointerError(tok);
347                 }
348             }
349         }
350     }

358     //-----------------------------------------------------------------------------
359     //-----------------------------------------------------------------------------
360     void CheckBool::checkComparisonOfBoolExpressionWithInt()
361     {
362         if (!_settings->isEnabled("warning"))
363             return;
364     
365         const SymbolDatabase* symbolDatabase = _tokenizer->getSymbolDatabase();
366     
367         const std::size_t functions = symbolDatabase->functionScopes.size();
368         for (std::size_t i = 0; i < functions; ++i) {
369             const Scope * scope = symbolDatabase->functionScopes[i];
370             for (const Token* tok = scope->classStart->next(); tok != scope->classEnd; tok = tok->next()) {
371                 if (!tok->isComparisonOp())
372                     continue;
373     
374                 const Token* numTok = 0;
375                 const Token* boolExpr = 0;
376                 bool numInRhs;
377                 if (astIsBool(tok->astOperand1())) {
378                     boolExpr = tok->astOperand1();
379                     numTok = tok->astOperand2();
380                     numInRhs = true;
381                 } else if (astIsBool(tok->astOperand2())) {
382                     boolExpr = tok->astOperand2();
383                     numTok = tok->astOperand1();
384                     numInRhs = false;
385                 } else {
386                     continue;
387                 }
388     
389                 if (!numTok || !boolExpr)
390                     continue;
391     
392                 if (Token::Match(boolExpr,"%bool%"))
393                     // The CheckBool::checkComparisonOfBoolWithInt warns about this.
394                     continue;
395     
396                 if (boolExpr->isOp() && numTok->isName() && Token::Match(tok, "==|!="))
397                     // there is weird code such as:  ((a<b)==c)
398                     // but it is probably written this way by design.
399                     continue;
400     
401                 if (numTok->isNumber()) {
402                     if (numTok->str() == "0" &&
403                         (numInRhs ? Token::Match(tok, ">|==|!=")
404                          : Token::Match(tok, "<|==|!=")))
405                         continue;
406                     if (numTok->str() == "1" &&
407                         (numInRhs ? Token::Match(tok, "<|==|!=")
408                          : Token::Match(tok, ">|==|!=")))
409                         continue;
410                     comparisonOfBoolExpressionWithIntError(tok, true);
411                 } else if (isNonBoolStdType(numTok->variable()))
412                     comparisonOfBoolExpressionWithIntError(tok, false);
413             }
414         }
415     }

428     void CheckBool::pointerArithBool()
429     {
430         const SymbolDatabase* symbolDatabase = _tokenizer->getSymbolDatabase();
431     
432         for (std::list<Scope>::const_iterator scope = symbolDatabase->scopeList.begin(); scope != symbolDatabase->scopeList.end(); ++scope) {
433             if (scope->type != Scope::eIf && scope->type != Scope::eWhile && scope->type != Scope::eDo && scope->type != Scope::eFor)
434                 continue;
435             const Token* tok = scope->classDef->next()->astOperand2();
436             if (scope->type == Scope::eFor) {
437                 tok = Token::findsimplematch(scope->classDef->tokAt(2), ";");
438                 if (tok)
439                     tok = tok->astOperand2();
440                 if (tok)
441                     tok = tok->astOperand1();
442             } else if (scope->type == Scope::eDo)
443                 tok = (scope->classEnd->tokAt(2)) ? scope->classEnd->tokAt(2)->astOperand2() : nullptr;
444     
445             pointerArithBoolCond(tok);
446         }
447     }
448     
449     void CheckBool::pointerArithBoolCond(const Token *tok)
450     {
451         if (!tok)
452             return;
453         if (Token::Match(tok, "&&|%oror%")) {
454             pointerArithBoolCond(tok->astOperand1());
455             pointerArithBoolCond(tok->astOperand2());
456             return;
457         }
458         if (tok->str() != "+" && tok->str() != "-")
459             return;
460     
461         if (tok->astOperand1() &&
462             tok->astOperand2() &&
463             tok->astOperand1()->isName() &&
464             tok->astOperand1()->variable() &&
465             tok->astOperand1()->variable()->isPointer() &&
466             tok->astOperand2()->isNumber())
467             pointerArithBoolError(tok);
468     }

479     void CheckBool::checkAssignBoolToFloat()
480     {
481         if (!_tokenizer->isCPP())
482             return;
483         if (!_settings->isEnabled("style"))
484             return;
485         const SymbolDatabase *symbolDatabase = _tokenizer->getSymbolDatabase();
486         const std::size_t functions = symbolDatabase->functionScopes.size();
487         for (std::size_t i = 0; i < functions; ++i) {
488             const Scope * scope = symbolDatabase->functionScopes[i];
489             for (const Token* tok = scope->classStart; tok != scope->classEnd; tok = tok->next()) {
490                 if (Token::Match(tok, "%var% =")) {
491                     const Variable * const var = tok->variable();
492                     if (var && var->isFloatingType() && !var->isArrayOrPointer() && astIsBool(tok->next()->astOperand2()))
493                         assignBoolToFloatError(tok->next());
494                 }
495             }
496         }
497     }
