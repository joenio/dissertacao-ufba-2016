36     /**
37      * @brief Interface class that cppcheck uses to communicate with the checks.
38      * All checking classes must inherit from this class
39      */
40     class CPPCHECKLIB Check {
41     public:
42         /** This constructor is used when registering the CheckClass */
43         explicit Check(const std::string &aname);
44     
45         /** This constructor is used when running checks. */
46         Check(const std::string &aname, const Tokenizer *tokenizer, const Settings *settings, ErrorLogger *errorLogger)
47             : _tokenizer(tokenizer), _settings(settings), _errorLogger(errorLogger), _name(aname) {
48         }
49     
50         virtual ~Check() {
51             if (!_tokenizer)
52                 instances().remove(this);
53         }
54     
55         /** List of registered check classes. This is used by Cppcheck to run checks and generate documentation */
56         static std::list<Check *> &instances();
57     
58         /** run checks, the token list is not simplified */
59         virtual void runChecks(const Tokenizer *, const Settings *, ErrorLogger *) {
60         }
61     
62         /** run checks, the token list is simplified */
63         virtual void runSimplifiedChecks(const Tokenizer *tokenizer, const Settings *settings, ErrorLogger *errorLogger) = 0;
64     
65         /** get error messages */
66         virtual void getErrorMessages(ErrorLogger *errorLogger, const Settings *settings) const = 0;
67     
68         /** class name, used to generate documentation */
69         const std::string& name() const {
70             return _name;
71         }
72     
73         /** get information about this class, used to generate documentation */
74         virtual std::string classInfo() const = 0;
75     
76         /**
77          * Write given error to errorlogger or to out stream in xml format.
78          * This is for for printout out the error list with --errorlist
79          * @param errmsg Error message to write
80          */
81         static void reportError(const ErrorLogger::ErrorMessage &errmsg);
82     
83         bool inconclusiveFlag() const {
84             return _settings && _settings->inconclusive;
85         }
86     
87         /** Base class used for whole-program analysis */
88         class FileInfo {
89         public:
90             FileInfo() {}
91             virtual ~FileInfo() {}
92         };
93     
94         virtual FileInfo * getFileInfo(const Tokenizer *tokenizer, const Settings *settings) const {
95             (void)tokenizer;
96             (void)settings;
97             return nullptr;
98         }
99     
100         virtual void analyseWholeProgram(const std::list<FileInfo*> &fileInfo, const Settings& settings, ErrorLogger &errorLogger) {
101             (void)fileInfo;
102             (void)settings;
103             (void)errorLogger;
104         }
105     
106     protected:
107         const Tokenizer * const _tokenizer;
108         const Settings * const _settings;
109         ErrorLogger * const _errorLogger;
110     
111         /** report an error */
112         template<typename T, typename U>
113         void reportError(const Token *tok, const Severity::SeverityType severity, const T id, const U msg) {
114             reportError(tok, severity, id, msg, 0U, false);
115         }
116     
117         /** report an error */
118         template<typename T, typename U>
119         void reportError(const Token *tok, const Severity::SeverityType severity, const T id, const U msg, unsigned int cwe, bool inconclusive) {
120             std::list<const Token *> callstack(1, tok);
121             reportError(callstack, severity, id, msg, cwe, inconclusive);
122         }
123     
124         /** report an error */
125         template<typename T, typename U>
126         void reportError(const std::list<const Token *> &callstack, Severity::SeverityType severity, const T id, const U msg) {
127             reportError(callstack, severity, id, msg, 0U, false);
128         }
129     
130         /** report an error */
131         template<typename T, typename U>
132         void reportError(const std::list<const Token *> &callstack, Severity::SeverityType severity, const T id, const U msg, unsigned int cwe, bool inconclusive) {
133             ErrorLogger::ErrorMessage errmsg(callstack, _tokenizer?&_tokenizer->list:0, severity, id, msg, inconclusive);
134             errmsg._cwe = cwe;
135             if (_errorLogger)
136                 _errorLogger->reportErr(errmsg);
137             else
138                 reportError(errmsg);
139         }
140     
141     private:
142         const std::string _name;
143     
144         /** disabled assignment operator and copy constructor */
145         void operator=(const Check &);
146         explicit Check(const Check &);
147     };
148
