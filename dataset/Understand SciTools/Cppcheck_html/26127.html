40     void VarInfo::print()
41     {
42         std::cout << "size=" << alloctype.size() << std::endl;
43         std::map<unsigned int, AllocInfo>::const_iterator it;
44         for (it = alloctype.begin(); it != alloctype.end(); ++it) {
45             std::string strusage;
46             std::map<unsigned int, std::string>::const_iterator use = possibleUsage.find(it->first);
47             if (use != possibleUsage.end())
48                 strusage = use->second;
49     
50             std::cout << "alloctype='" << it->second.type << "' "
51                       << "possibleUsage='" << strusage << "'" << std::endl;
52         }
53     }

55     void VarInfo::possibleUsageAll(const std::string &functionName)
56     {
57         possibleUsage.clear();
58         std::map<unsigned int, AllocInfo>::const_iterator it;

110     void CheckLeakAutoVar::check()
111     {
112         const SymbolDatabase *symbolDatabase = _tokenizer->getSymbolDatabase();
113     
114         // Local variables that are known to be non-zero.
115         const std::set<unsigned int> notzero;
116     
117         // Check function scopes
118         const std::size_t functions = symbolDatabase->functionScopes.size();
119         for (std::size_t i = 0; i < functions; ++i) {
120             const Scope * scope = symbolDatabase->functionScopes[i];
121             // Empty variable info
122             VarInfo varInfo;
123     
124             checkScope(scope->classStart, &varInfo, notzero);
125     
126             varInfo.conditionalAlloc.clear();
127     
128             // Clear reference arguments from varInfo..
129             std::map<unsigned int, VarInfo::AllocInfo>::iterator it = varInfo.alloctype.begin();
130             while (it != varInfo.alloctype.end()) {
131                 const Variable *var = symbolDatabase->getVariableFromVarId(it->first);
132                 if (!var ||
133                     (var->isArgument() && var->isReference()) ||
134                     (!var->isArgument() && !var->isLocal()))
135                     varInfo.alloctype.erase(it++);
136                 else
137                     ++it;
138             }
139     
140             ret(scope->classEnd, varInfo);
141         }
142     }

144     void CheckLeakAutoVar::checkScope(const Token * const startToken,
145                                       VarInfo *varInfo,
146                                       std::set<unsigned int> notzero)
147     {
148         std::map<unsigned int, VarInfo::AllocInfo> &alloctype = varInfo->alloctype;
149         std::map<unsigned int, std::string> &possibleUsage = varInfo->possibleUsage;
150         const std::set<unsigned int> conditionalAlloc(varInfo->conditionalAlloc);
151     
152         // Parse all tokens
153         const Token * const endToken = startToken->link();
154         for (const Token *tok = startToken; tok && tok != endToken; tok = tok->next()) {
155             if (!tok->scope()->isExecutable()) {
156                 tok = tok->scope()->classEnd;
157                 if (!tok) // Ticket #6666 (crash upon invalid code)
158                     break;
159             }
160     
161             // Deallocation and then dereferencing pointer..
162             if (tok->varId() > 0) {
163                 const std::map<unsigned int, VarInfo::AllocInfo>::const_iterator var = alloctype.find(tok->varId());
164                 if (var != alloctype.end()) {
165                     if (var->second.status == VarInfo::DEALLOC && tok->strAt(-1) != "&" && (!Token::Match(tok, "%name% =") || tok->strAt(-1) == "*")) {
166                         deallocUseError(tok, tok->str());
167                     } else if (Token::simpleMatch(tok->tokAt(-2), "= &")) {
168                         varInfo->erase(tok->varId());
169                     } else if (tok->strAt(-1) == "=") {
170                         varInfo->erase(tok->varId());
171                     }
172                 } else if (Token::Match(tok->previous(), "& %name% = %var% ;")) {
173                     varInfo->referenced.insert(tok->tokAt(2)->varId());
174                 }
175             }
176     
177             if (tok->str() == "(" && tok->previous()->isName()) {
178                 VarInfo::AllocInfo allocation(0, VarInfo::NOALLOC);
179                 functionCall(tok->previous(), varInfo, allocation);
180                 tok = tok->link();
181                 continue;
182             }
183     
184             // look for end of statement
185             if (!Token::Match(tok, "[;{}]") || Token::Match(tok->next(), "[;{}]"))
186                 continue;
187     
188             tok = tok->next();
189             if (!tok || tok == endToken)
190                 break;
191     
192             // parse statement, skip to last member
193             while (Token::Match(tok, "%name% ::|. %name% !!("))
194                 tok = tok->tokAt(2);
195     
196             // assignment..
197             if (Token::Match(tok, "%var% =")) {
198                 // taking address of another variable..
199                 if (Token::Match(tok->next(), "= %var% [+;]")) {
200                     if (tok->tokAt(2)->varId() != tok->varId()) {
201                         // If variable points at allocated memory => error
202                         leakIfAllocated(tok, *varInfo);
203     
204                         // no multivariable checking currently => bail out for rhs variables
205                         for (const Token *tok2 = tok; tok2; tok2 = tok2->next()) {
206                             if (tok2->str() == ";") {
207                                 break;
208                             }
209                             if (tok2->varId()) {
210                                 varInfo->erase(tok2->varId());
211                             }
212                         }
213                     }
214                 }
215     
216                 // is variable used in rhs?
217                 bool used_in_rhs = false;
218                 for (const Token *tok2 = tok->tokAt(2); tok2; tok2 = tok2->next()) {
219                     if (tok2->str() == ";") {
220                         break;
221                     }
222                     if (tok->varId() == tok2->varId()) {
223                         used_in_rhs = true;
224                         break;
225                     }
226                 }
227                 // TODO: Better checking how the pointer is used in rhs?
228                 if (used_in_rhs)
229                     continue;
230     
231                 // Variable has already been allocated => error
232                 if (conditionalAlloc.find(tok->varId()) == conditionalAlloc.end())
233                     leakIfAllocated(tok, *varInfo);
234                 varInfo->erase(tok->varId());
235     
236                 // not a local variable nor argument?
237                 const Variable *var = tok->variable();
238                 if (var && !var->isArgument() && (!var->isLocal() || var->isStatic()))
239                     continue;
240     
241                 // Don't check reference variables
242                 if (var && var->isReference())
243                     continue;
244     
245                 // non-pod variable
246                 if (_tokenizer->isCPP()) {
247                     if (!var)
248                         continue;
249                     // Possibly automatically deallocated memory
250                     if (!var->typeStartToken()->isStandardType() && Token::Match(tok, "%var% = new"))
251                         continue;
252                 }
253     
254                 // allocation?
255                 if (tok->next()->astOperand2() && Token::Match(tok->next()->astOperand2()->previous(), "%type% (")) {
256                     int i = _settings->library.alloc(tok->next()->astOperand2()->previous());
257                     if (i > 0) {
258                         alloctype[tok->varId()].type = i;
259                         alloctype[tok->varId()].status = VarInfo::ALLOC;
260                     }
261                 } else if (_tokenizer->isCPP() && tok->strAt(2) == "new") {
262                     alloctype[tok->varId()].type = -1;
263                     alloctype[tok->varId()].status = VarInfo::ALLOC;
264                 }
265     
266                 // Assigning non-zero value variable. It might be used to
267                 // track the execution for a later if condition.
268                 if (Token::Match(tok->tokAt(2), "%num% ;") && MathLib::toLongNumber(tok->strAt(2)) != 0)
269                     notzero.insert(tok->varId());
270                 else if (Token::Match(tok->tokAt(2), "- %type% ;") && tok->tokAt(3)->isUpperCaseName())
271                     notzero.insert(tok->varId());
272                 else
273                     notzero.erase(tok->varId());
274             }
275     
276             // if/else
277             else if (Token::simpleMatch(tok, "if (")) {
278                 // Parse function calls inside the condition
279                 for (const Token *innerTok = tok->tokAt(2); innerTok; innerTok = innerTok->next()) {
280                     if (innerTok->str() == ")")
281                         break;
282                     if (innerTok->str() == "(" && innerTok->previous()->isName()) {
283                         VarInfo::AllocInfo allocation(_settings->library.dealloc(tok), VarInfo::DEALLOC);
284                         if (allocation.type == 0)
285                             allocation.status = VarInfo::NOALLOC;
286                         functionCall(innerTok->previous(), varInfo, allocation);
287                         innerTok = innerTok->link();
288                     }
289                 }
290     
291                 const Token *tok2 = tok->linkAt(1);
292                 if (Token::simpleMatch(tok2, ") {")) {
293                     VarInfo varInfo1(*varInfo);  // VarInfo for if code
294                     VarInfo varInfo2(*varInfo);  // VarInfo for else code
295     
296                     // Recursively scan variable comparisons in condition
297                     std::stack<const Token *> tokens;
298                     tokens.push(tok->next()->astOperand2());
299                     while (!tokens.empty()) {
300                         const Token *tok3 = tokens.top();
301                         tokens.pop();
302                         if (!tok3)
303                             continue;
304                         if (tok3->str() == "&&") {
305                             tokens.push(tok3->astOperand1());
306                             tokens.push(tok3->astOperand2());
307                             continue;
308                         }
309                         if (tok3->str() == "(" && Token::Match(tok3->astOperand1(), "UNLIKELY|LIKELY")) {
310                             tokens.push(tok3->astOperand2());
311                             continue;
312                         }
313     
314                         const Token *vartok = nullptr;
315                         if (astIsVariableComparison(tok3, "!=", "0", &vartok)) {
316                             varInfo2.erase(vartok->varId());
317                             if (notzero.find(vartok->varId()) != notzero.end())
318                                 varInfo2.clear();
319                         } else if (astIsVariableComparison(tok3, "==", "0", &vartok)) {
320                             varInfo1.erase(vartok->varId());
321                         } else if (astIsVariableComparison(tok3, "<", "0", &vartok)) {
322                             varInfo1.erase(vartok->varId());
323                         } else if (astIsVariableComparison(tok3, ">", "0", &vartok)) {
324                             varInfo2.erase(vartok->varId());
325                         } else if (astIsVariableComparison(tok3, "==", "-1", &vartok)) {
326                             varInfo1.erase(vartok->varId());
327                         }
328                     }
329     
330                     checkScope(tok2->next(), &varInfo1, notzero);
331                     tok2 = tok2->linkAt(1);
332                     if (Token::simpleMatch(tok2, "} else {")) {
333                         checkScope(tok2->tokAt(2), &varInfo2, notzero);
334                         tok = tok2->linkAt(2)->previous();
335                     } else {
336                         tok = tok2->previous();
337                     }
338     
339                     VarInfo old;
340                     old.swap(*varInfo);
341     
342                     // Conditional allocation in varInfo1
343                     std::map<unsigned int, VarInfo::AllocInfo>::const_iterator it;
344                     for (it = varInfo1.alloctype.begin(); it != varInfo1.alloctype.end(); ++it) {
345                         if (varInfo2.alloctype.find(it->first) == varInfo2.alloctype.end() &&
346                             old.alloctype.find(it->first) == old.alloctype.end()) {
347                             varInfo->conditionalAlloc.insert(it->first);
348                         }
349                     }
350     
351                     // Conditional allocation in varInfo2
352                     for (it = varInfo2.alloctype.begin(); it != varInfo2.alloctype.end(); ++it) {
353                         if (varInfo1.alloctype.find(it->first) == varInfo1.alloctype.end() &&
354                             old.alloctype.find(it->first) == old.alloctype.end()) {
355                             varInfo->conditionalAlloc.insert(it->first);
356                         }
357                     }
358     
359                     // Conditional allocation/deallocation
360                     for (it = varInfo1.alloctype.begin(); it != varInfo1.alloctype.end(); ++it) {
361                         if (it->second.status == VarInfo::DEALLOC && conditionalAlloc.find(it->first) != conditionalAlloc.end()) {
362                             varInfo->conditionalAlloc.erase(it->first);
363                             varInfo2.erase(it->first);
364                         }
365                     }
366                     for (it = varInfo2.alloctype.begin(); it != varInfo2.alloctype.end(); ++it) {
367                         if (it->second.status == VarInfo::DEALLOC && conditionalAlloc.find(it->first) != conditionalAlloc.end()) {
368                             varInfo->conditionalAlloc.erase(it->first);
369                             varInfo1.erase(it->first);
370                         }
371                     }
372     
373                     alloctype.insert(varInfo1.alloctype.begin(), varInfo1.alloctype.end());
374                     alloctype.insert(varInfo2.alloctype.begin(), varInfo2.alloctype.end());
375     
376                     possibleUsage.insert(varInfo1.possibleUsage.begin(), varInfo1.possibleUsage.end());
377                     possibleUsage.insert(varInfo2.possibleUsage.begin(), varInfo2.possibleUsage.end());
378                 }
379             }
380     
381             // unknown control.. (TODO: handle loops)
382             else if ((Token::Match(tok, "%type% (") && Token::simpleMatch(tok->linkAt(1), ") {")) || Token::simpleMatch(tok, "do {")) {
383                 varInfo->clear();
384                 break;
385             }
386     
387             // return
388             else if (tok->str() == "return") {
389                 ret(tok, *varInfo);
390                 varInfo->clear();
391             }
392     
393             // throw
394             else if (_tokenizer->isCPP() && tok->str() == "throw") {
395                 bool tryFound = false;
396                 const Scope* scope = tok->scope();
397                 while (scope && scope->isExecutable()) {
398                     if (scope->type == Scope::eTry)
399                         tryFound = true;
400                     scope = scope->nestedIn;
401                 }
402                 // If the execution leaves the function then treat it as return
403                 if (!tryFound)
404                     ret(tok, *varInfo);
405                 varInfo->clear();
406             }
407     
408             // Function call..
409             else if (Token::Match(tok, "%type% (")) {
410                 VarInfo::AllocInfo allocation(_settings->library.dealloc(tok), VarInfo::DEALLOC);
411                 if (allocation.type == 0)
412                     allocation.status = VarInfo::NOALLOC;
413                 functionCall(tok, varInfo, allocation);
414     
415                 tok = tok->next()->link();
416     
417                 // Handle scopes that might be noreturn
418                 if (allocation.status == VarInfo::NOALLOC && Token::simpleMatch(tok, ") ; }")) {
419                     const std::string &functionName(tok->link()->previous()->str());
420                     bool unknown = false;
421                     if (_tokenizer->IsScopeNoReturn(tok->tokAt(2), &unknown)) {
422                         if (!unknown)
423                             varInfo->clear();
424                         else if (_settings->library.leakignore.find(functionName) == _settings->library.leakignore.end() &&
425                                  _settings->library.use.find(functionName) == _settings->library.use.end())
426                             varInfo->possibleUsageAll(functionName);
427                     }
428                 }
429     
430                 continue;
431             }
432     
433             // delete
434             else if (_tokenizer->isCPP() && tok->str() == "delete") {
435                 if (tok->strAt(1) == "[")
436                     tok = tok->tokAt(3);
437                 else
438                     tok = tok->next();
439                 while (Token::Match(tok, "%name% ::|."))
440                     tok = tok->tokAt(2);
441                 if (tok->varId() && tok->strAt(1) != "[") {
442                     VarInfo::AllocInfo allocation(-1, VarInfo::DEALLOC);
443                     changeAllocStatus(varInfo, allocation, tok, tok);
444                 }
445             }
446     
447             // goto => weird execution path
448             else if (tok->str() == "goto") {
449                 varInfo->clear();
450             }
451     
452             // continue/break
453             else if (Token::Match(tok, "continue|break ;")) {
454                 varInfo->clear();
455             }
456         }
457     }

459     void CheckLeakAutoVar::changeAllocStatus(VarInfo *varInfo, const VarInfo::AllocInfo& allocation, const Token* tok, const Token* arg)
460     {
461         std::map<unsigned int, VarInfo::AllocInfo> &alloctype = varInfo->alloctype;
462         std::map<unsigned int, std::string> &possibleUsage = varInfo->possibleUsage;
463         const std::map<unsigned int, VarInfo::AllocInfo>::iterator var = alloctype.find(arg->varId());
464         if (var != alloctype.end()) {
465             if (allocation.status == VarInfo::NOALLOC) {
466                 // possible usage
467                 possibleUsage[arg->varId()] = tok->str();
468                 if (var->second.status == VarInfo::DEALLOC && arg->previous()->str() == "&")
469                     varInfo->erase(arg->varId());
470             } else if (var->second.status == VarInfo::DEALLOC) {
471                 doubleFreeError(tok, arg->str(), allocation.type);
472             } else if (var->second.type != allocation.type) {
473                 // mismatching allocation and deallocation
474                 mismatchError(tok, arg->str());
475                 varInfo->erase(arg->varId());
476             } else {
477                 // deallocation
478                 var->second.status = VarInfo::DEALLOC;
479                 var->second.type = allocation.type;
480             }
481         } else if (allocation.status != VarInfo::NOALLOC) {
482             alloctype[arg->varId()].status = VarInfo::DEALLOC;
483         }
484     }

486     void CheckLeakAutoVar::functionCall(const Token *tok, VarInfo *varInfo, const VarInfo::AllocInfo& allocation)
487     {
488         // Ignore function call?
489         const bool ignore = bool(_settings->library.leakignore.find(tok->str()) != _settings->library.leakignore.end());
490         if (ignore)
491             return;
492     
493         for (const Token *arg = tok->tokAt(2); arg; arg = arg->nextArgument()) {
494             if (_tokenizer->isCPP() && arg->str() == "new")
495                 arg = arg->next();
496     
497             if (Token::Match(arg, "%var% [-,)]") || Token::Match(arg, "& %var%")) {
498     
499                 // goto variable
500                 if (arg->str() == "&")
501                     arg = arg->next();
502     
503                 // Is variable allocated?
504                 changeAllocStatus(varInfo, allocation, tok, arg);
505             } else if (Token::Match(arg, "%name% (")) {
506                 functionCall(arg, varInfo, allocation);
507             }
508         }
509     }

512     void CheckLeakAutoVar::leakIfAllocated(const Token *vartok,
513                                            const VarInfo &varInfo)
514     {
515         const std::map<unsigned int, VarInfo::AllocInfo> &alloctype = varInfo.alloctype;
516         const std::map<unsigned int, std::string> &possibleUsage = varInfo.possibleUsage;
517     
518         const std::map<unsigned int, VarInfo::AllocInfo>::const_iterator var = alloctype.find(vartok->varId());
519         if (var != alloctype.end() && var->second.status != VarInfo::DEALLOC) {
520             const std::map<unsigned int, std::string>::const_iterator use = possibleUsage.find(vartok->varId());
521             if (use == possibleUsage.end()) {
522                 leakError(vartok, vartok->str(), var->second.type);
523             } else {
524                 configurationInfo(vartok, use->second);
525             }
526         }
527     }

529     void CheckLeakAutoVar::ret(const Token *tok, const VarInfo &varInfo)
530     {
531         const std::map<unsigned int, VarInfo::AllocInfo> &alloctype = varInfo.alloctype;
532         const std::map<unsigned int, std::string> &possibleUsage = varInfo.possibleUsage;
533     
534         const SymbolDatabase *symbolDatabase = _tokenizer->getSymbolDatabase();
535         for (std::map<unsigned int, VarInfo::AllocInfo>::const_iterator it = alloctype.begin(); it != alloctype.end(); ++it) {
536             // don't warn if variable is conditionally allocated
537             if (it->second.status != VarInfo::DEALLOC && varInfo.conditionalAlloc.find(it->first) != varInfo.conditionalAlloc.end())
538                 continue;
539     
540             // don't warn if there is a reference of the variable
541             if (varInfo.referenced.find(it->first) != varInfo.referenced.end())
542                 continue;
543     
544             const unsigned int varid = it->first;
545             const Variable *var = symbolDatabase->getVariableFromVarId(varid);
546             if (var) {
547                 bool used = false;
548                 for (const Token *tok2 = tok; tok2; tok2 = tok2->next()) {
549                     if (tok2->str() == ";")
550                         break;
551                     if (Token::Match(tok2, "return|(|, %varid% [);,]", varid)) {
552                         used = true;
553                         break;
554                     }
555                     if (Token::Match(tok2, "return|(|, & %varid% . %name% [);,]", varid)) {
556                         used = true;
557                         break;
558                     }
559                 }
560     
561                 // return deallocated pointer
562                 if (used && it->second.status == VarInfo::DEALLOC)
563                     deallocReturnError(tok, var->name());
564     
565                 else if (!used && it->second.status != VarInfo::DEALLOC) {
566                     const std::map<unsigned int, std::string>::const_iterator use = possibleUsage.find(varid);
567                     if (use == possibleUsage.end()) {
568                         leakError(tok, var->name(), it->second.type);
569                     } else {
570                         configurationInfo(tok, use->second);
571                     }
572                 }
573             }
574         }
575     }
