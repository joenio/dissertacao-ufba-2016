42     static void AddFilesToList(const std::string& FileList, std::vector<std::string>& PathNames)
43     {
44         // To keep things initially simple, if the file can't be opened, just be silent and move on.
45         std::istream *Files;
46         std::ifstream Infile;
47         if (FileList.compare("-") == 0) { // read from stdin
48             Files = &std::cin;
49         } else {
50             Infile.open(FileList.c_str());
51             Files = &Infile;
52         }
53         if (Files && *Files) {
54             std::string FileName;
55             while (std::getline(*Files, FileName)) { // next line
56                 if (!FileName.empty()) {
57                     PathNames.push_back(FileName);
58                 }
59             }
60         }
61     }

63     static void AddInclPathsToList(const std::string& FileList, std::list<std::string>* PathNames)
64     {
65         // To keep things initially simple, if the file can't be opened, just be silent and move on.
66         std::ifstream Files(FileList.c_str());
67         if (Files) {
68             std::string PathName;
69             while (std::getline(Files, PathName)) { // next line
70                 if (!PathName.empty()) {
71                     PathName = Path::removeQuotationMarks(PathName);
72                     PathName = Path::fromNativeSeparators(PathName);
73     
74                     // If path doesn't end with / or \, add it
75                     if (PathName.back() != '/')
76                         PathName += '/';
77     
78                     PathNames->push_back(PathName);
79                 }
80             }
81         }
82     }

84     static void AddPathsToSet(const std::string& FileName, std::set<std::string>* set)
85     {
86         std::list<std::string> templist;
87         AddInclPathsToList(FileName, &templist);
88         set->insert(templist.begin(), templist.end());
89     }

110     bool CmdLineParser::ParseFromArgs(int argc, const char* const argv[])
111     {
112         bool def = false;
113         bool maxconfigs = false;
114     
115         for (int i = 1; i < argc; i++) {
116             if (argv[i][0] == '-') {
117                 if (std::strcmp(argv[i], "--version") == 0) {
118                     _showVersion = true;
119                     _exitAfterPrint = true;
120                     return true;
121                 }
122     
123                 // Flag used for various purposes during debugging
124                 else if (std::strcmp(argv[i], "--debug") == 0)
125                     _settings->debug = _settings->debugwarnings = true;
126     
127                 // Show --debug output after the first simplifications
128                 else if (std::strcmp(argv[i], "--debug-normal") == 0)
129                     _settings->debugnormal = true;
130     
131                 // Show debug warnings
132                 else if (std::strcmp(argv[i], "--debug-warnings") == 0)
133                     _settings->debugwarnings = true;
134     
135                 // dump cppcheck data
136                 else if (std::strcmp(argv[i], "--dump") == 0)
137                     _settings->dump = true;
138     
139                 // (Experimental) exception handling inside cppcheck client
140                 else if (std::strcmp(argv[i], "--exception-handling") == 0)
141                     _settings->exceptionHandling = true;
142                 else if (std::strncmp(argv[i], "--exception-handling=", 21) == 0) {
143                     _settings->exceptionHandling = true;
144                     const std::string exceptionOutfilename = &(argv[i][21]);
145                     CppCheckExecutor::setExceptionOutput((exceptionOutfilename=="stderr") ? stderr : stdout);
146                 }
147     
148                 // Inconclusive checking
149                 else if (std::strcmp(argv[i], "--inconclusive") == 0)
150                     _settings->inconclusive = true;
151     
152                 // Enforce language (--language=, -x)
153                 else if (std::strncmp(argv[i], "--language=", 11) == 0 || std::strcmp(argv[i], "-x") == 0) {
154                     std::string str;
155                     if (argv[i][2]) {
156                         str = argv[i]+11;
157                     } else {
158                         i++;
159                         if (i >= argc || argv[i][0] == '-') {
160                             PrintMessage("cppcheck: No language given to '-x' option.");
161                             return false;
162                         }
163                         str = argv[i];
164                     }
165     
166                     if (str == "c")
167                         _settings->enforcedLang = Settings::C;
168                     else if (str == "c++")
169                         _settings->enforcedLang = Settings::CPP;
170                     else {
171                         PrintMessage("cppcheck: Unknown language '" + str + "' enforced.");
172                         return false;
173                     }
174                 }
175     
176                 // Filter errors
177                 else if (std::strncmp(argv[i], "--exitcode-suppressions=", 24) == 0) {
178                     // exitcode-suppressions=filename.txt
179                     std::string filename = 24 + argv[i];
180     
181                     std::ifstream f(filename.c_str());
182                     if (!f.is_open()) {
183                         PrintMessage("cppcheck: Couldn't open the file: \"" + filename + "\".");
184                         return false;
185                     }
186                     const std::string errmsg(_settings->nofail.parseFile(f));
187                     if (!errmsg.empty()) {
188                         PrintMessage(errmsg);
189                         return false;
190                     }
191                 }
192     
193                 // Filter errors
194                 else if (std::strncmp(argv[i], "--suppressions-list=", 20) == 0) {
195                     std::string filename = argv[i]+20;
196                     std::ifstream f(filename.c_str());
197                     if (!f.is_open()) {
198                         std::string message("cppcheck: Couldn't open the file: \"");
199                         message += filename;
200                         message += "\".";
201                         if (std::count(filename.begin(), filename.end(), ',') > 0 ||
202                             std::count(filename.begin(), filename.end(), '.') > 1) {
203                             // If user tried to pass multiple files (we can only guess that)
204                             // e.g. like this: --suppressions-list=a.txt,b.txt
205                             // print more detailed error message to tell user how he can solve the problem
206                             message += "\nIf you want to pass two files, you can do it e.g. like this:";
207                             message += "\n    cppcheck --suppressions-list=a.txt --suppressions-list=b.txt file.cpp";
208                         }
209     
210                         PrintMessage(message);
211                         return false;
212                     }
213                     const std::string errmsg(_settings->nomsg.parseFile(f));
214                     if (!errmsg.empty()) {
215                         PrintMessage(errmsg);
216                         return false;
217                     }
218                 }
219     
220                 else if (std::strncmp(argv[i], "--suppress=", 11) == 0) {
221                     std::string suppression = argv[i]+11;
222                     const std::string errmsg(_settings->nomsg.addSuppressionLine(suppression));
223                     if (!errmsg.empty()) {
224                         PrintMessage(errmsg);
225                         return false;
226                     }
227                 }
228     
229                 // Enables inline suppressions.
230                 else if (std::strcmp(argv[i], "--inline-suppr") == 0)
231                     _settings->inlineSuppressions = true;
232     
233                 // Verbose error messages (configuration info)
234                 else if (std::strcmp(argv[i], "-v") == 0 || std::strcmp(argv[i], "--verbose") == 0)
235                     _settings->verbose = true;
236     
237                 // Force checking of files that have "too many" configurations
238                 else if (std::strcmp(argv[i], "-f") == 0 || std::strcmp(argv[i], "--force") == 0)
239                     _settings->force = true;
240     
241                 // Output relative paths
242                 else if (std::strcmp(argv[i], "-rp") == 0 || std::strcmp(argv[i], "--relative-paths") == 0)
243                     _settings->relativePaths = true;
244                 else if (std::strncmp(argv[i], "-rp=", 4) == 0 || std::strncmp(argv[i], "--relative-paths=", 17) == 0) {
245                     _settings->relativePaths = true;
246                     if (argv[i][argv[i][3]=='='?4:17] != 0) {
247                         std::string paths = argv[i]+(argv[i][3]=='='?4:17);
248                         for (;;) {
249                             std::string::size_type pos = paths.find(';');
250                             if (pos == std::string::npos) {
251                                 _settings->basePaths.push_back(Path::fromNativeSeparators(paths));
252                                 break;
253                             } else {
254                                 _settings->basePaths.push_back(Path::fromNativeSeparators(paths.substr(0, pos)));
255                                 paths.erase(0, pos + 1);
256                             }
257                         }
258                     } else {
259                         PrintMessage("cppcheck: No paths specified for the '" + std::string(argv[i]) + "' option.");
260                         return false;
261                     }
262                 }
263     
264                 // Write results in results.xml
265                 else if (std::strcmp(argv[i], "--xml") == 0)
266                     _settings->xml = true;
267     
268                 // Define the XML file version (and enable XML output)
269                 else if (std::strncmp(argv[i], "--xml-version=", 14) == 0) {
270                     std::string numberString(argv[i]+14);
271     
272                     std::istringstream iss(numberString);
273                     if (!(iss >> _settings->xml_version)) {
274                         PrintMessage("cppcheck: argument to '--xml-version' is not a number.");
275                         return false;
276                     }
277     
278                     if (_settings->xml_version < 0 || _settings->xml_version > 2) {
279                         // We only have xml versions 1 and 2
280                         PrintMessage("cppcheck: '--xml-version' can only be 1 or 2.");
281                         return false;
282                     }
283     
284                     // Enable also XML if version is set
285                     _settings->xml = true;
286                 }
287     
288                 // Only print something when there are errors
289                 else if (std::strcmp(argv[i], "-q") == 0 || std::strcmp(argv[i], "--quiet") == 0)
290                     _settings->quiet = true;
291     
292                 // Append user-defined code to checked source code
293                 else if (std::strncmp(argv[i], "--append=", 9) == 0) {
294                     const std::string filename = 9 + argv[i];
295                     if (!_settings->append(filename)) {
296                         PrintMessage("cppcheck: Couldn't open the file: \"" + filename + "\".");
297                         return false;
298                     }
299                 }
300     
301                 // Check configuration
302                 else if (std::strcmp(argv[i], "--check-config") == 0) {
303                     _settings->checkConfiguration = true;
304                 }
305     
306                 // Check library definitions
307                 else if (std::strcmp(argv[i], "--check-library") == 0) {
308                     _settings->checkLibrary = true;
309                 }
310     
311                 else if (std::strncmp(argv[i], "--enable=", 9) == 0) {
312                     const std::string errmsg = _settings->addEnabled(argv[i] + 9);
313                     if (!errmsg.empty()) {
314                         PrintMessage(errmsg);
315                         return false;
316                     }
317                     // when "style" is enabled, also enable "warning", "performance" and "portability"
318                     if (_settings->isEnabled("style")) {
319                         _settings->addEnabled("warning");
320                         _settings->addEnabled("performance");
321                         _settings->addEnabled("portability");
322                     }
323                 }
324     
325                 // --error-exitcode=1
326                 else if (std::strncmp(argv[i], "--error-exitcode=", 17) == 0) {
327                     std::string temp = argv[i]+17;
328                     std::istringstream iss(temp);
329                     if (!(iss >> _settings->exitCode)) {
330                         _settings->exitCode = 0;
331                         PrintMessage("cppcheck: Argument must be an integer. Try something like '--error-exitcode=1'.");
332                         return false;
333                     }
334                 }
335     
336                 // User define
337                 else if (std::strncmp(argv[i], "-D", 2) == 0) {
338                     std::string define;
339     
340                     // "-D define"
341                     if (std::strcmp(argv[i], "-D") == 0) {
342                         ++i;
343                         if (i >= argc || argv[i][0] == '-') {
344                             PrintMessage("cppcheck: argument to '-D' is missing.");
345                             return false;
346                         }
347     
348                         define = argv[i];
349                     }
350                     // "-Ddefine"
351                     else {
352                         define = 2 + argv[i];
353                     }
354     
355                     // No "=", append a "=1"
356                     if (define.find('=') == std::string::npos)
357                         define += "=1";
358     
359                     if (!_settings->userDefines.empty())
360                         _settings->userDefines += ";";
361                     _settings->userDefines += define;
362     
363                     def = true;
364                 }
365                 // User undef
366                 else if (std::strncmp(argv[i], "-U", 2) == 0) {
367                     std::string undef;
368     
369                     // "-U undef"
370                     if (std::strcmp(argv[i], "-U") == 0) {
371                         ++i;
372                         if (i >= argc || argv[i][0] == '-') {
373                             PrintMessage("cppcheck: argument to '-U' is missing.");
374                             return false;
375                         }
376     
377                         undef = argv[i];
378                     }
379                     // "-Uundef"
380                     else {
381                         undef = 2 + argv[i];
382                     }
383     
384                     _settings->userUndefs.insert(undef);
385                 }
386     
387                 // -E
388                 else if (std::strcmp(argv[i], "-E") == 0) {
389                     _settings->preprocessOnly = true;
390                 }
391     
392                 // Include paths
393                 else if (std::strncmp(argv[i], "-I", 2) == 0) {
394                     std::string path;
395     
396                     // "-I path/"
397                     if (std::strcmp(argv[i], "-I") == 0) {
398                         ++i;
399                         if (i >= argc || argv[i][0] == '-') {
400                             PrintMessage("cppcheck: argument to '-I' is missing.");
401                             return false;
402                         }
403                         path = argv[i];
404                     }
405     
406                     // "-Ipath/"
407                     else {
408                         path = 2 + argv[i];
409                     }
410                     path = Path::removeQuotationMarks(path);
411                     path = Path::fromNativeSeparators(path);
412     
413                     // If path doesn't end with / or \, add it
414                     if (path.back() != '/')
415                         path += '/';
416     
417                     _settings->includePaths.push_back(path);
418                 } else if (std::strncmp(argv[i], "--include=", 10) == 0) {
419                     std::string path = argv[i] + 10;
420     
421                     path = Path::fromNativeSeparators(path);
422     
423                     _settings->userIncludes.push_back(path);
424                 } else if (std::strncmp(argv[i], "--includes-file=", 16) == 0) {
425                     // open this file and read every input file (1 file name per line)
426                     AddInclPathsToList(16 + argv[i], &_settings->includePaths);
427                 } else if (std::strncmp(argv[i], "--config-exclude=",17) ==0) {
428                     std::string path = argv[i] + 17;
429                     path = Path::fromNativeSeparators(path);
430                     _settings->configExcludePaths.insert(path);
431                 } else if (std::strncmp(argv[i], "--config-excludes-file=", 23) == 0) {
432                     // open this file and read every input file (1 file name per line)
433                     AddPathsToSet(23 + argv[i], &_settings->configExcludePaths);
434                 }
435     
436                 // file list specified
437                 else if (std::strncmp(argv[i], "--file-list=", 12) == 0) {
438                     // open this file and read every input file (1 file name per line)
439                     AddFilesToList(12 + argv[i], _pathnames);
440                 }
441     
442                 // Ignored paths
443                 else if (std::strncmp(argv[i], "-i", 2) == 0) {
444                     std::string path;
445     
446                     // "-i path/"
447                     if (std::strcmp(argv[i], "-i") == 0) {
448                         ++i;
449                         if (i >= argc || argv[i][0] == '-') {
450                             PrintMessage("cppcheck: argument to '-i' is missing.");
451                             return false;
452                         }
453                         path = argv[i];
454                     }
455     
456                     // "-ipath/"
457                     else {
458                         path = 2 + argv[i];
459                     }
460     
461                     if (!path.empty()) {
462                         path = Path::removeQuotationMarks(path);
463                         path = Path::fromNativeSeparators(path);
464                         path = Path::simplifyPath(path);
465     
466                         if (FileLister::isDirectory(path)) {
467                             // If directory name doesn't end with / or \, add it
468                             if (path.back() != '/')
469                                 path += '/';
470                         }
471                         _ignoredPaths.push_back(path);
472                     }
473                 }
474     
475                 // --library
476                 else if (std::strncmp(argv[i], "--library=", 10) == 0) {
477                     if (!CppCheckExecutor::tryLoadLibrary(_settings->library, argv[0], argv[i]+10))
478                         return false;
479                 }
480     
481                 // Report progress
482                 else if (std::strcmp(argv[i], "--report-progress") == 0) {
483                     _settings->reportProgress = true;
484                 }
485     
486                 // --std
487                 else if (std::strcmp(argv[i], "--std=posix") == 0) {
488                     _settings->standards.posix = true;
489                 } else if (std::strcmp(argv[i], "--std=c89") == 0) {
490                     _settings->standards.c = Standards::C89;
491                 } else if (std::strcmp(argv[i], "--std=c99") == 0) {
492                     _settings->standards.c = Standards::C99;
493                 } else if (std::strcmp(argv[i], "--std=c11") == 0) {
494                     _settings->standards.c = Standards::C11;
495                 } else if (std::strcmp(argv[i], "--std=c++03") == 0) {
496                     _settings->standards.cpp = Standards::CPP03;
497                 } else if (std::strcmp(argv[i], "--std=c++11") == 0) {
498                     _settings->standards.cpp = Standards::CPP11;
499                 }
500     
501                 // Output formatter
502                 else if (std::strcmp(argv[i], "--template") == 0 ||
503                          std::strncmp(argv[i], "--template=", 11) == 0) {
504                     // "--template path/"
505                     if (argv[i][10] == '=')
506                         _settings->outputFormat = argv[i] + 11;
507                     else if ((i+1) < argc && argv[i+1][0] != '-') {
508                         ++i;
509                         _settings->outputFormat = argv[i];
510                     } else {
511                         PrintMessage("cppcheck: argument to '--template' is missing.");
512                         return false;
513                     }
514     
515                     if (_settings->outputFormat == "gcc")
516                         _settings->outputFormat = "{file}:{line}: {severity}: {message}";
517                     else if (_settings->outputFormat == "vs")
518                         _settings->outputFormat = "{file}({line}): {severity}: {message}";
519                     else if (_settings->outputFormat == "edit")
520                         _settings->outputFormat = "{file} +{line}: {severity}: {message}";
521                 }
522     
523                 // Checking threads
524                 else if (std::strncmp(argv[i], "-j", 2) == 0) {
525                     std::string numberString;
526     
527                     // "-j 3"
528                     if (std::strcmp(argv[i], "-j") == 0) {
529                         ++i;
530                         if (i >= argc || argv[i][0] == '-') {
531                             PrintMessage("cppcheck: argument to '-j' is missing.");
532                             return false;
533                         }
534     
535                         numberString = argv[i];
536                     }
537     
538                     // "-j3"
539                     else
540                         numberString = argv[i]+2;
541     
542                     std::istringstream iss(numberString);
543                     if (!(iss >> _settings->jobs)) {
544                         PrintMessage("cppcheck: argument to '-j' is not a number.");
545                         return false;
546                     }
547     
548                     if (_settings->jobs > 10000) {
549                         // This limit is here just to catch typos. If someone has
550                         // need for more jobs, this value should be increased.
551                         PrintMessage("cppcheck: argument for '-j' is allowed to be 10000 at max.");
552                         return false;
553                     }
554                 } else if (std::strncmp(argv[i], "-l", 2) == 0) {
555                     std::string numberString;
556     
557                     // "-l 3"
558                     if (std::strcmp(argv[i], "-l") == 0) {
559                         ++i;
560                         if (i >= argc || argv[i][0] == '-') {
561                             PrintMessage("cppcheck: argument to '-l' is missing.");
562                             return false;
563                         }
564     
565                         numberString = argv[i];
566                     }
567     
568                     // "-l3"
569                     else
570                         numberString = argv[i]+2;
571     
572                     std::istringstream iss(numberString);
573                     if (!(iss >> _settings->loadAverage)) {
574                         PrintMessage("cppcheck: argument to '-l' is not a number.");
575                         return false;
576                     }
577                 }
578     
579                 // print all possible error messages..
580                 else if (std::strcmp(argv[i], "--errorlist") == 0) {
581                     _showErrorMessages = true;
582                     _settings->xml = true;
583                     _exitAfterPrint = true;
584                 }
585     
586                 // documentation..
587                 else if (std::strcmp(argv[i], "--doc") == 0) {
588                     std::ostringstream doc;
589                     // Get documentation..
590                     for (std::list<Check *>::iterator it = Check::instances().begin(); it != Check::instances().end(); ++it) {
591                         const std::string& name((*it)->name());
592                         const std::string info((*it)->classInfo());
593                         if (!name.empty() && !info.empty())
594                             doc << "## " << name << " ##\n"
595                                 << info << "\n";
596                     }
597     
598                     std::cout << doc.str();
599                     _exitAfterPrint = true;
600                     return true;
601                 }
602     
603                 // show timing information..
604                 else if (std::strncmp(argv[i], "--showtime=", 11) == 0) {
605                     const std::string showtimeMode = argv[i] + 11;
606                     if (showtimeMode == "file")
607                         _settings->showtime = SHOWTIME_FILE;
608                     else if (showtimeMode == "summary")
609                         _settings->showtime = SHOWTIME_SUMMARY;
610                     else if (showtimeMode == "top5")
611                         _settings->showtime = SHOWTIME_TOP5;
612                     else if (showtimeMode.empty())
613                         _settings->showtime = SHOWTIME_NONE;
614                     else {
615                         std::string message("cppcheck: error: unrecognized showtime mode: \"");
616                         message += showtimeMode;
617                         message += "\". Supported modes: file, summary, top5.";
618                         PrintMessage(message);
619                         return false;
620                     }
621                 }
622     
623     #ifdef HAVE_RULES
624                 // Rule given at command line
625                 else if (std::strncmp(argv[i], "--rule=", 7) == 0) {
626                     Settings::Rule rule;
627                     rule.pattern = 7 + argv[i];
628                     _settings->rules.push_back(rule);
629                 }
630     
631                 // Rule file
632                 else if (std::strncmp(argv[i], "--rule-file=", 12) == 0) {
633                     tinyxml2::XMLDocument doc;
634                     if (doc.LoadFile(12+argv[i]) == tinyxml2::XML_NO_ERROR) {
635                         tinyxml2::XMLElement *node = doc.FirstChildElement();
636                         for (; node && strcmp(node->Value(), "rule") == 0; node = node->NextSiblingElement()) {
637                             Settings::Rule rule;
638     
639                             tinyxml2::XMLElement *tokenlist = node->FirstChildElement("tokenlist");
640                             if (tokenlist)
641                                 rule.tokenlist = tokenlist->GetText();
642     
643                             tinyxml2::XMLElement *pattern = node->FirstChildElement("pattern");
644                             if (pattern) {
645                                 rule.pattern = pattern->GetText();
646                             }
647     
648                             tinyxml2::XMLElement *message = node->FirstChildElement("message");
649                             if (message) {
650                                 tinyxml2::XMLElement *severity = message->FirstChildElement("severity");
651                                 if (severity)
652                                     rule.severity = Severity::fromString(severity->GetText());
653     
654                                 tinyxml2::XMLElement *id = message->FirstChildElement("id");
655                                 if (id)
656                                     rule.id = id->GetText();
657     
658                                 tinyxml2::XMLElement *summary = message->FirstChildElement("summary");
659                                 if (summary)
660                                     rule.summary = summary->GetText() ? summary->GetText() : "";
661                             }
662     
663                             if (!rule.pattern.empty())
664                                 _settings->rules.push_back(rule);
665                         }
666                     }
667                 }
668     #endif
669     
670                 // Specify platform
671                 else if (std::strncmp(argv[i], "--platform=", 11) == 0) {
672                     std::string platform(11+argv[i]);
673     
674                     if (platform == "win32A")
675                         _settings->platform(Settings::Win32A);
676                     else if (platform == "win32W")
677                         _settings->platform(Settings::Win32W);
678                     else if (platform == "win64")
679                         _settings->platform(Settings::Win64);
680                     else if (platform == "unix32")
681                         _settings->platform(Settings::Unix32);
682                     else if (platform == "unix64")
683                         _settings->platform(Settings::Unix64);
684                     else if (platform == "native")
685                         _settings->platform(Settings::Unspecified);
686                     else if (!_settings->platformFile(platform)) {
687                         std::string message("cppcheck: error: unrecognized platform: \"");
688                         message += platform;
689                         message += "\".";
690                         PrintMessage(message);
691                         return false;
692                     }
693                 }
694     
695                 // Set maximum number of #ifdef configurations to check
696                 else if (std::strncmp(argv[i], "--max-configs=", 14) == 0) {
697                     _settings->force = false;
698     
699                     std::istringstream iss(14+argv[i]);
700                     if (!(iss >> _settings->maxConfigs)) {
701                         PrintMessage("cppcheck: argument to '--max-configs=' is not a number.");
702                         return false;
703                     }
704     
705                     if (_settings->maxConfigs < 1) {
706                         PrintMessage("cppcheck: argument to '--max-configs=' must be greater than 0.");
707                         return false;
708                     }
709     
710                     maxconfigs = true;
711                 }
712     
713                 // Print help
714                 else if (std::strcmp(argv[i], "-h") == 0 || std::strcmp(argv[i], "--help") == 0) {
715                     _pathnames.clear();
716                     _showHelp = true;
717                     _exitAfterPrint = true;
718                     break;
719                 }
720     
721                 else {
722                     std::string message("cppcheck: error: unrecognized command line option: \"");
723                     message += argv[i];
724                     message += "\".";
725                     PrintMessage(message);
726                     return false;
727                 }
728             }
729     
730             else {
731                 std::string path = Path::removeQuotationMarks(argv[i]);
732                 path = Path::fromNativeSeparators(path);
733                 _pathnames.push_back(path);
734             }
735         }
736     
737         if (_settings->force)
738             _settings->maxConfigs = ~0U;
739     
740         else if ((def || _settings->preprocessOnly) && !maxconfigs)
741             _settings->maxConfigs = 1U;
742     
743         if (_settings->isEnabled("unusedFunction") && _settings->jobs > 1) {
744             PrintMessage("cppcheck: unusedFunction check can't be used with '-j' option. Disabling unusedFunction check.");
745         }
746     
747         if (_settings->inconclusive && _settings->xml && _settings->xml_version == 1U) {
748             PrintMessage("cppcheck: inconclusive messages will not be shown, because the old xml format is not compatible. It's recommended to use the new xml format (use --xml-version=2).");
749         }
750     
751         if (argc <= 1) {
752             _showHelp = true;
753             _exitAfterPrint = true;
754         }
755     
756         if (_showHelp) {
757             PrintHelp();
758             return true;
759         }
760     
761         // Print error only if we have "real" command and expect files
762         if (!_exitAfterPrint && _pathnames.empty()) {
763             PrintMessage("cppcheck: No C or C++ source files found.");
764             return false;
765         }
766     
767         // Use paths _pathnames if no base paths for relative path output are given
768         if (_settings->basePaths.empty() && _settings->relativePaths)
769             _settings->basePaths = _pathnames;
770     
771         return true;
772     }
